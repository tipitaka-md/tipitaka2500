# 𑁭𑁇𑁧𑁇𑁭 𑀕𑁄𑀤𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀕𑁄𑀤𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(349.)

1519\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑁄𑀤𑀢𑁆𑀢𑁄 𑀫𑀘𑁆𑀙𑀺𑀓𑀸𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀫𑁆𑀩𑀸𑀝𑀓𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀕𑁄𑀤𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀕𑁄𑀤𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑁄𑀤𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀸 𑀘𑀸𑀬𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀬𑀸 𑀘 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀬𑀸 𑀘 𑀲𑀼𑀜𑁆𑀜𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀬𑀸 𑀘 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀣𑀸 𑀦𑀸𑀦𑀸𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑀢𑁆𑀣𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀦𑁆”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀣𑀸 𑀘𑁂𑀯 𑀦𑀸𑀦𑀸𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀘𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀢𑁆𑀣𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

1520\. “𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀣𑀸 𑀘𑁂𑀯 𑀦𑀸𑀦𑀸𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀘? 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀓𑀭𑀼𑀡𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂…  𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇

1521\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺? 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫, ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇

1522\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀜𑁆𑀜𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺? 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀲𑀼𑀜𑁆𑀜𑀫𑀺𑀤𑀁 𑀅𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀺𑀬𑁂𑀦 𑀯𑀸’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀜𑁆𑀜𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇

1523\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺? 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀣𑀸 𑀘𑁂𑀯 𑀦𑀸𑀦𑀸𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀘𑁇

1524\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀢𑁆𑀣𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁? 𑀭𑀸𑀕𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀫𑀸𑀡𑀓𑀭𑀡𑁄, 𑀤𑁄𑀲𑁄 𑀧𑀫𑀸𑀡𑀓𑀭𑀡𑁄, 𑀫𑁄𑀳𑁄 𑀧𑀫𑀸𑀡𑀓𑀭𑀡𑁄𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀢𑀸𑀲𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀲𑀸 𑀔𑁄 𑀧𑀦 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀸 𑀭𑀸𑀕𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀤𑁄𑀲𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑁄𑀳𑁂𑀦𑁇 𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀺𑀜𑁆𑀘𑀦𑀁, 𑀤𑁄𑀲𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁, 𑀫𑁄𑀳𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀢𑀸𑀲𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀲𑀸 𑀔𑁄 𑀧𑀦 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀸 𑀭𑀸𑀕𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀤𑁄𑀲𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑁄𑀳𑁂𑀦𑁇 𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀭𑀡𑁄, 𑀤𑁄𑀲𑁄 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀭𑀡𑁄, 𑀫𑁄𑀳𑁄 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀭𑀡𑁄𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀢𑀸𑀲𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀲𑀸 𑀔𑁄 𑀧𑀦 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀸 𑀭𑀸𑀕𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀤𑁄𑀲𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑁄𑀳𑁂𑀦𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀢𑁆𑀣𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀮𑀸𑀪𑀸 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺𑁇 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑁂 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀓𑀫𑀢𑀻”𑀢𑀺𑁇
