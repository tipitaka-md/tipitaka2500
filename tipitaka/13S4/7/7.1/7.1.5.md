# 𑁭𑁇𑁧𑁇𑁫 𑀧𑀞𑀫𑀓𑀸𑀫𑀪𑀽𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀓𑀸𑀫𑀪𑀽𑀲𑀼𑀢𑁆𑀢

(347.)

1500\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀸𑀫𑀪𑀽 𑀫𑀘𑁆𑀙𑀺𑀓𑀸𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀫𑁆𑀩𑀸𑀝𑀓𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀓𑀸𑀫𑀪𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀓𑀸𑀫𑀪𑀼𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀸𑀫𑀪𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—

1501\. “𑀯𑀼𑀢𑁆𑀢𑀫𑀺𑀤𑀁, 𑀕𑀳𑀧𑀢𑀺—

1502\. _‘𑀦𑁂𑀮𑀗𑁆𑀕𑁄 𑀲𑁂𑀢𑀧𑀘𑁆𑀙𑀸𑀤𑁄,_  
_𑀏𑀓𑀸𑀭𑁄 𑀯𑀢𑁆𑀢𑀢𑀻 𑀭𑀣𑁄;_  
_𑀅𑀦𑀻𑀖𑀁 𑀧𑀲𑁆𑀲 𑀆𑀬𑀦𑁆𑀢𑀁,_  
_𑀙𑀺𑀦𑁆𑀦𑀲𑁄𑀢𑀁 𑀅𑀩𑀦𑁆𑀥𑀦𑀦𑁆’𑀢𑀺𑁇_  


1503\. 𑀇𑀫𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀓𑀣𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺? “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀸𑀲𑀺𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀕𑀳𑀧𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀆𑀕𑀫𑁂𑀳𑀺 𑀬𑀸𑀯𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀧𑁂𑀓𑁆𑀔𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑀼𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀓𑀸𑀫𑀪𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1504\. “‘𑀦𑁂𑀮𑀗𑁆𑀕𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀻𑀮𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 ‘𑀲𑁂𑀢𑀧𑀘𑁆𑀙𑀸𑀤𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 ‘𑀏𑀓𑀸𑀭𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀢𑀺𑀬𑀸 𑀏𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 ‘𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 ‘𑀭𑀣𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑀲𑁆𑀲𑁂𑀢𑀁 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀲𑀫𑁆𑀪𑀯𑀲𑁆𑀲 𑀑𑀤𑀦𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀽𑀧𑀘𑀬𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀼𑀘𑁆𑀙𑀸𑀤𑀦𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀦𑀪𑁂𑀤𑀦𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑀦𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑁇 𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀻𑀖𑁄, 𑀤𑁄𑀲𑁄 𑀦𑀻𑀖𑁄, 𑀫𑁄𑀳𑁄 𑀦𑀻𑀖𑁄𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀲𑁆𑀫𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀦𑀻𑀖𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 ‘𑀆𑀬𑀦𑁆𑀢𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀳𑀢𑁄 𑀏𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 ‘𑀲𑁄𑀢𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀡𑁆𑀳𑀸𑀬𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀲𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀲𑁆𑀫𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀙𑀺𑀦𑁆𑀦𑀲𑁄𑀢𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀦𑁆𑀥𑀦𑀁, 𑀤𑁄𑀲𑁄 𑀩𑀦𑁆𑀥𑀦𑀁, 𑀫𑁄𑀳𑁄 𑀩𑀦𑁆𑀥𑀦𑀁𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀲𑁆𑀫𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀩𑀦𑁆𑀥𑀦𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀁 𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀁—

1505\. _‘𑀦𑁂𑀮𑀗𑁆𑀕𑁄 𑀲𑁂𑀢𑀧𑀘𑁆𑀙𑀸𑀤𑁄,_  
_𑀏𑀓𑀸𑀭𑁄 𑀯𑀢𑁆𑀢𑀢𑀻 𑀭𑀣𑁄;_  
_𑀅𑀦𑀻𑀖𑀁 𑀧𑀲𑁆𑀲 𑀆𑀬𑀦𑁆𑀢𑀁,_  
_𑀙𑀺𑀦𑁆𑀦𑀲𑁄𑀢𑀁 𑀅𑀩𑀦𑁆𑀥𑀦𑀦𑁆’𑀢𑀺𑁇_  


1506\. 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀮𑀸𑀪𑀸 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺𑁇 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑁂 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀓𑀫𑀢𑀻”𑀢𑀺𑁇
