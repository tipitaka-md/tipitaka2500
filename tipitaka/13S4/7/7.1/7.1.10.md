# 𑁭𑁇𑁧𑁇𑁧𑁦 𑀕𑀺𑀮𑀸𑀦𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀕𑀺𑀮𑀸𑀦𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

(352.)

1536. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀭𑀸𑀫𑀤𑁂𑀯𑀢𑀸 𑀯𑀦𑀤𑁂𑀯𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀤𑁂𑀯𑀢𑀸 𑀑𑀲𑀥𑀺𑀢𑀺𑀡𑀯𑀦𑀧𑁆𑀧𑀢𑀻𑀲𑀼 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀘𑀺𑀢𑁆𑀢𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀧𑀡𑀺𑀥𑁂𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀭𑀸𑀚𑀸 𑀅𑀲𑁆𑀲𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻”𑀢𑀺𑁇

1537. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀢𑀸 𑀆𑀭𑀸𑀫𑀤𑁂𑀯𑀢𑀸 𑀯𑀦𑀤𑁂𑀯𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀤𑁂𑀯𑀢𑀸 𑀑𑀲𑀥𑀺𑀢𑀺𑀡𑀯𑀦𑀧𑁆𑀧𑀢𑀻𑀲𑀼 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀫𑁆𑀧𑀺 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀼𑀯𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀧𑀳𑀸𑀬 𑀕𑀫𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑀸 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀢𑀺𑀁, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀳𑀺, 𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀮𑀧𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀢𑀸𑀳𑀁 𑀯𑀤𑀸𑀫𑀺 𑀬𑀁 𑀫𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀣—  ‘𑀲𑀢𑀺𑀁, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀳𑀺, 𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀮𑀧𑀻’”𑀢𑀺? “𑀏𑀯𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀯𑀤𑁂𑀲𑀺—  ‘𑀢𑀫𑁆𑀧𑀺 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀼𑀯𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀧𑀳𑀸𑀬 𑀕𑀫𑀦𑀻𑀬𑀦𑁆’”𑀢𑀺𑁇 “𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦 𑀫𑀁 𑀆𑀭𑀸𑀫𑀤𑁂𑀯𑀢𑀸 𑀯𑀦𑀤𑁂𑀯𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀤𑁂𑀯𑀢𑀸 𑀑𑀲𑀥𑀺𑀢𑀺𑀡𑀯𑀦𑀧𑁆𑀧𑀢𑀻𑀲𑀼 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀧𑀡𑀺𑀥𑁂𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀭𑀸𑀚𑀸 𑀅𑀲𑁆𑀲𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑀫𑁆𑀧𑀺 𑀅𑀦𑀺𑀘𑁆𑀘𑀁…𑀧𑁂…  𑀢𑀫𑁆𑀧𑀺 𑀧𑀳𑀸𑀬 𑀕𑀫𑀦𑀻𑀬𑀦𑁆’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑀸, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀆𑀭𑀸𑀫𑀤𑁂𑀯𑀢𑀸 𑀯𑀦𑀤𑁂𑀯𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀤𑁂𑀯𑀢𑀸 𑀑𑀲𑀥𑀺𑀢𑀺𑀡𑀯𑀦𑀧𑁆𑀧𑀢𑀻𑀲𑀼 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀧𑀡𑀺𑀥𑁂𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀭𑀸𑀚𑀸 𑀅𑀲𑁆𑀲𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻’”𑀢𑀺? “𑀢𑀸𑀲𑀁 𑀔𑁄 𑀆𑀭𑀸𑀫𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀯𑀦𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀭𑀼𑀓𑁆𑀔𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀑𑀲𑀥𑀺𑀢𑀺𑀡𑀯𑀦𑀧𑁆𑀧𑀢𑀻𑀲𑀼 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀻𑀮𑀯𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁄𑁇 𑀲𑀘𑁂 𑀧𑀡𑀺𑀤𑀳𑀺𑀲𑁆𑀲𑀢𑀺—  𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀭𑀸𑀚𑀸 𑀅𑀲𑁆𑀲𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻’𑀢𑀺, ‘𑀢𑀲𑁆𑀲 𑀔𑁄 𑀅𑀬𑀁 𑀇𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀻𑀮𑀯𑀢𑁄 𑀘𑁂𑀢𑁄𑀧𑀡𑀺𑀥𑀺 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀨𑀮𑀁 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀢𑀸 𑀆𑀭𑀸𑀫𑀤𑁂𑀯𑀢𑀸 𑀯𑀦𑀤𑁂𑀯𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀤𑁂𑀯𑀢𑀸 𑀑𑀲𑀥𑀺𑀢𑀺𑀡𑀯𑀦𑀧𑁆𑀧𑀢𑀻𑀲𑀼 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀧𑀡𑀺𑀥𑁂𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀭𑀸𑀚𑀸 𑀅𑀲𑁆𑀲𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑀫𑁆𑀧𑀺 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀼𑀯𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀧𑀳𑀸𑀬 𑀕𑀫𑀦𑀻𑀬𑀦𑁆’”𑀢𑀺𑁇

1538. “𑀢𑁂𑀦 𑀳𑀺, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀅𑀫𑁆𑀳𑁂𑀧𑀺 𑀑𑀯𑀤𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀯𑁄 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀉𑀚𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀜𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀏𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀬𑀁 𑀔𑁄 𑀧𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀼𑀮𑁂 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂𑀳𑀻𑀢𑀺 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑁂 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀲𑀁𑀖𑁂 𑀘 𑀘𑀸𑀕𑁂 𑀘 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀻𑀢𑀺𑁇
