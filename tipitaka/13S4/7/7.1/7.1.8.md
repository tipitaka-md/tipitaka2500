# 𑁭𑁇𑁧𑁇𑁮 𑀦𑀺𑀕𑀡𑁆𑀞𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀕𑀡𑁆𑀞𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(350.)

1526. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀘𑁆𑀙𑀺𑀓𑀸𑀲𑀡𑁆𑀟𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀢𑀺𑀬𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀭𑀺𑀲𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺—  “𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀓𑀺𑀭 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀘𑁆𑀙𑀺𑀓𑀸𑀲𑀡𑁆𑀟𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀫𑀳𑀢𑀺𑀬𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀭𑀺𑀲𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑁂𑀦 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀤𑁆𑀤𑀳𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺?

1527. “𑀦 𑀔𑁆𑀯𑀸𑀳𑀁 𑀏𑀢𑁆𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀤𑁆𑀥𑀸𑀬 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀪𑀯𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀼, 𑀬𑀸𑀯 𑀉𑀚𑀼𑀓𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀅𑀲𑀞𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀅𑀫𑀸𑀬𑀸𑀯𑀻 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀯𑀸𑀢𑀁 𑀯𑀸 𑀲𑁄 𑀚𑀸𑀮𑁂𑀦 𑀩𑀸𑀥𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂 𑀦𑀺𑀭𑁄𑀥𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬, 𑀲𑀓𑀫𑀼𑀝𑁆𑀞𑀺𑀦𑀸 𑀯𑀸 𑀲𑁄 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀆𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂 𑀦𑀺𑀭𑁄𑀥𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

1528. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀧𑀡𑀻𑀢𑀢𑀭𑀁—  𑀜𑀸𑀡𑀁 𑀯𑀸 𑀲𑀤𑁆𑀥𑀸 𑀯𑀸”𑀢𑀺? “𑀲𑀤𑁆𑀥𑀸𑀬 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀜𑀸𑀡𑀁𑀬𑁂𑀯 𑀧𑀡𑀻𑀢𑀢𑀭𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺, 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺, 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀦 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀓𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀲𑀤𑁆𑀥𑀸𑀬 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺𑁇

1529. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑀁 𑀧𑀭𑀺𑀲𑀁 𑀅𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀪𑀯𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀼, 𑀬𑀸𑀯 𑀅𑀦𑀼𑀚𑀼𑀓𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀲𑀞𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀫𑀸𑀬𑀸𑀯𑀻 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀻”𑀢𑀺𑁇

1530. “𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀢𑁂 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀸𑀲𑀺𑀢𑀁—  ‘𑀏𑀯𑀁 𑀆𑀚𑀸𑀦𑀸𑀫 𑀇𑀤𑀁 𑀪𑀯𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀼, 𑀬𑀸𑀯 𑀉𑀚𑀼𑀓𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀅𑀲𑀞𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀅𑀫𑀸𑀬𑀸𑀯𑀻 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀻’𑀢𑀺𑁇 𑀇𑀤𑀸𑀦𑁂𑀯 𑀘 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀸𑀲𑀺𑀢𑀁—  ‘𑀏𑀯𑀁 𑀆𑀚𑀸𑀦𑀸𑀫 𑀇𑀤𑀁 𑀪𑀯𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀼, 𑀬𑀸𑀯 𑀅𑀦𑀼𑀚𑀼𑀓𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀲𑀞𑁄 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀸𑀯 𑀫𑀸𑀬𑀸𑀯𑀻 𑀘𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀻’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀺𑀫𑀁 𑀲𑀘𑁆𑀘𑀁, 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀲𑀘𑁂 𑀧𑀦 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀺𑀫𑀁 𑀫𑀺𑀘𑁆𑀙𑀸, 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀢𑁂 𑀲𑀘𑁆𑀘𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀲 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀧𑀜𑁆𑀳𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀬𑀤𑀸 𑀦𑁂𑀲𑀁 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀣 𑀫𑀁 𑀧𑀝𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀭𑀺𑀲𑀸𑀬𑁇 𑀏𑀓𑁄 𑀧𑀜𑁆𑀳𑁄, 𑀏𑀓𑁄 𑀉𑀤𑁆𑀤𑁂𑀲𑁄, 𑀏𑀓𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁𑁇 𑀤𑁆𑀯𑁂 𑀧𑀜𑁆𑀳𑀸, 𑀤𑁆𑀯𑁂 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀤𑁆𑀯𑁂 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀢𑀬𑁄 𑀧𑀜𑁆𑀳𑀸, 𑀢𑀬𑁄 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀢𑀻𑀡𑀺 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀜𑁆𑀳𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀧𑀜𑁆𑀘 𑀧𑀜𑁆𑀳𑀸, 𑀧𑀜𑁆𑀘 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀧𑀜𑁆𑀘 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀙 𑀧𑀜𑁆𑀳𑀸, 𑀙 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀙 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀲𑀢𑁆𑀢 𑀧𑀜𑁆𑀳𑀸, 𑀲𑀢𑁆𑀢 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀲𑀢𑁆𑀢 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀅𑀝𑁆𑀞 𑀧𑀜𑁆𑀳𑀸, 𑀅𑀝𑁆𑀞 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀅𑀝𑁆𑀞 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀦𑀯 𑀧𑀜𑁆𑀳𑀸, 𑀦𑀯 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀦𑀯 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀤𑀲 𑀧𑀜𑁆𑀳𑀸, 𑀤𑀲 𑀉𑀤𑁆𑀤𑁂𑀲𑀸, 𑀤𑀲 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀦𑀺𑀕𑀡𑁆𑀞𑀁 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀇𑀫𑁂 𑀤𑀲 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀧𑀜𑁆𑀳𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀻𑀢𑀺𑁇

---
