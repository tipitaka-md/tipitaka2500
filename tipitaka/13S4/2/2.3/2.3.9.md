# 𑁨𑁇𑁩𑁇𑁯 𑀢𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀢𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(277.)

1218\. “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀯𑁂𑀤𑀦𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
