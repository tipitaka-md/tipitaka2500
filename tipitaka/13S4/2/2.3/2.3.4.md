# 𑁨𑁇𑁩𑁇𑁪 𑀧𑀼𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

(272.)

1208. “𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑁄, 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄, 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀓𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑁇 𑀨𑀲𑁆𑀲𑀲𑀫𑀼𑀤𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀢𑀡𑁆𑀳𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸…𑀧𑁂…  𑀬𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀁𑁇 𑀇𑀤𑀁 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆’”𑀢𑀺𑁇
