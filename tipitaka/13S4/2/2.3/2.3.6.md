# 𑁨𑁇𑁩𑁇𑁬 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(274.)

1212\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑁄, 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄, 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀓𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀨𑀲𑁆𑀲𑀲𑀫𑀼𑀤𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀢𑀡𑁆𑀳𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸𑁇 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸…𑀧𑁂…  𑀬𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀁𑁇 𑀇𑀤𑀁 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆”𑀢𑀺𑁇
