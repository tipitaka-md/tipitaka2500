# 𑁨𑁇𑁩𑁇𑁧𑁧 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀲𑀼𑀢𑁆𑀢

(279.)

1222\. “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑀸 𑀧𑀻𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀧𑀻𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀸 𑀧𑀻𑀢𑀺; 𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀁 𑀲𑀼𑀔𑀁; 𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀫𑀺𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸; 𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀫𑀺𑀲𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑀸 𑀧𑀻𑀢𑀺? 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇 𑀬𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀻𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑀸 𑀧𑀻𑀢𑀺𑁇

1223\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀧𑀻𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀧𑀻𑀢𑀺𑁇

1224\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀸 𑀧𑀻𑀢𑀺? 𑀬𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀫𑁄𑀳𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀻𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀸 𑀧𑀻𑀢𑀺𑁇

1225\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁? 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁𑁇

1226\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁𑁇

1227\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀁 𑀲𑀼𑀔𑀁? 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀫𑁄𑀳𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀁 𑀲𑀼𑀔𑀁𑁇

1228\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇 𑀬𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑁇

1229\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑁇

1230\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀬𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀫𑁄𑀳𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑁇

1231\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀺𑀲𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄? 𑀭𑀽𑀧𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀲𑀸𑀫𑀺𑀲𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1232\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄? 𑀅𑀭𑀽𑀧𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1233\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄? 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄, 𑀫𑁄𑀳𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀢𑀭𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄”𑀢𑀺𑁇
