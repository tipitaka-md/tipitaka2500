# 𑁨𑁇𑁩𑁇𑁨 𑀅𑀝𑁆𑀞𑀲𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀝𑁆𑀞𑀲𑀢𑀲𑀼𑀢𑁆𑀢

(270.)

1202. “𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄? 𑀤𑁆𑀯𑁂𑀧𑀺 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦; 𑀢𑀺𑀲𑁆𑀲𑁄𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦; 𑀧𑀜𑁆𑀘𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦; 𑀙𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦; 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀸𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦; 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦; 𑀅𑀝𑁆𑀞𑀲𑀢𑀫𑁆𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑁇

1203. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀯𑁂𑀤𑀦𑀸? 𑀓𑀸𑀬𑀺𑀓𑀸 𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀘—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀯𑁂𑀤𑀦𑀸? 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀯𑁂𑀤𑀦𑀸? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀯𑁂𑀤𑀦𑀸? 𑀙 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀽𑀧𑀯𑀺𑀘𑀸𑀭𑀸, 𑀙 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀽𑀧𑀯𑀺𑀘𑀸𑀭𑀸, 𑀙 𑀉𑀧𑁂𑀓𑁆𑀔𑀽𑀧𑀯𑀺𑀘𑀸𑀭𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀯𑁂𑀤𑀦𑀸? 𑀙 𑀕𑁂𑀳𑀲𑀺𑀢𑀸𑀦𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀺, 𑀙 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀺𑀢𑀸𑀦𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀺, 𑀙 𑀕𑁂𑀳𑀲𑀺𑀢𑀸𑀦𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀺, 𑀙 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀺𑀢𑀸𑀦𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀺, 𑀙 𑀕𑁂𑀳𑀲𑀺𑀢𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀙 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀺𑀢𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀲𑀢𑀁 𑀯𑁂𑀤𑀦𑀸? 𑀅𑀢𑀻𑀢𑀸 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀦𑀸𑀕𑀢𑀸 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀯𑁂𑀤𑀦𑀸, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀲𑀢𑀁 𑀯𑁂𑀤𑀦𑀸𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀲𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄”𑀢𑀺𑁇
