# 𑁨𑁇𑁨𑁇𑁪 𑀅𑀕𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀕𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(262.)

1162\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀕𑀸𑀭𑀁𑁇 𑀢𑀢𑁆𑀣 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀸𑀬𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺, 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀧𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺, 𑀯𑁂𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺, 𑀲𑀼𑀤𑁆𑀤𑀸𑀧𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀼𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀸𑀫𑀺𑀲𑀸𑀧𑀺 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀸𑀧𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸𑀧𑀺 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸𑀧𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
