# 𑁨𑁇𑁨𑁇𑁩 𑀤𑀼𑀢𑀺𑀬𑀆𑀓𑀸𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀆𑀓𑀸𑀲𑀲𑀼𑀢𑁆𑀢

(261.)

1160\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀓𑀸𑀲𑁂 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
