# 𑁨𑁇𑁨𑁇𑁧𑁦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(268.)

1190. “𑀤𑁆𑀯𑁂𑀧𑀺 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀢𑀺𑀲𑁆𑀲𑁄𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀧𑀜𑁆𑀘𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀙𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀸𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀅𑀝𑁆𑀞𑀲𑀢𑀫𑁆𑀧𑀺 𑀫𑀬𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑁇 𑀏𑀯𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀤𑁂𑀲𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀤𑁂𑀲𑀺𑀢𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀬𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀲𑀼𑀮𑀧𑀺𑀢𑀁 𑀦 𑀲𑀫𑀦𑀼𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀫𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀫𑀦𑀼𑀫𑁄𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀤𑁂𑀲𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀤𑁂𑀲𑀺𑀢𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀬𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀲𑀼𑀮𑀧𑀺𑀢𑀁 𑀲𑀫𑀦𑀼𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑀦𑀼𑀫𑁄𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀔𑀻𑀭𑁄𑀤𑀓𑀻𑀪𑀽𑀢𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀘𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺𑁇

1191. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀕𑀼𑀡𑀸…𑀧𑁂…  𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀆𑀳, 𑀢𑀜𑁆𑀘 𑀲𑀼𑀔𑀲𑁆𑀫𑀺𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺𑁇 𑀢𑀬𑀺𑀤𑀁 𑀓𑀺𑀁𑀲𑀼, 𑀢𑀬𑀺𑀤𑀁 𑀓𑀣𑀁𑀲𑀽’𑀢𑀺? 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀲𑀼𑀔𑀜𑁆𑀜𑁂𑀯 𑀯𑁂𑀤𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀲𑀼𑀔𑀲𑁆𑀫𑀺𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀼𑀔𑀁 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀬𑀳𑀺𑀁 𑀬𑀳𑀺𑀁, 𑀢𑀁 𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀼𑀔𑀲𑁆𑀫𑀺𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀻’”𑀢𑀺𑁇
