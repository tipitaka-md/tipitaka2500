# 𑁨𑁇𑁨𑁇𑁨 𑀧𑀞𑀫𑀆𑀓𑀸𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀆𑀓𑀸𑀲𑀲𑀼𑀢𑁆𑀢

(260.)

1153\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀓𑀸𑀲𑁂 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀲𑀭𑀚𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀅𑀭𑀚𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀲𑀻𑀢𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀉𑀡𑁆𑀳𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺𑁇

1154\. _𑀬𑀣𑀸𑀧𑀺 𑀯𑀸𑀢𑀸 𑀆𑀓𑀸𑀲𑁂,_  
_𑀯𑀸𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀺𑀥𑀸 𑀧𑀼𑀣𑀽;_  
_𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀘𑀸𑀧𑀺,_  
_𑀉𑀢𑁆𑀢𑀭𑀸 𑀅𑀣 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑁇_  


1155\. _𑀲𑀭𑀚𑀸 𑀅𑀭𑀚𑀸 𑀘𑀧𑀺,_  
_𑀲𑀻𑀢𑀸 𑀉𑀡𑁆𑀳𑀸 𑀘 𑀏𑀓𑀤𑀸;_  
_𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀘,_  
_𑀧𑀼𑀣𑀽 𑀯𑀸𑀬𑀦𑁆𑀢𑀺 𑀫𑀸𑀮𑀼𑀢𑀸𑁇_  


1156\. _𑀢𑀣𑁂𑀯𑀺𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀫𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸;_  
_𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺,_  
_𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀘 𑀬𑀸𑁇_  


1157\. _𑀬𑀢𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀢𑀸𑀧𑀻,_  
_𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀦 𑀭𑀺𑀜𑁆𑀘𑀢𑀺;_  
_𑀢𑀢𑁄 𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


1158\. _𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀤𑀺𑀝𑁆𑀞𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀦𑁄𑀧𑁂𑀢𑀺 𑀯𑁂𑀤𑀕𑀽”𑀢𑀺𑁇_  


---
