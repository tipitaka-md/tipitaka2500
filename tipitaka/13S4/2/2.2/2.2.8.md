# 𑁨𑁇𑁨𑁇𑁮 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀲𑀼𑀢𑁆𑀢

(266.)

1171\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑁄, 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄, 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀓𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂… ” “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀨𑀲𑁆𑀲𑀲𑀫𑀼𑀤𑀬𑀸…𑀧𑁂… 𑁇 (𑀬𑀣𑀸 𑀧𑀼𑀭𑀺𑀫𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁂, 𑀢𑀣𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇)
