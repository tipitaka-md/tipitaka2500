# 𑁨𑁇𑁨𑁇𑁫 𑀧𑀞𑀫𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(263.)

1164\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑁄, 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄, 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀓𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀨𑀲𑁆𑀲𑀲𑀫𑀼𑀤𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑁄; 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀬𑀁 𑀯𑁂𑀤𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀅𑀬𑀁 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀲𑁆𑀲𑀸𑀤𑁄𑁇 𑀬𑀁 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸, 𑀅𑀬𑀁 𑀯𑁂𑀤𑀦𑀸𑀬 𑀆𑀤𑀻𑀦𑀯𑁄𑁇 𑀬𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀁, 𑀇𑀤𑀁 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀫𑀬𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸𑀘𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁄 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑁄 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀫𑀬𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸𑀘𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁄 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑁄 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀫𑀬𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸𑀘𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
