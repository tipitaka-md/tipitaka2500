# 𑁨𑁇𑁨𑁇𑁧 𑀭𑀳𑁄𑀕𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

* 𑀭𑀳𑁄𑀕𑀢𑀲𑀼𑀢𑁆𑀢

(259.)

1150\. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸𑁇 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀦𑁆’𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀏𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀁—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀦𑁆’”𑀢𑀺?

1151\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸𑁇 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁, 𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀦𑁆’𑀢𑀺𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑀬𑁂𑀯 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀁—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀦𑁆’𑀢𑀺𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑀬𑁂𑀯 𑀔𑀬𑀥𑀫𑁆𑀫𑀢𑀁…𑀧𑁂…  𑀯𑀬𑀥𑀫𑁆𑀫𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀢𑀁…𑀧𑁂…  𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀢𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀁—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸𑀘𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁄 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑁄 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸𑀘𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁄 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑁄 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀙𑀬𑀺𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑁄𑁇 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸𑀘𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
