# 𑁨𑁇𑁧𑁇𑁧 𑀲𑀫𑀸𑀥𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

### 𑀲𑀫𑀸𑀥𑀺𑀲𑀼𑀢𑁆𑀢

(249.)

1094\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀢𑀺𑁇

1095\. _𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄,_  
_𑀲𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀯𑁂𑀤𑀦𑀸 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀯𑁂𑀤𑀦𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀁𑁇_  


1096\. _𑀬𑀢𑁆𑀣 𑀘𑁂𑀢𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀁;_  
_𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀔𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀦𑀺𑀘𑁆𑀙𑀸𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇_
