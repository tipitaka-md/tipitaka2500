# 𑁨𑁇𑁧𑁇𑁪 𑀧𑀸𑀢𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀢𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(252.)

1109\. “𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀬𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀧𑀸𑀢𑀸𑀮𑁄’𑀢𑀺𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀧𑀸𑀢𑀸𑀮𑁄’𑀢𑀺𑁇 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀦𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 ‘𑀧𑀸𑀢𑀸𑀮𑁄’𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑁄𑀘𑀢𑀺 𑀓𑀺𑀮𑀫𑀢𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀧𑀸𑀢𑀸𑀮𑁂 𑀦 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺, 𑀕𑀸𑀥𑀜𑁆𑀘 𑀦𑀸𑀚𑁆𑀛𑀕𑀸’𑁇 𑀲𑀼𑀢𑀯𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀦𑁂𑀯 𑀲𑁄𑀘𑀢𑀺, 𑀦 𑀓𑀺𑀮𑀫𑀢𑀺, 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀦 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀸𑀢𑀸𑀮𑁂 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺, 𑀕𑀸𑀥𑀜𑁆𑀘 𑀅𑀚𑁆𑀛𑀕𑀸’𑀢𑀺𑁇

1110\. _𑀬𑁄 𑀏𑀢𑀸 𑀦𑀸𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀺,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀔𑀸;_  
_𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸 𑀧𑀸𑀡𑀳𑀭𑀸,_  
_𑀬𑀸𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑁄 𑀧𑀯𑁂𑀥𑀢𑀺𑁇_  


1111\. _𑀅𑀓𑁆𑀓𑀦𑁆𑀤𑀢𑀺 𑀧𑀭𑁄𑀤𑀢𑀺,_  
_𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀅𑀧𑁆𑀧𑀣𑀸𑀫𑀓𑁄;_  
_𑀦 𑀲𑁄 𑀧𑀸𑀢𑀸𑀮𑁂 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺,_  
_𑀅𑀣𑁄 𑀕𑀸𑀥𑀫𑁆𑀧𑀺 𑀦𑀸𑀚𑁆𑀛𑀕𑀸𑁇_  


1112\. _𑀬𑁄 𑀘𑁂𑀢𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀺,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀔𑀸;_  
_𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸 𑀧𑀸𑀡𑀳𑀭𑀸,_  
_𑀬𑀸𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑁄 𑀦 𑀯𑁂𑀥𑀢𑀺;_  
_𑀲 𑀯𑁂 𑀧𑀸𑀢𑀸𑀮𑁂 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺,_  
_𑀅𑀣𑁄 𑀕𑀸𑀥𑀫𑁆𑀧𑀺 𑀅𑀚𑁆𑀛𑀕𑀸”𑀢𑀺𑁇_
