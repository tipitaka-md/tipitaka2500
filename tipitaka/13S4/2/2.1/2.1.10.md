# 𑁨𑁇𑁧𑁇𑁧𑁦 𑀨𑀲𑁆𑀲𑀫𑀽𑀮𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

### 𑀨𑀲𑁆𑀲𑀫𑀽𑀮𑀓𑀲𑀼𑀢𑁆𑀢

(258.)

1145. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸 𑀨𑀲𑁆𑀲𑀚𑀸 𑀨𑀲𑁆𑀲𑀫𑀽𑀮𑀓𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀤𑀸𑀦𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀲𑁆𑀲 𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸, 𑀬𑀁 𑀢𑀚𑁆𑀚𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀁 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀲𑀸 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀲𑁆𑀲 𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸, 𑀬𑀁 𑀢𑀚𑁆𑀚𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀁 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀲𑀸 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀲𑁆𑀲 𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸, 𑀬𑀁 𑀢𑀚𑁆𑀚𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀺𑀬𑀁 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀲𑀸 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀝𑁆𑀝𑀦𑀲𑀫𑁄𑀥𑀸𑀦𑀸 𑀉𑀲𑁆𑀫𑀸 𑀚𑀸𑀬𑀢𑀺, 𑀢𑁂𑀚𑁄 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀸𑀦𑀸𑀪𑀸𑀯𑀸 𑀯𑀺𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀸, 𑀬𑀸 𑀢𑀚𑁆𑀚𑀸 𑀉𑀲𑁆𑀫𑀸, 𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀲𑀸 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀨𑀲𑁆𑀲𑀚𑀸 𑀨𑀲𑁆𑀲𑀫𑀽𑀮𑀓𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀤𑀸𑀦𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀢𑀚𑁆𑀚𑀁 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀚𑁆𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑀚𑁆𑀚𑀲𑁆𑀲 𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀢𑀚𑁆𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
