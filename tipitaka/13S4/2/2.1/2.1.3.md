# 𑁨𑁇𑁧𑁇𑁩 𑀧𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

### 𑀧𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(251.)

1102. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀲𑀼𑀔𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀲𑁄 𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀢𑀡𑁆𑀳𑀁, 𑀯𑀺𑀯𑀢𑁆𑀢𑀬𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀫𑁆𑀫𑀸 𑀫𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀸 𑀅𑀦𑁆𑀢𑀫𑀓𑀸𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

1103. _𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀯𑁂𑀤𑀦𑀁 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀢𑁄;_  
_𑀲𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀳𑁄𑀢𑀺,_  
_𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀤𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


1104. _𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀯𑁂𑀤𑀦𑀁 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀢𑁄;_  
_𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀳𑁄𑀢𑀺,_  
_𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀤𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


1105. _𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀢𑀜𑁆𑀘𑀸𑀧𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺,_  
_𑀦𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1106. _𑀬𑀢𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀢𑀸𑀧𑀻,_  
_𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀦 𑀭𑀺𑀜𑁆𑀘𑀢𑀺;_  
_𑀢𑀢𑁄 𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


1107. _𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀤𑀺𑀝𑁆𑀞𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀦𑁄𑀧𑁂𑀢𑀺 𑀯𑁂𑀤𑀕𑀽”𑀢𑀺𑁇_  


---

1108. 𑀢𑀢𑀺𑀬𑀁𑁇
