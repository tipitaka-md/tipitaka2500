# 𑁨𑁇𑁧𑁇𑁬 𑀲𑀮𑁆𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

* 𑀲𑀮𑁆𑀮𑀲𑀼𑀢𑁆𑀢

(254.)

1118\. “𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀲𑀼𑀢𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑁄 𑀯𑀺𑀲𑁂𑀲𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀬𑀸𑀲𑁄 𑀓𑀺𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀦𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑁄𑀘𑀢𑀺 𑀓𑀺𑀮𑀫𑀢𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀤𑁆𑀯𑁂 𑀯𑁂𑀤𑀦𑀸 𑀯𑁂𑀤𑀬𑀢𑀺—  𑀓𑀸𑀬𑀺𑀓𑀜𑁆𑀘, 𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀲𑀮𑁆𑀮𑁂𑀦 𑀯𑀺𑀚𑁆𑀛𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀲𑀮𑁆𑀮𑁂𑀦 𑀅𑀦𑀼𑀯𑁂𑀥𑀁 𑀯𑀺𑀚𑁆𑀛𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑁆𑀯𑀺𑀲𑀮𑁆𑀮𑁂𑀦 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑁄𑀘𑀢𑀺 𑀓𑀺𑀮𑀫𑀢𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀤𑁆𑀯𑁂 𑀯𑁂𑀤𑀦𑀸 𑀯𑁂𑀤𑀬𑀢𑀺—  𑀓𑀸𑀬𑀺𑀓𑀜𑁆𑀘, 𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘𑁇 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀧𑀝𑀺𑀖𑀯𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀯𑀦𑁆𑀢𑀁, 𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄, 𑀲𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀦 𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀸𑀫𑀲𑀼𑀔𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁, 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀲𑀼𑀔𑀜𑁆𑀘 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑁄, 𑀬𑁄 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄, 𑀲𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀸𑀲𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀸𑀲𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀢𑁄, 𑀬𑁄 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄, 𑀲𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

1119\. 𑀲𑀼𑀢𑀯𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀦 𑀲𑁄𑀘𑀢𑀺, 𑀦 𑀓𑀺𑀮𑀫𑀢𑀺, 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀦 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀏𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺—  𑀓𑀸𑀬𑀺𑀓𑀁, 𑀦 𑀘𑁂𑀢𑀲𑀺𑀓𑀁𑁇

1120\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀲𑀮𑁆𑀮𑁂𑀦 𑀯𑀺𑀚𑁆𑀛𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀲𑀮𑁆𑀮𑁂𑀦 𑀅𑀦𑀼𑀯𑁂𑀥𑀁 𑀦 𑀯𑀺𑀚𑁆𑀛𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀏𑀓𑀲𑀮𑁆𑀮𑁂𑀦 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀦 𑀲𑁄𑀘𑀢𑀺, 𑀦 𑀓𑀺𑀮𑀫𑀢𑀺, 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀦 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀏𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺—  𑀓𑀸𑀬𑀺𑀓𑀁, 𑀦 𑀘𑁂𑀢𑀲𑀺𑀓𑀁𑁇 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀧𑀝𑀺𑀖𑀯𑀸 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀯𑀦𑁆𑀢𑀁, 𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄, 𑀲𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀸𑀫𑀲𑀼𑀔𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀲𑀼𑀔𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑁄 𑀬𑁄 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄, 𑀲𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀸𑀲𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀁 𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀸𑀲𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀢𑁄, 𑀬𑁄 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄, 𑀲𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀯𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀯𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀲𑁂𑀲𑁄, 𑀅𑀬𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑀬𑀸𑀲𑁄, 𑀇𑀤𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀦𑀸𑀢𑀺𑁇

1121\. _𑀦 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺 𑀲𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄𑀧𑀺;_  
_𑀅𑀬𑀜𑁆𑀘 𑀥𑀻𑀭𑀲𑁆𑀲 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀦,_  
_𑀫𑀳𑀸 𑀯𑀺𑀲𑁂𑀲𑁄 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇_  


1122\. _𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑀫𑀺𑀫𑀁 𑀧𑀭𑀜𑁆𑀘;_  
_𑀇𑀝𑁆𑀞𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀣𑁂𑀦𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀅𑀦𑀺𑀝𑁆𑀞𑀢𑁄 𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑀫𑁂𑀢𑀺𑁇_  


1123\. _𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀭𑁄𑀥𑀸 𑀅𑀣𑀯𑀸 𑀯𑀺𑀭𑁄𑀥𑀸,_  
_𑀯𑀺𑀥𑀽𑀧𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀕𑀢𑀸 𑀦 𑀲𑀦𑁆𑀢𑀺;_  
_𑀧𑀤𑀜𑁆𑀘 𑀜𑀢𑁆𑀯𑀸 𑀯𑀺𑀭𑀚𑀁 𑀅𑀲𑁄𑀓𑀁,_  
_𑀲𑀫𑁆𑀫𑀸 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀪𑀯𑀲𑁆𑀲 𑀧𑀸𑀭𑀕𑀽”𑀢𑀺𑁇_
