# 𑁨𑁇𑁧𑁇𑁫 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

* 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

(253.)

1114\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀲𑀼𑀔𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀲𑀮𑁆𑀮𑀢𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀲𑀮𑁆𑀮𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀲𑁄 𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀢𑀡𑁆𑀳𑀁, 𑀯𑀺𑀯𑀢𑁆𑀢𑀬𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀫𑁆𑀫𑀸 𑀫𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀸 𑀅𑀦𑁆𑀢𑀫𑀓𑀸𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

1115\. _𑀬𑁄 𑀲𑀼𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀤𑁆𑀤,_  
_𑀤𑀼𑀓𑁆𑀔𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀲𑀮𑁆𑀮𑀢𑁄;_  
_𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀦𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄𑁇_  


1116\. _𑀲 𑀯𑁂 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀯𑁂𑀤𑀦𑀸;_  
_𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀤𑀺𑀝𑁆𑀞𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀦𑁄𑀧𑁂𑀢𑀺 𑀯𑁂𑀤𑀕𑀽”𑀢𑀺𑁇_  


---

1117\. 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
