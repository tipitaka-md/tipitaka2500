# 𑁨𑁇𑁧𑁇𑁨 𑀲𑀼𑀔𑀲𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀔𑀲𑀼𑀢𑁆𑀢

(250.)

1098\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀢𑀺𑁇

1099\. _𑀲𑀼𑀔𑀁 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀲𑀳;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀜𑁆𑀘 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀢𑁆𑀣𑀺 𑀯𑁂𑀤𑀺𑀢𑀁𑁇_  


1100\. _𑀏𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀜𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑁄𑀲𑀥𑀫𑁆𑀫𑀁 𑀧𑀮𑁄𑀓𑀺𑀦𑀁;_  
_𑀨𑀼𑀲𑁆𑀲 𑀨𑀼𑀲𑁆𑀲 𑀯𑀬𑀁 𑀧𑀲𑁆𑀲𑀁,_  
_𑀏𑀯𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀭𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_
