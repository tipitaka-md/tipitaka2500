# 𑁨𑁇𑁧𑁇𑁯 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀕𑀸𑀣𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢

(257.)

1143\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀸”𑀢𑀺𑁇
