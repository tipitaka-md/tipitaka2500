# 𑁪𑁇𑁧𑁇𑁯 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀜𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀚𑀫𑁆𑀩𑀼𑀔𑀸𑀤𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀚𑀫𑁆𑀩𑀼𑀔𑀸𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀜𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(322.)

1344\. “‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑀢𑀺? “𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀡𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀢𑀫𑁄 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀓𑀢𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺? “𑀅𑀬𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀬𑀁 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀪𑀤𑁆𑀤𑀓𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀪𑀤𑁆𑀤𑀺𑀓𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬𑁇 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀸𑀬𑀸”𑀢𑀺𑁇
