# 𑁪𑁇𑁧𑁇𑁫 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀚𑀫𑁆𑀩𑀼𑀔𑀸𑀤𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀚𑀫𑁆𑀩𑀼𑀔𑀸𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(318.)

1336. “‘𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀬𑀢𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀦𑀁 𑀨𑀲𑁆𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀢𑀫𑁄 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀓𑀢𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺? “𑀅𑀬𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀬𑀁 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀪𑀤𑁆𑀤𑀓𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀪𑀤𑁆𑀤𑀺𑀓𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀸𑀲𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀸𑀬𑀸”𑀢𑀺𑁇
