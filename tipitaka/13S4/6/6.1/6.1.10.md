# 𑁬𑁇𑁧𑁇𑁧𑁦 𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(341.)

1401\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1402\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1403\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖…𑀧𑁂…  𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1404\. ( )



1405\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀙𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀢𑁆𑀢𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀝𑁆𑀞𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀤𑁂𑀯𑀢𑀸𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1406\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1407\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1409\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1410\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1411\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1412\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀉𑀚𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀜𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀏𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1413\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺 𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑁂𑀳𑀺 𑀅𑀲𑀩𑀮𑁂𑀳𑀺 𑀅𑀓𑀫𑁆𑀫𑀸𑀲𑁂𑀳𑀺 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑁂𑀳𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1414\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1415\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1416\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1417\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1419\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀙𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂… 𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀢𑁆𑀢𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂… 𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀝𑁆𑀞𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂… 𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀤𑁂𑀯𑀢𑀸𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1420\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1421\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1422\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1423\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1424\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1425\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1426\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1427\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1429\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1430\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1431\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1432\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇

1433\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1434\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

1435\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇

1437\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀙𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀢𑁆𑀢𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀝𑁆𑀞𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀤𑁂𑀯𑀢𑀸𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1438\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1439\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

1440\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇

1441\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇

1443\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1444\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1445\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1446\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1447\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇

1448\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1449\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1450\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1451\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇

1453\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀙𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀢𑁆𑀢𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀝𑁆𑀞𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀤𑁂𑀯𑀢𑀸𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1454\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1455\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1456\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀉𑀚𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀜𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀏𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1457\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺 𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑁂𑀳𑀺 𑀅𑀲𑀩𑀮𑁂𑀳𑀺 𑀅𑀓𑀫𑁆𑀫𑀸𑀲𑁂𑀳𑀺 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑁂𑀳𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇

1458\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1459\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1460\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸…𑀧𑁂…  𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀺𑁇

1461\. 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀤𑀲𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺—  𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀬𑀼𑀦𑀸, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀲𑀼𑀔𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀲𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀭𑀲𑁂𑀳𑀺, 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀳𑀻”𑀢𑀺𑁇
