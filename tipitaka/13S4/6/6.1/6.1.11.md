# 𑁬𑁇𑁧𑁇𑁧𑁧 𑀘𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀘𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

(342.)

1463\. 𑀅𑀣 𑀔𑁄 𑀘𑀦𑁆𑀤𑀦𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄…𑀧𑁂… 𑁇

1464\. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀬𑀸𑀫𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄…𑀧𑁂… 𑁇

1465\. 𑀅𑀣 𑀔𑁄 𑀲𑀦𑁆𑀢𑀼𑀲𑀺𑀢𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄…𑀧𑁂… 𑁇

1466\. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄…𑀧𑁂… 𑁇

1467\. 𑀅𑀣 𑀔𑁄 𑀯𑀲𑀯𑀢𑁆𑀢𑀺 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄…𑀧𑁂… 𑁇

1468\. (𑀬𑀣𑀸 𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀢𑀣𑀸 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇)

1471\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1472\. _𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀜𑁆𑀘,_  
_𑀲𑀼𑀔𑁂𑀦 𑀘 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄;_  
_𑀆𑀓𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀆𑀓𑀺𑀜𑁆𑀘𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀺𑀦𑀸;_  
_𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀘 𑀲𑀓𑁆𑀓𑁄 𑀘,_  
_𑀘𑀦𑁆𑀤𑀦𑁂𑀓𑀸𑀤𑀲𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_
