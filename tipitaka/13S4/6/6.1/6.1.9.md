# 𑁬𑁇𑁧𑁇𑁯 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀧𑀜𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀧𑀜𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(340.)

1398\. “‘𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀻’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀻𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀺𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀲𑀸𑀭𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺𑁇

1399\. 𑀅𑀣 𑀔𑁄 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁇 𑀫𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀁 𑀧𑀫𑀸𑀤𑁄, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀡𑁆𑀞𑀧𑁂𑀳𑀺, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑁄𑀤𑀺𑀁 𑀓𑀭𑁄𑀳𑀺, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀳𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀣𑀸𑀭𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀲𑀸𑀯𑀓𑁄 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄’𑀢𑀺, 𑀫𑀫𑀁 𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀣𑀸𑀭𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀲𑀸𑀯𑀓𑁄 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄’”𑀢𑀺𑁇
