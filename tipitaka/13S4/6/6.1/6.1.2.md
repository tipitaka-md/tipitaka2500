# 𑁬𑁇𑁧𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀛𑀸𑀦𑀧𑀜𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀛𑀸𑀦𑀧𑀜𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(333.)

1377\. “‘𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀦𑁆’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀦𑁆𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀺𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀭𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇

1378\. 𑀅𑀣 𑀔𑁄 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁇 𑀫𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀫𑀸𑀤𑁄, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀛𑀸𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀡𑁆𑀞𑀧𑁂𑀳𑀺, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀛𑀸𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑁄𑀤𑀺𑀁 𑀓𑀭𑁄𑀳𑀺, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀛𑀸𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀳𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀣𑀸𑀭𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀲𑀸𑀯𑀓𑁄 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄’𑀢𑀺, 𑀫𑀫𑀁 𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀣𑀸𑀭𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀲𑀸𑀯𑀓𑁄 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄’”𑀢𑀺𑁇
