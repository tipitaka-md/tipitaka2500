# 𑁧𑁦𑁇𑁧𑁇𑁮 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀯𑀕𑁆𑀕

### 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(417.)

1808\. 𑀅𑀣 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀯𑀘𑁆𑀙, 𑀫𑀬𑀸—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺…𑀧𑁂… 𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺? “𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀫𑀬𑀸—  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇

1809\. “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸? 𑀓𑁄 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀪𑁄𑀢𑁄 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀦 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺𑀧𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀻”𑀢𑀺?

1810\. “𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁, 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀯𑁂𑀤𑀦𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀢𑀲𑁆𑀫𑀸 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀦 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀦 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀦 𑀯𑁂𑀤𑀦𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀦 𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀦 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀦 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺𑀧𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀻”𑀢𑀺𑁇

1811\. 𑀅𑀣 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀯𑀘𑁆𑀙, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺…𑀧𑁂… 𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺? “𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇

1812\. “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸? 𑀓𑁄 𑀧𑀦, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀦 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺𑀧𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀻”𑀢𑀺?

1813\. “𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁, 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀯𑁂𑀤𑀦𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀢𑀲𑁆𑀫𑀸 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀦 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀦 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀦 𑀯𑁂𑀤𑀦𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀦 𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀦 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀦 𑀏𑀯𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺𑀧𑀺, ‘𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺𑀧𑀺, ‘𑀅𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄’𑀢𑀺𑀧𑀺, ‘𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄’𑀢𑀺𑀧𑀺, ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆’𑀢𑀺𑀧𑀺, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆’𑀢𑀺𑀧𑀺, ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺, ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺, ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀻”𑀢𑀺𑁇

1814\. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀣𑀼 𑀘 𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀘 𑀅𑀢𑁆𑀣𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀦 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀫𑁂𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀯𑀺𑀭𑁄𑀥𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁 𑀅𑀕𑁆𑀕𑀧𑀤𑀲𑁆𑀫𑀺𑀁𑁇 𑀇𑀤𑀸𑀦𑀸𑀳𑀁, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑁇 𑀲𑀫𑀡𑁄𑀧𑀺 𑀫𑁂 𑀕𑁄𑀢𑀫𑁄 𑀏𑀢𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀏𑀢𑁂𑀳𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀪𑀯𑀁 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄𑁇 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀣𑀼 𑀘 𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀘 𑀅𑀢𑁆𑀣𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀦 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀯𑀺𑀭𑁄𑀥𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁 𑀅𑀕𑁆𑀕𑀧𑀤𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇
