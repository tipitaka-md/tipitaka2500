# 𑁧𑁦𑁇𑁧𑁇𑁪 𑀤𑀼𑀢𑀺𑀬𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(413.)

1786\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂…𑀧𑁂…  (𑀲𑀸𑀬𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀸𑁇) “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑁂𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺? “𑀭𑀽𑀧𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀭𑀽𑀧𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺; ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺; ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺; ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀚𑀸𑀦𑀢𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺; ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺; ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺; ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇

1787\. 𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀭𑀽𑀧𑀲𑀫𑀼𑀤𑀬𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁, ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺; ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺; ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺; ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑁂𑀢𑀼 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑁂𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇
