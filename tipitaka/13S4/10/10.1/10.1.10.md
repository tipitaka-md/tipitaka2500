# 𑁧𑁦𑁇𑁧𑁇𑁧𑁦 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀯𑀕𑁆𑀕

### 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(419.)

1823. 𑀅𑀣 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀢𑁆𑀣𑀢𑁆𑀢𑀸”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀢𑁆𑀣𑀢𑁆𑀢𑀸”𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1824. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀦 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀻”𑀢𑀺? “𑀅𑀳𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 ‘𑀅𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁, 𑀬𑁂 𑀢𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀲𑁆𑀲𑀢𑀯𑀸𑀤𑀸 𑀢𑁂𑀲𑀫𑁂𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀯𑀺𑀲𑁆𑀲𑁇 𑀅𑀳𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 ‘𑀦𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁, 𑀬𑁂 𑀢𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀘𑁆𑀙𑁂𑀤𑀯𑀸𑀤𑀸 𑀢𑁂𑀲𑀫𑁂𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀯𑀺𑀲𑁆𑀲𑁇 𑀅𑀳𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 ‘𑀅𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀫𑁂 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀅𑀪𑀯𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬—  ‘𑀲𑀩𑁆𑀩𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀅𑀳𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 ‘𑀦𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀢𑁆𑀢𑀸’𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁, 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑀲𑁆𑀲, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑀫𑁆𑀫𑁄𑀳𑀸𑀬 𑀅𑀪𑀯𑀺𑀲𑁆𑀲—  ‘𑀅𑀳𑀼𑀯𑀸 𑀫𑁂 𑀦𑀽𑀦 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀢𑁆𑀢𑀸, 𑀲𑁄 𑀏𑀢𑀭𑀳𑀺 𑀦𑀢𑁆𑀣𑀻’”𑀢𑀺𑁇
