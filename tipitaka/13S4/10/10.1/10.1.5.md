# 𑁧𑁦𑁇𑁧𑁇𑁫 𑀢𑀢𑀺𑀬𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀯𑀕𑁆𑀕

### 𑀢𑀢𑀺𑀬𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(414.)

1789\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂…𑀧𑁂…  (𑀲𑀸𑀬𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀸𑁇) “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑁂𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺? “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀯𑀺𑀕𑀢𑀭𑀸𑀕𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀘𑁆𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀧𑀺𑀧𑀸𑀲𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀢𑀡𑁆𑀳𑀲𑁆𑀲 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀅𑀯𑀺𑀕𑀢𑀭𑀸𑀕𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀘𑁆𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀧𑀺𑀧𑀸𑀲𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀲𑁆𑀲 𑀅𑀯𑀺𑀕𑀢𑀢𑀡𑁆𑀳𑀲𑁆𑀲 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇 𑀭𑀽𑀧𑁂 𑀘 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀕𑀢𑀭𑀸𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀯𑀺𑀕𑀢𑀭𑀸𑀕𑀲𑁆𑀲 𑀯𑀺𑀕𑀢𑀘𑁆𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑀲𑁆𑀲 𑀯𑀺𑀕𑀢𑀧𑀺𑀧𑀸𑀲𑀲𑁆𑀲 𑀯𑀺𑀕𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀲𑁆𑀲 𑀯𑀺𑀕𑀢𑀢𑀡𑁆𑀳𑀲𑁆𑀲 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑁂𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇
