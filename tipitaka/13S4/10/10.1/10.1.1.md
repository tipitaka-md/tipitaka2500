# 𑁧𑁦𑁇𑁧𑁇𑁧 𑀔𑁂𑀫𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀯𑀕𑁆𑀕

### 𑀔𑁂𑀫𑀸𑀲𑀼𑀢𑁆𑀢

(410.)

1749\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀔𑁂𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑀸 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀲𑀸𑀓𑁂𑀢𑀁 𑀢𑁄𑀭𑀡𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀲𑀸𑀓𑁂𑀢𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄, 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀲𑀸𑀓𑁂𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀢𑁄𑀭𑀡𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀏𑀓𑀭𑀢𑁆𑀢𑀺𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀢𑁄𑀭𑀡𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀲𑀫𑀡𑀁 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀯𑀸 𑀚𑀸𑀦 𑀬𑀫𑀳𑀁 𑀅𑀚𑁆𑀚 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

1750\. “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀢𑁄𑀭𑀡𑀯𑀢𑁆𑀣𑀼𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄 𑀦𑀸𑀤𑁆𑀤𑀲 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀲𑀫𑀡𑀁 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀯𑀸 𑀬𑀁 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀔𑁂𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀢𑁄𑀭𑀡𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀧𑀲𑁂𑀦𑀤𑀺𑀁 𑀓𑁄𑀲𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1751\. “𑀦𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀤𑁂𑀯, 𑀢𑁄𑀭𑀡𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀢𑀣𑀸𑀭𑀽𑀧𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀬𑀁 𑀤𑁂𑀯𑁄 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄, 𑀤𑁂𑀯, 𑀔𑁂𑀫𑀸 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻, 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀺𑀓𑀸 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲𑀸 𑀔𑁄 𑀧𑀦 𑀅𑀬𑁆𑀬𑀸𑀬 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀧𑀡𑁆𑀟𑀺𑀢𑀸, 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀓𑀣𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀝𑀺𑀪𑀸𑀦𑀸’𑀢𑀺𑁇 𑀢𑀁 𑀤𑁂𑀯𑁄 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀽”𑀢𑀺𑁇

1752\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀬𑁂𑀦 𑀔𑁂𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀔𑁂𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀔𑁂𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑀬𑁆𑀬𑁂, 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑀬𑁆𑀬𑁂, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺𑁇 “𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇

1753\. “‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸—  ‘𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦𑀬𑁆𑀬𑁂, 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸—  ‘𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸—  ‘𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦𑀬𑁆𑀬𑁂, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸—  ‘𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦𑁂𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺?

1754\. “𑀢𑁂𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬 𑀢𑀣𑀸 𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀓𑁄𑀘𑀺 𑀕𑀡𑀓𑁄 𑀯𑀸 𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑁄 𑀯𑀸 𑀲𑀗𑁆𑀔𑀸𑀬𑀓𑁄 𑀯𑀸 𑀬𑁄 𑀧𑀳𑁄𑀢𑀺 𑀕𑀗𑁆𑀕𑀸𑀬 𑀯𑀸𑀮𑀼𑀓𑀁 𑀕𑀡𑁂𑀢𑀼𑀁—  𑀏𑀢𑁆𑀢𑀓𑀸 𑀯𑀸𑀮𑀼𑀓𑀸 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀸𑀮𑀼𑀓𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀸𑀮𑀼𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀸𑀮𑀼𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀅𑀬𑁆𑀬𑁂”𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦 𑀢𑁂 𑀓𑁄𑀘𑀺 𑀕𑀡𑀓𑁄 𑀯𑀸 𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑁄 𑀯𑀸 𑀲𑀗𑁆𑀔𑀸𑀬𑀓𑁄 𑀯𑀸 𑀬𑁄 𑀧𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁 𑀕𑀡𑁂𑀢𑀼𑀁—  𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀍𑀳𑀓𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀍𑀳𑀓𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀍𑀳𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀍𑀳𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀅𑀬𑁆𑀬𑁂”𑁇 “𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼”? “𑀫𑀳𑀸𑀬𑁆𑀬𑁂, 𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂𑀦 𑀭𑀽𑀧𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀁 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀁 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀁 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀁 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀁𑁇 𑀭𑀽𑀧𑀲𑀗𑁆𑀔𑀸𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑁇 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺𑁇

1755\. 𑀬𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬, 𑀲𑀸 𑀯𑁂𑀤𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀯𑁂𑀤𑀦𑀸𑀲𑀗𑁆𑀔𑀸𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑁇 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺𑁇

1756\. 𑀬𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀢𑀣𑀸𑀕𑀢𑀁…𑀧𑁂…  𑀬𑁂𑀳𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬, 𑀢𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀗𑁆𑀔𑀸𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑁇 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺𑁇

1757\. 𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬 𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀁 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀁 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀁 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀁 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀁𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀗𑁆𑀔𑀸𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑁇 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀔𑁂𑀫𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀔𑁂𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1758\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀬𑀸—  ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀫𑀬𑀸—  ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀬𑀸—  ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀫𑀬𑀸—  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄—  ‘𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀬𑀸—  𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺…𑀧𑁂… 𑁇 “‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄—  ‘𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀫𑀬𑀸—  𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀤𑁂𑀲𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑁂𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺?

1759\. “𑀢𑁂𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬 𑀢𑀣𑀸 𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀓𑁄𑀘𑀺 𑀕𑀡𑀓𑁄 𑀯𑀸 𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑁄 𑀯𑀸 𑀲𑀗𑁆𑀔𑀸𑀬𑀓𑁄 𑀯𑀸 𑀬𑁄 𑀧𑀳𑁄𑀢𑀺 𑀕𑀗𑁆𑀕𑀸𑀬 𑀯𑀸𑀮𑀼𑀓𑀁 𑀕𑀡𑁂𑀢𑀼𑀁—  𑀏𑀢𑁆𑀢𑀓𑀸 𑀯𑀸𑀮𑀼𑀓𑀸 𑀇𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀸𑀮𑀼𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦 𑀢𑁂 𑀓𑁄𑀘𑀺 𑀕𑀡𑀓𑁄 𑀯𑀸 𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑁄 𑀯𑀸 𑀲𑀗𑁆𑀔𑀸𑀬𑀓𑁄 𑀯𑀸 𑀬𑁄 𑀧𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁 𑀕𑀡𑁂𑀢𑀼𑀁—  𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀍𑀳𑀓𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀍𑀳𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼”? “𑀫𑀳𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂𑀦 𑀭𑀽𑀧𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬, 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀁 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀁 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀁 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀁 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀁𑁇 𑀭𑀽𑀧𑀲𑀗𑁆𑀔𑀸𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑁇 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺…𑀧𑁂…  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺𑁇 𑀬𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀬𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀬𑁂𑀳𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀳𑀺…𑀧𑁂… 𑁇

1760\. 𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬, 𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀁 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀁 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀁 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀁 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀁𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀗𑁆𑀔𑀸𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀧𑁆𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑁇 ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀻”𑀢𑀺𑁇

1761\. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀲𑀸𑀯𑀺𑀓𑀸𑀬 𑀘 𑀅𑀢𑁆𑀣𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀦 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀫𑁂𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀯𑀺𑀭𑁄𑀥𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀬𑀤𑀺𑀤𑀁 𑀅𑀕𑁆𑀕𑀧𑀤𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀫𑀺𑀤𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑀁 𑀔𑁂𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑁇 𑀲𑀸𑀧𑀺 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀏𑀢𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀏𑀢𑁂𑀳𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀪𑀕𑀯𑀸𑁇 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀲𑀸𑀯𑀺𑀓𑀸𑀬 𑀘 𑀅𑀢𑁆𑀣𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀦 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀫𑁂𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀯𑀺𑀭𑁄𑀥𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀬𑀤𑀺𑀤𑀁 𑀅𑀕𑁆𑀕𑀧𑀤𑀲𑁆𑀫𑀺𑀁𑁇 𑀳𑀦𑁆𑀤 𑀤𑀸𑀦𑀺 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀘𑁆𑀙𑀸𑀫𑁇 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑀸 𑀫𑀬𑀁 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑀸”𑀢𑀺𑁇 “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀻𑀢𑀺𑁇
