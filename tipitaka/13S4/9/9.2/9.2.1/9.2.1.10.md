# 𑁯𑁇𑁨𑁇𑁧𑁇𑁧𑁦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕

### 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕

1705\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁…𑀧𑁂…  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀁𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂… 𑁇 (𑁧𑁇𑁩𑁧--𑁩𑁭)
