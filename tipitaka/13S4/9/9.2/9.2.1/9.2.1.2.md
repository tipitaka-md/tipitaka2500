# 𑁯𑁇𑁨𑁇𑁧𑁇𑁨 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸

### 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸

1697\. 𑀅𑀲𑀗𑁆𑀔𑀢𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀁𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀁? 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀕𑀓𑁆𑀔𑀬𑁄 𑀤𑁄𑀲𑀓𑁆𑀔𑀬𑁄 𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀁𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄𑁇 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑀺𑀢𑀁 𑀯𑁄 𑀫𑀬𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁…𑀧𑁂…  𑀅𑀬𑀁 𑀯𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀢𑀺𑁇 (𑁧𑁇𑁨)
