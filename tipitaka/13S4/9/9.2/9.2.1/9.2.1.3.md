# 𑁯𑁇𑁨𑁇𑁧𑁇𑁩 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀫𑀸𑀥𑀺

### 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀫𑀸𑀥𑀺

1698. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂… 𑁇 (𑁧𑁇𑁩--𑁫)
