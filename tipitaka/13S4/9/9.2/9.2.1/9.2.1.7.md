# 𑁯𑁇𑁨𑁇𑁧𑁇𑁭 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤

* 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤

1702\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀻𑀫𑀁𑀲𑀲𑀫𑀸𑀥𑀺𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀗𑁆𑀔𑀢𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄…𑀧𑁂… 𑁇 (𑁧𑁇𑁧𑁭--𑁨𑁦)
