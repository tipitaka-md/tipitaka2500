# 𑁯𑁇𑁨𑁇𑁩𑁩 𑀧𑀭𑀸𑀬𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀲𑀗𑁆𑀔𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀭𑀸𑀬𑀦𑀲𑀼𑀢𑁆𑀢

(409.)

1741\. “𑀧𑀭𑀸𑀬𑀦𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀭𑀸𑀬𑀦𑀕𑀸𑀫𑀺𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀁𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀸𑀬𑀦𑀁? 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀕𑀓𑁆𑀔𑀬𑁄 𑀤𑁄𑀲𑀓𑁆𑀔𑀬𑁄 𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀸𑀬𑀦𑀁𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀸𑀬𑀦𑀕𑀸𑀫𑀻 𑀫𑀕𑁆𑀕𑁄? 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀸𑀬𑀦𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄𑁇 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑀺𑀢𑀁 𑀯𑁄 𑀫𑀬𑀸 𑀧𑀭𑀸𑀬𑀦𑀁, 𑀤𑁂𑀲𑀺𑀢𑁄 𑀧𑀭𑀸𑀬𑀦𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁄𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀣𑀸𑀭𑀸 𑀓𑀭𑀡𑀻𑀬𑀁 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁂𑀦 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀢𑀁 𑀯𑁄 𑀢𑀁 𑀫𑀬𑀸𑁇 𑀏𑀢𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀸𑀦𑀺, 𑀏𑀢𑀸𑀦𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀸𑀦𑀺𑁇 𑀛𑀸𑀬𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸 𑀧𑀫𑀸𑀤𑀢𑁆𑀣; 𑀫𑀸 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄 𑀅𑀳𑀼𑀯𑀢𑁆𑀣𑁇 𑀅𑀬𑀁 𑀯𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻”𑀢𑀺𑁇 (𑀬𑀣𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀢𑀣𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇) (𑁩𑁩)

1743\. 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄𑁇

1744\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1745\. _𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀅𑀦𑀢𑀁 𑀅𑀦𑀸𑀲𑀯𑀁,_  
_𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀧𑀸𑀭𑀁 𑀦𑀺𑀧𑀼𑀡𑀁 𑀲𑀼𑀤𑀼𑀤𑁆𑀤𑀲𑀁;_  
_𑀅𑀚𑀚𑁆𑀚𑀭𑀁 𑀥𑀼𑀯𑀁 𑀅𑀧𑀮𑁄𑀓𑀺𑀢𑀁,_  
_𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘 𑀲𑀦𑁆𑀢𑀁𑁇_  


1746\. _𑀅𑀫𑀢𑀁 𑀧𑀡𑀻𑀢𑀜𑁆𑀘 𑀲𑀺𑀯𑀜𑁆𑀘 𑀔𑁂𑀫𑀁,_  
_𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀜𑁆𑀘 𑀅𑀩𑁆𑀪𑀼𑀢𑀁;_  
_𑀅𑀦𑀻𑀢𑀺𑀓𑀁 𑀅𑀦𑀻𑀢𑀺𑀓𑀥𑀫𑁆𑀫𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑀼𑀕𑀢𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁𑁇_  


1747\. _𑀅𑀩𑁆𑀬𑀸𑀩𑀚𑁆𑀛𑁄 𑀯𑀺𑀭𑀸𑀕𑁄 𑀘,_  
_𑀲𑀼𑀤𑁆𑀥𑀺 𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀦𑀸𑀮𑀬𑁄;_  
_𑀤𑀻𑀧𑁄 𑀮𑁂𑀡𑀜𑁆𑀘 𑀢𑀸𑀡𑀜𑁆𑀘,_  
_𑀲𑀭𑀡𑀜𑁆𑀘 𑀧𑀭𑀸𑀬𑀦𑀦𑁆𑀢𑀺𑁇_
