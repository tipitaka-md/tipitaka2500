# 𑁧𑁇𑁪𑁇𑁪𑁇𑁯 𑀯𑀻𑀡𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀲𑀻𑀯𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀯𑀻𑀡𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(246.)

1067\. “𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀭𑀽𑀧𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀤𑁄𑀲𑁄 𑀯𑀸 𑀫𑁄𑀳𑁄 𑀯𑀸 𑀧𑀝𑀺𑀖𑀁 𑀯𑀸𑀧𑀺 𑀘𑁂𑀢𑀲𑁄, 𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑁇 𑀲𑀪𑀬𑁄 𑀘𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁄 𑀘 𑀲𑀓𑀡𑁆𑀝𑀓𑁄 𑀘 𑀲𑀕𑀳𑀦𑁄 𑀘 𑀉𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀘 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀘 𑀤𑀼𑀳𑀺𑀢𑀺𑀓𑁄 𑀘𑁇 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑁂𑀯𑀺𑀢𑁄 𑀘𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀦 𑀘𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀳𑀺 𑀲𑁂𑀯𑀺𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁 𑀏𑀢𑀁 𑀅𑀭𑀳𑀲𑀻𑀢𑀺𑁇 𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀯𑀸𑀭𑀬𑁂 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀳𑀺 𑀭𑀽𑀧𑁂𑀳𑀺…𑀧𑁂…  𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀭𑀲𑁂𑀲𑀼…𑀧𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀤𑁄𑀲𑁄 𑀯𑀸 𑀫𑁄𑀳𑁄 𑀯𑀸 𑀧𑀝𑀺𑀖𑀁 𑀯𑀸𑀧𑀺 𑀘𑁂𑀢𑀲𑁄 𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑁇 𑀲𑀪𑀬𑁄 𑀘𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁄 𑀘 𑀲𑀓𑀡𑁆𑀝𑀓𑁄 𑀘 𑀲𑀕𑀳𑀦𑁄 𑀘 𑀉𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀘 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀘 𑀤𑀼𑀳𑀺𑀢𑀺𑀓𑁄 𑀘𑁇 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑁂𑀯𑀺𑀢𑁄 𑀘𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀦 𑀘𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀳𑀺 𑀲𑁂𑀯𑀺𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁 𑀏𑀢𑀁 𑀅𑀭𑀳𑀲𑀻𑀢𑀺𑁇 𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀯𑀸𑀭𑀬𑁂 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺𑁇

1068\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀓𑀺𑀝𑁆𑀞𑀸𑀭𑀓𑁆𑀔𑁄 𑀘 𑀧𑀫𑀢𑁆𑀢𑁄, 𑀕𑁄𑀡𑁄 𑀘 𑀓𑀺𑀝𑁆𑀞𑀸𑀤𑁄 𑀅𑀤𑀼𑀁 𑀓𑀺𑀝𑁆𑀞𑀁 𑀑𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀙𑀲𑀼 𑀨𑀲𑁆𑀲𑀸𑀬𑀢𑀦𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀼𑀢𑀓𑀸𑀭𑀻 𑀧𑀜𑁆𑀘𑀲𑀼 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1069\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀓𑀺𑀝𑁆𑀞𑀸𑀭𑀓𑁆𑀔𑁄 𑀘 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀕𑁄𑀡𑁄 𑀘 𑀓𑀺𑀝𑁆𑀞𑀸𑀤𑁄 𑀅𑀤𑀼𑀁 𑀓𑀺𑀝𑁆𑀞𑀁 𑀑𑀢𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑁂𑀦𑀁 𑀓𑀺𑀝𑁆𑀞𑀸𑀭𑀓𑁆𑀔𑁄 𑀦𑀸𑀲𑀸𑀬𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑁇 𑀦𑀸𑀲𑀸𑀬𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀭𑀺𑀖𑀝𑀸𑀬𑀁 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑁇 𑀉𑀧𑀭𑀺𑀖𑀝𑀸𑀬𑀁 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀡𑁆𑀟𑁂𑀦 𑀲𑀼𑀢𑀸𑀍𑀇𑀢𑀁 𑀢𑀸𑀍𑀏𑀬𑁆𑀬𑁇 𑀤𑀡𑁆𑀟𑁂𑀦 𑀲𑀼𑀢𑀸𑀍𑀇𑀢𑀁 𑀢𑀸𑀍𑀏𑀢𑁆𑀯𑀸 𑀑𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀡𑁄 𑀓𑀺𑀝𑁆𑀞𑀸𑀤𑁄 𑀅𑀤𑀼𑀁 𑀓𑀺𑀝𑁆𑀞𑀁 𑀑𑀢𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑁂𑀦𑀁 𑀓𑀺𑀝𑁆𑀞𑀸𑀭𑀓𑁆𑀔𑁄 𑀦𑀸𑀲𑀸𑀬𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑁇 𑀦𑀸𑀲𑀸𑀬𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀭𑀺𑀖𑀝𑀸𑀬𑀁 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑁇 𑀉𑀧𑀭𑀺𑀖𑀝𑀸𑀬𑀁 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀡𑁆𑀟𑁂𑀦 𑀲𑀼𑀢𑀸𑀍𑀇𑀢𑀁 𑀢𑀸𑀍𑀏𑀬𑁆𑀬𑁇 𑀤𑀡𑁆𑀟𑁂𑀦 𑀲𑀼𑀢𑀸𑀍𑀇𑀢𑀁 𑀢𑀸𑀍𑀏𑀢𑁆𑀯𑀸 𑀑𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀡𑁄 𑀓𑀺𑀝𑁆𑀞𑀸𑀤𑁄 𑀕𑀸𑀫𑀕𑀢𑁄 𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸, 𑀞𑀸𑀦𑀩𑀳𑀼𑀮𑁄 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀦𑀺𑀲𑀚𑁆𑀚𑀩𑀳𑀼𑀮𑁄 𑀯𑀸 𑀦 𑀢𑀁 𑀓𑀺𑀝𑁆𑀞𑀁 𑀧𑀼𑀦 𑀑𑀢𑀭𑁂𑀬𑁆𑀬—  𑀢𑀫𑁂𑀯 𑀧𑀼𑀭𑀺𑀫𑀁 𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀲𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁄 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀙𑀲𑀼 𑀨𑀲𑁆𑀲𑀸𑀬𑀢𑀦𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀤𑀼𑀚𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀤𑀼𑀚𑀺𑀢𑀁, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀫𑁂𑀯 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀲𑀦𑁆𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀏𑀓𑁄𑀤𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇

1070\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀜𑁆𑀜𑁄 𑀯𑀸 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀯𑀻𑀡𑀸𑀬 𑀲𑀤𑁆𑀤𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑁄 𑀅𑀲𑁆𑀲𑁇 𑀲𑁄 𑀯𑀻𑀡𑀸𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀏𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀏𑀯𑀁𑀭𑀚𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀓𑀫𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀤𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀼𑀘𑁆𑀙𑀦𑀻𑀬𑁄 𑀏𑀯𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑀻𑀬𑁄’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀏𑀲𑀸, 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀻𑀡𑀸 𑀦𑀸𑀫, 𑀬𑀲𑁆𑀲𑀸 𑀏𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀏𑀯𑀁𑀭𑀚𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀓𑀫𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀤𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀼𑀘𑁆𑀙𑀦𑀻𑀬𑁄 𑀏𑀯𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑀻𑀬𑁄’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀕𑀘𑁆𑀙𑀣 𑀫𑁂, 𑀪𑁄, 𑀢𑀁 𑀯𑀻𑀡𑀁 𑀆𑀳𑀭𑀣𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀻𑀡𑀁 𑀆𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀻𑀡𑀸 𑀬𑀲𑁆𑀲𑀸 𑀏𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀏𑀯𑀁𑀭𑀚𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀓𑀫𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀤𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀼𑀘𑁆𑀙𑀦𑀻𑀬𑁄 𑀏𑀯𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑀻𑀬𑁄’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀮𑀁 𑀫𑁂, 𑀪𑁄, 𑀢𑀸𑀬 𑀯𑀻𑀡𑀸𑀬, 𑀢𑀫𑁂𑀯 𑀫𑁂 𑀲𑀤𑁆𑀤𑀁 𑀆𑀳𑀭𑀣𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀻𑀡𑀸 𑀦𑀸𑀫 𑀅𑀦𑁂𑀓𑀲𑀫𑁆𑀪𑀸𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑁆𑀪𑀸𑀭𑀸𑁇 𑀅𑀦𑁂𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸 𑀯𑀤𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑁄𑀡𑀺𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀫𑁆𑀫𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀡𑁆𑀟𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀥𑀸𑀭𑀡𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀦𑁆𑀢𑀺𑀬𑁄 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑁄𑀡𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀘 𑀢𑀚𑁆𑀚𑀁 𑀯𑀸𑀬𑀸𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀏𑀯𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀻𑀡𑀸 𑀦𑀸𑀫 𑀅𑀦𑁂𑀓𑀲𑀫𑁆𑀪𑀸𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑁆𑀪𑀸𑀭𑀸𑁇 𑀅𑀦𑁂𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸 𑀯𑀤𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀯𑀻𑀡𑀁 𑀤𑀲𑀥𑀸 𑀯𑀸 𑀲𑀢𑀥𑀸 𑀯𑀸 𑀨𑀸𑀮𑁂𑀬𑁆𑀬, 𑀤𑀲𑀥𑀸 𑀯𑀸 𑀲𑀢𑀥𑀸 𑀯𑀸 𑀢𑀁 𑀨𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑀮𑀺𑀓𑀁 𑀲𑀓𑀮𑀺𑀓𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀲𑀓𑀮𑀺𑀓𑀁 𑀲𑀓𑀮𑀺𑀓𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀟𑀳𑁂𑀬𑁆𑀬, 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀟𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀲𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀫𑀲𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀯𑀸𑀢𑁂 𑀯𑀸 𑀑𑀨𑀼𑀦𑁂𑀬𑁆𑀬, 𑀦𑀤𑀺𑀬𑀸 𑀯𑀸 𑀲𑀻𑀖𑀲𑁄𑀢𑀸𑀬 𑀧𑀯𑀸𑀳𑁂𑀬𑁆𑀬𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀲𑀢𑀻 𑀓𑀺𑀭𑀸𑀬𑀁, 𑀪𑁄, 𑀯𑀻𑀡𑀸 𑀦𑀸𑀫, 𑀬𑀣𑁂𑀯𑀁 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑀻𑀡𑀸 𑀦𑀸𑀫 𑀏𑀢𑁆𑀣 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀚𑀦𑁄 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀧𑀫𑀢𑁆𑀢𑁄 𑀧𑀮𑀍𑀇𑀢𑁄’𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀽𑀧𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑀺 𑀬𑀸𑀯𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀕𑀢𑀺, 𑀯𑁂𑀤𑀦𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑀺 𑀬𑀸𑀯𑀢𑀸 𑀯𑁂𑀤𑀦𑀸𑀬 𑀕𑀢𑀺, 𑀲𑀜𑁆𑀜𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑀺 𑀬𑀸𑀯𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀬 𑀕𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑀺 𑀬𑀸𑀯𑀢𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀕𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑀺 𑀬𑀸𑀯𑀢𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀕𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑁄 𑀬𑀸𑀯𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀕𑀢𑀺, 𑀯𑁂𑀤𑀦𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑁄…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀢𑁄 𑀬𑀸𑀯𑀢𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀕𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀳𑀦𑁆𑀢𑀺 𑀯𑀸 𑀫𑀫𑀦𑁆𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸 𑀢𑀫𑁆𑀧𑀺 𑀢𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
