# 𑁧𑁇𑁪𑁇𑁪𑁇𑁭 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀲𑀻𑀯𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(244.)

1047\. “𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑁂𑀲𑀁𑀬𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀣𑀸 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑀣𑀸𑀲𑁆𑀲 𑀓𑀸𑀫𑁂 𑀧𑀲𑁆𑀲𑀢𑁄, 𑀬𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀓𑀸𑀫𑀲𑁆𑀦𑁂𑀳𑁄 𑀓𑀸𑀫𑀫𑀼𑀘𑁆𑀙𑀸 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀲𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀢𑀣𑀸 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀘𑀸𑀭𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀘 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸 𑀘𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇

1048\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑁂𑀲𑀁𑀬𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺? ‘𑀇𑀢𑀺 𑀭𑀽𑀧𑀁, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸…  𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸…  𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄’𑀢𑀺—  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑁂𑀲𑀁𑀬𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1049\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀬𑀣𑀸𑀲𑁆𑀲 𑀓𑀸𑀫𑁂 𑀧𑀲𑁆𑀲𑀢𑁄, 𑀬𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀓𑀸𑀫𑀲𑁆𑀦𑁂𑀳𑁄 𑀓𑀸𑀫𑀫𑀼𑀘𑁆𑀙𑀸 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀲𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀼 𑀲𑀸𑀥𑀺𑀓𑀧𑁄𑀭𑀺𑀲𑀸 𑀧𑀼𑀡𑁆𑀡𑀸 𑀅𑀗𑁆𑀕𑀸𑀭𑀸𑀦𑀁 𑀯𑀻𑀢𑀘𑁆𑀘𑀺𑀓𑀸𑀦𑀁 𑀯𑀻𑀢𑀥𑀽𑀫𑀸𑀦𑀁𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀚𑀻𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀫𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀼𑀔𑀓𑀸𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀓𑀽𑀮𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑁆𑀯𑁂 𑀩𑀮𑀯𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀸 𑀦𑀸𑀦𑀸𑀩𑀸𑀳𑀸𑀲𑀼 𑀕𑀳𑁂𑀢𑁆𑀯𑀸, 𑀢𑀁 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀼𑀁 𑀉𑀧𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀲𑁄 𑀇𑀢𑀺𑀘𑀻𑀢𑀺𑀘𑁂𑀯 𑀓𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀸𑀫𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀜𑀸𑀢𑀜𑁆𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀇𑀫𑀜𑁆𑀘𑀸𑀳𑀁 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀼𑀁 𑀧𑀧𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑀣𑀸𑀲𑁆𑀲 𑀓𑀸𑀫𑁂 𑀧𑀲𑁆𑀲𑀢𑁄, 𑀬𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀓𑀸𑀫𑀲𑁆𑀦𑁂𑀳𑁄 𑀓𑀸𑀫𑀫𑀼𑀘𑁆𑀙𑀸 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀲𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

1050\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀸𑀭𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀘 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸 𑀘𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀼𑀲𑁆𑀲𑀯𑀦𑁆𑀢𑀺? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀩𑀳𑀼𑀓𑀡𑁆𑀝𑀓𑀁 𑀤𑀸𑀬𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄𑀧𑀺 𑀓𑀡𑁆𑀝𑀓𑁄, 𑀧𑀘𑁆𑀙𑀢𑁄𑀧𑀺 𑀓𑀡𑁆𑀝𑀓𑁄, 𑀉𑀢𑁆𑀢𑀭𑀢𑁄𑀧𑀺 𑀓𑀡𑁆𑀝𑀓𑁄, 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄𑀧𑀺 𑀓𑀡𑁆𑀝𑀓𑁄, 𑀳𑁂𑀝𑁆𑀞𑀢𑁄𑀧𑀺 𑀓𑀡𑁆𑀝𑀓𑁄, 𑀉𑀧𑀭𑀺𑀢𑁄𑀧𑀺 𑀓𑀡𑁆𑀝𑀓𑁄𑁇 𑀲𑁄 𑀲𑀢𑁄𑀯 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀲𑀢𑁄𑀯 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬—  ‘𑀫𑀸 𑀫𑀁 𑀓𑀡𑁆𑀝𑀓𑁄’𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀓𑀡𑁆𑀝𑀓𑁄”𑀢𑀺𑁇 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀯𑀭𑁄 𑀘 𑀅𑀲𑀁𑀯𑀭𑁄 𑀘 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄𑁇

1053\. 𑀢𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀘𑀭𑀢𑁄 𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀢𑁄 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀭𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸, 𑀤𑀦𑁆𑀥𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀺 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀭𑁄𑀢𑀺 𑀅𑀦𑀪𑀸𑀯𑀁 𑀕𑀫𑁂𑀢𑀺𑁇

1054\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀯𑀲𑀁𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑁂 𑀅𑀬𑁄𑀓𑀝𑀸𑀳𑁂 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀦𑀺𑀧𑀸𑀢𑁂𑀬𑁆𑀬𑁇 𑀤𑀦𑁆𑀥𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀁 𑀦𑀺𑀧𑀸𑀢𑁄, 𑀅𑀣 𑀔𑁄 𑀦𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀘𑀭𑀢𑁄, 𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀢𑁄 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀭𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸, 𑀤𑀦𑁆𑀥𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀺 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀭𑁄𑀢𑀺 𑀅𑀦𑀪𑀸𑀯𑀁 𑀕𑀫𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀸𑀭𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀘 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺; 𑀬𑀣𑀸 𑀘𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀼𑀲𑁆𑀲𑀯𑀦𑁆𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀯𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀯𑀸 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀯𑀸 𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸, 𑀪𑁄𑀕𑁂𑀳𑀺 𑀅𑀪𑀺𑀳𑀝𑁆𑀞𑀼𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀏𑀳𑀺, 𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀓𑀺𑀁 𑀢𑁂 𑀇𑀫𑁂 𑀓𑀸𑀲𑀸𑀯𑀸 𑀅𑀦𑀼𑀤𑀳𑀦𑁆𑀢𑀺, 𑀓𑀺𑀁 𑀫𑀼𑀡𑁆𑀟𑁄 𑀓𑀧𑀸𑀮𑀫𑀦𑀼𑀘𑀭𑀲𑀺, 𑀏𑀳𑀺 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼, 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀘𑀭𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

1055\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀗𑁆𑀕𑀸 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸𑁇 𑀅𑀣 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀓𑀼𑀤𑁆𑀤𑀸𑀮𑀧𑀺𑀝𑀓𑀁 𑀆𑀤𑀸𑀬—  ‘𑀫𑀬𑀁 𑀇𑀫𑀁 𑀕𑀗𑁆𑀕𑀁 𑀦𑀤𑀺𑀁 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀦𑁆𑀦𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫 𑀧𑀘𑁆𑀙𑀸𑀧𑁄𑀡𑀁 𑀧𑀘𑁆𑀙𑀸𑀧𑀩𑁆𑀪𑀸𑀭𑀦𑁆’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀕𑀗𑁆𑀕𑀁 𑀦𑀤𑀺𑀁 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀦𑁆𑀦𑀁 𑀓𑀭𑁂𑀬𑁆𑀬 𑀧𑀘𑁆𑀙𑀸𑀧𑁄𑀡𑀁 𑀧𑀘𑁆𑀙𑀸𑀧𑀩𑁆𑀪𑀸𑀭𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼”? “𑀕𑀗𑁆𑀕𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸; 𑀲𑀸 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀦𑁆𑀦𑀸 𑀓𑀸𑀢𑀼𑀁 𑀧𑀘𑁆𑀙𑀸𑀧𑁄𑀡𑀸 𑀧𑀘𑁆𑀙𑀸𑀧𑀩𑁆𑀪𑀸𑀭𑀸𑁇 𑀬𑀸𑀯𑀤𑁂𑀯 𑀘 𑀧𑀦 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀣𑀲𑁆𑀲 𑀯𑀺𑀖𑀸𑀢𑀲𑁆𑀲 𑀪𑀸𑀕𑀻 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀯𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀯𑀸 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀯𑀸 𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸 𑀪𑁄𑀕𑁂𑀳𑀺 𑀅𑀪𑀺𑀳𑀝𑁆𑀞𑀼𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀏𑀳𑀺, 𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀓𑀺𑀁 𑀢𑁂 𑀇𑀫𑁂 𑀓𑀸𑀲𑀸𑀯𑀸 𑀅𑀦𑀼𑀤𑀳𑀦𑁆𑀢𑀺, 𑀓𑀺𑀁 𑀫𑀼𑀡𑁆𑀟𑁄 𑀓𑀧𑀸𑀮𑀫𑀦𑀼𑀘𑀭𑀲𑀺, 𑀏𑀳𑀺 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼, 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀘𑀭𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀦𑁆𑀦𑀁 𑀯𑀺𑀯𑁂𑀓𑀧𑁄𑀡𑀁 𑀯𑀺𑀯𑁂𑀓𑀧𑀩𑁆𑀪𑀸𑀭𑀁, 𑀢𑀣𑀸 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
