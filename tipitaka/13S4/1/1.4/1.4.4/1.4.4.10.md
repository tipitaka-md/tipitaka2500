# 𑁧𑁇𑁪𑁇𑁪𑁇𑁧𑁦 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀆𑀲𑀻𑀯𑀺𑀲𑀯𑀕𑁆𑀕

* 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(247.)

1072\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀭𑀼𑀕𑀢𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀕𑀢𑁆𑀢𑁄 𑀲𑀭𑀯𑀦𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀓𑀡𑁆𑀝𑀓𑀸 𑀘𑁂𑀯 𑀧𑀸𑀤𑁂 𑀯𑀺𑀚𑁆𑀛𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀭𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀺𑀮𑁂𑀔𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀬𑁂𑀣𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀕𑀢𑁄 𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀮𑀪𑀢𑀺 𑀯𑀢𑁆𑀢𑀸𑀭𑀁—  ‘𑀅𑀬𑀜𑁆𑀘 𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁𑀓𑀸𑀭𑀻 𑀏𑀯𑀁𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀅𑀲𑀼𑀘𑀺𑀕𑀸𑀫𑀓𑀡𑁆𑀝𑀓𑁄’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀡𑁆𑀝𑀓𑁄𑀢𑀺 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀯𑀭𑁄 𑀘 𑀅𑀲𑀁𑀯𑀭𑁄 𑀘 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄𑁇

1073\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀁𑀯𑀭𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀭𑀽𑀧𑁂 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀭𑀽𑀧𑁂 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀓𑀸𑀬𑀲𑁆𑀲𑀢𑀺 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑁄𑁇 𑀢𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…  𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀥𑀫𑁆𑀫𑁂 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀓𑀸𑀬𑀲𑁆𑀲𑀢𑀺 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑁄, 𑀢𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

1074\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀯𑀺𑀲𑀬𑁂 𑀦𑀸𑀦𑀸𑀕𑁄𑀘𑀭𑁂 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀅𑀳𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀧𑀓𑁆𑀔𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀫𑀓𑁆𑀓𑀝𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀕𑀡𑁆𑀞𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀑𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀅𑀣 𑀔𑁄, 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑀸 𑀦𑀸𑀦𑀸𑀯𑀺𑀲𑀬𑀸 𑀦𑀸𑀦𑀸𑀕𑁄𑀘𑀭𑀸 𑀲𑀓𑀁 𑀲𑀓𑀁 𑀕𑁄𑀘𑀭𑀯𑀺𑀲𑀬𑀁 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  𑀅𑀳𑀺 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀯𑀫𑁆𑀫𑀺𑀓𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀉𑀤𑀓𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀧𑀓𑁆𑀔𑀻 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀆𑀓𑀸𑀲𑀁 𑀟𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀕𑀸𑀫𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀲𑀻𑀯𑀣𑀺𑀓𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀫𑀓𑁆𑀓𑀝𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀯𑀦𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑀸 𑀛𑀢𑁆𑀢𑀸 𑀅𑀲𑁆𑀲𑀼 𑀓𑀺𑀮𑀦𑁆𑀢𑀸, 𑀅𑀣 𑀔𑁄 𑀬𑁄 𑀦𑁂𑀲𑀁 𑀧𑀸𑀡𑀓𑀸𑀦𑀁 𑀩𑀮𑀯𑀢𑀭𑁄 𑀅𑀲𑁆𑀲 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀁, 𑀅𑀦𑀼𑀯𑀺𑀥𑀸𑀬𑁂𑀬𑁆𑀬𑀼𑀁 𑀯𑀲𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺 𑀅𑀪𑀸𑀯𑀺𑀢𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸, 𑀢𑀁 𑀘𑀓𑁆𑀔𑀼 𑀆𑀯𑀺𑀜𑁆𑀙𑀢𑀺 𑀫𑀦𑀸𑀧𑀺𑀬𑁂𑀲𑀼 𑀭𑀽𑀧𑁂𑀲𑀼, 𑀅𑀫𑀦𑀸𑀧𑀺𑀬𑀸 𑀭𑀽𑀧𑀸 𑀧𑀝𑀺𑀓𑀽𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀫𑀦𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑀢𑀺 𑀫𑀦𑀸𑀧𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀅𑀫𑀦𑀸𑀧𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀓𑀽𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀁𑀯𑀭𑁄 𑀳𑁄𑀢𑀺𑁇

1075\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀯𑀭𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀭𑀽𑀧𑁂 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀭𑀽𑀧𑁂 𑀦 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀓𑀸𑀬𑀲𑁆𑀲𑀢𑀺 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀘𑁂𑀢𑀲𑁄, 𑀢𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀓𑀸𑀬𑀲𑁆𑀲𑀢𑀺 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀘𑁂𑀢𑀲𑁄, 𑀢𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

1076\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀯𑀺𑀲𑀬𑁂 𑀦𑀸𑀦𑀸𑀕𑁄𑀘𑀭𑁂 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀅𑀳𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀧𑀓𑁆𑀔𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸…  𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸…  𑀫𑀓𑁆𑀓𑀝𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑁂 𑀔𑀻𑀮𑁂 𑀯𑀸 𑀣𑀫𑁆𑀪𑁂 𑀯𑀸 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑀸 𑀦𑀸𑀦𑀸𑀯𑀺𑀲𑀬𑀸 𑀦𑀸𑀦𑀸𑀕𑁄𑀘𑀭𑀸 𑀲𑀓𑀁 𑀲𑀓𑀁 𑀕𑁄𑀘𑀭𑀯𑀺𑀲𑀬𑀁 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  𑀅𑀳𑀺 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀯𑀫𑁆𑀫𑀺𑀓𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀉𑀤𑀓𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀧𑀓𑁆𑀔𑀻 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀆𑀓𑀸𑀲𑀁 𑀟𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀕𑀸𑀫𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀲𑀻𑀯𑀣𑀺𑀓𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺, 𑀫𑀓𑁆𑀓𑀝𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀯𑀦𑀁 𑀧𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀧𑁆𑀧𑀸𑀡𑀓𑀸 𑀛𑀢𑁆𑀢𑀸 𑀅𑀲𑁆𑀲𑀼 𑀓𑀺𑀮𑀦𑁆𑀢𑀸, 𑀅𑀣 𑀢𑀫𑁂𑀯 𑀔𑀻𑀮𑀁 𑀯𑀸 𑀣𑀫𑁆𑀪𑀁 𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀼𑀁, 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬𑀼𑀁, 𑀉𑀧𑀦𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸, 𑀢𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦𑀸𑀯𑀺𑀜𑁆𑀙𑀢𑀺 𑀫𑀦𑀸𑀧𑀺𑀬𑁂𑀲𑀼 𑀭𑀽𑀧𑁂𑀲𑀼, 𑀅𑀫𑀦𑀸𑀧𑀺𑀬𑀸 𑀭𑀽𑀧𑀸 𑀦𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀦𑀸𑀯𑀺𑀜𑁆𑀙𑀢𑀺 𑀫𑀦𑀸𑀧𑀺𑀬𑁂𑀲𑀼 𑀭𑀲𑁂𑀲𑀼…𑀧𑁂…  𑀫𑀦𑁄 𑀦𑀸𑀯𑀺𑀜𑁆𑀙𑀢𑀺 𑀫𑀦𑀸𑀧𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀅𑀫𑀦𑀸𑀧𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀯𑀭𑁄 𑀳𑁄𑀢𑀺𑁇

1077\. ‘𑀤𑀍𑀳𑁂 𑀔𑀻𑀮𑁂 𑀯𑀸 𑀣𑀫𑁆𑀪𑁂 𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀏𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀓𑀸𑀬𑀕𑀢𑀸 𑀦𑁄 𑀲𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
