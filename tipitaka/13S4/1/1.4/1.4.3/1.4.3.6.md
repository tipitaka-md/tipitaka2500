# 𑁧𑁇𑁪𑁇𑁩𑁇𑁬 𑀓𑀸𑀫𑀪𑀽𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀫𑀪𑀽𑀲𑀼𑀢𑁆𑀢

(233.)

964. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀆𑀦𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀸𑀫𑀪𑀽 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀸𑀫𑀪𑀽 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀸𑀫𑀪𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

965. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀓𑁆𑀔𑀼 𑀭𑀽𑀧𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀭𑀽𑀧𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀭𑀲𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀭𑀲𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀲𑀁𑀬𑁄𑀚𑀦𑀁…𑀧𑁂…  𑀫𑀦𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀦𑁆”𑀢𑀺?

966. “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀸𑀫𑀪𑀽, 𑀘𑀓𑁆𑀔𑀼 𑀭𑀽𑀧𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀦 𑀭𑀽𑀧𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀬𑀜𑁆𑀘 𑀢𑀢𑁆𑀣 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀲𑀁𑀬𑁄𑀚𑀦𑀁…𑀧𑁂…  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀭𑀲𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀦 𑀭𑀲𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀲𑀁𑀬𑁄𑀚𑀦𑀁…𑀧𑁂…  𑀦 𑀫𑀦𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀦 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀬𑀜𑁆𑀘 𑀢𑀢𑁆𑀣 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇

967. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀍𑀑 𑀘 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑁄 𑀑𑀤𑀸𑀢𑁄 𑀘 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑁄 𑀏𑀓𑁂𑀦 𑀤𑀸𑀫𑁂𑀦 𑀯𑀸 𑀬𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑁆𑀲𑀼𑁇 𑀬𑁄 𑀦𑀼 𑀔𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀓𑀸𑀍𑀑 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑁄 𑀑𑀤𑀸𑀢𑀲𑁆𑀲 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀑𑀤𑀸𑀢𑁄 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑁄 𑀓𑀸𑀍𑀅𑀲𑁆𑀲 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀦𑁆’𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸 𑀦𑀼 𑀔𑁄 𑀲𑁄 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄”𑁇 “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀍𑀑 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑁄 𑀑𑀤𑀸𑀢𑀲𑁆𑀲 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀦𑀧𑀺 𑀑𑀤𑀸𑀢𑁄 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑁄 𑀓𑀸𑀍𑀅𑀲𑁆𑀲 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀬𑁂𑀦 𑀘 𑀔𑁄 𑀢𑁂 𑀏𑀓𑁂𑀦 𑀤𑀸𑀫𑁂𑀦 𑀯𑀸 𑀬𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀦 𑀘𑀓𑁆𑀔𑀼 𑀭𑀽𑀧𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀦 𑀭𑀽𑀧𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀁…𑀧𑁂…  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸…𑀧𑁂…  𑀦 𑀫𑀦𑁄…𑀧𑁂…  𑀬𑀜𑁆𑀘 𑀢𑀢𑁆𑀣 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀲𑀁𑀬𑁄𑀚𑀦𑀦𑁆”𑀢𑀺𑁇
