### 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀢𑀻𑀢𑀸𑀤𑀺𑀬𑀤𑀦𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(210--212.)

910\. “𑀘𑀓𑁆𑀔𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀢𑁆𑀢𑀸 𑀅𑀢𑀻𑀢𑀁 𑀅𑀦𑀸𑀕𑀢𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀬𑀤𑀦𑀢𑁆𑀢𑀸, 𑀢𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀏𑀯𑀫𑁂𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀅𑀦𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀫𑀦𑁄 𑀅𑀦𑀢𑁆𑀢𑀸 𑀅𑀢𑀻𑀢𑁄 𑀅𑀦𑀸𑀕𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀬𑀤𑀦𑀢𑁆𑀢𑀸 𑀢𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀏𑀯𑀫𑁂𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁…𑀧𑁂…  𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
