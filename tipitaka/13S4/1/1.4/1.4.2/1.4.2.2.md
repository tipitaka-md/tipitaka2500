# 𑁧𑁇𑁪𑁇𑁨𑁇𑁨 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀅𑀦𑀺𑀘𑁆𑀘𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀝𑁆𑀞𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀅𑀦𑀺𑀘𑁆𑀘𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(169.)

891\. “𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁? 𑀘𑀓𑁆𑀔𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀫𑀦𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
