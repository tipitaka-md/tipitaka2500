# 𑁧𑁇𑁪𑁇𑁨𑁇𑁨𑁨--𑁨𑁪 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀢𑀻𑀢𑀸𑀤𑀺𑀤𑀼𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀝𑁆𑀞𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀢𑀻𑀢𑀸𑀤𑀺𑀤𑀼𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

(189--191.)

901\. “𑀘𑀓𑁆𑀔𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑀻𑀢𑀁 𑀅𑀦𑀸𑀕𑀢𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀢𑀻𑀢𑀸 𑀅𑀦𑀸𑀕𑀢𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸…𑀧𑁂…  𑀫𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑁄 𑀅𑀢𑀻𑀢𑁄 𑀅𑀦𑀸𑀕𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
