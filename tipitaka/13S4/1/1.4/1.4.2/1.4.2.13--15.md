# 𑁧𑁇𑁪𑁇𑁨𑁇𑁧𑁩--𑁧𑁫 𑀩𑀸𑀳𑀺𑀭𑀤𑀼𑀓𑁆𑀔𑀙𑀦𑁆𑀤𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀝𑁆𑀞𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀳𑀺𑀭𑀤𑀼𑀓𑁆𑀔𑀙𑀦𑁆𑀤𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

(180--182.)

896\. “𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀭𑀽𑀧𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀸; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀤𑁆𑀤𑀸…  𑀕𑀦𑁆𑀥𑀸…  𑀭𑀲𑀸…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸…  𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀸; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
