### 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀅𑀦𑀺𑀘𑁆𑀘𑀙𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(168.)

890\. “𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁? 𑀘𑀓𑁆𑀔𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀫𑀦𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
