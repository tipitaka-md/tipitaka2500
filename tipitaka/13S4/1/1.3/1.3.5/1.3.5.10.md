# 𑁧𑁇𑁩𑁇𑁫𑁇𑁧𑁦 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀧𑀼𑀘𑁆𑀙𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀦𑀯𑀧𑀼𑀭𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀧𑀼𑀘𑁆𑀙𑀲𑀼𑀢𑁆𑀢

(155.)

852. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “‘𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺?

853. “𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, ‘𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, ‘𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, ‘𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, ‘𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬…𑀧𑁂…  𑀫𑀦𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, ‘𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀫𑀦𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, ‘𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀫𑀦𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, ‘𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

---

854. 𑀤𑀲𑀫𑀁𑁇



---

855. 𑀦𑀯𑀧𑀼𑀭𑀸𑀡𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

856. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



857. _𑀓𑀫𑁆𑀫𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀧𑁆𑀧𑀸𑀬𑀸,_  
_𑀅𑀦𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺 𑀓𑀺𑀫𑀢𑁆𑀣𑀺𑀬𑀸;_  
_𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀧𑀭𑀺𑀬𑀸𑀬𑁄,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀓𑀣𑀺𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_  


---

858. 𑀲𑀍𑀆𑀬𑀢𑀦𑀯𑀕𑁆𑀕𑁂 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓𑁄 𑀲𑀫𑀢𑁆𑀢𑁄𑁇



---

859. 𑀢𑀲𑁆𑀲 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



860. _𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀺 𑀘 𑀮𑁄𑀓𑁄 𑀘,_  
_𑀕𑀳𑀧𑀢𑀺 𑀤𑁂𑀯𑀤𑀳𑁂𑀦 𑀘;_  
_𑀦𑀯𑀧𑀼𑀭𑀸𑀡𑁂𑀦 𑀧𑀡𑁆𑀡𑀸𑀲𑁄,_  
_𑀢𑀢𑀺𑀬𑁄 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
