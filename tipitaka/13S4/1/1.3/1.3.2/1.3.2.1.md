# 𑁧𑁇𑁩𑁇𑁨𑁇𑁧 𑀧𑀞𑀫𑀫𑀸𑀭𑀧𑀸𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀮𑁄𑀓𑀓𑀸𑀫𑀕𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀫𑀸𑀭𑀧𑀸𑀲𑀲𑀼𑀢𑁆𑀢

(114.)

613. “𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸, 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀅𑀪𑀺𑀯𑀤𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀯𑀸𑀲𑀕𑀢𑁄 𑀫𑀸𑀭𑀲𑁆𑀲, 𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀢𑁄, 𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄𑁇 𑀩𑀤𑁆𑀥𑁄 𑀲𑁄 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄…𑀧𑁂… 𑁇

614. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀲𑀸, 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀅𑀪𑀺𑀯𑀤𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀯𑀸𑀲𑀕𑀢𑁄 𑀫𑀸𑀭𑀲𑁆𑀲, 𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀢𑁄, 𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄𑁇 𑀩𑀤𑁆𑀥𑁄 𑀲𑁄 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦…𑀧𑁂… 𑁇

615. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀅𑀪𑀺𑀯𑀤𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀯𑀸𑀲𑀕𑀢𑁄 𑀫𑀸𑀭𑀲𑁆𑀲, 𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀢𑁄, 𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄𑁇 𑀩𑀤𑁆𑀥𑁄 𑀲𑁄 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄…𑀧𑁂…

616. 𑀲𑀦𑁆𑀢𑀺 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸, 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀦𑀸𑀪𑀺𑀯𑀤𑀢𑀺 𑀦𑀸𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀯𑀸𑀲𑀕𑀢𑁄 𑀫𑀸𑀭𑀲𑁆𑀲, 𑀦 𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀢𑁄, 𑀉𑀫𑁆𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄𑁇 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀦 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄…𑀧𑁂… 𑁇

617. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀲𑀸, 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀦𑀸𑀪𑀺𑀯𑀤𑀢𑀺 𑀦𑀸𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀯𑀸𑀲𑀕𑀢𑁄 𑀫𑀸𑀭𑀲𑁆𑀲, 𑀦 𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀢𑁄, 𑀉𑀫𑁆𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄𑁇 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀦 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄…𑀧𑁂… 𑁇

618. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀦𑀸𑀪𑀺𑀯𑀤𑀢𑀺 𑀦𑀸𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀯𑀸𑀲𑀕𑀢𑁄 𑀫𑀸𑀭𑀲𑁆𑀲, 𑀦 𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀢𑁄, 𑀉𑀫𑁆𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄𑁇 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀦 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄”𑀢𑀺𑁇
