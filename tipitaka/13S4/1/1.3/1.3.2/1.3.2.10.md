# 𑁧𑁇𑁩𑁇𑁨𑁇𑁧𑁦 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀮𑁄𑀓𑀓𑀸𑀫𑀕𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(123.)

720\. “𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀓𑀢𑀫𑀜𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁? 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑁄 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀲𑀸…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑁄 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

723\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



724\. _𑀫𑀸𑀭𑀧𑀸𑀲𑁂𑀦 𑀤𑁆𑀯𑁂 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀮𑁄𑀓𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀦 𑀘;_  
_𑀲𑀓𑁆𑀓𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀘𑁂𑀯,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀭𑀸𑀳𑀼𑀮𑁄;_

_𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
