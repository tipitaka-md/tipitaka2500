# 𑁧𑁇𑁩𑁇𑁪𑁇𑁯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀦𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀤𑁂𑀯𑀤𑀳𑀯𑀕𑁆𑀕

* 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀦𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀲𑀼𑀢𑁆𑀢

(142.)

799\. “𑀘𑀓𑁆𑀔𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀬𑁄𑀧𑀺 𑀳𑁂𑀢𑀼, 𑀬𑁄𑀧𑀺 𑀧𑀘𑁆𑀘𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬, 𑀲𑁄𑀧𑀺 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀅𑀦𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀽𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼 𑀓𑀼𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀬𑁄𑀧𑀺 𑀳𑁂𑀢𑀼 𑀬𑁄𑀧𑀺 𑀧𑀘𑁆𑀘𑀬𑁄 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬, 𑀲𑁄𑀧𑀺 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀅𑀦𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀽𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀺𑀯𑁆𑀳𑀸 𑀓𑀼𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀫𑀦𑁄 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀬𑁄𑀧𑀺 𑀳𑁂𑀢𑀼 𑀬𑁄𑀧𑀺 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬, 𑀲𑁄𑀧𑀺 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀅𑀦𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀽𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁄 𑀓𑀼𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁…𑀧𑁂…  𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
