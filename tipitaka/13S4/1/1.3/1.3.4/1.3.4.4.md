# 𑁧𑁇𑁩𑁇𑁪𑁇𑁪 𑀤𑀼𑀢𑀺𑀬𑀭𑀽𑀧𑀸𑀭𑀸𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑁂𑀯𑀤𑀳𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀭𑀽𑀧𑀸𑀭𑀸𑀫𑀲𑀼𑀢𑁆𑀢

(137.)

789\. “𑀭𑀽𑀧𑀸𑀭𑀸𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀭𑀽𑀧𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑁇 𑀭𑀽𑀧𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑀸…  𑀕𑀦𑁆𑀥𑀸𑀭𑀸𑀫𑀸…  𑀭𑀲𑀸𑀭𑀸𑀫𑀸…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑀸…  𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀥𑀫𑁆𑀫𑀭𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑁇 𑀥𑀫𑁆𑀫𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀭𑀽𑀧𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀦 𑀭𑀽𑀧𑀸𑀭𑀸𑀫𑁄 𑀦 𑀭𑀽𑀧𑀭𑀢𑁄 𑀦 𑀭𑀽𑀧𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑁄𑁇 𑀭𑀽𑀧𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀼𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀸𑀦𑀁…  𑀕𑀦𑁆𑀥𑀸𑀦𑀁…  𑀭𑀲𑀸𑀦𑀁…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀦𑀁…  𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑁄 𑀦 𑀥𑀫𑁆𑀫𑀭𑀢𑁄 𑀦 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑁄𑁇 𑀥𑀫𑁆𑀫𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀼𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇
