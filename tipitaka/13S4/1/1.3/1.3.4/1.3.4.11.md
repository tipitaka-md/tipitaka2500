# 𑁧𑁇𑁩𑁇𑁪𑁇𑁧𑁧 𑀩𑀸𑀳𑀺𑀭𑀤𑀼𑀓𑁆𑀔𑀳𑁂𑀢𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑁂𑀯𑀤𑀳𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀳𑀺𑀭𑀤𑀼𑀓𑁆𑀔𑀳𑁂𑀢𑀼𑀲𑀼𑀢𑁆𑀢

(144.)

803. “𑀭𑀽𑀧𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀸𑁇 𑀬𑁄𑀧𑀺 𑀳𑁂𑀢𑀼, 𑀬𑁄𑀧𑀺 𑀧𑀘𑁆𑀘𑀬𑁄 𑀭𑀽𑀧𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬, 𑀲𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑁆𑀪𑀽𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀸 𑀓𑀼𑀢𑁄 𑀲𑀼𑀔𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀸…  𑀕𑀦𑁆𑀥𑀸…  𑀭𑀲𑀸…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸…  𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑁇 𑀬𑁄𑀧𑀺 𑀳𑁂𑀢𑀼, 𑀬𑁄𑀧𑀺 𑀧𑀘𑁆𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬, 𑀲𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑁆𑀪𑀽𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀢𑁄 𑀲𑀼𑀔𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁…𑀧𑁂…  𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
