# 𑁧𑁇𑁨𑁇𑁧𑁇𑁭 𑀅𑀦𑀼𑀲𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀼𑀲𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀲𑀼𑀢𑁆𑀢

(59.)

177. “𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀲𑀬𑀸 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀘𑀓𑁆𑀔𑀼𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀅𑀦𑀼𑀲𑀬𑀸 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑁄𑀢𑀁…  𑀖𑀸𑀦𑀁…  𑀚𑀺𑀯𑁆𑀳𑀁…  𑀓𑀸𑀬𑀁…  𑀫𑀦𑀁…  𑀥𑀫𑁆𑀫𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁…  𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀢𑀫𑁆𑀧𑀺 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀅𑀦𑀼𑀲𑀬𑀸 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀅𑀦𑀼𑀲𑀬𑀸 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
