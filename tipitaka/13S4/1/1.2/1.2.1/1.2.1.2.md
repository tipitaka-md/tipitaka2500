# 𑁧𑁇𑁨𑁇𑁧𑁇𑁨 𑀲𑀁𑀬𑁄𑀚𑀦𑀧𑁆𑀧𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀕𑁆𑀕

### 𑀲𑀁𑀬𑁄𑀚𑀦𑀧𑁆𑀧𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(54.)

167\. “𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀦𑀢𑁄, 𑀓𑀣𑀁 𑀧𑀲𑁆𑀲𑀢𑁄, 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀘𑀓𑁆𑀔𑀼𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀭𑀽𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁…  𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀢𑀫𑁆𑀧𑀺 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀲𑁄𑀢𑀁…  𑀖𑀸𑀦𑀁…  𑀚𑀺𑀯𑁆𑀳𑀁…  𑀓𑀸𑀬𑀁…  𑀫𑀦𑀁…  𑀥𑀫𑁆𑀫𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁…  𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀢𑀫𑁆𑀧𑀺 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀚𑀸𑀦𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
