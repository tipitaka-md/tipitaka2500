# 𑁧𑁇𑁨𑁇𑁫𑁇𑁨 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(95.)

467\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼, 𑀬𑀫𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

468\. “𑀏𑀢𑁆𑀣 𑀤𑀸𑀦𑀺, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀁 𑀤𑀳𑀭𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀓𑁆𑀔𑀸𑀫𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀯𑀼𑀤𑁆𑀥𑁄 𑀫𑀳𑀮𑁆𑀮𑀓𑁄 𑀅𑀤𑁆𑀥𑀕𑀢𑁄 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀑𑀯𑀸𑀤𑀁 𑀬𑀸𑀘𑀲𑀻”𑀢𑀺𑁇

469\. “𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀯𑀼𑀤𑁆𑀥𑁄 𑀫𑀳𑀮𑁆𑀮𑀓𑁄 𑀅𑀤𑁆𑀥𑀕𑀢𑁄 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀤𑁂𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀁, 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀁, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀤𑀸𑀬𑀸𑀤𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇

470\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀬𑁂 𑀢𑁂 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸 𑀅𑀤𑀺𑀝𑁆𑀞𑀸 𑀅𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀸, 𑀦 𑀘 𑀧𑀲𑁆𑀲𑀲𑀺, 𑀦 𑀘 𑀢𑁂 𑀳𑁄𑀢𑀺 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

471\. “𑀬𑁂 𑀢𑁂 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀲𑀤𑁆𑀤𑀸 𑀅𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀲𑁆𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀸, 𑀦 𑀘 𑀲𑀼𑀡𑀸𑀲𑀺, 𑀦 𑀘 𑀢𑁂 𑀳𑁄𑀢𑀺 𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

472\. “𑀬𑁂 𑀢𑁂 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀕𑀦𑁆𑀥𑀸 𑀅𑀖𑀸𑀬𑀺𑀢𑀸 𑀅𑀖𑀸𑀬𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑀸, 𑀦 𑀘 𑀖𑀸𑀬𑀲𑀺, 𑀦 𑀘 𑀢𑁂 𑀳𑁄𑀢𑀺 𑀖𑀸𑀬𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

473\. “𑀬𑁂 𑀢𑁂 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀲𑀸 𑀅𑀲𑀸𑀬𑀺𑀢𑀸 𑀅𑀲𑀸𑀬𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑀸, 𑀦 𑀘 𑀲𑀸𑀬𑀲𑀺, 𑀦 𑀘 𑀢𑁂 𑀳𑁄𑀢𑀺 𑀲𑀸𑀬𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

474\. “𑀬𑁂 𑀢𑁂 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀅𑀲𑀫𑁆𑀨𑀼𑀝𑁆𑀞𑀸 𑀅𑀲𑀫𑁆𑀨𑀼𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀸, 𑀦 𑀘 𑀨𑀼𑀲𑀲𑀺, 𑀦 𑀘 𑀢𑁂 𑀳𑁄𑀢𑀺 𑀨𑀼𑀲𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

475\. “𑀬𑁂 𑀢𑁂 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀧𑀼𑀩𑁆𑀩𑀸, 𑀦 𑀘 𑀯𑀺𑀚𑀸𑀦𑀸𑀲𑀺, 𑀦 𑀘 𑀢𑁂 𑀳𑁄𑀢𑀺 𑀯𑀺𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

476\. “𑀏𑀢𑁆𑀣 𑀘 𑀢𑁂, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑀫𑀼𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀩𑁆𑀩𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑁂 𑀤𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀼𑀢𑁂 𑀲𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀼𑀢𑁂 𑀫𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑀫𑀼𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀩𑁆𑀩𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑁂 𑀤𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀼𑀢𑁂 𑀲𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀼𑀢𑁂 𑀫𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺; 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀦 𑀢𑁂𑀦𑁇 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀦 𑀢𑁂𑀦; 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀦 𑀢𑀢𑁆𑀣𑁇 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀦 𑀢𑀢𑁆𑀣; 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀦𑁂𑀯𑀺𑀥, 𑀦 𑀳𑀼𑀭𑀁, 𑀦 𑀉𑀪𑀬𑀫𑀦𑁆𑀢𑀭𑁂𑀦𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

477\. “𑀇𑀫𑀲𑁆𑀲 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺—

478\. _‘𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀢𑀺 𑀫𑀼𑀝𑁆𑀞𑀸,_  
_𑀧𑀺𑀬𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄;_  
_𑀲𑀸𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀅𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


479\. _𑀢𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸,_

480\. _𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀢𑀺 𑀫𑀼𑀝𑁆𑀞𑀸,_

481\. _𑀢𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸,_

482\. _𑀕𑀦𑁆𑀥𑀁 𑀖𑀢𑁆𑀯𑀸 𑀲𑀢𑀺 𑀫𑀼𑀝𑁆𑀞𑀸,_

483\. _𑀢𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸,_

_𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲𑀽𑀧𑀳𑀜𑁆𑀜𑀢𑀺;_

_𑀆𑀭𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


484\. _𑀭𑀲𑀁 𑀪𑁄𑀢𑁆𑀯𑀸 𑀲𑀢𑀺 𑀫𑀼𑀝𑁆𑀞𑀸,_

_𑀲𑀸𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀅𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


485\. _𑀢𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸,_

_𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲𑀽𑀧𑀳𑀜𑁆𑀜𑀢𑀺;_  
_𑀏𑀯𑀁 𑀆𑀘𑀺𑀦𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀆𑀭𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


486\. _𑀨𑀲𑁆𑀲𑀁 𑀨𑀼𑀲𑁆𑀲 𑀲𑀢𑀺 𑀫𑀼𑀝𑁆𑀞𑀸,_  
_𑀧𑀺𑀬𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄;_  
_𑀲𑀸𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀅𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


487\. _𑀢𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸,_  
_𑀅𑀦𑁂𑀓𑀸 𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀘 𑀯𑀺𑀳𑁂𑀲𑀸 𑀘,_  
_𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲𑀽𑀧𑀳𑀜𑁆𑀜𑀢𑀺;_  
_𑀏𑀯𑀁 𑀆𑀘𑀺𑀦𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀆𑀭𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


488\. _𑀥𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀯𑀸 𑀲𑀢𑀺 𑀫𑀼𑀝𑁆𑀞𑀸,_  
_𑀧𑀺𑀬𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄;_  
_𑀲𑀸𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀅𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


489\. _𑀢𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸,_  
_𑀅𑀦𑁂𑀓𑀸 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀘 𑀯𑀺𑀳𑁂𑀲𑀸 𑀘,_  
_𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲𑀽𑀧𑀳𑀜𑁆𑀜𑀢𑀺;_

_𑀆𑀭𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


490\. _𑀦 𑀲𑁄 𑀭𑀚𑁆𑀚𑀢𑀺 𑀭𑀽𑀧𑁂𑀲𑀼,_  
_𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀦𑀸𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


491\. _𑀬𑀣𑀸𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀢𑁄 𑀭𑀽𑀧𑀁,_  
_𑀲𑁂𑀯𑀢𑁄 𑀘𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀁;_  
_𑀔𑀻𑀬𑀢𑀺 𑀦𑁄𑀧𑀘𑀻𑀬𑀢𑀺,_

_𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀦𑀸𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


493\. _𑀬𑀣𑀸𑀲𑁆𑀲 𑀲𑀼𑀡𑀢𑁄 𑀲𑀤𑁆𑀤𑀁,_  
_𑀲𑁂𑀯𑀢𑁄 𑀘𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀁;_  
_𑀔𑀻𑀬𑀢𑀺 𑀦𑁄𑀧𑀘𑀻𑀬𑀢𑀺,_

494\. _𑀦 𑀲𑁄 𑀭𑀚𑁆𑀚𑀢𑀺 𑀕𑀦𑁆𑀥𑁂𑀲𑀼,_  
_𑀕𑀦𑁆𑀥𑀁 𑀖𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_

495\. _𑀬𑀣𑀸𑀲𑁆𑀲 𑀖𑀸𑀬𑀢𑁄 𑀕𑀦𑁆𑀥𑀁,_  
_𑀲𑁂𑀯𑀢𑁄 𑀘𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀁;_

_𑀏𑀯𑀁 𑀅𑀧𑀘𑀺𑀦𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁,_

496\. _𑀦 𑀲𑁄 𑀭𑀚𑁆𑀚𑀢𑀺 𑀭𑀲𑁂𑀲𑀼,_  
_𑀭𑀲𑀁 𑀪𑁄𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀦𑀸𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


497\. _𑀬𑀣𑀸𑀲𑁆𑀲 𑀲𑀸𑀬𑀢𑁄 𑀭𑀲𑀁,_  
_𑀲𑁂𑀯𑀢𑁄 𑀘𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀁;_  
_𑀔𑀻𑀬𑀢𑀺 𑀦𑁄𑀧𑀘𑀻𑀬𑀢𑀺,_

_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


498\. _𑀦 𑀲𑁄 𑀭𑀚𑁆𑀚𑀢𑀺 𑀨𑀲𑁆𑀲𑁂𑀲𑀼,_

_𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀦𑀸𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


499\. _𑀬𑀣𑀸𑀲𑁆𑀲 𑀨𑀼𑀲𑀢𑁄 𑀨𑀲𑁆𑀲𑀁,_  
_𑀲𑁂𑀯𑀢𑁄 𑀘𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀁;_

_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


500\. _𑀦 𑀲𑁄 𑀭𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀲𑀼,_  
_𑀥𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀦𑀸𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


501\. _𑀬𑀣𑀸𑀲𑁆𑀲 𑀚𑀸𑀦𑀢𑁄 𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑁂𑀯𑀢𑁄 𑀘𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀁;_  
_𑀔𑀻𑀬𑀢𑀺 𑀦𑁄𑀧𑀘𑀻𑀬𑀢𑀺,_  
_𑀏𑀯𑀁 𑀲𑁄 𑀘𑀭𑀢𑀻 𑀲𑀢𑁄;_  
_𑀏𑀯𑀁 𑀅𑀧𑀘𑀺𑀦𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀻’𑀢𑀺𑁇_  


502\. 𑀇𑀫𑀲𑁆𑀲 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀫𑀬𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺—

503\. _‘𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀢𑀺 𑀫𑀼𑀝𑁆𑀞𑀸,_  
_𑀧𑀺𑀬𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄;_  
_𑀲𑀸𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀅𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


504\. _𑀢𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸,_  
_𑀅𑀦𑁂𑀓𑀸 𑀭𑀽𑀧𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀘 𑀯𑀺𑀳𑁂𑀲𑀸 𑀘,_  
_𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲𑀽𑀧𑀳𑀜𑁆𑀜𑀢𑀺;_  
_𑀏𑀯𑀁 𑀆𑀘𑀺𑀦𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀆𑀭𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


---

505\. …𑀧𑁂…



506\. _𑀦 𑀲𑁄 𑀭𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀲𑀼,_  
_𑀥𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑁂𑀢𑀺,_  
_𑀢𑀜𑁆𑀘 𑀦𑀸𑀚𑁆𑀛𑁄𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


507\. _𑀬𑀣𑀸𑀲𑁆𑀲 𑀯𑀺𑀚𑀸𑀦𑀢𑁄 𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑁂𑀯𑀢𑁄 𑀘𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀁;_  
_𑀔𑀻𑀬𑀢𑀺 𑀦𑁄𑀧𑀘𑀻𑀬𑀢𑀺,_  
_𑀏𑀯𑀁 𑀲𑁄 𑀘𑀭𑀢𑀻 𑀲𑀢𑁄;_  
_𑀏𑀯𑀁 𑀅𑀧𑀘𑀺𑀦𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀻’𑀢𑀺𑁇_  


508\. 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀫𑀬𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

509\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇

---

510\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
