(11.)

23\. “𑀭𑀽𑀧𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀸; 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀢𑀻𑀢𑁂𑀲𑀼 𑀭𑀽𑀧𑁂𑀲𑀼 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺; 𑀅𑀦𑀸𑀕𑀢𑁂 𑀭𑀽𑀧𑁂 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺…𑀧𑁂… 𑁇
