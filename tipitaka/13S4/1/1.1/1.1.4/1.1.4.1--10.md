# 𑁧𑁇𑁧𑁇𑁪𑁇𑁧--𑁧𑁦 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀤𑀲𑀓

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀤𑀲𑀓

(33.)

115. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄…𑀧𑁂…  “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁𑁇 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀁 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁? 𑀘𑀓𑁆𑀔𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁𑁇 𑀭𑀽𑀧𑀸…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑁄𑁇 𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀢𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸…  𑀭𑀲𑀸…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄…  𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀢𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁𑁇 𑀓𑀸𑀬𑁄…𑀧𑁂…  𑀫𑀦𑁄 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑁄, 𑀥𑀫𑁆𑀫𑀸 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑁄𑁇 𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀢𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀭𑀽𑀧𑁂𑀲𑀼𑀧𑀺…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀧𑀺…  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀧𑀺…𑀧𑁂…  𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇

---

116. 𑀧𑀞𑀫𑀁𑁇



(34.)

117. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

118. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇



(35.)

119. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

120. 𑀢𑀢𑀺𑀬𑀁𑁇



(36.)

121. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

122. 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇



(37.)

123. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

124. 𑀧𑀜𑁆𑀘𑀫𑀁𑁇



(38.)

125. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

126. 𑀙𑀝𑁆𑀞𑀁𑁇



(39.)

127. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀬𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

128. 𑀲𑀢𑁆𑀢𑀫𑀁𑁇



(40.)

129. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀬𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

130. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇



(41.)

131. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

132. 𑀦𑀯𑀫𑀁𑁇



(42.)

133. “𑀲𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀁…  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇

---

134. 𑀤𑀲𑀫𑀁𑁇



---

135. 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇

136. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



137. _𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀩𑁆𑀬𑀸𑀥𑀺𑀫𑀭𑀡𑀁,_  
_𑀲𑁄𑀓𑁄 𑀘 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁;_
