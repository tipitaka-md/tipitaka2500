# 𑁧𑁇𑁧𑁇𑁨𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀤𑀼𑀓𑁆𑀔𑀼𑀧𑁆𑀧𑀸𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀤𑀼𑀓𑁆𑀔𑀼𑀧𑁆𑀧𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(22.)

57\. “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀲𑀤𑁆𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀬𑁄 𑀕𑀦𑁆𑀥𑀸𑀦𑀁…  𑀬𑁄 𑀭𑀲𑀸𑀦𑀁…  𑀬𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀦𑀁…  𑀬𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇

58\. 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑁇 𑀬𑁄 𑀲𑀤𑁆𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀬𑁄 𑀕𑀦𑁆𑀥𑀸𑀦𑀁…  𑀬𑁄 𑀭𑀲𑀸𑀦𑀁…  𑀬𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀦𑀁…  𑀬𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇

61\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁
