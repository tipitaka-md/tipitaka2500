# 𑁮𑁇𑁧𑁇𑁧 𑀘𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

* 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

* 𑀘𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

(353.)

1544\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀡𑁆𑀟𑁄 𑀕𑀸𑀫𑀡𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀘𑀡𑁆𑀟𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀘𑀡𑁆𑀟𑁄 𑀘𑀡𑁆𑀟𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑁄𑀭𑀢𑁄 𑀲𑁄𑀭𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺? “𑀇𑀥, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀭𑀸𑀕𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀭𑀸𑀕𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀢𑁆𑀢𑀸 𑀧𑀭𑁂 𑀓𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀭𑁂𑀳𑀺 𑀓𑁄𑀧𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑁄𑀧𑀁 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑀡𑁆𑀟𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀤𑁄𑀲𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀤𑁄𑀲𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀢𑁆𑀢𑀸 𑀧𑀭𑁂 𑀓𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀭𑁂𑀳𑀺 𑀓𑁄𑀧𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑁄𑀧𑀁 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑀡𑁆𑀟𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀫𑁄𑀳𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑁄𑀳𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀢𑁆𑀢𑀸 𑀧𑀭𑁂 𑀓𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀭𑁂𑀳𑀺 𑀓𑁄𑀧𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑁄𑀧𑀁 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑀡𑁆𑀟𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀘𑀡𑁆𑀟𑁄 𑀘𑀡𑁆𑀟𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

1545\. 𑀇𑀥 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀭𑀸𑀕𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀭𑀸𑀕𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀢𑁆𑀢𑀸 𑀧𑀭𑁂 𑀦 𑀓𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀭𑁂𑀳𑀺 𑀓𑁄𑀧𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀲𑁄𑀭𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀤𑁄𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀤𑁄𑀲𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀢𑁆𑀢𑀸 𑀧𑀭𑁂 𑀦 𑀓𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀭𑁂𑀳𑀺 𑀓𑁄𑀧𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀲𑁄𑀭𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀫𑁄𑀳𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑁄𑀳𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀢𑁆𑀢𑀸 𑀧𑀭𑁂 𑀦 𑀓𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀭𑁂𑀳𑀺 𑀓𑁄𑀧𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀲𑁄𑀭𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀳𑁂𑀢𑀼 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑁄𑀭𑀢𑁄 𑀲𑁄𑀭𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇

1546\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀘𑀡𑁆𑀟𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
