# 𑁮𑁇𑁧𑁇𑁧𑁧 𑀪𑀤𑁆𑀭𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀪𑀤𑁆𑀭𑀓𑀲𑀼𑀢𑁆𑀢

(363.)

1603\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀧𑁆𑀧𑀁 𑀦𑀸𑀫 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑀫𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀤𑁆𑀭𑀓𑁄 𑀕𑀸𑀫𑀡𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀪𑀤𑁆𑀭𑀓𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀢𑀽”𑀢𑀺𑁇 “𑀅𑀳𑀜𑁆𑀘𑁂 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁—  ‘𑀏𑀯𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀦𑁆’𑀢𑀺, 𑀢𑀢𑁆𑀭 𑀢𑁂 𑀲𑀺𑀬𑀸 𑀓𑀗𑁆𑀔𑀸, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀫𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑁂 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁—  ‘𑀏𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀦𑁆’𑀢𑀺, 𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀢𑁂 𑀲𑀺𑀬𑀸 𑀓𑀗𑁆𑀔𑀸, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀫𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀯 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀏𑀢𑁆𑀣𑁂𑀯 𑀢𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀪𑀤𑁆𑀭𑀓𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1605\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀘𑀺𑀭𑀯𑀸𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀥𑁂𑀦 𑀯𑀸 𑀩𑀦𑁆𑀥𑁂𑀦 𑀯𑀸 𑀚𑀸𑀦𑀺𑀬𑀸 𑀯𑀸 𑀕𑀭𑀳𑀸𑀬 𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸”𑀢𑀺? “𑀘𑀺𑀭𑀯𑀸𑀲𑀺𑀲𑁆𑀲 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀥𑁂𑀦 𑀯𑀸 𑀩𑀦𑁆𑀥𑁂𑀦 𑀯𑀸 𑀚𑀸𑀦𑀺𑀬𑀸 𑀯𑀸 𑀕𑀭𑀳𑀸𑀬 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲𑀧𑀺 𑀲𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑁂 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸”𑀢𑀺𑁇 “𑀇𑀫𑀺𑀦𑀸𑀧𑀺 𑀔𑁄 𑀏𑀢𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀙𑀦𑁆𑀤𑀫𑀽𑀮𑀓𑀁 𑀙𑀦𑁆𑀤𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀙𑀦𑁆𑀤𑁄 𑀳𑀺 𑀫𑀽𑀮𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

1606\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑀤𑀸 𑀢𑁂 𑀘𑀺𑀭𑀯𑀸𑀲𑀺𑀫𑀸𑀢𑀸 𑀅𑀤𑀺𑀝𑁆𑀞𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀅𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑁂 𑀘𑀺𑀭𑀯𑀸𑀲𑀺𑀫𑀸𑀢𑀼𑀬𑀸 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀤𑀲𑁆𑀲𑀦𑀁 𑀯𑀸 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀆𑀕𑀫𑁆𑀫 𑀲𑀯𑀦𑀁 𑀯𑀸 𑀏𑀯𑀁 𑀢𑁂 𑀅𑀳𑁄𑀲𑀺—  ‘𑀘𑀺𑀭𑀯𑀸𑀲𑀺𑀫𑀸𑀢𑀼𑀬𑀸 𑀙𑀦𑁆𑀤𑁄 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀯𑀸 𑀧𑁂𑀫𑀁 𑀯𑀸’”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1607\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀘𑀺𑀭𑀯𑀸𑀲𑀺𑀫𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀯𑀥𑁂𑀦 𑀯𑀸 𑀩𑀦𑁆𑀥𑁂𑀦 𑀯𑀸 𑀚𑀸𑀦𑀺𑀬𑀸 𑀯𑀸 𑀕𑀭𑀳𑀸𑀬 𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸”𑀢𑀺? “𑀘𑀺𑀭𑀯𑀸𑀲𑀺𑀫𑀸𑀢𑀼𑀬𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀥𑁂𑀦 𑀯𑀸 𑀩𑀦𑁆𑀥𑁂𑀦 𑀯𑀸 𑀚𑀸𑀦𑀺𑀬𑀸 𑀯𑀸 𑀕𑀭𑀳𑀸𑀬 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲𑀧𑀺 𑀲𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑁂 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸”𑀢𑀺𑁇 “𑀇𑀫𑀺𑀦𑀸𑀧𑀺 𑀔𑁄 𑀏𑀢𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀙𑀦𑁆𑀤𑀫𑀽𑀮𑀓𑀁 𑀙𑀦𑁆𑀤𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀙𑀦𑁆𑀤𑁄 𑀳𑀺 𑀫𑀽𑀮𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸’”𑀢𑀺𑁇
