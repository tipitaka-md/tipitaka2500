# 𑁮𑁇𑁧𑁇𑁫 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀲𑀼𑀢𑁆𑀢

(357.)

1558\. 𑀅𑀣 𑀔𑁄 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀕𑀸𑀫𑀡𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀉𑀲𑁆𑀲𑀳𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀉𑀲𑁆𑀲𑀳𑀦𑁆𑀢𑀁 𑀯𑀸𑀬𑀫𑀦𑁆𑀢𑀁 𑀧𑀭𑁂 𑀳𑀦𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀬𑀸𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀭𑀚𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺𑁇 𑀇𑀥 𑀪𑀕𑀯𑀸 𑀓𑀺𑀫𑀸𑀳𑀸”𑀢𑀺? “𑀅𑀮𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁; 𑀫𑀸 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀻”𑀢𑀺𑁇

1559\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀉𑀲𑁆𑀲𑀳𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀉𑀲𑁆𑀲𑀳𑀦𑁆𑀢𑀁 𑀯𑀸𑀬𑀫𑀦𑁆𑀢𑀁 𑀧𑀭𑁂 𑀳𑀦𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀬𑀸𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀭𑀚𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺𑁇 𑀇𑀥 𑀪𑀕𑀯𑀸 𑀓𑀺𑀫𑀸𑀳𑀸”𑀢𑀺?

1560\. “𑀅𑀤𑁆𑀥𑀸 𑀔𑁄 𑀢𑁆𑀬𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀦 𑀮𑀪𑀸𑀫𑀺—  ‘𑀅𑀮𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁; 𑀫𑀸 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀻’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀢𑁆𑀬𑀸𑀳𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑁄 𑀲𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀉𑀲𑁆𑀲𑀳𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀕𑀳𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀤𑀼𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀁—  ‘𑀇𑀫𑁂 𑀲𑀢𑁆𑀢𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀼 𑀯𑀸 𑀩𑀚𑁆𑀛𑀦𑁆𑀢𑀼 𑀯𑀸 𑀉𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀼 𑀯𑀸 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑀼 𑀯𑀸 𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀇𑀢𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀉𑀲𑁆𑀲𑀳𑀦𑁆𑀢𑀁 𑀯𑀸𑀬𑀫𑀦𑁆𑀢𑀁 𑀧𑀭𑁂 𑀳𑀦𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀬𑀸𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀭𑀚𑀺𑀢𑁄 𑀦𑀸𑀫 𑀦𑀺𑀭𑀬𑁄 𑀢𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺—  ‘𑀬𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀉𑀲𑁆𑀲𑀳𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀉𑀲𑁆𑀲𑀳𑀦𑁆𑀢𑀁 𑀯𑀸𑀬𑀫𑀦𑁆𑀢𑀁 𑀧𑀭𑁂 𑀳𑀦𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀬𑀸𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀭𑀚𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺, 𑀲𑀸𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀕𑀢𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀢𑀺𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀦𑀺𑀭𑀬𑀁 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀯𑀸”𑀢𑀺𑁇

1561\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀕𑀸𑀫𑀡𑀺 𑀧𑀭𑁄𑀤𑀺, 𑀅𑀲𑁆𑀲𑀽𑀦𑀺 𑀧𑀯𑀢𑁆𑀢𑁂𑀲𑀺𑁇 “𑀏𑀢𑀁 𑀔𑁄 𑀢𑁆𑀬𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀦𑀸𑀮𑀢𑁆𑀣𑀁—  ‘𑀅𑀮𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁; 𑀫𑀸 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀻’”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀁 𑀭𑁄𑀤𑀸𑀫𑀺 𑀬𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁂𑀳𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑀢𑁄 𑀯𑀜𑁆𑀘𑀺𑀢𑁄 𑀧𑀮𑀼𑀤𑁆𑀥𑁄—  ‘𑀬𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀉𑀲𑁆𑀲𑀳𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀉𑀲𑁆𑀲𑀳𑀦𑁆𑀢𑀁 𑀯𑀸𑀬𑀫𑀦𑁆𑀢𑀁 𑀧𑀭𑁂 𑀳𑀦𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀬𑀸𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀭𑀚𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’”𑀢𑀺𑁇 “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
