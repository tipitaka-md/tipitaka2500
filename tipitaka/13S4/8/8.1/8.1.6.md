# 𑁮𑁇𑁧𑁇𑁬 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(358.)

1563\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀸𑀯𑀸𑀭𑀺𑀓𑀫𑁆𑀩𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1564\. “𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀘𑁆𑀙𑀸 𑀪𑀽𑀫𑀓𑀸 𑀓𑀸𑀫𑀡𑁆𑀟𑀮𑀼𑀓𑀸 𑀲𑁂𑀯𑀸𑀮𑀫𑀸𑀮𑀺𑀓𑀸 𑀉𑀤𑀓𑁄𑀭𑁄𑀳𑀓𑀸 𑀅𑀕𑁆𑀕𑀺𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸𑁇 𑀢𑁂 𑀫𑀢𑀁 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀁 𑀉𑀬𑁆𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀸𑀫 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀸𑀫 𑀲𑀕𑁆𑀕𑀁 𑀦𑀸𑀫 𑀑𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀧𑀳𑁄𑀢𑀺 𑀢𑀣𑀸 𑀓𑀸𑀢𑀼𑀁 𑀬𑀣𑀸 𑀲𑀩𑁆𑀩𑁄 𑀮𑁄𑀓𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺?

1565\. “𑀢𑁂𑀦 𑀳𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬 𑀢𑀣𑀸 𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻𑀢𑀺𑁇

1566\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑁄 𑀨𑀭𑀼𑀲𑀯𑀸𑀘𑁄 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀻 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1567\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑀺𑀁 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀁 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀉𑀤𑀓𑀭𑀳𑀤𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬—  ‘𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚, 𑀪𑁄 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑁂, 𑀉𑀧𑁆𑀮𑀯, 𑀪𑁄 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑁂, 𑀣𑀮𑀫𑀼𑀧𑁆𑀮𑀯, 𑀪𑁄 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑁂’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑀸 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀸 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸 𑀉𑀧𑁆𑀮𑀯𑁂𑀬𑁆𑀬 𑀯𑀸 𑀣𑀮𑀁 𑀯𑀸 𑀉𑀧𑁆𑀮𑀯𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑁄 𑀨𑀭𑀼𑀲𑀯𑀸𑀘𑁄 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀻 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄𑁇 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀢𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇

1568\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1569\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀧𑁆𑀧𑀺𑀓𑀼𑀫𑁆𑀪𑀁 𑀯𑀸 𑀢𑁂𑀮𑀓𑀼𑀫𑁆𑀪𑀁 𑀯𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀉𑀤𑀓𑀭𑀳𑀤𑁂 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑁇 𑀢𑀢𑁆𑀭 𑀬𑀸𑀲𑁆𑀲 𑀲𑀓𑁆𑀔𑀭𑀸 𑀯𑀸 𑀓𑀞𑀮𑀸 𑀯𑀸 𑀲𑀸 𑀅𑀥𑁄𑀕𑀸𑀫𑀻 𑀅𑀲𑁆𑀲; 𑀬𑀜𑁆𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀢𑀢𑁆𑀭 𑀲𑀧𑁆𑀧𑀺 𑀯𑀸 𑀢𑁂𑀮𑀁 𑀯𑀸 𑀢𑀁 𑀉𑀤𑁆𑀥𑀁 𑀕𑀸𑀫𑀺 𑀅𑀲𑁆𑀲𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬—  ‘𑀑𑀲𑀻𑀤, 𑀪𑁄 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮, 𑀲𑀁𑀲𑀻𑀤, 𑀪𑁄 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮, 𑀅𑀥𑁄 𑀕𑀘𑁆𑀙, 𑀪𑁄 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮𑀸’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮𑀁 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀑𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀲𑀁𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀅𑀥𑁄 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1570\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼, 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄, 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀢𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
