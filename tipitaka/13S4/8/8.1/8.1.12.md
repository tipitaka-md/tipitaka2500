# 𑁮𑁇𑁧𑁇𑁧𑁨 𑀭𑀸𑀲𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀭𑀸𑀲𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(364.)

1609. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀲𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀲𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, ‘𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀩𑁆𑀩𑀁 𑀢𑀧𑀁 𑀕𑀭𑀳𑀢𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀁 𑀮𑀽𑀔𑀚𑀻𑀯𑀺𑀁 𑀏𑀓𑀁𑀲𑁂𑀦 𑀉𑀧𑀯𑀤𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑁄𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀩𑁆𑀩𑀁 𑀢𑀧𑀁 𑀕𑀭𑀳𑀢𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀁 𑀮𑀽𑀔𑀚𑀻𑀯𑀺𑀁 𑀏𑀓𑀁𑀲𑁂𑀦 𑀉𑀧𑀯𑀤𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑁄𑀲𑀢𑀻’𑀢𑀺, 𑀓𑀘𑁆𑀘𑀺 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀓𑁄𑀘𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀯𑀸𑀤𑀸𑀦𑀼𑀯𑀸𑀤𑁄 𑀕𑀸𑀭𑀬𑁆𑀳𑀁 𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺? “𑀬𑁂 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀩𑁆𑀩𑀁 𑀢𑀧𑀁 𑀕𑀭𑀳𑀢𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀁 𑀮𑀽𑀔𑀚𑀻𑀯𑀺𑀁 𑀏𑀓𑀁𑀲𑁂𑀦 𑀉𑀧𑀯𑀤𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑁄𑀲𑀢𑀻’𑀢𑀺, 𑀦 𑀫𑁂 𑀢𑁂 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀤𑀺𑀦𑁄, 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀘 𑀧𑀦 𑀫𑀁 𑀢𑁂 𑀅𑀲𑀢𑀸 𑀢𑀼𑀘𑁆𑀙𑀸 𑀅𑀪𑀽𑀢𑁂𑀦𑁇

1610. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀦𑁆𑀢𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸—  𑀬𑁄 𑀘𑀸𑀬𑀁 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀳𑀻𑀦𑁄 𑀕𑀫𑁆𑀫𑁄 𑀧𑁄𑀣𑀼𑀚𑁆𑀚𑀦𑀺𑀓𑁄 𑀅𑀦𑀭𑀺𑀬𑁄 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁄, 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀤𑀼𑀓𑁆𑀔𑁄 𑀅𑀦𑀭𑀺𑀬𑁄 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁄𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸—  𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀘 𑀲𑀸, 𑀕𑀸𑀫𑀡𑀺, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸—  𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀲𑀸, 𑀕𑀸𑀫𑀡𑀺, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸—  𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

1611. 𑀢𑀬𑁄 𑀔𑁄 𑀫𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀇𑀥, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀪𑁄𑀕𑁂 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀲𑀸𑀳𑀲𑁂𑀦 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀪𑁄𑀕𑁂 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀳𑀲𑁂𑀦 𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀔𑁂𑀢𑀺 𑀦 𑀧𑀻𑀡𑁂𑀢𑀺 𑀦 𑀲𑀁𑀯𑀺𑀪𑀚𑀢𑀺 𑀦 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀇𑀥 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀪𑁄𑀕𑁂 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺 𑀲𑀸𑀳𑀲𑁂𑀦𑁇 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀪𑁄𑀕𑁂 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀳𑀲𑁂𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀔𑁂𑀢𑀺 𑀧𑀻𑀡𑁂𑀢𑀺, 𑀦 𑀲𑀁𑀯𑀺𑀪𑀚𑀢𑀺 𑀦 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀇𑀥 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀪𑁄𑀕𑁂 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺 𑀲𑀸𑀳𑀲𑁂𑀦𑁇 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀪𑁄𑀕𑁂 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀳𑀲𑁂𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀔𑁂𑀢𑀺 𑀧𑀻𑀡𑁂𑀢𑀺 𑀲𑀁𑀯𑀺𑀪𑀚𑀢𑀺 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 (𑁧--𑁩)

1624. 𑀢𑀬𑁄𑀫𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀮𑀽𑀔𑀚𑀻𑀯𑀺𑀦𑁄 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀇𑀥, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 (𑁧)

1625. 𑀇𑀥 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀓𑀼𑀲𑀮𑀜𑁆𑀳𑀺 𑀔𑁄 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 (𑁨)

1626. 𑀇𑀥 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 (𑁩)

1627. 𑀢𑀢𑁆𑀭, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑁄? 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀧𑀞𑀫𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀢𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 𑀅𑀬𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻, 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 (𑁧)

1628. 𑀢𑀢𑁆𑀭, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀓𑀼𑀲𑀮𑀜𑁆𑀳𑀺 𑀔𑁄 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀤𑁆𑀯𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑁄, 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀧𑀸𑀲𑀁𑀲𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑁄? 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀧𑀞𑀫𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 𑀓𑀢𑀫𑁂𑀦 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀧𑀸𑀲𑀁𑀲𑁄? 𑀓𑀼𑀲𑀮𑀜𑁆𑀳𑀺 𑀔𑁄 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀧𑀸𑀲𑀁𑀲𑁄𑁇 𑀅𑀬𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀇𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑁄, 𑀇𑀫𑀺𑀦𑀸 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀧𑀸𑀲𑀁𑀲𑁄𑁇 (𑁨)

1629. 𑀢𑀢𑁆𑀭, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀲𑀁𑀲𑁄𑁇 𑀓𑀢𑀫𑁂𑀦 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄? 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀲𑀁𑀲𑁄? 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀧𑀞𑀫𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀧𑀸𑀲𑀁𑀲𑁄𑁇 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀧𑀸𑀲𑀁𑀲𑁄𑁇 𑀅𑀬𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀧𑀲𑁆𑀲𑀻 𑀮𑀽𑀔𑀚𑀻𑀯𑀻 𑀇𑀫𑀺𑀦𑀸 𑀏𑀓𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀕𑀸𑀭𑀬𑁆𑀳𑁄, 𑀇𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀲𑀁𑀲𑁄𑁇 (𑁩)

1630. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀕𑀸𑀫𑀡𑀺, 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀺𑀚𑁆𑀚𑀭𑀸 𑀅𑀓𑀸𑀮𑀺𑀓𑀸 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑀸 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑀸 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀬𑀁 𑀭𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀅𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺𑁇 𑀭𑀸𑀕𑁂 𑀧𑀳𑀻𑀦𑁂 𑀦𑁂𑀯𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀦 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀦 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺𑁇 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀺𑀚𑁆𑀚𑀭𑀸 𑀅𑀓𑀸𑀮𑀺𑀓𑀸 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑀸 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑀸 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺𑁇 𑀬𑀁 𑀤𑀼𑀝𑁆𑀞𑁄 𑀤𑁄𑀲𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀅𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺𑁇 𑀤𑁄𑀲𑁂 𑀧𑀳𑀻𑀦𑁂 𑀦𑁂𑀯𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀦 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀦 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺𑁇 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀺𑀚𑁆𑀚𑀭𑀸 𑀅𑀓𑀸𑀮𑀺𑀓𑀸 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑀸 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑀸 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺𑁇 𑀬𑀁 𑀫𑀽𑀍𑀳𑁄 𑀫𑁄𑀳𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀅𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀘𑁂𑀢𑁂𑀢𑀺𑁇 𑀫𑁄𑀳𑁂 𑀧𑀳𑀻𑀦𑁂 𑀦𑁂𑀯𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀦 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀦 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺𑁇 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀺𑀚𑁆𑀚𑀭𑀸 𑀅𑀓𑀸𑀮𑀺𑀓𑀸 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑀸 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑀸 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀺𑀚𑁆𑀚𑀭𑀸 𑀅𑀓𑀸𑀮𑀺𑀓𑀸 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑀸 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑀸 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻”𑀢𑀺𑁇

1631. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀭𑀸𑀲𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

---
