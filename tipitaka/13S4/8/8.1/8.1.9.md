# 𑁮𑁇𑁧𑁇𑁯 𑀓𑀼𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀮𑀲𑀼𑀢𑁆𑀢

(361.)

1593. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀸𑀯𑀸𑀭𑀺𑀓𑀫𑁆𑀩𑀯𑀦𑁂𑁇

1594. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀤𑁆𑀯𑀻𑀳𑀺𑀢𑀺𑀓𑀸 𑀲𑁂𑀢𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀮𑀸𑀓𑀸𑀯𑀼𑀢𑁆𑀢𑀸𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀫𑀳𑀢𑀺𑀬𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀭𑀺𑀲𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀦𑀺𑀕𑀡𑁆𑀞𑀲𑀸𑀯𑀓𑁄 𑀬𑁂𑀦 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀁 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑀁 𑀕𑀸𑀫𑀡𑀺𑀁 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀯𑀸𑀤𑀁 𑀆𑀭𑁄𑀧𑁂𑀳𑀺𑁇 𑀏𑀯𑀁 𑀢𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺—  ‘𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀕𑀸𑀫𑀡𑀺𑀦𑀸 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀲𑁆𑀲 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀲𑁆𑀲 𑀯𑀸𑀤𑁄 𑀆𑀭𑁄𑀧𑀺𑀢𑁄’”𑀢𑀺𑁇

1595. “𑀓𑀣𑀁 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀲𑁆𑀲 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀲𑁆𑀲 𑀯𑀸𑀤𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺? “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁂𑀦 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀺—  ‘𑀦𑀦𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀓𑀼𑀮𑀸𑀦𑀁 𑀅𑀦𑀼𑀤𑁆𑀤𑀬𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀻’𑀢𑀺? 𑀲𑀘𑁂 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀏𑀯𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀓𑀼𑀮𑀸𑀦𑀁 𑀅𑀦𑀼𑀤𑁆𑀤𑀬𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀻’𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  ‘𑀅𑀣 𑀓𑀺𑀜𑁆𑀘𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀤𑁆𑀯𑀻𑀳𑀺𑀢𑀺𑀓𑁂 𑀲𑁂𑀢𑀝𑁆𑀞𑀺𑀓𑁂 𑀲𑀮𑀸𑀓𑀸𑀯𑀼𑀢𑁆𑀢𑁂 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀢𑀺? 𑀉𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀦𑀬𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀉𑀧𑀖𑀸𑀢𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀉𑀪𑀢𑁄𑀓𑁄𑀝𑀺𑀓𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀦𑁂𑀯 𑀲𑀓𑁆𑀔𑀢𑀺 𑀉𑀕𑁆𑀕𑀺𑀮𑀺𑀢𑀼𑀁, 𑀦𑁂𑀯 𑀲𑀓𑁆𑀔𑀢𑀺 𑀑𑀕𑀺𑀮𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀦𑀺𑀕𑀡𑁆𑀞𑀲𑁆𑀲 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀁 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1596. “𑀦𑀦𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀓𑀼𑀮𑀸𑀦𑀁 𑀅𑀦𑀼𑀤𑁆𑀤𑀬𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀻”𑀢𑀺? “𑀏𑀯𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀓𑀼𑀮𑀸𑀦𑀁 𑀅𑀦𑀼𑀤𑁆𑀤𑀬𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀯𑀡𑁆𑀡𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀅𑀣 𑀓𑀺𑀜𑁆𑀘𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀤𑁆𑀯𑀻𑀳𑀺𑀢𑀺𑀓𑁂 𑀲𑁂𑀢𑀝𑁆𑀞𑀺𑀓𑁂 𑀲𑀮𑀸𑀓𑀸𑀯𑀼𑀢𑁆𑀢𑁂 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀢𑀺? 𑀉𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀦𑀬𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀉𑀧𑀖𑀸𑀢𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 “𑀇𑀢𑁄 𑀲𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀓𑀧𑁆𑀧𑁂 𑀬𑀫𑀳𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺, 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀼𑀮𑀁 𑀧𑀓𑁆𑀓𑀪𑀺𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑀫𑀢𑁆𑀢𑁂𑀦 𑀉𑀧𑀳𑀢𑀧𑀼𑀩𑁆𑀩𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀅𑀟𑁆𑀠𑀸𑀦𑀺 𑀫𑀳𑀤𑁆𑀥𑀦𑀸𑀦𑀺 𑀫𑀳𑀸𑀪𑁄𑀕𑀸𑀦𑀺 𑀧𑀳𑀽𑀢𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀸𑀦𑀺 𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑀸𑀦𑀺 𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀸𑀦𑀺, 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑀸𑀦𑀲𑀫𑁆𑀪𑀽𑀢𑀸𑀦𑀺 𑀘𑁂𑀯 𑀲𑀘𑁆𑀘𑀲𑀫𑁆𑀪𑀽𑀢𑀸𑀦𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀲𑀫𑁆𑀪𑀽𑀢𑀸𑀦𑀺 𑀘𑁇 𑀅𑀝𑁆𑀞 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀳𑁂𑀢𑀽, 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀉𑀧𑀖𑀸𑀢𑀸𑀬𑁇 𑀭𑀸𑀚𑀢𑁄 𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀘𑁄𑀭𑀢𑁄 𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀕𑁆𑀕𑀺𑀢𑁄 𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀉𑀤𑀓𑀢𑁄 𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑀺𑀳𑀺𑀢𑀁 𑀯𑀸 𑀞𑀸𑀦𑀸 𑀯𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑀼𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀸 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀯𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀓𑀼𑀮𑁂 𑀯𑀸 𑀓𑀼𑀮𑀗𑁆𑀕𑀸𑀭𑁄𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀬𑁄 𑀢𑁂 𑀪𑁄𑀕𑁂 𑀯𑀺𑀓𑀺𑀭𑀢𑀺 𑀯𑀺𑀥𑀫𑀢𑀺 𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺, 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸𑀬𑁂𑀯 𑀅𑀝𑁆𑀞𑀫𑀻𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽, 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀉𑀧𑀖𑀸𑀢𑀸𑀬𑁇 𑀇𑀫𑁂𑀲𑀼 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀝𑁆𑀞𑀲𑀼 𑀳𑁂𑀢𑀽𑀲𑀼, 𑀅𑀝𑁆𑀞𑀲𑀼 𑀧𑀘𑁆𑀘𑀬𑁂𑀲𑀼 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀲𑀼 𑀬𑁄 𑀫𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀉𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀦𑀬𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀉𑀧𑀖𑀸𑀢𑀸𑀬 𑀪𑀕𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄’𑀢𑀺, 𑀢𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀯𑀸𑀘𑀁 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂”𑀢𑀺𑁇

1597. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
