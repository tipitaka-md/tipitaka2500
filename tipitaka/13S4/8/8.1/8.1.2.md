# 𑁮𑁇𑁧𑁇𑁨 𑀢𑀸𑀮𑀧𑀼𑀝𑀲𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀢𑀸𑀮𑀧𑀼𑀝𑀲𑀼𑀢𑁆𑀢

(354.)

1548\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸𑀮𑀧𑀼𑀝𑁄 𑀦𑀝𑀕𑀸𑀫𑀡𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀢𑀸𑀮𑀧𑀼𑀝𑁄 𑀦𑀝𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀦𑀝𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀲𑁄 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀲𑀘𑁆𑀘𑀸𑀮𑀺𑀓𑁂𑀦 𑀚𑀦𑀁 𑀳𑀸𑀲𑁂𑀢𑀺 𑀭𑀫𑁂𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀳𑀸𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺𑁇 𑀇𑀥 𑀪𑀕𑀯𑀸 𑀓𑀺𑀫𑀸𑀳𑀸”𑀢𑀺? “𑀅𑀮𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁𑁇 𑀫𑀸 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑀸𑀮𑀧𑀼𑀝𑁄 𑀦𑀝𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀦𑀝𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀲𑁄 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀲𑀘𑁆𑀘𑀸𑀮𑀺𑀓𑁂𑀦 𑀚𑀦𑀁 𑀳𑀸𑀲𑁂𑀢𑀺 𑀭𑀫𑁂𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀳𑀸𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺𑁇 𑀇𑀥 𑀪𑀕𑀯𑀸 𑀓𑀺𑀫𑀸𑀳𑀸”𑀢𑀺? “𑀅𑀮𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁𑁇 𑀫𑀸 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀻”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑀸𑀮𑀧𑀼𑀝𑁄 𑀦𑀝𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀦𑀝𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀲𑁄 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀲𑀘𑁆𑀘𑀸𑀮𑀺𑀓𑁂𑀦 𑀚𑀦𑀁 𑀳𑀸𑀲𑁂𑀢𑀺 𑀭𑀫𑁂𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀳𑀸𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺𑁇 𑀇𑀥 𑀪𑀕𑀯𑀸 𑀓𑀺𑀫𑀸𑀳𑀸”𑀢𑀺?

1549\. “𑀅𑀤𑁆𑀥𑀸 𑀔𑁄 𑀢𑁆𑀬𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀦 𑀮𑀪𑀸𑀫𑀺—  ‘𑀅𑀮𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁, 𑀫𑀸 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀻’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀢𑁆𑀬𑀸𑀳𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀭𑀸𑀕𑀩𑀦𑁆𑀥𑀦𑀩𑀤𑁆𑀥𑀸𑁇 𑀢𑁂𑀲𑀁 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀭𑀚𑀦𑀻𑀬𑀸 𑀢𑁂 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀯𑀻𑀢𑀤𑁄𑀲𑀸 𑀤𑁄𑀲𑀩𑀦𑁆𑀥𑀦𑀩𑀤𑁆𑀥𑀸𑁇 𑀢𑁂𑀲𑀁 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑁄𑀲𑀦𑀻𑀬𑀸 𑀢𑁂 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀯𑀻𑀢𑀫𑁄𑀳𑀸 𑀫𑁄𑀳𑀩𑀦𑁆𑀥𑀦𑀩𑀤𑁆𑀥𑀸𑁇 𑀢𑁂𑀲𑀁 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑁄𑀳𑀦𑀻𑀬𑀸 𑀢𑁂 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬𑁇 𑀲𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀫𑀢𑁆𑀢𑁄 𑀧𑀫𑀢𑁆𑀢𑁄 𑀧𑀭𑁂 𑀫𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀫𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀳𑀸𑀲𑁄 𑀦𑀸𑀫 𑀦𑀺𑀭𑀬𑁄 𑀢𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺—  ‘𑀬𑁄 𑀲𑁄 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀲𑀘𑁆𑀘𑀸𑀮𑀺𑀓𑁂𑀦 𑀚𑀦𑀁 𑀳𑀸𑀲𑁂𑀢𑀺 𑀭𑀫𑁂𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀳𑀸𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺, 𑀲𑀸𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀕𑀢𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀢𑀺𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀦𑀺𑀭𑀬𑀁 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀯𑀸”𑀢𑀺𑁇

1550\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀢𑀸𑀮𑀧𑀼𑀝𑁄 𑀦𑀝𑀕𑀸𑀫𑀡𑀺 𑀧𑀭𑁄𑀤𑀺 𑀅𑀲𑁆𑀲𑀽𑀦𑀺 𑀧𑀯𑀢𑁆𑀢𑁂𑀲𑀺𑁇 “𑀏𑀢𑀁 𑀔𑁄 𑀢𑁆𑀬𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀦𑀸𑀮𑀢𑁆𑀣𑀁—  ‘𑀅𑀮𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁; 𑀫𑀸 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀻’”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀁 𑀭𑁄𑀤𑀸𑀫𑀺 𑀬𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀏𑀯𑀫𑀸𑀳; 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑁂𑀳𑀺 𑀦𑀝𑁂𑀳𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑀢𑁄 𑀯𑀜𑁆𑀘𑀺𑀢𑁄 𑀧𑀮𑀼𑀤𑁆𑀥𑁄—  ‘𑀬𑁄 𑀲𑁄 𑀦𑀝𑁄 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀚𑁆𑀚𑀫𑀚𑁆𑀛𑁂 𑀲𑀘𑁆𑀘𑀸𑀮𑀺𑀓𑁂𑀦 𑀚𑀦𑀁 𑀳𑀸𑀲𑁂𑀢𑀺 𑀭𑀫𑁂𑀢𑀺 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀳𑀸𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻’”𑀢𑀺𑁇 “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 𑀅𑀮𑀢𑁆𑀣 𑀔𑁄 𑀢𑀸𑀮𑀧𑀼𑀝𑁄 𑀦𑀝𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀢𑁆𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀅𑀘𑀺𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀢𑀸𑀮𑀧𑀼𑀝𑁄…𑀧𑁂…  𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇
