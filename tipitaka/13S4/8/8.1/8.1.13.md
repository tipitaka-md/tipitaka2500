# 𑁮𑁇𑁧𑁇𑁧𑁩 𑀧𑀸𑀝𑀮𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀝𑀮𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(365.)

1633. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀮𑀺𑀬𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑀁 𑀦𑀸𑀫 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀦𑀺𑀕𑀫𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂—  ‘𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀫𑀸𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺𑁇

1634. 𑀬𑁂 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀫𑀸𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺, 𑀓𑀘𑁆𑀘𑀺 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀓𑁄𑀘𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀯𑀸𑀤𑀸𑀦𑀼𑀯𑀸𑀤𑁄 𑀕𑀸𑀭𑀬𑁆𑀳𑀁 𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺? 𑀅𑀦𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀳𑀺 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀬𑁂 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀫𑀸𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺, 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀤𑀺𑀦𑁄 𑀘𑁂𑀯 𑀫𑁂, 𑀢𑁂 𑀦 𑀘 𑀫𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀓𑁄𑀘𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀯𑀸𑀤𑀸𑀦𑀼𑀯𑀸𑀤𑁄 𑀕𑀸𑀭𑀬𑁆𑀳𑀁 𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀘𑁆𑀘𑀁𑀬𑁂𑀯 𑀓𑀺𑀭, 𑀪𑁄, 𑀫𑀬𑀁 𑀢𑁂𑀲𑀁 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀦 𑀲𑀤𑁆𑀤𑀳𑀸𑀫—  ‘𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀫𑀸𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺, 𑀲𑀫𑀡𑁄 𑀔𑀮𑀼 𑀪𑁄 𑀕𑁄𑀢𑀫𑁄 𑀫𑀸𑀬𑀸𑀯𑀻”𑀢𑀺𑁇 “𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀅𑀳𑀁 𑀫𑀸𑀬𑀁 𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀅𑀳𑀁 𑀫𑀸𑀬𑀸𑀯𑀻’”𑀢𑀺𑁇 “𑀢𑀣𑁂𑀯 𑀢𑀁 𑀪𑀕𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀢𑀣𑁂𑀯 𑀢𑀁 𑀲𑀼𑀕𑀢 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺; 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬, 𑀢𑀣𑀸 𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺—

1635. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀚𑀸𑀦𑀸𑀲𑀺 𑀢𑁆𑀯𑀁 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑁂 𑀪𑀝𑁂”𑀢𑀺? “𑀚𑀸𑀦𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑁂 𑀪𑀝𑁂”𑀢𑀺𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀓𑀺𑀫𑀢𑁆𑀣𑀺𑀬𑀸 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑀸 𑀪𑀝𑀸”𑀢𑀺? “𑀬𑁂 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀘𑁄𑀭𑀸 𑀢𑁂 𑀘 𑀧𑀝𑀺𑀲𑁂𑀥𑁂𑀢𑀼𑀁, 𑀬𑀸𑀦𑀺 𑀘 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀤𑀽𑀢𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀘 𑀯𑀳𑀸𑀢𑀼𑀁, 𑀏𑀢𑀤𑀢𑁆𑀣𑀺𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑀸 𑀪𑀝𑀸”𑀢𑀺𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀚𑀸𑀦𑀸𑀲𑀺 𑀢𑁆𑀯𑀁 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑁂 𑀪𑀝𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀯𑀸 𑀢𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁂 𑀯𑀸”𑀢𑀺? “𑀚𑀸𑀦𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑁂 𑀪𑀝𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁂 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁂; 𑀬𑁂 𑀘 𑀮𑁄𑀓𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑀸 𑀪𑀝𑀸 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀸”𑀢𑀺𑁇 “𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀧𑀸𑀝𑀮𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀚𑀸𑀦𑀸𑀢𑀺 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑁂 𑀪𑀝𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁂 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁂, 𑀧𑀸𑀝𑀮𑀺𑀬𑁄𑀧𑀺 𑀕𑀸𑀫𑀡𑀺 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄’𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸 𑀦𑀼 𑀔𑁄 𑀲𑁄 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀅𑀜𑁆𑀜𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑀸 𑀪𑀝𑀸, 𑀅𑀜𑁆𑀜𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀅𑀜𑁆𑀜𑀣𑀸𑀥𑀫𑁆𑀫𑀸 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑀸 𑀪𑀝𑀸, 𑀅𑀜𑁆𑀜𑀣𑀸𑀥𑀫𑁆𑀫𑁄𑀳𑀫𑀲𑁆𑀫𑀻”𑀢𑀺𑁇 “𑀢𑁆𑀯𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀕𑀸𑀫𑀡𑀺, 𑀮𑀘𑁆𑀙𑀲𑀺—  ‘𑀧𑀸𑀝𑀮𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀚𑀸𑀦𑀸𑀢𑀺 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀮𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑁂 𑀪𑀝𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁂 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁂, 𑀦 𑀘 𑀧𑀸𑀝𑀮𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄’𑀢𑀺, 𑀓𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀦 𑀮𑀘𑁆𑀙𑀢𑀺—  ‘𑀢𑀣𑀸𑀕𑀢𑁄 𑀫𑀸𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀘 𑀢𑀣𑀸𑀕𑀢𑁄 𑀫𑀸𑀬𑀸𑀯𑀻’𑀢𑀺? 𑀫𑀸𑀬𑀜𑁆𑀘𑀸𑀳𑀁, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀚𑀸𑀦𑀸𑀫𑀺, 𑀫𑀸𑀬𑀸𑀬 𑀘 𑀯𑀺𑀧𑀸𑀓𑀁, 𑀬𑀣𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀫𑀸𑀬𑀸𑀯𑀻 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀫𑀺𑁇

1638. 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻 𑀲𑀼𑀦𑁆𑀳𑀸𑀢𑁄 𑀲𑀼𑀯𑀺𑀮𑀺𑀢𑁆𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻 𑀲𑀼𑀦𑁆𑀳𑀸𑀢𑁄 𑀲𑀼𑀯𑀺𑀮𑀺𑀢𑁆𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀁 𑀧𑀲𑀬𑁆𑀳 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀭𑀸𑀚𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀪𑀺𑀳𑀸𑀭𑀫𑀤𑀸𑀲𑀺𑁇 𑀢𑁂𑀦𑀸𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻 𑀲𑀼𑀦𑁆𑀳𑀸𑀢𑁄 𑀲𑀼𑀯𑀺𑀮𑀺𑀢𑁆𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼, 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺𑁇

1639. 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀬𑀸𑀬 𑀭𑀣𑀺𑀬𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀢𑁆𑀯𑀸, 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀢𑁆𑀯𑀸, 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀬𑀸𑀬 𑀭𑀣𑀺𑀬𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀸𑀚𑀯𑁂𑀭𑀻 𑀇𑀢𑁆𑀣𑀺𑀁 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀢𑁂𑀦 𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀡𑀁 𑀓𑀸𑀭𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇

1640. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑀜𑁆𑀘 𑀲𑀼𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇 “𑀢𑀢𑁆𑀭, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  ‘𑀬𑁄 𑀓𑁄𑀘𑀺 𑀧𑀸𑀡𑀫𑀢𑀺𑀧𑀸𑀢𑁂𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻’𑀢𑀺, 𑀲𑀘𑁆𑀘𑀁 𑀯𑀸 𑀢𑁂 𑀆𑀳𑀁𑀲𑀼 𑀫𑀼𑀲𑀸 𑀯𑀸”𑀢𑀺? “𑀫𑀼𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀢𑀼𑀘𑁆𑀙𑀁 𑀫𑀼𑀲𑀸 𑀯𑀺𑀮𑀧𑀦𑁆𑀢𑀺, 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀯𑀸 𑀢𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀯𑀸”𑀢𑀺? “𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀯𑀸”𑀢𑀺? “𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀯𑀸 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀯𑀸”𑀢𑀺? “𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑁂𑀲𑀼 𑀧𑀲𑀻𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 (𑁧)

1641. “𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻…𑀧𑁂…  𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻…𑀧𑁂…  𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀲𑁆𑀲 𑀧𑀲𑀬𑁆𑀳 𑀭𑀢𑀦𑀁 𑀅𑀳𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀭𑀸𑀚𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀪𑀺𑀳𑀸𑀭𑀫𑀤𑀸𑀲𑀺𑁇 𑀢𑁂𑀦𑀸𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻…𑀧𑁂…  𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺𑁇

1642. 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀕𑀸𑀫𑀸 𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀸 𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑀺𑁇 𑀢𑁂𑀦 𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀡𑀁 𑀓𑀸𑀭𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑀜𑁆𑀘 𑀲𑀼𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇 “𑀢𑀢𑁆𑀭, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  ‘𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻’𑀢𑀺, 𑀲𑀘𑁆𑀘𑀁 𑀯𑀸 𑀢𑁂 𑀆𑀳𑀁𑀲𑀼 𑀫𑀼𑀲𑀸 𑀯𑀸𑀢𑀺…𑀧𑁂…  𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑁂𑀲𑀼 𑀧𑀲𑀻𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 (𑁨)

1643. “𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻…𑀧𑁂…  𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻…𑀧𑁂…  𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀲𑁆𑀲 𑀤𑀸𑀭𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀭𑀸𑀚𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀪𑀺𑀳𑀸𑀭𑀫𑀤𑀸𑀲𑀺𑁇 𑀢𑁂𑀦𑀸𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻…𑀧𑁂…  𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺𑁇

1644. 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀼𑀮𑀺𑀢𑁆𑀣𑀻𑀲𑀼 𑀓𑀼𑀮𑀓𑀼𑀫𑀸𑀭𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀺, 𑀢𑁂𑀦 𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀡𑀁 𑀓𑀸𑀭𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑀜𑁆𑀘 𑀲𑀼𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇 “𑀢𑀢𑁆𑀭, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  ‘𑀬𑁄 𑀓𑁄𑀘𑀺 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀘𑀭𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻’𑀢𑀺, 𑀲𑀘𑁆𑀘𑀁 𑀯𑀸 𑀢𑁂 𑀆𑀳𑀁𑀲𑀼 𑀫𑀼𑀲𑀸 𑀯𑀸𑀢𑀺…𑀧𑁂…  𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑁂𑀲𑀼 𑀧𑀲𑀻𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 (𑁩)

1645. “𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄 𑀧𑀦, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻 𑀲𑀼𑀦𑁆𑀳𑀸𑀢𑁄 𑀲𑀼𑀯𑀺𑀮𑀺𑀢𑁆𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻 𑀲𑀼𑀦𑁆𑀳𑀸𑀢𑁄 𑀲𑀼𑀯𑀺𑀮𑀺𑀢𑁆𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀳𑀸𑀲𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀭𑀸𑀚𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀪𑀺𑀳𑀸𑀭𑀫𑀤𑀸𑀲𑀺𑁇 𑀢𑁂𑀦𑀸𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀸𑀮𑀻 𑀓𑀼𑀡𑁆𑀟𑀮𑀻 𑀲𑀼𑀦𑁆𑀳𑀸𑀢𑁄 𑀲𑀼𑀯𑀺𑀮𑀺𑀢𑁆𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀫𑁂𑀳𑀺 𑀭𑀸𑀚𑀸 𑀫𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀻’𑀢𑀺𑁇

1646. 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀬𑀸𑀬 𑀭𑀣𑀺𑀬𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁 𑀅𑀓𑀸𑀲𑀺 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀬𑀸𑀬 𑀭𑀣𑀺𑀬𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀢𑁆𑀯𑀸, 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀢𑁆𑀯𑀸, 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀻’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀪𑀜𑁆𑀚𑀺, 𑀢𑁂𑀦 𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀡𑀁 𑀓𑀸𑀭𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑀜𑁆𑀘 𑀲𑀼𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇 “𑀢𑀢𑁆𑀭, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  ‘𑀬𑁄 𑀓𑁄𑀘𑀺 𑀫𑀼𑀲𑀸 𑀪𑀡𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻’𑀢𑀺, 𑀲𑀘𑁆𑀘𑀁 𑀯𑀸 𑀢𑁂 𑀆𑀳𑀁𑀲𑀼 𑀫𑀼𑀲𑀸 𑀯𑀸”𑀢𑀺? “𑀫𑀼𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀢𑀼𑀘𑁆𑀙𑀁 𑀫𑀼𑀲𑀸 𑀯𑀺𑀮𑀧𑀦𑁆𑀢𑀺 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀯𑀸 𑀢𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀯𑀸”𑀢𑀺? “𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀯𑀸”𑀢𑀺? “𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀯𑀸 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀯𑀸”𑀢𑀺? “𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑁂𑀲𑀼 𑀧𑀲𑀻𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 (𑁪)

1652. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀳𑀼𑀤𑁂𑀯 𑀓𑀗𑁆𑀔𑀸, 𑀅𑀳𑀼 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸—  ‘𑀓𑁄𑀲𑀼 𑀦𑀸𑀫 𑀇𑀫𑁂𑀲𑀁 𑀪𑀯𑀢𑀁 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀲𑀘𑁆𑀘𑀁 𑀆𑀳, 𑀓𑁄 𑀫𑀼𑀲𑀸’”𑀢𑀺?

1653. “𑀅𑀮𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀕𑀸𑀫𑀡𑀺, 𑀓𑀗𑁆𑀔𑀺𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀼𑀁𑁇 𑀓𑀗𑁆𑀔𑀦𑀻𑀬𑁂 𑀘 𑀧𑀦 𑀢𑁂 𑀞𑀸𑀦𑁂 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀺𑁇 𑀧𑀳𑁄𑀢𑀺 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼𑀁 𑀬𑀣𑀸𑀳𑀁 𑀇𑀫𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

1654. “𑀅𑀢𑁆𑀣𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀥𑀫𑁆𑀫𑀲𑀫𑀸𑀥𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀢𑁆𑀯𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀏𑀯𑀁 𑀢𑁆𑀯𑀁 𑀇𑀫𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀕𑀸𑀫𑀡𑀺, 𑀥𑀫𑁆𑀫𑀲𑀫𑀸𑀥𑀺? 𑀇𑀥, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀁 𑀧𑀳𑀸𑀬 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀳𑀸𑀬 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀁 𑀧𑀳𑀸𑀬 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀁 𑀧𑀳𑀸𑀬 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀺𑀲𑀼𑀡𑀁 𑀯𑀸𑀘𑀁 𑀧𑀳𑀸𑀬 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀨𑀭𑀼𑀲𑀁 𑀯𑀸𑀘𑀁 𑀧𑀳𑀸𑀬 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀁 𑀧𑀳𑀸𑀬 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀪𑀺𑀚𑁆𑀛𑀁 𑀧𑀳𑀸𑀬 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀁 𑀧𑀳𑀸𑀬 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀳𑀸𑀬 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

1663. 𑀲 𑀔𑁄 𑀲𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑁄 𑀯𑀺𑀕𑀢𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁, 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑁆𑀯𑀸𑀬𑀁 𑀲𑀢𑁆𑀣𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺—  “𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀭𑀬𑀢𑁄, 𑀙𑀺𑀦𑁆𑀤𑀢𑁄 𑀙𑁂𑀤𑀸𑀧𑀬𑀢𑁄, 𑀧𑀘𑀢𑁄 𑀧𑀸𑀘𑀸𑀧𑀬𑀢𑁄, 𑀲𑁄𑀘𑀬𑀢𑁄 𑀲𑁄𑀘𑀸𑀧𑀬𑀢𑁄, 𑀓𑀺𑀮𑀫𑀢𑁄 𑀓𑀺𑀮𑀫𑀸𑀧𑀬𑀢𑁄, 𑀨𑀦𑁆𑀤𑀢𑁄 𑀨𑀦𑁆𑀤𑀸𑀧𑀬𑀢𑁄, 𑀧𑀸𑀡𑀫𑀢𑀺𑀧𑀸𑀢𑀬𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑁄, 𑀲𑀦𑁆𑀥𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑁄, 𑀦𑀺𑀮𑁆𑀮𑁄𑀧𑀁 𑀳𑀭𑀢𑁄, 𑀏𑀓𑀸𑀕𑀸𑀭𑀺𑀓𑀁 𑀓𑀭𑁄𑀢𑁄, 𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑁄, 𑀧𑀭𑀤𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑁄, 𑀫𑀼𑀲𑀸 𑀪𑀡𑀢𑁄, 𑀓𑀭𑁄𑀢𑁄 𑀓𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀁𑁇 𑀔𑀼𑀭𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁂𑀦 𑀘𑁂𑀧𑀺 𑀘𑀓𑁆𑀓𑁂𑀦 𑀬𑁄 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀸𑀡𑁂 𑀏𑀓𑀁 𑀫𑀁𑀲𑀔𑀮𑀁 𑀏𑀓𑀁 𑀫𑀁𑀲𑀧𑀼𑀜𑁆𑀚𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀸𑀧𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀆𑀕𑀫𑁄𑁇 𑀤𑀓𑁆𑀔𑀺𑀡𑀜𑁆𑀘𑁂𑀧𑀺 𑀕𑀗𑁆𑀕𑀸𑀬 𑀢𑀻𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀳𑀦𑀦𑁆𑀢𑁄 𑀖𑀸𑀢𑁂𑀦𑁆𑀢𑁄 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀙𑁂𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀧𑀘𑀦𑁆𑀢𑁄 𑀧𑀸𑀘𑀸𑀧𑁂𑀦𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀸𑀧𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀆𑀕𑀫𑁄𑁇 𑀉𑀢𑁆𑀢𑀭𑀜𑁆𑀘𑁂𑀧𑀺 𑀕𑀗𑁆𑀕𑀸𑀬 𑀢𑀻𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀤𑀤𑀦𑁆𑀢𑁄 𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑁄, 𑀬𑀚𑀦𑁆𑀢𑁄 𑀬𑀚𑀸𑀧𑁂𑀦𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀼𑀜𑁆𑀜𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄𑁇 𑀤𑀸𑀦𑁂𑀦 𑀤𑀫𑁂𑀦 𑀲𑀁𑀬𑀫𑁂𑀦 𑀲𑀘𑁆𑀘𑀯𑀚𑁆𑀚𑁂𑀦 𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀢𑀲𑁆𑀲 𑀪𑁄𑀢𑁄 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀲𑀘𑁆𑀘𑀁 𑀯𑀘𑀦𑀁, 𑀅𑀧𑀡𑁆𑀡𑀓𑀢𑀸𑀬 𑀫𑀬𑁆𑀳𑀁, 𑀬𑁆𑀯𑀸𑀳𑀁 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑁂𑀫𑀺 𑀢𑀲𑀁 𑀯𑀸 𑀣𑀸𑀯𑀭𑀁 𑀯𑀸? 𑀉𑀪𑀬𑀫𑁂𑀢𑁆𑀣 𑀓𑀝𑀕𑁆𑀕𑀸𑀳𑁄, 𑀬𑀁 𑀘𑀫𑁆𑀳𑀺 𑀓𑀸𑀬𑁂𑀦 𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀸𑀘𑀸𑀬 𑀲𑀁𑀯𑀼𑀢𑁄 𑀫𑀦𑀲𑀸 𑀲𑀁𑀯𑀼𑀢𑁄, 𑀬𑀜𑁆𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺𑁇 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺𑁇 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺𑁇 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀥𑀫𑁆𑀫𑀲𑀫𑀸𑀥𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀢𑁆𑀯𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀏𑀯𑀁 𑀢𑁆𑀯𑀁 𑀇𑀫𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇

1664. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀧𑀸𑀝𑀮𑀺𑀬𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
