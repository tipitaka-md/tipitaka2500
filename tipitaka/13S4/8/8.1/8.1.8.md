# 𑁮𑁇𑁧𑁇𑁮 𑀲𑀗𑁆𑀔𑀥𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀕𑀸𑀫𑀡𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀕𑀸𑀫𑀡𑀺𑀯𑀕𑁆𑀕

* 𑀲𑀗𑁆𑀔𑀥𑀫𑀲𑀼𑀢𑁆𑀢

(360.)

1578\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀸𑀯𑀸𑀭𑀺𑀓𑀫𑁆𑀩𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀦𑀺𑀕𑀡𑁆𑀞𑀲𑀸𑀯𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑀁 𑀕𑀸𑀫𑀡𑀺𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀻”𑀢𑀺? “𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀬𑁄 𑀓𑁄𑀘𑀺 𑀧𑀸𑀡𑀁 𑀅𑀢𑀺𑀧𑀸𑀢𑁂𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀘𑀭𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀫𑀼𑀲𑀸 𑀪𑀡𑀢𑀺 𑀲𑀩𑁆𑀩𑁄, 𑀲𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄𑁇 𑀬𑀁𑀩𑀳𑀼𑀮𑀁 𑀬𑀁𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑁂𑀦 𑀢𑁂𑀦 𑀦𑀻𑀬𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀻”𑀢𑀺𑁇 “‘𑀬𑀁𑀩𑀳𑀼𑀮𑀁 𑀬𑀁𑀩𑀳𑀼𑀮𑀜𑁆𑀘, 𑀕𑀸𑀫𑀡𑀺, 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑁂𑀦 𑀢𑁂𑀦 𑀦𑀻𑀬𑀢𑀺’, 𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂 𑀦 𑀓𑁄𑀘𑀺 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀣𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀲𑁆𑀲 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀘𑀦𑀁𑁇

1579\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀫𑀬𑀸𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀢𑀫𑁄 𑀩𑀳𑀼𑀢𑀭𑁄 𑀲𑀫𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀲𑁄 𑀧𑀸𑀡𑀫𑀢𑀺𑀧𑀸𑀢𑁂𑀢𑀺, 𑀬𑀁 𑀯𑀸 𑀲𑁄 𑀧𑀸𑀡𑀁 𑀦𑀸𑀢𑀺𑀧𑀸𑀢𑁂𑀢𑀻”𑀢𑀺? “𑀬𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀫𑀬𑀸𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀅𑀧𑁆𑀧𑀢𑀭𑁄 𑀲𑁄 𑀲𑀫𑀬𑁄 𑀬𑀁 𑀲𑁄 𑀧𑀸𑀡𑀫𑀢𑀺𑀧𑀸𑀢𑁂𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀲𑁆𑀯𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑁄 𑀲𑀫𑀬𑁄 𑀬𑀁 𑀲𑁄 𑀧𑀸𑀡𑀁 𑀦𑀸𑀢𑀺𑀧𑀸𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “‘𑀬𑀁𑀩𑀳𑀼𑀮𑀁 𑀬𑀁𑀩𑀳𑀼𑀮𑀜𑁆𑀘, 𑀕𑀸𑀫𑀡𑀺, 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑁂𑀦 𑀢𑁂𑀦 𑀦𑀻𑀬𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂 𑀦 𑀓𑁄𑀘𑀺 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀣𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀲𑁆𑀲 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀘𑀦𑀁𑁇

1580\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀫𑀬𑀸𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀢𑀫𑁄 𑀩𑀳𑀼𑀢𑀭𑁄 𑀲𑀫𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀬𑀁 𑀯𑀸 𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀦𑀸𑀤𑀺𑀬𑀢𑀻”𑀢𑀺𑁇 “𑀬𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀫𑀬𑀸𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀧𑁆𑀧𑀢𑀭𑁄 𑀲𑁄 𑀲𑀫𑀬𑁄, 𑀬𑀁 𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀲𑁆𑀯𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑁄 𑀲𑀫𑀬𑁄, 𑀬𑀁 𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀦𑀸𑀤𑀺𑀬𑀢𑀻”𑀢𑀺𑁇 “‘𑀬𑀁𑀩𑀳𑀼𑀮𑀁 𑀬𑀁𑀩𑀳𑀼𑀮𑀜𑁆𑀘, 𑀕𑀸𑀫𑀡𑀺, 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑁂𑀦 𑀢𑁂𑀦 𑀦𑀻𑀬𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂 𑀦 𑀓𑁄𑀘𑀺 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀣𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀲𑁆𑀲 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀘𑀦𑀁𑁇

1581\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀫𑀬𑀸𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀢𑀫𑁄 𑀩𑀳𑀼𑀢𑀭𑁄 𑀲𑀫𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀘𑀭𑀢𑀺, 𑀬𑀁 𑀯𑀸 𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀦 𑀘𑀭𑀢𑀻”𑀢𑀺? “𑀬𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀫𑀬𑀸𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀅𑀧𑁆𑀧𑀢𑀭𑁄 𑀲𑁄 𑀲𑀫𑀬𑁄 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀘𑀭𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀲𑁆𑀯𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑁄 𑀲𑀫𑀬𑁄, 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀦 𑀘𑀭𑀢𑀻”𑀢𑀺𑁇 “‘𑀬𑀁𑀩𑀳𑀼𑀮𑀁 𑀬𑀁𑀩𑀳𑀼𑀮𑀜𑁆𑀘, 𑀕𑀸𑀫𑀡𑀺, 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑁂𑀦 𑀢𑁂𑀦 𑀦𑀻𑀬𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂 𑀦 𑀓𑁄𑀘𑀺 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀣𑀸 𑀦𑀺𑀕𑀡𑁆𑀞𑀲𑁆𑀲 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀘𑀦𑀁𑁇

1588\. 𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀁 𑀧𑀳𑀸𑀬 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀳𑀸𑀬 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀁 𑀧𑀳𑀸𑀬 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀁 𑀧𑀳𑀸𑀬 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑀺𑀲𑀼𑀡𑀁 𑀯𑀸𑀘𑀁 𑀧𑀳𑀸𑀬 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀨𑀭𑀼𑀲𑀁 𑀯𑀸𑀘𑀁 𑀧𑀳𑀸𑀬 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀁 𑀧𑀳𑀸𑀬 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀪𑀺𑀚𑁆𑀛𑀁 𑀧𑀳𑀸𑀬 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀳𑁄𑀢𑀺𑁇 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑁆𑀧𑀤𑁄𑀲𑀁 𑀧𑀳𑀸𑀬 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀳𑀸𑀬 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

1589\. 𑀲 𑀔𑁄 𑀲𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑁄 𑀯𑀺𑀕𑀢𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀩𑀮𑀯𑀸 𑀲𑀗𑁆𑀔𑀥𑀫𑁄 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦𑁂𑀯 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀬 𑀬𑀁 𑀧𑀫𑀸𑀡𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁, 𑀦 𑀢𑀁 𑀢𑀢𑁆𑀭𑀸𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑀁 𑀢𑀢𑁆𑀭𑀸𑀯𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇

1590\. 𑀲 𑀔𑁄 𑀲𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑁄 𑀯𑀺𑀕𑀢𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄 𑀓𑀭𑀼𑀡𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂…  𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂… 𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀩𑀮𑀯𑀸 𑀲𑀗𑁆𑀔𑀥𑀫𑁄 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦𑁂𑀯 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀬 𑀬𑀁 𑀧𑀫𑀸𑀡𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀦 𑀢𑀁 𑀢𑀢𑁆𑀭𑀸𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑀁 𑀢𑀢𑁆𑀭𑀸𑀯𑀢𑀺𑀝𑁆𑀞𑀢𑀻”𑀢𑀺𑁇

1591\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
