# 𑁩𑁇𑁩𑁇𑁨 𑀧𑀲𑀬𑁆𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀩𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀲𑀬𑁆𑀳𑀲𑀼𑀢𑁆𑀢

(305.)

1301\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀩𑀮𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀭𑀽𑀧𑀩𑀮𑀁, 𑀪𑁄𑀕𑀩𑀮𑀁, 𑀜𑀸𑀢𑀺𑀩𑀮𑀁, 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑀁, 𑀲𑀻𑀮𑀩𑀮𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀩𑀮𑀸𑀦𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀩𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀸𑀫𑀺𑀓𑀁 𑀧𑀲𑀬𑁆𑀳 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀻”𑀢𑀺𑁇
