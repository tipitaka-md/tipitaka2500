# 𑁩𑁇𑁩𑁇𑁫 𑀅𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀩𑀮𑀯𑀕𑁆𑀕

### 𑀅𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(308.)

1307. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀩𑀮𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀭𑀽𑀧𑀩𑀮𑀁, 𑀪𑁄𑀕𑀩𑀮𑀁, 𑀜𑀸𑀢𑀺𑀩𑀮𑀁, 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑀁, 𑀲𑀻𑀮𑀩𑀮𑀁𑁇 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀪𑁄𑀕𑀩𑀮𑁂𑀦—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀪𑁄𑀕𑀩𑀮𑁂𑀦 𑀘—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀪𑁄𑀕𑀩𑀮𑁂𑀦 𑀘, 𑀦 𑀘 𑀜𑀸𑀢𑀺𑀩𑀮𑁂𑀦—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀪𑁄𑀕𑀩𑀮𑁂𑀦 𑀘, 𑀜𑀸𑀢𑀺𑀩𑀮𑁂𑀦 𑀘—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀪𑁄𑀕𑀩𑀮𑁂𑀦 𑀘, 𑀜𑀸𑀢𑀺𑀩𑀮𑁂𑀦 𑀘, 𑀦 𑀘 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑁂𑀦—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀪𑁄𑀕𑀩𑀮𑁂𑀦 𑀘, 𑀜𑀸𑀢𑀺𑀩𑀮𑁂𑀦 𑀘, 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑁂𑀦 𑀘—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀪𑁄𑀕𑀩𑀮𑁂𑀦 𑀘, 𑀜𑀸𑀢𑀺𑀩𑀮𑁂𑀦 𑀘, 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑁂𑀦 𑀘, 𑀦 𑀘 𑀲𑀻𑀮𑀩𑀮𑁂𑀦—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀭𑀽𑀧𑀩𑀮𑁂𑀦 𑀘 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀪𑁄𑀕𑀩𑀮𑁂𑀦 𑀘, 𑀜𑀸𑀢𑀺𑀩𑀮𑁂𑀦 𑀘, 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑁂𑀦 𑀘, 𑀲𑀻𑀮𑀩𑀮𑁂𑀦 𑀘—  𑀏𑀯𑀁 𑀲𑁄 𑀢𑁂𑀦𑀗𑁆𑀕𑁂𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀩𑀮𑀸𑀦𑀻”𑀢𑀺𑁇
