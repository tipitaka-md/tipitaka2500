# 𑁩𑁇𑁩𑁇𑁮 𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀩𑀮𑀯𑀕𑁆𑀕

* 𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(311.)

1313\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀞𑀸𑀦𑀸𑀦𑀺 𑀤𑀼𑀮𑁆𑀮𑀪𑀸𑀦𑀺 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀁 𑀞𑀸𑀦𑀁 𑀤𑀼𑀮𑁆𑀮𑀪𑀁 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀞𑀸𑀦𑀁 𑀤𑀼𑀮𑁆𑀮𑀪𑀁 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸, 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸, 𑀅𑀲𑀧𑀢𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑀁 𑀞𑀸𑀦𑀁 𑀤𑀼𑀮𑁆𑀮𑀪𑀁 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸, 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸, 𑀅𑀲𑀧𑀢𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀦𑁆𑀢𑀻 𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑀻 𑀅𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀞𑀸𑀦𑀁 𑀤𑀼𑀮𑁆𑀮𑀪𑀁 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸, 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸, 𑀅𑀲𑀧𑀢𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀦𑁆𑀢𑀻 𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑀻 𑀲𑀫𑀸𑀦𑀸 𑀲𑀸𑀫𑀺𑀓𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑀁 𑀞𑀸𑀦𑀁 𑀤𑀼𑀮𑁆𑀮𑀪𑀁 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀞𑀸𑀦𑀸𑀦𑀺 𑀤𑀼𑀮𑁆𑀮𑀪𑀸𑀦𑀺 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑀸𑀢𑀺𑁇

1314\. 𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀞𑀸𑀦𑀸𑀦𑀺 𑀲𑀼𑀮𑀪𑀸𑀦𑀺 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀁 𑀞𑀸𑀦𑀁 𑀲𑀼𑀮𑀪𑀁 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀞𑀸𑀦𑀁 𑀲𑀼𑀮𑀪𑀁 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀲𑀧𑀢𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑀁 𑀞𑀸𑀦𑀁 𑀲𑀼𑀮𑀪𑀁 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀲𑀧𑀢𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀦𑁆𑀢𑀻 𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑀻 𑀅𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀞𑀸𑀦𑀁 𑀲𑀼𑀮𑀪𑀁 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀲𑀧𑀢𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀦𑁆𑀢𑀻 𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑀻 𑀲𑀫𑀸𑀦𑀸 𑀲𑀸𑀫𑀺𑀓𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑀁 𑀞𑀸𑀦𑀁 𑀲𑀼𑀮𑀪𑀁 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀞𑀸𑀦𑀸𑀦𑀺 𑀲𑀼𑀮𑀪𑀸𑀦𑀺 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦𑀸”𑀢𑀺𑁇
