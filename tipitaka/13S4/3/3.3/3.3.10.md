# 𑁩𑁇𑁩𑁇𑁧𑁦 𑀯𑀟𑁆𑀠𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀩𑀮𑀯𑀕𑁆𑀕

* 𑀯𑀟𑁆𑀠𑀻𑀲𑀼𑀢𑁆𑀢

(313.)

1318\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀟𑁆𑀠𑀻𑀳𑀺 𑀯𑀟𑁆𑀠𑀫𑀸𑀦𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀺𑀓𑀸 𑀅𑀭𑀺𑀬𑀸𑀬 𑀯𑀟𑁆𑀠𑀺𑀬𑀸 𑀯𑀟𑁆𑀠𑀢𑀺 𑀲𑀸𑀭𑀸𑀤𑀸𑀬𑀺𑀦𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀯𑀭𑀸𑀤𑀸𑀬𑀺𑀦𑀻 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀲𑀤𑁆𑀥𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀲𑀻𑀮𑁂𑀦 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀲𑀼𑀢𑁂𑀦 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀘𑀸𑀕𑁂𑀦 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀟𑁆𑀠𑀻𑀳𑀺 𑀯𑀟𑁆𑀠𑀫𑀸𑀦𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀺𑀓𑀸 𑀅𑀭𑀺𑀬𑀸𑀬 𑀯𑀟𑁆𑀠𑀺𑀬𑀸 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀲𑀸𑀭𑀸𑀤𑀸𑀬𑀺𑀦𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀯𑀭𑀸𑀤𑀸𑀬𑀺𑀦𑀻 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲𑀸𑀢𑀺𑁇

1319\. _𑀲𑀤𑁆𑀥𑀸𑀬 𑀲𑀻𑀮𑁂𑀦 𑀘 𑀬𑀸𑀥 𑀯𑀟𑁆𑀠𑀢𑀺,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀸𑀕𑁂𑀦 𑀲𑀼𑀢𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀲𑀸 𑀢𑀸𑀤𑀺𑀲𑀻 𑀲𑀻𑀮𑀯𑀢𑀻 𑀉𑀧𑀸𑀲𑀺𑀓𑀸,_  
_𑀆𑀤𑀻𑀬𑀢𑀺 𑀲𑀸𑀭𑀫𑀺𑀥𑁂𑀯 𑀅𑀢𑁆𑀢𑀦𑁄”𑀢𑀺𑁇_

1322\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1323\. _𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀲𑀬𑁆𑀳 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬,_  
_𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀁;_  
_𑀦𑀸𑀲𑁂𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼 𑀞𑀸𑀦𑀜𑁆𑀘,_  
_𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀯𑀟𑁆𑀠𑀺𑀦𑀸 𑀤𑀲𑀸𑀢𑀺𑁇_
