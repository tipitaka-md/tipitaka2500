# 𑁩𑁇𑁩𑁇𑁭 𑀳𑁂𑀢𑀼𑀲𑀼𑀢𑁆𑀢

(310.)

1311\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀩𑀮𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀭𑀽𑀧𑀩𑀮𑀁, 𑀪𑁄𑀕𑀩𑀮𑀁, 𑀜𑀸𑀢𑀺𑀩𑀮𑀁, 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑀁, 𑀲𑀻𑀮𑀩𑀮𑀁𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀭𑀽𑀧𑀩𑀮𑀳𑁂𑀢𑀼 𑀯𑀸 𑀪𑁄𑀕𑀩𑀮𑀳𑁂𑀢𑀼 𑀯𑀸 𑀜𑀸𑀢𑀺𑀩𑀮𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀻𑀮𑀩𑀮𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀩𑀮𑀸𑀦𑀻”𑀢𑀺𑁇
