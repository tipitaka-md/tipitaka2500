### 𑀏𑀓𑀲𑀼𑀢𑁆𑀢

(307.)

1305\. “𑀏𑀓𑁂𑀦 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀮𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀦 𑀏𑀓𑁂𑀦 𑀩𑀮𑁂𑀦? 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀩𑀮𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀁 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀦𑁂𑀯 𑀭𑀽𑀧𑀩𑀮𑀁 𑀢𑀸𑀬𑀢𑀺, 𑀦 𑀪𑁄𑀕𑀩𑀮𑀁 𑀢𑀸𑀬𑀢𑀺, 𑀦 𑀜𑀸𑀢𑀺𑀩𑀮𑀁 𑀢𑀸𑀬𑀢𑀺, 𑀦 𑀧𑀼𑀢𑁆𑀢𑀩𑀮𑀁 𑀢𑀸𑀬𑀢𑀺, 𑀦 𑀲𑀻𑀮𑀩𑀮𑀁 𑀢𑀸𑀬𑀢𑀻”𑀢𑀺𑁇
