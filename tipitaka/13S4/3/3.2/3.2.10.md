# 𑁩𑁇𑁨𑁇𑁧𑁦 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀲𑀼𑀢𑁆𑀢

(303.)

1293. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
