# 𑁩𑁇𑁨𑁇𑁯 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(302.)

1290\. “𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀯𑀸 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
