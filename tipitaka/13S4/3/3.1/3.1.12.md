# 𑁩𑁇𑁧𑁇𑁧𑁨 𑀓𑀼𑀲𑀻𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀲𑀻𑀢𑀲𑀼𑀢𑁆𑀢

(291.)

1263\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄…𑀧𑁂…  𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀳𑀺𑀭𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀓𑀼𑀲𑀻𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀘 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄…𑀧𑁂…  𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
