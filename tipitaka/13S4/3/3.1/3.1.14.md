# 𑁩𑁇𑁧𑁇𑁧𑁪 𑀧𑀜𑁆𑀘𑀯𑁂𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀯𑁂𑀭𑀲𑀼𑀢𑁆𑀢

(293.)

1267\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄…𑀧𑁂…  𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀬𑀻 𑀘 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

1270\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1271\. _𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀘,_  
_𑀆𑀯𑁂𑀡𑀺𑀓𑀸 𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀘;_  
_𑀓𑁄𑀥𑀦𑁄 𑀉𑀧𑀦𑀸𑀳𑀻 𑀘,_  
_𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀫𑀘𑁆𑀙𑀭𑁂𑀦 𑀘;_  
_𑀅𑀢𑀺𑀘𑀸𑀭𑀻 𑀘 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄,_  
_𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁄 𑀘 𑀓𑀼𑀲𑀻𑀢𑁄;_  
_𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺 𑀧𑀜𑁆𑀘𑀯𑁂𑀭𑀁,_  
_𑀓𑀡𑁆𑀳𑀧𑀓𑁆𑀔𑁂 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇_
