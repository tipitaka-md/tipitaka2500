# 𑁧𑁇𑁧𑁇𑁯 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀼𑀢𑁆𑀢

(9.)

80\. “𑀦𑀯𑀬𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀦𑀯? 𑀅𑀭𑀳𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀦𑀸𑀕𑀸𑀫𑀻, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇
