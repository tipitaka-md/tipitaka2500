# 𑁧𑁇𑁩𑁇𑁯 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀢𑁆𑀢𑀸𑀯𑀸𑀲𑀯𑀕𑁆𑀕

* 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

(29.)

237\. “𑀦𑀯𑀬𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀦𑀯? ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀅𑀘𑀭𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; ‘𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; ‘𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀅𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀯 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀻”𑀢𑀺𑁇
