# 𑁧𑁇𑁫𑁇𑁨 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀸𑀫𑀜𑁆𑀜𑀯𑀕𑁆𑀕

* 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻𑀲𑀼𑀢𑁆𑀢

(43.)

370\. “‘𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻, 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺?

371\. 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀘 𑀢𑀤𑀸𑀬𑀢𑀦𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀦𑀁 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑁇 (𑁧)

372\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀘 𑀢𑀤𑀸𑀬𑀢𑀦𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀦𑀁 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑁇 (𑁨--𑁪)

373\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀘 𑀢𑀤𑀸𑀬𑀢𑀦𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀦𑀁 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦…𑀧𑁂… 𑁇 (𑁫--𑁮)

374\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀲𑁆𑀲 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀘 𑀢𑀤𑀸𑀬𑀢𑀦𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀦𑀁 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀦𑀺𑀧𑁆𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸”𑀢𑀺𑁇 (𑁯)
