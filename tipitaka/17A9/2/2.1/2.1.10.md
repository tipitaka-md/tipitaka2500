# 𑁨𑁇𑁧𑁇𑁧𑁦 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀭𑁄𑀥𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀔𑁂𑀫𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀭𑁄𑀥𑀲𑀼𑀢𑁆𑀢

(61.)

426\. “‘𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀭𑁄𑀥𑁄, 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺?

427\. “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦…𑀧𑁂… 𑁇

428\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀲𑁆𑀲 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀦𑀺𑀧𑁆𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸”𑀢𑀺𑁇
