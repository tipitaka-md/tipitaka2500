# 𑁨𑁇𑁧𑁇𑁧𑁧 𑀅𑀪𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

(62.)

430\. “𑀦𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀦𑀯? 𑀭𑀸𑀕𑀁, 𑀤𑁄𑀲𑀁, 𑀫𑁄𑀳𑀁, 𑀓𑁄𑀥𑀁, 𑀉𑀧𑀦𑀸𑀳𑀁, 𑀫𑀓𑁆𑀔𑀁, 𑀧𑀍𑀆𑀲𑀁, 𑀇𑀲𑁆𑀲𑀁, 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇

431\. 𑀦𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀦𑀯? 𑀭𑀸𑀕𑀁, 𑀤𑁄𑀲𑀁, 𑀫𑁄𑀳𑀁, 𑀓𑁄𑀥𑀁, 𑀉𑀧𑀦𑀸𑀳𑀁, 𑀫𑀓𑁆𑀔𑀁, 𑀧𑀍𑀆𑀲𑀁, 𑀇𑀲𑁆𑀲𑀁, 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀯 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
