# 𑁨𑁇𑁨𑁇𑁬 𑀕𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀕𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(68.)

451\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀢𑀺𑀬𑁄𑁇 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀘? 𑀦𑀺𑀭𑀬𑁄, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺, 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸, 𑀤𑁂𑀯𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀕𑀢𑀺𑀬𑁄𑁇

452\. 𑀇𑀫𑀸𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀕𑀢𑀻𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬…𑀧𑁂…  𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇

---
