# 𑁨𑁇𑁨𑁇𑁩 𑀓𑀸𑀫𑀕𑀼𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀫𑀕𑀼𑀡𑀲𑀼𑀢𑁆𑀢

(65.)

442. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀲𑀤𑁆𑀤𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀕𑀦𑁆𑀥𑀸…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀲𑀸…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇

443. 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬…𑀧𑁂…  𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
