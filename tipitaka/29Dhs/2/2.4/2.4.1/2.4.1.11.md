# 𑁨𑁇𑁪𑁇𑁧𑁇𑁧𑁧 𑀲𑁂𑀓𑁆𑀔𑀢𑁆𑀢𑀺𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀢𑀺𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀲𑁂𑀓𑁆𑀔𑀢𑁆𑀢𑀺𑀓

(1414.)

2120\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀓𑁆𑀔𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺 𑀘 𑀢𑀻𑀡𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀓𑁆𑀔𑀸𑁇

(1415.)

2121\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑀸? 𑀉𑀧𑀭𑀺𑀝𑁆𑀞𑀺𑀫𑀁 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑀸𑁇

(1416.)

2122\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸? 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇
