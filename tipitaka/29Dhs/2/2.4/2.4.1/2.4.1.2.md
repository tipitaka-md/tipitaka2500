# 𑁨𑁇𑁪𑁇𑁧𑁇𑁨 𑀯𑁂𑀤𑀦𑀸𑀢𑁆𑀢𑀺𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀢𑀺𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀯𑁂𑀤𑀦𑀸𑀢𑁆𑀢𑀺𑀓

(1387.)

2093\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘 𑀧𑀜𑁆𑀘, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀢𑀺𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀢𑀺𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(1388.)

2094\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀤𑁆𑀯𑁂 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(1389.)

2095\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀢𑁄 𑀙, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀤𑀲, 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀙, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀙, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺𑀧𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺𑀧𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺𑀧𑀺𑁇
