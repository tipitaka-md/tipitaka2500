# 𑁨𑁇𑁪𑁇𑁧𑁇𑁭 𑀧𑀻𑀢𑀺𑀢𑁆𑀢𑀺𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀢𑀺𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀧𑀻𑀢𑀺𑀢𑁆𑀢𑀺𑀓

(1402.)

2108\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀧𑀜𑁆𑀘, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀧𑀜𑁆𑀘, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓𑀢𑀺𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀤𑀼𑀓𑀢𑀺𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀻𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇

(1403.)

2109\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀙, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀧𑀜𑁆𑀘, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀢𑀺𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀢𑀺𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑁇

(1404.)

2110\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀢𑁄 𑀙, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀤𑀲, 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀙, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀙, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀭𑀼𑀧𑁆𑀧𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀧𑀻𑀢𑀺 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀲𑀼𑀔𑀁 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁, 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀦𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀬𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀭𑀽𑀧𑀜𑁆𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑀢𑀺𑀧𑀺, 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀢𑀺𑀧𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑀢𑀺𑀧𑀺𑁇
