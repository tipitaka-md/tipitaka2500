# 𑁨𑁇𑁪𑁇𑁧𑁇𑁧𑁪 𑀳𑀻𑀦𑀢𑁆𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀢𑀺𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀳𑀻𑀦𑀢𑁆𑀢𑀺𑀓

(1423.)

2129\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑀻𑀦𑀸? 𑀤𑁆𑀯𑀸𑀤𑀲 𑀅𑀓𑀼𑀲𑀮𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑀻𑀦𑀸𑁇

(1424.)

2130\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀸? 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑁇

(1425.)

2131\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀡𑀻𑀢𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀡𑀻𑀢𑀸𑁇
