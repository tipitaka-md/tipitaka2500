# 𑁨𑁇𑁪𑁇𑁧𑁇𑁬 𑀯𑀺𑀢𑀓𑁆𑀓𑀢𑁆𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀢𑀺𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀯𑀺𑀢𑀓𑁆𑀓𑀢𑁆𑀢𑀺𑀓

(1399.)

2105. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀏𑀓𑀸𑀤𑀲 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀤𑁆𑀯𑁂, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀏𑀓𑀸𑀤𑀲, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸𑁇

(1400.)

2106. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀧𑀜𑁆𑀘𑀓𑀦𑀬𑁂 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀧𑀜𑁆𑀘𑀓𑀦𑀬𑁂 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀘𑀸𑀭𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸𑁇

(1401.)

2107. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸? 𑀤𑁆𑀯𑁂𑀧𑀜𑁆𑀘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀺, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀢𑀺𑀓𑀢𑀺𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀭𑀼𑀧𑁆𑀧𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀢𑀺𑀓𑀢𑀺𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀧𑀜𑁆𑀘𑀓𑀦𑀬𑁂 𑀤𑀼𑀢𑀺𑀬𑁂 𑀛𑀸𑀦𑁂, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀯𑀺𑀘𑀸𑀭𑁄 𑀭𑀽𑀧𑀜𑁆𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀳𑀚𑀸𑀢𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄𑀢𑀺𑀧𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄𑀢𑀺𑀧𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄𑀢𑀺𑀧𑀺𑁇
