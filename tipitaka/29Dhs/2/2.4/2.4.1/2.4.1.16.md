# 𑁨𑁇𑁪𑁇𑁧𑁇𑁧𑁬 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀢𑀺𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

(1429.)

2135. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀦 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸; 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺𑀧𑀺, 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄𑀢𑀺𑀧𑀺𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀸 𑀦 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸; 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄𑀢𑀺𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀓𑀺𑀭𑀺𑀬𑀸𑀳𑁂𑀢𑀼𑀓𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸; 𑀦 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸, 𑀦 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄; 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺𑁇 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀩𑁆𑀩𑁄 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀙 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀢𑀺𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀭𑀼𑀧𑁆𑀧𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺𑀧𑀺, 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸𑀢𑀺𑀧𑀺, 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇
