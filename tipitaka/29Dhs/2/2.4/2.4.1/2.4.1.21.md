# 𑁨𑁇𑁪𑁇𑁧𑁇𑁨𑁧 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀢𑀺𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

(1436.)

2142\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸? 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇

(1437.)

2143\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀢𑀺𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆𑀢𑀺𑀧𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆𑀢𑀺𑀧𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇
