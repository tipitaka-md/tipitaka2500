# 𑁨𑁇𑁪𑁇𑁨𑁇𑁫𑁇𑁪 𑀕𑀦𑁆𑀣𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀕𑀦𑁆𑀣𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀕𑀦𑁆𑀣𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀤𑀼𑀓

(1495.)

2206\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀕𑀦𑁆𑀣𑀸 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘? 𑀢𑁂𑀯 𑀕𑀦𑁆𑀣𑀸 𑀕𑀦𑁆𑀣𑀸 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘𑁇

(1496.)

2207\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀸? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀦𑁆𑀣𑁂 𑀅𑀯𑀲𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀸𑁇 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  𑀕𑀦𑁆𑀣𑀸 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘𑀸𑀢𑀺𑀧𑀺, 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀸𑀢𑀺𑀧𑀺𑁇
