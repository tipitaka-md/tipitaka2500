# 𑁨𑁇𑁪𑁇𑁨𑁇𑁫𑁇𑁧 𑀕𑀦𑁆𑀣𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀕𑀦𑁆𑀣𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀕𑀦𑁆𑀣𑀤𑀼𑀓

(1489.)

2200\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀕𑀦𑁆𑀣𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀕𑀦𑁆𑀣𑀸—  𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄, 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄𑁇 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀅𑀝𑁆𑀞𑀲𑀼 𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀤𑁆𑀯𑀻𑀲𑀼 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀘 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀘 𑀘𑀢𑀽𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀕𑀦𑁆𑀣𑀸𑁇

(1490.)

2201\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀕𑀦𑁆𑀣𑀸? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀦𑁆𑀣𑁂, 𑀅𑀯𑀲𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀕𑀦𑁆𑀣𑀸𑁇
