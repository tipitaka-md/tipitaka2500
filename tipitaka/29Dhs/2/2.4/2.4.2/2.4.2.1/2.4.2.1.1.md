# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁇𑁧 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

(1441.)

2148\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽𑁇 𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀫𑁄𑀳𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

2149\. 𑀮𑁄𑀪𑁄 𑀅𑀝𑁆𑀞𑀲𑀼 𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑁄𑀲𑁄 𑀤𑁆𑀯𑀻𑀲𑀼 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀫𑁄𑀳𑁄 𑀲𑀩𑁆𑀩𑀸𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

2150\. 𑀅𑀮𑁄𑀪𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼 𑀅𑀤𑁄𑀲𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀫𑁄𑀳𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

2151\. 𑀅𑀮𑁄𑀪𑁄 𑀓𑀺𑀭𑀺𑀬𑀳𑁂𑀢𑀼 𑀅𑀤𑁄𑀲𑁄 𑀓𑀺𑀭𑀺𑀬𑀳𑁂𑀢𑀼, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀫𑁄𑀳𑁄 𑀓𑀺𑀭𑀺𑀬𑀳𑁂𑀢𑀼, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽𑁇

(1442.)

2152\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀳𑁂𑀢𑀽? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀳𑁂𑀢𑀽, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀳𑁂𑀢𑀽𑁇
