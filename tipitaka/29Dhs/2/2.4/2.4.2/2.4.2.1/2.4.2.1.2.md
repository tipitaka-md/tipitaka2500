# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁇𑁨 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

(1443.)

2153. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀳𑁂𑀢𑀼𑀓𑀸? 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀁 𑀫𑁄𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀳𑁂𑀢𑀼𑀓𑀸𑁇

(1444.)

2154. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀳𑁂𑀢𑀼𑀓𑀸? 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄, 𑀤𑁆𑀯𑁂𑀧𑀜𑁆𑀘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀺, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑁄, 𑀧𑀜𑁆𑀘 𑀘 𑀅𑀳𑁂𑀢𑀼𑀓𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑁄, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀳𑁂𑀢𑀼𑀓𑀸𑁇
