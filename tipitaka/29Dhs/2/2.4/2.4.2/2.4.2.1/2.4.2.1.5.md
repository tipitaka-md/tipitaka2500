# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁇𑁫 𑀳𑁂𑀢𑀼𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀳𑁂𑀢𑀼𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1449.)

2159\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘? 𑀬𑀢𑁆𑀣 𑀤𑁆𑀯𑁂 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀏𑀓𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁇

(1450.)

2160\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽? 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀅𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀳𑁂𑀢𑀽 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑁇 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸𑀢𑀺𑀧𑀺, 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑀢𑀺𑀧𑀺𑁇
