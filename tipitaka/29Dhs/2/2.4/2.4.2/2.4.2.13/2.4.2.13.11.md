# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁩𑁇𑁧𑁧 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀤𑀼𑀓

(1601.)

2312. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀩𑁆𑀩𑁄 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇

(1602.)

2313. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇
