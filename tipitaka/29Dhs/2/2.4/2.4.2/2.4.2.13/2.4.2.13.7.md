# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁩𑁇𑁭 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀤𑀼𑀓

(1593.)

2304\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀧𑀜𑁆𑀘, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀧𑀜𑁆𑀘, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓𑀢𑀺𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀤𑀼𑀓𑀢𑀺𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀻𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸𑁇

(1594.)

2305\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀅𑀓𑀼𑀲𑀮𑀢𑁄 𑀅𑀝𑁆𑀞, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀏𑀓𑀸𑀤𑀲, 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀲𑀢𑁆𑀢, 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀙, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓𑀤𑀼𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀭𑀼𑀧𑁆𑀧𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀤𑀼𑀓𑀤𑀼𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀧𑀻𑀢𑀺 𑀘, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸𑁇
