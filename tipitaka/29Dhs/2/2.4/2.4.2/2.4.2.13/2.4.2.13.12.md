# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁩𑁇𑁧𑁨 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓

(1603.)

2314\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀘𑀢𑀼𑀓𑁆𑀓𑀧𑀜𑁆𑀘𑀓𑀚𑁆𑀛𑀸𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁄 𑀘 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀘 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇

(1604.)

2315\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇
