# 𑁨𑁇𑁪𑁇𑁨𑁇𑁩𑁇𑁬 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀤𑀼𑀓

(1475.)

2186. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸? 𑀤𑁆𑀯𑀻𑀲𑀼 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀫𑁄𑀳𑁄, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸𑁇

(1476.)

2187. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀲𑀯𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸𑀢𑀺𑀧𑀺, 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀲𑀯𑀸𑀢𑀺𑀧𑀺𑁇
