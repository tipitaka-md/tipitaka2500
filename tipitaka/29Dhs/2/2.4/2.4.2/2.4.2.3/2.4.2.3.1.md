# 𑁨𑁇𑁪𑁇𑁨𑁇𑁩𑁇𑁧 𑀆𑀲𑀯𑀤𑀼𑀓

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀆𑀲𑀯𑀤𑀼𑀓

(1465.)

2176\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀲𑀯𑀸—  𑀓𑀸𑀫𑀸𑀲𑀯𑁄, 𑀪𑀯𑀸𑀲𑀯𑁄, 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄𑁇 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀅𑀝𑁆𑀞𑀲𑀼 𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀪𑀯𑀸𑀲𑀯𑁄 𑀘𑀢𑀽𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄 𑀘𑀢𑀽𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀲𑀩𑁆𑀩𑀸𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀸𑁇

(1466.)

2177\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀆𑀲𑀯𑀸? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀯𑁂 𑀅𑀯𑀲𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀆𑀲𑀯𑀸𑁇
