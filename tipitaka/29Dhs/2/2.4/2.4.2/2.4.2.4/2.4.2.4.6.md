# 𑁨𑁇𑁪𑁇𑁨𑁇𑁪𑁇𑁬 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀤𑀼𑀓

(1487.)

2198\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸? 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸𑁇

(1488.)

2199\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸𑁇 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸𑀢𑀺𑀧𑀺, 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸𑀢𑀺𑀧𑀺𑁇
