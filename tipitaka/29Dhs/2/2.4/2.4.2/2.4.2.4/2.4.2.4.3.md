# 𑁨𑁇𑁪𑁇𑁨𑁇𑁪𑁇𑁩 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1481.)

2192\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀁 𑀫𑁄𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(1482.)

2193\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
