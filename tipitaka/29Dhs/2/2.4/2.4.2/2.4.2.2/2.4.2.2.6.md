# 𑁨𑁇𑁪𑁇𑁨𑁇𑁨𑁇𑁬 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

### 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

(1463.)

2173. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀺𑀬𑀸? 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇

(1464.)

2174. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇
