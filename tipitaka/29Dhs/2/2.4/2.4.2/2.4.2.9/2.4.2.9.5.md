# 𑁨𑁇𑁪𑁇𑁨𑁇𑁯𑁇𑁫 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀤𑀼𑀓

(1523.)

2234\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀤𑁆𑀯𑁂 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇

(1524.)

2235\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑀢𑀺𑀧𑀺, 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑀢𑀺𑀧𑀺𑁇
