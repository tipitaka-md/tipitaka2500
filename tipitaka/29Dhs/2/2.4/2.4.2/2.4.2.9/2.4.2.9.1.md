# 𑁨𑁇𑁪𑁇𑁨𑁇𑁯𑁇𑁧 𑀧𑀭𑀸𑀫𑀸𑀲𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀤𑀼𑀓

(1515.)

2226\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸? 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑀢𑀽𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇

(1516.)

2227\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀸? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀅𑀯𑀲𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇
