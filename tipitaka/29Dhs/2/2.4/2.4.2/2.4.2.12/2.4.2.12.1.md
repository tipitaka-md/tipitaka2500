# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁨𑁇𑁧 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

### 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓

(1565.)

2276. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀮𑁂𑀲𑀸? 𑀤𑀲 𑀓𑀺𑀮𑁂𑀲𑀯𑀢𑁆𑀣𑀽𑀦𑀺—  𑀮𑁄𑀪𑁄, 𑀤𑁄𑀲𑁄, 𑀫𑁄𑀳𑁄, 𑀫𑀸𑀦𑁄, 𑀤𑀺𑀝𑁆𑀞𑀺, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸, 𑀣𑀺𑀦𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀅𑀳𑀺𑀭𑀺𑀓𑀁, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁𑁇 𑀮𑁄𑀪𑁄 𑀅𑀝𑁆𑀞𑀲𑀼 𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑁄𑀲𑁄 𑀤𑁆𑀯𑀻𑀲𑀼 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀫𑁄𑀳𑁄 𑀲𑀩𑁆𑀩𑀸𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀫𑀸𑀦𑁄 𑀘𑀢𑀽𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑀢𑀽𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀣𑀺𑀦𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀺𑀓𑁂𑀲𑀼 𑀅𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀜𑁆𑀘 𑀅𑀳𑀺𑀭𑀺𑀓𑀜𑁆𑀘 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀸𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀮𑁂𑀲𑀸𑁇

(1566.)

2277. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀓𑀺𑀮𑁂𑀲𑀸? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀺𑀮𑁂𑀲𑁂 𑀅𑀯𑀲𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀓𑀺𑀮𑁂𑀲𑀸𑁇
