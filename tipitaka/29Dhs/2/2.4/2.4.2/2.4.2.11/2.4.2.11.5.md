# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁧𑁇𑁫 𑀉𑀧𑀸𑀤𑀸𑀦𑁅𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀉𑀧𑀸𑀤𑀸𑀦𑁅𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1561.)

2272\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘? 𑀬𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀺 𑀘 𑀮𑁄𑀪𑁄 𑀘 𑀏𑀓𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁇

(1562.)

2273\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸? 𑀅𑀝𑁆𑀞 𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑁇 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸𑀢𑀺𑀧𑀺, 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀢𑀺𑀧𑀺𑁇
