# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁧𑁇𑁩 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀓𑀣𑀸𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀅𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1557.)

2268\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑁄𑀪𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(1558.)

2269\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀘𑀢𑀽𑀲𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀮𑁄𑀪𑁄, 𑀤𑁆𑀯𑁂 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀸, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁄, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀭𑀽𑀧𑀜𑁆𑀘, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
