1096\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸, 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁𑁇 𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀳𑁂𑀢𑀼, 𑀅𑀳𑁂𑀢𑀼𑀓𑀁, 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁, 𑀲𑀗𑁆𑀔𑀢𑀁, 𑀭𑀽𑀧𑀁, 𑀮𑁄𑀓𑀺𑀬𑀁, 𑀲𑀸𑀲𑀯𑀁, 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀁, 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀁, 𑀑𑀖𑀦𑀺𑀬𑀁, 𑀬𑁄𑀕𑀦𑀺𑀬𑀁, 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀁, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁, 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁, 𑀦 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀦 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀁, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁, 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁, 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁, 𑀦𑁂𑀯 𑀆𑀘𑀬𑀕𑀸𑀫𑀺 𑀦 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺, 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀁, 𑀧𑀭𑀺𑀢𑁆𑀢𑀁, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁, 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀦𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀅𑀦𑀺𑀬𑀢𑀁, 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀙𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀁𑁇
