# 𑁨𑁇𑁨𑁇𑁨𑁇𑁩 𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

### 𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀢𑀺𑀓

(585.)

1144\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀲𑀗𑁆𑀕𑀳𑁄—

1145\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇

1146\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

1147\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

1148\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

1149\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

1150\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1151\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

1152\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

1153\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇

1154\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑁇

1155\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇

1156\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

1157\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 [𑁧𑁩]

1158\. 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀢𑀺𑀓𑀁𑁇



1159\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇

1160\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇

1161\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 [𑁨𑁫]

1162\. 𑀯𑀢𑁆𑀣𑀼𑀢𑀺𑀓𑀁𑁇



1163\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇

1164\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 [𑁨𑁫]

---

1165\. 𑀆𑀭𑀫𑁆𑀫𑀡𑀢𑀺𑀓𑀁𑁇



1166\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁𑁇

1167\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦 𑀭𑀲𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁𑁇 [𑁧𑁦]

1168\. 𑀆𑀬𑀢𑀦𑀢𑀺𑀓𑀁𑁇



1169\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀖𑀸𑀦𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑁇

1170\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀭𑀲𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑁇 [𑁧𑁦]

---

1171\. 𑀥𑀸𑀢𑀼𑀢𑀺𑀓𑀁𑁇



1172\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1173\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1174\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1175\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 [𑁮]

1176\. 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀢𑀺𑀓𑀁𑁇



1177\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

1178\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

1179\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼𑁇

1180\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀆𑀧𑁄𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑁇

1181\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀮𑀳𑀼𑀢𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀮𑀳𑀼𑀢𑀸𑁇

1182\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀫𑀼𑀤𑀼𑀢𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀫𑀼𑀤𑀼𑀢𑀸𑁇

1183\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸𑁇

1184\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀉𑀧𑀘𑀬𑁄𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀉𑀧𑀘𑀬𑁄𑁇

1185\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀲𑀦𑁆𑀢𑀢𑀺𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀲𑀦𑁆𑀢𑀢𑀺𑁇

1186\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀚𑀭𑀢𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀚𑀭𑀢𑀸𑁇

1187\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸𑁇

1188\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀢𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇 [𑁧𑁨]
