# 𑁨𑁇𑁨𑁇𑁨𑁇𑁨 𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

* 𑀫𑀸𑀢𑀺𑀓𑀸

* 𑀤𑀼𑀓

1098\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀲𑀗𑁆𑀕𑀳𑁄—

1099\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇

1100\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

1101\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

1102\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

1103\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

1104\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1105\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

1106\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

1107\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇

1108\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑁇

1109\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇

1110\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁𑁇

1111\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

1112\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 [𑁧𑁪]

---

1113\. 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀤𑀼𑀓𑀁𑁇



1114\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇

1115\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 [𑁨𑁫]

---

1116\. 𑀯𑀢𑁆𑀣𑀼𑀤𑀼𑀓𑀁𑁇



1117\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇

1118\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 [𑁨𑁫]

---

1119\. 𑀆𑀭𑀫𑁆𑀫𑀡𑀤𑀼𑀓𑀁𑁇



1120\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁𑁇

1121\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁𑁇 [𑁧𑁦]

---

1122\. 𑀆𑀬𑀢𑀦𑀤𑀼𑀓𑀁𑁇



1123\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑁄𑀢𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀖𑀸𑀦𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀓𑀸𑀬𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑁇

1124\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀭𑀲𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑁇 [𑁧𑁦]

---

1125\. 𑀥𑀸𑀢𑀼𑀤𑀼𑀓𑀁𑁇



1126\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1127\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1128\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1129\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 [𑁮]

---

1130\. 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀤𑀼𑀓𑀁𑁇



1131\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

1132\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

1133\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼𑁇

1134\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑁇

1135\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀮𑀳𑀼𑀢𑀸𑁇

1136\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀫𑀼𑀤𑀼𑀢𑀸𑁇

1137\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸𑁇

1138\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀉𑀧𑀘𑀬𑁄𑁇

1139\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀲𑀦𑁆𑀢𑀢𑀺𑁇

1140\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀚𑀭𑀢𑀸𑁇

1141\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸𑁇

1142\. 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇 [𑁧𑁨]
