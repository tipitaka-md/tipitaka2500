# 𑁨𑁇𑁨𑁇𑁨𑁇𑁪 𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

* 𑀫𑀸𑀢𑀺𑀓𑀸

* 𑀘𑀢𑀼𑀓𑁆𑀓

(586.)

1190\. 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀲𑀗𑁆𑀕𑀳𑁄—

1191\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

1192\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

1193\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

1194\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

1195\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

1196\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

1197\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

1198\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

1199\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

1200\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

1201\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

1202\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

1203\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

1204\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

1205\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

1206\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1207\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

1208\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

1209\. 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇
