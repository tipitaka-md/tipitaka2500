# 𑁨𑁇𑁨𑁇𑁩𑁇𑁪 𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

### 𑀭𑀽𑀧𑀯𑀺𑀪𑀢𑁆𑀢𑀺

### 𑀘𑀢𑀼𑀓𑁆𑀓

(882.)

1538\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

(883.)

1539\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

(884.)

1540\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁? 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

(885.)

1541\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁? 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

(886.)

1542\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

(887.)

1543\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

(888.)

1544\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁? 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

(889.)

1545\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁? 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

(890.)

1546\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(891.)

1547\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(892.)

1548\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(893.)

1549\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(894.)

1550\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(895.)

1551\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(896.)

1552\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(897.)

1554\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀤𑀽𑀭𑁂? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀤𑀽𑀭𑁂𑁇

1555\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

1556\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀤𑀽𑀭𑁂? 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀤𑀽𑀭𑁂𑁇

(901.)

1557\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

(902.)

1558\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(903.)

1559\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(904.)

1560\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(905.)

1561\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(906.)

1562\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(907.)

1563\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(908.)

1564\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(909.)

1565\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(910.)

1566\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(911.)

1567\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(912.)

1568\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(913.)

1569\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(914.)

1570\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(915.)

1571\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(916.)

1572\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(917.)

1573\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(918.)

1574\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀤𑀽𑀭𑁂? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀤𑀽𑀭𑁂𑁇

(919.)

1575\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

(920.)

1576\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀤𑀽𑀭𑁂? 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀤𑀽𑀭𑁂𑁇

(921.)

1577\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

(922.)

1578\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(923.)

1579\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(924.)

1580\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(925.)

1581\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(926.)

1582\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(927.)

1583\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(928.)

1584\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(929.)

1585\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(930.)

1586\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(931.)

1587\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(932.)

1588\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(933.)

1589\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇

(934.)

1590\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(935.)

1591\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(936.)

1592\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(937.)

1593\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(938.)

1594\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀤𑀽𑀭𑁂? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀤𑀽𑀭𑁂𑁇

(939.)

1611\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

(956.)

1612\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀤𑀽𑀭𑁂? 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀤𑀽𑀭𑁂𑁇

(957.)

1613\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

(958.)

1614\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(959.)

1615\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(960.)

1616\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

(961.)

1617\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(962.)

1618\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀤𑀽𑀭𑁂? 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀤𑀽𑀭𑁂𑁇

(963.)

1619\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

(964.)

1620\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀤𑀽𑀭𑁂? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀤𑀽𑀭𑁂𑁇

(965.)

1621\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇

(966.)

1622\. 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀲𑀼𑀢𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀫𑀼𑀢𑀁, 𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁 𑀫𑀦𑀲𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀭𑀽𑀧𑀁𑁇

---

1623\. 𑀏𑀯𑀁 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀲𑀗𑁆𑀕𑀳𑁄𑁇
