# 𑁨𑁇𑁨𑁇𑁩𑁇𑁫 𑀧𑀜𑁆𑀘𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

* 𑀭𑀽𑀧𑀯𑀺𑀪𑀢𑁆𑀢𑀺

* 𑀧𑀜𑁆𑀘𑀓

(967.)

1624\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀔𑀭𑀕𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀪𑀸𑀯𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑁇

(968.)

1625\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀆𑀧𑁄 𑀆𑀧𑁄𑀕𑀢𑀁 𑀲𑀺𑀦𑁂𑀳𑁄 𑀲𑀺𑀦𑁂𑀳𑀕𑀢𑀁 𑀩𑀦𑁆𑀥𑀦𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑁇

(969.)

1626\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀢𑁂𑀚𑁄 𑀢𑁂𑀚𑁄𑀕𑀢𑀁 𑀉𑀲𑁆𑀫𑀸 𑀉𑀲𑁆𑀫𑀸𑀕𑀢𑀁 𑀉𑀲𑀼𑀫𑀁 𑀉𑀲𑀼𑀫𑀕𑀢𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑁇

(970.)

1627\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀯𑀸𑀬𑁄 𑀯𑀸𑀬𑁄𑀕𑀢𑀁 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀸—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑁇

(971.)

1628\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑁇

---
