# 𑁨𑁇𑁨𑁇𑁩𑁇𑁧𑁦 𑀤𑀲𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

### 𑀭𑀽𑀧𑀯𑀺𑀪𑀢𑁆𑀢𑀺

### 𑀤𑀲𑀓

(978.)

1640\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? 𑀬𑀁 𑀘𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄…𑀧𑁂…  𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

(979.)

1641\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? 𑀬𑁄 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀆𑀬𑀼 𑀞𑀺𑀢𑀺 𑀬𑀧𑀦𑀸 𑀬𑀸𑀧𑀦𑀸 𑀇𑀭𑀺𑀬𑀦𑀸 𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀸𑀮𑀦𑀸 𑀚𑀻𑀯𑀺𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

(980.)

1642\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(981.)

1643\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

---

1644\. 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀲𑀗𑁆𑀕𑀳𑁄𑁇
