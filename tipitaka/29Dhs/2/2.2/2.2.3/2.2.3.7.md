### 𑀲𑀢𑁆𑀢𑀓

(973.)

1632\. 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁𑁇
