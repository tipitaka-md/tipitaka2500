# 𑁨𑁇𑁨𑁇𑁩𑁇𑁧𑁧 𑀏𑀓𑀸𑀤𑀲𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

* 𑀭𑀽𑀧𑀯𑀺𑀪𑀢𑁆𑀢𑀺

* 𑀏𑀓𑀸𑀤𑀲𑀓

(982.)

1645\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄…𑀧𑁂…  𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇

(983.)

1646\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁? 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁𑁇

(984.)

1647\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇

---

1648\. 𑀏𑀯𑀁 𑀏𑀓𑀸𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀲𑀗𑁆𑀕𑀳𑁄𑁇

  
𑀏𑀓𑀸𑀤𑀲𑀓𑀁𑁇

  
𑀅𑀝𑁆𑀞𑀫𑁄 𑀪𑀸𑀡𑀯𑀸𑀭𑁄𑁇

  
𑀭𑀽𑀧𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑁇
