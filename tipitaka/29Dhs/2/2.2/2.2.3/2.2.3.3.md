# 𑁨𑁇𑁨𑁇𑁩𑁇𑁩 𑀢𑀺𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

* 𑀭𑀽𑀧𑀯𑀺𑀪𑀢𑁆𑀢𑀺

* 𑀢𑀺𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(744.)

1393\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀸? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀸𑁇

(745.)

1394\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀉𑀧𑀸𑀤𑀸? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀉𑀧𑀸𑀤𑀸𑁇

(746.)

1395\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸? 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇

(747.)

1396\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

(748.)

1397\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

(749.)

1398\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇

(750.)

1399\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

(751.)

1400\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

(752.)

1401\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇

(753.)

1402\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(754.)

1403\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(755.)

1404\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁? 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 …𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇

(756.)

1405\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(757.)

1406\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(758.)

1407\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇

(759.)

1523\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄? 𑀬𑁄 𑀆𑀬𑀢𑀦𑀸𑀦𑀁 𑀆𑀘𑀬𑁄, 𑀲𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄𑁇

(869.)

1524\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀉𑀧𑀘𑀬𑁄? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀉𑀧𑀘𑀬𑁄𑁇

(870.)

1525\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀲𑀦𑁆𑀢𑀢𑀺? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀲𑀦𑁆𑀢𑀢𑀺𑁇

(871.)

1526\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺? 𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀲𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀢𑀺𑁇

(872.)

1527\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀲𑀦𑁆𑀢𑀢𑀺? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀲𑀦𑁆𑀢𑀢𑀺𑁇

(873.)

1528\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀚𑀭𑀢𑀸? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀚𑀭𑀢𑀸𑁇

(874.)

1529\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸? 𑀬𑀸 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀸 𑀚𑀻𑀭𑀡𑀢𑀸 𑀔𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀧𑀸𑀮𑀺𑀘𑁆𑀘𑀁 𑀯𑀮𑀺𑀢𑁆𑀢𑀘𑀢𑀸 𑀆𑀬𑀼𑀦𑁄 𑀲𑀁𑀳𑀸𑀦𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀸𑀓𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀚𑀭𑀢𑀸𑁇

(875.)

1530\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀚𑀭𑀢𑀸? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀚𑀭𑀢𑀸𑁇

(876.)

1531\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸𑁇

(877.)

1532\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸? 𑀬𑁄 𑀭𑀽𑀧𑀲𑁆𑀲 𑀔𑀬𑁄 𑀯𑀬𑁄 𑀪𑁂𑀤𑁄 𑀧𑀭𑀺𑀪𑁂𑀤𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸𑁇

(878.)

1533\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸𑁇

(879.)

1534\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇

(880.)

1535\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄? 𑀑𑀤𑀦𑁄 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑁄 𑀲𑀢𑁆𑀢𑀼 𑀫𑀘𑁆𑀙𑁄 𑀫𑀁𑀲𑀁 𑀔𑀻𑀭𑀁 𑀤𑀥𑀺 𑀲𑀧𑁆𑀧𑀺 𑀦𑀯𑀦𑀻𑀢𑀁 𑀢𑁂𑀮𑀁 𑀫𑀥𑀼 𑀨𑀸𑀡𑀺𑀢𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀬𑀫𑁆𑀳𑀺 𑀬𑀫𑁆𑀳𑀺 𑀚𑀦𑀧𑀤𑁂 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀫𑀼𑀔𑀸𑀲𑀺𑀬𑀁 𑀤𑀦𑁆𑀢𑀯𑀺𑀔𑀸𑀤𑀦𑀁 𑀕𑀮𑀚𑁆𑀛𑁄𑀳𑀭𑀡𑀻𑀬𑀁 𑀓𑀼𑀘𑁆𑀙𑀺𑀯𑀺𑀢𑁆𑀣𑀫𑁆𑀪𑀦𑀁 𑀬𑀸𑀬 𑀑𑀚𑀸𑀬 𑀲𑀢𑁆𑀢𑀸 𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇

(881.)

1536\. 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄? 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸—  𑀇𑀤𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇

---

1537\. 𑀏𑀯𑀁 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀲𑀗𑁆𑀕𑀳𑁄𑁇
