# 𑁨𑁇𑁨𑁇𑁩𑁇𑁧 𑀏𑀓𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀭𑀽𑀧𑀓𑀡𑁆𑀟

* 𑀭𑀽𑀧𑀯𑀺𑀪𑀢𑁆𑀢𑀺

* 𑀏𑀓𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(594.)

1234\. 𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀳𑁂𑀢𑀼𑀫𑁂𑀯, 𑀅𑀳𑁂𑀢𑀼𑀓𑀫𑁂𑀯, 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀫𑁂𑀯, 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀫𑁂𑀯, 𑀲𑀗𑁆𑀔𑀢𑀫𑁂𑀯, 𑀭𑀽𑀧𑀫𑁂𑀯, 𑀮𑁄𑀓𑀺𑀬𑀫𑁂𑀯, 𑀲𑀸𑀲𑀯𑀫𑁂𑀯, 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀫𑁂𑀯, 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀫𑁂𑀯, 𑀑𑀖𑀦𑀺𑀬𑀫𑁂𑀯, 𑀬𑁄𑀕𑀦𑀺𑀬𑀫𑁂𑀯, 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀫𑁂𑀯, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀫𑁂𑀯, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀫𑁂𑀯, 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀫𑁂𑀯, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑁂𑀯, 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀫𑁂𑀯, 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀫𑁂𑀯, 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀫𑁂𑀯, 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀫𑁂𑀯, 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀫𑁂𑀯, 𑀦 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀫𑁂𑀯, 𑀦 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀫𑁂𑀯, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀫𑁂𑀯, 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀫𑁂𑀯, 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀫𑁂𑀯, 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀫𑁂𑀯, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀫𑁂𑀯, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀫𑁂𑀯, 𑀦𑁂𑀯 𑀆𑀘𑀬𑀕𑀸𑀫𑀺 𑀦 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀫𑁂𑀯, 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀫𑁂𑀯, 𑀧𑀭𑀺𑀢𑁆𑀢𑀫𑁂𑀯, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀫𑁂𑀯, 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀫𑁂𑀯, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀫𑁂𑀯, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑁂𑀯, 𑀦𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑁂𑀯, 𑀅𑀦𑀺𑀬𑀢𑀫𑁂𑀯, 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀫𑁂𑀯, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀙𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀫𑁂𑀯, 𑀅𑀦𑀺𑀘𑁆𑀘𑀫𑁂𑀯, 𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀫𑁂𑀯𑁇
