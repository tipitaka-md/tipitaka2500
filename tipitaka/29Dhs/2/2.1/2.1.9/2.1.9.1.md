# 𑁨𑁇𑁧𑁇𑁯𑁇𑁧 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑀸𑀓𑀧𑀜𑁆𑀘𑀯𑀺𑀜𑁆𑀜𑀸𑀡

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

* 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑀸𑀓𑀅𑀩𑁆𑀬𑀸𑀓𑀢

* 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑀸𑀓𑀧𑀜𑁆𑀘𑀯𑀺𑀜𑁆𑀜𑀸𑀡

(556.)

1052\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁…𑀧𑁂…  𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀤𑁆𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀕𑀦𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀭𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

(557.)

1053\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺𑁇

(558.)

1054\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀚𑁆𑀚𑀸𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

(559.)

1055\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

(560.)

1056\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

1057\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸…𑀧𑁂… 𑁇

(561.)

1058\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

---
