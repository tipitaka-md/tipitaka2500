# 𑁨𑁇𑁧𑁇𑁧𑁦𑁇𑁫 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀺𑀭𑀺𑀬𑀸

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀳𑁂𑀢𑀼𑀓𑀓𑀺𑀭𑀺𑀬𑀸𑀅𑀩𑁆𑀬𑀸𑀓𑀢

### 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀺𑀭𑀺𑀬𑀸

(577.)

1086\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀓𑀺𑀭𑀺𑀬𑀁 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀁 𑀦𑀸𑀓𑀼𑀲𑀮𑀁 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

(578.)

1087\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀓𑀺𑀭𑀺𑀬𑀁 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀁 𑀦𑀸𑀓𑀼𑀲𑀮𑀁 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀘𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇
