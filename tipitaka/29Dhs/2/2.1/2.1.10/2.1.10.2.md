# 𑁨𑁇𑁧𑁇𑁧𑁦𑁇𑁨 𑀓𑀺𑀭𑀺𑀬𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀅𑀳𑁂𑀢𑀼𑀓𑀓𑀺𑀭𑀺𑀬𑀸𑀅𑀩𑁆𑀬𑀸𑀓𑀢

### 𑀓𑀺𑀭𑀺𑀬𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢

(568.)

1072\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀸 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

(569.)

1073\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

(570.)

1074\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(571.)

1075\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(572.)

1076\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

1077\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀸 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸…𑀧𑁂… 𑁇

(573.)

1078\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀧𑀻𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

---

1079\. 𑀓𑀺𑀭𑀺𑀬𑀸 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀸𑁇
