# 𑁨𑁇𑁧𑁇𑁫𑁇𑁭 𑀅𑀥𑀺𑀧𑀢𑀺

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀓𑀼𑀲𑀮

### 𑀅𑀥𑀺𑀧𑀢𑀺

(358.)

820\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(359.)

821\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀘𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(360.)

822\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀘𑁆𑀘𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑀣𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀔𑀦𑁆𑀥𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀆𑀬𑀢𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀸𑀢𑀼𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀆𑀳𑀸𑀭𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀨𑀲𑁆𑀲𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑁂𑀤𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑁂𑀢𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

823\. 𑀅𑀥𑀺𑀧𑀢𑀺𑁇

  
𑀧𑀞𑀫𑁄 𑀫𑀕𑁆𑀕𑁄𑁇



(361.)

824\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸𑀦𑀁 𑀢𑀦𑀼𑀪𑀸𑀯𑀸𑀬 𑀤𑀼𑀢𑀺𑀬𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

825\. 𑀤𑀼𑀢𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄𑁇



(362.)

826\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬 𑀢𑀢𑀺𑀬𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

828\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀭𑀽𑀧𑀭𑀸𑀕𑀅𑀭𑀽𑀧𑀭𑀸𑀕𑀫𑀸𑀦𑁅𑀤𑁆𑀥𑀘𑁆𑀘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀅𑀦𑀯𑀲𑁂𑀲𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬 𑀘𑀢𑀼𑀢𑁆𑀣𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

(364.)

829\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀜𑀸𑀢𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇
