# 𑁨𑁇𑁧𑁇𑁨𑁇𑁧 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀬

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀬

(160.)

566\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(161.)

567\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

568\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀺𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

(162.)

569\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀧𑀻𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄…𑀧𑁂…  𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(163.)

570\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  “𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻”𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

571\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑀼𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

572\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄…𑀧𑁂…  𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(165.)

573\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

574\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑀼𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

(166.)

575\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄…𑀧𑁂…  𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇
