# 𑁨𑁇𑁧𑁇𑁨𑁇𑁨𑁦 𑀅𑀲𑀼𑀪𑀛𑀸𑀦𑀁 𑀲𑁄𑀍𑀅𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀓𑀁

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

## 710\. 𑀅𑀲𑀼𑀪𑀛𑀸𑀦𑀁 𑀲𑁄𑀍𑀅𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀓𑀁

(263.)

711\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(264.)

712\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑀺𑀦𑀻𑀮𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀼𑀩𑁆𑀩𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀓𑁆𑀔𑀸𑀬𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀳𑀢𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀮𑁄𑀳𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀧𑀼𑀍𑀅𑀯𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀅𑀝𑁆𑀞𑀺𑀓𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇
