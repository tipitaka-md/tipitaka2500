# 𑁨𑁇𑁧𑁇𑁮𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑀸𑀤𑀺𑀫𑀕𑁆𑀕𑀯𑀺𑀧𑀸𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

* 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀯𑀺𑀧𑀸𑀓-𑀧𑀞𑀫𑀫𑀕𑁆𑀕𑀯𑀺𑀧𑀸𑀓

* 𑀤𑀼𑀢𑀺𑀬𑀸𑀤𑀺𑀫𑀕𑁆𑀕𑀯𑀺𑀧𑀸𑀓

(553.)

1048\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸𑀦𑀁 𑀢𑀦𑀼𑀪𑀸𑀯𑀸𑀬 𑀤𑀼𑀢𑀺𑀬𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀫𑀭𑀸𑀕𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬 𑀢𑀢𑀺𑀬𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸…𑀧𑁂…  𑀭𑀽𑀧𑀭𑀸𑀕𑀅𑀭𑀽𑀧𑀭𑀸𑀕𑀫𑀸𑀦𑁅𑀤𑁆𑀥𑀘𑁆𑀘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀅𑀦𑀯𑀲𑁂𑀲𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬 𑀘𑀢𑀼𑀢𑁆𑀣𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

(554.)

1049\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

(555.)

1050\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇
