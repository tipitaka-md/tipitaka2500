1027. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

1028. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

1029. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

1046. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀘𑁆𑀘𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑀣𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀔𑀦𑁆𑀥𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀆𑀬𑀢𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀸𑀢𑀼𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀆𑀳𑀸𑀭𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀨𑀲𑁆𑀲𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑁂𑀤𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑁂𑀢𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁…𑀧𑁂…  𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀁 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇
