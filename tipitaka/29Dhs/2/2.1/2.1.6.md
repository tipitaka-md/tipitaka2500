# 𑁨𑁇𑁧𑁇𑁬 𑀤𑁆𑀯𑀸𑀤𑀲 𑀅𑀓𑀼𑀲𑀮

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

* 𑀤𑁆𑀯𑀸𑀤𑀲 𑀅𑀓𑀼𑀲𑀮

(365.)

831\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀲𑀤𑁆𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀕𑀦𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀭𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀮𑁄𑀪𑁄 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀣𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

(366.)

832\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺𑁇

(367.)

833\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

(368.)

834\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺𑁇

(369.)

835\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

(370.)

836\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(371.)

837\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀧𑁆𑀧𑀦𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀦𑀸 𑀘𑁂𑀢𑀲𑁄 𑀅𑀪𑀺𑀦𑀺𑀭𑁄𑀧𑀦𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

(372.)

838\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀸𑀭𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁄 𑀉𑀧𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀸𑀦𑀢𑀸 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇

(373.)

839\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀻𑀢𑀺 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀆𑀫𑁄𑀤𑀦𑀸 𑀧𑀫𑁄𑀤𑀦𑀸 𑀳𑀸𑀲𑁄 𑀧𑀳𑀸𑀲𑁄 𑀯𑀺𑀢𑁆𑀢𑀺 𑀑𑀤𑀕𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇

(374.)

840\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺𑁇

(375.)

841\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(376.)

842\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(377.)

843\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(378.)

844\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(379.)

845\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(380.)

846\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑁂𑀲𑀁 𑀅𑀭𑀽𑀧𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀆𑀬𑀼 𑀞𑀺𑀢𑀺 𑀬𑀧𑀦𑀸 𑀬𑀸𑀧𑀦𑀸 𑀇𑀭𑀺𑀬𑀦𑀸 𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀸𑀮𑀦𑀸 𑀚𑀻𑀯𑀺𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(381.)

847\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀦𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀕𑀸𑀳𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀳𑁄 𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀣𑁄 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀁 𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇

(382.)

848\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀧𑁆𑀧𑀦𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀦𑀸 𑀘𑁂𑀢𑀲𑁄 𑀅𑀪𑀺𑀦𑀺𑀭𑁄𑀧𑀦𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺𑁇

(383.)

849\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇

(384.)

850\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺𑁇

(385.)

851\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(386.)

852\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(387.)

853\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀭𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(388.)

854\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(389.)

855\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀪𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀪𑁄 𑀮𑀼𑀩𑁆𑀪𑀦𑀸 𑀮𑀼𑀩𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀲𑀸𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑀸 𑀲𑀸𑀭𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀪𑁄 𑀳𑁄𑀢𑀺𑁇

(390.)

856\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑁄 𑀅𑀲𑀫𑁆𑀩𑁄𑀥𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀅𑀲𑀗𑁆𑀕𑀸𑀳𑀦𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑀦𑀸 𑀅𑀲𑀫𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀫𑁆𑀫𑁂𑀚𑁆𑀛𑀁 𑀩𑀸𑀮𑁆𑀬𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀫𑁄𑀳𑁄 𑀧𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀫𑁄𑀳𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑁄𑀖𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄𑀕𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺𑁇

(391.)

857\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀪𑁄 𑀮𑀼𑀩𑁆𑀪𑀦𑀸 𑀮𑀼𑀩𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀲𑀸𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑀸 𑀲𑀸𑀭𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀳𑁄𑀢𑀺𑁇

(392.)

858\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀦𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀕𑀸𑀳𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀳𑁄 𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀣𑁄 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀁 𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇

(393.)

859\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀭𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀳𑁄𑀢𑀺𑁇

(394.)

860\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺𑁇

(395.)

861\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀣𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀣𑁄 𑀳𑁄𑀢𑀺𑁇

(396.)

862\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺𑁇

(397.)

863\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

864\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

(398.)

865\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀧𑀻𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀮𑁄𑀪𑁄 𑀫𑁄𑀳𑁄 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀲𑀫𑀣𑁄 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

(399.)

888\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

889\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

890\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀺𑀯𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

891\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀤𑁄𑀲𑁄 𑀫𑁄𑀳𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀲𑀫𑀣𑁄 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

898\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀫𑁄𑀳𑁄 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀕𑁆𑀕𑀸𑀳𑁄; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

(427.)

899\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸…𑀧𑁂…  𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀣𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

(428.)

900\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

(429.)

901\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀯𑀽𑀧𑀲𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀪𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀳𑁄𑀢𑀺 …𑀧𑁂…  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇

902\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀺𑀯𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀳𑁂𑀢𑀼 𑀳𑁄𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

(430.)

903\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀩𑀮𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀫𑁄𑀳𑁄 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀲𑀫𑀣𑁄 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑁇
