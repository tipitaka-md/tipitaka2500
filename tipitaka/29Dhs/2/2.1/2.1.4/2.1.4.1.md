# 𑁨𑁇𑁧𑁇𑁪𑁇𑁧 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀢𑁂𑀪𑀽𑀫𑀓𑀓𑀼𑀲𑀮

### 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

(269.)

720\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(270.)

721\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦…𑀧𑁂…  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇
