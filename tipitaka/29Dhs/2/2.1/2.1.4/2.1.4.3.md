# 𑁨𑁇𑁧𑁇𑁪𑁇𑁩 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀢𑁂𑀪𑀽𑀫𑀓𑀓𑀼𑀲𑀮

### 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

(273.)

726\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(274.)

727\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(275.)

728\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(276.)

729\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀫𑀚𑁆𑀛𑀺𑀫𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇
