# 𑁨𑁇𑁧𑁇𑁧𑁇𑁧 𑀧𑀤𑀪𑀸𑀚𑀦𑀻

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

### 𑀧𑀤𑀪𑀸𑀚𑀦𑀻

(1.)

368\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀲𑀤𑁆𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀕𑀦𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀭𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂—

369\. 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, (𑁧)

370\. 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, (𑁨)

371\. 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, (𑁩)

372\. 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺, (𑁪)

373\. 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀢𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀳𑀺𑀭𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, (𑁫)

374\. 𑀅𑀮𑁄𑀪𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, (𑁬)

375\. 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺, (𑁭)

376\. 𑀓𑀸𑀬𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀮𑀳𑀼𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀮𑀳𑀼𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀫𑀼𑀤𑀼𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑀼𑀤𑀼𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀼𑀚𑀼𑀓𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀼𑀚𑀼𑀓𑀢𑀸 𑀳𑁄𑀢𑀺, (𑁮)

377\. 𑀲𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀳𑁄𑀢𑀺, (𑁯)

378\. 𑀲𑀫𑀣𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀳𑁄𑀢𑀺, (𑁧𑁦)

379\. 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺; (𑁧𑁧)

380\. 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(2.)

381\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺𑁇

(3.)

382\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

(4.)

383\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺𑁇

(5.)

384\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

(6.)

385\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(7.)

386\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀧𑁆𑀧𑀦𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀦𑀸 𑀘𑁂𑀢𑀲𑁄 𑀅𑀪𑀺𑀦𑀺𑀭𑁄𑀧𑀦𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

(8.)

387\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀸𑀭𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁄 𑀉𑀧𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀸𑀦𑀢𑀸 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇

(9.)

388\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀻𑀢𑀺 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀆𑀫𑁄𑀤𑀦𑀸 𑀧𑀫𑁄𑀤𑀦𑀸 𑀳𑀸𑀲𑁄 𑀧𑀳𑀸𑀲𑁄 𑀯𑀺𑀢𑁆𑀢𑀺 𑀑𑀤𑀕𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇

(10.)

389\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺𑁇

(11.)

390\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(12.)

391\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀤𑁆𑀥𑀸 𑀲𑀤𑁆𑀤𑀳𑀦𑀸 𑀑𑀓𑀧𑁆𑀧𑀦𑀸 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀸𑀤𑁄 𑀲𑀤𑁆𑀥𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(13.)

392\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(14.)

393\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑀺 𑀲𑀭𑀡𑀢𑀸 𑀥𑀸𑀭𑀡𑀢𑀸 𑀅𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀅𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸 𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(15.)

394\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(16.)

395\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(17.)

396\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(18.)

397\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(19.)

398\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑁂𑀲𑀁 𑀅𑀭𑀽𑀧𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀆𑀬𑀼 𑀞𑀺𑀢𑀺 𑀬𑀧𑀦𑀸 𑀬𑀸𑀧𑀦𑀸 𑀇𑀭𑀺𑀬𑀦𑀸 𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀸𑀮𑀦𑀸 𑀚𑀻𑀯𑀺𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇

(20.)

399\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇

(21.)

400\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀧𑁆𑀧𑀦𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀦𑀸 𑀘𑁂𑀢𑀲𑁄 𑀅𑀪𑀺𑀦𑀺𑀭𑁄𑀧𑀦𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺𑁇

(22.)

401\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇

(23.)

402\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑀺 𑀲𑀭𑀡𑀢𑀸 𑀥𑀸𑀭𑀡𑀢𑀸 𑀅𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀅𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸 𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀳𑁄𑀢𑀺𑁇

(24.)

403\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺𑁇

(25.)

404\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀤𑁆𑀥𑀸 𑀲𑀤𑁆𑀤𑀳𑀦𑀸 𑀑𑀓𑀧𑁆𑀧𑀦𑀸 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀸𑀤𑁄 𑀲𑀤𑁆𑀥𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(26.)

405\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(27.)

406\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑀺 𑀲𑀭𑀡𑀢𑀸 𑀥𑀸𑀭𑀡𑀢𑀸 𑀅𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀅𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸 𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(28.)

407\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(29.)

408\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(30.)

409\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑀺𑀭𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀭𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑀺𑀭𑀺𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(31.)

410\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀳𑁄𑀢𑀺𑁇

(32.)

411\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀮𑁄𑀪𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀮𑁄𑀪𑁄 𑀅𑀮𑀼𑀩𑁆𑀪𑀦𑀸 𑀅𑀮𑀼𑀩𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑀸 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀮𑁄𑀪𑁄 𑀳𑁄𑀢𑀺𑁇

(33.)

412\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀤𑁄𑀲𑁄 𑀅𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀅𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺𑁇

(34.)

413\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀫𑁄𑀳𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺𑁇

(35.)

414\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀮𑁄𑀪𑁄 𑀅𑀮𑀼𑀩𑁆𑀪𑀦𑀸 𑀅𑀮𑀼𑀩𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑀸 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀳𑁄𑀢𑀺𑁇

(36.)

415\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀤𑁄𑀲𑁄 𑀅𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀅𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺𑁇

(37.)

416\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇

(38.)

417\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀭𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺𑁇

(39.)

418\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺𑁇

(40.)

419\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀳𑁄𑀢𑀺𑁇

(41.)

420\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀳𑁄𑀢𑀺𑁇

(42.)

421\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀮𑀳𑀼𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀮𑀳𑀼𑀧𑀭𑀺𑀡𑀸𑀫𑀢𑀸 𑀅𑀤𑀦𑁆𑀥𑀦𑀢𑀸 𑀅𑀯𑀺𑀢𑁆𑀣𑀦𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀮𑀳𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(43.)

422\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀮𑀳𑀼𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀮𑀳𑀼𑀢𑀸 𑀮𑀳𑀼𑀧𑀭𑀺𑀡𑀸𑀫𑀢𑀸 𑀅𑀤𑀦𑁆𑀥𑀦𑀢𑀸 𑀅𑀯𑀺𑀢𑁆𑀣𑀦𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀮𑀳𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(44.)

423\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀫𑀼𑀤𑀼𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀫𑀤𑁆𑀤𑀯𑀢𑀸 𑀅𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑀸 𑀅𑀓𑀣𑀺𑀦𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀫𑀼𑀤𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(45.)

424\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀫𑀼𑀤𑀼𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀫𑀼𑀤𑀼𑀢𑀸 𑀫𑀤𑁆𑀤𑀯𑀢𑀸 𑀅𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑀸 𑀅𑀓𑀣𑀺𑀦𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀫𑀼𑀤𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(46.)

425\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀪𑀸𑀯𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(47.)

426\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀪𑀸𑀯𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(48.)

427\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀕𑀼𑀡𑀢𑀸 𑀧𑀕𑀼𑀡𑀢𑁆𑀢𑀁 𑀧𑀕𑀼𑀡𑀪𑀸𑀯𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(49.)

428\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀕𑀼𑀡𑀢𑀸 𑀧𑀕𑀼𑀡𑀢𑁆𑀢𑀁 𑀧𑀕𑀼𑀡𑀪𑀸𑀯𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(50.)

429\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀼𑀚𑀼𑀓𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀉𑀚𑀼𑀢𑀸 𑀉𑀚𑀼𑀓𑀢𑀸 𑀅𑀚𑀺𑀫𑁆𑀳𑀢𑀸 𑀅𑀯𑀗𑁆𑀓𑀢𑀸 𑀅𑀓𑀼𑀝𑀺𑀮𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀬𑀼𑀚𑀼𑀓𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(51.)

430\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀚𑀼𑀓𑀢𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀉𑀚𑀼𑀢𑀸 𑀉𑀚𑀼𑀓𑀢𑀸 𑀅𑀚𑀺𑀫𑁆𑀳𑀢𑀸 𑀅𑀯𑀗𑁆𑀓𑀢𑀸 𑀅𑀓𑀼𑀝𑀺𑀮𑀢𑀸—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀼𑀚𑀼𑀓𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(52.)

431\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑀺 𑀲𑀭𑀡𑀢𑀸 𑀥𑀸𑀭𑀡𑀢𑀸 𑀅𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀅𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸 𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀳𑁄𑀢𑀺𑁇

(53.)

432\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀳𑁄𑀢𑀺𑁇

(54.)

433\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀣𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑀣𑁄 𑀳𑁄𑀢𑀺𑁇

(55.)

434\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

(56.)

435\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺? 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀉𑀬𑁆𑀬𑀸𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀣𑀸𑀫𑁄 𑀥𑀺𑀢𑀺 𑀅𑀲𑀺𑀣𑀺𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀥𑀼𑀭𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀳𑁄𑀢𑀺𑁇

(57.)

436\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀅𑀯𑀺𑀲𑀸𑀳𑀸𑀭𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀺𑀲𑀸𑀳𑀝𑀫𑀸𑀦𑀲𑀢𑀸 𑀲𑀫𑀣𑁄 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇

437\. 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

---
