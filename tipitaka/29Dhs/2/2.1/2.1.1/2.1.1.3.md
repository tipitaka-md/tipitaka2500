# 𑁨𑁇𑁧𑁇𑁧𑁇𑁩 𑀲𑀼𑀜𑁆𑀜𑀢𑀯𑀸𑀭

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀡𑁆𑀟

### 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

### 𑀲𑀼𑀜𑁆𑀜𑀢𑀯𑀸𑀭

(121.)

513\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇

(122.)

514\. 𑀓𑀢𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(123.)

515\. 𑀓𑀢𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(124.)

516\. 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(125.)

517\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(126.)

518\. 𑀓𑀢𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄—  𑀇𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(127.)

519\. 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(128.)

520\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺? 𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀯𑀺𑀘𑀸𑀭𑁄, 𑀧𑀻𑀢𑀺, 𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸—  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

(129.)

521\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺𑁇

(130.)

522\. 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁, 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁, 𑀲𑀢𑀺𑀩𑀮𑀁, 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁, 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁, 𑀳𑀺𑀭𑀺𑀩𑀮𑀁, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(131.)

523\. 𑀓𑀢𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀅𑀮𑁄𑀪𑁄, 𑀅𑀤𑁄𑀲𑁄, 𑀅𑀫𑁄𑀳𑁄—  𑀇𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

(132.)

524\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺𑁇

(133.)

525\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

(134.)

526\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺𑁇

(135.)

527\. 𑀓𑀢𑀫𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

(136.)

528\. 𑀓𑀢𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(137.)

529\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

(138.)

530\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

(139.)

531\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

(140.)

553\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

554\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀳𑀺𑀭𑀺𑀩𑀮𑀁 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀅𑀮𑁄𑀪𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑀺𑀭𑀻 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀓𑀸𑀬𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀘𑀺𑀢𑁆𑀢𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀓𑀸𑀬𑀮𑀳𑀼𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀮𑀳𑀼𑀢𑀸 𑀓𑀸𑀬𑀫𑀼𑀤𑀼𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀫𑀼𑀤𑀼𑀢𑀸 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀓𑀸𑀬𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀓𑀸𑀬𑀼𑀚𑀼𑀓𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀼𑀚𑀼𑀓𑀢𑀸 𑀲𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀲𑀫𑀣𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄𑁇

560\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀙 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀳𑁂𑀢𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸…𑀧𑁂… 𑁇

561\. 𑀓𑀢𑀫𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺? 𑀨𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀳𑀺𑀭𑀺𑀩𑀮𑀁 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀮𑀁 𑀅𑀮𑁄𑀪𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀳𑀺𑀭𑀻 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀓𑀸𑀬𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀘𑀺𑀢𑁆𑀢𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀓𑀸𑀬𑀮𑀳𑀼𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀮𑀳𑀼𑀢𑀸 𑀓𑀸𑀬𑀫𑀼𑀤𑀼𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀫𑀼𑀤𑀼𑀢𑀸 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀓𑀸𑀬𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀧𑀸𑀕𑀼𑀜𑁆𑀜𑀢𑀸 𑀓𑀸𑀬𑀼𑀚𑀼𑀓𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀼𑀚𑀼𑀓𑀢𑀸 𑀲𑀢𑀺 𑀲𑀫𑀣𑁄 𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄𑁇

564\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇
