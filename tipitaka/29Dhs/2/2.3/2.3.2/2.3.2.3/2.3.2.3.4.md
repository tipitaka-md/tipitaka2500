# 𑁨𑁇𑁩𑁇𑁨𑁇𑁩𑁇𑁪 𑀆𑀲𑀯𑀲𑀸𑀲𑀯𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀆𑀲𑀯𑀲𑀸𑀲𑀯𑀤𑀼𑀓

(1112.)

1803\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘? 𑀢𑁂𑀬𑁂𑀯 𑀆𑀲𑀯𑀸 𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘𑁇

(1113.)

1804\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀲𑀯𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸; 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸𑁇
