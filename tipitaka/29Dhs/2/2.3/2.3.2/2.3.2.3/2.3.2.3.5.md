# 𑁨𑁇𑁩𑁇𑁨𑁇𑁩𑁇𑁫 𑀆𑀲𑀯𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀆𑀲𑀯𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1114.)

1805\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘? 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁂𑀦 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀓𑀸𑀫𑀸𑀲𑀯𑁂𑀦 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀪𑀯𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁂𑀦 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀪𑀯𑀸𑀲𑀯𑁂𑀦 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁂𑀦 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁂𑀦 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁇

(1115.)

1806\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸𑁇
