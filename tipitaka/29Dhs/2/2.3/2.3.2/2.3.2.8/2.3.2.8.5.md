# 𑁨𑁇𑁩𑁇𑁨𑁇𑁮𑁇𑁫 𑀦𑀻𑀯𑀭𑀡𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀦𑀻𑀯𑀭𑀡𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1176.)

1871\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀻𑀯𑀭𑀡𑀸 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘? 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀻𑀯𑀭𑀡𑀸 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁇

(1177.)

1872\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑁇
