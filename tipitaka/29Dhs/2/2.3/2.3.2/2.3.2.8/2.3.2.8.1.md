# 𑁨𑁇𑁩𑁇𑁨𑁇𑁮𑁇𑁧 𑀦𑀻𑀯𑀭𑀡𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀦𑀻𑀯𑀭𑀡𑀤𑀼𑀓

(1158.)

1852\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀻𑀯𑀭𑀡𑀸? 𑀙 𑀦𑀻𑀯𑀭𑀡𑀸—  𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇

(1159.)

1853\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁? 𑀬𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑁄 𑀓𑀸𑀫𑀦𑀦𑁆𑀤𑀻 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀓𑀸𑀫𑀲𑀺𑀦𑁂𑀳𑁄 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀓𑀸𑀫𑀫𑀼𑀘𑁆𑀙𑀸 𑀓𑀸𑀫𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁𑁇

(1160.)

1854\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁? 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀅𑀘𑀭𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺; 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺; 𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀅𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀝𑁆𑀞𑀸𑀦𑁂 𑀯𑀸 𑀧𑀦 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀓𑁄𑀧𑁄 𑀧𑀓𑁄𑀧𑁄 𑀲𑀫𑁆𑀧𑀓𑁄𑀧𑁄 𑀤𑁄𑀲𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑀫𑁆𑀧𑀤𑁄𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑁄 𑀓𑁄𑀥𑁄 𑀓𑀼𑀚𑁆𑀛𑀦𑀸 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀢𑁆𑀢𑀁 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁𑁇

(1161.)

1855\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀣𑀺𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀤𑁆𑀥𑀁𑁇

(1162.)

1856\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀣𑀺𑀦𑀁? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀮𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀑𑀮𑀻𑀬𑀦𑀸 𑀲𑀮𑁆𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀦𑀁 𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀣𑀺𑀦𑀁 𑀣𑀺𑀬𑀦𑀸 𑀣𑀺𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀣𑀺𑀦𑀁𑁇

(1163.)

1857\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀺𑀤𑁆𑀥𑀁? 𑀬𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀮𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀑𑀦𑀸𑀳𑁄 𑀧𑀭𑀺𑀬𑁄𑀦𑀸𑀳𑁄 𑀅𑀦𑁆𑀢𑁄𑀲𑀫𑁄𑀭𑁄𑀥𑁄 𑀫𑀺𑀤𑁆𑀥𑀁 𑀲𑁄𑀧𑁆𑀧𑀁 𑀧𑀘𑀮𑀸𑀬𑀺𑀓𑀸 𑀲𑁄𑀧𑁆𑀧𑀁 𑀲𑀼𑀧𑀦𑀸 𑀲𑀼𑀧𑀺𑀢𑀢𑁆𑀢𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀫𑀺𑀤𑁆𑀥𑀁𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀣𑀺𑀦𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀫𑀺𑀤𑁆𑀥𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁𑁇

(1164.)

1858\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁𑁇

(1165.)

1859\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀯𑀽𑀧𑀲𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀪𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁𑁇

(1166.)

1860\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁? 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀅𑀯𑀚𑁆𑀚𑁂 𑀯𑀚𑁆𑀚𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀯𑀚𑁆𑀚𑁂 𑀅𑀯𑀚𑁆𑀚𑀲𑀜𑁆𑀜𑀺𑀢𑀸𑁇 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 𑀫𑀦𑁄𑀯𑀺𑀮𑁂𑀔𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁𑁇

(1167.)

1861\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁? 𑀲𑀢𑁆𑀣𑀭𑀺 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀲𑀁𑀖𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺𑁇 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀓𑀗𑁆𑀔𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀦𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀤𑁆𑀯𑁂𑀍𑀳𑀓𑀁 𑀤𑁆𑀯𑁂𑀥𑀸𑀧𑀣𑁄 𑀲𑀁𑀲𑀬𑁄 𑀅𑀦𑁂𑀓𑀁𑀲𑀕𑁆𑀕𑀸𑀳𑁄 𑀆𑀲𑀧𑁆𑀧𑀦𑀸 𑀧𑀭𑀺𑀲𑀧𑁆𑀧𑀦𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑀦𑀸 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀦𑁄𑀯𑀺𑀮𑁂𑀔𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇

(1168.)

1862\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀜𑁆𑀜𑀸𑀡𑀁; 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑁄 𑀅𑀲𑀫𑁆𑀩𑁄𑀥𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀅𑀲𑀗𑁆𑀕𑀸𑀳𑀦𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑀦𑀸 𑀅𑀲𑀫𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀡𑀸 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀫𑁆𑀫𑁂𑀚𑁆𑀛𑀁 𑀩𑀸𑀮𑁆𑀬𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀫𑁄𑀳𑁄 𑀧𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀫𑁄𑀳𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑁄𑀖𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄𑀕𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇

1863\. 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀻𑀯𑀭𑀡𑀸𑁇

(1169.)

1864\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀦𑀻𑀯𑀭𑀡𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀦𑀻𑀯𑀭𑀡𑀸𑁇
