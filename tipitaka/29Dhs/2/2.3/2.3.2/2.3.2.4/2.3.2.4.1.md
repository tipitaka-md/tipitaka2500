# 𑁨𑁇𑁩𑁇𑁨𑁇𑁪𑁇𑁧 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀤𑀼𑀓

(1118.)

1810\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸? 𑀤𑀲 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸𑀦𑀺—  𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀫𑀸𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀪𑀯𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀇𑀲𑁆𑀲𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1119.)

1811\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀬𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑁄 𑀓𑀸𑀫𑀦𑀦𑁆𑀤𑀻 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀓𑀸𑀫𑀲𑀺𑀦𑁂𑀳𑁄 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀓𑀸𑀫𑀫𑀼𑀘𑁆𑀙𑀸 𑀓𑀸𑀫𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1120.)

1812\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀅𑀘𑀭𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀅𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀝𑁆𑀞𑀸𑀦𑁂 𑀯𑀸 𑀧𑀦 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀓𑁄𑀧𑁄 𑀧𑀓𑁄𑀧𑁄 𑀲𑀫𑁆𑀧𑀓𑁄𑀧𑁄 𑀤𑁄𑀲𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑀫𑁆𑀧𑀤𑁄𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑁄 𑀓𑁄𑀥𑁄 𑀓𑀼𑀚𑁆𑀛𑀦𑀸 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀢𑁆𑀢𑀁 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1121.)

1813\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀸𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄, 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄, 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀫𑀸𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1122.)

1814\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸; 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀦𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀕𑀸𑀳𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀳𑁄 𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀣𑁄 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀁 𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1123.)

1815\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀲𑀢𑁆𑀣𑀭𑀺 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀲𑀁𑀖𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺; 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀓𑀗𑁆𑀔𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀦𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀤𑁆𑀯𑁂𑀍𑀳𑀓𑀁 𑀤𑁆𑀯𑁂𑀥𑀸𑀧𑀣𑁄 𑀲𑀁𑀲𑀬𑁄 𑀅𑀦𑁂𑀓𑀁𑀲𑀕𑁆𑀕𑀸𑀳𑁄 𑀆𑀲𑀧𑁆𑀧𑀦𑀸 𑀧𑀭𑀺𑀲𑀧𑁆𑀧𑀦𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑀦𑀸 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀦𑁄𑀯𑀺𑀮𑁂𑀔𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1124.)

1816\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀲𑀻𑀮𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀺 𑀯𑀢𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀺 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀻𑀢𑀺; 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀦𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀕𑀸𑀳𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀳𑁄 𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀣𑁄 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀁 𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1125.)

1817\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀪𑀯𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀬𑁄 𑀪𑀯𑁂𑀲𑀼 𑀪𑀯𑀙𑀦𑁆𑀤𑁄 𑀪𑀯𑀭𑀸𑀕𑁄 𑀪𑀯𑀦𑀦𑁆𑀤𑀻 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀪𑀯𑀲𑀺𑀦𑁂𑀳𑁄 𑀪𑀯𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀪𑀯𑀫𑀼𑀘𑁆𑀙𑀸 𑀪𑀯𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀪𑀯𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1126.)

1818\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀇𑀲𑁆𑀲𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀬𑀸 𑀧𑀭𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀕𑀭𑀼𑀓𑀸𑀭𑀫𑀸𑀦𑀦𑀯𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀸𑀲𑀼 𑀇𑀲𑁆𑀲𑀸 𑀇𑀲𑁆𑀲𑀸𑀬𑀦𑀸 𑀇𑀲𑁆𑀲𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀲𑀽𑀬𑀸 𑀉𑀲𑀽𑀬𑀦𑀸 𑀉𑀲𑀽𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀇𑀲𑁆𑀲𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1127.)

1819\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀧𑀜𑁆𑀘 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀸𑀦𑀺—  𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀫𑀘𑁆𑀙𑁂𑀭𑀁 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀦𑀸 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑁂𑀯𑀺𑀘𑁆𑀙𑀁 𑀓𑀤𑀭𑀺𑀬𑀁 𑀓𑀝𑀼𑀓𑀜𑁆𑀘𑀼𑀓𑀢𑀸 𑀅𑀕𑁆𑀕𑀳𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

(1128.)

1820\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀜𑁆𑀜𑀸𑀡𑀁; 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑁄 𑀅𑀲𑀫𑁆𑀩𑁄𑀥𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀅𑀲𑀗𑁆𑀕𑀸𑀳𑀦𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑀦𑀸 𑀅𑀲𑀫𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀡𑀸 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀫𑁆𑀫𑁂𑀚𑁆𑀛𑀁 𑀩𑀸𑀮𑁆𑀬𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀫𑁄𑀳𑁄 𑀧𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀫𑁄𑀳𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑁄𑀖𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄𑀕𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇

1821\. 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸𑁇

(1129.)

1822\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸𑁇
