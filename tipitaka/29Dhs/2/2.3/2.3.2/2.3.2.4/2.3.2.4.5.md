# 𑁨𑁇𑁩𑁇𑁨𑁇𑁪𑁇𑁫 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1136.)

1829. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘? 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀫𑀸𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀫𑀸𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀪𑀯𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀪𑀯𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀇𑀲𑁆𑀲𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀇𑀲𑁆𑀲𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂𑀦 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁇

(1137.)

1830. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸𑁇
