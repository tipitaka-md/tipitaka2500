# 𑁨𑁇𑁩𑁇𑁨𑁇𑁪𑁇𑁬 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀤𑀼𑀓

(1138.)

1831\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸; 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸𑁇

(1139.)

1832\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸? 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀫𑀕𑁆𑀕𑀸 𑀘, 𑀫𑀕𑁆𑀕𑀨𑀮𑀸𑀦𑀺 𑀘, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀺𑀬𑀸𑁇
