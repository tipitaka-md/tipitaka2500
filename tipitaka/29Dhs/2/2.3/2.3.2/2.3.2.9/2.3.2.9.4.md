# 𑁨𑁇𑁩𑁇𑁨𑁇𑁯𑁇𑁪 𑀧𑀭𑀸𑀫𑀸𑀲𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀤𑀼𑀓

(1187.)

1883\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘? 𑀲𑁆𑀯𑁂𑀯 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁇

(1188.)

1884\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸; 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇
