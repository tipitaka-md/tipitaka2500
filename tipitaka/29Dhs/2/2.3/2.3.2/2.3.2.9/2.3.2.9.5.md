# 𑁨𑁇𑁩𑁇𑁨𑁇𑁯𑁇𑁫 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀤𑀼𑀓

(1189.)

1885. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸; 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇

(1190.)

1886. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸? 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀫𑀕𑁆𑀕𑀸 𑀘, 𑀫𑀕𑁆𑀕𑀨𑀮𑀸𑀦𑀺 𑀘, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇
