# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁧𑁇𑁬 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀤𑀼𑀓

(1233.)

1930. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸; 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇

(1234.)

1931. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸? 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀫𑀕𑁆𑀕𑀸 𑀘, 𑀫𑀕𑁆𑀕𑀨𑀮𑀸𑀦𑀺 𑀘, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇

---

1932. 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟𑁂 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀡𑀯𑀸𑀭𑁄𑁇
