# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁨𑁇𑁬 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀤𑀼𑀓

(1255.)

1954\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘? 𑀢𑁂𑀯 𑀓𑀺𑀮𑁂𑀲𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁇

(1256.)

1955\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑁇
