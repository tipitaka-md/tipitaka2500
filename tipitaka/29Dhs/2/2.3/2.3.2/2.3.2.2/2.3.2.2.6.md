# 𑁨𑁇𑁩𑁇𑁨𑁇𑁨𑁇𑁬 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

* 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

(1099.)

1768\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀺𑀬𑀸? 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇

(1100.)

1769\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸? 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀫𑀕𑁆𑀕𑀸 𑀘, 𑀫𑀕𑁆𑀕𑀨𑀮𑀸𑀦𑀺 𑀘, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇
