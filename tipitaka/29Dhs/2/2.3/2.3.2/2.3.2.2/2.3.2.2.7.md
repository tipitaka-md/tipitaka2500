# 𑁨𑁇𑁩𑁇𑁨𑁇𑁨𑁇𑁭 𑀓𑁂𑀦𑀘𑀺𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀤𑀼𑀓

### 𑀓𑁂𑀦𑀘𑀺𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀤𑀼𑀓

(1101.)

1770. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?

1771. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1772. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1773. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1774. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1775. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1776. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1777. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1778. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1779. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1780. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1781. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1782. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1783. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1784. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1785. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1786. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1787. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1788. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1789. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇

1790. 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸; 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇
