# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁦𑁇𑁩 𑀘𑁂𑀢𑀲𑀺𑀓𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

### 𑀘𑁂𑀢𑀲𑀺𑀓𑀤𑀼𑀓

(1195.)

1891\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀸? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑁇

(1196.)

1892\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸? 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑁇
