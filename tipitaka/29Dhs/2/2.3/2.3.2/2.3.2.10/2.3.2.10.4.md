# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁦𑁇𑁪 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

* 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1197.)

1893\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(1198.)

1894\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀧𑀺, 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀧𑀺𑁇
