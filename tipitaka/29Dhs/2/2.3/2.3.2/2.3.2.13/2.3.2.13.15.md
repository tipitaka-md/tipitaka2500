# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁩𑁇𑁧𑁫 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀤𑀼𑀓

(1295.)

1996. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑁇

(1296.)

1997. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑁇
