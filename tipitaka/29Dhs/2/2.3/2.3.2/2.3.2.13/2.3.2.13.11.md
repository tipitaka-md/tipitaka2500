# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁩𑁇𑁧𑁧 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀤𑀼𑀓

(1287.)

1988\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸? 𑀳𑁂𑀝𑁆𑀞𑀢𑁄 𑀅𑀯𑀺𑀘𑀺𑀦𑀺𑀭𑀬𑀁 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀉𑀧𑀭𑀺𑀢𑁄 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀤𑁂𑀯𑁂 𑀅𑀦𑁆𑀢𑁄 𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀬𑀁 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑁂 𑀏𑀢𑁆𑀣𑀸𑀯𑀘𑀭𑀸 𑀏𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀔𑀦𑁆𑀥𑀥𑀸𑀢𑀼 𑀆𑀬𑀢𑀦𑀸, 𑀭𑀽𑀧𑀁 𑀯𑁂𑀤𑀦𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇

(1288.)

1989\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇
