# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁩𑁇𑁮 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀤𑀼𑀓

(1281.)

1982\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸? 𑀧𑀻𑀢𑀺𑀪𑀽𑀫𑀺𑀬𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂, 𑀧𑀻𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀢𑀁𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇

(1282.)

1983\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸? 𑀦 𑀧𑀻𑀢𑀺𑀪𑀽𑀫𑀺𑀬𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀧𑀻𑀢𑀺 𑀘, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇
