# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁩𑁇𑁧𑁮 𑀲𑀭𑀡𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀲𑀭𑀡𑀤𑀼𑀓

(1301.)

2002\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀭𑀡𑀸? 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀮𑁄𑀪𑁄, 𑀤𑁄𑀲𑁄, 𑀫𑁄𑀳𑁄; 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀢𑀁𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀢𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁, 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁, 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀭𑀡𑀸𑁇

(1302.)

2003\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀭𑀡𑀸? 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀭𑀡𑀸𑁇

---

2004\. 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀤𑀼𑀓𑀁𑁇
