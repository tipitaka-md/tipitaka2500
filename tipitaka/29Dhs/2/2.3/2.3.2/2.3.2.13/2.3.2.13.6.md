# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁩𑁇𑁬 𑀲𑀯𑀺𑀘𑀸𑀭𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀲𑀯𑀺𑀘𑀸𑀭𑀤𑀼𑀓

(1277.)

1978\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸? 𑀲𑀯𑀺𑀘𑀸𑀭𑀪𑀽𑀫𑀺𑀬𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂, 𑀯𑀺𑀘𑀸𑀭𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀢𑀁𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸𑁇

(1278.)

1979\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸? 𑀅𑀯𑀺𑀘𑀸𑀭𑀪𑀽𑀫𑀺𑀬𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀯𑀺𑀘𑀸𑀭𑁄 𑀘, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇
