# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁇𑁧 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

(1059.)

1725\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽, 𑀦𑀯 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽 𑀙 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽, 𑀙 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽, 𑀙 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀳𑁂𑀢𑀽𑁇

(1060.)

1726\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽? 𑀅𑀮𑁄𑀪𑁄, 𑀅𑀤𑁄𑀲𑁄, 𑀅𑀫𑁄𑀳𑁄𑁇

(1061.)

1727\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀮𑁄𑀪𑁄? 𑀬𑁄 𑀅𑀮𑁄𑀪𑁄 𑀅𑀮𑀼𑀩𑁆𑀪𑀦𑀸 𑀅𑀮𑀼𑀩𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑀸 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀮𑁄𑀪𑁄𑁇

(1062.)

1728\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀤𑁄𑀲𑁄? 𑀬𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀅𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑁂𑀢𑁆𑀢𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀦𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀤𑀸 𑀅𑀦𑀼𑀤𑁆𑀤𑀸𑀬𑀦𑀸 𑀅𑀦𑀼𑀤𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑁂𑀲𑀺𑀢𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀤𑁄𑀲𑁄𑁇

(1063.)

1729\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀫𑁄𑀳𑁄? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀫𑁄𑀳𑁄𑁇

1730\. 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽𑁇

(1064.)

1731\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽? 𑀮𑁄𑀪𑁄, 𑀤𑁄𑀲𑁄, 𑀫𑁄𑀳𑁄𑁇

(1065.)

1732\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑁄𑀪𑁄? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀦𑀼𑀦𑀬𑁄 𑀅𑀦𑀼𑀭𑁄𑀥𑁄 𑀦𑀦𑁆𑀤𑀻 𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄 𑀇𑀘𑁆𑀙𑀸 𑀫𑀼𑀘𑁆𑀙𑀸 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁 𑀕𑁂𑀥𑁄 𑀧𑀮𑀺𑀕𑁂𑀥𑁄 𑀲𑀗𑁆𑀕𑁄 𑀧𑀗𑁆𑀓𑁄 𑀏𑀚𑀸 𑀫𑀸𑀬𑀸 𑀚𑀦𑀺𑀓𑀸 𑀲𑀜𑁆𑀚𑀦𑀦𑀻 𑀲𑀺𑀩𑁆𑀩𑀺𑀦𑀻 𑀚𑀸𑀮𑀺𑀦𑀻 𑀲𑀭𑀺𑀢𑀸 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀝𑀸 𑀆𑀬𑀽𑀳𑀺𑀦𑀻 𑀤𑀼𑀢𑀺𑀬𑀸 𑀧𑀡𑀺𑀥𑀺 𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀯𑀦𑀁 𑀯𑀦𑀣𑁄 𑀲𑀦𑁆𑀣𑀯𑁄 𑀲𑀺𑀦𑁂𑀳𑁄 𑀅𑀧𑁂𑀓𑁆𑀔𑀸 𑀧𑀝𑀺𑀩𑀦𑁆𑀥𑀼 𑀆𑀲𑀸 𑀆𑀲𑀺𑀲𑀦𑀸 𑀆𑀲𑀺𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀸𑀲𑀸 𑀲𑀤𑁆𑀤𑀸𑀲𑀸 𑀕𑀦𑁆𑀥𑀸𑀲𑀸 𑀭𑀲𑀸𑀲𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀲𑀸 𑀮𑀸𑀪𑀸𑀲𑀸 𑀥𑀦𑀸𑀲𑀸 𑀧𑀼𑀢𑁆𑀢𑀸𑀲𑀸 𑀚𑀻𑀯𑀺𑀢𑀸𑀲𑀸 𑀚𑀧𑁆𑀧𑀸 𑀧𑀚𑀧𑁆𑀧𑀸 𑀅𑀪𑀺𑀚𑀧𑁆𑀧𑀸 𑀚𑀧𑁆𑀧𑀸 𑀚𑀧𑁆𑀧𑀦𑀸 𑀚𑀧𑁆𑀧𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀁 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀸𑀬𑀦𑀸 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀜𑁆𑀚𑀺𑀓𑀢𑀸 𑀲𑀸𑀥𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀥𑀫𑁆𑀫𑀭𑀸𑀕𑁄 𑀯𑀺𑀲𑀫𑀮𑁄𑀪𑁄 𑀦𑀺𑀓𑀦𑁆𑀢𑀺 𑀦𑀺𑀓𑀸𑀫𑀦𑀸 𑀧𑀢𑁆𑀣𑀦𑀸 𑀧𑀺𑀳𑀦𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀣𑀦𑀸 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀭𑁄𑀥𑀢𑀡𑁆𑀳𑀸 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸 𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸 𑀭𑀲𑀢𑀡𑁆𑀳𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸 𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸 𑀑𑀖𑁄 𑀬𑁄𑀕𑁄 𑀕𑀦𑁆𑀣𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀆𑀯𑀭𑀡𑀁 𑀦𑀻𑀯𑀭𑀡𑀁 𑀙𑀸𑀤𑀦𑀁 𑀩𑀦𑁆𑀥𑀦𑀁 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀅𑀦𑀼𑀲𑀬𑁄 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀮𑀢𑀸 𑀯𑁂𑀯𑀺𑀘𑁆𑀙𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀪𑀯𑁄 𑀫𑀸𑀭𑀧𑀸𑀲𑁄 𑀫𑀸𑀭𑀩𑀍𑀇𑀲𑀁 𑀫𑀸𑀭𑀯𑀺𑀲𑀬𑁄 𑀢𑀡𑁆𑀳𑀸𑀦𑀤𑀻 𑀢𑀡𑁆𑀳𑀸𑀚𑀸𑀮𑀁 𑀢𑀡𑁆𑀳𑀸𑀕𑀤𑁆𑀤𑀼𑀮𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀮𑁄𑀪𑁄𑁇

(1066.)

1733\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑁄𑀲𑁄? 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀅𑀘𑀭𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀅𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀝𑁆𑀞𑀸𑀦𑁂 𑀯𑀸 𑀧𑀦 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀓𑁄𑀧𑁄 𑀧𑀓𑁄𑀧𑁄 𑀲𑀫𑁆𑀧𑀓𑁄𑀧𑁄 𑀤𑁄𑀲𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑀫𑁆𑀧𑀤𑁄𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑁄 𑀓𑁄𑀥𑁄 𑀓𑀼𑀚𑁆𑀛𑀦𑀸 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀢𑁆𑀢𑀁 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑁄𑀲𑁄𑁇

(1067.)

1734\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑁄𑀳𑁄? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑁄 𑀅𑀲𑀫𑁆𑀩𑁄𑀥𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀅𑀲𑀗𑁆𑀕𑀸𑀳𑀦𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑀦𑀸 𑀅𑀲𑀫𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀡𑀸 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀫𑁆𑀫𑁂𑀚𑁆𑀛𑀁 𑀩𑀸𑀮𑁆𑀬𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀫𑁄𑀳𑁄 𑀧𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀫𑁄𑀳𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑁄𑀖𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄𑀕𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀫𑁄𑀳𑁄𑁇

1735\. 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽𑁇

(1068.)

1736\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽? 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂𑀲𑀼 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀮𑁄𑀪𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽𑁇

(1069.)

1737\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀦𑀯 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽—  𑀇𑀫𑁂 𑀦𑀯 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽𑁇

(1070.)

1738\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀙 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽—  𑀇𑀫𑁂 𑀙 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽𑁇

(1071.)

1739\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀙 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽—  𑀇𑀫𑁂 𑀙 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀳𑁂𑀢𑀽𑁇

(1072.)

1740\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀙 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽—  𑀇𑀫𑁂 𑀙 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀳𑁂𑀢𑀽𑁇

(1073.)

1741\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽? 𑀅𑀮𑁄𑀪𑁄, 𑀅𑀤𑁄𑀲𑁄, 𑀅𑀫𑁄𑀳𑁄𑁇

(1074.)

1742\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀮𑁄𑀪𑁄? 𑀬𑁄 𑀅𑀮𑁄𑀪𑁄 𑀅𑀮𑀼𑀩𑁆𑀪𑀦𑀸 𑀅𑀮𑀼𑀩𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑀸 𑀅𑀲𑀸𑀭𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀮𑁄𑀪𑁄𑁇

(1075.)

1743\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀤𑁄𑀲𑁄? 𑀬𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀅𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀤𑁄𑀲𑁄𑁇

(1076.)

1744\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀫𑁄𑀳𑁄? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻 𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀫𑁄𑀳𑁄𑁇

1745\. 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽𑁇

(1077.)

1746\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽? 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀅𑀮𑁄𑀪𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽𑁇 𑀇𑀫𑁂 𑀙 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀳𑁂𑀢𑀽—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽𑁇

(1078.)

1747\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀳𑁂𑀢𑀽? 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀳𑁂𑀢𑀽𑁇
