# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁇𑁫 𑀳𑁂𑀢𑀼𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀳𑁂𑀢𑀼𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(1085.)

1754\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘? 𑀮𑁄𑀪𑁄 𑀫𑁄𑀳𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀫𑁄𑀳𑁄 𑀮𑁄𑀪𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀤𑁄𑀲𑁄 𑀫𑁄𑀳𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀫𑁄𑀳𑁄 𑀤𑁄𑀲𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘; 𑀅𑀮𑁄𑀪𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄, 𑀢𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁇

(1086.)

1755\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑁇
