### 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

(1083.)

1752\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘? 𑀮𑁄𑀪𑁄 𑀫𑁄𑀳𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘, 𑀫𑁄𑀳𑁄 𑀮𑁄𑀪𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘, 𑀤𑁄𑀲𑁄 𑀫𑁄𑀳𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘, 𑀫𑁄𑀳𑁄 𑀤𑁄𑀲𑁂𑀦 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘; 𑀅𑀮𑁄𑀪𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄, 𑀢𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁇

(1084.)

1753\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽? 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑁇
