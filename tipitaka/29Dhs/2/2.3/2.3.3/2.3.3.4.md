# 𑁨𑁇𑁩𑁇𑁩𑁇𑁪 𑀓𑀡𑁆𑀳𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀓𑀡𑁆𑀳𑀤𑀼𑀓

(1309.)

2011\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀡𑁆𑀳𑀸? 𑀅𑀳𑀺𑀭𑀻𑀓𑀜𑁆𑀘 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀡𑁆𑀳𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀡𑁆𑀳𑀸𑁇

(1310.)

2012\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀓𑁆𑀓𑀸? 𑀳𑀺𑀭𑀻 𑀘 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀓𑁆𑀓𑀸? 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀓𑁆𑀓𑀸𑁇
