# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩𑁦 𑀲𑀢𑀺𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀲𑀢𑀺𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀤𑀼𑀓

(1358.)

2063\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑀺 𑀲𑀭𑀡𑀢𑀸 𑀥𑀸𑀭𑀡𑀢𑀸 𑀅𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀅𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸 𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀢𑀺𑁇

(1359.)

2064\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁𑁇
