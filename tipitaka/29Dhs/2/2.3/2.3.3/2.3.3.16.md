# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧𑁬 𑀳𑀺𑀭𑀺𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀳𑀺𑀭𑀺𑀤𑀼𑀓

(1330.)

2035\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀳𑀺𑀭𑀻? 𑀬𑀁 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀭𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑁂𑀦, 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀳𑀺𑀭𑀻𑁇

(1331.)

2036\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁? 𑀬𑀁 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑁂𑀦, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁𑁇
