# 𑁨𑁇𑁩𑁇𑁩𑁇𑁪𑁧 𑀯𑀺𑀚𑁆𑀚𑀸𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀯𑀺𑀚𑁆𑀚𑀸𑀤𑀼𑀓

(1380.)

2085\. **𑀯𑀺𑀚𑁆𑀚𑀸**𑀢𑀺—  𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸—  𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀜𑀸𑀡𑀁 𑀯𑀺𑀚𑁆𑀚𑀸, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁂 𑀜𑀸𑀡𑀁 𑀯𑀺𑀚𑁆𑀚𑀸, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁂 𑀜𑀸𑀡𑀁 𑀯𑀺𑀚𑁆𑀚𑀸𑁇

(1381.)

2086\. **𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀻**𑀢𑀺—  𑀤𑁆𑀯𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘𑁇
