# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩 𑀩𑀸𑀮𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀩𑀸𑀮𑀤𑀼𑀓

(1307.)

2009\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀩𑀸𑀮𑀸? 𑀅𑀳𑀺𑀭𑀻𑀓𑀜𑁆𑀘 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀩𑀸𑀮𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀩𑀸𑀮𑀸𑁇

(1308.)

2010\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸? 𑀳𑀺𑀭𑀻 𑀘 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇
