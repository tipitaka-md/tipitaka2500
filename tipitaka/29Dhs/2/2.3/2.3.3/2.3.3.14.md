# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧𑁪 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

(1326.)

2031. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺? 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑁇

(1327.)

2032. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑀭𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺? 𑀅𑀧𑀭𑀦𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀧𑀭𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑁇
