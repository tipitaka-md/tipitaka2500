# 𑁨𑁇𑁩𑁇𑁩𑁇𑁫 𑀢𑀧𑀦𑀻𑀬𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀢𑀧𑀦𑀻𑀬𑀤𑀼𑀓

(1311.)

2013\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀢𑀧𑀦𑀻𑀬𑀸? 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀢𑀧𑀦𑀻𑀬𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀢𑀧𑀦𑀻𑀬𑀸𑁇

(1312.)

2014\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀢𑀧𑀦𑀻𑀬𑀸? 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑀁, 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑀁, 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀢𑀧𑀦𑀻𑀬𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀢𑀧𑀦𑀻𑀬𑀸𑁇
