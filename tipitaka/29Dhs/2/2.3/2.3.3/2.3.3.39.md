# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩𑁯 𑀲𑀁𑀯𑁂𑀕𑀲𑀁𑀯𑁂𑀚𑀦𑀦𑀺𑀬𑀝𑁆𑀞𑀸𑀦𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀲𑀁𑀯𑁂𑀕𑀲𑀁𑀯𑁂𑀚𑀦𑀦𑀺𑀬𑀝𑁆𑀞𑀸𑀦𑀤𑀼𑀓

(1376.)

2081\. **𑀲𑀁𑀯𑁂𑀕𑁄**𑀢𑀺—  𑀚𑀸𑀢𑀺𑀪𑀬𑀁 𑀚𑀭𑀸𑀪𑀬𑀁 𑀩𑁆𑀬𑀸𑀥𑀺𑀪𑀬𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇 𑀲𑀁𑀯𑁂𑀚𑀦𑀺𑀬𑀁 𑀞𑀸𑀦𑀦𑁆𑀢𑀺—  𑀚𑀸𑀢𑀺 𑀚𑀭𑀸 𑀩𑁆𑀬𑀸𑀥𑀺 𑀫𑀭𑀡𑀁𑁇

(1377.)

2082\. **𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀲𑁆𑀲 𑀘 𑀬𑁄𑀦𑀺𑀲𑁄 𑀧𑀥𑀸𑀦𑀦𑁆**𑀢𑀺—  𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇
