# 𑁨𑁇𑁩𑁇𑁩𑁇𑁨𑁫 𑀔𑀦𑁆𑀢𑀺𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀔𑀦𑁆𑀢𑀺𑀤𑀼𑀓

(1348.)

2053. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀔𑀦𑁆𑀢𑀺? 𑀬𑀸 𑀔𑀦𑁆𑀢𑀺 𑀔𑀫𑀦𑀢𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀢𑀸 𑀅𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀦𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀔𑀦𑁆𑀢𑀺𑁇

(1349.)

2054. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑁄𑀭𑀘𑁆𑀘𑀁? 𑀬𑁄 𑀓𑀸𑀬𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑁄𑀭𑀘𑁆𑀘𑀁𑁇 𑀲𑀩𑁆𑀩𑁄𑀧𑀺 𑀲𑀻𑀮𑀲𑀁𑀯𑀭𑁄 𑀲𑁄𑀭𑀘𑁆𑀘𑀁𑁇
