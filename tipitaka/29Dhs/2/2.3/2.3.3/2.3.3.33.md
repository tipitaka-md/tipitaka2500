# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩𑁩 𑀲𑀫𑀣𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀲𑀫𑀣𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀤𑀼𑀓

(1364.)

2069. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑀣𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀫𑀣𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁𑁇

(1365.)

2070. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀕𑁆𑀕𑀸𑀳𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀕𑁆𑀕𑀸𑀳𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁𑁇
