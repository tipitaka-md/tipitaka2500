# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩𑁧 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀩𑀮𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀩𑀮𑀤𑀼𑀓

(1360.)

2065\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀩𑀮𑀁? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀩𑀮𑀁𑁇

(1361.)

2066\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀪𑀸𑀯𑀦𑀸𑀩𑀮𑀁? 𑀬𑀸 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀆𑀲𑁂𑀯𑀦𑀸 𑀪𑀸𑀯𑀦𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀫𑁆𑀫𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀪𑀸𑀯𑀦𑀸𑀩𑀮𑀁𑁇 𑀲𑀢𑁆𑀢𑀧𑀺 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀦𑀸𑀩𑀮𑀁𑁇
