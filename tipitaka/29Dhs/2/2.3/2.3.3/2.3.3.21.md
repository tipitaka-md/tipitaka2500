# 𑁨𑁇𑁩𑁇𑁩𑁇𑁨𑁧 𑀥𑀸𑀢𑀼𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀥𑀸𑀢𑀼𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

(1340.)

2045\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀥𑀸𑀢𑀼𑀓𑀼𑀲𑀮𑀢𑀸? 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼 𑀭𑀲𑀥𑀸𑀢𑀼 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀥𑀸𑀢𑀼 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇 𑀬𑀸 𑀢𑀸𑀲𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀥𑀸𑀢𑀼𑀓𑀼𑀲𑀮𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀥𑀸𑀢𑀼𑀓𑀼𑀲𑀮𑀢𑀸𑁇

(1341.)

2046\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀓𑀼𑀲𑀮𑀢𑀸? 𑀬𑀸 𑀢𑀸𑀲𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀓𑀼𑀲𑀮𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀓𑀼𑀲𑀮𑀢𑀸𑁇
