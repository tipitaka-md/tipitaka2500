# 𑁨𑁇𑁩𑁇𑁩𑁇𑁨𑁦 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

(1338.)

2043\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸? 𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀬𑀸 𑀢𑀸𑀲𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸𑁇

(1339.)

2044\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸? 𑀬𑀸 𑀢𑀸𑀳𑀺 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸𑁇
