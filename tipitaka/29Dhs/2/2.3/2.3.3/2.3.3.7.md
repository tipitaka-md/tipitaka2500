# 𑁨𑁇𑁩𑁇𑁩𑁇𑁭 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀤𑀼𑀓

(1314.)

2017\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑁄𑀳𑀸𑀭𑁄 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀓𑀫𑁆𑀫𑀁 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀅𑀪𑀺𑀮𑀸𑀧𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑁇

2018\. 𑀲𑀩𑁆𑀩𑁂𑀯 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀣𑀸𑁇
