# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧𑁨 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

(1322.)

2027\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺? 𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸𑀢𑀺, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑁇

(1323.)

2028\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺? 𑀉𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸𑀢𑀺, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺𑁇
