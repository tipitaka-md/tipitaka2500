# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀻𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀻𑀤𑀼𑀓

(1303.)

2005\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀦𑁄? 𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀦𑁄𑁇

(1304.)

2006\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀦𑁄? 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀦𑁄𑁇
