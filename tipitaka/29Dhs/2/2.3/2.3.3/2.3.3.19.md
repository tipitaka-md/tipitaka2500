# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧𑁯 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

(1336.)

2041\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸? 𑀧𑀜𑁆𑀘𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄, 𑀲𑀢𑁆𑀢𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀬𑀸 𑀢𑀸𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸𑁇

(1337.)

2042\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸? 𑀬𑀸 𑀢𑀸𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸𑁇
