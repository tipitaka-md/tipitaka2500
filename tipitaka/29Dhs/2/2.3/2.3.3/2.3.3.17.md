# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧𑁭 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸𑀤𑀼𑀓

(1332.)

2037. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸? 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁂 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀸𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀺𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀕𑁆𑀕𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀲𑀸𑀢𑀢𑀸 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀅𑀦𑀸𑀤𑀭𑀢𑀸 𑀅𑀕𑀸𑀭𑀯𑀢𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑁆𑀲𑀯𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸𑁇

(1333.)

2038. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸? 𑀬𑁂 𑀢𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀸 𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸, 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀲𑁂𑀯𑀦𑀸 𑀦𑀺𑀲𑁂𑀯𑀦𑀸 𑀲𑀁𑀲𑁂𑀯𑀦𑀸 𑀪𑀚𑀦𑀸 𑀲𑀫𑁆𑀪𑀚𑀦𑀸 𑀪𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀺 𑀢𑀁𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑁇
