# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩𑁪 𑀧𑀕𑁆𑀕𑀸𑀳𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀧𑀕𑁆𑀕𑀸𑀳𑀤𑀼𑀓

(1366.)

2071\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀕𑁆𑀕𑀸𑀳𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀕𑁆𑀕𑀸𑀳𑁄𑁇

(1367.)

2072\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄𑁇
