# 𑁨𑁇𑁩𑁇𑁩𑁇𑁪𑁦 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀓𑀼𑀲𑀮𑀥𑀫𑁆𑀫𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀓𑀼𑀲𑀮𑀥𑀫𑁆𑀫𑀤𑀼𑀓

(1378.)

2083\. **𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀘 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀽**𑀢𑀺—  𑀬𑀸 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀸𑁇

(1379.)

2084\. **𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀺𑀢𑀸 𑀘 𑀧𑀥𑀸𑀦𑀲𑁆𑀫𑀺𑀦𑁆**𑀢𑀺—  𑀬𑀸 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀝𑁆𑀞𑀺𑀢𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀦𑁄𑀮𑀻𑀦𑀯𑀼𑀢𑁆𑀢𑀺𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀆𑀲𑁂𑀯𑀦𑀸 𑀪𑀸𑀯𑀦𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀫𑁆𑀫𑀁𑁇
