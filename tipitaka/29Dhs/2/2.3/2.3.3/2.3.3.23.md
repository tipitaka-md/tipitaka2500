# 𑁨𑁇𑁩𑁇𑁩𑁇𑁨𑁩 𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸𑀤𑀼𑀓

(1344.)

2049\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸? 𑀬𑁂 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀬𑁂𑀲𑀁 𑀬𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀳𑁂𑀢𑀽 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀢𑀁 𑀢𑀁 𑀞𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸𑁇

(1345.)

2050\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸? 𑀬𑁂 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀬𑁂𑀲𑀁 𑀬𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀦 𑀳𑁂𑀢𑀽 𑀦 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀢𑀁 𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸𑁇
