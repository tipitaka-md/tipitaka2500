# 𑁨𑁇𑁩𑁇𑁩𑁇𑁨𑁪 𑀅𑀚𑁆𑀚𑀯𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀅𑀚𑁆𑀚𑀯𑀤𑀼𑀓

(1346.)

2051\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀚𑁆𑀚𑀯𑁄? 𑀬𑀸 𑀅𑀚𑁆𑀚𑀯𑀢𑀸 𑀅𑀚𑀺𑀫𑁆𑀳𑀢𑀸 𑀅𑀯𑀗𑁆𑀓𑀢𑀸 𑀅𑀓𑀼𑀝𑀺𑀮𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀚𑁆𑀚𑀯𑁄𑁇

(1347.)

2052\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑀤𑁆𑀤𑀯𑁄? 𑀬𑀸 𑀫𑀼𑀤𑀼𑀢𑀸 𑀫𑀤𑁆𑀤𑀯𑀢𑀸 𑀅𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑀸 𑀅𑀓𑀣𑀺𑀦𑀢𑀸 𑀦𑀻𑀘𑀘𑀺𑀢𑁆𑀢𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀫𑀤𑁆𑀤𑀯𑁄𑁇
