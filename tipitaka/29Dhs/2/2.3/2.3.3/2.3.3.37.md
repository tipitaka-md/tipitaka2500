# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩𑁭 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀤𑀼𑀓

(1372.)

2077\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺? 𑀬𑁄 𑀓𑀸𑀬𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑁇 𑀲𑀩𑁆𑀩𑁄𑀧𑀺 𑀲𑀻𑀮𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑁇

(1373.)

2078\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺? 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀓𑀢𑀜𑀸𑀡𑀁 𑀲𑀘𑁆𑀘𑀸𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀜𑀸𑀡𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀁 𑀨𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀁𑁇
