# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧𑁩 𑀅𑀦𑁆𑀢𑀯𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀅𑀦𑁆𑀢𑀯𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

(1324.)

2029\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀦𑁆𑀢𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀺? 𑀅𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸𑀢𑀺, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀦𑁆𑀢𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑁇

(1325.)

2030\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀺? 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸𑀢𑀺, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑁇
