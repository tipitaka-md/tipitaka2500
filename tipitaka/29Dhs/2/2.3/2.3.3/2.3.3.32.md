* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀤𑀼𑀓

(1362.)

2067\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑀣𑁄? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀫𑀣𑁄𑁇

(1363.)

2068\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑁇
