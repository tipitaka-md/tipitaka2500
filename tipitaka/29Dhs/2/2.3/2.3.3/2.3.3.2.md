# 𑁨𑁇𑁩𑁇𑁩𑁇𑁨 𑀯𑀺𑀚𑁆𑀚𑀽𑀧𑀫𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀯𑀺𑀚𑁆𑀚𑀽𑀧𑀫𑀤𑀼𑀓

(1305.)

2007. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀚𑁆𑀚𑀽𑀧𑀫𑀸? 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑁂𑀲𑀼 𑀢𑀻𑀲𑀼 𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀚𑁆𑀚𑀽𑀧𑀫𑀸𑁇

(1306.)

2008. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀚𑀺𑀭𑀽𑀧𑀫𑀸? 𑀉𑀧𑀭𑀺𑀝𑁆𑀞𑀺𑀫𑁂 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑁂 𑀧𑀜𑁆𑀜𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀚𑀺𑀭𑀽𑀧𑀫𑀸𑁇
