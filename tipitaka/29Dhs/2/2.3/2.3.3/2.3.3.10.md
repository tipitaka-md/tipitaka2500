# 𑁨𑁇𑁩𑁇𑁩𑁇𑁧𑁦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀤𑀼𑀓

(1318.)

2023\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑁇

(1319.)

2024\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸? 𑀬𑁄 𑀪𑀯𑁂𑀲𑀼 𑀪𑀯𑀙𑀦𑁆𑀤𑁄…𑀧𑁂…  𑀪𑀯𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑁇
