# 𑁨𑁇𑁩𑁇𑁩𑁇𑁨𑁬 𑀲𑀸𑀔𑀮𑁆𑀬𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀤𑀼𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀲𑀸𑀔𑀮𑁆𑀬𑀤𑀼𑀓

(1350.)

2055. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀸𑀔𑀮𑁆𑀬𑀁? 𑀬𑀸 𑀲𑀸 𑀯𑀸𑀘𑀸 𑀅𑀡𑁆𑀟𑀓𑀸 𑀓𑀓𑁆𑀓𑀲𑀸 𑀧𑀭𑀓𑀝𑀼𑀓𑀸 𑀧𑀭𑀸𑀪𑀺𑀲𑀚𑁆𑀚𑀦𑀻 𑀓𑁄𑀥𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀅𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸, 𑀢𑀣𑀸𑀭𑀽𑀧𑀺𑀁 𑀯𑀸𑀘𑀁 𑀧𑀳𑀸𑀬 𑀬𑀸 𑀲𑀸 𑀯𑀸𑀘𑀸 𑀦𑁂𑀍𑀆 𑀓𑀡𑁆𑀡𑀲𑀼𑀔𑀸 𑀧𑁂𑀫𑀦𑀻𑀬𑀸 𑀳𑀤𑀬𑀗𑁆𑀕𑀫𑀸 𑀧𑁄𑀭𑀻 𑀩𑀳𑀼𑀚𑀦𑀓𑀦𑁆𑀢𑀸 𑀩𑀳𑀼𑀚𑀦𑀫𑀦𑀸𑀧𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀺𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀲𑀡𑁆𑀳𑀯𑀸𑀘𑀢𑀸 𑀲𑀔𑀺𑀮𑀯𑀸𑀘𑀢𑀸 𑀅𑀨𑀭𑀼𑀲𑀯𑀸𑀘𑀢𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀸𑀔𑀮𑁆𑀬𑀁𑁇

(1351.)

2056. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄? 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀸—  𑀆𑀫𑀺𑀲𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄 𑀘 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄 𑀘𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀆𑀫𑀺𑀲𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄𑁇
