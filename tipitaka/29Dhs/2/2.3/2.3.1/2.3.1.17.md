# 𑁨𑁇𑁩𑁇𑁧𑁇𑁧𑁭 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀢𑀺𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓

(1041.)

1706\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸? 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀸𑀢𑀸 𑀪𑀽𑀢𑀸 𑀲𑀜𑁆𑀚𑀸𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀉𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀭𑀽𑀧𑀁, 𑀯𑁂𑀤𑀦𑀸, 𑀲𑀜𑁆𑀜𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇

(1042.)

1707\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸? 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀚𑀸𑀢𑀸 𑀅𑀪𑀽𑀢𑀸 𑀅𑀲𑀜𑁆𑀚𑀸𑀢𑀸 𑀅𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀅𑀦𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀅𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀭𑀽𑀧𑀁, 𑀯𑁂𑀤𑀦𑀸, 𑀲𑀜𑁆𑀜𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇

(1043.)

1708\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄? 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀯𑀺𑀧𑀓𑁆𑀓𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄𑁇
