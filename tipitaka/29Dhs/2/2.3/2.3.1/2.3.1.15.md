# 𑁨𑁇𑁩𑁇𑁧𑁇𑁧𑁫 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀢𑁆𑀢𑀺𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

* 𑀢𑀺𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀢𑁆𑀢𑀺𑀓

(1035.)

1700\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸? 𑀧𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀆𑀦𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀦𑀺, 𑀬𑀸 𑀘 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀬𑀢𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸𑁇

(1036.)

1701\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸𑁇

(1037.)

1702\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀺𑀬𑀢𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇
