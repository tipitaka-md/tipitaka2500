# 𑁨𑁇𑁩𑁇𑁧𑁇𑁧𑁩 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀢𑀺𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

(1029.)

1694\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸? 𑀧𑀭𑀺𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇

(1030.)

1695\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸? 𑀫𑀳𑀕𑁆𑀕𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇

(1031.)

1696\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸? 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇
