# 𑁨𑁇𑁩𑁇𑁧𑁇𑁧𑁧 𑀲𑁂𑀓𑁆𑀔𑀢𑁆𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀡𑁆𑀟

### 𑀢𑀺𑀓𑀦𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀲𑁂𑀓𑁆𑀔𑀢𑁆𑀢𑀺𑀓

(1023.)

1688\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀓𑁆𑀔𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺 𑀘 𑀢𑀻𑀡𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀓𑁆𑀔𑀸𑁇

(1024.)

1689\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑀸? 𑀉𑀧𑀭𑀺𑀝𑁆𑀞𑀺𑀫𑀁 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑀸𑁇

(1025.)

1690\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸; 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇
