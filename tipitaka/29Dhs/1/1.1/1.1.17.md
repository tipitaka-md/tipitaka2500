# 𑁧𑁇𑁧𑁇𑁧𑁭 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓

### 𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀢𑀺𑀓𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓

(17.)

50. 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (**𑁧𑁦𑁪𑁧, 𑁧𑁪𑁩𑁦**)
