# 𑁧𑁇𑁧𑁇𑁯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀢𑁆𑀢𑀺𑀓

* 𑀫𑀸𑀢𑀺𑀓𑀸

* 𑀢𑀺𑀓𑀫𑀸𑀢𑀺𑀓𑀸

* 𑀤𑀲𑁆𑀲𑀦𑁂𑀦𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀢𑁆𑀢𑀺𑀓

(9.)

26\. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (**𑁧𑁦𑁧𑁩, 𑁧𑁪𑁦𑁮**)

27\. 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (**𑁧𑁦𑁧𑁮, 𑁧𑁪𑁦𑁯**)

28\. 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (**𑁧𑁦𑁧𑁯, 𑁧𑁪𑁧𑁦**)
