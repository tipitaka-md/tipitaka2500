# 𑁧𑁇𑁧𑁇𑁧𑁩 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

* 𑀫𑀸𑀢𑀺𑀓𑀸

* 𑀢𑀺𑀓𑀫𑀸𑀢𑀺𑀓𑀸

* 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

(13.)

38\. 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (**𑁧𑁦𑁨𑁯, 𑁧𑁪𑁨𑁦**)

39\. 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (**𑁧𑁦𑁩𑁦, 𑁧𑁪𑁨𑁧**)

40\. 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (**𑁧𑁦𑁩𑁧, 𑁧𑁪𑁨𑁨**)
