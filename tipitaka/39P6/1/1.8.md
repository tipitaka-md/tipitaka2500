# 𑁧𑁇𑁮 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀤𑀼𑀓

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀤𑀼𑀓

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀤𑀼𑀓

(23.)

58. 𑀳𑁂𑀢𑀼𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

59. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

60. 𑀳𑁂𑀢𑀼𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀜𑁆𑀘 𑀦𑀳𑁂𑀢𑀼𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

61. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

62. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑀫𑁆𑀧𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
