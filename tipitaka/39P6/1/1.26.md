# 𑁧𑁇𑁨𑁬 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑀤𑀼𑀓

(50.)

146\. 𑀳𑁂𑀢𑀼𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

147\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁄𑁇)
