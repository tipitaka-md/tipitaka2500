# 𑁧𑁇𑁧𑁇𑁯 𑀲𑀤𑁆𑀥𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀤𑁆𑀥𑀲𑀼𑀢𑁆𑀢

(9.)

64\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀤𑁆𑀥𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

65\. “𑀆𑀚𑀸𑀦𑀻𑀬𑀛𑀸𑀬𑀺𑀢𑀁 𑀔𑁄, 𑀲𑀤𑁆𑀥, 𑀛𑀸𑀬; 𑀫𑀸 𑀔𑀍𑀉𑀗𑁆𑀓𑀛𑀸𑀬𑀺𑀢𑀁𑁇 𑀓𑀣𑀜𑁆𑀘, 𑀔𑀍𑀉𑀗𑁆𑀓𑀛𑀸𑀬𑀺𑀢𑀁 𑀳𑁄𑀢𑀺? 𑀅𑀲𑁆𑀲𑀔𑀍𑀉𑀗𑁆𑀓𑁄 𑀳𑀺, 𑀲𑀤𑁆𑀥, 𑀤𑁄𑀡𑀺𑀬𑀸 𑀩𑀤𑁆𑀥𑁄 ‘𑀬𑀯𑀲𑀁 𑀬𑀯𑀲𑀦𑁆’𑀢𑀺 𑀛𑀸𑀬𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀦 𑀳𑀺, 𑀲𑀤𑁆𑀥, 𑀅𑀲𑁆𑀲𑀔𑀍𑀉𑀗𑁆𑀓𑀲𑁆𑀲 𑀤𑁄𑀡𑀺𑀬𑀸 𑀩𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀁 𑀅𑀚𑁆𑀚 𑀅𑀲𑁆𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀓𑀸𑀭𑀡𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀓𑀺𑀫𑀲𑁆𑀲𑀸𑀳𑀁 𑀧𑀝𑀺𑀓𑀭𑁄𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀤𑁄𑀡𑀺𑀬𑀸 𑀩𑀤𑁆𑀥𑁄 ‘𑀬𑀯𑀲𑀁 𑀬𑀯𑀲𑀦𑁆’𑀢𑀺 𑀛𑀸𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀲𑀤𑁆𑀥, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀭𑀺𑀲𑀔𑀍𑀉𑀗𑁆𑀓𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀧𑀭𑁂𑀢𑁂𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀁𑀬𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀁 𑀓𑀢𑁆𑀯𑀸 𑀛𑀸𑀬𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀅𑀯𑀚𑁆𑀛𑀸𑀬𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀧𑀭𑁂𑀢𑁂𑀦, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁𑀬𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀁 𑀓𑀢𑁆𑀯𑀸 𑀛𑀸𑀬𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀅𑀯𑀚𑁆𑀛𑀸𑀬𑀢𑀺𑁇 𑀲𑁄 𑀧𑀣𑀯𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀆𑀧𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀢𑁂𑀚𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀯𑀸𑀬𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀇𑀥𑀮𑁄𑀓𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀧𑀭𑀮𑁄𑀓𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸, 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀲𑀤𑁆𑀥, 𑀧𑀼𑀭𑀺𑀲𑀔𑀍𑀉𑀗𑁆𑀓𑀛𑀸𑀬𑀺𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

66\. 𑀓𑀣𑀜𑁆𑀘, 𑀲𑀤𑁆𑀥, 𑀆𑀚𑀸𑀦𑀻𑀬𑀛𑀸𑀬𑀺𑀢𑀁 𑀳𑁄𑀢𑀺? 𑀪𑀤𑁆𑀭𑁄 𑀳𑀺, 𑀲𑀤𑁆𑀥, 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀤𑁄𑀡𑀺𑀬𑀸 𑀩𑀤𑁆𑀥𑁄 𑀦 ‘𑀬𑀯𑀲𑀁 𑀬𑀯𑀲𑀦𑁆’𑀢𑀺 𑀛𑀸𑀬𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀪𑀤𑁆𑀭𑀲𑁆𑀲 𑀳𑀺, 𑀲𑀤𑁆𑀥, 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀲𑁆𑀲 𑀤𑁄𑀡𑀺𑀬𑀸 𑀩𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀁 𑀅𑀚𑁆𑀚 𑀅𑀲𑁆𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀓𑀸𑀭𑀡𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀓𑀺𑀫𑀲𑁆𑀲𑀸𑀳𑀁 𑀧𑀝𑀺𑀓𑀭𑁄𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀤𑁄𑀡𑀺𑀬𑀸 𑀩𑀤𑁆𑀥𑁄 𑀦 ‘𑀬𑀯𑀲𑀁 𑀬𑀯𑀲𑀦𑁆’𑀢𑀺 𑀛𑀸𑀬𑀢𑀺𑁇 𑀪𑀤𑁆𑀭𑁄 𑀳𑀺, 𑀲𑀤𑁆𑀥, 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀬𑀣𑀸 𑀇𑀡𑀁 𑀬𑀣𑀸 𑀩𑀦𑁆𑀥𑀁 𑀬𑀣𑀸 𑀚𑀸𑀦𑀺𑀁 𑀬𑀣𑀸 𑀓𑀮𑀺𑀁 𑀏𑀯𑀁 𑀧𑀢𑁄𑀤𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑁄𑀳𑀭𑀡𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀲𑀤𑁆𑀥, 𑀪𑀤𑁆𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀦 𑀓𑀸𑀫𑀭𑀸𑀕𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦 𑀓𑀸𑀫𑀭𑀸𑀕𑀧𑀭𑁂𑀢𑁂𑀦, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀦 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀦 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀧𑀭𑁂𑀢𑁂𑀦, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀦𑁂𑀯 𑀧𑀣𑀯𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀆𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀢𑁂𑀚𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀇𑀥𑀮𑁄𑀓𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀧𑀭𑀮𑁄𑀓𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸, 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦 𑀛𑀸𑀬𑀢𑀺; 𑀛𑀸𑀬𑀢𑀺 𑀘 𑀧𑀦𑁇 𑀏𑀯𑀁 𑀛𑀸𑀬𑀺𑀜𑁆𑀘 𑀧𑀦, 𑀲𑀤𑁆𑀥, 𑀪𑀤𑁆𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀁 𑀲𑁃𑀦𑁆𑀤𑀸 𑀤𑁂𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀸 𑀆𑀭𑀓𑀸𑀯 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺—

67\. _‘𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀬𑀲𑁆𑀲 𑀢𑁂 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀬𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀲𑀻’”𑀢𑀺𑁇_  


68\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀣𑀁 𑀛𑀸𑀬𑀻 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀤𑁆𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀦𑁂𑀯 𑀧𑀣𑀯𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀆𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀢𑁂𑀚𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀇𑀥𑀮𑁄𑀓𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀦 𑀧𑀭𑀮𑁄𑀓𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺, 𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸, 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦 𑀛𑀸𑀬𑀢𑀺; 𑀛𑀸𑀬𑀢𑀺 𑀘 𑀧𑀦? 𑀓𑀣𑀁 𑀛𑀸𑀬𑀺𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀤𑁆𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀁 𑀲𑁃𑀦𑁆𑀤𑀸 𑀤𑁂𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀸 𑀆𑀭𑀓𑀸𑀯 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺—

69\. _‘𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀬𑀲𑁆𑀲 𑀢𑁂 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀬𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀲𑀻’”𑀢𑀺𑁇_  


70\. “𑀇𑀥, 𑀲𑀤𑁆𑀥, 𑀪𑀤𑁆𑀭𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀲𑁆𑀲 𑀧𑀣𑀯𑀺𑀬𑀁 𑀧𑀣𑀯𑀺𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀆𑀧𑀲𑁆𑀫𑀺𑀁 𑀆𑀧𑁄𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀢𑁂𑀚𑀲𑁆𑀫𑀺𑀁 𑀢𑁂𑀚𑁄𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀸𑀬𑁄𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑁂 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑁂 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑁂 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀇𑀥𑀮𑁄𑀓𑁂 𑀇𑀥𑀮𑁄𑀓𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀮𑁄𑀓𑁂 𑀧𑀭𑀮𑁄𑀓𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸, 𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀪𑀽𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀛𑀸𑀬𑀻 𑀔𑁄, 𑀲𑀤𑁆𑀥, 𑀪𑀤𑁆𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀦𑁂𑀯 𑀧𑀣𑀯𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺…𑀧𑁂…  𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸, 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦 𑀛𑀸𑀬𑀢𑀺; 𑀛𑀸𑀬𑀢𑀺 𑀘 𑀧𑀦𑁇 𑀏𑀯𑀁 𑀛𑀸𑀬𑀺𑀜𑁆𑀘 𑀧𑀦, 𑀲𑀤𑁆𑀥, 𑀪𑀤𑁆𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀁 𑀲𑁃𑀦𑁆𑀤𑀸 𑀤𑁂𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀸 𑀆𑀭𑀓𑀸𑀯 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺—

71\. _‘𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀬𑀲𑁆𑀲 𑀢𑁂 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀬𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀲𑀻’”𑀢𑀺𑁇_  


---

72\. 𑀦𑀯𑀫𑀁𑁇
