# 𑁧𑁇𑁧𑁇𑁧𑁦 𑀫𑁄𑀭𑀦𑀺𑀯𑀸𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀕𑁆𑀕

### 𑀫𑁄𑀭𑀦𑀺𑀯𑀸𑀧𑀲𑀼𑀢𑁆𑀢

(10.)

73\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑁄𑀭𑀦𑀺𑀯𑀸𑀧𑁂 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

74\. “𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀅𑀲𑁂𑀔𑁂𑀦 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀅𑀲𑁂𑀔𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀅𑀲𑁂𑀔𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦—  𑀇𑀫𑁂𑀳𑀺, 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇

75\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑁂𑀦, 𑀆𑀤𑁂𑀲𑀦𑀸𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑁂𑀦, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑁂𑀦—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇

76\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸, 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑁂𑀦, 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇

77\. 𑀤𑁆𑀯𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀯𑀺𑀚𑁆𑀚𑀸𑀬, 𑀘𑀭𑀡𑁂𑀦—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀧𑁂𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁂𑀦 𑀕𑀸𑀣𑀸 𑀪𑀸𑀲𑀺𑀢𑀸—

78\. _‘𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀚𑀦𑁂𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑁂 𑀕𑁄𑀢𑁆𑀢𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑁂’𑀢𑀺𑁇_  


79\. 𑀲𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁂𑀦 𑀕𑀸𑀣𑀸 𑀪𑀸𑀲𑀺𑀢𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸, 𑀦𑁄 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸; 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀸, 𑀦𑁄 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀸; 𑀅𑀦𑀼𑀫𑀢𑀸 𑀫𑀬𑀸𑁇 𑀅𑀳𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—

80\. _‘𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀚𑀦𑁂𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑁂 𑀕𑁄𑀢𑁆𑀢𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑁂’”𑀢𑀺𑁇_
