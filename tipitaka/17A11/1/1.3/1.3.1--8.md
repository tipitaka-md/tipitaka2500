# 𑁧𑁇𑁩𑁇𑁧--𑁮

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀸𑀫𑀜𑁆𑀜𑀯𑀕𑁆𑀕

(22--29.)

212. “𑀏𑀓𑀸𑀤𑀲𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀕𑁄𑀕𑀡𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀢𑀼𑀁 𑀨𑀸𑀢𑀺𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀦 𑀭𑀽𑀧𑀜𑁆𑀜𑀽 𑀳𑁄𑀢𑀺, 𑀦 𑀮𑀓𑁆𑀔𑀡𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀆𑀲𑀸𑀝𑀺𑀓𑀁 𑀳𑀸𑀭𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀯𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀥𑀽𑀫𑀁 𑀓𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀢𑀺𑀢𑁆𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀧𑀻𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀯𑀻𑀣𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀦𑀯𑀲𑁂𑀲𑀤𑁄𑀳𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀢𑁂 𑀉𑀲𑀪𑀸 𑀕𑁄𑀧𑀺𑀢𑀭𑁄 𑀕𑁄𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀸 𑀢𑁂 𑀦 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀽𑀚𑀸𑀬 𑀧𑀽𑀚𑁂𑀢𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀕𑁄𑀕𑀡𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀢𑀼𑀁 𑀨𑀸𑀢𑀺𑀁 𑀓𑀸𑀢𑀼𑀁𑁇

213. 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…𑀧𑁂…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀔𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀭𑀸𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀭𑁄𑀥𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁”𑁇
