# 𑁧𑁇𑁩𑁇𑁪𑁮𑁧--𑁯𑁬𑁦

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀸𑀫𑀜𑁆𑀜𑀯𑀕𑁆𑀕

(502.)

224\. 𑀏𑀓𑀸𑀤𑀲𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀪𑀩𑁆𑀩𑁄 𑀕𑁄𑀕𑀡𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀢𑀼𑀁 𑀨𑀸𑀢𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀭𑀽𑀧𑀜𑁆𑀜𑀽 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

225\. 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁…𑀧𑁂…  𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀦𑁆𑀢𑀺𑁇
