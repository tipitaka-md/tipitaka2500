# 𑁧𑁇𑁨𑁇𑁪 𑀲𑀼𑀪𑀽𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀪𑀽𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(14.)

112\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀪𑀽𑀢𑀺 𑀲𑀤𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀪𑀽𑀢𑀺𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀸𑀫𑀸𑀬𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀲𑀤𑁆𑀥𑁄 𑀦𑀸𑀫𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀼𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑁄, 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺𑁇

113\. “𑀓𑀘𑁆𑀘𑀺 𑀧𑀦𑀸𑀬𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑁂𑀲𑀽”𑀢𑀺? “𑀏𑀢𑀲𑁆𑀲, 𑀪𑀕𑀯𑀸, 𑀓𑀸𑀮𑁄; 𑀏𑀢𑀲𑁆𑀲, 𑀲𑀼𑀕𑀢, 𑀓𑀸𑀮𑁄, 𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀸𑀦𑀺 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀇𑀤𑀸𑀦𑀸𑀳𑀁 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀤𑀺 𑀯𑀸 𑀅𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑁂𑀲𑀼 𑀬𑀤𑀺 𑀯𑀸 𑀦𑁄”𑀢𑀺𑁇

114\. “𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀼𑀡𑀸𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀪𑀽𑀢𑀺 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

115\. “𑀇𑀥, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻, 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁧)

116\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀢𑀥𑀭𑁄 𑀲𑀼𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄; 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀥𑀸𑀢𑀸 𑀯𑀘𑀲𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁨)

117\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁩)

118\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀔𑀫𑁄 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀔𑀫𑁄 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁪)

119\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀢𑀢𑁆𑀭 𑀤𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑁄 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀢𑀢𑁆𑀭 𑀤𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑁄 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁫)

120\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀺𑀬𑀲𑀫𑀼𑀤𑀸𑀳𑀸𑀭𑁄 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁂 𑀉𑀍𑀆𑀭𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁄𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀺𑀬𑀲𑀫𑀼𑀤𑀸𑀳𑀸𑀭𑁄 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁂 𑀉𑀍𑀆𑀭𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁄, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁬)

121\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁭)

122\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁮)

123\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀏𑀓𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀁 𑀤𑁆𑀯𑁂𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀢𑀺𑀲𑁆𑀲𑁄𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀘𑀢𑀲𑁆𑀲𑁄𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀧𑀜𑁆𑀘𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀤𑀲𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀯𑀻𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀢𑀺𑀁𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀚𑀸𑀢𑀺𑀲𑀢𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀅𑀦𑁂𑀓𑁂𑀧𑀺 𑀲𑀁𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂 𑀅𑀦𑁂𑀓𑁂𑀧𑀺 𑀯𑀺𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂 𑀅𑀦𑁂𑀓𑁂𑀧𑀺 𑀲𑀁𑀯𑀝𑁆𑀝𑀯𑀺𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂—  ‘𑀅𑀫𑀼𑀢𑁆𑀭𑀸𑀲𑀺𑀁 𑀏𑀯𑀁𑀦𑀸𑀫𑁄 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑁄 𑀏𑀯𑀁𑀯𑀡𑁆𑀡𑁄 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑁄 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀏𑀯𑀫𑀸𑀬𑀼𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀤𑀧𑀸𑀤𑀺𑀁; 𑀢𑀢𑁆𑀭𑀸𑀧𑀸𑀲𑀺𑀁 𑀏𑀯𑀁𑀦𑀸𑀫𑁄 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑁄 𑀏𑀯𑀁𑀯𑀡𑁆𑀡𑁄 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑁄 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀏𑀯𑀫𑀸𑀬𑀼𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀇𑀥𑀽𑀧𑀧𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀲𑀸𑀓𑀸𑀭𑀁 𑀲𑁅𑀤𑁆𑀤𑁂𑀲𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀏𑀓𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀁 𑀤𑁆𑀯𑁂𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄…𑀧𑁂…  𑀇𑀢𑀺 𑀲𑀸𑀓𑀸𑀭𑀁 𑀲𑁅𑀤𑁆𑀤𑁂𑀲𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺𑁇 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁯)

124\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀢𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂, 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀇𑀫𑁂 𑀯𑀢 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀉𑀧𑀯𑀸𑀤𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸, 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇 𑀇𑀫𑁂 𑀯𑀸 𑀧𑀦 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀸𑀤𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸, 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀢𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂, 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦…𑀧𑁂…  𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁦)

125\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸…𑀧𑁂…  𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁧𑁧)

126\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀪𑀽𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀸𑀦𑀺 𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀺, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀸𑀦𑀺 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀅𑀬𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀢𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇

127\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻, 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇

128\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀢𑀥𑀭𑁄 𑀲𑀼𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄; 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀥𑀸𑀢𑀸 𑀯𑀘𑀲𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇

129\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇

130\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇

131\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀢𑀢𑁆𑀣 𑀤𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑁄 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇

132\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀺𑀬𑀲𑀫𑀼𑀤𑀸𑀳𑀸𑀭𑁄 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁂 𑀉𑀍𑀆𑀭𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁄𑁇

133\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇

134\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇

135\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀏𑀓𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀁 𑀤𑁆𑀯𑁂𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄…𑀧𑁂…  𑀇𑀢𑀺 𑀲𑀸𑀓𑀸𑀭𑀁 𑀲𑁅𑀤𑁆𑀤𑁂𑀲𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺𑁇

136\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦…𑀧𑁂…  𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

137\. 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸…𑀧𑁂…  𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀸𑀦𑀺 𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀺, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀸𑀦𑀺 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀅𑀬𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀢𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

138\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀼𑀪𑀽𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀲𑀤𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀬𑀤𑀸 𑀘 𑀢𑁆𑀯𑀁, 𑀲𑀼𑀪𑀽𑀢𑀺, 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀇𑀫𑀺𑀦𑀸 𑀲𑀤𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇
