# 𑁧𑁇𑁨𑁇𑁩 𑀦𑀦𑁆𑀤𑀺𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(13.)

100\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇

101\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁄—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀫𑁆𑀧𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀕𑀯𑀦𑁆𑀢𑀜𑁆𑀘 𑀮𑀘𑁆𑀙𑀸𑀫𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

102\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑁇 𑀦𑀦𑁆𑀤𑀺𑀬𑁄𑀧𑀺 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺, 𑀪𑀕𑀯𑀦𑁆𑀢𑀜𑁆𑀘 𑀮𑀪𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  “𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

103\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁄—  “𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  ‘𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂—  ‘𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀦𑀸𑀯𑀺𑀳𑀸𑀭𑁂𑀳𑀺 𑀯𑀺𑀳𑀭𑀢𑀁 𑀓𑁂𑀦𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺?

104\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀦𑀦𑁆𑀤𑀺𑀬𑁇 𑀏𑀢𑀁 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁, 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀣—  ‘𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀦𑀸𑀯𑀺𑀳𑀸𑀭𑁂𑀳𑀺 𑀯𑀺𑀳𑀭𑀢𑀁 𑀓𑁂𑀦𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺? 𑀲𑀤𑁆𑀥𑁄 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀆𑀭𑀸𑀥𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄; 𑀲𑀻𑀮𑀯𑀸 𑀆𑀭𑀸𑀥𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄; 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀆𑀭𑀸𑀥𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀓𑀼𑀲𑀻𑀢𑁄; 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀆𑀭𑀸𑀥𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺; 𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀆𑀭𑀸𑀥𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑁄; 𑀧𑀜𑁆𑀜𑀯𑀸 𑀆𑀭𑀸𑀥𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄𑁇 𑀇𑀫𑁂𑀲𑀼 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬 𑀧𑀜𑁆𑀘𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

105\. 𑀇𑀥 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺, 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

106\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀭𑀩𑁆𑀪 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

107\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁂 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  ‘𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀬𑀲𑁆𑀲 𑀫𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀑𑀯𑀸𑀤𑀓𑀸 𑀅𑀦𑀼𑀲𑀸𑀲𑀓𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

108\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀘𑀸𑀕𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  ‘𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀬𑁄𑀳𑀁 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀚𑀸𑀬 𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀸𑀫𑀺 𑀫𑀼𑀢𑁆𑀢𑀘𑀸𑀕𑁄 𑀧𑀬𑀢𑀧𑀸𑀡𑀺 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀭𑀢𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑁄’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀘𑀸𑀕𑀁 𑀆𑀭𑀩𑁆𑀪 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

109\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀤𑁂𑀯𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  ‘𑀬𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸, 𑀢𑀸 𑀓𑀭𑀡𑀻𑀬𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀦 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀢𑀺𑀘𑀬𑀁’𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑀫𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀓𑀭𑀡𑀻𑀬𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀦 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺 𑀓𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀢𑀺𑀘𑀬𑀁; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀬𑀸 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸, 𑀢𑀸 𑀓𑀭𑀡𑀻𑀬𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀦 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀢𑀺𑀘𑀬𑀁𑁇 𑀇𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀤𑁂𑀯𑀢𑀸 𑀆𑀭𑀩𑁆𑀪 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

110\. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀚𑀳𑀢𑁂𑀯 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂, 𑀦 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀓𑀼𑀫𑁆𑀪𑁄 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁄 𑀯𑀫𑀢𑁂𑀯 𑀉𑀤𑀓𑀁, 𑀦𑁄 𑀯𑀦𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀸𑀯𑀫𑀢𑀺; 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀯𑀸 𑀧𑀦, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀲𑀼𑀓𑁆𑀔𑁂 𑀢𑀺𑀡𑀤𑀸𑀬𑁂 𑀅𑀕𑁆𑀕𑀺 𑀫𑀼𑀢𑁆𑀢𑁄 𑀟𑀳𑀜𑁆𑀜𑁂𑀯 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦𑁄 𑀤𑀟𑁆𑀠𑀁 𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀇𑀫𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀚𑀳𑀢𑁂𑀯 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂, 𑀦 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀻”𑀢𑀺𑁇
