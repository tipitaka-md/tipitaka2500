# 𑁧𑁇𑁧 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

(1.)

2\. 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼, 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼, 𑀭𑀲𑀥𑀸𑀢𑀼, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑁇
