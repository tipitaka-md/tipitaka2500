# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀥𑀸𑀢𑀼𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(15.)

194\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺?

195\. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀸𑀢𑀽 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀥𑀸𑀢𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀥𑀸𑀢𑀽𑁇

196\. «  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

197\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺?

198\. 𑀲𑁄𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂… 𑁇

199\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

200\. «  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

201\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺?

202\. 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀁…  𑀕𑀦𑁆𑀥𑀁…  𑀭𑀲𑀁…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂… 𑁇

203\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

204\. «  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
