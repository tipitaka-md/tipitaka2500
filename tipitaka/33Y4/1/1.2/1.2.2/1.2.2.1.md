# 𑁧𑁇𑁨𑁇𑁨𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(12.)

163. »  𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀽𑀢𑀺?

164. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼, 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘𑁇

165. «  𑀥𑀸𑀢𑀽 𑀲𑁄𑀢𑀥𑀸𑀢𑀽𑀢𑀺?

166. 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀥𑀸𑀢𑀼 𑀘𑁂𑀯 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀸𑀢𑀽 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑁇

167. »  𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀽𑀢𑀺?

168. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼, 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘𑁇

169. «  𑀥𑀸𑀢𑀽 𑀖𑀸𑀦𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀥𑀸𑀢𑀽 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀽𑀢𑀺?

170. 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀥𑀸𑀢𑀼 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀸𑀢𑀽 𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑁇

171. (𑀬𑀣𑀸 𑀆𑀬𑀢𑀦𑀬𑀫𑀓𑁂 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺 𑀏𑀯𑀁 𑀥𑀸𑀢𑀼𑀬𑀫𑀓𑁂𑀧𑀺 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑁇 𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇)
