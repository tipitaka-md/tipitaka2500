# 𑁧𑁇𑁨𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀥𑀸𑀢𑀼𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(17.)

212\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺?

213\. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀸𑀢𑀽 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀥𑀸𑀢𑀼𑁇

214\. «  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

215\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺?

216\. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀸𑀢𑀽 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀥𑀸𑀢𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀥𑀸𑀢𑀼𑁇

217\. «  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀖𑀸𑀦𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

218\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

219\. «  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

220\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

221\. «  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀦 𑀥𑀸𑀢𑀽 𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

---

222\. (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇)



223\. (𑀬𑀣𑀸 𑀆𑀬𑀢𑀦𑀬𑀫𑀓𑀲𑁆𑀲 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺 𑀏𑀯𑀁 𑀥𑀸𑀢𑀼𑀬𑀫𑀓𑀲𑁆𑀲 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇)
