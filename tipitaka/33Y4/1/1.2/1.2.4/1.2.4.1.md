# 𑁧𑁇𑁨𑁇𑁪𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀥𑀸𑀢𑀼𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(16.)

205\. »  𑀘𑀓𑁆𑀔𑀼 𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

206\. «  𑀥𑀸𑀢𑀽 𑀲𑁄𑀢𑀥𑀸𑀢𑀽𑀢𑀺?

207\. 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀥𑀸𑀢𑀼 𑀘𑁂𑀯 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀸𑀢𑀽 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑁇

208\. »  𑀘𑀓𑁆𑀔𑀼 𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

209\. «  𑀥𑀸𑀢𑀽 𑀖𑀸𑀦𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀥𑀸𑀢𑀽 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀽𑀢𑀺?

210\. 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀥𑀸𑀢𑀼 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀸𑀢𑀽 𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑁇
