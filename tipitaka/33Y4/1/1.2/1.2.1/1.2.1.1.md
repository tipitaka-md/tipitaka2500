# 𑁧𑁇𑁨𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(10.)

96\. »  𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀽𑀢𑀺?

97\. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼, 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘𑁇

98\. «  𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘𑀓𑁆𑀔𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

99\. »  𑀲𑁄𑀢𑀁 𑀲𑁄𑀢𑀥𑀸𑀢𑀽𑀢𑀺?

100\. 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑁄𑀢𑀁 𑀲𑁄𑀢𑀁, 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑁇 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀲𑁄𑀢𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀘𑁇

101\. «  𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀲𑁄𑀢𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

102\. »  𑀖𑀸𑀦𑀁 𑀖𑀸𑀦𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

103\. «  𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀖𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

---

104\. (𑀚𑀺𑀯𑁆𑀳𑀸𑀧𑀺 𑀖𑀸𑀦𑀥𑀸𑀢𑀼𑀲𑀤𑀺𑀲𑀸𑁇)



105\. »  𑀓𑀸𑀬𑁄 𑀓𑀸𑀬𑀥𑀸𑀢𑀽𑀢𑀺?

106\. 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀓𑀸𑀬𑁄, 𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑁇 𑀓𑀸𑀬𑀥𑀸𑀢𑀼 𑀓𑀸𑀬𑁄 𑀘𑁂𑀯 𑀓𑀸𑀬𑀥𑀸𑀢𑀼 𑀘𑁇

107\. «  𑀓𑀸𑀬𑀥𑀸𑀢𑀼 𑀓𑀸𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

108\. »  𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀽𑀢𑀺?

109\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀜𑁆𑀘𑁂𑀯 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀘𑁇

110\. «  𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

---

111\. (𑀲𑀤𑁆𑀤𑁄 𑀖𑀸𑀦𑀲𑀤𑀺𑀲𑁄𑁇)



112\. »  𑀕𑀦𑁆𑀥𑁄 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀽𑀢𑀺?

113\. 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁄 𑀲𑀫𑀸𑀥𑀺𑀕𑀦𑁆𑀥𑁄 𑀧𑀜𑁆𑀜𑀸𑀕𑀦𑁆𑀥𑁄 𑀕𑀦𑁆𑀥𑁄, 𑀦 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑁇 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼 𑀕𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼 𑀘𑁇

114\. «  𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼 𑀕𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

115\. »  𑀭𑀲𑁄 𑀭𑀲𑀥𑀸𑀢𑀽𑀢𑀺?

116\. 𑀅𑀢𑁆𑀣𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀭𑀲𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑁄 𑀭𑀲𑁄, 𑀦 𑀭𑀲𑀥𑀸𑀢𑀼𑁇 𑀭𑀲𑀥𑀸𑀢𑀼 𑀭𑀲𑁄 𑀘𑁂𑀯 𑀭𑀲𑀥𑀸𑀢𑀼 𑀘𑁇

117\. «  𑀭𑀲𑀥𑀸𑀢𑀼 𑀭𑀲𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

---

118\. (𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄 𑀖𑀸𑀦𑀲𑀤𑀺𑀲𑁄𑁇)



119\. »  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

120\. «  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

121\. 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁… 𑁇

122\. »  𑀫𑀦𑁄 𑀫𑀦𑁄𑀥𑀸𑀢𑀽𑀢𑀺?

123\. 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀫𑀦𑁄, 𑀦 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑁇 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀫𑀦𑁄 𑀘𑁂𑀯 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀘𑁇

124\. «  𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀫𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

125\. »  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

126\. «  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

127\. »  𑀥𑀫𑁆𑀫𑁄 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀽𑀢𑀺?

128\. 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀘𑁇

129\. «  𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀥𑀫𑁆𑀫𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
