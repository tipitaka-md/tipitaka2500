# 𑁧𑁇𑁨𑁇𑁧𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(11.)

130\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

131\. «  𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀦 𑀘𑀓𑁆𑀔𑀽𑀢𑀺?

132\. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀘𑀓𑁆𑀔𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑁇

133\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

134\. «  𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀦 𑀲𑁄𑀢𑀦𑁆𑀢𑀺?

135\. 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑁄𑀢𑀁 𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀁𑁇 𑀲𑁄𑀢𑀜𑁆𑀘 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀲𑁄𑀢𑀁 𑀦 𑀘 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑁇

136\. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀖𑀸𑀦𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

137\. «  𑀦 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀦 𑀖𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

138\. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸… 𑁇

---

139\. (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁, 𑀉𑀪𑀢𑁄 𑀆𑀫𑀦𑁆𑀢𑀸𑁇)



140\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

141\. «  𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀼 𑀦 𑀓𑀸𑀬𑁄𑀢𑀺?

142\. 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀓𑀸𑀬𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑁄𑁇 𑀓𑀸𑀬𑀜𑁆𑀘 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑁇

143\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

144\. «  𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀦 𑀭𑀽𑀧𑀦𑁆𑀢𑀺?

145\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀁𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀭𑀽𑀧𑀁 𑀦 𑀘 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑁇

146\. »  𑀦 𑀲𑀤𑁆𑀤𑁄…𑀧𑁂…  𑀦 𑀕𑀦𑁆𑀥𑁄 𑀦 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

147\. «  𑀦 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼 𑀦 𑀕𑀦𑁆𑀥𑁄𑀢𑀺?

148\. 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁄 𑀲𑀫𑀸𑀥𑀺𑀕𑀦𑁆𑀥𑁄 𑀧𑀜𑁆𑀜𑀸𑀕𑀦𑁆𑀥𑁄 𑀦 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼, 𑀕𑀦𑁆𑀥𑁄𑁇 𑀕𑀦𑁆𑀥𑀜𑁆𑀘 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀕𑀦𑁆𑀥𑁄 𑀦 𑀘 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑁇

149\. »  𑀦 𑀭𑀲𑁄 𑀦 𑀭𑀲𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

150\. «  𑀦 𑀭𑀲𑀥𑀸𑀢𑀼 𑀦 𑀭𑀲𑁄𑀢𑀺?

151\. 𑀅𑀢𑁆𑀣𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀭𑀲𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑁄 𑀦 𑀭𑀲𑀥𑀸𑀢𑀼, 𑀭𑀲𑁄𑁇 𑀭𑀲𑀜𑁆𑀘 𑀭𑀲𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀭𑀲𑁄 𑀦 𑀘 𑀭𑀲𑀥𑀸𑀢𑀼𑁇

152\. »  𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

153\. «  𑀦 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

154\. »  𑀦 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀦 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀦 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁… 𑁇

155\. »  𑀦 𑀫𑀦𑁄 𑀦 𑀫𑀦𑁄𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

156\. «  𑀦 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀦 𑀫𑀦𑁄𑀢𑀺?

157\. 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑁇 𑀫𑀦𑀜𑁆𑀘 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀫𑀦𑁄 𑀦 𑀘 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑁇

158\. »  𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

159\. «  𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

160\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

161\. «  𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀦 𑀥𑀫𑁆𑀫𑁄𑀢𑀺?

162\. 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑁄𑁇 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑁇
