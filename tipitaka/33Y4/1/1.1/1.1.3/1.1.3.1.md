# 𑁧𑁇𑁧𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀥𑀸𑀢𑀼𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫

(6.)

43\. »  𑀘𑀓𑁆𑀔𑀼 𑀥𑀸𑀢𑀼?

44\. «  𑀥𑀸𑀢𑀽 𑀘𑀓𑁆𑀔𑀼?

45\. »  𑀲𑁄𑀢𑀁 𑀥𑀸𑀢𑀼?

46\. «  𑀥𑀸𑀢𑀽 𑀲𑁄𑀢𑀁?

47\. 𑀖𑀸𑀦𑀁 𑀥𑀸𑀢𑀼? …𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀥𑀸𑀢𑀼? …  𑀓𑀸𑀬𑁄 𑀥𑀸𑀢𑀼? …  𑀭𑀽𑀧𑀁 𑀥𑀸𑀢𑀼? …  𑀲𑀤𑁆𑀤𑁄 𑀥𑀸𑀢𑀼? …  𑀕𑀦𑁆𑀥𑁄 𑀥𑀸𑀢𑀼? …  𑀭𑀲𑁄 𑀥𑀸𑀢𑀼? …  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄 𑀥𑀸𑀢𑀼? …

48\. »  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀥𑀸𑀢𑀼?

49\. «  𑀥𑀸𑀢𑀽 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

50\. »  𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀥𑀸𑀢𑀼?

51\. «  𑀥𑀸𑀢𑀽 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

52\. …  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? …𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? …  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

53\. »  𑀫𑀦𑁄 𑀥𑀸𑀢𑀼?

54\. «  𑀥𑀸𑀢𑀽 𑀫𑀦𑁄?

55\. »  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀥𑀸𑀢𑀼?

56\. «  𑀥𑀸𑀢𑀽 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

57\. »  𑀥𑀫𑁆𑀫𑁄 𑀥𑀸𑀢𑀼?

58\. «  𑀥𑀸𑀢𑀽 𑀥𑀫𑁆𑀫𑁄?
