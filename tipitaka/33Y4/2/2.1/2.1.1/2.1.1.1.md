# 𑁨𑁇𑁧𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(18.)

225\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

226\. 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀅𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀲𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

227\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

228\. 𑀲𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

229\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

230\. 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀲𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

231\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

232\. 𑀲𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

233\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

234\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

235\. 𑀲𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

236\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

237\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

238\. 𑀲𑀘𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

239\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

240\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

241\. 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

242\. (𑀬𑀣𑀸 𑀆𑀬𑀢𑀦𑀬𑀫𑀓𑀁 𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀏𑀯𑀁 𑀥𑀸𑀢𑀼𑀬𑀫𑀓𑀫𑁆𑀧𑀺 𑀯𑀺𑀪𑀚𑀺𑀢𑀩𑁆𑀩𑀁, 𑀲𑀤𑀺𑀲𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)
