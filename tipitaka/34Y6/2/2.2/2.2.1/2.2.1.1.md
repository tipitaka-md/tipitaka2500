# 𑁨𑁇𑁨𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(79.)

369\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

370\. 𑀯𑀺𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

371\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

372\. 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

373\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

374\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

375\. 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

(80.)

376\. »  𑀬𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

377\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

378\. 𑀯𑀺𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇
