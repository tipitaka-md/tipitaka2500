# 𑁨𑁇𑁨𑁇𑁪𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀢𑀻𑀢𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(98.)

439\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

440\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

441\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

442\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

443\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

444\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

(99.)

445\. »  𑀬𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

446\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

447\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀯𑀺𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇
