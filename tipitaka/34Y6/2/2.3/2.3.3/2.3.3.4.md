# 𑁨𑁇𑁩𑁇𑁩𑁇𑁪 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

(142.)

610\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

611\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

612\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

613\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

614\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

615\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(143.)

616\. »  𑀬𑀲𑁆𑀲 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

617\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

618\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
