# 𑁨𑁇𑁩𑁇𑁩𑁇𑁩 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

(140.)

600. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

601. 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

602. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

603. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀇𑀢𑀭𑁂𑀲𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

604. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

605. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

606. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀇𑀢𑀭𑁂𑀲𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

(141.)

607. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

608. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

609. 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀇𑀢𑀭𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇
