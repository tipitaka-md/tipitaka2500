# 𑁨𑁇𑁧𑁇𑁫𑁇𑁬 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

(67.)

316\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

317\. 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

318\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

319\. 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

320\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

321\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀯𑀺𑀦𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

322\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(68.)

323\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

324\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀯𑀺𑀦𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

325\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

326\. 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇
