# 𑁨𑁇𑁧𑁇𑁧𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(21.)

135\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

136\. 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

137\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

138\. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

139\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

140\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

141\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑁂 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

(22.)

142\. »  𑀬𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

143\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

144\. 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇
