# 𑁨𑁇𑁧𑁇𑁧𑁇𑁫 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

(26.)

157\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

158\. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

159\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

160\. 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

161\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

162\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

163\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(27.)

164\. »  𑀬𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

165\. 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑁂 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

166\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
