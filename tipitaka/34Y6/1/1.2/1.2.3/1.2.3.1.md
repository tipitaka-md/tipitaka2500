# 𑁧𑁇𑁨𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀲𑀗𑁆𑀔𑀸𑀭𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(16.)

106\. »  𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

107\. «  𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

108\. »  𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

109\. «  𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

110\. »  𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

111\. «  𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇
