# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀲𑀗𑁆𑀔𑀸𑀭𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(17.)

112\. »  𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

113\. 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

114\. «  𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

115\. 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

116\. »  𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

117\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

118\. «  𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

119\. 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

(18.)

120\. »  𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

121\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

122\. «  𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

123\. 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

---
