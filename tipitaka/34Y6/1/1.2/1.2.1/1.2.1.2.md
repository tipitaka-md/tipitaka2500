# 𑁧𑁇𑁨𑁇𑁧𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(9.)

58. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

59. 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑁄, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀓𑀸𑀬𑀜𑁆𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

60. «  𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑁄𑀢𑀺?

61. 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀓𑀸𑀬𑁄𑁇 𑀓𑀸𑀬𑀜𑁆𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

62. »  𑀦 𑀯𑀘𑀻 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

63. 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀯𑀘𑀺𑀜𑁆𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀘𑀻 𑀦 𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

64. «  𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻𑀢𑀺?

65. 𑀯𑀘𑀻 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀯𑀘𑀻𑁇 𑀯𑀘𑀺𑀜𑁆𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀘𑀻 𑀦 𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

66. »  𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

67. 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

68. «  𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀦𑁆𑀢𑀺?

69. 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇
