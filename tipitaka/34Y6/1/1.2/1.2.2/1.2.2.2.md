# 𑁧𑁇𑁨𑁇𑁨𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(13.)

88\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

89\. 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑁄, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀓𑀸𑀬𑀜𑁆𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

90\. «  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

91\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

92\. 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀓𑀸𑀬𑁄, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀓𑀸𑀬𑀜𑁆𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

93\. «  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(14.)

94\. »  𑀦 𑀯𑀘𑀻 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

95\. 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀯𑀘𑀺𑀜𑁆𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀘𑀻 𑀦 𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

96\. «  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

97\. »  𑀦 𑀯𑀘𑀻 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

98\. 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀯𑀘𑀻, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀯𑀘𑀺𑀜𑁆𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀘𑀻 𑀦 𑀘 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

99\. «  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(15.)

100\. »  𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

101\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

102\. «  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

103\. »  𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

104\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

105\. «  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
