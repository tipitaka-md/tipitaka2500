# 𑁧𑁇𑁨𑁇𑁨𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫

(10.)

70\. »  𑀓𑀸𑀬𑁄 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

71\. «  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

72\. 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁂𑀯 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

73\. »  𑀓𑀸𑀬𑁄 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

74\. «  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

75\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

(11.)

76\. »  𑀯𑀘𑀻 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

77\. «  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

78\. 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁂𑀯 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

79\. »  𑀯𑀘𑀻 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

80\. «  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

81\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

(12.)

82\. »  𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

83\. «  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

84\. 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁂𑀯 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

85\. »  𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺? 𑀦𑁄𑁇

86\. «  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑀺?

87\. 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁂𑀯 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇
