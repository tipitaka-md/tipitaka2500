# 𑁧𑁇𑁧𑁇𑁪𑁇𑁪 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

* 𑀉𑀤𑁆𑀤𑁂𑀲

* 𑀫𑀽𑀮𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀸𑀫𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

(46.)

92\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

93\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸?

(47.)

94\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

95\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸?

(48.)

96\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

97\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸? (𑁪)
