# 𑁧𑁇𑁧𑁇𑁪𑁇𑁧 𑀫𑀽𑀮𑀦𑀬

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀸𑀫𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀦𑀬

(37.)

74\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑀸?

75\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸?

(38.)

76\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀸?

77\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸?

(39.)

78\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀸?

79\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸? (𑁧)
