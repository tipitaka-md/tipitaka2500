# 𑁧𑁇𑁧𑁇𑁪𑁇𑁨 𑀫𑀽𑀮𑀫𑀽𑀮𑀦𑀬

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀸𑀫𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀫𑀽𑀮𑀦𑀬

(40.)

80\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑀫𑀽𑀮𑀸?

81\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸?

(41.)

82\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸?

83\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸?

(42.)

84\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸?

85\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸? (𑁨)
