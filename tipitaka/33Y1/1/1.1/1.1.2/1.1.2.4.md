# 𑁧𑁇𑁧𑁇𑁨𑁇𑁪 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀓𑀼𑀲𑀮𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

(22.)

44\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

45\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸?

(23.)

46\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

47\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸?

(24.)

48\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

49\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸? (𑁪)
