# 𑁧𑁇𑁧𑁇𑁧𑁇𑁪 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

### 𑀓𑀼𑀲𑀮𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

(10.)

20\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

21\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸?

(11.)

22\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

23\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸?

(12.)

24\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸?

25\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? (𑁪)
