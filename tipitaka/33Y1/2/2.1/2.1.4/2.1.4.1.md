# 𑁨𑁇𑁧𑁇𑁪𑁇𑁧 𑀫𑀽𑀮𑀦𑀬

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀸𑀫𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀦𑀬

(86.)

214\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑀸𑀢𑀺?

215\. 𑀦𑀯𑁂𑀯 𑀦𑀸𑀫𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑀸𑁇

216\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(87.)

217\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀸𑀢𑀺?

218\. 𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀦 𑀏𑀓𑀫𑀽𑀮𑀁𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀁𑁇

219\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸𑀢𑀺?

220\. 𑀦𑀸𑀫𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀁, 𑀦 𑀦𑀸𑀫𑀁𑁇 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀜𑁆𑀘𑁂𑀯 𑀦𑀸𑀫𑀜𑁆𑀘𑁇

(88.)

221\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀸𑀢𑀺?

222\. 𑀫𑀽𑀮𑀸𑀦𑀺 𑀬𑀸𑀦𑀺 𑀏𑀓𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀸𑀫𑀫𑀽𑀮𑀸𑀦𑀺 𑀏𑀓𑀫𑀽𑀮𑀸𑀦𑀺 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀸𑀦𑀺 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦𑀸𑀫𑀫𑀽𑀮𑀲𑀳𑀚𑀸𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀸, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀸𑁇

223\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁧)
