# 𑁨𑁇𑁧𑁇𑁪𑁇𑁩 𑀫𑀽𑀮𑀓𑀦𑀬

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀫𑀽𑀮𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀸𑀫𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀫𑀽𑀮𑀓𑀦𑀬

(92.)

234\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑀓𑀸𑀢𑀺?

235\. 𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀦𑀸𑀫𑀁 𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑀓𑀁𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑀓𑀁𑁇

236\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸𑀢𑀺?

237\. 𑀦𑀸𑀫𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑀓𑀁, 𑀦 𑀦𑀸𑀫𑀁𑁇 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑀸𑀫𑀜𑁆𑀘𑁇

(93.)

238\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀸𑀢𑀺?

239\. 𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇

240\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸𑀢𑀺?

241\. 𑀦𑀸𑀫𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀁, 𑀦 𑀦𑀸𑀫𑀁𑁇 𑀦𑀸𑀫𑀁 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑀸𑀫𑀜𑁆𑀘𑁇

(94.)

242\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀓𑀸𑀢𑀺?

243\. 𑀫𑀽𑀮𑀸𑀦𑀺 𑀬𑀸𑀦𑀺 𑀏𑀓𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀸𑀫𑀫𑀽𑀮𑀸𑀦𑀺 𑀏𑀓𑀫𑀽𑀮𑀓𑀸𑀦𑀺 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀓𑀸𑀦𑀺 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦𑀸𑀫𑀫𑀽𑀮𑀲𑀳𑀚𑀸𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀓𑀸, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀓𑀸𑁇

244\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁩)
