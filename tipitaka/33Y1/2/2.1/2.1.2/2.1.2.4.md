# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀓𑀼𑀲𑀮𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀦𑀬

(71.)

167. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸𑀢𑀺?

168. 𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀁𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀁𑁇

169. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀢𑀺?

170. 𑀅𑀓𑀼𑀲𑀮𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀁 𑀦 𑀅𑀓𑀼𑀲𑀮𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀜𑁆𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘𑁇

(72.)

171. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸𑀢𑀺?

172. 𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀁𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀁𑁇

173. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀢𑀺?

174. 𑀅𑀓𑀼𑀲𑀮𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀁, 𑀦 𑀅𑀓𑀼𑀲𑀮𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀜𑁆𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘𑁇

(73.)

175. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸𑀢𑀺?

176. 𑀫𑀽𑀮𑀸𑀦𑀺 𑀬𑀸𑀦𑀺 𑀏𑀓𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸𑀦𑀺 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸𑀦𑀺 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀲𑀳𑀚𑀸𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸𑁇

177. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁪)
