# 𑁨𑁇𑁧𑁇𑁨𑁇𑁨 𑀫𑀽𑀮𑀫𑀽𑀮𑀦𑀬

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀓𑀼𑀲𑀮𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀫𑀽𑀮𑀦𑀬

(65.)

146. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀢𑀺?

147. 𑀢𑀻𑀡𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑁇

148. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(66.)

149. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀢𑀺?

150. 𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀁𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀁𑁇

151. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀢𑀺?

152. 𑀅𑀓𑀼𑀲𑀮𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀁, 𑀦 𑀅𑀓𑀼𑀲𑀮𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀜𑁆𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘𑁇

(67.)

153. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀢𑀺?

154. 𑀫𑀽𑀮𑀸𑀦𑀺 𑀬𑀸𑀦𑀺 𑀏𑀓𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀲𑀳𑀚𑀸𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑁇

155. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁨)
