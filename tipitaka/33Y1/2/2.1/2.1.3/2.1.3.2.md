# 𑁨𑁇𑁧𑁇𑁩𑁇𑁨 𑀫𑀽𑀮𑀫𑀽𑀮𑀦𑀬

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀫𑀽𑀮𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀧𑀤 𑀦𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀫𑀽𑀮𑀫𑀽𑀮𑀦𑀬

(77.)

187\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀢𑀺?

188\. 𑀢𑀻𑀡𑁂𑀯 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑁇

189\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(78.)

190\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀢𑀺?

191\. 𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀁𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀁𑁇

192\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(79.)

193\. »  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀢𑀺?

194\. 𑀫𑀽𑀮𑀸𑀦𑀺 𑀬𑀸𑀦𑀺 𑀏𑀓𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑀸𑀦𑀺 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑀲𑀳𑀚𑀸𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀏𑀓𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸𑁇

195\. «  𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑀽𑀮𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀽𑀮𑀫𑀽𑀮𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁨)
