# 𑁩𑁇𑁨𑁇𑁨 𑀘𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀘𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(61.)

393\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

394\. “𑀢𑀻𑀡𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀽𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺? 𑀫𑀁𑀲𑀘𑀓𑁆𑀔𑀼, 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼, 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀡𑀺 𑀘𑀓𑁆𑀔𑀽𑀦𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

395\. _“𑀫𑀁𑀲𑀘𑀓𑁆𑀔𑀼 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼,_  
_𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀏𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀘𑀓𑁆𑀔𑀽𑀦𑀺,_  
_𑀅𑀓𑁆𑀔𑀸𑀲𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


396\. _𑀫𑀁𑀲𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄,_  
_𑀫𑀕𑁆𑀕𑁄 𑀤𑀺𑀩𑁆𑀩𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀬𑀢𑁄 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺,_  
_𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀮𑀸𑀪𑀸,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


397\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
