# 𑁩𑁇𑁨𑁇𑁪 𑀅𑀤𑁆𑀥𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀅𑀤𑁆𑀥𑀸𑀲𑀼𑀢𑁆𑀢

(63.)

406\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

407\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀢𑀻𑀢𑁄 𑀅𑀤𑁆𑀥𑀸, 𑀅𑀦𑀸𑀕𑀢𑁄 𑀅𑀤𑁆𑀥𑀸, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀤𑁆𑀥𑀸—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀅𑀤𑁆𑀥𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

408\. _“𑀅𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀲𑀢𑁆𑀢𑀸,_  
_𑀅𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀅𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀁 𑀅𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀬𑁄𑀕𑀫𑀸𑀬𑀦𑁆𑀢𑀺 𑀫𑀘𑁆𑀘𑀼𑀦𑁄𑁇_  


409\. _𑀅𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑀭𑀁 𑀦 𑀫𑀜𑁆𑀜𑀢𑀺;_  
_𑀨𑀼𑀝𑁆𑀞𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀫𑀦𑀲𑀸,_  
_𑀲𑀦𑁆𑀢𑀺𑀧𑀤𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


410\. _𑀲 𑀯𑁂 𑀅𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀦𑁆𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀧𑀤𑁂 𑀭𑀢𑁄;_  
_𑀲𑀗𑁆𑀔𑀸𑀬𑀲𑁂𑀯𑀻 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀦𑁄𑀧𑁂𑀢𑀺 𑀯𑁂𑀤𑀕𑀽”𑀢𑀺𑁇_  


411\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
