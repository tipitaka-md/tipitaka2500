# 𑁩𑁇𑁨𑁇𑁯 𑀧𑀞𑀫𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(68.)

435\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

436\. “𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀕𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀤𑁄𑀲𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀫𑁄𑀳𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀩𑀤𑁆𑀥𑁄 𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄’𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀤𑁄𑀲𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀫𑁄𑀳𑁄 𑀧𑀳𑀻𑀦𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀅𑀩𑀤𑁆𑀥𑁄 𑀫𑀸𑀭𑀲𑁆𑀲 𑀑𑀫𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀸𑀭𑀧𑀸𑀲𑁄 𑀦 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄’”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

437\. _“𑀬𑀲𑁆𑀲 𑀭𑀸𑀕𑁄 𑀘 𑀤𑁄𑀲𑁄 𑀘,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀭𑀸𑀚𑀺𑀢𑀸;_  
_𑀢𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀜𑁆𑀜𑀢𑀭𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀪𑀽𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑁂𑀭𑀪𑀬𑀸𑀢𑀻𑀢𑀁,_  
_𑀆𑀳𑀼 𑀲𑀩𑁆𑀩𑀧𑁆𑀧𑀳𑀸𑀬𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_  


438\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
