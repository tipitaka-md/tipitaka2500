# 𑁩𑁇𑁨𑁇𑁬 𑀲𑀼𑀘𑀭𑀺𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀘𑀭𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(65.)

419\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

420\. “𑀢𑀻𑀡𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀘𑀭𑀺𑀢𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺? 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑀁, 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑀁, 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀡𑀺 𑀲𑀼𑀘𑀭𑀺𑀢𑀸𑀦𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

421\. _“𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀘;_  
_𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑀜𑁆𑀘𑀜𑁆𑀜𑀁 𑀤𑁄𑀲𑀲𑀁𑀳𑀺𑀢𑀁𑁇_  


422\. _𑀅𑀓𑀢𑁆𑀯𑀸𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀓𑀼𑀲𑀮𑀁 𑀩𑀳𑀼𑀁;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀕𑁆𑀕𑀁 𑀲𑁄 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


423\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
