# 𑁩𑁇𑁨𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(69.)

440\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

441\. “𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀤𑁄𑀲𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀫𑁄𑀳𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀅𑀢𑀺𑀡𑁆𑀡𑁄 𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀲𑀊𑀫𑀺𑀁 𑀲𑀯𑀻𑀘𑀺𑀁 𑀲𑀸𑀯𑀝𑁆𑀝𑀁 𑀲𑀕𑀳𑀁 𑀲𑀭𑀓𑁆𑀔𑀲𑀁’𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀤𑁄𑀲𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀫𑁄𑀳𑁄 𑀧𑀳𑀻𑀦𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀅𑀢𑀭𑀺 𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀲𑀊𑀫𑀺𑀁 𑀲𑀯𑀻𑀘𑀺𑀁 𑀲𑀸𑀯𑀝𑁆𑀝𑀁 𑀲𑀕𑀳𑀁 𑀲𑀭𑀓𑁆𑀔𑀲𑀁, 𑀢𑀺𑀡𑁆𑀡𑁄 𑀧𑀸𑀭𑀗𑁆𑀕𑀢𑁄 𑀣𑀮𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

442\. _“𑀬𑀲𑁆𑀲 𑀭𑀸𑀕𑁄 𑀘 𑀤𑁄𑀲𑁄 𑀘,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀭𑀸𑀚𑀺𑀢𑀸;_  
_𑀲𑁄𑀫𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀲𑀕𑀳𑀁 𑀲𑀭𑀓𑁆𑀔𑀲𑀁,_  
_𑀲𑀊𑀫𑀺𑀪𑀬𑀁 𑀤𑀼𑀢𑁆𑀢𑀭𑀁 𑀅𑀘𑁆𑀘𑀢𑀸𑀭𑀺𑁇_  


443\. _𑀲𑀗𑁆𑀕𑀸𑀢𑀺𑀕𑁄 𑀫𑀘𑁆𑀘𑀼𑀚𑀳𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺,_  
_𑀧𑀳𑀸𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀬;_  
_𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑁄 𑀲𑁄 𑀦 𑀧𑀫𑀸𑀡𑀫𑁂𑀢𑀺,_  
_𑀅𑀫𑁄𑀳𑀬𑀺 𑀫𑀘𑁆𑀘𑀼𑀭𑀸𑀚𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀫𑀻”𑀢𑀺𑁇_  


444\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

447\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



448\. _𑀧𑀼𑀜𑁆𑀜𑀁 𑀘𑀓𑁆𑀔𑀼 𑀅𑀣 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺,_  
_𑀅𑀤𑁆𑀥𑀸 𑀘 𑀘𑀭𑀺𑀢𑀁 𑀤𑀼𑀯𑁂 𑀲𑁄𑀘𑀺;_
