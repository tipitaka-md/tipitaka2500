# 𑁩𑁇𑁪𑁇𑁬 𑀅𑀲𑀼𑀪𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀅𑀲𑀼𑀪𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻𑀲𑀼𑀢𑁆𑀢

(85.)

591\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

592\. “𑀅𑀲𑀼𑀪𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀣; 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺 𑀘 𑀯𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀼; 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀣𑁇 𑀅𑀲𑀼𑀪𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀬𑁄 𑀲𑀼𑀪𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀳𑀻𑀬𑀢𑀺𑁇 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀺𑀢𑀸𑀬 𑀬𑁂 𑀩𑀸𑀳𑀺𑀭𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀲𑀬𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑀸, 𑀢𑁂 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀬𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀲𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀬𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

593\. _“𑀅𑀲𑀼𑀪𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁,_  
_𑀆𑀦𑀸𑀧𑀸𑀦𑁂 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀣𑀁,_  
_𑀧𑀲𑁆𑀲𑀁 𑀆𑀢𑀸𑀧𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


594\. _𑀲 𑀯𑁂 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀬𑀢𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀯𑁄𑀲𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲 𑀯𑁂 𑀬𑁄𑀕𑀸𑀢𑀺𑀕𑁄 𑀫𑀼𑀦𑀻”𑀢𑀺𑁇_  


595\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
