# 𑁩𑁇𑁪𑁇𑁭 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

(86.)

597\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

598\. “𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀬𑀫𑀦𑀼𑀥𑀫𑁆𑀫𑁄 𑀳𑁄𑀢𑀺 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀬—  𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄𑀬𑀦𑁆𑀢𑀺 𑀪𑀸𑀲𑀫𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑀜𑁆𑀜𑁂𑀯 𑀪𑀸𑀲𑀢𑀺 𑀦𑁄 𑀅𑀥𑀫𑁆𑀫𑀁, 𑀯𑀺𑀢𑀓𑁆𑀓𑀬𑀫𑀸𑀦𑁄 𑀯𑀸 𑀥𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀜𑁆𑀜𑁂𑀯 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀦𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀁, 𑀢𑀤𑀼𑀪𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

599\. _“𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑁄 𑀥𑀫𑁆𑀫𑀭𑀢𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀬𑀁;_  
_𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺𑁇_  


600\. _𑀘𑀭𑀁 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀁,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀉𑀤 𑀯𑀸 𑀲𑀬𑀁;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀲𑀦𑁆𑀢𑀺𑀫𑁂𑀯𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇_  


601\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
