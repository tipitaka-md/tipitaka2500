# 𑁩𑁇𑁪𑁇𑁧 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(80.)

535\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

536\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀦𑀯𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀧𑀭𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀢𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

537\. _“𑀅𑀦𑀯𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀕𑀸𑀭𑀯𑁄;_  
_𑀲𑀳𑀦𑀦𑁆𑀤𑀻 𑀅𑀫𑀘𑁆𑀘𑁂𑀳𑀺,_  
_𑀆𑀭𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑀸𑁇_  


538\. _𑀬𑁄 𑀘 𑀧𑀼𑀢𑁆𑀢𑀧𑀲𑀼𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀯𑀸𑀳𑁂 𑀲𑀁𑀳𑀭𑀸𑀦𑀺 𑀘;_  
_𑀪𑀩𑁆𑀩𑁄 𑀲𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀨𑀼𑀝𑁆𑀞𑀼𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀦𑁆”𑀢𑀺𑁇_  


539\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
