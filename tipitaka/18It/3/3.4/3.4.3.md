# 𑁩𑁇𑁪𑁇𑁩 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(82.)

553\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

554\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀺 𑀲𑀫𑀬𑀸 𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺—  ‘𑀏𑀲𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀫𑀸𑀭𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀬 𑀘𑁂𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺 𑀲𑀫𑀬𑀸 𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇

555\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺—  ‘𑀏𑀲𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀫𑀸𑀭𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑁂𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺 𑀲𑀫𑀬𑀸 𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇

556\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺—  ‘𑀏𑀲𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫𑁄 𑀢𑀫𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀫𑀲𑀻𑀲𑀁 𑀅𑀪𑀺𑀯𑀺𑀚𑀺𑀬 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺 𑀲𑀫𑀬𑀸 𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀤𑁂𑀯𑀲𑀤𑁆𑀤𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀺 𑀲𑀫𑀬𑀸 𑀲𑀫𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

557\. _“𑀤𑀺𑀲𑁆𑀯𑀸 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑀁;_  
_𑀤𑁂𑀯𑀢𑀸𑀧𑀺 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁𑁇_  


558\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀬𑁄 𑀢𑁆𑀯𑀁 𑀤𑀼𑀚𑁆𑀚𑀬𑀫𑀚𑁆𑀛𑀪𑀽;_  
_𑀚𑁂𑀢𑁆𑀯𑀸𑀦 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁂𑀦 𑀅𑀦𑀸𑀯𑀭𑀁𑁇_  


559\. _𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀯𑀢𑀸 𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀁;_  
_𑀢𑀜𑁆𑀳𑀺 𑀢𑀲𑁆𑀲 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀬𑁂𑀦 𑀫𑀘𑁆𑀘𑀼𑀯𑀲𑀁 𑀯𑀚𑁂”𑀢𑀺𑁇_  


560\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
