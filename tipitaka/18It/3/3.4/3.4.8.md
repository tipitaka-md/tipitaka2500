# 𑁩𑁇𑁪𑁇𑁮 𑀅𑀦𑁆𑀥𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀅𑀦𑁆𑀥𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(87.)

603\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

604\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀦𑁆𑀥𑀓𑀭𑀡𑀸 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀸 𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑀭𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑀸 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀥𑀓𑀭𑀡𑁄 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑁄 𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑀭𑀡𑁄 𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑁄 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀥𑀓𑀭𑀡𑁄 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑁄 𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑀭𑀡𑁄 𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑁄 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀥𑀓𑀭𑀡𑁄 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑁄 𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑀭𑀡𑁄 𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑁄 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀦𑁆𑀥𑀓𑀭𑀡𑀸 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀸 𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑀭𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑀸 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑁇

605\. 𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀦𑀦𑁆𑀥𑀓𑀭𑀡𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀸 𑀜𑀸𑀡𑀓𑀭𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀓𑀸 𑀅𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀦𑁆𑀥𑀓𑀭𑀡𑁄 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑁄 𑀜𑀸𑀡𑀓𑀭𑀡𑁄 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀓𑁄 𑀅𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀦𑁆𑀥𑀓𑀭𑀡𑁄 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑁄 𑀜𑀸𑀡𑀓𑀭𑀡𑁄 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀓𑁄 𑀅𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀦𑁆𑀥𑀓𑀭𑀡𑁄 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑁄 𑀜𑀸𑀡𑀓𑀭𑀡𑁄 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀓𑁄 𑀅𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀦𑀦𑁆𑀥𑀓𑀭𑀡𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀸 𑀜𑀸𑀡𑀓𑀭𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀓𑀸 𑀅𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

606\. _“𑀢𑀬𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀓𑀼𑀲𑀮𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑀬𑁂,_  
_𑀢𑀬𑁄 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑁂 𑀦𑀺𑀭𑀸𑀓𑀭𑁂;_  
_𑀲 𑀯𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀦𑀺 𑀯𑀺𑀘𑀸𑀭𑀺𑀢𑀸𑀦𑀺,_  
_𑀲𑀫𑁂𑀢𑀺 𑀯𑀼𑀝𑁆𑀞𑀻𑀯 𑀭𑀚𑀁 𑀲𑀫𑀽𑀳𑀢𑀁;_  
_𑀲 𑀯𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑀽𑀧𑀲𑀫𑁂𑀦 𑀘𑁂𑀢𑀲𑀸,_  
_𑀇𑀥𑁂𑀯 𑀲𑁄 𑀲𑀦𑁆𑀢𑀺𑀧𑀤𑀁 𑀲𑀫𑀚𑁆𑀛𑀕𑀸”𑀢𑀺𑁇_  


607\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
