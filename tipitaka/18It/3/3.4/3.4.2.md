# 𑁩𑁇𑁪𑁇𑁨 𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(81.)

541\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

542\. “𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀸, 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇

543\. 𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀸, 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇

544\. 𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘 𑀢𑀤𑀼𑀪𑀬𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀸, 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇

545\. 𑀢𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑀤𑀸𑀫𑀺; ( ) 𑀅𑀧𑀺 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀤𑁂𑀯 𑀫𑁂 𑀲𑀸𑀫𑀁 𑀜𑀸𑀢𑀁 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀸𑀫𑀁 𑀯𑀺𑀤𑀺𑀢𑀁 𑀢𑀫𑁂𑀯𑀸𑀳𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

546\. 𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀸, 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇

547\. 𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀸, 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇

548\. 𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘 𑀢𑀤𑀼𑀪𑀬𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀸, 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

549\. _“𑀬𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑀭𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀲𑀫𑀸𑀥𑀺 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄𑁇_  


550\. _𑀢𑀁 𑀛𑀸𑀬𑀺𑀦𑀁 𑀲𑀸𑀢𑀢𑀺𑀓𑀁,_  
_𑀲𑀼𑀔𑀼𑀫𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑀁;_  
_𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀬𑀸𑀭𑀸𑀫𑀁,_  
_𑀆𑀳𑀼 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀢𑀻”𑀢𑀺𑁇_  


551\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
