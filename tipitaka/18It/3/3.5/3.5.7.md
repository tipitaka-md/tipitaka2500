# 𑁩𑁇𑁫𑁇𑁭 𑀓𑀸𑀫𑀬𑁄𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀫𑀬𑁄𑀕𑀲𑀼𑀢𑁆𑀢

(96.)

685\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

686\. “𑀓𑀸𑀫𑀬𑁄𑀕𑀬𑀼𑀢𑁆𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑀬𑁄𑀕𑀬𑀼𑀢𑁆𑀢𑁄 𑀆𑀕𑀸𑀫𑀻 𑀳𑁄𑀢𑀺 𑀆𑀕𑀦𑁆𑀢𑀸 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀁𑁇 𑀓𑀸𑀫𑀬𑁄𑀕𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑀬𑁄𑀕𑀬𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀸𑀕𑀦𑁆𑀢𑀸 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀁𑁇 𑀓𑀸𑀫𑀬𑁄𑀕𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑀬𑁄𑀕𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀅𑀭𑀳𑀸 𑀳𑁄𑀢𑀺, 𑀔𑀻𑀡𑀸𑀲𑀯𑁄”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

687\. _“𑀓𑀸𑀫𑀬𑁄𑀕𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀪𑀯𑀬𑁄𑀕𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀕𑀸𑀫𑀺𑀦𑁄𑁇_  


688\. _𑀬𑁂 𑀘 𑀓𑀸𑀫𑁂 𑀧𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁;_  
_𑀪𑀯𑀬𑁄𑀕𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀅𑀦𑀸𑀕𑀸𑀫𑀻𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀭𑁂𑁇_  


689\. _𑀬𑁂 𑀘 𑀔𑁄 𑀙𑀺𑀦𑁆𑀦𑀲𑀁𑀲𑀬𑀸,_  
_𑀔𑀻𑀡𑀫𑀸𑀦𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸;_  
_𑀢𑁂 𑀯𑁂 𑀧𑀸𑀭𑀗𑁆𑀕𑀢𑀸 𑀮𑁄𑀓𑁂,_  
_𑀬𑁂 𑀧𑀢𑁆𑀢𑀸 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀦𑁆”𑀢𑀺𑁇_  


690\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
