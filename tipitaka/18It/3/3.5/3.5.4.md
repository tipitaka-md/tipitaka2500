# 𑁩𑁇𑁫𑁇𑁪 𑀅𑀕𑁆𑀕𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀕𑁆𑀕𑀺𑀲𑀼𑀢𑁆𑀢

(93.)

661\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

662\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀕𑁆𑀕𑀻𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀭𑀸𑀕𑀕𑁆𑀕𑀺, 𑀤𑁄𑀲𑀕𑁆𑀕𑀺, 𑀫𑁄𑀳𑀕𑁆𑀕𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀅𑀕𑁆𑀕𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

663\. _“𑀭𑀸𑀕𑀕𑁆𑀕𑀺 𑀤𑀳𑀢𑀺 𑀫𑀘𑁆𑀘𑁂,_  
_𑀭𑀢𑁆𑀢𑁂 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁂;_  
_𑀤𑁄𑀲𑀕𑁆𑀕𑀺 𑀧𑀦 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑁂,_  
_𑀦𑀭𑁂 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀺𑀦𑁄𑁇_  


664\. _𑀫𑁄𑀳𑀕𑁆𑀕𑀺 𑀧𑀦 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁂,_  
_𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀅𑀓𑁄𑀯𑀺𑀤𑁂;_  
_𑀏𑀢𑁂 𑀅𑀕𑁆𑀕𑀻 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸,_  
_𑀲𑀓𑁆𑀓𑀸𑀬𑀸𑀪𑀺𑀭𑀢𑀸 𑀧𑀚𑀸𑁇_  


665\. _𑀢𑁂 𑀯𑀟𑁆𑀠𑀬𑀦𑁆𑀢𑀺 𑀦𑀺𑀭𑀬𑀁,_  
_𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀜𑁆𑀘 𑀬𑁄𑀦𑀺𑀬𑁄;_  
_𑀅𑀲𑀼𑀭𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁,_  
_𑀅𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑀸𑁇_  


666\. _𑀬𑁂 𑀘 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸 𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂;_  
_𑀢𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀭𑀸𑀕𑀕𑁆𑀕𑀺𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀺𑀦𑁄𑁇_  


667\. _𑀤𑁄𑀲𑀕𑁆𑀕𑀺𑀁 𑀧𑀦 𑀫𑁂𑀢𑁆𑀢𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀸;_  
_𑀫𑁄𑀳𑀕𑁆𑀕𑀺𑀁 𑀧𑀦 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀬𑀸𑀬𑀁 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀕𑀸𑀫𑀺𑀦𑀻𑁇_  


668\. _𑀢𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀧𑀓𑀸,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑀸;_  
_𑀅𑀲𑁂𑀲𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀺,_  
_𑀅𑀲𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀘𑁆𑀘𑀕𑀼𑀁𑁇_  


669\. _𑀅𑀭𑀺𑀬𑀤𑁆𑀤𑀲𑀸 𑀯𑁂𑀤𑀕𑀼𑀦𑁄,_  
_𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀚𑀸𑀢𑀺𑀓𑁆𑀔𑀬𑀫𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇_  


670\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
