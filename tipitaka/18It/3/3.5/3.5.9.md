# 𑁩𑁇𑁫𑁇𑁯 𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

### 𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(98.)

705\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

706\. “𑀤𑁆𑀯𑁂𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀦𑀸𑀦𑀺—  𑀆𑀫𑀺𑀲𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀸𑀦𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀥𑀫𑁆𑀫𑀤𑀸𑀦𑀁𑁇

707\. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀸—  𑀆𑀫𑀺𑀲𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑁄 𑀘 𑀥𑀫𑁆𑀫𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑁄 𑀘𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀥𑀫𑁆𑀫𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑁄𑁇

708\. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀸—  𑀆𑀫𑀺𑀲𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑁄 𑀘𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑁄”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

709\. _“𑀬𑀫𑀸𑀳𑀼 𑀤𑀸𑀦𑀁 𑀧𑀭𑀫𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁,_  
_𑀬𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑀡𑁆𑀡𑀬𑀺;_  
_𑀅𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀚𑀸𑀦𑀁 𑀓𑁄 𑀦 𑀬𑀚𑁂𑀣 𑀓𑀸𑀮𑁂𑁇_  


710\. _𑀬𑁂 𑀘𑁂𑀯 𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀘𑀽𑀪𑀬𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂;_  
_𑀢𑁂𑀲𑀁 𑀲𑁄 𑀅𑀢𑁆𑀣𑁄 𑀧𑀭𑀫𑁄 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀺,_  
_𑀬𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂”𑀢𑀺𑁇_  


711\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
