# 𑁩𑁇𑁩𑁇𑁬 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(75.)

487. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

488. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑀲𑀫𑁄, 𑀧𑀤𑁂𑀲𑀯𑀲𑁆𑀲𑀻, 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀸𑀪𑀺𑀯𑀲𑁆𑀲𑀻𑁇

489. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑀲𑀫𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀩𑁆𑀩𑁂𑀲𑀜𑁆𑀜𑁂𑀯 𑀦 𑀤𑀸𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀧𑀡𑀤𑁆𑀥𑀺𑀓𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀬𑀸𑀘𑀓𑀸𑀦𑀁 𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀁 𑀬𑀸𑀦𑀁 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀸𑀯𑀲𑀣𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑀲𑀫𑁄 𑀳𑁄𑀢𑀺𑁇

490. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀤𑁂𑀲𑀯𑀲𑁆𑀲𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀤𑀸𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀦 𑀤𑀸𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀧𑀡𑀤𑁆𑀥𑀺𑀓𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀬𑀸𑀘𑀓𑀸𑀦𑀁 𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀁 𑀬𑀸𑀦𑀁 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀸𑀯𑀲𑀣𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀤𑁂𑀲𑀯𑀲𑁆𑀲𑀻 𑀳𑁄𑀢𑀺𑁇

491. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀸𑀪𑀺𑀯𑀲𑁆𑀲𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀩𑁆𑀩𑁂𑀲𑀁𑀯 𑀤𑁂𑀢𑀺, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀧𑀡𑀤𑁆𑀥𑀺𑀓𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀬𑀸𑀘𑀓𑀸𑀦𑀁 𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀁 𑀬𑀸𑀦𑀁 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀸𑀯𑀲𑀣𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀸𑀪𑀺𑀯𑀲𑁆𑀲𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

492. _“𑀦 𑀲𑀫𑀡𑁂 𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂,_  
_𑀦 𑀓𑀧𑀡𑀤𑁆𑀥𑀺𑀓𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁂;_  
_𑀮𑀤𑁆𑀥𑀸𑀦 𑀲𑀁𑀯𑀺𑀪𑀸𑀚𑁂𑀢𑀺,_  
_𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀁;_  
_𑀢𑀁 𑀯𑁂 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑀲𑀫𑁄𑀢𑀺,_  
_𑀆𑀳𑀼 𑀦𑀁 𑀧𑀼𑀭𑀺𑀲𑀸𑀥𑀫𑀁𑁇_  


493. _𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀦 𑀤𑀤𑀸𑀢𑀺,_  
_𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀧𑀯𑁂𑀘𑁆𑀙𑀢𑀺;_  
_𑀢𑀁 𑀯𑁂 𑀧𑀤𑁂𑀲𑀯𑀲𑁆𑀲𑀻𑀢𑀺,_  
_𑀆𑀳𑀼 𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑁄 𑀚𑀦𑀸𑁇_  


494. _𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀯𑀸𑀘𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄;_  
_𑀆𑀫𑁄𑀤𑀫𑀸𑀦𑁄 𑀧𑀓𑀺𑀭𑁂𑀢𑀺,_  
_𑀤𑁂𑀣 𑀤𑁂𑀣𑀸𑀢𑀺 𑀪𑀸𑀲𑀢𑀺𑁇_  


495. _𑀬𑀣𑀸𑀧𑀺 𑀫𑁂𑀖𑁄 𑀣𑀦𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀕𑀚𑁆𑀚𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀣𑀮𑀁 𑀦𑀺𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀽𑀭𑁂𑀢𑀺,_  
_𑀅𑀪𑀺𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑁄𑀯 𑀯𑀸𑀭𑀺𑀦𑀸𑁇_  


496. _𑀏𑀯𑀫𑁂𑀯𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄,_  
_𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀳𑁄𑀢𑀺 𑀢𑀸𑀤𑀺𑀲𑁄;_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀁𑀳𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀝𑁆𑀞𑀸𑀦𑀸𑀥𑀺𑀕𑀢𑀁 𑀥𑀦𑀁;_  
_𑀢𑀧𑁆𑀧𑁂𑀢𑀺 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦,_  
_𑀲𑀫𑁆𑀫𑀸 𑀧𑀢𑁆𑀢𑁂 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁂”𑀢𑀺𑁇_  


497. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
