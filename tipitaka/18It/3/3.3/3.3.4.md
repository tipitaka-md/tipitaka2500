# 𑁩𑁇𑁩𑁇𑁪 𑀲𑀦𑁆𑀢𑀢𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀦𑁆𑀢𑀢𑀭𑀲𑀼𑀢𑁆𑀢

(73.)

471\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

472\. “𑀭𑀽𑀧𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀽𑀧𑀸 𑀲𑀦𑁆𑀢𑀢𑀭𑀸, 𑀅𑀭𑀽𑀧𑁂𑀳𑀺 𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀦𑁆𑀢𑀢𑀭𑁄”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

473\. _“𑀬𑁂 𑀘 𑀭𑀽𑀧𑀽𑀧𑀕𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀝𑁆𑀞𑀸𑀬𑀺𑀦𑁄;_  
_𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀸,_  
_𑀆𑀕𑀦𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀁𑁇_  


474\. _𑀬𑁂 𑀘 𑀭𑀽𑀧𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀭𑀽𑀧𑁂𑀲𑀼 𑀅𑀲𑀡𑁆𑀞𑀺𑀢𑀸;_  
_𑀦𑀺𑀭𑁄𑀥𑁂 𑀬𑁂 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀢𑁂 𑀚𑀦𑀸 𑀫𑀘𑁆𑀘𑀼𑀳𑀸𑀬𑀺𑀦𑁄𑁇_  


475\. _𑀓𑀸𑀬𑁂𑀦 𑀅𑀫𑀢𑀥𑀸𑀢𑀼𑀁,_  
_𑀨𑀼𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁;_  
_𑀉𑀧𑀥𑀺𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀁,_  
_𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀤𑁂𑀲𑁂𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀲𑁄𑀓𑀁 𑀯𑀺𑀭𑀚𑀁 𑀧𑀤𑀦𑁆”𑀢𑀺𑁇_  


476\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
