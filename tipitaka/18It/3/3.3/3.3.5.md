# 𑁩𑁇𑁩𑁇𑁫 𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(74.)

478\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

479\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀢𑀺𑀚𑀸𑀢𑁄, 𑀅𑀦𑀼𑀚𑀸𑀢𑁄, 𑀅𑀯𑀚𑀸𑀢𑁄𑀢𑀺𑁇

480\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑀺𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀦 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀦 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀦 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸𑁇 𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀦𑁂𑀲𑀁 𑀳𑁄𑀢𑀺—  𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄, 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑀺𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

481\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸𑁇 𑀧𑀼𑀢𑁆𑀢𑁄𑀧𑀺 𑀦𑁂𑀲𑀁 𑀳𑁄𑀢𑀺—  𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄, 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

482\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀯𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸𑁇 𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀦𑁂𑀲𑀁 𑀳𑁄𑀢𑀺—  𑀦 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄, 𑀦 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄, 𑀦 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀯𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

483\. _“𑀅𑀢𑀺𑀚𑀸𑀢𑀁 𑀅𑀦𑀼𑀚𑀸𑀢𑀁,_  
_𑀧𑀼𑀢𑁆𑀢𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀅𑀯𑀚𑀸𑀢𑀁 𑀦 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀳𑁄𑀢𑀺 𑀓𑀼𑀮𑀕𑀦𑁆𑀥𑀦𑁄𑁇_  


484\. _𑀏𑀢𑁂 𑀔𑁄 𑀧𑀼𑀢𑁆𑀢𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑁂 𑀪𑀯𑀦𑁆𑀢𑀺 𑀉𑀧𑀸𑀲𑀓𑀸;_  
_𑀲𑀤𑁆𑀥𑀸 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀯𑀤𑀜𑁆𑀜𑀽 𑀯𑀻𑀢𑀫𑀘𑁆𑀙𑀭𑀸;_  
_𑀘𑀦𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀖𑀦𑀸 𑀫𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀭𑀺𑀲𑀸𑀲𑀼 𑀯𑀺𑀭𑁄𑀘𑀭𑁂”𑀢𑀺𑁇_  


485\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

486\. 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
