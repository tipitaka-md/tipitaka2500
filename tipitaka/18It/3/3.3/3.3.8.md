# 𑁩𑁇𑁩𑁇𑁮 𑀪𑀺𑀤𑀼𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀤𑀼𑀭𑀲𑀼𑀢𑁆𑀢

(77.)

510. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

511. “𑀪𑀺𑀤𑀼𑀭𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀬𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀁, 𑀲𑀩𑁆𑀩𑁂 𑀉𑀧𑀥𑀻 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

512. _“𑀓𑀸𑀬𑀜𑁆𑀘 𑀪𑀺𑀤𑀼𑀭𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘 𑀯𑀺𑀭𑀸𑀕𑀼𑀦𑀁;_  
_𑀉𑀧𑀥𑀻𑀲𑀼 𑀪𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀫𑀘𑁆𑀘𑀕𑀸;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀯𑀸 𑀧𑀭𑀫𑀁 𑀲𑀦𑁆𑀢𑀺𑀁,_  
_𑀓𑀸𑀮𑀁 𑀓𑀗𑁆𑀔𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑁄”𑀢𑀺𑁇_  


513. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
