# 𑁩𑁇𑁩𑁇𑁩 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀺𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(72.)

465\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

466\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀺𑀬𑀸 𑀥𑀸𑀢𑀼𑀬𑁄𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀤𑀺𑀤𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁, 𑀭𑀽𑀧𑀸𑀦𑀫𑁂𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀤𑀺𑀤𑀁 𑀆𑀭𑀼𑀧𑁆𑀧𑀁, 𑀬𑀁 𑀔𑁄 𑀧𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀺𑀬𑀸 𑀥𑀸𑀢𑀼𑀬𑁄”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

467\. _“𑀓𑀸𑀫𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀭𑀽𑀧𑀸𑀦𑀜𑁆𑀘 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀁;_  
_𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀣𑀁,_  
_𑀨𑀼𑀲𑀁 𑀆𑀢𑀸𑀧𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


468\. _𑀲 𑀯𑁂 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀬𑀢𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀯𑁄𑀲𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲 𑀯𑁂 𑀬𑁄𑀕𑀸𑀢𑀺𑀕𑁄 𑀫𑀼𑀦𑀻”𑀢𑀺𑁇_  


469\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
