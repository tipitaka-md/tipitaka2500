# 𑁩𑁇𑁩𑁇𑁨 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(71.)

457\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

458\. “𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀸𑀤𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸𑁇 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇

459\. 𑀢𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑀤𑀸𑀫𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀸𑀤𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸𑁇 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇 𑀅𑀧𑀺 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀤𑁂𑀯 𑀲𑀸𑀫𑀁 𑀜𑀸𑀢𑀁 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀸𑀫𑀁 𑀯𑀺𑀤𑀺𑀢𑀁 𑀢𑀤𑁂𑀯𑀸𑀳𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

460\. 𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀸𑀤𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸𑁇 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

461\. _“𑀲𑀫𑁆𑀫𑀸 𑀫𑀦𑀁 𑀧𑀡𑀺𑀥𑀸𑀬,_  
_𑀲𑀫𑁆𑀫𑀸 𑀯𑀸𑀘𑀜𑁆𑀘 𑀪𑀸𑀲𑀺𑀬;_  
_𑀲𑀫𑁆𑀫𑀸 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀸𑀬𑁂𑀦 𑀇𑀥 𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


462\. _𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀧𑀼𑀜𑁆𑀜𑀓𑀭𑁄,_  
_𑀅𑀧𑁆𑀧𑀲𑁆𑀫𑀺𑀁 𑀇𑀥 𑀚𑀻𑀯𑀺𑀢𑁂;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀕𑁆𑀕𑀁 𑀲𑁄 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


463\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

464\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
