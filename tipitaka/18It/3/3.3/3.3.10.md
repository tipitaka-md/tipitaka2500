# 𑁩𑁇𑁩𑁇𑁧𑁦 𑀧𑀭𑀺𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀭𑀺𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(79.)

525. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

526. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀫𑁆𑀫𑀭𑀢𑁄, 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄; 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀪𑀲𑁆𑀲𑀭𑀢𑁄, 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄; 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀢𑁄, 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇

527. 𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀓𑀫𑁆𑀫𑀭𑀢𑁄, 𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄; 𑀦 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀪𑀲𑁆𑀲𑀭𑀢𑁄, 𑀦 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄; 𑀦 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀢𑁄, 𑀦 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

528. _“𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑁄 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑁄,_  
_𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑁄 𑀘 𑀉𑀤𑁆𑀥𑀢𑁄;_  
_𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀨𑀼𑀝𑁆𑀞𑀼𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


529. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀧𑁆𑀧𑀓𑀺𑀘𑁆𑀘𑀲𑁆𑀲,_  
_𑀅𑀧𑁆𑀧𑀫𑀺𑀤𑁆𑀥𑁄 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄;_  
_𑀪𑀩𑁆𑀩𑁄 𑀲𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀨𑀼𑀝𑁆𑀞𑀼𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀦𑁆”𑀢𑀺𑁇_  


530. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
