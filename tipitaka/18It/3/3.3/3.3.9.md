# 𑁩𑁇𑁩𑁇𑁯 𑀥𑀸𑀢𑀼𑀲𑁄𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀥𑀸𑀢𑀼𑀲𑁄𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

(78.)

515. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

516. “𑀥𑀸𑀢𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺, 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

517. 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼, 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇

518. 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

519. 𑀏𑀢𑀭𑀳𑀺𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺, 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

520. _“𑀲𑀁𑀲𑀕𑁆𑀕𑀸 𑀯𑀦𑀣𑁄 𑀚𑀸𑀢𑁄,_  
_𑀅𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀤𑀸𑀭𑀼𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀬𑀣𑀸 𑀲𑀻𑀤𑁂 𑀫𑀳𑀡𑁆𑀡𑀯𑁂𑁇_  


521. _𑀏𑀯𑀁 𑀓𑀼𑀲𑀻𑀢𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀲𑀸𑀥𑀼𑀚𑀻𑀯𑀻𑀧𑀺 𑀲𑀻𑀤𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀢𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀓𑀼𑀲𑀻𑀢𑀁 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑀁𑁇_  


522. _𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀅𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁂𑀳𑀺 𑀛𑀸𑀬𑀺𑀪𑀺;_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀆𑀭𑀤𑁆𑀥𑀯𑀺𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁂𑀳𑀺 𑀲𑀳𑀸𑀯𑀲𑁂”𑀢𑀺𑁇_  


523. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
