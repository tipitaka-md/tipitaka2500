# 𑁩𑁇𑁧𑁇𑁯 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(58.)

373\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

374\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀡𑁆𑀳𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸, 𑀪𑀯𑀢𑀡𑁆𑀳𑀸, 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

375\. _“𑀢𑀡𑁆𑀳𑀸𑀬𑁄𑀕𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀢𑁂 𑀬𑁄𑀕𑀬𑀼𑀢𑁆𑀢𑀸 𑀫𑀸𑀭𑀲𑁆𑀲,_  
_𑀅𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀺𑀦𑁄 𑀚𑀦𑀸;_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀚𑀸𑀢𑀻𑀫𑀭𑀡𑀕𑀸𑀫𑀺𑀦𑁄𑁇_  


376\. _𑀬𑁂 𑀘 𑀢𑀡𑁆𑀳𑀁 𑀧𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀸 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀢𑁂 𑀯𑁂 𑀧𑀸𑀭𑀗𑁆𑀕𑀢𑀸 𑀮𑁄𑀓𑁂,_  
_𑀬𑁂 𑀧𑀢𑁆𑀢𑀸 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀦𑁆”𑀢𑀺𑁇_  


377\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
