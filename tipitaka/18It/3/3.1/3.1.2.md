# 𑁩𑁇𑁧𑁇𑁨 𑀥𑀸𑀢𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀥𑀸𑀢𑀼𑀲𑀼𑀢𑁆𑀢

(51.)

332\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

333\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀬𑁄𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀦𑀺𑀭𑁄𑀥𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀥𑀸𑀢𑀼𑀬𑁄”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

334\. _“𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀆𑀭𑀼𑀧𑁆𑀧𑁂𑀲𑀼 𑀅𑀲𑀡𑁆𑀞𑀺𑀢𑀸;_  
_𑀦𑀺𑀭𑁄𑀥𑁂 𑀬𑁂 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀢𑁂 𑀚𑀦𑀸 𑀫𑀘𑁆𑀘𑀼𑀳𑀸𑀬𑀺𑀦𑁄𑁇_  


335\. _𑀓𑀸𑀬𑁂𑀦 𑀅𑀫𑀢𑀁 𑀥𑀸𑀢𑀼𑀁,_  
_𑀨𑀼𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁;_  
_𑀉𑀧𑀥𑀺𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀁,_  
_𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀤𑁂𑀲𑁂𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀲𑁄𑀓𑀁 𑀯𑀺𑀭𑀚𑀁 𑀧𑀤𑀦𑁆”𑀢𑀺𑁇_  


336\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
