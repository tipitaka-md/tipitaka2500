# 𑁩𑁇𑁧𑁇𑁧 𑀫𑀽𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀽𑀮𑀲𑀼𑀢𑁆𑀢

(50.)

327. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

328. “𑀢𑀻𑀡𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺? 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀤𑁄𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

329. _“𑀮𑁄𑀪𑁄 𑀤𑁄𑀲𑁄 𑀘 𑀫𑁄𑀳𑁄 𑀘,_  
_𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀸𑀧𑀘𑁂𑀢𑀲𑀁;_  
_𑀳𑀺𑀁𑀲𑀦𑁆𑀢𑀺 𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀽𑀢𑀸,_  
_𑀢𑀘𑀲𑀸𑀭𑀁𑀯 𑀲𑀫𑁆𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


330. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
