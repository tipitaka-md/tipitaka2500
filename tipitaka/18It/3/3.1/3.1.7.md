# 𑁩𑁇𑁧𑁇𑁭 𑀧𑀞𑀫𑀆𑀲𑀯𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀆𑀲𑀯𑀲𑀼𑀢𑁆𑀢

(56.)

362\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

363\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀓𑀸𑀫𑀸𑀲𑀯𑁄, 𑀪𑀯𑀸𑀲𑀯𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀆𑀲𑀯𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

364\. _“𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄,_  
_𑀲𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀆𑀲𑀯𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀆𑀲𑀯𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀁𑁇_  


365\. _𑀬𑀢𑁆𑀣 𑀘𑁂𑀢𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀁;_  
_𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀦𑀺𑀘𑁆𑀙𑀸𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇_  


366\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
