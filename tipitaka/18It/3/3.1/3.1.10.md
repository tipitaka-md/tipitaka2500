# 𑁩𑁇𑁧𑁇𑁧𑁦 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

(59.)

379\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

380\. “𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀁 𑀆𑀤𑀺𑀘𑁆𑀘𑁄𑀯 𑀯𑀺𑀭𑁄𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁂𑀔𑁂𑀦 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀲𑁂𑀔𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀲𑁂𑀔𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀁 𑀆𑀤𑀺𑀘𑁆𑀘𑁄𑀯 𑀯𑀺𑀭𑁄𑀘𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

381\. _“𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀥𑀺 𑀧𑀜𑁆𑀜𑀸 𑀘,_  
_𑀬𑀲𑁆𑀲 𑀏𑀢𑁂 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀸;_  
_𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀁,_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑁄𑀯 𑀯𑀺𑀭𑁄𑀘𑀢𑀻”𑀢𑀺𑁇_  


382\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
