# 𑁧𑁇𑁧𑁇𑁮 𑀫𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

(8.)

37. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

38. “𑀫𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀪𑀺𑀚𑀸𑀦𑀁 𑀅𑀧𑀭𑀺𑀚𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀯𑀺𑀭𑀸𑀚𑀬𑀁 𑀅𑀧𑁆𑀧𑀚𑀳𑀁 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬𑁇 𑀫𑀸𑀦𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀚𑀸𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑀸𑀚𑀬𑀁 𑀧𑀚𑀳𑀁 𑀪𑀩𑁆𑀩𑁄 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

39. _“𑀫𑀸𑀦𑀼𑀧𑁂𑀢𑀸 𑀅𑀬𑀁 𑀧𑀚𑀸,_  
_𑀫𑀸𑀦𑀕𑀦𑁆𑀣𑀸 𑀪𑀯𑁂 𑀭𑀢𑀸;_  
_𑀫𑀸𑀦𑀁 𑀅𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀸,_  
_𑀆𑀕𑀦𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀁𑁇_  


40. _𑀬𑁂 𑀘 𑀫𑀸𑀦𑀁 𑀧𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀸𑀦𑀲𑀗𑁆𑀔𑀬𑁂;_  
_𑀢𑁂 𑀫𑀸𑀦𑀕𑀦𑁆𑀣𑀸𑀪𑀺𑀪𑀼𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀫𑀼𑀧𑀘𑁆𑀘𑀕𑀼𑀦𑁆”𑀢𑀺𑁇_  


41. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

42. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
