# 𑁧𑁇𑁧𑁇𑁨 𑀤𑁄𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀤𑁄𑀲𑀲𑀼𑀢𑁆𑀢

(2.)

7\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

8\. “𑀏𑀓𑀥𑀫𑁆𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀚𑀳𑀣; 𑀅𑀳𑀁 𑀯𑁄 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀢𑀸𑀬𑁇 𑀓𑀢𑀫𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀁? 𑀤𑁄𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑀣; 𑀅𑀳𑀁 𑀯𑁄 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀢𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

9\. _“𑀬𑁂𑀦 𑀤𑁄𑀲𑁂𑀦 𑀤𑀼𑀝𑁆𑀞𑀸𑀲𑁂,_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁;_  
_𑀢𑀁 𑀤𑁄𑀲𑀁 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀬,_  
_𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀧𑀳𑀸𑀬 𑀦 𑀧𑀼𑀦𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀓𑀼𑀤𑀸𑀘𑀦𑀦𑁆”𑀢𑀺𑁇_  


10\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

11\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
