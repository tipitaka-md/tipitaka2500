# 𑁧𑁇𑁨𑁇𑁧𑁦 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(20.)

103\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

104\. “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀫𑁆𑀳𑀺 𑀘𑀸𑀬𑀁 𑀲𑀫𑀬𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁂𑀬𑁆𑀬 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂’𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀤𑀼𑀝𑁆𑀞𑀁𑁇 𑀘𑁂𑀢𑁄𑀧𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

105\. _“𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑀁 𑀜𑀢𑁆𑀯𑀸𑀦,_  
_𑀏𑀓𑀘𑁆𑀘𑀁 𑀇𑀥 𑀧𑀼𑀕𑁆𑀕𑀮𑀁;_  
_𑀏𑀢𑀫𑀢𑁆𑀣𑀜𑁆𑀘 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


106\. _𑀇𑀫𑀫𑁆𑀳𑀺 𑀘𑀸𑀬𑀁 𑀲𑀫𑀬𑁂,_  
_𑀓𑀸𑀮𑀁 𑀓𑀬𑀺𑀭𑀸𑀣 𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑀺𑀲𑁆𑀲 𑀧𑀤𑀽𑀲𑀺𑀢𑀁𑁇_  


107\. _𑀬𑀣𑀸 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬,_  
_𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑀣𑀸𑀯𑀺𑀥𑁄;_  
_𑀘𑁂𑀢𑁄𑀧𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀺,_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀦𑁆”𑀢𑀺𑁇_  


108\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
