# 𑁧𑁇𑁨𑁇𑁨 𑀓𑁄𑀥𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀓𑁄𑀥𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

(12.)

61\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

62\. “𑀓𑁄𑀥𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀪𑀺𑀚𑀸𑀦𑀁 𑀅𑀧𑀭𑀺𑀚𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀯𑀺𑀭𑀸𑀚𑀬𑀁 𑀅𑀧𑁆𑀧𑀚𑀳𑀁 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬𑁇 𑀓𑁄𑀥𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀚𑀸𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑀸𑀚𑀬𑀁 𑀧𑀚𑀳𑀁 𑀪𑀩𑁆𑀩𑁄 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

63\. _“𑀬𑁂𑀦 𑀓𑁄𑀥𑁂𑀦 𑀓𑀼𑀤𑁆𑀥𑀸𑀲𑁂,_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁;_  
_𑀢𑀁 𑀓𑁄𑀥𑀁 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀬,_  
_𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀧𑀳𑀸𑀬 𑀦 𑀧𑀼𑀦𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀓𑀼𑀤𑀸𑀘𑀦𑀦𑁆”𑀢𑀺𑁇_  


64\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

65\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
