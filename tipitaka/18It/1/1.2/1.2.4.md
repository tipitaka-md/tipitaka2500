# 𑁧𑁇𑁨𑁇𑁪 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(14.)

71\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

72\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀦𑀻𑀯𑀭𑀡𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀦𑀺𑀯𑀼𑀢𑀸 𑀧𑀚𑀸 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀯𑀼𑀢𑀸 𑀧𑀚𑀸 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

73\. _“𑀦𑀢𑁆𑀣𑀜𑁆𑀜𑁄 𑀏𑀓𑀥𑀫𑁆𑀫𑁄𑀧𑀺,_  
_𑀬𑁂𑀦𑁂𑀯𑀁 𑀦𑀺𑀯𑀼𑀢𑀸 𑀧𑀚𑀸;_  
_𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑀺 𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀁,_  
_𑀬𑀣𑀸 𑀫𑁄𑀳𑁂𑀦 𑀆𑀯𑀼𑀢𑀸𑁇_  


74\. _𑀬𑁂 𑀘 𑀫𑁄𑀳𑀁 𑀧𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑀁 𑀧𑀤𑀸𑀮𑀬𑀼𑀁;_  
_𑀦 𑀢𑁂 𑀧𑀼𑀦 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑀺,_  
_𑀳𑁂𑀢𑀼 𑀢𑁂𑀲𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


75\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
