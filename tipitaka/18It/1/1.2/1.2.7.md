# 𑁧𑁇𑁨𑁇𑁭 𑀤𑀼𑀢𑀺𑀬𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

(17.)

88. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

89. “𑀲𑁂𑀔𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀲𑁆𑀲 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀩𑀸𑀳𑀺𑀭𑀁 𑀅𑀗𑁆𑀕𑀦𑁆𑀢𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀜𑁆𑀜𑀁 𑀏𑀓𑀗𑁆𑀕𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀁 𑀏𑀯𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀁 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀼𑀲𑀮𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀓𑀼𑀲𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

90. _“𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀕𑀸𑀭𑀯𑁄;_  
_𑀓𑀭𑀁 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀘𑀦𑀁,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀧𑀸𑀧𑀼𑀡𑁂 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑀦𑁆”𑀢𑀺𑁇_  


91. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

92. 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
