# 𑁧𑁇𑁩𑁇𑁭 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(27.)

155\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

156\. “𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀧𑀥𑀺𑀓𑀸𑀦𑀺 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑁂𑀯 𑀢𑀸𑀦𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀸𑀲𑀢𑁂 𑀘 𑀢𑀧𑀢𑁂 𑀘 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀘𑁇

157\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀢𑀸𑀭𑀓𑀭𑀽𑀧𑀸𑀦𑀁 𑀧𑀪𑀸 𑀲𑀩𑁆𑀩𑀸 𑀢𑀸 𑀘𑀦𑁆𑀤𑀺𑀬𑀸 𑀧𑀪𑀸𑀬 𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁, 𑀘𑀦𑁆𑀤𑀧𑀪𑀸𑀬𑁂𑀯 𑀢𑀸 𑀅𑀥𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀸𑀲𑀢𑁂 𑀘 𑀢𑀧𑀢𑁂 𑀘 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀘; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀑𑀧𑀥𑀺𑀓𑀸𑀦𑀺 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁, 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑁂𑀯 𑀢𑀸𑀦𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀸𑀲𑀢𑁂 𑀘 𑀢𑀧𑀢𑁂 𑀘 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀘𑁇

158\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀲𑀭𑀤𑀲𑀫𑀬𑁂 𑀯𑀺𑀤𑁆𑀥𑁂 𑀯𑀺𑀕𑀢𑀯𑀮𑀸𑀳𑀓𑁂 𑀤𑁂𑀯𑁂 𑀆𑀤𑀺𑀘𑁆𑀘𑁄 𑀦𑀪𑀁 𑀅𑀩𑁆𑀪𑀼𑀲𑁆𑀲𑀓𑁆𑀓𑀫𑀸𑀦𑁄 𑀲𑀩𑁆𑀩𑀁 𑀆𑀓𑀸𑀲𑀕𑀢𑀁 𑀢𑀫𑀕𑀢𑀁 𑀅𑀪𑀺𑀯𑀺𑀳𑀘𑁆𑀘 𑀪𑀸𑀲𑀢𑁂 𑀘 𑀢𑀧𑀢𑁂 𑀘 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀘; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀑𑀧𑀥𑀺𑀓𑀸𑀦𑀺 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁, 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑁂𑀯 𑀢𑀸𑀦𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀸𑀲𑀢𑁂 𑀘 𑀢𑀧𑀢𑁂 𑀘 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀘𑁇

159\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀑𑀲𑀥𑀺𑀢𑀸𑀭𑀓𑀸 𑀪𑀸𑀲𑀢𑁂 𑀘 𑀢𑀧𑀢𑁂 𑀘 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀘; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀑𑀧𑀥𑀺𑀓𑀸𑀦𑀺 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁, 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑁂𑀯 𑀢𑀸𑀦𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀸𑀲𑀢𑁂 𑀘 𑀢𑀧𑀢𑁂 𑀘 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

160\. _“𑀬𑁄 𑀘 𑀫𑁂𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑀬𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀢𑀦𑀽 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀧𑀲𑁆𑀲𑀢𑁄 𑀉𑀧𑀥𑀺𑀓𑁆𑀔𑀬𑀁𑁇_  


161\. _𑀏𑀓𑀫𑁆𑀧𑀺 𑀘𑁂 𑀧𑀸𑀡𑀫𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀢𑀺 𑀓𑀼𑀲𑀮𑁄 𑀢𑁂𑀦 𑀳𑁄𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑁂 𑀘 𑀧𑀸𑀡𑁂 𑀫𑀦𑀲𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀁,_  
_𑀧𑀳𑀽𑀢𑀫𑀭𑀺𑀬𑁄 𑀧𑀓𑀭𑁄𑀢𑀺 𑀧𑀼𑀜𑁆𑀜𑀁𑁇_  


162\. _𑀬𑁂 𑀲𑀢𑁆𑀢𑀲𑀡𑁆𑀟𑀁 𑀧𑀣𑀯𑀺𑀁 𑀯𑀺𑀚𑀺𑀢𑁆𑀯𑀸,_  
_𑀭𑀸𑀚𑀺𑀲𑀬𑁄 𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀬𑀕𑀸;_  
_𑀅𑀲𑁆𑀲𑀫𑁂𑀥𑀁 𑀧𑀼𑀭𑀺𑀲𑀫𑁂𑀥𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀧𑀸𑀲𑀁 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀦𑀺𑀭𑀕𑁆𑀕𑀍𑀅𑀁𑁇_  


163\. _𑀫𑁂𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀲𑁆𑀲,_  
_𑀓𑀮𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦𑀸𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁;_  
_𑀘𑀦𑁆𑀤𑀧𑁆𑀧𑀪𑀸 𑀢𑀸𑀭𑀕𑀡𑀸𑀯 𑀲𑀩𑁆𑀩𑁂𑁇_  


164\. _𑀬𑁄 𑀦 𑀳𑀦𑁆𑀢𑀺 𑀦 𑀖𑀸𑀢𑁂𑀢𑀺,_  
_𑀦 𑀚𑀺𑀦𑀸𑀢𑀺 𑀦 𑀚𑀸𑀧𑀬𑁂;_  
_𑀫𑁂𑀢𑁆𑀢𑀁𑀲𑁄 𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑁂𑀲𑀼,_  
_𑀯𑁂𑀭𑀁 𑀢𑀲𑁆𑀲 𑀦 𑀓𑁂𑀦𑀘𑀻”𑀢𑀺𑁇_  


165\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

167\. 𑀢𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄𑁇

168\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



169\. _𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑁂𑀢𑁆𑀢𑀁 𑀉𑀪𑁄 𑀅𑀢𑁆𑀣𑁂,_  
_𑀧𑀼𑀜𑁆𑀚𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄,_  
_𑀤𑀸𑀦𑀜𑁆𑀘 𑀫𑁂𑀢𑁆𑀢𑀪𑀸𑀯𑀦𑀸𑁇_  


170\. _𑀲𑀢𑁆𑀢𑀺𑀫𑀸𑀦𑀺 𑀘 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀦𑀺 𑀘 𑀯𑀻𑀲𑀢𑀺;_  
_𑀏𑀓𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸,_  
_𑀲𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀲𑀗𑁆𑀕𑀳𑀸𑀢𑀺𑁇_
