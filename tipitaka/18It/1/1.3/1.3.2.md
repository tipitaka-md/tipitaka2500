# 𑁧𑁇𑁩𑁇𑁨 𑀫𑁂𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀫𑁂𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(22.)

120\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

121\. “𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀪𑀸𑀬𑀺𑀢𑁆𑀣𑁇 𑀲𑀼𑀔𑀲𑁆𑀲𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀇𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀬𑀤𑀺𑀤𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺𑁇 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀓𑀢𑀸𑀦𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀇𑀝𑁆𑀞𑀁 𑀓𑀦𑁆𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁 𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁𑁇 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢 𑀲𑀁𑀯𑀝𑁆𑀝𑀯𑀺𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂 𑀦𑀬𑀺𑀫𑀁 𑀮𑁄𑀓𑀁 𑀧𑀼𑀦𑀭𑀸𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀲𑀁𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑁂 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀧𑁆𑀧𑁂 𑀆𑀪𑀲𑁆𑀲𑀭𑀽𑀧𑀕𑁄 𑀳𑁄𑀫𑀺; 𑀯𑀺𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑁂 𑀓𑀧𑁆𑀧𑁂 𑀲𑀼𑀜𑁆𑀜𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀺𑀫𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺𑁇

122\. 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀳𑁄𑀫𑀺 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀅𑀪𑀺𑀪𑀽 𑀅𑀦𑀪𑀺𑀪𑀽𑀢𑁄 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼𑀤𑀲𑁄 𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑁇 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄; 𑀅𑀦𑁂𑀓𑀲𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀚𑀦𑀧𑀤𑀢𑁆𑀣𑀸𑀯𑀭𑀺𑀬𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀲𑁆𑀲𑁇

123\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀨𑀮𑀁, 𑀓𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀬𑁂𑀦𑀸𑀳𑀁 𑀏𑀢𑀭𑀳𑀺 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀢𑀺𑀡𑁆𑀡𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀬𑁂𑀦𑀸𑀳𑀁 𑀏𑀢𑀭𑀳𑀺 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀲𑁆𑀲, 𑀤𑀫𑀲𑁆𑀲, 𑀲𑀜𑁆𑀜𑀫𑀲𑁆𑀲𑀸’”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

124\. _“𑀧𑀼𑀜𑁆𑀜𑀫𑁂𑀯 𑀲𑁄 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬,_  
_𑀆𑀬𑀢𑀕𑁆𑀕𑀁 𑀲𑀼𑀔𑀼𑀤𑁆𑀭𑀬𑀁;_  
_𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑀘𑀭𑀺𑀬𑀜𑁆𑀘,_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀸𑀯𑀬𑁂𑁇_  


125\. _𑀏𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀪𑀸𑀯𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀢𑀬𑁄 𑀲𑀼𑀔𑀲𑀫𑀼𑀤𑁆𑀤𑀬𑁂;_  
_𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀁 𑀲𑀼𑀔𑀁 𑀮𑁄𑀓𑀁,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


126\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

127\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
