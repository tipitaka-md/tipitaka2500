# 𑁧𑁇𑁩𑁇𑁩 𑀉𑀪𑀬𑀢𑁆𑀣𑀲𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀉𑀪𑀬𑀢𑁆𑀣𑀲𑀼𑀢𑁆𑀢

(23.)

128\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

129\. “𑀏𑀓𑀥𑀫𑁆𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀉𑀪𑁄 𑀅𑀢𑁆𑀣𑁂 𑀲𑀫𑀥𑀺𑀕𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀅𑀢𑁆𑀣𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀜𑁆𑀘𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑀥𑀫𑁆𑀫𑁄? 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀉𑀪𑁄 𑀅𑀢𑁆𑀣𑁂 𑀲𑀫𑀥𑀺𑀕𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀢𑀺—  𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀅𑀢𑁆𑀣𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

130\. _“𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀁 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀸𑀲𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀉𑀪𑁄 𑀅𑀢𑁆𑀣𑁂,_  
_𑀅𑀥𑀺𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


131\. _𑀤𑀺𑀝𑁆𑀞𑁂 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀬𑁄 𑀅𑀢𑁆𑀣𑁄,_  
_𑀬𑁄 𑀘𑀢𑁆𑀣𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑁄;_  
_𑀅𑀢𑁆𑀣𑀸𑀪𑀺𑀲𑀫𑀬𑀸 𑀥𑀻𑀭𑁄,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


132\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

133\. 𑀢𑀢𑀺𑀬𑀁𑁇
