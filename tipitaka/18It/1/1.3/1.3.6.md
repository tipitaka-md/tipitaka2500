# 𑁧𑁇𑁩𑁇𑁬 𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(26.)

147. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

148. “𑀏𑀯𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀁 𑀬𑀣𑀸𑀳𑀁 𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦 𑀅𑀤𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀘 𑀦𑁂𑀲𑀁 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑁇 𑀬𑁄𑀧𑀺 𑀦𑁂𑀲𑀁 𑀅𑀲𑁆𑀲 𑀘𑀭𑀺𑀫𑁄 𑀆𑀮𑁄𑀧𑁄 𑀘𑀭𑀺𑀫𑀁 𑀓𑀩𑀍𑀅𑀁, 𑀢𑀢𑁄𑀧𑀺 𑀦 𑀅𑀲𑀁𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀘𑁂 𑀦𑁂𑀲𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀸 𑀅𑀲𑁆𑀲𑀼𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀦 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀁 𑀬𑀣𑀸𑀳𑀁 𑀚𑀸𑀦𑀸𑀫𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀅𑀤𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀜𑁆𑀘 𑀦𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

149. _“𑀏𑀯𑀜𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀬𑀣𑀸𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀸;_  
_𑀯𑀺𑀧𑀸𑀓𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀲𑁆𑀲,_  
_𑀬𑀣𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


150. _𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀤𑀚𑁆𑀚𑀼𑀁 𑀓𑀸𑀮𑁂𑀦 𑀅𑀭𑀺𑀬𑁂𑀲𑀼,_  
_𑀬𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


151. _𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀤𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀦𑁄,_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀇𑀢𑁄 𑀘𑀼𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀸,_  
_𑀲𑀕𑁆𑀕𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀸𑀬𑀓𑀸𑁇_  


152. _𑀢𑁂 𑀘 𑀲𑀕𑁆𑀕𑀕𑀢𑀸 𑀢𑀢𑁆𑀣,_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀓𑀸𑀫𑀺𑀦𑁄;_  
_𑀯𑀺𑀧𑀸𑀓𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀲𑁆𑀲,_  
_𑀅𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀺 𑀅𑀫𑀘𑁆𑀙𑀭𑀸”𑀢𑀺𑁇_  


153. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
