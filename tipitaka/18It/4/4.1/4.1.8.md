# 𑁪𑁇𑁧𑁇𑁮 𑀩𑀳𑀼𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑀯𑀕𑁆𑀕

* 𑀩𑀳𑀼𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(107.)

778\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

779\. “𑀩𑀳𑀼𑀓𑀸𑀭𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀬𑁂 𑀯𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺𑁇 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀼𑀓𑀸𑀭𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀬𑀁 𑀦𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀣 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁, 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀣𑁇 𑀏𑀯𑀫𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀺 𑀑𑀖𑀲𑁆𑀲 𑀦𑀺𑀢𑁆𑀣𑀭𑀡𑀢𑁆𑀣𑀸𑀬 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

780\. _“𑀲𑀸𑀕𑀸𑀭𑀸 𑀅𑀦𑀕𑀸𑀭𑀸 𑀘,_  
_𑀉𑀪𑁄 𑀅𑀜𑁆𑀜𑁄𑀜𑁆𑀜𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸;_  
_𑀆𑀭𑀸𑀥𑀬𑀦𑁆𑀢𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


781\. _𑀲𑀸𑀕𑀸𑀭𑁂𑀲𑀼 𑀘 𑀘𑀻𑀯𑀭𑀁,_  
_𑀧𑀘𑁆𑀘𑀬𑀁 𑀲𑀬𑀦𑀸𑀲𑀦𑀁;_  
_𑀅𑀦𑀕𑀸𑀭𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀧𑀭𑀺𑀲𑁆𑀲𑀬𑀯𑀺𑀦𑁄𑀤𑀦𑀁𑁇_  


782\. _𑀲𑀼𑀕𑀢𑀁 𑀧𑀦 𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀕𑀳𑀝𑁆𑀞𑀸 𑀖𑀭𑀫𑁂𑀲𑀺𑀦𑁄;_  
_𑀲𑀤𑁆𑀤𑀳𑀸𑀦𑀸 𑀅𑀭𑀳𑀢𑀁,_  
_𑀅𑀭𑀺𑀬𑀧𑀜𑁆𑀜𑀸𑀬 𑀛𑀸𑀬𑀺𑀦𑁄𑁇_  


783\. _𑀇𑀥 𑀥𑀫𑁆𑀫𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀕𑁆𑀕𑀁 𑀲𑀼𑀕𑀢𑀺𑀕𑀸𑀫𑀺𑀦𑀁;_  
_𑀦𑀦𑁆𑀤𑀺𑀦𑁄 𑀤𑁂𑀯𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀓𑀸𑀫𑀺𑀦𑁄”𑀢𑀺𑁇_  


784\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
