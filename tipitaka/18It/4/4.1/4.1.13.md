# 𑁪𑁇𑁧𑁇𑁧𑁩 𑀮𑁄𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑀯𑀕𑁆𑀕

### 𑀮𑁄𑀓𑀲𑀼𑀢𑁆𑀢

(112.)

832. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

833. “𑀮𑁄𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀮𑁄𑀓𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀮𑁄𑀓𑀲𑀫𑀼𑀤𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀮𑁄𑀓𑀲𑀫𑀼𑀤𑀬𑁄 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑁄𑁇 𑀮𑁄𑀓𑀦𑀺𑀭𑁄𑀥𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀮𑁄𑀓𑀦𑀺𑀭𑁄𑀥𑁄 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁄𑁇 𑀮𑁄𑀓𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀮𑁄𑀓𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀸𑀯𑀺𑀢𑀸𑁇

834. 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲 𑀲𑀫𑀸𑀭𑀓𑀲𑁆𑀲 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀲𑁆𑀲 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸 𑀬𑀲𑁆𑀫𑀸 𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

835. 𑀬𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺, 𑀬𑀁 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑁂 𑀪𑀸𑀲𑀢𑀺 𑀮𑀧𑀢𑀺 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀢𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀢𑀣𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀅𑀜𑁆𑀜𑀣𑀸, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

836. 𑀬𑀣𑀸𑀯𑀸𑀤𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀢𑀣𑀸𑀓𑀸𑀭𑀻, 𑀬𑀣𑀸𑀓𑀸𑀭𑀻 𑀢𑀣𑀸𑀯𑀸𑀤𑀻, 𑀇𑀢𑀺 𑀬𑀣𑀸𑀯𑀸𑀤𑀻 𑀢𑀣𑀸𑀓𑀸𑀭𑀻 𑀬𑀣𑀸𑀓𑀸𑀭𑀻 𑀢𑀣𑀸𑀯𑀸𑀤𑀻, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

837. 𑀲𑀤𑁂𑀯𑀓𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀪𑀺𑀪𑀽 𑀅𑀦𑀪𑀺𑀪𑀽𑀢𑁄 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼𑀤𑀲𑁄 𑀯𑀲𑀯𑀢𑁆𑀢𑀻, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

838. _“𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂 𑀅𑀦𑀽𑀧𑀬𑁄𑁇_  


839. _𑀲 𑀯𑁂 𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀪𑀽 𑀥𑀻𑀭𑁄,_  
_𑀲𑀩𑁆𑀩𑀕𑀦𑁆𑀣𑀧𑁆𑀧𑀫𑁄𑀘𑀦𑁄;_  
_𑀨𑀼𑀝𑁆𑀞𑀸𑀲𑁆𑀲 𑀧𑀭𑀫𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑀁𑁇_  


840. _𑀏𑀲 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀦𑀻𑀖𑁄 𑀙𑀺𑀦𑁆𑀦𑀲𑀁𑀲𑀬𑁄;_  
_𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀬𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀉𑀧𑀥𑀺𑀲𑀗𑁆𑀔𑀬𑁂𑁇_  


841. _𑀏𑀲 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀏𑀲 𑀲𑀻𑀳𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀬𑀺𑁇_  


842. _𑀇𑀢𑀺 𑀤𑁂𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘,_  
_𑀬𑁂 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸;_  
_𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁𑁇_  


843. _𑀤𑀦𑁆𑀢𑁄 𑀤𑀫𑀬𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀲𑀦𑁆𑀢𑁄 𑀲𑀫𑀬𑀢𑀁 𑀇𑀲𑀺;_  
_𑀫𑀼𑀢𑁆𑀢𑁄 𑀫𑁄𑀘𑀬𑀢𑀁 𑀅𑀕𑁆𑀕𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑁄 𑀢𑀸𑀭𑀬𑀢𑀁 𑀯𑀭𑁄𑁇_  


844. _𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀢𑁂 𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄”𑀢𑀺𑁇_  


845. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

846. 𑀢𑁂𑀭𑀲𑀫𑀁𑁇



---

847. 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

848. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



849. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀮𑀪𑀸 𑀚𑀸𑀦𑀁,_  
_𑀲𑀫𑀡𑀲𑀻𑀮𑀸 𑀢𑀡𑁆𑀳𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸;_  
_𑀩𑀳𑀼𑀓𑀸𑀭𑀸 𑀓𑀼𑀳𑀧𑀼𑀭𑀺𑀲𑀸,_  
_𑀘𑀭 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀮𑁄𑀓𑁂𑀦 𑀢𑁂𑀭𑀲𑀸𑀢𑀺𑁇_  


---

850. 𑀲𑀼𑀢𑁆𑀢𑀲𑀗𑁆𑀕𑀳𑁄



851. _𑀲𑀢𑁆𑀢𑀯𑀺𑀲𑁂𑀓𑀦𑀺𑀧𑀸𑀢𑀁,_  
_𑀤𑀼𑀓𑁆𑀓𑀁 𑀩𑀸𑀯𑀻𑀲𑀲𑀼𑀢𑁆𑀢𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁;_  
_𑀲𑀫𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀣𑀢𑀺𑀓𑀁,_  
_𑀢𑁂𑀭𑀲 𑀘𑀢𑀼𑀓𑁆𑀓𑀜𑁆𑀘 𑀇𑀢𑀺 𑀬𑀫𑀺𑀤𑀁𑁇_  


852. _𑀤𑁆𑀯𑀺𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢𑀲𑀢𑁂,_  
_𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀳𑀺𑀁𑀲𑀼 𑀧𑀼𑀭𑀸;_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸,_  
_𑀢𑀫𑀸𑀳𑀼 𑀦𑀸𑀫𑁂𑀦 𑀇𑀢𑀺𑀯𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇_  


---

853. 𑀇𑀢𑀺𑀯𑀼𑀢𑁆𑀢𑀓𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
