# 𑁪𑁇𑁧𑁇𑁫 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

(104.)

757\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

758\. “𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀑𑀯𑀸𑀤𑀓𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑀸 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑀸 𑀲𑀫𑀸𑀤𑀧𑀓𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑀸 𑀅𑀮𑀁𑀲𑀫𑀓𑁆𑀔𑀸𑀢𑀸𑀭𑁄 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀫𑁆𑀧𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀁 𑀯𑀤𑀸𑀫𑀺; 𑀲𑀯𑀦𑀫𑁆𑀧𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀁 𑀯𑀤𑀸𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑀫𑁆𑀧𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀁 𑀯𑀤𑀸𑀫𑀺; 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑀫𑁆𑀧𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀁 𑀯𑀤𑀸𑀫𑀺; 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀡𑀫𑁆𑀧𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀁 𑀯𑀤𑀸𑀫𑀺; 𑀅𑀦𑀼𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀫𑁆𑀧𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸𑀭𑀽𑀧𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑁂𑀯𑀢𑁄 𑀪𑀚𑀢𑁄 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑁄 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄𑀧𑀺 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄𑀧𑀺 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄𑀧𑀺 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄𑀧𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄𑀧𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑀸 𑀘 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀢𑁆𑀣𑀸𑀭𑁄𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀭𑀡𑀜𑁆𑀚𑀳𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀢𑀫𑁄𑀦𑀼𑀤𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀆𑀮𑁄𑀓𑀓𑀭𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀑𑀪𑀸𑀲𑀓𑀭𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀧𑀚𑁆𑀚𑁄𑀢𑀓𑀭𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀉𑀓𑁆𑀓𑀸𑀥𑀸𑀭𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀧𑀪𑀗𑁆𑀓𑀭𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀅𑀭𑀺𑀬𑀸𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄𑀢𑀺𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

759\. _“𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀓𑀭𑀡𑀁 𑀞𑀸𑀦𑀁,_  
_𑀏𑀢𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀚𑀸𑀦𑀢𑀁;_  
_𑀬𑀤𑀺𑀤𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀚𑀻𑀯𑀺𑀦𑀁𑁇_  


760\. _𑀢𑁂 𑀚𑁄𑀢𑀬𑀦𑁆𑀢𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀪𑀸𑀲𑀬𑀦𑁆𑀢𑀺 𑀧𑀪𑀗𑁆𑀓𑀭𑀸;_  
_𑀆𑀮𑁄𑀓𑀓𑀭𑀡𑀸 𑀥𑀻𑀭𑀸,_  
_𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀡𑀜𑁆𑀚𑀳𑀸𑁇_  


761\. _𑀬𑁂𑀲𑀁 𑀯𑁂 𑀲𑀸𑀲𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀚𑀸𑀢𑀺𑀓𑁆𑀔𑀬𑀫𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇_  


762\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
