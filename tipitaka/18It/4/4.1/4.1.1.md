# 𑁪𑁇𑁧𑁇𑁧 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(100.)

726. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

727. “𑀅𑀳𑀫𑀲𑁆𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀲𑀤𑀸 𑀧𑀬𑀢𑀧𑀸𑀡𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀤𑁂𑀳𑀥𑀭𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀪𑀺𑀲𑀓𑁆𑀓𑁄 𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀧𑀼𑀢𑁆𑀢𑀸 𑀑𑀭𑀲𑀸 𑀫𑀼𑀔𑀢𑁄 𑀚𑀸𑀢𑀸 𑀥𑀫𑁆𑀫𑀚𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸, 𑀦𑁄 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸𑁇

728. 𑀤𑁆𑀯𑁂𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀦𑀸𑀦𑀺—  𑀆𑀫𑀺𑀲𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀸𑀦𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀥𑀫𑁆𑀫𑀤𑀸𑀦𑀁𑁇

729. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀸—  𑀆𑀫𑀺𑀲𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑁄 𑀘 𑀥𑀫𑁆𑀫𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑁄 𑀘𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀥𑀫𑁆𑀫𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑁄𑁇

730. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀸—  𑀆𑀫𑀺𑀲𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑁄 𑀘𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑁄𑁇

731. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀕𑀸—  𑀆𑀫𑀺𑀲𑀬𑀸𑀕𑁄 𑀘 𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑁄 𑀘𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀬𑀸𑀕𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑁄”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

732. _“𑀬𑁄 𑀥𑀫𑁆𑀫𑀬𑀸𑀕𑀁 𑀅𑀬𑀚𑀻 𑀅𑀫𑀘𑁆𑀙𑀭𑀻,_  
_𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻;_  
_𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀲𑀢𑁆𑀢𑀸 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀪𑀯𑀲𑁆𑀲 𑀧𑀸𑀭𑀕𑀼𑀦𑁆”𑀢𑀺𑁇_  


733. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
