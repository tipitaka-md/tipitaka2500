# 𑁨𑁇𑁧𑁇𑁩 𑀢𑀧𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀢𑀧𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(30.)

186\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

187\. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀢𑀧𑀦𑀻𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀅𑀓𑀢𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀓𑀢𑀓𑀼𑀲𑀮𑁄, 𑀅𑀓𑀢𑀪𑀻𑀭𑀼𑀢𑁆𑀢𑀸𑀡𑁄, 𑀓𑀢𑀧𑀸𑀧𑁄, 𑀓𑀢𑀮𑀼𑀤𑁆𑀤𑁄, 𑀓𑀢𑀓𑀺𑀩𑁆𑀩𑀺𑀲𑁄𑁇 𑀲𑁄 ‘𑀅𑀓𑀢𑀁 𑀫𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀦𑁆’𑀢𑀺𑀧𑀺 𑀢𑀧𑁆𑀧𑀢𑀺, ‘𑀓𑀢𑀁 𑀫𑁂 𑀧𑀸𑀧𑀦𑁆’𑀢𑀺𑀧𑀺 𑀢𑀧𑁆𑀧𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀢𑀧𑀦𑀻𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

188\. _“𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀘;_  
_𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀬𑀜𑁆𑀘𑀜𑁆𑀜𑀁 𑀤𑁄𑀲𑀲𑀜𑁆𑀳𑀺𑀢𑀁𑁇_  


189\. _𑀅𑀓𑀢𑁆𑀯𑀸 𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀢𑁆𑀯𑀸𑀦𑀸𑀓𑀼𑀲𑀮𑀁 𑀩𑀳𑀼𑀁;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀦𑀺𑀭𑀬𑀁 𑀲𑁄𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


190\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
