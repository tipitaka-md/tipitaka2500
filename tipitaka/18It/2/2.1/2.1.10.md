# 𑁨𑁇𑁧𑁇𑁧𑁦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

(37.)

226\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

227\. “𑀤𑁆𑀯𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀔𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀩𑀳𑀼𑀮𑁄 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀬𑁄𑀦𑀺 𑀘𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑁂𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼 𑀲𑀁𑀯𑁂𑀚𑀦𑁂𑀦, 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀲𑁆𑀲 𑀘 𑀬𑁄𑀦𑀺𑀲𑁄 𑀧𑀥𑀸𑀦𑁂𑀦𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀔𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀩𑀳𑀼𑀮𑁄 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀬𑁄𑀦𑀺 𑀘𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

228\. _“𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼,_  
_𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑁂𑀣𑁂𑀯 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀆𑀢𑀸𑀧𑀻 𑀦𑀺𑀧𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀬𑁇_  


229\. _𑀏𑀯𑀁 𑀯𑀺𑀳𑀸𑀭𑀻 𑀆𑀢𑀸𑀧𑀻,_  
_𑀲𑀦𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄;_  
_𑀘𑁂𑀢𑁄𑀲𑀫𑀣𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀔𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀸𑀧𑀼𑀡𑁂”𑀢𑀺𑁇_  


230\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
