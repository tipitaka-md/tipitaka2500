# 𑁨𑁇𑁧𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑀚𑀦𑀦𑀓𑀼𑀳𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀚𑀦𑀦𑀓𑀼𑀳𑀦𑀲𑀼𑀢𑁆𑀢

(36.)

220\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

221\. “𑀦𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀺 𑀚𑀦𑀓𑀼𑀳𑀦𑀢𑁆𑀣𑀁, 𑀦 𑀚𑀦𑀮𑀧𑀦𑀢𑁆𑀣𑀁, 𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀸𑀦𑀺𑀲𑀁𑀲𑀢𑁆𑀣𑀁, 𑀦 ‘𑀇𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀺 𑀅𑀪𑀺𑀜𑁆𑀜𑀢𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀧𑀭𑀺𑀜𑁆𑀜𑀢𑁆𑀣𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

222\. _“𑀅𑀪𑀺𑀜𑁆𑀜𑀢𑁆𑀣𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀢𑁆𑀣𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀦𑀻𑀢𑀺𑀳𑀁;_  
_𑀅𑀤𑁂𑀲𑀬𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑁄𑀕𑀥𑀕𑀸𑀫𑀺𑀦𑀁𑁇_  


223\. _𑀏𑀲 𑀫𑀕𑁆𑀕𑁄 𑀫𑀳𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀅𑀦𑀼𑀬𑀸𑀢𑁄 𑀫𑀳𑁂𑀲𑀺𑀪𑀺;_  
_𑀬𑁂 𑀬𑁂 𑀢𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀬𑀣𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀦𑁄”𑀢𑀺𑁇_  


224\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
