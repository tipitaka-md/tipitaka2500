# 𑁨𑁇𑁧𑁇𑁭 𑀆𑀢𑀸𑀧𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀆𑀢𑀸𑀧𑀻𑀲𑀼𑀢𑁆𑀢

(34.)

208\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

209\. “𑀅𑀦𑀸𑀢𑀸𑀧𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁄𑀢𑁆𑀢𑀸𑀧𑀻 𑀅𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀆𑀢𑀸𑀧𑀻 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀑𑀢𑁆𑀢𑀸𑀧𑀻 𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬, 𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬, 𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

210\. _“𑀅𑀦𑀸𑀢𑀸𑀧𑀻 𑀅𑀦𑁄𑀢𑁆𑀢𑀸𑀧𑀻,_  
_𑀓𑀼𑀲𑀻𑀢𑁄 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑁄;_  
_𑀬𑁄 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀩𑀳𑀼𑀮𑁄,_  
_𑀅𑀳𑀺𑀭𑀻𑀓𑁄 𑀅𑀦𑀸𑀤𑀭𑁄;_  
_𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀨𑀼𑀝𑁆𑀞𑀼𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


211\. _𑀬𑁄 𑀘 𑀲𑀢𑀺𑀫𑀸 𑀦𑀺𑀧𑀓𑁄 𑀛𑀸𑀬𑀻,_  
_𑀆𑀢𑀸𑀧𑀻 𑀑𑀢𑁆𑀢𑀸𑀧𑀻 𑀘 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄;_  
_𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀬 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀥𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀨𑀼𑀲𑁂”𑀢𑀺𑁇_  


212\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---

213\. 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
