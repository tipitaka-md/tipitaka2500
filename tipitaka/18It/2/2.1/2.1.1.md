# 𑁨𑁇𑁧𑁇𑁧 𑀤𑀼𑀓𑁆𑀔𑀯𑀺𑀳𑀸𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀓𑁆𑀔𑀯𑀺𑀳𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(28.)

172\. ( ) 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

173\. “𑀤𑁆𑀯𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀯𑀺𑀖𑀸𑀢𑀁 𑀲𑁅𑀧𑀸𑀬𑀸𑀲𑀁 𑀲𑀧𑀭𑀺𑀍𑀆𑀳𑀁; 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸𑀬 𑀘, 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀘𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀯𑀺𑀖𑀸𑀢𑀁 𑀲𑁅𑀧𑀸𑀬𑀸𑀲𑀁 𑀲𑀧𑀭𑀺𑀍𑀆𑀳𑀁; 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

174\. _“𑀘𑀓𑁆𑀔𑀼 𑀲𑁄𑀢𑀜𑁆𑀘 𑀖𑀸𑀦𑀜𑁆𑀘,_  
_𑀚𑀺𑀯𑁆𑀳𑀸 𑀓𑀸𑀬𑁄 𑀢𑀣𑀸 𑀫𑀦𑁄;_  
_𑀏𑀢𑀸𑀦𑀺 𑀬𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀭𑀸𑀦𑀺,_  
_𑀅𑀕𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


175\. _𑀪𑁄𑀚𑀦𑀫𑁆𑀳𑀺 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀼𑀢𑁄;_  
_𑀓𑀸𑀬𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑁄𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑁄 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


176\. _𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁂𑀦 𑀓𑀸𑀬𑁂𑀦,_  
_𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀤𑀺𑀯𑀸 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑀸𑀤𑀺𑀲𑁄”𑀢𑀺𑁇_  


177\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
