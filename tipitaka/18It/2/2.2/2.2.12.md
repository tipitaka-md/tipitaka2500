# 𑁨𑁇𑁨𑁇𑁧𑁨 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀼𑀢𑁆𑀢

(49.)

313\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

314\. “𑀤𑁆𑀯𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀑𑀮𑀻𑀬𑀦𑁆𑀢𑀺 𑀏𑀓𑁂, 𑀅𑀢𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀏𑀓𑁂; 𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀘 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

315\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀮𑀻𑀬𑀦𑁆𑀢𑀺 𑀏𑀓𑁂? 𑀪𑀯𑀸𑀭𑀸𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀯𑀭𑀢𑀸 𑀪𑀯𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸 𑀢𑁂𑀲𑀁 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀮𑀻𑀬𑀦𑁆𑀢𑀺 𑀏𑀓𑁂𑁇

316\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀏𑀓𑁂? 𑀪𑀯𑁂𑀦𑁂𑀯 𑀔𑁄 𑀧𑀦𑁂𑀓𑁂 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑀸 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑀸 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑀸 𑀯𑀺𑀪𑀯𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺—  𑀬𑀢𑁄 𑀓𑀺𑀭, 𑀪𑁄, 𑀅𑀬𑀁 𑀅𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀉𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸; 𑀏𑀢𑀁 𑀲𑀦𑁆𑀢𑀁 𑀏𑀢𑀁 𑀧𑀡𑀻𑀢𑀁 𑀏𑀢𑀁 𑀬𑀸𑀣𑀸𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀏𑀓𑁂𑁇

317\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀽𑀢𑀁 𑀪𑀽𑀢𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺; 𑀪𑀽𑀢𑀁 𑀪𑀽𑀢𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀽𑀢𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

318\. _“𑀬𑁂 𑀪𑀽𑀢𑀁 𑀪𑀽𑀢𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀪𑀽𑀢𑀲𑁆𑀲 𑀘 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀁;_  
_𑀬𑀣𑀸𑀪𑀽𑀢𑁂 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑁇_  


319\. _𑀲 𑀯𑁂 𑀪𑀽𑀢𑀧𑀭𑀺𑀜𑁆𑀜𑁄 𑀲𑁄,_  
_𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑁄 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀪𑀽𑀢𑀲𑁆𑀲 𑀯𑀺𑀪𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀦𑀸𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇_  


320\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

322\. 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄𑁇

323\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



324\. _𑀤𑁆𑀯𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀢𑀧𑀦𑀻𑀬𑀸,_  
_𑀲𑀻𑀮𑁂𑀦 𑀅𑀧𑀭𑁂 𑀤𑀼𑀯𑁂;_  
_𑀅𑀦𑁄𑀢𑁆𑀢𑀸𑀧𑀻 𑀓𑀼𑀳𑀦𑀸 𑀤𑁆𑀯𑁂 𑀘,_  
_𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑁂𑀦 𑀢𑁂 𑀤𑀲𑁇_  


325\. _𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀤𑁂𑀲𑀦𑀸 𑀯𑀺𑀚𑁆𑀚𑀸,_  
_𑀧𑀜𑁆𑀜𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀁;_  
_𑀅𑀚𑀸𑀢𑀁 𑀥𑀸𑀢𑀼𑀲𑀮𑁆𑀮𑀸𑀦𑀁,_  
_𑀲𑀺𑀓𑁆𑀔𑀸 𑀚𑀸𑀕𑀭𑀺𑀬𑁂𑀦 𑀘;_  
_𑀅𑀧𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀘𑁂𑀯,_  
_𑀩𑀸𑀯𑀻𑀲𑀢𑀺 𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑀢𑀺𑁇_
