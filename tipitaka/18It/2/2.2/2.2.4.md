# 𑁨𑁇𑁨𑁇𑁪 𑀧𑀜𑁆𑀜𑀸𑀧𑀭𑀺𑀳𑀻𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀜𑁆𑀜𑀸𑀧𑀭𑀺𑀳𑀻𑀦𑀲𑀼𑀢𑁆𑀢

(41.)

259. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

260. “𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀼𑀧𑀭𑀺𑀳𑀻𑀦𑀸 𑀬𑁂 𑀅𑀭𑀺𑀬𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀳𑀻𑀦𑀸𑁇 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀲𑀯𑀺𑀖𑀸𑀢𑀁 𑀲𑁅𑀧𑀸𑀬𑀸𑀲𑀁 𑀲𑀧𑀭𑀺𑀍𑀆𑀳𑀁; 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀺𑀳𑀻𑀦𑀸 𑀬𑁂 𑀅𑀭𑀺𑀬𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑀭𑀺𑀳𑀻𑀦𑀸𑁇 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀔𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀅𑀯𑀺𑀖𑀸𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀬𑀸𑀲𑀁 𑀅𑀧𑀭𑀺𑀍𑀆𑀳𑀁; 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

261. _“𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀳𑀸𑀦𑁂𑀦,_  
_𑀧𑀲𑁆𑀲 𑀮𑁄𑀓𑀁 𑀲𑀤𑁂𑀯𑀓𑀁;_  
_𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁,_  
_𑀇𑀤𑀁 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑀢𑀺𑁇_  


262. _𑀧𑀜𑁆𑀜𑀸 𑀳𑀺 𑀲𑁂𑀝𑁆𑀞𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑀸𑀬𑀁 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀕𑀸𑀫𑀺𑀦𑀻;_  
_𑀬𑀸𑀬 𑀲𑀫𑁆𑀫𑀸 𑀧𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀚𑀸𑀢𑀺𑀪𑀯𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁𑁇_  


263. _𑀢𑁂𑀲𑀁 𑀤𑁂𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀢𑀻𑀫𑀢𑀁;_  
_𑀧𑀺𑀳𑀬𑀦𑁆𑀢𑀺 𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑀸𑀦𑀁,_  
_𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺𑀫𑀥𑀸𑀭𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_  


264. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
