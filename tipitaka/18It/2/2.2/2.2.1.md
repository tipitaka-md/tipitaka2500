# 𑁨𑁇𑁨𑁇𑁧 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(38.)

235\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

236\. “𑀢𑀣𑀸𑀕𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀩𑀳𑀼𑀮𑀁 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺—  𑀔𑁂𑀫𑁄 𑀘 𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀧𑀯𑀺𑀯𑁂𑀓𑁄 𑀘𑁇 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀭𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑀁 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀭𑀢𑀁 𑀏𑀲𑁂𑀯 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀩𑀳𑀼𑀮𑀁 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀺—  ‘𑀇𑀫𑀸𑀬𑀸𑀳𑀁 𑀇𑀭𑀺𑀬𑀸𑀬 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑁂𑀫𑀺 𑀢𑀲𑀁 𑀯𑀸 𑀣𑀸𑀯𑀭𑀁 𑀯𑀸’𑀢𑀺𑁇

237\. 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀭𑀸𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀯𑀺𑀯𑁂𑀓𑀭𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀭𑀸𑀫𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀭𑀢𑀁 𑀏𑀲𑁂𑀯 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀩𑀳𑀼𑀮𑀁 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀺—  ‘𑀬𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀢𑀁 𑀧𑀳𑀻𑀦𑀦𑁆’𑀢𑀺𑁇

238\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑀸 𑀯𑀺𑀳𑀭𑀣 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀭𑀢𑀸𑁇 𑀢𑁂𑀲𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑀭𑀢𑀸𑀦𑀁 𑀏𑀲𑁂𑀯 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀩𑀳𑀼𑀮𑀁 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺—  ‘𑀇𑀫𑀸𑀬 𑀫𑀬𑀁 𑀇𑀭𑀺𑀬𑀸𑀬 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑁂𑀫 𑀢𑀲𑀁 𑀯𑀸 𑀣𑀸𑀯𑀭𑀁 𑀯𑀸’𑀢𑀺𑁇

239\. 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀭𑀸𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀳𑀭𑀣 𑀧𑀯𑀺𑀯𑁂𑀓𑀭𑀢𑀸𑁇 𑀢𑁂𑀲𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀭𑀸𑀫𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀭𑀢𑀸𑀦𑀁 𑀏𑀲𑁂𑀯 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀩𑀳𑀼𑀮𑀁 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺—  ‘𑀓𑀺𑀁 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀓𑀺𑀁 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀁, 𑀓𑀺𑀁 𑀧𑀚𑀳𑀸𑀫𑀸’”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

240\. _“𑀢𑀣𑀸𑀕𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀫𑀲𑀬𑁆𑀳𑀲𑀸𑀳𑀺𑀦𑀁,_  
_𑀤𑀼𑀯𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀦𑀁;_  
_𑀔𑁂𑀫𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀧𑀞𑀫𑁄 𑀉𑀤𑀻𑀭𑀺𑀢𑁄,_  
_𑀢𑀢𑁄 𑀯𑀺𑀯𑁂𑀓𑁄 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇_  


241\. _𑀢𑀫𑁄𑀦𑀼𑀤𑀁 𑀧𑀸𑀭𑀕𑀢𑀁 𑀫𑀳𑁂𑀲𑀺𑀁,_  
_𑀢𑀁 𑀧𑀢𑁆𑀢𑀺𑀧𑀢𑁆𑀢𑀁 𑀯𑀲𑀺𑀫𑀁 𑀅𑀦𑀸𑀲𑀯𑀁;_  
_𑀯𑀺𑀲𑀦𑁆𑀢𑀭𑀁 𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀁 𑀅𑀦𑁆𑀢𑀺𑀫𑀤𑁂𑀳𑀥𑀸𑀭𑀺𑀁;_  
_𑀫𑀸𑀭𑀜𑁆𑀚𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀚𑀭𑀸𑀬 𑀧𑀸𑀭𑀕𑀼𑀁𑁇_  


242\. _𑀲𑁂𑀮𑁂 𑀬𑀣𑀸 𑀧𑀩𑁆𑀩𑀢𑀫𑀼𑀤𑁆𑀥𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄,_  
_𑀬𑀣𑀸𑀧𑀺 𑀧𑀲𑁆𑀲𑁂 𑀚𑀦𑀢𑀁 𑀲𑀫𑀦𑁆𑀢𑀢𑁄;_  
_𑀢𑀣𑀽𑀧𑀫𑀁 𑀥𑀫𑁆𑀫𑀫𑀬𑀁 𑀲𑀼𑀫𑁂𑀥𑁄,_  
_𑀧𑀸𑀲𑀸𑀤𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀲𑀫𑀦𑁆𑀢𑀘𑀓𑁆𑀔𑀼;_  
_𑀲𑁄𑀓𑀸𑀯𑀢𑀺𑀡𑁆𑀡𑀁 𑀚𑀦𑀢𑀫𑀧𑁂𑀢𑀲𑁄𑀓𑁄,_  
_𑀅𑀯𑁂𑀓𑁆𑀔𑀢𑀺 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀦𑁆”𑀢𑀺𑁇_  


243\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
