# 𑁨𑁇𑁨𑁇𑁧𑁦 𑀚𑀸𑀕𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀚𑀸𑀕𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(47.)

299. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

300. “𑀚𑀸𑀕𑀭𑁄 𑀘𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀧𑀫𑀼𑀤𑀺𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀘 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀚𑀸𑀕𑀭𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀨𑀮𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀨𑀮𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀸, 𑀲𑀢𑀺 𑀯𑀸 𑀉𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑁂 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀢𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

301. _“𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑀸 𑀲𑀼𑀡𑀸𑀣𑁂𑀢𑀁,_  
_𑀬𑁂 𑀲𑀼𑀢𑁆𑀢𑀸 𑀢𑁂 𑀧𑀩𑀼𑀚𑁆𑀛𑀣;_  
_𑀲𑀼𑀢𑁆𑀢𑀸 𑀚𑀸𑀕𑀭𑀺𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀚𑀸𑀕𑀭𑀢𑁄 𑀪𑀬𑀁𑁇_  


302. _𑀬𑁄 𑀚𑀸𑀕𑀭𑁄 𑀘 𑀲𑀢𑀺𑀫𑀸 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄,_  
_𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀫𑀼𑀤𑀺𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀘;_  
_𑀓𑀸𑀮𑁂𑀦 𑀲𑁄 𑀲𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄,_  
_𑀏𑀓𑁄𑀤𑀺𑀪𑀽𑀢𑁄 𑀯𑀺𑀳𑀦𑁂 𑀢𑀫𑀁 𑀲𑁄𑁇_  


303. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀯𑁂 𑀚𑀸𑀕𑀭𑀺𑀬𑀁 𑀪𑀚𑁂𑀣,_  
_𑀆𑀢𑀸𑀧𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀧𑀓𑁄 𑀛𑀸𑀦𑀮𑀸𑀪𑀻;_  
_𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀬 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀥𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀨𑀼𑀲𑁂”𑀢𑀺𑁇_  


304. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
