# 𑁨𑁇𑁨𑁇𑁩 𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀼𑀢𑁆𑀢

(40.)

251\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

252\. “𑀅𑀯𑀺𑀚𑁆𑀚𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑁆𑀯𑀤𑁂𑀯 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁; 𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑁆𑀯𑀤𑁂𑀯 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀦𑁆”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

253\. _“𑀬𑀸 𑀓𑀸𑀘𑀺𑀫𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀫𑁆𑀳𑀺 𑀘;_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀫𑀽𑀮𑀺𑀓𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀇𑀘𑁆𑀙𑀸𑀮𑁄𑀪𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀸𑁇_  


254\. _𑀬𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄,_  
_𑀅𑀳𑀺𑀭𑀻𑀓𑁄 𑀅𑀦𑀸𑀤𑀭𑁄;_  
_𑀢𑀢𑁄 𑀧𑀸𑀧𑀁 𑀧𑀲𑀯𑀢𑀺,_  
_𑀅𑀧𑀸𑀬𑀁 𑀢𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


255\. _𑀢𑀲𑁆𑀫𑀸 𑀙𑀦𑁆𑀤𑀜𑁆𑀘 𑀮𑁄𑀪𑀜𑁆𑀘,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀜𑁆𑀘 𑀯𑀺𑀭𑀸𑀚𑀬𑀁;_  
_𑀯𑀺𑀚𑁆𑀚𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀩𑁆𑀩𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄 𑀚𑀳𑁂”𑀢𑀺𑁇_  


256\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
