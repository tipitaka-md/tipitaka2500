# 𑁨𑁇𑁨𑁇𑁧𑁧 𑀆𑀧𑀸𑀬𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀆𑀧𑀸𑀬𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(48.)

306\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

307\. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸 𑀇𑀤𑀫𑀧𑁆𑀧𑀳𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀬𑁄 𑀘 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀧𑀝𑀺𑀜𑁆𑀜𑁄, 𑀬𑁄 𑀘 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸 𑀇𑀤𑀫𑀧𑁆𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

308\. _“𑀅𑀪𑀽𑀢𑀯𑀸𑀤𑀻 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑁂𑀢𑀺,_  
_𑀬𑁄 𑀯𑀸𑀧𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦 𑀓𑀭𑁄𑀫𑀺 𑀘𑀸𑀳;_  
_𑀉𑀪𑁄𑀧𑀺 𑀢𑁂 𑀧𑁂𑀘𑁆𑀘 𑀲𑀫𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀳𑀻𑀦𑀓𑀫𑁆𑀫𑀸 𑀫𑀦𑀼𑀚𑀸 𑀧𑀭𑀢𑁆𑀣𑁇_  


309\. _𑀓𑀸𑀲𑀸𑀯𑀓𑀡𑁆𑀞𑀸 𑀩𑀳𑀯𑁄,_  
_𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀜𑁆𑀜𑀢𑀸;_  
_𑀧𑀸𑀧𑀸 𑀧𑀸𑀧𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀳𑀺,_  
_𑀦𑀺𑀭𑀬𑀁 𑀢𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀭𑁂𑁇_  


310\. _𑀲𑁂𑀬𑁆𑀬𑁄 𑀅𑀬𑁄𑀕𑀼𑀍𑀑 𑀪𑀼𑀢𑁆𑀢𑁄,_  
_𑀢𑀢𑁆𑀢𑁄 𑀅𑀕𑁆𑀕𑀺𑀲𑀺𑀔𑀽𑀧𑀫𑁄;_  
_𑀬𑀜𑁆𑀘𑁂 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄,_  
_𑀭𑀝𑁆𑀞𑀧𑀺𑀡𑁆𑀟𑀫𑀲𑀜𑁆𑀜𑀢𑁄”𑀢𑀺𑁇_  


311\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
