# 𑁨𑁇𑁨𑁇𑁮 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(45.)

287\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

288\. “𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀭𑀸𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀳𑀭𑀣 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀭𑀢𑀸, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀣𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀅𑀦𑀺𑀭𑀸𑀓𑀢𑀚𑁆𑀛𑀸𑀦𑀸, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀩𑁆𑀭𑀽𑀳𑁂𑀢𑀸𑀭𑁄 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀸𑀦𑀁𑁇 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀭𑀸𑀫𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀳𑀭𑀢𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀭𑀢𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀣𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀦𑀺𑀭𑀸𑀓𑀢𑀚𑁆𑀛𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀽𑀳𑁂𑀢𑀸𑀦𑀁 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀸𑀦𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀨𑀮𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀨𑀮𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀸, 𑀲𑀢𑀺 𑀯𑀸 𑀉𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑁂 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀢𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

289\. _“𑀬𑁂 𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀦𑀺𑀧𑀓𑀸,_  
_𑀲𑀢𑀺𑀫𑀦𑁆𑀢𑁄 𑀘 𑀛𑀸𑀬𑀺𑀦𑁄;_  
_𑀲𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁄𑁇_  


290\. _𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀭𑀢𑀸 𑀲𑀦𑁆𑀢𑀸,_  
_𑀧𑀫𑀸𑀤𑁂 𑀪𑀬𑀤𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀅𑀪𑀩𑁆𑀩𑀸 𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲𑁂𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑀢𑀺𑁇_  


291\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇

---
