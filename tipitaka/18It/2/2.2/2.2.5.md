# 𑁨𑁇𑁨𑁇𑁫 𑀲𑀼𑀓𑁆𑀓𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀓𑁆𑀓𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(42.)

266\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

267\. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀓𑁆𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀁 𑀧𑀸𑀮𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀳𑀺𑀭𑀻 𑀘, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘𑁇 𑀇𑀫𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀲𑀼𑀓𑁆𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀁 𑀦 𑀧𑀸𑀮𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦𑀬𑀺𑀥 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣 𑀫𑀸𑀢𑀸𑀢𑀺 𑀯𑀸 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀯𑀸 𑀫𑀸𑀢𑀼𑀮𑀸𑀦𑀻𑀢𑀺 𑀯𑀸 𑀆𑀘𑀭𑀺𑀬𑀪𑀭𑀺𑀬𑀸𑀢𑀺 𑀯𑀸 𑀕𑀭𑀽𑀦𑀁 𑀤𑀸𑀭𑀸𑀢𑀺 𑀯𑀸𑁇 𑀲𑀫𑁆𑀪𑁂𑀤𑀁 𑀮𑁄𑀓𑁄 𑀅𑀕𑀫𑀺𑀲𑁆𑀲 𑀬𑀣𑀸 𑀅𑀚𑁂𑀍𑀅𑀓𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀸 𑀲𑁄𑀡𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀸𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀲𑀼𑀓𑁆𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀁 𑀧𑀸𑀮𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀫𑀸𑀢𑀸𑀢𑀺 𑀯𑀸 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀯𑀸 𑀫𑀸𑀢𑀼𑀮𑀸𑀦𑀻𑀢𑀺 𑀯𑀸 𑀆𑀘𑀭𑀺𑀬𑀪𑀭𑀺𑀬𑀸𑀢𑀺 𑀯𑀸 𑀕𑀭𑀽𑀦𑀁 𑀤𑀸𑀭𑀸𑀢𑀺 𑀯𑀸”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

268\. _“𑀬𑁂𑀲𑀜𑁆𑀘𑁂 𑀳𑀺𑀭𑀺𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁,_  
_𑀲𑀩𑁆𑀩𑀤𑀸 𑀘 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀯𑁄𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀸 𑀢𑁂,_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀕𑀸𑀫𑀺𑀦𑁄𑁇_  


269\. _𑀬𑁂𑀲𑀜𑁆𑀘 𑀳𑀺𑀭𑀺𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁,_  
_𑀲𑀤𑀸 𑀲𑀫𑁆𑀫𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀯𑀺𑀭𑀽𑀍𑀳𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀢𑁂,_  
_𑀲𑀦𑁆𑀢𑁄 𑀔𑀻𑀡𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸”𑀢𑀺𑁇_  


270\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
