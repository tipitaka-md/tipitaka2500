# 𑁨𑁇𑁨𑁇𑁨 𑀤𑁂𑀲𑀦𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀤𑁂𑀲𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(39.)

245\. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

246\. “𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀤𑁆𑀯𑁂? ‘𑀧𑀸𑀧𑀁 𑀧𑀸𑀧𑀓𑀢𑁄 𑀧𑀲𑁆𑀲𑀣𑀸’𑀢𑀺—  𑀅𑀬𑀁 𑀧𑀞𑀫𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸; ‘𑀧𑀸𑀧𑀁 𑀧𑀸𑀧𑀓𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀢𑀢𑁆𑀣 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀣 𑀯𑀺𑀭𑀚𑁆𑀚𑀣 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀣𑀸’𑀢𑀺—  𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀇𑀫𑀸 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀪𑀯𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

247\. _“𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄;_  
_𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀘𑀦𑀁 𑀧𑀲𑁆𑀲,_  
_𑀤𑁆𑀯𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑁇_  


248\. _𑀧𑀸𑀧𑀓𑀁 𑀧𑀲𑁆𑀲𑀣 𑀘𑁂𑀢𑀁,_  
_𑀢𑀢𑁆𑀣 𑀘𑀸𑀧𑀺 𑀯𑀺𑀭𑀚𑁆𑀚𑀣;_  
_𑀢𑀢𑁄 𑀯𑀺𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀲𑁂,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀣𑀸”𑀢𑀺𑁇_  


249\. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
