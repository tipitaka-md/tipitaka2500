# 𑁨𑁇𑁨𑁇𑁬 𑀅𑀚𑀸𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀚𑀸𑀢𑀲𑀼𑀢𑁆𑀢

(43.)

272. 𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀫𑀭𑀳𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁—

273. “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁𑁇 𑀦𑁄 𑀘𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀯𑀺𑀲𑁆𑀲 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁, 𑀦𑀬𑀺𑀥 𑀚𑀸𑀢𑀲𑁆𑀲 𑀪𑀽𑀢𑀲𑁆𑀲 𑀓𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁, 𑀢𑀲𑁆𑀫𑀸 𑀚𑀸𑀢𑀲𑁆𑀲 𑀪𑀽𑀢𑀲𑁆𑀲 𑀓𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀻”𑀢𑀺𑁇 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 𑀢𑀢𑁆𑀣𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—

274. _“𑀚𑀸𑀢𑀁 𑀪𑀽𑀢𑀁 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀓𑀢𑀁 𑀲𑀗𑁆𑀔𑀢𑀫𑀤𑁆𑀥𑀼𑀯𑀁;_  
_𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀗𑁆𑀖𑀸𑀝𑀁,_  
_𑀭𑁄𑀕𑀦𑀻𑀍𑀅𑀁 𑀧𑀪𑀗𑁆𑀕𑀼𑀭𑀁𑁇_  


275. _𑀆𑀳𑀸𑀭𑀦𑁂𑀢𑁆𑀢𑀺𑀧𑁆𑀧𑀪𑀯𑀁,_  
_𑀦𑀸𑀮𑀁 𑀢𑀤𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀼𑀁;_  
_𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀅𑀢𑀓𑁆𑀓𑀸𑀯𑀘𑀭𑀁 𑀥𑀼𑀯𑀁𑁇_  


276. _𑀅𑀚𑀸𑀢𑀁 𑀅𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀅𑀲𑁄𑀓𑀁 𑀯𑀺𑀭𑀚𑀁 𑀧𑀤𑀁;_  
_𑀦𑀺𑀭𑁄𑀥𑁄 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀲𑀗𑁆𑀔𑀸𑀭𑀽𑀧𑀲𑀫𑁄 𑀲𑀼𑀔𑁄”𑀢𑀺𑁇_  


277. 𑀅𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀦𑁆𑀢𑀺𑁇
