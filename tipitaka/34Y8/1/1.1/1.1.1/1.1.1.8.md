# 𑁧𑁇𑁧𑁇𑁧𑁇𑁮 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

(14.)

54\. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁?

55\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁?

56\. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁?

57\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁?
