# 𑁧𑁇𑁧𑁇𑁧𑁇𑁪 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

(4.)

14. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣?

15. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺?

16. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣?

17. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺?

(5.)

18. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺?

19. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺?

20. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺?

21. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺?

(6.)

22. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺?

23. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣?

24. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺?

25. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣?
