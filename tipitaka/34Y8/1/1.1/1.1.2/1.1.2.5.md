# 𑁧𑁇𑁧𑁇𑁨𑁇𑁫 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

### 𑀥𑀫𑁆𑀫𑀯𑀸𑀭

### 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

(27.)

110\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣?

111\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺?

112\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣?

113\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺?

(28.)

114\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺?

115\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺?

116\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺?

117\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺?

(29.)

118\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺?

119\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣?

120\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺?

121\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣?
