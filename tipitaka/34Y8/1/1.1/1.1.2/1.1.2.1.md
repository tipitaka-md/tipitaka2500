# 𑁧𑁇𑁧𑁇𑁨𑁇𑁧 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀓𑀸𑀮𑀲𑀫𑁆𑀪𑁂𑀤𑀯𑀸𑀭

* 𑀉𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

* 𑀥𑀫𑁆𑀫𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀓𑀸𑀮𑀲𑀫𑁆𑀪𑁂𑀤𑀯𑀸𑀭

(21.)

86\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺?

87\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺?

88\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺?

89\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺?
