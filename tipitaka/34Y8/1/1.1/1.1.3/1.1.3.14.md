# 𑁧𑁇𑁧𑁇𑁩𑁇𑁧𑁪 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀯𑀸𑀭

* 𑀉𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

* 𑀧𑀼𑀕𑁆𑀕𑀮𑀥𑀫𑁆𑀫𑀯𑀸𑀭

* 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀯𑀸𑀭

(60.)

250\. »  𑀬𑀲𑁆𑀲 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁?

251\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁?

252\. »  𑀬𑀲𑁆𑀲 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁?

253\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁?
