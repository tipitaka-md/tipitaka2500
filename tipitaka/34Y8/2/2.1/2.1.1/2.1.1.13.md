# 𑁨𑁇𑁧𑁇𑁧𑁇𑁧𑁩 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀯𑀸𑀭

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀸𑀭

### 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀯𑀸𑀭

(82.)

366\. »  𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀦𑁆𑀢𑀺? 𑀦𑁄𑁇

367\. «  𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀦𑁆𑀢𑀺? 𑀦𑁄𑁇

368\. »  𑀦 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀦𑁆𑀢𑀺?

369\. 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀦 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀁, 𑀦𑁄 𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁𑁇 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀅𑀦𑀸𑀕𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑁂𑀯 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀁 𑀦 𑀘 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁𑁇

370\. «  𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀦𑁆𑀢𑀺?

371\. 𑀅𑀢𑀻𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁, 𑀦𑁄 𑀘 𑀦 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀁𑁇 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀅𑀦𑀸𑀕𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑁂𑀯 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀦 𑀘 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀁𑁇
