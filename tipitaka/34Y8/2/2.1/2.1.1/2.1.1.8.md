# 𑁨𑁇𑁧𑁇𑁧𑁇𑁮 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

* 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

(76.)

333\. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

334\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀦𑁆𑀢𑀺?

335\. 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀜𑁆𑀘𑁇

336\. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺?

337\. 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑁂𑀯 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀦 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇

338\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
