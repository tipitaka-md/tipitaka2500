# 𑁨𑁇𑁧𑁇𑁧𑁇𑁧𑁪 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀯𑀸𑀭

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀸𑀭

### 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀯𑀸𑀭

(83.)

372. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆𑀢𑀺?

373. 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑀁 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀅𑀢𑀻𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑀜𑁆𑀘 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑀜𑁆𑀘 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁𑁇

374. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆𑀢𑀺?

375. 𑀅𑀢𑀻𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

376. »  𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆𑀢𑀺?

377. 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀅𑀦𑀸𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

378. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁, 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀔𑀡𑀁 𑀔𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀸𑀮𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆𑀢𑀺?

379. 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑀁 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀦𑁄 𑀘 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑀁 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀅𑀦𑀸𑀕𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑀜𑁆𑀘 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑀜𑁆𑀘 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁𑁇
