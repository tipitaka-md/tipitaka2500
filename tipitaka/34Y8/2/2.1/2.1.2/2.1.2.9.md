# 𑁨𑁇𑁧𑁇𑁨𑁇𑁯 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

### 𑀥𑀫𑁆𑀫𑀯𑀸𑀭

### 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

(98.)

462\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

463\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀦𑁆𑀢𑀺?

464\. 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀦𑁄 𑀘 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁𑁇 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀜𑁆𑀘𑁂𑀯 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀜𑁆𑀘𑁇

465\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺?

466\. 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁, 𑀦𑁄 𑀘 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑁂𑀯 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀁 𑀦 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇

467\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
