# 𑁨𑁇𑁧𑁇𑁨𑁇𑁮 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀲𑀸𑀫𑀜𑁆𑀜

### 𑀥𑀫𑁆𑀫𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

(97.)

456\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

457\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀦𑁆𑀢𑀺?

458\. 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀦𑁄 𑀘 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁𑁇 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀜𑁆𑀘𑁇

459\. »  𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺?

460\. 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁, 𑀦𑁄 𑀘 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑁂𑀯 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁 𑀦 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇

461\. «  𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
