# 𑁧𑁮 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

* 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

(30.)

79\. 𑀓𑀼𑀲𑀮𑀁 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

80\. 𑀓𑀼𑀲𑀮𑀁 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

81\. 𑀓𑀼𑀲𑀮𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇
