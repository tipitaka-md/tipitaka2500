# 𑁧𑁬𑁇𑁧𑁇𑁪 𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(769.)

1108. _“𑀧𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀓𑀢𑀁 𑀩𑀺𑀫𑁆𑀩𑀁,_  
_𑀅𑀭𑀼𑀓𑀸𑀬𑀁 𑀲𑀫𑀼𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀆𑀢𑀼𑀭𑀁 𑀩𑀳𑀼𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁,_  
_𑀬𑀲𑁆𑀲 𑀦𑀢𑁆𑀣𑀺 𑀥𑀼𑀯𑀁 𑀞𑀺𑀢𑀺𑁇_  


(770.)

1109. _𑀧𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀓𑀢𑀁 𑀭𑀽𑀧𑀁,_  
_𑀫𑀡𑀺𑀦𑀸 𑀓𑀼𑀡𑁆𑀟𑀮𑁂𑀦 𑀘;_  
_𑀅𑀝𑁆𑀞𑀺𑀁 𑀢𑀘𑁂𑀦 𑀑𑀦𑀤𑁆𑀥𑀁,_  
_𑀲𑀳 𑀯𑀢𑁆𑀣𑁂𑀳𑀺 𑀲𑁄𑀪𑀢𑀺𑁇_  


(771.)

1110. _𑀅𑀮𑀢𑁆𑀢𑀓𑀓𑀢𑀸 𑀧𑀸𑀤𑀸,_  
_𑀫𑀼𑀔𑀁 𑀘𑀼𑀡𑁆𑀡𑀓𑀫𑀓𑁆𑀔𑀺𑀢𑀁;_  
_𑀅𑀮𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀫𑁄𑀳𑀸𑀬,_  
_𑀦𑁄 𑀘 𑀧𑀸𑀭𑀕𑀯𑁂𑀲𑀺𑀦𑁄𑁇_  


(772.)

1111. _𑀅𑀝𑁆𑀞𑀧𑀤𑀓𑀢𑀸 𑀓𑁂𑀲𑀸,_  
_𑀦𑁂𑀢𑁆𑀢𑀸 𑀅𑀜𑁆𑀚𑀦𑀫𑀓𑁆𑀔𑀺𑀢𑀸;_  
_𑀅𑀮𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀫𑁄𑀳𑀸𑀬,_  
_𑀦𑁄 𑀘 𑀧𑀸𑀭𑀕𑀯𑁂𑀲𑀺𑀦𑁄𑁇_  


(773.)

1112. _𑀅𑀜𑁆𑀚𑀦𑀻𑀯 𑀦𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀧𑀽𑀢𑀺𑀓𑀸𑀬𑁄 𑀅𑀮𑀗𑁆𑀓𑀢𑁄;_  
_𑀅𑀮𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀫𑁄𑀳𑀸𑀬,_  
_𑀦𑁄 𑀘 𑀧𑀸𑀭𑀕𑀯𑁂𑀲𑀺𑀦𑁄𑁇_  


(774.)

1113. _𑀑𑀤𑀳𑀺 𑀫𑀺𑀕𑀯𑁄 𑀧𑀸𑀲𑀁,_  
_𑀦𑀸𑀲𑀤𑀸 𑀯𑀸𑀕𑀼𑀭𑀁 𑀫𑀺𑀕𑁄;_  
_𑀪𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀧𑀁 𑀕𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑁂 𑀫𑀺𑀕𑀩𑀦𑁆𑀥𑀓𑁂𑁇_  


(775.)

1114. _𑀙𑀺𑀦𑁆𑀦𑁄 𑀧𑀸𑀲𑁄 𑀫𑀺𑀕𑀯𑀲𑁆𑀲,_  
_𑀦𑀸𑀲𑀤𑀸 𑀯𑀸𑀕𑀼𑀭𑀁 𑀫𑀺𑀕𑁄;_  
_𑀪𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀧𑀁 𑀕𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑁄𑀘𑀦𑁆𑀢𑁂 𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑀓𑁂𑁇_  


(776.)

1115. _𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀮𑁄𑀓𑁂 𑀲𑀥𑀦𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂,_  
_𑀮𑀤𑁆𑀥𑀸𑀦 𑀯𑀺𑀢𑁆𑀢𑀁 𑀦 𑀤𑀤𑀦𑁆𑀢𑀺 𑀫𑁄𑀳𑀸;_  
_𑀮𑀼𑀤𑁆𑀥𑀸 𑀥𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺,_  
_𑀪𑀺𑀬𑁆𑀬𑁄𑀯 𑀓𑀸𑀫𑁂 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀦𑁆𑀢𑀺𑁇_  


(777.)

1116. _𑀭𑀸𑀚𑀸 𑀧𑀲𑀬𑁆𑀳𑀧𑁆𑀧𑀣𑀯𑀺𑀁 𑀯𑀺𑀚𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀲𑀸𑀕𑀭𑀦𑁆𑀢𑀁 𑀫𑀳𑀺𑀫𑀸𑀯𑀲𑀦𑁆𑀢𑁄;_  
_𑀑𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀅𑀢𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑁄,_  
_𑀧𑀸𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲𑀧𑀺 𑀧𑀢𑁆𑀣𑀬𑁂𑀣𑁇_  


(778.)

1117. _𑀭𑀸𑀚𑀸 𑀘 𑀅𑀜𑁆𑀜𑁂 𑀘 𑀩𑀳𑀽 𑀫𑀦𑀼𑀲𑁆𑀲𑀸,_  
_𑀅𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀸 𑀫𑀭𑀡𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺;_  
_𑀊𑀦𑀸𑀯 𑀳𑀼𑀢𑁆𑀯𑀸𑀦 𑀚𑀳𑀦𑁆𑀢𑀺 𑀤𑁂𑀳𑀁,_  
_𑀓𑀸𑀫𑁂𑀳𑀺 𑀮𑁄𑀓𑀫𑁆𑀳𑀺 𑀦 𑀳𑀢𑁆𑀣𑀺 𑀢𑀺𑀢𑁆𑀢𑀺𑁇_  


(779.)

1118. _𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀦𑀁 𑀜𑀸𑀢𑀻 𑀧𑀓𑀺𑀭𑀺𑀬 𑀓𑁂𑀲𑁂,_  
_𑀅𑀳𑁄 𑀯𑀢𑀸 𑀦𑁄 𑀅𑀫𑀭𑀸𑀢𑀺 𑀘𑀸𑀳𑀼;_  
_𑀯𑀢𑁆𑀣𑁂𑀦 𑀦𑀁 𑀧𑀸𑀭𑀼𑀢𑀁 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀘𑀺𑀢𑀁 𑀲𑀫𑁄𑀥𑀸𑀬 𑀢𑀢𑁄 𑀟𑀳𑀦𑁆𑀢𑀺𑁇_  


(780.)

1119. _𑀲𑁄 𑀟𑀬𑁆𑀳𑀢𑀺 𑀲𑀽𑀮𑁂𑀳𑀺 𑀢𑀼𑀚𑁆𑀚𑀫𑀸𑀦𑁄,_  
_𑀏𑀓𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦 𑀧𑀳𑀸𑀬 𑀪𑁄𑀕𑁂;_  
_𑀦 𑀫𑀻𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑀸𑀡𑀸,_  
_𑀜𑀸𑀢𑀻 𑀘 𑀫𑀺𑀢𑁆𑀢𑀸 𑀅𑀣 𑀯𑀸 𑀲𑀳𑀸𑀬𑀸𑁇_  


(781.)

1120. _𑀤𑀸𑀬𑀸𑀤𑀓𑀸 𑀢𑀲𑁆𑀲 𑀥𑀦𑀁 𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀲𑀢𑁆𑀢𑁄 𑀧𑀦 𑀕𑀘𑁆𑀙𑀢𑀺 𑀬𑁂𑀦 𑀓𑀫𑁆𑀫𑀁;_  
_𑀦 𑀫𑀻𑀬𑀫𑀸𑀦𑀁 𑀥𑀦𑀫𑀦𑁆𑀯𑁂𑀢𑀺 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀤𑀸𑀭𑀸 𑀘 𑀥𑀦𑀜𑁆𑀘 𑀭𑀝𑁆𑀞𑀁𑁇_  


(782.)

1121. _𑀦 𑀤𑀻𑀖𑀫𑀸𑀬𑀼𑀁 𑀮𑀪𑀢𑁂 𑀥𑀦𑁂𑀦,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀯𑀺𑀢𑁆𑀢𑁂𑀦 𑀚𑀭𑀁 𑀯𑀺𑀳𑀦𑁆𑀢𑀺;_  
_𑀅𑀧𑁆𑀧𑀁 𑀳𑀺𑀤𑀁 𑀚𑀻𑀯𑀺𑀢𑀫𑀸𑀳𑀼 𑀥𑀻𑀭𑀸,_  
_𑀅𑀲𑀲𑁆𑀲𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁𑁇_  


(783.)

1122. _𑀅𑀟𑁆𑀠𑀸 𑀤𑀮𑀺𑀤𑁆𑀤𑀸 𑀘 𑀨𑀼𑀲𑀦𑁆𑀢𑀺 𑀨𑀲𑁆𑀲𑀁,_  
_𑀩𑀸𑀮𑁄 𑀘 𑀥𑀻𑀭𑁄 𑀘 𑀢𑀣𑁂𑀯 𑀨𑀼𑀝𑁆𑀞𑁄;_  
_𑀩𑀸𑀮𑁄 𑀳𑀺 𑀩𑀸𑀮𑁆𑀬𑀸 𑀯𑀥𑀺𑀢𑁄𑀯 𑀲𑁂𑀢𑀺,_  
_𑀥𑀻𑀭𑁄 𑀘 𑀦𑁄 𑀯𑁂𑀥𑀢𑀺 𑀨𑀲𑁆𑀲𑀨𑀼𑀝𑁆𑀞𑁄𑁇_  


(784.)

1123. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀧𑀜𑁆𑀜𑀸𑀯 𑀥𑀦𑁂𑀦 𑀲𑁂𑀬𑁆𑀬𑀸,_  
_𑀬𑀸𑀬 𑀯𑁄𑀲𑀸𑀦𑀫𑀺𑀥𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀅𑀩𑁆𑀬𑁄𑀲𑀺𑀢𑀢𑁆𑀢𑀸 𑀳𑀺 𑀪𑀯𑀸𑀪𑀯𑁂𑀲𑀼,_  
_𑀧𑀸𑀧𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺 𑀫𑁄𑀳𑀸𑁇_  


(785.)

1124. _𑀉𑀧𑁂𑀢𑀺 𑀕𑀩𑁆𑀪𑀜𑁆𑀘 𑀧𑀭𑀜𑁆𑀘 𑀮𑁄𑀓𑀁,_  
_𑀲𑀁𑀲𑀸𑀭𑀫𑀸𑀧𑀚𑁆𑀚 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀬;_  
_𑀢𑀲𑁆𑀲𑀧𑁆𑀧𑀧𑀜𑁆𑀜𑁄 𑀅𑀪𑀺𑀲𑀤𑁆𑀤𑀳𑀦𑁆𑀢𑁄,_  
_𑀉𑀧𑁂𑀢𑀺 𑀕𑀩𑁆𑀪𑀜𑁆𑀘 𑀧𑀭𑀜𑁆𑀘 𑀮𑁄𑀓𑀁𑁇_  


(786.)

1125. _𑀘𑁄𑀭𑁄 𑀬𑀣𑀸 𑀲𑀦𑁆𑀥𑀺𑀫𑀼𑀔𑁂 𑀕𑀳𑀻𑀢𑁄,_  
_𑀲𑀓𑀫𑁆𑀫𑀼𑀦𑀸 𑀳𑀜𑁆𑀜𑀢𑀺 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄;_  
_𑀏𑀯𑀁 𑀧𑀚𑀸 𑀧𑁂𑀘𑁆𑀘 𑀧𑀭𑀫𑁆𑀳𑀺 𑀮𑁄𑀓𑁂,_  
_𑀲𑀓𑀫𑁆𑀫𑀼𑀦𑀸 𑀳𑀜𑁆𑀜𑀢𑀺 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄𑁇_  


(787.)

1126. _𑀓𑀸𑀫𑀸 𑀳𑀺 𑀘𑀺𑀢𑁆𑀭𑀸 𑀫𑀥𑀼𑀭𑀸 𑀫𑀦𑁄𑀭𑀫𑀸,_  
_𑀯𑀺𑀭𑀽𑀧𑀭𑀽𑀧𑁂𑀦 𑀫𑀣𑁂𑀦𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁;_  
_𑀆𑀤𑀻𑀦𑀯𑀁 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀢𑀲𑁆𑀫𑀸 𑀅𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀭𑀸𑀚𑁇_  


(788.)

1127. _𑀤𑀼𑀫𑀧𑁆𑀨𑀮𑀸𑀦𑀻𑀯 𑀧𑀢𑀦𑁆𑀢𑀺 𑀫𑀸𑀡𑀯𑀸,_  
_𑀤𑀳𑀭𑀸 𑀘 𑀯𑀼𑀟𑁆𑀠𑀸 𑀘 𑀲𑀭𑀻𑀭𑀪𑁂𑀤𑀸;_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀭𑀸𑀚,_  
_𑀅𑀧𑀡𑁆𑀡𑀓𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀫𑁂𑀯 𑀲𑁂𑀬𑁆𑀬𑁄𑁇_  


(789.)

1128. _𑀲𑀤𑁆𑀥𑀸𑀬𑀸𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀉𑀧𑁂𑀢𑁄 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂;_  
_𑀅𑀯𑀜𑁆𑀛𑀸 𑀫𑀬𑁆𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸,_  
_𑀅𑀦𑀡𑁄 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺 𑀪𑁄𑀚𑀦𑀁𑁇_  


(790.)

1129. _𑀓𑀸𑀫𑁂 𑀆𑀤𑀺𑀢𑁆𑀢𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀚𑀸𑀢𑀭𑀽𑀧𑀸𑀦𑀺 𑀲𑀢𑁆𑀣𑀢𑁄;_  
_𑀕𑀩𑁆𑀪𑀯𑁄𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀦𑀺𑀭𑀬𑁂𑀲𑀼 𑀫𑀳𑀩𑁆𑀪𑀬𑀁𑁇_  


(791.)

1130. _𑀏𑀢𑀫𑀸𑀤𑀻𑀦𑀯𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀯𑁂𑀕𑀁 𑀅𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀲𑁄𑀳𑀁 𑀯𑀺𑀤𑁆𑀥𑁄 𑀢𑀤𑀸 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁄 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(792.)

1131. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(793.)

1132. _𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄”𑀢𑀺𑁇_  


---

1133. …  𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑁄 𑀣𑁂𑀭𑁄… 𑁇
