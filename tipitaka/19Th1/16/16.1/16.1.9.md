# 𑁧𑁬𑁇𑁧𑁇𑁯 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(892.)

1236\. _“𑀧𑀳𑀸𑀬 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀪𑀕𑀺𑀦𑀻 𑀜𑀸𑀢𑀺𑀪𑀸𑀢𑀭𑁄;_  
_𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄𑀯 𑀛𑀸𑀬𑀢𑀺𑁇_  


(893.)

1237\. _𑀲𑀫𑁂𑀢𑁄 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑁂𑀳𑀺,_  
_𑀲𑀫𑁆𑀫𑀢𑀸𑀍𑀅𑀧𑁆𑀧𑀩𑁄𑀥𑀦𑁄;_  
_𑀦 𑀢𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀺𑀫𑀚𑁆𑀛𑀕𑀁,_  
_𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀺𑀲𑀬𑁂 𑀭𑀢𑁄𑁇_  


(894.)

1238\. _𑀏𑀢𑀜𑁆𑀘 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀭𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂;_  
_𑀲𑀩𑁆𑀩𑁄𑀖𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄𑀯 𑀛𑀸𑀬𑀢𑀺𑁇_  


(895.)

1239\. _𑀭𑀽𑀧𑀸 𑀲𑀤𑁆𑀤𑀸 𑀭𑀲𑀸 𑀕𑀦𑁆𑀥𑀸,_  
_𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀘 𑀫𑀦𑁄𑀭𑀫𑀸;_  
_𑀏𑀢𑁂 𑀘 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄𑀯 𑀛𑀸𑀬𑀢𑀺𑁇_  


(896.)

1240\. _𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄,_  
_𑀏𑀓𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀫𑀼𑀦𑀺;_  
_𑀏𑀲𑀢𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺,_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(897.)

1241\. _𑀯𑀺𑀘𑀺𑀦𑀻 𑀅𑀕𑁆𑀕𑀳𑀻 𑀥𑁄𑀯𑀺,_  
_𑀭𑀚𑀬𑀻 𑀥𑀸𑀭𑀬𑀻 𑀫𑀼𑀦𑀺;_  
_𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀫𑀢𑀺𑀫𑀸,_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(898.)

1242\. _𑀫𑀳𑀺𑀘𑁆𑀙𑁄 𑀘 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀬𑁄 𑀘 𑀉𑀤𑁆𑀥𑀢𑁄;_  
_𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀇𑀫𑁂 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀧𑀸𑀧𑀓𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇_  


(899.)

1243\. _𑀲𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄,_  
_𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀅𑀯𑀺𑀖𑀸𑀢𑀯𑀸;_  
_𑀧𑀯𑀺𑀯𑁂𑀓𑀭𑀢𑁄 𑀯𑀺𑀢𑁆𑀢𑁄,_  
_𑀦𑀺𑀘𑁆𑀘𑀫𑀸𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄𑁇_  


(900.)

1244\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀇𑀫𑁂 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀓𑀼𑀲𑀮𑀸 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑀸;_  
_𑀅𑀦𑀸𑀲𑀯𑁄 𑀘 𑀲𑁄 𑀳𑁄𑀢𑀺,_  
_𑀇𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀸𑁇_  


(901.)

1245\. _𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀫𑀦𑁄𑀫𑀬𑁂𑀦 𑀓𑀸𑀬𑁂𑀦,_  
_𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇_  


(902.)

1246\. _𑀬𑀤𑀸 𑀫𑁂 𑀅𑀳𑀼 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀢𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀤𑁂𑀲𑀬𑀺;_  
_𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀫𑀤𑁂𑀲𑀬𑀺𑁇_  


(903.)

1247\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(904.)

1248\. _𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀯𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀬𑀢𑁄 𑀦𑁂𑀲𑀚𑁆𑀚𑀺𑀓𑁄 𑀅𑀳𑀁;_  
_𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀬𑀢𑁄 𑀫𑀺𑀤𑁆𑀥𑀁 𑀲𑀫𑀽𑀳𑀢𑀁𑁇_  


(905.)

1249\. _𑀦𑀸𑀳𑀼 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸,_  
_𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀅𑀦𑁂𑀚𑁄 𑀲𑀦𑁆𑀢𑀺𑀫𑀸𑀭𑀩𑁆𑀪,_  
_𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇_  


(906.)

1250\. _𑀅𑀲𑀮𑁆𑀮𑀻𑀦𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀯𑁂𑀤𑀦𑀁 𑀅𑀚𑁆𑀛𑀯𑀸𑀲𑀬𑀺;_  
_𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑁆𑀲𑁂𑀯 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀘𑁂𑀢𑀲𑁄 𑀅𑀳𑀼𑁇_  


(907.)

1251\. _𑀏𑀢𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑀸 𑀤𑀸𑀦𑀺,_  
_𑀫𑀼𑀦𑀺𑀦𑁄 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀘𑀫𑀸;_  
_𑀦𑀸𑀜𑁆𑀜𑁂 𑀥𑀫𑁆𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂𑁇_  


(908.)

1252\. _𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀸𑀯𑀸𑀲𑁄,_  
_𑀤𑁂𑀯𑀓𑀸𑀬𑀲𑁆𑀫𑀺 𑀚𑀸𑀮𑀺𑀦𑀺;_  
_𑀯𑀺𑀓𑁆𑀔𑀻𑀡𑁄 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(909.)

1253\. _𑀬𑀲𑁆𑀲 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀳𑀲𑁆𑀲𑀥𑀸,_  
_𑀮𑁄𑀓𑁄 𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀧𑁆𑀧𑁄;_  
_𑀯𑀲𑀻 𑀇𑀤𑁆𑀥𑀺𑀕𑀼𑀡𑁂 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁂,_  
_𑀓𑀸𑀮𑁂 𑀧𑀲𑁆𑀲𑀢𑀺 𑀤𑁂𑀯𑀢𑀸 𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑁇_  


(910.)

1254\. _𑀅𑀦𑁆𑀦𑀪𑀸𑀭𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀤𑀮𑀺𑀤𑁆𑀤𑁄 𑀖𑀸𑀲𑀳𑀸𑀭𑀓𑁄;_  
_𑀲𑀫𑀡𑀁 𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀺𑀁,_  
_𑀉𑀧𑀭𑀺𑀝𑁆𑀞𑀁 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀁𑁇_  


(911.)

1255\. _𑀲𑁄𑀫𑁆𑀳𑀺 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑁂 𑀚𑀸𑀢𑁄,_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀫𑀁 𑀯𑀺𑀤𑀽;_  
_𑀉𑀧𑁂𑀢𑁄 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑁂𑀳𑀺,_  
_𑀲𑀫𑁆𑀫𑀢𑀸𑀍𑀅𑀧𑁆𑀧𑀩𑁄𑀥𑀦𑁄𑁇_  


(912.)

1256\. _𑀅𑀣𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑀁;_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(913.)

1257\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀬𑀢𑁆𑀣 𑀫𑁂 𑀯𑀼𑀲𑀺𑀢𑀁 𑀧𑀼𑀭𑁂;_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑀁 𑀲𑀓𑁆𑀓𑀚𑀸𑀢𑀺𑀬𑀸𑁇_  


(914.)

1258\. _𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁄,_  
_𑀅𑀳𑀁 𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻,_  
_𑀚𑀫𑁆𑀩𑀼𑀲𑀡𑁆𑀟𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑁄;_  
_𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦,_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀦𑀼𑀲𑀸𑀲𑀬𑀺𑀁𑁇_  


(915.)

1259\. _𑀇𑀢𑁄 𑀲𑀢𑁆𑀢 𑀢𑀢𑁄 𑀲𑀢𑁆𑀢,_  
_𑀲𑀁𑀲𑀸𑀭𑀸𑀦𑀺 𑀘𑀢𑀼𑀤𑁆𑀤𑀲;_  
_𑀦𑀺𑀯𑀸𑀲𑀫𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀁,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀞𑀺𑀢𑁄 𑀢𑀤𑀸𑁇_  


(916.)

1260\. _𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁂 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀺,_  
_𑀲𑀦𑁆𑀢𑁂 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀺𑀢𑁂;_  
_𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀮𑀤𑁆𑀥𑀫𑁆𑀳𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀺 𑀫𑁂𑁇_  


(917.)

1261\. _𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀕𑀢𑀺𑀁 𑀕𑀢𑀺𑀁;_  
_𑀇𑀢𑁆𑀣𑀪𑀸𑀯𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀁,_  
_𑀛𑀸𑀦𑁂 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁂 𑀞𑀺𑀢𑁄𑁇_

1262\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_
