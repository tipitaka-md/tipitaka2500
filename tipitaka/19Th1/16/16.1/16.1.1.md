# 𑁧𑁬𑁇𑁧𑁇𑁧 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(705.)

1041\. _“𑀬𑀜𑁆𑀜𑀢𑁆𑀣𑀁 𑀯𑀸 𑀥𑀦𑀢𑁆𑀣𑀁 𑀯𑀸,_  
_𑀬𑁂 𑀳𑀦𑀸𑀫 𑀫𑀬𑀁 𑀧𑀼𑀭𑁂;_  
_𑀅𑀯𑀲𑁂𑀲𑀁 𑀪𑀬𑀁 𑀳𑁄𑀢𑀺,_  
_𑀯𑁂𑀥𑀦𑁆𑀢𑀺 𑀯𑀺𑀮𑀧𑀦𑁆𑀢𑀺 𑀘𑁇_  


(706.)

1042\. _𑀢𑀲𑁆𑀲 𑀢𑁂 𑀦𑀢𑁆𑀣𑀺 𑀪𑀻𑀢𑀢𑁆𑀢𑀁,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀯𑀡𑁆𑀡𑁄 𑀧𑀲𑀻𑀤𑀢𑀺;_  
_𑀓𑀲𑁆𑀫𑀸 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀲𑀺,_  
_𑀏𑀯𑀭𑀽𑀧𑁂 𑀫𑀳𑀩𑁆𑀪𑀬𑁂”𑁇_  


(707.)

1043\. _“𑀦𑀢𑁆𑀣𑀺 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀲𑁆𑀲 𑀕𑀸𑀫𑀡𑀺;_  
_𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀪𑀬𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀔𑀻𑀡𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀯𑁂𑁇_  


(708.)

1044\. _𑀔𑀻𑀡𑀸𑀬 𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺𑀬𑀸,_  
_𑀤𑀺𑀝𑁆𑀞𑁂 𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀢𑀣𑁂;_  
_𑀦 𑀪𑀬𑀁 𑀫𑀭𑀡𑁂 𑀳𑁄𑀢𑀺,_  
_𑀪𑀸𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀦𑁂 𑀬𑀣𑀸𑁇_  


(709.)

1045\. _𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀫𑁂,_  
_𑀫𑀕𑁆𑀕𑁄 𑀘𑀸𑀧𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄;_  
_𑀫𑀭𑀡𑁂 𑀫𑁂 𑀪𑀬𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀭𑁄𑀕𑀸𑀦𑀫𑀺𑀯 𑀲𑀗𑁆𑀔𑀬𑁂𑁇_  


(710.)

1046\. _𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀫𑁂,_  
_𑀫𑀕𑁆𑀕𑁄 𑀘𑀸𑀧𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄;_  
_𑀦𑀺𑀭𑀲𑁆𑀲𑀸𑀤𑀸 𑀪𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀸,_  
_𑀯𑀺𑀲𑀁 𑀧𑀺𑀢𑁆𑀯𑀸𑀯 𑀙𑀟𑁆𑀟𑀺𑀢𑀁𑁇_  


(711.)

1047\. _𑀧𑀸𑀭𑀕𑀽 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁄 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀢𑀼𑀝𑁆𑀞𑁄 𑀆𑀬𑀼𑀓𑁆𑀔𑀬𑀸 𑀳𑁄𑀢𑀺,_  
_𑀫𑀼𑀢𑁆𑀢𑁄 𑀆𑀖𑀸𑀢𑀦𑀸 𑀬𑀣𑀸𑁇_  


(712.)

1048\. _𑀉𑀢𑁆𑀢𑀫𑀁 𑀥𑀫𑁆𑀫𑀢𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀆𑀤𑀺𑀢𑁆𑀢𑀸𑀯 𑀖𑀭𑀸 𑀫𑀼𑀢𑁆𑀢𑁄,_  
_𑀫𑀭𑀡𑀲𑁆𑀫𑀺𑀁 𑀦 𑀲𑁄𑀘𑀢𑀺𑁇_  


(713.)

1049\. _𑀬𑀤𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀕𑀢𑀁 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀪𑀯𑁄 𑀯𑀸 𑀬𑀢𑁆𑀣 𑀮𑀩𑁆𑀪𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑀁 𑀏𑀢𑀁,_  
_𑀇𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀸𑁇_  


(714.)

1050\. _𑀬𑁄 𑀢𑀁 𑀢𑀣𑀸 𑀧𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀬𑀣𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀦 𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀪𑀯𑀁 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀲𑀼𑀢𑀢𑁆𑀢𑀁𑀯 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀁𑁇_  


(715.)

1051\. _𑀦 𑀫𑁂 𑀳𑁄𑀢𑀺 ‘𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺,_  
_‘𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆’𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺 𑀫𑁂;_  
_𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀯𑀺𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑀸𑁇_  


(716.)

1052\. _𑀲𑀼𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁,_  
_𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀦𑁆𑀢𑀢𑀺𑀁;_  
_𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀪𑀽𑀢𑀁,_  
_𑀦 𑀪𑀬𑀁 𑀳𑁄𑀢𑀺 𑀕𑀸𑀫𑀡𑀺𑁇_  


(717.)

1053\. _𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀫𑀁 𑀮𑁄𑀓𑀁,_  
_𑀬𑀤𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑀫𑀢𑁆𑀢𑀁 𑀲𑁄 𑀅𑀲𑀁𑀯𑀺𑀦𑁆𑀤𑀁,_  
_‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂’𑀢𑀺 𑀦 𑀲𑁄𑀘𑀢𑀺𑁇_  


(718.)

1054\. _𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀸𑀫𑀺 𑀲𑀭𑀻𑀭𑁂𑀦,_  
_𑀪𑀯𑁂𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀲𑁄𑀬𑀁 𑀪𑀺𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀬𑁄,_  
_𑀅𑀜𑁆𑀜𑁄 𑀘 𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(719.)

1055\. _𑀬𑀁 𑀯𑁄 𑀓𑀺𑀘𑁆𑀘𑀁 𑀲𑀭𑀻𑀭𑁂𑀦,_  
_𑀢𑀁 𑀓𑀭𑁄𑀣 𑀬𑀤𑀺𑀘𑁆𑀙𑀣;_  
_𑀦 𑀫𑁂 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀢𑁆𑀣,_  
_𑀤𑁄𑀲𑁄 𑀧𑁂𑀫𑀜𑁆𑀘 𑀳𑁂𑀳𑀺𑀢𑀺”𑁇_  


(720.)

1056\. _𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀸𑀡𑀯𑀸 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀼𑀁𑁇_  


(721.)

1057\. _“𑀓𑀺𑀁 𑀪𑀤𑀦𑁆𑀢𑁂 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑁄 𑀯𑀸 𑀆𑀘𑀭𑀺𑀬𑁄 𑀢𑀯;_  
_𑀓𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀮𑀩𑁆𑀪𑀢𑁂 𑀢𑀁 𑀅𑀲𑁄𑀓𑀢𑀸”𑁇_  


(722.)

1058\. _“𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽 𑀲𑀩𑁆𑀩𑀤𑀲𑁆𑀲𑀸𑀯𑀻,_  
_𑀚𑀺𑀦𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀫𑀫;_  
_𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑁄𑁇_  


(723.)

1059\. _𑀢𑁂𑀦𑀸𑀬𑀁 𑀤𑁂𑀲𑀺𑀢𑁄 𑀥𑀫𑁆𑀫𑁄,_  
_𑀔𑀬𑀕𑀸𑀫𑀻 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀮𑀩𑁆𑀪𑀢𑁂 𑀢𑀁 𑀅𑀲𑁄𑀓𑀢𑀸”𑁇_  


(724.)

1060\. _𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀘𑁄𑀭𑀸 𑀇𑀲𑀺𑀦𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁,_  
_𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁆𑀧 𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀘 𑀆𑀯𑀼𑀥𑀸𑀦𑀺 𑀘;_  
_𑀢𑀫𑁆𑀳𑀸 𑀘 𑀓𑀫𑁆𑀫𑀸 𑀯𑀺𑀭𑀫𑀺𑀁𑀲𑀼 𑀏𑀓𑁂,_  
_𑀏𑀓𑁂 𑀘 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀫𑀭𑁄𑀘𑀬𑀺𑀁𑀲𑀼𑁇_  


(725.)

1061\. _𑀢𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂,_  
_𑀪𑀸𑀯𑁂𑀢𑁆𑀯 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀩𑀮𑀸𑀦𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸 𑀓𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸,_  
_𑀨𑀼𑀲𑀺𑀁𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀤𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀦𑁆𑀢𑀺𑁇_  


---

1062\. … 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
