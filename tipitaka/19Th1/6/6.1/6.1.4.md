# 𑁬𑁇𑁧𑁇𑁪 𑀓𑀼𑀮𑁆𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀮𑁆𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(393.)

664\. _“𑀓𑀼𑀮𑁆𑀮𑁄 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀤𑁆𑀤𑀲 𑀇𑀢𑁆𑀣𑀺𑀫𑀼𑀚𑁆𑀛𑀺𑀢𑀁;_  
_𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀲𑀼𑀲𑀸𑀦𑀲𑁆𑀫𑀺𑀁,_  
_𑀔𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀁 𑀓𑀺𑀫𑀺𑀳𑀻 𑀨𑀼𑀝𑀁𑁇_  


(394.)

665\. _𑀆𑀢𑀼𑀭𑀁 𑀅𑀲𑀼𑀘𑀺𑀁 𑀧𑀽𑀢𑀺𑀁,_  
_𑀧𑀲𑁆𑀲 𑀓𑀼𑀮𑁆𑀮 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀁;_  
_𑀉𑀕𑁆𑀖𑀭𑀦𑁆𑀢𑀁 𑀧𑀕𑁆𑀖𑀭𑀦𑁆𑀢𑀁,_  
_𑀩𑀸𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀁𑁇_  


(395.)

666\. _𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀁 𑀇𑀫𑀁 𑀓𑀸𑀬𑀁,_  
_𑀢𑀼𑀘𑁆𑀙𑀁 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀁𑁇_  


(396.)

667\. _𑀬𑀣𑀸 𑀇𑀤𑀁 𑀢𑀣𑀸 𑀏𑀢𑀁,_  
_𑀬𑀣𑀸 𑀏𑀢𑀁 𑀢𑀣𑀸 𑀇𑀤𑀁;_  
_𑀬𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁,_  
_𑀬𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄𑁇_  


(397.)

668\. _𑀬𑀣𑀸 𑀤𑀺𑀯𑀸 𑀢𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀬𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀣𑀸 𑀤𑀺𑀯𑀸;_  
_𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸,_  
_𑀬𑀣𑀸 𑀧𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀧𑀼𑀭𑁂𑁇_  


(398.)

669\. _𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀢𑀼𑀭𑀺𑀬𑁂𑀦,_  
_𑀦 𑀭𑀢𑀻 𑀳𑁄𑀢𑀺 𑀢𑀸𑀤𑀺𑀲𑀻;_  
_𑀬𑀣𑀸 𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑁄”𑀢𑀺𑁇_  


---

670\. …  𑀓𑀼𑀮𑁆𑀮𑁄 𑀣𑁂𑀭𑁄… 𑁇
