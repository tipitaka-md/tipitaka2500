# 𑁬𑁇𑁧𑁇𑁧 𑀉𑀭𑀼𑀯𑁂𑀍𑀅𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀉𑀭𑀼𑀯𑁂𑀍𑀅𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(375.)

643. _“𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀸𑀦𑀺,_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀦 𑀢𑀸𑀯𑀸𑀳𑀁 𑀧𑀡𑀺𑀧𑀢𑀺𑀁,_  
_𑀇𑀲𑁆𑀲𑀸𑀫𑀸𑀦𑁂𑀦 𑀯𑀜𑁆𑀘𑀺𑀢𑁄𑁇_  


(376.)

644. _𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀘𑁄𑀤𑁂𑀲𑀺 𑀦𑀭𑀲𑀸𑀭𑀣𑀺;_  
_𑀢𑀢𑁄 𑀫𑁂 𑀆𑀲𑀺 𑀲𑀁𑀯𑁂𑀕𑁄,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑁄𑁇_  


(377.)

645. _𑀧𑀼𑀩𑁆𑀩𑁂 𑀚𑀝𑀺𑀮𑀪𑀽𑀢𑀲𑁆𑀲,_  
_𑀬𑀸 𑀫𑁂 𑀲𑀺𑀤𑁆𑀥𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀺𑀓𑀸;_  
_𑀢𑀸𑀳𑀁 𑀢𑀤𑀸 𑀦𑀺𑀭𑀸𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂𑁇_  


(378.)

646. _𑀧𑀼𑀩𑁆𑀩𑁂 𑀬𑀜𑁆𑀜𑁂𑀦 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀧𑀘𑁆𑀙𑀸 𑀭𑀸𑀕𑀜𑁆𑀘 𑀤𑁄𑀲𑀜𑁆𑀘,_  
_𑀫𑁄𑀳𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀫𑀽𑀳𑀦𑀺𑀁𑁇_  


(379.)

647. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀜𑀽,_  
_𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀜𑁆𑀘 𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(380.)

648. _𑀬𑀲𑁆𑀲 𑀘𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄”𑀢𑀺𑁇_  


---

649. …  𑀉𑀭𑀼𑀯𑁂𑀍𑀅𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄… 𑁇
