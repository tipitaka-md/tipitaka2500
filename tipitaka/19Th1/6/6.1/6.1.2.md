# 𑁬𑁇𑁧𑁇𑁨 𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑀓𑀸𑀭𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑀓𑀸𑀭𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(381.)

650\. _“𑀅𑀢𑀺𑀳𑀺𑀢𑀸 𑀯𑀻𑀳𑀺,_  
_𑀔𑀮𑀕𑀢𑀸 𑀲𑀸𑀮𑀻;_  
_𑀦 𑀘 𑀮𑀪𑁂 𑀧𑀺𑀡𑁆𑀟𑀁,_  
_𑀓𑀣𑀫𑀳𑀁 𑀓𑀲𑁆𑀲𑀁”𑁇_  


(382.)

651\. _“𑀩𑀼𑀤𑁆𑀥𑀫𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭 𑀧𑀲𑀦𑁆𑀦𑁄,_  
_𑀧𑀻𑀢𑀺𑀬𑀸 𑀨𑀼𑀝𑀲𑀭𑀻𑀭𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀲𑀢𑀢𑀫𑀼𑀤𑀕𑁆𑀕𑁄𑁇_  


(383.)

652\. _𑀥𑀫𑁆𑀫𑀫𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭 𑀧𑀲𑀦𑁆𑀦𑁄,_  
_𑀧𑀻𑀢𑀺𑀬𑀸 𑀨𑀼𑀝𑀲𑀭𑀻𑀭𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀲𑀢𑀢𑀫𑀼𑀤𑀕𑁆𑀕𑁄𑁇_  


(384.)

653\. _𑀲𑀁𑀖𑀫𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭 𑀧𑀲𑀦𑁆𑀦𑁄,_  
_𑀧𑀻𑀢𑀺𑀬𑀸 𑀨𑀼𑀝𑀲𑀭𑀻𑀭𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀲𑀢𑀢𑀫𑀼𑀤𑀕𑁆𑀕𑁄”𑁇_  


(385.)

654\. _“𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀯𑀺𑀳𑀭𑀲𑀺,_  
_𑀲𑀻𑀢𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸 𑀇𑀫𑀸 𑀭𑀢𑁆𑀬𑁄;_  
_𑀫𑀸 𑀲𑀻𑀢𑁂𑀦 𑀧𑀭𑁂𑀢𑁄 𑀯𑀺𑀳𑀜𑁆𑀜𑀺𑀢𑁆𑀣𑁄,_  
_𑀧𑀯𑀺𑀲 𑀢𑁆𑀯𑀁 𑀯𑀺𑀳𑀸𑀭𑀁 𑀨𑀼𑀲𑀺𑀢𑀕𑁆𑀕𑀍𑀅𑀁”𑁇_  


(386.)

655\. _“𑀨𑀼𑀲𑀺𑀲𑁆𑀲𑀁 𑀘𑀢𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄,_  
_𑀢𑀸𑀳𑀺 𑀘 𑀲𑀼𑀔𑀺𑀢𑁄 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀁;_  
_𑀦𑀸𑀳𑀁 𑀲𑀻𑀢𑁂𑀦 𑀯𑀺𑀳𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀁,_  
_𑀅𑀦𑀺𑀜𑁆𑀚𑀺𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄”𑀢𑀺𑁇_  


---

656\. …  𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑀓𑀸𑀭𑀻 𑀣𑁂𑀭𑁄… 𑁇
