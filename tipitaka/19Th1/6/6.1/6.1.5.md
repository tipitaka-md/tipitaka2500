# 𑁬𑁇𑁧𑁇𑁫 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(399.)

671. _“𑀫𑀦𑀼𑀚𑀲𑁆𑀲 𑀧𑀫𑀢𑁆𑀢𑀘𑀸𑀭𑀺𑀦𑁄,_  
_𑀢𑀡𑁆𑀳𑀸 𑀯𑀟𑁆𑀠𑀢𑀺 𑀫𑀸𑀮𑀼𑀯𑀸 𑀯𑀺𑀬;_  
_𑀲𑁄 𑀧𑁆𑀮𑀯𑀢𑀻 𑀳𑀼𑀭𑀸 𑀳𑀼𑀭𑀁,_  
_𑀨𑀮𑀫𑀺𑀘𑁆𑀙𑀁𑀯 𑀯𑀦𑀲𑁆𑀫𑀺 𑀯𑀸𑀦𑀭𑁄𑁇_  


(400.)

672. _𑀬𑀁 𑀏𑀲𑀸 𑀲𑀳𑀢𑁂 𑀚𑀫𑁆𑀫𑀻,_  
_𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀸;_  
_𑀲𑁄𑀓𑀸 𑀢𑀲𑁆𑀲 𑀧𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺,_  
_𑀅𑀪𑀺𑀯𑀝𑁆𑀞𑀁𑀯 𑀩𑀻𑀭𑀡𑀁𑁇_  


(401.)

673. _𑀬𑁄 𑀘𑁂𑀢𑀁 𑀲𑀳𑀢𑁂 𑀚𑀫𑁆𑀫𑀺𑀁,_  
_𑀢𑀡𑁆𑀳𑀁 𑀮𑁄𑀓𑁂 𑀤𑀼𑀭𑀘𑁆𑀘𑀬𑀁;_  
_𑀲𑁄𑀓𑀸 𑀢𑀫𑁆𑀳𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺,_  
_𑀉𑀤𑀩𑀺𑀦𑁆𑀤𑀽𑀯 𑀧𑁄𑀓𑁆𑀔𑀭𑀸𑁇_  


(402.)

674. _𑀢𑀁 𑀯𑁄 𑀯𑀤𑀸𑀫𑀺 𑀪𑀤𑁆𑀤𑀁 𑀯𑁄,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀢𑀡𑁆𑀳𑀸𑀬 𑀫𑀽𑀮𑀁 𑀔𑀡𑀣,_  
_𑀉𑀲𑀻𑀭𑀢𑁆𑀣𑁄𑀯 𑀩𑀻𑀭𑀡𑀁;_  
_𑀫𑀸 𑀯𑁄 𑀦𑀍𑀅𑀁𑀯 𑀲𑁄𑀢𑁄𑀯,_  
_𑀫𑀸𑀭𑁄 𑀪𑀜𑁆𑀚𑀺 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁𑁇_  


(403.)

675. _𑀓𑀭𑁄𑀣 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑀁,_  
_𑀔𑀡𑁄 𑀯𑁄 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀔𑀡𑀸𑀢𑀻𑀢𑀸 𑀳𑀺 𑀲𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(404.)

676. _𑀧𑀫𑀸𑀤𑁄 𑀭𑀚𑁄 𑀧𑀫𑀸𑀤𑁄,_  
_𑀧𑀫𑀸𑀤𑀸𑀦𑀼𑀧𑀢𑀺𑀢𑁄 𑀭𑀚𑁄;_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦 𑀯𑀺𑀚𑁆𑀚𑀸𑀬,_  
_𑀅𑀩𑁆𑀩𑀳𑁂 𑀲𑀮𑁆𑀮𑀫𑀢𑁆𑀢𑀦𑁄”𑀢𑀺𑁇_  


---

677. …  𑀫𑀸𑀮𑀼𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
