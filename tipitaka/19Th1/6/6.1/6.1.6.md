# 𑁬𑁇𑁧𑁇𑁬 𑀲𑀧𑁆𑀧𑀤𑀸𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀧𑁆𑀧𑀤𑀸𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(405.)

678\. _“𑀧𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀬𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁;_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀖𑀸𑀢𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺,_  
_𑀘𑁂𑀢𑁄𑀲𑀦𑁆𑀢𑀺𑀫𑀦𑀚𑁆𑀛𑀕𑀁𑁇_  


(406.)

679\. _𑀅𑀮𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀁,_  
_𑀓𑀸𑀫𑀭𑀸𑀕𑁂𑀦 𑀅𑀝𑁆𑀝𑀺𑀢𑁄;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀳𑀸𑀭𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀁𑁇_  


(407.)

680\. _𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀆𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀓𑁄 𑀅𑀢𑁆𑀣𑁄 𑀚𑀻𑀯𑀺𑀢𑁂𑀦 𑀫𑁂;_  
_𑀓𑀣𑀁 𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀁,_  
_𑀓𑀸𑀮𑀁 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀫𑀸𑀤𑀺𑀲𑁄𑁇_  


(408.)

681\. _𑀢𑀤𑀸𑀳𑀁 𑀔𑀼𑀭𑀫𑀸𑀤𑀸𑀬,_  
_𑀫𑀜𑁆𑀘𑀓𑀫𑁆𑀳𑀺 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑀁;_  
_𑀧𑀭𑀺𑀦𑀻𑀢𑁄 𑀔𑀼𑀭𑁄 𑀆𑀲𑀺,_  
_𑀥𑀫𑀦𑀺𑀁 𑀙𑁂𑀢𑁆𑀢𑀼𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


(409.)

682\. _𑀢𑀢𑁄 𑀫𑁂 𑀫𑀦𑀲𑀻𑀓𑀸𑀭𑁄,_  
_𑀬𑁄𑀦𑀺𑀲𑁄 𑀉𑀤𑀧𑀚𑁆𑀚𑀣;_  
_𑀆𑀤𑀻𑀦𑀯𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑀼,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀲𑀫𑀢𑀺𑀝𑁆𑀞𑀣𑁇_  


(410.)

683\. _𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺 𑀫𑁂,_  
_𑀧𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

684\. …  𑀲𑀧𑁆𑀧𑀤𑀸𑀲𑁄 𑀣𑁂𑀭𑁄… 𑁇
