# 𑁬𑁇𑁧𑁇𑁧𑁨 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(441.)

720\. _“𑀅𑀓𑁆𑀓𑁄𑀥𑀲𑁆𑀲 𑀓𑀼𑀢𑁄 𑀓𑁄𑀥𑁄,_  
_𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑀚𑀻𑀯𑀺𑀦𑁄;_  
_𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(442.)

721\. _𑀢𑀲𑁆𑀲𑁂𑀯 𑀢𑁂𑀦 𑀧𑀸𑀧𑀺𑀬𑁄,_  
_𑀬𑁄 𑀓𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀓𑀼𑀤𑁆𑀥𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑁄,_  
_𑀲𑀗𑁆𑀕𑀸𑀫𑀁 𑀚𑁂𑀢𑀺 𑀤𑀼𑀚𑁆𑀚𑀬𑀁𑁇_  


(443.)

722\. _𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀭𑀲𑁆𑀲 𑀘;_  
_𑀧𑀭𑀁 𑀲𑀗𑁆𑀓𑀼𑀧𑀺𑀢𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀬𑁄 𑀲𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


(444.)

723\. _𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀢𑀁,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀭𑀲𑁆𑀲 𑀘;_  
_𑀚𑀦𑀸 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑁄𑀢𑀺,_  
_𑀬𑁂 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑀸𑁇_  


(445.)

724\. _𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂 𑀢𑁂 𑀲𑀘𑁂 𑀓𑁄𑀥𑁄,_  
_𑀆𑀯𑀚𑁆𑀚 𑀓𑀓𑀘𑀽𑀧𑀫𑀁;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂 𑀘𑁂 𑀭𑀲𑁂 𑀢𑀡𑁆𑀳𑀸,_  
_𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑀽𑀧𑀫𑀁 𑀲𑀭𑁇_  


(446.)

725\. _𑀲𑀘𑁂 𑀥𑀸𑀯𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀢𑁂,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀘 𑀪𑀯𑁂𑀲𑀼 𑀘;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳 𑀲𑀢𑀺𑀬𑀸,_  
_𑀓𑀺𑀝𑁆𑀞𑀸𑀤𑀁 𑀯𑀺𑀬 𑀤𑀼𑀧𑁆𑀧𑀲𑀼𑀦𑁆”𑀢𑀺𑁇_  


---

726\. …  𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
