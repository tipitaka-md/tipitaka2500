# 𑁧𑁧𑁇𑁧𑁇𑁧 𑀲𑀗𑁆𑀓𑀺𑀘𑁆𑀘𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀗𑁆𑀓𑀺𑀘𑁆𑀘𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(597.)

910\. _“𑀓𑀺𑀁 𑀢𑀯𑀢𑁆𑀣𑁄 𑀯𑀦𑁂 𑀢𑀸𑀢,_  
_𑀉𑀚𑁆𑀚𑀼𑀳𑀸𑀦𑁄𑀯 𑀧𑀸𑀯𑀼𑀲𑁂;_  
_𑀯𑁂𑀭𑀫𑁆𑀪𑀸 𑀭𑀫𑀡𑀻𑀬𑀸 𑀢𑁂,_  
_𑀧𑀯𑀺𑀯𑁂𑀓𑁄 𑀳𑀺 𑀛𑀸𑀬𑀺𑀦𑀁”𑁇_  


(598.)

911\. _“𑀬𑀣𑀸 𑀅𑀩𑁆𑀪𑀸𑀦𑀺 𑀯𑁂𑀭𑀫𑁆𑀪𑁄,_  
_𑀯𑀸𑀢𑁄 𑀦𑀼𑀤𑀢𑀺 𑀧𑀸𑀯𑀼𑀲𑁂;_  
_𑀲𑀜𑁆𑀜𑀸 𑀫𑁂 𑀅𑀪𑀺𑀓𑀺𑀭𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀯𑁂𑀓𑀧𑀝𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑀸𑁇_  


(599.)

912\. _𑀅𑀧𑀡𑁆𑀟𑀭𑁄 𑀅𑀡𑁆𑀟𑀲𑀫𑁆𑀪𑀯𑁄,_  
_𑀲𑀻𑀯𑀣𑀺𑀓𑀸𑀬 𑀦𑀺𑀓𑁂𑀢𑀘𑀸𑀭𑀺𑀓𑁄;_  
_𑀉𑀧𑁆𑀧𑀸𑀤𑀬𑀢𑁂𑀯 𑀫𑁂 𑀲𑀢𑀺𑀁,_  
_𑀲𑀦𑁆𑀤𑁂𑀳𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁𑁇_  


(600.)

913\. _𑀬𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑁂 𑀦 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀘 𑀅𑀜𑁆𑀜𑁂 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺;_  
_𑀲 𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀁 𑀲𑁂𑀢𑀺,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀯𑀸𑁇_  


(601.)

914\. _𑀅𑀘𑁆𑀙𑁄𑀤𑀺𑀓𑀸 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀸,_  
_𑀕𑁄𑀦𑀗𑁆𑀕𑀼𑀮𑀫𑀺𑀕𑀸𑀬𑀼𑀢𑀸;_  
_𑀅𑀫𑁆𑀩𑀼𑀲𑁂𑀯𑀸𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(602.)

915\. _𑀯𑀲𑀺𑀢𑀁 𑀫𑁂 𑀅𑀭𑀜𑁆𑀜𑁂𑀲𑀼,_  
_𑀓𑀦𑁆𑀤𑀭𑀸𑀲𑀼 𑀕𑀼𑀳𑀸𑀲𑀼 𑀘;_  
_𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀧𑀦𑁆𑀢𑁂𑀲𑀼,_  
_𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑁂𑁇_  


(603.)

916\. _‘𑀇𑀫𑁂 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀼 𑀯𑀚𑁆𑀛𑀦𑁆𑀢𑀼,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀧𑁆𑀧𑁄𑀦𑁆𑀢𑀼 𑀧𑀸𑀡𑀺𑀦𑁄’;_  
_𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀦𑀭𑀺𑀬𑀁 𑀤𑁄𑀲𑀲𑀁𑀳𑀺𑀢𑀁𑁇_  


(604.)

917\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(605.)

918\. _𑀬𑀲𑁆𑀲 𑀘𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄𑁇_  


(606.)

919\. _𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀫𑀭𑀡𑀁,_  
_𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀓𑀸𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀲𑀁 𑀪𑀢𑀓𑁄 𑀬𑀣𑀸𑁇_  


(607.)

920\. _𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀫𑀭𑀡𑀁,_  
_𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀓𑀸𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄”𑀢𑀺𑁇_  


---

921\. …  𑀲𑀗𑁆𑀓𑀺𑀘𑁆𑀘𑁄 𑀣𑁂𑀭𑁄… 𑁇



---

922\. 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

923\. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



924\. _𑀲𑀗𑁆𑀓𑀺𑀘𑁆𑀘𑀣𑁂𑀭𑁄 𑀏𑀓𑁄𑀯,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁄 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀏𑀓𑀸𑀤𑀲𑀦𑀺𑀧𑀸𑀢𑀫𑁆𑀳𑀺,_  
_𑀕𑀸𑀣𑀸 𑀏𑀓𑀸𑀤𑀲𑁂𑀯 𑀘𑀸𑀢𑀺𑁇_
