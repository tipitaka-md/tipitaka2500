# 𑁫𑁇𑁧𑁇𑁧 𑀭𑀸𑀚𑀤𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀭𑀸𑀚𑀤𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(315.)

567. _“𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀤𑁆𑀤𑀲 𑀇𑀢𑁆𑀣𑀺𑀫𑀼𑀚𑁆𑀛𑀺𑀢𑀁;_  
_𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀲𑀼𑀲𑀸𑀦𑀲𑁆𑀫𑀺𑀁,_  
_𑀔𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀁 𑀓𑀺𑀫𑀺𑀳𑀻 𑀨𑀼𑀝𑀁𑁇_  


(316.)

568. _𑀬𑀜𑁆𑀳𑀺 𑀏𑀓𑁂 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀫𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀧𑀸𑀧𑀓𑀁;_  
_𑀓𑀸𑀫𑀭𑀸𑀕𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑀼,_  
_𑀅𑀦𑁆𑀥𑁄𑀯 𑀲𑀯𑀢𑀻 𑀅𑀳𑀼𑀁𑁇_  


(317.)

569. _𑀑𑀭𑀁 𑀑𑀤𑀦𑀧𑀸𑀓𑀫𑁆𑀳𑀸,_  
_𑀢𑀫𑁆𑀳𑀸 𑀞𑀸𑀦𑀸 𑀅𑀧𑀓𑁆𑀓𑀫𑀺𑀁;_  
_𑀲𑀢𑀺𑀫𑀸 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑀳𑀁,_  
_𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑁇_  


(318.)

570. _𑀢𑀢𑁄 𑀫𑁂 𑀫𑀦𑀲𑀻𑀓𑀸𑀭𑁄,_  
_𑀬𑁄𑀦𑀺𑀲𑁄 𑀉𑀤𑀧𑀚𑁆𑀚𑀣;_  
_𑀆𑀤𑀻𑀦𑀯𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑀼,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀲𑀫𑀢𑀺𑀝𑁆𑀞𑀣𑁇_  


(319.)

571. _𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺 𑀫𑁂,_  
_𑀧𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

572. …  𑀭𑀸𑀚𑀤𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
