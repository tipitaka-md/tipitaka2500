# 𑁫𑁇𑁧𑁇𑁧𑁧 𑀲𑁄𑀡𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀡𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(365.)

627\. _“𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀘 𑀫𑁂 𑀮𑀤𑁆𑀥𑀸,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀘𑀫𑁆𑀳𑀺 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀲𑁄 𑀘 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑁄,_  
_𑀯𑀺𑀳𑀸𑀭𑁂 𑀘 𑀲𑀳𑀸𑀯𑀲𑀺𑀁𑁇_  


(366.)

628\. _𑀩𑀳𑀼𑀤𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀁 𑀪𑀕𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂𑀢𑀺𑀦𑀸𑀫𑀬𑀺;_  
_𑀯𑀺𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀻 𑀢𑀤𑀸𑁇_  


(367.)

629\. _𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁,_  
_𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺 𑀕𑁄𑀢𑀫𑁄;_  
_𑀲𑀻𑀳𑁄 𑀲𑁂𑀮𑀕𑀼𑀳𑀸𑀬𑀁𑀯,_  
_𑀧𑀳𑀻𑀦𑀪𑀬𑀪𑁂𑀭𑀯𑁄𑁇_  


(368.)

630\. _𑀢𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑁄;_  
_𑀲𑁄𑀡𑁄 𑀅𑀪𑀸𑀲𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑁇_  


(369.)

631\. _𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀪𑀸𑀯𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀲𑀁;_  
_𑀧𑀧𑁆𑀧𑀼𑀬𑁆𑀬 𑀧𑀭𑀫𑀁 𑀲𑀦𑁆𑀢𑀺𑀁,_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀢𑁆𑀬𑀦𑀸𑀲𑀯𑁄”𑀢𑀺𑁇_  


---

632\. …  𑀲𑁄𑀡𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀢𑁆𑀣𑁂𑀭𑁄… 𑁇
