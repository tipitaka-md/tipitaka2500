# 𑁫𑁇𑁧𑁇𑁬 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(340.)

597\. _“𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀢 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀤𑀺𑀁 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀁 𑀅𑀕𑀸;_  
_𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑀬𑀺𑀁𑁇_  


(341.)

598\. _𑀬𑀚𑀺𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑁂 𑀬𑀜𑁆𑀜𑁂,_  
_𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑁆𑀢𑀁 𑀚𑀼𑀳𑀺𑀁 𑀅𑀳𑀁;_  
_‘𑀏𑀲𑀸 𑀲𑀼𑀤𑁆𑀥𑀻’𑀢𑀺 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑁄,_  
_𑀅𑀦𑁆𑀥𑀪𑀽𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄𑁇_  


(342.)

599\. _𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑁄,_  
_𑀧𑀭𑀸𑀫𑀸𑀲𑁂𑀦 𑀫𑁄𑀳𑀺𑀢𑁄;_  
_𑀅𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑀁 𑀲𑀼𑀤𑁆𑀥𑀺𑀁,_  
_𑀅𑀦𑁆𑀥𑀪𑀽𑀢𑁄 𑀅𑀯𑀺𑀤𑁆𑀤𑀲𑀼𑁇_  


(343.)

600\. _𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀳𑀻𑀦𑀸 𑀫𑁂,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀧𑀤𑀸𑀮𑀺𑀢𑀸;_  
_𑀚𑀼𑀳𑀸𑀫𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀕𑁆𑀕𑀺𑀁,_  
_𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀣𑀸𑀕𑀢𑀁𑁇_  


(344.)

601\. _𑀫𑁄𑀳𑀸 𑀲𑀩𑁆𑀩𑁂 𑀧𑀳𑀻𑀦𑀸 𑀫𑁂,_  
_𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀧𑀤𑀸𑀮𑀺𑀢𑀸;_  
_𑀯𑀺𑀓𑁆𑀔𑀻𑀡𑁄 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇_  


---

602\. …  𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄… 𑁇
