# 𑁫𑁇𑁧𑁇𑁨 𑀲𑀼𑀪𑀽𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀪𑀽𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(320.)

573\. _“𑀅𑀬𑁄𑀕𑁂 𑀬𑀼𑀜𑁆𑀚𑀫𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀘𑁆𑀘𑀫𑀺𑀘𑁆𑀙𑀓𑁄;_  
_𑀘𑀭𑀜𑁆𑀘𑁂 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_‘𑀢𑀁 𑀫𑁂 𑀤𑀼𑀩𑁆𑀪𑀕𑀮𑀓𑁆𑀔𑀡𑀁’𑁇_  


(321.)

574\. _𑀅𑀩𑁆𑀩𑀽𑀍𑀳𑀁 𑀅𑀖𑀕𑀢𑀁 𑀯𑀺𑀚𑀺𑀢𑀁,_  
_𑀏𑀓𑀜𑁆𑀘𑁂 𑀑𑀲𑁆𑀲𑀚𑁂𑀬𑁆𑀬 𑀓𑀮𑀻𑀯 𑀲𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀘𑁂 𑀑𑀲𑁆𑀲𑀚𑁂𑀬𑁆𑀬 𑀅𑀦𑁆𑀥𑁄𑀯 𑀲𑀺𑀬𑀸,_  
_𑀲𑀫𑀯𑀺𑀲𑀫𑀲𑁆𑀲 𑀅𑀤𑀲𑁆𑀲𑀦𑀢𑁄𑁇_  


(322.)

575\. _𑀬𑀜𑁆𑀳𑀺 𑀓𑀬𑀺𑀭𑀸 𑀢𑀜𑁆𑀳𑀺 𑀯𑀤𑁂,_  
_𑀬𑀁 𑀦 𑀓𑀬𑀺𑀭𑀸 𑀦 𑀢𑀁 𑀯𑀤𑁂;_  
_𑀅𑀓𑀭𑁄𑀦𑁆𑀢𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀁,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


(323.)

576\. _𑀬𑀣𑀸𑀧𑀺 𑀭𑀼𑀘𑀺𑀭𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑀁 𑀅𑀕𑀦𑁆𑀥𑀓𑀁;_  
_𑀏𑀯𑀁 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸 𑀯𑀸𑀘𑀸,_  
_𑀅𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀅𑀓𑀼𑀩𑁆𑀩𑀢𑁄𑁇_  


(324.)

577\. _𑀬𑀣𑀸𑀧𑀺 𑀭𑀼𑀘𑀺𑀭𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀕𑀦𑁆𑀥𑀓𑀁;_  
_𑀏𑀯𑀁 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸 𑀯𑀸𑀘𑀸,_  
_𑀲𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀓𑀼𑀩𑁆𑀩𑀢𑁄”𑀢𑀺𑁇_  


---

578\. …  𑀲𑀼𑀪𑀽𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
