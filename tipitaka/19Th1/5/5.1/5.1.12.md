# 𑁫𑁇𑁧𑁇𑁧𑁨 𑀓𑁄𑀲𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀓𑁄𑀲𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(370.)

633\. _“𑀬𑁄 𑀯𑁂 𑀕𑀭𑀽𑀦𑀁 𑀯𑀘𑀦𑀜𑁆𑀜𑀼 𑀥𑀻𑀭𑁄,_  
_𑀯𑀲𑁂 𑀘 𑀢𑀫𑁆𑀳𑀺 𑀚𑀦𑀬𑁂𑀣 𑀧𑁂𑀫𑀁;_  
_𑀲𑁄 𑀪𑀢𑁆𑀢𑀺𑀫𑀸 𑀦𑀸𑀫 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄,_  
_𑀜𑀢𑁆𑀯𑀸 𑀘 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀲𑁂𑀲𑀺 𑀅𑀲𑁆𑀲𑁇_  


(371.)

634\. _𑀬𑀁 𑀆𑀧𑀤𑀸 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀸 𑀉𑀍𑀆𑀭𑀸,_  
_𑀦𑀓𑁆𑀔𑀫𑁆𑀪𑀬𑀦𑁆𑀢𑁂 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀬𑀦𑁆𑀢𑀁;_  
_𑀲𑁄 𑀣𑀸𑀫𑀯𑀸 𑀦𑀸𑀫 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄,_  
_𑀜𑀢𑁆𑀯𑀸 𑀘 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀲𑁂𑀲𑀺 𑀅𑀲𑁆𑀲𑁇_  


(372.)

635\. _𑀬𑁄 𑀯𑁂 𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑀯 𑀞𑀺𑀢𑁄 𑀅𑀦𑁂𑀚𑁄,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀜𑁄 𑀦𑀺𑀧𑀼𑀡𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻;_  
_𑀅𑀲𑀁𑀳𑀸𑀭𑀺𑀬𑁄 𑀦𑀸𑀫 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄,_  
_𑀜𑀢𑁆𑀯𑀸 𑀘 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀲𑁂𑀲𑀺 𑀅𑀲𑁆𑀲𑁇_  


(373.)

636\. _𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻;_  
_𑀲𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀦𑀸𑀫 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄,_  
_𑀜𑀢𑁆𑀯𑀸 𑀘 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀲𑁂𑀲𑀺 𑀅𑀲𑁆𑀲𑁇_  


(374.)

637\. _𑀅𑀢𑁆𑀣𑀜𑁆𑀘 𑀬𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲,_  
_𑀅𑀢𑁆𑀣𑀜𑁆𑀘 𑀜𑀢𑁆𑀯𑀸𑀦 𑀢𑀣𑀸 𑀓𑀭𑁄𑀢𑀺;_  
_𑀅𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀲 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄,_  
_𑀜𑀢𑁆𑀯𑀸 𑀘 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀲𑁂𑀲𑀺 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇_  


---

638\. …  𑀓𑁄𑀲𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇



---

639\. 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

640\. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



641\. _𑀭𑀸𑀚𑀤𑀢𑁆𑀢𑁄 𑀲𑀼𑀪𑀽𑀢𑁄 𑀘,_  
_𑀕𑀺𑀭𑀺𑀫𑀸𑀦𑀦𑁆𑀤𑀲𑀼𑀫𑀦𑀸;_  
_𑀯𑀟𑁆𑀠𑁄 𑀘 𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄,_  
_𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑀯𑀓𑁆𑀓𑀮𑀻𑁇_  


642\. _𑀯𑀺𑀚𑀺𑀢𑁄 𑀬𑀲𑀤𑀢𑁆𑀢𑁄 𑀘,_  
_𑀲𑁄𑀡𑁄 𑀓𑁄𑀲𑀺𑀬𑀲𑀯𑁆𑀳𑀬𑁄;_  
_𑀲𑀝𑁆𑀞𑀺 𑀘 𑀧𑀜𑁆𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀣𑁂𑀭𑀸 𑀘 𑀏𑀢𑁆𑀣 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀢𑀺𑁇_
