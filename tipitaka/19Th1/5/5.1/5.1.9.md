# 𑁫𑁇𑁧𑁇𑁯 𑀯𑀺𑀚𑀺𑀢𑀲𑁂𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀚𑀺𑀢𑀲𑁂𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(355.)

615\. _“𑀑𑀮𑀕𑁆𑀕𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑁂 𑀘𑀺𑀢𑁆𑀢,_  
_𑀆𑀡𑀺𑀤𑁆𑀯𑀸𑀭𑁂𑀯 𑀳𑀢𑁆𑀣𑀺𑀦𑀁;_  
_𑀦 𑀢𑀁 𑀧𑀸𑀧𑁂 𑀦𑀺𑀬𑁄𑀚𑁂𑀲𑁆𑀲𑀁,_  
_𑀓𑀸𑀫𑀚𑀸𑀮 𑀲𑀭𑀻𑀭𑀚𑁇_  


(356.)

616\. _𑀢𑁆𑀯𑀁 𑀑𑀮𑀕𑁆𑀕𑁄 𑀦 𑀕𑀘𑁆𑀙𑀲𑀺,_  
_𑀤𑁆𑀯𑀸𑀭𑀯𑀺𑀯𑀭𑀁 𑀕𑀚𑁄𑀯 𑀅𑀮𑀪𑀦𑁆𑀢𑁄;_  
_𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀓𑀮𑀺 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁,_  
_𑀧𑀲𑀓𑁆𑀓 𑀧𑀸𑀧𑀭𑀢𑁄 𑀘𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(357.)

617\. _𑀬𑀣𑀸 𑀓𑀼𑀜𑁆𑀚𑀭𑀁 𑀅𑀤𑀦𑁆𑀢𑀁,_  
_𑀦𑀯𑀕𑁆𑀕𑀳𑀫𑀗𑁆𑀓𑀼𑀲𑀕𑁆𑀕𑀳𑁄;_  
_𑀩𑀮𑀯𑀸 𑀆𑀯𑀢𑁆𑀢𑁂𑀢𑀺 𑀅𑀓𑀸𑀫𑀁,_  
_𑀏𑀯𑀁 𑀆𑀯𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀁 𑀢𑀁𑁇_  


(358.)

618\. _𑀬𑀣𑀸 𑀯𑀭𑀳𑀬𑀤𑀫𑀓𑀼𑀲𑀮𑁄,_  
_𑀲𑀸𑀭𑀣𑀺 𑀧𑀯𑀭𑁄 𑀤𑀫𑁂𑀢𑀺 𑀆𑀚𑀜𑁆𑀜𑀁;_  
_𑀏𑀯𑀁 𑀤𑀫𑀬𑀺𑀲𑁆𑀲𑀁 𑀢𑀁,_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀜𑁆𑀘𑀲𑀼 𑀩𑀮𑁂𑀲𑀼𑁇_  


(359.)

619\. _𑀲𑀢𑀺𑀬𑀸 𑀢𑀁 𑀦𑀺𑀩𑀦𑁆𑀥𑀺𑀲𑁆𑀲𑀁,_  
_𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑁂 𑀤𑀫𑁂𑀲𑁆𑀲𑀸𑀫𑀺;_  
_𑀯𑀻𑀭𑀺𑀬𑀥𑀼𑀭𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑁄,_  
_𑀦 𑀬𑀺𑀢𑁄 𑀤𑀽𑀭𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑁂 𑀘𑀺𑀢𑁆𑀢𑀸”𑀢𑀺𑁇_  


---

620\. …  𑀯𑀺𑀚𑀺𑀢𑀲𑁂𑀦𑁄 𑀣𑁂𑀭𑁄… 𑁇
