# 𑁫𑁇𑁧𑁇𑁮 𑀯𑀓𑁆𑀓𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀯𑀓𑁆𑀓𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(350.)

609\. _“𑀯𑀸𑀢𑀭𑁄𑀕𑀸𑀪𑀺𑀦𑀻𑀢𑁄 𑀢𑁆𑀯𑀁,_  
_𑀯𑀺𑀳𑀭𑀁 𑀓𑀸𑀦𑀦𑁂 𑀯𑀦𑁂;_  
_𑀧𑀯𑀺𑀝𑁆𑀞𑀕𑁄𑀘𑀭𑁂 𑀮𑀽𑀔𑁂,_  
_𑀓𑀣𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(351.)

610\. _“𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑁂𑀦 𑀯𑀺𑀧𑀼𑀮𑁂𑀦,_  
_𑀨𑀭𑀫𑀸𑀦𑁄 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀁;_  
_𑀮𑀽𑀔𑀫𑁆𑀧𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀪𑁄𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀸𑀦𑀦𑁂𑁇_  


(352.)

611\. _𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀩𑀮𑀸𑀦𑀺 𑀘;_  
_𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑀦𑀺 𑀘 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀸𑀦𑀦𑁂𑁇_  


(353.)

612\. _𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁂 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁂,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁂;_  
_𑀲𑀫𑀕𑁆𑀕𑁂 𑀲𑀳𑀺𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀸𑀦𑀦𑁂𑁇_  


(354.)

613\. _𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀅𑀕𑁆𑀕𑀁 𑀤𑀦𑁆𑀢𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁,_  
_𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀸𑀦𑀦𑁂”𑀢𑀺𑁇_  


---

614\. …  𑀯𑀓𑁆𑀓𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑁄… 𑁇
