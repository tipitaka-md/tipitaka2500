# 𑀦𑀺𑀤𑀸𑀦𑀕𑀸𑀣𑀸

### 𑀦𑀺𑀤𑀸𑀦𑀕𑀸𑀣𑀸

(814.)

2. _𑀲𑀻𑀳𑀸𑀦𑀁𑀯 𑀦𑀤𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀤𑀸𑀞𑀻𑀦𑀁 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑁂;_  
_𑀲𑀼𑀡𑀸𑀣 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀕𑀸𑀣𑀸 𑀅𑀢𑁆𑀣𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸𑁇_  


3. _𑀬𑀣𑀸𑀦𑀸𑀫𑀸 𑀬𑀣𑀸𑀕𑁄𑀢𑁆𑀢𑀸,_  
_𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄;_  
_𑀬𑀣𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀺𑀳𑀭𑀺𑀁𑀲𑀼 𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑀸𑁇_  


4. _𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀘𑀼𑀢𑀁 𑀧𑀤𑀁;_  
_𑀓𑀢𑀦𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀸,_  
_𑀇𑀫𑀫𑀢𑁆𑀣𑀫𑀪𑀸𑀲𑀺𑀲𑀼𑀁𑁇_
