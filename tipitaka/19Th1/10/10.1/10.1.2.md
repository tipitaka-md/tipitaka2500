# 𑁧𑁦𑁇𑁧𑁇𑁨 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(537.)

841\. _“𑀧𑀼𑀭𑀢𑁄 𑀧𑀘𑁆𑀙𑀢𑁄 𑀯𑀸𑀧𑀺,_  
_𑀅𑀧𑀭𑁄 𑀘𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀢𑀻𑀯 𑀨𑀸𑀲𑀼 𑀪𑀯𑀢𑀺,_  
_𑀏𑀓𑀲𑁆𑀲 𑀯𑀲𑀢𑁄 𑀯𑀦𑁂𑁇_  


(538.)

842\. _𑀳𑀦𑁆𑀤 𑀏𑀓𑁄 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀅𑀭𑀜𑁆𑀜𑀁 𑀩𑀼𑀤𑁆𑀥𑀯𑀡𑁆𑀡𑀺𑀢𑀁;_  
_𑀨𑀸𑀲𑀼 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


(539.)

843\. _𑀬𑁄𑀕𑀺-𑀧𑀻𑀢𑀺𑀓𑀭𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀫𑀢𑁆𑀢𑀓𑀼𑀜𑁆𑀚𑀭𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀏𑀓𑁄 𑀅𑀢𑁆𑀢𑀯𑀲𑀻 𑀔𑀺𑀧𑁆𑀧𑀁,_  
_𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀸𑀦𑀦𑀁𑁇_  


(540.)

844\. _𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁂 𑀲𑀻𑀢𑀯𑀦𑁂,_  
_𑀲𑀻𑀢𑀮𑁂 𑀕𑀺𑀭𑀺𑀓𑀦𑁆𑀤𑀭𑁂;_  
_𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸,_  
_𑀘𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀏𑀓𑀓𑁄𑁇_  


(541.)

845\. _𑀏𑀓𑀸𑀓𑀺𑀬𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄,_  
_𑀭𑀫𑀡𑀻𑀬𑁂 𑀫𑀳𑀸𑀯𑀦𑁂;_  
_𑀓𑀤𑀸𑀳𑀁 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁄 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(542.)

846\. _𑀏𑀯𑀁 𑀫𑁂 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑀲𑁆𑀲,_  
_𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀼;_  
_𑀲𑀸𑀥𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁𑀬𑁂𑀯,_  
_𑀦𑀸𑀜𑁆𑀜𑁄 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀓𑀸𑀭𑀓𑁄𑁇_  


(543.)

847\. _𑀏𑀲 𑀩𑀦𑁆𑀥𑀸𑀫𑀺 𑀲𑀦𑁆𑀦𑀸𑀳𑀁,_  
_𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀸𑀦𑀦𑀁;_  
_𑀦 𑀢𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(544.)

848\. _𑀫𑀸𑀮𑀼𑀢𑁂 𑀉𑀧𑀯𑀸𑀬𑀦𑁆𑀢𑁂,_  
_𑀲𑀻𑀢𑁂 𑀲𑀼𑀭𑀪𑀺𑀕𑀦𑁆𑀥𑀺𑀓𑁂;_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀁 𑀤𑀸𑀮𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀦𑀕𑀫𑀼𑀤𑁆𑀥𑀦𑀺𑁇_  


(545.)

849\. _𑀯𑀦𑁂 𑀓𑀼𑀲𑀼𑀫𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑁂,_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑁂 𑀦𑀽𑀦 𑀲𑀻𑀢𑀮𑁂;_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀼𑀔𑁂𑀦 𑀲𑀼𑀔𑀺𑀢𑁄,_  
_𑀭𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑁂𑁇_  


(546.)

850\. _𑀲𑁄𑀳𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀘𑀦𑁆𑀤𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑁄 𑀬𑀣𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇_  


---

851\. …  𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
