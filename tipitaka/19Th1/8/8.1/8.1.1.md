# 𑁮𑁇𑁧𑁇𑁧 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(494.)

787\. _“𑀓𑀫𑁆𑀫𑀁 𑀩𑀳𑀼𑀓𑀁 𑀦 𑀓𑀸𑀭𑀬𑁂,_  
_𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀚𑀦𑀁 𑀦 𑀉𑀬𑁆𑀬𑀫𑁂;_  
_𑀲𑁄 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀭𑀲𑀸𑀦𑀼𑀕𑀺𑀤𑁆𑀥𑁄,_  
_𑀅𑀢𑁆𑀣𑀁 𑀭𑀺𑀜𑁆𑀘𑀢𑀺 𑀬𑁄 𑀲𑀼𑀔𑀸𑀥𑀺𑀯𑀸𑀳𑁄𑁇_  


(495.)

788\. _𑀧𑀗𑁆𑀓𑁄𑀢𑀺 𑀳𑀺 𑀦𑀁 𑀅𑀯𑁂𑀤𑀬𑀼𑀁,_  
_𑀬𑀸𑀬𑀁 𑀯𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀸 𑀓𑀼𑀮𑁂𑀲𑀼;_  
_𑀲𑀼𑀔𑀼𑀫𑀁 𑀲𑀮𑁆𑀮𑀁 𑀤𑀼𑀭𑀼𑀩𑁆𑀩𑀳𑀁,_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀤𑀼𑀚𑁆𑀚𑀳𑁄𑁇_  


(496.)

789\. _𑀦 𑀧𑀭𑀲𑁆𑀲𑀼𑀧𑀦𑀺𑀥𑀸𑀬,_  
_𑀓𑀫𑁆𑀫𑀁 𑀫𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑀁;_  
_𑀅𑀢𑁆𑀢𑀦𑀸 𑀢𑀁 𑀦 𑀲𑁂𑀯𑁂𑀬𑁆𑀬,_  
_𑀓𑀫𑁆𑀫𑀩𑀦𑁆𑀥𑀽 𑀳𑀺 𑀫𑀸𑀢𑀺𑀬𑀸𑁇_  


(497.)

790\. _𑀦 𑀧𑀭𑁂 𑀯𑀘𑀦𑀸 𑀘𑁄𑀭𑁄,_  
_𑀦 𑀧𑀭𑁂 𑀯𑀘𑀦𑀸 𑀫𑀼𑀦𑀺;_  
_𑀅𑀢𑁆𑀢𑀸 𑀘 𑀦𑀁 𑀬𑀣𑀸𑀯𑁂𑀤𑀺,_  
_𑀤𑁂𑀯𑀸𑀧𑀺 𑀦𑀁 𑀢𑀣𑀸 𑀯𑀺𑀤𑀽𑁇_  


(498.)

791\. _𑀧𑀭𑁂 𑀘 𑀦 𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀬𑀫𑀸𑀫𑀲𑁂;_  
_𑀬𑁂 𑀘 𑀢𑀢𑁆𑀣 𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀢𑀢𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀫𑁂𑀥𑀕𑀸𑁇_  


(499.)

792\. _𑀚𑀻𑀯𑀢𑁂 𑀯𑀸𑀧𑀺 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀅𑀧𑀺 𑀯𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀅𑀮𑀸𑀪𑁂𑀦,_  
_𑀯𑀺𑀢𑁆𑀢𑀯𑀸𑀧𑀺 𑀦 𑀚𑀻𑀯𑀢𑀺𑁇_  


(500.)

793\. _𑀲𑀩𑁆𑀩𑀁 𑀲𑀼𑀡𑀸𑀢𑀺 𑀲𑁄𑀢𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑀸;_  
_𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀥𑀻𑀭𑁄,_  
_𑀲𑀩𑁆𑀩𑀁 𑀉𑀚𑁆𑀛𑀺𑀢𑀼𑀫𑀭𑀳𑀢𑀺𑁇_  


(501.)

794\. _𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑁆𑀲 𑀬𑀣𑀸 𑀅𑀦𑁆𑀥𑁄,_  
_𑀲𑁄𑀢𑀯𑀸 𑀩𑀥𑀺𑀭𑁄 𑀬𑀣𑀸;_  
_𑀧𑀜𑁆𑀜𑀯𑀸𑀲𑁆𑀲 𑀬𑀣𑀸 𑀫𑀽𑀕𑁄,_  
_𑀩𑀮𑀯𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁄𑀭𑀺𑀯;_  
_𑀅𑀣 𑀅𑀢𑁆𑀣𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀲𑀬𑁂𑀣 𑀫𑀢𑀲𑀸𑀬𑀺𑀓𑀦𑁆”𑀢𑀺𑁇_  


---

795\. …  𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 𑀣𑁂𑀭𑁄… 𑁇
