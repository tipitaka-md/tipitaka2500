# 𑁪𑁇𑁧𑁇𑁧𑁨 𑀫𑀼𑀤𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀼𑀤𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(311.)

558\. _“𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀚𑀻𑀯𑀺𑀓𑀢𑁆𑀣𑁄𑀳𑀁,_  
_𑀮𑀤𑁆𑀥𑀸𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁;_  
_𑀢𑀢𑁄 𑀲𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀁,_  
_𑀤𑀍𑀳𑀯𑀻𑀭𑀺𑀬𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑀺𑀁𑁇_  


(312.)

559\. _𑀓𑀸𑀫𑀁 𑀪𑀺𑀚𑁆𑀚𑀢𑀼𑀬𑀁 𑀓𑀸𑀬𑁄,_  
_𑀫𑀁𑀲𑀧𑁂𑀲𑀻 𑀯𑀺𑀲𑀻𑀬𑀭𑀼𑀁;_  
_𑀉𑀪𑁄 𑀚𑀡𑁆𑀡𑀼𑀓𑀲𑀦𑁆𑀥𑀻𑀳𑀺,_  
_𑀚𑀗𑁆𑀖𑀸𑀬𑁄 𑀧𑀧𑀢𑀦𑁆𑀢𑀼 𑀫𑁂𑁇_  


(313.)

560\. _𑀦𑀸𑀲𑀺𑀲𑁆𑀲𑀁 𑀦 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀯𑀺𑀳𑀸𑀭𑀸 𑀘 𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑁂;_  
_𑀦𑀧𑀺 𑀧𑀲𑁆𑀲𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑁆𑀲𑀁,_  
_𑀢𑀡𑁆𑀳𑀸𑀲𑀮𑁆𑀮𑁂 𑀅𑀦𑀽𑀳𑀢𑁂𑁇_  


(314.)

561\. _𑀢𑀲𑁆𑀲 𑀫𑁂𑀯𑀁 𑀯𑀺𑀳𑀭𑀢𑁄,_  
_𑀧𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀧𑀭𑀓𑁆𑀓𑀫𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

562\. …  𑀫𑀼𑀤𑀺𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇



---

563\. 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

564\. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



565\. _𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀪𑀕𑀼 𑀘,_  
_𑀲𑀪𑀺𑀬𑁄 𑀦𑀦𑁆𑀤𑀓𑁄𑀧𑀺 𑀘;_  
_𑀚𑀫𑁆𑀩𑀼𑀓𑁄 𑀲𑁂𑀦𑀓𑁄 𑀣𑁂𑀭𑁄,_  
_𑀲𑀫𑁆𑀪𑀽𑀢𑁄 𑀭𑀸𑀳𑀼𑀮𑁄𑀧𑀺 𑀘𑁇_  


566\. _𑀪𑀯𑀢𑀺 𑀘𑀦𑁆𑀤𑀦𑁄 𑀣𑁂𑀭𑁄,_  
_𑀤𑀲𑁂𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑀸;_  
_𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀲𑀧𑁆𑀧𑀓𑁄 𑀣𑁂𑀭𑁄,_  
_𑀫𑀼𑀤𑀺𑀢𑁄 𑀘𑀸𑀧𑀺 𑀢𑁂 𑀢𑀬𑁄;_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀤𑁆𑀯𑁂 𑀘 𑀧𑀜𑁆𑀜𑀸𑀲,_  
_𑀣𑁂𑀭𑀸 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀢𑁂𑀭𑀲𑀸𑀢𑀺𑁇_
