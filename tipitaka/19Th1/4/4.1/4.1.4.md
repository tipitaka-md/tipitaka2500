# 𑁪𑁇𑁧𑁇𑁪 𑀦𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀦𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(279.)

518\. _“𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀧𑀽𑀭𑁂 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑁂,_  
_𑀫𑀸𑀭𑀧𑀓𑁆𑀔𑁂 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁂;_  
_𑀦𑀯𑀲𑁄𑀢𑀸𑀦𑀺 𑀢𑁂 𑀓𑀸𑀬𑁂,_  
_𑀬𑀸𑀦𑀺 𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


(280.)

519\. _𑀫𑀸 𑀧𑀼𑀭𑀸𑀡𑀁 𑀅𑀫𑀜𑁆𑀜𑀺𑀢𑁆𑀣𑁄,_  
_𑀫𑀸𑀲𑀸𑀤𑁂𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑁂;_  
_𑀲𑀕𑁆𑀕𑁂𑀧𑀺 𑀢𑁂 𑀦 𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀓𑀺𑀫𑀗𑁆𑀕𑀁 𑀧𑀦 𑀫𑀸𑀦𑀼𑀲𑁂𑁇_  


(281.)

520\. _𑀬𑁂 𑀘 𑀔𑁄 𑀩𑀸𑀮𑀸 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑀸,_  
_𑀤𑀼𑀫𑁆𑀫𑀦𑁆𑀢𑀻 𑀫𑁄𑀳𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀢𑀸𑀤𑀺𑀲𑀸 𑀢𑀢𑁆𑀣 𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀫𑀸𑀭𑀔𑀺𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀩𑀦𑁆𑀥𑀦𑁂𑁇_  


(282.)

521\. _𑀬𑁂𑀲𑀁 𑀭𑀸𑀕𑁄 𑀘 𑀤𑁄𑀲𑁄 𑀘,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀭𑀸𑀚𑀺𑀢𑀸;_  
_𑀢𑀸𑀤𑀻 𑀢𑀢𑁆𑀣 𑀦 𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀙𑀺𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢𑀸 𑀅𑀩𑀦𑁆𑀥𑀦𑀸”𑀢𑀺𑁇_  


---

522\. …  𑀦𑀦𑁆𑀤𑀓𑁄 𑀣𑁂𑀭𑁄… 𑁇
