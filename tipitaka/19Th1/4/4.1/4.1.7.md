# 𑁪𑁇𑁧𑁇𑁭 𑀲𑀫𑁆𑀪𑀽𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀫𑁆𑀪𑀽𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(291.)

533\. _“𑀬𑁄 𑀤𑀦𑁆𑀥𑀓𑀸𑀮𑁂 𑀢𑀭𑀢𑀺,_  
_𑀢𑀭𑀡𑀻𑀬𑁂 𑀘 𑀤𑀦𑁆𑀥𑀬𑁂;_  
_𑀅𑀬𑁄𑀦𑀺 𑀲𑀁𑀯𑀺𑀥𑀸𑀦𑁂𑀦,_  
_𑀩𑀸𑀮𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(292.)

534\. _𑀢𑀲𑁆𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀓𑀸𑀍𑀅𑀧𑀓𑁆𑀔𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸;_  
_𑀆𑀬𑀲𑀓𑁆𑀬𑀜𑁆𑀘 𑀧𑀧𑁆𑀧𑁄𑀢𑀺,_  
_𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀘 𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇_  


(293.)

535\. _𑀬𑁄 𑀤𑀦𑁆𑀥𑀓𑀸𑀮𑁂 𑀤𑀦𑁆𑀥𑁂𑀢𑀺,_  
_𑀢𑀭𑀡𑀻𑀬𑁂 𑀘 𑀢𑀸𑀭𑀬𑁂;_  
_𑀬𑁄𑀦𑀺𑀲𑁄 𑀲𑀁𑀯𑀺𑀥𑀸𑀦𑁂𑀦,_  
_𑀲𑀼𑀔𑀁 𑀧𑀧𑁆𑀧𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


(294.)

536\. _𑀢𑀲𑁆𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀲𑀼𑀓𑁆𑀓𑀧𑀓𑁆𑀔𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸;_  
_𑀬𑀲𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀧𑀧𑁆𑀧𑁄𑀢𑀺,_  
_𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀦 𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_  


---

537\. …  𑀲𑀫𑁆𑀪𑀽𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
