# 𑁧𑁮𑁇𑁧𑁇𑁧 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(1054.)

1409\. _“𑀦 𑀕𑀡𑁂𑀦 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄 𑀘𑀭𑁂,_  
_𑀯𑀺𑀫𑀦𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀥𑀺 𑀤𑀼𑀮𑁆𑀮𑀪𑁄;_  
_𑀦𑀸𑀦𑀸𑀚𑀦𑀲𑀗𑁆𑀕𑀳𑁄 𑀤𑀼𑀔𑁄,_  
_𑀇𑀢𑀺 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀕𑀡𑀁 𑀦 𑀭𑁄𑀘𑀬𑁂𑁇_  


(1055.)

1410\. _𑀦 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀩𑁆𑀩𑀚𑁂 𑀫𑀼𑀦𑀺,_  
_𑀯𑀺𑀫𑀦𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀥𑀺 𑀤𑀼𑀮𑁆𑀮𑀪𑁄;_  
_𑀲𑁄 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀭𑀲𑀸𑀦𑀼𑀕𑀺𑀤𑁆𑀥𑁄,_  
_𑀅𑀢𑁆𑀣𑀁 𑀭𑀺𑀜𑁆𑀘𑀢𑀺 𑀬𑁄 𑀲𑀼𑀔𑀸𑀯𑀳𑁄𑁇_  


(1056.)

1411\. _𑀧𑀗𑁆𑀓𑁄𑀢𑀺 𑀳𑀺 𑀦𑀁 𑀅𑀯𑁂𑀤𑀬𑀼𑀁,_  
_𑀬𑀸𑀬𑀁 𑀯𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀸 𑀓𑀼𑀮𑁂𑀲𑀼;_  
_𑀲𑀼𑀔𑀼𑀫𑀁 𑀲𑀮𑁆𑀮𑀁 𑀤𑀼𑀭𑀼𑀩𑁆𑀩𑀳𑀁,_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀤𑀼𑀚𑁆𑀚𑀳𑁄𑁇_  


(1057.)

1412\. _𑀲𑁂𑀦𑀸𑀲𑀦𑀫𑁆𑀳𑀸 𑀑𑀭𑀼𑀬𑁆𑀳,_  
_𑀦𑀕𑀭𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑀁;_  
_𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀓𑀼𑀝𑁆𑀞𑀺𑀁,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑁇_  


(1058.)

1413\. _𑀲𑁄 𑀫𑁂 𑀧𑀓𑁆𑀓𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦,_  
_𑀆𑀮𑁄𑀧𑀁 𑀉𑀧𑀦𑀸𑀫𑀬𑀺;_  
_𑀆𑀮𑁄𑀧𑀁 𑀧𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀅𑀗𑁆𑀕𑀼𑀮𑀺 𑀘𑁂𑀢𑁆𑀣 𑀙𑀺𑀚𑁆𑀚𑀣𑁇_  


(1059.)

1414\. _𑀓𑀼𑀝𑁆𑀝𑀫𑀽𑀮𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀆𑀮𑁄𑀧𑀁 𑀢𑀁 𑀅𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑀁;_  
_𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑁂 𑀯𑀸 𑀪𑀼𑀢𑁆𑀢𑁂 𑀯𑀸,_  
_𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀁 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(1060.)

1415\. _𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑀧𑀺𑀡𑁆𑀟𑁄 𑀆𑀳𑀸𑀭𑁄,_  
_𑀧𑀽𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀑𑀲𑀥𑀁;_  
_𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁,_  
_𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀜𑁆𑀘 𑀘𑀻𑀯𑀭𑀁;_  
_𑀬𑀲𑁆𑀲𑁂𑀢𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀪𑀼𑀢𑁆𑀯𑀸,_  
_𑀲 𑀯𑁂 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑁄 𑀦𑀭𑁄𑁇_  


(1061.)

1416\. _𑀬𑀢𑁆𑀣 𑀏𑀓𑁂 𑀯𑀺𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺,_  
_𑀆𑀭𑀼𑀳𑀦𑁆𑀢𑀸 𑀲𑀺𑀮𑀼𑀘𑁆𑀘𑀬𑀁;_  
_𑀢𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀇𑀤𑁆𑀥𑀺𑀩𑀮𑁂𑀦𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑁄,_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀪𑀺𑀭𑀽𑀳𑀢𑀺𑁇_  


(1062.)

1417\. _𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄,_  
_𑀲𑁂𑀮𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀓𑀲𑁆𑀲𑀧𑁄;_  
_𑀛𑀸𑀬𑀢𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄,_  
_𑀧𑀳𑀻𑀦𑀪𑀬𑀪𑁂𑀭𑀯𑁄𑁇_  


(1063.)

1418\. _𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄,_  
_𑀲𑁂𑀮𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀓𑀲𑁆𑀲𑀧𑁄;_  
_𑀛𑀸𑀬𑀢𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄,_  
_𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇_  


(1064.)

1419\. _𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄,_  
_𑀲𑁂𑀮𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀓𑀲𑁆𑀲𑀧𑁄;_  
_𑀛𑀸𑀬𑀢𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁄 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(1065.)

1420\. _𑀓𑀭𑁂𑀭𑀺𑀫𑀸𑀮𑀸𑀯𑀺𑀢𑀢𑀸,_  
_𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑀸 𑀫𑀦𑁄𑀭𑀫𑀸;_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀸𑀪𑀺𑀭𑀼𑀤𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(1066.)

1421\. _𑀦𑀻𑀮𑀩𑁆𑀪𑀯𑀡𑁆𑀡𑀸 𑀭𑀼𑀘𑀺𑀭𑀸,_  
_𑀯𑀸𑀭𑀺𑀲𑀻𑀢𑀸 𑀲𑀼𑀘𑀺𑀦𑁆𑀥𑀭𑀸;_  
_𑀇𑀦𑁆𑀤𑀕𑁄𑀧𑀓𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(1067.)

1422\. _𑀦𑀻𑀮𑀩𑁆𑀪𑀓𑀽𑀝𑀲𑀤𑀺𑀲𑀸,_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀯𑀭𑀽𑀧𑀫𑀸;_  
_𑀯𑀸𑀭𑀡𑀸𑀪𑀺𑀭𑀼𑀤𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(1068.)

1423\. _𑀅𑀪𑀺𑀯𑀼𑀝𑁆𑀞𑀸 𑀭𑀫𑁆𑀫𑀢𑀮𑀸,_  
_𑀦𑀕𑀸 𑀇𑀲𑀺𑀪𑀺 𑀲𑁂𑀯𑀺𑀢𑀸;_  
_𑀅𑀩𑁆𑀪𑀼𑀦𑁆𑀦𑀤𑀺𑀢𑀸 𑀲𑀺𑀔𑀻𑀳𑀺,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(1069.)

1424\. _𑀅𑀮𑀁 𑀛𑀸𑀬𑀺𑀢𑀼𑀓𑀸𑀫𑀲𑁆𑀲,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄;_  
_𑀅𑀮𑀁 𑀫𑁂 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀲𑁆𑀲,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


(1070.)

1425\. _𑀅𑀮𑀁 𑀫𑁂 𑀨𑀸𑀲𑀼𑀓𑀸𑀫𑀲𑁆𑀲,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀅𑀮𑀁 𑀫𑁂 𑀬𑁄𑀕𑀓𑀸𑀫𑀲𑁆𑀲,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(1071.)

1426\. _𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑁂𑀦 𑀲𑀫𑀸𑀦𑀸,_  
_𑀕𑀕𑀦𑀸𑀯𑀩𑁆𑀪𑀙𑀸𑀤𑀺𑀢𑀸;_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀚𑀕𑀡𑀸𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(1072.)

1427\. _𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑀸 𑀕𑀳𑀝𑁆𑀞𑁂𑀳𑀺,_  
_𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑀸;_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀚𑀕𑀡𑀸𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(1073.)

1428\. _𑀅𑀘𑁆𑀙𑁄𑀤𑀺𑀓𑀸 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀸,_  
_𑀕𑁄𑀦𑀗𑁆𑀕𑀼𑀮𑀫𑀺𑀕𑀸𑀬𑀼𑀢𑀸;_  
_𑀅𑀫𑁆𑀩𑀼𑀲𑁂𑀯𑀸𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(1074.)

1429\. _𑀦 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀢𑀼𑀭𑀺𑀬𑁂𑀦,_  
_𑀭𑀢𑀺 𑀫𑁂 𑀳𑁄𑀢𑀺 𑀢𑀸𑀤𑀺𑀲𑀻;_  
_𑀬𑀣𑀸 𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑁄𑁇_  


(1075.)

1430\. _𑀓𑀫𑁆𑀫𑀁 𑀩𑀳𑀼𑀓𑀁 𑀦 𑀓𑀸𑀭𑀬𑁂,_  
_𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀚𑀦𑀁 𑀦 𑀉𑀬𑁆𑀬𑀫𑁂;_  
_𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀲𑁄 𑀭𑀲𑀸𑀦𑀼𑀕𑀺𑀤𑁆𑀥𑁄,_  
_𑀅𑀢𑁆𑀣𑀁 𑀭𑀺𑀜𑁆𑀘𑀢𑀺 𑀬𑁄 𑀲𑀼𑀔𑀸𑀯𑀳𑁄𑁇_  


(1076.)

1431\. _𑀓𑀫𑁆𑀫𑀁 𑀩𑀳𑀼𑀓𑀁 𑀦 𑀓𑀸𑀭𑀬𑁂,_  
_𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀦𑀢𑁆𑀢𑀦𑁂𑀬𑁆𑀬𑀫𑁂𑀢𑀁;_  
_𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀢𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀲𑁄 𑀲𑀫𑀣𑀁 𑀦 𑀯𑀺𑀦𑁆𑀤𑀢𑀺𑁇_  


(1077.)

1432\. _𑀑𑀝𑁆𑀞𑀧𑁆𑀧𑀳𑀢𑀫𑀢𑁆𑀢𑁂𑀦,_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀕𑀻𑀯𑁄 𑀘𑀭𑀢𑀺,_  
_𑀅𑀳𑀁 𑀲𑁂𑀬𑁆𑀬𑁄𑀢𑀺 𑀫𑀜𑁆𑀜𑀢𑀺𑁇_  


(1078.)

1433\. _𑀅𑀲𑁂𑀬𑁆𑀬𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀫𑀸𑀦𑀁,_  
_𑀩𑀸𑀮𑁄 𑀫𑀜𑁆𑀜𑀢𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁;_  
_𑀦 𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺,_  
_𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀫𑀸𑀦𑀲𑀁 𑀦𑀭𑀁𑁇_  


(1079.)

1434\. _𑀬𑁄 𑀘 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺,_  
_𑀦𑀸𑀳𑀁 𑀲𑁂𑀬𑁆𑀬𑁄𑀢𑀺 𑀯𑀸 𑀧𑀦;_  
_𑀳𑀻𑀦𑁄 𑀢𑀁𑀲𑀤𑀺𑀲𑁄 𑀯𑀸𑀢𑀺,_  
_𑀯𑀺𑀥𑀸𑀲𑀼 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑀢𑀺𑁇_  


(1080.)

1435\. _𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑀁 𑀢𑀣𑀸 𑀢𑀸𑀤𑀺𑀁,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀘𑁂𑀢𑁄𑀲𑀫𑀣𑀫𑀦𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀜𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀲𑀁𑀲𑀭𑁂𑁇_  


(1081.)

1436\. _𑀬𑀲𑁆𑀲 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼,_  
_𑀕𑀸𑀭𑀯𑁄 𑀦𑀽𑀧𑀮𑀩𑁆𑀪𑀢𑀺;_  
_𑀆𑀭𑀓𑀸 𑀳𑁄𑀢𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸,_  
_𑀦𑀪𑀢𑁄 𑀧𑀼𑀣𑀯𑀻 𑀬𑀣𑀸𑁇_  


(1082.)

1437\. _𑀬𑁂𑀲𑀜𑁆𑀘 𑀳𑀺𑀭𑀺𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁,_  
_𑀲𑀤𑀸 𑀲𑀫𑁆𑀫𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀯𑀺𑀭𑀽𑀍𑀳𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀢𑁂,_  
_𑀢𑁂𑀲𑀁 𑀔𑀻𑀡𑀸 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑁇_  


(1083.)

1438\. _𑀉𑀤𑁆𑀥𑀢𑁄 𑀘𑀧𑀮𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑁂𑀦 𑀧𑀸𑀭𑀼𑀢𑁄;_  
_𑀓𑀧𑀻𑀯 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑁂𑀦,_  
_𑀦 𑀲𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑁄𑀪𑀢𑀺𑁇_  


(1084.)

1439\. _𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄 𑀅𑀘𑀧𑀮𑁄,_  
_𑀦𑀺𑀧𑀓𑁄 𑀲𑀁𑀯𑀼𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀲𑁄𑀪𑀢𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑁂𑀦,_  
_𑀲𑀻𑀳𑁄𑀯 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑁂𑁇_  


(1085.)

1440\. _𑀏𑀢𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀤𑁂𑀯𑀸,_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀤𑀲𑀤𑁂𑀯𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸𑁇_  


(1086.)

1441\. _𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀁 𑀯𑀻𑀭𑀁,_  
_𑀫𑀳𑀸𑀛𑀸𑀬𑀺𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀚𑀮𑀻𑀓𑀢𑀸𑁇_  


(1087.)

1442\. _‘𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀬𑀲𑁆𑀲 𑀢𑁂 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀬𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺𑁇_  


(1088.)

1443\. _𑀅𑀘𑁆𑀙𑁂𑀭𑀁 𑀯𑀢 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀕𑁄𑀘𑀭𑁄 𑀲𑀓𑁄;_  
_𑀬𑁂 𑀫𑀬𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀯𑀸𑀮𑀯𑁂𑀥𑀺𑀲𑀫𑀸𑀕𑀢𑀸’𑁇_  


(1089.)

1444\. _𑀢𑀁 𑀢𑀣𑀸 𑀤𑁂𑀯𑀓𑀸𑀬𑁂𑀳𑀺,_  
_𑀧𑀽𑀚𑀺𑀢𑀁 𑀧𑀽𑀚𑀦𑀸𑀭𑀳𑀁;_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀢𑀤𑀸 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑁆𑀲 𑀲𑀺𑀢𑀁 𑀅𑀳𑀼𑁇_  


(1090.)

1445\. _𑀬𑀸𑀯𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀞𑀧𑀬𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁;_  
_𑀥𑀼𑀢𑀕𑀼𑀡𑁂 𑀯𑀺𑀲𑀺𑀝𑁆𑀞𑁄𑀳𑀁,_  
_𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(1091.)

1446\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(1092.)

1447\. _𑀦 𑀘𑀻𑀯𑀭𑁂 𑀦 𑀲𑀬𑀦𑁂,_  
_𑀪𑁄𑀚𑀦𑁂 𑀦𑀼𑀧𑀮𑀺𑀫𑁆𑀧𑀢𑀺;_  
_𑀕𑁄𑀢𑀫𑁄 𑀅𑀦𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄,_  
_𑀫𑀼𑀍𑀆𑀮𑀧𑀼𑀧𑁆𑀨𑀁 𑀯𑀺𑀫𑀮𑀁𑀯;_  
_𑀅𑀫𑁆𑀩𑀼𑀦𑀸 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀦𑀺𑀦𑁆𑀦𑁄,_  
_𑀢𑀺𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁄𑁇_  


(1093.)

1448\. _𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀕𑀻𑀯𑁄 𑀲𑁄,_  
_𑀲𑀤𑁆𑀥𑀸𑀳𑀢𑁆𑀣𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀧𑀜𑁆𑀜𑀸𑀲𑀻𑀲𑁄 𑀫𑀳𑀸𑀜𑀸𑀡𑀻,_  
_𑀲𑀤𑀸 𑀘𑀭𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇_  


---

1449\. …  𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄… 𑁇



---

1450\. 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

1451\. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1452\. _𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀦𑀺𑀧𑀸𑀢𑀫𑁆𑀳𑀺,_  
_𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑀯𑁆𑀳𑀬𑁄;_  
_𑀏𑀓𑁄𑀯 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲 𑀤𑀼𑀯𑁂𑀧𑀺 𑀘𑀸𑀢𑀺𑁇_
