# 𑁧𑁇𑁫𑁇𑁧 𑀲𑀺𑀭𑀺𑀯𑀟𑁆𑀠𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀺𑀭𑀺𑀯𑀟𑁆𑀠𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(41.)

94\. _“𑀯𑀺𑀯𑀭𑀫𑀦𑀼𑀧𑀢𑀦𑁆𑀢𑀺 𑀯𑀺𑀚𑁆𑀚𑀼𑀢𑀸,_  
_𑀯𑁂𑀪𑀸𑀭𑀲𑁆𑀲 𑀘 𑀧𑀡𑁆𑀟𑀯𑀲𑁆𑀲 𑀘;_  
_𑀦𑀕𑀯𑀺𑀯𑀭𑀕𑀢𑁄 𑀘 𑀛𑀸𑀬𑀢𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀫𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄”𑀢𑀺𑁇_  


---

95\. …  𑀲𑀺𑀭𑀺𑀯𑀟𑁆𑀠𑁄 𑀣𑁂𑀭𑁄… 𑁇
