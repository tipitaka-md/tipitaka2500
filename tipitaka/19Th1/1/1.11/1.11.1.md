# 𑁧𑁇𑁧𑁧𑁇𑁧 𑀩𑁂𑀮𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀏𑀓𑀸𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀩𑁂𑀮𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(101.)

227. _“𑀳𑀺𑀢𑁆𑀯𑀸 𑀕𑀺𑀳𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀯𑁄𑀲𑀺𑀢𑀢𑁆𑀢𑁄,_  
_𑀫𑀼𑀔𑀦𑀗𑁆𑀕𑀮𑀻 𑀑𑀤𑀭𑀺𑀓𑁄 𑀓𑀼𑀲𑀻𑀢𑁄;_  
_𑀫𑀳𑀸𑀯𑀭𑀸𑀳𑁄𑀯 𑀦𑀺𑀯𑀸𑀧𑀧𑀼𑀝𑁆𑀞𑁄,_  
_𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀕𑀩𑁆𑀪𑀫𑀼𑀧𑁂𑀢𑀺 𑀫𑀦𑁆𑀤𑁄”𑀢𑀺𑁇_  


---

228. …  𑀩𑁂𑀮𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑁄 𑀣𑁂𑀭𑁄… 𑁇
