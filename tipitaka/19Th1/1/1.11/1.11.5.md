# 𑁧𑁇𑁧𑁧𑁇𑁫 𑀫𑀮𑀺𑀢𑀯𑀫𑁆𑀪𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀏𑀓𑀸𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀮𑀺𑀢𑀯𑀫𑁆𑀪𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(105.)

235\. _“𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄𑀧𑀺 𑀦 𑀯𑀲𑁂,_  
_𑀭𑀫𑀫𑀸𑀦𑁄𑀧𑀺 𑀧𑀓𑁆𑀓𑀫𑁂;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯𑀸𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁,_  
_𑀯𑀲𑁂 𑀯𑀸𑀲𑀁 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄”𑀢𑀺𑁇_  


---

236\. …  𑀫𑀮𑀺𑀢𑀯𑀫𑁆𑀪𑁄 𑀣𑁂𑀭𑁄… 𑁇
