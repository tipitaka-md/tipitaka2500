# 𑁧𑁇𑁧𑁧𑁇𑁯 𑀲𑀁𑀖𑀭𑀓𑁆𑀔𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀏𑀓𑀸𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀁𑀖𑀭𑀓𑁆𑀔𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(109.)

243\. _“𑀦 𑀦𑀽𑀦𑀸𑀬𑀁 𑀧𑀭𑀫𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄,_  
_𑀭𑀳𑁄𑀕𑀢𑁄 𑀅𑀦𑀼𑀯𑀺𑀕𑀡𑁂𑀢𑀺 𑀲𑀸𑀲𑀦𑀁;_  
_𑀢𑀣𑀸𑀳𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀸𑀓𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄,_  
_𑀫𑀺𑀕𑀻 𑀬𑀣𑀸 𑀢𑀭𑀼𑀡𑀚𑀸𑀢𑀺𑀓𑀸 𑀯𑀦𑁂”𑀢𑀺𑁇_  


---

244\. …  𑀲𑀁𑀖𑀭𑀓𑁆𑀔𑀺𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
