# 𑁧𑁇𑁩𑁇𑁩 𑀕𑁄𑀲𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀕𑁄𑀲𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(23.)

54\. _“𑀅𑀳𑀁 𑀔𑁄 𑀯𑁂𑀍𑀉𑀕𑀼𑀫𑁆𑀩𑀲𑁆𑀫𑀺𑀁,_  
_𑀪𑀼𑀢𑁆𑀯𑀸𑀦 𑀫𑀥𑀼𑀧𑀸𑀬𑀲𑀁;_  
_𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀲𑀫𑁆𑀫𑀲𑀦𑁆𑀢𑁄,_  
_𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁;_  
_𑀲𑀸𑀦𑀼𑀁 𑀧𑀝𑀺𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

55\. …  𑀕𑁄𑀲𑀸𑀮𑁄 𑀣𑁂𑀭𑁄… 𑁇
