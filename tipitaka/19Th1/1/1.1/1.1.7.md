# 𑁧𑁇𑁧𑁇𑁭 𑀪𑀮𑁆𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀪𑀮𑁆𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(7.)

17\. _“𑀬𑁄𑀧𑀸𑀦𑀼𑀤𑀻 𑀫𑀘𑁆𑀘𑀼𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑁂𑀦𑀁,_  
_𑀦𑀍𑀅𑀲𑁂𑀢𑀼𑀁𑀯 𑀲𑀼𑀤𑀼𑀩𑁆𑀩𑀮𑀁 𑀫𑀳𑁄𑀖𑁄;_  
_𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀅𑀧𑁂𑀢𑀪𑁂𑀭𑀯𑁄 𑀳𑀺,_  
_𑀤𑀦𑁆𑀢𑁄 𑀲𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀞𑀺𑀢𑀢𑁆𑀢𑁄”𑀢𑀺𑁇_  


18\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀮𑁆𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
