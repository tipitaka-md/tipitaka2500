# 𑁧𑁇𑁧𑁇𑁯 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(9.)

21\. _“𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀦 𑀤𑀼𑀭𑀸𑀕𑀢𑀁,_  
_𑀦𑀬𑀺𑀤𑀁 𑀤𑀼𑀫𑀦𑁆𑀢𑀺𑀢𑀁 𑀫𑀫;_  
_𑀲𑀁𑀯𑀺𑀪𑀢𑁆𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼,_  
_𑀬𑀁 𑀲𑁂𑀝𑁆𑀞𑀁 𑀢𑀤𑀼𑀧𑀸𑀕𑀫𑀺𑀦𑁆”𑀢𑀺𑁇_  


22\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
