# 𑁧𑁇𑁧𑁇𑁧𑁦 𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(10.)

23\. _“𑀯𑀺𑀳𑀭𑀺 𑀅𑀧𑁂𑀓𑁆𑀔𑀁 𑀇𑀥 𑀯𑀸 𑀳𑀼𑀭𑀁 𑀯𑀸,_  
_𑀬𑁄 𑀯𑁂𑀤𑀕𑀽 𑀲𑀫𑀺𑀢𑁄 𑀬𑀢𑀢𑁆𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀦𑀽𑀧𑀮𑀺𑀢𑁆𑀢𑁄,_  
_𑀮𑁄𑀓𑀲𑁆𑀲 𑀚𑀜𑁆𑀜𑀸 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀜𑁆𑀘𑀸”𑀢𑀺𑁇_  


24\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

25\. 𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

26\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



27\. _𑀲𑀼𑀪𑀽𑀢𑀺 𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀣𑁂𑀭𑁄,_  
_𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑀲𑀫𑁆𑀫𑀢𑁄;_  
_𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀤𑀩𑁆𑀩𑁄 𑀘,_  
_𑀲𑀻𑀢𑀯𑀦𑀺𑀬𑁄 𑀘 𑀪𑀮𑁆𑀮𑀺𑀬𑁄;_  
_𑀯𑀻𑀭𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀘,_  
_𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑁄 𑀢𑀫𑁄𑀦𑀼𑀤𑁄𑀢𑀺𑁇_
