# 𑁧𑁇𑁧𑁇𑁬 𑀲𑀻𑀢𑀯𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀻𑀢𑀯𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(6.)

15\. _“𑀬𑁄 𑀲𑀻𑀢𑀯𑀦𑀁 𑀉𑀧𑀕𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀏𑀓𑁄 𑀲𑀦𑁆𑀢𑀼𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑀢𑁆𑀢𑁄;_  
_𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀅𑀧𑁂𑀢𑀮𑁄𑀫𑀳𑀁𑀲𑁄,_  
_𑀭𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀁 𑀥𑀺𑀢𑀺𑀫𑀸”𑀢𑀺𑁇_  


16\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀻𑀢𑀯𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
