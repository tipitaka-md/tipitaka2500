# 𑁧𑁇𑁧𑁇𑁩 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(3.)

9\. _“𑀧𑀜𑁆𑀜𑀁 𑀇𑀫𑀁 𑀧𑀲𑁆𑀲 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁,_  
_𑀅𑀕𑁆𑀕𑀺 𑀬𑀣𑀸 𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀦𑀺𑀲𑀻𑀣𑁂;_  
_𑀆𑀮𑁄𑀓𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀤𑀤𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀆𑀕𑀢𑀸𑀦𑀁 𑀯𑀺𑀦𑀬𑀦𑁆𑀢𑀺 𑀓𑀗𑁆𑀔𑀦𑁆”𑀢𑀺𑁇_  


10\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
