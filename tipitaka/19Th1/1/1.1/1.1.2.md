# 𑁧𑁇𑁧𑁇𑁨 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(2.)

7\. _“𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀉𑀧𑀭𑀢𑁄,_  
_𑀫𑀦𑁆𑀢𑀪𑀸𑀡𑀻 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄;_  
_𑀥𑀼𑀦𑀸𑀢𑀺 𑀧𑀸𑀧𑀓𑁂 𑀥𑀫𑁆𑀫𑁂,_  
_𑀤𑀼𑀫𑀧𑀢𑁆𑀢𑀁𑀯 𑀫𑀸𑀮𑀼𑀢𑁄”𑀢𑀺𑁇_  


8\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
