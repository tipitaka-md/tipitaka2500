# 𑁧𑁇𑁪𑁇𑁪 𑀧𑁄𑀲𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀧𑁄𑀲𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(34.)

78\. _“𑀅𑀦𑀸𑀲𑀦𑁆𑀦𑀯𑀭𑀸 𑀏𑀢𑀸,_  
_𑀦𑀺𑀘𑁆𑀘𑀫𑁂𑀯 𑀯𑀺𑀚𑀸𑀦𑀢𑀸;_  
_𑀕𑀸𑀫𑀸 𑀅𑀭𑀜𑁆𑀜𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀢𑀢𑁄 𑀕𑁂𑀳𑀁 𑀉𑀧𑀸𑀯𑀺𑀲𑀺;_  
_𑀢𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬 𑀧𑀓𑁆𑀓𑀸𑀫𑀺,_  
_𑀅𑀦𑀸𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀧𑁄𑀲𑀺𑀬𑁄”𑀢𑀺𑁇_  


---

79\. …  𑀧𑁄𑀲𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
