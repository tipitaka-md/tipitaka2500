# 𑁧𑁇𑁪𑁇𑁫 𑀲𑀸𑀫𑀜𑁆𑀜𑀓𑀸𑀦𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀫𑀜𑁆𑀜𑀓𑀸𑀦𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(35.)

80. _“𑀲𑀼𑀔𑀁 𑀲𑀼𑀔𑀢𑁆𑀣𑁄 𑀮𑀪𑀢𑁂 𑀢𑀤𑀸𑀘𑀭𑀁,_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀧𑀧𑁆𑀧𑁄𑀢𑀺 𑀬𑀲𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀢𑀺;_  
_𑀬𑁄 𑀅𑀭𑀺𑀬𑀫𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀫𑀜𑁆𑀚𑀲𑀁 𑀉𑀚𑀼𑀁,_  
_𑀪𑀸𑀯𑁂𑀢𑀺 𑀫𑀕𑁆𑀕𑀁 𑀅𑀫𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇_  


---

81. …  𑀲𑀸𑀫𑀜𑁆𑀜𑀓𑀸𑀦𑀺𑀢𑁆𑀣𑁂𑀭𑁄… 𑁇
