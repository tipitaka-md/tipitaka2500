# 𑁧𑁇𑁪𑁇𑁯 𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(39.)

88\. _“𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀬 𑀑𑀫𑀝𑁆𑀞𑁄,_

_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂”𑀢𑀺𑁇_
