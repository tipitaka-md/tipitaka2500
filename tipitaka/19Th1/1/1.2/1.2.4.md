# 𑁧𑁇𑁨𑁇𑁪 𑀲𑀺𑀯𑀓𑀲𑀸𑀫𑀡𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀺𑀯𑀓𑀲𑀸𑀫𑀡𑁂𑀭𑀕𑀸𑀣𑀸

(14.)

34\. _“𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀫𑀁 𑀅𑀯𑀘,_  
_𑀇𑀢𑁄 𑀕𑀘𑁆𑀙𑀸𑀫 𑀲𑀻𑀯𑀓;_  
_𑀕𑀸𑀫𑁂 𑀫𑁂 𑀯𑀲𑀢𑀺 𑀓𑀸𑀬𑁄,_  
_𑀅𑀭𑀜𑁆𑀜𑀁 𑀫𑁂 𑀕𑀢𑁄 𑀫𑀦𑁄;_  
_𑀲𑁂𑀫𑀸𑀦𑀓𑁄𑀧𑀺 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀦𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀕𑁄 𑀯𑀺𑀚𑀸𑀦𑀢𑀦𑁆”𑀢𑀺𑁇_  


---

35\. …  𑀯𑀦𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀣𑁂𑀭𑀲𑁆𑀲 (𑀲𑀺𑀯𑀺𑀓𑁄) 𑀲𑀸𑀫𑀡𑁂𑀭𑁄… 𑁇
