# 𑁧𑁇𑁨𑁇𑁬 𑀩𑁂𑀮𑀝𑁆𑀞𑀲𑀻𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀩𑁂𑀮𑀝𑁆𑀞𑀲𑀻𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(16.)

38\. _“𑀬𑀣𑀸𑀧𑀺 𑀪𑀤𑁆𑀤𑁄 𑀆𑀚𑀜𑁆𑀜𑁄,_  
_𑀦𑀗𑁆𑀕𑀮𑀸𑀯𑀢𑁆𑀢𑀦𑀻 𑀲𑀺𑀔𑀻;_  
_𑀕𑀘𑁆𑀙𑀢𑀺 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦,_  
_𑀏𑀯𑀁 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸 𑀫𑀫;_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦,_  
_𑀲𑀼𑀔𑁂 𑀮𑀤𑁆𑀥𑁂 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁂”𑀢𑀺𑁇_  


---

39\. …  𑀩𑁂𑀮𑀝𑁆𑀞𑀲𑀻𑀲𑁄 𑀣𑁂𑀭𑁄… 𑁇
