# 𑁧𑁇𑁨𑁇𑁮 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀧𑀺𑀢𑀼𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀧𑀺𑀢𑀼𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(18.)

42\. _“𑀅𑀳𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂;_  
_𑀓𑁂𑀯𑀮𑀁 𑀅𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀨𑀭𑀻 𑀧𑀣𑀯𑀺𑀁 𑀇𑀫𑀁;_  
_𑀫𑀜𑁆𑀜𑁂𑀳𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀁 𑀲𑁄,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀳𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

43\. …  𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀧𑀺𑀢𑀸 𑀣𑁂𑀭𑁄… 𑁇
