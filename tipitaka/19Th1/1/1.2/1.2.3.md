# 𑁧𑁇𑁨𑁇𑁩 𑀯𑀦𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀯𑀦𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(13.)

32\. _“𑀦𑀻𑀮𑀩𑁆𑀪𑀯𑀡𑁆𑀡𑀸 𑀭𑀼𑀘𑀺𑀭𑀸,_  
_𑀲𑀻𑀢𑀯𑀸𑀭𑀻 𑀲𑀼𑀘𑀺𑀦𑁆𑀥𑀭𑀸;_  
_𑀇𑀦𑁆𑀤𑀕𑁄𑀧𑀓𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀢𑁂 𑀲𑁂𑀮𑀸 𑀭𑀫𑀬𑀦𑁆𑀢𑀺 𑀫𑀦𑁆”𑀢𑀺𑁇_  


---

33\. …  𑀯𑀦𑀯𑀘𑁆𑀙𑁄 𑀣𑁂𑀭𑁄… 𑁇
