# 𑁧𑁇𑁧𑁨𑁇𑁫 𑀫𑀳𑀸𑀦𑀸𑀫𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀦𑀸𑀫𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(115.)

257. _“𑀏𑀲𑀸𑀯𑀳𑀺𑀬𑁆𑀬𑀲𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑀦,_  
_𑀩𑀳𑀼𑀓𑀼𑀝𑀚𑀲𑀮𑁆𑀮𑀓𑀺𑀓𑁂𑀦;_  
_𑀦𑁂𑀲𑀸𑀤𑀓𑁂𑀦 𑀕𑀺𑀭𑀺𑀦𑀸,_  
_𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀸 𑀧𑀭𑀺𑀘𑁆𑀙𑀤𑁂𑀦𑀸”𑀢𑀺𑁇_  


---
