# 𑁧𑁇𑁧𑁨𑁇𑁧 𑀚𑁂𑀦𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀯𑀕𑁆𑀕

* 𑀚𑁂𑀦𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(111.)

249\. _“𑀤𑀼𑀧𑁆𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀯𑁂 𑀤𑀼𑀭𑀥𑀺𑀯𑀸𑀲𑀸 𑀕𑁂𑀳𑀸,_  
_𑀥𑀫𑁆𑀫𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀤𑀼𑀭𑀥𑀺𑀕𑀫𑀸 𑀪𑁄𑀕𑀸;_  
_𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀼𑀢𑁆𑀢𑀺 𑀦𑁄 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦𑁂𑀯,_  
_𑀬𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀢𑀼𑀁 𑀲𑀢𑀢𑀫𑀦𑀺𑀘𑁆𑀘𑀢𑀦𑁆”𑀢𑀺𑁇_  


---

250\. …  𑀚𑁂𑀦𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
