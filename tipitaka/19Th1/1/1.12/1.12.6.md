# 𑁧𑁇𑁧𑁨𑁇𑁬 𑀧𑀸𑀭𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(116.)

259\. _“𑀙𑀨𑀲𑁆𑀲𑀸𑀬𑀢𑀦𑁂 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁄 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑁄;_  
_𑀅𑀖𑀫𑀽𑀮𑀁 𑀯𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀢𑁄 𑀫𑁂 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑁄”𑀢𑀺𑁇_  


---

260\. …  𑀧𑀸𑀭𑀸𑀧𑀭𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
