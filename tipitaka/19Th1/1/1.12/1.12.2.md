# 𑁧𑁇𑁧𑁨𑁇𑁨 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

(112.)

251. _“𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑁄𑀳𑀁 𑀫𑀳𑀸𑀛𑀸𑀬𑀻,_  
_𑀘𑁂𑀢𑁄𑀲𑀫𑀣𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀲𑀤𑀢𑁆𑀣𑁄 𑀫𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_
