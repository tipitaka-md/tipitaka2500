# 𑁧𑁇𑁧𑁨𑁇𑁮 𑀓𑀺𑀫𑀺𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀺𑀫𑀺𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(118.)

263. _“𑀅𑀪𑀺𑀲𑀢𑁆𑀢𑁄𑀯 𑀦𑀺𑀧𑀢𑀢𑀺,_  
_𑀯𑀬𑁄 𑀭𑀽𑀧𑀁 𑀅𑀜𑁆𑀜𑀫𑀺𑀯 𑀢𑀣𑁂𑀯 𑀲𑀦𑁆𑀢𑀁;_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀢𑁄,_  
_𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑁂𑀯 𑀲𑀭𑀸𑀫𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀦𑁆”𑀢𑀺𑁇_
