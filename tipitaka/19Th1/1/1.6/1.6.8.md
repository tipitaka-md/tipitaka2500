# 𑁧𑁇𑁬𑁇𑁮 𑀭𑀫𑀡𑀻𑀬𑀓𑀼𑀝𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀙𑀝𑁆𑀞𑀯𑀕𑁆𑀕

* 𑀭𑀫𑀡𑀻𑀬𑀓𑀼𑀝𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(58.)

131\. _“𑀭𑀫𑀡𑀻𑀬𑀸 𑀫𑁂 𑀓𑀼𑀝𑀺𑀓𑀸,_  
_𑀲𑀤𑁆𑀥𑀸𑀤𑁂𑀬𑁆𑀬𑀸 𑀫𑀦𑁄𑀭𑀫𑀸;_  
_𑀦 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀓𑀼𑀫𑀸𑀭𑀻𑀳𑀺,_
