# 𑁧𑁇𑁬𑁇𑁩 𑀯𑀮𑁆𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀙𑀝𑁆𑀞𑀯𑀕𑁆𑀕

### 𑀯𑀮𑁆𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(53.)

121. _“𑀯𑀲𑁆𑀲𑀢𑀺 𑀤𑁂𑀯𑁄 𑀬𑀣𑀸 𑀲𑀼𑀕𑀻𑀢𑀁,_  
_𑀙𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀝𑀺𑀓𑀸 𑀲𑀼𑀔𑀸 𑀦𑀺𑀯𑀸𑀢𑀸;_  
_𑀢𑀲𑁆𑀲𑀁 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄,_  
_𑀅𑀣 𑀘𑁂 𑀧𑀢𑁆𑀣𑀬𑀲𑀺 𑀧𑀯𑀲𑁆𑀲 𑀤𑁂𑀯𑀸”𑀢𑀺𑁇_  


---

122. …  𑀯𑀮𑁆𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
