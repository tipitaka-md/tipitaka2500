# 𑁧𑁇𑁬𑁇𑁬 𑀓𑀼𑀝𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀙𑀝𑁆𑀞𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀝𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(56.)

127\. _“𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀼𑀝𑀺𑀓𑀸𑀬𑀁,_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀆𑀯𑀼𑀲𑁄,_  
_𑀅𑀫𑁄𑀖𑀸 𑀢𑁂 𑀓𑀼𑀝𑀺𑀓𑀸 𑀓𑀢𑀸”𑀢𑀺𑁇_  


---

128\. …  𑀓𑀼𑀝𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀢𑁆𑀣𑁂𑀭𑁄… 𑁇
