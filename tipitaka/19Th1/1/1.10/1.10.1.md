# 𑁧𑁇𑁧𑁦𑁇𑁧 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(91.)

205\. _“𑀦 𑀢𑀣𑀸 𑀫𑀢𑀁 𑀲𑀢𑀭𑀲𑀁,_  
_𑀲𑀼𑀥𑀦𑁆𑀦𑀁 𑀬𑀁 𑀫𑀬𑀚𑁆𑀚 𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀁;_  
_𑀅𑀧𑀭𑀺𑀫𑀺𑀢𑀤𑀲𑁆𑀲𑀺𑀦𑀸 𑀕𑁄𑀢𑀫𑁂𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇_  


---

206\. …  𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀓𑁄 𑀣𑁂𑀭𑁄… 𑁇
