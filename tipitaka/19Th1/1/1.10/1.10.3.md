# 𑁧𑁇𑁧𑁦𑁇𑁩 𑀏𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀏𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(93.)

209\. _“𑀤𑀼𑀓𑁆𑀔𑀸 𑀓𑀸𑀫𑀸 𑀏𑀭𑀓,_  
_𑀦 𑀲𑀼𑀔𑀸 𑀓𑀸𑀫𑀸 𑀏𑀭𑀓;_  
_𑀬𑁄 𑀓𑀸𑀫𑁂 𑀓𑀸𑀫𑀬𑀢𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑁄 𑀓𑀸𑀫𑀬𑀢𑀺 𑀏𑀭𑀓;_  
_𑀬𑁄 𑀓𑀸𑀫𑁂 𑀦 𑀓𑀸𑀫𑀬𑀢𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑁄 𑀦 𑀓𑀸𑀫𑀬𑀢𑀺 𑀏𑀭𑀓𑀸”𑀢𑀺𑁇_  


---

210\. …  𑀏𑀭𑀓𑁄 𑀣𑁂𑀭𑁄… 𑁇
