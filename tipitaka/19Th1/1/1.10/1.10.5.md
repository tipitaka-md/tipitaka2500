# 𑁧𑁇𑁧𑁦𑁇𑁫 𑀘𑀓𑁆𑀔𑀼𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀘𑀓𑁆𑀔𑀼𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(95.)

213\. _“𑀅𑀦𑁆𑀥𑁄𑀳𑀁 𑀳𑀢𑀦𑁂𑀢𑁆𑀢𑁄𑀲𑁆𑀫𑀺,_  
_𑀓𑀦𑁆𑀢𑀸𑀭𑀤𑁆𑀥𑀸𑀦𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑁄;_  
_𑀲𑀬𑀫𑀸𑀦𑁄𑀧𑀺 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀁,_  
_𑀦 𑀲𑀳𑀸𑀬𑁂𑀦 𑀧𑀸𑀧𑁂𑀦𑀸”𑀢𑀺𑁇_
