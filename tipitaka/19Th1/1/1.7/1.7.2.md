# 𑁧𑁇𑁭𑁇𑁨 𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀢𑁆𑀢𑀫𑀯𑀕𑁆𑀕

### 𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(62.)

141\. _“𑀏𑀓𑀓𑀸 𑀫𑀬𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀸𑀫,_  
_𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀁𑀯 𑀯𑀦𑀲𑁆𑀫𑀺𑀁 𑀤𑀸𑀭𑀼𑀓𑀁;_  
_𑀢𑀲𑁆𑀲 𑀫𑁂 𑀩𑀳𑀼𑀓𑀸 𑀧𑀺𑀳𑀬𑀦𑁆𑀢𑀺,_  
_𑀦𑁂𑀭𑀬𑀺𑀓𑀸 𑀯𑀺𑀬 𑀲𑀕𑁆𑀕𑀕𑀸𑀫𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_
