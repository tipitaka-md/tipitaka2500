# 𑁧𑁇𑁭𑁇𑁬 𑀫𑁂𑀖𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀢𑁆𑀢𑀫𑀯𑀕𑁆𑀕

* 𑀫𑁂𑀖𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(66.)

149\. _“𑀅𑀦𑀼𑀲𑀸𑀲𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀲𑀢𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

150\. …  𑀫𑁂𑀖𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
