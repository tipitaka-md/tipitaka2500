# 𑁧𑁇𑁭𑁇𑁮 𑀏𑀓𑀼𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀢𑁆𑀢𑀫𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀼𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(68.)

153\. _“𑀅𑀥𑀺𑀘𑁂𑀢𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑁄,_  
_𑀫𑀼𑀦𑀺𑀦𑁄 𑀫𑁄𑀦𑀧𑀣𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀢𑁄;_  
_𑀲𑁄𑀓𑀸 𑀦 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑀸𑀤𑀺𑀦𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑀸 𑀲𑀢𑀻𑀫𑀢𑁄”𑀢𑀺𑁇_  


---

154\. …  𑀏𑀓𑀼𑀤𑀸𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
