# 𑁧𑁇𑁯𑁇𑁬 𑀦𑀸𑀕𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀯𑀫𑀯𑀕𑁆𑀕

### 𑀦𑀸𑀕𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(86.)

193. _“𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀼𑀣𑀼 𑀅𑀜𑁆𑀜𑀯𑀸𑀤𑀺𑀦𑀁,_  
_𑀫𑀕𑁆𑀕𑁄 𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀫𑁄 𑀬𑀣𑀸 𑀅𑀬𑀁;_  
_𑀇𑀢𑀺𑀲𑁆𑀲𑀼 𑀲𑀁𑀖𑀁 𑀪𑀕𑀯𑀸𑀦𑀼𑀲𑀸𑀲𑀢𑀺,_  
_𑀲𑀢𑁆𑀣𑀸 𑀲𑀬𑀁 𑀧𑀸𑀡𑀺𑀢𑀮𑁂𑀯 𑀤𑀲𑁆𑀲𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

194. …  𑀦𑀸𑀕𑀺𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
