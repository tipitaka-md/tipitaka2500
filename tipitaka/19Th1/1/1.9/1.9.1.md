# 𑁧𑁇𑁯𑁇𑁧 𑀲𑀫𑀺𑀢𑀺𑀕𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀯𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀫𑀺𑀢𑀺𑀕𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(81.)

183\. _“𑀬𑀁 𑀫𑀬𑀸 𑀧𑀓𑀢𑀁 𑀧𑀸𑀧𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀜𑁆𑀜𑀸𑀲𑀼 𑀚𑀸𑀢𑀺𑀲𑀼;_  
_𑀇𑀥𑁂𑀯 𑀢𑀁 𑀯𑁂𑀤𑀦𑀻𑀬𑀁,_  
_𑀯𑀢𑁆𑀣𑀼 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


---

184\. …  𑀲𑀫𑀺𑀢𑀺𑀕𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
