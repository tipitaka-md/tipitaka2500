# 𑁧𑁇𑁯𑁇𑁮 𑀅𑀚𑁆𑀚𑀼𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀯𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀚𑁆𑀚𑀼𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(88.)

197. _“𑀅𑀲𑀓𑁆𑀔𑀺𑀁 𑀯𑀢 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀉𑀤𑁆𑀥𑀸𑀢𑀼𑀁 𑀉𑀤𑀓𑀸 𑀣𑀮𑀁;_  
_𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀫𑀳𑁄𑀖𑁂𑀯,_  
_𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀳𑀦𑁆”𑀢𑀺𑁇_
