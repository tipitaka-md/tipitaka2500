# 𑁧𑁇𑁯𑁇𑁩 𑀲𑀻𑀳𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀯𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀳𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(83.)

187\. _“𑀲𑀻𑀳𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑁄;_  
_𑀪𑀸𑀯𑁂𑀳𑀺 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀚𑀳 𑀲𑀻𑀖𑀁 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀦𑁆”𑀢𑀺𑁇_
