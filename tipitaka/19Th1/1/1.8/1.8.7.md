# 𑁧𑁇𑁮𑁇𑁭 𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀝𑁆𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(77.)

173\. _“𑀇𑀤𑀁 𑀧𑀼𑀭𑁂 𑀘𑀺𑀢𑁆𑀢𑀫𑀘𑀸𑀭𑀺 𑀘𑀸𑀭𑀺𑀓𑀁,_  
_𑀬𑁂𑀦𑀺𑀘𑁆𑀙𑀓𑀁 𑀬𑀢𑁆𑀣𑀓𑀸𑀫𑀁 𑀬𑀣𑀸𑀲𑀼𑀔𑀁;_  
_𑀢𑀤𑀚𑁆𑀚𑀳𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁄𑀦𑀺𑀲𑁄,_  
_𑀳𑀢𑁆𑀣𑀺𑀧𑁆𑀧𑀪𑀺𑀦𑁆𑀦𑀁 𑀯𑀺𑀬 𑀅𑀗𑁆𑀓𑀼𑀲𑀕𑁆𑀕𑀳𑁄”𑀢𑀺𑁇_  


---

174\. …  𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀧𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
