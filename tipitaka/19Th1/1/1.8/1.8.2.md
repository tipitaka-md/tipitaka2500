# 𑁧𑁇𑁮𑁇𑁨 𑀆𑀢𑀼𑀫𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀝𑁆𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀆𑀢𑀼𑀫𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(72.)

163\. _“𑀬𑀣𑀸 𑀓𑀍𑀈𑀭𑁄 𑀲𑀼𑀲𑀼 𑀯𑀟𑁆𑀠𑀺𑀢𑀕𑁆𑀕𑁄,_  
_𑀤𑀼𑀦𑁆𑀦𑀺𑀓𑁆𑀔𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀲𑀸𑀔𑀚𑀸𑀢𑁄;_  
_𑀏𑀯𑀁 𑀅𑀳𑀁 𑀪𑀭𑀺𑀬𑀸𑀬𑀸𑀦𑀺𑀢𑀸𑀬,_  
_𑀅𑀦𑀼𑀫𑀜𑁆𑀜𑀁 𑀫𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀤𑀸𑀦𑀻”𑀢𑀺𑁇_  


---

164\. …  𑀆𑀢𑀼𑀫𑁄 𑀣𑁂𑀭𑁄… 𑁇
