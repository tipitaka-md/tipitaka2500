# 𑁧𑁇𑁮𑁇𑁮 𑀫𑁂𑀡𑁆𑀟𑀲𑀺𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀝𑁆𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑁂𑀡𑁆𑀟𑀲𑀺𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(78.)

175. _“𑀅𑀦𑁂𑀓𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀁 𑀅𑀦𑀺𑀩𑁆𑀩𑀺𑀲𑀁;_  
_𑀢𑀲𑁆𑀲 𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀚𑀸𑀢𑀲𑁆𑀲,_  
_𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀧𑀭𑀤𑁆𑀥𑁄”𑀢𑀺𑁇_
