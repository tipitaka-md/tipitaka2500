# 𑁨𑁧𑁇𑁧𑁇𑁧 𑀯𑀗𑁆𑀕𑀻𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀯𑀗𑁆𑀕𑀻𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(1218.)

1585\. _“𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀁 𑀯𑀢 𑀫𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀉𑀧𑀥𑀸𑀯𑀦𑁆𑀢𑀺,_  
_𑀧𑀕𑀩𑁆𑀪𑀸 𑀓𑀡𑁆𑀳𑀢𑁄 𑀇𑀫𑁂𑁇_  


(1219.)

1586\. _𑀉𑀕𑁆𑀕𑀧𑀼𑀢𑁆𑀢𑀸 𑀫𑀳𑀺𑀲𑁆𑀲𑀸𑀲𑀸,_  
_𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀤𑀍𑀳𑀥𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀺𑀭𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀲𑀳𑀲𑁆𑀲𑀁 𑀅𑀧𑀮𑀸𑀬𑀺𑀦𑀁𑁇_  


(1220.)

1587\. _𑀲𑀘𑁂𑀧𑀺 𑀏𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀬𑁆𑀬𑁄,_  
_𑀆𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀇𑀢𑁆𑀣𑀺𑀬𑁄;_  
_𑀦𑁂𑀯 𑀫𑀁 𑀩𑁆𑀬𑀸𑀥𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀥𑀫𑁆𑀫𑁂 𑀲𑀫𑁆𑀳𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇_  


(1221.)

1588\. _𑀲𑀓𑁆𑀔𑀻 𑀳𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀏𑀢𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑁄;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀫𑀦𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀢𑀢𑁆𑀣 𑀫𑁂 𑀦𑀺𑀭𑀢𑁄 𑀫𑀦𑁄𑁇_  


(1222.)

1589\. _𑀏𑀯𑀜𑁆𑀘𑁂 𑀫𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁,_  
_𑀧𑀸𑀧𑀺𑀫 𑀉𑀧𑀕𑀘𑁆𑀙𑀲𑀺;_  
_𑀢𑀣𑀸 𑀫𑀘𑁆𑀘𑀼 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀦 𑀫𑁂 𑀫𑀕𑁆𑀕𑀫𑁆𑀧𑀺 𑀤𑀓𑁆𑀔𑀲𑀺𑁇_  


(1223.)

1590\. _𑀅𑀭𑀢𑀺𑀜𑁆𑀘 𑀭𑀢𑀺𑀜𑁆𑀘 𑀧𑀳𑀸𑀬,_  
_𑀲𑀩𑁆𑀩𑀲𑁄 𑀕𑁂𑀳𑀲𑀺𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑀓𑁆𑀓𑀁;_  
_𑀯𑀦𑀣𑀁 𑀦 𑀓𑀭𑁂𑀬𑁆𑀬 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀦𑀣𑁄 𑀅𑀯𑀦𑀣𑁄 𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑁇_  


(1224.)

1591\. _𑀬𑀫𑀺𑀥 𑀧𑀣𑀯𑀺𑀜𑁆𑀘 𑀯𑁂𑀳𑀸𑀲𑀁,_  
_𑀭𑀽𑀧𑀕𑀢𑀁 𑀚𑀕𑀢𑁄𑀕𑀥𑀁 𑀓𑀺𑀜𑁆𑀘𑀺;_  
_𑀧𑀭𑀺𑀚𑀻𑀬𑀢𑀺 𑀲𑀩𑁆𑀩𑀫𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀏𑀯𑀁 𑀲𑀫𑁂𑀘𑁆𑀘 𑀘𑀭𑀦𑁆𑀢𑀺 𑀫𑀼𑀢𑀢𑁆𑀢𑀸𑁇_  


(1225.)

1592\. _𑀉𑀧𑀥𑀻𑀲𑀼 𑀚𑀦𑀸 𑀕𑀥𑀺𑀢𑀸𑀲𑁂,_  
_𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑁂 𑀧𑀝𑀺𑀖𑁂 𑀘 𑀫𑀼𑀢𑁂 𑀘;_  
_𑀏𑀢𑁆𑀣 𑀯𑀺𑀦𑁄𑀤𑀬 𑀙𑀦𑁆𑀤𑀫𑀦𑁂𑀚𑁄,_  
_𑀬𑁄 𑀳𑁂𑀢𑁆𑀣 𑀦 𑀮𑀺𑀫𑁆𑀧𑀢𑀺 𑀫𑀼𑀦𑀺 𑀢𑀫𑀸𑀳𑀼𑁇_  


(1226.)

1593\. _𑀅𑀣 𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀢𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸,_  
_𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀢𑀸𑀬 𑀅𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀸;_  
_𑀦 𑀘 𑀯𑀕𑁆𑀕𑀕𑀢𑀲𑁆𑀲 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺,_  
_𑀦𑁄 𑀧𑀦 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀕𑀸𑀳𑀻 𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑁇_  


(1227.)

1594\. _𑀤𑀩𑁆𑀩𑁄 𑀘𑀺𑀭𑀭𑀢𑁆𑀢𑀲𑀫𑀸𑀳𑀺𑀢𑁄,_  
_𑀅𑀓𑀼𑀳𑀓𑁄 𑀦𑀺𑀧𑀓𑁄 𑀅𑀧𑀺𑀳𑀸𑀮𑀼;_  
_𑀲𑀦𑁆𑀢𑀁 𑀧𑀤𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀸 𑀫𑀼𑀦𑀺,_  
_𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀓𑀗𑁆𑀔𑀢𑀺 𑀓𑀸𑀮𑀁𑁇_  


(1228.)

1595\. _𑀫𑀸𑀦𑀁 𑀧𑀚𑀳𑀲𑁆𑀲𑀼 𑀕𑁄𑀢𑀫,_  
_𑀫𑀸𑀦𑀧𑀣𑀜𑁆𑀘 𑀚𑀳𑀲𑁆𑀲𑀼 𑀅𑀲𑁂𑀲𑀁;_  
_𑀫𑀸𑀦𑀧𑀣𑀫𑁆𑀳𑀺 𑀲 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄,_  
_𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻𑀳𑀼𑀯𑀸 𑀘𑀺𑀭𑀭𑀢𑁆𑀢𑀁𑁇_  


(1229.)

1596\. _𑀫𑀓𑁆𑀔𑁂𑀦 𑀫𑀓𑁆𑀔𑀺𑀢𑀸 𑀧𑀚𑀸,_  
_𑀫𑀸𑀦𑀳𑀢𑀸 𑀦𑀺𑀭𑀬𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺;_  
_𑀲𑁄𑀘𑀦𑁆𑀢𑀺 𑀚𑀦𑀸 𑀘𑀺𑀭𑀭𑀢𑁆𑀢𑀁,_  
_𑀫𑀸𑀦𑀳𑀢𑀸 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇_  


(1230.)

1597\. _𑀦 𑀳𑀺 𑀲𑁄𑀘𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀤𑀸𑀘𑀺,_  
_𑀫𑀕𑁆𑀕𑀚𑀺𑀦𑁄 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄;_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀲𑀼𑀔𑀜𑁆𑀘𑀸𑀦𑀼𑀪𑁄𑀢𑀺,_  
_𑀥𑀫𑁆𑀫𑀤𑀲𑁄𑀢𑀺 𑀢𑀫𑀸𑀳𑀼 𑀢𑀣𑀢𑁆𑀢𑀁𑁇_  


(1231.)

1598\. _𑀢𑀲𑁆𑀫𑀸 𑀅𑀔𑀺𑀮𑁄 𑀇𑀥 𑀧𑀥𑀸𑀦𑀯𑀸,_  
_𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀺 𑀧𑀳𑀸𑀬 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁄;_  
_𑀫𑀸𑀦𑀜𑁆𑀘 𑀧𑀳𑀸𑀬 𑀅𑀲𑁂𑀲𑀁,_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑀦𑁆𑀢𑀓𑀭𑁄 𑀲𑀫𑀺𑀢𑀸𑀯𑀻”𑁇_  


(1232.)

1599\. _“𑀓𑀸𑀫𑀭𑀸𑀕𑁂𑀦 𑀟𑀬𑁆𑀳𑀸𑀫𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑁂 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀢𑀺;_  
_𑀲𑀸𑀥𑀼 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀕𑁄𑀢𑀫”𑁇_  


(1233.)

1600\. _“𑀲𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀧𑀭𑀺𑀬𑁂𑀲𑀸,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀢𑁂 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀢𑀺;_  
_𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀳𑀺,_  
_𑀲𑀼𑀪𑀁 𑀭𑀸𑀕𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁𑁇 ( )_  


(1234.)

1601\. _𑀅𑀲𑀼𑀪𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀳𑀺,_  
_𑀏𑀓𑀕𑁆𑀕𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀲𑀢𑀺 𑀓𑀸𑀬𑀕𑀢𑀸 𑀢𑁆𑀬𑀢𑁆𑀣𑀼,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀩𑀳𑀼𑀮𑁄 𑀪𑀯𑁇_  


(1235.)

1602\. _𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀸𑀯𑁂𑀳𑀺,_  
_𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀫𑀼𑀚𑁆𑀚𑀳;_  
_𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀸,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀘𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(1236.)

1603\. _“𑀢𑀫𑁂𑀯 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀬𑀸𑀬𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀢𑀸𑀧𑀬𑁂;_  
_𑀧𑀭𑁂 𑀘 𑀦 𑀯𑀺𑀳𑀺𑀁𑀲𑁂𑀬𑁆𑀬,_  
_𑀲𑀸 𑀯𑁂 𑀯𑀸𑀘𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸𑁇_  


(1237.)

1604\. _𑀧𑀺𑀬𑀯𑀸𑀘𑀫𑁂𑀯 𑀪𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀬𑀸 𑀯𑀸𑀘𑀸 𑀧𑀝𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸;_  
_𑀬𑀁 𑀅𑀦𑀸𑀤𑀸𑀬 𑀧𑀸𑀧𑀸𑀦𑀺,_  
_𑀧𑀭𑁂𑀲𑀁 𑀪𑀸𑀲𑀢𑁂 𑀧𑀺𑀬𑀁𑁇_  


(1238.)

1605\. _𑀲𑀘𑁆𑀘𑀁 𑀯𑁂 𑀅𑀫𑀢𑀸 𑀯𑀸𑀘𑀸,_  
_𑀏𑀲 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑀦𑁆𑀢𑀦𑁄;_  
_𑀲𑀘𑁆𑀘𑁂 𑀅𑀢𑁆𑀣𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘,_  
_𑀆𑀳𑀼 𑀲𑀦𑁆𑀢𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇_  


(1239.)

1606\. _𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀸𑀲𑀢𑀺 𑀯𑀸𑀘𑀁,_  
_𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬,_  
_𑀲𑀸 𑀯𑁂 𑀯𑀸𑀘𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑀸”𑁇_  


(1240.)

1607\. _“𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀜𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀫𑀕𑁆𑀕𑀸𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀁𑁇_  


(1241.)

1608\. _𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀤𑁂𑀲𑁂𑀢𑀺,_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦𑀧𑀺 𑀪𑀸𑀲𑀢𑀺;_  
_𑀲𑀸𑀮𑀺𑀓𑀸𑀬𑀺𑀯 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑁄,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑀁 𑀉𑀤𑀺𑀬𑁆𑀬𑀢𑀺𑁇_  


(1242.)

1609\. _𑀢𑀲𑁆𑀲 𑀢𑀁 𑀤𑁂𑀲𑀬𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀫𑀥𑀼𑀭𑀁 𑀕𑀺𑀭𑀁;_  
_𑀲𑀭𑁂𑀦 𑀭𑀚𑀦𑀻𑀬𑁂𑀦,_  
_𑀲𑀯𑀦𑀻𑀬𑁂𑀦 𑀯𑀕𑁆𑀕𑀼𑀦𑀸;_  
_𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸 𑀫𑀼𑀤𑀺𑀢𑀸,_  
_𑀲𑁄𑀢𑀁 𑀑𑀥𑁂𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑁇_  


(1243.)

1610\. _“𑀅𑀚𑁆𑀚 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸,_  
_𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀜𑁆𑀘𑀲𑀢𑀸 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀲𑀁𑀬𑁄𑀚𑀦𑀩𑀦𑁆𑀥𑀦𑀘𑁆𑀙𑀺𑀤𑀸,_  
_𑀅𑀦𑀻𑀖𑀸 𑀔𑀻𑀡𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸 𑀇𑀲𑀻𑁇_  


(1244.)

1611\. _𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀬𑀣𑀸 𑀭𑀸𑀚𑀸,_  
_𑀅𑀫𑀘𑁆𑀘𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑁂𑀢𑀺,_  
_𑀲𑀸𑀕𑀭𑀦𑁆𑀢𑀁 𑀫𑀳𑀺𑀁 𑀇𑀫𑀁𑁇_  


(1245.)

1612\. _𑀏𑀯𑀁 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫𑀁,_  
_𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀲𑀸𑀯𑀓𑀸 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸 𑀫𑀘𑁆𑀘𑀼𑀳𑀸𑀬𑀺𑀦𑁄𑁇_  


(1246.)

1613\. _𑀲𑀩𑁆𑀩𑁂 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀢𑁆𑀢𑀸,_  
_𑀧𑀮𑀸𑀧𑁂𑀢𑁆𑀣 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀢𑀡𑁆𑀳𑀸𑀲𑀮𑁆𑀮𑀲𑁆𑀲 𑀳𑀦𑁆𑀢𑀸𑀭𑀁,_  
_𑀯𑀦𑁆𑀤𑁂 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀁𑁇_  


(1247.)

1614\. _𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁,_  
_𑀲𑀼𑀕𑀢𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺;_  
_𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀁 𑀯𑀺𑀭𑀚𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑀁𑁇_  


(1248.)

1615\. _𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀫𑀮𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀲𑁄𑀪𑀢𑀺 𑀯𑀢 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(1249.)

1616\. _‘𑀦𑀸𑀕𑀦𑀸𑀫𑁄’𑀲𑀺 𑀪𑀕𑀯𑀸,_  
_𑀇𑀲𑀻𑀦𑀁 𑀇𑀲𑀺𑀲𑀢𑁆𑀢𑀫𑁄;_  
_𑀫𑀳𑀸𑀫𑁂𑀖𑁄𑀯 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀸𑀯𑀓𑁂 𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀲𑀺𑁇_  


(1250.)

1617\. _𑀤𑀺𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀲𑀢𑁆𑀣𑀼𑀤𑀲𑁆𑀲𑀦𑀓𑀫𑁆𑀬𑀢𑀸;_  
_𑀲𑀸𑀯𑀓𑁄 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀢𑀺 𑀯𑀗𑁆𑀕𑀺𑀲𑁄”𑁇_  


(1251.)

1618\. _“𑀉𑀫𑁆𑀫𑀕𑁆𑀕𑀧𑀣𑀁 𑀫𑀸𑀭𑀲𑁆𑀲,_  
_𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀘𑀭𑀢𑀺 𑀧𑀪𑀺𑀚𑁆𑀚 𑀔𑀻𑀮𑀸𑀦𑀺;_  
_𑀢𑀁 𑀧𑀲𑁆𑀲𑀣 𑀩𑀦𑁆𑀥𑀧𑀫𑀼𑀜𑁆𑀘𑀓𑀭𑀁,_  
_𑀅𑀲𑀺𑀢𑀁𑀯 𑀪𑀸𑀕𑀲𑁄 𑀧𑀯𑀺𑀪𑀚𑁆𑀚𑁇_  


(1252.)

1619\. _𑀑𑀖𑀲𑁆𑀲 𑀳𑀺 𑀦𑀺𑀢𑀭𑀡𑀢𑁆𑀣𑀁,_  
_𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀫𑀕𑁆𑀕𑀁 𑀅𑀓𑁆𑀔𑀸𑀲𑀺;_  
_𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀅𑀫𑀢𑁂 𑀅𑀓𑁆𑀔𑀸𑀢𑁂,_  
_𑀥𑀫𑁆𑀫𑀤𑀲𑀸 𑀞𑀺𑀢𑀸 𑀅𑀲𑀁𑀳𑀻𑀭𑀸𑁇_  


(1253.)

1620\. _𑀧𑀚𑁆𑀚𑁄𑀢𑀓𑀭𑁄 𑀅𑀢𑀺𑀯𑀺𑀚𑁆𑀛,_  
_𑀲𑀩𑁆𑀩𑀞𑀺𑀢𑀻𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀫𑀤𑁆𑀤𑀲;_  
_𑀜𑀢𑁆𑀯𑀸 𑀘 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀘,_  
_𑀅𑀕𑁆𑀕𑀁 𑀲𑁄 𑀤𑁂𑀲𑀬𑀺 𑀤𑀲𑀤𑁆𑀥𑀸𑀦𑀁𑁇_  


(1254.)

1621\. _𑀏𑀯𑀁 𑀲𑀼𑀤𑁂𑀲𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑁂,_  
_𑀓𑁄 𑀧𑀫𑀸𑀤𑁄 𑀯𑀺𑀚𑀸𑀦𑀢𑀁 𑀥𑀫𑁆𑀫𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀲𑀦𑁂,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀤𑀸 𑀦𑀫𑀲𑁆𑀲𑀫𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑁂”𑁇_  


(1255.)

1622\. _“𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑁄 𑀬𑁄 𑀣𑁂𑀭𑁄,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀢𑀺𑀩𑁆𑀩𑀦𑀺𑀓𑁆𑀓𑀫𑁄;_  
_𑀮𑀸𑀪𑀻 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁,_  
_𑀯𑀺𑀯𑁂𑀓𑀸𑀦𑀁 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄𑁇_  


(1256.)

1623\. _𑀬𑀁 𑀲𑀸𑀯𑀓𑁂𑀦 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀦𑀸;_  
_𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀢𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀺𑀓𑁆𑀔𑀢𑁄𑁇_  


(1257.)

1624\. _𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑁄,_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀩𑀼𑀤𑁆𑀥𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀢𑀺 𑀲𑀢𑁆𑀣𑀼𑀦𑁄”𑁇_  


(1258.)

1625\. _“𑀦𑀕𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑁂 𑀆𑀲𑀻𑀦𑀁,_  
_𑀫𑀼𑀦𑀺𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀸𑀭𑀕𑀼𑀁;_  
_𑀲𑀸𑀯𑀓𑀸 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸 𑀫𑀘𑁆𑀘𑀼𑀳𑀸𑀬𑀺𑀦𑁄𑁇_  


(1259.)

1626\. _𑀘𑁂𑀢𑀲𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑁂𑀢𑀺,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄;_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑁂𑀲𑀁 𑀲𑀫𑀦𑁆𑀯𑁂𑀲𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁𑁇_  


(1260.)

1627\. _𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀫𑀼𑀦𑀺𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀸𑀭𑀕𑀼𑀁;_  
_𑀅𑀦𑁂𑀓𑀸𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑁆𑀢𑀺 𑀕𑁄𑀢𑀫𑀁”𑁇_  


(1261.)

1628\. _“𑀘𑀦𑁆𑀤𑁄 𑀬𑀣𑀸 𑀯𑀺𑀕𑀢𑀯𑀮𑀸𑀳𑀓𑁂 𑀦𑀪𑁂,_  
_𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀻𑀢𑀫𑀮𑁄𑀯 𑀪𑀸𑀡𑀼𑀫𑀸;_  
_𑀏𑀯𑀫𑁆𑀧𑀺 𑀅𑀗𑁆𑀕𑀻𑀭𑀲 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺,_  
_𑀅𑀢𑀺𑀭𑁄𑀘𑀲𑀺 𑀬𑀲𑀲𑀸 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀁”𑁇_  


(1262.)

1629\. _“𑀓𑀸𑀯𑁂𑀬𑁆𑀬𑀫𑀢𑁆𑀢𑀸 𑀯𑀺𑀘𑀭𑀺𑀫𑁆𑀳 𑀧𑀼𑀩𑁆𑀩𑁂,_  
_𑀕𑀸𑀫𑀸 𑀕𑀸𑀫𑀁 𑀧𑀼𑀭𑀸 𑀧𑀼𑀭𑀁;_  
_𑀅𑀣𑀤𑁆𑀤𑀲𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀼𑀁𑁇_  


(1263.)

1630\. _𑀲𑁄 𑀫𑁂 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀫𑀼𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀸𑀭𑀕𑀽;_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀲𑀻𑀤𑀺𑀫𑁆𑀳,_  
_𑀲𑀤𑁆𑀥𑀸 𑀦𑁄 𑀉𑀤𑀧𑀚𑁆𑀚𑀣𑁇_  


(1264.)

1631\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀔𑀦𑁆𑀥𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀘;_  
_𑀥𑀸𑀢𑀼𑀬𑁄 𑀘 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(1265.)

1632\. _𑀩𑀳𑀽𑀦𑀁 𑀯𑀢 𑀅𑀢𑁆𑀣𑀸𑀬,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀸;_  
_𑀇𑀢𑁆𑀣𑀻𑀦𑀁 𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀜𑁆𑀘,_  
_𑀬𑁂 𑀢𑁂 𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀓𑀸𑁇_  


(1266.)

1633\. _𑀢𑁂𑀲𑀁 𑀔𑁄 𑀯𑀢 𑀅𑀢𑁆𑀣𑀸𑀬,_  
_𑀩𑁄𑀥𑀺𑀫𑀚𑁆𑀛𑀕𑀫𑀸 𑀫𑀼𑀦𑀺;_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘,_  
_𑀬𑁂 𑀦𑀺𑀭𑀸𑀫𑀕𑀢𑀤𑁆𑀤𑀲𑀸𑁇_  


(1267.)

1634\. _𑀲𑀼𑀤𑁂𑀲𑀺𑀢𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(1268.)

1635\. _𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀁;_  
_𑀅𑀭𑀺𑀬𑀁 𑀘𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀲𑀫𑀕𑀸𑀫𑀺𑀦𑀁𑁇_  


(1269.)

1636\. _𑀏𑀯𑀫𑁂𑀢𑁂 𑀢𑀣𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑁂 𑀢𑁂 𑀬𑀣𑀸 𑀢𑀣𑀸;_  
_𑀲𑀤𑀢𑁆𑀣𑁄 𑀫𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(1270.)

1637\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼,_  
_𑀬𑀁 𑀲𑁂𑀝𑁆𑀞𑀁 𑀢𑀤𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(1271.)

1638\. _𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀸;_  
_𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀓𑁄𑀯𑀺𑀤𑁄”𑁇_  


(1272.)

1639\. _“𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀢𑁆𑀣𑀸𑀭𑀫𑀦𑁄𑀫𑀧𑀜𑁆𑀜𑀁,_  
_𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀁 𑀙𑁂𑀢𑁆𑀢𑀸;_  
_𑀅𑀕𑁆𑀕𑀸𑀍𑀅𑀯𑁂 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀜𑀸𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀻 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀢𑁆𑀢𑁄𑁇_  


(1273.)

1640\. _𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀓𑀧𑁆𑀧𑁄 𑀇𑀢𑀺 𑀢𑀲𑁆𑀲 𑀦𑀸𑀫𑀁,_  
_𑀢𑀬𑀸 𑀓𑀢𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲;_  
_𑀲𑁄 𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀁 𑀅𑀘𑀭𑀺 𑀫𑀼𑀢𑁆𑀬𑀧𑁂𑀔𑁄,_  
_𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀤𑀍𑀳𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻𑁇_  


(1274.)

1641\. _𑀢𑀁 𑀲𑀸𑀯𑀓𑀁 𑀲𑀓𑁆𑀓 𑀫𑀬𑀫𑁆𑀧𑀺 𑀲𑀩𑁆𑀩𑁂,_  
_𑀅𑀜𑁆𑀜𑀸𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀫 𑀲𑀫𑀦𑁆𑀢𑀘𑀓𑁆𑀔𑀼;_  
_𑀲𑀫𑀯𑀝𑁆𑀞𑀺𑀢𑀸 𑀦𑁄 𑀲𑀯𑀦𑀸𑀬 𑀲𑁄𑀢𑀸,_  
_𑀢𑀼𑀯𑀁 𑀦𑁄 𑀲𑀢𑁆𑀣𑀸 𑀢𑁆𑀯𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑁄𑀲𑀺𑁇_  


(1275.)

1642\. _𑀙𑀺𑀦𑁆𑀤 𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁 𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑁂𑀢𑀁,_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀁 𑀯𑁂𑀤𑀬 𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜;_  
_𑀫𑀚𑁆𑀛𑁂𑀯 𑀦𑁄 𑀪𑀸𑀲 𑀲𑀫𑀦𑁆𑀢𑀘𑀓𑁆𑀔𑀼,_  
_𑀲𑀓𑁆𑀓𑁄𑀯 𑀤𑁂𑀯𑀸𑀦 𑀲𑀳𑀲𑁆𑀲𑀦𑁂𑀢𑁆𑀢𑁄𑁇_  


(1276.)

1643\. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀕𑀦𑁆𑀣𑀸 𑀇𑀥 𑀫𑁄𑀳𑀫𑀕𑁆𑀕𑀸,_  
_𑀅𑀜𑁆𑀜𑀸𑀡𑀧𑀓𑁆𑀔𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀞𑀸𑀦𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀢𑁆𑀯𑀸 𑀦 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀳𑀺 𑀏𑀢𑀁 𑀧𑀭𑀫𑀁 𑀦𑀭𑀸𑀦𑀁𑁇_  


(1277.)

1644\. _𑀦𑁄 𑀘𑁂 𑀳𑀺 𑀚𑀸𑀢𑀼 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀮𑁂𑀲𑁂,_  
_𑀯𑀸𑀢𑁄 𑀬𑀣𑀸 𑀅𑀩𑁆𑀪𑀖𑀦𑀁 𑀯𑀺𑀳𑀸𑀦𑁂;_  
_𑀢𑀫𑁄𑀯𑀲𑁆𑀲 𑀦𑀺𑀯𑀼𑀢𑁄 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁄,_  
_𑀚𑁄𑀢𑀺𑀫𑀦𑁆𑀢𑁄𑀧𑀺 𑀦 𑀧𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑀼𑀁𑁇_  


(1278.)

1645\. _𑀥𑀻𑀭𑀸 𑀘 𑀧𑀚𑁆𑀚𑁄𑀢𑀓𑀭𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀢𑀁 𑀅𑀳𑀁 𑀯𑀻𑀭 𑀢𑀣𑁂𑀯 𑀫𑀜𑁆𑀜𑁂;_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑀁 𑀚𑀸𑀦𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀫𑁆𑀳,_  
_𑀧𑀭𑀺𑀲𑀸𑀲𑀼 𑀦𑁄 𑀆𑀯𑀺𑀓𑀭𑁄𑀳𑀺 𑀓𑀧𑁆𑀧𑀁𑁇_  


(1279.)

1646\. _𑀔𑀺𑀧𑁆𑀧𑀁 𑀕𑀺𑀭𑀁 𑀏𑀭𑀬 𑀯𑀕𑁆𑀕𑀼 𑀯𑀕𑁆𑀕𑀼𑀁,_  
_𑀳𑀁𑀲𑁄𑀯 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀲𑀡𑀺𑀓𑀁 𑀦𑀺𑀓𑀽𑀚;_  
_𑀩𑀺𑀦𑁆𑀤𑀼𑀲𑁆𑀲𑀭𑁂𑀦 𑀲𑀼𑀯𑀺𑀓𑀧𑁆𑀧𑀺𑀢𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑁂 𑀉𑀚𑁆𑀚𑀼𑀕𑀢𑀸 𑀲𑀼𑀡𑁄𑀫𑁇_  


(1280.)

1647\. _𑀧𑀳𑀻𑀦𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀁 𑀅𑀲𑁂𑀲𑀁,_  
_𑀦𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀥𑁄𑀦𑀁 𑀯𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀁;_  
_𑀦 𑀓𑀸𑀫𑀓𑀸𑀭𑁄 𑀳𑀺 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸𑀦𑀁,_  
_𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀓𑀸𑀭𑁄 𑀘 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁𑁇_  


(1281.)

1648\. _𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀢𑀯𑁂𑀤𑀁,_  
_𑀲𑀫𑀼𑀚𑁆𑀚𑀼𑀧𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀼𑀕𑁆𑀕𑀳𑀻𑀢𑀁;_  
_𑀅𑀬𑀫𑀜𑁆𑀚𑀮𑀺 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀲𑀼𑀧𑁆𑀧𑀡𑀸𑀫𑀺𑀢𑁄,_  
_𑀫𑀸 𑀫𑁄𑀳𑀬𑀻 𑀚𑀸𑀦𑀫𑀦𑁄𑀫𑀧𑀜𑁆𑀜𑁇_  


(1282.)

1649\. _𑀧𑀭𑁄𑀧𑀭𑀁 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀸 𑀫𑁄𑀳𑀬𑀻 𑀚𑀸𑀦𑀫𑀦𑁄𑀫𑀯𑀻𑀭𑀺𑀬;_  
_𑀯𑀸𑀭𑀺𑀁 𑀬𑀣𑀸 𑀖𑀫𑁆𑀫𑀦𑀺 𑀖𑀫𑁆𑀫𑀢𑀢𑁆𑀢𑁄,_  
_𑀯𑀸𑀘𑀸𑀪𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺 𑀲𑀼𑀢𑀁 𑀧𑀯𑀲𑁆𑀲𑁇_  


(1283.)

1650\. _𑀬𑀤𑀢𑁆𑀣𑀺𑀓𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀘𑀭𑀻,_  
_𑀓𑀧𑁆𑀧𑀸𑀬𑀦𑁄 𑀓𑀘𑁆𑀘𑀺𑀲𑁆𑀲𑀢𑀁 𑀅𑀫𑁄𑀖𑀁;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺 𑀲𑁄 𑀆𑀤𑀼 𑀲𑁅𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑁄,_  
_𑀬𑀣𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑀼 𑀢𑀁 𑀲𑀼𑀡𑁄𑀫”𑁇_  


(1284.)

1651\. _“𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀢𑀡𑁆𑀳𑀁 𑀇𑀥 𑀦𑀸𑀫𑀭𑀽𑀧𑁂, __(𑀇𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀓𑀡𑁆𑀳𑀲𑁆𑀲 𑀲𑁄𑀢𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀸𑀦𑀼𑀲𑀬𑀺𑀢𑀁;_  
_𑀅𑀢𑀸𑀭𑀺 𑀚𑀸𑀢𑀺𑀁 𑀫𑀭𑀡𑀁 𑀅𑀲𑁂𑀲𑀁,_  
_𑀇𑀘𑁆𑀘𑀩𑁆𑀭𑀯𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑁂𑀝𑁆𑀞𑁄”𑁇_  


(1285.)

1652\. _“𑀏𑀲 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀲𑀻𑀤𑀸𑀫𑀺,_  
_𑀯𑀘𑁄 𑀢𑁂 𑀇𑀲𑀺𑀲𑀢𑁆𑀢𑀫;_  
_𑀅𑀫𑁄𑀖𑀁 𑀓𑀺𑀭 𑀫𑁂 𑀧𑀼𑀝𑁆𑀞𑀁,_  
_𑀦 𑀫𑀁 𑀯𑀜𑁆𑀘𑁂𑀲𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


(1286.)

1653\. _𑀬𑀣𑀸 𑀯𑀸𑀤𑀻 𑀢𑀣𑀸 𑀓𑀸𑀭𑀻,_  
_𑀅𑀳𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀚𑀸𑀮𑀁,_  
_𑀢𑀢𑀁 𑀫𑀸𑀬𑀸𑀯𑀺𑀦𑁄 𑀤𑀍𑀳𑀁𑁇_  


(1287.)

1654\. _𑀅𑀤𑁆𑀤𑀲 𑀪𑀕𑀯𑀸 𑀆𑀤𑀺𑀁,_  
_𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀓𑀧𑁆𑀧𑀺𑀬𑁄;_  
_𑀅𑀘𑁆𑀘𑀕𑀸 𑀯𑀢 𑀓𑀧𑁆𑀧𑀸𑀦𑁄,_  
_𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀤𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


(1288.)

1655\. _𑀢𑀁 𑀤𑁂𑀯𑀤𑁂𑀯𑀁 𑀯𑀦𑁆𑀤𑀸𑀫𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑀁 𑀢𑁂 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫;_  
_𑀅𑀦𑀼𑀚𑀸𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀦𑀸𑀕𑀁 𑀦𑀸𑀕𑀲𑁆𑀲 𑀑𑀭𑀲𑀦𑁆”𑀢𑀺𑁇_  


1656\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑀗𑁆𑀕𑀻𑀲𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1657\. 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

1658\. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1659\. _𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑁆𑀳𑀺 𑀦𑀺𑀧𑀸𑀢𑀫𑁆𑀳𑀺,_  
_𑀯𑀗𑁆𑀕𑀻𑀲𑁄 𑀧𑀝𑀺𑀪𑀸𑀡𑀯𑀸;_  
_𑀏𑀓𑁄𑀯 𑀣𑁂𑀭𑁄 𑀦𑀢𑁆𑀣𑀜𑁆𑀜𑁄,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀏𑀓𑀲𑀢𑁆𑀢𑀢𑀻𑀢𑀺𑁇_  


---

1660\. 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸 𑀣𑁂𑀭𑀕𑀸𑀣𑀸𑀬𑁄𑁇



---

1661\. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1662\. _𑀲𑀳𑀲𑁆𑀲𑀁 𑀳𑁄𑀦𑁆𑀢𑀺 𑀢𑀸 𑀕𑀸𑀣𑀸,_  
_𑀢𑀻𑀡𑀺 𑀲𑀝𑁆𑀞𑀺𑀲𑀢𑀸𑀦𑀺 𑀘;_  
_𑀣𑁂𑀭𑀸 𑀘 𑀤𑁆𑀯𑁂 𑀲𑀢𑀸 𑀲𑀝𑁆𑀞𑀺,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑁇_  


1663\. _𑀲𑀻𑀳𑀦𑀸𑀤𑀁 𑀦𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀲𑀯𑀸;_  
_𑀔𑁂𑀫𑀦𑁆𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀕𑁆𑀕𑀺𑀔𑀦𑁆𑀥𑀸𑀯 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸𑀢𑀺𑁇_  


---

1664\. 𑀣𑁂𑀭𑀕𑀸𑀣𑀸𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
