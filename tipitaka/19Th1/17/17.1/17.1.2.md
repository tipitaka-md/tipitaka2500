# 𑁧𑁭𑁇𑁧𑁇𑁨 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀁𑀲𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(981.)

1331. _“𑀬𑀣𑀸𑀘𑀸𑀭𑀻 𑀬𑀣𑀸𑀲𑀢𑁄 𑀲𑀢𑀻𑀫𑀸,_  
_𑀬𑀢𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀚𑁆𑀛𑀸𑀬𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀭𑀢𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑀢𑁆𑀢𑁄,_  
_𑀏𑀓𑁄 𑀲𑀦𑁆𑀢𑀼𑀲𑀺𑀢𑁄 𑀢𑀫𑀸𑀳𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀁𑁇_  


(982.)

1332. _𑀅𑀮𑁆𑀮𑀁 𑀲𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄,_  
_𑀦 𑀩𑀸𑀍𑀳𑀁 𑀲𑀼𑀳𑀺𑀢𑁄 𑀲𑀺𑀬𑀸;_  
_𑀊𑀦𑀽𑀤𑀭𑁄 𑀫𑀺𑀢𑀸𑀳𑀸𑀭𑁄,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂𑁇_  


(983.)

1333. _𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀜𑁆𑀘 𑀆𑀮𑁄𑀧𑁂,_  
_𑀅𑀪𑀼𑀢𑁆𑀯𑀸 𑀉𑀤𑀓𑀁 𑀧𑀺𑀯𑁂;_  
_𑀅𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


(984.)

1334. _𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀢𑀜𑁆𑀘𑁂 𑀙𑀸𑀤𑁂𑀢𑀺,_  
_𑀘𑀻𑀯𑀭𑀁 𑀇𑀤𑀫𑀢𑁆𑀣𑀺𑀓𑀁;_  
_𑀅𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


(985.)

1335. _𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀚𑀡𑁆𑀡𑀼𑀓𑁂 𑀦𑀸𑀪𑀺𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


(986.)

1336. _𑀬𑁄 𑀲𑀼𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀤𑁆𑀤,_  
_𑀤𑀼𑀓𑁆𑀔𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀲𑀮𑁆𑀮𑀢𑁄;_  
_𑀉𑀪𑀬𑀦𑁆𑀢𑀭𑁂𑀦 𑀦𑀸𑀳𑁄𑀲𑀺,_  
_𑀓𑁂𑀦 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀓𑀺𑀁 𑀲𑀺𑀬𑀸𑁇_  


(987.)

1337. _𑀫𑀸 𑀫𑁂 𑀓𑀤𑀸𑀘𑀺 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄,_  
_𑀓𑀼𑀲𑀻𑀢𑁄 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑁄;_  
_𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁄 𑀅𑀦𑀸𑀤𑀭𑁄,_  
_𑀓𑁂𑀦 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀓𑀺𑀁 𑀲𑀺𑀬𑀸𑁇_  


(988.)

1338. _𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀘𑁂𑀢𑁄𑀲𑀫𑀣𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀅𑀧𑀺 𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀢𑀼𑁇_  


(989.)

1339. _𑀬𑁄 𑀧𑀧𑀜𑁆𑀘𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀧𑀜𑁆𑀘𑀸𑀪𑀺𑀭𑀢𑁄 𑀫𑀕𑁄;_  
_𑀯𑀺𑀭𑀸𑀥𑀬𑀻 𑀲𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


(990.)

1340. _𑀬𑁄 𑀘 𑀧𑀧𑀜𑁆𑀘𑀁 𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀧𑀣𑁂 𑀭𑀢𑁄;_  
_𑀆𑀭𑀸𑀥𑀬𑀻 𑀲𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


(991.)

1341. _𑀕𑀸𑀫𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀦𑀺𑀦𑁆𑀦𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀣𑀮𑁂;_  
_𑀬𑀢𑁆𑀣 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀪𑀽𑀫𑀺𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁𑁇_  


(992.)

1342. _𑀭𑀫𑀡𑀻𑀬𑀸𑀦𑀺 𑀅𑀭𑀜𑁆𑀜𑀸𑀦𑀺,_  
_𑀬𑀢𑁆𑀣 𑀦 𑀭𑀫𑀢𑀻 𑀚𑀦𑁄;_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀭𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀦 𑀢𑁂 𑀓𑀸𑀫𑀕𑀯𑁂𑀲𑀺𑀦𑁄𑁇_  


(993.)

1343. _𑀦𑀺𑀥𑀻𑀦𑀁𑀯 𑀧𑀯𑀢𑁆𑀢𑀸𑀭𑀁,_  
_𑀬𑀁 𑀧𑀲𑁆𑀲𑁂 𑀯𑀚𑁆𑀚𑀤𑀲𑁆𑀲𑀺𑀦𑀁;_  
_𑀦𑀺𑀕𑁆𑀕𑀬𑁆𑀳𑀯𑀸𑀤𑀺𑀁 𑀫𑁂𑀥𑀸𑀯𑀺𑀁,_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀧𑀡𑁆𑀟𑀺𑀢𑀁 𑀪𑀚𑁂;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀚𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀲𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀧𑀸𑀧𑀺𑀬𑁄𑁇_  


(994.)

1344. _𑀑𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀦𑀼𑀲𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀅𑀲𑀩𑁆𑀪𑀸 𑀘 𑀦𑀺𑀯𑀸𑀭𑀬𑁂;_  
_𑀲𑀢𑀜𑁆𑀳𑀺 𑀲𑁄 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺,_  
_𑀅𑀲𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀺𑀬𑁄𑁇_  


(995.)

1345. _𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀲𑁄𑀢𑀫𑁄𑀥𑁂𑀲𑀺𑀫𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀢𑀁 𑀫𑁂 𑀅𑀫𑁄𑀖𑀁 𑀲𑀯𑀦𑀁,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(996.)

1346. _𑀦𑁂𑀯 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀬,_  
_𑀦𑀧𑀺 𑀤𑀺𑀩𑁆𑀩𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀬𑀸,_  
_𑀘𑀼𑀢𑀺𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸,_  
_𑀧𑀡𑀺𑀥𑀻 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(997.)

1347. _“𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀫𑀼𑀡𑁆𑀟𑁄 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀸𑀭𑀼𑀢𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀉𑀢𑁆𑀢𑀫𑁄 𑀣𑁂𑀭𑁄,_  
_𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄𑀯 𑀛𑀸𑀬𑀢𑀺𑁇_  


(998.)

1348. _𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑁄;_  
_𑀅𑀭𑀺𑀬𑁂𑀦 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦,_  
_𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(999.)

1349. _𑀬𑀣𑀸𑀧𑀺 𑀧𑀩𑁆𑀩𑀢𑁄 𑀲𑁂𑀮𑁄,_  
_𑀅𑀘𑀮𑁄 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀩𑁆𑀩𑀢𑁄𑀯 𑀦 𑀯𑁂𑀥𑀢𑀺”𑁇_  


(1000.)

1350. _“𑀅𑀦𑀗𑁆𑀕𑀡𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑀼𑀘𑀺𑀕𑀯𑁂𑀲𑀺𑀦𑁄;_  
_𑀯𑀸𑀮𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀁 𑀧𑀸𑀧𑀲𑁆𑀲,_  
_𑀅𑀩𑁆𑀪𑀫𑀢𑁆𑀢𑀁𑀯 𑀔𑀸𑀬𑀢𑀺𑁇_  


(1001.)

1351. _𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀫𑀭𑀡𑀁,_  
_𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀁 𑀇𑀫𑀁 𑀓𑀸𑀬𑀁,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄𑁇_  


(1002.)

1352. _𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀫𑀭𑀡𑀁,_  
_𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀓𑀸𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀲𑀁 𑀪𑀢𑀓𑁄 𑀬𑀣𑀸”𑁇_  


(1003.)

1353. _“𑀉𑀪𑀬𑁂𑀦 𑀫𑀺𑀤𑀁 𑀫𑀭𑀡𑀫𑁂𑀯,_  
_𑀦𑀸𑀫𑀭𑀡𑀁 𑀧𑀘𑁆𑀙𑀸 𑀯𑀸 𑀧𑀼𑀭𑁂 𑀯𑀸;_  
_𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀣 𑀫𑀸 𑀯𑀺𑀦𑀲𑁆𑀲𑀣,_  
_𑀔𑀡𑁄 𑀯𑁄 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸𑁇_  


(1004.)

1354. _𑀦𑀕𑀭𑀁 𑀬𑀣𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀕𑀼𑀢𑁆𑀢𑀁 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀁;_  
_𑀏𑀯𑀁 𑀕𑁄𑀧𑁂𑀣 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀔𑀡𑁄 𑀯𑁄 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀔𑀡𑀸𑀢𑀻𑀢𑀸 𑀳𑀺 𑀲𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸”𑁇_  


(1005.)

1355. _“𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀉𑀧𑀭𑀢𑁄,_  
_𑀫𑀦𑁆𑀢𑀪𑀸𑀡𑀻 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄;_  
_𑀥𑀼𑀦𑀸𑀢𑀺 𑀧𑀸𑀧𑀓𑁂 𑀥𑀫𑁆𑀫𑁂,_  
_𑀤𑀼𑀫𑀧𑀢𑁆𑀢𑀁𑀯 𑀫𑀸𑀮𑀼𑀢𑁄𑁇_  


(1006.)

1356. _𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀉𑀧𑀭𑀢𑁄,_  
_𑀫𑀦𑁆𑀢𑀪𑀸𑀡𑀻 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄;_  
_𑀅𑀧𑁆𑀧𑀸𑀲𑀺 𑀧𑀸𑀧𑀓𑁂 𑀥𑀫𑁆𑀫𑁂,_  
_𑀤𑀼𑀫𑀧𑀢𑁆𑀢𑀁𑀯 𑀫𑀸𑀮𑀼𑀢𑁄𑁇_  


(1007.)

1357. _𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑀸𑀬𑀸𑀲𑁄,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀅𑀦𑀸𑀯𑀺𑀮𑁄;_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀻𑀮𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀓𑀭𑁄 𑀲𑀺𑀬𑀸”𑁇_  


(1008.)

1358. _“𑀦 𑀯𑀺𑀲𑁆𑀲𑀲𑁂 𑀏𑀓𑀢𑀺𑀬𑁂𑀲𑀼 𑀏𑀯𑀁,_  
_𑀅𑀕𑀸𑀭𑀺𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀲𑀼 𑀘𑀸𑀧𑀺;_  
_𑀲𑀸𑀥𑀽𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀦 𑀅𑀲𑀸𑀥𑀼 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀅𑀲𑀸𑀥𑀼 𑀳𑀼𑀢𑁆𑀯𑀸 𑀧𑀼𑀦 𑀲𑀸𑀥𑀼 𑀳𑁄𑀦𑁆𑀢𑀺𑁇_  


(1009.)

1359. _𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄,_  
_𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀘,_  
_𑀧𑀜𑁆𑀘𑁂𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀓𑁂𑀮𑀺𑀲𑀸𑁇_  


(1010.)

1360. _𑀬𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑀭𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀲𑀫𑀸𑀥𑀺 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄𑁇_  


(1011.)

1361. _𑀢𑀁 𑀛𑀸𑀬𑀺𑀦𑀁 𑀲𑀸𑀢𑀢𑀺𑀓𑀁,_  
_𑀲𑀼𑀔𑀼𑀫𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑀁;_  
_𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀬𑀸𑀭𑀸𑀫𑀁,_  
_𑀆𑀳𑀼 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀢𑀺𑁇_  


(1012.)

1362. _𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀧𑀣𑀯𑀻,_  
_𑀧𑀩𑁆𑀩𑀢𑁄 𑀅𑀦𑀺𑀮𑁄𑀧𑀺 𑀘;_  
_𑀉𑀧𑀫𑀸𑀬 𑀦 𑀬𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀲𑀢𑁆𑀣𑀼 𑀯𑀭𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(1013.)

1363. _𑀘𑀓𑁆𑀓𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑁄 𑀣𑁂𑀭𑁄,_  
_𑀫𑀳𑀸𑀜𑀸𑀡𑀻 𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀧𑀣𑀯𑀸𑀧𑀕𑁆𑀕𑀺𑀲𑀫𑀸𑀦𑁄,_  
_𑀦 𑀭𑀚𑁆𑀚𑀢𑀺 𑀦 𑀤𑀼𑀲𑁆𑀲𑀢𑀺𑁇_  


(1014.)

1364. _𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀫𑀳𑀸𑀩𑀼𑀤𑁆𑀥𑀺 𑀫𑀳𑀸𑀫𑀢𑀺;_  
_𑀅𑀚𑀍𑀑 𑀚𑀍𑀅𑀲𑀫𑀸𑀦𑁄,_  
_𑀲𑀤𑀸 𑀘𑀭𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇_  


(1015.)

1365. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_…𑀧𑁂…_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(1016.)

1366. _𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀣𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦,_  
_𑀏𑀲𑀸 𑀫𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻;_  
_𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀲𑀩𑁆𑀩𑀥𑀻”𑀢𑀺𑁇_  


---

1367. …  𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
