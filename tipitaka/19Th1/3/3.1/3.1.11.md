# 𑁩𑁇𑁧𑁇𑁧𑁧 𑀉𑀧𑀸𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(249.)

474. _“𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀦𑀯𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀯𑁄;_  
_𑀫𑀺𑀢𑁆𑀢𑁂 𑀪𑀚𑁂𑀬𑁆𑀬 𑀓𑀮𑁆𑀬𑀸𑀡𑁂,_  
_𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁂 𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑁂𑁇_  


(250.)

475. _𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀦𑀯𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀯𑁄;_  
_𑀲𑀁𑀖𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀺𑀓𑁆𑀔𑁂𑀣 𑀯𑀺𑀦𑀬𑀁 𑀩𑀼𑀥𑁄𑁇_  


(251.)

476. _𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀦𑀯𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀯𑁄;_  
_𑀓𑀧𑁆𑀧𑀸𑀓𑀧𑁆𑀧𑁂𑀲𑀼 𑀓𑀼𑀲𑀮𑁄,_  
_𑀘𑀭𑁂𑀬𑁆𑀬 𑀅𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄”𑀢𑀺𑁇_  


---

477. …  𑀉𑀧𑀸𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑁄… 𑁇
