# 𑁩𑁇𑁧𑁇𑁧𑁦 𑀲𑀸𑀝𑀺𑀫𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀝𑀺𑀫𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(246.)

470\. _“𑀅𑀳𑀼 𑀢𑀼𑀬𑁆𑀳𑀁 𑀧𑀼𑀭𑁂 𑀲𑀤𑁆𑀥𑀸,_  
_𑀲𑀸 𑀢𑁂 𑀅𑀚𑁆𑀚 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀬𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀢𑀼𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀫𑀫𑁇_  


(247.)

471\. _𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀳𑀺 𑀘𑀮𑀸 𑀲𑀤𑁆𑀥𑀸,_  
_𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑀺 𑀲𑀸 𑀫𑀬𑀸;_  
_𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀢𑀢𑁆𑀣 𑀓𑀺𑀁 𑀚𑀺𑀬𑁆𑀬𑀢𑁂 𑀫𑀼𑀦𑀺𑁇_  


(248.)

472\. _𑀧𑀘𑁆𑀘𑀢𑀺 𑀫𑀼𑀦𑀺𑀦𑁄 𑀪𑀢𑁆𑀢𑀁,_  
_𑀣𑁄𑀓𑀁 𑀣𑁄𑀓𑀁 𑀓𑀼𑀮𑁂 𑀓𑀼𑀮𑁂;_  
_𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀅𑀢𑁆𑀣𑀺 𑀚𑀗𑁆𑀖𑀩𑀮𑀁 𑀫𑀫𑀸”𑀢𑀺𑁇_  


---

473\. …  𑀲𑀸𑀝𑀺𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
