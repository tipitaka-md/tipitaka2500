# 𑁩𑁇𑁧𑁇𑁫 𑀫𑀸𑀢𑀗𑁆𑀕𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀢𑀗𑁆𑀕𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(231.)

450. _“𑀅𑀢𑀺𑀲𑀻𑀢𑀁 𑀅𑀢𑀺𑀉𑀡𑁆𑀳𑀁,_  
_𑀅𑀢𑀺𑀲𑀸𑀬𑀫𑀺𑀤𑀁 𑀅𑀳𑀼;_  
_𑀇𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁂,_  
_𑀔𑀡𑀸 𑀅𑀘𑁆𑀘𑁂𑀦𑁆𑀢𑀺 𑀫𑀸𑀡𑀯𑁂𑁇_  


(232.)

451. _𑀬𑁄 𑀘 𑀲𑀻𑀢𑀜𑁆𑀘 𑀉𑀡𑁆𑀳𑀜𑁆𑀘,_  
_𑀢𑀺𑀡𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀦 𑀫𑀜𑁆𑀜𑀢𑀺;_  
_𑀓𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀺,_  
_𑀲𑁄 𑀲𑀼𑀔𑀸 𑀦 𑀯𑀺𑀳𑀸𑀬𑀢𑀺𑁇_  


(233.)

452. _𑀤𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀁 𑀧𑁄𑀝𑀓𑀺𑀮𑀁,_  
_𑀉𑀲𑀻𑀭𑀁 𑀫𑀼𑀜𑁆𑀚𑀧𑀩𑁆𑀩𑀚𑀁;_  
_𑀉𑀭𑀲𑀸 𑀧𑀦𑀼𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

453. …  𑀫𑀸𑀢𑀗𑁆𑀕𑀧𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
