# 𑁩𑁇𑁧𑁇𑁨 𑀧𑀘𑁆𑀘𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀧𑀘𑁆𑀘𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(222.)

438\. _“𑀧𑀜𑁆𑀘𑀸𑀳𑀸𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀲𑁂𑀔𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑁄;_  
_𑀯𑀺𑀳𑀸𑀭𑀁 𑀫𑁂 𑀧𑀯𑀺𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀘𑁂𑀢𑀲𑁄 𑀧𑀡𑀺𑀥𑀻 𑀅𑀳𑀼𑁇_  


(223.)

439\. _𑀦𑀸𑀲𑀺𑀲𑁆𑀲𑀁 𑀦 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀯𑀺𑀳𑀸𑀭𑀢𑁄 𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑁂;_  
_𑀦𑀧𑀺 𑀧𑀲𑁆𑀲𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑁆𑀲𑀁,_  
_𑀢𑀡𑁆𑀳𑀸𑀲𑀮𑁆𑀮𑁂 𑀅𑀦𑀽𑀳𑀢𑁂𑁇_  


(224.)

440\. _𑀢𑀲𑁆𑀲 𑀫𑁂𑀯𑀁 𑀯𑀺𑀳𑀭𑀢𑁄,_  
_𑀧𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀧𑀭𑀓𑁆𑀓𑀫𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

441\. …  𑀧𑀘𑁆𑀘𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
