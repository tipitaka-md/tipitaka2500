# 𑁩𑁇𑁧𑁇𑁬 𑀔𑀼𑀚𑁆𑀚𑀲𑁄𑀪𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀔𑀼𑀚𑁆𑀚𑀲𑁄𑀪𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(234.)

454\. _“𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀓𑀣𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸,_  
_𑀲𑀫𑀡𑀸 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀯𑀸𑀲𑀺𑀦𑁄;_  
_𑀢𑁂𑀲𑀜𑁆𑀜𑀢𑀭𑁄𑀬𑀫𑀸𑀬𑀼𑀯𑀸,_  
_𑀤𑁆𑀯𑀸𑀭𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀔𑀼𑀚𑁆𑀚𑀲𑁄𑀪𑀺𑀢𑁄”𑁇_  


(235.)

455\. _“𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀓𑀣𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸,_  
_𑀲𑀫𑀡𑀸 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀯𑀸𑀲𑀺𑀦𑁄;_  
_𑀢𑁂𑀲𑀜𑁆𑀜𑀢𑀭𑁄𑀬𑀫𑀸𑀬𑀼𑀯𑀸,_  
_𑀤𑁆𑀯𑀸𑀭𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀫𑀸𑀮𑀼𑀢𑁂𑀭𑀺𑀢𑁄”𑁇_  


(236.)

456\. _“𑀲𑀼𑀬𑀼𑀤𑁆𑀥𑁂𑀦 𑀲𑀼𑀬𑀺𑀝𑁆𑀞𑁂𑀦,_  
_𑀲𑀗𑁆𑀕𑀸𑀫𑀯𑀺𑀚𑀬𑁂𑀦 𑀘;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀦𑀼𑀘𑀺𑀡𑁆𑀡𑁂𑀦,_  
_𑀏𑀯𑀸𑀬𑀁 𑀲𑀼𑀔𑀫𑁂𑀥𑀢𑀻”𑀢𑀺𑁇_  


---

457\. …  𑀔𑀼𑀚𑁆𑀚𑀲𑁄𑀪𑀺𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
