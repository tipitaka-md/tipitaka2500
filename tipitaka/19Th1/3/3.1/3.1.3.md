# 𑁩𑁇𑁧𑁇𑁩 𑀩𑀸𑀓𑀼𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀓𑀼𑀮𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(225.)

442. _“𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺,_  
_𑀧𑀘𑁆𑀙𑀸 𑀲𑁄 𑀓𑀸𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀢𑀺;_  
_𑀲𑀼𑀔𑀸 𑀲𑁄 𑀥𑀁𑀲𑀢𑁂 𑀞𑀸𑀦𑀸,_  
_𑀧𑀘𑁆𑀙𑀸 𑀘 𑀫𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀺𑁇_  


(226.)

443. _𑀬𑀜𑁆𑀳𑀺 𑀓𑀬𑀺𑀭𑀸 𑀢𑀜𑁆𑀳𑀺 𑀯𑀤𑁂,_  
_𑀬𑀁 𑀦 𑀓𑀬𑀺𑀭𑀸 𑀦 𑀢𑀁 𑀯𑀤𑁂;_  
_𑀅𑀓𑀭𑁄𑀦𑁆𑀢𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀁,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


(227.)

444. _𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀅𑀲𑁄𑀓𑀁 𑀯𑀺𑀭𑀚𑀁 𑀔𑁂𑀫𑀁,_  
_𑀬𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_  


---

445. …  𑀩𑀸𑀓𑀼𑀮𑁄 𑀣𑁂𑀭𑁄… 𑁇
