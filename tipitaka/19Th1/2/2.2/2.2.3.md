# 𑁨𑁇𑁨𑁇𑁩 𑀳𑁂𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀳𑁂𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(145.)

312\. _“𑀅𑀘𑁆𑀘𑀬𑀦𑁆𑀢𑀺 𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀸,_  
_𑀚𑀻𑀯𑀺𑀢𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀆𑀬𑀼 𑀔𑀻𑀬𑀢𑀺 𑀫𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀓𑀼𑀦𑁆𑀦𑀤𑀻𑀦𑀁𑀯 𑀑𑀤𑀓𑀁𑁇_  


(146.)

313\. _𑀅𑀣 𑀧𑀸𑀧𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺,_  
_𑀓𑀭𑀁 𑀩𑀸𑀮𑁄 𑀦 𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀧𑀘𑁆𑀙𑀸𑀲𑁆𑀲 𑀓𑀝𑀼𑀓𑀁 𑀳𑁄𑀢𑀺,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑀺𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑁄”𑀢𑀺𑁇_  


---

314\. …  𑀳𑁂𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺𑀢𑁆𑀣𑁂𑀭𑁄… 𑁇
