# 𑁨𑁇𑁨𑁇𑁭 𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(153.)

324\. _“𑀩𑀳𑀽 𑀲𑀧𑀢𑁆𑀢𑁂 𑀮𑀪𑀢𑀺,_  
_𑀫𑀼𑀡𑁆𑀟𑁄 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀸𑀭𑀼𑀢𑁄;_  
_𑀮𑀸𑀪𑀻 𑀅𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀸𑀦𑀲𑁆𑀲,_  
_𑀯𑀢𑁆𑀣𑀲𑁆𑀲 𑀲𑀬𑀦𑀲𑁆𑀲 𑀘𑁇_  


(154.)

325\. _𑀏𑀢𑀫𑀸𑀤𑀻𑀦𑀯𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀲𑀼 𑀫𑀳𑀩𑁆𑀪𑀬𑀁;_  
_𑀅𑀧𑁆𑀧𑀮𑀸𑀪𑁄 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂”𑀢𑀺𑁇_  


---

326\. …  𑀢𑀺𑀲𑁆𑀲𑁄 𑀣𑁂𑀭𑁄… 𑁇
