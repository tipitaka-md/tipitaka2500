# 𑁨𑁇𑁨𑁇𑁬 𑀫𑀳𑀸𑀓𑀸𑀍𑀅𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀓𑀸𑀍𑀅𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(151.)

321. _“𑀓𑀸𑀍𑀈 𑀇𑀢𑁆𑀣𑀻 𑀩𑁆𑀭𑀳𑀢𑀻 𑀥𑀗𑁆𑀓𑀭𑀽𑀧𑀸,_  
_𑀲𑀢𑁆𑀣𑀺𑀜𑁆𑀘 𑀪𑁂𑀢𑁆𑀯𑀸 𑀅𑀧𑀭𑀜𑁆𑀘 𑀲𑀢𑁆𑀣𑀺𑀁;_  
_𑀩𑀸𑀳𑀜𑁆𑀘 𑀪𑁂𑀢𑁆𑀯𑀸 𑀅𑀧𑀭𑀜𑁆𑀘 𑀩𑀸𑀳𑀁,_  
_𑀲𑀻𑀲𑀜𑁆𑀘 𑀪𑁂𑀢𑁆𑀯𑀸 𑀤𑀥𑀺𑀣𑀸𑀮𑀓𑀁𑀯;_  
_𑀏𑀲𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀅𑀪𑀺𑀲𑀦𑁆𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑁇_  


(152.)

322. _𑀬𑁄 𑀯𑁂 𑀅𑀯𑀺𑀤𑁆𑀯𑀸 𑀉𑀧𑀥𑀺𑀁 𑀓𑀭𑁄𑀢𑀺,_  
_𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀼𑀧𑁂𑀢𑀺 𑀫𑀦𑁆𑀤𑁄;_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀚𑀸𑀦𑀁 𑀉𑀧𑀥𑀺𑀁 𑀦 𑀓𑀬𑀺𑀭𑀸,_  
_𑀫𑀸𑀳𑀁 𑀧𑀼𑀦 𑀪𑀺𑀦𑁆𑀦𑀲𑀺𑀭𑁄 𑀲𑀬𑀺𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇_  


---

323. …  𑀫𑀳𑀸𑀓𑀸𑀍𑀑 𑀣𑁂𑀭𑁄… 𑁇
