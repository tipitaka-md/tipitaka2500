# 𑁨𑁇𑁨𑁇𑁫 𑀲𑀩𑁆𑀩𑀫𑀺𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀩𑁆𑀩𑀫𑀺𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(149.)

318\. _“𑀚𑀦𑁄 𑀚𑀦𑀫𑁆𑀳𑀺 𑀲𑀫𑁆𑀩𑀤𑁆𑀥𑁄,_  
_𑀚𑀦𑀫𑁂𑀯𑀲𑁆𑀲𑀺𑀢𑁄 𑀚𑀦𑁄;_  
_𑀚𑀦𑁄 𑀚𑀦𑁂𑀦 𑀳𑁂𑀞𑀻𑀬𑀢𑀺,_  
_𑀳𑁂𑀞𑁂𑀢𑀺 𑀘 𑀚𑀦𑁄 𑀚𑀦𑀁𑁇_  


(150.)

319\. _𑀓𑁄 𑀳𑀺 𑀢𑀲𑁆𑀲 𑀚𑀦𑁂𑀦𑀢𑁆𑀣𑁄,_  
_𑀚𑀦𑁂𑀦 𑀚𑀦𑀺𑀢𑁂𑀦 𑀯𑀸;_  
_𑀚𑀦𑀁 𑀑𑀳𑀸𑀬 𑀕𑀘𑁆𑀙𑀁 𑀢𑀁,_  
_𑀳𑁂𑀞𑀬𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀁 𑀚𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

320\. …  𑀲𑀩𑁆𑀩𑀫𑀺𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
