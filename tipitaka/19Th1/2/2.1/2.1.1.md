# 𑁨𑁇𑁧𑁇𑁧 𑀉𑀢𑁆𑀢𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀉𑀢𑁆𑀢𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(121.)

274\. _“𑀦𑀢𑁆𑀣𑀺 𑀓𑁄𑀘𑀺 𑀪𑀯𑁄 𑀦𑀺𑀘𑁆𑀘𑁄,_  
_𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀯𑀸𑀧𑀺 𑀲𑀲𑁆𑀲𑀢𑀸;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀔𑀦𑁆𑀥𑀸,_  
_𑀘𑀯𑀦𑁆𑀢𑀺 𑀅𑀧𑀭𑀸𑀧𑀭𑀁𑁇_  


(122.)

275\. _𑀏𑀢𑀫𑀸𑀤𑀻𑀦𑀯𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀪𑀯𑁂𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀦𑀺𑀲𑁆𑀲𑀝𑁄 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑁂𑀳𑀺,_  
_𑀧𑀢𑁆𑀢𑁄 𑀫𑁂 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑁄”𑀢𑀺𑁇_  


276\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
