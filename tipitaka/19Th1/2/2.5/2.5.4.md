# 𑁨𑁇𑁫𑁇𑁪 𑀫𑁄𑀖𑀭𑀸𑀚𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

### 𑀫𑁄𑀖𑀭𑀸𑀚𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(207.)

411\. _“𑀙𑀯𑀺𑀧𑀸𑀧𑀓 𑀘𑀺𑀢𑁆𑀢𑀪𑀤𑁆𑀤𑀓,_  
_𑀫𑁄𑀖𑀭𑀸𑀚 𑀲𑀢𑀢𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀲𑀻𑀢𑀓𑀸𑀮𑀭𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁆𑀯𑀁𑀲𑀺 𑀓𑀣𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(208.)

412\. _“𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀲𑁆𑀲𑀸 𑀫𑀕𑀥𑀸,_  
_𑀓𑁂𑀯𑀮𑀸 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀧𑀮𑀸𑀮𑀘𑁆𑀙𑀦𑁆𑀦𑀓𑁄 𑀲𑁂𑀬𑁆𑀬𑀁,_  
_𑀬𑀣𑀜𑁆𑀜𑁂 𑀲𑀼𑀔𑀚𑀻𑀯𑀺𑀦𑁄”𑀢𑀺𑁇_  


---

413\. …  𑀫𑁄𑀖𑀭𑀸𑀚𑀸 𑀣𑁂𑀭𑁄… 𑁇
