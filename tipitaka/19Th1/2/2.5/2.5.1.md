# 𑁨𑁇𑁫𑁇𑁧 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(201.)

402\. _“𑀅𑀳𑁄 𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀳𑁄 𑀥𑀫𑁆𑀫𑀸,_  
_𑀅𑀳𑁄 𑀦𑁄 𑀲𑀢𑁆𑀣𑀼 𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀬𑀢𑁆𑀣 𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀸𑀯𑀓𑁄 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀳𑀺𑀢𑀺𑁇_  


(202.)

403\. _𑀅𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀓𑀧𑁆𑀧𑁂𑀲𑀼,_  
_𑀲𑀓𑁆𑀓𑀸𑀬𑀸𑀥𑀺𑀕𑀢𑀸 𑀅𑀳𑀽;_  
_𑀢𑁂𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑁄,_  
_𑀘𑀭𑀺𑀫𑁄𑀬𑀁 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑁄;_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀲𑀁𑀲𑀸𑀭𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇_  


---

404\. …  𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄… 𑁇
