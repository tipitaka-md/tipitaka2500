# 𑁨𑁇𑁫𑁇𑁯 𑀲𑀦𑁆𑀥𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀦𑁆𑀥𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(217.)

426\. _“𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑁂 𑀳𑀭𑀺𑀢𑁄𑀪𑀸𑀲𑁂,_  
_𑀲𑀁𑀯𑀺𑀭𑀽𑀍𑀳𑀫𑁆𑀳𑀺 𑀧𑀸𑀤𑀧𑁂;_  
_𑀏𑀓𑀁 𑀩𑀼𑀤𑁆𑀥𑀕𑀢𑀁 𑀲𑀜𑁆𑀜𑀁,_  
_𑀅𑀮𑀪𑀺𑀢𑁆𑀣𑀁 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄𑁇_  


(218.)

427\. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀢𑀲𑁆𑀲𑀸 𑀲𑀜𑁆𑀜𑀸𑀬 𑀯𑀸𑀳𑀲𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀫𑁂 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑁄”𑀢𑀺𑁇_  


---

428\. …  𑀲𑀦𑁆𑀥𑀺𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇

  
𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

429\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



430\. _𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄,_  
_𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑁄 𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀮𑀺;_  
_𑀫𑁄𑀖𑀭𑀸𑀚𑀸 𑀯𑀺𑀲𑀸𑀔𑁄 𑀘,_  
_𑀘𑀽𑀍𑀅𑀓𑁄 𑀘 𑀅𑀦𑀽𑀧𑀫𑁄;_  
_𑀯𑀚𑁆𑀚𑀺𑀢𑁄 𑀲𑀦𑁆𑀥𑀺𑀢𑁄 𑀣𑁂𑀭𑁄,_  
_𑀓𑀺𑀮𑁂𑀲𑀭𑀚𑀯𑀸𑀳𑀦𑁄𑀢𑀺𑁇_  


---

431\. 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

432\. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



433\. _𑀕𑀸𑀣𑀸𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢𑀫𑁆𑀳𑀺,_  
_𑀦𑀯𑀼𑀢𑀺 𑀘𑁂𑀯 𑀅𑀝𑁆𑀞 𑀘;_  
_𑀣𑁂𑀭𑀸 𑀏𑀓𑀽𑀦𑀧𑀜𑁆𑀜𑀸𑀲𑀁,_  
_𑀪𑀸𑀲𑀺𑀢𑀸 𑀦𑀬𑀓𑁄𑀯𑀺𑀤𑀸𑀢𑀺𑁇_
