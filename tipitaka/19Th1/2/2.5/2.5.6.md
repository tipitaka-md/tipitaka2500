# 𑁨𑁇𑁫𑁇𑁬 𑀘𑀽𑀍𑀅𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

* 𑀘𑀽𑀍𑀅𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(211.)

417\. _“𑀦𑀤𑀦𑁆𑀢𑀺 𑀫𑁄𑀭𑀸 𑀲𑀼𑀲𑀺𑀔𑀸 𑀲𑀼𑀧𑁂𑀔𑀼𑀡𑀸,_  
_𑀲𑀼𑀦𑀻𑀮𑀕𑀻𑀯𑀸 𑀲𑀼𑀫𑀼𑀔𑀸 𑀲𑀼𑀕𑀚𑁆𑀚𑀺𑀦𑁄;_  
_𑀲𑀼𑀲𑀤𑁆𑀤𑀮𑀸 𑀘𑀸𑀧𑀺 𑀫𑀳𑀸𑀫𑀳𑀻 𑀅𑀬𑀁,_  
_𑀲𑀼𑀩𑁆𑀬𑀸𑀧𑀺𑀢𑀫𑁆𑀩𑀼 𑀲𑀼𑀯𑀮𑀸𑀳𑀓𑀁 𑀦𑀪𑀁𑁇_  


(212.)

418\. _𑀲𑀼𑀓𑀮𑁆𑀮𑀭𑀽𑀧𑁄 𑀲𑀼𑀫𑀦𑀲𑁆𑀲 𑀛𑀸𑀬𑀢𑀁,_  
_𑀲𑀼𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀲𑀸𑀥𑀼 𑀲𑀼𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂;_  
_𑀲𑀼𑀲𑀼𑀓𑁆𑀓𑀲𑀼𑀓𑁆𑀓𑀁 𑀦𑀺𑀧𑀼𑀡𑀁 𑀲𑀼𑀤𑀼𑀤𑁆𑀤𑀲𑀁,_  
_𑀨𑀼𑀲𑀸𑀳𑀺 𑀢𑀁 𑀉𑀢𑁆𑀢𑀫𑀫𑀘𑁆𑀘𑀼𑀢𑀁 𑀧𑀤𑀦𑁆”𑀢𑀺𑁇_  


---

419\. …  𑀘𑀽𑀍𑀅𑀓𑁄 𑀣𑁂𑀭𑁄… 𑁇
