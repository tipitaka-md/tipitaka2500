# 𑁨𑁇𑁩𑁇𑁯 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(177.)

362. _“𑀦𑀤𑀦𑁆𑀢𑀺 𑀏𑀯𑀁 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀸,_  
_𑀲𑀻𑀳𑀸𑀯 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑁂;_  
_𑀯𑀻𑀭𑀸 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫𑀸,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀁𑁇_  


(178.)

363. _𑀲𑀢𑁆𑀣𑀸 𑀘 𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑁂,_  
_𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀖𑁄 𑀘 𑀧𑀽𑀚𑀺𑀢𑁄;_  
_𑀅𑀳𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀼𑀢𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀦𑀸𑀲𑀯𑀦𑁆”𑀢𑀺𑁇_  


---

364. …  𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀣𑁂𑀭𑁄… 𑁇
