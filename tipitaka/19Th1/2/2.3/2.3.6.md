# 𑁨𑁇𑁩𑁇𑁬 𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(171.)

353\. _“𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


(172.)

354\. _𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀁 𑀇𑀫𑀁 𑀓𑀸𑀬𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀁;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀜𑁆𑀘 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘,_  
_𑀢𑀼𑀘𑁆𑀙𑁄 𑀓𑀸𑀬𑁄 𑀅𑀤𑀺𑀲𑁆𑀲𑀣𑀸”𑀢𑀺𑁇_  


---

355\. …  𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑁄 𑀣𑁂𑀭𑁄… 𑁇
