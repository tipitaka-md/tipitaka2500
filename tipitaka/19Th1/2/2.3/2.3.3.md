# 𑁨𑁇𑁩𑁇𑁩 𑀲𑁄𑀪𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀪𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(165.)

344. _“𑀲𑀢𑀺𑀫𑀸 𑀧𑀜𑁆𑀜𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀆𑀭𑀤𑁆𑀥𑀩𑀮𑀯𑀻𑀭𑀺𑀬𑁄;_  
_𑀧𑀜𑁆𑀘 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀸𑀳𑀁,_  
_𑀏𑀓𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁𑁇_  


(166.)

345. _𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂,_  
_𑀲𑀢𑁆𑀢 𑀅𑀝𑁆𑀞 𑀘 𑀪𑀸𑀯𑀬𑀁;_  
_𑀧𑀜𑁆𑀘 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀸𑀳𑀁,_  
_𑀏𑀓𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

346. …  𑀲𑁄𑀪𑀺𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
