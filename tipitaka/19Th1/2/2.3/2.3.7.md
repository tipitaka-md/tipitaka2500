# 𑁨𑁇𑁩𑁇𑁭 𑀦𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(173.)

356\. _“𑀬𑀣𑀸𑀧𑀺 𑀪𑀤𑁆𑀤𑁄 𑀆𑀚𑀜𑁆𑀜𑁄,_  
_𑀔𑀮𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀢𑀺𑀝𑁆𑀞𑀢𑀺;_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀮𑀤𑁆𑀥𑀸𑀦 𑀲𑀁𑀯𑁂𑀕𑀁,_  
_𑀅𑀤𑀻𑀦𑁄 𑀯𑀳𑀢𑁂 𑀥𑀼𑀭𑀁𑁇_  


(174.)

357\. _𑀏𑀯𑀁 𑀤𑀲𑁆𑀲𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑀁;_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑀁 𑀫𑀁 𑀥𑀸𑀭𑁂𑀣,_  
_𑀧𑀼𑀢𑁆𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀑𑀭𑀲𑀦𑁆”𑀢𑀺𑁇_  


---

358\. …  𑀦𑀦𑁆𑀤𑀓𑁄 𑀣𑁂𑀭𑁄… 𑁇
