# 𑁨𑁇𑁩𑁇𑁧𑁦 𑀓𑀡𑁆𑀳𑀤𑀺𑀦𑁆𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀓𑀡𑁆𑀳𑀤𑀺𑀦𑁆𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(179.)

365\. _“𑀉𑀧𑀸𑀲𑀺𑀢𑀸 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸,_  
_𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄;_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀁,_  
_𑀅𑀜𑁆𑀚𑀲𑀁 𑀅𑀫𑀢𑁄𑀕𑀥𑀁𑁇_  


(180.)

366\. _𑀪𑀯𑀭𑀸𑀕𑀳𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄,_  
_𑀪𑀯𑀭𑀸𑀕𑁄 𑀧𑀼𑀦 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀦 𑀘𑀸𑀳𑀼 𑀦 𑀘 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺,_  
_𑀦 𑀘 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


---

367\. …  𑀓𑀡𑁆𑀳𑀤𑀺𑀦𑁆𑀦𑁄 𑀣𑁂𑀭𑁄… 𑁇

  
𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

368\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



369\. _𑀉𑀢𑁆𑀢𑀭𑁄 𑀪𑀤𑁆𑀤𑀚𑀺𑀢𑁆𑀣𑁂𑀭𑁄,_  
_𑀲𑁄𑀪𑀺𑀢𑁄 𑀯𑀮𑁆𑀮𑀺𑀬𑁄 𑀇𑀲𑀺;_  
_𑀯𑀻𑀢𑀲𑁄𑀓𑁄 𑀘 𑀬𑁄 𑀣𑁂𑀭𑁄,_  
_𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑁄 𑀘 𑀦𑀦𑁆𑀤𑀓𑁄;_  
_𑀪𑀭𑀢𑁄 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀘,_  
_𑀓𑀡𑁆𑀳𑀤𑀺𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀻𑀢𑀺𑁇_
