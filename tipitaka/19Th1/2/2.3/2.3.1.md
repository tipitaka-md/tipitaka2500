# 𑁨𑁇𑁩𑁇𑁧 𑀉𑀢𑁆𑀢𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀉𑀢𑁆𑀢𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(161.)

338\. _“𑀔𑀦𑁆𑀥𑀸 𑀫𑀬𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀸,_  
_𑀢𑀡𑁆𑀳𑀸 𑀫𑁂 𑀲𑀼𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀪𑀸𑀯𑀺𑀢𑀸 𑀫𑀫 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀫𑁂 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑁄𑁇_  


(162.)

339\. _𑀲𑁄𑀳𑀁 𑀔𑀦𑁆𑀥𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀩𑁆𑀩𑀳𑀺𑀢𑁆𑀯𑀸𑀦 𑀚𑀸𑀮𑀺𑀦𑀺𑀁;_  
_𑀪𑀸𑀯𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀁 𑀅𑀦𑀸𑀲𑀯𑁄”𑀢𑀺𑁇_  


---

340\. …  𑀉𑀢𑁆𑀢𑀭𑁄 𑀣𑁂𑀭𑁄… 𑁇
