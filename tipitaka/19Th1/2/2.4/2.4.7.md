# 𑁨𑁇𑁪𑁇𑁭 𑀲𑁄𑀡𑀧𑁄𑀝𑀺𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀡𑀧𑁄𑀝𑀺𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(193.)

388. _“𑀦 𑀢𑀸𑀯 𑀲𑀼𑀧𑀺𑀢𑀼𑀁 𑀳𑁄𑀢𑀺,_  
_𑀭𑀢𑁆𑀢𑀺 𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀫𑀸𑀮𑀺𑀦𑀻;_  
_𑀧𑀝𑀺𑀚𑀕𑁆𑀕𑀺𑀢𑀼𑀫𑁂𑀯𑁂𑀲𑀸,_  
_𑀭𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀯𑀺𑀚𑀸𑀦𑀢𑀸”𑁇_  


(194.)

389. _“𑀳𑀢𑁆𑀣𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀯𑀧𑀢𑀺𑀢𑀁,_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑁄 𑀘𑁂 𑀅𑀦𑀼𑀓𑁆𑀓𑀫𑁂;_  
_𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀫𑁂 𑀫𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑀜𑁆𑀘𑁂 𑀚𑀻𑀯𑁂 𑀧𑀭𑀸𑀚𑀺𑀢𑁄”𑀢𑀺𑁇_  


---

390. …  𑀲𑁄𑀡𑁄 𑀧𑁄𑀝𑀺𑀭𑀺𑀬𑁄 𑀣𑁂𑀭𑁄… 𑁇
