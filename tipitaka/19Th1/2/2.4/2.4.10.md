# 𑁨𑁇𑁪𑁇𑁧𑁦 𑀓𑀧𑁆𑀧𑀝𑀓𑀼𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀓𑀧𑁆𑀧𑀝𑀓𑀼𑀭𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(199.)

397\. _“𑀅𑀬𑀫𑀺𑀢𑀺 𑀓𑀧𑁆𑀧𑀝𑁄 𑀓𑀧𑁆𑀧𑀝𑀓𑀼𑀭𑁄,_  
_𑀅𑀘𑁆𑀙𑀸𑀬 𑀅𑀢𑀺𑀪𑀭𑀺𑀢𑀸𑀬;_  
_𑀅𑀫𑀢𑀖𑀝𑀺𑀓𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑀓𑀝𑀫𑀢𑁆𑀢𑁄,_  
_𑀓𑀢𑀧𑀤𑀁 𑀛𑀸𑀦𑀸𑀦𑀺 𑀑𑀘𑁂𑀢𑀼𑀁𑁇_  


(200.)

398\. _𑀫𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁 𑀓𑀧𑁆𑀧𑀝 𑀧𑀘𑀸𑀮𑁂𑀲𑀺,_  
_𑀫𑀸 𑀢𑁆𑀯𑀁 𑀉𑀧𑀓𑀡𑁆𑀡𑀫𑁆𑀳𑀺 𑀢𑀸𑀍𑀏𑀲𑁆𑀲𑀁;_  
_𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁 𑀓𑀧𑁆𑀧𑀝 𑀫𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀲𑀺,_  
_𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑀫𑁆𑀳𑀺 𑀧𑀘𑀮𑀸𑀬𑀫𑀸𑀦𑁄”𑀢𑀺𑁇_  


---

399\. …  𑀓𑀧𑁆𑀧𑀝𑀓𑀼𑀭𑁄 𑀣𑁂𑀭𑁄… 𑁇

  
𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇



---

400\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



401\. _𑀫𑀺𑀕𑀲𑀺𑀭𑁄 𑀲𑀺𑀯𑀓𑁄 𑀘,_  
_𑀉𑀧𑀯𑀸𑀡𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀇𑀲𑀺𑀤𑀺𑀦𑁆𑀦𑁄 𑀘 𑀓𑀘𑁆𑀘𑀸𑀦𑁄,_  
_𑀦𑀺𑀢𑀓𑁄 𑀘 𑀫𑀳𑀸𑀯𑀲𑀻;_  
_𑀧𑁄𑀝𑀺𑀭𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑀺𑀲𑀪𑁄,_  
_𑀉𑀲𑀪𑁄 𑀓𑀧𑁆𑀧𑀝𑀓𑀼𑀭𑁄𑀢𑀺𑁇_
