# 𑁨𑁦𑁇𑁧𑁇𑁧 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀲𑀝𑁆𑀞𑀺𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(1149.)

1512. _“𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸,_  
_𑀉𑀜𑁆𑀙𑀸𑀧𑀢𑁆𑀢𑀸𑀕𑀢𑁂 𑀭𑀢𑀸;_  
_𑀤𑀸𑀮𑁂𑀫𑀼 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁,_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀸𑁇_  


(1150.)

1513. _𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸,_  
_𑀉𑀜𑁆𑀙𑀸𑀧𑀢𑁆𑀢𑀸𑀕𑀢𑁂 𑀭𑀢𑀸;_  
_𑀥𑀼𑀦𑀸𑀫 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁,_  
_𑀦𑀍𑀆𑀕𑀸𑀭𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄𑁇_  


(1151.)

1514. _𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀺𑀓𑀸 𑀲𑀸𑀢𑀢𑀺𑀓𑀸,_  
_𑀉𑀜𑁆𑀙𑀸𑀧𑀢𑁆𑀢𑀸𑀕𑀢𑁂 𑀭𑀢𑀸;_  
_𑀤𑀸𑀮𑁂𑀫𑀼 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁,_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀸𑁇_  


(1152.)

1515. _𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀺𑀓𑀸 𑀲𑀸𑀢𑀢𑀺𑀓𑀸,_  
_𑀉𑀜𑁆𑀙𑀸𑀧𑀢𑁆𑀢𑀸𑀕𑀢𑁂 𑀭𑀢𑀸;_  
_𑀥𑀼𑀦𑀸𑀫 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁,_  
_𑀦𑀍𑀆𑀕𑀸𑀭𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄”𑁇_  


(1153.)

1516. _“𑀅𑀝𑁆𑀞𑀺𑀓𑀗𑁆𑀓𑀮𑀓𑀼𑀝𑀺𑀓𑁂,_  
_𑀫𑀁𑀲𑀦𑁆𑀳𑀸𑀭𑀼𑀧𑀲𑀺𑀩𑁆𑀩𑀺𑀢𑁂;_  
_𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀧𑀼𑀭𑁂 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑁂,_  
_𑀧𑀭𑀕𑀢𑁆𑀢𑁂 𑀫𑀫𑀸𑀬𑀲𑁂𑁇_  


(1154.)

1517. _𑀕𑀽𑀣𑀪𑀲𑁆𑀢𑁂 𑀢𑀘𑁄𑀦𑀤𑁆𑀥𑁂,_  
_𑀉𑀭𑀕𑀡𑁆𑀟𑀺𑀧𑀺𑀲𑀸𑀘𑀺𑀦𑀺;_  
_𑀦𑀯 𑀲𑁄𑀢𑀸𑀦𑀺 𑀢𑁂 𑀓𑀸𑀬𑁂,_  
_𑀬𑀸𑀦𑀺 𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


(1155.)

1518. _𑀢𑀯 𑀲𑀭𑀻𑀭𑀁 𑀦𑀯𑀲𑁄𑀢𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀓𑀭𑀁 𑀧𑀭𑀺𑀩𑀦𑁆𑀥𑀁;_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀢𑁂 𑀢𑀁,_  
_𑀫𑀻𑀍𑀳𑀁 𑀘 𑀬𑀣𑀸 𑀲𑀼𑀘𑀺𑀓𑀸𑀫𑁄𑁇_  


(1156.)

1519. _𑀏𑀯𑀜𑁆𑀘𑁂 𑀢𑀁 𑀚𑀦𑁄 𑀚𑀜𑁆𑀜𑀸,_  
_𑀬𑀣𑀸 𑀚𑀸𑀦𑀸𑀫𑀺 𑀢𑀁 𑀅𑀳𑀁;_  
_𑀆𑀭𑀓𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀕𑀽𑀣𑀝𑁆𑀞𑀸𑀦𑀁𑀯 𑀧𑀸𑀯𑀼𑀲𑁂”𑁇_  


(1157.)

1520. _“𑀏𑀯𑀫𑁂𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀬𑀣𑀸 𑀲𑀫𑀡 𑀪𑀸𑀲𑀲𑀺;_  
_𑀏𑀢𑁆𑀣 𑀘𑁂𑀓𑁂 𑀯𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺,_  
_𑀧𑀗𑁆𑀓𑀫𑁆𑀳𑀺𑀯 𑀚𑀭𑀕𑁆𑀕𑀯𑁄”𑁇_  


(1158.)

1521. _“𑀆𑀓𑀸𑀲𑀫𑁆𑀳𑀺 𑀳𑀮𑀺𑀤𑁆𑀤𑀺𑀬𑀸,_  
_𑀬𑁄 𑀫𑀜𑁆𑀜𑁂𑀣 𑀭𑀚𑁂𑀢𑀯𑁂;_  
_𑀅𑀜𑁆𑀜𑁂𑀦 𑀯𑀸𑀧𑀺 𑀭𑀗𑁆𑀕𑁂𑀦,_  
_𑀯𑀺𑀖𑀸𑀢𑀼𑀤𑀬𑀫𑁂𑀯 𑀢𑀁𑁇_  


(1159.)

1522. _𑀢𑀤𑀸𑀓𑀸𑀲𑀲𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀫𑀸 𑀧𑀸𑀧𑀘𑀺𑀢𑁆𑀢𑁂 𑀆𑀲𑀸𑀤𑀺,_  
_𑀅𑀕𑁆𑀕𑀺𑀔𑀦𑁆𑀥𑀁𑀯 𑀧𑀓𑁆𑀔𑀺𑀫𑀸”𑁇_  


(1160.)

1523. _“𑀧𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀓𑀢𑀁 𑀩𑀺𑀫𑁆𑀩𑀁,_  
_𑀅𑀭𑀼𑀓𑀸𑀬𑀁 𑀲𑀫𑀼𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀆𑀢𑀼𑀭𑀁 𑀩𑀳𑀼𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁,_  
_𑀬𑀲𑁆𑀲 𑀦𑀢𑁆𑀣𑀺 𑀥𑀼𑀯𑀁 𑀞𑀺𑀢𑀺𑁇_  


(1161.)

1524. _𑀧𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀓𑀢𑀁 𑀭𑀽𑀧𑀁,_  
_𑀫𑀡𑀺𑀦𑀸 𑀓𑀼𑀡𑁆𑀟𑀮𑁂𑀦 𑀘;_  
_𑀅𑀝𑁆𑀞𑀺𑀁 𑀢𑀘𑁂𑀦 𑀑𑀦𑀤𑁆𑀥𑀁,_  
_𑀲𑀳 𑀯𑀢𑁆𑀣𑁂𑀳𑀺 𑀲𑁄𑀪𑀢𑀺𑁇_  


(1162.)

1525. _𑀅𑀮𑀢𑁆𑀢𑀓𑀓𑀢𑀸 𑀧𑀸𑀤𑀸,_  
_𑀫𑀼𑀔𑀁 𑀘𑀼𑀡𑁆𑀡𑀓𑀫𑀓𑁆𑀔𑀺𑀢𑀁;_  
_𑀅𑀮𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀫𑁄𑀳𑀸𑀬,_  
_𑀦𑁄 𑀘 𑀧𑀸𑀭𑀕𑀯𑁂𑀲𑀺𑀦𑁄𑁇_  


(1163.)

1526. _𑀅𑀝𑁆𑀞𑀧𑀤𑀓𑀢𑀸 𑀓𑁂𑀲𑀸,_  
_𑀦𑁂𑀢𑁆𑀢𑀸 𑀅𑀜𑁆𑀚𑀦𑀫𑀓𑁆𑀔𑀺𑀢𑀸;_  
_𑀅𑀮𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀫𑁄𑀳𑀸𑀬,_  
_𑀦𑁄 𑀘 𑀧𑀸𑀭𑀕𑀯𑁂𑀲𑀺𑀦𑁄𑁇_  


(1164.)

1527. _𑀅𑀜𑁆𑀚𑀦𑀻𑀯 𑀦𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀧𑀽𑀢𑀺𑀓𑀸𑀬𑁄 𑀅𑀮𑀗𑁆𑀓𑀢𑁄;_  
_𑀅𑀮𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀫𑁄𑀳𑀸𑀬,_  
_𑀦𑁄 𑀘 𑀧𑀸𑀭𑀕𑀯𑁂𑀲𑀺𑀦𑁄𑁇_  


(1165.)

1528. _𑀑𑀤𑀳𑀺 𑀫𑀺𑀕𑀯𑁄 𑀧𑀸𑀲𑀁,_  
_𑀦𑀸𑀲𑀤𑀸 𑀯𑀸𑀕𑀼𑀭𑀁 𑀫𑀺𑀕𑁄;_  
_𑀪𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀧𑀁 𑀕𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀤𑁆𑀤𑀦𑁆𑀢𑁂 𑀫𑀺𑀕𑀩𑀦𑁆𑀥𑀓𑁂𑁇_  


(1166.)

1529. _𑀙𑀺𑀦𑁆𑀦𑁄 𑀧𑀸𑀲𑁄 𑀫𑀺𑀕𑀯𑀲𑁆𑀲,_  
_𑀦𑀸𑀲𑀤𑀸 𑀯𑀸𑀕𑀼𑀭𑀁 𑀫𑀺𑀕𑁄;_  
_𑀪𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀧𑀁 𑀕𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑁄𑀘𑀦𑁆𑀢𑁂 𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑀓𑁂”𑁇_  


(1167.)

1530. _“𑀢𑀤𑀸𑀲𑀺 𑀬𑀁 𑀪𑀺𑀁𑀲𑀦𑀓𑀁,_  
_𑀢𑀤𑀸𑀲𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀅𑀦𑁂𑀓𑀸𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂𑁇_  


(1168.)

1531. _𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀢 𑀲𑀗𑁆𑀔𑀸𑀭𑀸,_  
_𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀬𑀥𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀉𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀲𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀲𑀼𑀔𑁄𑁇_  


(1169.)

1532. _𑀲𑀼𑀔𑀼𑀫𑀁 𑀢𑁂 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀯𑀸𑀮𑀕𑁆𑀕𑀁 𑀉𑀲𑀼𑀦𑀸 𑀬𑀣𑀸;_  
_𑀬𑁂 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀧𑀭𑀢𑁄 𑀦𑁄 𑀘 𑀅𑀢𑁆𑀢𑀢𑁄𑁇_  


(1170.)

1533. _𑀬𑁂 𑀘 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂,_  
_𑀧𑀭𑀢𑁄 𑀦𑁄 𑀘 𑀅𑀢𑁆𑀢𑀢𑁄;_  
_𑀧𑀘𑁆𑀘𑀩𑁆𑀬𑀸𑀥𑀺𑀁𑀲𑀼 𑀦𑀺𑀧𑀼𑀡𑀁,_  
_𑀯𑀸𑀮𑀕𑁆𑀕𑀁 𑀉𑀲𑀼𑀦𑀸 𑀬𑀣𑀸𑁇_  


(1171.)

1534. _𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀬 𑀑𑀫𑀝𑁆𑀞𑁄,_  
_𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄𑀯 𑀫𑀢𑁆𑀣𑀓𑁂;_  
_𑀓𑀸𑀫𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂𑁇_  


(1172.)

1535. _𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀬 𑀑𑀫𑀝𑁆𑀞𑁄,_  
_𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄𑀯 𑀫𑀢𑁆𑀣𑀓𑁂;_  
_𑀪𑀯𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂𑁇_  


(1173.)

1536. _𑀘𑁄𑀤𑀺𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺𑀫𑀥𑀸𑀭𑀺𑀦𑀸;_  
_𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑀁,_  
_𑀧𑀸𑀤𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑁂𑀦 𑀓𑀫𑁆𑀧𑀬𑀺𑀁𑁇_  


(1174.)

1537. _𑀦𑀬𑀺𑀤𑀁 𑀲𑀺𑀣𑀺𑀮𑀫𑀸𑀭𑀩𑁆𑀪,_  
_𑀦𑀬𑀺𑀤𑀁 𑀅𑀧𑁆𑀧𑁂𑀦 𑀣𑀸𑀫𑀲𑀸;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑀥𑀺𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀲𑀩𑁆𑀩𑀕𑀦𑁆𑀣𑀧𑀫𑁄𑀘𑀦𑀁𑁇_  


(1175.)

1538. _𑀅𑀬𑀜𑁆𑀘 𑀤𑀳𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀅𑀬𑀫𑀼𑀢𑁆𑀢𑀫𑀧𑁄𑀭𑀺𑀲𑁄;_  
_𑀥𑀸𑀭𑁂𑀢𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀁𑁇_  


(1176.)

1539. _𑀯𑀺𑀯𑀭𑀫𑀦𑀼𑀧𑀪𑀦𑁆𑀢𑀺 𑀯𑀺𑀚𑁆𑀚𑀼𑀢𑀸,_  
_𑀯𑁂𑀪𑀸𑀭𑀲𑁆𑀲 𑀘 𑀧𑀡𑁆𑀟𑀯𑀲𑁆𑀲 𑀘;_  
_𑀦𑀕𑀯𑀺𑀯𑀭𑀕𑀢𑁄 𑀛𑀸𑀬𑀢𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀫𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(1177.)

1540. _𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀉𑀧𑀭𑀢𑁄,_  
_𑀧𑀦𑁆𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑁄 𑀫𑀼𑀦𑀺;_  
_𑀤𑀸𑀬𑀸𑀤𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀅𑀪𑀺𑀯𑀦𑁆𑀤𑀺𑀢𑁄”𑁇_  


(1178.)

1541. _“𑀉𑀧𑀲𑀦𑁆𑀢𑀁 𑀉𑀧𑀭𑀢𑀁,_  
_𑀧𑀦𑁆𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀫𑀼𑀦𑀺𑀁;_  
_𑀤𑀸𑀬𑀸𑀤𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀯𑀦𑁆𑀤 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀓𑀲𑁆𑀲𑀧𑀁𑁇_  


(1179.)

1542. _𑀬𑁄 𑀘 𑀚𑀸𑀢𑀺𑀲𑀢𑀁 𑀕𑀘𑁆𑀙𑁂,_  
_𑀲𑀩𑁆𑀩𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀚𑀸𑀢𑀺𑀬𑁄;_  
_𑀲𑁄𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑁂𑀤𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁𑁇_  


(1180.)

1543. _𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄𑀧𑀺 𑀘𑁂 𑀅𑀲𑁆𑀲,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀏𑀢𑀲𑁆𑀲 𑀯𑀦𑁆𑀤𑀦𑀸𑀬𑁂𑀢𑀁,_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


(1181.)

1544. _𑀬𑁄 𑀲𑁄 𑀅𑀝𑁆𑀞 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦𑀺,_  
_𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀅𑀨𑀲𑁆𑀲𑀬𑀺;_  
_𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀧𑀝𑀺𑀮𑁄𑀫𑀁,_  
_𑀢𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(1182.)

1545. _𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀫𑀸𑀲𑀸𑀤𑀺,_  
_𑀫𑀸𑀢𑁆𑀢𑀸𑀦𑀁 𑀔𑀡𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀳𑀺 𑀫𑀦𑀁,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑀫𑁆𑀳𑀺 𑀢𑀸𑀤𑀺𑀦𑁂;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀯𑀦𑁆𑀤,_  
_𑀫𑀸 𑀢𑁂 𑀯𑀺𑀚𑀝𑀺 𑀫𑀢𑁆𑀣𑀓𑀁”𑁇_  


(1183.)

1546. _“𑀦𑁂𑀲𑁄 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀁𑀲𑀸𑀭𑁂𑀦 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀅𑀥𑁄𑀕𑀫𑀁 𑀚𑀺𑀫𑁆𑀳𑀧𑀣𑀁,_  
_𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀫𑀦𑀼𑀥𑀸𑀯𑀢𑀺𑁇_  


(1184.)

1547. _𑀓𑀺𑀫𑀻𑀯 𑀫𑀻𑀍𑀳𑀲𑀮𑁆𑀮𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄;_  
_𑀧𑀕𑀸𑀍𑀳𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑁂,_  
_𑀢𑀼𑀘𑁆𑀙𑁄 𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑁄𑀝𑁆𑀞𑀺𑀮𑁄”𑁇_  


(1185.)

1548. _“𑀇𑀫𑀜𑁆𑀘 𑀧𑀲𑁆𑀲 𑀆𑀬𑀦𑁆𑀢𑀁,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑁂,_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀁𑁇_  


(1186.)

1549. _𑀯𑀺𑀲𑀮𑁆𑀮𑀁 𑀔𑀻𑀡𑀲𑀁𑀬𑁄𑀕𑀁,_  
_𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀁 𑀫𑀘𑁆𑀘𑀼𑀳𑀸𑀬𑀺𑀦𑀁;_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁”𑁇_  


(1187.)

1550. _“𑀏𑀢𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀤𑁂𑀯𑀸,_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀤𑀲 𑀤𑁂𑀯𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀸;_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀚𑀮𑀻𑀓𑀢𑀸𑁇_  


(1188.)

1551. _‘𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀬𑀲𑁆𑀲 𑀢𑁂 𑀆𑀲𑀯𑀸 𑀔𑀻𑀡𑀸,_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄𑀲𑀺 𑀫𑀸𑀭𑀺𑀲’𑁇_  


(1189.)

1552. _𑀧𑀽𑀚𑀺𑀢𑁄 𑀦𑀭𑀤𑁂𑀯𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀫𑀭𑀡𑀸𑀪𑀺𑀪𑀽;_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀁𑀯 𑀢𑁄𑀬𑁂𑀦,_  
_𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦𑀼𑀧𑀮𑀺𑀧𑁆𑀧𑀢𑀺𑁇_  


(1190.)

1553. _𑀬𑀲𑁆𑀲 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀳𑀲𑁆𑀲𑀥𑀸 𑀮𑁄𑀓𑁄,_  
_𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀧𑁆𑀧𑁄 𑀯𑀲𑀺;_  
_𑀇𑀤𑁆𑀥𑀺𑀕𑀼𑀡𑁂 𑀘𑀼𑀢𑀼𑀧𑀧𑀸𑀢𑁂 𑀓𑀸𑀮𑁂,_  
_𑀧𑀲𑁆𑀲𑀢𑀺 𑀤𑁂𑀯𑀢𑀸 𑀲 𑀪𑀺𑀓𑁆𑀔𑀼”𑁇_  


(1191.)

1554. _“𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀯 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀻𑀮𑁂𑀦 𑀉𑀧𑀲𑀫𑁂𑀦 𑀘;_  
_𑀬𑁄𑀧𑀺 𑀧𑀸𑀭𑀗𑁆𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀏𑀢𑀸𑀯𑀧𑀭𑀫𑁄 𑀲𑀺𑀬𑀸𑁇_  


(1192.)

1555. _𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲,_  
_𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀁 𑀔𑀡𑁂𑀦 𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑁂;_  
_𑀅𑀳𑀁 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀸𑀲𑀼 𑀓𑀼𑀲𑀮𑁄,_  
_𑀯𑀲𑀻𑀪𑀽𑀢𑁄𑀫𑁆𑀳𑀺 𑀇𑀤𑁆𑀥𑀺𑀬𑀸𑁇_  


(1193.)

1556. _𑀲𑀫𑀸𑀥𑀺𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀲𑀺𑀧𑀸𑀭𑀫𑀻𑀕𑀢𑁄,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀕𑁄𑀢𑁆𑀢𑁄 𑀅𑀲𑀺𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂;_  
_𑀥𑀻𑀭𑁄 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺 𑀲𑀫𑀸𑀳𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄,_  
_𑀦𑀸𑀕𑁄 𑀬𑀣𑀸 𑀧𑀽𑀢𑀺𑀮𑀢𑀁𑀯 𑀩𑀦𑁆𑀥𑀦𑀁𑁇_  


(1194.)

1557. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(1195.)

1558. _𑀬𑀲𑁆𑀲 𑀘𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄𑁇_  


(1196.)

1559. _𑀓𑀻𑀤𑀺𑀲𑁄 𑀦𑀺𑀭𑀬𑁄 𑀆𑀲𑀺,_  
_𑀬𑀢𑁆𑀣 𑀤𑀼𑀲𑁆𑀲𑀻 𑀅𑀧𑀘𑁆𑀘𑀣;_  
_𑀯𑀺𑀥𑀼𑀭𑀁 𑀲𑀸𑀯𑀓𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_  


(1197.)

1560. _𑀲𑀢𑀁 𑀆𑀲𑀺 𑀅𑀬𑁄𑀲𑀗𑁆𑀓𑀽,_  
_𑀲𑀩𑁆𑀩𑁂 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀯𑁂𑀤𑀦𑀸;_  
_𑀈𑀤𑀺𑀲𑁄 𑀦𑀺𑀭𑀬𑁄 𑀆𑀲𑀺,_  
_𑀬𑀢𑁆𑀣 𑀤𑀼𑀲𑁆𑀲𑀻 𑀅𑀧𑀘𑁆𑀘𑀣;_  
_𑀯𑀺𑀥𑀼𑀭𑀁 𑀲𑀸𑀯𑀓𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_  


(1198.)

1561. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(1199.)

1562. _𑀫𑀚𑁆𑀛𑁂𑀲𑀭𑀲𑁆𑀫𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀫𑀸𑀦𑀸 𑀓𑀧𑁆𑀧𑀞𑀸𑀬𑀺𑀦𑁄;_  
_𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀯𑀡𑁆𑀡𑀸 𑀭𑀼𑀘𑀺𑀭𑀸,_  
_𑀅𑀘𑁆𑀘𑀺𑀫𑀦𑁆𑀢𑁄 𑀧𑀪𑀲𑁆𑀲𑀭𑀸;_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀢𑀢𑁆𑀣 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀣𑀼 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀬𑁄𑁇_  


(1200.)

1563. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(1201.)

1564. _𑀬𑁄 𑀯𑁂 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑁂𑀓𑁆𑀔𑀢𑁄;_  
_𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑀁,_  
_𑀧𑀸𑀤𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑁂𑀦 𑀓𑀫𑁆𑀧𑀬𑀺𑁇_  


(1202.)

1565. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(1203.)

1566. _𑀬𑁄 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑀁,_  
_𑀧𑀸𑀤𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑁂𑀦 𑀓𑀫𑁆𑀧𑀬𑀺;_  
_𑀇𑀤𑁆𑀥𑀺𑀩𑀮𑁂𑀦𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑁄,_  
_𑀲𑀁𑀯𑁂𑀚𑁂𑀲𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑁇_  


(1204.)

1567. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(1205.)

1568. _𑀬𑁄 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑁂,_  
_𑀲𑀓𑁆𑀓𑀁 𑀲𑁄 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺;_  
_𑀅𑀧𑀺 𑀆𑀯𑀼𑀲𑁄 𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀢𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀬𑀣𑀸𑀢𑀣𑀁”𑁇_  


(1206.)

1569. _“𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(1207.)

1570. _𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀞𑀺𑀢𑁄 𑀲𑀪𑀁;_  
_𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀢𑁆𑀬𑀸𑀯𑀼𑀲𑁄 𑀲𑀸 𑀤𑀺𑀝𑁆𑀞𑀺,_  
_𑀬𑀸 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀼𑀭𑁂 𑀅𑀳𑀼;_  
_𑀧𑀲𑁆𑀲𑀲𑀺 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀧𑀪𑀲𑁆𑀲𑀭𑀁𑁇_  


(1208.)

1571. _𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀦 𑀫𑁂 𑀫𑀸𑀭𑀺𑀲 𑀲𑀸 𑀤𑀺𑀝𑁆𑀞𑀺,_  
_𑀬𑀸 𑀫𑁂 𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀼𑀭𑁂 𑀅𑀳𑀼𑁇_  


(1209.)

1572. _𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀧𑀪𑀲𑁆𑀲𑀭𑀁;_  
_𑀲𑁄𑀳𑀁 𑀅𑀚𑁆𑀚 𑀓𑀣𑀁 𑀯𑀚𑁆𑀚𑀁,_  
_𑀅𑀳𑀁 𑀦𑀺𑀘𑁆𑀘𑁄𑀫𑁆𑀳𑀺 𑀲𑀲𑁆𑀲𑀢𑁄”𑁇_  


(1210.)

1573. _“𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(1211.)

1574. _𑀬𑁄 𑀫𑀳𑀸𑀦𑁂𑀭𑀼𑀦𑁄 𑀓𑀽𑀝𑀁,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁂𑀦 𑀅𑀨𑀲𑁆𑀲𑀬𑀺;_  
_𑀯𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁,_  
_𑀬𑁂 𑀘 𑀪𑀽𑀫𑀺𑀲𑀬𑀸 𑀦𑀭𑀸𑁇_  


(1212.)

1575. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(1213.)

1576. _𑀦 𑀯𑁂 𑀅𑀕𑁆𑀕𑀺 𑀘𑁂𑀢𑀬𑀢𑀺,_  
_𑀅𑀳𑀁 𑀩𑀸𑀮𑀁 𑀟𑀳𑀸𑀫𑀻𑀢𑀺;_  
_𑀩𑀸𑀮𑁄𑀯 𑀚𑀮𑀺𑀢𑀁 𑀅𑀕𑁆𑀕𑀺𑀁,_  
_𑀆𑀲𑀚𑁆𑀚 𑀦𑀁 𑀧𑀟𑀬𑁆𑀳𑀢𑀺𑁇_  


(1214.)

1577. _𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑀼𑀯𑀁 𑀫𑀸𑀭,_  
_𑀆𑀲𑀚𑁆𑀚 𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑀁;_  
_𑀲𑀬𑀁 𑀟𑀳𑀺𑀲𑁆𑀲𑀲𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀩𑀸𑀮𑁄 𑀅𑀕𑁆𑀕𑀺𑀁𑀯 𑀲𑀫𑁆𑀨𑀼𑀲𑀁𑁇_  


(1215.)

1578. _𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀻 𑀫𑀸𑀭𑁄,_  
_𑀆𑀲𑀚𑁆𑀚 𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑀁;_  
_𑀓𑀺𑀁 𑀦𑀼 𑀫𑀜𑁆𑀜𑀲𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦 𑀫𑁂 𑀧𑀸𑀧𑀁 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀺𑁇_  


(1216.)

1579. _𑀓𑀭𑀢𑁄 𑀢𑁂 𑀘𑀻𑀬𑀢𑁂 𑀧𑀸𑀧𑀁,_  
_𑀘𑀺𑀭𑀭𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑁆𑀢𑀓;_  
_𑀫𑀸𑀭 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤 𑀩𑀼𑀤𑁆𑀥𑀫𑁆𑀳𑀸,_  
_𑀆𑀲𑀁 𑀫𑀸𑀓𑀸𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼”𑁇_  


(1217.)

1580. _“𑀇𑀢𑀺 𑀫𑀸𑀭𑀁 𑀅𑀢𑀚𑁆𑀚𑁂𑀲𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂;_  
_𑀢𑀢𑁄 𑀲𑁄 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀬𑀓𑁆𑀔𑁄,_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀣𑀸”𑀢𑀺𑁇_  


1581. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀣𑁂𑀭𑁄 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1582. 𑀲𑀝𑁆𑀞𑀺𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

1583. 𑀢𑀢𑁆𑀭𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1584. _𑀲𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀳𑀺 𑀦𑀺𑀧𑀸𑀢𑀫𑁆𑀳𑀺,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄;_  
_𑀏𑀓𑁄𑀯 𑀣𑁂𑀭𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑀸𑀢𑀺𑁇_
