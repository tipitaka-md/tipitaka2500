# 𑁧𑁨𑁇𑁧𑁇𑁧 𑀲𑀻𑀮𑀯𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

### 𑀤𑁆𑀯𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀮𑀯𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(608.)

925. _“𑀲𑀻𑀮𑀫𑁂𑀯𑀺𑀥 𑀲𑀺𑀓𑁆𑀔𑁂𑀣,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀁;_  
_𑀲𑀻𑀮𑀜𑁆𑀳𑀺 𑀲𑀩𑁆𑀩𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁,_  
_𑀉𑀧𑀦𑀸𑀫𑁂𑀢𑀺 𑀲𑁂𑀯𑀺𑀢𑀁𑁇_  


(609.)

926. _𑀲𑀻𑀮𑀁 𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄 𑀢𑀬𑁄 𑀲𑀼𑀔𑁂;_  
_𑀧𑀲𑀁𑀲𑀁 𑀯𑀺𑀢𑁆𑀢𑀺𑀮𑀸𑀪𑀜𑁆𑀘,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑀕𑁆𑀕𑁂 𑀧𑀫𑁄𑀤𑀦𑀁𑁇_  


(610.)

927. _𑀲𑀻𑀮𑀯𑀸 𑀳𑀺 𑀩𑀳𑀽 𑀫𑀺𑀢𑁆𑀢𑁂,_  
_𑀲𑀜𑁆𑀜𑀫𑁂𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀦 𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀥𑀁𑀲𑀢𑁂 𑀧𑀸𑀧𑀫𑀸𑀘𑀭𑀁𑁇_  


(611.)

928. _𑀅𑀯𑀡𑁆𑀡𑀜𑁆𑀘 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀘,_  
_𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀮𑀪𑀢𑁂 𑀦𑀭𑁄;_  
_𑀯𑀡𑁆𑀡𑀁 𑀓𑀺𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑀁𑀲𑀜𑁆𑀘,_  
_𑀲𑀤𑀸 𑀮𑀪𑀢𑀺 𑀲𑀻𑀮𑀯𑀸𑁇_  


(612.)

929. _𑀆𑀤𑀺 𑀲𑀻𑀮𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀘,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀦𑀜𑁆𑀘 𑀫𑀸𑀢𑀼𑀓𑀁;_  
_𑀧𑀫𑀼𑀔𑀁 𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀲𑀻𑀮𑀁 𑀯𑀺𑀲𑁄𑀥𑀬𑁂𑁇_  


(613.)

930. _𑀯𑁂𑀮𑀸 𑀘 𑀲𑀁𑀯𑀭𑀁 𑀲𑀻𑀮𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀪𑀺𑀳𑀸𑀲𑀦𑀁;_  
_𑀢𑀺𑀢𑁆𑀣𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀲𑀻𑀮𑀁 𑀯𑀺𑀲𑁄𑀥𑀬𑁂𑁇_  


(614.)

931. _𑀲𑀻𑀮𑀁 𑀩𑀮𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀫𑀁,_  
_𑀲𑀻𑀮𑀁 𑀆𑀯𑀼𑀥𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀻𑀮𑀫𑀸𑀪𑀭𑀡𑀁 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀲𑀻𑀮𑀁 𑀓𑀯𑀘𑀫𑀩𑁆𑀪𑀼𑀢𑀁𑁇_  


(615.)

932. _𑀲𑀻𑀮𑀁 𑀲𑁂𑀢𑀼 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑁄,_  
_𑀲𑀻𑀮𑀁 𑀕𑀦𑁆𑀥𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀲𑀻𑀮𑀁 𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀬𑁂𑀦 𑀯𑀸𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁𑁇_  


(616.)

933. _𑀲𑀻𑀮𑀁 𑀲𑀫𑁆𑀩𑀮𑀫𑁂𑀯𑀕𑁆𑀕𑀁,_  
_𑀲𑀻𑀮𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀻𑀮𑀁 𑀲𑁂𑀝𑁆𑀞𑁄 𑀅𑀢𑀺𑀯𑀸𑀳𑁄,_  
_𑀬𑁂𑀦 𑀬𑀸𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁𑁇_  


(617.)

934. _𑀇𑀥𑁂𑀯 𑀦𑀺𑀦𑁆𑀤𑀁 𑀮𑀪𑀢𑀺,_  
_𑀧𑁂𑀘𑁆𑀘𑀸𑀧𑀸𑀬𑁂 𑀘 𑀤𑀼𑀫𑁆𑀫𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀩𑀸𑀮𑁄,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


(618.)

935. _𑀇𑀥𑁂𑀯 𑀓𑀺𑀢𑁆𑀢𑀺𑀁 𑀮𑀪𑀢𑀺,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑀕𑁆𑀕𑁂 𑀘 𑀲𑀼𑀫𑁆𑀫𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀫𑀦𑁄 𑀥𑀻𑀭𑁄,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


(619.)

936. _𑀲𑀻𑀮𑀫𑁂𑀯 𑀇𑀥 𑀅𑀕𑁆𑀕𑀁,_  
_𑀧𑀜𑁆𑀜𑀯𑀸 𑀧𑀦 𑀉𑀢𑁆𑀢𑀫𑁄;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀘 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀲𑀻𑀮𑀧𑀜𑁆𑀜𑀸𑀡𑀢𑁄 𑀚𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

937. …  𑀲𑀻𑀮𑀯𑁄 𑀣𑁂𑀭𑁄… 𑁇
