# 𑁧𑁪𑁇𑁧𑁇𑁧 𑀔𑀤𑀺𑀭𑀯𑀦𑀺𑀬𑀭𑁂𑀯𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

* 𑀘𑀼𑀤𑁆𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀔𑀤𑀺𑀭𑀯𑀦𑀺𑀬𑀭𑁂𑀯𑀢𑀢𑁆𑀣𑁂𑀭𑀕𑀸𑀣𑀸

(645.)

971\. _“𑀬𑀤𑀸 𑀅𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁,_  
_𑀅𑀦𑀭𑀺𑀬𑀁 𑀤𑁄𑀲𑀲𑀁𑀳𑀺𑀢𑀁𑁇_  


(646.)

972\. _‘𑀇𑀫𑁂 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀼 𑀯𑀚𑁆𑀛𑀦𑁆𑀢𑀼,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀧𑁆𑀧𑁄𑀦𑁆𑀢𑀼 𑀧𑀸𑀡𑀺𑀦𑁄’;_  
_𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀤𑀻𑀖𑀫𑀦𑁆𑀢𑀭𑁂𑁇_  


(647.)

973\. _𑀫𑁂𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁;_  
_𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀘𑀺𑀢𑀁,_  
_𑀬𑀣𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁𑁇_  


(648.)

974\. _𑀲𑀩𑁆𑀩𑀫𑀺𑀢𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀲𑀔𑁄,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄;_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀸𑀯𑁂𑀫𑀺,_  
_𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀭𑀢𑁄 𑀲𑀤𑀸𑁇_  


(649.)

975\. _𑀅𑀲𑀁𑀳𑀻𑀭𑀁 𑀅𑀲𑀁𑀓𑀼𑀧𑁆𑀧𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀆𑀫𑁄𑀤𑀬𑀸𑀫𑀳𑀁;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀺𑀳𑀸𑀭𑀁 𑀪𑀸𑀯𑁂𑀫𑀺,_  
_𑀅𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁂𑀯𑀺𑀢𑀁𑁇_  


(650.)

976\. _𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑁄;_  
_𑀅𑀭𑀺𑀬𑁂𑀦 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦,_  
_𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(651.)

977\. _𑀬𑀣𑀸𑀧𑀺 𑀧𑀩𑁆𑀩𑀢𑁄 𑀲𑁂𑀮𑁄,_  
_𑀅𑀘𑀮𑁄 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀩𑁆𑀩𑀢𑁄𑀯 𑀦 𑀯𑁂𑀥𑀢𑀺𑁇_  


(652.)

978\. _𑀅𑀦𑀗𑁆𑀕𑀡𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑀼𑀘𑀺𑀕𑀯𑁂𑀲𑀺𑀦𑁄;_  
_𑀯𑀸𑀮𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀁 𑀧𑀸𑀧𑀲𑁆𑀲,_  
_𑀅𑀩𑁆𑀪𑀫𑀢𑁆𑀢𑀁𑀯 𑀔𑀸𑀬𑀢𑀺𑁇_  


(653.)

979\. _𑀦𑀕𑀭𑀁 𑀬𑀣𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀕𑀼𑀢𑁆𑀢𑀁 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀁;_  
_𑀏𑀯𑀁 𑀕𑁄𑀧𑁂𑀣 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀔𑀡𑁄 𑀯𑁄 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸𑁇_  


(654.)

980\. _𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀫𑀭𑀡𑀁,_  
_𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀓𑀸𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀲𑀁 𑀪𑀢𑀓𑁄 𑀬𑀣𑀸𑁇_  


(655.)

981\. _𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺 𑀫𑀭𑀡𑀁,_  
_…𑀧𑁂…_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄𑁇_  


(656.)

982\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(657.)

983\. _𑀬𑀲𑁆𑀲 𑀘𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄𑁇_  


(658.)

984\. _𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀣𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦,_  
_𑀏𑀲𑀸 𑀫𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻;_  
_𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀲𑀩𑁆𑀩𑀥𑀻”𑀢𑀺𑁇_  


---

985\. …  𑀔𑀤𑀺𑀭𑀯𑀦𑀺𑀬𑀭𑁂𑀯𑀢𑁄 𑀣𑁂𑀭𑁄… 𑁇
