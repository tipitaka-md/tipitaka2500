# 𑁩𑁇𑁧 𑀓𑀓𑀘𑀽𑀧𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀓𑀓𑀘𑀽𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(222.)

746\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀲𑀘𑁂 𑀓𑁄𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀢𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀢𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑀘𑁂 𑀧𑀦 𑀓𑁄𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀢𑁂𑀦 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

747\. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀲𑁆𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀲𑀘𑁂 𑀓𑁄𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀢𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀢𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑀘𑁂 𑀧𑀦 𑀓𑁄𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀢𑁂𑀦 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(223.)

748\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀨𑀕𑁆𑀕𑀼𑀦, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀨𑀕𑁆𑀕𑀼𑀦, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

754\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀭𑀸𑀥𑀬𑀺𑀁𑀲𑀼 𑀯𑀢 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺𑀁—  𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀸𑀲𑀦𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺𑁇 𑀏𑀓𑀸𑀲𑀦𑀪𑁄𑀚𑀦𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑁄 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀢𑀜𑁆𑀘 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀢𑀜𑁆𑀘 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀩𑀮𑀜𑁆𑀘 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀜𑁆𑀘𑁇 𑀏𑀣 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀸𑀲𑀦𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀣𑁇 𑀏𑀓𑀸𑀲𑀦𑀪𑁄𑀚𑀦𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀢𑀜𑁆𑀘 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀣 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀢𑀜𑁆𑀘 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀩𑀮𑀜𑁆𑀘 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀜𑁆𑀘𑀸𑀢𑀺𑁇 𑀦 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀳𑁄𑀲𑀺; 𑀲𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀭𑀡𑀻𑀬𑀫𑁂𑀯 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀳𑁄𑀲𑀺𑁇

755\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀪𑀽𑀫𑀺𑀬𑀁 𑀘𑀢𑀼𑀫𑀳𑀸𑀧𑀣𑁂 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑁄 𑀬𑀼𑀢𑁆𑀢𑁄 𑀅𑀲𑁆𑀲 𑀞𑀺𑀢𑁄 𑀑𑀥𑀲𑁆𑀢𑀧𑀢𑁄𑀤𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀓𑁆𑀔𑁄 𑀬𑁄𑀕𑁆𑀕𑀸𑀘𑀭𑀺𑀬𑁄 𑀅𑀲𑁆𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸, 𑀯𑀸𑀫𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀭𑀲𑁆𑀫𑀺𑀬𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸, 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀢𑁄𑀤𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸, 𑀬𑁂𑀦𑀺𑀘𑁆𑀙𑀓𑀁 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁 𑀲𑀸𑀭𑁂𑀬𑁆𑀬𑀧𑀺 𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀭𑁂𑀬𑁆𑀬𑀧𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦 𑀫𑁂 𑀢𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀲𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀭𑀡𑀻𑀬𑀫𑁂𑀯 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀁 𑀧𑀚𑀳𑀣, 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀬𑁄𑀕𑀁 𑀓𑀭𑁄𑀣𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀣𑁇

756\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀸𑀫𑀲𑁆𑀲 𑀯𑀸 𑀦𑀺𑀕𑀫𑀲𑁆𑀲 𑀯𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀫𑀳𑀦𑁆𑀢𑀁 𑀲𑀸𑀮𑀯𑀦𑀁𑁇 𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀏𑀍𑀅𑀡𑁆𑀟𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑁄 𑀳𑀺𑀢𑀓𑀸𑀫𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑁄𑁇 𑀲𑁄 𑀬𑀸 𑀢𑀸 𑀲𑀸𑀮𑀮𑀝𑁆𑀞𑀺𑀬𑁄 𑀓𑀼𑀝𑀺𑀮𑀸 𑀑𑀚𑀸𑀧𑀳𑀭𑀡𑀺𑀬𑁄 𑀢𑀸 𑀙𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀦𑀻𑀳𑀭𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀢𑁄𑀯𑀦𑀁 𑀲𑀼𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀬𑁆𑀬𑁇 𑀬𑀸 𑀧𑀦 𑀢𑀸 𑀲𑀸𑀮𑀮𑀝𑁆𑀞𑀺𑀬𑁄 𑀉𑀚𑀼𑀓𑀸 𑀲𑀼𑀚𑀸𑀢𑀸 𑀢𑀸 𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀮𑀯𑀦𑀁 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀁 𑀧𑀚𑀳𑀣, 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀬𑁄𑀕𑀁 𑀓𑀭𑁄𑀣𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀣𑁇

757\. 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀺𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀦𑀸𑀫 𑀕𑀳𑀧𑀢𑀸𑀦𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀬𑀸 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀲𑁄𑀭𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻, 𑀦𑀺𑀯𑀸𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻, 𑀉𑀧𑀲𑀦𑁆𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻’𑀢𑀺𑁇 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸𑀬 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀬𑀸 𑀓𑀸𑀍𑀈 𑀦𑀸𑀫 𑀤𑀸𑀲𑀻 𑀅𑀳𑁄𑀲𑀺 𑀤𑀓𑁆𑀔𑀸 𑀅𑀦𑀮𑀲𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸𑁇

758\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀍𑀇𑀬𑀸 𑀤𑀸𑀲𑀺𑀬𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀫𑀬𑁆𑀳𑀁 𑀔𑁄 𑀅𑀬𑁆𑀬𑀸𑀬 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  “𑀲𑁄𑀭𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻, 𑀦𑀺𑀯𑀸𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻, 𑀉𑀧𑀲𑀦𑁆𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻”𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀦𑀼 𑀔𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺 𑀉𑀤𑀸𑀳𑀼 𑀅𑀲𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀳𑀼 𑀫𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀬𑁂𑀦 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺, 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀁? 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀅𑀬𑁆𑀬𑀁 𑀯𑀻𑀫𑀁𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀍𑀈 𑀤𑀸𑀲𑀻 𑀤𑀺𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻 𑀓𑀸𑀍𑀇𑀁 𑀤𑀸𑀲𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀳𑁂 𑀚𑁂 𑀓𑀸𑀍𑀈’𑀢𑀺𑁇 ‘𑀓𑀺𑀁, 𑀅𑀬𑁆𑀬𑁂’𑀢𑀺? ‘𑀓𑀺𑀁, 𑀚𑁂, 𑀤𑀺𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁆𑀯𑀬𑁆𑀬𑁂, 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺𑁇 ‘𑀦𑁄 𑀯𑀢 𑀭𑁂 𑀓𑀺𑀜𑁆𑀘𑀺, 𑀧𑀸𑀧𑀺 𑀤𑀸𑀲𑀺, 𑀤𑀺𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀪𑀸𑀓𑀼𑀝𑀺𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀍𑀇𑀬𑀸 𑀤𑀸𑀲𑀺𑀬𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺, 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀁; 𑀫𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸, 𑀬𑁂𑀦 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺, 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀬𑁆𑀬𑀁 𑀯𑀻𑀫𑀁𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇

759\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀍𑀈 𑀤𑀸𑀲𑀻 𑀤𑀺𑀯𑀸𑀢𑀭𑀁𑀬𑁂𑀯 𑀉𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻 𑀓𑀸𑀍𑀇𑀁 𑀤𑀸𑀲𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀳𑁂 𑀚𑁂 𑀓𑀸𑀍𑀈’𑀢𑀺𑁇 ‘𑀓𑀺𑀁, 𑀅𑀬𑁆𑀬𑁂’𑀢𑀺? ‘𑀓𑀺𑀁, 𑀚𑁂, 𑀤𑀺𑀯𑀸𑀢𑀭𑀁 𑀉𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁆𑀯𑀬𑁆𑀬𑁂, 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺𑁇 ‘𑀦𑁄 𑀯𑀢 𑀭𑁂 𑀓𑀺𑀜𑁆𑀘𑀺, 𑀧𑀸𑀧𑀺 𑀤𑀸𑀲𑀺, 𑀤𑀺𑀯𑀸𑀢𑀭𑀁 𑀉𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀯𑀸𑀘𑀁 𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀍𑀇𑀬𑀸 𑀤𑀸𑀲𑀺𑀬𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺, 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀁𑁇 𑀫𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸, 𑀬𑁂𑀦 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑁄𑀧𑀁 𑀦 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺, 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀬𑁆𑀬𑀁 𑀯𑀻𑀫𑀁𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇

760\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀍𑀈 𑀤𑀸𑀲𑀻 𑀤𑀺𑀯𑀸𑀢𑀭𑀁𑀬𑁂𑀯 𑀉𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻 𑀓𑀸𑀍𑀇𑀁 𑀤𑀸𑀲𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀳𑁂 𑀚𑁂 𑀓𑀸𑀍𑀈’𑀢𑀺𑁇 ‘𑀓𑀺𑀁, 𑀅𑀬𑁆𑀬𑁂’𑀢𑀺? ‘𑀓𑀺𑀁, 𑀚𑁂, 𑀤𑀺𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁆𑀯𑀬𑁆𑀬𑁂, 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺𑁇 ‘𑀦𑁄 𑀯𑀢 𑀭𑁂 𑀓𑀺𑀜𑁆𑀘𑀺, 𑀧𑀸𑀧𑀺 𑀤𑀸𑀲𑀺, 𑀤𑀺𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀕𑁆𑀕𑀍𑀅𑀲𑀽𑀘𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀻𑀲𑁂 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺, 𑀲𑀻𑀲𑀁 𑀯𑁄𑀪𑀺𑀦𑁆𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀍𑀈 𑀤𑀸𑀲𑀻 𑀪𑀺𑀦𑁆𑀦𑁂𑀦 𑀲𑀻𑀲𑁂𑀦 𑀮𑁄𑀳𑀺𑀢𑁂𑀦 𑀕𑀮𑀦𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀯𑀺𑀲𑁆𑀲𑀓𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀲𑀺—  ‘𑀧𑀲𑁆𑀲𑀣𑀬𑁆𑀬𑁂, 𑀲𑁄𑀭𑀢𑀸𑀬 𑀓𑀫𑁆𑀫𑀁; 𑀧𑀲𑁆𑀲𑀣𑀬𑁆𑀬𑁂, 𑀦𑀺𑀯𑀸𑀢𑀸𑀬 𑀓𑀫𑁆𑀫𑀁, 𑀧𑀲𑁆𑀲𑀣𑀬𑁆𑀬𑁂, 𑀉𑀧𑀲𑀦𑁆𑀢𑀸𑀬 𑀓𑀫𑁆𑀫𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀏𑀓𑀤𑀸𑀲𑀺𑀓𑀸𑀬 𑀤𑀺𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀲𑀻𑀢𑀺 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀕𑁆𑀕𑀍𑀅𑀲𑀽𑀘𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀻𑀲𑁂 𑀧𑀳𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀢𑀺, 𑀲𑀻𑀲𑀁 𑀯𑁄𑀪𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

761\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸𑀬 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀬𑀸 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀏𑀯𑀁 𑀧𑀸𑀧𑀓𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀺—  ‘𑀘𑀡𑁆𑀟𑀻 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻, 𑀅𑀦𑀺𑀯𑀸𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻, 𑀅𑀦𑀼𑀧𑀲𑀦𑁆𑀢𑀸 𑀯𑁂𑀤𑁂𑀳𑀺𑀓𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻’𑀢𑀺𑁇

762\. 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀯𑀤𑁂𑀯 𑀲𑁄𑀭𑀢𑀲𑁄𑀭𑀢𑁄 𑀳𑁄𑀢𑀺 𑀦𑀺𑀯𑀸𑀢𑀦𑀺𑀯𑀸𑀢𑁄 𑀳𑁄𑀢𑀺 𑀉𑀧𑀲𑀦𑁆𑀢𑀽𑀧𑀲𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀬𑀸𑀯 𑀦 𑀅𑀫𑀦𑀸𑀧𑀸 𑀯𑀘𑀦𑀧𑀣𑀸 𑀨𑀼𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀫𑀦𑀸𑀧𑀸 𑀯𑀘𑀦𑀧𑀣𑀸 𑀨𑀼𑀲𑀦𑁆𑀢𑀺, 𑀅𑀣 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀲𑁄𑀭𑀢𑁄’𑀢𑀺 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄, ‘𑀦𑀺𑀯𑀸𑀢𑁄’𑀢𑀺 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄, ‘𑀉𑀧𑀲𑀦𑁆𑀢𑁄’𑀢𑀺 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀦𑀸𑀳𑀁 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 ‘𑀲𑀼𑀯𑀘𑁄’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺 𑀬𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀳𑁂𑀢𑀼 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀦 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀧𑀘𑀸𑀬𑀫𑀸𑀦𑁄 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀫𑀳𑀁 ‘𑀲𑀼𑀯𑀘𑁄’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀧𑀘𑀸𑀬𑀫𑀸𑀦𑀸 𑀲𑀼𑀯𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫, 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇
