# 𑁩𑁇𑁪 𑀭𑀣𑀯𑀺𑀦𑀻𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀭𑀣𑀯𑀺𑀦𑀻𑀢𑀲𑀼𑀢𑁆𑀢

(252.)

855\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀚𑀸𑀢𑀺𑀪𑀽𑀫𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀢𑀺𑀪𑀽𑀫𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

856\. “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺𑀪𑀽𑀫𑀺𑀬𑀁 𑀚𑀸𑀢𑀺𑀪𑀽𑀫𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀏𑀯𑀁 𑀲𑀫𑁆𑀪𑀸𑀯𑀺𑀢𑁄—  ‘𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀧𑀯𑀺𑀯𑁂𑀓𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀲𑀁𑀲𑀕𑁆𑀕𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀑𑀯𑀸𑀤𑀓𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑁄 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀦𑁆’”𑀢𑀺? “𑀧𑀼𑀡𑁆𑀡𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀚𑀸𑀢𑀺𑀪𑀽𑀫𑀺𑀬𑀁 𑀚𑀸𑀢𑀺𑀪𑀽𑀫𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀏𑀯𑀁 𑀲𑀫𑁆𑀪𑀸𑀯𑀺𑀢𑁄—  ‘𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄…𑀧𑁂…  𑀑𑀯𑀸𑀤𑀓𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑁄 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀦𑁆’”𑀢𑀺𑁇

(253.)

857\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀮𑀸𑀪𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀼𑀡𑁆𑀡𑀲𑁆𑀲 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀼𑀡𑁆𑀡𑀲𑁆𑀲 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲, 𑀬𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀽 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀢𑁆𑀣𑀼 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀅𑀦𑀼𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀫𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀦𑁆𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀲𑀢𑁆𑀣𑀸 𑀅𑀩𑁆𑀪𑀦𑀼𑀫𑁄𑀤𑀢𑀺𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀫𑀬𑀫𑁆𑀧𑀺 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀧𑀼𑀡𑁆𑀡𑁂𑀦 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀲𑀺𑀬𑀸 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄”𑀢𑀺𑁇

(254.)

858\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄; 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂”𑀢𑀺𑁇

(255.)

859\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇

(256.)

860\. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀲𑁆𑀲 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀼𑀡𑁆𑀡𑀲𑁆𑀲 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀢𑁂𑀦 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇

861\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑁄 𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀆𑀤𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺 𑀲𑀻𑀲𑀸𑀦𑀼𑀮𑁄𑀓𑀻𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

862\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀧𑀼𑀡𑁆𑀡𑁂𑀦 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

(257.)

863\. “𑀪𑀕𑀯𑀢𑀺 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺?

864\. “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇

865\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺?

866\. “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀆𑀯𑀼𑀲𑁄”𑁇

867\. “𑀓𑀺𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺?

868\. “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀆𑀯𑀼𑀲𑁄”𑁇

869\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺?

870\. “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀆𑀯𑀼𑀲𑁄”𑁇

901\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄𑀦𑀸𑀫𑁄 𑀆𑀬𑀲𑁆𑀫𑀸, 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀚𑀸𑀦𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀧𑀼𑀡𑁆𑀡𑁄𑀢𑀺 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀸𑀫𑀁; 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀢𑀺 𑀘 𑀧𑀦 𑀫𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀚𑀸𑀦𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄𑁇 𑀬𑀣𑀸 𑀢𑀁 𑀲𑀼𑀢𑀯𑀢𑀸 𑀲𑀸𑀯𑀓𑁂𑀦 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦, 𑀏𑀯𑀫𑁂𑀯 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀧𑀼𑀡𑁆𑀡𑁂𑀦 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀕𑀫𑁆𑀪𑀻𑀭𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀳𑀸 𑀅𑀦𑀼𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀫𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀮𑀸𑀪𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁, 𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁, 𑀬𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀮𑀪𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀧𑀬𑀺𑀭𑀽𑀧𑀸𑀲𑀦𑀸𑀬𑁇 𑀘𑁂𑀮𑀡𑁆𑀟𑀼𑀓𑁂𑀦 𑀘𑁂𑀧𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀫𑀼𑀤𑁆𑀥𑀦𑀸 𑀧𑀭𑀺𑀳𑀭𑀦𑁆𑀢𑀸 𑀮𑀪𑁂𑀬𑁆𑀬𑀼𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀮𑀪𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀬𑀺𑀭𑀽𑀧𑀸𑀲𑀦𑀸𑀬, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀮𑀸𑀪𑀸 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀲𑀼𑀮𑀤𑁆𑀥𑀁, 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀮𑀸𑀪𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀲𑀼𑀮𑀤𑁆𑀥𑀁, 𑀬𑁂 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀮𑀪𑀸𑀫 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀮𑀪𑀸𑀫 𑀧𑀬𑀺𑀭𑀽𑀧𑀸𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

902\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀸𑀫𑁄 𑀆𑀬𑀲𑁆𑀫𑀸, 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀚𑀸𑀦𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄𑀢𑀺 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀸𑀫𑀁; 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀢𑀺 𑀘 𑀧𑀦 𑀫𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀚𑀸𑀦𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀢𑁆𑀣𑀼𑀓𑀧𑁆𑀧𑁂𑀦 𑀯𑀢 𑀓𑀺𑀭, 𑀪𑁄, 𑀲𑀸𑀯𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀫𑀦𑁆𑀢𑀬𑀫𑀸𑀦𑀸 𑀦 𑀚𑀸𑀦𑀺𑀫𑁆𑀳—  ‘𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀳𑀺 𑀫𑀬𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫 ‘𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄’𑀢𑀺, 𑀏𑀢𑁆𑀢𑀓𑀫𑁆𑀧𑀺 𑀦𑁄 𑀦𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄𑁇 𑀬𑀣𑀸 𑀢𑀁 𑀲𑀼𑀢𑀯𑀢𑀸 𑀲𑀸𑀯𑀓𑁂𑀦 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦, 𑀏𑀯𑀫𑁂𑀯 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀕𑀫𑁆𑀪𑀻𑀭𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀳𑀸 𑀅𑀦𑀼𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀫𑀲𑁆𑀲 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸𑁇 𑀮𑀸𑀪𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁, 𑀬𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀮𑀪𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀧𑀬𑀺𑀭𑀽𑀧𑀸𑀲𑀦𑀸𑀬𑁇 𑀘𑁂𑀮𑀡𑁆𑀟𑀼𑀓𑁂𑀦 𑀘𑁂𑀧𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀫𑀼𑀤𑁆𑀥𑀦𑀸 𑀧𑀭𑀺𑀳𑀭𑀦𑁆𑀢𑀸 𑀮𑀪𑁂𑀬𑁆𑀬𑀼𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀮𑀪𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀬𑀺𑀭𑀽𑀧𑀸𑀲𑀦𑀸𑀬, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀮𑀸𑀪𑀸 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀲𑀼𑀮𑀤𑁆𑀥𑀁, 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀮𑀸𑀪𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀲𑀼𑀮𑀤𑁆𑀥𑀁, 𑀬𑁂 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀮𑀪𑀸𑀫 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀮𑀪𑀸𑀫 𑀧𑀬𑀺𑀭𑀽𑀧𑀸𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

903\. 𑀇𑀢𑀺𑀳 𑀢𑁂 𑀉𑀪𑁄𑀧𑀺 𑀫𑀳𑀸𑀦𑀸𑀕𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀲𑀫𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀽𑀢𑀺𑁇
