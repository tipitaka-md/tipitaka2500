# 𑁩𑁇𑁯 𑀫𑀳𑀸𑀲𑀸𑀭𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀲𑀸𑀭𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(307.)

1077\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

1078\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀑𑀢𑀺𑀡𑁆𑀡𑁄𑀫𑁆𑀳𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀤𑀼𑀓𑁆𑀔𑁄𑀢𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀧𑀭𑁂𑀢𑁄, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀇𑀫𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑁆𑀓𑀁𑀲𑁂𑀢𑀺 𑀧𑀭𑀁 𑀯𑀫𑁆𑀪𑁂𑀢𑀺—  ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀯𑀸, 𑀇𑀫𑁂 𑀧𑀦𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀢𑀸 𑀅𑀧𑁆𑀧𑁂𑀲𑀓𑁆𑀔𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

1079\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬 ‘𑀲𑀸𑀭𑀦𑁆’𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦 𑀯𑀢𑀸𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀭𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀨𑁂𑀕𑁆𑀕𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀢𑀘𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁𑁇 𑀢𑀣𑀸 𑀳𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 “𑀲𑀸𑀭𑀦𑁆”𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀭𑁂𑀦 𑀲𑀸𑀭𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀦𑀸𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀑𑀢𑀺𑀡𑁆𑀡𑁄𑀫𑁆𑀳𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀤𑀼𑀓𑁆𑀔𑁄𑀢𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀧𑀭𑁂𑀢𑁄, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀇𑀫𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑁆𑀓𑀁𑀲𑁂𑀢𑀺, 𑀧𑀭𑀁 𑀯𑀫𑁆𑀪𑁂𑀢𑀺 ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀯𑀸, 𑀇𑀫𑁂 𑀧𑀦𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀢𑀸 𑀅𑀧𑁆𑀧𑁂𑀲𑀓𑁆𑀔𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲; 𑀢𑁂𑀦 𑀘 𑀯𑁄𑀲𑀸𑀦𑀁 𑀆𑀧𑀸𑀤𑀺𑁇

1081\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁, 𑀧𑀧𑀝𑀺𑀓𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬 ‘𑀲𑀸𑀭𑀦𑁆’𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦 𑀯𑀢𑀸𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀭𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀨𑁂𑀕𑁆𑀕𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀢𑀘𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁𑁇 𑀢𑀣𑀸 𑀳𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁, 𑀧𑀧𑀝𑀺𑀓𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 “𑀲𑀸𑀭𑀦𑁆”𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄; 𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀭𑁂𑀦 𑀲𑀸𑀭𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀦𑀸𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

1090\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀲𑀸𑀭𑀜𑁆𑀜𑁂𑀯 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬 ‘𑀲𑀸𑀭𑀦𑁆’𑀢𑀺 𑀚𑀸𑀦𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀯𑀢𑀸𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀁, 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀨𑁂𑀕𑁆𑀕𑀼𑀁, 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀢𑀘𑀁, 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁𑁇 𑀢𑀣𑀸 𑀳𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀲𑀸𑀭𑀜𑁆𑀜𑁂𑀯 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 “𑀲𑀸𑀭𑀦𑁆”𑀢𑀺 𑀚𑀸𑀦𑀫𑀸𑀦𑁄𑁇 𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀭𑁂𑀦 𑀲𑀸𑀭𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

1091\. 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀑𑀢𑀺𑀡𑁆𑀡𑁄𑀫𑁆𑀳𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀤𑀼𑀓𑁆𑀔𑁄𑀢𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀧𑀭𑁂𑀢𑁄, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀇𑀫𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀦 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑁆𑀓𑀁𑀲𑁂𑀢𑀺, 𑀦 𑀧𑀭𑀁 𑀯𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑁆𑀓𑀁𑀲𑁂𑀢𑀺, 𑀦 𑀧𑀭𑀁 𑀯𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑁆𑀓𑀁𑀲𑁂𑀢𑀺, 𑀦 𑀧𑀭𑀁 𑀯𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑁆𑀓𑀁𑀲𑁂𑀢𑀺, 𑀦 𑀧𑀭𑀁 𑀯𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀅𑀲𑀫𑀬𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺𑁇 𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀬 𑀅𑀲𑀫𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑁂𑀣𑁇

1092\. 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀸𑀦𑀺𑀲𑀁𑀲𑀁, 𑀦 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀺𑀲𑀁𑀲𑀁, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀺𑀲𑀁𑀲𑀁, 𑀦 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀁𑁇 𑀬𑀸 𑀘 𑀔𑁄 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀏𑀢𑀤𑀢𑁆𑀣𑀫𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀏𑀢𑀁 𑀲𑀸𑀭𑀁 𑀏𑀢𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

1093\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇
