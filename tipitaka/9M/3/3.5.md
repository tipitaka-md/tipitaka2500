# 𑁩𑁇𑁫 𑀦𑀺𑀯𑀸𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀯𑀸𑀧𑀲𑀼𑀢𑁆𑀢

(261.)

905\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

906\. “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑁄 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀧𑀢𑀺 𑀫𑀺𑀕𑀚𑀸𑀢𑀸𑀦𑀁—  ‘𑀇𑀫𑀁 𑀫𑁂 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀘𑀺𑀭𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀽’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑁄 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀧𑀢𑀺 𑀫𑀺𑀕𑀚𑀸𑀢𑀸𑀦𑀁—  ‘𑀇𑀫𑀁 𑀫𑁂 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀯𑀸𑀧𑁂’𑀢𑀺𑁇

(262.)

907\. 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀸 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀅𑀫𑀼𑀁 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼, 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀸 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇

(263.)

908\. 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑀸 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀏𑀯𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁—  ‘𑀬𑁂 𑀔𑁄 𑀢𑁂 𑀧𑀞𑀫𑀸 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀅𑀫𑀼𑀁 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂 𑀧𑀞𑀫𑀸 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑀺𑀯𑀸𑀧𑀪𑁄𑀚𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀫𑁂𑀬𑁆𑀬𑀸𑀫, 𑀪𑀬𑀪𑁄𑀕𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑀺𑀯𑀸𑀧𑀪𑁄𑀚𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀫𑀺𑀁𑀲𑀼, 𑀪𑀬𑀪𑁄𑀕𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂, 𑀢𑀺𑀡𑁄𑀤𑀓𑀲𑀗𑁆𑀔𑀬𑁂, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀓𑀲𑀺𑀫𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑁄 𑀓𑀸𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀓𑀲𑀺𑀫𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀓𑀸𑀬𑀸𑀦𑀁 𑀩𑀮𑀯𑀻𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀳𑀸𑀬𑀺𑁇 𑀩𑀮𑀯𑀻𑀭𑀺𑀬𑁂 𑀧𑀭𑀺𑀳𑀻𑀦𑁂 𑀢𑀫𑁂𑀯 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑀸𑀧𑀺 𑀫𑀺𑀕𑀚𑀸𑀢𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀦𑁂𑀯𑀸𑀧𑀺𑀓𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇

917\. 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁—  ‘𑀬𑁂 𑀔𑁄 𑀢𑁂 𑀧𑀞𑀫𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀫𑀼𑀁 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀫𑀽𑀦𑀺 𑀘 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀸𑀦𑀺…𑀧𑁂… 𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂 𑀧𑀞𑀫𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀫𑀸𑀭𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂 𑀤𑀼𑀢𑀺𑀬𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁—  “𑀬𑁂 𑀔𑁄 𑀢𑁂 𑀧𑀞𑀫𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀫𑀼𑀁 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀫𑀽𑀦𑀺 𑀘 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀸𑀦𑀺…𑀧𑁂… 𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂 𑀧𑀞𑀫𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀫𑀸𑀭𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑀺𑀯𑀸𑀧𑀪𑁄𑀚𑀦𑀸 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀫𑁂𑀬𑁆𑀬𑀸𑀫, 𑀪𑀬𑀪𑁄𑀕𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑀺𑀯𑀸𑀧𑀪𑁄𑀚𑀦𑀸 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀫𑀺𑀁𑀲𑀼𑁇 𑀪𑀬𑀪𑁄𑀕𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀲𑀸𑀓𑀪𑀓𑁆𑀔𑀸𑀧𑀺 𑀅𑀳𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀧𑀯𑀢𑁆𑀢𑀨𑀮𑀪𑁄𑀚𑀻𑁇 𑀢𑁂𑀲𑀁 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀢𑀺𑀡𑁄𑀤𑀓𑀲𑀗𑁆𑀔𑀬𑁂 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀓𑀲𑀺𑀫𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑁄 𑀓𑀸𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀓𑀲𑀺𑀫𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀓𑀸𑀬𑀸𑀦𑀁 𑀩𑀮𑀯𑀻𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀳𑀸𑀬𑀺, 𑀩𑀮𑀯𑀻𑀭𑀺𑀬𑁂 𑀧𑀭𑀺𑀳𑀻𑀦𑁂 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀧𑀭𑀺𑀳𑀸𑀬𑀺, 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀳𑀻𑀦𑀸𑀬 𑀢𑀫𑁂𑀯 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀺𑀁𑀲𑀼 𑀢𑀸𑀦𑀺 𑀘 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀸𑀦𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀯𑀸𑀧𑁂 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑁂𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂 𑀤𑀼𑀢𑀺𑀬𑀸𑀧𑀺 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀫𑀸𑀭𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀅𑀫𑀼𑀁 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀫𑀽𑀦𑀺 𑀘 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀸𑀦𑀺 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀆𑀲𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀸𑀫, 𑀢𑀢𑁆𑀭𑀸𑀲𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀫𑀼𑀁 𑀦𑀺𑀯𑀸𑀧𑀁 𑀦𑀺𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀫𑀽𑀦𑀺 𑀘 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀸𑀦𑀺 𑀅𑀦𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀦𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀦 𑀫𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀦 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀦 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀯𑀸𑀧𑁂 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑁂’𑀢𑀺𑁇

922\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀕𑀢𑀺 𑀫𑀸𑀭𑀲𑁆𑀲 𑀘 𑀫𑀸𑀭𑀧𑀭𑀺𑀲𑀸𑀬 𑀘? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁆𑀥𑀫𑀓𑀸𑀲𑀺 𑀫𑀸𑀭𑀁, 𑀅𑀧𑀤𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀕𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

924\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

925\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

926\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

927\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

928\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

929\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

930\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀲𑁆𑀲 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁆𑀥𑀫𑀓𑀸𑀲𑀺 𑀫𑀸𑀭𑀁, 𑀅𑀧𑀤𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀕𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑁄 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀦𑁆”𑀢𑀺𑁇

931\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---
