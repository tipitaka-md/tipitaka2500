# 𑁩𑁇𑁨 𑀅𑀮𑀕𑀤𑁆𑀤𑀽𑀧𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀅𑀮𑀕𑀤𑁆𑀤𑀽𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(234.)

772\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀭𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺—  “𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀅𑀭𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀺𑀭 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺𑁇

773\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺𑁇 “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸”𑀢𑀺𑁇

774\. 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀦𑁆𑀢𑀺 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀺—  “𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀅𑀯𑀘, 𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺; 𑀦 𑀳𑀺 𑀲𑀸𑀥𑀼 𑀪𑀕𑀯𑀢𑁄 𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁, 𑀦 𑀳𑀺 𑀪𑀕𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸, 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀗𑁆𑀓𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀫𑀁𑀲𑀧𑁂𑀲𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀢𑀺𑀡𑀼𑀓𑁆𑀓𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀲𑀼𑀧𑀺𑀦𑀓𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀬𑀸𑀘𑀺𑀢𑀓𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀭𑀼𑀓𑁆𑀔𑀨𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀅𑀲𑀺𑀲𑀽𑀦𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀲𑀢𑁆𑀢𑀺𑀲𑀽𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…  𑀲𑀧𑁆𑀧𑀲𑀺𑀭𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄”𑀢𑀺𑁇

775\. 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑁄 𑀢𑀤𑁂𑀯 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀣𑀸𑀫𑀲𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀯𑁄𑀳𑀭𑀢𑀺—  “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸”𑀢𑀺𑁇

(235.)

776\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀁, 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀭𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀼𑀫𑁆𑀳 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂—  ‘𑀅𑀭𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀺𑀭 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’𑀢𑀺𑁇

777\. 𑀅𑀣 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀫𑁆𑀳; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀫𑁆𑀳—  ‘𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’𑀢𑀺?

778\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀅𑀫𑁆𑀳𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀫𑁆𑀳 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀫𑁆𑀳 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀫𑁆𑀳—  ‘𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀅𑀯𑀘, 𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺; 𑀦 𑀳𑀺 𑀲𑀸𑀥𑀼 𑀪𑀕𑀯𑀢𑁄 𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁, 𑀦 𑀳𑀺 𑀪𑀕𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸, 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀗𑁆𑀓𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…𑀧𑁂…  𑀲𑀧𑁆𑀧𑀲𑀺𑀭𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄’𑀢𑀺𑁇

779\. 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑁄 𑀢𑀤𑁂𑀯 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀣𑀸𑀫𑀲𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀯𑁄𑀳𑀭𑀢𑀺—  ‘𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀬𑀢𑁄 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀫𑁆𑀳 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀁, 𑀅𑀣 𑀫𑀬𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀫𑀸”𑀢𑀺𑁇

780\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸, 𑀬𑁂𑀦 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺? “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺𑁇

781\. “𑀓𑀲𑁆𑀲 𑀔𑁄 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀬𑀸 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺? 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸? 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸, 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀗𑁆𑀓𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀫𑀁𑀲𑀧𑁂𑀲𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀢𑀺𑀡𑀼𑀓𑁆𑀓𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀲𑀼𑀧𑀺𑀦𑀓𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀬𑀸𑀘𑀺𑀢𑀓𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀭𑀼𑀓𑁆𑀔𑀨𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀅𑀲𑀺𑀲𑀽𑀦𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀲𑀢𑁆𑀢𑀺𑀲𑀽𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…  𑀲𑀧𑁆𑀧𑀲𑀺𑀭𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸, 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀅𑀫𑁆𑀳𑁂 𑀘𑁂𑀯 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀲𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑀲𑀺, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀲𑀺𑁇 𑀢𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

782\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀸𑀬𑀁 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀉𑀲𑁆𑀫𑀻𑀓𑀢𑁄𑀧𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂”𑀢𑀺? “𑀓𑀺𑀜𑁆𑀳𑀺 𑀲𑀺𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂; 𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑁄 𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑁄 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀥𑁄𑀫𑀼𑀔𑁄 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀁 𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑀁 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀅𑀥𑁄𑀫𑀼𑀔𑀁 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀏𑀢𑁂𑀦 𑀲𑀓𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑁂𑀦𑁇 𑀇𑀥𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

783\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀣 𑀬𑀣𑀸𑀬𑀁 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀅𑀫𑁆𑀳𑁂 𑀘𑁂𑀯 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀢𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑀢𑀺, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀻”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀳𑀺 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸; 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀗𑁆𑀓𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸…𑀧𑁂…  𑀲𑀧𑁆𑀧𑀲𑀺𑀭𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄”𑀢𑀺𑁇

784\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀥𑀼, 𑀔𑁄 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀣𑁇 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸, 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇

785\. 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸, 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀗𑁆𑀓𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…𑀧𑁂…  𑀲𑀧𑁆𑀧𑀲𑀺𑀭𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸, 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀣 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀅𑀫𑁆𑀳𑁂 𑀘𑁂𑀯 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀢𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑀢𑀺, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀢𑀜𑁆𑀳𑀺 𑀢𑀲𑁆𑀲 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀲𑁄 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺 𑀓𑀸𑀫𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

794\. 𑀲𑀼𑀢𑀯𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑁄𑀯𑀺𑀤𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑁄, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑁄𑀯𑀺𑀤𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑁄, 𑀭𑀽𑀧𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺; 𑀯𑁂𑀤𑀦𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺; 𑀲𑀜𑁆𑀜𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺; 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺; 𑀬𑀫𑁆𑀧𑀺 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁, 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸, 𑀢𑀫𑁆𑀧𑀺 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺; 𑀬𑀫𑁆𑀧𑀺 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀝𑁆𑀞𑀸𑀦𑀁—  𑀲𑁄 𑀮𑁄𑀓𑁄 𑀲𑁄 𑀅𑀢𑁆𑀢𑀸, 𑀲𑁄 𑀧𑁂𑀘𑁆𑀘 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀺𑀘𑁆𑀘𑁄 𑀥𑀼𑀯𑁄 𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄, 𑀲𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀁 𑀢𑀣𑁂𑀯 𑀞𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺—  𑀢𑀫𑁆𑀧𑀺 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀅𑀲𑀢𑀺 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

819\. “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀭𑀽𑀧𑀁 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸, 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀯𑀸 𑀲𑀼𑀔𑀼𑀫𑀁 𑀯𑀸, 𑀳𑀻𑀦𑀁 𑀯𑀸 𑀧𑀡𑀻𑀢𑀁 𑀯𑀸, 𑀬𑀁 𑀤𑀽𑀭𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺—  𑀏𑀯𑀫𑁂𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀬𑀸 𑀓𑀸𑀘𑀺 𑀲𑀜𑁆𑀜𑀸…  𑀬𑁂 𑀓𑁂𑀘𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸, 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀯𑀸 𑀲𑀼𑀔𑀼𑀫𑀁 𑀯𑀸, 𑀳𑀻𑀦𑀁 𑀯𑀸 𑀧𑀡𑀻𑀢𑀁 𑀯𑀸, 𑀬𑀁 𑀤𑀽𑀭𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺—  𑀏𑀯𑀫𑁂𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇

(245.)

820\. 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀚𑁆𑀚𑀢𑀺, 𑀯𑀺𑀭𑀸𑀕𑀸 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀧𑀮𑀺𑀖𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀲𑀁𑀓𑀺𑀡𑁆𑀡𑀧𑀭𑀺𑀓𑁆𑀔𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀅𑀩𑁆𑀩𑀽𑀍𑀳𑁂𑀲𑀺𑀓𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀦𑀺𑀭𑀕𑁆𑀕𑀍𑀑 𑀇𑀢𑀺𑀧𑀺, 𑀅𑀭𑀺𑀬𑁄 𑀧𑀦𑁆𑀦𑀤𑁆𑀥𑀚𑁄 𑀧𑀦𑁆𑀦𑀪𑀸𑀭𑁄 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀇𑀢𑀺𑀧𑀺𑁇

821\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀧𑀮𑀺𑀖𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸, 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀧𑀮𑀺𑀖𑁄 𑀳𑁄𑀢𑀺𑁇

822\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀓𑀺𑀡𑁆𑀡𑀧𑀭𑀺𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑁄𑀦𑁄𑀩𑁆𑀪𑀯𑀺𑀓𑁄 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄, 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀓𑀺𑀡𑁆𑀡𑀧𑀭𑀺𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇

823\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀩𑁆𑀩𑀽𑀍𑀳𑁂𑀲𑀺𑀓𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸, 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀩𑁆𑀩𑀽𑀍𑀳𑁂𑀲𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

824\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀭𑀕𑁆𑀕𑀍𑀑 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀜𑁆𑀘 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸𑀦𑀺 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸𑀦𑀺, 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑀦𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀭𑀕𑁆𑀕𑀍𑀑 𑀳𑁄𑀢𑀺𑁇

825\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀺𑀬𑁄 𑀧𑀦𑁆𑀦𑀤𑁆𑀥𑀚𑁄 𑀧𑀦𑁆𑀦𑀪𑀸𑀭𑁄 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀲𑁆𑀫𑀺𑀫𑀸𑀦𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄, 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀺𑀬𑁄 𑀧𑀦𑁆𑀦𑀤𑁆𑀥𑀚𑁄 𑀧𑀦𑁆𑀦𑀪𑀸𑀭𑁄 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

826\. 𑀏𑀯𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑁃𑀦𑁆𑀤𑀸 𑀤𑁂𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀸 𑀅𑀦𑁆𑀯𑁂𑀲𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  ‘𑀇𑀤𑀁 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀤𑀺𑀝𑁆𑀞𑁂𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀦𑀦𑀼𑀯𑀺𑀚𑁆𑀚𑁄𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀁 𑀔𑁄 𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀫𑀓𑁆𑀔𑀸𑀬𑀺𑀁 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀲𑀢𑀸 𑀢𑀼𑀘𑁆𑀙𑀸 𑀫𑀼𑀲𑀸 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀯𑁂𑀦𑀬𑀺𑀓𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄, 𑀲𑀢𑁄 𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀘𑁆𑀙𑁂𑀤𑀁 𑀯𑀺𑀦𑀸𑀲𑀁 𑀯𑀺𑀪𑀯𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀻’𑀢𑀺𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳𑀁 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀣𑀸 𑀘𑀸𑀳𑀁 𑀦 𑀯𑀤𑀸𑀫𑀺, 𑀢𑀣𑀸 𑀫𑀁 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀲𑀢𑀸 𑀢𑀼𑀘𑁆𑀙𑀸 𑀫𑀼𑀲𑀸 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀯𑁂𑀦𑀬𑀺𑀓𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄, 𑀲𑀢𑁄 𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀘𑁆𑀙𑁂𑀤𑀁 𑀯𑀺𑀦𑀸𑀲𑀁 𑀯𑀺𑀪𑀯𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀻’𑀢𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀘 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂𑀯 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀫𑀺, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀦𑀺𑀭𑁄𑀥𑀁𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑁂𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀦 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀦 𑀘𑁂𑀢𑀲𑁄 𑀅𑀦𑀪𑀺𑀭𑀤𑁆𑀥𑀺𑁇

(248.)

829\. 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀯𑀺𑀯𑀝𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁄𑁇 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀢𑁆𑀢𑀸𑀦𑁂 𑀯𑀺𑀯𑀝𑁂 𑀧𑀓𑀸𑀲𑀺𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁂 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀯𑀼𑀲𑀺𑀢𑀯𑀦𑁆𑀢𑁄 𑀓𑀢𑀓𑀭𑀡𑀻𑀬𑀸 𑀑𑀳𑀺𑀢𑀪𑀸𑀭𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀤𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀪𑀯𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸, 𑀯𑀝𑁆𑀝𑀁 𑀢𑁂𑀲𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀜𑀸𑀧𑀦𑀸𑀬𑁇

830\. 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀯𑀺𑀯𑀝𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁄𑁇 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀢𑁆𑀢𑀸𑀦𑁂 𑀯𑀺𑀯𑀝𑁂 𑀧𑀓𑀸𑀲𑀺𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁂 𑀬𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀜𑁆𑀘𑁄𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀦𑁄, 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀸 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇

831\. 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀯𑀺𑀯𑀝𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁄𑁇 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀢𑁆𑀢𑀸𑀦𑁂 𑀯𑀺𑀯𑀝𑁂 𑀧𑀓𑀸𑀲𑀺𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁂 𑀬𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀻𑀡𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀸 𑀢𑀦𑀼𑀪𑀽𑀢𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄, 𑀲𑀓𑀺𑀤𑁂𑀯 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

832\. 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀯𑀺𑀯𑀝𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁄𑁇 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀢𑁆𑀢𑀸𑀦𑁂 𑀯𑀺𑀯𑀝𑁂 𑀧𑀓𑀸𑀲𑀺𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁂 𑀬𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀻𑀡𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸, 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸, 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑀸𑁇

833\. 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀯𑀺𑀯𑀝𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁄𑁇 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀢𑁆𑀢𑀸𑀦𑁂 𑀯𑀺𑀯𑀝𑁂 𑀧𑀓𑀸𑀲𑀺𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁂 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄 𑀲𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑀸𑁇

834\. 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀯𑀺𑀯𑀝𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁄𑁇 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀢𑁆𑀢𑀸𑀦𑁂 𑀯𑀺𑀯𑀝𑁂 𑀧𑀓𑀸𑀲𑀺𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑁂 𑀬𑁂𑀲𑀁 𑀫𑀬𑀺 𑀲𑀤𑁆𑀥𑀸𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀫𑀢𑁆𑀢𑀁 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀲𑀕𑁆𑀕𑀧𑀭𑀸𑀬𑀦𑀸”𑀢𑀺𑁇

835\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---
