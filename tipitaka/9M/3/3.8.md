# 𑁩𑁇𑁮 𑀫𑀳𑀸𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(300.)

1053\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀯𑀼𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺𑁇 “𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

1054\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀚𑀗𑁆𑀕𑀮𑀸𑀦𑀁 𑀧𑀸𑀡𑀸𑀦𑀁 𑀧𑀤𑀚𑀸𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑁂 𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑀁 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀦𑁆𑀢𑀢𑁆𑀢𑁂𑀦; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑁂𑀢𑁂 𑀘𑀢𑀽𑀲𑀼 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑁂𑀲𑀼 𑀲𑀗𑁆𑀕𑀳𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑁂, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑁂, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑁂, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑁂𑁇

(301.)

1055\. 𑀓𑀢𑀫𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁? 𑀚𑀸𑀢𑀺𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀚𑀭𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁; 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦, 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸? 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

1056\. 𑀓𑀢𑀫𑁄 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀺, 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀭𑀽𑀧𑀁𑁇

1057\. 𑀓𑀢𑀫𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀪𑀽𑀢𑀸? 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼, 𑀆𑀧𑁄𑀥𑀸𑀢𑀼, 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑁇

(302.)

1058\. 𑀓𑀢𑀫𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼? 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀔𑀭𑀺𑀕𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀦𑀔𑀸 𑀤𑀦𑁆𑀢𑀸 𑀢𑀘𑁄 𑀫𑀁𑀲𑀁 𑀦𑁆𑀳𑀸𑀭𑀼 𑀅𑀝𑁆𑀞𑀺 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀯𑀓𑁆𑀓𑀁 𑀳𑀤𑀬𑀁 𑀬𑀓𑀦𑀁 𑀓𑀺𑀮𑁄𑀫𑀓𑀁 𑀧𑀺𑀳𑀓𑀁 𑀧𑀧𑁆𑀨𑀸𑀲𑀁 𑀅𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀕𑀼𑀡𑀁 𑀉𑀤𑀭𑀺𑀬𑀁 𑀓𑀭𑀻𑀲𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀔𑀭𑀺𑀕𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑁇 𑀬𑀸 𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼, 𑀬𑀸 𑀘 𑀩𑀸𑀳𑀺𑀭𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀭𑁂𑀯𑁂𑀲𑀸𑁇 ‘𑀢𑀁 𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺—  𑀏𑀯𑀫𑁂𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑀸𑀚𑁂𑀢𑀺𑁇

1059\. 𑀳𑁄𑀢𑀺 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑀬𑁄 𑀬𑀁 𑀩𑀸𑀳𑀺𑀭𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀧𑀓𑀼𑀧𑁆𑀧𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀩𑀸𑀳𑀺𑀭𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀳𑀺 𑀦𑀸𑀫, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑀸𑀳𑀺𑀭𑀸𑀬 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀸𑀯 𑀫𑀳𑀮𑁆𑀮𑀺𑀓𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀔𑀬𑀥𑀫𑁆𑀫𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀺𑀁 𑀧𑀦𑀺𑀫𑀲𑁆𑀲 𑀫𑀢𑁆𑀢𑀝𑁆𑀞𑀓𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀢𑀡𑁆𑀳𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 ‘𑀅𑀳𑀦𑁆𑀢𑀺 𑀯𑀸 𑀫𑀫𑀦𑁆𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀫𑀻’𑀢𑀺 𑀯𑀸? 𑀅𑀣 𑀔𑁆𑀯𑀸𑀲𑁆𑀲 𑀦𑁄𑀢𑁂𑀯𑁂𑀢𑁆𑀣 𑀳𑁄𑀢𑀺𑁇

1060\. 𑀢𑀜𑁆𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀭𑁂 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀸𑁇 𑀲𑀸 𑀘 𑀔𑁄 𑀧𑀝𑀺𑀘𑁆𑀘, 𑀦𑁄 𑀅𑀧𑀝𑀺𑀘𑁆𑀘𑁇 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘? 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘’𑁇 𑀲𑁄 𑀨𑀲𑁆𑀲𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀥𑀸𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀫𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1061\. 𑀢𑀜𑁆𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀭𑁂 𑀅𑀦𑀺𑀝𑁆𑀞𑁂𑀳𑀺 𑀅𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀅𑀫𑀦𑀸𑀧𑁂𑀳𑀺 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺—  𑀧𑀸𑀡𑀺𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺 𑀮𑁂𑀟𑁆𑀟𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺 𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺 𑀲𑀢𑁆𑀣𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀢𑀣𑀸𑀪𑀽𑀢𑁄 𑀔𑁄 𑀅𑀬𑀁 𑀓𑀸𑀬𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀧𑀸𑀡𑀺𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺, 𑀮𑁂𑀟𑁆𑀟𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺, 𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀣𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀓𑀓𑀘𑀽𑀧𑀫𑁄𑀯𑀸𑀤𑁂—  “𑀉𑀪𑀢𑁄𑀤𑀡𑁆𑀟𑀓𑁂𑀦 𑀘𑁂𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀓𑀘𑁂𑀦 𑀘𑁄𑀭𑀸 𑀑𑀘𑀭𑀓𑀸 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀑𑀓𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀬𑁄 𑀫𑀦𑁄 𑀧𑀤𑀽𑀲𑁂𑀬𑁆𑀬 𑀦 𑀫𑁂 𑀲𑁄 𑀢𑁂𑀦 𑀲𑀸𑀲𑀦𑀓𑀭𑁄”𑀢𑀺𑁇 𑀆𑀭𑀤𑁆𑀥𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀯𑀻𑀭𑀺𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑀮𑁆𑀮𑀻𑀦𑀁, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀢𑀺 𑀅𑀲𑀫𑁆𑀫𑀼𑀝𑁆𑀞𑀸, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀓𑀸𑀬𑁄 𑀅𑀲𑀸𑀭𑀤𑁆𑀥𑁄, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑀕𑁆𑀕𑀁𑁇 𑀓𑀸𑀫𑀁 𑀤𑀸𑀦𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀧𑀸𑀡𑀺𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼, 𑀮𑁂𑀟𑁆𑀟𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼, 𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼, 𑀲𑀢𑁆𑀣𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼, 𑀓𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀤𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀦𑁆’𑀢𑀺𑁇

1068\. 𑀳𑁄𑀢𑀺 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑀬𑁄 𑀬𑀁 𑀩𑀸𑀳𑀺𑀭𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼 𑀧𑀓𑀼𑀧𑁆𑀧𑀢𑀺𑁇 𑀲𑀸 𑀕𑀸𑀫𑀫𑁆𑀧𑀺 𑀯𑀳𑀢𑀺, 𑀦𑀺𑀕𑀫𑀫𑁆𑀧𑀺 𑀯𑀳𑀢𑀺, 𑀦𑀕𑀭𑀫𑁆𑀧𑀺 𑀯𑀳𑀢𑀺, 𑀚𑀦𑀧𑀤𑀫𑁆𑀧𑀺 𑀯𑀳𑀢𑀺, 𑀚𑀦𑀧𑀤𑀧𑀤𑁂𑀲𑀫𑁆𑀧𑀺 𑀯𑀳𑀢𑀺𑁇 𑀳𑁄𑀢𑀺 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑀬𑁄 𑀬𑀁 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑁂𑀦𑀧𑀺 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦𑀧𑀺 𑀯𑀸𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺, 𑀑𑀲𑁆𑀲𑀯𑀦𑁂𑀧𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀦 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀳𑀺 𑀦𑀸𑀫, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑀸𑀳𑀺𑀭𑀸𑀬 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀸𑀯 𑀫𑀳𑀮𑁆𑀮𑀺𑀓𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀔𑀬𑀥𑀫𑁆𑀫𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀺𑀁 𑀧𑀦𑀺𑀫𑀲𑁆𑀲 𑀫𑀢𑁆𑀢𑀝𑁆𑀞𑀓𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀢𑀡𑁆𑀳𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 ‘𑀅𑀳𑀦𑁆𑀢𑀺 𑀯𑀸 𑀫𑀫𑀦𑁆𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀫𑀻’𑀢𑀺 𑀯𑀸? 𑀅𑀣 𑀔𑁆𑀯𑀸𑀲𑁆𑀲 𑀦𑁄𑀢𑁂𑀯𑁂𑀢𑁆𑀣 𑀳𑁄𑀢𑀺𑁇

1069\. 𑀢𑀜𑁆𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀭𑁂 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀲𑀸 𑀘 𑀔𑁄 𑀧𑀝𑀺𑀘𑁆𑀘, 𑀦𑁄 𑀅𑀧𑀝𑀺𑀘𑁆𑀘𑁇 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘? 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑁇 𑀲𑁄𑀧𑀺 𑀨𑀲𑁆𑀲𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀥𑀸𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀫𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1070\. 𑀢𑀜𑁆𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀭𑁂 𑀅𑀦𑀺𑀝𑁆𑀞𑁂𑀳𑀺 𑀅𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀅𑀫𑀦𑀸𑀧𑁂𑀳𑀺 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺, 𑀧𑀸𑀡𑀺𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺 𑀮𑁂𑀟𑁆𑀟𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺 𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺 𑀲𑀢𑁆𑀣𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀦𑀧𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 ‘𑀢𑀣𑀸𑀪𑀽𑀢𑁄 𑀔𑁄 𑀅𑀬𑀁 𑀓𑀸𑀬𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀧𑀸𑀡𑀺𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺, 𑀮𑁂𑀟𑁆𑀟𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺, 𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀣𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀓𑀓𑀘𑀽𑀧𑀫𑁄𑀯𑀸𑀤𑁂 “𑀉𑀪𑀢𑁄𑀤𑀡𑁆𑀟𑀓𑁂𑀦 𑀘𑁂𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀓𑀘𑁂𑀦 𑀘𑁄𑀭𑀸 𑀑𑀘𑀭𑀓𑀸 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀑𑀓𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀬𑁄 𑀫𑀦𑁄 𑀧𑀤𑀽𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂 𑀲𑁄 𑀢𑁂𑀦 𑀲𑀸𑀲𑀦𑀓𑀭𑁄”𑀢𑀺𑁇 𑀆𑀭𑀤𑁆𑀥𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀯𑀻𑀭𑀺𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑀮𑁆𑀮𑀻𑀦𑀁, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀢𑀺 𑀅𑀲𑀫𑁆𑀫𑀼𑀝𑁆𑀞𑀸, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀓𑀸𑀬𑁄 𑀅𑀲𑀸𑀭𑀤𑁆𑀥𑁄, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑀕𑁆𑀕𑀁𑁇 𑀓𑀸𑀫𑀁 𑀤𑀸𑀦𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀧𑀸𑀡𑀺𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼, 𑀮𑁂𑀟𑁆𑀟𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼, 𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼, 𑀲𑀢𑁆𑀣𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀫𑀦𑁆𑀢𑀼𑁇 𑀓𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀤𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀦𑁆’𑀢𑀺𑁇

1073\. 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀏𑀯𑀜𑁆𑀳𑀺 𑀓𑀺𑀭 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀲𑀗𑁆𑀕𑀳𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀲𑀫𑀯𑀸𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  “𑀬𑁄 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀧𑀲𑁆𑀲𑀢𑀺; 𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁 𑀧𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑁄 𑀧𑀦𑀺𑀫𑁂 𑀬𑀤𑀺𑀤𑀁 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 𑀬𑁄 𑀇𑀫𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑁄 𑀆𑀮𑀬𑁄 𑀅𑀦𑀼𑀦𑀬𑁄 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀬𑁄 𑀇𑀫𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀁, 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀩𑀳𑀼𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

1074\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀜𑁆𑀘𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑁄𑀢𑀁 𑀅𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀖𑀸𑀦𑀁 𑀅𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀅𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺…  𑀓𑀸𑀬𑁄 𑀅𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…  𑀫𑀦𑁄 𑀅𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀳𑀺𑀭𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀆𑀧𑀸𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀦𑁄 𑀘 𑀢𑀚𑁆𑀚𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀢𑀚𑁆𑀚𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀪𑀸𑀕𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑁄 𑀘𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀦𑁄 𑀅𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀳𑀺𑀭𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀆𑀧𑀸𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀚𑁆𑀚𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀢𑀚𑁆𑀚𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀪𑀸𑀕𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑁄 𑀘𑁂𑀯 𑀫𑀦𑁄 𑀅𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀳𑀺𑀭𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀆𑀧𑀸𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀢𑀚𑁆𑀚𑁄 𑀘 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀁 𑀢𑀚𑁆𑀚𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀪𑀸𑀕𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀁 𑀢𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀁 𑀢𑀁 𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀗𑁆𑀕𑀳𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀬𑀸 𑀢𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀲 𑀯𑁂𑀤𑀦𑀸 𑀲𑀸 𑀯𑁂𑀤𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀗𑁆𑀕𑀳𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀬𑀸 𑀢𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀲𑀸 𑀲𑀜𑁆𑀜𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀗𑁆𑀕𑀳𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀬𑁂 𑀢𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀢𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀗𑁆𑀕𑀳𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀬𑀁 𑀢𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀗𑁆𑀕𑀳𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀏𑀯𑀜𑁆𑀳𑀺 𑀓𑀺𑀭 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀲𑀗𑁆𑀕𑀳𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀲𑀫𑀯𑀸𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  “𑀬𑁄 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀧𑀲𑁆𑀲𑀢𑀺; 𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁 𑀧𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑁄 𑀧𑀦𑀺𑀫𑁂 𑀬𑀤𑀺𑀤𑀁 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 𑀬𑁄 𑀇𑀫𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑁄 𑀆𑀮𑀬𑁄 𑀅𑀦𑀼𑀦𑀬𑁄 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀬𑁄 𑀇𑀫𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀁 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀩𑀳𑀼𑀓𑀢𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1075\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇
