# 𑁩𑁇𑁬 𑀧𑀸𑀲𑀭𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀧𑀸𑀲𑀭𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

(272.)

933\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀘𑀺𑀭𑀲𑁆𑀲𑀼𑀢𑀸 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀥𑀫𑁆𑀫𑀻 𑀓𑀣𑀸𑁇 𑀲𑀸𑀥𑀼 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀲𑀯𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀬𑁂𑀦 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀣; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀣 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀲𑀯𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇

934\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑁄 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑁄 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀧𑀼𑀩𑁆𑀩𑀓𑁄𑀝𑁆𑀞𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇

(273.)

935\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀧𑀼𑀩𑁆𑀩𑀓𑁄𑀝𑁆𑀞𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑀼𑀁𑁇 𑀧𑀼𑀩𑁆𑀩𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀘𑀻𑀯𑀭𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀬𑀫𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂𑁇 𑀭𑀫𑀡𑀻𑀬𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁄; 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁄𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

936\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁂 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀓𑀣𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀆𑀕𑀫𑀬𑀫𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀓𑀣𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀆𑀓𑁄𑀝𑁂𑀲𑀺𑁇 𑀯𑀺𑀯𑀭𑀺𑀁𑀲𑀼 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀤𑁆𑀯𑀸𑀭𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀫𑁆𑀫𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀸𑀬 𑀦𑀼𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸? 𑀓𑀸 𑀘 𑀧𑀦 𑀯𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸”𑀢𑀺? “𑀪𑀕𑀯𑀦𑁆𑀢𑀫𑁂𑀯 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀭𑀩𑁆𑀪 𑀥𑀫𑁆𑀫𑀻 𑀓𑀣𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸, 𑀅𑀣 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇

937\. “𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂𑁇 𑀏𑀢𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸𑀦𑀁 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀬𑀁 𑀓𑀭𑀡𑀻𑀬𑀁—  𑀥𑀫𑁆𑀫𑀻 𑀯𑀸 𑀓𑀣𑀸, 𑀅𑀭𑀺𑀬𑁄 𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄𑁇

938\. 𑀤𑁆𑀯𑁂𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸—  𑀅𑀭𑀺𑀬𑀸 𑀘 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸, 𑀅𑀦𑀭𑀺𑀬𑀸 𑀘 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑁇

939\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀭𑀺𑀬𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺𑁇

940\. 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁 𑀯𑀤𑁂𑀣? 𑀧𑀼𑀢𑁆𑀢𑀪𑀭𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁, 𑀤𑀸𑀲𑀺𑀤𑀸𑀲𑀁 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁, 𑀅𑀚𑁂𑀍𑀅𑀓𑀁 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀁 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁, 𑀳𑀢𑁆𑀣𑀺𑀕𑀯𑀸𑀲𑁆𑀲𑀯𑀍𑀅𑀯𑀁 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁, 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁𑁇 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀥𑀬𑁄𑁇 𑀏𑀢𑁆𑀣𑀸𑀬𑀁 𑀕𑀣𑀺𑀢𑁄 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺𑁇

942\. 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁 𑀯𑀤𑁂𑀣? 𑀧𑀼𑀢𑁆𑀢𑀪𑀭𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁, 𑀤𑀸𑀲𑀺𑀤𑀸𑀲𑀁 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁, 𑀅𑀚𑁂𑀍𑀅𑀓𑀁 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀁 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁, 𑀳𑀢𑁆𑀣𑀺𑀕𑀯𑀸𑀲𑁆𑀲𑀯𑀍𑀅𑀯𑀁 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁𑁇 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀥𑀬𑁄𑁇 𑀏𑀢𑁆𑀣𑀸𑀬𑀁 𑀕𑀣𑀺𑀢𑁄 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺𑁇

943\. 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁 𑀯𑀤𑁂𑀣? 𑀧𑀼𑀢𑁆𑀢𑀪𑀭𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁, 𑀤𑀸𑀲𑀺𑀤𑀸𑀲𑀁 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁, 𑀅𑀚𑁂𑀍𑀅𑀓𑀁 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀁 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁, 𑀳𑀢𑁆𑀣𑀺𑀕𑀯𑀸𑀲𑁆𑀲𑀯𑀍𑀅𑀯𑀁 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁𑁇 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀥𑀬𑁄𑁇 𑀏𑀢𑁆𑀣𑀸𑀬𑀁 𑀕𑀣𑀺𑀢𑁄 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺𑁇

944\. 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁 𑀯𑀤𑁂𑀣? 𑀧𑀼𑀢𑁆𑀢𑀪𑀭𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁, 𑀤𑀸𑀲𑀺𑀤𑀸𑀲𑀁 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁, 𑀅𑀚𑁂𑀍𑀅𑀓𑀁 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀁 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁, 𑀳𑀢𑁆𑀣𑀺𑀕𑀯𑀸𑀲𑁆𑀲𑀯𑀍𑀅𑀯𑀁 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁𑁇 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀥𑀬𑁄𑁇 𑀏𑀢𑁆𑀣𑀸𑀬𑀁 𑀕𑀣𑀺𑀢𑁄 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺𑁇

948\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑀳𑀭𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 𑀲𑀼𑀲𑀼𑀓𑀸𑀍𑀅𑀓𑁂𑀲𑁄, 𑀪𑀤𑁆𑀭𑁂𑀦 𑀬𑁄𑀩𑁆𑀩𑀦𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀧𑀞𑀫𑁂𑀦 𑀯𑀬𑀲𑀸 𑀅𑀓𑀸𑀫𑀓𑀸𑀦𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀅𑀲𑁆𑀲𑀼𑀫𑀼𑀔𑀸𑀦𑀁 𑀭𑀼𑀤𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀁𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀓𑀺𑀁𑀓𑀼𑀲𑀮𑀕𑀯𑁂𑀲𑀻 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀦𑁆𑀢𑀺𑀯𑀭𑀧𑀤𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑁇 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀍𑀆𑀭𑀁 𑀓𑀸𑀮𑀸𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀸𑀮𑀸𑀫, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀯𑀺𑀳𑀭𑀢𑀸𑀬𑀲𑁆𑀫𑀸; 𑀢𑀸𑀤𑀺𑀲𑁄 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀬𑀢𑁆𑀣 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀁𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀯𑀢𑀓𑁂𑀦𑁂𑀯 𑀑𑀝𑁆𑀞𑀧𑀳𑀢𑀫𑀢𑁆𑀢𑁂𑀦 𑀮𑀧𑀺𑀢𑀮𑀸𑀧𑀦𑀫𑀢𑁆𑀢𑁂𑀦 𑀜𑀸𑀡𑀯𑀸𑀤𑀜𑁆𑀘 𑀯𑀤𑀸𑀫𑀺 𑀣𑁂𑀭𑀯𑀸𑀤𑀜𑁆𑀘, ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀘 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑁂 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦 𑀔𑁄 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀓𑁂𑀯𑀮𑀁 𑀲𑀤𑁆𑀥𑀸𑀫𑀢𑁆𑀢𑀓𑁂𑀦 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻𑀢𑀺 𑀧𑀯𑁂𑀤𑁂𑀢𑀺; 𑀅𑀤𑁆𑀥𑀸 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀁 𑀯𑀺𑀳𑀭𑀢𑀻’𑀢𑀺𑁇

949\. 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀍𑀆𑀭𑀁 𑀓𑀸𑀮𑀸𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀸𑀮𑀸𑀫, 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻𑀢𑀺 𑀧𑀯𑁂𑀤𑁂𑀲𑀻’𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀧𑀯𑁂𑀤𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦 𑀔𑁄 𑀆𑀍𑀆𑀭𑀲𑁆𑀲𑁂𑀯 𑀓𑀸𑀮𑀸𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺 𑀲𑀤𑁆𑀥𑀸, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀲𑀤𑁆𑀥𑀸; 𑀦 𑀔𑁄 𑀆𑀍𑀆𑀭𑀲𑁆𑀲𑁂𑀯 𑀓𑀸𑀮𑀸𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺 𑀯𑀻𑀭𑀺𑀬𑀁, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀯𑀻𑀭𑀺𑀬𑀁; 𑀦 𑀔𑁄 𑀆𑀍𑀆𑀭𑀲𑁆𑀲𑁂𑀯 𑀓𑀸𑀮𑀸𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑀺, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀲𑀢𑀺; 𑀦 𑀔𑁄 𑀆𑀍𑀆𑀭𑀲𑁆𑀲𑁂𑀯 𑀓𑀸𑀮𑀸𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑀸𑀥𑀺, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀲𑀫𑀸𑀥𑀺; 𑀦 𑀔𑁄 𑀆𑀍𑀆𑀭𑀲𑁆𑀲𑁂𑀯 𑀓𑀸𑀮𑀸𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀜𑀸, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀜𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻𑀢𑀺 𑀧𑀯𑁂𑀤𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀤𑀳𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇

950\. 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀍𑀆𑀭𑀁 𑀓𑀸𑀮𑀸𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—

951\. ‘𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀸𑀮𑀸𑀫, 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀧𑀯𑁂𑀤𑁂𑀲𑀻’𑀢𑀺?

952\. ‘𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀧𑀯𑁂𑀤𑁂𑀫𑀻’𑀢𑀺𑁇

953\. ‘𑀅𑀳𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇

956\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁 𑀓𑀼𑀲𑀮𑀕𑀯𑁂𑀲𑀻 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀦𑁆𑀢𑀺𑀯𑀭𑀧𑀤𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀓𑀁 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀯𑀺𑀳𑀭𑀢𑀸𑀬𑀲𑁆𑀫𑀸; 𑀢𑀸𑀤𑀺𑀲𑁄 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀬𑀢𑁆𑀣 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀁𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀯𑀢𑀓𑁂𑀦𑁂𑀯 𑀑𑀝𑁆𑀞𑀧𑀳𑀢𑀫𑀢𑁆𑀢𑁂𑀦 𑀮𑀧𑀺𑀢𑀮𑀸𑀧𑀦𑀫𑀢𑁆𑀢𑁂𑀦 𑀜𑀸𑀡𑀯𑀸𑀤𑀜𑁆𑀘 𑀯𑀤𑀸𑀫𑀺 𑀣𑁂𑀭𑀯𑀸𑀤𑀜𑁆𑀘, ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀘 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑁂 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦 𑀔𑁄 𑀭𑀸𑀫𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀓𑁂𑀯𑀮𑀁 𑀲𑀤𑁆𑀥𑀸𑀫𑀢𑁆𑀢𑀓𑁂𑀦 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻𑀢𑀺 𑀧𑀯𑁂𑀤𑁂𑀲𑀺; 𑀅𑀤𑁆𑀥𑀸 𑀭𑀸𑀫𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀁 𑀯𑀺𑀳𑀸𑀲𑀻’𑀢𑀺𑁇

957\. 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀓𑀁 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀭𑀸𑀫𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻𑀢𑀺 𑀧𑀯𑁂𑀤𑁂𑀲𑀻’𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀧𑀯𑁂𑀤𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦 𑀔𑁄 𑀭𑀸𑀫𑀲𑁆𑀲𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀲𑀤𑁆𑀥𑀸, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀲𑀤𑁆𑀥𑀸; 𑀦 𑀔𑁄 𑀭𑀸𑀫𑀲𑁆𑀲𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀯𑀻𑀭𑀺𑀬𑀁, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀯𑀻𑀭𑀺𑀬𑀁; 𑀦 𑀔𑁄 𑀭𑀸𑀫𑀲𑁆𑀲𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀲𑀢𑀺, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀲𑀢𑀺; 𑀦 𑀔𑁄 𑀭𑀸𑀫𑀲𑁆𑀲𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀲𑀫𑀸𑀥𑀺, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀲𑀫𑀸𑀥𑀺, 𑀦 𑀔𑁄 𑀭𑀸𑀫𑀲𑁆𑀲𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀧𑀜𑁆𑀜𑀸, 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀜𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑀸𑀫𑁄 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻𑀢𑀺 𑀧𑀯𑁂𑀤𑁂𑀲𑀺, 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀤𑀳𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇

958\. 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀓𑀁 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—

959\. ‘𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀭𑀸𑀫𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀧𑀯𑁂𑀤𑁂𑀲𑀻’𑀢𑀺?

960\. ‘𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀭𑀸𑀫𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀧𑀯𑁂𑀤𑁂𑀲𑀻’𑀢𑀺𑁇

965\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑀸𑀢𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀅𑀚𑀸𑀢𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑀭𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀅𑀚𑀭𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀩𑁆𑀬𑀸𑀥𑀺𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀅𑀩𑁆𑀬𑀸𑀥𑀺𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑀢𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑁄𑀓𑀥𑀫𑁆𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁄𑀓𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀥𑀫𑁆𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀁𑁇 𑀜𑀸𑀡𑀜𑁆𑀘 𑀧𑀦 𑀫𑁂 𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀫𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀅𑀬𑀫𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀸𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄’𑀢𑀺𑁇

966\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀥𑀺𑀕𑀢𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀤𑀼𑀤𑁆𑀤𑀲𑁄 𑀤𑀼𑀭𑀦𑀼𑀩𑁄𑀥𑁄 𑀲𑀦𑁆𑀢𑁄 𑀧𑀡𑀻𑀢𑁄 𑀅𑀢𑀓𑁆𑀓𑀸𑀯𑀘𑀭𑁄 𑀦𑀺𑀧𑀼𑀡𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑁂𑀤𑀦𑀻𑀬𑁄𑁇 𑀆𑀮𑀬𑀭𑀸𑀫𑀸 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀧𑀚𑀸 𑀆𑀮𑀬𑀭𑀢𑀸 𑀆𑀮𑀬𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑁇 𑀆𑀮𑀬𑀭𑀸𑀫𑀸𑀬 𑀔𑁄 𑀧𑀦 𑀧𑀚𑀸𑀬 𑀆𑀮𑀬𑀭𑀢𑀸𑀬 𑀆𑀮𑀬𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀇𑀤𑀁 𑀞𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄𑁇 𑀇𑀤𑀫𑁆𑀧𑀺 𑀔𑁄 𑀞𑀸𑀦𑀁 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀬𑀤𑀺𑀤𑀁—  𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀣𑁄 𑀲𑀩𑁆𑀩𑀽𑀧𑀥𑀺𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁄 𑀯𑀺𑀭𑀸𑀕𑁄 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑁂 𑀘 𑀫𑁂 𑀦 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑁄 𑀫𑀫𑀲𑁆𑀲 𑀓𑀺𑀮𑀫𑀣𑁄, 𑀲𑀸 𑀫𑀫𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀸’𑀢𑀺𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼 𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀧𑀝𑀺𑀪𑀁𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀲𑁆𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀸—

967\. _‘𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀫𑁂 𑀅𑀥𑀺𑀕𑀢𑀁,_  
_𑀳𑀮𑀁 𑀤𑀸𑀦𑀺 𑀧𑀓𑀸𑀲𑀺𑀢𑀼𑀁;_  
_𑀭𑀸𑀕𑀤𑁄𑀲𑀧𑀭𑁂𑀢𑁂𑀳𑀺,_  
_𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀲𑀫𑁆𑀩𑀼𑀥𑁄𑁇_  


968\. _𑀧𑀝𑀺𑀲𑁄𑀢𑀕𑀸𑀫𑀺𑀁 𑀦𑀺𑀧𑀼𑀡𑀁,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀅𑀡𑀼𑀁;_  
_𑀭𑀸𑀕𑀭𑀢𑁆𑀢𑀸 𑀦 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀺,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁂𑀦 𑀆𑀯𑀼𑀝𑀸’𑀢𑀺𑁇_

969\. 𑀇𑀢𑀺𑀳 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀢𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀢𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀲𑀳𑀫𑁆𑀧𑀢𑀺𑀲𑁆𑀲 𑀫𑀫 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦𑀲𑁆𑀲𑀢𑀺 𑀯𑀢 𑀪𑁄 𑀮𑁄𑀓𑁄, 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺 𑀯𑀢 𑀪𑁄 𑀮𑁄𑀓𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀢𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀢𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀫 𑀧𑀼𑀭𑀢𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀤𑁂𑀲𑁂𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁, 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸, 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

970\. _‘𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀫𑀕𑀥𑁂𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂,_  
_𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀼𑀤𑁆𑀥𑁄 𑀲𑀫𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑁄;_  
_𑀅𑀧𑀸𑀧𑀼𑀭𑁂𑀢𑀁 𑀅𑀫𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀭𑀁,_  
_𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀫𑀮𑁂𑀦𑀸𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁𑁇_  


971\. _𑀲𑁂𑀮𑁂 𑀬𑀣𑀸 𑀧𑀩𑁆𑀩𑀢𑀫𑀼𑀤𑁆𑀥𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄,_  
_𑀬𑀣𑀸𑀧𑀺 𑀧𑀲𑁆𑀲𑁂 𑀚𑀦𑀢𑀁 𑀲𑀫𑀦𑁆𑀢𑀢𑁄;_  
_𑀢𑀣𑀽𑀧𑀫𑀁 𑀥𑀫𑁆𑀫𑀫𑀬𑀁 𑀲𑀼𑀫𑁂𑀥,_  
_𑀧𑀸𑀲𑀸𑀤𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀲𑀫𑀦𑁆𑀢𑀘𑀓𑁆𑀔𑀼;_  
_𑀲𑁄𑀓𑀸𑀯𑀢𑀺𑀡𑁆𑀡𑀁 𑀚𑀦𑀢𑀫𑀧𑁂𑀢𑀲𑁄𑀓𑁄,_  
_𑀅𑀯𑁂𑀓𑁆𑀔𑀲𑁆𑀲𑀼 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀁𑁇_  


972\. _𑀉𑀝𑁆𑀞𑁂𑀳𑀺 𑀯𑀻𑀭 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫,_  
_𑀲𑀢𑁆𑀣𑀯𑀸𑀳 𑀅𑀡𑀡 𑀯𑀺𑀘𑀭 𑀮𑁄𑀓𑁂;_  
_𑀤𑁂𑀲𑀲𑁆𑀲𑀼 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁,_  
_𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇_  


(283.)

973\. 𑀅𑀣 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀘 𑀅𑀚𑁆𑀛𑁂𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀘 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀮𑁄𑀓𑀁 𑀯𑁄𑀮𑁄𑀓𑁂𑀲𑀺𑀁𑁇 𑀅𑀤𑁆𑀤𑀲𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀼𑀤𑁆𑀥𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀮𑁄𑀓𑀁 𑀯𑁄𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑁂 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑁂 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑁂, 𑀢𑀺𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀫𑀼𑀤𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂, 𑀲𑁆𑀯𑀸𑀓𑀸𑀭𑁂 𑀤𑁆𑀯𑀸𑀓𑀸𑀭𑁂, 𑀲𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂 𑀤𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀉𑀧𑁆𑀧𑀮𑀺𑀦𑀺𑀬𑀁 𑀯𑀸 𑀧𑀤𑀼𑀫𑀺𑀦𑀺𑀬𑀁 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀺𑀦𑀺𑀬𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀮𑀸𑀦𑀺 𑀯𑀸 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂 𑀲𑀁𑀯𑀟𑁆𑀠𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀼𑀕𑁆𑀕𑀢𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑁄𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑀧𑁄𑀲𑀻𑀦𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀮𑀸𑀦𑀺 𑀯𑀸 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂 𑀲𑀁𑀯𑀟𑁆𑀠𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀼𑀕𑁆𑀕𑀢𑀸𑀦𑀺 𑀲𑀫𑁄𑀤𑀓𑀁 𑀞𑀺𑀢𑀸𑀦𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀮𑀸𑀦𑀺 𑀯𑀸 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂 𑀲𑀁𑀯𑀟𑁆𑀠𑀸𑀦𑀺 𑀉𑀤𑀓𑀁 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑁆𑀫 𑀞𑀺𑀢𑀸𑀦𑀺 𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂𑀦; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀼𑀤𑁆𑀥𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀮𑁄𑀓𑀁 𑀯𑁄𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀢𑁆𑀢𑁂 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑁂 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑁂, 𑀢𑀺𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀫𑀼𑀤𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂, 𑀲𑁆𑀯𑀸𑀓𑀸𑀭𑁂 𑀤𑁆𑀯𑀸𑀓𑀸𑀭𑁂, 𑀲𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂 𑀤𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀲𑀳𑀫𑁆𑀧𑀢𑀺𑀁 𑀕𑀸𑀣𑀸𑀬 𑀧𑀘𑁆𑀘𑀪𑀸𑀲𑀺𑀁—

974\. _‘𑀅𑀧𑀸𑀭𑀼𑀢𑀸 𑀢𑁂𑀲𑀁 𑀅𑀫𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀭𑀸,_  
_𑀬𑁂 𑀲𑁄𑀢𑀯𑀦𑁆𑀢𑁄 𑀧𑀫𑀼𑀜𑁆𑀘𑀦𑁆𑀢𑀼 𑀲𑀤𑁆𑀥𑀁;_  
_𑀯𑀺𑀳𑀺𑀁𑀲𑀲𑀜𑁆𑀜𑀻 𑀧𑀕𑀼𑀡𑀁 𑀦 𑀪𑀸𑀲𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀧𑀡𑀻𑀢𑀁 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑁂’𑀢𑀺𑁇_  


975\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 ‘𑀓𑀢𑀸𑀯𑀓𑀸𑀲𑁄 𑀔𑁄𑀫𑁆𑀳𑀺 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀸’𑀢𑀺 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇

(284.)

976\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁; 𑀓𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀆𑀍𑀆𑀭𑀲𑁆𑀲 𑀓𑀸𑀮𑀸𑀫𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀢𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄’𑀢𑀺𑁇 𑀜𑀸𑀡𑀜𑁆𑀘 𑀧𑀦 𑀫𑁂 𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀫𑀳𑀸𑀚𑀸𑀦𑀺𑀬𑁄 𑀔𑁄 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄𑁇 𑀲𑀘𑁂 𑀳𑀺 𑀲𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬, 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

977\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁; 𑀓𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀉𑀤𑀓𑀲𑁆𑀲 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀢𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀅𑀪𑀺𑀤𑁄𑀲𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄’𑀢𑀺𑁇 𑀜𑀸𑀡𑀜𑁆𑀘 𑀧𑀦 𑀫𑁂 𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀅𑀪𑀺𑀤𑁄𑀲𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀫𑀳𑀸𑀚𑀸𑀦𑀺𑀬𑁄 𑀔𑁄 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄𑁇 𑀲𑀘𑁂 𑀳𑀺 𑀲𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬, 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

978\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁; 𑀓𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀩𑀳𑀼𑀓𑀸𑀭𑀸 𑀔𑁄 𑀫𑁂 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀬𑁂 𑀫𑀁 𑀧𑀥𑀸𑀦𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑀲𑀼𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀳𑀁 𑀦𑀼 𑀔𑁄 𑀏𑀢𑀭𑀳𑀺 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻’𑀢𑀺? 𑀅𑀤𑁆𑀤𑀲𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀭𑀼𑀯𑁂𑀮𑀸𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑁇

(285.)

979\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀓𑁄 𑀆𑀚𑀻𑀯𑀓𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀕𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀩𑁄𑀥𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀓𑀁𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀓𑁄 𑀯𑀸 𑀢𑁂 𑀲𑀢𑁆𑀣𑀸, 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀲𑀻’𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀓𑀁 𑀆𑀚𑀻𑀯𑀓𑀁 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺𑀁—

980\. _‘𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀪𑀽 𑀲𑀩𑁆𑀩𑀯𑀺𑀤𑀽𑀳𑀫𑀲𑁆𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀦𑀽𑀧𑀮𑀺𑀢𑁆𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀚𑀳𑁄 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄,_  
_𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀓𑀫𑀼𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀁𑁇_  


981\. _𑀦 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


982\. _𑀅𑀳𑀜𑁆𑀳𑀺 𑀅𑀭𑀳𑀸 𑀮𑁄𑀓𑁂,_  
_𑀅𑀳𑀁 𑀲𑀢𑁆𑀣𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀏𑀓𑁄𑀫𑁆𑀳𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑁄𑀲𑁆𑀫𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇_  


983\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀼𑀁,_  
_𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀓𑀸𑀲𑀺𑀦𑀁 𑀧𑀼𑀭𑀁;_  
_𑀅𑀦𑁆𑀥𑀻𑀪𑀽𑀢𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀆𑀳𑀜𑁆𑀙𑀁 𑀅𑀫𑀢𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀦𑁆’𑀢𑀺𑁇_  


984\. ‘𑀬𑀣𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀲𑀺, 𑀅𑀭𑀳𑀲𑀺 𑀅𑀦𑀦𑁆𑀢𑀚𑀺𑀦𑁄’𑀢𑀺𑁇

985\. _‘𑀫𑀸𑀤𑀺𑀲𑀸 𑀯𑁂 𑀚𑀺𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀧𑀢𑁆𑀢𑀸 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁;_  
_𑀚𑀺𑀢𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀢𑀲𑁆𑀫𑀸𑀳𑀫𑀼𑀧𑀓 𑀚𑀺𑀦𑁄’𑀢𑀺𑁇_  


986\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀓𑁄 𑀆𑀚𑀻𑀯𑀓𑁄 ‘𑀳𑀼𑀧𑁂𑀬𑁆𑀬𑀧𑀸𑀯𑀼𑀲𑁄’𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀲𑀻𑀲𑀁 𑀑𑀓𑀫𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀫𑁆𑀫𑀕𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

(286.)

987\. 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀇𑀲𑀺𑀧𑀢𑀦𑀁 𑀫𑀺𑀕𑀤𑀸𑀬𑁄 𑀬𑁂𑀦 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑁇 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀔𑁄 𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀭𑀢𑁄 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀡𑁆𑀞𑀧𑁂𑀲𑀼𑀁—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀺𑀓𑁄 𑀧𑀥𑀸𑀦𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑁄 𑀆𑀯𑀢𑁆𑀢𑁄 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬𑁇 𑀲𑁄 𑀦𑁂𑀯 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄; 𑀦𑀸𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀆𑀲𑀦𑀁 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑀁, 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀺 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀲𑀓𑀸𑀬 𑀓𑀢𑀺𑀓𑀸𑀬 𑀲𑀡𑁆𑀞𑀸𑀢𑀼𑀁𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀼𑀁, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀼𑀁, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑀼𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀫𑀁 𑀦𑀸𑀫𑁂𑀦 𑀘 𑀆𑀯𑀼𑀲𑁄𑀯𑀸𑀤𑁂𑀦 𑀘 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇

988\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀸𑀫𑁂𑀦 𑀘 𑀆𑀯𑀼𑀲𑁄𑀯𑀸𑀤𑁂𑀦 𑀘 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀣𑁇 𑀅𑀭𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀑𑀤𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀢𑀁, 𑀅𑀫𑀢𑀫𑀥𑀺𑀕𑀢𑀁, 𑀅𑀳𑀫𑀦𑀼𑀲𑀸𑀲𑀸𑀫𑀺, 𑀅𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀺𑁇 𑀬𑀣𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀁 𑀢𑀣𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀢𑀸𑀬𑀧𑀺 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀕𑁄𑀢𑀫, 𑀇𑀭𑀺𑀬𑀸𑀬 𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸𑀬 𑀦𑀸𑀚𑁆𑀛𑀕𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁, 𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁 𑀏𑀢𑀭𑀳𑀺 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀺𑀓𑁄 𑀧𑀥𑀸𑀦𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑁄 𑀆𑀯𑀢𑁆𑀢𑁄 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀅𑀥𑀺𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀦𑁆’𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀺𑀓𑁄, 𑀦 𑀧𑀥𑀸𑀦𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑁄, 𑀦 𑀆𑀯𑀢𑁆𑀢𑁄 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬𑁇 𑀅𑀭𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀑𑀤𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀢𑀁, 𑀅𑀫𑀢𑀫𑀥𑀺𑀕𑀢𑀁, 𑀅𑀳𑀫𑀦𑀼𑀲𑀸𑀲𑀸𑀫𑀺, 𑀅𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀺𑁇 𑀬𑀣𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀁 𑀢𑀣𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀣𑀸’𑀢𑀺𑁇

998\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

999\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

1000\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

1001\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

1002\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

1003\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

1004\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇

1005\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀲𑁆𑀲 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁆𑀥𑀫𑀓𑀸𑀲𑀺 𑀫𑀸𑀭𑀁 𑀅𑀧𑀤𑀁, 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀕𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇 𑀢𑀺𑀡𑁆𑀡𑁄 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀯𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁄 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁄 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁄 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁄 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀸𑀧𑀸𑀣𑀕𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀧𑀺𑀫𑀢𑁄”𑀢𑀺𑁇

1006\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

1007\. 𑀧𑀸𑀲𑀭𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
