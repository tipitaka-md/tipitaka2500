# 𑁩𑁇𑁧𑁦 𑀘𑀽𑀍𑀅𑀲𑀸𑀭𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀲𑀸𑀭𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(312.)

1095\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀺𑀗𑁆𑀕𑀮𑀓𑁄𑀘𑁆𑀙𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀧𑀺𑀗𑁆𑀕𑀮𑀓𑁄𑀘𑁆𑀙𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁂𑀫𑁂, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀁𑀖𑀺𑀦𑁄 𑀕𑀡𑀺𑀦𑁄 𑀕𑀡𑀸𑀘𑀭𑀺𑀬𑀸 𑀜𑀸𑀢𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄 𑀢𑀺𑀢𑁆𑀣𑀓𑀭𑀸 𑀲𑀸𑀥𑀼𑀲𑀫𑁆𑀫𑀢𑀸, 𑀩𑀳𑀼𑀚𑀦𑀲𑁆𑀲, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀧𑀽𑀭𑀡𑁄 𑀓𑀲𑁆𑀲𑀧𑁄, 𑀫𑀓𑁆𑀔𑀮𑀺 𑀕𑁄𑀲𑀸𑀮𑁄, 𑀅𑀚𑀺𑀢𑁄 𑀓𑁂𑀲𑀓𑀫𑁆𑀩𑀮𑁄, 𑀧𑀓𑀼𑀥𑁄 𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄, 𑀲𑀜𑁆𑀘𑀬𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄, 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄, 𑀲𑀩𑁆𑀩𑁂𑀢𑁂 𑀲𑀓𑀸𑀬 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼 𑀲𑀩𑁆𑀩𑁂𑀯 𑀦𑀸𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑀘𑁆𑀘𑁂 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼 𑀏𑀓𑀘𑁆𑀘𑁂 𑀦𑀸𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀽”𑀢𑀺? “𑀅𑀮𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁—  𑀲𑀩𑁆𑀩𑁂𑀢𑁂 𑀲𑀓𑀸𑀬 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼 𑀲𑀩𑁆𑀩𑁂𑀯 𑀦𑀸𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑀘𑁆𑀘𑁂 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼 𑀏𑀓𑀘𑁆𑀘𑁂 𑀦𑀸𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀽𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀁 𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀧𑀺𑀗𑁆𑀕𑀮𑀓𑁄𑀘𑁆𑀙𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

(313.)

1096\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬 ‘𑀲𑀸𑀭𑀦𑁆’𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦 𑀯𑀢𑀸𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀭𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀨𑁂𑀕𑁆𑀕𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀢𑀘𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁𑁇 𑀢𑀣𑀸 𑀳𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 “𑀲𑀸𑀭𑀦𑁆”𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀭𑁂𑀦 𑀲𑀸𑀭𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀦𑀸𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

(314.)

1097\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀯𑀸 𑀧𑀦, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁, 𑀧𑀧𑀝𑀺𑀓𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬 ‘𑀲𑀸𑀭𑀦𑁆’𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦 𑀯𑀢𑀸𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀭𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀨𑁂𑀕𑁆𑀕𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀢𑀘𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀧𑀧𑀝𑀺𑀓𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁𑁇 𑀢𑀣𑀸 𑀳𑀬𑀁 𑀪𑀯𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀲𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀘𑀁 𑀧𑀧𑀝𑀺𑀓𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 “𑀲𑀸𑀭𑀦𑁆”𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀭𑁂𑀦 𑀲𑀸𑀭𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀦𑀸𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

(315.)

1106\. 𑀓𑀢𑀫𑁂 𑀘, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑀸 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑀸 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑀸 𑀘? 𑀇𑀥, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1107\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1108\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1109\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1110\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1111\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1112\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1113\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

1114\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀲𑁆𑀲 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑁄 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀥𑀫𑁆𑀫𑀸 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑀸 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑀸 𑀘𑁇

(324.)

1115\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀸𑀭𑀕𑀯𑁂𑀲𑀻 𑀲𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀲𑀸𑀭𑀯𑀢𑁄 𑀲𑀸𑀭𑀁𑀬𑁂𑀯 𑀙𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 ‘𑀲𑀸𑀭𑀦𑁆’𑀢𑀺 𑀚𑀸𑀦𑀫𑀸𑀦𑁄𑁇 𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀭𑁂𑀦 𑀲𑀸𑀭𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀣𑀽𑀧𑀫𑀸𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

1116\. 𑀇𑀢𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀦𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀸𑀦𑀺𑀲𑀁𑀲𑀁, 𑀦 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀺𑀲𑀁𑀲𑀁, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀺𑀲𑀁𑀲𑀁, 𑀦 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀁𑁇 𑀬𑀸 𑀘 𑀔𑁄 𑀅𑀬𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀏𑀢𑀤𑀢𑁆𑀣𑀫𑀺𑀤𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀏𑀢𑀁 𑀲𑀸𑀭𑀁 𑀏𑀢𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

1117\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀧𑀺𑀗𑁆𑀕𑀮𑀓𑁄𑀘𑁆𑀙𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

---

1118\. 𑀘𑀽𑀍𑀅𑀲𑀸𑀭𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀲𑀫𑀁𑁇

1119\. 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀢𑀢𑀺𑀬𑁄𑁇

1120\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1121\. _𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑀭𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑀸𑀫𑁄,_  
_𑀅𑀦𑁆𑀥𑀯𑀦𑁂 𑀓𑀣𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀦𑀺𑀯𑀸𑀧𑁄;_  
_𑀭𑀸𑀲𑀺𑀓𑀡𑁂𑀭𑀼𑀫𑀳𑀸𑀕𑀚𑀦𑀸𑀫𑁄,_  
_𑀲𑀸𑀭𑀽𑀧𑀫𑁄 𑀧𑀼𑀦 𑀧𑀺𑀗𑁆𑀕𑀮𑀓𑁄𑀘𑁆𑀙𑁄𑁇_
