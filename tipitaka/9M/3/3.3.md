# 𑁩𑁇𑁩 𑀯𑀫𑁆𑀫𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀯𑀫𑁆𑀫𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(249.)

837. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀦𑁆𑀥𑀯𑀦𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

838. “𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀯𑀫𑁆𑀫𑀺𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀁 𑀥𑀽𑀫𑀸𑀬𑀢𑀺, 𑀤𑀺𑀯𑀸 𑀧𑀚𑁆𑀚𑀮𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀮𑀗𑁆𑀕𑀺𑀁 ‘𑀮𑀗𑁆𑀕𑀻, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀉𑀓𑁆𑀔𑀺𑀧 𑀮𑀗𑁆𑀕𑀺𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀬𑀺𑀓𑀁𑁇 ‘𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀬𑀺𑀓𑀸, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀉𑀓𑁆𑀔𑀺𑀧 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀬𑀺𑀓𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑀁𑁇 ‘𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑁄, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀉𑀓𑁆𑀔𑀺𑀧 𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀘𑀗𑁆𑀕𑀯𑀸𑀭𑀁𑁇 ‘𑀘𑀗𑁆𑀕𑀯𑀸𑀭𑁄, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀉𑀓𑁆𑀔𑀺𑀧 𑀘𑀗𑁆𑀕𑀯𑀸𑀭𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀓𑀼𑀫𑁆𑀫𑀁𑁇 ‘𑀓𑀼𑀫𑁆𑀫𑁄, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀉𑀓𑁆𑀔𑀺𑀧 𑀓𑀼𑀫𑁆𑀫𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀅𑀲𑀺𑀲𑀽𑀦𑀁𑁇 ‘𑀅𑀲𑀺𑀲𑀽𑀦𑀸, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀉𑀓𑁆𑀔𑀺𑀧 𑀅𑀲𑀺𑀲𑀽𑀦𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀫𑀁𑀲𑀧𑁂𑀲𑀺𑀁𑁇 ‘𑀫𑀁𑀲𑀧𑁂𑀲𑀺, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀉𑀓𑁆𑀔𑀺𑀧 𑀫𑀁𑀲𑀧𑁂𑀲𑀺𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬 𑀅𑀤𑁆𑀤𑀲 𑀦𑀸𑀕𑀁𑁇 ‘𑀦𑀸𑀕𑁄, 𑀪𑀤𑀦𑁆𑀢𑁂’𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀦𑀸𑀕𑁄, 𑀫𑀸 𑀦𑀸𑀕𑀁 𑀖𑀝𑁆𑀝𑁂𑀲𑀺; 𑀦𑀫𑁄 𑀓𑀭𑁄𑀳𑀺 𑀦𑀸𑀕𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

839. 𑀇𑀫𑁂 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀜𑁆𑀳𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀬𑀣𑀸 𑀘 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺 𑀢𑀣𑀸 𑀦𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀦𑀸𑀳𑀁 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬, 𑀬𑁄 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀳𑀸𑀦𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀆𑀭𑀸𑀥𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀯𑀸, 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑁂𑀦 𑀯𑀸, 𑀇𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀲𑀼𑀢𑁆𑀯𑀸”𑀢𑀺—  𑀇𑀤𑀫𑀯𑁄𑀘 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇

(250.)

840. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀯𑀫𑁆𑀫𑀺𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀁 𑀥𑀽𑀫𑀸𑀬𑀢𑀺, 𑀤𑀺𑀯𑀸 𑀧𑀚𑁆𑀚𑀮𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆𑀢𑁄 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬…𑀧𑁂…  𑀇𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀲𑀼𑀢𑁆𑀯𑀸’𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀫𑁆𑀫𑀺𑀓𑁄, 𑀓𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀥𑀽𑀫𑀸𑀬𑀦𑀸, 𑀓𑀸 𑀤𑀺𑀯𑀸 𑀧𑀚𑁆𑀚𑀮𑀦𑀸, 𑀓𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄, 𑀓𑁄 𑀲𑀼𑀫𑁂𑀥𑁄, 𑀓𑀺𑀁 𑀲𑀢𑁆𑀣𑀁, 𑀓𑀺𑀁 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁, 𑀓𑀸 𑀮𑀗𑁆𑀕𑀻, 𑀓𑀸 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀬𑀺𑀓𑀸, 𑀓𑁄 𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑁄, 𑀓𑀺𑀁 𑀘𑀗𑁆𑀕𑀯𑀸𑀭𑀁, 𑀓𑁄 𑀓𑀼𑀫𑁆𑀫𑁄, 𑀓𑀸 𑀅𑀲𑀺𑀲𑀽𑀦𑀸, 𑀓𑀸 𑀫𑀁𑀲𑀧𑁂𑀲𑀺, 𑀓𑁄 𑀦𑀸𑀕𑁄”𑀢𑀺?

(251.)

841. “‘𑀯𑀫𑁆𑀫𑀺𑀓𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀫𑀲𑁆𑀲𑁂𑀢𑀁 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀥𑀺𑀯𑀘𑀦𑀁, 𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀲𑀫𑁆𑀪𑀯𑀲𑁆𑀲 𑀑𑀤𑀦𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀽𑀧𑀘𑀬𑀲𑁆𑀲 𑀅𑀦𑀺𑀘𑁆𑀘𑀼𑀘𑁆𑀙𑀸𑀤𑀦𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀦𑀪𑁂𑀤𑀦𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑀦𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑁇 (𑁧)

842. 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀯𑀸 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀭𑀩𑁆𑀪 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀭𑀢𑁆𑀢𑀺𑀁 𑀥𑀽𑀫𑀸𑀬𑀦𑀸𑁇 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀤𑀺𑀯𑀸 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀬𑁄𑀚𑁂𑀢𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 ‘𑀫𑀦𑀲𑀸’—  𑀅𑀬𑀁 𑀤𑀺𑀯𑀸 𑀧𑀚𑁆𑀚𑀮𑀦𑀸𑁇 (𑁨--𑁩)

843. ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀲𑀼𑀫𑁂𑀥𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑁂𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 (𑁪--𑁫)

844. ‘𑀲𑀢𑁆𑀣𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀭𑀺𑀬𑀸𑀬𑁂𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 ‘𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 (𑁬--𑁭)

845. ‘𑀮𑀗𑁆𑀕𑀻’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀉𑀓𑁆𑀔𑀺𑀧 𑀮𑀗𑁆𑀕𑀺𑀁, 𑀧𑀚𑀳 𑀅𑀯𑀺𑀚𑁆𑀚𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 (𑁮)

846. ‘𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀬𑀺𑀓𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄𑀥𑀽𑀧𑀸𑀬𑀸𑀲𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀉𑀓𑁆𑀔𑀺𑀧 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀬𑀺𑀓𑀁, 𑀧𑀚𑀳 𑀓𑁄𑀥𑀽𑀧𑀸𑀬𑀸𑀲𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 (𑁯)

847. ‘𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀉𑀓𑁆𑀔𑀺𑀧 𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑀁, 𑀧𑀚𑀳 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 (𑁧𑁦)

848. ‘𑀘𑀗𑁆𑀕𑀯𑀸𑀭𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑁂𑀢𑀁 𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲𑁇 𑀉𑀓𑁆𑀔𑀺𑀧 𑀘𑀗𑁆𑀕𑀯𑀸𑀭𑀁, 𑀧𑀚𑀳 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 (𑁧𑁧)

849. ‘𑀓𑀼𑀫𑁆𑀫𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑁂𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲, 𑀯𑁂𑀤𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲, 𑀲𑀜𑁆𑀜𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲, 𑀲𑀗𑁆𑀔𑀸𑀭𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲𑁇 𑀉𑀓𑁆𑀔𑀺𑀧 𑀓𑀼𑀫𑁆𑀫𑀁, 𑀧𑀚𑀳 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 (𑁧𑁨)

850. ‘𑀅𑀲𑀺𑀲𑀽𑀦𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑁂𑀢𑀁 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀇𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀫𑀦𑀸𑀧𑀸𑀦𑀁 𑀧𑀺𑀬𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸𑀦𑀁 𑀭𑀚𑀦𑀻𑀬𑀸𑀦𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑀦𑀁 𑀕𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑀦𑀁 𑀭𑀲𑀸𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀦𑀁 𑀇𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀫𑀦𑀸𑀧𑀸𑀦𑀁 𑀧𑀺𑀬𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸𑀦𑀁 𑀭𑀚𑀦𑀻𑀬𑀸𑀦𑀁𑁇 𑀉𑀓𑁆𑀔𑀺𑀧 𑀅𑀲𑀺𑀲𑀽𑀦𑀁, 𑀧𑀚𑀳 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 (𑁧𑁩)

851. ‘𑀫𑀁𑀲𑀧𑁂𑀲𑀻’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀉𑀓𑁆𑀔𑀺𑀧 𑀫𑀁𑀲𑀧𑁂𑀲𑀺𑀁, 𑀧𑀚𑀳 𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑀁; 𑀅𑀪𑀺𑀓𑁆𑀔𑀡, 𑀲𑀼𑀫𑁂𑀥, 𑀲𑀢𑁆𑀣𑀁 𑀆𑀤𑀸𑀬𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 (𑁧𑁪)

852. ‘𑀦𑀸𑀕𑁄’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲𑁂𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀦𑀸𑀕𑁄, 𑀫𑀸 𑀦𑀸𑀕𑀁 𑀖𑀝𑁆𑀝𑁂𑀲𑀺; 𑀦𑀫𑁄 𑀓𑀭𑁄𑀳𑀺 𑀦𑀸𑀕𑀲𑁆𑀲𑀸𑀢𑀺 𑀅𑀬𑀫𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺𑁇 (𑁧𑁫)

853. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀻𑀢𑀺𑁇

---

854. 𑀯𑀫𑁆𑀫𑀺𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
