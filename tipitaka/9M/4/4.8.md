# 𑁪𑁇𑁮 𑀫𑀳𑀸𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

(396.)

1282\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀢𑀺𑀲𑁆𑀲 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺—  “𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀲𑀸𑀢𑀺𑀲𑁆𑀲 𑀓𑀺𑀭 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀲𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆’”𑀢𑀺? “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀦𑁆𑀢𑀺 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀺—  “𑀫𑀸 𑀏𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀅𑀯𑀘, 𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺, 𑀦 𑀳𑀺 𑀲𑀸𑀥𑀼 𑀪𑀕𑀯𑀢𑁄 𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁, 𑀦 𑀳𑀺 𑀪𑀕𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀢𑁆𑀣𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑁆𑀪𑀯𑁄”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑁄 𑀢𑀤𑁂𑀯 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀣𑀸𑀫𑀲𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀯𑁄𑀳𑀭𑀢𑀺—  “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇

1283\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀁, 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀸𑀢𑀺𑀲𑁆𑀲 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆’𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀼𑀫𑁆𑀳 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀢𑀺𑀲𑁆𑀲 𑀓𑀺𑀭 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀲𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀫𑁆𑀳; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀫𑁆𑀳—  ‘𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  “𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆”’𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀫𑁆𑀳𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀫𑁆𑀳 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀫𑁆𑀳 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀫𑁆𑀳—  ‘𑀫𑀸 𑀏𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀅𑀯𑀘, 𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺, 𑀦 𑀳𑀺 𑀲𑀸𑀥𑀼 𑀪𑀕𑀯𑀢𑁄 𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁, 𑀦 𑀳𑀺 𑀪𑀕𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀢𑁆𑀣𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑁆𑀪𑀯𑁄’𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑁄 𑀢𑀤𑁂𑀯 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀣𑀸𑀫𑀲𑀸 𑀧𑀭𑀸𑀫𑀲𑀸 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀯𑁄𑀳𑀭𑀢𑀺—  ‘𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆’𑀢𑀺𑁇 𑀬𑀢𑁄 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀫𑁆𑀳 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀁, 𑀅𑀣 𑀫𑀬𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀫𑀸”𑀢𑀺𑁇

1284\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀢𑀺, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀲𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀲𑀸𑀢𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀲𑀸𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆’”𑀢𑀺? “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀣𑀸 𑀢𑀤𑁂𑀯𑀺𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀺 𑀲𑀁𑀲𑀭𑀢𑀺, 𑀅𑀦𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 “𑀓𑀢𑀫𑀁 𑀢𑀁, 𑀲𑀸𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆”𑀢𑀺? “𑀬𑁆𑀯𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀤𑁄 𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑁄 𑀢𑀢𑁆𑀭 𑀢𑀢𑁆𑀭 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀬𑀸 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺? 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀢𑁆𑀣𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑁆𑀪𑀯𑁄𑀢𑀺? 𑀅𑀣 𑀘 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀅𑀫𑁆𑀳𑁂 𑀘𑁂𑀯 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀲𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘 𑀔𑀡𑀲𑀺, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀲𑀺𑁇 𑀢𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

1332\. 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀭𑀸𑀫𑀭𑀡𑀁; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

1334\. “𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑁄, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀯𑁄; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄”𑀢𑀺𑁇

1338\. “𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀨𑀲𑁆𑀲𑁄, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀨𑀲𑁆𑀲𑁄; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑀢𑀺𑁇

1350\. “𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺𑁇

1351\. “𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑁄; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺𑁇

1352\. “𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺𑁇 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺?

1353\. “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺𑁇 “𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦𑁄 𑀯𑀸, 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄; 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺𑁇

1356\. 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀁 𑀯𑀸 𑀧𑀝𑀺𑀥𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀣—  ‘𑀅𑀳𑁂𑀲𑀼𑀫𑁆𑀳 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀦𑀦𑀼 𑀔𑁄 𑀅𑀳𑁂𑀲𑀼𑀫𑁆𑀳 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑁂𑀲𑀼𑀫𑁆𑀳 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑁂𑀲𑀼𑀫𑁆𑀳 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀓𑀺𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀓𑀺𑀁 𑀅𑀳𑁂𑀲𑀼𑀫𑁆𑀳 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀦𑁆’”𑀢𑀺?

1357\. “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1358\. “𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀅𑀧𑀭𑀦𑁆𑀢𑀁 𑀯𑀸 𑀧𑀝𑀺𑀥𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀣—  𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀦𑀦𑀼 𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁, 𑀓𑀺𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀓𑀺𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀦𑁆”𑀢𑀺?

1359\. “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1360\. “𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀏𑀢𑀭𑀳𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀣𑀁𑀓𑀣𑀻 𑀅𑀲𑁆𑀲𑀣—  𑀅𑀳𑀁 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺, 𑀦𑁄 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺, 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺, 𑀅𑀬𑀁 𑀦𑀼 𑀔𑁄 𑀲𑀢𑁆𑀢𑁄 𑀓𑀼𑀢𑁄 𑀆𑀕𑀢𑁄, 𑀲𑁄 𑀓𑀼𑀳𑀺𑀁 𑀕𑀸𑀫𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺?

1361\. “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1362\. “𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀣—  𑀲𑀢𑁆𑀣𑀸 𑀦𑁄 𑀕𑀭𑀼, 𑀲𑀢𑁆𑀣𑀼𑀕𑀸𑀭𑀯𑁂𑀦 𑀘 𑀫𑀬𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫𑀸”𑀢𑀺?

1363\. “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1364\. “𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀣—  𑀲𑀫𑀡𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀲𑀫𑀡𑀸 𑀘 𑀦𑀸𑀫 𑀫𑀬𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫𑀸”𑀢𑀺?

1365\. “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1366\. “𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺?

1367\. “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1368\. “𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀼𑀣𑀼𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀯𑀢 𑀓𑁄𑀢𑀽𑀳𑀮𑀫𑀗𑁆𑀕𑀮𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀸𑀭𑀢𑁄 𑀧𑀘𑁆𑀘𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺?

1369\. “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1370\. “𑀦𑀦𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀤𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀸𑀫𑀁 𑀜𑀸𑀢𑀁 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀸𑀫𑀁 𑀯𑀺𑀤𑀺𑀢𑀁, 𑀢𑀤𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀤𑁂𑀣𑀸”𑀢𑀺𑁇

1371\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

1372\. “𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀦𑀻𑀢𑀸 𑀔𑁄 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀺𑀦𑀸 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀓𑀸𑀮𑀺𑀓𑁂𑀦 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁂𑀦 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁂𑀦 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺𑁇 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺—  𑀇𑀢𑀺 𑀬𑀦𑁆𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇

1373\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀕𑀩𑁆𑀪𑀲𑁆𑀲𑀸𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀇𑀥 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀘 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀫𑀸𑀢𑀸 𑀘 𑀦 𑀉𑀢𑀼𑀦𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄 𑀘 𑀦 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀕𑀩𑁆𑀪𑀲𑁆𑀲𑀸𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀇𑀥 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀘 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀫𑀸𑀢𑀸 𑀘 𑀉𑀢𑀼𑀦𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄 𑀘 𑀦 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀕𑀩𑁆𑀪𑀲𑁆𑀲𑀸𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀘 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀫𑀸𑀢𑀸 𑀘 𑀉𑀢𑀼𑀦𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄 𑀘 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀏𑀯𑀁 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀕𑀩𑁆𑀪𑀲𑁆𑀲𑀸𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀸 𑀦𑀯 𑀯𑀸 𑀤𑀲 𑀯𑀸 𑀫𑀸𑀲𑁂 𑀕𑀩𑁆𑀪𑀁 𑀓𑀼𑀘𑁆𑀙𑀺𑀦𑀸 𑀧𑀭𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀢𑀸 𑀲𑀁𑀲𑀬𑁂𑀦 𑀕𑀭𑀼𑀪𑀸𑀭𑀁𑁇 𑀢𑀫𑁂𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀸 𑀦𑀯𑀦𑁆𑀦𑀁 𑀯𑀸 𑀤𑀲𑀦𑁆𑀦𑀁 𑀯𑀸 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀯𑀺𑀚𑀸𑀬𑀢𑀺 𑀫𑀳𑀢𑀸 𑀲𑀁𑀲𑀬𑁂𑀦 𑀕𑀭𑀼𑀪𑀸𑀭𑀁𑁇 𑀢𑀫𑁂𑀦𑀁 𑀚𑀸𑀢𑀁 𑀲𑀫𑀸𑀦𑀁 𑀲𑀓𑁂𑀦 𑀮𑁄𑀳𑀺𑀢𑁂𑀦 𑀧𑁄𑀲𑁂𑀢𑀺𑁇 𑀮𑁄𑀳𑀺𑀢𑀜𑁆𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀬𑀤𑀺𑀤𑀁 𑀫𑀸𑀢𑀼𑀣𑀜𑁆𑀜𑀁𑁇 𑀲 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀯𑀼𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀯𑀸𑀬 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀸𑀓𑀫𑀦𑁆𑀯𑀸𑀬 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀓𑀼𑀫𑀸𑀭𑀓𑀸𑀦𑀁 𑀓𑀻𑀍𑀆𑀧𑀦𑀓𑀸𑀦𑀺 𑀢𑁂𑀳𑀺 𑀓𑀻𑀍𑀅𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀯𑀗𑁆𑀓𑀓𑀁 𑀖𑀝𑀺𑀓𑀁 𑀫𑁄𑀓𑁆𑀔𑀘𑀺𑀓𑀁 𑀘𑀺𑀗𑁆𑀕𑀼𑀮𑀓𑀁 𑀧𑀢𑁆𑀢𑀸𑀍𑀳𑀓𑀁 𑀭𑀣𑀓𑀁 𑀥𑀦𑀼𑀓𑀁𑁇

1377\. 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀸𑀚𑀻𑀯𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀁 𑀧𑀳𑀸𑀬 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀺𑀳𑀺𑀢𑀤𑀡𑁆𑀟𑁄 𑀦𑀺𑀳𑀺𑀢𑀲𑀢𑁆𑀣𑁄 𑀮𑀚𑁆𑀚𑀻 𑀤𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀪𑀽𑀢𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

1378\. 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀳𑀸𑀬 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀤𑀺𑀦𑁆𑀦𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀻 𑀅𑀣𑁂𑀦𑁂𑀦 𑀲𑀼𑀘𑀺𑀪𑀽𑀢𑁂𑀦 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

1379\. 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀳𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀆𑀭𑀸𑀘𑀸𑀭𑀻 𑀯𑀺𑀭𑀢𑁄 𑀫𑁂𑀣𑀼𑀦𑀸 𑀕𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇

1380\. 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀁 𑀧𑀳𑀸𑀬 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀻 𑀲𑀘𑁆𑀘𑀲𑀦𑁆𑀥𑁄 𑀣𑁂𑀢𑁄 𑀧𑀘𑁆𑀘𑀬𑀺𑀓𑁄 𑀅𑀯𑀺𑀲𑀁𑀯𑀸𑀤𑀓𑁄 𑀮𑁄𑀓𑀲𑁆𑀲𑁇

1381\. 𑀧𑀺𑀲𑀼𑀡𑀁 𑀯𑀸𑀘𑀁 𑀧𑀳𑀸𑀬 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀅𑀫𑀼𑀢𑁆𑀭 𑀅𑀓𑁆𑀔𑀸𑀢𑀸 𑀇𑀫𑁂𑀲𑀁 𑀪𑁂𑀤𑀸𑀬, 𑀅𑀫𑀼𑀢𑁆𑀭 𑀯𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀇𑀫𑁂𑀲𑀁 𑀅𑀓𑁆𑀔𑀸𑀢𑀸 𑀅𑀫𑀽𑀲𑀁 𑀪𑁂𑀤𑀸𑀬𑁇 𑀇𑀢𑀺 𑀪𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀲𑀦𑁆𑀥𑀸𑀢𑀸, 𑀲𑀳𑀺𑀢𑀸𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀢𑀸 𑀲𑀫𑀕𑁆𑀕𑀸𑀭𑀸𑀫𑁄 𑀲𑀫𑀕𑁆𑀕𑀭𑀢𑁄 𑀲𑀫𑀕𑁆𑀕𑀦𑀦𑁆𑀤𑀻, 𑀲𑀫𑀕𑁆𑀕𑀓𑀭𑀡𑀺𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

1382\. 𑀨𑀭𑀼𑀲𑀁 𑀯𑀸𑀘𑀁 𑀧𑀳𑀸𑀬 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀬𑀸 𑀲𑀸 𑀯𑀸𑀘𑀸 𑀦𑁂𑀮𑀸 𑀓𑀡𑁆𑀡𑀲𑀼𑀔𑀸 𑀧𑁂𑀫𑀦𑀻𑀬𑀸 𑀳𑀤𑀬𑀗𑁆𑀕𑀫𑀸 𑀧𑁄𑀭𑀻 𑀩𑀳𑀼𑀚𑀦𑀓𑀦𑁆𑀢𑀸 𑀩𑀳𑀼𑀚𑀦𑀫𑀦𑀸𑀧𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀺𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

1383\. 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀁 𑀧𑀳𑀸𑀬 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀮𑀯𑀸𑀤𑀻 𑀪𑀽𑀢𑀯𑀸𑀤𑀻 𑀅𑀢𑁆𑀣𑀯𑀸𑀤𑀻 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻, 𑀦𑀺𑀥𑀸𑀦𑀯𑀢𑀺𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀺𑀢𑀸 𑀓𑀸𑀮𑁂𑀦, 𑀲𑀸𑀧𑀤𑁂𑀲𑀁 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀯𑀢𑀺𑀁 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁𑁇

1384\. 𑀲𑁄 𑀩𑀻𑀚𑀕𑀸𑀫𑀪𑀽𑀢𑀕𑀸𑀫𑀲𑀫𑀸𑀭𑀫𑁆𑀪𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀪𑀢𑁆𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀭𑀢𑁆𑀢𑀽𑀧𑀭𑀢𑁄, 𑀯𑀺𑀭𑀢𑁄 𑀯𑀺𑀓𑀸𑀮𑀪𑁄𑀚𑀦𑀸𑁇 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑀯𑀸𑀤𑀺𑀢𑀯𑀺𑀲𑀽𑀓𑀤𑀲𑁆𑀲𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀥𑀸𑀭𑀡𑀫𑀡𑁆𑀟𑀦𑀯𑀺𑀪𑀽𑀲𑀦𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀲𑀬𑀦𑀫𑀳𑀸𑀲𑀬𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀫𑀓𑀫𑀁𑀲𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀇𑀢𑁆𑀣𑀺𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑀸𑀲𑀺𑀤𑀸𑀲𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀚𑁂𑀍𑀅𑀓𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀕𑀯𑀸𑀲𑁆𑀲𑀯𑀍𑀅𑀯𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀔𑁂𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑀽𑀢𑁂𑀬𑁆𑀬𑀧𑀳𑀺𑀡𑀕𑀫𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀼𑀮𑀸𑀓𑀽𑀝𑀓𑀁𑀲𑀓𑀽𑀝𑀫𑀸𑀦𑀓𑀽𑀝𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀯𑀜𑁆𑀘𑀦𑀦𑀺𑀓𑀢𑀺𑀲𑀸𑀘𑀺𑀬𑁄𑀕𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀙𑁂𑀤𑀦𑀯𑀥𑀩𑀦𑁆𑀥𑀦𑀯𑀺𑀧𑀭𑀸𑀫𑁄𑀲𑀆𑀮𑁄𑀧𑀲𑀳𑀲𑀸𑀓𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

1385\. 𑀲𑁄 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀓𑀸𑀬𑀧𑀭𑀺𑀳𑀸𑀭𑀺𑀓𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦 𑀓𑀼𑀘𑁆𑀙𑀺𑀧𑀭𑀺𑀳𑀸𑀭𑀺𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀦𑁇 𑀲𑁄 𑀬𑁂𑀦 𑀬𑁂𑀦𑁂𑀯 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀲𑀫𑀸𑀤𑀸𑀬𑁂𑀯 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀓𑁆𑀔𑀻 𑀲𑀓𑀼𑀡𑁄 𑀬𑁂𑀦 𑀬𑁂𑀦𑁂𑀯 𑀟𑁂𑀢𑀺 𑀲𑀧𑀢𑁆𑀢𑀪𑀸𑀭𑁄𑀯 𑀟𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀓𑀸𑀬𑀧𑀭𑀺𑀳𑀸𑀭𑀺𑀓𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦, 𑀓𑀼𑀘𑁆𑀙𑀺𑀧𑀭𑀺𑀳𑀸𑀭𑀺𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀦𑁇 𑀲𑁄 𑀬𑁂𑀦 𑀬𑁂𑀦𑁂𑀯 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀲𑀫𑀸𑀤𑀸𑀬𑁂𑀯 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀲𑁄 𑀇𑀫𑀺𑀦𑀸 𑀅𑀭𑀺𑀬𑁂𑀦 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀦𑀯𑀚𑁆𑀚𑀲𑀼𑀔𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇

1386\. 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀦𑀸𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀭𑀓𑁆𑀔𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀦𑀸𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀭𑀓𑁆𑀔𑀢𑀺 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀇𑀫𑀺𑀦𑀸 𑀅𑀭𑀺𑀬𑁂𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀯𑀭𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀲𑁂𑀓𑀲𑀼𑀔𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇

1389\. 𑀲𑁄 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀳𑀸𑀬 𑀘𑁂𑀢𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀭𑀡𑁂, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

1390\. 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀭𑀽𑀧𑁂 𑀦 𑀲𑀸𑀭𑀚𑁆𑀚𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀭𑀽𑀧𑁂 𑀦 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀓𑀸𑀬𑀲𑀢𑀺 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀘𑁂𑀢𑀲𑁄𑁇 𑀢𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀅𑀦𑀼𑀭𑁄𑀥𑀯𑀺𑀭𑁄𑀥𑀯𑀺𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸, 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀦𑀸𑀪𑀺𑀯𑀤𑀢𑀺 𑀦𑀸𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑁄 𑀅𑀦𑀪𑀺𑀯𑀤𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀦𑀦𑁆𑀤𑀻 𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀦𑀦𑁆𑀤𑀻𑀦𑀺𑀭𑁄𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄, 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄, 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑁄, 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀲𑀸𑀭𑀚𑁆𑀚𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀓𑀸𑀬𑀲𑀢𑀺 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀘𑁂𑀢𑀲𑁄, 𑀢𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀅𑀦𑀼𑀭𑁄𑀥𑀯𑀺𑀭𑁄𑀥𑀯𑀺𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸, 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺 𑀦𑀸𑀪𑀺𑀯𑀤𑀢𑀺 𑀦𑀸𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑁄 𑀅𑀦𑀪𑀺𑀯𑀤𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀦𑀦𑁆𑀤𑀻 𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀦𑀦𑁆𑀤𑀻𑀦𑀺𑀭𑁄𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄, 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄, 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑁄, 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀥𑀸𑀭𑁂𑀣, 𑀲𑀸𑀢𑀺𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀧𑀼𑀢𑁆𑀢𑀁 𑀫𑀳𑀸𑀢𑀡𑁆𑀳𑀸𑀚𑀸𑀮𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀖𑀸𑀝𑀧𑁆𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑀦𑁆”𑀢𑀺𑁇

1391\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇
