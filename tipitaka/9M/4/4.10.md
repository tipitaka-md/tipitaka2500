# 𑁪𑁇𑁧𑁦 𑀘𑀽𑀍𑀅𑀅𑀲𑁆𑀲𑀧𑀼𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀅𑀲𑁆𑀲𑀧𑀼𑀭𑀲𑀼𑀢𑁆𑀢

(435.)

1428\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀅𑀗𑁆𑀕𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀲𑁆𑀲𑀧𑀼𑀭𑀁 𑀦𑀸𑀫 𑀅𑀗𑁆𑀕𑀸𑀦𑀁 𑀦𑀺𑀕𑀫𑁄𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1429\. “𑀲𑀫𑀡𑀸 𑀲𑀫𑀡𑀸𑀢𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀦𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀼𑀫𑁆𑀳𑁂 𑀘 𑀧𑀦 ‘𑀓𑁂 𑀢𑀼𑀫𑁆𑀳𑁂’𑀢𑀺 𑀧𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸 ‘𑀲𑀫𑀡𑀸𑀫𑁆𑀳𑀸’𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀣𑁇 𑀢𑁂𑀲𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁𑀲𑀫𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀢𑀁 𑀏𑀯𑀁𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀢𑀁—  ‘𑀬𑀸 𑀲𑀫𑀡𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫; 𑀏𑀯𑀁 𑀦𑁄 𑀅𑀬𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀫𑀜𑁆𑀜𑀸 𑀘 𑀲𑀘𑁆𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀜𑁆𑀜𑀸 𑀘 𑀪𑀽𑀢𑀸; 𑀬𑁂𑀲𑀜𑁆𑀘 𑀫𑀬𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀸𑀫, 𑀢𑁂𑀲𑀁 𑀢𑁂 𑀓𑀸𑀭𑀸 𑀅𑀫𑁆𑀳𑁂𑀲𑀼 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸, 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁂𑀯𑀸𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀅𑀯𑀜𑁆𑀛𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀨𑀮𑀸 𑀲𑁅𑀤𑁆𑀭𑀬𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(436.)

1430\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀫𑀡𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼𑀲𑁆𑀲 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀓𑁄𑀥𑀦𑀲𑁆𑀲 𑀓𑁄𑀥𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑀦𑀸𑀳𑀺𑀲𑁆𑀲 𑀉𑀧𑀦𑀸𑀳𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀓𑁆𑀔𑀺𑀲𑁆𑀲 𑀫𑀓𑁆𑀔𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀍𑀆𑀲𑀺𑀲𑁆𑀲 𑀧𑀍𑀆𑀲𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀇𑀲𑁆𑀲𑀼𑀓𑀺𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀘𑁆𑀙𑀭𑀺𑀲𑁆𑀲 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀞𑀲𑁆𑀲 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀸𑀬𑀸𑀯𑀺𑀲𑁆𑀲 𑀫𑀸𑀬𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀓𑀸 𑀇𑀘𑁆𑀙𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀲𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀫𑀮𑀸𑀦𑀁 𑀲𑀫𑀡𑀤𑁄𑀲𑀸𑀦𑀁 𑀲𑀫𑀡𑀓𑀲𑀝𑀸𑀦𑀁 𑀆𑀧𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑁂𑀤𑀦𑀺𑀬𑀸𑀦𑀁 𑀅𑀧𑁆𑀧𑀳𑀸𑀦𑀸 ‘𑀦 𑀲𑀫𑀡𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀢𑀚𑀁 𑀦𑀸𑀫 𑀆𑀯𑀼𑀥𑀚𑀸𑀢𑀁 𑀉𑀪𑀢𑁄𑀥𑀸𑀭𑀁 𑀧𑀻𑀢𑀦𑀺𑀲𑀺𑀢𑀁𑁇 𑀢𑀤𑀲𑁆𑀲 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀬𑀸 𑀲𑀫𑁆𑀧𑀸𑀭𑀼𑀢𑀁 𑀲𑀫𑁆𑀧𑀮𑀺𑀯𑁂𑀞𑀺𑀢𑀁𑁇 𑀢𑀣𑀽𑀧𑀫𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

1431\. 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀲𑁆𑀲 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀥𑀸𑀭𑀡𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁂𑀮𑀓𑀲𑁆𑀲 𑀅𑀘𑁂𑀮𑀓𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀺𑀓𑀲𑁆𑀲 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀺𑀓𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀓𑁄𑀭𑁄𑀳𑀓𑀲𑁆𑀲 𑀉𑀤𑀓𑁄𑀭𑁄𑀳𑀡𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀺𑀓𑀲𑁆𑀲 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀺𑀓𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀺𑀓𑀲𑁆𑀲 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀺𑀓𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀩𑁆𑀪𑀝𑁆𑀞𑀓𑀲𑁆𑀲 𑀉𑀩𑁆𑀪𑀝𑁆𑀞𑀓𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑀪𑀢𑁆𑀢𑀺𑀓𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑀸𑀬𑀪𑀢𑁆𑀢𑀺𑀓𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁆𑀢𑀚𑁆𑀛𑀸𑀬𑀓𑀲𑁆𑀲 𑀫𑀦𑁆𑀢𑀚𑁆𑀛𑀸𑀬𑀓𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀝𑀺𑀮𑀓𑀲𑁆𑀲 𑀚𑀝𑀸𑀥𑀸𑀭𑀡𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

1436\. 𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀸 𑀘𑁂𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀸 𑀘𑁂𑀧𑀺…𑀧𑁂…  𑀯𑁂𑀲𑁆𑀲𑀓𑀼𑀮𑀸 𑀘𑁂𑀧𑀺…  𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑀸 𑀘𑁂𑀧𑀺…  𑀬𑀲𑁆𑀫𑀸 𑀓𑀲𑁆𑀫𑀸 𑀘𑁂𑀧𑀺 𑀓𑀼𑀮𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1437\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

1439\. 𑀫𑀳𑀸𑀬𑀫𑀓𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇

1440\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1441\. _𑀕𑀺𑀜𑁆𑀚𑀓𑀲𑀸𑀮𑀯𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀢𑀼𑀁,_
