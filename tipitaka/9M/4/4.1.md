# 𑁪𑁇𑁧 𑀘𑀽𑀍𑀅𑀕𑁄𑀲𑀺𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀕𑁄𑀲𑀺𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(325.)

1122\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄 𑀕𑁄𑀲𑀺𑀗𑁆𑀕𑀲𑀸𑀮𑀯𑀦𑀤𑀸𑀬𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦 𑀕𑁄𑀲𑀺𑀗𑁆𑀕𑀲𑀸𑀮𑀯𑀦𑀤𑀸𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀤𑀸𑀬𑀧𑀸𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸, 𑀲𑀫𑀡, 𑀏𑀢𑀁 𑀤𑀸𑀬𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀲𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀭𑀽𑀧𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀫𑀸 𑀢𑁂𑀲𑀁 𑀅𑀨𑀸𑀲𑀼𑀫𑀓𑀸𑀲𑀻”𑀢𑀺𑁇

1123\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀤𑀸𑀬𑀧𑀸𑀮𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀫𑀦𑁆𑀢𑀬𑀫𑀸𑀦𑀲𑁆𑀲𑁇 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀤𑀸𑀬𑀧𑀸𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀤𑀸𑀬𑀧𑀸𑀮, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀸𑀭𑁂𑀲𑀺𑁇 𑀲𑀢𑁆𑀣𑀸 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀓𑀺𑀫𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀲𑀢𑁆𑀣𑀸 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸—  𑀏𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺, 𑀏𑀓𑁄 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺, 𑀏𑀓𑁄 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀲𑀺𑁇 𑀢𑁂𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1124\. “𑀓𑀘𑁆𑀘𑀺 𑀯𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀦 𑀓𑀺𑀮𑀫𑀣𑀸”𑀢𑀺? “𑀔𑀫𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸, 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸; 𑀦 𑀘 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀓𑀘𑁆𑀘𑀺 𑀧𑀦 𑀯𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀔𑀻𑀭𑁄𑀤𑀓𑀻𑀪𑀽𑀢𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀘𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣𑀸”𑀢𑀺? “𑀢𑀕𑁆𑀖 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀔𑀻𑀭𑁄𑀤𑀓𑀻𑀪𑀽𑀢𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀘𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀔𑀻𑀭𑁄𑀤𑀓𑀻𑀪𑀽𑀢𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀘𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣𑀸”𑀢𑀺? “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀬𑁄𑀳𑀁 𑀏𑀯𑀭𑀽𑀧𑁂𑀳𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑁂𑀲𑀼 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂𑀲𑀼 𑀫𑁂𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘; 𑀫𑁂𑀢𑁆𑀢𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘; 𑀫𑁂𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀲𑁂𑀦 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀲𑁂𑀦 𑀯𑀢𑁆𑀢𑀸𑀫𑀺𑁇 𑀦𑀸𑀦𑀸 𑀳𑀺 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀬𑀸 𑀏𑀓𑀜𑁆𑀘 𑀧𑀦 𑀫𑀜𑁆𑀜𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

1125\. 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀓𑀺𑀫𑀺𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀬𑁄𑀳𑀁 𑀏𑀯𑀭𑀽𑀧𑁂𑀳𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑁂𑀲𑀼 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂𑀲𑀼 𑀫𑁂𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘, 𑀫𑁂𑀢𑁆𑀢𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘, 𑀫𑁂𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀲𑁂𑀦 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀲𑁂𑀦 𑀯𑀢𑁆𑀢𑀸𑀫𑀺𑁇 𑀦𑀸𑀦𑀸 𑀳𑀺 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀬𑀸 𑀏𑀓𑀜𑁆𑀘 𑀧𑀦 𑀫𑀜𑁆𑀜𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

1131\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸𑁇 𑀏𑀢𑀲𑁆𑀲 𑀧𑀦 𑀯𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺? “𑀓𑀺𑀜𑁆𑀳𑀺 𑀦𑁄 𑀲𑀺𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀇𑀥 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑁇 𑀏𑀢𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀬𑀫𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺𑁇

1132\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸𑁇 𑀏𑀢𑀲𑁆𑀲 𑀧𑀦 𑀯𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺? “𑀓𑀺𑀜𑁆𑀳𑀺 𑀦𑁄 𑀲𑀺𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀇𑀥 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑁇 𑀏𑀢𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀬𑀫𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺𑁇

1133\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸𑁇 𑀏𑀢𑀲𑁆𑀲 𑀧𑀦 𑀯𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺? “𑀓𑀺𑀜𑁆𑀳𑀺 𑀦𑁄 𑀲𑀺𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀇𑀥 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑁇 𑀏𑀢𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀬𑀫𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺𑁇

1134\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸𑁇 𑀏𑀢𑀲𑁆𑀲 𑀧𑀦 𑀯𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺? “𑀓𑀺𑀜𑁆𑀳𑀺 𑀦𑁄 𑀲𑀺𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀇𑀥 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀸𑀫 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀦𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸𑁇 𑀏𑀢𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀏𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀬𑀫𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄𑁇 𑀇𑀫𑀫𑁆𑀳𑀸 𑀘 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑀁 𑀯𑀸 𑀧𑀡𑀻𑀢𑀢𑀭𑀁 𑀯𑀸 𑀦 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀸𑁇 𑀇𑀫𑀫𑁆𑀳𑀸 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄 𑀯𑀸 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀯𑀸 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀦𑀢𑁆𑀣𑀻”𑀢𑀺𑁇

1135\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀓𑀺𑀫𑀺𑀮𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀲𑀁𑀬𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀯𑀫𑀸𑀭𑁄𑀘𑀺𑀫𑁆𑀳—  ‘𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀫𑀬𑀁 𑀮𑀸𑀪𑀺𑀦𑁄’𑀢𑀺, 𑀬𑀁 𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀬𑀸𑀯 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀧𑀓𑀸𑀲𑁂𑀢𑀻”𑀢𑀺? “𑀦 𑀔𑁄 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀫𑀸𑀭𑁄𑀘𑁂𑀲𑀼𑀁—  ‘𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀫𑀬𑀁 𑀮𑀸𑀪𑀺𑀦𑁄’𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀯𑀺𑀤𑀺𑀢𑁄—  ‘𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀇𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀮𑀸𑀪𑀺𑀦𑁄’𑀢𑀺𑁇 𑀤𑁂𑀯𑀢𑀸𑀧𑀺 𑀫𑁂 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁—  ‘𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀇𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀮𑀸𑀪𑀺𑀦𑁄’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀳𑀸𑀪𑀺𑀧𑀼𑀝𑁆𑀞𑁂𑀦 𑀩𑁆𑀬𑀸𑀓𑀢𑀦𑁆”𑀢𑀺𑁇

1136\. 𑀅𑀣 𑀔𑁄 𑀤𑀻𑀖𑁄 𑀧𑀭𑀚𑀦𑁄 𑀬𑀓𑁆𑀔𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀤𑀻𑀖𑁄 𑀧𑀭𑀚𑀦𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀮𑀸𑀪𑀸 𑀯𑀢, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀚𑁆𑀚𑀻𑀦𑀁, 𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑀸 𑀯𑀚𑁆𑀚𑀺𑀧𑀚𑀸𑀬, 𑀬𑀢𑁆𑀣 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀇𑀫𑁂 𑀘 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸—  𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄”𑀢𑀺𑁇 𑀤𑀻𑀖𑀲𑁆𑀲 𑀧𑀭𑀚𑀦𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀼𑀫𑁆𑀫𑀸 𑀤𑁂𑀯𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁—  “𑀮𑀸𑀪𑀸 𑀯𑀢, 𑀪𑁄, 𑀯𑀚𑁆𑀚𑀻𑀦𑀁, 𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑀸 𑀯𑀚𑁆𑀚𑀺𑀧𑀚𑀸𑀬, 𑀬𑀢𑁆𑀣 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀇𑀫𑁂 𑀘 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸—  𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄”𑀢𑀺𑁇 𑀪𑀼𑀫𑁆𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀬𑀸𑀫𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀢𑀼𑀲𑀺𑀢𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁—  “𑀮𑀸𑀪𑀸 𑀯𑀢, 𑀪𑁄, 𑀯𑀚𑁆𑀚𑀻𑀦𑀁, 𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑀸 𑀯𑀚𑁆𑀚𑀺𑀧𑀚𑀸𑀬, 𑀬𑀢𑁆𑀣 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀇𑀫𑁂 𑀘 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸—  𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀺𑀫𑀺𑀮𑁄”𑀢𑀺𑁇 𑀇𑀢𑀺𑀳 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀢𑁂𑀦 𑀔𑀡𑁂𑀦 𑀢𑁂𑀦 𑀮𑀬𑁂𑀦 𑀢𑁂𑀦 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑁂𑀦 𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸 𑀯𑀺𑀤𑀺𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

1137\. “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀤𑀻𑀖, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀤𑀻𑀖𑁇 𑀬𑀲𑁆𑀫𑀸𑀧𑀺, 𑀤𑀻𑀖, 𑀓𑀼𑀮𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀢𑀜𑁆𑀘𑁂𑀧𑀺 𑀓𑀼𑀮𑀁 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀧𑀸𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀬𑀲𑁆𑀫𑀸𑀧𑀺, 𑀤𑀻𑀖, 𑀓𑀼𑀮𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀲𑁄 𑀘𑁂𑀧𑀺 𑀓𑀼𑀮𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑁄 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀧𑀸𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀬𑀲𑁆𑀫𑀸𑀧𑀺, 𑀤𑀻𑀖, 𑀕𑀸𑀫𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀲𑁄 𑀘𑁂𑀧𑀺 𑀕𑀸𑀫𑁄 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀧𑀸𑀲𑁆𑀲 𑀕𑀸𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀬𑀲𑁆𑀫𑀸𑀧𑀺, 𑀤𑀻𑀖, 𑀦𑀺𑀕𑀫𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀲𑁄 𑀘𑁂𑀧𑀺 𑀦𑀺𑀕𑀫𑁄 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀧𑀸𑀲𑁆𑀲 𑀦𑀺𑀕𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀬𑀲𑁆𑀫𑀸𑀧𑀺, 𑀤𑀻𑀖, 𑀦𑀕𑀭𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀢𑀜𑁆𑀘𑁂𑀧𑀺 𑀦𑀕𑀭𑀁 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀧𑀸𑀲𑁆𑀲 𑀦𑀕𑀭𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀬𑀲𑁆𑀫𑀸𑀧𑀺, 𑀤𑀻𑀖, 𑀚𑀦𑀧𑀤𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀲𑁄 𑀘𑁂𑀧𑀺 𑀚𑀦𑀧𑀤𑁄 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀧𑀸𑀲𑁆𑀲 𑀚𑀦𑀧𑀤𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀤𑀻𑀖, 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂𑀲𑀸𑀦𑀫𑁆𑀧𑀸𑀲𑁆𑀲 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀤𑀻𑀖, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸…𑀧𑁂…  𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀤𑀻𑀖, 𑀯𑁂𑀲𑁆𑀲𑀸…𑀧𑁂…  𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀤𑀻𑀖, 𑀲𑀼𑀤𑁆𑀤𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂𑀲𑀸𑀦𑀫𑁆𑀧𑀸𑀲𑁆𑀲 𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀲𑀤𑁂𑀯𑀓𑁄 𑀘𑁂𑀧𑀺, 𑀤𑀻𑀖, 𑀮𑁄𑀓𑁄 𑀲𑀫𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁄 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀧𑀚𑀸 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬, 𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲𑀧𑀸𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲 𑀲𑀫𑀸𑀭𑀓𑀲𑁆𑀲 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀲𑁆𑀲 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀧𑀲𑁆𑀲, 𑀤𑀻𑀖, 𑀬𑀸𑀯 𑀏𑀢𑁂 𑀢𑀬𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬, 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

1138\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀤𑀻𑀖𑁄 𑀧𑀭𑀚𑀦𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀻𑀢𑀺𑁇
