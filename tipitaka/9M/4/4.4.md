# 𑁪𑁇𑁪 𑀘𑀽𑀍𑀅𑀕𑁄𑀧𑀸𑀮𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀘𑀽𑀍𑀅𑀕𑁄𑀧𑀸𑀮𑀓𑀲𑀼𑀢𑁆𑀢

(350.)

1186\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑀚𑁆𑀚𑀻𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀓𑁆𑀓𑀘𑁂𑀮𑀸𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀢𑀻𑀭𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1187\. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀕𑀥𑀓𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀚𑀸𑀢𑀺𑀓𑁄, 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀲𑀭𑀤𑀲𑀫𑀬𑁂, 𑀅𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀅𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀅𑀢𑀺𑀢𑁆𑀣𑁂𑀦𑁂𑀯 𑀕𑀸𑀯𑁄 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑀁 𑀢𑀻𑀭𑀁 𑀲𑀼𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀸𑀯𑁄 𑀫𑀚𑁆𑀛𑁂𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀲𑁄𑀢𑁂 𑀆𑀫𑀡𑁆𑀟𑀮𑀺𑀬𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀕𑀥𑀓𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀚𑀸𑀢𑀺𑀓𑁄, 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀲𑀭𑀤𑀲𑀫𑀬𑁂, 𑀅𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀅𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀅𑀢𑀺𑀢𑁆𑀣𑁂𑀦𑁂𑀯 𑀕𑀸𑀯𑁄 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑀁 𑀢𑀻𑀭𑀁 𑀲𑀼𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀇𑀫𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀧𑀭𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀅𑀓𑀼𑀲𑀮𑀸 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀅𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲, 𑀅𑀓𑀼𑀲𑀮𑀸 𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀅𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲, 𑀢𑁂𑀲𑀁 𑀬𑁂 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀲𑀤𑁆𑀤𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇

(351.)

1188\. 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀕𑀥𑀓𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀚𑀸𑀢𑀺𑀓𑁄, 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀲𑀭𑀤𑀲𑀫𑀬𑁂, 𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀢𑀺𑀢𑁆𑀣𑁂𑀦𑁂𑀯 𑀕𑀸𑀯𑁄 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑀁 𑀢𑀻𑀭𑀁 𑀲𑀼𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁𑁇 𑀲𑁄 𑀧𑀞𑀫𑀁 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀬𑁂 𑀢𑁂 𑀉𑀲𑀪𑀸 𑀕𑁄𑀧𑀺𑀢𑀭𑁄 𑀕𑁄𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀸𑁇 𑀢𑁂 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀅𑀣𑀸𑀧𑀭𑁂 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀩𑀮𑀯𑀕𑀸𑀯𑁄 𑀤𑀫𑁆𑀫𑀕𑀸𑀯𑁄𑁇 𑀢𑁂𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀅𑀣𑀸𑀧𑀭𑁂 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀯𑀘𑁆𑀙𑀢𑀭𑁂 𑀯𑀘𑁆𑀙𑀢𑀭𑀺𑀬𑁄𑁇 𑀢𑁂𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀅𑀣𑀸𑀧𑀭𑁂 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀯𑀘𑁆𑀙𑀓𑁂 𑀓𑀺𑀲𑀸𑀩𑀮𑀓𑁂𑁇 𑀢𑁂𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀘𑁆𑀙𑀓𑁄 𑀢𑀭𑀼𑀡𑀓𑁄 𑀢𑀸𑀯𑀤𑁂𑀯 𑀚𑀸𑀢𑀓𑁄 𑀫𑀸𑀢𑀼𑀕𑁄𑀭𑀯𑀓𑁂𑀦 𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑁄, 𑀲𑁄𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀕𑀥𑀓𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀚𑀸𑀢𑀺𑀓𑁄, 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀲𑀭𑀤𑀲𑀫𑀬𑁂, 𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀲𑀫𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀢𑀺𑀢𑁆𑀣𑁂𑀦𑁂𑀯 𑀕𑀸𑀯𑁄 𑀧𑀢𑀸𑀭𑁂𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑀁 𑀢𑀻𑀭𑀁 𑀲𑀼𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀇𑀫𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀧𑀭𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀓𑀼𑀲𑀮𑀸 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀅𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲, 𑀓𑀼𑀲𑀮𑀸 𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀅𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲, 𑀢𑁂𑀲𑀁 𑀬𑁂 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀲𑀤𑁆𑀤𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇

1191\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀘𑁆𑀙𑀢𑀭𑀸 𑀯𑀘𑁆𑀙𑀢𑀭𑀺𑀬𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀁𑀲𑀼; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀸𑀦𑀁 𑀢𑀦𑀼𑀢𑁆𑀢𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄 𑀲𑀓𑀺𑀤𑁂𑀯 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

1192\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀘𑁆𑀙𑀓𑀸 𑀓𑀺𑀲𑀸𑀩𑀮𑀓𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀁𑀲𑀼; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑀸, 𑀢𑁂𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

1193\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀘𑁆𑀙𑀓𑁄 𑀢𑀭𑀼𑀡𑀓𑁄 𑀢𑀸𑀯𑀤𑁂𑀯 𑀚𑀸𑀢𑀓𑁄 𑀫𑀸𑀢𑀼𑀕𑁄𑀭𑀯𑀓𑁂𑀦 𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀅𑀕𑀫𑀸𑀲𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄 𑀲𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄, 𑀢𑁂𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀲𑁄𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀧𑀸𑀭𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

1194\. 𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀲𑀮𑁄 𑀇𑀫𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑁄 𑀧𑀭𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀓𑀼𑀲𑀮𑁄 𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑁄 𑀅𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲, 𑀓𑀼𑀲𑀮𑁄 𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑁄 𑀅𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀲𑀤𑁆𑀤𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

1195\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

1196\. _“𑀅𑀬𑀁 𑀮𑁄𑀓𑁄 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄,_  
_𑀚𑀸𑀦𑀢𑀸 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑁄;_  
_𑀬𑀜𑁆𑀘 𑀫𑀸𑀭𑁂𑀦 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀁 𑀬𑀜𑁆𑀘 𑀫𑀘𑁆𑀘𑀼𑀦𑀸𑁇_  


1197\. _𑀲𑀩𑁆𑀩𑀁 𑀮𑁄𑀓𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀚𑀸𑀦𑀢𑀸;_  
_𑀯𑀺𑀯𑀝𑀁 𑀅𑀫𑀢𑀤𑁆𑀯𑀸𑀭𑀁,_  
_𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


1198\. _𑀙𑀺𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀲𑁄𑀢𑀁,_  
_𑀯𑀺𑀤𑁆𑀥𑀲𑁆𑀢𑀁 𑀯𑀺𑀦𑀍𑀈𑀓𑀢𑀁;_  
_𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀩𑀳𑀼𑀮𑀸 𑀳𑁄𑀣,_  
_𑀔𑁂𑀫𑀁 𑀧𑀢𑁆𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇_  


---

1199\. 𑀘𑀽𑀍𑀅𑀕𑁄𑀧𑀸𑀮𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
