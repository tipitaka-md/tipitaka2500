# 𑁪𑁇𑁫 𑀘𑀽𑀍𑀅𑀲𑀘𑁆𑀘𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀲𑀘𑁆𑀘𑀓𑀲𑀼𑀢𑁆𑀢

(353.)

1200\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀪𑀲𑁆𑀲𑀧𑁆𑀧𑀯𑀸𑀤𑀓𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑀸𑀤𑁄 𑀲𑀸𑀥𑀼𑀲𑀫𑁆𑀫𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀲𑁆𑀲𑁇 𑀲𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀭𑀺𑀲𑀢𑀺 𑀏𑀯𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀢𑀺—  “𑀦𑀸𑀳𑀁 𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀫𑀡𑀁 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀯𑀸, 𑀲𑀁𑀖𑀺𑀁 𑀕𑀡𑀺𑀁 𑀕𑀡𑀸𑀘𑀭𑀺𑀬𑀁, 𑀅𑀧𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑀁, 𑀬𑁄 𑀫𑀬𑀸 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑁄 𑀦 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑁂𑀬𑁆𑀬, 𑀬𑀲𑁆𑀲 𑀦 𑀓𑀘𑁆𑀙𑁂𑀳𑀺 𑀲𑁂𑀤𑀸 𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀣𑀽𑀡𑀁 𑀘𑁂𑀧𑀸𑀳𑀁 𑀅𑀘𑁂𑀢𑀦𑀁 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀲𑀫𑀸𑀭𑀪𑁂𑀬𑁆𑀬𑀁, 𑀲𑀸𑀧𑀺 𑀫𑀬𑀸 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑁂𑀬𑁆𑀬𑁇 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲𑀸”𑀢𑀺?

1201\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀚𑀺 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀚𑀗𑁆𑀖𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀦𑀼𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀫𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀚𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀅𑀲𑁆𑀲𑀚𑀺𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀣𑀁 𑀧𑀦, 𑀪𑁄 𑀅𑀲𑁆𑀲𑀚𑀺, 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀸𑀯𑀓𑁂 𑀯𑀺𑀦𑁂𑀢𑀺, 𑀓𑀣𑀁𑀪𑀸𑀕𑀸 𑀘 𑀧𑀦 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁂𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀩𑀳𑀼𑀮𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺? “𑀏𑀯𑀁 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀓𑁂 𑀯𑀺𑀦𑁂𑀢𑀺, 𑀏𑀯𑀁𑀪𑀸𑀕𑀸 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑁂𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀩𑀳𑀼𑀮𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀺—  ‘𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁𑁇 𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀲𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀢𑁆𑀢𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀓𑁂 𑀯𑀺𑀦𑁂𑀢𑀺, 𑀏𑀯𑀁𑀪𑀸𑀕𑀸 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑁂𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀩𑀳𑀼𑀮𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇 “𑀤𑀼𑀲𑁆𑀲𑀼𑀢𑀁 𑀯𑀢, 𑀪𑁄 𑀅𑀲𑁆𑀲𑀚𑀺, 𑀅𑀲𑁆𑀲𑀼𑀫𑁆𑀳 𑀬𑁂 𑀫𑀬𑀁 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀅𑀲𑁆𑀲𑀼𑀫𑁆𑀳𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀫𑀬𑀁 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀢𑁂𑀦 𑀪𑁄𑀢𑀸 𑀕𑁄𑀢𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀲𑀺𑀬𑀸 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇

1202\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀲𑀢𑀸𑀦𑀺 𑀲𑀦𑁆𑀣𑀸𑀕𑀸𑀭𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀻, 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀻, 𑀅𑀚𑁆𑀚 𑀫𑁂 𑀲𑀫𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀘𑁂 𑀫𑁂 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀢𑀣𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀢𑀺 𑀬𑀣𑀸 𑀘 𑀫𑁂 𑀜𑀸𑀢𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀲𑀸𑀯𑀓𑁂𑀦 𑀅𑀲𑁆𑀲𑀚𑀺𑀦𑀸 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀻𑀖𑀮𑁄𑀫𑀺𑀓𑀁 𑀏𑀍𑀅𑀓𑀁 𑀮𑁄𑀫𑁂𑀲𑀼 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀸𑀳𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀆𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑁄 𑀫𑀳𑀦𑁆𑀢𑀁 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀸𑀓𑀺𑀍𑀅𑀜𑁆𑀚𑀁 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀉𑀤𑀓𑀭𑀳𑀤𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀓𑀡𑁆𑀡𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀸𑀳𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀆𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀸𑀥𑀼𑀢𑁆𑀢𑁄 𑀯𑀸𑀮𑀁 𑀓𑀡𑁆𑀡𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀑𑀥𑀼𑀦𑁂𑀬𑁆𑀬 𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑁂𑀬𑁆𑀬 𑀦𑀺𑀧𑁆𑀨𑁄𑀝𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀸𑀳𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀑𑀥𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀺𑀧𑁆𑀨𑁄𑀝𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀓𑀼𑀜𑁆𑀚𑀭𑁄 𑀲𑀝𑁆𑀞𑀺𑀳𑀸𑀬𑀦𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀡𑀥𑁄𑀯𑀺𑀓𑀁 𑀦𑀸𑀫 𑀓𑀻𑀍𑀇𑀢𑀚𑀸𑀢𑀁 𑀓𑀻𑀍𑀅𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀸𑀳𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀸𑀡𑀥𑁄𑀯𑀺𑀓𑀁 𑀫𑀜𑁆𑀜𑁂 𑀓𑀻𑀍𑀇𑀢𑀚𑀸𑀢𑀁 𑀓𑀻𑀍𑀇𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀻, 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀻, 𑀅𑀚𑁆𑀚 𑀫𑁂 𑀲𑀫𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑁂𑀓𑀘𑁆𑀘𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀁 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀘𑁆𑀘𑀓𑀲𑁆𑀲 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸𑀤𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀯𑀸𑀤𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? 𑀏𑀓𑀘𑁆𑀘𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀁 𑀲𑁄 𑀪𑀯𑀫𑀸𑀦𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁄 𑀪𑀕𑀯𑀢𑁄 𑀯𑀸𑀤𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀘𑁆𑀘𑀓𑀲𑁆𑀲 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸𑀤𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀲𑀢𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀬𑁂𑀦 𑀫𑀳𑀸𑀯𑀦𑀁 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇

1204\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀳𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀓𑀺𑀜𑁆𑀘𑀺𑀤𑁂𑀯 𑀤𑁂𑀲𑀁, 𑀲𑀘𑁂 𑀫𑁂 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀑𑀓𑀸𑀲𑀁 𑀓𑀭𑁄𑀢𑀺 𑀧𑀜𑁆𑀳𑀲𑁆𑀲 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀧𑀼𑀘𑁆𑀙, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀤𑀸𑀓𑀗𑁆𑀔𑀲𑀻”𑀢𑀺𑁇 “𑀓𑀣𑀁 𑀧𑀦 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀲𑀸𑀯𑀓𑁂 𑀯𑀺𑀦𑁂𑀢𑀺, 𑀓𑀣𑀁𑀪𑀸𑀕𑀸 𑀘 𑀧𑀦 𑀪𑁄𑀢𑁄 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁂𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀩𑀳𑀼𑀮𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺? “𑀏𑀯𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀲𑀸𑀯𑀓𑁂 𑀯𑀺𑀦𑁂𑀫𑀺, 𑀏𑀯𑀁𑀪𑀸𑀕𑀸 𑀘 𑀧𑀦 𑀫𑁂 𑀲𑀸𑀯𑀓𑁂𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀩𑀳𑀼𑀮𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀺—  ‘𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁𑁇 𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀢𑁆𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀢𑁆𑀢𑀸𑁇 𑀲𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀢𑁆𑀢𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀲𑀸𑀯𑀓𑁂 𑀯𑀺𑀦𑁂𑀫𑀺, 𑀏𑀯𑀁𑀪𑀸𑀕𑀸 𑀘 𑀧𑀦 𑀫𑁂 𑀲𑀸𑀯𑀓𑁂𑀲𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀩𑀳𑀼𑀮𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇

1209\. “𑀢𑁂𑀦 𑀳𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬 𑀢𑀣𑀸 𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬 𑀭𑀜𑁆𑀜𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀓𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀚𑀺𑀢𑁂 𑀯𑀲𑁄—  𑀖𑀸𑀢𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀖𑀸𑀢𑁂𑀢𑀼𑀁, 𑀚𑀸𑀧𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀚𑀸𑀧𑁂𑀢𑀼𑀁, 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀯𑀸 𑀧𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼𑀲𑁆𑀲 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺? “𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀭𑀜𑁆𑀜𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀓𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀚𑀺𑀢𑁂 𑀯𑀲𑁄—  𑀖𑀸𑀢𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀖𑀸𑀢𑁂𑀢𑀼𑀁, 𑀚𑀸𑀧𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀚𑀸𑀧𑁂𑀢𑀼𑀁, 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀯𑀸 𑀧𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼𑀲𑁆𑀲 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲𑁇 𑀇𑀫𑁂𑀲𑀫𑁆𑀧𑀺 𑀳𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀁𑀖𑀸𑀦𑀁 𑀕𑀡𑀸𑀦𑀁—  𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀯𑀚𑁆𑀚𑀻𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁—  𑀯𑀢𑁆𑀢𑀢𑀺 𑀲𑀓𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀚𑀺𑀢𑁂 𑀯𑀲𑁄—  𑀖𑀸𑀢𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀖𑀸𑀢𑁂𑀢𑀼𑀁, 𑀚𑀸𑀧𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀚𑀸𑀧𑁂𑀢𑀼𑀁, 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀸𑀬𑀁 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼𑀁𑁇 𑀓𑀺𑀁 𑀧𑀦 𑀭𑀜𑁆𑀜𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀯𑀸 𑀧𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼𑀲𑁆𑀲 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲? 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀯𑀢𑁆𑀢𑀺𑀢𑀼𑀜𑁆𑀘 𑀫𑀭𑀳𑀢𑀻”𑀢𑀺𑁇

1210\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  ‘𑀭𑀽𑀧𑀁 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀭𑀽𑀧𑁂 𑀯𑀲𑁄—  𑀏𑀯𑀁 𑀫𑁂 𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀼, 𑀏𑀯𑀁 𑀫𑁂 𑀭𑀽𑀧𑀁 𑀫𑀸 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀘𑁆𑀘𑀓𑀁 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  ‘𑀭𑀽𑀧𑀁 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀭𑀽𑀧𑁂 𑀯𑀲𑁄—  𑀏𑀯𑀁 𑀫𑁂 𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀼, 𑀏𑀯𑀁 𑀫𑁂 𑀭𑀽𑀧𑀁 𑀫𑀸 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀘𑁆𑀘𑀓𑀁 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺 𑀤𑀸𑀦𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀦 𑀤𑀸𑀦𑀺 𑀢𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑀲𑁆𑀲 𑀓𑀸𑀮𑁄𑁇 𑀬𑁄 𑀓𑁄𑀘𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺, 𑀏𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀥𑀸 𑀫𑀼𑀤𑁆𑀥𑀸 𑀨𑀮𑀢𑀻”𑀢𑀺𑁇

1211\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀚𑀺𑀭𑀧𑀸𑀡𑀺 𑀬𑀓𑁆𑀔𑁄 𑀆𑀬𑀲𑀁 𑀯𑀚𑀺𑀭𑀁 𑀆𑀤𑀸𑀬 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁 𑀲𑀘𑁆𑀘𑀓𑀲𑁆𑀲 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀭𑀺𑀯𑁂𑀳𑀸𑀲𑀁 𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  “𑀲𑀘𑀸𑀬𑀁 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀏𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀥𑀸 𑀫𑀼𑀤𑁆𑀥𑀁 𑀨𑀸𑀮𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀯𑀚𑀺𑀭𑀧𑀸𑀡𑀺𑀁 𑀬𑀓𑁆𑀔𑀁 𑀪𑀕𑀯𑀸 𑀘𑁂𑀯 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑀘𑁆𑀘𑀓𑁄 𑀘 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀻𑀢𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑁄 𑀮𑁄𑀫𑀳𑀝𑁆𑀞𑀚𑀸𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀢𑀸𑀡𑀁 𑀕𑀯𑁂𑀲𑀻 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀮𑁂𑀡𑀁 𑀕𑀯𑁂𑀲𑀻 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀲𑀭𑀡𑀁 𑀕𑀯𑁂𑀲𑀻 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀼𑀘𑁆𑀙𑀢𑀼 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄, 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

1212\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  ‘𑀭𑀽𑀧𑀁 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀭𑀽𑀧𑁂 𑀯𑀲𑁄—  𑀏𑀯𑀁 𑀫𑁂 𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀼, 𑀏𑀯𑀁 𑀫𑁂 𑀭𑀽𑀧𑀁 𑀫𑀸 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇

1213\. “𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦; 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺𑁇 𑀦 𑀔𑁄 𑀢𑁂 𑀲𑀦𑁆𑀥𑀺𑀬𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  ‘𑀯𑁂𑀤𑀦𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀢𑀺𑀲𑁆𑀲𑀁 𑀯𑁂𑀤𑀦𑀸𑀬𑀁 𑀯𑀲𑁄—  𑀏𑀯𑀁 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀼, 𑀏𑀯𑀁 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀫𑀸 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇

1214\. “𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦; 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺𑁇 𑀦 𑀔𑁄 𑀢𑁂 𑀲𑀦𑁆𑀥𑀺𑀬𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁, 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  ‘𑀲𑀜𑁆𑀜𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀢𑀺𑀲𑁆𑀲𑀁 𑀲𑀜𑁆𑀜𑀸𑀬𑀁 𑀯𑀲𑁄—  𑀏𑀯𑀁 𑀫𑁂 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀼, 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀜𑁆𑀜𑀸 𑀫𑀸 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇

1215\. “𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦; 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺𑁇 𑀦 𑀔𑁄 𑀢𑁂 𑀲𑀦𑁆𑀥𑀺𑀬𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁, 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  ‘𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀢𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀯𑀲𑁄—  𑀏𑀯𑀁 𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀼, 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇

1216\. “𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦; 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺𑁇 𑀦 𑀔𑁄 𑀢𑁂 𑀲𑀦𑁆𑀥𑀺𑀬𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁, 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  ‘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀯𑀲𑁄—  𑀏𑀯𑀁 𑀫𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀼, 𑀏𑀯𑀁 𑀫𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀫𑀸 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇

1217\. “𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦; 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺𑁇 𑀦 𑀔𑁄 𑀢𑁂 𑀲𑀦𑁆𑀥𑀺𑀬𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁, 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀭𑀽𑀧𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇

1218\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…𑀧𑁂…  𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫”𑁇

1219\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀬𑁄 𑀦𑀼 𑀔𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀮𑁆𑀮𑀻𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑀕𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑁄, 𑀤𑀼𑀓𑁆𑀔𑀁 ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀲𑁄 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀓𑀺𑀜𑁆𑀳𑀺 𑀲𑀺𑀬𑀸, 𑀪𑁄 𑀕𑁄𑀢𑀫? 𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫𑀸”𑀢𑀺𑁇

1225\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑀫𑁂𑀯, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀥𑀁𑀲𑀻, 𑀫𑀬𑀁 𑀧𑀕𑀩𑁆𑀩𑀸, 𑀬𑁂 𑀫𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀆𑀲𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀅𑀫𑀜𑁆𑀜𑀺𑀫𑁆𑀳𑁇 𑀲𑀺𑀬𑀸 𑀳𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀳𑀢𑁆𑀣𑀺𑀁 𑀧𑀪𑀺𑀦𑁆𑀦𑀁 𑀆𑀲𑀚𑁆𑀚 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁄𑀢𑁆𑀣𑀺𑀪𑀸𑀯𑁄, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀆𑀲𑀚𑁆𑀚 𑀲𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁄𑀢𑁆𑀣𑀺𑀪𑀸𑀯𑁄𑁇 𑀲𑀺𑀬𑀸 𑀳𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀆𑀲𑀚𑁆𑀚 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁄𑀢𑁆𑀣𑀺𑀪𑀸𑀯𑁄, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀆𑀲𑀚𑁆𑀚 𑀲𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁄𑀢𑁆𑀣𑀺𑀪𑀸𑀯𑁄𑁇 𑀲𑀺𑀬𑀸 𑀳𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀆𑀲𑀻𑀯𑀺𑀲𑀁 𑀖𑁄𑀭𑀯𑀺𑀲𑀁 𑀆𑀲𑀚𑁆𑀚 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁄𑀢𑁆𑀣𑀺𑀪𑀸𑀯𑁄, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀆𑀲𑀚𑁆𑀚 𑀲𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁄𑀢𑁆𑀣𑀺𑀪𑀸𑀯𑁄𑁇 𑀫𑀬𑀫𑁂𑀯, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀥𑀁𑀲𑀻, 𑀫𑀬𑀁 𑀧𑀕𑀩𑁆𑀩𑀸, 𑀬𑁂 𑀫𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀯𑀸𑀤𑁂𑀦 𑀯𑀸𑀤𑀁 𑀆𑀲𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀅𑀫𑀜𑁆𑀜𑀺𑀫𑁆𑀳𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀫𑁂 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

1226\. 𑀅𑀣 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀻, 𑀲𑀫𑀡𑁄 𑀫𑁂 𑀕𑁄𑀢𑀫𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀢𑁂𑀦 𑀫𑁂 𑀅𑀪𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀣 𑀬𑀫𑀲𑁆𑀲 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀘𑁆𑀘𑀓𑀲𑁆𑀲 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀣𑀸𑀮𑀺𑀧𑀸𑀓𑀲𑀢𑀸𑀦𑀺 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑀁 𑀅𑀪𑀺𑀳𑀭𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁂 𑀆𑀭𑀸𑀫𑁂 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀘𑁆𑀘𑀓𑀲𑁆𑀲 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀦𑀻𑀘𑀁 𑀆𑀲𑀦𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀘𑁆𑀘𑀓𑁄 𑀦𑀺𑀕𑀡𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀫𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀤𑀸𑀦𑁂 𑀧𑀼𑀜𑁆𑀜𑀜𑁆𑀘 𑀧𑀼𑀜𑁆𑀜𑀫𑀳𑀻 𑀘 𑀢𑀁 𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀽”𑀢𑀺𑁇 “𑀬𑀁 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀢𑀸𑀤𑀺𑀲𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀅𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀅𑀯𑀻𑀢𑀫𑁄𑀳𑀁, 𑀢𑀁 𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀁 𑀔𑁄, 𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀲𑁆𑀲𑀦, 𑀫𑀸𑀤𑀺𑀲𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀯𑀻𑀢𑀫𑁄𑀳𑀁, 𑀢𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇
