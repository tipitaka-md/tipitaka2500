# 𑁧𑁇𑁮𑁇𑁫 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀭𑀺𑀬𑀸𑀬

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀭𑀺𑀬𑀸𑀬

(87.)

252\. 𑀲𑁄 𑀯𑀢, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀮𑀺𑀧𑀧𑀮𑀺𑀧𑀦𑁆𑀦𑁄 𑀧𑀭𑀁 𑀧𑀮𑀺𑀧𑀧𑀮𑀺𑀧𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀧𑀮𑀺𑀧𑀧𑀮𑀺𑀧𑀦𑁆𑀦𑁄 𑀧𑀭𑀁 𑀧𑀮𑀺𑀧𑀧𑀮𑀺𑀧𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀤𑀦𑁆𑀢𑁄 𑀅𑀯𑀺𑀦𑀻𑀢𑁄 𑀅𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀧𑀭𑀁 𑀤𑀫𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀦𑁂𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀤𑀦𑁆𑀢𑁄 𑀯𑀺𑀦𑀻𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀧𑀭𑀁 𑀤𑀫𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀦𑁂𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀳𑀺𑀁𑀲𑀓𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬, 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀯𑁂𑀭𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀨𑀭𑀼𑀲𑀯𑀸𑀘𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀯𑁂𑀭𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 (𑁧--𑁧𑁦)

253\. 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀜𑀸𑀡𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 (𑁧𑁧--𑁨𑁦)

254\. 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀺𑀕𑀢𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀉𑀤𑁆𑀥𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀦𑀼𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀓𑁄𑀥𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀓𑁆𑀓𑁄𑀥𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀉𑀧𑀦𑀸𑀳𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀦𑀸𑀳𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀓𑁆𑀔𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀫𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀧𑀍𑀆𑀲𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀧𑀍𑀆𑀲𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀇𑀲𑁆𑀲𑀼𑀓𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀦𑀺𑀲𑁆𑀲𑀼𑀓𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀘𑁆𑀙𑀭𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀲𑀞𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀸𑀬𑀸𑀯𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀫𑀸𑀬𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀣𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀤𑁆𑀥𑀺𑀬𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀢𑀺𑀫𑀸𑀦𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀦𑀢𑀺𑀫𑀸𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀤𑀼𑀩𑁆𑀩𑀘𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀳𑀺𑀭𑀺𑀓𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀦𑁄𑀢𑁆𑀢𑀸𑀧𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀩𑀸𑀳𑀼𑀲𑀘𑁆𑀘𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀤𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀸𑀫𑀸𑀲𑀺𑀆𑀥𑀸𑀦𑀕𑁆𑀕𑀸𑀳𑀺𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀸𑀫𑀸𑀲𑀺𑀅𑀦𑀸𑀥𑀸𑀦𑀕𑁆𑀕𑀸𑀳𑀺𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬𑁇 (𑁨𑁧--𑁪𑁪)

(88.)

255\. 𑀇𑀢𑀺 𑀔𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀤𑁂𑀲𑀺𑀢𑁄 𑀫𑀬𑀸 𑀲𑀮𑁆𑀮𑁂𑀔𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀤𑁂𑀲𑀺𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀤𑁂𑀲𑀺𑀢𑁄 𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀤𑁂𑀲𑀺𑀢𑁄 𑀉𑀧𑀭𑀺𑀪𑀸𑀕𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀤𑁂𑀲𑀺𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀭𑀺𑀬𑀸𑀬𑁄𑁇 𑀬𑀁 𑀔𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀲𑀢𑁆𑀣𑀸𑀭𑀸 𑀓𑀭𑀡𑀻𑀬𑀁 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁂𑀦 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀢𑀁 𑀯𑁄 𑀢𑀁 𑀫𑀬𑀸𑁇 𑀏𑀢𑀸𑀦𑀺, 𑀘𑀼𑀦𑁆𑀤, 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀸𑀦𑀺, 𑀏𑀢𑀸𑀦𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀸𑀦𑀺, 𑀛𑀸𑀬𑀣, 𑀘𑀼𑀦𑁆𑀤, 𑀫𑀸 𑀧𑀫𑀸𑀤𑀢𑁆𑀣, 𑀫𑀸 𑀧𑀘𑁆𑀙𑀸𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄 𑀅𑀳𑀼𑀯𑀢𑁆𑀣—  𑀅𑀬𑀁 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻”𑀢𑀺𑁇

256\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀘𑀼𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀻𑀢𑀺𑁇

257\. _𑀘𑀢𑀼𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀧𑀤𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀲𑀦𑁆𑀥𑀬𑁄 𑀧𑀜𑁆𑀘 𑀤𑁂𑀲𑀺𑀢𑀸;_  
_𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀦𑀸𑀫 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁄,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀲𑀸𑀕𑀭𑀽𑀧𑀫𑁄𑀢𑀺𑁇_  


---

258\. 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
