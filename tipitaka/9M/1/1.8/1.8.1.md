# 𑁧𑁇𑁮𑁇𑁧 𑀲𑀮𑁆𑀮𑁂𑀔𑀧𑀭𑀺𑀬𑀸𑀬

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀧𑀭𑀺𑀬𑀸𑀬

(83.)

202\. 𑀇𑀥 𑀔𑁄 𑀧𑀦 𑀯𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇

203\. ‘𑀧𑀭𑁂 𑀯𑀺𑀳𑀺𑀁𑀲𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧)

204\. ‘𑀧𑀭𑁂 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨)

205\. ‘𑀧𑀭𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩)

206\. ‘𑀧𑀭𑁂 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁪)

207\. ‘𑀧𑀭𑁂 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁫)

208\. ‘𑀧𑀭𑁂 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁬)

209\. ‘𑀧𑀭𑁂 𑀨𑀭𑀼𑀲𑀯𑀸𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁭)

210\. ‘𑀧𑀭𑁂 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁮)

211\. ‘𑀧𑀭𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀽 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀽 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁯)

212\. ‘𑀧𑀭𑁂 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁦)

213\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁧)

214\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁨)

215\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁩)

216\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁪)

217\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁫)

218\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁬)

219\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁭)

220\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁮)

221\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀜𑀸𑀡𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁧𑁯)

222\. ‘𑀧𑀭𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁦)

223\. ‘𑀧𑀭𑁂 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀯𑀺𑀕𑀢𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁧)

224\. ‘𑀧𑀭𑁂 𑀉𑀤𑁆𑀥𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁨)

225\. ‘𑀧𑀭𑁂 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁩)

226\. ‘𑀧𑀭𑁂 𑀓𑁄𑀥𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁪)

227\. ‘𑀧𑀭𑁂 𑀉𑀧𑀦𑀸𑀳𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀦𑀸𑀳𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁫)

228\. ‘𑀧𑀭𑁂 𑀫𑀓𑁆𑀔𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀫𑀓𑁆𑀔𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁬)

229\. ‘𑀧𑀭𑁂 𑀧𑀍𑀆𑀲𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀧𑀍𑀆𑀲𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁭)

230\. ‘𑀧𑀭𑁂 𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀦𑀺𑀲𑁆𑀲𑀼𑀓𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁮)

231\. ‘𑀧𑀭𑁂 𑀫𑀘𑁆𑀙𑀭𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀫𑀘𑁆𑀙𑀭𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁨𑁯)

232\. ‘𑀧𑀭𑁂 𑀲𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀲𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁦)

233\. ‘𑀧𑀭𑁂 𑀫𑀸𑀬𑀸𑀯𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀫𑀸𑀬𑀸𑀯𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁧)

234\. ‘𑀧𑀭𑁂 𑀣𑀤𑁆𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀤𑁆𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁨)

235\. ‘𑀧𑀭𑁂 𑀅𑀢𑀺𑀫𑀸𑀦𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀦𑀢𑀺𑀫𑀸𑀦𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁩)

236\. ‘𑀧𑀭𑁂 𑀤𑀼𑀩𑁆𑀩𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀼𑀯𑀘𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁪)

237\. ‘𑀧𑀭𑁂 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁫)

238\. ‘𑀧𑀭𑁂 𑀧𑀫𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁬)

239\. ‘𑀧𑀭𑁂 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀲𑀤𑁆𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁭)

240\. ‘𑀧𑀭𑁂 𑀅𑀳𑀺𑀭𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀳𑀺𑀭𑀺𑀫𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁮)

241\. ‘𑀧𑀭𑁂 𑀅𑀦𑁄𑀢𑁆𑀢𑀸𑀧𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀑𑀢𑁆𑀢𑀸𑀧𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁩𑁯)

242\. ‘𑀧𑀭𑁂 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁪𑁦)

243\. ‘𑀧𑀭𑁂 𑀓𑀼𑀲𑀻𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁪𑁧)

244\. ‘𑀧𑀭𑁂 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁪𑁨)

245\. ‘𑀧𑀭𑁂 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁪𑁩)

246\. ‘𑀧𑀭𑁂 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀸𑀫𑀸𑀲𑀻 𑀆𑀥𑀸𑀦𑀕𑁆𑀕𑀸𑀳𑀻 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀅𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀸𑀫𑀸𑀲𑀻 𑀅𑀦𑀸𑀥𑀸𑀦𑀕𑁆𑀕𑀸𑀳𑀻 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀲𑀮𑁆𑀮𑁂𑀔𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 (𑁪𑁪)
