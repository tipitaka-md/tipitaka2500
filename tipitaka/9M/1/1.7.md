# 𑁧𑁇𑁭 𑀯𑀢𑁆𑀣𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀯𑀢𑁆𑀣𑀲𑀼𑀢𑁆𑀢

(70.)

175\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

176\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀢𑁆𑀣𑀁 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀫𑀮𑀕𑁆𑀕𑀳𑀺𑀢𑀁; 𑀢𑀫𑁂𑀦𑀁 𑀭𑀚𑀓𑁄 𑀬𑀲𑁆𑀫𑀺𑀁 𑀬𑀲𑁆𑀫𑀺𑀁 𑀭𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀉𑀧𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬—  𑀬𑀤𑀺 𑀦𑀻𑀮𑀓𑀸𑀬 𑀬𑀤𑀺 𑀧𑀻𑀢𑀓𑀸𑀬 𑀬𑀤𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀬 𑀬𑀤𑀺 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀸𑀬 𑀤𑀼𑀭𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀫𑁂𑀯𑀲𑁆𑀲 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀯𑀡𑁆𑀡𑀫𑁂𑀯𑀲𑁆𑀲𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀢𑁆𑀣𑀲𑁆𑀲𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂, 𑀤𑀼𑀕𑁆𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀢𑁆𑀣𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀁; 𑀢𑀫𑁂𑀦𑀁 𑀭𑀚𑀓𑁄 𑀬𑀲𑁆𑀫𑀺𑀁 𑀬𑀲𑁆𑀫𑀺𑀁 𑀭𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀉𑀧𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬—  𑀬𑀤𑀺 𑀦𑀻𑀮𑀓𑀸𑀬 𑀬𑀤𑀺 𑀧𑀻𑀢𑀓𑀸𑀬 𑀬𑀤𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀬 𑀬𑀤𑀺 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀸𑀬—  𑀲𑀼𑀭𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀫𑁂𑀯𑀲𑁆𑀲 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀯𑀡𑁆𑀡𑀫𑁂𑀯𑀲𑁆𑀲𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀢𑁆𑀣𑀲𑁆𑀲𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑁂 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂, 𑀲𑀼𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇

(71.)

177\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀸? 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀯𑀺𑀲𑀫𑀮𑁄𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀓𑁄𑀥𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀉𑀧𑀦𑀸𑀳𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀫𑀓𑁆𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀧𑀍𑀆𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀇𑀲𑁆𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀫𑀸𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀣𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀲𑀸𑀭𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀅𑀢𑀺𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀫𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄, 𑀧𑀫𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄𑁇

(72.)

178\. 𑀲 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀯𑀺𑀲𑀫𑀮𑁄𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀯𑀺𑀲𑀫𑀮𑁄𑀪𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀓𑁄𑀥𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀓𑁄𑀥𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀉𑀧𑀦𑀸𑀳𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀦𑀸𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀫𑀓𑁆𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀓𑁆𑀔𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀧𑀍𑀆𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀍𑀆𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀇𑀲𑁆𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀇𑀲𑁆𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀫𑀸𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀣𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀣𑀫𑁆𑀪𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀲𑀸𑀭𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀅𑀢𑀺𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀫𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀫𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺; ‘𑀧𑀫𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀫𑀸𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀚𑀳𑀢𑀺𑁇

(73.)

179\. 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 ‘𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀯𑀺𑀲𑀫𑀮𑁄𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀯𑀺𑀲𑀫𑀮𑁄𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, ‘𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀓𑁄𑀥𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀓𑁄𑀥𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀉𑀧𑀦𑀸𑀳𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀦𑀸𑀳𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀫𑀓𑁆𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀓𑁆𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀧𑀍𑀆𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀍𑀆𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀇𑀲𑁆𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀇𑀲𑁆𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀫𑀸𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀣𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀣𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀲𑀸𑀭𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑀫𑁆𑀪𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀅𑀢𑀺𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀫𑀸𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀫𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; ‘𑀧𑀫𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀫𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

(74.)

180\. 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺; 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺; 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀉𑀚𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀜𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸𑀦𑀺, 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑁇 𑀏𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

(75.)

181\. 𑀬𑀣𑁄𑀥𑀺 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀯𑀦𑁆𑀢𑀁 𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀳𑀻𑀦𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀁, 𑀲𑁄 ‘𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑀫𑁆𑀳𑀻’𑀢𑀺 𑀮𑀪𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁𑁇 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺; ‘𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑀫𑁆𑀳𑀻’𑀢𑀺 𑀮𑀪𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁; 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 ‘𑀬𑀣𑁄𑀥𑀺 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀘𑀢𑁆𑀢𑀁 𑀯𑀦𑁆𑀢𑀁 𑀫𑀼𑀢𑁆𑀢𑀁 𑀧𑀳𑀻𑀦𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀦𑁆’𑀢𑀺 𑀮𑀪𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁; 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇

(76.)

182\. 𑀲 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁𑀲𑀻𑀮𑁄 𑀏𑀯𑀁𑀥𑀫𑁆𑀫𑁄 𑀏𑀯𑀁𑀧𑀜𑁆𑀜𑁄 𑀲𑀸𑀮𑀻𑀦𑀜𑁆𑀘𑁂𑀧𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀺𑀘𑀺𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀅𑀦𑁂𑀓𑀲𑀽𑀧𑀁 𑀅𑀦𑁂𑀓𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁, 𑀦𑁂𑀯𑀲𑁆𑀲 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀢𑁆𑀣𑀁 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀫𑀮𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑀁 𑀆𑀕𑀫𑁆𑀫 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀁, 𑀉𑀓𑁆𑀓𑀸𑀫𑀼𑀔𑀁 𑀯𑀸 𑀧𑀦𑀸𑀕𑀫𑁆𑀫 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁𑀲𑀻𑀮𑁄 𑀏𑀯𑀁𑀥𑀫𑁆𑀫𑁄 𑀏𑀯𑀁𑀧𑀜𑁆𑀜𑁄 𑀲𑀸𑀮𑀻𑀦𑀜𑁆𑀘𑁂𑀧𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀺𑀘𑀺𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀅𑀦𑁂𑀓𑀲𑀽𑀧𑀁 𑀅𑀦𑁂𑀓𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁, 𑀦𑁂𑀯𑀲𑁆𑀲 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇

(77.)

183\. 𑀲𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀓𑀭𑀼𑀡𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂…  𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

(78.)

184\. 𑀲𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀇𑀤𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀇𑀫𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀕𑀢𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀺 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀓𑀸𑀫𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀯𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀦𑀸𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑁂𑀦 𑀲𑀺𑀦𑀸𑀦𑁂𑀦𑀸’”𑀢𑀺𑁇

(79.)

185\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀦 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀩𑀸𑀳𑀼𑀓𑀁 𑀦𑀤𑀺𑀁 𑀲𑀺𑀦𑀸𑀬𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀓𑀺𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀩𑀸𑀳𑀼𑀓𑀸𑀬 𑀦𑀤𑀺𑀬𑀸? 𑀓𑀺𑀁 𑀩𑀸𑀳𑀼𑀓𑀸 𑀦𑀤𑀻 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀮𑁄𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀸 𑀳𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑀸𑀳𑀼𑀓𑀸 𑀦𑀤𑀻 𑀩𑀳𑀼𑀚𑀦𑀲𑁆𑀲, 𑀧𑀼𑀜𑁆𑀜𑀲𑀫𑁆𑀫𑀢𑀸 𑀳𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑀸𑀳𑀼𑀓𑀸 𑀦𑀤𑀻 𑀩𑀳𑀼𑀚𑀦𑀲𑁆𑀲, 𑀩𑀸𑀳𑀼𑀓𑀸𑀬 𑀧𑀦 𑀦𑀤𑀺𑀬𑀸 𑀩𑀳𑀼𑀚𑀦𑁄 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀧𑀯𑀸𑀳𑁂𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

186\. _“𑀩𑀸𑀳𑀼𑀓𑀁 𑀅𑀥𑀺𑀓𑀓𑁆𑀓𑀜𑁆𑀘,_  
_𑀕𑀬𑀁 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀁 𑀫𑀧𑀺;_  
_𑀲𑀭𑀲𑁆𑀲𑀢𑀺𑀁 𑀧𑀬𑀸𑀕𑀜𑁆𑀘,_  
_𑀅𑀣𑁄 𑀩𑀸𑀳𑀼𑀫𑀢𑀺𑀁 𑀦𑀤𑀺𑀁;_  
_𑀦𑀺𑀘𑁆𑀘𑀫𑁆𑀧𑀺 𑀩𑀸𑀮𑁄 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑁄,_  
_𑀓𑀡𑁆𑀳𑀓𑀫𑁆𑀫𑁄 𑀦 𑀲𑀼𑀚𑁆𑀛𑀢𑀺𑁇_  


187\. _𑀓𑀺𑀁 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺,_  
_𑀓𑀺𑀁 𑀧𑀬𑀸𑀕𑀸 𑀓𑀺𑀁 𑀩𑀸𑀳𑀼𑀓𑀸 𑀦𑀤𑀻;_  
_𑀯𑁂𑀭𑀺𑀁 𑀓𑀢𑀓𑀺𑀩𑁆𑀩𑀺𑀲𑀁 𑀦𑀭𑀁,_  
_𑀦 𑀳𑀺 𑀦𑀁 𑀲𑁄𑀥𑀬𑁂 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀺𑀦𑀁𑁇_  


188\. _𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑁂 𑀲𑀤𑀸 𑀨𑀕𑁆𑀕𑀼,_  
_𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀼𑀧𑁄𑀲𑀣𑁄 𑀲𑀤𑀸;_  
_𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀼𑀘𑀺𑀓𑀫𑁆𑀫𑀲𑁆𑀲,_  
_𑀲𑀤𑀸 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀢𑁂 𑀯𑀢𑀁;_  
_𑀇𑀥𑁂𑀯 𑀲𑀺𑀦𑀸𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑁂𑀲𑀼 𑀓𑀭𑁄𑀳𑀺 𑀔𑁂𑀫𑀢𑀁𑁇_  


189\. _𑀲𑀘𑁂 𑀫𑀼𑀲𑀸 𑀦 𑀪𑀡𑀲𑀺,_  
_𑀲𑀘𑁂 𑀧𑀸𑀡𑀁 𑀦 𑀳𑀺𑀁𑀲𑀲𑀺;_  
_𑀲𑀘𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀦𑀸𑀤𑀺𑀬𑀲𑀺,_  
_𑀲𑀤𑁆𑀤𑀳𑀸𑀦𑁄 𑀅𑀫𑀘𑁆𑀙𑀭𑀻;_  
_𑀓𑀺𑀁 𑀓𑀸𑀳𑀲𑀺 𑀕𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀉𑀤𑀧𑀸𑀦𑁄𑀧𑀺 𑀢𑁂 𑀕𑀬𑀸”𑀢𑀺𑁇_  


(80.)

190\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑁄𑀢𑀸 𑀕𑁄𑀢𑀫𑁂𑀦 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀳𑀁 𑀪𑁄𑀢𑁄 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 𑀅𑀮𑀢𑁆𑀣 𑀔𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀢𑁆𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀅𑀘𑀺𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇
