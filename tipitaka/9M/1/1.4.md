# 𑁧𑁇𑁪 𑀪𑀬𑀪𑁂𑀭𑀯𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀪𑀬𑀪𑁂𑀭𑀯𑀲𑀼𑀢𑁆𑀢

(34.)

98\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁂𑀫𑁂, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀪𑀯𑀁 𑀢𑁂𑀲𑀁 𑀕𑁄𑀢𑀫𑁄 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑁄, 𑀪𑀯𑀁 𑀢𑁂𑀲𑀁 𑀕𑁄𑀢𑀫𑁄 𑀩𑀳𑀼𑀓𑀸𑀭𑁄, 𑀪𑀯𑀁 𑀢𑁂𑀲𑀁 𑀕𑁄𑀢𑀫𑁄 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸; 𑀪𑁄𑀢𑁄 𑀘 𑀧𑀦 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁇 𑀬𑁂 𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀫𑀫𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀅𑀳𑀁 𑀢𑁂𑀲𑀁 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑁄, 𑀅𑀳𑀁 𑀢𑁂𑀲𑀁 𑀩𑀳𑀼𑀓𑀸𑀭𑁄, 𑀅𑀳𑀁 𑀢𑁂𑀲𑀁 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸; 𑀫𑀫 𑀘 𑀧𑀦 𑀲𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇 “𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀸𑀦𑀺 𑀳𑀺 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺, 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀁, 𑀤𑀼𑀭𑀪𑀺𑀭𑀫𑀁 𑀏𑀓𑀢𑁆𑀢𑁂, 𑀳𑀭𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀫𑀦𑁄 𑀯𑀦𑀸𑀦𑀺 𑀲𑀫𑀸𑀥𑀺𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁇 𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀸𑀦𑀺 𑀳𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺, 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀁, 𑀤𑀼𑀭𑀪𑀺𑀭𑀫𑀁 𑀏𑀓𑀢𑁆𑀢𑁂, 𑀳𑀭𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀫𑀦𑁄 𑀯𑀦𑀸𑀦𑀺 𑀲𑀫𑀸𑀥𑀺𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇

(35.)

99\. 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀸𑀦𑀺 𑀳𑀺 𑀔𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺, 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀁, 𑀤𑀼𑀭𑀪𑀺𑀭𑀫𑀁 𑀏𑀓𑀢𑁆𑀢𑁂, 𑀳𑀭𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀫𑀦𑁄 𑀯𑀦𑀸𑀦𑀺 𑀲𑀫𑀸𑀥𑀺𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧)

(36.)

100\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁨--𑁪)

101\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀽 𑀓𑀸𑀫𑁂𑀲𑀼 𑀢𑀺𑀩𑁆𑀩𑀲𑀸𑀭𑀸𑀕𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼𑀓𑀸𑀫𑁂𑀲𑀼𑀢𑀺𑀩𑁆𑀩𑀲𑀸𑀭𑀸𑀕𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀓𑀸𑀫𑁂𑀲𑀼 𑀢𑀺𑀩𑁆𑀩𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀽𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀽 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁫)

(38.)

102\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀤𑀼𑀝𑁆𑀞𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀧𑀤𑀼𑀝𑁆𑀞𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀤𑀼𑀝𑁆𑀞𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁬)

(39.)

103\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀯𑀺𑀕𑀢𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀯𑀺𑀕𑀢𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀺𑀕𑀢𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁭)

(40.)

104\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀉𑀤𑁆𑀥𑀢𑀸 𑀅𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀉𑀤𑁆𑀥𑀢𑀅𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀉𑀤𑁆𑀥𑀢𑁄 𑀅𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁮)

(41.)

105\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀓𑀗𑁆𑀔𑀻 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀻 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀓𑀗𑁆𑀔𑀺𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀓𑀗𑁆𑀔𑀻 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀻 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁯)

(42.)

106\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀢𑁆𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀓𑀸 𑀧𑀭𑀯𑀫𑁆𑀪𑀻 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀅𑀢𑁆𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀦𑀧𑀭𑀯𑀫𑁆𑀪𑀦𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀅𑀢𑁆𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀓𑁄 𑀧𑀭𑀯𑀫𑁆𑀪𑀻 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀅𑀦𑀢𑁆𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀓𑁄 𑀅𑀧𑀭𑀯𑀫𑁆𑀪𑀻𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀅𑀦𑀢𑁆𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀓𑀸 𑀅𑀧𑀭𑀯𑀫𑁆𑀪𑀻 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀦𑀢𑁆𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀓𑀢𑀁 𑀅𑀧𑀭𑀯𑀫𑁆𑀪𑀺𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧𑁦)

(43.)

107\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀙𑀫𑁆𑀪𑀻 𑀪𑀻𑀭𑀼𑀓𑀚𑀸𑀢𑀺𑀓𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀙𑀫𑁆𑀪𑀺𑀪𑀻𑀭𑀼𑀓𑀚𑀸𑀢𑀺𑀓𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀙𑀫𑁆𑀪𑀻 𑀪𑀻𑀭𑀼𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀯𑀺𑀕𑀢𑀮𑁄𑀫𑀳𑀁𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀯𑀺𑀕𑀢𑀮𑁄𑀫𑀳𑀁𑀲𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀺𑀕𑀢𑀮𑁄𑀫𑀳𑀁𑀲𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧𑁧)

(44.)

108\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀦𑀺𑀓𑀸𑀫𑀬𑀫𑀸𑀦𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀦𑀺𑀓𑀸𑀫𑀦𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀦𑀺𑀓𑀸𑀫𑀬𑀫𑀸𑀦𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧𑁨)

(45.)

109\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀓𑀼𑀲𑀻𑀢𑀸 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀻𑀢𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀓𑀼𑀲𑀻𑀢𑁄 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧𑁩)

(46.)

110\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀻 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧𑁪)

(47.)

111\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧𑁫)

(48.)

112\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀏𑀍𑀅𑀫𑀽𑀕𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀏𑀍𑀅𑀫𑀽𑀕𑀲𑀦𑁆𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀳𑀯𑁂 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀯𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀏𑀍𑀅𑀫𑀽𑀕𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺; 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀯𑁄 𑀅𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀫𑀳𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’𑀢𑀺𑁇 𑀏𑀢𑀫𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑁆𑀮𑁄𑀫𑀫𑀸𑀧𑀸𑀤𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 (𑁧𑁬)

114\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑀸 𑀢𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀅𑀪𑀺𑀮𑀓𑁆𑀔𑀺𑀢𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀻 𑀧𑀜𑁆𑀘𑀤𑀲𑀻 𑀅𑀝𑁆𑀞𑀫𑀻 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲—  𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀆𑀭𑀸𑀫𑀘𑁂𑀢𑀺𑀬𑀸𑀦𑀺 𑀯𑀦𑀘𑁂𑀢𑀺𑀬𑀸𑀦𑀺 𑀭𑀼𑀓𑁆𑀔𑀘𑁂𑀢𑀺𑀬𑀸𑀦𑀺 𑀪𑀺𑀁𑀲𑀦𑀓𑀸𑀦𑀺 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑀸𑀦𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀁 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀸𑀳𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀬𑀸 𑀢𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀅𑀪𑀺𑀮𑀓𑁆𑀔𑀺𑀢𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀻 𑀧𑀜𑁆𑀘𑀤𑀲𑀻 𑀅𑀝𑁆𑀞𑀫𑀻 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲—  𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀆𑀭𑀸𑀫𑀘𑁂𑀢𑀺𑀬𑀸𑀦𑀺 𑀯𑀦𑀘𑁂𑀢𑀺𑀬𑀸𑀦𑀺 𑀭𑀼𑀓𑁆𑀔𑀘𑁂𑀢𑀺𑀬𑀸𑀦𑀺 𑀪𑀺𑀁𑀲𑀦𑀓𑀸𑀦𑀺 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑀸𑀦𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀢𑁆𑀣 𑀘 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀺𑀳𑀭𑀢𑁄 𑀫𑀕𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑁄𑀭𑁄 𑀯𑀸 𑀓𑀝𑁆𑀞𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀯𑀸𑀢𑁄 𑀯𑀸 𑀧𑀡𑁆𑀡𑀓𑀲𑀝𑀁 𑀏𑀭𑁂𑀢𑀺; 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀏𑀢𑀁 𑀦𑀽𑀦 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼 𑀪𑀬𑀧𑀝𑀺𑀓𑀗𑁆𑀔𑀻 𑀯𑀺𑀳𑀭𑀸𑀫𑀺? 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀲 𑀫𑁂 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀢𑀣𑀸𑀪𑀽𑀢𑀁 𑀢𑀣𑀸𑀪𑀽𑀢𑁄𑀯 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀦 𑀦𑀺𑀲𑀻𑀤𑀸𑀫𑀺 𑀦 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁄𑀯 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀞𑀺𑀢𑀲𑁆𑀲 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀘𑀗𑁆𑀓𑀫𑀸𑀫𑀺 𑀦 𑀦𑀺𑀲𑀻𑀤𑀸𑀫𑀺 𑀦 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀺𑁇 𑀬𑀸𑀯 𑀞𑀺𑀢𑁄𑀯 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀺 𑀦 𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀦 𑀘𑀗𑁆𑀓𑀫𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀯 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀦𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀦𑀺𑀲𑀻𑀤𑀸𑀫𑀺 𑀦 𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀦 𑀘𑀗𑁆𑀓𑀫𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀦𑀺𑀧𑀦𑁆𑀦𑁄𑀯 𑀢𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀫𑀺𑁇

118\. 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁂 𑀅𑀦𑀗𑁆𑀕𑀡𑁂 𑀯𑀺𑀕𑀢𑀽𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀫𑀼𑀤𑀼𑀪𑀽𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀞𑀺𑀢𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑀜𑀸𑀡𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀲𑀺𑀁𑁇 𑀲𑁄 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑁂 𑀯𑀢 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀉𑀧𑀯𑀸𑀤𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸; 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇 𑀇𑀫𑁂 𑀯𑀸 𑀧𑀦 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀸𑀤𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸; 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀤𑀼𑀢𑀺𑀬𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀥𑀺𑀕𑀢𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀯𑀺𑀳𑀢𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀢𑀫𑁄 𑀯𑀺𑀳𑀢𑁄 𑀆𑀮𑁄𑀓𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇

(54.)

119\. 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁂 𑀅𑀦𑀗𑁆𑀕𑀡𑁂 𑀯𑀺𑀕𑀢𑀽𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀫𑀼𑀤𑀼𑀪𑀽𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀞𑀺𑀢𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀜𑀸𑀡𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀲𑀺𑀁𑁇 𑀲𑁄 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 ‘𑀇𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀯𑀁 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀓𑀸𑀫𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣, 𑀪𑀯𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣𑁇 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀢𑀢𑀺𑀬𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀥𑀺𑀕𑀢𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀯𑀺𑀳𑀢𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀢𑀫𑁄 𑀯𑀺𑀳𑀢𑁄 𑀆𑀮𑁄𑀓𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇

(55.)

120\. 𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦 𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀦𑀽𑀦 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀅𑀯𑀻𑀢𑀤𑁄𑀲𑁄 𑀅𑀯𑀻𑀢𑀫𑁄𑀳𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀏𑀯𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀤𑁆𑀯𑁂 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀸𑀫𑀺—  𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄, 𑀧𑀘𑁆𑀙𑀺𑀫𑀜𑁆𑀘 𑀚𑀦𑀢𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀫𑀸𑀦𑁄”𑀢𑀺𑁇

(56.)

121\. “𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀢𑀭𑀽𑀧𑀸 𑀯𑀢𑀸𑀬𑀁 𑀪𑁄𑀢𑀸 𑀕𑁄𑀢𑀫𑁂𑀦 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸, 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑁄𑀢𑀸 𑀕𑁄𑀢𑀫𑁂𑀦 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
