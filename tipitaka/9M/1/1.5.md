# 𑁧𑁇𑁫 𑀅𑀦𑀗𑁆𑀕𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀗𑁆𑀕𑀡𑀲𑀼𑀢𑁆𑀢

(57.)

123\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀯𑀼𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺𑁇 “𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

124\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀲𑀸𑀗𑁆𑀕𑀡𑀸𑀦𑀁𑀬𑁂𑀯 𑀲𑀢𑀁 𑀳𑀻𑀦𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀲𑀸𑀗𑁆𑀕𑀡𑀸𑀦𑀁𑀬𑁂𑀯 𑀲𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀦𑀗𑁆𑀕𑀡𑀸𑀦𑀁𑀬𑁂𑀯 𑀲𑀢𑀁 𑀳𑀻𑀦𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀦𑀗𑁆𑀕𑀡𑀸𑀦𑀁𑀬𑁂𑀯 𑀲𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀻”𑀢𑀺𑁇

(58.)

125\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

126\. “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦𑀺𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀲𑀸𑀗𑁆𑀕𑀡𑀸𑀦𑀁𑀬𑁂𑀯 𑀲𑀢𑀁 𑀏𑀓𑁄 𑀳𑀻𑀦𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀏𑀓𑁄 𑀲𑁂𑀝𑁆𑀞𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀺? 𑀓𑁄 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦𑀺𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀦𑀗𑁆𑀕𑀡𑀸𑀦𑀁𑀬𑁂𑀯 𑀲𑀢𑀁 𑀏𑀓𑁄 𑀳𑀻𑀦𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀏𑀓𑁄 𑀲𑁂𑀝𑁆𑀞𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀬𑀢𑀻”𑀢𑀺?

(59.)

127\. “𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀦 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀯𑀸𑀬𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲𑀗𑁆𑀕𑀡𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬; 𑀲𑁄 𑀲𑀭𑀸𑀕𑁄 𑀲𑀤𑁄𑀲𑁄 𑀲𑀫𑁄𑀳𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀆𑀪𑀢𑀸 𑀆𑀧𑀡𑀸 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀭𑀓𑀼𑀮𑀸 𑀯𑀸 𑀭𑀚𑁂𑀦 𑀘 𑀫𑀮𑁂𑀦 𑀘 𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑀸𑁇 𑀢𑀫𑁂𑀦𑀁 𑀲𑀸𑀫𑀺𑀓𑀸 𑀦 𑀘𑁂𑀯 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀦 𑀘 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑁂𑀬𑁆𑀬𑀼𑀁, 𑀭𑀚𑀸𑀧𑀣𑁂 𑀘 𑀦𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀢𑀭𑀸 𑀅𑀲𑁆𑀲 𑀫𑀮𑀕𑁆𑀕𑀳𑀺𑀢𑀸”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀦 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀯𑀸𑀬𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲𑀗𑁆𑀕𑀡𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬; 𑀲𑁄 𑀲𑀭𑀸𑀕𑁄 𑀲𑀤𑁄𑀲𑁄 𑀲𑀫𑁄𑀳𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇

128\. 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸𑀬𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲𑀗𑁆𑀕𑀡𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬; 𑀲𑁄 𑀅𑀭𑀸𑀕𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀆𑀪𑀢𑀸 𑀆𑀧𑀡𑀸 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀭𑀓𑀼𑀮𑀸 𑀯𑀸 𑀭𑀚𑁂𑀦 𑀘 𑀫𑀮𑁂𑀦 𑀘 𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑀸𑁇 𑀢𑀫𑁂𑀦𑀁 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂𑀯 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘, 𑀦 𑀘 𑀦𑀁 𑀭𑀚𑀸𑀧𑀣𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑀭𑀸 𑀅𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸𑀬𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲𑀗𑁆𑀕𑀡𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬; 𑀲𑁄 𑀅𑀭𑀸𑀕𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇

129\. 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀲𑁆𑀲𑀢𑀺; 𑀲𑁄 𑀲𑀭𑀸𑀕𑁄 𑀲𑀤𑁄𑀲𑁄 𑀲𑀫𑁄𑀳𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀆𑀪𑀢𑀸 𑀆𑀧𑀡𑀸 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀭𑀓𑀼𑀮𑀸 𑀯𑀸 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑁇 𑀢𑀫𑁂𑀦𑀁 𑀲𑀸𑀫𑀺𑀓𑀸 𑀦 𑀘𑁂𑀯 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀦 𑀘 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑁂𑀬𑁆𑀬𑀼𑀁, 𑀭𑀚𑀸𑀧𑀣𑁂 𑀘 𑀦𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀢𑀭𑀸 𑀅𑀲𑁆𑀲 𑀫𑀮𑀕𑁆𑀕𑀳𑀺𑀢𑀸”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀲𑁆𑀲𑀢𑀺; 𑀲𑁄 𑀲𑀭𑀸𑀕𑁄 𑀲𑀤𑁄𑀲𑁄 𑀲𑀫𑁄𑀳𑁄 𑀲𑀸𑀗𑁆𑀕𑀡𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇

130\. 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀸𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀲𑁆𑀲𑀢𑀺; 𑀲𑁄 𑀅𑀭𑀸𑀕𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀆𑀪𑀢𑀸 𑀆𑀧𑀡𑀸 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀭𑀓𑀼𑀮𑀸 𑀯𑀸 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑁇 𑀢𑀫𑁂𑀦𑀁 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂𑀯 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘, 𑀦 𑀘 𑀦𑀁 𑀭𑀚𑀸𑀧𑀣𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀁𑀲𑀧𑀸𑀢𑀺 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑀭𑀸 𑀅𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀸𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀲𑁆𑀲𑀢𑀺; 𑀲𑁄 𑀅𑀭𑀸𑀕𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇

132\. “𑀅𑀗𑁆𑀕𑀡𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆”𑀢𑀺? “𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀇𑀘𑁆𑀙𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁, 𑀬𑀤𑀺𑀤𑀁 𑀅𑀗𑁆𑀕𑀡𑀦𑁆”𑀢𑀺𑁇

133\. “𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀆𑀧𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀯𑀢 𑀆𑀧𑀦𑁆𑀦𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀦 𑀘 𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 ‘𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

134\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀆𑀧𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀯𑀢 𑀆𑀧𑀦𑁆𑀦𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀅𑀦𑀼𑀭𑀳𑁄 𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀘𑁄𑀤𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦𑁄 𑀅𑀦𑀼𑀭𑀳𑁄𑁇 ‘𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀦𑀼𑀭𑀳𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

135\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀆𑀧𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀯𑀢 𑀆𑀧𑀦𑁆𑀦𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀲𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀁 𑀘𑁄𑀤𑁂𑀬𑁆𑀬, 𑀦𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀘𑁄𑀤𑁂𑀬𑁆𑀬, 𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇 ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀁 𑀘𑁄𑀤𑁂𑀢𑀺, 𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

136\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀫𑁂𑀯 𑀲𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑁇 ‘𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀦 𑀫𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀻’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

137\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀁 𑀪𑀢𑁆𑀢𑀸𑀬 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀁 𑀪𑀢𑁆𑀢𑀸𑀬 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀁 𑀪𑀢𑁆𑀢𑀸𑀬 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀁 𑀪𑀢𑁆𑀢𑀸𑀬 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀼𑀁𑁇 ‘𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀁 𑀪𑀢𑁆𑀢𑀸𑀬 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀁 𑀪𑀢𑁆𑀢𑀸𑀬 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

138\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀅𑀳𑀫𑁂𑀯 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀕𑁆𑀕𑀸𑀲𑀦𑀁 𑀅𑀕𑁆𑀕𑁄𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀧𑀺𑀡𑁆𑀟𑀁, 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀪𑁂𑀬𑁆𑀬 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀕𑁆𑀕𑀸𑀲𑀦𑀁 𑀅𑀕𑁆𑀕𑁄𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀧𑀺𑀡𑁆𑀟𑀦𑁆’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀪𑁂𑀬𑁆𑀬 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀕𑁆𑀕𑀸𑀲𑀦𑀁 𑀅𑀕𑁆𑀕𑁄𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀧𑀺𑀡𑁆𑀟𑀁, 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀪𑁂𑀬𑁆𑀬 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀕𑁆𑀕𑀸𑀲𑀦𑀁 𑀅𑀕𑁆𑀕𑁄𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀧𑀺𑀡𑁆𑀟𑀁𑁇 ‘𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀪𑀢𑀺 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀕𑁆𑀕𑀸𑀲𑀦𑀁 𑀅𑀕𑁆𑀕𑁄𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀧𑀺𑀡𑁆𑀟𑀁, 𑀦𑀸𑀳𑀁 𑀮𑀪𑀸𑀫𑀺 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀕𑁆𑀕𑀸𑀲𑀦𑀁 𑀅𑀕𑁆𑀕𑁄𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀧𑀺𑀡𑁆𑀟𑀦𑁆’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

139\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀅𑀳𑀫𑁂𑀯 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀅𑀦𑀼𑀫𑁄𑀤𑁂𑀬𑁆𑀬𑀁, 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀅𑀦𑀼𑀫𑁄𑀤𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀅𑀦𑀼𑀫𑁄𑀤𑁂𑀬𑁆𑀬, 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀅𑀦𑀼𑀫𑁄𑀤𑁂𑀬𑁆𑀬𑁇 ‘𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀅𑀦𑀼𑀫𑁄𑀤𑀢𑀺, 𑀦𑀸𑀳𑀁 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀅𑀦𑀼𑀫𑁄𑀤𑀸𑀫𑀻’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

140\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀅𑀳𑀫𑁂𑀯 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁, 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑁇 ‘𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀦𑀸𑀳𑀁 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀻’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

141\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀅𑀳𑀫𑁂𑀯 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁, 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑁇 ‘𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀦𑀸𑀳𑀁 𑀆𑀭𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀻’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

142\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 ‘𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

143\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀸…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 ‘𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀫𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

144\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀅𑀳𑀫𑁂𑀯 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲𑀁 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁, 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁, 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁𑁇 ‘𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁, 𑀦𑀸𑀳𑀁 𑀮𑀸𑀪𑀻 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀦𑁆’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

145\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀅𑀳𑀫𑁂𑀯 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲𑀁 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁, 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁, 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 ‘𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀻 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁, 𑀦𑀸𑀳𑀁 𑀮𑀸𑀪𑀻 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀦𑁆’𑀢𑀺—  𑀇𑀢𑀺 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄𑁇 𑀬𑁄 𑀘𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀧𑁄 𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀅𑀗𑁆𑀕𑀡𑀁𑁇

(62.)

149\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀉𑀧𑀫𑀸 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀝𑀺𑀪𑀸𑀢𑀻”𑀢𑀺𑁇 “𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸”𑀢𑀺𑁇 “𑀏𑀓𑀫𑀺𑀤𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑀬𑀁 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑁂𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑀻𑀢𑀺 𑀬𑀸𑀦𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀣𑀲𑁆𑀲 𑀦𑁂𑀫𑀺𑀁 𑀢𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀧𑀡𑁆𑀟𑀼𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀚𑀻𑀯𑀓𑁄 𑀧𑀼𑀭𑀸𑀡𑀬𑀸𑀦𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀡𑁆𑀟𑀼𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀬𑀸𑀦𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀬𑀁 𑀲𑀫𑀻𑀢𑀺 𑀬𑀸𑀦𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀦𑁂𑀫𑀺𑀬𑀸 𑀇𑀫𑀜𑁆𑀘 𑀯𑀗𑁆𑀓𑀁 𑀇𑀫𑀜𑁆𑀘 𑀚𑀺𑀫𑁆𑀳𑀁 𑀇𑀫𑀜𑁆𑀘 𑀤𑁄𑀲𑀁 𑀢𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀏𑀯𑀸𑀬𑀁 𑀦𑁂𑀫𑀺 𑀅𑀧𑀕𑀢𑀯𑀗𑁆𑀓𑀸 𑀅𑀧𑀕𑀢𑀚𑀺𑀫𑁆𑀳𑀸 𑀅𑀧𑀕𑀢𑀤𑁄𑀲𑀸 𑀲𑀼𑀤𑁆𑀥𑀸 𑀅𑀲𑁆𑀲 𑀲𑀸𑀭𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸’𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀡𑁆𑀟𑀼𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀬𑀸𑀦𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑀫𑀻𑀢𑀺 𑀬𑀸𑀦𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀦𑁂𑀫𑀺𑀬𑀸 𑀢𑀜𑁆𑀘 𑀯𑀗𑁆𑀓𑀁 𑀢𑀜𑁆𑀘 𑀚𑀺𑀫𑁆𑀳𑀁 𑀢𑀜𑁆𑀘 𑀤𑁄𑀲𑀁 𑀢𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀡𑁆𑀟𑀼𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀚𑀻𑀯𑀓𑁄 𑀧𑀼𑀭𑀸𑀡𑀬𑀸𑀦𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑀯𑀸𑀘𑀁 𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑁂𑀲𑀺—  ‘𑀳𑀤𑀬𑀸 𑀳𑀤𑀬𑀁 𑀫𑀜𑁆𑀜𑁂 𑀅𑀜𑁆𑀜𑀸𑀬 𑀢𑀘𑁆𑀙𑀢𑀻’𑀢𑀺𑁇

150\. 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁂 𑀢𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸, 𑀚𑀻𑀯𑀺𑀓𑀢𑁆𑀣𑀸 𑀦 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀲𑀞𑀸 𑀫𑀸𑀬𑀸𑀯𑀺𑀦𑁄 𑀓𑁂𑀢𑀩𑀺𑀦𑁄 𑀉𑀤𑁆𑀥𑀢𑀸 𑀉𑀦𑁆𑀦𑀍𑀆 𑀘𑀧𑀮𑀸 𑀫𑀼𑀔𑀭𑀸 𑀯𑀺𑀓𑀺𑀡𑁆𑀡𑀯𑀸𑀘𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀸, 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀦𑁄, 𑀚𑀸𑀕𑀭𑀺𑀬𑀁 𑀅𑀦𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀸𑀫𑀜𑁆𑀜𑁂 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀯𑀦𑁆𑀢𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀦 𑀢𑀺𑀩𑁆𑀩𑀕𑀸𑀭𑀯𑀸, 𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸 𑀲𑀸𑀣𑀮𑀺𑀓𑀸, 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸, 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸, 𑀓𑀼𑀲𑀻𑀢𑀸 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑀸 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀻 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀏𑀍𑀅𑀫𑀽𑀕𑀸, 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀇𑀫𑀺𑀦𑀸 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀳𑀤𑀬𑀸 𑀳𑀤𑀬𑀁 𑀫𑀜𑁆𑀜𑁂 𑀅𑀜𑁆𑀜𑀸𑀬 𑀢𑀘𑁆𑀙𑀢𑀺𑁇

151\. 𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀅𑀲𑀞𑀸 𑀅𑀫𑀸𑀬𑀸𑀯𑀺𑀦𑁄 𑀅𑀓𑁂𑀢𑀩𑀺𑀦𑁄 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑀸 𑀅𑀦𑀼𑀦𑁆𑀦𑀍𑀆 𑀅𑀘𑀧𑀮𑀸 𑀅𑀫𑀼𑀔𑀭𑀸 𑀅𑀯𑀺𑀓𑀺𑀡𑁆𑀡𑀯𑀸𑀘𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀸, 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀦𑁄, 𑀚𑀸𑀕𑀭𑀺𑀬𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀸𑀫𑀜𑁆𑀜𑁂 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀦𑁆𑀢𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀢𑀺𑀩𑁆𑀩𑀕𑀸𑀭𑀯𑀸, 𑀦 𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸 𑀦 𑀲𑀸𑀣𑀮𑀺𑀓𑀸, 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸, 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸 𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁄 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑀸, 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀺𑀯𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂, 𑀖𑀲𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀯𑀘𑀲𑀸 𑀘𑁂𑀯 𑀫𑀦𑀲𑀸 𑀘—  ‘𑀲𑀸𑀥𑀼 𑀯𑀢, 𑀪𑁄, 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀓𑀼𑀲𑀮𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀼𑀲𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀻’𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀢𑁆𑀣𑀻 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀸 𑀤𑀳𑀭𑁄 𑀬𑀼𑀯𑀸 𑀫𑀡𑁆𑀟𑀦𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑁄 𑀉𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀁 𑀯𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀫𑀸𑀮𑀁 𑀯𑀸 𑀅𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀓𑀫𑀸𑀮𑀁 𑀯𑀸 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀉𑀪𑁄𑀳𑀺 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑁂 𑀲𑀺𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀧𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁂 𑀢𑁂 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸, 𑀅𑀲𑀞𑀸 𑀅𑀫𑀸𑀬𑀸𑀯𑀺𑀦𑁄 𑀅𑀓𑁂𑀢𑀩𑀺𑀦𑁄 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑀸 𑀅𑀦𑀼𑀦𑁆𑀦𑀍𑀆 𑀅𑀘𑀧𑀮𑀸 𑀅𑀫𑀼𑀔𑀭𑀸 𑀅𑀯𑀺𑀓𑀺𑀡𑁆𑀡𑀯𑀸𑀘𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀸, 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀦𑁄, 𑀚𑀸𑀕𑀭𑀺𑀬𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀸𑀫𑀜𑁆𑀜𑁂 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀦𑁆𑀢𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀢𑀺𑀩𑁆𑀩𑀕𑀸𑀭𑀯𑀸, 𑀦 𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸 𑀦 𑀲𑀸𑀣𑀮𑀺𑀓𑀸, 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸, 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸 𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁄 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑀸, 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀺𑀯𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂, 𑀖𑀲𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀯𑀘𑀲𑀸 𑀘𑁂𑀯 𑀫𑀦𑀲𑀸 𑀘—  ‘𑀲𑀸𑀥𑀼 𑀯𑀢, 𑀪𑁄, 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀓𑀼𑀲𑀮𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀼𑀲𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀻’”𑀢𑀺𑁇 𑀇𑀢𑀺𑀳 𑀢𑁂 𑀉𑀪𑁄 𑀫𑀳𑀸𑀦𑀸𑀕𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀲𑀫𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀽𑀢𑀺𑁇
