# 𑁧𑁇𑁨 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

(14.)

51\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀁𑀯𑀭𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

(15.)

52\. “𑀚𑀸𑀦𑀢𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀲𑁆𑀲𑀢𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀁 𑀯𑀤𑀸𑀫𑀺, 𑀦𑁄 𑀅𑀚𑀸𑀦𑀢𑁄 𑀦𑁄 𑀅𑀧𑀲𑁆𑀲𑀢𑁄𑁇 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀦𑀢𑁄 𑀓𑀺𑀜𑁆𑀘 𑀧𑀲𑁆𑀲𑀢𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀁 𑀯𑀤𑀸𑀫𑀺? 𑀬𑁄𑀦𑀺𑀲𑁄 𑀘 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀁 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀘 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀁𑁇 𑀅𑀬𑁄𑀦𑀺𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺; 𑀬𑁄𑀦𑀺𑀲𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

(16.)

53\. 𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀸 𑀲𑀁𑀯𑀭𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀸 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀦𑁄𑀤𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀸 𑀪𑀸𑀯𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇
