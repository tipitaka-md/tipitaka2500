# 𑁧𑁇𑁨𑁇𑁭 𑀪𑀸𑀯𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀸𑀯𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

(27.)

78\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀪𑀸𑀯𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀬𑁄𑀦𑀺𑀲𑁄 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀁; 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀬𑁄𑀦𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀁𑁇

79\. 𑀬𑀜𑁆𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀸𑀯𑀬𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀪𑀸𑀯𑀬𑀢𑁄 𑀏𑀯𑀁𑀲 𑀢𑁂 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀪𑀸𑀯𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇

(28.)

80\. 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑁂 𑀆𑀲𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀤𑀲𑁆𑀲𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀆𑀲𑀯𑀸 𑀲𑀁𑀯𑀭𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀲𑀁𑀯𑀭𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀆𑀲𑀯𑀸 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀆𑀲𑀯𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀆𑀲𑀯𑀸 𑀯𑀺𑀦𑁄𑀤𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀯𑀺𑀦𑁄𑀤𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀆𑀲𑀯𑀸 𑀪𑀸𑀯𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀪𑀸𑀯𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺; 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀢𑀡𑁆𑀳𑀁, 𑀯𑀺𑀯𑀢𑁆𑀢𑀬𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀫𑁆𑀫𑀸 𑀫𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀸 𑀅𑀦𑁆𑀢𑀫𑀓𑀸𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸’”𑀢𑀺𑁇

81\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

82\. 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
