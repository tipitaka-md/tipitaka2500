# 𑁧𑁇𑁨𑁇𑁪 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

(24.)

72\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀬𑁄𑀦𑀺𑀲𑁄 𑀔𑀫𑁄 𑀳𑁄𑀢𑀺 𑀲𑀻𑀢𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑀲𑁆𑀲, 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀬 𑀧𑀺𑀧𑀸𑀲𑀸𑀬𑁇 𑀟𑀁𑀲𑀫𑀓𑀲𑀯𑀸𑀢𑀸𑀢𑀧𑀲𑀭𑀻𑀲𑀧𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀦𑀁, 𑀤𑀼𑀭𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀼𑀭𑀸𑀕𑀢𑀸𑀦𑀁 𑀯𑀘𑀦𑀧𑀣𑀸𑀦𑀁, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀁 𑀢𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀔𑀭𑀸𑀦𑀁 𑀓𑀝𑀼𑀓𑀸𑀦𑀁 𑀅𑀲𑀸𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀸𑀧𑀸𑀦𑀁 𑀧𑀸𑀡𑀳𑀭𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀸𑀲𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

73\. 𑀬𑀜𑁆𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀥𑀺𑀯𑀸𑀲𑀬𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀅𑀥𑀺𑀯𑀸𑀲𑀬𑀢𑁄 𑀏𑀯𑀁𑀲 𑀢𑁂 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇
