# 𑁧𑁇𑁨𑁇𑁧 𑀤𑀲𑁆𑀲𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀲𑁆𑀲𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

(17.)

54\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄—  𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄—  𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀅𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑁄, 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇

55\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺? 𑀬𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇

56\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺? 𑀬𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀧𑀳𑀻𑀬𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀧𑀳𑀻𑀬𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀧𑀳𑀻𑀬𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇

57\. 𑀢𑀲𑁆𑀲 𑀅𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺𑁇

(18.)

58\. 𑀲𑁄 𑀏𑀯𑀁 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺—  ‘𑀅𑀳𑁄𑀲𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀦 𑀦𑀼 𑀔𑁄 𑀅𑀳𑁄𑀲𑀺𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑁄𑀲𑀺𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑁄𑀲𑀺𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀓𑀺𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀓𑀺𑀁 𑀅𑀳𑁄𑀲𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀦 𑀦𑀼 𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁? 𑀓𑀺𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀓𑀺𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀦𑁆’𑀢𑀺? 𑀏𑀢𑀭𑀳𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀣𑀁𑀓𑀣𑀻 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺? 𑀦𑁄 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺? 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺? 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄𑀲𑁆𑀫𑀺? 𑀅𑀬𑀁 𑀦𑀼 𑀔𑁄 𑀲𑀢𑁆𑀢𑁄 𑀓𑀼𑀢𑁄 𑀆𑀕𑀢𑁄? 𑀲𑁄 𑀓𑀼𑀳𑀺𑀁 𑀕𑀸𑀫𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺?

(19.)

59\. 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀙𑀦𑁆𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀲𑀘𑁆𑀘𑀢𑁄 𑀣𑁂𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀲𑀘𑁆𑀘𑀢𑁄 𑀣𑁂𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; ‘𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀲𑀘𑁆𑀘𑀢𑁄 𑀣𑁂𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; ‘𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀲𑀘𑁆𑀘𑀢𑁄 𑀣𑁂𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; ‘𑀅𑀦𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀲𑀘𑁆𑀘𑀢𑁄 𑀣𑁂𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀣 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺—  ‘𑀬𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀢𑁆𑀢𑀸 𑀯𑀤𑁄 𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑁄 𑀢𑀢𑁆𑀭 𑀢𑀢𑁆𑀭 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀲𑁄 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀢𑁆𑀢𑀸 𑀦𑀺𑀘𑁆𑀘𑁄 𑀥𑀼𑀯𑁄 𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄 𑀲𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀁 𑀢𑀣𑁂𑀯 𑀞𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀦𑁆𑀢𑀸𑀭𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀽𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺; ‘𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

(20.)

60\. 𑀲𑀼𑀢𑀯𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄—  𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑁄𑀯𑀺𑀤𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑁄, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑁄𑀯𑀺𑀤𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑁄—  𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀅𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀅𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇

61\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺? 𑀬𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸, 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇

62\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺? 𑀬𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀧𑀳𑀻𑀬𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀪𑀯𑀸𑀲𑀯𑁄 𑀧𑀳𑀻𑀬𑀢𑀺; 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀧𑀳𑀻𑀬𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇

63\. 𑀢𑀲𑁆𑀲 𑀅𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇

(21.)

64\. 𑀲𑁄 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀢𑀻𑀡𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺—  𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇
