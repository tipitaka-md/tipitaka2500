# 𑁧𑁇𑁨𑁇𑁫 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀆𑀲𑀯

(25.)

74\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀬𑁄𑀦𑀺𑀲𑁄 𑀘𑀡𑁆𑀟𑀁 𑀳𑀢𑁆𑀣𑀺𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺, 𑀘𑀡𑁆𑀟𑀁 𑀅𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺, 𑀘𑀡𑁆𑀟𑀁 𑀕𑁄𑀡𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺, 𑀘𑀡𑁆𑀟𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺, 𑀅𑀳𑀺𑀁 𑀔𑀸𑀡𑀼𑀁 𑀓𑀡𑁆𑀝𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀲𑁄𑀩𑁆𑀪𑀁 𑀧𑀧𑀸𑀢𑀁 𑀘𑀦𑁆𑀤𑀦𑀺𑀓𑀁 𑀑𑀍𑀇𑀕𑀮𑁆𑀮𑀁𑁇 𑀬𑀣𑀸𑀭𑀽𑀧𑁂 𑀅𑀦𑀸𑀲𑀦𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀬𑀣𑀸𑀭𑀽𑀧𑁂 𑀅𑀕𑁄𑀘𑀭𑁂 𑀘𑀭𑀦𑁆𑀢𑀁 𑀬𑀣𑀸𑀭𑀽𑀧𑁂 𑀧𑀸𑀧𑀓𑁂 𑀫𑀺𑀢𑁆𑀢𑁂 𑀪𑀚𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀽 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀧𑀸𑀧𑀓𑁂𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼 𑀑𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑁄 𑀢𑀜𑁆𑀘 𑀅𑀦𑀸𑀲𑀦𑀁 𑀢𑀜𑁆𑀘 𑀅𑀕𑁄𑀘𑀭𑀁 𑀢𑁂 𑀘 𑀧𑀸𑀧𑀓𑁂 𑀫𑀺𑀢𑁆𑀢𑁂 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀬𑁄𑀦𑀺𑀲𑁄 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺𑁇

75\. 𑀬𑀜𑁆𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀢𑁄 𑀏𑀯𑀁𑀲 𑀢𑁂 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇
