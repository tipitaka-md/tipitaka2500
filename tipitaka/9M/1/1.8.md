# 𑁧𑁇𑁮 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

(81.)

192. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀘𑀼𑀦𑁆𑀤𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀘𑀼𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀸 𑀇𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀸 𑀮𑁄𑀓𑀯𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀸—  𑀆𑀤𑀺𑀫𑁂𑀯 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀏𑀯𑀫𑁂𑀢𑀸𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀻𑀦𑀁 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀫𑁂𑀢𑀸𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀻𑀦𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺?

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

(82.)

193. “𑀬𑀸 𑀇𑀫𑀸, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀸 𑀮𑁄𑀓𑀯𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀸—  𑀬𑀢𑁆𑀣 𑀘𑁂𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀬𑀢𑁆𑀣 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀬𑀢𑁆𑀣 𑀘 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀢𑀁 ‘𑀦𑁂𑀢𑀁 𑀫𑀫, 𑀦𑁂𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦 𑀫𑁂 𑀲𑁄 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺—  𑀏𑀯𑀫𑁂𑀢𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀧𑀲𑁆𑀲𑀢𑁄 𑀏𑀯𑀫𑁂𑀢𑀸𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀻𑀦𑀁 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀫𑁂𑀢𑀸𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀻𑀦𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

194. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸 𑀏𑀢𑁂 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

195. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸 𑀏𑀢𑁂 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

196. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬, 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀬𑁆𑀬, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸 𑀏𑀢𑁂 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

197. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸 𑀏𑀢𑁂 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

198. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸, 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸, 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸, ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸 𑀏𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑀸 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

199. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸 𑀏𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑀸 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

200. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸 𑀏𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑀸 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑀼𑀢𑁆𑀢

201. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀮𑁆𑀮𑁂𑀔𑁂𑀦 𑀯𑀺𑀳𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸 𑀏𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑀸 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇
