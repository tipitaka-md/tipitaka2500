# 𑁧𑁇𑁧𑁦𑁇𑁧𑁇𑁧 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀆𑀦𑀸𑀧𑀸𑀦𑀧𑀩𑁆𑀩

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

### 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀆𑀦𑀸𑀧𑀸𑀦𑀧𑀩𑁆𑀩

(107.)

305. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸, 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬, 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇 𑀲𑁄 𑀲𑀢𑁄𑀯 𑀅𑀲𑁆𑀲𑀲𑀢𑀺, 𑀲𑀢𑁄𑀯 𑀧𑀲𑁆𑀲𑀲𑀢𑀺𑁇 𑀤𑀻𑀖𑀁 𑀯𑀸 𑀅𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀤𑀻𑀖𑀁 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀻𑀖𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀤𑀻𑀖𑀁 𑀧𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀲𑁆𑀲𑀁 𑀯𑀸 𑀅𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀭𑀲𑁆𑀲𑀁 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀲𑁆𑀲𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀭𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

306. ‘𑀲𑀩𑁆𑀩𑀓𑀸𑀬𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀲𑀩𑁆𑀩𑀓𑀸𑀬𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇

307. ‘𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀬𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀬𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇

308. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀓𑁆𑀔𑁄 𑀪𑀫𑀓𑀸𑀭𑁄 𑀯𑀸 𑀪𑀫𑀓𑀸𑀭𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀻 𑀯𑀸 𑀤𑀻𑀖𑀁 𑀯𑀸 𑀅𑀜𑁆𑀙𑀦𑁆𑀢𑁄 ‘𑀤𑀻𑀖𑀁 𑀅𑀜𑁆𑀙𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀲𑁆𑀲𑀁 𑀯𑀸 𑀅𑀜𑁆𑀙𑀦𑁆𑀢𑁄 ‘𑀭𑀲𑁆𑀲𑀁 𑀅𑀜𑁆𑀙𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀻𑀖𑀁 𑀯𑀸 𑀅𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀤𑀻𑀖𑀁 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀻𑀖𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀤𑀻𑀖𑀁 𑀧𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀲𑁆𑀲𑀁 𑀯𑀸 𑀅𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀭𑀲𑁆𑀲𑀁 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀲𑁆𑀲𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀭𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺; ‘𑀲𑀩𑁆𑀩𑀓𑀸𑀬𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀲𑀩𑁆𑀩𑀓𑀸𑀬𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺; ‘𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀬𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀬𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇

309. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 ‘𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑁄’𑀢𑀺 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸𑀯𑀤𑁂𑀯 𑀜𑀸𑀡𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

---

310. 𑀆𑀦𑀸𑀧𑀸𑀦𑀧𑀩𑁆𑀩𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
