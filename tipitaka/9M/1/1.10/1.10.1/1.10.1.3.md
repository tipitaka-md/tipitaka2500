# 𑁧𑁇𑁧𑁦𑁇𑁧𑁇𑁩 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀧𑀩𑁆𑀩

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

### 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀧𑀩𑁆𑀩

(109.)

314\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀆𑀮𑁄𑀓𑀺𑀢𑁂 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂 𑀧𑀲𑀸𑀭𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀺𑀢𑁂 𑀧𑀻𑀢𑁂 𑀔𑀸𑀬𑀺𑀢𑁂 𑀲𑀸𑀬𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑀓𑀫𑁆𑀫𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀢𑁂 𑀞𑀺𑀢𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀚𑀸𑀕𑀭𑀺𑀢𑁂 𑀪𑀸𑀲𑀺𑀢𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇

315\. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

---

316\. 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀧𑀩𑁆𑀩𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
