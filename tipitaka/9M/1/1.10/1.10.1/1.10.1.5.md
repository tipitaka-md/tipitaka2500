# 𑁧𑁇𑁧𑁦𑁇𑁧𑁇𑁫 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀥𑀸𑀢𑀼𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀧𑀩𑁆𑀩

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

* 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀥𑀸𑀢𑀼𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀧𑀩𑁆𑀩

(111.)

321\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑀫𑁂𑀯 𑀓𑀸𑀬𑀁 𑀬𑀣𑀸𑀞𑀺𑀢𑀁 𑀬𑀣𑀸𑀧𑀡𑀺𑀳𑀺𑀢𑀁 𑀥𑀸𑀢𑀼𑀲𑁄 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀽’𑀢𑀺𑁇

322\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀓𑁆𑀔𑁄 𑀕𑁄𑀖𑀸𑀢𑀓𑁄 𑀯𑀸 𑀕𑁄𑀖𑀸𑀢𑀓𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀻 𑀯𑀸 𑀕𑀸𑀯𑀺𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀫𑀳𑀸𑀧𑀣𑁂 𑀩𑀺𑀮𑀲𑁄 𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀅𑀲𑁆𑀲𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑀫𑁂𑀯 𑀓𑀸𑀬𑀁 𑀬𑀣𑀸𑀞𑀺𑀢𑀁 𑀬𑀣𑀸𑀧𑀡𑀺𑀳𑀺𑀢𑀁 𑀥𑀸𑀢𑀼𑀲𑁄 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀽’𑀢𑀺𑁇

323\. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

---

324\. 𑀥𑀸𑀢𑀼𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀧𑀩𑁆𑀩𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
