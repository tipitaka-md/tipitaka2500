# 𑁧𑁇𑁧𑁦𑁇𑁪𑁇𑁩 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀆𑀬𑀢𑀦𑀧𑀩𑁆𑀩

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀆𑀬𑀢𑀦𑀧𑀩𑁆𑀩

(117.)

375\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀙𑀲𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀩𑀸𑀳𑀺𑀭𑁂𑀲𑀼 𑀆𑀬𑀢𑀦𑁂𑀲𑀼𑁇 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀙𑀲𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀩𑀸𑀳𑀺𑀭𑁂𑀲𑀼 𑀆𑀬𑀢𑀦𑁂𑀲𑀼?

376\. 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀽𑀧𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧)

377\. 𑀲𑁄𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀤𑁆𑀤𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁨)

378\. 𑀖𑀸𑀦𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀕𑀦𑁆𑀥𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁩)

379\. 𑀚𑀺𑀯𑁆𑀳𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀲𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁪)

380\. 𑀓𑀸𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁫)

381\. 𑀫𑀦𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁬)

382\. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 ‘𑀅𑀢𑁆𑀣𑀺 𑀥𑀫𑁆𑀫𑀸’𑀢𑀺 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸𑀯𑀤𑁂𑀯 𑀜𑀸𑀡𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀙𑀲𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀩𑀸𑀳𑀺𑀭𑁂𑀲𑀼 𑀆𑀬𑀢𑀦𑁂𑀲𑀼𑁇
