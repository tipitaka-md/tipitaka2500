# 𑁧𑁇𑁧𑁦𑁇𑁪𑁇𑁪 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀧𑀩𑁆𑀩

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀧𑀩𑁆𑀩

(118.)

384\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁆𑀢𑀲𑀼 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂𑀲𑀼𑁇 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁆𑀢𑀲𑀼 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂𑀲𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧)

385\. 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁨)

386\. 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁩)

387\. 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁪)

388\. 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁫)

389\. 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁬)

390\. 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 ‘𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁭)

391\. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 ‘𑀅𑀢𑁆𑀣𑀺 𑀥𑀫𑁆𑀫𑀸’𑀢𑀺 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸𑀯𑀤𑁂𑀯 𑀜𑀸𑀡𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁆𑀢𑀲𑀼 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂𑀲𑀼𑁇
