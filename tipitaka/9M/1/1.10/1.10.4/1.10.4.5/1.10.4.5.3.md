# 𑁧𑁇𑁧𑁦𑁇𑁪𑁇𑁫𑁇𑁩 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀲𑀘𑁆𑀘𑀧𑀩𑁆𑀩

* 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(134.)

420\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁? 𑀬𑁄 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑁄 𑀘𑀸𑀕𑁄 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀦𑀸𑀮𑀬𑁄𑁇

421\. 𑀲𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀡𑁆𑀳𑀸 𑀓𑀢𑁆𑀣 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀓𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

422\. 𑀓𑀺𑀜𑁆𑀘 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀼 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑁄𑀢𑀁 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀖𑀸𑀦𑀁 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑁄 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

423\. 𑀭𑀽𑀧𑀸 𑀮𑁄𑀓𑁂…  𑀲𑀤𑁆𑀤𑀸 𑀮𑁄𑀓𑁂…  𑀕𑀦𑁆𑀥𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

424\. 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

425\. 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

426\. 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

427\. 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀕𑀦𑁆𑀥𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

428\. 𑀭𑀽𑀧𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀕𑀦𑁆𑀥𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

429\. 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

430\. 𑀭𑀽𑀧𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀲𑀤𑁆𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀕𑀦𑁆𑀥𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

431\. 𑀭𑀽𑀧𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂…  𑀲𑀤𑁆𑀤𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂…  𑀕𑀦𑁆𑀥𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁𑁇
