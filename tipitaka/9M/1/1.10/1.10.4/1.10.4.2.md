# 𑁧𑁇𑁧𑁦𑁇𑁪𑁇𑁨 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀔𑀦𑁆𑀥𑀧𑀩𑁆𑀩

* 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀔𑀦𑁆𑀥𑀧𑀩𑁆𑀩

(116.)

373\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼𑁇 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼—  ‘𑀇𑀢𑀺 𑀭𑀽𑀧𑀁, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸, 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸, 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀬 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄’𑀢𑀺; 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 ‘𑀅𑀢𑁆𑀣𑀺 𑀥𑀫𑁆𑀫𑀸’𑀢𑀺 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸𑀯𑀤𑁂𑀯 𑀜𑀸𑀡𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼𑁇
