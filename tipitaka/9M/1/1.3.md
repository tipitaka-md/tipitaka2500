# 𑁧𑁇𑁩 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(29.)

83\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

84\. “𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑀣, 𑀫𑀸 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂𑀲𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀫𑁂 𑀲𑀸𑀯𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦𑁄 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸’𑀢𑀺𑁇 𑀢𑀼𑀫𑁆𑀳𑁂 𑀘 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀣 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸, 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀢𑁂𑀦 𑀆𑀤𑀺𑀬𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀣—  ‘𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸’𑀢𑀺; 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑁂𑀦 𑀆𑀤𑀺𑀬𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀁—  ‘𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸’𑀢𑀺𑁇 𑀢𑀼𑀫𑁆𑀳𑁂 𑀘 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀣, 𑀦𑁄 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸, 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀢𑁂𑀦 𑀦 𑀆𑀤𑀺𑀬𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀣—  ‘𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸’𑀢𑀺; 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑁂𑀦 𑀦 𑀆𑀤𑀺𑀬𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀁—  ‘𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀪𑀯𑀣, 𑀫𑀸 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂𑀲𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀫𑁂 𑀲𑀸𑀯𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦𑁄 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸’𑀢𑀺𑁇

(30.)

85\. 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀅𑀲𑁆𑀲𑀁 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑁄 𑀲𑀼𑀳𑀺𑀢𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄; 𑀲𑀺𑀬𑀸 𑀘 𑀫𑁂 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀅𑀢𑀺𑀭𑁂𑀓𑀥𑀫𑁆𑀫𑁄 𑀙𑀟𑁆𑀟𑀦𑀻𑀬𑀥𑀫𑁆𑀫𑁄𑁇 𑀅𑀣 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀧𑀭𑁂𑀢𑀸𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄𑀫𑁆𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑁄 𑀲𑀼𑀳𑀺𑀢𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄; 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀫𑁂 𑀅𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀅𑀢𑀺𑀭𑁂𑀓𑀥𑀫𑁆𑀫𑁄 𑀙𑀟𑁆𑀟𑀦𑀻𑀬𑀥𑀫𑁆𑀫𑁄𑁇 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀣, 𑀪𑀼𑀜𑁆𑀚𑀣, 𑀦𑁄 𑀘𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀣, 𑀇𑀤𑀸𑀦𑀸𑀳𑀁 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀑𑀧𑀺𑀮𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇

86\. 𑀢𑀢𑁆𑀭𑁂𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀪𑀕𑀯𑀸 𑀔𑁄 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑁄 𑀲𑀼𑀳𑀺𑀢𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄; 𑀅𑀢𑁆𑀣𑀺 𑀘𑀸𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀅𑀢𑀺𑀭𑁂𑀓𑀥𑀫𑁆𑀫𑁄 𑀙𑀟𑁆𑀟𑀦𑀻𑀬𑀥𑀫𑁆𑀫𑁄𑁇 𑀲𑀘𑁂 𑀫𑀬𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀇𑀤𑀸𑀦𑀺 𑀪𑀕𑀯𑀸 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀑𑀧𑀺𑀮𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  “𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑀣, 𑀫𑀸 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸”𑀢𑀺𑁇 𑀆𑀫𑀺𑀲𑀜𑁆𑀜𑀢𑀭𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀬𑀤𑀺𑀤𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀅𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑀺𑀦𑀸𑀯 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑁂𑀦 𑀏𑀯𑀁 𑀇𑀫𑀁 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀅𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀦𑁂𑀯 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑁂𑀦 𑀏𑀯𑀁 𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀬𑁆𑀬𑁇

87\. 𑀅𑀣 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀪𑀕𑀯𑀸 𑀔𑁄 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑁄 𑀲𑀼𑀳𑀺𑀢𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄; 𑀅𑀢𑁆𑀣𑀺 𑀘𑀸𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀅𑀢𑀺𑀭𑁂𑀓𑀥𑀫𑁆𑀫𑁄 𑀙𑀟𑁆𑀟𑀦𑀻𑀬𑀥𑀫𑁆𑀫𑁄𑁇 𑀲𑀘𑁂 𑀫𑀬𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀇𑀤𑀸𑀦𑀺 𑀪𑀕𑀯𑀸 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀑𑀧𑀺𑀮𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀇𑀫𑀁 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀬𑁆𑀬𑁇

88\. 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀔𑁄 𑀅𑀲𑀼𑀬𑁂𑀯 𑀫𑁂 𑀧𑀼𑀭𑀺𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀚𑁆𑀚𑀢𑀭𑁄 𑀘 𑀧𑀸𑀲𑀁𑀲𑀢𑀭𑁄 𑀘𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀜𑁆𑀳𑀺 𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀮𑁆𑀮𑁂𑀔𑀸𑀬 𑀲𑀼𑀪𑀭𑀢𑀸𑀬 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀪𑀯𑀣, 𑀫𑀸 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂𑀲𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀫𑁂 𑀲𑀸𑀯𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦𑁄 𑀆𑀫𑀺𑀲𑀤𑀸𑀬𑀸𑀤𑀸’”𑀢𑀺𑁇

89\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

90\. 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀯𑀼𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺𑁇 “𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

91\. “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀁 𑀦𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀘 𑀧𑀦 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀤𑀽𑀭𑀢𑁄𑀧𑀺 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀕𑀘𑁆𑀙𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀏𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀢𑀼𑀁𑁇 𑀲𑀸𑀥𑀼 𑀯𑀢𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀏𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄; 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀸𑀯𑀼𑀲𑁄, 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

92\. “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀁 𑀦𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀁 𑀦𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀬𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀳𑀸𑀦𑀫𑀸𑀳, 𑀢𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺, 𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀲𑀸𑀣𑀮𑀺𑀓𑀸, 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸, 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀁 𑀦𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀧𑀞𑀫𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀬𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀳𑀸𑀦𑀫𑀸𑀳 𑀢𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸 𑀘, 𑀲𑀸𑀣𑀮𑀺𑀓𑀸, 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸, 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀢𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀣𑁂𑀭𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀽…𑀧𑁂…  𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀁 𑀦𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀧𑀞𑀫𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀬𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀳𑀸𑀦𑀫𑀸𑀳 𑀢𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀲𑀸𑀣𑀮𑀺𑀓𑀸, 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸, 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀢𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀦𑀯𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀁 𑀦𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇

93\. 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀘 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  𑀬𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀳𑀸𑀦𑀫𑀸𑀳 𑀢𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀧𑀚𑀳𑀦𑁆𑀢𑀺; 𑀦 𑀘 𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀸𑀣𑀮𑀺𑀓𑀸 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀧𑀞𑀫𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀬𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀳𑀸𑀦𑀫𑀸𑀳 𑀢𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀧𑀚𑀳𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀦 𑀘 𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸, 𑀦 𑀲𑀸𑀣𑀮𑀺𑀓𑀸 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀢𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀣𑁂𑀭𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀽…𑀧𑁂…  𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀧𑀞𑀫𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀬𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀣𑀸 𑀧𑀳𑀸𑀦𑀫𑀸𑀳 𑀢𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀧𑀚𑀳𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 ‘𑀦 𑀘 𑀩𑀸𑀳𑀼𑀮𑀺𑀓𑀸, 𑀦 𑀲𑀸𑀣𑀮𑀺𑀓𑀸 𑀑𑀓𑁆𑀓𑀫𑀦𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀸 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸’𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀢𑀢𑀺𑀬𑁂𑀦 𑀞𑀸𑀦𑁂𑀦 𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀦𑀯𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀲𑀁𑀲𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇

(33.)

94\. 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀮𑁄𑀪𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄 𑀤𑁄𑀲𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄𑁇 𑀮𑁄𑀪𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸𑀬 𑀤𑁄𑀲𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸𑀬 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀘 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

95\. 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀓𑁄𑀥𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄 𑀉𑀧𑀦𑀸𑀳𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄…𑀧𑁂…  𑀫𑀓𑁆𑀔𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄 𑀧𑀍𑀆𑀲𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄, 𑀇𑀲𑁆𑀲𑀸 𑀘 𑀧𑀸𑀧𑀺𑀓𑀸 𑀫𑀘𑁆𑀙𑁂𑀭𑀜𑁆𑀘 𑀧𑀸𑀧𑀓𑀁, 𑀫𑀸𑀬𑀸 𑀘 𑀧𑀸𑀧𑀺𑀓𑀸 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀧𑀸𑀧𑀓𑀁, 𑀣𑀫𑁆𑀪𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄 𑀲𑀸𑀭𑀫𑁆𑀪𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄, 𑀫𑀸𑀦𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄 𑀅𑀢𑀺𑀫𑀸𑀦𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄, 𑀫𑀤𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄 𑀧𑀫𑀸𑀤𑁄 𑀘 𑀧𑀸𑀧𑀓𑁄𑁇 𑀫𑀤𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀫𑀸𑀤𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸𑀬 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀘 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀓𑀭𑀡𑀻 𑀜𑀸𑀡𑀓𑀭𑀡𑀻 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇

96\. 𑀇𑀤𑀫𑀯𑁄𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

97\. 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
