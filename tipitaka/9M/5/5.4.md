# 𑁫𑁇𑁪 𑀘𑀽𑀍𑀅𑀯𑁂𑀤𑀮𑁆𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀽𑀍𑀅𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀘𑀽𑀍𑀅𑀯𑁂𑀤𑀮𑁆𑀮𑀲𑀼𑀢𑁆𑀢

(460.)

1577\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “‘𑀲𑀓𑁆𑀓𑀸𑀬𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑁄’𑀢𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑀸𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺? “𑀧𑀜𑁆𑀘 𑀔𑁄 𑀇𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀲𑀓𑁆𑀓𑀸𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀲𑀓𑁆𑀓𑀸𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇

1578\. “𑀲𑀸𑀥𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺—  “‘𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑀫𑀼𑀤𑀬𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑀫𑀼𑀤𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺? “𑀬𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀢𑀡𑁆𑀳𑀸 𑀧𑁄𑀦𑁄𑀩𑁆𑀪𑀯𑀺𑀓𑀸 𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑀲𑀳𑀕𑀢𑀸 𑀢𑀢𑁆𑀭𑀢𑀢𑁆𑀭𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀦𑀻, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸; 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑀫𑀼𑀤𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇

1579\. “‘𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺?

1580\. “𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑁄 𑀘𑀸𑀕𑁄 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀦𑀸𑀮𑀬𑁄; 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇

1581\. “‘𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺?

1582\. “𑀅𑀬𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀻”𑀢𑀺𑁇

1583\. “𑀢𑀜𑁆𑀜𑁂𑀯 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀢𑁂 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀉𑀤𑀸𑀳𑀼 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺? “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀢𑀜𑁆𑀜𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀢𑁂 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸, 𑀦𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

1584\. “𑀓𑀣𑀁 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀇𑀥𑀸𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄, 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄, 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁, 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1599\. “𑀓𑀲𑁆𑀫𑀸 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀓𑀲𑁆𑀫𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀓𑀲𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑀢𑀺?

1600\. “𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀓𑀸𑀬𑀺𑀓𑀸 𑀏𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀸, 𑀢𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀯𑀸𑀘𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀏𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀸, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀘 𑀯𑁂𑀤𑀦𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑀢𑀺𑁇

(464.)

1601\. “𑀓𑀣𑀁 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀻”𑀢𑀺?

1602\. “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆’𑀢𑀺 𑀯𑀸, ‘𑀅𑀳𑀁 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺 𑀯𑀸, ‘𑀅𑀳𑀁 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄’𑀢𑀺 𑀯𑀸𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀢𑀣𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀉𑀧𑀦𑁂𑀢𑀻”𑀢𑀺𑁇

1603\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀞𑀫𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺—  𑀬𑀤𑀺 𑀯𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀬𑀤𑀺 𑀯𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀬𑀤𑀺 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑀢𑀺?

1604\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀞𑀫𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀢𑀢𑁄 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑀢𑀺𑁇

1605\. “𑀓𑀣𑀁 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺?

1606\. “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀦𑁆’𑀢𑀺 𑀯𑀸, ‘𑀅𑀳𑀁 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀸𑀫𑀻’𑀢𑀺 𑀯𑀸, ‘𑀅𑀳𑀁 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄’𑀢𑀺 𑀯𑀸𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀢𑀣𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀉𑀧𑀦𑁂𑀢𑀻”𑀢𑀺𑁇

1607\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀞𑀫𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀬𑀤𑀺 𑀯𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀬𑀤𑀺 𑀯𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀬𑀤𑀺 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑀢𑀺?

1608\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀞𑀫𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀢𑀢𑁄 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀢𑀢𑁄 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑀢𑀺𑁇

1609\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑀢𑀺 𑀨𑀲𑁆𑀲𑀸 𑀨𑀼𑀲𑀦𑁆𑀢𑀻”𑀢𑀺?

1610\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀢𑀬𑁄 𑀨𑀲𑁆𑀲𑀸 𑀨𑀼𑀲𑀦𑁆𑀢𑀺—  𑀲𑀼𑀜𑁆𑀜𑀢𑁄 𑀨𑀲𑁆𑀲𑁄, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀨𑀲𑁆𑀲𑁄, 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁄 𑀨𑀲𑁆𑀲𑁄”𑀢𑀺𑁇

1611\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀺𑀁𑀦𑀺𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀓𑀺𑀁𑀧𑁄𑀡𑀁 𑀓𑀺𑀁𑀧𑀩𑁆𑀪𑀸𑀭𑀦𑁆”𑀢𑀺?

1612\. “𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀯𑁂𑀓𑀧𑁄𑀡𑀁 𑀯𑀺𑀯𑁂𑀓𑀧𑀩𑁆𑀪𑀸𑀭𑀦𑁆”𑀢𑀺𑁇

1613\. “𑀓𑀢𑀺 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺?

1614\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀔𑁄 𑀇𑀫𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺𑁇

1615\. “𑀓𑀢𑀫𑀸 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀢𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺?

1616\. “𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀓𑀸𑀬𑀺𑀓𑀁 𑀯𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀲𑀸𑀢𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁—  𑀅𑀬𑀁 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀓𑀸𑀬𑀺𑀓𑀁 𑀯𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀲𑀸𑀢𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁—  𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀓𑀸𑀬𑀺𑀓𑀁 𑀯𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀯𑀸 𑀦𑁂𑀯 𑀲𑀸𑀢𑀁 𑀦𑀸𑀲𑀸𑀢𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁—  𑀅𑀬𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺𑁇

1617\. “𑀲𑀼𑀔𑀸 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀯𑁂𑀤𑀦𑀸 𑀓𑀺𑀁𑀲𑀼𑀔𑀸 𑀓𑀺𑀁𑀤𑀼𑀓𑁆𑀔𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀓𑀺𑀁𑀲𑀼𑀔𑀸 𑀓𑀺𑀁𑀤𑀼𑀓𑁆𑀔𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀓𑀺𑀁𑀲𑀼𑀔𑀸 𑀓𑀺𑀁𑀤𑀼𑀓𑁆𑀔𑀸”𑀢𑀺?

1618\. “𑀲𑀼𑀔𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀯𑁂𑀤𑀦𑀸 𑀞𑀺𑀢𑀺𑀲𑀼𑀔𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀤𑀼𑀓𑁆𑀔𑀸; 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀞𑀺𑀢𑀺𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀲𑀼𑀔𑀸; 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀜𑀸𑀡𑀲𑀼𑀔𑀸 𑀅𑀜𑁆𑀜𑀸𑀡𑀤𑀼𑀓𑁆𑀔𑀸”𑀢𑀺𑁇

1619\. “𑀲𑀼𑀔𑀸𑀬 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀯𑁂𑀤𑀦𑀸𑀬 𑀓𑀺𑀁 𑀅𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀓𑀺𑀁 𑀅𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀓𑀺𑀁 𑀅𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻”𑀢𑀺?

1620\. “𑀲𑀼𑀔𑀸𑀬 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻”𑀢𑀺𑁇

1621\. “𑀲𑀩𑁆𑀩𑀸𑀬 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀲𑀩𑁆𑀩𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀲𑀩𑁆𑀩𑀸𑀬 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻”𑀢𑀺?

1622\. “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀲𑀩𑁆𑀩𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀸𑀬 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻”𑀢𑀺𑁇

1623\. “𑀲𑀼𑀔𑀸𑀬 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀯𑁂𑀤𑀦𑀸𑀬 𑀓𑀺𑀁 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀓𑀺𑀁 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀓𑀺𑀁 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺?

1624\. “𑀲𑀼𑀔𑀸𑀬 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

1625\. “𑀲𑀩𑁆𑀩𑀸𑀬 𑀦𑀼 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀲𑀩𑁆𑀩𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀲𑀩𑁆𑀩𑀸𑀬 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺?

1639\. “𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀧𑀦𑀸𑀬𑁆𑀬𑁂, 𑀓𑀺𑀁 𑀧𑀝𑀺𑀪𑀸𑀕𑁄”𑀢𑀺? “𑀅𑀘𑁆𑀘𑀬𑀸𑀲𑀺, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀧𑀜𑁆𑀳𑀁, 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀜𑁆𑀳𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑀼𑀁𑁇 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑁄𑀕𑀥𑀜𑁆𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀭𑀸𑀬𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁𑁇 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀘 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀯𑀺𑀲𑀸𑀔, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀬𑀣𑀸 𑀘 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺 𑀢𑀣𑀸 𑀦𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇

1640\. 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑀸𑀯𑀢𑀓𑁄 𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀯𑀺𑀲𑀸𑀔𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀡𑁆𑀟𑀺𑀢𑀸, 𑀯𑀺𑀲𑀸𑀔, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻, 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸, 𑀯𑀺𑀲𑀸𑀔, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑁇 𑀫𑀜𑁆𑀘𑁂𑀧𑀺 𑀢𑁆𑀯𑀁, 𑀯𑀺𑀲𑀸𑀔, 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀁 𑀏𑀯𑀫𑁂𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁, 𑀬𑀣𑀸 𑀢𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀏𑀲𑁄 𑀘𑁂𑀯𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 𑀏𑀯𑀜𑁆𑀘 𑀦𑀁 𑀥𑀸𑀭𑁂𑀳𑀻”𑀢𑀺𑁇

1641\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀯𑀺𑀲𑀸𑀔𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀻𑀢𑀺𑁇
