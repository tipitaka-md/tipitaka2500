# 𑁫𑁇𑁮 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀽𑀍𑀅𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(491.)

1692\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀘 𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1693\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀯𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣, 𑀢𑁂 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀣, 𑀦 𑀘 𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀣, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀣, 𑀦 𑀘 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀣𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀲𑀫𑀬𑁂 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑁂𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘, 𑀫𑁂𑀢𑁆𑀢𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀫𑁂𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀲𑀫𑀬𑁂 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣, 𑀦𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀫𑁂𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘, 𑀦 𑀫𑁂𑀢𑁆𑀢𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀦 𑀫𑁂𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀣 𑀓𑀺𑀜𑁆𑀘𑀭𑀳𑀺 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀓𑀺𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀓𑀺𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣, 𑀢𑁂 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀣, 𑀦 𑀘 𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀣, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀣, 𑀦 𑀘 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁂𑀣? 𑀢𑀜𑁆𑀳𑀺 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

(492.)

1694\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀙𑀬𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀭𑀡𑀻𑀬𑀸 𑀧𑀺𑀬𑀓𑀭𑀡𑀸 𑀕𑀭𑀼𑀓𑀭𑀡𑀸 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁂𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧)

1695\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁂𑀢𑁆𑀢𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁨)

1696\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁂𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁩)

1697\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂 𑀢𑁂 𑀮𑀸𑀪𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀸 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀧𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀳𑀺 𑀮𑀸𑀪𑁂𑀳𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀪𑀢𑁆𑀢𑀪𑁄𑀕𑀻 𑀳𑁄𑀢𑀺 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀲𑀸𑀥𑀸𑀭𑀡𑀪𑁄𑀕𑀻𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁪)

1698\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑀸𑀦𑀺 𑀅𑀔𑀡𑁆𑀟𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀸𑀦𑀺 𑀅𑀲𑀩𑀮𑀸𑀦𑀺 𑀅𑀓𑀫𑁆𑀫𑀸𑀲𑀸𑀦𑀺 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑀸𑀦𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑀦𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀦𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀻𑀮𑀲𑀸𑀫𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁫)

1699\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀭𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑀺 𑀢𑀓𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀸𑀫𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁬)

1702\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀇𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀆𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀮𑀪𑀸𑀫𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀲𑀫𑀣𑀁, 𑀮𑀪𑀸𑀫𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀺𑀦𑁆’𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀇𑀫𑀁 𑀔𑁄 𑀅𑀳𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀆𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀮𑀪𑀸𑀫𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀲𑀫𑀣𑀁, 𑀮𑀪𑀸𑀫𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀺𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀜𑀸𑀡𑀁 𑀅𑀥𑀺𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀳𑀺𑁇 (𑁨)

(495.)

1703\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬𑀸𑀳𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀜𑁆𑀜𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬𑀸𑀳𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀜𑁆𑀜𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀜𑀸𑀡𑀁 𑀅𑀥𑀺𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀳𑀺𑁇 (𑁩)

(496.)

1704\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀓𑀣𑀁𑀭𑀽𑀧𑀸𑀬 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄? 𑀥𑀫𑁆𑀫𑀢𑀸 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲—  ‘𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀦𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀢𑁆𑀣𑀭𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀲𑀼 𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀤𑁂𑀲𑁂𑀢𑀺 𑀯𑀺𑀯𑀭𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀢𑀺; 𑀤𑁂𑀲𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀢𑁆𑀯𑀸 𑀆𑀬𑀢𑀺𑀁 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺’𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀳𑀭𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑀦𑁆𑀤𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑀲𑁂𑀬𑁆𑀬𑀓𑁄 𑀳𑀢𑁆𑀣𑁂𑀦 𑀯𑀸 𑀧𑀸𑀤𑁂𑀦 𑀯𑀸 𑀅𑀗𑁆𑀕𑀸𑀭𑀁 𑀅𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀝𑀺𑀲𑀁𑀳𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀢𑀸 𑀏𑀲𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲—  ‘𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀦𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀢𑁆𑀣𑀭𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀲𑀼 𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀤𑁂𑀲𑁂𑀢𑀺 𑀯𑀺𑀯𑀭𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀢𑀺; 𑀤𑁂𑀲𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀢𑁆𑀯𑀸 𑀆𑀬𑀢𑀺𑀁 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺’𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀜𑀸𑀡𑀁 𑀅𑀥𑀺𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀳𑀺𑁇 (𑁪)

1705\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀓𑀣𑀁𑀭𑀽𑀧𑀸𑀬 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄? 𑀥𑀫𑁆𑀫𑀢𑀸 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲—  ‘𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀢𑀢𑁆𑀣 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀔𑁆𑀯𑀸𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀬’𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀸𑀯𑀻 𑀢𑀭𑀼𑀡𑀯𑀘𑁆𑀙𑀸 𑀣𑀫𑁆𑀩𑀜𑁆𑀘 𑀆𑀮𑀼𑀫𑁆𑀧𑀢𑀺 𑀯𑀘𑁆𑀙𑀓𑀜𑁆𑀘 𑀅𑀧𑀘𑀺𑀦𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀢𑀸 𑀏𑀲𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲—  ‘𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀢𑀢𑁆𑀣 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀔𑁆𑀯𑀸𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀬’𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀥𑀫𑁆𑀫𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀜𑀸𑀡𑀁 𑀅𑀥𑀺𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀳𑀺𑁇 (𑁫)

(498.)

1706\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀓𑀣𑀁𑀭𑀽𑀧𑀸𑀬 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀮𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄? 𑀩𑀮𑀢𑀸 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀬𑀁 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀤𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑁂 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺𑀓𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀑𑀳𑀺𑀢𑀲𑁄𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀸𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀙𑀝𑁆𑀞𑀁 𑀜𑀸𑀡𑀁 𑀅𑀥𑀺𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀳𑀺𑁇 (𑁬)

(499.)

1707\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀓𑀣𑀁𑀭𑀽𑀧𑀸𑀬 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀮𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄? 𑀩𑀮𑀢𑀸 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀬𑀁 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀤𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑁂 𑀮𑀪𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀁, 𑀮𑀪𑀢𑀺 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀩𑀮𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀫𑀁 𑀜𑀸𑀡𑀁 𑀅𑀥𑀺𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀳𑀺𑁇 (𑁭)

(500.)

1708\. 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀼𑀲𑀫𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1709\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

1710\. 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
