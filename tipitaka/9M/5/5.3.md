# 𑁫𑁇𑁩 𑀫𑀳𑀸𑀯𑁂𑀤𑀮𑁆𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀽𑀍𑀅𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀯𑁂𑀤𑀮𑁆𑀮𑀲𑀼𑀢𑁆𑀢

(449.)

1502\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1503\. “‘𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺?

1504\. “‘𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺? ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇

1505\. “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺—

1506\. “‘𑀧𑀜𑁆𑀜𑀯𑀸 𑀧𑀜𑁆𑀜𑀯𑀸’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀜𑀯𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺?

1507\. “‘𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀯𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺? ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀯𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇

1508\. “‘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺?

1509\. “‘𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺? 𑀲𑀼𑀔𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇

1510\. “𑀬𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀜𑀸 𑀬𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀉𑀤𑀸𑀳𑀼 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸? 𑀮𑀩𑁆𑀪𑀸 𑀘 𑀧𑀦𑀺𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀬𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀜𑀸 𑀬𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀦 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀁 𑀳𑀸𑀯𑀼𑀲𑁄, 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀦 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1511\. “𑀬𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀜𑀸 𑀬𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀺𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀬𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀜𑀸 𑀬𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁𑁇 𑀇𑀤𑀁 𑀦𑁂𑀲𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

(450.)

1512\. “‘𑀯𑁂𑀤𑀦𑀸 𑀯𑁂𑀤𑀦𑀸’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑁂𑀤𑀦𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺?

1513\. “‘𑀯𑁂𑀤𑁂𑀢𑀺 𑀯𑁂𑀤𑁂𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘 𑀯𑁂𑀤𑁂𑀢𑀺? 𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 ‘𑀯𑁂𑀤𑁂𑀢𑀺 𑀯𑁂𑀤𑁂𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇

1514\. “‘𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸’𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀜𑁆𑀜𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺?

1515\. “‘𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺? 𑀦𑀻𑀮𑀓𑀫𑁆𑀧𑀺 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀻𑀢𑀓𑀫𑁆𑀧𑀺 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑁄𑀳𑀺𑀢𑀓𑀫𑁆𑀧𑀺 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀑𑀤𑀸𑀢𑀫𑁆𑀧𑀺 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇

1516\. “𑀬𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀯𑁂𑀤𑀦𑀸 𑀬𑀸 𑀘 𑀲𑀜𑁆𑀜𑀸 𑀬𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀉𑀤𑀸𑀳𑀼 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸? 𑀮𑀩𑁆𑀪𑀸 𑀘 𑀧𑀦𑀺𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀬𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀯𑁂𑀤𑀦𑀸 𑀬𑀸 𑀘 𑀲𑀜𑁆𑀜𑀸 𑀬𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀦 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀁 𑀳𑀸𑀯𑀼𑀲𑁄, 𑀯𑁂𑀤𑁂𑀢𑀺 𑀢𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀦 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(451.)

1517\. “𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀳𑀸𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀓𑀺𑀁 𑀦𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺?

1518\. “𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀬𑁆𑀬𑀁, ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀬𑁆𑀬𑀁, ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

1519\. “𑀦𑁂𑀬𑁆𑀬𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀓𑁂𑀦 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺?

1520\. “𑀦𑁂𑀬𑁆𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇

1521\. “𑀧𑀜𑁆𑀜𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀓𑀺𑀫𑀢𑁆𑀣𑀺𑀬𑀸”𑀢𑀺?

1522\. “𑀧𑀜𑁆𑀜𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀪𑀺𑀜𑁆𑀜𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀢𑁆𑀣𑀸 𑀧𑀳𑀸𑀦𑀢𑁆𑀣𑀸”𑀢𑀺𑁇

(452.)

1523\. “𑀓𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬𑀸”𑀢𑀺?

1524\. “𑀤𑁆𑀯𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬—  𑀧𑀭𑀢𑁄 𑀘 𑀖𑁄𑀲𑁄, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀘 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁆𑀯𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬𑀸”𑀢𑀺𑁇

1525\. “𑀓𑀢𑀺𑀳𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸 𑀘 𑀳𑁄𑀢𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀘, 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀘𑀸”𑀢𑀺?

1526\. “𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸 𑀘 𑀳𑁄𑀢𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀘, 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀘𑁇 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀻𑀮𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀢𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀸𑀓𑀘𑁆𑀙𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀣𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀘 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀘 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸 𑀘 𑀳𑁄𑀢𑀺 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀘, 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀘𑀸”𑀢𑀺𑁇

(453.)

1545\. “𑀉𑀲𑁆𑀫𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀻”𑀢𑀺?

1546\. “𑀉𑀲𑁆𑀫𑀸 𑀆𑀬𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀻”𑀢𑀺𑁇

1547\. “𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀏𑀯𑀁 𑀆𑀚𑀸𑀦𑀸𑀫—  ‘𑀆𑀬𑀼 𑀉𑀲𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀻’𑀢𑀺𑁇 𑀇𑀤𑀸𑀦𑁂𑀯 𑀧𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀏𑀯𑀁 𑀆𑀚𑀸𑀦𑀸𑀫—  ‘𑀉𑀲𑁆𑀫𑀸 𑀆𑀬𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀻’𑀢𑀺𑁇 𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺?

1548\. “𑀢𑁂𑀦 𑀳𑀸𑀯𑀼𑀲𑁄, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺; 𑀉𑀧𑀫𑀸𑀬𑀧𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑀲𑁆𑀲 𑀛𑀸𑀬𑀢𑁄 𑀅𑀘𑁆𑀘𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀪𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀆𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀘𑁆𑀘𑀺 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀼 𑀉𑀲𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀉𑀲𑁆𑀫𑀸 𑀆𑀬𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀻”𑀢𑀺𑁇

(457.)

1549\. “𑀢𑁂𑀯 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀢𑁂 𑀯𑁂𑀤𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀤𑀸𑀳𑀼 𑀅𑀜𑁆𑀜𑁂 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑁂 𑀯𑁂𑀤𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸”𑀢𑀺?

1550\. “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀯 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀢𑁂 𑀯𑁂𑀤𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑁂 𑀘 𑀳𑀸𑀯𑀼𑀲𑁄, 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀪𑀯𑀺𑀁𑀲𑀼 𑀢𑁂 𑀯𑁂𑀤𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀬𑀺𑀤𑀁 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀜𑁆𑀜𑁂 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑁂 𑀯𑁂𑀤𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀻”𑀢𑀺𑁇

1551\. “𑀬𑀤𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀁 𑀓𑀸𑀬𑀁 𑀓𑀢𑀺 𑀥𑀫𑁆𑀫𑀸 𑀚𑀳𑀦𑁆𑀢𑀺; 𑀅𑀣𑀸𑀬𑀁 𑀓𑀸𑀬𑁄 𑀉𑀚𑁆𑀛𑀺𑀢𑁄 𑀅𑀯𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀲𑁂𑀢𑀺, 𑀬𑀣𑀸 𑀓𑀝𑁆𑀞𑀁 𑀅𑀘𑁂𑀢𑀦𑀦𑁆”𑀢𑀺?

1552\. “𑀬𑀤𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀁 𑀓𑀸𑀬𑀁 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀚𑀳𑀦𑁆𑀢𑀺—  𑀆𑀬𑀼 𑀉𑀲𑁆𑀫𑀸 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀅𑀣𑀸𑀬𑀁 𑀓𑀸𑀬𑁄 𑀉𑀚𑁆𑀛𑀺𑀢𑁄 𑀅𑀯𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀲𑁂𑀢𑀺, 𑀬𑀣𑀸 𑀓𑀝𑁆𑀞𑀁 𑀅𑀘𑁂𑀢𑀦𑀦𑁆”𑀢𑀺𑁇

1553\. “𑀬𑁆𑀯𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀢𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄—  𑀇𑀫𑁂𑀲𑀁 𑀓𑀺𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀦𑁆”𑀢𑀺?

1554\. “𑀬𑁆𑀯𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀢𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸, 𑀆𑀬𑀼 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁄, 𑀉𑀲𑁆𑀫𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑀸𑀦𑀺𑁇 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀢𑀲𑁆𑀲𑀧𑀺 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸, 𑀆𑀬𑀼 𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁄, 𑀉𑀲𑁆𑀫𑀸 𑀅𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺𑁇 𑀬𑁆𑀯𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀢𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄—  𑀇𑀤𑀁 𑀦𑁂𑀲𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

(458.)

1555\. “𑀓𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺?

1556\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇

1557\. “𑀓𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺?

1561\. “𑀓𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀬𑀸”𑀢𑀺?

1566\. 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀩𑀚𑁆𑀛𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀓𑀭𑀼𑀡𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂…  𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀩𑀚𑁆𑀛𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇

1567\. 𑀓𑀢𑀫𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺?

1568\. 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇

1569\. 𑀓𑀢𑀫𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀲𑀼𑀜𑁆𑀜𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺?

1570\. 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀲𑀼𑀜𑁆𑀜𑀫𑀺𑀤𑀁 𑀅𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀺𑀬𑁂𑀦 𑀯𑀸’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀼𑀜𑁆𑀜𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇

1571\. 𑀓𑀢𑀫𑀸 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺?

1572\. 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀦𑀸𑀢𑁆𑀣𑀸 𑀘𑁂𑀯 𑀦𑀸𑀦𑀸𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀘𑁇

1573\. 𑀓𑀢𑀫𑁄 𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀢𑁆𑀣𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁?

1574\. 𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀫𑀸𑀡𑀓𑀭𑀡𑁄, 𑀤𑁄𑀲𑁄 𑀧𑀫𑀸𑀡𑀓𑀭𑀡𑁄, 𑀫𑁄𑀳𑁄 𑀧𑀫𑀸𑀡𑀓𑀭𑀡𑁄𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀢𑀸𑀲𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀲𑀸 𑀔𑁄 𑀧𑀦𑀸𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀸 𑀭𑀸𑀕𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀤𑁄𑀲𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑁄𑀳𑁂𑀦𑁇 𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀺𑀜𑁆𑀘𑀦𑁄, 𑀤𑁄𑀲𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑁄, 𑀫𑁄𑀳𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑁄𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀢𑀸𑀲𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀲𑀸 𑀔𑁄 𑀧𑀦𑀸𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀸 𑀭𑀸𑀕𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀤𑁄𑀲𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑁄𑀳𑁂𑀦𑁇 𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀭𑀡𑁄, 𑀤𑁄𑀲𑁄 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀭𑀡𑁄, 𑀫𑁄𑀳𑁄 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀭𑀡𑁄𑁇 𑀢𑁂 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀢𑀸𑀲𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺𑁇 𑀲𑀸 𑀔𑁄 𑀧𑀦𑀸𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀸 𑀭𑀸𑀕𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀤𑁄𑀲𑁂𑀦, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑁄𑀳𑁂𑀦𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀢𑁆𑀣𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀦𑁆”𑀢𑀺𑁇
