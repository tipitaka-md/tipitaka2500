# 𑁫𑁇𑁧𑁦 𑀫𑀸𑀭𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀭𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(506.)

1733\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀕𑁆𑀕𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀕𑀺𑀭𑁂 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑁄𑀝𑁆𑀞𑀫𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑁂 𑀓𑀼𑀘𑁆𑀙𑀺 𑀕𑀭𑀼𑀕𑀭𑁄 𑀯𑀺𑀬? 𑀫𑀸𑀲𑀸𑀘𑀺𑀢𑀁 𑀫𑀜𑁆𑀜𑁂”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀘𑀗𑁆𑀓𑀫𑀸 𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀸𑀓𑀸𑀲𑀺𑁇

1734\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑀁 𑀓𑁄𑀝𑁆𑀞𑀫𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫; 𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫𑁇 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀲𑀺, 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑀁𑁇 𑀫𑀸 𑀢𑁂 𑀅𑀳𑁄𑀲𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

1735\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀚𑀸𑀦𑀫𑁂𑀯 𑀔𑁄 𑀫𑀁 𑀅𑀬𑀁 𑀲𑀫𑀡𑁄 𑀅𑀧𑀲𑁆𑀲𑀁 𑀏𑀯𑀫𑀸𑀳—  ‘𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫; 𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫𑁇 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀲𑀺, 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑀁𑁇 𑀫𑀸 𑀢𑁂 𑀅𑀳𑁄𑀲𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸’𑀢𑀺𑁇 𑀬𑁄𑀧𑀺𑀲𑁆𑀲 𑀲𑁄 𑀲𑀢𑁆𑀣𑀸 𑀲𑁄𑀧𑀺 𑀫𑀁 𑀦𑁂𑀯 𑀔𑀺𑀧𑁆𑀧𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀓𑀼𑀢𑁄 𑀧𑀦 𑀫𑀁 𑀅𑀬𑀁 𑀲𑀸𑀯𑀓𑁄 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺?

1736\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀚𑀸𑀦𑀸𑀫𑀺, 𑀫𑀸 𑀢𑁆𑀯𑀁 𑀫𑀜𑁆𑀜𑀺𑀢𑁆𑀣𑁄—  ‘𑀦 𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀻’𑀢𑀺𑁇 𑀫𑀸𑀭𑁄 𑀢𑁆𑀯𑀫𑀲𑀺, 𑀧𑀸𑀧𑀺𑀫; 𑀢𑀼𑀬𑁆𑀳𑀜𑁆𑀳𑀺, 𑀧𑀸𑀧𑀺𑀫, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀚𑀸𑀦𑀫𑁂𑀯 𑀔𑁄 𑀫𑀁 𑀅𑀬𑀁 𑀲𑀫𑀡𑁄 𑀅𑀧𑀲𑁆𑀲𑀁 𑀏𑀯𑀫𑀸𑀳—  𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫; 𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫𑁇 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀲𑀺, 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑀁𑁇 𑀫𑀸 𑀢𑁂 𑀅𑀳𑁄𑀲𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸𑀢𑀺𑁇 𑀬𑁄𑀧𑀺𑀲𑁆𑀲 𑀲𑁄 𑀲𑀢𑁆𑀣𑀸 𑀲𑁄𑀧𑀺 𑀫𑀁 𑀦𑁂𑀯 𑀔𑀺𑀧𑁆𑀧𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀓𑀼𑀢𑁄 𑀧𑀦 𑀫𑀁 𑀅𑀬𑀁 𑀲𑀸𑀯𑀓𑁄 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺?

1737\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀚𑀸𑀦𑀫𑁂𑀯 𑀔𑁄 𑀫𑀁 𑀅𑀬𑀁 𑀲𑀫𑀡𑁄 𑀧𑀲𑁆𑀲𑀁 𑀏𑀯𑀫𑀸𑀳—  ‘𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫; 𑀦𑀺𑀓𑁆𑀔𑀫, 𑀧𑀸𑀧𑀺𑀫𑁇 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀲𑀺, 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑀁𑁇 𑀫𑀸 𑀢𑁂 𑀅𑀳𑁄𑀲𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀫𑀼𑀔𑀢𑁄 𑀉𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀕𑁆𑀕𑀍𑀏 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇

1738\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀕𑁆𑀕𑀍𑀏 𑀞𑀺𑀢𑀁; 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀢𑁆𑀣𑀸𑀧𑀺 𑀔𑁄 𑀢𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀲𑁆𑀲𑀸𑀫𑀺; 𑀫𑀸 𑀢𑁆𑀯𑀁 𑀫𑀜𑁆𑀜𑀺𑀢𑁆𑀣𑁄 ‘𑀦 𑀫𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀲𑁄 𑀢𑁆𑀯𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀘𑁆𑀘𑀕𑁆𑀕𑀍𑀏 𑀞𑀺𑀢𑁄𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀤𑀽𑀲𑀻 𑀦𑀸𑀫 𑀫𑀸𑀭𑁄 𑀅𑀳𑁄𑀲𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀍𑀈 𑀦𑀸𑀫 𑀪𑀕𑀺𑀦𑀻𑁇 𑀢𑀲𑁆𑀲𑀸 𑀢𑁆𑀯𑀁 𑀧𑀼𑀢𑁆𑀢𑁄𑁇 𑀲𑁄 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦, 𑀧𑀸𑀧𑀺𑀫, 𑀲𑀫𑀬𑁂𑀦 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀧𑀸𑀧𑀺𑀫, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀺𑀥𑀼𑀭𑀲𑀜𑁆𑀚𑀻𑀯𑀁 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀕𑁆𑀕𑀁 𑀪𑀤𑁆𑀤𑀬𑀼𑀕𑀁𑁇 𑀬𑀸𑀯𑀢𑀸 𑀔𑁄 𑀧𑀦, 𑀧𑀸𑀧𑀺𑀫, 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑀸𑁇 𑀢𑁂𑀲𑀼 𑀦 𑀘 𑀓𑁄𑀘𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀯𑀺𑀥𑀼𑀭𑁂𑀦 𑀲𑀫𑀲𑀫𑁄 𑀳𑁄𑀢𑀺 𑀬𑀤𑀺𑀤𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑁇 𑀇𑀫𑀺𑀦𑀸 𑀔𑁄 𑀏𑀯𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀯𑀺𑀥𑀼𑀭𑀲𑁆𑀲 𑀯𑀺𑀥𑀼𑀭𑁄𑀢𑁂𑀯 𑀲𑀫𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

1740\. 𑀅𑀣 𑀔𑁄, 𑀧𑀸𑀧𑀺𑀫, 𑀤𑀽𑀲𑀺𑀲𑁆𑀲 𑀫𑀸𑀭𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀫𑁂𑀲𑀁 𑀔𑁄 𑀅𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀦𑁂𑀯 𑀚𑀸𑀦𑀸𑀫𑀺 𑀆𑀕𑀢𑀺𑀁 𑀯𑀸 𑀕𑀢𑀺𑀁 𑀯𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂 𑀅𑀦𑁆𑀯𑀸𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀁—  𑀏𑀣, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂 𑀅𑀓𑁆𑀓𑁄𑀲𑀣 𑀧𑀭𑀺𑀪𑀸𑀲𑀣 𑀭𑁄𑀲𑁂𑀣 𑀯𑀺𑀳𑁂𑀲𑁂𑀣𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀭𑁄𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁, 𑀬𑀣𑀸 𑀢𑀁 𑀤𑀽𑀲𑀻 𑀫𑀸𑀭𑁄 𑀮𑀪𑁂𑀣 𑀑𑀢𑀸𑀭𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂, 𑀧𑀸𑀧𑀺𑀫, 𑀤𑀽𑀲𑀻 𑀫𑀸𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂 𑀅𑀦𑁆𑀯𑀸𑀯𑀺𑀲𑀺—  ‘𑀏𑀣, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂 𑀅𑀓𑁆𑀓𑁄𑀲𑀣 𑀧𑀭𑀺𑀪𑀸𑀲𑀣 𑀭𑁄𑀲𑁂𑀣 𑀯𑀺𑀳𑁂𑀲𑁂𑀣𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀭𑁄𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁, 𑀬𑀣𑀸 𑀢𑀁 𑀤𑀽𑀲𑀻 𑀫𑀸𑀭𑁄 𑀮𑀪𑁂𑀣 𑀑𑀢𑀸𑀭𑀦𑁆’𑀢𑀺𑁇

1741\. 𑀅𑀣 𑀔𑁄 𑀢𑁂, 𑀧𑀸𑀧𑀺𑀫, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀅𑀦𑁆𑀯𑀸𑀯𑀺𑀲𑀺𑀝𑁆𑀞𑀸 𑀤𑀽𑀲𑀺𑀦𑀸 𑀫𑀸𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀇𑀫𑁂 𑀧𑀦 𑀫𑀼𑀡𑁆𑀟𑀓𑀸 𑀲𑀫𑀡𑀓𑀸 𑀇𑀩𑁆𑀪𑀸 𑀓𑀺𑀡𑁆𑀳𑀸 𑀩𑀦𑁆𑀥𑀼𑀧𑀸𑀤𑀸𑀧𑀘𑁆𑀘𑀸 “𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸 𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸”𑀢𑀺 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀅𑀥𑁄𑀫𑀼𑀔𑀸 𑀫𑀥𑀼𑀭𑀓𑀚𑀸𑀢𑀸 𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀉𑀮𑀽𑀓𑁄 𑀭𑀼𑀓𑁆𑀔𑀲𑀸𑀔𑀸𑀬𑀁 𑀫𑀽𑀲𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀬𑀫𑀸𑀦𑁄 𑀛𑀸𑀬𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀺𑀫𑁂 𑀫𑀼𑀡𑁆𑀟𑀓𑀸 𑀲𑀫𑀡𑀓𑀸 𑀇𑀩𑁆𑀪𑀸 𑀓𑀺𑀡𑁆𑀳𑀸 𑀩𑀦𑁆𑀥𑀼𑀧𑀸𑀤𑀸𑀧𑀘𑁆𑀘𑀸 “𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸 𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸”𑀢𑀺 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀅𑀥𑁄𑀫𑀼𑀔𑀸 𑀫𑀥𑀼𑀭𑀓𑀚𑀸𑀢𑀸 𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀓𑁄𑀢𑁆𑀣𑀼 𑀦𑀤𑀻𑀢𑀻𑀭𑁂 𑀫𑀘𑁆𑀙𑁂 𑀫𑀕𑁆𑀕𑀬𑀫𑀸𑀦𑁄 𑀛𑀸𑀬𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀺𑀫𑁂 𑀫𑀼𑀡𑁆𑀟𑀓𑀸 𑀲𑀫𑀡𑀓𑀸 𑀇𑀩𑁆𑀪𑀸 𑀓𑀺𑀡𑁆𑀳𑀸 𑀩𑀦𑁆𑀥𑀼𑀧𑀸𑀤𑀸𑀧𑀘𑁆𑀘𑀸 “𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸 𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸”𑀢𑀺 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀅𑀥𑁄𑀫𑀼𑀔𑀸 𑀫𑀥𑀼𑀭𑀓𑀚𑀸𑀢𑀸 𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀺𑀍𑀆𑀭𑁄 𑀲𑀦𑁆𑀥𑀺𑀲𑀫𑀮𑀲𑀗𑁆𑀓𑀝𑀻𑀭𑁂 𑀫𑀽𑀲𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀬𑀫𑀸𑀦𑁄 𑀛𑀸𑀬𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀺𑀫𑁂 𑀫𑀼𑀡𑁆𑀟𑀓𑀸 𑀲𑀫𑀡𑀓𑀸 𑀇𑀩𑁆𑀪𑀸 𑀓𑀺𑀡𑁆𑀳𑀸 𑀩𑀦𑁆𑀥𑀼𑀧𑀸𑀤𑀸𑀧𑀘𑁆𑀘𑀸 “𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸 𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸”𑀢𑀺 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀅𑀥𑁄𑀫𑀼𑀔𑀸 𑀫𑀥𑀼𑀭𑀓𑀚𑀸𑀢𑀸 𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀤𑁆𑀭𑀪𑁄 𑀯𑀳𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀲𑀦𑁆𑀥𑀺𑀲𑀫𑀮𑀲𑀗𑁆𑀓𑀝𑀻𑀭𑁂 𑀛𑀸𑀬𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀺𑀫𑁂 𑀫𑀼𑀡𑁆𑀟𑀓𑀸 𑀲𑀫𑀡𑀓𑀸 𑀇𑀩𑁆𑀪𑀸 𑀓𑀺𑀡𑁆𑀳𑀸 𑀩𑀦𑁆𑀥𑀼𑀧𑀸𑀤𑀸𑀧𑀘𑁆𑀘𑀸 “𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸 𑀛𑀸𑀬𑀺𑀦𑁄𑀲𑁆𑀫𑀸”𑀢𑀺 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀅𑀥𑁄𑀫𑀼𑀔𑀸 𑀫𑀥𑀼𑀭𑀓𑀚𑀸𑀢𑀸 𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀻’𑀢𑀺𑁇

1751\. 𑀢𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀧𑀸𑀧𑀺𑀫, 𑀫𑀳𑀸𑀦𑀺𑀭𑀬𑀲𑁆𑀲 𑀢𑀬𑁄 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀙𑀨𑀲𑁆𑀲𑀸𑀬𑀢𑀦𑀺𑀓𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀲𑀗𑁆𑀓𑀼𑀲𑀫𑀸𑀳𑀢𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀯𑁂𑀤𑀦𑀺𑀬𑁄 𑀇𑀢𑀺𑀧𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀦𑀺𑀭𑀬𑀧𑀸𑀮𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀬𑀤𑀸 𑀔𑁄 𑀢𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑀗𑁆𑀓𑀼𑀦𑀸 𑀲𑀗𑁆𑀓𑀼 𑀳𑀤𑀬𑁂 𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀅𑀣 𑀦𑀁 𑀢𑁆𑀯𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  “𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀁 𑀫𑁂 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀫𑀸𑀦𑀲𑁆𑀲𑀸”’𑀢𑀺𑁇

1752\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑀳𑀸𑀦𑀺𑀭𑀬𑁂 𑀅𑀧𑀘𑁆𑀘𑀺𑀁𑁇 𑀤𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀲𑁆𑀲𑁂𑀯 𑀫𑀳𑀸𑀦𑀺𑀭𑀬𑀲𑁆𑀲 𑀉𑀲𑁆𑀲𑀤𑁂 𑀅𑀧𑀘𑁆𑀘𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀫𑀁 𑀦𑀸𑀫 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀫𑀸𑀦𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀏𑀯𑀭𑀽𑀧𑁄 𑀓𑀸𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑁇 𑀏𑀯𑀭𑀽𑀧𑀁 𑀲𑀻𑀲𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀘𑁆𑀙𑀲𑁆𑀲𑁇

1753\. _𑀓𑀻𑀤𑀺𑀲𑁄 𑀦𑀺𑀭𑀬𑁄 𑀆𑀲𑀺,_  
_𑀬𑀢𑁆𑀣 𑀤𑀽𑀲𑀻 𑀅𑀧𑀘𑁆𑀘𑀣;_  
_𑀯𑀺𑀥𑀼𑀭𑀁 𑀲𑀸𑀯𑀓𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_  


1754\. _𑀲𑀢𑀁 𑀆𑀲𑀺 𑀅𑀬𑁄𑀲𑀗𑁆𑀓𑀽,_  
_𑀲𑀩𑁆𑀩𑁂 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀯𑁂𑀤𑀦𑀸;_  
_𑀈𑀤𑀺𑀲𑁄 𑀦𑀺𑀭𑀬𑁄 𑀆𑀲𑀺,_  
_𑀬𑀢𑁆𑀣 𑀤𑀽𑀲𑀻 𑀅𑀧𑀘𑁆𑀘𑀣;_  
_𑀯𑀺𑀥𑀼𑀭𑀁 𑀲𑀸𑀯𑀓𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_  


1755\. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


1756\. _𑀫𑀚𑁆𑀛𑁂 𑀲𑀭𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀫𑀸𑀦𑀸 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑀸𑀬𑀺𑀦𑁄;_  
_𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀯𑀡𑁆𑀡𑀸 𑀭𑀼𑀘𑀺𑀭𑀸,_  
_𑀅𑀘𑁆𑀘𑀺𑀫𑀦𑁆𑀢𑁄 𑀧𑀪𑀲𑁆𑀲𑀭𑀸;_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀢𑀢𑁆𑀣 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀣𑀼 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀬𑁄𑁇_  


1757\. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


1758\. _𑀬𑁄 𑀯𑁂 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑁂𑀓𑁆𑀔𑀢𑁄;_  
_𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑀁,_  
_𑀧𑀸𑀤𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑁂𑀦 𑀓𑀫𑁆𑀧𑀬𑀺𑁇_  


1759\. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


1760\. _𑀬𑁄 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀁 𑀧𑀸𑀲𑀸𑀤𑀁,_  
_𑀧𑀸𑀤𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑁂𑀦 𑀓𑀫𑁆𑀧𑀬𑀺;_  
_𑀇𑀤𑁆𑀥𑀺𑀩𑀮𑁂𑀦𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑁄,_  
_𑀲𑀁𑀯𑁂𑀚𑁂𑀲𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑁇_  


1761\. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


1762\. _𑀬𑁄 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑁂,_  
_𑀲𑀓𑁆𑀓𑀁 𑀲𑁄 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺;_  
_𑀅𑀧𑀺 𑀯𑀸𑀲𑀯 𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀢𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀬𑀣𑀸𑀢𑀣𑀁𑁇_  


1763\. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


1764\. _𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀸𑀪𑀺𑀢𑁄 𑀲𑀪𑀁;_  
_𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀢𑁆𑀬𑀸𑀯𑀼𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀺,_  
_𑀬𑀸 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀼𑀭𑁂 𑀅𑀳𑀼;_  
_𑀧𑀲𑁆𑀲𑀲𑀺 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀧𑀪𑀲𑁆𑀲𑀭𑀁𑁇_  


1765\. _𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀁 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀦 𑀫𑁂 𑀫𑀸𑀭𑀺𑀲 𑀲𑀸 𑀤𑀺𑀝𑁆𑀞𑀺,_  
_𑀬𑀸 𑀫𑁂 𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀼𑀭𑁂 𑀅𑀳𑀼𑁇_  


1766\. _𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀧𑀪𑀲𑁆𑀲𑀭𑀁;_  
_𑀲𑁄𑀳𑀁 𑀅𑀚𑁆𑀚 𑀓𑀣𑀁 𑀯𑀚𑁆𑀚𑀁,_  
_𑀅𑀳𑀁 𑀦𑀺𑀘𑁆𑀘𑁄𑀫𑁆𑀳𑀺 𑀲𑀲𑁆𑀲𑀢𑁄𑁇_  


1767\. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


1768\. _𑀬𑁄 𑀫𑀳𑀸𑀫𑁂𑀭𑀼𑀦𑁄 𑀓𑀽𑀝𑀁,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁂𑀦 𑀅𑀨𑀲𑁆𑀲𑀬𑀺;_  
_𑀯𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁,_  
_𑀬𑁂 𑀘 𑀪𑀽𑀫𑀺𑀲𑀬𑀸 𑀦𑀭𑀸𑁇_  


1769\. _𑀬𑁄 𑀏𑀢𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀓𑀡𑁆𑀳 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


1770\. _𑀦 𑀯𑁂 𑀅𑀕𑁆𑀕𑀺 𑀘𑁂𑀢𑀬𑀢𑀺,_  
_‘𑀅𑀳𑀁 𑀩𑀸𑀮𑀁 𑀟𑀳𑀸𑀫𑀻’𑀢𑀺;_  
_𑀩𑀸𑀮𑁄 𑀘 𑀚𑀮𑀺𑀢𑀁 𑀅𑀕𑁆𑀕𑀺𑀁,_  
_𑀆𑀲𑀚𑁆𑀚 𑀦𑀁 𑀲 𑀟𑀬𑁆𑀳𑀢𑀺𑁇_  


1771\. _𑀏𑀯𑀫𑁂𑀯 𑀢𑀼𑀯𑀁 𑀫𑀸𑀭,_  
_𑀆𑀲𑀚𑁆𑀚 𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑀁;_  
_𑀲𑀬𑀁 𑀟𑀳𑀺𑀲𑁆𑀲𑀲𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀩𑀸𑀮𑁄 𑀅𑀕𑁆𑀕𑀺𑀁𑀯 𑀲𑀫𑁆𑀨𑀼𑀲𑀁𑁇_  


1772\. _𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀻 𑀫𑀸𑀭𑁄,_  
_𑀆𑀲𑀚𑁆𑀚 𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑀁;_  
_𑀓𑀺𑀁 𑀦𑀼 𑀫𑀜𑁆𑀜𑀲𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦 𑀫𑁂 𑀧𑀸𑀧𑀁 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀺𑁇_  


1773\. _𑀓𑀭𑁄𑀢𑁄 𑀘𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀁,_  
_𑀘𑀺𑀭𑀭𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑁆𑀢𑀓;_  
_𑀫𑀸𑀭 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤 𑀩𑀼𑀤𑁆𑀥𑀫𑁆𑀳𑀸,_  
_𑀆𑀲𑀁 𑀫𑀸𑀓𑀸𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑁇_  


1774\. _𑀇𑀢𑀺 𑀫𑀸𑀭𑀁 𑀅𑀢𑀚𑁆𑀚𑁂𑀲𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂;_  
_𑀢𑀢𑁄 𑀲𑁄 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀬𑀓𑁆𑀔𑁄,_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀣𑀸”𑀢𑀺𑁇_  


---

1775\. 𑀫𑀸𑀭𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀲𑀫𑀁𑁇



---

1776\. 𑀘𑀽𑀍𑀅𑀬𑀫𑀓𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

1777\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1778\. _𑀲𑀸𑀮𑁂𑀬𑁆𑀬 𑀯𑁂𑀭𑀜𑁆𑀚𑀤𑀼𑀯𑁂 𑀘 𑀢𑀼𑀝𑁆𑀞𑀺,_  
_𑀘𑀽𑀍𑀅𑀫𑀳𑀸𑀥𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀜𑁆𑀘;_  
_𑀯𑀻𑀫𑀁𑀲𑀓𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄,_  
_𑀤𑀽𑀲𑀻 𑀘 𑀫𑀸𑀭𑁄 𑀤𑀲𑀫𑁄 𑀘 𑀯𑀕𑁆𑀕𑁄𑁇_  


---

1779\. 𑀇𑀤𑀁 𑀯𑀕𑁆𑀕𑀸𑀦𑀫𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1780\. _𑀫𑀽𑀮𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀘𑁂𑀯,_  
_𑀲𑀻𑀳𑀦𑀸𑀤𑁄 𑀘 𑀉𑀢𑁆𑀢𑀫𑁄;_  
_𑀓𑀓𑀘𑁄 𑀘𑁂𑀯 𑀕𑁄𑀲𑀺𑀗𑁆𑀕𑁄,_  
_𑀲𑀸𑀮𑁂𑀬𑁆𑀬𑁄 𑀘 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘𑁇_  


---

1781\. 𑀫𑀽𑀮𑀧𑀡𑁆𑀡𑀸𑀲𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
