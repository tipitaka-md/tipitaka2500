# 𑁫𑁇𑁨 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀲𑀼𑀢𑁆𑀢

(444.)

1479\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸—  “𑀲𑀫𑀡𑁄 𑀔𑀮𑀼, 𑀪𑁄, 𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀲𑀤𑁂𑀯𑀓𑀁 𑀲𑀫𑀸𑀭𑀓𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀁 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀧𑀚𑀁 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁; 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1480\. 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼; 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼; 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼; 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀲𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼; 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀓𑁄 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺?

1481\. “𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀯𑀺𑀲𑀫𑀘𑀭𑀺𑀬𑀸𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀲𑀫𑀘𑀭𑀺𑀬𑀸𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1482\. “𑀦 𑀔𑁄 𑀫𑀬𑀁 𑀇𑀫𑀲𑁆𑀲 𑀪𑁄𑀢𑁄 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲, 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲, 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑁇 𑀲𑀸𑀥𑀼 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼 𑀬𑀣𑀸 𑀫𑀬𑀁 𑀇𑀫𑀲𑁆𑀲 𑀪𑁄𑀢𑁄 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲, 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲, 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀲𑀼𑀡𑀸𑀣 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1483\. “𑀢𑀺𑀯𑀺𑀥𑀁 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀓𑀸𑀬𑁂𑀦 𑀅𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀯𑀺𑀲𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑀁 𑀯𑀸𑀘𑀸𑀬 𑀅𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀯𑀺𑀲𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀺𑀯𑀺𑀥𑀁 𑀫𑀦𑀲𑀸 𑀅𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀯𑀺𑀲𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇

1491\. 𑀓𑀣𑀜𑁆𑀘, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀢𑀺𑀯𑀺𑀥𑀁 𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀳𑁄𑀢𑀺𑁇 𑀬𑀁 𑀢𑀁 𑀧𑀭𑀲𑁆𑀲 𑀧𑀭𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑀁 𑀢𑀁 𑀦𑀸𑀪𑀺𑀚𑁆𑀛𑀸𑀢𑀸 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀬𑀁 𑀧𑀭𑀲𑁆𑀲, 𑀢𑀁 𑀫𑀫𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀤𑀼𑀝𑁆𑀞𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  ‘𑀇𑀫𑁂 𑀲𑀢𑁆𑀢𑀸 𑀅𑀯𑁂𑀭𑀸 𑀅𑀩𑁆𑀬𑀸𑀩𑀚𑁆𑀛𑀸 𑀅𑀦𑀻𑀖𑀸 𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀦𑁆𑀢𑀽’𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀧𑀭𑀻𑀢𑀤𑀲𑁆𑀲𑀦𑁄—  ‘𑀅𑀢𑁆𑀣𑀺 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀬𑀺𑀝𑁆𑀞𑀁…  𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀢𑀺𑀯𑀺𑀥𑀁 𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇

1492\. 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀲𑀫𑀘𑀭𑀺𑀬𑀸𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1493\. 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺; 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻𑁇

1494\. 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁…𑀧𑁂…  𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺; 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻𑁇

1495\. 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺; 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻𑁇

1496\. 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀬𑀸𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀢𑀼𑀲𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺; 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻𑁇

1497\. 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀆𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺; 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀆𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻𑁇

1498\. 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀆𑀪𑀲𑁆𑀲𑀭𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑀼𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀲𑀼𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀲𑀼𑀪𑀓𑀺𑀡𑁆𑀳𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀯𑁂𑀳𑀧𑁆𑀨𑀮𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀅𑀯𑀺𑀳𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀅𑀢𑀧𑁆𑀧𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀲𑀼𑀤𑀲𑁆𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀲𑀼𑀤𑀲𑁆𑀲𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁…  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺; 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻𑁇

1499\. 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺; 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, ‘𑀬𑀁 𑀲𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀫𑀘𑀸𑀭𑀻’”𑀢𑀺𑁇

1500\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑁄𑀢𑀸 𑀕𑁄𑀢𑀫𑁂𑀦 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀢𑁂 𑀫𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑁂 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑁂 𑀲𑀭𑀡𑀁 𑀕𑀢𑁂”𑀢𑀺𑁇
