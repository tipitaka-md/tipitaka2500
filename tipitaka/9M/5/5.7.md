# 𑁫𑁇𑁭 𑀯𑀻𑀫𑀁𑀲𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀯𑀻𑀫𑀁𑀲𑀓𑀲𑀼𑀢𑁆𑀢

(487.)

1677. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀻𑀫𑀁𑀲𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀭𑀲𑁆𑀲 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁂 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀦𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸 ‘𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀸 𑀦𑁄 𑀯𑀸’ 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸, 𑀪𑀕𑀯𑀁𑀦𑁂𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀁𑀧𑀝𑀺𑀲𑀭𑀡𑀸; 𑀲𑀸𑀥𑀼 𑀯𑀢, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀏𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄; 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

(488.)

1678. “𑀯𑀻𑀫𑀁𑀲𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀭𑀲𑁆𑀲 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦 𑀤𑁆𑀯𑀻𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼—  ‘𑀬𑁂 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀯𑀸 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦𑁄 𑀯𑀸’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑁂 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 (𑁧)

1679. 𑀬𑀢𑁄 𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑁂 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀢𑀢𑁄 𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀢𑀺—  ‘𑀬𑁂 𑀯𑀻𑀢𑀺𑀫𑀺𑀲𑁆𑀲𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀯𑀸 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦𑁄 𑀯𑀸’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑁂 𑀯𑀻𑀢𑀺𑀫𑀺𑀲𑁆𑀲𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 (𑁨)

1680. 𑀬𑀢𑁄 𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑁂 𑀯𑀻𑀢𑀺𑀫𑀺𑀲𑁆𑀲𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀢𑀢𑁄 𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀢𑀺—  ‘𑀬𑁂 𑀯𑁄𑀤𑀸𑀢𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀯𑀸 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦𑁄 𑀯𑀸’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑁂 𑀯𑁄𑀤𑀸𑀢𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 (𑁩)

1681. 𑀬𑀢𑁄 𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑁂 𑀯𑁄𑀤𑀸𑀢𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀢𑀢𑁄 𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀢𑀺—  ‘𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀇𑀢𑁆𑀢𑀭𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁, 𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀢𑁆𑀢𑀭𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 (𑁪)

1682. 𑀬𑀢𑁄 𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁, 𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀢𑁆𑀢𑀭𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄’𑀢𑀺, 𑀢𑀢𑁄 𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀢𑀺—  ‘𑀜𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀤𑀻𑀦𑀯𑀸’𑀢𑀺? 𑀦 𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀬𑀸𑀯 𑀦 𑀜𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀬𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀜𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀬𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀅𑀣𑀲𑁆𑀲 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀜𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀦𑀸𑀲𑁆𑀲 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 (𑁫)

1683. 𑀬𑀢𑁄 𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀜𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀦𑀸𑀲𑁆𑀲 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀢𑀢𑁄 𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀢𑀺—  ‘𑀅𑀪𑀬𑀽𑀧𑀭𑀢𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸, 𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀬𑀽𑀧𑀭𑀢𑁄; 𑀯𑀻𑀢𑀭𑀸𑀕𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑁂 𑀦 𑀲𑁂𑀯𑀢𑀺 𑀔𑀬𑀸 𑀭𑀸𑀕𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀢𑀫𑁂𑀦𑀁 𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀅𑀪𑀬𑀽𑀧𑀭𑀢𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸, 𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀬𑀽𑀧𑀭𑀢𑁄; 𑀯𑀻𑀢𑀭𑀸𑀕𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑁂 𑀦 𑀲𑁂𑀯𑀢𑀺 𑀔𑀬𑀸 𑀭𑀸𑀕𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 (𑁬)

1684. 𑀢𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀭𑁂 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑁂 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀓𑀸𑀭𑀸, 𑀓𑁂 𑀅𑀦𑁆𑀯𑀬𑀸, 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  𑀅𑀪𑀬𑀽𑀧𑀭𑀢𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸, 𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀬𑀽𑀧𑀭𑀢𑁄; 𑀯𑀻𑀢𑀭𑀸𑀕𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑁂 𑀦 𑀲𑁂𑀯𑀢𑀺 𑀔𑀬𑀸 𑀭𑀸𑀕𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀁𑀖𑁂 𑀯𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀏𑀓𑁄 𑀯𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄, 𑀬𑁂 𑀘 𑀢𑀢𑁆𑀣 𑀲𑀼𑀕𑀢𑀸 𑀬𑁂 𑀘 𑀢𑀢𑁆𑀣 𑀤𑀼𑀕𑁆𑀕𑀢𑀸, 𑀬𑁂 𑀘 𑀢𑀢𑁆𑀣 𑀕𑀡𑀫𑀦𑀼𑀲𑀸𑀲𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀘 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀫𑀺𑀲𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀘 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀫𑀺𑀲𑁂𑀦 𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑀸, 𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀢𑀁 𑀢𑁂𑀦 𑀅𑀯𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  𑀅𑀪𑀬𑀽𑀧𑀭𑀢𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀦𑀸𑀳𑀫𑀲𑁆𑀫𑀺 𑀪𑀬𑀽𑀧𑀭𑀢𑁄, 𑀯𑀻𑀢𑀭𑀸𑀕𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑁂 𑀦 𑀲𑁂𑀯𑀸𑀫𑀺 𑀔𑀬𑀸 𑀭𑀸𑀕𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

(489.)

1685. 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄𑀯 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄—  ‘𑀬𑁂 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀯𑀸 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦𑁄 𑀯𑀸’𑀢𑀺? 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀬𑁂 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 (𑁧)

1686. ‘𑀬𑁂 𑀯𑀻𑀢𑀺𑀫𑀺𑀲𑁆𑀲𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀯𑀸 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦𑁄 𑀯𑀸’𑀢𑀺? 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀬𑁂 𑀯𑀻𑀢𑀺𑀫𑀺𑀲𑁆𑀲𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 (𑁨)

1687. ‘𑀬𑁂 𑀯𑁄𑀤𑀸𑀢𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀯𑀸 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦𑁄 𑀯𑀸’𑀢𑀺? 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀬𑁂 𑀯𑁄𑀤𑀸𑀢𑀸 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲; 𑀏𑀢𑀁 𑀧𑀣𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀢𑀁 𑀕𑁄𑀘𑀭𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀦 𑀢𑀫𑁆𑀫𑀬𑁄’𑀢𑀺𑁇 (𑁩)

1688. 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀅𑀭𑀳𑀢𑀺 𑀲𑀸𑀯𑀓𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀧𑀡𑀻𑀢𑀧𑀡𑀻𑀢𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀁𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀧𑀡𑀻𑀢𑀧𑀡𑀻𑀢𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀦𑀺𑀝𑁆𑀞𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀢𑁆𑀣𑀭𑀺 𑀧𑀲𑀻𑀤𑀢𑀺—  ‘𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸, 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄, 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁄’𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀭𑁂 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑁂 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀓𑀸𑀭𑀸, 𑀓𑁂 𑀅𑀦𑁆𑀯𑀬𑀸, 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺—  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸, 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄, 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁄’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀧𑀡𑀻𑀢𑀧𑀡𑀻𑀢𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀁𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀧𑀡𑀻𑀢𑀧𑀡𑀻𑀢𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸𑀳𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀦𑀺𑀝𑁆𑀞𑀫𑀕𑀫𑀁, 𑀲𑀢𑁆𑀣𑀭𑀺 𑀧𑀲𑀻𑀤𑀺𑀁—  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸, 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀥𑀫𑁆𑀫𑁄, 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁄’𑀢𑀺𑁇

(490.)

1689. 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀇𑀫𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀇𑀫𑁂𑀳𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑁂 𑀲𑀤𑁆𑀥𑀸 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀢𑀺 𑀫𑀽𑀮𑀚𑀸𑀢𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀓𑀸𑀭𑀯𑀢𑀻 𑀲𑀤𑁆𑀥𑀸 𑀤𑀲𑁆𑀲𑀦𑀫𑀽𑀮𑀺𑀓𑀸, 𑀤𑀍𑀳𑀸; 𑀅𑀲𑀁𑀳𑀸𑀭𑀺𑀬𑀸 𑀲𑀫𑀡𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀦 𑀯𑀸 𑀤𑁂𑀯𑁂𑀦 𑀯𑀸 𑀫𑀸𑀭𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀯𑀸 𑀓𑁂𑀦𑀘𑀺 𑀯𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂 𑀥𑀫𑁆𑀫𑀲𑀫𑀦𑁆𑀦𑁂𑀲𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀢𑀸𑀲𑀼𑀲𑀫𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1690. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

1691. 𑀯𑀻𑀫𑀁𑀲𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
