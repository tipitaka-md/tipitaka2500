# 𑁨𑁇𑁭 𑀯𑀦𑀧𑀢𑁆𑀣𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀻𑀳𑀦𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀯𑀦𑀧𑀢𑁆𑀣𑀲𑀼𑀢𑁆𑀢

(190.)

689\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀦𑀧𑀢𑁆𑀣𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

(191.)

690\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀸𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸—  𑀢𑁂 𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀇𑀫𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀸𑀫𑀺, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀇𑀫𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀸𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀫𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸—  𑀢𑁂 𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀢𑁆𑀢𑀺𑀪𑀸𑀕𑀁 𑀯𑀸 𑀤𑀺𑀯𑀲𑀪𑀸𑀕𑀁 𑀯𑀸 𑀢𑀫𑁆𑀳𑀸 𑀯𑀦𑀧𑀢𑁆𑀣𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇

695\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀸𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸, 𑀢𑁂 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀸𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀫𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸—  𑀢𑁂 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀘𑀻𑀯𑀭𑀳𑁂𑀢𑀼 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀳𑁂𑀢𑀼…𑀧𑁂…  𑀦 𑀲𑁂𑀦𑀸𑀲𑀦𑀳𑁂𑀢𑀼…𑀧𑁂…  𑀦 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀳𑁂𑀢𑀼 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀫𑁂 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀦𑀸𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀗𑁆𑀔𑀸𑀧𑀺 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀆𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦𑀸𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑁄𑁇

696\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸—  𑀢𑁂 𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀫𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸—  𑀢𑁂 𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀘𑀻𑀯𑀭𑀳𑁂𑀢𑀼 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀳𑁂𑀢𑀼…𑀧𑁂…  𑀦 𑀲𑁂𑀦𑀸𑀲𑀦𑀳𑁂𑀢𑀼…𑀧𑁂…  𑀦 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀳𑁂𑀢𑀼 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀫𑁂 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀗𑁆𑀔𑀸𑀧𑀺 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(198.)

697\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸—  𑀢𑁂 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀘 𑀆𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀫𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀇𑀫𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀦𑁂𑀢𑀩𑁆𑀩𑀸—  𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸—  𑀢𑁂 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀫𑁆𑀧𑀺 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀧𑀺 𑀧𑀦𑀼𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀦𑀧𑀻”𑀢𑀺𑁇

698\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

699\. 𑀯𑀦𑀧𑀢𑁆𑀣𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
