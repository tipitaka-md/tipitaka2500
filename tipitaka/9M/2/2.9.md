# 𑁨𑁇𑁯 𑀤𑁆𑀯𑁂𑀥𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀻𑀳𑀦𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀤𑁆𑀯𑁂𑀥𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(206.)

717\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

718\. “𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀤𑁆𑀯𑀺𑀥𑀸 𑀓𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀺𑀥𑀸 𑀓𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀬𑁄 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀬𑁄 𑀘 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄—  𑀇𑀫𑀁 𑀏𑀓𑀁 𑀪𑀸𑀕𑀫𑀓𑀸𑀲𑀺𑀁; 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀬𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀬𑁄 𑀘 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄—  𑀇𑀫𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀪𑀸𑀕𑀫𑀓𑀸𑀲𑀺𑀁𑁇

(207.)

719\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀘 𑀔𑁄 𑀅𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑁄 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄’𑁇 ‘𑀅𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; ‘𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; ‘𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; ‘𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑁄 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀧𑀚𑀳𑀫𑁂𑀯 𑀯𑀺𑀦𑁄𑀤𑀫𑁂𑀯 𑀩𑁆𑀬𑀦𑁆𑀢𑀫𑁂𑀯 𑀦𑀁 𑀅𑀓𑀸𑀲𑀺𑀁𑁇

(208.)

720\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀘 𑀔𑁄 𑀅𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬𑀧𑀺 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑁄 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄’𑁇 ‘𑀅𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; ‘𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; ‘𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; ‘𑀧𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀺𑀓𑁄 𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀅𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄’𑀢𑀺𑀧𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀧𑀚𑀳𑀫𑁂𑀯 𑀯𑀺𑀦𑁄𑀤𑀫𑁂𑀯 𑀩𑁆𑀬𑀦𑁆𑀢𑀫𑁂𑀯 𑀦𑀁 𑀅𑀓𑀸𑀲𑀺𑀁𑁇

721\. 𑀬𑀜𑁆𑀜𑀤𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀮𑀫𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺, 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀦𑀢𑀺 𑀳𑁄𑀢𑀺 𑀘𑁂𑀢𑀲𑁄𑁇 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀮𑀫𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺, 𑀧𑀳𑀸𑀲𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀁, 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀩𑀳𑀼𑀮𑀫𑀓𑀸𑀲𑀺, 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀢𑀺𑁇 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀮𑀫𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺, 𑀧𑀳𑀸𑀲𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀁, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀩𑀳𑀼𑀮𑀫𑀓𑀸𑀲𑀺, 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀲𑀭𑀤𑀲𑀫𑀬𑁂 𑀓𑀺𑀝𑁆𑀞𑀲𑀫𑁆𑀩𑀸𑀥𑁂 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀕𑀸𑀯𑁄 𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑁇 𑀲𑁄 𑀢𑀸 𑀕𑀸𑀯𑁄 𑀢𑀢𑁄 𑀢𑀢𑁄 𑀤𑀡𑁆𑀟𑁂𑀦 𑀆𑀓𑁄𑀝𑁂𑀬𑁆𑀬 𑀧𑀝𑀺𑀓𑁄𑀝𑁂𑀬𑁆𑀬 𑀲𑀦𑁆𑀦𑀺𑀭𑀼𑀦𑁆𑀥𑁂𑀬𑁆𑀬 𑀲𑀦𑁆𑀦𑀺𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀧𑀲𑁆𑀲𑀢𑀺 𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀯𑀥𑀁 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑀁 𑀯𑀸 𑀚𑀸𑀦𑀺𑀁 𑀯𑀸 𑀕𑀭𑀳𑀁 𑀯𑀸𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀯𑁄𑀤𑀸𑀦𑀧𑀓𑁆𑀔𑀁𑁇

722\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀘 𑀔𑁄 𑀦𑁂𑀯𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀦 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀦 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀓𑁄 𑀅𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄’𑁇 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘𑁂𑀧𑀺 𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀬𑁆𑀬𑀁, 𑀦𑁂𑀯 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀪𑀬𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀺𑀯𑀲𑀜𑁆𑀘𑁂𑀧𑀺 𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀬𑁆𑀬𑀁, 𑀦𑁂𑀯 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀪𑀬𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀜𑁆𑀘𑁂𑀧𑀺 𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀬𑁆𑀬𑀁, 𑀦𑁂𑀯 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀪𑀬𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀫𑁂 𑀅𑀢𑀺𑀘𑀺𑀭𑀁 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑀬𑀢𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑀬𑀢𑁄 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑁇 𑀓𑀸𑀬𑁂 𑀓𑀺𑀮𑀦𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀊𑀳𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁇 𑀊𑀳𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀆𑀭𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀫𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀡𑁆𑀞𑀧𑁂𑀫𑀺 𑀲𑀦𑁆𑀦𑀺𑀲𑀸𑀤𑁂𑀫𑀺 𑀏𑀓𑁄𑀤𑀺𑀁 𑀓𑀭𑁄𑀫𑀺 𑀲𑀫𑀸𑀤𑀳𑀸𑀫𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? ‘𑀫𑀸 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀊𑀳𑀜𑁆𑀜𑀻’𑀢𑀺𑁇

723\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀲𑁄 𑀘 𑀔𑁄 𑀦𑁂𑀯𑀢𑁆𑀢𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀦 𑀧𑀭𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀦 𑀉𑀪𑀬𑀩𑁆𑀬𑀸𑀩𑀸𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀓𑁄 𑀅𑀯𑀺𑀖𑀸𑀢𑀧𑀓𑁆𑀔𑀺𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄’𑁇 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘𑁂𑀧𑀺 𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀬𑁆𑀬𑀁, 𑀦𑁂𑀯 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀪𑀬𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀺𑀯𑀲𑀜𑁆𑀘𑁂𑀧𑀺 𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀬𑁆𑀬𑀁, 𑀦𑁂𑀯 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀪𑀬𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀜𑁆𑀘𑁂𑀧𑀺 𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀬𑁆𑀬𑀁, 𑀦𑁂𑀯 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀪𑀬𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀫𑁂 𑀅𑀢𑀺𑀘𑀺𑀭𑀁 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑀬𑀢𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑀬𑀢𑁄 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑁇 𑀓𑀸𑀬𑁂 𑀓𑀺𑀮𑀦𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀊𑀳𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁇 𑀊𑀳𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀆𑀭𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀫𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀡𑁆𑀞𑀧𑁂𑀫𑀺, 𑀲𑀦𑁆𑀦𑀺𑀲𑀸𑀤𑁂𑀫𑀺, 𑀏𑀓𑁄𑀤𑀺𑀁 𑀓𑀭𑁄𑀫𑀺 𑀲𑀫𑀸𑀤𑀳𑀸𑀫𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? ‘𑀫𑀸 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀊𑀳𑀜𑁆𑀜𑀻’𑀢𑀺𑁇

725\. 𑀆𑀭𑀤𑁆𑀥𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀲𑀮𑁆𑀮𑀻𑀦𑀁, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀢𑀺 𑀅𑀲𑀫𑁆𑀫𑀼𑀝𑁆𑀞𑀸, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀓𑀸𑀬𑁄 𑀅𑀲𑀸𑀭𑀤𑁆𑀥𑁄, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑀕𑁆𑀕𑀁𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀲𑀺𑀁, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺, 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇

727\. 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁂 𑀅𑀦𑀗𑁆𑀕𑀡𑁂 𑀯𑀺𑀕𑀢𑀽𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀫𑀼𑀤𑀼𑀪𑀽𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀞𑀺𑀢𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑀜𑀸𑀡𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀲𑀺𑀁𑁇 𑀲𑁄 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂…𑀧𑁂…  𑀇𑀫𑁂 𑀯𑀢 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸…𑀧𑁂…  𑀇𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂, 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀤𑀼𑀢𑀺𑀬𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀥𑀺𑀕𑀢𑀸; 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀯𑀺𑀳𑀢𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸; 𑀢𑀫𑁄 𑀯𑀺𑀳𑀢𑁄 𑀆𑀮𑁄𑀓𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄; 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇

(214.)

728\. 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁂 𑀅𑀦𑀗𑁆𑀕𑀡𑁂 𑀯𑀺𑀕𑀢𑀽𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀫𑀼𑀤𑀼𑀪𑀽𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀞𑀺𑀢𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀜𑀸𑀡𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀲𑀺𑀁𑁇 𑀲𑁄 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 ‘𑀇𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀯𑀁 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀓𑀸𑀫𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣, 𑀪𑀯𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀅𑀳𑁄𑀲𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀢𑀢𑀺𑀬𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀥𑀺𑀕𑀢𑀸; 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀯𑀺𑀳𑀢𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸; 𑀢𑀫𑁄 𑀯𑀺𑀳𑀢𑁄 𑀆𑀮𑁄𑀓𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄; 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇

(215.)

729\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀜𑁆𑀜𑁂 𑀧𑀯𑀦𑁂 𑀫𑀳𑀦𑁆𑀢𑀁 𑀦𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀮𑀁𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁄 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀦𑀢𑁆𑀣𑀓𑀸𑀫𑁄 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑁄 𑀅𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑁄𑁇 𑀲𑁄 𑀬𑁆𑀯𑀸𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑁄 𑀔𑁂𑀫𑁄 𑀲𑁄𑀯𑀢𑁆𑀣𑀺𑀓𑁄 𑀧𑀻𑀢𑀺𑀕𑀫𑀦𑀻𑀬𑁄 𑀢𑀁 𑀫𑀕𑁆𑀕𑀁 𑀧𑀺𑀤𑀳𑁂𑀬𑁆𑀬, 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀁, 𑀑𑀤𑀳𑁂𑀬𑁆𑀬 𑀑𑀓𑀘𑀭𑀁, 𑀞𑀧𑁂𑀬𑁆𑀬 𑀑𑀓𑀘𑀸𑀭𑀺𑀓𑀁𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀢𑁄 𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑀲𑁆𑀲 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑁄 𑀳𑀺𑀢𑀓𑀸𑀫𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑁄𑁇 𑀲𑁄 𑀬𑁆𑀯𑀸𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑁄 𑀔𑁂𑀫𑁄 𑀲𑁄𑀯𑀢𑁆𑀣𑀺𑀓𑁄 𑀧𑀻𑀢𑀺𑀕𑀫𑀦𑀻𑀬𑁄 𑀢𑀁 𑀫𑀕𑁆𑀕𑀁 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀧𑀺𑀤𑀳𑁂𑀬𑁆𑀬 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀁, 𑀊𑀳𑀦𑁂𑀬𑁆𑀬 𑀑𑀓𑀘𑀭𑀁, 𑀦𑀸𑀲𑁂𑀬𑁆𑀬 𑀑𑀓𑀘𑀸𑀭𑀺𑀓𑀁𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇

730\. 𑀉𑀧𑀫𑀸 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀸 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀸𑀬𑁇 𑀅𑀬𑀁 𑀘𑁂𑀯𑁂𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑁄—  𑀫𑀳𑀦𑁆𑀢𑀁 𑀦𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀮𑀦𑁆𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀫𑀳𑀸𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀦𑀢𑁆𑀣𑀓𑀸𑀫𑁄 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑁄 𑀅𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀭𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀲𑁆𑀲𑁂𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀅𑀥𑀺𑀯𑀘𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸𑀬 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲𑁇 𑀑𑀓𑀘𑀭𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀑𑀓𑀘𑀸𑀭𑀺𑀓𑀸𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑁄 𑀳𑀺𑀢𑀓𑀸𑀫𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀔𑁂𑀫𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑁄𑀯𑀢𑁆𑀣𑀺𑀓𑁄 𑀧𑀻𑀢𑀺𑀕𑀫𑀦𑀻𑀬𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀅𑀥𑀺𑀯𑀘𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲𑁇

731\. 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀯𑀝𑁄 𑀫𑀬𑀸 𑀔𑁂𑀫𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑁄𑀯𑀢𑁆𑀣𑀺𑀓𑁄 𑀧𑀻𑀢𑀺𑀕𑀫𑀦𑀻𑀬𑁄, 𑀧𑀺𑀳𑀺𑀢𑁄 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄, 𑀊𑀳𑀢𑁄 𑀑𑀓𑀘𑀭𑁄, 𑀦𑀸𑀲𑀺𑀢𑀸 𑀑𑀓𑀘𑀸𑀭𑀺𑀓𑀸𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀣𑀸𑀭𑀸 𑀓𑀭𑀡𑀻𑀬𑀁 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁂𑀦 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀢𑀁 𑀯𑁄 𑀢𑀁 𑀫𑀬𑀸𑁇 𑀏𑀢𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀸𑀦𑀺, 𑀏𑀢𑀸𑀦𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀸𑀦𑀺; 𑀛𑀸𑀬𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸 𑀧𑀫𑀸𑀤𑀢𑁆𑀣; 𑀫𑀸 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄 𑀅𑀳𑀼𑀯𑀢𑁆𑀣𑁇 𑀅𑀬𑀁 𑀯𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻”𑀢𑀺𑁇

732\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

733\. 𑀤𑁆𑀯𑁂𑀥𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀦𑀯𑀫𑀁𑁇
