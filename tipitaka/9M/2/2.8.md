# 𑁨𑁇𑁮 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀻𑀳𑀦𑀸𑀤𑀯𑀕𑁆𑀕

* 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(199.)

700\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀬𑁂𑀦 𑀫𑀳𑀸𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀫𑀳𑀸𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀩𑁂𑀮𑀼𑀯𑀮𑀝𑁆𑀞𑀺𑀓𑀸𑀬 𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺𑀧𑀺 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀚𑀗𑁆𑀖𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀦𑀼𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀫𑀳𑀸𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀫𑀳𑀸𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑁂𑀮𑀼𑀯𑀮𑀝𑁆𑀞𑀺𑀓𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀤𑀡𑁆𑀟𑀫𑁄𑀮𑀼𑀩𑁆𑀪 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁𑀯𑀸𑀤𑀻 𑀲𑀫𑀡𑁄 𑀓𑀺𑀫𑀓𑁆𑀔𑀸𑀬𑀻”𑀢𑀺? “𑀬𑀣𑀸𑀯𑀸𑀤𑀻 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀦 𑀓𑁂𑀦𑀘𑀺 𑀮𑁄𑀓𑁂 𑀯𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀦 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀅𑀓𑀣𑀁𑀓𑀣𑀺𑀁 𑀙𑀺𑀦𑁆𑀦𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀪𑀯𑀸𑀪𑀯𑁂 𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀁 𑀲𑀜𑁆𑀜𑀸 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺—  𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀫𑀓𑁆𑀔𑀸𑀬𑀻”𑀢𑀺𑁇

701\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺 𑀲𑀓𑁆𑀓𑁄 𑀲𑀻𑀲𑀁 𑀑𑀓𑀫𑁆𑀧𑁂𑀢𑁆𑀯𑀸, 𑀚𑀺𑀯𑁆𑀳𑀁 𑀦𑀺𑀮𑁆𑀮𑀸𑀍𑀏𑀢𑁆𑀯𑀸, 𑀢𑀺𑀯𑀺𑀲𑀸𑀔𑀁 𑀦𑀮𑀸𑀝𑀺𑀓𑀁 𑀦𑀮𑀸𑀝𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑀡𑁆𑀟𑀫𑁄𑀮𑀼𑀩𑁆𑀪 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

(200.)

702\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑁇 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀬𑁂𑀦 𑀫𑀳𑀸𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀫𑀳𑀸𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀩𑁂𑀮𑀼𑀯𑀮𑀝𑁆𑀞𑀺𑀓𑀸𑀬 𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑁇 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀚𑀗𑁆𑀖𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀦𑀼𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀫𑀳𑀸𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀫𑀳𑀸𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑁂𑀮𑀼𑀯𑀮𑀝𑁆𑀞𑀺𑀓𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀤𑀡𑁆𑀟𑀫𑁄𑀮𑀼𑀩𑁆𑀪 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺 𑀲𑀓𑁆𑀓𑁄 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀓𑀺𑀁𑀯𑀸𑀤𑀻 𑀲𑀫𑀡𑁄 𑀓𑀺𑀫𑀓𑁆𑀔𑀸𑀬𑀻’𑀢𑀺?

703\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺𑀁 𑀲𑀓𑁆𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀬𑀣𑀸𑀯𑀸𑀤𑀻 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀦 𑀓𑁂𑀦𑀘𑀺 𑀮𑁄𑀓𑁂 𑀯𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀦 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀅𑀓𑀣𑀁𑀓𑀣𑀺𑀁 𑀙𑀺𑀦𑁆𑀦𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀪𑀯𑀸𑀪𑀯𑁂 𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀁 𑀲𑀜𑁆𑀜𑀸 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺—  𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀫𑀓𑁆𑀔𑀸𑀬𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺 𑀲𑀓𑁆𑀓𑁄 𑀲𑀻𑀲𑀁 𑀑𑀓𑀫𑁆𑀧𑁂𑀢𑁆𑀯𑀸, 𑀚𑀺𑀯𑁆𑀳𑀁 𑀦𑀺𑀮𑁆𑀮𑀸𑀍𑀏𑀢𑁆𑀯𑀸, 𑀢𑀺𑀯𑀺𑀲𑀸𑀔𑀁 𑀦𑀮𑀸𑀝𑀺𑀓𑀁 𑀦𑀮𑀸𑀝𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑀡𑁆𑀟𑀫𑁄𑀮𑀼𑀩𑁆𑀪 𑀧𑀓𑁆𑀓𑀸𑀫𑀻”𑀢𑀺𑁇

(201.)

704\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁𑀯𑀸𑀤𑀻 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀦 𑀓𑁂𑀦𑀘𑀺 𑀮𑁄𑀓𑁂 𑀯𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀢𑀺? 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀅𑀓𑀣𑀁𑀓𑀣𑀺𑀁 𑀙𑀺𑀦𑁆𑀦𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀪𑀯𑀸𑀪𑀯𑁂 𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀁 𑀲𑀜𑁆𑀜𑀸 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀘𑁂 𑀦𑀢𑁆𑀣𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀡𑁆𑀟𑀸𑀤𑀸𑀦𑀲𑀢𑁆𑀣𑀸𑀤𑀸𑀦𑀓𑀮𑀳𑀯𑀺𑀕𑁆𑀕𑀳𑀯𑀺𑀯𑀸𑀤𑀢𑀼𑀯𑀁𑀢𑀼𑀯𑀁𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸𑀦𑀁𑁇 𑀏𑀢𑁆𑀣𑁂𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

705\. 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀤𑀁 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸, 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸, 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄—  ‘𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀘𑁂 𑀦𑀢𑁆𑀣𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑁆𑀣𑁂𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀇𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀪𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀬𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀲𑀁𑀯𑀡𑁆𑀡𑀺𑀢𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑀺𑀢𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀧𑀳𑁄𑀢𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀪𑀚𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇

706\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀤𑀁 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄—  ‘𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀘𑁂 𑀦𑀢𑁆𑀣𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑁆𑀣𑁂𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀤𑀁 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄—  “𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀘𑁂 𑀦𑀢𑁆𑀣𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑁆𑀣𑁂𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻”’𑀢𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀇𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀪𑀚𑁂𑀬𑁆𑀬𑀸𑀢𑀺? 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀲𑀁𑀯𑀡𑁆𑀡𑀺𑀢𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑀺𑀢𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁, 𑀧𑀳𑁄𑀢𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀪𑀚𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀯𑀺𑀪𑀚𑀢𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄”𑀢𑀺𑁇

708\. “𑀬𑀁 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄—  ‘𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀘𑁂 𑀦𑀢𑁆𑀣𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑁆𑀣𑁂𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺—

709\. 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀽𑀧𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀗𑁆𑀕𑀢𑀺 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀬𑀁 𑀯𑁂𑀤𑁂𑀢𑀺 𑀢𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺, 𑀬𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀢𑀁 𑀧𑀧𑀜𑁆𑀘𑁂𑀢𑀺, 𑀬𑀁 𑀧𑀧𑀜𑁆𑀘𑁂𑀢𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀭𑀽𑀧𑁂𑀲𑀼𑁇 𑀲𑁄𑀢𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀤𑁆𑀤𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀕𑀦𑁆𑀥𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀲𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀫𑀦𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀗𑁆𑀕𑀢𑀺 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀬𑀁 𑀯𑁂𑀤𑁂𑀢𑀺 𑀢𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺, 𑀬𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀢𑀁 𑀧𑀧𑀜𑁆𑀘𑁂𑀢𑀺, 𑀬𑀁 𑀧𑀧𑀜𑁆𑀘𑁂𑀢𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇

710\. 𑀲𑁄 𑀯𑀢𑀸𑀯𑀼𑀲𑁄, 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀭𑀽𑀧𑁂 𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑁄𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀲𑀤𑁆𑀤𑁂 𑀲𑀢𑀺…𑀧𑁂…  𑀖𑀸𑀦𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀕𑀦𑁆𑀥𑁂 𑀲𑀢𑀺…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀲𑀢𑀺 𑀭𑀲𑁂 𑀲𑀢𑀺…𑀧𑁂…  𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀲𑀢𑀺…𑀧𑁂…  𑀫𑀦𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑁂 𑀲𑀢𑀺 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

711\. 𑀲𑁄 𑀯𑀢𑀸𑀯𑀼𑀲𑁄, 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀭𑀽𑀧𑁂 𑀅𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀅𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑁄𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀲𑀤𑁆𑀤𑁂 𑀅𑀲𑀢𑀺…𑀧𑁂…  𑀖𑀸𑀦𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀕𑀦𑁆𑀥𑁂 𑀅𑀲𑀢𑀺…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀅𑀲𑀢𑀺 𑀭𑀲𑁂 𑀅𑀲𑀢𑀺…𑀧𑁂…  𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀅𑀲𑀢𑀺…𑀧𑁂…  𑀫𑀦𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑁂 𑀅𑀲𑀢𑀺 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀅𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀨𑀲𑁆𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

713\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀬𑀁 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄—  ‘𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀘𑁂 𑀦𑀢𑁆𑀣𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑁆𑀣𑁂𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀤𑀁 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄—  “𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀘𑁂 𑀦𑀢𑁆𑀣𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀸𑀦𑀁, 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀡𑁆𑀟𑀸𑀤𑀸𑀦𑀲𑀢𑁆𑀣𑀸𑀤𑀸𑀦𑀓𑀮𑀳𑀯𑀺𑀕𑁆𑀕𑀳𑀯𑀺𑀯𑀸𑀤𑀢𑀼𑀯𑀁𑀢𑀼𑀯𑀁𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸𑀦𑀁𑁇 𑀏𑀢𑁆𑀣𑁂𑀢𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀇𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀪𑀚𑁂𑀬𑁆𑀬𑀸’𑀢𑀺? 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀲𑀁𑀯𑀡𑁆𑀡𑀺𑀢𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑀺𑀢𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁, 𑀧𑀳𑁄𑀢𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀪𑀚𑀺𑀢𑀼𑀁, 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀫𑁆𑀳; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁂𑀦 𑀇𑀫𑁂𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀇𑀫𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀇𑀫𑁂𑀳𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀺𑀪𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄; 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄𑁇 𑀫𑀜𑁆𑀘𑁂𑀧𑀺 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀣, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀁 𑀏𑀯𑀫𑁂𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁂𑀦 𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀏𑀲𑁄 𑀘𑁂𑀯𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 𑀏𑀯𑀜𑁆𑀘 𑀦𑀁 𑀥𑀸𑀭𑁂𑀣𑀸”𑀢𑀺𑁇

714\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀧𑀭𑁂𑀢𑁄 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀬𑀢𑁄 𑀬𑀢𑁄 𑀲𑀸𑀬𑁂𑀬𑁆𑀬, 𑀮𑀪𑁂𑀣𑁂𑀯 𑀲𑀸𑀤𑀼𑀭𑀲𑀁 𑀅𑀲𑁂𑀘𑀦𑀓𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑁂𑀢𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀩𑁆𑀩𑀚𑀸𑀢𑀺𑀓𑁄, 𑀬𑀢𑁄 𑀬𑀢𑁄 𑀇𑀫𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀢𑁆𑀣𑀁 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀮𑀪𑁂𑀣𑁂𑀯 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀁, 𑀮𑀪𑁂𑀣𑁂𑀯 𑀘𑁂𑀢𑀲𑁄 𑀧𑀲𑀸𑀤𑀁𑁇 𑀓𑁄 𑀦𑀸𑀫𑁄 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄”𑀢𑀺? “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁄𑀢𑁆𑀯𑁂𑀯 𑀦𑀁 𑀥𑀸𑀭𑁂𑀳𑀻”𑀢𑀺𑁇

715\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀻𑀢𑀺𑁇
