# 𑁨𑁇𑁧 𑀘𑀽𑀍𑀅𑀲𑀻𑀳𑀦𑀸𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀻𑀳𑀦𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀲𑀻𑀳𑀦𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(139.)

452\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

453\. “𑀇𑀥𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀢𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀲𑀫𑀡𑁄; 𑀲𑀼𑀜𑁆𑀜𑀸 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀸 𑀲𑀫𑀡𑁂𑀪𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀻𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀲𑀻𑀳𑀦𑀸𑀤𑀁 𑀦𑀤𑀣𑁇

(140.)

454\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑁄, 𑀓𑀺𑀁 𑀩𑀮𑀁, 𑀬𑁂𑀦 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀣—  𑀇𑀥𑁂𑀯 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀢𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀲𑀫𑀡𑁄; 𑀲𑀼𑀜𑁆𑀜𑀸 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀸 𑀲𑀫𑀡𑁂𑀪𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀻’𑀢𑀺? 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀅𑀢𑁆𑀣𑀺 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀸 𑀬𑁂 𑀫𑀬𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫—  𑀇𑀥𑁂𑀯 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀢𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀲𑀫𑀡𑁄; 𑀲𑀼𑀜𑁆𑀜𑀸 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀸 𑀲𑀫𑀡𑁂𑀪𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀻𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀣𑀭𑀺 𑀧𑀲𑀸𑀤𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀥𑀫𑁆𑀫𑁂 𑀧𑀲𑀸𑀤𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀢𑀸; 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀔𑁄 𑀧𑀦 𑀧𑀺𑀬𑀸 𑀫𑀦𑀸𑀧𑀸—  𑀕𑀳𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀘𑁇 𑀇𑀫𑁂 𑀔𑁄 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀸 𑀬𑁂 𑀫𑀬𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫—  𑀇𑀥𑁂𑀯 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀢𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀲𑀫𑀡𑁄; 𑀲𑀼𑀜𑁆𑀜𑀸 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀸 𑀲𑀫𑀡𑁂𑀪𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀻’𑀢𑀺𑁇

(141.)

455\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀣𑀭𑀺 𑀧𑀲𑀸𑀤𑁄 𑀬𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀢𑁆𑀣𑀸, 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀥𑀫𑁆𑀫𑁂 𑀧𑀲𑀸𑀤𑁄 𑀬𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄 𑀬𑀸𑀦𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀻𑀮𑀸𑀦𑀺, 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀧𑀺𑀬𑀸 𑀫𑀦𑀸𑀧𑀸—  𑀕𑀳𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀘𑁇 𑀇𑀥 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄 𑀯𑀺𑀲𑁂𑀲𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀬𑀸𑀲𑁄 𑀓𑀺𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀬𑀤𑀺𑀤𑀁 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁂𑀯 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑀸’𑀢𑀺?

456\. 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀓𑀺𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀏𑀓𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀉𑀤𑀸𑀳𑀼 𑀧𑀼𑀣𑀼 𑀦𑀺𑀝𑁆𑀞𑀸’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀏𑀓𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀧𑀼𑀣𑀼 𑀦𑀺𑀝𑁆𑀞𑀸’𑀢𑀺𑁇

457\. ‘𑀲𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀭𑀸𑀕𑀲𑁆𑀲 𑀉𑀤𑀸𑀳𑀼 𑀯𑀻𑀢𑀭𑀸𑀕𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀯𑀻𑀢𑀭𑀸𑀕𑀲𑁆𑀲𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀭𑀸𑀕𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

458\. ‘𑀲𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀤𑁄𑀲𑀲𑁆𑀲 𑀉𑀤𑀸𑀳𑀼 𑀯𑀻𑀢𑀤𑁄𑀲𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀯𑀻𑀢𑀤𑁄𑀲𑀲𑁆𑀲𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀤𑁄𑀲𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

459\. ‘𑀲𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀫𑁄𑀳𑀲𑁆𑀲 𑀉𑀤𑀸𑀳𑀼 𑀯𑀻𑀢𑀫𑁄𑀳𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀯𑀻𑀢𑀫𑁄𑀳𑀲𑁆𑀲𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀫𑁄𑀳𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

460\. ‘𑀲𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀢𑀡𑁆𑀳𑀲𑁆𑀲 𑀉𑀤𑀸𑀳𑀼 𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀲𑁆𑀲𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑀢𑀡𑁆𑀳𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

461\. ‘𑀲𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀉𑀤𑀸𑀳𑀼 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

462\. ‘𑀲𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸 𑀯𑀺𑀤𑁆𑀤𑀲𑀼𑀦𑁄 𑀉𑀤𑀸𑀳𑀼 𑀅𑀯𑀺𑀤𑁆𑀤𑀲𑀼𑀦𑁄’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀯𑀺𑀤𑁆𑀤𑀲𑀼𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀯𑀺𑀤𑁆𑀤𑀲𑀼𑀦𑁄’𑀢𑀺𑁇

463\. ‘𑀲𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀉𑀤𑀸𑀳𑀼 𑀅𑀦𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀦𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀸 𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

466\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺? 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑀸𑁇 𑀢𑁂 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺—  𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀇𑀫𑀸𑀦𑀺 𑀳𑀺 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀢𑀻𑀡𑀺 𑀞𑀸𑀦𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑀸; 𑀢𑁂 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺—  𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇

467\. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑀸𑁇 𑀢𑁂 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺—  𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀇𑀫𑀸𑀦𑀺 𑀳𑀺 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀤𑁆𑀯𑁂 𑀞𑀸𑀦𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑀸; 𑀢𑁂 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺—  𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇

468\. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑀸𑁇 𑀢𑁂 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺—  𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀇𑀫𑀜𑁆𑀳𑀺 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀓𑀁 𑀞𑀸𑀦𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑀸; 𑀢𑁂 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺—  𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇

471\. 𑀇𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑁇 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸? 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀤𑀸𑀦𑀸 𑀢𑀡𑁆𑀳𑀸𑀲𑀫𑀼𑀤𑀬𑀸 𑀢𑀡𑁆𑀳𑀸𑀚𑀸𑀢𑀺𑀓𑀸 𑀢𑀡𑁆𑀳𑀸𑀧𑀪𑀯𑀸𑁇 𑀢𑀡𑁆𑀳𑀸 𑀘𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸? 𑀢𑀡𑁆𑀳𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀤𑀸𑀦𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀚𑀸𑀢𑀺𑀓𑀸 𑀯𑁂𑀤𑀦𑀸𑀧𑀪𑀯𑀸𑁇 𑀯𑁂𑀤𑀦𑀸 𑀘𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸? 𑀯𑁂𑀤𑀦𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀤𑀸𑀦𑀸 𑀨𑀲𑁆𑀲𑀲𑀫𑀼𑀤𑀬𑀸 𑀨𑀲𑁆𑀲𑀚𑀸𑀢𑀺𑀓𑀸 𑀨𑀲𑁆𑀲𑀧𑀪𑀯𑀸𑁇 𑀨𑀲𑁆𑀲𑁄 𑀘𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑁄 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑁄 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑁄 𑀓𑀺𑀁𑀧𑀪𑀯𑁄? 𑀨𑀲𑁆𑀲𑁄 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀤𑀸𑀦𑁄 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑀫𑀼𑀤𑀬𑁄 𑀲𑀍𑀆𑀬𑀢𑀦𑀚𑀸𑀢𑀺𑀓𑁄 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀪𑀯𑁄𑁇 𑀲𑀍𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁 𑀓𑀺𑀁𑀧𑀪𑀯𑀁? 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀤𑀸𑀦𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀚𑀸𑀢𑀺𑀓𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀪𑀯𑀁𑁇 𑀦𑀸𑀫𑀭𑀽𑀧𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁 𑀓𑀺𑀁𑀧𑀪𑀯𑀁? 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀤𑀸𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀚𑀸𑀢𑀺𑀓𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀪𑀯𑀁𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁 𑀓𑀺𑀁𑀧𑀪𑀯𑀁? 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀤𑀸𑀦𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀚𑀸𑀢𑀺𑀓𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀪𑀯𑀁𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸? 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀤𑀸𑀦𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀫𑀼𑀤𑀬𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀚𑀸𑀢𑀺𑀓𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀪𑀯𑀸𑁇

472\. 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀭𑀸𑀕𑀸 𑀯𑀺𑀚𑁆𑀚𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀦𑁂𑀯 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺, 𑀦 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺, 𑀦 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀜𑁆𑀜𑁂𑀯 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇

473\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

474\. 𑀘𑀽𑀍𑀅𑀲𑀻𑀳𑀦𑀸𑀤𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀧𑀞𑀫𑀁𑁇
