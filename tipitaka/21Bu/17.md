# 𑁧𑁭 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

722\. _“𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀡𑁆𑀟𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀢𑀫𑀦𑁆𑀥𑀓𑀸𑀭𑀁 𑀯𑀺𑀥𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁂𑁇_  


(2.)

723\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀅𑀢𑀼𑀮𑀢𑁂𑀚𑀲𑁆𑀲,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀦𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(3.)

724\. _𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻,_  
_𑀯𑀺𑀦𑁂𑀲𑀺 𑀲𑀜𑁆𑀚𑀬𑀁 𑀇𑀲𑀺𑀁;_  
_𑀢𑀤𑀸 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

725\. _𑀬𑀤𑀸 𑀲𑀓𑁆𑀓𑁄 𑀉𑀧𑀸𑀕𑀜𑁆𑀙𑀺,_  
_𑀲𑀧𑀭𑀺𑀲𑁄 𑀯𑀺𑀦𑀸𑀬𑀓𑀁;_  
_𑀢𑀤𑀸 𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

726\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀤𑁂𑀯𑀤𑁂𑀯𑀲𑁆𑀲,_  
_𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀅𑀳𑀼𑀁;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(6.)

727\. _𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻,_  
_𑀲𑀭𑀡𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀸𑀕𑀫𑀺;_  
_𑀢𑀤𑀸 𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(7.)

728\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀤𑁂𑀯𑀢𑁄 𑀏𑀢𑀺 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀢𑀤𑀸𑀧𑀺 𑀲𑀢𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(8.)

729\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀧𑀓𑀸𑀲𑁂𑀲𑀺 𑀥𑀼𑀢𑁂 𑀕𑀼𑀡𑁂;_  
_𑀢𑀤𑀸 𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(9.)

730\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀲𑀓𑁆𑀓𑁄 𑀆𑀲𑀺𑀁 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑁂𑀦,_  
_𑀢𑀽𑀭𑀺𑀬𑁂𑀦𑀸𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(10.)

731\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀤𑁂𑀯𑀫𑀚𑁆𑀛𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀬;_  
_‘𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(11.)

732\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(12.)

733\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(13.)

734\. _𑀲𑀭𑀡𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀲𑀭𑀡𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀼𑀦𑀦𑁆𑀤𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(14.)

735\. _𑀅𑀝𑁆𑀞𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀅𑀭𑀚𑁄 𑀯𑀺𑀭𑀚𑁄 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(15.)

736\. _𑀢𑀺𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀯𑀺𑀘𑀺𑀓𑁄𑀍𑀇 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀅𑀢𑁆𑀭𑀚𑁄 𑀧𑀼𑀜𑁆𑀜𑀯𑀟𑁆𑀠𑀦𑁄𑁇_  


(16.)

737\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀸𑀲𑀸𑀤𑁂𑀦𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(17.)

738\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(18.)

739\. _𑀧𑀤𑀼𑀫𑁄 𑀨𑀼𑀲𑁆𑀲𑀤𑁂𑀯𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑀼𑀦𑁂𑀢𑁆𑀢𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(19.)

740\. _𑀔𑁂𑀫𑀸 𑀘 𑀲𑀘𑁆𑀘𑀦𑀸𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀩𑀺𑀫𑁆𑀩𑀺𑀚𑀸𑀮𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(20.)

741\. _𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀓𑀝𑀺𑀲𑁆𑀲𑀳𑁄 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀲𑀸𑀍𑀇𑀬𑀸 𑀘 𑀓𑀍𑀇𑀬𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(21.)

742\. _𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀅𑀲𑀻𑀢𑀺𑀳𑀢𑁆𑀣𑀫𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀢𑁂𑀚𑁂𑀦,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑁆𑀳𑀺 𑀥𑀸𑀢𑀼𑀬𑀸𑁇_  


(22.)

743\. _𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑁄 𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯,_  
_𑀯𑀺𑀚𑁆𑀚𑀽𑀯 𑀕𑀕𑀦𑁂 𑀬𑀣𑀸;_  
_𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂𑀯 𑀲𑀽𑀭𑀺𑀬𑁄,_  
_𑀏𑀯𑀁 𑀲𑁄 𑀉𑀧𑀲𑁄𑀪𑀣𑁇_  


(23.)

744\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀅𑀢𑀼𑀮𑀢𑁂𑀚𑀲𑁆𑀲,_  
_𑀲𑀫𑀓𑀁 𑀆𑀲𑀺 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀮𑁄𑀓𑁂 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(24.)

745\. _𑀑𑀪𑀸𑀲𑀁 𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀫𑀮𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀲𑀸𑀲𑀦𑀁;_  
_𑀘𑀯𑀺 𑀘𑀦𑁆𑀤𑁄𑀯 𑀕𑀕𑀦𑁂,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(25.)

746\. _𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀸𑀮𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀣𑀽𑀧𑀯𑀭𑁄,_  
_𑀢𑀻𑀡𑀺𑀬𑁄𑀚𑀦𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁭𑁨𑁯)_  


---

747\. 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑀫𑁄𑁇
