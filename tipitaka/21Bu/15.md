# 𑁧𑁫 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

667\. _“𑀲𑀼𑀚𑀸𑀢𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀤𑀼𑀭𑀸𑀲𑀤𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(2.)

668\. _𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑁄𑀯 𑀯𑀺𑀭𑁄𑀘𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀁 𑀦𑀺𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀬𑀺𑁇_  


(3.)

669\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀅𑀢𑀼𑀮𑀢𑁂𑀚𑀲𑁆𑀲,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀸 𑀢𑀬𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

670\. _𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀤𑁂𑀯𑀭𑀸𑀚𑀸,_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀫𑀭𑁄𑀘𑀬𑀺;_  
_𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀯𑀺𑀦𑁄𑀤𑁂𑀦𑁆𑀢𑁄,_  
_𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑀬𑀺𑁇_  


(5.)

671\. _𑀚𑀦𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀢𑀼𑀮𑁄,_  
_𑀫𑀳𑀸𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀻 𑀢𑀤𑀸;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

672\. _𑀬𑀤𑀸 𑀤𑁄𑀡𑀫𑀼𑀔𑀁 𑀳𑀢𑁆𑀣𑀺𑀁,_  
_𑀯𑀺𑀦𑁂𑀲𑀺 𑀦𑀭𑀲𑀸𑀭𑀣𑀺;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(7.)

673\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(8.)

674\. _𑀢𑀢𑁄 𑀧𑀭𑀁 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀻,_  
_𑀲𑀫𑀺𑀁𑀲𑀼 𑀏𑀓𑀢𑁄 𑀫𑀼𑀦𑀻;_  
_𑀢𑀢𑀺𑀬𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀫𑁆𑀳𑀺,_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀬𑁄 𑀅𑀳𑀽𑁇_  


(9.)

675\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽𑁇_  


(10.)

676\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀲𑀸𑀤𑀁 𑀚𑀦𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀲𑀁𑀖𑀸𑀭𑀸𑀫𑀁 𑀅𑀫𑀸𑀧𑀬𑀺𑀁𑁇_  


(11.)

677\. _𑀢𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦 𑀆𑀭𑀸𑀫𑀁,_  
_𑀳𑀝𑁆𑀞𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑁄;_  
_𑀲𑀭𑀡𑁂 𑀧𑀜𑁆𑀘 𑀲𑀻𑀮𑁂 𑀘,_  
_𑀤𑀍𑀳𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀁𑁇_  


(12.)

678\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀬;_  
_‘𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(13.)

679\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(14.)

680\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(15.)

681\. _𑀲𑀼𑀥𑀜𑁆𑀜𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀲𑀼𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀘𑀦𑁆𑀤𑀸 𑀦𑀸𑀫𑀸𑀲𑀺 𑀚𑀦𑀺𑀓𑀸,_  
_𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(16.)

682\. _𑀦𑀯𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀮𑀯𑀺𑀫𑀮𑀕𑀺𑀭𑀺𑀕𑀼𑀳𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(17.)

683\. _𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀘,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀯𑀺𑀫𑀮𑀸 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻 𑀘,_  
_𑀓𑀜𑁆𑀘𑀦𑀸𑀯𑁂𑀍𑀑 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(18.)

684\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀭𑀣𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀙𑀫𑀸𑀲𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(19.)

685\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀉𑀲𑀪𑀼𑀬𑁆𑀬𑀸𑀦𑁂 𑀫𑀦𑁄𑀭𑀫𑁂𑁇_  


(20.)

686\. _𑀧𑀸𑀮𑀺𑀢𑁄 𑀲𑀩𑁆𑀩𑀤𑀲𑁆𑀲𑀻 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑁄𑀪𑀺𑀢𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(21.)

687\. _𑀲𑀼𑀚𑀸𑀢𑀸 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀓𑀓𑀼𑀥𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(22.)

688\. _𑀲𑀦𑁆𑀥𑀓𑁄 𑀥𑀫𑁆𑀫𑀓𑁄 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀯𑀺𑀲𑀸𑀔𑀸 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(23.)

689\. _𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄;_  
_𑀅𑀲𑀻𑀢𑀺𑀳𑀢𑁆𑀣𑀫𑀼𑀩𑁆𑀩𑁂𑀥𑁄,_  
_𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(24.)

690\. _𑀅𑀕𑁆𑀕𑀺𑀘𑀦𑁆𑀤𑀲𑀽𑀭𑀺𑀬𑀸𑀦𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀢𑀸𑀤𑀺𑀲𑀺𑀓𑀸 𑀧𑀪𑀸;_  
_𑀬𑀣𑀸 𑀅𑀳𑀼 𑀧𑀪𑀸 𑀢𑀲𑁆𑀲,_  
_𑀅𑀲𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(25.)

691\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀤𑁂𑀯𑀤𑁂𑀯𑀲𑁆𑀲,_  
_𑀆𑀬𑀼 𑀢𑀸𑀯𑀢𑀓𑀁 𑀅𑀳𑀼;_  
_𑀦𑀯𑀼𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀮𑁄𑀓𑁂 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(26.)

692\. _𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀬𑀼𑀕𑀸𑀦𑀺𑀧𑀺 𑀢𑀸𑀦𑀺 𑀅𑀢𑀼𑀮𑀺𑀬𑀸𑀦𑀺;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(27.)

693\. _𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻 𑀫𑀼𑀦𑀺𑀯𑀭𑁄,_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀚𑀺𑀦𑀣𑀽𑀧𑁄,_  
_𑀢𑀻𑀡𑀺𑀬𑁄𑀚𑀦𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁬𑁭𑁮)_  


---

694\. 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀢𑁂𑀭𑀲𑀫𑁄𑁇
