# 𑁧𑁩 𑀲𑀼𑀫𑁂𑀥𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀲𑀼𑀫𑁂𑀥𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

598\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀤𑀼𑀭𑀸𑀲𑀤𑁄 𑀉𑀕𑁆𑀕𑀢𑁂𑀚𑁄,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀫𑁄 𑀫𑀼𑀦𑀺𑁇_  


(2.)

599\. _𑀧𑀲𑀦𑁆𑀦𑀦𑁂𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀼𑀔𑁄,_  
_𑀩𑁆𑀭𑀳𑀸 𑀉𑀚𑀼 𑀧𑀢𑀸𑀧𑀯𑀸;_  
_𑀳𑀺𑀢𑁂𑀲𑀻 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀩𑀳𑀽 𑀫𑁄𑀘𑁂𑀲𑀺 𑀩𑀦𑁆𑀥𑀦𑀸𑁇_  


(3.)

600\. _𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄 𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑁂𑀯𑀮𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀫𑁆𑀳𑀺 𑀦𑀕𑀭𑁂,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀬𑀺𑁇_  


(4.)

601\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀅𑀪𑀺𑀲𑀫𑀬𑀸 𑀢𑀻𑀡𑀺,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

602\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀓𑀼𑀫𑁆𑀪𑀓𑀡𑁆𑀡𑀁,_  
_𑀬𑀓𑁆𑀔𑀁 𑀲𑁄 𑀤𑀫𑀬𑀻 𑀚𑀺𑀦𑁄;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

603\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀁 𑀧𑀓𑀸𑀲𑀬𑀺;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(7.)

604\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀲𑀼𑀫𑁂𑀥𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(8.)

605\. _𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀉𑀧𑀕𑀜𑁆𑀙𑀺 𑀚𑀺𑀦𑁄 𑀬𑀤𑀸;_  
_𑀢𑀤𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽,_  
_𑀲𑀫𑀺𑀁𑀲𑀼 𑀲𑀢𑀓𑁄𑀝𑀺𑀬𑁄𑁇_  


(9.)

606\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀤𑁂𑀯𑀓𑀽𑀝𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀢𑁂;_  
_𑀢𑀤𑀸 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(10.)

607\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀤𑀲𑀩𑀮𑁄,_  
_𑀬𑀤𑀸 𑀘𑀭𑀢𑀺 𑀘𑀸𑀭𑀺𑀓𑀁;_  
_𑀢𑀤𑀸 𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(11.)

608\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀫𑀸𑀡𑀯𑁄;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀬𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀖𑀭𑁂 𑀲𑀦𑁆𑀦𑀺𑀘𑀺𑀢𑀁 𑀥𑀦𑀁𑁇_  


(12.)

609\. _𑀓𑁂𑀯𑀮𑀁 𑀲𑀩𑁆𑀩𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀲𑀁𑀖𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀲𑀭𑀡𑀁 𑀢𑀲𑁆𑀲𑀼𑀧𑀕𑀜𑁆𑀙𑀺𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀜𑁆𑀘𑀸𑀪𑀺𑀭𑁄𑀘𑀬𑀺𑀁𑁇_  


(13.)

610\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑀁;_  
_‘𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(14.)

611\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(15.)

612\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(16.)

613\. _𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀯𑀺𑀦𑀬𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀦𑀯𑀗𑁆𑀕𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑁄𑀪𑀬𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(17.)

614\. _𑀢𑀢𑁆𑀣𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀲𑀚𑁆𑀚𑀝𑁆𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑁂;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀫𑀕𑀜𑁆𑀙𑀳𑀁𑁇_  


(18.)

615\. _𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀲𑀼𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀼𑀤𑀢𑁆𑀢𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀲𑀼𑀫𑁂𑀥𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(19.)

616\. _𑀦𑀯𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀲𑀼𑀘𑀦𑁆𑀤𑀓𑀜𑁆𑀘𑀦𑀲𑀺𑀭𑀺𑀯𑀟𑁆𑀠𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(20.)

617\. _𑀢𑀺𑀲𑁄𑀍𑀅𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑀼𑀫𑀦𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(21.)

618\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀓𑀁 𑀅𑀟𑁆𑀠𑀫𑀸𑀲𑀁,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(22.)

619\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀼𑀬𑁆𑀬𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(23.)

620\. _𑀲𑀭𑀡𑁄 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑀸𑀕𑀭𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀲𑀼𑀫𑁂𑀥𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(24.)

621\. _𑀭𑀸𑀫𑀸 𑀘𑁂𑀯 𑀲𑀼𑀭𑀸𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀫𑀳𑀸𑀦𑀻𑀧𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(25.)

622\. _𑀉𑀭𑀼𑀯𑁂𑀮𑀸 𑀬𑀲𑀯𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀬𑀲𑁄𑀥𑀭𑀸 𑀲𑀺𑀭𑀺𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(26.)

623\. _𑀅𑀝𑁆𑀞𑀸𑀲𑀻𑀢𑀺𑀭𑀢𑀦𑀸𑀦𑀺,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀑𑀪𑀸𑀲𑁂𑀢𑀺 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀘𑀦𑁆𑀤𑁄 𑀢𑀸𑀭𑀕𑀡𑁂 𑀬𑀣𑀸𑁇_  


(27.)

624\. _𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀫𑀡𑀻 𑀦𑀸𑀫,_  
_𑀬𑀣𑀸 𑀢𑀧𑀢𑀺 𑀬𑁄𑀚𑀦𑀁;_  
_𑀢𑀣𑁂𑀯 𑀢𑀲𑁆𑀲 𑀭𑀢𑀦𑀁,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀨𑀭𑀢𑀺 𑀬𑁄𑀚𑀦𑀁𑁇_  


(28.)

625\. _𑀦𑀯𑀼𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(29.)

626\. _𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀳𑀺,_  
_𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀳𑀺;_  
_𑀲𑀫𑀸𑀓𑀼𑀮𑀫𑀺𑀤𑀁 𑀆𑀲𑀺,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀸𑀥𑀼𑀳𑀺𑁇_  


(30.)

627\. _𑀢𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑁂 𑀅𑀫𑀺𑀢𑀬𑀲𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀦𑀺𑀭𑀽𑀧𑀥𑀻;_  
_𑀜𑀸𑀡𑀸𑀮𑁄𑀓𑀁 𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀢𑁂 𑀫𑀳𑀸𑀬𑀲𑀸𑁇_  


(31.)

628\. _𑀲𑀼𑀫𑁂𑀥𑁄 𑀚𑀺𑀦𑀯𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀫𑁂𑀥𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄”𑀢𑀺𑁇 (𑁬𑁧𑁫)_  


---
