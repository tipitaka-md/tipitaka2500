# 𑁨𑁪 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

926\. _“𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀭𑀸𑀲𑀤𑁄𑁇_  


(2.)

927\. _𑀉𑀕𑁆𑀖𑀸𑀝𑁂𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀪𑀯𑀁,_  
_𑀘𑀭𑀺𑀬𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑁄;_  
_𑀲𑀻𑀳𑁄𑀯 𑀧𑀜𑁆𑀚𑀭𑀁 𑀪𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(3.)

928\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

929\. _𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀫𑁆𑀳𑀺 𑀆𑀓𑀸𑀲𑁂,_  
_𑀬𑀫𑀓𑀁 𑀓𑀢𑁆𑀯𑀸 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀁;_  
_𑀢𑀺𑀁𑀲𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀩𑁄𑀥𑁂𑀲𑀺 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑁂𑁇_  


(5.)

930\. _𑀦𑀭𑀤𑁂𑀯𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀧𑁆𑀧𑀓𑀸𑀲𑀦𑁂;_  
_𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀢𑀲𑁆𑀲,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑁄𑁇_  


(6.)

931\. _𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(7.)

932\. _𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀤𑀦𑁆𑀢𑀪𑀽𑀫𑀺𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀆𑀲𑀯𑀸𑀭𑀺𑀕𑀡𑀓𑁆𑀔𑀬𑀸𑁇_  


(8.)

933\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀔𑁂𑀫𑁄 𑀦𑀸𑀫𑀸𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀢𑀣𑀸𑀕𑀢𑁂 𑀚𑀺𑀦𑀧𑀼𑀢𑁆𑀢𑁂,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀦𑀧𑁆𑀧𑀓𑀁𑁇_  


(9.)

934\. _𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀅𑀜𑁆𑀚𑀦𑀁 𑀫𑀥𑀼𑀮𑀝𑁆𑀞𑀺𑀓𑀁;_  
_𑀇𑀫𑁂𑀢𑀁 𑀧𑀢𑁆𑀣𑀺𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀫𑀺 𑀯𑀭𑀁 𑀯𑀭𑀁𑁇_  


(10.)

935\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑁄 𑀯𑀺𑀦𑀸𑀬𑀓𑁄;_  
_‘𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(11.)

936\. _𑀅𑀳𑀼 𑀓𑀧𑀺𑀮𑀯𑁆𑀳𑀬𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(12.)

937\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(13.)

938\. _𑀦𑀕𑀭𑀁 𑀔𑁂𑀫𑀸𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀔𑁂𑀫𑁄 𑀦𑀸𑀫𑀸𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(14.)

939\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀅𑀕𑁆𑀕𑀺𑀤𑀢𑁆𑀢𑁄 𑀘,_  
_𑀆𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑁄 𑀧𑀺𑀢𑀸;_  
_𑀯𑀺𑀲𑀸𑀔𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(15.)

940\. _𑀯𑀲𑀢𑁂 𑀢𑀢𑁆𑀣 𑀔𑁂𑀫𑁂 𑀧𑀼𑀭𑁂,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀫𑀳𑀸𑀓𑀼𑀮𑀁;_  
_𑀦𑀭𑀸𑀦𑀁 𑀧𑀯𑀭𑀁 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀚𑀸𑀢𑀺𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀬𑀲𑀁𑁇_  


(16.)

941\. _𑀘𑀢𑀼𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀓𑀸𑀫𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀓𑀸𑀫𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑀸𑀫𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(17.)

942\. _𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀭𑁄𑀘𑀺𑀦𑀻 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(18.)

943\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀭𑀣𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀅𑀝𑁆𑀞𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(19.)

944\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑁄 𑀯𑀺𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(20.)

945\. _𑀯𑀺𑀥𑀼𑀭𑁄 𑀘 𑀲𑀜𑁆𑀚𑀻𑀯𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑀺𑀚𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(21.)

946\. _𑀲𑀸𑀫𑀸 𑀘 𑀘𑀫𑁆𑀧𑀸𑀦𑀸𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀺𑀭𑀻𑀲𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(22.)

947\. _𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀼𑀫𑀦𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀦𑀦𑁆𑀤𑀸 𑀘𑁂𑀯 𑀲𑀼𑀦𑀦𑁆𑀤𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(23.)

948\. _𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀭𑀢𑀦𑀸𑀦𑀺,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀓𑀦𑀓𑀧𑁆𑀧𑀪𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀤𑀲𑀬𑁄𑀚𑀦𑀁𑁇_  


(24.)

949\. _𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(25.)

950\. _𑀥𑀫𑁆𑀫𑀸𑀧𑀡𑀁 𑀧𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀦𑀭𑀦𑀸𑀭𑀻𑀦𑀁 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀦𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀻𑀳𑀦𑀸𑀤𑀁𑀯,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(26.)

951\. _𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀯𑀘𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀸𑀦𑀺 𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(27.)

952\. _𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑁄 𑀚𑀺𑀦𑀯𑀭𑁄,_  
_𑀔𑁂𑀫𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀣𑀽𑀧𑀯𑀭𑁄,_  
_𑀕𑀸𑀯𑀼𑀢𑀁 𑀦𑀪𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁯𑁨𑁮)_
