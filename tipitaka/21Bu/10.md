# 𑁧𑁦 𑀧𑀤𑀼𑀫𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀧𑀤𑀼𑀫𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

502\. _“𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀧𑀤𑀼𑀫𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀅𑀲𑀫𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


(2.)

503\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀅𑀲𑀫𑀁 𑀲𑀻𑀮𑀁,_  
_𑀲𑀫𑀸𑀥𑀺𑀧𑀺 𑀅𑀦𑀦𑁆𑀢𑀓𑁄;_  
_𑀅𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀁 𑀜𑀸𑀡𑀯𑀭𑀁,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀺 𑀅𑀦𑀽𑀧𑀫𑀸𑁇_  


(3.)

504\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀅𑀢𑀼𑀮𑀢𑁂𑀚𑀲𑁆𑀲,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀦𑁂;_  
_𑀅𑀪𑀺𑀲𑀫𑀬𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀫𑀳𑀸𑀢𑀫𑀧𑀯𑀸𑀳𑀦𑀸𑁇_  


(4.)

505\. _𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀫𑀩𑁄𑀥𑀬𑀺;_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁂 𑀥𑀻𑀭𑁄,_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀫𑀩𑁄𑀥𑀬𑀺𑁇_  


(5.)

506\. _𑀬𑀤𑀸 𑀘 𑀧𑀤𑀼𑀫𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀑𑀯𑀤𑀻 𑀲𑀓𑀫𑀢𑁆𑀭𑀚𑀁;_  
_𑀢𑀤𑀸 𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

507\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(7.)

508\. _𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀸𑀭𑀲𑀫𑀬𑁂,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀓𑀣𑀺𑀦𑀘𑀻𑀯𑀭𑁂;_  
_𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀢𑁆𑀣𑀸𑀬,_  
_𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀺𑀩𑁆𑀩𑀺𑀁𑀲𑀼 𑀘𑀻𑀯𑀭𑀁𑁇_  


(8.)

509\. _𑀢𑀤𑀸 𑀢𑁂 𑀯𑀺𑀫𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽,_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸;_  
_𑀢𑀻𑀡𑀺 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀫𑀺𑀁𑀲𑀼 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀸𑁇_  


(9.)

510\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀲𑁄 𑀦𑀭𑀸𑀲𑀪𑁄,_  
_𑀧𑀯𑀦𑁂 𑀯𑀸𑀲𑀁 𑀉𑀧𑀸𑀕𑀫𑀺;_  
_𑀢𑀤𑀸 𑀲𑀫𑀸𑀕𑀫𑁄 𑀆𑀲𑀺,_  
_𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀦𑀁𑁇_  


(10.)

511\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀲𑀻𑀳𑁄 𑀆𑀲𑀺𑀁 𑀫𑀺𑀕𑀸𑀥𑀺𑀪𑀽;_  
_𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑀦𑁆𑀢𑀁,_  
_𑀧𑀯𑀦𑁂 𑀅𑀤𑁆𑀤𑀲𑀁 𑀚𑀺𑀦𑀁𑁇_  


(11.)

512\. _𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀺𑀭𑀲𑀸 𑀧𑀸𑀤𑁂,_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀅𑀪𑀺𑀦𑀸𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀚𑀺𑀦𑀫𑀼𑀧𑀝𑁆𑀞𑀳𑀁𑁇_  


(12.)

513\. _𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀯𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸,_  
_𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑁄;_  
_𑀫𑀦𑀲𑀸 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑁄𑀝𑀺𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑀬𑀺𑁇_  


(13.)

514\. _𑀢𑀤𑀸𑀧𑀺 𑀲𑁄 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀢𑁂𑀲𑀁 𑀫𑀚𑁆𑀛𑁂 𑀯𑀺𑀬𑀸𑀓𑀭𑀺;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(14.)

515\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(15.)

516\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(16.)

517\. _𑀘𑀫𑁆𑀧𑀓𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀅𑀲𑀫𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀅𑀲𑀫𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(17.)

518\. _𑀤𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀦𑀦𑁆𑀤𑀸-𑀯𑀲𑀼-𑀬𑀲𑀼𑀢𑁆𑀢𑀭𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(18.)

519\. _𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀘 𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀉𑀢𑁆𑀢𑀭𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀭𑀫𑁆𑀫𑁄 𑀦𑀸𑀫𑀸𑀲𑀺 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(19.)

520\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀭𑀣𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀅𑀝𑁆𑀞𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(20.)

521\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀧𑀤𑀼𑀫𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀥𑀦𑀜𑁆𑀘𑀼𑀬𑁆𑀬𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(21.)

522\. _𑀲𑀸𑀮𑁄 𑀘 𑀉𑀧𑀲𑀸𑀮𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀯𑀭𑀼𑀡𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(22.)

523\. _𑀭𑀸𑀥𑀸 𑀘𑁂𑀯 𑀲𑀼𑀭𑀸𑀥𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀫𑀳𑀸𑀲𑁄𑀡𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(23.)

524\. _𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑁂𑀯 𑀅𑀲𑀫𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀭𑀼𑀘𑀻 𑀘 𑀦𑀦𑁆𑀤𑀭𑀸𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(24.)

525\. _𑀅𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀭𑀢𑀦𑀁,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀧𑀪𑀸 𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀢𑀻 𑀢𑀲𑁆𑀲,_  
_𑀅𑀲𑀫𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀤𑀺𑀲𑀸𑁇_  


(25.)

526\. _𑀘𑀦𑁆𑀤𑀧𑁆𑀧𑀪𑀸 𑀲𑀽𑀭𑀺𑀬𑀧𑁆𑀧𑀪𑀸,_  
_𑀭𑀢𑀦𑀕𑁆𑀕𑀺𑀫𑀡𑀺𑀧𑁆𑀧𑀪𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀢𑀸 𑀳𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀧𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀧𑀪𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(26.)

527\. _𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(27.)

528\. _𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀫𑀸𑀦𑀲𑁂 𑀲𑀢𑁆𑀢𑁂,_  
_𑀩𑁄𑀥𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁂𑀲𑀢𑁄;_  
_𑀲𑁂𑀲𑀓𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(28.)

529\. _𑀉𑀭𑀕𑁄𑀯 𑀢𑀘𑀁 𑀚𑀺𑀡𑁆𑀡𑀁,_  
_𑀯𑀤𑁆𑀥𑀧𑀢𑁆𑀢𑀁𑀯 𑀧𑀸𑀤𑀧𑁄;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀬𑀣𑀸 𑀲𑀺𑀔𑀻𑁇_  


(29.)

530\. _𑀧𑀤𑀼𑀫𑁄 𑀚𑀺𑀦𑀯𑀭𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄”𑀢𑀺𑁇 (𑁫𑁨𑁦)_
