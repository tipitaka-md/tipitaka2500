# 𑁨𑁩 𑀯𑁂𑀲𑁆𑀲𑀪𑀽𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀯𑁂𑀲𑁆𑀲𑀪𑀽𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

895\. _“𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀡𑁆𑀟𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀅𑀲𑀫𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀽 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(2.)

896\. _𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀯𑀢 𑀭𑀸𑀕𑀕𑁆𑀕𑀺,_  
_𑀢𑀡𑁆𑀳𑀸𑀦𑀁 𑀯𑀺𑀚𑀺𑀢𑀁 𑀢𑀤𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(3.)

897\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀽𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

898\. _𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀘𑀸𑀭𑀺𑀓𑀁 𑀭𑀝𑁆𑀞𑁂,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁂 𑀦𑀭𑀸𑀲𑀪𑁂;_  
_𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

899\. _𑀫𑀳𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀯𑀺𑀦𑁄𑀤𑁂𑀦𑁆𑀢𑁄,_  
_𑀧𑀸𑀝𑀺𑀳𑁂𑀭𑀁 𑀓𑀭𑁄𑀢𑀺 𑀲𑁄;_  
_𑀲𑀫𑀸𑀕𑀢𑀸 𑀦𑀭𑀫𑀭𑀽,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀻 𑀲𑀤𑁂𑀯𑀓𑁂𑁇_  


(6.)

900\. _𑀫𑀳𑀸𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀤𑁂𑀯𑀸 𑀘𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘,_  
_𑀩𑀼𑀚𑁆𑀛𑀭𑁂 𑀲𑀝𑁆𑀞𑀺𑀓𑁄𑀝𑀺𑀬𑁄𑁇_  


(7.)

901\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(8.)

902\. _𑀅𑀲𑀻𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀲𑀢𑁆𑀢𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(9.)

903\. _𑀲𑀝𑁆𑀞𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀚𑀭𑀸𑀤𑀺𑀪𑀬𑀪𑀻𑀢𑀸𑀦𑀁,_  
_𑀑𑀭𑀲𑀸𑀦𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(10.)

904\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀭𑀳𑀁;_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦,_  
_𑀲𑀲𑀁𑀖𑀁 𑀚𑀺𑀦𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(11.)

905\. _𑀢𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀲𑀫𑀲𑁆𑀲,_  
_𑀘𑀓𑁆𑀓𑀁 𑀯𑀢𑁆𑀢𑀺𑀢𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀧𑀡𑀺𑀢𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀫𑀪𑀺𑀭𑁄𑀘𑀬𑀺𑀁𑁇_  


(12.)

906\. _𑀫𑀳𑀸𑀤𑀸𑀦𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑁄;_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀕𑀼𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀚𑀺𑀦𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(13.)

907\. _𑀆𑀘𑀸𑀭𑀕𑀼𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀯𑀢𑁆𑀢𑀲𑀻𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀭𑀫𑀸𑀫𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂𑁇_  


(14.)

908\. _𑀲𑀤𑁆𑀥𑀸𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀦𑁆𑀤𑀸𑀫𑀺 𑀲𑀢𑁆𑀣𑀭𑀁;_  
_𑀧𑀻𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀬𑁆𑀳𑀁,_  
_𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀓𑀸𑀭𑀡𑀸𑁇_  


(15.)

909\. _𑀅𑀦𑀺𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀺;_  
_‘𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(16.)

910\. _𑀅𑀳𑀼 𑀓𑀧𑀺𑀮𑀯𑁆𑀳𑀬𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(17.)

911\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(18.)

912\. _𑀅𑀦𑁄𑀫𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀲𑀼𑀧𑁆𑀧𑀢𑀻𑀢𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀫𑀸𑀢𑀸 𑀬𑀲𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(19.)

913\. _𑀙 𑀘 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀭𑀼𑀘𑀺 𑀲𑀼𑀭𑀼𑀘𑀺 𑀭𑀢𑀺𑀯𑀟𑁆𑀠𑀦𑁄,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(20.)

914\. _𑀅𑀦𑀽𑀦𑀢𑀺𑀁𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑀼𑀘𑀺𑀢𑁆𑀢𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(21.)

915\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀺𑀯𑀺𑀓𑀸𑀬𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀙𑀫𑀸𑀲𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(22.)

916\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀽𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀅𑀭𑀼𑀡𑀸𑀭𑀸𑀫𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(23.)

917\. _𑀲𑁄𑀡𑁄 𑀘 𑀉𑀢𑁆𑀢𑀭𑁄 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(24.)

918\. _𑀭𑀸𑀫𑀸 𑀘𑁂𑀯 𑀲𑀫𑀸𑀮𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀫𑀳𑀸𑀲𑀸𑀮𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(25.)

919\. _𑀲𑁄𑀢𑁆𑀣𑀺𑀓𑁄 𑀘𑁂𑀯 𑀭𑀫𑁆𑀪𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀕𑁄𑀢𑀫𑀻 𑀲𑀺𑀭𑀺𑀫𑀸 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(26.)

920\. _𑀲𑀝𑁆𑀞𑀺𑀭𑀢𑀦𑀫𑀼𑀩𑁆𑀩𑁂𑀥𑁄,_  
_𑀳𑁂𑀫𑀬𑀽𑀧𑀲𑀫𑀽𑀧𑀫𑁄;_  
_𑀓𑀸𑀬𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺 𑀭𑀲𑁆𑀫𑀺,_  
_𑀭𑀢𑁆𑀢𑀺𑀁𑀯 𑀧𑀩𑁆𑀩𑀢𑁂 𑀲𑀺𑀔𑀻𑁇_  


(27.)

921\. _𑀲𑀝𑁆𑀞𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(28.)

922\. _𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀥𑀫𑁆𑀫𑀦𑀸𑀯𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(29.)

923\. _𑀤𑀲𑁆𑀲𑀦𑁂𑀬𑁆𑀬𑀁 𑀲𑀩𑁆𑀩𑀚𑀦𑀁,_  
_𑀯𑀺𑀳𑀸𑀭𑀁 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(30.)

924\. _𑀯𑁂𑀲𑁆𑀲𑀪𑀽 𑀚𑀺𑀦𑀯𑀭𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀔𑁂𑀫𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄”𑀢𑀺𑁇 (𑁯𑁦𑁧)_  


---

925\. 𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀏𑀓𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇
