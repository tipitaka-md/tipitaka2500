# 𑁮 𑀲𑁄𑀪𑀺𑀢𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀲𑁄𑀪𑀺𑀢𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

441\. _“𑀭𑁂𑀯𑀢𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑁄𑀪𑀺𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀅𑀲𑀫𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


(2.)

442\. _𑀲𑁄 𑀚𑀺𑀦𑁄 𑀲𑀓𑀕𑁂𑀳𑀫𑁆𑀳𑀺,_  
_𑀫𑀸𑀦𑀲𑀁 𑀯𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑀬𑀺;_  
_𑀧𑀢𑁆𑀯𑀸𑀦 𑀓𑁂𑀯𑀮𑀁 𑀩𑁄𑀥𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀬𑀺𑁇_  


(3.)

443\. _𑀬𑀸𑀯 𑀳𑁂𑀝𑁆𑀞𑀸 𑀅𑀯𑀻𑀘𑀺𑀢𑁄,_  
_𑀪𑀯𑀕𑁆𑀕𑀸 𑀘𑀸𑀧𑀺 𑀉𑀤𑁆𑀥𑀢𑁄;_  
_𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀏𑀓𑀧𑀭𑀺𑀲𑀸,_  
_𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂𑁇_  


(4.)

444\. _𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀬𑀺;_  
_𑀕𑀡𑀦𑀸𑀬 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

445\. _𑀢𑀢𑁄 𑀧𑀭𑀫𑁆𑀧𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁂,_  
_𑀫𑀭𑀽𑀦𑀜𑁆𑀘 𑀲𑀫𑀸𑀕𑀫𑁂;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

446\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀚𑀬𑀲𑁂𑀦𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀆𑀭𑀸𑀫𑀁 𑀭𑁄𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑀬𑀻 𑀢𑀤𑀸𑁇_  


(7.)

447\. _𑀢𑀲𑁆𑀲 𑀬𑀸𑀕𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀢𑀤𑀸 𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(8.)

448\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀲𑁄𑀪𑀺𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(9.)

449\. _𑀉𑀕𑁆𑀕𑀢𑁄 𑀦𑀸𑀫 𑀲𑁄 𑀭𑀸𑀚𑀸,_  
_𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀫𑁆𑀳𑀺 𑀤𑀸𑀦𑁂 𑀲𑀫𑀸𑀕𑀜𑁆𑀙𑀼𑀁,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑀸 𑀲𑀢𑀓𑁄𑀝𑀺𑀬𑁄𑁇_  


(10.)

450\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀧𑀼𑀭𑀕𑀡𑁄,_  
_𑀤𑁂𑀢𑀺 𑀤𑀸𑀦𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀤𑀸 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(11.)

451\. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀯𑀲𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀬𑀤𑀸 𑀑𑀭𑁄𑀳𑀢𑀻 𑀚𑀺𑀦𑁄;_  
_𑀢𑀤𑀸 𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(12.)

452\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀲𑀼𑀚𑀸𑀢𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀢𑀤𑀸 𑀲𑀲𑀸𑀯𑀓𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀢𑀧𑁆𑀧𑀬𑀺𑀁𑁇_  


(13.)

453\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀲𑁄𑀪𑀺𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(14.)

454\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(15.)

455\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀳𑀝𑁆𑀞𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑁄;_  
_𑀢𑀫𑁂𑀯𑀢𑁆𑀣𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑀸,_  
_𑀉𑀕𑁆𑀕𑀁 𑀥𑀺𑀢𑀺𑀫𑀓𑀸𑀲𑀳𑀁𑁇_  


(16.)

456\. _𑀲𑀼𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀲𑁄𑀪𑀺𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(17.)

457\. _𑀦𑀯𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀓𑀼𑀫𑀼𑀤𑁄 𑀦𑀸𑀍𑀇𑀦𑁄 𑀧𑀤𑀼𑀫𑁄,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(18.)

458\. _𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀫𑀡𑀺𑀮𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀲𑀻𑀳𑁄 𑀦𑀸𑀫𑀸𑀲𑀺 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(19.)

459\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀸𑀲𑀸𑀤𑁂𑀦𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(20.)

460\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑁄𑀪𑀺𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀼𑀬𑁆𑀬𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(21.)

461\. _𑀅𑀲𑀫𑁄 𑀘 𑀲𑀼𑀦𑁂𑀢𑁆𑀢𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀅𑀦𑁄𑀫𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀲𑁄𑀪𑀺𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(22.)

462\. _𑀦𑀓𑀼𑀮𑀸 𑀘 𑀲𑀼𑀚𑀸𑀢𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑁄 𑀘 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀸𑀕𑀫𑀽𑀮𑁂 𑀅𑀩𑀼𑀚𑁆𑀛𑀣𑁇_  


(23.)

463\. _𑀭𑀫𑁆𑀫𑁄 𑀘𑁂𑀯 𑀲𑀼𑀤𑀢𑁆𑀢𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀦𑀓𑀼𑀮𑀸 𑀘𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(24.)

464\. _𑀅𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀭𑀢𑀦𑀁,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀑𑀪𑀸𑀲𑁂𑀢𑀺 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀲𑀢𑀭𑀁𑀲𑀻𑀯 𑀉𑀕𑁆𑀕𑀢𑁄𑁇_  


(25.)

465\. _𑀬𑀣𑀸 𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀁 𑀧𑀯𑀦𑀁,_  
_𑀦𑀸𑀦𑀸𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀥𑀽𑀧𑀺𑀢𑀁;_  
_𑀢𑀣𑁂𑀯 𑀢𑀲𑁆𑀲 𑀧𑀸𑀯𑀘𑀦𑀁,_  
_𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀥𑀽𑀧𑀺𑀢𑀁𑁇_  


(26.)

466\. _𑀬𑀣𑀸𑀧𑀺 𑀲𑀸𑀕𑀭𑁄 𑀦𑀸𑀫,_  
_𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀅𑀢𑀧𑁆𑀧𑀺𑀬𑁄;_  
_𑀢𑀣𑁂𑀯 𑀢𑀲𑁆𑀲 𑀧𑀸𑀯𑀘𑀦𑀁,_  
_𑀲𑀯𑀦𑁂𑀦 𑀅𑀢𑀧𑁆𑀧𑀺𑀬𑀁𑁇_  


(27.)

467\. _𑀦𑀯𑀼𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(28.)

468\. _𑀑𑀯𑀸𑀤𑀁 𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺𑀜𑁆𑀘,_  
_𑀤𑀢𑁆𑀯𑀸𑀦 𑀲𑁂𑀲𑀓𑁂 𑀚𑀦𑁂;_  
_𑀳𑀼𑀢𑀸𑀲𑀦𑁄𑀯 𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(29.)

469\. _𑀲𑁄 𑀘 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀢𑁂𑀧𑀺 𑀲𑀸𑀯𑀓𑀸 𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(30.)

470\. _𑀲𑁄𑀪𑀺𑀢𑁄 𑀯𑀭𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀻𑀳𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄”𑀢𑀺𑁇 (𑁪𑁬𑁨)_
