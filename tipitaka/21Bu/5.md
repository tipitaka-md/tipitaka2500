# 𑁫 𑀫𑀗𑁆𑀕𑀮𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀫𑀗𑁆𑀕𑀮𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

343. _“𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀫𑀗𑁆𑀕𑀮𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀢𑀫𑀁 𑀮𑁄𑀓𑁂 𑀦𑀺𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀥𑀫𑁆𑀫𑁄𑀓𑁆𑀓𑀫𑀪𑀺𑀥𑀸𑀭𑀬𑀺𑁇_  


(2.)

344. _𑀅𑀢𑀼𑀮𑀸𑀲𑀺 𑀧𑀪𑀸 𑀢𑀲𑁆𑀲,_  
_𑀚𑀺𑀦𑁂𑀳𑀜𑁆𑀜𑁂𑀳𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁;_  
_𑀘𑀦𑁆𑀤𑀲𑀽𑀭𑀺𑀬𑀧𑀪𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀻 𑀯𑀺𑀭𑁄𑀘𑀢𑀺𑁇_  


(3.)

345. _𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀧𑀓𑀸𑀲𑁂𑀲𑀺,_  
_𑀘𑀢𑀼𑀭𑁄 𑀲𑀘𑁆𑀘𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑁂 𑀢𑁂 𑀲𑀘𑁆𑀘𑀭𑀲𑀁 𑀧𑀻𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀦𑁄𑀤𑁂𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀢𑀫𑀁𑁇_  


(4.)

346. _𑀧𑀢𑁆𑀯𑀸𑀦 𑀩𑁄𑀥𑀺𑀫𑀢𑀼𑀮𑀁,_  
_𑀧𑀞𑀫𑁂 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

347. _𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁂𑀯𑀪𑀯𑀦𑁂,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑀬𑀺;_  
_𑀢𑀤𑀸 𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

348. _𑀬𑀤𑀸 𑀲𑀼𑀦𑀦𑁆𑀤𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺;_  
_𑀢𑀤𑀸 𑀆𑀳𑀦𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀥𑀫𑁆𑀫𑀪𑁂𑀭𑀺𑀁 𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(7.)

349. _𑀲𑀼𑀦𑀦𑁆𑀤𑀲𑁆𑀲𑀸𑀦𑀼𑀘𑀭𑀸 𑀚𑀦𑀢𑀸,_  
_𑀢𑀤𑀸𑀲𑀼𑀁 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀬𑁄;_  
_𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀢𑁂 𑀦𑀺𑀭𑀯𑀲𑁂𑀲𑀸,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀏𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀸𑁇_  


(8.)

350. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀫𑀗𑁆𑀕𑀮𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(9.)

351. _𑀤𑀼𑀢𑀺𑀬𑁄 𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀦𑀁;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(10.)

352. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀲𑀼𑀭𑀼𑀘𑀻 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽𑁇_  


(11.)

353. _𑀢𑀫𑀳𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀲𑀭𑀡𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀲𑀁𑀖𑀁,_  
_𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑁂𑀦 𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑁂𑀦,_  
_𑀕𑀯𑀧𑀸𑀦𑁂𑀦 𑀢𑀧𑁆𑀧𑀬𑀺𑀁𑁇_  


(12.)

354. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀫𑀗𑁆𑀕𑀮𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(13.)

355. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(14.)

356. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(15.)

357. _𑀢𑀤𑀸 𑀧𑀻𑀢𑀺𑀫𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑀦𑁆𑀢𑁄,_  
_𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀯𑀭𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀤𑀢𑁆𑀯𑀸𑀦 𑀫𑀁 𑀕𑁂𑀳𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(16.)

358. _𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀯𑀺𑀦𑀬𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀦𑀯𑀗𑁆𑀕𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑁄𑀪𑀬𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(17.)

359. _𑀢𑀢𑁆𑀣𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯 𑀪𑀸𑀯𑀦𑀁;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(18.)

360. _𑀉𑀢𑁆𑀢𑀭𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀉𑀢𑁆𑀢𑀭𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀫𑀗𑁆𑀕𑀮𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(19.)

361. _𑀦𑀯𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀬𑀲𑀯𑀸 𑀲𑀼𑀘𑀺𑀫𑀸 𑀲𑀺𑀭𑀻𑀫𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(20.)

362. _𑀲𑀫𑀢𑀺𑀁𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀬𑀲𑀯𑀢𑀻 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀲𑀻𑀯𑀮𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(21.)

363. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀲𑁆𑀲𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀅𑀝𑁆𑀞𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(22.)

364. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑀗𑁆𑀕𑀮𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀯𑀦𑁂 𑀲𑀺𑀭𑀻𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(23.)

365. _𑀲𑀼𑀤𑁂𑀯𑁄 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀧𑀸𑀮𑀺𑀢𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀫𑀗𑁆𑀕𑀮𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(24.)

366. _𑀲𑀻𑀯𑀮𑀸 𑀘 𑀅𑀲𑁄𑀓𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀦𑀸𑀕𑀭𑀼𑀓𑁆𑀔𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(25.)

367. _𑀦𑀦𑁆𑀤𑁄 𑀘𑁂𑀯 𑀯𑀺𑀲𑀸𑀔𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀅𑀦𑀼𑀮𑀸 𑀘𑁂𑀯 𑀲𑀼𑀢𑀦𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(26.)

368. _𑀅𑀝𑁆𑀞𑀸𑀲𑀻𑀢𑀺 𑀭𑀢𑀦𑀸𑀦𑀺,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀢𑀢𑁄 𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀢𑀻 𑀭𑀁𑀲𑀻,_  
_𑀅𑀦𑁂𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀬𑁄𑁇_  


(27.)

369. _𑀦𑀯𑀼𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(28.)

370. _𑀬𑀣𑀸𑀧𑀺 𑀲𑀸𑀕𑀭𑁂 𑀊𑀫𑀻,_  
_𑀦 𑀲𑀓𑁆𑀓𑀸 𑀢𑀸 𑀕𑀡𑁂𑀢𑀼𑀬𑁂;_  
_𑀢𑀣𑁂𑀯 𑀲𑀸𑀯𑀓𑀸 𑀢𑀲𑁆𑀲,_  
_𑀦 𑀲𑀓𑁆𑀓𑀸 𑀢𑁂 𑀕𑀡𑁂𑀢𑀼𑀬𑁂𑁇_  


(29.)

371. _𑀬𑀸𑀯 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀫𑀗𑁆𑀕𑀮𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀦 𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀓𑀺𑀮𑁂𑀲𑀫𑀭𑀡𑀁 𑀢𑀤𑀸𑁇_  


(30.)

372. _𑀥𑀫𑁆𑀫𑁄𑀓𑁆𑀓𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀦𑁆𑀢𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀚𑀮𑀺𑀢𑁆𑀯𑀸 𑀥𑀽𑀫𑀓𑁂𑀢𑀽𑀯,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(31.)

373. _𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀲𑀪𑀸𑀯𑀢𑁆𑀣𑀁,_  
_𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀚𑀮𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀯,_  
_𑀲𑀽𑀭𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑁄 𑀬𑀣𑀸𑁇_  


(32.)

374. _𑀉𑀬𑁆𑀬𑀸𑀦𑁂 𑀯𑀲𑁆𑀲𑀭𑁂 𑀦𑀸𑀫,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺 𑀫𑀗𑁆𑀕𑀮𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀚𑀺𑀦𑀣𑀽𑀧𑁄,_  
_𑀢𑀺𑀁𑀲𑀬𑁄𑀚𑀦𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁩𑁬𑁯)_  


---

375. 𑀫𑀗𑁆𑀕𑀮𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
