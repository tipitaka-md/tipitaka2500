# 𑁭 𑀭𑁂𑀯𑀢𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

* 𑀭𑁂𑀯𑀢𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

411\. _𑀲𑀼𑀫𑀦𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀭𑁂𑀯𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀅𑀦𑀽𑀧𑀫𑁄 𑀅𑀲𑀤𑀺𑀲𑁄,_  
_𑀅𑀢𑀼𑀮𑁄 𑀉𑀢𑁆𑀢𑀫𑁄 𑀚𑀺𑀦𑁄𑁇_  


(2.)

412\. _𑀲𑁄𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀅𑀪𑀺𑀬𑀸𑀘𑀺𑀢𑁄;_  
_𑀔𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁,_  
_𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀁 𑀪𑀯𑀸𑀪𑀯𑁂𑁇_  


(3.)

413\. _𑀢𑀲𑁆𑀲𑀸𑀪𑀺𑀲𑀫𑀬𑀸 𑀢𑀻𑀡𑀺,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂;_  
_𑀕𑀡𑀦𑀸𑀬 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

414\. _𑀬𑀤𑀸 𑀅𑀭𑀺𑀦𑁆𑀤𑀫𑀁 𑀭𑀸𑀚𑀁,_  
_𑀯𑀺𑀦𑁂𑀲𑀺 𑀭𑁂𑀯𑀢𑁄 𑀫𑀼𑀦𑀺;_  
_𑀢𑀤𑀸 𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

415\. _𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸,_  
_𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀁 𑀦𑀭𑀫𑀭𑀽𑀦𑀁,_  
_𑀯𑀺𑀦𑁂𑀲𑀺 𑀉𑀢𑁆𑀢𑀫𑁂 𑀨𑀮𑁂𑁇_  


(6.)

416\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀭𑁂𑀯𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(7.)

417\. _𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀕𑀡𑀦𑀧𑀣𑀁,_  
_𑀧𑀞𑀫𑀁 𑀬𑁂 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(8.)

418\. _𑀬𑁄𑀧𑀺 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀲𑀫𑁄,_  
_𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀓𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑁄;_  
_𑀲𑁄 𑀢𑀤𑀸 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀆𑀲𑀺,_  
_𑀧𑀢𑁆𑀢𑁄 𑀚𑀻𑀯𑀺𑀢𑀲𑀁𑀲𑀬𑀁𑁇_  


(9.)

419\. _𑀢𑀲𑁆𑀲 𑀕𑀺𑀮𑀸𑀦𑀧𑀼𑀘𑁆𑀙𑀸𑀬,_  
_𑀬𑁂 𑀢𑀤𑀸 𑀉𑀧𑀕𑀢𑀸 𑀫𑀼𑀦𑀻;_  
_𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸 𑀅𑀭𑀳𑀦𑁆𑀢𑁄,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(10.)

420\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀅𑀢𑀺𑀤𑁂𑀯𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀭𑁂𑀯𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀭𑀡𑀁 𑀢𑀲𑁆𑀲 𑀕𑀜𑁆𑀙𑀳𑀁𑁇_  


(11.)

421\. _𑀢𑀲𑁆𑀲 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀥𑀺𑀜𑁆𑀘,_  
_𑀧𑀜𑁆𑀜𑀸𑀕𑀼𑀡𑀫𑀦𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀣𑁄𑀫𑀬𑀺𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀣𑀸𑀫𑀁,_  
_𑀉𑀢𑁆𑀢𑀭𑀻𑀬𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(12.)

422\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀭𑁂𑀯𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_“𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(13.)

423\. _‘𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(14.)

424\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(15.)

425\. _𑀢𑀤𑀸𑀧𑀺 𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑀬𑀺𑀁;_  
_𑀆𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀬𑀁 𑀫𑀬𑁆𑀳𑀁 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑀁𑁇_  


(16.)

426\. _𑀦𑀕𑀭𑀁 𑀲𑀼𑀥𑀜𑁆𑀜𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀯𑀺𑀧𑀼𑀮𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀯𑀺𑀧𑀼𑀮𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀭𑁂𑀯𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(17.)

427\. _𑀙 𑀘 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀭𑀢𑀦𑀕𑁆𑀖𑀺,_  
_𑀆𑀯𑁂𑀍𑀑 𑀘 𑀯𑀺𑀪𑀽𑀲𑀺𑀢𑁄;_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(18.)

428\. _𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀘 𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀸 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀯𑀭𑀼𑀡𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(19.)

429\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀭𑀣𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(20.)

430\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀭𑁂𑀯𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀯𑀭𑀼𑀡𑀸𑀭𑀸𑀫𑁂 𑀲𑀺𑀭𑀻𑀖𑀭𑁂𑁇_  


(21.)

431\. _𑀯𑀭𑀼𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑁂𑀯𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑀫𑁆𑀪𑀯𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀭𑁂𑀯𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(22.)

432\. _𑀪𑀤𑁆𑀤𑀸 𑀘𑁂𑀯 𑀲𑀼𑀪𑀤𑁆𑀤𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀦𑀸𑀕𑀫𑀽𑀮𑁂 𑀅𑀩𑀼𑀚𑁆𑀛𑀣𑁇_  


(23.)

433\. _𑀧𑀤𑀼𑀫𑁄 𑀓𑀼𑀜𑁆𑀚𑀭𑁄 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀲𑀺𑀭𑀻𑀫𑀸 𑀘𑁂𑀯 𑀬𑀲𑀯𑀢𑀻,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(24.)

434\. _𑀉𑀘𑁆𑀘𑀢𑁆𑀢𑀦𑁂𑀦 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀲𑀻𑀢𑀺𑀳𑀢𑁆𑀣𑀫𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀑𑀪𑀸𑀲𑁂𑀢𑀺 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀇𑀦𑁆𑀤𑀓𑁂𑀢𑀼𑀯 𑀉𑀕𑁆𑀕𑀢𑁄𑁇_  


(25.)

435\. _𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸,_  
_𑀧𑀪𑀸𑀫𑀸𑀮𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸;_  
_𑀤𑀺𑀯𑀸 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀨𑀭𑀢𑀺 𑀬𑁄𑀚𑀦𑀁𑁇_  


(26.)

436\. _𑀲𑀝𑁆𑀞𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(27.)

437\. _𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀩𑀮𑀁,_  
_𑀅𑀫𑀢𑀁 𑀮𑁄𑀓𑁂 𑀧𑀓𑀸𑀲𑀬𑀁;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄,_  
_𑀬𑀣𑀕𑁆𑀕𑀼𑀧𑀸𑀤𑀸𑀦𑀲𑀗𑁆𑀔𑀬𑀸𑁇_  


(28.)

438\. _𑀲𑁄 𑀘 𑀓𑀸𑀬𑁄 𑀭𑀢𑀦𑀦𑀺𑀪𑁄,_  
_𑀲𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀸𑀤𑀺𑀲𑁄;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_

439\. _𑀭𑁂𑀯𑀢𑁄 𑀬𑀲𑀥𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀫𑀳𑀸𑀧𑀼𑀭𑁂;_

_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄”𑀢𑀺𑁇 (𑁪𑁩𑁨)_
