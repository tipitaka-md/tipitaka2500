# 𑁩 𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

* 𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

272\. _“𑀢𑀤𑀸 𑀢𑁂 𑀪𑁄𑀚𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀲𑀁𑀖𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀼𑀁 𑀲𑀭𑀡𑀁 𑀢𑀲𑁆𑀲,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(2.)

273\. _𑀲𑀭𑀡𑀸𑀕𑀫𑀦𑁂 𑀓𑀜𑁆𑀘𑀺,_  
_𑀦𑀺𑀯𑁂𑀲𑁂𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑁄;_  
_𑀓𑀜𑁆𑀘𑀺 𑀧𑀜𑁆𑀘𑀲𑀼 𑀲𑀻𑀮𑁂𑀲𑀼,_  
_𑀲𑀻𑀮𑁂 𑀤𑀲𑀯𑀺𑀥𑁂 𑀧𑀭𑀁𑁇_  


(3.)

274\. _𑀓𑀲𑁆𑀲𑀘𑀺 𑀤𑁂𑀢𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀁,_  
_𑀘𑀢𑀼𑀭𑁄 𑀨𑀮𑀫𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀲𑀫𑁂 𑀥𑀫𑁆𑀫𑁂,_  
_𑀤𑁂𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇_  


(4.)

275\. _𑀓𑀲𑁆𑀲𑀘𑀺 𑀯𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀅𑀝𑁆𑀞 𑀤𑁂𑀢𑀺 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀓𑀲𑁆𑀲𑀘𑀺 𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄,_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀯𑁂𑀘𑁆𑀙𑀢𑀺𑁇_  


(5.)

276\. _𑀢𑁂𑀦 𑀬𑁄𑀕𑁂𑀦 𑀚𑀦𑀓𑀸𑀬𑀁,_  
_𑀑𑀯𑀤𑀢𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀢𑁂𑀦 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(6.)

277\. _𑀫𑀳𑀸𑀳𑀦𑀼𑀲𑀪𑀓𑁆𑀔𑀦𑁆𑀥𑁄,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀦𑀸𑀫𑀓𑁄;_  
_𑀩𑀳𑀽 𑀚𑀦𑁂 𑀢𑀸𑀭𑀬𑀢𑀺,_  
_𑀧𑀭𑀺𑀫𑁄𑀘𑁂𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁𑁇_  


(7.)

278\. _𑀩𑁄𑀥𑀦𑁂𑀬𑁆𑀬𑀁 𑀚𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀧𑀺 𑀬𑁄𑀚𑀦𑁂;_  
_𑀔𑀡𑁂𑀦 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑁄𑀥𑁂𑀢𑀺 𑀢𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(8.)

279\. _𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀫𑀩𑁄𑀥𑀬𑀺;_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁂 𑀦𑀸𑀣𑁄,_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀫𑀩𑁄𑀥𑀬𑀺𑁇_  


(9.)

280\. _𑀬𑀤𑀸 𑀘 𑀤𑁂𑀯𑀪𑀯𑀦𑀫𑁆𑀳𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑀬𑀺;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(10.)

281\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(11.)

282\. _𑀧𑀼𑀦 𑀦𑀸𑀭𑀤𑀓𑀽𑀝𑀫𑁆𑀳𑀺,_  
_𑀧𑀯𑀺𑀯𑁂𑀓𑀕𑀢𑁂 𑀚𑀺𑀦𑁂;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀯𑀻𑀢𑀫𑀮𑀸,_  
_𑀲𑀫𑀺𑀁𑀲𑀼 𑀲𑀢𑀓𑁄𑀝𑀺𑀬𑁄𑁇_  


(12.)

283\. _𑀬𑀫𑁆𑀳𑀺 𑀓𑀸𑀮𑁂 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀺𑀮𑀼𑀘𑁆𑀘𑀬𑁂;_  
_𑀦𑀯𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀧𑀯𑀸𑀭𑁂𑀲𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(13.)

284\. _𑀤𑀲𑀯𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼;_  
_𑀏𑀓𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀸,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀸𑁇_  


(14.)

285\. _𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁,_  
_𑀇𑀤𑁆𑀥𑀁 𑀨𑀻𑀢𑀁 𑀅𑀳𑀽 𑀢𑀤𑀸;_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀸𑀲𑀦𑀁 𑀲𑀼𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁𑁇_  


(15.)

286\. _𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸;_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀁 𑀮𑁄𑀓𑀯𑀺𑀤𑀼𑀁,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


(16.)

287\. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀚𑀳𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀼𑀲𑀁 𑀪𑀯𑀁;_  
_𑀅𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀸 𑀲𑁂𑀔𑀸,_  
_𑀕𑀭𑀳𑀺𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(17.)

288\. _𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀧𑀸𑀯𑀘𑀦𑀁,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀳𑀺;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑁂𑀳𑀺 𑀯𑀺𑀫𑀮𑁂𑀳𑀺,_  
_𑀉𑀧𑀲𑁄𑀪𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


(18.)

289\. _𑀦𑀕𑀭𑀁 𑀭𑀫𑁆𑀫𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀲𑀼𑀤𑁂𑀯𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀼𑀫𑁂𑀥𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(19.)

290\. _𑀤𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀳𑀁𑀲𑀸 𑀓𑁄𑀜𑁆𑀘𑀸 𑀫𑀬𑀽𑀭𑀸 𑀘,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(20.)

291\. _𑀢𑀻𑀡𑀺 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀧𑀤𑀼𑀫𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀉𑀲𑀪𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(21.)

292\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀤𑀲𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑁂 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(22.)

293\. _𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀩𑀼𑀚𑁆𑀛𑀺 𑀫𑀸𑀦𑀲𑀁 𑀫𑀼𑀦𑀺;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(23.)

294\. _𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀦𑀦𑁆𑀤𑀸𑀭𑀸𑀫𑁂 𑀲𑀺𑀭𑀻𑀖𑀭𑁂;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀲𑀺𑀭𑀻𑀲𑀫𑀽𑀮𑀫𑁆𑀳𑀺,_  
_𑀅𑀓𑀸 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀫𑀤𑁆𑀤𑀦𑀁𑁇_  


(24.)

295\. _𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑁄 𑀘 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑀸𑀕𑀢𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(25.)

296\. _𑀦𑀦𑁆𑀤𑀸 𑀘𑁂𑀯 𑀲𑀼𑀦𑀦𑁆𑀤𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀧𑀺𑀧𑁆𑀨𑀮𑀻𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(26.)

297\. _𑀢𑀧𑀼𑀲𑁆𑀲𑀪𑀮𑁆𑀮𑀺𑀓𑀸 𑀦𑀸𑀫,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀲𑀺𑀭𑀺𑀫𑀸 𑀓𑁄𑀡𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀓𑀸,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(27.)

298\. _𑀅𑀲𑀻𑀢𑀺𑀳𑀢𑁆𑀣𑀫𑀼𑀩𑁆𑀩𑁂𑀥𑁄,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀲𑁄𑀪𑀢𑀺 𑀤𑀻𑀧𑀭𑀼𑀓𑁆𑀔𑁄𑀯,_  
_𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯 𑀨𑀼𑀮𑁆𑀮𑀺𑀢𑁄𑁇_  


(28.)

299\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀯𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(29.)

300\. _𑀚𑁄𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀦𑁆𑀢𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀚𑀮𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀯,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(30.)

301\. _𑀲𑀸 𑀘 𑀇𑀤𑁆𑀥𑀺 𑀲𑁄 𑀘 𑀬𑀲𑁄,_  
_𑀢𑀸𑀦𑀺 𑀘 𑀧𑀸𑀤𑁂𑀲𑀼 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀸𑀦𑀺;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(31.)

302\. _𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑁄 𑀚𑀺𑀦𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀦𑀦𑁆𑀤𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀚𑀺𑀦𑀣𑀽𑀧𑁄,_  
_𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀼𑀩𑁆𑀩𑁂𑀥𑀬𑁄𑀚𑀦𑁄”𑀢𑀺𑁇 (𑁨𑁯𑁯)_
