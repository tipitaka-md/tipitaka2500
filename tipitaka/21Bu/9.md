# 𑁯 𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

* 𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

472\. _“𑀲𑁄𑀪𑀺𑀢𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀢𑁂𑀚𑀲𑁆𑀲𑀻 𑀤𑀼𑀭𑀢𑀺𑀓𑁆𑀓𑀫𑁄𑁇_  


(2.)

473\. _𑀲𑁄 𑀙𑁂𑀢𑁆𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀢𑀬𑁄 𑀪𑀯𑁂;_  
_𑀅𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀕𑀫𑀦𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀤𑁂𑀲𑁂𑀲𑀺 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑁂𑁇_  


(3.)

474\. _𑀲𑀸𑀕𑀭𑁄𑀯 𑀅𑀲𑀗𑁆𑀔𑁄𑀩𑁆𑀪𑁄,_  
_𑀧𑀩𑁆𑀩𑀢𑁄𑀯 𑀤𑀼𑀭𑀸𑀲𑀤𑁄;_  
_𑀆𑀓𑀸𑀲𑁄𑀯 𑀅𑀦𑀦𑁆𑀢𑁄 𑀲𑁄,_  
_𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯 𑀨𑀼𑀮𑁆𑀮𑀺𑀢𑁄𑁇_  


(4.)

475\. _𑀤𑀲𑁆𑀲𑀦𑁂𑀦𑀧𑀺 𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀢𑁄𑀲𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑁄;_  
_𑀩𑁆𑀬𑀸𑀳𑀭𑀦𑁆𑀢𑀁 𑀕𑀺𑀭𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀅𑀫𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(5.)

476\. _𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀢𑀲𑁆𑀲,_  
_𑀇𑀤𑁆𑀥𑁄 𑀨𑀻𑀢𑁄 𑀢𑀤𑀸 𑀅𑀳𑀼;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀲𑀫𑀺𑀁𑀲𑀼,_  
_𑀧𑀞𑀫𑁂 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂𑁇_  


(6.)

477\. _𑀢𑀢𑁄 𑀧𑀭𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑁂,_  
_𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀼𑀝𑁆𑀞𑀺𑀬𑁄;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀬𑁄𑀪𑀺𑀲𑀫𑀺𑀁𑀲𑀼,_  
_𑀤𑀼𑀢𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂𑁇_  


(7.)

478\. _𑀢𑀢𑁄𑀧𑀭𑀜𑁆𑀳𑀺 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂,_  
_𑀢𑀧𑁆𑀧𑀬𑀦𑁆𑀢𑁂 𑀘 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀅𑀝𑁆𑀞𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(8.)

479\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀘 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸𑀦𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(9.)

480\. _𑀅𑀝𑁆𑀞𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀢𑀤𑀸 𑀅𑀳𑀼;_  
_𑀧𑀳𑀻𑀦𑀫𑀤𑀫𑁄𑀳𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(10.)

481\. _𑀲𑀢𑁆𑀢𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀅𑀦𑀗𑁆𑀕𑀡𑀸𑀦𑀁 𑀯𑀺𑀭𑀚𑀸𑀦𑀁,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(11.)

482\. _𑀙𑀦𑁆𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸𑀦𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀁𑁇_  


(12.)

483\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀬𑀓𑁆𑀔𑁄 𑀆𑀲𑀺𑀁 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄;_  
_𑀦𑁂𑀓𑀸𑀦𑀁 𑀬𑀓𑁆𑀔𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀫𑁆𑀳𑀺 𑀇𑀲𑁆𑀲𑀭𑁄𑁇_  


(13.)

484\. _𑀢𑀤𑀸𑀧𑀺 𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀯𑀭𑀁,_  
_𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀳𑁂𑀲𑀺𑀦𑀁;_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀢𑀧𑁆𑀧𑁂𑀲𑀺𑀁,_  
_𑀲𑀲𑀁𑀖𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(14.)

485\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀢𑀤𑀸 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀦𑀬𑀦𑁄 𑀫𑀼𑀦𑀺;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(15.)

486\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(16.)

487\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀳𑀝𑁆𑀞𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑁄;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(17.)

488\. _𑀦𑀕𑀭𑀁 𑀘𑀦𑁆𑀤𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀬𑀲𑀯𑀸 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀫𑀸𑀢𑀸 𑀬𑀲𑁄𑀥𑀭𑀸 𑀦𑀸𑀫,_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(18.)

489\. _𑀤𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀲𑀺𑀭𑀻 𑀉𑀧𑀲𑀺𑀭𑀻 𑀯𑀟𑁆𑀠𑁄,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(19.)

490\. _𑀢𑁂𑀯𑀻𑀲𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑀺𑀭𑀺𑀫𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀉𑀧𑀯𑀸𑀡𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(20.)

491\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀺𑀯𑀺𑀓𑀸𑀬𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀤𑀲𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(21.)

492\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀉𑀬𑁆𑀬𑀸𑀦𑁂 𑀲𑁄 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁂𑁇_  


(22.)

493\. _𑀦𑀺𑀲𑀪𑁄 𑀘 𑀅𑀦𑁄𑀫𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀯𑀭𑀼𑀡𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(23.)

494\. _𑀲𑀼𑀦𑁆𑀤𑀭𑀻 𑀘 𑀲𑀼𑀫𑀦𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀅𑀚𑁆𑀚𑀼𑀦𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(24.)

495\. _𑀦𑀦𑁆𑀤𑀺𑀯𑀟𑁆𑀠𑁄 𑀲𑀺𑀭𑀺𑀯𑀟𑁆𑀠𑁄,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀉𑀧𑁆𑀧𑀮𑀸 𑀘𑁂𑀯 𑀧𑀤𑀼𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(25.)

496\. _𑀅𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀭𑀢𑀦𑀁,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀧𑀪𑀸 𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀢𑀻 𑀢𑀲𑁆𑀲,_  
_𑀲𑀢𑀭𑀁𑀲𑀻𑀯 𑀉𑀕𑁆𑀕𑀢𑁄𑁇_  


(26.)

497\. _𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(27.)

498\. _𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀧𑀸𑀯𑀘𑀦𑀁,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀳𑀺;_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑁂𑀳𑀺 𑀯𑀺𑀫𑀮𑁂𑀳𑀺,_  
_𑀲𑁄𑀪𑀺𑀢𑁆𑀣 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(28.)

499\. _𑀲𑁄 𑀘 𑀲𑀢𑁆𑀣𑀸 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀬𑀼𑀕𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀅𑀢𑀼𑀮𑀺𑀬𑀸𑀦𑀺;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(29.)

500\. _𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀚𑀺𑀦𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀚𑀺𑀦𑀣𑀽𑀧𑁄,_  
_𑀉𑀩𑁆𑀩𑁂𑀥𑁄 𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀻”𑀢𑀺𑁇 (𑁪𑁯𑁧)_
