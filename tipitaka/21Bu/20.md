# 𑁨𑁦 𑀨𑀼𑀲𑁆𑀲𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀨𑀼𑀲𑁆𑀲𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

802\. _“𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀡𑁆𑀟𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀅𑀳𑀼 𑀲𑀢𑁆𑀣𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀅𑀦𑀽𑀧𑀫𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀨𑀼𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄𑁇_  


(2.)

803\. _𑀲𑁄𑀧𑀺 𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀚𑀝𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀚𑀝𑀁;_  
_𑀲𑀤𑁂𑀯𑀓𑀁 𑀢𑀧𑁆𑀧𑀬𑀦𑁆𑀢𑁄,_  
_𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀺 𑀅𑀫𑀢𑀫𑁆𑀩𑀼𑀦𑀸𑁇_  


(3.)

804\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀨𑀼𑀲𑁆𑀲𑁂 𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑀮𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

805\. _𑀦𑀯𑀼𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼;_  
_𑀅𑀲𑀻𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

806\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀨𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀧𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(6.)

807\. _𑀲𑀝𑁆𑀞𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(7.)

808\. _𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀯𑁄𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀦𑀁𑁇_  


(8.)

809\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀙𑀟𑁆𑀟𑀬𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(9.)

810\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀨𑀼𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_‘𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(10.)

811\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(11.)

812\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(12.)

813\. _𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀯𑀺𑀦𑀬𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀦𑀯𑀗𑁆𑀕𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑁄𑀪𑀬𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(13.)

814\. _𑀢𑀢𑁆𑀣𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯 𑀪𑀸𑀯𑀦𑀁;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀫𑀕𑀜𑁆𑀙𑀳𑀁𑁇_  


(14.)

815\. _𑀓𑀸𑀲𑀺𑀓𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀚𑀬𑀲𑁂𑀦𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀺𑀭𑀺𑀫𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀨𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀧𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(15.)

816\. _𑀦𑀯𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀕𑀭𑀼𑀍𑀅𑀧𑀓𑁆𑀔 𑀳𑀁𑀲 𑀲𑀼𑀯𑀡𑁆𑀡𑀪𑀸𑀭𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(16.)

817\. _𑀢𑀺𑀁𑀲𑁃𑀢𑁆𑀣𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀅𑀦𑀽𑀧𑀫𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(17.)

818\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀙𑀫𑀸𑀲𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(18.)

819\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀨𑀼𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(19.)

820\. _𑀲𑀼𑀭𑀓𑁆𑀔𑀺𑀢𑁄 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑁄,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑀪𑀺𑀬𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀨𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀧𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(20.)

821\. _𑀘𑀸𑀮𑀸 𑀘 𑀉𑀧𑀘𑀸𑀮𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀆𑀫𑀡𑁆𑀟𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(21.)

822\. _𑀥𑀦𑀜𑁆𑀘𑀬𑁄 𑀯𑀺𑀲𑀸𑀔𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀧𑀤𑀼𑀫𑀸 𑀘𑁂𑀯 𑀦𑀸𑀕𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(22.)

823\. _𑀅𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀭𑀢𑀦𑀁,_  
_𑀲𑁄𑀧𑀺 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀼𑀦𑀺;_  
_𑀲𑁄𑀪𑀢𑁂 𑀲𑀢𑀭𑀁𑀲𑀻𑀯,_  
_𑀉𑀍𑀉𑀭𑀸𑀚𑀸𑀯 𑀧𑀽𑀭𑀺𑀢𑁄𑁇_  


(23.)

824\. _𑀦𑀯𑀼𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(24.)

825\. _𑀑𑀯𑀤𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀲𑀢𑁆𑀢𑁂,_  
_𑀲𑀦𑁆𑀢𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀚𑀦𑁂;_  
_𑀲𑁄𑀧𑀺 𑀲𑀢𑁆𑀣𑀸 𑀅𑀢𑀼𑀮𑀬𑀲𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(25.)

826\. _𑀨𑀼𑀲𑁆𑀲𑁄 𑀚𑀺𑀦𑀯𑀭𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑁂𑀦𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄”𑀢𑀺𑁇 (𑁮𑁦𑁬)_  


---

827\. 𑀨𑀼𑀲𑁆𑀲𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀫𑁄𑁇
