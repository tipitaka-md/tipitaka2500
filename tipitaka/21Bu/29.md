# 𑁨𑁯 𑀥𑀸𑀢𑀼𑀪𑀸𑀚𑀦𑀻𑀬𑀓𑀣𑀸

* 𑀥𑀸𑀢𑀼𑀪𑀸𑀚𑀦𑀻𑀬𑀓𑀣𑀸

(1.)

1082\. _𑀫𑀳𑀸𑀕𑁄𑀢𑀫𑁄 𑀚𑀺𑀦𑀯𑀭𑁄,_  
_𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄𑁇_  


(2.)

1083\. _𑀏𑀓𑁄 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼𑀲𑁆𑀲,_  
_𑀏𑀓𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀸 𑀧𑀼𑀭𑁂;_  
_𑀏𑀓𑁄 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀏𑀓𑁄 𑀘 𑀅𑀮𑁆𑀮𑀓𑀧𑁆𑀧𑀓𑁂𑁇_  


(3.)

1084\. _𑀏𑀓𑁄 𑀘 𑀭𑀸𑀫𑀕𑀸𑀫𑀫𑁆𑀳𑀺,_  
_𑀏𑀓𑁄 𑀘 𑀯𑁂𑀞𑀤𑀻𑀧𑀓𑁂;_  
_𑀏𑀓𑁄 𑀧𑀸𑀯𑁂𑀬𑁆𑀬𑀓𑁂 𑀫𑀮𑁆𑀮𑁂,_  
_𑀏𑀓𑁄 𑀘 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑁂𑁇_  


(4.)

1085\. _𑀓𑀼𑀫𑁆𑀪𑀲𑁆𑀲 𑀣𑀽𑀧𑀁 𑀓𑀸𑀭𑁂𑀲𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀤𑁄𑀡𑀲𑀯𑁆𑀳𑀬𑁄;_  
_𑀅𑀗𑁆𑀕𑀸𑀭𑀣𑀽𑀧𑀁 𑀓𑀸𑀭𑁂𑀲𑀼𑀁,_  
_𑀫𑁄𑀭𑀺𑀬𑀸 𑀢𑀼𑀝𑁆𑀞𑀫𑀸𑀦𑀲𑀸𑁇_  


(5.)

1086\. _𑀅𑀝𑁆𑀞 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸 𑀣𑀽𑀧𑀸,_  
_𑀦𑀯𑀫𑁄 𑀓𑀼𑀫𑁆𑀪𑀘𑁂𑀢𑀺𑀬𑁄;_  
_𑀅𑀗𑁆𑀕𑀸𑀭𑀣𑀽𑀧𑁄 𑀤𑀲𑀫𑁄,_  
_𑀢𑀤𑀸𑀬𑁂𑀯 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇_  


(6.)

1087\. _𑀉𑀡𑁆𑀳𑀻𑀲𑀁 𑀘𑀢𑀲𑁆𑀲𑁄 𑀤𑀸𑀞𑀸,_  
_𑀅𑀓𑁆𑀔𑀓𑀸 𑀤𑁆𑀯𑁂 𑀘 𑀥𑀸𑀢𑀼𑀬𑁄;_  
_𑀅𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀸 𑀇𑀫𑀸 𑀲𑀢𑁆𑀢,_  
_𑀲𑁂𑀲𑀸 𑀪𑀺𑀦𑁆𑀦𑀸𑀯 𑀥𑀸𑀢𑀼𑀬𑁄𑁇_  


(7.)

1088\. _𑀫𑀳𑀦𑁆𑀢𑀸 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀸 𑀘,_  
_𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀪𑀺𑀦𑁆𑀦𑀢𑀡𑁆𑀟𑀼𑀮𑀸;_  
_𑀔𑀼𑀤𑁆𑀤𑀓𑀸 𑀲𑀸𑀲𑀧𑀫𑀢𑁆𑀢𑀸 𑀘,_  
_𑀦𑀸𑀦𑀸𑀯𑀡𑁆𑀡𑀸 𑀘 𑀥𑀸𑀢𑀼𑀬𑁄𑁇_  


(8.)

1089\. _𑀫𑀳𑀦𑁆𑀢𑀸 𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀫𑀼𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀘 𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀔𑀼𑀤𑁆𑀤𑀓𑀸 𑀫𑀓𑀼𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀲𑁄𑀍𑀅𑀲𑀤𑁄𑀡𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑁇_  


(9.)

1090\. _𑀫𑀳𑀦𑁆𑀢𑀸 𑀧𑀜𑁆𑀘 𑀦𑀸𑀍𑀇𑀬𑁄,_  
_𑀦𑀸𑀍𑀇𑀬𑁄 𑀧𑀜𑁆𑀘 𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀔𑀼𑀤𑁆𑀤𑀓𑀸 𑀙 𑀦𑀸𑀍𑀈 𑀘𑁂𑀯,_  
_𑀏𑀢𑀸 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀥𑀸𑀢𑀼𑀬𑁄𑁇_  


(10.)

1091\. _𑀉𑀡𑁆𑀳𑀻𑀲𑀁 𑀲𑀻𑀳𑀍𑀏 𑀤𑀻𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀘 𑀯𑀸𑀫𑀓𑀁;_  
_𑀲𑀻𑀳𑀍𑀏 𑀤𑀓𑁆𑀔𑀺𑀡𑀓𑁆𑀔𑀜𑁆𑀘,_  
_𑀲𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇_  


(11.)

1092\. _𑀏𑀓𑀸 𑀤𑀸𑀞𑀸 𑀢𑀺𑀤𑀲𑀧𑀼𑀭𑁂,_  
_𑀏𑀓𑀸 𑀦𑀸𑀕𑀧𑀼𑀭𑁂 𑀅𑀳𑀼;_  
_𑀏𑀓𑀸 𑀕𑀦𑁆𑀥𑀸𑀭𑀯𑀺𑀲𑀬𑁂,_  
_𑀏𑀓𑀸 𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀸𑀚𑀺𑀦𑁄𑁇_  


(12.)

1093\. _𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀲𑀫𑀸 𑀤𑀦𑁆𑀢𑀸,_  
_𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀘 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀤𑁂𑀯𑀸 𑀳𑀭𑀺𑀁𑀲𑀼 𑀏𑀓𑁂𑀓𑀁,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀍𑀅𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑁇_  


(13.)

1094\. _𑀯𑀚𑀺𑀭𑀸𑀬𑀁 𑀪𑀕𑀯𑀢𑁄,_  
_𑀧𑀢𑁆𑀢𑁄 𑀤𑀡𑁆𑀟𑀜𑁆𑀘 𑀘𑀻𑀯𑀭𑀁;_  
_𑀦𑀺𑀯𑀸𑀲𑀦𑀁 𑀓𑀼𑀮𑀖𑀭𑁂,_  
_𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀭𑀡𑀁 𑀓𑀧𑀺𑀮𑀯𑁆𑀳𑀬𑁂𑁇_  


(14.)

1095\. _𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀧𑀼𑀭𑀫𑁆𑀳𑀺,_  
_𑀓𑀭𑀡𑀁 𑀓𑀸𑀬𑀩𑀦𑁆𑀥𑀦𑀁;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀼𑀤𑀓𑀲𑀸𑀝𑀺𑀬𑀁,_  
_𑀉𑀡𑁆𑀡𑀮𑁄𑀫𑀜𑁆𑀘 𑀓𑁄𑀲𑀮𑁂𑁇_  


(15.)

1096\. _𑀓𑀸𑀲𑀸𑀯𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀘,_  
_𑀯𑁂𑀞𑀦𑀁 𑀢𑀺𑀤𑀲𑁂 𑀧𑀼𑀭𑁂;_  
_𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀅𑀯𑀦𑁆𑀢𑀻𑀲𑀼,_  
_𑀭𑀝𑁆𑀞𑁂 𑀅𑀢𑁆𑀣𑀭𑀡𑀁 𑀢𑀤𑀸𑁇_  


(16.)

1097\. _𑀅𑀭𑀡𑀻 𑀘 𑀫𑀺𑀣𑀺𑀮𑀸𑀬𑀁,_  
_𑀯𑀺𑀤𑁂𑀳𑁂 𑀧𑀭𑀺𑀲𑀸𑀯𑀦𑀁;_  
_𑀯𑀸𑀲𑀺 𑀲𑀽𑀘𑀺𑀖𑀭𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀇𑀦𑁆𑀤𑀧𑀢𑁆𑀣𑀧𑀼𑀭𑁂 𑀢𑀤𑀸𑁇_  


(17.)

1098\. _𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀅𑀯𑀲𑁂𑀲𑀸,_  
_𑀚𑀦𑀧𑀤𑁂 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑁂;_  
_𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀫𑀼𑀦𑀺𑀦𑀸,_  
_𑀅𑀓𑀁𑀲𑀼 𑀫𑀦𑀼𑀚𑀸 𑀢𑀤𑀸𑁇_  


(18.)

1099\. _𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀧𑀸𑀡𑀻𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬,_  
_𑀅𑀳𑀼 𑀧𑁄𑀭𑀸𑀡𑀺𑀓𑀁 𑀢𑀤𑀸𑀢𑀺𑁇 (𑁧𑁦𑁭𑁦)_
