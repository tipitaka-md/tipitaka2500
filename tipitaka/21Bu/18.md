# 𑁧𑁮 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

* 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

748\. _“𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀦𑀺𑀳𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀫𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀲𑀽𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄 𑀬𑀣𑀸𑁇_  


(2.)

749\. _𑀲𑁄𑀧𑀺 𑀧𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁,_  
_𑀲𑀦𑁆𑀢𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀲𑀤𑁂𑀯𑀓𑀁;_  
_𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀺 𑀥𑀫𑁆𑀫𑀫𑁂𑀖𑁂𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀲𑀤𑁂𑀯𑀓𑀁𑁇_  


(3.)

750\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀅𑀢𑀼𑀮𑀢𑁂𑀚𑀲𑁆𑀲,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀸 𑀢𑀬𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

751\. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀪𑀻𑀫𑀭𑀣𑁂,_  
_𑀬𑀤𑀸 𑀆𑀳𑀦𑀺 𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀁;_  
_𑀢𑀤𑀸 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

752\. _𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀯𑁂𑀪𑀸𑀭𑁂 𑀲𑁄 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀤𑀸 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀻𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

753\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀢𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(7.)

754\. _𑀓𑁄𑀝𑀺𑀲𑀢𑀸𑀦𑀁 𑀦𑀯𑀼𑀢𑀻𑀦𑀁,_  
_𑀅𑀲𑀻𑀢𑀺𑀬𑀸𑀧𑀺 𑀘 𑀓𑁄𑀝𑀺𑀦𑀁;_  
_𑀏𑀢𑁂 𑀆𑀲𑀼𑀁 𑀢𑀬𑁄 𑀞𑀸𑀦𑀸,_  
_𑀯𑀺𑀫𑀮𑀸𑀦𑀁 𑀲𑀫𑀸𑀕𑀫𑁂𑁇_  


(8.)

755\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀫𑀗𑁆𑀕𑀮𑁄 𑀦𑀸𑀫 𑀢𑀸𑀧𑀲𑁄;_  
_𑀉𑀕𑁆𑀕𑀢𑁂𑀚𑁄 𑀤𑀼𑀧𑁆𑀧𑀲𑀳𑁄,_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


(9.)

756\. _𑀚𑀫𑁆𑀩𑀼𑀢𑁄 𑀨𑀮𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑁇_  


(10.)

757\. _‘𑀧𑀲𑁆𑀲𑀣 𑀇𑀫𑀁 𑀢𑀸𑀧𑀲𑀁,_  
_𑀚𑀝𑀺𑀮𑀁 𑀉𑀕𑁆𑀕𑀢𑀸𑀧𑀦𑀁;_  
_𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(11.)

758\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(12.)

759\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(13.)

760\. _𑀯𑁂𑀪𑀸𑀭𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀉𑀤𑁂𑀦𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀼𑀨𑀲𑁆𑀲𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(14.)

761\. _𑀤𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀓𑁄𑀓𑀸𑀲𑀼𑀧𑁆𑀧𑀮𑀓𑁄𑀓𑀦𑀤𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(15.)

762\. _𑀢𑀺𑀲𑁄𑀍𑀅𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀅𑀦𑀽𑀧𑀫𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(16.)

763\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀺𑀯𑀺𑀓𑀸𑀬𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀤𑀲𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(17.)

764\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(18.)

765\. _𑀲𑀫𑁆𑀩𑀮𑁄 𑀘 𑀲𑀼𑀫𑀺𑀢𑁆𑀢𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀭𑁂𑀯𑀢𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(19.)

766\. _𑀲𑀻𑀯𑀮𑀸 𑀘 𑀲𑀼𑀭𑀸𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀓𑀡𑀺𑀓𑀸𑀭𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(20.)

767\. _𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀭𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀲𑀼𑀭𑀫𑁆𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(21.)

768\. _𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀝𑁆𑀞𑀺𑀭𑀢𑀦𑀁,_  
_𑀅𑀳𑁄𑀲𑀺 𑀦𑀪𑀫𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀻 𑀯𑀺𑀭𑁄𑀘𑀢𑀺𑁇_  


(22.)

769\. _𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀅𑀢𑀼𑀮𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀮𑁄𑀓𑁂 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(23.)

770\. _𑀯𑀺𑀧𑀼𑀮𑀁 𑀧𑀪𑀁 𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀼𑀧𑁆𑀨𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀸𑀯𑀓𑁂;_  
_𑀯𑀺𑀮𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀬𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(24.)

771\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀫𑀼𑀦𑀺𑀯𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀦𑁄𑀫𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀣𑀽𑀧𑀯𑀭𑁄,_  
_𑀘𑀢𑀼𑀬𑁄𑀚𑀦𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁭𑁫𑁩)_  


---

772\. 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀲𑁄𑀍𑀅𑀲𑀫𑁄𑁇
