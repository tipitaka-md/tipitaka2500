# 𑁧𑁧 𑀦𑀸𑀭𑀤𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

* 𑀦𑀸𑀭𑀤𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

532\. _“𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀦𑀸𑀭𑀤𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀅𑀲𑀫𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


(2.)

533\. _𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲,_  
_𑀚𑁂𑀝𑁆𑀞𑁄 𑀤𑀬𑀺𑀢𑀑𑀭𑀲𑁄;_  
_𑀆𑀫𑀼𑀓𑁆𑀓𑀫𑀸𑀮𑀸𑀪𑀭𑀡𑁄,_  
_𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇_  


(3.)

534\. _𑀢𑀢𑁆𑀣𑀸𑀲𑀺 𑀭𑀼𑀓𑁆𑀔𑁄 𑀬𑀲𑀯𑀺𑀧𑀼𑀮𑁄,_  
_𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀩𑁆𑀭𑀳𑀸 𑀲𑀼𑀘𑀺;_  
_𑀢𑀫𑀚𑁆𑀛𑀧𑀢𑁆𑀯𑀸 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑀺,_  
_𑀫𑀳𑀸𑀲𑁄𑀡𑀲𑁆𑀲 𑀳𑁂𑀝𑁆𑀞𑀢𑁄𑁇_  


(4.)

535\. _𑀢𑀢𑁆𑀣 𑀜𑀸𑀡𑀯𑀭𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺,_  
_𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀚𑀺𑀭𑀽𑀧𑀫𑀁;_  
_𑀢𑁂𑀦 𑀯𑀺𑀘𑀺𑀦𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂,_  
_𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀫𑀯𑀓𑀼𑀚𑁆𑀚𑀓𑀁𑁇_  


(5.)

536\. _𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀺,_  
_𑀅𑀲𑁂𑀲𑀫𑀪𑀺𑀯𑀸𑀳𑀬𑀺;_  
_𑀧𑀸𑀧𑀼𑀡𑀻 𑀓𑁂𑀯𑀮𑀁 𑀩𑁄𑀥𑀺𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀜𑀸𑀡𑁂 𑀘 𑀘𑀼𑀤𑁆𑀤𑀲𑁇_  


(6.)

537\. _𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀬𑀺;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(7.)

538\. _𑀫𑀳𑀸𑀤𑁄𑀡𑀁 𑀦𑀸𑀕𑀭𑀸𑀚𑀁,_  
_𑀯𑀺𑀦𑀬𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀧𑀸𑀝𑀺𑀳𑁂𑀭𑀁 𑀢𑀤𑀸𑀓𑀸𑀲𑀺,_  
_𑀤𑀲𑁆𑀲𑀬𑀦𑁆𑀢𑁄 𑀲𑀤𑁂𑀯𑀓𑁂𑁇_  


(8.)

539\. _𑀢𑀤𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀫𑁆𑀳𑀺 𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀓𑀸𑀲𑀦𑁂;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀭𑀺𑀁𑀲𑀼 𑀲𑀩𑁆𑀩𑀲𑀁𑀲𑀬𑀁𑁇_  


(9.)

540\. _𑀬𑀫𑁆𑀳𑀺 𑀓𑀸𑀮𑁂 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀑𑀯𑀤𑀻 𑀲𑀓𑀫𑀢𑁆𑀭𑀚𑀁;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(10.)

541\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(11.)

542\. _𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑀼𑀤𑁆𑀥𑀕𑀼𑀡𑀁,_  
_𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀓𑀸𑀲𑀬𑀺;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀫𑀺𑀁𑀲𑀼 𑀯𑀺𑀫𑀮𑀸 𑀢𑀤𑀸𑁇_  


(12.)

543\. _𑀬𑀤𑀸 𑀯𑁂𑀭𑁄𑀘𑀦𑁄 𑀦𑀸𑀕𑁄,_  
_𑀤𑀸𑀦𑀁 𑀤𑀤𑀸𑀢𑀺 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀢𑀤𑀸 𑀲𑀫𑀺𑀁𑀲𑀼 𑀚𑀺𑀦𑀧𑀼𑀢𑁆𑀢𑀸,_  
_𑀅𑀲𑀻𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀬𑁄𑁇_  


(13.)

544\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀚𑀝𑀺𑀮𑁄 𑀉𑀕𑁆𑀕𑀢𑀸𑀧𑀦𑁄;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀘𑀭𑁄 𑀆𑀲𑀺𑀁,_  
_𑀧𑀜𑁆𑀘𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀧𑀸𑀭𑀕𑀽𑁇_  


(14.)

545\. _𑀢𑀤𑀸𑀧𑀸𑀳𑀁 𑀅𑀲𑀫𑀲𑀫𑀁,_  
_𑀲𑀲𑀁𑀖𑀁 𑀲𑀧𑀭𑀺𑀚𑁆𑀚𑀦𑀁;_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀘𑀦𑁆𑀤𑀦𑁂𑀦𑀸𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(15.)

546\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀢𑀤𑀸 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀦𑀸𑀭𑀤𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(16.)

547\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(17.)

548\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀳𑀸𑀲𑁂𑀢𑁆𑀯 𑀫𑀸𑀦𑀲𑀁;_  
_𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀁 𑀯𑀢𑀁 𑀉𑀕𑁆𑀕𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(18.)

549\. _𑀦𑀕𑀭𑀁 𑀥𑀜𑁆𑀜𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀲𑀼𑀤𑁂𑀯𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀅𑀦𑁄𑀫𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(19.)

550\. _𑀦𑀯𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀚𑀺𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀪𑀺𑀭𑀸𑀫𑁄,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(20.)

551\. _𑀢𑀺𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀯𑀺𑀚𑀺𑀢𑀲𑁂𑀦𑀸 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀦𑀦𑁆𑀤𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(21.)

552\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀤𑀲𑀸 𑀕𑀫𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(22.)

553\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀦𑀸𑀭𑀤𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀥𑀦𑀜𑁆𑀘𑀼𑀬𑁆𑀬𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(23.)

554\. _𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑁄 𑀚𑀺𑀢𑀫𑀺𑀢𑁆𑀢𑁄,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(24.)

555\. _𑀉𑀢𑁆𑀢𑀭𑀸 𑀨𑀕𑁆𑀕𑀼𑀦𑀻 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀫𑀳𑀸𑀲𑁄𑀡𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(25.)

556\. _𑀉𑀕𑁆𑀕𑀭𑀺𑀦𑁆𑀤𑁄 𑀯𑀲𑀪𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀇𑀦𑁆𑀤𑀸𑀯𑀭𑀻 𑀘 𑀯𑀡𑁆𑀟𑀻 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(26.)

557\. _𑀅𑀝𑁆𑀞𑀸𑀲𑀻𑀢𑀺𑀭𑀢𑀦𑀸𑀦𑀺,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀻 𑀯𑀺𑀭𑁄𑀘𑀢𑀺𑁇_  


(27.)

558\. _𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀫𑀧𑁆𑀧𑀪𑀸 𑀓𑀸𑀬𑀸,_  
_𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁;_  
_𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀁 𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀬𑁄𑀚𑀦𑀁 𑀨𑀭𑀢𑁂 𑀲𑀤𑀸𑁇_  


(28.)

559\. _𑀦 𑀓𑁂𑀘𑀺 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑁂 𑀚𑀦𑀸;_  
_𑀉𑀓𑁆𑀓𑀸𑀧𑀤𑀻𑀧𑁂 𑀉𑀚𑁆𑀚𑀸𑀮𑁂𑀦𑁆𑀢𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀭𑀁𑀲𑀻𑀳𑀺 𑀑𑀢𑁆𑀣𑀝𑀸𑁇_  


(29.)

560\. _𑀦𑀯𑀼𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(30.)

561\. _𑀬𑀣𑀸 𑀉𑀍𑀊𑀳𑀺 𑀕𑀕𑀦𑀁,_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑁄𑀪𑀢𑀺;_  
_𑀢𑀣𑁂𑀯 𑀲𑀸𑀲𑀦𑀁 𑀢𑀲𑁆𑀲,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑁄𑀪𑀢𑀺𑁇_  


(31.)

562\. _𑀲𑀁𑀲𑀸𑀭𑀲𑁄𑀢𑀁 𑀢𑀭𑀡𑀸𑀬,_  
_𑀲𑁂𑀲𑀓𑁂 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀓𑁂;_  
_𑀥𑀫𑁆𑀫𑀲𑁂𑀢𑀼𑀁 𑀤𑀍𑀳𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀦𑀭𑀸𑀲𑀪𑁄𑁇_  


(32.)

563\. _𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀫𑀲𑀫𑁄,_  
_𑀢𑁂𑀧𑀺 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀅𑀢𑀼𑀮𑀢𑁂𑀚𑀸;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(33.)

564\. _𑀦𑀸𑀭𑀤𑁄 𑀚𑀺𑀦𑀯𑀲𑀪𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁂 𑀧𑀼𑀭𑁂;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀣𑀽𑀧𑀯𑀭𑁄,_  
_𑀘𑀢𑀼𑀬𑁄𑀚𑀦𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁫𑁫𑁩)_  


---

565\. 𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀦𑀯𑀫𑁄𑁇
