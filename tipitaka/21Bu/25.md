# 𑁨𑁫 𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

* 𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

954\. _“𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄𑁇_  


(2.)

955\. _𑀤𑀲 𑀥𑀫𑁆𑀫𑁂 𑀧𑀽𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀦𑁆𑀢𑀸𑀭𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀺;_  
_𑀧𑀯𑀸𑀳𑀺𑀬 𑀫𑀮𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(3.)

956\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀦𑀸𑀬𑀓𑁂;_  
_𑀢𑀺𑀁𑀲𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

957\. _𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂 𑀘,_  
_𑀧𑀭𑀯𑀸𑀤𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑁂;_  
_𑀯𑀻𑀲𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

958\. _𑀢𑀢𑁄 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀚𑀺𑀦𑁄 𑀤𑁂𑀯𑀧𑀼𑀭𑀁 𑀕𑀢𑁄;_  
_𑀯𑀲𑀢𑁂 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀺𑀮𑀸𑀬 𑀧𑀡𑁆𑀟𑀼𑀓𑀫𑁆𑀩𑀮𑁂𑁇_  


(6.)

959\. _𑀧𑀓𑀭𑀡𑁂 𑀲𑀢𑁆𑀢 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄,_  
_𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀢𑀺 𑀲𑁄 𑀫𑀼𑀦𑀺;_  
_𑀤𑀲𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(7.)

960\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀤𑁂𑀯𑀤𑁂𑀯𑀲𑁆𑀲,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(8.)

961\. _𑀢𑀺𑀁𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀑𑀖𑀸𑀦𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀪𑀺𑀚𑁆𑀚𑀺𑀢𑀸𑀦𑀜𑁆𑀘 𑀫𑀘𑁆𑀘𑀼𑀬𑀸𑁇_  


(9.)

962\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀧𑀩𑁆𑀩𑀢𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀅𑀦𑀦𑁆𑀢𑀩𑀮𑀯𑀸𑀳𑀦𑁄𑁇_  


(10.)

963\. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀤𑀲𑁆𑀲𑀦𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀲𑀚𑀺𑀦𑀲𑀁𑀖𑀁,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(11.)

964\. _𑀧𑀢𑁆𑀢𑀼𑀡𑁆𑀡𑀁 𑀘𑀻𑀦𑀧𑀝𑁆𑀝𑀜𑁆𑀘,_  
_𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀫𑁆𑀩𑀮𑀫𑁆𑀧𑀺 𑀘;_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑀧𑀸𑀤𑀼𑀓𑀜𑁆𑀘𑁂𑀯,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀯𑀓𑁂𑁇_  


(12.)

965\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀬;_  
_‘𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(13.)

966\. _𑀅𑀳𑀼 𑀓𑀧𑀺𑀮𑀯𑁆𑀳𑀬𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(14.)

967\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(15.)

968\. _𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀑𑀳𑀸𑀬𑀸𑀳𑀁 𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀚𑀺𑀦𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(16.)

969\. _𑀦𑀕𑀭𑀁 𑀲𑁄𑀪𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀲𑁄𑀪𑁄 𑀦𑀸𑀫𑀸𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀯𑀲𑀢𑁂 𑀢𑀢𑁆𑀣 𑀦𑀕𑀭𑁂,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀫𑀳𑀸𑀓𑀼𑀮𑀁𑁇_  


(17.)

970\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑀜𑁆𑀜𑀤𑀢𑁆𑀢𑁄 𑀘,_  
_𑀆𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑁄 𑀧𑀺𑀢𑀸;_  
_𑀉𑀢𑁆𑀢𑀭𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(18.)

971\. _𑀢𑀻𑀡𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀢𑀼𑀲𑀺𑀢𑀲𑀦𑁆𑀢𑀼𑀲𑀺𑀢𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(19.)

972\. _𑀅𑀦𑀽𑀦𑀲𑁄𑀍𑀅𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀭𑀼𑀘𑀺𑀕𑀢𑁆𑀢𑀸 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(20.)

973\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀙𑀫𑀸𑀲𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(21.)

974\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(22.)

975\. _𑀪𑀺𑀬𑁆𑀬𑀲𑁄 𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑁄𑀢𑁆𑀣𑀺𑀚𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(23.)

976\. _𑀲𑀫𑀼𑀤𑁆𑀤𑀸 𑀉𑀢𑁆𑀢𑀭𑀸 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(24.)

977\. _𑀉𑀕𑁆𑀕𑁄 𑀘 𑀲𑁄𑀫𑀤𑁂𑀯𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀲𑀻𑀯𑀮𑀸 𑀘𑁂𑀯 𑀲𑀸𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(25.)

978\. _𑀉𑀘𑁆𑀘𑀢𑁆𑀢𑀦𑁂𑀦 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀺𑀁𑀲𑀳𑀢𑁆𑀣𑀲𑀫𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀉𑀓𑁆𑀓𑀸𑀫𑀼𑀔𑁂 𑀬𑀣𑀸 𑀓𑀫𑁆𑀩𑀼,_  
_𑀏𑀯𑀁 𑀭𑀁𑀲𑀻𑀳𑀺 𑀫𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


(26.)

979\. _𑀢𑀺𑀁𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(27.)

980\. _𑀥𑀫𑁆𑀫𑀘𑁂𑀢𑀺𑀁 𑀲𑀫𑀼𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸,_  
_𑀥𑀫𑁆𑀫𑀤𑀼𑀲𑁆𑀲𑀯𑀺𑀪𑀽𑀲𑀺𑀢𑀁;_  
_𑀥𑀫𑁆𑀫𑀧𑀼𑀧𑁆𑀨𑀕𑀼𑀍𑀅𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(28.)

981\. _𑀫𑀳𑀸𑀯𑀺𑀮𑀸𑀲𑁄 𑀢𑀲𑁆𑀲 𑀚𑀦𑁄,_  
_𑀲𑀺𑀭𑀺𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀓𑀸𑀲𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(29.)

982\. _𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀧𑀩𑁆𑀩𑀢𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀢𑀼𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀆𑀲𑀺,_  
_𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀢𑁄”𑀢𑀺𑁇 (𑁯𑁫𑁭)_
