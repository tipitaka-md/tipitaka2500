# 𑁨𑁨 𑀲𑀺𑀔𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀲𑀺𑀔𑀻𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

866\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀲𑀺𑀔𑀺𑀯𑁆𑀳𑀬𑁄 𑀆𑀲𑀺 𑀚𑀺𑀦𑁄,_  
_𑀅𑀲𑀫𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


(2.)

867\. _𑀫𑀸𑀭𑀲𑁂𑀦𑀁 𑀧𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀲𑀺,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(3.)

868\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀲𑀺𑀔𑀺𑀫𑁆𑀳𑀺 𑀚𑀺𑀦𑀧𑀼𑀗𑁆𑀕𑀯𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

869\. _𑀅𑀧𑀭𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁂,_  
_𑀕𑀡𑀲𑁂𑀝𑁆𑀞𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

870\. _𑀬𑀫𑀓𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀜𑁆𑀘,_  
_𑀤𑀲𑁆𑀲𑀬𑀦𑁆𑀢𑁂 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

871\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀲𑀺𑀔𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(7.)

872\. _𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀅𑀲𑀻𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(8.)

873\. _𑀲𑀢𑁆𑀢𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑁄 𑀧𑀤𑀼𑀫𑀁𑀯,_  
_𑀢𑁄𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑁆𑀧𑀯𑀟𑁆𑀠𑀺𑀢𑀁𑁇_  


(9.)

874\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀅𑀭𑀺𑀦𑁆𑀤𑀫𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀲𑀁𑀖𑀁,_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀢𑀧𑁆𑀧𑀬𑀺𑀁𑁇_  


(10.)

875\. _𑀩𑀳𑀼𑀁 𑀤𑀼𑀲𑁆𑀲𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀤𑀼𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀁 𑀅𑀦𑀧𑁆𑀧𑀓𑀁;_  
_𑀅𑀮𑀗𑁆𑀓𑀢𑀁 𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(11.)

876\. _𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀦𑀸𑀫𑀬𑀺𑀁;_  
_𑀧𑀽𑀭𑀬𑀺𑀁 𑀫𑀸𑀦𑀲𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀍𑀳𑀫𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁𑁇_  


(12.)

877\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀲𑀺𑀔𑀻 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_‘𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(13.)

878\. _𑀅𑀳𑀼 𑀓𑀧𑀺𑀮𑀯𑁆𑀳𑀬𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(14.)

879\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(15.)

880\. _𑀦𑀕𑀭𑀁 𑀅𑀭𑀼𑀡𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀅𑀭𑀼𑀡𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀧𑀪𑀸𑀯𑀢𑀻 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀲𑀺𑀔𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(16.)

881\. _𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀲𑀼𑀘𑀦𑁆𑀤𑀓𑁄 𑀕𑀺𑀭𑀺 𑀯𑀲𑀪𑁄,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(17.)

882\. _𑀘𑀢𑀼𑀯𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀸 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀅𑀢𑀼𑀮𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(18.)

883\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀝𑁆𑀞𑀫𑀸𑀲𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(19.)

884\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀺𑀔𑀻 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(20.)

885\. _𑀅𑀪𑀺𑀪𑀽 𑀲𑀫𑁆𑀪𑀯𑁄 𑀘𑁂𑀯,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀔𑁂𑀫𑀗𑁆𑀓𑀭𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀲𑀺𑀔𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(21.)

886\. _𑀲𑀔𑀺𑀮𑀸 𑀘 𑀧𑀤𑀼𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(22.)

887\. _𑀲𑀺𑀭𑀺𑀯𑀟𑁆𑀠𑁄 𑀘 𑀦𑀦𑁆𑀤𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀘𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀲𑀼𑀕𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(23.)

888\. _𑀉𑀘𑁆𑀘𑀢𑁆𑀢𑀦𑁂𑀦 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀢𑁆𑀢𑀢𑀺𑀳𑀢𑁆𑀣𑀫𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄𑁇_  


(24.)

889\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀩𑁆𑀬𑀸𑀫𑀧𑁆𑀧𑀪𑀸 𑀓𑀸𑀬𑀸,_  
_𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁 𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀁;_  
_𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀺,_  
_𑀢𑀻𑀡𑀺𑀬𑁄𑀚𑀦𑀲𑁄 𑀧𑀪𑀸𑁇_  


(25.)

890\. _𑀲𑀢𑁆𑀢𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(26.)

891\. _𑀥𑀫𑁆𑀫𑀫𑁂𑀖𑀁 𑀧𑀯𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑁂𑀫𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀔𑁂𑀫𑀦𑁆𑀢𑀁 𑀧𑀸𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(27.)

892\. _𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(28.)

893\. _𑀲𑀺𑀔𑀻 𑀫𑀼𑀦𑀺𑀯𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_
