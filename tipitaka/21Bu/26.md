# 𑁨𑁬 𑀓𑀲𑁆𑀲𑀧𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

* 𑀓𑀲𑁆𑀲𑀧𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

984\. _“𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀧𑀪𑀗𑁆𑀓𑀭𑁄𑁇_  


(2.)

985\. _𑀲𑀜𑁆𑀙𑀟𑁆𑀟𑀺𑀢𑀁 𑀓𑀼𑀮𑀫𑀽𑀮𑀁,_  
_𑀩𑀳𑁆𑀯𑀦𑁆𑀦𑀧𑀸𑀦𑀪𑁄𑀚𑀦𑀁;_  
_𑀤𑀢𑁆𑀯𑀸𑀦 𑀬𑀸𑀘𑀓𑁂 𑀤𑀸𑀦𑀁,_  
_𑀧𑀽𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀦𑀲𑀁;_  
_𑀉𑀲𑀪𑁄𑀯 𑀆𑀍𑀅𑀓𑀁 𑀪𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(3.)

986\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀓𑀲𑁆𑀲𑀧𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀯𑀻𑀲𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(4.)

987\. _𑀘𑀢𑀼𑀫𑀸𑀲𑀁 𑀬𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀮𑁄𑀓𑁂 𑀘𑀭𑀢𑀺 𑀘𑀸𑀭𑀺𑀓𑀁;_  
_𑀤𑀲𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

988\. _𑀬𑀫𑀓𑀁 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑀬𑀺;_  
_𑀧𑀜𑁆𑀘𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

989\. _𑀲𑀼𑀥𑀫𑁆𑀫𑀸 𑀤𑁂𑀯𑀧𑀼𑀭𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑀬𑀺;_  
_𑀢𑀻𑀡𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑁂𑀯𑀸𑀦𑀁 𑀩𑁄𑀥𑀬𑀻 𑀚𑀺𑀦𑁄𑁇_  


(7.)

990\. _𑀦𑀭𑀤𑁂𑀯𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀅𑀧𑀭𑁂 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂;_  
_𑀏𑀢𑁂𑀲𑀸𑀦𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀸,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀸𑁇_  


(8.)

991\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀤𑁂𑀯𑀤𑁂𑀯𑀲𑁆𑀲,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(9.)

992\. _𑀯𑀻𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀪𑀯𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀳𑀺𑀭𑀺𑀲𑀻𑀮𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(10.)

993\. _𑀅𑀳𑀁 𑀢𑀤𑀸 𑀫𑀸𑀡𑀯𑀓𑁄,_  
_𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁄𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽𑁇_  


(11.)

994\. _𑀮𑀓𑁆𑀔𑀡𑁂 𑀇𑀢𑀺𑀳𑀸𑀲𑁂 𑀘,_  
_𑀲𑀥𑀫𑁆𑀫𑁂 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑁄;_  
_𑀪𑀽𑀫𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀓𑀼𑀲𑀮𑁄,_  
_𑀓𑀢𑀯𑀺𑀚𑁆𑀚𑁄 𑀅𑀦𑀸𑀯𑀬𑁄𑁇_  


(12.)

995\. _𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀖𑀝𑀻𑀓𑀸𑀭𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄;_  
_𑀲𑀕𑀸𑀭𑀯𑁄 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀢𑀢𑀺𑀬𑁂 𑀨𑀮𑁂𑁇_  


(13.)

996\. _𑀆𑀤𑀸𑀬 𑀫𑀁 𑀖𑀝𑀻𑀓𑀸𑀭𑁄,_  
_𑀉𑀧𑀕𑀜𑁆𑀙𑀺 𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀺𑀦𑀁;_  
_𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(14.)

997\. _𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀯𑀢𑁆𑀢𑀸𑀯𑀢𑁆𑀢𑁂𑀲𑀼 𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀦 𑀓𑁆𑀯𑀘𑀺 𑀧𑀭𑀺𑀳𑀸𑀬𑀸𑀫𑀺,_  
_𑀧𑀽𑀭𑁂𑀲𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(15.)

998\. _𑀬𑀸𑀯𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀪𑀡𑀺𑀢𑀁,_  
_𑀦𑀯𑀗𑁆𑀕𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑁄𑀪𑀬𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(16.)

999\. _𑀫𑀫 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀭𑀺;_  
_‘𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(17.)

1000\. _𑀅𑀳𑀼 𑀓𑀧𑀺𑀮𑀯𑁆𑀳𑀬𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑁄;_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀁𑁇_  


(18.)

1001\. _𑀅𑀚𑀧𑀸𑀮𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂,_  
_𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑁄;_  
_𑀢𑀢𑁆𑀣 𑀧𑀸𑀬𑀸𑀲𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀫𑀼𑀧𑁂𑀳𑀺𑀢𑀺𑁇_  


(19.)

1002\. _𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀸𑀬 𑀢𑀻𑀭𑀫𑁆𑀳𑀺,_  
_𑀧𑀸𑀬𑀸𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀬;_  
_𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀯𑀭𑀫𑀕𑁆𑀕𑁂𑀦,_  
_𑀩𑁄𑀥𑀺𑀫𑀽𑀮𑀫𑀼𑀧𑁂𑀳𑀺𑀢𑀺𑁇_  


(20.)

1003\. _𑀢𑀢𑁄 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀩𑁄𑀥𑀺𑀫𑀡𑁆𑀟𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀝𑁆𑀞𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀩𑁄𑀥𑀺𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(21.)

1004\. _𑀇𑀫𑀲𑁆𑀲 𑀚𑀦𑀺𑀓𑀸 𑀫𑀸𑀢𑀸,_  
_𑀫𑀸𑀬𑀸 𑀦𑀸𑀫 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀺𑀢𑀸 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀦𑀸𑀫,_  
_𑀅𑀬𑀁 𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀕𑁄𑀢𑀫𑁄𑁇_  


(22.)

1005\. _𑀅𑀦𑀸𑀲𑀯𑀸 𑀯𑀻𑀢𑀭𑀸𑀕𑀸,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀓𑁄𑀮𑀺𑀢𑁄 𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀘,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀸𑀯𑀓𑀸;_  
_𑀆𑀦𑀦𑁆𑀤𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀁 𑀚𑀺𑀦𑀁𑁇_  


(23.)

1006\. _𑀔𑁂𑀫𑀸 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀅𑀦𑀸𑀲𑀯𑀸 𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(24.)

1007\. _𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑀢𑁆𑀣𑀸𑀍𑀅𑀯𑀓𑁄 𑀘,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀦𑀦𑁆𑀤𑀫𑀸𑀢𑀸 𑀘 𑀉𑀢𑁆𑀢𑀭𑀸,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸’𑁇_  


(25.)

1008\. _𑀇𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀯𑀘𑀦𑀁,_  
_𑀅𑀲𑁆𑀲𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀆𑀫𑁄𑀤𑀺𑀢𑀸 𑀦𑀭𑀫𑀭𑀽,_  
_𑀩𑀼𑀤𑁆𑀥𑀩𑀻𑀚𑀁 𑀓𑀺𑀭 𑀅𑀬𑀁𑁇_  


(26.)

1009\. _𑀉𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑀺𑀲𑀤𑁆𑀤𑀸 𑀧𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀅𑀧𑁆𑀨𑁄𑀝𑁂𑀦𑁆𑀢𑀺 𑀳𑀲𑀦𑁆𑀢𑀺 𑀘;_  
_𑀓𑀢𑀜𑁆𑀚𑀮𑀻 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀻 𑀲𑀤𑁂𑀯𑀓𑀸𑁇_  


(27.)

1010\. _‘𑀬𑀤𑀺𑀫𑀲𑁆𑀲 𑀮𑁄𑀓𑀦𑀸𑀣𑀲𑁆𑀲,_  
_𑀯𑀺𑀭𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀸𑀫 𑀲𑀸𑀲𑀦𑀁;_  
_𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀳𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑁂,_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁𑁇_  


(28.)

1011\. _𑀬𑀣𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀁 𑀢𑀭𑀦𑁆𑀢𑀸,_  
_𑀧𑀝𑀺𑀢𑀺𑀢𑁆𑀣𑀁 𑀯𑀺𑀭𑀚𑁆𑀛𑀺𑀬;_  
_𑀳𑁂𑀝𑁆𑀞𑀸 𑀢𑀺𑀢𑁆𑀣𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀢𑁆𑀢𑀭𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀤𑀺𑀁𑁇_  


(29.)

1012\. _𑀏𑀯𑀫𑁂𑀯𑀁 𑀫𑀬𑀁 𑀲𑀩𑁆𑀩𑁂,_  
_𑀬𑀤𑀺 𑀫𑀼𑀜𑁆𑀘𑀸𑀫𑀺𑀫𑀁 𑀚𑀺𑀦𑀁;_  
_𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀳𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑁂,_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(30.)

1013\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(31.)

1014\. _𑀏𑀯𑀫𑀳𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀸𑀘𑀭𑀁;_  
_𑀤𑀼𑀓𑁆𑀓𑀭𑀜𑁆𑀘 𑀓𑀢𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀓𑀸𑀭𑀡𑀸𑁇_  


(32.)

1015\. _𑀦𑀕𑀭𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀦𑀸𑀫,_  
_𑀓𑀺𑀓𑀻 𑀦𑀸𑀫𑀸𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀯𑀲𑀢𑁂 𑀢𑀢𑁆𑀣 𑀦𑀕𑀭𑁂,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀫𑀳𑀸𑀓𑀼𑀮𑀁𑁇_  


(33.)

1016\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄𑀯,_  
_𑀆𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑁄 𑀧𑀺𑀢𑀸;_  
_𑀥𑀦𑀯𑀢𑀻 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(34.)

1017\. _𑀤𑀼𑀯𑁂 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀳𑀁𑀲𑁄 𑀬𑀲𑁄 𑀲𑀺𑀭𑀺𑀦𑀦𑁆𑀤𑁄,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(35.)

1018\. _𑀢𑀺𑀲𑁄𑀍𑀅𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑀼𑀦𑀦𑁆𑀤𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀦𑀸𑀭𑀻,_  
_𑀯𑀺𑀚𑀺𑀢𑀲𑁂𑀦𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(36.)

1019\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀸𑀲𑀸𑀤𑁂𑀦𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀧𑀥𑀸𑀦𑀘𑀸𑀭𑀁,_  
_𑀅𑀘𑀭𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(37.)

1020\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(38.)

1021\. _𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑀩𑁆𑀩𑀫𑀺𑀢𑁆𑀢𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(39.)

1022\. _𑀅𑀦𑀼𑀍𑀆 𑀉𑀭𑀼𑀯𑁂𑀮𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(40.)

1023\. _𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑁄 𑀖𑀝𑀺𑀓𑀸𑀭𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀯𑀺𑀘𑀺𑀢𑀲𑁂𑀦𑀸 𑀪𑀤𑁆𑀤𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(41.)

1024\. _𑀉𑀘𑁆𑀘𑀢𑁆𑀢𑀦𑁂𑀦 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀻𑀲𑀢𑀺𑀭𑀢𑀦𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑀝𑁆𑀞𑀻𑀯 𑀆𑀓𑀸𑀲𑁂,_  
_𑀘𑀦𑁆𑀤𑁄𑀯 𑀕𑀳𑀧𑀽𑀭𑀺𑀢𑁄𑁇_  


(42.)

1025\. _𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(43.)

1026\. _𑀥𑀫𑁆𑀫𑀢𑀍𑀆𑀓𑀁 𑀫𑀸𑀧𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀻𑀮𑀁 𑀤𑀢𑁆𑀯𑀸 𑀯𑀺𑀮𑁂𑀧𑀦𑀁;_  
_𑀥𑀫𑁆𑀫𑀤𑀼𑀲𑁆𑀲𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸,_  
_𑀥𑀫𑁆𑀫𑀫𑀸𑀮𑀁 𑀯𑀺𑀪𑀚𑁆𑀚𑀺𑀬𑁇_  


(44.)

1027\. _𑀥𑀫𑁆𑀫𑀯𑀺𑀫𑀮𑀫𑀸𑀤𑀸𑀲𑀁,_  
_𑀞𑀧𑀬𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀚𑀦𑁂;_  
_𑀓𑁂𑀘𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀢𑁆𑀣𑁂𑀦𑁆𑀢𑀸,_  
_𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀼 𑀫𑁂 𑀅𑀮𑀗𑁆𑀓𑀭𑀁𑁇_  


(45.)

1028\. _𑀲𑀻𑀮𑀓𑀜𑁆𑀘𑀼𑀓𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀛𑀸𑀦𑀓𑀯𑀘𑀯𑀫𑁆𑀫𑀺𑀢𑀁;_  
_𑀥𑀫𑁆𑀫𑀘𑀫𑁆𑀫𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑁆𑀯𑀸,_  
_𑀤𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀦𑀸𑀳𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(46.)

1029\. _𑀲𑀢𑀺𑀨𑀮𑀓𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀺𑀔𑀺𑀡𑀜𑀸𑀡𑀓𑀼𑀦𑁆𑀢𑀺𑀫𑀁;_  
_𑀥𑀫𑁆𑀫𑀔𑀕𑁆𑀕𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀻𑀮𑀲𑀁𑀲𑀕𑁆𑀕𑀫𑀤𑁆𑀤𑀦𑀁𑁇_  


(47.)

1030\. _𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀽𑀲𑀦𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀯𑁂𑀍𑀅𑀁 𑀘𑀢𑀼𑀭𑁄 𑀨𑀮𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀪𑀭𑀡𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀥𑀫𑁆𑀫𑀧𑀼𑀧𑁆𑀨𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀦𑀁𑁇_  


(48.)

1031\. _𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀧𑀡𑁆𑀟𑀭𑀘𑁆𑀙𑀢𑁆𑀢𑀁,_  
_𑀤𑀢𑁆𑀯𑀸 𑀧𑀸𑀧𑀦𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀫𑀸𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀪𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀲𑀲𑀸𑀯𑀓𑁄𑁇_  


(49.)

1032\. _𑀏𑀲𑁄 𑀳𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀭𑀸𑀲𑀤𑁄;_  
_𑀏𑀲𑁄 𑀳𑀺 𑀥𑀫𑁆𑀫𑀭𑀢𑀦𑁄,_  
_𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄𑁇_  


(50.)

1033\. _𑀏𑀲𑁄 𑀳𑀺 𑀲𑀁𑀖𑀭𑀢𑀦𑁄,_  
_𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(51.)

1034\. _𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀺𑀦𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑁂𑀢𑀩𑁆𑀬𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀲𑁆𑀲 𑀚𑀺𑀦𑀣𑀽𑀧𑁄,_  
_𑀬𑁄𑀚𑀦𑀼𑀩𑁆𑀩𑁂𑀥𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁧𑁦𑁦𑁮)_  


---

1035\. 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇
