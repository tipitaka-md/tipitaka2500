# 𑁬 𑀲𑀼𑀫𑀦𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀲𑀼𑀫𑀦𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

376\. _“𑀫𑀗𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀲𑀼𑀫𑀦𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀲𑀫𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(2.)

377\. _𑀢𑀤𑀸 𑀅𑀫𑀢𑀪𑁂𑀭𑀺𑀁 𑀲𑁄,_  
_𑀆𑀳𑀦𑀻 𑀫𑁂𑀔𑀮𑁂 𑀧𑀼𑀭𑁂;_  
_𑀥𑀫𑁆𑀫𑀲𑀗𑁆𑀔𑀲𑀫𑀸𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀦𑀯𑀗𑁆𑀕𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(3.)

378\. _𑀦𑀺𑀚𑁆𑀚𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀓𑀺𑀮𑁂𑀲𑁂 𑀲𑁄,_  
_𑀧𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀫𑀸𑀧𑁂𑀲𑀺 𑀦𑀕𑀭𑀁 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀧𑀼𑀭𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(4.)

379\. _𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀁 𑀅𑀓𑀼𑀝𑀺𑀮𑀁,_  
_𑀉𑀚𑀼𑀁 𑀯𑀺𑀧𑀼𑀮𑀯𑀺𑀢𑁆𑀣𑀢𑀁;_  
_𑀫𑀸𑀧𑁂𑀲𑀺 𑀲𑁄 𑀫𑀳𑀸𑀯𑀻𑀣𑀺𑀁,_  
_𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(5.)

380\. _𑀨𑀮𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑁂,_  
_𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀝𑁆𑀞𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻,_  
_𑀧𑀲𑀸𑀭𑁂𑀲𑀺 𑀢𑀢𑁆𑀣 𑀯𑀻𑀣𑀺𑀬𑀁𑁇_  


(6.)

381\. _𑀬𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀅𑀔𑀺𑀮𑀸,_  
_𑀳𑀺𑀭𑀺𑀯𑀻𑀭𑀺𑀬𑁂𑀳𑀼𑀧𑀸𑀕𑀢𑀸;_  
_𑀢𑁂 𑀢𑁂 𑀇𑀫𑁂 𑀕𑀼𑀡𑀯𑀭𑁂,_  
_𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑀺 𑀬𑀣𑀸 𑀲𑀼𑀔𑀁𑁇_  


(7.)

382\. _𑀏𑀯𑀫𑁂𑀢𑁂𑀦 𑀬𑁄𑀕𑁂𑀦,_  
_𑀉𑀤𑁆𑀥𑀭𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀩𑁄𑀥𑁂𑀲𑀺 𑀧𑀞𑀫𑀁 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀬𑁄𑁇_  


(8.)

383\. _𑀬𑀫𑁆𑀳𑀺 𑀓𑀸𑀮𑁂 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀑𑀯𑀤𑀻 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀕𑀡𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀪𑀺𑀲𑀫𑀺𑀁𑀲𑀼,_  
_𑀤𑀼𑀢𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂𑁇_  


(9.)

384\. _𑀬𑀤𑀸 𑀤𑁂𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘,_  
_𑀲𑀫𑀕𑁆𑀕𑀸 𑀏𑀓𑀫𑀸𑀦𑀲𑀸;_  
_𑀦𑀺𑀭𑁄𑀥𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼,_  
_𑀲𑀁𑀲𑀬𑀜𑁆𑀘𑀸𑀧𑀺 𑀫𑀸𑀦𑀲𑀁𑁇_  


(10.)

385\. _𑀢𑀤𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑁂,_  
_𑀦𑀺𑀭𑁄𑀥𑀧𑀭𑀺𑀤𑀻𑀧𑀦𑁂;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(11.)

386\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀲𑀼𑀫𑀦𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(12.)

387\. _𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀅𑀪𑀺𑀖𑀼𑀝𑁆𑀞𑁂 𑀧𑀯𑀸𑀭𑀡𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀧𑀯𑀸𑀭𑁂𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑁄𑁇_  


(13.)

388\. _𑀢𑀢𑁄𑀧𑀭𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂,_  
_𑀯𑀺𑀫𑀮𑁂 𑀓𑀜𑁆𑀘𑀦𑀧𑀩𑁆𑀩𑀢𑁂;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(14.)

389\. _𑀬𑀤𑀸 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀭𑀸𑀚𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀤𑀲𑁆𑀲𑀦𑀼𑀧𑀸𑀕𑀫𑀺;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(15.)

390\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄;_  
_𑀅𑀢𑀼𑀮𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀉𑀲𑁆𑀲𑀦𑁆𑀦𑀓𑀼𑀲𑀮𑀲𑀜𑁆𑀘𑀬𑁄𑁇_  


(16.)

391\. _𑀢𑀤𑀸𑀳𑀁 𑀦𑀸𑀕𑀪𑀯𑀦𑀸,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀜𑀸𑀢𑀺𑀪𑀺;_  
_𑀦𑀸𑀕𑀸𑀦𑀁 𑀤𑀺𑀩𑁆𑀩𑀢𑀼𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀲𑀲𑀁𑀖𑀁 𑀚𑀺𑀦𑀫𑀼𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑁇_  


(17.)

392\. _𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀢𑀧𑁆𑀧𑀬𑀺𑀁;_  
_𑀧𑀘𑁆𑀘𑁂𑀓𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀭𑀡𑀁 𑀢𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(18.)

393\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀲𑀼𑀫𑀦𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(19.)

394\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_…𑀧𑁂…_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(20.)

395\. _𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀤𑀲𑀧𑀸𑀭𑀫𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇_  


(21.)

396\. _𑀦𑀕𑀭𑀁 𑀫𑁂𑀔𑀮𑀁 𑀦𑀸𑀫,_  
_𑀲𑀼𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀺𑀭𑀺𑀫𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀲𑀼𑀫𑀦𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(22.)

397\. _𑀦𑀯𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀘𑀦𑁆𑀤𑁄 𑀲𑀼𑀘𑀦𑁆𑀤𑁄 𑀯𑀝𑀁𑀲𑁄 𑀘,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(23.)

398\. _𑀢𑁂𑀲𑀝𑁆𑀞𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀯𑀝𑀁𑀲𑀺𑀓𑀸 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀅𑀦𑀽𑀧𑀫𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(24.)

399\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀤𑀲𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(25.)

400\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀫𑀦𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀫𑁂𑀔𑀮𑁂 𑀧𑀼𑀭𑀫𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(26.)

401\. _𑀲𑀭𑀡𑁄 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀉𑀤𑁂𑀦𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀲𑀼𑀫𑀦𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(27.)

402\. _𑀲𑁄𑀡𑀸 𑀘 𑀉𑀧𑀲𑁄𑀡𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀲𑁄𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀦𑀸𑀕𑀫𑀽𑀮𑁂 𑀅𑀩𑀼𑀚𑁆𑀛𑀣𑁇_  


(28.)

403\. _𑀯𑀭𑀼𑀡𑁄 𑀘𑁂𑀯 𑀲𑀭𑀡𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀘𑀸𑀮𑀸 𑀘 𑀉𑀧𑀘𑀸𑀮𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(29.)

404\. _𑀉𑀘𑁆𑀘𑀢𑁆𑀢𑀦𑁂𑀦 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀯𑀼𑀢𑀺𑀳𑀢𑁆𑀣𑀫𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀻 𑀯𑀺𑀭𑁄𑀘𑀢𑀺𑁇_  


(30.)

405\. _𑀦𑀯𑀼𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(31.)

406\. _𑀢𑀸𑀭𑀡𑀻𑀬𑁂 𑀢𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑁄𑀥𑀦𑀻𑀬𑁂 𑀘 𑀩𑁄𑀥𑀬𑀺;_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀉𑀍𑀉𑀭𑀸𑀚𑀸𑀯 𑀅𑀢𑁆𑀣𑀫𑀺𑁇_  


(32.)

407\. _𑀢𑁂 𑀘 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽,_  
_𑀲𑁄 𑀘 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀸𑀤𑀺𑀲𑁄;_  
_𑀅𑀢𑀼𑀮𑀧𑁆𑀧𑀪𑀁 𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀬𑁂 𑀫𑀳𑀸𑀬𑀲𑀸𑁇_  


(33.)

408\. _𑀢𑀜𑁆𑀘 𑀜𑀸𑀡𑀁 𑀅𑀢𑀼𑀮𑀺𑀬𑀁,_  
_𑀢𑀸𑀦𑀺 𑀘 𑀅𑀢𑀼𑀮𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(34.)

409\. _𑀲𑀼𑀫𑀦𑁄 𑀬𑀲𑀥𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀗𑁆𑀕𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀢𑀲𑁆𑀲 𑀚𑀺𑀦𑀣𑀽𑀧𑁄,_  
_𑀘𑀢𑀼𑀬𑁄𑀚𑀦𑀫𑀼𑀕𑁆𑀕𑀢𑁄”𑀢𑀺𑁇 (𑁪𑁦𑁩)_  


---

410\. 𑀲𑀼𑀫𑀦𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
