# 𑁪 𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

### 𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀩𑀼𑀤𑁆𑀥𑀯𑀁𑀲

(1.)

304\. _“𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀦𑀸𑀫 𑀦𑀸𑀬𑀓𑁄;_  
_𑀅𑀦𑀦𑁆𑀢𑀢𑁂𑀚𑁄 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀤𑀼𑀭𑀸𑀲𑀤𑁄𑁇_  


(2.)

305\. _𑀥𑀭𑀡𑀽𑀧𑀫𑁄 𑀔𑀫𑀦𑁂𑀦,_  
_𑀲𑀻𑀮𑁂𑀦 𑀲𑀸𑀕𑀭𑀽𑀧𑀫𑁄;_  
_𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀫𑁂𑀭𑀽𑀧𑀫𑁄,_  
_𑀜𑀸𑀡𑁂𑀦 𑀕𑀕𑀦𑀽𑀧𑀫𑁄𑁇_  


(3.)

306\. _𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀩𑀮𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕-_  
_𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀧𑁆𑀧𑀓𑀸𑀲𑀦𑀁;_  
_𑀧𑀓𑀸𑀲𑁂𑀲𑀺 𑀲𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀳𑀺𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(4.)

307\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(5.)

308\. _𑀢𑀢𑁄 𑀧𑀭𑀫𑁆𑀧𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁂,_  
_𑀦𑀭𑀫𑀭𑀽𑀦𑀁 𑀲𑀫𑀸𑀕𑀫𑁂;_  
_𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(6.)

309\. _𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀅𑀪𑀺𑀫𑀤𑁆𑀤𑀦𑁆𑀢𑁄,_  
_𑀬𑀤𑀸 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑀬𑀺;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(7.)

310\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀢𑀬𑁄 𑀆𑀲𑀼𑀁,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁 𑀯𑀺𑀫𑀮𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(8.)

311\. _𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀞𑀫𑁄 𑀆𑀲𑀺 𑀲𑀫𑀸𑀕𑀫𑁄;_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀦𑀯𑀼𑀢𑀺𑀓𑁄𑀝𑀺𑀦𑀁𑁇_  


(9.)

312\. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀦𑁆𑀢𑀫𑀦𑁆𑀢𑁂𑀦,_  
_𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀯𑀢𑁆𑀢𑀬𑀸𑀫𑀳𑀁𑁇_  


(10.)

313\. _𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀯𑀺𑀫𑀮𑀸𑀦𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀁;_  
_𑀲𑀳 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀣𑁂𑀦,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑁂𑀦 𑀢𑀧𑁆𑀧𑀬𑀺𑀁𑁇_  


(11.)

314\. _𑀲𑁄𑀧𑀺 𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(12.)

315\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀁;_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀫𑀽𑀮𑁂 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀩𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(13.)

316\. _𑀇𑀫𑀲𑁆𑀲 𑀚𑀦𑀺𑀓𑀸 𑀫𑀸𑀢𑀸,_  
_𑀫𑀸𑀬𑀸 𑀦𑀸𑀫 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀺𑀢𑀸 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀦𑀸𑀫,_  
_𑀅𑀬𑀁 𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀕𑁄𑀢𑀫𑁄𑁇_  


(14.)

317\. _𑀓𑁄𑀮𑀺𑀢𑁄 𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀘,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀸𑀯𑀓𑀸;_  
_𑀆𑀦𑀦𑁆𑀤𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀁 𑀚𑀺𑀦𑀁𑁇_  


(15.)

318\. _𑀔𑁂𑀫𑀸 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(16.)

319\. _𑀘𑀺𑀢𑁆𑀢𑁄 𑀘 𑀳𑀢𑁆𑀣𑀸𑀍𑀅𑀯𑀓𑁄,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀦𑀦𑁆𑀤𑀫𑀸𑀢𑀸 𑀘 𑀉𑀢𑁆𑀢𑀭𑀸,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸;_  
_𑀆𑀬𑀼 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀢𑀲𑁆𑀲,_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄’𑁇_  


(17.)

320\. _𑀇𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀯𑀘𑀦𑀁,_  
_𑀅𑀲𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀆𑀫𑁄𑀤𑀺𑀢𑀸 𑀦𑀭𑀫𑀭𑀽,_  
_‘𑀩𑀼𑀤𑁆𑀥𑀩𑀻𑀚𑀁 𑀓𑀺𑀭 𑀅𑀬𑀁’𑁇_  


(18.)

321\. _𑀉𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑀺𑀲𑀤𑁆𑀤𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀅𑀧𑁆𑀨𑁄𑀝𑁂𑀦𑁆𑀢𑀺 𑀳𑀲𑀦𑁆𑀢𑀺 𑀘;_  
_𑀓𑀢𑀜𑁆𑀚𑀮𑀻 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀺𑀤𑁂𑀯𑀢𑀸𑁇_  


(19.)

322\. _‘𑀬𑀤𑀺𑀫𑀲𑁆𑀲 𑀮𑁄𑀓𑀦𑀸𑀣𑀲𑁆𑀲,_  
_𑀯𑀺𑀭𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀸𑀫 𑀲𑀸𑀲𑀦𑀁;_  
_𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀳𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑁂,_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁𑁇_  


(20.)

323\. _𑀬𑀣𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀁 𑀢𑀭𑀦𑁆𑀢𑀸,_  
_𑀧𑀝𑀺𑀢𑀺𑀢𑁆𑀣𑀁 𑀯𑀺𑀭𑀚𑁆𑀛𑀺𑀬;_  
_𑀳𑁂𑀝𑁆𑀞𑀸𑀢𑀺𑀢𑁆𑀣𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀢𑁆𑀢𑀭𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀤𑀺𑀁𑁇_  


(21.)

324\. _𑀏𑀯𑀫𑁂𑀯𑀁 𑀫𑀬𑀁 𑀲𑀩𑁆𑀩𑁂,_  
_𑀬𑀤𑀺 𑀫𑀼𑀜𑁆𑀘𑀸𑀫𑀺𑀫𑀁 𑀚𑀺𑀦𑀁;_  
_𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀳𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑁂,_  
_𑀳𑁂𑀲𑁆𑀲𑀸𑀫 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀁’𑁇_  


(22.)

325\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀢𑀫𑁂𑀯 𑀅𑀢𑁆𑀣𑀁 𑀲𑀸𑀥𑁂𑀦𑁆𑀢𑁄,_  
_𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀁 𑀚𑀺𑀦𑁂 𑀅𑀤𑀁;_  
_𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀁 𑀤𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(23.)

326\. _𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀯𑀺𑀦𑀬𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀦𑀯𑀗𑁆𑀕𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑁄𑀪𑀬𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(24.)

327\. _𑀢𑀢𑁆𑀣𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀲𑀚𑁆𑀚𑀝𑁆𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑁂;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀫𑀕𑀜𑁆𑀙𑀳𑀁𑁇_  


(25.)

328\. _𑀦𑀕𑀭𑀁 𑀭𑀫𑁆𑀫𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀲𑀼𑀦𑀦𑁆𑀤𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀼𑀚𑀸𑀢𑀸 𑀦𑀸𑀫 𑀚𑀦𑀺𑀓𑀸,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(26.)

329\. _𑀤𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛 𑀲𑁄 𑀯𑀲𑀺;_  
_𑀲𑀼𑀘𑀺 𑀲𑀼𑀭𑀼𑀘𑀺 𑀲𑀼𑀪𑁄 𑀘,_  
_𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑁇_  


(27.)

330\. _𑀢𑀻𑀡𑀺 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀭𑀼𑀘𑀺𑀤𑁂𑀯𑀻 𑀦𑀸𑀫 𑀦𑀸𑀭𑀻,_  
_𑀯𑀺𑀚𑀺𑀢𑀲𑁂𑀦𑁄 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(28.)

331\. _𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀭𑀣𑀬𑀸𑀦𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺;_  
_𑀅𑀦𑀽𑀦𑀤𑀲𑀫𑀸𑀲𑀸𑀦𑀺,_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀻 𑀚𑀺𑀦𑁄𑁇_  


(29.)

332\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀬𑀸𑀘𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀯𑀢𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀤𑁂𑀯𑀸𑀦𑀁 𑀦𑀕𑀭𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(30.)

333\. _𑀪𑀤𑁆𑀤𑁄 𑀘𑁂𑀯 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸;_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(31.)

334\. _𑀢𑀺𑀲𑁆𑀲𑀸 𑀘 𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑁄 𑀩𑁄𑀥𑀺,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(32.)

335\. _𑀲𑁄𑀡𑁄 𑀘 𑀉𑀧𑀲𑁄𑀡𑁄 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀓𑀸;_  
_𑀦𑀦𑁆𑀤𑀸 𑀘𑁂𑀯 𑀲𑀺𑀭𑀻𑀫𑀸 𑀘,_  
_𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(33.)

336\. _𑀲𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀻𑀢𑀺 𑀳𑀢𑁆𑀣𑀸𑀦𑀺,_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀲𑁄𑀪𑀢𑁂 𑀉𑀍𑀉𑀭𑀸𑀚𑀸𑀯,_  
_𑀲𑀽𑀭𑀺𑀬𑁄 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀬𑀣𑀸𑁇_  


(34.)

337\. _𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(35.)

338\. _𑀔𑀻𑀡𑀸𑀲𑀯𑁂𑀳𑀺 𑀯𑀺𑀫𑀮𑁂𑀳𑀺,_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀸 𑀆𑀲𑀺 𑀫𑁂𑀤𑀦𑀻;_  
_𑀬𑀣𑀸 𑀕𑀕𑀦𑀫𑀼𑀍𑀊𑀳𑀺,_  
_𑀏𑀯𑀁 𑀲𑁄 𑀉𑀧𑀲𑁄𑀪𑀣𑁇_  


(36.)

339\. _𑀢𑁂𑀧𑀺 𑀦𑀸𑀕𑀸 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀸,_  
_𑀅𑀲𑀗𑁆𑀔𑁄𑀩𑁆𑀪𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸;_  
_𑀯𑀺𑀚𑁆𑀚𑀼𑀧𑀸𑀢𑀁𑀯 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀢𑁂 𑀫𑀳𑀸𑀬𑀲𑀸𑁇_  


(37.)

340\. _𑀲𑀸 𑀘 𑀅𑀢𑀼𑀮𑀺𑀬𑀸 𑀚𑀺𑀦𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀺,_  
_𑀜𑀸𑀡𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁄 𑀘 𑀲𑀫𑀸𑀥𑀺;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀁,_  
_𑀦𑀦𑀼 𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇_  


(38.)

341\. _𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀧𑀯𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀘𑀦𑁆𑀤𑀸𑀭𑀸𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀘𑁂𑀢𑀺𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀢𑁆𑀢 𑀬𑁄𑀚𑀦𑀫𑀼𑀲𑁆𑀲𑀺𑀢𑁄”𑀢𑀺𑁇 (𑁩𑁩𑁭)_  


---

342\. 𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀁𑀲𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
