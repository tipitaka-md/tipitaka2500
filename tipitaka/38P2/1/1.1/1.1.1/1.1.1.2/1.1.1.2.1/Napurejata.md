# 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢

### 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢

(8.)

19. 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀅𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄, 𑀫𑁄𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀮𑁄𑀪𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

20. 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀳𑁂𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀳𑁂𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

21. 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘 𑀦𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀅𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 (𑀘𑀓𑁆𑀓𑀁); 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)

(9.)

22. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀽 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁧)

23. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑀳𑁂𑀢𑀽 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀽; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

24. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘 𑀦𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁂𑀢𑀼 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)

(10.)

25. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀦𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀅𑀮𑁄𑀪𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀅𑀭𑀽𑀧𑁂 𑀮𑁄𑀪𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 (𑀘𑀓𑁆𑀓𑀁); 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

26. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀦𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀽 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

27. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀦𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘 𑀦𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀅𑀮𑁄𑀪𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀦𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀫𑁄𑀳𑁄 𑀘 (𑀘𑀓𑁆𑀓𑀁); 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)
