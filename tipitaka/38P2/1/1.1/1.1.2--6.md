# 𑁧𑁇𑁧𑁇𑁨--𑁬 𑀲𑀳𑀚𑀸𑀢-𑀧𑀘𑁆𑀘𑀬-𑀦𑀺𑀲𑁆𑀲𑀬-𑀲𑀁𑀲𑀝𑁆𑀞-𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭

* 𑀲𑀳𑀚𑀸𑀢-𑀧𑀘𑁆𑀘𑀬-𑀦𑀺𑀲𑁆𑀲𑀬-𑀲𑀁𑀲𑀝𑁆𑀞-𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭

40\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁄𑀧𑀺 𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭𑁄𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀸𑀭𑁄𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀸𑀬𑁂𑀯 𑀧𑀜𑁆𑀳𑀸𑁇 𑀫𑀳𑀸𑀪𑀽𑀢𑁂𑀲𑀼 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁂𑀲𑀼 “𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸”𑀢𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀧𑀺 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑁂𑀧𑀺 𑀬𑀣𑀸 𑀮𑀩𑁆𑀪𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭𑁄𑀧𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭𑁄𑀧𑀺 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄𑁇 𑀭𑀽𑀧𑀁 𑀦𑀢𑁆𑀣𑀺, 𑀅𑀭𑀽𑀧𑀫𑁂𑀯𑁇)
