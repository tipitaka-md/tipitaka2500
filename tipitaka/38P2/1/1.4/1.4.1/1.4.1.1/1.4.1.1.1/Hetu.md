# 𑀳𑁂𑀢𑀼

### 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(127.)

353\. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 (𑀘𑀓𑁆𑀓𑀁); 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀅𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁧)

354\. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀳𑁂𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

355\. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 (𑀘𑀓𑁆𑀓𑀁); 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)

(128.)

356\. 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

357\. 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀳𑁂𑀢𑀼𑀓𑁂 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀽; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

358\. 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀳𑁂𑀢𑀼 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)

(129.)

359\. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀮𑁄𑀪𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀮𑁄𑀪𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 (𑀘𑀓𑁆𑀓𑀁); 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

360\. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

361\. 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀅𑀮𑁄𑀪𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)

362\. (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
