# 𑀓𑀫𑁆𑀫

* 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀓𑀫𑁆𑀫

(148.)

413\. 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀲𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

414\. 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀳𑁂𑀢𑀽𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀳𑁂𑀢𑀽𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

415\. 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀳𑁂𑀢𑀽𑀦𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀳𑁂𑀢𑀽𑀦𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
