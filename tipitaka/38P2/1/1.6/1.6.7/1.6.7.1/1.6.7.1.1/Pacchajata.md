### 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢

(184.)

502. 𑀦𑀳𑁂𑀢𑀼 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸 𑀦𑀳𑁂𑀢𑀽 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

503. 𑀦𑀳𑁂𑀢𑀼 𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸 𑀦𑀳𑁂𑀢𑀽 𑀅𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)
