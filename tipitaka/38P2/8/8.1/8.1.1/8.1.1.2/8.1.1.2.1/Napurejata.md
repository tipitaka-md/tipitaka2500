# 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢

* 𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀦𑀻𑀯𑀭𑀡𑀤𑀼𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢

(8.)

1907\. 𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀅𑀭𑀽𑀧𑁂 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀅𑀭𑀽𑀧𑁂 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀅𑀭𑀽𑀧𑁂 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 (𑁧)

1908\. 𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀜𑁆𑀳𑀸 𑀲𑀩𑁆𑀩𑁂 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀭𑀽𑀧𑀁 𑀧𑀞𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁, 𑀭𑀽𑀧𑀁 𑀧𑀘𑁆𑀙𑀸 𑀬𑀣𑀸 𑀮𑀪𑀢𑀺𑁇)

(9.)

1909\. 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁧)

1910\. 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀦𑀻𑀯𑀭𑀡𑁂 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀦𑀻𑀯𑀭𑀡𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁨)

1911\. 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁩)
