# 𑀳𑁂𑀢𑀼

* 𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀦𑀻𑀯𑀭𑀡𑀤𑀼𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀳𑁂𑀢𑀼

(1.)

1889\. 𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 (𑁧)

1890\. 𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁨)

1891\. 𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁩)

(2.)

1892\. 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

1893\. 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑁇 (𑁨)

1894\. 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦𑀻𑀯𑀭𑀡𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

(3.)

1895\. 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁧)

1896\. 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀦𑀻𑀯𑀭𑀡𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁨)

1897\. 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀦𑀻𑀯𑀭𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁩)
