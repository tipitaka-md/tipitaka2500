# 𑁮𑁇𑁨𑁇𑁧--𑁭 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

### 𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀤𑀼𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

(51.)

1990\. 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂… 𑁇 (𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀤𑀼𑀓𑀁 𑀬𑀣𑀸 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓𑀁, 𑀏𑀯𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀦𑁆𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁𑁇)

1991\. _𑀤𑁆𑀯𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁂𑀦,_  
_𑀘𑀢𑀼𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀝𑀺𑀖𑁂𑀦 𑀘;_  
_𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀘,_  
_𑀉𑀪𑁄𑀧𑁂𑀢𑁂 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁;_  
_𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀦𑀻𑀯𑀭𑀡𑁂𑀳𑀺,_  
_𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑀁 𑀧𑀬𑁄𑀚𑀦𑀁𑁇_  


---

1992\. (𑀦𑀻𑀯𑀭𑀡𑀤𑀼𑀓𑀲𑁆𑀲 𑀫𑀸𑀢𑀺𑀓𑀸 𑀇𑀥 𑀓𑀢𑀸𑁇)



---

1993\. 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀤𑀼𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
