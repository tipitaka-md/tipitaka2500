# 𑀓𑀫𑁆𑀫𑀸𑀤𑀺

### 𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀓𑀫𑁆𑀫𑀸𑀤𑀺

(77.)

2051. 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2052. 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑀸…  𑀏𑀓𑀁𑁇
