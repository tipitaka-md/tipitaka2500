# 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

### 𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(73.)

2044\. 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀭𑀸𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀫𑀸𑀦𑀁…  𑀤𑀺𑀝𑁆𑀞𑀺𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸 𑀭𑀸𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2045\. 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀭𑀸𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁…  𑀫𑀕𑁆𑀕𑀁…  𑀅𑀪𑀺𑀜𑁆𑀜𑀁…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺; 𑀤𑁄𑀲𑀁…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺; 𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲, 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

(74.)

2046\. 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀫𑀕𑁆𑀕𑀁…𑀧𑁂…  𑀅𑀪𑀺𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺; 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2047\. 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁, 𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀫𑀸𑀦𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)
