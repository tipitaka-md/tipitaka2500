### 𑀅𑀥𑀺𑀧𑀢𑁆𑀬𑀸𑀤𑀺

(102.)

2113\. 𑀦𑀻𑀯𑀭𑀡𑁄 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺…  𑀢𑀻𑀡𑀺𑁇 (𑀕𑀭𑀼𑀓𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬𑁂𑀯𑁇)

2114\. 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀔𑀦𑁆𑀥𑁂 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀔𑀦𑁆𑀥𑁂 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀻𑀯𑀭𑀡𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀔𑀦𑁆𑀥𑁂 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀻𑀯𑀭𑀡𑀸 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀜𑁆𑀘 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(103.)

2115\. 𑀦𑀻𑀯𑀭𑀡𑁄 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀻𑀯𑀭𑀡𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺…  𑀢𑀻𑀡𑀺𑁇 (𑁩)

2116\. 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄 (𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀫𑁆𑀧𑀺 𑀦𑀢𑁆𑀣𑀺, 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸)…  𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯…  𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯…  𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯…  𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯 (𑀆𑀭𑀫𑁆𑀫𑀡𑀲𑀤𑀺𑀲𑀁, 𑀯𑀺𑀧𑀸𑀓𑁄 𑀦𑀢𑁆𑀣𑀺)…  𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀧𑀜𑁆𑀘𑁇
