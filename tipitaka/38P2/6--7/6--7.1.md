# 𑁬--𑁭𑁇𑁧 𑀑𑀖𑀤𑀼𑀓𑀸𑀤𑀺

### 𑀑𑀖𑀬𑁄𑀕𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀑𑀖𑀤𑀼𑀓𑀸𑀤𑀺

(1.)

1886\. 𑀑𑀖𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀑𑀖𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂… 𑁇

(2.)

1887\. 𑀬𑁄𑀕𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀬𑁄𑀕𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂… 𑁇 (𑀤𑁆𑀯𑁂𑀧𑀺 𑀕𑁄𑀘𑁆𑀙𑀓𑀸 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓𑀲𑀤𑀺𑀲𑀸 𑀦𑀺𑀦𑁆𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀸𑁇)

---

1888\. 𑀑𑀖𑀬𑁄𑀕𑀕𑁄𑀘𑁆𑀙𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
