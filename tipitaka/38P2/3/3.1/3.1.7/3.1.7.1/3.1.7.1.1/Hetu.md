### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(16.)

1009\. 𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀆𑀲𑀯𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀪𑀯𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁧)

1010\. 𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀆𑀲𑀯𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀲𑀯𑀸 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

1011\. 𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀆𑀲𑀯𑀲𑁆𑀲 𑀘 𑀦𑁄𑀆𑀲𑀯𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀓𑀸𑀫𑀸𑀲𑀯𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(17.)

1012\. 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀆𑀲𑀯𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑁄𑀆𑀲𑀯𑀸 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

1013\. 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀆𑀲𑀯𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑁄𑀆𑀲𑀯𑀸 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

1014\. 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀆𑀲𑀯𑀲𑁆𑀲 𑀘 𑀦𑁄𑀆𑀲𑀯𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑁄𑀆𑀲𑀯𑀸 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

1015\. 𑀆𑀲𑀯𑁄 𑀘 𑀦𑁄𑀆𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄𑀆𑀲𑀯𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀲𑀯𑀸 𑀘 𑀦𑁄𑀆𑀲𑀯𑀸 𑀘 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)
