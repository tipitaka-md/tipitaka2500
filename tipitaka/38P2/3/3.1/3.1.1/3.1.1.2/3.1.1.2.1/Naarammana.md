# 𑀦𑀆𑀭𑀫𑁆𑀫𑀡

* 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀆𑀲𑀯𑀤𑀼𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀦𑀆𑀭𑀫𑁆𑀫𑀡

(6.)

982\. 𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀆𑀲𑀯𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁧)

983\. 𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀆𑀲𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀢𑁆𑀣𑀼, 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑀬𑀸𑀯 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑁇) (𑁧)

984\. 𑀆𑀲𑀯𑀜𑁆𑀘 𑀦𑁄𑀆𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀆𑀲𑀯𑁂 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁧)
