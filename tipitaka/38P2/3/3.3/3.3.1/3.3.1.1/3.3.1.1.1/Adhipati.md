# 𑀅𑀥𑀺𑀧𑀢𑀺

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀅𑀥𑀺𑀧𑀢𑀺

(53.)

1110\. 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

1111\. 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀫𑁄𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀁𑁇 (𑁧)

1112\. 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀫𑁄𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

1113\. 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀫𑁄𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁩)

(54.)

1114\. 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀫𑁄𑀳𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

1115\. 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑁄𑀳𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁨)

1116\. 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀫𑁄𑀳𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

---

1117\. (𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)
