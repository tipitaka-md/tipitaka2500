# 𑀳𑁂𑀢𑀼

### 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀆𑀲𑀯𑀲𑀸𑀲𑀯𑀤𑀼𑀓

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(109.)

1257\. 𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀸𑀲𑀯𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁) 𑀪𑀯𑀸𑀲𑀯𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁) 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄𑁇 (𑁧)

1258\. 𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀆𑀲𑀯𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁨)

1259\. 𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀸𑀲𑀯𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀪𑀯𑀸𑀲𑀯𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇) (𑁩)

(110.)

1260\. 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀢𑁆𑀣𑀼, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀔𑀦𑁆𑀥𑀸, 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁧)

1261\. 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀸𑀲𑀯𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑀸𑁇 (𑁨)

1262\. 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀆𑀲𑀯𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

(111.)

1263\. 𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀜𑁆𑀘 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀸𑀲𑀯𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄𑁇 (𑀏𑀯𑀁 𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇) (𑁧)

1264\. 𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀜𑁆𑀘 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀆𑀲𑀯𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑁨)

1265\. 𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀜𑁆𑀘 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀓𑀸𑀫𑀸𑀲𑀯𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁩)
