# 𑀆𑀭𑀫𑁆𑀫𑀡

### 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀆𑀭𑀫𑁆𑀫𑀡

(156.)

2699\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸…  𑀏𑀓𑀁𑁇 (𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀤𑀺𑀲𑀁𑁇) (𑁧)

2700\. 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

2701\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁧)
