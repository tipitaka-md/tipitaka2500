### 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(344.)

3169. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

3170. 𑀩𑀸𑀳𑀺𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀩𑀸𑀳𑀺𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀢𑀻𑀡𑀺𑀧𑀺 𑀧𑀽𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀢𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘𑀸𑀢𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)

3171. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑁄 𑀘 𑀩𑀸𑀳𑀺𑀭𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇
