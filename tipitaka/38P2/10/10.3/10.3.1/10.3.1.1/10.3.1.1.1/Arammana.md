### 𑀆𑀭𑀫𑁆𑀫𑀡

(95.)

2551. 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

2552. 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

2553. 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)

(96.)

2554. 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 (𑁧)

2555. 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

2556. 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁩)

2557. 𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑁧)

2558. 𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 (𑁨)

2559. 𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑁩)
