2593\. 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2594\. (𑀦𑀯 𑀧𑀜𑁆𑀳𑀸 𑀧𑀜𑁆𑀘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀫𑁆𑀧𑀺 𑀬𑀣𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀯𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁, 𑀢𑀻𑀲𑀼𑀬𑁂𑀯 𑀫𑁄𑀳𑁄𑁇 𑀲𑀩𑁆𑀩𑁂 𑀧𑀜𑁆𑀳𑀸 𑀧𑀯𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀬𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀳𑀦𑁆𑀢𑁂𑀦𑁇)
