### 𑀆𑀭𑀫𑁆𑀫𑀡

(120.)

2611\. 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇) 𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇) 𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)

(121.)

2612\. 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀨𑀮𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲, 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲, 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄…𑀧𑁂…  𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺𑁇 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑁂𑀦 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2613\. 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺 (𑀧𑀞𑀫𑀕𑀫𑀦𑀲𑀤𑀺𑀲𑀁); 𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺𑁇 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑁂𑀦 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲…𑀧𑁂…  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2614\. 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑀲𑁆𑀲 𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺 (𑀧𑀞𑀫𑀕𑀫𑀦𑀲𑀤𑀺𑀲𑀁); 𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄…𑀧𑁂…  𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2615\. 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀆𑀭𑀩𑁆𑀪 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇) 𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀆𑀭𑀩𑁆𑀪 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇) 𑀘𑁂𑀢𑀲𑀺𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀆𑀭𑀩𑁆𑀪 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)
