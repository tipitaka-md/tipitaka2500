# 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

### 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

### 𑀘𑀺𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

(75.)

2497\. 𑀘𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀁 𑀧𑀼𑀭𑀺𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2498\. 𑀘𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀁 𑀧𑀼𑀭𑀺𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2499\. 𑀘𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀁 𑀧𑀼𑀭𑀺𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(76.)

2500\. 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀧𑀽𑀭𑁂𑀢𑀼𑀓𑀸𑀫𑁂𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀧𑀼𑀭𑀺𑀫𑀕𑀫𑀦𑀲𑀤𑀺𑀲𑀁𑁇)

2501\. 𑀘𑀺𑀢𑁆𑀢𑁄 𑀘 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀁 𑀧𑀼𑀭𑀺𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀁 𑀧𑀼𑀭𑀺𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀁 𑀧𑀼𑀭𑀺𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2502\. 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄 (𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀸)…  𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀧𑀜𑁆𑀘𑁇 (𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀸𑁇)
