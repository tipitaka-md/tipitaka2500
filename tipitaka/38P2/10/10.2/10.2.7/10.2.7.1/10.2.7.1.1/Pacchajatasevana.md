### 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸𑀲𑁂𑀯𑀦

(79.)

2510\. 𑀘𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁧)

2511\. 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁧)

2512\. 𑀘𑀺𑀢𑁆𑀢𑁄 𑀘 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁧)

2513\. 𑀘𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯𑁇
