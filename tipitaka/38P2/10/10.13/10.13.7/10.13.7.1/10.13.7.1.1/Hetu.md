### 𑀳𑁂𑀢𑀼

(394.)

3318\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

3319\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀳𑁂𑀢𑀽 𑀉𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁨)

3320\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸 𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸 𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁩)
