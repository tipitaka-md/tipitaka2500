# 𑀅𑀥𑀺𑀧𑀢𑀺

* 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

* 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀅𑀥𑀺𑀧𑀢𑀺

(30.)

2380\. 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀸𑀦𑀺…𑀧𑁂…  𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀕𑀭𑀼𑀁…𑀧𑁂…  𑀨𑀮𑀁 𑀕𑀭𑀼𑀁…𑀧𑁂…  𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑁂 𑀔𑀦𑁆𑀥𑁂 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2381\. 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2382\. 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀘 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(31.)

2383\. 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀅𑀭𑀺𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺; 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲, 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲, 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑁆𑀲, 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧)
