# 𑁧𑁦𑁇𑁧𑁪𑁇𑁫𑁇𑁧--𑁪 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

* 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

* 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀤𑀼𑀓

* 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

(443.)

3435\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇

3436\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

3437\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀏𑀓𑀁, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀤𑁆𑀯𑁂, 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀤𑁆𑀯𑁂, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀤𑁆𑀯𑁂, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀤𑁆𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀤𑁆𑀯𑁂, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀤𑁆𑀯𑁂, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀤𑁆𑀯𑁂, 𑀆𑀲𑁂𑀯𑀦𑁂 𑀏𑀓𑀁, 𑀓𑀫𑁆𑀫𑁂 𑀤𑁆𑀯𑁂, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀤𑁆𑀯𑁂𑁇

---

3438\. 𑀅𑀦𑀼𑀮𑁄𑀫𑀁𑁇



(444.)

3439\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀅𑀳𑁂𑀢𑀼𑀓𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑁧)

3440\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁂 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁄 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄𑁇 (𑁧)

3441\. 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂, 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂, 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀤𑁆𑀯𑁂, 𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀤𑁆𑀯𑁂, 𑀦𑀆𑀲𑁂𑀯𑀦𑁂 𑀤𑁆𑀯𑁂, 𑀦𑀓𑀫𑁆𑀫𑁂 𑀏𑀓𑀁, 𑀦𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁, 𑀦𑀛𑀸𑀦𑁂 𑀏𑀓𑀁, 𑀦𑀫𑀕𑁆𑀕𑁂 𑀤𑁆𑀯𑁂, 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀤𑁆𑀯𑁂𑁇
