### 𑀅𑀥𑀺𑀧𑀢𑀺

(259.)

2957\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽 𑀥𑀫𑁆𑀫𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇 (𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺𑀧𑀺 𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺𑀧𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)

2958\. 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇 (𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺𑀧𑀺 𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺𑀧𑀺 𑀇𑀫𑁂𑀲𑀫𑁆𑀧𑀺 𑀢𑀺𑀡𑁆𑀡𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀯𑀧𑀺 𑀧𑀜𑁆𑀳𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀸𑁇 𑀅𑀦𑁆𑀢𑁂 𑀢𑀻𑀡𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺𑀬𑁂𑀯𑁇)
