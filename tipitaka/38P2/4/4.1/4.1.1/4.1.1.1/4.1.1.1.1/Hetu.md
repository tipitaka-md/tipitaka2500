# 𑀳𑁂𑀢𑀼

### 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀤𑀼𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(1.)

1322\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀸𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀇𑀲𑁆𑀲𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀫𑀸𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀪𑀯𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇 (𑁧)

1323\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁨)

1324\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇) (𑁩)

(2.)

1325\. 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀢𑁆𑀣𑀼, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀔𑀦𑁆𑀥𑀸, 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀁𑁇 (𑁧)

1326\. 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸𑁇 (𑁨)

1327\. 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

(3.)

1328\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇) (𑁧)

1329\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑁨)

1330\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂… 𑁇 (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇) (𑁩)

1331\. (𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂 𑀭𑀽𑀧𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁄 𑀳𑁂𑀢𑀼𑀲𑀤𑀺𑀲𑁄, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀦𑀢𑁆𑀣𑀺𑁇) 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇
