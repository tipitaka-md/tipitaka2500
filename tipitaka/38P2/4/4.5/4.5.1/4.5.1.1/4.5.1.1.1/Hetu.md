# 𑀳𑁂𑀢𑀼

### 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(100.)

1571\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁧)

1572\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

1573\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁩)

(101.)

1574\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

1575\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸 (𑁨)

1576\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀸 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

(102.)

1577\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁧)

1578\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑁨)

1579\. 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁩)
