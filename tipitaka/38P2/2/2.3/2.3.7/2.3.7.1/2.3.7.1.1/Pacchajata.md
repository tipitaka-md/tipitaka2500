### 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢

(41.)

629. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

630. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

631. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
