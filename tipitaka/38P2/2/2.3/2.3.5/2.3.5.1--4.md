# 𑁨𑁇𑁩𑁇𑁫𑁇𑁧--𑁪 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

* 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

* 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀤𑀼𑀓

* 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

(33.)

610\. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇

611\. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀁 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀕𑀡𑀦𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)
