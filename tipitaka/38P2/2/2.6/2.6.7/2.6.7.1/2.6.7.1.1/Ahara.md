# 𑀆𑀳𑀸𑀭

918\. 𑀮𑁄𑀓𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀮𑁄𑀓𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀮𑁄𑀓𑀺𑀬𑀸 𑀆𑀳𑀸𑀭𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄**—  𑀇𑀫𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

919\. 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇
