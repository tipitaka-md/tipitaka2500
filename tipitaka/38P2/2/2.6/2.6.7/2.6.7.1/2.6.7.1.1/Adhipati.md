### 𑀅𑀥𑀺𑀧𑀢𑀺

(151.)

893\. 𑀮𑁄𑀓𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀮𑁄𑀓𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀧𑀼𑀩𑁆𑀩𑁂…𑀧𑁂…  𑀛𑀸𑀦𑀸…𑀧𑁂…  𑀲𑁂𑀓𑁆𑀔𑀸 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀯𑁄𑀤𑀸𑀦𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺; 𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀮𑁄𑀓𑀺𑀬𑁂 𑀔𑀦𑁆𑀥𑁂 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀮𑁄𑀓𑀺𑀬𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

(152.)

894\. 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

895\. 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀮𑁄𑀓𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀨𑀮𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺; 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲, 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑀥𑀺𑀧𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

896\. 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀮𑁄𑀓𑀺𑀬𑀲𑁆𑀲 𑀘 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺**—  𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
