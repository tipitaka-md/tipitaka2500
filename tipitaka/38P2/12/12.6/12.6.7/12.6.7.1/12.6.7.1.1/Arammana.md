# 𑀆𑀭𑀫𑁆𑀫𑀡

* 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀆𑀭𑀫𑁆𑀫𑀡

(87.)

4035\. 𑀓𑀺𑀮𑁂𑀲𑁄 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀓𑀺𑀮𑁂𑀲𑁂 𑀆𑀭𑀩𑁆𑀪 𑀓𑀺𑀮𑁂𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀓𑀺𑀮𑁂𑀲𑁂 𑀆𑀭𑀩𑁆𑀪 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀓𑀺𑀮𑁂𑀲𑁂 𑀆𑀭𑀩𑁆𑀪 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)

(88.)

4036\. 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀓𑀺𑀮𑁂𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)

4037\. (𑀇𑀢𑀭𑁂𑀧𑀺 𑀢𑀻𑀡𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)
