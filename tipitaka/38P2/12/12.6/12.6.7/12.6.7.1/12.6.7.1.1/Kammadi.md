# 𑀓𑀫𑁆𑀫𑀸𑀤𑀺

### 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀤𑀼𑀓

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀓𑀫𑁆𑀫𑀸𑀤𑀺

(91.)

4041. 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

4042. 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

4043. 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀘 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

4044. 𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺…  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺…  𑀛𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺…  𑀫𑀕𑁆𑀕𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯…  𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯…  𑀦𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯𑁇
