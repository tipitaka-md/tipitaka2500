* 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸𑀲𑁂𑀯𑀦

(26.)

3880\. 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁧)

3881\. 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑁧)

3882\. 𑀓𑀺𑀮𑁂𑀲𑁄 𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁) 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀯𑁇
