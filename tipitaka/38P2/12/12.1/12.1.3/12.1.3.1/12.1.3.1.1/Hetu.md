# 𑀳𑁂𑀢𑀼

### 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓

### 𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(10.)

3834\. 𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀤𑀺𑀲𑀸𑁇)

3835\. 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  (𑀬𑀸𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀫𑀳𑀸𑀪𑀽𑀢𑀸) 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁧)

3836\. 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑀸𑁇 (𑁨)

3837\. 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑁄 𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁩)

(11.)

3838\. 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀮𑁄𑀪𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀫𑁄𑀳𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀮𑁄𑀪𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑀸𑁇 (𑁧)

3839\. 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀓𑀺𑀮𑁂𑀲𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀓𑀺𑀮𑁂𑀲𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

3840\. 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑁄 𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀫𑁄𑀳𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀮𑁄𑀪𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀫𑁄𑀳𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁩)

3841\. (𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀫𑀽𑀮𑁂 𑀧𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)
