# 𑀦𑀓𑀫𑁆𑀫

* 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀦𑀓𑀫𑁆𑀫

(8.)

3829\. 𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 (𑁧)

3830\. 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘𑁂𑀢𑀦𑀸; 𑀩𑀸𑀳𑀺𑀭𑀁…  𑀆𑀳𑀸𑀭𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀉𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…𑀧𑁂… 𑁇 (𑁧)

3831\. 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀺𑀮𑁂𑀲𑁂 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 (𑁧) (𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)
