### 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀸𑀤𑀺

(7.)

3827\. 𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁, 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁, 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀫𑀸𑀦𑁄 𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁, 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀫𑀸𑀦𑁄 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁, 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁, 𑀮𑁄𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁; 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁; 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑁄𑀳𑁄 𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁𑁇 (𑀅𑀭𑀽𑀧𑁂 𑀤𑁄𑀲𑀫𑀽𑀮𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇) (𑁧)

3828\. 𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 (𑀏𑀯𑀁 𑀦𑀯𑀧𑀺 𑀧𑀜𑁆𑀳𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸), 𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸, 𑀦𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑁇
