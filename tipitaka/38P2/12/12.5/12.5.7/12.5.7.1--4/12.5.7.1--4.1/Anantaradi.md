# 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

### 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀤𑀼𑀓

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

(79.)

4013\. 𑀓𑀺𑀮𑁂𑀲𑁄 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇 (𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀸𑁇)

4014\. 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲…  𑀆𑀯𑀚𑁆𑀚𑀦𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

4015\. (𑀇𑀢𑀭𑁂 𑀤𑁆𑀯𑁂 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀸, 𑀦𑀺𑀦𑁆𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀸𑁇 𑀖𑀝𑀦𑀸𑀦𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀺 𑀲𑀩𑁆𑀩𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀸, 𑀦𑀺𑀦𑁆𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀸𑁇 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀦𑀢𑁆𑀣𑀺, 𑀇𑀤𑀁 𑀤𑀼𑀓𑀁 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀁, 𑀦𑀺𑀦𑁆𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁𑁇)

---

4016\. 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀤𑀼𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
