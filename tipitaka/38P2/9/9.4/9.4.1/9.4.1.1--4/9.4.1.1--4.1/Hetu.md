# 𑀳𑁂𑀢𑀼

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀤𑀼𑀓

### 𑀳𑁂𑀢𑀼

(86.)

2297\. 𑀧𑀭𑀸𑀫𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁧)

2298\. 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 (𑀬𑀸𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑁇) (𑁧)

2299\. 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄𑁇 (𑁨)

2300\. 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

2301\. 𑀧𑀭𑀸𑀫𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)
