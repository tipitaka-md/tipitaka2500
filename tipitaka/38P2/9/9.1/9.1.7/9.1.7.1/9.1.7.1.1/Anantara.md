### 𑀅𑀦𑀦𑁆𑀢𑀭

(23.)

2169. 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(24.)

2170. 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀆𑀯𑀚𑁆𑀚𑀦𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀆𑀯𑀚𑁆𑀚𑀦𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(25.)

2171. 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀼𑀭𑀺𑀫𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
