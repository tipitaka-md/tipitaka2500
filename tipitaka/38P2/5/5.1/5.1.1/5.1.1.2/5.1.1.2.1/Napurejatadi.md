1630\. 𑀕𑀦𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀕𑀦𑁆𑀣𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀁 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 (𑀅𑀭𑀽𑀧𑁂 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀦𑀢𑁆𑀣𑀺, 𑀏𑀯𑀁 𑀦𑀯 𑀧𑀜𑁆𑀳𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸), 𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀦𑀯, 𑀦𑀆𑀲𑁂𑀯𑀦𑁂 𑀦𑀯, 𑀦𑀓𑀫𑁆𑀫𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀯𑀺𑀧𑀸𑀓𑁂 𑀦𑀯, 𑀦𑀆𑀳𑀸𑀭𑁂 𑀏𑀓𑀁, 𑀦𑁃𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀏𑀓𑀁, 𑀦𑀛𑀸𑀦𑁂 𑀏𑀓𑀁, 𑀦𑀫𑀕𑁆𑀕𑁂 𑀏𑀓𑀁, 𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀦𑀯, 𑀦𑁄𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇
