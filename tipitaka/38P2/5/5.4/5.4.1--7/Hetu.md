# 𑀳𑁂𑀢𑀼

### 𑀕𑀦𑁆𑀣𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀕𑀦𑁆𑀣𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀤𑀼𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(103.)

1842\. 𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄, 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄𑁇 (𑁧)

1843\. 𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀕𑀦𑁆𑀣𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁨)

1844\. 𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)
