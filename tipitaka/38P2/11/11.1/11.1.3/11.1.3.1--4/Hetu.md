# 𑀳𑁂𑀢𑀼

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀤𑀼𑀓

* 𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

* 𑀳𑁂𑀢𑀼

(12.)

3549\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁…  𑀢𑀻𑀡𑀺𑁇 (𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀤𑀺𑀲𑀸𑁇)

3550\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀢𑁆𑀣𑀼…𑀧𑁂…  (𑀬𑀸𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀫𑀳𑀸𑀪𑀽𑀢𑀸) 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁧)

3551\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑁇 (𑁨)

3552\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁩)

(13.)

3553\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁 (𑀘𑀓𑁆𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁)𑁇 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁𑁇 (𑀘𑀓𑁆𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) (𑁧)

3554\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  (𑀘𑀓𑁆𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁)𑁇 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

3555\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  (𑀘𑀓𑁆𑀓𑀁)𑁇 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂… 𑁇 (𑀘𑀓𑁆𑀓𑀁𑁇) (𑁩)

3556\. 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸… 𑁇 (𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀫𑀽𑀮𑀓𑁂 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)

3557\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯), 𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

3558\. 𑀅𑀦𑀼𑀮𑁄𑀫𑀁𑁇



(14.)

3559\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀅𑀳𑁂𑀢𑀼𑀓𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  (𑀬𑀸𑀯 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸) 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀳𑁂𑀢𑀼𑀓𑀸 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁂 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑁄 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀫𑁄𑀳𑁄𑁇

3560\. 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯, 𑀦𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀦𑀯…𑀧𑁂…  𑀦𑀓𑀫𑁆𑀫𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀯𑀺𑀧𑀸𑀓𑁂 𑀦𑀯 (𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀤𑀺𑀲𑀁)…𑀧𑁂…  𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇
