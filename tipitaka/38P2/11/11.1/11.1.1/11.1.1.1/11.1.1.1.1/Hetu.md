# 𑀳𑁂𑀢𑀼

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀤𑀼𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(1.)

3527. 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀁𑁇 (𑁧)

3528. 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁨)

3529. 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀁 𑀘𑀓𑁆𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) (𑁩)

(2.)

3530. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀢𑁆𑀣𑀼, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀔𑀦𑁆𑀥𑀸, 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀁𑁇 (𑁧)

3531. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑁇 (𑁨)

3532. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁩)

(3.)

3533. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑁂 𑀘 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁𑁇 (𑀲𑀩𑁆𑀩𑁂 𑀘𑀓𑁆𑀓𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇) (𑁧)

3534. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁨)

3535. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘…𑀧𑁂… 𑁇 (𑀘𑀓𑁆𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) (𑁩)
