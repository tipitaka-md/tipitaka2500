* 𑀉𑀧𑀸𑀤𑀸𑀦𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀤𑀼𑀓

* 𑀳𑁂𑀢𑀼

(98.)

3768\. 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀘 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

3769\. 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀘 𑀳𑁂𑀢𑀽 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀉𑀧𑀸𑀤𑀸𑀦𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀸 𑀦𑀺𑀦𑁆𑀦𑀸𑀦𑀸, 𑀦𑀯 𑀧𑀜𑁆𑀳𑀸𑁇)
