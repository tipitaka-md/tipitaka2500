# 𑁧𑁧𑁇𑁫𑁇𑁨--𑁬 𑀲𑀳𑀚𑀸𑀢-𑀧𑀘𑁆𑀘𑀬-𑀦𑀺𑀲𑁆𑀲𑀬-𑀲𑀁𑀲𑀝𑁆𑀞-𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭

* 𑀉𑀧𑀸𑀤𑀸𑀦𑁅𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

3780\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀆𑀫𑀲𑀦𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀁 𑀦𑀯 𑀧𑀜𑁆𑀳𑀸𑁇 𑀭𑀽𑀧𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀯𑀸𑀭𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀭𑀽𑀧𑀁𑀬𑁂𑀯𑁇)
