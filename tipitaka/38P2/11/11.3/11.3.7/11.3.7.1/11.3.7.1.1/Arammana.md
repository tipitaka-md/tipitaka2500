# 𑀆𑀭𑀫𑁆𑀫𑀡

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀆𑀭𑀫𑁆𑀫𑀡

(66.)

3698\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀮𑁄𑀪𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑀩𑁆𑀪 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀮𑁄𑀪𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)

(67.)

3699\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀸𑀦𑀺…𑀧𑁂…  𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀛𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀛𑀸𑀦𑁂 𑀧𑀭𑀺𑀳𑀻𑀦𑁂 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀲𑁆𑀲 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀨𑀮𑀁…𑀧𑁂…  𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲, 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲, 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀭𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀧𑀳𑀻𑀦𑁂 𑀓𑀺𑀮𑁂𑀲𑁂…𑀧𑁂…  𑀯𑀺𑀓𑁆𑀔𑀫𑁆𑀪𑀺𑀢𑁂 𑀓𑀺𑀮𑁂𑀲𑁂…𑀧𑁂…  𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀫𑀼𑀤𑀸𑀘𑀺𑀡𑁆𑀡𑁂 𑀓𑀺𑀮𑁂𑀲𑁂…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄…𑀧𑁂…  𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  (𑀲𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁), 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸…  (𑀬𑀸𑀯 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁), 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀲𑁆𑀲, 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

3700\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨)

3701\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀆𑀭𑀩𑁆𑀪 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀮𑁄𑀪𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)

(68.)

3702\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀆𑀭𑀩𑁆𑀪 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀆𑀭𑀩𑁆𑀪 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀮𑁄𑀪𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀮𑁄𑀪𑀜𑁆𑀘 𑀆𑀭𑀩𑁆𑀪 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀮𑁄𑀪𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀮𑁄𑀪𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)
