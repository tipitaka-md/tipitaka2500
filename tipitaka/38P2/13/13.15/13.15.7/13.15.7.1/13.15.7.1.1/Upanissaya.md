# 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(318.)

4752\. 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀧𑀞𑀫𑁄 𑀫𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄…𑀧𑁂…  𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬…𑀧𑁂…  𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬 𑀞𑀸𑀦𑀸𑀞𑀸𑀦𑀓𑁄𑀲𑀮𑁆𑀮𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

(319.)

4753\. 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀛𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺…  𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁…  𑀅𑀪𑀺𑀜𑁆𑀜𑀁…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺…  𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺…  𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…  𑀭𑀸𑀕𑀁…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲…  𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀧𑀞𑀫𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀧𑀞𑀫𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)
