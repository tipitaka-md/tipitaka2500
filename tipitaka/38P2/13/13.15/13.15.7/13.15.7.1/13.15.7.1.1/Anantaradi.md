# 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

(317.)

4749\. 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀫𑀕𑁆𑀕𑁄 𑀨𑀮𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

4750\. 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲…  𑀨𑀮𑀁 𑀨𑀮𑀲𑁆𑀲…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸…  𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀕𑁄𑀢𑁆𑀭𑀪𑀼 𑀫𑀕𑁆𑀕𑀲𑁆𑀲; 𑀯𑁄𑀤𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

4751\. 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀧𑀜𑁆𑀘…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀤𑁆𑀯𑁂…  𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀲𑀢𑁆𑀢𑁇
