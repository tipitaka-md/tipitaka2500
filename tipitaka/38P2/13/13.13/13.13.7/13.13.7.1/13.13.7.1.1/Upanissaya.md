# 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(285.)

4688\. 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲…𑀧𑁂…  𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀫𑀕𑁆𑀕𑀁…𑀧𑁂…  𑀅𑀪𑀺𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺…  𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺…  𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬…  𑀭𑀸𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲…  𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

(286.)

4689\. 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀫𑀕𑁆𑀕𑀁…𑀧𑁂…  𑀅𑀪𑀺𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺…  𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀭𑀸𑀕𑀁…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺…  𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲, 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

4690\. 𑀦𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)
