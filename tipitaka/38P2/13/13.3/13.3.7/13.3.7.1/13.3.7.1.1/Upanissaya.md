# 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

### 𑀤𑀲𑁆𑀲𑀦𑁂𑀦𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

### 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(78.)

4240\. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀅𑀯𑀲𑁂𑀲𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄𑁇) (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(79.)

4241\. 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺; 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…  𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁 𑀭𑀸𑀕𑀁…  𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀫𑀸𑀦𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺; 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

4242\. 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…  𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁 𑀭𑀸𑀕𑀁…  𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀫𑀸𑀦𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

4243\. 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸, 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀭𑀸𑀕𑁄…  𑀤𑁄𑀲𑁄…  𑀫𑁄𑀳𑁄…  𑀫𑀸𑀦𑁄…  𑀧𑀢𑁆𑀣𑀦𑀸…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(80.)

4244\. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀫𑁄𑀳𑁄 𑀘 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀫𑁄𑀳𑁄 𑀘 𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀫𑁄𑀳𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
