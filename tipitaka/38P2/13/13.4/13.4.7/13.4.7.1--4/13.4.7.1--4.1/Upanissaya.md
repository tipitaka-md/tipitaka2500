# 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

* 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀪𑀸𑀯𑀦𑀸𑀬𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(99.)

4279\. 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀲𑀓𑀪𑀡𑁆𑀟𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀭𑀪𑀡𑁆𑀟𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀲𑀓𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀭𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(100.)

4280\. 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…  𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁 𑀭𑀸𑀕𑀁…  𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀤𑀺𑀝𑁆𑀞𑀺𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬…  𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲…𑀧𑁂…  𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

4281\. 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀫𑀸𑀦𑀲𑁆𑀲…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

4282\. 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀫𑁄𑀳𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀖𑀝𑀦𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀸𑀧𑀺 𑀢𑀻𑀡𑀺𑀧𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇) (𑁩)
