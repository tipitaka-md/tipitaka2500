# 𑀓𑀫𑁆𑀫𑀸𑀤𑀺

### 𑀓𑀫𑁆𑀫𑀸𑀤𑀺

(255.)

4633. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀘 𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

4634. 𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂… 𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄… 𑁇 (𑁧)

4635. 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺…  𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺…  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺…  𑀛𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺…  𑀫𑀕𑁆𑀕𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀤𑁆𑀯𑁂𑁇
