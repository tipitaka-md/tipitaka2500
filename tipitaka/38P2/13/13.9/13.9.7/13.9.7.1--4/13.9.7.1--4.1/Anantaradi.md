# 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀸𑀤𑀺

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀤𑀺

(171.)

4449\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀘𑀼𑀢𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀪𑀯𑀗𑁆𑀕𑀁 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀧𑀸𑀓𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀓𑀺𑀭𑀺𑀬𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀪𑀯𑀗𑁆𑀕𑀁 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀪𑀯𑀗𑁆𑀕𑀲𑁆𑀲…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲…  𑀓𑀺𑀭𑀺𑀬𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲…  𑀨𑀮𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

4450\. 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸…𑀧𑁂… 𑁇 (𑀫𑀽𑀮𑀁𑁇 𑀢𑀻𑀡𑀺𑀧𑀺 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀸𑁇)

(172.)

4451\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀘 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀘𑀼𑀢𑀺𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀪𑀯𑀗𑁆𑀕𑀜𑁆𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀯𑀺𑀧𑀸𑀓𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀺𑀭𑀺𑀬𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀪𑀯𑀗𑁆𑀕𑀜𑁆𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀪𑀯𑀗𑁆𑀕𑀲𑁆𑀲…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲…  𑀓𑀺𑀭𑀺𑀬𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲…  𑀨𑀮𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀫𑀽𑀮𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇) 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀲𑀼𑀔𑀜𑁆𑀘 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

4452\. 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇
