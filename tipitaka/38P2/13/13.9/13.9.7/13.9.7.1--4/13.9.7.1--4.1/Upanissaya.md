### 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(173.)

4453\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

4454\. 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…  𑀭𑀸𑀕𑀁…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀼𑀔𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺; 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀼𑀔𑀜𑁆𑀘 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬…  𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲, 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

4455\. 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…  𑀭𑀸𑀕𑀁…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀼𑀔𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺; 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺; 𑀫𑀼𑀲𑀸…𑀧𑁂…  𑀧𑀺𑀲𑀼𑀡𑀁…𑀧𑁂…  𑀲𑀫𑁆𑀨𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀥𑀺𑀁…𑀧𑁂…  𑀦𑀺𑀮𑁆𑀮𑁄𑀧𑀁…𑀧𑁂…  𑀏𑀓𑀸𑀕𑀸𑀭𑀺𑀓𑀁…𑀧𑁂…  𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑁂…𑀧𑁂…  𑀧𑀭𑀤𑀸𑀭𑀁…𑀧𑁂…  𑀕𑀸𑀫𑀖𑀸𑀢𑀁…𑀧𑁂…  𑀦𑀺𑀕𑀫𑀖𑀸𑀢𑀁 𑀓𑀭𑁄𑀢𑀺; 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀼𑀔𑀜𑁆𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬…  𑀭𑀸𑀕𑀲𑁆𑀲…  𑀫𑀸𑀦𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲…  𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

4456\. 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  (𑀤𑀼𑀢𑀺𑀬𑀕𑀫𑀦𑀲𑀤𑀺𑀲𑀁) 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀼𑀔𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺…𑀧𑁂…  𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀼𑀔𑀜𑁆𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀬…  𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬…  𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

4457\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀘 𑀦𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇
