# 𑁧𑁦𑁇𑁧𑁇𑁭 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀏𑀓𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑀼𑀢𑁆𑀢

(983.)

1665\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀁 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

1666\. “𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸”𑀢𑀺? “𑀬𑀤𑀸𑀧𑀺 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀏𑀓𑀁 𑀯𑀸 𑀭𑀳𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁, 𑀢𑀤𑀸𑀧𑀺 𑀫𑀬𑀁 𑀢𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫 𑀓𑀸𑀬𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸”𑀢𑀺𑁇

1667\. “𑀬𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀦𑁂𑀯 𑀓𑀸𑀬𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸, 𑀢𑀲𑁆𑀲 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀓𑀢𑀫𑀲𑁆𑀲 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀦𑁂𑀯 𑀓𑀸𑀬𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸?

1668\. 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀦𑁂𑀯 𑀓𑀸𑀬𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸𑁇 𑀓𑀣𑀁 𑀪𑀸𑀯𑀺𑀢𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀺 𑀓𑀣𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀦𑁂𑀯 𑀓𑀸𑀬𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸?

1669\. 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇 𑀲𑁄 𑀲𑀢𑁄𑀯 𑀅𑀲𑁆𑀲𑀲𑀢𑀺, 𑀲𑀢𑁄𑀯 𑀧𑀲𑁆𑀲𑀲𑀢𑀺…𑀧𑁂…  ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁂 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀦𑁂𑀯 𑀓𑀸𑀬𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀇𑀜𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀨𑀦𑁆𑀤𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸”𑀢𑀺𑁇
