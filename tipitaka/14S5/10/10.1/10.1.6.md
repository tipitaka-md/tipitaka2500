# 𑁧𑁦𑁇𑁧𑁇𑁬 𑀅𑀭𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

### 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀭𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

(982.)

1661\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀸𑀯𑁂𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀦𑁆”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀸𑀯𑁂𑀫𑀺 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀪𑀸𑀯𑁂𑀲𑀺 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀦𑁆”𑀢𑀺? “𑀅𑀢𑀻𑀢𑁂𑀲𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀅𑀦𑀸𑀕𑀢𑁂𑀲𑀼 𑀫𑁂 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀯𑀺𑀕𑀢𑁄, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘 𑀫𑁂 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀦𑀻𑀢𑀸𑁇 𑀲𑁄 𑀲𑀢𑁄𑀯 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀢𑁄𑀯 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀏𑀯𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀸𑀯𑁂𑀫𑀺 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀦𑁆”𑀢𑀺𑁇

1662\. “‘𑀅𑀢𑁆𑀣𑁂𑀲𑀸, 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺, 𑀦𑁂𑀲𑀸 𑀦𑀢𑁆𑀣𑀻’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀧𑀺 𑀘, 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀬𑀣𑀸 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸 𑀳𑁄𑀢𑀺 𑀢𑀁 𑀲𑀼𑀡𑀸𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1663\. “𑀓𑀣𑀜𑁆𑀘, 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇 𑀲𑁄 𑀲𑀢𑁄𑀯 𑀅𑀲𑁆𑀲𑀲𑀢𑀺, 𑀲𑀢𑁄𑀯 𑀧𑀲𑁆𑀲𑀲𑀢𑀺𑁇 𑀤𑀻𑀖𑀁 𑀯𑀸 𑀅𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀤𑀻𑀖𑀁 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀅𑀭𑀺𑀝𑁆𑀞, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
