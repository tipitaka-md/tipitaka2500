# 𑁧𑁦𑁇𑁧𑁇𑁯 𑀯𑁂𑀲𑀸𑀮𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀏𑀓𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀯𑁂𑀲𑀸𑀮𑀻𑀲𑀼𑀢𑁆𑀢

(985.)

1694\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀲𑀼𑀪𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀅𑀲𑀼𑀪𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇

1695\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀟𑁆𑀠𑀫𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁𑁇 𑀦𑀸𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑁆𑀲𑀼𑀥 𑀓𑁄𑀘𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑁇

1696\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀲𑀼𑀪𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀅𑀲𑀼𑀪𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀻”𑀢𑀺 𑀅𑀦𑁂𑀓𑀸𑀓𑀸𑀭𑀯𑁄𑀓𑀸𑀭𑀁 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀇𑀫𑀺𑀦𑀸 𑀓𑀸𑀬𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑀸 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑀸 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑀸 𑀲𑀢𑁆𑀣𑀳𑀸𑀭𑀓𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺𑁇 𑀤𑀲𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀲𑀢𑁆𑀣𑀁 𑀆𑀳𑀭𑀦𑁆𑀢𑀺, 𑀯𑀻𑀲𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀢𑀺𑀁𑀲𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀲𑀢𑁆𑀣𑀁 𑀆𑀳𑀭𑀦𑁆𑀢𑀺𑁇

1697\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀲𑁆𑀲 𑀅𑀟𑁆𑀠𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀦𑀼𑀪𑀽𑀢𑁄 𑀯𑀺𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄”𑀢𑀺? “𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, ‘𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀲𑀼𑀪𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀅𑀲𑀼𑀪𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀻’𑀢𑀺 𑀅𑀦𑁂𑀓𑀸𑀓𑀸𑀭𑀯𑁄𑀓𑀸𑀭𑀁 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀇𑀫𑀺𑀦𑀸 𑀓𑀸𑀬𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑀸 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑀸 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑀸 𑀲𑀢𑁆𑀣𑀳𑀸𑀭𑀓𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺𑁇 𑀤𑀲𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀲𑀢𑁆𑀣𑀁 𑀆𑀳𑀭𑀦𑁆𑀢𑀺, 𑀯𑀻𑀲𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽…  𑀢𑀺𑀁𑀲𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀲𑀢𑁆𑀣𑀁 𑀆𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀼 𑀬𑀣𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀅𑀜𑁆𑀜𑀸𑀬 𑀲𑀡𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

1698\. “𑀢𑁂𑀦𑀳𑀸𑀦𑀦𑁆𑀤, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄𑁇 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

1699\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀦𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀡𑀻𑀢𑁄 𑀘 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀘 𑀲𑀼𑀔𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀘 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺𑁇

1700\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀊𑀳𑀢𑀁 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁, 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸𑀅𑀓𑀸𑀮𑀫𑁂𑀖𑁄 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀦𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀡𑀻𑀢𑁄 𑀘 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀘 𑀲𑀼𑀔𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀘 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺𑁇 𑀓𑀣𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀓𑀣𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀦𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀡𑀻𑀢𑁄 𑀘 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀘 𑀲𑀼𑀔𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀘 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺?

1701\. 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇 𑀲𑁄 𑀲𑀢𑁄𑀯 𑀅𑀲𑁆𑀲𑀲𑀢𑀺, 𑀲𑀢𑁄𑀯 𑀧𑀲𑁆𑀲𑀲𑀢𑀺…𑀧𑁂…  ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀦𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀡𑀻𑀢𑁄 𑀘 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀘 𑀲𑀼𑀔𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀘 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀻”𑀢𑀺𑁇
