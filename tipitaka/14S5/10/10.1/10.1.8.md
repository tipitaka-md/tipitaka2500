# 𑁧𑁦𑁇𑁧𑁇𑁮 𑀧𑀤𑀻𑀧𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀏𑀓𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀤𑀻𑀧𑁄𑀧𑀫𑀲𑀼𑀢𑁆𑀢

(984.)

1671\. “𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄𑁇 𑀓𑀣𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀓𑀣𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄?

1672\. 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇 𑀲𑁄 𑀲𑀢𑁄𑀯 𑀅𑀲𑁆𑀲𑀲𑀢𑀺, 𑀲𑀢𑁄𑀯 𑀧𑀲𑁆𑀲𑀲𑀢𑀺𑁇 𑀤𑀻𑀖𑀁 𑀯𑀸 𑀅𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 ‘𑀤𑀻𑀖𑀁 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, ‘𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄𑁇

1673\. 𑀅𑀳𑀫𑁆𑀧𑀺 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀺𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀺𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑁂𑀯 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀢𑀺 𑀦 𑀘𑀓𑁆𑀔𑀽𑀦𑀺; 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀘 𑀫𑁂 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑁇

1674\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀦𑁂𑀯 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬 𑀦 𑀘𑀓𑁆𑀔𑀽𑀦𑀺, 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀘 𑀫𑁂 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬𑀸’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1675\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀬𑁂 𑀫𑁂 𑀕𑁂𑀳𑀲𑀺𑀢𑀸 𑀲𑀭𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀢𑁂 𑀧𑀳𑀻𑀬𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1676\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1677\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1678\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1679\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1680\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀁 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1681\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1682\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1683\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀁 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑀁, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1684\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1685\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1686\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1687\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1688\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1689\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀧𑀺 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬—  ‘𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁂𑀯 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

1690\. 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂, 𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀸 ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺; 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀸 ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺; 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, ‘𑀲𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1691\. 𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺; 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺; 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀉𑀤𑁆𑀥𑀁 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀇𑀥𑁂𑀯 𑀲𑀩𑁆𑀩𑀯𑁂𑀤𑀬𑀺𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀲𑀻𑀢𑀻𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1692\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘, 𑀯𑀝𑁆𑀝𑀺𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁄 𑀛𑀸𑀬𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑁂𑀯 𑀢𑁂𑀮𑀲𑁆𑀲 𑀘 𑀯𑀝𑁆𑀝𑀺𑀬𑀸 𑀘 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀅𑀦𑀸𑀳𑀸𑀭𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀉𑀤𑁆𑀥𑀁 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀇𑀥𑁂𑀯 𑀲𑀩𑁆𑀩𑀯𑁂𑀤𑀬𑀺𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀲𑀻𑀢𑀻𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
