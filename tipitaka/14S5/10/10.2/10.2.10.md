# 𑁧𑁦𑁇𑁨𑁇𑁧𑁦 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

(996.)

1785\. “… 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀣𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀓𑀣𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀦𑀼𑀲𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀤𑁆𑀥𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸…𑀧𑁂…  𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀼𑀲𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…𑀧𑁂…  𑀅𑀤𑁆𑀥𑀸𑀦𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…𑀧𑁂…  𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
