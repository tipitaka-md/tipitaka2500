# 𑁧𑁦𑁇𑁨𑁇𑁨 𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

(988.)

1721. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑁄𑀫𑀲𑀓𑀁𑀪𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀮𑁄𑀫𑀲𑀓𑀁𑀪𑀺𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀮𑁄𑀫𑀲𑀓𑀁𑀪𑀺𑀬𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀮𑁄𑀫𑀲𑀓𑀁𑀪𑀺𑀬𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑁄 𑀏𑀯 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁂𑀔𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀜𑁆𑀜𑁄𑀯 𑀲𑁂𑀔𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀜𑁆𑀜𑁄 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺?

1722. “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀲𑁆𑀯𑁂𑀯 𑀲𑁂𑀔𑁄 𑀯𑀺𑀳𑀸𑀭𑁄, 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄𑁇 𑀅𑀜𑁆𑀜𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀲𑁂𑀔𑁄 𑀯𑀺𑀳𑀸𑀭𑁄, 𑀅𑀜𑁆𑀜𑁄 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄𑁇 𑀬𑁂 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑁂𑀔𑀸 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀳𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀳𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁…𑀧𑁂…  𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁…  𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀳𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇

1723. 𑀬𑁂𑀧𑀺 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑁂𑀔𑀸 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀳𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇

1724. 𑀬𑁂 𑀘 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀯𑀼𑀲𑀺𑀢𑀯𑀦𑁆𑀢𑁄 𑀓𑀢𑀓𑀭𑀡𑀻𑀬𑀸 𑀑𑀳𑀺𑀢𑀪𑀸𑀭𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀤𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀪𑀯𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀲𑀁 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀳𑀻𑀦𑀁 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀁 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀁 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀁 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀁; 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀳𑀻𑀦𑀁…𑀧𑁂…  𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁…  𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀧𑀳𑀻𑀦𑀁 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀁 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀁 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀁 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀁𑁇

1725. 𑀬𑁂 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀯𑀼𑀲𑀺𑀢𑀯𑀦𑁆𑀢𑁄 𑀓𑀢𑀓𑀭𑀡𑀻𑀬𑀸 𑀑𑀳𑀺𑀢𑀪𑀸𑀭𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀤𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀪𑀯𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀲𑀁 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀤𑀫𑀺𑀦𑀸𑀧𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀬𑀣𑀸—  𑀅𑀜𑁆𑀜𑁄𑀯 𑀲𑁂𑀔𑁄 𑀯𑀺𑀳𑀸𑀭𑁄, 𑀅𑀜𑁆𑀜𑁄 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄𑁇

1726. 𑀏𑀓𑀫𑀺𑀤𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁𑁇 𑀦𑀸𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑁆𑀲𑀼𑀥 𑀓𑁄𑀘𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑁇

1727. 𑀅𑀣 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀢𑀲𑁆𑀲 𑀢𑁂𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀲𑀘𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  “𑀓𑀢𑀫𑁂𑀦𑀸𑀯𑀼𑀲𑁄, 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀸𑀲𑀻”𑀢𑀺, 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀣—  “𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀸𑀲𑀻”’𑀢𑀺𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁄 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀺, 𑀲𑀢𑁄 𑀧𑀲𑁆𑀲𑀲𑀸𑀫𑀺𑁇 𑀤𑀻𑀖𑀁 𑀅𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 𑀤𑀻𑀖𑀁 𑀅𑀲𑁆𑀲𑀲𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀺, 𑀤𑀻𑀖𑀁 𑀧𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 𑀤𑀻𑀖𑀁 𑀧𑀲𑁆𑀲𑀲𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀺…𑀧𑁂…  𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀅𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀺, 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀧𑀲𑁆𑀲𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀺𑁇

1728. 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  𑀅𑀭𑀺𑀬𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺𑁇 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  𑀅𑀭𑀺𑀬𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺𑁇

1729. 𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑁂𑀔𑀸 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

1730. 𑀬𑁂 𑀘 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀯𑀼𑀲𑀺𑀢𑀯𑀦𑁆𑀢𑁄 𑀓𑀢𑀓𑀭𑀡𑀻𑀬𑀸 𑀑𑀳𑀺𑀢𑀪𑀸𑀭𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀤𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀪𑀯𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀲𑀁 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀘𑁂𑀯 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀸𑀬 𑀘𑁇

1731. 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  𑀅𑀭𑀺𑀬𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺𑁇 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  𑀅𑀭𑀺𑀬𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀻𑀢𑀺𑁇 𑀇𑀫𑀺𑀦𑀸 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑀁, 𑀬𑀣𑀸—  𑀅𑀜𑁆𑀜𑁄𑀯 𑀲𑁂𑀔𑁄 𑀯𑀺𑀳𑀸𑀭𑁄, 𑀅𑀜𑁆𑀜𑁄 𑀢𑀣𑀸𑀕𑀢𑀯𑀺𑀳𑀸𑀭𑁄”𑀢𑀺𑁇

---
