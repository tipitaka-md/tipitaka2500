# 𑁧𑁦𑁇𑁨𑁇𑁫 𑀧𑀞𑀫𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(991.)

1756. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1757. “𑀓𑀢𑀫𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻𑀢𑀺𑁇

1758. 𑀓𑀣𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀓𑀣𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸…𑀧𑁂…  𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
