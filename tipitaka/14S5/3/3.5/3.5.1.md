# 𑁩𑁇𑁫𑁇𑁧 𑀅𑀫𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀫𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀫𑀢𑀲𑀼𑀢𑁆𑀢

(407.)

939. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑀽𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀯𑁄 𑀅𑀫𑀢𑀁 𑀧𑀦𑀲𑁆𑀲𑁇 𑀓𑀢𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁; 𑀯𑁂𑀤𑀦𑀸𑀲𑀼…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀇𑀫𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀲𑀼 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀯𑁄 𑀅𑀫𑀢𑀁 𑀧𑀦𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
