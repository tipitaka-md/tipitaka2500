# 𑁩𑁇𑁫𑁇𑁮 𑀫𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀫𑀢𑀯𑀕𑁆𑀕

### 𑀫𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(414.)

963\. “𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀸𑀣, 𑀬𑁂 𑀘 𑀔𑁄 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸, 𑀢𑁂 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

964\. 𑀓𑀢𑀫𑁂𑀲𑀁, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁; 𑀯𑁂𑀤𑀦𑀸𑀲𑀼…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀸𑀣, 𑀬𑁂 𑀘 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸, 𑀢𑁂 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
