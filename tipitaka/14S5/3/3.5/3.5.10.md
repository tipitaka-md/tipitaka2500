# 𑁩𑁇𑁫𑁇𑁧𑁦 𑀆𑀲𑀯𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀫𑀢𑀯𑀕𑁆𑀕

### 𑀆𑀲𑀯𑀲𑀼𑀢𑁆𑀢

(416.)

969\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂 𑀆𑀲𑀯𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀓𑀸𑀫𑀸𑀲𑀯𑁄, 𑀪𑀯𑀸𑀲𑀯𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀆𑀲𑀯𑀸𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇

970\. 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁; 𑀯𑁂𑀤𑀦𑀸𑀲𑀼…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
