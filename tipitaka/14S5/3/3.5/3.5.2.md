# 𑁩𑁇𑁫𑁇𑁨 𑀲𑀫𑀼𑀤𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀫𑀢𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀼𑀤𑀬𑀲𑀼𑀢𑁆𑀢

(408.)

941\. “𑀘𑀢𑀼𑀦𑁆𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄? 𑀆𑀳𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄; 𑀆𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑁇 𑀨𑀲𑁆𑀲𑀲𑀫𑀼𑀤𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑁄; 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑁇 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑀫𑀼𑀤𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄; 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑁇 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑁄; 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇
