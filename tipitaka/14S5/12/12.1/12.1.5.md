# 𑁧𑁨𑁇𑁧𑁇𑁫 𑀧𑀞𑀫𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀫𑀸𑀥𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(1075.)

2180\. “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀺𑀁𑀲𑀼, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

2181\. 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺? 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀺𑀁𑀲𑀼…𑀧𑁂…  𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

2182\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇

---
