# 𑁧𑁨𑁇𑁧𑁇𑁮 𑀘𑀺𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀫𑀸𑀥𑀺𑀯𑀕𑁆𑀕

### 𑀘𑀺𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(1078.)

2192\. “𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀧𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀸𑀣—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸 ‘𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸, ‘𑀅𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄’𑀢𑀺 𑀯𑀸, ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆’𑀢𑀺 𑀯𑀸 ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆’𑀢𑀺 𑀯𑀸, ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸 ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸, ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸, ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺 𑀯𑀸𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀦𑁂𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀸 𑀦𑀸𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀓𑀸 𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀦 𑀉𑀧𑀲𑀫𑀸𑀬 𑀦 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀦 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

2193\. 𑀘𑀺𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀸 𑀘 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀘𑀺𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀸𑀣, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀘𑀺𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀸𑀣, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀘𑀺𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀸𑀣, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀘𑀺𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀸, 𑀏𑀲𑀸 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀓𑀸, 𑀏𑀲𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

2194\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
