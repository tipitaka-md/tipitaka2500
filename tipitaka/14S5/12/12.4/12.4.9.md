# 𑁧𑁨𑁇𑁪𑁇𑁯 𑀇𑀦𑁆𑀤𑀔𑀻𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀻𑀲𑀧𑀸𑀯𑀦𑀯𑀕𑁆𑀕

* 𑀇𑀦𑁆𑀤𑀔𑀻𑀮𑀲𑀼𑀢𑁆𑀢

(1109.)

2342\. “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀫𑀼𑀔𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀦𑀽𑀦 𑀪𑀯𑀁 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

2343\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀽𑀮𑀧𑀺𑀘𑀼 𑀯𑀸 𑀓𑀧𑁆𑀧𑀸𑀲𑀧𑀺𑀘𑀼 𑀯𑀸 𑀮𑀳𑀼𑀓𑁄 𑀯𑀸𑀢𑀽𑀧𑀸𑀤𑀸𑀦𑁄 𑀲𑀫𑁂 𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁄 𑀯𑀸𑀢𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬, 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀯𑀸𑀢𑁄 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂𑀦 𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬, 𑀉𑀢𑁆𑀢𑀭𑁄 𑀯𑀸𑀢𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬, 𑀤𑀓𑁆𑀔𑀺𑀡𑁄 𑀯𑀸𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀮𑀳𑀼𑀓𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀧𑁆𑀧𑀸𑀲𑀧𑀺𑀘𑀼𑀦𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀫𑀼𑀔𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀦𑀽𑀦 𑀪𑀯𑀁 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀤𑀺𑀝𑁆𑀞𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁𑁇

2344\. 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀦 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀫𑀼𑀔𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀦𑀽𑀦 𑀪𑀯𑀁 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

2345\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀬𑁄𑀔𑀻𑀮𑁄 𑀯𑀸 𑀇𑀦𑁆𑀤𑀔𑀻𑀮𑁄 𑀯𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑀦𑁂𑀫𑁄 𑀲𑀼𑀦𑀺𑀔𑀸𑀢𑁄 𑀅𑀘𑀮𑁄 𑀅𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀻𑁇 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀪𑀼𑀲𑀸 𑀯𑀸𑀢𑀯𑀼𑀝𑁆𑀞𑀺, 𑀦𑁂𑀯 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀬𑁆𑀬; 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀪𑀼𑀲𑀸 𑀯𑀸𑀢𑀯𑀼𑀝𑁆𑀞𑀺, 𑀦𑁂𑀯 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀕𑀫𑁆𑀪𑀻𑀭𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀫𑀲𑁆𑀲 𑀲𑀼𑀦𑀺𑀔𑀸𑀢𑀢𑁆𑀢𑀸 𑀇𑀦𑁆𑀤𑀔𑀻𑀮𑀲𑁆𑀲𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀦 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀫𑀼𑀔𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀦𑀽𑀦 𑀪𑀯𑀁 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲𑁇

2346\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
