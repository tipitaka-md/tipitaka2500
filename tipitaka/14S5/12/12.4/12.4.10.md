# 𑁧𑁨𑁇𑁪𑁇𑁧𑁦 𑀯𑀸𑀤𑀢𑁆𑀣𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀻𑀲𑀧𑀸𑀯𑀦𑀯𑀕𑁆𑀕

* 𑀯𑀸𑀤𑀢𑁆𑀣𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(1110.)

2348\. “𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀯𑀸𑀤𑀢𑁆𑀣𑀺𑀓𑁄 𑀯𑀸𑀤𑀕𑀯𑁂𑀲𑀻—  ‘𑀯𑀸𑀤𑀫𑀲𑁆𑀲 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺, 𑀢𑀁 𑀯𑀢 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀯𑀸𑀤𑀢𑁆𑀣𑀺𑀓𑁄 𑀯𑀸𑀤𑀕𑀯𑁂𑀲𑀻—  ‘𑀯𑀸𑀤𑀫𑀲𑁆𑀲 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺, 𑀢𑀁 𑀯𑀢 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

2349\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀮𑀸𑀬𑀽𑀧𑁄 𑀲𑁄𑀍𑀅𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀓𑁄𑁇 𑀢𑀲𑁆𑀲𑀲𑁆𑀲𑀼 𑀅𑀝𑁆𑀞 𑀓𑀼𑀓𑁆𑀓𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀦𑁂𑀫𑀗𑁆𑀕𑀫𑀸, 𑀅𑀝𑁆𑀞 𑀓𑀼𑀓𑁆𑀓𑀼 𑀉𑀧𑀭𑀺𑀦𑁂𑀫𑀲𑁆𑀲𑁇 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀪𑀼𑀲𑀸 𑀯𑀸𑀢𑀯𑀼𑀝𑁆𑀞𑀺, 𑀦𑁂𑀯 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀬𑁆𑀬; 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀪𑀼𑀲𑀸 𑀯𑀸𑀢𑀯𑀼𑀝𑁆𑀞𑀺, 𑀦𑁂𑀯 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀦 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀕𑀫𑁆𑀪𑀻𑀭𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀫𑀲𑁆𑀲 𑀲𑀼𑀦𑀺𑀔𑀸𑀢𑀢𑁆𑀢𑀸 𑀲𑀺𑀮𑀸𑀬𑀽𑀧𑀲𑁆𑀲𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺; 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀯𑀸𑀤𑀢𑁆𑀣𑀺𑀓𑁄 𑀯𑀸𑀤𑀕𑀯𑁂𑀲𑀻 ‘𑀯𑀸𑀤𑀫𑀲𑁆𑀲 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺, 𑀢𑀁 𑀯𑀢 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀘𑁂𑀧𑀺 𑀤𑀺𑀲𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀯𑀸𑀤𑀢𑁆𑀣𑀺𑀓𑁄 𑀯𑀸𑀤𑀕𑀯𑁂𑀲𑀻—  ‘𑀯𑀸𑀤𑀫𑀲𑁆𑀲 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺, 𑀢𑀁 𑀯𑀢 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀧𑀘𑀸𑀮𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀸𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲𑁇

2350\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
