# 𑁧𑁨𑁇𑁪𑁇𑁪 𑀘𑁂𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀻𑀲𑀧𑀸𑀯𑀦𑀯𑀕𑁆𑀕

* 𑀘𑁂𑀮𑀲𑀼𑀢𑁆𑀢

(1104.)

2322\. “𑀆𑀤𑀺𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀮𑁂 𑀯𑀸 𑀲𑀻𑀲𑁂 𑀯𑀸 𑀓𑀺𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀻𑀬𑀦𑁆”𑀢𑀺? “𑀆𑀤𑀺𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑁂𑀮𑁂 𑀯𑀸 𑀲𑀻𑀲𑁂 𑀯𑀸, 𑀢𑀲𑁆𑀲𑁂𑀯 𑀘𑁂𑀮𑀲𑁆𑀲 𑀯𑀸 𑀲𑀻𑀲𑀲𑁆𑀲 𑀯𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀙𑀦𑁆𑀤𑁄 𑀘 𑀯𑀸𑀬𑀸𑀫𑁄 𑀘 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀘 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀻 𑀘 𑀲𑀢𑀺 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘 𑀓𑀭𑀡𑀻𑀬𑀦𑁆”𑀢𑀺𑁇

2323\. “𑀆𑀤𑀺𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀮𑀁 𑀯𑀸 𑀲𑀻𑀲𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑀦𑀲𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁂𑀢𑀸𑀦𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀸𑀬 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀙𑀦𑁆𑀤𑁄 𑀘 𑀯𑀸𑀬𑀸𑀫𑁄 𑀘 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀘 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀻 𑀘 𑀲𑀢𑀺 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲𑁇

2324\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
