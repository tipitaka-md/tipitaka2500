# 𑁧𑁨𑁇𑁪𑁇𑁬 𑀧𑀸𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀻𑀲𑀧𑀸𑀯𑀦𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀡𑀲𑀼𑀢𑁆𑀢

(1106.)

2330\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀢𑀘𑁆𑀙𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬; 𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀁𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀽𑀮𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀲𑀽𑀮𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀫𑀳𑀦𑁆𑀢𑀓𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀫𑀳𑀦𑁆𑀢𑀓𑁂𑀲𑀼 𑀲𑀽𑀮𑁂𑀲𑀼 𑀆𑀯𑀼𑀦𑁂𑀬𑁆𑀬, 𑀬𑁂 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑀓𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑀓𑁂𑀲𑀼 𑀲𑀽𑀮𑁂𑀲𑀼 𑀆𑀯𑀼𑀦𑁂𑀬𑁆𑀬, 𑀬𑁂 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀲𑀼𑀔𑀼𑀫𑀓𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀲𑀼𑀔𑀼𑀫𑀓𑁂𑀲𑀼 𑀲𑀽𑀮𑁂𑀲𑀼 𑀆𑀯𑀼𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀧𑀸𑀡𑀸 𑀅𑀲𑁆𑀲𑀼𑁇

2331\. 𑀅𑀣 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀇𑀢𑁄 𑀩𑀳𑀼𑀢𑀭𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀲𑀼𑀔𑀼𑀫𑀓𑀸 𑀧𑀸𑀡𑀸, 𑀬𑁂 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀲𑀽𑀮𑁂𑀲𑀼 𑀆𑀯𑀼𑀦𑀺𑀢𑀼𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀲𑀼𑀔𑀼𑀫𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀲𑁆𑀲𑁇 𑀏𑀯𑀁 𑀫𑀳𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀸𑀬𑁄𑁇 𑀏𑀯𑀁 𑀫𑀳𑀦𑁆𑀢𑀲𑁆𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

2332\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇

---

2333\. 𑀙𑀝𑁆𑀞𑀁𑁇
