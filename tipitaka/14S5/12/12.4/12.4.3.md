# 𑁧𑁨𑁇𑁪𑁇𑁩 𑀤𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀻𑀲𑀧𑀸𑀯𑀦𑀯𑀕𑁆𑀕

* 𑀤𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

(1103.)

2319\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀡𑁆𑀟𑁄 𑀉𑀧𑀭𑀺𑀯𑁂𑀳𑀸𑀲𑀁 𑀔𑀺𑀢𑁆𑀢𑁄 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀫𑀽𑀮𑁂𑀦 𑀦𑀺𑀧𑀢𑀢𑀺, 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀅𑀕𑁆𑀕𑁂𑀦 𑀦𑀺𑀧𑀢𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸 𑀲𑀢𑁆𑀢𑀸 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀦𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀸 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑀸 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀅𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸 𑀧𑀭𑀁 𑀮𑁄𑀓𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀧𑀭𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀤𑀺𑀝𑁆𑀞𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲𑁇

2320\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
