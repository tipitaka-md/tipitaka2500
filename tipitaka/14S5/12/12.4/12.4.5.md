# 𑁧𑁨𑁇𑁪𑁇𑁫 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀻𑀲𑀧𑀸𑀯𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑀲𑀼𑀢𑁆𑀢

(1105.)

2326. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀬𑀼𑀓𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀚𑀻𑀯𑀻𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀏𑀳𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀢𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀤𑀺𑀯𑀲𑁂 𑀤𑀺𑀯𑀲𑁂 𑀢𑀻𑀳𑀺 𑀢𑀻𑀳𑀺 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀳𑀺 𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀬𑀼𑀓𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀚𑀻𑀯𑀻 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀅𑀦𑀪𑀺𑀲𑀫𑁂𑀢𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀪𑀺𑀲𑀫𑁂𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇

2327. 𑀅𑀢𑁆𑀣𑀯𑀲𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀮𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄; 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀲𑀢𑁆𑀢𑀺𑀧𑁆𑀧𑀳𑀸𑀭𑀸𑀦𑀁 𑀅𑀲𑀺𑀧𑁆𑀧𑀳𑀸𑀭𑀸𑀦𑀁 𑀉𑀲𑀼𑀧𑁆𑀧𑀳𑀸𑀭𑀸𑀦𑀁 𑀨𑀭𑀲𑀼𑀧𑁆𑀧𑀳𑀸𑀭𑀸𑀦𑀁𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀳 𑀤𑀼𑀓𑁆𑀔𑁂𑀦, 𑀲𑀳 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀦 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀁 𑀯𑀤𑀸𑀫𑀺; 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀳𑀸𑀯 𑀲𑀼𑀔𑁂𑀦, 𑀲𑀳𑀸𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀦 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲𑁇

2328. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
