### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀢𑀼𑀮𑀸𑀓𑀽𑀝𑀲𑀼𑀢𑁆𑀢

(1164.)

2511\. …  “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀢𑀼𑀮𑀸𑀓𑀽𑀝𑀓𑀁𑀲𑀓𑀽𑀝𑀫𑀸𑀦𑀓𑀽𑀝𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀢𑀼𑀮𑀸𑀓𑀽𑀝𑀓𑀁𑀲𑀓𑀽𑀝𑀫𑀸𑀦𑀓𑀽𑀝𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸…𑀧𑁂… 𑁇
