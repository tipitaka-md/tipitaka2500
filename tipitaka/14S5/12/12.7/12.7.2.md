# 𑁧𑁨𑁇𑁭𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(1132.)

2437\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁—  𑀬𑁄 𑀯𑀸𑀬𑀁 𑀫𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼 𑀆𑀭𑁄𑀧𑀺𑀢𑁄, 𑀅𑀬𑀁 𑀯𑀸 𑀫𑀳𑀸𑀧𑀣𑀯𑀻”𑀢𑀺? “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑀤𑀺𑀤𑀁—  𑀫𑀳𑀸𑀧𑀣𑀯𑀻; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀸𑀬𑀁 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼 𑀆𑀭𑁄𑀧𑀺𑀢𑁄𑁇 𑀲𑀗𑁆𑀔𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, 𑀉𑀧𑀦𑀺𑀥𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, 𑀓𑀮𑀪𑀸𑀕𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼 𑀆𑀭𑁄𑀧𑀺𑀢𑁄”𑀢𑀺𑁇

2438\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁂𑀲𑀼 𑀫𑀺𑀮𑀓𑁆𑀔𑁂𑀲𑀼…𑀧𑁂… 𑁇
