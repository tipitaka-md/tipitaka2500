# 𑁧𑁨𑁇𑁭𑁇𑁧𑁦 𑀧𑀘𑀸𑀬𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀘𑀸𑀬𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(1140.)

2454\. …  “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀓𑀼𑀮𑁂 𑀚𑁂𑀝𑁆𑀞𑀸𑀧𑀘𑀸𑀬𑀺𑀦𑁄; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀓𑀼𑀮𑁂 𑀅𑀚𑁂𑀝𑁆𑀞𑀸𑀧𑀘𑀸𑀬𑀺𑀦𑁄𑀢𑀺…𑀧𑁂… 𑁇

---

2455\. 𑀤𑀲𑀫𑀁𑁇



---

2456\. 𑀧𑀞𑀫𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇



---

2457\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2458\. _𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀁 𑀧𑀜𑁆𑀜𑀸,_  
_𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀑𑀤𑀓𑀸;_  
_𑀫𑀢𑁆𑀢𑁂𑀬𑁆𑀬 𑀧𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀸 𑀘𑀸𑀧𑀺,_  
_𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀘𑀸𑀬𑀺𑀓𑀦𑁆𑀢𑀺𑁇_
