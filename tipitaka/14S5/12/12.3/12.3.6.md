# 𑁧𑁨𑁇𑁩𑁇𑁬 𑀫𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(1096.)

2288\. “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀸𑀣, 𑀬𑁂 𑀘 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀼𑀁—  𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸—  𑀢𑁂 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

2289\. 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀸𑀣, 𑀬𑁂 𑀘 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀼𑀁—  𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸—  𑀢𑁂 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇

2290\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
