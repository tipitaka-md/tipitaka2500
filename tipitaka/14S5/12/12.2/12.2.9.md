# 𑁧𑁨𑁇𑁨𑁇𑁯 𑀲𑀗𑁆𑀓𑀸𑀲𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀦𑀯𑀕𑁆𑀕

* 𑀲𑀗𑁆𑀓𑀸𑀲𑀦𑀲𑀼𑀢𑁆𑀢

(1089.)

2255\. “‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀦𑁆’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀢𑀢𑁆𑀣 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀯𑀡𑁆𑀡𑀸 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀲𑀗𑁆𑀓𑀸𑀲𑀦𑀸—  ‘𑀇𑀢𑀺𑀧𑀺𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀦𑁆’𑀢𑀺; 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁…𑀧𑁂…  ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀦𑁆’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀢𑀢𑁆𑀣 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀯𑀡𑁆𑀡𑀸 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀲𑀗𑁆𑀓𑀸𑀲𑀦𑀸—  ‘𑀇𑀢𑀺𑀧𑀺𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀦𑁆’𑀢𑀺𑁇

2256\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
