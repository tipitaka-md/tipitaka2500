# 𑁧𑁨𑁇𑁨𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀥𑀸𑀭𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀦𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀥𑀸𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(1086.)

2244. “𑀥𑀸𑀭𑁂𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀻”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀸𑀭𑁂𑀫𑀺 𑀪𑀕𑀯𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀻”𑀢𑀺𑁇

2245. “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀥𑀸𑀭𑁂𑀲𑀺 𑀫𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀻”𑀢𑀺? “𑀤𑀼𑀓𑁆𑀔𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀥𑀸𑀭𑁂𑀫𑀺𑁇 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀁 𑀲𑀫𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁𑁇 𑀅𑀳𑀫𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀅𑀜𑁆𑀜𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀥𑀸𑀭𑁂𑀫𑀺𑁇 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀁 𑀲𑀫𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁𑁇 𑀅𑀳𑀫𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀅𑀜𑁆𑀜𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀸𑀭𑁂𑀫𑀺 𑀪𑀕𑀯𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀻”𑀢𑀺𑁇

2246. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀥𑀸𑀭𑁂𑀲𑀺 𑀫𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀻𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀤𑁂𑀲𑀺𑀢𑀁, 𑀢𑀣𑀸 𑀦𑀁 𑀥𑀸𑀭𑁂𑀳𑀺𑁇 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀁 𑀲𑀫𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁𑁇 𑀅𑀳𑀫𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀅𑀜𑁆𑀜𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀞𑀫𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀤𑁂𑀲𑀺𑀢𑀁, 𑀢𑀣𑀸 𑀦𑀁 𑀥𑀸𑀭𑁂𑀳𑀺𑁇 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀁 𑀲𑀫𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁𑁇 𑀅𑀳𑀫𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀅𑀜𑁆𑀜𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀥𑀸𑀭𑁂𑀳𑀺 𑀫𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀻𑀢𑀺𑁇

2247. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀼, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
