# 𑁧𑁨𑁇𑁧𑁧𑁇𑁧𑁬--𑁧𑁮 𑀦𑀺𑀭𑀬𑀤𑁂𑀯𑀦𑀺𑀭𑀬𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀕𑀢𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀭𑀬𑀤𑁂𑀯𑀦𑀺𑀭𑀬𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

(1187--1189.)

2535. …  “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀦𑀺𑀭𑀬𑀸 𑀘𑀼𑀢𑀸 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀦𑀺𑀭𑀬𑀸 𑀘𑀼𑀢𑀸 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇
