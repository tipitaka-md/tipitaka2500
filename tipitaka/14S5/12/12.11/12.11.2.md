# 𑁧𑁨𑁇𑁧𑁧𑁇𑁨 𑀫𑀦𑀼𑀲𑁆𑀲𑀘𑀼𑀢𑀺𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀕𑀢𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀫𑀦𑀼𑀲𑁆𑀲𑀘𑀼𑀢𑀺𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(1173.)

2523\. …  “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘𑀼𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘𑀼𑀢𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇
