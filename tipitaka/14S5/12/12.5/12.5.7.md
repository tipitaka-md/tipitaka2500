# 𑁧𑁨𑁇𑁫𑁇𑁭 𑀧𑀞𑀫𑀙𑀺𑀕𑁆𑀕𑀍𑀅𑀬𑀼𑀕𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀧𑀸𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀙𑀺𑀕𑁆𑀕𑀍𑀅𑀬𑀼𑀕𑀲𑀼𑀢𑁆𑀢

(1117.)

2390\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀏𑀓𑀘𑁆𑀙𑀺𑀕𑁆𑀕𑀍𑀅𑀁 𑀬𑀼𑀕𑀁 𑀧𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑁇 𑀢𑀢𑁆𑀭𑀸𑀧𑀺𑀲𑁆𑀲 𑀓𑀸𑀡𑁄 𑀓𑀘𑁆𑀙𑀧𑁄𑁇 𑀲𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀓𑀸𑀡𑁄 𑀓𑀘𑁆𑀙𑀧𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀏𑀓𑀘𑁆𑀙𑀺𑀕𑁆𑀕𑀍𑀏 𑀬𑀼𑀕𑁂 𑀕𑀻𑀯𑀁 𑀧𑀯𑁂𑀲𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀬𑀤𑀺 𑀦𑀽𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀤𑀻𑀖𑀲𑁆𑀲 𑀅𑀤𑁆𑀥𑀼𑀦𑁄 𑀅𑀘𑁆𑀘𑀬𑁂𑀦𑀸”𑀢𑀺𑁇

2391\. “𑀔𑀺𑀧𑁆𑀧𑀢𑀭𑀁 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀡𑁄 𑀓𑀘𑁆𑀙𑀧𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀏𑀓𑀘𑁆𑀙𑀺𑀕𑁆𑀕𑀍𑀏 𑀬𑀼𑀕𑁂 𑀕𑀻𑀯𑀁 𑀧𑀯𑁂𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀢𑁆𑀯𑁂𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀕𑀢𑁂𑀦 𑀩𑀸𑀮𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

2392\. 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀦 𑀳𑁂𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸, 𑀲𑀫𑀘𑀭𑀺𑀬𑀸, 𑀓𑀼𑀲𑀮𑀓𑀺𑀭𑀺𑀬𑀸, 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀸𑁇 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀔𑀸𑀤𑀺𑀓𑀸 𑀏𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀢𑁆𑀢𑀢𑀺 𑀤𑀼𑀩𑁆𑀩𑀮𑀔𑀸𑀤𑀺𑀓𑀸𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀤𑀺𑀝𑁆𑀞𑀢𑁆𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲𑁇

2393\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
