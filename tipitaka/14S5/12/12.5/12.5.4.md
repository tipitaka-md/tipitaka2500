# 𑁧𑁨𑁇𑁫𑁇𑁪 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀧𑀸𑀢𑀯𑀕𑁆𑀕

### 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(1114.)

2373\. “𑀬𑁄 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

2374\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀁 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀁 𑀖𑀭𑀁 𑀅𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀭𑀺𑀫𑀁 𑀖𑀭𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

2375\. 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

2376\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀁 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀁 𑀖𑀭𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀭𑀺𑀫𑀁 𑀖𑀭𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀪𑀺𑀲𑀫𑁂𑀘𑁆𑀘 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

2377\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
