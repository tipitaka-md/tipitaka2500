# 𑁧𑁨𑁇𑁫𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀦𑁂𑀭𑀼𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀧𑀸𑀢𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀦𑁂𑀭𑀼𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑀼𑀢𑁆𑀢

(1120.)

2402\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀦𑁂𑀭𑀼𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀸𑀬𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁—  𑀬𑀁 𑀯𑀸 𑀲𑀺𑀦𑁂𑀭𑀼𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀬𑀸 𑀯𑀸 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸”𑀢𑀺? “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀲𑀺𑀦𑁂𑀭𑀼𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀬𑀤𑀺𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑁇 𑀲𑀗𑁆𑀔𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀉𑀧𑀦𑀺𑀥𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀓𑀮𑀪𑀸𑀕𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑀺𑀦𑁂𑀭𑀼𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀪𑀺𑀲𑀫𑁂𑀢𑀸𑀯𑀺𑀦𑁄 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀁 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀁𑁇 𑀲𑀗𑁆𑀔𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, 𑀉𑀧𑀦𑀺𑀥𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, 𑀓𑀮𑀪𑀸𑀕𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺, 𑀧𑀼𑀭𑀺𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀤𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀧𑀭𑀫𑀢𑀸; 𑀬𑁄 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

2403\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇

2406\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2407\. _𑀘𑀺𑀦𑁆𑀢𑀸 𑀧𑀧𑀸𑀢𑁄 𑀧𑀭𑀺𑀍𑀆𑀳𑁄,_
