# 𑁧𑁨𑁇𑁫𑁇𑁬 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀧𑀸𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(1116.)

2384. “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑁄𑀓𑀦𑁆𑀢𑀭𑀺𑀓𑀸 𑀅𑀖𑀸 𑀅𑀲𑀁𑀯𑀼𑀢𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀢𑀺𑀫𑀺𑀲𑀸, 𑀬𑀢𑁆𑀣𑀫𑀺𑀫𑁂𑀲𑀁 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑀸𑀦𑀁 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸𑀦𑀁 𑀏𑀯𑀁 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑀦𑀁 𑀆𑀪𑀸𑀬 𑀦𑀸𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2385. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀳𑀸 𑀯𑀢 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄, 𑀲𑀼𑀫𑀳𑀸 𑀯𑀢 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀫𑁆𑀳𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀫𑀳𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀪𑀬𑀸𑀦𑀓𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀢𑀫𑁆𑀳𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀫𑀳𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀪𑀬𑀸𑀦𑀓𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇

2386. “𑀓𑀢𑀫𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀫𑁆𑀳𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀫𑀳𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀪𑀬𑀸𑀦𑀓𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺? “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀪𑀺𑀭𑀢𑀸…𑀧𑁂…  𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀚𑀸𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀚𑀭𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺𑁇 ‘𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

2387. 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀸𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀪𑀺𑀭𑀢𑀸…𑀧𑁂…  𑀦𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀚𑀸𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀚𑀭𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺𑁇 ‘𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

2388. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
