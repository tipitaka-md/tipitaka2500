# 𑁧𑁨𑁇𑁫𑁇𑁨 𑀧𑀧𑀸𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀧𑀸𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀧𑀸𑀢𑀲𑀼𑀢𑁆𑀢

(1112.)

2362\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀧𑀝𑀺𑀪𑀸𑀦𑀓𑀽𑀝𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀧𑀝𑀺𑀪𑀸𑀦𑀓𑀽𑀝𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀪𑀸𑀦𑀓𑀽𑀝𑁂 𑀫𑀳𑀦𑁆𑀢𑀁 𑀧𑀧𑀸𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀧𑀸𑀢𑁄 𑀲𑀼𑀪𑀬𑀸𑀦𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀧𑀸𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑀫𑁆𑀳𑀸 𑀧𑀧𑀸𑀢𑀸 𑀅𑀜𑁆𑀜𑁄 𑀧𑀧𑀸𑀢𑁄 𑀫𑀳𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀪𑀬𑀸𑀦𑀓𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀫𑀫𑁆𑀳𑀸 𑀧𑀧𑀸𑀢𑀸 𑀅𑀜𑁆𑀜𑁄 𑀧𑀧𑀸𑀢𑁄 𑀫𑀳𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀪𑀬𑀸𑀦𑀓𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇

2363\. “𑀓𑀢𑀫𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑀫𑁆𑀳𑀸 𑀧𑀧𑀸𑀢𑀸 𑀅𑀜𑁆𑀜𑁄 𑀧𑀧𑀸𑀢𑁄 𑀫𑀳𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀪𑀬𑀸𑀦𑀓𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺? “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀢𑀸 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀢𑀸 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀢𑀸 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀪𑀺𑀭𑀢𑀸 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺, 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀚𑀸𑀢𑀺𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀚𑀭𑀸𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀧𑀧𑀢𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺𑁇 ‘𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

2364\. 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀸𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀸𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀸𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀸𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀪𑀺𑀭𑀢𑀸, 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀪𑀺𑀭𑀢𑀸, 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀪𑀺𑀭𑀢𑀸, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀪𑀺𑀭𑀢𑀸, 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀦𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺, 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀦𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀦𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀦𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀚𑀸𑀢𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀦𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸, 𑀚𑀭𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀦𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸, 𑀫𑀭𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀦𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀦𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸, 𑀚𑀸𑀢𑀺𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀚𑀭𑀸𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀧𑀧𑀸𑀢𑀫𑁆𑀧𑀺 𑀦𑀧𑁆𑀧𑀧𑀢𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺𑁇 ‘𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

2365\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
