# 𑁧𑁨𑁇𑁬𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀫𑀳𑀸𑀧𑀣𑀯𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀪𑀺𑀲𑀫𑀬𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀫𑀳𑀸𑀧𑀣𑀯𑀻𑀲𑀼𑀢𑁆𑀢

(1126.)

2419\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀧𑀣𑀯𑀻 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢 𑀓𑁄𑀮𑀝𑁆𑀞𑀺𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀕𑀼𑀍𑀇𑀓𑀸𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁—  𑀬𑀁 𑀯𑀸 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀬𑀸 𑀯𑀸 𑀲𑀢𑁆𑀢 𑀓𑁄𑀮𑀝𑁆𑀞𑀺𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀕𑀼𑀍𑀇𑀓𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸”𑀢𑀺? “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀬𑀸 𑀬𑀤𑀺𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢 𑀓𑁄𑀮𑀝𑁆𑀞𑀺𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀕𑀼𑀍𑀇𑀓𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑁇 𑀲𑀗𑁆𑀔𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀉𑀧𑀦𑀺𑀥𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀓𑀮𑀪𑀸𑀕𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀲𑀢𑁆𑀢 𑀓𑁄𑀮𑀝𑁆𑀞𑀺𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀕𑀼𑀍𑀇𑀓𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
