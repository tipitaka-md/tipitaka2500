# 𑁧𑁨𑁇𑁬𑁇𑁪 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀪𑁂𑀚𑁆𑀚𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀪𑀺𑀲𑀫𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀪𑁂𑀚𑁆𑀚𑀲𑀼𑀢𑁆𑀢

(1124.)

2415. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣𑀺𑀫𑀸 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀕𑀗𑁆𑀕𑀸, 𑀬𑀫𑀼𑀦𑀸, 𑀅𑀘𑀺𑀭𑀯𑀢𑀻, 𑀲𑀭𑀪𑀽, 𑀫𑀳𑀻, 𑀢𑀁 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁—  𑀬𑀁 𑀯𑀸 𑀲𑀁𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀬𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀻”𑀢𑀺? “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀲𑀁𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀺𑁇 𑀲𑀗𑁆𑀔𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀉𑀧𑀦𑀺𑀥𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺, 𑀓𑀮𑀪𑀸𑀕𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑀁𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇
