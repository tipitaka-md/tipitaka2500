# 𑁬𑁇𑁫𑁇𑁧--𑁧𑁦 𑀑𑀖𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀩𑀮𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀑𑀖𑀯𑀕𑁆𑀕

### 𑀑𑀖𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

(749--758.)

1324\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀅𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀫𑀸𑀦𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀼𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀁, 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁…𑀧𑁂…  𑀲𑀢𑀺𑀩𑀮𑀁…  𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁…  𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀁𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺𑁇 (𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇)
