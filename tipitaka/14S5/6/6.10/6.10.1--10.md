# 𑁬𑁇𑁧𑁦𑁇𑁧--𑁧𑁦 𑀧𑀼𑀦𑀑𑀖𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀩𑀮𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀼𑀦𑀑𑀖𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀦𑀑𑀖𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

(803--812.)

1335\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀅𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀫𑀸𑀦𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀼𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑁄𑀲𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀫𑁄𑀳𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺𑁇
