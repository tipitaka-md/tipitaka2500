# 𑁭𑁇𑁨𑁇𑁫 𑀉𑀡𑁆𑀡𑀸𑀪𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀸𑀲𑀸𑀤𑀓𑀫𑁆𑀧𑀦𑀯𑀕𑁆𑀕

* 𑀉𑀡𑁆𑀡𑀸𑀪𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(827.)

1414\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀡𑁆𑀡𑀸𑀪𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀉𑀡𑁆𑀡𑀸𑀪𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀫𑀢𑁆𑀣𑀺𑀬𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀡𑁂 𑀕𑁄𑀢𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀙𑀦𑁆𑀤𑀧𑁆𑀧𑀳𑀸𑀦𑀢𑁆𑀣𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1415\. “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀧𑀤𑀸 𑀏𑀢𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀫𑀕𑁆𑀕𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀧𑀤𑀸 𑀏𑀢𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

1416\. “𑀓𑀢𑀫𑁄 𑀧𑀦, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀕𑁆𑀕𑁄 𑀓𑀢𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀏𑀢𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺? “𑀇𑀥, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀫𑀕𑁆𑀕𑁄 𑀅𑀬𑀁 𑀧𑀝𑀺𑀧𑀤𑀸 𑀏𑀢𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

1417\. “𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀦𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀓𑀁𑁇 𑀙𑀦𑁆𑀤𑁂𑀦𑁂𑀯 𑀙𑀦𑁆𑀤𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬 𑀢𑀣𑀸 𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀳𑁄𑀲𑀺 𑀢𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀙𑀦𑁆𑀤𑁄 ‘𑀆𑀭𑀸𑀫𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀆𑀭𑀸𑀫𑀕𑀢𑀲𑁆𑀲 𑀬𑁄 𑀢𑀚𑁆𑀚𑁄 𑀙𑀦𑁆𑀤𑁄 𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑁄”𑁇 “𑀅𑀳𑁄𑀲𑀺 𑀢𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀻𑀭𑀺𑀬𑀁 ‘𑀆𑀭𑀸𑀫𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀆𑀭𑀸𑀫𑀕𑀢𑀲𑁆𑀲 𑀬𑀁 𑀢𑀚𑁆𑀚𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀢𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑁄”𑁇 “𑀅𑀳𑁄𑀲𑀺 𑀢𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀆𑀭𑀸𑀫𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀆𑀭𑀸𑀫𑀕𑀢𑀲𑁆𑀲 𑀬𑀁 𑀢𑀚𑁆𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀢𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑁄”𑁇 “𑀅𑀳𑁄𑀲𑀺 𑀢𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀻𑀫𑀁𑀲𑀸 ‘𑀆𑀭𑀸𑀫𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀆𑀭𑀸𑀫𑀕𑀢𑀲𑁆𑀲 𑀬𑀸 𑀢𑀚𑁆𑀚𑀸 𑀯𑀻𑀫𑀁𑀲𑀸 𑀲𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑁄”𑁇

1418\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀬𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀯𑀼𑀲𑀺𑀢𑀯𑀸 𑀓𑀢𑀓𑀭𑀡𑀻𑀬𑁄 𑀑𑀳𑀺𑀢𑀪𑀸𑀭𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀤𑀢𑁆𑀣𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀪𑀯𑀲𑀁𑀬𑁄𑀚𑀦𑁄 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀙𑀦𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀬𑁄 𑀢𑀚𑁆𑀚𑁄 𑀙𑀦𑁆𑀤𑁄 𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄; 𑀬𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀬𑀁 𑀢𑀚𑁆𑀚𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀢𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁; 𑀬𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀬𑀁 𑀢𑀚𑁆𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀢𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁; 𑀬𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀻𑀫𑀁𑀲𑀸 𑀅𑀳𑁄𑀲𑀺 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀬𑀸 𑀢𑀚𑁆𑀚𑀸 𑀯𑀻𑀫𑀁𑀲𑀸 𑀲𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀇𑀢𑀺 𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂, 𑀲𑀦𑁆𑀢𑀓𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀓𑀁 𑀯𑀸”𑀢𑀺?

1419\. “𑀅𑀤𑁆𑀥𑀸, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂, 𑀲𑀦𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀓𑀁𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑁄𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀆𑀦𑀦𑁆𑀤𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
