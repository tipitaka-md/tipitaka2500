# 𑁭𑁇𑁨𑁇𑁬 𑀧𑀞𑀫𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀸𑀲𑀸𑀤𑀓𑀫𑁆𑀧𑀦𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(828.)

1421\. “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇

1422\. 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸”𑀢𑀺𑁇
