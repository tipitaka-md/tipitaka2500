# 𑁭𑁇𑁨𑁇𑁧𑁦 𑀯𑀺𑀪𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀲𑀸𑀤𑀓𑀫𑁆𑀧𑀦𑀯𑀕𑁆𑀕

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(832.)

1440\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸”𑁇

1441\. 𑀓𑀣𑀁 𑀪𑀸𑀯𑀺𑀢𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀓𑀣𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺—  𑀇𑀢𑀺 𑀫𑁂 𑀙𑀦𑁆𑀤𑁄 𑀦 𑀘 𑀅𑀢𑀺𑀮𑀻𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀸𑀧𑀼𑀭𑁂𑀲𑀜𑁆𑀜𑀻 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀬𑀣𑀸 𑀧𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀧𑀼𑀭𑁂; 𑀬𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁, 𑀬𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄; 𑀬𑀣𑀸 𑀤𑀺𑀯𑀸 𑀢𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀣𑀸 𑀤𑀺𑀯𑀸𑁇 𑀇𑀢𑀺 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺—  𑀇𑀢𑀺 𑀫𑁂 𑀯𑀻𑀫𑀁𑀲𑀸 𑀦 𑀘 𑀅𑀢𑀺𑀮𑀻𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀸𑀧𑀼𑀭𑁂𑀲𑀜𑁆𑀜𑀻 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀬𑀣𑀸 𑀧𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀧𑀼𑀭𑁂; 𑀬𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁, 𑀬𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄; 𑀬𑀣𑀸 𑀤𑀺𑀯𑀸 𑀢𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁, 𑀬𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀣𑀸 𑀤𑀺𑀯𑀸𑁇 𑀇𑀢𑀺 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇

1442\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀮𑀻𑀦𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀤𑁄 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑀳𑀕𑀢𑁄 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀮𑀻𑀦𑁄 𑀙𑀦𑁆𑀤𑁄𑁇

1443\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀤𑁄 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑁄 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀙𑀦𑁆𑀤𑁄𑁇

1444\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀤𑁄 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀲𑀳𑀕𑀢𑁄 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁄 𑀙𑀦𑁆𑀤𑁄𑁇

1445\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀤𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑀼𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀯𑀺𑀲𑀝𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀙𑀦𑁆𑀤𑁄𑁇

1446\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀘𑁆𑀙𑀸𑀧𑀼𑀭𑁂𑀲𑀜𑁆𑀜𑀻 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀬𑀣𑀸 𑀧𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀧𑀼𑀭𑁂? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀘𑁆𑀙𑀸𑀧𑀼𑀭𑁂𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀸 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀘𑁆𑀙𑀸𑀧𑀼𑀭𑁂𑀲𑀜𑁆𑀜𑀻 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀬𑀣𑀸 𑀧𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀧𑀼𑀭𑁂𑁇

1447\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁, 𑀬𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑀫𑁂𑀯 𑀓𑀸𑀬𑀁 𑀉𑀤𑁆𑀥𑀁 𑀧𑀸𑀤𑀢𑀮𑀸 𑀅𑀥𑁄 𑀓𑁂𑀲𑀫𑀢𑁆𑀣𑀓𑀸 𑀢𑀘𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀧𑀽𑀭𑀁 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺𑀦𑁄 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀦𑀔𑀸 𑀤𑀦𑁆𑀢𑀸 𑀢𑀘𑁄 𑀫𑀁𑀲𑀁 𑀦𑁆𑀳𑀸𑀭𑀼 𑀅𑀝𑁆𑀞𑀺 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀯𑀓𑁆𑀓𑀁 𑀳𑀤𑀬𑀁 𑀬𑀓𑀦𑀁 𑀓𑀺𑀮𑁄𑀫𑀓𑀁 𑀧𑀺𑀳𑀓𑀁 𑀧𑀧𑁆𑀨𑀸𑀲𑀁 𑀅𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀕𑀼𑀡𑀁 𑀉𑀤𑀭𑀺𑀬𑀁 𑀓𑀭𑀻𑀲𑀁 𑀧𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀫𑁆𑀳𑀁 𑀧𑀼𑀩𑁆𑀩𑁄 𑀮𑁄𑀳𑀺𑀢𑀁 𑀲𑁂𑀤𑁄 𑀫𑁂𑀤𑁄 𑀅𑀲𑁆𑀲𑀼 𑀯𑀲𑀸 𑀔𑁂𑀍𑀑 𑀲𑀺𑀗𑁆𑀖𑀸𑀡𑀺𑀓𑀸 𑀮𑀲𑀺𑀓𑀸 𑀫𑀼𑀢𑁆𑀢𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁, 𑀬𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

1453\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀁? 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀭𑀺𑀬𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑀼𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀲𑀝𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀁…𑀧𑁂… 𑁇

1454\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀮𑁄𑀓𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑀺𑀯𑀸𑀲𑀜𑁆𑀜𑀸 𑀲𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇

1455\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀮𑀻𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑀳𑀕𑀢𑀁 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀮𑀻𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

1456\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

1457\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀲𑀳𑀕𑀢𑀁 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

1458\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑀼𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀲𑀝𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇

1459\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀮𑀻𑀦𑀸 𑀯𑀻𑀫𑀁𑀲𑀸? 𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀫𑀁𑀲𑀸 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑀳𑀕𑀢𑀸 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀮𑀻𑀦𑀸 𑀯𑀻𑀫𑀁𑀲𑀸𑁇

1460\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀯𑀻𑀫𑀁𑀲𑀸? 𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀫𑀁𑀲𑀸 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀸 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀯𑀻𑀫𑀁𑀲𑀸𑁇

1461\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸 𑀯𑀻𑀫𑀁𑀲𑀸? 𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀫𑀁𑀲𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀲𑀳𑀕𑀢𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸 𑀯𑀻𑀫𑀁𑀲𑀸𑁇

1462\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀯𑀻𑀫𑀁𑀲𑀸? 𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀫𑀁𑀲𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑀼𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀯𑀺𑀲𑀝𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀯𑀻𑀫𑀁𑀲𑀸…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸𑁇

1463\. 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁂𑀲𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀽𑀲𑀼 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂𑀲𑀼, 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀥𑀸𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁂𑀲𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀽𑀲𑀼 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂𑀲𑀼, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

1465\. 𑀧𑀸𑀲𑀸𑀤𑀓𑀫𑁆𑀧𑀦𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇

1466\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1467\. _𑀧𑀼𑀩𑁆𑀩𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀙𑀦𑁆𑀤𑀁,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀜𑁆𑀘 𑀉𑀡𑁆𑀡𑀸𑀪𑀁;_  
_𑀤𑁆𑀯𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀤𑁂𑀲𑀦𑀸 𑀯𑀺𑀪𑀗𑁆𑀕𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_
