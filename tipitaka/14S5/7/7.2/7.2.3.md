# 𑁭𑁇𑁨𑁇𑁩 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀲𑀸𑀤𑀓𑀫𑁆𑀧𑀦𑀯𑀕𑁆𑀕

### 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀲𑀼𑀢𑁆𑀢

(825.)

1403. “𑀙𑀦𑁆𑀤𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀮𑀪𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁, 𑀮𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑁇 𑀲𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 ‘𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀙𑀦𑁆𑀤𑁄, 𑀅𑀬𑀜𑁆𑀘 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺, 𑀇𑀫𑁂 𑀘 𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄𑁇

1404. 𑀯𑀻𑀭𑀺𑀬𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀮𑀪𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁, 𑀮𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺’𑁇 𑀲𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 ‘𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀯𑀻𑀭𑀺𑀬𑀁, 𑀅𑀬𑀜𑁆𑀘 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺, 𑀇𑀫𑁂 𑀘 𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄𑁇

1405. 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀮𑀪𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁, 𑀮𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺’𑁇 𑀲𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 ‘𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀅𑀬𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺, 𑀇𑀫𑁂 𑀘 𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄𑁇

1406. 𑀯𑀻𑀫𑀁𑀲𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀮𑀪𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁, 𑀮𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺’𑁇 𑀲𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 ‘𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀯𑀻𑀫𑀁𑀲𑀸, 𑀅𑀬𑀜𑁆𑀘 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺, 𑀇𑀫𑁂 𑀘 𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄”𑀢𑀺𑁇
