# 𑁭𑁇𑁨𑁇𑁭 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀲𑀸𑀤𑀓𑀫𑁆𑀧𑀦𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(829.)

1424\. “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑀼𑀁—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀩𑀳𑀼𑀥𑀸𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀅𑀳𑁂𑀲𑀼𑀁; 𑀆𑀯𑀺𑀪𑀸𑀯𑀁, 𑀢𑀺𑀭𑁄𑀪𑀸𑀯𑀁; 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀧𑀸𑀓𑀸𑀭𑀁 𑀢𑀺𑀭𑁄𑀧𑀩𑁆𑀩𑀢𑀁 𑀅𑀲𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀕𑀫𑀁𑀲𑀼, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀆𑀓𑀸𑀲𑁂; 𑀧𑀣𑀯𑀺𑀬𑀸𑀧𑀺 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀁 𑀅𑀓𑀁𑀲𑀼, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀉𑀤𑀓𑁂; 𑀉𑀤𑀓𑁂𑀧𑀺 𑀅𑀪𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀅𑀕𑀫𑀁𑀲𑀼, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀣𑀯𑀺𑀬𑀁; 𑀆𑀓𑀸𑀲𑁂𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀓𑀫𑀺𑀁𑀲𑀼, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀓𑁆𑀔𑀻 𑀲𑀓𑀼𑀡𑁄; 𑀇𑀫𑁂𑀧𑀺 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑁂 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁂 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂 𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑀭𑀺𑀫𑀲𑀺𑀁𑀲𑀼 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀺𑀁𑀲𑀼; 𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀲𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇

1425\. 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀩𑀳𑀼𑀥𑀸𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀆𑀯𑀺𑀪𑀸𑀯𑀁, 𑀢𑀺𑀭𑁄𑀪𑀸𑀯𑀁; 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀧𑀸𑀓𑀸𑀭𑀁 𑀢𑀺𑀭𑁄𑀧𑀩𑁆𑀩𑀢𑀁 𑀅𑀲𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀆𑀓𑀸𑀲𑁂; 𑀧𑀣𑀯𑀺𑀬𑀸𑀧𑀺 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀉𑀤𑀓𑁂; 𑀉𑀤𑀓𑁂𑀧𑀺 𑀅𑀪𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀣𑀯𑀺𑀬𑀁; 𑀆𑀓𑀸𑀲𑁂𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀓𑁆𑀔𑀻 𑀲𑀓𑀼𑀡𑁄; 𑀇𑀫𑁂𑀧𑀺 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑁂 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁂 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂 𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑀭𑀺𑀫𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇

1426\. 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀩𑀳𑀼𑀥𑀸𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀳𑁄𑀦𑁆𑀢𑀺; 𑀆𑀯𑀺𑀪𑀸𑀯𑀁, 𑀢𑀺𑀭𑁄𑀪𑀸𑀯𑀁; 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀧𑀸𑀓𑀸𑀭𑀁 𑀢𑀺𑀭𑁄𑀧𑀩𑁆𑀩𑀢𑀁 𑀅𑀲𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀆𑀓𑀸𑀲𑁂; 𑀧𑀣𑀯𑀺𑀬𑀸𑀧𑀺 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀉𑀤𑀓𑁂; 𑀉𑀤𑀓𑁂𑀧𑀺 𑀅𑀪𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀣𑀯𑀺𑀬𑀁; 𑀆𑀓𑀸𑀲𑁂𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀓𑀫𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀓𑁆𑀔𑀻 𑀲𑀓𑀼𑀡𑁄; 𑀇𑀫𑁂𑀧𑀺 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑁂 𑀏𑀯𑀁𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁂 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂 𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑀭𑀺𑀫𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺; 𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑀢𑀺𑁇

1427\. 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑀼𑀁—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀅𑀳𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀲𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇

1428\. 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇

1429\. 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸”𑀢𑀺𑁇
