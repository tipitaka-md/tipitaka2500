# 𑁭𑁇𑁩𑁇𑁨 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀯𑀕𑁆𑀕

* 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀲𑀼𑀢𑁆𑀢

(834.)

1472\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀫𑀦𑁄𑀫𑀬𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸”𑀢𑀺? “𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀫𑀦𑁄𑀫𑀬𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸”𑀢𑀺𑁇 “𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀇𑀫𑀺𑀦𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸”𑀢𑀺? “𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑀺𑀦𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸”𑀢𑀺𑁇

1473\. “𑀬𑀜𑁆𑀘 𑀔𑁄 𑀑𑀫𑀸𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀫𑀦𑁄𑀫𑀬𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁, 𑀬𑀜𑁆𑀘 𑀔𑁄 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀇𑀫𑀺𑀦𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸, 𑀢𑀬𑀺𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀅𑀩𑁆𑀪𑀼𑀢𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀘𑁂𑀯, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀘, 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀘𑁂𑀯, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀥𑀫𑁆𑀫𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀘𑁇

1474\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀓𑀸𑀬𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺, 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀮𑀳𑀼𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀓𑀸𑀬𑁂 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀢𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀮𑀳𑀼𑀢𑀭𑁄 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀫𑀼𑀤𑀼𑀢𑀭𑁄 𑀘 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀢𑀭𑁄 𑀘 𑀧𑀪𑀲𑁆𑀲𑀭𑀢𑀭𑁄 𑀘𑁇

1475\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀬𑁄𑀕𑀼𑀍𑀑 𑀤𑀺𑀯𑀲𑀁 𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑁄 𑀮𑀳𑀼𑀢𑀭𑁄 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀫𑀼𑀤𑀼𑀢𑀭𑁄 𑀘 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀢𑀭𑁄 𑀘 𑀧𑀪𑀲𑁆𑀲𑀭𑀢𑀭𑁄 𑀘; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀓𑀸𑀬𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺, 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀮𑀳𑀼𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀓𑀸𑀬𑁂 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀢𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀮𑀳𑀼𑀢𑀭𑁄 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀫𑀼𑀤𑀼𑀢𑀭𑁄 𑀘 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀢𑀭𑁄 𑀘 𑀧𑀪𑀲𑁆𑀲𑀭𑀢𑀭𑁄 𑀘𑁇

1476\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀓𑀸𑀬𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺, 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀮𑀳𑀼𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀓𑀸𑀬𑁂 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀢𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦𑁂𑀯 𑀧𑀣𑀯𑀺𑀬𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑁄 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀥𑀸𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀢𑀺𑁇

1477\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀽𑀮𑀧𑀺𑀘𑀼 𑀯𑀸 𑀓𑀧𑁆𑀧𑀸𑀲𑀧𑀺𑀘𑀼 𑀯𑀸 𑀮𑀳𑀼𑀓𑁄 𑀯𑀸𑀢𑀽𑀧𑀸𑀤𑀸𑀦𑁄 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦𑁂𑀯 𑀧𑀣𑀯𑀺𑀬𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀓𑀸𑀬𑁂 𑀲𑀫𑁄𑀤𑀳𑀢𑀺, 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀮𑀳𑀼𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀓𑀸𑀬𑁂 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀢𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀅𑀧𑁆𑀧𑀓𑀲𑀺𑀭𑁂𑀦𑁂𑀯 𑀧𑀣𑀯𑀺𑀬𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑁄 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇
