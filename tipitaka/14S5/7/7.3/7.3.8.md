# 𑁭𑁇𑁩𑁇𑁮 𑀤𑀼𑀢𑀺𑀬𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(840.)

1494\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺, 𑀓𑀢𑀫𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄, 𑀓𑀢𑀫𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸, 𑀓𑀢𑀫𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀪𑀕𑀯𑀁𑀦𑁂𑀢𑁆𑀢𑀺𑀓𑀸…𑀧𑁂… 𑁇

1495\. “𑀇𑀥𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑁇

1496\. 𑀓𑀢𑀫𑁄 𑀘𑀸𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄? 𑀬𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀕𑁆𑀕𑁄 𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀇𑀤𑁆𑀥𑀺𑀮𑀸𑀪𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄𑁇

1497\. 𑀓𑀢𑀫𑀸 𑀘𑀸𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸? 𑀇𑀥𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑁇

1498\. 𑀓𑀢𑀫𑀸 𑀘𑀸𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀦𑀦𑁆𑀤, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺𑁇

---

1499\. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
