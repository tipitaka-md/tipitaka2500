# 𑁭𑁇𑁩𑁇𑁯 𑀧𑀞𑀫𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(841.)

1500\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀤𑁆𑀥𑀺, 𑀓𑀢𑀫𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄, 𑀓𑀢𑀫𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸, 𑀓𑀢𑀫𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺?

1501\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑁇

1502\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄? 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀕𑁆𑀕𑁄, 𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀇𑀤𑁆𑀥𑀺𑀮𑀸𑀪𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄𑁇

1503\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑁇

1504\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺𑁇
