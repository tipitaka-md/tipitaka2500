# 𑁭𑁇𑁩𑁇𑁧 𑀫𑀕𑁆𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀫𑀕𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(833.)

1468. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀓𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀪𑀸𑀯𑀦𑀸𑀬𑀸’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺—  𑀇𑀢𑀺 𑀫𑁂 𑀙𑀦𑁆𑀤𑁄 𑀦 𑀘 𑀅𑀢𑀺𑀮𑀻𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀸𑀧𑀼𑀭𑁂𑀲𑀜𑁆𑀜𑀻 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀬𑀣𑀸 𑀧𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀧𑀼𑀭𑁂; 𑀬𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁, 𑀬𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄; 𑀬𑀣𑀸 𑀤𑀺𑀯𑀸 𑀢𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁, 𑀬𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀣𑀸 𑀤𑀺𑀯𑀸𑁇 𑀇𑀢𑀺 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺—  𑀇𑀢𑀺 𑀫𑁂 𑀯𑀻𑀫𑀁𑀲𑀸 𑀦 𑀘 𑀅𑀢𑀺𑀮𑀻𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀢𑀺𑀧𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀘 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀸𑀧𑀼𑀭𑁂𑀲𑀜𑁆𑀜𑀻 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀬𑀣𑀸 𑀧𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀧𑀼𑀭𑁂; 𑀬𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁, 𑀬𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄; 𑀬𑀣𑀸 𑀤𑀺𑀯𑀸 𑀢𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀬𑀣𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀣𑀸 𑀤𑀺𑀯𑀸’—  𑀇𑀢𑀺 𑀯𑀺𑀯𑀝𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑁂𑀦 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇

1469. 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁂𑀲𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀽𑀲𑀼 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂𑀲𑀼 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺—  𑀏𑀓𑁄𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀥𑀸 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀥𑀸𑀧𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀧𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁂𑀲𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀽𑀲𑀼 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀏𑀯𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂𑀲𑀼, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇
