# 𑁭𑁇𑁧𑁇𑁬 𑀲𑀫𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀸𑀧𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀲𑀫𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(818.)

1352\. “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀢𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀲𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀢𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀢𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇

1353\. 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺…𑀧𑁂…  𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀢𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀲𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀢𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀢𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸”𑀢𑀺𑁇
