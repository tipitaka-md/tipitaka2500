# 𑁭𑁇𑁧𑁇𑁯 𑀜𑀸𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀸𑀧𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀜𑀸𑀡𑀲𑀼𑀢𑁆𑀢

(821.)

1360\. “‘𑀅𑀬𑀁 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 ‘𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  ‘𑀪𑀸𑀯𑀺𑀢𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

1361\. ‘𑀅𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 ‘𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  ‘𑀪𑀸𑀯𑀺𑀢𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

1362\. ‘𑀅𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 ‘𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  ‘𑀪𑀸𑀯𑀺𑀢𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

1363\. ‘𑀅𑀬𑀁 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 ‘𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  ‘𑀪𑀸𑀯𑀺𑀢𑁄’𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀻”𑀢𑀺𑁇
