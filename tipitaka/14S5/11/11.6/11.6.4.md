# 𑁧𑁧𑁇𑁬𑁇𑁪 𑀕𑀺𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀯𑀕𑁆𑀕

### 𑀕𑀺𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(1050.)

2110\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  “𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄—  “𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  ‘𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂—  ‘𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀦𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁂𑀦 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄 𑀑𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

2111\. “𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁂𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄 𑀘𑀢𑀽𑀳𑀺 𑀅𑀲𑁆𑀲𑀸𑀲𑀦𑀻𑀬𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀅𑀲𑁆𑀲𑀸𑀲𑀢𑀸𑀬𑀲𑁆𑀫𑀸—  𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁄 𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀸𑀲𑀢𑀸𑀬𑀲𑁆𑀫𑀸—  𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑀸𑀦𑀺 𑀅𑀔𑀡𑁆𑀟𑀸𑀦𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀦𑀻’𑀢𑀺𑁇

2112\. 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁂𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀅𑀲𑁆𑀲𑀸𑀲𑀦𑀻𑀬𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲𑁄 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑁄𑁇 𑀲𑀘𑁂𑀧𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑁂𑀯; 𑀦𑁄 𑀘𑁂𑀧𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑁂𑀯𑁇 𑀲𑀸𑀥𑀸𑀬𑀲𑁆𑀫𑀸, 𑀬𑀸 𑀢𑁂 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸 𑀢𑀁 𑀧𑀚𑀳𑀸’𑀢𑀺𑁇

2113\. 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀬𑀸 𑀫𑁂 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸 𑀲𑀸 𑀧𑀳𑀻𑀦𑀸’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀢𑁆𑀣𑀺 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲𑁄 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑁄𑁇 𑀲𑀘𑁂𑀧𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑁂𑀯; 𑀦𑁄 𑀘𑁂𑀧𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑁂𑀯𑁇 𑀲𑀸𑀥𑀸𑀬𑀲𑁆𑀫𑀸, 𑀬𑀸 𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸 𑀢𑀁 𑀧𑀚𑀳𑀸’𑀢𑀺𑁇

2114\. 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀬𑀸 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸 𑀲𑀸 𑀧𑀳𑀻𑀦𑀸’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀢𑁆𑀣𑀺 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀅𑀧𑁂𑀓𑁆𑀔𑀸’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀫𑁂𑀳𑀺 𑀤𑀺𑀩𑁆𑀩𑀸 𑀓𑀸𑀫𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀢𑀭𑀸 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑀸 𑀘𑁇 𑀲𑀸𑀥𑀸𑀬𑀲𑁆𑀫𑀸, 𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀳𑀺 𑀓𑀸𑀫𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀫𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇

2115\. 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀳𑀺 𑀫𑁂 𑀓𑀸𑀫𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁, 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀫𑁄𑀘𑀺𑀢𑀦𑁆’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁂𑀯𑁂𑀳𑀺 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀤𑁂𑀯𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀢𑀭𑀸 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑀸 𑀘𑁇 𑀲𑀸𑀥𑀸𑀬𑀲𑁆𑀫𑀸, 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀳𑀺 𑀤𑁂𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀫𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇

2116\. 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀳𑀺 𑀫𑁂 𑀤𑁂𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁, 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀫𑁄𑀘𑀺𑀢𑀦𑁆’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁂𑀯𑁂𑀳𑀺 𑀬𑀸𑀫𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀢𑀼𑀲𑀺𑀢𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁂𑀯𑁂𑀳𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇 𑀲𑀸𑀥𑀸𑀬𑀲𑁆𑀫𑀸, 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀳𑀺 𑀤𑁂𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀫𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀳𑀺 𑀫𑁂 𑀤𑁂𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁, 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀫𑁄𑀘𑀺𑀢𑀦𑁆’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁄𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀺𑀘𑁆𑀘𑁄 𑀅𑀤𑁆𑀥𑀼𑀯𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀲𑀸𑀥𑀸𑀬𑀲𑁆𑀫𑀸, 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀸𑀳𑀻’𑀢𑀺𑁇

2117\. 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁, 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀸𑀫𑀻’𑀢𑀺; 𑀏𑀯𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀆𑀲𑀯𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀯𑀤𑀸𑀫𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇
