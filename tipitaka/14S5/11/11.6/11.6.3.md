# 𑁧𑁧𑁇𑁬𑁇𑁩 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀯𑀕𑁆𑀕

* 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

(1049.)

2105\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀑𑀯𑀤𑀢𑀼 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸; 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀼 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀬𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

2106\. “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀯𑁄, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀬𑁂 𑀢𑁂 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸 𑀢𑀣𑀸𑀕𑀢𑀪𑀸𑀲𑀺𑀢𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸 𑀲𑀼𑀜𑁆𑀜𑀢𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀢𑁂 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 “𑀦 𑀔𑁄 𑀦𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀓𑀭𑀁 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀬𑀦𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀦𑁆𑀢𑁂𑀳𑀺 𑀓𑀸𑀲𑀺𑀓𑀘𑀦𑁆𑀤𑀦𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑁂𑀳𑀺 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑁂𑀳𑀺 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁂𑀳𑀺—  𑀬𑁂 𑀢𑁂 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸 𑀢𑀣𑀸𑀕𑀢𑀪𑀸𑀲𑀺𑀢𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸 𑀲𑀼𑀜𑁆𑀜𑀢𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀢𑁂 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀜𑁆𑀘𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀞𑀺𑀢𑀸𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀽”𑀢𑀺𑁇

2107\. “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀯𑁄, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

2108\. “𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀺, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀫𑁆𑀳𑁂𑀲𑀼, 𑀫𑀬𑀜𑁆𑀘 𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀫𑀬𑀜𑁆𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀮𑀸𑀪𑀸 𑀯𑁄, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑁄, 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑁇 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀩𑁆𑀬𑀸𑀓𑀢𑀦𑁆”𑀢𑀺𑁇
