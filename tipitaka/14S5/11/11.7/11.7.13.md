# 𑁧𑁧𑁇𑁭𑁇𑁧𑁩 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

(1070.)

2160\. “…  𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑁂𑀯𑁄, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2162\. 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇



---

2163\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2164\. _𑀫𑀳𑀸 𑀧𑀼𑀣𑀼 𑀯𑀺𑀧𑀼𑀮 𑀕𑀫𑁆𑀪𑀻𑀭𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢 𑀪𑀽𑀭𑀺 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀁;_  
_𑀲𑀻𑀖 𑀮𑀳𑀼 𑀳𑀸𑀲 𑀚𑀯𑀦,_  
_𑀢𑀺𑀓𑁆𑀔 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀸𑀬 𑀘𑀸𑀢𑀺𑁇_
