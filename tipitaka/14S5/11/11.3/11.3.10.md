# 𑁧𑁧𑁇𑁩𑁇𑁧𑁦 𑀦𑀦𑁆𑀤𑀓𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀭𑀡𑀸𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀦𑀦𑁆𑀤𑀓𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀲𑀼𑀢𑁆𑀢

(1026.)

1988\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀦𑁆𑀤𑀓𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀦𑀦𑁆𑀤𑀓𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1989\. “𑀘𑀢𑀽𑀳𑀺 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀓, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀦𑀦𑁆𑀤𑀓, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀓, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄𑁇

1990\. 𑀇𑀫𑁂𑀳𑀺 𑀘 𑀧𑀦, 𑀦𑀦𑁆𑀤𑀓, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀬𑀼𑀦𑀸 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦𑀧𑀺 𑀫𑀸𑀦𑀼𑀲𑁂𑀦𑀧𑀺; 𑀯𑀡𑁆𑀡𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦𑀧𑀺 𑀫𑀸𑀦𑀼𑀲𑁂𑀦𑀧𑀺; 𑀲𑀼𑀔𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦𑀧𑀺 𑀫𑀸𑀦𑀼𑀲𑁂𑀦𑀧𑀺; 𑀬𑀲𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦𑀧𑀺 𑀫𑀸𑀦𑀼𑀲𑁂𑀦𑀧𑀺; 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦𑀧𑀺 𑀫𑀸𑀦𑀼𑀲𑁂𑀦𑀧𑀺𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀦𑀦𑁆𑀤𑀓, 𑀦𑀸𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀬𑀤𑁂𑀯 𑀫𑀬𑀸 𑀲𑀸𑀫𑀁 𑀜𑀸𑀢𑀁 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀸𑀫𑀁 𑀯𑀺𑀤𑀺𑀢𑀁, 𑀢𑀤𑁂𑀯𑀸𑀳𑀁 𑀯𑀤𑀸𑀫𑀻”𑀢𑀺𑁇

1991\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦𑀦𑁆𑀤𑀓𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀳𑀸𑀦𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀅𑀮𑀁 𑀤𑀸𑀦𑀺, 𑀪𑀡𑁂, 𑀏𑀢𑁂𑀦 𑀩𑀸𑀳𑀺𑀭𑁂𑀦 𑀦𑀳𑀸𑀦𑁂𑀦𑁇 𑀅𑀮𑀫𑀺𑀤𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀦𑀳𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀪𑀕𑀯𑀢𑀺 𑀧𑀲𑀸𑀤𑁄”𑀢𑀺𑁇

---

1992\. 𑀤𑀲𑀫𑀁𑁇



---

1993\. 𑀲𑀭𑀡𑀸𑀦𑀺𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

1994\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1995\. _𑀫𑀳𑀸𑀦𑀸𑀫𑁂𑀦 𑀤𑁆𑀯𑁂 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀕𑁄𑀥𑀸 𑀘 𑀲𑀭𑀡𑀸 𑀤𑀼𑀯𑁂;_  
_𑀤𑀼𑀯𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀸,_  
_𑀤𑀼𑀯𑁂 𑀯𑁂𑀭𑀪𑀬𑁂𑀦 𑀘;_  
_𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀤𑀲𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄,_  
_𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
