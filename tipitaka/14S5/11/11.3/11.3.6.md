# 𑁧𑁧𑁇𑁩𑁇𑁬 𑀧𑀞𑀫𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀭𑀡𑀸𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(1022.)

1942\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤—  ‘𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀳𑀺—  ‘𑀲𑀸𑀥𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’”𑀢𑀺𑁇

1943\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1944\. “𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑀢𑀺—  ‘𑀲𑀸𑀥𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

1945\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀧𑀘𑁆𑀙𑀸𑀲𑀫𑀡𑁂𑀦 𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀔𑀫𑀦𑀻𑀬𑀁 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁? 𑀓𑀘𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁄𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁄”𑀢𑀺? “𑀦 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀦 𑀬𑀸𑀧𑀦𑀻𑀬𑀁𑁇 𑀩𑀸𑀍𑀳𑀸 𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁄𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁄”𑀢𑀺𑁇

1946\. “𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁄 𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁄 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁄—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑁇

1947\. 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑁄 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁄 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁄—  𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑁇

1948\. 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀁𑀖𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑁄 𑀢𑁂 𑀲𑀁𑀖𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁄 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁄—  𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑁇

1949\. 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀢𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑀸𑀦𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀦𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘 𑀧𑀦 𑀢𑁂 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑁇

1950\. 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑁇

1957\. 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑁄 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑁇

1958\. 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀜𑀸𑀡𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀜𑀸𑀡𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀁𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑁇

1959\. 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀦𑀢𑁆𑀣𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

1960\. 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀞𑀸𑀦𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀆𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁂𑀦𑁂𑀯 𑀣𑀸𑀮𑀺𑀧𑀸𑀓𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀦𑀻𑀘𑀸𑀲𑀦𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—

1961\. _“𑀬𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀢𑀣𑀸𑀕𑀢𑁂,_  
_𑀅𑀘𑀮𑀸 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀲𑀻𑀮𑀜𑁆𑀘 𑀬𑀲𑁆𑀲 𑀓𑀮𑁆𑀬𑀸𑀡𑀁,_  
_𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑀁 𑀧𑀲𑀁𑀲𑀺𑀢𑀁𑁇_  


1962\. _𑀲𑀁𑀖𑁂 𑀧𑀲𑀸𑀤𑁄 𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀺,_  
_𑀉𑀚𑀼𑀪𑀽𑀢𑀜𑁆𑀘 𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀅𑀤𑀮𑀺𑀤𑁆𑀤𑁄𑀢𑀺 𑀢𑀁 𑀆𑀳𑀼,_  
_𑀅𑀫𑁄𑀖𑀁 𑀢𑀲𑁆𑀲 𑀚𑀻𑀯𑀺𑀢𑀁𑁇_  


1963\. _𑀢𑀲𑁆𑀫𑀸 𑀲𑀤𑁆𑀥𑀜𑁆𑀘 𑀲𑀻𑀮𑀜𑁆𑀘,_  
_𑀧𑀲𑀸𑀤𑀁 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑁂𑀣 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀲𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


1964\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1965\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀳𑀦𑁆𑀤 𑀓𑀼𑀢𑁄 𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀆𑀕𑀘𑁆𑀙𑀲𑀺 𑀤𑀺𑀯𑀸𑀤𑀺𑀯𑀲𑁆𑀲𑀸”𑀢𑀺? “𑀆𑀬𑀲𑁆𑀫𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀑𑀯𑀸𑀤𑁂𑀦 𑀑𑀯𑀤𑀺𑀢𑁄”𑀢𑀺𑁇 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄; 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺 𑀤𑀲𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

---

1966\. 𑀙𑀝𑁆𑀞𑀁𑁇
