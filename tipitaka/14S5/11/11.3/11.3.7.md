# 𑁧𑁧𑁇𑁩𑁇𑁭 𑀤𑀼𑀢𑀺𑀬𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀭𑀡𑀸𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(1023.)

1967. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤—  ‘𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀳𑀺—  ‘𑀲𑀸𑀥𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’”𑀢𑀺𑁇

1968. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑀢𑀺—  ‘𑀲𑀸𑀥𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

1969. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁? 𑀓𑀘𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁄𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁄”𑀢𑀺? “𑀦 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀫𑀦𑀻𑀬𑀁 𑀦 𑀬𑀸𑀧𑀦𑀻𑀬𑀁𑁇 𑀩𑀸𑀍𑀳𑀸 𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁄𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁄”𑀢𑀺𑁇

1970. “𑀘𑀢𑀽𑀳𑀺 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

1971. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

1972. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀲𑀁𑀖𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦𑀲𑁆𑀲 𑀲𑀁𑀖𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

1973. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦𑀲𑁆𑀲 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

1974. 𑀘𑀢𑀽𑀳𑀺 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀦 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

1975. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀦 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

1976. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀧𑀦𑀲𑁆𑀲 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀦 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

1977. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘 𑀧𑀦𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑁄 𑀦 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀫𑀭𑀡𑀪𑀬𑀦𑁆”𑀢𑀺𑁇

1978. “𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀸𑀬𑀸𑀫𑀺𑁇 𑀓𑁆𑀬𑀸𑀳𑀁 𑀪𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀫𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀫𑀺—  𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀬𑀸𑀦𑀺 𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀕𑀺𑀳𑀺𑀲𑀸𑀫𑀻𑀘𑀺𑀓𑀸𑀦𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀺, 𑀦𑀸𑀳𑀁 𑀢𑁂𑀲𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀢𑁆𑀢𑀦𑀺 𑀔𑀡𑁆𑀟𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀮𑀸𑀪𑀸 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺𑁇 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀢𑀬𑀸, 𑀕𑀳𑀧𑀢𑀺, 𑀩𑁆𑀬𑀸𑀓𑀢𑀦𑁆”𑀢𑀺𑁇

---

1979. 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
