# 𑁧𑁧𑁇𑁩𑁇𑁪 𑀧𑀞𑀫𑀲𑀭𑀡𑀸𑀦𑀺𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀭𑀡𑀸𑀦𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀲𑀭𑀡𑀸𑀦𑀺𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(1020.)

1920\. 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀭𑀡𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄—  “𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄”𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀓𑁆𑀓𑀸 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀻𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀏𑀢𑁆𑀣 𑀤𑀸𑀦𑀺 𑀓𑁄 𑀦 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀭𑀡𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄; 𑀲𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄—  ‘𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑀢𑀺𑁇 𑀲𑀭𑀡𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑁄 𑀲𑀺𑀓𑁆𑀔𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀫𑀸𑀧𑀸𑀤𑀺, 𑀫𑀚𑁆𑀚𑀧𑀸𑀦𑀁 𑀅𑀧𑀸𑀬𑀻”𑀢𑀺𑁇

1921\. 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀭𑀡𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀲𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄—  ‘𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀓𑁆𑀓𑀸 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀻𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢, 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢, 𑀪𑁄𑁇 𑀏𑀢𑁆𑀣 𑀤𑀸𑀦𑀺 𑀓𑁄 𑀦 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀭𑀡𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄; 𑀲𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄—  ‘𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑀢𑀺𑁇 𑀲𑀭𑀡𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑁄 𑀲𑀺𑀓𑁆𑀔𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀫𑀸𑀧𑀸𑀤𑀺, 𑀫𑀚𑁆𑀚𑀧𑀸𑀦𑀁 𑀅𑀧𑀸𑀬𑀻”𑀢𑀺𑁇

1922\. “𑀬𑁄 𑀲𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄, 𑀲𑁄 𑀓𑀣𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄’𑀢𑀺, 𑀲𑀭𑀡𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀲𑀭𑀡𑀸𑀦𑀺, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀲𑀓𑁆𑀓𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄𑁇 𑀲𑁄 𑀓𑀣𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇

1923\. 𑀇𑀥, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑁄 𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀘 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀲𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀦𑀺𑀭𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀸𑁇

1924\. 𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑁄 𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑁄 𑀦 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀲𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀦𑀺𑀭𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀸𑁇

1925\. 𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀦 𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑁄 𑀦 𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑁄 𑀦 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀲𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀸𑀦𑀁 𑀢𑀦𑀼𑀢𑁆𑀢𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀓𑀺𑀤𑁂𑀯 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀦𑀺𑀭𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀸𑁇

1926\. 𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺; 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀦 𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑁄 𑀦 𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑁄 𑀦 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀲𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀦𑀺𑀭𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀸𑁇

1927\. 𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀦 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀦 𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀦 𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑁄 𑀦 𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑁄 𑀦 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀅𑀧𑀺 𑀘𑀲𑁆𑀲 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑀸 𑀘𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑀢𑁆𑀢𑀲𑁄 𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀁 𑀔𑀫𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀕𑀦𑁆𑀢𑀸 𑀦𑀺𑀭𑀬𑀁 𑀅𑀕𑀦𑁆𑀢𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀅𑀕𑀦𑁆𑀢𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁 𑀅𑀕𑀦𑁆𑀢𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁𑁇

1928\. 𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀦 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀦 𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀦 𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑁄 𑀦 𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑁄 𑀦 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀧𑀺 𑀘𑀲𑁆𑀲 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀢𑀣𑀸𑀕𑀢𑁂 𑀘𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀫𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑁂𑀫𑀫𑀢𑁆𑀢𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀕𑀦𑁆𑀢𑀸 𑀦𑀺𑀭𑀬𑀁 𑀅𑀕𑀦𑁆𑀢𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀅𑀕𑀦𑁆𑀢𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁 𑀅𑀕𑀦𑁆𑀢𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁𑁇 𑀇𑀫𑁂 𑀘𑁂𑀧𑀺, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁, 𑀇𑀫𑁂 𑀘𑀸𑀳𑀁 𑀫𑀳𑀸𑀲𑀸𑀮𑁂 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁—  ‘𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸’𑀢𑀺; 𑀓𑀺𑀫𑀗𑁆𑀕𑀁 𑀧𑀦 𑀲𑀭𑀡𑀸𑀦𑀺𑀁 𑀲𑀓𑁆𑀓𑀁𑁇 𑀲𑀭𑀡𑀸𑀦𑀺, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀲𑀓𑁆𑀓𑁄 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀻”𑀢𑀺𑁇
