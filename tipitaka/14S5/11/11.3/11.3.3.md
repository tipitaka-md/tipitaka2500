# 𑁧𑁧𑁇𑁩𑁇𑁩 𑀕𑁄𑀥𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀭𑀡𑀸𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀕𑁄𑀥𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(1019.)

1911\. 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦 𑀕𑁄𑀥𑀸 𑀲𑀓𑁆𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀕𑁄𑀥𑀁 𑀲𑀓𑁆𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀺𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑁄𑀥𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀦𑁆”𑀢𑀺?

1912\. “𑀢𑀻𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀥, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀁𑁇

1913\. 𑀢𑁆𑀯𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀓𑀢𑀺𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀦𑁆”𑀢𑀺? “𑀘𑀢𑀽𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀕𑁄𑀥𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀕𑁄𑀥𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀕𑁄𑀥𑁂, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀦𑁆”𑀢𑀺𑁇

1914\. “𑀆𑀕𑀫𑁂𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀆𑀕𑀫𑁂𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫𑁇 𑀪𑀕𑀯𑀸𑀯 𑀏𑀢𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬 𑀏𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀯𑀸 𑀅𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀯𑀸”𑀢𑀺𑁇 “𑀆𑀬𑀸𑀫, 𑀕𑁄𑀥𑁂, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀕𑁄𑀥𑀸 𑀘 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1915\. “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀕𑁄𑀥𑀸 𑀲𑀓𑁆𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀕𑁄𑀥𑀁 𑀲𑀓𑁆𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀓𑀢𑀺𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑁄𑀥𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀁’? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑁄𑀥𑀸 𑀲𑀓𑁆𑀓𑁄 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1916\. 𑀢𑀻𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀥, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀁𑁇 𑀢𑁆𑀯𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀓𑀢𑀫𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀦𑁆’𑀢𑀺?

1917\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑁄𑀥𑀁 𑀲𑀓𑁆𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀘𑀢𑀽𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀕𑁄𑀥𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀕𑁄𑀥𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀕𑁄𑀥𑁂, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀬𑀢𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀦𑁆’𑀢𑀺𑁇

1918\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑁄𑀥𑀸 𑀲𑀓𑁆𑀓𑁄 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀆𑀕𑀫𑁂𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫, 𑀆𑀕𑀫𑁂𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀦𑀸𑀫𑁇 𑀪𑀕𑀯𑀸𑀯 𑀏𑀢𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬 𑀏𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀯𑀸 𑀅𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀯𑀸’𑀢𑀺𑁇 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀏𑀓𑀢𑁄 𑀅𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀏𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀘𑁇 𑀬𑁂𑀦𑁂𑀯 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑁂𑀯𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼𑁇 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀏𑀓𑀢𑁄 𑀅𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀏𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀘𑁇 𑀬𑁂𑀦𑁂𑀯 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑁂𑀯𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼𑁇 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀏𑀓𑀢𑁄 𑀅𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀏𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀘 𑀉𑀧𑀸𑀲𑀓𑀸 𑀘𑁇 𑀬𑁂𑀦𑁂𑀯 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑁂𑀯𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼𑁇 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀏𑀓𑀢𑁄 𑀅𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀏𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀉𑀧𑀸𑀲𑀓𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄 𑀘𑁇 𑀬𑁂𑀦𑁂𑀯 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑁂𑀯𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼𑁇 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀏𑀓𑀢𑁄 𑀅𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀏𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀉𑀧𑀸𑀲𑀓𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄 𑀲𑀤𑁂𑀯𑀓𑁄 𑀘 𑀮𑁄𑀓𑁄 𑀲𑀫𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁄 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀧𑀚𑀸 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑁇 𑀬𑁂𑀦𑁂𑀯 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑁂𑀯𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀁𑁇 𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀽”𑀢𑀺𑁇 “𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀢𑁆𑀯𑀁, 𑀕𑁄𑀥𑁂, 𑀫𑀳𑀸𑀦𑀸𑀫𑀁 𑀲𑀓𑁆𑀓𑀁 𑀓𑀺𑀁 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀁𑀯𑀸𑀤𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀦𑀸𑀫𑀁 𑀲𑀓𑁆𑀓𑀁 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑀤𑀸𑀫𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀼𑀲𑀮𑀸”𑀢𑀺𑁇
