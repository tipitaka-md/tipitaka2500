# 𑁧𑁧𑁇𑁫𑁇𑁧𑁦 𑀅𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀕𑀸𑀣𑀓𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀅𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(1046.)

2086\. “𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺? 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑁂𑀯𑁄, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀻”𑀢𑀺𑁇

2089\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2090\. _𑀅𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀢𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀼𑀯𑁂 𑀫𑀳𑀤𑁆𑀥𑀦𑁂𑀦 𑀘;_
