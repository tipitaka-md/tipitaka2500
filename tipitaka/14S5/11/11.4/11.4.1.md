# 𑁧𑁧𑁇𑁪𑁇𑁧 𑀧𑀞𑀫𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(1027.)

1996. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄𑁇

1997. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄𑁇

1998. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄𑁇

1999. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑀸”𑀢𑀺𑁇
