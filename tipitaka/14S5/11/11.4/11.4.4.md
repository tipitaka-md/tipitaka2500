# 𑁧𑁧𑁇𑁪𑁇𑁪 𑀧𑀞𑀫𑀤𑁂𑀯𑀧𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀤𑁂𑀯𑀧𑀤𑀲𑀼𑀢𑁆𑀢

(1030.)

2009\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀦𑀁 𑀤𑁂𑀯𑀧𑀤𑀸𑀦𑀺 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑁇

2010\. 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀤𑁂𑀯𑀧𑀤𑀁 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑁇

2011\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂… 𑁇

2012\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀤𑁂𑀯𑀧𑀤𑀁 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀤𑁂𑀯𑀸𑀦𑀁 𑀤𑁂𑀯𑀧𑀤𑀸𑀦𑀺 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑀸”𑀢𑀺𑁇
