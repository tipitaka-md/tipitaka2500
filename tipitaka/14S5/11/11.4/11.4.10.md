# 𑁧𑁧𑁇𑁪𑁇𑁧𑁦 𑀦𑀦𑁆𑀤𑀺𑀬𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀺𑀬𑀲𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(1036.)

2035\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀲𑁆𑀲𑁂𑀯 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀦𑀢𑁆𑀣𑀺 𑀲𑁆𑀯𑁂𑀯 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻”𑀢𑀺𑁇

2036\. “‘𑀬𑀲𑁆𑀲 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀦𑀢𑁆𑀣𑀺 𑀢𑀫𑀳𑀁 𑀩𑀸𑀳𑀺𑀭𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀧𑀓𑁆𑀔𑁂 𑀞𑀺𑀢𑁄’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀧𑀺 𑀘, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀬𑀣𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻 𑀘 𑀢𑀁 𑀲𑀼𑀡𑀸𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

2037\. “𑀓𑀣𑀜𑁆𑀘, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀦 𑀉𑀢𑁆𑀢𑀭𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀤𑀺𑀯𑀸 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀬, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁂 𑀅𑀲𑀢𑀺, 𑀧𑀻𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀧𑀻𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀲𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

2038\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀲𑁄 𑀢𑁂𑀳𑀺 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀦 𑀉𑀢𑁆𑀢𑀭𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀤𑀺𑀯𑀸 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁂 𑀅𑀲𑀢𑀺, 𑀧𑀻𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀧𑀻𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀲𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇

2039\. 𑀓𑀣𑀜𑁆𑀘, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀤𑀺𑀯𑀸 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺𑁇 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺𑁇 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺𑁇 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺𑁇 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

2040\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀲𑁄 𑀢𑁂𑀳𑀺 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀤𑀺𑀯𑀸 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺𑁇 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺𑁇 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺𑁇 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺𑁇 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀦𑀦𑁆𑀤𑀺𑀬, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
