# 𑁧𑁧𑁇𑁪𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(1028.)

2001. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄𑁇

2002. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂… 𑁇

2003. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺 𑀫𑀼𑀢𑁆𑀢𑀘𑀸𑀕𑁄 𑀧𑀬𑀢𑀧𑀸𑀡𑀺 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀭𑀢𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑁄𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀓𑀼𑀲𑀮𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀸 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑀸”𑀢𑀺𑁇

---

2004. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
