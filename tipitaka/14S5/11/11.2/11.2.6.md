# 𑁧𑁧𑁇𑁨𑁇𑁬 𑀧𑀞𑀫𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀭𑀸𑀚𑀓𑀸𑀭𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢

(1012.)

1883\. “𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀸𑀣, 𑀬𑁂 𑀘 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀼𑀁—  𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸—  𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀲𑀼 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑁂𑀲𑀼 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸, 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼? 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸, 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀲𑀼 𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸, 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸 𑀅𑀔𑀡𑁆𑀟𑁂𑀲𑀼…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀸𑀣, 𑀬𑁂 𑀘 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀼𑀁—  𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸—  𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑁂𑀲𑀼 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸, 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
