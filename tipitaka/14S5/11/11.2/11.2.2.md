# 𑁧𑁧𑁇𑁨𑁇𑁨 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀸𑀚𑀓𑀸𑀭𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(1008.)

1869. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀦𑀸𑀫 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀲𑀸𑀯𑀓𑀁 𑀏𑀯𑀁 𑀲𑀫𑀸𑀤𑀧𑁂𑀦𑁆𑀢𑀺—  ‘𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀓𑀸𑀮𑀲𑁆𑀲𑁂𑀯 𑀉𑀝𑁆𑀞𑀸𑀬 𑀧𑀸𑀘𑀻𑀦𑀫𑀼𑀔𑁄 𑀬𑀸𑀳𑀺𑁇 𑀲𑁄 𑀢𑁆𑀯𑀁 𑀫𑀸 𑀲𑁄𑀩𑁆𑀪𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀳𑀺, 𑀫𑀸 𑀧𑀧𑀸𑀢𑀁, 𑀫𑀸 𑀔𑀸𑀡𑀼𑀁, 𑀫𑀸 𑀓𑀡𑁆𑀟𑀓𑀞𑀸𑀦𑀁, 𑀫𑀸 𑀘𑀦𑁆𑀤𑀦𑀺𑀬𑀁, 𑀫𑀸 𑀑𑀍𑀇𑀕𑀮𑁆𑀮𑀁𑁇 𑀬𑀢𑁆𑀣 𑀧𑀧𑀢𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀭𑀡𑀁 𑀆𑀕𑀫𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀏𑀯𑀁 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇

1870. 𑀢𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀩𑀸𑀮𑀕𑀫𑀦𑀫𑁂𑀢𑀁 𑀫𑀽𑀍𑀳𑀕𑀫𑀦𑀫𑁂𑀢𑀁 𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀦 𑀉𑀧𑀲𑀫𑀸𑀬 𑀦 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀦 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀉𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀫𑀺; 𑀬𑀸 𑀏𑀓𑀦𑁆𑀢𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

1871. 𑀓𑀢𑀫𑀸 𑀘 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸; 𑀬𑀸 𑀏𑀓𑀦𑁆𑀢𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬…𑀧𑁂…  𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺; 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬…𑀧𑁂…  𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
