# 𑁧𑁧𑁇𑁨𑁇𑁧𑁦 𑀢𑀢𑀺𑀬𑀤𑁂𑀯𑀘𑀸𑀭𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀭𑀸𑀚𑀓𑀸𑀭𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀢𑀢𑀺𑀬𑀤𑁂𑀯𑀘𑀸𑀭𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(1016.)

1896\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1897\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸”𑀢𑀺𑁇

1898\. “𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀏𑀯𑀫𑀬𑀁 𑀧𑀚𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀁 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀫𑀦𑀳𑁂𑀢𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀏𑀯𑀫𑀬𑀁 𑀧𑀚𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸”𑀢𑀺𑁇

1901\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1902\. _𑀲𑀳𑀲𑁆𑀲𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀦𑁆𑀤,_

_𑀢𑀬𑁄 𑀘 𑀤𑁂𑀯𑀘𑀸𑀭𑀺𑀓𑀸𑀢𑀺𑁇_
