# 𑁧𑁧𑁇𑁧𑁇𑁫 𑀤𑀼𑀢𑀺𑀬𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(1001.)

1808\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “‘𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀦𑁆’𑀢𑀺 𑀳𑀺𑀤𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀦𑁆”𑀢𑀺? “𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑁂𑀯𑁄 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑁂𑀯𑁄 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀁𑁇

1809\. ‘𑀲𑁄𑀢𑁄, 𑀲𑁄𑀢𑁄’𑀢𑀺 𑀳𑀺𑀤𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑁄𑀢𑁄”𑀢𑀺? “𑀅𑀬𑀫𑁂𑀯 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑁄𑀢𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀅𑀬𑀫𑁂𑀯 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑁄𑀢𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇

1810\. ‘𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄’𑀢𑀺 𑀳𑀺𑀤𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? “𑀬𑁄 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀭𑀺𑀬𑁂𑀦 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁆𑀯𑀸𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁𑀦𑀸𑀫𑁄 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀬𑁄 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀭𑀺𑀬𑁂𑀦 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁆𑀯𑀸𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁𑀦𑀸𑀫𑁄 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑁄”𑀢𑀺𑁇
