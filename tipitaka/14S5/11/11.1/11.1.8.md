# 𑁧𑁧𑁇𑁧𑁇𑁮 𑀧𑀞𑀫𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑀲𑀼𑀢𑁆𑀢

(1004.)

1844\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀜𑀸𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1845\. “𑀲𑀸𑀍𑀳𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄; 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀦𑀦𑁆𑀤𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸; 𑀢𑀲𑁆𑀲𑀸 𑀓𑀸 𑀕𑀢𑀺 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀲𑀼𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄; 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀲𑀼𑀚𑀸𑀢𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸; 𑀢𑀲𑁆𑀲𑀸 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄”𑀢𑀺?

1846\. “𑀲𑀸𑀍𑀳𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 𑀦𑀦𑁆𑀤𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀦𑀻 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀸 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇 𑀲𑀼𑀤𑀢𑁆𑀢𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀸𑀦𑀁 𑀢𑀦𑀼𑀢𑁆𑀢𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻; 𑀲𑀓𑀺𑀤𑁂𑀯 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀼𑀚𑀸𑀢𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸𑁇

1847\. 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁂𑀬𑁆𑀬; 𑀢𑀲𑁆𑀫𑀺𑀁 𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑁂 𑀫𑀁 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀣𑁇 𑀯𑀺𑀳𑁂𑀲𑀸 𑀧𑁂𑀲𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀲𑁆𑀲 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑀁 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺; 𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑁇

1848\. 𑀓𑀢𑀫𑁄 𑀘 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄; 𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’?

1849\. 𑀇𑀥, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄; 𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’”𑀢𑀺𑁇
