# 𑁧𑁧𑁇𑁧𑁇𑁬 𑀣𑀧𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀣𑀧𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(1002.)

1812\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  “𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑀸 𑀣𑀧𑀢𑀬𑁄 𑀲𑀸𑀥𑀼𑀓𑁂 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑀸 𑀣𑀧𑀢𑀬𑁄—  “𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  ‘𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇

1813\. 𑀅𑀣 𑀔𑁄 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑀸 𑀣𑀧𑀢𑀬𑁄 𑀫𑀕𑁆𑀕𑁂 𑀧𑀼𑀭𑀺𑀲𑀁 𑀞𑀧𑁂𑀲𑀼𑀁—  “𑀬𑀤𑀸 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁, 𑀅𑀣 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 𑀤𑁆𑀯𑀻𑀳𑀢𑀻𑀳𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀤𑁆𑀤𑀲 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑀸 𑀣𑀧𑀢𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑁂 𑀣𑀧𑀢𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀣𑀸”𑀢𑀺𑁇

1814\. 𑀅𑀣 𑀔𑁄 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑀸 𑀣𑀧𑀢𑀬𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑀸 𑀣𑀧𑀢𑀬𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀇𑀲𑀺𑀤𑀢𑁆𑀢𑀧𑀼𑀭𑀸𑀡𑀸 𑀣𑀧𑀢𑀬𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—

1815\. “𑀬𑀤𑀸 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1816\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀓𑁄𑀲𑀮𑁂𑀳𑀺 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀓𑁄𑀲𑀮𑁂𑀳𑀺 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1817\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀫𑀮𑁆𑀮𑁂𑀳𑀺 𑀯𑀚𑁆𑀚𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀫𑀮𑁆𑀮𑁂𑀳𑀺 𑀯𑀚𑁆𑀚𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1818\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀯𑀚𑁆𑀚𑀻𑀳𑀺 𑀓𑀸𑀲𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀯𑀚𑁆𑀚𑀻𑀳𑀺 𑀓𑀸𑀲𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1819\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀓𑀸𑀲𑀻𑀳𑀺 𑀫𑀸𑀕𑀥𑁂 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀓𑀸𑀲𑀻𑀳𑀺 𑀫𑀸𑀕𑀥𑁂 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀧𑁆𑀧𑀓𑀸 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀦𑀧𑁆𑀧𑀓𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀤𑀽𑀭𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1820\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀫𑀸𑀕𑀥𑁂𑀳𑀺 𑀓𑀸𑀲𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀫𑀸𑀕𑀥𑁂𑀳𑀺 𑀓𑀸𑀲𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1821\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀓𑀸𑀲𑀻𑀳𑀺 𑀯𑀚𑁆𑀚𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀓𑀸𑀲𑀻𑀳𑀺 𑀯𑀚𑁆𑀚𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1822\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀯𑀚𑁆𑀚𑀻𑀳𑀺 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀯𑀚𑁆𑀚𑀻𑀳𑀺 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1823\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀫𑀮𑁆𑀮𑁂𑀳𑀺 𑀓𑁄𑀲𑀮𑁂 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀫𑀮𑁆𑀮𑁂𑀳𑀺 𑀓𑁄𑀲𑀮𑁂 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1824\. 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀓𑁄𑀲𑀮𑁂𑀳𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀼𑀡𑀸𑀫—  ‘𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂’𑀢𑀺, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀧𑁆𑀧𑀓𑀸 𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀦𑀧𑁆𑀧𑀓𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  ‘𑀆𑀲𑀦𑁆𑀦𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸’”𑀢𑀺𑁇

1825\. “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀣𑀧𑀢𑀬𑁄, 𑀲𑀫𑁆𑀩𑀸𑀥𑁄 𑀖𑀭𑀸𑀯𑀸𑀲𑁄 𑀭𑀚𑀸𑀧𑀣𑁄, 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑁇 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀯𑁄, 𑀣𑀧𑀢𑀬𑁄, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑁆𑀩𑀸𑀥𑀸 𑀅𑀜𑁆𑀜𑁄 𑀲𑀫𑁆𑀩𑀸𑀥𑁄 𑀲𑀫𑁆𑀩𑀸𑀥𑀢𑀭𑁄 𑀘𑁂𑀯 𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀗𑁆𑀔𑀸𑀢𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇 “𑀓𑀢𑀫𑁄 𑀧𑀦 𑀯𑁄, 𑀣𑀧𑀢𑀬𑁄, 𑀏𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑁆𑀩𑀸𑀥𑀸 𑀅𑀜𑁆𑀜𑁄 𑀲𑀫𑁆𑀩𑀸𑀥𑁄 𑀲𑀫𑁆𑀩𑀸𑀥𑀢𑀭𑁄 𑀘𑁂𑀯 𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀗𑁆𑀔𑀸𑀢𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺?

1826\. “𑀇𑀥 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀤𑀸 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀢𑁂 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀦𑀸𑀕𑀸 𑀑𑀧𑀯𑀬𑁆𑀳𑀸 𑀢𑁂 𑀓𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸, 𑀬𑀸 𑀢𑀸 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀧𑀺𑀬𑀸 𑀫𑀦𑀸𑀧𑀸 𑀢𑀸 𑀏𑀓𑀁 𑀧𑀼𑀭𑀢𑁄 𑀏𑀓𑀁 𑀧𑀘𑁆𑀙𑀢𑁄 𑀦𑀺𑀲𑀻𑀤𑀸𑀧𑁂𑀫𑁇 𑀢𑀸𑀲𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀏𑀯𑀭𑀽𑀧𑁄 𑀕𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀦𑁆𑀥𑀓𑀭𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀢𑀸𑀯𑀤𑁂𑀯 𑀯𑀺𑀯𑀭𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲, 𑀬𑀣𑀸 𑀢𑀁 𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀦𑀁 𑀕𑀦𑁆𑀥𑁂𑀦 𑀯𑀺𑀪𑀽𑀲𑀺𑀢𑀸𑀦𑀁𑁇 𑀢𑀸𑀲𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀏𑀯𑀭𑀽𑀧𑁄 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀢𑀽𑀮𑀧𑀺𑀘𑀼𑀦𑁄 𑀯𑀸 𑀓𑀧𑁆𑀧𑀸𑀲𑀧𑀺𑀘𑀼𑀦𑁄 𑀯𑀸, 𑀬𑀣𑀸 𑀢𑀁 𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸𑀦𑀁𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑁂 𑀦𑀸𑀕𑁄𑀧𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀸𑀧𑀺 𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀭𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀅𑀢𑁆𑀢𑀸𑀧𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫 𑀢𑀸𑀲𑀼 𑀪𑀕𑀺𑀦𑀻𑀲𑀼 𑀧𑀸𑀧𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀸𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑁆𑀩𑀸𑀥𑀸 𑀅𑀜𑁆𑀜𑁄 𑀲𑀫𑁆𑀩𑀸𑀥𑁄 𑀲𑀫𑁆𑀩𑀸𑀥𑀢𑀭𑁄 𑀘𑁂𑀯 𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀗𑁆𑀔𑀸𑀢𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇

1827\. “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀣𑀧𑀢𑀬𑁄, 𑀲𑀫𑁆𑀩𑀸𑀥𑁄 𑀖𑀭𑀸𑀯𑀸𑀲𑁄 𑀭𑀚𑀸𑀧𑀣𑁄, 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑁇 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀯𑁄, 𑀣𑀧𑀢𑀬𑁄, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀸𑀬𑁇 𑀘𑀢𑀽𑀳𑀺 𑀔𑁄, 𑀣𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄𑁇

1828\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀣𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀚𑁆𑀛𑀸𑀕𑀸𑀭𑀁 𑀯𑀲𑀢𑀺 𑀫𑀼𑀢𑁆𑀢𑀘𑀸𑀕𑁄 𑀧𑀬𑀢𑀧𑀸𑀡𑀺 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀭𑀢𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑁄𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀣𑀧𑀢𑀬𑁄, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄𑁇

1829\. 𑀢𑀼𑀫𑁆𑀳𑁂 𑀔𑁄, 𑀣𑀧𑀢𑀬𑁄, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀬𑀁 𑀔𑁄 𑀧𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀼𑀮𑁂 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂𑀳𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀣𑀧𑀢𑀬𑁄, 𑀓𑀢𑀺𑀯𑀺𑀥𑀸 𑀢𑁂 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀬𑁂 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀫𑀲𑀫𑀸, 𑀬𑀤𑀺𑀤𑀁—  𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑁂”𑀢𑀺? “𑀮𑀸𑀪𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀬𑁂𑀲𑀁 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
