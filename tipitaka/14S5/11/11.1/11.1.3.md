# 𑁧𑁧𑁇𑁧𑁇𑁩 𑀤𑀻𑀖𑀸𑀯𑀼𑀉𑀧𑀸𑀲𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀤𑀻𑀖𑀸𑀯𑀼𑀉𑀧𑀸𑀲𑀓𑀲𑀼𑀢𑁆𑀢

(999.)

1799\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑀻𑀖𑀸𑀯𑀼 𑀉𑀧𑀸𑀲𑀓𑁄 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀤𑀻𑀖𑀸𑀯𑀼 𑀉𑀧𑀸𑀲𑀓𑁄 𑀧𑀺𑀢𑀭𑀁 𑀚𑁄𑀢𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤—  ‘𑀤𑀻𑀖𑀸𑀯𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀲𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀳𑀺—  ‘𑀲𑀸𑀥𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀤𑀻𑀖𑀸𑀯𑀼𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀢𑀸𑀢𑀸”𑀢𑀺 𑀔𑁄 𑀚𑁄𑀢𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀤𑀻𑀖𑀸𑀯𑀼𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀚𑁄𑀢𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀻𑀖𑀸𑀯𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀲𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀢𑀺—  ‘𑀲𑀸𑀥𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀤𑀻𑀖𑀸𑀯𑀼𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

1800\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀤𑀻𑀖𑀸𑀯𑀼𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀻𑀖𑀸𑀯𑀼𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺 𑀢𑁂, 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁? 𑀓𑀘𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁄𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁄”𑀢𑀺? “𑀦 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀦 𑀬𑀸𑀧𑀦𑀻𑀬𑀁𑁇 𑀩𑀸𑀍𑀳𑀸 𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁄𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁄”𑀢𑀺𑁇 “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁂, 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺’𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

1801\. “𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀺, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀬𑀺, 𑀅𑀳𑀜𑁆𑀘 𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁆𑀯𑀁, 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀇𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑁂𑀲𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬 𑀙 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀉𑀢𑁆𑀢𑀭𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀇𑀥 𑀢𑁆𑀯𑀁, 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀸𑀳𑀺, 𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀻, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀻 𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀻𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

1802\. “𑀬𑁂𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀙 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑁂𑀲𑀺𑀢𑀸, 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀫𑀬𑀺, 𑀅𑀳𑀜𑁆𑀘 𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀸𑀫𑀺, 𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀻, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀻 𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀻𑁇 𑀅𑀧𑀺 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀸 𑀳𑁂𑀯𑀸𑀬𑀁 𑀚𑁄𑀢𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀫𑀫𑀘𑁆𑀘𑀬𑁂𑀦 𑀯𑀺𑀖𑀸𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀻’”𑀢𑀺𑁇 “𑀫𑀸 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀏𑀯𑀁 𑀫𑀦𑀲𑀸𑀓𑀸𑀲𑀺𑁇 𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀬𑀤𑁂𑀯 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀆𑀳, 𑀢𑀤𑁂𑀯 𑀢𑁆𑀯𑀁 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇

1803\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀻𑀖𑀸𑀯𑀼𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀇𑀫𑀺𑀦𑀸 𑀑𑀯𑀸𑀤𑁂𑀦 𑀑𑀯𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀤𑀻𑀖𑀸𑀯𑀼 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀬𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀻𑀖𑀸𑀯𑀼 𑀦𑀸𑀫 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀑𑀯𑀸𑀤𑁂𑀦 𑀑𑀯𑀤𑀺𑀢𑁄 𑀲𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄”𑀢𑀺? “𑀧𑀡𑁆𑀟𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀸𑀯𑀼 𑀉𑀧𑀸𑀲𑀓𑁄, 𑀧𑀘𑁆𑀘𑀧𑀸𑀤𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁, 𑀦 𑀘 𑀫𑀁 𑀥𑀫𑁆𑀫𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀲𑀺𑁇 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁄 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸”𑀢𑀺𑁇
