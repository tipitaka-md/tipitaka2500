# 𑁧𑁧𑁇𑁧𑁇𑁭 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

(1003.)

1831\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑀁 𑀦𑀸𑀫 𑀓𑁄𑀲𑀮𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀢𑁂 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸—  “𑀲𑀫𑀡𑁄 𑀔𑀮𑀼, 𑀪𑁄, 𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀲𑀤𑁂𑀯𑀓𑀁 𑀲𑀫𑀸𑀭𑀓𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀁 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀧𑀚𑀁 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁, 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1832\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼; 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀲𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀫𑀬𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀏𑀯𑀁𑀓𑀸𑀫𑀸 𑀏𑀯𑀁𑀙𑀦𑁆𑀤𑀸 𑀏𑀯𑀁𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀸—  𑀧𑀼𑀢𑁆𑀢𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀬𑀦𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫, 𑀓𑀸𑀲𑀺𑀓𑀘𑀦𑁆𑀤𑀦𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀫, 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀸𑀫, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀫𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀯𑀁𑀓𑀸𑀫𑀸𑀦𑀁 𑀏𑀯𑀁𑀙𑀦𑁆𑀤𑀸𑀦𑀁 𑀏𑀯𑀁𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀁 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼 𑀬𑀣𑀸 𑀫𑀬𑀁 𑀧𑀼𑀢𑁆𑀢𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀬𑀦𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫…𑀧𑁂…  𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇

1833\. “𑀅𑀢𑁆𑀢𑀼𑀧𑀦𑀸𑀬𑀺𑀓𑀁 𑀯𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1834\. “𑀓𑀢𑀫𑁄 𑀘, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀢𑁆𑀢𑀼𑀧𑀦𑀸𑀬𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄? 𑀇𑀥, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄𑀲𑁆𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀫𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀼𑀔𑀓𑀸𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁄𑁇 𑀬𑁄 𑀔𑁄 𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑀁 𑀅𑀫𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑀁 𑀲𑀼𑀔𑀓𑀸𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂𑀢𑀁 𑀅𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑀁 𑀅𑀫𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑀁 𑀲𑀼𑀔𑀓𑀸𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑀲𑁆𑀲𑀧𑀺 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀧𑀭𑀲𑁆𑀲 𑀧𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀓𑀣𑀸𑀳𑀁 𑀧𑀭𑀁 𑀢𑁂𑀦 𑀲𑀁𑀬𑁄𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀓𑀸𑀬𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀢𑀺𑀓𑁄𑀝𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

1835\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂𑀢𑀁 𑀅𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀭𑀲𑁆𑀲 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑀲𑁆𑀲𑀧𑀺 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀧𑀭𑀲𑁆𑀲 𑀧𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀓𑀣𑀸𑀳𑀁 𑀧𑀭𑀁 𑀢𑁂𑀦 𑀲𑀁𑀬𑁄𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀓𑀸𑀬𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀢𑀺𑀓𑁄𑀝𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

1836\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑁄 𑀔𑁄 𑀫𑁂 𑀤𑀸𑀭𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂𑀢𑀁 𑀅𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀭𑀲𑁆𑀲 𑀤𑀸𑀭𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑀲𑁆𑀲𑀧𑀺 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀧𑀭𑀲𑁆𑀲 𑀧𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀓𑀣𑀸𑀳𑀁 𑀧𑀭𑀁 𑀢𑁂𑀦 𑀲𑀁𑀬𑁄𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀓𑀸𑀬𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀢𑀺𑀓𑁄𑀝𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

1837\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑁄 𑀔𑁄 𑀫𑁂 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀪𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂𑀢𑀁 𑀅𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀭𑀲𑁆𑀲 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀪𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑀲𑁆𑀲𑀧𑀺 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀧𑀭𑀲𑁆𑀲 𑀧𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀓𑀣𑀸𑀳𑀁 𑀧𑀭𑀁 𑀢𑁂𑀦 𑀲𑀁𑀬𑁄𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀯𑀘𑀻𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀢𑀺𑀓𑁄𑀝𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

1838\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  “𑀬𑁄 𑀔𑁄 𑀫𑀁 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀫𑀺𑀢𑁆𑀢𑁂 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂𑀢𑀁 𑀅𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀭𑀁 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀫𑀺𑀢𑁆𑀢𑁂 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑀲𑁆𑀲𑀧𑀺 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁…𑀧𑁂…  𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀯𑀘𑀻𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀢𑀺𑀓𑁄𑀝𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

1839\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  “𑀬𑁄 𑀔𑁄 𑀫𑀁 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂𑀢𑀁 𑀅𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀭𑀁 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑀲𑁆𑀲𑀧𑀺 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄…𑀧𑁂…  𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀯𑀘𑀻𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀢𑀺𑀓𑁄𑀝𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

1840\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑁄 𑀔𑁄 𑀫𑀁 𑀲𑀫𑁆𑀨𑀪𑀸𑀲𑁂𑀦 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀪𑀸𑀲𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁂𑀢𑀁 𑀅𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀭𑀁 𑀲𑀫𑁆𑀨𑀪𑀸𑀲𑁂𑀦 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀪𑀸𑀲𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑀲𑁆𑀲𑀧𑀺 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀧𑀭𑀲𑁆𑀲 𑀧𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄𑁇 𑀬𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀓𑀣𑀸𑀳𑀁 𑀧𑀭𑀁 𑀢𑁂𑀦 𑀲𑀁𑀬𑁄𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀯𑀘𑀻𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀢𑀺𑀓𑁄𑀝𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

1841\. 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀢𑀺; 𑀥𑀫𑁆𑀫𑁂 …𑀧𑁂…  𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀫𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀳𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀆𑀓𑀗𑁆𑀔𑀺𑀬𑁂𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’”𑀢𑀺𑁇

1842\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀯𑁂𑀍𑀉𑀤𑁆𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀏𑀢𑁂 𑀫𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑁂 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑁂 𑀲𑀭𑀡𑀁 𑀕𑀢𑁂”𑀢𑀺𑁇
