# 𑁪𑁇𑁨𑁇𑁧 𑀧𑀝𑀺𑀮𑀸𑀪𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀫𑀼𑀤𑀼𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀧𑀝𑀺𑀮𑀸𑀪𑀲𑀼𑀢𑁆𑀢

(481.)

1033\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂… 𑁇 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀁—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1034\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑁂 𑀆𑀭𑀩𑁆𑀪 𑀯𑀻𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1035\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀆𑀭𑀩𑁆𑀪 𑀲𑀢𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1036\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀮𑀪𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁, 𑀮𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1037\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀤𑀬𑀢𑁆𑀣𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀻”𑀢𑀺𑁇
