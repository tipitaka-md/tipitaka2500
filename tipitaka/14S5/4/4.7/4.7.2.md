# 𑁪𑁇𑁭𑁇𑁨 𑀅𑀦𑀼𑀲𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀼𑀲𑀬𑀲𑀼𑀢𑁆𑀢

(532.)

1231\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀼𑀲𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀼𑀲𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
