# 𑁪𑁇𑁩𑁇𑁭 𑀅𑀭𑀳𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀙𑀍𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀭𑀳𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(497.)

1073. “𑀙𑀬𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀙? 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑁂𑀲𑀁 𑀙𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀯𑀼𑀲𑀺𑀢𑀯𑀸 𑀓𑀢𑀓𑀭𑀡𑀻𑀬𑁄 𑀑𑀳𑀺𑀢𑀪𑀸𑀭𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑀤𑀢𑁆𑀣𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀪𑀯𑀲𑀁𑀬𑁄𑀚𑀦𑁄 𑀲𑀫𑁆𑀫𑀤𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄’”𑀢𑀺𑁇
