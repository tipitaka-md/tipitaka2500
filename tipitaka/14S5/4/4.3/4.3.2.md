# 𑁪𑁇𑁩𑁇𑁨 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀙𑀍𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(492.)

1063\. “𑀢𑀻𑀡𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀡𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀻”𑀢𑀺𑁇
