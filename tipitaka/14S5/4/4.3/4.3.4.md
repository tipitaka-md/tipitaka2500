# 𑁪𑁇𑁩𑁇𑁪 𑀏𑀓𑀩𑀻𑀚𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀙𑀍𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀩𑀻𑀚𑀻𑀲𑀼𑀢𑁆𑀢

(494.)

1067. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀲𑀫𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀧𑀽𑀭𑀢𑁆𑀢𑀸 𑀅𑀭𑀳𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀉𑀧𑀳𑀘𑁆𑀘𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀉𑀤𑁆𑀥𑀁𑀲𑁄𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑀕𑀸𑀫𑀻, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀏𑀓𑀩𑀻𑀚𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀧𑀭𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀤𑀼𑀢𑀭𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀭𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
