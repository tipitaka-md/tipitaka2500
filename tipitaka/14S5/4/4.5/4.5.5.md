# 𑁪𑁇𑁫𑁇𑁫 𑀧𑀞𑀫𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑀲𑀼𑀢𑁆𑀢

(515.)

1162\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑁂 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀢𑀺𑀦𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺?

1163\. “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  “𑀏𑀓𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻𑀢𑀺𑁇 𑀓𑀢𑀫𑀲𑁆𑀲 𑀏𑀓𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀯𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀢𑀤𑀦𑁆𑀯𑀬𑀸 𑀲𑀤𑁆𑀥𑀸 𑀲𑀡𑁆𑀞𑀸𑀢𑀺, 𑀢𑀤𑀦𑁆𑀯𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀲𑀡𑁆𑀞𑀸𑀢𑀺, 𑀢𑀤𑀦𑁆𑀯𑀬𑀸 𑀲𑀢𑀺 𑀲𑀡𑁆𑀞𑀸𑀢𑀺, 𑀢𑀤𑀦𑁆𑀯𑀬𑁄 𑀲𑀫𑀸𑀥𑀺 𑀲𑀡𑁆𑀞𑀸𑀢𑀺𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀲𑁆𑀲 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇
