# 𑁪𑁇𑁫𑁇𑁩 𑀲𑀸𑀓𑁂𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀓𑁂𑀢𑀲𑀼𑀢𑁆𑀢

(513.)

1151\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀓𑁂𑀢𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀚𑀦𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀻”𑀢𑀺?

1152\. “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀪𑀕𑀯𑀁𑀦𑁂𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀁𑀧𑀝𑀺𑀲𑀭𑀡𑀸𑁇 𑀲𑀸𑀥𑀼 𑀯𑀢, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀏𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…

1153\. “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

1154\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁, 𑀬𑀁 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁, 𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀲𑀢𑀺𑀩𑀮𑀁, 𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀢𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁, 𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁, 𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀢𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸, 𑀢𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂 𑀤𑀻𑀧𑁄𑁇 𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀢𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀬𑀸 𑀏𑀓𑁄 𑀲𑁄𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀢𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀲𑁄𑀢𑀸𑀦𑀺𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

1155\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀢𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀬𑀸 𑀏𑀓𑁄 𑀲𑁄𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺? 𑀬𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀤𑀻𑀧𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀫𑀦𑁆𑀢𑁂 𑀉𑀤𑀓𑀁, 𑀬𑀜𑁆𑀘 𑀧𑀘𑁆𑀙𑀺𑀫𑀦𑁆𑀢𑁂 𑀉𑀤𑀓𑀁—  𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀢𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀬𑀸 𑀏𑀓𑁄 𑀲𑁄𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

1156\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀢𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀲𑁄𑀢𑀸𑀦𑀺𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺? 𑀬𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀤𑀻𑀧𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀦𑁆𑀢𑁂 𑀉𑀤𑀓𑀁, 𑀬𑀜𑁆𑀘 𑀤𑀓𑁆𑀔𑀺𑀡𑀦𑁆𑀢𑁂 𑀉𑀤𑀓𑀁—  𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀢𑀲𑁆𑀲𑀸 𑀦𑀤𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀲𑁄𑀢𑀸𑀦𑀺𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀁 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁, 𑀬𑀁 𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁, 𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀲𑀢𑀺𑀩𑀮𑀁, 𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀢𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁, 𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁; 𑀬𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁, 𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀢𑀁 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇
