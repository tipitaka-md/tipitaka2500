# 𑁪𑁇𑁫𑁇𑁧 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

* 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(511.)

1137\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑁂 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀘𑁆𑀙𑀸𑀢𑀧𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀺𑀝𑁆𑀞𑀺𑀁 𑀑𑀢𑀸𑀧𑀬𑀫𑀸𑀦𑁄𑁇

1138\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀸𑀡𑀺𑀦𑀸 𑀅𑀦𑁄𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀦 𑀘𑁂𑀯𑀁 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀢𑀸𑀯 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄, 𑀲𑀺𑀣𑀺𑀮𑀸𑀦𑀺 𑀘 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀯𑀮𑀺𑀬𑀚𑀸𑀢𑀸𑀦𑀺, 𑀧𑀼𑀭𑀢𑁄 𑀧𑀩𑁆𑀪𑀸𑀭𑁄 𑀘 𑀓𑀸𑀬𑁄, 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁—  𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1139\. “𑀏𑀯𑀜𑁆𑀳𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀳𑁄𑀢𑀺—  𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑁄 𑀬𑁄𑀩𑁆𑀩𑀜𑁆𑀜𑁂, 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑁄 𑀆𑀭𑁄𑀕𑁆𑀬𑁂, 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑁄 𑀚𑀻𑀯𑀺𑀢𑁂𑁇 𑀦 𑀘𑁂𑀯 𑀢𑀸𑀯 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄, 𑀲𑀺𑀣𑀺𑀮𑀸𑀦𑀺 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀯𑀮𑀺𑀬𑀚𑀸𑀢𑀸𑀦𑀺, 𑀧𑀼𑀭𑀢𑁄 𑀧𑀩𑁆𑀪𑀸𑀭𑁄 𑀘 𑀓𑀸𑀬𑁄, 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁—  𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1140\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀘 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

1141\. _“𑀥𑀻 𑀢𑀁 𑀚𑀫𑁆𑀫𑀺 𑀚𑀭𑁂 𑀅𑀢𑁆𑀣𑀼,_  
_𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀓𑀭𑀡𑀻 𑀚𑀭𑁂;_  
_𑀢𑀸𑀯 𑀫𑀦𑁄𑀭𑀫𑀁 𑀩𑀺𑀫𑁆𑀩𑀁,_  
_𑀚𑀭𑀸𑀬 𑀅𑀪𑀺𑀫𑀤𑁆𑀤𑀺𑀢𑀁𑁇_  


1142\. _𑀬𑁄𑀧𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀚𑀻𑀯𑁂,_  
_𑀲𑁄𑀧𑀺 𑀫𑀘𑁆𑀘𑀼𑀧𑀭𑀸𑀬𑀡𑁄;_  
_𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺,_  
_𑀲𑀩𑁆𑀩𑀫𑁂𑀯𑀸𑀪𑀺𑀫𑀤𑁆𑀤𑀢𑀻”𑀢𑀺𑁇_
