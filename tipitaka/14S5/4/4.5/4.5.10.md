# 𑁪𑁇𑁫𑁇𑁧𑁦 𑀆𑀧𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

### 𑀆𑀧𑀡𑀲𑀼𑀢𑁆𑀢

(520.)

1178. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀅𑀗𑁆𑀕𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀧𑀡𑀁 𑀦𑀸𑀫 𑀅𑀗𑁆𑀕𑀸𑀦𑀁 𑀦𑀺𑀕𑀫𑁄𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑁄 𑀲𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀢𑀣𑀸𑀕𑀢𑁂 𑀏𑀓𑀦𑁆𑀢𑀕𑀢𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄, 𑀦 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀯𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀯𑀸”𑀢𑀺?

1179. “𑀬𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀢𑀣𑀸𑀕𑀢𑁂 𑀏𑀓𑀦𑁆𑀢𑀕𑀢𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄, 𑀦 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀯𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀯𑀸𑁇 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀺—  𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀬𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀻𑀭𑀺𑀬𑀁 𑀢𑀤𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1180. 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀲𑀢𑀺𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀭𑀫𑁂𑀦 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀘𑀺𑀭𑀓𑀢𑀫𑁆𑀧𑀺 𑀘𑀺𑀭𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑀺 𑀲𑀭𑀺𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀢𑀺 𑀢𑀤𑀲𑁆𑀲 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1181. 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀦𑁄 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀮𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁, 𑀮𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀁𑁇 𑀬𑁄 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀸𑀥𑀺 𑀢𑀤𑀲𑁆𑀲 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1182. 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀦𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺—  𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄 𑀔𑁄 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯 𑀢𑀫𑁄𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀦𑁆𑀢𑀫𑁂𑀢𑀁 𑀧𑀤𑀁 𑀧𑀡𑀻𑀢𑀫𑁂𑀢𑀁 𑀧𑀤𑀁, 𑀬𑀤𑀺𑀤𑀁—  𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀣𑁄 𑀲𑀩𑁆𑀩𑀽𑀧𑀥𑀺𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁄 𑀯𑀺𑀭𑀸𑀕𑁄 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀜𑀸 𑀢𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1183. 𑀲𑀤𑁆𑀥𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀲𑀫𑀸𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀅𑀪𑀺𑀲𑀤𑁆𑀤𑀳𑀢𑀺—  ‘𑀇𑀫𑁂 𑀔𑁄 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀬𑁂 𑀫𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀢𑀯𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀢𑁂𑀦𑀸𑀳𑀁 𑀏𑀢𑀭𑀳𑀺 𑀓𑀸𑀬𑁂𑀦 𑀘 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀸𑀫𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀅𑀢𑀺𑀯𑀺𑀚𑁆𑀛 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀤𑁆𑀥𑀸 𑀢𑀤𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇

1184. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀬𑁄 𑀲𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀢𑀣𑀸𑀕𑀢𑁂 𑀏𑀓𑀦𑁆𑀢𑀕𑀢𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄, 𑀦 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀯𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀯𑀸𑁇 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀺—  𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀬𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀯𑀻𑀭𑀺𑀬𑀁 𑀢𑀤𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1185. 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀲𑀢𑀺𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀭𑀫𑁂𑀦 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀘𑀺𑀭𑀓𑀢𑀫𑁆𑀧𑀺 𑀘𑀺𑀭𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑀺 𑀲𑀭𑀺𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀢𑀺 𑀢𑀤𑀲𑁆𑀲 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1186. 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀦𑁄 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀮𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁, 𑀮𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀁𑁇 𑀬𑁄 𑀳𑀺𑀲𑁆𑀲, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀫𑀸𑀥𑀺 𑀢𑀤𑀲𑁆𑀲 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1187. 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀦𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀬𑀁 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺—  𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄 𑀔𑁄 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯 𑀢𑀫𑁄𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀦𑁆𑀢𑀫𑁂𑀢𑀁 𑀧𑀤𑀁 𑀧𑀡𑀻𑀢𑀫𑁂𑀢𑀁 𑀧𑀤𑀁, 𑀬𑀤𑀺𑀤𑀁—  𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀣𑁄 𑀲𑀩𑁆𑀩𑀽𑀧𑀥𑀺𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁄 𑀯𑀺𑀭𑀸𑀕𑁄 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀜𑁆𑀜𑀸 𑀢𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1188. 𑀲𑀤𑁆𑀥𑁄 𑀲𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀲𑀫𑀸𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀅𑀪𑀺𑀲𑀤𑁆𑀤𑀳𑀢𑀺—  ‘𑀇𑀫𑁂 𑀔𑁄 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀬𑁂 𑀫𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀢𑀯𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀢𑁂𑀦𑀸𑀳𑀁 𑀏𑀢𑀭𑀳𑀺 𑀓𑀸𑀬𑁂𑀦 𑀘 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀸𑀫𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀅𑀢𑀺𑀯𑀺𑀚𑁆𑀛 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀤𑁆𑀥𑀸 𑀢𑀤𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇
