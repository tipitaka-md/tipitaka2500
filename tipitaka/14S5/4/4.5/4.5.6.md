# 𑁪𑁇𑁫𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑀲𑀼𑀢𑁆𑀢

(516.)

1165\. 𑀢𑀁𑀬𑁂𑀯 𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀓𑀢𑀺𑀦𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺?

1166\. “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  “𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁? 𑀅𑀭𑀺𑀬𑀸𑀬 𑀘 𑀧𑀜𑁆𑀜𑀸𑀬, 𑀅𑀭𑀺𑀬𑀸𑀬 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀢𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀢𑀤𑀲𑁆𑀲 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇
