# 𑁪𑁇𑁪𑁇𑁧𑁦 𑀉𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀝𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀉𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀝𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(510.)

1128\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁𑁇 𑀢𑀜𑁆𑀘 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀺𑀤𑀸𑀦𑀁 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀓𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺’𑁇

1129\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁𑁇 𑀢𑀜𑁆𑀘 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀺𑀤𑀸𑀦𑀁 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀓𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺’𑁇

1130\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁𑁇 𑀢𑀜𑁆𑀘 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀺𑀤𑀸𑀦𑀁 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀓𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺’𑁇

1131\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁𑁇 𑀢𑀜𑁆𑀘 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀺𑀤𑀸𑀦𑀁 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀓𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺’𑁇

1132\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁𑁇 𑀢𑀜𑁆𑀘 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀺𑀤𑀸𑀦𑀁 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀓𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀻’”𑀢𑀺𑁇
