# 𑁪𑁇𑁪𑁇𑁫 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(505.)

1094\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺; 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺; 𑀦 𑀫𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀡𑁂𑀲𑀼 𑀯𑀸 𑀲𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸, 𑀦 𑀘 𑀧𑀦𑁂𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀁 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀁 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇

1095\. 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺; 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘 𑀔𑁄 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀡𑁂𑀲𑀼 𑀘𑁂𑀯 𑀲𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸, 𑀢𑁂 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀜𑁆𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
