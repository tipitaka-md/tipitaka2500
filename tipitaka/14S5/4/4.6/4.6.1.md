# 𑁪𑁇𑁬𑁇𑁧 𑀲𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀽𑀓𑀭𑀔𑀢𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(521.)

1193. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀸𑀮𑀸𑀬 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸 𑀧𑀸𑀡𑀸, 𑀲𑀻𑀳𑁄 𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀣𑀸𑀫𑁂𑀦 𑀚𑀯𑁂𑀦 𑀲𑀽𑀭𑁂𑀦; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀩𑁄𑀥𑀸𑀬𑁇

1194. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸 𑀧𑀸𑀡𑀸, 𑀲𑀻𑀳𑁄 𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀣𑀸𑀫𑁂𑀦 𑀚𑀯𑁂𑀦 𑀲𑀽𑀭𑁂𑀦; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀩𑁄𑀥𑀸𑀬𑀸”𑀢𑀺𑁇
