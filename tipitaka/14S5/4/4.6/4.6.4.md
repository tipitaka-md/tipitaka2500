# 𑁪𑁇𑁬𑁇𑁪 𑀧𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀽𑀓𑀭𑀔𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀤𑀲𑀼𑀢𑁆𑀢

(524.)

1208\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀚𑀗𑁆𑀕𑀮𑀸𑀦𑀁 𑀧𑀸𑀡𑀸𑀦𑀁 𑀧𑀤𑀚𑀸𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑁂 𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑀁 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀫𑀳𑀦𑁆𑀢𑀢𑁆𑀢𑁂𑀦; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀧𑀤𑀸𑀦𑀺 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀤𑀁 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀩𑁄𑀥𑀸𑀬𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀤𑀸𑀦𑀺 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀤𑀁, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀤𑀁, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀤𑀁, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀤𑀁, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺; 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀤𑀁, 𑀢𑀁 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀚𑀗𑁆𑀕𑀮𑀸𑀦𑀁 𑀧𑀸𑀡𑀸𑀦𑀁 𑀧𑀤𑀚𑀸𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑁂 𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀧𑀤𑀁 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀫𑀳𑀦𑁆𑀢𑀢𑁆𑀢𑁂𑀦; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀧𑀤𑀸𑀦𑀺 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀤𑀁 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀩𑁄𑀥𑀸𑀬𑀸”𑀢𑀺𑁇
