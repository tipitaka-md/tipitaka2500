# 𑁪𑁇𑁬𑁇𑁭 𑀲𑀳𑀫𑁆𑀧𑀢𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀽𑀓𑀭𑀔𑀢𑀯𑀕𑁆𑀕

### 𑀲𑀳𑀫𑁆𑀧𑀢𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(527.)

1214\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦𑀚𑁆𑀚𑀸 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀸𑀬 𑀢𑀻𑀭𑁂 𑀅𑀚𑀧𑀸𑀮𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁂 𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀦𑀺 𑀅𑀫𑀢𑁄𑀕𑀥𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀸𑀦𑀺 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀅𑀫𑀢𑁄𑀕𑀥𑀁 𑀳𑁄𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀁 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁𑁇 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀅𑀫𑀢𑁄𑀕𑀥𑀁 𑀳𑁄𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀁 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀦𑀺 𑀅𑀫𑀢𑁄𑀕𑀥𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀸𑀦𑀺 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑀦𑀻”𑀢𑀺𑁇

1215\. 𑀅𑀣 𑀔𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀪𑀕𑀯𑀢𑁄 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀕𑀯𑀸, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀲𑀼𑀕𑀢𑁇 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀦𑀺 𑀅𑀫𑀢𑁄𑀕𑀥𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀸𑀦𑀺 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀅𑀫𑀢𑁄𑀕𑀥𑀁 𑀳𑁄𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀁 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀅𑀫𑀢𑁄𑀕𑀥𑀁 𑀳𑁄𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀁 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀦𑀺 𑀅𑀫𑀢𑁄𑀕𑀥𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀸𑀦𑀺 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑀦𑀺𑁇

1216\. 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀘𑀭𑀺𑀁𑁇 𑀢𑀢𑁆𑀭𑀧𑀺 𑀫𑀁 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀲𑀳𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀳𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 𑀯𑀺𑀭𑀸𑀚𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄𑁇 𑀢𑀢𑁆𑀭𑀧𑀺 𑀫𑀁 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀕𑀯𑀸, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀲𑀼𑀕𑀢𑁇 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺, 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀣𑀸 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀦𑀺 𑀅𑀫𑀢𑁄𑀕𑀥𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀫𑀢𑀧𑀭𑀸𑀬𑀡𑀸𑀦𑀺 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑀦𑀻”𑀢𑀺𑁇
