# 𑁪𑁇𑁬𑁇𑁩 𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀽𑀓𑀭𑀔𑀢𑀯𑀕𑁆𑀕

* 𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

(523.)

1200\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀅𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸”𑀢𑀺?

1201\. “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀅𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇

1202\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1203\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀜𑁆𑀜𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀬𑁄 𑀏𑀯𑀁 𑀪𑀽𑀢𑀁 𑀢𑀘𑁆𑀙𑀁 𑀢𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀬𑀣𑀸 𑀪𑀕𑀯𑀸’𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀦𑀢𑁆𑀣𑀺 𑀔𑁄 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀜𑁆𑀜𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀬𑁄 𑀏𑀯𑀁 𑀪𑀽𑀢𑀁 𑀢𑀘𑁆𑀙𑀁 𑀢𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀬𑀣𑀸 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1204\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁𑀕𑀢𑀺𑀓𑀸𑀦𑀺 𑀬𑀁𑀧𑀭𑀫𑀸𑀦𑀺 𑀬𑀁𑀨𑀮𑀸𑀦𑀺 𑀬𑀁𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑀦𑀺𑁇 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀅𑀢𑀺𑀯𑀺𑀚𑁆𑀛 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1205\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀅𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀬𑀁𑀕𑀢𑀺𑀓𑀸𑀦𑀺 𑀬𑀁𑀧𑀭𑀫𑀸𑀦𑀺 𑀬𑀁𑀨𑀮𑀸𑀦𑀺 𑀬𑀁𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀸𑀬𑁂𑀦 𑀘 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀅𑀢𑀺𑀯𑀺𑀚𑁆𑀛 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀅𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1206\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀦 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺 𑀓𑀺𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁂𑀔𑀪𑀽𑀫𑀺𑀬𑀁 𑀞𑀺𑀢𑁄 ‘𑀅𑀲𑁂𑀔𑁄𑀲𑁆𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
