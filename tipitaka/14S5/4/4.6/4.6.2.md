# 𑁪𑁇𑁬𑁇𑁨 𑀫𑀮𑁆𑀮𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀽𑀓𑀭𑀔𑀢𑀯𑀕𑁆𑀕

* 𑀫𑀮𑁆𑀮𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(522.)

1196\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀧𑁆𑀧𑀁 𑀦𑀸𑀫 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑀫𑁄𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺𑁇

1197\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀓𑀽𑀝𑀁 𑀦 𑀉𑀲𑁆𑀲𑀺𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀕𑁄𑀧𑀸𑀦𑀲𑀻𑀦𑀁 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀕𑁄𑀧𑀸𑀦𑀲𑀻𑀦𑀁 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀓𑀽𑀝𑀁 𑀉𑀲𑁆𑀲𑀺𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀕𑁄𑀧𑀸𑀦𑀲𑀻𑀦𑀁 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀕𑁄𑀧𑀸𑀦𑀲𑀻𑀦𑀁 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀲𑀡𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀺 𑀳𑁄𑀢𑀺𑁇

1198\. 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀜𑁆𑀜𑀯𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀢𑀤𑀦𑁆𑀯𑀬𑀸 𑀲𑀤𑁆𑀥𑀸 𑀲𑀡𑁆𑀞𑀸𑀢𑀺, 𑀢𑀤𑀦𑁆𑀯𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀲𑀡𑁆𑀞𑀸𑀢𑀺, 𑀢𑀤𑀦𑁆𑀯𑀬𑀸 𑀲𑀢𑀺 𑀲𑀡𑁆𑀞𑀸𑀢𑀺, 𑀢𑀤𑀦𑁆𑀯𑀬𑁄 𑀲𑀫𑀸𑀥𑀺 𑀲𑀡𑁆𑀞𑀸𑀢𑀻”𑀢𑀺𑁇
