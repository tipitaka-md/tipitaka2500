# 𑁪𑁇𑁬𑁇𑁮 𑀲𑀽𑀓𑀭𑀔𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀽𑀓𑀭𑀔𑀢𑀯𑀕𑁆𑀕

### 𑀲𑀽𑀓𑀭𑀔𑀢𑀲𑀼𑀢𑁆𑀢

(528.)

1218. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀲𑀽𑀓𑀭𑀔𑀢𑀸𑀬𑀁𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺? “𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀜𑁆𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀜𑁆𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀺𑁇

1219. 𑀓𑀢𑀫𑁄 𑀘, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑁄 𑀬𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺? “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀉𑀧𑀲𑀫𑀕𑀸𑀫𑀺𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀺𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀉𑀧𑀲𑀫𑀕𑀸𑀫𑀺𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀺𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑁄 𑀬𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀏𑀲𑁄 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑁄 𑀬𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻𑀢𑀺𑁇

1220. 𑀓𑀢𑀫𑁄 𑀘, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑁄 𑀬𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺? “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀭𑀺 𑀲𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂 𑀲𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀲𑀁𑀖𑁂 𑀲𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀲𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑁄 𑀬𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀏𑀲𑁄 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑁄 𑀬𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀲𑀦𑁂 𑀯𑀸 𑀧𑀭𑀫𑀦𑀺𑀧𑀘𑁆𑀘𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
