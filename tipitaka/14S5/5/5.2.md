# 𑁫𑁇𑁨 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕

1291\. (𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀯𑀲𑁂𑀦 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇)

---

1292\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1293\. _𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀤𑀁 𑀓𑀽𑀝𑀁,_  
_𑀫𑀽𑀮𑀁 𑀲𑀸𑀭𑁂𑀦 𑀯𑀲𑁆𑀲𑀺𑀓𑀁;_  
_𑀭𑀸𑀚𑀸 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑀸,_  
_𑀯𑀢𑁆𑀣𑁂𑀦 𑀤𑀲𑀫𑀁 𑀧𑀤𑀦𑁆𑀢𑀺𑁇_
