# 𑁧𑁇𑁯𑁇𑁨--𑁫 𑀤𑀼𑀢𑀺𑀬𑀸𑀤𑀺𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀕𑀗𑁆𑀕𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀸𑀤𑀺𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢𑀘𑀢𑀼𑀓𑁆𑀓

(92--95.)

202\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀫𑀼𑀦𑀸 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑀺𑀭𑀯𑀢𑀻 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀭𑀪𑀽 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀻 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂… 𑁇
