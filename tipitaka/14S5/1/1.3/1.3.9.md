# 𑁧𑁇𑁩𑁇𑁯 𑀯𑁂𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(29.)

110\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑁇 𑀇𑀫𑀸𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁄 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄? 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀇𑀫𑀸𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
