# 𑁧𑁇𑁩𑁇𑁩 𑀧𑀞𑀫𑀧𑀝𑀺𑀧𑀤𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀝𑀺𑀧𑀤𑀸𑀲𑀼𑀢𑁆𑀢

(23.)

92\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀤𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀤𑀜𑁆𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀤𑀸? 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀤𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀤𑀸? 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺𑁇
