# 𑁧𑁇𑁧𑁩𑁇𑁩 𑀆𑀲𑀯𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀏𑀲𑀦𑀸𑀯𑀕𑁆𑀕

* 𑀆𑀲𑀯𑀲𑀼𑀢𑁆𑀢

(163.)

314\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀓𑀸𑀫𑀸𑀲𑀯𑁄, 𑀪𑀯𑀸𑀲𑀯𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀆𑀲𑀯𑀸𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬…𑀧𑁂…  𑀅𑀬𑀁 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
