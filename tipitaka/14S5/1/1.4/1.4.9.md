# 𑁧𑁇𑁪𑁇𑁯 𑀧𑀞𑀫𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(39.)

139\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀨𑀮𑀸𑀦𑀺? 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁, 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀨𑀮𑀸𑀦𑀻”𑀢𑀺𑁇
