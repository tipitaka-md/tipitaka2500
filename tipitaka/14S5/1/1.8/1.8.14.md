# 𑁧𑁇𑁮𑁇𑁧𑁪 𑀤𑀼𑀢𑀺𑀬𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀤𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀤𑀸𑀲𑀼𑀢𑁆𑀢

(90.)

195\. “𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀤𑀸𑁇 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑁄𑀲𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀫𑁄𑀳𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑁄𑀲𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀫𑁄𑀳𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀻”𑀢𑀺𑁇

198\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



199\. _𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀻𑀮𑀜𑁆𑀘,_  
_𑀙𑀦𑁆𑀤𑁄 𑀘 𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀤𑀺𑀝𑁆𑀞𑀺 𑀘 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄 𑀘,_  
_𑀬𑁄𑀦𑀺𑀲𑁄 𑀪𑀯𑀢𑀺 𑀲𑀢𑁆𑀢𑀫𑀁𑁇_
