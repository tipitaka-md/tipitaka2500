# 𑁧𑁇𑁧𑁨𑁇𑁨 𑀩𑀻𑀚𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀩𑀮𑀓𑀭𑀡𑀻𑀬𑀯𑀕𑁆𑀕

### 𑀩𑀻𑀚𑀲𑀼𑀢𑁆𑀢

(150.)

278\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀓𑁂𑀘𑀺𑀫𑁂 𑀩𑀻𑀚𑀕𑀸𑀫𑀪𑀽𑀢𑀕𑀸𑀫𑀸 𑀯𑀼𑀟𑁆𑀠𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀧𑀣𑀯𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀣𑀯𑀺𑀬𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬 𑀏𑀯𑀫𑁂𑀢𑁂 𑀩𑀻𑀚𑀕𑀸𑀫𑀪𑀽𑀢𑀕𑀸𑀫𑀸 𑀯𑀼𑀟𑁆𑀠𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀯𑀼𑀟𑁆𑀠𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀯𑀼𑀟𑁆𑀠𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀲𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀁…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀯𑀼𑀟𑁆𑀠𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀲𑀽”𑀢𑀺𑁇
