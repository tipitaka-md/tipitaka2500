# 𑁮𑁇𑁧𑁇𑁮 𑀲𑀮𑀍𑀆𑀕𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

### 𑀲𑀮𑀍𑀆𑀕𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(906.)

1578\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀮𑀍𑀆𑀕𑀸𑀭𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺…𑀧𑁂…  𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀗𑁆𑀕𑀸 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸𑁇 𑀅𑀣 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀓𑀼𑀤𑀸𑀮𑀧𑀺𑀝𑀓𑀁 𑀆𑀤𑀸𑀬—  ‘𑀫𑀬𑀁 𑀇𑀫𑀁 𑀕𑀗𑁆𑀕𑀸𑀦𑀤𑀺𑀁 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀦𑁆𑀦𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫 𑀧𑀘𑁆𑀙𑀸𑀧𑁄𑀡𑀁 𑀧𑀘𑁆𑀙𑀸𑀧𑀩𑁆𑀪𑀸𑀭𑀦𑁆’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣𑀸𑀯𑀼𑀲𑁄, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀕𑀗𑁆𑀕𑀸𑀦𑀤𑀺𑀁 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀦𑁆𑀦𑀁 𑀓𑀭𑁂𑀬𑁆𑀬 𑀧𑀘𑁆𑀙𑀸𑀧𑁄𑀡𑀁 𑀧𑀘𑁆𑀙𑀸𑀧𑀩𑁆𑀪𑀸𑀭𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄”𑁇 “𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼”? “𑀕𑀗𑁆𑀕𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀤𑀻 𑀧𑀸𑀘𑀻𑀦𑀦𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑁄𑀡𑀸 𑀧𑀸𑀘𑀻𑀦𑀧𑀩𑁆𑀪𑀸𑀭𑀸𑁇 𑀲𑀸 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀦𑁆𑀦𑀁 𑀓𑀸𑀢𑀼𑀁 𑀧𑀘𑁆𑀙𑀸𑀧𑁄𑀡𑀁 𑀧𑀘𑁆𑀙𑀸𑀧𑀩𑁆𑀪𑀸𑀭𑀁𑁇 𑀬𑀸𑀯𑀤𑁂𑀯 𑀘 𑀧𑀦 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀣𑀲𑁆𑀲 𑀯𑀺𑀖𑀸𑀢𑀲𑁆𑀲 𑀪𑀸𑀕𑀻 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1579\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀯𑀸 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀯𑀸 𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀻 𑀯𑀸 𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸 𑀪𑁄𑀕𑁂𑀳𑀺 𑀅𑀪𑀺𑀳𑀝𑁆𑀞𑀼𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀏𑀳𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀓𑀺𑀁 𑀢𑁂 𑀇𑀫𑁂 𑀓𑀸𑀲𑀸𑀯𑀸 𑀅𑀦𑀼𑀤𑀳𑀦𑁆𑀢𑀺? 𑀓𑀺𑀁 𑀫𑀼𑀡𑁆𑀟𑁄 𑀓𑀧𑀸𑀮𑀫𑀦𑀼𑀲𑀜𑁆𑀘𑀭𑀲𑀺? 𑀏𑀳𑀺 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀳𑀻’𑀢𑀺𑁇

1580\. 𑀲𑁄 𑀯𑀢, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀦𑁆𑀦𑀁 𑀯𑀺𑀯𑁂𑀓𑀧𑁄𑀡𑀁 𑀯𑀺𑀯𑁂𑀓𑀧𑀩𑁆𑀪𑀸𑀭𑀁 𑀢𑀁 𑀯𑀢 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺—  𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀻𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀸𑀲𑀼…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀻”𑀢𑀺𑁇
