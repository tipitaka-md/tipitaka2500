# 𑁮𑁇𑁧𑁇𑁩 𑀲𑀼𑀢𑀦𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀢𑀦𑀼𑀲𑀼𑀢𑁆𑀢

(901.)

1565\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀢𑀦𑀼𑀢𑀻𑀭𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀢𑀫𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄”𑀢𑀺?

1566\. “𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁; 𑀯𑁂𑀤𑀦𑀸𑀲𑀼…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  𑀇𑀫𑁂𑀲𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄𑁇 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀧𑀦𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀳𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀳𑀻𑀦𑀢𑁄 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀢𑁄 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, 𑀧𑀡𑀻𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀡𑀻𑀢𑀢𑁄 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀦𑁆”𑀢𑀺𑁇
