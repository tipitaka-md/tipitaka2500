# 𑁮𑁇𑁧𑁇𑁧 𑀧𑀞𑀫𑀭𑀳𑁄𑀕𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀭𑀳𑁄𑀕𑀢𑀲𑀼𑀢𑁆𑀢

(899.)

1546\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑁂𑀲𑀁 𑀓𑁂𑀲𑀜𑁆𑀘𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑀺𑀭𑀤𑁆𑀥𑀸, 𑀯𑀺𑀭𑀤𑁆𑀥𑁄 𑀢𑁂𑀲𑀁 𑀅𑀭𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀻𑁇 𑀬𑁂𑀲𑀁 𑀓𑁂𑀲𑀜𑁆𑀘𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀆𑀭𑀤𑁆𑀥𑀸, 𑀆𑀭𑀤𑁆𑀥𑁄 𑀢𑁂𑀲𑀁 𑀅𑀭𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀻”𑀢𑀺𑁇

1547\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀆𑀭𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀻”𑀢𑀺?

1548\. “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑁂 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑁂 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑁂 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀬𑁂 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀬𑁂 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀬𑁂 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀬𑁂 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀬𑁂 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀬𑁂 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇

1549\. 𑀲𑁄 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀁 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄’𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑁇

1550\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇

1551\. 𑀲𑁄 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀽𑀮𑀜𑁆𑀘 𑀢𑀤𑀼𑀪𑀬𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀁 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄’𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑁇

1552\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻…𑀧𑁂…  𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇

1553\. 𑀲𑁄 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑁇

1554\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼…𑀧𑁂…  𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼…𑀧𑁂…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇

1555\. 𑀲𑁄 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺—  ‘𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀆𑀭𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
