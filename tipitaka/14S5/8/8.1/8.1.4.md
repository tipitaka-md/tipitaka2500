# 𑁮𑁇𑁧𑁇𑁪 𑀧𑀞𑀫𑀓𑀡𑁆𑀟𑀓𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀓𑀡𑁆𑀟𑀓𑀻𑀲𑀼𑀢𑁆𑀢

(902.)

1568\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀲𑀸𑀓𑁂𑀢𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀓𑀡𑁆𑀟𑀓𑀻𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑁂𑀔𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀢𑀩𑁆𑀩𑀸”𑀢𑀺?

1569\. “𑀲𑁂𑀔𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁; 𑀯𑁂𑀤𑀦𑀸𑀲𑀼…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  𑀲𑁂𑀔𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
