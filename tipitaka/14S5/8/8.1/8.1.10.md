# 𑁮𑁇𑁧𑁇𑁧𑁦 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀭𑀳𑁄𑀕𑀢𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(908.)

1586\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑁆𑀥𑀯𑀦𑀲𑁆𑀫𑀺𑀁 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—

1587\. “𑀓𑀢𑀫𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀘𑀢𑀽𑀲𑀼 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼? 𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀸𑀫𑀺…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀸𑀲𑀼…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁—  𑀇𑀫𑁂𑀲𑀼 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀢𑀽𑀲𑀼 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
