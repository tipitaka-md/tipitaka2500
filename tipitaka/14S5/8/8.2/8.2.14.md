# 𑁮𑁇𑁨𑁇𑁧𑁪 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

(922.)

1619\. “𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀧𑀦𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀻”𑀢𑀺𑁇

1622\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1623\. _𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀇𑀤𑁆𑀥𑀺 𑀤𑀺𑀩𑁆𑀩𑀁,_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀁 𑀞𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀥𑀸𑀢𑀼𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀢𑀺𑁇_
