# 𑁯𑁇𑁫𑁇𑁧--𑁧𑁦 𑀑𑀖𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀑𑀖𑀯𑀕𑁆𑀕

* 𑀑𑀖𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

(967--976.)

1640\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀅𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀫𑀸𑀦𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀼𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀛𑀸𑀦𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀛𑀸𑀦𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇

---

1641\. (𑀬𑀣𑀸 𑀫𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀢𑀣𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)

  
𑀑𑀖𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

1642\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1643\. _𑀑𑀖𑁄 𑀬𑁄𑀕𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀁,_  
_𑀕𑀦𑁆𑀣𑀸 𑀅𑀦𑀼𑀲𑀬𑁂𑀦 𑀘;_  
_𑀓𑀸𑀫𑀕𑀼𑀡𑀸 𑀦𑀻𑀯𑀭𑀡𑀸,_  
_𑀔𑀦𑁆𑀥𑀸 𑀑𑀭𑀼𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀢𑀺𑁇_  


---

1644\. 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀯𑀫𑀁𑁇
