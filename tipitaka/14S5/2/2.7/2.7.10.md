# 𑁨𑁇𑁭𑁇𑁧𑁦 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀆𑀦𑀸𑀧𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(247.)

680\. “𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑀸…𑀧𑁂…  𑀤𑀲𑀫𑀁𑁇

---

681\. 𑀆𑀦𑀸𑀧𑀸𑀦𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇



---

682\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



683\. _𑀅𑀝𑁆𑀞𑀺𑀓𑀧𑀼𑀍𑀅𑀯𑀓𑀁 𑀯𑀺𑀦𑀻𑀮𑀓𑀁,_  
_𑀯𑀺𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀓𑀁 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑀸 𑀓𑀭𑀼𑀡𑀸 𑀫𑀼𑀤𑀺𑀢𑀸,_  
_𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀆𑀦𑀸𑀧𑀸𑀦𑁂𑀦 𑀢𑁂 𑀤𑀲𑀸𑀢𑀺𑁇_
