# 𑁨𑁇𑁫𑁇𑁬 𑀤𑀮𑀺𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀤𑀮𑀺𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(227.)

542\. “‘𑀤𑀮𑀺𑀤𑁆𑀤𑁄, 𑀤𑀮𑀺𑀤𑁆𑀤𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, ‘𑀤𑀮𑀺𑀤𑁆𑀤𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺? “𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑀦𑀁 𑀅𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 ‘𑀤𑀮𑀺𑀤𑁆𑀤𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲—  𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑀦𑀁 𑀅𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 ‘𑀤𑀮𑀺𑀤𑁆𑀤𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇
