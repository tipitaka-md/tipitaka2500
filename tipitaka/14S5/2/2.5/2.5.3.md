# 𑁨𑁇𑁫𑁇𑁩 𑀫𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(224.)

536. “𑀫𑀸𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀕𑁆𑀕𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺; 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑁄 𑀫𑀕𑁆𑀕𑁄? 𑀬𑀤𑀺𑀤𑀁—  𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑁄 𑀫𑀕𑁆𑀕𑁄”𑀢𑀺𑁇
