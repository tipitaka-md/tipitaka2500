# 𑁨𑁇𑁨𑁇𑁭 𑀧𑀸𑀭𑀗𑁆𑀕𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀕𑀺𑀮𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀧𑀸𑀭𑀗𑁆𑀕𑀫𑀲𑀼𑀢𑁆𑀢

(198.)

446\. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀅𑀧𑀸𑀭𑀸 𑀧𑀸𑀭𑀁 𑀕𑀫𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀅𑀧𑀸𑀭𑀸 𑀧𑀸𑀭𑀁 𑀕𑀫𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻𑀢𑀺𑁇

447\. _𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀬𑁂 𑀚𑀦𑀸 𑀧𑀸𑀭𑀕𑀸𑀫𑀺𑀦𑁄;_  
_𑀅𑀣𑀸𑀬𑀁 𑀇𑀢𑀭𑀸 𑀧𑀚𑀸,_  
_𑀢𑀻𑀭𑀫𑁂𑀯𑀸𑀦𑀼𑀥𑀸𑀯𑀢𑀺𑁇_  


448\. _𑀬𑁂 𑀘 𑀔𑁄 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑁂,_  
_𑀥𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀢𑁂 𑀚𑀦𑀸 𑀧𑀸𑀭𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀤𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


449\. _𑀓𑀡𑁆𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀬,_  
_𑀲𑀼𑀓𑁆𑀓𑀁 𑀪𑀸𑀯𑁂𑀣 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀑𑀓𑀸 𑀅𑀦𑁄𑀓𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀯𑀺𑀯𑁂𑀓𑁂 𑀬𑀢𑁆𑀣 𑀤𑀽𑀭𑀫𑀁𑁇_  


450\. _𑀢𑀢𑁆𑀭𑀸𑀪𑀺𑀭𑀢𑀺𑀫𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂 𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑁄;_  
_𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑁂𑀬𑁆𑀬 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀮𑁂𑀲𑁂𑀳𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


451\. _𑀬𑁂𑀲𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀬𑀗𑁆𑀕𑁂𑀲𑀼,_  
_𑀲𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁;_  
_𑀆𑀤𑀸𑀦𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂,_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀬𑁂 𑀭𑀢𑀸;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀢𑁂 𑀮𑁄𑀓𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸”𑀢𑀺𑁇_
