# 𑁨𑁇𑁧𑁦𑁇𑁧--𑁧𑁦 𑀢𑀣𑀸𑀕𑀢𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀢𑀣𑀸𑀕𑀢𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢

(270--279.)

704. “𑀬𑀸𑀯𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑀤𑀸 𑀯𑀸 𑀤𑁆𑀯𑀺𑀧𑀤𑀸 𑀯𑀸 𑀘𑀢𑀼𑀧𑁆𑀧𑀤𑀸 𑀯𑀸 𑀩𑀳𑀼𑀧𑁆𑀧𑀤𑀸 𑀯𑀸”𑀢𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇

---

705. 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕𑁄 𑀤𑀲𑀫𑁄𑁇



---

706. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



707. _𑀢𑀣𑀸𑀕𑀢𑀁 𑀧𑀤𑀁 𑀓𑀽𑀝𑀁,_  
_𑀫𑀽𑀮𑀁 𑀲𑀸𑀭𑁂𑀦 𑀯𑀲𑁆𑀲𑀺𑀓𑀁;_  
_𑀭𑀸𑀚𑀸 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑀸 𑀘,_  
_𑀯𑀢𑁆𑀣𑁂𑀦 𑀤𑀲𑀫𑀁 𑀧𑀤𑀦𑁆𑀢𑀺𑁇_  


708. (𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕𑁄 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀯𑀲𑁂𑀦 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇)
