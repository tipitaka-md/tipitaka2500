# 𑁨𑁇𑁧𑁇𑁯 𑀧𑀞𑀫𑁅𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀩𑁆𑀩𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑁅𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

(190.)

416\. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
