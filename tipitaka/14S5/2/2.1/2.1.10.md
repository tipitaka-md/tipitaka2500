# 𑁨𑁇𑁧𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑁅𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀩𑁆𑀩𑀢𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑁅𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

(191.)

418\. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀼𑀕𑀢𑀯𑀺𑀦𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀼𑀕𑀢𑀯𑀺𑀦𑀬𑀸”𑀢𑀺𑁇
