# 𑁨𑁇𑁪𑁇𑁬 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀻𑀯𑀭𑀡𑀯𑀕𑁆𑀕

### 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(217.)

511. “𑀬𑁄𑀦𑀺𑀲𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘𑁂𑀯 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘𑁂𑀯 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇
