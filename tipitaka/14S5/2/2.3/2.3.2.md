# 𑁨𑁇𑁩𑁇𑁨 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀤𑁂𑀲𑀦𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀉𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕

* 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀤𑁂𑀲𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(203.)

465\. “𑀲𑀢𑁆𑀢 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺; 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸”𑀢𑀺𑁇
