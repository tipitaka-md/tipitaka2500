# 𑁫 𑀦𑀬𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀦𑀬𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(79.)

547. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀬𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁? 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀘 𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑀬 𑀘, 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸 𑀲𑀢𑁆𑀢𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀓𑁆𑀔𑁂𑀦 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀘𑀭𑀺𑀢𑀸𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀢𑁆𑀢𑀸 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀢𑀡𑁆𑀳𑀸𑀧𑀓𑁆𑀔𑁂𑀦 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀢𑀡𑁆𑀳𑀸𑀘𑀭𑀺𑀢𑀸𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀺𑀘𑀭𑀺𑀢𑀸 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀘𑀭𑀺𑀢𑀸 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇

548. 𑀢𑀢𑁆𑀣 𑀓𑀺𑀁𑀓𑀸𑀭𑀡𑀁 𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀘𑀭𑀺𑀢𑀸 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑀡𑁆𑀳𑀸𑀘𑀭𑀺𑀢𑀸 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺? 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀦𑀢𑁆𑀣𑀺 𑀲𑀘𑁆𑀘𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁, 𑀓𑀼𑀢𑁄 𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀧𑁆𑀧𑀓𑀸𑀲𑀦𑀸 𑀯𑀸 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀯𑀸 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀧𑁆𑀧𑀢𑁆𑀢𑀺 𑀯𑀸𑁇 𑀢𑁂 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀅𑀦𑀪𑀺𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀭𑀻𑀢𑀘𑁂𑀢𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼 “𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑁂𑀦 𑀲𑀼𑀔𑀁, 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀦𑀸𑀫 𑀲𑀼𑀔𑀁 𑀅𑀥𑀺𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀬𑁄 𑀓𑀸𑀫𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺, 𑀲𑁄 𑀮𑁄𑀓𑀁 𑀯𑀟𑁆𑀠𑀬𑀢𑀺, 𑀬𑁄 𑀮𑁄𑀓𑀁 𑀯𑀟𑁆𑀠𑀬𑀢𑀺, 𑀲𑁄 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀻𑀢𑀺 𑀢𑁂 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀻 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀲𑀼𑀔𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀧𑀼𑀜𑁆𑀜𑀲𑀜𑁆𑀜𑀻 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀘 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀘, 𑀢𑁂 𑀢𑀤𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀦𑁆𑀢𑀸 𑀭𑁄𑀕𑀫𑁂𑀯 𑀯𑀟𑁆𑀠𑀬𑀦𑁆𑀢𑀺, 𑀕𑀡𑁆𑀟𑀫𑁂𑀯 𑀯𑀟𑁆𑀠𑀬𑀦𑁆𑀢𑀺, 𑀲𑀮𑁆𑀮𑀫𑁂𑀯 𑀯𑀟𑁆𑀠𑀬𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀭𑁄𑀕𑀸𑀪𑀺𑀢𑀼𑀦𑁆𑀦𑀸 𑀕𑀡𑁆𑀟𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑀸 𑀲𑀮𑁆𑀮𑀸𑀦𑀼𑀯𑀺𑀤𑁆𑀥𑀸 𑀦𑀺𑀭𑀬𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀧𑁂𑀢𑀸𑀲𑀼𑀭𑁂𑀲𑀼 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀉𑀕𑁆𑀖𑀸𑀢𑀦𑀺𑀕𑁆𑀖𑀸𑀢𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀸 𑀭𑁄𑀕𑀕𑀡𑁆𑀟𑀲𑀮𑁆𑀮𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀦 𑀯𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀘 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑁄, 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀯𑁄𑀤𑀸𑀦𑀁𑁇 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀘 𑀭𑁄𑀕𑁄, 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀭𑁄𑀕𑀦𑀺𑀕𑁆𑀖𑀸𑀢𑀓𑀪𑁂𑀲𑀚𑁆𑀚𑀁𑁇 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀘 𑀕𑀡𑁆𑀟𑁄, 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀕𑀡𑁆𑀟𑀦𑀺𑀕𑁆𑀖𑀸𑀢𑀓𑀪𑁂𑀲𑀚𑁆𑀚𑀁𑁇 𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀘 𑀲𑀮𑁆𑀮𑁄, 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀮𑁆𑀮𑀼𑀤𑁆𑀥𑀭𑀡𑀪𑁂𑀲𑀚𑁆𑀚𑀁𑁇

549. 𑀢𑀢𑁆𑀣 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀢𑀤𑀪𑀺𑀲𑀗𑁆𑀕𑁄 𑀢𑀡𑁆𑀳𑀸 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄, 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁, 𑀲𑀫𑀼𑀤𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄, 𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

550. 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀺𑀘𑀭𑀺𑀢𑀸 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀘𑀭𑀺𑀢𑀸 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁, 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀯𑁂𑀤𑀦𑀸𑀯𑀦𑁆𑀢𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀯𑀦𑁆𑀢𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑀯𑀦𑁆𑀢𑀁…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀯𑀻𑀲𑀢𑀺𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑁇

555. 𑀢𑀢𑁆𑀣 𑀓𑀺𑀁𑀓𑀸𑀭𑀡𑀁, 𑀬𑀁 𑀢𑀡𑁆𑀳𑀸𑀘𑀭𑀺𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀘 𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀳𑀺 𑀓𑀸𑀫𑀸 𑀅𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑁂𑀘𑀺𑀬𑀫𑀸𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀭𑀢𑀺 𑀤𑀦𑁆𑀥𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀆𑀚𑀸𑀦𑀸𑀢𑀺? 𑀬𑁄 𑀧𑀦𑀸𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀘𑀭𑀺𑀢𑁄 𑀅𑀬𑀁 𑀆𑀤𑀺𑀢𑁄𑀬𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄 𑀪𑀯𑀢𑀺𑁇 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀯𑁂𑀘𑀺𑀬𑀫𑀸𑀦𑁄 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀘 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀭𑀢𑀺, 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀆𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀺 𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀼𑀯𑀺𑀥𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀘 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀘𑁇 𑀲𑀼𑀔𑀸𑀧𑀺 𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀼𑀯𑀺𑀥𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀘 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀘𑁇 𑀲𑀢𑁆𑀢𑀸𑀧𑀺 𑀤𑀼𑀯𑀺𑀥𑀸 𑀫𑀼𑀤𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀧𑀺 𑀢𑀺𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀧𑀺𑁇 𑀬𑁂 𑀫𑀼𑀤𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸, 𑀢𑁂 𑀤𑀦𑁆𑀥𑀜𑁆𑀘 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀺 𑀤𑀦𑁆𑀥𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑀺𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸, 𑀢𑁂 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀘 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀺, 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀇𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀧𑀤𑀸𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀦𑀺𑀬𑁆𑀬𑀺𑀁𑀲𑀼 𑀯𑀸 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀸 𑀦𑀺𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀢𑁂 𑀇𑀫𑀸𑀳𑀺 𑀏𑀯 𑀘𑀢𑀽𑀳𑀺 𑀧𑀝𑀺𑀧𑀤𑀸𑀳𑀺𑁇 𑀏𑀯𑀁 𑀅𑀭𑀺𑀬𑀸 𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀕𑁆𑀕𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀅𑀩𑀼𑀥𑀚𑀦𑀲𑁂𑀯𑀺𑀢𑀸𑀬 𑀩𑀸𑀮𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀯𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀦𑀦𑁆𑀤𑀺𑀬𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀯𑀝𑁆𑀝𑀦𑀢𑁆𑀣𑀁𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀯𑀝𑁆𑀝𑀲𑁆𑀲 𑀦𑀬𑀲𑁆𑀲 𑀪𑀽𑀫𑀻𑀢𑀺, 𑀢𑁂𑀦𑀸𑀳 “𑀢𑀡𑁆𑀳𑀜𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀘 𑀲𑀫𑀣𑁂𑀦𑀸”𑀢𑀺𑁇

556. 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀲𑀼 𑀳𑀺 𑀬𑁂 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀢𑀺 𑀢𑁂 𑀤𑀼𑀯𑀺𑀥𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀸—  𑀮𑁄𑀓𑀯𑀝𑁆𑀝𑀸𑀦𑀼𑀲𑀸𑀭𑀻 𑀘 𑀮𑁄𑀓𑀯𑀺𑀯𑀝𑁆𑀝𑀸𑀦𑀼𑀲𑀸𑀭𑀻 𑀘𑁇 𑀯𑀝𑁆𑀝𑀁 𑀦𑀸𑀫 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀯𑀺𑀯𑀝𑁆𑀝𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁𑁇 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑀸 𑀳𑁂𑀢𑀼 𑀲𑀁𑀲𑀸𑀭𑀲𑁆𑀲𑁇 𑀢𑀢𑁆𑀣 𑀓𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀦𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀁 𑀓𑀣𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁? 𑀉𑀧𑀘𑀬𑁂𑀦 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂𑀳𑀺 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑁂 𑀓𑀢𑁆𑀣 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸? 𑀤𑀲 𑀯𑀢𑁆𑀣𑀼𑀓𑁂 𑀓𑀺𑀮𑁂𑀲𑀧𑀼𑀜𑁆𑀚𑁂𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀤𑀲 𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀬𑁄𑀕𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀕𑀦𑁆𑀣𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀲𑀯𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀑𑀖𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀮𑁆𑀮𑀸, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀕𑀢𑀺𑀕𑀫𑀦𑀸𑀦𑀺𑁇 𑀧𑀞𑀫𑁂 𑀆𑀳𑀸𑀭𑁂 𑀧𑀞𑀫𑁄 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀆𑀳𑀸𑀭𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄, 𑀢𑀢𑀺𑀬𑁂 𑀆𑀳𑀸𑀭𑁂 𑀢𑀢𑀺𑀬𑁄 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀆𑀳𑀸𑀭𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄𑁇 𑀧𑀞𑀫𑁂 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂 𑀧𑀞𑀫𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀤𑀼𑀢𑀺𑀬𑁂 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂 𑀤𑀼𑀢𑀺𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀢𑀢𑀺𑀬𑁂 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂 𑀢𑀢𑀺𑀬𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀧𑀞𑀫𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀧𑀞𑀫𑁄 𑀬𑁄𑀕𑁄, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀬𑁄𑀕𑁄, 𑀢𑀢𑀺𑀬𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀢𑀢𑀺𑀬𑁄 𑀬𑁄𑀕𑁄, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀬𑁄𑀕𑁄𑁇 𑀧𑀞𑀫𑁂 𑀬𑁄𑀕𑁂 𑀧𑀞𑀫𑁄 𑀕𑀦𑁆𑀣𑁄, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀬𑁄𑀕𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀕𑀦𑁆𑀣𑁄, 𑀢𑀢𑀺𑀬𑁂 𑀬𑁄𑀕𑁂 𑀢𑀢𑀺𑀬𑁄 𑀕𑀦𑁆𑀣𑁄, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀬𑁄𑀕𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀕𑀦𑁆𑀣𑁄𑁇 𑀧𑀞𑀫𑁂 𑀕𑀦𑁆𑀣𑁂 𑀧𑀞𑀫𑁄 𑀆𑀲𑀯𑁄, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀕𑀦𑁆𑀣𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀲𑀯𑁄, 𑀢𑀢𑀺𑀬𑁂 𑀕𑀦𑁆𑀣𑁂 𑀢𑀢𑀺𑀬𑁄 𑀆𑀲𑀯𑁄, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀕𑀦𑁆𑀣𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀆𑀲𑀯𑁄𑁇 𑀧𑀞𑀫𑁂 𑀆𑀲𑀯𑁂 𑀧𑀞𑀫𑁄 𑀑𑀖𑁄, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀆𑀲𑀯𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀑𑀖𑁄, 𑀢𑀢𑀺𑀬𑁂 𑀆𑀲𑀯𑁂 𑀢𑀢𑀺𑀬𑁄 𑀑𑀖𑁄, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀆𑀲𑀯𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀑𑀖𑁄𑁇 𑀧𑀞𑀫𑁂 𑀑𑀖𑁂 𑀧𑀞𑀫𑁄 𑀲𑀮𑁆𑀮𑁄, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀑𑀖𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀮𑁆𑀮𑁄, 𑀢𑀢𑀺𑀬𑁂 𑀑𑀖𑁂 𑀢𑀢𑀺𑀬𑁄 𑀲𑀮𑁆𑀮𑁄, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀑𑀖𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀲𑀮𑁆𑀮𑁄𑁇 𑀧𑀞𑀫𑁂 𑀲𑀮𑁆𑀮𑁂 𑀧𑀞𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺, 𑀤𑀼𑀢𑀺𑀬𑁂 𑀲𑀮𑁆𑀮𑁂 𑀤𑀼𑀢𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺, 𑀢𑀢𑀺𑀬𑁂 𑀲𑀮𑁆𑀮𑁂 𑀢𑀢𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺, 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀲𑀮𑁆𑀮𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀧𑀞𑀫𑀸𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀁 𑀧𑀞𑀫𑀁 𑀅𑀕𑀢𑀺𑀕𑀫𑀦𑀁, 𑀤𑀼𑀢𑀺𑀬𑀸𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀕𑀢𑀺𑀕𑀫𑀦𑀁, 𑀢𑀢𑀺𑀬𑀸𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀁 𑀢𑀢𑀺𑀬𑀁 𑀅𑀕𑀢𑀺𑀕𑀫𑀦𑀁, 𑀘𑀢𑀼𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀕𑀢𑀺𑀕𑀫𑀦𑀁𑁇

559. 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑁂𑀦 𑀓𑀸𑀫𑁂𑀳𑀺 𑀲𑀁𑀬𑀼𑀚𑁆𑀚𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀓𑀸𑀫𑀬𑁄𑀕𑁄; 𑀪𑀯𑀼𑀧𑀸𑀤𑀸𑀦𑁂𑀦 𑀪𑀯𑁂𑀳𑀺 𑀲𑀁𑀬𑀼𑀚𑁆𑀚𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀪𑀯𑀬𑁄𑀕𑁄; 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑁂𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀁𑀬𑀼𑀚𑁆𑀚𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑁄𑀕𑁄; 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑁂𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀁𑀬𑀼𑀚𑁆𑀚𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄𑀕𑁄𑁇

560. 𑀧𑀞𑀫𑁂 𑀬𑁄𑀕𑁂 𑀞𑀺𑀢𑁄 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀓𑀸𑀬𑀁 𑀕𑀦𑁆𑀣𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄; 𑀤𑀼𑀢𑀺𑀬𑁂 𑀬𑁄𑀕𑁂 𑀞𑀺𑀢𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂𑀦 𑀓𑀸𑀬𑀁 𑀕𑀦𑁆𑀣𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄; 𑀢𑀢𑀺𑀬𑁂 𑀬𑁄𑀕𑁂 𑀞𑀺𑀢𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁂𑀦 𑀓𑀸𑀬𑀁 𑀕𑀦𑁆𑀣𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄; 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀬𑁄𑀕𑁂 𑀞𑀺𑀢𑁄 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁂𑀦 𑀓𑀸𑀬𑀁 𑀕𑀦𑁆𑀣𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄𑁇

561. 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁𑀕𑀦𑁆𑀣𑀺𑀢𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀆𑀲𑀯𑀦𑁆𑀢𑀺𑁇 𑀓𑀼𑀢𑁄 𑀘 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀆𑀲𑀯𑀦𑁆𑀢𑀻𑀢𑀺? 𑀅𑀦𑀼𑀲𑀬𑀢𑁄 𑀯𑀸 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀢𑁄 𑀯𑀸𑁇 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁂𑀦 𑀓𑀸𑀫𑀸𑀲𑀯𑁄, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁂𑀦 𑀪𑀯𑀸𑀲𑀯𑁄, 𑀧𑀭𑀸𑀫𑀸𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁂𑀦 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁄, 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁂𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄𑁇

562. 𑀢𑀲𑁆𑀲 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀲𑀯𑀸 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀕𑀢𑀸 𑀑𑀖𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀇𑀢𑀺 𑀆𑀲𑀯𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸 𑀑𑀖𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀸𑀲𑀯𑁂𑀦 𑀓𑀸𑀫𑁄𑀖𑁄, 𑀪𑀯𑀸𑀲𑀯𑁂𑀦 𑀪𑀯𑁄𑀖𑁄, 𑀤𑀺𑀝𑁆𑀞𑀸𑀲𑀯𑁂𑀦 𑀤𑀺𑀝𑁆𑀞𑁄𑀖𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁂𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑁄𑀖𑁄𑁇

563. 𑀢𑀲𑁆𑀲 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀑𑀖𑀸 𑀅𑀦𑀼𑀲𑀬𑀲𑀳𑀕𑀢𑀸 𑀅𑀚𑁆𑀛𑀸𑀲𑀬𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑀸 𑀳𑀤𑀬𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀲𑀮𑁆𑀮𑀸𑀇𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑁄𑀖𑁂𑀦 𑀭𑀸𑀕𑀲𑀮𑁆𑀮𑁄, 𑀪𑀯𑁄𑀖𑁂𑀦 𑀤𑁄𑀲𑀲𑀮𑁆𑀮𑁄, 𑀤𑀺𑀝𑁆𑀞𑁄𑀖𑁂𑀦 𑀫𑀸𑀦𑀲𑀮𑁆𑀮𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑁄𑀖𑁂𑀦 𑀫𑁄𑀳𑀲𑀮𑁆𑀮𑁄𑁇

564. 𑀢𑀲𑁆𑀲 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀲𑀮𑁆𑀮𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀘𑀢𑀽𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀡𑁆𑀞𑀳𑀢𑀺 𑀭𑀽𑀧𑁂 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼𑁇 𑀢𑀢𑁆𑀣 𑀭𑀸𑀕𑀲𑀮𑁆𑀮𑁂𑀦 𑀦𑀦𑁆𑀤𑀽𑀧𑀲𑁂𑀘𑀦𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀭𑀽𑀧𑀽𑀧𑀕𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺, 𑀤𑁄𑀲𑀲𑀮𑁆𑀮𑁂𑀦 𑀦𑀦𑁆𑀤𑀽𑀧𑀲𑁂𑀘𑀦𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀯𑁂𑀤𑀦𑀽𑀧𑀕𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺, 𑀫𑀸𑀦𑀲𑀮𑁆𑀮𑁂𑀦 𑀦𑀦𑁆𑀤𑀽𑀧𑀲𑁂𑀘𑀦𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀲𑀜𑁆𑀜𑀽𑀧𑀕𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺, 𑀫𑁄𑀳𑀲𑀮𑁆𑀮𑁂𑀦 𑀦𑀦𑁆𑀤𑀽𑀧𑀲𑁂𑀘𑀦𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀽𑀧𑀕𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇

574. 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀘 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀬𑁄 𑀘 𑀨𑀲𑁆𑀲𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀇𑀫𑁂 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁂𑀦 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀫𑀼𑀔𑁂𑀦 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄 𑀲𑀼𑀜𑁆𑀜𑀢𑀸𑀬, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦, 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀘 “𑀅𑀲𑀼𑀪𑁂 𑀲𑀼𑀪𑀦𑁆”𑀢𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄, 𑀬𑁄 𑀘 “𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄, 𑀇𑀫𑁂 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁂𑀦 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀫𑀼𑀔𑁂𑀦 𑀧𑀳𑀸𑀦𑀁 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 “𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀦𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄 𑀲𑀼𑀜𑁆𑀜𑀢𑀸𑀬, “𑀅𑀦𑀢𑁆𑀢𑀦𑀺 𑀅𑀢𑁆𑀢𑀸”𑀢𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦𑁇 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀪𑀯𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁂𑀦 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀫𑀼𑀔𑁂𑀦 𑀧𑀳𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀸𑀬, 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦𑁇 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀬𑁄𑀕𑁄 𑀘 𑀪𑀯𑀬𑁄𑀕𑁄 𑀘 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁂𑀦 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀫𑀼𑀔𑁂𑀦 𑀧𑀳𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑁄𑀕𑁄 𑀲𑀼𑀜𑁆𑀜𑀢𑀸𑀬, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄𑀕𑁄 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦𑁇 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀘 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁂𑀦 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀫𑀼𑀔𑁂𑀦 𑀧𑀳𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀭𑀸𑀫𑀸𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀲𑀼𑀜𑁆𑀜𑀢𑀸𑀬, 𑀇𑀤𑀁𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀓𑀸𑀬𑀕𑀦𑁆𑀣𑁄 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦𑁇

605. 𑀢𑁂𑀲𑀁 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀬𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑁄, 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁 𑀤𑁄𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀢𑀻𑀡𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺—  𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁; 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸—  𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀲𑀜𑁆𑀜𑀸—  𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀧𑀭𑀻𑀢𑀲𑀜𑁆𑀜𑀸—  𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑀸—  𑀤𑀼𑀓𑁆𑀔𑀤𑀼𑀓𑁆𑀔𑀢𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀤𑀼𑀓𑁆𑀔𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀤𑀼𑀓𑁆𑀔𑀢𑀸; 𑀢𑀬𑁄 𑀅𑀕𑁆𑀕𑀻—  𑀭𑀸𑀕𑀕𑁆𑀕𑀺 𑀤𑁄𑀲𑀕𑁆𑀕𑀺 𑀫𑁄𑀳𑀕𑁆𑀕𑀺; 𑀢𑀬𑁄 𑀲𑀮𑁆𑀮𑀸—  𑀭𑀸𑀕𑀲𑀮𑁆𑀮𑁄 𑀤𑁄𑀲𑀲𑀮𑁆𑀮𑁄 𑀫𑁄𑀳𑀲𑀮𑁆𑀮𑁄; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀚𑀝𑀸—  𑀭𑀸𑀕𑀚𑀝𑀸 𑀤𑁄𑀲𑀚𑀝𑀸 𑀫𑁄𑀳𑀚𑀝𑀸; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸—  𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄—  𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀇𑀤𑀁 𑀯𑁄𑀤𑀸𑀦𑀁𑁇 𑀢𑀻𑀡𑀺 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁 𑀅𑀫𑁄𑀳𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀼𑀘𑀭𑀺𑀢𑀸𑀦𑀺—  𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑀁𑁇 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸—  𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀢𑀬𑁄 𑀲𑀫𑀸𑀥𑀻—  𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀥𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀓𑀼𑀲𑀮𑀲𑀜𑁆𑀜𑀸—  𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀯𑀺𑀧𑀭𑀻𑀢𑀲𑀜𑁆𑀜𑀸—  𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀓𑀼𑀲𑀮𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸—  𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀼𑀲𑀮𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀓𑀼𑀲𑀮𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀸𑀦𑀺—  𑀓𑀸𑀬𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀁 𑀯𑀘𑀻𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀁 𑀫𑀦𑁄𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀁; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄—  𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀺𑀓𑁆𑀔𑀸—  𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁆𑀔𑀸; 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸—  𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀢𑀻𑀡𑀺 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀫𑀼𑀔𑀸𑀦𑀺—  𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀦𑁆𑀢𑀺𑁇

606. 𑀇𑀢𑀺 𑀔𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀢𑀻𑀡𑀺 𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑀻𑀡𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑀡𑁆𑀳𑀸𑀘𑀭𑀺𑀢𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀘𑀭𑀺𑀢𑁄 𑀘𑁇

607. 𑀢𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀬𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑁄, 𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀅𑀳𑀺𑀭𑀺𑀓𑀜𑁆𑀘 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀢𑀺 𑀘 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀘 𑀓𑁄𑀲𑀚𑁆𑀚𑀜𑁆𑀘 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀜𑁆𑀘 𑀅𑀳𑀗𑁆𑀓𑀸𑀭𑁄 𑀘 𑀫𑀫𑀗𑁆𑀓𑀸𑀭𑁄 𑀘 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀘 𑀧𑀫𑀸𑀤𑁄 𑀘 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀜𑁆𑀘 𑀅𑀲𑀁𑀯𑀭𑁄 𑀘 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘 𑀓𑁄𑀥𑁄 𑀘 𑀉𑀧𑀦𑀸𑀳𑁄 𑀘 𑀫𑀓𑁆𑀔𑁄 𑀘 𑀧𑀮𑀸𑀲𑁄 𑀘 𑀇𑀲𑁆𑀲𑀸 𑀘 𑀫𑀘𑁆𑀙𑁂𑀭𑀜𑁆𑀘 𑀫𑀸𑀬𑀸 𑀘 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺 𑀘 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑀸𑀢𑀺𑁇

608. 𑀢𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀇𑀤𑀁 𑀯𑁄𑀤𑀸𑀦𑀁, 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘 𑀳𑀺𑀭𑀻 𑀘 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘 𑀲𑀢𑀺 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀘 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 𑀘 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀜𑀸𑀡𑀜𑁆𑀘 𑀅𑀦𑁆𑀯𑀬𑁂 𑀜𑀸𑀡𑀜𑁆𑀘 𑀔𑀬𑁂 𑀜𑀸𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀜𑀸𑀡𑀜𑁆𑀘 𑀲𑀤𑁆𑀥𑀸 𑀘 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀜𑁆𑀘 𑀲𑀁𑀯𑀭𑁄 𑀘 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸 𑀘 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀘 𑀭𑀸𑀕𑀯𑀺𑀭𑀸𑀕𑀸 𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀭𑀸𑀕𑀸 𑀘 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀪𑀺𑀲𑀫𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸 𑀘 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺 𑀘 𑀅𑀓𑁆𑀓𑁄𑀥𑁄 𑀘 𑀅𑀦𑀼𑀧𑀦𑀸𑀳𑁄 𑀘 𑀅𑀫𑀓𑁆𑀔𑁄 𑀘 𑀅𑀧𑀮𑀸𑀲𑁄 𑀘 𑀇𑀲𑁆𑀲𑀸𑀧𑀳𑀸𑀦𑀜𑁆𑀘 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀧𑁆𑀧𑀳𑀸𑀦𑀜𑁆𑀘 𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀘 𑀲𑀗𑁆𑀔𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀘 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀅𑀲𑀗𑁆𑀔𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀘 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀲𑁅𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀽𑀢𑀺𑁇

609. 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀢𑀺𑀧𑀼𑀓𑁆𑀔𑀮𑀲𑁆𑀲 𑀘 𑀦𑀬𑀲𑁆𑀲 𑀅𑀗𑁆𑀓𑀼𑀲𑀲𑁆𑀲 𑀘 𑀦𑀬𑀲𑁆𑀲 𑀪𑀽𑀫𑀻𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 “𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑁂 𑀲𑀫𑀽𑀮𑁂𑀳𑀺 𑀦𑁂𑀢𑀻”𑀢𑀺 “𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀤𑀺𑀲𑀮𑁄𑀘𑀦𑁂𑀦𑀸”𑀢𑀺 𑀘𑁇
