# 𑁪𑁇𑁧𑁇𑁧𑁧 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(39.)

297\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑁄? “𑀏𑀓𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀯𑀺𑀥𑀸𑀳𑀺 𑀤𑁂𑀲𑁂𑀢𑀻”𑀢𑀺𑁇

298\. 𑀬𑀸 𑀧𑀓𑀢𑀺𑀓𑀣𑀸𑀬 𑀤𑁂𑀲𑀦𑀸𑁇 𑀅𑀬𑀁 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀓𑀸 𑀘 𑀧𑀓𑀢𑀺𑀓𑀣𑀸𑀬 𑀤𑁂𑀲𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀬𑀣𑀸 𑀪𑀕𑀯𑀸 𑀆𑀳—  “𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺 𑀅𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀙𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑁆𑀦𑀁 𑀆𑀬𑀢𑀦𑀸𑀦𑀁 𑀤𑀲𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

299\. 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀅𑀢𑁆𑀣𑀺 𑀭𑀸𑀕𑁄 𑀅𑀢𑁆𑀣𑀺 𑀦𑀦𑁆𑀤𑀻 𑀅𑀢𑁆𑀣𑀺 𑀢𑀡𑁆𑀳𑀸, 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀁𑁇 𑀬𑀢𑁆𑀣 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀲𑀲𑁄𑀓𑀁 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑀭𑀁 𑀲𑁅𑀧𑀸𑀬𑀸𑀲𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

300\. 𑀨𑀲𑁆𑀲𑁂 𑀘𑁂…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀅𑀢𑁆𑀣𑀺 𑀭𑀸𑀕𑁄 𑀅𑀢𑁆𑀣𑀺 𑀦𑀦𑁆𑀤𑀻 𑀅𑀢𑁆𑀣𑀺 𑀢𑀡𑁆𑀳𑀸, 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀁𑁇 𑀬𑀢𑁆𑀣 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀬𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀲𑀲𑁄𑀓𑀁 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑀭𑀁 𑀲𑁅𑀧𑀸𑀬𑀸𑀲𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀧𑀪𑀯𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀘𑁇

301\. 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀦𑀢𑁆𑀣𑀺 𑀭𑀸𑀕𑁄 𑀦𑀢𑁆𑀣𑀺 𑀦𑀦𑁆𑀤𑀻 𑀦𑀢𑁆𑀣𑀺 𑀢𑀡𑁆𑀳𑀸, 𑀅𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀯𑀺𑀭𑀽𑀍𑀳𑀁𑁇 𑀬𑀢𑁆𑀣 𑀅𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀯𑀺𑀭𑀽𑀍𑀳𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀅𑀲𑁄𑀓𑀁 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑀭𑀁 𑀅𑀦𑀼𑀧𑀸𑀬𑀸𑀲𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

302\. 𑀨𑀲𑁆𑀲𑁂 𑀘𑁂…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀦𑀢𑁆𑀣𑀺 𑀭𑀸𑀕𑁄 𑀦𑀢𑁆𑀣𑀺 𑀦𑀦𑁆𑀤𑀻 𑀦𑀢𑁆𑀣𑀺 𑀢𑀡𑁆𑀳𑀸, 𑀅𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀯𑀺𑀭𑀽𑀍𑀳𑀁𑁇 𑀬𑀢𑁆𑀣 𑀅𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀯𑀺𑀭𑀽𑀍𑀳𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀅𑀲𑁄𑀓𑀁 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑀭𑀁 𑀅𑀦𑀼𑀧𑀸𑀬𑀸𑀲𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

309\. “𑀲𑁄 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

310\. 𑀅𑀬𑀁 𑀧𑀝𑀺𑀯𑁂𑀥𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀁, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀪𑀽𑀫𑀺𑀬𑀸, 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑁆𑀲𑁇 “𑀲𑁄 𑀇𑀫𑁂 𑀆𑀲𑀯𑀸”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, “𑀅𑀬𑀁 𑀆𑀲𑀯𑀲𑀫𑀼𑀤𑀬𑁄”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, “𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 “𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 “𑀇𑀫𑁂 𑀆𑀲𑀯𑀸 𑀅𑀲𑁂𑀲𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

311\. 𑀅𑀬𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀔𑀬𑁂 𑀜𑀸𑀡𑀲𑁆𑀲, 𑀑𑀓𑀸𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀜𑀸𑀡𑀲𑁆𑀲, 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲, 𑀧𑀳𑀸𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲, 𑀆𑀭𑀫𑁆𑀪𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲, 𑀆𑀳𑀝𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀆𑀲𑀸𑀝𑀺𑀓𑀸𑀦𑀁, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀦𑀸𑀪𑀽𑀫𑀺𑀬𑀸, 𑀅𑀪𑀺𑀦𑀺𑀖𑀸𑀢𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁𑁇

312\. 𑀇𑀤𑀁 “𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 𑀅𑀬𑀁 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀅𑀬𑀁 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂… 𑁇 𑀅𑀬𑀁 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

313\. 𑀅𑀬𑀁 𑀤𑁂𑀲𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀁, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀼𑀢𑀫𑀬𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲, 𑀧𑀯𑀢𑁆𑀢𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀲𑁆𑀲𑁇

314\. “𑀢𑀁 𑀔𑁄 𑀧𑀦𑀺𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 “𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  “𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  “𑀲𑀸 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

315\. 𑀅𑀬𑀁 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬, 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲𑁇

316\. “𑀢𑀁 𑀔𑁄 𑀧𑀦𑀺𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 “𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄 𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁄”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  “𑀲𑀸 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸”𑀢𑀺 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

317\. 𑀅𑀬𑀁 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬, 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲, 𑀧𑀯𑀢𑁆𑀢𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀲𑁆𑀲𑁇

318\. _“𑀢𑀼𑀮𑀫𑀢𑀼𑀮𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀁,_  
_𑀪𑀯𑀲𑀗𑁆𑀔𑀸𑀭𑀫𑀯𑀲𑁆𑀲𑀚𑀺 𑀫𑀼𑀦𑀺;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀭𑀢𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑁄,_  
_𑀅𑀪𑀺𑀦𑁆𑀤𑀺 𑀓𑀯𑀘𑀫𑀺𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇_  


319\. “𑀢𑀼𑀮𑀦𑁆”𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀥𑀸𑀢𑀼𑁇 “𑀅𑀢𑀼𑀮𑀦𑁆”𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼, “𑀢𑀼𑀮𑀫𑀢𑀼𑀮𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀦𑁆”𑀢𑀺 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦𑀁𑁇 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬𑁇 “𑀪𑀯𑀲𑀗𑁆𑀔𑀸𑀭𑀫𑀯𑀲𑁆𑀲𑀚𑀺 𑀫𑀼𑀦𑀻”𑀢𑀺 𑀧𑀭𑀺𑀘𑁆𑀘𑀸𑀕𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲𑁇 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁇 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀭𑀢𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑁄”𑀢𑀺 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸𑁇 𑀞𑀺𑀢𑀺𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑀬𑁇 “𑀅𑀪𑀺𑀦𑁆𑀤𑀺 𑀓𑀯𑀘𑀫𑀺𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀯𑀦𑁆”𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀸𑀬, 𑀧𑀤𑀸𑀮𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀡𑁆𑀟𑀓𑁄𑀲𑀸𑀦𑀁𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀢𑀼𑀮𑀫𑀢𑀼𑀮𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇

320\. _“𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀲𑁄 𑀚𑀦𑁆𑀢𑀼 𑀓𑀣𑀁 𑀦𑀫𑁂𑀬𑁆𑀬;_  
_𑀓𑀸𑀫𑀸 𑀳𑀺 𑀮𑁄𑀓𑁂 𑀲𑀗𑁆𑀕𑁄𑀢𑀺 𑀜𑀢𑁆𑀯𑀸,_  
_𑀢𑁂𑀲𑀁 𑀲𑀢𑀻𑀫𑀸 𑀯𑀺𑀦𑀬𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑁂”𑀢𑀺𑁇_  


321\. **“𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”**𑀢𑀺 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 **“𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀦𑁆”**𑀢𑀺 𑀧𑀪𑀯𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 **“𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀻”**𑀢𑀺 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘 𑀜𑀸𑀡𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑁂𑀥𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 **“𑀓𑀸𑀫𑁂𑀲𑀼 𑀲𑁄 𑀚𑀦𑁆𑀢𑀼 𑀓𑀣𑀁 𑀦𑀫𑁂𑀬𑁆𑀬𑀸”**𑀢𑀺 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 **“𑀓𑀸𑀫𑀸 𑀳𑀺 𑀮𑁄𑀓𑁂 𑀲𑀗𑁆𑀕𑁄𑀢𑀺 𑀜𑀢𑁆𑀯𑀸”**𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀢𑁄 𑀤𑀲𑁆𑀲𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀓𑀸𑀫𑀸𑀦𑀁𑁇 𑀓𑀸𑀫𑀸 𑀳𑀺 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀽𑀧𑀫𑀸 𑀫𑀁𑀲𑀧𑁂𑀲𑀽𑀧𑀫𑀸 𑀧𑀸𑀯𑀓𑀓𑀧𑁆𑀧𑀸 𑀧𑀧𑀸𑀢𑁅𑀭𑀕𑁄𑀧𑀫𑀸 𑀘𑁇 **“𑀢𑁂𑀲𑀁 𑀲𑀢𑀻𑀫𑀸”**𑀢𑀺 𑀅𑀧𑀘𑀬𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸, 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑁆𑀲𑁇 **“𑀯𑀺𑀦𑀬𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑁂”**𑀢𑀺 𑀧𑀝𑀺𑀯𑁂𑀥𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀤𑁄𑀲𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀫𑁄𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲𑁇 **“𑀚𑀦𑁆𑀢𑀽”**𑀢𑀺 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀬𑁄𑀕𑀺𑀲𑁆𑀲𑁇 𑀬𑀤𑀸 𑀳𑀺 𑀬𑁄𑀕𑀻 𑀓𑀸𑀫𑀸 𑀲𑀗𑁆𑀕𑁄𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀉𑀧𑁆𑀧𑀸𑀤𑀬𑀢𑀺, 𑀲𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀯𑀸𑀬𑀫𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀸𑀬𑀸𑀫𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀑𑀭𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸𑁇 𑀢𑀢𑁆𑀣 𑀲𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀯𑀸𑀬𑀫𑀢𑀻𑀢𑀺 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀦𑀸𑀬, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲, 𑀆𑀭𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀞𑀺𑀢𑀺𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

322\. _“𑀫𑁄𑀳𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑁄 𑀮𑁄𑀓𑁄,_  
_𑀪𑀩𑁆𑀩𑀭𑀽𑀧𑁄𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀉𑀧𑀥𑀺𑀩𑀦𑁆𑀥𑀦𑁄 𑀩𑀸𑀮𑁄,_  
_𑀢𑀫𑀲𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄;_  
_𑀅𑀲𑁆𑀲𑀺𑀭𑀻 𑀯𑀺𑀬 𑀔𑀸𑀬𑀢𑀺,_  
_𑀧𑀲𑁆𑀲𑀢𑁄 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀦𑁆”𑀢𑀺𑁇_  


323\. **“𑀫𑁄𑀳𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑁄 𑀮𑁄𑀓𑁄”**𑀢𑀺 𑀤𑁂𑀲𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀸𑀦𑀁𑁇 **“𑀪𑀩𑁆𑀩𑀭𑀽𑀧𑁄𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀻”**𑀢𑀺 𑀯𑀺𑀧𑀭𑀻𑀢𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀮𑁄𑀓𑀲𑁆𑀲𑁇 **“𑀉𑀧𑀥𑀺𑀩𑀦𑁆𑀥𑀦𑁄 𑀩𑀸𑀮𑁄”**𑀢𑀺 𑀧𑀪𑀯𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀇𑀘𑁆𑀙𑀸𑀯𑀘𑀭𑀸𑀦𑀁, 𑀓𑀺𑀘𑁆𑀘𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁𑁇 𑀩𑀮𑀯𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁𑁇 𑀯𑀺𑀭𑀽𑀳𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 **“𑀢𑀫𑀲𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄”**𑀢𑀺 𑀤𑁂𑀲𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀥𑀓𑀸𑀭𑀲𑁆𑀲 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 **“𑀅𑀲𑁆𑀲𑀺𑀭𑀻 𑀯𑀺𑀬 𑀔𑀸𑀬𑀢𑀻”**𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑁇 **“𑀧𑀲𑁆𑀲𑀢𑁄 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀦𑁆”**𑀢𑀺 𑀧𑀝𑀺𑀯𑁂𑀥𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀦𑀁, 𑀭𑀸𑀕𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁 𑀤𑁄𑀲𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁 𑀫𑁄𑀳𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀫𑁄𑀳𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺𑁇

324\. “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁, 𑀦𑁄 𑀘𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀯𑀺𑀲𑁆𑀲 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁𑁇 𑀦𑀬𑀺𑀥 𑀚𑀸𑀢𑀲𑁆𑀲 𑀪𑀽𑀢𑀲𑁆𑀲 𑀓𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁, 𑀢𑀲𑁆𑀫𑀸 𑀚𑀸𑀢𑀲𑁆𑀲 𑀪𑀽𑀢𑀲𑁆𑀲 𑀓𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀻”𑀢𑀺𑁇

325\. “𑀦𑁄 𑀘𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀯𑀺𑀲𑁆𑀲 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀦𑁆”𑀢𑀺 𑀤𑁂𑀲𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 “𑀦𑀬𑀺𑀥 𑀚𑀸𑀢𑀲𑁆𑀲 𑀪𑀽𑀢𑀲𑁆𑀲 𑀓𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀗𑁆𑀔𑀢𑀲𑁆𑀲 𑀉𑀧𑀦𑀬𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 “𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀓𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀦𑁆”𑀢𑀺 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀚𑁄𑀢𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀘𑁇 “𑀢𑀲𑁆𑀫𑀸 𑀚𑀸𑀢𑀲𑁆𑀲 𑀪𑀽𑀢𑀲𑁆𑀲 𑀓𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀻”𑀢𑀺 𑀅𑀬𑀁 𑀯𑁂𑀯𑀘𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲, 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑁆𑀲, 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀁𑀲𑀸𑀭𑀢𑁄𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀦𑁄 𑀘𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀯𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀏𑀓𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀯𑀺𑀥𑀸𑀳𑀺 𑀤𑁂𑀲𑁂𑀢𑀻”𑀢𑀺𑁇
