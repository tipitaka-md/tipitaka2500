# 𑁪𑁇𑁧𑁇𑁬 𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

* 𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(25.)

191\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑁄 𑀳𑀸𑀭𑁄? “𑀦𑁂𑀭𑀼𑀢𑁆𑀢𑀫𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺 𑀅𑀬𑀁𑁇 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀦 𑀲𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀦𑁂𑀭𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀘 𑀦𑀺𑀤𑀸𑀦𑀜𑁆𑀘 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀲𑀦𑁆𑀥𑀺 𑀘 𑀕𑀯𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑁂𑀭𑀼𑀢𑁆𑀢𑀁, 𑀬𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀤𑀲𑀁𑀳𑀺𑀢𑀸, 𑀬𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀦𑀸𑀫𑀲𑁄 𑀜𑀸𑀡𑀁𑁇 𑀬𑀤𑀸 𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀘 𑀦𑀸𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀦𑀸𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀦𑀁 𑀅𑀪𑀺𑀦𑀺𑀭𑁄𑀧𑁂𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀢𑁆𑀣𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑀓𑀼𑀲𑀮𑁄 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀓𑀼𑀲𑀮𑁄 𑀤𑁂𑀲𑀦𑀸𑀓𑀼𑀲𑀮𑁄 𑀅𑀢𑀻𑀢𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄 𑀅𑀦𑀸𑀕𑀢𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄 𑀇𑀢𑁆𑀣𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄 𑀧𑀼𑀭𑀺𑀲𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀧𑀼𑀁𑀲𑀓𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄 𑀏𑀓𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄 𑀅𑀦𑁂𑀓𑀸𑀥𑀺𑀯𑀘𑀦𑀓𑀼𑀲𑀮𑁄, 𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑀦𑀺 𑀚𑀦𑀧𑀤𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸 𑀘 𑀚𑀦𑀧𑀤𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀅𑀬𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀤𑀲𑀁𑀳𑀺𑀢𑀸𑁇 (𑁧)

(26.)

192\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄?

193\. _“𑀥𑀫𑁆𑀫𑁄 𑀳𑀯𑁂 𑀭𑀓𑁆𑀔𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀁,_  
_𑀙𑀢𑁆𑀢𑀁 𑀫𑀳𑀦𑁆𑀢𑀁 𑀬𑀣 𑀯𑀲𑁆𑀲𑀓𑀸𑀮𑁂;_  
_𑀏𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑁄 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁂,_  
_𑀦 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻”𑀢𑀺𑁇_  


194\. 𑀇𑀥 𑀪𑀕𑀯𑀢𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄? 𑀬𑁂 𑀅𑀧𑀸𑀬𑁂𑀳𑀺 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀅𑀬𑀁 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇

195\. _“𑀘𑁄𑀭𑁄 𑀬𑀣𑀸 𑀲𑀦𑁆𑀥𑀺𑀫𑀼𑀔𑁂 𑀕𑀳𑀻𑀢𑁄,_  
_𑀲𑀓𑀫𑁆𑀫𑀼𑀦𑀸 𑀳𑀜𑁆𑀜𑀢𑀺 𑀩𑀚𑁆𑀛𑀢𑁂 𑀘;_  
_𑀏𑀯𑀁 𑀅𑀬𑀁 𑀧𑁂𑀘𑁆𑀘 𑀧𑀚𑀸 𑀧𑀭𑀢𑁆𑀣,_  
_𑀲𑀓𑀫𑁆𑀫𑀼𑀦𑀸 𑀳𑀜𑁆𑀜𑀢𑀺 𑀩𑀚𑁆𑀛𑀢𑁂 𑀘𑀸”𑀢𑀺𑁇_  


196\. 𑀇𑀥 𑀪𑀕𑀯𑀢𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄? 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀦𑀁 𑀓𑀢𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀘𑀺𑀢𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀻𑀬𑀸𑀦𑀁 𑀅𑀦𑀺𑀝𑁆𑀞𑀁 𑀅𑀲𑀸𑀢𑀁 𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀅𑀬𑀁 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇

197\. _“𑀲𑀼𑀔𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀬𑁄 𑀤𑀡𑁆𑀟𑁂𑀦 𑀯𑀺𑀳𑀺𑀁𑀲𑀢𑀺;_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀲𑀼𑀔𑀫𑁂𑀲𑀸𑀦𑁄,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑁄 𑀦 𑀮𑀪𑀢𑁂 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇_  


198\. 𑀇𑀥 𑀪𑀕𑀯𑀢𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄? 𑀬𑁂 𑀲𑀼𑀔𑁂𑀦 𑀅𑀢𑁆𑀣𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀦 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀅𑀬𑀁 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇

199\. _“𑀫𑀺𑀤𑁆𑀥𑀻 𑀬𑀤𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀕𑁆𑀖𑀲𑁄 𑀘,_  
_𑀦𑀺𑀤𑁆𑀤𑀸𑀬𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀲𑀸𑀬𑀻;_  
_𑀫𑀳𑀸𑀯𑀭𑀸𑀳𑁄𑀯 𑀦𑀺𑀯𑀸𑀧𑀧𑀼𑀝𑁆𑀞𑁄,_  
_𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀕𑀩𑁆𑀪𑀫𑀼𑀧𑁂𑀢𑀺 𑀫𑀦𑁆𑀤𑁄”𑀢𑀺𑁇_  


200\. 𑀇𑀥 𑀪𑀕𑀯𑀢𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄? 𑀬𑁂 𑀚𑀭𑀸𑀫𑀭𑀡𑁂𑀦 𑀅𑀝𑁆𑀝𑀺𑀬𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀦𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀸 𑀧𑀼𑀩𑁆𑀩𑀭𑀢𑁆𑀢𑀸𑀧𑀭𑀭𑀢𑁆𑀢𑀁 𑀚𑀸𑀕𑀭𑀺𑀬𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑀸 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀕𑀸𑀭𑀯𑀸 𑀘 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀣𑁂𑀭𑁂𑀲𑀼 𑀦𑀯𑁂𑀲𑀼 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀽𑀢𑀺 𑀅𑀬𑀁 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇

201\. _“𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄 𑀅𑀫𑀢𑀧𑀤𑀁,_  
_𑀧𑀫𑀸𑀤𑁄 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀧𑀤𑀁;_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀦 𑀫𑀻𑀬𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀧𑀫𑀢𑁆𑀢𑀸 𑀬𑀣𑀸 𑀫𑀢𑀸”𑀢𑀺𑁇_  


202\. 𑀇𑀥 𑀪𑀕𑀯𑀢𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄? 𑀬𑁂 𑀅𑀫𑀢𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀅𑀬𑀁 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇 𑀅𑀬𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇 (𑁨)

(27.)

203\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀺𑀤𑀸𑀦𑀁? 𑀬𑀣𑀸 𑀲𑁄 𑀥𑀦𑀺𑀬𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀆𑀳—

204\. _“𑀦𑀦𑁆𑀤𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑀺𑀫𑀸,_  
_𑀕𑁄𑀫𑀸 𑀕𑁄𑀳𑀺 𑀢𑀣𑁂𑀯 𑀦𑀦𑁆𑀤𑀢𑀺;_  
_𑀉𑀧𑀥𑀻 𑀳𑀺 𑀦𑀭𑀲𑁆𑀲 𑀦𑀦𑁆𑀤𑀦𑀸,_  
_𑀦 𑀳𑀺 𑀲𑁄 𑀦𑀦𑁆𑀤𑀢𑀺 𑀬𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀻”𑀢𑀺𑁇_  


205\. 𑀪𑀕𑀯𑀸 𑀆𑀳—

206\. _“𑀲𑁄𑀘𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑀺𑀫𑀸,_  
_𑀕𑁄𑀧𑀺𑀓𑁄 𑀕𑁄𑀳𑀺 𑀢𑀣𑁂𑀯 𑀲𑁄𑀘𑀢𑀺;_  
_𑀉𑀧𑀥𑀻 𑀳𑀺 𑀦𑀭𑀲𑁆𑀲 𑀲𑁄𑀘𑀦𑀸,_  
_𑀦 𑀳𑀺 𑀲𑁄 𑀲𑁄𑀘𑀢𑀺 𑀬𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀻”𑀢𑀺𑁇_  


207\. 𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀇𑀫𑀺𑀦𑀸 𑀦𑀺𑀤𑀸𑀦𑁂𑀦 𑀏𑀯𑀁 𑀜𑀸𑀬𑀢𑀺 “𑀇𑀥 𑀪𑀕𑀯𑀸 𑀩𑀸𑀳𑀺𑀭𑀁 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀁 𑀉𑀧𑀥𑀺𑀁 𑀆𑀳𑀸”𑀢𑀺𑁇 𑀬𑀣𑀸 𑀘 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀁 𑀧𑀸𑀢𑁂𑀲𑀺, 𑀪𑀕𑀯𑀸 𑀆𑀳—

208\. _“𑀲𑀘𑁂𑀧𑀺 𑀓𑁂𑀯𑀮𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀁 𑀘𑀮𑁂𑀲𑁆𑀲𑀲𑀺;_  
_𑀦𑁂𑀯 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺 𑀇𑀜𑁆𑀚𑀺𑀢𑀁𑁇_  


209\. _𑀦𑀪𑀁 𑀨𑀮𑁂𑀬𑁆𑀬𑀧𑁆𑀧𑀣𑀯𑀻 𑀘𑀮𑁂𑀬𑁆𑀬,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀧𑀸𑀡𑀸 𑀉𑀤 𑀲𑀦𑁆𑀢𑀲𑁂𑀬𑁆𑀬𑀼𑀁;_  
_𑀲𑀮𑁆𑀮𑀫𑁆𑀧𑀺 𑀘𑁂 𑀉𑀭𑀲𑀺 𑀧𑀓𑀧𑁆𑀧𑀬𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀉𑀧𑀥𑀻𑀲𑀼 𑀢𑀸𑀡𑀁 𑀦 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀸”𑀢𑀺𑁇_  


210\. 𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀇𑀫𑀺𑀦𑀸 𑀦𑀺𑀤𑀸𑀦𑁂𑀦 𑀏𑀯𑀁 𑀜𑀸𑀬𑀢𑀺 “𑀇𑀥 𑀪𑀕𑀯𑀸 𑀓𑀸𑀬𑀁 𑀉𑀧𑀥𑀺𑀁 𑀆𑀳𑀸”𑀢𑀺𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳—

211\. _“𑀦 𑀢𑀁 𑀤𑀍𑀳𑀁 𑀩𑀦𑁆𑀥𑀦𑀫𑀸𑀳𑀼 𑀥𑀻𑀭𑀸,_  
_𑀬𑀤𑀸𑀬𑀲𑀁 𑀤𑀸𑀭𑀼𑀚𑀧𑀩𑁆𑀩𑀚𑀜𑁆𑀘;_  
_𑀲𑀸𑀭𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀸 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑁂𑀲𑀼,_  
_𑀧𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀤𑀸𑀭𑁂𑀲𑀼 𑀘 𑀬𑀸 𑀅𑀧𑁂𑀓𑁆𑀔𑀸”𑀢𑀺𑁇_  


212\. 𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀇𑀫𑀺𑀦𑀸 𑀦𑀺𑀤𑀸𑀦𑁂𑀦 𑀏𑀯𑀁 𑀜𑀸𑀬𑀢𑀺 “𑀇𑀥 𑀪𑀕𑀯𑀸 𑀩𑀸𑀳𑀺𑀭𑁂𑀲𑀼 𑀯𑀢𑁆𑀣𑀽𑀲𑀼 𑀢𑀡𑁆𑀳𑀁 𑀆𑀳𑀸”𑀢𑀺𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳—

213\. _“𑀏𑀢𑀁 𑀤𑀍𑀳𑀁 𑀩𑀦𑁆𑀥𑀦𑀫𑀸𑀳𑀼 𑀥𑀻𑀭𑀸,_  
_𑀑𑀳𑀸𑀭𑀺𑀦𑀁 𑀲𑀺𑀣𑀺𑀮𑀁 𑀤𑀼𑀧𑁆𑀧𑀫𑀼𑀜𑁆𑀘𑀁;_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀙𑁂𑀢𑁆𑀯𑀸𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇_  


214\. 𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀇𑀫𑀺𑀦𑀸 𑀦𑀺𑀤𑀸𑀦𑁂𑀦 𑀏𑀯𑀁 𑀜𑀸𑀬𑀢𑀺 “𑀇𑀥 𑀪𑀕𑀯𑀸 𑀩𑀸𑀳𑀺𑀭𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀬 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀳𑀸𑀦𑀁 𑀆𑀳𑀸”𑀢𑀺𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳—

215\. _“𑀆𑀢𑀼𑀭𑀁 𑀅𑀲𑀼𑀘𑀺𑀁 𑀧𑀽𑀢𑀺𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀤𑁂𑀳𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀧𑀕𑁆𑀖𑀭𑀦𑁆𑀢𑀁 𑀤𑀺𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀩𑀸𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀦𑁆”𑀢𑀺𑁇_  


216\. 𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀇𑀫𑀺𑀦𑀸 𑀦𑀺𑀤𑀸𑀦𑁂𑀦 𑀏𑀯𑀁 𑀜𑀸𑀬𑀢𑀺 “𑀇𑀥 𑀪𑀕𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀬 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀳𑀸𑀦𑀁 𑀆𑀳𑀸”𑀢𑀺𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳—

217\. _“𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀤 𑀲𑀺𑀦𑁂𑀳𑀫𑀢𑁆𑀢𑀦𑁄,_  
_𑀓𑀼𑀫𑀼𑀤𑀁 𑀲𑀸𑀭𑀤𑀺𑀓𑀁𑀯 𑀧𑀸𑀡𑀺𑀦𑀸;_  
_𑀲𑀦𑁆𑀢𑀺𑀫𑀕𑁆𑀕𑀫𑁂𑀯 𑀩𑁆𑀭𑀽𑀳𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀼𑀕𑀢𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀦𑁆”𑀢𑀺𑁇_  


218\. 𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀇𑀫𑀺𑀦𑀸 𑀦𑀺𑀤𑀸𑀦𑁂𑀦 𑀏𑀯𑀁 𑀜𑀸𑀬𑀢𑀺 “𑀇𑀥 𑀪𑀕𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀬 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀳𑀸𑀦𑀁 𑀆𑀳𑀸”𑀢𑀺𑁇 𑀇𑀤𑀁 𑀦𑀺𑀤𑀸𑀦𑀁𑁇 (𑁩)

219\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀲𑀦𑁆𑀥𑀺𑁇 𑀬𑀣𑀸𑀳—

220\. _“𑀓𑀸𑀫𑀦𑁆𑀥𑀸 𑀚𑀸𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀢𑀡𑁆𑀳𑀸𑀙𑀤𑀦𑀙𑀸𑀤𑀺𑀢𑀸;_  
_𑀧𑀫𑀢𑁆𑀢𑀩𑀦𑁆𑀥𑀦𑀸 𑀩𑀤𑁆𑀥𑀸,_  
_𑀫𑀘𑁆𑀙𑀸𑀯 𑀓𑀼𑀫𑀺𑀦𑀸𑀫𑀼𑀔𑁂;_  
_𑀚𑀭𑀸𑀫𑀭𑀡𑀫𑀦𑁆𑀯𑁂𑀦𑁆𑀢𑀺,_  
_𑀯𑀘𑁆𑀙𑁄 𑀔𑀻𑀭𑀧𑀓𑁄𑀯 𑀫𑀸𑀢𑀭𑀦𑁆”𑀢𑀺𑁇_  


221\. 𑀅𑀬𑀁 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀯𑀼𑀢𑁆𑀢𑀸𑁇 𑀲𑀸 𑀓𑀢𑀫𑁂𑀦 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑁂𑀦 𑀬𑀼𑀚𑁆𑀚𑀢𑀺? 𑀬𑀣𑀸𑀳—

222\. _“𑀭𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀭𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀦𑁆𑀥𑀦𑁆𑀢𑀫𑀁 𑀢𑀤𑀸 𑀳𑁄𑀢𑀺,_  
_𑀬𑀁 𑀭𑀸𑀕𑁄 𑀲𑀳𑀢𑁂 𑀦𑀭𑀦𑁆”𑀢𑀺𑁇_  


223\. 𑀇𑀢𑀺 𑀅𑀦𑁆𑀥𑀢𑀸𑀬 𑀘 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀢𑀸𑀬 𑀘 𑀲𑀸𑀬𑁂𑀯 𑀢𑀡𑁆𑀳𑀸 𑀅𑀪𑀺𑀮𑀧𑀺𑀢𑀸𑁇 𑀬𑀜𑁆𑀘𑀸𑀳 𑀓𑀸𑀫𑀦𑁆𑀥𑀸 𑀚𑀸𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸, 𑀢𑀡𑁆𑀳𑀸𑀙𑀤𑀦𑀙𑀸𑀤𑀺𑀢𑀸𑀢𑀺𑁇 𑀬𑀜𑁆𑀘𑀸𑀳 𑀭𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀭𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀧𑀲𑁆𑀲𑀢𑀻𑀢𑀺, 𑀇𑀫𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀸𑀬𑁂𑀯 𑀢𑀡𑁆𑀳𑀸 𑀅𑀪𑀺𑀮𑀧𑀺𑀢𑀸𑁇 𑀬𑀁 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀁, 𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄, 𑀬𑀸 𑀘 𑀢𑀡𑁆𑀳𑀸 𑀧𑁄𑀦𑁄𑀪𑀯𑀺𑀓𑀸, 𑀬𑀜𑁆𑀘𑀸𑀳 𑀓𑀸𑀫𑀸𑀢𑀺 𑀇𑀫𑁂 𑀓𑀺𑀮𑁂𑀲𑀓𑀸𑀫𑀸𑁇 𑀬𑀜𑁆𑀘𑀸𑀳 𑀚𑀸𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸𑀢𑀺 𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀓𑀸𑀫𑀸𑀦𑀁 𑀧𑀬𑁄𑀕𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀓𑀺𑀮𑁂𑀲𑀯𑀲𑁂𑀦 𑀘 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀯𑀲𑁂𑀦 𑀘 𑀢𑀡𑁆𑀳𑀸𑀩𑀦𑁆𑀥𑀦𑀁 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 𑀬𑁂 𑀏𑀤𑀺𑀲𑀺𑀓𑀸, 𑀢𑁂 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀅𑀦𑁆𑀯𑁂𑀦𑁆𑀢𑀺, 𑀅𑀬𑀁 𑀪𑀕𑀯𑀢𑀸 𑀬𑀣𑀸𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀕𑀸𑀣𑀸𑀩𑀮𑁂𑀦 𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀫𑀦𑁆𑀯𑁂𑀦𑁆𑀢𑀻𑀢𑀺𑁇

224\. _“𑀬𑀲𑁆𑀲 𑀧𑀧𑀜𑁆𑀘𑀸 𑀞𑀺𑀢𑀻 𑀘 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀦𑁆𑀤𑀸𑀦𑀁 𑀧𑀮𑀺𑀖𑀜𑁆𑀘 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁄;_  
_𑀢𑀁 𑀦𑀺𑀢𑁆𑀢𑀡𑁆𑀳𑀁 𑀫𑀼𑀦𑀺𑀁 𑀘𑀭𑀦𑁆𑀢𑀁,_  
_𑀦 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁄𑀧𑀺 𑀮𑁄𑀓𑁄”𑀢𑀺𑁇_  


225\. 𑀧𑀧𑀜𑁆𑀘𑀸 𑀦𑀸𑀫 𑀢𑀡𑁆𑀳𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀸, 𑀢𑀤𑀪𑀺𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇 𑀞𑀺𑀢𑀺 𑀦𑀸𑀫 𑀅𑀦𑀼𑀲𑀬𑀸𑁇 𑀲𑀦𑁆𑀤𑀸𑀦𑀁 𑀦𑀸𑀫 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀬𑀸𑀦𑀺 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀢𑀡𑁆𑀳𑀸𑀬 𑀚𑀸𑀮𑀺𑀦𑀺𑀬𑀸 𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺𑁇 𑀧𑀮𑀺𑀖𑁄 𑀦𑀸𑀫 𑀫𑁄𑀳𑁄𑁇 𑀬𑁂 𑀘 𑀧𑀧𑀜𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀬𑀸 𑀘 𑀞𑀺𑀢𑀺 𑀬𑀁 𑀲𑀦𑁆𑀤𑀸𑀦𑀜𑁆𑀘 𑀬𑀁 𑀧𑀮𑀺𑀖𑀜𑁆𑀘 𑀬𑁄 𑀏𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀦𑀺𑀢𑁆𑀢𑀡𑁆𑀳𑁄 𑀇𑀢𑀺𑁇

(28.)

226\. 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀦𑀻𑀬𑀸 𑀯𑀸 𑀉𑀧𑀧𑀚𑁆𑀚𑀯𑁂𑀤𑀦𑀻𑀬𑀸 𑀯𑀸 𑀅𑀧𑀭𑀸𑀧𑀭𑀺𑀬𑀯𑁂𑀤𑀦𑀻𑀬𑀸 𑀯𑀸, 𑀏𑀯𑀁 𑀢𑀡𑁆𑀳𑀸 𑀢𑀺𑀯𑀺𑀥𑀁 𑀨𑀮𑀁 𑀤𑁂𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑁂 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑁂 𑀯𑀸 𑀅𑀧𑀭𑁂 𑀯𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀸 𑀆𑀳—  “𑀬𑀁 𑀮𑁄𑀪𑀧𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀫𑀦𑀲𑀸 𑀯𑀸, 𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀁 𑀅𑀦𑀼𑀪𑁄𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑁂 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑁂 𑀯𑀸 𑀅𑀧𑀭𑁂 𑀯𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂”𑀢𑀺𑁇 𑀇𑀤𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑁂𑀦 𑀬𑀼𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀯𑁂𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀅𑀧𑀭𑀸𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑁂𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁, 𑀏𑀯𑀁 𑀓𑀫𑁆𑀫𑀁 𑀢𑀺𑀥𑀸 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑁂 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑁂 𑀯𑀸 𑀅𑀧𑀭𑁂 𑀯𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑁇 𑀬𑀣𑀸𑀳—

227\. “𑀬𑀜𑁆𑀘𑁂 𑀩𑀸𑀮𑁄 𑀇𑀥 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑁂 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂 𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀉𑀧𑀧𑀚𑁆𑀚𑁂 𑀯𑀸 𑀅𑀧𑀭𑁂 𑀯𑀸 𑀧𑀭𑀺𑀬𑀸𑀬𑁂”𑀢𑀺𑁇 𑀇𑀤𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑁂𑀦 𑀬𑀼𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀩𑀮𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀤𑀲𑁆𑀲𑀦𑀩𑀮𑁂𑀦, 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀪𑀸𑀯𑀦𑀸𑀩𑀮𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑀦𑀻𑀢𑀺 𑀏𑀯𑀁 𑀢𑀡𑁆𑀳𑀸𑀧𑀺 𑀢𑀺𑀥𑀸 𑀧𑀳𑀻𑀬𑀢𑀺𑁇 𑀬𑀸 𑀦𑀺𑀢𑁆𑀢𑀡𑁆𑀳𑀸𑀢𑀸 𑀅𑀬𑀁 𑀲𑁅𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑁇 𑀪𑁂𑀤𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀬𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑁇

228\. 𑀧𑀧𑀜𑁆𑀘𑁄 𑀦𑀸𑀫 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑁄𑁇 𑀬𑀜𑁆𑀘𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀧𑀧𑀜𑁆𑀘𑁂𑀢𑀺 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁 𑀆𑀭𑀩𑁆𑀪𑀸”𑀢𑀺𑁇 𑀬𑀜𑁆𑀘𑀸𑀳 𑀪𑀕𑀯𑀸—  “𑀅𑀢𑀻𑀢𑁂, 𑀭𑀸𑀥, 𑀭𑀽𑀧𑁂 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄 𑀳𑁄𑀳𑀺, 𑀅𑀦𑀸𑀕𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀸 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸”𑀢𑀺𑁇 𑀇𑀤𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑁂𑀦 𑀬𑀼𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁄 𑀘𑀸𑀧𑀺 𑀧𑀧𑀜𑁆𑀘𑁄 𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀬𑀸 𑀘 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑀸, 𑀇𑀤𑀁 𑀏𑀓𑀢𑁆𑀣𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀅𑀓𑁆𑀔𑀭𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸𑁇 𑀏𑀯𑀁 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀼𑀢𑁆𑀢𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁄𑀚𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀪𑀯𑀢𑀺𑁇

229\. 𑀲𑁄 𑀘𑀸𑀬𑀁 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑁄 𑀲𑀦𑁆𑀥𑀺 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁄 𑀅𑀢𑁆𑀣𑀲𑀦𑁆𑀥𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑀦𑁆𑀥𑀺 𑀤𑁂𑀲𑀦𑀸𑀲𑀦𑁆𑀥𑀺 𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀲𑀦𑁆𑀥𑀻𑀢𑀺𑁇

230\. 𑀢𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀲𑀦𑁆𑀥𑀺 𑀙𑀧𑁆𑀧𑀤𑀸𑀦𑀺 𑀲𑀗𑁆𑀓𑀸𑀲𑀦𑀸 𑀧𑀓𑀸𑀲𑀦𑀸 𑀯𑀺𑀯𑀭𑀡𑀸 𑀯𑀺𑀪𑀚𑀦𑀸 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀫𑁆𑀫𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺𑁇

231\. 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑀦𑁆𑀥𑀺 𑀙𑀧𑁆𑀧𑀤𑀸𑀦𑀺 𑀅𑀓𑁆𑀔𑀭𑀁 𑀧𑀤𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀆𑀓𑀸𑀭𑁄 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺 𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑁄𑀢𑀺𑁇

232\. 𑀤𑁂𑀲𑀦𑀸𑀲𑀦𑁆𑀥𑀺 𑀦 𑀘 𑀧𑀣𑀯𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺 𑀛𑀸𑀬𑀻 𑀛𑀸𑀬𑀢𑀺 𑀘𑁇 𑀦 𑀘 𑀆𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺 𑀛𑀸𑀬𑀻 𑀛𑀸𑀬𑀢𑀺 𑀘, 𑀦 𑀘 𑀢𑁂𑀚𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺 𑀛𑀸𑀬𑀻 𑀛𑀸𑀬𑀢𑀺 𑀘, 𑀦 𑀘 𑀯𑀸𑀬𑀼𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺 𑀛𑀸𑀬𑀻 𑀛𑀸𑀬𑀢𑀺 𑀘…𑀧𑁂…  𑀦 𑀘 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬…  𑀦 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬…  𑀦 𑀘 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬…  𑀦 𑀘 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬…  𑀦 𑀘 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬…  𑀦 𑀘 𑀧𑀭𑀮𑁄𑀓𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀛𑀸𑀬𑀢𑀺 𑀛𑀸𑀬𑀻 𑀛𑀸𑀬𑀢𑀺 𑀘𑁇 𑀬𑀫𑀺𑀤𑀁 𑀉𑀪𑀬𑀫𑀦𑁆𑀢𑀭𑁂𑀦 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑀺𑀢𑀁 𑀯𑀺𑀘𑀸𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸𑀦𑀼𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦 𑀛𑀸𑀬𑀢𑀺 𑀛𑀸𑀬𑀻 𑀛𑀸𑀬𑀢𑀺 𑀘𑁇 𑀅𑀬𑀁 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀦 𑀜𑀸𑀬𑀢𑀺 𑀛𑀸𑀬𑀦𑁆𑀢𑁄𑁇

234\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀲𑀦𑁆𑀥𑀺? 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁂𑀦 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁂𑀦 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀓𑀺𑀮𑁂𑀲𑁂𑀦 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀯𑁄𑀤𑀸𑀦𑁂𑀦 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀸𑀭𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀸𑀭𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑀣𑁂𑀦 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑀺𑀭𑀺𑀓𑁂𑀦 𑀘 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀺𑀬𑀸 𑀘 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀲𑀢𑀺𑀬𑀸 𑀘 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀢𑀺𑀬𑀸 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀬𑁄𑀦𑀺𑀬𑀸 𑀘 𑀅𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀬𑁄𑀦𑀺𑀬𑀸 𑀘 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀓𑁄𑀲𑀚𑁆𑀚𑁂𑀦 𑀘 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁂𑀦 𑀘 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑁂𑀦 𑀘 𑀧𑀫𑀸𑀤𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀤𑁆𑀥𑀸𑀬 𑀘 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑁂𑀦 𑀘 𑀅𑀲𑀁𑀯𑀭𑀡𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑁂𑀦 𑀘 𑀲𑀁𑀯𑀭𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀘 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂𑀦 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀦𑀻𑀯𑀭𑀡𑁂𑀳𑀺 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀳𑀺 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀭𑀸𑀕𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀘 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀘 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑁅𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀬𑀁 𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀲𑀦𑁆𑀥𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀦𑁂𑀭𑀼𑀢𑁆𑀢𑀫𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺𑁇
