# 𑁪𑁇𑁧𑁇𑁧𑁩 𑀲𑁄𑀥𑀦𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁄𑀥𑀦𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(45.)

384. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑁄𑀥𑀦𑁄 𑀳𑀸𑀭𑁄? “𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀫𑁆𑀳𑀺 𑀧𑀜𑁆𑀳𑁂”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 𑀬𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑀺𑀢𑁄 𑀧𑀸𑀭𑀸𑀬𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺—

385. _“𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀺𑀯𑀼𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀧𑁆𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀓𑀺𑀲𑁆𑀲𑀸𑀪𑀺𑀮𑁂𑀧𑀦𑀁 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀦𑁆”𑀢𑀺𑁇_  


386. _“𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀺𑀯𑀼𑀢𑁄 𑀮𑁄𑀓𑁄, __(𑀅𑀚𑀺𑀢𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀯𑀺𑀯𑀺𑀘𑁆𑀙𑀸 𑀧𑀫𑀸𑀤𑀸 𑀦𑀧𑁆𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀚𑀧𑁆𑀧𑀸𑀪𑀺𑀮𑁂𑀧𑀦𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀦𑁆”𑀢𑀺𑁇_  


387. “𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀺𑀯𑀼𑀢𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺 𑀧𑀜𑁆𑀳𑁂 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀺𑀯𑀼𑀢𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀧𑀤𑀁 𑀲𑁄𑀥𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀆𑀭𑀫𑁆𑀪𑀁𑁇 “𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀧𑁆𑀧𑀓𑀸𑀲𑀢𑀻”𑀢𑀺 𑀧𑀜𑁆𑀳𑁂 “𑀯𑀺𑀯𑀺𑀘𑁆𑀙𑀸 𑀧𑀫𑀸𑀤𑀸 𑀦𑀧𑁆𑀧𑀓𑀸𑀲𑀢𑀻”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀧𑀤𑀁 𑀲𑁄𑀥𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀆𑀭𑀫𑁆𑀪𑀁𑁇 “𑀓𑀺𑀲𑁆𑀲𑀸𑀪𑀺𑀮𑁂𑀧𑀦𑀁 𑀩𑁆𑀭𑀽𑀲𑀻”𑀢𑀺 𑀧𑀜𑁆𑀳𑁂 “𑀚𑀧𑁆𑀧𑀸𑀪𑀺𑀮𑁂𑀧𑀦𑀁 𑀩𑁆𑀭𑀽𑀫𑀻”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀧𑀤𑀁 𑀲𑁄𑀥𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀆𑀭𑀫𑁆𑀪𑀁𑁇 “𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀦𑁆”𑀢𑀺 𑀧𑀜𑁆𑀳𑁂 “𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀦𑁆”𑀢𑀺 𑀲𑀼𑀤𑁆𑀥𑁄 𑀆𑀭𑀫𑁆𑀪𑁄𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀺𑀯𑀼𑀢𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺𑁇

388. _“𑀲𑀯𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑁄𑀢𑀸, __(𑀇𑀘𑁆𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑀺𑀢𑁄)___  
_𑀲𑁄𑀢𑀸𑀦𑀁 𑀓𑀺𑀁 𑀦𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀲𑁄𑀢𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑁂𑀦 𑀲𑁄𑀢𑀸 𑀧𑀺𑀥𑀻𑀬𑀭𑁂”𑀢𑀺𑁇_  


389. _“𑀬𑀸𑀦𑀺 𑀲𑁄𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, __(𑀅𑀚𑀺𑀢𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀲𑀢𑀺 𑀢𑁂𑀲𑀁 𑀦𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀲𑁄𑀢𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀢𑁂 𑀧𑀺𑀥𑀻𑀬𑀭𑁂”𑀢𑀺𑁇_  


390. “𑀲𑀯𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑁄𑀢𑀸, 𑀲𑁄𑀢𑀸𑀦𑀁 𑀓𑀺𑀁 𑀦𑀺𑀯𑀸𑀭𑀡𑀦𑁆”𑀢𑀺 𑀧𑀜𑁆𑀳𑁂 “𑀬𑀸𑀦𑀺 𑀲𑁄𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀲𑀢𑀺 𑀢𑁂𑀲𑀁 𑀦𑀺𑀯𑀸𑀭𑀡𑀦𑁆”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀧𑀤𑀁 𑀲𑁄𑀥𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀆𑀭𑀫𑁆𑀪𑀁𑁇 “𑀲𑁄𑀢𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀁 𑀩𑁆𑀭𑀽𑀳𑀺, 𑀓𑁂𑀦 𑀲𑁄𑀢𑀸 𑀧𑀺𑀥𑀻𑀬𑀭𑁂”𑀢𑀺 𑀧𑀜𑁆𑀳𑁂 “𑀲𑁄𑀢𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀁 𑀩𑁆𑀭𑀽𑀫𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀢𑁂 𑀧𑀺𑀥𑀻𑀬𑀭𑁂”𑀢𑀺 𑀲𑀼𑀤𑁆𑀥𑁄 𑀆𑀭𑀫𑁆𑀪𑁄𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀬𑀸𑀦𑀺 𑀲𑁄𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇

391. _“𑀧𑀜𑁆𑀜𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀘, __(𑀇𑀘𑁆𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑀺𑀢𑁄)___  
_𑀦𑀸𑀫𑀭𑀽𑀧𑀜𑁆𑀘 𑀫𑀸𑀭𑀺𑀲;_  
_𑀏𑀢𑀁 𑀫𑁂 𑀧𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀢𑁆𑀣𑁂𑀢𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_  


392. 𑀧𑀜𑁆𑀳𑁂—

393. _“𑀬𑀫𑁂𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺,_  
_𑀅𑀚𑀺𑀢 𑀢𑀁 𑀯𑀤𑀸𑀫𑀺 𑀢𑁂;_  
_𑀬𑀢𑁆𑀣 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘,_  
_𑀅𑀲𑁂𑀲𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁂𑀦,_  
_𑀏𑀢𑁆𑀣𑁂𑀢𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_  


394. 𑀲𑀼𑀤𑁆𑀥𑁄 𑀆𑀭𑀫𑁆𑀪𑁄𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀬𑀫𑁂𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀻”𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀏𑀯𑀁 𑀲𑀼𑀤𑁆𑀥𑁄 𑀆𑀭𑀫𑁆𑀪𑁄, 𑀲𑁄 𑀧𑀜𑁆𑀳𑁄 𑀯𑀺𑀲𑀚𑁆𑀚𑀺𑀢𑁄 𑀪𑀯𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀧𑀦 𑀆𑀭𑀫𑁆𑀪𑁄 𑀅𑀲𑀼𑀤𑁆𑀥𑁄, 𑀦 𑀢𑀸𑀯 𑀲𑁄 𑀧𑀜𑁆𑀳𑁄 𑀯𑀺𑀲𑀚𑁆𑀚𑀺𑀢𑁄 𑀪𑀯𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀫𑁆𑀳𑀺 𑀧𑀜𑁆𑀳𑁂”𑀢𑀺𑁇
