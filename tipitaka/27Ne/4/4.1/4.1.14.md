# 𑁪𑁇𑁧𑁇𑁧𑁪 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(46.)

396\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁄 𑀳𑀸𑀭𑁄? “𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑀬 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂𑀧𑀺 𑀘 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑀬 𑀦𑀺𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸”𑀢𑀺𑁇

397\. 𑀬𑁂 𑀢𑀢𑁆𑀣 𑀦𑀺𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸, 𑀢𑀣𑀸 𑀢𑁂 𑀥𑀸𑀭𑀬𑀺𑀢𑀩𑁆𑀩𑀸𑁇

398\. “𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀚𑀸𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀚𑀭𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀩𑁆𑀬𑀸𑀥𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀫𑀭𑀡𑀁 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀧𑁆𑀧𑀺𑀬𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑁄 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀧𑀺𑀬𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑁄𑀕𑁄 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀭𑀽𑀧𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀲𑀜𑁆𑀜𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

399\. “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀬𑀸𑀬𑀁 𑀢𑀡𑁆𑀳𑀸 𑀧𑁄𑀦𑁄𑀪𑀯𑀺𑀓𑀸 𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑀲𑀳𑀕𑀢𑀸 𑀢𑀢𑁆𑀭𑀢𑀢𑁆𑀭𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀦𑀻𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

400\. “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀬𑁄 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑁄 𑀘𑀸𑀕𑁄 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀦𑀸𑀮𑀬𑁄𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

401\. “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

402\. “𑀫𑀕𑁆𑀕𑁄”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑀕𑁆𑀕𑁄? 𑀦𑀺𑀭𑀬𑀕𑀸𑀫𑀻 𑀫𑀕𑁆𑀕𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀕𑀸𑀫𑀻 𑀫𑀕𑁆𑀕𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀕𑀸𑀫𑀻 𑀫𑀕𑁆𑀕𑁄 𑀅𑀲𑀼𑀭𑀬𑁄𑀦𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀕𑁆𑀕𑀕𑀸𑀫𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀕𑀸𑀫𑀻 𑀫𑀕𑁆𑀕𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀸𑀫𑀻 𑀫𑀕𑁆𑀕𑁄𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

403\. “𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀦𑀺𑀭𑁄𑀥𑁄? 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀺𑀭𑁄𑀥𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀺𑀭𑁄𑀥𑁄 𑀅𑀦𑀼𑀦𑀬𑀦𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀖𑀦𑀺𑀭𑁄𑀥𑁄 𑀫𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄 𑀫𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄 𑀧𑀍𑀆𑀲𑀦𑀺𑀭𑁄𑀥𑁄 𑀇𑀲𑁆𑀲𑀸𑀦𑀺𑀭𑁄𑀥𑁄 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

404\. “𑀭𑀽𑀧𑀦𑁆”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑀁 𑀭𑀽𑀧𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀭𑀽𑀧𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀺? 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑁇

(47.)

405\. 𑀤𑁆𑀯𑀻𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀲𑀗𑁆𑀔𑁂𑀧𑁂𑀦 𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀘𑁇 𑀓𑀣𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺? 𑀯𑀻𑀲𑀢𑀺𑀬𑀸 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀘𑀢𑀽𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀙𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇

406\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀯𑀻𑀲𑀢𑀺𑀬𑀸 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀦𑀔𑀸 𑀤𑀦𑁆𑀢𑀸 𑀢𑀘𑁄, 𑀫𑀁𑀲𑀁 𑀦𑁆𑀳𑀸𑀭𑀼 𑀅𑀝𑁆𑀞𑀺 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀯𑀓𑁆𑀓𑀁, 𑀳𑀤𑀬𑀁 𑀬𑀓𑀦𑀁 𑀓𑀺𑀮𑁄𑀫𑀓𑀁 𑀧𑀺𑀳𑀓𑀁 𑀧𑀧𑁆𑀨𑀸𑀲𑀁, 𑀅𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀕𑀼𑀡𑀁 𑀉𑀤𑀭𑀺𑀬𑀁 𑀓𑀭𑀻𑀲𑀁 𑀫𑀢𑁆𑀣𑀓𑁂 𑀫𑀢𑁆𑀣𑀮𑀼𑀗𑁆𑀕𑀦𑁆𑀢𑀺 𑀇𑀫𑁂𑀳𑀺 𑀯𑀻𑀲𑀢𑀺𑀬𑀸 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇

407\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺? 𑀅𑀢𑁆𑀣𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀧𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀫𑁆𑀳𑀁 𑀧𑀼𑀩𑁆𑀩𑁄 𑀮𑁄𑀳𑀺𑀢𑀁 𑀲𑁂𑀤𑁄 𑀫𑁂𑀤𑁄 𑀅𑀲𑁆𑀲𑀼 𑀯𑀲𑀸 𑀔𑁂𑀍𑀑 𑀲𑀺𑀗𑁆𑀖𑀸𑀡𑀺𑀓𑀸 𑀮𑀲𑀺𑀓𑀸 𑀫𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀇𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇

408\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺? 𑀬𑁂𑀦 𑀘 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀢𑀺, 𑀬𑁂𑀦 𑀘 𑀚𑀻𑀭𑀻𑀬𑀢𑀺, 𑀬𑁂𑀦 𑀘 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀢𑀺, 𑀬𑁂𑀦 𑀘 𑀅𑀲𑀺𑀢𑀧𑀻𑀢𑀔𑀸𑀬𑀺𑀢𑀲𑀸𑀬𑀺𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇

409\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀙𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺? 𑀉𑀤𑁆𑀥𑀗𑁆𑀕𑀫𑀸 𑀯𑀸𑀢𑀸, 𑀅𑀥𑁄𑀕𑀫𑀸 𑀯𑀸𑀢𑀸, 𑀓𑀼𑀘𑁆𑀙𑀺𑀲𑀬𑀸 𑀯𑀸𑀢𑀸, 𑀓𑁄𑀝𑁆𑀞𑀸𑀲𑀬𑀸 𑀯𑀸𑀢𑀸, 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄 𑀯𑀸𑀢𑀸, 𑀅𑀲𑁆𑀲𑀸𑀲𑁄 𑀧𑀲𑁆𑀲𑀸𑀲𑁄 𑀇𑀢𑀺, 𑀇𑀫𑁂𑀳𑀺 𑀙𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇

415\. “𑀲𑀫𑀸𑀥𑀻”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑀸𑀥𑀺? 𑀲𑀭𑀡𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀭𑀡𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀲𑀯𑁂𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀯𑁂𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀲𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀦𑀺𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀲𑀸𑀫𑀺𑀲𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀏𑀓𑀁𑀲𑀪𑀸𑀯𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀉𑀪𑀬𑀁𑀲𑀪𑀸𑀯𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀉𑀪𑀬𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑀪𑀸𑀯𑀦𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀬𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀬𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀮𑁄𑀓𑀺𑀬𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

416\. “𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀆𑀕𑀸𑀍𑀳𑀧𑀝𑀺𑀧𑀤𑀸, 𑀦𑀺𑀚𑁆𑀛𑀸𑀫𑀧𑀝𑀺𑀧𑀤𑀸, 𑀫𑀚𑁆𑀛𑀺𑀫𑀧𑀝𑀺𑀧𑀤𑀸, 𑀅𑀓𑁆𑀔𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀔𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀲𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀤𑀫𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸, 𑀲𑀼𑀔𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸, 𑀲𑀼𑀔𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

417\. “𑀓𑀸𑀬𑁄”𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀸𑀬𑁄? 𑀦𑀸𑀫𑀓𑀸𑀬𑁄 𑀭𑀽𑀧𑀓𑀸𑀬𑁄 𑀘𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀭𑀽𑀧𑀓𑀸𑀬𑁄? 𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀦𑀔𑀸 𑀤𑀦𑁆𑀢𑀸 𑀢𑀘𑁄 𑀫𑀁𑀲𑀁 𑀦𑁆𑀳𑀸𑀭𑀼 𑀅𑀝𑁆𑀞𑀺 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀯𑀓𑁆𑀓𑀁 𑀳𑀤𑀬𑀁 𑀬𑀓𑀦𑀁 𑀓𑀺𑀮𑁄𑀫𑀓𑀁 𑀧𑀺𑀳𑀓𑀁 𑀧𑀧𑁆𑀨𑀸𑀲𑀁 𑀅𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀕𑀼𑀡𑀁 𑀉𑀤𑀭𑀺𑀬𑀁 𑀓𑀭𑀻𑀲𑀁 𑀧𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀫𑁆𑀳𑀁 𑀧𑀼𑀩𑁆𑀩𑁄 𑀮𑁄𑀳𑀺𑀢𑀁 𑀲𑁂𑀤𑁄 𑀫𑁂𑀤𑁄 𑀅𑀲𑁆𑀲𑀼 𑀯𑀲𑀸 𑀔𑁂𑀍𑀑 𑀲𑀺𑀗𑁆𑀖𑀸𑀡𑀺𑀓𑀸 𑀮𑀲𑀺𑀓𑀸 𑀫𑀼𑀢𑁆𑀢𑀁 𑀫𑀢𑁆𑀣𑀮𑀼𑀗𑁆𑀕𑀦𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀭𑀽𑀧𑀓𑀸𑀬𑁄𑁇 𑀦𑀸𑀫𑀓𑀸𑀬𑁄 𑀦𑀸𑀫 𑀯𑁂𑀤𑀦𑀸 𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀨𑀲𑁆𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄𑀢𑀺—  𑀅𑀬𑀁 𑀦𑀸𑀫𑀓𑀸𑀬𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇

418\. 𑀏𑀯𑀁 𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀫𑀸𑀦𑀪𑀸𑀯𑁄, 𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑀬 𑀏𑀓𑀻 𑀪𑀯𑀢𑀺𑁇 𑀬𑁂𑀦 𑀬𑁂𑀦 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀮𑀓𑁆𑀔𑀡𑁄, 𑀢𑁂𑀦 𑀢𑁂𑀦 𑀯𑁂𑀫𑀢𑁆𑀢𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀼𑀢𑁆𑀢𑁂 𑀯𑀸 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂 𑀯𑀸 𑀕𑀸𑀣𑀸𑀬𑀁 𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂𑀦 𑀯𑀻𑀫𑀁𑀲𑀬𑀺𑀢𑀩𑁆𑀩𑀁, 𑀓𑀺𑀁 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑀬 𑀧𑀼𑀘𑁆𑀙𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑀬𑀸𑀢𑀺𑁇 𑀬𑀤𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑀬 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀁, 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑀬 𑀯𑀺𑀲𑀚𑁆𑀚𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀤𑀺 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑀬 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀁, 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑀬 𑀯𑀺𑀲𑀚𑁆𑀚𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀤𑀺 𑀲𑀢𑁆𑀢𑀸𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀁, 𑀲𑀢𑁆𑀢𑀸𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀺𑀲𑀚𑁆𑀚𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀤𑀺 𑀥𑀫𑁆𑀫𑀸𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀁, 𑀥𑀫𑁆𑀫𑀸𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀺𑀲𑀚𑁆𑀚𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀁 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀯𑀺𑀲𑀚𑁆𑀚𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑀬 𑀥𑀫𑁆𑀫𑀸”𑀢𑀺𑁇
