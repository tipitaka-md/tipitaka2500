# 𑁪𑁇𑁧𑁇𑁧𑁫 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(49.)

420. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄 𑀳𑀸𑀭𑁄? “𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀚𑀦𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

421. 𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀚𑀦𑀬𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀲𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄𑁇 𑀓𑀺𑀁𑀮𑀓𑁆𑀔𑀡𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄? 𑀚𑀦𑀓𑀮𑀓𑁆𑀔𑀡𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄𑁇 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀚𑀦𑀬𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼 𑀘 𑀧𑀘𑁆𑀘𑀬𑁄 𑀘𑁇 𑀢𑀢𑁆𑀣 𑀓𑀺𑀁𑀮𑀓𑁆𑀔𑀡𑁄 𑀳𑁂𑀢𑀼, 𑀓𑀺𑀁𑀮𑀓𑁆𑀔𑀡𑁄 𑀧𑀘𑁆𑀘𑀬𑁄? 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀮𑀓𑁆𑀔𑀡𑁄 𑀳𑁂𑀢𑀼, 𑀲𑀸𑀥𑀸𑀭𑀡𑀮𑀓𑁆𑀔𑀡𑁄 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀬𑀣𑀸 𑀓𑀺𑀁 𑀪𑀯𑁂? 𑀬𑀣𑀸 𑀅𑀗𑁆𑀓𑀼𑀭𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀬𑀸 𑀩𑀻𑀚𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁, 𑀧𑀣𑀯𑀻 𑀆𑀧𑁄 𑀘 𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 𑀅𑀗𑁆𑀓𑀼𑀭𑀲𑁆𑀲 𑀳𑀺 𑀧𑀣𑀯𑀻 𑀆𑀧𑁄 𑀘 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀖𑀝𑁂 𑀤𑀼𑀤𑁆𑀥𑀁 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀤𑀥𑀺 𑀪𑀯𑀢𑀺, 𑀦 𑀘𑀢𑁆𑀣𑀺 𑀏𑀓𑀓𑀸𑀮𑀲𑀫𑀯𑀥𑀸𑀦𑀁 𑀤𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘 𑀤𑀥𑀺𑀲𑁆𑀲 𑀘𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀦𑀢𑁆𑀣𑀺 𑀏𑀓𑀓𑀸𑀮𑀲𑀫𑀯𑀥𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀧𑀘𑁆𑀘𑀬𑀲𑁆𑀲 𑀘𑁇

422. 𑀅𑀬𑀜𑁆𑀳𑀺 𑀲𑀁𑀲𑀸𑀭𑁄 𑀲𑀳𑁂𑀢𑀼 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁄𑁇 𑀯𑀼𑀢𑁆𑀢𑀁 𑀳𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑁄 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄𑁇 𑀇𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀳𑁂𑀢𑀼 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀺𑀓𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀳𑁂𑀢𑀼𑁇 𑀢𑀢𑁆𑀣 𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀺𑀓𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀺𑀓𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀭𑀽𑀳𑀦𑀸𑀬, 𑀩𑀻𑀚𑀗𑁆𑀓𑀼𑀭𑁄 𑀯𑀺𑀬 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀳𑁂𑀢𑀼𑀢𑀸𑀬𑁇 𑀬𑀁 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀨𑀮𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀺, 𑀇𑀤𑀫𑀲𑁆𑀲 𑀧𑀭𑀫𑁆𑀧𑀭𑀳𑁂𑀢𑀼𑀢𑀸𑀬 𑀳𑁂𑀢𑀼𑀪𑀽𑀢𑀁𑁇 𑀤𑀼𑀯𑀺𑀥𑁄 𑀳𑀺 𑀳𑁂𑀢𑀼 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀳𑁂𑀢𑀼 𑀧𑀭𑀫𑁆𑀧𑀭𑀳𑁂𑀢𑀼 𑀘, 𑀏𑀯𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑀧𑀺 𑀤𑀼𑀯𑀺𑀥𑁄 𑀳𑁂𑀢𑀼 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀳𑁂𑀢𑀼 𑀧𑀭𑀫𑁆𑀧𑀭𑀳𑁂𑀢𑀼 𑀘𑁇

423. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀣𑀸𑀮𑀓𑀜𑁆𑀘 𑀯𑀝𑁆𑀝𑀺 𑀘 𑀢𑁂𑀮𑀜𑁆𑀘 𑀧𑀤𑀻𑀧𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑀪𑀽𑀢𑀁 𑀦 𑀲𑀪𑀸𑀯𑀳𑁂𑀢𑀼, 𑀦 𑀳𑀺 𑀲𑀓𑁆𑀓𑀸 𑀣𑀸𑀮𑀓𑀜𑁆𑀘 𑀯𑀝𑁆𑀝𑀺𑀜𑁆𑀘 𑀢𑁂𑀮𑀜𑁆𑀘 𑀅𑀦𑀕𑁆𑀕𑀺𑀓𑀁 𑀤𑀻𑀧𑁂𑀢𑀼𑀁 𑀧𑀤𑀻𑀧𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑀪𑀽𑀢𑀁𑁇 𑀧𑀤𑀻𑀧𑁄 𑀯𑀺𑀬 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼 𑀳𑁄𑀢𑀺𑁇 𑀇𑀢𑀺 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼, 𑀧𑀭𑀪𑀸𑀯𑁄 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑁄 𑀳𑁂𑀢𑀼, 𑀩𑀸𑀳𑀺𑀭𑁄 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀚𑀦𑀓𑁄 𑀳𑁂𑀢𑀼, 𑀧𑀭𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑁄 𑀳𑁂𑀢𑀼, 𑀲𑀸𑀥𑀸𑀭𑀡𑁄 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

424. 𑀅𑀯𑀼𑀧𑀘𑁆𑀙𑁂𑀤𑀢𑁆𑀣𑁄 𑀲𑀦𑁆𑀢𑀢𑀺 𑀅𑀢𑁆𑀣𑁄, 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀅𑀢𑁆𑀣𑁄 𑀨𑀮𑀢𑁆𑀣𑁄, 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺 𑀅𑀢𑁆𑀣𑁄 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀢𑁆𑀣𑁄, 𑀧𑀮𑀺𑀩𑁄𑀥𑀢𑁆𑀣𑁄 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀢𑁆𑀣𑁄, 𑀅𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀲𑀬𑀢𑁆𑀣𑁄, 𑀅𑀲𑀫𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀢𑁆𑀣𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀢𑁆𑀣𑁄, 𑀅𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀢𑁆𑀣𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀩𑀻𑀚𑀢𑁆𑀣𑁄𑁇 𑀬𑀢𑁆𑀣 𑀅𑀯𑀼𑀧𑀘𑁆𑀙𑁂𑀤𑁄 𑀢𑀢𑁆𑀣 𑀲𑀦𑁆𑀢𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀲𑀦𑁆𑀢𑀢𑀺 𑀢𑀢𑁆𑀣 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀨𑀮𑀁, 𑀬𑀢𑁆𑀣 𑀨𑀮𑀁 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺, 𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺 𑀢𑀢𑁆𑀣 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄, 𑀬𑀢𑁆𑀣 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄 𑀢𑀢𑁆𑀣 𑀧𑀮𑀺𑀩𑁄𑀥𑁄, 𑀬𑀢𑁆𑀣 𑀧𑀮𑀺𑀩𑁄𑀥𑁄 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀬𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀅𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄𑁇 𑀬𑀢𑁆𑀣 𑀅𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀲𑀬𑁄, 𑀬𑀢𑁆𑀣 𑀅𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁆𑀣 𑀅𑀲𑀫𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄, 𑀬𑀢𑁆𑀣 𑀅𑀲𑀫𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸, 𑀬𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀢𑀢𑁆𑀣 𑀲𑀸𑀲𑀯𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀁, 𑀬𑀢𑁆𑀣 𑀲𑀸𑀲𑀯𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀢𑀢𑁆𑀣 𑀩𑀻𑀚𑀢𑁆𑀣𑁄𑁇

425. 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀢𑀺𑀢𑁆𑀣𑀜𑁆𑀜𑀼𑀢𑀸 𑀧𑀻𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀧𑀻𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀧𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀧𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀅𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

426. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀽𑀧𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀼 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀧𑀘𑁆𑀘𑀬𑀢𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀭𑀽𑀧𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀢𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀆𑀮𑁄𑀓𑁄 𑀲𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀬𑀢𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀨𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀯𑁂𑀤𑀦𑀸 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀯𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀪𑀯𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀚𑀸𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀲𑁄𑀓𑁄 𑀧𑀭𑀺𑀤𑁂𑀯𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀧𑀭𑀺𑀤𑁂𑀯𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑀸𑀬𑀸𑀲𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀲𑀪𑀸𑀯𑁄 𑀳𑁂𑀢𑀼𑁇 𑀏𑀯𑀁 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀚𑀦𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
