# 𑁪𑁇𑁧𑁇𑁯 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

* 𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(35.)

273\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁄 𑀳𑀸𑀭𑁄? “𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂”𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀪𑀯𑀢𑀺𑁇 𑀬𑁂 𑀘𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀅𑀦𑁂𑀓𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀲𑁆𑀲 𑀅𑀦𑁂𑀓𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑁄 𑀪𑀯𑀢𑀺𑁇 𑀬𑁂 𑀘𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀅𑀦𑁂𑀓𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀲𑁆𑀲 𑀅𑀦𑁂𑀓𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀁 𑀪𑀯𑀢𑀺𑁇 𑀬𑁂 𑀘𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀅𑀦𑁂𑀓𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀲𑁆𑀲 𑀅𑀦𑁂𑀓𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

(36.)

274\. 𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀲𑁆𑀲 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀲𑁆𑀲 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀳𑀻𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀲𑁆𑀲 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀳𑀻𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀲𑁆𑀲 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑀲𑁆𑀲 𑀧𑀺𑀲𑀼𑀡𑀸 𑀯𑀸𑀘𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀡𑁆𑀳𑀯𑀸𑀘𑀲𑁆𑀲 𑀨𑀭𑀼𑀲𑀸 𑀯𑀸𑀘𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀓𑀸𑀮𑀯𑀸𑀤𑀺𑀲𑁆𑀲 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼𑀲𑁆𑀲 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

275\. 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁 𑀕𑀭𑀳𑀦𑁆𑀢𑀺, 𑀦𑁂𑀲𑀁 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀕𑀸𑀭𑀬𑁆𑀳𑀸 𑀯𑀸𑀤𑀸𑀦𑀼𑀯𑀸𑀤𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀜𑁆𑀘 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀕𑀭𑀳𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸, 𑀢𑁂𑀲𑀁 𑀪𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀼𑀚𑁆𑀚𑀸 𑀘 𑀧𑀸𑀲𑀁𑀲𑀸 𑀘𑁇 𑀏𑀯𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀁 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀜𑁆𑀘 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀕𑀭𑀳𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀬𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀸, 𑀢𑁂𑀲𑀁 𑀪𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀼𑀚𑁆𑀚𑀸 𑀘 𑀧𑀸𑀲𑀁𑀲𑀸 𑀘𑁇

276\. 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼 “𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸 𑀓𑀸𑀫𑀸, 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸 𑀓𑀸𑀫𑀸, 𑀆𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀓𑀸𑀫𑀸, 𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀓𑀸𑀫𑀸, 𑀪𑀸𑀯𑀬𑀺𑀢𑀩𑁆𑀩𑀸 𑀓𑀸𑀫𑀸, 𑀩𑀳𑀼𑀮𑀻𑀓𑀸𑀢𑀩𑁆𑀩𑀸 𑀓𑀸𑀫𑀸”𑀢𑀺𑁇 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑁂𑀭𑀫𑀡𑀻 𑀢𑁂𑀲𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑁇

277\. 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀓𑁂𑀘𑀺 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼 “𑀅𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑀸𑀦𑀼𑀬𑁄𑀕𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀥𑀫𑁆𑀫𑁄𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼 “𑀤𑀼𑀓𑁆𑀔𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇 𑀲𑀼𑀔𑁄 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀥𑀫𑁆𑀫𑁄𑁇 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀲𑀼𑀪𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀅𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑀬𑀢𑀺 𑀯𑀸 𑀉𑀧𑀕𑀘𑁆𑀙𑀢𑀺 𑀯𑀸, 𑀢𑀲𑁆𑀲 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀬𑁄 𑀧𑀝𑀺𑀧𑀓𑁆𑀔𑁄, 𑀲𑁆𑀯𑀲𑁆𑀲 𑀅𑀦𑀺𑀝𑁆𑀞𑀢𑁄 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀪𑀯𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀥𑀫𑁆𑀫𑁂”𑀢𑀺𑁇
