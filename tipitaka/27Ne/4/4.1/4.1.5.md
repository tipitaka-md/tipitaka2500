# 𑁪𑁇𑁧𑁇𑁫 𑀮𑀓𑁆𑀔𑀡𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀮𑀓𑁆𑀔𑀡𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(23.)

180\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑀓𑁆𑀔𑀡𑁄 𑀳𑀸𑀭𑁄? “𑀯𑀼𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀏𑀓𑀥𑀫𑁆𑀫𑁂”𑀢𑀺, 𑀅𑀬𑀁 𑀮𑀓𑁆𑀔𑀡𑁄 𑀳𑀸𑀭𑁄𑁇 𑀓𑀺𑀁 𑀮𑀓𑁆𑀔𑀬𑀢𑀺? 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀸, 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀏𑀓𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀯𑀼𑀢𑁆𑀢𑁂 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀬𑀣𑀸 𑀓𑀺𑀁 𑀪𑀯𑁂? 𑀬𑀣𑀸𑀳 𑀪𑀕𑀯𑀸—

181\. “𑀘𑀓𑁆𑀔𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀝𑁆𑀞𑀺𑀢𑀁 𑀇𑀢𑁆𑀢𑀭𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀧𑀪𑀗𑁆𑀕𑀼 𑀧𑀭𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀩𑁆𑀬𑀲𑀦𑀁 𑀘𑀮𑀦𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀍𑀅𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀯𑀥𑀓𑀁 𑀅𑀫𑀺𑀢𑁆𑀢𑀫𑀚𑁆𑀛𑁂”𑁇 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀢𑁆𑀢𑁂 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀺 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦? 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀳𑀺 𑀙 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀺 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀯𑀥𑀓𑀝𑁆𑀞𑁂𑀦 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀸𑀦𑀺𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳 𑀪𑀕𑀯𑀸—

182\. “𑀅𑀢𑀻𑀢𑁂, 𑀭𑀸𑀥, 𑀭𑀽𑀧𑁂 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄 𑀳𑁄𑀳𑀺, 𑀅𑀦𑀸𑀕𑀢𑀁 𑀭𑀽𑀧𑀁 𑀫𑀸 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀘𑀸𑀕𑀸𑀬 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚”𑁇 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀯𑀼𑀢𑁆𑀢𑁂 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸 𑀔𑀦𑁆𑀥𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦? 𑀲𑀩𑁆𑀩𑁂 𑀳𑀺 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀬𑀫𑀓𑁄𑀯𑀸𑀤𑀲𑀼𑀢𑁆𑀢𑁂 𑀯𑀥𑀓𑀝𑁆𑀞𑁂𑀦 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀸 𑀯𑀼𑀢𑁆𑀢𑀸𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳 𑀪𑀕𑀯𑀸—

183\. _“𑀬𑁂𑀲𑀜𑁆𑀘 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺;_  
_𑀅𑀓𑀺𑀘𑁆𑀘𑀁 𑀢𑁂 𑀦 𑀲𑁂𑀯𑀦𑁆𑀢𑀺,_  
_𑀓𑀺𑀘𑁆𑀘𑁂 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀺𑀦𑁄”𑁇_  


184\. 𑀇𑀢𑀺 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸𑀬 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀕𑀢𑀸 𑀲𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀕𑀢𑀸 𑀘𑁇 𑀢𑀣𑀸 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸 𑀫𑀼𑀢𑀁 𑀯𑀸𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑁂 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀯𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀬𑀣𑀸 𑀘𑀸𑀳 𑀪𑀕𑀯𑀸—

185\. “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁆𑀯𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀸𑀳𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁”𑁇 “𑀆𑀢𑀸𑀧𑀻”𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, “𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, “𑀲𑀢𑀺𑀫𑀸”𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, “𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆”𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀏𑀯𑀁 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦, 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀢𑁆𑀢𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁𑁇

(24.)

186\. 𑀘𑀢𑀽𑀲𑀼 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀘𑀢𑀽𑀲𑀼 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀘𑀢𑀽𑀲𑀼 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀲𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀲𑀼 𑀩𑀮𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢𑀲𑀼 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀩𑁄𑀥𑀗𑁆𑀕𑀫𑀸 𑀥𑀫𑁆𑀫𑀸 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦, 𑀲𑀩𑁆𑀩𑁂 𑀳𑀺 𑀩𑁄𑀥𑀗𑁆𑀕𑀫𑀸 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀸 𑀦𑁂𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀮𑀓𑁆𑀔𑀡𑁂𑀦 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀸, 𑀢𑁂 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

187\. 𑀏𑀯𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀢𑁆𑀢𑀸 𑀧𑀳𑀸𑀦𑀁 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀽𑀲𑀼 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺, 𑀆𑀳𑀸𑀭𑀸 𑀘𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀳𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄 𑀪𑀯𑀢𑀺, 𑀬𑁄𑀕𑁂𑀳𑀺 𑀘 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀪𑀯𑀢𑀺, 𑀕𑀦𑁆𑀣𑁂𑀳𑀺 𑀘 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀪𑀯𑀢𑀺, 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘 𑀅𑀦𑀸𑀲𑀯𑁄 𑀪𑀯𑀢𑀺, 𑀑𑀖𑁂𑀳𑀺 𑀘 𑀦𑀺𑀢𑁆𑀣𑀺𑀡𑁆𑀡𑁄 𑀪𑀯𑀢𑀺, 𑀲𑀮𑁆𑀮𑁂𑀳𑀺 𑀘 𑀯𑀺𑀲𑀮𑁆𑀮𑁄 𑀪𑀯𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑁄 𑀘𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀕𑀢𑀺𑀕𑀫𑀦𑁂𑀳𑀺 𑀦 𑀅𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀏𑀯𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀢𑁆𑀢𑀸 𑀧𑀳𑀸𑀦𑀁 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

188\. 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀤𑁂𑀲𑀺𑀢𑀁, 𑀤𑁂𑀲𑀺𑀢𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀭𑀽𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑁂𑀲𑀺𑀢𑀸, 𑀤𑁂𑀲𑀺𑀢𑀁 𑀢𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄 𑀘 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁𑁇 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑁂𑀲𑀺𑀢𑀸, 𑀤𑁂𑀲𑀺𑀢𑀁 𑀢𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁𑁇 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑁂𑀲𑀺𑀢𑀸, 𑀤𑁂𑀲𑀺𑀢𑀁 𑀢𑀢𑁆𑀣 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑁄 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄𑁇 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁, 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺, 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘 𑀲𑀭𑀸𑀕𑀲𑀤𑁄𑀲𑀲𑀫𑁄𑀳𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀧𑀓𑁆𑀔𑁂𑀦 𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀯𑀻𑀢𑀭𑀸𑀕𑀯𑀻𑀢𑀤𑁄𑀲𑀯𑀻𑀢𑀫𑁄𑀳𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇

189\. 𑀏𑀯𑀁 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀮𑀓𑁆𑀔𑀡𑀸 𑀓𑀺𑀘𑁆𑀘𑀢𑁄 𑀘 𑀮𑀓𑁆𑀔𑀡𑀢𑁄 𑀘 𑀲𑀸𑀫𑀜𑁆𑀜𑀢𑁄 𑀘 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑀢𑁄 𑀘, 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀏𑀓𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀯𑀼𑀢𑁆𑀢𑁂 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀯𑀼𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀏𑀓𑀥𑀫𑁆𑀫𑁂”𑀢𑀺𑁇

---
