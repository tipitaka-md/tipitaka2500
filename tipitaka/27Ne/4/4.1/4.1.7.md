# 𑁪𑁇𑁧𑁇𑁭 𑀆𑀯𑀝𑁆𑀝𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀆𑀯𑀝𑁆𑀝𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(29.)

236\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀆𑀯𑀝𑁆𑀝𑁄 𑀳𑀸𑀭𑁄? “𑀏𑀓𑀫𑁆𑀳𑀺 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑁂”𑀢𑀺 𑀅𑀬𑀁𑁇

237\. _“𑀆𑀭𑀫𑁆𑀪𑀣 𑀦𑀺𑀓𑁆𑀓𑀫𑀣,_  
_𑀬𑀼𑀜𑁆𑀚𑀣 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂;_  
_𑀥𑀼𑀦𑀸𑀣 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁,_  
_𑀦𑀍𑀆𑀕𑀸𑀭𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄”𑀢𑀺𑁇_  


238\. “𑀆𑀭𑀫𑁆𑀪𑀣 𑀦𑀺𑀓𑁆𑀓𑀫𑀣𑀸”𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 “𑀬𑀼𑀜𑁆𑀚𑀣 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂”𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 “𑀥𑀼𑀦𑀸𑀣 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁, 𑀦𑀍𑀆𑀕𑀸𑀭𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄”𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 “𑀆𑀭𑀫𑁆𑀪𑀣 𑀦𑀺𑀓𑁆𑀓𑀫𑀣𑀸”𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 “𑀬𑀼𑀜𑁆𑀚𑀣 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂”𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 “𑀥𑀼𑀦𑀸𑀣 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁, 𑀦𑀍𑀆𑀕𑀸𑀭𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄”𑀢𑀺 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀺 𑀤𑁂𑀲𑀦𑀸𑁇

239\. 𑀅𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀬𑁄𑀕𑁂, 𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀸 𑀆𑀭𑀫𑁆𑀪𑁄𑁇

240\. 𑀢𑀢𑁆𑀣 𑀬𑁂 𑀦 𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀧𑀫𑀸𑀤𑀫𑀽𑀮𑀓𑀸 𑀦 𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀧𑀫𑀸𑀤𑁄 𑀤𑀼𑀯𑀺𑀥𑁄 𑀢𑀡𑁆𑀳𑀸𑀫𑀽𑀮𑀓𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀫𑀽𑀮𑀓𑁄 𑀘𑁇 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀫𑀽𑀮𑀓𑁄 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀦𑀺𑀯𑀼𑀢𑁄 𑀜𑁂𑀬𑁆𑀬𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀫𑀽𑀮𑀓𑁄𑁇 𑀬𑁄 𑀢𑀡𑁆𑀳𑀸𑀫𑀽𑀮𑀓𑁄, 𑀲𑁄 𑀢𑀺𑀯𑀺𑀥𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑁄𑀕𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑁄 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑁄𑀕𑀸𑀦𑀁 𑀆𑀭𑀓𑁆𑀔𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀪𑁄𑀕𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀫𑀸𑀤𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀬𑀁 𑀮𑁄𑀓𑁂 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁄 𑀧𑀫𑀸𑀤𑁄 𑀏𑀓𑀯𑀺𑀥𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑀺𑀯𑀺𑀥𑁄 𑀢𑀡𑁆𑀳𑀸𑀬𑁇 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀸𑀫𑀓𑀸𑀬𑁄 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀢𑀡𑁆𑀳𑀸𑀬 𑀭𑀽𑀧𑀓𑀸𑀬𑁄 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼, 𑀭𑀽𑀧𑀻𑀲𑀼 𑀪𑀯𑁂𑀲𑀼 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁, 𑀅𑀭𑀽𑀧𑀻𑀲𑀼 𑀲𑀫𑁆𑀫𑁄𑀳𑁄? 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑀸𑀬𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀸𑀫𑀓𑀸𑀬𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀔𑀦𑁆𑀥𑀸𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀓𑀢𑀫𑁂𑀦 𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀦 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀸, 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀘? 𑀢𑀢𑁆𑀣 𑀢𑀡𑁆𑀳𑀸 𑀤𑁆𑀯𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀤𑁆𑀯𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀳𑀺 𑀬𑁂 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺, 𑀅𑀬𑀁 𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀘 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑁇

241\. 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀢𑀺𑀯𑀺𑀥𑁄 𑀢𑀡𑁆𑀳𑀸𑀫𑀽𑀮𑀓𑁄 𑀧𑀫𑀸𑀤𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑁄𑀕𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑁄𑀕𑀸𑀦𑀁 𑀆𑀭𑀓𑁆𑀔𑀡𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺 𑀧𑀭𑀺𑀪𑁄𑀕𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘, 𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁂𑀦 𑀭𑀓𑁆𑀔𑀡𑀸 𑀧𑀝𑀺𑀲𑀁𑀳𑀭𑀡𑀸, 𑀅𑀬𑀁 𑀲𑀫𑀣𑁄𑁇

242\. 𑀲𑁄 𑀓𑀣𑀁 𑀪𑀯𑀢𑀺? 𑀬𑀤𑀸 𑀚𑀸𑀦𑀸𑀢𑀺 𑀓𑀸𑀫𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀢𑁄 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀢𑁄 𑀑𑀓𑀸𑀭𑀜𑁆𑀘 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀯𑁄𑀤𑀸𑀦𑀜𑁆𑀘 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀁𑁇 𑀢𑀢𑁆𑀣 𑀬𑀸 𑀯𑀻𑀫𑀁𑀲𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀅𑀬𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘𑁇 𑀇𑀫𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀪𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺 𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘, 𑀇𑀫𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀧𑀳𑀻𑀦𑁂𑀲𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄, 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑁄, 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀓𑀸𑀦𑀺 𑀘 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀘, 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘 𑀫𑀕𑁆𑀕𑁄𑁇 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀆𑀭𑀫𑁆𑀪𑀣 𑀦𑀺𑀓𑁆𑀓𑀫𑀣𑀸”𑀢𑀺𑁇

243\. _“𑀬𑀣𑀸𑀧𑀺 𑀫𑀽𑀮𑁂 𑀅𑀦𑀼𑀧𑀤𑁆𑀤𑀯𑁂 𑀤𑀍𑀳𑁂,_  
_𑀙𑀺𑀦𑁆𑀦𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑁄 𑀧𑀼𑀦𑀭𑁂𑀯 𑀭𑀽𑀳𑀢𑀺;_  
_𑀏𑀯𑀫𑁆𑀧𑀺 𑀢𑀡𑁆𑀳𑀸𑀦𑀼𑀲𑀬𑁂 𑀅𑀦𑀽𑀳𑀢𑁂,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀻 𑀤𑀼𑀓𑁆𑀔𑀫𑀺𑀤𑀁 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁”𑁇_  


244\. 𑀅𑀬𑀁 𑀢𑀡𑁆𑀳𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀢𑀫𑀲𑁆𑀲𑀸 𑀢𑀡𑁆𑀳𑀸𑀬? 𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑀬𑁇 𑀬𑁄 𑀏𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀬𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑀺 𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀓𑀺𑀮𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘𑁇 𑀢𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 𑀢𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀳𑀺 𑀬𑁂 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 𑀅𑀬𑀁 𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀘 𑀧𑀳𑀸𑀦𑀸𑀬 𑀘 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑁇

245\. 𑀬𑁂𑀦 𑀢𑀡𑁆𑀳𑀸𑀦𑀼𑀲𑀬𑀁 𑀲𑀫𑀽𑀳𑀦𑀢𑀺, 𑀅𑀬𑀁 𑀲𑀫𑀣𑁄𑁇 𑀬𑁂𑀦 𑀢𑀡𑁆𑀳𑀸𑀦𑀼𑀲𑀬𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀁 𑀯𑀸𑀭𑀬𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘𑁇 𑀢𑀢𑁆𑀣 𑀲𑀫𑀣𑀲𑁆𑀲 𑀨𑀮𑀁 𑀭𑀸𑀕𑀯𑀺𑀭𑀸𑀕𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀨𑀮𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀭𑀸𑀕𑀸 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇 𑀇𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀓𑀸𑀦𑀺 𑀘 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀘, 𑀲𑀫𑀣𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘 𑀫𑀕𑁆𑀕𑁄, 𑀤𑁆𑀯𑁂 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑁄 𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀬𑀣𑀸𑀧𑀺 𑀫𑀽𑀮𑁂”𑀢𑀺𑁇

246\. _“𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀲𑁆𑀲 𑀅𑀓𑀭𑀡𑀁,_  
_𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀲𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑀦𑀁,_  
_𑀏𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


247\. 𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀁 𑀦𑀸𑀫 𑀢𑀻𑀡𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀢𑁂 𑀤𑀲 𑀅𑀓𑀼𑀲𑀮𑀓𑀫𑁆𑀫𑀧𑀣𑀸 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀧𑀺𑀲𑀼𑀡𑀸 𑀯𑀸𑀘𑀸 𑀨𑀭𑀼𑀲𑀸 𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑁄 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀘𑁂𑀢𑀦𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀜𑁆𑀘𑁇 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀘 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄 𑀬𑀸 𑀘 𑀧𑀺𑀲𑀼𑀡𑀸 𑀯𑀸𑀘𑀸 𑀬𑀸 𑀘 𑀨𑀭𑀼𑀲𑀸 𑀯𑀸𑀘𑀸, 𑀇𑀤𑀁 𑀤𑁄𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀬𑀜𑁆𑀘 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀬𑁄 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑁄 𑀬𑁄 𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄, 𑀇𑀤𑀁 𑀮𑁄𑀪𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀬𑁄 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑁄, 𑀇𑀤𑀁 𑀫𑁄𑀳𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀓𑀸𑀭𑀡𑀸𑀦𑀺 𑀘𑁂𑀢𑀦𑀸𑀓𑀫𑁆𑀫𑀁𑁇 𑀬𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸, 𑀅𑀬𑀁 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁𑁇 𑀬𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄, 𑀅𑀬𑀁 𑀤𑁄𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁𑁇 𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀬𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀕𑁆𑀕𑁄𑁇 𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀓𑀸𑀭𑀡𑀸𑀦𑀺 𑀘𑁂𑀢𑀲𑀺𑀓𑀓𑀫𑁆𑀫𑀁𑁇 𑀢𑁂𑀦𑀸𑀳 “𑀘𑁂𑀢𑀦𑀸𑀓𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇

248\. 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁 𑀧𑀬𑁄𑀕𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑀁 𑀅𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀪𑀬𑀸 𑀫𑁄𑀳𑀸𑁇 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀙𑀦𑁆𑀤𑀸 𑀅𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀇𑀤𑀁 𑀮𑁄𑀪𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀬𑀁 𑀤𑁄𑀲𑀸 𑀅𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀇𑀤𑀁 𑀤𑁄𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀬𑀁 𑀪𑀬𑀸 𑀘 𑀫𑁄𑀳𑀸 𑀘 𑀅𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀇𑀤𑀁 𑀫𑁄𑀳𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀢𑀢𑁆𑀣 𑀮𑁄𑀪𑁄 𑀅𑀲𑀼𑀪𑀸𑀬 𑀧𑀳𑀻𑀬𑀢𑀺𑁇 𑀤𑁄𑀲𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑁇 𑀫𑁄𑀳𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁇 𑀢𑀣𑀸 𑀮𑁄𑀪𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀧𑀳𑀻𑀬𑀢𑀺𑁇 𑀤𑁄𑀲𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘 𑀓𑀭𑀼𑀡𑀸𑀬 𑀘𑁇 𑀫𑁄𑀳𑁄 𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀧𑀳𑀸𑀦𑀁 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀲𑁆𑀲 𑀅𑀓𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

(31.)

249\. 𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀁 𑀦𑀸𑀫 𑀅𑀝𑁆𑀞 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸𑀦𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀁𑁇 𑀇𑀫𑁂𑀲𑀁 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸𑀦𑀁 𑀬𑀸 𑀅𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀓𑀭𑀡𑀁 𑀅𑀦𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑁄, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀲𑁆𑀲 𑀅𑀓𑀭𑀡𑀁𑁇

250\. 𑀅𑀝𑁆𑀞𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑁂𑀲𑀼 𑀧𑀳𑀻𑀦𑁂𑀲𑀼 𑀅𑀝𑁆𑀞 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀬𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀧𑀸𑀤𑀦𑀁, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇 𑀲𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑀦𑀦𑁆𑀢𑀺 𑀅𑀢𑀻𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀓𑀺𑀭𑀺𑀬𑀁 𑀤𑀲𑁆𑀲𑀬𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑀺𑀢𑁂 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑀺𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺, 𑀏𑀯𑀜𑁆𑀳𑀺 𑀪𑀕𑀯𑀸 𑀆𑀳—  “𑀘𑁂𑀢𑁄𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀤𑀼𑀯𑀺𑀥𑀸 𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑀦𑀸 𑀦𑀻𑀯𑀭𑀡𑀧𑁆𑀧𑀳𑀸𑀦𑀜𑁆𑀘 𑀅𑀦𑀼𑀲𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑀦𑀪𑀽𑀫𑀺𑀬𑁄 𑀤𑀲𑁆𑀲𑀦𑀪𑀽𑀫𑀺 𑀘, 𑀪𑀸𑀯𑀦𑀸𑀪𑀽𑀫𑀺 𑀘, 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀧𑀝𑀺𑀯𑁂𑀥𑁂𑀦 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑁂𑀢𑀺, 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀬𑀢𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑁂𑀢𑀺, 𑀅𑀬𑀁 𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀬𑁂𑀦 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑁂𑀢𑀺, 𑀅𑀬𑀁 𑀫𑀕𑁆𑀕𑁄𑁇 𑀬𑀁 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑀺𑀢𑀁, 𑀅𑀬𑀁 𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀲𑁆𑀲 𑀅𑀓𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

251\. _“𑀥𑀫𑁆𑀫𑁄 𑀳𑀯𑁂 𑀭𑀓𑁆𑀔𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀁,_  
_𑀙𑀢𑁆𑀢𑀁 𑀫𑀳𑀦𑁆𑀢𑀁 𑀬𑀣 𑀯𑀲𑁆𑀲𑀓𑀸𑀮𑁂;_  
_𑀏𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑁄 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁂,_  
_𑀦 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻”𑀢𑀺𑁇_  


252\. 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀫 𑀤𑀼𑀯𑀺𑀥𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀯𑀭𑁄 𑀫𑀕𑁆𑀕𑁄 𑀘𑁇 𑀤𑀼𑀕𑁆𑀕𑀢𑀺 𑀦𑀸𑀫 𑀤𑀼𑀯𑀺𑀥𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀅𑀧𑀸𑀬𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀯𑀸 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀲𑀩𑁆𑀩𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀬𑀸 𑀲𑀁𑀯𑀭𑀲𑀻𑀮𑁂 𑀅𑀔𑀡𑁆𑀟𑀓𑀸𑀭𑀺𑀢𑀸, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁄 𑀅𑀧𑀸𑀬𑁂𑀳𑀺 𑀭𑀓𑁆𑀔𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀸 𑀆𑀳—  “𑀤𑁆𑀯𑁂𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀮𑀯𑀢𑁄 𑀕𑀢𑀺𑀬𑁄 𑀤𑁂𑀯𑀸 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘”𑁇 𑀏𑀯𑀜𑁆𑀘 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀦𑀺𑀕𑀫𑁂 𑀅𑀲𑀺𑀩𑀦𑁆𑀥𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀡𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

253\. “𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀘𑁆𑀙𑀸𑀪𑀽𑀫𑀓𑀸 𑀓𑀸𑀫𑀡𑁆𑀟𑀮𑀼𑀓𑀸 𑀲𑁂𑀯𑀸𑀮𑀫𑀸𑀮𑀺𑀓𑀸 𑀉𑀤𑀓𑁄𑀭𑁄𑀳𑀓𑀸 𑀅𑀕𑁆𑀕𑀺𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸, 𑀢𑁂 𑀫𑀢𑀁 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀁 𑀉𑀬𑁆𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀸𑀫, 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀸𑀫, 𑀲𑀕𑁆𑀕𑀁 𑀦𑀸𑀫 𑀑𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀧𑀳𑁄𑀢𑀺 𑀢𑀣𑀸 𑀓𑀸𑀢𑀼𑀁, 𑀬𑀣𑀸 𑀲𑀩𑁆𑀩𑁄 𑀮𑁄𑀓𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

254\. “𑀢𑁂𑀦 𑀳𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬, 𑀢𑀣𑀸 𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻𑀢𑀺𑁇

255\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑁄 𑀨𑀭𑀼𑀲𑀯𑀸𑀘𑁄 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀻 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄, 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬 ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

256\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑀺𑀁 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀁 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀉𑀤𑀓𑀭𑀳𑀤𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬, 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬 ‘𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚, 𑀪𑁄, 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑁂, 𑀉𑀧𑁆𑀮𑀯 𑀪𑁄 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑁂, 𑀣𑀮𑀫𑀼𑀧𑁆𑀮𑀯, 𑀪𑁄 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑁂’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑀸 𑀫𑀳𑀢𑀻 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀸 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸 𑀉𑀧𑁆𑀮𑀯𑁂𑀬𑁆𑀬 𑀯𑀸 𑀣𑀮𑀁 𑀯𑀸 𑀉𑀧𑁆𑀮𑀯𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄, 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬 ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇

257\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀇𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄, 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬 ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

258\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀕𑀸𑀫𑀡𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀧𑁆𑀧𑀺𑀓𑀼𑀫𑁆𑀪𑀁 𑀯𑀸 𑀢𑁂𑀮𑀓𑀼𑀫𑁆𑀪𑀁 𑀯𑀸 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀉𑀤𑀓𑀭𑀳𑀤𑁂 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑁇 𑀢𑀢𑁆𑀭 𑀬𑀸𑀲𑁆𑀲 𑀲𑀓𑁆𑀔𑀭𑀓𑀞𑀮𑀁, 𑀲𑀸 𑀅𑀥𑁄𑀕𑀸𑀫𑀻 𑀅𑀲𑁆𑀲𑁇 𑀬𑀜𑁆𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀢𑀢𑁆𑀭 𑀲𑀧𑁆𑀧𑀺 𑀯𑀸 𑀢𑁂𑀮𑀁 𑀯𑀸, 𑀢𑀁 𑀉𑀤𑁆𑀥𑀗𑁆𑀕𑀸𑀫𑀺 𑀅𑀲𑁆𑀲𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬 ‘𑀑𑀲𑀻𑀤, 𑀪𑁄 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮, 𑀲𑀁𑀲𑀻𑀤, 𑀪𑁄 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮, 𑀅𑀥𑁄 𑀕𑀘𑁆𑀙, 𑀪𑁄 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮𑀸’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺 𑀕𑀸𑀫𑀡𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮𑀁 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀬𑀸𑀘𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀣𑁄𑀫𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 ‘𑀑𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀲𑀁𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀅𑀥𑁄 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’”𑀢𑀺𑁇 “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

259\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀕𑀸𑀫𑀡𑀺, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄, 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀦𑀁 𑀫𑀳𑀸 𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑁂𑀬𑁆𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬 ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀇𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁄 𑀅𑀧𑀸𑀬𑁂𑀳𑀺 𑀭𑀓𑁆𑀔𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀬𑀸 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀢𑀺𑀓𑁆𑀔𑀢𑀸 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀢𑀸, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁄 𑀲𑀩𑁆𑀩𑀸𑀳𑀺 𑀉𑀧𑀧𑀢𑁆𑀢𑀻𑀳𑀺 𑀭𑀓𑁆𑀔𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀸 𑀆𑀳—

260\. _“𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄;_  
_𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄,_  
_𑀜𑀢𑁆𑀯𑀸𑀦 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁;_  
_𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸𑀪𑀺𑀪𑀽 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀩𑁆𑀩𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄 𑀚𑀳𑁂”𑀢𑀺𑁇_  


(32.)

261\. 𑀢𑀢𑁆𑀣 𑀤𑀼𑀕𑁆𑀕𑀢𑀻𑀦𑀁 𑀳𑁂𑀢𑀼 𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘, 𑀢𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺, 𑀢𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀳𑀺 𑀬𑁂 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺, 𑀅𑀬𑀁 𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀘 𑀧𑀳𑀸𑀦𑀸𑀬 𑀘 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑁇 𑀢𑀢𑁆𑀣 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀭𑀽𑀧𑀻𑀦𑀺 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀭𑀽𑀧𑀻𑀦𑀺 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀲𑀫𑀸𑀥𑀺𑀁 𑀪𑀸𑀯𑀬𑀢𑀺, 𑀢𑀡𑁆𑀳𑀜𑁆𑀘 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁 𑀪𑀸𑀯𑀬𑀢𑀺, 𑀅𑀯𑀺𑀚𑁆𑀚𑀜𑁆𑀘 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀦 𑀤𑁆𑀯𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀦 𑀤𑁆𑀯𑁂 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀘𑀢𑀽𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀲𑀼 𑀧𑀳𑀻𑀦𑁂𑀲𑀼 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑁇

262\. 𑀢𑀢𑁆𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀨𑀮𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀅𑀕𑁆𑀕𑀨𑀮𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀨𑀮𑀸𑀦𑀺𑁇 𑀇𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀓𑀸𑀦𑀺 𑀘 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀘𑁇 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀨𑀮𑀸𑀦𑀺 𑀘 𑀢𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀘 𑀅𑀲𑀗𑁆𑀔𑀢𑀸𑀥𑀸𑀢𑀼 𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀥𑀫𑁆𑀫𑁄 𑀳𑀯𑁂 𑀭𑀓𑁆𑀔𑀢𑀻”𑀢𑀺𑁇

263\. 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀧𑀝𑀺𑀯𑁂𑀥𑁂𑀦 𑀭𑀓𑁆𑀔𑀢𑀺, 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀬𑀢𑁄 𑀭𑀓𑁆𑀔𑀢𑀺, 𑀅𑀬𑀁 𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀬𑁂𑀦 𑀭𑀓𑁆𑀔𑀢𑀺, 𑀅𑀬𑀁 𑀫𑀕𑁆𑀕𑁄𑁇 𑀬𑀁 𑀭𑀓𑁆𑀔𑀢𑀺, 𑀅𑀬𑀁 𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀏𑀓𑀫𑁆𑀳𑀺 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑁂”𑀢𑀺𑁇
