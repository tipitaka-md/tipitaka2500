# 𑁪𑁇𑁧𑁇𑁮 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀯𑀺𑀪𑀗𑁆𑀕

(33.)

265. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑁄? “𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀪𑀽𑀫𑀺𑀜𑁆𑀘𑀸”𑀢𑀺𑁇

266. 𑀤𑁆𑀯𑁂 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀸𑀲𑀦𑀸𑀪𑀸𑀕𑀺𑀬𑀜𑁆𑀘 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀬𑀜𑁆𑀘𑁇 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀧𑀤𑀸 𑀧𑀼𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀸 𑀘 𑀨𑀮𑀪𑀸𑀕𑀺𑀬𑀸 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀲𑀻𑀮𑀸𑀦𑀺 𑀲𑀁𑀯𑀭𑀲𑀻𑀮𑀜𑁆𑀘 𑀧𑀳𑀸𑀦𑀲𑀻𑀮𑀜𑁆𑀘, 𑀢𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀯𑀸𑀲𑀦𑀸𑀪𑀸𑀕𑀺𑀬𑀁 𑀲𑀼𑀢𑁆𑀢𑀁 𑀧𑀼𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀤𑁂𑀲𑀬𑀢𑀺, 𑀲𑁄 𑀲𑀁𑀯𑀭𑀲𑀻𑀮𑁂 𑀞𑀺𑀢𑁄 𑀢𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀪𑀯𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀬𑀁 𑀲𑀼𑀢𑁆𑀢𑀁 𑀨𑀮𑀪𑀸𑀕𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀤𑁂𑀲𑀬𑀢𑀺, 𑀲𑁄 𑀧𑀳𑀸𑀦𑀲𑀻𑀮𑁂 𑀞𑀺𑀢𑁄 𑀢𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀪𑀯𑀢𑀺𑁇

267. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀸𑀲𑀦𑀸𑀪𑀸𑀕𑀺𑀬𑀁 𑀲𑀼𑀢𑁆𑀢𑀁? 𑀯𑀸𑀲𑀦𑀸𑀪𑀸𑀕𑀺𑀬𑀁 𑀦𑀸𑀫 𑀲𑀼𑀢𑁆𑀢𑀁 𑀤𑀸𑀦𑀓𑀣𑀸 𑀲𑀻𑀮𑀓𑀣𑀸 𑀲𑀕𑁆𑀕𑀓𑀣𑀸 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑁄 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑁄𑀢𑀺𑁇

268. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀬𑀁 𑀲𑀼𑀢𑁆𑀢𑀁? 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀬𑀁 𑀦𑀸𑀫 𑀲𑀼𑀢𑁆𑀢𑀁 𑀬𑀸 𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀧𑁆𑀧𑀓𑀸𑀲𑀦𑀸, 𑀯𑀸𑀲𑀦𑀸𑀪𑀸𑀕𑀺𑀬𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀦𑀢𑁆𑀣𑀺 𑀧𑀚𑀸𑀦𑀦𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀨𑀮𑀁𑁇 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀬𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀅𑀢𑁆𑀣𑀺 𑀧𑀚𑀸𑀦𑀦𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀨𑀮𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑁂𑀲𑀦𑀸𑀬 𑀨𑀮𑁂𑀦 𑀲𑀻𑀮𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀲𑀩𑁆𑀩𑀢𑁄 𑀯𑀺𑀘𑀬𑁂𑀦 𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀬𑀼𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑁂𑀦 𑀬𑁄𑀚𑀬𑀺𑀢𑀩𑁆𑀩𑀸 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀜𑀸𑀡𑀲𑁆𑀲 𑀪𑀽𑀫𑀺𑁇

(34.)

269. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀦𑀸𑀫𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀯𑀢𑁆𑀣𑀼𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀘𑁇 𑀬𑀁 𑀯𑀸 𑀧𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀏𑀯𑀁 𑀚𑀸𑀢𑀺𑀬𑀁, 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀦𑀺𑀬𑀢𑀸𑀦𑀜𑁆𑀘 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀧𑁆𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀘 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀮𑁄𑀓𑀺𑀬𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑀩𑁆𑀩𑀸 𑀲𑀸 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁂𑀳𑀺 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀳𑀺 𑀥𑀫𑁆𑀫𑀸 𑀏𑀯𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀭𑀁 𑀧𑀭𑀁 𑀲𑀓𑀁 𑀲𑀓𑀁 𑀯𑀺𑀲𑀬𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀬𑁄𑀧𑀺 𑀇𑀫𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀦 𑀲𑁄 𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀉𑀧𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇

270. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀬𑀸𑀯 𑀤𑁂𑀲𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀕𑀯𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀸 𑀲𑁂𑀓𑁆𑀔𑀸𑀲𑁂𑀓𑁆𑀔𑀸 𑀪𑀩𑁆𑀩𑀸𑀪𑀩𑁆𑀩𑀸𑀢𑀺, 𑀅𑀝𑁆𑀞𑀫𑀓𑀲𑁆𑀲 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀥𑀫𑁆𑀫𑀢𑀸 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀅𑀝𑁆𑀞𑀫𑀓𑀲𑁆𑀲 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀘 𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀥𑀫𑁆𑀫𑀢𑀸 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑁂𑀓𑁆𑀔𑀸𑀦𑀁 𑀦𑀸𑀫𑀁 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀢𑀸 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀓𑀸𑀦𑀁 𑀦𑀸𑀫𑀁 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀢𑀸 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑁂𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑁂𑀓𑁆𑀔𑀲𑀻𑀮𑀁 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀢𑀸 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 𑀏𑀯𑀁 𑀯𑀺𑀲𑁂𑀲𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑀸 𑀳𑀻𑀦𑀼𑀓𑁆𑀓𑀝𑁆𑀞𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀕𑀯𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇

271. 𑀤𑀲𑁆𑀲𑀦𑀪𑀽𑀫𑀺 𑀦𑀺𑀬𑀸𑀫𑀸𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀪𑀸𑀯𑀦𑀸𑀪𑀽𑀫𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀓𑀸𑀦𑀁 𑀨𑀮𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀼𑀔𑀸 𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀤𑀸𑀦𑀫𑀬𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀼 𑀧𑀭𑀢𑁄 𑀖𑁄𑀲𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀻𑀮𑀫𑀬𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀼 𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀼 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀤𑀸𑀦𑀫𑀬𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀼 𑀧𑀭𑀢𑁄 𑀘 𑀖𑁄𑀲𑀲𑁆𑀲 𑀲𑀼𑀢𑀫𑀬𑀺𑀬𑀸 𑀘 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀲𑀻𑀮𑀫𑀬𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀼 𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀺𑀬𑀸 𑀘 𑀧𑀜𑁆𑀜𑀸𑀬 𑀬𑁄𑀦𑀺𑀲𑁄 𑀘 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀯𑀢𑁆𑀣𑀼 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀺𑀬𑀸 𑀘 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀧𑀢𑀺𑀭𑀽𑀧𑀤𑁂𑀲𑀯𑀸𑀲𑁄 𑀯𑀺𑀯𑁂𑀓𑀲𑁆𑀲 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀘 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀢𑀺𑀡𑁆𑀡𑀜𑁆𑀘 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑀸𑀦𑀁 𑀲𑀫𑀣𑀲𑁆𑀲 𑀘 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀫𑀸𑀧𑀡𑀺𑀥𑀸𑀦𑀁 𑀳𑀺𑀭𑀺𑀬𑀸 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀘 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀅𑀓𑀼𑀲𑀮𑀧𑀭𑀺𑀘𑁆𑀘𑀸𑀕𑁄 𑀓𑀼𑀲𑀮𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀘 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲 𑀘 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀥𑀫𑁆𑀫𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑀢𑀸 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀭𑁄𑀧𑀦𑀸𑀬 𑀘 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀁𑀖𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀢𑀸 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀢𑁆𑀣𑀼𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀧𑀲𑀸𑀤𑀸𑀬 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁, 𑀅𑀧𑁆𑀧𑀝𑀺𑀳𑀢𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀢𑀸 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬 𑀧𑁂𑀲𑀮𑀸𑀦𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀢𑁂𑀦𑀸𑀳 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇
