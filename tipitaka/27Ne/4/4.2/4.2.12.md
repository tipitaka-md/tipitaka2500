# 𑁪𑁇𑁨𑁇𑁧𑁨 𑀑𑀢𑀭𑀡𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

* 𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀑𑀢𑀭𑀡𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(74.)

524\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀑𑀢𑀭𑀡𑁄 𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

525\. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑁇 “𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄”𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀕𑀳𑀺𑀢𑀸𑀬 𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀅𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺 𑀑𑀢𑀭𑀡𑀸𑁇

526\. 𑀢𑀸𑀦𑀺𑀬𑁂𑀯 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀯𑀺𑀚𑁆𑀚𑀸, 𑀯𑀺𑀚𑁆𑀚𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄, 𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀁, 𑀅𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀦 𑀑𑀢𑀭𑀡𑀸𑁇

527\. 𑀢𑀸𑀦𑀺𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺—  𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀅𑀬𑀁 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀑𑀢𑀭𑀡𑀸𑁇

528\. 𑀢𑀸𑀦𑀺 𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀺𑁇 𑀬𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀸𑀲𑀯𑀸 𑀦𑁄 𑀘 𑀪𑀯𑀗𑁆𑀕𑀸, 𑀢𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀅𑀬𑀁 𑀥𑀸𑀢𑀽𑀳𑀺 𑀑𑀢𑀭𑀡𑀸𑁇

529\. 𑀲𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀬𑀁 𑀆𑀬𑀢𑀦𑀁 𑀅𑀦𑀸𑀲𑀯𑀁 𑀦𑁄 𑀘 𑀪𑀯𑀗𑁆𑀕𑀁, 𑀅𑀬𑀁 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀑𑀢𑀭𑀡𑀸𑁇

---

530\. 𑀦𑀺𑀬𑀼𑀢𑁆𑀢𑁄 𑀑𑀢𑀭𑀡𑁄 𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄𑁇
