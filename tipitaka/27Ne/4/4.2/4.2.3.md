# 𑁪𑁇𑁨𑁇𑁩 𑀬𑀼𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀬𑀼𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(65.)

496\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀬𑀼𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

497\. _“𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄;_  
_𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄,_  
_𑀜𑀢𑁆𑀯𑀸𑀦 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁;_  
_𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸𑀪𑀺𑀪𑀽 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀩𑁆𑀩𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄 𑀚𑀳𑁂”𑀢𑀺𑁇_  


498\. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀬𑀼𑀚𑁆𑀚𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀬𑀼𑀚𑁆𑀚𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀬𑀼𑀚𑁆𑀚𑀢𑀺, 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀦𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄 𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀬𑀼𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄 𑀚𑀳𑀦𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀪𑀬𑀸𑀦𑀺 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀬𑀼𑀚𑁆𑀚𑀢𑀻𑀢𑀺𑁇

---
