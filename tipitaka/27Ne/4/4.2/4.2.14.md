# 𑁪𑁇𑁨𑁇𑁧𑁪 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

* 𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(76.)

534\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁄 𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

535\. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀦𑀸𑀫 𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑁂 𑀜𑀸𑀡𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀜𑀸𑀡𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀜𑀸𑀡𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀜𑀸𑀡𑀁 𑀫𑀕𑁆𑀕𑁂 𑀜𑀸𑀡𑀁 𑀳𑁂𑀢𑀼𑀫𑁆𑀳𑀺 𑀜𑀸𑀡𑀁 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀘𑁆𑀘𑀬𑁂 𑀜𑀸𑀡𑀁 𑀧𑀘𑁆𑀘𑀬𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁, 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑁄 𑀲𑀫𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀲𑀘𑁆𑀘𑀸𑀕𑀫𑀦𑀁, 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇 𑀜𑀢𑁆𑀯𑀸𑀦 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀦𑁆𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸, 𑀉𑀤𑀬𑁂𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀪𑀯𑀢𑀺𑁇 𑀯𑀬𑁂𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄, 𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸𑀪𑀺𑀪𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸, 𑀣𑀺𑀦𑀁 𑀦𑀸𑀫 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀮𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀢𑀸, 𑀫𑀺𑀤𑁆𑀥𑀁 𑀦𑀸𑀫 𑀬𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀮𑀻𑀦𑀢𑁆𑀢𑀁, 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇 𑀲𑀩𑁆𑀩𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄 𑀚𑀳𑁂𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀢𑀸, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀅𑀧𑀸𑀬𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀯𑀸 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀲𑀩𑁆𑀩𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀤𑀼𑀕𑁆𑀕𑀢𑀺, 𑀅𑀬𑀁 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑁇
