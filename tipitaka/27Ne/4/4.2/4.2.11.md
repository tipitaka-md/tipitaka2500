# 𑁪𑁇𑁨𑁇𑁧𑁧 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(73.)

521\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

522\. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀢𑀺𑀬𑀸𑁇 “𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀲𑀫𑀣𑀲𑁆𑀲𑁇 “𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄, 𑀜𑀢𑁆𑀯𑀸𑀦 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀦𑁆”𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 “𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸𑀪𑀺𑀪𑀽 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀅𑀦𑀯𑀲𑁂𑀲𑀧𑁆𑀧𑀳𑀸𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, “𑀲𑀩𑁆𑀩𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀬𑁄 𑀚𑀳𑁂”𑀢𑀺 𑀪𑀸𑀯𑀦𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑁆𑀲𑁇

---
