# 𑁪𑁇𑁨𑁇𑁯 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

* 𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(71.)

515\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁄 𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

516\. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 𑀲𑀫𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑁄 𑀨𑀮𑀁, 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀧𑀝𑀺𑀧𑀓𑁆𑀔𑁂𑀦𑁇

---
