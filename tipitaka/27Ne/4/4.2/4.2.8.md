# 𑁪𑁇𑁨𑁇𑁮 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(70.)

512. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

513. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁄 𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁂𑀦 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁄 𑀅𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁂𑀦 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇

---

514. 𑀦𑀺𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄𑁇
