# 𑁪𑁇𑁨𑁇𑁧 𑀤𑁂𑀲𑀦𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

* 𑀲𑀫𑁆𑀧𑀸𑀢

* 𑀤𑁂𑀲𑀦𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(52.)

443\. _“𑀲𑁄𑀍𑀅𑀲 𑀳𑀸𑀭𑀸 𑀧𑀞𑀫𑀁,_  
_𑀤𑀺𑀲𑀮𑁄𑀘𑀦𑀢𑁄 𑀤𑀺𑀲𑀸 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸;_  
_𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀬 𑀅𑀗𑁆𑀓𑀼𑀲𑁂𑀦 𑀳𑀺,_  
_𑀦𑀬𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑁂 𑀲𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇_  


444\. 𑀯𑀼𑀢𑁆𑀢𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑁄 𑀓𑀼𑀳𑀺𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄? 𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁂𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑁂𑀲𑀦𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

445\. _“𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀳𑀢𑁂𑀦 𑀘;_  
_𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸𑀪𑀺𑀪𑀽𑀢𑁂𑀦,_  
_𑀯𑀲𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇_  


446\. **𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦𑀸**𑀢𑀺 𑀓𑀺𑀁 𑀤𑁂𑀲𑀬𑀢𑀺, 𑀧𑀫𑀸𑀤𑀁 𑀢𑀁 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀧𑀤𑀁𑁇 **𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀳𑀢𑁂𑀦 𑀘𑀸**𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀳𑀢𑀁 𑀦𑀸𑀫 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀬𑀤𑀸 “𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀦𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄𑁇 𑀲𑁄 𑀧𑀦 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄 𑀓𑀺𑀁𑀮𑀓𑁆𑀔𑀡𑁄? 𑀯𑀺𑀧𑀭𑀻𑀢𑀕𑁆𑀕𑀸𑀳𑀮𑀓𑁆𑀔𑀡𑁄 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄𑁇 𑀲𑁄 𑀓𑀺𑀁 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀬𑀢𑀺? 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑁂 𑀲𑀜𑁆𑀜𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀫𑀺𑀢𑀺𑁇 𑀲𑁄 𑀓𑀼𑀳𑀺𑀁 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀬𑀢𑀺? 𑀘𑀢𑀽𑀲𑀼 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀯𑀢𑁆𑀣𑀽𑀲𑀼, 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁, 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀏𑀯𑀁 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇

447\. 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀁 𑀧𑀞𑀫𑀁 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀯𑀢𑁆𑀣𑀼 “𑀅𑀲𑀼𑀪𑁂 𑀲𑀼𑀪𑀦𑁆”𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀯𑀢𑁆𑀣𑀼 “𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇 𑀲𑀜𑁆𑀜𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘 𑀢𑀢𑀺𑀬𑀁 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀯𑀢𑁆𑀣𑀼 “𑀅𑀦𑀢𑁆𑀢𑀦𑀺 𑀅𑀢𑁆𑀢𑀸”𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀯𑀢𑁆𑀣𑀼 “𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀦𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀸—  𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘𑁇 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀯𑀼𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂𑀳𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀻𑀬𑀢𑀺 “𑀅𑀲𑀼𑀪𑁂 𑀲𑀼𑀪𑀦𑁆”𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀯𑀼𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂𑀳𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀻𑀬𑀢𑀺 “𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀦𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺 “𑀅𑀦𑀢𑁆𑀢𑀦𑀺 𑀅𑀢𑁆𑀢𑀸”𑀢𑀺𑁇

448\. 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄, 𑀲𑁄 𑀅𑀢𑀻𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀢𑀻𑀢𑀁 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀅𑀢𑀻𑀢𑀁 𑀲𑀜𑁆𑀜𑀁, 𑀅𑀢𑀻𑀢𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀅𑀢𑀻𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁄, 𑀲𑁄 𑀅𑀦𑀸𑀕𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀅𑀦𑀸𑀕𑀢𑀁 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀅𑀦𑀸𑀕𑀢𑀁 𑀲𑀜𑁆𑀜𑀁, 𑀅𑀦𑀸𑀕𑀢𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑁂, 𑀅𑀦𑀸𑀕𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀸—  𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘𑁇 𑀢𑀸𑀳𑀺 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀲𑀦𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀓𑀺𑀁 𑀦𑀺𑀭𑀬𑀁 𑀲𑀓𑀺𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀲𑀓𑀺𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁 𑀲𑀓𑀺𑀁 𑀅𑀲𑀼𑀭𑀓𑀸𑀬𑀁 𑀲𑀓𑀺𑀁 𑀤𑁂𑀯𑁂 𑀲𑀓𑀺𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑁇

449\. **𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸𑀪𑀺𑀪𑀽𑀢𑁂𑀦𑀸**𑀢𑀺𑁇 𑀣𑀺𑀦𑀁 𑀦𑀸𑀫 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀮𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀢𑀸; 𑀫𑀺𑀤𑁆𑀥𑀁 𑀦𑀸𑀫 𑀬𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀮𑀻𑀦𑀢𑁆𑀢𑀁𑁇 **𑀯𑀲𑀁 𑀫𑀸𑀭𑀲𑁆𑀲 𑀕𑀘𑁆𑀙𑀢𑀻**𑀢𑀺 𑀓𑀺𑀮𑁂𑀲𑀫𑀸𑀭𑀲𑁆𑀲 𑀘 𑀲𑀢𑁆𑀢𑀫𑀸𑀭𑀲𑁆𑀲 𑀘 𑀯𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑁄 𑀳𑀺 𑀦𑀺𑀯𑀼𑀢𑁄 𑀲𑀁𑀲𑀸𑀭𑀸𑀪𑀺𑀫𑀼𑀔𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑀸 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀘𑁇 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀘 𑀧𑀳𑀸𑀦𑀸𑀬 𑀘 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬𑁇 𑀬𑁂𑀦 𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀬𑁂𑀦 𑀘 𑀧𑀚𑀳𑀢𑀺, 𑀅𑀬𑀁 𑀫𑀕𑁆𑀕𑁄𑁇 𑀬𑀁 𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀘 𑀧𑀳𑀸𑀦𑀁, 𑀅𑀬𑀁 𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇 𑀢𑁂𑀦𑀸𑀳 𑀪𑀕𑀯𑀸 “𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦𑀸”𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 “𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀤𑀻𑀦𑀯𑀢𑀸”𑀢𑀺𑁇
