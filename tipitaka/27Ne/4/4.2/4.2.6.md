# 𑁪𑁇𑁨𑁇𑁬 𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(68.)

506\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑁄 𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄𑁇

507\. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑀁 𑀧𑀭𑀺𑀧𑀸𑀮𑀻𑀬𑀢𑀻𑀢𑀺 𑀏𑀲𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑁇 𑀇𑀥 𑀪𑀕𑀯𑀢𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄? 𑀬𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑀻𑀳𑀺 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀅𑀬𑀁 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇 𑀓𑁄𑀓𑀸𑀮𑀺𑀓𑁄 𑀳𑀺 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀲𑀼 𑀣𑁂𑀭𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀤𑁄𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀧𑀤𑀼𑀫𑀦𑀺𑀭𑀬𑁂 𑀉𑀧𑀧𑀦𑁆𑀦𑁄𑁇 𑀪𑀕𑀯𑀸 𑀘 𑀲𑀢𑀺𑀆𑀭𑀓𑁆𑀔𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀲𑀼𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀯𑀼𑀢𑁆𑀢𑀁—  “𑀲𑀢𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀭𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
