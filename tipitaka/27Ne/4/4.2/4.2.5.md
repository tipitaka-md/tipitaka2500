# 𑁪𑁇𑁨𑁇𑁫 𑀮𑀓𑁆𑀔𑀡𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀧𑀝𑀺𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀲𑀫𑁆𑀧𑀸𑀢

### 𑀮𑀓𑁆𑀔𑀡𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢

(67.)

503\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑀓𑁆𑀔𑀡𑁄 𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀸𑀢𑁄?

504\. “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀕𑀸𑀣𑀸𑁇 “𑀢𑀲𑁆𑀫𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀇𑀤𑀁 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀕𑀳𑀺𑀢𑁂 𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺𑁇 “𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄”𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀕𑀳𑀺𑀢𑀸𑀬 𑀕𑀳𑀺𑀢𑁄 𑀪𑀯𑀢𑀺 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀧𑀪𑀯𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀪𑀯𑀢𑀺𑁇

---
