# 𑁩𑁇𑁨 𑀦𑀬𑀲𑀗𑁆𑀔𑁂𑀧

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

* 𑀦𑀬𑀲𑀗𑁆𑀔𑁂𑀧

41\. _𑀢𑀡𑁆𑀳𑀜𑁆𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀘,_  
_𑀲𑀫𑀣𑁂𑀦 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀬𑁄 𑀦𑁂𑀢𑀺;_  
_𑀲𑀘𑁆𑀘𑁂𑀳𑀺 𑀬𑁄𑀚𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀬𑀁 𑀦𑀬𑁄 **𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀯𑀝𑁆𑀝𑁄**𑁇_  


42\. _𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑁂 𑀲𑀫𑀽𑀮𑁂𑀳𑀺,_  
_𑀦𑁂𑀢𑀺 𑀓𑀼𑀲𑀮𑁂 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀳𑀺;_  
_𑀪𑀽𑀢𑀁 𑀢𑀣𑀁 𑀅𑀯𑀺𑀢𑀣𑀁,_  
_**𑀢𑀺𑀧𑀼𑀓𑁆𑀔𑀮𑀁** 𑀢𑀁 𑀦𑀬𑀁 𑀆𑀳𑀼𑁇_  


43\. _𑀬𑁄 𑀦𑁂𑀢𑀺 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑁂𑀳𑀺,_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂;_  
_𑀏𑀢𑀁 𑀦𑀬𑀁 𑀦𑀬𑀯𑀺𑀤𑀽,_  
_**𑀲𑀻𑀳𑀯𑀺𑀓𑁆𑀓𑀻𑀍𑀇𑀢𑀁** 𑀆𑀳𑀼𑁇_  


44\. _𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀲𑀼 𑀳𑀺 𑀬𑁂,_  
_𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁 𑀯𑀼𑀢𑁆𑀢𑀸;_  
_𑀫𑀦𑀲𑀸 𑀯𑁄𑀮𑁄𑀓𑀬𑀢𑁂,_  
_𑀢𑀁 𑀔𑀼 **𑀤𑀺𑀲𑀸𑀮𑁄𑀘𑀦𑀁** 𑀆𑀳𑀼𑁇_  


45\. _𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀤𑀺𑀲𑀮𑁄𑀘𑀦𑁂𑀦,_  
_𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀬 𑀬𑀁 𑀲𑀫𑀸𑀦𑁂𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑁂 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂,_  
_𑀅𑀬𑀁 𑀦𑀬𑁄 **𑀅𑀗𑁆𑀓𑀼𑀲𑁄** 𑀦𑀸𑀫𑁇_  


46\. _𑀲𑁄𑀍𑀅𑀲 𑀳𑀸𑀭𑀸 𑀧𑀞𑀫𑀁,_  
_𑀤𑀺𑀲𑀮𑁄𑀘𑀦𑀢𑁄 𑀤𑀺𑀲𑀸 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸;_  
_𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀬 𑀅𑀗𑁆𑀓𑀼𑀲𑁂𑀦 𑀳𑀺,_  
_𑀦𑀬𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑁂 𑀲𑀼𑀢𑁆𑀢𑀁𑁇_
