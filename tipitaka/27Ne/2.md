# 𑁨 𑀉𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

### 𑀉𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭

(1.)

8\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀲𑁄𑀍𑀅𑀲 𑀳𑀸𑀭𑀸? 𑀤𑁂𑀲𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀬𑀼𑀢𑁆𑀢𑀺 𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑁄 𑀮𑀓𑁆𑀔𑀡𑁄 𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑁄 𑀆𑀯𑀝𑁆𑀝𑁄 𑀯𑀺𑀪𑀢𑁆𑀢𑀺 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁄 𑀯𑁂𑀯𑀘𑀦𑁄 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀑𑀢𑀭𑀡𑁄 𑀲𑁄𑀥𑀦𑁄 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄 𑀲𑀫𑀸𑀭𑁄𑀧𑀦𑁄 𑀇𑀢𑀺𑁇

---

9\. 𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀕𑀻𑀢𑀺



10\. _𑀤𑁂𑀲𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀬𑀼𑀢𑁆𑀢𑀺,_  
_𑀧𑀤𑀝𑁆𑀞𑀸𑀦𑁄 𑀘 𑀮𑀓𑁆𑀔𑀡𑁄;_  
_𑀘𑀢𑀼𑀩𑁆𑀬𑀽𑀳𑁄 𑀘 𑀆𑀯𑀝𑁆𑀝𑁄,_  
_𑀯𑀺𑀪𑀢𑁆𑀢𑀺 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁄𑁇_  


11\. _𑀯𑁂𑀯𑀘𑀦𑁄 𑀘 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺,_  
_𑀑𑀢𑀭𑀡𑁄 𑀘 𑀲𑁄𑀥𑀦𑁄;_  
_𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄,_  
_𑀲𑀫𑀸𑀭𑁄𑀧𑀦𑁄 𑀲𑁄𑀍𑀅𑀲𑁄𑁇_  


12\. _𑀏𑀢𑁂 𑀲𑁄𑀍𑀅𑀲 𑀳𑀸𑀭𑀸,_  
_𑀧𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀢𑁄 𑀅𑀲𑀁𑀓𑀺𑀡𑁆𑀡𑀸;_  
_𑀏𑀢𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀪𑀯𑀢𑀺,_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀢𑀬𑀸 𑀦𑀬𑀯𑀺𑀪𑀢𑁆𑀢𑀻𑀢𑀺𑁇_  


(2.)

13\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀬𑀸? 𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀯𑀝𑁆𑀝𑁄 𑀢𑀺𑀧𑀼𑀓𑁆𑀔𑀮𑁄 𑀲𑀻𑀳𑀯𑀺𑀓𑁆𑀓𑀻𑀍𑀇𑀢𑁄 𑀤𑀺𑀲𑀸𑀮𑁄𑀘𑀦𑁄 𑀅𑀗𑁆𑀓𑀼𑀲𑁄 𑀇𑀢𑀺𑁇

---

14\. 𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀕𑀻𑀢𑀺



15\. _𑀧𑀞𑀫𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀯𑀝𑁆𑀝𑁄,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀘 𑀢𑀺𑀧𑀼𑀓𑁆𑀔𑀮𑁄;_  
_𑀲𑀻𑀳𑀯𑀺𑀓𑁆𑀓𑀻𑀍𑀇𑀢𑁄 𑀦𑀸𑀫,_  
_𑀢𑀢𑀺𑀬𑁄 𑀦𑀬𑀮𑀜𑁆𑀚𑀓𑁄𑁇_  


16\. _𑀤𑀺𑀲𑀸𑀮𑁄𑀘𑀦𑀫𑀸𑀳𑀁𑀲𑀼,_  
_𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀦𑀬𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀧𑀜𑁆𑀘𑀫𑁄 𑀅𑀗𑁆𑀓𑀼𑀲𑁄 𑀦𑀸𑀫,_  
_𑀲𑀩𑁆𑀩𑁂 𑀧𑀜𑁆𑀘 𑀦𑀬𑀸 𑀕𑀢𑀸𑀢𑀺𑁇_  


(3.)

17\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀫𑀽𑀮𑀧𑀤𑀸𑀦𑀺? 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺 𑀓𑀼𑀲𑀮𑀸𑀦𑀺 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀺, 𑀢𑀡𑁆𑀳𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀮𑁄𑀪𑁄 𑀤𑁄𑀲𑁄 𑀫𑁄𑀳𑁄 𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀢𑀺, 𑀇𑀫𑀸𑀦𑀺 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀺, 𑀬𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁄 𑀅𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁄 𑀲𑀗𑁆𑀕𑀳𑀁 𑀲𑀫𑁄𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

18\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺 𑀓𑀼𑀲𑀮𑀸𑀦𑀺? 𑀲𑀫𑀣𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀅𑀮𑁄𑀪𑁄 𑀅𑀤𑁄𑀲𑁄 𑀅𑀫𑁄𑀳𑁄 𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀢𑀺, 𑀇𑀫𑀸𑀦𑀺 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺 𑀓𑀼𑀲𑀮𑀸𑀦𑀺, 𑀬𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁄 𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑁄 𑀲𑀗𑁆𑀕𑀳𑀁 𑀲𑀫𑁄𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

---

19\. 𑀢𑀢𑁆𑀭𑀺𑀤𑀁 𑀉𑀤𑁆𑀤𑀸𑀦𑀁



20\. _𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀺 𑀘,_  
_𑀮𑁄𑀪𑁄 𑀤𑁄𑀲𑁄 𑀢𑀣𑁂𑀯 𑀫𑁄𑀳𑁄 𑀘;_  
_𑀘𑀢𑀼𑀭𑁄 𑀘 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀸,_  
_𑀓𑀺𑀮𑁂𑀲𑀪𑀽𑀫𑀻 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺𑁇_  


21\. _𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘,_  
_𑀓𑀼𑀲𑀮𑀸𑀦𑀺 𑀘 𑀬𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀫𑀽𑀮𑀸𑀦𑀺;_  
_𑀘𑀢𑀼𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀪𑀽𑀫𑀻 𑀦𑀯 𑀧𑀤𑀸𑀦𑀺𑁇_  


22\. _𑀦𑀯𑀳𑀺 𑀘 𑀧𑀤𑁂𑀳𑀺 𑀓𑀼𑀲𑀮𑀸,_  
_𑀦𑀯𑀳𑀺 𑀘 𑀬𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀧𑀓𑁆𑀔𑀸;_  
_𑀏𑀢𑁂 𑀔𑁄 𑀫𑀽𑀮𑀧𑀤𑀸,_  
_𑀪𑀯𑀦𑁆𑀢𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀧𑀤𑀸𑀦𑀻𑀢𑀺𑁇_  


---

23\. 𑀉𑀤𑁆𑀤𑁂𑀲𑀯𑀸𑀭𑁄𑁇
