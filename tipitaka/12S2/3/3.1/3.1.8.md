# 𑁩𑁇𑁧𑁇𑁮 𑀦𑁄𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀦𑁄𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(92.)

665\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁; 𑀦𑁄 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁𑁇 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁? 𑀭𑀽𑀧𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁𑁇

666\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁; 𑀦𑁄 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁?

667\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸; 𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑁇

668\. 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁; 𑀦𑁄 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇
