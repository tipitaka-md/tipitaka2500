# 𑁩𑁇𑁧𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀩𑀸𑀳𑀺𑀭𑀨𑀲𑁆𑀲𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀩𑀸𑀳𑀺𑀭𑀨𑀲𑁆𑀲𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(94.)

675\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀨𑀲𑁆𑀲…  𑀯𑁂𑀤𑀦𑀸…  𑀙𑀦𑁆𑀤…  𑀧𑀭𑀺𑀍𑀆𑀳…  𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀦𑀸𑀦𑀢𑁆𑀢𑀁; 𑀦𑁄 𑀮𑀸𑀪𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀙𑀦𑁆𑀤…  𑀯𑁂𑀤𑀦𑀸…  𑀨𑀲𑁆𑀲…  𑀲𑀗𑁆𑀓𑀧𑁆𑀧…  𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁𑁇 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁? 𑀭𑀽𑀧𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁𑁇

676\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁? 𑀨𑀲𑁆𑀲…  𑀯𑁂𑀤𑀦𑀸…  𑀙𑀦𑁆𑀤…  𑀧𑀭𑀺𑀍𑀆𑀳…  𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸…  𑀮𑀸𑀪…  𑀦𑁄 𑀮𑀸𑀪𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀍𑀆𑀳…  𑀙𑀦𑁆𑀤…  𑀯𑁂𑀤𑀦𑀸…  𑀨𑀲𑁆𑀲…  𑀦𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁?

677\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸, 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀮𑀸𑀪𑁄; 𑀦𑁄 𑀥𑀫𑁆𑀫𑀮𑀸𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀯𑁂𑀤𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑁇

678\. 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀲𑀗𑁆𑀓𑀧𑁆𑀧…  𑀨𑀲𑁆𑀲…  𑀯𑁂𑀤𑀦𑀸…  𑀙𑀦𑁆𑀤…  𑀧𑀭𑀺𑀍𑀆𑀳…  𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸…  𑀮𑀸𑀪…  𑀦𑁄 𑀮𑀸𑀪𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀧𑀭𑀺𑀍𑀆𑀳𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀯𑁂𑀤𑀦𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀨𑀲𑁆𑀲𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀨𑀲𑁆𑀲𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁, 𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

681\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



682\. _𑀥𑀸𑀢𑀼𑀨𑀲𑁆𑀲𑀜𑁆𑀘 𑀦𑁄 𑀘𑁂𑀢𑀁,_  
_𑀯𑁂𑀤𑀦𑀸 𑀅𑀧𑀭𑁂 𑀤𑀼𑀯𑁂;_  
_𑀏𑀢𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀧𑀜𑁆𑀘𑀓𑀁,_  
_𑀥𑀸𑀢𑀼𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀦𑁄 𑀘𑁂𑀢𑀁;_  
_𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀧𑀭𑁂 𑀤𑀼𑀯𑁂,_  
_𑀏𑀢𑀁 𑀩𑀸𑀳𑀺𑀭𑀧𑀜𑁆𑀘𑀓𑀦𑁆𑀢𑀺𑁇_
