# 𑁩𑁇𑁩𑁇𑁩 𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀫𑁆𑀫𑀧𑀣𑀯𑀕𑁆𑀕

### 𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀲𑀼𑀢𑁆𑀢

(109.)

768. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀺𑀦𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀺𑀦𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀺𑀦𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀬𑀺𑀦𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀬𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

769. 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
