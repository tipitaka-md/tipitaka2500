# 𑁩𑁇𑁩𑁇𑁬 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀫𑁆𑀫𑀧𑀣𑀯𑀕𑁆𑀕

### 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(112.)

777. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺𑀦𑁄…  𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑀦𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

778. 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸…  𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸…  𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀸…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀸…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀦𑁄…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀦𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
