# 𑁩𑁇𑁩𑁇𑁭 𑀤𑀲𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀫𑁆𑀫𑀧𑀣𑀯𑀕𑁆𑀕

### 𑀤𑀲𑀗𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(113.)

780. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀸…  𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺𑀦𑁄…  𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑀦𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀫𑀺𑀘𑁆𑀙𑀸𑀜𑀸𑀡𑀺𑀦𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀜𑀸𑀡𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

781. 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸…  𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸…  𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀸…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀸…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀦𑁄…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀦𑁄…  𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀺𑀦𑁄 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

---

782. 𑀲𑀢𑁆𑀢𑀫𑀁𑁇



---

783. (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)



---

784. 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀤𑁆𑀤𑀸𑀦𑀁



785. _𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀁,_  
_𑀧𑀜𑁆𑀘 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺 𑀘;_  
_𑀲𑀢𑁆𑀢 𑀓𑀫𑁆𑀫𑀧𑀣𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀲𑀓𑀫𑁆𑀫𑀧𑀣𑁂𑀦 𑀘;_  
_𑀙𑀝𑁆𑀞𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀯𑀼𑀢𑁆𑀢𑁄,_  
_𑀤𑀲𑀗𑁆𑀕𑁂𑀦 𑀘 𑀲𑀢𑁆𑀢𑀫𑀁𑁇_  


---

786. 𑀓𑀫𑁆𑀫𑀧𑀣𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
