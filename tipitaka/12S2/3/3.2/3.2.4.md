# 𑁩𑁇𑁨𑁇𑁪 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(98.)

706\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

707\. 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇

708\. 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

709\. 𑀏𑀢𑀭𑀳𑀺𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
