# 𑁩𑁇𑁨𑁇𑁫 𑀘𑀗𑁆𑀓𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀘𑀗𑁆𑀓𑀫𑀲𑀼𑀢𑁆𑀢

(99.)

711\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺; 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺; 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺; 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺; 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀦𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺; 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺; 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺; 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀧𑀺 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺𑁇

712\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸𑁇 𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸𑁇 𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀲𑁆𑀲𑀧𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀼𑀢𑀯𑀸𑀤𑀸𑁇 𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑁇 𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀡𑁆𑀡𑀁 𑀫𑀦𑁆𑀢𑀸𑀦𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸𑁇 𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀮𑀺𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀦𑀬𑀥𑀭𑀸𑁇 𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀦𑁆𑀤𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸𑁇 𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀩𑁆𑀩𑁂 𑀔𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀸𑁇

713\. 𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇

714\. 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

715\. 𑀏𑀢𑀭𑀳𑀺𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

---
