# 𑁩𑁇𑁨𑁇𑁧𑁨 𑀓𑀼𑀲𑀻𑀢𑀫𑀽𑀮𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀲𑀻𑀢𑀫𑀽𑀮𑀓𑀲𑀼𑀢𑁆𑀢

(106.)

754\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀼𑀲𑀻𑀢𑀸 𑀓𑀼𑀲𑀻𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀦𑁄 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀦𑁄 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺; 𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂… 𑁇

757\. 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄𑁇



---

758\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



759\. _𑀲𑀢𑁆𑀢𑀺𑀫𑀸 𑀲𑀦𑀺𑀤𑀸𑀦𑀜𑁆𑀘,_  
_𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑁂𑀦 𑀘;_  
_𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀘𑀗𑁆𑀓𑀫𑀁,_  
_𑀲𑀕𑀸𑀣𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀲𑀢𑁆𑀢𑀫𑀁𑁇_  


760\. _𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀫𑀽𑀮𑀓𑀸 𑀧𑀜𑁆𑀘,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀳𑀺𑀭𑀺𑀓𑀫𑀽𑀮𑀓𑀸;_  
_𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀫𑀽𑀮𑀓𑀸 𑀢𑀻𑀡𑀺,_  
_𑀤𑀼𑀯𑁂 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁂𑀦 𑀘𑁇_  


761\. _𑀓𑀼𑀲𑀻𑀢𑀁 𑀏𑀓𑀓𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸 𑀢𑀻𑀡𑀺 𑀧𑀜𑁆𑀘𑀓𑀸;_  
_𑀩𑀸𑀯𑀻𑀲𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸 𑀲𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
