# 𑁩𑁇𑁨𑁇𑁬 𑀲𑀕𑀸𑀣𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀕𑀸𑀣𑀸𑀲𑀼𑀢𑁆𑀢

(100.)

717\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇

718\. 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

719\. 𑀏𑀢𑀭𑀳𑀺𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

720\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀽𑀣𑁄 𑀕𑀽𑀣𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀫𑀼𑀢𑁆𑀢𑀁 𑀫𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀔𑁂𑀍𑀑 𑀔𑁂𑀍𑀏𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀧𑀼𑀩𑁆𑀩𑁄 𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀮𑁄𑀳𑀺𑀢𑀁 𑀮𑁄𑀳𑀺𑀢𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑀭𑀳𑀺𑀧𑀺 𑀔𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

721\. 𑀥𑀸𑀢𑀼𑀲𑁄𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀫𑀺𑀁𑀲𑀼𑁇

722\. 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑀭𑀳𑀺𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇

723\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀻𑀭𑀁 𑀔𑀻𑀭𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀢𑁂𑀮𑀁 𑀢𑁂𑀮𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀲𑀧𑁆𑀧𑀺 𑀲𑀧𑁆𑀧𑀺𑀦𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀫𑀥𑀼 𑀫𑀥𑀼𑀦𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀨𑀸𑀡𑀺𑀢𑀁 𑀨𑀸𑀡𑀺𑀢𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀁…  𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀁…  𑀏𑀢𑀭𑀳𑀺𑀧𑀺 𑀔𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀥𑀸𑀢𑀼𑀲𑁄𑀯 𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

724\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

725\. _“𑀲𑀁𑀲𑀕𑁆𑀕𑀸 𑀯𑀦𑀣𑁄 𑀚𑀸𑀢𑁄,_  
_𑀅𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀤𑀸𑀭𑀼𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀬𑀣𑀸 𑀲𑀻𑀤𑁂 𑀫𑀳𑀡𑁆𑀡𑀯𑁂𑁇_  


726\. _𑀏𑀯𑀁 𑀓𑀼𑀲𑀻𑀢𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀲𑀸𑀥𑀼𑀚𑀻𑀯𑀺𑀧𑀺 𑀲𑀻𑀤𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀢𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀓𑀼𑀲𑀻𑀢𑀁 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑀁𑁇_  


727\. _𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀅𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁂𑀳𑀺 𑀛𑀸𑀬𑀻𑀳𑀺;_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁂𑀳𑀺 𑀲𑀳𑀸𑀯𑀲𑁂”𑀢𑀺𑁇_
