# 𑁩𑁇𑁨𑁇𑁨 𑀲𑀦𑀺𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀦𑀺𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(96.)

689\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁𑁇

690\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁? 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸, 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀓𑀸𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄, 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀓𑀸𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑁇 𑀓𑀸𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦, 𑀯𑀸𑀘𑀸𑀬, 𑀫𑀦𑀲𑀸𑁇

691\. 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄…𑀧𑁂…  𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀘𑁆𑀙𑀦𑁆𑀤𑁄…  𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀍𑀆𑀳𑁄…  𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸…  𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦, 𑀯𑀸𑀘𑀸𑀬, 𑀫𑀦𑀲𑀸𑁇

692\. 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸; 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄…𑀧𑁂…  𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀙𑀦𑁆𑀤𑁄…  𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀭𑀺𑀍𑀆𑀳𑁄…  𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸…  𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦, 𑀯𑀸𑀘𑀸𑀬, 𑀫𑀦𑀲𑀸𑁇

693\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀢𑀺𑀡𑀼𑀓𑁆𑀓𑀁 𑀲𑀼𑀓𑁆𑀔𑁂 𑀢𑀺𑀡𑀤𑀸𑀬𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬; 𑀦𑁄 𑀘𑁂 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀘 𑀧𑀸𑀤𑁂𑀳𑀺 𑀘 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀲𑀫𑀕𑀢𑀁 𑀲𑀜𑁆𑀜𑀁 𑀦 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀺 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀭𑁄𑀢𑀺 𑀅𑀦𑀪𑀸𑀯𑀁 𑀕𑀫𑁂𑀢𑀺, 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀯𑀺𑀖𑀸𑀢𑀁 𑀲𑁅𑀧𑀸𑀬𑀸𑀲𑀁 𑀲𑀧𑀭𑀺𑀍𑀆𑀳𑀁; 𑀓𑀸𑀬𑀲𑁆𑀲 𑀘 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇

694\. 𑀲𑀦𑀺𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁𑁇

695\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁; 𑀲𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀦𑁄 𑀅𑀦𑀺𑀤𑀸𑀦𑀁? 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸; 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦, 𑀯𑀸𑀘𑀸𑀬, 𑀫𑀦𑀲𑀸𑁇

696\. 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸, 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄…𑀧𑁂…  𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀘𑁆𑀙𑀦𑁆𑀤𑁄…  𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀍𑀆𑀳𑁄…  𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸, 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦, 𑀯𑀸𑀘𑀸𑀬, 𑀫𑀦𑀲𑀸𑁇

697\. 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸, 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀙𑀦𑁆𑀤𑁄, 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀭𑀺𑀍𑀆𑀳𑁄, 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀭𑀺𑀍𑀆𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸; 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦, 𑀯𑀸𑀘𑀸𑀬, 𑀫𑀦𑀲𑀸𑁇

698\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀢𑀺𑀡𑀼𑀓𑁆𑀓𑀁 𑀲𑀼𑀓𑁆𑀔𑁂 𑀢𑀺𑀡𑀤𑀸𑀬𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬; 𑀢𑀫𑁂𑀦𑀁 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀘 𑀧𑀸𑀤𑁂𑀳𑀺 𑀘 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀦 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀲𑀫𑀕𑀢𑀁 𑀲𑀜𑁆𑀜𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀺 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀭𑁄𑀢𑀺 𑀅𑀦𑀪𑀸𑀯𑀁 𑀕𑀫𑁂𑀢𑀺, 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀔𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀯𑀺𑀖𑀸𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀬𑀸𑀲𑀁 𑀅𑀧𑀭𑀺𑀍𑀆𑀳𑀁; 𑀓𑀸𑀬𑀲𑁆𑀲 𑀘 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸”𑀢𑀺𑁇

---

699\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
