# 𑁩𑁇𑁪𑁇𑁭 𑀉𑀧𑁆𑀧𑀸𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(120.)

812\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀬𑁄 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀬𑀸…  𑀬𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇

813\. 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑁇 𑀬𑁄 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀬𑁄 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀬𑀸…  𑀬𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇
