# 𑁩𑁇𑁪𑁇𑁪 𑀦𑁄𑀘𑁂𑀤𑀁𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀦𑁄𑀘𑁂𑀤𑀁𑀲𑀼𑀢𑁆𑀢

(117.)

799\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀅𑀪𑀯𑀺𑀲𑁆𑀲, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀸𑀭𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄 𑀅𑀪𑀯𑀺𑀲𑁆𑀲, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀅𑀪𑀯𑀺𑀲𑁆𑀲, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀺𑁇

800\. 𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀅𑀪𑀯𑀺𑀲𑁆𑀲…𑀧𑁂…  𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀅𑀪𑀯𑀺𑀲𑁆𑀲, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀸𑀭𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀸𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄 𑀅𑀪𑀯𑀺𑀲𑁆𑀲, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀦𑁄 𑀘𑁂𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀅𑀪𑀯𑀺𑀲𑁆𑀲, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀺𑁇

801\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀇𑀫𑀸𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀢𑁄 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀦𑁂𑀯 𑀢𑀸𑀯𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀤𑁂𑀯𑀓𑀸 𑀮𑁄𑀓𑀸 𑀲𑀫𑀸𑀭𑀓𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀝𑀸 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀫𑀭𑀺𑀬𑀸𑀤𑀺𑀓𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀺𑀁𑀲𑀼𑁇

802\. 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀇𑀫𑀸𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀢𑁄 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀅𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀤𑁂𑀯𑀓𑀸 𑀮𑁄𑀓𑀸 𑀲𑀫𑀸𑀭𑀓𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀝𑀸 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀫𑀭𑀺𑀬𑀸𑀤𑀺𑀓𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
