# 𑁩𑁇𑁪𑁇𑁧𑁦 𑀢𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

### 𑀢𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(123.)

820\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…  𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…  𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀦 𑀫𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀡𑁂𑀲𑀼 𑀯𑀸 𑀲𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸; 𑀦 𑀘 𑀧𑀦 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀁 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀁 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇

821\. 𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…  𑀬𑁂 𑀘 𑀔𑁄 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸…𑀧𑁂…  𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…  𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…  𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘 𑀔𑁄 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀲𑀫𑀡𑁂𑀲𑀼 𑀘𑁂𑀯 𑀲𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸; 𑀢𑁂 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀜𑁆𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
