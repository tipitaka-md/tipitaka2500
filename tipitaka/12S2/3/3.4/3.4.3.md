# 𑁩𑁇𑁪𑁇𑁩 𑀅𑀘𑀭𑀺𑀁𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀅𑀘𑀭𑀺𑀁𑀲𑀼𑀢𑁆𑀢

(116.)

794\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀸𑀤𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀅𑀘𑀭𑀺𑀁, 𑀬𑁄 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀢𑀤𑀚𑁆𑀛𑀕𑀫𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑁂 𑀲𑁄 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄𑁇 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀅𑀘𑀭𑀺𑀁, 𑀬𑁄 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄 𑀢𑀤𑀚𑁆𑀛𑀕𑀫𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑁂 𑀲𑁄 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄𑁇 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀅𑀘𑀭𑀺𑀁, 𑀬𑀁 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀢𑀤𑀚𑁆𑀛𑀕𑀫𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑁂 𑀢𑀁 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀁𑁇

795\. 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀸𑀤𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀅𑀘𑀭𑀺𑀁, 𑀬𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀢𑀤𑀚𑁆𑀛𑀕𑀫𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑁂 𑀲𑁄 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄𑁇 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀅𑀘𑀭𑀺𑀁, 𑀬𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄 𑀢𑀤𑀚𑁆𑀛𑀕𑀫𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑁂 𑀲𑁄 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄𑁇 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀁 𑀅𑀘𑀭𑀺𑀁, 𑀬𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀢𑀤𑀚𑁆𑀛𑀕𑀫𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑁂 𑀢𑀁 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀁𑁇

796\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀢𑁄 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, 𑀦𑁂𑀯 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀧𑀘𑁆𑀘𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇

797\. 𑀬𑀢𑁄 𑀘 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀢𑁄 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, 𑀅𑀣𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀧𑀘𑁆𑀘𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀜𑀸𑀡𑀜𑁆𑀘 𑀧𑀦 𑀫𑁂 𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀫𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀅𑀬𑀫𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀸𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄’”𑀢𑀺𑁇
