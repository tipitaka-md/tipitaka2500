# 𑁩𑁇𑁪𑁇𑁬 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(119.)

809\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑁄 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀅𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀬𑁄 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀬𑁄 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁…  𑀬𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑁄 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀅𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

810\. 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑁄 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀬𑁄 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁…𑀧𑁂…  𑀬𑁄 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁…  𑀬𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑁄 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀻”𑀢𑀺𑁇
