# 𑁩𑁇𑁪𑁇𑁨 𑀧𑀼𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀸𑀢𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀢𑁆𑀢

(115.)

789\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁; 𑀓𑁄 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁; 𑀓𑁄 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁; 𑀓𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄, 𑀓𑁄 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀓𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆’𑀢𑀺?

790\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀁 𑀔𑁄 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀅𑀬𑀁 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄; 𑀬𑀁 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸, 𑀅𑀬𑀁 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄; 𑀬𑁄 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀁, 𑀇𑀤𑀁 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 𑀬𑀁 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀬𑀁 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀬𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀅𑀬𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑁄; 𑀬𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸, 𑀅𑀬𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑁄; 𑀬𑁄 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀁, 𑀇𑀤𑀁 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁’𑁇

791\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀏𑀯𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀢𑁄 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, 𑀦𑁂𑀯 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀧𑀘𑁆𑀘𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇

792\. 𑀬𑀢𑁄 𑀘 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀏𑀯𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀢𑁄 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀢𑁄 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, 𑀅𑀣𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀧𑀘𑁆𑀘𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀜𑀸𑀡𑀜𑁆𑀘 𑀧𑀦 𑀫𑁂 𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀫𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺, 𑀅𑀬𑀫𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀸𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄’”𑀢𑀺𑁇
