# 𑁬𑁇𑁩𑁇𑁪 𑀏𑀓𑀥𑀻𑀢𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀥𑀻𑀢𑀼𑀲𑀼𑀢𑁆𑀢

(173.)

1066\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀲𑀤𑁆𑀥𑀸 𑀪𑀺𑀓𑁆𑀔𑀯𑁂 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀏𑀓𑀁 𑀥𑀻𑀢𑀭𑀁 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁 𑀏𑀯𑀁 𑀲𑀫𑁆𑀫𑀸 𑀆𑀬𑀸𑀘𑀫𑀸𑀦𑀸 𑀆𑀬𑀸𑀘𑁂𑀬𑁆𑀬—  ‘𑀢𑀸𑀤𑀺𑀲𑀸, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀯𑀸𑀳𑀺 𑀬𑀸𑀤𑀺𑀲𑀸 𑀔𑀼𑀚𑁆𑀚𑀼𑀢𑁆𑀢𑀭𑀸 𑀘 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀯𑁂𑀍𑀉𑀓𑀡𑁆𑀟𑀓𑀺𑀬𑀸 𑀘 𑀦𑀦𑁆𑀤𑀫𑀸𑀢𑀸’𑀢𑀺𑁇 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀮𑀸 𑀏𑀢𑀁 𑀧𑀫𑀸𑀡𑀁 𑀫𑀫 𑀲𑀸𑀯𑀺𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁, 𑀬𑀤𑀺𑀤𑀁 𑀔𑀼𑀚𑁆𑀚𑀼𑀢𑁆𑀢𑀭𑀸 𑀘 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀯𑁂𑀍𑀉𑀓𑀡𑁆𑀟𑀓𑀺𑀬𑀸 𑀘 𑀦𑀦𑁆𑀤𑀫𑀸𑀢𑀸𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀲𑀺; 𑀢𑀸𑀤𑀺𑀲𑀸, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀯𑀸𑀳𑀺 𑀬𑀸𑀤𑀺𑀲𑀸 𑀔𑁂𑀫𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘𑀸𑀢𑀺𑁇 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀮𑀸 𑀏𑀢𑀁 𑀧𑀫𑀸𑀡𑀁 𑀫𑀫 𑀲𑀸𑀯𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁, 𑀬𑀤𑀺𑀤𑀁 𑀔𑁂𑀫𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘𑁇 𑀫𑀸 𑀘 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑁂𑀔𑀁 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀽𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀲𑁂𑀔𑀁 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺, 𑀲𑁄 𑀢𑀲𑁆𑀲𑀸 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
