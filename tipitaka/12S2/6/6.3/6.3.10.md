# 𑁬𑁇𑁩𑁇𑁧𑁦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(179.)

1079\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑁄𑀧𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀳𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬 𑀯𑀤𑀸𑀫𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸”𑀢𑀺? “𑀬𑀸 𑀳𑀺𑀲𑁆𑀲 𑀲𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀦𑀸𑀳𑀁 𑀢𑀲𑁆𑀲𑀸 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬 𑀯𑀤𑀸𑀫𑀺𑁇 𑀬𑁂 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸 𑀅𑀥𑀺𑀕𑀢𑀸 𑀢𑁂𑀲𑀸𑀳𑀫𑀲𑁆𑀲 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬 𑀯𑀤𑀸𑀫𑀺𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀓𑀝𑀼𑀓𑁄 𑀨𑀭𑀼𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀞𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
