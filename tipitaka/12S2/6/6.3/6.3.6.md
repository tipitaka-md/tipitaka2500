# 𑁬𑁇𑁩𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(175.)

1070\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
