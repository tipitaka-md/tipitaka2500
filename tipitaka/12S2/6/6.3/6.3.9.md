# 𑁬𑁇𑁩𑁇𑁯 𑀭𑀚𑁆𑀚𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀭𑀚𑁆𑀚𑀼𑀲𑀼𑀢𑁆𑀢

(178.)

1076. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄𑁇 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀯𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀙𑀯𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀘𑀫𑁆𑀫𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀘𑀫𑁆𑀫𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀫𑀁𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇

1077. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀍𑀳𑀸𑀬 𑀯𑀸𑀍𑀅𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀚𑀗𑁆𑀖𑀁 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀖𑀁𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀸 𑀙𑀯𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬, 𑀙𑀯𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀘𑀫𑁆𑀫𑀁 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬, 𑀘𑀫𑁆𑀫𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬, 𑀫𑀁𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬, 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬, 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀙𑀯𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀙𑀯𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀘𑀫𑁆𑀫𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀘𑀫𑁆𑀫𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀫𑀁𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
