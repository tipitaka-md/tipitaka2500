# 𑁬𑁇𑁪𑁇𑁮--𑁧𑁩 𑀧𑀺𑀢𑀼𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀙𑀓𑁆𑀓

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀢𑀼𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀙𑀓𑁆𑀓

(187.)

1105\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀓𑀝𑀼𑀓𑁄 𑀨𑀭𑀼𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀦 𑀘𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀢𑀼𑀧𑀺 𑀳𑁂𑀢𑀼…𑀧𑁂…  (𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁)…  𑀪𑀸𑀢𑀼𑀧𑀺 𑀳𑁂𑀢𑀼…  𑀪𑀕𑀺𑀦𑀺𑀬𑀸𑀧𑀺 𑀳𑁂𑀢𑀼…  𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲𑀧𑀺 𑀳𑁂𑀢𑀼…  𑀥𑀻𑀢𑀼𑀬𑀸𑀧𑀺 𑀳𑁂𑀢𑀼…  𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑀸𑀧𑀺 𑀳𑁂𑀢𑀼 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀦𑁆𑀢𑀁𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀓𑀝𑀼𑀓𑁄 𑀨𑀭𑀼𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀞𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
