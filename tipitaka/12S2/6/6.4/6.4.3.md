# 𑁬𑁇𑁪𑁇𑁩 𑀓𑀼𑀲𑀮𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀲𑀮𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(182.)

1088\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄𑁇 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀫𑀕𑀫𑀸𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
