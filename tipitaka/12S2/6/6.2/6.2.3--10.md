# 𑁬𑁇𑁨𑁇𑁩--𑁧𑁦 𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑀺𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀅𑀝𑁆𑀞𑀓

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑀺𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀅𑀝𑁆𑀞𑀓

(169.)

1055\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀦 𑀘𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑀺𑀓𑁆𑀔𑀲𑁆𑀲𑀸𑀧𑀺 𑀳𑁂𑀢𑀼…𑀧𑁂…  𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑀺𑀓𑁆𑀔𑀲𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀳𑁂𑀢𑀼…  𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑁆𑀲𑀸𑀧𑀺 𑀳𑁂𑀢𑀼…  𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀳𑁂𑀢𑀼…  𑀧𑀣𑀯𑀺𑀬𑀸𑀧𑀺 𑀚𑀸𑀢𑀭𑀽𑀧𑀧𑀭𑀺𑀧𑀽𑀭𑀸𑀬 𑀳𑁂𑀢𑀼…  𑀆𑀫𑀺𑀲𑀓𑀺𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀳𑁂𑀢𑀼𑀧𑀺…  𑀚𑀻𑀯𑀺𑀢𑀳𑁂𑀢𑀼𑀧𑀺…  𑀚𑀦𑀧𑀤𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀬𑀸𑀧𑀺 𑀳𑁂𑀢𑀼 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀦𑁆𑀢𑀁𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
