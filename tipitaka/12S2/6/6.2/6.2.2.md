# 𑁬𑁇𑁨𑁇𑁨 𑀭𑀽𑀧𑀺𑀬𑀧𑀸𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀭𑀽𑀧𑀺𑀬𑀧𑀸𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(168.)

1053\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀦 𑀘𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀭𑀽𑀧𑀺𑀬𑀧𑀸𑀢𑀺𑀬𑀸𑀧𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀘𑀼𑀡𑁆𑀡𑀧𑀭𑀺𑀧𑀽𑀭𑀸𑀬 𑀳𑁂𑀢𑀼 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀦𑁆𑀢𑀁𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
