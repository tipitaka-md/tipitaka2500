# 𑁬𑁇𑁧𑁇𑁬 𑀅𑀲𑀦𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀅𑀲𑀦𑀺𑀲𑀼𑀢𑁆𑀢

(162.)

1033\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀦𑀺𑀯𑀺𑀘𑀓𑁆𑀓𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀼, 𑀲𑁂𑀔𑀁 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀼𑁇

1034\. 𑀅𑀲𑀦𑀺𑀯𑀺𑀘𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
