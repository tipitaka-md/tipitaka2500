# 𑁬𑁇𑁧𑁇𑁧𑁦 𑀲𑀕𑀸𑀣𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀕𑀸𑀣𑀓𑀲𑀼𑀢𑁆𑀢

(166.)

1043\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀁, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀁𑁇 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀁, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀁𑁇 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘 𑀢𑀤𑀼𑀪𑀬𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀁, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀁𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

1044\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

1045\. _“𑀬𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑀭𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀲𑀫𑀸𑀥𑀺 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄𑁇_  


1046\. _𑀢𑀁 𑀛𑀸𑀬𑀺𑀦𑀁 𑀲𑀸𑀢𑀢𑀺𑀓𑀁,_  
_𑀲𑀼𑀔𑀼𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑀁;_  
_𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀬𑀸𑀭𑀸𑀫𑀁,_  
_𑀆𑀳𑀼 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀢𑀻”𑀢𑀺𑁇_

1050\. _𑀤𑀸𑀭𑀼𑀡𑁄 𑀩𑀍𑀇𑀲𑀁 𑀓𑀼𑀫𑁆𑀫𑀁,_
