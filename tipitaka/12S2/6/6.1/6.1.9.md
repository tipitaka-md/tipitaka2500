# 𑁬𑁇𑁧𑁇𑁯 𑀯𑁂𑀭𑀫𑁆𑀪𑀲𑀼𑀢𑁆𑀢

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀯𑁂𑀭𑀫𑁆𑀪𑀲𑀼𑀢𑁆𑀢

(165.)

1041\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀉𑀧𑀭𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀓𑀸𑀲𑁂 𑀯𑁂𑀭𑀫𑁆𑀪𑀸 𑀦𑀸𑀫 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀧𑀓𑁆𑀔𑀻 𑀕𑀘𑁆𑀙𑀢𑀺 𑀢𑀫𑁂𑀦𑀁 𑀯𑁂𑀭𑀫𑁆𑀪𑀸 𑀯𑀸𑀢𑀸 𑀔𑀺𑀧𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀯𑁂𑀭𑀫𑁆𑀪𑀯𑀸𑀢𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑁂𑀦𑁂𑀯 𑀧𑀸𑀤𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀧𑀓𑁆𑀔𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀲𑀻𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀸𑀬𑁄 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀢𑀺 𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦𑁂𑀯 𑀓𑀸𑀬𑁂𑀦 𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦, 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸, 𑀅𑀲𑀁𑀯𑀼𑀢𑁂𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀧𑀲𑁆𑀲𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀣𑀁 𑀯𑀸 𑀤𑀼𑀧𑁆𑀧𑀸𑀭𑀼𑀢𑀁 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀼𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀣𑀁 𑀯𑀸 𑀤𑀼𑀧𑁆𑀧𑀸𑀭𑀼𑀢𑀁 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑁂 𑀘𑀻𑀯𑀭𑀁 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂 𑀧𑀢𑁆𑀢𑀁 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂 𑀲𑀽𑀘𑀺𑀖𑀭𑀁 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀯𑁂𑀭𑀫𑁆𑀪𑀯𑀸𑀢𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀓𑀼𑀡𑀲𑁆𑀲𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
