# 𑁬𑁇𑁧𑁇𑁭 𑀤𑀺𑀤𑁆𑀥𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀤𑀺𑀤𑁆𑀥𑀲𑀼𑀢𑁆𑀢

(163.)

1036\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀤𑁆𑀥𑀕𑀢𑁂𑀦 𑀯𑀺𑀲𑀮𑁆𑀮𑁂𑀦 𑀲𑀮𑁆𑀮𑁂𑀦 𑀯𑀺𑀚𑁆𑀛𑀢𑀼, 𑀲𑁂𑀔𑀁 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀼𑁇

1037\. 𑀲𑀮𑁆𑀮𑀦𑁆𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
