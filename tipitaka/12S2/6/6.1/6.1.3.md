# 𑁬𑁇𑁧𑁇𑁩 𑀓𑀼𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(159.)

1026\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀉𑀤𑀓𑀭𑀳𑀤𑁂 𑀫𑀳𑀸𑀓𑀼𑀫𑁆𑀫𑀓𑀼𑀮𑀁 𑀘𑀺𑀭𑀦𑀺𑀯𑀸𑀲𑀺 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀓𑀼𑀫𑁆𑀫𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀓𑀼𑀫𑁆𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀫𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀏𑀢𑀁 𑀧𑀤𑁂𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀻’𑀢𑀺𑁇 𑀅𑀕𑀫𑀸𑀲𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑁄 𑀢𑀁 𑀧𑀤𑁂𑀲𑀁𑁇 𑀢𑀫𑁂𑀦𑀁 𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀧𑀢𑀸𑀬 𑀯𑀺𑀚𑁆𑀛𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑁄 𑀬𑁂𑀦 𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑁄 𑀢𑀁 𑀓𑀼𑀫𑁆𑀫𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀢𑀁 𑀓𑀼𑀫𑁆𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀓𑀘𑁆𑀘𑀺 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀦 𑀢𑀁 𑀧𑀤𑁂𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀻’𑀢𑀺? ‘𑀅𑀕𑀫𑀸𑀲𑀺𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀢𑀁 𑀧𑀤𑁂𑀲𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀘𑁆𑀘𑀺 𑀧𑀦𑀸𑀲𑀺, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀅𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀼𑀧𑀳𑀢𑁄’𑀢𑀺? ‘𑀅𑀓𑁆𑀔𑀢𑁄 𑀔𑁄𑀫𑁆𑀳𑀺, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀅𑀦𑀼𑀧𑀳𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀫𑁂 𑀇𑀤𑀁 𑀲𑀼𑀢𑁆𑀢𑀓𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀦𑁆’𑀢𑀺𑁇 ‘𑀢𑀕𑁆𑀖𑀲𑀺, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀔𑀢𑁄, 𑀢𑀕𑁆𑀖 𑀉𑀧𑀳𑀢𑁄𑁇 𑀏𑀢𑁂𑀦 𑀳𑀺 𑀢𑁂, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀲𑀼𑀢𑁆𑀢𑀓𑁂𑀦 𑀧𑀺𑀢𑀭𑁄 𑀘 𑀧𑀺𑀢𑀸𑀫𑀳𑀸 𑀘 𑀅𑀦𑀬𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑀸𑁇 𑀕𑀘𑁆𑀙 𑀤𑀸𑀦𑀺 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀓𑀼𑀫𑁆𑀫, 𑀦 𑀤𑀸𑀦𑀺 𑀢𑁆𑀯𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀦𑁆’𑀢𑀺𑁇

1027\. 𑀮𑀼𑀤𑁆𑀤𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀭𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀧𑀧𑀢𑀸𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀲𑀼𑀢𑁆𑀢𑀓𑀦𑁆𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀦𑀺𑀓𑀸𑀫𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀤𑁆𑀥𑁄 𑀧𑀧𑀢𑀸𑀬 𑀅𑀦𑀬𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
