# 𑁬𑁇𑁧𑁇𑁫 𑀫𑀻𑀍𑀳𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀻𑀍𑀳𑀓𑀲𑀼𑀢𑁆𑀢

(161.)

1031\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀻𑀍𑀳𑀓𑀸 𑀕𑀽𑀣𑀸𑀤𑀻 𑀕𑀽𑀣𑀧𑀽𑀭𑀸 𑀧𑀼𑀡𑁆𑀡𑀸 𑀕𑀽𑀣𑀲𑁆𑀲𑁇 𑀧𑀼𑀭𑀢𑁄 𑀘𑀲𑁆𑀲 𑀫𑀳𑀸𑀕𑀽𑀣𑀧𑀼𑀜𑁆𑀚𑁄𑁇 𑀲𑀸 𑀢𑁂𑀦 𑀅𑀜𑁆𑀜𑀸 𑀫𑀻𑀍𑀳𑀓𑀸 𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀫𑁆𑀳𑀺 𑀕𑀽𑀣𑀸𑀤𑀻 𑀕𑀽𑀣𑀧𑀽𑀭𑀸 𑀧𑀼𑀡𑁆𑀡𑀸 𑀕𑀽𑀣𑀲𑁆𑀲, 𑀧𑀼𑀭𑀢𑁄 𑀘 𑀫𑁆𑀬𑀸𑀬𑀁 𑀫𑀳𑀸𑀕𑀽𑀣𑀧𑀼𑀜𑁆𑀚𑁄’𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄, 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀘 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀘𑀲𑁆𑀲 𑀧𑀽𑀭𑁄𑁇 𑀲𑁄 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀕𑀡𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂 𑀯𑀺𑀓𑀢𑁆𑀣𑀢𑀺—  ‘𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀘𑀫𑁆𑀳𑀺 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄, 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀘𑀫𑁆𑀳𑀺 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀘 𑀫𑁆𑀬𑀸𑀬𑀁 𑀧𑀽𑀭𑁄, 𑀮𑀸𑀪𑀻 𑀘𑀫𑁆𑀳𑀺 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁, 𑀇𑀫𑁂 𑀧𑀦𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀧𑀼𑀜𑁆𑀜𑀸 𑀅𑀧𑁆𑀧𑁂𑀲𑀓𑁆𑀔𑀸 𑀦 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑁂 𑀧𑁂𑀲𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺𑁇 𑀢𑀜𑁆𑀳𑀺 𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
