# 𑁬𑁇𑁧𑁇𑁮 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(164.)

1039\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀅𑀲𑁆𑀲𑀼𑀢𑁆𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀚𑀭𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀫𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀏𑀲𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀭𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀉𑀓𑁆𑀓𑀡𑁆𑀟𑀓𑁂𑀦 𑀦𑀸𑀫 𑀭𑁄𑀕𑀚𑀸𑀢𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑁄 𑀦𑁂𑀯 𑀩𑀺𑀮𑀕𑀢𑁄 𑀭𑀫𑀢𑀺, 𑀦 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀭𑀫𑀢𑀺, 𑀦 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑀕𑀢𑁄 𑀭𑀫𑀢𑀺; 𑀬𑁂𑀦 𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁂𑀦 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑁄 𑀦𑁂𑀯 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀭𑀫𑀢𑀺, 𑀦 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀭𑀫𑀢𑀺, 𑀦 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑀕𑀢𑁄 𑀭𑀫𑀢𑀺; 𑀬𑁂𑀦 𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
