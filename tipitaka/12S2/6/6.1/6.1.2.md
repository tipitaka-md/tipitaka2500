# 𑁬𑁇𑁧𑁇𑁨 𑀩𑀍𑀇𑀲𑀲𑀼𑀢𑁆𑀢

* 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀩𑀍𑀇𑀲𑀲𑀼𑀢𑁆𑀢

(158.)

1023\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀤𑀸𑀭𑀼𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀓𑀝𑀼𑀓𑁄 𑀨𑀭𑀼𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑁄 𑀆𑀫𑀺𑀲𑀕𑀢𑀁 𑀩𑀍𑀇𑀲𑀁 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀉𑀤𑀓𑀭𑀳𑀤𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑁂𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀆𑀫𑀺𑀲𑀘𑀓𑁆𑀔𑀼 𑀫𑀘𑁆𑀙𑁄 𑀕𑀺𑀮𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀘𑁆𑀙𑁄 𑀕𑀺𑀮𑀩𑀍𑀇𑀲𑁄 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀲𑁆𑀲 𑀅𑀦𑀬𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀲𑁆𑀲𑁇

1024\. 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑁄𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀭𑀲𑁆𑀲𑁂𑀢𑀁 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀩𑀍𑀇𑀲𑀦𑁆𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀦𑀺𑀓𑀸𑀫𑁂𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀩𑀍𑀇𑀲𑁄 𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀦𑀬𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀬𑀣𑀸𑀓𑀸𑀫𑀓𑀭𑀡𑀻𑀬𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇 𑀏𑀯𑀁 𑀤𑀸𑀭𑀼𑀡𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀓𑀝𑀼𑀓𑁄 𑀨𑀭𑀼𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀞𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
