# 𑁧𑁇𑁫𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀧𑀜𑁆𑀘𑀯𑁂𑀭𑀪𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀜𑁆𑀘𑀯𑁂𑀭𑀪𑀬𑀲𑀼𑀢𑁆𑀢

(42.)

307\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀪𑀬𑀸𑀦𑀺 𑀯𑁂𑀭𑀸𑀦𑀺 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀘𑀢𑀽𑀳𑀺 𑀘 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀭𑀺𑀬𑁄 𑀘𑀲𑁆𑀲 𑀜𑀸𑀬𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑁄, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑁄’𑀢𑀺𑁇

308\. 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑀬𑀸𑀦𑀺 𑀯𑁂𑀭𑀸𑀦𑀺 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 …𑀧𑁂…  𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻…  𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻…  𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻…  𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀬𑀻…𑀧𑁂…  𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑀬𑀸𑀦𑀺 𑀯𑁂𑀭𑀸𑀦𑀺 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

309\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂…  𑀲𑀁𑀖𑁂…  𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

310\. 𑀓𑀢𑀫𑁄 𑀘𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑁄 𑀜𑀸𑀬𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀜𑁆𑀜𑁂𑀯 𑀲𑀸𑀥𑀼𑀓𑀁 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀫𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑁄 𑀜𑀸𑀬𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑁄𑁇

311\. 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑀬𑀸𑀦𑀺 𑀯𑁂𑀭𑀸𑀦𑀺 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑁄 𑀜𑀸𑀬𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑁄, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑁄’”𑀢𑀺𑁇

---

312\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
