# 𑁧𑁇𑁫𑁇𑁮 𑀮𑁄𑀓𑀸𑀬𑀢𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀮𑁄𑀓𑀸𑀬𑀢𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(48.)

342\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀮𑁄𑀓𑀸𑀬𑀢𑀺𑀓𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀮𑁄𑀓𑀸𑀬𑀢𑀺𑀓𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

343\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀩𑁆𑀩𑀫𑀢𑁆𑀣𑀻”𑀢𑀺? “‘𑀲𑀩𑁆𑀩𑀫𑀢𑁆𑀣𑀻’𑀢𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀚𑁂𑀝𑁆𑀞𑀫𑁂𑀢𑀁 𑀮𑁄𑀓𑀸𑀬𑀢𑀁”𑁇

344\. “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀩𑁆𑀩𑀁 𑀦𑀢𑁆𑀣𑀻”𑀢𑀺? “‘𑀲𑀩𑁆𑀩𑀁 𑀦𑀢𑁆𑀣𑀻’𑀢𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀤𑀼𑀢𑀺𑀬𑀫𑁂𑀢𑀁 𑀮𑁄𑀓𑀸𑀬𑀢𑀁”𑁇

345\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀩𑁆𑀩𑀫𑁂𑀓𑀢𑁆𑀢𑀦𑁆”𑀢𑀺? “‘𑀲𑀩𑁆𑀩𑀫𑁂𑀓𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀢𑀺𑀬𑀫𑁂𑀢𑀁 𑀮𑁄𑀓𑀸𑀬𑀢𑀁”𑁇

346\. “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀩𑁆𑀩𑀁 𑀧𑀼𑀣𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺? “‘𑀲𑀩𑁆𑀩𑀁 𑀧𑀼𑀣𑀼𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀘𑀢𑀼𑀢𑁆𑀣𑀫𑁂𑀢𑀁 𑀮𑁄𑀓𑀸𑀬𑀢𑀁𑁇

347\. 𑀏𑀢𑁂 𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻’”𑀢𑀺𑁇

348\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀮𑁄𑀓𑀸𑀬𑀢𑀺𑀓𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

---

349\. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
