# 𑁧𑁇𑁫𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

(50.)

356\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀓𑀺𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀲𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀓𑀺𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀢𑀡𑁆𑀳𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀪𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀻’𑀢𑀺?

357\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀜𑀸𑀡𑀫𑁂𑀯𑁂𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀤𑀁 𑀳𑁄𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀇𑀤𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺; 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀺; 𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺; 𑀲𑀍𑀆𑀬𑀢𑀦𑁂 𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺; 𑀨𑀲𑁆𑀲𑁂 𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺; 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀢𑀺 𑀢𑀡𑁆𑀳𑀸 𑀳𑁄𑀢𑀺; 𑀢𑀡𑁆𑀳𑀸𑀬 𑀲𑀢𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺; 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀲𑀢𑀺 𑀪𑀯𑁄 𑀳𑁄𑀢𑀺; 𑀪𑀯𑁂 𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀳𑁄𑀢𑀺; 𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀏𑀯𑀫𑀬𑀁 𑀮𑁄𑀓𑁄 𑀲𑀫𑀼𑀤𑀬𑀢𑀻’𑀢𑀺𑁇

358\. 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀓𑀺𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀓𑀺𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀢𑀡𑁆𑀳𑀸 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑀁…  𑀪𑀯𑁄…  𑀚𑀸𑀢𑀺…  𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺?

359\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀜𑀸𑀡𑀫𑁂𑀯𑁂𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀇𑀤𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀇𑀤𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀅𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺; 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀅𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀅𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀏𑀯𑀫𑀬𑀁 𑀮𑁄𑀓𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻’𑀢𑀺𑁇

360\. 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀮𑁄𑀓𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀤𑀲𑁆𑀲𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀆𑀕𑀢𑁄 𑀇𑀫𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀧𑀲𑁆𑀲𑀢𑀺 𑀇𑀫𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀲𑁂𑀓𑁆𑀔𑀸𑀬 𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀥𑀫𑁆𑀫𑀲𑁄𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀅𑀭𑀺𑀬𑁄 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀅𑀫𑀢𑀤𑁆𑀯𑀸𑀭𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀇𑀢𑀺𑀧𑀻”𑀢𑀺𑁇

363\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



364\. _𑀤𑁆𑀯𑁂 𑀧𑀜𑁆𑀘𑀯𑁂𑀭𑀪𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀮𑁄𑀓𑁄 𑀘 𑀜𑀸𑀢𑀺𑀓𑀁;_  
_𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺 𑀘,_  
_𑀮𑁄𑀓𑀸𑀬𑀢𑀺𑀓𑁂𑀦 𑀅𑀝𑁆𑀞𑀫𑀁;_  
_𑀤𑁆𑀯𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
