# 𑁧𑁇𑁫𑁇𑁯 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

(49.)

350\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄—  𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀓𑀺𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀲𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀓𑀺𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀢𑀡𑁆𑀳𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀪𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀻’𑀢𑀺?

351\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀜𑀸𑀡𑀫𑁂𑀯𑁂𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀤𑀁 𑀳𑁄𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀇𑀤𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺; 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀺; 𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺; 𑀲𑀍𑀆𑀬𑀢𑀦𑁂 𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺; 𑀨𑀲𑁆𑀲𑁂 𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺; 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀢𑀺 𑀢𑀡𑁆𑀳𑀸 𑀳𑁄𑀢𑀺; 𑀢𑀡𑁆𑀳𑀸𑀬 𑀲𑀢𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺; 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀲𑀢𑀺 𑀪𑀯𑁄 𑀳𑁄𑀢𑀺; 𑀪𑀯𑁂 𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀳𑁄𑀢𑀺; 𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀏𑀯𑀫𑀬𑀁 𑀮𑁄𑀓𑁄 𑀲𑀫𑀼𑀤𑀬𑀢𑀻’𑀢𑀺𑁇

352\. 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄—  𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀓𑀺𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀓𑀺𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺? 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀨𑀲𑁆𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀢𑀡𑁆𑀳𑀸 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀪𑀯𑁄 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺?

353\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀜𑀸𑀡𑀫𑁂𑀯𑁂𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀇𑀤𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀇𑀤𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀅𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺; 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀅𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀅𑀲𑀢𑀺 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀪𑀯𑁄 𑀦 𑀳𑁄𑀢𑀺…  𑀚𑀸𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺…  𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀏𑀯𑀫𑀬𑀁 𑀮𑁄𑀓𑁄 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻’𑀢𑀺𑁇

354\. 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀮𑁄𑀓𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀇𑀢𑀺𑀧𑀺…𑀧𑁂…  𑀅𑀫𑀢𑀤𑁆𑀯𑀸𑀭𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀇𑀢𑀺𑀧𑀻”𑀢𑀺𑁇
