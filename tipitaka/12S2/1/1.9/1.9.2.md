# 𑁧𑁇𑁯𑁇𑁨 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀏𑀓𑀸𑀤𑀲𑀓

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮

* 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀏𑀓𑀸𑀤𑀲𑀓

585\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑁂 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸𑁇 (𑁯𑁩--𑁧𑁦𑁩:**𑁩𑁬𑁪**--**𑁩𑁭𑁪**)

---

586\. (𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑁄𑁇 𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀺𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)



587\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀬𑁄𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂… 𑁇 (𑁧𑁦𑁪--𑁧𑁧𑁪:**𑁩𑁭𑁫**--**𑁩𑁮𑁫**)

588\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀙𑀦𑁆𑀤𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂… 𑁇 (𑁧𑁧𑁫--𑁧𑁨𑁫:**𑁩𑁮𑁬**--**𑁩𑁯𑁬**)

589\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀻 𑀓𑀭𑀡𑀻𑀬𑀸…𑀧𑁂… 𑁇 (𑁧𑁨𑁬--𑁧𑁩𑁬:**𑁩𑁯𑁭**--**𑁪𑁦𑁭**)

590\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀻 𑀓𑀭𑀡𑀻𑀬𑀸…𑀧𑁂… 𑁇 (𑁧𑁩𑁭--𑁧𑁪𑁭:**𑁪𑁦𑁮**--**𑁪𑁧𑁮**)

591\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑀡𑀻𑀬𑀁…𑀧𑁂… 𑁇 (𑁧𑁪𑁮--𑁧𑁫𑁮:**𑁪𑁧𑁯**--**𑁪𑁨𑁯**)

592\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀯𑀻𑀭𑀺𑀬𑀁 𑀓𑀭𑀡𑀻𑀬𑀁…𑀧𑁂… 𑁇 (𑁧𑁫𑁯--𑁧𑁬𑁯:**𑁪𑁩𑁦**--**𑁪𑁪𑁦**)

593\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀲𑀸𑀢𑀘𑁆𑀘𑀁 𑀓𑀭𑀡𑀻𑀬𑀁…𑀧𑁂… 𑁇 (𑁧𑁭𑁦--𑁧𑁮𑁦:**𑁪𑁪𑁧**--**𑁪𑁫𑁧**)

594\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀲𑀢𑀺 𑀓𑀭𑀡𑀻𑀬𑀸…𑀧𑁂… 𑁇 (𑁧𑁮𑁧--𑁧𑁯𑁧:**𑁪𑁫𑁨**--**𑁪𑁬𑁨**)

595\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀓𑀭𑀡𑀻𑀬𑀁…𑀧𑁂… 𑁇 (𑁧𑁯𑁨--𑁨𑁦𑁨:**𑁪𑁬𑁩**--**𑁪𑁭𑁩**)

596\. “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸…𑀧𑁂…  𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄 𑀓𑀭𑀡𑀻𑀬𑁄…𑀧𑁂… 𑁇 (𑁨𑁦𑁩--𑁨𑁧𑁩:**𑁪𑁭𑁪**--**𑁪𑁮𑁪**)

---

597\. 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑁄 𑀦𑀯𑀫𑁄𑁇



---

598\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



599\. _𑀲𑀢𑁆𑀣𑀸 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀬𑁄𑀕𑁄 𑀘,_  
_𑀙𑀦𑁆𑀤𑁄 𑀉𑀲𑁆𑀲𑁄𑀍𑀳𑀺𑀧𑀜𑁆𑀘𑀫𑀻;_  
_𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀺 𑀬𑀸𑀢𑀧𑁆𑀧𑀁,_  
_𑀯𑀻𑀭𑀺𑀬𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀫𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀲𑀢𑀺 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀢𑀺𑁇_  


---

600\. 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇



601\. _𑀧𑀭𑁂 𑀢𑁂 𑀤𑁆𑀯𑀸𑀤𑀲 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀲𑀼𑀢𑁆𑀢𑀸 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲 𑀲𑀢𑀸𑀦𑀺;_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑁂𑀦 𑀢𑁂 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀅𑀦𑁆𑀢𑀭𑀫𑁆𑀳𑀺 𑀬𑁂𑀢𑀺𑁇_  


---

602\. 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑁂𑀲𑀼 𑀉𑀤𑁆𑀤𑀸𑀦𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇



---

603\. 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
