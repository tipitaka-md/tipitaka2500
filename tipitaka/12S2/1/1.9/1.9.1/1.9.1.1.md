# 𑁧𑁇𑁯𑁇𑁧𑁇𑁧 𑀲𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

570\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑁂 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄; 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑁂 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄; 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁂 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄; 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 (𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁄 𑀏𑀓𑁄𑁇) (𑁮𑁨:**𑁩𑁫𑁩**)
