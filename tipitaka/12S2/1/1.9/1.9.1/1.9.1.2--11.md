# 𑁧𑁇𑁯𑁇𑁧𑁇𑁨--𑁧𑁧 𑀤𑀼𑀢𑀺𑀬𑀲𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀤𑀲𑀓

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮

* 𑀲𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺

* 𑀤𑀼𑀢𑀺𑀬𑀲𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀤𑀲𑀓

573\. “𑀚𑀸𑀢𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁮𑁩:**𑁩𑁫𑁪**)

574\. “𑀪𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁮𑁪:**𑁩𑁫𑁫**)

575\. “𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁮𑁫:**𑁩𑁫𑁬**)

576\. “𑀢𑀡𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁮𑁬:**𑁩𑁫𑁭**)

577\. “𑀯𑁂𑀤𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁮𑁭:**𑁩𑁫𑁮**)

578\. “𑀨𑀲𑁆𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁮𑁮:**𑁩𑁫𑁯**)

579\. “𑀲𑀍𑀆𑀬𑀢𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁮𑁯:**𑁩𑁬𑁦**)

580\. “𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁯𑁦:**𑁩𑁬𑁧**)

581\. “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 (𑁯𑁧:**𑁩𑁬𑁨**)

582\. “𑀲𑀗𑁆𑀔𑀸𑀭𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄; 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑁂 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁂 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀚𑀸𑀦𑀢𑀸 𑀅𑀧𑀲𑁆𑀲𑀢𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀜𑀸𑀡𑀸𑀬 𑀲𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 (𑁯𑁨:**𑁩𑁬𑁩**)
