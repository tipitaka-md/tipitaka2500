# 𑁧𑁇𑁩𑁇𑁫 𑀪𑀽𑀫𑀺𑀚𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀲𑀩𑀮𑀯𑀕𑁆𑀕

* 𑀪𑀽𑀫𑀺𑀚𑀲𑀼𑀢𑁆𑀢

(25.)

165\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀽𑀫𑀺𑀚𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀽𑀫𑀺𑀚𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

166\. “𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀧𑀭𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀇𑀥 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀪𑀕𑀯𑀸 𑀓𑀺𑀁𑀯𑀸𑀤𑀻 𑀓𑀺𑀫𑀓𑁆𑀔𑀸𑀬𑀻, 𑀓𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑀸 𑀘 𑀫𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀤𑀺𑀦𑁄 𑀘𑁂𑀯 𑀪𑀕𑀯𑀢𑁄 𑀅𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸𑀫, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀘 𑀓𑁄𑀘𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀯𑀸𑀤𑀸𑀦𑀼𑀧𑀸𑀢𑁄 𑀕𑀸𑀭𑀬𑁆𑀳𑀁 𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸”𑀢𑀺?

167\. “𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸𑁇 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘? 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑁇 𑀇𑀢𑀺 𑀯𑀤𑀁 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀤𑀻 𑀘𑁂𑀯 𑀪𑀕𑀯𑀢𑁄 𑀅𑀲𑁆𑀲, 𑀦 𑀘 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀓𑁄𑀘𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀯𑀸𑀤𑀸𑀦𑀼𑀧𑀸𑀢𑁄 𑀕𑀸𑀭𑀬𑁆𑀳𑀁 𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇

168\. 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀢𑀤𑀧𑀺 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀢𑀤𑀧𑀺 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑁇

169\. 𑀢𑀢𑁆𑀭𑀸𑀯𑀼𑀲𑁄, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀯𑀢 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀨𑀲𑁆𑀲𑀸 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀯𑀢 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀨𑀲𑁆𑀲𑀸 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

170\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀪𑀽𑀫𑀺𑀚𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀇𑀫𑀁 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑀸𑀯𑀢𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀪𑀽𑀫𑀺𑀚𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇

171\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀣𑀸 𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀬𑀸𑁇 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘? 𑀨𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑁇 𑀇𑀢𑀺 𑀯𑀤𑀁 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀤𑀻 𑀘𑁂𑀯 𑀫𑁂 𑀅𑀲𑁆𑀲, 𑀦 𑀘 𑀫𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀓𑁄𑀘𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀯𑀸𑀤𑀸𑀦𑀼𑀧𑀸𑀢𑁄 𑀕𑀸𑀭𑀬𑁆𑀳𑀁 𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇

172\. 𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀢𑀤𑀧𑀺 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀢𑀤𑀧𑀺 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑁇

173\. 𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀯𑀢 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀨𑀲𑁆𑀲𑀸 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂…𑀧𑁂…  𑀬𑁂𑀧𑀺 𑀢𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀫𑁆𑀫𑀯𑀸𑀤𑀸 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀯𑀢 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀨𑀲𑁆𑀲𑀸 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

174\. 𑀓𑀸𑀬𑁂 𑀯𑀸 𑀳𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑀺 𑀓𑀸𑀬𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑁂𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀳𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑀺 𑀯𑀘𑀻𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑁂𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀫𑀦𑁂 𑀯𑀸 𑀳𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑀺 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑁂𑀢𑀼 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑁇

175\. 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺, 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀧𑀭𑁂 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺, 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇

176\. 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀧𑀭𑁂 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤…𑀧𑁂…  𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇

177\. 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀦𑁄𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀧𑀭𑁂 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀦𑁄𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤…𑀧𑁂…  𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀸 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀦𑁄𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇

178\. 𑀇𑀫𑁂𑀲𑀼, 𑀆𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑀸𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑁄 𑀓𑀸𑀬𑁄 𑀦 𑀳𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑀸 𑀯𑀸𑀘𑀸 𑀦 𑀳𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑁄 𑀫𑀦𑁄 𑀦 𑀳𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀔𑁂𑀢𑁆𑀢𑀁 𑀢𑀁 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼 𑀢𑀁 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀢𑀦𑀁 𑀢𑀁 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀢𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀬𑀁𑀧𑀘𑁆𑀘𑀬𑀸𑀲𑁆𑀲 𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺𑁇
