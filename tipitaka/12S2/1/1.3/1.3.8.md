# 𑁧𑁇𑁩𑁇𑁮 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀲𑀩𑀮𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(28.)

192\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀢𑀢𑁆𑀭 𑀔𑁄…𑀧𑁂…  𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  𑀪𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀯𑁂𑀤𑀦𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀨𑀲𑁆𑀲𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

193\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀭𑀸𑀫𑀭𑀡𑀁? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀺 𑀢𑀫𑁆𑀳𑀺 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑁂 𑀚𑀭𑀸 𑀚𑀻𑀭𑀡𑀢𑀸 𑀔𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀧𑀸𑀮𑀺𑀘𑁆𑀘𑀁 𑀯𑀮𑀺𑀢𑁆𑀢𑀘𑀢𑀸 𑀆𑀬𑀼𑀦𑁄 𑀲𑀁𑀳𑀸𑀦𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀸𑀓𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀚𑀭𑀸𑁇 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀸 𑀢𑀫𑁆𑀳𑀸 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀸 𑀘𑀼𑀢𑀺 𑀘𑀯𑀦𑀢𑀸 𑀪𑁂𑀤𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀁 𑀫𑀘𑁆𑀘𑀼 𑀫𑀭𑀡𑀁 𑀓𑀸𑀮𑀓𑀺𑀭𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀪𑁂𑀤𑁄 𑀓𑀍𑀏𑀯𑀭𑀲𑁆𑀲 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁄; 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀫𑀭𑀡𑀁𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀚𑀭𑀸 𑀇𑀤𑀜𑁆𑀘 𑀫𑀭𑀡𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀚𑀸𑀢𑀺𑀲𑀫𑀼𑀤𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑁄; 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇

194\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑁄…  𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…  𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀡𑁆𑀳𑀸…  𑀯𑁂𑀤𑀦𑀸…  𑀨𑀲𑁆𑀲𑁄…  𑀲𑀍𑀆𑀬𑀢𑀦𑀁…  𑀦𑀸𑀫𑀭𑀽𑀧𑀁…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁… 𑁇

195\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸? 𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸—  𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀫𑀼𑀤𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑁄; 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇

196\. 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀏𑀯𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀏𑀯𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀏𑀯𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀏𑀯𑀁 𑀚𑀸𑀢𑀺𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  𑀪𑀯𑀁…  𑀉𑀧𑀸𑀤𑀸𑀦𑀁…  𑀢𑀡𑁆𑀳𑀁…  𑀯𑁂𑀤𑀦𑀁…  𑀨𑀲𑁆𑀲𑀁…  𑀲𑀍𑀆𑀬𑀢𑀦𑀁…  𑀦𑀸𑀫𑀭𑀽𑀧𑀁…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀁…  𑀏𑀯𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀤𑀲𑁆𑀲𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀆𑀕𑀢𑁄 𑀇𑀫𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀧𑀲𑁆𑀲𑀢𑀺 𑀇𑀫𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀲𑁂𑀓𑁆𑀔𑀸𑀬 𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀥𑀫𑁆𑀫𑀲𑁄𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀅𑀭𑀺𑀬𑁄 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀅𑀫𑀢𑀤𑁆𑀯𑀸𑀭𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀇𑀢𑀺𑀧𑀻”𑀢𑀺𑁇

---

197\. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
