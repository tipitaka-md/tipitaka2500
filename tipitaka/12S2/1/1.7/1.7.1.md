# 𑁧𑁇𑁭𑁇𑁧 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸𑀲𑀼𑀢𑁆𑀢

(61.)

436. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂…  “𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀧𑀺 𑀯𑀺𑀭𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀧𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬𑀧𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀲𑁆𑀲 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀆𑀘𑀬𑁄𑀧𑀺 𑀅𑀧𑀘𑀬𑁄𑀧𑀺 𑀆𑀤𑀸𑀦𑀫𑁆𑀧𑀺 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀦𑀫𑁆𑀧𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑀢𑁆𑀭𑀸𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀧𑀺 𑀯𑀺𑀭𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀧𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬𑀧𑀺𑁇

437. 𑀬𑀜𑁆𑀘 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀫𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀢𑀢𑁆𑀭𑀸𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀦𑀸𑀮𑀁 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀁 𑀦𑀸𑀮𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀺𑀢𑀼𑀁 𑀦𑀸𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀁 𑀫𑀫𑀸𑀬𑀺𑀢𑀁 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀁—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑀢𑁆𑀭𑀸𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀦𑀸𑀮𑀁 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀁 𑀦𑀸𑀮𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀺𑀢𑀼𑀁 𑀦𑀸𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀁𑁇

438. 𑀯𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀇𑀫𑀁 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑀁 𑀓𑀸𑀬𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀤𑀺𑀲𑁆𑀲𑀢𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑁄 𑀓𑀸𑀬𑁄 𑀏𑀓𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀁 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀤𑁆𑀯𑁂𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀢𑀻𑀡𑀺𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀘𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀤𑀲𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀯𑀻𑀲𑀢𑀺𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀢𑀺𑀁𑀲𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀫𑁆𑀧𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄, 𑀪𑀺𑀬𑁆𑀬𑁄𑀧𑀺 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄𑁇

439. 𑀬𑀜𑁆𑀘 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀫𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀘 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀘 𑀅𑀜𑁆𑀜𑀤𑁂𑀯 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀓𑁆𑀓𑀝𑁄 𑀅𑀭𑀜𑁆𑀜𑁂 𑀧𑀯𑀦𑁂 𑀘𑀭𑀫𑀸𑀦𑁄 𑀲𑀸𑀔𑀁 𑀕𑀡𑁆𑀳𑀢𑀺, 𑀢𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀕𑀡𑁆𑀳𑀢𑀺, 𑀢𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀕𑀡𑁆𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀫𑀺𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀫𑀦𑁄 𑀇𑀢𑀺𑀧𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀢𑀺𑀧𑀺, 𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀘 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀘 𑀅𑀜𑁆𑀜𑀤𑁂𑀯 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

440. 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁𑀬𑁂𑀯 𑀲𑀸𑀥𑀼𑀓𑀁 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺—  ‘𑀇𑀢𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀤𑀁 𑀳𑁄𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀇𑀤𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀇𑀤𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀇𑀤𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺—  𑀬𑀤𑀺𑀤𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇

441. 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺; 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀢𑀺, 𑀯𑀺𑀭𑀸𑀕𑀸 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
