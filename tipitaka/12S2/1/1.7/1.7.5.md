# 𑁧𑁇𑁭𑁇𑁫 𑀦𑀕𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀦𑀕𑀭𑀲𑀼𑀢𑁆𑀢

(65.)

466\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀢𑀸𑀬𑀁 𑀮𑁄𑀓𑁄 𑀆𑀧𑀦𑁆𑀦𑁄 𑀚𑀸𑀬𑀢𑀺 𑀘 𑀚𑀻𑀬𑀢𑀺 𑀘 𑀫𑀻𑀬𑀢𑀺 𑀘 𑀘𑀯𑀢𑀺 𑀘 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑁇 𑀅𑀣 𑀘 𑀧𑀦𑀺𑀫𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲𑁇 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀦𑀸𑀫 𑀇𑀫𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲𑀸’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀁𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀳𑀼 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀪𑀺𑀲𑀫𑀬𑁄—  ‘𑀚𑀸𑀢𑀺𑀬𑀸 𑀔𑁄 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺𑁇

467\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀪𑀯𑁄 𑀳𑁄𑀢𑀺…  𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀢𑀡𑁆𑀳𑀸 𑀳𑁄𑀢𑀺…  𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺…  𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…  𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀺…  𑀓𑀺𑀁𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀳𑀼 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀪𑀺𑀲𑀫𑀬𑁄—  ‘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀔𑁄 𑀲𑀢𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀁𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀳𑀼 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀪𑀺𑀲𑀫𑀬𑁄—  ‘𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀔𑁄 𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺𑁇

468\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑀢𑀺 𑀔𑁄 𑀇𑀤𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀫𑁆𑀳𑀸 𑀦 𑀧𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀚𑀸𑀬𑁂𑀣 𑀯𑀸 𑀚𑀻𑀬𑁂𑀣 𑀯𑀸 𑀫𑀻𑀬𑁂𑀣 𑀯𑀸 𑀘𑀯𑁂𑀣 𑀯𑀸 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀣 𑀯𑀸, 𑀬𑀤𑀺𑀤𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁; 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁; 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 ‘𑀲𑀫𑀼𑀤𑀬𑁄, 𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀔𑁄 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

469\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀲𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀓𑀺𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀳𑀼 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀪𑀺𑀲𑀫𑀬𑁄—  ‘𑀚𑀸𑀢𑀺𑀬𑀸 𑀔𑁄 𑀅𑀲𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀪𑀯𑁄 𑀦 𑀳𑁄𑀢𑀺…  𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺…  𑀢𑀡𑁆𑀳𑀸 𑀦 𑀳𑁄𑀢𑀺…  𑀯𑁂𑀤𑀦𑀸 𑀦 𑀳𑁄𑀢𑀺…  𑀨𑀲𑁆𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺…  𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺…  𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀓𑀺𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀳𑀼 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀪𑀺𑀲𑀫𑀬𑁄—  ‘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀔𑁄 𑀅𑀲𑀢𑀺, 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺𑁇

470\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀓𑀺𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀅𑀳𑀼 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀪𑀺𑀲𑀫𑀬𑁄—  ‘𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀔𑁄 𑀅𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺𑁇

471\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  𑀅𑀥𑀺𑀕𑀢𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀫𑀕𑁆𑀕𑁄 𑀩𑁄𑀥𑀸𑀬 𑀬𑀤𑀺𑀤𑀁—  𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄; 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄; 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 ‘𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀔𑁄 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺 𑀧𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀤𑀧𑀸𑀤𑀺 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

472\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀭𑀜𑁆𑀜𑁂 𑀧𑀯𑀦𑁂 𑀘𑀭𑀫𑀸𑀦𑁄 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬 𑀧𑀼𑀭𑀸𑀡𑀁 𑀫𑀕𑁆𑀕𑀁 𑀧𑀼𑀭𑀸𑀡𑀜𑁆𑀚𑀲𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀅𑀦𑀼𑀬𑀸𑀢𑀁𑁇 𑀲𑁄 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬 𑀧𑀼𑀭𑀸𑀡𑀁 𑀦𑀕𑀭𑀁 𑀧𑀼𑀭𑀸𑀡𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀅𑀚𑁆𑀛𑀸𑀯𑀼𑀝𑁆𑀞𑀁 𑀆𑀭𑀸𑀫𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀸𑀧𑀯𑀦𑁆𑀢𑀁 𑀭𑀫𑀡𑀻𑀬𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀜𑁆𑀜𑁄 𑀯𑀸 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬—  ‘𑀬𑀕𑁆𑀖𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺—  𑀅𑀳𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀧𑀯𑀦𑁂 𑀘𑀭𑀫𑀸𑀦𑁄 𑀧𑀼𑀭𑀸𑀡𑀁 𑀫𑀕𑁆𑀕𑀁 𑀧𑀼𑀭𑀸𑀡𑀜𑁆𑀚𑀲𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀅𑀦𑀼𑀬𑀸𑀢𑀁 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀺𑀁𑁇 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀧𑀼𑀭𑀸𑀡𑀁 𑀦𑀕𑀭𑀁 𑀧𑀼𑀭𑀸𑀡𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀅𑀚𑁆𑀛𑀸𑀯𑀼𑀝𑁆𑀞𑀁 𑀆𑀭𑀸𑀫𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀤𑁆𑀥𑀸𑀧𑀯𑀦𑁆𑀢𑀁 𑀭𑀫𑀡𑀻𑀬𑀁𑁇 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀕𑀭𑀁 𑀫𑀸𑀧𑁂𑀳𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀚𑀸 𑀯𑀸 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀯𑀸 𑀢𑀁 𑀦𑀕𑀭𑀁 𑀫𑀸𑀧𑁂𑀬𑁆𑀬𑁇 𑀢𑀤𑀲𑁆𑀲 𑀦𑀕𑀭𑀁 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀯𑀼𑀤𑁆𑀥𑀺𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀤𑀲𑀁 𑀧𑀼𑀭𑀸𑀡𑀁 𑀫𑀕𑁆𑀕𑀁 𑀧𑀼𑀭𑀸𑀡𑀜𑁆𑀚𑀲𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀅𑀦𑀼𑀬𑀸𑀢𑀁𑁇

473\. 𑀓𑀢𑀫𑁄 𑀘 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀸𑀡𑀫𑀕𑁆𑀕𑁄 𑀧𑀼𑀭𑀸𑀡𑀜𑁆𑀚𑀲𑁄 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀅𑀦𑀼𑀬𑀸𑀢𑁄? 𑀅𑀬𑀫𑁂𑀯 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀸𑀡𑀫𑀕𑁆𑀕𑁄 𑀧𑀼𑀭𑀸𑀡𑀜𑁆𑀚𑀲𑁄 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀅𑀦𑀼𑀬𑀸𑀢𑁄, 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀺𑀁; 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁; 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁; 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁; 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀺𑀁; 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀚𑀸𑀢𑀺𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…𑀧𑁂…  𑀪𑀯𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…  𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…  𑀢𑀡𑁆𑀳𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…  𑀯𑁂𑀤𑀦𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…  𑀨𑀲𑁆𑀲𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…  𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…  𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀺𑀁; 𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁; 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁; 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀢𑀤𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁𑁇 𑀢𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀦𑁆”𑀢𑀺𑁇

---
