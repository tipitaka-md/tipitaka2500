# 𑁧𑁇𑁭𑁇𑁬 𑀲𑀫𑁆𑀫𑀲𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀲𑀫𑁆𑀫𑀲𑀲𑀼𑀢𑁆𑀢

(66.)

475\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑀼𑀭𑀽𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀲𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀫 𑀓𑀼𑀭𑀽𑀦𑀁 𑀦𑀺𑀕𑀫𑁄𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀫𑁆𑀫𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀦𑁆”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑁆𑀫𑀲𑀸𑀫𑀺 𑀅𑀦𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀦𑁆”𑀢𑀺𑁇 “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀫𑁆𑀫𑀲𑀲𑀺 𑀅𑀦𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑁇 𑀬𑀣𑀸 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀆𑀭𑀸𑀥𑁂𑀲𑀺𑁇

476\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀢𑀲𑁆𑀲, 𑀪𑀕𑀯𑀸, 𑀓𑀸𑀮𑁄; 𑀏𑀢𑀲𑁆𑀲, 𑀲𑀼𑀕𑀢, 𑀓𑀸𑀮𑁄; 𑀬𑀁 𑀪𑀕𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀁 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦𑀳𑀸𑀦𑀦𑁆𑀤, 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

477\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑁆𑀫𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑁆𑀫𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀁—  ‘𑀬𑀁 𑀔𑁄 𑀇𑀤𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀥𑀁 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀇𑀤𑀁 𑀔𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀲𑀫𑁆𑀫𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑀁 𑀔𑁄 𑀇𑀤𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀥𑀁 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀇𑀤𑀁 𑀔𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑀥𑀺𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀧𑀥𑀺𑀲𑀫𑀼𑀤𑀬𑀁 𑀉𑀧𑀥𑀺𑀚𑀸𑀢𑀺𑀓𑀁 𑀉𑀧𑀥𑀺𑀧𑀪𑀯𑀁, 𑀉𑀧𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀧𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀬𑀸 𑀘 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀣𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀸𑀬𑁇

478\. 𑀅𑀣𑀸𑀧𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑁆𑀫𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀁—  ‘𑀉𑀧𑀥𑀺 𑀧𑀦𑀸𑀬𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑁄 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑁄 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑁄 𑀓𑀺𑀁𑀧𑀪𑀯𑁄, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀉𑀧𑀥𑀺 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀉𑀧𑀥𑀺 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀲𑀫𑁆𑀫𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀉𑀧𑀥𑀺 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀤𑀸𑀦𑁄 𑀢𑀡𑁆𑀳𑀸𑀲𑀫𑀼𑀤𑀬𑁄 𑀢𑀡𑁆𑀳𑀸𑀚𑀸𑀢𑀺𑀓𑁄 𑀢𑀡𑁆𑀳𑀸𑀧𑀪𑀯𑁄, 𑀢𑀡𑁆𑀳𑀸𑀬 𑀲𑀢𑀺 𑀉𑀧𑀥𑀺 𑀳𑁄𑀢𑀺, 𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀲𑀢𑀺 𑀉𑀧𑀥𑀺 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀉𑀧𑀥𑀺𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀉𑀧𑀥𑀺𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀉𑀧𑀥𑀺𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀬𑀸 𑀘 𑀉𑀧𑀥𑀺𑀦𑀺𑀭𑁄𑀥𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀣𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀉𑀧𑀥𑀺𑀦𑀺𑀭𑁄𑀥𑀸𑀬𑁇

479\. 𑀅𑀣𑀸𑀧𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑁆𑀫𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀲𑀁—  ‘𑀢𑀡𑁆𑀳𑀸 𑀧𑀦𑀸𑀬𑀁 𑀓𑀢𑁆𑀣 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀓𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀲𑀫𑁆𑀫𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀬𑀁 𑀔𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁, 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑁄𑀢𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁…  𑀓𑀸𑀬𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁…  𑀫𑀦𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

480\. 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀢𑀁 𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀲𑀼𑀔𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀆𑀭𑁄𑀕𑁆𑀬𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀔𑁂𑀫𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁𑁇 𑀢𑁂 𑀢𑀡𑁆𑀳𑀁 𑀯𑀟𑁆𑀠𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀢𑀡𑁆𑀳𑀁 𑀯𑀟𑁆𑀠𑁂𑀲𑀼𑀁 𑀢𑁂 𑀉𑀧𑀥𑀺𑀁 𑀯𑀟𑁆𑀠𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀉𑀧𑀥𑀺𑀁 𑀯𑀟𑁆𑀠𑁂𑀲𑀼𑀁 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀟𑁆𑀠𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀟𑁆𑀠𑁂𑀲𑀼𑀁 𑀢𑁂 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

485\. 𑀬𑁂𑀧𑀺 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀢𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀕𑀢𑁄 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀪𑀬𑀢𑁄 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

486\. 𑀬𑁂𑀧𑀺 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀢𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀕𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀪𑀬𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑁂 𑀉𑀧𑀥𑀺𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀉𑀧𑀥𑀺𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑁂 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

487\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀸𑀦𑀻𑀬𑀓𑀁𑀲𑁄 𑀯𑀡𑁆𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀕𑀦𑁆𑀥𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀭𑀲𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀲𑁄 𑀘 𑀔𑁄 𑀯𑀺𑀲𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑁄𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀖𑀫𑁆𑀫𑀸𑀪𑀺𑀢𑀢𑁆𑀢𑁄 𑀖𑀫𑁆𑀫𑀧𑀭𑁂𑀢𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀢𑀲𑀺𑀢𑁄 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀆𑀧𑀸𑀦𑀻𑀬𑀓𑀁𑀲𑁄 𑀯𑀡𑁆𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀕𑀦𑁆𑀥𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀭𑀲𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑁄 𑀘 𑀔𑁄 𑀯𑀺𑀲𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑁄𑁇 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀲𑀺 𑀧𑀺𑀯𑁇 𑀧𑀺𑀯𑀢𑁄 𑀳𑀺 𑀔𑁄 𑀢𑀁 𑀙𑀸𑀤𑁂𑀲𑁆𑀲𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦𑀧𑀺 𑀕𑀦𑁆𑀥𑁂𑀦𑀧𑀺 𑀭𑀲𑁂𑀦𑀧𑀺; 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘 𑀧𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀲𑀓𑁆𑀓𑀸 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀲𑀼𑀭𑀸𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑀸 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑀺𑀦𑁂𑀢𑀼𑀁 𑀤𑀥𑀺𑀫𑀡𑁆𑀟𑀓𑁂𑀦 𑀯𑀸 𑀯𑀺𑀦𑁂𑀢𑀼𑀁 𑀪𑀝𑁆𑀞𑀮𑁄𑀡𑀺𑀓𑀸𑀬 𑀯𑀸 𑀯𑀺𑀦𑁂𑀢𑀼𑀁 𑀮𑁄𑀡𑀲𑁄𑀯𑀻𑀭𑀓𑁂𑀦 𑀯𑀸 𑀯𑀺𑀦𑁂𑀢𑀼𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯𑀸𑀳𑀁 𑀢𑀁 𑀧𑀺𑀯𑁂𑀬𑁆𑀬𑀁, 𑀬𑀁 𑀫𑀫 𑀅𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀧𑀸𑀦𑀻𑀬𑀓𑀁𑀲𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀦 𑀧𑀺𑀯𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀲𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀦 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀢𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀭𑁄𑀕𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁 𑀪𑀬𑀢𑁄 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁, 𑀢𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀺𑀁𑀲𑀼 𑀢𑁂 𑀉𑀧𑀥𑀺𑀁 𑀧𑀚𑀳𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀉𑀧𑀥𑀺𑀁 𑀧𑀚𑀳𑀺𑀁𑀲𑀼 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀚𑀳𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀚𑀳𑀺𑀁𑀲𑀼 𑀢𑁂 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

488\. 𑀬𑁂𑀧𑀺 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀢𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀕𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀪𑀬𑀢𑁄 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑀡𑁆𑀳𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑁂 𑀉𑀧𑀥𑀺𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀉𑀧𑀥𑀺𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑁂 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺, 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀻”𑀢𑀺𑁇
