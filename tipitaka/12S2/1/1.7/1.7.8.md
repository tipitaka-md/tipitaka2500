# 𑁧𑁇𑁭𑁇𑁮 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀲𑀼𑀢𑁆𑀢

(68.)

500\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀼𑀲𑀺𑀮𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀸𑀭𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀆𑀦𑀦𑁆𑀤𑁄 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀼𑀲𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀼𑀲𑀺𑀮, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀼𑀲𑀺𑀮𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’”𑀢𑀺𑁇

501\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀼𑀲𑀺𑀮, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀼𑀲𑀺𑀮𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀻𑀢𑀺…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄𑀢𑀺…  𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆𑀢𑀺…  𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸𑀢𑀺…  𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀢𑀺…  𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄𑀢𑀺…  𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’”𑀢𑀺𑁇

502\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀼𑀲𑀺𑀮, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀼𑀲𑀺𑀮𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺𑁇

503\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀼𑀲𑀺𑀮, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀼𑀲𑀺𑀮𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺𑁇

504\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀫𑀼𑀲𑀺𑀮, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀼𑀲𑀺𑀮𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’”𑀢𑀺𑁇

505\. “𑀢𑁂𑀦𑀳𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀼𑀲𑀺𑀮𑁄 𑀅𑀭𑀳𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀼𑀲𑀺𑀮𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

506\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀭𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀅𑀳𑀁 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀮𑀪𑁂𑀬𑁆𑀬𑀁𑁇 𑀫𑀁 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑁇 𑀅𑀳𑀁 𑀢𑁂 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀮𑀪𑀢𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀭𑀤𑁄 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁𑁇 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀭𑀤𑀁 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁𑁇 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀼 𑀘 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀭𑀤𑁄 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁𑁇

507\. 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀦𑀸𑀭𑀤, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’”𑀢𑀺𑁇

508\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀦𑀸𑀭𑀤, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’”𑀢𑀺𑁇

509\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀦𑀸𑀭𑀤, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺𑁇

510\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀦𑀸𑀭𑀤, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁄’”𑀢𑀺𑁇

511\. “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀦𑀸𑀭𑀤, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑀸𑀭𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀫𑁂𑀯 𑀜𑀸𑀡𑀁—  ‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀲𑀤𑁆𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀭𑀼𑀘𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀫𑁂𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’”𑀢𑀺𑁇

512\. “𑀢𑁂𑀦𑀳𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀭𑀤𑁄 𑀅𑀭𑀳𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄”𑀢𑀺? “‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’𑀢𑀺 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀁, 𑀦 𑀘𑀫𑁆𑀳𑀺 𑀅𑀭𑀳𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀦𑁆𑀢𑀸𑀭𑀫𑀕𑁆𑀕𑁂 𑀉𑀤𑀧𑀸𑀦𑁄, 𑀢𑀢𑁆𑀭 𑀦𑁂𑀯𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀼 𑀦 𑀉𑀤𑀓𑀯𑀸𑀭𑀓𑁄𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀖𑀫𑁆𑀫𑀸𑀪𑀺𑀢𑀢𑁆𑀢𑁄 𑀖𑀫𑁆𑀫𑀧𑀭𑁂𑀢𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀢𑀲𑀺𑀢𑁄 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄, 𑀲𑁄 𑀢𑀁 𑀉𑀤𑀧𑀸𑀦𑀁 𑀑𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀲 ‘𑀉𑀤𑀓𑀦𑁆’𑀢𑀺 𑀳𑀺 𑀔𑁄 𑀜𑀸𑀡𑀁 𑀅𑀲𑁆𑀲, 𑀦 𑀘 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, ‘𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀁, 𑀦 𑀘𑀫𑁆𑀳𑀺 𑀅𑀭𑀳𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑁄”𑀢𑀺𑁇

513\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀯𑀺𑀝𑁆𑀞, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀭𑀤𑀁 𑀓𑀺𑀁 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀁𑀯𑀸𑀤𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀭𑀤𑀁 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑀤𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀼𑀲𑀮𑀸”𑀢𑀺𑁇

---
