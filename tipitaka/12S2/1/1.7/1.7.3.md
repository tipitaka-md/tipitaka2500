# 𑁧𑁇𑁭𑁇𑁩 𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑀲𑀼𑀢𑁆𑀢

(63.)

449\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁…  “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑀸 𑀪𑀽𑀢𑀸𑀦𑀁 𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀲𑀫𑁆𑀪𑀯𑁂𑀲𑀻𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀑𑀍𑀆𑀭𑀺𑀓𑁄 𑀯𑀸 𑀲𑀼𑀔𑀼𑀫𑁄 𑀯𑀸, 𑀨𑀲𑁆𑀲𑁄 𑀤𑀼𑀢𑀺𑀬𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀢𑀢𑀺𑀬𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸 𑀪𑀽𑀢𑀸𑀦𑀁 𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀲𑀫𑁆𑀪𑀯𑁂𑀲𑀻𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇

450\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀚𑀸𑀬𑀫𑁆𑀧𑀢𑀺𑀓𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀩𑀮𑀁 𑀆𑀤𑀸𑀬 𑀓𑀦𑁆𑀢𑀸𑀭𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀢𑁂𑀲𑀫𑀲𑁆𑀲 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀚𑀸𑀬𑀫𑁆𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀓𑀦𑁆𑀢𑀸𑀭𑀕𑀢𑀸𑀦𑀁 𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀩𑀮𑀫𑀢𑁆𑀢𑀸, 𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀲𑀺𑀬𑀸 𑀘 𑀦𑁂𑀲𑀁 𑀓𑀦𑁆𑀢𑀸𑀭𑀸𑀯𑀲𑁂𑀲𑁄 𑀅𑀦𑀢𑀺𑀡𑁆𑀡𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀚𑀸𑀬𑀫𑁆𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀔𑁄 𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀩𑀮𑀫𑀢𑁆𑀢𑀸 𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘𑀸𑀬𑀁 𑀓𑀦𑁆𑀢𑀸𑀭𑀸𑀯𑀲𑁂𑀲𑁄 𑀅𑀦𑀺𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀇𑀫𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑀁 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀮𑁆𑀮𑀽𑀭𑀜𑁆𑀘 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀜𑁆𑀘 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑀸𑀦𑀺 𑀔𑀸𑀤𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀢𑀁 𑀓𑀦𑁆𑀢𑀸𑀭𑀸𑀯𑀲𑁂𑀲𑀁 𑀦𑀺𑀢𑁆𑀣𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀫𑀸 𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑀬𑁄 𑀯𑀺𑀦𑀲𑁆𑀲𑀺𑀫𑁆𑀳𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀚𑀸𑀬𑀫𑁆𑀧𑀢𑀺𑀓𑀸 𑀢𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑀁 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀮𑁆𑀮𑀽𑀭𑀜𑁆𑀘 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀜𑁆𑀘 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑀸𑀦𑀺 𑀔𑀸𑀤𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀢𑀁 𑀓𑀦𑁆𑀢𑀸𑀭𑀸𑀯𑀲𑁂𑀲𑀁 𑀦𑀺𑀢𑁆𑀣𑀭𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑀸𑀦𑀺 𑀘𑁂𑀯 𑀔𑀸𑀤𑁂𑀬𑁆𑀬𑀼𑀁, 𑀉𑀭𑁂 𑀘 𑀧𑀝𑀺𑀧𑀺𑀲𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀳𑀁, 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓, 𑀓𑀳𑀁, 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑀸’𑀢𑀺𑁇

451\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁂 𑀤𑀯𑀸𑀬 𑀯𑀸 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀫𑀤𑀸𑀬 𑀯𑀸 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀫𑀡𑁆𑀟𑀦𑀸𑀬 𑀯𑀸 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀯𑀺𑀪𑀽𑀲𑀦𑀸𑀬 𑀯𑀸 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀦𑀦𑀼 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀓𑀦𑁆𑀢𑀸𑀭𑀲𑁆𑀲 𑀦𑀺𑀢𑁆𑀣𑀭𑀡𑀢𑁆𑀣𑀸𑀬 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “‘𑀏𑀯𑀫𑁂𑀯 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀺𑀓𑁄 𑀭𑀸𑀕𑁄 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀺𑀓𑁂 𑀭𑀸𑀕𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀦𑀢𑁆𑀣𑀺 𑀢𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀬𑁂𑀦 𑀲𑀁𑀬𑁄𑀚𑀦𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀼𑀦 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇

452\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀸𑀯𑀻 𑀦𑀺𑀘𑁆𑀘𑀫𑁆𑀫𑀸 𑀓𑀼𑀝𑁆𑀝𑀜𑁆𑀘𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑁇 𑀬𑁂 𑀓𑀼𑀝𑁆𑀝𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀦𑀁 𑀔𑀸𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀭𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬, 𑀬𑁂 𑀭𑀼𑀓𑁆𑀔𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀦𑀁 𑀔𑀸𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀉𑀤𑀓𑀜𑁆𑀘𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬, 𑀬𑁂 𑀉𑀤𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀦𑀁 𑀔𑀸𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀆𑀓𑀸𑀲𑀜𑁆𑀘𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬, 𑀬𑁂 𑀆𑀓𑀸𑀲𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀦𑀁 𑀔𑀸𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑀁 𑀬𑀤𑁂𑀯 𑀳𑀺 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀸𑀯𑀻 𑀦𑀺𑀘𑁆𑀘𑀫𑁆𑀫𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬, 𑀬𑁂 𑀢𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀦𑀁 𑀔𑀸𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀨𑀲𑁆𑀲𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀲𑀼 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀓𑀭𑀡𑀻𑀬𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

453\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀼 𑀲𑀸𑀥𑀺𑀓𑀧𑁄𑀭𑀺𑀲𑀸 𑀧𑀼𑀡𑁆𑀡𑀸 𑀅𑀗𑁆𑀕𑀸𑀭𑀸𑀦𑀁 𑀯𑀻𑀢𑀘𑁆𑀘𑀺𑀓𑀸𑀦𑀁 𑀯𑀻𑀢𑀥𑀽𑀫𑀸𑀦𑀁𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀚𑀻𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀫𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀼𑀔𑀓𑀸𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑁆𑀯𑁂 𑀩𑀮𑀯𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀸 𑀦𑀸𑀦𑀸𑀩𑀸𑀳𑀸𑀲𑀼 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀼𑀁 𑀉𑀧𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀆𑀭𑀓𑀸𑀯𑀲𑁆𑀲 𑀘𑁂𑀢𑀦𑀸 𑀆𑀭𑀓𑀸 𑀧𑀢𑁆𑀣𑀦𑀸 𑀆𑀭𑀓𑀸 𑀧𑀡𑀺𑀥𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀏𑀯𑀜𑁆𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀜𑁆𑀘𑀸𑀳𑀁 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀼𑀁 𑀧𑀧𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀻𑀲𑀼 𑀢𑀡𑁆𑀳𑀸𑀲𑀼 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀲𑀼 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀓𑀭𑀡𑀻𑀬𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

454\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁄𑀭𑀁 𑀆𑀕𑀼𑀘𑀸𑀭𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀤𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀤𑁂𑀯, 𑀘𑁄𑀭𑁄 𑀆𑀕𑀼𑀘𑀸𑀭𑀻, 𑀇𑀫𑀲𑁆𑀲 𑀬𑀁 𑀇𑀘𑁆𑀙𑀲𑀺 𑀢𑀁 𑀤𑀡𑁆𑀟𑀁 𑀧𑀡𑁂𑀳𑀻’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀕𑀘𑁆𑀙𑀣, 𑀪𑁄, 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑀣𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀅𑀣 𑀭𑀸𑀚𑀸 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀫𑀬𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀓𑀣𑀁 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄’𑀢𑀺? ‘𑀢𑀣𑁂𑀯, 𑀤𑁂𑀯, 𑀚𑀻𑀯𑀢𑀻’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀕𑀘𑁆𑀙𑀣, 𑀪𑁄, 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑀣𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀅𑀣 𑀭𑀸𑀚𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀓𑀣𑀁 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄’𑀢𑀺? ‘𑀢𑀣𑁂𑀯, 𑀤𑁂𑀯, 𑀚𑀻𑀯𑀢𑀻’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀕𑀘𑁆𑀙𑀣, 𑀪𑁄, 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑀣𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀦 𑀳𑀦𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀯𑀲𑀁 𑀢𑀻𑀳𑀺 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀳𑀺 𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀏𑀓𑀺𑀲𑁆𑀲𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀬𑁂𑀣; 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀢𑀻𑀳𑀺 𑀲𑀢𑁆𑀢𑀺𑀲𑀢𑁂𑀳𑀺 𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀳𑀸𑀭𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀓𑀭𑀡𑀻𑀬𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀻”𑀢𑀺𑁇
