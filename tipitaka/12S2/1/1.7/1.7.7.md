# 𑁧𑁇𑁭𑁇𑁭 𑀦𑀍𑀅𑀓𑀮𑀸𑀧𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀦𑀍𑀅𑀓𑀮𑀸𑀧𑀻𑀲𑀼𑀢𑁆𑀢

(67.)

490\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀬𑀁𑀓𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀧𑀭𑀁𑀓𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑁄𑀝𑁆𑀞𑀺𑀓, 𑀲𑀬𑀁𑀓𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀦 𑀧𑀭𑀁𑀓𑀢𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀦 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀦𑀸𑀧𑀺 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

491\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀬𑀁𑀓𑀢𑀸 𑀚𑀸𑀢𑀺, 𑀧𑀭𑀁𑀓𑀢𑀸 𑀚𑀸𑀢𑀺, 𑀲𑀬𑀁𑀓𑀢𑀸 𑀘 𑀧𑀭𑀁𑀓𑀢𑀸 𑀘 𑀚𑀸𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀲𑀬𑀗𑁆𑀓𑀸𑀭𑀸 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀸 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀚𑀸𑀢𑀻”𑀢𑀺? “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑁄𑀝𑁆𑀞𑀺𑀓, 𑀲𑀬𑀁𑀓𑀢𑀸 𑀚𑀸𑀢𑀺, 𑀦 𑀧𑀭𑀁𑀓𑀢𑀸 𑀚𑀸𑀢𑀺, 𑀦 𑀲𑀬𑀁𑀓𑀢𑀸 𑀘 𑀧𑀭𑀁𑀓𑀢𑀸 𑀘 𑀚𑀸𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀅𑀲𑀬𑀗𑁆𑀓𑀸𑀭𑀸 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀸 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀚𑀸𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀻”𑀢𑀺𑁇

492\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀬𑀁𑀓𑀢𑁄 𑀪𑀯𑁄…𑀧𑁂…  𑀲𑀬𑀁𑀓𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…  𑀲𑀬𑀁𑀓𑀢𑀸 𑀢𑀡𑁆𑀳𑀸…  𑀲𑀬𑀁𑀓𑀢𑀸 𑀯𑁂𑀤𑀦𑀸…  𑀲𑀬𑀁𑀓𑀢𑁄 𑀨𑀲𑁆𑀲𑁄…  𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀍𑀆𑀬𑀢𑀦𑀁…  𑀲𑀬𑀁𑀓𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀧𑀭𑀁𑀓𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆”𑀢𑀺? “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑁄𑀝𑁆𑀞𑀺𑀓, 𑀲𑀬𑀁𑀓𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦 𑀧𑀭𑀁𑀓𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀧𑀺 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁, 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆”𑀢𑀺𑁇

493\. “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀬𑀁𑀓𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀭𑀁𑀓𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆”𑀢𑀺? “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑁄𑀝𑁆𑀞𑀺𑀓, 𑀲𑀬𑀁𑀓𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦 𑀧𑀭𑀁𑀓𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦𑀸𑀧𑀺 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆”𑀢𑀺𑁇

494\. “𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀏𑀯𑀁 𑀆𑀚𑀸𑀦𑀸𑀫—  ‘𑀦 𑀔𑁆𑀯𑀸𑀯𑀼𑀲𑁄 𑀓𑁄𑀝𑁆𑀞𑀺𑀓, 𑀲𑀬𑀁𑀓𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦 𑀧𑀭𑀁𑀓𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀧𑀺 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆’𑀢𑀺𑁇

495\. 𑀇𑀤𑀸𑀦𑁂𑀯 𑀘 𑀧𑀦 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀏𑀯𑀁 𑀆𑀚𑀸𑀦𑀸𑀫—  ‘𑀦 𑀔𑁆𑀯𑀸𑀯𑀼𑀲𑁄 𑀓𑁄𑀝𑁆𑀞𑀺𑀓, 𑀲𑀬𑀁𑀓𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦 𑀧𑀭𑀁𑀓𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀦𑀸𑀧𑀺 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺𑁇

496\. 𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺? “𑀢𑁂𑀦𑀳𑀸𑀯𑀼𑀲𑁄, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀉𑀧𑀫𑀸𑀬𑀧𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁆𑀯𑁂 𑀦𑀍𑀅𑀓𑀮𑀸𑀧𑀺𑀬𑁄 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁; 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁; 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀸𑀲𑀜𑁆𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀍𑀅𑀓𑀮𑀸𑀧𑀻𑀦𑀁 𑀏𑀓𑀁 𑀆𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬, 𑀏𑀓𑀸 𑀧𑀧𑀢𑁂𑀬𑁆𑀬; 𑀅𑀧𑀭𑀜𑁆𑀘𑁂 𑀆𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬, 𑀅𑀧𑀭𑀸 𑀧𑀧𑀢𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁄; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄; 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄; 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

497\. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢; 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀬𑀸𑀯𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀜𑁆𑀘𑀺𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦𑁇 𑀇𑀤𑀜𑁆𑀘 𑀧𑀦 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀇𑀫𑁂𑀳𑀺 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀸𑀫—

498\. ‘𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀚𑀸𑀢𑀺𑀬𑀸 𑀘𑁂…  𑀪𑀯𑀲𑁆𑀲 𑀘𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀘𑁂…  𑀢𑀡𑁆𑀳𑀸𑀬 𑀘𑁂…  𑀯𑁂𑀤𑀦𑀸𑀬 𑀘𑁂…  𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀘𑁂…  𑀲𑀍𑀆𑀬𑀢𑀦𑀲𑁆𑀲 𑀘𑁂…  𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀘𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀘𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀜𑁆𑀘𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀘𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑀸’”𑀢𑀺𑁇

---
