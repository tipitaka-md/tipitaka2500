# 𑁧𑁇𑁭𑁇𑁧𑁦 𑀲𑀼𑀲𑀺𑀫𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀲𑀺𑀫𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀼𑀢𑁆𑀢

(70.)

518\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄𑀧𑀺 𑀲𑀓𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀅𑀲𑀓𑁆𑀓𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀕𑀭𑀼𑀓𑀢𑀸 𑀅𑀫𑀸𑀦𑀺𑀢𑀸 𑀅𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀦𑀧𑀘𑀺𑀢𑀸, 𑀦 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇

519\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀲𑀺𑀫𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀫𑀳𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀧𑀭𑀺𑀲𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀲𑀺𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀧𑀭𑀺𑀲𑀸 𑀲𑀼𑀲𑀺𑀫𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀼𑀲𑀺𑀫, 𑀲𑀫𑀡𑁂 𑀕𑁄𑀢𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑁇 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑁆𑀳𑁂 𑀯𑀸𑀘𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀁 𑀫𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀕𑀺𑀳𑀻𑀦𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀏𑀯𑀁 𑀫𑀬𑀫𑁆𑀧𑀺 𑀲𑀓𑁆𑀓𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀕𑀭𑀼𑀓𑀢𑀸 𑀫𑀸𑀦𑀺𑀢𑀸 𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀧𑀘𑀺𑀢𑀸 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀲𑀼𑀲𑀺𑀫𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀲𑀓𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀲𑀺𑀫𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

520\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀲𑀼𑀲𑀺𑀫𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀲𑀺𑀫𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀦𑁆’”𑀢𑀺𑁇 “𑀢𑁂𑀦𑀳𑀸𑀦𑀦𑁆𑀤, 𑀲𑀼𑀲𑀺𑀫𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀣𑀸”𑀢𑀺𑁇 𑀅𑀮𑀢𑁆𑀣 𑀔𑁄 𑀲𑀼𑀲𑀺𑀫𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀢𑁆𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇

521\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀲𑀺𑀫𑁄—  “𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀲𑀺𑀫𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀲𑀺𑀫𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂𑀳𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀸”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇

526\. “𑀅𑀧𑀺 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀣 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂, 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀣—  ‘𑀇𑀫𑁂 𑀯𑀢 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀉𑀧𑀯𑀸𑀤𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸, 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸; 𑀇𑀫𑁂 𑀯𑀸 𑀧𑀦 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀸𑀤𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀣 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂, 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄”𑁇

527\. “𑀅𑀧𑀺 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀬𑁂 𑀢𑁂 𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀭𑀽𑀧𑁂 𑀆𑀭𑀼𑀧𑁆𑀧𑀸, 𑀢𑁂 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄”𑁇

528\. “𑀏𑀢𑁆𑀣 𑀤𑀸𑀦𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀤𑀜𑁆𑀘 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺; 𑀇𑀤𑀁 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀣𑀦𑁆”𑀢𑀺? “𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀼𑀲𑀺𑀫𑀸”𑀢𑀺𑁇

529\. “𑀦 𑀔𑁆𑀯𑀸𑀳𑀁 𑀇𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀲𑀸𑀥𑀼 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀢𑀣𑀸 𑀪𑀸𑀲𑀦𑁆𑀢𑀼 𑀬𑀣𑀸𑀳𑀁 𑀇𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀯𑀸 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀼𑀲𑀺𑀫, 𑀦 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀅𑀣 𑀔𑁄 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇

530\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀲𑀺𑀫𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀲𑀺𑀫𑁄 𑀬𑀸𑀯𑀢𑀓𑁄 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀧𑀼𑀩𑁆𑀩𑁂 𑀔𑁄, 𑀲𑀼𑀲𑀺𑀫, 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀜𑀸𑀡𑀁, 𑀧𑀘𑁆𑀙𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑁂 𑀜𑀸𑀡𑀦𑁆”𑀢𑀺𑁇

542\. “𑀅𑀧𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀲𑀼𑀲𑀺𑀫, 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀬𑁂 𑀢𑁂 𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀭𑀽𑀧𑁂 𑀆𑀭𑀼𑀧𑁆𑀧𑀸, 𑀢𑁂 𑀓𑀸𑀬𑁂𑀦 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀲𑀻”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

543\. “𑀏𑀢𑁆𑀣 𑀤𑀸𑀦𑀺, 𑀲𑀼𑀲𑀺𑀫, 𑀇𑀤𑀜𑁆𑀘 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺, 𑀇𑀤𑀁 𑀦𑁄, 𑀲𑀼𑀲𑀺𑀫, 𑀓𑀣𑀦𑁆”𑀢𑀺?

544\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀲𑀺𑀫𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀘𑀬𑁄 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀘𑁆𑀘𑀕𑀫𑀸 𑀬𑀣𑀸𑀩𑀸𑀮𑀁 𑀬𑀣𑀸𑀫𑀽𑀍𑀳𑀁 𑀬𑀣𑀸𑀅𑀓𑀼𑀲𑀮𑀁, 𑀬𑁆𑀯𑀸𑀳𑀁 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀥𑀫𑁆𑀫𑀢𑁆𑀣𑁂𑀦𑀓𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀆𑀬𑀢𑀺𑀁 𑀲𑀁𑀯𑀭𑀸𑀬𑀸”𑀢𑀺𑁇

545\. “𑀢𑀕𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀲𑀼𑀲𑀺𑀫, 𑀅𑀘𑁆𑀘𑀬𑁄 𑀅𑀘𑁆𑀘𑀕𑀫𑀸 𑀬𑀣𑀸𑀩𑀸𑀮𑀁 𑀬𑀣𑀸𑀫𑀽𑀍𑀳𑀁 𑀬𑀣𑀸𑀅𑀓𑀼𑀲𑀮𑀁, 𑀬𑁄 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀥𑀫𑁆𑀫𑀢𑁆𑀣𑁂𑀦𑀓𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀲𑀼𑀲𑀺𑀫, 𑀘𑁄𑀭𑀁 𑀆𑀕𑀼𑀘𑀸𑀭𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀤𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀤𑁂𑀯, 𑀘𑁄𑀭𑁄 𑀆𑀕𑀼𑀘𑀸𑀭𑀻, 𑀇𑀫𑀲𑁆𑀲 𑀬𑀁 𑀇𑀘𑁆𑀙𑀲𑀺 𑀢𑀁 𑀤𑀡𑁆𑀟𑀁 𑀧𑀡𑁂𑀳𑀻’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀕𑀘𑁆𑀙𑀣, 𑀪𑁄, 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀬𑀸𑀬 𑀭𑀣𑀺𑀬𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀣𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀜𑁆𑀜𑁄 𑀧𑀼𑀭𑀺𑀲𑀸 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀬𑀸𑀬 𑀭𑀣𑀺𑀬𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀲𑀼𑀲𑀺𑀫, 𑀅𑀧𑀺 𑀦𑀼 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

546\. “𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀲𑀼𑀲𑀺𑀫, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀬𑁂𑀣𑁇 𑀬𑀸 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀥𑀫𑁆𑀫𑀢𑁆𑀣𑁂𑀦𑀓𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀅𑀬𑀁 𑀢𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀯𑀺𑀧𑀸𑀓𑀢𑀭𑀸 𑀘 𑀓𑀝𑀼𑀓𑀯𑀺𑀧𑀸𑀓𑀢𑀭𑀸 𑀘, 𑀅𑀧𑀺 𑀘 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀼𑀲𑀺𑀫, 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀭𑁄𑀲𑀺 𑀢𑀁 𑀢𑁂 𑀫𑀬𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫𑁇 𑀯𑀼𑀤𑁆𑀥𑀺 𑀳𑁂𑀲𑀸, 𑀲𑀼𑀲𑀺𑀫, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀬𑁄 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀭𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

_𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑁂𑀦 𑀘𑀸𑀧𑀭𑀁;_
