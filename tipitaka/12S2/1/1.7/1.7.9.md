# 𑁧𑁇𑁭𑁇𑁯 𑀉𑀧𑀬𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀬𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(69.)

515\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄…𑀧𑁂…  “𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀬𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄 𑀉𑀧𑀬𑀦𑁆𑀢𑀺𑀬𑁄 𑀓𑀼𑀦𑁆𑀦𑀤𑀺𑀬𑁄 𑀉𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀓𑀼𑀦𑁆𑀦𑀤𑀺𑀬𑁄 𑀉𑀧𑀬𑀦𑁆𑀢𑀺𑀬𑁄 𑀫𑀳𑀸𑀲𑁄𑀩𑁆𑀪𑁂 𑀉𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸𑀲𑁄𑀩𑁆𑀪𑀸 𑀉𑀧𑀬𑀦𑁆𑀢𑀸 𑀓𑀼𑀲𑁄𑀩𑁆𑀪𑁂 𑀉𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑀬𑀦𑁆𑀢𑀻 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀉𑀧𑀬𑀦𑁆𑀢𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀬𑀦𑁆𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀉𑀧𑀬𑀦𑁆𑀢𑀁 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀉𑀧𑀬𑀦𑁆𑀢𑀁 𑀨𑀲𑁆𑀲𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀨𑀲𑁆𑀲𑁄 𑀉𑀧𑀬𑀦𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑀬𑀦𑁆𑀢𑀻 𑀢𑀡𑁆𑀳𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑀬𑀦𑁆𑀢𑀻 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀉𑀧𑀬𑀦𑁆𑀢𑀁 𑀪𑀯𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀪𑀯𑁄 𑀉𑀧𑀬𑀦𑁆𑀢𑁄 𑀚𑀸𑀢𑀺𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀚𑀸𑀢𑀺 𑀉𑀧𑀬𑀦𑁆𑀢𑀻 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀺𑁇

516\. 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀬𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄 𑀅𑀧𑀬𑀦𑁆𑀢𑀺𑀬𑁄 𑀓𑀼𑀦𑁆𑀦𑀤𑀺𑀬𑁄 𑀅𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀓𑀼𑀦𑁆𑀦𑀤𑀺𑀬𑁄 𑀅𑀧𑀬𑀦𑁆𑀢𑀺𑀬𑁄 𑀫𑀳𑀸𑀲𑁄𑀩𑁆𑀪𑁂 𑀅𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸𑀲𑁄𑀩𑁆𑀪𑀸 𑀅𑀧𑀬𑀦𑁆𑀢𑀸 𑀓𑀼𑀲𑁄𑀩𑁆𑀪𑁂 𑀅𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀧𑀬𑀦𑁆𑀢𑀻 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀧𑀬𑀦𑁆𑀢𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀧𑀬𑀦𑁆𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀅𑀧𑀬𑀦𑁆𑀢𑀁 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀅𑀧𑀬𑀦𑁆𑀢𑀁 𑀨𑀲𑁆𑀲𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀨𑀲𑁆𑀲𑁄 𑀅𑀧𑀬𑀦𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀧𑀬𑀦𑁆𑀢𑀻 𑀢𑀡𑁆𑀳𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀢𑀡𑁆𑀳𑀸 𑀅𑀧𑀬𑀦𑁆𑀢𑀻 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀧𑀬𑀦𑁆𑀢𑀁 𑀪𑀯𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀪𑀯𑁄 𑀅𑀧𑀬𑀦𑁆𑀢𑁄 𑀚𑀸𑀢𑀺𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀺, 𑀚𑀸𑀢𑀺 𑀅𑀧𑀬𑀦𑁆𑀢𑀻 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀻”𑀢𑀺𑁇
