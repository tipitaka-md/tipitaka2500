# 𑁧𑁇𑁮𑁇𑁨--𑁧𑁧 𑀚𑀸𑀢𑀺𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀤𑀲𑀓

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀚𑀸𑀢𑀺𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀤𑀲𑀓

(72.)

554\. “𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀚𑀸𑀢𑀺𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

555\. “𑀪𑀯𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

556\. “𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

557\. “𑀢𑀡𑁆𑀳𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

558\. “𑀯𑁂𑀤𑀦𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

559\. “𑀨𑀲𑁆𑀲𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

560\. “𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

561\. “𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

562\. “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

563\. “𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

---

564\. 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇



---

565\. 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇



---

566\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



567\. _𑀧𑀘𑁆𑀘𑀬𑁂𑀓𑀸𑀤𑀲 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀯𑀺𑀪𑀚𑁆𑀚𑀦𑀸;_  
_𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕𑁄,_  
_𑀦𑀺𑀤𑀸𑀦𑁂 𑀪𑀯𑀢𑀺 𑀅𑀝𑁆𑀞𑀫𑁄𑁇_  


---

568\. 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



569\. _𑀩𑀼𑀤𑁆𑀥𑁄 𑀆𑀳𑀸𑀭𑁄 𑀤𑀲𑀩𑀮𑁄,_  
_𑀓𑀍𑀆𑀭𑁄 𑀕𑀳𑀧𑀢𑀺𑀧𑀜𑁆𑀘𑀫𑁄;_  
_𑀤𑀼𑀓𑁆𑀔𑀯𑀕𑁆𑀕𑁄 𑀫𑀳𑀸𑀯𑀕𑁆𑀕𑁄,_  
_𑀅𑀝𑁆𑀞𑀫𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀢𑀺𑁇_
