# 𑁧𑁇𑁪𑁇𑁧𑁦 𑀢𑀢𑀺𑀬𑀘𑁂𑀢𑀦𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀢𑀢𑀺𑀬𑀘𑁂𑀢𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(40.)

286. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀢𑁂𑀢𑀺 𑀬𑀜𑁆𑀘 𑀧𑀓𑀧𑁆𑀧𑁂𑀢𑀺 𑀬𑀜𑁆𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑁂𑀢𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀞𑀺𑀢𑀺𑀬𑀸𑁇 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑀢𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀯𑀺𑀭𑀽𑀍𑀳𑁂 𑀦𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀦𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀆𑀕𑀢𑀺𑀕𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀆𑀕𑀢𑀺𑀕𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁂 𑀲𑀢𑀺 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

287. 𑀦𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀢𑁂𑀢𑀺 𑀦𑁄 𑀘𑁂 𑀧𑀓𑀧𑁆𑀧𑁂𑀢𑀺 𑀅𑀣 𑀘𑁂 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑁂𑀢𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀞𑀺𑀢𑀺𑀬𑀸𑁇 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑀢𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀯𑀺𑀭𑀽𑀍𑀳𑁂 𑀦𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀦𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀆𑀕𑀢𑀺𑀕𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀆𑀕𑀢𑀺𑀕𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁂 𑀲𑀢𑀺 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

288. 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁄 𑀘𑁂𑀯 𑀘𑁂𑀢𑁂𑀢𑀺 𑀦𑁄 𑀘 𑀧𑀓𑀧𑁆𑀧𑁂𑀢𑀺 𑀦𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑁂𑀢𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀞𑀺𑀢𑀺𑀬𑀸𑁇 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀅𑀲𑀢𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀅𑀯𑀺𑀭𑀽𑀍𑀳𑁂 𑀦𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀦𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀆𑀕𑀢𑀺𑀕𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀆𑀕𑀢𑀺𑀕𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁂 𑀅𑀲𑀢𑀺 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
