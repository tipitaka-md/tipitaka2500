# 𑁧𑁇𑁪𑁇𑁫 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀲𑀼𑀢𑁆𑀢

(35.)

260\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “‘𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺? ‘𑀦𑁄 𑀓𑀮𑁆𑀮𑁄 𑀧𑀜𑁆𑀳𑁄’𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘, ‘𑀓𑀢𑀫𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺, 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀏𑀓𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁𑁇 𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’”𑀢𑀺𑁇

261\. “𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀢𑀺, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀚𑀸𑀢𑀻”𑀢𑀺? “𑀦𑁄 𑀓𑀮𑁆𑀮𑁄 𑀧𑀜𑁆𑀳𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘, “‘𑀓𑀢𑀫𑀸 𑀚𑀸𑀢𑀺, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀚𑀸𑀢𑀻’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, ‘𑀅𑀜𑁆𑀜𑀸 𑀚𑀸𑀢𑀺 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀚𑀸𑀢𑀻’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀏𑀓𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁𑁇 𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀻’”𑀢𑀺𑁇

262\. “𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀯𑁄, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀪𑀯𑁄”𑀢𑀺? “𑀦𑁄 𑀓𑀮𑁆𑀮𑁄 𑀧𑀜𑁆𑀳𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘, “‘𑀓𑀢𑀫𑁄 𑀪𑀯𑁄, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀪𑀯𑁄’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, ‘𑀅𑀜𑁆𑀜𑁄 𑀪𑀯𑁄 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀪𑀯𑁄’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀏𑀓𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁𑁇 𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺; 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄’𑀢𑀺…𑀧𑁂…  ‘𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆𑀢𑀺…  𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸𑀢𑀺…  𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀢𑀺…  𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄𑀢𑀺…  𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’”𑀢𑀺𑁇

263\. “𑀓𑀢𑀫𑁂 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸”𑀢𑀺? “𑀦𑁄 𑀓𑀮𑁆𑀮𑁄 𑀧𑀜𑁆𑀳𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘, “‘𑀓𑀢𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, ‘𑀅𑀜𑁆𑀜𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀏𑀓𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁𑁇 𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺; 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’”𑀢𑀺𑁇

264\. “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀺 𑀯𑀺𑀲𑁂𑀯𑀺𑀢𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀸𑀦𑀺𑀘𑀺𑁇 ‘𑀓𑀢𑀫𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸𑁇 𑀲𑀩𑁆𑀩𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸𑀦𑀺 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸𑀦𑀺 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑀦𑀺𑁇

265\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀺 𑀯𑀺𑀲𑁂𑀯𑀺𑀢𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀸𑀦𑀺𑀘𑀺𑁇 ‘𑀓𑀢𑀫𑀸 𑀚𑀸𑀢𑀺, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀚𑀸𑀢𑀺’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀸 𑀚𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀚𑀸𑀢𑀺’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸𑁇 𑀲𑀩𑁆𑀩𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸𑀦𑀺 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸𑀦𑀺 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑀦𑀺𑁇

266\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀺 𑀯𑀺𑀲𑁂𑀯𑀺𑀢𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀸𑀦𑀺𑀘𑀺𑁇 𑀓𑀢𑀫𑁄 𑀪𑀯𑁄…𑀧𑁂…  𑀓𑀢𑀫𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…  𑀓𑀢𑀫𑀸 𑀢𑀡𑁆𑀳𑀸…  𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸…  𑀓𑀢𑀫𑁄 𑀨𑀲𑁆𑀲𑁄…  𑀓𑀢𑀫𑀁 𑀲𑀍𑀆𑀬𑀢𑀦𑀁…  𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁…  𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂… 𑁇

267\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀺 𑀯𑀺𑀲𑁂𑀯𑀺𑀢𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀸𑀦𑀺𑀘𑀺𑁇 ‘𑀓𑀢𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸𑁇 𑀲𑀩𑁆𑀩𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸𑀦𑀺 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸𑀦𑀺 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑀦𑀻”𑀢𑀺𑁇
