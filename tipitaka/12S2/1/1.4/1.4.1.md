# 𑁧𑁇𑁪𑁇𑁧 𑀪𑀽𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀪𑀽𑀢𑀲𑀼𑀢𑁆𑀢

(31.)

207. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀯𑀼𑀢𑁆𑀢𑀫𑀺𑀤𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀸𑀭𑀸𑀬𑀦𑁂 𑀅𑀚𑀺𑀢𑀧𑀜𑁆𑀳𑁂—

208. _‘𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀸𑀲𑁂,_  
_𑀬𑁂 𑀘 𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀼𑀣𑀽 𑀇𑀥;_  
_𑀢𑁂𑀲𑀁 𑀫𑁂 𑀦𑀺𑀧𑀓𑁄 𑀇𑀭𑀺𑀬𑀁,_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑀸𑀭𑀺𑀲𑀸’𑀢𑀺𑁇_  


209. 𑀇𑀫𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀓𑀣𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

210. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀯𑀼𑀢𑁆𑀢𑀫𑀺𑀤𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀸𑀭𑀸𑀬𑀦𑁂 𑀅𑀚𑀺𑀢𑀧𑀜𑁆𑀳𑁂—

211. _‘𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀸𑀲𑁂,_  
_𑀬𑁂 𑀘 𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀼𑀣𑀽 𑀇𑀥;_  
_𑀢𑁂𑀲𑀁 𑀫𑁂 𑀦𑀺𑀧𑀓𑁄 𑀇𑀭𑀺𑀬𑀁,_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑀸𑀭𑀺𑀲𑀸’𑀢𑀺𑁇_  


212. 𑀇𑀫𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀓𑀣𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

213. “𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? “𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀽𑀢𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁂𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇

214. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀳𑁄𑀢𑀺? 𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀽𑀢𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀧𑀸𑀭𑀸𑀬𑀦𑁂 𑀅𑀚𑀺𑀢𑀧𑀜𑁆𑀳𑁂—

215. _‘𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀸𑀲𑁂,_  
_𑀬𑁂 𑀘 𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀼𑀣𑀽 𑀇𑀥;_  
_𑀢𑁂𑀲𑀁 𑀫𑁂 𑀦𑀺𑀧𑀓𑁄 𑀇𑀭𑀺𑀬𑀁,_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑀸𑀭𑀺𑀲𑀸’𑀢𑀺𑁇_  


216. 𑀇𑀫𑀲𑁆𑀲 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇

217. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀽𑀢𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑁂𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇

218. 𑀓𑀣𑀜𑁆𑀘, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀳𑁄𑀢𑀺? 𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀪𑀽𑀢𑀫𑀺𑀤𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀽𑀢𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀳𑀸𑀭𑀲𑀫𑁆𑀪𑀯𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀤𑀸𑀳𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀁 𑀪𑀽𑀢𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀧𑀸𑀭𑀸𑀬𑀦𑁂 𑀅𑀚𑀺𑀢𑀧𑀜𑁆𑀳𑁂—

219. _‘𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀸𑀲𑁂,_  
_𑀬𑁂 𑀘 𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀼𑀣𑀽 𑀇𑀥;_  
_𑀢𑁂𑀲𑀁 𑀫𑁂 𑀦𑀺𑀧𑀓𑁄 𑀇𑀭𑀺𑀬𑀁,_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑀸𑀭𑀺𑀲𑀸’𑀢𑀺𑁇_  


220. 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
