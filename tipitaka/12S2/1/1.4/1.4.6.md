# 𑁧𑁇𑁪𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀲𑀼𑀢𑁆𑀢

(36.)

269\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

270\. ‘𑀓𑀢𑀫𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀏𑀓𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁𑁇 ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆’𑀢𑀺𑁇

271\. 𑀓𑀢𑀫𑀸 𑀚𑀸𑀢𑀺…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀪𑀯𑁄…  𑀓𑀢𑀫𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…  𑀓𑀢𑀫𑀸 𑀢𑀡𑁆𑀳𑀸…  𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸…  𑀓𑀢𑀫𑁄 𑀨𑀲𑁆𑀲𑁄…  𑀓𑀢𑀫𑀁 𑀲𑀍𑀆𑀬𑀢𑀦𑀁…  𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁…  𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  𑀓𑀢𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, ‘𑀅𑀜𑁆𑀜𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀏𑀓𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁂𑀯 𑀦𑀸𑀦𑀁𑁇 ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀫𑀚𑁆𑀛𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’𑀢𑀺𑁇

272\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀺 𑀯𑀺𑀲𑁂𑀯𑀺𑀢𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀸𑀦𑀺𑀘𑀺𑁇 ‘𑀓𑀢𑀫𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀤𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸𑁇 𑀲𑀩𑁆𑀩𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸𑀦𑀺 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸𑀦𑀺 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑀦𑀺𑁇

273\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀢𑁆𑀯𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁂𑀲𑀯𑀺𑀭𑀸𑀕𑀦𑀺𑀭𑁄𑀥𑀸 𑀬𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀺 𑀯𑀺𑀲𑁂𑀯𑀺𑀢𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀸𑀦𑀺𑀘𑀺𑁇 𑀓𑀢𑀫𑀸 𑀚𑀸𑀢𑀺…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀪𑀯𑁄…  𑀓𑀢𑀫𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…  𑀓𑀢𑀫𑀸 𑀢𑀡𑁆𑀳𑀸…  𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸…  𑀓𑀢𑀫𑁄 𑀨𑀲𑁆𑀲𑁄…  𑀓𑀢𑀫𑀁 𑀲𑀍𑀆𑀬𑀢𑀦𑀁…  𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁…  𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…  ‘𑀓𑀢𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀓𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀧𑀦𑀺𑀫𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸’ 𑀇𑀢𑀺 𑀯𑀸; ‘𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸, ‘𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁’ 𑀇𑀢𑀺 𑀯𑀸𑁇 𑀲𑀩𑁆𑀩𑀸𑀦𑀺𑀲𑁆𑀲 𑀢𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸𑀦𑀺 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸𑀦𑀺 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑀦𑀻”𑀢𑀺𑁇

274\. 𑀙𑀝𑁆𑀞𑀁𑁇
