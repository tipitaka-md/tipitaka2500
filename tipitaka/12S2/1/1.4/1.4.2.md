# 𑁧𑁇𑁪𑁇𑁨 𑀓𑀍𑀆𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀓𑀍𑀆𑀭𑀲𑀼𑀢𑁆𑀢

(32.)

222\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑁄𑀍𑀇𑀬𑀨𑀕𑁆𑀕𑀼𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 “𑀦 𑀳𑀺 𑀦𑀽𑀦 𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀲𑁆𑀲𑀸𑀲𑀫𑀮𑀢𑁆𑀣𑀸”𑀢𑀺𑁇

223\. “𑀢𑁂𑀦 𑀳𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀲𑁆𑀲𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑁄”𑀢𑀺?

224\. “𑀦 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀗𑁆𑀔𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀆𑀬𑀢𑀺𑀁, 𑀧𑀦𑀸𑀯𑀼𑀲𑁄”𑀢𑀺?

225\. “𑀦 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

226\. 𑀅𑀣 𑀔𑁄 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀲𑁆𑀫𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇

227\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀬𑀸, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺? “𑀦 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀏𑀢𑁂𑀳𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 “𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺𑀧𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀢𑀢𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀅𑀳𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀦 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀏𑀢𑁂𑀳𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄 𑀯𑀼𑀢𑁆𑀢𑁄’”𑀢𑀺𑁇

228\. “𑀲𑀘𑁂 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀣𑀁 𑀚𑀸𑀦𑀢𑀸 𑀧𑀦 𑀢𑀬𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀣𑀁 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺?

229\. “𑀲𑀘𑁂 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀣𑀁 𑀚𑀸𑀦𑀢𑀸 𑀧𑀦 𑀢𑀬𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀣𑀁 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺; 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁—  ‘𑀬𑀁𑀦𑀺𑀤𑀸𑀦𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀦𑀺𑀤𑀸𑀦𑀲𑁆𑀲 𑀔𑀬𑀸 𑀔𑀻𑀡𑀲𑁆𑀫𑀺𑀁 𑀔𑀻𑀡𑀸𑀫𑁆𑀳𑀻𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑀁𑁇 𑀔𑀻𑀡𑀸𑀫𑁆𑀳𑀻𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸—  𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

230\. “𑀲𑀘𑁂 𑀧𑀦 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀚𑀸𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸’𑀢𑀺? 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁂 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀚𑀸𑀢𑀺 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸’𑀢𑀺? 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁—  ‘𑀚𑀸𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀯𑀦𑀺𑀤𑀸𑀦𑀸 𑀪𑀯𑀲𑀫𑀼𑀤𑀬𑀸 𑀪𑀯𑀚𑀸𑀢𑀺𑀓𑀸 𑀪𑀯𑀧𑁆𑀧𑀪𑀯𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

231\. “𑀲𑀘𑁂 𑀧𑀦 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀪𑀯𑁄 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑁄 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑁄 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑁄 𑀓𑀺𑀁𑀧𑀪𑀯𑁄’𑀢𑀺? 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁂 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀪𑀯𑁄 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑁄 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑁄 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑁄 𑀓𑀺𑀁𑀧𑀪𑀯𑁄’𑀢𑀺? 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁—  ‘𑀪𑀯𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀤𑀸𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑀼𑀤𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀚𑀸𑀢𑀺𑀓𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑁆𑀧𑀪𑀯𑁄’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

232\. “𑀲𑀘𑁂 𑀧𑀦 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄…𑀧𑁂…  𑀲𑀘𑁂 𑀧𑀦 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  𑀢𑀡𑁆𑀳𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸’𑀢𑀺? 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁂 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀢𑀡𑁆𑀳𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸’𑀢𑀺? 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁—  ‘𑀢𑀡𑁆𑀳𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀤𑀸𑀦𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀼𑀤𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀚𑀸𑀢𑀺𑀓𑀸 𑀯𑁂𑀤𑀦𑀸𑀧𑀪𑀯𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

233\. “𑀲𑀘𑁂 𑀧𑀦 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀣𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀦 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀣𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀦𑀦𑁆𑀤𑀻 𑀲𑀸 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁂 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀣𑀁 𑀚𑀸𑀦𑀢𑁄 𑀧𑀦 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀣𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀦𑀦𑁆𑀤𑀻 𑀲𑀸 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁—  ‘𑀢𑀺𑀲𑁆𑀲𑁄 𑀔𑁄 𑀇𑀫𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑁂𑀤𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑁇 𑀬𑀤𑀦𑀺𑀘𑁆𑀘𑀁 𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑀁, 𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀦𑀦𑁆𑀤𑀻 𑀲𑀸 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁, 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

234\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀏𑀢𑀲𑁆𑀲𑁂𑀯 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀬—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀦𑁆’”𑀢𑀺𑁇

235\. “𑀲𑀘𑁂 𑀧𑀦 𑀢𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀣𑀁 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸 𑀧𑀦 𑀢𑀬𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺? 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁂 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀣𑀁 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸 𑀧𑀦 𑀢𑀬𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀜𑁆𑀜𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸—  𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁—  ‘𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀩𑁆𑀩𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀬𑀸 𑀢𑀣𑀸 𑀲𑀢𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀬𑀣𑀸 𑀲𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀆𑀲𑀯𑀸 𑀦𑀸𑀦𑀼𑀲𑁆𑀲𑀯𑀦𑁆𑀢𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘 𑀦𑀸𑀯𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

236\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀏𑀢𑀲𑁆𑀲𑁂𑀯 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀬—  𑀬𑁂 𑀆𑀲𑀯𑀸 𑀲𑀫𑀡𑁂𑀦 𑀯𑀼𑀢𑁆𑀢𑀸 𑀢𑁂𑀲𑁆𑀯𑀸𑀳𑀁 𑀦 𑀓𑀗𑁆𑀔𑀸𑀫𑀺, 𑀢𑁂 𑀫𑁂 𑀧𑀳𑀻𑀦𑀸𑀢𑀺 𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀲𑀼𑀕𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

238\. 𑀅𑀣 𑀔𑁄 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀓𑀍𑀆𑀭𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀲𑁆𑀫𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀻𑀳𑀦𑀸𑀤𑁄 𑀦𑀤𑀺𑀢𑁄—  𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀧𑀞𑀫𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀅𑀳𑁄𑀲𑀺 𑀤𑀦𑁆𑀥𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀧𑀞𑀫𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀦𑀼𑀫𑁄𑀤𑀺, 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  𑀤𑀺𑀯𑀲𑀜𑁆𑀘𑁂𑀧𑀺 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀺, 𑀤𑀺𑀯𑀲𑀫𑁆𑀧𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀺; 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘𑁂𑀧𑀺…𑀧𑁂…  𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀜𑁆𑀘𑁂𑀧𑀺 𑀫𑀁 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀺 𑀫𑀁 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀢𑀻𑀡𑀺…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺…  𑀧𑀜𑁆𑀘…  𑀙…  𑀲𑀢𑁆𑀢 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀺 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀺, 𑀲𑀢𑁆𑀢 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺𑀧𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀻”𑀢𑀺𑁇

239\. “𑀲𑀸 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸, 𑀬𑀲𑁆𑀲𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀢𑁆𑀢𑀸 𑀤𑀺𑀯𑀲𑀁 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀺, 𑀤𑀺𑀯𑀲𑀫𑁆𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀺𑁇 𑀭𑀢𑁆𑀢𑀺𑀁 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀺, 𑀭𑀢𑁆𑀢𑀺𑀫𑁆𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬…𑀧𑁂…  𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑁆𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬…  𑀤𑁆𑀯𑁂 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀤𑁆𑀯𑁂 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬…  𑀢𑀻𑀡𑀺 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀢𑀻𑀡𑀺 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬…  𑀧𑀜𑁆𑀘 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀧𑀜𑁆𑀘 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬…  𑀙 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀙 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬…  𑀲𑀢𑁆𑀢 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺 𑀘𑁂𑀧𑀸𑀳𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀺, 𑀲𑀢𑁆𑀢 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸𑀦𑀺𑀧𑀺 𑀫𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀤𑁂𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀳𑀻”𑀢𑀺𑁇
