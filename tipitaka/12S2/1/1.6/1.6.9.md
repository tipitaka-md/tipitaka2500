# 𑁧𑁇𑁬𑁇𑁯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀼𑀢𑁆𑀢

(59.)

421\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

422\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑁄𑁇 𑀢𑀲𑁆𑀲 𑀬𑀸𑀦𑀺 𑀘𑁂𑀯 𑀫𑀽𑀮𑀸𑀦𑀺…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

423\. 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

424\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑁄𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀓𑀼𑀤𑁆𑀤𑀸𑀮𑀧𑀺𑀝𑀓𑀁 𑀆𑀤𑀸𑀬…𑀧𑁂…  𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
