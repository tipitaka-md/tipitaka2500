# 𑁧𑁇𑁬𑁇𑁧 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑁆𑀔𑀯𑀕𑁆𑀕

* 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀦𑀲𑀼𑀢𑁆𑀢

(51.)

365\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

366\. “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑁂𑀬𑁆𑀬 𑀲𑀩𑁆𑀩𑀲𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬𑀸”𑀢𑀺? “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀪𑀕𑀯𑀁𑀦𑁂𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀁𑀧𑀝𑀺𑀲𑀭𑀡𑀸𑁇 𑀲𑀸𑀥𑀼 𑀯𑀢, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀏𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

367\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀢𑀺—  ‘𑀬𑀁 𑀔𑁄 𑀇𑀤𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀥𑀁 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁; 𑀇𑀤𑀁 𑀦𑀼 𑀔𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁 𑀓𑀺𑀁𑀧𑀪𑀯𑀁? 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀬𑀁 𑀔𑁄 𑀇𑀤𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀥𑀁 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁, 𑀇𑀤𑀁 𑀔𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀚𑀸𑀢𑀺𑀦𑀺𑀤𑀸𑀦𑀁 𑀚𑀸𑀢𑀺𑀲𑀫𑀼𑀤𑀬𑀁 𑀚𑀸𑀢𑀺𑀚𑀸𑀢𑀺𑀓𑀁 𑀚𑀸𑀢𑀺𑀧𑁆𑀧𑀪𑀯𑀁𑁇 𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇

368\. 𑀲𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀸 𑀘 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀣𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻; 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀸𑀬𑁇

369\. 𑀅𑀣𑀸𑀧𑀭𑀁 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀢𑀺—  ‘𑀚𑀸𑀢𑀺 𑀧𑀦𑀸𑀬𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀚𑀸𑀢𑀺 𑀪𑀯𑀦𑀺𑀤𑀸𑀦𑀸 𑀪𑀯𑀲𑀫𑀼𑀤𑀬𑀸 𑀪𑀯𑀚𑀸𑀢𑀺𑀓𑀸 𑀪𑀯𑀧𑁆𑀧𑀪𑀯𑀸; 𑀪𑀯𑁂 𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀪𑀯𑁂 𑀅𑀲𑀢𑀺 𑀚𑀸𑀢𑀺 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇

370\. 𑀲𑁄 𑀚𑀸𑀢𑀺𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀸 𑀘 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀣𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻; 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸𑀬𑁇

371\. 𑀅𑀣𑀸𑀧𑀭𑀁 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀢𑀺—  ‘𑀪𑀯𑁄 𑀧𑀦𑀸𑀬𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑁄…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀦𑀺𑀤𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁…  𑀢𑀡𑁆𑀳𑀸 𑀧𑀦𑀸𑀬𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸…  𑀯𑁂𑀤𑀦𑀸…  𑀨𑀲𑁆𑀲𑁄…  𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀧𑀦𑀺𑀤𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁…  𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀧𑀦𑀺𑀤𑀁…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀦𑀺𑀤𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀧𑀦𑀺𑀫𑁂 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀸 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀸 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀸 𑀓𑀺𑀁𑀧𑀪𑀯𑀸; 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀧𑀭𑀺𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀤𑀸𑀦𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀫𑀼𑀤𑀬𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀚𑀸𑀢𑀺𑀓𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀪𑀯𑀸; 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀅𑀲𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀻’𑀢𑀺𑁇

372\. 𑀲𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀸 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀣𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻; 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀲𑁄 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸𑀬𑁇

373\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀕𑀢𑁄 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀼𑀜𑁆𑀜𑀜𑁆𑀘𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺, 𑀧𑀼𑀜𑁆𑀜𑀽𑀧𑀕𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀅𑀧𑀼𑀜𑁆𑀜𑀜𑁆𑀘𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺, 𑀅𑀧𑀼𑀜𑁆𑀜𑀽𑀧𑀕𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀆𑀦𑁂𑀜𑁆𑀚𑀜𑁆𑀘𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀆𑀦𑁂𑀜𑁆𑀚𑀽𑀧𑀕𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀭𑀸𑀕𑀸 𑀯𑀺𑀚𑁆𑀚𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀦𑁂𑀯 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀦 𑀅𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺 𑀦 𑀆𑀦𑁂𑀜𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺𑁇 𑀅𑀦𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀦𑁆𑀢𑁄 𑀅𑀦𑀪𑀺𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀦𑁆𑀢𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺; 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀜𑁆𑀜𑁂𑀯 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

374\. 𑀲𑁄 𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀲𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀜𑁆𑀘𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺, 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇

375\. 𑀲𑁄 𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀉𑀤𑁆𑀥𑀁 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀇𑀥𑁂𑀯 𑀲𑀩𑁆𑀩𑀯𑁂𑀤𑀬𑀺𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀲𑀻𑀢𑀻𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀭𑀻𑀭𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

376\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀧𑀸𑀓𑀸 𑀉𑀡𑁆𑀳𑀁 𑀓𑀼𑀫𑁆𑀪𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁂 𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑁂 𑀧𑀝𑀺𑀲𑀺𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑁇 𑀢𑀢𑁆𑀭 𑀬𑀸𑀬𑀁 𑀉𑀲𑁆𑀫𑀸 𑀲𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀯𑀽𑀧𑀲𑀫𑁂𑀬𑁆𑀬, 𑀓𑀧𑀮𑁆𑀮𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀉𑀤𑁆𑀥𑀁 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀇𑀥𑁂𑀯 𑀲𑀩𑁆𑀩𑀯𑁂𑀤𑀬𑀺𑀢𑀸𑀦𑀺 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺 𑀲𑀻𑀢𑀻𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀭𑀻𑀭𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

377\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁂𑀬𑁆𑀬 𑀅𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁂𑀬𑁆𑀬 𑀆𑀦𑁂𑀜𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

378\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀲𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

379\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀅𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫𑀭𑀽𑀧𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

380\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀦𑀸𑀫𑀭𑀽𑀧𑁂 𑀅𑀲𑀢𑀺, 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

381\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀲𑀍𑀆𑀬𑀢𑀦𑁂 𑀅𑀲𑀢𑀺, 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀨𑀲𑁆𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

382\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀨𑀲𑁆𑀲𑁂 𑀅𑀲𑀢𑀺, 𑀨𑀲𑁆𑀲𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀯𑁂𑀤𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

383\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀲𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

384\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀢𑀡𑁆𑀳𑀸𑀬 𑀅𑀲𑀢𑀺, 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

385\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀅𑀲𑀢𑀺, 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀪𑀯𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺𑁇 “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

386\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑁂 𑀅𑀲𑀢𑀺, 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀚𑀸𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

387\. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀸 𑀧𑀦 𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀲𑀢𑀺, 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

388\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀢𑀁 𑀅𑀜𑁆𑀜𑀣𑀸𑁇 𑀲𑀤𑁆𑀤𑀳𑀣 𑀫𑁂 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀣, 𑀦𑀺𑀓𑁆𑀓𑀗𑁆𑀔𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀣 𑀦𑀺𑀩𑁆𑀩𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
