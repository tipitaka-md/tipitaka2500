# 𑁧𑁇𑁬𑁇𑁭 𑀢𑀭𑀼𑀡𑀭𑀼𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓𑁆𑀔𑀯𑀕𑁆𑀕

* 𑀢𑀭𑀼𑀡𑀭𑀼𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

(57.)

411\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀯𑀟𑁆𑀠𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

412\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀭𑀼𑀡𑁄 𑀭𑀼𑀓𑁆𑀔𑁄𑁇 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀽𑀮𑀸𑀦𑀺 𑀧𑀮𑀺𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀁𑀲𑀼𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀉𑀤𑀓𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀭𑀼𑀡𑁄 𑀭𑀼𑀓𑁆𑀔𑁄 𑀢𑀤𑀸𑀳𑀸𑀭𑁄 𑀢𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑁄 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀯𑀟𑁆𑀠𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

413\. 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

414\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀭𑀼𑀡𑁄 𑀭𑀼𑀓𑁆𑀔𑁄𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀓𑀼𑀤𑁆𑀤𑀸𑀮𑀧𑀺𑀝𑀓𑀁 𑀆𑀤𑀸𑀬…𑀧𑁂…  𑀦𑀤𑀺𑀬𑀸 𑀯𑀸 𑀲𑀻𑀖𑀲𑁄𑀢𑀸𑀬 𑀧𑀯𑀸𑀳𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀭𑀼𑀡𑁄 𑀭𑀼𑀓𑁆𑀔𑁄 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀅𑀲𑁆𑀲 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
