# 𑁧𑁇𑁬𑁇𑁩 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀼𑀢𑁆𑀢

(53.)

395\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀯𑀟𑁆𑀠𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁; 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄; 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺; 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

396\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀝𑁆𑀝𑀺𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁄 𑀛𑀸𑀬𑁂𑀬𑁆𑀬𑁇 𑀢𑀢𑁆𑀭 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀢𑁂𑀮𑀁 𑀆𑀲𑀺𑀜𑁆𑀘𑁂𑀬𑁆𑀬 𑀯𑀝𑁆𑀝𑀺𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁄 𑀢𑀤𑀸𑀳𑀸𑀭𑁄 𑀢𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑁄 𑀘𑀺𑀭𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀚𑀮𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀯𑀟𑁆𑀠𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁; 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄; 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺; 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

397\. 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄; 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑀸 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑁄; 𑀪𑀯𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑁄; 𑀚𑀸𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

398\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀝𑁆𑀝𑀺𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁄 𑀛𑀸𑀬𑁂𑀬𑁆𑀬𑁇 𑀢𑀢𑁆𑀭 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀢𑁂𑀮𑀁 𑀆𑀲𑀺𑀜𑁆𑀘𑁂𑀬𑁆𑀬 𑀦 𑀯𑀝𑁆𑀝𑀺𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁄 𑀧𑀼𑀭𑀺𑀫𑀲𑁆𑀲 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀳𑀸𑀭𑀸 𑀅𑀦𑀸𑀳𑀸𑀭𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
