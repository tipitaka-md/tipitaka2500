# 𑁧𑁇𑁬𑁇𑁮 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀤𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

(58.)

416\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

417\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑁄𑁇 𑀢𑀲𑁆𑀲 𑀬𑀸𑀦𑀺 𑀘𑁂𑀯 𑀫𑀽𑀮𑀸𑀦𑀺 𑀅𑀥𑁄𑀕𑀫𑀸𑀦𑀺, 𑀬𑀸𑀦𑀺 𑀘 𑀢𑀺𑀭𑀺𑀬𑀗𑁆𑀕𑀫𑀸𑀦𑀺, 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀉𑀤𑁆𑀥𑀁 𑀑𑀚𑀁 𑀅𑀪𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑁄 𑀢𑀤𑀸𑀳𑀸𑀭𑁄 𑀢𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑁄 𑀘𑀺𑀭𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑁆𑀲𑀸𑀤𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇

418\. 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

419\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑁄𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀓𑀼𑀤𑁆𑀤𑀸𑀮𑀧𑀺𑀝𑀓𑀁 𑀆𑀤𑀸𑀬…𑀧𑁂…  𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀸𑀫𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
