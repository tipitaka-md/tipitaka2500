# 𑁪𑁇𑁧𑁇𑁯 𑀤𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀤𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

(132.)

851\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀡𑁆𑀟𑁄 𑀉𑀧𑀭𑀺𑀯𑁂𑀳𑀸𑀲𑀁 𑀔𑀺𑀢𑁆𑀢𑁄 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀫𑀽𑀮𑁂𑀦 𑀦𑀺𑀧𑀢𑀢𑀺, 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀫𑀚𑁆𑀛𑁂𑀦 𑀦𑀺𑀧𑀢𑀢𑀺, 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀅𑀦𑁆𑀢𑁂𑀦 𑀦𑀺𑀧𑀢𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸 𑀲𑀢𑁆𑀢𑀸 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀲𑀦𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀸 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑀸 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀅𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸 𑀧𑀭𑀁 𑀮𑁄𑀓𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀧𑀭𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄…𑀧𑁂…  𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
