# 𑁪𑁇𑁧𑁇𑁪 𑀔𑀻𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀔𑀻𑀭𑀲𑀼𑀢𑁆𑀢

(127.)

835\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑀁 𑀯𑀸 𑀯𑁄 𑀇𑀫𑀺𑀦𑀸 𑀤𑀻𑀖𑁂𑀦 𑀅𑀤𑁆𑀥𑀼𑀦𑀸 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁 𑀫𑀸𑀢𑀼𑀣𑀜𑁆𑀜𑀁 𑀧𑀻𑀢𑀁, 𑀬𑀁 𑀯𑀸 𑀘𑀢𑀽𑀲𑀼 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂𑀲𑀼 𑀉𑀤𑀓𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫, 𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀬𑀁 𑀦𑁄 𑀇𑀫𑀺𑀦𑀸 𑀤𑀻𑀖𑁂𑀦 𑀅𑀤𑁆𑀥𑀼𑀦𑀸 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁 𑀫𑀸𑀢𑀼𑀣𑀜𑁆𑀜𑀁 𑀧𑀻𑀢𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘𑀢𑀽𑀲𑀼 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂𑀲𑀼 𑀉𑀤𑀓𑀦𑁆”𑀢𑀺𑁇

836\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀣𑁇 𑀏𑀢𑀤𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀬𑀁 𑀯𑁄 𑀇𑀫𑀺𑀦𑀸 𑀤𑀻𑀖𑁂𑀦 𑀅𑀤𑁆𑀥𑀼𑀦𑀸 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁 𑀫𑀸𑀢𑀼𑀣𑀜𑁆𑀜𑀁 𑀧𑀻𑀢𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘𑀢𑀽𑀲𑀼 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂𑀲𑀼 𑀉𑀤𑀓𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄…𑀧𑁂…  𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
