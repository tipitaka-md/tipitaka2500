# 𑁪𑁇𑁧𑁇𑁮 𑀕𑀗𑁆𑀕𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀕𑀗𑁆𑀕𑀸𑀲𑀼𑀢𑁆𑀢

(131.)

847\. 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀻𑀯𑀩𑀳𑀼𑀓𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀓𑀧𑁆𑀧𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸”𑀢𑀺? “𑀩𑀳𑀼𑀓𑀸 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀓𑀧𑁆𑀧𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑁇 𑀢𑁂 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁—  ‘𑀏𑀢𑁆𑀢𑀓𑀸 𑀓𑀧𑁆𑀧𑀸 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸’”𑀢𑀺𑁇

848\. “𑀲𑀓𑁆𑀓𑀸 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀉𑀧𑀫𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀓𑁆𑀓𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀬𑀢𑁄 𑀘𑀸𑀬𑀁 𑀕𑀗𑁆𑀕𑀸 𑀦𑀤𑀻 𑀧𑀪𑀯𑀢𑀺 𑀬𑀢𑁆𑀣 𑀘 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑁂𑀢𑀺, 𑀬𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑁂 𑀯𑀸𑀮𑀺𑀓𑀸 𑀲𑀸 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁—  ‘𑀏𑀢𑁆𑀢𑀓𑀸 𑀯𑀸𑀮𑀺𑀓𑀸 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀸𑀮𑀺𑀓𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀸𑀮𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀸𑀮𑀺𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀢𑀢𑁄 𑀩𑀳𑀼𑀢𑀭𑀸 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀓𑀧𑁆𑀧𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑁇 𑀢𑁂 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁—  ‘𑀏𑀢𑁆𑀢𑀓𑀸 𑀓𑀧𑁆𑀧𑀸 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀏𑀯𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁 𑀢𑀺𑀩𑁆𑀩𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁 𑀩𑁆𑀬𑀲𑀦𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁, 𑀓𑀝𑀲𑀻 𑀯𑀟𑁆𑀠𑀺𑀢𑀸𑁇 𑀬𑀸𑀯𑀜𑁆𑀘𑀺𑀤𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀮𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀺𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

849\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
