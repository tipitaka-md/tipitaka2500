# 𑁪𑁇𑁧𑁇𑁭 𑀲𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

(130.)

844\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀻𑀯𑀩𑀳𑀼𑀓𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀧𑁆𑀧𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸”𑀢𑀺? “𑀩𑀳𑀼𑀓𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀧𑁆𑀧𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑁇 𑀢𑁂 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁—  ‘𑀏𑀢𑁆𑀢𑀓𑀸 𑀓𑀧𑁆𑀧𑀸 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸’”𑀢𑀺𑁇

845\. “𑀲𑀓𑁆𑀓𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀫𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀓𑁆𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 “𑀇𑀥𑀲𑁆𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀬𑀼𑀓𑀸 𑀯𑀲𑁆𑀲𑀲𑀢𑀚𑀻𑀯𑀺𑀦𑁄𑁇 𑀢𑁂 𑀤𑀺𑀯𑀲𑁂 𑀤𑀺𑀯𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀳𑀺 𑀓𑀧𑁆𑀧𑀸 𑀅𑀲𑁆𑀲𑀼, 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀬𑀼𑀓𑀸 𑀯𑀲𑁆𑀲𑀲𑀢𑀚𑀻𑀯𑀺𑀦𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀓𑀸𑀮𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀓𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀧𑁆𑀧𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑁇 𑀢𑁂 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁—  ‘𑀏𑀢𑁆𑀢𑀓𑀸 𑀓𑀧𑁆𑀧𑀸 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄…𑀧𑁂…  𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
