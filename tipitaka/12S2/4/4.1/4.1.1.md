# 𑁪𑁇𑁧𑁇𑁧 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

(124.)

827\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

828\. “𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀢𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀁𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀼𑀮𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀼𑀮𑀁 𑀖𑀝𑀺𑀓𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬—  ‘𑀅𑀬𑀁 𑀫𑁂 𑀫𑀸𑀢𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀫𑀸𑀢𑀼 𑀅𑀬𑀁 𑀫𑀸𑀢𑀸’𑀢𑀺, 𑀅𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀫𑀸𑀢𑀭𑁄 𑀅𑀲𑁆𑀲𑀼, 𑀅𑀣 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀏𑀯𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁 𑀢𑀺𑀩𑁆𑀩𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁 𑀩𑁆𑀬𑀲𑀦𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁, 𑀓𑀝𑀲𑀻 𑀯𑀟𑁆𑀠𑀺𑀢𑀸𑁇 𑀬𑀸𑀯𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀮𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀁 𑀅𑀮𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀺𑀢𑀼𑀁 𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
