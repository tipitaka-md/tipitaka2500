# 𑁪𑁇𑁧𑁇𑁫 𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

(128.)

838. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀆𑀭𑀸𑀫𑁂…𑀧𑁂… 𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀻𑀯𑀤𑀻𑀖𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀧𑁆𑀧𑁄”𑀢𑀺? “𑀤𑀻𑀖𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑀧𑁆𑀧𑁄𑁇 𑀲𑁄 𑀦 𑀲𑀼𑀓𑀭𑁄 𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸, 𑀏𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀢𑀺 𑀯𑀸”𑀢𑀺𑁇

839. “𑀲𑀓𑁆𑀓𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀫𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀓𑁆𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀳𑀸𑀲𑁂𑀮𑁄 𑀧𑀩𑁆𑀩𑀢𑁄 𑀬𑁄𑀚𑀦𑀁 𑀆𑀬𑀸𑀫𑁂𑀦 𑀬𑁄𑀚𑀦𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀬𑁄𑀚𑀦𑀁 𑀉𑀩𑁆𑀩𑁂𑀥𑁂𑀦 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀲𑀼𑀲𑀺𑀭𑁄 𑀏𑀓𑀕𑁆𑀖𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀓𑀸𑀲𑀺𑀓𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀔𑀺𑀧𑁆𑀧𑀢𑀭𑀁 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀳𑀸𑀲𑁂𑀮𑁄 𑀧𑀩𑁆𑀩𑀢𑁄 𑀇𑀫𑀺𑀦𑀸 𑀉𑀧𑀓𑁆𑀓𑀫𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀓𑀧𑁆𑀧𑁄𑁇 𑀏𑀯𑀁 𑀤𑀻𑀖𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑀧𑁆𑀧𑁄𑁇 𑀏𑀯𑀁 𑀤𑀻𑀖𑀸𑀦𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀦𑁂𑀓𑁄 𑀓𑀧𑁆𑀧𑁄 𑀲𑀁𑀲𑀺𑀢𑁄, 𑀦𑁂𑀓𑀁 𑀓𑀧𑁆𑀧𑀲𑀢𑀁 𑀲𑀁𑀲𑀺𑀢𑀁, 𑀦𑁂𑀓𑀁 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀁 𑀲𑀁𑀲𑀺𑀢𑀁, 𑀦𑁂𑀓𑀁 𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀲𑀁𑀲𑀺𑀢𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺…𑀧𑁂…  𑀬𑀸𑀯𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀮𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀺𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
