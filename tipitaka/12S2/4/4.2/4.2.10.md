# 𑁪𑁇𑁨𑁇𑁧𑁦 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

(143.)

885. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

886. “𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀢𑀁 𑀲𑀁𑀲𑀭𑀢𑀁𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀲𑁆𑀲 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 ‘𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑁄’𑀢𑁆𑀯𑁂𑀯 𑀲𑀫𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 ‘𑀢𑀺𑀯𑀭𑀸’𑀢𑁆𑀯𑁂𑀯 𑀲𑀫𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 𑀢𑀺𑀯𑀭𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀺𑀯𑀭𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀘𑀢𑀽𑀳𑁂𑀦 𑀆𑀭𑁄𑀳𑀦𑁆𑀢𑀺, 𑀘𑀢𑀽𑀳𑁂𑀦 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂𑀦 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀺𑀥𑀼𑀭𑀲𑀜𑁆𑀚𑀻𑀯𑀁 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀕𑁆𑀕𑀁 𑀪𑀤𑁆𑀤𑀬𑀼𑀕𑀁𑁇 𑀧𑀲𑁆𑀲𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸 𑀘𑁂𑀯𑀺𑀫𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀲𑀫𑀜𑁆𑀜𑀸 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸, 𑀢𑁂 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸, 𑀲𑁄 𑀘 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇 𑀏𑀯𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀏𑀯𑀁 𑀅𑀤𑁆𑀥𑀼𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀏𑀯𑀁 𑀅𑀦𑀲𑁆𑀲𑀸𑀲𑀺𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑁇 𑀬𑀸𑀯𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀮𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀺𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀁𑁇

_𑀢𑀺𑀁𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑁂𑀦 𑀘;_  
_𑀪𑀸𑀢𑀸 𑀪𑀕𑀺𑀦𑀻 𑀧𑀼𑀢𑁆𑀢𑁄 𑀘,_  
_𑀥𑀻𑀢𑀸 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀧𑀩𑁆𑀩𑀢𑀁𑁇_
