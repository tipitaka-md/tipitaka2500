# 𑁪𑁇𑁨𑁇𑁨 𑀲𑀼𑀔𑀺𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀼𑀔𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(135.)

866\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄…𑀧𑁂…  𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀣 𑀲𑀼𑀔𑀺𑀢𑀁 𑀲𑀼𑀲𑀚𑁆𑀚𑀺𑀢𑀁, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀫𑁆𑀳𑁂𑀳𑀺𑀧𑀺 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁 𑀇𑀫𑀺𑀦𑀸 𑀤𑀻𑀖𑁂𑀦 𑀅𑀤𑁆𑀥𑀼𑀦𑀸’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑀸𑀭𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺…𑀧𑁂…  𑀅𑀮𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
