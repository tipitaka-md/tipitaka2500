# 𑁨𑁇𑁧𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀪𑀺𑀲𑀫𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀪𑀺𑀲𑀫𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

(83.)

632\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀳𑀺𑀫𑀯𑀸 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢 𑀲𑀸𑀲𑀧𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑀁 𑀯𑀸 𑀳𑀺𑀫𑀯𑀢𑁄 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁 𑀬𑀸 𑀯𑀸 𑀲𑀢𑁆𑀢 𑀲𑀸𑀲𑀧𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸”𑀢𑀺?

633\. “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀳𑀺𑀫𑀯𑀢𑁄 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀬𑀤𑀺𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢 𑀲𑀸𑀲𑀧𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑁇 𑀦𑁂𑀯 𑀲𑀢𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀳𑀺𑀫𑀯𑀢𑁄 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀲𑀢𑁆𑀢 𑀲𑀸𑀲𑀧𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸”𑀢𑀺𑁇

634\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀪𑀺𑀲𑀫𑁂𑀢𑀸𑀯𑀺𑀦𑁄 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀁 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀁𑁇 𑀦𑁂𑀯 𑀲𑀢𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀢𑀺 𑀦 𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀢𑀺 𑀦 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀤𑀺𑀤𑀁 𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁𑀧𑀭𑀫𑀢𑀸𑁇 𑀏𑀯𑀁 𑀫𑀳𑀢𑁆𑀣𑀺𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄, 𑀏𑀯𑀁 𑀫𑀳𑀢𑁆𑀣𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀧𑀝𑀺𑀮𑀸𑀪𑁄”𑀢𑀺𑁇
