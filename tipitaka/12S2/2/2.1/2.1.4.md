# 𑁨𑁇𑁧𑁇𑁪 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀪𑀺𑀲𑀫𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀪𑀺𑀲𑀫𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀲𑀼𑀢𑁆𑀢

(77.)

614\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣𑀺𑀫𑀸 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄 𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀲𑀫𑁂𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀕𑀗𑁆𑀕𑀸 𑀬𑀫𑀼𑀦𑀸 𑀅𑀘𑀺𑀭𑀯𑀢𑀻 𑀲𑀭𑀪𑀽 𑀫𑀳𑀻, 𑀢𑀁 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑀁 𑀯𑀸 𑀲𑀫𑁆𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁 𑀬𑀸𑀦𑀺 𑀯𑀸 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀻”𑀢𑀺?

615\. “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀲𑀫𑁆𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀺𑁇 𑀦𑁂𑀯 𑀲𑀢𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀪𑁂𑀚𑁆𑀚𑁅𑀤𑀓𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀡𑁆𑀡𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀧𑀝𑀺𑀮𑀸𑀪𑁄”𑀢𑀺𑁇
