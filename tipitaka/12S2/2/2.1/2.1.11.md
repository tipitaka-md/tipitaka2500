# 𑁨𑁇𑁧𑁇𑁧𑁧 𑀢𑀢𑀺𑀬𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀪𑀺𑀲𑀫𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀪𑀺𑀲𑀫𑀬𑀯𑀕𑁆𑀕

* 𑀢𑀢𑀺𑀬𑀧𑀩𑁆𑀩𑀢𑀲𑀼𑀢𑁆𑀢

(84.)

636\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀺𑀦𑁂𑀭𑀼𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑀸 𑀯𑀸 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀬𑁄 𑀯𑀸 𑀲𑀺𑀦𑁂𑀭𑀼 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀸”𑀢𑀺?

637\. “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀺𑀦𑁂𑀭𑀼 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀸; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑁇 𑀦𑁂𑀯 𑀲𑀢𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑀺𑀦𑁂𑀭𑀼𑀁 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀸𑀦𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀲𑀢𑁆𑀢 𑀫𑀼𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸”𑀢𑀺𑁇

638\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀅𑀥𑀺𑀕𑀫𑁄 𑀦𑁂𑀯 𑀲𑀢𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀢𑀺 𑀦 𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀢𑀺 𑀦 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀫𑀳𑀸𑀥𑀺𑀕𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄, 𑀏𑀯𑀁 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑁄”𑀢𑀺𑁇
