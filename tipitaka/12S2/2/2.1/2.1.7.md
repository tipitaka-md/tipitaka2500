# 𑁨𑁇𑁧𑁇𑁭 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀪𑀺𑀲𑀫𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀪𑀺𑀲𑀫𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(80.)

623\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀢𑁄 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀉𑀤𑁆𑀥𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑀸𑀦𑀺 𑀯𑀸 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀉𑀩𑁆𑀪𑀢𑀸𑀦𑀺 𑀬𑀁 𑀯𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀦𑁆”𑀢𑀺?

624\. “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁; 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀉𑀩𑁆𑀪𑀢𑀸𑀦𑀺𑁇 𑀦𑁂𑀯 𑀲𑀢𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀦 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀫𑀁 𑀓𑀮𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀻𑀡𑀺 𑀯𑀸 𑀉𑀤𑀓𑀨𑀼𑀲𑀺𑀢𑀸𑀦𑀺 𑀉𑀩𑁆𑀪𑀢𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀧𑀝𑀺𑀮𑀸𑀪𑁄”𑀢𑀺𑁇
