# 𑁫𑁇𑁧𑁇𑁪 𑀓𑀼𑀮𑀽𑀧𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀮𑀽𑀧𑀓𑀲𑀼𑀢𑁆𑀢

(147.)

920. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀣𑀁𑀭𑀽𑀧𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀢𑀺 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀼𑀁, 𑀓𑀣𑀁𑀭𑀽𑀧𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀅𑀭𑀳𑀢𑀺 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀼𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

921. “𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺—  ‘𑀤𑁂𑀦𑁆𑀢𑀼𑀬𑁂𑀯 𑀫𑁂, 𑀫𑀸 𑀦𑀸𑀤𑀁𑀲𑀼; 𑀩𑀳𑀼𑀓𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀣𑁄𑀓𑀁; 𑀧𑀡𑀻𑀢𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀮𑀽𑀔𑀁; 𑀲𑀻𑀖𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀤𑀦𑁆𑀥𑀁; 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑁄 𑀦 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀣𑁄𑀓𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀩𑀳𑀼𑀓𑀁…𑀧𑁂…  𑀮𑀽𑀔𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀡𑀻𑀢𑀁…  𑀤𑀦𑁆𑀥𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀲𑀻𑀖𑀁, 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁; 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀅𑀭𑀳𑀢𑀺 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀼𑀁𑁇

922. 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺—  ‘𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣 𑀮𑀩𑁆𑀪𑀸 𑀧𑀭𑀓𑀼𑀮𑁂𑀲𑀼—  𑀤𑁂𑀦𑁆𑀢𑀼𑀬𑁂𑀯 𑀫𑁂, 𑀫𑀸 𑀦𑀸𑀤𑀁𑀲𑀼; 𑀩𑀳𑀼𑀓𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀣𑁄𑀓𑀁; 𑀧𑀡𑀻𑀢𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀮𑀽𑀔𑀁; 𑀲𑀻𑀖𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀤𑀦𑁆𑀥𑀁; 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑁄 𑀦 𑀤𑁂𑀦𑁆𑀢𑀺; 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀣𑁄𑀓𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀩𑀳𑀼𑀓𑀁; 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀮𑀽𑀔𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀡𑀻𑀢𑀁; 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀤𑀦𑁆𑀥𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀲𑀻𑀖𑀁; 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁; 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀢𑀺 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀼𑀁𑁇

923. 𑀓𑀲𑁆𑀲𑀧𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺—  ‘𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣 𑀮𑀩𑁆𑀪𑀸 𑀧𑀭𑀓𑀼𑀮𑁂𑀲𑀼—  𑀤𑁂𑀦𑁆𑀢𑀼𑀬𑁂𑀯 𑀫𑁂, 𑀫𑀸 𑀦𑀸𑀤𑀁𑀲𑀼; 𑀩𑀳𑀼𑀓𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀣𑁄𑀓𑀁; 𑀧𑀡𑀻𑀢𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀮𑀽𑀔𑀁; 𑀲𑀻𑀖𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀤𑀦𑁆𑀥𑀁; 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀜𑁆𑀜𑁂𑀯 𑀫𑁂 𑀤𑁂𑀦𑁆𑀢𑀼, 𑀫𑀸 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀏𑀯𑀁𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑁄 𑀦 𑀤𑁂𑀦𑁆𑀢𑀺; 𑀢𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀣𑁄𑀓𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀩𑀳𑀼𑀓𑀁; 𑀢𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀮𑀽𑀔𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀡𑀻𑀢𑀁; 𑀢𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀤𑀦𑁆𑀥𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀲𑀻𑀖𑀁; 𑀢𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁; 𑀢𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀦 𑀲𑀦𑁆𑀤𑀻𑀬𑀢𑀺; 𑀲𑁄 𑀦 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀯𑀸 𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀯𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁄 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀧𑀲𑀤𑀺𑀲𑁄𑁇 𑀑𑀯𑀤𑀺𑀢𑁂𑀳𑀺 𑀘 𑀧𑀦 𑀯𑁄 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
