# 𑁫𑁇𑁧𑁇𑁬 𑀑𑀯𑀸𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

* 𑀑𑀯𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(149.)

933\. 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑁂𑀍𑀉𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀑𑀯𑀤, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽; 𑀓𑀭𑁄𑀳𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁𑁇 𑀅𑀳𑀁 𑀯𑀸, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀯𑀤𑁂𑀬𑁆𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀸; 𑀅𑀳𑀁 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀸”𑀢𑀺𑁇

934\. “𑀤𑀼𑀩𑁆𑀩𑀘𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀅𑀓𑁆𑀔𑀫𑀸, 𑀅𑀧𑁆𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀺𑀦𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀤𑁆𑀤𑀲𑀁 𑀪𑀡𑁆𑀟𑀜𑁆𑀘 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀅𑀪𑀺𑀚𑀺𑀓𑀜𑁆𑀘 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀦𑁆𑀢𑁂—  ‘𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀩𑀳𑀼𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀘𑀺𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇

935\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀡𑁆𑀟𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀅𑀪𑀺𑀚𑀺𑀓𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇

936\. “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀣—  ‘𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀩𑀳𑀼𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀘𑀺𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀣—  ‘𑀏𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀣—  𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀩𑀳𑀼𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀘𑀺𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀦𑁄 𑀘𑁂 𑀓𑀺𑀭 𑀫𑁂 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀣, 𑀅𑀣 𑀓𑀺𑀜𑁆𑀘𑀭𑀳𑀺 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀓𑀺𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀓𑀺𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀣—  ‘𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀩𑀳𑀼𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀘𑀺𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇

937\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀘𑁆𑀘𑀬𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀘𑁆𑀘𑀕𑀫𑀸, 𑀬𑀣𑀸𑀩𑀸𑀮𑁂 𑀬𑀣𑀸𑀫𑀽𑀍𑀳𑁂 𑀬𑀣𑀸𑀅𑀓𑀼𑀲𑀮𑁂, 𑀬𑁂 𑀫𑀬𑀁 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀺𑀫𑁆𑀳—  ‘𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀩𑀳𑀼𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀘𑀺𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀆𑀬𑀢𑀺𑀁 𑀲𑀁𑀯𑀭𑀸𑀬𑀸”𑀢𑀺𑁇

938\. “𑀢𑀕𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁆𑀘𑀬𑁄 𑀅𑀘𑁆𑀘𑀕𑀫𑀸 𑀬𑀣𑀸𑀩𑀸𑀮𑁂 𑀬𑀣𑀸𑀫𑀽𑀍𑀳𑁂 𑀬𑀣𑀸𑀅𑀓𑀼𑀲𑀮𑁂, 𑀬𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀺𑀢𑁆𑀣—  ‘𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀩𑀳𑀼𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑁄 𑀘𑀺𑀭𑀢𑀭𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀭𑁄𑀣, 𑀢𑀁 𑀯𑁄 𑀫𑀬𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫𑁇 𑀯𑀼𑀤𑁆𑀥𑀺 𑀳𑁂𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀬𑁄 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀭𑁄𑀢𑀺 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
