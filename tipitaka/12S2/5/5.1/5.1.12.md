# 𑁫𑁇𑁧𑁇𑁧𑁨 𑀧𑀭𑀁𑀫𑀭𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

### 𑀧𑀭𑀁𑀫𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(155.)

1007. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺, 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀓𑀲𑁆𑀫𑀸 𑀘𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺? “𑀦 𑀳𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁 𑀦𑀸𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀓𑀁 𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀦 𑀉𑀧𑀲𑀫𑀸𑀬 𑀦 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀦 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇

1008. “𑀅𑀣 𑀓𑀺𑀜𑁆𑀘𑀭𑀳𑀸𑀯𑀼𑀲𑁄, 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺? “𑀇𑀤𑀁 ‘𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸; 𑀅𑀬𑀁 ‘𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸; 𑀅𑀬𑀁 ‘𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸; 𑀅𑀬𑀁 ‘𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇 “𑀓𑀲𑁆𑀫𑀸 𑀘𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺? “𑀏𑀢𑀜𑁆𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁 𑀏𑀢𑀁 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀓𑀁 𑀏𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀪𑀕𑀯𑀢𑀸”𑀢𑀺𑁇
