# 𑁫𑁇𑁧𑁇𑁧𑁦 𑀉𑀧𑀲𑁆𑀲𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀲𑁆𑀲𑀬𑀲𑀼𑀢𑁆𑀢

(153.)

977\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀢𑁆𑀯𑀁 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀢𑁆𑀯𑀁 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

978\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀧𑀘𑁆𑀙𑀸𑀲𑀫𑀡𑁂𑀦 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

979\. 𑀅𑀣 𑀔𑁄 𑀣𑀼𑀮𑁆𑀮𑀢𑀺𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀯𑀸𑀘𑀁 𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀧𑀦 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀯𑁂𑀤𑁂𑀳𑀫𑀼𑀦𑀺𑀦𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀢𑀺? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀽𑀘𑀺𑀯𑀸𑀡𑀺𑀚𑀓𑁄 𑀲𑀽𑀘𑀺𑀓𑀸𑀭𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀲𑀽𑀘𑀺𑀁 𑀯𑀺𑀓𑁆𑀓𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀯𑁂𑀤𑁂𑀳𑀫𑀼𑀦𑀺𑀦𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

980\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑀼𑀮𑁆𑀮𑀢𑀺𑀲𑁆𑀲𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀇𑀫𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀬𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀳𑀁 𑀲𑀽𑀘𑀺𑀯𑀸𑀡𑀺𑀚𑀓𑁄, 𑀢𑁆𑀯𑀁 𑀲𑀽𑀘𑀺𑀓𑀸𑀭𑁄; 𑀉𑀤𑀸𑀳𑀼 𑀅𑀳𑀁 𑀲𑀽𑀘𑀺𑀓𑀸𑀭𑁄, 𑀢𑁆𑀯𑀁 𑀲𑀽𑀘𑀺𑀯𑀸𑀡𑀺𑀚𑀓𑁄”𑀢𑀺? “𑀔𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀩𑀸𑀮𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄”𑀢𑀺𑁇 “𑀆𑀕𑀫𑁂𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸 𑀢𑁂 𑀲𑀁𑀖𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑁇

981\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁆𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀉𑀧𑀦𑀻𑀢𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀤𑁂 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀆𑀦𑀦𑁆𑀤𑁄𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀤𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

982\. “𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀉𑀧𑀦𑀻𑀢𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀤𑁂 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀓𑀲𑁆𑀲𑀧𑁄𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀤𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻’𑀢𑀺…𑀧𑁂… 𑁇 (𑀦𑀯𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀜𑁆𑀘 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁄 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄𑁇)

983\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁆𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀉𑀧𑀦𑀻𑀢𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀆𑀦𑀦𑁆𑀤𑁄𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

984\. “𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀉𑀧𑀦𑀻𑀢𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀓𑀲𑁆𑀲𑀧𑁄𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻’𑀢𑀺𑁇

985\. 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀁 𑀯𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀸𑀕𑀁 𑀅𑀟𑁆𑀠𑀝𑁆𑀞𑀫𑀭𑀢𑀦𑀁 𑀯𑀸 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀙𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀫𑁂 𑀙 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀙𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

986\. 𑀘𑀯𑀺𑀢𑁆𑀣 𑀘 𑀧𑀦 𑀣𑀼𑀮𑁆𑀮𑀢𑀺𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀸𑀢𑀺𑁇

---
