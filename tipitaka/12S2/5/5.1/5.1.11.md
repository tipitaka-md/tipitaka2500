# 𑁫𑁇𑁧𑁇𑁧𑁧 𑀘𑀻𑀯𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

### 𑀘𑀻𑀯𑀭𑀲𑀼𑀢𑁆𑀢

(154.)

988\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀢𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁𑁇

989\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀺𑀁𑀲𑀫𑀢𑁆𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀓𑀼𑀫𑀸𑀭𑀪𑀽𑀢𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀓𑀼𑀮𑁂𑀲𑀼 𑀢𑀺𑀓𑀪𑁄𑀚𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆”𑀢𑀺?

990\. “𑀢𑀬𑁄 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀓𑀼𑀮𑁂𑀲𑀼 𑀢𑀺𑀓𑀪𑁄𑀚𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬, 𑀫𑀸 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀔𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀼𑀁, 𑀓𑀼𑀮𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀢𑀸𑀬 𑀘𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀢𑀬𑁄 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀓𑀼𑀮𑁂𑀲𑀼 𑀢𑀺𑀓𑀪𑁄𑀚𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

991\. “𑀅𑀣 𑀓𑀺𑀜𑁆𑀘𑀭𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑁂𑀳𑀺 𑀦𑀯𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁂𑀳𑀺 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽𑀳𑀺 𑀚𑀸𑀕𑀭𑀺𑀬𑀁 𑀅𑀦𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀲𑀺? 𑀲𑀲𑁆𑀲𑀖𑀸𑀢𑀁 𑀫𑀜𑁆𑀜𑁂 𑀘𑀭𑀲𑀺, 𑀓𑀼𑀮𑀽𑀧𑀖𑀸𑀢𑀁 𑀫𑀜𑁆𑀜𑁂 𑀘𑀭𑀲𑀺𑁇 𑀑𑀮𑀼𑀚𑁆𑀚𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀭𑀺𑀲𑀸; 𑀧𑀮𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀯𑀧𑁆𑀧𑀸𑀬𑀸𑁇 𑀦 𑀯𑀸𑀬𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑁄 𑀫𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀲𑀻”𑀢𑀺𑁇

992\. “𑀅𑀧𑀺 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀺𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀮𑀺𑀢𑀸𑀦𑀺 𑀚𑀸𑀢𑀸𑀦𑀺𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀫𑀬𑀁 𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀓𑀯𑀸𑀤𑀸 𑀦 𑀫𑀼𑀘𑁆𑀘𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑁂𑀳𑀺 𑀦𑀯𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁂𑀳𑀺 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽𑀳𑀺 𑀚𑀸𑀕𑀭𑀺𑀬𑀁 𑀅𑀦𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀲𑀺, 𑀲𑀲𑁆𑀲𑀖𑀸𑀢𑀁 𑀫𑀜𑁆𑀜𑁂 𑀘𑀭𑀲𑀺, 𑀓𑀼𑀮𑀽𑀧𑀖𑀸𑀢𑀁 𑀫𑀜𑁆𑀜𑁂 𑀘𑀭𑀲𑀺𑁇 𑀑𑀮𑀼𑀚𑁆𑀚𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀭𑀺𑀲𑀸; 𑀧𑀮𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀯𑀧𑁆𑀧𑀸𑀬𑀸𑁇 𑀦 𑀯𑀸𑀬𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑁄 𑀫𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀲𑀻”𑀢𑀺𑁇

993\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀅𑀬𑁆𑀬𑁂𑀦 𑀓𑀺𑀭 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀅𑀬𑁆𑀬𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀯𑁂𑀤𑁂𑀳𑀫𑀼𑀦𑀺 𑀓𑀼𑀫𑀸𑀭𑀓𑀯𑀸𑀤𑁂𑀦 𑀅𑀧𑀲𑀸𑀤𑀺𑀢𑁄”𑀢𑀺𑁇

994\. 𑀅𑀣 𑀔𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀯𑀸𑀘𑀁 𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀧𑀦 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀼𑀩𑁆𑀩𑁄 𑀲𑀫𑀸𑀦𑁄 𑀅𑀬𑁆𑀬𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀯𑁂𑀤𑁂𑀳𑀫𑀼𑀦𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀯𑀸𑀤𑁂𑀦 𑀅𑀧𑀲𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀇𑀫𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀬𑁇

995\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀕𑁆𑀖𑀸𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀳𑀲𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀯𑀸𑀘𑀸 𑀪𑀸𑀲𑀺𑀢𑀸𑁇 𑀬𑀢𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀸, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑀲𑁆𑀲 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀫𑁆𑀩𑀸𑀥𑁄 𑀖𑀭𑀸𑀯𑀸𑀲𑁄 𑀭𑀚𑀸𑀧𑀣𑁄, 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑁇 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀝𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀬𑁂 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀢𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀁𑁇

996\. 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀭𑀸𑀚𑀕𑀳𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀁 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀢𑁆𑀣𑀸𑀭𑀜𑁆𑀘 𑀯𑀢𑀸𑀳𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁, 𑀪𑀕𑀯𑀦𑁆𑀢𑀫𑁂𑀯 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁; 𑀲𑀼𑀕𑀢𑀜𑁆𑀘 𑀯𑀢𑀸𑀳𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁, 𑀪𑀕𑀯𑀦𑁆𑀢𑀫𑁂𑀯 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁; 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑀸𑀳𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁; 𑀪𑀕𑀯𑀦𑁆𑀢𑀫𑁂𑀯 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀲𑀢𑁆𑀣𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀯𑀓𑁄𑀳𑀫𑀲𑁆𑀫𑀺; 𑀲𑀢𑁆𑀣𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀯𑀓𑁄𑀳𑀫𑀲𑁆𑀫𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀬𑁄 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀸𑀯𑀓𑀁 𑀅𑀚𑀸𑀦𑀜𑁆𑀜𑁂𑀯 𑀯𑀤𑁂𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀻𑀢𑀺, 𑀅𑀧𑀲𑁆𑀲𑀜𑁆𑀜𑁂𑀯 𑀯𑀤𑁂𑀬𑁆𑀬 𑀧𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺, 𑀫𑀼𑀤𑁆𑀥𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑁇 𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦, 𑀓𑀲𑁆𑀲𑀧, 𑀚𑀸𑀦𑀜𑁆𑀜𑁂𑀯 𑀯𑀤𑀸𑀫𑀺 𑀚𑀸𑀦𑀸𑀫𑀻𑀢𑀺, 𑀧𑀲𑁆𑀲𑀜𑁆𑀜𑁂𑀯 𑀯𑀤𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑁇

997\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  “𑀢𑀺𑀩𑁆𑀩𑀁 𑀫𑁂 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀣𑁂𑀭𑁂𑀲𑀼 𑀦𑀯𑁂𑀲𑀼 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀽”𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

998\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀼𑀲𑀮𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀑𑀳𑀺𑀢𑀲𑁄𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

999\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀸𑀢𑀲𑀳𑀕𑀢𑀸 𑀘 𑀫𑁂 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺 𑀦 𑀯𑀺𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺’𑁇

1000\. 𑀅𑀣 𑀔𑁄 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀇𑀫𑀺𑀦𑀸 𑀑𑀯𑀸𑀤𑁂𑀦 𑀑𑀯𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀫𑁂𑀯 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀭𑀡𑁄 𑀭𑀝𑁆𑀞𑀧𑀺𑀡𑁆𑀟𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀁 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

1001\. 𑀅𑀣 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀝𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀢𑀼, 𑀬𑀁 𑀫𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸’𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀫𑀼𑀤𑀼𑀓𑀸 𑀔𑁄 𑀢𑁆𑀬𑀸𑀬𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀝𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀻’𑀢𑀺𑁇 ‘𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀝𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 ‘𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀲𑀺 𑀧𑀦 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀸𑀡𑀸𑀦𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀲𑀦𑀸𑀦𑀻’𑀢𑀺𑁇 ‘𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀡𑀸𑀦𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀲𑀦𑀸𑀦𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀝𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑀸𑀲𑀺𑀁𑁇 𑀅𑀳𑀁 𑀧𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀡𑀸𑀦𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀁𑁇

1002\. 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀢𑁆𑀢𑁄 𑀑𑀭𑀲𑁄 𑀫𑀼𑀔𑀢𑁄 𑀚𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑀚𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑁄, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀲𑀸𑀡𑀸𑀦𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀲𑀦𑀸𑀦𑀻’𑀢𑀺, 𑀫𑀫𑀁 𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀢𑁆𑀢𑁄 𑀑𑀭𑀲𑁄 𑀫𑀼𑀔𑀢𑁄 𑀚𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑀚𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑁄, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀲𑀸𑀡𑀸𑀦𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀲𑀦𑀸𑀦𑀻’𑀢𑀺𑁇

1003\. 𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀯𑀤𑁂 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀯𑀤𑁂 𑀆𑀓𑀗𑁆𑀔𑀸𑀫𑀺…𑀧𑁂…  (𑀦𑀯𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀜𑁆𑀘 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁄 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄𑁇)

1004\. 𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺; 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀁 𑀯𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀸𑀕𑀁 𑀅𑀟𑁆𑀠𑀝𑁆𑀞𑀫𑀭𑀢𑀦𑀁 𑀯𑀸 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀙𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀫𑁂 𑀙 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀙𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

1005\. 𑀘𑀯𑀺𑀢𑁆𑀣 𑀘 𑀧𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀸𑀢𑀺𑁇

---
