# 𑁫𑁇𑁧𑁇𑁧 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

* 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

* 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

(144.)

898\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦, 𑀇𑀢𑀭𑀻𑀢𑀭𑀘𑀻𑀯𑀭𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻; 𑀦 𑀘 𑀘𑀻𑀯𑀭𑀳𑁂𑀢𑀼 𑀅𑀦𑁂𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀮𑀤𑁆𑀥𑀸 𑀘 𑀘𑀻𑀯𑀭𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺; 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀘𑀻𑀯𑀭𑀁 𑀅𑀕𑀥𑀺𑀢𑁄 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇

899\. 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀦, 𑀇𑀢𑀭𑀻𑀢𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻; 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀳𑁂𑀢𑀼 𑀅𑀦𑁂𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀮𑀤𑁆𑀥𑀸 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺; 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀅𑀕𑀥𑀺𑀢𑁄 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇

900\. 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀦, 𑀇𑀢𑀭𑀻𑀢𑀭𑀲𑁂𑀦𑀸𑀲𑀦𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻; 𑀦 𑀘 𑀲𑁂𑀦𑀸𑀲𑀦𑀳𑁂𑀢𑀼 𑀅𑀦𑁂𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀮𑀤𑁆𑀥𑀸 𑀘 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺; 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀅𑀕𑀥𑀺𑀢𑁄 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇

901\. 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦, 𑀇𑀢𑀭𑀻𑀢𑀭𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻; 𑀦 𑀘 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀳𑁂𑀢𑀼 𑀅𑀦𑁂𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀮𑀤𑁆𑀥𑀸 𑀘 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺; 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀅𑀕𑀥𑀺𑀢𑁄 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇

902\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦, 𑀇𑀢𑀭𑀻𑀢𑀭𑀘𑀻𑀯𑀭𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄; 𑀦 𑀘 𑀘𑀻𑀯𑀭𑀳𑁂𑀢𑀼 𑀅𑀦𑁂𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫; 𑀅𑀮𑀤𑁆𑀥𑀸 𑀘 𑀘𑀻𑀯𑀭𑀁 𑀦 𑀘 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫; 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀘𑀻𑀯𑀭𑀁 𑀅𑀕𑀥𑀺𑀢𑀸 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀸 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫’𑁇 (𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)

903\. ‘𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀦…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀦…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦, 𑀇𑀢𑀭𑀻𑀢𑀭𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄; 𑀦 𑀘 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀳𑁂𑀢𑀼 𑀅𑀦𑁂𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀮𑀤𑁆𑀥𑀸 𑀘 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫; 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀅𑀕𑀥𑀺𑀢𑀸 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀸 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀯𑀸 𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀯𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁄 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀧𑀲𑀤𑀺𑀲𑁄, 𑀑𑀯𑀤𑀺𑀢𑁂𑀳𑀺 𑀘 𑀧𑀦 𑀯𑁄 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
