# 𑁫𑁇𑁧𑁇𑁮 𑀢𑀢𑀺𑀬𑀑𑀯𑀸𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

### 𑀢𑀢𑀺𑀬𑀑𑀯𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(151.)

953. 𑀭𑀸𑀚𑀕𑀳𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀑𑀯𑀤, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽; 𑀓𑀭𑁄𑀳𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁𑁇 𑀅𑀳𑀁 𑀯𑀸, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀑𑀯𑀤𑁂𑀬𑁆𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀸; 𑀅𑀳𑀁 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀸”𑀢𑀺𑁇

954. “𑀤𑀼𑀩𑁆𑀩𑀘𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀅𑀓𑁆𑀔𑀫𑀸, 𑀅𑀧𑁆𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀺𑀦𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄𑁇

955. 𑀢𑀢𑁆𑀭 𑀬𑁄 𑀳𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘𑁂𑀯 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘𑁂𑀯 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘𑁂𑀯 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑁄 𑀘𑁂𑀯 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀘𑁂𑀯 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀅𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀢𑀁 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀲𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺—  ‘𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀦𑀸𑀫𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀤𑁆𑀤𑀓𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼; 𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀤𑀁 𑀆𑀲𑀦𑀁 𑀦𑀺𑀲𑀻𑀤𑀸𑀳𑀻’𑀢𑀺𑁇

956. 𑀢𑀢𑁆𑀭, 𑀓𑀲𑁆𑀲𑀧, 𑀦𑀯𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀬𑁄 𑀓𑀺𑀭 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘𑁂𑀯 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘𑁂𑀯…𑀧𑁂…  𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀘𑁂𑀯…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀯…  𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀢𑀁 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀲𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺—  𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀦𑀸𑀫𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀤𑁆𑀤𑀓𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼; 𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀤𑀁 𑀆𑀲𑀦𑀁 𑀦𑀺𑀲𑀻𑀤𑀸𑀳𑀻’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺; 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇

957. 𑀏𑀢𑀭𑀳𑀺 𑀧𑀦, 𑀓𑀲𑁆𑀲𑀧, 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀘𑁂𑀯 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀦 𑀘 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸 𑀦 𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀸 𑀦 𑀘 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀦 𑀘 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀦 𑀘 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸 𑀦 𑀘 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀦 𑀘 𑀅𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀘𑁂𑀯 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀦 𑀘 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄𑁇

958. 𑀢𑀢𑁆𑀭 𑀬𑁄 𑀳𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀜𑀸𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀻 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀢𑀁 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀲𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺—  ‘𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀦𑀸𑀫𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀤𑁆𑀤𑀓𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀓𑀸𑀫𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼; 𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀤𑀁 𑀆𑀲𑀦𑀁 𑀦𑀺𑀲𑀻𑀤𑀸𑀳𑀻’𑀢𑀺𑁇

959. 𑀢𑀢𑁆𑀭, 𑀓𑀲𑁆𑀲𑀧, 𑀦𑀯𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀬𑁄 𑀓𑀺𑀭 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀜𑀸𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀻 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀢𑀁 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀲𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺—  𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑁄 𑀦𑀸𑀫𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀤𑁆𑀤𑀓𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀓𑀸𑀫𑁄 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼; 𑀏𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀤𑀁 𑀆𑀲𑀦𑀁 𑀦𑀺𑀲𑀻𑀤𑀸𑀳𑀻’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀉𑀧𑀤𑁆𑀤𑀼𑀢𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀽𑀧𑀤𑁆𑀤𑀯𑁂𑀦 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀦𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀦𑁂𑀦𑀸’𑀢𑀺, 𑀏𑀢𑀭𑀳𑀺 𑀢𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀉𑀧𑀤𑁆𑀤𑀼𑀢𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀽𑀧𑀤𑁆𑀤𑀯𑁂𑀦 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀦𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀦𑁂𑀦𑀸’”𑀢𑀺𑁇

---

960. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
