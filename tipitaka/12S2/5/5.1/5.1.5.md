# 𑁫𑁇𑁧𑁇𑁫 𑀚𑀺𑀡𑁆𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

### 𑀚𑀺𑀡𑁆𑀡𑀲𑀼𑀢𑁆𑀢

(148.)

925. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁…  𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑁂𑀍𑀉𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀚𑀺𑀡𑁆𑀡𑁄𑀲𑀺 𑀤𑀸𑀦𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀕𑀭𑀼𑀓𑀸𑀦𑀺 𑀘 𑀢𑁂 𑀇𑀫𑀸𑀦𑀺 𑀲𑀸𑀡𑀸𑀦𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀲𑀦𑀸𑀦𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀕𑀳𑀧𑀢𑀸𑀦𑀺 𑀘𑁂𑀯 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀳𑀺, 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀸𑀦𑀺 𑀘 𑀪𑀼𑀜𑁆𑀚𑀸𑀳𑀺, 𑀫𑀫 𑀘 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀭𑀸𑀳𑀻”𑀢𑀺𑁇

926. “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘𑁂𑀯 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘𑁂𑀯 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘𑁂𑀯 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑁄 𑀘𑁂𑀯 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀘𑁂𑀯 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀅𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻”𑀢𑀺𑁇

927. “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘𑁂𑀯 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘𑁂𑀯…𑀧𑁂…  𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀘𑁂𑀯…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀯…  𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻”𑀢𑀺?

928. “𑀤𑁆𑀯𑁂 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘𑁂𑀯 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘𑁂𑀯…𑀧𑁂…  𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀘𑁂𑀯…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀯…  𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻𑁇

929. 𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄, 𑀧𑀘𑁆𑀙𑀺𑀫𑀜𑁆𑀘 𑀚𑀦𑀢𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀫𑀸𑀦𑁄—  ‘𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑁂 𑀓𑀺𑀭 𑀢𑁂 𑀅𑀳𑁂𑀲𑀼𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑀸 𑀢𑁂 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄…𑀧𑁂…  𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 …𑀧𑁂…  𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁…  𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁…  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁…  𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁…  𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁…  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀘𑁂𑀯 𑀅𑀳𑁂𑀲𑀼𑀁 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀺𑀦𑁄’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇

930. 𑀇𑀫𑁂 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘𑁂𑀯 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘𑁂𑀯…𑀧𑁂…  𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑁄 𑀘𑁂𑀯…  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀘𑁂𑀯…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀘𑁂𑀯…  𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀘𑁂𑀯…  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻”𑀢𑀺𑁇

931. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀓𑀲𑁆𑀲𑀧𑁇 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀸𑀡𑀸𑀦𑀺 𑀘𑁂𑀯 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀲𑀦𑀸𑀦𑀺, 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘 𑀘𑀭𑀸𑀳𑀺, 𑀅𑀭𑀜𑁆𑀜𑁂 𑀘 𑀯𑀺𑀳𑀭𑀸𑀳𑀻”𑀢𑀺𑁇

---
