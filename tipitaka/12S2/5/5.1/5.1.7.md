# 𑁫𑁇𑁧𑁇𑁭 𑀤𑀼𑀢𑀺𑀬𑀑𑀯𑀸𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀑𑀯𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(150.)

940. 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀑𑀯𑀤, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽; 𑀓𑀭𑁄𑀳𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁𑁇 𑀅𑀳𑀁 𑀯𑀸, 𑀓𑀲𑁆𑀲𑀧, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀯𑀤𑁂𑀬𑁆𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀸; 𑀅𑀳𑀁 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀸”𑀢𑀺𑁇

941. “𑀤𑀼𑀩𑁆𑀩𑀘𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀓𑁆𑀔𑀫𑀸 𑀅𑀧𑁆𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀺𑀦𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀤𑁆𑀥𑀸 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀳𑀺𑀭𑀻 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀯𑀻𑀭𑀺𑀬𑀁 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀧𑀜𑁆𑀜𑀸 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀳𑀸𑀦𑀺𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀦𑁄 𑀯𑀼𑀤𑁆𑀥𑀺𑁇

942. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀍𑀅𑀧𑀓𑁆𑀔𑁂 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀳𑀸𑀬𑀢𑁂𑀯 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀳𑀸𑀬𑀢𑀺 𑀫𑀡𑁆𑀟𑀮𑁂𑀦, 𑀳𑀸𑀬𑀢𑀺 𑀆𑀪𑀸𑀬, 𑀳𑀸𑀬𑀢𑀺 𑀆𑀭𑁄𑀳𑀧𑀭𑀺𑀡𑀸𑀳𑁂𑀦𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀲𑀤𑁆𑀥𑀸 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼…𑀧𑁂…  𑀳𑀺𑀭𑀻 𑀦𑀢𑁆𑀣𑀺…  𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀦𑀢𑁆𑀣𑀺…  𑀯𑀻𑀭𑀺𑀬𑀁 𑀦𑀢𑁆𑀣𑀺…  𑀧𑀜𑁆𑀜𑀸 𑀦𑀢𑁆𑀣𑀺…  𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀳𑀸𑀦𑀺𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀦𑁄 𑀯𑀼𑀤𑁆𑀥𑀺𑁇

943. ‘𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀅𑀳𑀺𑀭𑀺𑀓𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀓𑀼𑀲𑀻𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀓𑁄𑀥𑀦𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀉𑀧𑀦𑀸𑀳𑀻 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀦 𑀲𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀯𑀸𑀤𑀓𑀸’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇

944. 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀤𑁆𑀥𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀳𑀺𑀭𑀻 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

945. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀼𑀡𑁆𑀳𑀧𑀓𑁆𑀔𑁂 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀟𑁆𑀠𑀢𑁂𑀯 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀯𑀟𑁆𑀠𑀢𑀺 𑀫𑀡𑁆𑀟𑀮𑁂𑀦, 𑀯𑀟𑁆𑀠𑀢𑀺 𑀆𑀪𑀸𑀬, 𑀯𑀟𑁆𑀠𑀢𑀺 𑀆𑀭𑁄𑀳𑀧𑀭𑀺𑀡𑀸𑀳𑁂𑀦𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀲𑀤𑁆𑀥𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼…  𑀳𑀺𑀭𑀻 𑀅𑀢𑁆𑀣𑀺…𑀧𑁂…  𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀅𑀢𑁆𑀣𑀺…  𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀢𑁆𑀣𑀺…  𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

949. 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀤𑁆𑀥𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼…𑀧𑁂…  𑀳𑀺𑀭𑀻 𑀅𑀢𑁆𑀣𑀺…  𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀅𑀢𑁆𑀣𑀺…  𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀢𑁆𑀣𑀺…  𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

950. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀚𑀼𑀡𑁆𑀳𑀧𑀓𑁆𑀔𑁂 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀟𑁆𑀠𑀢𑁂𑀯 𑀯𑀡𑁆𑀡𑁂𑀦, 𑀯𑀟𑁆𑀠𑀢𑀺 𑀫𑀡𑁆𑀟𑀮𑁂𑀦, 𑀯𑀟𑁆𑀠𑀢𑀺 𑀆𑀪𑀸𑀬, 𑀯𑀟𑁆𑀠𑀢𑀺 𑀆𑀭𑁄𑀳𑀧𑀭𑀺𑀡𑀸𑀳𑁂𑀦𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀲𑀤𑁆𑀥𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀳𑀺𑀭𑀻 𑀅𑀢𑁆𑀣𑀺…  𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀅𑀢𑁆𑀣𑀺…  𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀢𑁆𑀣𑀺…  𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀭𑀢𑁆𑀢𑀺 𑀯𑀸 𑀤𑀺𑀯𑀲𑁄 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

951. ‘𑀲𑀤𑁆𑀥𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; 𑀳𑀺𑀭𑀺𑀫𑀸…𑀧𑁂…  𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻…  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄…  𑀧𑀜𑁆𑀜𑀯𑀸…  𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄…  ‘𑀅𑀦𑀼𑀧𑀦𑀸𑀳𑀻 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁; ‘𑀲𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀯𑀸𑀤𑀓𑀸’𑀢𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀦𑁆”𑀢𑀺𑁇
