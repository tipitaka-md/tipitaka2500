# 𑁫𑁇𑁧𑁇𑁨 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀲𑁆𑀲𑀧𑀯𑀕𑁆𑀕

### 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻𑀲𑀼𑀢𑁆𑀢

(145.)

905\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀳𑀺𑀤𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀦𑀸𑀢𑀸𑀧𑀻 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀅𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬; 𑀆𑀢𑀸𑀧𑀻 𑀘 𑀔𑁄 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑀸”𑀢𑀺𑁇

906\. “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀸𑀢𑀸𑀧𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀅𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬; 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀘 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀆𑀢𑀸𑀧𑀻 𑀳𑁄𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑀸”𑀢𑀺? “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀢𑀺, ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀢𑀺, ‘𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀢𑀺, ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀸𑀢𑀸𑀧𑀻 𑀳𑁄𑀢𑀺𑁇

907\. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺, ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺, ‘𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺, ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀸𑀢𑀸𑀧𑀻 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀅𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑁇

908\. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀆𑀢𑀸𑀧𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀢𑀺, ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀆𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀢𑀸𑀧𑀻 𑀳𑁄𑀢𑀺𑁇

909\. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺, ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀬𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺, ‘𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺, ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀫𑀸𑀦𑀸 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀢𑀸𑀧𑀻 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀪𑀩𑁆𑀩𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬𑀸”𑀢𑀺𑁇
