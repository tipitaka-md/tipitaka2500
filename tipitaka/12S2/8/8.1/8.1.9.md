# 𑁮𑁇𑁧𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑀲𑀽𑀘𑀺𑀮𑁄𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀓𑁆𑀔𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀽𑀘𑀺𑀮𑁄𑀫𑀲𑀼𑀢𑁆𑀢

(210.)

1177\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀽𑀘𑀺𑀮𑁄𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀢𑀲𑁆𑀲 𑀢𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀲𑀻𑀲𑁂 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀔𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀺; 𑀫𑀼𑀔𑁂 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀭𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀺; 𑀉𑀭𑁂 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀭𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀺; 𑀉𑀤𑀭𑁂 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀊𑀭𑀽𑀳𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀺; 𑀊𑀭𑀽𑀲𑀼 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀚𑀗𑁆𑀖𑀸𑀳𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀺; 𑀚𑀗𑁆𑀖𑀸𑀲𑀼 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑁂𑀳𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀺; 𑀲𑁄 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑀽𑀘𑀓𑁄 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇
