# 𑁮𑁇𑁧𑁇𑁬 𑀲𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀓𑁆𑀔𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(207.)

1171\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀢𑁆𑀢𑀺𑀮𑁄𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀢𑀲𑁆𑀲 𑀢𑀸 𑀲𑀢𑁆𑀢𑀺𑀬𑁄 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀸𑀬𑁂 𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀫𑀸𑀕𑀯𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇
