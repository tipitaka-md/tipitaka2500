# 𑁮𑁇𑁧𑁇𑁧 𑀅𑀝𑁆𑀞𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀓𑁆𑀔𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀝𑁆𑀞𑀺𑀲𑀼𑀢𑁆𑀢

(202.)

1158\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀮𑀓𑁆𑀔𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀸𑀫𑀸𑀯𑀼𑀲𑁄 𑀮𑀓𑁆𑀔𑀡, 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑁆𑀯𑀸𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀓𑀫𑁆𑀫𑀸𑀬𑀸”𑀢𑀺? “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀮𑀓𑁆𑀔𑀡, 𑀏𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀳𑀲𑁆𑀲𑁇 𑀪𑀕𑀯𑀢𑁄 𑀫𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀸”𑀢𑀺𑁇

1159\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑁆𑀯𑀸𑀓𑀸𑀲𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀓𑀫𑁆𑀫𑀸𑀬𑀸”𑀢𑀺?

1160\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀝𑁆𑀞𑀺𑀓𑀲𑀗𑁆𑀔𑀮𑀺𑀓𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀁𑁇 𑀢𑀫𑁂𑀦𑀁 𑀕𑀺𑀚𑁆𑀛𑀸𑀧𑀺 𑀓𑀸𑀓𑀸𑀧𑀺 𑀓𑀼𑀮𑀮𑀸𑀧𑀺 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀨𑀸𑀲𑀼𑀍𑀅𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀢𑀘𑁆𑀙𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀸 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀬𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇

1161\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀘𑀓𑁆𑀔𑀼𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺; 𑀜𑀸𑀡𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀁 𑀜𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀤𑀓𑁆𑀔𑀢𑀺 𑀯𑀸 𑀲𑀓𑁆𑀔𑀺𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀦 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑁂 𑀘 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑁂 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀕𑁄𑀖𑀸𑀢𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀸𑀯𑀲𑁂𑀲𑁂𑀦 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺𑁇 (𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸𑀦𑀁 𑀏𑀲𑁂𑀯 𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑁄𑁇)
