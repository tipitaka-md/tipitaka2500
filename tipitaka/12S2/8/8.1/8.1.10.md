# 𑁮𑁇𑁧𑁇𑁧𑁦 𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

(211.)

1179\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀲𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄𑀧𑀺 𑀢𑁂𑀯 𑀅𑀡𑁆𑀟𑁂 𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑁄𑀧𑀺 𑀢𑁂𑀲𑁆𑀯𑁂𑀯 𑀅𑀡𑁆𑀟𑁂𑀲𑀼 𑀦𑀺𑀲𑀻𑀤𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀕𑀺𑀚𑁆𑀛𑀸𑀧𑀺 𑀓𑀸𑀓𑀸𑀧𑀺 𑀓𑀼𑀮𑀮𑀸𑀧𑀺 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑀘𑁆𑀙𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀕𑀸𑀫𑀓𑀽𑀝𑀓𑁄 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇

1182\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1183\. _𑀅𑀝𑁆𑀞𑀺 𑀧𑁂𑀲𑀺 𑀉𑀪𑁄 𑀕𑀸𑀯𑀖𑀸𑀢𑀓𑀸,_  
_𑀧𑀺𑀡𑁆𑀟𑁄 𑀲𑀸𑀓𑀼𑀡𑀺𑀬𑁄 𑀦𑀺𑀘𑁆𑀙𑀯𑁄𑀭𑀩𑁆𑀪𑀺;_  
_𑀅𑀲𑀺 𑀲𑀽𑀓𑀭𑀺𑀓𑁄 𑀲𑀢𑁆𑀢𑀺𑀫𑀸𑀕𑀯𑀺,_  
_𑀉𑀲𑀼 𑀓𑀸𑀭𑀡𑀺𑀓𑁄 𑀲𑀽𑀘𑀺 𑀲𑀸𑀭𑀣𑀺;_  
_𑀬𑁄 𑀘 𑀲𑀺𑀩𑁆𑀩𑀺𑀬𑀢𑀺 𑀲𑀽𑀘𑀓𑁄 𑀳𑀺 𑀲𑁄,_  
_𑀅𑀡𑁆𑀟𑀪𑀸𑀭𑀺 𑀅𑀳𑀼 𑀕𑀸𑀫𑀓𑀽𑀝𑀓𑁄𑀢𑀺𑁇_
