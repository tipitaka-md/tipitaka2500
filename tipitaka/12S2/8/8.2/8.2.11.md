# 𑁮𑁇𑁨𑁇𑁧𑁧 𑀧𑀸𑀧𑀲𑀸𑀫𑀡𑁂𑀭𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀓𑁆𑀔𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀧𑀲𑀸𑀫𑀡𑁂𑀭𑀻𑀲𑀼𑀢𑁆𑀢

(222.)

1204. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀁𑁇 𑀢𑀲𑁆𑀲𑀸 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀸 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀸, 𑀧𑀢𑁆𑀢𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄, 𑀓𑀸𑀬𑀩𑀦𑁆𑀥𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁, 𑀓𑀸𑀬𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄𑁇 𑀲𑀸 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀬𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇

1205. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀘𑀓𑁆𑀔𑀼𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺; 𑀜𑀸𑀡𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀁 𑀜𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀤𑀓𑁆𑀔𑀢𑀺 𑀯𑀸 𑀲𑀓𑁆𑀔𑀺𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀤𑀺𑀝𑁆𑀞𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀦 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑁂 𑀘 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑁂 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀯𑀘𑀦𑁂 𑀧𑀸𑀧𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑀸 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀸𑀯𑀲𑁂𑀲𑁂𑀦 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺𑁇

1207. 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄𑁇



---

1208. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1209. _𑀓𑀽𑀧𑁂 𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑁄 𑀳𑀺 𑀲𑁄 𑀧𑀸𑀭𑀤𑀸𑀭𑀺𑀓𑁄,_  
_𑀕𑀽𑀣𑀔𑀸𑀤𑀺 𑀅𑀳𑀼 𑀤𑀼𑀝𑁆𑀞𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀦𑀺𑀘𑁆𑀙𑀯𑀺𑀢𑁆𑀣𑀺 𑀅𑀢𑀺𑀘𑀸𑀭𑀺𑀦𑀻 𑀅𑀳𑀼,_  
_𑀫𑀗𑁆𑀕𑀼𑀮𑀺𑀢𑁆𑀣𑀺 𑀅𑀳𑀼 𑀇𑀓𑁆𑀔𑀡𑀺𑀢𑁆𑀣𑀺𑀓𑀸;_  
_𑀑𑀓𑀺𑀮𑀺𑀦𑀺 𑀲𑀧𑀢𑁆𑀢𑀗𑁆𑀕𑀸𑀭𑁄𑀓𑀺𑀭𑀺,_  
_𑀲𑀻𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀳𑀼 𑀘𑁄𑀭𑀖𑀸𑀢𑀓𑁄𑁇_  


1210. _𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸,_  
_𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀅𑀣 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀓𑀸;_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀁𑀲𑀼 𑀢𑀸𑀯𑀤𑁂𑀢𑀺𑁇_  


---

1211. 𑀮𑀓𑁆𑀔𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
