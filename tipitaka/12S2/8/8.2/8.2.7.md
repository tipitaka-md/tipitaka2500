# 𑁮𑁇𑁨𑁇𑁭 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀮𑀓𑁆𑀔𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(218.)

1196\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀸 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀸, 𑀧𑀢𑁆𑀢𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄, 𑀓𑀸𑀬𑀩𑀦𑁆𑀥𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁, 𑀓𑀸𑀬𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄𑁇 𑀲𑁄 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀯𑀘𑀦𑁂 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇
