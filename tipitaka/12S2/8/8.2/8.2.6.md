### 𑀅𑀲𑀻𑀲𑀓𑀲𑀼𑀢𑁆𑀢

(217.)

1194\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀲𑀻𑀲𑀓𑀁 𑀓𑀩𑀦𑁆𑀥𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀢𑀲𑁆𑀲 𑀉𑀭𑁂 𑀅𑀓𑁆𑀔𑀻𑀦𑀺 𑀘𑁂𑀯 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀼𑀔𑀜𑁆𑀘𑁇 𑀢𑀫𑁂𑀦𑀁 𑀕𑀺𑀚𑁆𑀛𑀸𑀧𑀺 𑀓𑀸𑀓𑀸𑀧𑀺 𑀓𑀼𑀮𑀮𑀸𑀧𑀺 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑀘𑁆𑀙𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀳𑀸𑀭𑀺𑀓𑁄 𑀦𑀸𑀫 𑀘𑁄𑀭𑀖𑀸𑀢𑀓𑁄 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇
