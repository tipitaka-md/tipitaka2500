### 𑀫𑀗𑁆𑀕𑀼𑀮𑀺𑀢𑁆𑀣𑀺𑀲𑀼𑀢𑁆𑀢

(215.)

1190\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀫𑀗𑁆𑀕𑀼𑀮𑀺𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀁𑁇 𑀢𑀫𑁂𑀦𑀁 𑀕𑀺𑀚𑁆𑀛𑀸𑀧𑀺 𑀓𑀸𑀓𑀸𑀧𑀺 𑀓𑀼𑀮𑀮𑀸𑀧𑀺 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀢𑀘𑁆𑀙𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀸 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀢𑁆𑀣𑀻 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀇𑀓𑁆𑀔𑀡𑀺𑀓𑀸 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇
