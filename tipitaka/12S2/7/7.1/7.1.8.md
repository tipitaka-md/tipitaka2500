# 𑁭𑁇𑁧𑁇𑁮 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀭𑀸𑀳𑀼𑀮𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(195.)

1128. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀭𑀸𑀳𑀼𑀮, 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀪𑀦𑁆𑀢𑁂”…  “𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸…  𑀭𑀲𑀢𑀡𑁆𑀳𑀸…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸…  𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀪𑀦𑁆𑀢𑁂”…  “𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀭𑀸𑀳𑀼𑀮, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀭𑀲𑀢𑀡𑁆𑀳𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
