# 𑁭𑁇𑁧𑁇𑁯 𑀥𑀸𑀢𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀭𑀸𑀳𑀼𑀮𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀥𑀸𑀢𑀼𑀲𑀼𑀢𑁆𑀢

(196.)

1130\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀭𑀸𑀳𑀼𑀮, 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀪𑀦𑁆𑀢𑁂”…  “𑀆𑀧𑁄𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼…  𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼…  𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀪𑀦𑁆𑀢𑁂”…  “𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀭𑀸𑀳𑀼𑀮, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇
