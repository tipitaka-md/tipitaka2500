# 𑁯𑁇𑁧𑁇𑁧𑁦 𑀩𑀺𑀍𑀆𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀩𑀺𑀍𑀆𑀭𑀲𑀼𑀢𑁆𑀢

(232.)

1239\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀦 𑀯𑀺𑀭𑀫𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀦 𑀯𑀺𑀭𑀫𑀢𑀻”𑀢𑀺𑁇

1240\. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀺𑀍𑀆𑀭𑁄 𑀲𑀦𑁆𑀥𑀺𑀲𑀫𑀮𑀲𑀗𑁆𑀓𑀝𑀻𑀭𑁂 𑀞𑀺𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀫𑀼𑀤𑀼𑀫𑀽𑀲𑀺𑀁 𑀫𑀕𑁆𑀕𑀬𑀫𑀸𑀦𑁄—  ‘𑀬𑀤𑀸𑀬𑀁 𑀫𑀼𑀤𑀼𑀫𑀽𑀲𑀺 𑀕𑁄𑀘𑀭𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀦𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀼𑀤𑀼𑀫𑀽𑀲𑀺 𑀕𑁄𑀘𑀭𑀸𑀬 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀩𑀺𑀍𑀆𑀭𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀳𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑁄 𑀫𑀼𑀤𑀼𑀫𑀽𑀲𑀺 𑀅𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀔𑀸𑀤𑀺, 𑀅𑀦𑁆𑀢𑀕𑀼𑀡𑀫𑁆𑀧𑀺 𑀔𑀸𑀤𑀺𑁇 𑀲𑁄 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀦𑀺𑀕𑀘𑁆𑀙𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇

1241\. 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀢𑀺 𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦𑁂𑀯 𑀓𑀸𑀬𑁂𑀦 𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀅𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦, 𑀅𑀦𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸, 𑀅𑀲𑀁𑀯𑀼𑀢𑁂𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀧𑀲𑁆𑀲𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀣𑀁 𑀯𑀸 𑀤𑀼𑀧𑁆𑀧𑀸𑀭𑀼𑀢𑀁 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀼𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀣𑀁 𑀯𑀸 𑀤𑀼𑀧𑁆𑀧𑀸𑀭𑀼𑀢𑀁 𑀯𑀸 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺𑁇 𑀲𑁄 𑀭𑀸𑀕𑀸𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀫𑀭𑀡𑀜𑁆𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀬𑁄 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀬𑀤𑀺𑀤𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦𑁂𑀯 𑀓𑀸𑀬𑁂𑀦 𑀭𑀓𑁆𑀔𑀺𑀢𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸, 𑀲𑀁𑀯𑀼𑀢𑁂𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
