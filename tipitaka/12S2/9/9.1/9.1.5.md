# 𑁯𑁇𑁧𑁇𑁫 𑀲𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(227.)

1220\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀺 𑀢𑀺𑀡𑁆𑀳𑀨𑀮𑀸𑁇 𑀅𑀣 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀁 𑀇𑀫𑀁 𑀲𑀢𑁆𑀢𑀺𑀁 𑀢𑀺𑀡𑁆𑀳𑀨𑀮𑀁 𑀧𑀸𑀡𑀺𑀦𑀸 𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀺𑀦𑀸 𑀯𑀸 𑀧𑀝𑀺𑀮𑁂𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀝𑀺𑀓𑁄𑀝𑁆𑀝𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀝𑀺𑀯𑀝𑁆𑀝𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀩𑁆𑀩𑁄 𑀦𑀼 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀫𑀼𑀁 𑀲𑀢𑁆𑀢𑀺𑀁 𑀢𑀺𑀡𑁆𑀳𑀨𑀮𑀁 𑀧𑀸𑀡𑀺𑀦𑀸 𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀺𑀦𑀸 𑀯𑀸 𑀧𑀝𑀺𑀮𑁂𑀡𑁂𑀢𑀼𑀁 𑀧𑀝𑀺𑀓𑁄𑀝𑁆𑀝𑁂𑀢𑀼𑀁 𑀧𑀝𑀺𑀯𑀝𑁆𑀝𑁂𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼”? “𑀅𑀲𑀼 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀢𑁆𑀢𑀺 𑀢𑀺𑀡𑁆𑀳𑀨𑀮𑀸 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀧𑀸𑀡𑀺𑀦𑀸 𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀺𑀦𑀸 𑀯𑀸 𑀧𑀝𑀺𑀮𑁂𑀡𑁂𑀢𑀼𑀁 𑀧𑀝𑀺𑀓𑁄𑀝𑁆𑀝𑁂𑀢𑀼𑀁 𑀧𑀝𑀺𑀯𑀝𑁆𑀝𑁂𑀢𑀼𑀁𑁇 𑀬𑀸𑀯𑀤𑁂𑀯 𑀘 𑀧𑀦 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀮𑀫𑀣𑀲𑁆𑀲 𑀯𑀺𑀖𑀸𑀢𑀲𑁆𑀲 𑀪𑀸𑀕𑀻 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1221\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀢𑀲𑁆𑀲 𑀘𑁂 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬; 𑀅𑀣 𑀔𑁄 𑀲𑁆𑀯𑁂𑀯 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀓𑀺𑀮𑀫𑀣𑀲𑁆𑀲 𑀯𑀺𑀖𑀸𑀢𑀲𑁆𑀲 𑀪𑀸𑀕𑀻 𑀅𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀫𑁂𑀢𑁆𑀢𑀸 𑀦𑁄 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
