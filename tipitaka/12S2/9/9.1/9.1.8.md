# 𑁯𑁇𑁧𑁇𑁮 𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀲𑀼𑀢𑁆𑀢

(230.)

1230. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1231. “𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀽𑀧𑀥𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀉𑀧𑀸𑀲𑀦𑀲𑁆𑀫𑀺𑀁𑁇 𑀢𑁂𑀲𑀁 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀑𑀢𑀸𑀭𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑀸 𑀫𑀼𑀤𑀼𑀢𑀮𑀼𑀦𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀸 𑀢𑁂 𑀫𑀼𑀤𑀼𑀓𑀸𑀲𑀼 𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀼 𑀢𑀽𑀮𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀸𑀲𑀼 𑀬𑀸𑀯𑀲𑀽𑀭𑀺𑀬𑀼𑀕𑁆𑀕𑀫𑀦𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀮𑀘𑁆𑀙𑀢𑀺 𑀑𑀢𑀸𑀭𑀁 𑀮𑀘𑁆𑀙𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀁𑁇

1232. 𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀽𑀧𑀥𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀥𑀸𑀦𑀲𑁆𑀫𑀺𑀁𑁇 𑀢𑁂𑀲𑀁 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀦 𑀮𑀪𑀢𑀺 𑀑𑀢𑀸𑀭𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀔𑀼𑀫𑀸 𑀫𑀼𑀤𑀼𑀢𑀮𑀼𑀦𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀸𑁇 𑀢𑁂 𑀫𑀼𑀤𑀼𑀓𑀸𑀲𑀼 𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀼 𑀢𑀽𑀮𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀸𑀲𑀼 𑀬𑀸𑀯𑀲𑀽𑀭𑀺𑀬𑀼𑀕𑁆𑀕𑀫𑀦𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀮𑀘𑁆𑀙𑀢𑀺 𑀑𑀢𑀸𑀭𑀁 𑀮𑀘𑁆𑀙𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀽𑀧𑀥𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀥𑀸𑀦𑀲𑁆𑀫𑀺𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
