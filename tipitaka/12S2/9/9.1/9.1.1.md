# 𑁯𑁇𑁧𑁇𑁧 𑀓𑀽𑀝𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀓𑀽𑀝𑀲𑀼𑀢𑁆𑀢

(223.)

1212\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀬𑁄 𑀲𑀩𑁆𑀩𑀸 𑀢𑀸 𑀓𑀽𑀝𑀗𑁆𑀕𑀫𑀸 𑀓𑀽𑀝𑀲𑀫𑁄𑀲𑀭𑀡𑀸 𑀓𑀽𑀝𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸 𑀲𑀩𑁆𑀩𑀸 𑀢𑀸 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀫𑀽𑀮𑀓𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀫𑁄𑀲𑀭𑀡𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
