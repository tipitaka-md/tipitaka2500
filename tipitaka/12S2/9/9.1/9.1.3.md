# 𑁯𑁇𑁧𑁇𑁩 𑀓𑀼𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀮𑀲𑀼𑀢𑁆𑀢

(225.)

1216\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀩𑀳𑀼𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀘𑁄𑀭𑁂𑀳𑀺 𑀓𑀼𑀫𑁆𑀪𑀢𑁆𑀣𑁂𑀦𑀓𑁂𑀳𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀪𑀸𑀯𑀺𑀢𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑁄 𑀲𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀬𑁄 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀺𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀺 𑀩𑀳𑀼𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀘𑁄𑀭𑁂𑀳𑀺 𑀓𑀼𑀫𑁆𑀪𑀢𑁆𑀣𑁂𑀦𑀓𑁂𑀳𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑁄 𑀤𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀬𑁄 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀫𑁂𑀢𑁆𑀢𑀸 𑀦𑁄 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
