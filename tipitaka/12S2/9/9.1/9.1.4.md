# 𑁯𑁇𑁧𑁇𑁪 𑀑𑀓𑁆𑀔𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀑𑀓𑁆𑀔𑀸𑀲𑀼𑀢𑁆𑀢

(226.)

1218\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀑𑀓𑁆𑀔𑀸𑀲𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀫𑀬𑀁 𑀑𑀓𑁆𑀔𑀸𑀲𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀑𑀓𑁆𑀔𑀸𑀲𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀤𑁆𑀤𑀼𑀳𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀯𑀸 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀫𑀬𑀁 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀤𑁆𑀤𑀼𑀳𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀤𑁆𑀤𑀼𑀳𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀬𑁆𑀬, 𑀇𑀤𑀁 𑀢𑀢𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑀢𑀭𑀁𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀫𑁂𑀢𑁆𑀢𑀸 𑀦𑁄 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
