# 𑁯𑁇𑁧𑁇𑁨 𑀦𑀔𑀲𑀺𑀔𑀲𑀼𑀢𑁆𑀢

* 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀔𑀲𑀺𑀔𑀲𑀼𑀢𑁆𑀢

(224.)

1214\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁, 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀫𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼 𑀆𑀭𑁄𑀧𑀺𑀢𑁄 𑀬𑀸 𑀘𑀸𑀬𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀻”𑀢𑀺? “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀻𑁇 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄𑀬𑀁 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼 𑀆𑀭𑁄𑀧𑀺𑀢𑁄𑁇 𑀲𑀗𑁆𑀔𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺 𑀉𑀧𑀦𑀺𑀥𑀺𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺 𑀓𑀮𑀪𑀸𑀕𑀫𑁆𑀧𑀺 𑀦 𑀉𑀧𑁂𑀢𑀺 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀪𑀕𑀯𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀔𑀲𑀺𑀔𑀸𑀬𑀁 𑀧𑀁𑀲𑀼 𑀆𑀭𑁄𑀧𑀺𑀢𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀬𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
