# 𑁯𑁇𑁧𑁇𑁯 𑀦𑀸𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀦𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(231.)

1234\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀦𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀓𑀺𑀫𑀗𑁆𑀕𑀁 𑀧𑀦𑀸𑀳𑀦𑁆”𑀢𑀺?

1235\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀦𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀓𑀺𑀫𑀗𑁆𑀕𑀁 𑀧𑀦𑀸𑀳𑀦𑁆’”𑀢𑀺𑁇

1236\. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑁂 𑀫𑀳𑀸𑀲𑀭𑀲𑀻𑁇 𑀢𑀁 𑀦𑀸𑀕𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀲𑀭𑀲𑀺𑀁 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀸𑀬 𑀪𑀺𑀲𑀫𑀼𑀍𑀆𑀮𑀁 𑀅𑀩𑁆𑀩𑀼𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀯𑀺𑀓𑁆𑀔𑀸𑀮𑀺𑀢𑀁 𑀯𑀺𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀅𑀓𑀤𑁆𑀤𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀯𑀡𑁆𑀡𑀸𑀬 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀩𑀮𑀸𑀬 𑀘, 𑀦 𑀘 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀦𑀸𑀕𑀸𑀦𑀁 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀢𑀭𑀼𑀡𑀸 𑀪𑀺𑀗𑁆𑀓𑀘𑁆𑀙𑀸𑀧𑀸 𑀢𑀁 𑀲𑀭𑀲𑀺𑀁 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀸𑀬 𑀪𑀺𑀲𑀫𑀼𑀍𑀆𑀮𑀁 𑀅𑀩𑁆𑀩𑀼𑀳𑁂𑀢𑁆𑀯𑀸 𑀦 𑀲𑀼𑀯𑀺𑀓𑁆𑀔𑀸𑀮𑀺𑀢𑀁 𑀯𑀺𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑀤𑁆𑀤𑀫𑀁 𑀅𑀲𑀗𑁆𑀔𑀸𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀦𑁂𑀯 𑀯𑀡𑁆𑀡𑀸𑀬 𑀳𑁄𑀢𑀺 𑀦 𑀩𑀮𑀸𑀬𑁇 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇

1237\. 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀕𑀺𑀳𑀻 𑀧𑀲𑀦𑁆𑀦𑀸𑀓𑀸𑀭𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀮𑀸𑀪𑀁 𑀅𑀕𑀥𑀺𑀢𑀸 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀅𑀦𑀚𑁆𑀛𑁄𑀧𑀦𑁆𑀦𑀸 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀯𑀡𑁆𑀡𑀸𑀬 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀩𑀮𑀸𑀬 𑀘, 𑀦 𑀘 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑁂𑀭𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀦𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀕𑀺𑀳𑀻 𑀧𑀲𑀦𑁆𑀦𑀸𑀓𑀸𑀭𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀮𑀸𑀪𑀁 𑀕𑀥𑀺𑀢𑀸 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀅𑀚𑁆𑀛𑁄𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀸𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁄 𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀦𑁂𑀯 𑀯𑀡𑁆𑀡𑀸𑀬 𑀳𑁄𑀢𑀺 𑀦 𑀩𑀮𑀸𑀬, 𑀢𑁂 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀕𑀥𑀺𑀢𑀸 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀅𑀦𑀚𑁆𑀛𑁄𑀧𑀦𑁆𑀦𑀸 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑀸 𑀢𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
