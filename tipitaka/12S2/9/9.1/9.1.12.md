# 𑁯𑁇𑁧𑁇𑁧𑁨 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀑𑀧𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(234.)

1245. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 “𑀅𑀲𑁆𑀲𑀼𑀢𑁆𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀚𑀭𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀫𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀺𑀬𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀚𑀭𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁂 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀓𑀢𑀯𑁂𑀤𑀺𑀢𑀸, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀧𑀝𑀺𑀜𑁆𑀜𑁂 𑀲𑀺𑀬𑀸 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀓𑀢𑀯𑁂𑀤𑀺𑀢𑀸𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀓𑀢𑀜𑁆𑀜𑀼𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀓𑀢𑀯𑁂𑀤𑀺𑀦𑁄; 𑀦 𑀘 𑀦𑁄 𑀅𑀫𑁆𑀳𑁂𑀲𑀼 𑀅𑀧𑁆𑀧𑀓𑀫𑁆𑀧𑀺 𑀓𑀢𑀁 𑀦𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

1248. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1249. _𑀓𑀽𑀝𑀁 𑀦𑀔𑀲𑀺𑀔𑀁 𑀓𑀼𑀮𑀁,_
