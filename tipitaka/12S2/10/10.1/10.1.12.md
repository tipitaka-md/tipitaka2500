# 𑁧𑁦𑁇𑁧𑁇𑁧𑁨 𑀲𑀳𑀸𑀬𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀲𑀳𑀸𑀬𑀓𑀲𑀼𑀢𑁆𑀢

(246.)

1307. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀳𑀸𑀬𑀓𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀧𑀲𑁆𑀲𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀳𑀸𑀬𑀓𑁂 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀏𑀢𑁂 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇 𑀦 𑀘 𑀲𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺 𑀲𑀼𑀮𑀪𑀭𑀽𑀧𑀸, 𑀬𑀸 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀸𑁇 𑀬𑀲𑁆𑀲 𑀘𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1308. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

1309. _“𑀲𑀳𑀸𑀬𑀸𑀯𑀢𑀺𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽,_  
_𑀘𑀺𑀭𑀭𑀢𑁆𑀢𑀁 𑀲𑀫𑁂𑀢𑀺𑀓𑀸;_  
_𑀲𑀫𑁂𑀢𑀺 𑀦𑁂𑀲𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄,_  
_𑀥𑀫𑁆𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂𑁇_  


1310. _𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑀸 𑀓𑀧𑁆𑀧𑀺𑀦𑁂𑀦,_  
_𑀥𑀫𑁆𑀫𑁂 𑀅𑀭𑀺𑀬𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂;_  
_𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

1311. 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁𑁇



---

1312. 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

1313. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1314. _𑀓𑁄𑀮𑀺𑀢𑁄 𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀘,_  
_𑀖𑀝𑁄 𑀘𑀸𑀧𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀦𑀯𑁄 𑀲𑀼𑀚𑀸𑀢𑁄 𑀪𑀤𑁆𑀤𑀺 𑀘,_  
_𑀯𑀺𑀲𑀸𑀔𑁄 𑀦𑀦𑁆𑀤𑁄 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘;_  
_𑀣𑁂𑀭𑀦𑀸𑀫𑁄 𑀘 𑀓𑀧𑁆𑀧𑀺𑀦𑁄,_  
_𑀲𑀳𑀸𑀬𑁂𑀦 𑀘 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀢𑀺𑁇_  


---

1315. 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇



---

1316. 𑀦𑀺𑀤𑀸𑀦𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



---

1317. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1318. _𑀦𑀺𑀤𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀥𑀸𑀢𑀼,_  
_𑀅𑀦𑀫𑀢𑀕𑁆𑀕𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑀁;_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑀭𑀸𑀳𑀼𑀮𑀮𑀓𑁆𑀔𑀡𑁄,_  
_𑀑𑀧𑀫𑁆𑀫𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀕𑁆𑀕𑁄𑁇_  


---

1319. 𑀤𑀼𑀢𑀺𑀬𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇



---

1320. 𑀦𑀺𑀤𑀸𑀦𑀯𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
