# 𑁧𑁦𑁇𑁧𑁇𑁯 𑀢𑀺𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

* 𑀢𑀺𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

(243.)

1292\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀺𑀢𑀼𑀘𑁆𑀙𑀸𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀅𑀲𑁆𑀲𑀽𑀦𑀺 𑀧𑀯𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀢𑀺𑀲𑁆𑀲𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀢𑀺𑀲𑁆𑀲, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀅𑀲𑁆𑀲𑀽𑀦𑀺 𑀧𑀯𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑁄”𑀢𑀺? “𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀦𑁆𑀢𑀸 𑀯𑀸𑀘𑀸𑀬𑀲𑀦𑁆𑀦𑀺𑀢𑁄𑀤𑀓𑁂𑀦 𑀲𑀜𑁆𑀚𑀫𑁆𑀪𑀭𑀺𑀫𑀓𑀁𑀲𑀽”𑀢𑀺𑁇 “𑀢𑀣𑀸𑀳𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀢𑀺𑀲𑁆𑀲, 𑀯𑀢𑁆𑀢𑀸 𑀦𑁄 𑀘 𑀯𑀘𑀦𑀓𑁆𑀔𑀫𑁄; 𑀦 𑀔𑁄 𑀢𑁂 𑀢𑀁, 𑀢𑀺𑀲𑁆𑀲, 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑁆𑀲, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀢𑁆𑀢𑀸 𑀦𑁄 𑀘 𑀯𑀘𑀦𑀓𑁆𑀔𑀫𑁄𑁇 𑀏𑀢𑀁 𑀔𑁄 𑀢𑁂, 𑀢𑀺𑀲𑁆𑀲, 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑁆𑀲—  ‘𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀢𑁆𑀢𑀸 𑀘 𑀅𑀲𑁆𑀲 𑀯𑀘𑀦𑀓𑁆𑀔𑀫𑁄 𑀘𑀸’”𑀢𑀺𑁇

1293\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

1294\. _“𑀓𑀺𑀁 𑀦𑀼 𑀓𑀼𑀚𑁆𑀛𑀲𑀺 𑀫𑀸 𑀓𑀼𑀚𑁆𑀛𑀺,_  
_𑀅𑀓𑁆𑀓𑁄𑀥𑁄 𑀢𑀺𑀲𑁆𑀲 𑀢𑁂 𑀯𑀭𑀁;_  
_𑀓𑁄𑀥𑀫𑀸𑀦𑀫𑀓𑁆𑀔𑀯𑀺𑀦𑀬𑀢𑁆𑀣𑀜𑁆𑀳𑀺,_  
_𑀢𑀺𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_
