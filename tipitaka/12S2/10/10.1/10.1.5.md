# 𑁧𑁦𑁇𑁧𑁇𑁫 𑀲𑀼𑀚𑀸𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀚𑀸𑀢𑀲𑀼𑀢𑁆𑀢

(239.)

1274\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀚𑀸𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀚𑀸𑀢𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀉𑀪𑀬𑁂𑀦𑁂𑀯𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑁄𑀪𑀢𑀺—  𑀬𑀜𑁆𑀘 𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁄 𑀧𑀭𑀫𑀸𑀬 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀬𑀲𑁆𑀲 𑀘𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸—

1275\. _“𑀲𑁄𑀪𑀢𑀺 𑀯𑀢𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀉𑀚𑀼𑀪𑀽𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀥𑀸𑀭𑁂𑀢𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀦𑁆”𑀢𑀺𑁇_
