# 𑁧𑁦𑁇𑁧𑁇𑁧 𑀓𑁄𑀮𑀺𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀓𑁄𑀮𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(235.)

1251\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀯𑀼𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺𑁇 “𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇

1252\. 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀅𑀭𑀺𑀬𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄, 𑀅𑀭𑀺𑀬𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀭𑀺𑀬𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀭𑀺𑀬𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀺𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀭𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺𑁇

1253\. 𑀅𑀣 𑀔𑁄 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀫𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀭𑀺𑀬𑀁 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑀁 𑀧𑀫𑀸𑀤𑁄, 𑀅𑀭𑀺𑀬𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀡𑁆𑀞𑀧𑁂𑀳𑀺, 𑀅𑀭𑀺𑀬𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀓𑀭𑁄𑀳𑀺, 𑀅𑀭𑀺𑀬𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀳𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀣𑀸𑀭𑀸 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀲𑀸𑀯𑀓𑁄 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄’𑀢𑀺, 𑀫𑀫𑀁 𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀣𑀸𑀭𑀸 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀲𑀸𑀯𑀓𑁄 𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀢𑀁 𑀧𑀢𑁆𑀢𑁄’”𑀢𑀺𑁇
