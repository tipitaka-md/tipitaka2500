# 𑁧𑁦𑁇𑁧𑁇𑁮 𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(242.)

1288. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀓𑁄𑀝𑀺𑀢𑀧𑀘𑁆𑀘𑀸𑀓𑁄𑀝𑀺𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀓𑁆𑀔𑀻𑀦𑀺 𑀅𑀜𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀁 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦 𑀔𑁄 𑀢𑁂 𑀢𑀁, 𑀦𑀦𑁆𑀤, 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑁆𑀲, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀆𑀓𑁄𑀝𑀺𑀢𑀧𑀘𑁆𑀘𑀸𑀓𑁄𑀝𑀺𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀧𑀸𑀭𑀼𑀧𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀓𑁆𑀔𑀻𑀦𑀺 𑀘 𑀅𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀘𑁆𑀙𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀏𑀢𑀁 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤, 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑁆𑀲, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘 𑀅𑀲𑁆𑀲𑀲𑀺, 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼 𑀘 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸—

1289. _“𑀓𑀤𑀸𑀳𑀁 𑀦𑀦𑁆𑀤𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁,_  
_𑀆𑀭𑀜𑁆𑀜𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀁;_  
_𑀅𑀜𑁆𑀜𑀸𑀢𑀼𑀜𑁆𑀙𑁂𑀦 𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀁,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_  


1290. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼 𑀘 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄 𑀯𑀺𑀳𑀸𑀲𑀻𑀢𑀺𑁇
