# 𑁧𑁦𑁇𑁧𑁇𑁩 𑀖𑀝𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀖𑀝𑀲𑀼𑀢𑁆𑀢

(237.)

1259\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1260\. “𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺; 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄 𑀲𑀦𑁆𑀢𑁂𑀦 𑀦𑀽𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀅𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀑𑀍𑀆𑀭𑀺𑀓𑁂𑀦 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀫𑁂 𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑀻 𑀓𑀣𑀸”𑀢𑀺𑁇 “𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑀻 𑀓𑀣𑀸”𑀢𑀺? “𑀪𑀕𑀯𑀢𑀸 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑀻 𑀓𑀣𑀸”𑀢𑀺𑁇 “𑀤𑀽𑀭𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀏𑀢𑀭𑀳𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀤𑀸𑀳𑀼 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀻”𑀢𑀺? “𑀦 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀦𑀧𑀺 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀫𑁂 𑀬𑀸𑀯𑀢𑀸 𑀪𑀕𑀯𑀸 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀺 𑀤𑀺𑀩𑁆𑀩𑀸 𑀘 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑁇 𑀪𑀕𑀯𑀢𑁄𑀧𑀺 𑀬𑀸𑀯𑀢𑀸𑀳𑀁 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀺 𑀤𑀺𑀩𑁆𑀩𑀸 𑀘 𑀲𑁄𑀢𑀥𑀸𑀢𑀽”𑀢𑀺𑁇 “𑀬𑀣𑀸𑀓𑀣𑀁 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑀻 𑀓𑀣𑀸”𑀢𑀺?

1261\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀇𑀥, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀓𑀸𑀫𑀁 𑀢𑀘𑁄 𑀘 𑀦𑁆𑀳𑀸𑀭𑀼 𑀘 𑀅𑀝𑁆𑀞𑀺 𑀘 𑀅𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀼, 𑀲𑀭𑀻𑀭𑁂 𑀉𑀧𑀲𑁆𑀲𑀼𑀲𑁆𑀲𑀢𑀼 𑀫𑀁𑀲𑀮𑁄𑀳𑀺𑀢𑀁, 𑀬𑀁 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀣𑀸𑀫𑁂𑀦 𑀧𑀼𑀭𑀺𑀲𑀯𑀻𑀭𑀺𑀬𑁂𑀦 𑀧𑀼𑀭𑀺𑀲𑀧𑀭𑀓𑁆𑀓𑀫𑁂𑀦 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀦 𑀢𑀁 𑀅𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑀻 𑀓𑀣𑀸”𑀢𑀺𑁇

1262\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑀺𑀫𑀯𑀢𑁄 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀲𑁆𑀲 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸 𑀬𑀸𑀯𑀤𑁂𑀯 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀦𑀫𑀢𑁆𑀢𑀸𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑁂𑀯 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀦𑀫𑀢𑁆𑀢𑀸𑀬𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀳𑀺 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀓𑀧𑁆𑀧𑀁 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

1263\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀳𑀢𑀺𑀬𑀸 𑀮𑁄𑀡𑀖𑀝𑀸𑀬 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀮𑁄𑀡𑀲𑀓𑁆𑀔𑀭𑀸𑀬 𑀬𑀸𑀯𑀤𑁂𑀯 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀦𑀫𑀢𑁆𑀢𑀸𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑁂𑀯 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀦𑀫𑀢𑁆𑀢𑀸𑀬𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀳𑀺 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀣𑁄𑀫𑀺𑀢𑁄 𑀯𑀡𑁆𑀡𑀺𑀢𑁄 𑀧𑀲𑀢𑁆𑀣𑁄—

1264\. _‘𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀯 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀻𑀮𑁂𑀦 𑀉𑀧𑀲𑀫𑁂𑀦 𑀘;_  
_𑀬𑁄𑀧𑀺 𑀧𑀸𑀭𑀗𑁆𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀏𑀢𑀸𑀯𑀧𑀭𑀫𑁄 𑀲𑀺𑀬𑀸’”𑀢𑀺𑁇_
