# 𑀳𑁂𑀢𑀼

### 𑀢𑀺𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(35.)

403. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

404. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

405. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀧𑀜𑁆𑀘𑁇

406. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

407. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

408. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢𑀭𑀲…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀲𑀢𑁆𑀢𑀭𑀲𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(36.)

409. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

410. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

411. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁄𑁇)
