# 𑁧𑁇𑁧𑁇𑁭 𑀲𑁄𑀪𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀪𑀡𑁆𑀟𑀕𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀲𑁄𑀪𑀦𑀲𑀼𑀢𑁆𑀢

(7.)

44\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑀁 𑀲𑁄𑀪𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀯𑀺𑀦𑀻𑀢𑁄 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀁 𑀲𑁄𑀪𑁂𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑀁 𑀲𑁄𑀪𑁂𑀢𑀺𑁇 𑀉𑀧𑀸𑀲𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀯𑀺𑀦𑀻𑀢𑁄 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀁 𑀲𑁄𑀪𑁂𑀢𑀺𑁇 𑀉𑀧𑀸𑀲𑀺𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑀁 𑀲𑁄𑀪𑁂𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑀁 𑀲𑁄𑀪𑁂𑀦𑁆𑀢𑀻𑀢𑀺𑁇

45\. _𑀬𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀘,_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺;_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻,_  
_𑀲 𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀁𑀖𑀲𑁄𑀪𑀦𑁄𑁇_  


46\. _𑀪𑀺𑀓𑁆𑀔𑀼 𑀘 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀉𑀧𑀸𑀲𑀓𑁄 𑀘 𑀬𑁄 𑀲𑀤𑁆𑀥𑁄,_  
_𑀬𑀸 𑀘 𑀲𑀤𑁆𑀥𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸;_  
_𑀏𑀢𑁂 𑀔𑁄 𑀲𑀁𑀖𑀁 𑀲𑁄𑀪𑁂𑀦𑁆𑀢𑀺,_  
_𑀏𑀢𑁂 𑀳𑀺 𑀲𑀁𑀖𑀲𑁄𑀪𑀦𑀸”𑀢𑀺𑁇_  


---

47\. 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
