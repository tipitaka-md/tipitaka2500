# 𑁧𑁇𑁧𑁇𑁪 𑀤𑀼𑀢𑀺𑀬𑀔𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀪𑀡𑁆𑀟𑀕𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀔𑀢𑀲𑀼𑀢𑁆𑀢

(4.)

18\. “𑀘𑀢𑀽𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼? 𑀫𑀸𑀢𑀭𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀧𑀺𑀢𑀭𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀢𑀣𑀸𑀕𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇

19\. 𑀘𑀢𑀽𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀢𑀁 𑀅𑀦𑀼𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀅𑀦𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼? 𑀫𑀸𑀢𑀭𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀢𑀁 𑀅𑀦𑀼𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀅𑀦𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀧𑀺𑀢𑀭𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀢𑀣𑀸𑀕𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀢𑀁 𑀅𑀦𑀼𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀅𑀦𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀲𑀼 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀓𑁆𑀔𑀢𑀁 𑀅𑀦𑀼𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀅𑀦𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀦𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀻𑀢𑀺𑁇

20\. _𑀫𑀸𑀢𑀭𑀺 𑀧𑀺𑀢𑀭𑀺 𑀘𑀸𑀧𑀺,_  
_𑀬𑁄 𑀫𑀺𑀘𑁆𑀙𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺;_  
_𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂,_  
_𑀅𑀣 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁂;_  
_𑀩𑀳𑀼𑀜𑁆𑀘 𑀲𑁄 𑀧𑀲𑀯𑀢𑀺,_  
_𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀢𑀸𑀤𑀺𑀲𑁄 𑀦𑀭𑁄𑁇_  


21\. _𑀢𑀸𑀬 𑀦𑀁 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀇𑀥𑁂𑀯 𑀦𑀁 𑀕𑀭𑀳𑀦𑁆𑀢𑀺,_  
_𑀧𑁂𑀘𑁆𑀘𑀸𑀧𑀸𑀬𑀜𑁆𑀘 𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


22\. _𑀫𑀸𑀢𑀭𑀺 𑀧𑀺𑀢𑀭𑀺 𑀘𑀸𑀧𑀺,_  
_𑀬𑁄 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺;_  
_𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂,_  
_𑀅𑀣 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁂;_  
_𑀩𑀳𑀼𑀜𑁆𑀘 𑀲𑁄 𑀧𑀲𑀯𑀢𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀁 𑀏𑀢𑀸𑀤𑀺𑀲𑁄 𑀦𑀭𑁄𑁇_  


23\. _𑀢𑀸𑀬 𑀦𑀁 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀇𑀥𑁂𑀯 𑀦𑀁 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑀕𑁆𑀕𑁂 𑀧𑀫𑁄𑀤𑀢𑀻”𑀢𑀺𑁇_  


---

24\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
