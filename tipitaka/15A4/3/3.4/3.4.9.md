# 𑁩𑁇𑁪𑁇𑁯 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(139.)

872\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀅𑀧𑁆𑀧𑀜𑁆𑀘 𑀪𑀸𑀲𑀢𑀺 𑀅𑀲𑀳𑀺𑀢𑀜𑁆𑀘; 𑀧𑀭𑀺𑀲𑀸 𑀘𑀲𑁆𑀲 𑀦 𑀓𑀼𑀲𑀮𑀸 𑀳𑁄𑀢𑀺 𑀲𑀳𑀺𑀢𑀸𑀲𑀳𑀺𑀢𑀲𑁆𑀲𑁇 𑀏𑀯𑀭𑀽𑀧𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

873\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀅𑀧𑁆𑀧𑀜𑁆𑀘 𑀪𑀸𑀲𑀢𑀺 𑀲𑀳𑀺𑀢𑀜𑁆𑀘; 𑀧𑀭𑀺𑀲𑀸 𑀘𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀳𑁄𑀢𑀺 𑀲𑀳𑀺𑀢𑀸𑀲𑀳𑀺𑀢𑀲𑁆𑀲𑁇 𑀏𑀯𑀭𑀽𑀧𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

874\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀩𑀳𑀼𑀜𑁆𑀘 𑀪𑀸𑀲𑀢𑀺 𑀅𑀲𑀳𑀺𑀢𑀜𑁆𑀘; 𑀧𑀭𑀺𑀲𑀸 𑀘𑀲𑁆𑀲 𑀦 𑀓𑀼𑀲𑀮𑀸 𑀳𑁄𑀢𑀺 𑀲𑀳𑀺𑀢𑀸𑀲𑀳𑀺𑀢𑀲𑁆𑀲𑁇 𑀏𑀯𑀭𑀽𑀧𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

875\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀩𑀳𑀼𑀜𑁆𑀘 𑀪𑀸𑀲𑀢𑀺 𑀲𑀳𑀺𑀢𑀜𑁆𑀘; 𑀧𑀭𑀺𑀲𑀸 𑀘𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀳𑁄𑀢𑀺 𑀲𑀳𑀺𑀢𑀸𑀲𑀳𑀺𑀢𑀲𑁆𑀲𑁇 𑀏𑀯𑀭𑀽𑀧𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸”𑀢𑀺𑁇
