# 𑁩𑁇𑁪𑁇𑁨 𑀧𑀝𑀺𑀪𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀝𑀺𑀪𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(132.)

844\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀬𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄, 𑀦𑁄 𑀫𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄; 𑀫𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄, 𑀦𑁄 𑀬𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄; 𑀬𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄 𑀘 𑀫𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄 𑀘; 𑀦𑁂𑀯 𑀬𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄 𑀦 𑀫𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇
