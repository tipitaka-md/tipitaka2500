# 𑁩𑁇𑁪𑁇𑁧 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

* 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀼𑀢𑁆𑀢

(131.)

835\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

836\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

837\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

838\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

839\. 𑀓𑀢𑀫𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺? 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺𑁇

840\. 𑀓𑀢𑀫𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺? 𑀉𑀤𑁆𑀥𑀁𑀲𑁄𑀢𑀲𑁆𑀲 𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺𑁇

841\. 𑀓𑀢𑀫𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺? 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀦𑀺𑁇

842\. 𑀓𑀢𑀫𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺? 𑀅𑀭𑀳𑀢𑁄𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺, 𑀪𑀯𑀧𑀝𑀺𑀮𑀸𑀪𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇
