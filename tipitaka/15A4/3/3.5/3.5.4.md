# 𑁩𑁇𑁫𑁇𑁪 𑀑𑀪𑀸𑀲𑀲𑀼𑀢𑁆𑀢

* 𑀆𑀪𑀸𑀯𑀕𑁆𑀕

* 𑀑𑀪𑀸𑀲𑀲𑀼𑀢𑁆𑀢

(144.)

888\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀪𑀸𑀲𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀘𑀦𑁆𑀤𑁄𑀪𑀸𑀲𑁄, 𑀲𑀽𑀭𑀺𑀬𑁄𑀪𑀸𑀲𑁄, 𑀅𑀕𑁆𑀕𑁄𑀪𑀸𑀲𑁄, 𑀧𑀜𑁆𑀜𑁄𑀪𑀸𑀲𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀑𑀪𑀸𑀲𑀸𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀑𑀪𑀸𑀲𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀜𑁆𑀜𑁄𑀪𑀸𑀲𑁄”𑀢𑀺𑁇
