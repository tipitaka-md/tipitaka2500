# 𑁩𑁇𑁫𑁇𑁬 𑀧𑀞𑀫𑀓𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀆𑀪𑀸𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀓𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(146.)

892. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀮𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀓𑀸𑀮𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁, 𑀓𑀸𑀮𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑀸𑀓𑀘𑁆𑀙𑀸, 𑀓𑀸𑀮𑁂𑀦 𑀲𑀫𑁆𑀫𑀲𑀦𑀸, 𑀓𑀸𑀮𑁂𑀦 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀓𑀸𑀮𑀸”𑀢𑀺𑁇
