# 𑁨𑁇𑁧𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀚𑀻𑀯𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑀚𑀻𑀯𑀻𑀲𑀼𑀢𑁆𑀢

(56.)

412\. “𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀪𑁄 𑀚𑀸𑀦𑀺𑀧𑀢𑀬𑁄 𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁 𑀉𑀪𑁄𑀯 𑀅𑀲𑁆𑀲𑀼 𑀲𑀫𑀲𑀤𑁆𑀥𑀸 𑀲𑀫𑀲𑀻𑀮𑀸 𑀲𑀫𑀘𑀸𑀕𑀸 𑀲𑀫𑀧𑀜𑁆𑀜𑀸, 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺𑁇

413\. _𑀉𑀪𑁄 𑀲𑀤𑁆𑀥𑀸 𑀯𑀤𑀜𑁆𑀜𑀽 𑀘,_  
_𑀲𑀜𑁆𑀜𑀢𑀸 𑀥𑀫𑁆𑀫𑀚𑀻𑀯𑀺𑀦𑁄;_  
_𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀧𑀢𑀬𑁄,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀁𑀯𑀤𑀸𑁇_  


414\. _𑀅𑀢𑁆𑀣𑀸𑀲𑀁 𑀧𑀘𑀼𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀨𑀸𑀲𑀼𑀓𑀁 𑀉𑀧𑀚𑀸𑀬𑀢𑀺;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀤𑀼𑀫𑁆𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀲𑀫𑀲𑀻𑀮𑀺𑀦𑀁𑁇_  


415\. _𑀇𑀥 𑀥𑀫𑁆𑀫𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀫𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀸 𑀉𑀪𑁄;_  
_𑀦𑀦𑁆𑀤𑀺𑀦𑁄 𑀤𑁂𑀯𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀓𑀸𑀫𑀺𑀦𑁄”𑀢𑀺𑁇_
