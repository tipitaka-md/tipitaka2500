# 𑁨𑁇𑁨𑁇𑁬 𑀲𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀧𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(66.)

484\. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀲𑀭𑀸𑀕𑁄, 𑀲𑀤𑁄𑀲𑁄, 𑀲𑀫𑁄𑀳𑁄, 𑀲𑀫𑀸𑀦𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺𑁇

485\. _𑀲𑀸𑀭𑀢𑁆𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑁂𑀲𑀼,_  
_𑀧𑀺𑀬𑀭𑀽𑀧𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀦𑁄;_  
_𑀫𑁄𑀳𑁂𑀦 𑀆𑀯𑀼𑀢𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀩𑀤𑁆𑀥𑀸 𑀯𑀟𑁆𑀠𑁂𑀦𑁆𑀢𑀺 𑀩𑀦𑁆𑀥𑀦𑀁𑁇_  


486\. _𑀭𑀸𑀕𑀚𑀁 𑀤𑁄𑀲𑀚𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀫𑁄𑀳𑀚𑀁 𑀘𑀸𑀧𑀯𑀺𑀤𑁆𑀤𑀲𑀽;_  
_𑀓𑀭𑁄𑀦𑁆𑀢𑀸𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀯𑀺𑀖𑀸𑀢𑀁 𑀤𑀼𑀔𑀼𑀤𑁆𑀭𑀬𑀁𑁇_  


487\. _𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀯𑀼𑀢𑀸 𑀧𑁄𑀲𑀸,_  
_𑀅𑀦𑁆𑀥𑀪𑀽𑀢𑀸 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸;_  
_𑀬𑀣𑀸 𑀥𑀫𑁆𑀫𑀸 𑀢𑀣𑀸 𑀲𑀦𑁆𑀢𑀸,_  
_𑀦 𑀢𑀲𑁆𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑀭𑁂”𑀢𑀺𑁇_  


---
