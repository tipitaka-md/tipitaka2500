### 𑀧𑀥𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(71.)

523\. “𑀘𑀢𑀽𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀧𑀡𑁆𑀡𑀓𑀧𑁆𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀬𑁄𑀦𑀺 𑀘𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀧𑀡𑁆𑀡𑀓𑀧𑁆𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀬𑁄𑀦𑀺 𑀘𑀲𑁆𑀲 𑀆𑀭𑀤𑁆𑀥𑀸 𑀳𑁄𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑀸”𑀢𑀺𑁇
