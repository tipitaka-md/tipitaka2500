# 𑁫𑁇𑁨𑁇𑁮 𑀤𑀼𑀢𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀧𑀣𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀧𑀭𑀺𑀲𑀸𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀧𑀣𑀲𑀼𑀢𑁆𑀢

(218.)

1390\. …𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑁂 𑀅𑀤𑀺𑀝𑁆𑀞𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀢𑁂 𑀅𑀲𑀼𑀢𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀢𑁂 𑀅𑀫𑀼𑀢𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑁂 𑀤𑀺𑀝𑁆𑀞𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀢𑁂 𑀲𑀼𑀢𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀢𑁂 𑀫𑀼𑀢𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺𑁇
