# 𑁫𑁇𑁭𑁇𑁫 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀓𑀫𑁆𑀫𑀧𑀣𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑀸𑀲𑀼𑀢𑁆𑀢

(268.)

1580. …𑀧𑁂…  𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀘 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺…𑀧𑁂… 𑁇

1581. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺…𑀧𑁂… 𑁇
