# 𑁫𑁇𑁭𑁇𑁪 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀓𑀫𑁆𑀫𑀧𑀣𑀯𑀕𑁆𑀕

### 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻𑀲𑀼𑀢𑁆𑀢

(267.)

1577\. …𑀧𑁂…  𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀘 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄…𑀧𑁂… 𑁇

1578\. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺…𑀧𑁂… 𑁇
