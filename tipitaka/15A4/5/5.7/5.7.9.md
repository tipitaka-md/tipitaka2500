# 𑁫𑁇𑁭𑁇𑁯 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀓𑀫𑁆𑀫𑀧𑀣𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(272.)

1592\. …𑀧𑁂…  𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀘 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺…𑀧𑁂… 𑁇

1593\. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀘 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑁆𑀲 𑀘 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺…𑀧𑁂… 𑁇
