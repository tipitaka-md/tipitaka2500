# 𑁫𑁇𑁩𑁇𑁪 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻𑀲𑀼𑀢𑁆𑀢

(224.)

1407. …𑀧𑁂…  𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇
