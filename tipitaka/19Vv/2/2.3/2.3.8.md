# 𑁨𑁇𑁩𑁇𑁮 𑀅𑀦𑁂𑀓𑀯𑀡𑁆𑀡𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀲𑀼𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀅𑀦𑁂𑀓𑀯𑀡𑁆𑀡𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(1199.)

1050\. _“𑀅𑀦𑁂𑀓𑀯𑀡𑁆𑀡𑀁 𑀤𑀭𑀲𑁄𑀓𑀦𑀸𑀲𑀦𑀁,_  
_𑀯𑀺𑀫𑀸𑀦𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀅𑀦𑁂𑀓𑀘𑀺𑀢𑁆𑀢𑀁;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀕𑀡𑁂𑀦,_  
_𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄 𑀪𑀽𑀢𑀧𑀢𑀻𑀯 𑀫𑁄𑀤𑀲𑀺𑁇_  


(1200.)

1051\. _𑀲𑀫𑀲𑁆𑀲𑀫𑁄 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀢𑁄 𑀧𑀦𑀼𑀢𑁆𑀢𑀭𑁄,_  
_𑀬𑀲𑁂𑀦 𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀘 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀘;_  
_𑀲𑀩𑁆𑀩𑁂 𑀘 𑀤𑁂𑀯𑀸 𑀢𑀺𑀤𑀲𑀕𑀡𑀸 𑀲𑀫𑁂𑀘𑁆𑀘,_  
_𑀢𑀁 𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀲𑀺𑀁𑀯 𑀤𑁂𑀯𑀸;_  
_𑀇𑀫𑀸 𑀘 𑀢𑁂 𑀅𑀘𑁆𑀙𑀭𑀸𑀬𑁄 𑀲𑀫𑀦𑁆𑀢𑀢𑁄,_  
_𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀫𑁄𑀤𑀬𑀦𑁆𑀢𑀺𑁇_  


(1201.)

1052\. _𑀤𑁂𑀯𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(1202.)

1053\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(1203.)

1054\. _“𑀅𑀳𑀁 𑀪𑀤𑀦𑁆𑀢𑁂 𑀅𑀳𑀼𑀯𑀸𑀲𑀺 𑀧𑀼𑀩𑁆𑀩𑁂,_  
_𑀲𑀼𑀫𑁂𑀥𑀦𑀸𑀫𑀲𑁆𑀲 𑀚𑀺𑀦𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑁄𑀳𑀫𑀲𑁆𑀫𑀺,_  
_𑀲𑁄 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀳𑀁𑁇_  


(1204.)

1055\. _𑀲𑁄𑀳𑀁 𑀲𑀼𑀫𑁂𑀥𑀲𑁆𑀲 𑀚𑀺𑀦𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄,_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲𑁄𑀖𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀭𑀢𑀦𑀼𑀘𑁆𑀘𑀬𑀁 𑀳𑁂𑀫𑀚𑀸𑀮𑁂𑀦 𑀙𑀦𑁆𑀦𑀁,_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀣𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀫𑀦𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁𑁇_  


(1205.)

1056\. _𑀦 𑀫𑀸𑀲𑀺 𑀤𑀸𑀦𑀁 𑀦 𑀘 𑀫𑀢𑁆𑀣𑀺 𑀤𑀸𑀢𑀼𑀁,_  
_𑀧𑀭𑁂 𑀘 𑀔𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺𑀁;_  
_𑀧𑀽𑀚𑁂𑀣 𑀦𑀁 𑀧𑀽𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀥𑀸𑀢𑀼𑀁,_  
_𑀏𑀯𑀁 𑀓𑀺𑀭 𑀲𑀕𑁆𑀕𑀫𑀺𑀢𑁄 𑀕𑀫𑀺𑀲𑁆𑀲𑀣𑁇_  


(1206.)

1057\. _𑀢𑀤𑁂𑀯 𑀓𑀫𑁆𑀫𑀁 𑀓𑀼𑀲𑀮𑀁 𑀓𑀢𑀁 𑀫𑀬𑀸,_  
_𑀲𑀼𑀔𑀜𑁆𑀘 𑀤𑀺𑀩𑁆𑀩𑀁 𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀅𑀢𑁆𑀢𑀦𑀸;_  
_𑀫𑁄𑀤𑀸𑀫𑀳𑀁 𑀢𑀺𑀤𑀲𑀕𑀡𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂,_  
_𑀦 𑀢𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀔𑀬𑀫𑁆𑀧𑀺 𑀅𑀚𑁆𑀛𑀕𑀦𑁆”𑀢𑀺𑁇_  


---

1058\. 𑀅𑀦𑁂𑀓𑀯𑀡𑁆𑀡𑀯𑀺𑀫𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
