# 𑁨𑁇𑁩𑁇𑁩 𑀫𑀡𑀺𑀣𑀽𑀡𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀲𑀼𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀫𑀡𑀺𑀣𑀽𑀡𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(1126.)

975\. _“𑀉𑀘𑁆𑀘𑀫𑀺𑀤𑀁 𑀫𑀡𑀺𑀣𑀽𑀡𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀤𑁆𑀯𑀸𑀤𑀲 𑀬𑁄𑀚𑀦𑀸𑀦𑀺;_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀉𑀍𑀆𑀭𑀸,_  
_𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀣𑀫𑁆𑀪𑀸 𑀭𑀼𑀘𑀓𑀢𑁆𑀣𑀢𑀸 𑀲𑀼𑀪𑀸𑁇_  


(1127.)

976\. _𑀢𑀢𑁆𑀣𑀘𑁆𑀙𑀲𑀺 𑀧𑀺𑀯𑀲𑀺 𑀔𑀸𑀤𑀲𑀺 𑀘,_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀘 𑀯𑀻𑀡𑀸 𑀧𑀯𑀤𑀦𑁆𑀢𑀺 𑀯𑀕𑁆𑀕𑀼𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀭𑀲𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀢𑁆𑀣 𑀧𑀜𑁆𑀘,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀘 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀙𑀦𑁆𑀦𑀸𑁇_  


(1128--1129.)

977\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(1130.)

978\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(1131.)

979\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄,_  
_𑀯𑀺𑀯𑀦𑁂 𑀧𑀣𑁂 𑀲𑀗𑁆𑀓𑀫𑀦𑀁 𑀅𑀓𑀸𑀲𑀺𑀁;_  
_𑀆𑀭𑀸𑀫𑀭𑀼𑀓𑁆𑀔𑀸𑀦𑀺 𑀘 𑀭𑁄𑀧𑀬𑀺𑀲𑁆𑀲𑀁,_  
_𑀧𑀺𑀬𑀸 𑀘 𑀫𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀅𑀳𑁂𑀲𑀼𑀁;_  
_𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀸𑀦𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀤𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀅𑀤𑀸𑀲𑀺𑀁𑁇_  


(1132--1133.)

980\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

981\. 𑀫𑀡𑀺𑀣𑀽𑀡𑀯𑀺𑀫𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
