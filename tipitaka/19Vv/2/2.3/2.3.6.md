# 𑁨𑁇𑁩𑁇𑁬 𑀕𑁄𑀧𑀸𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

### 𑀲𑀼𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀕𑁄𑀧𑀸𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(1159.)

1008\. _“𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀤𑁂𑀯𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀉𑀘𑁆𑀘𑁂 𑀯𑀺𑀫𑀸𑀦𑀫𑁆𑀳𑀺 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑁂;_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀸𑀪𑀭𑀡𑀁 𑀬𑀲𑀲𑁆𑀲𑀺𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑁂 𑀯𑀺𑀫𑀸𑀦𑀫𑁆𑀳𑀺 𑀬𑀣𑀸𑀧𑀺 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑁇_  


(1160.)

1009\. _𑀅𑀮𑀗𑁆𑀓𑀢𑁄 𑀫𑀮𑁆𑀬𑀥𑀭𑁄 𑀲𑀼𑀯𑀢𑁆𑀣𑁄,_  
_𑀲𑀼𑀓𑀼𑀡𑁆𑀟𑀮𑀻 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼;_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀸𑀪𑀭𑀡𑁄 𑀬𑀲𑀲𑁆𑀲𑀻,_  
_𑀤𑀺𑀩𑁆𑀩𑁂 𑀯𑀺𑀫𑀸𑀦𑀫𑁆𑀳𑀺 𑀬𑀣𑀸𑀧𑀺 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑁇_  


(1161.)

1010\. _𑀤𑀺𑀩𑁆𑀩𑀸 𑀘 𑀯𑀻𑀡𑀸 𑀧𑀯𑀤𑀦𑁆𑀢𑀺 𑀯𑀕𑁆𑀕𑀼𑀁,_  
_𑀅𑀝𑁆𑀞𑀝𑁆𑀞𑀓𑀸 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀲𑀸𑀥𑀼𑀭𑀽𑀧𑀸;_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀘 𑀓𑀜𑁆𑀜𑀸 𑀢𑀺𑀤𑀲𑀘𑀭𑀸 𑀉𑀍𑀆𑀭𑀸,_  
_𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀫𑁄𑀤𑀬𑀦𑁆𑀢𑀺𑁇_  


(1162.)

1011\. _𑀤𑁂𑀯𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(1163.)

1012\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(1164.)

1013\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄,_  
_𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀭𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀁 𑀧𑀭𑁂𑀲𑀁 𑀥𑁂𑀦𑀼𑀬𑁄;_  
_𑀢𑀢𑁄 𑀘 𑀆𑀕𑀸 𑀲𑀫𑀡𑁄 𑀫𑀫𑀦𑁆𑀢𑀺𑀓𑁂,_  
_𑀕𑀸𑀯𑁄 𑀘 𑀫𑀸𑀲𑁂 𑀅𑀕𑀫𑀁𑀲𑀼 𑀔𑀸𑀤𑀺𑀢𑀼𑀁𑁇_  


(1165.)

1014\. _𑀤𑁆𑀯𑀬𑀚𑁆𑀚 𑀓𑀺𑀘𑁆𑀘𑀁 𑀉𑀪𑀬𑀜𑁆𑀘 𑀓𑀸𑀭𑀺𑀬𑀁,_  
_𑀇𑀘𑁆𑀘𑁂𑀯𑀳𑀁 𑀪𑀦𑁆𑀢𑁂 𑀢𑀤𑀸 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀁;_  
_𑀢𑀢𑁄 𑀘 𑀲𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀮𑀤𑁆𑀥 𑀬𑁄𑀦𑀺𑀲𑁄,_  
_𑀤𑀤𑀸𑀫𑀺 𑀪𑀦𑁆𑀢𑁂𑀢𑀺 𑀔𑀺𑀧𑀺𑀁 𑀅𑀦𑀦𑁆𑀢𑀓𑀁𑁇_  


(1166.)

1015\. _𑀲𑁄 𑀫𑀸𑀲𑀔𑁂𑀢𑁆𑀢𑀁 𑀢𑀼𑀭𑀺𑀢𑁄 𑀅𑀯𑀸𑀲𑀭𑀺𑀁,_  
_𑀧𑀼𑀭𑀸 𑀅𑀬𑀁 𑀪𑀜𑁆𑀚𑀢𑀺 𑀬𑀲𑁆𑀲𑀺𑀤𑀁 𑀥𑀦𑀁;_  
_𑀢𑀢𑁄 𑀘 𑀓𑀡𑁆𑀳𑁄 𑀉𑀭𑀕𑁄 𑀫𑀳𑀸𑀯𑀺𑀲𑁄,_  
_𑀅𑀟𑀁𑀲𑀺 𑀧𑀸𑀤𑁂 𑀢𑀼𑀭𑀺𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄𑁇_  


(1167.)

1016\. _𑀲𑁆𑀯𑀸𑀳𑀁 𑀅𑀝𑁆𑀝𑁄𑀫𑁆𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀧𑀻𑀍𑀇𑀢𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀘 𑀢𑀁 𑀲𑀸𑀫𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸𑀦𑀦𑁆𑀢𑀓𑀁;_  
_𑀅𑀳𑀸𑀲𑀺 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁 𑀫𑀫𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀬𑀸,_  
_𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀓𑀸𑀮𑀓𑀢𑁄𑀫𑁆𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑁇_  


(1168.)

1017\. _𑀢𑀤𑁂𑀯 𑀓𑀫𑁆𑀫𑀁 𑀓𑀼𑀲𑀮𑀁 𑀓𑀢𑀁 𑀫𑀬𑀸,_  
_𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀅𑀢𑁆𑀢𑀦𑀸;_  
_𑀢𑀬𑀸 𑀳𑀺 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀢𑁄 𑀪𑀼𑀲𑀁,_  
_𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀅𑀪𑀺𑀯𑀸𑀤𑀬𑀸𑀫𑀺 𑀢𑀁𑁇_  


(1169.)

1018\. _𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀘,_  
_𑀅𑀜𑁆𑀜𑁄 𑀫𑀼𑀦𑀺 𑀦𑀢𑁆𑀣𑀺 𑀢𑀬𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄;_  
_𑀢𑀬𑀸 𑀳𑀺 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀢𑁄 𑀪𑀼𑀲𑀁,_  
_𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀅𑀪𑀺𑀯𑀸𑀤𑀬𑀸𑀫𑀺 𑀢𑀁𑁇_  


(1170.)

1019\. _𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀧𑀦,_  
_𑀅𑀜𑁆𑀜𑁄 𑀫𑀼𑀦𑀻 𑀦𑀢𑁆𑀣𑀺 𑀢𑀬𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄;_  
_𑀢𑀬𑀸 𑀳𑀺 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀢𑁄 𑀪𑀼𑀲𑀁,_  
_𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀅𑀪𑀺𑀯𑀸𑀤𑀬𑀸𑀫𑀺 𑀢𑀦𑁆”𑀢𑀺𑁇_  


---
