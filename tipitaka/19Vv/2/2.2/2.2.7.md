# 𑁨𑁇𑁨𑁇𑁭 𑀬𑀯𑀧𑀸𑀮𑀓𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀯𑀕𑁆𑀕

* 𑀬𑀯𑀧𑀸𑀮𑀓𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(1087--1088.)

933\. _“𑀉𑀘𑁆𑀘𑀫𑀺𑀤𑀁 𑀫𑀡𑀺𑀣𑀽𑀡𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(1089.)

934\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(1090.)

935\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀬𑀯𑀧𑀸𑀮𑀓𑁄;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀁𑁇_  


(1091.)

936\. _𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁 𑀪𑀸𑀕𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺;_  
_𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑁄𑀤𑀸𑀫𑀺 𑀦𑀦𑁆𑀤𑀦𑁂 𑀯𑀦𑁂𑁇_  


(1092--1093.)

937\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

938\. 𑀬𑀯𑀧𑀸𑀮𑀓𑀯𑀺𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
