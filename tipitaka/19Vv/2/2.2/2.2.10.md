# 𑁨𑁇𑁨𑁇𑁧𑁦 (𑀉𑀢𑁆𑀢𑀭) 𑀧𑀸𑀬𑀸𑀲𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀯𑀕𑁆𑀕

* (𑀉𑀢𑁆𑀢𑀭) 𑀧𑀸𑀬𑀸𑀲𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(1108.)

953\. _“𑀬𑀸 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑀪𑀸 𑀲𑀼𑀥𑀫𑁆𑀫𑀸,_  
_𑀬𑀢𑁆𑀣𑀘𑁆𑀙𑀢𑀺 𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑁄 𑀲𑀫𑀕𑁆𑀕𑁄;_  
_𑀢𑀣𑀽𑀧𑀫𑀁 𑀢𑀼𑀬𑁆𑀳𑀫𑀺𑀤𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀑𑀪𑀸𑀲𑀬𑀁 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂𑁇_  


(1109.)

954\. _𑀤𑁂𑀯𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(1110.)

955\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(1111.)

956\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄,_  
_𑀭𑀜𑁆𑀜𑁄 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑁆𑀲 𑀅𑀳𑁄𑀲𑀺𑀁 𑀫𑀸𑀡𑀯𑁄;_  
_𑀮𑀤𑁆𑀥𑀸 𑀥𑀦𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀁 𑀅𑀓𑀸𑀲𑀺𑀁,_  
_𑀧𑀺𑀬𑀸 𑀘 𑀫𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀅𑀳𑁂𑀲𑀼𑀁;_  
_𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀸𑀦𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀤𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀅𑀤𑀸𑀲𑀺𑀁𑁇_  


(1112--1113.)

957\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

958\. 𑀧𑀸𑀬𑀸𑀲𑀺𑀯𑀺𑀫𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

959\. 𑀧𑀸𑀬𑀸𑀲𑀺𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇



---

960\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



961\. _𑀤𑁆𑀯𑁂 𑀅𑀕𑀸𑀭𑀺𑀦𑁄 𑀨𑀮𑀤𑀸𑀬𑀻,_  
_𑀤𑁆𑀯𑁂 𑀉𑀧𑀲𑁆𑀲𑀬𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀸𑀬 𑀤𑀸𑀬𑀻;_  
_𑀬𑀯𑀧𑀸𑀮𑀓𑁄 𑀘𑁂𑀯 𑀤𑁆𑀯𑁂,_  
_𑀓𑀼𑀡𑁆𑀟𑀮𑀺𑀦𑁄 𑀧𑀸𑀬𑀸𑀲𑀻𑀢𑀺𑁇_  


---

962\. 𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇
