# 𑁨𑁇𑁨𑁇𑁬 𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀓𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀓𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(1081.)

927\. _“𑀉𑀘𑁆𑀘𑀫𑀺𑀤𑀁 𑀫𑀡𑀺𑀣𑀽𑀡𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀤𑁆𑀯𑀸𑀤𑀲 𑀬𑁄𑀚𑀦𑀸𑀦𑀺;_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀉𑀍𑀆𑀭𑀸,_  
_𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀣𑀫𑁆𑀪𑀸 𑀭𑀼𑀘𑀓𑀢𑁆𑀣𑀢𑀸 𑀲𑀼𑀪𑀸𑁇_  


(1082.)

928\. _𑀤𑁂𑀯𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(1083.)

929\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(1084.)

930\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄,_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀢𑀲𑀺𑀢𑀁 𑀓𑀺𑀮𑀦𑁆𑀢𑀁;_  
_𑀏𑀓𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀁 𑀧𑀝𑀺𑀧𑀸𑀤𑀬𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀫𑀗𑁆𑀕𑀺 𑀪𑀢𑁆𑀢𑁂𑀦 𑀢𑀤𑀸 𑀅𑀓𑀸𑀲𑀺𑀁𑁇_  


(1085--1086.)

931\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

932\. 𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀓𑀯𑀺𑀫𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
