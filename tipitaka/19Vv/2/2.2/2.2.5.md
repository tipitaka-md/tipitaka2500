### 𑀤𑀼𑀢𑀺𑀬𑁅𑀧𑀲𑁆𑀲𑀬𑀤𑀸𑀬𑀓𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(1075--1078.)

923. _𑀲𑀼𑀭𑀺𑀬𑁄 𑀬𑀣𑀸 𑀯𑀺𑀕𑀢𑀯𑀮𑀸𑀳𑀓𑁂 𑀦𑀪𑁂,_  
_…𑀧𑁂…_

924. (𑀬𑀣𑀸 𑀧𑀼𑀭𑀺𑀫𑀯𑀺𑀫𑀸𑀦𑀁 𑀢𑀣𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)



(1079--1080.)

925. _“𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_
