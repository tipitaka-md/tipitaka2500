# 𑁨𑁇𑁧𑁇𑁧𑁧 𑀤𑀼𑀢𑀺𑀬𑀦𑀸𑀕𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀫𑀳𑀸𑀭𑀣𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀦𑀸𑀕𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(968.)

811\. _“𑀫𑀳𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀁 𑀅𑀪𑀺𑀭𑀼𑀬𑁆𑀳,_  
_𑀲𑀩𑁆𑀩𑀲𑁂𑀢𑀁 𑀕𑀚𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀯𑀦𑀸 𑀯𑀦𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑀸𑀲𑀺,_  
_𑀦𑀸𑀭𑀻𑀕𑀡𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑁄 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(969--970.)

812\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(971.)

813\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_𑀯𑀗𑁆𑀕𑀻𑀲𑁂𑀦𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(972.)

814\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄,_  
_𑀉𑀧𑀸𑀲𑀓𑁄 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑀁;_  
_𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑀁,_  
_𑀮𑁄𑀓𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀺𑀲𑁆𑀲𑀁𑁇_  


(973.)

815\. _𑀅𑀫𑀚𑁆𑀚𑀧𑁄 𑀦𑁄 𑀘 𑀫𑀼𑀲𑀸 𑀅𑀪𑀸𑀡𑀺𑀁,_  
_𑀲𑀓𑁂𑀦 𑀤𑀸𑀭𑁂𑀦 𑀘 𑀢𑀼𑀝𑁆𑀞𑁄 𑀅𑀳𑁄𑀲𑀺𑀁;_  
_𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀸𑀦𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀤𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀅𑀤𑀸𑀲𑀺𑀁𑁇_  


(974--975.)

816\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

817\. 𑀤𑀼𑀢𑀺𑀬𑀦𑀸𑀕𑀯𑀺𑀫𑀸𑀦𑀁 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇
