# 𑁨𑁇𑁧𑁇𑁭 𑀤𑀼𑀢𑀺𑀬𑀓𑀭𑀡𑀻𑀬𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀫𑀳𑀸𑀭𑀣𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀓𑀭𑀡𑀻𑀬𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(935.)

781\. _“𑀉𑀘𑁆𑀘𑀫𑀺𑀤𑀁 𑀫𑀡𑀺𑀣𑀽𑀡𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀤𑁆𑀯𑀸𑀤𑀲 𑀬𑁄𑀚𑀦𑀸𑀦𑀺;_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀉𑀍𑀆𑀭𑀸,_  
_𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀣𑀫𑁆𑀪𑀸 𑀭𑀼𑀘𑀓𑀢𑁆𑀣𑀢𑀸 𑀲𑀼𑀪𑀸𑁇_  


(936.)

782\. _𑀢𑀢𑁆𑀣𑀘𑁆𑀙𑀲𑀺 𑀧𑀺𑀯𑀲𑀺 𑀔𑀸𑀤𑀲𑀺 𑀘,_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀘 𑀯𑀻𑀡𑀸 𑀧𑀯𑀤𑀦𑁆𑀢𑀺 𑀯𑀕𑁆𑀕𑀼𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀭𑀲𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀢𑁆𑀣 𑀧𑀜𑁆𑀘,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀘 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀙𑀦𑁆𑀦𑀸𑁇_  


(937--938.)

783\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(939.)

784\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(940.)

785\. _“𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁂𑀦 𑀯𑀺𑀚𑀸𑀦𑀢𑀸;_  
_𑀲𑀫𑁆𑀫𑀕𑁆𑀕𑀢𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼,_  
_𑀬𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(941.)

786\. _𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀢 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀅𑀭𑀜𑁆𑀜𑀸 𑀕𑀸𑀫𑀫𑀸𑀕𑀢𑁄;_  
_𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀽𑀧𑀕𑁄 𑀅𑀳𑀁𑁇_  


(942--943.)

787\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

788\. 𑀤𑀼𑀢𑀺𑀬𑀓𑀭𑀡𑀻𑀬𑀯𑀺𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
