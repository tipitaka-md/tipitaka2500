# 𑁨𑁇𑁧𑁇𑁧𑁨 𑀢𑀢𑀺𑀬𑀦𑀸𑀕𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

* 𑀫𑀳𑀸𑀭𑀣𑀯𑀕𑁆𑀕

* 𑀢𑀢𑀺𑀬𑀦𑀸𑀕𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(976.)

818\. _“𑀓𑁄 𑀦𑀼 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀬𑀸𑀦𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀲𑁂𑀢𑁂𑀦 𑀳𑀢𑁆𑀣𑀺𑀦𑀸;_  
_𑀢𑀽𑀭𑀺𑀬𑀢𑀸𑀍𑀇𑀢𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑁄,_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀫𑀳𑀻𑀬𑀢𑀺𑁇_  


(977.)

819\. _𑀤𑁂𑀯𑀢𑀸 𑀦𑀼𑀲𑀺 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀅𑀤𑀼 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇_  


(978.)

820\. _“𑀦𑀫𑁆𑀳𑀺 𑀤𑁂𑀯𑁄 𑀦 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀦𑀸𑀧𑀺 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫 𑀬𑁂 𑀤𑁂𑀯𑀸,_  
_𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇_  


(979.)

821\. _“𑀧𑀼𑀘𑁆𑀙𑀸𑀫 𑀤𑁂𑀯𑀁 𑀲𑀼𑀥𑀫𑁆𑀫𑀁,_  
_𑀧𑀼𑀣𑀼𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀓𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑁂 𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


(980.)

822\. _“𑀉𑀘𑁆𑀙𑀸𑀕𑀸𑀭𑀁 𑀢𑀺𑀡𑀸𑀕𑀸𑀭𑀁,_  
_𑀯𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀬𑁄 𑀤𑀤𑁂;_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


---

823\. 𑀢𑀢𑀺𑀬𑀦𑀸𑀕𑀯𑀺𑀫𑀸𑀦𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁𑁇
