# 𑁨𑁇𑁧𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑀲𑀽𑀘𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀫𑀸𑀦

### 𑀫𑀳𑀸𑀭𑀣𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀽𑀘𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(952.)

796\. _“𑀉𑀘𑁆𑀘𑀫𑀺𑀤𑀁 𑀫𑀡𑀺𑀣𑀽𑀡𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀤𑁆𑀯𑀸𑀤𑀲 𑀬𑁄𑀚𑀦𑀸𑀦𑀺;_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀉𑀍𑀆𑀭𑀸,_  
_𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀣𑀫𑁆𑀪𑀸 𑀭𑀼𑀘𑀓𑀢𑁆𑀣𑀢𑀸 𑀲𑀼𑀪𑀸𑁇_  


(953.)

797\. _𑀢𑀢𑁆𑀣𑀘𑁆𑀙𑀲𑀺 𑀧𑀺𑀯𑀲𑀺 𑀔𑀸𑀤𑀲𑀺 𑀘,_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀘 𑀯𑀻𑀡𑀸 𑀧𑀯𑀤𑀦𑁆𑀢𑀺 𑀯𑀕𑁆𑀕𑀼𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀭𑀲𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀢𑁆𑀣 𑀧𑀜𑁆𑀘,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀘 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀙𑀦𑁆𑀦𑀸𑁇_  


(954--955.)

798\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(956.)

799\. _𑀲𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(957.)

800\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄,_  
_𑀧𑀼𑀭𑀺𑀫𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂𑁇_  


(958.)

801\. _𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀁;_  
_𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁 𑀲𑀽𑀘𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(959--960.)

802\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---
