# 𑁧𑁇𑁩𑁇𑁧 𑀉𑀍𑀆𑀭𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀯𑀕𑁆𑀕

* 𑀉𑀍𑀆𑀭𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(286.)

308\. _“𑀉𑀍𑀆𑀭𑁄 𑀢𑁂 𑀬𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀲𑀩𑁆𑀩𑀸 𑀑𑀪𑀸𑀲𑀢𑁂 𑀤𑀺𑀲𑀸;_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀮𑀗𑁆𑀓𑀢𑀸𑁇_  


(287.)

309\. _𑀫𑁄𑀤𑁂𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑀯 𑀧𑀽𑀚𑀸𑀬 𑀤𑁂𑀯𑀢𑁂;_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀯𑀺𑀫𑀸𑀦𑀸𑀦𑀺,_  
_𑀢𑀯𑀺𑀫𑀸𑀦𑀺 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁂𑁇_  


(288.)

310\. _𑀢𑀼𑀯𑀁𑀲𑀺 𑀇𑀲𑁆𑀲𑀭𑀸 𑀢𑁂𑀲𑀁,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀦𑀻;_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀸 𑀫𑀳𑀦𑁆𑀢𑀸𑀲𑀺,_  
_𑀤𑁂𑀯𑀓𑀸𑀬𑁂 𑀧𑀫𑁄𑀤𑀲𑀺;_  
_𑀤𑁂𑀯𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸𑀘𑀺𑀓𑁆𑀔,_  
_𑀓𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


(289.)

311\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂;_  
_𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀓𑀼𑀮𑁂 𑀲𑀼𑀡𑀺𑀲𑀸 𑀅𑀳𑁄𑀲𑀺𑀁,_  
_𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁂𑀲𑀼 𑀓𑀤𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀳𑀁𑁇_  


(290.)

312\. _𑀲𑀤𑁆𑀥𑀸 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑀸 𑀲𑀤𑀸;_  
_𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀧𑀽𑀯𑀁 𑀢𑁂 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(291.)

313\. _𑀢𑀤𑀸𑀳𑀁 𑀲𑀲𑁆𑀲𑀼𑀬𑀸𑀘𑀺𑀓𑁆𑀔𑀺𑀁,_  
_𑀲𑀫𑀡𑁄 𑀆𑀕𑀢𑁄 𑀇𑀥;_  
_𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁 𑀧𑀽𑀯𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(292.)

314\. _𑀇𑀢𑀺𑀲𑁆𑀲𑀸 𑀲𑀲𑁆𑀲𑀼 𑀧𑀭𑀺𑀪𑀸𑀲𑀺,_  
_𑀅𑀯𑀺𑀦𑀻𑀢𑀸𑀲𑀺 𑀢𑁆𑀯𑀁 𑀯𑀥𑀼;_  
_𑀦 𑀫𑀁 𑀲𑀫𑁆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁 𑀇𑀘𑁆𑀙𑀺,_  
_𑀲𑀫𑀡𑀲𑁆𑀲 𑀤𑀤𑀸𑀫𑀳𑀁𑁇_  


(293.)

315\. _𑀢𑀢𑁄 𑀫𑁂 𑀲𑀲𑁆𑀲𑀼 𑀓𑀼𑀧𑀺𑀢𑀸,_  
_𑀧𑀳𑀸𑀲𑀺 𑀫𑀼𑀲𑀮𑁂𑀦 𑀫𑀁;_  
_𑀓𑀽𑀝𑀗𑁆𑀕𑀘𑁆𑀙𑀺 𑀅𑀯𑀥𑀺 𑀫𑀁,_  
_𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀚𑀻𑀯𑀺𑀢𑀼𑀁 𑀘𑀺𑀭𑀁𑁇_  


(294.)

316\. _𑀲𑀸 𑀅𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀢𑀢𑁄 𑀘𑀼𑀢𑀸;_  
_𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑀸 𑀲𑀳𑀩𑁆𑀬𑀢𑀁𑁇_  


(295.)

317\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

318\. 𑀉𑀍𑀆𑀭𑀯𑀺𑀫𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
