# 𑁧𑁇𑁩𑁇𑁫𑁇𑁭 𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀯𑀕𑁆𑀕

* 𑀕𑀼𑀢𑁆𑀢𑀺𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(377--379.)

399\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(380.)

400\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(381.)

401\. _“𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀅𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘,_  
_𑀅𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀧𑀦𑁆𑀣𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂;_  
_𑀢𑁂𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀺𑀲𑁆𑀲𑀁𑁇_  


(382.)

402\. _𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀧𑀲𑁆𑀲 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀺𑀦𑀻𑀳𑀫𑀲𑁆𑀫𑀺;_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀧𑀯𑀭𑀸 𑀧𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁𑁇_  


(383--384.)

403\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_
