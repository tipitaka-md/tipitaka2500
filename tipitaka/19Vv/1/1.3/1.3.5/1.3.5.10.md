# 𑁧𑁇𑁩𑁇𑁫𑁇𑁧𑁦 𑀅𑀧𑀭𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀺𑀦𑀻𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀼𑀢𑁆𑀢𑀺𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀅𑀧𑀭𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀺𑀦𑀻𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(401--403.)

414\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(404.)

415\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(405.)

416\. _“𑀧𑀭𑀓𑀫𑁆𑀫𑀓𑀭𑀻 𑀆𑀲𑀺𑀁,_  
_𑀅𑀢𑁆𑀣𑁂𑀦𑀸𑀢𑀦𑁆𑀤𑀺𑀢𑀸 𑀤𑀸𑀲𑀻;_  
_𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑀸𑀦𑀢𑀺𑀫𑀸𑀦𑀺𑀦𑀻,_  
_𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀲𑀓𑀲𑁆𑀲 𑀪𑀸𑀕𑀲𑁆𑀲𑁇_  


(406.)

417\. _𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀧𑀲𑁆𑀲 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀺𑀦𑀻𑀳𑀫𑀲𑁆𑀫𑀺;_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀧𑀯𑀭𑀸 𑀧𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁𑁇_  


(407--408.)

418\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_
