# 𑁧𑁇𑁩𑁇𑁫𑁇𑁮 𑀉𑀤𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀼𑀢𑁆𑀢𑀺𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀉𑀤𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(385--387.)

404. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(388.)

405. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(389.)

406. _“𑀉𑀤𑀓𑁂 𑀞𑀺𑀢𑀸 𑀉𑀤𑀓𑀫𑀤𑀸𑀲𑀺𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦;_  
_𑀏𑀯𑀁 𑀧𑀺𑀬𑀭𑀽𑀧𑀤𑀸𑀬𑀺𑀓𑀸 𑀫𑀦𑀸𑀧𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀁 𑀲𑀸 𑀮𑀪𑀢𑁂 𑀉𑀧𑁂𑀘𑁆𑀘 𑀞𑀸𑀦𑀁𑁇_  


(390.)

407. _𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀧𑀲𑁆𑀲 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀺𑀦𑀻𑀳𑀫𑀲𑁆𑀫𑀺;_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀧𑀯𑀭𑀸 𑀧𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁𑁇_  


(391--392.)

408. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_
