# 𑁧𑁇𑁩𑁇𑁫𑁇𑁬 𑀕𑀦𑁆𑀥𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀼𑀢𑁆𑀢𑀺𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀕𑀦𑁆𑀥𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(369.)

392\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(370--371.)

393\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(372.)

394\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(373.)

395\. _“𑀕𑀦𑁆𑀥𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀓𑀁 𑀅𑀳𑀫𑀤𑀸𑀲𑀺𑀁,_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀣𑀽𑀧𑀫𑁆𑀳𑀺;_  
_𑀏𑀯𑀁 𑀧𑀺𑀬𑀭𑀽𑀧𑀤𑀸𑀬𑀺𑀓𑀸 𑀫𑀦𑀸𑀧𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀁 𑀲𑀸 𑀮𑀪𑀢𑁂 𑀉𑀧𑁂𑀘𑁆𑀘 𑀞𑀸𑀦𑀁𑁇_  


(374.)

396\. _𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀧𑀲𑁆𑀲 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀺𑀦𑀻𑀳𑀫𑀲𑁆𑀫𑀺;_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀧𑀯𑀭𑀸 𑀧𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁𑁇_  


(375--376.)

397\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


398\. (𑀅𑀦𑀦𑁆𑀢𑀭𑀁 𑀘𑀢𑀼𑀭𑀯𑀺𑀫𑀸𑀦𑀁 𑀬𑀣𑀸 𑀕𑀦𑁆𑀥𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀢𑀣𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
