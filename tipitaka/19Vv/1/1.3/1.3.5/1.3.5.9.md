# 𑁧𑁇𑁩𑁇𑁫𑁇𑁯 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀯𑀕𑁆𑀕

* 𑀕𑀼𑀢𑁆𑀢𑀺𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(393--395.)

409\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(396.)

410\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(397.)

411\. _“𑀲𑀲𑁆𑀲𑀼𑀜𑁆𑀘𑀸𑀳𑀁 𑀲𑀲𑀼𑀭𑀜𑁆𑀘,_  
_𑀘𑀡𑁆𑀟𑀺𑀓𑁂 𑀓𑁄𑀥𑀦𑁂 𑀘 𑀨𑀭𑀼𑀲𑁂 𑀘;_  
_𑀅𑀦𑀼𑀲𑀽𑀬𑀺𑀓𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀓𑁂𑀦 𑀲𑀻𑀮𑁂𑀦𑁇_  


(398.)

412\. _𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀧𑀲𑁆𑀲 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀺𑀦𑀻𑀳𑀫𑀲𑁆𑀫𑀺;_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀧𑀯𑀭𑀸 𑀧𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁𑁇_  


(399--400.)

413\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_
