# 𑁧𑁇𑁩𑁇𑁬 𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀯𑀕𑁆𑀕

* 𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(619.)

458\. _“𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀫𑀸𑀦𑀸 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀲𑀲𑀸 𑀘 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀻;_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂,_  
_𑀯𑀡𑁆𑀡𑁂𑀦 𑀅𑀢𑀺𑀭𑁄𑀘𑀲𑀺𑁇_  


(620.)

459\. _𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀇𑀤𑀁 𑀧𑀞𑀫𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀓𑀲𑁆𑀫𑀸 𑀓𑀸𑀬𑀸 𑀦𑀼 𑀆𑀕𑀫𑁆𑀫,_  
_𑀦𑀸𑀫𑁂𑀦 𑀪𑀸𑀲𑀲𑁂 𑀫𑀫𑀦𑁆”𑀢𑀺𑁇_  


(621.)

460\. _“𑀅𑀳𑀁 𑀪𑀤𑁆𑀤𑁂 𑀲𑀼𑀪𑀤𑁆𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀸𑀦𑀼𑀲𑀓𑁂 𑀪𑀯𑁂;_  
_𑀲𑀳𑀪𑀭𑀺𑀬𑀸 𑀘 𑀢𑁂 𑀆𑀲𑀺𑀁,_  
_𑀪𑀕𑀺𑀦𑀻 𑀘 𑀓𑀦𑀺𑀝𑁆𑀞𑀺𑀓𑀸𑁇_  


(622.)

461\. _𑀲𑀸 𑀅𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀢𑀢𑁄 𑀘𑀼𑀢𑀸;_  
_𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑀸 𑀲𑀳𑀩𑁆𑀬𑀢𑀦𑁆”𑀢𑀺𑁇_  


(623.)

462\. _“𑀧𑀳𑀽𑀢𑀓𑀢𑀓𑀮𑁆𑀬𑀸𑀡𑀸,_  
_𑀢𑁂 𑀤𑁂𑀯𑁂 𑀬𑀦𑁆𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑁄;_  
_𑀬𑁂𑀲𑀁 𑀢𑁆𑀯𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀲𑀺,_  
_𑀲𑀼𑀪𑀤𑁆𑀤𑁂 𑀚𑀸𑀢𑀺𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


(624.)

463\. _𑀅𑀣 𑀢𑁆𑀯𑀁 𑀓𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀓𑁂𑀦 𑀯𑀸 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀸;_  
_𑀓𑀻𑀤𑀺𑀲𑁂𑀦𑁂𑀯 𑀤𑀸𑀦𑁂𑀦,_  
_𑀲𑀼𑀩𑁆𑀩𑀢𑁂𑀦 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀻𑁇_  


(625.)

464\. _𑀬𑀲𑀁 𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀧𑀢𑁆𑀢𑀸,_  
_𑀯𑀺𑀲𑁂𑀲𑀁 𑀯𑀺𑀧𑀼𑀮𑀫𑀚𑁆𑀛𑀕𑀸;_  
_𑀤𑁂𑀯𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸𑀘𑀺𑀓𑁆𑀔,_  
_𑀓𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


(626.)

465\. _“𑀅𑀝𑁆𑀞𑁂𑀯 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀸𑀦𑀺,_  
_𑀬𑀁 𑀤𑀸𑀦𑀁 𑀅𑀤𑀤𑀁 𑀧𑀼𑀭𑁂;_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀲𑀁𑀖𑀲𑁆𑀲,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(627--628.)

466\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(629.)

467\. _“𑀅𑀳𑀁 𑀢𑀬𑀸 𑀩𑀳𑀼𑀢𑀭𑁂 𑀪𑀺𑀓𑁆𑀔𑀽,_  
_𑀲𑀜𑁆𑀜𑀢𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀬𑁄;_  
_𑀢𑀧𑁆𑀧𑁂𑀲𑀺𑀁 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(630.)

468\. _𑀢𑀬𑀸 𑀩𑀳𑀼𑀢𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀳𑀻𑀦𑀓𑀸𑀬𑀽𑀧𑀕𑀸 𑀅𑀳𑀁;_  
_𑀓𑀣𑀁 𑀢𑁆𑀯𑀁 𑀅𑀧𑁆𑀧𑀢𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀲𑁂𑀲𑀁 𑀯𑀺𑀧𑀼𑀮𑀫𑀚𑁆𑀛𑀕𑀸;_  
_𑀤𑁂𑀯𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸𑀘𑀺𑀓𑁆𑀔,_  
_𑀓𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


(631.)

469\. _“𑀫𑀦𑁄𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑁂 𑀧𑀼𑀭𑁂 𑀅𑀳𑀼;_  
_𑀢𑀸𑀳𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀲𑀺𑀁,_  
_𑀭𑁂𑀯𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀝𑁆𑀞𑀫𑀁𑁇_  


(632.)

470\. _𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑁄,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀭𑁂𑀯𑀢𑁄;_  
_𑀲𑀁𑀖𑁂 𑀤𑁂𑀳𑀻𑀢𑀺 𑀫𑀁𑀯𑁄𑀘,_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀓𑀭𑀺𑀁𑁇_  


(633.)

471\. _𑀲𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀲𑀁𑀖𑀕𑀢𑀸,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑀲𑀼 𑀢𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀦 𑀢𑀁 𑀢𑀯 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


(634.)

472\. _“𑀇𑀤𑀸𑀦𑁂𑀯𑀸𑀳𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀁𑀖𑁂 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁;_  
_𑀲𑀸𑀳𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁,_  
_𑀯𑀤𑀜𑁆𑀜𑀽 𑀯𑀻𑀢𑀫𑀘𑁆𑀙𑀭𑀸;_  
_𑀲𑀁𑀖𑁂 𑀤𑀸𑀦𑀸𑀦𑀺 𑀤𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀦𑁆”𑀢𑀺𑁇_  


(635.)

473\. _“𑀓𑀸 𑀏𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀤𑁆𑀤𑁂,_  
_𑀢𑀬𑀸 𑀫𑀦𑁆𑀢𑀬𑀢𑁂 𑀲𑀳;_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂,_  
_𑀯𑀡𑁆𑀡𑁂𑀦 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀻”𑀢𑀺𑁇_  


(636.)

474\. _“𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀤𑁂𑀯𑀺𑀦𑁆𑀤,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀸𑀦𑀼𑀲𑀓𑁂 𑀪𑀯𑁂;_  
_𑀲𑀳𑀪𑀭𑀺𑀬𑀸 𑀘 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀪𑀕𑀺𑀦𑀻 𑀘 𑀓𑀦𑀺𑀝𑁆𑀞𑀺𑀓𑀸;_  
_𑀲𑀁𑀖𑁂 𑀤𑀸𑀦𑀸𑀦𑀺 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑀸 𑀯𑀺𑀭𑁄𑀘𑀢𑀻”𑀢𑀺𑁇_  


(637.)

475\. _“𑀥𑀫𑁆𑀫𑁂𑀦 𑀧𑀼𑀩𑁆𑀩𑁂 𑀪𑀕𑀺𑀦𑀻,_  
_𑀢𑀬𑀸 𑀪𑀤𑁆𑀤𑁂 𑀯𑀺𑀭𑁄𑀘𑀢𑀺;_  
_𑀬𑀁 𑀲𑀁𑀖𑀫𑁆𑀳𑀺 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁂,_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁”𑁇_  


(638.)

476\. _“𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀳𑀺 𑀫𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀫𑁆𑀳𑀺 𑀧𑀩𑁆𑀩𑀢𑁂;_  
_𑀯𑀺𑀧𑀸𑀓𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀲𑁆𑀲,_  
_𑀬𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(639.)

477\. _𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀧𑁂𑀓𑁆𑀔𑀸𑀦 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀓𑀭𑁄𑀢𑀁 𑀑𑀧𑀥𑀺𑀓𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀬𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(640.)

478\. _𑀢𑀁 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀚𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀨𑀮𑀁 𑀲𑀓𑀁;_  
_𑀯𑀺𑀧𑀸𑀓𑀁 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀲𑁆𑀲,_  
_𑀬𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(641.)

479\. _𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀨𑀮𑁂 𑀞𑀺𑀢𑀸;_  
_𑀏𑀲 𑀲𑀁𑀖𑁄 𑀉𑀚𑀼𑀪𑀽𑀢𑁄,_  
_𑀧𑀜𑁆𑀜𑀸𑀲𑀻𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


(642.)

480\. _𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀧𑁂𑀓𑁆𑀔𑀸𑀦 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀓𑀭𑁄𑀢𑀁 𑀑𑀧𑀥𑀺𑀓𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀲𑀁𑀖𑁂 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(643.)

481\. _𑀏𑀲𑁄 𑀳𑀺 𑀲𑀁𑀖𑁄 𑀯𑀺𑀧𑀼𑀮𑁄 𑀫𑀳𑀕𑁆𑀕𑀢𑁄,_  
_𑀏𑀲𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀉𑀤𑀥𑀻𑀯 𑀲𑀸𑀕𑀭𑁄;_  
_𑀏𑀢𑁂 𑀳𑀺 𑀲𑁂𑀝𑁆𑀞𑀸 𑀦𑀭𑀯𑀻𑀭𑀲𑀸𑀯𑀓𑀸,_  
_𑀧𑀪𑀗𑁆𑀓𑀭𑀸 𑀥𑀫𑁆𑀫𑀫𑀼𑀤𑀻𑀭𑀬𑀦𑁆𑀢𑀺𑁇_  


(644.)

482\. _𑀢𑁂𑀲𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀼𑀳𑀼𑀢𑀁 𑀲𑀼𑀬𑀺𑀝𑁆𑀞𑀁,_  
_𑀬𑁂 𑀲𑀁𑀖𑀫𑀼𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀤𑀤𑀦𑁆𑀢𑀺 𑀤𑀸𑀦𑀁;_  
_𑀲𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀲𑀁𑀖𑀕𑀢𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸,_  
_𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀮𑁄𑀓𑀯𑀺𑀤𑀽𑀦 𑀯𑀡𑁆𑀡𑀺𑀢𑀸𑁇_  


(645.)

483\. _𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀬𑀜𑁆𑀜𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀸,_  
_𑀬𑁂 𑀯𑁂𑀤𑀚𑀸𑀢𑀸 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀺 𑀮𑁄𑀓𑁂;_  
_𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀁 𑀲𑀫𑀽𑀮𑀁,_  
_𑀅𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑀸 𑀲𑀕𑁆𑀕𑀫𑀼𑀧𑁂𑀦𑁆𑀢𑀺 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇_
