# 𑁧𑁇𑁪𑁇𑁪 𑀅𑀮𑁄𑀫𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀮𑁄𑀫𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(711.)

560\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀸 𑀢𑁆𑀯𑀁 𑀢𑀺𑀝𑁆𑀞𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀻 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(712--713.)

561\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(714.)

562\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(715.)

563\. _“𑀅𑀳𑀜𑁆𑀘 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑁄;_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀲𑀼𑀓𑁆𑀔𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(716.)

564\. _𑀲𑀼𑀓𑁆𑀔𑀸𑀬 𑀅𑀮𑁄𑀡𑀺𑀓𑀸𑀬 𑀘,_  
_𑀧𑀲𑁆𑀲 𑀨𑀮𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀺𑀬𑀸;_  
_𑀅𑀮𑁄𑀫𑀁 𑀲𑀼𑀔𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀓𑁄 𑀧𑀼𑀜𑁆𑀜𑀁 𑀦 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(717--718.)

565\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---
