# 𑁧𑁇𑁪𑁇𑁯 𑀧𑀻𑀢𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀻𑀢𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(795.)

632. _“𑀧𑀻𑀢𑀯𑀢𑁆𑀣𑁂 𑀧𑀻𑀢𑀥𑀚𑁂,_  
_𑀧𑀻𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑁂;_  
_𑀧𑀻𑀢𑀘𑀦𑁆𑀤𑀦𑀮𑀺𑀢𑁆𑀢𑀗𑁆𑀕𑁂,_  
_𑀧𑀻𑀢𑁅𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀺𑀦𑀻𑁇_  


(796.)

633. _𑀧𑀻𑀢𑀧𑀸𑀲𑀸𑀤𑀲𑀬𑀦𑁂,_  
_𑀧𑀻𑀢𑀸𑀲𑀦𑁂 𑀧𑀻𑀢𑀪𑀸𑀚𑀦𑁂;_  
_𑀧𑀻𑀢𑀙𑀢𑁆𑀢𑁂 𑀧𑀻𑀢𑀭𑀣𑁂,_  
_𑀧𑀻𑀢𑀲𑁆𑀲𑁂 𑀧𑀻𑀢𑀩𑀻𑀚𑀦𑁂𑁇_  


(797.)

634. _𑀓𑀺𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀻 𑀪𑀤𑁆𑀤𑁂,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀸𑀦𑀼𑀲𑀓𑁂 𑀪𑀯𑁂;_  
_𑀤𑁂𑀯𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸𑀘𑀺𑀓𑁆𑀔,_  
_𑀓𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


(798.)

635. _“𑀓𑁄𑀲𑀸𑀢𑀓𑀻 𑀦𑀸𑀫 𑀮𑀢𑀢𑁆𑀣𑀺 𑀪𑀦𑁆𑀢𑁂,_  
_𑀢𑀺𑀢𑁆𑀢𑀺𑀓𑀸 𑀅𑀦𑀪𑀺𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀢𑀲𑁆𑀲𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀣𑀽𑀧𑀁 𑀅𑀪𑀺𑀳𑀭𑀺𑀁 𑀅𑀳𑀁𑁇_  


(799.)

636. _𑀲𑀢𑁆𑀣𑀼 𑀲𑀭𑀻𑀭𑀫𑀼𑀤𑁆𑀤𑀺𑀲𑁆𑀲,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀦𑀸𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀁 𑀅𑀯𑁂𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀁,_  
_𑀦 𑀢𑀕𑁆𑀕𑀫𑀦𑀲𑀸 𑀲𑀢𑀻𑁇_  


(800.)

637. _𑀢𑀢𑁄 𑀫𑀁 𑀅𑀯𑀥𑀻 𑀕𑀸𑀯𑀻,_  
_𑀣𑀽𑀧𑀁 𑀅𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀁;_  
_𑀢𑀜𑁆𑀘𑀸𑀳𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀘𑁂𑀬𑁆𑀬𑀁,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀦𑀽𑀦 𑀇𑀢𑁄 𑀲𑀺𑀬𑀸𑁇_  


(801.)

638. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁂𑀯𑀺𑀦𑁆𑀤,_  
_𑀫𑀖𑀯𑀸 𑀤𑁂𑀯𑀓𑀼𑀜𑁆𑀚𑀭;_  
_𑀧𑀳𑀸𑀬 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀯 𑀲𑀳𑀩𑁆𑀬𑀫𑀸𑀕𑀢𑀸”𑀢𑀺𑁇_  


(802.)

639. _𑀇𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀺𑀤𑀲𑀸𑀥𑀺𑀧𑀢𑀺,_  
_𑀫𑀖𑀯𑀸 𑀤𑁂𑀯𑀓𑀼𑀜𑁆𑀚𑀭𑁄;_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑁄,_  
_𑀫𑀸𑀢𑀮𑀺𑀁 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀺𑁇_  


(803.)

640. _“𑀧𑀲𑁆𑀲 𑀫𑀸𑀢𑀮𑀺 𑀅𑀘𑁆𑀙𑁂𑀭𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀓𑀫𑁆𑀫𑀨𑀮𑀁 𑀇𑀤𑀁;_  
_𑀅𑀧𑁆𑀧𑀓𑀫𑁆𑀧𑀺 𑀓𑀢𑀁 𑀤𑁂𑀬𑁆𑀬𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(804.)

641. _𑀦𑀢𑁆𑀣𑀺 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀫𑁆𑀳𑀺,_  
_𑀅𑀧𑁆𑀧𑀓𑀸 𑀦𑀸𑀫 𑀤𑀓𑁆𑀔𑀺𑀡𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑁂 𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂,_  
_𑀅𑀣 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁂𑁇_  


(805.)

642. _𑀏𑀳𑀺 𑀫𑀸𑀢𑀮𑀺 𑀅𑀫𑁆𑀳𑁂𑀧𑀺,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀪𑀺𑀬𑁆𑀬𑁄 𑀫𑀳𑁂𑀫𑀲𑁂;_  
_𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀥𑀸𑀢𑀼𑀬𑁄,_  
_𑀲𑀼𑀔𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀦 𑀫𑀼𑀘𑁆𑀘𑀬𑁄𑁇_  


(806.)

643. _𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁂 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀘𑀸𑀧𑀺,_  
_𑀲𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀫𑀁 𑀨𑀮𑀁;_  
_𑀘𑁂𑀢𑁄𑀧𑀡𑀺𑀥𑀺𑀳𑁂𑀢𑀼 𑀳𑀺,_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀼𑀕𑁆𑀕𑀢𑀺𑀁𑁇_  


(807.)

644. _𑀩𑀳𑀽𑀦𑀁 𑀯𑀢 𑀅𑀢𑁆𑀣𑀸𑀬,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀸;_  
_𑀬𑀢𑁆𑀣 𑀓𑀸𑀭𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀕𑁆𑀕𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀸𑀬𑀓𑀸”𑀢𑀺𑁇_  


---

645. 𑀧𑀻𑀢𑀯𑀺𑀫𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
