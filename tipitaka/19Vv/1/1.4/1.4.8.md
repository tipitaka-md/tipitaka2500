# 𑁧𑁇𑁪𑁇𑁮 𑀅𑀫𑁆𑀩𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀫𑁆𑀩𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(783.)

621\. _“𑀤𑀺𑀩𑁆𑀩𑀁 𑀢𑁂 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀧𑀸𑀲𑀸𑀤𑁂𑀢𑁆𑀣 𑀫𑀳𑀮𑁆𑀮𑀓𑁄;_  
_𑀦𑀸𑀦𑀸𑀢𑀽𑀭𑀺𑀬𑀲𑀗𑁆𑀖𑀼𑀝𑁆𑀞𑁄,_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀕𑀡𑀖𑁄𑀲𑀺𑀢𑁄𑁇_  


(784.)

622\. _𑀧𑀤𑀻𑀧𑁄 𑀘𑁂𑀢𑁆𑀣 𑀚𑀮𑀢𑀺,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀬𑁄 𑀫𑀳𑀸;_  
_𑀤𑀼𑀲𑁆𑀲𑀨𑀮𑁂𑀳𑀺 𑀭𑀼𑀓𑁆𑀔𑁂𑀳𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄𑁇_  


(785--786.)

623\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(787.)

624\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(788.)

625\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂;_  
_𑀯𑀺𑀳𑀸𑀭𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀸𑀭𑁂𑀲𑀺𑀁,_  
_𑀅𑀫𑁆𑀩𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑀁𑁇_  


(789.)

626\. _𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑁂,_  
_𑀓𑀸𑀭𑁂𑀦𑁆𑀢𑁂 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀫𑀳𑁂;_  
_𑀅𑀫𑁆𑀩𑁂𑀳𑀺 𑀙𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀫𑀬𑁂 𑀨𑀮𑁂𑁇_  


(790.)

627\. _𑀧𑀤𑀻𑀧𑀁 𑀢𑀢𑁆𑀣 𑀚𑀸𑀮𑁂𑀢𑁆𑀯𑀸,_  
_𑀪𑁄𑀚𑀬𑀺𑀢𑁆𑀯𑀸 𑀕𑀡𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀲𑀺𑀁 𑀢𑀁 𑀲𑀁𑀖𑀲𑁆𑀲,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(791.)

628\. _𑀢𑁂𑀦 𑀫𑁂 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀧𑀸𑀲𑀸𑀤𑁂𑀢𑁆𑀣 𑀫𑀳𑀮𑁆𑀮𑀓𑁄;_  
_𑀦𑀸𑀦𑀸𑀢𑀽𑀭𑀺𑀬𑀲𑀗𑁆𑀖𑀼𑀝𑁆𑀞𑁄,_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀕𑀡𑀖𑁄𑀲𑀺𑀢𑁄𑁇_  


(792.)

629\. _𑀧𑀤𑀻𑀧𑁄 𑀘𑁂𑀢𑁆𑀣 𑀚𑀮𑀢𑀺,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀬𑁄 𑀫𑀳𑀸;_  
_𑀤𑀼𑀲𑁆𑀲𑀨𑀮𑁂𑀳𑀺 𑀭𑀼𑀓𑁆𑀔𑁂𑀳𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄𑁇_  


(793--794.)

630\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

631\. 𑀅𑀫𑁆𑀩𑀯𑀺𑀫𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
