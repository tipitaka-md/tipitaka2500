# 𑁧𑁇𑁪𑁇𑁭 𑀘𑀢𑀼𑀭𑀺𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕

* 𑀘𑀢𑀼𑀭𑀺𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(755--757.)

604\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(758.)

605\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(759.)

606\. _“𑀇𑀦𑁆𑀤𑀻𑀯𑀭𑀸𑀦𑀁 𑀳𑀢𑁆𑀣𑀓𑀁 𑀅𑀳𑀫𑀤𑀸𑀲𑀺𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲;_  
_𑀏𑀲𑀺𑀓𑀸𑀦𑀁 𑀉𑀡𑁆𑀡𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀕𑀭𑀯𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀢𑁂 𑀭𑀫𑁆𑀫𑁂𑁇_  


(760--761.)

607\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(762--764.)

608\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(765.)

609\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(766.)

610\. _“𑀦𑀻𑀮𑀼𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀓𑀁 𑀅𑀳𑀫𑀤𑀸𑀲𑀺𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲;_  
_𑀏𑀲𑀺𑀓𑀸𑀦𑀁 𑀉𑀡𑁆𑀡𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀕𑀭𑀯𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀢𑁂 𑀭𑀫𑁆𑀫𑁂𑁇_  


(767--768.)

611\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(769--771.)

612\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(772.)

613\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(773.)

614\. _“𑀑𑀤𑀸𑀢𑀫𑀽𑀮𑀓𑀁 𑀳𑀭𑀺𑀢𑀧𑀢𑁆𑀢𑀁,_  
_𑀉𑀤𑀓𑀲𑁆𑀫𑀺𑀁 𑀲𑀭𑁂 𑀚𑀸𑀢𑀁 𑀅𑀳𑀫𑀤𑀸𑀲𑀺𑀁;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀏𑀲𑀺𑀓𑀸𑀦𑀁 𑀉𑀡𑁆𑀡𑀢𑀲𑁆𑀫𑀺𑀁;_  
_𑀦𑀕𑀭𑀯𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀢𑁂 𑀭𑀫𑁆𑀫𑁂𑁇_  


(774--775.)

615\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(776--778.)

616\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(779.)

617\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(780.)

618\. _“𑀅𑀳𑀁 𑀲𑀼𑀫𑀦𑀸 𑀲𑀼𑀫𑀦𑀲𑁆𑀲 𑀲𑀼𑀫𑀦𑀫𑀓𑀼𑀍𑀆𑀦𑀺,_  
_𑀤𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀅𑀳𑀫𑀤𑀸𑀲𑀺𑀁;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀏𑀲𑀺𑀓𑀸𑀦𑀁 𑀉𑀡𑁆𑀡𑀢𑀲𑁆𑀫𑀺𑀁;_  
_𑀦𑀕𑀭𑀯𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀢𑁂 𑀭𑀫𑁆𑀫𑁂𑁇_  


(781--782.)

619\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

620\. 𑀘𑀢𑀼𑀭𑀺𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
