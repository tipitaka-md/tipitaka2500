# 𑁧𑁇𑁨𑁇𑁧𑁧 𑀤𑀼𑀢𑀺𑀬𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(278.)

297. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀸 𑀢𑁆𑀯𑀁 𑀢𑀺𑀝𑁆𑀞𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀻 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(279--280.)

298. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(281.)

299. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(282.)

300. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂𑁇_  


(283.)

301. _𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀁;_  
_𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(284--285.)

302. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

303. 𑀤𑀼𑀢𑀺𑀬𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇



---

304. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



305. _𑀤𑀸𑀲𑀻 𑀘𑁂𑀯 𑀮𑀔𑀼𑀫𑀸 𑀘,_  
_𑀅𑀣 𑀆𑀘𑀸𑀫𑀤𑀸𑀬𑀺𑀓𑀸;_  
_𑀘𑀡𑁆𑀟𑀸𑀮𑀻 𑀪𑀤𑁆𑀤𑀺𑀢𑁆𑀣𑀻 𑀘𑁂𑀯,_  
_𑀲𑁄𑀡𑀤𑀺𑀦𑁆𑀦𑀸 𑀉𑀧𑁄𑀲𑀣𑀸;_  
_𑀦𑀺𑀤𑁆𑀤𑀸 𑀘𑁂𑀯 𑀲𑀼𑀦𑀺𑀤𑁆𑀤𑀸 𑀘,_  
_𑀤𑁆𑀯𑁂 𑀘 𑀪𑀺𑀓𑁆𑀔𑀸𑀬 𑀤𑀸𑀬𑀺𑀓𑀸;_  
_𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_  


---

306. 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄𑁇



---
