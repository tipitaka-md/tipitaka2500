# 𑁧𑁇𑁨𑁇𑁯 𑀲𑀼𑀦𑀺𑀤𑁆𑀤𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀦𑀺𑀤𑁆𑀤𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(258.)

282\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_…𑀧𑁂…_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(259--260.)

283\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(261.)

284\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(262.)

285\. _“𑀲𑀼𑀦𑀺𑀤𑁆𑀤𑀸𑀢𑀺 𑀫𑀁 𑀅𑀜𑁆𑀜𑀁𑀲𑀼,_  
_𑀭𑀸𑀚𑀕𑀳𑀲𑁆𑀫𑀺𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸;_  
_𑀲𑀤𑁆𑀥𑀸 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑀸 𑀲𑀤𑀸𑁇_  


(263--266.)

286\. _( )_  


(267.)

287\. _𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂 𑀭𑀢𑀸,_  
_𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦 𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄,_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(268--269.)

288\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

289\. 𑀲𑀼𑀦𑀺𑀤𑁆𑀤𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
