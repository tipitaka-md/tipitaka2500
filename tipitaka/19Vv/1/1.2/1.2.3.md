# 𑁧𑁇𑁨𑁇𑁩 𑀆𑀘𑀸𑀫𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀆𑀘𑀸𑀫𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(185.)

208. _“𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀢𑁂 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄;_  
_𑀤𑀮𑀺𑀤𑁆𑀤𑀸 𑀓𑀧𑀡𑀸 𑀦𑀸𑀭𑀻,_  
_𑀧𑀭𑀸𑀕𑀸𑀭𑀁 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑀸𑁇_  


(186.)

209. _𑀬𑀸 𑀢𑁂 𑀅𑀤𑀸𑀲𑀺 𑀆𑀘𑀸𑀫𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺;_  
_𑀲𑀸 𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀓𑀁 𑀦𑀼 𑀲𑀸 𑀤𑀺𑀲𑀢𑀁 𑀕𑀢𑀸”𑀢𑀺𑁇_  


(187.)

210. _“𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀫𑁂 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑁄;_  
_𑀤𑀮𑀺𑀤𑁆𑀤𑀸 𑀓𑀧𑀡𑀸 𑀦𑀸𑀭𑀻,_  
_𑀧𑀭𑀸𑀕𑀸𑀭𑀁 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑀸𑁇_  


(188.)

211. _𑀬𑀸 𑀫𑁂 𑀅𑀤𑀸𑀲𑀺 𑀆𑀘𑀸𑀫𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺;_  
_𑀲𑀸 𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀇𑀢𑁄 𑀘𑀼𑀢𑀸𑁇_  


(189.)

212. _𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀺𑀦𑁄 𑀦𑀸𑀫,_  
_𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸;_  
_𑀢𑀢𑁆𑀣 𑀲𑀸 𑀲𑀼𑀔𑀺𑀢𑀸 𑀦𑀸𑀭𑀻,_  
_𑀫𑁄𑀤𑀢𑀸𑀘𑀸𑀫𑀤𑀸𑀬𑀺𑀓𑀸”𑀢𑀺𑁇_  


(190.)

213. _“𑀅𑀳𑁄 𑀤𑀸𑀦𑀁 𑀯𑀭𑀸𑀓𑀺𑀬𑀸,_  
_𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀧𑀭𑀸𑀪𑀢𑁂𑀦 𑀤𑀸𑀦𑁂𑀦,_  
_𑀇𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀯𑀢 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑁇_  


(191.)

214. _𑀬𑀸 𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀭𑀸𑀚𑀺𑀦𑁄;_  
_𑀦𑀸𑀭𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀓𑀮𑁆𑀬𑀸𑀡𑀻,_  
_𑀪𑀢𑁆𑀢𑀼 𑀘𑀸𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀓𑀸;_  
_𑀏𑀢𑀲𑁆𑀲𑀸𑀘𑀸𑀫𑀤𑀸𑀦𑀲𑁆𑀲,_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


(192.)

215. _𑀲𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀸 𑀲𑀢𑀁 𑀅𑀲𑁆𑀲𑀸,_  
_𑀲𑀢𑀁 𑀅𑀲𑁆𑀲𑀢𑀭𑀻𑀭𑀣𑀸;_  
_𑀲𑀢𑀁 𑀓𑀜𑁆𑀜𑀸𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀸;_  
_𑀏𑀢𑀲𑁆𑀲𑀸𑀘𑀸𑀫𑀤𑀸𑀦𑀲𑁆𑀲,_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


(193.)

216. _𑀲𑀢𑀁 𑀳𑁂𑀫𑀯𑀢𑀸 𑀦𑀸𑀕𑀸,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑀸 𑀉𑀭𑀽𑀍𑀳𑀯𑀸;_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀘𑁆𑀙𑀸 𑀫𑀸𑀢𑀗𑁆𑀕𑀸,_  
_𑀳𑁂𑀫𑀓𑀧𑁆𑀧𑀦𑀯𑀸𑀲𑀲𑀸;_  
_𑀏𑀢𑀲𑁆𑀲𑀸𑀘𑀸𑀫𑀤𑀸𑀦𑀲𑁆𑀲,_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


(194.)

217. _𑀘𑀢𑀼𑀦𑁆𑀦𑀫𑀧𑀺 𑀤𑀻𑀧𑀸𑀦𑀁,_  
_𑀇𑀲𑁆𑀲𑀭𑀁 𑀬𑁄𑀥 𑀓𑀸𑀭𑀬𑁂;_  
_𑀏𑀢𑀲𑁆𑀲𑀸𑀘𑀸𑀫𑀤𑀸𑀦𑀲𑁆𑀲,_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

218. 𑀆𑀘𑀸𑀫𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
