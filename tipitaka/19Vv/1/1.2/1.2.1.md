# 𑁧𑁇𑁨𑁇𑁧 𑀤𑀸𑀲𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀕𑁆𑀕

* 𑀤𑀸𑀲𑀺𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(157.)

178\. _“𑀅𑀧𑀺 𑀲𑀓𑁆𑀓𑁄𑀯 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀭𑀫𑁆𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀦𑁂;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑀸𑀲𑀺,_  
_𑀦𑀸𑀭𑀻𑀕𑀡𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀸;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀻 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(158.)

179\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀓𑁂𑀦 𑀢𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(159.)

180\. _𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀤𑁂𑀯𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(160.)

181\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(161.)

182\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀤𑀸𑀲𑀻 𑀅𑀳𑁄𑀲𑀺𑀁 𑀧𑀭𑀧𑁂𑀲𑁆𑀲𑀺𑀬𑀸 𑀓𑀼𑀮𑁂𑁇_  


(162.)

183\. _𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄,_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀆𑀲𑀺,_  
_𑀲𑀸𑀲𑀦𑁂 𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(163.)

184\. _𑀓𑀸𑀫𑀁 𑀪𑀺𑀚𑁆𑀚𑀢𑀼𑀬𑀁 𑀓𑀸𑀬𑁄,_  
_𑀦𑁂𑀯 𑀅𑀢𑁆𑀣𑁂𑀢𑁆𑀣 𑀲𑀡𑁆𑀞𑀦𑀁;_  
_𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁,_  
_𑀫𑀕𑁆𑀕𑁄 𑀲𑁄𑀯𑀢𑁆𑀣𑀺𑀓𑁄 𑀲𑀺𑀯𑁄𑁇_  


(164.)

185\. _𑀅𑀓𑀡𑁆𑀝𑀓𑁄 𑀅𑀕𑀳𑀦𑁄,_  
_𑀉𑀚𑀼 𑀲𑀩𑁆𑀪𑀺 𑀧𑀯𑁂𑀤𑀺𑀢𑁄;_  
_𑀦𑀺𑀓𑁆𑀓𑀫𑀲𑁆𑀲 𑀨𑀮𑀁 𑀧𑀲𑁆𑀲,_  
_𑀬𑀣𑀺𑀤𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀣𑀺𑀓𑀸𑁇_  


(165.)

186\. _𑀆𑀫𑀦𑁆𑀢𑀦𑀺𑀓𑀸 𑀭𑀜𑁆𑀜𑁄𑀫𑁆𑀳𑀺,_  
_𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀲𑀝𑁆𑀞𑀺 𑀢𑀽𑀭𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀝𑀺𑀩𑁄𑀥𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀫𑁂𑁇_  


(166.)

187\. _𑀆𑀮𑀫𑁆𑀩𑁄 𑀕𑀕𑁆𑀕𑀭𑁄 𑀪𑀻𑀫𑁄,_  
_𑀲𑀸𑀥𑀼𑀯𑀸𑀤𑀻 𑀘 𑀲𑀁𑀲𑀬𑁄;_  
_𑀧𑁄𑀓𑁆𑀔𑀭𑁄 𑀘 𑀲𑀼𑀨𑀲𑁆𑀲𑁄 𑀘,_  
_𑀯𑀺𑀡𑀸𑀫𑁄𑀓𑁆𑀔𑀸 𑀘 𑀦𑀸𑀭𑀺𑀬𑁄𑁇_  


(167.)

188\. _𑀦𑀦𑁆𑀤𑀸 𑀘𑁂𑀯 𑀲𑀼𑀦𑀦𑁆𑀤𑀸 𑀘,_  
_𑀲𑁄𑀡𑀤𑀺𑀦𑁆𑀦𑀸 𑀲𑀼𑀘𑀺𑀫𑁆𑀳𑀺𑀢𑀸;_  
_𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀸 𑀫𑀺𑀲𑁆𑀲𑀓𑁂𑀲𑀻 𑀘,_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀢𑀺 𑀤𑀸𑀭𑀼𑀡𑀻𑁇_  


(168.)

189\. _𑀏𑀡𑀻𑀨𑀲𑁆𑀲𑀸 𑀲𑀼𑀨𑀲𑁆𑀲𑀸 𑀘,_  
_𑀲𑀼𑀪𑀤𑁆𑀤𑀸 𑀫𑀼𑀤𑀼𑀯𑀸𑀤𑀺𑀦𑀻;_  
_𑀏𑀢𑀸 𑀘𑀜𑁆𑀜𑀸 𑀘 𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑁂,_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀦𑀁 𑀧𑀩𑁄𑀥𑀺𑀓𑀸𑁇_  


(169.)

190\. _𑀢𑀸 𑀫𑀁 𑀓𑀸𑀮𑁂𑀦𑀼𑀧𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀪𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀯𑀢𑀸;_  
_𑀳𑀦𑁆𑀤 𑀦𑀘𑁆𑀘𑀸𑀫 𑀕𑀸𑀬𑀸𑀫,_  
_𑀳𑀦𑁆𑀤 𑀢𑀁 𑀭𑀫𑀬𑀸𑀫𑀲𑁂𑁇_  


(170.)

191\. _𑀦𑀬𑀺𑀤𑀁 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁,_  
_𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀫𑁂𑀯𑀺𑀤𑀁;_  
_𑀅𑀲𑁄𑀓𑀁 𑀦𑀦𑁆𑀤𑀦𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀢𑀺𑀤𑀲𑀸𑀦𑀁 𑀫𑀳𑀸𑀯𑀦𑀁𑁇_  


(171.)

192\. _𑀲𑀼𑀔𑀁 𑀅𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁,_  
_𑀇𑀥 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑀢𑁆𑀣 𑀘;_  
_𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁,_  
_𑀇𑀥 𑀘𑁂𑀯 𑀧𑀭𑀢𑁆𑀣 𑀘𑁇_  


(172.)

193\. _𑀢𑁂𑀲𑀁 𑀲𑀳𑀩𑁆𑀬𑀓𑀸𑀫𑀸𑀦𑀁,_  
_𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀮𑀁 𑀩𑀳𑀼𑀁;_  
_𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑀸 𑀳𑀺 𑀫𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀲𑀕𑁆𑀕𑁂 𑀪𑁄𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄”𑀢𑀺𑁇_  


---

194\. 𑀤𑀸𑀲𑀺𑀯𑀺𑀫𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
