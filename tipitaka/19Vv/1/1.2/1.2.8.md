# 𑁧𑁇𑁨𑁇𑁮 𑀦𑀺𑀤𑁆𑀤𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(246.)

271. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀸 𑀢𑁆𑀯𑀁 𑀢𑀺𑀝𑁆𑀞𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀻 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(247--248.)

272. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(249.)

273. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(250.)

274. _“𑀦𑀺𑀤𑁆𑀤𑀸𑀢𑀺 𑀫𑀫𑀁 𑀅𑀜𑁆𑀜𑀁𑀲𑀼,_  
_𑀭𑀸𑀚𑀕𑀳𑀲𑁆𑀫𑀺𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸;_  
_𑀲𑀤𑁆𑀥𑀸 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑀸 𑀲𑀤𑀸𑁇_  


(251.)

275. _𑀅𑀘𑁆𑀙𑀸𑀤𑀦𑀜𑁆𑀘 𑀪𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀤𑀻𑀧𑀺𑀬𑀁;_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀉𑀚𑀼𑀪𑀽𑀢𑁂𑀲𑀼,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(252.)

276. _𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀁 𑀧𑀜𑁆𑀘𑀤𑀲𑀺𑀁,_  
_𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀫𑀻;_  
_𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀧𑀓𑁆𑀔𑀜𑁆𑀘,_  
_𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑀁𑁇_  


(253.)

277. _𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀤𑀸 𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀁𑀯𑀼𑀢𑀸;_  
_𑀲𑀜𑁆𑀜𑀫𑀸 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀸 𑀘,_  
_𑀯𑀺𑀫𑀸𑀦𑀁 𑀆𑀯𑀲𑀸𑀫𑀳𑀁𑁇_  


(254.)

278. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀢𑀸,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀘 𑀲𑀜𑁆𑀜𑀢𑀸;_  
_𑀣𑁂𑀬𑁆𑀬𑀸 𑀘 𑀅𑀢𑀺𑀘𑀸𑀭𑀸 𑀘,_  
_𑀫𑀚𑁆𑀚𑀧𑀸𑀦𑀸 𑀘 𑀆𑀭𑀓𑀸𑁇_  


(255.)

279. _𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂 𑀭𑀢𑀸,_  
_𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦 𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄,_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(256--257.)

280. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

281. 𑀦𑀺𑀤𑁆𑀤𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
