# 𑁧𑁇𑁨𑁇𑁧𑁦 𑀧𑀞𑀫𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(270.)

290\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀸 𑀢𑁆𑀯𑀁 𑀢𑀺𑀝𑁆𑀞𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀻 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(271--272.)

291\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(273.)

292\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_…𑀧𑁂…_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(274.)

293\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂𑁇_  


(275.)

294\. _𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀁;_  
_𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(276--277.)

295\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

296\. 𑀧𑀞𑀫𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇
