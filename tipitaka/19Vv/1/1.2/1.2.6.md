# 𑁧𑁇𑁨𑁇𑁬 𑀲𑁄𑀡𑀤𑀺𑀦𑁆𑀦𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀡𑀤𑀺𑀦𑁆𑀦𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(217.)

243. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀸 𑀢𑁆𑀯𑀁 𑀢𑀺𑀝𑁆𑀞𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀻 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(218.)

244. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀓𑁂𑀦 𑀢𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(219.)

245. _𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀤𑁂𑀯𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(220.)

246. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(221.)

247. _“𑀲𑁄𑀡𑀤𑀺𑀦𑁆𑀦𑀸𑀢𑀺 𑀫𑀁 𑀅𑀜𑁆𑀜𑀁𑀲𑀼,_  
_𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸;_  
_𑀲𑀤𑁆𑀥𑀸 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑀸 𑀲𑀤𑀸𑁇_  


(222.)

248. _𑀅𑀘𑁆𑀙𑀸𑀤𑀦𑀜𑁆𑀘 𑀪𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀤𑀻𑀧𑀺𑀬𑀁;_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀉𑀚𑀼𑀪𑀽𑀢𑁂𑀲𑀼,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(223.)

249. _𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀁 𑀧𑀜𑁆𑀘𑀤𑀲𑀺𑀁,_  
_𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀫𑀻;_  
_𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀧𑀓𑁆𑀔𑀜𑁆𑀘,_  
_𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑀁𑁇_  


(224.)

250. _𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀤𑀸 𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀁𑀯𑀼𑀢𑀸;_  
_𑀲𑀜𑁆𑀜𑀫𑀸 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀸 𑀘,_  
_𑀯𑀺𑀫𑀸𑀦𑀁 𑀆𑀯𑀲𑀸𑀫𑀳𑀁𑁇_  


(225.)

251. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀢𑀸,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀘 𑀲𑀜𑁆𑀜𑀢𑀸;_  
_𑀣𑁂𑀬𑁆𑀬𑀸 𑀘 𑀅𑀢𑀺𑀘𑀸𑀭𑀸 𑀘,_  
_𑀫𑀚𑁆𑀚𑀧𑀸𑀦𑀸 𑀘 𑀆𑀭𑀓𑀸𑁇_  


(226.)

252. _𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂 𑀭𑀢𑀸,_  
_𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦 𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄,_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(227--228.)

253. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_…𑀧𑁂…_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

254. 𑀲𑁄𑀡𑀤𑀺𑀦𑁆𑀦𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
