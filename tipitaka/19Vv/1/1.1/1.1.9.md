# 𑁧𑁇𑁧𑁇𑁯 𑀤𑀻𑀧𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀧𑀻𑀞𑀯𑀕𑁆𑀕

### 𑀤𑀻𑀧𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(75.)

84\. _“𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀸 𑀢𑁆𑀯𑀁 𑀢𑀺𑀝𑁆𑀞𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀻 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸𑁇_  


(76.)

85\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀓𑁂𑀦 𑀢𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(77.)

86\. _𑀓𑁂𑀦 𑀢𑁆𑀯𑀁 𑀯𑀺𑀫𑀮𑁄𑀪𑀸𑀲𑀸,_  
_𑀅𑀢𑀺𑀭𑁄𑀘𑀲𑀺 𑀤𑁂𑀯𑀢𑀸;_  
_𑀓𑁂𑀦 𑀢𑁂 𑀲𑀩𑁆𑀩𑀕𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀲𑀩𑁆𑀩𑀸 𑀑𑀪𑀸𑀲𑀢𑁂 𑀤𑀺𑀲𑀸𑁇_  


(78.)

87\. _𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀤𑁂𑀯𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(79.)

88\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(80.)

89\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂;_  
_𑀢𑀫𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀳𑀺 𑀢𑀺𑀫𑀻𑀲𑀺𑀓𑀸𑀬𑀁,_  
_𑀧𑀤𑀻𑀧𑀓𑀸𑀮𑀫𑁆𑀳𑀺 𑀅𑀤𑀸𑀲𑀺 𑀤𑀻𑀧𑀁𑁇_  


(81.)

90\. _𑀬𑁄 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀳𑀺 𑀢𑀺𑀫𑀻𑀲𑀺𑀓𑀸𑀬𑀁,_  
_𑀧𑀤𑀻𑀧𑀓𑀸𑀮𑀫𑁆𑀳𑀺 𑀤𑀤𑀸𑀢𑀺 𑀤𑀻𑀧𑀁;_  
_𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀚𑁄𑀢𑀺𑀭𑀲𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀧𑀳𑀽𑀢𑀫𑀮𑁆𑀬𑀁 𑀩𑀳𑀼𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀁𑁇_  


(82.)

91\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀫𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(83.)

92\. _𑀢𑁂𑀦𑀸𑀳𑀁 𑀯𑀺𑀫𑀮𑁄𑀪𑀸𑀲𑀸,_  
_𑀅𑀢𑀺𑀭𑁄𑀘𑀸𑀫𑀺 𑀤𑁂𑀯𑀢𑀸;_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀩𑁆𑀩𑀕𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀲𑀩𑁆𑀩𑀸 𑀑𑀪𑀸𑀲𑀢𑁂 𑀤𑀺𑀲𑀸𑁇_  


(84.)

93\. _𑀅𑀓𑁆𑀔𑀸𑀫𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀬𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

94\. 𑀤𑀻𑀧𑀯𑀺𑀫𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
