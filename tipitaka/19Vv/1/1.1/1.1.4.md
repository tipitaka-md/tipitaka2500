# 𑁧𑁇𑁧𑁇𑁪 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀻𑀞𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀧𑀻𑀞𑀯𑀕𑁆𑀕

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀻𑀞𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(23.)

27\. _“𑀧𑀻𑀞𑀁 𑀢𑁂 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀁 𑀉𑀍𑀆𑀭𑀁,_  
_𑀫𑀦𑁄𑀚𑀯𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀬𑁂𑀦𑀓𑀸𑀫𑀁;_  
_𑀅𑀮𑀗𑁆𑀓𑀢𑁂 𑀫𑀮𑁆𑀬𑀥𑀭𑁂 𑀲𑀼𑀯𑀢𑁆𑀣𑁂,_  
_𑀑𑀪𑀸𑀲𑀲𑀺 𑀯𑀺𑀚𑁆𑀚𑀼𑀭𑀺𑀯𑀩𑁆𑀪𑀓𑀽𑀝𑀁𑁇_  


(24.)

28\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀓𑁂𑀦 𑀢𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(25.)

29\. _𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀤𑁂𑀯𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(26.)

30\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(27.)

31\. _“𑀅𑀧𑁆𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀨𑀮𑀁 𑀫𑀫𑁂𑀤𑀁,_  
_𑀬𑁂𑀦𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸;_  
_𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂𑁇_  


(28.)

32\. _𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀁;_  
_𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁 𑀧𑀻𑀞𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(29.)

33\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀫𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(30.)

34\. _𑀅𑀓𑁆𑀔𑀸𑀫𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀬𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

35\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀻𑀞𑀯𑀺𑀫𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
