# 𑁧𑁇𑁧𑁇𑁧𑁧 𑀧𑀞𑀫𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

* 𑀧𑀻𑀞𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(93.)

104\. _“𑀓𑁄𑀜𑁆𑀘𑀸 𑀫𑀬𑀽𑀭𑀸 𑀤𑀺𑀯𑀺𑀬𑀸 𑀘 𑀳𑀁𑀲𑀸,_  
_𑀯𑀕𑁆𑀕𑀼𑀲𑁆𑀲𑀭𑀸 𑀓𑁄𑀓𑀺𑀮𑀸 𑀲𑀫𑁆𑀧𑀢𑀦𑁆𑀢𑀺;_  
_𑀧𑀼𑀧𑁆𑀨𑀸𑀪𑀺𑀓𑀺𑀡𑁆𑀡𑀁 𑀭𑀫𑁆𑀫𑀫𑀺𑀤𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀅𑀦𑁂𑀓𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀭𑀦𑀸𑀭𑀺𑀲𑁂𑀯𑀺𑀢𑀁𑁇_  


(94.)

105\. _𑀢𑀢𑁆𑀣𑀘𑁆𑀙𑀲𑀺 𑀤𑁂𑀯𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀇𑀤𑁆𑀥𑀻 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺 𑀅𑀦𑁂𑀓𑀭𑀽𑀧𑀸;_  
_𑀇𑀫𑀸 𑀘 𑀢𑁂 𑀅𑀘𑁆𑀙𑀭𑀸𑀬𑁄 𑀲𑀫𑀦𑁆𑀢𑀢𑁄,_  
_𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀫𑁄𑀤𑀬𑀦𑁆𑀢𑀺 𑀘𑁇_  


(95.)

106\. _𑀤𑁂𑀯𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑀸𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(96.)

107\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(97.)

108\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀦𑀜𑁆𑀜𑀫𑀦𑀸 𑀅𑀳𑁄𑀲𑀺𑀁;_  
_𑀫𑀸𑀢𑀸𑀯 𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀫𑀸𑀦𑀸,_  
_𑀓𑀼𑀤𑁆𑀥𑀸𑀧𑀺𑀳𑀁 𑀦𑀧𑁆𑀨𑀭𑀼𑀲𑀁 𑀅𑀯𑁄𑀘𑀁𑁇_  


(98.)

109\. _𑀲𑀘𑁆𑀘𑁂 𑀞𑀺𑀢𑀸 𑀫𑁄𑀲𑀯𑀚𑁆𑀚𑀁 𑀧𑀳𑀸𑀬,_  
_𑀤𑀸𑀦𑁂 𑀭𑀢𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀢𑁆𑀢𑀪𑀸𑀯𑀸;_  
_𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀸𑀦𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀤𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀅𑀤𑀸𑀲𑀺𑀁𑁇_  


(99.)

110\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀫𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(100.)

111\. _𑀅𑀓𑁆𑀔𑀸𑀫𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀬𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

112\. 𑀧𑀞𑀫𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇
