# 𑁧𑁇𑁧𑁇𑁧𑁨 𑀤𑀼𑀢𑀺𑀬𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀧𑀻𑀞𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(101.)

113. _“𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀣𑀫𑁆𑀪𑀁 𑀭𑀼𑀘𑀺𑀭𑀁 𑀧𑀪𑀲𑁆𑀲𑀭𑀁,_  
_𑀯𑀺𑀫𑀸𑀦𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀅𑀦𑁂𑀓𑀘𑀺𑀢𑁆𑀢𑀁;_  
_𑀢𑀢𑁆𑀣𑀘𑁆𑀙𑀲𑀺 𑀤𑁂𑀯𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸 𑀇𑀤𑁆𑀥𑀺 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀫𑀸𑀦𑀸;_  
_𑀇𑀫𑀸 𑀘 𑀢𑁂 𑀅𑀘𑁆𑀙𑀭𑀸𑀬𑁄 𑀲𑀫𑀦𑁆𑀢𑀢𑁄,_  
_𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀫𑁄𑀤𑀬𑀦𑁆𑀢𑀺 𑀘𑁇_  


(102.)

114. _𑀤𑁂𑀯𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑀸𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(103.)

115. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(104.)

116. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑀁;_  
_𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑀁,_  
_𑀮𑁄𑀓𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀺𑀲𑁆𑀲𑀁𑁇_  


(105.)

117. _𑀅𑀫𑀚𑁆𑀚𑀧𑀸 𑀦𑁄 𑀘 𑀫𑀼𑀲𑀸 𑀅𑀪𑀸𑀡𑀺𑀁,_  
_𑀲𑀓𑁂𑀦 𑀲𑀸𑀫𑀺𑀦𑀸 𑀅𑀳𑁄𑀲𑀺𑀁 𑀢𑀼𑀝𑁆𑀞𑀸;_  
_𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀸𑀦𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀤𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀅𑀤𑀸𑀲𑀺𑀁𑁇_  


(106.)

118. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀫𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(107.)

119. _𑀅𑀓𑁆𑀔𑀸𑀫𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀬𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

120. 𑀤𑀼𑀢𑀺𑀬𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁𑁇
