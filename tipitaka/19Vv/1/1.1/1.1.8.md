# 𑁧𑁇𑁧𑁇𑁮 𑀢𑀢𑀺𑀬𑀦𑀸𑀯𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀺𑀫𑀸𑀦

### 𑀧𑀻𑀞𑀯𑀕𑁆𑀕

### 𑀢𑀢𑀺𑀬𑀦𑀸𑀯𑀸𑀯𑀺𑀫𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(63.)

71\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀘𑁆𑀙𑀤𑀦𑀁 𑀦𑀸𑀯𑀁,_  
_𑀦𑀸𑀭𑀺 𑀆𑀭𑀼𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀲𑀺;_  
_𑀑𑀕𑀸𑀳𑀲𑀺 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁,_  
_𑀧𑀤𑁆𑀫𑀁 𑀙𑀺𑀦𑁆𑀤𑀲𑀺 𑀧𑀸𑀡𑀺𑀦𑀸𑁇_  


(64.)

72\. _𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀦𑀺𑀯𑁂𑀲𑀸 𑀢𑁂,_  
_𑀯𑀺𑀪𑀢𑁆𑀢𑀸 𑀪𑀸𑀕𑀲𑁄 𑀫𑀺𑀢𑀸;_  
_𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀫𑀸𑀦𑀸 𑀆𑀪𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑀸𑁇_  


(65.)

73\. _𑀓𑁂𑀦 𑀢𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀓𑁂𑀦 𑀢𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(66.)

74\. _𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀤𑁂𑀯𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀓𑀺𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_  


(67.)

75\. _𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(68.)

76\. _“𑀅𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑀺𑀢𑁂 𑀓𑀺𑀮𑀦𑁆𑀢𑁂,_  
_𑀉𑀝𑁆𑀞𑀸𑀬 𑀧𑀸𑀢𑀼𑀁 𑀉𑀤𑀓𑀁 𑀅𑀤𑀸𑀲𑀺𑀁𑁇_  


(69.)

77\. _𑀬𑁄 𑀯𑁂 𑀓𑀺𑀮𑀦𑁆𑀢𑀸𑀦 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑀸𑀦𑀁,_  
_𑀉𑀝𑁆𑀞𑀸𑀬 𑀧𑀸𑀢𑀼𑀁 𑀉𑀤𑀓𑀁 𑀤𑀤𑀸𑀢𑀺;_  
_𑀲𑀻𑀢𑁄𑀤𑀓𑀸 𑀢𑀲𑁆𑀲 𑀪𑀯𑀦𑁆𑀢𑀺 𑀦𑀚𑁆𑀚𑁄,_  
_𑀧𑀳𑀽𑀢𑀫𑀮𑁆𑀬𑀸 𑀩𑀳𑀼𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑁇_  


(70.)

78\. _𑀢𑀁 𑀆𑀧𑀕𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸,_  
_𑀲𑀻𑀢𑁄𑀤𑀓𑀸 𑀯𑀸𑀮𑀼𑀓𑀲𑀦𑁆𑀣𑀢𑀸 𑀦𑀤𑀻;_  
_𑀅𑀫𑁆𑀩𑀸 𑀘 𑀲𑀸𑀮𑀸 𑀢𑀺𑀮𑀓𑀸 𑀘 𑀚𑀫𑁆𑀩𑀼𑀬𑁄,_  
_𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀸 𑀧𑀸𑀝𑀮𑀺𑀬𑁄 𑀘 𑀨𑀼𑀮𑁆𑀮𑀸𑁇_  


(71.)

79\. _𑀢𑀁 𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑁂𑀳𑀺 𑀉𑀧𑁂𑀢𑀭𑀽𑀧𑀁,_  
_𑀯𑀺𑀫𑀸𑀦𑀲𑁂𑀝𑁆𑀞𑀁 𑀪𑀼𑀲 𑀲𑁄𑀪𑀫𑀸𑀦𑀁;_  
_𑀢𑀲𑁆𑀲𑀻𑀥 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄,_  
_𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀢𑀸 𑀮𑀪𑀦𑁆𑀢𑀺𑁇_  


(72.)

80\. _𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀦𑀺𑀯𑁂𑀲𑀸 𑀫𑁂,_  
_𑀯𑀺𑀪𑀢𑁆𑀢𑀸 𑀪𑀸𑀕𑀲𑁄 𑀫𑀺𑀢𑀸;_  
_𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀫𑀸𑀦𑀸 𑀆𑀪𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑀸𑁇_  


(73.)

81\. _𑀢𑁂𑀦 𑀫𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀯𑀡𑁆𑀡𑁄,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀇𑀥 𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀫𑁂 𑀪𑁄𑀕𑀸,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(74.)

82\. _𑀅𑀓𑁆𑀔𑀸𑀫𑀺 𑀢𑁂 𑀩𑀼𑀤𑁆𑀥 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀸 𑀬𑀫𑀓𑀸𑀲𑀺 𑀧𑀼𑀜𑁆𑀜𑀁;_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀏𑀯𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑀸,_  
_𑀯𑀡𑁆𑀡𑁄 𑀘 𑀫𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀺;_  
_𑀏𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀨𑀮𑀁 𑀫𑀫𑁂𑀤𑀁,_  
_𑀅𑀢𑁆𑀣𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑁄 𑀉𑀤𑀓𑀁 𑀅𑀧𑀸𑀬𑀻”𑀢𑀺𑁇_  


---

83\. 𑀢𑀢𑀺𑀬𑀦𑀸𑀯𑀸𑀯𑀺𑀫𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
