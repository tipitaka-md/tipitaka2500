# 𑁧𑁇𑁩𑁇𑁩 𑀪𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀪𑀬𑀲𑀼𑀢𑁆𑀢

(23.)

142\. “‘𑀪𑀬𑀦𑁆’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁; ‘𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁; ‘𑀭𑁄𑀕𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁; ‘𑀕𑀡𑁆𑀟𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁; ‘𑀲𑀗𑁆𑀕𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁; ‘𑀧𑀗𑁆𑀓𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇

143\. 𑀓𑀲𑁆𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀪𑀬𑀦𑁆’𑀢𑀺 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁? 𑀓𑀸𑀫𑀭𑀸𑀕𑀭𑀢𑁆𑀢𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀺𑀩𑀤𑁆𑀥𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀧𑀺 𑀪𑀬𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀧𑀺 𑀪𑀬𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 ‘𑀪𑀬𑀦𑁆’𑀢𑀺 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀓𑀲𑁆𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀭𑁄𑀕𑁄𑀢𑀺…  𑀕𑀡𑁆𑀟𑁄𑀢𑀺…  𑀲𑀗𑁆𑀕𑁄𑀢𑀺…  𑀧𑀗𑁆𑀓𑁄𑀢𑀺 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁? 𑀓𑀸𑀫𑀭𑀸𑀕𑀭𑀢𑁆𑀢𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑀺𑀩𑀤𑁆𑀥𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀧𑀺 𑀧𑀗𑁆𑀓𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀧𑀺 𑀧𑀗𑁆𑀓𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 ‘𑀧𑀗𑁆𑀓𑁄’𑀢𑀺 𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀦𑁆𑀢𑀺𑁇

144\. _𑀪𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀭𑁄𑀕𑁄 𑀕𑀡𑁆𑀟𑁄,_  
_𑀲𑀗𑁆𑀕𑁄 𑀧𑀗𑁆𑀓𑁄 𑀘 𑀉𑀪𑀬𑀁;_  
_𑀏𑀢𑁂 𑀓𑀸𑀫𑀸 𑀧𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀬𑀢𑁆𑀣 𑀲𑀢𑁆𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄𑁇_  


145\. _𑀉𑀧𑀸𑀤𑀸𑀦𑁂 𑀪𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀲𑀫𑁆𑀪𑀯𑁂;_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀲𑀗𑁆𑀔𑀬𑁂𑁇_  


146\. _𑀢𑁂 𑀔𑁂𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀲𑀼𑀔𑀺𑀦𑁄,_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑀯𑁂𑀭𑀪𑀬𑀸𑀢𑀻𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑀘𑁆𑀘𑀕𑀼𑀦𑁆”𑀢𑀺𑁇_  


---

147\. 𑀢𑀢𑀺𑀬𑀁𑁇
