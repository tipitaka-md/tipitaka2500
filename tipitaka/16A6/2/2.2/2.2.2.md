# 𑁨𑁇𑁨𑁇𑁨 𑀅𑀭𑀳𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀤𑁂𑀯𑀢𑀸𑀯𑀕𑁆𑀕

* 𑀅𑀭𑀳𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(66.)

621\. “𑀙, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀣𑀺𑀦𑀁, 𑀫𑀺𑀤𑁆𑀥𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁, 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀁, 𑀧𑀫𑀸𑀤𑀁—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇

622\. 𑀙, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀣𑀺𑀦𑀁, 𑀫𑀺𑀤𑁆𑀥𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁, 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀁, 𑀧𑀫𑀸𑀤𑀁—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

---
