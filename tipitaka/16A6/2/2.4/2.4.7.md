# 𑁨𑁇𑁪𑁇𑁭 𑀅𑀪𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀻𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀪𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

(91.)

701. “𑀙, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀁, 𑀅𑀧𑀸𑀬𑀕𑀫𑀦𑀻𑀬𑀁 𑀭𑀸𑀕𑀁, 𑀅𑀧𑀸𑀬𑀕𑀫𑀦𑀻𑀬𑀁 𑀤𑁄𑀲𑀁, 𑀅𑀧𑀸𑀬𑀕𑀫𑀦𑀻𑀬𑀁 𑀫𑁄𑀳𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
