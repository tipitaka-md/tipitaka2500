# 𑁨𑁇𑁪𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀻𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(93.)

705\. “𑀙𑀬𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀙? 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀜𑁆𑀘𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀜𑁆𑀘𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀲𑀼𑀔𑀢𑁄 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀆𑀦𑀦𑁆𑀢𑀭𑀺𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑁄𑀢𑀽𑀳𑀮𑀫𑀗𑁆𑀕𑀮𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀁 𑀕𑀯𑁂𑀲𑀺𑀢𑀼𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀻”𑀢𑀺𑁇
