# 𑁨𑁇𑁪𑁇𑁧𑁧 𑀘𑀢𑀼𑀢𑁆𑀣𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀻𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(95.)

709\. “𑀙𑀬𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀙? 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀭𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀼𑀁, 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀲𑀬𑀗𑁆𑀓𑀸𑀭𑀜𑁆𑀘 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀜𑁆𑀘 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀳𑁂𑀢𑀼 𑀘 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀻”𑀢𑀺𑁇
