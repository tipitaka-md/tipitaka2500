# 𑁨𑁇𑁮𑁇𑁧

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀭𑀸𑀕𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(140.)

775\. “𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀙 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁, 𑀲𑀯𑀦𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁, 𑀲𑀺𑀓𑁆𑀔𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁, 𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁𑁇 𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀫𑁂 𑀙 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
