# 𑁨𑁇𑁬𑁇𑁬 𑀅𑀲𑁆𑀲𑀸𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀢𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀲𑁆𑀲𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(112.)

752\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀲𑁆𑀲𑀸𑀤𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀲𑁆𑀲𑀸𑀤𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀳𑀸𑀦𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸, 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀳𑀸𑀦𑀸𑀬 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀳𑀸𑀦𑀸𑀬 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
