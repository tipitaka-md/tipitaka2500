# 𑁨𑁇𑁬𑁇𑁭 𑀅𑀭𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀢𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀭𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(113.)

754\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀭𑀢𑀺, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸, 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀭𑀢𑀺𑀬𑀸 𑀧𑀳𑀸𑀦𑀸𑀬 𑀫𑀼𑀤𑀺𑀢𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸, 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀬 𑀧𑀳𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
