# 𑁨𑁇𑁫𑁇𑁨 𑀆𑀦𑀺𑀲𑀁𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀦𑀺𑀲𑀁𑀲𑀯𑀕𑁆𑀕

### 𑀆𑀦𑀺𑀲𑀁𑀲𑀲𑀼𑀢𑁆𑀢

(97.)

716\. “𑀙𑀬𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀦𑀺𑀬𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀥𑀫𑁆𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀓𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀳𑁂𑀢𑀼 𑀘𑀲𑁆𑀲 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄, 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇
