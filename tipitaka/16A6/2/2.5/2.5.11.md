# 𑁨𑁇𑁫𑁇𑁧𑁧 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀦𑀺𑀲𑀁𑀲𑀯𑀕𑁆𑀕

### 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(106.)

736\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀢𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸? 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸, 𑀪𑀯𑀢𑀡𑁆𑀳𑀸, 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸—  𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀫𑀸𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸? 𑀫𑀸𑀦𑁄, 𑀑𑀫𑀸𑀦𑁄, 𑀅𑀢𑀺𑀫𑀸𑀦𑁄—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀫𑀸𑀦𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀇𑀫𑁂 𑀘 𑀢𑀬𑁄 𑀫𑀸𑀦𑀸; 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀢𑀡𑁆𑀳𑀁, 𑀯𑀺𑀯𑀢𑁆𑀢𑀬𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀫𑁆𑀫𑀸 𑀫𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀸 𑀅𑀦𑁆𑀢𑀫𑀓𑀸𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

739\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



740\. _𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄,_  
_𑀅𑀦𑀺𑀘𑁆𑀘𑀤𑀼𑀓𑁆𑀔𑀅𑀦𑀢𑁆𑀢𑀢𑁄;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀯𑀢𑁆𑀣𑀺,_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀲𑀺 𑀅𑀢𑀫𑁆𑀫𑀬𑁄;_  
_𑀪𑀯𑀸 𑀢𑀡𑁆𑀳𑀸𑀬𑁂𑀓𑀸 𑀤𑀲𑀸𑀢𑀺𑁇_
