# 𑁨𑁇𑁫𑁇𑁧 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀆𑀦𑀺𑀲𑀁𑀲𑀯𑀕𑁆𑀕

* 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀲𑀼𑀢𑁆𑀢

(96.)

714\. “𑀙𑀦𑁆𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀙𑀦𑁆𑀦𑀁? 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀤𑁂𑀲𑁂𑀢𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀅𑀭𑀺𑀬𑀸𑀬𑀢𑀦𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑀺 𑀤𑀼𑀮𑁆𑀮𑀪𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀯𑁂𑀓𑀮𑁆𑀮𑀢𑀸 𑀤𑀼𑀮𑁆𑀮𑀪𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀅𑀚𑀍𑀅𑀢𑀸 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑀢𑀸 𑀤𑀼𑀮𑁆𑀮𑀪𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀙𑀦𑁆𑀤𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇
