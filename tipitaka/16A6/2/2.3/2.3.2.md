# 𑁨𑁇𑁩𑁇𑁨 𑀅𑀭𑀳𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀭𑀳𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀭𑀳𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(76.)

656. “𑀙, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀫𑀸𑀦𑀁, 𑀑𑀫𑀸𑀦𑀁, 𑀅𑀢𑀺𑀫𑀸𑀦𑀁, 𑀅𑀥𑀺𑀫𑀸𑀦𑀁, 𑀣𑀫𑁆𑀪𑀁, 𑀅𑀢𑀺𑀦𑀺𑀧𑀸𑀢𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇

657. 𑀙, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀙? 𑀫𑀸𑀦𑀁, 𑀑𑀫𑀸𑀦𑀁, 𑀅𑀢𑀺𑀫𑀸𑀦𑀁, 𑀅𑀥𑀺𑀫𑀸𑀦𑀁, 𑀣𑀫𑁆𑀪𑀁, 𑀅𑀢𑀺𑀦𑀺𑀧𑀸𑀢𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
