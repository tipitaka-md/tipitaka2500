# 𑁪 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

(241.)

535\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑁂 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑁂𑀦 𑀬𑀫𑀓𑀲𑀸𑀮𑀸𑀦𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀫𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀔𑀼𑀤𑁆𑀤𑀓𑀦𑀕𑀭𑀓𑁂 𑀉𑀚𑁆𑀚𑀗𑁆𑀕𑀮𑀦𑀕𑀭𑀓𑁂 𑀲𑀸𑀔𑀸𑀦𑀕𑀭𑀓𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑁇 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀸𑀦𑀺 𑀫𑀳𑀸𑀦𑀕𑀭𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀫𑁆𑀧𑀸, 𑀭𑀸𑀚𑀕𑀳𑀁, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺, 𑀲𑀸𑀓𑁂𑀢𑀁, 𑀓𑁄𑀲𑀫𑁆𑀩𑀻, 𑀩𑀸𑀭𑀸𑀡𑀲𑀻; 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀼𑁇 𑀏𑀢𑁆𑀣 𑀩𑀳𑀽 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀢𑀣𑀸𑀕𑀢𑁂 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸, 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀧𑀽𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

(242.)
