# 𑁧𑁦𑁇𑁧 𑀧𑀸𑀬𑀸𑀲𑀺𑀭𑀸𑀚𑀜𑁆𑀜𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀭𑀸𑀚𑀜𑁆𑀜𑀯𑀢𑁆𑀣𑀼

(407.)

1091\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑁆𑀲 𑀭𑀸𑀚𑀜𑁆𑀜𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸—  “𑀲𑀫𑀡𑁄 𑀔𑀮𑀼 𑀪𑁄 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺 𑀲𑁂𑀢𑀩𑁆𑀬𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑁂𑀢𑀩𑁆𑀬𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀲𑁂𑀢𑀩𑁆𑀬𑀁 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑁂𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀯𑀦𑁆𑀢𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀓𑀣𑀻 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀝𑀺𑀪𑀸𑀦𑁄 𑀯𑀼𑀤𑁆𑀥𑁄 𑀘𑁂𑀯 𑀅𑀭𑀳𑀸 𑀘𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀲𑁂𑀢𑀩𑁆𑀬𑀸𑀬 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀖𑀲𑀗𑁆𑀖𑀻 𑀕𑀡𑀻𑀪𑀽𑀢𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀦𑀫𑀼𑀔𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀬𑁂𑀦 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑀁𑁇

(408.)

1092\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀉𑀧𑀭𑀺𑀧𑀸𑀲𑀸𑀤𑁂 𑀤𑀺𑀯𑀸𑀲𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂 𑀲𑁂𑀢𑀩𑁆𑀬𑀸𑀬 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀖𑀲𑀗𑁆𑀖𑀻 𑀕𑀡𑀻𑀪𑀽𑀢𑁂 𑀉𑀢𑁆𑀢𑀭𑁂𑀦𑀫𑀼𑀔𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀬𑁂𑀦 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸 𑀔𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀔𑀢𑁆𑀢𑁂, 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀲𑁂𑀢𑀩𑁆𑀬𑀸𑀬 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀖𑀲𑀗𑁆𑀖𑀻 𑀕𑀡𑀻𑀪𑀽𑀢𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀦𑀫𑀼𑀔𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀬𑁂𑀦 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑀦𑁆”𑀢𑀺?

1093\. “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀪𑁄, 𑀲𑀫𑀡𑁄 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄, 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺 𑀲𑁂𑀢𑀩𑁆𑀬𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑁂𑀢𑀩𑁆𑀬𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀲𑁂𑀢𑀩𑁆𑀬𑀁 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑁂𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀯𑀦𑁆𑀢𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀓𑀣𑀻 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀝𑀺𑀪𑀸𑀦𑁄 𑀯𑀼𑀤𑁆𑀥𑁄 𑀘𑁂𑀯 𑀅𑀭𑀳𑀸 𑀘𑀸’𑀢𑀺𑁇 𑀢𑀫𑁂𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄 𑀔𑀢𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀺—  ‘𑀧𑀸𑀬𑀸𑀲𑀺, 𑀪𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀏𑀯𑀫𑀸𑀳—  𑀆𑀕𑀫𑁂𑀦𑁆𑀢𑀼 𑀓𑀺𑀭 𑀪𑀯𑀦𑁆𑀢𑁄, 𑀧𑀸𑀬𑀸𑀲𑀺𑀧𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀲𑀫𑀡𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀧𑀼𑀭𑀸 𑀲𑀫𑀡𑁄 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂 𑀩𑀸𑀮𑁂 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁂 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’𑀢𑀺𑁇 𑀦𑀢𑁆𑀣𑀺 𑀳𑀺, 𑀪𑁄 𑀔𑀢𑁆𑀢𑁂, 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀔𑀢𑁆𑀢𑀸 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑁆𑀲 𑀭𑀸𑀚𑀜𑁆𑀜𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀸𑀬𑀸𑀲𑀺, 𑀪𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀆𑀕𑀫𑁂𑀦𑁆𑀢𑀼 𑀓𑀺𑀭 𑀪𑀯𑀦𑁆𑀢𑁄, 𑀧𑀸𑀬𑀸𑀲𑀺𑀧𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀲𑀫𑀡𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

(409.)

1094\. 𑀅𑀣 𑀔𑁄 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑁂𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀬𑁂𑀦 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀲𑁂𑀢𑀩𑁆𑀬𑀓𑀸𑀧𑀺 𑀔𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼; 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼; 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀲𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇
