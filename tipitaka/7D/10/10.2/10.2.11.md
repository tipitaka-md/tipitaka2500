# 𑁧𑁦𑁇𑁨𑁇𑁧𑁧 𑀤𑁆𑀯𑁂𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁅𑀧𑀫𑀸

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀢𑁆𑀣𑀺𑀓𑀯𑀸𑀤

* 𑀤𑁆𑀯𑁂𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁅𑀧𑀫𑀸

(430.)

1121\. “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀉𑀧𑀫𑀸𑀬 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀫𑀳𑀸𑀲𑀓𑀝𑀲𑀢𑁆𑀣𑁄 𑀲𑀓𑀝𑀲𑀳𑀲𑁆𑀲𑀁 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸 𑀚𑀦𑀧𑀤𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀚𑀦𑀧𑀤𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀲𑁄 𑀬𑁂𑀦 𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀺, 𑀔𑀺𑀧𑁆𑀧𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀢𑀺 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑁄𑀤𑀓𑀁 𑀳𑀭𑀺𑀢𑀓𑀧𑀡𑁆𑀡𑀁𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀢𑁆𑀣𑁂 𑀤𑁆𑀯𑁂 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀏𑀓𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀁, 𑀏𑀓𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀫𑀳𑀸𑀲𑀓𑀝𑀲𑀢𑁆𑀣𑁄 𑀲𑀓𑀝𑀲𑀳𑀲𑁆𑀲𑀁; 𑀢𑁂 𑀫𑀬𑀁 𑀬𑁂𑀦 𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀸𑀫, 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀢𑀺 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑁄𑀤𑀓𑀁 𑀳𑀭𑀺𑀢𑀓𑀧𑀡𑁆𑀡𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀇𑀫𑀁 𑀲𑀢𑁆𑀣𑀁 𑀤𑁆𑀯𑀺𑀥𑀸 𑀯𑀺𑀪𑀚𑁂𑀬𑁆𑀬𑀸𑀫—  𑀏𑀓𑀢𑁄 𑀧𑀜𑁆𑀘 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀏𑀓𑀢𑁄 𑀧𑀜𑁆𑀘 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀻’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀲𑀢𑁆𑀣𑀁 𑀤𑁆𑀯𑀺𑀥𑀸 𑀯𑀺𑀪𑀚𑀺𑀁𑀲𑀼 𑀏𑀓𑀢𑁄 𑀧𑀜𑁆𑀘 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺, 𑀏𑀓𑀢𑁄 𑀧𑀜𑁆𑀘 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺𑁇 𑀏𑀓𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄 𑀩𑀳𑀼𑀁 𑀢𑀺𑀡𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀉𑀤𑀓𑀜𑁆𑀘 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀧𑀬𑀸𑀧𑁂𑀲𑀺𑁇 𑀤𑁆𑀯𑀻𑀳𑀢𑀻𑀳𑀧𑀬𑀸𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀲𑁄 𑀲𑀢𑁆𑀣𑁄 𑀅𑀤𑁆𑀤𑀲 𑀧𑀼𑀭𑀺𑀲𑀁 𑀓𑀸𑀍𑀅𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑁆𑀔𑀁 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀓𑀮𑀸𑀧𑀁 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀁 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀁 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀁 𑀓𑀤𑁆𑀤𑀫𑀫𑀓𑁆𑀔𑀺𑀢𑁂𑀳𑀺 𑀘𑀓𑁆𑀓𑁂𑀳𑀺 𑀪𑀤𑁆𑀭𑁂𑀦 𑀭𑀣𑁂𑀦 𑀧𑀝𑀺𑀧𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀓𑀼𑀢𑁄, 𑀪𑁄, 𑀆𑀕𑀘𑁆𑀙𑀲𑀻’𑀢𑀺? ‘𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀚𑀦𑀧𑀤𑀸’𑀢𑀺𑁇 ‘𑀓𑀼𑀳𑀺𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀻’𑀢𑀺? ‘𑀅𑀫𑀼𑀓𑀁 𑀦𑀸𑀫 𑀚𑀦𑀧𑀤𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀘𑁆𑀘𑀺, 𑀪𑁄, 𑀧𑀼𑀭𑀢𑁄 𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀯𑀼𑀝𑁆𑀞𑁄’𑀢𑀺? ‘𑀏𑀯𑀁, 𑀪𑁄, 𑀧𑀼𑀭𑀢𑁄 𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀯𑀼𑀝𑁆𑀞𑁄, 𑀆𑀲𑀺𑀢𑁆𑀢𑁄𑀤𑀓𑀸𑀦𑀺 𑀯𑀝𑀼𑀫𑀸𑀦𑀺, 𑀩𑀳𑀼 𑀢𑀺𑀡𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀉𑀤𑀓𑀜𑁆𑀘𑁇 𑀙𑀟𑁆𑀟𑁂𑀣, 𑀪𑁄, 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀺, 𑀮𑀳𑀼𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀓𑀝𑁂𑀳𑀺 𑀲𑀻𑀖𑀁 𑀲𑀻𑀖𑀁 𑀕𑀘𑁆𑀙𑀣, 𑀫𑀸 𑀬𑁄𑀕𑁆𑀕𑀸𑀦𑀺 𑀓𑀺𑀮𑀫𑀺𑀢𑁆𑀣𑀸’𑀢𑀺𑁇

1122\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄 𑀲𑀢𑁆𑀣𑀺𑀓𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀅𑀬𑀁, 𑀪𑁄, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀧𑀼𑀭𑀢𑁄 𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀯𑀼𑀝𑁆𑀞𑁄, 𑀆𑀲𑀺𑀢𑁆𑀢𑁄𑀤𑀓𑀸𑀦𑀺 𑀯𑀝𑀼𑀫𑀸𑀦𑀺, 𑀩𑀳𑀼 𑀢𑀺𑀡𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀉𑀤𑀓𑀜𑁆𑀘𑁇 𑀙𑀟𑁆𑀟𑁂𑀣, 𑀪𑁄, 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀺, 𑀮𑀳𑀼𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀓𑀝𑁂𑀳𑀺 𑀲𑀻𑀖𑀁 𑀲𑀻𑀖𑀁 𑀕𑀘𑁆𑀙𑀣, 𑀫𑀸 𑀬𑁄𑀕𑁆𑀕𑀸𑀦𑀺 𑀓𑀺𑀮𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 𑀙𑀟𑁆𑀟𑁂𑀣, 𑀪𑁄, 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀺, 𑀮𑀳𑀼𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀓𑀝𑁂𑀳𑀺 𑀲𑀢𑁆𑀣𑀁 𑀧𑀬𑀸𑀧𑁂𑀣𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑁄’𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀺 𑀮𑀳𑀼𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀓𑀝𑁂𑀳𑀺 𑀲𑀢𑁆𑀣𑀁 𑀧𑀬𑀸𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑁂 𑀧𑀞𑀫𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂 𑀦 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀢𑀺𑀡𑀁 𑀯𑀸 𑀓𑀝𑁆𑀞𑀁 𑀯𑀸 𑀉𑀤𑀓𑀁 𑀯𑀸𑁇 𑀤𑀼𑀢𑀺𑀬𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀢𑀢𑀺𑀬𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀧𑀜𑁆𑀘𑀫𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀙𑀝𑁆𑀞𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀲𑀢𑁆𑀢𑀫𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂 𑀦 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀢𑀺𑀡𑀁 𑀯𑀸 𑀓𑀝𑁆𑀞𑀁 𑀯𑀸 𑀉𑀤𑀓𑀁 𑀯𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀯 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀘 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑁆𑀣𑁂 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀸 𑀧𑀲𑀽 𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀲𑁄 𑀬𑀓𑁆𑀔𑁄 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀪𑀓𑁆𑀔𑁂𑀲𑀺𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀸𑀦𑁂𑀯 𑀲𑁂𑀲𑀸𑀦𑀺𑁇

1123\. 𑀬𑀤𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄—  ‘𑀩𑀳𑀼𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀔𑁄, 𑀪𑁄, 𑀤𑀸𑀦𑀺 𑀲𑁄 𑀲𑀢𑁆𑀣𑁄’𑀢𑀺 𑀩𑀳𑀼𑀁 𑀢𑀺𑀡𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀉𑀤𑀓𑀜𑁆𑀘 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀧𑀬𑀸𑀧𑁂𑀲𑀺𑁇 𑀤𑁆𑀯𑀻𑀳𑀢𑀻𑀳𑀧𑀬𑀸𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀲𑁄 𑀲𑀢𑁆𑀣𑁄 𑀅𑀤𑁆𑀤𑀲 𑀧𑀼𑀭𑀺𑀲𑀁 𑀓𑀸𑀍𑀅𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑁆𑀔𑀁 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀓𑀮𑀸𑀧𑀁 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀁 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀁 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀁 𑀓𑀤𑁆𑀤𑀫𑀫𑀓𑁆𑀔𑀺𑀢𑁂𑀳𑀺 𑀘𑀓𑁆𑀓𑁂𑀳𑀺 𑀪𑀤𑁆𑀭𑁂𑀦 𑀭𑀣𑁂𑀦 𑀧𑀝𑀺𑀧𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀓𑀼𑀢𑁄, 𑀪𑁄, 𑀆𑀕𑀘𑁆𑀙𑀲𑀻’𑀢𑀺? ‘𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀚𑀦𑀧𑀤𑀸’𑀢𑀺𑁇 ‘𑀓𑀼𑀳𑀺𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀻’𑀢𑀺? ‘𑀅𑀫𑀼𑀓𑀁 𑀦𑀸𑀫 𑀚𑀦𑀧𑀤𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀘𑁆𑀘𑀺, 𑀪𑁄, 𑀧𑀼𑀭𑀢𑁄 𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀯𑀼𑀝𑁆𑀞𑁄’𑀢𑀺? ‘𑀏𑀯𑀁, 𑀪𑁄, 𑀧𑀼𑀭𑀢𑁄 𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀯𑀼𑀝𑁆𑀞𑁄𑁇 𑀆𑀲𑀺𑀢𑁆𑀢𑁄𑀤𑀓𑀸𑀦𑀺 𑀯𑀝𑀼𑀫𑀸𑀦𑀺, 𑀩𑀳𑀼 𑀢𑀺𑀡𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀉𑀤𑀓𑀜𑁆𑀘𑁇 𑀙𑀟𑁆𑀟𑁂𑀣, 𑀪𑁄, 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀺, 𑀮𑀳𑀼𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀓𑀝𑁂𑀳𑀺 𑀲𑀻𑀖𑀁 𑀲𑀻𑀖𑀁 𑀕𑀘𑁆𑀙𑀣, 𑀫𑀸 𑀬𑁄𑀕𑁆𑀕𑀸𑀦𑀺 𑀓𑀺𑀮𑀫𑀺𑀢𑁆𑀣𑀸’𑀢𑀺𑁇

1124\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄 𑀲𑀢𑁆𑀣𑀺𑀓𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀅𑀬𑀁, 𑀪𑁄, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀧𑀼𑀭𑀢𑁄 𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀯𑀼𑀝𑁆𑀞𑁄, 𑀆𑀲𑀺𑀢𑁆𑀢𑁄𑀤𑀓𑀸𑀦𑀺 𑀯𑀝𑀼𑀫𑀸𑀦𑀺, 𑀩𑀳𑀼 𑀢𑀺𑀡𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀉𑀤𑀓𑀜𑁆𑀘𑁇 𑀙𑀟𑁆𑀟𑁂𑀣, 𑀪𑁄, 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀺, 𑀮𑀳𑀼𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀓𑀝𑁂𑀳𑀺 𑀲𑀻𑀖𑀁 𑀲𑀻𑀖𑀁 𑀕𑀘𑁆𑀙𑀣; 𑀫𑀸 𑀬𑁄𑀕𑁆𑀕𑀸𑀦𑀺 𑀓𑀺𑀮𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 𑀅𑀬𑀁 𑀪𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦𑁂𑀯 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀫𑀺𑀢𑁆𑀢𑁄, 𑀦 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑁄, 𑀓𑀣𑀁 𑀫𑀬𑀁 𑀇𑀫𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸𑀬 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀦 𑀯𑁄 𑀙𑀟𑁆𑀟𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀺, 𑀬𑀣𑀸𑀪𑀢𑁂𑀦 𑀪𑀡𑁆𑀟𑁂𑀦 𑀲𑀢𑁆𑀣𑀁 𑀧𑀬𑀸𑀧𑁂𑀣𑁇 𑀦 𑀦𑁄 𑀧𑀼𑀭𑀸𑀡𑀁 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑁄’𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀪𑀢𑁂𑀦 𑀪𑀡𑁆𑀟𑁂𑀦 𑀲𑀢𑁆𑀣𑀁 𑀧𑀬𑀸𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑁂 𑀧𑀞𑀫𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂 𑀦 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀢𑀺𑀡𑀁 𑀯𑀸 𑀓𑀝𑁆𑀞𑀁 𑀯𑀸 𑀉𑀤𑀓𑀁 𑀯𑀸𑁇 𑀤𑀼𑀢𑀺𑀬𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀢𑀢𑀺𑀬𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀧𑀜𑁆𑀘𑀫𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀙𑀝𑁆𑀞𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂…  𑀲𑀢𑁆𑀢𑀫𑁂𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂 𑀦 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀢𑀺𑀡𑀁 𑀯𑀸 𑀓𑀝𑁆𑀞𑀁 𑀯𑀸 𑀉𑀤𑀓𑀁 𑀯𑀸𑁇 𑀢𑀜𑁆𑀘 𑀲𑀢𑁆𑀣𑀁 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑀁𑁇 𑀬𑁂 𑀘 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑁆𑀣𑁂𑀧𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀸 𑀧𑀲𑀽 𑀯𑀸, 𑀢𑁂𑀲𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀺𑀓𑀸𑀦𑁂𑀯 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀢𑁂𑀦 𑀬𑀓𑁆𑀔𑁂𑀦 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀦 𑀪𑀓𑁆𑀔𑀺𑀢𑀸𑀦𑀁𑁇

1125\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄 𑀲𑀢𑁆𑀣𑀺𑀓𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀪𑁄, 𑀲𑀢𑁆𑀣𑁄 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑁄, 𑀬𑀣𑀸 𑀢𑀁 𑀢𑁂𑀦 𑀩𑀸𑀮𑁂𑀦 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁂𑀦 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑁂𑀦𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀬𑀸𑀦𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀢𑁆𑀣𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀭𑀸𑀦𑀺 𑀧𑀡𑀺𑀬𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀙𑀟𑁆𑀟𑁂𑀢𑁆𑀯𑀸, 𑀬𑀸𑀦𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑁆𑀣𑁂 𑀫𑀳𑀸𑀲𑀸𑀭𑀸𑀦𑀺 𑀧𑀡𑀺𑀬𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀆𑀤𑀺𑀬𑀣𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑁄’𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑀸𑀦𑀺 𑀲𑀓𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑁆𑀣𑁂 𑀅𑀧𑁆𑀧𑀲𑀸𑀭𑀸𑀦𑀺 𑀧𑀡𑀺𑀬𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀙𑀟𑁆𑀟𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑁆𑀣𑁂 𑀫𑀳𑀸𑀲𑀸𑀭𑀸𑀦𑀺 𑀧𑀡𑀺𑀬𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀢𑀁 𑀓𑀦𑁆𑀢𑀸𑀭𑀁 𑀦𑀺𑀢𑁆𑀣𑀭𑀺𑀁𑀲𑀼, 𑀬𑀣𑀸 𑀢𑀁 𑀧𑀡𑁆𑀟𑀺𑀢𑁂𑀦 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁂𑀦 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑁂𑀦𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀧𑀭𑀮𑁄𑀓𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑁄 𑀧𑀼𑀭𑀺𑀫𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄𑁇 𑀬𑁂𑀧𑀺 𑀢𑀯 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀲𑀤𑁆𑀥𑀸𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀧𑀺 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑁂 𑀲𑀢𑁆𑀣𑀺𑀓𑀸𑁇 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁; 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁𑁇 𑀫𑀸 𑀢𑁂 𑀅𑀳𑁄𑀲𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

1126\. “𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀅𑀣 𑀔𑁄 𑀦𑁂𑀯𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀇𑀤𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀭𑀸𑀚𑀸𑀧𑀺 𑀫𑀁 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀺𑀭𑁄𑀭𑀸𑀚𑀸𑀦𑁄𑀧𑀺—  ‘𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀻—  “𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁄”’𑀢𑀺𑁇 𑀲𑀘𑀸𑀳𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀇𑀤𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑁂 𑀯𑀢𑁆𑀢𑀸𑀭𑁄—  ‘𑀬𑀸𑀯 𑀩𑀸𑀮𑁄 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄, 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀕𑀸𑀳𑀻’𑀢𑀺𑁇 𑀓𑁄𑀧𑁂𑀦𑀧𑀺 𑀦𑀁 𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀫𑀓𑁆𑀔𑁂𑀦𑀧𑀺 𑀦𑀁 𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀮𑀸𑀲𑁂𑀦𑀧𑀺 𑀦𑀁 𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇
