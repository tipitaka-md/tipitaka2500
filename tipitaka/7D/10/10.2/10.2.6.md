# 𑁧𑁦𑁇𑁨𑁇𑁬 𑀕𑀩𑁆𑀪𑀺𑀦𑀻𑀉𑀧𑀫𑀸

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀢𑁆𑀣𑀺𑀓𑀯𑀸𑀤

* 𑀕𑀩𑁆𑀪𑀺𑀦𑀻𑀉𑀧𑀫𑀸

(420.)

1107\. “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀉𑀧𑀫𑀸𑀬 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀤𑀲𑀯𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑁂𑀲𑀺𑀓𑁄 𑀯𑀸 𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑁂𑀲𑀺𑀓𑁄 𑀯𑀸, 𑀏𑀓𑀸 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀉𑀧𑀯𑀺𑀚𑀜𑁆𑀜𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀫𑀸𑀢𑀼𑀲𑀧𑀢𑁆𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀬𑀫𑀺𑀤𑀁, 𑀪𑁄𑀢𑀺, 𑀥𑀦𑀁 𑀯𑀸 𑀥𑀜𑁆𑀜𑀁 𑀯𑀸 𑀭𑀚𑀢𑀁 𑀯𑀸 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀫𑀬𑁆𑀳𑀁; 𑀦𑀢𑁆𑀣𑀺 𑀢𑀼𑀬𑁆𑀳𑁂𑀢𑁆𑀣 𑀓𑀺𑀜𑁆𑀘𑀺𑁇 𑀧𑀺𑀢𑀼 𑀫𑁂, 𑀪𑁄𑀢𑀺, 𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀢𑀁 𑀫𑀸𑀡𑀯𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀆𑀕𑀫𑁂𑀳𑀺 𑀢𑀸𑀯, 𑀢𑀸𑀢, 𑀬𑀸𑀯 𑀯𑀺𑀚𑀸𑀬𑀸𑀫𑀺𑁇 𑀲𑀘𑁂 𑀓𑀼𑀫𑀸𑀭𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀺 𑀏𑀓𑀤𑁂𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺; 𑀲𑀘𑁂 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀸𑀧𑀺 𑀢𑁂 𑀑𑀧𑀪𑁄𑀕𑁆𑀕𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀫𑀸𑀢𑀼𑀲𑀧𑀢𑁆𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀬𑀫𑀺𑀤𑀁, 𑀪𑁄𑀢𑀺, 𑀥𑀦𑀁 𑀯𑀸 𑀥𑀜𑁆𑀜𑀁 𑀯𑀸 𑀭𑀚𑀢𑀁 𑀯𑀸 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀫𑀬𑁆𑀳𑀁; 𑀦𑀢𑁆𑀣𑀺 𑀢𑀼𑀬𑁆𑀳𑁂𑀢𑁆𑀣 𑀓𑀺𑀜𑁆𑀘𑀺𑁇 𑀧𑀺𑀢𑀼 𑀫𑁂, 𑀪𑁄𑀢𑀺, 𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀢𑀁 𑀫𑀸𑀡𑀯𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀆𑀕𑀫𑁂𑀳𑀺 𑀢𑀸𑀯, 𑀢𑀸𑀢, 𑀬𑀸𑀯 𑀯𑀺𑀚𑀸𑀬𑀸𑀫𑀺𑁇 𑀲𑀘𑁂 𑀓𑀼𑀫𑀸𑀭𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀺 𑀏𑀓𑀤𑁂𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺; 𑀲𑀘𑁂 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀸𑀧𑀺 𑀢𑁂 𑀑𑀧𑀪𑁄𑀕𑁆𑀕𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀫𑀸𑀢𑀼𑀲𑀧𑀢𑁆𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀬𑀫𑀺𑀤𑀁, 𑀪𑁄𑀢𑀺, 𑀥𑀦𑀁 𑀯𑀸 𑀥𑀜𑁆𑀜𑀁 𑀯𑀸 𑀭𑀚𑀢𑀁 𑀯𑀸 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀫𑀬𑁆𑀳𑀁; 𑀦𑀢𑁆𑀣𑀺 𑀢𑀼𑀬𑁆𑀳𑁂𑀢𑁆𑀣 𑀓𑀺𑀜𑁆𑀘𑀺𑁇 𑀧𑀺𑀢𑀼 𑀫𑁂, 𑀪𑁄𑀢𑀺, 𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀳𑀻’𑀢𑀺𑁇

1108\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀲𑀢𑁆𑀣𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀑𑀯𑀭𑀓𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀭𑀁 𑀑𑀧𑀸𑀤𑁂𑀲𑀺—  ‘𑀬𑀸𑀯 𑀯𑀺𑀚𑀸𑀬𑀸𑀫𑀺 𑀬𑀤𑀺 𑀯𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑁄 𑀬𑀤𑀺 𑀯𑀸 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸’𑀢𑀺𑁇 𑀲𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀚𑀻𑀯𑀺𑀢𑀜𑁆𑀘 𑀕𑀩𑁆𑀪𑀜𑁆𑀘 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀯𑀺𑀦𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀣𑀸 𑀢𑀁 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑀻, 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀧𑀭𑀮𑁄𑀓𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄; 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑀻𑁇 𑀦 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀓𑁆𑀓𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺; 𑀅𑀧𑀺 𑀘 𑀧𑀭𑀺𑀧𑀸𑀓𑀁 𑀆𑀕𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑁄 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑁂𑀦𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀭𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀺, 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀘 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑀺𑀦𑀸𑀧𑀺 𑀔𑁄 𑀢𑁂, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀼—  ‘𑀇𑀢𑀺𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇

(421.)

1109\. “𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀅𑀣 𑀔𑁄 𑀏𑀯𑀁 𑀫𑁂 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬, 𑀭𑀸𑀚𑀜𑁆𑀜𑀸”𑀢𑀺? “𑀇𑀥 𑀫𑁂, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀼𑀭𑀺𑀲𑀸 𑀘𑁄𑀭𑀁 𑀆𑀕𑀼𑀘𑀸𑀭𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑁄𑀭𑁄 𑀆𑀕𑀼𑀘𑀸𑀭𑀻; 𑀇𑀫𑀲𑁆𑀲 𑀬𑀁 𑀇𑀘𑁆𑀙𑀲𑀺, 𑀢𑀁 𑀤𑀡𑁆𑀟𑀁 𑀧𑀡𑁂𑀳𑀻’𑀢𑀺𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀓𑀼𑀫𑁆𑀪𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀔𑀁 𑀧𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑁆𑀮𑁂𑀦 𑀘𑀫𑁆𑀫𑁂𑀦 𑀑𑀦𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑁆𑀮𑀸𑀬 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀩𑀳𑀮𑀸𑀯𑀮𑁂𑀧𑀦𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑁆𑀥𑀦𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀁 𑀤𑁂𑀣𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀫𑁂 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀓𑀼𑀫𑁆𑀪𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀔𑀁 𑀧𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑁆𑀮𑁂𑀦 𑀘𑀫𑁆𑀫𑁂𑀦 𑀑𑀦𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑁆𑀮𑀸𑀬 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀩𑀳𑀮𑀸𑀯𑀮𑁂𑀧𑀦𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑁆𑀥𑀦𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑀤𑀸 𑀫𑀬𑀁 𑀚𑀸𑀦𑀸𑀫 ‘𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄’𑀢𑀺, 𑀅𑀣 𑀦𑀁 𑀓𑀼𑀫𑁆𑀪𑀺𑀁 𑀑𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀩𑁆𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀔𑀁 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀡𑀺𑀓𑀁 𑀦𑀺𑀮𑁆𑀮𑁄𑀓𑁂𑀫—  ‘𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀲𑁆𑀲 𑀚𑀻𑀯𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀦𑁂𑀯𑀲𑁆𑀲 𑀫𑀬𑀁 𑀚𑀻𑀯𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀬𑁂𑀦 𑀫𑁂 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇
