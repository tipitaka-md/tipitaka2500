# 𑁧𑁦𑁇𑁨𑁇𑁧𑁩 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀓𑁅𑀧𑀫𑀸

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀢𑁆𑀣𑀺𑀓𑀯𑀸𑀤

### 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀓𑁅𑀧𑀫𑀸

(434.)

1129\. “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀉𑀧𑀫𑀸𑀬 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀤𑁆𑀯𑁂 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀸 𑀅𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑀺𑀩𑁆𑀩𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑁄 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀆𑀕𑀢𑀸𑀕𑀢𑀁 𑀓𑀮𑀺𑀁 𑀕𑀺𑀮𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀢𑀁 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀁 𑀆𑀕𑀢𑀸𑀕𑀢𑀁 𑀓𑀮𑀺𑀁 𑀕𑀺𑀮𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸 𑀢𑀁 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀢𑁆𑀯𑀁 𑀔𑁄, 𑀲𑀫𑁆𑀫, 𑀏𑀓𑀦𑁆𑀢𑀺𑀓𑁂𑀦 𑀚𑀺𑀦𑀸𑀲𑀺, 𑀤𑁂𑀳𑀺 𑀫𑁂, 𑀲𑀫𑁆𑀫, 𑀅𑀓𑁆𑀔𑁂 𑀧𑀚𑁄𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀲𑀫𑁆𑀫𑀸’𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑁆𑀔𑁂 𑀧𑀸𑀤𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀅𑀓𑁆𑀔𑁂 𑀯𑀺𑀲𑁂𑀦 𑀧𑀭𑀺𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀏𑀳𑀺 𑀔𑁄, 𑀲𑀫𑁆𑀫, 𑀅𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀲𑀫𑁆𑀫𑀸’𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑁂 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀸 𑀅𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑀺𑀩𑁆𑀩𑀺𑀁𑀲𑀼𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀆𑀕𑀢𑀸𑀕𑀢𑀁 𑀓𑀮𑀺𑀁 𑀕𑀺𑀮𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀢𑀁 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀁 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀆𑀕𑀢𑀸𑀕𑀢𑀁 𑀓𑀮𑀺𑀁 𑀕𑀺𑀮𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸 𑀢𑀁 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1130\. _‘𑀮𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀫𑁂𑀦 𑀢𑁂𑀚𑀲𑀸,_  
_𑀕𑀺𑀮𑀫𑀓𑁆𑀔𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀕𑀺𑀮 𑀭𑁂 𑀕𑀺𑀮 𑀧𑀸𑀧𑀥𑀼𑀢𑁆𑀢𑀓,_  
_𑀧𑀘𑁆𑀙𑀸 𑀢𑁂 𑀓𑀝𑀼𑀓𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇_  


1131\. 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑀓𑀽𑀧𑀫𑁄 𑀫𑀜𑁆𑀜𑁂 𑀧𑀝𑀺𑀪𑀸𑀲𑀺𑁇 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁; 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁𑁇 𑀫𑀸 𑀢𑁂 𑀅𑀳𑁄𑀲𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

(435.)

1132\. “𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀅𑀣 𑀔𑁄 𑀦𑁂𑀯𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀇𑀤𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀭𑀸𑀚𑀸𑀧𑀺 𑀫𑀁 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀺𑀭𑁄𑀭𑀸𑀚𑀸𑀦𑁄𑀧𑀺—  ‘𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀻—  “𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁄”’𑀢𑀺𑁇 𑀲𑀘𑀸𑀳𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀇𑀤𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑁂 𑀯𑀢𑁆𑀢𑀸𑀭𑁄—  ‘𑀬𑀸𑀯 𑀩𑀸𑀮𑁄 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀕𑀸𑀳𑀻’𑀢𑀺𑁇 𑀓𑁄𑀧𑁂𑀦𑀧𑀺 𑀦𑀁 𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀫𑀓𑁆𑀔𑁂𑀦𑀧𑀺 𑀦𑀁 𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀮𑀸𑀲𑁂𑀦𑀧𑀺 𑀦𑀁 𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇
