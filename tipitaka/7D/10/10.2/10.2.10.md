# 𑁧𑁦𑁇𑁨𑁇𑁧𑁦 𑀅𑀕𑁆𑀕𑀺𑀓𑀚𑀝𑀺𑀮𑁅𑀧𑀫𑀸

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀢𑁆𑀣𑀺𑀓𑀯𑀸𑀤

### 𑀅𑀕𑁆𑀕𑀺𑀓𑀚𑀝𑀺𑀮𑁅𑀧𑀫𑀸

(428.)

1117. “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀉𑀧𑀫𑀸𑀬 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀅𑀕𑁆𑀕𑀺𑀓𑁄 𑀚𑀝𑀺𑀮𑁄 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑁂 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀚𑀦𑀧𑀤𑁂 𑀲𑀢𑁆𑀣𑁄 𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑀢𑁆𑀣𑁄 𑀢𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀏𑀓𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀢𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑁂𑀦 𑀲𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀁, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑁂𑀢𑁆𑀣 𑀓𑀺𑀜𑁆𑀘𑀺 𑀉𑀧𑀓𑀭𑀡𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀕𑁆𑀕𑀺𑀓𑁄 𑀚𑀝𑀺𑀮𑁄 𑀓𑀸𑀮𑀲𑁆𑀲𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀬 𑀬𑁂𑀦 𑀲𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀤𑁆𑀤𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑁆𑀣𑀯𑀸𑀲𑁂 𑀤𑀳𑀭𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀫𑀦𑁆𑀤𑀁 𑀉𑀢𑁆𑀢𑀸𑀦𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀙𑀟𑁆𑀟𑀺𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦 𑀔𑁄 𑀫𑁂 𑀢𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀬𑀁 𑀫𑁂 𑀧𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀲𑁆𑀲 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀓𑀸𑀮𑀁𑀓𑀭𑁂𑀬𑁆𑀬; 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀤𑀸𑀭𑀓𑀁 𑀅𑀲𑁆𑀲𑀫𑀁 𑀦𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀸𑀤𑁂𑀬𑁆𑀬𑀁 𑀧𑁄𑀲𑁂𑀬𑁆𑀬𑀁 𑀯𑀟𑁆𑀠𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀕𑁆𑀕𑀺𑀓𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀅𑀲𑁆𑀲𑀫𑀁 𑀦𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀸𑀤𑁂𑀲𑀺 𑀧𑁄𑀲𑁂𑀲𑀺 𑀯𑀟𑁆𑀠𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀲𑁄 𑀤𑀸𑀭𑀓𑁄 𑀤𑀲𑀯𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑁂𑀲𑀺𑀓𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑁂𑀲𑀺𑀓𑁄 𑀯𑀸, 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀚𑀦𑀧𑀤𑁂 𑀓𑀜𑁆𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀕𑁆𑀕𑀺𑀓𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀢𑀸𑀢, 𑀚𑀦𑀧𑀤𑀁 𑀕𑀦𑁆𑀢𑀼𑀁; 𑀅𑀕𑁆𑀕𑀺𑀁, 𑀢𑀸𑀢, 𑀧𑀭𑀺𑀘𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀫𑀸 𑀘 𑀢𑁂 𑀅𑀕𑁆𑀕𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑁇 𑀲𑀘𑁂 𑀘 𑀢𑁂 𑀅𑀕𑁆𑀕𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑁂𑀬𑁆𑀬, 𑀅𑀬𑀁 𑀯𑀸𑀲𑀻 𑀇𑀫𑀸𑀦𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀇𑀤𑀁 𑀅𑀭𑀡𑀺𑀲𑀳𑀺𑀢𑀁, 𑀅𑀕𑁆𑀕𑀺𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀁 𑀧𑀭𑀺𑀘𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀕𑁆𑀕𑀺𑀓𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀏𑀯𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀚𑀦𑀧𑀤𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀔𑀺𑀟𑁆𑀟𑀸𑀧𑀲𑀼𑀢𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑁇
