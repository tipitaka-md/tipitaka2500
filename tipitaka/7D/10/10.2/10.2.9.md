# 𑁧𑁦𑁇𑁨𑁇𑁯 𑀲𑀗𑁆𑀔𑀥𑀫𑁅𑀧𑀫𑀸

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀢𑁆𑀣𑀺𑀓𑀯𑀸𑀤

### 𑀲𑀗𑁆𑀔𑀥𑀫𑁅𑀧𑀫𑀸

(426.)

1114\. “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀉𑀧𑀫𑀸𑀬 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀲𑀗𑁆𑀔𑀥𑀫𑁄 𑀲𑀗𑁆𑀔𑀁 𑀆𑀤𑀸𑀬 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀁 𑀚𑀦𑀧𑀤𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀲𑁄 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀕𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀕𑀸𑀫𑀲𑁆𑀲 𑀞𑀺𑀢𑁄 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀲𑀗𑁆𑀔𑀁 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀔𑀁 𑀪𑀽𑀫𑀺𑀬𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀢𑁂𑀲𑀁 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀚𑀦𑀧𑀤𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀫𑁆𑀪𑁄 𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀏𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀏𑀯𑀁𑀭𑀚𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀓𑀫𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀤𑀦𑀻𑀬𑁄 𑀏𑀯𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀼𑀘𑁆𑀙𑀦𑀻𑀬𑁄’𑀢𑀺𑁇 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀲𑀗𑁆𑀔𑀥𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀅𑀫𑁆𑀪𑁄, 𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀏𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀏𑀯𑀁𑀭𑀚𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀓𑀫𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀤𑀦𑀻𑀬𑁄 𑀏𑀯𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀼𑀘𑁆𑀙𑀦𑀻𑀬𑁄’𑀢𑀺𑁇 ‘𑀏𑀲𑁄 𑀔𑁄, 𑀪𑁄, 𑀲𑀗𑁆𑀔𑁄 𑀦𑀸𑀫 𑀬𑀲𑁆𑀲𑁂𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀏𑀯𑀁𑀭𑀚𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀓𑀫𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀤𑀦𑀻𑀬𑁄 𑀏𑀯𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑀻𑀬𑁄 𑀏𑀯𑀁𑀫𑀼𑀘𑁆𑀙𑀦𑀻𑀬𑁄’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀲𑀗𑁆𑀔𑀁 𑀉𑀢𑁆𑀢𑀸𑀦𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑀼𑀁—  ‘𑀯𑀤𑁂𑀳𑀺, 𑀪𑁄 𑀲𑀗𑁆𑀔, 𑀯𑀤𑁂𑀳𑀺, 𑀪𑁄 𑀲𑀗𑁆𑀔𑀸’𑀢𑀺𑁇 𑀦𑁂𑀯 𑀲𑁄 𑀲𑀗𑁆𑀔𑁄 𑀲𑀤𑁆𑀤𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀲𑀗𑁆𑀔𑀁 𑀅𑀯𑀓𑀼𑀚𑁆𑀚𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑀼𑀁, 𑀧𑀲𑁆𑀲𑁂𑀦 𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑀼𑀁, 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑀼𑀁, 𑀉𑀤𑁆𑀥𑀁 𑀞𑀧𑁂𑀲𑀼𑀁, 𑀑𑀫𑀼𑀤𑁆𑀥𑀓𑀁 𑀞𑀧𑁂𑀲𑀼𑀁, 𑀧𑀸𑀡𑀺𑀦𑀸 𑀆𑀓𑁄𑀝𑁂𑀲𑀼𑀁, 𑀮𑁂𑀟𑁆𑀟𑀼𑀦𑀸 𑀆𑀓𑁄𑀝𑁂𑀲𑀼𑀁, 𑀤𑀡𑁆𑀟𑁂𑀦 𑀆𑀓𑁄𑀝𑁂𑀲𑀼𑀁, 𑀲𑀢𑁆𑀣𑁂𑀦 𑀆𑀓𑁄𑀝𑁂𑀲𑀼𑀁, 𑀑𑀥𑀼𑀦𑀺𑀁𑀲𑀼 𑀲𑀦𑁆𑀥𑀼𑀦𑀺𑀁𑀲𑀼 𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀺𑀁𑀲𑀼—  ‘𑀯𑀤𑁂𑀳𑀺, 𑀪𑁄 𑀲𑀗𑁆𑀔, 𑀯𑀤𑁂𑀳𑀺, 𑀪𑁄 𑀲𑀗𑁆𑀔𑀸’𑀢𑀺𑁇 𑀦𑁂𑀯 𑀲𑁄 𑀲𑀗𑁆𑀔𑁄 𑀲𑀤𑁆𑀤𑀫𑀓𑀸𑀲𑀺𑁇

1115\. 𑀅𑀣 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀥𑀫𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀸𑀯 𑀩𑀸𑀮𑀸 𑀇𑀫𑁂 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀚𑀦𑀧𑀤𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸, 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀲𑀗𑁆𑀔𑀲𑀤𑁆𑀤𑀁 𑀕𑀯𑁂𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀧𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀦𑀁 𑀲𑀗𑁆𑀔𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀲𑀗𑁆𑀔𑀁 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀔𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀢𑁂𑀲𑀁 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀚𑀦𑀧𑀤𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀤𑀸 𑀓𑀺𑀭, 𑀪𑁄, 𑀅𑀬𑀁 𑀲𑀗𑁆𑀔𑁄 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀲𑀲𑀳𑀕𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀯𑀸𑀬𑀸𑀫𑀲𑀳𑀕𑀢𑁄 𑀘 𑀯𑀸𑀬𑀼𑀲𑀳𑀕𑀢𑁄 𑀘, 𑀢𑀤𑀸𑀬𑀁 𑀲𑀗𑁆𑀔𑁄 𑀲𑀤𑁆𑀤𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀬𑀤𑀸 𑀧𑀦𑀸𑀬𑀁 𑀲𑀗𑁆𑀔𑁄 𑀦𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑀲𑀳𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀯𑀸𑀬𑀸𑀫𑀲𑀳𑀕𑀢𑁄 𑀦 𑀯𑀸𑀬𑀼𑀲𑀳𑀕𑀢𑁄, 𑀦𑀸𑀬𑀁 𑀲𑀗𑁆𑀔𑁄 𑀲𑀤𑁆𑀤𑀁 𑀓𑀭𑁄𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀤𑀸𑀬𑀁 𑀓𑀸𑀬𑁄 𑀆𑀬𑀼𑀲𑀳𑀕𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀉𑀲𑁆𑀫𑀸𑀲𑀳𑀕𑀢𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑁄 𑀘, 𑀢𑀤𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀢𑀺𑀧𑀺 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀢𑀺𑀧𑀺 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑀧𑀺 𑀦𑀺𑀲𑀻𑀤𑀢𑀺𑀧𑀺 𑀲𑁂𑀬𑁆𑀬𑀫𑁆𑀧𑀺 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀘𑀓𑁆𑀔𑀼𑀦𑀸𑀧𑀺 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄𑀢𑁂𑀦𑀧𑀺 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀖𑀸𑀦𑁂𑀦𑀧𑀺 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀢𑀺, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀧𑀺 𑀭𑀲𑀁 𑀲𑀸𑀬𑀢𑀺, 𑀓𑀸𑀬𑁂𑀦𑀧𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀢𑀺, 𑀫𑀦𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦𑀸𑀬𑀁 𑀓𑀸𑀬𑁄 𑀦𑁂𑀯 𑀆𑀬𑀼𑀲𑀳𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀉𑀲𑁆𑀫𑀸𑀲𑀳𑀕𑀢𑁄, 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑁄, 𑀢𑀤𑀸 𑀦𑁂𑀯 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀢𑀺 𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀢𑀺 𑀦 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀘𑀓𑁆𑀔𑀼𑀦𑀸𑀧𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄𑀢𑁂𑀦𑀧𑀺 𑀲𑀤𑁆𑀤𑀁 𑀦 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀖𑀸𑀦𑁂𑀦𑀧𑀺 𑀕𑀦𑁆𑀥𑀁 𑀦 𑀖𑀸𑀬𑀢𑀺, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀧𑀺 𑀭𑀲𑀁 𑀦 𑀲𑀸𑀬𑀢𑀺, 𑀓𑀸𑀬𑁂𑀦𑀧𑀺 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀦 𑀨𑀼𑀲𑀢𑀺, 𑀫𑀦𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀇𑀫𑀺𑀦𑀸𑀧𑀺 𑀔𑁄 𑀢𑁂, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀼—  ‘𑀇𑀢𑀺𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇

(427.)

1116\. “𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀅𑀣 𑀔𑁄 𑀏𑀯𑀁 𑀫𑁂 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬, 𑀭𑀸𑀚𑀜𑁆𑀜𑀸”𑀢𑀺? “𑀇𑀥 𑀫𑁂, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀼𑀭𑀺𑀲𑀸 𑀘𑁄𑀭𑀁 𑀆𑀕𑀼𑀘𑀸𑀭𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑁄𑀭𑁄 𑀆𑀕𑀼𑀘𑀸𑀭𑀻, 𑀇𑀫𑀲𑁆𑀲 𑀬𑀁 𑀇𑀘𑁆𑀙𑀲𑀺, 𑀢𑀁 𑀤𑀡𑁆𑀟𑀁 𑀧𑀡𑁂𑀳𑀻’𑀢𑀺𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀙𑀯𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀣, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀲𑁆𑀲 𑀚𑀻𑀯𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀙𑀯𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀯𑀲𑁆𑀲 𑀫𑀬𑀁 𑀚𑀻𑀯𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀘𑀫𑁆𑀫𑀁 𑀙𑀺𑀦𑁆𑀤𑀣, 𑀫𑀁𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀣, 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑀺𑀦𑁆𑀤𑀣, 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀣, 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀙𑀺𑀦𑁆𑀤𑀣, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀲𑁆𑀲 𑀚𑀻𑀯𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀦𑁂𑀯𑀲𑁆𑀲 𑀫𑀬𑀁 𑀚𑀻𑀯𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀬𑁂𑀦 𑀫𑁂 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇
