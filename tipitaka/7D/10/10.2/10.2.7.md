# 𑁧𑁦𑁇𑁨𑁇𑁭 𑀲𑀼𑀧𑀺𑀦𑀓𑁅𑀧𑀫𑀸

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀢𑁆𑀣𑀺𑀓𑀯𑀸𑀤

### 𑀲𑀼𑀧𑀺𑀦𑀓𑁅𑀧𑀫𑀸

(422.)

1110\. “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬, 𑀢𑀣𑀸 𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀲𑀺 𑀦𑁄 𑀢𑁆𑀯𑀁, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀤𑀺𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀕𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀓𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀸 𑀆𑀭𑀸𑀫𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀯𑀦𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀪𑀽𑀫𑀺𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀦𑁆”𑀢𑀺? “𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀳𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀤𑀺𑀯𑀸𑀲𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀕𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀓𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀸 𑀆𑀭𑀸𑀫𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀯𑀦𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀪𑀽𑀫𑀺𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀦𑁆”𑀢𑀺𑁇 “𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀢𑀁 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂 𑀔𑀼𑀚𑁆𑀚𑀸𑀧𑀺 𑀯𑀸𑀫𑀦𑀓𑀸𑀧𑀺 𑀯𑁂𑀮𑀸𑀲𑀺𑀓𑀸𑀧𑀺 𑀓𑁄𑀫𑀸𑀭𑀺𑀓𑀸𑀧𑀻”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀫𑀁 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂 𑀔𑀼𑀚𑁆𑀚𑀸𑀧𑀺 𑀯𑀸𑀫𑀦𑀓𑀸𑀧𑀺 𑀯𑁂𑀮𑀸𑀲𑀺𑀓𑀸𑀧𑀺 𑀓𑁄𑀫𑀸𑀭𑀺𑀓𑀸𑀧𑀻”𑀢𑀺𑁇 “𑀅𑀧𑀺 𑀦𑀼 𑀢𑀸 𑀢𑀼𑀬𑁆𑀳𑀁 𑀚𑀻𑀯𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧”𑁇 “𑀢𑀸 𑀳𑀺 𑀦𑀸𑀫, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀢𑀼𑀬𑁆𑀳𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀚𑀻𑀯𑀦𑁆𑀢𑀺𑀬𑁄 𑀚𑀻𑀯𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀯𑀸𑁇 𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀲𑁆𑀲 𑀚𑀻𑀯𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀲𑀺 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀯𑀸𑁇 𑀇𑀫𑀺𑀦𑀸𑀧𑀺 𑀔𑁄 𑀢𑁂, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀼—  ‘𑀇𑀢𑀺𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇

(423.)

1111\. “𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀅𑀣 𑀔𑁄 𑀏𑀯𑀁 𑀫𑁂 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  “𑀅𑀢𑁆𑀣𑀺, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬, 𑀭𑀸𑀚𑀜𑁆𑀜𑀸”𑀢𑀺? “𑀇𑀥 𑀫𑁂, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀼𑀭𑀺𑀲𑀸 𑀘𑁄𑀭𑀁 𑀆𑀕𑀼𑀘𑀸𑀭𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑁄𑀭𑁄 𑀆𑀕𑀼𑀘𑀸𑀭𑀻; 𑀇𑀫𑀲𑁆𑀲 𑀬𑀁 𑀇𑀘𑁆𑀙𑀲𑀺, 𑀢𑀁 𑀤𑀡𑁆𑀟𑀁 𑀧𑀡𑁂𑀳𑀻’𑀢𑀺𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑁂𑀢𑁆𑀯𑀸 𑀚𑀺𑀬𑀸𑀬 𑀅𑀦𑀲𑁆𑀲𑀸𑀲𑀓𑀁 𑀫𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑁂𑀣𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀫𑁂 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑁂𑀢𑁆𑀯𑀸 𑀚𑀺𑀬𑀸𑀬 𑀅𑀦𑀲𑁆𑀲𑀸𑀲𑀓𑀁 𑀫𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑀤𑀸 𑀲𑁄 𑀚𑀻𑀯𑀢𑀺, 𑀢𑀤𑀸 𑀮𑀳𑀼𑀢𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀼𑀤𑀼𑀢𑀭𑁄 𑀘 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀭𑁄 𑀘𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀲𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺 𑀢𑀤𑀸 𑀕𑀭𑀼𑀢𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀢𑁆𑀣𑀺𑀦𑁆𑀦𑀢𑀭𑁄 𑀘 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀭𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀬𑁂𑀦 𑀫𑁂 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇
