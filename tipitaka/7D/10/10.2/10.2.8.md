# 𑁧𑁦𑁇𑁨𑁇𑁮 𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀅𑀬𑁄𑀕𑀼𑀍𑀒𑀧𑀫𑀸

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀢𑁆𑀣𑀺𑀓𑀯𑀸𑀤

### 𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀅𑀬𑁄𑀕𑀼𑀍𑀒𑀧𑀫𑀸

(424.)

1112. “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀉𑀧𑀫𑀸𑀬 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀯𑀲𑀁 𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀅𑀬𑁄𑀕𑀼𑀍𑀅𑀁 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑁂𑀬𑁆𑀬𑁇 𑀢𑀫𑁂𑀦𑀁 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀻𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀁 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑁂𑀬𑁆𑀬𑁇 𑀓𑀤𑀸 𑀦𑀼 𑀔𑁄 𑀲𑁄 𑀅𑀬𑁄𑀕𑀼𑀍𑀑 𑀮𑀳𑀼𑀢𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀫𑀼𑀤𑀼𑀢𑀭𑁄 𑀯𑀸 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸, 𑀬𑀤𑀸 𑀯𑀸 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄, 𑀬𑀤𑀸 𑀯𑀸 𑀲𑀻𑀢𑁄 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺? “𑀬𑀤𑀸 𑀲𑁄, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀬𑁄𑀕𑀼𑀍𑀑 𑀢𑁂𑀚𑁄𑀲𑀳𑀕𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀯𑀸𑀬𑁄𑀲𑀳𑀕𑀢𑁄 𑀘 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄, 𑀢𑀤𑀸 𑀮𑀳𑀼𑀢𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀼𑀤𑀼𑀢𑀭𑁄 𑀘 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀭𑁄 𑀘𑁇 𑀬𑀤𑀸 𑀧𑀦 𑀲𑁄 𑀅𑀬𑁄𑀕𑀼𑀍𑀑 𑀦𑁂𑀯 𑀢𑁂𑀚𑁄𑀲𑀳𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀯𑀸𑀬𑁄𑀲𑀳𑀕𑀢𑁄 𑀲𑀻𑀢𑁄 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀢𑀤𑀸 𑀕𑀭𑀼𑀢𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀢𑁆𑀣𑀺𑀦𑁆𑀦𑀢𑀭𑁄 𑀘 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀤𑀸𑀬𑀁 𑀓𑀸𑀬𑁄 𑀆𑀬𑀼𑀲𑀳𑀕𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀉𑀲𑁆𑀫𑀸𑀲𑀳𑀕𑀢𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑁄 𑀘, 𑀢𑀤𑀸 𑀮𑀳𑀼𑀢𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀼𑀤𑀼𑀢𑀭𑁄 𑀘 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀭𑁄 𑀘𑁇 𑀬𑀤𑀸 𑀧𑀦𑀸𑀬𑀁 𑀓𑀸𑀬𑁄 𑀦𑁂𑀯 𑀆𑀬𑀼𑀲𑀳𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀉𑀲𑁆𑀫𑀸𑀲𑀳𑀕𑀢𑁄 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑁄 𑀢𑀤𑀸 𑀕𑀭𑀼𑀢𑀭𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀢𑁆𑀣𑀺𑀦𑁆𑀦𑀢𑀭𑁄 𑀘 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀭𑁄 𑀘𑁇 𑀇𑀫𑀺𑀦𑀸𑀧𑀺 𑀔𑁄 𑀢𑁂, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀼—  ‘𑀇𑀢𑀺𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇

(425.)

1113. “𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀏𑀯𑀫𑀸𑀳, 𑀅𑀣 𑀔𑁄 𑀏𑀯𑀁 𑀫𑁂 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄…𑀧𑁂…  𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬, 𑀭𑀸𑀚𑀜𑁆𑀜𑀸”𑀢𑀺? “𑀇𑀥 𑀫𑁂, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀼𑀭𑀺𑀲𑀸 𑀘𑁄𑀭𑀁 𑀆𑀕𑀼𑀘𑀸𑀭𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑁄𑀭𑁄 𑀆𑀕𑀼𑀘𑀸𑀭𑀻; 𑀇𑀫𑀲𑁆𑀲 𑀬𑀁 𑀇𑀘𑁆𑀙𑀲𑀺, 𑀢𑀁 𑀤𑀡𑁆𑀟𑀁 𑀧𑀡𑁂𑀳𑀻’𑀢𑀺𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀅𑀦𑀼𑀧𑀳𑀘𑁆𑀘 𑀙𑀯𑀺𑀜𑁆𑀘 𑀘𑀫𑁆𑀫𑀜𑁆𑀘 𑀫𑀁𑀲𑀜𑁆𑀘 𑀦𑁆𑀳𑀸𑀭𑀼𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀜𑁆𑀘 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀣, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀲𑁆𑀲 𑀚𑀻𑀯𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀫𑁂 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀅𑀦𑀼𑀧𑀳𑀘𑁆𑀘 𑀙𑀯𑀺𑀜𑁆𑀘…𑀧𑁂…  𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑀤𑀸 𑀲𑁄 𑀆𑀫𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀉𑀢𑁆𑀢𑀸𑀦𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀣, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀲𑁆𑀲 𑀚𑀻𑀯𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀉𑀢𑁆𑀢𑀸𑀦𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀯𑀲𑁆𑀲 𑀫𑀬𑀁 𑀚𑀻𑀯𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀅𑀯𑀓𑀼𑀚𑁆𑀚𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀣…  𑀧𑀲𑁆𑀲𑁂𑀦 𑀦𑀺𑀧𑀸𑀢𑁂𑀣…  𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀦𑀺𑀧𑀸𑀢𑁂𑀣…  𑀉𑀤𑁆𑀥𑀁 𑀞𑀧𑁂𑀣…  𑀑𑀫𑀼𑀤𑁆𑀥𑀓𑀁 𑀞𑀧𑁂𑀣…  𑀧𑀸𑀡𑀺𑀦𑀸 𑀆𑀓𑁄𑀝𑁂𑀣…  𑀮𑁂𑀟𑁆𑀟𑀼𑀦𑀸 𑀆𑀓𑁄𑀝𑁂𑀣…  𑀤𑀡𑁆𑀟𑁂𑀦 𑀆𑀓𑁄𑀝𑁂𑀣…  𑀲𑀢𑁆𑀣𑁂𑀦 𑀆𑀓𑁄𑀝𑁂𑀣…  𑀑𑀥𑀼𑀦𑀸𑀣 𑀲𑀦𑁆𑀥𑀼𑀦𑀸𑀣 𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀸𑀣, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀲𑁆𑀲 𑀚𑀻𑀯𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀑𑀥𑀼𑀦𑀦𑁆𑀢𑀺 𑀲𑀦𑁆𑀥𑀼𑀦𑀦𑁆𑀢𑀺 𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀯𑀲𑁆𑀲 𑀫𑀬𑀁 𑀚𑀻𑀯𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑁇 𑀢𑀲𑁆𑀲 𑀢𑀤𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀳𑁄𑀢𑀺 𑀢𑁂 𑀭𑀽𑀧𑀸, 𑀢𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀢𑀤𑁂𑀯 𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑁂 𑀲𑀤𑁆𑀤𑀸, 𑀢𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀢𑀤𑁂𑀯 𑀖𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑁂 𑀕𑀦𑁆𑀥𑀸, 𑀢𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑀸𑀯 𑀚𑀺𑀯𑁆𑀳𑀸 𑀳𑁄𑀢𑀺 𑀢𑁂 𑀭𑀲𑀸, 𑀢𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑁆𑀯𑁂𑀯 𑀓𑀸𑀬𑁄 𑀳𑁄𑀢𑀺 𑀢𑁂 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸, 𑀢𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀬𑁂𑀦 𑀫𑁂 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺𑁇
