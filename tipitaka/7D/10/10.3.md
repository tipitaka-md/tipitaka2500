# 𑁧𑁦𑁇𑁩 𑀲𑀭𑀡𑀕𑀫𑀦

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀭𑀡𑀕𑀫𑀦

(437.)

1137. “𑀧𑀼𑀭𑀺𑀫𑁂𑀦𑁂𑀯 𑀅𑀳𑀁 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀪𑁄𑀢𑁄 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀪𑀺𑀭𑀤𑁆𑀥𑁄𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀇𑀫𑀸𑀦𑀺 𑀯𑀺𑀘𑀺𑀢𑁆𑀭𑀸𑀦𑀺 𑀧𑀜𑁆𑀳𑀸𑀧𑀝𑀺𑀪𑀸𑀦𑀸𑀦𑀺 𑀲𑁄𑀢𑀼𑀓𑀸𑀫𑁄 𑀏𑀯𑀸𑀳𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀓𑀲𑁆𑀲𑀧𑀁 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀅𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀁𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑁄𑀢𑀸 𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀢𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀁𑁇

1138. 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀘𑀸𑀳𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧, 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑀢𑀼𑀁, 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀼 𑀫𑀁 𑀪𑀯𑀁 𑀓𑀲𑁆𑀲𑀧𑁄, 𑀬𑀁 𑀫𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇
