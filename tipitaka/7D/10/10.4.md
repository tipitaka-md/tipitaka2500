# 𑁧𑁦𑁇𑁪 𑀬𑀜𑁆𑀜𑀓𑀣𑀸

* 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀬𑀜𑁆𑀜𑀓𑀣𑀸

(438.)

1139\. “𑀬𑀣𑀸𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀜𑁆𑀜𑁂 𑀕𑀸𑀯𑁄 𑀯𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀅𑀚𑁂𑀍𑀅𑀓𑀸 𑀯𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀸 𑀯𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑀸 𑀧𑀸𑀡𑀸 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀸 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀻 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀻, 𑀏𑀯𑀭𑀽𑀧𑁄 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀜𑁆𑀜𑁄 𑀦 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄 𑀦 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑁄 𑀦 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀓𑀲𑁆𑀲𑀓𑁄 𑀩𑀻𑀚𑀦𑀗𑁆𑀕𑀮𑀁 𑀆𑀤𑀸𑀬 𑀯𑀦𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀤𑀼𑀩𑁆𑀪𑀽𑀫𑁂 𑀅𑀯𑀺𑀳𑀢𑀔𑀸𑀡𑀼𑀓𑀡𑁆𑀝𑀓𑁂 𑀩𑀻𑀚𑀸𑀦𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬 𑀔𑀡𑁆𑀟𑀸𑀦𑀺 𑀧𑀽𑀢𑀻𑀦𑀺 𑀯𑀸𑀢𑀸𑀢𑀧𑀳𑀢𑀸𑀦𑀺 𑀅𑀲𑀸𑀭𑀤𑀸𑀦𑀺 𑀅𑀲𑀼𑀔𑀲𑀬𑀺𑀢𑀸𑀦𑀺𑁇 𑀤𑁂𑀯𑁄 𑀘 𑀦 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀥𑀸𑀭𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀯𑁂𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀸𑀦𑀺 𑀩𑀻𑀚𑀸𑀦𑀺 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀓𑀲𑁆𑀲𑀓𑁄 𑀯𑀸 𑀯𑀺𑀧𑀼𑀮𑀁 𑀨𑀮𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑀺𑀤𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧”𑁇

1140\. “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀣𑀸𑀭𑀽𑀧𑁂 𑀬𑀜𑁆𑀜𑁂 𑀕𑀸𑀯𑁄 𑀯𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀅𑀚𑁂𑀍𑀅𑀓𑀸 𑀯𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀸 𑀯𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑀸 𑀧𑀸𑀡𑀸 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀸 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀻 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀻, 𑀏𑀯𑀭𑀽𑀧𑁄 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀜𑁆𑀜𑁄 𑀦 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄 𑀦 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑁄 𑀦 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑁄𑁇

1141\. 𑀬𑀣𑀸𑀭𑀽𑀧𑁂 𑀘 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀜𑁆𑀜𑁂 𑀦𑁂𑀯 𑀕𑀸𑀯𑁄 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀚𑁂𑀍𑀅𑀓𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀦 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑀸 𑀧𑀸𑀡𑀸 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀸 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀻 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀻, 𑀏𑀯𑀭𑀽𑀧𑁄 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀜𑁆𑀜𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑁄 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀓𑀲𑁆𑀲𑀓𑁄 𑀩𑀻𑀚𑀦𑀗𑁆𑀕𑀮𑀁 𑀆𑀤𑀸𑀬 𑀯𑀦𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀲𑀼𑀔𑁂𑀢𑁆𑀢𑁂 𑀲𑀼𑀪𑀽𑀫𑁂 𑀲𑀼𑀯𑀺𑀳𑀢𑀔𑀸𑀡𑀼𑀓𑀡𑁆𑀝𑀓𑁂 𑀩𑀻𑀚𑀸𑀦𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀧𑁂𑀬𑁆𑀬 𑀅𑀔𑀡𑁆𑀟𑀸𑀦𑀺 𑀅𑀧𑀽𑀢𑀻𑀦𑀺 𑀅𑀯𑀸𑀢𑀸𑀢𑀧𑀳𑀢𑀸𑀦𑀺 𑀲𑀸𑀭𑀤𑀸𑀦𑀺 𑀲𑀼𑀔𑀲𑀬𑀺𑀢𑀸𑀦𑀺𑁇 𑀤𑁂𑀯𑁄 𑀘 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀥𑀸𑀭𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀯𑁂𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀸𑀦𑀺 𑀩𑀻𑀚𑀸𑀦𑀺 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀓𑀲𑁆𑀲𑀓𑁄 𑀯𑀸 𑀯𑀺𑀧𑀼𑀮𑀁 𑀨𑀮𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑁄 𑀓𑀲𑁆𑀲𑀧”𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀣𑀸𑀭𑀽𑀧𑁂 𑀬𑀜𑁆𑀜𑁂 𑀦𑁂𑀯 𑀕𑀸𑀯𑁄 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀚𑁂𑀍𑀅𑀓𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀦 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀽𑀓𑀭𑀸 𑀳𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑀸 𑀧𑀸𑀡𑀸 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀸 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀻 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀻, 𑀏𑀯𑀭𑀽𑀧𑁄 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀬𑀜𑁆𑀜𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑁄 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑁄”𑀢𑀺𑁇
