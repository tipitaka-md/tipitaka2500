# 𑁧𑁦𑁇𑁬 𑀧𑀸𑀬𑀸𑀲𑀺𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀬𑀸𑀲𑀺𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢

(441.)

1144\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀯𑀫𑁆𑀧𑀢𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀲𑀼𑀜𑁆𑀜𑀁 𑀲𑁂𑀭𑀻𑀲𑀓𑀁 𑀯𑀺𑀫𑀸𑀦𑀁 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀸𑀬𑀸𑀲𑀺 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀕𑀯𑀫𑁆𑀧𑀢𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀕𑀯𑀫𑁆𑀧𑀢𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀧𑀸𑀬𑀸𑀲𑀺𑀁 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀯𑀫𑁆𑀧𑀢𑀺 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺? “𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑀁—  ‘𑀇𑀢𑀺𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀅𑀬𑁆𑀬𑁂𑀦 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑀺𑀢𑁄”𑀢𑀺𑁇 “𑀬𑁄 𑀧𑀦 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀸𑀦𑁂 𑀯𑀸𑀯𑀝𑁄 𑀅𑀳𑁄𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀫𑀸𑀡𑀯𑁄, 𑀲𑁄 𑀓𑀼𑀳𑀺𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? “𑀬𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀸𑀦𑁂 𑀯𑀸𑀯𑀝𑁄 𑀅𑀳𑁄𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀫𑀸𑀡𑀯𑁄, 𑀲𑁄 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀦𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁𑁇 𑀅𑀳𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀜𑁆𑀜𑀁 𑀲𑁂𑀭𑀻𑀲𑀓𑀁 𑀯𑀺𑀫𑀸𑀦𑀁𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀕𑀯𑀫𑁆𑀧𑀢𑀺, 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀭𑁄𑀘𑁂𑀳𑀺—  ‘𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣, 𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣, 𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣, 𑀅𑀦𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣𑁇 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀜𑁆𑀜𑀁 𑀲𑁂𑀭𑀻𑀲𑀓𑀁 𑀯𑀺𑀫𑀸𑀦𑀁𑁇 𑀬𑁄 𑀧𑀦 𑀢𑀲𑁆𑀲 𑀤𑀸𑀦𑁂 𑀯𑀸𑀯𑀝𑁄 𑀅𑀳𑁄𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀫𑀸𑀡𑀯𑁄, 𑀲𑁄 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀦𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀦𑁆’”𑀢𑀺𑁇

1145\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀯𑀫𑁆𑀧𑀢𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣, 𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣, 𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣, 𑀅𑀦𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀣𑁇 𑀧𑀸𑀬𑀸𑀲𑀺 𑀭𑀸𑀚𑀜𑁆𑀜𑁄 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀜𑁆𑀜𑀁 𑀲𑁂𑀭𑀻𑀲𑀓𑀁 𑀯𑀺𑀫𑀸𑀦𑀁𑁇 𑀬𑁄 𑀧𑀦 𑀢𑀲𑁆𑀲 𑀤𑀸𑀦𑁂 𑀯𑀸𑀯𑀝𑁄 𑀅𑀳𑁄𑀲𑀺 𑀉𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀫𑀸𑀡𑀯𑁄, 𑀲𑁄 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀦𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀦𑁆”𑀢𑀺𑁇

---

1146\. 𑀧𑀸𑀬𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀲𑀫𑀁𑁇



---

1147\. 𑀫𑀳𑀸𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

1148\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1149\. _𑀫𑀳𑀸𑀧𑀤𑀸𑀦 𑀦𑀺𑀤𑀸𑀦𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀚𑀦𑀯𑀲𑀪 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁,_  
_𑀲𑀫𑀬𑀁 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀓𑀁;_
