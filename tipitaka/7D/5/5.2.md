# 𑁫𑁇𑁨 𑀆𑀦𑀦𑁆𑀤𑀧𑀭𑀺𑀓𑀣𑀸

* 𑀚𑀦𑀯𑀲𑀪𑀲𑀼𑀢𑁆𑀢

* 𑀆𑀦𑀦𑁆𑀤𑀧𑀭𑀺𑀓𑀣𑀸

(276.)

599\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑁂 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀸𑀕𑀥𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀩𑀳𑀽 𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸𑁇 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑀜𑁆𑀜𑁂 𑀅𑀗𑁆𑀕𑀫𑀕𑀥𑀸 𑀅𑀗𑁆𑀕𑀫𑀸𑀕𑀥𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁂𑀳𑀺 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂𑀳𑀺𑁇 𑀢𑁂 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀲𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑁂 𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑁂 𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄𑁇 𑀢𑁂 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸; 𑀢𑁂𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀲𑀸𑀥𑀼 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁, 𑀩𑀳𑀼𑀚𑀦𑁄 𑀧𑀲𑀻𑀤𑁂𑀬𑁆𑀬, 𑀢𑀢𑁄 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀼𑀕𑀢𑀺𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺 𑀅𑀳𑁄𑀲𑀺 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀳𑀺𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀦𑁂𑀕𑀫𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀚𑀸𑀦𑀧𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼𑀤𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑀭𑀽𑀧𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  ‘𑀏𑀯𑀁 𑀦𑁄 𑀲𑁄 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀲𑀼𑀔𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀭𑀜𑁆𑀜𑁄 𑀯𑀺𑀚𑀺𑀢𑁂 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀺𑀫𑁆𑀳𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺 𑀅𑀳𑁄𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀲𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁂 𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁂 𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼𑀤𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀬𑀸𑀯 𑀫𑀭𑀡𑀓𑀸𑀮𑀸𑀧𑀺 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑀭𑀽𑀧𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀢𑀲𑁆𑀲𑀧𑀺𑀲𑁆𑀲 𑀲𑀸𑀥𑀼 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀩𑀳𑀼𑀚𑀦𑁄 𑀧𑀲𑀻𑀤𑁂𑀬𑁆𑀬, 𑀢𑀢𑁄 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀼𑀕𑀢𑀺𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀲𑀫𑁆𑀩𑁄𑀥𑀺 𑀫𑀕𑀥𑁂𑀲𑀼𑁇 𑀬𑀢𑁆𑀣 𑀔𑁄 𑀧𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺 𑀫𑀕𑀥𑁂𑀲𑀼, 𑀓𑀣𑀁 𑀢𑀢𑁆𑀭 𑀪𑀕𑀯𑀸 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀉𑀧𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀪𑀕𑀯𑀸 𑀘𑁂 𑀔𑁄 𑀧𑀦 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀉𑀧𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬, 𑀤𑀻𑀦𑀫𑀦𑀸 𑀢𑁂𑀦𑀲𑁆𑀲𑀼 𑀫𑀸𑀕𑀥𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸; 𑀬𑁂𑀦 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲𑀼 𑀤𑀻𑀦𑀫𑀦𑀸 𑀫𑀸𑀕𑀥𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀓𑀣𑀁 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺?

(277.)

600\. 𑀇𑀤𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀆𑀭𑀩𑁆𑀪 𑀏𑀓𑁄 𑀭𑀳𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂—  ‘𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀧𑀭𑀺𑀢𑁄 𑀧𑀭𑀺𑀢𑁄 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀉𑀧𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺 𑀓𑀸𑀲𑀺𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑀺𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀘𑁂𑀢𑀺𑀯𑀁𑀲𑁂𑀲𑀼 𑀓𑀼𑀭𑀼𑀧𑀜𑁆𑀘𑀸𑀮𑁂𑀲𑀼 𑀫𑀚𑁆𑀛𑀲𑀽𑀭𑀲𑁂𑀦𑁂𑀲𑀼—  “𑀅𑀲𑀼 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀧𑀧𑀦𑁆𑀦𑁄, 𑀅𑀲𑀼 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀧𑀧𑀦𑁆𑀦𑁄𑁇 𑀧𑀭𑁄𑀧𑀜𑁆𑀜𑀸𑀲 𑀦𑀸𑀢𑀺𑀓𑀺𑀬𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀦𑁄 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀸 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇 𑀲𑀸𑀥𑀺𑀓𑀸 𑀦𑀯𑀼𑀢𑀺 𑀦𑀸𑀢𑀺𑀓𑀺𑀬𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀸𑀦𑀁 𑀢𑀦𑀼𑀢𑁆𑀢𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄, 𑀲𑀓𑀺𑀤𑁂𑀯 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀢𑀺𑀭𑁂𑀓𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺 𑀦𑀸𑀢𑀺𑀓𑀺𑀬𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸𑀢𑀺𑁇 𑀢𑁂𑀦 𑀘 𑀦𑀸𑀢𑀺𑀓𑀺𑀬𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀧𑀫𑀼𑀤𑀺𑀢𑀸 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑁆𑀯𑀸”’𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀸𑀕𑀥𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀩𑀳𑀽 𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸𑁇 𑀲𑀼𑀜𑁆𑀜𑀸 𑀫𑀜𑁆𑀜𑁂 𑀅𑀗𑁆𑀕𑀫𑀕𑀥𑀸 𑀅𑀗𑁆𑀕𑀫𑀸𑀕𑀥𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁂𑀳𑀺 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂𑀳𑀺𑁇 𑀢𑁂 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀳𑁂𑀲𑀼𑀁 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀲𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑁂 𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑁂 𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄, 𑀢𑁂 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀢𑁂𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀲𑀸𑀥𑀼 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁, 𑀩𑀳𑀼𑀚𑀦𑁄 𑀧𑀲𑀻𑀤𑁂𑀬𑁆𑀬, 𑀢𑀢𑁄 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀼𑀕𑀢𑀺𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀳𑁄𑀲𑀺 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀳𑀺𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀦𑁂𑀕𑀫𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀚𑀸𑀦𑀧𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼𑀤𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑀭𑀽𑀧𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  ‘𑀏𑀯𑀁 𑀦𑁄 𑀲𑁄 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀲𑀼𑀔𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀭𑀜𑁆𑀜𑁄 𑀯𑀺𑀚𑀺𑀢𑁂 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀺𑀫𑁆𑀳𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀳𑁄𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀲𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁂 𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁂 𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼𑀤𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀬𑀸𑀯 𑀫𑀭𑀡𑀓𑀸𑀮𑀸𑀧𑀺 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑀭𑀽𑀧𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀢𑀲𑁆𑀲𑀧𑀺𑀲𑁆𑀲 𑀲𑀸𑀥𑀼 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁, 𑀩𑀳𑀼𑀚𑀦𑁄 𑀧𑀲𑀻𑀤𑁂𑀬𑁆𑀬, 𑀢𑀢𑁄 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀼𑀕𑀢𑀺𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑁆𑀩𑁄𑀥𑀺 𑀫𑀕𑀥𑁂𑀲𑀼𑁇 𑀬𑀢𑁆𑀣 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺 𑀫𑀕𑀥𑁂𑀲𑀼, 𑀓𑀣𑀁 𑀢𑀢𑁆𑀭 𑀪𑀕𑀯𑀸 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀉𑀧𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬? 𑀪𑀕𑀯𑀸 𑀘𑁂 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀉𑀧𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬, 𑀤𑀻𑀦𑀫𑀦𑀸 𑀢𑁂𑀦𑀲𑁆𑀲𑀼 𑀫𑀸𑀕𑀥𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸; 𑀬𑁂𑀦 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲𑀼 𑀤𑀻𑀦𑀫𑀦𑀸 𑀫𑀸𑀕𑀥𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀓𑀣𑀁 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀇𑀤𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀆𑀭𑀩𑁆𑀪 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀭𑀺𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

(278.)

601\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤𑁂 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀦𑀸𑀢𑀺𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀦𑀸𑀢𑀺𑀓𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺𑀓𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀁 𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺—  “𑀕𑀢𑀺𑀁 𑀦𑁂𑀲𑀁 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀁, 𑀬𑀁𑀕𑀢𑀺𑀓𑀸 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀬𑀁𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂—  “𑀬𑀁𑀕𑀢𑀺𑀓𑀸 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀬𑀁𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀧𑀘𑁆𑀙𑀸𑀬𑀸𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

(279.)

602\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀉𑀧𑀲𑀦𑁆𑀢𑀧𑀤𑀺𑀲𑁆𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀸𑀢𑀺𑀭𑀺𑀯 𑀪𑀕𑀯𑀢𑁄 𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀢𑁆𑀢𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁𑁇 𑀲𑀦𑁆𑀢𑁂𑀦 𑀦𑀽𑀦𑀚𑁆𑀚, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀲𑀻”𑀢𑀺?

603\. “𑀬𑀤𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀆𑀭𑀩𑁆𑀪 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀭𑀺𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄, 𑀢𑀤𑁂𑀯𑀸𑀳𑀁 𑀦𑀸𑀢𑀺𑀓𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺𑀓𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀁 𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀁—  ‘𑀕𑀢𑀺𑀁 𑀦𑁂𑀲𑀁 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀁, 𑀬𑀁𑀕𑀢𑀺𑀓𑀸 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀬𑀁𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 ‘𑀬𑀁𑀕𑀢𑀺𑀓𑀸 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀬𑀁𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀸’𑀢𑀺𑁇
