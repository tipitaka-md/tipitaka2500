# 𑁫𑁇𑁬 𑀪𑀸𑀯𑀺𑀢𑁃𑀤𑁆𑀥𑀺𑀧𑀸𑀤

### 𑀚𑀦𑀯𑀲𑀪𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀸𑀯𑀺𑀢𑁃𑀤𑁆𑀥𑀺𑀧𑀸𑀤

(287.)

631\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸, 𑀬𑀸𑀯 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀘𑀺𑀫𑁂 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀳𑀼𑀢𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀲𑀯𑀺𑀢𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀢𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥 𑀪𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀳𑀼𑀢𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀲𑀯𑀺𑀢𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀢𑀸𑀬𑁇

632\. 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀪𑁄 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂𑀧𑀺 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀪𑁄 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀬𑁂𑀧𑀺 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀪𑁄 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦𑁄, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸, 𑀫𑀫𑀧𑀺𑀫𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑁂”𑀢𑀺𑁇 “𑀅𑀳𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑁄 𑀇𑀫𑁂𑀲𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀢𑁆𑀢𑀸 𑀏𑀯𑀁 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀏𑀯𑀁𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄”𑀢𑀺𑁇 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑁇 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—
