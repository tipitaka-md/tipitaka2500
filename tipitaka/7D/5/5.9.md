# 𑁫𑁇𑁯 𑀲𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭

* 𑀚𑀦𑀯𑀲𑀪𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀢𑁆𑀢𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭

(290.)

637\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸, 𑀬𑀸𑀯 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀘𑀺𑀫𑁂 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀲𑀢𑁆𑀢 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀧𑀭𑀺𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑁇 𑀬𑀸 𑀔𑁄, 𑀪𑁄, 𑀇𑀫𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀕𑁆𑀕𑀢𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀢𑀸, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑁄, 𑀅𑀭𑀺𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀲𑁅𑀧𑀦𑀺𑀲𑁄 𑀇𑀢𑀺𑀧𑀺 𑀲𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲, 𑀪𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀧𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀧𑀳𑁄𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀧𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀧𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀧𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀧𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀁 𑀧𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀧𑀳𑁄𑀢𑀺𑁇 𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀪𑁄, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺 𑀅𑀧𑀸𑀭𑀼𑀢𑀸 𑀅𑀫𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀭𑀸’𑀢𑀺 𑀇𑀤𑀫𑁂𑀯 𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀳𑀺, 𑀪𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄, 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺 𑀅𑀧𑀸𑀭𑀼𑀢𑀸 𑀅𑀫𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀭𑀸𑁇

638\. 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑁄, 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀬𑁂 𑀘𑀺𑀫𑁂 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀻𑀢𑀸 𑀲𑀸𑀢𑀺𑀭𑁂𑀓𑀸𑀦𑀺 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀫𑀸𑀕𑀥𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘𑁂𑀯𑁂𑀢𑁆𑀣 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄𑁇

639\. _𑀅𑀢𑁆𑀣𑀸𑀬𑀁 𑀇𑀢𑀭𑀸 𑀧𑀚𑀸,_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀸𑀕𑀸𑀢𑀺 𑀫𑁂 𑀫𑀦𑁄;_  
_𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁 𑀦𑁄𑀧𑀺 𑀲𑀓𑁆𑀓𑁄𑀫𑀺,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀲𑁆𑀲 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀦𑁆”𑀢𑀺𑁇_  


(291.)

640\. 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀣, 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀉𑀍𑀆𑀭𑁄 𑀲𑀢𑁆𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀍𑀆𑀭𑀁 𑀥𑀫𑁆𑀫𑀓𑁆𑀔𑀸𑀦𑀁, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀉𑀍𑀆𑀭𑀸 𑀯𑀺𑀲𑁂𑀲𑀸𑀥𑀺𑀕𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑀁 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀢𑀺 𑀪𑀯𑀁 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀏𑀯𑀭𑀽𑀧𑁄 𑀉𑀍𑀆𑀭𑁄 𑀲𑀢𑁆𑀣𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀍𑀆𑀭𑀁 𑀥𑀫𑁆𑀫𑀓𑁆𑀔𑀸𑀦𑀁, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀉𑀍𑀆𑀭𑀸 𑀯𑀺𑀲𑁂𑀲𑀸𑀥𑀺𑀕𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀏𑀯𑀭𑀽𑀧𑁄 𑀉𑀍𑀆𑀭𑁄 𑀲𑀢𑁆𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀍𑀆𑀭𑀁 𑀥𑀫𑁆𑀫𑀓𑁆𑀔𑀸𑀦𑀁, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀉𑀍𑀆𑀭𑀸 𑀯𑀺𑀲𑁂𑀲𑀸𑀥𑀺𑀕𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

(292.)

641\. 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀅𑀪𑀸𑀲𑀺, 𑀇𑀫𑀫𑀢𑁆𑀣𑀁 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀪𑀸𑀲𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀲𑀬𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺’”𑁇

642\. 𑀇𑀫𑀫𑀢𑁆𑀣𑀁 𑀚𑀦𑀯𑀲𑀪𑁄 𑀬𑀓𑁆𑀔𑁄 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑀬𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀪𑀸𑀲𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀇𑀫𑀫𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀚𑀦𑀯𑀲𑀪𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀜𑁆𑀘 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑀺, 𑀇𑀫𑀫𑀢𑁆𑀣𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁𑁇 𑀢𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀦𑁆𑀢𑀺𑁇
