# 𑁫𑁇𑁩 𑀚𑀦𑀯𑀲𑀪𑀬𑀓𑁆𑀔

### 𑀚𑀦𑀯𑀲𑀪𑀲𑀼𑀢𑁆𑀢

### 𑀚𑀦𑀯𑀲𑀪𑀬𑀓𑁆𑀔

(280.)

604\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀬𑀓𑁆𑀔𑁄 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀺—  ‘𑀚𑀦𑀯𑀲𑀪𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸; 𑀚𑀦𑀯𑀲𑀪𑁄 𑀅𑀳𑀁, 𑀲𑀼𑀕𑀢𑀸’𑀢𑀺𑁇 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀲𑀺 𑀦𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀢𑀁 𑀬𑀤𑀺𑀤𑀁 𑀚𑀦𑀯𑀲𑀪𑁄”𑀢𑀺?

605\. “𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀇𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀢𑀁 𑀬𑀤𑀺𑀤𑀁 𑀚𑀦𑀯𑀲𑀪𑁄𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀮𑁄𑀫𑀸𑀦𑀺 𑀳𑀝𑁆𑀞𑀸𑀦𑀺 ‘𑀚𑀦𑀯𑀲𑀪𑁄’𑀢𑀺 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦 𑀳𑀺 𑀦𑀽𑀦 𑀲𑁄 𑀑𑀭𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀬𑀤𑀺𑀤𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀬𑀤𑀺𑀤𑀁 𑀚𑀦𑀯𑀲𑀪𑁄’”𑀢𑀺𑁇 “𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀤𑁆𑀤𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀉𑀍𑀆𑀭𑀯𑀡𑁆𑀡𑁄 𑀫𑁂 𑀬𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀓𑀫𑁆𑀧𑀺 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀺—  ‘𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸; 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀅𑀳𑀁, 𑀲𑀼𑀕𑀢𑀸𑀢𑀺𑁇 𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺, 𑀲𑁄 𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀚𑀸 𑀪𑀯𑀺𑀢𑀼𑀁 𑀧𑀳𑁄𑀫𑀺𑁇

606\. _𑀇𑀢𑁄 𑀲𑀢𑁆𑀢 𑀢𑀢𑁄 𑀲𑀢𑁆𑀢,_  
_𑀲𑀁𑀲𑀸𑀭𑀸𑀦𑀺 𑀘𑀢𑀼𑀤𑁆𑀤𑀲;_  
_𑀦𑀺𑀯𑀸𑀲𑀫𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀬𑀢𑁆𑀣 𑀫𑁂 𑀯𑀼𑀲𑀺𑀢𑀁 𑀧𑀼𑀭𑁂𑁇_  


(281.)

607\. 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀆𑀲𑀸 𑀘 𑀧𑀦 𑀫𑁂 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀢𑀸𑀬𑀸’”𑀢𑀺𑁇

608\. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀫𑀺𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀚𑀦𑀯𑀲𑀪𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲, 𑀅𑀩𑁆𑀪𑀼𑀢𑀫𑀺𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀚𑀦𑀯𑀲𑀪𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲𑁇 ‘𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺 𑀘 𑀯𑀤𑁂𑀲𑀺, ‘𑀆𑀲𑀸 𑀘 𑀧𑀦 𑀫𑁂 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀢𑀸𑀬𑀸’𑀢𑀺 𑀘 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀼𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀦𑀯𑀲𑀪𑁄 𑀬𑀓𑁆𑀔𑁄 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀍𑀆𑀭𑀁 𑀯𑀺𑀲𑁂𑀲𑀸𑀥𑀺𑀕𑀫𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺? “‘𑀦 𑀅𑀜𑁆𑀜𑀢𑁆𑀭, 𑀪𑀕𑀯𑀸, 𑀢𑀯 𑀲𑀸𑀲𑀦𑀸, 𑀦 𑀅𑀜𑁆𑀜𑀢𑁆𑀭, 𑀲𑀼𑀕𑀢, 𑀢𑀯 𑀲𑀸𑀲𑀦𑀸; 𑀬𑀤𑀕𑁆𑀕𑁂 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀺 𑀏𑀓𑀦𑁆𑀢𑀺𑀓𑀢𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄, 𑀢𑀤𑀕𑁆𑀕𑁂 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀆𑀲𑀸 𑀘 𑀧𑀦 𑀫𑁂 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀢𑀸𑀬𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁂𑀦 𑀫𑀳𑀸𑀭𑀸𑀚𑁂𑀦 𑀧𑁂𑀲𑀺𑀢𑁄 𑀯𑀺𑀭𑀽𑀍𑀳𑀓𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀅𑀤𑁆𑀤𑀲𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀕𑀥𑀓𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁂 𑀆𑀭𑀩𑁆𑀪 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺𑀓𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀁 𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁—  “𑀕𑀢𑀺𑀁 𑀦𑁂𑀲𑀁 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀁, 𑀬𑀁𑀕𑀢𑀺𑀓𑀸 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀬𑀁𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀁 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀪𑀸𑀲𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  “𑀬𑀁𑀕𑀢𑀺𑀓𑀸 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑁄 𑀬𑀁𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀦𑁆𑀢𑀜𑁆𑀘 𑀤𑀓𑁆𑀔𑀸𑀫𑀺, 𑀇𑀤𑀜𑁆𑀘 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁆𑀯𑁂𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇
