# 𑁫𑁇𑁭 𑀢𑀺𑀯𑀺𑀥𑀑𑀓𑀸𑀲𑀸𑀥𑀺𑀕𑀫

### 𑀚𑀦𑀯𑀲𑀪𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀺𑀯𑀺𑀥𑀑𑀓𑀸𑀲𑀸𑀥𑀺𑀕𑀫

(288.)

633\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸, 𑀬𑀸𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀢𑀬𑁄 𑀑𑀓𑀸𑀲𑀸𑀥𑀺𑀕𑀫𑀸 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀸 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀇𑀥 𑀪𑁄 𑀏𑀓𑀘𑁆𑀘𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀸𑀫𑁂𑀳𑀺 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺𑁇 𑀲𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁 𑀆𑀕𑀫𑁆𑀫 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀸𑀫𑁂𑀳𑀺 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺𑁇 𑀢𑀲𑁆𑀲 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀸𑀫𑁂𑀳𑀺 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁, 𑀲𑀼𑀔𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄, 𑀧𑀫𑀼𑀤𑀸 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑁂𑀣; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑁄, 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀸𑀫𑁂𑀳𑀺 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁, 𑀲𑀼𑀔𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀞𑀫𑁄 𑀑𑀓𑀸𑀲𑀸𑀥𑀺𑀕𑀫𑁄 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀥𑀺𑀕𑀫𑀸𑀬𑁇

634\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑁄, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁 𑀆𑀕𑀫𑁆𑀫 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑁆𑀢𑀺, 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑁆𑀢𑀺, 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀑𑀍𑀆𑀭𑀺𑀓𑀸𑀦𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸𑀦𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁, 𑀲𑀼𑀔𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄, 𑀧𑀫𑀼𑀤𑀸 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑁂𑀣; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀪𑁄 𑀑𑀍𑀆𑀭𑀺𑀓𑀸𑀦𑀁 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸𑀦𑀁 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁, 𑀲𑀼𑀔𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑀼𑀢𑀺𑀬𑁄 𑀑𑀓𑀸𑀲𑀸𑀥𑀺𑀕𑀫𑁄 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀥𑀺𑀕𑀫𑀸𑀬𑁇

635\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑁄, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 ‘𑀇𑀤𑀁 𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀇𑀤𑀁 𑀅𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀇𑀤𑀁 𑀲𑀸𑀯𑀚𑁆𑀚𑀁 𑀇𑀤𑀁 𑀅𑀦𑀯𑀚𑁆𑀚𑀁, 𑀇𑀤𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀇𑀤𑀁 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀁 𑀳𑀻𑀦𑀁 𑀇𑀤𑀁 𑀧𑀡𑀻𑀢𑀁, 𑀇𑀤𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁 𑀆𑀕𑀫𑁆𑀫 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀁, ‘𑀇𑀤𑀁 𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀇𑀤𑀁 𑀅𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀇𑀤𑀁 𑀲𑀸𑀯𑀚𑁆𑀚𑀁 𑀇𑀤𑀁 𑀅𑀦𑀯𑀚𑁆𑀚𑀁, 𑀇𑀤𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀇𑀤𑀁 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀁 𑀳𑀻𑀦𑀁 𑀇𑀤𑀁 𑀧𑀡𑀻𑀢𑀁, 𑀇𑀤𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀳𑀻𑀬𑀢𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀭𑀸𑀕𑀸 𑀯𑀺𑀚𑁆𑀚𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁, 𑀲𑀼𑀔𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄, 𑀧𑀫𑀼𑀤𑀸 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑁂𑀣; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀭𑀸𑀕𑀸 𑀯𑀺𑀚𑁆𑀚𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁, 𑀲𑀼𑀔𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀢𑀢𑀺𑀬𑁄 𑀑𑀓𑀸𑀲𑀸𑀥𑀺𑀕𑀫𑁄 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀢𑀬𑁄 𑀑𑀓𑀸𑀲𑀸𑀥𑀺𑀕𑀫𑀸 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀸 𑀲𑀼𑀔𑀲𑁆𑀲𑀸𑀥𑀺𑀕𑀫𑀸𑀬𑀸”𑀢𑀺𑁇 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀣, 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—
