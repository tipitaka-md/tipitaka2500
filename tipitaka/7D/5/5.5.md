# 𑁫𑁇𑁫 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀓𑀣𑀸

### 𑀚𑀦𑀯𑀲𑀪𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀓𑀣𑀸

(284.)

618. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀬𑀣𑀸𑀲𑀓𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼—  “𑀑𑀪𑀸𑀲𑀫𑁂𑀢𑀁 𑀜𑀲𑁆𑀲𑀸𑀫, 𑀬𑀁𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸𑀯 𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀧𑀺 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀬𑀣𑀸𑀲𑀓𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼—  “𑀑𑀪𑀸𑀲𑀫𑁂𑀢𑀁 𑀜𑀲𑁆𑀲𑀸𑀫 𑀬𑀁𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸𑀯 𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀇𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀏𑀓𑀕𑁆𑀕𑀸 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼—  “𑀑𑀪𑀸𑀲𑀫𑁂𑀢𑀁 𑀜𑀲𑁆𑀲𑀸𑀫, 𑀬𑀁𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸𑀯 𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

619. 𑀬𑀤𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇 𑀬𑁄 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀧𑀓𑀢𑀺𑀯𑀡𑁆𑀡𑁄 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀦𑀻𑀬𑁄 𑀲𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑀲𑁆𑀫𑀺𑀁𑁇 𑀬𑀤𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀲𑁄 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁄𑀯𑀡𑁆𑀡𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀫𑀸𑀦𑀼𑀲𑀁 𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀤𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀲𑁄 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑁇 𑀬𑀤𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀓𑁄𑀘𑀺 𑀤𑁂𑀯𑁄 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺—  “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀤𑁂𑀯𑀲𑁆𑀲 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀇𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄, 𑀢𑀲𑁆𑀲 𑀤𑁂𑀯𑀲𑁆𑀲 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

620. 𑀬𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀯𑁂𑀤𑀧𑀝𑀺𑀮𑀸𑀪𑀁; 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀧𑀝𑀺𑀮𑀸𑀪𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀸𑀚𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑁄 𑀅𑀥𑀼𑀦𑀸𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑁄 𑀭𑀚𑁆𑀚𑁂𑀦, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀯𑁂𑀤𑀧𑀝𑀺𑀮𑀸𑀪𑀁, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀧𑀝𑀺𑀮𑀸𑀪𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑁆𑀲 𑀤𑁂𑀯𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀯𑁂𑀤𑀧𑀝𑀺𑀮𑀸𑀪𑀁, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀧𑀝𑀺𑀮𑀸𑀪𑀁𑁇 𑀅𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀫𑀸𑀭𑀯𑀡𑁆𑀡𑀻 𑀳𑀼𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀲𑁄 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀢𑁂 𑀯𑀸 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀲𑀫𑁂 𑀯𑀸 𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—

621. _“𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀯𑀢 𑀪𑁄 𑀤𑁂𑀯𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀳𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀁𑁇_  


622. _𑀦𑀯𑁂 𑀤𑁂𑀯𑁂 𑀘 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁂 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁂;_  
_𑀲𑀼𑀕𑀢𑀲𑁆𑀫𑀺𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦 𑀇𑀥𑀸𑀕𑀢𑁂𑁇_  


623. _𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀯𑀡𑁆𑀡𑁂𑀦 𑀬𑀲𑀲𑀸𑀬𑀼𑀦𑀸;_  
_𑀲𑀸𑀯𑀓𑀸 𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀲𑁆𑀲,_  
_𑀯𑀺𑀲𑁂𑀲𑀽𑀧𑀕𑀢𑀸 𑀇𑀥𑁇_  


624. _𑀇𑀤𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀳𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀦𑁆”𑀢𑀺𑁇_  


(285.)

625. 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀣; 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀢𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄 𑀘 𑀫𑀜𑁆𑀚𑀼 𑀘 𑀲𑀯𑀦𑀻𑀬𑁄 𑀘 𑀩𑀺𑀦𑁆𑀤𑀼 𑀘 𑀅𑀯𑀺𑀲𑀸𑀭𑀻 𑀘 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀘 𑀦𑀺𑀦𑁆𑀦𑀸𑀤𑀻 𑀘𑁇 𑀬𑀣𑀸𑀧𑀭𑀺𑀲𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀭𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺; 𑀦 𑀘𑀲𑁆𑀲 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀖𑁄𑀲𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺𑁇 𑀬𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑁆𑀲𑀭𑁄”𑀢𑀺𑁇

626. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑁂 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑁂𑀓𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀲𑀼 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸, 𑀬𑀸𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑁄, 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀲𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑀸𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀲𑀩𑁆𑀩𑀦𑀺𑀳𑀻𑀦𑀁 𑀓𑀸𑀬𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀓𑀸𑀬𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

(286.)

627. 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀣; 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀢𑁄 𑀖𑁄𑀲𑁄𑀬𑁂𑀯 𑀤𑁂𑀯𑀸 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺—  “𑀬𑁆𑀯𑀸𑀬𑀁 𑀫𑀫 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀲𑁆𑀯𑀸𑀬𑀁 𑀏𑀓𑁄𑀯 𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇

628. _𑀏𑀓𑀲𑁆𑀫𑀺𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑁂 𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀏𑀓𑀲𑁆𑀫𑀺𑀁 𑀢𑀼𑀡𑁆𑀳𑀺𑀫𑀸𑀲𑀻𑀦𑁂,_  
_𑀲𑀩𑁆𑀩𑁂 𑀢𑀼𑀡𑁆𑀳𑀻 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


629. _𑀢𑀤𑀸𑀲𑀼 𑀤𑁂𑀯𑀸 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀳𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀬𑁆𑀯𑀸𑀬𑀁 𑀫𑀫 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑁆𑀯𑀸𑀬𑀁 𑀏𑀓𑁄𑀯 𑀪𑀸𑀲𑀢𑀻𑀢𑀺𑁇_  


630. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀏𑀓𑀢𑁆𑀢𑁂𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺, 𑀏𑀓𑀢𑁆𑀢𑁂𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—
