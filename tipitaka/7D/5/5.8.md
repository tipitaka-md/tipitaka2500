# 𑁫𑁇𑁮 𑀘𑀢𑀼𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦

### 𑀚𑀦𑀯𑀲𑀪𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦

(289.)

636\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸, 𑀬𑀸𑀯 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀘𑀺𑀫𑁂 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀓𑀼𑀲𑀮𑀲𑁆𑀲𑀸𑀥𑀺𑀕𑀫𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀭𑀓𑀸𑀬𑁂 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀭𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑁂 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀭𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀓𑀼𑀲𑀮𑀲𑁆𑀲𑀸𑀥𑀺𑀕𑀫𑀸𑀬𑀸”𑀢𑀺𑁇 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑁇 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—
