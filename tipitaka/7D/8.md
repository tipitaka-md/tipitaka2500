# 𑁮 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

(344.)

823\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀫𑀕𑀥𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀸𑀘𑀻𑀦𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀲𑀡𑁆𑀟𑀸 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄, 𑀢𑀲𑁆𑀲𑀼𑀢𑁆𑀢𑀭𑀢𑁄 𑀯𑁂𑀤𑀺𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀸𑀬𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀉𑀤𑀧𑀸𑀤𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀳𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀭𑀳𑀺 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄”𑀢𑀺? 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀫𑀕𑀥𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀧𑀸𑀘𑀻𑀦𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀲𑀡𑁆𑀟𑀸 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄, 𑀢𑀲𑁆𑀲𑀼𑀢𑁆𑀢𑀭𑀢𑁄 𑀯𑁂𑀤𑀺𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀸𑀬𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀬𑀁, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀸 𑀫𑀕𑀥𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀸𑀘𑀻𑀦𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀲𑀡𑁆𑀟𑀸 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄, 𑀢𑀲𑁆𑀲𑀼𑀢𑁆𑀢𑀭𑀢𑁄 𑀯𑁂𑀤𑀺𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀸𑀬𑀁𑁇 𑀬𑀤𑀺 𑀧𑀦, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀫𑀬𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀫 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇

### 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

(345.)

824\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑀁 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀬𑀁, 𑀢𑀸𑀢 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀪𑀕𑀯𑀸 𑀫𑀕𑀥𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀸𑀘𑀻𑀦𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀲𑀡𑁆𑀟𑀸 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄, 𑀢𑀲𑁆𑀲𑀼𑀢𑁆𑀢𑀭𑀢𑁄 𑀯𑁂𑀤𑀺𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀸𑀬𑀁𑁇 𑀬𑀤𑀺 𑀧𑀦, 𑀢𑀸𑀢 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀫𑀬𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀫 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑁂𑀮𑀼𑀯𑀧𑀡𑁆𑀟𑀼𑀯𑀻𑀡𑀁 𑀆𑀤𑀸𑀬 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀅𑀦𑀼𑀘𑀭𑀺𑀬𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇

### 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

(346.)

825\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑁂𑀳𑀺 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁂𑀦 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀕𑀥𑁂𑀲𑀼 𑀧𑀸𑀘𑀻𑀦𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀲𑀡𑁆𑀟𑀸 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄, 𑀢𑀲𑁆𑀲𑀼𑀢𑁆𑀢𑀭𑀢𑁄 𑀯𑁂𑀤𑀺𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀤𑀺𑀬𑀓𑁄 𑀧𑀩𑁆𑀩𑀢𑁄 𑀅𑀢𑀺𑀭𑀺𑀯 𑀑𑀪𑀸𑀲𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀫𑁆𑀩𑀲𑀡𑁆𑀟𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄 𑀬𑀣𑀸 𑀢𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀼𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼𑀤𑀁 𑀧𑀭𑀺𑀢𑁄 𑀕𑀸𑀫𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀆𑀤𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀼 𑀦𑀸𑀫𑀚𑁆𑀚 𑀯𑁂𑀤𑀺𑀬𑀓𑁄 𑀧𑀩𑁆𑀩𑀢𑁄 𑀛𑀸𑀬𑀢𑀺𑀲𑀼 𑀦𑀸𑀫𑀚𑁆𑀚 𑀯𑁂𑀤𑀺𑀬𑀓𑁄 𑀧𑀩𑁆𑀩𑀢𑁄 𑀚𑀮𑀢𑀺𑀲𑀼𑁇 𑀦𑀸𑀫𑀚𑁆𑀚 𑀯𑁂𑀤𑀺𑀬𑀓𑁄 𑀧𑀩𑁆𑀩𑀢𑁄𑁇 𑀓𑀺𑀁𑀲𑀼 𑀦𑀸𑀫𑀚𑁆𑀚 𑀯𑁂𑀤𑀺𑀬𑀓𑁄 𑀧𑀩𑁆𑀩𑀢𑁄 𑀅𑀢𑀺𑀭𑀺𑀯 𑀑𑀪𑀸𑀲𑀚𑀸𑀢𑁄 𑀅𑀫𑁆𑀩𑀲𑀡𑁆𑀟𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄”𑀢𑀺; 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀸 𑀮𑁄𑀫𑀳𑀝𑁆𑀞𑀚𑀸𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

### 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

(347.)

826\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑀁 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀤𑀼𑀭𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀸 𑀔𑁄, 𑀢𑀸𑀢 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀫𑀸𑀤𑀺𑀲𑁂𑀦, 𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑀸, 𑀢𑀤𑀦𑁆𑀢𑀭𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀸𑁇 𑀬𑀤𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀞𑀫𑀁 𑀧𑀲𑀸𑀤𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀢𑀬𑀸, 𑀢𑀸𑀢, 𑀧𑀞𑀫𑀁 𑀧𑀲𑀸𑀤𑀺𑀢𑀁 𑀧𑀘𑁆𑀙𑀸 𑀫𑀬𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸𑀫 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑁂𑀮𑀼𑀯𑀧𑀡𑁆𑀟𑀼𑀯𑀻𑀡𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸—  “𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀦𑁂𑀯 𑀅𑀢𑀺𑀤𑀽𑀭𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀦𑀸𑀘𑁆𑀘𑀸𑀲𑀦𑁆𑀦𑁂, 𑀲𑀤𑁆𑀤𑀜𑁆𑀘 𑀫𑁂 𑀲𑁄𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇
