# 𑁬𑁇𑁩 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀓𑀣𑀸

(300.)

667\. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀬𑀣𑀸𑀲𑀓𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼—  ‘𑀑𑀪𑀸𑀲𑀫𑁂𑀢𑀁 𑀜𑀲𑁆𑀲𑀸𑀫, 𑀬𑀁𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸𑀯 𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀧𑀺 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀬𑀣𑀸𑀲𑀓𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼—  ‘𑀑𑀪𑀸𑀲𑀫𑁂𑀢𑀁 𑀜𑀲𑁆𑀲𑀸𑀫, 𑀬𑀁𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸𑀯 𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀇𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀏𑀓𑀕𑁆𑀕𑀸 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼—  ‘𑀑𑀪𑀸𑀲𑀫𑁂𑀢𑀁 𑀜𑀲𑁆𑀲𑀸𑀫, 𑀬𑀁𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸𑀯 𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇

668\. 𑀬𑀤𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇 𑀬𑁄 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀧𑀓𑀢𑀺𑀯𑀡𑁆𑀡𑁄, 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀦𑀻𑀬𑁄 𑀲𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑀲𑁆𑀫𑀺𑀁𑁇 𑀬𑀤𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀲𑁄 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁄 𑀯𑀡𑁆𑀡𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀫𑀸𑀦𑀼𑀲𑀁 𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀤𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀲𑁄 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑁇 𑀬𑀤𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀓𑁄𑀘𑀺 𑀤𑁂𑀯𑁄 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺—  ‘𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀤𑁂𑀯𑀲𑁆𑀲 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀇𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄, 𑀢𑀲𑁆𑀲 𑀤𑁂𑀯𑀲𑁆𑀲 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀯𑁂𑀤𑀧𑀝𑀺𑀮𑀸𑀪𑀁, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀧𑀝𑀺𑀮𑀸𑀪𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀸𑀚𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑁄 𑀅𑀥𑀼𑀦𑀸𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑁄 𑀭𑀚𑁆𑀚𑁂𑀦, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀯𑁂𑀤𑀧𑀝𑀺𑀮𑀸𑀪𑀁, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀧𑀝𑀺𑀮𑀸𑀪𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑁆𑀲 𑀤𑁂𑀯𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀯𑁂𑀤𑀧𑀝𑀺𑀮𑀸𑀪𑀁, 𑀉𑀍𑀆𑀭𑀁 𑀲𑁄 𑀮𑀪𑀢𑀺 𑀤𑁂𑀯𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀧𑀝𑀺𑀮𑀸𑀪𑀁𑁇 𑀅𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—

669\. _‘𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀯𑀢 𑀪𑁄 𑀤𑁂𑀯𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀳𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀁𑁇_  


670\. _𑀦𑀯𑁂 𑀤𑁂𑀯𑁂 𑀘 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁂 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁂;_  
_𑀲𑀼𑀕𑀢𑀲𑁆𑀫𑀺𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦 𑀇𑀥𑀸𑀕𑀢𑁂𑁇_  


671\. _𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀯𑀡𑁆𑀡𑁂𑀦 𑀬𑀲𑀲𑀸𑀬𑀼𑀦𑀸;_  
_𑀲𑀸𑀯𑀓𑀸 𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀲𑁆𑀲,_  
_𑀯𑀺𑀲𑁂𑀲𑀽𑀧𑀕𑀢𑀸 𑀇𑀥𑁇_  


672\. _𑀇𑀤𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀳𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀦𑁆’𑀢𑀺𑁇_  


(301.)

673\. 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑁇 𑀇𑀫𑀫𑀢𑁆𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀢𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄 𑀘 𑀫𑀜𑁆𑀚𑀼 𑀘 𑀲𑀯𑀦𑀻𑀬𑁄 𑀘 𑀩𑀺𑀦𑁆𑀤𑀼 𑀘 𑀅𑀯𑀺𑀲𑀸𑀭𑀻 𑀘 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀘 𑀦𑀺𑀦𑁆𑀦𑀸𑀤𑀻 𑀘𑁇 𑀬𑀣𑀸𑀧𑀭𑀺𑀲𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀭𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀘𑀲𑁆𑀲 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀖𑁄𑀲𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀺𑁇 𑀬𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑁆𑀲𑀭𑁄’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀲𑀸𑀥𑀼, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑁂, 𑀏𑀢𑀤𑁂𑀯 𑀫𑀬𑀁 𑀲𑀗𑁆𑀔𑀸𑀬 𑀫𑁄𑀤𑀸𑀫; 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀲𑀓𑁆𑀓𑁂𑀦 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁂𑀦 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀝𑁆𑀞 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑀸 𑀯𑀡𑁆𑀡𑀸 𑀪𑀸𑀲𑀺𑀢𑀸; 𑀢𑁂 𑀘 𑀫𑀬𑀁 𑀲𑀗𑁆𑀔𑀸𑀬 𑀫𑁄𑀤𑀸𑀫𑀸’𑀢𑀺𑁇
