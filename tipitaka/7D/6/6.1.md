# 𑁬𑁇𑁧 𑀤𑁂𑀯𑀲𑀪𑀸

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀲𑀪𑀸

(294.)

645\. “𑀧𑀼𑀭𑀺𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀺𑀯𑀲𑀸𑀦𑀺 𑀧𑀼𑀭𑀺𑀫𑀢𑀭𑀸𑀦𑀺 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀼𑀡𑁆𑀡𑀸𑀬 𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀸 𑀘 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀲𑀪𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸; 𑀫𑀳𑀢𑀻 𑀘 𑀤𑀺𑀩𑁆𑀩𑀧𑀭𑀺𑀲𑀸 𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺; 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀥𑀢𑀭𑀝𑁆𑀞𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀪𑀺𑀫𑀼𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀤𑁂𑀯𑁂 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸; 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀯𑀺𑀭𑀽𑀍𑀳𑀓𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀉𑀢𑁆𑀢𑀭𑀸𑀪𑀺𑀫𑀼𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀤𑁂𑀯𑁂 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸; 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀯𑀺𑀭𑀽𑀧𑀓𑁆𑀔𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀧𑀼𑀭𑀢𑁆𑀣𑀸𑀪𑀺𑀫𑀼𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀤𑁂𑀯𑁂 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸; 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀪𑀺𑀫𑀼𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀤𑁂𑀯𑁂 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸𑁇 𑀬𑀤𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀸 𑀘 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀲𑀪𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸, 𑀫𑀳𑀢𑀻 𑀘 𑀤𑀺𑀩𑁆𑀩𑀧𑀭𑀺𑀲𑀸 𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀇𑀤𑀁 𑀦𑁂𑀲𑀁 𑀳𑁄𑀢𑀺 𑀆𑀲𑀦𑀲𑁆𑀫𑀺𑀁; 𑀅𑀣 𑀧𑀘𑁆𑀙𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀲𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

646\. 𑀬𑁂 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀼𑀦𑀽𑀧𑀧𑀦𑁆𑀦𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀓𑀸𑀬𑀁, 𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑁇 𑀢𑁂𑀦 𑀲𑀼𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀫𑀼𑀤𑀺𑀢𑀸 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀸; ‘𑀤𑀺𑀩𑁆𑀩𑀸 𑀯𑀢 𑀪𑁄 𑀓𑀸𑀬𑀸 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀳𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀓𑀸𑀬𑀸’𑀢𑀺𑁇

(295.)

647\. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—

648\. _‘𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀯𑀢 𑀪𑁄 𑀤𑁂𑀯𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀳𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀁𑁇_  


649\. _𑀦𑀯𑁂 𑀤𑁂𑀯𑁂 𑀘 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁂 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁂;_  
_𑀲𑀼𑀕𑀢𑀲𑁆𑀫𑀺𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦 𑀇𑀥𑀸𑀕𑀢𑁂𑁇_  


650\. _𑀢𑁂 𑀅𑀜𑁆𑀜𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀯𑀡𑁆𑀡𑁂𑀦 𑀬𑀲𑀲𑀸𑀬𑀼𑀦𑀸;_  
_𑀲𑀸𑀯𑀓𑀸 𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀲𑁆𑀲,_  
_𑀯𑀺𑀲𑁂𑀲𑀽𑀧𑀕𑀢𑀸 𑀇𑀥𑁇_  


651\. _𑀇𑀤𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀳𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀦𑁆’𑀢𑀺𑁇_  


652\. 𑀢𑁂𑀦 𑀲𑀼𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀫𑀼𑀤𑀺𑀢𑀸 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀸; ‘𑀤𑀺𑀩𑁆𑀩𑀸 𑀯𑀢 𑀪𑁄, 𑀓𑀸𑀬𑀸 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀳𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀓𑀸𑀬𑀸’𑀢𑀺𑁇
