# 𑁬𑁇𑁨 𑀅𑀝𑁆𑀞𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑀯𑀡𑁆𑀡

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀝𑁆𑀞𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑀯𑀡𑁆𑀡

(296.)

653\. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀣 𑀦𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀝𑁆𑀞 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑁂 𑀯𑀡𑁆𑀡𑁂 𑀲𑁄𑀢𑀼𑀦𑁆’𑀢𑀺? ‘𑀇𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁, 𑀫𑀸𑀭𑀺𑀲, 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀝𑁆𑀞 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑁂 𑀯𑀡𑁆𑀡𑁂 𑀲𑁄𑀢𑀼𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀝𑁆𑀞 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑁂 𑀯𑀡𑁆𑀡𑁂 𑀧𑀬𑀺𑀭𑀼𑀤𑀸𑀳𑀸𑀲𑀺—

654\. ‘𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸? 𑀬𑀸𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸𑁇 (𑁧)

655\. 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀺𑁇 𑀏𑀯𑀁 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀤𑁂𑀲𑁂𑀢𑀸𑀭𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸𑁇 (𑁨)

656\. “𑀇𑀤𑀁 𑀓𑀼𑀲𑀮𑀦𑁆”𑀢𑀺 𑀔𑁄 𑀧𑀦 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, “𑀇𑀤𑀁 𑀅𑀓𑀼𑀲𑀮𑀦𑁆”𑀢𑀺 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 “𑀇𑀤𑀁 𑀲𑀸𑀯𑀚𑁆𑀚𑀁 𑀇𑀤𑀁 𑀅𑀦𑀯𑀚𑁆𑀚𑀁, 𑀇𑀤𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀇𑀤𑀁 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀁 𑀳𑀻𑀦𑀁 𑀇𑀤𑀁 𑀧𑀡𑀻𑀢𑀁, 𑀇𑀤𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀦𑁆”𑀢𑀺 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀏𑀯𑀁 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀲𑀸𑀯𑀚𑁆𑀚𑀸𑀦𑀯𑀚𑁆𑀚𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀳𑀻𑀦𑀧𑀡𑀻𑀢𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀸𑀭𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸𑁇 (𑁩)

657\. 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀔𑁄 𑀧𑀦 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀗𑁆𑀕𑁄𑀤𑀓𑀁 𑀬𑀫𑀼𑀦𑁄𑀤𑀓𑁂𑀦 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀫𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀧𑀤𑀸 𑀘𑁇 𑀏𑀯𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀸𑀭𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸𑁇 (𑁪)

658\. 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀦𑁆𑀦𑁄 𑀔𑁄 𑀧𑀦 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀮𑀸𑀪𑁄 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀦𑁆𑀦𑁄 𑀲𑀺𑀮𑁄𑀓𑁄, 𑀬𑀸𑀯 𑀫𑀜𑁆𑀜𑁂 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁆𑀧𑀺𑀬𑀸𑀬𑀫𑀸𑀦𑀭𑀽𑀧𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀯𑀺𑀕𑀢𑀫𑀤𑁄 𑀔𑁄 𑀧𑀦 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀺𑀕𑀢𑀫𑀤𑀁 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀭𑀬𑀫𑀸𑀦𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸𑁇 (𑁫)

659\. 𑀮𑀤𑁆𑀥𑀲𑀳𑀸𑀬𑁄 𑀔𑁄 𑀧𑀦 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑁂𑀔𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀜𑁆𑀘 𑀯𑀼𑀲𑀺𑀢𑀯𑀢𑀁𑁇 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀅𑀧𑀦𑀼𑀚𑁆𑀚 𑀏𑀓𑀸𑀭𑀸𑀫𑀢𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀯𑀁 𑀏𑀓𑀸𑀭𑀸𑀫𑀢𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸𑁇 (𑁬)

660\. 𑀬𑀣𑀸𑀯𑀸𑀤𑀻 𑀔𑁄 𑀧𑀦 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀢𑀣𑀸𑀓𑀸𑀭𑀻, 𑀬𑀣𑀸𑀓𑀸𑀭𑀻 𑀢𑀣𑀸𑀯𑀸𑀤𑀻, 𑀇𑀢𑀺 𑀬𑀣𑀸𑀯𑀸𑀤𑀻 𑀢𑀣𑀸𑀓𑀸𑀭𑀻, 𑀬𑀣𑀸𑀓𑀸𑀭𑀻 𑀢𑀣𑀸𑀯𑀸𑀤𑀻𑁇 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸𑁇 (𑁭)

661\. 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀔𑁄 𑀧𑀦 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑁄 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀚𑁆𑀛𑀸𑀲𑀬𑀁 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁𑁇 𑀏𑀯𑀁 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀅𑀚𑁆𑀛𑀸𑀲𑀬𑀁 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀇𑀫𑀺𑀦𑀸𑀧𑀗𑁆𑀕𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀦𑁂𑀯 𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀧𑀦𑁂𑀢𑀭𑀳𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸’𑀢𑀺𑁇 (𑁮)

(297.)

662\. 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀪𑀕𑀯𑀢𑁄 𑀅𑀝𑁆𑀞 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑁂 𑀯𑀡𑁆𑀡𑁂 𑀧𑀬𑀺𑀭𑀼𑀤𑀸𑀳𑀸𑀲𑀺𑁇 𑀢𑁂𑀦 𑀲𑀼𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀫𑀼𑀤𑀺𑀢𑀸 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀝𑁆𑀞 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑁂 𑀯𑀡𑁆𑀡𑁂 𑀲𑀼𑀢𑁆𑀯𑀸𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀤𑁂𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀳𑁄 𑀯𑀢, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀼𑀁 𑀬𑀣𑀭𑀺𑀯 𑀪𑀕𑀯𑀸𑁇 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀤𑁂𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀼, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀅𑀳𑁄 𑀯𑀢, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀢𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀼𑀁 𑀬𑀣𑀭𑀺𑀯 𑀪𑀕𑀯𑀸𑁇 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀤𑁂𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀼, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀢𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀅𑀳𑁄 𑀯𑀢, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀤𑁆𑀯𑁂 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀼𑀁 𑀬𑀣𑀭𑀺𑀯 𑀪𑀕𑀯𑀸𑁇 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆’𑀢𑀺𑁇

(298.)

663\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀅𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄, 𑀬𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀳𑁄 𑀯𑀢, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑁄 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑁄 𑀘𑀺𑀭𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑁇 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦𑀢𑁆𑀣𑁂𑀦 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀲𑀪𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸, 𑀢𑀁 𑀅𑀢𑁆𑀣𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀁 𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀢𑁆𑀢𑀯𑀘𑀦𑀸𑀧𑀺 𑀢𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀧𑀘𑁆𑀘𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀯𑀘𑀦𑀸𑀧𑀺 𑀢𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀲𑀓𑁂𑀲𑀼 𑀲𑀓𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀞𑀺𑀢𑀸 𑀅𑀯𑀺𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑁇

664\. _𑀢𑁂 𑀯𑀼𑀢𑁆𑀢𑀯𑀸𑀓𑁆𑀬𑀸 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀬𑁆𑀳𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁;_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸 𑀲𑀦𑁆𑀢𑀸,_  
_𑀅𑀝𑁆𑀞𑀁𑀲𑀼 𑀲𑀫𑁆𑀳𑀺 𑀆𑀲𑀦𑁂𑀢𑀺𑁇_  


(299.)

665\. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀉𑀍𑀆𑀭𑁄 𑀆𑀮𑁄𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀬𑀺, 𑀑𑀪𑀸𑀲𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀤𑁂𑀯𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀼𑀪𑀸𑀯𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀬𑀣𑀸 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀉𑀍𑀆𑀭𑁄 𑀆𑀮𑁄𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀬𑀢𑀺, 𑀑𑀪𑀸𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺; 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀳𑁂𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬, 𑀬𑀤𑀺𑀤𑀁 𑀆𑀮𑁄𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀬𑀢𑀺 𑀑𑀪𑀸𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀻𑀢𑀺𑁇

666\. _𑀬𑀣𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀳𑁂𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁,_  
_𑀑𑀪𑀸𑀲𑁄 𑀯𑀺𑀧𑀼𑀮𑁄 𑀫𑀳𑀸’𑀢𑀺𑁇_
