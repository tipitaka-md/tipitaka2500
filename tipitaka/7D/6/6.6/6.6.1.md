# 𑁬𑁇𑁬𑁇𑁧 𑀭𑀚𑁆𑀚𑀲𑀁𑀯𑀺𑀪𑀚𑀦

* 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀯𑀢𑁆𑀣𑀼

* 𑀭𑀚𑁆𑀚𑀲𑀁𑀯𑀺𑀪𑀚𑀦

(306.)

687\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀭𑀸𑀚𑀸 𑀚𑀺𑀡𑁆𑀡𑁄 𑀯𑀼𑀤𑁆𑀥𑁄 𑀫𑀳𑀮𑁆𑀮𑀓𑁄 𑀅𑀤𑁆𑀥𑀕𑀢𑁄 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑁄, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀁? 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺𑀫𑁆𑀳𑀺 𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀭𑀸𑀚𑀓𑀢𑁆𑀢𑀸𑀭𑁄 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀭𑀚𑁆𑀚𑁂 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀆𑀬𑀦𑁆𑀢𑀼, 𑀪𑁄𑀦𑁆𑀢𑁄, 𑀬𑁂𑀦 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀣; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀣—  ‘𑀫𑀬𑀁 𑀔𑁄 𑀪𑁄𑀢𑁄 𑀭𑁂𑀡𑀼𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀸 𑀧𑀺𑀬𑀸 𑀫𑀦𑀸𑀧𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀸, 𑀬𑀁𑀲𑀼𑀔𑁄 𑀪𑀯𑀁 𑀢𑀁𑀲𑀼𑀔𑀸 𑀫𑀬𑀁, 𑀬𑀁𑀤𑀼𑀓𑁆𑀔𑁄 𑀪𑀯𑀁 𑀢𑀁𑀤𑀼𑀓𑁆𑀔𑀸 𑀫𑀬𑀁𑁇 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀭𑀸𑀚𑀸 𑀚𑀺𑀡𑁆𑀡𑁄 𑀯𑀼𑀤𑁆𑀥𑁄 𑀫𑀳𑀮𑁆𑀮𑀓𑁄 𑀅𑀤𑁆𑀥𑀕𑀢𑁄 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑁄, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀁? 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺𑀫𑁆𑀳𑀺 𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀭𑀸𑀚𑀓𑀢𑁆𑀢𑀸𑀭𑁄 𑀪𑀯𑀦𑁆𑀢𑀁 𑀭𑁂𑀡𑀼𑀁 𑀭𑀚𑁆𑀚𑁂 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀲𑀘𑁂 𑀪𑀯𑀁 𑀭𑁂𑀡𑀼 𑀭𑀚𑁆𑀚𑀁 𑀮𑀪𑁂𑀣, 𑀲𑀁𑀯𑀺𑀪𑀚𑁂𑀣 𑀦𑁄 𑀭𑀚𑁆𑀚𑁂𑀦𑀸’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀫𑀬𑀁 𑀔𑁄 𑀪𑁄𑀢𑁄 𑀭𑁂𑀡𑀼𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀸 𑀧𑀺𑀬𑀸 𑀫𑀦𑀸𑀧𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀸; 𑀬𑀁𑀲𑀼𑀔𑁄 𑀪𑀯𑀁 𑀢𑀁𑀲𑀼𑀔𑀸 𑀫𑀬𑀁, 𑀬𑀁𑀤𑀼𑀓𑁆𑀔𑁄 𑀪𑀯𑀁 𑀢𑀁𑀤𑀼𑀓𑁆𑀔𑀸 𑀫𑀬𑀁𑁇 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀭𑀸𑀚𑀸 𑀚𑀺𑀡𑁆𑀡𑁄 𑀯𑀼𑀤𑁆𑀥𑁄 𑀫𑀳𑀮𑁆𑀮𑀓𑁄 𑀅𑀤𑁆𑀥𑀕𑀢𑁄 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦 𑀪𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀁? 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺𑀫𑁆𑀳𑀺 𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀭𑀸𑀚𑀓𑀢𑁆𑀢𑀸𑀭𑁄 𑀪𑀯𑀦𑁆𑀢𑀁 𑀭𑁂𑀡𑀼𑀁 𑀭𑀚𑁆𑀚𑁂 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀲𑀘𑁂 𑀪𑀯𑀁 𑀭𑁂𑀡𑀼 𑀭𑀚𑁆𑀚𑀁 𑀮𑀪𑁂𑀣, 𑀲𑀁𑀯𑀺𑀪𑀚𑁂𑀣 𑀦𑁄 𑀭𑀚𑁆𑀚𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄, 𑀅𑀜𑁆𑀜𑁄 𑀫𑀫 𑀯𑀺𑀚𑀺𑀢𑁂 𑀲𑀼𑀔𑁄 𑀪𑀯𑁂𑀣, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀯𑀦𑁆𑀢𑁂𑀪𑀺? 𑀲𑀘𑀸𑀳𑀁, 𑀪𑁄, 𑀭𑀚𑁆𑀚𑀁 𑀮𑀪𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀁𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄 𑀭𑀚𑁆𑀚𑁂𑀦𑀸”𑀢𑀺𑁇

(307.)

688\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺𑀫𑁆𑀳𑀺 𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀭𑀸𑀚𑀓𑀢𑁆𑀢𑀸𑀭𑁄 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀭𑀚𑁆𑀚𑁂 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑀺𑀁𑀲𑀼𑁇 𑀅𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑁄 𑀭𑁂𑀡𑀼 𑀭𑀚𑁆𑀚𑁂𑀦 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀭𑀸𑀚𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀅𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑁄 𑀭𑁂𑀡𑀼 𑀭𑀚𑁆𑀚𑁂𑀦 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑁄, 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀫𑀤𑀦𑀻𑀬𑀸 𑀓𑀸𑀫𑀸? 𑀆𑀬𑀦𑁆𑀢𑀼, 𑀪𑁄𑀦𑁆𑀢𑁄, 𑀬𑁂𑀦 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀣; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀣—  ‘𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀭𑀸𑀚𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀅𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑁄 𑀪𑀯𑀁 𑀭𑁂𑀡𑀼 𑀭𑀚𑁆𑀚𑁂𑀦, 𑀲𑀭𑀢𑀺 𑀪𑀯𑀁 𑀢𑀁 𑀯𑀘𑀦𑀦𑁆’”𑀢𑀺?

(308.)

689\. “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀭𑀸𑀚𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀅𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑁄 𑀪𑀯𑀁 𑀭𑁂𑀡𑀼 𑀭𑀚𑁆𑀚𑁂𑀦, 𑀲𑀭𑀢𑀺 𑀪𑀯𑀁 𑀢𑀁 𑀯𑀘𑀦𑀦𑁆”𑀢𑀺? “𑀲𑀭𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀢𑀁 𑀯𑀘𑀦𑀁𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄, 𑀧𑀳𑁄𑀢𑀺 𑀇𑀫𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀆𑀬𑀢𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀲𑀓𑀝𑀫𑀼𑀔𑀁 𑀲𑀢𑁆𑀢𑀥𑀸 𑀲𑀫𑀁 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀯𑀺𑀪𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄, 𑀅𑀜𑁆𑀜𑁄 𑀧𑀳𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁂𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀦𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀬𑁂𑀦 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀺—  ‘𑀭𑀸𑀚𑀸 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑁂𑀡𑀼 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑁂𑀡𑀼𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀭𑀸𑀚𑀸 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑁂𑀡𑀼 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑁂𑀡𑀼𑀦𑀸 𑀭𑀜𑁆𑀜𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀢𑀼, 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀇𑀫𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀆𑀬𑀢𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀲𑀓𑀝𑀫𑀼𑀔𑀁 𑀲𑀢𑁆𑀢𑀥𑀸 𑀲𑀫𑀁 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀯𑀺𑀪𑀚𑀢𑀽”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀪𑁄 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀭𑁂𑀡𑀼𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀇𑀫𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀆𑀬𑀢𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀲𑀓𑀝𑀫𑀼𑀔𑀁 𑀲𑀢𑁆𑀢𑀥𑀸 𑀲𑀫𑀁 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀯𑀺𑀪𑀚𑀺𑁇 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀲𑀓𑀝𑀫𑀼𑀔𑀸𑀦𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀫𑀚𑁆𑀛𑁂 𑀭𑁂𑀡𑀼𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀚𑀦𑀧𑀤𑁄 𑀳𑁄𑀢𑀺𑁇

(309.)

690\. _𑀤𑀦𑁆𑀢𑀧𑀼𑀭𑀁 𑀓𑀮𑀺𑀗𑁆𑀕𑀸𑀦𑀁,_  
_𑀅𑀲𑁆𑀲𑀓𑀸𑀦𑀜𑁆𑀘 𑀧𑁄𑀢𑀦𑀁;_  
_𑀫𑀳𑁂𑀲𑀬𑀁 𑀅𑀯𑀦𑁆𑀢𑀻𑀦𑀁,_  
_𑀲𑁄𑀯𑀻𑀭𑀸𑀦𑀜𑁆𑀘 𑀭𑁄𑀭𑀼𑀓𑀁𑁇_  


691\. _𑀫𑀺𑀣𑀺𑀮𑀸 𑀘 𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁,_  
_𑀘𑀫𑁆𑀧𑀸 𑀅𑀗𑁆𑀕𑁂𑀲𑀼 𑀫𑀸𑀧𑀺𑀢𑀸;_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀘 𑀓𑀸𑀲𑀻𑀦𑀁,_  
_𑀏𑀢𑁂 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀫𑀸𑀧𑀺𑀢𑀸𑀢𑀺𑁇_  


(310.)

692\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀬𑀣𑀸𑀲𑀓𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸—  “𑀬𑀁 𑀯𑀢 𑀦𑁄 𑀅𑀳𑁄𑀲𑀺 𑀇𑀘𑁆𑀙𑀺𑀢𑀁, 𑀬𑀁 𑀆𑀓𑀗𑁆𑀔𑀺𑀢𑀁, 𑀬𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀁, 𑀬𑀁 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑀁, 𑀢𑀁 𑀦𑁄 𑀮𑀤𑁆𑀥𑀦𑁆”𑀢𑀺𑁇

693\. _𑀲𑀢𑁆𑀢𑀪𑀽 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀘,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀽 𑀪𑀭𑀢𑁄 𑀲𑀳;_  
_𑀭𑁂𑀡𑀼 𑀤𑁆𑀯𑁂 𑀥𑀢𑀭𑀝𑁆𑀞𑀸 𑀘,_  
_𑀢𑀤𑀸𑀲𑀼𑀁 𑀲𑀢𑁆𑀢 𑀪𑀸𑀭𑀥𑀸𑀢𑀺𑁇_
