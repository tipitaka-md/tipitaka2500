# 𑁬𑁇𑁬𑁇𑁭 𑀪𑀭𑀺𑀬𑀸𑀦𑀁𑀆𑀫𑀦𑁆𑀢𑀦𑀸

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀯𑀢𑁆𑀣𑀼

### 𑀪𑀭𑀺𑀬𑀸𑀦𑀁𑀆𑀫𑀦𑁆𑀢𑀦𑀸

(325.)

737\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸 𑀪𑀭𑀺𑀬𑀸 𑀲𑀸𑀤𑀺𑀲𑀺𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸 𑀪𑀭𑀺𑀬𑀸 𑀲𑀸𑀤𑀺𑀲𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀸 𑀪𑁄𑀢𑀻𑀦𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑀓𑀸𑀦𑀺 𑀯𑀸 𑀜𑀸𑀢𑀺𑀓𑀼𑀮𑀸𑀦𑀺 𑀕𑀘𑁆𑀙𑀢𑀼 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀪𑀢𑁆𑀢𑀸𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀼𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄𑀢𑀻, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸𑁇 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄𑀢𑀻, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁆𑀯𑀜𑁆𑀜𑁂𑀯 𑀦𑁄 𑀜𑀸𑀢𑀺 𑀜𑀸𑀢𑀺𑀓𑀸𑀫𑀸𑀦𑀁, 𑀢𑁆𑀯𑀁 𑀧𑀦 𑀪𑀢𑁆𑀢𑀸 𑀪𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑀸𑀦𑀁𑁇 𑀲𑀘𑁂 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇
