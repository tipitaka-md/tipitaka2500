# 𑁬𑁇𑁬𑁇𑁫 𑀙𑀔𑀢𑁆𑀢𑀺𑀬𑀆𑀫𑀦𑁆𑀢𑀦𑀸

* 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀯𑀢𑁆𑀣𑀼

* 𑀙𑀔𑀢𑁆𑀢𑀺𑀬𑀆𑀫𑀦𑁆𑀢𑀦𑀸

(322.)

726\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀜𑁆𑀜𑀁 𑀤𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀼, 𑀬𑁄 𑀪𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀭𑀚𑁆𑀚𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸𑁇 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀓𑁆𑀓𑀫𑁆𑀫 𑀏𑀯𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁—  “𑀇𑀫𑁂 𑀔𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀦𑀸𑀫 𑀥𑀦𑀮𑀼𑀤𑁆𑀥𑀸; 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀥𑀦𑁂𑀦 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀇𑀫𑁂𑀲𑀼 𑀲𑀢𑁆𑀢𑀲𑀼 𑀭𑀚𑁆𑀚𑁂𑀲𑀼 𑀧𑀳𑀽𑀢𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁, 𑀢𑀢𑁄 𑀪𑁄𑀢𑁄 𑀬𑀸𑀯𑀢𑀓𑁂𑀦 𑀅𑀢𑁆𑀣𑁄, 𑀢𑀸𑀯𑀢𑀓𑀁 𑀆𑀳𑀭𑀻𑀬𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀪𑁄, 𑀫𑀫𑀧𑀺𑀤𑀁 𑀧𑀳𑀽𑀢𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀪𑀯𑀦𑁆𑀢𑀸𑀦𑀁𑀬𑁂𑀯 𑀯𑀸𑀳𑀲𑀸𑁇 𑀢𑀫𑀳𑀁 𑀲𑀩𑁆𑀩𑀁 𑀧𑀳𑀸𑀬 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀓𑁆𑀓𑀫𑁆𑀫 𑀏𑀯𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁—  “𑀇𑀫𑁂 𑀔𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀦𑀸𑀫 𑀇𑀢𑁆𑀣𑀺𑀮𑀼𑀤𑁆𑀥𑀸; 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀇𑀢𑁆𑀣𑀻𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀔𑁄, 𑀪𑁄, 𑀇𑀫𑁂𑀲𑀼 𑀲𑀢𑁆𑀢𑀲𑀼 𑀭𑀚𑁆𑀚𑁂𑀲𑀼 𑀧𑀳𑀽𑀢𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄, 𑀢𑀢𑁄 𑀪𑁄𑀢𑁄 𑀬𑀸𑀯𑀢𑀺𑀓𑀸𑀳𑀺 𑀅𑀢𑁆𑀣𑁄, 𑀢𑀸𑀯𑀢𑀺𑀓𑀸 𑀆𑀦𑀻𑀬𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀪𑁄, 𑀫𑀫𑀧𑀺𑀫𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸 𑀪𑀭𑀺𑀬𑀸 𑀲𑀸𑀤𑀺𑀲𑀺𑀬𑁄𑁇 𑀢𑀸𑀧𑀸𑀳𑀁 𑀲𑀩𑁆𑀩𑀸 𑀧𑀳𑀸𑀬 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆𑀢𑀺”𑁇

(323.)

727\. “𑀲𑀘𑁂 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

728\. _“𑀲𑀘𑁂 𑀚𑀳𑀣 𑀓𑀸𑀫𑀸𑀦𑀺,_  
_𑀬𑀢𑁆𑀣 𑀲𑀢𑁆𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄;_  
_𑀆𑀭𑀫𑁆𑀪𑀯𑁆𑀳𑁄 𑀤𑀍𑀳𑀸 𑀳𑁄𑀣,_  
_𑀔𑀦𑁆𑀢𑀻𑀩𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑀸𑁇_  


729\. _𑀏𑀲 𑀫𑀕𑁆𑀕𑁄 𑀉𑀚𑀼𑀫𑀕𑁆𑀕𑁄,_  
_𑀏𑀲 𑀫𑀕𑁆𑀕𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄 𑀲𑀩𑁆𑀪𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇_  


730\. “𑀢𑁂𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

731\. “𑀅𑀢𑀺𑀘𑀺𑀭𑀁 𑀔𑁄, 𑀪𑁄, 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺, 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺, 𑀪𑀯𑀦𑁆𑀢𑁂, 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼𑀁𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑁄, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀸𑀦𑀁𑁇 𑀕𑀫𑀦𑀻𑀬𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄, 𑀫𑀦𑁆𑀢𑀸𑀬𑀁 𑀩𑁄𑀤𑁆𑀥𑀩𑁆𑀩𑀁, 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀚𑀸𑀢𑀲𑁆𑀲 𑀅𑀫𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀢𑀻𑀡𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀆𑀕𑀫𑁂𑀢𑀼, 𑀏𑀓𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

732\. “𑀅𑀢𑀺𑀘𑀺𑀭𑀁 𑀔𑁄, 𑀪𑁄, 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁, 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀪𑀯𑀦𑁆𑀢𑁂 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀆𑀕𑀫𑁂𑀢𑀼𑀁𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑁄, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀸𑀦𑀁𑁇 𑀕𑀫𑀦𑀻𑀬𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄, 𑀫𑀦𑁆𑀢𑀸𑀬𑀁 𑀩𑁄𑀤𑁆𑀥𑀩𑁆𑀩𑀁, 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀚𑀸𑀢𑀲𑁆𑀲 𑀅𑀫𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀲𑀢𑁆𑀢 𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼, 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

733\. “𑀅𑀢𑀺𑀘𑀺𑀭𑀁 𑀔𑁄, 𑀪𑁄, 𑀲𑀢𑁆𑀢 𑀫𑀸𑀲𑀸𑀦𑀺, 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀪𑀯𑀦𑁆𑀢𑁂 𑀲𑀢𑁆𑀢 𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼𑀁𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑁄, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀸𑀦𑀁𑁇 𑀕𑀫𑀦𑀻𑀬𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄, 𑀫𑀦𑁆𑀢𑀸𑀬𑀁 𑀩𑁄𑀤𑁆𑀥𑀩𑁆𑀩𑀁, 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀚𑀸𑀢𑀲𑁆𑀲 𑀅𑀫𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇

734\. “𑀢𑁂𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀙 𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀢𑀻𑀡𑀺 𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀤𑁆𑀯𑁂 𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀏𑀓𑀁 𑀫𑀸𑀲𑀁 𑀆𑀕𑀫𑁂𑀢𑀼…  𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀆𑀕𑀫𑁂𑀢𑀼, 𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

735\. “𑀅𑀢𑀺𑀘𑀺𑀭𑀁 𑀔𑁄, 𑀪𑁄, 𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑁄, 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀪𑀯𑀦𑁆𑀢𑁂 𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀆𑀕𑀫𑁂𑀢𑀼𑀁𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑁄, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀸𑀦𑀁𑁇 𑀕𑀫𑀦𑀻𑀬𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄, 𑀫𑀦𑁆𑀢𑀸𑀬𑀁 𑀩𑁄𑀤𑁆𑀥𑀩𑁆𑀩𑀁, 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀚𑀸𑀢𑀲𑁆𑀲 𑀅𑀫𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀆𑀕𑀫𑁂𑀢𑀼, 𑀬𑀸𑀯 𑀫𑀬𑀁 𑀲𑀓𑁂 𑀧𑀼𑀢𑁆𑀢𑀪𑀸𑀢𑀭𑁄 𑀭𑀚𑁆𑀚𑁂𑀦 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫, 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀦 𑀘𑀺𑀭𑀁 𑀔𑁄, 𑀪𑁄, 𑀲𑀢𑁆𑀢𑀸𑀳𑀁, 𑀆𑀕𑀫𑁂𑀲𑁆𑀲𑀸𑀫𑀳𑀁 𑀪𑀯𑀦𑁆𑀢𑁂 𑀲𑀢𑁆𑀢𑀸𑀳𑀦𑁆”𑀢𑀺𑁇
