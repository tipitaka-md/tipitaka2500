# 𑁬𑁇𑁬𑁇𑁨 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀫𑀦

* 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀯𑀢𑁆𑀣𑀼

* 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀫𑀦

(311.)

695\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀬𑁂𑀦 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀬𑀣𑀸 𑀔𑁄 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀭𑁂𑀡𑀼𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀲𑀳𑀸𑀬𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀲𑀳𑀸𑀬𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑁄, 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀼 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄; 𑀫𑀸 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀩𑁆𑀬𑀸𑀳𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑁂𑀲𑀁 𑀙𑀦𑁆𑀦𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀢𑁆𑀢 𑀘 𑀭𑀸𑀚𑀸𑀦𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑁂 𑀭𑀚𑁆𑀚𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀺, 𑀲𑀢𑁆𑀢 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁂 𑀲𑀢𑁆𑀢 𑀘 𑀦𑁆𑀳𑀸𑀢𑀓𑀲𑀢𑀸𑀦𑀺 𑀫𑀦𑁆𑀢𑁂 𑀯𑀸𑀘𑁂𑀲𑀺𑁇

(312.)

696\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀺—  “𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀫𑀺𑁇 𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀢𑀺, 𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑁂𑀬𑁆𑀬𑀁, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

(313.)

697\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄, 𑀪𑁄, 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀫𑀺𑁇 𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀢𑀺, 𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀺𑀢𑀼𑀁; 𑀦𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

(314.)

698\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀙 𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄, 𑀪𑁄, 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀫𑀺𑁇 𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁, ‘𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀢𑀺, 𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀺𑀢𑀼𑀁; 𑀦𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

699\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀲𑀢𑁆𑀢 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀲𑀢𑁆𑀢 𑀘 𑀦𑁆𑀳𑀸𑀢𑀓𑀲𑀢𑀸𑀦𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁂 𑀲𑀢𑁆𑀢 𑀘 𑀦𑁆𑀳𑀸𑀢𑀓𑀲𑀢𑀸𑀦𑀺 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄, 𑀪𑁄, 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀫𑀺𑁇 𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀢𑀺, 𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑁄, 𑀬𑀣𑀸𑀲𑀼𑀢𑁂 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑁂 𑀫𑀦𑁆𑀢𑁂 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀚𑁆𑀛𑀸𑀬𑀁 𑀓𑀭𑁄𑀣, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀜𑁆𑀘 𑀫𑀦𑁆𑀢𑁂 𑀯𑀸𑀘𑁂𑀣; 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀺𑀢𑀼𑀁; 𑀦𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

(316.)

700\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸 𑀪𑀭𑀺𑀬𑀸 𑀲𑀸𑀤𑀺𑀲𑀺𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸 𑀪𑀭𑀺𑀬𑀸 𑀲𑀸𑀤𑀺𑀲𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄, 𑀪𑁄𑀢𑀻, 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀓𑁆𑀔𑀺 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑁄𑀢𑀻, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀫𑀺𑁇 𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁 ‘𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀢𑀺, 𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺, 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄𑀢𑀻, 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀺𑀢𑀼𑀁; 𑀦𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

(317.)

701\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂𑀦 𑀦𑀕𑀭𑀲𑁆𑀲 𑀦𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀕𑀸𑀭𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀺; 𑀦𑀸𑀲𑁆𑀲𑀼𑀥 𑀓𑁄𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀅𑀳𑀼𑀤𑁂𑀯 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀦𑀸 𑀅𑀳𑀼 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀦𑀸—  “𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀓𑀭𑀼𑀡𑀁 𑀛𑀸𑀦𑀁 𑀛𑀸𑀬𑀢𑀺, 𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀸𑀓𑀘𑁆𑀙𑁂𑀫𑀺 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺 𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀫𑀦𑁆𑀢𑁂𑀫𑀻”𑀢𑀺𑁇
