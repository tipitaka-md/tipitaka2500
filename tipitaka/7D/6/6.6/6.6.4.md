# 𑁬𑁇𑁬𑁇𑁪 𑀭𑁂𑀡𑀼𑀭𑀸𑀚𑀆𑀫𑀦𑁆𑀢𑀦𑀸

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀯𑀢𑁆𑀣𑀼

### 𑀭𑁂𑀡𑀼𑀭𑀸𑀚𑀆𑀫𑀦𑁆𑀢𑀦𑀸

(321.)

716\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀭𑁂𑀡𑀼 𑀭𑀸𑀚𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑁂𑀡𑀼𑀁 𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀜𑁆𑀜𑀁 𑀤𑀸𑀦𑀺 𑀪𑀯𑀁 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀼, 𑀬𑁄 𑀪𑁄𑀢𑁄 𑀭𑀚𑁆𑀚𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸𑁇 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇

717\. _“𑀆𑀫𑀦𑁆𑀢𑀬𑀸𑀫𑀺 𑀭𑀸𑀚𑀸𑀦𑀁,_  
_𑀭𑁂𑀡𑀼𑀁 𑀪𑀽𑀫𑀺𑀧𑀢𑀺𑀁 𑀅𑀳𑀁;_  
_𑀢𑁆𑀯𑀁 𑀧𑀚𑀸𑀦𑀲𑁆𑀲𑀼 𑀭𑀚𑁆𑀚𑁂𑀦,_  
_𑀦𑀸𑀳𑀁 𑀧𑁄𑀭𑁄𑀳𑀺𑀘𑁆𑀘𑁂 𑀭𑀫𑁂”𑁇_  


718\. _“𑀲𑀘𑁂 𑀢𑁂 𑀊𑀦𑀁 𑀓𑀸𑀫𑁂𑀳𑀺,_  
_𑀅𑀳𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑀬𑀸𑀫𑀺 𑀢𑁂;_  
_𑀬𑁄 𑀢𑀁 𑀳𑀺𑀁𑀲𑀢𑀺 𑀯𑀸𑀭𑁂𑀫𑀺,_  
_𑀪𑀽𑀫𑀺𑀲𑁂𑀦𑀸𑀧𑀢𑀺 𑀅𑀳𑀁;_  
_𑀢𑀼𑀯𑀁 𑀧𑀺𑀢𑀸 𑀅𑀳𑀁 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀫𑀸 𑀦𑁄 𑀕𑁄𑀯𑀺𑀦𑁆𑀤 𑀧𑀸𑀚𑀳𑀺”𑁇_  


719\. _“𑀦𑀫𑀢𑁆𑀣𑀺 𑀊𑀦𑀁 𑀓𑀸𑀫𑁂𑀳𑀺,_  
_𑀳𑀺𑀁𑀲𑀺𑀢𑀸 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀘𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀢𑀲𑁆𑀫𑀸𑀳𑀁 𑀦 𑀕𑀳𑁂 𑀭𑀫𑁂”𑁇_  


720\. _“𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀓𑀣𑀁𑀯𑀡𑁆𑀡𑁄,_  
_𑀓𑀺𑀁 𑀢𑁂 𑀅𑀢𑁆𑀣𑀁 𑀅𑀪𑀸𑀲𑀣;_  
_𑀬𑀜𑁆𑀘 𑀲𑀼𑀢𑁆𑀯𑀸 𑀚𑀳𑀸𑀲𑀺 𑀦𑁄,_  
_𑀕𑁂𑀳𑁂 𑀅𑀫𑁆𑀳𑁂 𑀘 𑀓𑁂𑀯𑀮𑀻”𑁇_  


721\. _“𑀉𑀧𑀯𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑁂 𑀧𑀼𑀩𑁆𑀩𑁂,_  
_𑀬𑀺𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄;_  
_𑀅𑀕𑁆𑀕𑀺 𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀆𑀲𑀺,_  
_𑀓𑀼𑀲𑀧𑀢𑁆𑀢𑀧𑀭𑀺𑀢𑁆𑀣𑀢𑁄𑁇_  


722\. _𑀢𑀢𑁄 𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀧𑀸𑀢𑀼𑀭𑀳𑀼,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸 𑀲𑀦𑀦𑁆𑀢𑀦𑁄;_  
_𑀲𑁄 𑀫𑁂 𑀧𑀜𑁆𑀳𑀁 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀕𑀳𑁂 𑀭𑀫𑁂”𑁇_  


723\. _“𑀲𑀤𑁆𑀤𑀳𑀸𑀫𑀺 𑀅𑀳𑀁 𑀪𑁄𑀢𑁄,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤 𑀪𑀸𑀲𑀲𑀺;_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀘𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀓𑀣𑀁 𑀯𑀢𑁆𑀢𑁂𑀣 𑀅𑀜𑁆𑀜𑀣𑀸𑁇_  


724\. _𑀢𑁂 𑀢𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀲𑀢𑁆𑀣𑀸 𑀕𑁄𑀯𑀺𑀦𑁆𑀤 𑀦𑁄 𑀪𑀯𑀁;_  
_𑀫𑀡𑀺 𑀬𑀣𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑁄,_  
_𑀅𑀓𑀸𑀘𑁄 𑀯𑀺𑀫𑀮𑁄 𑀲𑀼𑀪𑁄;_  
_𑀏𑀯𑀁 𑀲𑀼𑀤𑁆𑀥𑀸 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑁂𑀢𑀺𑁇_  


725\. 𑀲𑀘𑁂 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇
