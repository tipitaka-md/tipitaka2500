# 𑁬𑁇𑁬𑁇𑁬 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀤𑀻𑀦𑀁𑀆𑀫𑀦𑁆𑀢𑀦𑀸

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀯𑀢𑁆𑀣𑀼

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀤𑀻𑀦𑀁𑀆𑀫𑀦𑁆𑀢𑀦𑀸

(324.)

736\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀲𑀢𑁆𑀢 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀲𑀢𑁆𑀢 𑀘 𑀦𑁆𑀳𑀸𑀢𑀓𑀲𑀢𑀸𑀦𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁂 𑀲𑀢𑁆𑀢 𑀘 𑀦𑁆𑀳𑀸𑀢𑀓𑀲𑀢𑀸𑀦𑀺 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀜𑁆𑀜𑀁 𑀤𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑁄 𑀆𑀘𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀼, 𑀬𑁄 𑀪𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀫𑀦𑁆𑀢𑁂 𑀯𑀸𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲𑁇 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀫𑀸 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑁇 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀪𑁄, 𑀅𑀧𑁆𑀧𑁂𑀲𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑁆𑀧𑀮𑀸𑀪𑀸 𑀘; 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀜𑁆𑀘 𑀫𑀳𑀸𑀮𑀸𑀪𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 “𑀫𑀸 𑀪𑀯𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀅𑀯𑀘𑀼𑀢𑁆𑀣—  ‘𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀅𑀧𑁆𑀧𑁂𑀲𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑁆𑀧𑀮𑀸𑀪𑀸 𑀘, 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀜𑁆𑀘 𑀫𑀳𑀸𑀮𑀸𑀪𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀫𑀬𑀸 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀮𑀸𑀪𑀢𑀭𑁄 𑀯𑀸𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀪𑁄, 𑀏𑀢𑀭𑀳𑀺 𑀭𑀸𑀚𑀸𑀯 𑀭𑀜𑁆𑀜𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀯 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀤𑁂𑀯𑀢𑀸𑀯 𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁𑁇 𑀢𑀫𑀳𑀁 𑀲𑀩𑁆𑀩𑀁 𑀧𑀳𑀸𑀬 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲, 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸𑁇 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀘𑁂 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫, 𑀅𑀣 𑀬𑀸 𑀢𑁂 𑀕𑀢𑀺, 𑀲𑀸 𑀦𑁄 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇
