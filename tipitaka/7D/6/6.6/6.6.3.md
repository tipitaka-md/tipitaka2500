# 𑁬𑁇𑁬𑁇𑁩 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸𑀲𑀸𑀓𑀘𑁆𑀙𑀸

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀯𑀢𑁆𑀣𑀼

### 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸𑀲𑀸𑀓𑀘𑁆𑀙𑀸

(318.)

702\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀳𑀼𑀤𑁂𑀯 𑀪𑀬𑀁 𑀅𑀳𑀼 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀳𑀼 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀻𑀢𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑁄 𑀮𑁄𑀫𑀳𑀝𑁆𑀞𑀚𑀸𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

703\. _“𑀯𑀡𑁆𑀡𑀯𑀸 𑀬𑀲𑀯𑀸 𑀲𑀺𑀭𑀺𑀫𑀸,_  
_𑀓𑁄 𑀦𑀼 𑀢𑁆𑀯𑀫𑀲𑀺 𑀫𑀸𑀭𑀺𑀲;_  
_𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇_  


704\. _“𑀫𑀁 𑀯𑁂 𑀓𑀼𑀫𑀸𑀭𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀲𑀦𑀦𑁆𑀢𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀫𑀁 𑀤𑁂𑀯𑀸,_  
_𑀏𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤 𑀚𑀸𑀦𑀳𑀺”𑁇_  


705\. _“𑀆𑀲𑀦𑀁 𑀉𑀤𑀓𑀁 𑀧𑀚𑁆𑀚𑀁,_  
_𑀫𑀥𑀼𑀲𑀸𑀓𑀜𑁆𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄;_  
_𑀅𑀕𑁆𑀖𑁂 𑀪𑀯𑀦𑁆𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀅𑀕𑁆𑀖𑀁 𑀓𑀼𑀭𑀼𑀢𑀼 𑀦𑁄 𑀪𑀯𑀁”𑁇_  


706\. _“𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫 𑀢𑁂 𑀅𑀕𑁆𑀖𑀁,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤 𑀪𑀸𑀲𑀲𑀺;_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀳𑀺𑀢𑀢𑁆𑀣𑀸𑀬,_  
_𑀲𑀫𑁆𑀧𑀭𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀘;_  
_𑀓𑀢𑀸𑀯𑀓𑀸𑀲𑁄 𑀧𑀼𑀘𑁆𑀙𑀲𑁆𑀲𑀼,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑀦𑁆”𑀢𑀺𑁇_  


(319.)

707\. 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀢𑀸𑀯𑀓𑀸𑀲𑁄 𑀔𑁄𑀫𑁆𑀳𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁂𑀦𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀸 𑀅𑀢𑁆𑀣𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁 𑀯𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀼𑀲𑀮𑁄 𑀔𑁄 𑀅𑀳𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀦𑀁, 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀅𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀜𑁆𑀜𑁂𑀯 𑀅𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑁄, 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

708\. _“𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀁,_  
_𑀓𑀗𑁆𑀔𑀻 𑀅𑀓𑀗𑁆𑀔𑀺𑀁 𑀧𑀭𑀯𑁂𑀤𑀺𑀬𑁂𑀲𑀼;_  
_𑀓𑀢𑁆𑀣𑀝𑁆𑀞𑀺𑀢𑁄 𑀓𑀺𑀫𑁆𑀳𑀺 𑀘 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑁄,_  
_𑀧𑀧𑁆𑀧𑁄𑀢𑀺 𑀫𑀘𑁆𑀘𑁄 𑀅𑀫𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀦𑁆”𑀢𑀺𑁇_  


709\. _“𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀫𑀢𑁆𑀢𑀁 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀏𑀓𑁄𑀤𑀺𑀪𑀽𑀢𑁄 𑀓𑀭𑀼𑀡𑁂𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄;_  
_𑀦𑀺𑀭𑀸𑀫𑀕𑀦𑁆𑀥𑁄 𑀯𑀺𑀭𑀢𑁄 𑀫𑁂𑀣𑀼𑀦𑀲𑁆𑀫𑀸,_  
_𑀏𑀢𑁆𑀣𑀝𑁆𑀞𑀺𑀢𑁄 𑀏𑀢𑁆𑀣 𑀘 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑁄;_  
_𑀧𑀧𑁆𑀧𑁄𑀢𑀺 𑀫𑀘𑁆𑀘𑁄 𑀅𑀫𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀦𑁆”𑀢𑀺𑁇_  


(320.)

710\. “‘𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀫𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀅𑀧𑁆𑀧𑀁 𑀯𑀸 𑀪𑁄𑀕𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀳𑀸𑀬 𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀸 𑀪𑁄𑀕𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀳𑀸𑀬 𑀅𑀧𑁆𑀧𑀁 𑀯𑀸 𑀜𑀸𑀢𑀺𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀁 𑀧𑀳𑀸𑀬 𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀸 𑀜𑀸𑀢𑀺𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀁 𑀧𑀳𑀸𑀬 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀢𑀺, 𑀇𑀢𑀺 ‘𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀫𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇

711\. ‘𑀏𑀓𑁄𑀤𑀺𑀪𑀽𑀢𑁄’𑀢𑀺 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀚𑀢𑀺 𑀅𑀭𑀜𑁆𑀜𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀓𑀦𑁆𑀤𑀭𑀁 𑀕𑀺𑀭𑀺𑀕𑀼𑀳𑀁 𑀲𑀼𑀲𑀸𑀦𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀁 𑀧𑀮𑀸𑀮𑀧𑀼𑀜𑁆𑀚𑀁, 𑀇𑀢𑀺 ‘𑀏𑀓𑁄𑀤𑀺𑀪𑀽𑀢𑁄’𑀢𑀺 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇

712\. ‘𑀓𑀭𑀼𑀡𑁂𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄’𑀢𑀺 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀭𑀼𑀡𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀓𑀭𑀼𑀡𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀢𑀺 ‘𑀓𑀭𑀼𑀡𑁂𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄’𑀢𑀺 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀘 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲 𑀦 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇

713\. _𑀓𑁂 𑀆𑀫𑀕𑀦𑁆𑀥𑀸 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀏𑀢𑁂 𑀅𑀯𑀺𑀤𑁆𑀯𑀸 𑀇𑀥 𑀩𑁆𑀭𑀽𑀳𑀺 𑀥𑀻𑀭;_  
_𑀓𑁂𑀦𑀸𑀯𑀝𑀸 𑀯𑀸𑀢𑀺 𑀧𑀚𑀸 𑀓𑀼𑀭𑀼𑀢𑀼,_  
_𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑀺𑀯𑀼𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸”𑀢𑀺𑁇_  


714\. _“𑀓𑁄𑀥𑁄 𑀫𑁄𑀲𑀯𑀚𑁆𑀚𑀁 𑀦𑀺𑀓𑀢𑀺 𑀘 𑀤𑀼𑀩𑁆𑀪𑁄,_  
_𑀓𑀤𑀭𑀺𑀬𑀢𑀸 𑀅𑀢𑀺𑀫𑀸𑀦𑁄 𑀉𑀲𑀽𑀬𑀸;_  
_𑀇𑀘𑁆𑀙𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀙𑀸 𑀧𑀭𑀳𑁂𑀞𑀦𑀸 𑀘,_  
_𑀮𑁄𑀪𑁄 𑀘 𑀤𑁄𑀲𑁄 𑀘 𑀫𑀤𑁄 𑀘 𑀫𑁄𑀳𑁄;_  
_𑀏𑀢𑁂𑀲𑀼 𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀺𑀭𑀸𑀫𑀕𑀦𑁆𑀥𑀸,_  
_𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑀺𑀯𑀼𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸”𑀢𑀺𑁇_  


715\. “𑀬𑀣𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑁄𑀢𑁄, 𑀆𑀫𑀕𑀦𑁆𑀥𑁂 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀢𑁂 𑀦 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀤𑀬𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸𑁇 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀪𑀯𑀁 𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑁄 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇
