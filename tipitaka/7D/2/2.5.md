# 𑁨𑁇𑁫 𑀲𑀢𑁆𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺

### 𑀫𑀳𑀸𑀦𑀺𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺

(127.)

236\. 𑀲𑀢𑁆𑀢 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑁄, 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸 𑀲𑀢𑁆𑀢? 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘 𑀤𑁂𑀯𑀸, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀺𑀓𑀸𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀏𑀓𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀤𑁂𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸 𑀧𑀞𑀫𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀸 𑀏𑀓𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀤𑁂𑀯𑀸 𑀆𑀪𑀲𑁆𑀲𑀭𑀸𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀸 𑀏𑀓𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀏𑀓𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀤𑁂𑀯𑀸 𑀲𑀼𑀪𑀓𑀺𑀡𑁆𑀳𑀸𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀫𑁂𑀯 𑀤𑀼𑀢𑀺𑀬𑀁𑁇

(128.)

237\. 𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑀸𑀬𑀁 𑀧𑀞𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺 𑀦𑀸𑀦𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘 𑀤𑁂𑀯𑀸, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀺𑀓𑀸𑁇 𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸 𑀘 𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸 𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸 𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸 𑀘 𑀆𑀤𑀻𑀦𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸 𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑁂𑀦 𑀢𑀤𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  “𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑀫𑀺𑀤𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀆𑀤𑀻𑀦𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑁂𑀦 𑀢𑀤𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑀫𑀺𑀤𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁𑁇 𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀆𑀤𑀻𑀦𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑁂𑀦 𑀢𑀤𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀢𑁄 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑀸𑀲𑀜𑁆𑀘 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀻𑀦𑀁 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀆𑀬𑀢𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀤𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀜𑁆𑀘 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄𑁇
