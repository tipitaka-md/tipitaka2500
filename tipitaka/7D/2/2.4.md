# 𑁨𑁇𑁪 𑀅𑀢𑁆𑀢𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

* 𑀫𑀳𑀸𑀦𑀺𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀢𑁆𑀢𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

(121.)

230\. 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀘, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺? 𑀯𑁂𑀤𑀦𑀁 𑀯𑀸 𑀳𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺—  ‘𑀯𑁂𑀤𑀦𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺𑁇 ‘𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀢𑀸, 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀦𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸 𑀳𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇 ‘𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀢𑀸, 𑀦𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀦𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀸 𑀫𑁂 𑀯𑁂𑀤𑀺𑀬𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀥𑀫𑁆𑀫𑁄 𑀳𑀺 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀇𑀢𑀺 𑀯𑀸 𑀳𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇

(122.)

231\. 𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑁄 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀯𑁂𑀤𑀦𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀢𑀺𑀲𑁆𑀲𑁄 𑀔𑁄 𑀇𑀫𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀇𑀫𑀸𑀲𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁 𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀲𑀻’𑀢𑀺? 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀦 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺; 𑀲𑀼𑀔𑀁𑀬𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀦 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺; 𑀤𑀼𑀓𑁆𑀔𑀁𑀬𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀦𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺; 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁𑀬𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇

(123.)

232\. 𑀲𑀼𑀔𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀧𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀬𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀲𑁆𑀲 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲 ‘𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸 ‘𑀩𑁆𑀬𑀕𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲 ‘𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸 ‘𑀩𑁆𑀬𑀕𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲 ‘𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸 ‘𑀩𑁆𑀬𑀕𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀇𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀯𑁄𑀓𑀺𑀡𑁆𑀡𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀬𑀥𑀫𑁆𑀫𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀬𑁄 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀯𑁂𑀤𑀦𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀏𑀢𑁂𑀦 𑀧𑁂𑀢𑀁 𑀦𑀓𑁆𑀔𑀫𑀢𑀺—  ‘𑀯𑁂𑀤𑀦𑀸 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇

(124.)

233\. 𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑁄 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀢𑀸, 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀦𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀢𑁆𑀣 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀦𑀢𑁆𑀣𑀺 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀢𑀢𑁆𑀣 “𑀅𑀬𑀫𑀳𑀫𑀲𑁆𑀫𑀻”𑀢𑀺 𑀲𑀺𑀬𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀏𑀢𑁂𑀦 𑀧𑁂𑀢𑀁 𑀦𑀓𑁆𑀔𑀫𑀢𑀺—  ‘𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀢𑀸, 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀦𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇

(125.)

234\. 𑀢𑀢𑁆𑀭𑀸𑀦𑀦𑁆𑀤, 𑀬𑁄 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀢𑀸, 𑀦𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀦𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀸 𑀫𑁂 𑀯𑁂𑀤𑀺𑀬𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀥𑀫𑁆𑀫𑁄 𑀳𑀺 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  𑀯𑁂𑀤𑀦𑀸 𑀘 𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀅𑀧𑀭𑀺𑀲𑁂𑀲𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀸 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀢𑀢𑁆𑀣 ‘𑀅𑀬𑀫𑀳𑀫𑀲𑁆𑀫𑀻’𑀢𑀺 𑀲𑀺𑀬𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀏𑀢𑁂𑀦 𑀧𑁂𑀢𑀁 𑀦𑀓𑁆𑀔𑀫𑀢𑀺—  ‘𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀢𑀸, 𑀦𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀦𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀢𑀸 𑀫𑁂 𑀯𑁂𑀤𑀺𑀬𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀥𑀫𑁆𑀫𑁄 𑀳𑀺 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇

(126.)

235\. 𑀬𑀢𑁄 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑁂𑀯 𑀯𑁂𑀤𑀦𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀦𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄𑀧𑀺 ‘𑀅𑀢𑁆𑀢𑀸 𑀫𑁂 𑀯𑁂𑀤𑀺𑀬𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀥𑀫𑁆𑀫𑁄 𑀳𑀺 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀦 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺, 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀜𑁆𑀜𑁂𑀯 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺, ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀇𑀢𑀺𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀻’𑀢𑀺, 𑀢𑀤𑀓𑀮𑁆𑀮𑀁𑁇 ‘𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀇𑀢𑀺𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀻’𑀢𑀺, 𑀢𑀤𑀓𑀮𑁆𑀮𑀁𑁇 ‘𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀇𑀢𑀺𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀻’𑀢𑀺, 𑀢𑀤𑀓𑀮𑁆𑀮𑀁𑁇 ‘𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀇𑀢𑀺𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀻’𑀢𑀺, 𑀢𑀤𑀓𑀮𑁆𑀮𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀬𑀸𑀯𑀢𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀬𑀸𑀯𑀢𑀸 𑀅𑀥𑀺𑀯𑀘𑀦𑀧𑀣𑁄, 𑀬𑀸𑀯𑀢𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺 𑀬𑀸𑀯𑀢𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀣𑁄, 𑀬𑀸𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀬𑀸𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀧𑀣𑁄, 𑀬𑀸𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀬𑀸𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀯𑀘𑀭𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀯𑀝𑁆𑀝𑀁, 𑀬𑀸𑀯𑀢𑀸 𑀯𑀝𑁆𑀝𑀢𑀺, 𑀢𑀤𑀪𑀺𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀢𑀤𑀪𑀺𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 ‘𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺 𑀇𑀢𑀺𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀻’𑀢𑀺, 𑀢𑀤𑀓𑀮𑁆𑀮𑀁𑁇
