# 𑁯𑁇𑁧𑁇𑁨 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀧𑀩𑁆𑀩

### 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

### 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀧𑀩𑁆𑀩

(375.)

953\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀯𑀸 ‘𑀕𑀘𑁆𑀙𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀞𑀺𑀢𑁄 𑀯𑀸 ‘𑀞𑀺𑀢𑁄𑀫𑁆𑀳𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀸 ‘𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀬𑀸𑀦𑁄 𑀯𑀸 ‘𑀲𑀬𑀸𑀦𑁄𑀫𑁆𑀳𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀡𑀺𑀳𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀦𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

954\. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 ‘𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑁄’𑀢𑀺 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀬𑀸𑀯𑀤𑁂𑀯 𑀜𑀸𑀡𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇
