# 𑁯𑁇𑁨 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

### 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

(380.)

977\. 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧)

978\. 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁨)

979\. 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁩)

980\. 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀲𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁪)

981\. 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁫)

982\. 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀲𑀸𑀫𑀺𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁬)

983\. 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁭)

984\. 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀲𑀸𑀫𑀺𑀲𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁮)

985\. 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 ‘𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁯)

986\. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 ‘𑀅𑀢𑁆𑀣𑀺 𑀯𑁂𑀤𑀦𑀸’𑀢𑀺 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀬𑀸𑀯𑀤𑁂𑀯 𑀜𑀸𑀡𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

---

987\. 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
