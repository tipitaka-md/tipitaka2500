# 𑁯𑁇𑁪𑁇𑁫𑁇𑁧 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀲𑀘𑁆𑀘𑀧𑀩𑁆𑀩

* 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(387.)

1036\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁? 𑀚𑀸𑀢𑀺𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀚𑀭𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀅𑀧𑁆𑀧𑀺𑀬𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀧𑀺𑀬𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑁄𑀕𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑁇

(388.)

1037\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀺 𑀢𑀫𑁆𑀳𑀺 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑁂 𑀚𑀸𑀢𑀺 𑀲𑀜𑁆𑀚𑀸𑀢𑀺 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀆𑀬𑀢𑀦𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑁄, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀢𑀺𑁇 (𑁧)

(389.)

1038\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀭𑀸? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀺 𑀢𑀫𑁆𑀳𑀺 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑁂 𑀚𑀭𑀸 𑀚𑀻𑀭𑀡𑀢𑀸 𑀔𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀧𑀸𑀮𑀺𑀘𑁆𑀘𑀁 𑀯𑀮𑀺𑀢𑁆𑀢𑀘𑀢𑀸 𑀆𑀬𑀼𑀦𑁄 𑀲𑀁𑀳𑀸𑀦𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀸𑀓𑁄, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀭𑀸𑁇 (𑁨)

(390.)

1039\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀭𑀡𑀁? 𑀬𑀁 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀸 𑀢𑀫𑁆𑀳𑀸 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀸 𑀘𑀼𑀢𑀺 𑀘𑀯𑀦𑀢𑀸 𑀪𑁂𑀤𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀁 𑀫𑀘𑁆𑀘𑀼 𑀫𑀭𑀡𑀁 𑀓𑀸𑀮𑀓𑀺𑀭𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀪𑁂𑀤𑁄 𑀓𑀍𑀏𑀯𑀭𑀲𑁆𑀲 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲𑀼𑀧𑀘𑁆𑀙𑁂𑀤𑁄, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀭𑀡𑀁𑁇 (𑁩)

(391.)

1040\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀓𑁄? 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑁄𑀓𑁄 𑀲𑁄𑀘𑀦𑀸 𑀲𑁄𑀘𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑁄𑀲𑁄𑀓𑁄 𑀅𑀦𑁆𑀢𑁄𑀧𑀭𑀺𑀲𑁄𑀓𑁄, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀓𑁄𑁇 (𑁪)

(392.)

1041\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀤𑁂𑀯𑁄? 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀆𑀤𑁂𑀯𑁄 𑀧𑀭𑀺𑀤𑁂𑀯𑁄 𑀆𑀤𑁂𑀯𑀦𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑀸 𑀆𑀤𑁂𑀯𑀺𑀢𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀢𑁆𑀢𑀁, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀤𑁂𑀯𑁄𑁇 (𑁫)

(393.)

1042\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀲𑀸𑀢𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 (𑁬)

(394.)

1043\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁? 𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀲𑀸𑀢𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 (𑁭)

(395.)

1044\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀬𑀸𑀲𑁄? 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀆𑀬𑀸𑀲𑁄 𑀉𑀧𑀸𑀬𑀸𑀲𑁄 𑀆𑀬𑀸𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀬𑀸𑀲𑀺𑀢𑀢𑁆𑀢𑀁, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀬𑀸𑀲𑁄𑁇 (𑁮)

(396.)

1045\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀺𑀬𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑁄 𑀤𑀼𑀓𑁆𑀔𑁄? 𑀇𑀥 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀓𑀦𑁆𑀢𑀸 𑀅𑀫𑀦𑀸𑀧𑀸 𑀭𑀽𑀧𑀸 𑀲𑀤𑁆𑀤𑀸 𑀕𑀦𑁆𑀥𑀸 𑀭𑀲𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀦𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑀸 𑀅𑀨𑀸𑀲𑀼𑀓𑀓𑀸𑀫𑀸 𑀅𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑀸, 𑀬𑀸 𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀗𑁆𑀕𑀢𑀺 𑀲𑀫𑀸𑀕𑀫𑁄 𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀫𑀺𑀲𑁆𑀲𑀻𑀪𑀸𑀯𑁄, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀺𑀬𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑁄 𑀤𑀼𑀓𑁆𑀔𑁄𑁇 (𑁯)

(397.)

1046\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀺𑀬𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑁄𑀕𑁄 𑀤𑀼𑀓𑁆𑀔𑁄? 𑀇𑀥 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀭𑀽𑀧𑀸 𑀲𑀤𑁆𑀤𑀸 𑀕𑀦𑁆𑀥𑀸 𑀭𑀲𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀳𑀺𑀢𑀓𑀸𑀫𑀸 𑀨𑀸𑀲𑀼𑀓𑀓𑀸𑀫𑀸 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑀸 𑀫𑀸𑀢𑀸 𑀯𑀸 𑀧𑀺𑀢𑀸 𑀯𑀸 𑀪𑀸𑀢𑀸 𑀯𑀸 𑀪𑀕𑀺𑀦𑀻 𑀯𑀸 𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀸 𑀅𑀫𑀘𑁆𑀘𑀸 𑀯𑀸 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀯𑀸, 𑀬𑀸 𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀲𑀗𑁆𑀕𑀢𑀺 𑀅𑀲𑀫𑀸𑀕𑀫𑁄 𑀅𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀅𑀫𑀺𑀲𑁆𑀲𑀻𑀪𑀸𑀯𑁄, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀺𑀬𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑁄𑀕𑁄 𑀤𑀼𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦)

(398.)

1047\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀬𑀁 𑀦 𑀚𑀸𑀢𑀺𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀯𑀢 𑀦𑁄 𑀚𑀸𑀢𑀺 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀇𑀘𑁆𑀙𑀸𑀬 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀬𑀁 𑀦 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀯𑀢 𑀦𑁄 𑀚𑀭𑀸 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀇𑀘𑁆𑀙𑀸𑀬 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀬𑀁 𑀦 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀯𑀢 𑀦𑁄 𑀩𑁆𑀬𑀸𑀥𑀺 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀇𑀘𑁆𑀙𑀸𑀬 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀬𑀁 𑀦 𑀫𑀭𑀡𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀯𑀢 𑀦𑁄 𑀫𑀭𑀡𑀁 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀇𑀘𑁆𑀙𑀸𑀬 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀯𑀁 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀬𑀁 𑀦 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀸𑀫, 𑀦 𑀘 𑀯𑀢 𑀦𑁄 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀥𑀫𑁆𑀫𑀸 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀇𑀘𑁆𑀙𑀸𑀬 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀬𑀫𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀦 𑀮𑀪𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 (𑁧𑁧)

(399.)

1048\. 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀸? 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁𑁇
