# 𑁯𑁇𑁩 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

### 𑀫𑀳𑀸𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸

(381.)

988. 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀲𑀭𑀸𑀕𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧)

989. 𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁨)

990. 𑀲𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀲𑀤𑁄𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁩)

991. 𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁪)

992. 𑀲𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀲𑀫𑁄𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁫)

993. 𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁬)

994. 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁭)

995. 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁮)

996. 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁯)

997. 𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁦)

998. 𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁧)

999. 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁨)

1000. 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁩)

1001. 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁪)

1002. 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁫)

1003. 𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 ‘𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁫)

1004. 𑀇𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀫𑀼𑀤𑀬𑀯𑀬𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, ‘𑀅𑀢𑁆𑀣𑀺 𑀘𑀺𑀢𑁆𑀢𑀦𑁆’𑀢𑀺 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀬𑀸𑀯𑀤𑁂𑀯 𑀜𑀸𑀡𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

---

1005. 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
