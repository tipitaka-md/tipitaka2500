# 𑁩𑁇𑁧𑁭 𑀅𑀝𑁆𑀞𑀧𑀭𑀺𑀲𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀝𑁆𑀞𑀧𑀭𑀺𑀲𑀸

(172.)

374. 𑀅𑀝𑁆𑀞 𑀔𑁄 𑀇𑀫𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀭𑀺𑀲𑀸𑁇 𑀓𑀢𑀫𑀸 𑀅𑀝𑁆𑀞? 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀸, 𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀸, 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀸, 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀧𑀭𑀺𑀲𑀸, 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀧𑀭𑀺𑀲𑀸, 𑀫𑀸𑀭𑀧𑀭𑀺𑀲𑀸, 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀭𑀺𑀲𑀸𑁇 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑁂𑀓𑀲𑀢𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸𑁇 𑀢𑀢𑁆𑀭𑀧𑀺 𑀫𑀬𑀸 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑁂𑀯 𑀲𑀮𑁆𑀮𑀧𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘 𑀲𑀸𑀓𑀘𑁆𑀙𑀸 𑀘 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀣 𑀬𑀸𑀤𑀺𑀲𑀓𑁄 𑀢𑁂𑀲𑀁 𑀯𑀡𑁆𑀡𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀸𑀤𑀺𑀲𑀓𑁄 𑀫𑀬𑁆𑀳𑀁 𑀯𑀡𑁆𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸𑀤𑀺𑀲𑀓𑁄 𑀢𑁂𑀲𑀁 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀸𑀤𑀺𑀲𑀓𑁄 𑀫𑀬𑁆𑀳𑀁 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀫𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀫𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀫𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀫𑀺𑁇 𑀪𑀸𑀲𑀫𑀸𑀦𑀜𑁆𑀘 𑀫𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀁 𑀪𑀸𑀲𑀢𑀺 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸’𑀢𑀺? 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀸𑀫𑀺𑁇 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀜𑁆𑀘 𑀫𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸’𑀢𑀺? 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑁂𑀓𑀲𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀁…𑀧𑁂…  𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀁…  𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀁…  𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀧𑀭𑀺𑀲𑀁…  𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀧𑀭𑀺𑀲𑀁…  𑀫𑀸𑀭𑀧𑀭𑀺𑀲𑀁…  𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸𑁇 𑀢𑀢𑁆𑀭𑀧𑀺 𑀫𑀬𑀸 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑁂𑀯 𑀲𑀮𑁆𑀮𑀧𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘 𑀲𑀸𑀓𑀘𑁆𑀙𑀸 𑀘 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀣 𑀬𑀸𑀤𑀺𑀲𑀓𑁄 𑀢𑁂𑀲𑀁 𑀯𑀡𑁆𑀡𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀸𑀤𑀺𑀲𑀓𑁄 𑀫𑀬𑁆𑀳𑀁 𑀯𑀡𑁆𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸𑀤𑀺𑀲𑀓𑁄 𑀢𑁂𑀲𑀁 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀸𑀤𑀺𑀲𑀓𑁄 𑀫𑀬𑁆𑀳𑀁 𑀲𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀫𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀫𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀫𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀫𑀺𑁇 𑀪𑀸𑀲𑀫𑀸𑀦𑀜𑁆𑀘 𑀫𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀁 𑀪𑀸𑀲𑀢𑀺 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸’𑀢𑀺? 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀸𑀫𑀺𑁇 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀜𑁆𑀘 𑀫𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸’𑀢𑀺? 𑀇𑀫𑀸 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀝𑁆𑀞 𑀧𑀭𑀺𑀲𑀸𑁇
