# 𑁩𑁇𑁩𑁧 𑀆𑀦𑀦𑁆𑀤𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀥𑀫𑁆𑀫

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀆𑀦𑀦𑁆𑀤𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀥𑀫𑁆𑀫

(207.)

465\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑀧𑀺𑀲𑀻𑀲𑀁 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀫𑀸𑀦𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀺—  “𑀅𑀳𑀜𑁆𑀘 𑀯𑀢𑀫𑁆𑀳𑀺 𑀲𑁂𑀔𑁄 𑀲𑀓𑀭𑀡𑀻𑀬𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀘 𑀫𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑁄 𑀫𑀫 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀳𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀦𑁆𑀤𑁄”𑀢𑀺? “𑀏𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑀧𑀺𑀲𑀻𑀲𑀁 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀫𑀸𑀦𑁄 𑀞𑀺𑀢𑁄—  ‘𑀅𑀳𑀜𑁆𑀘 𑀯𑀢𑀫𑁆𑀳𑀺 𑀲𑁂𑀔𑁄 𑀲𑀓𑀭𑀡𑀻𑀬𑁄, 𑀲𑀢𑁆𑀣𑀼 𑀘 𑀫𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑁄 𑀫𑀫 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

466\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀮𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸 𑀲𑁄𑀘𑀺 𑀫𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀺, 𑀦𑀦𑀼 𑀏𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀬𑀸 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀅𑀓𑁆𑀔𑀸𑀢𑀁—  ‘𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀧𑀺𑀬𑁂𑀳𑀺 𑀫𑀦𑀸𑀧𑁂𑀳𑀺 𑀦𑀸𑀦𑀸𑀪𑀸𑀯𑁄 𑀯𑀺𑀦𑀸𑀪𑀸𑀯𑁄 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑁄’; 𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣, 𑀆𑀦𑀦𑁆𑀤, 𑀮𑀩𑁆𑀪𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀚𑀸𑀢𑀁 𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀢𑀁 𑀧𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀁, ‘𑀢𑀁 𑀯𑀢 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀲𑀭𑀻𑀭𑀁 𑀫𑀸 𑀧𑀮𑀼𑀚𑁆𑀚𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀔𑁄 𑀢𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀫𑁂𑀢𑁆𑀢𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦 𑀳𑀺𑀢𑁂𑀦 𑀲𑀼𑀔𑁂𑀦 𑀅𑀤𑁆𑀯𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦, 𑀫𑁂𑀢𑁆𑀢𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦 𑀳𑀺𑀢𑁂𑀦 𑀲𑀼𑀔𑁂𑀦 𑀅𑀤𑁆𑀯𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦, 𑀫𑁂𑀢𑁆𑀢𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦 𑀳𑀺𑀢𑁂𑀦 𑀲𑀼𑀔𑁂𑀦 𑀅𑀤𑁆𑀯𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦𑁇 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁄𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀥𑀸𑀦𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚, 𑀔𑀺𑀧𑁆𑀧𑀁 𑀳𑁄𑀳𑀺𑀲𑀺 𑀅𑀦𑀸𑀲𑀯𑁄”𑀢𑀺𑁇

(208.)

467\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑁂𑀧𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀏𑀢𑀧𑁆𑀧𑀭𑀫𑀸𑀬𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀬𑁆𑀳𑀁 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀏𑀢𑀧𑁆𑀧𑀭𑀫𑀸𑀬𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀬𑁆𑀳𑀁 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 𑀧𑀡𑁆𑀟𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀦𑁆𑀤𑁄; 𑀫𑁂𑀥𑀸𑀯𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 𑀚𑀸𑀦𑀸𑀢𑀺 ‘𑀅𑀬𑀁 𑀓𑀸𑀮𑁄 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁, 𑀅𑀬𑀁 𑀓𑀸𑀮𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁, 𑀅𑀬𑀁 𑀓𑀸𑀮𑁄 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁, 𑀅𑀬𑀁 𑀓𑀸𑀮𑁄 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁, 𑀅𑀬𑀁 𑀓𑀸𑀮𑁄 𑀭𑀜𑁆𑀜𑁄 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑀸𑀦𑀦𑁆’𑀢𑀺𑁇

(209.)

468\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀭𑀺𑀲𑀸 𑀆𑀦𑀦𑁆𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀆𑀦𑀦𑁆𑀤𑁄 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀧𑀭𑀺𑀲𑀸 𑀆𑀦𑀦𑁆𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀆𑀦𑀦𑁆𑀤𑁄 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀓𑀧𑀭𑀺𑀲𑀸 𑀆𑀦𑀦𑁆𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀆𑀦𑀦𑁆𑀤𑁄 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀓𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀧𑀭𑀺𑀲𑀸 𑀆𑀦𑀦𑁆𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀆𑀦𑀦𑁆𑀤𑁄 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑁇

469\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀭𑀜𑁆𑀜𑁂 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀫𑁆𑀳𑀺𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀸…𑀧𑁂…  𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀸…𑀧𑁂…  𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑁇 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀭𑀺𑀲𑀸 𑀆𑀦𑀦𑁆𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀆𑀦𑀦𑁆𑀤𑁄 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀧𑀭𑀺𑀲𑀸…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀧𑀭𑀺𑀲𑀸…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀧𑀭𑀺𑀲𑀸 𑀆𑀦𑀦𑁆𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀆𑀦𑀦𑁆𑀤𑁄 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀧𑀺 𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁂”𑀢𑀺𑁇
