# 𑁩𑁇𑁩𑁪 𑀲𑀼𑀪𑀤𑁆𑀤𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀼𑀪𑀤𑁆𑀤𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀯𑀢𑁆𑀣𑀼

(212.)

475\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀦𑀸𑀫 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀝𑀺𑀯𑀲𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄—  “𑀅𑀚𑁆𑀚 𑀓𑀺𑀭 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀅𑀚𑁆𑀚𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀫𑁂 𑀅𑀬𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑁄 𑀅𑀳𑀁 𑀲𑀫𑀡𑁂 𑀕𑁄𑀢𑀫𑁂, ‘𑀧𑀳𑁄𑀢𑀺 𑀫𑁂 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼𑀁, 𑀬𑀣𑀸𑀳𑀁 𑀇𑀫𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀦𑁆’”𑀢𑀺𑁇

476\. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀬𑁂𑀦 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑀁, 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀅𑀚𑁆𑀚𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀫𑁂 𑀅𑀬𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄—  𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑁄 𑀅𑀳𑀁 𑀲𑀫𑀡𑁂 𑀕𑁄𑀢𑀫𑁂 ‘𑀧𑀳𑁄𑀢𑀺 𑀫𑁂 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼𑀁, 𑀬𑀣𑀸𑀳𑀁 𑀇𑀫𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀳𑀁, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀲𑀺, 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀧𑀸𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀦𑀁—  ‘𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀅𑀚𑁆𑀚𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘 𑀫𑁂 𑀅𑀬𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄—  𑀏𑀯𑀁 𑀧𑀲𑀦𑁆𑀦𑁄 𑀅𑀳𑀁 𑀲𑀫𑀡𑁂 𑀕𑁄𑀢𑀫𑁂, ‘𑀧𑀳𑁄𑀢𑀺 𑀫𑁂 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼𑀁, 𑀬𑀣𑀸𑀳𑀁 𑀇𑀫𑀁 𑀓𑀗𑁆𑀔𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀳𑀁, 𑀪𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀲𑀺, 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇

477\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀼𑀪𑀤𑁆𑀤𑁂𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀇𑀫𑀁 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀮𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸 𑀲𑀼𑀪𑀤𑁆𑀤𑀁 𑀯𑀸𑀭𑁂𑀲𑀺, 𑀮𑀪𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀫𑀁 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀅𑀜𑁆𑀜𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑀯 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀯𑀺𑀳𑁂𑀲𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀬𑀜𑁆𑀘𑀲𑁆𑀲𑀸𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀓𑀭𑁄𑀢𑀺 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀑𑀓𑀸𑀲𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁂𑀫𑁂, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀁𑀖𑀺𑀦𑁄 𑀕𑀡𑀺𑀦𑁄 𑀕𑀡𑀸𑀘𑀭𑀺𑀬𑀸 𑀜𑀸𑀢𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄 𑀢𑀺𑀢𑁆𑀣𑀓𑀭𑀸 𑀲𑀸𑀥𑀼𑀲𑀫𑁆𑀫𑀢𑀸 𑀩𑀳𑀼𑀚𑀦𑀲𑁆𑀲, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀧𑀽𑀭𑀡𑁄 𑀓𑀲𑁆𑀲𑀧𑁄, 𑀫𑀓𑁆𑀔𑀮𑀺 𑀕𑁄𑀲𑀸𑀮𑁄, 𑀅𑀚𑀺𑀢𑁄 𑀓𑁂𑀲𑀓𑀫𑁆𑀩𑀮𑁄, 𑀧𑀓𑀼𑀥𑁄 𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄, 𑀲𑀜𑁆𑀘𑀬𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑀧𑀼𑀢𑁆𑀢𑁄, 𑀦𑀺𑀕𑀡𑁆𑀞𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁄, 𑀲𑀩𑁆𑀩𑁂𑀢𑁂 𑀲𑀓𑀸𑀬 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀼, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀦 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀼, 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑀘𑁆𑀘𑁂 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀼, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀦 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀽”𑀢𑀺? “𑀅𑀮𑀁, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀢𑀺𑀝𑁆𑀞𑀢𑁂𑀢𑀁—  ‘𑀲𑀩𑁆𑀩𑁂𑀢𑁂 𑀲𑀓𑀸𑀬 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀼, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀦 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀼, 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑀘𑁆𑀘𑁂 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀼, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀦 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀽’𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀁 𑀢𑁂, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺; 𑀢𑀁 𑀲𑀼𑀡𑀸𑀳𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀳𑀺, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

(214.)

478\. “𑀬𑀲𑁆𑀫𑀺𑀁 𑀔𑁄, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺, 𑀲𑀫𑀡𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀦 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀡𑁄 𑀦 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀡𑁄 𑀦 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀡𑁄 𑀦 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀔𑁄, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺, 𑀲𑀫𑀡𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺, 𑀤𑀼𑀢𑀺𑀬𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀡𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺, 𑀢𑀢𑀺𑀬𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀡𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀡𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺𑁇 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀔𑁄, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺, 𑀇𑀥𑁂𑀯, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀢𑀢𑀺𑀬𑁄 𑀲𑀫𑀡𑁄, 𑀇𑀥 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀲𑀫𑀡𑁄, 𑀲𑀼𑀜𑁆𑀜𑀸 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀸 𑀲𑀫𑀡𑁂𑀪𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺𑁇 𑀇𑀫𑁂 𑀘, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑁆𑀫𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀅𑀲𑀼𑀜𑁆𑀜𑁄 𑀮𑁄𑀓𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀸𑀢𑀺𑁇

479\. _𑀏𑀓𑀽𑀦𑀢𑀺𑀁𑀲𑁄 𑀯𑀬𑀲𑀸 𑀲𑀼𑀪𑀤𑁆𑀤,_  
_𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀓𑀺𑀁𑀓𑀼𑀲𑀮𑀸𑀦𑀼𑀏𑀲𑀻;_  
_𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀸𑀲 𑀲𑀫𑀸𑀥𑀺𑀓𑀸𑀦𑀺,_  
_𑀬𑀢𑁄 𑀅𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀼𑀪𑀤𑁆𑀤;_  
_𑀜𑀸𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀤𑁂𑀲𑀯𑀢𑁆𑀢𑀻,_  
_𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀲𑀫𑀡𑁄𑀧𑀺 𑀦𑀢𑁆𑀣𑀺𑁇_  


480\. 𑀤𑀼𑀢𑀺𑀬𑁄𑀧𑀺 𑀲𑀫𑀡𑁄 𑀦𑀢𑁆𑀣𑀺𑁇 𑀢𑀢𑀺𑀬𑁄𑀧𑀺 𑀲𑀫𑀡𑁄 𑀦𑀢𑁆𑀣𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑀧𑀺 𑀲𑀫𑀡𑁄 𑀦𑀢𑁆𑀣𑀺𑁇 𑀲𑀼𑀜𑁆𑀜𑀸 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀸 𑀲𑀫𑀡𑁂𑀪𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺𑁇 𑀇𑀫𑁂 𑀘, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑁆𑀫𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀅𑀲𑀼𑀜𑁆𑀜𑁄 𑀮𑁄𑀓𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

(215.)

481\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀬𑁄 𑀔𑁄, 𑀲𑀼𑀪𑀤𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀼𑀩𑁆𑀩𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁, 𑀲𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀭𑀺𑀯𑀲𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀆𑀭𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀅𑀧𑀺 𑀘 𑀫𑁂𑀢𑁆𑀣 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀯𑀺𑀤𑀺𑀢𑀸”𑀢𑀺𑁇 “𑀲𑀘𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀼𑀩𑁆𑀩𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀆𑀓𑀗𑁆𑀔𑀦𑁆𑀢𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀆𑀓𑀗𑁆𑀔𑀦𑁆𑀢𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀺, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀆𑀭𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀅𑀳𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀧𑀭𑀺𑀯𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀆𑀭𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀼 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀸𑀬𑀸”𑀢𑀺𑁇

482\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦𑀳𑀸𑀦𑀦𑁆𑀤, 𑀲𑀼𑀪𑀤𑁆𑀤𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀮𑀸𑀪𑀸 𑀯𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤; 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀬𑁂 𑀏𑀢𑁆𑀣 𑀲𑀢𑁆𑀣𑀼 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸𑀪𑀺𑀲𑁂𑀓𑁂𑀦 𑀅𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑀸”𑀢𑀺𑁇 𑀅𑀮𑀢𑁆𑀣 𑀔𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀢𑁆𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀅𑀘𑀺𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺—  𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀲𑀓𑁆𑀔𑀺𑀲𑀸𑀯𑀓𑁄 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇

---
