# 𑁩𑁇𑁨𑁮 𑀘𑀢𑀼𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀞𑀸𑀦

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀞𑀸𑀦

(202.)

454\. “𑀧𑀼𑀩𑁆𑀩𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑀸𑀲𑀼 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀢𑁂 𑀫𑀬𑀁 𑀮𑀪𑀸𑀫 𑀫𑀦𑁄𑀪𑀸𑀯𑀦𑀻𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀮𑀪𑀸𑀫 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑀸𑀬𑁇 𑀪𑀕𑀯𑀢𑁄 𑀧𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀦 𑀮𑀪𑀺𑀲𑁆𑀲𑀸𑀫 𑀫𑀦𑁄𑀪𑀸𑀯𑀦𑀻𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀦 𑀮𑀪𑀺𑀲𑁆𑀲𑀸𑀫 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

455\. “𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀫𑀸𑀦𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸𑀦𑀺 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀸𑀦𑀺 𑀞𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺? ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁄 𑀚𑀸𑀢𑁄’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀁 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀁 𑀞𑀸𑀦𑀁𑁇 ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀁 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀁 𑀞𑀸𑀦𑀁𑁇 ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀦𑁆’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀁 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀁 𑀞𑀸𑀦𑀁𑁇 ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀁 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀁 𑀞𑀸𑀦𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸𑀦𑀺 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑀸𑀦𑀺 𑀞𑀸𑀦𑀸𑀦𑀺𑁇

456\. 𑀆𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀤𑁆𑀥𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄—  ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁄 𑀚𑀸𑀢𑁄’𑀢𑀺𑀧𑀺, ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄’𑀢𑀺𑀧𑀺, ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀦𑁆’𑀢𑀺𑀧𑀺, ‘𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄’𑀢𑀺𑀧𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑁂𑀢𑀺𑀬𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
