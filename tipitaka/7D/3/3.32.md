# 𑁩𑁇𑁩𑁨 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢𑀤𑁂𑀲𑀦𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢𑀤𑁂𑀲𑀦𑀸

(210.)

470\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀔𑀼𑀤𑁆𑀤𑀓𑀦𑀕𑀭𑀓𑁂 𑀉𑀚𑁆𑀚𑀗𑁆𑀕𑀮𑀦𑀕𑀭𑀓𑁂 𑀲𑀸𑀔𑀸𑀦𑀕𑀭𑀓𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑁇 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀸𑀦𑀺 𑀫𑀳𑀸𑀦𑀕𑀭𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀫𑁆𑀧𑀸 𑀭𑀸𑀚𑀕𑀳𑀁 𑀲𑀸𑀯𑀢𑁆𑀣𑀻 𑀲𑀸𑀓𑁂𑀢𑀁 𑀓𑁄𑀲𑀫𑁆𑀩𑀻 𑀩𑀸𑀭𑀸𑀡𑀲𑀻; 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀼𑁇 𑀏𑀢𑁆𑀣 𑀩𑀳𑀽 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀢𑀣𑀸𑀕𑀢𑁂 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑁇 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀧𑀽𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺 “𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀯𑀘; 𑀫𑀸 𑀳𑁂𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀯𑀘—  ‘𑀔𑀼𑀤𑁆𑀤𑀓𑀦𑀕𑀭𑀓𑀁 𑀉𑀚𑁆𑀚𑀗𑁆𑀕𑀮𑀦𑀕𑀭𑀓𑀁 𑀲𑀸𑀔𑀸𑀦𑀕𑀭𑀓𑀦𑁆’𑀢𑀺𑁇

471\. 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀦𑀸𑀫 𑀅𑀳𑁄𑀲𑀺 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀚𑀦𑀧𑀤𑀢𑁆𑀣𑀸𑀯𑀭𑀺𑀬𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀅𑀬𑀁 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀓𑀼𑀲𑀸𑀯𑀢𑀻 𑀦𑀸𑀫 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀅𑀳𑁄𑀲𑀺, 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂𑀦 𑀘 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀘 𑀤𑁆𑀯𑀸𑀤𑀲𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀆𑀬𑀸𑀫𑁂𑀦; 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀘 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀘 𑀲𑀢𑁆𑀢𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦𑁇 𑀓𑀼𑀲𑀸𑀯𑀢𑀻, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀨𑀻𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀍𑀅𑀓𑀫𑀦𑁆𑀤𑀸 𑀦𑀸𑀫 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀨𑀻𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀬𑀓𑁆𑀔𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀼𑀲𑀸𑀯𑀢𑀻 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀨𑀻𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘𑁇 𑀓𑀼𑀲𑀸𑀯𑀢𑀻, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀤𑀲𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺 𑀅𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁄𑀲𑀺 𑀤𑀺𑀯𑀸 𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀳𑀢𑁆𑀣𑀺𑀲𑀤𑁆𑀤𑁂𑀦 𑀅𑀲𑁆𑀲𑀲𑀤𑁆𑀤𑁂𑀦 𑀭𑀣𑀲𑀤𑁆𑀤𑁂𑀦 𑀪𑁂𑀭𑀺𑀲𑀤𑁆𑀤𑁂𑀦 𑀫𑀼𑀤𑀺𑀗𑁆𑀕𑀲𑀤𑁆𑀤𑁂𑀦 𑀯𑀻𑀡𑀸𑀲𑀤𑁆𑀤𑁂𑀦 𑀕𑀻𑀢𑀲𑀤𑁆𑀤𑁂𑀦 𑀲𑀗𑁆𑀔𑀲𑀤𑁆𑀤𑁂𑀦 𑀲𑀫𑁆𑀫𑀲𑀤𑁆𑀤𑁂𑀦 𑀧𑀸𑀡𑀺𑀢𑀸𑀍𑀅𑀲𑀤𑁆𑀤𑁂𑀦 ‘𑀅𑀲𑁆𑀦𑀸𑀣 𑀧𑀺𑀯𑀣 𑀔𑀸𑀤𑀣𑀸’𑀢𑀺 𑀤𑀲𑀫𑁂𑀦 𑀲𑀤𑁆𑀤𑁂𑀦𑁇

472\. 𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀺—  ‘𑀅𑀚𑁆𑀚 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀣, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀣, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸𑁇 𑀫𑀸 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄 𑀅𑀳𑀼𑀯𑀢𑁆𑀣—  𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘 𑀦𑁄 𑀕𑀸𑀫𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀦 𑀫𑀬𑀁 𑀮𑀪𑀺𑀫𑁆𑀳𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀓𑀸𑀮𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀅𑀢𑁆𑀢𑀤𑀼𑀢𑀺𑀬𑁄 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇
