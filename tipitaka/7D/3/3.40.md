# 𑁩𑁇𑁪𑁦 𑀥𑀸𑀢𑀼𑀣𑀽𑀧𑀧𑀽𑀚𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀥𑀸𑀢𑀼𑀣𑀽𑀧𑀧𑀽𑀚𑀸

(239.)

528\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀸𑀲𑀺𑁇 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸𑀧𑀺 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀁𑀲𑀼𑁇 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀯𑀸𑀲𑀻𑀧𑀺 𑀲𑀓𑁆𑀬𑀸 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀁𑀲𑀼𑁇 𑀅𑀮𑁆𑀮𑀓𑀧𑁆𑀧𑀓𑀸𑀧𑀺 𑀩𑀼𑀮𑀬𑁄 𑀅𑀮𑁆𑀮𑀓𑀧𑁆𑀧𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀁𑀲𑀼𑁇 𑀭𑀸𑀫𑀕𑀸𑀫𑀓𑀸𑀧𑀺 𑀓𑁄𑀍𑀇𑀬𑀸 𑀭𑀸𑀫𑀕𑀸𑀫𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀁𑀲𑀼𑁇 𑀯𑁂𑀝𑁆𑀞𑀤𑀻𑀧𑀓𑁄𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑁂𑀝𑁆𑀞𑀤𑀻𑀧𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀸𑀲𑀺𑁇 𑀧𑀸𑀯𑁂𑀬𑁆𑀬𑀓𑀸𑀧𑀺 𑀫𑀮𑁆𑀮𑀸 𑀧𑀸𑀯𑀸𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀁𑀲𑀼𑁇 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀧𑀺 𑀫𑀮𑁆𑀮𑀸 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀁𑀲𑀼𑁇 𑀤𑁄𑀡𑁄𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀼𑀫𑁆𑀩𑀲𑁆𑀲 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀸𑀲𑀺𑁇 𑀧𑀺𑀧𑁆𑀧𑀮𑀺𑀯𑀦𑀺𑀬𑀸𑀧𑀺 𑀫𑁄𑀭𑀺𑀬𑀸 𑀧𑀺𑀧𑁆𑀧𑀮𑀺𑀯𑀦𑁂 𑀅𑀗𑁆𑀕𑀸𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀅𑀓𑀁𑀲𑀼𑁇 𑀇𑀢𑀺 𑀅𑀝𑁆𑀞 𑀲𑀭𑀻𑀭𑀣𑀽𑀧𑀸 𑀦𑀯𑀫𑁄 𑀢𑀼𑀫𑁆𑀩𑀣𑀽𑀧𑁄 𑀤𑀲𑀫𑁄 𑀅𑀗𑁆𑀕𑀸𑀭𑀣𑀽𑀧𑁄𑁇 𑀏𑀯𑀫𑁂𑀢𑀁 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺𑁇

(240.)

529\. _𑀅𑀝𑁆𑀞𑀤𑁄𑀡𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄 𑀲𑀭𑀻𑀭𑀁,_  
_𑀲𑀢𑁆𑀢𑀤𑁄𑀡𑀁 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀫𑀳𑁂𑀦𑁆𑀢𑀺;_  
_𑀏𑀓𑀜𑁆𑀘 𑀤𑁄𑀡𑀁 𑀧𑀼𑀭𑀺𑀲𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀲𑁆𑀲,_  
_𑀭𑀸𑀫𑀕𑀸𑀫𑁂 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀫𑀳𑁂𑀢𑀺𑁇_  


530\. _𑀏𑀓𑀸𑀳𑀺 𑀤𑀸𑀞𑀸 𑀢𑀺𑀤𑀺𑀯𑁂𑀳𑀺 𑀧𑀽𑀚𑀺𑀢𑀸,_  
_𑀏𑀓𑀸 𑀧𑀦 𑀕𑀦𑁆𑀥𑀸𑀭𑀧𑀼𑀭𑁂 𑀫𑀳𑀻𑀬𑀢𑀺;_  
_𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀭𑀜𑁆𑀜𑁄 𑀯𑀺𑀚𑀺𑀢𑁂 𑀧𑀼𑀦𑁂𑀓𑀁,_  
_𑀏𑀓𑀁 𑀧𑀦 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀫𑀳𑁂𑀢𑀺𑁇_  


531\. _𑀢𑀲𑁆𑀲𑁂𑀯 𑀢𑁂𑀚𑁂𑀦 𑀅𑀬𑀁 𑀯𑀲𑀼𑀦𑁆𑀥𑀭𑀸,_  
_𑀆𑀬𑀸𑀕𑀲𑁂𑀝𑁆𑀞𑁂𑀳𑀺 𑀫𑀳𑀻 𑀅𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀏𑀯𑀁 𑀇𑀫𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄 𑀲𑀭𑀻𑀭𑀁,_  
_𑀲𑀼𑀲𑀓𑁆𑀓𑀢𑀁 𑀲𑀓𑁆𑀓𑀢𑀲𑀓𑁆𑀓𑀢𑁂𑀳𑀺𑁇_  


532\. _𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑀦𑀸𑀕𑀺𑀦𑁆𑀤𑀦𑀭𑀺𑀦𑁆𑀤𑀧𑀽𑀚𑀺𑀢𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑀲𑁂𑀝𑁆𑀞𑁂𑀳𑀺 𑀢𑀣𑁂𑀯 𑀧𑀽𑀚𑀺𑀢𑁄;_  
_𑀢𑀁 𑀯𑀦𑁆𑀤𑀣 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀮𑀪𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀳𑀯𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂𑀳𑀺 𑀤𑀼𑀮𑁆𑀮𑀪𑁄𑀢𑀺𑁇_  


533\. _𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲 𑀲𑀫𑀸 𑀤𑀦𑁆𑀢𑀸,_  
_𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀘 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀤𑁂𑀯𑀸 𑀳𑀭𑀺𑀁𑀲𑀼 𑀏𑀓𑁂𑀓𑀁,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀍𑀅𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀢𑀺𑁇_  


---

534\. 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
