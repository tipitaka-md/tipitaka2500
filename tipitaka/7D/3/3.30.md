# 𑁩𑁇𑁩𑁦 𑀣𑀽𑀧𑀸𑀭𑀳𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀣𑀽𑀧𑀸𑀭𑀳𑀧𑀼𑀕𑁆𑀕𑀮

(206.)

460\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀣𑀽𑀧𑀸𑀭𑀳𑀸𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄, 𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄, 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀣𑀽𑀧𑀸𑀭𑀳𑁄𑀢𑀺𑁇

461\. 𑀓𑀺𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄? ‘𑀅𑀬𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀣𑀽𑀧𑁄’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀇𑀤𑀁 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄𑁇

462\. 𑀓𑀺𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄? ‘𑀅𑀬𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀣𑀽𑀧𑁄’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀇𑀤𑀁 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄𑁇

463\. 𑀓𑀺𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄? ‘𑀅𑀬𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀣𑀽𑀧𑁄’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀇𑀤𑀁 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑁄𑁇

464\. 𑀓𑀺𑀜𑁆𑀘𑀸𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀣𑀽𑀧𑀸𑀭𑀳𑁄? ‘𑀅𑀬𑀁 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀭𑀜𑁆𑀜𑁄 𑀣𑀽𑀧𑁄’𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀇𑀤𑀁 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀣𑀽𑀧𑀸𑀭𑀳𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀣𑀽𑀧𑀸𑀭𑀳𑀸”𑀢𑀺𑁇
