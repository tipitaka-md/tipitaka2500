# 𑁩𑁇𑁧𑁯 𑀅𑀝𑁆𑀞𑀯𑀺𑀫𑁄𑀓𑁆𑀔

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀝𑁆𑀞𑀯𑀺𑀫𑁄𑀓𑁆𑀔

(174.)

383. 𑀅𑀝𑁆𑀞 𑀔𑁄 𑀇𑀫𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀭𑀽𑀧𑀻 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀲𑀼𑀪𑀦𑁆𑀢𑁂𑀯 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀝𑁆𑀞 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑁇

385. 𑀏𑀓𑀫𑀺𑀤𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑀁 𑀉𑀭𑀼𑀯𑁂𑀮𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀦𑀚𑁆𑀚𑀸 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀸𑀬 𑀢𑀻𑀭𑁂 𑀅𑀚𑀧𑀸𑀮𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁂 𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸; 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—

386. ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀲𑀸𑀯𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

387. 𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦 𑀲𑀸𑀯𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀺𑀬𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

390. 𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀫𑁂 𑀇𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀦 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀦𑁆’𑀢𑀺𑁇

391. 𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀚𑁆𑀚 𑀘𑀸𑀧𑀸𑀮𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  “𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀲𑀸𑀯𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦 𑀲𑀸𑀯𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀫𑁂 𑀉𑀧𑀸𑀲𑀓𑀸 𑀦 𑀲𑀸𑀯𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀫𑁂 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀦 𑀲𑀸𑀯𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀫𑁂 𑀇𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀦 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁, 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀦𑁆”𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁, 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀁𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄’𑀢𑀺𑁇

(177.)

392. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑁆𑀯𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀳𑁄𑀳𑀺, 𑀦 𑀘𑀺𑀭𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀚𑁆𑀚 𑀘𑀸𑀧𑀸𑀮𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁂𑀦 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀑𑀲𑁆𑀲𑀝𑁆𑀞𑁄”𑀢𑀺𑁇
