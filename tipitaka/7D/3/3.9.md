# 𑁩𑁇𑁯 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡

(156.)

330\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀦𑀸𑀢𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀦𑀸𑀢𑀺𑀓𑀸 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭𑀧𑀺 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀍𑀳𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀢𑀺𑀓𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀦𑀦𑁆𑀤𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀦𑀸𑀢𑀺𑀓𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀲𑀼𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀦𑀸𑀢𑀺𑀓𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀲𑀼𑀚𑀸𑀢𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀦𑀸𑀢𑀺𑀓𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀦𑀸𑀢𑀺𑀓𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄? 𑀓𑀸𑀍𑀇𑀫𑁆𑀩𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄…𑀧𑁂…  𑀦𑀺𑀓𑀝𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀓𑀝𑀺𑀲𑁆𑀲𑀳𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀢𑀼𑀝𑁆𑀞𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀪𑀤𑁆𑀤𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀲𑀼𑀪𑀤𑁆𑀤𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀦𑀸𑀢𑀺𑀓𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄”𑀢𑀺?

(157.)

331\. “𑀲𑀸𑀍𑀳𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 𑀦𑀦𑁆𑀤𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀦𑀻 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀸 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇 𑀲𑀼𑀤𑀢𑁆𑀢𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀸𑀦𑀁 𑀢𑀦𑀼𑀢𑁆𑀢𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻 𑀲𑀓𑀺𑀤𑁂𑀯 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀼𑀚𑀸𑀢𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸𑁇 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁄 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇 𑀓𑀸𑀍𑀇𑀫𑁆𑀩𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄…𑀧𑁂…  𑀦𑀺𑀓𑀝𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀓𑀝𑀺𑀲𑁆𑀲𑀳𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀢𑀼𑀝𑁆𑀞𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀪𑀤𑁆𑀤𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄…  𑀲𑀼𑀪𑀤𑁆𑀤𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁄 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑁄 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇 𑀧𑀭𑁄𑀧𑀜𑁆𑀜𑀸𑀲𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀸𑀢𑀺𑀓𑁂 𑀉𑀧𑀸𑀲𑀓𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀢𑀢𑁆𑀣 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀦𑁄 𑀅𑀦𑀸𑀯𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀸 𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸𑁇 𑀲𑀸𑀥𑀺𑀓𑀸 𑀦𑀯𑀼𑀢𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀸𑀢𑀺𑀓𑁂 𑀉𑀧𑀸𑀲𑀓𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀸𑀦𑀁 𑀢𑀦𑀼𑀢𑁆𑀢𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄 𑀲𑀓𑀺𑀤𑁂𑀯 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀢𑀺𑀭𑁂𑀓𑀸𑀦𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺 𑀦𑀸𑀢𑀺𑀓𑁂 𑀉𑀧𑀸𑀲𑀓𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀬𑀢𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑀸𑁇
