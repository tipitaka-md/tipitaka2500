# 𑁩𑁇𑁨𑁭 𑀉𑀧𑀯𑀸𑀡𑀢𑁆𑀣𑁂𑀭

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀉𑀧𑀯𑀸𑀡𑀢𑁆𑀣𑁂𑀭

(200.)

449\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀯𑀸𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀩𑀻𑀚𑀬𑀫𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀯𑀸𑀡𑀁 𑀅𑀧𑀲𑀸𑀭𑁂𑀲𑀺—  “𑀅𑀧𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀫𑁂 𑀧𑀼𑀭𑀢𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀬𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀯𑀸𑀡𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑀸𑀯𑀘𑀭𑁄 𑀲𑀫𑀻𑀧𑀘𑀸𑀭𑀻𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀓𑀸𑀮𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀯𑀸𑀡𑀁 𑀅𑀧𑀲𑀸𑀭𑁂𑀢𑀺—  ‘𑀅𑀧𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀫𑁂 𑀧𑀼𑀭𑀢𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑀁 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀯𑀸𑀡𑀁 𑀅𑀧𑀲𑀸𑀭𑁂𑀢𑀺—  ‘𑀅𑀧𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀫𑁂 𑀧𑀼𑀭𑀢𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀻’”𑀢𑀺?

450\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀯𑀸𑀡𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑀸𑀯𑀘𑀭𑁄 𑀲𑀫𑀻𑀧𑀘𑀸𑀭𑀻𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀓𑀸𑀮𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀯𑀸𑀡𑀁 𑀅𑀧𑀲𑀸𑀭𑁂𑀢𑀺—  ‘𑀅𑀧𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀫𑁂 𑀧𑀼𑀭𑀢𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀻’𑀢𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑀁 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀯𑀸𑀡𑀁 𑀅𑀧𑀲𑀸𑀭𑁂𑀢𑀺—  ‘𑀅𑀧𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀫𑁂 𑀧𑀼𑀭𑀢𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀻’”𑀢𑀺? “𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑀲𑀲𑀼 𑀮𑁄𑀓𑀥𑀸𑀢𑀽𑀲𑀼 𑀤𑁂𑀯𑀢𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀬𑀸𑀯𑀢𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑀁 𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀤𑁆𑀯𑀸𑀤𑀲 𑀬𑁄𑀚𑀦𑀸𑀦𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀲𑁄 𑀧𑀤𑁂𑀲𑁄 𑀯𑀸𑀮𑀕𑁆𑀕𑀓𑁄𑀝𑀺𑀦𑀺𑀢𑁆𑀢𑀼𑀤𑀦𑀫𑀢𑁆𑀢𑁄𑀧𑀺 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀸𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀅𑀧𑁆𑀨𑀼𑀝𑁄𑁇 𑀤𑁂𑀯𑀢𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺—  ‘𑀤𑀽𑀭𑀸 𑀘 𑀯𑀢𑀫𑁆𑀳 𑀆𑀕𑀢𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸𑁇 𑀅𑀚𑁆𑀚𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀞𑀺𑀢𑁄 𑀑𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑁄, 𑀦 𑀫𑀬𑀁 𑀮𑀪𑀸𑀫 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀓𑀸𑀮𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’”𑀢𑀺𑁇

(201.)

451\. “𑀓𑀣𑀁𑀪𑀽𑀢𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑁂𑀯𑀢𑀸 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀤𑁂𑀯𑀢𑀸 𑀆𑀓𑀸𑀲𑁂 𑀧𑀣𑀯𑀻𑀲𑀜𑁆𑀜𑀺𑀦𑀺𑀬𑁄 𑀓𑁂𑀲𑁂 𑀧𑀓𑀺𑀭𑀺𑀬 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀧𑀸𑀢𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀆𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀼𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

452\. 𑀲𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀤, 𑀤𑁂𑀯𑀢𑀸 𑀧𑀣𑀯𑀺𑀬𑀁 𑀧𑀣𑀯𑀻𑀲𑀜𑁆𑀜𑀺𑀦𑀺𑀬𑁄 𑀓𑁂𑀲𑁂 𑀧𑀓𑀺𑀭𑀺𑀬 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀧𑀸𑀢𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀆𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀼𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

453\. 𑀬𑀸 𑀧𑀦 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀻𑀢𑀭𑀸𑀕𑀸, 𑀢𑀸 𑀲𑀢𑀸 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣 𑀮𑀩𑁆𑀪𑀸’”𑀢𑀺𑁇
