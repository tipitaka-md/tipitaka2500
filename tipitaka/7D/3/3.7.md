# 𑁩𑁇𑁭 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀦𑀕𑀭𑀫𑀸𑀧𑀦

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀦𑀕𑀭𑀫𑀸𑀧𑀦

(152.)

311\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁂 𑀦𑀕𑀭𑀁 𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀚𑁆𑀚𑀻𑀦𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀸𑀬𑁇 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀲𑀳𑀲𑁆𑀲𑁂𑀯 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁂 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀭𑀜𑁆𑀜𑀁 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀫𑀦𑁆𑀢𑀺 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀫𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀭𑀜𑁆𑀜𑀁 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀫𑀦𑁆𑀢𑀺 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀫𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀦𑀻𑀘𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀦𑀻𑀘𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀭𑀜𑁆𑀜𑀁 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀫𑀦𑁆𑀢𑀺 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀫𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀲𑀳𑀲𑁆𑀲𑁂𑀯 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁂 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑀬𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

312\. “𑀓𑁂 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁂 𑀦𑀕𑀭𑀁 𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁂 𑀦𑀕𑀭𑀁 𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀚𑁆𑀚𑀻𑀦𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑁂𑀯𑁂𑀳𑀺 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁂 𑀦𑀕𑀭𑀁 𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀚𑁆𑀚𑀻𑀦𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀸𑀬𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀤𑁆𑀤𑀲𑀁 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀲𑀳𑀲𑁆𑀲𑁂𑀯 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁂 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑀬𑁄𑁇 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀤𑁂𑀲𑁂 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀭𑀜𑁆𑀜𑀁 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀫𑀦𑁆𑀢𑀺 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀫𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀭𑀜𑁆𑀜𑀁 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀫𑀦𑁆𑀢𑀺 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀫𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀦𑀻𑀘𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀦𑀻𑀘𑀸𑀦𑀁 𑀢𑀢𑁆𑀣 𑀭𑀜𑁆𑀜𑀁 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀫𑀦𑁆𑀢𑀺 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀫𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀸𑀯𑀢𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀁 𑀆𑀬𑀢𑀦𑀁 𑀬𑀸𑀯𑀢𑀸 𑀯𑀡𑀺𑀧𑁆𑀧𑀣𑁄 𑀇𑀤𑀁 𑀅𑀕𑁆𑀕𑀦𑀕𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀼𑀝𑀪𑁂𑀤𑀦𑀁𑁇 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀬𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  𑀅𑀕𑁆𑀕𑀺𑀢𑁄 𑀯𑀸 𑀉𑀤𑀓𑀢𑁄 𑀯𑀸 𑀫𑀺𑀣𑀼𑀪𑁂𑀤𑀸 𑀯𑀸”𑀢𑀺𑁇

(153.)

313\. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀦𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀅𑀚𑁆𑀚𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀓𑁄 𑀆𑀯𑀲𑀣𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀓𑁂 𑀆𑀯𑀲𑀣𑁂 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀼𑀁—  “𑀓𑀸𑀮𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

314\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀬𑁂𑀦 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸𑀦𑀁 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀯𑀲𑀣𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀼𑀁 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀦𑀻𑀘𑀁 𑀆𑀲𑀦𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀔𑁄 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑁂 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂 𑀪𑀕𑀯𑀸 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—

315\. _“𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀓𑀧𑁆𑀧𑁂𑀢𑀺,_  
_𑀯𑀸𑀲𑀁 𑀧𑀡𑁆𑀟𑀺𑀢𑀚𑀸𑀢𑀺𑀬𑁄;_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀪𑁄𑀚𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀜𑁆𑀜𑀢𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀬𑁄𑁇_  


316\. _𑀬𑀸 𑀢𑀢𑁆𑀣 𑀤𑁂𑀯𑀢𑀸 𑀆𑀲𑀼𑀁,_  
_𑀢𑀸𑀲𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀫𑀸𑀤𑀺𑀲𑁂;_  
_𑀢𑀸 𑀧𑀽𑀚𑀺𑀢𑀸 𑀧𑀽𑀚𑀬𑀦𑁆𑀢𑀺,_  
_𑀫𑀸𑀦𑀺𑀢𑀸 𑀫𑀸𑀦𑀬𑀦𑁆𑀢𑀺 𑀦𑀁𑁇_  


317\. _𑀢𑀢𑁄 𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀦𑁆𑀢𑀺,_  
_𑀫𑀸𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀁𑀯 𑀑𑀭𑀲𑀁;_  
_𑀤𑁂𑀯𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀲𑀤𑀸 𑀪𑀤𑁆𑀭𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


318\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑁂 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

(154.)

319\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑀺𑀥𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑀸 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  “𑀬𑁂𑀦𑀚𑁆𑀚 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀁 𑀕𑁄𑀢𑀫𑀤𑁆𑀯𑀸𑀭𑀁 𑀦𑀸𑀫 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑁂𑀦 𑀢𑀺𑀢𑁆𑀣𑁂𑀦 𑀕𑀗𑁆𑀕𑀁 𑀦𑀤𑀺𑀁 𑀢𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀁 𑀕𑁄𑀢𑀫𑀢𑀺𑀢𑁆𑀣𑀁 𑀦𑀸𑀫 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

320\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺, 𑀢𑀁 𑀕𑁄𑀢𑀫𑀤𑁆𑀯𑀸𑀭𑀁 𑀦𑀸𑀫 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀕𑀗𑁆𑀕𑀸 𑀦𑀤𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀕𑀗𑁆𑀕𑀸 𑀦𑀤𑀻 𑀧𑀽𑀭𑀸 𑀳𑁄𑀢𑀺 𑀲𑀫𑀢𑀺𑀢𑁆𑀢𑀺𑀓𑀸 𑀓𑀸𑀓𑀧𑁂𑀬𑁆𑀬𑀸𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦𑀸𑀯𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀉𑀍𑀉𑀫𑁆𑀧𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀓𑀼𑀮𑁆𑀮𑀁 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀅𑀧𑀸𑀭𑀸, 𑀧𑀸𑀭𑀁 𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀑𑀭𑀺𑀫𑀢𑀻𑀭𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀧𑀸𑀭𑀺𑀫𑀢𑀻𑀭𑁂 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇

321\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦𑀸𑀯𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑁂 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀉𑀍𑀉𑀫𑁆𑀧𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑁂 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀓𑀼𑀮𑁆𑀮𑀁 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑁂 𑀅𑀧𑀸𑀭𑀸 𑀧𑀸𑀭𑀁 𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑁂𑁇

322\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

323\. _“𑀬𑁂 𑀢𑀭𑀦𑁆𑀢𑀺 𑀅𑀡𑁆𑀡𑀯𑀁 𑀲𑀭𑀁,_  
_𑀲𑁂𑀢𑀼𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀯𑀺𑀲𑀚𑁆𑀚 𑀧𑀮𑁆𑀮𑀮𑀸𑀦𑀺;_  
_𑀓𑀼𑀮𑁆𑀮𑀜𑁆𑀳𑀺 𑀚𑀦𑁄 𑀩𑀦𑁆𑀥𑀢𑀺,_  
_𑀢𑀺𑀡𑁆𑀡𑀸 𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑁄 𑀚𑀦𑀸”𑀢𑀺𑁇_  


---
