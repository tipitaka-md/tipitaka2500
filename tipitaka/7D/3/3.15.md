# 𑁩𑁇𑁧𑁫 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀑𑀲𑁆𑀲𑀚𑁆𑀚𑀦

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀑𑀲𑁆𑀲𑀚𑁆𑀚𑀦

(169.)

362\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀘𑀸𑀧𑀸𑀮𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀑𑀲𑁆𑀲𑀚𑀺𑁇 𑀑𑀲𑁆𑀲𑀝𑁆𑀞𑁂 𑀘 𑀪𑀕𑀯𑀢𑀸 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀫𑀳𑀸𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀅𑀳𑁄𑀲𑀺 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

363\. _“𑀢𑀼𑀮𑀫𑀢𑀼𑀮𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀁,_  
_𑀪𑀯𑀲𑀗𑁆𑀔𑀸𑀭𑀫𑀯𑀲𑁆𑀲𑀚𑀺 𑀫𑀼𑀦𑀺;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀭𑀢𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑁄,_  
_𑀅𑀪𑀺𑀦𑁆𑀤𑀺 𑀓𑀯𑀘𑀫𑀺𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇_
