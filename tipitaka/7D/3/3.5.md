# 𑁩𑁇𑁫 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀆𑀤𑀻𑀦𑀯

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀆𑀤𑀻𑀦𑀯

(148.)

298\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑁄 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀦𑁆”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀲𑀦𑁆𑀣𑀭𑀺𑀁 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀤𑀓𑀫𑀡𑀺𑀓𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑁂𑀮𑀧𑀤𑀻𑀧𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀩𑁆𑀩𑀲𑀦𑁆𑀣𑀭𑀺𑀲𑀦𑁆𑀣𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁, 𑀆𑀲𑀦𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑀦𑀺, 𑀉𑀤𑀓𑀫𑀡𑀺𑀓𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁄, 𑀢𑁂𑀮𑀧𑀤𑀻𑀧𑁄 𑀆𑀭𑁄𑀧𑀺𑀢𑁄; 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀬𑁂𑀦 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀣𑀫𑁆𑀪𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀼𑀭𑀢𑁆𑀣𑀸𑀪𑀺𑀫𑀼𑀔𑁄 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄𑀧𑀺 𑀔𑁄 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀪𑀺𑀢𑁆𑀢𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀼𑀭𑀢𑁆𑀣𑀸𑀪𑀺𑀫𑀼𑀔𑁄 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀫𑁂𑀯 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸𑁇 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑀸𑀧𑀺 𑀔𑁄 𑀉𑀧𑀸𑀲𑀓𑀸 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀆𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀁 𑀪𑀺𑀢𑁆𑀢𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀪𑀺𑀫𑀼𑀔𑀸 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀫𑁂𑀯 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸𑁇

300\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑁂 𑀉𑀧𑀸𑀲𑀓𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀇𑀥, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀧𑀫𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀫𑀳𑀢𑀺𑀁 𑀪𑁄𑀕𑀚𑀸𑀦𑀺𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇

301\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇

302\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀬𑀜𑁆𑀜𑀤𑁂𑀯 𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺—  𑀬𑀤𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀁—  𑀅𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑁄𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇

303\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇

304\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇
