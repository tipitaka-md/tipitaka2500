# 𑁩𑁇𑁧𑁩 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄𑀪𑀸𑀲𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄𑀪𑀸𑀲𑀓𑀣𑀸

(166.)

353\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀲𑀻𑀤𑀦𑀁, 𑀬𑁂𑀦 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫 𑀤𑀺𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀆𑀤𑀸𑀬 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

(167.)

354\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀭𑀫𑀡𑀻𑀬𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀲𑀸𑀮𑀻, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀉𑀤𑁂𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀕𑁄𑀢𑀫𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁆𑀩𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀸𑀭𑀦𑁆𑀤𑀤𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁; 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀬𑀸𑀘𑀺—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺, 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑁇

355\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀭𑀫𑀡𑀻𑀬𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀲𑀸𑀮𑀻, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀉𑀤𑁂𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀕𑁄𑀢𑀫𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁆𑀩𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀸𑀭𑀦𑁆𑀤𑀤𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁; 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀬𑀸𑀘𑀺—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺, 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇
