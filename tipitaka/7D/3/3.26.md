# 𑁩𑁇𑁨𑁬 𑀬𑀫𑀓𑀲𑀸𑀮𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀬𑀫𑀓𑀲𑀸𑀮𑀸

(198.)

446\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀳𑀺𑀭𑀜𑁆𑀜𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀧𑀸𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀬𑁂𑀦 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀳𑀺𑀭𑀜𑁆𑀜𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀧𑀸𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀁, 𑀬𑁂𑀦 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑁆𑀢𑀭𑁂𑀦 𑀬𑀫𑀓𑀲𑀸𑀮𑀸𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀲𑀻𑀲𑀓𑀁 𑀫𑀜𑁆𑀘𑀓𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀳𑀺, 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑁂𑀦 𑀬𑀫𑀓𑀲𑀸𑀮𑀸𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀲𑀻𑀲𑀓𑀁 𑀫𑀜𑁆𑀘𑀓𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀻𑀳𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺 𑀧𑀸𑀤𑁂 𑀧𑀸𑀤𑀁 𑀅𑀘𑁆𑀘𑀸𑀥𑀸𑀬 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑁇

447\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀬𑀫𑀓𑀲𑀸𑀮𑀸 𑀲𑀩𑁆𑀩𑀨𑀸𑀮𑀺𑀨𑀼𑀮𑁆𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀓𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺𑁇 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀑𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀧𑁆𑀧𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀫𑀦𑁆𑀤𑀸𑀭𑀯𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀢𑀸𑀦𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀑𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀧𑁆𑀧𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀘𑀦𑁆𑀤𑀦𑀘𑀼𑀡𑁆𑀡𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀢𑀸𑀦𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀑𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀧𑁆𑀧𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀢𑀽𑀭𑀺𑀬𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀯𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀲𑀗𑁆𑀕𑀻𑀢𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇

(199.)

448\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀩𑁆𑀩𑀨𑀸𑀮𑀺𑀨𑀼𑀮𑁆𑀮𑀸 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀫𑀓𑀲𑀸𑀮𑀸 𑀅𑀓𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺𑁇 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀑𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀧𑁆𑀧𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀫𑀦𑁆𑀤𑀸𑀭𑀯𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀢𑀸𑀦𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀑𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀧𑁆𑀧𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀘𑀦𑁆𑀤𑀦𑀘𑀼𑀡𑁆𑀡𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀢𑀸𑀦𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀑𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀧𑁆𑀧𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀢𑀽𑀭𑀺𑀬𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀯𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑀺 𑀲𑀗𑁆𑀕𑀻𑀢𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀽𑀚𑀸𑀬𑁇 𑀦 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀓𑁆𑀓𑀢𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑁄 𑀯𑀸 𑀫𑀸𑀦𑀺𑀢𑁄 𑀯𑀸 𑀧𑀽𑀚𑀺𑀢𑁄 𑀯𑀸 𑀅𑀧𑀘𑀺𑀢𑁄 𑀯𑀸𑁇 𑀬𑁄 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑀸 𑀉𑀧𑀸𑀲𑀓𑁄 𑀯𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻, 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀁 𑀲𑀓𑁆𑀓𑀭𑁄𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀢𑀺 𑀫𑀸𑀦𑁂𑀢𑀺 𑀧𑀽𑀚𑁂𑀢𑀺 𑀅𑀧𑀘𑀺𑀬𑀢𑀺, 𑀧𑀭𑀫𑀸𑀬 𑀧𑀽𑀚𑀸𑀬𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
