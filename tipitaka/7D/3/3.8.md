# 𑁩𑁇𑁮 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀓𑀣𑀸

(155.)

325. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁄 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

326. “𑀘𑀢𑀼𑀦𑁆𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀢𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸, 𑀔𑀻𑀡𑀸 𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

327. _“𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀲𑀁𑀲𑀺𑀢𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀢𑀸𑀲𑀼 𑀢𑀸𑀲𑁆𑀯𑁂𑀯 𑀚𑀸𑀢𑀺𑀲𑀼𑁇_  


328. _𑀢𑀸𑀦𑀺 𑀏𑀢𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀺,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀫𑀽𑀮𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇_  


329. 𑀢𑀢𑁆𑀭𑀧𑀺 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀮𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀇𑀢𑀺 𑀲𑀻𑀮𑀁, 𑀇𑀢𑀺 𑀲𑀫𑀸𑀥𑀺, 𑀇𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑁇 𑀲𑀻𑀮𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄𑁇 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸𑁇 𑀧𑀜𑁆𑀜𑀸𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀆𑀲𑀯𑁂𑀳𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀸𑀲𑀯𑀸, 𑀪𑀯𑀸𑀲𑀯𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸”𑀢𑀺𑁇
