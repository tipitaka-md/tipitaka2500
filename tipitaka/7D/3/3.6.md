# 𑁩𑁇𑁬 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀆𑀦𑀺𑀲𑀁𑀲

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀆𑀦𑀺𑀲𑀁𑀲

(150.)

305\. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀇𑀥, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀫𑀳𑀦𑁆𑀢𑀁 𑀪𑁄𑀕𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇

306\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇

307\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀬𑀜𑁆𑀜𑀤𑁂𑀯 𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺—  𑀬𑀤𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀁 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀅𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑁄𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇

308\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇

309\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑀸”𑀢𑀺𑁇

(151.)

310\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑁂 𑀉𑀧𑀸𑀲𑀓𑁂 𑀩𑀳𑀼𑀤𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀲𑀺—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀬𑁄, 𑀭𑀢𑁆𑀢𑀺, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀢𑀼𑀫𑁆𑀳𑁂 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀣𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀲𑀼 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀓𑁂𑀲𑀼 𑀉𑀧𑀸𑀲𑀓𑁂𑀲𑀼 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇
