# 𑁩𑁇𑁩 𑀪𑀺𑀓𑁆𑀔𑀼𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑀥𑀫𑁆𑀫

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑀥𑀫𑁆𑀫

(136.)

254\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀯𑀲𑁆𑀲𑀓𑀸𑀭𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀫𑀕𑀥𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀸𑀚𑀕𑀳𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀸𑀚𑀕𑀳𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

255\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀢𑁆𑀢 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

256\. “𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀩𑀳𑀼𑀮𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁧)

257\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀁𑀖𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁨)

258\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀦 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑀣𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀲𑀫𑀸𑀤𑀸𑀬 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁩)

259\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑀸 𑀭𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘𑀺𑀭𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀲𑀁𑀖𑀧𑀺𑀢𑀭𑁄 𑀲𑀁𑀖𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀸, 𑀢𑁂 𑀲𑀓𑁆𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀽𑀚𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀜𑁆𑀘 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁪)

260\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑁄𑀦𑁄𑀩𑁆𑀪𑀯𑀺𑀓𑀸𑀬 𑀦 𑀯𑀲𑀁 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁫)

261\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀲𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁬)

262\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀜𑁆𑀜𑁂𑀯 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀕𑀢𑀸 𑀘 𑀧𑁂𑀲𑀮𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁, 𑀆𑀕𑀢𑀸 𑀘 𑀧𑁂𑀲𑀮𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁭)

263\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀲𑀢𑁆𑀢𑀲𑀼 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

(137.)

264\. 𑀅𑀧𑀭𑁂𑀧𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

265\. “𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦 𑀓𑀫𑁆𑀫𑀭𑀢𑀸 𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁧)

266\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦 𑀪𑀲𑁆𑀲𑀭𑀢𑀸 𑀦 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁨)

267\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀢𑀸 𑀦 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁩)

268\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀭𑀢𑀸 𑀦 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁪)

269\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀦𑀁 𑀇𑀘𑁆𑀙𑀸𑀦𑀁 𑀯𑀲𑀁 𑀕𑀢𑀸, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁫)

270\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦 𑀧𑀸𑀧𑀲𑀳𑀸𑀬𑀸 𑀦 𑀧𑀸𑀧𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑀸, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁬)

271\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀑𑀭𑀫𑀢𑁆𑀢𑀓𑁂𑀦 𑀯𑀺𑀲𑁂𑀲𑀸𑀥𑀺𑀕𑀫𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀸𑀯𑁄𑀲𑀸𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁭)

272\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀲𑀢𑁆𑀢𑀲𑀼 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

(138.)

273\. 𑀅𑀧𑀭𑁂𑀧𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺…𑀧𑁂… 𑁇 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀤𑁆𑀥𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀳𑀺𑀭𑀺𑀫𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀲𑀢𑁆𑀢𑀲𑀼 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

274\. 𑀅𑀧𑀭𑁂𑀧𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

278\. “𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀆𑀤𑀻𑀦𑀯𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺…  𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

279\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀲𑀢𑁆𑀢𑀲𑀼 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇

280\. 𑀙 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

283\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀬𑁂 𑀢𑁂 𑀮𑀸𑀪𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀸 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀧𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀳𑀺 𑀮𑀸𑀪𑁂𑀳𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀪𑀢𑁆𑀢𑀪𑁄𑀕𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀲𑀸𑀥𑀸𑀭𑀡𑀪𑁄𑀕𑀻, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁪)

284\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑀸𑀦𑀺 𑀅𑀔𑀡𑁆𑀟𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀸𑀦𑀺 𑀅𑀲𑀩𑀮𑀸𑀦𑀺 𑀅𑀓𑀫𑁆𑀫𑀸𑀲𑀸𑀦𑀺 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑀸𑀦𑀺 𑀯𑀺𑀜𑁆𑀜𑀽𑀧𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑀦𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀦𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀻𑀮𑀲𑀸𑀫𑀜𑁆𑀜𑀕𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁫)

285\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑀸𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀭𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸, 𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑀺 𑀢𑀓𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀸𑀫𑀜𑁆𑀜𑀕𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑁇 (𑁬)

286\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀙 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀙𑀲𑀼 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀻”𑀢𑀺𑁇

288\. 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀮𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀇𑀢𑀺 𑀲𑀻𑀮𑀁, 𑀇𑀢𑀺 𑀲𑀫𑀸𑀥𑀺, 𑀇𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑁇 𑀲𑀻𑀮𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄𑁇 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸𑁇 𑀧𑀜𑁆𑀜𑀸𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀆𑀲𑀯𑁂𑀳𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀸𑀲𑀯𑀸, 𑀪𑀯𑀸𑀲𑀯𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸”𑀢𑀺𑁇

(143.)

289\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀮𑀝𑁆𑀞𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀮𑀝𑁆𑀞𑀺𑀓𑀸 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀩𑀮𑀝𑁆𑀞𑀺𑀓𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀭𑀸𑀚𑀸𑀕𑀸𑀭𑀓𑁂𑁇 𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀩𑀮𑀝𑁆𑀞𑀺𑀓𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀭𑀸𑀚𑀸𑀕𑀸𑀭𑀓𑁂 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀮𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀇𑀢𑀺 𑀲𑀻𑀮𑀁 𑀇𑀢𑀺 𑀲𑀫𑀸𑀥𑀺 𑀇𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑁇 𑀲𑀻𑀮𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄𑁇 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸𑁇 𑀧𑀜𑁆𑀜𑀸𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀆𑀲𑀯𑁂𑀳𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀸𑀲𑀯𑀸, 𑀪𑀯𑀸𑀲𑀯𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸”𑀢𑀺𑁇

290\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀩𑀮𑀝𑁆𑀞𑀺𑀓𑀸𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸 𑀢𑀤𑀯𑀲𑀭𑀺, 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀍𑀅𑀦𑁆𑀤𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀸𑀯𑀸𑀭𑀺𑀓𑀫𑁆𑀩𑀯𑀦𑁂𑁇
