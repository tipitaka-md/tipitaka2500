# 𑁩𑁇𑁩𑁭 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀻𑀭𑀧𑀽𑀚𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀩𑀼𑀤𑁆𑀥𑀲𑀭𑀻𑀭𑀧𑀽𑀚𑀸

(227.)

503. 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀧𑀼𑀭𑀺𑀲𑁂 𑀆𑀡𑀸𑀧𑁂𑀲𑀼𑀁—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀢𑀸𑀍𑀆𑀯𑀘𑀭𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀢𑀸𑀍𑀆𑀯𑀘𑀭𑀁 𑀧𑀜𑁆𑀘 𑀘 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀢𑀸𑀦𑀺 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑀁, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀯𑀸𑀤𑀺𑀢𑁂𑀳𑀺 𑀫𑀸𑀮𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀸 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀸 𑀘𑁂𑀮𑀯𑀺𑀢𑀸𑀦𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀏 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀦𑁆𑀢𑀸 𑀏𑀓𑀤𑀺𑀯𑀲𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀼𑀁𑁇

504. 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀢𑀺𑀯𑀺𑀓𑀸𑀮𑁄 𑀔𑁄 𑀅𑀚𑁆𑀚 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀛𑀸𑀧𑁂𑀢𑀼𑀁, 𑀲𑁆𑀯𑁂 𑀤𑀸𑀦𑀺 𑀫𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀛𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀯𑀸𑀤𑀺𑀢𑁂𑀳𑀺 𑀫𑀸𑀮𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀸 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀸 𑀘𑁂𑀮𑀯𑀺𑀢𑀸𑀦𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀏 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀦𑁆𑀢𑀸 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀤𑀺𑀯𑀲𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀼𑀁, 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀤𑀺𑀯𑀲𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀼𑀁, 𑀘𑀢𑀼𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀤𑀺𑀯𑀲𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀼𑀁, 𑀧𑀜𑁆𑀘𑀫𑀫𑁆𑀧𑀺 𑀤𑀺𑀯𑀲𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀼𑀁, 𑀙𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀤𑀺𑀯𑀲𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀼𑀁𑁇

505. 𑀅𑀣 𑀔𑁄 𑀲𑀢𑁆𑀢𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀯𑀸𑀤𑀺𑀢𑁂𑀳𑀺 𑀫𑀸𑀮𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀸 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀦𑀕𑀭𑀲𑁆𑀲 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀩𑀸𑀳𑀺𑀭𑁂𑀦 𑀩𑀸𑀳𑀺𑀭𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀛𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

(228.)

506. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀝𑁆𑀞 𑀫𑀮𑁆𑀮𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑀸 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑀸 𑀅𑀳𑀢𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀦𑀺𑀯𑀢𑁆𑀣𑀸—  “𑀫𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑀺𑀫𑁂 𑀅𑀝𑁆𑀞 𑀫𑀮𑁆𑀮𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑀸 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑀸 𑀅𑀳𑀢𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀦𑀺𑀯𑀢𑁆𑀣𑀸—  ‘𑀫𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀣𑀸 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄, 𑀅𑀜𑁆𑀜𑀣𑀸 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺𑁇 “𑀓𑀣𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺? “𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄—  ‘𑀫𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀯𑀸𑀤𑀺𑀢𑁂𑀳𑀺 𑀫𑀸𑀮𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀸 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀦𑀕𑀭𑀲𑁆𑀲 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀩𑀸𑀳𑀺𑀭𑁂𑀦 𑀩𑀸𑀳𑀺𑀭𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀛𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺; 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄—  ‘𑀫𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀯𑀸𑀤𑀺𑀢𑁂𑀳𑀺 𑀫𑀸𑀮𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀸 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀁 𑀦𑀕𑀭𑀲𑁆𑀲 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀕𑀭𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂𑀦 𑀫𑀚𑁆𑀛𑀁 𑀦𑀕𑀭𑀲𑁆𑀲 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀫𑀓𑀼𑀝𑀩𑀦𑁆𑀥𑀦𑀁 𑀦𑀸𑀫 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀛𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸’”𑀢𑀺𑁇 “𑀬𑀣𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄, 𑀢𑀣𑀸 𑀳𑁄𑀢𑀽”𑀢𑀺𑁇

(229.)

507. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀬𑀸𑀯 𑀲𑀦𑁆𑀥𑀺𑀲𑀫𑀮𑀲𑀗𑁆𑀓𑀝𑀻𑀭𑀸 𑀚𑀡𑁆𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀦 𑀑𑀥𑀺𑀦𑀸 𑀫𑀦𑁆𑀤𑀸𑀭𑀯𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺 𑀲𑀦𑁆𑀣𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀤𑁂𑀯𑀢𑀸 𑀘 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀘 𑀫𑀮𑁆𑀮𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀘 𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀳𑀺 𑀘 𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀯𑀸𑀤𑀺𑀢𑁂𑀳𑀺 𑀫𑀸𑀮𑁂𑀳𑀺 𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀸 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀁 𑀦𑀕𑀭𑀲𑁆𑀲 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀕𑀭𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂𑀦 𑀫𑀚𑁆𑀛𑀁 𑀦𑀕𑀭𑀲𑁆𑀲 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀫𑀓𑀼𑀝𑀩𑀦𑁆𑀥𑀦𑀁 𑀦𑀸𑀫 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀏𑀢𑁆𑀣 𑀘 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀁𑀲𑀼𑁇

508. 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀣𑀁 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀸”𑀢𑀺? “𑀬𑀣𑀸 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀏𑀯𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 “𑀓𑀣𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀭𑀜𑁆𑀜𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀅𑀳𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦 𑀯𑁂𑀞𑁂𑀦𑁆𑀢𑀺, 𑀅𑀳𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀢𑁂𑀦 𑀓𑀧𑁆𑀧𑀸𑀲𑁂𑀦 𑀯𑁂𑀞𑁂𑀦𑁆𑀢𑀺, 𑀯𑀺𑀳𑀢𑁂𑀦 𑀓𑀧𑁆𑀧𑀸𑀲𑁂𑀦 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀅𑀳𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦 𑀯𑁂𑀞𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁂𑀦 𑀉𑀧𑀸𑀬𑁂𑀦 𑀧𑀜𑁆𑀘𑀳𑀺 𑀬𑀼𑀕𑀲𑀢𑁂𑀳𑀺 𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑀸𑀬 𑀢𑁂𑀮𑀤𑁄𑀡𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀆𑀬𑀲𑀸𑀬 𑀤𑁄𑀡𑀺𑀬𑀸 𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑀓𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁 𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀧𑀣𑁂 𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀣𑀽𑀧𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀬𑀣𑀸 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀏𑀯𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀧𑀣𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀣𑀽𑀧𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑁆𑀣 𑀬𑁂 𑀫𑀸𑀮𑀁 𑀯𑀸 𑀕𑀦𑁆𑀥𑀁 𑀯𑀸 𑀘𑀼𑀡𑁆𑀡𑀓𑀁 𑀯𑀸 𑀆𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀧𑀼𑀭𑀺𑀲𑁂 𑀆𑀡𑀸𑀧𑁂𑀲𑀼𑀁—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀯𑀺𑀳𑀢𑀁 𑀓𑀧𑁆𑀧𑀸𑀲𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀣𑀸”𑀢𑀺𑁇

509. 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀅𑀳𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀢𑁂𑀦 𑀓𑀧𑁆𑀧𑀸𑀲𑁂𑀦 𑀯𑁂𑀞𑁂𑀲𑀼𑀁, 𑀯𑀺𑀳𑀢𑁂𑀦 𑀓𑀧𑁆𑀧𑀸𑀲𑁂𑀦 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀅𑀳𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑁂𑀦 𑀯𑁂𑀞𑁂𑀲𑀼𑀁𑁇 𑀏𑀢𑁂𑀦 𑀉𑀧𑀸𑀬𑁂𑀦 𑀧𑀜𑁆𑀘𑀳𑀺 𑀬𑀼𑀕𑀲𑀢𑁂𑀳𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑀸𑀬 𑀢𑁂𑀮𑀤𑁄𑀡𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀆𑀬𑀲𑀸𑀬 𑀤𑁄𑀡𑀺𑀬𑀸 𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑀓𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀁 𑀘𑀺𑀢𑀓𑀁 𑀆𑀭𑁄𑀧𑁂𑀲𑀼𑀁𑁇
