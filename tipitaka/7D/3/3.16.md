# 𑁩𑁇𑁧𑁬 𑀫𑀳𑀸𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀳𑁂𑀢𑀼

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀳𑁂𑀢𑀼

(170.)

364\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄; 𑀲𑀼𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑁄; 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑀸”𑀢𑀺?

365\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄; 𑀲𑀼𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑁄; 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑀸”𑀢𑀺?

(171.)

366\. “𑀅𑀝𑁆𑀞 𑀔𑁄 𑀇𑀫𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀳𑁂𑀢𑀽, 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀧𑀣𑀯𑀻 𑀉𑀤𑀓𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸, 𑀉𑀤𑀓𑀁 𑀯𑀸𑀢𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁, 𑀯𑀸𑀢𑁄 𑀆𑀓𑀸𑀲𑀝𑁆𑀞𑁄𑁇 𑀳𑁄𑀢𑀺 𑀔𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁄, 𑀬𑀁 𑀫𑀳𑀸𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀫𑀳𑀸𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀸 𑀉𑀤𑀓𑀁 𑀓𑀫𑁆𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀉𑀤𑀓𑀁 𑀓𑀫𑁆𑀧𑀺𑀢𑀁 𑀧𑀣𑀯𑀺𑀁 𑀓𑀫𑁆𑀧𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀳𑁂𑀢𑀼 𑀧𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇

367\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀘𑁂𑀢𑁄𑀯𑀲𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀆𑀧𑁄𑀲𑀜𑁆𑀜𑀸𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀧𑀣𑀯𑀺𑀁 𑀓𑀫𑁆𑀧𑁂𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇

368\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀼𑀲𑀺𑀢𑀓𑀸𑀬𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼 𑀢𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇

369\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀫𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀳𑁂𑀢𑀼 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇

370\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀳𑁂𑀢𑀼 𑀧𑀜𑁆𑀘𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇

371\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀳𑁂𑀢𑀼 𑀙𑀝𑁆𑀞𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇

372\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀑𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀳𑁂𑀢𑀼 𑀲𑀢𑁆𑀢𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇

373\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀳𑁂𑀢𑀼 𑀅𑀝𑁆𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽, 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇
