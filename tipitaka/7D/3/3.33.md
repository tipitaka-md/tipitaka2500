# 𑁩𑁇𑁩𑁩 𑀫𑀮𑁆𑀮𑀸𑀦𑀁𑀯𑀦𑁆𑀤𑀦𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀮𑁆𑀮𑀸𑀦𑀁𑀯𑀦𑁆𑀤𑀦𑀸

(211.)

473. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀲𑀦𑁆𑀥𑀸𑀕𑀸𑀭𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀕𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀅𑀚𑁆𑀚 𑀔𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀣, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀣, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸𑁇 𑀫𑀸 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄 𑀅𑀳𑀼𑀯𑀢𑁆𑀣—  ‘𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘 𑀦𑁄 𑀕𑀸𑀫𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀦 𑀫𑀬𑀁 𑀮𑀪𑀺𑀫𑁆𑀳𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀓𑀸𑀮𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’”𑀢𑀺𑁇

474. 𑀇𑀤𑀫𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀮𑁆𑀮𑀸 𑀘 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀫𑀮𑁆𑀮𑀲𑀼𑀡𑀺𑀲𑀸 𑀘 𑀫𑀮𑁆𑀮𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀘 𑀅𑀖𑀸𑀯𑀺𑀦𑁄 𑀤𑀼𑀫𑁆𑀫𑀦𑀸 𑀘𑁂𑀢𑁄𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀓𑁂𑀲𑁂 𑀧𑀓𑀺𑀭𑀺𑀬 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀧𑀸𑀢𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀆𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺—  “𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀼𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀮𑁆𑀮𑀸 𑀘 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀫𑀮𑁆𑀮𑀲𑀼𑀡𑀺𑀲𑀸 𑀘 𑀫𑀮𑁆𑀮𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀘 𑀅𑀖𑀸𑀯𑀺𑀦𑁄 𑀤𑀼𑀫𑁆𑀫𑀦𑀸 𑀘𑁂𑀢𑁄𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸 𑀬𑁂𑀦 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀅𑀳𑀁 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑁂 𑀫𑀮𑁆𑀮𑁂 𑀏𑀓𑀫𑁂𑀓𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀯𑀦𑁆𑀤𑀺𑀢𑁄 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑁂𑀳𑀺 𑀫𑀮𑁆𑀮𑁂𑀳𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀭𑀢𑁆𑀢𑀺 𑀯𑀺𑀪𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑁂 𑀫𑀮𑁆𑀮𑁂 𑀓𑀼𑀮𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀲𑁄 𑀓𑀼𑀮𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀲𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬𑀁—  ‘𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀮𑁆𑀮𑁄 𑀲𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀪𑀭𑀺𑀬𑁄 𑀲𑀧𑀭𑀺𑀲𑁄 𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑁂 𑀫𑀮𑁆𑀮𑁂 𑀓𑀼𑀮𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀲𑁄 𑀓𑀼𑀮𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀲𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀮𑁆𑀮𑁄 𑀲𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀪𑀭𑀺𑀬𑁄 𑀲𑀧𑀭𑀺𑀲𑁄 𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀏𑀢𑁂𑀦 𑀉𑀧𑀸𑀬𑁂𑀦 𑀧𑀞𑀫𑁂𑀦𑁂𑀯 𑀬𑀸𑀫𑁂𑀦 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑁂 𑀫𑀮𑁆𑀮𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺𑁇
