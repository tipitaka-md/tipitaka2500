# 𑁩𑁇𑁩𑁬 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀓𑀣𑀸

(219.)

487. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇

488. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀦𑀸𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄”𑀢𑀺𑁇

489. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑁇

(220.)

490. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀲𑀳 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸 𑀫𑀳𑀸𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀅𑀳𑁄𑀲𑀺 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑁄𑁇 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀲𑀳 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

491. _“𑀲𑀩𑁆𑀩𑁂𑀯 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀪𑀽𑀢𑀸 𑀮𑁄𑀓𑁂 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀁;_  
_𑀬𑀢𑁆𑀣 𑀏𑀢𑀸𑀤𑀺𑀲𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀮𑁄𑀓𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀢𑀣𑀸𑀕𑀢𑁄 𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇_  


(221.)

492. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀲𑀳 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

493. _“𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀢 𑀲𑀗𑁆𑀔𑀸𑀭𑀸,_  
_𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀬𑀥𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀲𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀲𑀼𑀔𑁄”𑀢𑀺𑁇_  


(222.)

494. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀲𑀳 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

495. _“𑀦𑀸𑀳𑀼 𑀅𑀲𑁆𑀲𑀸𑀲𑀧𑀲𑁆𑀲𑀸𑀲𑁄,_  
_𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀅𑀦𑁂𑀚𑁄 𑀲𑀦𑁆𑀢𑀺𑀫𑀸𑀭𑀩𑁆𑀪,_  
_𑀬𑀁 𑀓𑀸𑀮𑀫𑀓𑀭𑀻 𑀫𑀼𑀦𑀺𑁇_  


496. _𑀅𑀲𑀮𑁆𑀮𑀻𑀦𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀯𑁂𑀤𑀦𑀁 𑀅𑀚𑁆𑀛𑀯𑀸𑀲𑀬𑀺;_  
_𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑁆𑀲𑁂𑀯 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀘𑁂𑀢𑀲𑁄 𑀅𑀳𑀽”𑀢𑀺𑁇_  


(223.)

497. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀲𑀳 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

498. _“𑀢𑀤𑀸𑀲𑀺 𑀬𑀁 𑀪𑀺𑀁𑀲𑀦𑀓𑀁,_  
_𑀢𑀤𑀸𑀲𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀓𑀸𑀭𑀯𑀭𑀽𑀧𑁂𑀢𑁂,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂”𑀢𑀺𑁇_  


(224.)

499. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀬𑁂 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀧𑀸𑀢𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀆𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺, “𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀼𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄”𑀢𑀺𑁇 𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀻𑀢𑀭𑀸𑀕𑀸, 𑀢𑁂 𑀲𑀢𑀸 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣 𑀮𑀩𑁆𑀪𑀸”𑀢𑀺𑁇

(225.)

500. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸 𑀲𑁄𑀘𑀺𑀢𑁆𑀣 𑀫𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑁆𑀣𑁇 𑀦𑀦𑀼 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀅𑀓𑁆𑀔𑀸𑀢𑀁—  ‘𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀧𑀺𑀬𑁂𑀳𑀺 𑀫𑀦𑀸𑀧𑁂𑀳𑀺 𑀦𑀸𑀦𑀸𑀪𑀸𑀯𑁄 𑀯𑀺𑀦𑀸𑀪𑀸𑀯𑁄 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑁄’𑁇 𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣, 𑀆𑀯𑀼𑀲𑁄, 𑀮𑀩𑁆𑀪𑀸𑁇 ‘𑀬𑀁 𑀢𑀁 𑀚𑀸𑀢𑀁 𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀢𑀁 𑀧𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀁, 𑀢𑀁 𑀯𑀢 𑀫𑀸 𑀧𑀮𑀼𑀚𑁆𑀚𑀻’𑀢𑀺, 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑁂𑀯𑀢𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀣𑀁𑀪𑀽𑀢𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀤𑁂𑀯𑀢𑀸 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀻”𑀢𑀺?

501. “𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀤𑁂𑀯𑀢𑀸 𑀆𑀓𑀸𑀲𑁂 𑀧𑀣𑀯𑀻𑀲𑀜𑁆𑀜𑀺𑀦𑀺𑀬𑁄 𑀓𑁂𑀲𑁂 𑀧𑀓𑀺𑀭𑀺𑀬 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀧𑀸𑀢𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀆𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀼𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄’𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀤𑁂𑀯𑀢𑀸 𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀣𑀯𑀻𑀲𑀜𑁆𑀜𑀺𑀦𑀺𑀬𑁄 𑀓𑁂𑀲𑁂 𑀧𑀓𑀺𑀭𑀺𑀬 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀧𑀸𑀢𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀆𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀼𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄’𑀢𑀺𑁇 𑀬𑀸 𑀧𑀦 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀯𑀻𑀢𑀭𑀸𑀕𑀸, 𑀢𑀸 𑀲𑀢𑀸 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣 𑀮𑀩𑁆𑀪𑀸’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀁 𑀭𑀢𑁆𑀢𑀸𑀯𑀲𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀼𑀁𑁇

(226.)

502. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀺—  ‘𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀪𑀕𑀯𑀸, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀣𑀸’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀅𑀢𑁆𑀢𑀤𑀼𑀢𑀺𑀬𑁄 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀲𑀦𑁆𑀥𑀸𑀕𑀸𑀭𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀢𑁂𑀦𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀕𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀸, 𑀪𑀕𑀯𑀸, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀣𑀸”𑀢𑀺𑁇 𑀇𑀤𑀫𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀮𑁆𑀮𑀸 𑀘 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀫𑀮𑁆𑀮𑀲𑀼𑀡𑀺𑀲𑀸 𑀘 𑀫𑀮𑁆𑀮𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀘 𑀅𑀖𑀸𑀯𑀺𑀦𑁄 𑀤𑀼𑀫𑁆𑀫𑀦𑀸 𑀘𑁂𑀢𑁄𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀓𑁂𑀲𑁂 𑀧𑀓𑀺𑀭𑀺𑀬 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀧𑀸𑀢𑀁 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀆𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑁆𑀢𑀺—  “𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀼𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀅𑀢𑀺𑀔𑀺𑀧𑁆𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄”𑀢𑀺𑁇
