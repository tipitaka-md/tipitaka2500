# 𑁩𑁇𑁨𑁨 𑀘𑀢𑀼𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀓𑀣𑀸

(187.)

414\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀡𑁆𑀟𑀕𑀸𑀫𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀳𑀢𑁆𑀣𑀺𑀕𑀸𑀫𑁄, 𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀕𑀸𑀫𑁄, 𑀬𑁂𑀦 𑀚𑀫𑁆𑀩𑀼𑀕𑀸𑀫𑁄, 𑀬𑁂𑀦 𑀪𑁄𑀕𑀦𑀕𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀪𑁄𑀕𑀦𑀕𑀭𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀪𑁄𑀕𑀦𑀕𑀭𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀦𑀦𑁆𑀤𑁂 𑀘𑁂𑀢𑀺𑀬𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

(188.)

415\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀫𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀦𑁂𑀯 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀦𑀺 𑀧𑀤𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸𑀦𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀦 𑀘𑁂𑀯 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀦 𑀘𑁂𑀯 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀇𑀫𑀲𑁆𑀲 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀢𑀸𑀦𑀺 𑀘𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀯 𑀑𑀲𑀭𑀦𑁆𑀢𑀺, 𑀯𑀺𑀦𑀬𑁂 𑀘 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀇𑀫𑀲𑁆𑀲 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀁 𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑁇 (𑁧)

416\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀦𑀸𑀫 𑀆𑀯𑀸𑀲𑁂 𑀲𑀁𑀖𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀣𑁂𑀭𑁄 𑀲𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀦𑁂𑀯 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀦𑀺 𑀧𑀤𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸𑀦𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀦 𑀘𑁂𑀯 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀦 𑀘𑁂𑀯 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀢𑀲𑁆𑀲 𑀘 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀢𑀺𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀢𑀸𑀦𑀺 𑀘𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀯 𑀑𑀲𑀭𑀦𑁆𑀢𑀺, 𑀯𑀺𑀦𑀬𑁂 𑀘 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀢𑀲𑁆𑀲 𑀘 𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑁇 (𑁨)

417\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀦𑀸𑀫 𑀆𑀯𑀸𑀲𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀆𑀕𑀢𑀸𑀕𑀫𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀯𑀺𑀦𑀬𑀥𑀭𑀸 𑀫𑀸𑀢𑀺𑀓𑀸𑀥𑀭𑀸𑁇 𑀢𑁂𑀲𑀁 𑀫𑁂 𑀣𑁂𑀭𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀦𑁂𑀯 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀦 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀦 𑀘𑁂𑀯 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀢𑁂𑀲𑀜𑁆𑀘 𑀣𑁂𑀭𑀸𑀦𑀁 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀢𑀺𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀢𑀸𑀦𑀺 𑀘𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺…𑀧𑁂…  𑀯𑀺𑀦𑀬𑁂 𑀘 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀢𑁂𑀲𑀜𑁆𑀘 𑀣𑁂𑀭𑀸𑀦𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑀁 𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑁇 (𑁩)

418\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀦𑀸𑀫 𑀆𑀯𑀸𑀲𑁂 𑀏𑀓𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀆𑀕𑀢𑀸𑀕𑀫𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 𑀫𑀸𑀢𑀺𑀓𑀸𑀥𑀭𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀣𑁂𑀭𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀦𑁂𑀯 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀦𑀺 𑀧𑀤𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸𑀦𑀺 𑀲𑀸𑀥𑀼𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀦 𑀘𑁂𑀯 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀦 𑀘𑁂𑀯 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀢𑀲𑁆𑀲 𑀘 𑀣𑁂𑀭𑀲𑁆𑀲 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀢𑀺𑀳𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀢𑀸𑀦𑀺 𑀘 𑀲𑀼𑀢𑁆𑀢𑁂 𑀑𑀲𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀯𑀺𑀦𑀬𑁂 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑁂 𑀘𑁂𑀯 𑀑𑀲𑀭𑀦𑁆𑀢𑀺, 𑀯𑀺𑀦𑀬𑁂 𑀘 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀤𑁆𑀥𑀸 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁; 𑀢𑀲𑁆𑀲 𑀘 𑀣𑁂𑀭𑀲𑁆𑀲 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑁇 (𑁪) 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑁂 𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺𑁇

419\. 𑀢𑀢𑁆𑀭𑀧𑀺 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀪𑁄𑀕𑀦𑀕𑀭𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀮𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀇𑀢𑀺 𑀲𑀻𑀮𑀁, 𑀇𑀢𑀺 𑀲𑀫𑀸𑀥𑀺, 𑀇𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑁇 𑀲𑀻𑀮𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄𑁇 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸𑁇 𑀧𑀜𑁆𑀜𑀸𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀆𑀲𑀯𑁂𑀳𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀸𑀲𑀯𑀸, 𑀪𑀯𑀸𑀲𑀯𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸”𑀢𑀺𑁇
