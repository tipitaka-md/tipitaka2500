# 𑁩𑁇𑁨𑁫 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(192.)

429\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀍𑀆𑀭𑀲𑁆𑀲 𑀓𑀸𑀮𑀸𑀫𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬 𑀧𑀸𑀯𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀦𑁆𑀢𑁂𑀦 𑀯𑀢, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀆𑀍𑀆𑀭𑀁 𑀓𑀸𑀮𑀸𑀫𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀲𑁆𑀲 𑀲𑀓𑀝𑀲𑀢𑁆𑀣𑀲𑁆𑀲 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀬𑁂𑀦 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀍𑀆𑀭𑀁 𑀓𑀸𑀮𑀸𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀅𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀅𑀤𑁆𑀤𑀲𑀸’𑀢𑀺? ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀤𑁆𑀤𑀲𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀤𑁆𑀤𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀤𑁆𑀤𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀜𑁆𑀜𑀻 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺? ‘𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄’𑀢𑀺𑁇 ‘𑀲𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀕𑀭𑁄 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀦𑁂𑀯 𑀅𑀤𑁆𑀤𑀲, 𑀦 𑀧𑀦 𑀲𑀤𑁆𑀤𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀺; 𑀅𑀧𑀺𑀲𑀼 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺 𑀭𑀚𑁂𑀦 𑀑𑀓𑀺𑀡𑁆𑀡𑀸’𑀢𑀺? ‘𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀲𑀦𑁆𑀢𑁂𑀦 𑀯𑀢 𑀪𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀕𑀭𑁄 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀦𑁂𑀯 𑀤𑀓𑁆𑀔𑀢𑀺, 𑀦 𑀧𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑁄𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀆𑀍𑀆𑀭𑁂 𑀓𑀸𑀮𑀸𑀫𑁂 𑀉𑀍𑀆𑀭𑀁 𑀧𑀲𑀸𑀤𑀁 𑀧𑀯𑁂𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀻”𑀢𑀺𑁇

430\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀤𑀼𑀓𑁆𑀓𑀭𑀢𑀭𑀁 𑀯𑀸 𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀢𑀭𑀁 𑀯𑀸—  𑀬𑁄 𑀯𑀸 𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀕𑀭𑁄 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀦𑁂𑀯 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀧𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬; 𑀬𑁄 𑀯𑀸 𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀕𑀭𑁄 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑁂 𑀕𑀍𑀅𑀕𑀍𑀆𑀬𑀦𑁆𑀢𑁂 𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑁆𑀮𑀢𑀸𑀲𑀼 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀻𑀲𑀼 𑀅𑀲𑀦𑀺𑀬𑀸 𑀨𑀮𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀧𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀓𑀺𑀜𑁆𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀘 𑀯𑀸 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀙 𑀯𑀸 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀯𑀸 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀯𑀸 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀦𑀯 𑀯𑀸 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺, 𑀲𑀓𑀝𑀲𑀳𑀲𑁆𑀲𑀁 𑀯𑀸 𑀲𑀓𑀝𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀯𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀏𑀢𑀤𑁂𑀯 𑀤𑀼𑀓𑁆𑀓𑀭𑀢𑀭𑀜𑁆𑀘𑁂𑀯 𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀢𑀭𑀜𑁆𑀘 𑀬𑁄 𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀕𑀭𑁄 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑁂 𑀕𑀍𑀅𑀕𑀍𑀆𑀬𑀦𑁆𑀢𑁂 𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑁆𑀮𑀢𑀸𑀲𑀼 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀻𑀲𑀼 𑀅𑀲𑀦𑀺𑀬𑀸 𑀨𑀮𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀧𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

431\. “𑀏𑀓𑀫𑀺𑀤𑀸𑀳𑀁, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀲𑀫𑀬𑀁 𑀆𑀢𑀼𑀫𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀪𑀼𑀲𑀸𑀕𑀸𑀭𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑁂 𑀕𑀍𑀅𑀕𑀍𑀆𑀬𑀦𑁆𑀢𑁂 𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑁆𑀮𑀢𑀸𑀲𑀼 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀻𑀲𑀼 𑀅𑀲𑀦𑀺𑀬𑀸 𑀨𑀮𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀪𑀼𑀲𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀓𑀲𑁆𑀲𑀓𑀸 𑀪𑀸𑀢𑀭𑁄 𑀳𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀩𑀮𑀺𑀩𑀤𑁆𑀤𑀸𑁇 𑀅𑀣 𑀔𑁄, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀆𑀢𑀼𑀫𑀸𑀬 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁂 𑀤𑁆𑀯𑁂 𑀓𑀲𑁆𑀲𑀓𑀸 𑀪𑀸𑀢𑀭𑁄 𑀳𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀩𑀮𑀺𑀩𑀤𑁆𑀤𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀲𑀫𑀬𑁂𑀦 𑀪𑀼𑀲𑀸𑀕𑀸𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀼𑀲𑀸𑀕𑀸𑀭𑀤𑁆𑀯𑀸𑀭𑁂 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀏𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄’𑀢𑀺? ‘𑀇𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑁂 𑀕𑀍𑀅𑀕𑀍𑀆𑀬𑀦𑁆𑀢𑁂 𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑁆𑀮𑀢𑀸𑀲𑀼 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀻𑀲𑀼 𑀅𑀲𑀦𑀺𑀬𑀸 𑀨𑀮𑀦𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀓𑀲𑁆𑀲𑀓𑀸 𑀪𑀸𑀢𑀭𑁄 𑀳𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀩𑀮𑀺𑀩𑀤𑁆𑀤𑀸𑁇 𑀏𑀢𑁆𑀣𑁂𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄𑁇 𑀢𑁆𑀯𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁆𑀯 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺? ‘𑀇𑀥𑁂𑀯 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀤𑁆𑀤𑀲𑀸’𑀢𑀺? ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀤𑁆𑀤𑀲𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀤𑁆𑀤𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀤𑁆𑀤𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀜𑁆𑀜𑀻 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺? ‘𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄’𑀢𑀺𑁇 ‘𑀲𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀕𑀭𑁄 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑁂 𑀕𑀍𑀅𑀕𑀍𑀆𑀬𑀦𑁆𑀢𑁂 𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑁆𑀮𑀢𑀸𑀲𑀼 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀻𑀲𑀼 𑀅𑀲𑀦𑀺𑀬𑀸 𑀨𑀮𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀅𑀤𑁆𑀤𑀲, 𑀦 𑀧𑀦 𑀲𑀤𑁆𑀤𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀻’𑀢𑀺? ‘𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄’𑀢𑀺?

432\. 𑀅𑀣 𑀔𑁄, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀲𑀦𑁆𑀢𑁂𑀦 𑀯𑀢 𑀪𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀦𑁄 𑀚𑀸𑀕𑀭𑁄 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑁂 𑀕𑀍𑀅𑀕𑀍𑀆𑀬𑀦𑁆𑀢𑁂 𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑁆𑀮𑀢𑀸𑀲𑀼 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀻𑀲𑀼 𑀅𑀲𑀦𑀺𑀬𑀸 𑀨𑀮𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀤𑀓𑁆𑀔𑀢𑀺, 𑀦 𑀧𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑁄𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀫𑀬𑀺 𑀉𑀍𑀆𑀭𑀁 𑀧𑀲𑀸𑀤𑀁 𑀧𑀯𑁂𑀤𑁂𑀢𑁆𑀯𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀻”𑀢𑀺𑁇

433\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁄 𑀫𑁂 𑀆𑀍𑀆𑀭𑁂 𑀓𑀸𑀮𑀸𑀫𑁂 𑀧𑀲𑀸𑀤𑁄 𑀢𑀁 𑀫𑀳𑀸𑀯𑀸𑀢𑁂 𑀯𑀸 𑀑𑀨𑀼𑀡𑀸𑀫𑀺 𑀲𑀻𑀖𑀲𑁄𑀢𑀸𑀬 𑀯𑀸 𑀦𑀤𑀺𑀬𑀸 𑀧𑀯𑀸𑀳𑁂𑀫𑀺𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

434\. 𑀅𑀣 𑀔𑁄 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀪𑀡𑁂, 𑀲𑀺𑀗𑁆𑀕𑀻𑀯𑀡𑁆𑀡𑀁 𑀬𑀼𑀕𑀫𑀝𑁆𑀞𑀁 𑀥𑀸𑀭𑀡𑀻𑀬𑀁 𑀆𑀳𑀭𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀲𑁆𑀲 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀲𑀺𑀗𑁆𑀕𑀻𑀯𑀡𑁆𑀡𑀁 𑀬𑀼𑀕𑀫𑀝𑁆𑀞𑀁 𑀥𑀸𑀭𑀡𑀻𑀬𑀁 𑀆𑀳𑀭𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀁 𑀲𑀺𑀗𑁆𑀕𑀻𑀯𑀡𑁆𑀡𑀁 𑀬𑀼𑀕𑀫𑀝𑁆𑀞𑀁 𑀥𑀸𑀭𑀡𑀻𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀺𑀗𑁆𑀕𑀻𑀯𑀡𑁆𑀡𑀁 𑀬𑀼𑀕𑀫𑀝𑁆𑀞𑀁 𑀥𑀸𑀭𑀡𑀻𑀬𑀁, 𑀢𑀁 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀧𑀼𑀓𑁆𑀓𑀼𑀲, 𑀏𑀓𑁂𑀦 𑀫𑀁 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀳𑀺, 𑀏𑀓𑁂𑀦 𑀆𑀦𑀦𑁆𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁂𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀏𑀓𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

435\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁂 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁂 𑀢𑀁 𑀲𑀺𑀗𑁆𑀕𑀻𑀯𑀡𑁆𑀡𑀁 𑀬𑀼𑀕𑀫𑀝𑁆𑀞𑀁 𑀥𑀸𑀭𑀡𑀻𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺𑁇 𑀢𑀁 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀸𑀫𑀺𑀢𑀁 𑀳𑀢𑀘𑁆𑀘𑀺𑀓𑀁 𑀯𑀺𑀬 𑀔𑀸𑀬𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀺𑀗𑁆𑀕𑀻𑀯𑀡𑁆𑀡𑀁 𑀬𑀼𑀕𑀫𑀝𑁆𑀞𑀁 𑀥𑀸𑀭𑀡𑀻𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀸𑀫𑀺𑀢𑀁 𑀳𑀢𑀘𑁆𑀘𑀺𑀓𑀁 𑀯𑀺𑀬 𑀔𑀸𑀬𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑁆𑀯𑀻𑀲𑀼 𑀓𑀸𑀮𑁂𑀲𑀼 𑀅𑀢𑀺𑀯𑀺𑀬 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀓𑀢𑀫𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼? 𑀬𑀜𑁆𑀘, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑁆𑀯𑀻𑀲𑀼 𑀓𑀸𑀮𑁂𑀲𑀼 𑀅𑀢𑀺𑀯𑀺𑀬 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀅𑀚𑁆𑀚 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀉𑀧𑀯𑀢𑁆𑀢𑀦𑁂 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀲𑀸𑀮𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑁂𑀦 𑀬𑀫𑀓𑀲𑀸𑀮𑀸𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀓𑀓𑀼𑀥𑀸 𑀦𑀤𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇

438\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀓𑀓𑀼𑀥𑀸 𑀦𑀤𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀓𑀼𑀥𑀁 𑀦𑀤𑀺𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑁆𑀳𑀢𑁆𑀯𑀸 𑀘 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘 𑀧𑀘𑁆𑀘𑀼𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀘𑀼𑀦𑁆𑀤𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀘𑀼𑀦𑁆𑀤𑀓, 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀳𑀺, 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑀲𑁆𑀫𑀺, 𑀘𑀼𑀦𑁆𑀤𑀓, 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

439\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀼𑀦𑁆𑀤𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀻𑀳𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺 𑀧𑀸𑀤𑁂 𑀧𑀸𑀤𑀁 𑀅𑀘𑁆𑀘𑀸𑀥𑀸𑀬 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀦 𑀘𑀼𑀦𑁆𑀤𑀓𑁄 𑀢𑀢𑁆𑀣𑁂𑀯 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

440\. _𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑀤𑀺𑀓𑀁 𑀓𑀓𑀼𑀥𑀁,_  
_𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑀁 𑀲𑀸𑀢𑀼𑀤𑀓𑀁 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀁;_  
_𑀑𑀕𑀸𑀳𑀺 𑀲𑀢𑁆𑀣𑀸 𑀅𑀓𑀺𑀮𑀦𑁆𑀢𑀭𑀽𑀧𑁄,_  
_𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀫𑁄 𑀘 𑀮𑁄𑀓𑁂𑁇_  


441\. _𑀦𑁆𑀳𑀢𑁆𑀯𑀸 𑀘 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘𑀼𑀤𑀢𑀸𑀭𑀺 𑀲𑀢𑁆𑀣𑀸,_  
_𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀕𑀡𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂;_  
_𑀯𑀢𑁆𑀢𑀸 𑀧𑀯𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀸 𑀇𑀥 𑀥𑀫𑁆𑀫𑁂,_  
_𑀉𑀧𑀸𑀕𑀫𑀺 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀫𑀳𑁂𑀲𑀺𑁇_  


442\. _𑀆𑀫𑀦𑁆𑀢𑀬𑀺 𑀘𑀼𑀦𑁆𑀤𑀓𑀁 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀦𑁆𑀣𑀭 𑀫𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀁;_  
_𑀲𑁄 𑀘𑁄𑀤𑀺𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑁂𑀦 𑀘𑀼𑀦𑁆𑀤𑁄,_  
_𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀦𑁆𑀣𑀭𑀺 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯;_  
_𑀦𑀺𑀧𑀚𑁆𑀚𑀺 𑀲𑀢𑁆𑀣𑀸 𑀅𑀓𑀺𑀮𑀦𑁆𑀢𑀭𑀽𑀧𑁄,_  
_𑀘𑀼𑀦𑁆𑀤𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀧𑀫𑀼𑀔𑁂 𑀦𑀺𑀲𑀻𑀤𑀻𑀢𑀺𑁇_  


(197.)

443\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑁄𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬—  ‘𑀢𑀲𑁆𑀲 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀮𑀸𑀪𑀸 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁, 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄’𑀢𑀺𑁇 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀢𑀲𑁆𑀲 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀘𑀼𑀦𑁆𑀤, 𑀮𑀸𑀪𑀸 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀲𑀼𑀮𑀤𑁆𑀥𑀁, 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀫𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀘𑀼𑀦𑁆𑀤, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  𑀤𑁆𑀯𑁂𑀫𑁂 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀸 𑀲𑀫𑀲𑀫𑀨𑀮𑀸 𑀲𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀅𑀢𑀺𑀯𑀺𑀬 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀳𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀢𑀭𑀸 𑀘 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀢𑀭𑀸 𑀘𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀬𑀜𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀸 𑀲𑀫𑀲𑀫𑀨𑀮𑀸 𑀲𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀅𑀢𑀺𑀯𑀺𑀬 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀳𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀢𑀭𑀸 𑀘 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀢𑀭𑀸 𑀘𑁇 𑀆𑀬𑀼𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀯𑀡𑁆𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀲𑀼𑀔𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀬𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀲𑀕𑁆𑀕𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

444\. _“𑀤𑀤𑀢𑁄 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀯𑀟𑁆𑀠𑀢𑀺,_  
_𑀲𑀁𑀬𑀫𑀢𑁄 𑀯𑁂𑀭𑀁 𑀦 𑀘𑀻𑀬𑀢𑀺;_  
_𑀓𑀼𑀲𑀮𑁄 𑀘 𑀚𑀳𑀸𑀢𑀺 𑀧𑀸𑀧𑀓𑀁,_
