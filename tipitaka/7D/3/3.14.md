# 𑁩𑁇𑁧𑁪 𑀫𑀸𑀭𑀬𑀸𑀘𑀦𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀸𑀭𑀬𑀸𑀘𑀦𑀓𑀣𑀸

(168.)

356. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤𑁂 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀲𑀸𑀯𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇

359. 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀫𑁂 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀦 𑀲𑀸𑀯𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀺𑀬𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀺𑀓𑀸 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀺𑀬𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇

360. 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀯 𑀫𑁂 𑀇𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀦 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁, 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀁𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄”𑀢𑀺𑁇

361. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑁆𑀯𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀳𑁄𑀳𑀺, 𑀦 𑀘𑀺𑀭𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇
