# 𑁩𑁇𑁧𑁦 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬

(158.)

332\. 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀣, 𑀯𑀺𑀳𑁂𑀲𑀸 𑀳𑁂𑀲𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑀁 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑀢𑀺𑁇

(159.)

333\. 𑀓𑀢𑀫𑁄 𑀘 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑀢𑀺?

334\. 𑀇𑀥𑀸𑀦𑀦𑁆𑀤, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

335\. 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇

336\. 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀉𑀚𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀜𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸, 𑀏𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

337\. 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺 𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑁂𑀳𑀺 𑀅𑀲𑀩𑀮𑁂𑀳𑀺 𑀅𑀓𑀫𑁆𑀫𑀸𑀲𑁂𑀳𑀺 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀽𑀧𑀲𑀢𑁆𑀣𑁂𑀳𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇

338\. 𑀅𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑀸𑀤𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄, 𑀬𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’”𑀢𑀺𑁇

339\. 𑀢𑀢𑁆𑀭𑀧𑀺 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑁂 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀮𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀇𑀢𑀺 𑀲𑀻𑀮𑀁 𑀇𑀢𑀺 𑀲𑀫𑀸𑀥𑀺 𑀇𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑁇 𑀲𑀻𑀮𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑁄𑁇 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸𑁇 𑀧𑀜𑁆𑀜𑀸𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀆𑀲𑀯𑁂𑀳𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀸𑀲𑀯𑀸, 𑀪𑀯𑀸𑀲𑀯𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸”𑀢𑀺𑁇

341\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀦𑀸𑀢𑀺𑀓𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀯𑀦𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

342\. “𑀲𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀅𑀬𑀁 𑀯𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑁇 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

343\. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀆𑀮𑁄𑀓𑀺𑀢𑁂 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂 𑀧𑀲𑀸𑀭𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀺𑀢𑁂 𑀧𑀻𑀢𑁂 𑀔𑀸𑀬𑀺𑀢𑁂 𑀲𑀸𑀬𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑀓𑀫𑁆𑀫𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀢𑁂 𑀞𑀺𑀢𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀚𑀸𑀕𑀭𑀺𑀢𑁂 𑀪𑀸𑀲𑀺𑀢𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀅𑀬𑀁 𑀯𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻”𑀢𑀺𑁇
