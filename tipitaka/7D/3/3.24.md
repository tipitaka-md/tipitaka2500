# 𑁩𑁇𑁨𑁪 𑀧𑀸𑀦𑀻𑀬𑀸𑀳𑀭𑀡

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀸𑀦𑀻𑀬𑀸𑀳𑀭𑀡

(191.)

426\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀳𑀺, 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀆𑀳𑀭, 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺, 𑀢𑀁 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀮𑀼𑀍𑀇𑀢𑀁 𑀆𑀯𑀺𑀮𑀁 𑀲𑀦𑁆𑀤𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀓𑀼𑀥𑀸 𑀦𑀤𑀻 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑀸 𑀲𑀸𑀢𑁄𑀤𑀓𑀸 𑀲𑀻𑀢𑁄𑀤𑀓𑀸 𑀲𑁂𑀢𑁄𑀤𑀓𑀸 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀢𑁆𑀣𑀸 𑀭𑀫𑀡𑀻𑀬𑀸𑁇 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀜𑁆𑀘 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀲𑀻𑀢𑀻 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

427\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀆𑀳𑀭, 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺, 𑀢𑀁 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀮𑀼𑀍𑀇𑀢𑀁 𑀆𑀯𑀺𑀮𑀁 𑀲𑀦𑁆𑀤𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀓𑀼𑀥𑀸 𑀦𑀤𑀻 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑀸 𑀲𑀸𑀢𑁄𑀤𑀓𑀸 𑀲𑀻𑀢𑁄𑀤𑀓𑀸 𑀲𑁂𑀢𑁄𑀤𑀓𑀸 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀢𑁆𑀣𑀸 𑀭𑀫𑀡𑀻𑀬𑀸𑁇 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀜𑁆𑀘 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀲𑀻𑀢𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

428\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀆𑀳𑀭, 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸 𑀦𑀤𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀦𑀤𑀺𑀓𑀸 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀮𑀼𑀍𑀇𑀢𑀸 𑀆𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀅𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀅𑀦𑀸𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀺𑀢𑁆𑀣𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀸𑁇 𑀅𑀬𑀜𑁆𑀳𑀺 𑀲𑀸 𑀦𑀤𑀺𑀓𑀸 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀮𑀼𑀍𑀇𑀢𑀸 𑀆𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸 𑀫𑀬𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀅𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀅𑀦𑀸𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀢𑀻”𑀢𑀺𑁇 𑀧𑀢𑁆𑀢𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀸𑁇 𑀇𑀤𑀸𑀦𑀺 𑀲𑀸 𑀪𑀦𑁆𑀢𑁂 𑀦𑀤𑀺𑀓𑀸 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀮𑀼𑀍𑀇𑀢𑀸 𑀆𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸 𑀫𑀬𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀅𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀅𑀦𑀸𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀺𑀢𑁆𑀣𑁇 𑀧𑀺𑀯𑀢𑀼 𑀪𑀕𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀺𑀯𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀧𑀸𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀁 𑀅𑀧𑀸𑀬𑀺𑁇
