# 𑁩𑁇𑁨𑁦 𑀆𑀦𑀦𑁆𑀤𑀬𑀸𑀘𑀦𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀆𑀦𑀦𑁆𑀤𑀬𑀸𑀘𑀦𑀓𑀣𑀸

(178.)

393\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

394\. “𑀅𑀮𑀁 𑀤𑀸𑀦𑀺, 𑀆𑀦𑀦𑁆𑀤𑁇 𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀬𑀸𑀘𑀺, 𑀅𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀬𑀸𑀘𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

395\. “𑀲𑀤𑁆𑀤𑀳𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀅𑀣 𑀓𑀺𑀜𑁆𑀘𑀭𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀧𑀻𑀍𑀏𑀲𑀻”𑀢𑀺? “𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  ‘𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑁇 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸’”𑀢𑀺𑁇 “𑀲𑀤𑁆𑀤𑀳𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀢𑀼𑀬𑁆𑀳𑁂𑀯𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀢𑀼𑀬𑁆𑀳𑁂𑀯𑁂𑀢𑀁 𑀅𑀧𑀭𑀤𑁆𑀥𑀁, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀏𑀯𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁, 𑀦 𑀢𑀣𑀸𑀕𑀢𑀁 𑀬𑀸𑀘𑀺—  ‘𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀤𑁆𑀯𑁂𑀯 𑀢𑁂 𑀯𑀸𑀘𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀢𑀢𑀺𑀬𑀓𑀁 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀢𑀼𑀬𑁆𑀳𑁂𑀯𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀢𑀼𑀬𑁆𑀳𑁂𑀯𑁂𑀢𑀁 𑀅𑀧𑀭𑀤𑁆𑀥𑀁𑁇

(179.)

396\. 𑀏𑀓𑀫𑀺𑀤𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑀁 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀔𑁄 𑀢𑀸𑀳𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺𑀁—  ‘𑀭𑀫𑀡𑀻𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀕𑀳𑀁, 𑀭𑀫𑀡𑀻𑀬𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁄 𑀧𑀩𑁆𑀩𑀢𑁄𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸’𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁, 𑀦 𑀢𑀣𑀸𑀕𑀢𑀁 𑀬𑀸𑀘𑀺—  ‘𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀤𑁆𑀯𑁂𑀯 𑀢𑁂 𑀯𑀸𑀘𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀢𑀢𑀺𑀬𑀓𑀁 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳𑀸𑀦𑀦𑁆𑀤, 𑀢𑀼𑀬𑁆𑀳𑁂𑀯𑁂𑀢𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀢𑀼𑀬𑁆𑀳𑁂𑀯𑁂𑀢𑀁 𑀅𑀧𑀭𑀤𑁆𑀥𑀁𑁇

(180.)

403\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑁂 𑀫𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀤𑁂𑀲𑀺𑀢𑀸, 𑀢𑁂 𑀯𑁄 𑀲𑀸𑀥𑀼𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀬𑀣𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀥𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁, 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂 𑀘 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀫𑀬𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀤𑁂𑀲𑀺𑀢𑀸, 𑀬𑁂 𑀯𑁄 𑀲𑀸𑀥𑀼𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀬𑀣𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀥𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁, 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀫𑀬𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀤𑁂𑀲𑀺𑀢𑀸, 𑀬𑁂 𑀯𑁄 𑀲𑀸𑀥𑀼𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀬𑀣𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀥𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁, 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

(185.)

404\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀳𑀦𑁆𑀤 𑀤𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀫𑀦𑁆𑀢𑀬𑀸𑀫𑀺 𑀯𑁄, 𑀯𑀬𑀥𑀫𑁆𑀫𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀣𑁇 𑀦𑀘𑀺𑀭𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

405\. _“𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁄 𑀯𑀬𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀫𑀫 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀧𑀳𑀸𑀬 𑀯𑁄 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀓𑀢𑀁 𑀫𑁂 𑀲𑀭𑀡𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


406\. _𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀲𑀢𑀻𑀫𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀲𑀻𑀮𑀸 𑀳𑁄𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄;_  
_𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸,_  
_𑀲𑀘𑀺𑀢𑁆𑀢𑀫𑀦𑀼𑀭𑀓𑁆𑀔𑀣𑁇_  


407\. _𑀬𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀳𑀸𑀬 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


---
