# 𑁩𑁇𑁩𑁯 𑀲𑀭𑀻𑀭𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀚𑀦

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀭𑀻𑀭𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀚𑀦

(236.)

515\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀅𑀚𑀸𑀢𑀲𑀢𑁆𑀢𑀼 𑀯𑁂𑀤𑁂𑀳𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀪𑀕𑀯𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀳𑀫𑁆𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

516\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀻—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀪𑀕𑀯𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀬𑀫𑁆𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

517\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀯𑀸𑀲𑀻 𑀲𑀓𑁆𑀬𑀸—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀯𑀸𑀲𑀻 𑀲𑀓𑁆𑀬𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀜𑀸𑀢𑀺𑀲𑁂𑀝𑁆𑀞𑁄, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

518\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀅𑀮𑁆𑀮𑀓𑀧𑁆𑀧𑀓𑀸 𑀩𑀼𑀮𑀬𑁄—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀮𑁆𑀮𑀓𑀧𑁆𑀧𑀓𑀸 𑀩𑀼𑀮𑀬𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀪𑀕𑀯𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀬𑀫𑁆𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

519\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀭𑀸𑀫𑀕𑀸𑀫𑀓𑀸 𑀓𑁄𑀍𑀇𑀬𑀸—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀫𑀕𑀸𑀫𑀓𑀸 𑀓𑁄𑀍𑀇𑀬𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀪𑀕𑀯𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀬𑀫𑁆𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

520\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀯𑁂𑀝𑁆𑀞𑀤𑀻𑀧𑀓𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀝𑁆𑀞𑀤𑀻𑀧𑀓𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀪𑀕𑀯𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀳𑀫𑁆𑀧𑀺𑀲𑁆𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

521\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀧𑀸𑀯𑁂𑀬𑁆𑀬𑀓𑀸 𑀫𑀮𑁆𑀮𑀸—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀸𑀯𑁂𑀬𑁆𑀬𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀪𑀕𑀯𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀬𑀫𑁆𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

522\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀢𑁂 𑀲𑀁𑀖𑁂 𑀕𑀡𑁂 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀕𑀸𑀫𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀦 𑀫𑀬𑀁 𑀤𑀲𑁆𑀲𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀦𑁆”𑀢𑀺𑁇

(237.)

523\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀤𑁄𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑁂 𑀲𑀁𑀖𑁂 𑀕𑀡𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—

524\. _“𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀫𑀫 𑀏𑀓𑀯𑀸𑀘𑀁,_  
_𑀅𑀫𑁆𑀳𑀸𑀓 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀳𑀼 𑀔𑀦𑁆𑀢𑀺𑀯𑀸𑀤𑁄;_  
_𑀦 𑀳𑀺 𑀲𑀸𑀥𑀼 𑀬𑀁 𑀉𑀢𑁆𑀢𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲,_  
_𑀲𑀭𑀻𑀭𑀪𑀸𑀕𑁂 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀧𑀳𑀸𑀭𑁄𑁇_  


525\. _𑀲𑀩𑁆𑀩𑁂𑀯 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀳𑀺𑀢𑀸 𑀲𑀫𑀕𑁆𑀕𑀸,_  
_𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀓𑀭𑁄𑀫𑀝𑁆𑀞𑀪𑀸𑀕𑁂;_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀼 𑀤𑀺𑀲𑀸𑀲𑀼 𑀣𑀽𑀧𑀸,_  
_𑀩𑀳𑀽 𑀚𑀦𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑁄 𑀧𑀲𑀦𑁆𑀦𑀸”𑀢𑀺𑁇_  


(238.)

526\. “𑀢𑁂𑀦 𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑁆𑀯𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀥𑀸 𑀲𑀫𑀁 𑀲𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀯𑀺𑀪𑀚𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀤𑁄𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑁂𑀲𑀁 𑀲𑀁𑀖𑀸𑀦𑀁 𑀕𑀡𑀸𑀦𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀥𑀸 𑀲𑀫𑀁 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀲𑀁𑀖𑁂 𑀕𑀡𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑀁 𑀫𑁂 𑀪𑁄𑀦𑁆𑀢𑁄 𑀢𑀼𑀫𑁆𑀩𑀁 𑀤𑀤𑀦𑁆𑀢𑀼 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀼𑀫𑁆𑀩𑀲𑁆𑀲 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀤𑀁𑀲𑀼 𑀔𑁄 𑀢𑁂 𑀤𑁄𑀡𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀩𑀁𑁇

527\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀧𑀺𑀧𑁆𑀧𑀮𑀺𑀯𑀦𑀺𑀬𑀸 𑀫𑁄𑀭𑀺𑀬𑀸—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀺𑀧𑁆𑀧𑀮𑀺𑀯𑀦𑀺𑀬𑀸 𑀫𑁄𑀭𑀺𑀬𑀸 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀪𑀕𑀯𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀬𑀫𑁆𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀅𑀭𑀳𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀣𑀽𑀧𑀜𑁆𑀘 𑀫𑀳𑀜𑁆𑀘 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀦𑀢𑁆𑀣𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀁 𑀪𑀸𑀕𑁄, 𑀯𑀺𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀭𑀻𑀭𑀸𑀦𑀺𑁇 𑀇𑀢𑁄 𑀅𑀗𑁆𑀕𑀸𑀭𑀁 𑀳𑀭𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀗𑁆𑀕𑀸𑀭𑀁 𑀳𑀭𑀺𑀁𑀲𑀼𑁇
