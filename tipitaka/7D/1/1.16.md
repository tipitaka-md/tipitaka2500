# 𑁧𑁇𑁧𑁬 𑀘𑀸𑀭𑀺𑀓𑀸𑀅𑀦𑀼𑀚𑀸𑀦𑀦

* 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀸𑀭𑀺𑀓𑀸𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(86.)

171\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀫𑀳𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀫𑀳𑀸 𑀔𑁄 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁, 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  “𑀘𑀭𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀭𑀺𑀓𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁; 𑀫𑀸 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀅𑀕𑀫𑀺𑀢𑁆𑀣; 𑀤𑁂𑀲𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸, 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄𑁇 𑀅𑀧𑀺 𑀘 𑀙𑀦𑁆𑀦𑀁 𑀙𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀻 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸”’𑀢𑀺𑁇

(87.)

172\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀕𑀯𑀸, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀲𑀼𑀕𑀢𑁇 𑀫𑀳𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁, 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀘𑀭𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀭𑀺𑀓𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁; 𑀫𑀸 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀅𑀕𑀫𑀺𑀢𑁆𑀣; 𑀤𑁂𑀲𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸, 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄”𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀬𑀁 𑀢𑀣𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫 𑀬𑀣𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀦𑁆𑀦𑀁 𑀙𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸’𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇

173\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀫𑀳𑀸 𑀔𑁄 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  𑀘𑀭𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀭𑀺𑀓𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁; 𑀫𑀸 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀅𑀕𑀫𑀺𑀢𑁆𑀣; 𑀤𑁂𑀲𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸, 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄𑁇 𑀅𑀧𑀺 𑀘 𑀙𑀦𑁆𑀦𑀁 𑀙𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀻 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸”’𑀢𑀺𑁇

174\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀫𑀫 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀫 𑀧𑀼𑀭𑀢𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀕𑀯𑀸, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀲𑀼𑀕𑀢𑁇 𑀫𑀳𑀸 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  𑀘𑀭𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀭𑀺𑀓𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁; 𑀫𑀸 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀅𑀕𑀫𑀺𑀢𑁆𑀣; 𑀤𑁂𑀲𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸, 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀬𑀁 𑀢𑀣𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫, 𑀬𑀣𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀦𑁆𑀦𑀁 𑀙𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸’𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇

175\. ‘𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀭𑀣 𑀘𑀸𑀭𑀺𑀓𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁; 𑀫𑀸 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀅𑀕𑀫𑀺𑀢𑁆𑀣; 𑀤𑁂𑀲𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸, 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄𑁇 𑀅𑀧𑀺 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀦𑀁 𑀙𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀻 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀏𑀓𑀸𑀳𑁂𑀦𑁂𑀯 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇

(89.)

176\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀆𑀯𑀸𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑀫𑁆𑀳𑀺 𑀳𑀺 𑀯𑀲𑁆𑀲𑁂 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑀢𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁—  ‘𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀁 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁; 𑀧𑀜𑁆𑀘 𑀤𑀸𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑁂𑀲𑀸𑀦𑀺; 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀻 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸’𑀢𑀺𑁇 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑀲𑁆𑀲𑁂𑀲𑀼 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂𑀲𑀼…  𑀢𑀻𑀲𑀼 𑀯𑀲𑁆𑀲𑁂𑀲𑀼 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂𑀲𑀼…  𑀘𑀢𑀽𑀲𑀼 𑀯𑀲𑁆𑀲𑁂𑀲𑀼 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂𑀲𑀼…  𑀧𑀜𑁆𑀘𑀲𑀼 𑀯𑀲𑁆𑀲𑁂𑀲𑀼 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂𑀲𑀼 𑀤𑁂𑀯𑀢𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁—  ‘𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑀸𑀦𑀺; 𑀏𑀓𑀁 𑀤𑀸𑀦𑀺 𑀯𑀲𑁆𑀲𑀁 𑀲𑁂𑀲𑀁; 𑀏𑀓𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀻 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸’𑀢𑀺𑁇 𑀙𑀲𑀼 𑀯𑀲𑁆𑀲𑁂𑀲𑀼 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂𑀲𑀼 𑀤𑁂𑀯𑀢𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁—  ‘𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺, 𑀲𑀫𑀬𑁄 𑀤𑀸𑀦𑀺 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀓𑁂𑀦 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁂𑀦 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁂𑀦 𑀏𑀓𑀸𑀳𑁂𑀦𑁂𑀯 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀬𑀸𑀢𑀺𑁇

(90.)

177\. 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀏𑀯𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺—

178\. _‘𑀔𑀦𑁆𑀢𑀻 𑀧𑀭𑀫𑀁 𑀢𑀧𑁄 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀫𑀁 𑀯𑀤𑀦𑁆𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀸;_  
_𑀦 𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀧𑀭𑀽𑀧𑀖𑀸𑀢𑀻,_  
_𑀦 𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀁 𑀯𑀺𑀳𑁂𑀞𑀬𑀦𑁆𑀢𑁄𑁇_  


179\. _𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀲𑁆𑀲 𑀅𑀓𑀭𑀡𑀁,_  
_𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀲𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀁,_  
_𑀏𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


180\. _𑀅𑀦𑀽𑀧𑀯𑀸𑀤𑁄 𑀅𑀦𑀽𑀧𑀖𑀸𑀢𑁄,_  
_𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀘 𑀲𑀁𑀯𑀭𑁄;_  
_𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀘 𑀪𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀦𑁆𑀢𑀜𑁆𑀘 𑀲𑀬𑀦𑀸𑀲𑀦𑀁;_  
_𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑁂 𑀘 𑀆𑀬𑁄𑀕𑁄,_  
_𑀏𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀲𑀸𑀲𑀦𑀦𑁆’𑀢𑀺𑁇_
