# 𑁧𑁇𑁧𑁨 𑀩𑁆𑀭𑀳𑁆𑀫𑀬𑀸𑀘𑀦𑀓𑀣𑀸

* 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁆𑀭𑀳𑁆𑀫𑀬𑀸𑀘𑀦𑀓𑀣𑀸

(64.)

139\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀥𑀺𑀕𑀢𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀤𑀼𑀤𑁆𑀤𑀲𑁄 𑀤𑀼𑀭𑀦𑀼𑀩𑁄𑀥𑁄 𑀲𑀦𑁆𑀢𑁄 𑀧𑀡𑀻𑀢𑁄 𑀅𑀢𑀓𑁆𑀓𑀸𑀯𑀘𑀭𑁄 𑀦𑀺𑀧𑀼𑀡𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑁂𑀤𑀦𑀻𑀬𑁄𑁇 𑀆𑀮𑀬𑀭𑀸𑀫𑀸 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀧𑀚𑀸 𑀆𑀮𑀬𑀭𑀢𑀸 𑀆𑀮𑀬𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑁇 𑀆𑀮𑀬𑀭𑀸𑀫𑀸𑀬 𑀔𑁄 𑀧𑀦 𑀧𑀚𑀸𑀬 𑀆𑀮𑀬𑀭𑀢𑀸𑀬 𑀆𑀮𑀬𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀇𑀤𑀁 𑀞𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄𑁇 𑀇𑀤𑀫𑁆𑀧𑀺 𑀔𑁄 𑀞𑀸𑀦𑀁 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀣𑁄 𑀲𑀩𑁆𑀩𑀽𑀧𑀥𑀺𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁄 𑀯𑀺𑀭𑀸𑀕𑁄 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑁂 𑀘 𑀫𑁂 𑀦 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁; 𑀲𑁄 𑀫𑀫𑀲𑁆𑀲 𑀓𑀺𑀮𑀫𑀣𑁄, 𑀲𑀸 𑀫𑀫𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀸’𑀢𑀺𑁇

(65.)

140\. 𑀅𑀧𑀺𑀲𑁆𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀇𑀫𑀸 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀧𑀝𑀺𑀪𑀁𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀲𑁆𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀸—

141\. _‘𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀫𑁂 𑀅𑀥𑀺𑀕𑀢𑀁,_  
_𑀳𑀮𑀁 𑀤𑀸𑀦𑀺 𑀧𑀓𑀸𑀲𑀺𑀢𑀼𑀁;_  
_𑀭𑀸𑀕𑀤𑁄𑀲𑀧𑀭𑁂𑀢𑁂𑀳𑀺,_  
_𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀲𑀫𑁆𑀩𑀼𑀥𑁄𑁇_  


142\. _𑀧𑀝𑀺𑀲𑁄𑀢𑀕𑀸𑀫𑀺𑀁 𑀦𑀺𑀧𑀼𑀡𑀁,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀅𑀡𑀼𑀁;_  
_𑀭𑀸𑀕𑀭𑀢𑁆𑀢𑀸 𑀦 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀺,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁂𑀦 𑀆𑀯𑀼𑀝𑀸’𑀢𑀺𑁇_  


143\. 𑀇𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀢𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑁇

(66.)

144\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀦𑀲𑁆𑀲𑀢𑀺 𑀯𑀢 𑀪𑁄 𑀮𑁄𑀓𑁄, 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺 𑀯𑀢 𑀪𑁄 𑀮𑁄𑀓𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀢𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀢𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀧𑀣𑀯𑀺𑀬𑀁 𑀦𑀺𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀤𑁂𑀲𑁂𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁, 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁, 𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸; 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄’𑀢𑀺𑁇

(67.)

145\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑀁 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀩𑁆𑀭𑀳𑁆𑀫𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀥𑀺𑀕𑀢𑁄 𑀔𑁄 𑀫𑁆𑀬𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄 𑀤𑀼𑀤𑁆𑀤𑀲𑁄 𑀤𑀼𑀭𑀦𑀼𑀩𑁄𑀥𑁄 𑀲𑀦𑁆𑀢𑁄 𑀧𑀡𑀻𑀢𑁄 𑀅𑀢𑀓𑁆𑀓𑀸𑀯𑀘𑀭𑁄 𑀦𑀺𑀧𑀼𑀡𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑁂𑀤𑀦𑀻𑀬𑁄𑁇 𑀆𑀮𑀬𑀭𑀸𑀫𑀸 𑀔𑁄 𑀧𑀦𑀸𑀬𑀁 𑀧𑀚𑀸 𑀆𑀮𑀬𑀭𑀢𑀸 𑀆𑀮𑀬𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑁇 𑀆𑀮𑀬𑀭𑀸𑀫𑀸𑀬 𑀔𑁄 𑀧𑀦 𑀧𑀚𑀸𑀬 𑀆𑀮𑀬𑀭𑀢𑀸𑀬 𑀆𑀮𑀬𑀲𑀫𑁆𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀇𑀤𑀁 𑀞𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄𑁇 𑀇𑀤𑀫𑁆𑀧𑀺 𑀔𑁄 𑀞𑀸𑀦𑀁 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀣𑁄 𑀲𑀩𑁆𑀩𑀽𑀧𑀥𑀺𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁄 𑀯𑀺𑀭𑀸𑀕𑁄 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁, 𑀧𑀭𑁂 𑀘 𑀫𑁂 𑀦 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁; 𑀲𑁄 𑀫𑀫𑀲𑁆𑀲 𑀓𑀺𑀮𑀫𑀣𑁄, 𑀲𑀸 𑀫𑀫𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀸”𑀢𑀺𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼 𑀫𑀁, 𑀩𑁆𑀭𑀳𑁆𑀫𑁂, 𑀇𑀫𑀸 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀧𑀝𑀺𑀪𑀁𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀲𑁆𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀸—

146\. _“𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀫𑁂 𑀅𑀥𑀺𑀕𑀢𑀁,_  
_𑀳𑀮𑀁 𑀤𑀸𑀦𑀺 𑀧𑀓𑀸𑀲𑀺𑀢𑀼𑀁;_  
_𑀭𑀸𑀕𑀤𑁄𑀲𑀧𑀭𑁂𑀢𑁂𑀳𑀺,_  
_𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀲𑀫𑁆𑀩𑀼𑀥𑁄𑁇_  


147\. _𑀧𑀝𑀺𑀲𑁄𑀢𑀕𑀸𑀫𑀺𑀁 𑀦𑀺𑀧𑀼𑀡𑀁,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀅𑀡𑀼𑀁;_  
_𑀭𑀸𑀕𑀭𑀢𑁆𑀢𑀸 𑀦 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀺,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁂𑀦 𑀆𑀯𑀼𑀝𑀸”𑀢𑀺𑁇_  


148\. 𑀇𑀢𑀺𑀳 𑀫𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑁂, 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑁄 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀢𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇

(68.)

149\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀤𑁂𑀲𑁂𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁, 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁, 𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸, 𑀅𑀲𑁆𑀲𑀯𑀦𑀢𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄’𑀢𑀺𑁇

(69.)

150\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑁄 𑀘 𑀅𑀚𑁆𑀛𑁂𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀘 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀮𑁄𑀓𑀁 𑀯𑁄𑀮𑁄𑀓𑁂𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑀼𑀤𑁆𑀥𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀮𑁄𑀓𑀁 𑀯𑁄𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑁂 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑁂 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑁂 𑀢𑀺𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀫𑀼𑀤𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀲𑁆𑀯𑀸𑀓𑀸𑀭𑁂 𑀤𑁆𑀯𑀸𑀓𑀸𑀭𑁂 𑀲𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂 𑀤𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀉𑀧𑁆𑀧𑀮𑀺𑀦𑀺𑀬𑀁 𑀯𑀸 𑀧𑀤𑀼𑀫𑀺𑀦𑀺𑀬𑀁 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀺𑀦𑀺𑀬𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀮𑀸𑀦𑀺 𑀯𑀸 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂 𑀲𑀁𑀯𑀟𑁆𑀠𑀸𑀦𑀺 𑀉𑀤𑀓𑀸𑀦𑀼𑀕𑁆𑀕𑀢𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑁄 𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑀧𑁄𑀲𑀻𑀦𑀺𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀮𑀸𑀦𑀺 𑀯𑀸 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂 𑀲𑀁𑀯𑀟𑁆𑀠𑀸𑀦𑀺 𑀲𑀫𑁄𑀤𑀓𑀁 𑀞𑀺𑀢𑀸𑀦𑀺𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀮𑀸𑀦𑀺 𑀯𑀸 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂 𑀲𑀁𑀯𑀟𑁆𑀠𑀸𑀦𑀺 𑀉𑀤𑀓𑀸 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑁆𑀫 𑀞𑀺𑀢𑀸𑀦𑀺 𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀉𑀤𑀓𑁂𑀦𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑀼𑀤𑁆𑀥𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀮𑁄𑀓𑀁 𑀯𑁄𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀲𑀢𑁆𑀢𑁂 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑁂 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑁂 𑀢𑀺𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀫𑀼𑀤𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀲𑁆𑀯𑀸𑀓𑀸𑀭𑁂 𑀤𑁆𑀯𑀸𑀓𑀸𑀭𑁂 𑀲𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂 𑀤𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀬𑁂 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦 𑀧𑀭𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑁇

(70.)

151\. 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

152\. _‘𑀲𑁂𑀮𑁂 𑀬𑀣𑀸 𑀧𑀩𑁆𑀩𑀢𑀫𑀼𑀤𑁆𑀥𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄,_  
_𑀬𑀣𑀸𑀧𑀺 𑀧𑀲𑁆𑀲𑁂 𑀚𑀦𑀢𑀁 𑀲𑀫𑀦𑁆𑀢𑀢𑁄;_  
_𑀢𑀣𑀽𑀧𑀫𑀁 𑀥𑀫𑁆𑀫𑀫𑀬𑀁 𑀲𑀼𑀫𑁂𑀥,_  
_𑀧𑀸𑀲𑀸𑀤𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀲𑀫𑀦𑁆𑀢𑀘𑀓𑁆𑀔𑀼;_  
_𑀲𑁄𑀓𑀸𑀯𑀢𑀺𑀡𑁆𑀡𑀁 𑀚𑀦𑀢𑀫𑀧𑁂𑀢𑀲𑁄𑀓𑁄,_  
_𑀅𑀯𑁂𑀓𑁆𑀔𑀲𑁆𑀲𑀼 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀁𑁇_  


153\. _𑀉𑀝𑁆𑀞𑁂𑀳𑀺 𑀯𑀻𑀭 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫,_  
_𑀲𑀢𑁆𑀣𑀯𑀸𑀳 𑀅𑀡𑀡 𑀯𑀺𑀘𑀭 𑀮𑁄𑀓𑁂;_  
_𑀤𑁂𑀲𑀲𑁆𑀲𑀼 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁,_  
_𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇_  


(71.)

154\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑀁 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

155\. _‘𑀅𑀧𑀸𑀭𑀼𑀢𑀸 𑀢𑁂𑀲𑀁 𑀅𑀫𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀭𑀸,_  
_𑀬𑁂 𑀲𑁄𑀢𑀯𑀦𑁆𑀢𑁄 𑀧𑀫𑀼𑀜𑁆𑀘𑀦𑁆𑀢𑀼 𑀲𑀤𑁆𑀥𑀁;_  
_𑀯𑀺𑀳𑀺𑀁𑀲𑀲𑀜𑁆𑀜𑀻 𑀧𑀕𑀼𑀡𑀁 𑀦 𑀪𑀸𑀲𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀧𑀡𑀻𑀢𑀁 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑁂’𑀢𑀺𑁇_  


156\. 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸—  ‘𑀓𑀢𑀸𑀯𑀓𑀸𑀲𑁄 𑀔𑁄𑀫𑁆𑀳𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑀸 𑀪𑀕𑀯𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀸’𑀢𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇
