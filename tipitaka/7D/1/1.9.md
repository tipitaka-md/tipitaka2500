# 𑁧𑁇𑁯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸

* 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸

(54.)

108\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀭𑀣𑀁 𑀆𑀤𑀸𑀬 𑀇𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀳𑀺𑁇 𑀅𑀳𑀁 𑀧𑀦 𑀇𑀥𑁂𑀯 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀣𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀭𑀣𑀁 𑀆𑀤𑀸𑀬 𑀢𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀧𑀦 𑀓𑀼𑀫𑀸𑀭𑁄 𑀢𑀢𑁆𑀣𑁂𑀯 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑁇
