# 𑁧𑁇𑁬 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑀧𑀼𑀭𑀺𑀲

* 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑀧𑀼𑀭𑀺𑀲

(46.)

96\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—

97\. ‘𑀫𑀸 𑀳𑁂𑀯 𑀔𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀦 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀲𑀺, 𑀫𑀸 𑀳𑁂𑀯 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺, 𑀫𑀸 𑀳𑁂𑀯 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀲𑀘𑁆𑀘𑀁 𑀅𑀲𑁆𑀲 𑀯𑀘𑀦𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑀦𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺—  ‘𑀬𑀣𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀦 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸 𑀅𑀲𑁆𑀲 𑀯𑀘𑀦𑀦𑁆’𑀢𑀺𑁇

98\. 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀩𑀳𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁…𑀧𑁂…

(47.)

99\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀁 𑀆𑀩𑀸𑀥𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑀁 𑀲𑀓𑁂 𑀫𑀼𑀢𑁆𑀢𑀓𑀭𑀻𑀲𑁂 𑀧𑀮𑀺𑀧𑀦𑁆𑀦𑀁 𑀲𑁂𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀬𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀲𑀁𑀯𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀅𑀬𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁𑀓𑀢𑁄? 𑀅𑀓𑁆𑀔𑀻𑀦𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀁, 𑀲𑀭𑁄𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆’𑀢𑀺? ‘𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦𑁂𑀲𑁄, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺? ‘𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀦𑀸𑀫 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀢𑀫𑁆𑀳𑀸 𑀆𑀩𑀸𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑁄, 𑀩𑁆𑀬𑀸𑀥𑀺𑀁 𑀅𑀦𑀢𑀻𑀢𑁄’𑀢𑀺? ‘𑀢𑁆𑀯𑀜𑁆𑀘, 𑀤𑁂𑀯, 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳 𑀲𑀩𑁆𑀩𑁂 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀸, 𑀩𑁆𑀬𑀸𑀥𑀺𑀁 𑀅𑀦𑀢𑀻𑀢𑀸’𑀢𑀺𑁇 ‘𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀮𑀁 𑀤𑀸𑀦𑀚𑁆𑀚 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸, 𑀇𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀳𑀻’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀣𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀕𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺—  ‘𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀓𑀺𑀭, 𑀪𑁄, 𑀚𑀸𑀢𑀺 𑀦𑀸𑀫, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀚𑀸𑀢𑀲𑁆𑀲 𑀚𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀩𑁆𑀬𑀸𑀥𑀺 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

(48.)

100\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀓𑀘𑁆𑀘𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀪𑀺𑀭𑀫𑀺𑀢𑁆𑀣, 𑀓𑀘𑁆𑀘𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀪𑀺𑀭𑀫𑀺𑀢𑁆𑀣, 𑀦 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀤𑁆𑀤𑀲 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄’𑀢𑀺? ‘𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀁 𑀆𑀩𑀸𑀥𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑀁 𑀲𑀓𑁂 𑀫𑀼𑀢𑁆𑀢𑀓𑀭𑀻𑀲𑁂 𑀧𑀮𑀺𑀧𑀦𑁆𑀦𑀁 𑀲𑁂𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀬𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀲𑀁𑀯𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁𑀓𑀢𑁄, 𑀅𑀓𑁆𑀔𑀻𑀦𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀁, 𑀲𑀭𑁄𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆”𑀢𑀺? “𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀦𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑁂𑀲𑁄, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀦𑀸𑀫𑀸”𑀢𑀺? “𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀦𑀸𑀫 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀢𑀫𑁆𑀳𑀸 𑀆𑀩𑀸𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑁄, 𑀩𑁆𑀬𑀸𑀥𑀺𑀁 𑀅𑀦𑀢𑀻𑀢𑁄”𑀢𑀺? “𑀢𑁆𑀯𑀜𑁆𑀘, 𑀤𑁂𑀯, 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳 𑀲𑀩𑁆𑀩𑁂 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀸, 𑀩𑁆𑀬𑀸𑀥𑀺𑀁 𑀅𑀦𑀢𑀻𑀢𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀮𑀁 𑀤𑀸𑀦𑀚𑁆𑀚 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸, 𑀇𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀅𑀳𑀁, 𑀤𑁂𑀯, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺𑀁𑁇 𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀕𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺—  “𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀓𑀺𑀭, 𑀪𑁄, 𑀚𑀸𑀢𑀺 𑀦𑀸𑀫, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀚𑀸𑀢𑀲𑁆𑀲 𑀚𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀩𑁆𑀬𑀸𑀥𑀺 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”’𑀢𑀺𑁇
