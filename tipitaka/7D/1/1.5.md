# 𑁧𑁇𑁫 𑀚𑀺𑀡𑁆𑀡𑀧𑀼𑀭𑀺𑀲

* 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀚𑀺𑀡𑁆𑀡𑀧𑀼𑀭𑀺𑀲

(43.)

92\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀩𑀳𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀬𑁄𑀚𑁂𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀲𑀼𑀪𑀽𑀫𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀣𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀲𑀺—  ‘𑀬𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀤𑁂𑀯, 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀤𑁆𑀤𑀁 𑀪𑀤𑁆𑀤𑀁 𑀬𑀸𑀦𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀤𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑁂𑀳𑀺 𑀬𑀸𑀦𑁂𑀳𑀺 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺𑁇

(44.)

93\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀁 𑀚𑀺𑀡𑁆𑀡𑀁 𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀯𑀗𑁆𑀓𑀁 𑀪𑁄𑀕𑁆𑀕𑀁 𑀤𑀡𑁆𑀟𑀧𑀭𑀸𑀬𑀦𑀁 𑀧𑀯𑁂𑀥𑀫𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀆𑀢𑀼𑀭𑀁 𑀕𑀢𑀬𑁄𑀩𑁆𑀩𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀅𑀬𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁𑀓𑀢𑁄? 𑀓𑁂𑀲𑀸𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀁, 𑀓𑀸𑀬𑁄𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆’𑀢𑀺𑁇 ‘𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦𑁂𑀲𑁄, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺? ‘𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀸𑀫𑁇 𑀦 𑀤𑀸𑀦𑀺 𑀢𑁂𑀦 𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑁄, 𑀚𑀭𑀁 𑀅𑀦𑀢𑀻𑀢𑁄’𑀢𑀺? ‘𑀢𑁆𑀯𑀜𑁆𑀘, 𑀤𑁂𑀯, 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳 𑀲𑀩𑁆𑀩𑁂 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀸, 𑀚𑀭𑀁 𑀅𑀦𑀢𑀻𑀢𑀸’𑀢𑀺𑁇 ‘𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀮𑀁 𑀤𑀸𑀦𑀚𑁆𑀚 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸𑁇 𑀇𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀳𑀻’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀣𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀕𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺—  ‘𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀓𑀺𑀭, 𑀪𑁄, 𑀚𑀸𑀢𑀺 𑀦𑀸𑀫, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀚𑀸𑀢𑀲𑁆𑀲 𑀚𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

(45.)

94\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀓𑀘𑁆𑀘𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀪𑀺𑀭𑀫𑀺𑀢𑁆𑀣? 𑀓𑀘𑁆𑀘𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺? ‘𑀦 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀪𑀺𑀭𑀫𑀺𑀢𑁆𑀣, 𑀦 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀤𑁆𑀤𑀲 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄’𑀢𑀺? ‘𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀁 𑀚𑀺𑀡𑁆𑀡𑀁 𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀯𑀗𑁆𑀓𑀁 𑀪𑁄𑀕𑁆𑀕𑀁 𑀤𑀡𑁆𑀟𑀧𑀭𑀸𑀬𑀦𑀁 𑀧𑀯𑁂𑀥𑀫𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀆𑀢𑀼𑀭𑀁 𑀕𑀢𑀬𑁄𑀩𑁆𑀩𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁𑀓𑀢𑁄, 𑀓𑁂𑀲𑀸𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀁, 𑀓𑀸𑀬𑁄𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆”𑀢𑀺? “𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑁂𑀲𑁄, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀸𑀫𑀸”𑀢𑀺? “𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀸𑀫 𑀦 𑀤𑀸𑀦𑀺 𑀢𑁂𑀦 𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑁄, 𑀚𑀭𑀁 𑀅𑀦𑀢𑀻𑀢𑁄”𑀢𑀺? “𑀢𑁆𑀯𑀜𑁆𑀘, 𑀤𑁂𑀯, 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳 𑀲𑀩𑁆𑀩𑁂 𑀚𑀭𑀸𑀥𑀫𑁆𑀫𑀸, 𑀚𑀭𑀁 𑀅𑀦𑀢𑀻𑀢𑀸”𑀢𑀺𑁇

95\. “𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀅𑀮𑀁 𑀤𑀸𑀦𑀚𑁆𑀚 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸, 𑀇𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀅𑀳𑀁, 𑀤𑁂𑀯, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀢𑁄𑀯 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺𑀁𑁇 𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀕𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺—  “𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀓𑀺𑀭, 𑀪𑁄, 𑀚𑀸𑀢𑀺 𑀦𑀸𑀫, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀚𑀸𑀢𑀲𑁆𑀲 𑀚𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”’𑀢𑀺𑁇
