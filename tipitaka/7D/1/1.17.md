# 𑁧𑁇𑁧𑁭 𑀤𑁂𑀯𑀢𑀸𑀭𑁄𑀘𑀦

### 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀭𑁄𑀘𑀦

181\. 𑀏𑀓𑀫𑀺𑀤𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑀁 𑀉𑀓𑁆𑀓𑀝𑁆𑀞𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀲𑀼𑀪𑀕𑀯𑀦𑁂 𑀲𑀸𑀮𑀭𑀸𑀚𑀫𑀽𑀮𑁂𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀦 𑀔𑁄 𑀲𑁄 𑀲𑀢𑁆𑀢𑀸𑀯𑀸𑀲𑁄 𑀲𑀼𑀮𑀪𑀭𑀽𑀧𑁄, 𑀬𑁄 𑀫𑀬𑀸 𑀅𑀦𑀸𑀯𑀼𑀢𑁆𑀣𑀧𑀼𑀩𑁆𑀩𑁄 𑀇𑀫𑀺𑀦𑀸 𑀤𑀻𑀖𑁂𑀦 𑀅𑀤𑁆𑀥𑀼𑀦𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑁂𑀳𑀺 𑀤𑁂𑀯𑁂𑀳𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀸 𑀤𑁂𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀉𑀓𑁆𑀓𑀝𑁆𑀞𑀸𑀬𑀁 𑀲𑀼𑀪𑀕𑀯𑀦𑁂 𑀲𑀸𑀮𑀭𑀸𑀚𑀫𑀽𑀮𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀅𑀯𑀺𑀳𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑀁𑁇 𑀢𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑁂 𑀅𑀦𑁂𑀓𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀇𑀢𑁄 𑀲𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀓𑀧𑁆𑀧𑁂 𑀬𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀻, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑁂 𑀉𑀤𑀧𑀸𑀤𑀺𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀻, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀅𑀳𑁄𑀲𑀺𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀲𑀻𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀻, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀧𑀸𑀝𑀮𑀺𑀬𑀸 𑀫𑀽𑀮𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀔𑀡𑁆𑀟𑀢𑀺𑀲𑁆𑀲𑀁 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀕𑁆𑀕𑀁 𑀪𑀤𑁆𑀤𑀬𑀼𑀕𑀁𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀢𑀬𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀏𑀓𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁𑁇 𑀏𑀓𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁𑁇 𑀏𑀓𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀅𑀲𑀻𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀩𑁆𑀩𑁂𑀲𑀁𑀬𑁂𑀯 𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀲𑁄𑀓𑁄 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀅𑀳𑁄𑀲𑀺 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀦𑀸𑀫 𑀭𑀸𑀚𑀸 𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀻 𑀦𑀸𑀫 𑀤𑁂𑀯𑀻 𑀫𑀸𑀢𑀸 𑀅𑀳𑁄𑀲𑀺 𑀚𑀦𑁂𑀢𑁆𑀢𑀺𑁇 𑀩𑀦𑁆𑀥𑀼𑀫𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀻 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀯𑀁 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑀁 𑀅𑀳𑁄𑀲𑀺 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀏𑀯𑀁 𑀧𑀥𑀸𑀦𑀁 𑀏𑀯𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀥𑀺 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀦𑀁𑁇 𑀢𑁂 𑀫𑀬𑀁, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀫𑁆𑀳𑀺 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 𑀯𑀺𑀭𑀸𑀚𑁂𑀢𑁆𑀯𑀸 𑀇𑀥𑀽𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺 …𑀧𑁂…

(93.)

185\. 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑁂 𑀅𑀦𑁂𑀓𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀤𑁆𑀤𑀓𑀧𑁆𑀧𑁂 𑀪𑀕𑀯𑀸 𑀏𑀢𑀭𑀳𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀪𑀕𑀯𑀸, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀚𑀸𑀢𑀺𑀬𑀸, 𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀪𑀕𑀯𑀸, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀕𑁄𑀢𑀫𑁄 𑀕𑁄𑀢𑁆𑀢𑁂𑀦𑁇 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀮𑀳𑀼𑀓𑀁 𑀬𑁄 𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀢𑀺, 𑀲𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀅𑀧𑁆𑀧𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀪𑀕𑀯𑀸, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀽𑀮𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀕𑁆𑀕𑀁 𑀪𑀤𑁆𑀤𑀬𑀼𑀕𑀁𑁇 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀏𑀓𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀅𑀟𑁆𑀠𑀢𑁂𑀍𑀅𑀲𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺𑁇 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀅𑀬𑀁 𑀏𑀓𑁄 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑀩𑁆𑀩𑁂𑀲𑀁𑀬𑁂𑀯 𑀔𑀻𑀡𑀸𑀲𑀯𑀸𑀦𑀁𑁇 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀆𑀦𑀦𑁆𑀤𑁄 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀅𑀕𑁆𑀕𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀦𑀸𑀫 𑀭𑀸𑀚𑀸 𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀫𑀸𑀬𑀸 𑀦𑀸𑀫 𑀤𑁂𑀯𑀻 𑀫𑀸𑀢𑀸 𑀅𑀳𑁄𑀲𑀺 𑀚𑀦𑁂𑀢𑁆𑀢𑀺𑁇 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀏𑀯𑀁 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀏𑀯𑀁 𑀧𑀥𑀸𑀦𑀁, 𑀏𑀯𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑁄𑀥𑀺, 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀦𑀁𑁇 𑀢𑁂 𑀫𑀬𑀁, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 𑀯𑀺𑀭𑀸𑀚𑁂𑀢𑁆𑀯𑀸 𑀇𑀥𑀽𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺𑁇

(94.)

186\. 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀯𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸, 𑀬𑀲𑁆𑀲𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀢𑁆𑀢𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀢𑀻𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀧𑀜𑁆𑀘𑁂 𑀙𑀺𑀦𑁆𑀦𑀯𑀝𑀼𑀫𑁂 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀯𑀝𑁆𑀝𑁂 𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁂 𑀚𑀸𑀢𑀺𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀦𑀸𑀫𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀕𑁄𑀢𑁆𑀢𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀲𑀸𑀯𑀓𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺 ‘𑀏𑀯𑀁𑀚𑀘𑁆𑀘𑀸 𑀢𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑁄 𑀅𑀳𑁂𑀲𑀼𑀁’ 𑀇𑀢𑀺𑀧𑀺𑁇 ‘𑀏𑀯𑀁𑀦𑀸𑀫𑀸 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑀸 𑀏𑀯𑀁𑀲𑀻𑀮𑀸 𑀏𑀯𑀁𑀥𑀫𑁆𑀫𑀸 𑀏𑀯𑀁𑀧𑀜𑁆𑀜𑀸 𑀏𑀯𑀁𑀯𑀺𑀳𑀸𑀭𑀻 𑀏𑀯𑀁𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀢𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑁄 𑀅𑀳𑁂𑀲𑀼𑀁’ 𑀇𑀢𑀺𑀧𑀻𑀢𑀺𑁇

187\. 𑀤𑁂𑀯𑀢𑀸𑀧𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁, 𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀢𑀻𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀙𑀺𑀦𑁆𑀦𑀧𑀧𑀜𑁆𑀘𑁂 𑀙𑀺𑀦𑁆𑀦𑀯𑀝𑀼𑀫𑁂 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀯𑀝𑁆𑀝𑁂 𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁂 𑀚𑀸𑀢𑀺𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀦𑀸𑀫𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀕𑁄𑀢𑁆𑀢𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺, 𑀲𑀸𑀯𑀓𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀢𑁄𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀺 ‘𑀏𑀯𑀁𑀚𑀘𑁆𑀘𑀸 𑀢𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑁄 𑀅𑀳𑁂𑀲𑀼𑀁’ 𑀇𑀢𑀺𑀧𑀺𑁇 ‘𑀏𑀯𑀁𑀦𑀸𑀫𑀸 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑀸 𑀏𑀯𑀁𑀲𑀻𑀮𑀸 𑀏𑀯𑀁𑀥𑀫𑁆𑀫𑀸 𑀏𑀯𑀁𑀧𑀜𑁆𑀜𑀸 𑀏𑀯𑀁𑀯𑀺𑀳𑀸𑀭𑀻 𑀏𑀯𑀁𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀢𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑁄 𑀅𑀳𑁂𑀲𑀼𑀁’ 𑀇𑀢𑀺𑀧𑀻”𑀢𑀺𑁇

188\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---
