# 𑁧𑁇𑁮 𑀧𑀩𑁆𑀩𑀚𑀺𑀢

### 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀩𑁆𑀩𑀚𑀺𑀢

(52.)

105\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀫𑀸 𑀳𑁂𑀯 𑀔𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀦 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀲𑀺, 𑀫𑀸 𑀳𑁂𑀯 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺, 𑀫𑀸 𑀳𑁂𑀯 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀲𑀘𑁆𑀘𑀁 𑀅𑀲𑁆𑀲 𑀯𑀘𑀦𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑀦𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺—  ‘𑀬𑀣𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀦 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸 𑀅𑀲𑁆𑀲 𑀯𑀘𑀦𑀦𑁆’𑀢𑀺𑁇

106\. 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀩𑀳𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀬𑁄𑀚𑁂𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺, 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀲𑀼𑀪𑀽𑀫𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀣𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀲𑀺—  ‘𑀬𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀤𑁂𑀯, 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀤𑁆𑀤𑀁 𑀪𑀤𑁆𑀤𑀁 𑀬𑀸𑀦𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀤𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑁂𑀳𑀺 𑀬𑀸𑀦𑁂𑀳𑀺 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺𑁇

(53.)

107\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑀁 𑀪𑀡𑁆𑀟𑀼𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀁 𑀓𑀸𑀲𑀸𑀬𑀯𑀲𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀸𑀭𑀣𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀅𑀬𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀁𑀓𑀢𑁄? 𑀲𑀻𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀁, 𑀯𑀢𑁆𑀣𑀸𑀦𑀺𑀧𑀺𑀲𑁆𑀲 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆’𑀢𑀺? ‘𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦𑁂𑀲𑁄, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺? ‘𑀏𑀲𑁄 𑀔𑁄, 𑀤𑁂𑀯, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫 𑀲𑀸𑀥𑀼 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀲𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀓𑀼𑀲𑀮𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸 𑀲𑀸𑀥𑀼 𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸’𑀢𑀺𑁇 ‘𑀲𑀸𑀥𑀼 𑀔𑁄 𑀲𑁄, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫, 𑀲𑀸𑀥𑀼 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀲𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀓𑀼𑀲𑀮𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸 𑀲𑀸𑀥𑀼 𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀸𑀭𑀣𑀺, 𑀬𑁂𑀦 𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀢𑁂𑀦 𑀭𑀣𑀁 𑀧𑁂𑀲𑁂𑀳𑀻’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀣𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀢𑁂𑀦 𑀭𑀣𑀁 𑀧𑁂𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀢𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀢𑁆𑀯𑀁 𑀧𑀦, 𑀲𑀫𑁆𑀫, 𑀓𑀺𑀁𑀓𑀢𑁄, 𑀲𑀻𑀲𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀁, 𑀯𑀢𑁆𑀣𑀸𑀦𑀺𑀧𑀺 𑀢𑁂 𑀦 𑀬𑀣𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆’𑀢𑀺? ‘𑀅𑀳𑀁 𑀔𑁄, 𑀤𑁂𑀯, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺𑁇 ‘𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀲𑀫𑁆𑀫, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺? ‘𑀅𑀳𑀁 𑀔𑁄, 𑀤𑁂𑀯, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫, 𑀲𑀸𑀥𑀼 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀲𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀓𑀼𑀲𑀮𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸 𑀲𑀸𑀥𑀼 𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸’𑀢𑀺𑁇 ‘𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀫𑁆𑀫, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀫 𑀲𑀸𑀥𑀼 𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀲𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀓𑀼𑀲𑀮𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀧𑀼𑀜𑁆𑀜𑀓𑀺𑀭𑀺𑀬𑀸 𑀲𑀸𑀥𑀼 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸 𑀲𑀸𑀥𑀼 𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸’𑀢𑀺𑁇
