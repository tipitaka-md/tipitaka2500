# 𑁧𑁇𑁪 𑀯𑀺𑀧𑀲𑁆𑀲𑀻𑀲𑀫𑀜𑁆𑀜𑀸

### 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀺𑀧𑀲𑁆𑀲𑀻𑀲𑀫𑀜𑁆𑀜𑀸

(37.)

85. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀅𑀳𑀢𑁂𑀳𑀺 𑀯𑀢𑁆𑀣𑁂𑀳𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑁂𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀥𑀸𑀢𑀺𑀬𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀸 𑀔𑀻𑀭𑀁 𑀧𑀸𑀬𑁂𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀸 𑀦𑁆𑀳𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀸 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀸 𑀅𑀗𑁆𑀓𑁂𑀦 𑀧𑀭𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀚𑀸𑀢𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀣 𑀤𑀺𑀯𑀸 𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘—  ‘𑀫𑀸 𑀦𑀁 𑀲𑀻𑀢𑀁 𑀯𑀸 𑀉𑀡𑁆𑀳𑀁 𑀯𑀸 𑀢𑀺𑀡𑀁 𑀯𑀸 𑀭𑀚𑁄 𑀯𑀸 𑀉𑀲𑁆𑀲𑀸𑀯𑁄 𑀯𑀸 𑀩𑀸𑀥𑀬𑀺𑀢𑁆𑀣𑀸’𑀢𑀺𑁇 𑀚𑀸𑀢𑁄 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀧𑀺𑀬𑁄 𑀅𑀳𑁄𑀲𑀺 𑀫𑀦𑀸𑀧𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁆𑀧𑀮𑀁 𑀯𑀸 𑀧𑀤𑀼𑀫𑀁 𑀯𑀸 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀁 𑀯𑀸 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀧𑀺𑀬𑁄 𑀅𑀳𑁄𑀲𑀺 𑀫𑀦𑀸𑀧𑁄𑁇 𑀲𑁆𑀯𑀸𑀲𑁆𑀲𑀼𑀤𑀁 𑀅𑀗𑁆𑀓𑁂𑀦𑁂𑀯 𑀅𑀗𑁆𑀓𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀬𑀢𑀺𑁇

(38.)

86. 𑀚𑀸𑀢𑁄 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑀜𑁆𑀚𑀼𑀲𑁆𑀲𑀭𑁄 𑀘 𑀅𑀳𑁄𑀲𑀺 𑀯𑀕𑁆𑀕𑀼𑀲𑁆𑀲𑀭𑁄 𑀘 𑀫𑀥𑀼𑀭𑀲𑁆𑀲𑀭𑁄 𑀘 𑀧𑁂𑀫𑀦𑀺𑀬𑀲𑁆𑀲𑀭𑁄 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀓𑀭𑀯𑀻𑀓𑀸 𑀦𑀸𑀫 𑀲𑀓𑀼𑀡𑀚𑀸𑀢𑀺 𑀫𑀜𑁆𑀚𑀼𑀲𑁆𑀲𑀭𑀸 𑀘 𑀯𑀕𑁆𑀕𑀼𑀲𑁆𑀲𑀭𑀸 𑀘 𑀫𑀥𑀼𑀭𑀲𑁆𑀲𑀭𑀸 𑀘 𑀧𑁂𑀫𑀦𑀺𑀬𑀲𑁆𑀲𑀭𑀸 𑀘; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑀜𑁆𑀚𑀼𑀲𑁆𑀲𑀭𑁄 𑀘 𑀅𑀳𑁄𑀲𑀺 𑀯𑀕𑁆𑀕𑀼𑀲𑁆𑀲𑀭𑁄 𑀘 𑀫𑀥𑀼𑀭𑀲𑁆𑀲𑀭𑁄 𑀘 𑀧𑁂𑀫𑀦𑀺𑀬𑀲𑁆𑀲𑀭𑁄 𑀘𑁇

(39.)

87. 𑀚𑀸𑀢𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀚𑀁 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀬𑁂𑀦 𑀲𑀼𑀤𑀁 𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀤𑀺𑀯𑀸 𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘𑁇

(40.)

88. 𑀚𑀸𑀢𑁄 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀦𑀺𑀫𑀺𑀲𑀦𑁆𑀢𑁄 𑀧𑁂𑀓𑁆𑀔𑀢𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑁇 ‘𑀅𑀦𑀺𑀫𑀺𑀲𑀦𑁆𑀢𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑁂𑀓𑁆𑀔𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 ‘𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀧𑀲𑁆𑀲𑀻’ 𑀢𑁆𑀯𑁂𑀯 𑀲𑀫𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

(41.)

89. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀅𑀢𑁆𑀣𑀓𑀭𑀡𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀅𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀣𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀺𑀢𑀼𑀅𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀅𑀢𑁆𑀣𑁂 𑀧𑀦𑀸𑀬𑀢𑀺 𑀜𑀸𑀬𑁂𑀦𑁇 𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀢𑁆𑀣𑁂 𑀧𑀦𑀸𑀬𑀢𑀺 𑀜𑀸𑀬𑁂𑀦𑀸𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 ‘𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀧𑀲𑁆𑀲𑀻’ 𑀢𑁆𑀯𑁂𑀯 𑀲𑀫𑀜𑁆𑀜𑀸 𑀉𑀤𑀧𑀸𑀤𑀺𑁇

(42.)

90. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑁂 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑀺, 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀏𑀓𑀁 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀁 𑀏𑀓𑀁 𑀕𑀺𑀫𑁆𑀳𑀺𑀓𑀁; 𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑀦𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀓𑀼𑀫𑀸𑀭𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀧𑀸𑀲𑀸𑀤𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀳𑀺 𑀢𑀽𑀭𑀺𑀬𑁂𑀳𑀺 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀫𑀸𑀦𑁄 𑀦 𑀳𑁂𑀝𑁆𑀞𑀸𑀧𑀸𑀲𑀸𑀤𑀁 𑀑𑀭𑁄𑀳𑀢𑀻𑀢𑀺𑁇

---

91. 𑀧𑀞𑀫𑀪𑀸𑀡𑀯𑀸𑀭𑁄𑁇
