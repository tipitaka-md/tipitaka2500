# 𑁧𑁇𑁩 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀸

* 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀸

(33.)

50\. 𑀚𑀸𑀢𑁂 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀫𑁆𑀳𑀺 𑀓𑀼𑀫𑀸𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀲𑀼𑀁—  ‘𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂, 𑀤𑁂𑀯, 𑀚𑀸𑀢𑁄, 𑀢𑀁 𑀤𑁂𑀯𑁄 𑀧𑀲𑁆𑀲𑀢𑀽’𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀦𑁆𑀥𑀼𑀫𑀸 𑀭𑀸𑀚𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀆𑀫𑀦𑁆𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀼𑀫𑀸𑀭𑀦𑁆’𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸 𑀩𑀦𑁆𑀥𑀼𑀫𑀦𑁆𑀢𑀁 𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀅𑀢𑁆𑀢𑀫𑀦𑁄, 𑀤𑁂𑀯, 𑀳𑁄𑀳𑀺, 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑁄 𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀮𑀸𑀪𑀸 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀓𑀼𑀮𑁂 𑀏𑀯𑀭𑀽𑀧𑁄 𑀧𑀼𑀢𑁆𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀬𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂𑀯 𑀕𑀢𑀺𑀬𑁄 𑀪𑀯𑀦𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑁇 𑀲𑀘𑁂 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺, 𑀭𑀸𑀚𑀸 𑀳𑁄𑀢𑀺 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀚𑀦𑀧𑀤𑀢𑁆𑀣𑀸𑀯𑀭𑀺𑀬𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑀲𑁆𑀲𑀺𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀇𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀫𑁂𑀯 𑀲𑀢𑁆𑀢𑀫𑀁𑁇 𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺 𑀲𑀽𑀭𑀸 𑀯𑀻𑀭𑀗𑁆𑀕𑀭𑀽𑀧𑀸 𑀧𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑀸𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀧𑀣𑀯𑀺𑀁 𑀲𑀸𑀕𑀭𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀪𑀺𑀯𑀺𑀚𑀺𑀬 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀧𑀦 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀢𑀺, 𑀅𑀭𑀳𑀁 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀯𑀺𑀯𑀝𑀘𑁆𑀙𑀤𑁄𑁇

(34.)

51\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀸𑀬𑀁, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀬𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂𑀯 𑀕𑀢𑀺𑀬𑁄 𑀪𑀯𑀦𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑁇 𑀲𑀘𑁂 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺, 𑀭𑀸𑀚𑀸 𑀳𑁄𑀢𑀺 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀚𑀦𑀧𑀤𑀢𑁆𑀣𑀸𑀯𑀭𑀺𑀬𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑀲𑁆𑀲𑀺𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀇𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀫𑁂𑀯 𑀲𑀢𑁆𑀢𑀫𑀁𑁇 𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺 𑀲𑀽𑀭𑀸 𑀯𑀻𑀭𑀗𑁆𑀕𑀭𑀽𑀧𑀸 𑀧𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑀸𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀧𑀣𑀯𑀺𑀁 𑀲𑀸𑀕𑀭𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀪𑀺𑀯𑀺𑀚𑀺𑀬 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀧𑀦 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀢𑀺, 𑀅𑀭𑀳𑀁 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀯𑀺𑀯𑀝𑀘𑁆𑀙𑀤𑁄𑁇

(35.)

52\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀧𑀸𑀤𑁄𑁇 𑀬𑀁 𑀧𑀸𑀬𑀁, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀧𑀸𑀤𑁄𑁇 𑀇𑀤𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀁 𑀪𑀯𑀢𑀺𑁇 (𑁧)

53\. 𑀇𑀫𑀲𑁆𑀲, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀳𑁂𑀝𑁆𑀞𑀸 𑀧𑀸𑀤𑀢𑀮𑁂𑀲𑀼 𑀘𑀓𑁆𑀓𑀸𑀦𑀺 𑀚𑀸𑀢𑀸𑀦𑀺 𑀲𑀳𑀲𑁆𑀲𑀸𑀭𑀸𑀦𑀺 𑀲𑀦𑁂𑀫𑀺𑀓𑀸𑀦𑀺 𑀲𑀦𑀸𑀪𑀺𑀓𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀓𑀸𑀭𑀧𑀭𑀺𑀧𑀽𑀭𑀸𑀦𑀺𑁇 𑀬𑀫𑁆𑀧𑀺 𑀇𑀫𑀲𑁆𑀲, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀳𑁂𑀝𑁆𑀞𑀸 𑀧𑀸𑀤𑀢𑀮𑁂𑀲𑀼 𑀘𑀓𑁆𑀓𑀸𑀦𑀺 𑀚𑀸𑀢𑀸𑀦𑀺 𑀲𑀳𑀲𑁆𑀲𑀸𑀭𑀸𑀦𑀺 𑀲𑀦𑁂𑀫𑀺𑀓𑀸𑀦𑀺 𑀲𑀦𑀸𑀪𑀺𑀓𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀓𑀸𑀭𑀧𑀭𑀺𑀧𑀽𑀭𑀸𑀦𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀁 𑀪𑀯𑀢𑀺𑁇 (𑁨)

54\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀆𑀬𑀢𑀧𑀡𑁆𑀳𑀻…𑀧𑁂…  (𑁩)

55\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑀻𑀖𑀗𑁆𑀕𑀼𑀮𑀻…𑀧𑁂…  (𑁪)

56\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑀼𑀤𑀼𑀢𑀮𑀼𑀦𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑁄…𑀧𑁂…  (𑁫)

57\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯 𑀓𑀼𑀫𑀸𑀭𑁄 𑀚𑀸𑀮𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑁄…𑀧𑁂…  (𑁬)

58\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀲𑁆𑀲𑀗𑁆𑀔𑀧𑀸𑀤𑁄…𑀧𑁂…  (𑁭)

59\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀏𑀡𑀺𑀚𑀗𑁆𑀖𑁄…𑀧𑁂…  (𑁮)

60\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀞𑀺𑀢𑀓𑁄𑀯 𑀅𑀦𑁄𑀦𑀫𑀦𑁆𑀢𑁄 𑀉𑀪𑁄𑀳𑀺 𑀧𑀸𑀡𑀺𑀢𑀮𑁂𑀳𑀺 𑀚𑀡𑁆𑀡𑀼𑀓𑀸𑀦𑀺 𑀧𑀭𑀺𑀫𑀲𑀢𑀺 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  (𑁯)

61\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀓𑁄𑀲𑁄𑀳𑀺𑀢𑀯𑀢𑁆𑀣𑀕𑀼𑀬𑁆𑀳𑁄…𑀧𑁂…  (𑁧𑁦)

62\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑁄 𑀓𑀜𑁆𑀘𑀦𑀲𑀦𑁆𑀦𑀺𑀪𑀢𑁆𑀢𑀘𑁄…𑀧𑁂…  (𑁧𑁧)

63\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀼𑀔𑀼𑀫𑀘𑁆𑀙𑀯𑀻; 𑀲𑀼𑀔𑀼𑀫𑀢𑁆𑀢𑀸 𑀙𑀯𑀺𑀬𑀸 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀓𑀸𑀬𑁂 𑀦 𑀉𑀧𑀮𑀺𑀫𑁆𑀧𑀢𑀺…𑀧𑁂…  (𑁧𑁨)

64\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀏𑀓𑁂𑀓𑀮𑁄𑀫𑁄; 𑀏𑀓𑁂𑀓𑀸𑀦𑀺 𑀮𑁄𑀫𑀸𑀦𑀺 𑀮𑁄𑀫𑀓𑀽𑀧𑁂𑀲𑀼 𑀚𑀸𑀢𑀸𑀦𑀺…𑀧𑁂…  (𑁧𑁩)

65\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀤𑁆𑀥𑀕𑁆𑀕𑀮𑁄𑀫𑁄; 𑀉𑀤𑁆𑀥𑀕𑁆𑀕𑀸𑀦𑀺 𑀮𑁄𑀫𑀸𑀦𑀺 𑀚𑀸𑀢𑀸𑀦𑀺 𑀦𑀻𑀮𑀸𑀦𑀺 𑀅𑀜𑁆𑀚𑀦𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀓𑀼𑀡𑁆𑀟𑀮𑀸𑀯𑀝𑁆𑀝𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀯𑀝𑁆𑀝𑀓𑀚𑀸𑀢𑀸𑀦𑀺…𑀧𑁂…  (𑁧𑁪)

66\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀚𑀼𑀕𑀢𑁆𑀢𑁄…𑀧𑁂…  (𑁧𑁫)

67\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀢𑁆𑀢𑀼𑀲𑁆𑀲𑀤𑁄…𑀧𑁂…  (𑁧𑁬)

68\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀻𑀳𑀧𑀼𑀩𑁆𑀩𑀤𑁆𑀥𑀓𑀸𑀬𑁄…𑀧𑁂…  (𑁧𑁭)

69\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀘𑀺𑀢𑀦𑁆𑀢𑀭𑀁𑀲𑁄…𑀧𑁂…  (𑁧𑁮)

70\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑁄 𑀬𑀸𑀯𑀢𑀓𑁆𑀯𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀢𑀸𑀯𑀢𑀓𑁆𑀯𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀫𑁄, 𑀬𑀸𑀯𑀢𑀓𑁆𑀯𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀫𑁄, 𑀢𑀸𑀯𑀢𑀓𑁆𑀯𑀲𑁆𑀲 𑀓𑀸𑀬𑁄…𑀧𑁂…  (𑁧𑁯)

71\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀫𑀯𑀝𑁆𑀝𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  (𑁨𑁦)

72\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀭𑀲𑀕𑁆𑀕𑀲𑀕𑁆𑀕𑀻…𑀧𑁂…  (𑁨𑁧)

73\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀻𑀳𑀳𑀦𑀼…𑀧𑁂…  (𑁨𑁨)

74\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀤𑀦𑁆𑀢𑁄…𑀧𑁂…  (𑁨𑁩)

75\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀫𑀤𑀦𑁆𑀢𑁄…𑀧𑁂…  (𑁨𑁪)

76\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀯𑀺𑀭𑀍𑀅𑀤𑀦𑁆𑀢𑁄…𑀧𑁂…  (𑁨𑁫)

77\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀼𑀲𑀼𑀓𑁆𑀓𑀤𑀸𑀞𑁄…𑀧𑁂…  (𑁨𑁬)

78\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀧𑀳𑀽𑀢𑀚𑀺𑀯𑁆𑀳𑁄…𑀧𑁂…  (𑁨𑁭)

79\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑁆𑀲𑀭𑁄 𑀓𑀭𑀯𑀻𑀓𑀪𑀸𑀡𑀻…𑀧𑁂…  (𑁨𑁮)

80\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀪𑀺𑀦𑀻𑀮𑀦𑁂𑀢𑁆𑀢𑁄…𑀧𑁂…  (𑁨𑁯)

81\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀕𑁄𑀧𑀔𑀼𑀫𑁄…𑀧𑁂…  (𑁩𑁦)

82\. 𑀇𑀫𑀲𑁆𑀲, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀉𑀡𑁆𑀡𑀸 𑀪𑀫𑀼𑀓𑀦𑁆𑀢𑀭𑁂 𑀚𑀸𑀢𑀸 𑀑𑀤𑀸𑀢𑀸 𑀫𑀼𑀤𑀼𑀢𑀽𑀮𑀲𑀦𑁆𑀦𑀺𑀪𑀸𑁇 𑀬𑀫𑁆𑀧𑀺 𑀇𑀫𑀲𑁆𑀲, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀉𑀡𑁆𑀡𑀸 𑀪𑀫𑀼𑀓𑀦𑁆𑀢𑀭𑁂 𑀚𑀸𑀢𑀸 𑀑𑀤𑀸𑀢𑀸 𑀫𑀼𑀤𑀼𑀢𑀽𑀮𑀲𑀦𑁆𑀦𑀺𑀪𑀸, 𑀇𑀤𑀫𑁆𑀧𑀺𑀫𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀁 𑀪𑀯𑀢𑀺𑁇 (𑁩𑁧)

83\. 𑀅𑀬𑀜𑁆𑀳𑀺, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀡𑁆𑀳𑀻𑀲𑀲𑀻𑀲𑁄𑁇 𑀬𑀁 𑀧𑀸𑀬𑀁, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀉𑀡𑁆𑀳𑀻𑀲𑀲𑀻𑀲𑁄, 𑀇𑀤𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀁 𑀪𑀯𑀢𑀺𑁇 (𑁩𑁨)

(36.)

84\. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄 𑀅𑀬𑀁, 𑀤𑁂𑀯, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀬𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂𑀯 𑀕𑀢𑀺𑀬𑁄 𑀪𑀯𑀦𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑁇 𑀲𑀘𑁂 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺, 𑀭𑀸𑀚𑀸 𑀳𑁄𑀢𑀺 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀚𑀦𑀧𑀤𑀢𑁆𑀣𑀸𑀯𑀭𑀺𑀬𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑀲𑁆𑀲𑀺𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀸𑀦𑀺 𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀇𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀫𑁂𑀯 𑀲𑀢𑁆𑀢𑀫𑀁𑁇 𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺 𑀲𑀽𑀭𑀸 𑀯𑀻𑀭𑀗𑁆𑀕𑀭𑀽𑀧𑀸 𑀧𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑀸𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀧𑀣𑀯𑀺𑀁 𑀲𑀸𑀕𑀭𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀪𑀺𑀯𑀺𑀚𑀺𑀬 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀧𑀦 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀢𑀺, 𑀅𑀭𑀳𑀁 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀯𑀺𑀯𑀝𑀘𑁆𑀙𑀤𑁄’𑀢𑀺𑁇
