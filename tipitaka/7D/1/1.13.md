# 𑁧𑁇𑁧𑁩 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀬𑀼𑀕

* 𑀫𑀳𑀸𑀧𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀬𑀼𑀕

(72.)

157\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁, 𑀓𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺? 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀔𑀡𑁆𑀟𑁄 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀭𑀚𑀓𑁆𑀔𑀚𑀸𑀢𑀺𑀓𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀔𑀡𑁆𑀟𑀲𑁆𑀲 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲, 𑀢𑀺𑀲𑁆𑀲𑀲𑁆𑀲 𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀁, 𑀢𑁂 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇

(73.)

158\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀩𑁄𑀥𑀺𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀔𑁂𑀫𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑀸𑀬𑀧𑀸𑀮𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀫𑁆𑀫 𑀤𑀸𑀬𑀧𑀸𑀮, 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀔𑀡𑁆𑀟𑀜𑁆𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀢𑀺𑀲𑁆𑀲𑀜𑁆𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀺—  𑀯𑀺𑀧𑀲𑁆𑀲𑀻, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀔𑁂𑀫𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑁄’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀬𑀧𑀸𑀮𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀔𑀡𑁆𑀟𑀜𑁆𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀢𑀺𑀲𑁆𑀲𑀜𑁆𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀯𑀺𑀧𑀲𑁆𑀲𑀻, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀔𑁂𑀫𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑁄 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑁄’𑀢𑀺𑁇

(74.)

159\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀡𑁆𑀟𑁄 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀤𑀁 𑀪𑀤𑁆𑀤𑀁 𑀬𑀸𑀦𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀤𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑁂𑀳𑀺 𑀬𑀸𑀦𑁂𑀳𑀺 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀺𑀁𑀲𑀼𑁇 𑀬𑁂𑀦 𑀔𑁂𑀫𑁄 𑀫𑀺𑀕𑀤𑀸𑀬𑁄 𑀢𑁂𑀦 𑀧𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀬𑀸𑀦𑀲𑁆𑀲 𑀪𑀽𑀫𑀺, 𑀬𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑀸𑀦𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀯 𑀬𑁂𑀦 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇

(75.)

160\. 𑀢𑁂𑀲𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑁂 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑁂 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁂, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀺𑀭𑁄𑀥𑀁 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑀡𑁆𑀟𑀲𑁆𑀲 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀺𑀲𑁆𑀲𑀲𑁆𑀲 𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆’𑀢𑀺𑁇

(76.)

161\. 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑀸 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑀸 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  “𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻”𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀢𑁂 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀥𑀫𑁆𑀫𑀜𑁆𑀘𑁇 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆’𑀢𑀺𑁇

(77.)

162\. 𑀅𑀮𑀢𑁆𑀣𑀼𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀡𑁆𑀟𑁄 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀅𑀮𑀢𑁆𑀣𑀼𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀢𑁂 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺; 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀢𑁂𑀲𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑀸 𑀪𑀕𑀯𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀲𑀫𑀸𑀤𑀧𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼𑁇
