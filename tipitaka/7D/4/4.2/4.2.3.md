# 𑁪𑁇𑁨𑁇𑁩 𑀅𑀲𑁆𑀲𑀭𑀢𑀦

### 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢

### 𑀅𑀲𑁆𑀲𑀭𑀢𑀦

(247.)

544\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀲𑀩𑁆𑀩𑀲𑁂𑀢𑁄 𑀓𑀸𑀍𑀅𑀲𑀻𑀲𑁄 𑀫𑀼𑀜𑁆𑀚𑀓𑁂𑀲𑁄 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀯𑁂𑀳𑀸𑀲𑀗𑁆𑀕𑀫𑁄 𑀯𑀮𑀸𑀳𑀓𑁄 𑀦𑀸𑀫 𑀅𑀲𑁆𑀲𑀭𑀸𑀚𑀸𑁇 𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀺—  ‘𑀪𑀤𑁆𑀤𑀓𑀁 𑀯𑀢 𑀪𑁄 𑀅𑀲𑁆𑀲𑀬𑀸𑀦𑀁 𑀲𑀘𑁂 𑀤𑀫𑀣𑀁 𑀉𑀧𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀪𑀤𑁆𑀤𑁄 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀺𑀬𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀼𑀧𑀭𑀺𑀤𑀦𑁆𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀤𑀫𑀣𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀫𑁂𑀯 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀧𑀣𑀯𑀺𑀁 𑀅𑀦𑀼𑀬𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀲𑀸𑀯𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀸𑀢𑀭𑀸𑀲𑀫𑀓𑀸𑀲𑀺𑁇 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇
