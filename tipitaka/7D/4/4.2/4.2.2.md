# 𑁪𑁇𑁨𑁇𑁨 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦

### 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢

### 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦

(246.)

543. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀲𑀩𑁆𑀩𑀲𑁂𑀢𑁄 𑀲𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑁄 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀯𑁂𑀳𑀸𑀲𑀗𑁆𑀕𑀫𑁄 𑀉𑀧𑁄𑀲𑀣𑁄 𑀦𑀸𑀫 𑀦𑀸𑀕𑀭𑀸𑀚𑀸𑁇 𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀺—  ‘𑀪𑀤𑁆𑀤𑀓𑀁 𑀯𑀢 𑀪𑁄 𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁, 𑀲𑀘𑁂 𑀤𑀫𑀣𑀁 𑀉𑀧𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀦𑁆𑀥𑀳𑀢𑁆𑀣𑀸𑀚𑀸𑀦𑀺𑀬𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀼𑀧𑀭𑀺𑀤𑀦𑁆𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀤𑀫𑀣𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀫𑁂𑀯 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀧𑀣𑀯𑀺𑀁 𑀅𑀦𑀼𑀬𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀲𑀸𑀯𑀢𑀺𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀸𑀢𑀭𑀸𑀲𑀫𑀓𑀸𑀲𑀺𑁇 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇
