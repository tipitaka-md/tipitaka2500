# 𑁪𑁇𑁨𑁇𑁭 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦

### 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢

### 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦

(251.)

548. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑀺𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀧𑀝𑀺𑀩𑀮𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁 𑀉𑀧𑀬𑀸𑀧𑁂𑀢𑀼𑀁, 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁 𑀅𑀧𑀬𑀸𑀧𑁂𑀢𑀼𑀁, 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑀁 𑀞𑀧𑁂𑀢𑀼𑀁𑁇 𑀲𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳—  ‘𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑁆𑀯𑀁, 𑀤𑁂𑀯, 𑀳𑁄𑀳𑀺, 𑀅𑀳𑀫𑀦𑀼𑀲𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇

549. 𑀭𑀸𑀚𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀇𑀫𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀳𑀺 𑀭𑀢𑀦𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇
