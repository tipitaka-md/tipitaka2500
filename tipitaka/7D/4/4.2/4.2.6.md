# 𑁪𑁇𑁨𑁇𑁬 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦

* 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢

* 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦

(250.)

547\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀚𑀁 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀬𑁂𑀦 𑀦𑀺𑀥𑀺𑀁 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑀲𑁆𑀲𑀸𑀫𑀺𑀓𑀫𑁆𑀧𑀺 𑀅𑀲𑁆𑀲𑀸𑀫𑀺𑀓𑀫𑁆𑀧𑀺𑁇 𑀲𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳—  ‘𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑁆𑀯𑀁, 𑀤𑁂𑀯, 𑀳𑁄𑀳𑀺, 𑀅𑀳𑀁 𑀢𑁂 𑀥𑀦𑁂𑀦 𑀥𑀦𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀫𑁂𑀯 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀦𑀸𑀯𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀲𑁄𑀢𑀁 𑀑𑀕𑀸𑀳𑀺𑀢𑁆𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀅𑀢𑁆𑀣𑁄 𑀫𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀳𑀺𑀭𑀜𑁆𑀜𑀲𑀼𑀯𑀡𑁆𑀡𑁂𑀦𑀸’𑀢𑀺𑁇 ‘𑀢𑁂𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀁 𑀢𑀻𑀭𑀁 𑀦𑀸𑀯𑀸 𑀉𑀧𑁂𑀢𑀽’𑀢𑀺𑁇 ‘𑀇𑀥𑁂𑀯 𑀫𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀢𑁆𑀣𑁄 𑀳𑀺𑀭𑀜𑁆𑀜𑀲𑀼𑀯𑀡𑁆𑀡𑁂𑀦𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀉𑀪𑁄𑀳𑀺 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀉𑀤𑀓𑀁 𑀑𑀫𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀽𑀭𑀁 𑀳𑀺𑀭𑀜𑁆𑀜𑀲𑀼𑀯𑀡𑁆𑀡𑀲𑁆𑀲 𑀓𑀼𑀫𑁆𑀪𑀺𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀅𑀮𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀢𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀽𑀚𑀺𑀢𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸’𑀢𑀺? 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀅𑀮𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀢𑀸, 𑀕𑀳𑀧𑀢𑀺, 𑀓𑀢𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀢𑀸, 𑀕𑀳𑀧𑀢𑀺, 𑀧𑀽𑀚𑀺𑀢𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀢𑀸, 𑀕𑀳𑀧𑀢𑀻’𑀢𑀺𑁇 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇
