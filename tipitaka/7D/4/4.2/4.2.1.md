# 𑁪𑁇𑁨𑁇𑁧 𑀘𑀓𑁆𑀓𑀭𑀢𑀦

* 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢

* 𑀘𑀓𑁆𑀓𑀭𑀢𑀦

(243.)

540\. 𑀭𑀸𑀚𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀢𑁆𑀢𑀳𑀺 𑀭𑀢𑀦𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀘𑀢𑀽𑀳𑀺 𑀘 𑀇𑀤𑁆𑀥𑀻𑀳𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀳𑀺? 𑀇𑀥𑀸𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑀲𑁆𑀲 𑀉𑀧𑁄𑀲𑀣𑀺𑀓𑀲𑁆𑀲 𑀉𑀧𑀭𑀺𑀧𑀸𑀲𑀸𑀤𑀯𑀭𑀕𑀢𑀲𑁆𑀲 𑀤𑀺𑀩𑁆𑀩𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀲𑀳𑀲𑁆𑀲𑀸𑀭𑀁 𑀲𑀦𑁂𑀫𑀺𑀓𑀁 𑀲𑀦𑀸𑀪𑀺𑀓𑀁 𑀲𑀩𑁆𑀩𑀸𑀓𑀸𑀭𑀧𑀭𑀺𑀧𑀽𑀭𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀼𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂𑀢𑀁—  “𑀬𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑀲𑁆𑀲 𑀉𑀧𑁄𑀲𑀣𑀺𑀓𑀲𑁆𑀲 𑀉𑀧𑀭𑀺𑀧𑀸𑀲𑀸𑀤𑀯𑀭𑀕𑀢𑀲𑁆𑀲 𑀤𑀺𑀩𑁆𑀩𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺 𑀲𑀳𑀲𑁆𑀲𑀸𑀭𑀁 𑀲𑀦𑁂𑀫𑀺𑀓𑀁 𑀲𑀦𑀸𑀪𑀺𑀓𑀁 𑀲𑀩𑁆𑀩𑀸𑀓𑀸𑀭𑀧𑀭𑀺𑀧𑀽𑀭𑀁, 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻’𑀢𑀺𑁇

(244.)

541\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸𑀫𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀲𑀼𑀯𑀡𑁆𑀡𑀪𑀺𑀗𑁆𑀓𑀸𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀅𑀩𑁆𑀪𑀼𑀓𑁆𑀓𑀺𑀭𑀺—  ‘𑀧𑀯𑀢𑁆𑀢𑀢𑀼 𑀪𑀯𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁, 𑀅𑀪𑀺𑀯𑀺𑀚𑀺𑀦𑀸𑀢𑀼 𑀪𑀯𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀁 𑀤𑀺𑀲𑀁 𑀧𑀯𑀢𑁆𑀢𑀺, 𑀅𑀦𑁆𑀯𑀤𑁂𑀯 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀧𑀤𑁂𑀲𑁂 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀲𑀺, 𑀢𑀢𑁆𑀣 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬𑁇 𑀬𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀧𑀝𑀺𑀭𑀸𑀚𑀸𑀦𑁄, 𑀢𑁂 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀏𑀳𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑀁 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑀼𑀲𑀸𑀲, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸’𑀢𑀺𑁇 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀧𑀸𑀡𑁄 𑀦 𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀁, 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀦 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀸, 𑀫𑀼𑀲𑀸 𑀦 𑀪𑀡𑀺𑀢𑀩𑁆𑀩𑀸, 𑀫𑀚𑁆𑀚𑀁 𑀦 𑀧𑀸𑀢𑀩𑁆𑀩𑀁, 𑀬𑀣𑀸𑀪𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀼𑀜𑁆𑀚𑀣𑀸’𑀢𑀺𑁇 𑀬𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀧𑀝𑀺𑀭𑀸𑀚𑀸𑀦𑁄, 𑀢𑁂 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀤𑀺𑀲𑀁 𑀧𑀯𑀢𑁆𑀢𑀺…𑀧𑁂…  𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀤𑀺𑀲𑀁 𑀧𑀯𑀢𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀁 𑀤𑀺𑀲𑀁 𑀧𑀯𑀢𑁆𑀢𑀺, 𑀅𑀦𑁆𑀯𑀤𑁂𑀯 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀧𑀤𑁂𑀲𑁂 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀲𑀺, 𑀢𑀢𑁆𑀣 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬𑁇 𑀬𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀧𑀝𑀺𑀭𑀸𑀚𑀸𑀦𑁄, 𑀢𑁂 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀏𑀳𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑀁 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑀼𑀲𑀸𑀲, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸’𑀢𑀺𑁇 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀧𑀸𑀡𑁄 𑀦 𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀁, 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀺𑀘𑁆𑀙𑀸 𑀦 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀸, 𑀫𑀼𑀲𑀸 𑀦 𑀪𑀡𑀺𑀢𑀩𑁆𑀩𑀸, 𑀫𑀚𑁆𑀚𑀁 𑀦 𑀧𑀸𑀢𑀩𑁆𑀩𑀁, 𑀬𑀣𑀸𑀪𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀼𑀜𑁆𑀚𑀣𑀸’𑀢𑀺𑁇 𑀬𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀧𑀝𑀺𑀭𑀸𑀚𑀸𑀦𑁄, 𑀢𑁂 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

(245.)
