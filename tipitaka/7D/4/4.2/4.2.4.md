# 𑁪𑁇𑁨𑁇𑁪 𑀫𑀡𑀺𑀭𑀢𑀦

### 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢

### 𑀫𑀡𑀺𑀭𑀢𑀦

(248.)

545\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀫𑀡𑀺 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑁄 𑀲𑀼𑀪𑁄 𑀚𑀸𑀢𑀺𑀫𑀸 𑀅𑀝𑁆𑀞𑀁𑀲𑁄 𑀲𑀼𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀘𑁆𑀙𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀅𑀦𑀸𑀯𑀺𑀮𑁄 𑀲𑀩𑁆𑀩𑀸𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀢𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀫𑀡𑀺𑀭𑀢𑀦𑀲𑁆𑀲 𑀆𑀪𑀸 𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀁 𑀨𑀼𑀝𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀫𑁂𑀯 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀁 𑀲𑁂𑀦𑀁 𑀲𑀦𑁆𑀦𑀬𑁆𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀡𑀺𑀁 𑀥𑀚𑀕𑁆𑀕𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀢𑀺𑀫𑀺𑀲𑀸𑀬 𑀧𑀸𑀬𑀸𑀲𑀺𑁇 𑀬𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀲𑀫𑀦𑁆𑀢𑀸 𑀕𑀸𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀢𑁂 𑀢𑁂𑀦𑁄𑀪𑀸𑀲𑁂𑀦 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀬𑁄𑀚𑁂𑀲𑀼𑀁 𑀤𑀺𑀯𑀸𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑀸𑁇 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇
