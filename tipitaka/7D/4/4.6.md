# 𑁪𑁇𑁬 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀦𑀕𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀤𑀺

* 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀦𑀕𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀤𑀺

(263.)

579\. 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀦𑀕𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀓𑀼𑀲𑀸𑀯𑀢𑀻𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀧𑀸𑀲𑀸𑀤𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀥𑀫𑁆𑀫𑀧𑀸𑀲𑀸𑀤𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀳𑀸𑀯𑀺𑀬𑀽𑀳𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸𑀦𑀺 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸𑀦𑀺 𑀤𑀦𑁆𑀢𑀫𑀬𑀸𑀦𑀺 𑀲𑀸𑀭𑀫𑀬𑀸𑀦𑀺 𑀕𑁄𑀦𑀓𑀢𑁆𑀣𑀢𑀸𑀦𑀺 𑀧𑀝𑀺𑀓𑀢𑁆𑀣𑀢𑀸𑀦𑀺 𑀧𑀝𑀮𑀺𑀓𑀢𑁆𑀣𑀢𑀸𑀦𑀺 𑀓𑀤𑀮𑀺𑀫𑀺𑀕𑀧𑀯𑀭𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺 𑀲𑁅𑀢𑁆𑀢𑀭𑀘𑁆𑀙𑀤𑀸𑀦𑀺 𑀉𑀪𑀢𑁄𑀮𑁄𑀳𑀺𑀢𑀓𑀽𑀧𑀥𑀸𑀦𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑀦𑀺 𑀲𑁄𑀯𑀡𑁆𑀡𑀥𑀚𑀸𑀦𑀺 𑀳𑁂𑀫𑀚𑀸𑀮𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀺 𑀉𑀧𑁄𑀲𑀣𑀦𑀸𑀕𑀭𑀸𑀚𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀅𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑀦𑀺 𑀲𑁄𑀯𑀡𑁆𑀡𑀥𑀚𑀸𑀦𑀺 𑀳𑁂𑀫𑀚𑀸𑀮𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀺 𑀯𑀮𑀸𑀳𑀓𑀅𑀲𑁆𑀲𑀭𑀸𑀚𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀭𑀣𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀯𑀸𑀭𑀸𑀦𑀺 𑀩𑁆𑀬𑀕𑁆𑀖𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀯𑀸𑀭𑀸𑀦𑀺 𑀤𑀻𑀧𑀺𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀯𑀸𑀭𑀸𑀦𑀺 𑀧𑀡𑁆𑀟𑀼𑀓𑀫𑁆𑀩𑀮𑀧𑀭𑀺𑀯𑀸𑀭𑀸𑀦𑀺 𑀲𑁄𑀯𑀡𑁆𑀡𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑀦𑀺 𑀲𑁄𑀯𑀡𑁆𑀡𑀥𑀚𑀸𑀦𑀺 𑀳𑁂𑀫𑀚𑀸𑀮𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀺 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀭𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀫𑀡𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀫𑀡𑀺𑀭𑀢𑀦𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀇𑀢𑁆𑀣𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀼𑀪𑀤𑁆𑀤𑀸𑀤𑁂𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀕𑀳𑀧𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀥𑁂𑀦𑀼𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀤𑀼𑀳𑀲𑀦𑁆𑀤𑀦𑀸𑀦𑀺 𑀓𑀁𑀲𑀽𑀧𑀥𑀸𑀭𑀡𑀸𑀦𑀺; 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀯𑀢𑁆𑀣𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀔𑁄𑀫𑀲𑀼𑀔𑀼𑀫𑀸𑀦𑀁 𑀓𑀧𑁆𑀧𑀸𑀲𑀺𑀓𑀲𑀼𑀔𑀼𑀫𑀸𑀦𑀁 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀲𑀼𑀔𑀼𑀫𑀸𑀦𑀁 𑀓𑀫𑁆𑀩𑀮𑀲𑀼𑀔𑀼𑀫𑀸𑀦𑀁; 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀣𑀸𑀮𑀺𑀧𑀸𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑀁 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑁄 𑀅𑀪𑀺𑀳𑀭𑀺𑀬𑀺𑀢𑁆𑀣𑁇

(264.)

580\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀫𑀸𑀦𑀺 𑀔𑁄 𑀫𑁂 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀬𑀦𑁆𑀦𑀽𑀦 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀇𑀫𑀸𑀦𑀺 𑀔𑁄 𑀫𑁂, 𑀲𑀫𑁆𑀫 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦, 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦, 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀽’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀁 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀓𑀺𑀁 𑀲𑀓𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀫𑀁𑀲𑀼𑁇
