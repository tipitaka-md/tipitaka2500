# 𑁪𑁇𑁧 𑀓𑀼𑀲𑀸𑀯𑀢𑀻𑀭𑀸𑀚𑀥𑀸𑀦𑀻

### 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀼𑀲𑀸𑀯𑀢𑀻𑀭𑀸𑀚𑀥𑀸𑀦𑀻

537\. 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀦𑀸𑀫 𑀅𑀳𑁄𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑁄 𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀚𑀦𑀧𑀤𑀢𑁆𑀣𑀸𑀯𑀭𑀺𑀬𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀭𑀜𑁆𑀜𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀅𑀬𑀁 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀓𑀼𑀲𑀸𑀯𑀢𑀻 𑀦𑀸𑀫 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂𑀦 𑀘 𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀘 𑀤𑁆𑀯𑀸𑀤𑀲𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀆𑀬𑀸𑀫𑁂𑀦, 𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀘 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀘 𑀲𑀢𑁆𑀢𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦𑁇 𑀓𑀼𑀲𑀸𑀯𑀢𑀻, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀨𑀻𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀍𑀅𑀓𑀫𑀦𑁆𑀤𑀸 𑀦𑀸𑀫 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀨𑀻𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀬𑀓𑁆𑀔𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀼𑀲𑀸𑀯𑀢𑀻 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀨𑀻𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘𑁇 𑀓𑀼𑀲𑀸𑀯𑀢𑀻, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀤𑀲𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺 𑀅𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁄𑀲𑀺 𑀤𑀺𑀯𑀸 𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀳𑀢𑁆𑀣𑀺𑀲𑀤𑁆𑀤𑁂𑀦 𑀅𑀲𑁆𑀲𑀲𑀤𑁆𑀤𑁂𑀦 𑀭𑀣𑀲𑀤𑁆𑀤𑁂𑀦 𑀪𑁂𑀭𑀺𑀲𑀤𑁆𑀤𑁂𑀦 𑀫𑀼𑀤𑀺𑀗𑁆𑀕𑀲𑀤𑁆𑀤𑁂𑀦 𑀯𑀻𑀡𑀸𑀲𑀤𑁆𑀤𑁂𑀦 𑀕𑀻𑀢𑀲𑀤𑁆𑀤𑁂𑀦 𑀲𑀗𑁆𑀔𑀲𑀤𑁆𑀤𑁂𑀦 𑀲𑀫𑁆𑀫𑀲𑀤𑁆𑀤𑁂𑀦 𑀧𑀸𑀡𑀺𑀢𑀸𑀍𑀅𑀲𑀤𑁆𑀤𑁂𑀦 ‘𑀅𑀲𑁆𑀦𑀸𑀣 𑀧𑀺𑀯𑀣 𑀔𑀸𑀤𑀣𑀸’𑀢𑀺 𑀤𑀲𑀫𑁂𑀦 𑀲𑀤𑁆𑀤𑁂𑀦𑁇

538\. 𑀓𑀼𑀲𑀸𑀯𑀢𑀻, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀲𑀢𑁆𑀢𑀳𑀺 𑀧𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀏𑀓𑁄 𑀧𑀸𑀓𑀸𑀭𑁄 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑁄, 𑀏𑀓𑁄 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑁄, 𑀏𑀓𑁄 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑁄, 𑀏𑀓𑁄 𑀨𑀮𑀺𑀓𑀫𑀬𑁄, 𑀏𑀓𑁄 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑁄, 𑀏𑀓𑁄 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑁄, 𑀏𑀓𑁄 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑁄𑁇 𑀓𑀼𑀲𑀸𑀯𑀢𑀺𑀬𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀡𑁆𑀡𑀸𑀦𑀁 𑀤𑁆𑀯𑀸𑀭𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀏𑀓𑀁 𑀤𑁆𑀯𑀸𑀭𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀁, 𑀏𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀁, 𑀏𑀓𑀁 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀁, 𑀏𑀓𑀁 𑀨𑀮𑀺𑀓𑀫𑀬𑀁𑁇 𑀏𑀓𑁂𑀓𑀲𑁆𑀫𑀺𑀁 𑀤𑁆𑀯𑀸𑀭𑁂 𑀲𑀢𑁆𑀢 𑀲𑀢𑁆𑀢 𑀏𑀲𑀺𑀓𑀸 𑀦𑀺𑀔𑀸𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀢𑀺𑀧𑁄𑀭𑀺𑀲𑀗𑁆𑀕𑀸 𑀢𑀺𑀧𑁄𑀭𑀺𑀲𑀦𑀺𑀔𑀸𑀢𑀸 𑀤𑁆𑀯𑀸𑀤𑀲𑀧𑁄𑀭𑀺𑀲𑀸 𑀉𑀩𑁆𑀩𑁂𑀥𑁂𑀦𑁇 𑀏𑀓𑀸 𑀏𑀲𑀺𑀓𑀸 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀨𑀮𑀺𑀓𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑀸𑁇

539\. 𑀓𑀼𑀲𑀸𑀯𑀢𑀻, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀥𑀸𑀦𑀻 𑀲𑀢𑁆𑀢𑀳𑀺 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀳𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀏𑀓𑀸 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀺 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀨𑀮𑀺𑀓𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑀸𑁇 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀨𑀮𑀺𑀓𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀨𑀮𑀺𑀓𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀨𑀮𑀺𑀓𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀢𑀸𑀲𑀁 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀸𑀢𑁂𑀭𑀺𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀔𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀢𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀢𑀸𑀍𑀇𑀢𑀲𑁆𑀲 𑀲𑀼𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀔𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀸𑀲𑀁 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀸𑀢𑁂𑀭𑀺𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀔𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘𑁇 𑀬𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑀼𑀲𑀸𑀯𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀥𑀼𑀢𑁆𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑁄𑀡𑁆𑀟𑀸 𑀧𑀺𑀧𑀸𑀲𑀸, 𑀢𑁂 𑀢𑀸𑀲𑀁 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀸𑀢𑁂𑀭𑀺𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁂𑀦 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀲𑀼𑀁𑁇
