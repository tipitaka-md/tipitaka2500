# 𑁪𑁇𑁪 𑀥𑀫𑁆𑀫𑀧𑀸𑀲𑀸𑀤𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻

* 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀥𑀫𑁆𑀫𑀧𑀸𑀲𑀸𑀤𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻

(253.)

555\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀲𑀼 𑀢𑀸𑀮𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀲𑀼 𑀥𑀦𑀼𑀲𑀢𑁂 𑀥𑀦𑀼𑀲𑀢𑁂 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑁄 𑀫𑀸𑀧𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇

556\. 𑀫𑀸𑀧𑁂𑀲𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀸𑀲𑀼 𑀢𑀸𑀮𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀲𑀼 𑀥𑀦𑀼𑀲𑀢𑁂 𑀥𑀦𑀼𑀲𑀢𑁂 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑁄𑁇 𑀢𑀸 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀡𑁆𑀡𑀸𑀦𑀁 𑀇𑀝𑁆𑀞𑀓𑀸𑀳𑀺 𑀘𑀺𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁—  𑀏𑀓𑀸 𑀇𑀝𑁆𑀞𑀓𑀸 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀨𑀮𑀺𑀓𑀫𑀬𑀸𑁇

557\. 𑀢𑀸𑀲𑀼 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑁄𑀧𑀸𑀦𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀡𑁆𑀡𑀸𑀦𑀁, 𑀏𑀓𑀁 𑀲𑁄𑀧𑀸𑀦𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀁 𑀏𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀁 𑀏𑀓𑀁 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀁 𑀏𑀓𑀁 𑀨𑀮𑀺𑀓𑀫𑀬𑀁𑁇 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀲𑁆𑀲 𑀲𑁄𑀧𑀸𑀦𑀲𑁆𑀲 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀲𑁆𑀲 𑀲𑁄𑀧𑀸𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀲𑁆𑀲 𑀲𑁄𑀧𑀸𑀦𑀲𑁆𑀲 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀨𑀮𑀺𑀓𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇 𑀨𑀮𑀺𑀓𑀫𑀬𑀲𑁆𑀲 𑀲𑁄𑀧𑀸𑀦𑀲𑁆𑀲 𑀨𑀮𑀺𑀓𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇 𑀢𑀸 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑁄 𑀤𑁆𑀯𑀻𑀳𑀺 𑀯𑁂𑀤𑀺𑀓𑀸𑀳𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀏𑀓𑀸 𑀯𑁂𑀤𑀺𑀓𑀸 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸𑁇 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸𑀬 𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸𑀬 𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇

558\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀲𑀼 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀼 𑀏𑀯𑀭𑀽𑀧𑀁 𑀫𑀸𑀮𑀁 𑀭𑁄𑀧𑀸𑀧𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑁆𑀧𑀮𑀁 𑀧𑀤𑀼𑀫𑀁 𑀓𑀼𑀫𑀼𑀤𑀁 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀁 𑀲𑀩𑁆𑀩𑁄𑀢𑀼𑀓𑀁 𑀲𑀩𑁆𑀩𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀸𑀯𑀝𑀦𑁆’𑀢𑀺𑁇 𑀭𑁄𑀧𑀸𑀧𑁂𑀲𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀸𑀲𑀼 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀼 𑀏𑀯𑀭𑀽𑀧𑀁 𑀫𑀸𑀮𑀁 𑀉𑀧𑁆𑀧𑀮𑀁 𑀧𑀤𑀼𑀫𑀁 𑀓𑀼𑀫𑀼𑀤𑀁 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀁 𑀲𑀩𑁆𑀩𑁄𑀢𑀼𑀓𑀁 𑀲𑀩𑁆𑀩𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀸𑀯𑀝𑀁𑁇

(254.)

559\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀲𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀦𑀁 𑀢𑀻𑀭𑁂 𑀦𑁆𑀳𑀸𑀧𑀓𑁂 𑀧𑀼𑀭𑀺𑀲𑁂 𑀞𑀧𑁂𑀬𑁆𑀬𑀁, 𑀬𑁂 𑀆𑀕𑀢𑀸𑀕𑀢𑀁 𑀚𑀦𑀁 𑀦𑁆𑀳𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀞𑀧𑁂𑀲𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀸𑀲𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀦𑀁 𑀢𑀻𑀭𑁂 𑀦𑁆𑀳𑀸𑀧𑀓𑁂 𑀧𑀼𑀭𑀺𑀲𑁂, 𑀬𑁂 𑀆𑀕𑀢𑀸𑀕𑀢𑀁 𑀚𑀦𑀁 𑀦𑁆𑀳𑀸𑀧𑁂𑀲𑀼𑀁𑁇

560\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀲𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀦𑀁 𑀢𑀻𑀭𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀸𑀦𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀬𑁆𑀬𑀁—  𑀅𑀦𑁆𑀦𑀁 𑀅𑀦𑁆𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀧𑀸𑀦𑀁 𑀧𑀸𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀯𑀢𑁆𑀣𑀁 𑀯𑀢𑁆𑀣𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀬𑀸𑀦𑀁 𑀬𑀸𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀲𑀬𑀦𑀁 𑀲𑀬𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀇𑀢𑁆𑀣𑀺𑀁 𑀇𑀢𑁆𑀣𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀳𑀺𑀭𑀜𑁆𑀜𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀸𑀲𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀦𑀁 𑀢𑀻𑀭𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀸𑀦𑀁—  𑀅𑀦𑁆𑀦𑀁 𑀅𑀦𑁆𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀧𑀸𑀦𑀁 𑀧𑀸𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀯𑀢𑁆𑀣𑀁 𑀯𑀢𑁆𑀣𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀬𑀸𑀦𑀁 𑀬𑀸𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀲𑀬𑀦𑀁 𑀲𑀬𑀦𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀇𑀢𑁆𑀣𑀺𑀁 𑀇𑀢𑁆𑀣𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀳𑀺𑀭𑀜𑁆𑀜𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲𑁇

(255.)

561\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀧𑀳𑀽𑀢𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀆𑀤𑀸𑀬 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀇𑀤𑀁, 𑀤𑁂𑀯, 𑀧𑀳𑀽𑀢𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀤𑁂𑀯𑀜𑁆𑀜𑁂𑀯 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀆𑀪𑀢𑀁, 𑀢𑀁 𑀤𑁂𑀯𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀢𑀽’𑀢𑀺𑁇 ‘𑀅𑀮𑀁, 𑀪𑁄, 𑀫𑀫𑀧𑀺𑀤𑀁 𑀧𑀳𑀽𑀢𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀥𑀫𑁆𑀫𑀺𑀓𑁂𑀦 𑀩𑀮𑀺𑀦𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀢𑀁, 𑀢𑀜𑁆𑀘 𑀯𑁄 𑀳𑁄𑀢𑀼, 𑀇𑀢𑁄 𑀘 𑀪𑀺𑀬𑁆𑀬𑁄 𑀳𑀭𑀣𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀭𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀓𑁆𑀓𑀫𑁆𑀫 𑀏𑀯𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁—  ‘𑀦 𑀔𑁄 𑀏𑀢𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁, 𑀬𑀁 𑀫𑀬𑀁 𑀇𑀫𑀸𑀦𑀺 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀓𑀸𑀦𑀺 𑀖𑀭𑀸𑀦𑀺 𑀧𑀝𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀫𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂, 𑀤𑁂𑀯, 𑀫𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

(256.)

562\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀫𑁆𑀫 𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫, 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀫𑀸𑀧𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀫 𑀧𑀸𑀲𑀸𑀤𑀦𑁆’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂, 𑀤𑁂𑀯, 𑀫𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀫 𑀧𑀸𑀲𑀸𑀤𑀦𑁆’𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

571\. 𑀥𑀫𑁆𑀫𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀤𑁆𑀯𑀻𑀳𑀺 𑀯𑁂𑀤𑀺𑀓𑀸𑀳𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁄𑀲𑀺—  𑀏𑀓𑀸 𑀯𑁂𑀤𑀺𑀓𑀸 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸𑁇 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸𑀬 𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸𑀬 𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸 𑀣𑀫𑁆𑀪𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸 𑀲𑀽𑀘𑀺𑀬𑁄 𑀘 𑀉𑀡𑁆𑀳𑀻𑀲𑀜𑁆𑀘𑁇

572\. 𑀥𑀫𑁆𑀫𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀲𑀢𑁆𑀢𑀳𑀺 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀳𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁄𑀲𑀺—  𑀏𑀓𑀸 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀺 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀨𑀮𑀺𑀓𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀸, 𑀏𑀓𑀸 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑀸𑁇 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺 𑀨𑀮𑀺𑀓𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀨𑀮𑀺𑀓𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀨𑀮𑀺𑀓𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑀲𑁆𑀲 𑀢𑀸𑀮𑀲𑁆𑀲 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑁄 𑀔𑀦𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀫𑀬𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀨𑀮𑀸𑀦𑀺 𑀘𑁇 𑀢𑀸𑀲𑀁 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀸𑀢𑁂𑀭𑀺𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀔𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀢𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀢𑀸𑀍𑀇𑀢𑀲𑁆𑀲 𑀲𑀼𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀔𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀸𑀲𑀁 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀸𑀢𑁂𑀭𑀺𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀔𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘𑁇 𑀬𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑀼𑀲𑀸𑀯𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑀸 𑀥𑀼𑀢𑁆𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑁄𑀡𑁆𑀟𑀸 𑀧𑀺𑀧𑀸𑀲𑀸, 𑀢𑁂 𑀢𑀸𑀲𑀁 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀸𑀢𑁂𑀭𑀺𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁂𑀦 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀲𑀼𑀁𑁇

573\. 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀥𑀫𑁆𑀫𑁂 𑀧𑀸𑀲𑀸𑀤𑁂 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀬 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑀸 𑀭𑀸𑀚𑀸 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 ‘𑀬𑁂 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑀡𑁂𑀲𑀼 𑀯𑀸 𑀲𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀫𑁆𑀫𑀢𑀸’, 𑀢𑁂 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑁂𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀧𑀸𑀲𑀸𑀤𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑁇
