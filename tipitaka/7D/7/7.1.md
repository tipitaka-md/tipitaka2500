# 𑁭𑁇𑁧 𑀤𑁂𑀯𑀢𑀸𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀳𑀸𑀲𑀫𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢

(333.)

755. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀲𑀲𑀼 𑀮𑁄𑀓𑀥𑀸𑀢𑀽𑀲𑀼 𑀤𑁂𑀯𑀢𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀳𑁂𑀲𑀼𑀁 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀏𑀢𑀧𑁆𑀧𑀭𑀫𑀸𑀬𑁂𑀯 𑀤𑁂𑀯𑀢𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀬𑁆𑀳𑀁 𑀏𑀢𑀭𑀳𑀺𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀏𑀢𑀧𑁆𑀧𑀭𑀫𑀸𑀬𑁂𑀯 𑀤𑁂𑀯𑀢𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀬𑁆𑀳𑀁 𑀏𑀢𑀭𑀳𑀺𑁇 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀓𑀸𑀬𑀸𑀦𑀁 𑀦𑀸𑀫𑀸𑀦𑀺; 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀓𑀸𑀬𑀸𑀦𑀁 𑀦𑀸𑀫𑀸𑀦𑀺; 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀓𑀸𑀬𑀸𑀦𑀁 𑀦𑀸𑀫𑀸𑀦𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇

(334.)

756. 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

757. _“𑀲𑀺𑀮𑁄𑀓𑀫𑀦𑀼𑀓𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀢𑁆𑀣 𑀪𑀼𑀫𑁆𑀫𑀸 𑀢𑀤𑀲𑁆𑀲𑀺𑀢𑀸;_  
_𑀬𑁂 𑀲𑀺𑀢𑀸 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑀁,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸𑁇_  


758. _𑀧𑀼𑀣𑀽𑀲𑀻𑀳𑀸𑀯 𑀲𑀮𑁆𑀮𑀻𑀦𑀸,_  
_𑀮𑁄𑀫𑀳𑀁𑀲𑀸𑀪𑀺𑀲𑀫𑁆𑀪𑀼𑀦𑁄;_  
_𑀑𑀤𑀸𑀢𑀫𑀦𑀲𑀸 𑀲𑀼𑀤𑁆𑀥𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀸𑁇_  


759. _𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀜𑁆𑀘𑀲𑀢𑁂 𑀜𑀢𑁆𑀯𑀸,_  
_𑀯𑀦𑁂 𑀓𑀸𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀯𑁂;_  
_𑀢𑀢𑁄 𑀆𑀫𑀦𑁆𑀢𑀬𑀻 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑀸𑀯𑀓𑁂 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑁂𑁇_  


760. _‘𑀤𑁂𑀯𑀓𑀸𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸,_  
_𑀢𑁂 𑀯𑀺𑀚𑀸𑀦𑀸𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄’;_  
_𑀢𑁂 𑀘 𑀆𑀢𑀧𑁆𑀧𑀫𑀓𑀭𑀼𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


761. _𑀢𑁂𑀲𑀁 𑀧𑀸𑀢𑀼𑀭𑀳𑀼 𑀜𑀸𑀡𑀁,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀅𑀧𑁆𑀧𑁂𑀓𑁂 𑀲𑀢𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁,_  
_𑀲𑀳𑀲𑁆𑀲𑀁 𑀅𑀣 𑀲𑀢𑁆𑀢𑀭𑀺𑀁𑁇_  


762. _𑀲𑀢𑀁 𑀏𑀓𑁂 𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀫𑀤𑁆𑀤𑀲𑀼𑀁;_  
_𑀅𑀧𑁆𑀧𑁂𑀓𑁂𑀦𑀦𑁆𑀢𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁,_  
_𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸 𑀨𑀼𑀝𑀸 𑀅𑀳𑀼𑀁𑁇_  


763. _𑀢𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀯𑀢𑁆𑀣𑀺𑀢𑁆𑀯𑀸𑀦 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀢𑀢𑁄 𑀆𑀫𑀦𑁆𑀢𑀬𑀻 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑀸𑀯𑀓𑁂 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑁂𑁇_  


764. _‘𑀤𑁂𑀯𑀓𑀸𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸,_  
_𑀢𑁂 𑀯𑀺𑀚𑀸𑀦𑀸𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄;_  
_𑀬𑁂 𑀯𑁄𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀕𑀺𑀭𑀸𑀳𑀺 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑁄’𑁇_  


(335.)

765. _𑀲𑀢𑁆𑀢𑀲𑀳𑀲𑁆𑀲𑀸 𑀢𑁂 𑀬𑀓𑁆𑀔𑀸,_  
_𑀪𑀼𑀫𑁆𑀫𑀸 𑀓𑀸𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀯𑀸;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


766. _𑀙𑀲𑀳𑀲𑁆𑀲𑀸 𑀳𑁂𑀫𑀯𑀢𑀸,_  
_𑀬𑀓𑁆𑀔𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


767. _𑀲𑀸𑀢𑀸𑀕𑀺𑀭𑀸 𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸,_  
_𑀬𑀓𑁆𑀔𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


768. _𑀇𑀘𑁆𑀘𑁂𑀢𑁂 𑀲𑁄𑀍𑀅𑀲𑀲𑀳𑀲𑁆𑀲𑀸,_  
_𑀬𑀓𑁆𑀔𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


769. _𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀘𑀲𑀢𑀸,_  
_𑀬𑀓𑁆𑀔𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


770. _𑀓𑀼𑀫𑁆𑀪𑀻𑀭𑁄 𑀭𑀸𑀚𑀕𑀳𑀺𑀓𑁄,_  
_𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁;_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀬𑀓𑁆𑀔𑀸𑀦𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺;_  
_𑀓𑀼𑀫𑁆𑀪𑀻𑀭𑁄 𑀭𑀸𑀚𑀕𑀳𑀺𑀓𑁄,_  
_𑀲𑁄𑀧𑀸𑀕𑀸 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


(336.)

771. _𑀧𑀼𑀭𑀺𑀫𑀜𑁆𑀘 𑀤𑀺𑀲𑀁 𑀭𑀸𑀚𑀸,_  
_𑀥𑀢𑀭𑀝𑁆𑀞𑁄 𑀧𑀲𑀸𑀲𑀢𑀺;_  
_𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀲𑁄𑁇_  


772. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


773. _𑀤𑀓𑁆𑀔𑀺𑀡𑀜𑁆𑀘 𑀤𑀺𑀲𑀁 𑀭𑀸𑀚𑀸,_  
_𑀯𑀺𑀭𑀽𑀍𑀳𑁄 𑀢𑀁 𑀧𑀲𑀸𑀲𑀢𑀺;_  
_𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀲𑁄𑁇_  


774. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


775. _𑀧𑀘𑁆𑀙𑀺𑀫𑀜𑁆𑀘 𑀤𑀺𑀲𑀁 𑀭𑀸𑀚𑀸,_  
_𑀯𑀺𑀭𑀽𑀧𑀓𑁆𑀔𑁄 𑀧𑀲𑀸𑀲𑀢𑀺;_  
_𑀦𑀸𑀕𑀸𑀦𑀜𑁆𑀘 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀲𑁄𑁇_  


776. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


777. _𑀉𑀢𑁆𑀢𑀭𑀜𑁆𑀘 𑀤𑀺𑀲𑀁 𑀭𑀸𑀚𑀸,_  
_𑀓𑀼𑀯𑁂𑀭𑁄 𑀢𑀁 𑀧𑀲𑀸𑀲𑀢𑀺;_  
_𑀬𑀓𑁆𑀔𑀸𑀦𑀜𑁆𑀘 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀲𑁄𑁇_  


778. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


779. _𑀧𑀼𑀭𑀺𑀫𑀁 𑀤𑀺𑀲𑀁 𑀥𑀢𑀭𑀝𑁆𑀞𑁄,_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀯𑀺𑀭𑀽𑀍𑀳𑀓𑁄;_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑁂𑀦 𑀯𑀺𑀭𑀽𑀧𑀓𑁆𑀔𑁄,_  
_𑀓𑀼𑀯𑁂𑀭𑁄 𑀉𑀢𑁆𑀢𑀭𑀁 𑀤𑀺𑀲𑀁𑁇_  


780. _𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑀸;_  
_𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀫𑀸𑀦𑀸 𑀅𑀝𑁆𑀞𑀁𑀲𑀼,_  
_𑀯𑀦𑁂 𑀓𑀸𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀯𑁂𑁇_  


(337.)

781. _𑀢𑁂𑀲𑀁 𑀫𑀸𑀬𑀸𑀯𑀺𑀦𑁄 𑀤𑀸𑀲𑀸,_  
_𑀆𑀕𑀼𑀁 𑀯𑀜𑁆𑀘𑀦𑀺𑀓𑀸 𑀲𑀞𑀸;_  
_𑀫𑀸𑀬𑀸 𑀓𑀼𑀝𑁂𑀡𑁆𑀟𑀼 𑀯𑀺𑀝𑁂𑀡𑁆𑀟𑀼,_  
_𑀯𑀺𑀝𑀼𑀘𑁆𑀘 𑀯𑀺𑀝𑀼𑀝𑁄 𑀲𑀳𑁇_  


782. _𑀘𑀦𑁆𑀤𑀦𑁄 𑀓𑀸𑀫𑀲𑁂𑀝𑁆𑀞𑁄 𑀘,_  
_𑀓𑀺𑀦𑁆𑀦𑀺𑀖𑀡𑁆𑀟𑀼 𑀦𑀺𑀖𑀡𑁆𑀟𑀼 𑀘;_  
_𑀧𑀦𑀸𑀤𑁄 𑀑𑀧𑀫𑀜𑁆𑀜𑁄 𑀘,_  
_𑀤𑁂𑀯𑀲𑀽𑀢𑁄 𑀘 𑀫𑀸𑀢𑀮𑀺𑁇_  


783. _𑀘𑀺𑀢𑁆𑀢𑀲𑁂𑀦𑁄 𑀘 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀦𑀍𑀑𑀭𑀸𑀚𑀸 𑀚𑀦𑁂𑀲𑀪𑁄;_  
_𑀆𑀕𑀸 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀘𑁂𑀯,_  
_𑀢𑀺𑀫𑁆𑀩𑀭𑀽 𑀲𑀽𑀭𑀺𑀬𑀯𑀘𑁆𑀙𑀲𑀸𑁇_  


784. _𑀏𑀢𑁂 𑀘𑀜𑁆𑀜𑁂 𑀘 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸 𑀲𑀳 𑀭𑀸𑀚𑀼𑀪𑀺;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


(338.)

785. _𑀅𑀣𑀸𑀕𑀼𑀁 𑀦𑀸𑀕𑀲𑀸 𑀦𑀸𑀕𑀸,_  
_𑀯𑁂𑀲𑀸𑀮𑀸 𑀲𑀳𑀢𑀘𑁆𑀙𑀓𑀸;_  
_𑀓𑀫𑁆𑀩𑀮𑀲𑁆𑀲𑀢𑀭𑀸 𑀆𑀕𑀼𑀁,_  
_𑀧𑀸𑀬𑀸𑀕𑀸 𑀲𑀳 𑀜𑀸𑀢𑀺𑀪𑀺𑁇_  


786. _𑀬𑀸𑀫𑀼𑀦𑀸 𑀥𑀢𑀭𑀝𑁆𑀞𑀸 𑀘,_  
_𑀆𑀕𑀽 𑀦𑀸𑀕𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀏𑀭𑀸𑀯𑀡𑁄 𑀫𑀳𑀸𑀦𑀸𑀕𑁄,_  
_𑀲𑁄𑀧𑀸𑀕𑀸 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


787. _𑀬𑁂 𑀦𑀸𑀕𑀭𑀸𑀚𑁂 𑀲𑀳𑀲𑀸 𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀸 𑀤𑀺𑀚𑀸 𑀧𑀓𑁆𑀔𑀺 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀘𑀓𑁆𑀔𑀽;_  
_𑀯𑁂𑀳𑀸𑀬𑀲𑀸 𑀢𑁂 𑀯𑀦𑀫𑀚𑁆𑀛𑀧𑀢𑁆𑀢𑀸,_  
_𑀘𑀺𑀢𑁆𑀭𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀇𑀢𑀺 𑀢𑁂𑀲 𑀦𑀸𑀫𑀁𑁇_  


788. _𑀅𑀪𑀬𑀁 𑀢𑀤𑀸 𑀦𑀸𑀕𑀭𑀸𑀚𑀸𑀦𑀫𑀸𑀲𑀺,_  
_𑀲𑀼𑀧𑀡𑁆𑀡𑀢𑁄 𑀔𑁂𑀫𑀫𑀓𑀸𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑁄;_  
_𑀲𑀡𑁆𑀳𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀉𑀧𑀯𑁆𑀳𑀬𑀦𑁆𑀢𑀸,_  
_𑀦𑀸𑀕𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀲𑀭𑀡𑀫𑀓𑀁𑀲𑀼 𑀩𑀼𑀤𑁆𑀥𑀁𑁇_  


(339.)

789. _𑀚𑀺𑀢𑀸 𑀯𑀚𑀺𑀭𑀳𑀢𑁆𑀣𑁂𑀦,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀲𑀼𑀭𑀸𑀲𑀺𑀢𑀸;_  
_𑀪𑀸𑀢𑀭𑁄 𑀯𑀸𑀲𑀯𑀲𑁆𑀲𑁂𑀢𑁂,_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


790. _𑀓𑀸𑀮𑀓𑀜𑁆𑀘𑀸 𑀫𑀳𑀸𑀪𑀺𑀲𑁆𑀫𑀸,_  
_𑀅𑀲𑀼𑀭𑀸 𑀤𑀸𑀦𑀯𑁂𑀖𑀲𑀸;_  
_𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀲𑀼𑀘𑀺𑀢𑁆𑀢𑀺 𑀘,_  
_𑀧𑀳𑀸𑀭𑀸𑀤𑁄 𑀦𑀫𑀼𑀘𑀻 𑀲𑀳𑁇_  


791. _𑀲𑀢𑀜𑁆𑀘 𑀩𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀲𑀩𑁆𑀩𑁂 𑀯𑁂𑀭𑁄𑀘𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑀦𑁆𑀦𑀬𑁆𑀳𑀺𑀢𑁆𑀯𑀸 𑀩𑀮𑀺𑀲𑁂𑀦𑀁,_  
_𑀭𑀸𑀳𑀼𑀪𑀤𑁆𑀤𑀫𑀼𑀧𑀸𑀕𑀫𑀼𑀁;_  
_‘𑀲𑀫𑀬𑁄 𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁’𑁇_  


(340.)

792. _𑀆𑀧𑁄 𑀘 𑀤𑁂𑀯𑀸 𑀧𑀣𑀯𑀻,_  
_𑀢𑁂𑀚𑁄 𑀯𑀸𑀬𑁄 𑀢𑀤𑀸𑀕𑀫𑀼𑀁;_  
_𑀯𑀭𑀼𑀡𑀸 𑀯𑀸𑀭𑀡𑀸 𑀤𑁂𑀯𑀸,_  
_𑀲𑁄𑀫𑁄 𑀘 𑀬𑀲𑀲𑀸 𑀲𑀳𑁇_  


793. _𑀫𑁂𑀢𑁆𑀢𑀸 𑀓𑀭𑀼𑀡𑀸 𑀓𑀸𑀬𑀺𑀓𑀸,_  
_𑀆𑀕𑀼𑀁 𑀤𑁂𑀯𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀤𑀲𑁂𑀢𑁂 𑀤𑀲𑀥𑀸 𑀓𑀸𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄𑁇_  


794. _𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


795. _𑀯𑁂𑀡𑁆𑀟𑀼𑀤𑁂𑀯𑀸 𑀲𑀳𑀮𑀺 𑀘,_  
_𑀅𑀲𑀫𑀸 𑀘 𑀤𑀼𑀯𑁂 𑀬𑀫𑀸;_  
_𑀘𑀦𑁆𑀤𑀲𑁆𑀲𑀽𑀧𑀦𑀺𑀲𑀸 𑀤𑁂𑀯𑀸,_  
_𑀘𑀦𑁆𑀤𑀫𑀸𑀕𑀼𑀁 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸𑁇_  


796. _𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲𑀽𑀧𑀦𑀺𑀲𑀸 𑀤𑁂𑀯𑀸,_  
_𑀲𑀽𑀭𑀺𑀬𑀫𑀸𑀕𑀼𑀁 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸;_  
_𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸,_  
_𑀆𑀕𑀼𑀁 𑀫𑀦𑁆𑀤𑀯𑀮𑀸𑀳𑀓𑀸𑁇_  


797. _𑀯𑀲𑀽𑀦𑀁 𑀯𑀸𑀲𑀯𑁄 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀲𑀓𑁆𑀓𑁄𑀧𑀸𑀕𑀸 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀤𑀲𑁂𑀢𑁂 𑀤𑀲𑀥𑀸 𑀓𑀸𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄𑁇_  


798. _𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


799. _𑀅𑀣𑀸𑀕𑀼𑀁 𑀲𑀳𑀪𑀽 𑀤𑁂𑀯𑀸,_  
_𑀚𑀮𑀫𑀕𑁆𑀕𑀺 𑀲𑀺𑀔𑀸𑀭𑀺𑀯;_  
_𑀅𑀭𑀺𑀝𑁆𑀞𑀓𑀸 𑀘 𑀭𑁄𑀚𑀸 𑀘,_  
_𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀦𑀺𑀪𑀸𑀲𑀺𑀦𑁄𑁇_  


800. _𑀯𑀭𑀼𑀡𑀸 𑀲𑀳𑀥𑀫𑁆𑀫𑀸 𑀘,_  
_𑀅𑀘𑁆𑀘𑀼𑀢𑀸 𑀘 𑀅𑀦𑁂𑀚𑀓𑀸;_  
_𑀲𑀽𑀮𑁂𑀬𑁆𑀬𑀭𑀼𑀘𑀺𑀭𑀸 𑀆𑀕𑀼𑀁,_  
_𑀆𑀕𑀼𑀁 𑀯𑀸𑀲𑀯𑀦𑁂𑀲𑀺𑀦𑁄;_  
_𑀤𑀲𑁂𑀢𑁂 𑀤𑀲𑀥𑀸 𑀓𑀸𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄𑁇_  


801. _𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


802. _𑀲𑀫𑀸𑀦𑀸 𑀫𑀳𑀸𑀲𑀫𑀦𑀸,_  
_𑀫𑀸𑀦𑀼𑀲𑀸 𑀫𑀸𑀦𑀼𑀲𑀼𑀢𑁆𑀢𑀫𑀸;_  
_𑀔𑀺𑀟𑁆𑀟𑀸𑀧𑀤𑁄𑀲𑀺𑀓𑀸 𑀆𑀕𑀼𑀁,_  
_𑀆𑀕𑀼𑀁 𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑀺𑀓𑀸𑁇_  


803. _𑀅𑀣𑀸𑀕𑀼𑀁 𑀳𑀭𑀬𑁄 𑀤𑁂𑀯𑀸,_  
_𑀬𑁂 𑀘 𑀮𑁄𑀳𑀺𑀢𑀯𑀸𑀲𑀺𑀦𑁄;_  
_𑀧𑀸𑀭𑀕𑀸 𑀫𑀳𑀸𑀧𑀸𑀭𑀕𑀸,_  
_𑀆𑀕𑀼𑀁 𑀤𑁂𑀯𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀤𑀲𑁂𑀢𑁂 𑀤𑀲𑀥𑀸 𑀓𑀸𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄𑁇_  


804. _𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


805. _𑀲𑀼𑀓𑁆𑀓𑀸 𑀓𑀭𑀫𑁆𑀪𑀸 𑀅𑀭𑀼𑀡𑀸,_  
_𑀆𑀕𑀼𑀁 𑀯𑁂𑀖𑀦𑀲𑀸 𑀲𑀳;_  
_𑀑𑀤𑀸𑀢𑀕𑀬𑁆𑀳𑀸 𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑀸,_  
_𑀆𑀕𑀼𑀁 𑀤𑁂𑀯𑀸 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑀸𑁇_  


806. _𑀲𑀤𑀸𑀫𑀢𑁆𑀢𑀸 𑀳𑀸𑀭𑀕𑀚𑀸,_  
_𑀫𑀺𑀲𑁆𑀲𑀓𑀸 𑀘 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀣𑀦𑀬𑀁 𑀆𑀕 𑀧𑀚𑁆𑀚𑀼𑀦𑁆𑀦𑁄,_  
_𑀬𑁄 𑀤𑀺𑀲𑀸 𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀢𑀺𑁇_  


807. _𑀤𑀲𑁂𑀢𑁂 𑀤𑀲𑀥𑀸 𑀓𑀸𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


808. _𑀔𑁂𑀫𑀺𑀬𑀸 𑀢𑀼𑀲𑀺𑀢𑀸 𑀬𑀸𑀫𑀸,_  
_𑀓𑀝𑁆𑀞𑀓𑀸 𑀘 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀮𑀫𑁆𑀩𑀻𑀢𑀓𑀸 𑀮𑀸𑀫𑀲𑁂𑀝𑁆𑀞𑀸,_  
_𑀚𑁄𑀢𑀺𑀦𑀸𑀫𑀸 𑀘 𑀆𑀲𑀯𑀸;_  
_𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀺𑀦𑁄 𑀆𑀕𑀼𑀁,_  
_𑀅𑀣𑀸𑀕𑀼𑀁 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸𑁇_  


809. _𑀤𑀲𑁂𑀢𑁂 𑀤𑀲𑀥𑀸 𑀓𑀸𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


810. _𑀲𑀝𑁆𑀞𑁂𑀢𑁂 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑀺𑀦𑁄;_  
_𑀦𑀸𑀫𑀦𑁆𑀯𑀬𑁂𑀦 𑀆𑀕𑀘𑁆𑀙𑀼𑀁,_  
_𑀬𑁂 𑀘𑀜𑁆𑀜𑁂 𑀲𑀤𑀺𑀲𑀸 𑀲𑀳𑁇_  


811. _‘𑀧𑀯𑀼𑀝𑁆𑀞𑀚𑀸𑀢𑀺𑀫𑀔𑀺𑀮𑀁,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑀫𑀦𑀸𑀲𑀯𑀁;_  
_𑀤𑀓𑁆𑀔𑁂𑀫𑁄𑀖𑀢𑀭𑀁 𑀦𑀸𑀕𑀁,_  
_𑀘𑀦𑁆𑀤𑀁𑀯 𑀅𑀲𑀺𑀢𑀸𑀢𑀺𑀕𑀁’𑁇_  


(341.)

812. _𑀲𑀼𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀧𑀭𑀫𑀢𑁆𑀢𑁄 𑀘,_  
_𑀧𑀼𑀢𑁆𑀢𑀸 𑀇𑀤𑁆𑀥𑀺𑀫𑀢𑁄 𑀲𑀳;_  
_𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘,_  
_𑀲𑁄𑀧𑀸𑀕 𑀲𑀫𑀺𑀢𑀺𑀁 𑀯𑀦𑀁𑁇_  


813. _𑀲𑀳𑀲𑁆𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸𑀦𑀁,_  
_𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀪𑀺𑀢𑀺𑀝𑁆𑀞𑀢𑀺;_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀪𑀺𑀲𑁆𑀫𑀸𑀓𑀸𑀬𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀲𑁄𑁇_  


814. _𑀤𑀲𑁂𑀢𑁆𑀣 𑀇𑀲𑁆𑀲𑀭𑀸 𑀆𑀕𑀼𑀁,_  
_𑀧𑀘𑁆𑀘𑁂𑀓𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀢𑁂𑀲𑀜𑁆𑀘 𑀫𑀚𑁆𑀛𑀢𑁄 𑀆𑀕,_  
_𑀳𑀸𑀭𑀺𑀢𑁄 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄𑁇_  


(342.)

815. _𑀢𑁂 𑀘 𑀲𑀩𑁆𑀩𑁂 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂,_  
_𑀲𑁃𑀦𑁆𑀤𑁂 𑀤𑁂𑀯𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂;_  
_𑀫𑀸𑀭𑀲𑁂𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀸𑀫𑀺,_  
_𑀧𑀲𑁆𑀲 𑀓𑀡𑁆𑀳𑀲𑁆𑀲 𑀫𑀦𑁆𑀤𑀺𑀬𑀁𑁇_  


816. _‘𑀏𑀣 𑀕𑀡𑁆𑀳𑀣 𑀩𑀦𑁆𑀥𑀣,_  
_𑀭𑀸𑀕𑁂𑀦 𑀩𑀤𑁆𑀥𑀫𑀢𑁆𑀣𑀼 𑀯𑁄;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀣,_  
_𑀫𑀸 𑀯𑁄 𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀣 𑀓𑁄𑀘𑀺 𑀦𑀁’𑁇_  


817. _𑀇𑀢𑀺 𑀢𑀢𑁆𑀣 𑀫𑀳𑀸𑀲𑁂𑀦𑁄,_  
_𑀓𑀡𑁆𑀳𑁄 𑀲𑁂𑀦𑀁 𑀅𑀧𑁂𑀲𑀬𑀺;_  
_𑀧𑀸𑀡𑀺𑀦𑀸 𑀢𑀮𑀫𑀸𑀳𑀘𑁆𑀘,_  
_𑀲𑀭𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀪𑁂𑀭𑀯𑀁𑁇_  


818. _𑀬𑀣𑀸 𑀧𑀸𑀯𑀼𑀲𑁆𑀲𑀓𑁄 𑀫𑁂𑀖𑁄,_  
_𑀣𑀦𑀬𑀦𑁆𑀢𑁄 𑀲𑀯𑀺𑀚𑁆𑀚𑀼𑀓𑁄;_  
_𑀢𑀤𑀸 𑀲𑁄 𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑀺,_  
_𑀲𑀗𑁆𑀓𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀬𑀁𑀯𑀲𑁂𑁇_  


(343.)

819. _𑀢𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀯𑀢𑁆𑀣𑀺𑀢𑁆𑀯𑀸𑀦 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀢𑀢𑁄 𑀆𑀫𑀦𑁆𑀢𑀬𑀻 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑀸𑀯𑀓𑁂 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑁂𑁇_  


820. _‘𑀫𑀸𑀭𑀲𑁂𑀦𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸,_  
_𑀢𑁂 𑀯𑀺𑀚𑀸𑀦𑀸𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄’;_  
_𑀢𑁂 𑀘 𑀆𑀢𑀧𑁆𑀧𑀫𑀓𑀭𑀼𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑁂𑀳𑀺 𑀧𑀓𑁆𑀓𑀸𑀫𑀼𑀁,_  
_𑀦𑁂𑀲𑀁 𑀮𑁄𑀫𑀸𑀧𑀺 𑀇𑀜𑁆𑀚𑀬𑀼𑀁𑁇_  


821. _‘𑀲𑀩𑁆𑀩𑁂 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫𑀸,_  
_𑀪𑀬𑀸𑀢𑀻𑀢𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀲𑀳 𑀪𑀽𑀢𑁂𑀳𑀺,_  
_𑀲𑀸𑀯𑀓𑀸 𑀢𑁂 𑀚𑀦𑁂𑀲𑀼𑀢𑀸’”𑀢𑀺𑁇_  


---

822. 𑀫𑀳𑀸𑀲𑀫𑀬𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
