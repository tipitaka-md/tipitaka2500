# 𑁮𑁇𑁨𑁇𑁪 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀯𑀭

* 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀽𑀧𑀲𑀗𑁆𑀓𑀫

* 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀯𑀭

(365.)

895\. 𑀇𑀢𑀺𑀳 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺—

896\. “𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀧𑀦, 𑀫𑀸𑀭𑀺𑀲, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀯𑀭𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀲𑀤𑁆𑀤𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀕𑀦𑁆𑀥𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀲𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀻”𑀢𑀺𑁇

897\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

898\. “𑀇𑀫𑀲𑁆𑀲 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀬𑀣𑀸𑀭𑀽𑀧𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀣𑀸𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀣𑀸𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀲𑀤𑁆𑀤𑀁 𑀲𑁂𑀯𑀢𑁄…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀕𑀦𑁆𑀥𑀁 𑀲𑁂𑀯𑀢𑁄…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀲𑀁 𑀲𑁂𑀯𑀢𑁄…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀲𑁂𑀯𑀢𑁄…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀣𑀸𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑁄𑁇

899\. 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑀸 𑀫𑁂𑀢𑁆𑀣 𑀓𑀗𑁆𑀔𑀸 𑀯𑀺𑀕𑀢𑀸 𑀓𑀣𑀁𑀓𑀣𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑁆𑀯𑀸”𑀢𑀺𑁇

(366.)

900\. 𑀇𑀢𑀺𑀳 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺—

901\. “𑀲𑀩𑁆𑀩𑁂𑀯 𑀦𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀓𑀦𑁆𑀢𑀯𑀸𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀲𑀻𑀮𑀸 𑀏𑀓𑀦𑁆𑀢𑀙𑀦𑁆𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀸”𑀢𑀺? “𑀦 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀓𑀦𑁆𑀢𑀯𑀸𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀲𑀻𑀮𑀸 𑀏𑀓𑀦𑁆𑀢𑀙𑀦𑁆𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀸”𑀢𑀺𑁇

902\. “𑀓𑀲𑁆𑀫𑀸 𑀧𑀦, 𑀫𑀸𑀭𑀺𑀲, 𑀦 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀓𑀦𑁆𑀢𑀯𑀸𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀲𑀻𑀮𑀸 𑀏𑀓𑀦𑁆𑀢𑀙𑀦𑁆𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀸”𑀢𑀺? “𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼 𑀦𑀸𑀦𑀸𑀥𑀸𑀢𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀮𑁄𑀓𑁄𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼𑀦𑀸𑀦𑀸𑀥𑀸𑀢𑀼𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀬𑀁 𑀬𑀤𑁂𑀯 𑀲𑀢𑁆𑀢𑀸 𑀥𑀸𑀢𑀼𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑀦𑁆𑀢𑀺, 𑀢𑀁 𑀢𑀤𑁂𑀯 𑀣𑀸𑀫𑀲𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑀺—  ‘𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸 𑀦 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀓𑀦𑁆𑀢𑀯𑀸𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀲𑀻𑀮𑀸 𑀏𑀓𑀦𑁆𑀢𑀙𑀦𑁆𑀤𑀸 𑀏𑀓𑀦𑁆𑀢𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀸”𑀢𑀺𑁇

903\. “𑀲𑀩𑁆𑀩𑁂𑀯 𑀦𑀼 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸”𑀢𑀺? “𑀦 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸”𑀢𑀺𑁇

904\. “𑀓𑀲𑁆𑀫𑀸 𑀧𑀦, 𑀫𑀸𑀭𑀺𑀲, 𑀦 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸”𑀢𑀺? “𑀬𑁂 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀢𑁂 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸𑁇 𑀢𑀲𑁆𑀫𑀸 𑀦 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀦𑀺𑀝𑁆𑀞𑀸 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸”𑀢𑀺𑁇

905\. 𑀇𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—  “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀕𑀯𑀸, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀲𑀼𑀕𑀢𑁇 𑀢𑀺𑀡𑁆𑀡𑀸 𑀫𑁂𑀢𑁆𑀣 𑀓𑀗𑁆𑀔𑀸 𑀯𑀺𑀕𑀢𑀸 𑀓𑀣𑀁𑀓𑀣𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑁆𑀯𑀸”𑀢𑀺𑁇

906\. 𑀇𑀢𑀺𑀳 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

907\. “𑀏𑀚𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑁄𑀕𑁄, 𑀏𑀚𑀸 𑀕𑀡𑁆𑀟𑁄, 𑀏𑀚𑀸 𑀲𑀮𑁆𑀮𑀁, 𑀏𑀚𑀸 𑀇𑀫𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑁂𑀯 𑀪𑀯𑀲𑁆𑀲 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀢𑀲𑁆𑀫𑀸 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁂𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀳𑀸𑀦𑀁 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀜𑁆𑀜𑁂𑀲𑀼 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀑𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀦𑀸𑀮𑀢𑁆𑀣𑀁, 𑀢𑁂 𑀫𑁂 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀸𑀦𑀼𑀲𑀬𑀺𑀢𑀜𑁆𑀘 𑀧𑀦 𑀫𑁂 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀓𑀣𑀁𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀁, 𑀢𑀜𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀩𑀼𑀍𑀳𑀦𑁆”𑀢𑀺𑁇

908\. “𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀲𑀺 𑀦𑁄 𑀢𑁆𑀯𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀇𑀫𑁂 𑀧𑀜𑁆𑀳𑁂 𑀅𑀜𑁆𑀜𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸”𑀢𑀺? “𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑁂 𑀧𑀜𑁆𑀳𑁂 𑀅𑀜𑁆𑀜𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸”𑀢𑀺𑁇 “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦 𑀢𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀩𑁆𑀬𑀸𑀓𑀁𑀲𑀼? 𑀲𑀘𑁂 𑀢𑁂 𑀅𑀕𑀭𑀼 𑀪𑀸𑀲𑀲𑁆𑀲𑀽”𑀢𑀺𑁇 “𑀦 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀭𑀼 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀭𑀽𑀧𑁄 𑀯𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀪𑀸𑀲𑀲𑁆𑀲𑀽”𑀢𑀺𑁇 “𑀬𑁂𑀲𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀜𑁆𑀜𑀸𑀫𑀺 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀧𑀦𑁆𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀢𑀺, 𑀢𑁆𑀬𑀸𑀳𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑁂 𑀧𑀜𑁆𑀳𑁂 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺, 𑀢𑁂 𑀫𑀬𑀸 𑀧𑀼𑀝𑁆𑀞𑀸 𑀦 𑀲𑀫𑁆𑀧𑀸𑀬𑀦𑁆𑀢𑀺, 𑀅𑀲𑀫𑁆𑀧𑀸𑀬𑀦𑁆𑀢𑀸 𑀫𑀫𑀁𑀬𑁂𑀯 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  ‘𑀓𑁄 𑀦𑀸𑀫𑁄 𑀆𑀬𑀲𑁆𑀫𑀸’𑀢𑀺? 𑀢𑁂𑀲𑀸𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀫𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄’𑀢𑀺𑁇 𑀢𑁂 𑀫𑀫𑀁𑀬𑁂𑀯 𑀉𑀢𑁆𑀢𑀭𑀺 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  ‘𑀓𑀺𑀁 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀇𑀫𑀁 𑀞𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑁄’𑀢𑀺? 𑀢𑁂𑀲𑀸𑀳𑀁 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀺𑁇 𑀢𑁂 𑀢𑀸𑀯𑀢𑀓𑁂𑀦𑁂𑀯 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  ‘𑀲𑀓𑁆𑀓𑁄 𑀘 𑀦𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀬𑀜𑁆𑀘 𑀦𑁄 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀸, 𑀢𑀜𑁆𑀘 𑀦𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀻’𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼 𑀫𑀫𑀁𑀬𑁂𑀯 𑀲𑀸𑀯𑀓𑀸 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀸𑀳𑀁 𑀢𑁂𑀲𑀁𑁇 𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄”𑀢𑀺𑁇
