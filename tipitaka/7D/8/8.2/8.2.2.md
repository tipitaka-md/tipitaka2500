# 𑁮𑁇𑁨𑁇𑁨 𑀯𑁂𑀤𑀦𑀸𑀓𑀫𑁆𑀫𑀝𑁆𑀞𑀸𑀦

### 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀽𑀧𑀲𑀗𑁆𑀓𑀫

### 𑀯𑁂𑀤𑀦𑀸𑀓𑀫𑁆𑀫𑀝𑁆𑀞𑀸𑀦

(359.)

884\. “𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺—  𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺𑁇

(360.)

885\. 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀻𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 ‘𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 ‘𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀢𑁆𑀣 𑀬𑀜𑁆𑀘𑁂 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀬𑀜𑁆𑀘𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁, 𑀬𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀅𑀯𑀺𑀘𑀸𑀭𑁂, 𑀢𑁂 𑀧𑀡𑀻𑀢𑀢𑀭𑁂𑁇 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀻𑀢𑀺𑁇 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇

(361.)

886\. 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀻𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 ‘𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 ‘𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀢𑁆𑀣 𑀬𑀜𑁆𑀘𑁂 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀬𑀜𑁆𑀘𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁, 𑀬𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀅𑀯𑀺𑀘𑀸𑀭𑁂, 𑀢𑁂 𑀧𑀡𑀻𑀢𑀢𑀭𑁂𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀻𑀢𑀺 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇

(362.)

887\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀻𑀢𑀺 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀁 ‘𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀉𑀧𑁂𑀓𑁆𑀔𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀣 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀁 ‘𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀉𑀧𑁂𑀓𑁆𑀔𑀁 𑀲𑁂𑀯𑀢𑁄 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺, 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀣 𑀬𑀜𑁆𑀘𑁂 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀬𑀜𑁆𑀘𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁, 𑀬𑁂 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀅𑀯𑀺𑀘𑀸𑀭𑁂, 𑀢𑁂 𑀧𑀡𑀻𑀢𑀢𑀭𑁂𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀫𑁆𑀧𑀸𑀳𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀤𑀸𑀫𑀺 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀅𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀫𑁆𑀧𑀻𑀢𑀺 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇

(363.)

888\. 𑀏𑀯𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸𑀦𑀺𑀭𑁄𑀥𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 𑀇𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—  “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀕𑀯𑀸, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀲𑀼𑀕𑀢𑁇 𑀢𑀺𑀡𑁆𑀡𑀸 𑀫𑁂𑀢𑁆𑀣 𑀓𑀗𑁆𑀔𑀸 𑀯𑀺𑀕𑀢𑀸 𑀓𑀣𑀁𑀓𑀣𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑁆𑀯𑀸”𑀢𑀺𑁇
