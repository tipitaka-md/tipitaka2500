# 𑁮𑁇𑁨 𑀲𑀓𑁆𑀓𑀽𑀧𑀲𑀗𑁆𑀓𑀫

* 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀽𑀧𑀲𑀗𑁆𑀓𑀫

(350.)

847\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀢𑀺 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸, 𑀪𑀕𑀯𑀸 𑀘 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑀁 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀳𑀺 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁—  ‘𑀲𑀓𑁆𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀲𑀧𑀭𑀺𑀚𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀺—  “𑀲𑀓𑁆𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀲𑀧𑀭𑀺𑀚𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁 𑀲𑀼𑀔𑀻 𑀳𑁄𑀢𑀼, 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀲𑀧𑀭𑀺𑀚𑀦𑁄; 𑀲𑀼𑀔𑀓𑀸𑀫𑀸 𑀳𑀺 𑀤𑁂𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀅𑀲𑀼𑀭𑀸 𑀦𑀸𑀕𑀸 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸 𑀬𑁂 𑀘𑀜𑁆𑀜𑁂 𑀲𑀦𑁆𑀢𑀺 𑀧𑀼𑀣𑀼𑀓𑀸𑀬𑀸”𑀢𑀺𑁇

* 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

(351.)

848\. 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀸 𑀏𑀯𑀭𑀽𑀧𑁂 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑁂 𑀬𑀓𑁆𑀔𑁂 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺𑁇 𑀅𑀪𑀺𑀯𑀤𑀺𑀢𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀤𑁂𑀯𑀸𑀧𑀺 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄𑀧𑀺 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇

* 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀽𑀧𑀲𑀗𑁆𑀓𑀫

849\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀇𑀦𑁆𑀤𑀲𑀸𑀮𑀕𑀼𑀳𑀸 𑀯𑀺𑀲𑀫𑀸 𑀲𑀦𑁆𑀢𑀻 𑀲𑀫𑀸 𑀲𑀫𑀧𑀸𑀤𑀺, 𑀲𑀫𑁆𑀩𑀸𑀥𑀸 𑀲𑀦𑁆𑀢𑀻 𑀉𑀭𑀼𑀦𑁆𑀤𑀸 𑀲𑀫𑀧𑀸𑀤𑀺, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀕𑀼𑀳𑀸𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺 𑀬𑀣𑀸 𑀢𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀼𑀪𑀸𑀯𑁂𑀦𑁇

* 𑀲𑀓𑁆𑀓𑀽𑀧𑀲𑀗𑁆𑀓𑀫

(352.)

850\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀫𑀺𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀓𑁄𑀲𑀺𑀬𑀲𑁆𑀲, 𑀅𑀩𑁆𑀪𑀼𑀢𑀫𑀺𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀓𑁄𑀲𑀺𑀬𑀲𑁆𑀲 𑀢𑀸𑀯 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑀲𑁆𑀲 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑀲𑁆𑀲 𑀬𑀤𑀺𑀤𑀁 𑀇𑀥𑀸𑀕𑀫𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀘𑀺𑀭𑀧𑀝𑀺𑀓𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀓𑀸𑀫𑁄; 𑀅𑀧𑀺 𑀘 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀓𑁂𑀳𑀺𑀘𑀺 𑀓𑁂𑀳𑀺𑀘𑀺 𑀓𑀺𑀘𑁆𑀘𑀓𑀭𑀡𑀻𑀬𑁂𑀳𑀺 𑀩𑁆𑀬𑀸𑀯𑀝𑁄; 𑀏𑀯𑀸𑀳𑀁 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀏𑀓𑀫𑀺𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀮𑀍𑀆𑀕𑀸𑀭𑀓𑁂𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀪𑀽𑀚𑀢𑀺 𑀘 𑀦𑀸𑀫 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀦𑀫𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀽𑀚𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀳𑀺 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁—  “𑀲𑀓𑁆𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀲𑀧𑀭𑀺𑀚𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻”’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸 𑀪𑀽𑀚𑀢𑀺 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀅𑀓𑀸𑀮𑁄 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬; 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀕𑀺𑀦𑀺, 𑀬𑀤𑀸 𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀳𑀺—  “𑀲𑀓𑁆𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀲𑀧𑀭𑀺𑀚𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻”’𑀢𑀺𑁇 𑀓𑀘𑁆𑀘𑀺 𑀫𑁂 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀲𑀺? 𑀲𑀭𑀢𑀺 𑀪𑀕𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀪𑀕𑀺𑀦𑀺𑀬𑀸 𑀯𑀘𑀦𑀦𑁆”𑀢𑀺? “𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀲𑀺 𑀫𑀁 𑀲𑀸, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀪𑀕𑀺𑀦𑀻, 𑀲𑀭𑀸𑀫𑀳𑀁 𑀢𑀲𑁆𑀲𑀸 𑀪𑀕𑀺𑀦𑀺𑀬𑀸 𑀯𑀘𑀦𑀁𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑁂𑀫𑀺𑀲𑀤𑁆𑀤𑁂𑀦 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄”𑀢𑀺𑁇 “𑀬𑁂 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀞𑀫𑀢𑀭𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸, 𑀢𑁂𑀲𑀁 𑀫𑁂 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  ‘𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸, 𑀤𑀺𑀩𑁆𑀩𑀸 𑀓𑀸𑀬𑀸 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀳𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀓𑀸𑀬𑀸’𑀢𑀺𑁇 𑀢𑀁 𑀫𑁂 𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀔𑀺𑀤𑀺𑀝𑁆𑀞𑀁 𑀬𑀢𑁄 𑀢𑀣𑀸𑀕𑀢𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀤𑀺𑀩𑁆𑀩𑀸 𑀓𑀸𑀬𑀸 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺, 𑀳𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀓𑀸𑀬𑀸𑀢𑀺𑁇
