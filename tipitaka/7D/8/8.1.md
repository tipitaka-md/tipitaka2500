# 𑁮𑁇𑁧 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑀕𑀻𑀢𑀕𑀸𑀣𑀸

### 𑀲𑀓𑁆𑀓𑀧𑀜𑁆𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑀕𑀻𑀢𑀕𑀸𑀣𑀸

(348.)

827\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑁄 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀩𑁂𑀮𑀼𑀯𑀧𑀡𑁆𑀟𑀼𑀯𑀻𑀡𑀁 𑀅𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀺, 𑀇𑀫𑀸 𑀘 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀲𑀁𑀖𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀅𑀭𑀳𑀦𑁆𑀢𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸—

828\. _“𑀯𑀦𑁆𑀤𑁂 𑀢𑁂 𑀧𑀺𑀢𑀭𑀁 𑀪𑀤𑁆𑀤𑁂,_  
_𑀢𑀺𑀫𑁆𑀩𑀭𑀼𑀁 𑀲𑀽𑀭𑀺𑀬𑀯𑀘𑁆𑀙𑀲𑁂;_  
_𑀬𑁂𑀦 𑀚𑀸𑀢𑀸𑀲𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀻,_  
_𑀆𑀦𑀦𑁆𑀤𑀚𑀦𑀦𑀻 𑀫𑀫𑁇_  


829\. _𑀯𑀸𑀢𑁄𑀯 𑀲𑁂𑀤𑀢𑀁 𑀓𑀦𑁆𑀢𑁄,_  
_𑀧𑀸𑀦𑀻𑀬𑀁𑀯 𑀧𑀺𑀧𑀸𑀲𑀢𑁄;_  
_𑀅𑀗𑁆𑀕𑀻𑀭𑀲𑀺 𑀧𑀺𑀬𑀸𑀫𑁂𑀲𑀺,_  
_𑀥𑀫𑁆𑀫𑁄 𑀅𑀭𑀳𑀢𑀸𑀫𑀺𑀯𑁇_  


830\. _𑀆𑀢𑀼𑀭𑀲𑁆𑀲𑁂𑀯 𑀪𑁂𑀲𑀚𑁆𑀚𑀁,_  
_𑀪𑁄𑀚𑀦𑀁𑀯 𑀚𑀺𑀖𑀘𑁆𑀙𑀢𑁄;_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀬 𑀫𑀁 𑀪𑀤𑁆𑀤𑁂,_  
_𑀚𑀮𑀦𑁆𑀢𑀫𑀺𑀯 𑀯𑀸𑀭𑀺𑀦𑀸𑁇_  


831\. _𑀲𑀻𑀢𑁄𑀤𑀓𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁,_  
_𑀬𑀼𑀢𑁆𑀢𑀁 𑀓𑀺𑀜𑁆𑀚𑀓𑁆𑀔𑀭𑁂𑀡𑀼𑀦𑀸;_  
_𑀦𑀸𑀕𑁄 𑀖𑀫𑁆𑀫𑀸𑀪𑀺𑀢𑀢𑁆𑀢𑁄𑀯,_  
_𑀑𑀕𑀸𑀳𑁂 𑀢𑁂 𑀣𑀦𑀽𑀤𑀭𑀁𑁇_  


832\. _𑀅𑀘𑁆𑀘𑀗𑁆𑀓𑀼𑀲𑁄𑀯 𑀦𑀸𑀕𑁄𑀯,_  
_𑀚𑀺𑀢𑀁 𑀫𑁂 𑀢𑀼𑀢𑁆𑀢𑀢𑁄𑀫𑀭𑀁;_  
_𑀓𑀸𑀭𑀡𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑁄 𑀮𑀓𑁆𑀔𑀡𑀽𑀭𑀼𑀬𑀸𑁇_  


833\. _𑀢𑀬𑀺 𑀕𑁂𑀥𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑀲𑁆𑀫𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀢𑀁;_  
_𑀧𑀝𑀺𑀕𑀦𑁆𑀢𑀼𑀁 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺,_  
_𑀯𑀗𑁆𑀓𑀖𑀲𑁆𑀢𑁄𑀯 𑀅𑀫𑁆𑀩𑀼𑀚𑁄𑁇_  


834\. _𑀯𑀸𑀫𑀽𑀭𑀼 𑀲𑀚 𑀫𑀁 𑀪𑀤𑁆𑀤𑁂,_  
_𑀲𑀚 𑀫𑀁 𑀫𑀦𑁆𑀤𑀮𑁄𑀘𑀦𑁂;_  
_𑀧𑀮𑀺𑀲𑁆𑀲𑀚 𑀫𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀺,_  
_𑀏𑀢𑀁 𑀫𑁂 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑀁𑁇_  


835\. _𑀅𑀧𑁆𑀧𑀓𑁄 𑀯𑀢 𑀫𑁂 𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀸𑀫𑁄 𑀯𑁂𑀮𑁆𑀮𑀺𑀢𑀓𑁂𑀲𑀺𑀬𑀸;_  
_𑀅𑀦𑁂𑀓𑀪𑀸𑀯𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀺,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀯 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑁇_  


836\. _𑀬𑀁 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺 𑀓𑀢𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀲𑀼 𑀢𑀸𑀤𑀺𑀲𑀼;_  
_𑀢𑀁 𑀫𑁂 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀓𑀮𑁆𑀬𑀸𑀡𑀺,_  
_𑀢𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀁𑁇_  


837\. _𑀬𑀁 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺 𑀓𑀢𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀧𑀣𑀯𑀺𑀫𑀡𑁆𑀟𑀮𑁂;_  
_𑀢𑀁 𑀫𑁂 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀓𑀮𑁆𑀬𑀸𑀡𑀺,_  
_𑀢𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀁𑁇_  


838\. _𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄𑀯 𑀛𑀸𑀦𑁂𑀦,_  
_𑀏𑀓𑁄𑀤𑀺 𑀦𑀺𑀧𑀓𑁄 𑀲𑀢𑁄;_  
_𑀅𑀫𑀢𑀁 𑀫𑀼𑀦𑀺 𑀚𑀺𑀕𑀻𑀲𑀸𑀦𑁄,_  
_𑀢𑀫𑀳𑀁 𑀲𑀽𑀭𑀺𑀬𑀯𑀘𑁆𑀙𑀲𑁂𑁇_  


839\. _𑀬𑀣𑀸𑀧𑀺 𑀫𑀼𑀦𑀺 𑀦𑀦𑁆𑀤𑁂𑀬𑁆𑀬,_  
_𑀧𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀏𑀯𑀁 𑀦𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀺,_  
_𑀫𑀺𑀲𑁆𑀲𑀻𑀪𑀸𑀯𑀁 𑀕𑀢𑁄 𑀢𑀬𑀸𑁇_  


840\. _𑀲𑀓𑁆𑀓𑁄 𑀘𑁂 𑀫𑁂 𑀯𑀭𑀁 𑀤𑀚𑁆𑀚𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀫𑀺𑀲𑁆𑀲𑀭𑁄;_  
_𑀢𑀸𑀳𑀁 𑀪𑀤𑁆𑀤𑁂 𑀯𑀭𑁂𑀬𑁆𑀬𑀸𑀳𑁂,_  
_𑀏𑀯𑀁 𑀓𑀸𑀫𑁄 𑀤𑀍𑀳𑁄 𑀫𑀫𑁇_  


841\. _𑀲𑀸𑀮𑀁𑀯 𑀦 𑀘𑀺𑀭𑀁 𑀨𑀼𑀮𑁆𑀮𑀁,_  
_𑀧𑀺𑀢𑀭𑀁 𑀢𑁂 𑀲𑀼𑀫𑁂𑀥𑀲𑁂;_  
_𑀯𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀲𑁆𑀲𑀸𑀲𑁂𑀢𑀸𑀤𑀺𑀲𑀻 𑀧𑀚𑀸”𑀢𑀺𑁇_  


(349.)

842\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑀺𑀔𑀁 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀁𑀲𑀦𑁆𑀤𑀢𑀺 𑀔𑁄 𑀢𑁂, 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀢𑀦𑁆𑀢𑀺𑀲𑁆𑀲𑀭𑁄 𑀕𑀻𑀢𑀲𑁆𑀲𑀭𑁂𑀦, 𑀕𑀻𑀢𑀲𑁆𑀲𑀭𑁄 𑀘 𑀢𑀦𑁆𑀢𑀺𑀲𑁆𑀲𑀭𑁂𑀦; 𑀦 𑀘 𑀧𑀦 𑀢𑁂, 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀢𑀦𑁆𑀢𑀺𑀲𑁆𑀲𑀭𑁄 𑀕𑀻𑀢𑀲𑁆𑀲𑀭𑀁 𑀅𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺, 𑀕𑀻𑀢𑀲𑁆𑀲𑀭𑁄 𑀘 𑀢𑀦𑁆𑀢𑀺𑀲𑁆𑀲𑀭𑀁𑁇 𑀓𑀤𑀸 𑀲𑀁𑀬𑀽𑀍𑀳𑀸 𑀧𑀦 𑀢𑁂, 𑀧𑀜𑁆𑀘𑀲𑀺𑀔, 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀩𑀼𑀤𑁆𑀥𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀲𑀁𑀖𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀅𑀭𑀳𑀦𑁆𑀢𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸”𑀢𑀺? “𑀏𑀓𑀫𑀺𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦𑀚𑁆𑀚𑀸 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀸𑀬 𑀢𑀻𑀭𑁂 𑀅𑀚𑀧𑀸𑀮𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁂 𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀸 𑀦𑀸𑀫 𑀲𑀽𑀭𑀺𑀬𑀯𑀘𑁆𑀙𑀲𑀸 𑀢𑀺𑀫𑁆𑀩𑀭𑀼𑀦𑁄 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀭𑀜𑁆𑀜𑁄 𑀥𑀻𑀢𑀸, 𑀢𑀫𑀪𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺𑁇 𑀲𑀸 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻 𑀧𑀭𑀓𑀸𑀫𑀺𑀦𑀻 𑀳𑁄𑀢𑀺; 𑀲𑀺𑀔𑀡𑁆𑀟𑀻 𑀦𑀸𑀫 𑀫𑀸𑀢𑀮𑀺𑀲𑁆𑀲 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑁄, 𑀢𑀫𑀪𑀺𑀓𑀗𑁆𑀔𑀢𑀺𑁇 𑀬𑀢𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀁 𑀪𑀕𑀺𑀦𑀺𑀁 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀓𑁂𑀦𑀘𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑁇 𑀅𑀣𑀸𑀳𑀁 𑀩𑁂𑀮𑀼𑀯𑀧𑀡𑁆𑀟𑀼𑀯𑀻𑀡𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀢𑀺𑀫𑁆𑀩𑀭𑀼𑀦𑁄 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀭𑀜𑁆𑀜𑁄 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀩𑁂𑀮𑀼𑀯𑀧𑀡𑁆𑀟𑀼𑀯𑀻𑀡𑀁 𑀅𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀺𑀁, 𑀇𑀫𑀸 𑀘 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀲𑀁𑀖𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀅𑀭𑀳𑀦𑁆𑀢𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸—

843\. _𑀯𑀦𑁆𑀤𑁂 𑀢𑁂 𑀧𑀺𑀢𑀭𑀁 𑀪𑀤𑁆𑀤𑁂,_  
_𑀢𑀺𑀫𑁆𑀩𑀭𑀼𑀁 𑀲𑀽𑀭𑀺𑀬𑀯𑀘𑁆𑀙𑀲𑁂;_  
_𑀬𑁂𑀦 𑀚𑀸𑀢𑀸𑀲𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀻,_  
_𑀆𑀦𑀦𑁆𑀤𑀚𑀦𑀦𑀻 𑀫𑀫𑁇_

844\. …𑀧𑁂…



845\. _𑀲𑀸𑀮𑀁𑀯 𑀦 𑀘𑀺𑀭𑀁 𑀨𑀼𑀮𑁆𑀮𑀁,_  
_𑀧𑀺𑀢𑀭𑀁 𑀢𑁂 𑀲𑀼𑀫𑁂𑀥𑀲𑁂;_  
_𑀯𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀲𑁆𑀲𑀸𑀲𑁂𑀢𑀸𑀤𑀺𑀲𑀻 𑀧𑀚𑀸𑀢𑀺𑁇_  


846\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀤𑁆𑀤𑀸 𑀲𑀽𑀭𑀺𑀬𑀯𑀘𑁆𑀙𑀲𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀦 𑀔𑁄 𑀫𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀅𑀧𑀺 𑀘 𑀲𑀼𑀢𑁄𑀬𑁂𑀯 𑀫𑁂 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀲𑀪𑀸𑀬𑀁 𑀉𑀧𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀬𑀸𑁇 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀸𑀭𑀺𑀲, 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀓𑀺𑀢𑁆𑀢𑁂𑀲𑀺, 𑀳𑁄𑀢𑀼 𑀦𑁄 𑀅𑀚𑁆𑀚 𑀲𑀫𑀸𑀕𑀫𑁄’𑀢𑀺𑁇 𑀲𑁄𑀬𑁂𑀯 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀲𑁆𑀲𑀸 𑀪𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑀸𑀕𑀫𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀦 𑀘 𑀤𑀸𑀦𑀺 𑀢𑀢𑁄 𑀧𑀘𑁆𑀙𑀸”𑀢𑀺𑁇
