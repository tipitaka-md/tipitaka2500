# 𑁪 𑀓𑀼𑀫𑀸𑀭𑀧𑀜𑁆𑀳𑀸

### 𑀓𑀼𑀫𑀸𑀭𑀧𑀜𑁆𑀳𑀸

(1.)

20. 𑀏𑀓𑀁 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑀸 𑀆𑀳𑀸𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀸𑁇

(2.)

21. 𑀤𑁆𑀯𑁂 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘𑁇

(3.)

22. 𑀢𑀻𑀡𑀺 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑁇

(4.)

23. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺𑁇

(5.)

24. 𑀧𑀜𑁆𑀘 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇

(6.)

25. 𑀙 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀙 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀺 𑀆𑀬𑀢𑀦𑀸𑀦𑀺𑁇

(7.)

26. 𑀲𑀢𑁆𑀢 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇

(8.)

27. 𑀅𑀝𑁆𑀞 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄𑁇

(9.)

28. 𑀦𑀯 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀦𑀯 𑀲𑀢𑁆𑀢𑀸𑀯𑀸𑀲𑀸𑁇

(10.)

29. 𑀤𑀲 𑀦𑀸𑀫 𑀓𑀺𑀁? 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 “𑀅𑀭𑀳𑀸”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇

---

30. 𑀓𑀼𑀫𑀸𑀭𑀧𑀜𑁆𑀳𑀸𑁇
