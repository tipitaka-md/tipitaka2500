# 𑁧 𑀲𑀭𑀡𑀢𑁆𑀢𑀬

### 𑀲𑀭𑀡𑀢𑁆𑀢𑀬

(1.)

2\. _𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇_  


3\. _𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇_  


4\. _𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇_  


---

5\. 𑀲𑀭𑀡𑀢𑁆𑀢𑀬𑀁𑁇
