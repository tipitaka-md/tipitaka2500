# 𑀆𑀭𑀫𑁆𑀫𑀡

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀆𑀭𑀫𑁆𑀫𑀡

(2.)

2596\. 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑁄𑀢𑀁…  𑀖𑀸𑀦𑀁…  𑀚𑀺𑀯𑁆𑀳𑀁…  𑀓𑀸𑀬𑀁…  𑀭𑀽𑀧𑁂…  𑀲𑀤𑁆𑀤𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

(3.)

2597\. 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀭𑀽𑀧𑁂…  𑀲𑀤𑁆𑀤𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂…  𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

(4.)

2598\. 𑀉𑀧𑁆𑀧𑀸𑀤𑀻 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀁 𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀁…  𑀭𑀽𑀧𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁…  𑀉𑀧𑁆𑀧𑀸𑀤𑀻 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀧𑁆𑀧𑀸𑀤𑀻 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)
