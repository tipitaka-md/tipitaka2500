# 𑀆𑀲𑁂𑀯𑀦

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀆𑀲𑁂𑀯𑀦

(40.)

2836. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲…  𑀕𑁄𑀢𑁆𑀭𑀪𑀼 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…  𑀯𑁄𑀤𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2837. 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸…𑀧𑁂… 𑁇 (𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀲𑀤𑀺𑀲𑀁𑀬𑁂𑀯𑁇)
