# 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(31.)

2720\. 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2721\. 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2722\. 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(32.)

2723\. 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2724\. 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2725\. 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(33.)

2726\. 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2727\. 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2728\. 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀬 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬, 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
