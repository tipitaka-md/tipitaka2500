99\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀢𑁂𑀯𑀻𑀲, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀅𑀝𑁆𑀞𑀯𑀻𑀲, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀏𑀓𑀸𑀤𑀲, 𑀆𑀲𑁂𑀯𑀦𑁂 𑀏𑀓𑀸𑀤𑀲, 𑀓𑀫𑁆𑀫𑁂 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀆𑀳𑀸𑀭𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀛𑀸𑀦𑁂 𑀫𑀕𑁆𑀕𑁂 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲, 𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀏𑀓𑀯𑀻𑀲, 𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀅𑀯𑀺𑀕𑀢𑁂 𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲𑁇
