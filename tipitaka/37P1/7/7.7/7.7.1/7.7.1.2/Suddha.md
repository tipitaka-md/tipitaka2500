(138.)

591\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀸𑀤𑀲, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀢𑁂𑀯𑀻𑀲, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀧𑀜𑁆𑀘𑀯𑀻𑀲, 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀧𑀜𑁆𑀘𑀯𑀻𑀲, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀢𑀺𑀁𑀲, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀅𑀝𑁆𑀞𑀯𑀻𑀲, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀢𑀺𑀁𑀲, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀧𑀜𑁆𑀘𑀯𑀻𑀲, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀧𑀜𑁆𑀘, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀧𑀜𑁆𑀘, 𑀆𑀲𑁂𑀯𑀦𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀓𑀫𑁆𑀫𑁂 𑀏𑀓𑀸𑀤𑀲, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀆𑀳𑀸𑀭𑁂 𑀏𑀓𑀸𑀤𑀲, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀏𑀓𑀸𑀤𑀲, 𑀛𑀸𑀦𑁂 𑀏𑀓𑀯𑀻𑀲, 𑀫𑀕𑁆𑀕𑁂 𑀲𑁄𑀍𑀅𑀲, 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀏𑀓𑀸𑀤𑀲, 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀦𑀯, 𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑀺𑀁𑀲, 𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀯𑀻𑀲, 𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑀯𑀻𑀲, 𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀺𑀁𑀲𑁇

592\. (𑀖𑀝𑀦𑀸 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓𑀲𑀤𑀺𑀲𑀸𑀬𑁂𑀯𑁇 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑀕𑀡𑀦𑀁 𑀏𑀯𑀁 𑀅𑀲𑀫𑁆𑀫𑁄𑀳𑀦𑁆𑀢𑁂𑀦 𑀕𑀡𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
