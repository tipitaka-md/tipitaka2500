# 𑀆𑀭𑀫𑁆𑀫𑀡

* 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀢𑁆𑀢𑀺𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀆𑀭𑀫𑁆𑀫𑀡

(37.)

2349\. 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁂 𑀧𑀳𑀻𑀦𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀫𑀼𑀤𑀸𑀘𑀺𑀡𑁆𑀡𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄…𑀧𑁂…  𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀲𑁆𑀲, 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2350\. 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

(38.)

2351\. 𑀅𑀦𑀺𑀬𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀸𑀦𑀺 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺, 𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀛𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺, 𑀅𑀭𑀺𑀬𑀸 𑀨𑀮𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲, 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲, 𑀨𑀮𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀭𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑁂 𑀧𑀳𑀻𑀦𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀯𑀺𑀓𑁆𑀔𑀫𑁆𑀪𑀺𑀢𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀫𑀼𑀤𑀸𑀘𑀺𑀡𑁆𑀡𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀘𑀓𑁆𑀔𑀼𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀦𑀺𑀬𑀢𑁂 𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑀺𑀬𑀢𑁄 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑁂𑀦 𑀅𑀦𑀺𑀬𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲…𑀧𑁂…  𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀺𑀬𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑀜𑀸𑀡𑀲𑁆𑀲, 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀲𑁆𑀲, 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑀜𑀸𑀡𑀲𑁆𑀲, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑀜𑀸𑀡𑀲𑁆𑀲, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2352\. 𑀅𑀦𑀺𑀬𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀭𑀽𑀧𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀺𑀓𑀫𑁆𑀫𑀲𑁆𑀲…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀺𑀓𑀫𑁆𑀫𑀲𑁆𑀲…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀺𑀓𑀫𑁆𑀫𑀲𑁆𑀲…  𑀭𑀼𑀳𑀺𑀭𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀬𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀸𑀫𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2353\. 𑀅𑀦𑀺𑀬𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
