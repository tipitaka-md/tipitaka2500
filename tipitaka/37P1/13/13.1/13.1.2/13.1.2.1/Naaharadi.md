# 𑀦𑀆𑀳𑀸𑀭𑀸𑀤𑀺

### 𑀧𑀭𑀺𑀢𑁆𑀢𑀢𑁆𑀢𑀺𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀦𑀆𑀳𑀸𑀭𑀸𑀤𑀺

(24.)

1948. 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸—  𑀩𑀸𑀳𑀺𑀭𑀁…  𑀉𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 (𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁)…  𑀦𑁃𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀧𑀘𑁆𑀘𑀬𑀸—  𑀩𑀸𑀳𑀺𑀭𑀁…  𑀆𑀳𑀸𑀭𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀉𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀽𑀧𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀦𑀛𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀜𑁆𑀘𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀳𑀕𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁…𑀧𑁂…  𑀩𑀸𑀳𑀺𑀭𑀁…𑀧𑁂…  𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂…  (𑀲𑀩𑁆𑀩𑁂 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸), 𑀦𑀫𑀕𑁆𑀕𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁…𑀧𑁂…  𑀅𑀳𑁂𑀢𑀼𑀓𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀓𑀁…𑀧𑁂…  (𑀲𑀩𑁆𑀩𑁂 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸) 𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀩𑀸𑀳𑀺𑀭𑀁…  𑀆𑀳𑀸𑀭𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀉𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁…𑀧𑁂… 𑁇

1949. 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀳𑀕𑁆𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁…𑀧𑁂… 𑁇 (𑁧)

1950. 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀭𑀽𑀧𑁂 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁…𑀧𑁂…  𑀦𑁄𑀦𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑁄𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇
