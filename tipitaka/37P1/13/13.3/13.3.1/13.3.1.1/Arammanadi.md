# 𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀤𑀺

* 𑀧𑀭𑀺𑀢𑁆𑀢𑀢𑁆𑀢𑀺𑀓

* 𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀤𑀺

(31.)

1971\. 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸—  𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 (𑀅𑀯𑀲𑁂𑀲𑀸 𑀙 𑀧𑀜𑁆𑀳𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀲𑀤𑀺𑀲𑀸 𑀲𑀢𑁆𑀢 𑀓𑀸𑀢𑀩𑁆𑀩𑀸), 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 (𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺 𑀦𑀢𑁆𑀣𑀺, 𑀲𑀢𑁆𑀢𑀭𑀲 𑀧𑀜𑁆𑀳𑀸 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸), 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇
