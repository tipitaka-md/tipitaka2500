(90.)

2150\. 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀦𑁆𑀦𑀭𑀲, 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀧𑀦𑁆𑀦𑀭𑀲, 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀧𑀦𑁆𑀦𑀭𑀲, 𑀦𑀲𑀳𑀚𑀸𑀢𑁂 𑀤𑁆𑀯𑀸𑀤𑀲, 𑀦𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀤𑁆𑀯𑀸𑀤𑀲, 𑀦𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀤𑁆𑀯𑀸𑀤𑀲, 𑀦𑁅𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀧𑀦𑁆𑀦𑀭𑀲, 𑀦𑀆𑀲𑁂𑀯𑀦𑁂 𑀧𑀦𑁆𑀦𑀭𑀲…𑀧𑁂…  𑀦𑀫𑀕𑁆𑀕𑁂 𑀧𑀦𑁆𑀦𑀭𑀲, 𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀤𑁆𑀯𑀸𑀤𑀲, 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀤𑀲, 𑀦𑁄𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀤𑀲, 𑀦𑁄𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀦𑁆𑀦𑀭𑀲, 𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀧𑀦𑁆𑀦𑀭𑀲, 𑀦𑁄𑀅𑀯𑀺𑀕𑀢𑁂 𑀤𑀲𑁇 (𑀏𑀯𑀁 𑀕𑀡𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
