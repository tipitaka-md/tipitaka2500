### 𑀓𑀫𑁆𑀫

(78.)

2105\. 𑀧𑀭𑀺𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2106\. 𑀫𑀳𑀕𑁆𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀕𑁆𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2107\. 𑀫𑀳𑀕𑁆𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2108\. 𑀫𑀳𑀕𑁆𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀫𑀳𑀕𑁆𑀕𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(79.)

2109\. 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2110\. 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2111\. 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
