# 𑀳𑁂𑀢𑀼𑀲𑀸𑀫𑀜𑁆𑀜𑀖𑀝𑀦𑀸

* 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀢𑁆𑀢𑀺𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀧𑀘𑁆𑀘𑀦𑀻𑀬

* 𑀳𑁂𑀢𑀼𑀲𑀸𑀫𑀜𑁆𑀜𑀖𑀝𑀦𑀸

(65.)

3096\. 𑀳𑁂𑀢𑀼 𑀲𑀳𑀚𑀸𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑀢𑁆𑀢…𑀧𑁂…  𑀦𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀢𑁆𑀢, 𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀢𑁆𑀢, 𑀦𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀲𑀢𑁆𑀢 (𑀇𑀥𑀸𑀧𑀺 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁), 𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀏𑀓𑀁, 𑀦𑁄𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀲𑀢𑁆𑀢𑁇

3097\. 𑀳𑁂𑀢𑀼 𑀲𑀳𑀚𑀸𑀢 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜 𑀦𑀺𑀲𑁆𑀲𑀬 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁), 𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀏𑀯𑀁 𑀕𑀡𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
