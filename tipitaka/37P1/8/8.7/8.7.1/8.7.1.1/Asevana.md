# 𑀆𑀲𑁂𑀯𑀦

### 𑀧𑀻𑀢𑀺𑀢𑁆𑀢𑀺𑀓

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀆𑀲𑁂𑀯𑀦

(37.)

742\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲…  𑀕𑁄𑀢𑁆𑀭𑀪𑀼 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…  𑀯𑁄𑀤𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

743\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀯𑁄𑀤𑀸𑀦𑀁 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

744\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑀦𑀜𑁆𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀯𑁄𑀤𑀸𑀦𑀁 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(38.)

745\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀧𑀻𑀢𑀺𑀦𑀬𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)

746\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀯𑁄𑀤𑀸𑀦𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

(39.)

747\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲…𑀧𑁂…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀘 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑀦𑀜𑁆𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀜𑁆𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀜𑁆𑀘 𑀯𑁄𑀤𑀸𑀦𑀁 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀘 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
