# 𑀳𑁂𑀢𑀼

### 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(1.)

2430\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

2431\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄𑁇 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁨)

2432\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

(2.)

2433\. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

2434\. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁨)

2435\. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

(3.)

2436\. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

2437\. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁨)

2438\. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

2439\. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁪)

2440\. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁫)

(4.)

2441\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

2442\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁨)

2443\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)

(5.)

2444\. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁧)

2445\. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁨)

2446\. 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀜𑁆𑀘 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀘 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻 𑀘 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂…𑀧𑁂… 𑁇 (𑁩)
