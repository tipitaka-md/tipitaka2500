* 𑀆𑀲𑁂𑀯𑀦

(45.)

2561\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

2562\. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀅𑀦𑀦𑁆𑀢𑀭𑀲𑀤𑀺𑀲𑀁𑁇 𑀦𑀯 𑀧𑀜𑁆𑀳𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀆𑀯𑀚𑁆𑀚𑀦𑀸 𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)
