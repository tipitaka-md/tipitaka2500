# 𑁧𑁯𑁇𑁭𑁇𑁪 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀸𑀦𑀼𑀮𑁄𑀫

(21.)

2661\. 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀢𑀻𑀡𑀺, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀳𑀚𑀸𑀢𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀏𑀓𑀁, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀢𑀻𑀡𑀺, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀏𑀓𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀆𑀲𑁂𑀯𑀦𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀓𑀫𑁆𑀫𑁂 𑀤𑁆𑀯𑁂, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁 (𑀇𑀫𑁂𑀲𑀼 𑀧𑀤𑁂𑀲𑀼 𑀏𑀓𑀁𑀬𑁂𑀯), 𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀏𑀯𑀁 𑀕𑀡𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
