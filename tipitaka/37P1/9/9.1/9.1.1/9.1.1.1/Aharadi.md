# 𑀆𑀳𑀸𑀭𑀸𑀤𑀺

* 𑀤𑀲𑁆𑀲𑀦𑁂𑀦𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀓

* 𑀆𑀳𑀸𑀭𑀸𑀤𑀺

(14.)

818\. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 (𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘 𑀆𑀳𑀸𑀭𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘)…  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀧𑀘𑁆𑀘𑀬𑀸 (𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀲𑀤𑀺𑀲𑀁)…  𑀛𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸…  𑀫𑀕𑁆𑀕𑀧𑀘𑁆𑀘𑀬𑀸 (𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀲𑀤𑀺𑀲𑀁)…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸 (𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀲𑀤𑀺𑀲𑀁)…  𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸 (𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓𑁂 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀲𑀤𑀺𑀲𑀁)…  𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑀸 (𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀲𑀤𑀺𑀲𑀁)…  𑀦𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇
