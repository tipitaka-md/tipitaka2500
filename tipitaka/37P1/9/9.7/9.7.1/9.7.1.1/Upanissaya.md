### 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(83.)

990\. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀭𑀸𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀤𑀺𑀝𑁆𑀞𑀺𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀭𑀸𑀕𑁄…  𑀤𑁄𑀲𑁄…  𑀫𑁄𑀳𑁄…  𑀤𑀺𑀝𑁆𑀞𑀺…  𑀧𑀢𑁆𑀣𑀦𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

991\. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁂𑀯𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀭𑀸𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀤𑁄𑀲𑀁…𑀧𑁂…  𑀧𑀢𑁆𑀣𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀭𑀸𑀕𑁄…  𑀤𑁄𑀲𑁄…  𑀫𑁄𑀳𑁄…  𑀤𑀺𑀝𑁆𑀞𑀺…  𑀧𑀢𑁆𑀣𑀦𑀸 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

(84.)

992\. 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀭𑀸𑀕𑁄…  𑀤𑁄𑀲𑁄…  𑀫𑁄𑀳𑁄…  𑀫𑀸𑀦𑁄…  𑀧𑀢𑁆𑀣𑀦𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀫𑀸𑀦𑀲𑁆𑀲…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

993\. 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀭𑀸𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀫𑀸𑀦𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀭𑀸𑀕𑁄…  𑀤𑁄𑀲𑁄…  𑀫𑁄𑀳𑁄…  𑀫𑀸𑀦𑁄…  𑀧𑀢𑁆𑀣𑀦𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀓𑀪𑀡𑁆𑀟𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀭𑀪𑀡𑁆𑀟𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀓𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀭𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

994\. 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁂𑀯𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀭𑀸𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀫𑀸𑀦𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀭𑀸𑀕𑁄…  𑀤𑁄𑀲𑁄…  𑀫𑁄𑀳𑁄…  𑀫𑀸𑀦𑁄…  𑀧𑀢𑁆𑀣𑀦𑀸 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(85.)

995\. 𑀦𑁂𑀯𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁂𑀯𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀲𑀻𑀮𑀁…  𑀲𑀼𑀢𑀁…  𑀘𑀸𑀕𑀁…  𑀧𑀜𑁆𑀜𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀲𑀤𑁆𑀥𑀸…  𑀲𑀻𑀮𑀁…  𑀲𑀼𑀢𑀁…  𑀘𑀸𑀕𑁄…  𑀧𑀜𑁆𑀜𑀸…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀤𑁆𑀥𑀸𑀬…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸𑀬 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

996\. 𑀦𑁂𑀯𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

997\. 𑀦𑁂𑀯𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀸…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀲𑁆𑀲 𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀫𑀸𑀦𑀲𑁆𑀲…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
