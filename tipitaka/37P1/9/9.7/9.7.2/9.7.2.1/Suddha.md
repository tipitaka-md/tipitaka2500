1048\. 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀲𑀳𑀚𑀸𑀢𑁂 𑀤𑀲, 𑀦𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀤𑀲, 𑀦𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀤𑀲, 𑀦𑁅𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀤𑁆𑀯𑀸𑀤𑀲, 𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀆𑀲𑁂𑀯𑀦𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀓𑀫𑁆𑀫𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀯𑀺𑀧𑀸𑀓𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀆𑀳𑀸𑀭𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑁃𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀛𑀸𑀦𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀫𑀕𑁆𑀕𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀤𑀲, 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀅𑀝𑁆𑀞, 𑀦𑁄𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀅𑀝𑁆𑀞, 𑀦𑁄𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀘𑀼𑀤𑁆𑀤𑀲, 𑀦𑁄𑀅𑀯𑀺𑀕𑀢𑁂 𑀅𑀝𑁆𑀞𑁇 (𑀏𑀯𑀁 𑀕𑀡𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
