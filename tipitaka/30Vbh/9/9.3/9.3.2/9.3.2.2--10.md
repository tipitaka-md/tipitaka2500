# 𑁯𑁇𑁩𑁇𑁨𑁇𑁨--𑁧𑁦 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓𑀸𑀤𑀺

* 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀤𑀼𑀓

* 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓𑀸𑀤𑀺

1550\. 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀲𑀗𑁆𑀔𑀢𑀸𑁇 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑁇 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀅𑀭𑀽𑀧𑀸𑁇 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 𑀦𑁄 𑀆𑀲𑀯𑀸𑁇 𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘𑀸𑀢𑀺𑀧𑀺, 𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸𑀢𑀺𑀧𑀺𑁇 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸𑀢𑀺𑀧𑀺, 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸𑀢𑀺𑀧𑀺𑁇 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀅𑀦𑀸𑀲𑀯𑀸𑁇

1551\. 𑀦𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀸…𑀧𑁂…  𑀦𑁄 𑀕𑀦𑁆𑀣𑀸…𑀧𑁂…  𑀦𑁄 𑀑𑀖𑀸…𑀧𑁂…  𑀦𑁄 𑀬𑁄𑀕𑀸…𑀧𑁂…  𑀦𑁄 𑀦𑀻𑀯𑀭𑀡𑀸…𑀧𑁂…  𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀸…𑀧𑁂…  𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀸, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑁄𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄”𑀢𑀺𑀧𑀺𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑁄”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁄”𑀢𑀺𑀧𑀺𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁄𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀦𑁄, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁄𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀦𑁄, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀢𑀬𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀩𑀸𑀳𑀺𑀭𑀸, 𑀘𑀺𑀢𑁆𑀢𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑁄𑁇
