# 𑁯𑁇𑁧𑁇𑁪 𑀯𑀻𑀫𑀁𑀲𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤

### 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀯𑀻𑀫𑀁𑀲𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤

(441.)

1482\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀺? 𑀯𑀻𑀫𑀁𑀲𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀺𑀧𑀢𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀮𑀪𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀁 𑀮𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺”𑁇 𑀲𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬…𑀧𑁂…  𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺, 𑀇𑀫𑁂 𑀘 𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀸; 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

(442.)

1483\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀻𑀫𑀁𑀲𑀸? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀫𑀁𑀲𑀸”𑁇

1484\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑀸𑀥𑀺? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑀸𑀥𑀺”𑁇

1485\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇 𑀇𑀢𑀺 𑀇𑀫𑀸𑀬 𑀘 𑀯𑀻𑀫𑀁𑀲𑀸𑀬, 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀲𑀫𑀸𑀥𑀺𑀦𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀫𑀁𑀲𑀸𑀲𑀫𑀸𑀥𑀺𑀧𑀥𑀸𑀦𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄”𑀢𑀺𑁇

(443.)

1486\. “𑀇𑀤𑁆𑀥𑀻”𑀢𑀺𑁇 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀇𑀤𑁆𑀥𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑀺 𑀇𑀚𑁆𑀛𑀦𑀸 𑀲𑀫𑀺𑀚𑁆𑀛𑀦𑀸 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

1487\. “𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁄”𑀢𑀺𑁇 𑀢𑀣𑀸𑀪𑀽𑀢𑀲𑁆𑀲 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

1488\. “𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀁 𑀪𑀸𑀯𑁂𑀢𑀻”𑀢𑀺𑁇

---
