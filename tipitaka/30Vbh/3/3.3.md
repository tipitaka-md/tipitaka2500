# 𑁩𑁇𑁩 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

(185.)

669\. 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼, 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀥𑀸𑀢𑀼, 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼, 𑀭𑀲𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀥𑀸𑀢𑀼, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇

### 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

(186.)

670\. 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀓𑀢𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸…𑀧𑁂…  𑀓𑀢𑀺 𑀲𑀭𑀡𑀸, 𑀓𑀢𑀺 𑀅𑀭𑀡𑀸?
