# 𑁩𑁇𑁨 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

(183.)

644\. 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼, 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀥𑀸𑀢𑀼, 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼, 𑀭𑀲𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀥𑀸𑀢𑀼, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇

(184.)

645\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀘𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄…𑀧𑁂…  𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼”𑁇 (𑁧)

646\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸…𑀧𑁂…  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼”𑁇 (𑁨)

647\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼? 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀽𑀧𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼”𑁇 (𑁩)

648\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑁄𑀢𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀲𑁄𑀢𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄…𑀧𑁂…  𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁄𑀢𑀥𑀸𑀢𑀼”𑁇 (𑁪)

649\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼? 𑀬𑁄 𑀲𑀤𑁆𑀤𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼”𑁇 (𑁫)

650\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼? 𑀲𑁄𑀢𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀤𑁆𑀤𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼”𑁇 (𑁬)

651\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀖𑀸𑀦𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀖𑀸𑀦𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄…𑀧𑁂…  𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀖𑀸𑀦𑀥𑀸𑀢𑀼”𑁇 (𑁭)

652\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼? 𑀬𑁄 𑀕𑀦𑁆𑀥𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼”𑁇 (𑁮)

653\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼? 𑀖𑀸𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀕𑀦𑁆𑀥𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼”𑁇 (𑁯)

654\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼? 𑀬𑀸 𑀚𑀺𑀯𑁆𑀳𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄…𑀧𑁂…  𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁦)

655\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀭𑀲𑀥𑀸𑀢𑀼? 𑀬𑁄 𑀭𑀲𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄…𑀧𑁂…  𑀭𑀲𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀲𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁧)

656\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼? 𑀚𑀺𑀯𑁆𑀳𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀭𑀲𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁨)

657\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀬𑀥𑀸𑀢𑀼? 𑀬𑁄 𑀓𑀸𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄…𑀧𑁂…  𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀬𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁩)

658\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼? 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁪)

659\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼? 𑀓𑀸𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁫)

660\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀥𑀸𑀢𑀼; 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀸 𑀧𑀦 𑀧𑀞𑀫𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀦𑁄𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁬)

661\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼𑁇

662\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄”𑁇 (𑁧)

663\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄”𑁇 (𑁨)

664\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑁇 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄”𑁇 (𑁩)

665\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 “𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁”𑁇 (𑁪)

666\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀥𑀸𑀢𑀼? 𑀭𑀸𑀕𑀓𑁆𑀔𑀬𑁄, 𑀤𑁄𑀲𑀓𑁆𑀔𑀬𑁄, 𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀥𑀸𑀢𑀼”𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼”𑁇 (𑁫--𑁧𑁭)

667\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼; 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀫𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼”𑁇 (𑁧𑁮)
