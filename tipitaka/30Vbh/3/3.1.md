# 𑁩𑁇𑁧 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

(172.)

613\. 𑀙 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼, 𑀆𑀧𑁄𑀥𑀸𑀢𑀼, 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼, 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇

(173.)

614\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼? 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼𑀤𑁆𑀯𑀬𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀔𑀭𑀺𑀕𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀪𑀸𑀯𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀦𑀔𑀸 𑀤𑀦𑁆𑀢𑀸 𑀢𑀘𑁄 𑀫𑀁𑀲𑀁 𑀦𑁆𑀳𑀸𑀭𑀼 𑀅𑀝𑁆𑀞𑀺 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀯𑀓𑁆𑀓𑀁 𑀳𑀤𑀬𑀁 𑀬𑀓𑀦𑀁 𑀓𑀺𑀮𑁄𑀫𑀓𑀁 𑀧𑀺𑀳𑀓𑀁 𑀧𑀧𑁆𑀨𑀸𑀲𑀁 𑀅𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀕𑀼𑀡𑀁 𑀉𑀤𑀭𑀺𑀬𑀁 𑀓𑀭𑀻𑀲𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀔𑀭𑀺𑀕𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀪𑀸𑀯𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼”𑁇

615\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑀸𑀳𑀺𑀭𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀔𑀭𑀺𑀕𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀪𑀸𑀯𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀅𑀬𑁄 𑀮𑁄𑀳𑀁 𑀢𑀺𑀧𑀼 𑀲𑀻𑀲𑀁 𑀲𑀚𑁆𑀛𑀁 𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀡𑀺 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑁄 𑀲𑀗𑁆𑀔𑁄 𑀲𑀺𑀮𑀸 𑀧𑀯𑀸𑀍𑀅𑀁 𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑁄 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀁 𑀢𑀺𑀡𑀁 𑀓𑀝𑁆𑀞𑀁 𑀲𑀓𑁆𑀔𑀭𑀸 𑀓𑀞𑀮𑀁 𑀪𑀽𑀫𑀺 𑀧𑀸𑀲𑀸𑀡𑁄 𑀧𑀩𑁆𑀩𑀢𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀔𑀭𑀺𑀕𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀪𑀸𑀯𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑀸𑀳𑀺𑀭𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼”𑁇 𑀬𑀸 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀬𑀸 𑀘 𑀩𑀸𑀳𑀺𑀭𑀸 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼”𑁇 (𑁧)

(174.)

616\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼? 𑀆𑀧𑁄𑀥𑀸𑀢𑀼𑀤𑁆𑀯𑀬𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀆𑀧𑁄 𑀆𑀧𑁄𑀕𑀢𑀁 𑀲𑀺𑀦𑁂𑀳𑁄 𑀲𑀺𑀦𑁂𑀳𑀕𑀢𑀁 𑀩𑀦𑁆𑀥𑀦𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀧𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀫𑁆𑀳𑀁 𑀧𑀼𑀩𑁆𑀩𑁄 𑀮𑁄𑀳𑀺𑀢𑀁 𑀲𑁂𑀤𑁄 𑀫𑁂𑀤𑁄 𑀅𑀲𑁆𑀲𑀼 𑀯𑀲𑀸 𑀔𑁂𑀍𑀑 𑀲𑀺𑀗𑁆𑀖𑀸𑀡𑀺𑀓𑀸 𑀮𑀲𑀺𑀓𑀸 𑀫𑀼𑀢𑁆𑀢𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀆𑀧𑁄 𑀆𑀧𑁄𑀕𑀢𑀁 𑀲𑀺𑀦𑁂𑀳𑁄 𑀲𑀺𑀦𑁂𑀳𑀕𑀢𑀁 𑀩𑀦𑁆𑀥𑀦𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼”𑁇

617\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑀸𑀳𑀺𑀭𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀆𑀧𑁄 𑀆𑀧𑁄𑀕𑀢𑀁 𑀲𑀺𑀦𑁂𑀳𑁄 𑀲𑀺𑀦𑁂𑀳𑀕𑀢𑀁 𑀩𑀦𑁆𑀥𑀦𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀫𑀽𑀮𑀭𑀲𑁄 𑀔𑀦𑁆𑀥𑀭𑀲𑁄 𑀢𑀘𑀭𑀲𑁄 𑀧𑀢𑁆𑀢𑀭𑀲𑁄 𑀧𑀼𑀧𑁆𑀨𑀭𑀲𑁄 𑀨𑀮𑀭𑀲𑁄 𑀔𑀻𑀭𑀁 𑀤𑀥𑀺 𑀲𑀧𑁆𑀧𑀺 𑀦𑀯𑀦𑀻𑀢𑀁 𑀢𑁂𑀮𑀁 𑀫𑀥𑀼 𑀨𑀸𑀡𑀺𑀢𑀁 𑀪𑀼𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀸 𑀉𑀤𑀓𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸𑀦𑀺 𑀯𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀁 𑀆𑀧𑁄 𑀆𑀧𑁄𑀕𑀢𑀁 𑀲𑀺𑀦𑁂𑀳𑁄 𑀲𑀺𑀦𑁂𑀳𑀕𑀢𑀁 𑀩𑀦𑁆𑀥𑀦𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑀸𑀳𑀺𑀭𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼”𑁇 𑀬𑀸 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼 𑀬𑀸 𑀘 𑀩𑀸𑀳𑀺𑀭𑀸 𑀆𑀧𑁄𑀥𑀸𑀢𑀼, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀧𑁄𑀥𑀸𑀢𑀼”𑁇 (𑁨)

(175.)

618\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼? 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀤𑁆𑀯𑀬𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀢𑁂𑀚𑁄 𑀢𑁂𑀚𑁄𑀕𑀢𑀁 𑀉𑀲𑁆𑀫𑀸 𑀉𑀲𑁆𑀫𑀸𑀕𑀢𑀁 𑀉𑀲𑀼𑀫𑀁 𑀉𑀲𑀼𑀫𑀕𑀢𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀬𑁂𑀦 𑀘 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀬𑁂𑀦 𑀘 𑀚𑀻𑀭𑀻𑀬𑀢𑀺 𑀬𑁂𑀦 𑀘 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀢𑀺 𑀬𑁂𑀦 𑀘 𑀅𑀲𑀺𑀢𑀧𑀻𑀢𑀔𑀸𑀬𑀺𑀢𑀲𑀸𑀬𑀺𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀢𑁂𑀚𑁄 𑀢𑁂𑀚𑁄𑀕𑀢𑀁 𑀉𑀲𑁆𑀫𑀸 𑀉𑀲𑁆𑀫𑀸𑀕𑀢𑀁 𑀉𑀲𑀼𑀫𑀁 𑀉𑀲𑀼𑀫𑀕𑀢𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼”𑁇

619\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑀸𑀳𑀺𑀭𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀢𑁂𑀚𑁄 𑀢𑁂𑀚𑁄𑀕𑀢𑀁 𑀉𑀲𑁆𑀫𑀸 𑀉𑀲𑁆𑀫𑀸𑀕𑀢𑀁 𑀉𑀲𑀼𑀫𑀁 𑀉𑀲𑀼𑀫𑀕𑀢𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀝𑁆𑀞𑀕𑁆𑀕𑀺 𑀧𑀮𑀸𑀮𑀕𑁆𑀕𑀺 𑀢𑀺𑀡𑀕𑁆𑀕𑀺 𑀕𑁄𑀫𑀬𑀕𑁆𑀕𑀺 𑀣𑀼𑀲𑀕𑁆𑀕𑀺 𑀲𑀗𑁆𑀓𑀸𑀭𑀕𑁆𑀕𑀺 𑀇𑀦𑁆𑀤𑀕𑁆𑀕𑀺 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑁄 𑀲𑀽𑀭𑀺𑀬𑀲𑀦𑁆𑀢𑀸𑀧𑁄 𑀓𑀝𑁆𑀞𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀲𑀦𑁆𑀢𑀸𑀧𑁄 𑀢𑀺𑀡𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀲𑀦𑁆𑀢𑀸𑀧𑁄 𑀥𑀜𑁆𑀜𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀲𑀦𑁆𑀢𑀸𑀧𑁄 𑀪𑀡𑁆𑀟𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀲𑀦𑁆𑀢𑀸𑀧𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀁 𑀢𑁂𑀚𑁄 𑀢𑁂𑀚𑁄𑀕𑀢𑀁 𑀉𑀲𑁆𑀫𑀸 𑀉𑀲𑁆𑀫𑀸𑀕𑀢𑀁 𑀉𑀲𑀼𑀫𑀁 𑀉𑀲𑀼𑀫𑀕𑀢𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑀸𑀳𑀺𑀭𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼”𑁇 𑀬𑀸 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼 𑀬𑀸 𑀘 𑀩𑀸𑀳𑀺𑀭𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼”𑁇 (𑁩)

(176.)

620\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼? 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼𑀤𑁆𑀯𑀬𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑀸𑀬𑁄 𑀯𑀸𑀬𑁄𑀕𑀢𑀁 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀉𑀤𑁆𑀥𑀗𑁆𑀕𑀫𑀸 𑀯𑀸𑀢𑀸 𑀅𑀥𑁄𑀕𑀫𑀸 𑀯𑀸𑀢𑀸 𑀓𑀼𑀘𑁆𑀙𑀺𑀲𑀬𑀸 𑀯𑀸𑀢𑀸 𑀓𑁄𑀝𑁆𑀞𑀸𑀲𑀬𑀸 𑀯𑀸𑀢𑀸 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄 𑀯𑀸𑀢𑀸 𑀲𑀢𑁆𑀣𑀓𑀯𑀸𑀢𑀸 𑀔𑀼𑀭𑀓𑀯𑀸𑀢𑀸 𑀉𑀧𑁆𑀧𑀮𑀓𑀯𑀸𑀢𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑁄 𑀧𑀲𑁆𑀲𑀸𑀲𑁄 𑀇𑀢𑀺 𑀯𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑀸𑀬𑁄 𑀯𑀸𑀬𑁄𑀕𑀢𑀁 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼”𑁇

621\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑀸𑀳𑀺𑀭𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀯𑀸𑀬𑁄 𑀯𑀸𑀬𑁄𑀕𑀢𑀁 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸 𑀯𑀸𑀢𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀯𑀸𑀢𑀸 𑀉𑀢𑁆𑀢𑀭𑀸 𑀯𑀸𑀢𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀯𑀸𑀢𑀸 𑀲𑀭𑀚𑀸 𑀯𑀸𑀢𑀸 𑀅𑀭𑀚𑀸 𑀯𑀸𑀢𑀸 𑀲𑀻𑀢𑀸 𑀯𑀸𑀢𑀸 𑀉𑀡𑁆𑀳𑀸 𑀯𑀸𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀯𑀸𑀢𑀸 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀯𑀸𑀢𑀸 𑀓𑀸𑀍𑀅𑀯𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑁆𑀪𑀯𑀸𑀢𑀸 𑀧𑀓𑁆𑀔𑀯𑀸𑀢𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀯𑀸𑀢𑀸 𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀯𑀸𑀢𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑀯𑀸𑀢𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀁 𑀯𑀸𑀬𑁄 𑀯𑀸𑀬𑁄𑀕𑀢𑀁 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑁆𑀲 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑀸𑀳𑀺𑀭𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼”𑁇 𑀬𑀸 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼 𑀬𑀸 𑀘 𑀩𑀸𑀳𑀺𑀭𑀸 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼”𑁇 (𑁪)

(177.)

622\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼? 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼𑀤𑁆𑀯𑀬𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀆𑀓𑀸𑀲𑁄 𑀆𑀓𑀸𑀲𑀕𑀢𑀁 𑀅𑀖𑀁 𑀅𑀖𑀕𑀢𑀁 𑀯𑀺𑀯𑀭𑁄 𑀯𑀺𑀯𑀭𑀕𑀢𑀁 𑀅𑀲𑀫𑁆𑀨𑀼𑀝𑁆𑀞𑀁 𑀫𑀁𑀲𑀮𑁄𑀳𑀺𑀢𑁂𑀳𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀡𑁆𑀡𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀁 𑀦𑀸𑀲𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀁 𑀫𑀼𑀔𑀤𑁆𑀯𑀸𑀭𑀁, 𑀬𑁂𑀦 𑀘 𑀅𑀲𑀺𑀢𑀧𑀻𑀢𑀔𑀸𑀬𑀺𑀢𑀲𑀸𑀬𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑁄𑀳𑀭𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀘 𑀅𑀲𑀺𑀢𑀧𑀻𑀢𑀔𑀸𑀬𑀺𑀢𑀲𑀸𑀬𑀺𑀢𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀬𑁂𑀦 𑀘 𑀅𑀲𑀺𑀢𑀧𑀻𑀢𑀔𑀸𑀬𑀺𑀢𑀲𑀸𑀬𑀺𑀢𑀁 𑀅𑀥𑁄𑀪𑀸𑀕𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀢𑀺, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀆𑀓𑀸𑀲𑁄 𑀆𑀓𑀸𑀲𑀕𑀢𑀁 𑀅𑀖𑀁 𑀅𑀖𑀕𑀢𑀁 𑀯𑀺𑀯𑀭𑁄 𑀯𑀺𑀯𑀭𑀕𑀢𑀁 𑀅𑀲𑀫𑁆𑀨𑀼𑀝𑁆𑀞𑀁 𑀫𑀁𑀲𑀮𑁄𑀳𑀺𑀢𑁂𑀳𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼”𑁇

623\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑀸𑀳𑀺𑀭𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀆𑀓𑀸𑀲𑁄 𑀆𑀓𑀸𑀲𑀕𑀢𑀁 𑀅𑀖𑀁 𑀅𑀖𑀕𑀢𑀁 𑀯𑀺𑀯𑀭𑁄 𑀯𑀺𑀯𑀭𑀕𑀢𑀁 𑀅𑀲𑀫𑁆𑀨𑀼𑀝𑁆𑀞𑀁 𑀘𑀢𑀽𑀳𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑁂𑀳𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑀸𑀳𑀺𑀭𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼”𑁇 𑀬𑀸 𑀘 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼 𑀬𑀸 𑀘 𑀩𑀸𑀳𑀺𑀭𑀸 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼”𑁇 (𑁫)

(178.)

624\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼”𑁇 (𑁬)

625\. 𑀇𑀫𑀸 𑀙 𑀥𑀸𑀢𑀼𑀬𑁄𑁇

626\. 𑀅𑀧𑀭𑀸𑀧𑀺 𑀙 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀲𑀼𑀔𑀥𑀸𑀢𑀼, 𑀤𑀼𑀓𑁆𑀔𑀥𑀸𑀢𑀼, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀥𑀸𑀢𑀼, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀥𑀸𑀢𑀼, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀥𑀸𑀢𑀼, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀥𑀸𑀢𑀼𑁇

(180.)

627\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀼𑀔𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀥𑀸𑀢𑀼”𑁇 (𑁧)

628\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀥𑀸𑀢𑀼”𑁇 (𑁨)

629\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀥𑀸𑀢𑀼”𑁇 (𑁩)

630\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀥𑀸𑀢𑀼”𑁇 (𑁪)

631\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀦𑁂𑀯 𑀲𑀸𑀢𑀁 𑀦𑀸𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀥𑀸𑀢𑀼”𑁇 (𑁫)

632\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀥𑀸𑀢𑀼? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑁄 𑀅𑀲𑀫𑁆𑀩𑁄𑀥𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀅𑀲𑀗𑁆𑀕𑀸𑀳𑀡𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑀡𑀸 𑀅𑀲𑀫𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀡𑀸 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀫𑁆𑀫𑁂𑀚𑁆𑀛𑀁 𑀩𑀸𑀮𑁆𑀬𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀫𑁄𑀳𑁄 𑀧𑀫𑁄𑀳𑁄 𑀲𑀫𑁆𑀫𑁄𑀳𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑁄𑀖𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬𑁄𑀕𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀥𑀸𑀢𑀼”𑁇 (𑁬)

634\. 𑀅𑀧𑀭𑀸𑀧𑀺 𑀙 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀓𑀸𑀫𑀥𑀸𑀢𑀼, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀥𑀸𑀢𑀼, 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼, 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼𑁇

(182.)

635\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼? 𑀓𑀸𑀫𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀧𑁆𑀧𑀦𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀦𑀸 𑀘𑁂𑀢𑀲𑁄 𑀅𑀪𑀺𑀦𑀺𑀭𑁄𑀧𑀦𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑁇 𑀳𑁂𑀝𑁆𑀞𑀢𑁄 𑀅𑀯𑀻𑀘𑀺𑀦𑀺𑀭𑀬𑀁 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀭𑀺𑀢𑁄 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀤𑁂𑀯𑁂 𑀅𑀦𑁆𑀢𑁄 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑀁 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑁂 𑀏𑀢𑁆𑀣𑀸𑀯𑀘𑀭𑀸 𑀏𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀔𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑀆𑀬𑀢𑀦𑀸 𑀭𑀽𑀧𑀸 𑀯𑁂𑀤𑀦𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼”𑁇 (𑁧)

636\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼? 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼”𑁇 𑀤𑀲𑀲𑀼 𑀯𑀸 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀓𑁄𑀧𑁄 𑀧𑀓𑁄𑀧𑁄 𑀲𑀫𑁆𑀧𑀓𑁄𑀧𑁄 𑀤𑁄𑀲𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑀫𑁆𑀧𑀤𑁄𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑁄 𑀓𑁄𑀥𑁄 𑀓𑀼𑀚𑁆𑀛𑀦𑀸 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀢𑁆𑀢𑀁 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼”𑁇 (𑁨)

637\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼? 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼”𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀸𑀡𑀺𑀦𑀸 𑀯𑀸 𑀮𑁂𑀟𑁆𑀟𑀼𑀦𑀸 𑀯𑀸 𑀤𑀡𑁆𑀟𑁂𑀦 𑀯𑀸 𑀲𑀢𑁆𑀣𑁂𑀦 𑀯𑀸 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀯𑀺𑀳𑁂𑀞𑁂𑀢𑀺, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀳𑁂𑀞𑀦𑀸 𑀯𑀺𑀳𑁂𑀞𑀦𑀸 𑀳𑀺𑀁𑀲𑀦𑀸 𑀯𑀺𑀳𑀺𑀁𑀲𑀦𑀸 𑀭𑁄𑀲𑀦𑀸 𑀯𑀺𑀭𑁄𑀲𑀦𑀸 𑀧𑀭𑀽𑀧𑀖𑀸𑀢𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼”𑁇 (𑁩)

638\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀥𑀸𑀢𑀼? 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀥𑀸𑀢𑀼”𑁇 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 “𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀥𑀸𑀢𑀼”𑁇 (𑁪)

639\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼? 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼”𑁇 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀫𑁂𑀢𑁆𑀢𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀦𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑁂𑀢𑁆𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼”𑁇 (𑁫)

640\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼? 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀧𑁆𑀧𑀦𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀦𑀸 𑀘𑁂𑀢𑀲𑁄 𑀅𑀪𑀺𑀦𑀺𑀭𑁄𑀧𑀦𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼”𑁇 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀓𑀭𑀼𑀡𑀸 𑀓𑀭𑀼𑀡𑀸𑀬𑀦𑀸 𑀓𑀭𑀼𑀡𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀓𑀭𑀼𑀡𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼”𑁇 (𑁬)
