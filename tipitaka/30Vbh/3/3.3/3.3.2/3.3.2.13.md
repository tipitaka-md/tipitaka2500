# 𑁩𑁇𑁩𑁇𑁨𑁇𑁧𑁩 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

709\. 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇

710\. 𑀧𑀦𑁆𑀦𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 𑀧𑀦𑁆𑀦𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀲𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇 𑀧𑀦𑁆𑀦𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀧𑀜𑁆𑀘 𑀥𑀸𑀢𑀼𑀬𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇

711\. 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑁅𑀢𑁆𑀢𑀭𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑁅𑀢𑁆𑀢𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀭𑀡𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀭𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀡𑀸𑀢𑀺𑁇 (𑁧𑁩)
