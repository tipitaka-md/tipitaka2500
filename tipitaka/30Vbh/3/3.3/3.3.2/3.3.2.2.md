# 𑁩𑁇𑁩𑁇𑁨𑁇𑁨 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

* 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀤𑀼𑀓

* 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

694\. 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀗𑁆𑀔𑀢𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀲𑀗𑁆𑀔𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀸𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑁇 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀅𑀝𑁆𑀞 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀭𑀽𑀧𑀸𑁇 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀭𑀽𑀧𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸; 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇
