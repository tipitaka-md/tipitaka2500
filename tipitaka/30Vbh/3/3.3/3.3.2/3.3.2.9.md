# 𑁩𑁇𑁩𑁇𑁨𑁇𑁯 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀤𑀼𑀓

* 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

700\. 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑁄, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑀸”𑀢𑀺 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄”𑀢𑀺𑀧𑀺𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺𑁇
