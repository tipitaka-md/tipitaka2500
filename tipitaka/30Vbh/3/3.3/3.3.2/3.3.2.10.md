# 𑁩𑁇𑁩𑁇𑁨𑁇𑁧𑁦 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

### 𑀥𑀸𑀢𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

701\. 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑁇 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺𑁇

702\. 𑀤𑁆𑀯𑀸𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀙 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀻, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀻𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽𑁇 𑀲𑀢𑁆𑀢𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀻, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀻𑁇

703\. 𑀤𑁆𑀯𑀸𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸𑁇 𑀙 𑀥𑀸𑀢𑀼𑀬𑁄 𑀩𑀸𑀳𑀺𑀭𑀸𑁇

704\. 𑀦𑀯 𑀥𑀸𑀢𑀼𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀅𑀝𑁆𑀞 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑁇 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑁇 𑀲𑀢𑁆𑀢𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑁇
