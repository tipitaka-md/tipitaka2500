# 𑁪𑁇𑁧𑁇𑁨 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘

### 𑀲𑀘𑁆𑀘𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘

(203.)

730\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁? 𑀬𑀸𑀬𑀁 𑀢𑀡𑁆𑀳𑀸 𑀧𑁄𑀦𑁄𑀪𑀯𑀺𑀓𑀸 𑀦𑀦𑁆𑀤𑀺𑀭𑀸𑀕𑀲𑀳𑀕𑀢𑀸 𑀢𑀢𑁆𑀭𑀢𑀢𑁆𑀭𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀦𑀻, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸, 𑀪𑀯𑀢𑀡𑁆𑀳𑀸, 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑁇

731\. 𑀲𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀓𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺? 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

732\. 𑀓𑀺𑀜𑁆𑀘 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀼𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁𑁇 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑁄𑀢𑀁 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀖𑀸𑀦𑀁 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑁄 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

733\. 𑀭𑀽𑀧𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀸 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

734\. 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

735\. 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

736\. 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

737\. 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

738\. 𑀭𑀽𑀧𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

739\. 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

740\. 𑀭𑀽𑀧𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀭𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂…  𑀥𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

741\. 𑀭𑀽𑀧𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀲𑀤𑁆𑀤𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀕𑀦𑁆𑀥𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀭𑀲𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁, 𑀏𑀢𑁆𑀣𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀏𑀢𑁆𑀣 𑀦𑀺𑀯𑀺𑀲𑀫𑀸𑀦𑀸 𑀦𑀺𑀯𑀺𑀲𑀢𑀺𑁇

---
