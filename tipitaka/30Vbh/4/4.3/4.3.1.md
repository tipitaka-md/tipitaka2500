# 𑁪𑁇𑁩𑁇𑁧 𑀢𑀺𑀓

### 𑀲𑀘𑁆𑀘𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀢𑀺𑀓

(217.)

822\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀓𑀼𑀲𑀮𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 (𑁧)

823\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁨)

824\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁𑁇 (𑁩)

825\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇 (𑁪)

826\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁𑁇 (𑁫)

827\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀁𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁬)

828\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁭)

829\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁮)

830\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁𑁇 (𑁯)

831\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀆𑀘𑀬𑀕𑀸𑀫𑀺, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑁇 (𑁧𑁦)

832\. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀲𑁂𑀓𑁆𑀔𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 (𑁧𑁧)

833\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀁𑁇 (𑁧𑁨)

834\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁩)

835\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀳𑀻𑀦𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀳𑀻𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀁𑁇 (𑁧𑁪)

836\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀺𑀬𑀢𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 (𑁧𑁫)

837\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻”𑀢𑀺𑀧𑀺𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻”𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀦 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁬)

838\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄”𑀢𑀺𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀉𑀧𑁆𑀧𑀸𑀤𑀻”𑀢𑀺𑀧𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑁇 (𑁧𑁭)

839\. 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀅𑀢𑀻𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁮)

840\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁯)

841\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁨𑁦)

842\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁨𑁧)

843\. 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 (𑁨𑁨)
