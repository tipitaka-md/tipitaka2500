# 𑁪𑁇𑁩𑁇𑁨𑁇𑁧𑁦 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

### 𑀲𑀘𑁆𑀘𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

851\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀁, 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇

852\. 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀩𑀸𑀳𑀺𑀭𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇
