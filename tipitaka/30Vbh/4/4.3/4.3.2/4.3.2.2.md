### 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

845\. 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀲𑀗𑁆𑀔𑀢𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀭𑀽𑀧𑀸𑀦𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸; 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇
