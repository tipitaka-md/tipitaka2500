# 𑁪𑁇𑁩𑁇𑁨𑁇𑁧𑁧 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀲𑀘𑁆𑀘𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

853. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸”𑀢𑀺𑀧𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸”𑀢𑀺𑀧𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺𑁇
