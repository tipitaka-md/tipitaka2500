# 𑁪𑁇𑁨 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀲𑀘𑁆𑀘𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

(206.)

767\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸𑁇

768\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

769\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀸𑀲𑀯𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

770\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

771\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇

772\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

773\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄? 𑀬𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄”𑁇

774\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸? 𑀬𑀸 𑀘𑀢𑀽𑀳𑀺 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀳𑀺 𑀆𑀭𑀢𑀺 𑀯𑀺𑀭𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀺 𑀯𑁂𑀭𑀫𑀡𑀻 𑀅𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀓𑀭𑀡𑀁 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑁂𑀮𑀸𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸”𑁇

775\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄? 𑀬𑀸 𑀢𑀻𑀳𑀺 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀳𑀺 𑀆𑀭𑀢𑀺 𑀯𑀺𑀭𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀺 𑀯𑁂𑀭𑀫𑀡𑀻 𑀅𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀓𑀭𑀡𑀁 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑁂𑀮𑀸𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄”𑁇

776\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄? 𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀸 𑀆𑀚𑀻𑀯𑀸 𑀆𑀭𑀢𑀺 𑀯𑀺𑀭𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀺 𑀯𑁂𑀭𑀫𑀡𑀻 𑀅𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀓𑀭𑀡𑀁 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑁂𑀮𑀸𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄”𑁇

777\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 …𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄”𑁇

778\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺”𑁇

779\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺 𑀲𑀡𑁆𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺”𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

780\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

781\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀸𑀲𑀯𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

782\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

783\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

784\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸 𑀘 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

785\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀸𑀲𑀯𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

786\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

789\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀸𑀲𑀯𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

790\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀢𑀺𑀡𑁆𑀡𑀜𑁆𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀁 𑀲𑀸𑀲𑀯𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

791\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(210.)

792\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

793\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

794\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀢𑀺𑀡𑁆𑀡𑀜𑁆𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀁 𑀲𑀸𑀲𑀯𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀲𑀸𑀲𑀯𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

795\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(211.)

796\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸𑁇

797\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

798\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀸𑀲𑀯𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

799\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

800\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇

801\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

802\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄? 𑀬𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄”𑁇

803\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄”𑁇

804\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺”𑁇

805\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺”𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(212.)

806\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

807\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

808\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀢𑀺𑀡𑁆𑀡𑀜𑁆𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀁 𑀲𑀸𑀲𑀯𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀲𑀸𑀲𑀯𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

809\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(213.)

810\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸𑁇

811\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

812\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀲𑀸𑀲𑀯𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

813\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

814\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇

(214.)

815\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄? 𑀢𑀡𑁆𑀳𑀸 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀢𑀻𑀡𑀺 𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 𑀲𑀸𑀲𑀯𑀸𑀦𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀘 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑁇

816\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦𑀸𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

817\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄? 𑀢𑀡𑁆𑀳𑀸𑀬 𑀘, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀢𑀺𑀡𑁆𑀡𑀜𑁆𑀘 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀁 𑀲𑀸𑀲𑀯𑀸𑀦𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀜𑁆𑀘 𑀲𑀸𑀲𑀯𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑁇

818\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑁇
