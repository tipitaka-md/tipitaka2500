# 𑁧𑁫𑁇𑁧𑁇𑁧 𑀲𑀗𑁆𑀕𑀳𑀯𑀸𑀭

* 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀲𑀗𑁆𑀕𑀳𑀯𑀸𑀭

(718.)

2118\. 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀅𑀢𑁆𑀣𑁂 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀥𑀫𑁆𑀫𑁂 𑀜𑀸𑀡𑀁 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀅𑀬𑀁 𑀲𑀗𑁆𑀕𑀳𑀯𑀸𑀭𑁄𑁇
