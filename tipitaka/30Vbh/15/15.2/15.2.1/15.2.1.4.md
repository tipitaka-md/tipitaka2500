# 𑁧𑁫𑁇𑁨𑁇𑁧𑁇𑁪 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭

* 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀓𑀼𑀲𑀮𑀯𑀸𑀭

* 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭

(729.)

2135\. 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

2136\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀇𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀢𑁂𑀲𑀁 𑀯𑀺𑀧𑀸𑀓𑁂 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀢𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀢𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀇𑀤𑀫𑀢𑁆𑀣𑀚𑁄𑀢𑀓𑀸𑀦𑀻”𑀢𑀺, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇
