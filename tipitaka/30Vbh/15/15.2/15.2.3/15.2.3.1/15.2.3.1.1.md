# 𑁧𑁫𑁇𑁨𑁇𑁩𑁇𑁧𑁇𑁧 𑀅𑀳𑁂𑀢𑀼𑀓𑀓𑀼𑀲𑀮

* 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀯𑀺𑀧𑀸𑀓𑀯𑀸𑀭

* 𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑀸𑀓

* 𑀅𑀳𑁂𑀢𑀼𑀓𑀓𑀼𑀲𑀮

(732.)

2141\. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

2142\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀇𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀢𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀢𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀇𑀤𑀫𑀢𑁆𑀣𑀚𑁄𑀢𑀓𑀸𑀦𑀻”𑀢𑀺, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

(733.)

2143\. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

2144\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀤𑁆𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀕𑀦𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀭𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀇𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀢𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀢𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀇𑀤𑀫𑀢𑁆𑀣𑀚𑁄𑀢𑀓𑀸𑀦𑀻”𑀢𑀺, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

(734.)

2145\. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

2146\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀇𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀢𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀢𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀇𑀤𑀫𑀢𑁆𑀣𑀚𑁄𑀢𑀓𑀸𑀦𑀻”𑀢𑀺, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

(735.)

2147\. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

2148\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀸 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀸 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀇𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀢𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀢𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀇𑀤𑀫𑀢𑁆𑀣𑀚𑁄𑀢𑀓𑀸𑀦𑀻”𑀢𑀺, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

(736.)

2149\. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

2150\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀸 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀜𑁆𑀜𑀸 𑀳𑁄𑀢𑀺, 𑀘𑁂𑀢𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀯𑀸 𑀧𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀇𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀢𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀢𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀇𑀤𑀫𑀢𑁆𑀣𑀚𑁄𑀢𑀓𑀸𑀦𑀻”𑀢𑀺, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇
