# 𑁧𑁫𑁇𑁨𑁇𑁩𑁇𑁧𑁇𑁩 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭

### 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀯𑀺𑀧𑀸𑀓𑀯𑀸𑀭

### 𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑀸𑀓

### 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭

(739.)

2155. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

2156. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀇𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀢𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀢𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀜𑀸𑀡𑀸𑀦𑀺 𑀇𑀤𑀫𑀢𑁆𑀣𑀚𑁄𑀢𑀓𑀸𑀦𑀻”𑀢𑀺, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇
