# 𑁧𑁫𑁇𑁩𑁇𑁨𑁇𑁧𑁧--𑁧𑁩 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

### 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

2200. 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀸…𑀧𑁂…  𑀦𑁄 𑀓𑀺𑀮𑁂𑀲𑀸…𑀧𑁂…  𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸; 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸; 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸𑁇 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑁇 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸; 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸; 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀅𑀦𑀺𑀬𑀢𑀸; 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀺𑀬𑀸 𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑁅𑀢𑁆𑀢𑀭𑀸; 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀲𑀺𑀬𑀸 𑀲𑁅𑀢𑁆𑀢𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀅𑀭𑀡𑀸𑀢𑀺𑁇
