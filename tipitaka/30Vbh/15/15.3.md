# 𑁧𑁫𑁇𑁩 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

(747.)

2171\. 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸—  𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇

* 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

(748.)

2172\. 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀦𑀁 𑀓𑀢𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸…𑀧𑁂…  𑀓𑀢𑀺 𑀲𑀭𑀡𑀸, 𑀓𑀢𑀺 𑀅𑀭𑀡𑀸?
