# 𑁧𑁩𑁇𑁧𑁇𑁩 𑀫𑀼𑀤𑀺𑀢𑀸

* 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀫𑀼𑀤𑀺𑀢𑀸

(663.)

1963\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀏𑀓𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑁄 𑀅𑀲𑁆𑀲; 𑀏𑀯𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑁂 𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀨𑀭𑀢𑀺𑁇

1964\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀼𑀤𑀺𑀢𑀸? 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀫𑀼𑀤𑀺𑀢𑀸 𑀫𑀼𑀤𑀺𑀢𑀸𑀬𑀦𑀸 𑀫𑀼𑀤𑀺𑀢𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑀼𑀤𑀺𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀼𑀤𑀺𑀢𑀸”𑁇

1965\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀲𑀳𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀳𑀚𑀸𑀢𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇

(664.)

1966\. “𑀏𑀓𑀁 𑀤𑀺𑀲𑀦𑁆”𑀢𑀺 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀉𑀤𑁆𑀥𑀁 𑀯𑀸 𑀅𑀥𑁄 𑀯𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀯𑀸 𑀯𑀺𑀤𑀺𑀲𑀁 𑀯𑀸𑁇

(665.)

1967\. “𑀨𑀭𑀺𑀢𑁆𑀯𑀸”𑀢𑀺 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸𑁇

(666.)

1968\. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(667.)

1969\. “𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀦𑁆”𑀢𑀺 𑀬𑀣𑁂𑀯 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀢𑀣𑀸 𑀯𑀺𑀤𑀺𑀲𑀁𑁇

(668.)

1970\. “𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀦𑁆”𑀢𑀺 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀅𑀲𑁂𑀲𑀁 𑀦𑀺𑀲𑁆𑀲𑁂𑀲𑀁𑁇 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬𑀯𑀘𑀦𑀫𑁂𑀢𑀁—  “𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀦𑁆”𑀢𑀺𑁇

(669.)

1971\. “𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀼𑀤𑀺𑀢𑀸? 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀫𑀼𑀤𑀺𑀢𑀸 𑀫𑀼𑀤𑀺𑀢𑀸𑀬𑀦𑀸 𑀫𑀼𑀤𑀺𑀢𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑀼𑀤𑀺𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀼𑀤𑀺𑀢𑀸”𑁇

1972\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀲𑀳𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀳𑀚𑀸𑀢𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇

(670.)

1973\. “𑀯𑀺𑀧𑀼𑀮𑁂𑀦𑀸”𑀢𑀺 𑀬𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀢𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀁, 𑀬𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁, 𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀲𑁄 𑀅𑀯𑁂𑀭𑁄, 𑀬𑁄 𑀅𑀯𑁂𑀭𑁄 𑀲𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄𑁇

(671.)

1974\. “𑀨𑀭𑀺𑀢𑁆𑀯𑀸”𑀢𑀺 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸𑁇

(672.)

1975\. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇
