# 𑁧𑁩𑁇𑁧𑁇𑁪 𑀉𑀧𑁂𑀓𑁆𑀔𑀸

### 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀉𑀧𑁂𑀓𑁆𑀔𑀸

(673.)

1976\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀏𑀓𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀦𑁂𑀯 𑀫𑀦𑀸𑀧𑀁 𑀦 𑀅𑀫𑀦𑀸𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀅𑀲𑁆𑀲; 𑀏𑀯𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑁂 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀨𑀭𑀢𑀺𑁇

1977\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬𑀦𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸”𑁇

1978\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀲𑀳𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀳𑀚𑀸𑀢𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇

(674.)

1979\. “𑀏𑀓𑀁 𑀤𑀺𑀲𑀦𑁆”𑀢𑀺 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀉𑀤𑁆𑀥𑀁 𑀯𑀸 𑀅𑀥𑁄 𑀯𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀯𑀸 𑀯𑀺𑀤𑀺𑀲𑀁 𑀯𑀸𑁇

(675.)

1980\. “𑀨𑀭𑀺𑀢𑁆𑀯𑀸”𑀢𑀺 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸𑁇

(676.)

1981\. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(677.)

1982\. “𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀦𑁆”𑀢𑀺 𑀬𑀣𑁂𑀯 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀢𑀣𑀸 𑀯𑀺𑀤𑀺𑀲𑀁𑁇

(678.)

1983\. “𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀦𑁆”𑀢𑀺 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀅𑀲𑁂𑀲𑀁 𑀦𑀺𑀲𑁆𑀲𑁂𑀲𑀁𑁇 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬𑀯𑀘𑀦𑀫𑁂𑀢𑀁—  “𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀦𑁆”𑀢𑀺𑁇

(679.)

1984\. “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬𑀦𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸”𑁇

1985\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀲𑀳𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀳𑀚𑀸𑀢𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇

(680.)

1986\. “𑀯𑀺𑀧𑀼𑀮𑁂𑀦𑀸”𑀢𑀺 𑀬𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀢𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀁, 𑀬𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁, 𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀲𑁄 𑀅𑀯𑁂𑀭𑁄, 𑀬𑁄 𑀅𑀯𑁂𑀭𑁄 𑀲𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄𑁇

(681.)

1987\. “𑀨𑀭𑀺𑀢𑁆𑀯𑀸”𑀢𑀺 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸𑁇

(682.)

1988\. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

---
