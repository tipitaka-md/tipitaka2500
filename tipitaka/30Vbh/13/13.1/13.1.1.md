# 𑁧𑁩𑁇𑁧𑁇𑁧 𑀫𑁂𑀢𑁆𑀢𑀸

* 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀫𑁂𑀢𑁆𑀢𑀸

(643.)

1937\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀏𑀓𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀺𑀬𑀁 𑀫𑀦𑀸𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑁂 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀨𑀭𑀢𑀺𑁇

1938\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑁂𑀢𑁆𑀢𑀸? 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀫𑁂𑀢𑁆𑀢𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀦𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑁂𑀢𑁆𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑁂𑀢𑁆𑀢𑀸”𑁇

1939\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀲𑀳𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀳𑀚𑀸𑀢𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇

(644.)

1940\. “𑀏𑀓𑀁 𑀤𑀺𑀲𑀦𑁆”𑀢𑀺 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀸 𑀤𑀺𑀲𑀁 𑀉𑀤𑁆𑀥𑀁 𑀯𑀸 𑀅𑀥𑁄 𑀯𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀯𑀸 𑀯𑀺𑀤𑀺𑀲𑀁 𑀯𑀸𑁇

(645.)

1941\. “𑀨𑀭𑀺𑀢𑁆𑀯𑀸”𑀢𑀺 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸𑁇

(646.)

1942\. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺 𑀇𑀭𑀺𑀬𑀢𑀺 𑀯𑀢𑁆𑀢𑀢𑀺 𑀧𑀸𑀮𑁂𑀢𑀺 𑀬𑀧𑁂𑀢𑀺 𑀬𑀸𑀧𑁂𑀢𑀺 𑀘𑀭𑀢𑀺 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(647.)

1943\. “𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀦𑁆”𑀢𑀺 𑀬𑀣𑁂𑀯 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀤𑀺𑀲𑀁 𑀢𑀣𑀸 𑀉𑀤𑁆𑀥𑀁 𑀢𑀣𑀸 𑀅𑀥𑁄 𑀢𑀣𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀢𑀣𑀸 𑀯𑀺𑀤𑀺𑀲𑀁𑁇

(648.)

1944\. “𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀦𑁆”𑀢𑀺 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀅𑀲𑁂𑀲𑀁 𑀦𑀺𑀲𑁆𑀲𑁂𑀲𑀁𑁇 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬𑀯𑀘𑀦𑀫𑁂𑀢𑀁—  “𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀦𑁆”𑀢𑀺𑁇

(649.)

1945\. “𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑁂𑀢𑁆𑀢𑀸? 𑀬𑀸 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀫𑁂𑀢𑁆𑀢𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀦𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑁂𑀢𑁆𑀢𑀸𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑁂𑀢𑁆𑀢𑀸”𑁇

1946\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀲𑀳𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀳𑀚𑀸𑀢𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇

(650.)

1947\. “𑀯𑀺𑀧𑀼𑀮𑁂𑀦𑀸”𑀢𑀺 𑀬𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀢𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀁, 𑀬𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁, 𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀲𑁄 𑀅𑀯𑁂𑀭𑁄, 𑀬𑁄 𑀅𑀯𑁂𑀭𑁄 𑀲𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄𑁇

(651.)

1948\. “𑀨𑀭𑀺𑀢𑁆𑀯𑀸”𑀢𑀺 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸𑁇

(652.)

1949\. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇
