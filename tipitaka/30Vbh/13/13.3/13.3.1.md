# 𑁧𑁩𑁇𑁩𑁇𑁧 𑀢𑀺𑀓

### 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀢𑀺𑀓

(701.)

2032. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 (𑁧)

2033. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁨)

2034. 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 (𑁩)

2035. 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 (𑁪)

2036. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 (𑁫)

2037. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸; 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 (𑁬)

2038. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑀢𑀺; 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 (𑁭)

2039. 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

2040. 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 (𑁯)

2041. 𑀲𑀺𑀬𑀸 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 (𑁧𑁦)

2042. 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 (𑁧𑁧)

2043. 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑁇 (𑁧𑁨)

2044. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁩)

2045. 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑁇 (𑁧𑁪)

2046. 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 (𑁧𑁫)

2047. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁬)

2048. 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄𑁇 (𑁧𑁭)

2049. 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 (𑁧𑁮)

2050. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁯)

2051. 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁨𑁦)

2052. 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁨𑁧)

2053. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 (𑁨𑁨)
