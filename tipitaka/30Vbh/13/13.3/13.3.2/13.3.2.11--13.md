# 𑁧𑁩𑁇𑁩𑁇𑁨𑁇𑁧𑁧--𑁧𑁩 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

2059\. 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀸…𑀧𑁂…  𑀦𑁄 𑀓𑀺𑀮𑁂𑀲𑀸…𑀧𑁂…  𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸, 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸; 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸; 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸; 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸; 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸, 𑀅𑀦𑀺𑀬𑀢𑀸, 𑀲𑁅𑀢𑁆𑀢𑀭𑀸, 𑀅𑀭𑀡𑀸𑀢𑀺𑁇
