# 𑁧𑁩𑁇𑁩𑁇𑁨𑁇𑁧 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

(702.)

2054. 𑀫𑁂𑀢𑁆𑀢𑀸 𑀳𑁂𑀢𑀼, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀦 𑀳𑁂𑀢𑀽, 𑀲𑀳𑁂𑀢𑀼𑀓𑀸, 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀫𑁂𑀢𑁆𑀢𑀸 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑁇 𑀫𑁂𑀢𑁆𑀢𑀸 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘; 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺, 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 𑀦 𑀳𑁂𑀢𑀽 𑀲𑀳𑁂𑀢𑀼𑀓𑀸; 𑀫𑁂𑀢𑁆𑀢𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀦 𑀳𑁂𑀢𑀼 𑀲𑀳𑁂𑀢𑀼𑀓𑀸”𑀢𑀺𑀧𑀺, “𑀦 𑀳𑁂𑀢𑀼 𑀅𑀳𑁂𑀢𑀼𑀓𑀸”𑀢𑀺𑀧𑀺𑁇
