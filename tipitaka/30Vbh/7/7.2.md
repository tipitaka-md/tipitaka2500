# 𑁭𑁇𑁨 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

(374.)

1369\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸—  𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

(375.)

1370\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(376.)

1371\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(377.)

1372\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(378.)

1373\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(379.)

1374\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(380.)

1375\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸—  𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

(381.)

1376\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(382.)

1377\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

1378\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(384.)

1379\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(385.)

1380\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
