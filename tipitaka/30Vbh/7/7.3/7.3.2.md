# 𑁭𑁇𑁩𑁇𑁨 𑀤𑀼𑀓

* 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀤𑀼𑀓

(389.)

1385\. 𑀦 𑀳𑁂𑀢𑀽𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑀸𑀢𑀺, 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑁇 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀳𑁂𑀢𑀽 𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸𑀢𑀺, 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑁇 𑀦 𑀳𑁂𑀢𑀽 𑀲𑀳𑁂𑀢𑀼𑀓𑀸𑁇

1386\. 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀲𑀗𑁆𑀔𑀢𑀸𑁇 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑁇 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀅𑀭𑀽𑀧𑀸𑁇 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 𑀦𑁄 𑀆𑀲𑀯𑀸𑁇 𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑀧𑀺𑁇 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑀧𑀺𑁇 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀦𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀸…𑀧𑁂…  𑀦𑁄 𑀕𑀦𑁆𑀣𑀸…𑀧𑁂…  𑀦𑁄 𑀑𑀖𑀸…𑀧𑁂…  𑀦𑁄 𑀬𑁄𑀕𑀸…𑀧𑁂…  𑀦𑁄 𑀦𑀻𑀯𑀭𑀡𑀸…𑀧𑁂…  𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀸…𑀧𑁂…  𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑁇 𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑁇 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀸…𑀧𑁂…  𑀦𑁄 𑀓𑀺𑀮𑁂𑀲𑀸…𑀧𑁂…  𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇

1387\. 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸𑁇 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸𑁇 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸𑁇 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇 𑀲𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑁇 𑀲𑀺𑀬𑀸 𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀅𑀭𑀡𑀸𑀢𑀺𑁇
