# 𑁭𑁇𑁧𑁇𑁪 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(367.)

1356\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 “𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀁 “𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁 “𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁 “𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀧𑀳𑀻𑀦𑀸𑀬 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1357\. 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 “𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 “𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 “𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀁 “𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀳𑁄𑀢𑀺 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑁆𑀲 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺, 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺𑁇

1366\. “𑀲𑀢𑀺𑀫𑀸”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺”𑁇 𑀇𑀫𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀫𑀸”𑀢𑀺𑁇

1367\. “𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑁄𑀓𑁄? 𑀢𑁂𑀯 𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑁄𑁇 𑀧𑀜𑁆𑀘𑀧𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀮𑁄𑀓𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀮𑁄𑀓𑁄”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀪𑀺𑀚𑁆𑀛𑀸”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀇𑀤𑀜𑁆𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀫𑀫𑁆𑀳𑀺 𑀮𑁄𑀓𑁂 𑀯𑀺𑀦𑀻𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀦𑀻𑀢𑀸 𑀲𑀦𑁆𑀢𑀸 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇
