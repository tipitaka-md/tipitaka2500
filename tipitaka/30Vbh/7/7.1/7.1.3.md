# 𑁭𑁇𑁧𑁇𑁩 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(365.)

1351\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀭𑀸𑀕𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀤𑁄𑀲𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑁄𑀳𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺𑁇

1352\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀭𑀸𑀕𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀭𑀸𑀕𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀭𑀸𑀕𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀤𑁄𑀲𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀤𑁄𑀲𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀤𑁄𑀲𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑁄𑀳𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑁄𑀳𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀫𑁄𑀳𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀫𑀳𑀕𑁆𑀕𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑁅𑀢𑁆𑀢𑀭𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑀸𑀳𑀺𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺𑁇

1353\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀭𑀸𑀕𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀤𑁄𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑁄𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇

1354\. “𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀆𑀢𑀸𑀧𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺…𑀧𑁂…  “𑀲𑀢𑀺𑀫𑀸”𑀢𑀺…𑀧𑁂…  “𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑁄𑀓𑁄? 𑀢𑀁𑀬𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑁄𑁇 𑀧𑀜𑁆𑀘𑀧𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀮𑁄𑀓𑁄𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀪𑀺𑀚𑁆𑀛𑀸”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀇𑀤𑀜𑁆𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀫𑀫𑁆𑀳𑀺 𑀮𑁄𑀓𑁂 𑀯𑀺𑀦𑀻𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀦𑀻𑀢𑀸 𑀲𑀦𑁆𑀢𑀸 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇
