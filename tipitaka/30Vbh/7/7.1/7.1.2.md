# 𑁭𑁇𑁧𑁇𑁨 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀦𑀺𑀤𑁆𑀤𑁂𑀲

1346. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀲𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀲𑀸𑀫𑀺𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀲𑀸𑀫𑀺𑀲𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁄 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀸𑀫𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺𑁇

1347. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀲𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀲𑀸𑀫𑀺𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀲𑀸𑀫𑀺𑀲𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑀁 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑁆𑀯𑀸𑀯𑀢𑁆𑀣𑀺𑀢𑀁 𑀯𑀯𑀢𑁆𑀣𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺𑁇

1348. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀲𑀸𑀫𑀺𑀲𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀲𑀸𑀫𑀺𑀲𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀲𑀸𑀫𑀺𑀲𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 “𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇

1349. “𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀆𑀢𑀸𑀧𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺…𑀧𑁂…  “𑀲𑀢𑀺𑀫𑀸”𑀢𑀺…𑀧𑁂…  “𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑁄𑀓𑁄? 𑀲𑀸𑀬𑁂𑀯 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑁄𑁇 𑀧𑀜𑁆𑀘𑀧𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀮𑁄𑀓𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀮𑁄𑀓𑁄”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀪𑀺𑀚𑁆𑀛𑀸”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀇𑀤𑀜𑁆𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀫𑀫𑁆𑀳𑀺 𑀮𑁄𑀓𑁂 𑀯𑀺𑀦𑀻𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀦𑀻𑀢𑀸 𑀲𑀦𑁆𑀢𑀸 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇
