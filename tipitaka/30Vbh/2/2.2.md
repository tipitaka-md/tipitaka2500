# 𑁨𑁇𑁨 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀆𑀬𑀢𑀦𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

(155.)

542\. 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺—  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇

(156.)

543\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀭𑀽𑀧𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀧𑀲𑁆𑀲𑀺 𑀯𑀸 𑀧𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀧𑀲𑁆𑀲𑁂 𑀯𑀸, 𑀘𑀓𑁆𑀔𑀼𑀫𑁆𑀧𑁂𑀢𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑁆𑀧𑁂𑀢𑀁 𑀮𑁄𑀓𑁄𑀧𑁂𑀲𑁄 𑀤𑁆𑀯𑀸𑀭𑀸𑀧𑁂𑀲𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑀧𑁂𑀲𑁄 𑀧𑀡𑁆𑀟𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀫𑁆𑀧𑁂𑀢𑀁 𑀦𑁂𑀢𑁆𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀦𑀬𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁧)

(157.)

544\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀲𑁄𑀢𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑁂𑀦 𑀲𑁄𑀢𑁂𑀦 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀲𑀼𑀡𑀺 𑀯𑀸 𑀲𑀼𑀡𑀸𑀢𑀺 𑀯𑀸 𑀲𑀼𑀡𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀼𑀡𑁂 𑀯𑀸, 𑀲𑁄𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑁆𑀧𑁂𑀢𑀁 𑀮𑁄𑀓𑁄𑀧𑁂𑀲𑁄 𑀤𑁆𑀯𑀸𑀭𑀸𑀧𑁂𑀲𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑀧𑁂𑀲𑁄 𑀧𑀡𑁆𑀟𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀫𑁆𑀧𑁂𑀢𑀁 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁨)

(158.)

545\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀖𑀸𑀦𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑁂𑀦 𑀖𑀸𑀦𑁂𑀦 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀖𑀸𑀬𑀺 𑀯𑀸 𑀖𑀸𑀬𑀢𑀺 𑀯𑀸 𑀖𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀖𑀸𑀬𑁂 𑀯𑀸, 𑀖𑀸𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀖𑀸𑀦𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑁆𑀧𑁂𑀢𑀁 𑀮𑁄𑀓𑁄𑀧𑁂𑀲𑁄 𑀤𑁆𑀯𑀸𑀭𑀸𑀧𑁂𑀲𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑀧𑁂𑀲𑁄 𑀧𑀡𑁆𑀟𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀫𑁆𑀧𑁂𑀢𑀁 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁩)

(159.)

546\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁? 𑀬𑀸 𑀚𑀺𑀯𑁆𑀳𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑀸𑀬 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑀬 𑀭𑀲𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀲𑀸𑀬𑀺 𑀯𑀸 𑀲𑀸𑀬𑀢𑀺 𑀯𑀸 𑀲𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀸𑀬𑁂 𑀯𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀧𑁂𑀲𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑁆𑀧𑁂𑀢𑀁 𑀮𑁄𑀓𑁄𑀧𑁂𑀲𑁄 𑀤𑁆𑀯𑀸𑀭𑀸𑀧𑁂𑀲𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑀧𑁂𑀲𑁄 𑀧𑀡𑁆𑀟𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀫𑁆𑀧𑁂𑀢𑀁 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁪)

(160.)

547\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁? 𑀬𑁄 𑀓𑀸𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀲𑀸𑀤𑁄 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀨𑀼𑀲𑀺 𑀯𑀸 𑀨𑀼𑀲𑀢𑀺 𑀯𑀸 𑀨𑀼𑀲𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀨𑀼𑀲𑁂 𑀯𑀸, 𑀓𑀸𑀬𑁄𑀧𑁂𑀲𑁄 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀓𑀸𑀬𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑁆𑀧𑁂𑀢𑀁 𑀮𑁄𑀓𑁄𑀧𑁂𑀲𑁄 𑀤𑁆𑀯𑀸𑀭𑀸𑀧𑁂𑀲𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁄𑀧𑁂𑀲𑁄 𑀧𑀡𑁆𑀟𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀫𑁆𑀧𑁂𑀢𑀁 𑀑𑀭𑀺𑀫𑀁 𑀢𑀻𑀭𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑀼𑀜𑁆𑀜𑁄 𑀕𑀸𑀫𑁄𑀧𑁂𑀲𑁄𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁫)

(161.)

548\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀁? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇

549\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑀁𑁇

550\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇

551\. 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇

552\. 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇

553\. 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀏𑀯𑀁 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇

554\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇

555\. 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇

556\. 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀏𑀯𑀁 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇

557\. 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇

558\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇

559\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑀁𑁇

560\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀦𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁬)

(162.)

561\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀦𑀻𑀮𑀁 𑀧𑀻𑀢𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀁 𑀑𑀤𑀸𑀢𑀁 𑀓𑀸𑀍𑀅𑀓𑀁 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀁 𑀳𑀭𑀺 𑀳𑀭𑀺𑀯𑀡𑁆𑀡𑀁 𑀅𑀫𑁆𑀩𑀗𑁆𑀓𑀼𑀭𑀯𑀡𑁆𑀡𑀁 𑀤𑀻𑀖𑀁 𑀭𑀲𑁆𑀲𑀁 𑀅𑀡𑀼𑀁 𑀣𑀽𑀮𑀁 𑀯𑀝𑁆𑀝𑀁 𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀁 𑀘𑀢𑀼𑀭𑀲𑁆𑀲𑀁 𑀙𑀍𑀅𑀁𑀲𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀁 𑀲𑁄𑀍𑀅𑀲𑀁𑀲𑀁 𑀦𑀺𑀦𑁆𑀦𑀁 𑀣𑀮𑀁 𑀙𑀸𑀬𑀸 𑀆𑀢𑀧𑁄 𑀆𑀮𑁄𑀓𑁄 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀅𑀩𑁆𑀪𑀸 𑀫𑀳𑀺𑀓𑀸 𑀥𑀽𑀫𑁄 𑀭𑀚𑁄 𑀘𑀦𑁆𑀤𑀫𑀡𑁆𑀟𑀮𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸 𑀲𑀽𑀭𑀺𑀬𑀫𑀡𑁆𑀟𑀮𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸 𑀢𑀸𑀭𑀓𑀭𑀽𑀧𑀸𑀦𑀁 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸 𑀆𑀤𑀸𑀲𑀫𑀡𑁆𑀟𑀮𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸 𑀫𑀡𑀺𑀲𑀗𑁆𑀔𑀫𑀼𑀢𑁆𑀢𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀧𑀲𑁆𑀲𑀺 𑀯𑀸 𑀧𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀧𑀲𑁆𑀲𑁂 𑀯𑀸, 𑀭𑀽𑀧𑀫𑁆𑀧𑁂𑀢𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁭)

(163.)

562\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁? 𑀬𑁄 𑀲𑀤𑁆𑀤𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄 𑀪𑁂𑀭𑀺𑀲𑀤𑁆𑀤𑁄 𑀫𑀼𑀤𑀺𑀗𑁆𑀕𑀲𑀤𑁆𑀤𑁄 𑀲𑀗𑁆𑀔𑀲𑀤𑁆𑀤𑁄 𑀧𑀡𑀯𑀲𑀤𑁆𑀤𑁄 𑀕𑀻𑀢𑀲𑀤𑁆𑀤𑁄 𑀯𑀸𑀤𑀺𑀢𑀲𑀤𑁆𑀤𑁄 𑀲𑀫𑁆𑀫𑀲𑀤𑁆𑀤𑁄 𑀧𑀸𑀡𑀺𑀲𑀤𑁆𑀤𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀲𑀤𑁆𑀤𑁄 𑀥𑀸𑀢𑀽𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀖𑀸𑀢𑀲𑀤𑁆𑀤𑁄 𑀯𑀸𑀢𑀲𑀤𑁆𑀤𑁄 𑀉𑀤𑀓𑀲𑀤𑁆𑀤𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀲𑀤𑁆𑀤𑁄 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀲𑀤𑁆𑀤𑁄, 𑀬𑁄 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑁄𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀲𑀤𑁆𑀤𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑀁 𑀲𑀤𑁆𑀤𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀲𑁄𑀢𑁂𑀦 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀲𑀼𑀡𑀺 𑀯𑀸 𑀲𑀼𑀡𑀸𑀢𑀺 𑀯𑀸 𑀲𑀼𑀡𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀼𑀡𑁂 𑀯𑀸, 𑀲𑀤𑁆𑀤𑁄𑀧𑁂𑀲𑁄 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁮)

(164.)

563\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁? 𑀬𑁄 𑀕𑀦𑁆𑀥𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄 𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑁄 𑀲𑀸𑀭𑀕𑀦𑁆𑀥𑁄 𑀢𑀘𑀕𑀦𑁆𑀥𑁄 𑀧𑀢𑁆𑀢𑀕𑀦𑁆𑀥𑁄 𑀧𑀼𑀧𑁆𑀨𑀕𑀦𑁆𑀥𑁄 𑀨𑀮𑀕𑀦𑁆𑀥𑁄 𑀆𑀫𑀕𑀦𑁆𑀥𑁄 𑀯𑀺𑀲𑁆𑀲𑀕𑀦𑁆𑀥𑁄 𑀲𑀼𑀕𑀦𑁆𑀥𑁄 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑁄, 𑀬𑁄 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑁄𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀥𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑀁 𑀕𑀦𑁆𑀥𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀖𑀸𑀦𑁂𑀦 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀖𑀸𑀬𑀺 𑀯𑀸 𑀖𑀸𑀬𑀢𑀺 𑀯𑀸 𑀖𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀖𑀸𑀬𑁂 𑀯𑀸, 𑀕𑀦𑁆𑀥𑁄𑀧𑁂𑀲𑁄 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁯)

(165.)

564\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀲𑀸𑀬𑀢𑀦𑀁? 𑀬𑁄 𑀭𑀲𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄 𑀫𑀽𑀮𑀭𑀲𑁄 𑀔𑀦𑁆𑀥𑀭𑀲𑁄 𑀢𑀘𑀭𑀲𑁄 𑀧𑀢𑁆𑀢𑀭𑀲𑁄 𑀧𑀼𑀧𑁆𑀨𑀭𑀲𑁄 𑀨𑀮𑀭𑀲𑁄 𑀅𑀫𑁆𑀩𑀺𑀮𑀁 𑀫𑀥𑀼𑀭𑀁 𑀢𑀺𑀢𑁆𑀢𑀓𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀡𑀺𑀓𑀁 𑀔𑀸𑀭𑀺𑀓𑀁 𑀮𑀫𑁆𑀩𑀺𑀓𑀁 𑀓𑀲𑀸𑀯𑁄 𑀲𑀸𑀤𑀼 𑀅𑀲𑀸𑀤𑀼, 𑀬𑁄 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑁄𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀲𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀬𑀁 𑀭𑀲𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑀬 𑀲𑀸𑀬𑀺 𑀯𑀸 𑀲𑀸𑀬𑀢𑀺 𑀯𑀸 𑀲𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀸𑀬𑁂 𑀯𑀸, 𑀭𑀲𑁄𑀧𑁂𑀲𑁄 𑀭𑀲𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀭𑀲𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀲𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁧𑁦)

(166.)

565\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁? 𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀫𑀼𑀤𑀼𑀓𑀁 𑀲𑀡𑁆𑀳𑀁 𑀨𑀭𑀼𑀲𑀁 𑀲𑀼𑀔𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀕𑀭𑀼𑀓𑀁 𑀮𑀳𑀼𑀓𑀁, 𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀓𑀸𑀬𑁂𑀦 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁂𑀦 𑀨𑀼𑀲𑀺 𑀯𑀸 𑀨𑀼𑀲𑀢𑀺 𑀯𑀸 𑀨𑀼𑀲𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀨𑀼𑀲𑁂 𑀯𑀸, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄𑀧𑁂𑀲𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑀧𑁂𑀲𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁”𑁇 (𑁧𑁧)

(167.)

566\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼𑁇

567\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄”𑁇 (𑁧)

568\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄”𑁇 (𑁨)

569\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑁇 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄”𑁇 (𑁩)

570\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 “𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁”𑁇 (𑁪)

571\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀥𑀸𑀢𑀼? 𑀭𑀸𑀕𑀓𑁆𑀔𑀬𑁄, 𑀤𑁄𑀲𑀓𑁆𑀔𑀬𑁄, 𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀥𑀸𑀢𑀼”𑁇 (𑁫)

---

572\. 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇
