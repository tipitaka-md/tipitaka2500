# 𑁨𑁇𑁧 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀆𑀬𑀢𑀦𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

(154.)

539\. 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺—  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇

540\. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁𑁇 𑀭𑀽𑀧𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀲𑁄𑀢𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁𑁇 𑀲𑀤𑁆𑀤𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀖𑀸𑀦𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁𑁇 𑀕𑀦𑁆𑀥𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀚𑀺𑀯𑁆𑀳𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀭𑀲𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀸𑀬𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄 𑀤𑀼𑀓𑁆𑀔𑁄 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄𑁇 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀫𑀦𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄 𑀤𑀼𑀓𑁆𑀔𑁄 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄𑁇 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀅𑀦𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇
