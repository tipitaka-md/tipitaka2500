# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁧 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓

606. 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀘𑀸”𑀢𑀺, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀸”𑀢𑀺𑀧𑀺𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺𑁇
