# 𑁨𑁇𑁩𑁇𑁨𑁇𑁫 𑀕𑀦𑁆𑀣𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀆𑀬𑀢𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀤𑀼𑀓

* 𑀕𑀦𑁆𑀣𑀕𑁄𑀘𑁆𑀙𑀓

601\. 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀕𑀦𑁆𑀣𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑁄, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀕𑀦𑁆𑀣𑁄𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀕𑀦𑁆𑀣𑀸 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘𑀸”𑀢𑀺, 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀕𑀦𑁆𑀣𑀸 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀸”𑀢𑀺𑀧𑀺𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺, “𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺𑁇
