# 𑁨𑁇𑁩𑁇𑁨𑁇𑁪 𑀲𑀁𑀬𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀆𑀬𑀢𑀦𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀲𑀁𑀬𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓

600. 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀜𑁆𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀦𑁆”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀸”𑀢𑀺𑀧𑀺𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀦𑁆”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸”𑀢𑀺𑀧𑀺𑁇
