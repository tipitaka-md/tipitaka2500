# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧𑁦 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

### 𑀆𑀬𑀢𑀦𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

604\. 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀁, 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀁𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀙𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀙𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇

605\. 𑀙𑀸𑀬𑀢𑀦𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸𑁇 𑀙𑀸𑀬𑀢𑀦𑀸 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 𑀦𑀯𑀸𑀬𑀢𑀦𑀸 𑀉𑀧𑀸𑀤𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑁇 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀙𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑁇
