### 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

598\. 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀲𑀗𑁆𑀔𑀢𑀸𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀗𑁆𑀔𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁𑁇 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀭𑀽𑀧𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀅𑀭𑀽𑀧𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀁𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸; 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇
