# 𑁧𑁬𑁇𑁨𑁇𑁭 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(806.)

2549\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢𑀲𑀢𑁆𑀢𑀢𑀺 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀲𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀦𑀢𑁆𑀣𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀲𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀦𑀢𑁆𑀣𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀲𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀦𑀢𑁆𑀣𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀁 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀜𑀸𑀡𑀁 𑀢𑀫𑁆𑀧𑀺 𑀔𑀬𑀥𑀫𑁆𑀫𑀁 𑀯𑀬𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁; 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀻𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀲𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀢𑀻𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀲𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀧𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀢𑀺 𑀜𑀸𑀡𑀁, 𑀅𑀲𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀢𑁆𑀣𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀢𑀺 𑀜𑀸𑀡𑀁𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀁 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀜𑀸𑀡𑀁 𑀢𑀫𑁆𑀧𑀺 𑀔𑀬𑀥𑀫𑁆𑀫𑀁 𑀯𑀬𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀭𑀸𑀕𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀜𑀸𑀡𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢𑀲𑀢𑁆𑀢𑀢𑀺 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
