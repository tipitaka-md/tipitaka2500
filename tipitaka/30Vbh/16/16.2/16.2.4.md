# 𑁧𑁬𑁇𑁨𑁇𑁪 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(793.)

2501\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀓𑀢𑀜𑀸𑀡𑀁? “𑀅𑀢𑁆𑀣𑀺 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀬𑀺𑀝𑁆𑀞𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀳𑀼𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀬𑀁 𑀮𑁄𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀸𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀺𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀫𑁆𑀫𑀕𑁆𑀕𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀬𑁂 𑀇𑀫𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀧𑀭𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀓𑀢𑀜𑀸𑀡𑀁”𑁇 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀘𑁆𑀘𑀸𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀜𑀸𑀡𑀁, 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀧𑀜𑁆𑀜𑀸 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀓𑀢𑀜𑀸𑀡𑀁𑁇

2502\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀘𑁆𑀘𑀸𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀜𑀸𑀡𑀁? “𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺 𑀯𑀸 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀻 𑀅𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀸 𑀔𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺 𑀭𑀼𑀘𑀺 𑀫𑀼𑀤𑀺 𑀧𑁂𑀓𑁆𑀔𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀘𑁆𑀘𑀸𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀜𑀸𑀡𑀁”𑁇

2503\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀁𑁇

2504\. 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀨𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀁𑁇 (𑁧)

(794.)

2505\. 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁𑁇

2506\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑁂 𑀜𑀸𑀡𑀁? 𑀤𑀼𑀓𑁆𑀔𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑁂 𑀜𑀸𑀡𑀁”𑁇

2507\. 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀆𑀭𑀩𑁆𑀪…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀆𑀭𑀩𑁆𑀪…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀜𑀸𑀡𑀁”𑁇 (𑁨)

(795.)

2508\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩)

(796.)

2509\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀥𑀫𑁆𑀫𑁂 𑀜𑀸𑀡𑀁? 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀥𑀫𑁆𑀫𑁂 𑀜𑀸𑀡𑀁𑁇

2510\. 𑀲𑁄 𑀇𑀫𑀺𑀦𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀜𑀸𑀢𑁂𑀦 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀤𑀺𑀢𑁂𑀦 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑁂𑀦 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑁂𑀦 𑀦𑀬𑀁 𑀦𑁂𑀢𑀺𑁇 “𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀁𑀲𑀼𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀅𑀦𑀸𑀕𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀇𑀫𑀜𑁆𑀜𑁂𑀯 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀬𑀸 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑁆𑀯𑀬𑁂 𑀜𑀸𑀡𑀁”𑁇

2511\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀭𑀺𑀬𑁂 𑀜𑀸𑀡𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀭𑀸𑀕𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀤𑁄𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀤𑁄𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑁄𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀻𑀢𑀫𑁄𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑁅𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 “𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺—  𑀬𑀸 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀭𑀺𑀬𑁂 𑀜𑀸𑀡𑀁”𑁇

2512\. 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑁂 𑀜𑀸𑀡𑀁 𑀅𑀦𑁆𑀯𑀬𑁂 𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁂 𑀜𑀸𑀡𑀁, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀜𑁆𑀜𑀸 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀜𑀸𑀡𑀁𑁇 (𑁪)

(797.)

2513\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀜𑀸 𑀆𑀘𑀬𑀸𑀬 𑀦𑁄 𑀅𑀧𑀘𑀬𑀸𑀬? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑁂 𑀧𑀜𑁆𑀜𑀸 𑀆𑀘𑀬𑀸𑀬 𑀦𑁄 𑀅𑀧𑀘𑀬𑀸𑀬𑁇

2514\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑀘𑀬𑀸𑀬 𑀦𑁄 𑀆𑀘𑀬𑀸𑀬𑁇

2515\. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑁂 𑀧𑀜𑁆𑀜𑀸 𑀆𑀘𑀬𑀸𑀬 𑀘𑁂𑀯 𑀅𑀧𑀘𑀬𑀸𑀬 𑀘𑁇

2516\. 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦𑁂𑀯 𑀆𑀘𑀬𑀸𑀬 𑀦𑁄 𑀅𑀧𑀘𑀬𑀸𑀬𑁇 (𑁫)

(798.)

2517\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀦𑁄 𑀧𑀝𑀺𑀯𑁂𑀥𑀸𑀬? 𑀬𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀓𑀸𑀫𑁂𑀲𑀼 𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬𑁄 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀢𑀺 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑀦𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀜𑁆𑀜𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀦𑁄 𑀧𑀝𑀺𑀯𑁂𑀥𑀸𑀬”𑁇

2518\. 𑀲𑁆𑀯𑁂𑀯 𑀧𑀜𑁆𑀜𑀸𑀬 𑀓𑀸𑀫𑁂𑀲𑀼 𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀲𑀫𑀸𑀦𑁄 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬𑁄 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀢𑀺 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑀦𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀯𑁂𑀥𑀸𑀬 𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬”𑁇

2519\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀘𑁂𑀯 𑀧𑀝𑀺𑀯𑁂𑀥𑀸𑀬 𑀘𑁇

2520\. 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦𑁂𑀯 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀦𑁄 𑀧𑀝𑀺𑀯𑁂𑀥𑀸𑀬𑁇 (𑁬)

(799.)

2521\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸? 𑀧𑀞𑀫𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀁 𑀓𑀸𑀫𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇

2522\. 𑀢𑀤𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇

2523\. 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇

2524\. 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀕𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑀤𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀕𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑀤𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀕𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑀤𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀕𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀁 𑀭𑀽𑀧𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑀤𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀕𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀁 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑀤𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀕𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑀤𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸 𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀞𑀺𑀢𑀺𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀲𑀳𑀕𑀢𑀸 𑀲𑀜𑁆𑀜𑀸𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀭𑀸𑀕𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀪𑀸𑀕𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁭)

(800.)

2525\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸? 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀅𑀢𑁆𑀣𑁂 𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀥𑀫𑁆𑀫𑁂 𑀜𑀸𑀡𑀁 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀢𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀸𑀪𑀺𑀮𑀸𑀧𑁂 𑀜𑀸𑀡𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸, 𑀜𑀸𑀡𑁂𑀲𑀼 𑀜𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 𑀇𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇 (𑁮)

(801.)

2526\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀧𑀤𑀸? 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀼𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀼𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2527\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀦𑁆𑀥𑀁 𑀢𑀡𑁆𑀞𑀸𑀦𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2528\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀧𑁆𑀧𑀁 𑀢𑀡𑁆𑀞𑀸𑀦𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2529\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀼𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀅𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀦𑁆𑀥𑀁 𑀢𑀡𑁆𑀞𑀸𑀦𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2530\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀼𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀅𑀓𑀲𑀺𑀭𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀧𑁆𑀧𑀁 𑀢𑀡𑁆𑀞𑀸𑀦𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀧𑀝𑀺𑀧𑀤𑀸 𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 𑀇𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀧𑀤𑀸𑁇 (𑁯)

(802.)

2531\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀦𑀺? 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2532\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀦 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀣𑁄𑀓𑀁 𑀨𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2533\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀦 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀨𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2534\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀣𑁄𑀓𑀁 𑀨𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2535\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀨𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀦𑀺𑁇 (𑁧𑁦)

2536\. 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁𑁇

2537\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑁂 𑀜𑀸𑀡𑀁? 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀭𑀸𑀫𑀭𑀡𑁂 𑀜𑀸𑀡𑀁”𑁇

2538\. 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀆𑀭𑀩𑁆𑀪…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀆𑀭𑀩𑁆𑀪…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀭𑀸𑀫𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀜𑀸𑀡𑀁”𑁇 (𑁧𑁧)

(803.)

2539\. 𑀥𑀫𑁆𑀫𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀜𑀸𑀡𑀁 𑀚𑀸𑀢𑀺𑀬𑀸𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀪𑀯𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀢𑀡𑁆𑀳𑀸𑀬𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀸𑀬𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀨𑀲𑁆𑀲𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀲𑀍𑀆𑀬𑀢𑀦𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀦𑀸𑀫𑀭𑀽𑀧𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑁂𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁, 𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬𑀧𑁂𑀢𑀁 𑀜𑀸𑀡𑀁𑁇 (𑁧𑁨--𑁨𑁧)

2540\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀜𑀸𑀡𑀁? 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀜𑀸𑀡𑀁”𑁇

2541\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀫𑀼𑀤𑀬𑀁 𑀆𑀭𑀩𑁆𑀪…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀁 𑀆𑀭𑀩𑁆𑀪…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀜𑀸𑀡𑀁”𑁇 (𑁨𑁨--𑁨𑁪)

2542\. 𑀏𑀯𑀁 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
