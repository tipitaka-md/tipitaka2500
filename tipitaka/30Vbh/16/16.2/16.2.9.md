# 𑁧𑁬𑁇𑁨𑁇𑁯 𑀦𑀯𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀯𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(808.)

2553\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑀯𑀲𑀼 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀧𑀜𑁆𑀜𑀸? 𑀧𑀞𑀫𑀚𑁆𑀛𑀸𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀡𑀸𑀜𑀸𑀡𑀁—  𑀇𑀫𑀸 𑀦𑀯𑀲𑀼 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀧𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
