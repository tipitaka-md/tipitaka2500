# 𑁧𑁬𑁇𑁨𑁇𑁨 𑀤𑀼𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀤𑀼𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(767.)

2415. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧)

2416. 𑀲𑀩𑁆𑀩𑀸𑀯 𑀧𑀜𑁆𑀜𑀸 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁨)

2417. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀲𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩)

2418. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁪)

2419. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁫)

2420. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁬)

2421. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁭)

2422. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁮)

2423. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀑𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁯)

2424. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀑𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁦)

2425. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁧)

2426. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁨)

2427. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁩)

2428. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁪)

2429. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁫)

2430. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁬)

2431. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁭)

2432. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁮)

2433. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁯)

2434. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁦)

2435. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁧)

2436. 𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁨)

2437. 𑀯𑀺𑀘𑀸𑀭𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀲𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀯𑀺𑀘𑀸𑀭𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁩)

2438. 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀧𑀻𑀢𑀺𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁪)

2439. 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀧𑀻𑀢𑀺𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁫)

2440. 𑀲𑀼𑀔𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀼𑀔𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁬)

2441. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁭)

2442. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁮)

2443. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁯)

2444. 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁦)

2445. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁧)

2446. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁨)

2447. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀦𑀺𑀬𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀬𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁩)

2448. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀲𑁅𑀢𑁆𑀢𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁪)

2449. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀢𑁆𑀣𑀚𑀸𑀧𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸?

2450. 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀅𑀭𑀳𑀢𑁄 𑀅𑀪𑀺𑀜𑁆𑀜𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀚𑀸𑀧𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀅𑀭𑀳𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀚𑀸𑀧𑀺𑀢𑀢𑁆𑀣𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁫)

---
