# 𑁧𑁬𑁇𑁨𑁇𑁮 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(807.)

2551\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸? 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀫𑀕𑁆𑀕𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑁂 𑀧𑀜𑁆𑀜𑀸—  𑀇𑀫𑀸 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
