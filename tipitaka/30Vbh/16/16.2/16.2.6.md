# 𑁧𑁬𑁇𑁨𑁇𑁬 𑀙𑀓𑁆𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(805.)

2547. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀙𑀲𑀼 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀧𑀜𑁆𑀜𑀸? 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀥𑁂 𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀜𑀸𑀡𑀁, 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀜𑀸𑀡𑀁, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁂 𑀜𑀸𑀡𑀁, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁂 𑀜𑀸𑀡𑀁—  𑀇𑀫𑀸 𑀙𑀲𑀼 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀧𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
