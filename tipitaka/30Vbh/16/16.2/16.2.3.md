# 𑁧𑁬𑁇𑁨𑁇𑁩 𑀢𑀺𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀢𑀺𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(768.)

2452\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀬𑁄𑀕𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀲𑀼 𑀬𑁄𑀕𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀲𑀼 𑀬𑁄𑀕𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀓𑀢𑀁 𑀯𑀸 𑀲𑀘𑁆𑀘𑀸𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀯𑀸 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀺𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀔𑀦𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀭𑀼𑀘𑀺𑀁 𑀫𑀼𑀤𑀺𑀁 𑀧𑁂𑀓𑁆𑀔𑀁 𑀥𑀫𑁆𑀫𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀁 𑀧𑀭𑀢𑁄 𑀅𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀮𑀪𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2453\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀼𑀢𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀬𑁄𑀕𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀲𑀼 𑀬𑁄𑀕𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀲𑀼 𑀬𑁄𑀕𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀓𑀢𑀁 𑀯𑀸 𑀲𑀘𑁆𑀘𑀸𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀯𑀸 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀯𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀺𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀔𑀦𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀭𑀼𑀘𑀺𑀁 𑀫𑀼𑀤𑀺𑀁 𑀧𑁂𑀓𑁆𑀔𑀁 𑀥𑀫𑁆𑀫𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀁 𑀧𑀭𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀮𑀪𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀢𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2454\. 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧)

(769.)

2455\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀸𑀦𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀤𑀸𑀦𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀸𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀸𑀦𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2456\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀻𑀮𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀲𑀻𑀮𑀁 𑀆𑀭𑀩𑁆𑀪 𑀲𑀻𑀮𑀸𑀥𑀺𑀕𑀘𑁆𑀙 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀻𑀮𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇

2457\. 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨)

(770.)

2458\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀧𑀜𑁆𑀜𑀸? 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀁 𑀲𑀁𑀯𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀧𑀜𑁆𑀜𑀸”𑁇

2459\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀜𑁆𑀜𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀜𑁆𑀜𑀸”𑁇

2460\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀜𑁆𑀜𑀸? 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁩)

(771.)

2461\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀆𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁? “𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀸 𑀧𑀦𑀺𑀫𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀬𑀸 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁”𑁇

2462\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀧𑀸𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁? “𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀯𑀸 𑀧𑀦𑀺𑀫𑁂 𑀥𑀫𑁆𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀬𑀸 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑀸𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁”𑁇

2463\. 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑀸𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁𑁇 (𑁪)

(772.)

2464\. 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀧𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀸𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2465\. 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀧𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2466\. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁫)

(773.)

2467\. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2468\. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2469\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁬)

(774.)

2470\. 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2471\. 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀘𑀸𑀭𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2472\. 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁭)

(775.)

2473\. 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2474\. 𑀲𑀼𑀔𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2475\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁮)

(776.)

2476\. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀧𑀜𑁆𑀜𑀸 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇

2477\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇

2478\. 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁯)

(777.)

2479\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀢𑀻𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2480\. 𑀉𑀧𑀭𑀺𑀝𑁆𑀞𑀺𑀫𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑁂 𑀧𑀜𑁆𑀜𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2481\. 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁦)

(778.)

2482\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2483\. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇

2484\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀨𑀮𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁧)

(779.)

2485\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀧𑀭𑀺𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁨:*𑁧*)

(780.)

2486\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀫𑀳𑀕𑁆𑀕𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁨:*𑁨*)

(781.)

2487\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁨:*𑁩*)

(782.)

2488\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁩:*𑁧*)

2489\. 𑀘𑀢𑀽𑀲𑀼 𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀸 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁩:*𑁨*)

(783.)

2490\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀁 𑀅𑀥𑀺𑀧𑀢𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁩:*𑁩*)

(784.)

2491\. 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀯𑀺𑀧𑀸𑀓𑁂 𑀧𑀜𑁆𑀜𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑀻, 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀢𑀺𑁇 𑀘𑀢𑀽𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀼𑀲𑀮𑁂 𑀢𑀻𑀲𑀼 𑀪𑀽𑀫𑀻𑀲𑀼 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀧𑀜𑁆𑀜𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑀻𑀢𑀺𑁇 (𑁧𑁪)

(785.)

2492\. 𑀲𑀩𑁆𑀩𑀸𑀯 𑀧𑀜𑁆𑀜𑀸 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 (𑁧𑁫)

(786.)

2493\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀢𑀻𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁬:*𑁧*)

(787.)

2494\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀦𑀸𑀕𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁬:*𑁨*)

(788.)

2495\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁬:*𑁩*)

(789.)

2496\. 𑀲𑀩𑁆𑀩𑀸𑀯 𑀧𑀜𑁆𑀜𑀸 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁧𑁭)

(790.)

2497\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁮:*𑁧*)

(791.)

2498\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁮:*𑁨*)

(792.)

2499\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸? 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑁂 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁮:*𑁩*)
