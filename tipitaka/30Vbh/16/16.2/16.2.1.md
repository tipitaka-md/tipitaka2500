# 𑁧𑁬𑁇𑁨𑁇𑁧 𑀏𑀓𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀏𑀓𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(761.)

2385\. 𑀧𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸 𑀦 𑀳𑁂𑀢𑀼𑀫𑁂𑀯, 𑀅𑀳𑁂𑀢𑀼𑀓𑀫𑁂𑀯, 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀫𑁂𑀯, 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀫𑁂𑀯, 𑀲𑀗𑁆𑀔𑀢𑀫𑁂𑀯, 𑀅𑀭𑀽𑀧𑀫𑁂𑀯, 𑀮𑁄𑀓𑀺𑀬𑀫𑁂𑀯, 𑀲𑀸𑀲𑀯𑀫𑁂𑀯, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀫𑁂𑀯, 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀫𑁂𑀯, 𑀑𑀖𑀦𑀺𑀬𑀫𑁂𑀯, 𑀬𑁄𑀕𑀦𑀺𑀬𑀫𑁂𑀯, 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀫𑁂𑀯, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀫𑁂𑀯, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀫𑁂𑀯, 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀫𑁂𑀯, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑁂𑀯, 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀫𑁂𑀯, 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀫𑁂𑀯, 𑀯𑀺𑀧𑀸𑀓𑀫𑁂𑀯, 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀫𑁂𑀯, 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀫𑁂𑀯, 𑀦 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀫𑁂𑀯, 𑀦 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀫𑁂𑀯, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀫𑁂𑀯, 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀫𑁂𑀯, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀫𑁂𑀯, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀫𑁂𑀯, 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀫𑁂𑀯, 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀫𑁂𑀯, 𑀧𑀭𑀺𑀢𑁆𑀢𑀫𑁂𑀯, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀫𑁂𑀯, 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀫𑁂𑀯, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀫𑁂𑀯, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑁂𑀯, 𑀦𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑁂𑀯, 𑀅𑀦𑀺𑀬𑀢𑀫𑁂𑀯, 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀫𑁂𑀯, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀫𑁂𑀯, 𑀅𑀦𑀺𑀘𑁆𑀘𑀫𑁂𑀯, 𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀫𑁂𑀯𑁇 (𑁧)

(762.)

2386\. 𑀧𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀢𑁆𑀣𑀼𑀓𑀸, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁨)

2387\. 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀸, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁩)

2388\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀯𑀢𑁆𑀣𑀼𑀓𑀸, 𑀩𑀸𑀳𑀺𑀭𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸, 𑀆𑀭𑀫𑁆𑀫𑀡𑀸 𑀩𑀸𑀳𑀺𑀭𑀸𑁇 (𑁪)

2389\. 𑀅𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀯𑀢𑁆𑀣𑀼𑀓𑀸, 𑀅𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺 𑀅𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑁂 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁫)

2390\. 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸, 𑀦𑀸𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀆𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘, 𑀅𑀜𑁆𑀜𑀁 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀆𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘, 𑀅𑀜𑁆𑀜𑀁 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀆𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘, 𑀅𑀜𑁆𑀜𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀆𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘, 𑀅𑀜𑁆𑀜𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀆𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘𑁇 (𑁬)

(763.)

2396\. 𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧𑁨)

(766.)

2397\. 𑀧𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸 𑀅𑀦𑀸𑀪𑁄𑀕𑀸𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀦𑀢𑁆𑀣𑀺 𑀆𑀯𑀝𑁆𑀝𑀦𑀸 𑀯𑀸 𑀆𑀪𑁄𑀕𑁄 𑀯𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑁄 𑀯𑀸 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀯𑀸𑁇 (𑁧𑁩)

2398\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁪)

2399\. 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀪𑀺𑀦𑀺𑀧𑀸𑀢𑀫𑀢𑁆𑀢𑀸𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀧𑀸𑀢𑀫𑀢𑁆𑀢𑀸𑁇

2400\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁫)

2401\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺—  𑀕𑀫𑀦𑀁 𑀯𑀸 𑀞𑀸𑀦𑀁 𑀯𑀸 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀯𑀸𑁇 (𑁧𑁬)

2402\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺—  𑀕𑀫𑀦𑀁 𑀯𑀸 𑀞𑀸𑀦𑀁 𑀯𑀸 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀯𑀸𑁇 (𑁧𑁭)

2403\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀺𑁇 (𑁧𑁮)

2404\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀺𑁇 (𑁧𑁯)

2405\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺𑁇 (𑁨𑁦)

2406\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺𑁇 (𑁨𑁧)

2407\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 (𑁨𑁨)

2408\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 (𑁨𑁩)

2409\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀘𑀯𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀘𑀯𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨𑁪)

2410\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀘𑀯𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦 𑀘𑀯𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨𑁫)

2411\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀲𑀼𑀧𑀢𑀺 𑀦 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺 𑀦 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀲𑀼𑀧𑀢𑀺 𑀦 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺 𑀦 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 (𑁨𑁬)

2412\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀲𑀼𑀧𑀢𑀺 𑀦 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺 𑀦 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸𑀧𑀺 𑀦 𑀲𑀼𑀧𑀢𑀺 𑀦 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺 𑀦 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 (𑁨𑁭)

2413\. 𑀏𑀯𑀁 𑀬𑀸𑀣𑀸𑀯𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀸𑀯𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇
