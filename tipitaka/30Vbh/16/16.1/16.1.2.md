# 𑁧𑁬𑁇𑁧𑁇𑁨 𑀤𑀼𑀓𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀤𑀼𑀓𑀫𑀸𑀢𑀺𑀓𑀸

(752.)

2231. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼—

2232. 𑀮𑁄𑀓𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧)

2233. 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨)

2234. 𑀲𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩)

2235. 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀲𑀯𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁪)

2236. 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁫)

2237. 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁬)

2238. 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁭)

2239. 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁮)

2240. 𑀑𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁯)

2241. 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀑𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁦)

2242. 𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁧)

2243. 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁨)

2244. 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁩)

2245. 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁪)

2246. 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁫)

2247. 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁬)

2248. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁭)

2249. 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁮)

2250. 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁯)

2251. 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁦)

2252. 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁧)

2253. 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁨)

2254. 𑀲𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁩)

2255. 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁪)

2256. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁫)

2257. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁬)

2258. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁭)

2259. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁮)

2260. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁯)

2261. 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁦)

2262. 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁧)

2263. 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁨)

2264. 𑀦𑀺𑀬𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀺𑀬𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁩)

2265. 𑀲𑁅𑀢𑁆𑀢𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁪)

2266. 𑀅𑀢𑁆𑀣𑀚𑀸𑀧𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀚𑀸𑀧𑀺𑀢𑀢𑁆𑀣𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩𑁫)

---

2267. 𑀏𑀯𑀁 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
