# 𑁧𑁬𑁇𑁧𑁇𑁩 𑀢𑀺𑀓𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀢𑀺𑀓𑀫𑀸𑀢𑀺𑀓𑀸

(753.)

2268. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼—

2269. 𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀼𑀢𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧)

2270. 𑀤𑀸𑀦𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀻𑀮𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨)

2271. 𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁩)

2272. 𑀆𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁, 𑀅𑀧𑀸𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁, 𑀉𑀧𑀸𑀬𑀓𑁄𑀲𑀮𑁆𑀮𑀁𑁇 (𑁪)

2273. 𑀯𑀺𑀧𑀸𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁫)

2274. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁬)

2275. 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁭)

2276. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁮)

2277. 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸, 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁯)

2278. 𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁦)

2279. 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁧)

2280. 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁨)

2281. 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁩)

2282. 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑀻 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁪)

2283. 𑀅𑀢𑀻𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁫)

2284. 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁬)

2285. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁭)

2286. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁧𑁮)

2287. 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 (𑁧𑁯)

2288. 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 (𑁨𑁦)

2289. 𑀅𑀢𑁆𑀣𑀺 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 (𑁨𑁧)

2290. 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻𑁇 (𑁨𑁨)

2291. 𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 (𑁨𑁩)

2292. 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑁇 (𑁨𑁪)

2293. 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁨𑁫)

2294. 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑀻𑁇 (𑁨𑁬)

2295. 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑀻𑁇 (𑁨𑁭)

2296. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 (𑁨𑁮)

2297. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁨𑁯)

2298. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁩𑁦)

2299. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁩𑁧)

2300. 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 (𑁩𑁨)

2301. 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 (𑁩𑁩)

2302. 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻𑁇 (𑁩𑁪)

2332. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀧𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 (𑁬𑁪)

2333. 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 (𑁬𑁫)

2334. 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀻𑁇 (𑁬𑁬)

2335. 𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 (𑁬𑁭)

2336. 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑁇 (𑁬𑁮)

2337. 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁬𑁯)

2338. 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑀻𑁇 (𑁭𑁦)

2339. 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑀻𑁇 (𑁭𑁧)

2340. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 (𑁭𑁨)

2341. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁭𑁩)

2342. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁭𑁪)

2343. 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁭𑁫)

---

2344. 𑀏𑀯𑀁 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
