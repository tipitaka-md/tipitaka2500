# 𑁧𑁬𑁇𑁧𑁇𑁧 𑀏𑀓𑀓𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀜𑀸𑀡𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀏𑀓𑀓𑀫𑀸𑀢𑀺𑀓𑀸

(751.)

2203\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼—  𑀧𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸 𑀦 𑀳𑁂𑀢𑀽, 𑀅𑀳𑁂𑀢𑀼𑀓𑀸, 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸, 𑀲𑀗𑁆𑀔𑀢𑀸, 𑀅𑀭𑀽𑀧𑀸, 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀸𑀲𑀯𑀸, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀸, 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀸, 𑀑𑀖𑀦𑀺𑀬𑀸, 𑀬𑁄𑀕𑀦𑀺𑀬𑀸, 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸, 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸, 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸, 𑀦 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸, 𑀦 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸, 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸, 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄, 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸, 𑀧𑀭𑀺𑀢𑁆𑀢𑀸, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀦𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀅𑀦𑀺𑀬𑀢𑀸, 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸𑁇 (𑁧)

2204\. 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁨)

2205\. 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁩)

2206\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀩𑀸𑀳𑀺𑀭𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁪)

2207\. 𑀅𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀅𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁫)

2208\. 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀦𑀸𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁬)

2209\. 𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀕𑁄𑀘𑀭𑀯𑀺𑀲𑀬𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀺𑁇 (𑁭)

2210\. 𑀦 𑀅𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁮)

2211\. 𑀦 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁯)

2212\. 𑀦 𑀅𑀩𑁆𑀩𑁄𑀓𑀺𑀡𑁆𑀡𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁧𑁦)

2213\. 𑀦 𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁧𑁧)

2214\. 𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁧𑁨)

2215\. 𑀧𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸 𑀅𑀦𑀸𑀪𑁄𑀕𑀸𑁇 (𑁧𑁩)

2216\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀪𑀺𑀦𑀺𑀧𑀸𑀢𑀫𑀢𑁆𑀢𑀸𑁇 (𑁧𑁪)

2217\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑁧𑁫)

2218\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 (𑁧𑁬)

2219\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 (𑁧𑁭)

2220\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀺𑁇 (𑁧𑁮)

2221\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀺𑁇 (𑁧𑁯)

2222\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺𑁇 (𑁨𑁦)

2223\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺𑁇 (𑁨𑁧)

2224\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 (𑁨𑁨)

2225\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 (𑁨𑁩)

2226\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀘𑀯𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨𑁪)

2227\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀘𑀯𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨𑁫)

2228\. 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀦 𑀲𑀼𑀧𑀢𑀺 𑀦 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺 𑀦 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 (𑁨𑁬)

2229\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀦𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀦 𑀲𑀼𑀧𑀢𑀺 𑀦 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺 𑀦 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀬𑀸𑀣𑀸𑀯𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀸𑀯𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑁇 (𑁨𑁭)

---

2230\. 𑀏𑀯𑀁 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀜𑀸𑀡𑀯𑀢𑁆𑀣𑀼𑁇
