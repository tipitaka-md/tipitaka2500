# 𑁧𑁦𑁇𑁩 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

(482.)

1594\. 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇

* 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

(483.)

1595\. 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑀦𑀁 𑀓𑀢𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸…𑀧𑁂…  𑀓𑀢𑀺 𑀲𑀭𑀡𑀸, 𑀓𑀢𑀺 𑀅𑀭𑀡𑀸?
