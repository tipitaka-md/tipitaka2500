# 𑁧𑁦𑁇𑁨 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

(472.)

1574\. 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇

(473.)

1575\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇

(474.)

1576\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 (𑁧)

1577\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 (𑁨)

1578\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 (𑁩)

1579\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀬𑀸 𑀧𑀻𑀢𑀺 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀆𑀫𑁄𑀤𑀦𑀸 𑀧𑀫𑁄𑀤𑀦𑀸 𑀳𑀸𑀲𑁄 𑀧𑀳𑀸𑀲𑁄 𑀯𑀺𑀢𑁆𑀢𑀺 𑀑𑀤𑀕𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 (𑁪)

1580\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 (𑁫)

1581\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 (𑁬)

1582\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀫𑀚𑁆𑀛𑀢𑁆𑀢𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀳𑀺 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁭)

(475.)

1583\. 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇

(476.)

1584\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

1585\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀫𑀚𑁆𑀛𑀢𑁆𑀢𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(477.)

1586\. 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇

(478.)

1587\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇

1590\. 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇

1591\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

1592\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀨𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀁 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁, 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀫𑀚𑁆𑀛𑀢𑁆𑀢𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
