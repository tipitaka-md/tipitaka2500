# 𑁧𑁇𑁧𑁇𑁩 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥

### 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥

(14.)

25\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀯𑀸 𑀲𑀼𑀔𑀼𑀫𑀸 𑀯𑀸 𑀳𑀻𑀦𑀸 𑀯𑀸 𑀧𑀡𑀻𑀢𑀸 𑀯𑀸 𑀬𑀸 𑀤𑀽𑀭𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀸 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(15.)

26\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑀻𑀢𑀸? 𑀬𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑀻𑀢𑀸 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀯𑀺𑀕𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀕𑀢𑀸 𑀅𑀢𑀻𑀢𑀸 𑀅𑀢𑀻𑀢𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑀻𑀢𑀸𑁇

27\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀸𑀕𑀢𑀸? 𑀬𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀚𑀸𑀢𑀸 𑀅𑀪𑀽𑀢𑀸 𑀅𑀲𑀜𑁆𑀚𑀸𑀢𑀸 𑀅𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀅𑀦𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀅𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀦𑀸𑀕𑀢𑀸 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑁇

28\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸? 𑀬𑀸 𑀲𑀜𑁆𑀜𑀸 𑀚𑀸𑀢𑀸 𑀪𑀽𑀢𑀸 𑀲𑀜𑁆𑀚𑀸𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀉𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇

(16.)

29\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸? 𑀬𑀸 𑀲𑀜𑁆𑀜𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀦𑀺𑀬𑀓𑀸 𑀧𑀸𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑀺𑀓𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑁇

30\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸? 𑀬𑀸 𑀲𑀜𑁆𑀜𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀧𑀭𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀦𑀺𑀬𑀓𑀸 𑀧𑀸𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑀺𑀓𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇

(17.)

31\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀲𑀼𑀔𑀼𑀫𑀸? 𑀧𑀝𑀺𑀖𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸, 𑀅𑀥𑀺𑀯𑀘𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑁇 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸, 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸, 𑀲𑀼𑀔𑀸𑀬 𑀘 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑁇 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑁇 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸, 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑁇 𑀲𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸, 𑀅𑀦𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑁇 𑀢𑀁 𑀢𑀁 𑀯𑀸 𑀧𑀦 𑀲𑀜𑁆𑀜𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀲𑀜𑁆𑀜𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀸 𑀲𑀼𑀔𑀼𑀫𑀸 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑁇

(18.)

32\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸 𑀧𑀡𑀻𑀢𑀸? 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸, 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸, 𑀲𑀼𑀔𑀸𑀬 𑀘 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸, 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀲𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸, 𑀅𑀦𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀢𑀁 𑀢𑀁 𑀯𑀸 𑀧𑀦 𑀲𑀜𑁆𑀜𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀲𑀜𑁆𑀜𑀸 𑀳𑀻𑀦𑀸 𑀧𑀡𑀻𑀢𑀸 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑁇

33\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀤𑀽𑀭𑁂? 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀳𑀺 𑀲𑀜𑁆𑀜𑀸𑀳𑀺 𑀤𑀽𑀭𑁂; 𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂; 𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀳𑀺 𑀲𑀜𑁆𑀜𑀸𑀳𑀺 𑀤𑀽𑀭𑁂; 𑀅𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀼𑀲𑀮𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀳𑀺 𑀲𑀜𑁆𑀜𑀸𑀳𑀺 𑀤𑀽𑀭𑁂; 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀸𑀬 𑀘 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀳𑀺 𑀲𑀜𑁆𑀜𑀸𑀳𑀺 𑀤𑀽𑀭𑁂; 𑀲𑀼𑀔𑀸𑀬 𑀘 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂; 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀘 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀳𑀺 𑀲𑀜𑁆𑀜𑀸𑀳𑀺 𑀤𑀽𑀭𑁂; 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀘 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀘 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂; 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀳𑀺 𑀲𑀜𑁆𑀜𑀸𑀳𑀺 𑀤𑀽𑀭𑁂; 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂𑁇 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂; 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂𑁇 𑀲𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀸𑀲𑀯𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂; 𑀅𑀦𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀸𑀲𑀯𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀤𑀽𑀭𑁂—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀤𑀽𑀭𑁂𑁇

34\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀓𑀼𑀲𑀮𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀲𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀸𑀲𑀯𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀅𑀦𑀸𑀲𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀸𑀲𑀯𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀢𑀁 𑀢𑀁 𑀯𑀸 𑀧𑀦 𑀲𑀜𑁆𑀜𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀲𑀜𑁆𑀜𑀸 𑀤𑀽𑀭𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑁇
