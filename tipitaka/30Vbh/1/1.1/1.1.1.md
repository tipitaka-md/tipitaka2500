# 𑁧𑁇𑁧𑁇𑁧 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥

### 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥

(2.)

3\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀭𑀽𑀧𑀁 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀯𑀸 𑀲𑀼𑀔𑀼𑀫𑀁 𑀯𑀸 𑀳𑀻𑀦𑀁 𑀯𑀸 𑀧𑀡𑀻𑀢𑀁 𑀯𑀸 𑀬𑀁 𑀤𑀽𑀭𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀸, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(3.)

4\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀅𑀢𑀻𑀢𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀅𑀢𑀻𑀢𑀁 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑀁 𑀯𑀺𑀕𑀢𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀁 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀕𑀢𑀁 𑀅𑀢𑀻𑀢𑀁 𑀅𑀢𑀻𑀢𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀅𑀢𑀻𑀢𑀁𑁇

5\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀅𑀦𑀸𑀕𑀢𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑀸𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀅𑀲𑀜𑁆𑀚𑀸𑀢𑀁 𑀅𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀁 𑀅𑀦𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀁 𑀅𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀅𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀅𑀦𑀸𑀕𑀢𑀁 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀸𑀕𑀢𑀁𑁇

6\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀚𑀸𑀢𑀁 𑀪𑀽𑀢𑀁 𑀲𑀜𑁆𑀚𑀸𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇

(4.)

7\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀦𑀺𑀬𑀓𑀁 𑀧𑀸𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁𑁇

8\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀧𑀭𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀦𑀺𑀬𑀓𑀁 𑀧𑀸𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇

(5.)

9\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁𑁇

10\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀲𑀼𑀔𑀼𑀫𑀁? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇

(6.)

11\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀳𑀻𑀦𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀜𑁆𑀜𑀸𑀢𑀁 𑀅𑀯𑀜𑁆𑀜𑀸𑀢𑀁 𑀳𑀻𑀍𑀇𑀢𑀁 𑀧𑀭𑀺𑀪𑀽𑀢𑀁 𑀅𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀳𑀻𑀦𑀁 𑀳𑀻𑀦𑀫𑀢𑀁 𑀳𑀻𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀅𑀦𑀺𑀝𑁆𑀞𑀁 𑀅𑀓𑀦𑁆𑀢𑀁 𑀅𑀫𑀦𑀸𑀧𑀁, 𑀭𑀽𑀧𑀸 𑀲𑀤𑁆𑀤𑀸 𑀕𑀦𑁆𑀥𑀸 𑀭𑀲𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀳𑀻𑀦𑀁𑁇

12\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀧𑀡𑀻𑀢𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀅𑀦𑀯𑀜𑁆𑀜𑀸𑀢𑀁 𑀅𑀳𑀻𑀍𑀇𑀢𑀁 𑀅𑀧𑀭𑀺𑀪𑀽𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑀁 𑀧𑀡𑀻𑀢𑀁 𑀧𑀡𑀻𑀢𑀫𑀢𑀁 𑀧𑀡𑀻𑀢𑀲𑀫𑁆𑀫𑀢𑀁 𑀇𑀝𑁆𑀞𑀁 𑀓𑀦𑁆𑀢𑀁 𑀫𑀦𑀸𑀧𑀁, 𑀭𑀽𑀧𑀸 𑀲𑀤𑁆𑀤𑀸 𑀕𑀦𑁆𑀥𑀸 𑀭𑀲𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀧𑀡𑀻𑀢𑀁𑁇 𑀢𑀁 𑀢𑀁 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀭𑀽𑀧𑀁 𑀳𑀻𑀦𑀁 𑀧𑀡𑀻𑀢𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇

(7.)

13\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀤𑀽𑀭𑁂? 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀸𑀲𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀓𑀝𑁆𑀞𑁂 𑀤𑀽𑀭𑁂 𑀅𑀲𑀦𑁆𑀢𑀺𑀓𑁂—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀤𑀽𑀭𑁂𑁇

14\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀆𑀲𑀦𑁆𑀦𑁂 𑀉𑀧𑀓𑀝𑁆𑀞𑁂 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀭𑀽𑀧𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀢𑀁 𑀢𑀁 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀭𑀽𑀧𑀁 𑀤𑀽𑀭𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇
