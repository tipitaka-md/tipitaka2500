# 𑁧𑁇𑁩𑁇𑁨𑁇𑁬--𑁮 𑀑𑀖𑀬𑁄𑀕𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀤𑀼𑀓

* 𑀑𑀖𑀬𑁄𑀕𑀦𑀻𑀯𑀭𑀡𑀕𑁄𑀘𑁆𑀙𑀓

528\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦𑁄 𑀑𑀖𑀸…𑀧𑁂…  𑀦𑁄 𑀬𑁄𑀕𑀸…𑀧𑁂…  𑀦𑁄 𑀦𑀻𑀯𑀭𑀡𑀸𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀦𑀻𑀯𑀭𑀡𑀁𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄 𑀘𑀸”𑀢𑀺, “𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀁”𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀦𑀻𑀯𑀭𑀡𑀸 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸”𑀢𑀺𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄 𑀘, 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀦𑀻𑀯𑀭𑀡𑀸 𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀸”𑀢𑀺𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀦𑀻𑀯𑀭𑀡𑀜𑁆𑀘𑁂𑀯 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀦𑀻𑀯𑀭𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸”𑀢𑀺𑀧𑀺, “𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀸”𑀢𑀺𑀧𑀺𑁇
