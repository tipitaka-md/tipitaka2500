# 𑁧𑁇𑁩𑁇𑁨𑁇𑁩 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

525\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦𑁄 𑀆𑀲𑀯𑀸𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑁄, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀆𑀲𑀯𑁄𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀸𑀲𑀯𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀲𑀸𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘𑀸”𑀢𑀺, 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄”𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄”𑀢𑀺𑀧𑀺𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄”𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀸”𑀢𑀺𑀧𑀺, “𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑀸”𑀢𑀺𑀧𑀺𑁇
