# 𑁧𑁇𑁩𑁇𑁨𑁇𑁨 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓

523\. 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸, 𑀲𑀗𑁆𑀔𑀢𑀸𑁇

524\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑁄𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀭𑀽𑀧𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀅𑀭𑀽𑀧𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀮𑁄𑀓𑀺𑀬𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸; 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇
