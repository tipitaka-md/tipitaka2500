# 𑁧𑁇𑁩𑁇𑁨𑁇𑁯 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

529\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑁄, 𑀲𑀺𑀬𑀸 𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄”𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑀸”𑀢𑀺, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸”𑀢𑀺𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑁄”𑀢𑀺𑀧𑀺𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺𑁇
