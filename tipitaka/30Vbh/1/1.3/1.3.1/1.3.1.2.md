# 𑁧𑁇𑁩𑁇𑁧𑁇𑁨 𑀯𑁂𑀤𑀦𑀸𑀢𑁆𑀢𑀺𑀓

* 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀢𑀺𑀓

* 𑀯𑁂𑀤𑀦𑀸𑀢𑁆𑀢𑀺𑀓

501\. 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
