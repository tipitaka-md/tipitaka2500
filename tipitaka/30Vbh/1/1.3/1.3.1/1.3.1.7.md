# 𑁧𑁇𑁩𑁇𑁧𑁇𑁭 𑀧𑀻𑀢𑀺𑀢𑁆𑀢𑀺𑀓

* 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

* 𑀢𑀺𑀓

* 𑀧𑀻𑀢𑀺𑀢𑁆𑀢𑀺𑀓

506\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄”𑀢𑀺𑀧𑀺, “𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄”𑀢𑀺𑀧𑀺, “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄”𑀢𑀺𑀧𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄”𑀢𑀺𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺𑁇
