# 𑁧𑁇𑁨𑁇𑁩 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥

### 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥

(62.)

172\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

173\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

174\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇

175\. 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇

176\. 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀏𑀯𑀁 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

177\. 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

178\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

179\. 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

180\. 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀏𑀯𑀁 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

181\. 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(63.)

182\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

183\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

184\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(64.)

185\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

186\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

187\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

188\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

189\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 200\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀲𑀭𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀡𑁄𑁇

201\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(69.)

202\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

203\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

204\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(70.)

205\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

206\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀲𑀭𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀡𑁄𑁇

207\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

208\. 𑀢𑀺𑀓𑀫𑀽𑀮𑀓𑀁𑁇



(71.)

210\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

211\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

213\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

214\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(73.)

215\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

216\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

217\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(74.)

218\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

219\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄𑁇

220\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(75.)

221\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

222\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄𑁇

223\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(76.)

224\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

225\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇

226\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(77.)

227\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

228\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

229\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(78.)

230\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

231\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑁄𑁇

232\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(79.)

233\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

234\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄𑁇

235\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(80.)

236\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

237\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

238\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀻…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(81.)

239\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

240\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄𑁇

241\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(82.)

242\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

243\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄𑁇

244\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(83.)

245\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

246\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

247\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

248\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

249\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄𑁇

250\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(85.)

251\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

252\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑁄𑁇

253\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(86.)

254\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

255\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

256\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(87.)

257\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

258\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀑𑀖𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑁄𑁇

259\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀻…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(88.)

260\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

261\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑁄𑁇

262\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(89.)

263\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

264\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

265\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(90.)

266\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

267\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀬𑁄𑀕𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑁄𑁇

268\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(91.)

269\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

270\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄𑁇

271\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

272\. 𑀉𑀪𑀢𑁄𑀯𑀟𑁆𑀠𑀓𑀁𑁇



273\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

274\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

275\. 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

276\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

277\. 𑀢𑀺𑀁𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀢𑀺𑀁𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

278\. 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

279\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇
