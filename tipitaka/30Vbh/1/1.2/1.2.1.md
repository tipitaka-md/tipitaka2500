# 𑁧𑁇𑁨𑁇𑁧 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥

### 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥

(33.)

57\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀲𑀩𑁆𑀩𑀁 𑀭𑀽𑀧𑀁 𑀦 𑀳𑁂𑀢𑀼, 𑀅𑀳𑁂𑀢𑀼𑀓𑀁, 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁, 𑀲𑀗𑁆𑀔𑀢𑀁, 𑀭𑀽𑀧𑀁, 𑀮𑁄𑀓𑀺𑀬𑀁, 𑀲𑀸𑀲𑀯𑀁, 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑀁, 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀁, 𑀑𑀖𑀦𑀺𑀬𑀁, 𑀬𑁄𑀕𑀦𑀺𑀬𑀁, 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑀁, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁, 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀁, 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁, 𑀦 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀦 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀁, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁, 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁, 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁, 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀻, 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀁, 𑀧𑀭𑀺𑀢𑁆𑀢𑀁, 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁, 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀦𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀅𑀦𑀺𑀬𑀢𑀁, 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀙𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀁𑁇 𑀏𑀯𑀁 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

58\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀫𑀳𑀸𑀪𑀽𑀢𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀼𑀔𑀼𑀫𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇 𑀏𑀯𑀁 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

59\. (𑀬𑀣𑀸 𑀭𑀽𑀧𑀓𑀡𑁆𑀟𑁂 𑀯𑀺𑀪𑀢𑁆𑀢𑀁, 𑀢𑀣𑀸 𑀇𑀥 𑀯𑀺𑀪𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇)

60\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀢𑀁 𑀉𑀧𑀸𑀤𑀸, 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀢𑀁 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁…𑀧𑁂…  𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀢𑀁 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀩𑀸𑀳𑀺𑀭𑀁 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄𑁇 𑀏𑀯𑀁 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

61\. 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁; 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁; 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁; 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁; 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀼𑀫𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀬𑀁 𑀢𑀁 𑀭𑀽𑀧𑀁 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀢𑀁 𑀅𑀢𑁆𑀣𑀺 𑀤𑀽𑀭𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑀁, 𑀲𑀼𑀢𑀁, 𑀫𑀼𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀭𑀽𑀧𑀁𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

62\. 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀧𑀣𑀯𑀻𑀥𑀸𑀢𑀼, 𑀆𑀧𑁄𑀥𑀸𑀢𑀼, 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼, 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼, 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑁇 𑀏𑀯𑀁 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

63\. 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁𑁇 𑀏𑀯𑀁 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

64\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

65\. 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀁𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

66\. 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀏𑀯𑀁 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

67\. 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

68\. 𑀏𑀓𑀸𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇 𑀏𑀯𑀁 𑀏𑀓𑀸𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

---
