# 𑁧𑁇𑁨𑁇𑁨 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥

### 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥

(34.)

70\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

71\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

72\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇

73\. 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇

74\. 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀏𑀯𑀁 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

75\. 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀏𑀯𑀁 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

76\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

77\. 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

78\. 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀏𑀯𑀁 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

79\. 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(35.)

80\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

81\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

82\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

83\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

84\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 91\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

92\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(39.)

93\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

94\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

95\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(40.)

96\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

97\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀲𑀭𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀡𑁄𑁇

98\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(41.)

99\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

100\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

101\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(42.)

102\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

104\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

138\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

139\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

140\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄𑁇

144\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

145\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

147\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(57.)

148\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

149\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑁄𑁇

150\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀻…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

151\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

152\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

153\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

156\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

158\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑁄𑁇

159\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

161\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

163\. 𑀉𑀪𑀢𑁄𑀯𑀟𑁆𑀠𑀓𑀁𑁇



164\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

165\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

166\. 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

168\. 𑀢𑀺𑀁𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀏𑀯𑀁 𑀢𑀺𑀁𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

169\. 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

170\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇
