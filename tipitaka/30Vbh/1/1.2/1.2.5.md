# 𑁧𑁇𑁨𑁇𑁫 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥

### 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥

(121.)

390\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

391\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

392\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇

393\. 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇

394\. 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀏𑀯𑀁 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

395\. 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀏𑀯𑀁 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

396\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

397\. 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

398\. 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀏𑀯𑀁 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

399\. 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(122.)

400\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

401\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑁄𑁇

405\. 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀑𑀖𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀑𑀖𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑁄𑀖𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀬𑁄𑀕𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀬𑁄𑀕𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀻𑀯𑀭𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀻𑀯𑀭𑀡𑀺𑀬𑁄𑁇

406\. 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄𑁇

407\. 𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀘𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀘𑀸𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑁅𑀢𑁆𑀢𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀭𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀡𑁄𑁇

477\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

479\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

480\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

481\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

483\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀨𑀲𑁆𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

492\. 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

493\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄; 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄; 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

494\. 𑀢𑀺𑀁𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀏𑀯𑀁 𑀢𑀺𑀁𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

495\. 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

496\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇
