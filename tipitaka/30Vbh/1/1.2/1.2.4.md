# 𑁧𑁇𑁨𑁇𑁪 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥

* 𑀔𑀦𑁆𑀥𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥

(92.)

281\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

282\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑁇

283\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇

284\. 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇

285\. 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀏𑀯𑀁 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

286\. 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀙𑀩𑁆𑀩𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

287\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

288\. 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

289\. 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀏𑀯𑀁 𑀦𑀯𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

290\. 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀳𑀕𑀢𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(93.)

291\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

292\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀳𑁂𑀢𑀼𑁇

293\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀻𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

337\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄𑁇

338\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

339\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

340\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀦𑀘𑀺 𑀦 𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑁄𑁇

341\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

342\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

(109.)

345\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

347\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀻…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(110.)

348\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

349\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

350\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁂𑀓𑁆𑀔𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(111.)

351\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

352\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁄 𑀘, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄𑁇

353\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(112.)

354\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

355\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄𑁇

356\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(113.)

357\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

358\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑁄𑁇

359\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

360\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

361\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁄 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇

362\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀺𑀬𑀢𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

363\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

365\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

370\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄 𑀘, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇

373\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇

(119.)

375\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

376\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀲𑀁𑀬𑁄𑀚𑀦𑀺𑀬𑁄𑁇

377\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(120.)

378\. 𑀏𑀓𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄𑁇

379\. 𑀤𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀕𑀦𑁆𑀣𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀦𑁄 𑀕𑀦𑁆𑀣𑁄𑁇

380\. 𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄…𑀧𑁂… 𑁇 𑀏𑀯𑀁 𑀤𑀲𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

381\. 𑀉𑀪𑀢𑁄𑀯𑀟𑁆𑀠𑀓𑀁𑁇



382\. 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

383\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄; 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄; 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀏𑀯𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

384\. 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

385\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

386\. 𑀢𑀺𑀁𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀢𑀺𑀁𑀲𑀢𑀺𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

387\. 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

388\. 𑀅𑀧𑀭𑁄𑀧𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸𑁇 𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇
