# 𑁮𑁇𑁧 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

(390.)

1390\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸—  𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(391.)

1391\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸? 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀮𑁄𑀪𑁄, 𑀤𑁄𑀲𑁄, 𑀫𑁄𑀳𑁄𑁇 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑁇 𑀢𑀁𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀢𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸”𑁇 𑀇𑀢𑀺 𑀇𑀫𑁂𑀲𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(392.)

1392\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄 𑀙𑀦𑁆𑀤𑁄 𑀙𑀦𑁆𑀤𑀺𑀓𑀢𑀸 𑀓𑀢𑁆𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑁄”𑁇 𑀇𑀫𑀁 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀲𑀜𑁆𑀚𑀦𑁂𑀢𑀺 𑀉𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇

(393.)

1393\. “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀸𑀫𑁄”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀯𑀸𑀬𑀸𑀫𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀼𑀧𑁂𑀢𑁄 𑀉𑀧𑀸𑀕𑀢𑁄 𑀲𑀫𑀼𑀧𑀸𑀕𑀢𑁄 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇

(394.)

1394\. “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀻𑀭𑀺𑀬𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀭𑀺𑀬𑀁”𑁇 𑀇𑀫𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀲𑀫𑀸𑀭𑀪𑀢𑀺 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺𑁇

(395.)

1395\. “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇

(396.)

1396\. “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀥𑀸𑀦𑀁”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀧𑀥𑀸𑀦𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇

(397.)

1397\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸? 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀮𑁄𑀪𑁄, 𑀤𑁄𑀲𑁄, 𑀫𑁄𑀳𑁄𑁇 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀸 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸𑁇 𑀢𑀁𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀢𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸”𑁇 𑀇𑀢𑀺 𑀇𑀫𑁂𑀲𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

1398\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄 𑀙𑀦𑁆𑀤𑁄 𑀙𑀦𑁆𑀤𑀺𑀓𑀢𑀸 𑀓𑀢𑁆𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑁄”𑁇 𑀇𑀫𑀁 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀲𑀜𑁆𑀚𑀦𑁂𑀢𑀺 𑀉𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇

1399\. “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀸𑀫𑁄”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀯𑀸𑀬𑀸𑀫𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇

1401\. “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇

1402\. “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀥𑀸𑀦𑀁”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀧𑀥𑀸𑀦𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇

1403\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸? 𑀢𑀻𑀡𑀺 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀅𑀮𑁄𑀪𑁄, 𑀅𑀤𑁄𑀲𑁄, 𑀅𑀫𑁄𑀳𑁄𑁇 𑀢𑀁𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀢𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁, 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁, 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸”𑁇 𑀇𑀢𑀺 𑀇𑀫𑁂𑀲𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

1404\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄 …𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀥𑀸𑀦𑀁”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀧𑀥𑀸𑀦𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇

(405.)

1405\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸? 𑀢𑀻𑀡𑀺 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀅𑀮𑁄𑀪𑁄, 𑀅𑀤𑁄𑀲𑁄, 𑀅𑀫𑁄𑀳𑁄𑁇 𑀢𑀁𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀢𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸”𑁇 𑀇𑀢𑀺 𑀇𑀫𑁂𑀲𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(406.)

1406\. “𑀞𑀺𑀢𑀺𑀬𑀸”𑀢𑀺𑁇 𑀬𑀸 𑀞𑀺𑀢𑀺 𑀲𑁄 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑁄, 𑀬𑁄 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑁄 𑀲𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑁄, 𑀬𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑁄 𑀢𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁, 𑀬𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀲𑀸 𑀪𑀸𑀯𑀦𑀸, 𑀬𑀸 𑀪𑀸𑀯𑀦𑀸 𑀲𑀸 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻𑁇

(407.)

1407\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀧𑀤𑀳𑀢𑀻𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀥𑀸𑀦𑀁”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀧𑀥𑀸𑀦𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇
