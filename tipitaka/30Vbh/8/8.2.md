# 𑁮𑁇𑁨 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

(408.)

1409\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸—  𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(409.)

1410\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(410.)

1411\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄 𑀙𑀦𑁆𑀤𑁄 𑀙𑀦𑁆𑀤𑀺𑀓𑀢𑀸 𑀓𑀢𑁆𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑁄”𑁇 𑀇𑀫𑀁 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀲𑀜𑁆𑀚𑀦𑁂𑀢𑀺 𑀉𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇

(411.)

1412\. “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀸𑀫𑁄”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀯𑀸𑀬𑀸𑀫𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀼𑀧𑁂𑀢𑁄 𑀉𑀧𑀸𑀕𑀢𑁄 𑀲𑀫𑀼𑀧𑀸𑀕𑀢𑁄 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇

(412.)

1413\. “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀻𑀭𑀺𑀬𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀭𑀺𑀬𑀁”𑁇 𑀇𑀫𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀲𑀫𑀸𑀭𑀪𑀢𑀺 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺𑁇

(413.)

1414\. “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇

(414.)

1415\. “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(415.)

1416\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(416.)

1417\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(417.)

1418\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(418.)

1419\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(419.)

1420\. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀞𑀺𑀢𑀺𑀬𑀸 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀧𑀤𑀳𑀢𑀺𑁇

(420.)

1421\. “𑀞𑀺𑀢𑀺𑀬𑀸”𑀢𑀺𑁇 𑀬𑀸 𑀞𑀺𑀢𑀺 𑀲𑁄 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑁄, 𑀬𑁄 𑀅𑀲𑀫𑁆𑀫𑁄𑀲𑁄 𑀲𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑁄, 𑀬𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑁄 𑀢𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁, 𑀬𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀲𑀸 𑀪𑀸𑀯𑀦𑀸, 𑀬𑀸 𑀪𑀸𑀯𑀦𑀸 𑀲𑀸 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀻𑁇

(421.)

1422\. “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀦𑁆𑀤𑁄? 𑀬𑁄 𑀙𑀦𑁆𑀤𑁄 𑀙𑀦𑁆𑀤𑀺𑀓𑀢𑀸 𑀓𑀢𑁆𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑁄”𑁇 𑀇𑀫𑀁 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀺 𑀲𑀜𑁆𑀚𑀦𑁂𑀢𑀺 𑀉𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀢𑀻”𑀢𑀺𑁇

(422.)

1423\. “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀸𑀬𑀸𑀫𑁄? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀸𑀫𑁄”𑁇 𑀇𑀫𑀺𑀦𑀸 𑀯𑀸𑀬𑀸𑀫𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀸𑀬𑀫𑀢𑀻”𑀢𑀺𑁇

(423.)

1424\. “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀻𑀭𑀺𑀬𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀭𑀺𑀬𑀁”𑁇 𑀇𑀫𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀲𑀫𑀸𑀭𑀪𑀢𑀺 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀻”𑀢𑀺𑁇

(424.)

1425\. “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇

(425.)

1426\. “𑀧𑀤𑀳𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁? 𑀬𑁄 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(426.)

1427\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀞𑀫𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀧𑀝𑀺𑀧𑀤𑀁 𑀤𑀦𑁆𑀥𑀸𑀪𑀺𑀜𑁆𑀜𑀁, 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀗𑁆𑀕𑀁 𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

---
