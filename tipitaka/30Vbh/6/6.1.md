# 𑁬𑁇𑁧 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

(225.)

941\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁, 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(226.)

942\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀡𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

943\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸? 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀅𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀆𑀦𑁂𑀜𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇

944\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀤𑀸𑀦𑀫𑀬𑀸 𑀲𑀻𑀮𑀫𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀫𑀬𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

945\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

946\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀆𑀦𑁂𑀜𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀦𑁂𑀜𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

947\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀓𑀸𑀬𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀯𑀘𑀻𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀯𑀘𑀻𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸”𑁇

(227.)

948\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

(228.)

949\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀪𑀽𑀢𑀸, 𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀭𑀽𑀧𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

(229.)

950\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁”𑁇

(230.)

951\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

(231.)

952\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸”𑁇

(232.)

953\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸? 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸, 𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸, 𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸, 𑀭𑀲𑀢𑀡𑁆𑀳𑀸, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸, 𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸”𑁇

(233.)

954\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁? 𑀓𑀸𑀫𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀤𑀺𑀝𑁆𑀞𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀼𑀧𑀸𑀤𑀸𑀦𑀁, 𑀅𑀢𑁆𑀢𑀯𑀸𑀤𑀼𑀧𑀸𑀤𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁”𑁇

(234.)

955\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄? 𑀪𑀯𑁄 𑀤𑀼𑀯𑀺𑀥𑁂𑀦—  𑀅𑀢𑁆𑀣𑀺 𑀓𑀫𑁆𑀫𑀪𑀯𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀪𑀯𑁄𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀫𑁆𑀫𑀪𑀯𑁄? 𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀅𑀧𑀼𑀜𑁆𑀜𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀆𑀦𑁂𑀜𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀫𑁆𑀫𑀪𑀯𑁄”𑁇 𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀪𑀯𑀕𑀸𑀫𑀺𑀓𑀫𑁆𑀫𑀁 𑀓𑀫𑁆𑀫𑀪𑀯𑁄𑁇

956\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀪𑀯𑁄? 𑀓𑀸𑀫𑀪𑀯𑁄, 𑀭𑀽𑀧𑀪𑀯𑁄, 𑀅𑀭𑀽𑀧𑀪𑀯𑁄, 𑀲𑀜𑁆𑀜𑀸𑀪𑀯𑁄, 𑀅𑀲𑀜𑁆𑀜𑀸𑀪𑀯𑁄, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀪𑀯𑁄, 𑀏𑀓𑀯𑁄𑀓𑀸𑀭𑀪𑀯𑁄, 𑀘𑀢𑀼𑀯𑁄𑀓𑀸𑀭𑀪𑀯𑁄, 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀪𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀪𑀯𑁄”𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄”𑁇

(235.)

957\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀺 𑀢𑀫𑁆𑀳𑀺 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑁂 𑀚𑀸𑀢𑀺 𑀲𑀜𑁆𑀚𑀸𑀢𑀺 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺, 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀆𑀬𑀢𑀦𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺”𑁇

(236.)

958\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀚𑀭𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀭𑀡𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀚𑀭𑀸? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀺 𑀢𑀫𑁆𑀳𑀺 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑁂 𑀚𑀭𑀸 𑀚𑀻𑀭𑀡𑀢𑀸 𑀔𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀧𑀸𑀮𑀺𑀘𑁆𑀘𑀁 𑀯𑀮𑀺𑀢𑁆𑀢𑀘𑀢𑀸 𑀆𑀬𑀼𑀦𑁄 𑀲𑀁𑀳𑀸𑀦𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀸𑀓𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀭𑀸”𑁇

959\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀭𑀡𑀁? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀫𑁆𑀳𑀸 𑀢𑀫𑁆𑀳𑀸 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀸 𑀘𑀼𑀢𑀺 𑀘𑀯𑀦𑀢𑀸 𑀪𑁂𑀤𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀁 𑀫𑀘𑁆𑀘𑀼 𑀫𑀭𑀡𑀁 𑀓𑀸𑀮𑀓𑀺𑀭𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀪𑁂𑀤𑁄 𑀓𑀍𑀏𑀯𑀭𑀲𑁆𑀲 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑁆𑀲𑀼𑀧𑀘𑁆𑀙𑁂𑀤𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀭𑀡𑀁”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀚𑀭𑀸, 𑀇𑀤𑀜𑁆𑀘 𑀫𑀭𑀡𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁”𑁇

(237.)

960\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑁄𑀓𑁄? 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑁄𑀓𑁄 𑀲𑁄𑀘𑀦𑀸 𑀲𑁄𑀘𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑁄𑀲𑁄𑀓𑁄 𑀅𑀦𑁆𑀢𑁄𑀧𑀭𑀺𑀲𑁄𑀓𑁄 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁄𑀓𑁄”𑁇

(238.)

961\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀭𑀺𑀤𑁂𑀯𑁄? 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀆𑀤𑁂𑀯𑁄 𑀧𑀭𑀺𑀤𑁂𑀯𑁄 𑀆𑀤𑁂𑀯𑀦𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑀸 𑀆𑀤𑁂𑀯𑀺𑀢𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸𑀘𑀸 𑀧𑀮𑀸𑀧𑁄 𑀯𑀺𑀧𑁆𑀧𑀮𑀸𑀧𑁄 𑀮𑀸𑀮𑀧𑁆𑀧𑁄 𑀮𑀸𑀮𑀧𑁆𑀧𑀦𑀸 𑀮𑀸𑀮𑀧𑁆𑀧𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀭𑀺𑀤𑁂𑀯𑁄”𑁇

(239.)

962\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀬𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇

(240.)

963\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁, 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁, 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁”𑁇

(241.)

964\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑀸𑀬𑀸𑀲𑁄? 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀆𑀬𑀸𑀲𑁄 𑀉𑀧𑀸𑀬𑀸𑀲𑁄 𑀆𑀬𑀸𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀬𑀸𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑀸𑀬𑀸𑀲𑁄”𑁇

(242.)

965\. 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻𑀢𑀺, 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀕𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀕𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
