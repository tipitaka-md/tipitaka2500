# 𑁬𑁇𑁨𑁇𑁯𑁇𑁪 𑀅𑀓𑀼𑀲𑀮𑀘𑀺𑀢𑁆𑀢

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀅𑀓𑀼𑀲𑀮𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀓𑀼𑀲𑀮𑀘𑀺𑀢𑁆𑀢

(286.)

1171. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀧𑀝𑀺𑀖𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀧𑀝𑀺𑀖𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀦 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀲𑀤𑁆𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀕𑀦𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀭𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁, 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀝𑀺𑀖𑀁, 𑀧𑀝𑀺𑀖𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄, 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(287.)

1172. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸” …𑀧𑁂…  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

1173. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸”𑁇

1174. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀝𑀺𑀖𑀁? 𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄…𑀧𑁂…  𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀝𑀺𑀖𑀁”𑁇

1175. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀝𑀺𑀖𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄? 𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑀸 𑀢𑀤𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀝𑀺𑀖𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄”𑁇

1176. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀁, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄” …𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
