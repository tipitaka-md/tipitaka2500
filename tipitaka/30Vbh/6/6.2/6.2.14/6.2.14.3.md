# 𑁬𑁇𑁨𑁇𑁧𑁪𑁇𑁩 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀦𑁆𑀢𑀻𑀭𑀡𑀘𑀺𑀢𑁆𑀢

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀓𑀯𑀺𑀧𑀸𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀦𑁆𑀢𑀻𑀭𑀡𑀘𑀺𑀢𑁆𑀢

(353.)

1321\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀢𑀢𑁆𑀢𑀸 𑀉𑀧𑀘𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀧𑀸𑀓𑀸 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁, 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄, 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(354.)

1322\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1323\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1324\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁”𑁇

1325\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁”𑁇

1326\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

1327\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀦𑁂𑀯 𑀲𑀸𑀢𑀁 𑀦𑀸𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸”𑁇

1328\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄? 𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑀸 𑀢𑀤𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄”𑁇

1329\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀁, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄”𑁇

1330\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀚𑀸𑀢𑀺 𑀲𑀜𑁆𑀚𑀸𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺”𑁇

1331\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀚𑀭𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀭𑀡𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀚𑀭𑀸? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀚𑀭𑀸 𑀚𑀻𑀭𑀡𑀢𑀸 𑀆𑀬𑀼𑀦𑁄 𑀲𑀁𑀳𑀸𑀦𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀭𑀸”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀭𑀡𑀁? 𑀬𑁄 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀔𑀬𑁄 𑀯𑀬𑁄 𑀪𑁂𑀤𑁄 𑀧𑀭𑀺𑀪𑁂𑀤𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀭𑀡𑀁”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀚𑀭𑀸, 𑀇𑀤𑀜𑁆𑀘 𑀫𑀭𑀡𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁”𑁇

1332\. 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀕𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀕𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

---

1333\. 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀓𑀯𑀺𑀧𑀸𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑁄𑁇

  
𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬𑀁𑁇

1334\. 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
