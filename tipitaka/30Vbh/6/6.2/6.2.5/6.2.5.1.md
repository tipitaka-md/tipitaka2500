# 𑁬𑁇𑁨𑁇𑁫𑁇𑁧 𑀅𑀓𑀼𑀲𑀮𑀘𑀺𑀢𑁆𑀢

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀅𑀓𑀼𑀲𑀮𑀘𑀺𑀢𑁆𑀢

(248.)

989\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀲𑀳𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀲𑀤𑁆𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀕𑀦𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀭𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀸 𑀬𑀁 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀸𑀭𑀩𑁆𑀪, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁, 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(249.)

990\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

991\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

992\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

993\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁”𑁇

994\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁 𑀳𑀤𑀬𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁”𑁇

995\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

996\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸”𑁇

997\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀦𑀼𑀦𑀬𑁄 𑀅𑀦𑀼𑀭𑁄𑀥𑁄 𑀦𑀦𑁆𑀤𑀻 𑀦𑀦𑁆𑀤𑀺𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸”𑁇

998\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁? 𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀦𑁆𑀢𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀽𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑁆𑀨𑀦𑁆𑀤𑀺𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑁄𑀚𑀦𑀁 𑀕𑀸𑀳𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀳𑁄 𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀣𑁄 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀁 𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁”𑁇

999\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄”𑁇

1000\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀚𑀸𑀢𑀺 𑀲𑀜𑁆𑀚𑀸𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺”𑁇

1001\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀚𑀭𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀭𑀡𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀚𑀭𑀸? 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀚𑀭𑀸 𑀚𑀻𑀭𑀡𑀢𑀸 𑀆𑀬𑀼𑀦𑁄 𑀲𑀁𑀳𑀸𑀦𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀭𑀸”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀭𑀡𑀁? 𑀬𑁄 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀔𑀬𑁄 𑀯𑀬𑁄 𑀪𑁂𑀤𑁄 𑀧𑀭𑀺𑀪𑁂𑀤𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀭𑀡𑀁”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀚𑀭𑀸, 𑀇𑀤𑀜𑁆𑀘 𑀫𑀭𑀡𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁”𑁇

1002\. 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀗𑁆𑀕𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀕𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁄𑀥𑀸𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁧:*𑁧*)

1003\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁, 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(251.)

1004\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1005\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1006\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1007\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁”𑁇

1008\. 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀨𑀲𑁆𑀲𑀁, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇

1009\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄” …𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁨:*𑁨*)

(252.)

1010\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(253.)

1011\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1012\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1013\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1014\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1015\. 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀯𑀢𑁆𑀢𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1016\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁”𑁇

1017\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄” …𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁩:*𑁩*)

(254.)

1018\. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(255.)

1019\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1020\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1021\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1022\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1023\. 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸, 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀯𑀢𑁆𑀢𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1024\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁”𑁇

1025\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄” …𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁪:*𑁪*)
