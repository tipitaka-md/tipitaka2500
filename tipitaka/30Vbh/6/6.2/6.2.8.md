# 𑁬𑁇𑁨𑁇𑁮 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀘𑀢𑀼𑀓𑁆𑀓

(272.)

1098. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁, 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀁; 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁; 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀨𑀲𑁆𑀲𑁄; 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸; 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀢𑀡𑁆𑀳𑀸; 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(273.)

1099. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1100. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1101. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1102. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1103. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1104. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀁”𑁇

1105. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1106. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁”𑁇

1107. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀁”𑁇

1108. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

1109. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁”𑁇

1110. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸”𑁇

1111. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀨𑀲𑁆𑀲𑁄”𑁇

1112. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸”𑁇

1113. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸”𑁇

1127. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1128. 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀨𑀲𑁆𑀲𑀁, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇

1129. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

1131. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁; 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁; 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀨𑀲𑁆𑀲𑁄; 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸; 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀢𑀡𑁆𑀳𑀸; 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

1132. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1133. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1134. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1135. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1136. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1137. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1138. 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀯𑀢𑁆𑀢𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1139. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1140. 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀯𑀢𑁆𑀢𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1141. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁”𑁇

1142. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1143. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

1144. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁” …𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁩:*𑁧𑁫*)

1145. 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸; 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄; 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁; 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁, 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁; 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁; 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀨𑀲𑁆𑀲𑁄; 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑁂𑀤𑀦𑀸; 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀢𑀡𑁆𑀳𑀸; 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

(279.)

1146. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1147. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1148. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

1149. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1150. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄? 𑀬𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁄”𑁇

1151. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀦𑀸𑀫𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1152. 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀬𑀁 𑀭𑀽𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀯𑀢𑁆𑀢𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1153. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁”𑁇

1154. 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸, 𑀬𑀜𑁆𑀘 𑀭𑀽𑀧𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀯𑀢𑁆𑀢𑀢𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1155. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁”𑁇

1156. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀦𑀸𑀫𑀁? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀫𑀁”𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀭𑀽𑀧𑀁? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀉𑀧𑀘𑀬𑁄, 𑀬𑀁 𑀯𑀸 𑀧𑀦𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀣𑀺 𑀭𑀽𑀧𑀁 𑀘𑀺𑀢𑁆𑀢𑀚𑀁 𑀘𑀺𑀢𑁆𑀢𑀳𑁂𑀢𑀼𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀦𑀸𑀫𑀁, 𑀇𑀤𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀦𑀸𑀫𑀭𑀽𑀧𑀁”𑁇

1157. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄? 𑀬𑁄 𑀨𑀲𑁆𑀲𑁄 𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀦𑀸 𑀲𑀫𑁆𑀨𑀼𑀲𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄”𑁇

1158. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀺 𑀙𑀝𑁆𑀞𑀸𑀬𑀢𑀦𑀁”𑁇

1159. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸” …𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁪:*𑁧𑁬*)
