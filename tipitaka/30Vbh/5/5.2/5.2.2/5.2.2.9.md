# 𑁫𑁇𑁨𑁇𑁨𑁇𑁯 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

925. 𑀦𑁄 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀲𑁄𑀍𑀅𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑀸”𑀢𑀺, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸”𑀢𑀺𑀧𑀺𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀧𑀭𑀸𑀫𑀸𑀲𑀸”𑀢𑀺𑁇 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸”𑀢𑀺𑀧𑀺𑁇
