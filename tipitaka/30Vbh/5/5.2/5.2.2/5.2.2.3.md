# 𑁫𑁇𑁨𑁇𑁨𑁇𑁩 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀤𑀼𑀓

### 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

914\. 𑀦𑁄 𑀆𑀲𑀯𑀸𑁇 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀸𑀲𑀯𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑀸𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑀧𑀺𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀲𑀸𑀲𑀯𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑁇

915\. 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺𑀧𑀺, “𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑀧𑀺𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑀸”𑀢𑀺, 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀸 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑀸”𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀸”𑀢𑀺𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑀸𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀲𑀸𑀲𑀯𑀸”𑀢𑀺𑀧𑀺, “𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀅𑀦𑀸𑀲𑀯𑀸”𑀢𑀺𑀧𑀺𑁇
