# 𑁫𑁇𑁨𑁇𑁧 𑀢𑀺𑀓

### 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀢𑀺𑀓

(223.)

888\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀓𑀼𑀲𑀮𑀁𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 (𑁧)

889\. 𑀤𑁆𑀯𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑀧𑀺𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁨)

890\. 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 (𑁩)

891\. 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 (𑁪)

892\. 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 (𑁫)

893\. 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀁𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀁𑁇 𑀏𑀓𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 (𑁬)

894\. 𑀏𑀓𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺𑁇 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀦 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀦𑁆”𑀢𑀺𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺, “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸”𑀢𑀺𑀧𑀺𑁇 (𑁭)

895\. 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

896\. 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀁𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 (𑁯)

897\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑁇 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 (𑁧𑁦)

898\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑁂𑀓𑁆𑀔𑀸𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑁂𑀓𑁆𑀔𑀁𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑁂𑀓𑁆𑀔𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 (𑁧𑁧)

899\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑁇 (𑁧𑁨)

900\. 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁩)

901\. 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀳𑀻𑀦𑀁𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀳𑀻𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 (𑁧𑁪)

902\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀁𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 (𑁧𑁫)

903\. 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄”𑀢𑀺𑀧𑀺𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻”𑀢𑀺𑀧𑀺𑁇 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻”𑀢𑀺𑀧𑀺𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁬)

904\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄, 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄”𑀢𑀺𑁇 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄𑁇 (𑁧𑁭)

905\. 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 (𑁧𑁮)

906\. 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺𑁇 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁯)

907\. 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁨𑁦)

908\. 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺𑁇 (𑁨𑁧)

909\. 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 𑀲𑀢𑁆𑀢𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 (𑁨𑁨)
