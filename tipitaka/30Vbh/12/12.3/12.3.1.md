# 𑁧𑁨𑁇𑁩𑁇𑁧 𑀢𑀺𑀓

### 𑀛𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀢𑀺𑀓

(640.)

1907\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 (𑁧)

1908\. 𑀢𑀻𑀡𑀺 𑀛𑀸𑀦𑀸𑀦𑀺—  𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁—  𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 (𑁨)

1909\. 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 (𑁩)

1910\. 𑀲𑀺𑀬𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 (𑁪)

1911\. 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 (𑁫)

1912\. 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁—  𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁, 𑀢𑀻𑀡𑀺 𑀛𑀸𑀦𑀸𑀦𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸𑁇 (𑁬)

1913\. 𑀤𑁆𑀯𑁂 𑀛𑀸𑀦𑀸𑀦𑀺—  𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀻𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀢𑀻𑀡𑀺 𑀛𑀸𑀦𑀸𑀦𑀺—  𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁—  𑀏𑀢𑁆𑀣𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁𑁇 (𑁭)

1914\. 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

1915\. 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 (𑁯)

1916\. 𑀲𑀺𑀬𑀸 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀸𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 (𑁧𑁦)

1917\. 𑀲𑀺𑀬𑀸 𑀲𑁂𑀓𑁆𑀔𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑀸, 𑀲𑀺𑀬𑀸 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 (𑁧𑁧)

1918\. 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑁇 (𑁧𑁨)

1919\. 𑀢𑀻𑀡𑀺 𑀛𑀸𑀦𑀸𑀦𑀺 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺; 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀺𑀬𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁; 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 “𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁩)

1920\. 𑀲𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀡𑀻𑀢𑀸𑁇 (𑁧𑁪)

1921\. 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 (𑁧𑁫)

1922\. 𑀢𑀻𑀡𑀺 𑀛𑀸𑀦𑀸𑀦𑀺 𑀦 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄”𑀢𑀺𑀧𑀺; 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 “𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀦𑁆”𑀢𑀺𑀧𑀺 “𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀻”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁬)

1923\. 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄𑁇 (𑁧𑁭)

1924\. 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 (𑁧𑁮)

1925\. 𑀢𑀻𑀡𑀺 𑀛𑀸𑀦𑀸𑀦𑀺 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸”𑀢𑀺𑀧𑀺; 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 “𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁧𑁯)

1926\. 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁨𑁦)

1927\. 𑀢𑀻𑀡𑀺 𑀛𑀸𑀦𑀸𑀦𑀺 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺, “𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑀧𑀺𑁇 (𑁨𑁧)

1928\. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 (𑁨𑁨)
