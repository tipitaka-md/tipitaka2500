# 𑁧𑁨𑁇𑁨𑁇𑁨 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

* 𑀛𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

(626.)

1880\. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑀜𑁆𑀜𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
