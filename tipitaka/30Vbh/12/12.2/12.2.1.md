# 𑁧𑁨𑁇𑁨𑁇𑁧 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

* 𑀛𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀪𑀸𑀚𑀦𑀻𑀬

* 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀓𑀼𑀲𑀮

(623.)

1868\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀛𑀸𑀦𑀸𑀦𑀺—  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁, 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁𑁇

(624.)

1869\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀯𑀺𑀘𑀸𑀭𑁄, 𑀧𑀻𑀢𑀺, 𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁧)

1870\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀺𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀧𑀻𑀢𑀺, 𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁨)

1871\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁩)

1872\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁪)

---

1873\. 𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑁇



(625.)

1874\. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀯𑀺𑀘𑀸𑀭𑁄, 𑀧𑀻𑀢𑀺, 𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁧)

1875\. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀯𑀺𑀘𑀸𑀭𑁄, 𑀧𑀻𑀢𑀺, 𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁨)

1876\. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀢𑀺𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀧𑀻𑀢𑀺, 𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁩)

1877\. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁪)

1878\. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀭𑀽𑀧𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀤𑀼𑀯𑀗𑁆𑀕𑀺𑀓𑀁 𑀛𑀸𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀜𑁆𑀘𑀫𑀁 𑀛𑀸𑀦𑀁”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀛𑀸𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁫)
