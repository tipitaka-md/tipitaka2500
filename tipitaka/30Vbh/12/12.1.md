# 𑁧𑁨𑁇𑁧 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

### 𑀛𑀸𑀦𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬

(508.)

1699. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻, 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁄 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀧𑀼𑀩𑁆𑀩𑀭𑀢𑁆𑀢𑀸𑀧𑀭𑀭𑀢𑁆𑀢𑀁 𑀚𑀸𑀕𑀭𑀺𑀬𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀢𑀘𑁆𑀘𑀁 𑀦𑁂𑀧𑀓𑁆𑀓𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑁇

1700. 𑀲𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀆𑀮𑁄𑀓𑀺𑀢𑁂 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂 𑀧𑀲𑀸𑀭𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀺𑀢𑁂 𑀧𑀻𑀢𑁂 𑀔𑀸𑀬𑀺𑀢𑁂 𑀲𑀸𑀬𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑀓𑀫𑁆𑀫𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀢𑁂 𑀞𑀺𑀢𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀚𑀸𑀕𑀭𑀺𑀢𑁂 𑀪𑀸𑀲𑀺𑀢𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇

1701. 𑀲𑁄 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀚𑀢𑀺 𑀅𑀭𑀜𑁆𑀜𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀓𑀦𑁆𑀤𑀭𑀁 𑀕𑀺𑀭𑀺𑀕𑀼𑀳𑀁 𑀲𑀼𑀲𑀸𑀦𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀁 𑀧𑀮𑀸𑀮𑀧𑀼𑀜𑁆𑀚𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁 𑀯𑀺𑀚𑀦𑀯𑀸𑀢𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀳𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀁𑁇 𑀲𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇

1702. 𑀲𑁄 𑀅𑀪𑀺𑀚𑁆𑀛𑀁 𑀮𑁄𑀓𑁂 𑀧𑀳𑀸𑀬 𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀁 𑀧𑀳𑀸𑀬 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀪𑀽𑀢𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁 𑀧𑀳𑀸𑀬 𑀯𑀺𑀕𑀢𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀮𑁄𑀓𑀲𑀜𑁆𑀜𑀻 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀧𑀳𑀸𑀬 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀁 𑀧𑀳𑀸𑀬 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀣𑀁𑀓𑀣𑀻 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇

1703. 𑀲𑁄 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀳𑀸𑀬 𑀘𑁂𑀢𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀧𑀜𑁆𑀜𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀭𑀡𑁂 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  “𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻”𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 “𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄”𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 “𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆”𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 “𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻”𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

1707. “𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀦𑁆”𑀢𑀺 𑀲𑀻𑀮𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀆𑀤𑀺 𑀘𑀭𑀡𑀁 𑀲𑀁𑀬𑀫𑁄 𑀲𑀁𑀯𑀭𑁄 𑀫𑁄𑀓𑁆𑀔𑀁 𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 “𑀲𑀁𑀯𑀭𑁄”𑀢𑀺 𑀓𑀸𑀬𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄𑁇 “𑀲𑀁𑀯𑀼𑀢𑁄”𑀢𑀺 𑀇𑀫𑀺𑀦𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀼𑀧𑁂𑀢𑁄 𑀉𑀧𑀸𑀕𑀢𑁄 𑀲𑀫𑀼𑀧𑀸𑀕𑀢𑁄 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄”𑀢𑀺𑁇

1708. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺 𑀇𑀭𑀺𑀬𑀢𑀺 𑀯𑀢𑁆𑀢𑀢𑀺 𑀧𑀸𑀮𑁂𑀢𑀺 𑀬𑀧𑁂𑀢𑀺 𑀬𑀸𑀧𑁂𑀢𑀺 𑀘𑀭𑀢𑀺 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

1709. “𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀸𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀦𑀸𑀘𑀸𑀭𑁄𑁇

1710. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀦𑀸𑀘𑀸𑀭𑁄? 𑀓𑀸𑀬𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀸𑀘𑀸𑀭𑁄”𑁇 𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁 𑀅𑀦𑀸𑀘𑀸𑀭𑁄𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀯𑁂𑀍𑀉𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀧𑀢𑁆𑀢𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀧𑀼𑀧𑁆𑀨𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀨𑀮𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀦𑀸𑀦𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀘𑀸𑀝𑀼𑀓𑀫𑁆𑀬𑀢𑀸𑀬 𑀯𑀸 𑀫𑀼𑀕𑁆𑀕𑀲𑀽𑀧𑁆𑀬𑀢𑀸𑀬 𑀯𑀸 𑀧𑀸𑀭𑀺𑀪𑀝𑁆𑀬𑀢𑀸𑀬 𑀯𑀸 𑀚𑀗𑁆𑀖𑀧𑁂𑀲𑀦𑀺𑀓𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁂𑀦 𑀚𑀻𑀯𑀺𑀓𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀸𑀘𑀸𑀭𑁄”𑁇

1711. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀆𑀘𑀸𑀭𑁄? 𑀓𑀸𑀬𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀅𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀘𑀸𑀭𑁄”𑁇 𑀲𑀩𑁆𑀩𑁄𑀧𑀺 𑀲𑀻𑀮𑀲𑀁𑀯𑀭𑁄 𑀆𑀘𑀸𑀭𑁄𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀦 𑀯𑁂𑀍𑀉𑀤𑀸𑀦𑁂𑀦 𑀦 𑀧𑀢𑁆𑀢𑀤𑀸𑀦𑁂𑀦 𑀦 𑀧𑀼𑀧𑁆𑀨𑀤𑀸𑀦𑁂𑀦 𑀦 𑀨𑀮𑀤𑀸𑀦𑁂𑀦 𑀦 𑀲𑀺𑀦𑀸𑀦𑀤𑀸𑀦𑁂𑀦 𑀦 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀤𑀸𑀦𑁂𑀦 𑀦 𑀘𑀸𑀝𑀼𑀓𑀫𑁆𑀬𑀢𑀸𑀬 𑀦 𑀫𑀼𑀕𑁆𑀕𑀲𑀽𑀧𑁆𑀬𑀢𑀸𑀬 𑀦 𑀧𑀸𑀭𑀺𑀪𑀝𑁆𑀬𑀢𑀸𑀬 𑀦 𑀚𑀗𑁆𑀖𑀧𑁂𑀲𑀦𑀺𑀓𑁂𑀦 𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁂𑀦 𑀚𑀻𑀯𑀺𑀓𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀘𑀸𑀭𑁄”𑁇

1712. “𑀕𑁄𑀘𑀭𑁄”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀕𑁄𑀘𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀕𑁄𑀘𑀭𑁄𑁇

1713. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀕𑁄𑀘𑀭𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀯𑁂𑀲𑀺𑀬𑀸𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀥𑀯𑀸𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀣𑀼𑀮𑁆𑀮𑀓𑀼𑀫𑀸𑀭𑀺𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀧𑀡𑁆𑀟𑀓𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀧𑀸𑀦𑀸𑀕𑀸𑀭𑀕𑁄𑀘𑀭𑁄 𑀯𑀸, 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁂𑀳𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀦 𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀦; 𑀬𑀸𑀦𑀺 𑀯𑀸 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀅𑀦𑁄𑀧𑀸𑀦𑀪𑀽𑀢𑀸𑀦𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀓𑀧𑀭𑀺𑀪𑀸𑀲𑀓𑀸𑀦𑀺 𑀅𑀦𑀢𑁆𑀣𑀓𑀸𑀫𑀸𑀦𑀺 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑀸𑀦𑀺 𑀅𑀨𑀸𑀲𑀼𑀓𑀓𑀸𑀫𑀸𑀦𑀺 𑀅𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀲𑁂𑀯𑀢𑀺 𑀪𑀚𑀢𑀺 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀕𑁄𑀘𑀭𑁄”𑁇

1714. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀕𑁄𑀘𑀭𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀦 𑀯𑁂𑀲𑀺𑀬𑀸𑀕𑁄𑀘𑀭𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀯𑀺𑀥𑀯𑀸𑀕𑁄𑀘𑀭𑁄 𑀦 𑀣𑀼𑀮𑁆𑀮𑀓𑀼𑀫𑀸𑀭𑀺𑀕𑁄𑀘𑀭𑁄 𑀦 𑀧𑀡𑁆𑀟𑀓𑀕𑁄𑀘𑀭𑁄 𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀕𑁄𑀘𑀭𑁄 𑀦 𑀧𑀸𑀦𑀸𑀕𑀸𑀭𑀕𑁄𑀘𑀭𑁄, 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁂𑀳𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀦 𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀦; 𑀬𑀸𑀦𑀺 𑀯𑀸 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀲𑀤𑁆𑀥𑀸𑀦𑀺 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀑𑀧𑀸𑀦𑀪𑀽𑀢𑀸𑀦𑀺 𑀓𑀸𑀲𑀸𑀯𑀧𑀚𑁆𑀚𑁄𑀢𑀸𑀦𑀺 𑀇𑀲𑀺𑀯𑀸𑀢𑀧𑀝𑀺𑀯𑀸𑀢𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸𑀦𑀺 𑀳𑀺𑀢𑀓𑀸𑀫𑀸𑀦𑀺 𑀨𑀸𑀲𑀼𑀓𑀓𑀸𑀫𑀸𑀦𑀺 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀲𑁂𑀯𑀢𑀺 𑀪𑀚𑀢𑀺 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀕𑁄𑀘𑀭𑁄”𑁇 𑀇𑀢𑀺 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀆𑀘𑀸𑀭𑁂𑀦 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀕𑁄𑀘𑀭𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺𑁇

1715. “𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑀸 𑀯𑀚𑁆𑀚𑀸? 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀯𑀚𑁆𑀚𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀑𑀭𑀫𑀢𑁆𑀢𑀓𑀸𑀦𑀺 𑀮𑀳𑀼𑀲𑀸𑀦𑀺 𑀮𑀳𑀼𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀲𑀁𑀬𑀫𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀲𑀁𑀯𑀭𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀸𑀦𑀺—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑀸 𑀯𑀚𑁆𑀚𑀸”𑁇 𑀇𑀢𑀺 𑀇𑀫𑁂𑀲𑀼 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀘 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀘𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻”𑀢𑀺𑁇

1716. “𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀽”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀺𑀓𑁆𑀔𑀸? 𑀘𑀢𑀲𑁆𑀲𑁄 𑀲𑀺𑀓𑁆𑀔𑀸—  𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀺𑀓𑁆𑀔𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀺𑀓𑁆𑀔𑀸, 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀲𑀺𑀓𑁆𑀔𑀸, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑁇 𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀲𑀺𑀓𑁆𑀔𑀸𑀬𑁄”𑁇 𑀇𑀢𑀺 𑀇𑀫𑀸𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀼 𑀲𑀩𑁆𑀩𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀲𑀩𑁆𑀩𑀣𑀸 𑀲𑀩𑁆𑀩𑀁 𑀅𑀲𑁂𑀲𑀁 𑀦𑀺𑀲𑁆𑀲𑁂𑀲𑀁 𑀲𑀫𑀸𑀤𑀸𑀬 𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀽”𑀢𑀺𑁇

1717. “𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁄”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸𑁇

1718. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀦 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀦 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀦 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀦 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀸 𑀇𑀫𑁂𑀲𑀁 𑀙𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀕𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁄𑀧𑀦𑀸 𑀅𑀦𑀸𑀭𑀓𑁆𑀔𑁄 𑀅𑀲𑀁𑀯𑀭𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸”𑁇

1719. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀦𑀸𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀭𑀓𑁆𑀔𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀦𑀸𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀜𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀭𑀓𑁆𑀔𑀢𑀺 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀸 𑀇𑀫𑁂𑀲𑀁 𑀙𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀕𑀼𑀢𑁆𑀢𑀺 𑀕𑁄𑀧𑀦𑀸 𑀆𑀭𑀓𑁆𑀔𑁄 𑀲𑀁𑀯𑀭𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸”𑁇 𑀇𑀫𑀸𑀬 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸𑀬 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀼𑀧𑁂𑀢𑁄…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁄”𑀢𑀺𑁇

1721. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀢𑀺 𑀤𑀯𑀸𑀬 𑀫𑀤𑀸𑀬 𑀫𑀡𑁆𑀟𑀦𑀸𑀬 𑀯𑀺𑀪𑀽𑀲𑀦𑀸𑀬𑁇 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀪𑁄𑀚𑀦𑁂—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸”𑀢𑀺𑁇

1722. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀬𑁄𑀦𑀺𑀲𑁄 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀢𑀺—  “𑀦𑁂𑀯 𑀤𑀯𑀸𑀬 𑀦 𑀫𑀤𑀸𑀬 𑀦 𑀫𑀡𑁆𑀟𑀦𑀸𑀬 𑀦 𑀯𑀺𑀪𑀽𑀲𑀦𑀸𑀬, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀇𑀫𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀞𑀺𑀢𑀺𑀬𑀸 𑀬𑀸𑀧𑀦𑀸𑀬 𑀯𑀺𑀳𑀺𑀁𑀲𑀽𑀧𑀭𑀢𑀺𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬, 𑀇𑀢𑀺 𑀧𑀼𑀭𑀸𑀡𑀜𑁆𑀘 𑀯𑁂𑀤𑀦𑀁 𑀧𑀝𑀺𑀳𑀗𑁆𑀔𑀸𑀫𑀺 𑀦𑀯𑀜𑁆𑀘 𑀯𑁂𑀤𑀦𑀁 𑀦 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀸𑀢𑁆𑀭𑀸 𑀘 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑀯𑀚𑁆𑀚𑀢𑀸 𑀘 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀸𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀪𑁄𑀚𑀦𑁂—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸”𑁇 𑀇𑀫𑀸𑀬 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽”𑀢𑀺𑁇

(519.)

1723. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑀭𑀢𑁆𑀢𑀸𑀧𑀭𑀭𑀢𑁆𑀢𑀁 𑀚𑀸𑀕𑀭𑀺𑀬𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀯𑀲𑀁 𑀘𑀗𑁆𑀓𑀫𑁂𑀦 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀆𑀯𑀭𑀡𑀻𑀬𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀞𑀫𑀬𑀸𑀫𑀁 𑀘𑀗𑁆𑀓𑀫𑁂𑀦 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀆𑀯𑀭𑀡𑀻𑀬𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀬𑀸𑀫𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀻𑀳𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺 𑀧𑀸𑀤𑁂 𑀧𑀸𑀤𑀁 𑀅𑀘𑁆𑀘𑀸𑀥𑀸𑀬 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀬𑀸𑀫𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀘𑀗𑁆𑀓𑀫𑁂𑀦 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀆𑀯𑀭𑀡𑀻𑀬𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑀭𑀢𑁆𑀢𑀸𑀧𑀭𑀭𑀢𑁆𑀢𑀁 𑀚𑀸𑀕𑀭𑀺𑀬𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑁇

1726. “𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸? 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸”𑁇 𑀇𑀢𑀺 𑀢𑁂 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑁂 𑀆𑀲𑁂𑀯𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄”𑀢𑀺𑁇

1727. 𑀓𑀣𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀆𑀮𑁄𑀓𑀺𑀢𑁂 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂 𑀧𑀲𑀸𑀭𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀺𑀢𑁂 𑀧𑀻𑀢𑁂 𑀔𑀸𑀬𑀺𑀢𑁂 𑀲𑀸𑀬𑀺𑀢𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑀓𑀫𑁆𑀫𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀕𑀢𑁂 𑀞𑀺𑀢𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀚𑀸𑀕𑀭𑀺𑀢𑁂 𑀪𑀸𑀲𑀺𑀢𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀆𑀮𑁄𑀓𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀲𑀸𑀭𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀺𑀢𑁂 𑀧𑀻𑀢𑁂 𑀔𑀸𑀬𑀺𑀢𑁂 𑀲𑀸𑀬𑀺𑀢𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑀓𑀫𑁆𑀫𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀢𑁂 𑀞𑀺𑀢𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀚𑀸𑀕𑀭𑀺𑀢𑁂 𑀪𑀸𑀲𑀺𑀢𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

(524.)

1728. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑀺 𑀲𑀭𑀡𑀢𑀸 𑀥𑀸𑀭𑀡𑀢𑀸 𑀅𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀅𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸 𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀢𑀺𑀩𑀮𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺”𑁇

(525.)

1729. “𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸 𑀯𑀺𑀘𑀬𑁄 𑀧𑀯𑀺𑀘𑀬𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸 𑀉𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀁 𑀓𑁄𑀲𑀮𑁆𑀮𑀁 𑀦𑁂𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑁂𑀪𑀩𑁆𑀬𑀸 𑀘𑀺𑀦𑁆𑀢𑀸 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸 𑀪𑀽𑀭𑀻𑀫𑁂𑀥𑀸 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁 𑀧𑀢𑁄𑀤𑁄 𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀢𑁆𑀣𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑁄 𑀧𑀜𑁆𑀜𑀸𑀆𑀮𑁄𑀓𑁄 𑀧𑀜𑁆𑀜𑀸𑀑𑀪𑀸𑀲𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀫𑀸𑀬 𑀘 𑀲𑀢𑀺𑀬𑀸 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀏𑀯𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀆𑀮𑁄𑀓𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀲𑀸𑀭𑁂𑀢𑀺, 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀺𑀢𑁂 𑀧𑀻𑀢𑁂 𑀔𑀸𑀬𑀺𑀢𑁂 𑀲𑀸𑀬𑀺𑀢𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑀓𑀫𑁆𑀫𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀕𑀢𑁂 𑀞𑀺𑀢𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀲𑀼𑀢𑁆𑀢𑁂 𑀚𑀸𑀕𑀭𑀺𑀢𑁂 𑀪𑀸𑀲𑀺𑀢𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇

(526.)

1730. “𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀦𑁆”𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀘𑁂𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑀁 𑀕𑀳𑀝𑁆𑀞𑁂𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀢𑀁 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁𑁇 𑀤𑀽𑀭𑁂 𑀘𑁂𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑀁 𑀕𑀳𑀝𑁆𑀞𑁂𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀢𑀁 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁𑁇

(527.)

1731. “𑀲𑁂𑀦𑀸𑀲𑀦𑀦𑁆”𑀢𑀺 𑀫𑀜𑁆𑀘𑁄𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀧𑀻𑀞𑀫𑁆𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀪𑀺𑀲𑀺𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀫𑁆𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀯𑀺𑀳𑀸𑀭𑁄𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀅𑀟𑁆𑀠𑀬𑁄𑀕𑁄𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀧𑀸𑀲𑀸𑀤𑁄𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀅𑀝𑁆𑀝𑁄𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀫𑀸𑀍𑀑𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀮𑁂𑀡𑀫𑁆𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀕𑀼𑀳𑀸𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀫𑁆𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁, 𑀯𑁂𑀍𑀉𑀕𑀼𑀫𑁆𑀩𑁄𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁𑁇 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁𑁇

(528.)

1732. “𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀚𑀢𑀻”𑀢𑀺 𑀇𑀫𑀁 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀚𑀢𑀺 𑀲𑀫𑁆𑀪𑀚𑀢𑀺 𑀲𑁂𑀯𑀢𑀺 𑀦𑀺𑀲𑁂𑀯𑀢𑀺 𑀲𑀁𑀲𑁂𑀯𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀚𑀢𑀻”𑀢𑀺𑁇

(529.)

1733. “𑀅𑀭𑀜𑁆𑀜𑀦𑁆”𑀢𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀺 𑀇𑀦𑁆𑀤𑀔𑀻𑀮𑀸 𑀲𑀩𑁆𑀩𑀫𑁂𑀢𑀁 𑀅𑀭𑀜𑁆𑀜𑀁𑁇

(530.)

1734. “𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀦𑁆”𑀢𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁𑀬𑁂𑀯 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁𑁇 𑀧𑀩𑁆𑀩𑀢𑁄𑀬𑁂𑀯 𑀧𑀩𑁆𑀩𑀢𑁄𑁇 𑀓𑀦𑁆𑀤𑀭𑀸𑀬𑁂𑀯 𑀓𑀦𑁆𑀤𑀭𑀸𑁇 𑀕𑀺𑀭𑀺𑀕𑀼𑀳𑀸𑀬𑁂𑀯 𑀕𑀺𑀭𑀺𑀕𑀼𑀳𑀸𑁇 𑀲𑀼𑀲𑀸𑀦𑀁𑀬𑁂𑀯 𑀲𑀼𑀲𑀸𑀦𑀁𑁇 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁄𑀬𑁂𑀯 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁄𑁇 𑀧𑀮𑀸𑀮𑀧𑀼𑀜𑁆𑀚𑁄𑀬𑁂𑀯 𑀧𑀮𑀸𑀮𑀧𑀼𑀜𑁆𑀚𑁄𑁇

(531.)

1735. “𑀯𑀦𑀧𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀤𑀽𑀭𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 “𑀯𑀦𑀧𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀯𑀦𑀲𑀡𑁆𑀟𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 “𑀯𑀦𑀧𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀪𑀻𑀲𑀦𑀓𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 “𑀯𑀦𑀧𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 “𑀯𑀦𑀧𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 “𑀯𑀦𑀧𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀽𑀧𑀘𑀸𑀭𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 “𑀯𑀦𑀧𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀸𑀦𑀫𑁂𑀢𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇

(532.)

1736. “𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀦𑁆”𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀘𑁂𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑀁 𑀕𑀳𑀝𑁆𑀞𑁂𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀢𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁𑁇 𑀤𑀽𑀭𑁂 𑀘𑁂𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑀁 𑀕𑀳𑀝𑁆𑀞𑁂𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀢𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁𑁇

(533.)

1737. “𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀦𑁆”𑀢𑀺 𑀬𑀤𑁂𑀯 𑀢𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀢𑀤𑁂𑀯 𑀢𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁𑁇 𑀬𑀤𑁂𑀯 𑀢𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁 𑀢𑀤𑁂𑀯 𑀢𑀁 𑀯𑀺𑀚𑀦𑀯𑀸𑀢𑀁𑁇 𑀬𑀤𑁂𑀯 𑀢𑀁 𑀯𑀺𑀚𑀦𑀯𑀸𑀢𑀁 𑀢𑀤𑁂𑀯 𑀢𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀳𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀓𑀁𑁇 𑀬𑀤𑁂𑀯 𑀢𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀳𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀢𑀤𑁂𑀯 𑀢𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀁𑁇

(534.)

1738. “𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸”𑀢𑀺 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄 𑀯𑀸𑁇

(535.)

1739. “𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸”𑀢𑀺 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸𑁇

(536.)

1740. “𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬𑀸”𑀢𑀺 𑀉𑀚𑀼𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀸𑀬𑁄 𑀞𑀺𑀢𑁄 𑀧𑀡𑀺𑀳𑀺𑀢𑁄𑁇

(537.)

1741. “𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺”𑁇 𑀅𑀬𑀁 𑀲𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀦𑀸𑀲𑀺𑀓𑀕𑁆𑀕𑁂 𑀯𑀸 𑀫𑀼𑀔𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀯𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸”𑀢𑀺𑁇

(538.)

1742. “𑀅𑀪𑀺𑀚𑁆𑀛𑀁 𑀮𑁄𑀓𑁂 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀪𑀺𑀚𑁆𑀛𑀸”𑁇

1743. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑁄𑀓𑁄? 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀮𑁄𑀓𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀮𑁄𑀓𑁄”𑁇 𑀅𑀬𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀇𑀫𑀫𑁆𑀳𑀺 𑀮𑁄𑀓𑁂 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀪𑀺𑀚𑁆𑀛𑀁 𑀮𑁄𑀓𑁂 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇

(539.)

1744. “𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑁂𑀦 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇

(540.)

1745. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺 𑀇𑀭𑀺𑀬𑀢𑀺 𑀯𑀢𑁆𑀢𑀢𑀺 𑀧𑀸𑀮𑁂𑀢𑀺 𑀬𑀧𑁂𑀢𑀺 𑀬𑀸𑀧𑁂𑀢𑀺 𑀘𑀭𑀢𑀺 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(541.)

1746. “𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀪𑀺𑀚𑁆𑀛𑀸? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄…𑀧𑁂…  𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀪𑀺𑀚𑁆𑀛𑀸”𑁇

1747. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀲𑁄𑀥𑁂𑀢𑀺 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀫𑁄𑀘𑁂𑀢𑀺 𑀯𑀺𑀫𑁄𑀘𑁂𑀢𑀺 𑀧𑀭𑀺𑀫𑁄𑀘𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

(542.)

1748. “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀤𑁄𑀲𑁄𑁇

1749. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄? 𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀓𑁄𑀧𑁄 𑀧𑀓𑁄𑀧𑁄 𑀲𑀫𑁆𑀧𑀓𑁄𑀧𑁄 𑀤𑁄𑀲𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑀫𑁆𑀧𑀤𑁄𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑁄 𑀓𑁄𑀥𑁄 𑀓𑀼𑀚𑁆𑀛𑀦𑀸 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀢𑁆𑀢𑀁 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄”𑁇

1750. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀤𑁄𑀲𑁄? 𑀬𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀲𑁄 𑀧𑀤𑁄𑀲𑁄, 𑀬𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀬𑀜𑁆𑀘 𑀧𑀤𑁄𑀲𑁄 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇

(543.)

1751. “𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄”𑀢𑀺𑁇

(544.)

1752. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(545.)

1753. “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀢𑁆𑀣𑀺 𑀧𑀤𑁄𑀲𑁄𑁇

1754. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄? 𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄…𑀧𑁂…  𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄”𑁇

1755. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀤𑁄𑀲𑁄? 𑀬𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀲𑁄 𑀧𑀤𑁄𑀲𑁄, 𑀬𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄𑁇

1756. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀫𑁆𑀳𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀸 𑀲𑁄𑀥𑁂𑀢𑀺 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀫𑁄𑀘𑁂𑀢𑀺 𑀯𑀺𑀫𑁄𑀘𑁂𑀢𑀺 𑀧𑀭𑀺𑀫𑁄𑀘𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

(546.)

1757. “𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀣𑀺𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀤𑁆𑀥𑀁𑁇

1758. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀣𑀺𑀦𑀁? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀬𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀑𑀮𑀻𑀬𑀦𑀸 𑀲𑀮𑁆𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀦𑀁 𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀣𑀺𑀦𑀁 𑀣𑀺𑀬𑀦𑀸 𑀣𑀺𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀣𑀺𑀦𑀁”𑁇

1759. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀺𑀤𑁆𑀥𑀁? 𑀬𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀬𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀑𑀦𑀸𑀳𑁄 𑀧𑀭𑀺𑀬𑁄𑀦𑀸𑀳𑁄 𑀅𑀦𑁆𑀢𑁄𑀲𑀫𑁄𑀭𑁄𑀥𑁄 𑀫𑀺𑀤𑁆𑀥𑀁 𑀲𑀼𑀧𑁆𑀧𑀁 𑀧𑀘𑀮𑀸𑀬𑀺𑀓𑀸 𑀲𑀼𑀧𑁆𑀧𑀁 𑀲𑀼𑀧𑁆𑀧𑀦𑀸 𑀲𑀼𑀧𑁆𑀧𑀺𑀢𑀢𑁆𑀢𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀺𑀤𑁆𑀥𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀣𑀺𑀦𑀁 𑀇𑀤𑀜𑁆𑀘 𑀫𑀺𑀤𑁆𑀥𑀁 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇

1762. “𑀆𑀮𑁄𑀓𑀲𑀜𑁆𑀜𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀜𑁆𑀜𑀸? 𑀬𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀜𑁆𑀜𑀸”𑁇 𑀅𑀬𑀁 𑀲𑀜𑁆𑀜𑀸 𑀆𑀮𑁄𑀓𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀝𑀸 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀮𑁄𑀓𑀲𑀜𑁆𑀜𑀻”𑀢𑀺𑁇

(550.)

1763. “𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺”𑁇

1764. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀫𑀸𑀬 𑀘 𑀲𑀢𑀺𑀬𑀸 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺𑁇

1765. “𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀣𑀺𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀤𑁆𑀥𑀁𑁇

1766. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀣𑀺𑀦𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀣𑀺𑀦𑀁”𑁇

1767. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀺𑀤𑁆𑀥𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀺𑀤𑁆𑀥𑀁”𑁇

1768. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀫𑁆𑀳𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸 𑀲𑁄𑀥𑁂𑀢𑀺 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀫𑁄𑀘𑁂𑀢𑀺 𑀯𑀺𑀫𑁄𑀘𑁂𑀢𑀺 𑀧𑀭𑀺𑀫𑁄𑀘𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

1769. “𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁𑁇

1770. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀯𑀽𑀧𑀲𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀪𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁”𑁇

1771. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁? 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀅𑀯𑀚𑁆𑀚𑁂 𑀯𑀚𑁆𑀚𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀯𑀚𑁆𑀚𑁂 𑀅𑀯𑀚𑁆𑀚𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 𑀫𑀦𑁄𑀯𑀺𑀮𑁂𑀔𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀇𑀤𑀜𑁆𑀘 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇

1781. “𑀅𑀓𑀣𑀁𑀓𑀣𑀻 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀽”𑀢𑀺 𑀇𑀫𑀸𑀬 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀦 𑀓𑀗𑁆𑀔𑀢𑀺 𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀅𑀓𑀣𑀁𑀓𑀣𑀻 𑀳𑁄𑀢𑀺 𑀦𑀺𑀓𑁆𑀓𑀣𑀁𑀓𑀣𑀻 𑀯𑀺𑀓𑀣𑀁𑀓𑀣𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀓𑀣𑀁𑀓𑀣𑀻 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀽”𑀢𑀺𑁇

1782. “𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸? 𑀬𑀸 𑀓𑀗𑁆𑀔𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀦𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀦𑁄𑀯𑀺𑀮𑁂𑀔𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸”𑁇

1783. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄 𑀫𑀸𑀦𑀲𑀁…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀸𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 𑀇𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀸𑀬 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀲𑁄𑀥𑁂𑀢𑀺 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀺 𑀫𑁄𑀘𑁂𑀢𑀺 𑀯𑀺𑀫𑁄𑀘𑁂𑀢𑀺 𑀧𑀭𑀺𑀫𑁄𑀘𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

1784. “𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇

1785. “𑀘𑁂𑀢𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂”𑀢𑀺 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀸𑁇

1806. “𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀦𑁆”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀯𑀺𑀘𑀸𑀭𑁄𑁇

1807. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀬𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇

1808. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀘𑀸𑀭𑁄? 𑀬𑁄 𑀘𑀸𑀭𑁄 𑀅𑀦𑀼𑀘𑀸𑀭𑁄 𑀯𑀺𑀘𑀸𑀭𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁄 𑀉𑀧𑀯𑀺𑀘𑀸𑀭𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀦𑀢𑀸 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀘𑀸𑀭𑁄”𑁇 𑀇𑀢𑀺 𑀅𑀬𑀜𑁆𑀘 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀦𑁆”𑀢𑀺𑁇

(577.)

1809. “𑀲𑀫𑀸𑀥𑀺𑀚𑀦𑁆”𑀢𑀺 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑁄 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁—  𑀢𑁂 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀺 𑀚𑀸𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀜𑁆𑀚𑀸𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑀸𑀥𑀺𑀚𑀦𑁆”𑀢𑀺𑁇

(578.)

1810. “𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀦𑁆”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀧𑀻𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀁𑁇

1811. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀻𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀻𑀢𑀺”𑁇

1812. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀼𑀔𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀁”𑁇 𑀇𑀤𑀁 𑀲𑀼𑀔𑀁 𑀇𑀫𑀸𑀬 𑀧𑀻𑀢𑀺𑀬𑀸 𑀲𑀳𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀳𑀚𑀸𑀢𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇

(579.)

1813. “𑀤𑀼𑀢𑀺𑀬𑀦𑁆”𑀢𑀺 𑀕𑀡𑀦𑀸𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀢𑀸 𑀤𑀼𑀢𑀺𑀬𑀁𑁇 𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺 𑀤𑀼𑀢𑀺𑀬𑀁𑁇

(580.)

1814. “𑀛𑀸𑀦𑀦𑁆”𑀢𑀺 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑁄, 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇

(581.)

1815. “𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀸”𑀢𑀺 𑀬𑁄 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

(582.)

1816. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(583.)

1817. “𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀻𑀢𑀺? 𑀬𑀸 𑀧𑀻𑀢𑀺 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀆𑀫𑁄𑀤𑀦𑀸 𑀧𑀫𑁄𑀤𑀦𑀸 𑀳𑀸𑀲𑁄 𑀧𑀳𑀸𑀲𑁄 𑀯𑀺𑀢𑁆𑀢𑀺 𑀑𑀤𑀕𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀻𑀢𑀺”𑁇 𑀅𑀬𑀁 𑀧𑀻𑀢𑀺 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸”𑀢𑀺𑁇

(584.)

1818. “𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀫𑀚𑁆𑀛𑀢𑁆𑀢𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸”𑁇 𑀇𑀫𑀸𑀬 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄”𑀢𑀺𑁇

(585.)

1819. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(586.)

1820. “𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺”𑁇

1821. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁? 𑀬𑀸 𑀧𑀜𑁆𑀜𑀸 𑀧𑀚𑀸𑀦𑀦𑀸…𑀧𑁂…  𑀅𑀫𑁄𑀳𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀫𑀸𑀬 𑀘 𑀲𑀢𑀺𑀬𑀸 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑁂𑀦 𑀉𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄”𑀢𑀺𑁇

(587.)

1822. “𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀻”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀼𑀔𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀁”𑁇

1823. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀸𑀬𑁄? 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀬𑁄”𑁇 𑀇𑀤𑀁 𑀲𑀼𑀔𑀁 𑀇𑀫𑀺𑀦𑀸 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀻”𑀢𑀺𑁇

(588.)

1824. “𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀭𑀺𑀬𑀸? 𑀅𑀭𑀺𑀬𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀘 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑀸 𑀘𑁇 𑀢𑁂 𑀇𑀫𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀧𑀓𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

(589.)

1825. “𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀫𑀚𑁆𑀛𑀢𑁆𑀢𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸”𑁇

1826. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺”𑁇

1827. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀼𑀔𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀫𑀸𑀬 𑀘 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀇𑀫𑀸𑀬 𑀘 𑀲𑀢𑀺𑀬𑀸 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀲𑀼𑀔𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀭𑀺𑀬𑀢𑀺 𑀯𑀢𑁆𑀢𑀢𑀺 𑀧𑀸𑀮𑁂𑀢𑀺 𑀬𑀧𑁂𑀢𑀺 𑀬𑀸𑀧𑁂𑀢𑀺 𑀘𑀭𑀢𑀺 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻”𑀢𑀺𑁇

(590.)

1828. “𑀢𑀢𑀺𑀬𑀦𑁆”𑀢𑀺 𑀕𑀡𑀦𑀸𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀢𑀸 𑀢𑀢𑀺𑀬𑀁𑁇 𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺 𑀢𑀢𑀺𑀬𑀁𑁇

(591.)

1829. “𑀛𑀸𑀦𑀦𑁆”𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀲𑀢𑀺, 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁, 𑀲𑀼𑀔𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇

(592.)

1830. “𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀸”𑀢𑀺 𑀬𑁄 𑀢𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

(593.)

1831. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(594.)

1832. “𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸”𑀢𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀔𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇

1833. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀼𑀔𑀁? 𑀬𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀁”𑁇

1834. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀁? 𑀬𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀓𑁆𑀔𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀲𑀼𑀔𑀁 𑀇𑀤𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸”𑀢𑀺𑁇

(595.)

1835. “𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸”𑀢𑀺 𑀅𑀢𑁆𑀣𑀺 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁𑁇

1836. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀲𑀸𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀸𑀢𑀸 𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁”𑁇

1837. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁? 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀅𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀲𑀸𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀲𑀸𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁”𑁇 𑀇𑀢𑀺 𑀇𑀤𑀜𑁆𑀘 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀤𑀜𑁆𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸”𑀢𑀺𑁇

(596.)

1838. “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀦𑁂𑀯 𑀲𑀸𑀢𑀁 𑀦𑀸𑀲𑀸𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀁 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀘𑁂𑀢𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇

(597.)

1839. “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆”𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸? 𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀸 𑀫𑀚𑁆𑀛𑀢𑁆𑀢𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸”𑁇

1840. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑀺? 𑀬𑀸 𑀲𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀢𑀺”𑁇 𑀅𑀬𑀁 𑀲𑀢𑀺 𑀇𑀫𑀸𑀬 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀬 𑀯𑀺𑀯𑀝𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆”𑀢𑀺𑁇

(598.)

1841. “𑀘𑀢𑀼𑀢𑁆𑀣𑀦𑁆”𑀢𑀺 𑀕𑀡𑀦𑀸𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀢𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁, 𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇

(599.)

1842. “𑀛𑀸𑀦𑀦𑁆”𑀢𑀺 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀲𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸𑁇

(600.)

1843. “𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀸”𑀢𑀺 𑀬𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

(601.)

1844. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(602.)

1845. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲 𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀬𑁄”𑁇 𑀇𑀫𑀸 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀬𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸”𑀢𑀺𑁇

(603.)

1846. “𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸? 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀜𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀬𑁄”𑁇 𑀇𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀬𑁄 𑀲𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀸 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀸 𑀩𑁆𑀬𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑁄𑀲𑀺𑀢𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀓𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸”𑀢𑀺𑁇

(604.)

1847. “𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸? 𑀅𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀦𑁄𑀥𑀸𑀢𑀼 𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀯𑀸 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀫𑀗𑁆𑀕𑀺𑀲𑁆𑀲 𑀯𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀬𑁄”𑁇 𑀇𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀬𑁄 𑀦 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸”𑀢𑀺𑁇

(605.)

1848. “𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄”𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀆𑀓𑀸𑀲𑁄? 𑀬𑁄 𑀆𑀓𑀸𑀲𑁄 𑀆𑀓𑀸𑀲𑀕𑀢𑀁 𑀅𑀖𑀁 𑀅𑀖𑀕𑀢𑀁 𑀯𑀺𑀯𑀭𑁄 𑀯𑀺𑀯𑀭𑀕𑀢𑀁 𑀅𑀲𑀫𑁆𑀨𑀼𑀝𑁆𑀞𑀁 𑀘𑀢𑀽𑀳𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑁂𑀳𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀆𑀓𑀸𑀲𑁄”𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀆𑀓𑀸𑀲𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀞𑀧𑁂𑀢𑀺 𑀲𑀡𑁆𑀞𑀧𑁂𑀢𑀺 𑀅𑀦𑀦𑁆𑀢𑀁 𑀨𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄”𑀢𑀺𑁇

(606.)

1849. “𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀦𑁆”𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇

(607.)

1850. “𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀸”𑀢𑀺 𑀬𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

(608.)

1851. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(609.)

1852. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑀸”𑀢𑀺 𑀇𑀫𑀁 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑀸”𑀢𑀺𑁇

(610.)

1853. “𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆”𑀢𑀺 𑀢𑀁𑀬𑁂𑀯 𑀆𑀓𑀸𑀲𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺 𑀅𑀦𑀦𑁆𑀢𑀁 𑀨𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆”𑀢𑀺𑁇

(611.)

1854. “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀦𑁆”𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇

(612.)

1855. “𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀸”𑀢𑀺 𑀬𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

(613.)

1856. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(614.)

1857. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑀸”𑀢𑀺 𑀇𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑀸”𑀢𑀺𑁇

(615.)

1858. “𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻”𑀢𑀺 𑀢𑀁𑀬𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀯𑀺𑀪𑀸𑀯𑁂𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺, “𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻”𑀢𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻”𑀢𑀺𑁇

(616.)

1859. “𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀦𑁆”𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇

(617.)

1860. “𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀸”𑀢𑀺 𑀬𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

(618.)

1861. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

(619.)

1862. “𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑀸”𑀢𑀺 𑀇𑀫𑀁 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑀸”𑀢𑀺𑁇

1863. “𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻”𑀢𑀺 𑀢𑀁𑀬𑁂𑀯 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀦𑁆𑀢𑀢𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀯𑀲𑁂𑀲𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻”𑀢𑀺𑁇

(620.)

1864. “𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀦𑁆”𑀢𑀺 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇

(621.)

1865. “𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀸”𑀢𑀺 𑀬𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀮𑀸𑀪𑁄 𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀨𑀼𑀲𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

(622.)

1866. “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺 𑀇𑀭𑀺𑀬𑀢𑀺 𑀯𑀢𑁆𑀢𑀢𑀺 𑀧𑀸𑀮𑁂𑀢𑀺 𑀬𑀧𑁂𑀢𑀺 𑀬𑀸𑀧𑁂𑀢𑀺 𑀘𑀭𑀢𑀺 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

---

1867. 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀪𑀸𑀚𑀦𑀻𑀬𑀁𑁇
