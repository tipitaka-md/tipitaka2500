# 𑁧𑁮𑁇𑁧𑁨 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

(1044.)

3253\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀓𑀢𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸? 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑁆𑀦𑀁 𑀆𑀬𑀢𑀦𑀸𑀦𑀁 𑀓𑀢𑀺 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀓𑀢𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸? 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀦𑁆𑀦𑀁 𑀥𑀸𑀢𑀽𑀦𑀁 𑀓𑀢𑀺 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀓𑀢𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀓𑀢𑀺 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀓𑀢𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀓𑀢𑀺 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸?

3254\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀮𑁄𑀓𑀺𑀬𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇

3255\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀙 𑀳𑁂𑀢𑀽 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀮𑁄𑀓𑀺𑀬𑁄𑁇 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀙 𑀨𑀲𑁆𑀲𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑁄, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄𑁇 𑀙 𑀯𑁂𑀤𑀦𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀙 𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀙 𑀘𑁂𑀢𑀦𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑁇 𑀙 𑀘𑀺𑀢𑁆𑀢𑀸 𑀮𑁄𑀓𑀺𑀬𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑀸, 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸𑀢𑀺𑁇

3256\. _𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀤𑁆𑀯𑁂 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸,_  
_𑀤𑀺𑀝𑁆𑀞𑀸 𑀓𑀼𑀲𑀮𑀯𑁂𑀤𑀦𑀸;_  
_𑀯𑀺𑀧𑀸𑀓𑀸 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀭𑀽𑀧𑀮𑁄𑀓𑀺𑀬𑀸𑀢𑀺𑁇_  


---

3257\. 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

3258\. 𑀯𑀺𑀪𑀗𑁆𑀕𑀧𑀓𑀭𑀡𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
