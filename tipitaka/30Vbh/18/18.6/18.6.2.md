# 𑁧𑁮𑁇𑁬𑁇𑁨 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡

* 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑀓𑀫𑁆𑀫𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀯𑀸𑀭

* 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡

(1022.)

3152\. 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀯𑀲𑁆𑀲𑀲𑀢𑀁, 𑀅𑀧𑁆𑀧𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 (𑁧)

(1023.)

3153\. 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀬𑀸𑀦𑀺 𑀫𑀸𑀦𑀼𑀲𑀓𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀸𑀲 𑀯𑀲𑁆𑀲𑀸𑀦𑀺, 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀏𑀲𑁄 𑀏𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑁄𑁇 𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀺𑀁𑀲 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀸𑀲𑁄𑁇 𑀢𑁂𑀦 𑀫𑀸𑀲𑁂𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀸𑀲𑀺𑀬𑁄 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁄𑁇 𑀢𑁂𑀦 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁂𑀦 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀕𑀡𑀦𑀸𑀬 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺? 𑀦𑀯𑀼𑀢𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨)

3154\. 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀬𑀁 𑀫𑀸𑀦𑀼𑀲𑀓𑀁 𑀯𑀲𑁆𑀲𑀲𑀢𑀁, 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀏𑀲𑁄 𑀏𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑁄𑁇 𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀺𑀁𑀲 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀸𑀲𑁄𑁇 𑀢𑁂𑀦 𑀫𑀸𑀲𑁂𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀸𑀲𑀺𑀬𑁄 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁄𑁇 𑀢𑁂𑀦 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁂𑀦 𑀤𑀺𑀩𑁆𑀩𑀁 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀕𑀡𑀦𑀸𑀬 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺? 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀬𑁄 𑀲𑀝𑁆𑀞𑀺 𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁩)

3155\. 𑀬𑀸𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀬𑀸𑀦𑀺 𑀫𑀸𑀦𑀼𑀲𑀓𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺, 𑀬𑀸𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀏𑀲𑁄 𑀏𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑁄𑁇 𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀺𑀁𑀲𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀸𑀲𑁄𑁇 𑀢𑁂𑀦 𑀫𑀸𑀲𑁂𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀸𑀲𑀺𑀬𑁄 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁄𑁇 𑀢𑁂𑀦 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁂𑀦 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀬𑀸𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀕𑀡𑀦𑀸𑀬 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺? 𑀘𑀼𑀤𑁆𑀤𑀲𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀬𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁪)

3156\. 𑀢𑀼𑀲𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀬𑀸𑀦𑀺 𑀫𑀸𑀦𑀼𑀲𑀓𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺, 𑀢𑀼𑀲𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀏𑀲𑁄 𑀏𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑁄𑁇 𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀺𑀁𑀲𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀸𑀲𑁄𑁇 𑀢𑁂𑀦 𑀫𑀸𑀲𑁂𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀸𑀲𑀺𑀬𑁄 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁄𑁇 𑀢𑁂𑀦 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁂𑀦 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀼𑀲𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀕𑀡𑀦𑀸𑀬 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺? 𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲 𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀬𑁄 𑀲𑀝𑁆𑀞𑀺 𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁫)

3157\. 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀬𑀸𑀦𑀺 𑀫𑀸𑀦𑀼𑀲𑀓𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺, 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀏𑀲𑁄 𑀏𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑁄𑁇 𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀺𑀁𑀲𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀸𑀲𑁄𑁇 𑀢𑁂𑀦 𑀫𑀸𑀲𑁂𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀸𑀲𑀺𑀬𑁄 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁄𑁇 𑀢𑁂𑀦 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁂𑀦 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀕𑀡𑀦𑀸𑀬 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺? 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀲𑀢𑀸𑀦𑀺 𑀢𑀺𑀁𑀲𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀬𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁬)

3158\. 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀬𑀸𑀦𑀺 𑀫𑀸𑀦𑀼𑀲𑀓𑀸𑀦𑀺 𑀲𑁄𑀍𑀅𑀲 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺, 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀏𑀲𑁄 𑀏𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑁄𑁇 𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀺𑀁𑀲𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀸𑀲𑁄𑁇 𑀢𑁂𑀦 𑀫𑀸𑀲𑁂𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀸𑀲𑀺𑀬𑁄 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁄𑁇 𑀢𑁂𑀦 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁂𑀦 𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀲𑁄𑀍𑀅𑀲 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀕𑀡𑀦𑀸𑀬 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺? 𑀦𑀯 𑀘 𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀲𑀢𑀸𑀦𑀺 𑀏𑀓𑀯𑀻𑀲𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀬𑁄 𑀲𑀝𑁆𑀞𑀺 𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀻𑀢𑀺𑁇 (𑁭)

3159\. _𑀙 𑀏𑀢𑁂 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀏𑀓𑀲𑀗𑁆𑀔𑀸𑀢𑁄,_  
_𑀆𑀬𑀼 𑀪𑀯𑀢𑀺 𑀓𑀺𑀢𑁆𑀢𑀓𑁄𑁇_  


3160\. _𑀤𑁆𑀯𑀸𑀤𑀲 𑀓𑁄𑀝𑀺𑀲𑀢𑀁 𑀢𑁂𑀲𑀁,_  
_𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀜𑁆𑀘 𑀓𑁄𑀝𑀺𑀬𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀲 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀲𑁆𑀲𑀕𑁆𑀕𑁂𑀦 𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑀢𑀺𑁇_

3161\. 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀸𑀭𑀺𑀲𑀚𑁆𑀚𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀓𑀧𑁆𑀧𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑁄 𑀪𑀸𑀕𑁄𑁇 (𑁮)

3162\. 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀉𑀧𑀟𑁆𑀠𑀓𑀧𑁆𑀧𑁄𑁇 (𑁯)

3163\. 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀧𑀡𑀻𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀧𑀡𑀻𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀓𑀧𑁆𑀧𑁄𑁇 (𑁧𑁦)

(1025.)

3164\. 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀤𑁆𑀯𑁂 𑀓𑀧𑁆𑀧𑀸𑁇 (𑁧𑁧)

3165\. 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀓𑀧𑁆𑀧𑀸𑁇 (𑁧𑁨)

3166\. 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀡𑀻𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀡𑀻𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀆𑀪𑀲𑁆𑀲𑀭𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀅𑀝𑁆𑀞 𑀓𑀧𑁆𑀧𑀸𑁇 (𑁧𑁩)

(1026.)

3167\. 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑀼𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀲𑁄𑀍𑀅𑀲 𑀓𑀧𑁆𑀧𑀸𑁇 (𑁧𑁪)

3168\. 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀲𑀼𑀪𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲 𑀓𑀧𑁆𑀧𑀸𑁇 (𑁧𑁫)

3169\. 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀡𑀻𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀧𑀡𑀻𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀪𑀓𑀺𑀡𑁆𑀳𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀘𑀢𑀼𑀲𑀝𑁆𑀞𑀺 𑀓𑀧𑁆𑀧𑀸𑁇 (𑁧𑁬)

(1027.)

3170\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀦𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀦𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀙𑀦𑁆𑀤𑀦𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀧𑀡𑀺𑀥𑀺𑀦𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑀦𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀦𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀦𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑁂𑀳𑀧𑁆𑀨𑀮𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀅𑀯𑀺𑀳𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀅𑀢𑀧𑁆𑀧𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀼𑀤𑀲𑁆𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀼𑀤𑀲𑁆𑀲𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁧𑁭)

3171\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘 𑀯𑁂𑀳𑀧𑁆𑀨𑀮𑀸𑀦𑀜𑁆𑀘 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀧𑀜𑁆𑀘𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀺𑁇 (𑁧𑁮)

3172\. 𑀅𑀯𑀺𑀳𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀁𑁇 (𑁧𑁯)

3173\. 𑀅𑀢𑀧𑁆𑀧𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀤𑁆𑀯𑁂 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨𑁦)

3174\. 𑀲𑀼𑀤𑀲𑁆𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨𑁧)

3175\. 𑀲𑀼𑀤𑀲𑁆𑀲𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀅𑀝𑁆𑀞 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨𑁨)

3176\. 𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀲𑁄𑀍𑀅𑀲 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨𑁩)

(1028.)

3177\. 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀯𑀻𑀲𑀢𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨𑁪)

3178\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨𑁫)

3179\. 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀲𑀝𑁆𑀞𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑁇 (𑁨𑁬)

3180\. 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀓𑀁 𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁? 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀻𑀢𑀺𑁇 (𑁨𑁭)

(1029.)

3181\. _𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀧𑀼𑀜𑁆𑀜𑀢𑁂𑀚𑁂𑀦,_  
_𑀓𑀸𑀫𑀭𑀽𑀧𑀕𑀢𑀺𑀁 𑀕𑀢𑀸;_  
_𑀪𑀯𑀕𑁆𑀕𑀢𑀫𑁆𑀧𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀧𑀼𑀦𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁𑁇_  


3182\. _𑀢𑀸𑀯 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀘𑀯𑀦𑁆𑀢𑀺 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀬𑀸;_  
_𑀦𑀢𑁆𑀣𑀺 𑀓𑁄𑀘𑀺 𑀪𑀯𑁄 𑀦𑀺𑀘𑁆𑀘𑁄,_  
_𑀇𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀸𑁇_  


3183\. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀥𑀻𑀭𑀸 𑀦𑀺𑀧𑀓𑀸,_  
_𑀦𑀺𑀧𑀼𑀡𑀸 𑀅𑀢𑁆𑀣𑀘𑀺𑀦𑁆𑀢𑀓𑀸;_  
_𑀚𑀭𑀸𑀫𑀭𑀡𑀫𑁄𑀓𑁆𑀔𑀸𑀬,_  
_𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


3184\. _𑀪𑀸𑀯𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀘𑀺𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑁄𑀕𑀥𑀕𑀸𑀫𑀺𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑀢𑀺𑁇_
