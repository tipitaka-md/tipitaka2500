# 𑁧𑁮𑁇𑁬𑁇𑁧 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑀓𑀫𑁆𑀫

* 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑀓𑀫𑁆𑀫𑀆𑀬𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑀓𑀫𑁆𑀫

(1021.)

3147\. 𑀤𑁂𑀯𑀸𑀢𑀺𑁇 𑀢𑀬𑁄 𑀤𑁂𑀯𑀸—  𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀤𑁂𑀯𑀸, 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀤𑁂𑀯𑀸, 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀤𑁂𑀯𑀸𑁇

3148\. 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀤𑁂𑀯𑀸 𑀦𑀸𑀫—  𑀭𑀸𑀚𑀸𑀦𑁄, 𑀤𑁂𑀯𑀺𑀬𑁄, 𑀓𑀼𑀫𑀸𑀭𑀸𑁇

3149\. 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀤𑁂𑀯𑀸 𑀦𑀸𑀫—  𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑁂 𑀤𑁂𑀯𑁂 𑀉𑀧𑀸𑀤𑀸𑀬 𑀢𑀤𑀼𑀧𑀭𑀺 𑀤𑁂𑀯𑀸𑁇

3150\. 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀤𑁂𑀯𑀸 𑀦𑀸𑀫—  𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

3151\. 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸, 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀓𑀢𑁆𑀣 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸, 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑀸𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀲𑀺𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇
