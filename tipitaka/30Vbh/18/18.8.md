# 𑁧𑁮𑁇𑁮 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀯𑀸𑀭

* 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀯𑀸𑀭

(1032.)

3195\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀓𑀢𑀺 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀓𑀢𑀺 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸?

(1033.)

3196\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁧)

3197\. 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 (𑁨)

3198\. 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁩)

3199\. 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 (𑁪)

3200\. 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀘𑀼𑀤𑁆𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀦𑀯 𑀳𑁂𑀢𑀽 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀲𑀢𑁆𑀢 𑀨𑀲𑁆𑀲𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁫--𑁧𑁨)

(1034.)

3201\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀓𑀢𑀺 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀓𑀢𑀺 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸?

(1035.)

3202\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁧)

3203\. 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 (𑁨)

3204\. 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀙 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁩)

3205\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 (𑁪)

3206\. 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁𑁇 (𑁫)

3207\. 𑀦𑀯 𑀳𑁂𑀢𑀽 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀙 𑀨𑀲𑁆𑀲𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑁄𑁇 𑀙 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀙 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀙 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀙 𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁬--𑁧𑁨)
