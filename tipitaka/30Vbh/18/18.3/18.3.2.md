# 𑁧𑁮𑁇𑁩𑁇𑁨 𑀭𑀽𑀧𑀥𑀸𑀢𑀼

* 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀯𑀸𑀭

* 𑀭𑀽𑀧𑀥𑀸𑀢𑀼

(1001.)

3052\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀓𑀢𑀺 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀓𑀢𑀺 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸?

(1002.)

3053\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3054\. 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3055\. 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3056\. 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇

3057\. 𑀢𑁂𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3058\. 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀙 𑀳𑁂𑀢𑀽 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3059\. 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3060\. 𑀙 𑀨𑀲𑁆𑀲𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇

3061\. 𑀙 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀙 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀙 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀙 𑀘𑀺𑀢𑁆𑀢𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇
