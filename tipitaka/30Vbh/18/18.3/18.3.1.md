# 𑁧𑁮𑁇𑁩𑁇𑁧 𑀓𑀸𑀫𑀥𑀸𑀢𑀼

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀓𑀸𑀫𑀥𑀸𑀢𑀼

(999.)

3042\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀓𑀢𑀺 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀓𑀢𑀺 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸?

(1000.)

3043\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3044\. 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3045\. 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3046\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁; 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇

3047\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀢𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3048\. 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀙 𑀳𑁂𑀢𑀽 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3049\. 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3050\. 𑀙 𑀨𑀲𑁆𑀲𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇

3051\. 𑀙 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀙 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀙 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀙 𑀘𑀺𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇
