# 𑁧𑁮𑁇𑁩𑁇𑁩 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼

(1003.)

3062. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀓𑀢𑀺 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀓𑀢𑀺 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸?

(1004.)

3063. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3064. 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3065. 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3066. 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀺𑁇

3067. 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇

3068. 𑀘𑀼𑀤𑁆𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3069. 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀙 𑀳𑁂𑀢𑀽 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3070. 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄; 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3071. 𑀙 𑀨𑀲𑁆𑀲𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄𑁇

3072. 𑀙 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀙 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀙 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀙 𑀘𑀺𑀢𑁆𑀢𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸; 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇
