# 𑁧𑁮𑁇𑁧𑁦𑁇𑁧 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀢𑀺𑀓𑀸𑀤𑀺𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

(1038.)

3214\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀓𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸?

3215\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀤𑁆𑀯𑀸𑀬𑀢𑀦𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀲𑁄𑀍𑀅𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀓𑀼𑀲𑀮𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇

3216\. 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀓𑀼𑀲𑀮𑀁𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀙 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇

3217\. 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽 𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽 𑀅𑀓𑀼𑀲𑀮𑀸𑁇 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀙 𑀨𑀲𑁆𑀲𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀙 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀙 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀙 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀙 𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇
