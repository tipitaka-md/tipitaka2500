# 𑁧𑁮𑁇𑁪𑁇𑁩 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

### 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢

(1017.)

3125\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀺 𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀓𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺?

3126\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺—  𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄; 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺—  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁; 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼; 𑀏𑀓𑀁 𑀲𑀘𑁆𑀘𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺—  𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁; 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺—  𑀭𑀽𑀧𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸 𑀤𑁂𑀯𑀸 𑀅𑀳𑁂𑀢𑀼𑀓𑀸 𑀅𑀦𑀸𑀳𑀸𑀭𑀸 𑀅𑀨𑀲𑁆𑀲𑀓𑀸 𑀅𑀯𑁂𑀤𑀦𑀓𑀸 𑀅𑀲𑀜𑁆𑀜𑀓𑀸 𑀅𑀘𑁂𑀢𑀦𑀓𑀸 𑀅𑀘𑀺𑀢𑁆𑀢𑀓𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇
