# 𑁧𑁮𑁇𑁪𑁇𑁨 𑀭𑀽𑀧𑀥𑀸𑀢𑀼

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

### 𑀭𑀽𑀧𑀥𑀸𑀢𑀼

(1015.)

3114. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀺 𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀓𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺?

3115. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁, 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀏𑀓𑀁 𑀲𑀘𑁆𑀘𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺, 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇

(1016.)

3116. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁧)

3117. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁨)

3118. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀘 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀫𑀸 𑀧𑀜𑁆𑀘 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁩)

3119. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑀁 𑀏𑀓𑀁 𑀲𑀘𑁆𑀘𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺? 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀤𑀁 𑀏𑀓𑀁 𑀲𑀘𑁆𑀘𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇 (𑁪)

3120. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑀸𑀦𑀺 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀯𑀸, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀫𑀸𑀦𑀺 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁫)

3121. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀅𑀮𑁄𑀪𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼, 𑀅𑀤𑁄𑀲𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼, 𑀅𑀫𑁄𑀳𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁬)

3122. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁭)

3123. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑁄 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀅𑀬𑀁 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇 (𑁮)

3124. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀫𑀸 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀇𑀤𑀁 𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇 (𑁯--𑁧𑁨)
