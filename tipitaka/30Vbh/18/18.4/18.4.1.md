# 𑁧𑁮𑁇𑁪𑁇𑁧 𑀓𑀸𑀫𑀥𑀸𑀢𑀼

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

### 𑀓𑀸𑀫𑀥𑀸𑀢𑀼

(1007.)

3083. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀢𑀺 𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀓𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺?

3084. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀧𑀭𑀸𑀦𑀺 𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀦𑀯𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀲𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀏𑀓𑀸𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀧𑀭𑀸 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀦𑀯 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀏𑀓𑀁 𑀲𑀘𑁆𑀘𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀘𑀼𑀤𑁆𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀢𑁂𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀧𑀭𑀸𑀦𑀺 𑀢𑁂𑀭𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀧𑀭𑀸𑀦𑀺 𑀦𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀧𑀭𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀤𑁆𑀯𑁂 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀳𑁂𑀢𑀼𑀓𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺; 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇

(1008.)

3085. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁧)

(1009.)

3086. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀲𑁆𑀲 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁, 𑀧𑀞𑀫𑀓𑀧𑁆𑀧𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁, 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀧𑁂𑀢𑀸𑀦𑀁, 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀅𑀲𑀼𑀭𑀸𑀦𑀁, 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀦𑀁, 𑀦𑁂𑀭𑀬𑀺𑀓𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸𑀬𑀢𑀦𑀸𑀦𑀁 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺—  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀢𑁂𑀲𑀁 𑀇𑀫𑀸𑀦𑀺 𑀏𑀓𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇

3087. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀲𑁆𑀲 𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀧𑁂𑀢𑀸𑀦𑀁, 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀅𑀲𑀼𑀭𑀸𑀦𑀁, 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀦𑀁 𑀦𑁂𑀭𑀬𑀺𑀓𑀸𑀦𑀁, 𑀚𑀘𑁆𑀘𑀦𑁆𑀥𑀸𑀦𑀁 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺—  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀢𑁂𑀲𑀁 𑀇𑀫𑀸𑀦𑀺 𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇

3108. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀓𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁, 𑀦𑀧𑀼𑀁𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀢𑁂𑀲𑀁 𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁫)

(1013.)

3109. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀲𑁆𑀲 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁, 𑀧𑀞𑀫𑀓𑀧𑁆𑀧𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁, 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀓𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀜𑀸𑀡𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺—  𑀅𑀮𑁄𑀪𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼, 𑀅𑀤𑁄𑀲𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼, 𑀅𑀫𑁄𑀳𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀢𑁂𑀲𑀁 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇

3110. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁, 𑀧𑀞𑀫𑀓𑀧𑁆𑀧𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁, 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀓𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀸𑀦𑀁 𑀜𑀸𑀡𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀤𑁆𑀯𑁂 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺—  𑀅𑀮𑁄𑀪𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼, 𑀅𑀤𑁄𑀲𑁄 𑀯𑀺𑀧𑀸𑀓𑀳𑁂𑀢𑀼𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀏𑀢𑁂𑀲𑀁 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀳𑁂𑀢𑀽 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀳𑁂𑀢𑀼𑀓𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁬)

(1014.)

3111. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺? 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄, 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄—  𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 (𑁭)

3112. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀓𑀢𑀫𑁄 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄—  𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀅𑀬𑀁 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇 (𑁮)

3113. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀓𑀢𑀫𑀸 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀉𑀧𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀡𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀇𑀤𑀁 𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀢𑀺𑁇 (𑁯--𑁧𑁨)
