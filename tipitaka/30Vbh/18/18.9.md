# 𑁧𑁮𑁇𑁯 𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑀸𑀤𑀺𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

* 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑀸𑀤𑀺𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

(1036.)

3208\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀸, 𑀓𑀢𑀺 𑀲𑀼𑀢𑀸, 𑀓𑀢𑀺 𑀫𑀼𑀢𑀸, 𑀓𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸, 𑀓𑀢𑀺 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀸, 𑀓𑀢𑀺 𑀲𑀼𑀢𑀸, 𑀓𑀢𑀺 𑀫𑀼𑀢𑀸, 𑀓𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸, 𑀓𑀢𑀺 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸?

(1037.)

3209\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀫𑀼𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦 𑀲𑀼𑀢𑁄 𑀦 𑀫𑀼𑀢𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇

3210\. 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀦 𑀲𑀼𑀢𑀁 𑀦 𑀫𑀼𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀦 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀲𑀼𑀢𑀁, 𑀦 𑀫𑀼𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀦 𑀤𑀺𑀝𑁆𑀞𑀁 𑀦 𑀲𑀼𑀢𑀁, 𑀫𑀼𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀲𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇

3211\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼 𑀤𑀺𑀝𑁆𑀞𑀸, 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸, 𑀲𑀼𑀢𑀸, 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀭𑀲𑀥𑀸𑀢𑀼…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸, 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀢𑁂𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇

3212\. 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀫𑀼𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀁 𑀦 𑀲𑀼𑀢𑀁 𑀦 𑀫𑀼𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁𑁇

3213\. 𑀩𑀸𑀯𑀻𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀦𑀯 𑀳𑁂𑀢𑀽 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀲𑀢𑁆𑀢 𑀨𑀲𑁆𑀲𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇 𑀲𑀢𑁆𑀢 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀘𑀺𑀢𑁆𑀢𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸 𑀦 𑀲𑀼𑀢𑀸 𑀦 𑀫𑀼𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑁇
