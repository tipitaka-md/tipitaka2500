# 𑁧𑁮𑁇𑁨𑁇𑁨 𑀭𑀽𑀧𑀥𑀸𑀢𑀼

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀭𑀽𑀧𑀥𑀸𑀢𑀼

(993.)

3004\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑀢𑀺 𑀔𑀦𑁆𑀥𑀸, 𑀓𑀢𑀺 𑀆𑀬𑀢𑀦𑀸, 𑀓𑀢𑀺 𑀥𑀸𑀢𑀼𑀬𑁄, 𑀓𑀢𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀓𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺…𑀧𑁂…  𑀓𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺?

3005\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸, 𑀙 𑀆𑀬𑀢𑀦𑀸𑀦𑀺, 𑀦𑀯 𑀥𑀸𑀢𑀼𑀬𑁄, 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀘𑀼𑀤𑁆𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀨𑀲𑁆𑀲𑀸, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀲𑀜𑁆𑀜𑀸, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺𑁇

(994.)

3006\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸”𑁇 (𑁧)

3007\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀙 𑀆𑀬𑀢𑀦𑀸𑀦𑀺? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀙 𑀆𑀬𑀢𑀦𑀸𑀦𑀺”𑁇 (𑁨)

3008\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀯 𑀥𑀸𑀢𑀼𑀬𑁄? 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼, 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀯 𑀥𑀸𑀢𑀼𑀬𑁄”𑁇 (𑁩)

3009\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺? 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁, 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁, 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺”𑁇 (𑁪)

3010\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀼𑀤𑁆𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺? 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀼𑀤𑁆𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺”𑁇 (𑁫)

3011\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀤𑁆𑀯𑁂 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽𑁇

3012\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽? 𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀅𑀫𑁄𑀳𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽𑁇

3013\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽? 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽𑁇

3014\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽? 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑁂𑀲𑀼 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀮𑁄𑀪𑁄, 𑀅𑀤𑁄𑀲𑁄, 𑀅𑀫𑁄𑀳𑁄—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽𑁇 (𑁬)

3015\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸? 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸”𑁇 (𑁭)

3016\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀨𑀲𑁆𑀲𑀸? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀨𑀲𑁆𑀲𑀸”𑁇 (𑁮)

3017\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀘𑀢𑀲𑁆𑀲𑁄 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀘𑀢𑀲𑁆𑀲𑁄 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺”𑁇 (𑁯--𑁧𑁨)
