# 𑁧𑁮𑁇𑁨𑁇𑁧 𑀓𑀸𑀫𑀥𑀸𑀢𑀼

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀓𑀸𑀫𑀥𑀸𑀢𑀼

(991.)

2990. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑀢𑀺 𑀔𑀦𑁆𑀥𑀸, 𑀓𑀢𑀺 𑀆𑀬𑀢𑀦𑀸𑀦𑀺, 𑀓𑀢𑀺 𑀥𑀸𑀢𑀼𑀬𑁄, 𑀓𑀢𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀓𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀓𑀢𑀺 𑀳𑁂𑀢𑀽, 𑀓𑀢𑀺 𑀆𑀳𑀸𑀭𑀸, 𑀓𑀢𑀺 𑀨𑀲𑁆𑀲𑀸, 𑀓𑀢𑀺 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀢𑀺 𑀲𑀜𑁆𑀜𑀸, 𑀓𑀢𑀺 𑀘𑁂𑀢𑀦𑀸, 𑀓𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺?

2991. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺, 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄, 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀩𑀸𑀯𑀻𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀦𑀯 𑀳𑁂𑀢𑀽, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸, 𑀲𑀢𑁆𑀢 𑀨𑀲𑁆𑀲𑀸, 𑀲𑀢𑁆𑀢 𑀯𑁂𑀤𑀦𑀸, 𑀲𑀢𑁆𑀢 𑀲𑀜𑁆𑀜𑀸, 𑀲𑀢𑁆𑀢 𑀘𑁂𑀢𑀦𑀸, 𑀲𑀢𑁆𑀢 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺𑁇

(992.)

2992. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸”𑁇 (𑁧)

2993. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺? 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁, 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺”𑁇 (𑁨)

2994. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄? 𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀥𑀸𑀢𑀼, 𑀲𑀤𑁆𑀤𑀥𑀸𑀢𑀼, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀥𑀸𑀢𑀼, 𑀕𑀦𑁆𑀥𑀥𑀸𑀢𑀼, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀥𑀸𑀢𑀼, 𑀭𑀲𑀥𑀸𑀢𑀼, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀥𑀸𑀢𑀼, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄”𑁇 (𑁩)

2995. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺? 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁, 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁, 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺”𑁇 (𑁪)

2996. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀩𑀸𑀯𑀻𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺? 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀩𑀸𑀯𑀻𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺”𑁇 (𑁫)

2997. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀯 𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽…𑀧𑁂…  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀦𑀯 𑀳𑁂𑀢𑀽”𑁇 (𑁬)

2998. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸? 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑁄, 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑀸”𑁇 (𑁭)

2999. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀨𑀲𑁆𑀲𑀸? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀨𑀲𑁆𑀲𑀸”𑁇 (𑁮)

3000. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀯𑁂𑀤𑀦𑀸? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀯𑁂𑀤𑀦𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀯𑁂𑀤𑀦𑀸”𑁇 (𑁯)

3001. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀲𑀜𑁆𑀜𑀸? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀲𑀜𑁆𑀜𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀲𑀜𑁆𑀜𑀸”𑁇 (𑁧𑁦)

3002. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀘𑁂𑀢𑀦𑀸? 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸 𑀘𑁂𑀢𑀦𑀸—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀘𑁂𑀢𑀦𑀸”𑁇 (𑁧𑁧)

3003. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺? 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀫𑀦𑁄𑀥𑀸𑀢𑀼, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺”𑁇 (𑁧𑁨)
