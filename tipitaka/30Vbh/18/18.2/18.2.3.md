# 𑁧𑁮𑁇𑁨𑁇𑁩 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼

(995.)

3018\. 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑀢𑀺 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀓𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺?

3019\. 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄, 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀏𑀓𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄, 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

(996.)

3020\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸”𑁇 (𑁧)

3021\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺? 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺”𑁇 (𑁨)

3022\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄”𑁇 (𑁩)

3023\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺? 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁, 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁, 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺”𑁇 (𑁪)

3024\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺? 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺”𑁇 (𑁫)

3025\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀤𑁆𑀯𑁂 𑀅𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽…𑀧𑁂…  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽”𑁇 (𑁬)

3026\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸? 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸”𑁇 (𑁭)

3027\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄”𑁇 (𑁮)

3028\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 (𑁯--𑁧𑁨)
