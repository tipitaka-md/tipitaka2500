# 𑁧𑁮𑁇𑁨𑁇𑁪 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦

(997.)

3029\. 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀓𑀢𑀺 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀓𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺?

3030\. 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺, 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄, 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀤𑁆𑀯𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀙 𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸, 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄, 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸, 𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

(998.)

3031\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸”𑁇 (𑁧)

3032\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺? 𑀫𑀦𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑁂 𑀆𑀬𑀢𑀦𑀸𑀦𑀺”𑁇 (𑁨)

3033\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼, 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼—  𑀇𑀫𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄”𑁇 (𑁩)

3034\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸𑀦𑀺? 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁, 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑁂 𑀲𑀘𑁆𑀘𑀸𑀦𑀺”𑁇 (𑁪)

3035\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺? 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀤𑁆𑀯𑀸𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺”𑁇 (𑁫)

3036\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀙 𑀳𑁂𑀢𑀽? 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽, 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽𑁇

3037\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽? 𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼, 𑀅𑀫𑁄𑀳𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀼—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀓𑀼𑀲𑀮𑀳𑁂𑀢𑀽𑁇

3038\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽? 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀢𑁄 𑀅𑀮𑁄𑀪𑁄, 𑀅𑀤𑁄𑀲𑁄, 𑀅𑀫𑁄𑀳𑁄—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀳𑁂𑀢𑀽𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀙 𑀳𑁂𑀢𑀽”𑁇 (𑁬)

3039\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸? 𑀨𑀲𑁆𑀲𑀸𑀳𑀸𑀭𑁄, 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀳𑀸𑀭𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀳𑀸𑀭𑁄—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸”𑁇 (𑁭)

3040\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀓𑁄 𑀨𑀲𑁆𑀲𑁄”𑁇 (𑁮)

3041\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀓𑀸 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀏𑀓𑀸 𑀘𑁂𑀢𑀦𑀸…𑀧𑁂…  𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁? 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁”𑁇 (𑁯--𑁧𑁨)
