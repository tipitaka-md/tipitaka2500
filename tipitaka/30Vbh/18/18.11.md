# 𑁧𑁮𑁇𑁧𑁧 𑀭𑀽𑀧𑀤𑀼𑀓

### 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀭𑀽𑀧𑀤𑀼𑀓

(1043.)

3251\. 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺 𑀭𑀽𑀧𑀸, 𑀓𑀢𑀺 𑀅𑀭𑀽𑀧𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑀺 𑀭𑀽𑀧𑀸, 𑀓𑀢𑀺 𑀅𑀭𑀽𑀧𑀸?

3252\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀭𑀽𑀧𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀔𑀦𑁆𑀥𑀸 𑀅𑀭𑀽𑀧𑀸𑁇 𑀤𑀲𑀸𑀬𑀢𑀦𑀸 𑀭𑀽𑀧𑀸𑁇 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀅𑀭𑀽𑀧𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀁𑁇 𑀤𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 𑀭𑀽𑀧𑀸𑁇 𑀲𑀢𑁆𑀢 𑀥𑀸𑀢𑀼𑀬𑁄 𑀅𑀭𑀽𑀧𑀸𑁇 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀸𑁇 𑀢𑀻𑀡𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀭𑀽𑀧𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀁𑁇 𑀲𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀭𑀽𑀧𑀸𑁇 𑀘𑀼𑀤𑁆𑀤𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀸𑁇 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀭𑀽𑀧𑀁𑁇 𑀦𑀯 𑀳𑁂𑀢𑀽 𑀅𑀭𑀽𑀧𑀸𑁇 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀭𑀽𑀧𑀁𑁇 𑀢𑀬𑁄 𑀆𑀳𑀸𑀭𑀸 𑀅𑀭𑀽𑀧𑀸𑁇 𑀲𑀢𑁆𑀢 𑀨𑀲𑁆𑀲𑀸 𑀅𑀭𑀽𑀧𑀸𑁇 𑀲𑀢𑁆𑀢 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀘𑁂𑀢𑀦𑀸 𑀲𑀢𑁆𑀢 𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀭𑀽𑀧𑀸𑁇
