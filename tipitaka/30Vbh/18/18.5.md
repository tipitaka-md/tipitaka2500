# 𑁧𑁮𑁇𑁫 𑀪𑀽𑀫𑀦𑁆𑀢𑀭𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

* 𑀥𑀫𑁆𑀫𑀳𑀤𑀬𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀪𑀽𑀫𑀦𑁆𑀢𑀭𑀤𑀲𑁆𑀲𑀦𑀯𑀸𑀭

(1020.)

3138\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑁇

3139\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸? 𑀳𑁂𑀝𑁆𑀞𑀢𑁄 𑀅𑀯𑀻𑀘𑀺𑀦𑀺𑀭𑀬𑀁 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀉𑀧𑀭𑀺𑀢𑁄 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀤𑁂𑀯𑁂 𑀅𑀦𑁆𑀢𑁄𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀬𑀁 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑁂 𑀏𑀢𑁆𑀣𑀸𑀯𑀘𑀭𑀸 𑀏𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀔𑀦𑁆𑀥𑀥𑀸𑀢𑀼𑀆𑀬𑀢𑀦𑀸; 𑀭𑀽𑀧𑀁, 𑀯𑁂𑀤𑀦𑀸, 𑀲𑀜𑁆𑀜𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇

3140\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸? 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸𑁇

3141\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸? 𑀳𑁂𑀝𑁆𑀞𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀉𑀧𑀭𑀺𑀢𑁄 𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑁂 𑀤𑁂𑀯𑁂 𑀅𑀦𑁆𑀢𑁄𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀬𑀁 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑁂 𑀏𑀢𑁆𑀣𑀸𑀯𑀘𑀭𑀸 𑀏𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇

3142\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇

3143\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸? 𑀳𑁂𑀝𑁆𑀞𑀢𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑁂 𑀤𑁂𑀯𑁂 𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀉𑀧𑀭𑀺𑀢𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑁂 𑀤𑁂𑀯𑁂 𑀅𑀦𑁆𑀢𑁄𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀬𑀁 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑁂 𑀏𑀢𑁆𑀣𑀸𑀯𑀘𑀭𑀸 𑀏𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀲𑁆𑀲 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇

3144\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸? 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑁇

3145\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸? 𑀲𑀸𑀲𑀯𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸, 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇

3146\. 𑀓𑀢𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸? 𑀫𑀕𑁆𑀕𑀸 𑀘, 𑀫𑀕𑁆𑀕𑀨𑀮𑀸𑀦𑀺 𑀘, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸𑁇
