# 𑁧𑁪𑁇𑁨𑁇𑁧 𑀢𑀺𑀓

### 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀜𑁆𑀳𑀸𑀧𑀼𑀘𑁆𑀙𑀓

### 𑀢𑀺𑀓

(716.)

2088\. 𑀓𑀼𑀲𑀮𑀸𑀬𑁂𑀯𑁇 (𑁧)

2089\. 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁨)

2090\. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸𑁇 (𑁩)

2091\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑁇 (𑁪)

2092\. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑁇 (𑁫)

2093\. 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸𑁇 (𑁬)

2094\. 𑀲𑀺𑀬𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸𑁇 (𑁭)

2095\. 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

2096\. 𑀦𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀦 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸𑁇 (𑁯)

2097\. 𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 (𑁧𑁦)

2098\. 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀸𑁇 (𑁧𑁧)

2099\. 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑁇 (𑁧𑁨)

2100\. 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁧𑁩)

2101\. 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑁇 (𑁧𑁪)

2102\. 𑀅𑀦𑀺𑀬𑀢𑀸𑁇 (𑁧𑁫)

2103\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀢𑀺𑀧𑀺, 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸𑀢𑀺𑀧𑀺, 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄𑀢𑀺𑀧𑀺𑁇 (𑁧𑁬)

2104\. 𑀲𑀺𑀬𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀦𑁄𑀢𑀺𑁇 (𑁧𑁭)

2105\. 𑀲𑀺𑀬𑀸 𑀅𑀢𑀻𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀦𑀸𑀕𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑁇 (𑁧𑁮)

2106\. 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁧𑁯)

2107\. 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸, 𑀲𑀺𑀬𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑁇 (𑁨𑁦)

2108\. 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑁇 (𑁨𑁧)

2109\. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀸𑁇 (𑁨𑁨)
