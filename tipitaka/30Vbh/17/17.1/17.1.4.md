* 𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀸𑀢𑀺𑀓𑀸

(835.)

2614\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀆𑀲𑀯𑀸 (𑁧) 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀕𑀦𑁆𑀣𑀸 (𑁨) 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀑𑀖𑀸 (𑁩) 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀬𑁄𑀕𑀸 (𑁪) 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀸𑀦𑀺 (𑁫) 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀢𑀡𑁆𑀳𑀼𑀧𑁆𑀧𑀸𑀤𑀸 (𑁬) 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀕𑀢𑀺𑀕𑀫𑀦𑀸𑀦𑀺 (𑁭) 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀸 (𑁮) 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀸 (𑁯) 𑀅𑀧𑀭𑁂𑀧𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀸 (𑁧𑁦) 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺 (𑁧𑁧) 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺 (𑁧𑁨) 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀪𑀬𑀸𑀦𑀺 (𑁧𑁩) 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀪𑀬𑀸𑀦𑀺 (𑁧𑁪) 𑀘𑀢𑀲𑁆𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑁄 (𑁧𑁫)𑁇
