# 𑁧𑁭𑁇𑁧𑁇𑁩 𑀢𑀺𑀓𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀔𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀢𑀺𑀓𑀫𑀸𑀢𑀺𑀓𑀸

(834.)

2612\. 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺 (𑁧) 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸 (𑁨) 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀲𑀜𑁆𑀜𑀸 (𑁩) 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀥𑀸𑀢𑀼𑀬𑁄 (𑁪) 𑀢𑀻𑀡𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺 (𑁫) 𑀢𑀬𑁄 𑀆𑀲𑀯𑀸 (𑁬) 𑀢𑀻𑀡𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 (𑁭) 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸 (𑁮) 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸 (𑁯) 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸 (𑁧𑁦) 𑀢𑀺𑀲𑁆𑀲𑁄 𑀏𑀲𑀦𑀸 (𑁧𑁧) 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀥𑀸 (𑁧𑁨) 𑀢𑀻𑀡𑀺 𑀪𑀬𑀸𑀦𑀺 (𑁧𑁩) 𑀢𑀻𑀡𑀺 𑀢𑀫𑀸𑀦𑀺 (𑁧𑁪) 𑀢𑀻𑀡𑀺 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀸𑀦𑀺 (𑁧𑁫) 𑀢𑀬𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀸 (𑁧𑁬) 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀡𑀸𑀦𑀺 (𑁧𑁭) 𑀢𑀻𑀡𑀺 𑀫𑀮𑀸𑀦𑀺 (𑁧𑁮) 𑀢𑀻𑀡𑀺 𑀯𑀺𑀲𑀫𑀸𑀦𑀺 (𑁧𑁯) 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀢𑀻𑀡𑀺 𑀯𑀺𑀲𑀫𑀸𑀦𑀺 (𑁨𑁦) 𑀢𑀬𑁄 𑀅𑀕𑁆𑀕𑀻 (𑁨𑁧) 𑀢𑀬𑁄 𑀓𑀲𑀸𑀯𑀸 (𑁨𑁨) 𑀅𑀧𑀭𑁂𑀧𑀺 𑀢𑀬𑁄 𑀓𑀲𑀸𑀯𑀸 (𑁨𑁩) 𑀅𑀲𑁆𑀲𑀸𑀤𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 (𑁨𑁪) 𑀅𑀭𑀢𑀺, 𑀯𑀺𑀳𑁂𑀲𑀸, 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸 (𑁨𑁫) 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸, 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸, 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸 (𑁨𑁬) 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀓𑁄𑀲𑀚𑁆𑀚𑀁, 𑀧𑀫𑀸𑀤𑁄 (𑁨𑁭) 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸, 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀢𑀸, 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸 (𑁨𑁮) 𑀅𑀳𑀺𑀭𑀺𑀓𑀁, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁, 𑀧𑀫𑀸𑀤𑁄 (𑁨𑁯) 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁, 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸, 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸 (𑁩𑁦) 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀁, 𑀅𑀯𑀤𑀜𑁆𑀜𑀼𑀢𑀸, 𑀓𑁄𑀲𑀚𑁆𑀚𑀁 (𑁩𑁧) 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀅𑀲𑀁𑀯𑀭𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁 (𑁩𑁨) 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀓𑀫𑁆𑀬𑀢𑀸, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀅𑀲𑁄𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸, 𑀉𑀧𑀸𑀭𑀫𑁆𑀪𑀘𑀺𑀢𑁆𑀢𑀢𑀸 (𑁩𑁩) 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀘𑁆𑀘𑀁, 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁, 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 (𑁩𑁪) 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄, 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀲𑁂𑀯𑀦𑀸, 𑀘𑁂𑀢𑀲𑁄 𑀘 𑀮𑀻𑀦𑀢𑁆𑀢𑀁 (𑁩𑁫)𑁇
