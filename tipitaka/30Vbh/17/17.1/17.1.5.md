### 𑀧𑀜𑁆𑀘𑀓𑀫𑀸𑀢𑀺𑀓𑀸

(836.)

2616\. 𑀧𑀜𑁆𑀘𑁄𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 (𑁧) 𑀧𑀜𑁆𑀘𑀼𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 (𑁨) 𑀧𑀜𑁆𑀘 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀸𑀦𑀺 (𑁩) 𑀧𑀜𑁆𑀘 𑀲𑀗𑁆𑀕𑀸 (𑁪) 𑀧𑀜𑁆𑀘 𑀲𑀮𑁆𑀮𑀸 (𑁫) 𑀧𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀔𑀺𑀮𑀸 (𑁬) 𑀧𑀜𑁆𑀘 𑀘𑁂𑀢𑀲𑁄𑀯𑀺𑀦𑀺𑀩𑀦𑁆𑀥𑀸 (𑁭) 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀺 (𑁮) 𑀧𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀆𑀦𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀦𑀺 (𑁯) 𑀧𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑁄 (𑁧𑁦) 𑀧𑀜𑁆𑀘 𑀯𑁂𑀭𑀸 (𑁧𑁧) 𑀧𑀜𑁆𑀘 𑀩𑁆𑀬𑀲𑀦𑀸 (𑁧𑁨) 𑀧𑀜𑁆𑀘 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀤𑀻𑀦𑀯𑀸 (𑁧𑁩) 𑀧𑀜𑁆𑀘 𑀪𑀬𑀸𑀦𑀺 (𑁧𑁪) 𑀧𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀯𑀸𑀤𑀸 (𑁧𑁫)𑁇
