# 𑁧𑁭𑁇𑁨𑁇𑁨 𑀤𑀼𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀔𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀤𑀼𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(891.)

2679\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑁄𑀥𑁄? 𑀬𑁄 𑀓𑁄𑀥𑁄 𑀓𑀼𑀚𑁆𑀛𑀦𑀸 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀢𑁆𑀢𑀁 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑁄𑀥𑁄”𑁇

2680\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑀦𑀸𑀳𑁄? 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀮𑀁 𑀓𑁄𑀥𑁄, 𑀅𑀧𑀭𑀓𑀸𑀮𑀁 𑀉𑀧𑀦𑀸𑀳𑁄𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀉𑀧𑀦𑀸𑀳𑁄 𑀉𑀧𑀦𑀬𑁆𑀳𑀦𑀸 𑀉𑀧𑀦𑀬𑁆𑀳𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀧𑀦𑀸 𑀞𑀧𑀦𑀸 𑀲𑀡𑁆𑀞𑀧𑀦𑀸 𑀅𑀦𑀼𑀲𑀁𑀲𑀦𑁆𑀤𑀦𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀩𑀦𑁆𑀥𑀦𑀸 𑀤𑀍𑀳𑀻𑀓𑀫𑁆𑀫𑀁 𑀓𑁄𑀥𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑀦𑀸𑀳𑁄”𑁇 (𑁧)

(892.)

2681\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑀓𑁆𑀔𑁄? 𑀬𑁄 𑀫𑀓𑁆𑀔𑁄 𑀫𑀓𑁆𑀔𑀸𑀬𑀦𑀸 𑀫𑀓𑁆𑀔𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀼𑀭𑀺𑀬𑀁 𑀦𑀺𑀝𑁆𑀞𑀼𑀭𑀺𑀬𑀓𑀫𑁆𑀫𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀓𑁆𑀔𑁄”𑁇

2682\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀍𑀆𑀲𑁄? 𑀬𑁄 𑀧𑀍𑀆𑀲𑁄 𑀧𑀍𑀆𑀲𑀸𑀬𑀦𑀸 𑀧𑀍𑀆𑀲𑀸𑀳𑀸𑀭𑁄 𑀯𑀺𑀯𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀬𑀼𑀕𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀍𑀆𑀲𑁄”𑁇 (𑁨)

(893.)

2683\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀇𑀲𑁆𑀲𑀸? 𑀬𑀸 𑀧𑀭𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀕𑀭𑀼𑀓𑀸𑀭𑀫𑀸𑀦𑀦𑀯𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀸𑀲𑀼 𑀇𑀲𑁆𑀲𑀸 𑀇𑀲𑁆𑀲𑀸𑀬𑀦𑀸 𑀇𑀲𑁆𑀲𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀲𑀽𑀬𑀸 𑀉𑀲𑀽𑀬𑀦𑀸 𑀉𑀲𑀽𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀇𑀲𑁆𑀲𑀸”𑁇

2684\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁? 𑀧𑀜𑁆𑀘 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀸𑀦𑀺—  𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀫𑀘𑁆𑀙𑁂𑀭𑀁 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀦𑀸 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑁂𑀯𑀺𑀘𑁆𑀙𑀁 𑀓𑀤𑀭𑀺𑀬𑀁 𑀓𑀝𑀼𑀓𑀜𑁆𑀘𑀼𑀓𑀢𑀸 𑀅𑀕𑁆𑀕𑀳𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁”𑁇 (𑁩)

(894.)

2685\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀸𑀬𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑁂𑀦 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸, 𑀯𑀸𑀘𑀸𑀬 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸, 𑀫𑀦𑀲𑀸 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀳𑁂𑀢𑀼𑀁 𑀧𑀸𑀧𑀺𑀓𑀁 𑀇𑀘𑁆𑀙𑀁 𑀧𑀡𑀺𑀤𑀳𑀢𑀺𑁇 “𑀫𑀸 𑀫𑀁 𑀚𑀜𑁆𑀜𑀸”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺𑁇 “𑀫𑀸 𑀫𑀁 𑀚𑀜𑁆𑀜𑀸”𑀢𑀺 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 “𑀫𑀸 𑀫𑀁 𑀚𑀜𑁆𑀜𑀸”𑀢𑀺 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 “𑀫𑀸 𑀫𑀁 𑀚𑀜𑁆𑀜𑀸”𑀢𑀺 𑀓𑀸𑀬𑁂𑀦 𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀫𑀸𑀬𑀸 𑀫𑀸𑀬𑀸𑀯𑀺𑀢𑀸 𑀅𑀘𑁆𑀘𑀸𑀲𑀭𑀸 𑀯𑀜𑁆𑀘𑀦𑀸 𑀦𑀺𑀓𑀢𑀺 𑀯𑀺𑀓𑀺𑀭𑀡𑀸 𑀧𑀭𑀺𑀳𑀭𑀡𑀸 𑀕𑀽𑀳𑀦𑀸 𑀧𑀭𑀺𑀕𑀽𑀳𑀦𑀸 𑀙𑀸𑀤𑀦𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀁 𑀯𑁄𑀘𑁆𑀙𑀸𑀤𑀦𑀸 𑀧𑀸𑀧𑀓𑀺𑀭𑀺𑀬𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀸𑀬𑀸”𑁇

2686\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀞𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀲𑀞𑁄𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀞𑀁 𑀲𑀞𑀢𑀸 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁 𑀓𑀓𑁆𑀓𑀭𑀢𑀸 𑀓𑀓𑁆𑀓𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀢𑀸 𑀧𑀸𑀭𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀺𑀬𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁”𑁇 (𑁪)

(895.)

2687\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸”𑁇

2688\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸? 𑀬𑁄 𑀪𑀯𑁂𑀲𑀼 𑀪𑀯𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀪𑀯𑀭𑀸𑀕𑁄 𑀪𑀯𑀦𑀦𑁆𑀤𑀻 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀪𑀯𑀲𑀺𑀦𑁂𑀳𑁄 𑀪𑀯𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀪𑀯𑀫𑀼𑀘𑁆𑀙𑀸 𑀪𑀯𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀢𑀡𑁆𑀳𑀸”𑁇 (𑁫)

(896.)

2689\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑀤𑀺𑀝𑁆𑀞𑀺? “𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

2690\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀪𑀯𑀤𑀺𑀝𑁆𑀞𑀺? “𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀪𑀯𑀤𑀺𑀝𑁆𑀞𑀺”𑁇 (𑁬)

(897.)

2691\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺? “𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

2692\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺? “𑀉𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺”𑁇 (𑁭)

(898.)

2693\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀦𑁆𑀢𑀯𑀸𑀤𑀺𑀝𑁆𑀞𑀺? “𑀅𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑁆𑀢𑀯𑀸𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

2694\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀦𑀦𑁆𑀢𑀯𑀸𑀤𑀺𑀝𑁆𑀞𑀺? “𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘𑀸”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀦𑁆𑀢𑀯𑀸𑀤𑀺𑀝𑁆𑀞𑀺”𑁇 (𑁮)

(899.)

2695\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺? 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

2696\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑀭𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺? 𑀅𑀧𑀭𑀦𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑀭𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺”𑁇 (𑁯)

(900.)

2697\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁? 𑀬𑀁 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀭𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑁂𑀦, 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀳𑀺𑀭𑀺𑀓𑀁”𑁇

2698\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁? 𑀬𑀁 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑁂𑀦 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁”𑁇 (𑁧𑁦)

(901.)

2699\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸? 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁂 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀸𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀺𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀮𑀕𑁆𑀕𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀲𑀸𑀢𑀢𑀸 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀅𑀦𑀸𑀤𑀭𑀢𑀸 𑀅𑀕𑀸𑀭𑀯𑀢𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀯𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸”𑁇

2700\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸? 𑀬𑁂 𑀢𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀸 𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸, 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀲𑁂𑀯𑀦𑀸 𑀦𑀺𑀲𑁂𑀯𑀦𑀸 𑀲𑀁𑀲𑁂𑀯𑀦𑀸 𑀪𑀚𑀦𑀸 𑀲𑀫𑁆𑀪𑀚𑀦𑀸 𑀪𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸”𑁇 (𑁧𑁧)

(902.)

2701\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀦𑀚𑁆𑀚𑀯𑁄? 𑀬𑁄 𑀅𑀦𑀚𑁆𑀚𑀯𑁄 𑀅𑀦𑀚𑁆𑀚𑀯𑀢𑀸 𑀚𑀺𑀫𑁆𑀳𑀢𑀸 𑀯𑀗𑁆𑀓𑀢𑀸 𑀓𑀼𑀝𑀺𑀮𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀚𑁆𑀚𑀯𑁄”𑁇

2702\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀫𑀤𑁆𑀤𑀯𑁄? 𑀬𑀸 𑀅𑀫𑀼𑀤𑀼𑀢𑀸 𑀅𑀫𑀤𑁆𑀤𑀯𑀢𑀸 𑀓𑀓𑁆𑀔𑀍𑀇𑀬𑀁 𑀨𑀸𑀭𑀼𑀲𑀺𑀬𑀁 𑀓𑀓𑁆𑀔𑀍𑀅𑀢𑀸 𑀓𑀞𑀺𑀦𑀢𑀸 𑀉𑀚𑀼𑀘𑀺𑀢𑁆𑀢𑀢𑀸 𑀅𑀫𑀼𑀤𑀼𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀫𑀤𑁆𑀤𑀯𑁄”𑁇 (𑁧𑁨)

(903.)

2703\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀬𑀸 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀅𑀓𑁆𑀔𑀫𑀦𑀢𑀸 𑀅𑀦𑀥𑀺𑀯𑀸𑀲𑀦𑀢𑀸 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺”𑁇

2704\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀲𑁄𑀭𑀘𑁆𑀘𑀁? 𑀓𑀸𑀬𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑁄𑀭𑀘𑁆𑀘𑀁”𑁇 𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁 𑀅𑀲𑁄𑀭𑀘𑁆𑀘𑀁𑁇 (𑁧𑁩)

(904.)

2705\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀲𑀸𑀔𑀮𑁆𑀬𑀁? 𑀬𑀸 𑀲𑀸 𑀯𑀸𑀘𑀸 𑀓𑀡𑁆𑀟𑀓𑀸 𑀓𑀓𑁆𑀓𑀲𑀸 𑀧𑀭𑀓𑀝𑀼𑀓𑀸 𑀧𑀭𑀸𑀪𑀺𑀲𑀚𑁆𑀚𑀦𑀻 𑀓𑁄𑀥𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀅𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀺𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀅𑀲𑀡𑁆𑀳𑀯𑀸𑀘𑀢𑀸 𑀅𑀲𑀔𑀺𑀮𑀯𑀸𑀘𑀢𑀸 𑀨𑀭𑀼𑀲𑀯𑀸𑀘𑀢𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀸𑀔𑀮𑁆𑀬𑀁”𑁇

2706\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄? 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀸—  𑀆𑀫𑀺𑀲𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄 𑀘 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄 𑀘𑁇 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀆𑀫𑀺𑀲𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄”𑁇 (𑁧𑁪)

(905.)

2707\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀦 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀦 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀦 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀦 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀸 𑀇𑀫𑁂𑀲𑀁 𑀙𑀦𑁆𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀕𑀼𑀢𑁆𑀢𑀺 𑀅𑀕𑁄𑀧𑀦𑀸 𑀅𑀦𑀸𑀭𑀓𑁆𑀔𑁄 𑀅𑀲𑀁𑀯𑀭𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸”𑁇

2708\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑁂𑀢𑀺 𑀤𑀯𑀸𑀬 𑀫𑀤𑀸𑀬 𑀫𑀡𑁆𑀟𑀦𑀸𑀬 𑀯𑀺𑀪𑀽𑀲𑀦𑀸𑀬𑁇 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀪𑁄𑀚𑀦𑁂—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸”𑁇 (𑁧𑁫)

(906.)

2709\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀘𑁆𑀘𑀁? 𑀬𑀸 𑀅𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑁆𑀲𑀭𑀡𑀢𑀸 𑀅𑀥𑀸𑀭𑀡𑀢𑀸 𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀘𑁆𑀘𑀁”𑁇

2710\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁”𑁇 (𑁧𑁬)

(907.)

2711\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺? 𑀬𑁄 𑀓𑀸𑀬𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺”𑁇 𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇

2712\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺? “𑀦𑀢𑁆𑀣𑀺 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀬𑀺𑀝𑁆𑀞𑀁…𑀧𑁂…  𑀬𑁂 𑀇𑀫𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀧𑀭𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺”𑁇 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇 (𑁧𑁭)

(908.)

2713\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑀁? 𑀧𑀜𑁆𑀘𑁄𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺—  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 𑀧𑀜𑁆𑀘𑀼𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺—  𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀁𑁇 (𑁧𑁮)
