# 𑁧𑁭𑁇𑁨𑁇𑁮 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀔𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(952.)

2891\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀓𑀺𑀮𑁂𑀲𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀮𑁄𑀪𑁄, 𑀤𑁄𑀲𑁄, 𑀫𑁄𑀳𑁄, 𑀫𑀸𑀦𑁄, 𑀤𑀺𑀝𑁆𑀞𑀺, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸, 𑀣𑀺𑀦𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀓𑀺𑀮𑁂𑀲𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 (𑁧)

(953.)

2892\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺?

2893\. 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

2894\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑀁𑁇 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

2895\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀫𑀕𑁆𑀕𑁄 𑀔𑁄 𑀫𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

2896\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀫𑀕𑁆𑀕𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

2897\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

2898\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄 𑀫𑀸𑀲𑀸𑀘𑀺𑀢𑀁 𑀫𑀜𑁆𑀜𑁂𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀙𑀝𑁆𑀞𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

2899\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀅𑀢𑁆𑀣𑀺 𑀓𑀧𑁆𑀧𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

2900\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 𑀇𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 (𑁨)

(954.)

2901\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂𑀲𑀼 𑀅𑀝𑁆𑀞𑀲𑀼 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀢𑁄? 𑀮𑀸𑀪𑁂 𑀲𑀸𑀭𑀸𑀕𑁄, 𑀅𑀮𑀸𑀪𑁂 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄, 𑀬𑀲𑁂 𑀲𑀸𑀭𑀸𑀕𑁄, 𑀅𑀬𑀲𑁂 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄, 𑀧𑀲𑀁𑀲𑀸𑀬 𑀲𑀸𑀭𑀸𑀕𑁄, 𑀦𑀺𑀦𑁆𑀤𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄, 𑀲𑀼𑀔𑁂 𑀲𑀸𑀭𑀸𑀕𑁄, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄—  𑀇𑀫𑁂𑀲𑀼 𑀅𑀝𑁆𑀞𑀲𑀼 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀢𑁄𑁇 (𑁩)

(955.)

2902\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀸? 𑀅𑀤𑀺𑀝𑁆𑀞𑁂 𑀤𑀺𑀝𑁆𑀞𑀯𑀸𑀤𑀺𑀢𑀸, 𑀅𑀲𑁆𑀲𑀼𑀢𑁂 𑀲𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀅𑀫𑀼𑀢𑁂 𑀫𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀤𑀺𑀝𑁆𑀞𑁂 𑀅𑀤𑀺𑀝𑁆𑀞𑀯𑀸𑀤𑀺𑀢𑀸, 𑀲𑀼𑀢𑁂 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀫𑀼𑀢𑁂 𑀅𑀫𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀯𑀸𑀤𑀺𑀢𑀸—  𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀸𑁇 (𑁪)

(956.)

2903\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸? 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸𑁇 (𑁫)

(957.)

2904\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑀸? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 “𑀦 𑀲𑀭𑀸𑀫𑀺 𑀦 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸𑀯 𑀦𑀺𑀩𑁆𑀩𑁂𑀞𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇

2905\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑀓𑀁𑀬𑁂𑀯 𑀧𑀝𑀺𑀧𑁆𑀨𑀭𑀢𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀼𑀬𑁆𑀳𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀡𑀺𑀢𑁂𑀦𑁇 𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀫𑀁 𑀪𑀡𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇

2906\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑀓𑀁𑀬𑁂𑀯 𑀧𑀘𑁆𑀘𑀸𑀭𑁄𑀧𑁂𑀢𑀺—  “𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀔𑁄𑀲𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀢𑁆𑀯𑀁 𑀢𑀸𑀯 𑀧𑀞𑀫𑀁 𑀧𑀝𑀺𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇

2907\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑁂𑀦𑀸𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀣𑀁 𑀅𑀧𑀦𑀸𑀫𑁂𑀢𑀺, 𑀓𑁄𑀧𑀜𑁆𑀘 𑀤𑁄𑀲𑀜𑁆𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀜𑁆𑀘 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇

2908\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀩𑀸𑀳𑀸𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀁 𑀪𑀡𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇

2909\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 “𑀦𑁂𑀯𑀸𑀳𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀺, 𑀦 𑀧𑀦𑀸𑀳𑀁 𑀅𑀦𑀸𑀧𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀻”𑀢𑀺 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑁄 𑀲𑀁𑀖𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇

2910\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁, 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀘𑁄𑀤𑀓𑀁, 𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄𑀯 𑀬𑁂𑀦𑀓𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇

2911\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀻𑀬𑀫𑀸𑀦𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀢𑀺𑀩𑀸𑀍𑀳𑀁 𑀫𑀬𑀺 𑀩𑁆𑀬𑀸𑀯𑀝𑀸𑁇 𑀇𑀤𑀸𑀦𑀸𑀳𑀁 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳—  “𑀇𑀤𑀸𑀦𑀺 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀣𑀸”𑀢𑀺𑁇 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑁄𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀤𑁄𑀲𑀸𑁇 (𑁬)

(958.)

2912\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸? “𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀅𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; 𑀅𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀭𑀽𑀧𑀻 𑀘 𑀅𑀭𑀽𑀧𑀻 𑀘…𑀧𑁂…  𑀦𑁂𑀯𑀭𑀽𑀧𑀻𑀦𑀸𑀭𑀽𑀧𑀻…𑀧𑁂…  “𑀅𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀅𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀅𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀅𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀅𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀦𑁂𑀯𑀦𑁆𑀢𑀯𑀸 𑀦𑀸𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀅𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸𑁇 (𑁭)

(959.)

2913\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸? “𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀅𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀭𑀽𑀧𑀻 𑀘 𑀅𑀭𑀽𑀧𑀻 𑀘 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀦𑁂𑀯𑀭𑀽𑀧𑀻𑀦𑀸𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀅𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀅𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺; “𑀦𑁂𑀯𑀦𑁆𑀢𑀯𑀸 𑀦𑀸𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺—  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸𑁇 (𑁮)
