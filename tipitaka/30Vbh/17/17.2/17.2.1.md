# 𑁧𑁭𑁇𑁨𑁇𑁧 𑀏𑀓𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀔𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀏𑀓𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(843.)

2630\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀚𑀸𑀢𑀺𑀫𑀤𑁄? 𑀚𑀸𑀢𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀤𑁄 𑀫𑀚𑁆𑀚𑀦𑀸 𑀫𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀸𑀢𑀺𑀫𑀤𑁄”𑁇 (𑁧)

(844.)

2631\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀕𑁄𑀢𑁆𑀢𑀫𑀤𑁄? 𑀕𑁄𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀆𑀭𑁄𑀕𑁆𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀬𑁄𑀩𑁆𑀩𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀚𑀻𑀯𑀺𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀮𑀸𑀪𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀲𑀓𑁆𑀓𑀸𑀭𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀕𑀭𑀼𑀓𑀸𑀭𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀭𑀺𑀯𑀸𑀭𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀪𑁄𑀕𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀯𑀡𑁆𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀲𑀼𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀝𑀺𑀪𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀭𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀅𑀦𑀯𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀇𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀬𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀲𑀻𑀮𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀛𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀲𑀺𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀆𑀭𑁄𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀭𑀺𑀡𑀸𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀤𑁄 𑀫𑀚𑁆𑀚𑀦𑀸 𑀫𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑁄 𑀘 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀫𑀤𑁄”𑁇 (𑁨--𑁨𑁭)

(845.)

2632\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑀤𑁄? 𑀬𑁄 𑀫𑀤𑁄 𑀫𑀚𑁆𑀚𑀦𑀸 𑀫𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀤𑁄”𑁇 (𑁨𑁮)

(846.)

2633\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀫𑀸𑀤𑁄? 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑀼 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀦𑀝𑁆𑀞𑀺𑀢𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀑𑀮𑀻𑀦𑀯𑀼𑀢𑁆𑀢𑀺𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀅𑀦𑀸𑀲𑁂𑀯𑀦𑀸 𑀅𑀪𑀸𑀯𑀦𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀧𑀫𑀸𑀤𑁄, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀧𑀫𑀸𑀤𑁄 𑀧𑀫𑀚𑁆𑀚𑀦𑀸 𑀧𑀫𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀫𑀸𑀤𑁄”𑁇 (𑁨𑁯)

(847.)

2634\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀣𑀫𑁆𑀪𑁄? 𑀬𑁄 𑀣𑀫𑁆𑀪𑁄 𑀣𑀫𑁆𑀪𑀦𑀸 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀓𑀓𑁆𑀔𑀍𑀇𑀬𑀁 𑀨𑀸𑀭𑀼𑀲𑀺𑀬𑀁 𑀉𑀚𑀼𑀘𑀺𑀢𑁆𑀢𑀢𑀸 𑀅𑀫𑀼𑀤𑀼𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀣𑀫𑁆𑀪𑁄”𑁇 (𑁩𑁦)

(848.)

2635\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀸𑀭𑀫𑁆𑀪𑁄? 𑀬𑁄 𑀲𑀸𑀭𑀫𑁆𑀪𑁄 𑀧𑀝𑀺𑀲𑀸𑀭𑀫𑁆𑀪𑁄 𑀲𑀸𑀭𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀲𑀸𑀭𑀫𑁆𑀪𑀦𑀸 𑀧𑀝𑀺𑀲𑀸𑀭𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀸𑀭𑀫𑁆𑀪𑁄”𑁇 (𑁩𑁧)

(849.)

2636\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀢𑁆𑀭𑀺𑀘𑁆𑀙𑀢𑀸? 𑀇𑀢𑀭𑀻𑀢𑀭𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀸, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀇𑀘𑁆𑀙𑀸 𑀇𑀘𑁆𑀙𑀸𑀕𑀢𑀸 𑀅𑀢𑁆𑀭𑀺𑀘𑁆𑀙𑀢𑀸 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀢𑁆𑀭𑀺𑀘𑁆𑀙𑀢𑀸”𑁇 (𑁩𑁨)

(850.)

2637\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸? 𑀇𑀢𑀭𑀻𑀢𑀭𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀸, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀇𑀘𑁆𑀙𑀸 𑀇𑀘𑁆𑀙𑀸𑀕𑀢𑀸 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸”𑁇 (𑁩𑁩)

(851.)

2638\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀲𑀤𑁆𑀥𑁄𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀲𑀻𑀮𑀯𑀸𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀓𑀼𑀲𑀻𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀻 𑀲𑀫𑀸𑀦𑁄 “𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀧𑀜𑁆𑀜𑀯𑀸𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀅𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀲𑀫𑀸𑀦𑁄 “𑀔𑀻𑀡𑀸𑀲𑀯𑁄𑀢𑀺 𑀫𑀁 𑀚𑀦𑁄 𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀇𑀘𑁆𑀙𑀸 𑀇𑀘𑁆𑀙𑀸𑀕𑀢𑀸 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀸 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀸”𑁇 (𑁩𑁪)

(852.)

2639\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀲𑀺𑀗𑁆𑀕𑀁? 𑀬𑀁 𑀲𑀺𑀗𑁆𑀕𑀁 𑀲𑀺𑀗𑁆𑀕𑀸𑀭𑀢𑀸 𑀘𑀸𑀢𑀼𑀭𑀢𑀸 𑀘𑀸𑀢𑀼𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀢𑀸 𑀧𑀸𑀭𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀺𑀬𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀺𑀗𑁆𑀕𑀁”𑁇 (𑁩𑁫)

(853.)

2640\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀢𑀺𑀦𑁆𑀢𑀺𑀡𑀁? 𑀬𑀁 𑀢𑀺𑀦𑁆𑀢𑀺𑀡𑀁 𑀢𑀺𑀦𑁆𑀢𑀺𑀡𑀸𑀬𑀦𑀸 𑀢𑀺𑀦𑁆𑀢𑀺𑀡𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀁 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀸𑀬𑀦𑀸 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀜𑁆𑀚𑀺𑀓𑀢𑀸 𑀲𑀸𑀥𑀼𑀓𑀫𑁆𑀬𑀢𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑀺𑀦𑁆𑀢𑀺𑀡𑀁”𑁇 (𑁩𑁬)

(854.)

2641\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑀸𑀧𑀮𑁆𑀬𑀁? 𑀘𑀻𑀯𑀭𑀫𑀡𑁆𑀟𑀦𑀸 𑀧𑀢𑁆𑀢𑀫𑀡𑁆𑀟𑀦𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀫𑀡𑁆𑀟𑀦𑀸 𑀇𑀫𑀲𑁆𑀲 𑀯𑀸 𑀧𑀽𑀢𑀺𑀓𑀸𑀬𑀲𑁆𑀲 𑀩𑀸𑀳𑀺𑀭𑀸𑀦𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀫𑀡𑁆𑀟𑀦𑀸 𑀯𑀺𑀪𑀽𑀲𑀦𑀸 𑀓𑁂𑀍𑀅𑀦𑀸 𑀧𑀭𑀺𑀓𑁂𑀍𑀅𑀦𑀸 𑀕𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀕𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑁆𑀢𑀁 𑀘𑀧𑀮𑀢𑀸 𑀘𑀸𑀧𑀮𑁆𑀬𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑀸𑀧𑀮𑁆𑀬𑀁”𑁇 (𑁩𑁭)

(855.)

2642\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀲𑀪𑀸𑀕𑀯𑀼𑀢𑁆𑀢𑀺? 𑀫𑀸𑀢𑀭𑀺 𑀯𑀸 𑀧𑀺𑀢𑀭𑀺 𑀯𑀸 𑀚𑁂𑀝𑁆𑀞𑁂 𑀯𑀸 𑀪𑀸𑀢𑀭𑀺 𑀯𑀸 𑀆𑀘𑀭𑀺𑀬𑁂𑀲𑀼 𑀯𑀸 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂 𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁂 𑀯𑀸 𑀲𑀸𑀯𑀓𑁂𑀲𑀼 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀲𑀼 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀕𑁆𑀕𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀲𑀸𑀢𑀢𑀸 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀢𑀸 𑀅𑀕𑀸𑀭𑀯𑀢𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀯𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀪𑀸𑀕𑀯𑀼𑀢𑁆𑀢𑀺”𑁇 (𑁩𑁮)

(856.)

2643\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀭𑀢𑀺? 𑀧𑀦𑁆𑀢𑁂𑀲𑀼 𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀲𑀼 𑀯𑀸 𑀅𑀥𑀺𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀭𑀢𑀺 𑀅𑀭𑀢𑀺𑀢𑀸 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀅𑀦𑀪𑀺𑀭𑀫𑀡𑀸 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀺𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀭𑀢𑀺”𑁇 (𑁩𑁯)

(857.)

2644\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑀦𑁆𑀤𑀻? 𑀬𑀸 𑀢𑀦𑁆𑀤𑀻 𑀢𑀦𑁆𑀤𑀺𑀬𑀦𑀸 𑀢𑀦𑁆𑀤𑀺𑀫𑀦𑀓𑀢𑀸 𑀆𑀮𑀲𑁆𑀬𑀁 𑀆𑀮𑀲𑁆𑀬𑀸𑀬𑀦𑀸 𑀆𑀮𑀲𑁆𑀬𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀢𑀦𑁆𑀤𑀻”𑁇 (𑁪𑁦)

(858.)

2645\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀚𑀫𑁆𑀪𑀺𑀢𑀸? 𑀬𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀚𑀫𑁆𑀪𑀦𑀸 𑀯𑀺𑀚𑀫𑁆𑀪𑀦𑀸 𑀆𑀦𑀫𑀦𑀸 𑀯𑀺𑀦𑀫𑀦𑀸 𑀲𑀦𑁆𑀦𑀫𑀦𑀸 𑀧𑀡𑀫𑀦𑀸 𑀩𑁆𑀬𑀸𑀥𑀺𑀬𑀓𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀚𑀫𑁆𑀪𑀺𑀢𑀸”𑁇 (𑁪𑁧)

(859.)

2646\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀪𑀢𑁆𑀢𑀲𑀫𑁆𑀫𑀤𑁄? 𑀬𑀸 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀫𑀼𑀘𑁆𑀙𑀸 𑀪𑀢𑁆𑀢𑀓𑀺𑀮𑀫𑀣𑁄 𑀪𑀢𑁆𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀓𑀸𑀬𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀢𑁆𑀢𑀲𑀫𑁆𑀫𑀤𑁄”𑁇 (𑁪𑁨)

(860.)

2647\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑁂𑀢𑀲𑁄 𑀘 𑀮𑀻𑀦𑀢𑁆𑀢𑀁? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀬𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀑𑀮𑀻𑀬𑀦𑀸 𑀲𑀮𑁆𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀦𑀁 𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀣𑀺𑀦𑀁 𑀣𑀺𑀬𑀦𑀸 𑀣𑀺𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑁂𑀢𑀲𑁄 𑀘 𑀮𑀻𑀦𑀢𑁆𑀢𑀁”𑁇 (𑁪𑁩)

(861.)

2648\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀼𑀳𑀦𑀸? 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀬𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀲𑀗𑁆𑀔𑀸𑀢𑁂𑀦 𑀯𑀸 𑀲𑀸𑀫𑀦𑁆𑀢𑀚𑀧𑁆𑀧𑀺𑀢𑁂𑀦 𑀯𑀸 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑁆𑀲 𑀯𑀸 𑀅𑀞𑀧𑀦𑀸 𑀞𑀧𑀦𑀸 𑀲𑀡𑁆𑀞𑀧𑀦𑀸 𑀪𑀸𑀓𑀼𑀝𑀺𑀢𑀸 𑀪𑀸𑀓𑀼𑀝𑀺𑀬𑀁 𑀓𑀼𑀳𑀦𑀸 𑀓𑀼𑀳𑀸𑀬𑀦𑀸 𑀓𑀼𑀳𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀼𑀳𑀦𑀸”𑁇 (𑁪𑁪)

(862.)

2649\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀮𑀧𑀦𑀸? 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑀲𑁆𑀲 𑀬𑀸 𑀧𑀭𑁂𑀲𑀁 𑀆𑀮𑀧𑀦𑀸 𑀮𑀧𑀦𑀸 𑀲𑀮𑁆𑀮𑀧𑀦𑀸 𑀉𑀮𑁆𑀮𑀧𑀦𑀸 𑀲𑀫𑀼𑀮𑁆𑀮𑀧𑀦𑀸 𑀉𑀦𑁆𑀦𑀳𑀦𑀸 𑀲𑀫𑀼𑀦𑁆𑀦𑀳𑀦𑀸 𑀉𑀓𑁆𑀓𑀸𑀘𑀦𑀸 𑀲𑀫𑀼𑀓𑁆𑀓𑀸𑀘𑀦𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀢𑀸 𑀘𑀸𑀝𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀫𑀼𑀕𑁆𑀕𑀲𑀽𑀧𑁆𑀬𑀢𑀸 𑀧𑀸𑀭𑀺𑀪𑀝𑀬𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀮𑀧𑀦𑀸”𑁇 (𑁪𑁫)

(863.)

2650\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀺𑀓𑀢𑀸? 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑀲𑁆𑀲 𑀬𑀁 𑀧𑀭𑁂𑀲𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀁 𑀑𑀪𑀸𑀲𑁄 𑀑𑀪𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀲𑀸𑀫𑀦𑁆𑀢𑀚𑀧𑁆𑀧𑀸 𑀧𑀭𑀺𑀓𑀣𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀺𑀓𑀢𑀸”𑁇 (𑁪𑁬)

(864.)

2651\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑀺𑀧𑁆𑀧𑁂𑀲𑀺𑀓𑀢𑀸? 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑀲𑁆𑀲 𑀬𑀸 𑀧𑀭𑁂𑀲𑀁 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀯𑀫𑁆𑀪𑀦𑀸 𑀕𑀭𑀳𑀡𑀸 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀸 𑀲𑀫𑀼𑀓𑁆𑀔𑁂𑀧𑀦𑀸 𑀔𑀺𑀧𑀦𑀸 𑀲𑀗𑁆𑀔𑀺𑀧𑀦𑀸 𑀧𑀸𑀧𑀦𑀸 𑀲𑀫𑁆𑀧𑀸𑀧𑀦𑀸 𑀅𑀯𑀡𑁆𑀡𑀳𑀸𑀭𑀺𑀓𑀸 𑀧𑀭𑀧𑀺𑀝𑁆𑀞𑀺𑀫𑀁𑀲𑀺𑀓𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀺𑀧𑁆𑀧𑁂𑀲𑀺𑀓𑀢𑀸”𑁇 (𑁪𑁭)

(865.)

2652\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀮𑀸𑀪𑁂𑀦 𑀮𑀸𑀪𑀁 𑀦𑀺𑀚𑀺𑀕𑀻𑀲𑀦𑀢𑀸? 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀇𑀢𑁄 𑀮𑀤𑁆𑀥𑀁 𑀆𑀫𑀺𑀲𑀁 𑀅𑀫𑀼𑀢𑁆𑀭 𑀳𑀭𑀢𑀺 𑀅𑀫𑀼𑀢𑁆𑀭 𑀯𑀸 𑀮𑀤𑁆𑀥𑀁 𑀆𑀫𑀺𑀲𑀁 𑀇𑀥 𑀆𑀳𑀭𑀢𑀺, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀆𑀫𑀺𑀲𑀲𑁆𑀲 𑀏𑀝𑁆𑀞𑀺 𑀕𑀯𑁂𑀝𑁆𑀞𑀺 𑀧𑀭𑀺𑀬𑁂𑀝𑁆𑀞𑀺 𑀏𑀲𑀦𑀸 𑀕𑀯𑁂𑀲𑀦𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀮𑀸𑀪𑁂𑀦 𑀮𑀸𑀪𑀁 𑀦𑀺𑀚𑀺𑀕𑀻𑀲𑀦𑀢𑀸”𑁇 (𑁪𑁮)

(866.)

2653\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁪𑁯)

(867.)

2654\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁦)

(868.)

2655\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀑𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀑𑀫𑀸𑀦𑁄 𑀑𑀫𑀜𑁆𑀜𑀦𑀸 𑀑𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀇𑀢𑀢𑁆𑀢𑀁 𑀅𑀢𑁆𑀢𑀼𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀯𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀧𑀭𑀺𑀪𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁧)

(869.)

2656\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁨)

(870.)

2657\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀲𑀤𑀺𑀲𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁩)

(871.)

2658\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀳𑀻𑀦𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀑𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀑𑀫𑀸𑀦𑁄 𑀑𑀫𑀜𑁆𑀜𑀦𑀸 𑀑𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀇𑀢𑀢𑁆𑀢𑀁 𑀅𑀢𑁆𑀢𑀼𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀯𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀧𑀭𑀺𑀪𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁪)

(872.)

2659\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀤𑀺𑀲𑀲𑁆𑀲 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀤𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑀺𑀲𑀲𑁆𑀲 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁫)

(873.)

2660\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀤𑀺𑀲𑀲𑁆𑀲 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀤𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀲𑀤𑀺𑀲𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑀺𑀲𑀲𑁆𑀲 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁬)

(874.)

2661\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀤𑀺𑀲𑀲𑁆𑀲 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀤𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀳𑀻𑀦𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀑𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀑𑀫𑀸𑀦𑁄 𑀑𑀫𑀜𑁆𑀜𑀦𑀸 𑀑𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀇𑀢𑀢𑁆𑀢𑀁 𑀅𑀢𑁆𑀢𑀼𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀯𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀧𑀭𑀺𑀪𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑀺𑀲𑀲𑁆𑀲 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁭)

(875.)

2662\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁮)

(876.)

2663\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀲𑀤𑀺𑀲𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀳𑀻𑀦𑀲𑁆𑀲 𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁫𑁯)

(877.)

2664\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀳𑀻𑀦𑀲𑁆𑀲 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀲𑀤𑀺𑀲𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺; 𑀲𑁄 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀑𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀑𑀫𑀸𑀦𑁄 𑀑𑀫𑀜𑁆𑀜𑀦𑀸 𑀑𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀇𑀢𑀢𑁆𑀢𑀁 𑀅𑀢𑁆𑀢𑀼𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀯𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀧𑀭𑀺𑀪𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀳𑀻𑀦𑀲𑁆𑀲 𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁦)

(878.)

2665\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑀸𑀦𑁄? 𑀬𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁧)

(879.)

2666\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀢𑀺𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀭𑁂𑀳𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀢𑀺𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁨)

(880.)

2667\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑀸𑀦𑀸𑀢𑀺𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀮𑀁 𑀧𑀭𑁂𑀳𑀺 𑀲𑀤𑀺𑀲𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀳𑀢𑀺, 𑀅𑀧𑀭𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀁 𑀤𑀳𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀸𑀦𑀸𑀢𑀺𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁩)

(881.)

2668\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀑𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀑𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀑𑀫𑀸𑀦𑁄 𑀑𑀫𑀜𑁆𑀜𑀦𑀸 𑀑𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀅𑀦𑀸 𑀑𑀳𑀻𑀍𑀇𑀢𑀢𑁆𑀢𑀁 𑀅𑀢𑁆𑀢𑀼𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀯𑀜𑁆𑀜𑀸 𑀅𑀢𑁆𑀢𑀧𑀭𑀺𑀪𑀯𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀑𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁪)

(882.)

2669\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀥𑀺𑀫𑀸𑀦𑁄? 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀧𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀅𑀓𑀢𑁂 𑀓𑀢𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑁂 𑀅𑀥𑀺𑀕𑀢𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁂 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀥𑀺𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁫)

(883.)

2670\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀲𑁆𑀫𑀺𑀫𑀸𑀦𑁄? 𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄, 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀙𑀦𑁆𑀤𑁄, 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀅𑀦𑀼𑀲𑀬𑁄, 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑁄, 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀙𑀦𑁆𑀤𑁄, 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀅𑀦𑀼𑀲𑀬𑁄, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑁆𑀫𑀺𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁬)

(884.)

2671\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀸𑀦𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀲𑀻𑀮𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀯𑀢𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀯𑀸 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑁂𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀫𑀸𑀦𑁄 𑀫𑀜𑁆𑀜𑀦𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀢𑀺 𑀉𑀦𑁆𑀦𑀸𑀫𑁄 𑀥𑀚𑁄 𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀸𑀳𑁄 𑀓𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀸𑀦𑁄”𑁇 (𑁬𑁭)

(885.)

2672\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀜𑀸𑀢𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀜𑀸𑀢𑀓𑁂 𑀆𑀭𑀩𑁆𑀪 𑀕𑁂𑀳𑀲𑀺𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀜𑀸𑀢𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 (𑁬𑁮)

(886.)

2673\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀚𑀦𑀧𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀚𑀦𑀧𑀤𑀁 𑀆𑀭𑀩𑁆𑀪 𑀕𑁂𑀳𑀲𑀺𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀦𑀧𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 (𑁬𑁯)

(887.)

2674\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀫𑀭𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀸 𑀕𑁂𑀳𑀲𑀺𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀫𑀭𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 (𑁭𑁦)

(888.)

2675\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀭𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀢𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀕𑀺𑀳𑀻𑀳𑀺 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀳𑀦𑀦𑁆𑀤𑀻 𑀲𑀳𑀲𑁄𑀓𑀻, 𑀲𑀼𑀔𑀺𑀢𑁂𑀲𑀼 𑀲𑀼𑀔𑀺𑀢𑁄, 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁂𑀲𑀼 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀓𑀺𑀘𑁆𑀘𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀸 𑀬𑁄𑀕𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁄 𑀢𑀢𑁆𑀣 𑀕𑁂𑀳𑀲𑀺𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀭𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀢𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 (𑁭𑁧)

(889.)

2676\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀆𑀭𑀩𑁆𑀪 𑀕𑁂𑀳𑀲𑀺𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 (𑁭𑁨)

(890.)

2677\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀦𑀯𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀸 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀮𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀦𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀸𑀬𑀢𑀦𑁂𑀦 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑀸 𑀫𑀸 𑀫𑀁 𑀧𑀭𑁂 𑀅𑀯𑀚𑀸𑀦𑀺𑀁𑀲𑀽𑀢𑀺𑁇 𑀬𑁄 𑀢𑀢𑁆𑀣 𑀕𑁂𑀳𑀲𑀺𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀯𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 (𑁭𑁩)
