# 𑁧𑁭𑁇𑁨𑁇𑁩 𑀢𑀺𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀔𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀢𑀺𑀓𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(909.)

2715\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺? 𑀮𑁄𑀪𑁄, 𑀤𑁄𑀲𑁄, 𑀫𑁄𑀳𑁄𑁇

2716\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀮𑁄𑀪𑁄? 𑀬𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀅𑀦𑀼𑀦𑀬𑁄 𑀅𑀦𑀼𑀭𑁄𑀥𑁄 𑀦𑀦𑁆𑀤𑀻 𑀦𑀦𑁆𑀤𑀺𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄 𑀇𑀘𑁆𑀙𑀸 𑀫𑀼𑀘𑁆𑀙𑀸 𑀅𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁 𑀕𑁂𑀥𑁄 𑀧𑀭𑀺𑀕𑁂𑀥𑁄 𑀲𑀗𑁆𑀕𑁄 𑀧𑀗𑁆𑀓𑁄 𑀏𑀚𑀸 𑀫𑀸𑀬𑀸 𑀚𑀦𑀺𑀓𑀸 𑀲𑀜𑁆𑀚𑀦𑀦𑀻 𑀲𑀺𑀩𑁆𑀩𑀺𑀦𑀻 𑀚𑀸𑀮𑀺𑀦𑀻 𑀲𑀭𑀺𑀢𑀸 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑁄𑀢𑀁 𑀯𑀺𑀲𑀝𑀸 𑀆𑀬𑀽𑀳𑀦𑀻 𑀤𑀼𑀢𑀺𑀬𑀸 𑀧𑀡𑀺𑀥𑀺 𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀯𑀦𑀁 𑀯𑀦𑀣𑁄 𑀲𑀦𑁆𑀣𑀯𑁄 𑀲𑀺𑀦𑁂𑀳𑁄 𑀅𑀧𑁂𑀓𑁆𑀔𑀸 𑀧𑀝𑀺𑀩𑀦𑁆𑀥𑀼 𑀆𑀲𑀸 𑀆𑀲𑀻𑀲𑀦𑀸 𑀆𑀲𑀻𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀸𑀲𑀸 𑀲𑀤𑁆𑀤𑀸𑀲𑀸 𑀕𑀦𑁆𑀥𑀸𑀲𑀸 𑀭𑀲𑀸𑀲𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀲𑀸 𑀮𑀸𑀪𑀸𑀲𑀸 𑀥𑀦𑀸𑀲𑀸 𑀧𑀼𑀢𑁆𑀢𑀸𑀲𑀸 𑀚𑀻𑀯𑀺𑀢𑀸𑀲𑀸 𑀚𑀧𑁆𑀧𑀸 𑀅𑀪𑀺𑀚𑀧𑁆𑀧𑀸 𑀚𑀧𑁆𑀧𑀦𑀸 𑀚𑀧𑁆𑀧𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀁 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀸𑀬𑀦𑀸 𑀮𑁄𑀮𑀼𑀧𑁆𑀧𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀜𑁆𑀚𑀺𑀓𑀢𑀸 𑀲𑀸𑀥𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀥𑀫𑁆𑀫𑀭𑀸𑀕𑁄 𑀯𑀺𑀲𑀫𑀮𑁄𑀪𑁄 𑀦𑀺𑀓𑀦𑁆𑀢𑀺 𑀦𑀺𑀓𑀸𑀫𑀦𑀸 𑀧𑀢𑁆𑀣𑀦𑀸 𑀧𑀺𑀳𑀦𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀣𑀦𑀸 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸 𑀦𑀺𑀭𑁄𑀥𑀢𑀡𑁆𑀳𑀸 𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸 𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸 𑀭𑀲𑀢𑀡𑁆𑀳𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸 𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸 𑀑𑀖𑁄 𑀬𑁄𑀕𑁄 𑀕𑀦𑁆𑀣𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀆𑀯𑀭𑀡𑀁 𑀦𑀻𑀯𑀭𑀡𑀁 𑀙𑀤𑀦𑀁 𑀩𑀦𑁆𑀥𑀦𑀁 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀅𑀦𑀼𑀲𑀬𑁄 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀮𑀢𑀸 𑀯𑁂𑀯𑀺𑀘𑁆𑀙𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀪𑀯𑁄 𑀫𑀸𑀭𑀧𑀸𑀲𑁄 𑀫𑀸𑀭𑀩𑀍𑀇𑀲𑀁 𑀫𑀸𑀭𑀯𑀺𑀲𑀬𑁄 𑀢𑀡𑁆𑀳𑀸𑀦𑀤𑀻 𑀢𑀡𑁆𑀳𑀸𑀚𑀸𑀮𑀁 𑀢𑀡𑁆𑀳𑀸𑀕𑀤𑁆𑀤𑀼𑀮𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀅𑀪𑀺𑀚𑁆𑀛𑀸 𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀮𑁄𑀪𑁄”𑁇

2717\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀤𑁄𑀲𑁄? “𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀅𑀘𑀭𑀻”𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, “𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀢𑀻”𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, “𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, “𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, “𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀅𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺, 𑀅𑀝𑁆𑀞𑀸𑀦𑁂 𑀯𑀸 𑀧𑀦 𑀆𑀖𑀸𑀢𑁄 𑀚𑀸𑀬𑀢𑀺𑁇 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀓𑁄𑀧𑁄 𑀧𑀓𑁄𑀧𑁄 𑀲𑀫𑁆𑀧𑀓𑁄𑀧𑁄 𑀤𑁄𑀲𑁄 𑀧𑀤𑁄𑀲𑁄 𑀲𑀫𑁆𑀧𑀤𑁄𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑁄 𑀓𑁄𑀥𑁄 𑀓𑀼𑀚𑁆𑀛𑀦𑀸 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀢𑁆𑀢𑀁 𑀤𑁄𑀲𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀦𑀸 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑁄𑀥𑁄 𑀧𑀝𑀺𑀯𑀺𑀭𑁄𑀥𑁄 𑀘𑀡𑁆𑀟𑀺𑀓𑁆𑀓𑀁 𑀅𑀲𑀼𑀭𑁄𑀧𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀲𑁄”𑁇

2718\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀫𑁄𑀳𑁄? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀜𑁆𑀜𑀸𑀡𑀁, 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑁄𑀳𑁄”𑁇 𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺𑁇 (𑁧)

(910.)

2719\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸? 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄𑁇

2720\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀓𑀸𑀫𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇

2721\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇

2722\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑁇 (𑁨)

(911.)

2723\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀲𑀜𑁆𑀜𑀸? 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸𑁇

2724\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸? 𑀓𑀸𑀫𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸”𑁇

2725\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸? 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸”𑁇

2726\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸? 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑀸 𑀲𑀜𑁆𑀚𑀸𑀦𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸”𑁇 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀲𑀜𑁆𑀜𑀸𑁇 (𑁩)

(912.)

2727\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀥𑀸𑀢𑀼𑀬𑁄? 𑀓𑀸𑀫𑀥𑀸𑀢𑀼, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼𑁇

2728\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼? 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑁇 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀥𑀸𑀢𑀼𑁇 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀥𑀸𑀢𑀼𑁇

2729\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀓𑀸𑀫𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇

2730\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇

2731\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄? 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑁄”𑁇 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀥𑀸𑀢𑀼𑀬𑁄𑁇 (𑁪)

(913.)

2732\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺? 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁𑁇

2733\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁”𑁇

2734\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁? 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄, 𑀧𑀺𑀲𑀼𑀡𑀸 𑀯𑀸𑀘𑀸, 𑀨𑀭𑀼𑀲𑀸 𑀯𑀸𑀘𑀸, 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁”𑁇

2735\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁? 𑀅𑀪𑀺𑀚𑁆𑀛𑀸, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁”𑁇

2736\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁? 𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁𑁇

2737\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁? 𑀅𑀓𑀼𑀲𑀮𑀸 𑀓𑀸𑀬𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁, 𑀅𑀓𑀼𑀲𑀮𑀸 𑀯𑀘𑀻𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁, 𑀅𑀓𑀼𑀲𑀮𑀸 𑀫𑀦𑁄𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺𑁇 (𑁫)

(914.)

2738\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀆𑀲𑀯𑀸? 𑀓𑀸𑀫𑀸𑀲𑀯𑁄, 𑀪𑀯𑀸𑀲𑀯𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄𑁇

2739\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀸𑀫𑀸𑀲𑀯𑁄? 𑀬𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑁄 𑀓𑀸𑀫𑀦𑀦𑁆𑀤𑀻 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀓𑀸𑀫𑀲𑀺𑀦𑁂𑀳𑁄 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀓𑀸𑀫𑀫𑀼𑀘𑁆𑀙𑀸 𑀓𑀸𑀫𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑀸𑀲𑀯𑁄”𑁇

2740\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀪𑀯𑀸𑀲𑀯𑁄? 𑀬𑁄 𑀪𑀯𑁂𑀲𑀼 𑀪𑀯𑀘𑁆𑀙𑀦𑁆𑀤𑁄…𑀧𑁂…  𑀪𑀯𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀸𑀲𑀯𑁄”𑁇

2741\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄? 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀜𑁆𑀜𑀸𑀡𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑁄”𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀆𑀲𑀯𑀸𑁇 (𑁬)

(915.)

2742\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺? 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄𑁇

2743\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺? 𑀇𑀥 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄—  𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

2744\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸? 𑀲𑀢𑁆𑀣𑀭𑀺 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀲𑀁𑀖𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀧𑀭𑀦𑁆𑀢𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺, 𑀇𑀤𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀢𑀸𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺𑁇 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀓𑀗𑁆𑀔𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀦𑀸 𑀓𑀗𑁆𑀔𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀣𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀦𑁄𑀯𑀺𑀮𑁂𑀔𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸”𑁇

2745\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄? “𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀲𑀻𑀮𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀺 𑀯𑀢𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀺 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑀻”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄”𑁇 𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑁇 (𑁭)

(916.)

2746\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸? 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸, 𑀪𑀯𑀢𑀡𑁆𑀳𑀸, 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑁇

2747\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸? 𑀪𑀯𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀳𑀕𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀢𑀡𑁆𑀳𑀸”𑁇

2748\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸? 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀳𑀕𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸”𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀢𑀡𑁆𑀳𑀸 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸𑁇

2749\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸? 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸”𑁇

2750\. ( ) 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑀢𑀡𑁆𑀳𑀸”𑁇

2751\. ( ) 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀳𑀕𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸”𑁇 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸𑁇 (𑁮)

(917.)

2752\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸? 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸, 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸, 𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸𑁇

2753\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸? 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸”𑁇

2754\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸? 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸”𑁇

2755\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸? 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸”𑁇 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸𑁇 (𑁯)

(918.)

2756\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸? 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸, 𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸, 𑀦𑀺𑀭𑁄𑀥𑀢𑀡𑁆𑀳𑀸𑁇

2757\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸? 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸”𑁇

2758\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸? 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸”𑁇

2759\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑀺𑀭𑁄𑀥𑀢𑀡𑁆𑀳𑀸? 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀳𑀕𑀢𑁄 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀺𑀭𑁄𑀥𑀢𑀡𑁆𑀳𑀸”𑁇 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸𑁇 (𑁧𑁦)

(919.)

2760\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀏𑀲𑀦𑀸? 𑀓𑀸𑀫𑁂𑀲𑀦𑀸, 𑀪𑀯𑁂𑀲𑀦𑀸, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑁇

2761\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑁂𑀲𑀦𑀸? 𑀬𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄…𑀧𑁂…  𑀓𑀸𑀫𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑁂𑀲𑀦𑀸”𑁇

2762\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑁂𑀲𑀦𑀸? 𑀬𑁄 𑀪𑀯𑁂𑀲𑀼 𑀪𑀯𑀘𑁆𑀙𑀦𑁆𑀤𑁄…𑀧𑁂…  𑀪𑀯𑀚𑁆𑀛𑁄𑀲𑀸𑀦𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑁂𑀲𑀦𑀸”𑁇

2763\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀲𑀦𑀸? “𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺 𑀯𑀸, “𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄”𑀢𑀺—  𑀯𑀸…𑀧𑁂…  “𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺 𑀯𑀸 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀲𑀦𑀸”𑁇

2764\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀸𑀫𑁂𑀲𑀦𑀸? 𑀓𑀸𑀫𑀭𑀸𑀕𑁄, 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀸𑀫𑁂𑀲𑀦𑀸”𑁇

2765\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀪𑀯𑁂𑀲𑀦𑀸? 𑀪𑀯𑀭𑀸𑀕𑁄 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀪𑀯𑁂𑀲𑀦𑀸”𑁇

2766\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀲𑀦𑀸? 𑀅𑀦𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀺𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀲𑀦𑀸”𑁇 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀏𑀲𑀦𑀸𑁇 (𑁧𑁧)

(920.)

2767\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀥𑀸? “𑀲𑁂𑀬𑁆𑀬𑁄𑀳𑀫𑀲𑁆𑀫𑀻”𑀢𑀺 𑀯𑀺𑀥𑀸, “𑀲𑀤𑀺𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀻”𑀢𑀺 𑀯𑀺𑀥𑀸, “𑀳𑀻𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀻”𑀢𑀺 𑀯𑀺𑀥𑀸—  𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀥𑀸𑁇 (𑁧𑁨)

(921.)

2768\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀪𑀬𑀸𑀦𑀺? 𑀚𑀸𑀢𑀺𑀪𑀬𑀁, 𑀚𑀭𑀸𑀪𑀬𑀁, 𑀫𑀭𑀡𑀪𑀬𑀁𑁇

2769\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀚𑀸𑀢𑀺𑀪𑀬𑀁? 𑀚𑀸𑀢𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀬𑀁 𑀪𑀬𑀸𑀦𑀓𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀘𑁂𑀢𑀲𑁄 𑀉𑀢𑁆𑀭𑀸𑀲𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀸𑀢𑀺𑀪𑀬𑀁”𑁇

2770\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀚𑀭𑀸𑀪𑀬𑀁? 𑀚𑀭𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀬𑀁 𑀪𑀬𑀸𑀦𑀓𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀘𑁂𑀢𑀲𑁄 𑀉𑀢𑁆𑀭𑀸𑀲𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀚𑀭𑀸𑀪𑀬𑀁”𑁇

2771\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀭𑀡𑀪𑀬𑀁? 𑀫𑀭𑀡𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀬𑀁 𑀪𑀬𑀸𑀦𑀓𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀘𑁂𑀢𑀲𑁄 𑀉𑀢𑁆𑀭𑀸𑀲𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀭𑀡𑀪𑀬𑀁”𑁇 𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀪𑀬𑀸𑀦𑀺𑁇 (𑁧𑁩)

(922.)

2772\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀢𑀫𑀸𑀦𑀺? 𑀅𑀢𑀻𑀢𑀁 𑀯𑀸 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀭𑀩𑁆𑀪 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀦 𑀲𑀫𑁆𑀧𑀲𑀻𑀤𑀢𑀺, 𑀅𑀦𑀸𑀕𑀢𑀁 𑀯𑀸 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀭𑀩𑁆𑀪 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀦 𑀲𑀫𑁆𑀧𑀲𑀻𑀤𑀢𑀺, 𑀏𑀢𑀭𑀳𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀭𑀩𑁆𑀪 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀦 𑀲𑀫𑁆𑀧𑀲𑀻𑀤𑀢𑀺—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀢𑀫𑀸𑀦𑀺𑁇 (𑁧𑁪)

(923.)

2773\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀸𑀦𑀺? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀻—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀸𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀳𑁂𑀢𑀽”𑀢𑀺; 𑀇𑀥 𑀧𑀦𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀻—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀸𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀇𑀲𑁆𑀲𑀭𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀳𑁂𑀢𑀽”𑀢𑀺; 𑀇𑀥 𑀧𑀦𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀻—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀸𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀅𑀳𑁂𑀢𑀼 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸”𑀢𑀺—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀸𑀦𑀺𑁇 (𑁧𑁫)

(924.)

2774\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀸? 𑀭𑀸𑀕𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁, 𑀤𑁄𑀲𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁, 𑀫𑁄𑀳𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀁—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀓𑀺𑀜𑁆𑀘𑀦𑀸𑁇 (𑁧𑁬)

2775\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀡𑀸𑀦𑀺? 𑀭𑀸𑀕𑁄 𑀅𑀗𑁆𑀕𑀡𑀁, 𑀤𑁄𑀲𑁄 𑀅𑀗𑁆𑀕𑀡𑀁, 𑀫𑁄𑀳𑁄 𑀅𑀗𑁆𑀕𑀡𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀡𑀸𑀦𑀺𑁇 (𑁧𑁭)

2776\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀫𑀮𑀸𑀦𑀺? 𑀭𑀸𑀕𑁄 𑀫𑀮𑀁, 𑀤𑁄𑀲𑁄 𑀫𑀮𑀁, 𑀫𑁄𑀳𑁄 𑀫𑀮𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀫𑀮𑀸𑀦𑀺𑁇 (𑁧𑁮)

2777\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀯𑀺𑀲𑀫𑀸𑀦𑀺? 𑀭𑀸𑀕𑁄 𑀯𑀺𑀲𑀫𑀁, 𑀤𑁄𑀲𑁄 𑀯𑀺𑀲𑀫𑀁, 𑀫𑁄𑀳𑁄 𑀯𑀺𑀲𑀫𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀯𑀺𑀲𑀫𑀸𑀦𑀺𑁇 (𑁧𑁯)

2778\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀢𑀻𑀡𑀺 𑀯𑀺𑀲𑀫𑀸𑀦𑀺? 𑀓𑀸𑀬𑀯𑀺𑀲𑀫𑀁, 𑀯𑀘𑀻𑀯𑀺𑀲𑀫𑀁, 𑀫𑀦𑁄𑀯𑀺𑀲𑀫𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀯𑀺𑀲𑀫𑀸𑀦𑀺𑁇 (𑁨𑁦)

2779\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀅𑀕𑁆𑀕𑀻? 𑀭𑀸𑀕𑀕𑁆𑀕𑀺, 𑀤𑁄𑀲𑀕𑁆𑀕𑀺, 𑀫𑁄𑀳𑀕𑁆𑀕𑀺—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀅𑀕𑁆𑀕𑀻𑁇 (𑁨𑁧)

2780\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀓𑀲𑀸𑀯𑀸? 𑀭𑀸𑀕𑀓𑀲𑀸𑀯𑁄, 𑀤𑁄𑀲𑀓𑀲𑀸𑀯𑁄, 𑀫𑁄𑀳𑀓𑀲𑀸𑀯𑁄—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀓𑀲𑀸𑀯𑀸𑁇 (𑁨𑁨)

2781\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀧𑀭𑁂𑀧𑀺 𑀢𑀬𑁄 𑀓𑀲𑀸𑀯𑀸? 𑀓𑀸𑀬𑀓𑀲𑀸𑀯𑁄, 𑀯𑀘𑀻𑀓𑀲𑀸𑀯𑁄, 𑀫𑀦𑁄𑀓𑀲𑀸𑀯𑁄—  𑀇𑀫𑁂 𑀢𑀬𑁄 𑀓𑀲𑀸𑀯𑀸𑁇 (𑁨𑁩)

(925.)

2782\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀲𑁆𑀲𑀸𑀤𑀤𑀺𑀝𑁆𑀞𑀺? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀻—  “𑀦𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑁂𑀲𑀼 𑀤𑁄𑀲𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀧𑀸𑀢𑀩𑁆𑀬𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑁆𑀲𑀸𑀤𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

2783\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺? 𑀇𑀥 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑀺𑀦𑀻𑀢𑁄—  𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀢𑀢𑁄 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀫𑀺𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺”𑁇

2784\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺? “𑀦𑀢𑁆𑀣𑀺 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀬𑀺𑀝𑁆𑀞𑀁…𑀧𑁂…  𑀬𑁂 𑀇𑀫𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀧𑀭𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀤𑀺𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺”𑁇 𑀲𑀲𑁆𑀲𑀢𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀲𑁆𑀲𑀸𑀤𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺, 𑀉𑀘𑁆𑀙𑁂𑀤𑀤𑀺𑀝𑁆𑀞𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 (𑁨𑁪)

(926.)

2785\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀭𑀢𑀺? 𑀧𑀦𑁆𑀢𑁂𑀲𑀼 𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀲𑀼 𑀯𑀸 𑀅𑀥𑀺𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀭𑀢𑀺 𑀅𑀭𑀢𑀺𑀢𑀸 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀅𑀦𑀪𑀺𑀭𑀫𑀡𑀸 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀺𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀭𑀢𑀺”𑁇

2786\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀯𑀺𑀳𑁂𑀲𑀸? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀸𑀡𑀺𑀦𑀸 𑀯𑀸 𑀮𑁂𑀟𑁆𑀟𑀼𑀦𑀸 𑀯𑀸 𑀤𑀡𑁆𑀟𑁂𑀦 𑀯𑀸 𑀲𑀢𑁆𑀣𑁂𑀦 𑀯𑀸 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀯𑀺𑀳𑁂𑀞𑁂𑀢𑀺, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀳𑁂𑀞𑀦𑀸 𑀯𑀺𑀳𑁂𑀞𑀦𑀸 𑀳𑀺𑀁𑀲𑀦𑀸 𑀯𑀺𑀳𑀺𑀁𑀲𑀦𑀸 𑀭𑁄𑀲𑀦𑀸 𑀯𑀺𑀭𑁄𑀲𑀦𑀸 𑀧𑀭𑀽𑀧𑀖𑀸𑀢𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀯𑀺𑀳𑁂𑀲𑀸”𑁇

2787\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸? 𑀓𑀸𑀬𑁂𑀦 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀯𑀺𑀲𑀫𑀘𑀭𑀺𑀬𑀸, 𑀯𑀸𑀘𑀸𑀬 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀯𑀺𑀲𑀫𑀘𑀭𑀺𑀬𑀸, 𑀫𑀦𑀲𑀸 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀯𑀺𑀲𑀫𑀘𑀭𑀺𑀬𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸”𑁇 (𑁨𑁫)

(927.)

2788\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸? 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁂 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀸𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀺𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀮𑀕𑁆𑀕𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀲𑀸𑀢𑀢𑀸 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀅𑀦𑀸𑀤𑀭𑀢𑀸 𑀅𑀕𑀸𑀭𑀯𑀢𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀯𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸”𑁇

2789\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸? 𑀬𑁂 𑀢𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀸 𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸, 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀲𑁂𑀯𑀦𑀸 𑀦𑀺𑀲𑁂𑀯𑀦𑀸 𑀲𑀁𑀲𑁂𑀯𑀦𑀸 𑀪𑀚𑀦𑀸 𑀲𑀫𑁆𑀪𑀚𑀦𑀸 𑀪𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀺 𑀢𑀁𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸”𑁇

2790\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸? 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀜𑁆𑀜𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸”𑁇 𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀜𑁆𑀜𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑁇 (𑁨𑁬)

(928.)

2791\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀯𑀽𑀧𑀲𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀪𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁”𑁇

2792\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑁄𑀲𑀚𑁆𑀚𑀁? 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑀼 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀦𑀝𑁆𑀞𑀺𑀢𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀑𑀮𑀻𑀦𑀯𑀼𑀢𑁆𑀢𑀺𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀅𑀦𑀸𑀲𑁂𑀯𑀦𑀸 𑀅𑀪𑀸𑀯𑀦𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀧𑀫𑀸𑀤𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑁄𑀲𑀚𑁆𑀚𑀁”𑁇

2793\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀫𑀸𑀤𑁄? 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑀼 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀦𑀝𑁆𑀞𑀺𑀢𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀑𑀮𑀻𑀦𑀯𑀼𑀢𑁆𑀢𑀺𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀅𑀦𑀸𑀲𑁂𑀯𑀦𑀸 𑀅𑀪𑀸𑀯𑀦𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀧𑀫𑀸𑀤𑁄, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀧𑀫𑀸𑀤𑁄 𑀧𑀫𑀚𑁆𑀚𑀦𑀸 𑀧𑀫𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀫𑀸𑀤𑁄”𑁇 (𑁨𑁭)

(929.)

2794\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸? 𑀇𑀢𑀭𑀻𑀢𑀭𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀸, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀇𑀘𑁆𑀙𑀸 𑀇𑀘𑁆𑀙𑀸𑀕𑀢𑀸 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸”𑁇

2795\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀢𑀸? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀢𑀸”𑁇

2796\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸? 𑀇𑀢𑀭𑀻𑀢𑀭𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀸, 𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀸 𑀇𑀘𑁆𑀙𑀸 𑀇𑀘𑁆𑀙𑀸𑀕𑀢𑀸 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸 𑀭𑀸𑀕𑁄 𑀲𑀸𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀭𑀸𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸”𑁇 (𑁨𑁮)

(930.)

2797\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑀁? 𑀬𑀁 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀳𑀺𑀭𑀻𑀬𑀺𑀢𑀩𑁆𑀩𑁂𑀦, 𑀦 𑀳𑀺𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀳𑀺𑀭𑀺𑀓𑀁”𑁇

2798\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁? 𑀬𑀁 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑁂𑀦, 𑀦 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁”𑁇

2799\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀧𑀫𑀸𑀤𑁄? 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑀼 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑁄 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀬 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀦𑀝𑁆𑀞𑀺𑀢𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀑𑀮𑀻𑀦𑀯𑀼𑀢𑁆𑀢𑀺𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀅𑀦𑀸𑀲𑁂𑀯𑀦𑀸 𑀅𑀪𑀸𑀯𑀦𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀧𑀫𑀸𑀤𑁄, 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁄 𑀧𑀫𑀸𑀤𑁄 𑀧𑀫𑀚𑁆𑀚𑀦𑀸 𑀧𑀫𑀚𑁆𑀚𑀺𑀢𑀢𑁆𑀢𑀁—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀫𑀸𑀤𑁄”𑁇 (𑁨𑁯)

(931.)

2800\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁? 𑀬𑀁 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀅𑀦𑀸𑀤𑀭𑀢𑀸 𑀅𑀕𑀸𑀭𑀯𑀢𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀯𑀢𑀸 𑀅𑀦𑀤𑁆𑀤𑀸 𑀅𑀦𑀤𑁆𑀤𑀸𑀬𑀦𑀸 𑀅𑀦𑀤𑁆𑀤𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀅𑀲𑀻𑀮𑁆𑀬𑀁 𑀅𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀸𑀭𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁”𑁇

2801\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸? 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁂 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀸𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀺𑀬𑀁 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀮𑀕𑁆𑀕𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀲𑀸𑀢𑀢𑀸 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀅𑀦𑀸𑀤𑀭𑀢𑀸 𑀅𑀕𑀸𑀭𑀯𑀢𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀯𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸”𑁇

2802\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸? 𑀬𑁂 𑀢𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀸 𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸, 𑀬𑀸 𑀢𑁂𑀲𑀁 𑀲𑁂𑀯𑀦𑀸 𑀦𑀺𑀲𑁂𑀯𑀦𑀸 𑀲𑀁𑀲𑁂𑀯𑀦𑀸 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀸 𑀪𑀚𑀦𑀸 𑀲𑀫𑁆𑀪𑀚𑀦𑀸 𑀪𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀺 𑀢𑀁𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸”𑁇 (𑁩𑁦)

(932.)

2803\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀁? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀸 𑀲𑀁𑀖𑀁 𑀯𑀸, 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀁 𑀅𑀲𑁆𑀲𑀤𑁆𑀤𑀳𑀦𑀸 𑀅𑀦𑁄𑀓𑀧𑁆𑀧𑀦𑀸 𑀅𑀦𑀪𑀺𑀧𑁆𑀧𑀲𑀸𑀤𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀁”𑁇

2804\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀯𑀤𑀜𑁆𑀜𑀼𑀢𑀸? 𑀧𑀜𑁆𑀘 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀸𑀦𑀺—  𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁𑁇 𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀫𑀘𑁆𑀙𑁂𑀭𑀁 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀦𑀸 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑁂𑀯𑀺𑀘𑁆𑀙𑀁 𑀓𑀤𑀭𑀺𑀬𑀁 𑀓𑀝𑀼𑀓𑀜𑁆𑀘𑀼𑀓𑀢𑀸 𑀅𑀕𑁆𑀕𑀳𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀯𑀤𑀜𑁆𑀜𑀼𑀢𑀸”𑁇

2805\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀓𑁄𑀲𑀚𑁆𑀚𑀁? 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂 𑀯𑀸 𑀧𑀜𑁆𑀘𑀲𑀼 𑀯𑀸 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑁄, 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀪𑀸𑀯𑀦𑀸𑀬 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀅𑀦𑀝𑁆𑀞𑀺𑀢𑀓𑀺𑀭𑀺𑀬𑀢𑀸 𑀑𑀮𑀻𑀦𑀯𑀼𑀢𑁆𑀢𑀺𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀙𑀦𑁆𑀤𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑀢𑀸 𑀅𑀦𑀸𑀲𑁂𑀯𑀦𑀸 𑀅𑀪𑀸𑀯𑀦𑀸 𑀅𑀩𑀳𑀼𑀮𑀻𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀧𑀫𑀸𑀤𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑁄𑀲𑀚𑁆𑀚𑀁”𑁇 (𑁩𑁧)

(933.)

2806\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀯𑀽𑀧𑀲𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀪𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁”𑁇

2807\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀲𑀁𑀯𑀭𑁄? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀦 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀦 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀨𑀼𑀲𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀕𑁆𑀕𑀸𑀳𑀻𑁇 𑀬𑀢𑁆𑀯𑀸𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀦𑀁 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀅𑀲𑀁𑀯𑀼𑀢𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀯𑀸𑀲𑁆𑀲𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀯𑀭𑀸𑀬 𑀦 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀦 𑀲𑀁𑀯𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀁𑀯𑀭𑁄”𑁇

2808\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁? 𑀓𑀸𑀬𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁄 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁆𑀬𑀁”𑁇 (𑁩𑁨)

(934.)

2809\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀓𑀫𑁆𑀬𑀢𑀸? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀅𑀭𑀺𑀬𑀸? 𑀅𑀭𑀺𑀬𑀸 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀘 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑀸 𑀘𑁇 𑀬𑀸 𑀇𑀫𑁂𑀲𑀁 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀤𑀝𑁆𑀞𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀲𑀫𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀲𑀫𑀸𑀕𑀦𑁆𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀓𑀫𑁆𑀬𑀢𑀸”𑁇

2810\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀅𑀲𑁄𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄”𑁇 𑀬𑀸 𑀇𑀫𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀲𑁄𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀲𑀯𑀦𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸 𑀅𑀥𑀸𑀭𑁂𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀅𑀲𑁄𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸”𑁇

2811\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀉𑀧𑀸𑀭𑀫𑁆𑀪𑀘𑀺𑀢𑁆𑀢𑀢𑀸? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀉𑀧𑀸𑀭𑀫𑁆𑀪𑁄? 𑀬𑁄 𑀉𑀧𑀸𑀭𑀫𑁆𑀪𑁄 𑀅𑀦𑀼𑀧𑀸𑀭𑀫𑁆𑀪𑁄 𑀉𑀧𑀸𑀭𑀫𑁆𑀪𑀦𑀸 𑀅𑀦𑀼𑀧𑀸𑀭𑀫𑁆𑀪𑀦𑀸 𑀅𑀦𑀼𑀧𑀸𑀭𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀉𑀜𑁆𑀜𑀸 𑀅𑀯𑀜𑁆𑀜𑀸 𑀧𑀭𑀺𑀪𑀯𑁄 𑀭𑀦𑁆𑀥𑀕𑀯𑁂𑀲𑀺𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀉𑀧𑀸𑀭𑀫𑁆𑀪𑀘𑀺𑀢𑁆𑀢𑀢𑀸”𑁇 (𑁩𑁩)

(935.)

2812\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀘𑁆𑀘𑀁? 𑀬𑀸 𑀅𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑁆𑀲𑀭𑀡𑀢𑀸 𑀅𑀥𑀸𑀭𑀡𑀢𑀸 𑀧𑀺𑀮𑀸𑀧𑀦𑀢𑀸 𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀦𑀢𑀸—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀘𑁆𑀘𑀁”𑁇

2813\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁? 𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀗𑁆𑀕𑀻 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁”𑁇

2814\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄? 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁 𑀅𑀯𑀽𑀧𑀲𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀪𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄”𑁇 (𑁩𑁪)

(936.)

2815\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄? 𑀅𑀦𑀺𑀘𑁆𑀘𑁂 “𑀦𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄, 𑀤𑀼𑀓𑁆𑀔𑁂 “𑀲𑀼𑀔𑀦𑁆”𑀢𑀺 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄, 𑀅𑀦𑀢𑁆𑀢𑀦𑀺 “𑀅𑀢𑁆𑀢𑀸”𑀢𑀺 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄, 𑀅𑀲𑀼𑀪𑁂 “𑀲𑀼𑀪𑀦𑁆”𑀢𑀺 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄, 𑀲𑀘𑁆𑀘𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀮𑁂𑀦 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀯𑀝𑁆𑀝𑀦𑀸 𑀅𑀦𑀸𑀯𑀝𑁆𑀝𑀦𑀸 𑀆𑀪𑁄𑀕𑁄 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄”𑁇

2816\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀲𑁂𑀯𑀦𑀸? 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁄 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄? 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁄”𑁇 𑀬𑀸 𑀇𑀫𑀲𑁆𑀲 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀲𑁂𑀯𑀦𑀸 𑀦𑀺𑀲𑁂𑀯𑀦𑀸 𑀲𑀁𑀲𑁂𑀯𑀦𑀸 𑀪𑀚𑀦𑀸 𑀲𑀫𑁆𑀪𑀚𑀦𑀸 𑀪𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀺 𑀢𑀁𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑀢𑀸—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀲𑁂𑀯𑀦𑀸”𑁇

2817\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀁 𑀘𑁂𑀢𑀲𑁄 𑀘 𑀮𑀻𑀦𑀢𑁆𑀢𑀁? 𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑀮𑁆𑀬𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀑𑀮𑀻𑀬𑀦𑀸 𑀲𑀮𑁆𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀦𑀁 𑀮𑀻𑀬𑀦𑀸 𑀮𑀻𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀣𑀺𑀦𑀁 𑀣𑀺𑀬𑀦𑀸 𑀣𑀺𑀬𑀺𑀢𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲—  𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀘𑁂𑀢𑀲𑁄 𑀘 𑀮𑀻𑀦𑀢𑁆𑀢𑀁”𑁇 (𑁩𑁫)
