# 𑁧𑁭𑁇𑁨𑁇𑁧𑁧 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀔𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(973.)

2930. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀸𑀬? 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀇𑀢𑁆𑀣𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀣𑀸𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀇𑀢𑁆𑀣𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀸𑀢𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀸𑀳𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀸𑀳𑀁 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀸𑀳𑀁 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀸𑀳𑀁 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇

(974.)

2931. 𑀓𑀣𑀜𑁆𑀘 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀫𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀅𑀲𑁆𑀫𑀻𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀫𑀸𑀦𑀺 𑀧𑀧𑀜𑁆𑀘𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀇𑀢𑁆𑀣𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸 𑀏𑀯𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸 𑀅𑀜𑁆𑀜𑀣𑀸𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸𑁇 (𑁧)

2932. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀢𑁆𑀣𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀔𑀢𑁆𑀢𑀺𑀬𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀯𑁂𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀲𑀼𑀤𑁆𑀤𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀕𑀳𑀝𑁆𑀞𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀤𑁂𑀯𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀅𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀅𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀢𑁆𑀣𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁨)

2933. 𑀓𑀣𑀜𑁆𑀘 𑀏𑀯𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀢𑀣𑀸𑀳𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀣𑀸𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀯𑁂𑀲𑁆𑀲𑁄 𑀢𑀣𑀸𑀳𑀁 𑀯𑁂𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸 𑀲𑁄 𑀲𑀼𑀤𑁆𑀤𑁄 𑀢𑀣𑀸𑀳𑀁 𑀲𑀼𑀤𑁆𑀤𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀕𑀳𑀝𑁆𑀞𑁄 𑀢𑀣𑀸𑀳𑀁 𑀕𑀳𑀝𑁆𑀞𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀢𑀣𑀸𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀤𑁂𑀯𑁄 𑀢𑀣𑀸𑀳𑀁 𑀤𑁂𑀯𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀢𑀣𑀸𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀭𑀽𑀧𑀻 𑀢𑀣𑀸𑀳𑀁 𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀅𑀭𑀽𑀧𑀻 𑀢𑀣𑀸𑀳𑀁 𑀅𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀲𑀜𑁆𑀜𑀻 𑀢𑀣𑀸𑀳𑀁 𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀅𑀲𑀜𑁆𑀜𑀻 𑀢𑀣𑀸𑀳𑀁 𑀅𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀢𑀣𑀸𑀳𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀏𑀯𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁩)

2934. 𑀓𑀣𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑀣𑀸𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀯𑁂𑀲𑁆𑀲𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀯𑁂𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀲𑀼𑀤𑁆𑀤𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀲𑀼𑀤𑁆𑀤𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀕𑀳𑀝𑁆𑀞𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀕𑀳𑀝𑁆𑀞𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀤𑁂𑀯𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀤𑁂𑀯𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀭𑀽𑀧𑀻 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀅𑀭𑀽𑀧𑀻 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀅𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀲𑀜𑁆𑀜𑀻 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀅𑀲𑀜𑁆𑀜𑀻 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀅𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀅𑀜𑁆𑀜𑀣𑀸𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁪)

2939. 𑀓𑀣𑀜𑁆𑀘 𑀅𑀲𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀺𑀘𑁆𑀘𑁄𑀲𑁆𑀫𑀺 𑀥𑀼𑀯𑁄𑀲𑁆𑀫𑀺 𑀲𑀲𑁆𑀲𑀢𑁄𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄𑀲𑁆𑀫𑀻𑀢𑀺—  𑀏𑀯𑀁 𑀅𑀲𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁯)

2940. 𑀓𑀣𑀜𑁆𑀘 𑀲𑀸𑀢𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺—  𑀏𑀯𑀁 𑀲𑀸𑀢𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁦)

2941. 𑀓𑀣𑀜𑁆𑀘 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀫𑀸𑀦𑀺 𑀧𑀧𑀜𑁆𑀘𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸𑁇 (𑁧𑁧)

2942. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀯𑁂𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀲𑀼𑀤𑁆𑀤𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀕𑀳𑀝𑁆𑀞𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀤𑁂𑀯𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀭𑀽𑀧𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀅𑀭𑀽𑀧𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀅𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁨)

2943. 𑀓𑀣𑀜𑁆𑀘 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀢𑀣𑀸𑀳𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀣𑀸𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀢𑀣𑀸𑀳𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁩)

2952. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀇𑀢𑁆𑀣𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀇𑀫𑀺𑀦𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀯𑁂𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀲𑀼𑀤𑁆𑀤𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀕𑀳𑀝𑁆𑀞𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀤𑁂𑀯𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀭𑀽𑀧𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀇𑀢𑁆𑀣𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁨)

2953. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁩)

2954. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀲𑁆𑀫𑀻𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁪)

2955. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…  𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀫𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀫𑀸𑀦𑀺 𑀧𑀧𑀜𑁆𑀘𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀇𑀢𑁆𑀣𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸𑁇 (𑁫)

2956. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀇𑀢𑁆𑀣𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀇𑀫𑀺𑀦𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀇𑀫𑀺𑀦𑀸 𑀅𑀭𑀽𑀧𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀲𑀜𑁆𑀜𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀲𑀜𑁆𑀜𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀇𑀢𑁆𑀣𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁬)

2957. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁭)

2958. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁮)

2959. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀲𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀫𑀺𑀦𑀸 𑀦𑀺𑀘𑁆𑀘𑁄𑀲𑁆𑀫𑀺 𑀥𑀼𑀯𑁄𑀲𑁆𑀫𑀺 𑀲𑀲𑁆𑀲𑀢𑁄𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄𑀲𑁆𑀫𑀻𑀢𑀺—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀲𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁯)

2960. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀲𑀸𑀢𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀫𑀺𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀲𑀸𑀢𑀲𑁆𑀫𑀻𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁦)

2961. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀁…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀫𑀺𑀦𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀫𑀸𑀦𑀺 𑀧𑀧𑀜𑁆𑀘𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸𑁇 (𑁧𑁧)

2963. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀢𑀣𑀸𑀳𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁩)

2964. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀦𑀸𑀳𑀁 𑀢𑀣𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁪)

2965. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀓𑀜𑁆𑀘𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀯𑀓𑀸𑀭𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀙𑀦𑁆𑀤𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀫𑀸𑀦𑀺 𑀧𑀧𑀜𑁆𑀘𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸𑁇 (𑁧𑁫)

2966. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀯𑁂𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀲𑀼𑀤𑁆𑀤𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀕𑀳𑀝𑁆𑀞𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀤𑁂𑀯𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀭𑀽𑀧𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀅𑀭𑀽𑀧𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀅𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀇𑀢𑁆𑀣𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁬)

2967. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀢𑀣𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀢𑀣𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀢𑀣𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀏𑀯𑀁 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁭)

2968. 𑀓𑀣𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺? 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀬𑀣𑀸 𑀲𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀦 𑀢𑀣𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀬𑀣𑀸 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀦 𑀢𑀣𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀬𑀣𑀸 𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀦 𑀢𑀣𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀯𑀸—  𑀏𑀯𑀁 𑀇𑀫𑀺𑀦𑀸 𑀅𑀧𑀸𑀳𑀁 𑀅𑀜𑁆𑀜𑀣𑀸 𑀲𑀺𑀬𑀦𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁧𑁮)

2969. 𑀇𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀩𑀸𑀳𑀺𑀭𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀸𑀬𑁇

2970. 𑀇𑀢𑀺 𑀇𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀇𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀩𑀸𑀳𑀺𑀭𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀇𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀅𑀢𑀻𑀢𑀸𑀦𑀺 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺, 𑀅𑀦𑀸𑀕𑀢𑀸𑀦𑀺 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀺 𑀙𑀢𑁆𑀢𑀺𑀁𑀲 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀸𑀦𑀺, 𑀢𑀤𑁂𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀜𑁆𑀜𑀽𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀘𑀭𑀺𑀢𑀲𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(977.)

2971. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀤𑁆𑀯𑀸𑀲𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀚𑀸𑀮𑁂 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑀸? 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀲𑁆𑀲𑀢𑀯𑀸𑀤𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀏𑀓𑀘𑁆𑀘𑀲𑀲𑁆𑀲𑀢𑀺𑀓𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀅𑀫𑀭𑀸𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑀺𑀓𑀸, 𑀤𑁆𑀯𑁂 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀺𑀓𑀸, 𑀲𑁄𑀍𑀅𑀲 𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸, 𑀅𑀝𑁆𑀞 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸, 𑀅𑀝𑁆𑀞 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸, 𑀲𑀢𑁆𑀢 𑀉𑀘𑁆𑀙𑁂𑀤𑀯𑀸𑀤𑀸, 𑀧𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀯𑀸𑀤𑀸—  𑀇𑀫𑀸𑀦𑀺 𑀤𑁆𑀯𑀸𑀲𑀝𑁆𑀞𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀚𑀸𑀮𑁂 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑀸𑀢𑀺𑁇
