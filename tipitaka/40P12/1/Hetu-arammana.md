# 𑀳𑁂𑀢𑀼-𑀆𑀭𑀫𑁆𑀫𑀡

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓𑀸𑀤𑀺

### 𑀳𑁂𑀢𑀼-𑀆𑀭𑀫𑁆𑀫𑀡

(3.)

7\. 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

8\. 𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

9\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺…𑀧𑁂…  𑀲𑀳𑀚𑀸𑀢𑁂 𑀢𑀻𑀡𑀺, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀏𑀓𑀁, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀆𑀲𑁂𑀯𑀦𑁂 𑀤𑁆𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀛𑀸𑀦𑁂 𑀏𑀓𑀁, 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀏𑀓𑀁, 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑁇

(4.)

10\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

11\. 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

12\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

13\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(5.)

14\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

15\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

(6.)

16\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

17\. 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

18\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

19\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁄𑁇)

(7.)

20\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀬𑀸𑀯 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀏𑀓𑀁𑁇)

21\. 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

22\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

23\. 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

24\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

25\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)
