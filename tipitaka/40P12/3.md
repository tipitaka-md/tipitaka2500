# 𑁩 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

(9.)

38\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

39\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀜𑁆𑀘𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

40\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀲𑀸𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀦𑀸𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

41\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

42\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

43\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁄𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀦𑀸𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

44\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀦𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀜𑁆𑀘𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

45\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

46\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀜𑁆𑀘𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

47\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀲𑀸𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀼𑀁 𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀅𑀦𑀸𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)
