# 𑁧𑁨 𑀉𑀧𑀸𑀤𑀸𑀦𑀤𑀼𑀓𑀸𑀤𑀺, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀤𑀼𑀓𑀸𑀤𑀺, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

(35.)

123\. 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

124\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇

(36.)

125\. 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

126\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇
