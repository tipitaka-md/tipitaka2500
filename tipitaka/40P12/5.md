# 𑁫 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓𑀸𑀤𑀺

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓𑀸𑀤𑀺

(11.)

52\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

53\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

54\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

55\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

56\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

57\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

58\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

59\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

60\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀩𑀸𑀳𑀺𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

61\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁅𑀧𑀸𑀤𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

62\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁅𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

63\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

64\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁄𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)
