# 𑁨𑁇𑁧𑁇𑁪𑁇𑁪 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀢𑀻𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

(354.)

3106\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

3107\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

3108\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

3109\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

3110\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

3111\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

3112\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

3113\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

3114\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

3115\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

3116\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

3117\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺…𑀧𑁂… 𑁇

(355.)

3118\. »  𑀬𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

3119\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇
