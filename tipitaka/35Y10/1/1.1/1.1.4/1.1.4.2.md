496\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁?

497\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

498\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸?

499\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

500\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀤𑁆𑀥𑀸?

501\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

503\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

504\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀢𑀺?

506\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀫𑀸𑀥𑀺?

507\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

508\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀸?

509\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

510\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺?

511\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

512\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀁?

513\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

514\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻?

550\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻?

(91.)

551\. »  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

552\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻?

(92.)

553\. »  𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

554\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻?

(93.)

555\. »  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

556\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀁?
