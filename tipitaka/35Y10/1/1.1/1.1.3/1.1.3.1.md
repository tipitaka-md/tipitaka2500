# 𑁧𑁇𑁧𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫

(48.)

301\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

302\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼?

303\. »  𑀲𑁄𑀢𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

304\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑁄𑀢𑀁?

305\. »  𑀖𑀸𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

306\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀖𑀸𑀦𑀁?

307\. »  𑀚𑀺𑀯𑁆𑀳𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

308\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀚𑀺𑀯𑁆𑀳𑀸?

309\. »  𑀓𑀸𑀬𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

310\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀓𑀸𑀬𑁄?

311\. »  𑀫𑀦𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

312\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀫𑀦𑁄?

313\. »  𑀇𑀢𑁆𑀣𑀻 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

314\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀇𑀢𑁆𑀣𑀻?

315\. »  𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

316\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑁄?

317\. »  𑀚𑀻𑀯𑀺𑀢𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

318\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀚𑀻𑀯𑀺𑀢𑀁?

319\. »  𑀲𑀼𑀔𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

320\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀼𑀔𑀁?

321\. »  𑀤𑀼𑀓𑁆𑀔𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

322\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀔𑀁?

323\. »  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

324\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁?

325\. »  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

326\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁?

327\. »  𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

328\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸?

329\. »  𑀲𑀤𑁆𑀥𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

330\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀸?

331\. »  𑀯𑀻𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

332\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀯𑀻𑀭𑀺𑀬𑀁?

333\. »  𑀲𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

334\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀢𑀺?

335\. »  𑀲𑀫𑀸𑀥𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

336\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀫𑀸𑀥𑀺?

337\. »  𑀧𑀜𑁆𑀜𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

338\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸?

339\. »  𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

340\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺?

341\. »  𑀅𑀜𑁆𑀜𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

342\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀁?

343\. »  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

344\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻?
