# 𑁧𑁇𑁧𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(49.)

345\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

346\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼?

347\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

348\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀁?

349\. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

350\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀁?

351\. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

352\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸?

353\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

354\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀓𑀸𑀬𑁄?

355\. »  𑀦 𑀫𑀦𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

356\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑁄?

357\. »  𑀦 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

358\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀇𑀢𑁆𑀣𑀻?

359\. »  𑀦 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

360\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀼𑀭𑀺𑀲𑁄?

361\. »  𑀦 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

362\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀻𑀯𑀺𑀢𑀁?

363\. »  𑀦 𑀲𑀼𑀔𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

364\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀁?

365\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

366\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁?

367\. »  𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

368\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁?

369\. »  𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

370\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁?

371\. »  𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

372\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸?

373\. »  𑀦 𑀲𑀤𑁆𑀥𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

374\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀤𑁆𑀥𑀸?

375\. »  𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

376\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁?

377\. »  𑀦 𑀲𑀢𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

378\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀢𑀺?

379\. »  𑀦 𑀲𑀫𑀸𑀥𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

380\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀫𑀸𑀥𑀺?

381\. »  𑀦 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

382\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀸?

383\. »  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

384\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺?

385\. »  𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

386\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀁?

387\. »  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

388\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻?
