173\. »  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

174\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

175\. »  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

176\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(16.)

177\. »  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

178\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

179\. »  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

180\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(17.)

181\. »  𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

182\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

183\. »  𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

184\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(18.)

185\. »  𑀲𑀤𑁆𑀥𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

186\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

187\. »  𑀲𑀤𑁆𑀥𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

188\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(19.)

189\. »  𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

190\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

191\. »  𑀯𑀻𑀭𑀺𑀬𑀁 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

192\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(20.)

193\. »  𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

194\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

195\. »  𑀲𑀢𑀺 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

196\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(21.)

197\. »  𑀲𑀫𑀸𑀥𑀺 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

198\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

199\. »  𑀲𑀫𑀸𑀥𑀺 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

200\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(22.)

201\. »  𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

202\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

203\. »  𑀧𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

204\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(23.)

205\. »  𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

206\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

207\. »  𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

208\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(24.)

209\. »  𑀅𑀜𑁆𑀜𑀁 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

210\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

211\. »  𑀅𑀜𑁆𑀜𑀁 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

212\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(25.)

213\. »  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

214\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂… 𑁇

215\. »  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

216\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?
