### 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(26.)

217\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

218\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

219\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

220\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

221\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

222\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

223\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

224\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

225\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

226\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

227\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

228\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

229\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

230\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

231\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

232\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

233\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

234\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

235\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

236\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

237\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

238\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

239\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

240\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

241\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

242\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

243\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

244\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

245\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

246\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

247\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

248\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

249\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

250\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

251\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

252\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

253\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

254\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

255\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

256\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

257\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

258\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(27.)

259\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

260\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(28.)

261\. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

262\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(29.)

263\. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

264\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(30.)

265\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

266\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(31.)

267\. »  𑀦 𑀫𑀦𑁄 𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

268\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(32.)

269\. »  𑀦 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

270\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(33.)

271\. »  𑀦 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

272\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(34.)

273\. »  𑀦 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

274\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(35.)

275\. »  𑀦 𑀲𑀼𑀔𑀁 𑀦 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

276\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(36.)

277\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

278\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(37.)

279\. »  𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

280\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(38.)

281\. »  𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

282\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(39.)

283\. »  𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

284\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(40.)

285\. »  𑀦 𑀲𑀤𑁆𑀥𑀸 𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

286\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(41.)

287\. »  𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀦 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

288\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(42.)

289\. »  𑀦 𑀲𑀢𑀺 𑀦 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

290\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(43.)

291\. »  𑀦 𑀲𑀫𑀸𑀥𑀺 𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

292\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(44.)

293\. »  𑀦 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

294\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(45.)

295\. »  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

296\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(46.)

297\. »  𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

298\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

(47.)

299\. »  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

300\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁? …𑀧𑁂…  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?
