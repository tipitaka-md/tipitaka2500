# 𑁧𑁇𑁧𑁇𑁧𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(3.)

47\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

48\. «  𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀘𑀓𑁆𑀔𑀼?

49\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

50\. «  𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑁄𑀢𑀁?

51\. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

52\. «  𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀖𑀸𑀦𑀁?

53\. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

54\. «  𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸?

55\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

56\. «  𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀓𑀸𑀬𑁄?

57\. »  𑀦 𑀫𑀦𑁄 𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

58\. «  𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀫𑀦𑁄?

59\. »  𑀦 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

60\. «  𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀇𑀢𑁆𑀣𑀻?

61\. »  𑀦 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

62\. «  𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀼𑀭𑀺𑀲𑁄?

63\. »  𑀦 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

64\. «  𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀚𑀻𑀯𑀺𑀢𑀁?

65\. »  𑀦 𑀲𑀼𑀔𑀁 𑀦 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

66\. «  𑀦 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀼𑀔𑀁?

67\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

68\. «  𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁?

69\. »  𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

70\. «  𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁?

71\. »  𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

72\. «  𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁?

73\. »  𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

74\. «  𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸?

75\. »  𑀦 𑀲𑀤𑁆𑀥𑀸 𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

76\. «  𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀤𑁆𑀥𑀸?

77\. »  𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀦 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

78\. «  𑀦 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁?

79\. »  𑀦 𑀲𑀢𑀺 𑀦 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

80\. «  𑀦 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀢𑀺?

81\. »  𑀦 𑀲𑀫𑀸𑀥𑀺 𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

82\. «  𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀫𑀸𑀥𑀺?

83\. »  𑀦 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

84\. «  𑀦 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀜𑁆𑀜𑀸?

85\. »  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

86\. «  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺?

87\. »  𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

88\. «  𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀅𑀜𑁆𑀜𑀁?

89\. »  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁?

90\. «  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻?
