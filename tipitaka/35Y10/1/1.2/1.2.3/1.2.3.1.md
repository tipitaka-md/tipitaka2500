# 𑁧𑁇𑁨𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(140.)

947\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

948\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

949\. 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

950\. »  𑀲𑁄𑀢𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

951\. 𑀬𑀁 𑀲𑁄𑀢𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀢𑀁 𑀲𑁄𑀢𑀜𑁆𑀘𑁂𑀯 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀁 𑀲𑁄𑀢𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

952\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

953\. 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

954\. »  𑀖𑀸𑀦𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

955\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

956\. 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

957\. »  𑀚𑀺𑀯𑁆𑀳𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

958\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

959\. 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

960\. »  𑀓𑀸𑀬𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

961\. 𑀬𑁄 𑀓𑀸𑀬𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑁄 𑀓𑀸𑀬𑁄 𑀘𑁂𑀯 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑁄 𑀓𑀸𑀬𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

962\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

963\. 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

964\. »  𑀫𑀦𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

965\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

966\. 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

967\. »  𑀇𑀢𑁆𑀣𑀻 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀦𑁄𑁇

968\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

969\. 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

970\. »  𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀦𑁄𑁇

971\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

972\. 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

973\. »  𑀚𑀻𑀯𑀺𑀢𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

974\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

975\. 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

976\. »  𑀲𑀼𑀔𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

977\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

978\. 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

979\. »  𑀤𑀼𑀓𑁆𑀔𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

980\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

981\. 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

982\. »  𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

983\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

984\. 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

985\. »  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

986\. »  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

987\. 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

988\. »  𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

989\. 𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀘𑁂𑀯 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

990\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

991\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

992\. »  𑀲𑀤𑁆𑀥𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

993\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

994\. 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

995\. »  𑀯𑀻𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

996\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

997\. 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

998\. »  𑀲𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

999\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1000\. 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1001\. »  𑀲𑀫𑀸𑀥𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1002\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1003\. 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1004\. »  𑀧𑀜𑁆𑀜𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1005\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1006\. 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1007\. »  𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1008\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1009\. 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1010\. »  𑀅𑀜𑁆𑀜𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1011\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1012\. 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1013\. »  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1014\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1015\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇
