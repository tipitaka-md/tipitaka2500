# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(141.)

1016\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1017\. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1018\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1019\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1020\. 𑀲𑁄𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑁄𑀢𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑁄𑀢𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1021\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1022\. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1023\. 𑀖𑀸𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀖𑀸𑀦𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀖𑀸𑀦𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1024\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1025\. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1026\. 𑀚𑀺𑀯𑁆𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀚𑀺𑀯𑁆𑀳𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1027\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1028\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1029\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1030\. »  𑀦 𑀫𑀦𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1031\. 𑀫𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀫𑀦𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀦𑁄 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1032\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1033\. »  𑀦 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1034\. 𑀇𑀢𑁆𑀣𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀇𑀢𑁆𑀣𑀻, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀇𑀢𑁆𑀣𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1035\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1036\. »  𑀦 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1037\. 𑀧𑀼𑀭𑀺𑀲𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀼𑀭𑀺𑀲𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀧𑀼𑀭𑀺𑀲𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1038\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1039\. »  𑀦 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1040\. 𑀚𑀻𑀯𑀺𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀻𑀯𑀺𑀢𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀚𑀻𑀯𑀺𑀢𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1041\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1042\. »  𑀦 𑀲𑀼𑀔𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1043\. 𑀲𑀼𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑀼𑀔𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀼𑀔𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1044\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1045\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1046\. 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1047\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1048\. »  𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1049\. 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1050\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1051\. »  𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1052\. 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1053\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1054\. »  𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1055\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1056\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1057\. »  𑀦 𑀲𑀤𑁆𑀥𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1058\. 𑀲𑀤𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀤𑁆𑀥𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑀤𑁆𑀥𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀤𑁆𑀥𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1059\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1060\. »  𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1061\. 𑀯𑀻𑀭𑀺𑀬𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀯𑀻𑀭𑀺𑀬𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1062\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1063\. »  𑀦 𑀲𑀢𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1064\. 𑀲𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀢𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑀢𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1065\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1066\. »  𑀦 𑀲𑀫𑀸𑀥𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1067\. 𑀲𑀫𑀸𑀥𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀫𑀸𑀥𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑀫𑀸𑀥𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀫𑀸𑀥𑀺 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1068\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1069\. »  𑀦 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1070\. 𑀧𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀧𑀜𑁆𑀜𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1071\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1072\. »  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1073\. 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1074\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1075\. »  𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1076\. 𑀅𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀜𑁆𑀜𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1077\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1078\. »  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1079\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1080\. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
