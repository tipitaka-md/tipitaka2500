# 𑁧𑁇𑁨𑁇𑁪𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(142.)

1081\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1082\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1083\. 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1084\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1085\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1086\. 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1087\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1088\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1089\. 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1090\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1091\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1092\. 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1093\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1094\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1095\. 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1096\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1097\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1098\. 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1099\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1100\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1101\. 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1102\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1103\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1104\. 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1105\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1106\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1107\. 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1108\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1109\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1110\. 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1111\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1112\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1113\. 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1114\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1115\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1116\. 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1117\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1118\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1119\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1120\. »  𑀘𑀓𑁆𑀔𑀼 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1121\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1122\. 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1209\. »  𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1210\. 𑀬𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀘𑁂𑀯 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

1211\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1212\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇

(156.)

1213\. »  𑀲𑀤𑁆𑀥𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(157.)

1214\. »  𑀯𑀻𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(158.)

1215\. »  𑀲𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(159.)

1216\. »  𑀲𑀫𑀸𑀥𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(160.)

1217\. »  𑀧𑀜𑁆𑀜𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(161.)

1218\. »  𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸 …𑀧𑁂… 𑁇

(162.)

1219\. »  𑀅𑀜𑁆𑀜𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(163.)

1220\. »  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1221\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1222\. 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁…𑀧𑁂… 𑁇

1223\. »  𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1224\. «  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1225\. 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇
