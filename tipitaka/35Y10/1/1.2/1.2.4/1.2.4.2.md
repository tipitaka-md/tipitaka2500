# 𑁧𑁇𑁨𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(164.)

1226. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1227. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1228. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1229. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1230. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1231. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(165.)

1232. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1233. 𑀲𑁄𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑁄𑀢𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑁄𑀢𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1234. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1235. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1236. 𑀲𑁄𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀲𑁄𑀢𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀲𑁄𑀢𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑁄𑀢𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1237. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(166.)

1238. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1239. 𑀖𑀸𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀖𑀸𑀦𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀖𑀸𑀦𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1240. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1241. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1242. 𑀖𑀸𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀖𑀸𑀦𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀖𑀸𑀦𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀖𑀸𑀦𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1243. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(167.)

1244. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1245. 𑀚𑀺𑀯𑁆𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀚𑀺𑀯𑁆𑀳𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1246. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1247. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1248. 𑀚𑀺𑀯𑁆𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀚𑀺𑀯𑁆𑀳𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1249. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(168.)

1250. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1251. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1252. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1253. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(169.)

1254. »  𑀦 𑀫𑀦𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1255. 𑀫𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀫𑀦𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀦𑁄 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1256. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1257. »  𑀦 𑀫𑀦𑁄 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀫𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀫𑀦𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀫𑀦𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀦𑁄 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1258. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(170.)

1259. »  𑀦 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1260. 𑀇𑀢𑁆𑀣𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀇𑀢𑁆𑀣𑀻, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀇𑀢𑁆𑀣𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

(182.)

1330. »  𑀦 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1331. 𑀧𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀧𑀜𑁆𑀜𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1332. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1333. »  𑀦 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1334. 𑀧𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀧𑀜𑁆𑀜𑀸, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀧𑀜𑁆𑀜𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀧𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1335. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(183.)

1336. »  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1337. 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1338. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1339. »  𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1340. 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1341. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(184.)

1342. »  𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1343. 𑀅𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀜𑁆𑀜𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1344. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1345. »  𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1346. 𑀅𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀜𑁆𑀜𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1347. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(185.)

1348. »  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1349. 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1350. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

1351. »  𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺?

1352. 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀻 𑀦 𑀘 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

1353. «  𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸 𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

---
