# 𑁩𑁇𑁬𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

### 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(475.)

4335. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

4336. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑁄𑁇

4337. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

4338. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑁄𑁇

4339. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

4340. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑁄𑁇

4341. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

4342. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑁄𑁇 (𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(476.)

4343. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

4344. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑁄𑁇

4345. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4346. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀺𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺𑁇

4347. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4348. 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑁇

4349. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4350. 𑀅𑀭𑀳𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀺𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4351. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4352. 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑁇 (𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(477.)

4353. »  𑀬𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4354. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4355. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4356. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦𑁄 𑀘 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 𑀧𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇

4357. »  𑀬𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4358. 𑀅𑀭𑀳𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4359. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4360. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 (𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)
