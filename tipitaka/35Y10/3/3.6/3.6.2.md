# 𑁩𑁇𑁬𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

### 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(479.)

4363\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4364\. 𑀧𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4365\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4366\. 𑀅𑀭𑀳𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇

4367\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4368\. 𑀬𑁂 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4369\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4370\. 𑀅𑀭𑀳𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇

4371\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4372\. 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4373\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4374\. 𑀅𑀭𑀳𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇

4375\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4376\. 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4377\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4378\. 𑀅𑀭𑀳𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇 𑀬𑁂 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇 (𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(480.)

4379\. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺?
