# 𑁩𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀢𑀻𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(463.)

4238\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4239\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑁇 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀢𑁆𑀣𑁇

4240\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀢𑁆𑀣 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4241\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4242\. 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇

4243\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4244\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4245\. 𑀅𑀭𑀳𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇

4246\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

4247\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

4248\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4249\. 𑀅𑀭𑀳𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇

4250\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

4251\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 (𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(464.)

4252\. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4253\. 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇

4254\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4255\. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4256\. 𑀅𑀭𑀳𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇

4257\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

4258\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇

4259\. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4260\. 𑀅𑀭𑀳𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇

4261\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

4262\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺𑁇 (𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(465.)

4263\. »  𑀬𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4264\. 𑀅𑀭𑀳𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣𑁇

4265\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀻𑀢𑀺?

4266\. 𑀅𑀝𑁆𑀞𑀫𑀓𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑀺𑀢𑁆𑀣 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀺𑁇

4267\. »  𑀬𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4268\. 𑀅𑀭𑀳𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇

4269\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀻𑀢𑀺?

4270\. 𑀅𑀝𑁆𑀞𑀫𑀓𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀺𑁇 (𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(466.)

4271\. »  𑀬𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

4272\. 𑀅𑀭𑀳𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣𑁇

4273\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀢𑁆𑀣 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀻𑀢𑀺?
