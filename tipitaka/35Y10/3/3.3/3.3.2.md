# 𑁩𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

### 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(455.)

4186. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4187. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4188. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4189. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4190. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺𑁇

4191. 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4192. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4193. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4194. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4195. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4196. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(456.)

4197. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4198. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4199. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4200. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4201. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4202. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4203. »  𑀬𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4204. 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4205. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(457.)

4206. »  𑀬𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4207. 𑀙 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4208. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

4209. »  𑀬𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4210. 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4211. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀜𑁆𑀜𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)

(458.)

4212. »  𑀬𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

4213. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

4214. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦 𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀅𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀫𑀽𑀮𑀓𑀁𑁇)
