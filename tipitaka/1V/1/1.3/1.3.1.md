# 𑁧𑁇𑁩𑁇𑁧 𑀧𑀞𑀫𑀅𑀦𑀺𑀬𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀧𑀞𑀫𑀅𑀦𑀺𑀬𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1802\. 𑀇𑀫𑁂 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀅𑀦𑀺𑀬𑀢𑀸 𑀥𑀫𑁆𑀫𑀸

  
𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇



(443.)

1803\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀓𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀓𑀳𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸”𑀢𑀺? 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀤𑀺𑀦𑁆𑀦𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑀼𑀓𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀢𑀫𑁆𑀧𑀺 𑀔𑁄 𑀓𑀼𑀮𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀬𑁂𑀦 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀓𑀳𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸”𑀢𑀺? 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀏𑀲𑀸𑀬𑁆𑀬, 𑀑𑀯𑀭𑀓𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀬𑁂𑀦 𑀲𑀸 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺 𑀓𑀸𑀮𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀼𑀮𑁆𑀮𑀧𑀦𑁆𑀢𑁄 𑀓𑀸𑀮𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀪𑀡𑀦𑁆𑀢𑁄𑁇

1804\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀦𑀢𑁆𑀢𑀸 𑀅𑀭𑁄𑀕𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀭𑁄𑀕𑀦𑀢𑁆𑀢𑀸 𑀅𑀪𑀺𑀫𑀗𑁆𑀕𑀮𑀲𑀫𑁆𑀫𑀢𑀸𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀬𑀜𑁆𑀜𑁂𑀲𑀼 𑀙𑀡𑁂𑀲𑀼 𑀉𑀲𑁆𑀲𑀯𑁂𑀲𑀼 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀧𑀞𑀫𑀁 𑀪𑁄𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀢𑀲𑁆𑀲𑀸 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑀁 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀦𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀬𑀁 𑀅𑀬𑁆𑀬𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄 𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦, 𑀅𑀧𑀺 𑀘 𑀤𑀼𑀲𑁆𑀲𑀤𑁆𑀥𑀸𑀧𑀬𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀦𑀸𑀤𑀺𑀬𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1805\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(444.)

1806\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬, 𑀢𑀫𑁂𑀦𑀁 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀤𑁂𑀬𑁆𑀬—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀯𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀯𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸, 𑀬𑁂𑀦 𑀯𑀸 𑀲𑀸 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀢𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑀢𑁄”**𑀢𑀺𑁇 (*𑁧*:**𑁧𑁮**)

(445.)

1807\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1808\. **𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄** 𑀦𑀸𑀫 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀻, 𑀦 𑀬𑀓𑁆𑀔𑀻 𑀦 𑀧𑁂𑀢𑀻 𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑁇 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀤𑀳𑀼𑀚𑀸𑀢𑀸𑀧𑀺 𑀤𑀸𑀭𑀺𑀓𑀸, 𑀧𑀕𑁂𑀯 𑀫𑀳𑀢𑁆𑀢𑀭𑀻𑁇

1809\. **𑀲𑀤𑁆𑀥𑀺𑀦𑁆**𑀢𑀺 𑀏𑀓𑀢𑁄𑁇

1810\. **𑀏𑀓𑁄 𑀏𑀓𑀸𑀬𑀸**𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀘𑁇

1811\. **𑀭𑀳𑁄** 𑀦𑀸𑀫 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀭𑀳𑁄, 𑀲𑁄𑀢𑀲𑁆𑀲 𑀭𑀳𑁄𑁇 **𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀭𑀳𑁄** 𑀦𑀸𑀫 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀅𑀓𑁆𑀔𑀺𑀁 𑀯𑀸 𑀦𑀺𑀔𑀡𑀻𑀬𑀫𑀸𑀦𑁂 𑀪𑀫𑀼𑀓𑀁 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀻𑀬𑀫𑀸𑀦𑁂 𑀲𑀻𑀲𑀁 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀻𑀬𑀫𑀸𑀦𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇 **𑀲𑁄𑀢𑀲𑁆𑀲 𑀭𑀳𑁄** 𑀦𑀸𑀫 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀧𑀓𑀢𑀺𑀓𑀣𑀸 𑀲𑁄𑀢𑀼𑀁𑁇

1812\. **𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁** 𑀦𑀸𑀫 𑀆𑀲𑀦𑀁 𑀓𑀼𑀝𑁆𑀝𑁂𑀦 𑀯𑀸 𑀓𑀯𑀸𑀝𑁂𑀦 𑀯𑀸 𑀓𑀺𑀮𑀜𑁆𑀚𑁂𑀦 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑁂𑀦 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑁂𑀦 𑀯𑀸 𑀣𑀫𑁆𑀪𑁂𑀦 𑀯𑀸 𑀓𑁄𑀢𑁆𑀣𑀍𑀇𑀬𑀸 𑀯𑀸 𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

1813\. **𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂**𑀢𑀺 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑀼𑀁𑁇

1814\. **𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀧𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀸𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀉𑀧𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀧𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀸𑁇 𑀉𑀪𑁄 𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀉𑀪𑁄 𑀯𑀸 𑀦𑀺𑀧𑀦𑁆𑀦𑀸𑁇

1815\. **𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸** 𑀦𑀸𑀫 𑀆𑀕𑀢𑀨𑀮𑀸 𑀅𑀪𑀺𑀲𑀫𑁂𑀢𑀸𑀯𑀺𑀦𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀲𑀸𑀲𑀦𑀸𑁇

1816\. **𑀉𑀧𑀸𑀲𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸𑁇

1817\. **𑀤𑀺𑀲𑁆𑀯𑀸**𑀢𑀺 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸𑁇

1818\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀤𑁂𑀬𑁆𑀬—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀯𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸𑁇 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀯𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸𑁇 𑀬𑁂𑀦 𑀯𑀸 𑀲𑀸 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀯𑀤𑁂𑀬𑁆𑀬 𑀢𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1820\. 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑁄”𑀢𑀺, 𑀲𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑁄”𑀢𑀺, 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀲𑀘𑁆𑀘𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀦𑁄 𑀘 𑀔𑁄 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀦𑁆”𑀢𑀺, 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑁄”𑀢𑀺, 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑁄”𑀢𑀺, 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀞𑀺𑀢𑁄”𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1823\. 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀲𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀞𑀺𑀢𑁄”𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

(450.)

1824\. 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀧𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀲𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀞𑀺𑀢𑁄”𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1825\. **𑀅𑀦𑀺𑀬𑀢𑁄**𑀢𑀺 𑀦 𑀦𑀺𑀬𑀢𑁄, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀯𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀯𑀸𑁇

(451.)

1826\. 𑀕𑀫𑀦𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀕𑀫𑀦𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀕𑀫𑀦𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀕𑀫𑀦𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1827\. 𑀕𑀫𑀦𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀕𑀫𑀦𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀕𑀫𑀦𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀕𑀫𑀦𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑀢𑀺𑁇

---
