# 𑁧𑁇𑁩𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀅𑀦𑀺𑀬𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀦𑀺𑀬𑀢𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀢𑀺𑀬𑀅𑀦𑀺𑀬𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(452.)

1829\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀅𑀮𑀁𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺 𑀓𑀸𑀮𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀼𑀮𑁆𑀮𑀧𑀦𑁆𑀢𑁄 𑀓𑀸𑀮𑀬𑀼𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀪𑀡𑀦𑁆𑀢𑁄𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀢𑀲𑁆𑀲𑀸𑀬𑁂𑀯 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑀁 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀦𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀬𑀁 𑀅𑀬𑁆𑀬𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄 𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦, 𑀅𑀧𑀺 𑀘 𑀤𑀼𑀲𑁆𑀲𑀤𑁆𑀥𑀸𑀧𑀬𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀦𑀸𑀤𑀺𑀬𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1830\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(453.)

1831\. **“𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀆𑀲𑀦𑀁 𑀳𑁄𑀢𑀺 𑀦𑀸𑀮𑀗𑁆𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁, 𑀅𑀮𑀜𑁆𑀘 𑀔𑁄 𑀳𑁄𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀺𑀢𑀼𑀁𑁇 𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀭𑀽𑀧𑁂 𑀆𑀲𑀦𑁂 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬, 𑀢𑀫𑁂𑀦𑀁 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀤𑁂𑀬𑁆𑀬—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸𑁇 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑁄 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸𑁇 𑀬𑁂𑀦 𑀯𑀸 𑀲𑀸 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀯𑀤𑁂𑀬𑁆𑀬 𑀢𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀬𑀢𑁄”**𑀢𑀺𑁇 (*𑁨*:**𑁧𑁯**)

(454.)

1832\. **𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀓𑀼𑀝𑁆𑀝𑁂𑀦 𑀯𑀸 𑀓𑀯𑀸𑀝𑁂𑀦 𑀯𑀸 𑀓𑀺𑀮𑀜𑁆𑀚𑁂𑀦 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑁂𑀦 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑁂𑀦 𑀯𑀸 𑀣𑀫𑁆𑀪𑁂𑀦 𑀯𑀸 𑀓𑁄𑀢𑁆𑀣𑀍𑀇𑀬𑀸 𑀯𑀸 𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

1833\. **𑀦𑀸𑀮𑀗𑁆𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀦𑁆**𑀢𑀺 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑀼𑀁𑁇

1834\. **𑀅𑀮𑀜𑁆𑀘 𑀔𑁄 𑀳𑁄𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀺𑀢𑀼𑀦𑁆**𑀢𑀺 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀺𑀢𑀼𑀁𑁇

1835\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1836\. **𑀢𑀣𑀸𑀭𑀽𑀧𑁂 𑀆𑀲𑀦𑁂**𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑁂 𑀆𑀲𑀦𑁂𑁇

1837\. **𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄** 𑀦𑀸𑀫 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀻, 𑀦 𑀬𑀓𑁆𑀔𑀻 𑀦 𑀧𑁂𑀢𑀻 𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀝𑀺𑀩𑀮𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀚𑀸𑀦𑀺𑀢𑀼𑀁𑁇

1838\. **𑀲𑀤𑁆𑀥𑀺𑀦𑁆**𑀢𑀺 𑀏𑀓𑀢𑁄𑁇

1839\. **𑀏𑀓𑁄 𑀏𑀓𑀸𑀬𑀸**𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀘𑁇

1840\. **𑀭𑀳𑁄** 𑀦𑀸𑀫 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀭𑀳𑁄, 𑀲𑁄𑀢𑀲𑁆𑀲 𑀭𑀳𑁄𑁇 **𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀭𑀳𑁄** 𑀦𑀸𑀫 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀅𑀓𑁆𑀔𑀺𑀁 𑀯𑀸 𑀦𑀺𑀔𑀡𑀻𑀬𑀫𑀸𑀦𑁂 𑀪𑀫𑀼𑀓𑀁 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀻𑀬𑀫𑀸𑀦𑁂 𑀲𑀻𑀲𑀁 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀻𑀬𑀫𑀸𑀦𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇 **𑀲𑁄𑀢𑀲𑁆𑀲 𑀭𑀳𑁄** 𑀦𑀸𑀫 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀧𑀓𑀢𑀺𑀓𑀣𑀸 𑀲𑁄𑀢𑀼𑀁𑁇

1841\. **𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀧𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀸𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀉𑀧𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀧𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀸𑁇 𑀉𑀪𑁄 𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀉𑀪𑁄 𑀯𑀸 𑀦𑀺𑀧𑀦𑁆𑀦𑀸𑁇

1842\. **𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸** 𑀦𑀸𑀫 𑀆𑀕𑀢𑀨𑀮𑀸 𑀅𑀪𑀺𑀲𑀫𑁂𑀢𑀸𑀯𑀺𑀦𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀲𑀸𑀲𑀦𑀸𑁇

1843\. **𑀉𑀧𑀸𑀲𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸, 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸𑁇

1844\. **𑀤𑀺𑀲𑁆𑀯𑀸**𑀢𑀺 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸𑁇

1845\. **𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀤𑁂𑀬𑁆𑀬 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸**𑁇 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑁄 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀯𑀸𑁇 𑀬𑁂𑀦 𑀯𑀸 𑀲𑀸 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀢𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

(455.)

1846\. 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄”𑀢𑀺, 𑀲𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄”𑀢𑀺, 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀲𑀘𑁆𑀘𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀦𑁄 𑀘 𑀔𑁄 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀦𑁆”𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀞𑀺𑀢𑁄”𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1847\. 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄”𑀢𑀺, 𑀲𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀦𑁄 𑀘 𑀔𑁄 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀦𑁆”𑀢𑀺, 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀞𑀺𑀢𑁄”𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1848\. 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀫𑀬𑀸 𑀲𑀼𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺, 𑀲𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀫𑀬𑀸 𑀲𑀼𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺, 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀲𑀘𑁆𑀘𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀦𑁄 𑀘 𑀔𑁄 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀺𑀦𑁆”𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀞𑀺𑀢𑁄”𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1849\. 𑀲𑀸 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀫𑀬𑀸 𑀲𑀼𑀢𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺, 𑀲𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀦𑁄 𑀘 𑀔𑁄 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀺𑀦𑁆𑀢𑀺, 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺, 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  “𑀦𑀸𑀳𑀁 𑀦𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀞𑀺𑀢𑁄”𑀢𑀺, 𑀦 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1857\. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀅𑀦𑀺𑀬𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄; 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻𑀢𑀺𑁇

---
