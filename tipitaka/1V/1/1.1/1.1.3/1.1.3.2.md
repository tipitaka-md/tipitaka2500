# 𑁧𑁇𑁧𑁇𑁩𑁇𑁨 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

(180.)

457\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀭𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀁𑀯𑀡𑁆𑀡𑁂𑀲𑀼𑀁𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺 “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧)

458\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀞𑀓𑁂 𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀸𑀬 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀤𑀸𑀭𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑁄 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺 “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑀩𑁆𑀩𑀁; 𑀬𑁄 𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨)

459\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑁄 𑀫𑀼𑀲𑀮𑁂 𑀉𑀲𑁆𑀲𑀺𑀢𑁂 𑀏𑀓𑀁 𑀫𑀼𑀲𑀮𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑁄 𑀫𑀼𑀲𑀮𑁄 𑀧𑀭𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀤𑀸𑀭𑀓𑀲𑁆𑀲 𑀫𑀢𑁆𑀣𑀓𑁂 𑀅𑀯𑀢𑁆𑀣𑀸𑀲𑀺𑁇 𑀲𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀸”𑀢𑀺𑁇 (𑁩)

460\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑁄 𑀉𑀤𑀼𑀓𑁆𑀔𑀮𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀅𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀯𑀝𑁆𑀝𑁂𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀤𑀸𑀭𑀓𑀁 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀸”𑀢𑀺𑁇 (𑁪)

461\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀺𑀢𑀸𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀓𑀸𑀮𑁂 𑀆𑀭𑁄𑀘𑀺𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀺𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀧𑀢𑀺𑀫𑀸𑀦𑁂𑀢𑀻”𑀢𑀺 𑀧𑀺𑀝𑁆𑀞𑀺𑀬𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀡𑀸𑀫𑁂𑀲𑀺𑁇 𑀲𑁄 𑀧𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫)

462\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀺𑀢𑀸𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀓𑀸𑀮𑁂 𑀆𑀭𑁄𑀘𑀺𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀺𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀧𑀢𑀺𑀫𑀸𑀦𑁂𑀢𑀻”𑀢𑀺 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀬𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀡𑀸𑀫𑁂𑀲𑀺𑁇 𑀲𑁄 𑀧𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁬)

463\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀺𑀢𑀸𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀓𑀸𑀮𑁂 𑀆𑀭𑁄𑀘𑀺𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀺𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀧𑀢𑀺𑀫𑀸𑀦𑁂𑀢𑀻”𑀢𑀺 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀬𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀡𑀸𑀫𑁂𑀲𑀺𑁇 𑀲𑁄 𑀧𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭)

(181.)

464\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀫𑀁𑀲𑀁 𑀓𑀡𑁆𑀞𑁂 𑀯𑀺𑀮𑀕𑁆𑀕𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀻𑀯𑀸𑀬𑀁 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑀮𑁄𑀳𑀺𑀢𑀁 𑀫𑀁𑀲𑀁 𑀧𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮)

465\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀫𑀁𑀲𑀁 𑀓𑀡𑁆𑀞𑁂 𑀯𑀺𑀮𑀕𑁆𑀕𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀻𑀯𑀸𑀬𑀁 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑀮𑁄𑀳𑀺𑀢𑀁 𑀫𑀁𑀲𑀁 𑀧𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁯)

466\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀫𑀁𑀲𑀁 𑀓𑀡𑁆𑀞𑁂 𑀯𑀺𑀮𑀕𑁆𑀕𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀻𑀯𑀸𑀬𑀁 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑀮𑁄𑀳𑀺𑀢𑀁 𑀫𑀁𑀲𑀁 𑀧𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦)

467\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑀕𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑀁 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀕𑁆𑀕𑀓𑀸𑀭𑀺𑀓𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀸𑀮𑀫𑀓𑀁𑀲𑀼𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁧)

468\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀯𑀻𑀫𑀁𑀲𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁨)

469\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀸𑀭𑀯𑀢𑁆𑀣𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀲𑀺𑀮𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀧𑀭𑀺𑀫𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀲𑀺𑀮𑀸 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀢𑁆𑀣𑀓𑁂 𑀅𑀯𑀢𑁆𑀣𑀸𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀸”𑀢𑀺𑁇 (𑁧𑁩)

470\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀸𑀭𑀯𑀢𑁆𑀣𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀲𑀺𑀮𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀧𑀭𑀺𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀢𑁆𑀣𑀓𑁂 𑀲𑀺𑀮𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪--𑁧𑁫)

472\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀓𑀼𑀝𑁆𑀝𑀁 𑀉𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀇𑀝𑁆𑀞𑀓𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀧𑀭𑀺𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀢𑁆𑀣𑀓𑁂 𑀇𑀝𑁆𑀞𑀓𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁭--𑁧𑁮)

473\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀺𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀧𑀭𑀺𑀫𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀯𑀸𑀲𑀻 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀢𑁆𑀣𑀓𑁂 𑀅𑀯𑀢𑁆𑀣𑀸𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀸”𑀢𑀺𑁇 (𑁧𑁯)

474\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀺𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀧𑀭𑀺𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀢𑁆𑀣𑀓𑁂 𑀯𑀸𑀲𑀺𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁦--𑁨𑁧)

475\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀧𑀭𑀺𑀫𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀕𑁄𑀧𑀸𑀦𑀲𑀻 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀢𑁆𑀣𑀓𑁂 𑀅𑀯𑀢𑁆𑀣𑀸𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀸”𑀢𑀺𑁇 (𑁨𑁨)

476\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁂𑀝𑁆𑀞𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀧𑀭𑀺𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀢𑁆𑀣𑀓𑁂 𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁩--𑁨𑁪)

477\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀅𑀝𑁆𑀝𑀓𑀁 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑁄 𑀩𑀦𑁆𑀥𑀸𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑁄 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁫)

478\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀅𑀝𑁆𑀝𑀓𑀁 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑁄 𑀩𑀦𑁆𑀥𑀸𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑁄 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁬--𑁨𑁭)

479\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀸𑀭𑀁 𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀑𑀢𑀭𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀇𑀢𑁄 𑀑𑀢𑀭𑀸𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀑𑀢𑀭𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁮)

480\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀸𑀭𑀁 𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀑𑀢𑀭𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀇𑀢𑁄 𑀑𑀢𑀭𑀸𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀑𑀢𑀭𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁯--𑁩𑁦)

481\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀬𑀸 𑀧𑀻𑀍𑀇𑀢𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀧𑀸𑀢𑁂 𑀧𑀧𑀢𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀺𑀮𑀻𑀯𑀓𑀸𑀭𑀁 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀸𑀢𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁧)

482\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀯𑀸𑀬 𑀲𑀺𑀮𑀁 𑀧𑀯𑀺𑀚𑁆𑀛𑀺𑀁𑀲𑀼𑁇 𑀲𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑁄𑀧𑀸𑀮𑀓𑀁 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀯𑀸𑀬 𑀲𑀺𑀮𑀸 𑀧𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀧𑀯𑀺𑀚𑁆𑀛𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁨)

501\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀧𑀯𑀼𑀢𑁆𑀣𑀧𑀢𑀺𑀓𑀸 𑀚𑀸𑀭𑁂𑀦 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀓𑀼𑀮𑀽𑀧𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀗𑁆𑀖𑀸𑀬𑁆𑀬, 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀤𑀸𑀭𑀓𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁬𑁦)

503\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀏𑀓𑀸 𑀯𑀜𑁆𑀛𑀸, 𑀏𑀓𑀸 𑀯𑀺𑀚𑀸𑀬𑀺𑀦𑀻𑁇 𑀯𑀜𑁆𑀛𑀸 𑀇𑀢𑁆𑀣𑀻 𑀓𑀼𑀮𑀽𑀧𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀚𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀓𑀼𑀝𑀼𑀫𑁆𑀩𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀗𑁆𑀖𑀸𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀸 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀫𑀸𑀢𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺, 𑀤𑀸𑀭𑀓𑁄 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁨)

504\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀏𑀓𑀸 𑀯𑀜𑁆𑀛𑀸, 𑀏𑀓𑀸 𑀯𑀺𑀚𑀸𑀬𑀺𑀦𑀻𑁇 𑀯𑀜𑁆𑀛𑀸 𑀇𑀢𑁆𑀣𑀻 𑀓𑀼𑀮𑀽𑀧𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀚𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀓𑀼𑀝𑀼𑀫𑁆𑀩𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀗𑁆𑀖𑀸𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲𑀸 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀉𑀪𑁄 𑀓𑀸𑀮𑀫𑀓𑀁𑀲𑀼…𑀧𑁂…  𑀉𑀪𑁄 𑀦 𑀓𑀸𑀮𑀫𑀓𑀁𑀲𑀼𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁩--𑁬𑁪)

505\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀇𑀢𑁆𑀣𑀻 𑀓𑀼𑀮𑀽𑀧𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀗𑁆𑀖𑀸𑀬𑁆𑀬, 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀕𑀺𑀦𑀺, 𑀫𑀤𑁆𑀤𑀲𑁆𑀲𑀽”𑀢𑀺𑁇 𑀲𑀸 𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀩𑁆𑀪𑀁 𑀧𑀸𑀢𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁬𑁫)

506\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀇𑀢𑁆𑀣𑀻 𑀓𑀼𑀮𑀽𑀧𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀗𑁆𑀖𑀸𑀬𑁆𑀬, 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀕𑀺𑀦𑀺, 𑀢𑀸𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀩𑁆𑀪𑀁 𑀧𑀸𑀢𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁬𑁬)

507\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀯𑀜𑁆𑀛𑀸 𑀇𑀢𑁆𑀣𑀻 𑀓𑀼𑀮𑀽𑀧𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀗𑁆𑀖𑀸𑀬𑁆𑀬, 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀬𑁂𑀦𑀸𑀳𑀁 𑀯𑀺𑀚𑀸𑀬𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁭)

508\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀯𑀺𑀚𑀸𑀬𑀺𑀦𑀻 𑀇𑀢𑁆𑀣𑀻 𑀓𑀼𑀮𑀽𑀧𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀗𑁆𑀖𑀸𑀬𑁆𑀬, 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀬𑁂𑀦𑀸𑀳𑀁 𑀦 𑀯𑀺𑀚𑀸𑀬𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁮)

509\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀢𑁆𑀢𑀭𑀲𑀯𑀕𑁆𑀕𑀺𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀧𑀢𑁄𑀤𑀓𑁂𑀦 𑀳𑀸𑀲𑁂𑀲𑀼𑀁𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀢𑁆𑀢𑀦𑁆𑀢𑁄 𑀅𑀦𑀲𑁆𑀲𑀸𑀲𑀓𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁯)

510\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀢𑁆𑀢𑀭𑀲𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸𑀢𑀺 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑀼𑀁𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁦)

511\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀽𑀢𑀯𑁂𑀚𑁆𑀚𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀓𑁆𑀔𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁧)

512\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑀸𑀍𑀅𑀬𑀓𑁆𑀔𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑁇 𑀢𑀁 𑀬𑀓𑁆𑀔𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁨)

513\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑀸𑀍𑀅𑀬𑀓𑁆𑀔𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑁇 𑀢𑀁 𑀬𑀓𑁆𑀔𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀢𑀁 𑀬𑀓𑁆𑀔𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀦 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁩--𑁭𑁪)

514\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑀸𑀍𑀅𑀓𑀦𑁆𑀢𑀸𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑁇 𑀢𑀁 𑀯𑀸𑀍𑀆 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁫)

515\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑀸𑀍𑀅𑀓𑀦𑁆𑀢𑀸𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑁇 𑀢𑀁 𑀯𑀸𑀍𑀆 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀢𑀁 𑀯𑀸𑀍𑀆 𑀚𑀻𑀯𑀺𑀢𑀸 𑀦 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁬--𑁭𑁭)

516\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀭𑀓𑀦𑁆𑀢𑀸𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑁇 𑀢𑀁 𑀘𑁄𑀭𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁮)

517\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀭𑀓𑀦𑁆𑀢𑀸𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑁇 𑀢𑀁 𑀘𑁄𑀭𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀢𑀁 𑀘𑁄𑀭𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀦 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁯--𑁮𑁦)

518\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁮𑁧--𑁮𑁪)

519\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀦 𑀕𑀳𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁫)

520\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀦 𑀕𑀳𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁬--𑁮𑁭)

521\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺𑁇 𑀲𑁄 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁮)

522\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺𑁇 𑀲𑁄 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀲𑁄 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁯--𑁯𑁦)

523\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑁂𑀭𑀬𑀺𑀓𑀲𑁆𑀲 𑀦𑀺𑀭𑀬𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺𑁇 𑀲𑁄 𑀉𑀢𑁆𑀢𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁧)

524\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑀲𑁆𑀲 𑀦𑀺𑀭𑀬𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺𑁇 𑀲𑁄 𑀉𑀢𑁆𑀢𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀲𑁄 𑀉𑀢𑁆𑀢𑀲𑀺𑀢𑁆𑀯𑀸 𑀦 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁨--𑁯𑁩)

(189.)

525\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀁 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀤𑀸𑀳𑀻”𑀢𑀺𑁇 𑀢𑀁 𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑀁 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀁 𑀭𑀼𑀓𑁆𑀔𑁄 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁪)

526\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀁 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑁄 𑀙𑀺𑀦𑁆𑀤𑀸𑀳𑀻”𑀢𑀺𑁇 𑀢𑀁 𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑀁 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀁 𑀭𑀼𑀓𑁆𑀔𑁄 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑀺…𑀧𑁂…  𑀭𑀼𑀓𑁆𑀔𑁄 𑀑𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀦 𑀫𑀸𑀭𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁫--𑁯𑁬)

(190.)

527\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀸𑀬𑀁 𑀆𑀮𑀺𑀫𑁆𑀧𑁂𑀲𑀼𑀁; 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀟𑁆𑀠𑀸 𑀓𑀸𑀮𑀫𑀓𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁭)

528\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀸 𑀤𑀸𑀬𑀁 𑀆𑀮𑀺𑀫𑁆𑀧𑁂𑀲𑀼𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀟𑁆𑀠𑀸 𑀓𑀸𑀮𑀫𑀓𑀁𑀲𑀼…𑀧𑁂…  𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀟𑁆𑀠𑀸 𑀦 𑀓𑀸𑀮𑀫𑀓𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁮--𑁯𑁯)

(191.)

529\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀖𑀸𑀢𑀦𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀘𑁄𑀭𑀖𑀸𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸𑀬𑀺𑀫𑀁 𑀓𑀺𑀮𑀫𑁂𑀲𑀺𑁇 𑀏𑀓𑁂𑀦 𑀧𑀳𑀸𑀭𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀏𑀓𑁂𑀦 𑀧𑀳𑀸𑀭𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁦)

530\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀖𑀸𑀢𑀦𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀘𑁄𑀭𑀖𑀸𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸𑀬𑀺𑀫𑀁 𑀓𑀺𑀮𑀫𑁂𑀲𑀺𑁇 𑀏𑀓𑁂𑀦 𑀧𑀳𑀸𑀭𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄—  “𑀦𑀸𑀳𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀯𑀘𑀦𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦𑁧)

(192.)

531\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀜𑀸𑀢𑀺𑀖𑀭𑁂 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀜𑀸𑀢𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀇𑀘𑁆𑀙𑀣 𑀇𑀫𑀲𑁆𑀲 𑀫𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀘𑁆𑀙𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑀓𑁆𑀓𑀁 𑀧𑀸𑀬𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀢𑀓𑁆𑀓𑀁 𑀧𑀸𑀬𑁂𑀲𑀼𑀁𑁇 𑀲𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁨)

532\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀼𑀮𑀖𑀭𑁂 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀜𑀸𑀢𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀇𑀘𑁆𑀙𑀣 𑀇𑀫𑀲𑁆𑀲 𑀫𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀆𑀫𑀸𑀬𑁆𑀬𑁂, 𑀇𑀘𑁆𑀙𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀮𑁄𑀡𑀲𑁄𑀯𑀻𑀭𑀓𑀁 𑀧𑀸𑀬𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀮𑁄𑀡𑀲𑁄𑀯𑀻𑀭𑀓𑀁 𑀧𑀸𑀬𑁂𑀲𑀼𑀁𑁇 𑀲𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁩)
