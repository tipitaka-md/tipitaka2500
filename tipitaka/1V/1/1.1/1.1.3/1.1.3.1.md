# 𑁧𑁇𑁧𑁇𑁩𑁇𑁧 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

451\. _𑀲𑀁𑀯𑀡𑁆𑀡𑀦𑀸 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑁄,_  
_𑀫𑀼𑀲𑀮𑁄𑀤𑀼𑀓𑁆𑀔𑀮𑁂𑀦 𑀘;_  
_𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸𑀪𑀺𑀲𑀦𑁆𑀦𑁄,_  
_𑀅𑀕𑁆𑀕𑀯𑀻𑀫𑀁𑀲𑀦𑀸𑀯𑀺𑀲𑀁𑁇_  


452\. _𑀢𑀬𑁄 𑀘 𑀯𑀢𑁆𑀣𑀼𑀓𑀫𑁆𑀫𑁂𑀳𑀺,_  
_𑀇𑀝𑁆𑀞𑀓𑀸𑀳𑀺𑀧𑀭𑁂 𑀢𑀬𑁄;_  
_𑀯𑀸𑀲𑀻 𑀕𑁄𑀧𑀸𑀦𑀲𑀻 𑀘𑁂𑀯,_  
_𑀅𑀝𑁆𑀝𑀓𑁄𑀢𑀭𑀡𑀁 𑀧𑀢𑀺𑁇_  


453\. _𑀲𑁂𑀤𑀁 𑀦𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀲𑀫𑁆𑀩𑀸𑀳𑁄,_  
_𑀦𑁆𑀳𑀸𑀧𑀦𑀩𑁆𑀪𑀜𑁆𑀚𑀦𑁂𑀦 𑀘;_  
_𑀉𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀦𑀺𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀫𑀸𑀭𑀡𑀁𑁇_  


454\. _𑀚𑀸𑀭𑀕𑀩𑁆𑀪𑁄 𑀲𑀧𑀢𑁆𑀢𑀻 𑀘,_  
_𑀫𑀸𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀁 𑀉𑀪𑁄 𑀯𑀥𑀺;_  
_𑀉𑀪𑁄 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂 𑀫𑀤𑁆𑀤𑀸,_  
_𑀢𑀸𑀧𑀁 𑀯𑀜𑁆𑀛𑀸 𑀯𑀺𑀚𑀸𑀬𑀺𑀦𑀻𑁇_  


455\. _𑀧𑀢𑁄𑀤𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂 𑀬𑀓𑁆𑀔𑁄,_  
_𑀯𑀸𑀍𑀅𑀬𑀓𑁆𑀔𑀜𑁆𑀘 𑀧𑀸𑀳𑀺𑀡𑀺;_  
_𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀧𑀳𑀭𑀺,_  
_𑀲𑀕𑁆𑀕𑀜𑁆𑀘 𑀦𑀺𑀭𑀬𑀁 𑀪𑀡𑁂𑁇_  


456\. _𑀆𑀍𑀅𑀯𑀺𑀬𑀸 𑀢𑀬𑁄 𑀭𑀼𑀓𑁆𑀔𑀸,_  
_𑀤𑀸𑀬𑁂𑀳𑀺 𑀅𑀧𑀭𑁂 𑀢𑀬𑁄;_  
_𑀫𑀸 𑀓𑀺𑀮𑀫𑁂𑀲𑀺 𑀦 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀢𑀓𑁆𑀓𑀁 𑀲𑁄𑀯𑀻𑀭𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_
