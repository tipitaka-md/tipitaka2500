# 𑁧𑁇𑁧𑁇𑁩 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(162.)

376\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀲𑀼𑀪𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀅𑀲𑀼𑀪𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀤𑀺𑀲𑁆𑀲 𑀅𑀲𑀼𑀪𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

377\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁𑁇 𑀦𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺, 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑁆𑀲𑀼𑀥 𑀓𑁄𑀘𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

378\. 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀸 𑀔𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀲𑀼𑀪𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀅𑀲𑀼𑀪𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀤𑀺𑀲𑁆𑀲 𑀅𑀲𑀼𑀪𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀻”𑀢𑀺, 𑀢𑁂 𑀅𑀦𑁂𑀓𑀸𑀓𑀸𑀭𑀯𑁄𑀓𑀸𑀭𑀁 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀲𑀓𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑀦𑁆𑀢𑀺 𑀳𑀭𑀸𑀬𑀦𑁆𑀢𑀺 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀇𑀢𑁆𑀣𑀻 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀸 𑀤𑀳𑀭𑁄 𑀬𑀼𑀯𑀸 𑀫𑀡𑁆𑀟𑀦𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑁄 𑀅𑀳𑀺𑀓𑀼𑀡𑀧𑁂𑀦 𑀯𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀓𑀼𑀡𑀧𑁂𑀦 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀼𑀡𑀧𑁂𑀦 𑀯𑀸 𑀓𑀡𑁆𑀞𑁂 𑀆𑀲𑀢𑁆𑀢𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑁂𑀬𑁆𑀬 𑀳𑀭𑀸𑀬𑁂𑀬𑁆𑀬 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑀦𑁆𑀢𑀸 𑀳𑀭𑀸𑀬𑀦𑁆𑀢𑀸 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑀫𑁆𑀧𑀺 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀲𑀸𑀥𑀼 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀺𑁇 𑀇𑀤𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑁄 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑁂𑀳𑀺 𑀪𑀝𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀮𑁄𑀳𑀺𑀢𑀓𑀁 𑀅𑀲𑀺𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸 𑀦𑀤𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(163.)

379\. 𑀅𑀣 𑀔𑁄 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀮𑁄𑀳𑀺𑀢𑀓𑀁 𑀢𑀁 𑀅𑀲𑀺𑀁 𑀥𑁄𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀳𑀼𑀤𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑀼 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄—  “𑀅𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀮𑀸𑀪𑀸; 𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀲𑀼𑀮𑀤𑁆𑀥𑀁𑁇 𑀩𑀳𑀼𑀁 𑀯𑀢 𑀫𑀬𑀸 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀼𑀢𑀁, 𑀬𑁄𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀫𑀸𑀭𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀪𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀉𑀤𑀓𑁂 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑀁 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲, 𑀮𑀸𑀪𑀸 𑀢𑁂, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀢𑁂, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁇 𑀩𑀳𑀼𑀁 𑀢𑀬𑀸, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲, 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀼𑀢𑀁, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀅𑀢𑀺𑀡𑁆𑀡𑁂 𑀢𑀸𑀭𑁂𑀲𑀻”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

380\. 𑀅𑀣 𑀔𑁄 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑁄 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑁄—  “𑀮𑀸𑀪𑀸 𑀓𑀺𑀭 𑀫𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀓𑀺𑀭 𑀫𑁂, 𑀩𑀳𑀼𑀁 𑀓𑀺𑀭 𑀫𑀬𑀸 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀼𑀢𑀁, 𑀅𑀢𑀺𑀡𑁆𑀡𑁂 𑀓𑀺𑀭𑀸𑀳𑀁 𑀢𑀸𑀭𑁂𑀫𑀻”𑀢𑀺, 𑀢𑀺𑀡𑁆𑀳𑀁 𑀅𑀲𑀺𑀁 𑀆𑀤𑀸𑀬 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀭𑀺𑀯𑁂𑀡𑁂𑀦 𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀓𑁄 𑀅𑀢𑀺𑀡𑁆𑀡𑁄, 𑀓𑀁 𑀢𑀸𑀭𑁂𑀫𑀻”𑀢𑀺? 𑀢𑀢𑁆𑀣 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀢𑁂𑀲𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑁄𑀢𑀺𑀬𑁂𑀯 𑀪𑀬𑀁 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑁄𑁇 𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀢𑁂𑀲𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺 𑀪𑀬𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑁄𑁇

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

381\. 𑀅𑀣 𑀔𑁄 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑁄 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑁄 𑀏𑀓𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀤𑁆𑀯𑁂𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀢𑀬𑁄𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀧𑀜𑁆𑀘𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀤𑀲𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀯𑀻𑀲𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀢𑀺𑀁𑀲𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺, 𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

382\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀲𑁆𑀲 𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀦𑀼𑀪𑀽𑀢𑁄 𑀯𑀺𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄”𑀢𑀺? “𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀲𑀼𑀪𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀅𑀲𑀼𑀪𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀤𑀺𑀲𑁆𑀲 𑀅𑀲𑀼𑀪𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀢𑁂 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽—  ‘𑀪𑀕𑀯𑀸 𑀔𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀲𑀼𑀪𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀅𑀲𑀼𑀪𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀤𑀺𑀲𑁆𑀲 𑀅𑀲𑀼𑀪𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀻’𑀢𑀺, 𑀢𑁂 𑀅𑀦𑁂𑀓𑀸𑀓𑀸𑀭𑀯𑁄𑀓𑀸𑀭𑀁 𑀅𑀲𑀼𑀪𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀲𑀓𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑀦𑁆𑀢𑀺 𑀳𑀭𑀸𑀬𑀦𑁆𑀢𑀺 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀇𑀢𑁆𑀣𑀻 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀸 𑀤𑀳𑀭𑁄 𑀬𑀼𑀯𑀸 𑀫𑀡𑁆𑀟𑀦𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑁄 𑀅𑀳𑀺𑀓𑀼𑀡𑀧𑁂𑀦 𑀯𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀓𑀼𑀡𑀧𑁂𑀦 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀼𑀡𑀧𑁂𑀦 𑀯𑀸 𑀓𑀡𑁆𑀞𑁂 𑀆𑀲𑀢𑁆𑀢𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑁂𑀬𑁆𑀬 𑀳𑀭𑀸𑀬𑁂𑀬𑁆𑀬 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀓𑁂𑀦 𑀓𑀸𑀬𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑀦𑁆𑀢𑀸 𑀳𑀭𑀸𑀬𑀦𑁆𑀢𑀸 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀸 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑀫𑁆𑀧𑀺 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  ‘𑀲𑀸𑀥𑀼 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀺𑁇 𑀇𑀤𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑁄 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑁂𑀳𑀺 𑀪𑀝𑁄 𑀏𑀓𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀳𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀼 𑀬𑀣𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀅𑀜𑁆𑀜𑀸𑀬 𑀲𑀡𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦𑀳𑀸𑀦𑀦𑁆𑀤, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺, 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄; 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇

383\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

384\. “𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀦𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀡𑀻𑀢𑁄 𑀘 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀘 𑀲𑀼𑀔𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀘 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀉𑀳𑀢𑀁 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀢𑀫𑁂𑀦𑀁 𑀫𑀳𑀸𑀅𑀓𑀸𑀮𑀫𑁂𑀖𑁄 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀺 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑀦𑁆𑀢𑁄 𑀘𑁂𑀯 𑀧𑀡𑀻𑀢𑁄 𑀘 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀘 𑀲𑀼𑀔𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀘 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀞𑀸𑀦𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺𑁇

390\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑀫𑁆𑀧𑀺 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  ‘𑀲𑀸𑀥𑀼 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀺, 𑀇𑀤𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀺𑀕𑀮𑀡𑁆𑀟𑀺𑀓𑀫𑁆𑀧𑀺 𑀲𑀫𑀡𑀓𑀼𑀢𑁆𑀢𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀲𑀸𑀥𑀼 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀺, 𑀇𑀤𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

391\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬 𑀲𑀢𑁆𑀣𑀳𑀸𑀭𑀓𑀁 𑀯𑀸𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁂𑀲𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑁄”**𑀢𑀺𑁇

393\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀚𑀸𑀧𑀢𑀺 𑀅𑀪𑀺𑀭𑀽𑀧𑀸 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀚𑀻𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀫𑀬𑀁 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀮𑀪𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀫𑀭𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀁𑀯𑀡𑁆𑀡𑁂𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀉𑀧𑀸𑀲𑀓, 𑀓𑀢𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀢𑀓𑀼𑀲𑀮𑁄 𑀓𑀢𑀪𑀻𑀭𑀼𑀢𑁆𑀢𑀸𑀡𑁄 𑀅𑀓𑀢𑀧𑀸𑀧𑁄 𑀅𑀓𑀢𑀮𑀼𑀤𑁆𑀤𑁄 𑀅𑀓𑀢𑀓𑀺𑀩𑁆𑀩𑀺𑀲𑁄𑁇 𑀓𑀢𑀁 𑀢𑀬𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀁, 𑀅𑀓𑀢𑀁 𑀢𑀬𑀸 𑀧𑀸𑀧𑀁𑁇 𑀓𑀺𑀁 𑀢𑀼𑀬𑁆𑀳𑀺𑀫𑀺𑀦𑀸 𑀧𑀸𑀧𑀓𑁂𑀦 𑀤𑀼𑀚𑁆𑀚𑀻𑀯𑀺𑀢𑁂𑀦𑁇 𑀫𑀢𑀁 𑀢𑁂 𑀚𑀻𑀯𑀺𑀢𑀸 𑀲𑁂𑀬𑁆𑀬𑁄𑁇 𑀇𑀢𑁄 𑀢𑁆𑀯𑀁 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀢𑀢𑁆𑀣 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀲𑁆𑀲𑀲𑀻”𑀢𑀺𑁇

397\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬 𑀲𑀢𑁆𑀣𑀳𑀸𑀭𑀓𑀁 𑀯𑀸𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁂𑀲𑁂𑀬𑁆𑀬 𑀫𑀭𑀡𑀯𑀡𑁆𑀡𑀁 𑀯𑀸 𑀲𑀁𑀯𑀡𑁆𑀡𑁂𑀬𑁆𑀬 𑀫𑀭𑀡𑀸𑀬 𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀬𑁆𑀬—  ‘𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀓𑀺𑀁 𑀢𑀼𑀬𑁆𑀳𑀺𑀫𑀺𑀦𑀸 𑀧𑀸𑀧𑀓𑁂𑀦 𑀤𑀼𑀚𑁆𑀚𑀻𑀯𑀺𑀢𑁂𑀦, 𑀫𑀢𑀁 𑀢𑁂 𑀚𑀻𑀯𑀺𑀢𑀸 𑀲𑁂𑀬𑁆𑀬𑁄’𑀢𑀺, 𑀇𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀫𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀫𑀭𑀡𑀯𑀡𑁆𑀡𑀁 𑀯𑀸 𑀲𑀁𑀯𑀡𑁆𑀡𑁂𑀬𑁆𑀬, 𑀫𑀭𑀡𑀸𑀬 𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑁄”**𑀢𑀺𑁇 (*𑁩*:**𑁩**)

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(172.)

398\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

399\. **𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀸**𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀘𑁂𑀘𑁆𑀘 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

400\. **𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑁄** 𑀦𑀸𑀫 𑀬𑀁 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀫𑀺𑀁 𑀧𑀞𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀞𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀁, 𑀬𑀸𑀯 𑀫𑀭𑀡𑀓𑀸𑀮𑀸 𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀏𑀲𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀦𑀸𑀫𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

401\. **𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀉𑀧𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀉𑀧𑀭𑁄𑀥𑁂𑀢𑀺 𑀲𑀦𑁆𑀢𑀢𑀺𑀁 𑀯𑀺𑀓𑁄𑀧𑁂𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

402\. **𑀲𑀢𑁆𑀣𑀳𑀸𑀭𑀓𑀁 𑀯𑀸𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁂𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀲𑀺𑀁 𑀯𑀸 𑀲𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀪𑁂𑀡𑁆𑀟𑀺𑀁 𑀯𑀸 𑀮𑀕𑀼𑀍𑀅𑀁 𑀯𑀸 𑀧𑀸𑀲𑀸𑀡𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀭𑀚𑁆𑀚𑀼𑀁 𑀯𑀸𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

403\. **𑀫𑀭𑀡𑀯𑀡𑁆𑀡𑀁 𑀯𑀸 𑀲𑀁𑀯𑀡𑁆𑀡𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑁂 𑀆𑀤𑀻𑀦𑀯𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑀺, 𑀫𑀭𑀡𑁂 𑀯𑀡𑁆𑀡𑀁 𑀪𑀡𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

404\. **𑀫𑀭𑀡𑀸𑀬 𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀆𑀳𑀭, 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀔𑀸𑀤, 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀯𑀸 𑀉𑀩𑁆𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀳𑀻𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

405\. **𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲𑀸**𑀢𑀺 𑀆𑀮𑀧𑀦𑀸𑀥𑀺𑀯𑀘𑀦𑀫𑁂𑀢𑀁𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

406\. **𑀓𑀺𑀁 𑀢𑀼𑀬𑁆𑀳𑀺𑀫𑀺𑀦𑀸 𑀧𑀸𑀧𑀓𑁂𑀦 𑀤𑀼𑀚𑁆𑀚𑀻𑀯𑀺𑀢𑁂𑀦𑀸**𑀢𑀺 𑀧𑀸𑀧𑀓𑀁 𑀦𑀸𑀫 𑀚𑀻𑀯𑀺𑀢𑀁 𑀅𑀟𑁆𑀠𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀤𑀮𑀺𑀤𑁆𑀤𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀧𑀸𑀧𑀓𑀁 𑀮𑀸𑀫𑀓𑀁, 𑀲𑀥𑀦𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀅𑀥𑀦𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀧𑀸𑀧𑀓𑀁, 𑀤𑁂𑀯𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀧𑀸𑀧𑀓𑀁𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

407\. **𑀤𑀼𑀚𑁆𑀚𑀻𑀯𑀺𑀢𑀁** 𑀦𑀸𑀫 𑀳𑀢𑁆𑀣𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀸𑀤𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀡𑁆𑀡𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀦𑀸𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀲𑁆𑀲, 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀧𑀸𑀧𑀓𑁂𑀦 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀤𑀼𑀚𑁆𑀚𑀻𑀯𑀺𑀢𑁂𑀦 𑀫𑀢𑀁 𑀢𑁂 𑀚𑀻𑀯𑀺𑀢𑀸 𑀲𑁂𑀬𑁆𑀬𑁄𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

408\. **𑀇𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀫𑀦𑁄**𑀢𑀺 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀢𑀁 𑀫𑀦𑁄, 𑀬𑀁 𑀫𑀦𑁄 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

409\. **𑀘𑀺𑀢𑁆𑀢𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄**𑀢𑀺 𑀫𑀭𑀡𑀲𑀜𑁆𑀜𑀻 𑀫𑀭𑀡𑀘𑁂𑀢𑀦𑁄 𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

410\. **𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸**𑀢𑀺 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑁂𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

422\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀅𑀜𑁆𑀜𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺; 𑀫𑀽𑀮𑀝𑁆𑀞𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀯𑀥𑀓𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

424\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀧𑀸𑀯𑀤—  ‘𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀧𑀸𑀯𑀤𑀢𑀼—  𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀽’”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀯𑀥𑀓𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀫𑀽𑀮𑀝𑁆𑀞𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺; 𑀆𑀡𑀸𑀧𑀓𑀲𑁆𑀲 𑀘 𑀯𑀥𑀓𑀲𑁆𑀲 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

425\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀼𑀦 𑀧𑀘𑁆𑀘𑀸𑀕𑀘𑁆𑀙𑀢𑀺—  “𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀧𑀼𑀦 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  “𑀬𑀤𑀸 𑀲𑀓𑁆𑀓𑁄𑀲𑀺 𑀢𑀤𑀸 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

426\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀆𑀡𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀦 𑀲𑀸𑀯𑁂𑀢𑀺—  “𑀫𑀸 𑀖𑀸𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

427\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀆𑀡𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀲𑀸𑀯𑁂𑀢𑀺—  “𑀫𑀸 𑀖𑀸𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄—  “𑀆𑀡𑀢𑁆𑀢𑁄 𑀅𑀳𑀁 𑀢𑀬𑀸”𑀢𑀺 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀫𑀽𑀮𑀝𑁆𑀞𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀯𑀥𑀓𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

428\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀆𑀡𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀲𑀸𑀯𑁂𑀢𑀺—  “𑀫𑀸 𑀖𑀸𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀸𑀥𑀽𑀢𑀺 𑀑𑀭𑀫𑀢𑀺, 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

433\. **𑀤𑀽𑀢𑁂𑀦 𑀲𑀁𑀯𑀡𑁆𑀡𑁂𑀢𑀺** 𑀦𑀸𑀫 𑀤𑀽𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀢𑀺—  “𑀬𑁄 𑀏𑀯𑀁 𑀫𑀭𑀢𑀺 𑀲𑁄 𑀥𑀦𑀁 𑀯𑀸 𑀮𑀪𑀢𑀺 𑀬𑀲𑀁 𑀯𑀸 𑀮𑀪𑀢𑀺 𑀲𑀕𑁆𑀕𑀁 𑀯𑀸 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀤𑀽𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

437\. **𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑀁** 𑀦𑀸𑀫 𑀅𑀲𑀺𑀁 𑀯𑀸 𑀲𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀪𑁂𑀡𑁆𑀟𑀺𑀁 𑀯𑀸 𑀮𑀕𑀼𑀍𑀅𑀁 𑀯𑀸 𑀧𑀸𑀲𑀸𑀡𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀭𑀚𑁆𑀚𑀼𑀁 𑀯𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀢𑀺—  “𑀇𑀫𑀺𑀦𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 “𑀢𑁂𑀦 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

438\. **𑀪𑁂𑀲𑀚𑁆𑀚𑀁** 𑀦𑀸𑀫 𑀲𑀧𑁆𑀧𑀺𑀁 𑀯𑀸 𑀦𑀯𑀦𑀻𑀢𑀁 𑀯𑀸 𑀢𑁂𑀮𑀁 𑀯𑀸 𑀫𑀥𑀼𑀁 𑀯𑀸 𑀨𑀸𑀡𑀺𑀢𑀁 𑀯𑀸 𑀤𑁂𑀢𑀺—  “𑀇𑀫𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀸𑀬𑀺𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

439\. **𑀭𑀽𑀧𑀽𑀧𑀳𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀅𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀪𑀬𑀸𑀦𑀓𑀁 𑀪𑁂𑀭𑀯𑀁—  “𑀇𑀫𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺—  “𑀇𑀫𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

440\. **𑀲𑀤𑁆𑀤𑀽𑀧𑀳𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀅𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀲𑀤𑁆𑀤𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀪𑀬𑀸𑀦𑀓𑀁 𑀪𑁂𑀭𑀯𑀁—  “𑀇𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀲𑀤𑁆𑀤𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀧𑁂𑀫𑀦𑀻𑀬𑀁 𑀳𑀤𑀬𑀗𑁆𑀕𑀫𑀁—  “𑀇𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

441\. **𑀕𑀦𑁆𑀥𑀽𑀧𑀳𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀅𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀕𑀦𑁆𑀥𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀁 𑀧𑀸𑀝𑀺𑀓𑀼𑀮𑁆𑀬𑀁—  “𑀇𑀫𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀢𑀸 𑀧𑀸𑀝𑀺𑀓𑀼𑀮𑁆𑀬𑀢𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀖𑀸𑀬𑀺𑀢𑁂 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀢𑀸 𑀧𑀸𑀝𑀺𑀓𑀼𑀮𑁆𑀬𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀕𑀦𑁆𑀥𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺—  “𑀇𑀫𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

442\. **𑀭𑀲𑀽𑀧𑀳𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀅𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀭𑀲𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀁 𑀧𑀸𑀝𑀺𑀓𑀼𑀮𑁆𑀬𑀁—  “𑀇𑀫𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀢𑀸 𑀧𑀸𑀝𑀺𑀓𑀼𑀮𑁆𑀬𑀢𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀸𑀬𑀺𑀢𑁂 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀢𑀸 𑀧𑀸𑀝𑀺𑀓𑀼𑀮𑁆𑀬𑀢𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀭𑀲𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺—  “𑀇𑀫𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

443\. **𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀽𑀧𑀳𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀅𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀔𑀭𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁—  “𑀇𑀫𑀺𑀦𑀸 𑀨𑀼𑀝𑁆𑀞𑁄 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺 𑀲𑀼𑀔𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀫𑀼𑀤𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁—  “𑀇𑀫𑀺𑀦𑀸 𑀨𑀼𑀝𑁆𑀞𑁄 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑁄 𑀅𑀮𑀸𑀪𑀓𑁂𑀦 𑀲𑀼𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

444\. **𑀥𑀫𑁆𑀫𑀽𑀧𑀳𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀦𑁂𑀭𑀬𑀺𑀓𑀲𑁆𑀲 𑀦𑀺𑀭𑀬𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺—  “𑀇𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺—  “𑀇𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(179.)

445\. **𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑀸** 𑀦𑀸𑀫 𑀧𑀼𑀝𑁆𑀞𑁄 𑀪𑀡𑀢𑀺—  “𑀏𑀯𑀁 𑀫𑀭𑀲𑁆𑀲𑀼𑁇 𑀬𑁄 𑀏𑀯𑀁 𑀫𑀭𑀢𑀺 𑀲𑁄 𑀥𑀦𑀁 𑀯𑀸 𑀮𑀪𑀢𑀺 𑀬𑀲𑀁 𑀯𑀸 𑀮𑀪𑀢𑀺 𑀲𑀕𑁆𑀕𑀁 𑀯𑀸 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀸𑀬 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑀸𑀬 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

446\. **𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻** 𑀦𑀸𑀫 𑀅𑀧𑀼𑀝𑁆𑀞𑁄 𑀪𑀡𑀢𑀺—  “𑀏𑀯𑀁 𑀫𑀭𑀲𑁆𑀲𑀼𑁇 𑀬𑁄 𑀏𑀯𑀁 𑀫𑀭𑀢𑀺 𑀲𑁄 𑀥𑀦𑀁 𑀯𑀸 𑀮𑀪𑀢𑀺 𑀬𑀲𑀁 𑀯𑀸 𑀮𑀪𑀢𑀺 𑀲𑀕𑁆𑀕𑀁 𑀯𑀸 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀸𑀬 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀬𑀸 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀫𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

447\. **𑀲𑀗𑁆𑀓𑁂𑀢𑀓𑀫𑁆𑀫𑀁** 𑀦𑀸𑀫 𑀲𑀗𑁆𑀓𑁂𑀢𑀁 𑀓𑀭𑁄𑀢𑀺 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀤𑀺𑀯𑀸 𑀯𑀸—  “𑀢𑁂𑀦 𑀲𑀗𑁆𑀓𑁂𑀢𑁂𑀦 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂𑀦 𑀲𑀗𑁆𑀓𑁂𑀢𑁂𑀦 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀢𑀁 𑀲𑀗𑁆𑀓𑁂𑀢𑀁 𑀧𑀼𑀭𑁂 𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀯𑀸 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀫𑀽𑀮𑀝𑁆𑀞𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀯𑀥𑀓𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

448\. **𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀁** 𑀦𑀸𑀫 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀅𑀓𑁆𑀔𑀺𑀁 𑀯𑀸 𑀦𑀺𑀔𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀫𑀼𑀓𑀁 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀻𑀲𑀁 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑁂𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀳𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀧𑀼𑀭𑁂 𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀯𑀸 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀫𑀽𑀮𑀝𑁆𑀞𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀯𑀥𑀓𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

449\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀫𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀢𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤
