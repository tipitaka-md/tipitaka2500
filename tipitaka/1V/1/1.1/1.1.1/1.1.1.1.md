# 𑁧𑁇𑁧𑁇𑁧𑁇𑁧 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀪𑀸𑀡𑀯𑀸𑀭

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀪𑀸𑀡𑀯𑀸𑀭

(24.)

41. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀓𑀮𑀦𑁆𑀤𑀕𑀸𑀫𑁄 𑀦𑀸𑀫 𑀅𑀢𑁆𑀣𑀺𑁇 𑀢𑀢𑁆𑀣 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀦𑀸𑀫 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀲𑀳𑀸𑀬𑀓𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀅𑀕𑀫𑀸𑀲𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀫𑀳𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀭𑀺𑀯𑀼𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀲𑀸 𑀧𑀭𑀺𑀲𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄 𑀅𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁; 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀧𑀭𑀺𑀲𑀸 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

(25.)

42. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀣𑀸 𑀬𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁; 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼 𑀫𑀁 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄𑀲𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺? “𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀦 𑀔𑁄, 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀲𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀣𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀣𑀸 𑀫𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇

(26.)

43. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀻𑀭𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀓𑀮𑀦𑁆𑀤𑀕𑀸𑀫𑁄 𑀬𑁂𑀦 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀫𑁆𑀫𑀢𑀸𑀢𑀸, 𑀬𑀣𑀸 𑀬𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁; 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀣 𑀫𑀁 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄 𑀲𑀼𑀔𑀧𑀭𑀺𑀳𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀚𑀸𑀦𑀸𑀲𑀺𑁇 𑀫𑀭𑀡𑁂𑀦𑀧𑀺 𑀫𑀬𑀁 𑀢𑁂 𑀅𑀓𑀸𑀫𑀓𑀸 𑀯𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑀬𑀁 𑀢𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇

44. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀫𑁆𑀫𑀢𑀸𑀢𑀸, 𑀬𑀣𑀸 𑀬𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁; 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀣 𑀫𑀁 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄 𑀲𑀼𑀔𑀧𑀭𑀺𑀳𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀚𑀸𑀦𑀸𑀲𑀺𑁇 𑀫𑀭𑀡𑁂𑀦𑀧𑀺 𑀫𑀬𑀁 𑀢𑁂 𑀅𑀓𑀸𑀫𑀓𑀸 𑀯𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑀬𑀁 𑀢𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇

45. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀫𑁆𑀫𑀢𑀸𑀢𑀸, 𑀬𑀣𑀸 𑀬𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁; 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀣 𑀫𑀁 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄 𑀲𑀼𑀔𑀧𑀭𑀺𑀳𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀚𑀸𑀦𑀸𑀲𑀺𑁇 𑀫𑀭𑀡𑁂𑀦𑀧𑀺 𑀫𑀬𑀁 𑀢𑁂 𑀅𑀓𑀸𑀫𑀓𑀸 𑀯𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑀬𑀁 𑀢𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇

46. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀦 𑀫𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀅𑀦𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀺—  “𑀇𑀥𑁂𑀯 𑀫𑁂 𑀫𑀭𑀡𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀯𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀓𑀫𑁆𑀧𑀺 𑀪𑀢𑁆𑀢𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺, 𑀤𑁆𑀯𑁂𑀧𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺, 𑀢𑀻𑀡𑀺𑀧𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀧𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺, 𑀧𑀜𑁆𑀘𑀧𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺, 𑀙𑀧𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺, 𑀲𑀢𑁆𑀢𑀧𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺𑁇

(28.)

47. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄 𑀲𑀼𑀔𑀧𑀭𑀺𑀳𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀚𑀸𑀦𑀸𑀲𑀺𑁇 𑀫𑀭𑀡𑁂𑀦𑀧𑀺 𑀫𑀬𑀁 𑀢𑁂 𑀅𑀓𑀸𑀫𑀓𑀸 𑀯𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑀬𑀁 𑀢𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬? 𑀉𑀝𑁆𑀞𑁂𑀳𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀪𑀼𑀜𑁆𑀚 𑀘 𑀧𑀺𑀯 𑀘 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀘, 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀧𑀺𑀯𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀅𑀪𑀺𑀭𑀫𑀲𑁆𑀲𑀼𑁇 𑀦 𑀢𑀁 𑀫𑀬𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

48. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄 𑀲𑀼𑀔𑀧𑀭𑀺𑀳𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀚𑀸𑀦𑀸𑀲𑀺𑁇 𑀫𑀭𑀡𑁂𑀦𑀧𑀺 𑀫𑀬𑀁 𑀢𑁂 𑀅𑀓𑀸𑀫𑀓𑀸 𑀯𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑀬𑀁 𑀢𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑁇 𑀉𑀝𑁆𑀞𑁂𑀳𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀪𑀼𑀜𑁆𑀚 𑀘 𑀧𑀺𑀯 𑀘 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀘, 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀧𑀺𑀯𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀅𑀪𑀺𑀭𑀫𑀲𑁆𑀲𑀼𑁇 𑀦 𑀢𑀁 𑀫𑀬𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

49. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀸 𑀬𑁂𑀦 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄 𑀲𑀼𑀔𑀧𑀭𑀺𑀳𑀢𑁄𑁇 𑀦 𑀢𑁆𑀯𑀁, 𑀲𑀫𑁆𑀫 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀚𑀸𑀦𑀸𑀲𑀺𑁇 𑀫𑀭𑀡𑁂𑀦𑀧𑀺 𑀢𑁂 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀅𑀓𑀸𑀫𑀓𑀸 𑀯𑀺𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀓𑀺𑀁 𑀧𑀦 𑀢𑀁 𑀚𑀻𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀸𑀬𑁇 𑀉𑀝𑁆𑀞𑁂𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀪𑀼𑀜𑁆𑀚 𑀘 𑀧𑀺𑀯 𑀘 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀘, 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀧𑀺𑀯𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀅𑀪𑀺𑀭𑀫𑀲𑁆𑀲𑀼, 𑀦 𑀢𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀼𑀤𑀺𑀦𑁆𑀦…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

50. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀸 𑀬𑁂𑀦 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀫𑁆𑀫𑀢𑀸𑀢𑀸, 𑀏𑀲𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀅𑀦𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀦𑀺𑀧𑀦𑁆𑀦𑁄—  ‘𑀇𑀥𑁂𑀯 𑀫𑁂 𑀫𑀭𑀡𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀦𑀸𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀣 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀭𑀡𑀁 𑀆𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀘𑁂 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀣 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀫𑁆𑀧𑀺 𑀦𑀁 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀣𑁇 𑀲𑀘𑁂 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀦𑀸𑀪𑀺𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬, 𑀓𑀸 𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸 𑀕𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀇𑀥𑁂𑀯 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀣 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫, 𑀢𑀸𑀢𑀸, 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀸 𑀬𑁂𑀦 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀉𑀝𑁆𑀞𑁂𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄𑀲𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇

51. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄𑀫𑁆𑀳𑀺 𑀓𑀺𑀭 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺, 𑀳𑀝𑁆𑀞𑁄 𑀉𑀤𑀕𑁆𑀕𑁄 𑀧𑀸𑀡𑀺𑀦𑀸 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀧𑀼𑀜𑁆𑀙𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀓𑀢𑀺𑀧𑀸𑀳𑀁 𑀩𑀮𑀁 𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑁇 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼 𑀫𑀁 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀅𑀮𑀢𑁆𑀣 𑀔𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀢𑁆𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀅𑀘𑀺𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀏𑀯𑀭𑀽𑀧𑁂 𑀥𑀼𑀢𑀕𑀼𑀡𑁂 𑀲𑀫𑀸𑀤𑀸𑀬 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀲𑀧𑀤𑀸𑀦𑀘𑀸𑀭𑀺𑀓𑁄, 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀚𑁆𑀚𑀺𑀕𑀸𑀫𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺𑁇

52. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀚𑁆𑀚𑀻 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀤𑁆𑀯𑀻𑀳𑀺𑀢𑀺𑀓𑀸 𑀲𑁂𑀢𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀮𑀸𑀓𑀸𑀯𑀼𑀢𑁆𑀢𑀸, 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀉𑀜𑁆𑀙𑁂𑀦 𑀧𑀕𑁆𑀕𑀳𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀯𑀚𑁆𑀚𑀻 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀸 𑀤𑁆𑀯𑀻𑀳𑀺𑀢𑀺𑀓𑀸 𑀲𑁂𑀢𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀮𑀸𑀓𑀸𑀯𑀼𑀢𑁆𑀢𑀸, 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀉𑀜𑁆𑀙𑁂𑀦 𑀧𑀕𑁆𑀕𑀳𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀩𑀳𑀽 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀜𑀸𑀢𑀻 𑀅𑀟𑁆𑀠𑀸 𑀫𑀳𑀤𑁆𑀥𑀦𑀸 𑀫𑀳𑀸𑀪𑁄𑀕𑀸 𑀧𑀳𑀽𑀢𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀸 𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑀸 𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀜𑀸𑀢𑀻 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀁𑁇 𑀜𑀸𑀢𑀻 𑀫𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀸𑀦𑀺 𑀤𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀮𑀸𑀪𑀁 𑀮𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀳𑀜𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀦 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

53. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑁂𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀜𑀸𑀢𑀓𑀸—  “𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀺𑀭 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀝𑁆𑀞𑀺𑀫𑀢𑁆𑀢𑁂 𑀣𑀸𑀮𑀺𑀧𑀸𑀓𑁂 𑀪𑀢𑁆𑀢𑀸𑀪𑀺𑀳𑀸𑀭𑀁 𑀅𑀪𑀺𑀳𑀭𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑁂 𑀲𑀝𑁆𑀞𑀺𑀫𑀢𑁆𑀢𑁂 𑀣𑀸𑀮𑀺𑀧𑀸𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀓𑀮𑀦𑁆𑀤𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀓𑀮𑀦𑁆𑀤𑀕𑀸𑀫𑁂 𑀲𑀧𑀤𑀸𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀓𑀧𑀺𑀢𑀼 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇

54. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀜𑀸𑀢𑀺𑀤𑀸𑀲𑀻 𑀆𑀪𑀺𑀤𑁄𑀲𑀺𑀓𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁 𑀙𑀟𑁆𑀟𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑀁 𑀜𑀸𑀢𑀺𑀤𑀸𑀲𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀢𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀙𑀟𑁆𑀟𑀦𑀻𑀬𑀥𑀫𑁆𑀫𑀁, 𑀇𑀥 𑀫𑁂 𑀧𑀢𑁆𑀢𑁂 𑀆𑀓𑀺𑀭𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀜𑀸𑀢𑀺𑀤𑀸𑀲𑀻 𑀢𑀁 𑀆𑀪𑀺𑀤𑁄𑀲𑀺𑀓𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑁂 𑀆𑀓𑀺𑀭𑀦𑁆𑀢𑀻 𑀳𑀢𑁆𑀣𑀸𑀦𑀜𑁆𑀘 𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀭𑀲𑁆𑀲 𑀘 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀜𑀸𑀢𑀺𑀤𑀸𑀲𑀻 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀕𑁆𑀖𑁂𑀬𑁆𑀬𑁂, 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 “𑀲𑀘𑁂, 𑀚𑁂, 𑀢𑁆𑀯𑀁 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺, 𑀅𑀤𑀸𑀲𑀺𑀁 𑀢𑀁 𑀓𑀭𑁄𑀫𑀻”𑀢𑀺𑁇

55. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑀁 𑀆𑀪𑀺𑀤𑁄𑀲𑀺𑀓𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀓𑀼𑀝𑁆𑀝𑀫𑀽𑀮𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇 𑀧𑀺𑀢𑀸𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀢𑀁 𑀆𑀪𑀺𑀤𑁄𑀲𑀺𑀓𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀓𑀼𑀝𑁆𑀝𑀫𑀽𑀮𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀆𑀪𑀺𑀤𑁄𑀲𑀺𑀓𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑀦𑀼 𑀦𑀸𑀫, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀲𑀓𑀁 𑀕𑁂𑀳𑀁 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀅𑀕𑀫𑀺𑀫𑁆𑀳 𑀔𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀕𑁂𑀳𑀁𑁇 𑀢𑀢𑀸𑀬𑀁 𑀆𑀪𑀺𑀤𑁄𑀲𑀺𑀓𑁄 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀩𑀸𑀳𑀸𑀬𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀖𑀭𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀬𑁂𑀦 𑀲𑀓𑀧𑀺𑀢𑀼 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀼𑀜𑁆𑀚, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀓𑀢𑀁 𑀫𑁂 𑀅𑀚𑁆𑀚 𑀪𑀢𑁆𑀢𑀓𑀺𑀘𑁆𑀘𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀳𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

57. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀓𑀧𑀺𑀢𑀼 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀧𑀼𑀜𑁆𑀚𑁂 𑀯𑀺𑀯𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀢𑁂, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀫𑀸𑀢𑀼 𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀇𑀢𑁆𑀣𑀺𑀥𑀦𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀁 𑀅𑀜𑁆𑀜𑀁 𑀧𑀺𑀢𑀸𑀫𑀳𑀁𑁇 𑀮𑀩𑁆𑀪𑀸, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑀸 𑀘 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀸𑀢𑀼𑀁𑁇 𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀢𑀸𑀢, 𑀦 𑀉𑀲𑁆𑀲𑀳𑀸𑀫𑀺 𑀦 𑀯𑀺𑀲𑀳𑀸𑀫𑀺, 𑀅𑀪𑀺𑀭𑀢𑁄 𑀅𑀳𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀢𑁂, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀫𑀸𑀢𑀼 𑀫𑀢𑁆𑀢𑀺𑀓𑀁, 𑀇𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀇𑀢𑁆𑀣𑀺𑀥𑀦𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀧𑀺𑀢𑀸𑀫𑀳𑀁𑁇 𑀮𑀩𑁆𑀪𑀸, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑀸 𑀘 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀸𑀢𑀼𑀁𑁇 𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀫 𑀔𑁄 𑀢𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀘𑁂 𑀢𑁆𑀯𑀁 𑀦𑀸𑀢𑀺𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀯𑀤𑁂𑀳𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀫𑀳𑀦𑁆𑀢𑁂 𑀫𑀳𑀦𑁆𑀢𑁂 𑀲𑀸𑀡𑀺𑀧𑀲𑀺𑀩𑁆𑀩𑀓𑁂 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀳𑀺𑀭𑀜𑁆𑀜𑀲𑀼𑀯𑀡𑁆𑀡𑀲𑁆𑀲 𑀧𑀽𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑀝𑁂𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑁂 𑀑𑀧𑀸𑀢𑁂𑀳𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀬𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸 𑀆𑀭𑀓𑁆𑀔𑁄 𑀯𑀸 𑀲𑁄 𑀢𑁂 𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀳𑁄𑀲𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀻”𑀢𑀺𑁇

58. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀯𑀥𑀼, 𑀢𑁆𑀯𑀁 𑀧𑀺𑀬𑀸 𑀘 𑀫𑀦𑀸𑀧𑀸 𑀘𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑀼𑀬𑁆𑀳𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀸𑀤𑁂𑀲𑀼 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀻𑀤𑀺𑀲𑀸 𑀦𑀸𑀫 𑀢𑀸, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀅𑀘𑁆𑀙𑀭𑀸𑀬𑁄 𑀬𑀸𑀲𑀁 𑀢𑁆𑀯𑀁 𑀳𑁂𑀢𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀲𑀻”𑀢𑀺? “𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀅𑀘𑁆𑀙𑀭𑀸𑀦𑀁 𑀳𑁂𑀢𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸—  “𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀫𑀁 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀪𑀕𑀺𑀦𑀺𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀻”𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀧𑀧𑀢𑀸𑁇

59. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀧𑀺𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀪𑁄𑀚𑀦𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀁 𑀤𑁂𑀣, 𑀫𑀸 𑀦𑁄 𑀯𑀺𑀳𑁂𑀞𑀬𑀺𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀘 𑀧𑀺𑀢𑀸 𑀘 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀼𑀁 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀼𑀮𑀁 𑀅𑀟𑁆𑀠𑀁 𑀫𑀳𑀤𑁆𑀥𑀦𑀁 𑀫𑀳𑀸𑀪𑁄𑀕𑀁 𑀧𑀳𑀽𑀢𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑀁 𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀁𑁇 𑀮𑀩𑁆𑀪𑀸, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑀸 𑀘 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀸𑀢𑀼𑀁𑁇 𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀫𑁆𑀫, 𑀦 𑀉𑀲𑁆𑀲𑀳𑀸𑀫𑀺 𑀦 𑀯𑀺𑀲𑀳𑀸𑀫𑀺, 𑀅𑀪𑀺𑀭𑀢𑁄 𑀅𑀳𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀻”𑀢𑀺𑁇

61. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀯𑀥𑀼, 𑀬𑀤𑀸 𑀉𑀢𑀼𑀦𑀻 𑀅𑀳𑁄𑀲𑀺, 𑀧𑀼𑀧𑁆𑀨𑀁 𑀢𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀣 𑀫𑁂 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀬𑀸 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀉𑀢𑀼𑀦𑀻 𑀅𑀳𑁄𑀲𑀺, 𑀧𑀼𑀧𑁆𑀨𑀁𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀉𑀢𑀼𑀦𑀻𑀫𑁆𑀳𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀧𑀼𑀧𑁆𑀨𑀁 𑀫𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀯𑀥𑀼, 𑀬𑁂𑀦 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀮𑀗𑁆𑀓𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀺𑀬𑀸 𑀅𑀳𑁄𑀲𑀺 𑀫𑀦𑀸𑀧𑀸 𑀢𑁂𑀦 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑁂𑀦 𑀅𑀮𑀗𑁆𑀓𑀭𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀬𑀸 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀫𑀳𑀸𑀯𑀦𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀼𑀮𑀁 𑀅𑀟𑁆𑀠𑀁 𑀫𑀳𑀤𑁆𑀥𑀦𑀁 𑀫𑀳𑀸𑀪𑁄𑀕𑀁 𑀧𑀳𑀽𑀢𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑀁 𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀁𑁇 𑀮𑀩𑁆𑀪𑀸, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑀸 𑀘 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀸𑀢𑀼𑀁𑁇 𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀫𑁆𑀫, 𑀦 𑀉𑀲𑁆𑀲𑀳𑀸𑀫𑀺 𑀦 𑀯𑀺𑀲𑀳𑀸𑀫𑀺, 𑀅𑀪𑀺𑀭𑀢𑁄 𑀅𑀳𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀻”𑀢𑀺𑁇

62. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀓𑀼𑀮𑀁 𑀅𑀟𑁆𑀠𑀁 𑀫𑀳𑀤𑁆𑀥𑀦𑀁 𑀫𑀳𑀸𑀪𑁄𑀕𑀁 𑀧𑀳𑀽𑀢𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑀁 𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀁𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀲𑀼𑀤𑀺𑀦𑁆𑀦, 𑀩𑀻𑀚𑀓𑀫𑁆𑀧𑀺 𑀤𑁂𑀳𑀺—  𑀫𑀸 𑀦𑁄 𑀅𑀧𑀼𑀢𑁆𑀢𑀓𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀬𑁄 𑀅𑀢𑀺𑀳𑀭𑀸𑀧𑁂𑀲𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀢𑀁 𑀔𑁄 𑀫𑁂, 𑀅𑀫𑁆𑀫, 𑀲𑀓𑁆𑀓𑀸 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀩𑀸𑀳𑀸𑀬𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂 𑀅𑀦𑀸𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑁄 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀪𑀺𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺𑁇 𑀲𑀸 𑀢𑁂𑀦 𑀕𑀩𑁆𑀪𑀁 𑀕𑀡𑁆𑀳𑀺𑁇

63. 𑀪𑀼𑀫𑁆𑀫𑀸 𑀤𑁂𑀯𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁—  “𑀦𑀺𑀭𑀩𑁆𑀩𑀼𑀤𑁄 𑀯𑀢 𑀪𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀦𑀺𑀭𑀸𑀤𑀻𑀦𑀯𑁄; 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁂𑀦 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑁆𑀩𑀼𑀤𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀢𑀁, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀢𑁄”𑀢𑀺𑁇 𑀪𑀼𑀫𑁆𑀫𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀤𑁂𑀯𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀤𑁂𑀯𑀸…  𑀬𑀸𑀫𑀸 𑀤𑁂𑀯𑀸…  𑀢𑀼𑀲𑀺𑀢𑀸 𑀤𑁂𑀯𑀸…  𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻 𑀤𑁂𑀯𑀸…  𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀤𑁂𑀯𑀸…  𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀸 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀼𑀁—  “𑀦𑀺𑀭𑀩𑁆𑀩𑀼𑀤𑁄 𑀯𑀢 𑀪𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀦𑀺𑀭𑀸𑀤𑀻𑀦𑀯𑁄; 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁂𑀦 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑁆𑀩𑀼𑀤𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀢𑀁, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀢𑁄”𑀢𑀺𑁇 𑀇𑀢𑀺𑀳 𑀢𑁂𑀦 𑀔𑀡𑁂𑀦 𑀢𑁂𑀦 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑁂𑀦 𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀸 𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀺𑁇

64. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀢𑀲𑁆𑀲 𑀕𑀩𑁆𑀪𑀲𑁆𑀲 𑀧𑀭𑀺𑀧𑀸𑀓𑀫𑀦𑁆𑀯𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀬𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀸 𑀢𑀲𑁆𑀲 𑀤𑀸𑀭𑀓𑀲𑁆𑀲 𑀩𑀻𑀚𑀓𑁄𑀢𑀺 𑀦𑀸𑀫𑀁 𑀅𑀓𑀁𑀲𑀼𑁇 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀩𑀻𑀚𑀓𑀫𑀸𑀢𑀸𑀢𑀺 𑀦𑀸𑀫𑀁 𑀅𑀓𑀁𑀲𑀼𑁇 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀩𑀻𑀚𑀓𑀧𑀺𑀢𑀸𑀢𑀺 𑀦𑀸𑀫𑀁 𑀅𑀓𑀁𑀲𑀼𑁇 𑀢𑁂 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀪𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀸𑀓𑀁𑀲𑀼𑁇

(37.)

65. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀳𑀼𑀤𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁, 𑀅𑀳𑀼 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄—  “𑀅𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀮𑀸𑀪𑀸; 𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀲𑀼𑀮𑀤𑁆𑀥𑀁; 𑀬𑁄𑀳𑀁 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑁂𑀦 𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁂𑀦 𑀓𑀺𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀮𑀽𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁄 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑁄𑀫𑀦𑁄 𑀮𑀻𑀦𑀫𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀧𑀚𑁆𑀛𑀸𑀬𑀺𑁇

(38.)

71. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀩𑁆𑀪𑀭𑀢𑀸𑀬 𑀤𑀼𑀧𑁆𑀧𑁄𑀲𑀢𑀸𑀬 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀬 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀼𑀪𑀭𑀢𑀸𑀬 𑀲𑀼𑀧𑁄𑀲𑀢𑀸𑀬 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑁆𑀲 𑀥𑀼𑀢𑀲𑁆𑀲 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑁆𑀲 𑀅𑀧𑀘𑀬𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀤𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁 𑀢𑀤𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬, 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑀢𑀸𑀬, 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬, 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

72. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑁂𑀬𑁆𑀬, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑁄”**𑀢𑀺𑁇

73. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

74. 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
