# 𑁧𑁇𑁧𑁇𑁧𑁇𑁪 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

140. _𑀫𑀓𑁆𑀓𑀝𑀻 𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀕𑀺𑀳𑀻 𑀦𑀕𑁆𑀕𑁄 𑀘 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸;_  
_𑀤𑀸𑀭𑀺𑀓𑀼𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀳𑀺𑀧𑀭𑁂 𑀤𑀼𑀯𑁂𑁇_  


141. _𑀫𑀸𑀢𑀸 𑀥𑀻𑀢𑀸 𑀪𑀕𑀺𑀦𑀻 𑀘,_  
_𑀚𑀸𑀬𑀸 𑀘 𑀫𑀼𑀤𑀼 𑀮𑀫𑁆𑀩𑀺𑀦𑀸;_  
_𑀤𑁆𑀯𑁂 𑀯𑀡𑀸 𑀮𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀤𑀸𑀭𑀼𑀥𑀻𑀢𑀮𑀺𑀓𑀸𑀬 𑀘𑁇_  


142. _𑀲𑀼𑀦𑁆𑀤𑀭𑁂𑀦 𑀲𑀳 𑀧𑀜𑁆𑀘,_  
_𑀧𑀜𑁆𑀘 𑀲𑀺𑀯𑀣𑀺𑀓𑀝𑁆𑀞𑀺𑀓𑀸;_  
_𑀦𑀸𑀕𑀻 𑀬𑀓𑁆𑀔𑀻 𑀘 𑀧𑁂𑀢𑀻 𑀘,_  
_𑀧𑀡𑁆𑀟𑀓𑁄𑀧𑀳𑀢𑁄 𑀙𑀼𑀧𑁂𑁇_  


143. _𑀪𑀤𑁆𑀤𑀺𑀬𑁂 𑀅𑀭𑀳𑀁 𑀲𑀼𑀢𑁆𑀢𑁄,_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀘𑀢𑀼𑀭𑁄 𑀧𑀭𑁂;_  
_𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀸 𑀢𑀬𑁄 𑀫𑀸𑀮𑀸,_  
_𑀲𑀼𑀧𑀺𑀦𑁂 𑀪𑀸𑀭𑀼𑀓𑀘𑁆𑀙𑀓𑁄𑁇_  


144. _𑀲𑀼𑀧𑀩𑁆𑀩𑀸 𑀲𑀤𑁆𑀥𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻,_  
_𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀘;_  
_𑀯𑁂𑀲𑀺𑀬𑀸 𑀧𑀡𑁆𑀟𑀓𑁄 𑀕𑀺𑀳𑀻,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀫𑀺𑀕𑁄𑀢𑀺𑁇_
