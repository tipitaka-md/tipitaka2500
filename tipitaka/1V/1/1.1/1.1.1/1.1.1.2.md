# 𑁧𑁇𑁧𑁇𑁧𑁇𑁨 𑀫𑀓𑁆𑀓𑀝𑀻𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀓𑁆𑀓𑀝𑀻𑀯𑀢𑁆𑀣𑀼

(40.)

75\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀫𑀳𑀸𑀯𑀦𑁂 𑀫𑀓𑁆𑀓𑀝𑀺𑀁 𑀆𑀫𑀺𑀲𑁂𑀦 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑁂𑀦𑀸𑀲𑀦𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀤𑁆𑀤𑀲 𑀔𑁄 𑀲𑀸 𑀫𑀓𑁆𑀓𑀝𑀻 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀼𑀭𑀢𑁄 𑀓𑀝𑀺𑀫𑁆𑀧𑀺 𑀘𑀸𑀮𑁂𑀲𑀺 𑀙𑁂𑀧𑁆𑀧𑀫𑁆𑀧𑀺 𑀘𑀸𑀮𑁂𑀲𑀺, 𑀓𑀝𑀺𑀫𑁆𑀧𑀺 𑀑𑀟𑁆𑀟𑀺, 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀅𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀻”𑀢𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀮𑀻𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀆𑀤𑀸𑀬 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀺𑁇

(41.)

76\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀫𑀓𑁆𑀓𑀝𑀻 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀏𑀓𑀤𑁂𑀲𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀤𑁂𑀲𑀁 𑀢𑀲𑁆𑀲𑀸 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀅𑀤𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀫𑀓𑁆𑀓𑀝𑀻 𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀝𑀺𑀁 𑀑𑀟𑁆𑀟𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲𑀸 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺𑁇

77\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀦𑀦𑀼, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁; 𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁; 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀺𑀬𑀸, 𑀦𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀦𑀦𑀼, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀣𑁂𑀯 𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀲𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀦𑀦𑀼, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄, 𑀦𑁄 𑀲𑀭𑀸𑀕𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬𑁇 𑀅𑀣 𑀔𑁆𑀯𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

(42.)

78\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀲𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄, 𑀦𑁄 𑀲𑀭𑀸𑀕𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀯𑀭𑀁 𑀢𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀆𑀲𑀺𑀯𑀺𑀲𑀲𑁆𑀲 𑀖𑁄𑀭𑀯𑀺𑀲𑀲𑁆𑀲 𑀫𑀼𑀔𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀯𑀭𑀁 𑀢𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑀲𑁆𑀲 𑀫𑀼𑀔𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀯𑀭𑀁 𑀢𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀼𑀬𑀸 𑀆𑀤𑀺𑀢𑁆𑀢𑀸𑀬 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀸𑀬 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀸𑀬 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀜𑁆𑀳𑀺, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀭𑀡𑀁 𑀯𑀸 𑀦𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁; 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀇𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀜𑁆𑀘 𑀔𑁄, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀕𑀸𑀫𑀥𑀫𑁆𑀫𑀁 𑀯𑀲𑀮𑀥𑀫𑁆𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀑𑀤𑀓𑀦𑁆𑀢𑀺𑀓𑀁 𑀭𑀳𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑀬𑀁𑀤𑁆𑀯𑀬𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

79\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑁂𑀬𑁆𑀬 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀬𑀧𑀺, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑁄”**𑀢𑀺𑁇

80\. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

---
