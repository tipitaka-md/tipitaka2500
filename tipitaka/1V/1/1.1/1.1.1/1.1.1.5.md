# 𑁧𑁇𑁧𑁇𑁧𑁇𑁫 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

(67.)

145\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧)

146\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸 𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀺𑀓𑁆𑀔𑀁 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀅𑀦𑀸𑀯𑀺𑀓𑀢𑁆𑀯𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨)

147\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀏𑀯𑀁 𑀫𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀕𑀺𑀳𑀺𑀮𑀺𑀗𑁆𑀕𑁂𑀦 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩)

148\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀏𑀯𑀁 𑀫𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀦𑀕𑁆𑀕𑁄 𑀳𑀼𑀢𑁆𑀯𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁪)

149\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀏𑀯𑀁 𑀫𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀓𑀼𑀲𑀘𑀻𑀭𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…  𑀯𑀸𑀓𑀘𑀻𑀭𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…  𑀨𑀮𑀓𑀘𑀻𑀭𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…  𑀓𑁂𑀲𑀓𑀫𑁆𑀩𑀮𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…  𑀯𑀸𑀮𑀓𑀫𑁆𑀩𑀮𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…  𑀉𑀮𑀽𑀓𑀧𑀓𑁆𑀔𑀺𑀓𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…  𑀅𑀚𑀺𑀦𑀓𑁆𑀔𑀺𑀧𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁫--𑁧𑁧)

150\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀞𑀓𑁂 𑀦𑀺𑀧𑀦𑁆𑀦𑀁 𑀤𑀸𑀭𑀺𑀓𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑀁 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀯𑁂𑀲𑁂𑀲𑀺𑁇 𑀲𑀸 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁨)

(68.)

151\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀝𑁆𑀞𑀸𑀬 𑀓𑀼𑀝𑀺𑀓𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀮𑀻𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑁇 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀓𑀼𑀝𑀺𑀓𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀜𑁆𑀘𑀓𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀽𑀲𑁂𑀲𑀺𑁇 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇 (𑁧𑁩)

(69.)

152\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀢𑁆𑀣𑀺𑀮𑀺𑀗𑁆𑀕𑀁 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀜𑁆𑀜𑁂𑀯 𑀉𑀧𑀚𑁆𑀛𑀁 𑀢𑀫𑁂𑀯 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀢𑀸𑀦𑀺𑀬𑁂𑀯 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀗𑁆𑀕𑀫𑀺𑀢𑀼𑀁𑁇 𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀼𑀁𑁇 𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀢𑀸𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀻𑀳𑀺 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻”𑀢𑀺𑁇 (𑁧𑁪)

153\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀮𑀺𑀗𑁆𑀕𑀁 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀜𑁆𑀜𑁂𑀯 𑀉𑀧𑀚𑁆𑀛𑀁 𑀢𑀫𑁂𑀯 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀢𑀸𑀦𑀺𑀬𑁂𑀯 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀫𑀺𑀢𑀼𑀁𑁇 𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀼𑀁𑁇 𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀢𑀸𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀻𑀳𑀺 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻”𑀢𑀺𑁇 (𑁧𑁫)

(70.)

154\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀏𑀯𑀁 𑀫𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀫𑀸𑀢𑀼𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺…  𑀥𑀻𑀢𑀼𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺…  𑀪𑀕𑀺𑀦𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺…  𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁬--𑁧𑁮)

155\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁯)

(71.)

156\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀼𑀤𑀼𑀧𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀬𑀸 𑀧𑀻𑀍𑀇𑀢𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀫𑀼𑀔𑁂𑀦 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁦)

157\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀮𑀫𑁆𑀩𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀬𑀸 𑀧𑀻𑀍𑀇𑀢𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀯𑀘𑁆𑀘𑀫𑀕𑁆𑀕𑀁 𑀧𑀯𑁂𑀲𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁧)

158\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀢𑀲𑀭𑀻𑀭𑀁 𑀧𑀲𑁆𑀲𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀲𑀭𑀻𑀭𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀯𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄—  “𑀏𑀯𑀁 𑀫𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑁆𑀯𑀸 𑀯𑀡𑁂𑀦 𑀦𑀻𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁨)

159\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀢𑀲𑀭𑀻𑀭𑀁 𑀧𑀲𑁆𑀲𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀲𑀭𑀻𑀭𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀯𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄—  “𑀏𑀯𑀁 𑀫𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀯𑀡𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂𑀦 𑀦𑀻𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁩)

160\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀮𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂𑀦 𑀙𑀼𑀧𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁪)

161\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀤𑀸𑀭𑀼𑀥𑀻𑀢𑀮𑀺𑀓𑀸𑀬 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂𑀦 𑀙𑀼𑀧𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁫)

(72.)

162\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑁄 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀸𑀚𑀕𑀳𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀭𑀣𑀺𑀓𑀸𑀬 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻—  “𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀯𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀲𑀸 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀻 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀔𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀲𑀸𑀤𑀺𑀬𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀤𑀺𑀬𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁬)

163\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀳𑀁 𑀯𑀸𑀬𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑁆𑀯𑀁 𑀫𑀸 𑀯𑀸𑀬𑀫𑀺, 𑀏𑀯𑀁 𑀢𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸 𑀅𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁭)

164\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑁆𑀯𑀁 𑀯𑀸𑀬𑀫, 𑀅𑀳𑀁 𑀦 𑀯𑀸𑀬𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀏𑀯𑀁 𑀢𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸 𑀅𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁮)

165\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀁 𑀖𑀝𑁆𑀝𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺 𑀫𑁄𑀘𑁂𑀳𑀺…𑀧𑁂…  𑀩𑀳𑀺 𑀖𑀝𑁆𑀝𑁂𑀢𑁆𑀯𑀸 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀁 𑀫𑁄𑀘𑁂𑀳𑀺, 𑀏𑀯𑀁 𑀢𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸 𑀅𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁯--𑁩𑁦)

(73.)

166\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀓𑁆𑀔𑀸𑀬𑀺𑀢𑀁 𑀲𑀭𑀻𑀭𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁧)

167\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀅𑀓𑁆𑀔𑀸𑀬𑀺𑀢𑀁 𑀲𑀭𑀻𑀭𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁨)

168\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀔𑀸𑀬𑀺𑀢𑀁 𑀲𑀭𑀻𑀭𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁩)

169\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀙𑀺𑀦𑁆𑀦𑀲𑀻𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀝𑁆𑀝𑀓𑀢𑁂 𑀫𑀼𑀔𑁂 𑀙𑀼𑀧𑀦𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀯𑁂𑀲𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁪)

170\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀙𑀺𑀦𑁆𑀦𑀲𑀻𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀝𑁆𑀝𑀓𑀢𑁂 𑀫𑀼𑀔𑁂 𑀅𑀘𑁆𑀙𑀼𑀧𑀦𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀯𑁂𑀲𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁫)

171\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀲𑀼𑀲𑀸𑀦𑁂 𑀙𑀟𑁆𑀟𑀺𑀢𑀸𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀺𑀡𑁆𑀡𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀯𑀣𑀺𑀓𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀓𑀸𑀦𑀺 𑀲𑀗𑁆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁬)

172\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀕𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺…  𑀬𑀓𑁆𑀔𑀺𑀦𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺…  𑀧𑁂𑀢𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺…  𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁭--𑁪𑁦)

173\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑀳𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄—  “𑀦𑀸𑀳𑀁 𑀯𑁂𑀤𑀺𑀬𑀸𑀫𑀺 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑁇 …𑀧𑁂…  𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀯𑁂𑀤𑀺𑀬𑀺 𑀯𑀸 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀯𑀸 𑀯𑁂𑀤𑀺𑀬𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪𑁧)

174\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺—  𑀙𑀼𑀧𑀺𑀢𑀫𑀢𑁆𑀢𑁂 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪𑁨)

(74.)

175\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀤𑁆𑀤𑀺𑀬𑁂 𑀚𑀸𑀢𑀺𑀬𑀸𑀯𑀦𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀯𑀸𑀢𑀽𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀺𑀮𑀺𑀦𑁆𑀦𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁 𑀳𑁄𑀢𑀺—  𑀭𑀸𑀕𑁂𑀦, 𑀯𑀘𑁆𑀘𑁂𑀦, 𑀧𑀲𑁆𑀲𑀸𑀯𑁂𑀦, 𑀯𑀸𑀢𑁂𑀦, 𑀉𑀘𑁆𑀘𑀸𑀮𑀺𑀗𑁆𑀕𑀧𑀸𑀡𑀓𑀤𑀝𑁆𑀞𑁂𑀦𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲𑁇 𑀅𑀭𑀳𑀁 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄”𑀢𑀺𑁇 (𑁪𑁩)

176\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀅𑀦𑁆𑀥𑀯𑀦𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀕𑁄𑀧𑀸𑀮𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀯𑁂𑀲𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀧𑀯𑀺𑀝𑁆𑀞𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀞𑀺𑀢𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀉𑀤𑁆𑀥𑀭𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁪𑁪)

177\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀅𑀦𑁆𑀥𑀯𑀦𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀅𑀚𑀧𑀸𑀮𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸…  𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀓𑀝𑁆𑀞𑀳𑀸𑀭𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸…  𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀕𑁄𑀫𑀬𑀳𑀸𑀭𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀯𑁂𑀲𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀧𑀯𑀺𑀝𑁆𑀞𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀞𑀺𑀢𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀉𑀤𑁆𑀥𑀭𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁪𑁫--𑁪𑁭)

(75.)

178\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀫𑀳𑀸𑀯𑀦𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀳𑀲𑀫𑀸𑀦𑀸 𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀼𑀬𑁆𑀳𑀺𑀤𑀁 𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀫𑀬𑁆𑀳𑀁 𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀲𑀸𑀤𑀺𑀬𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪𑁮)

(76.)

179\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀫𑀳𑀸𑀯𑀦𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀁 𑀅𑀧𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀳𑀲𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀲𑀸𑀤𑀺𑀬𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀤𑀺𑀬𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪𑁯)

180\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀫𑀳𑀸𑀯𑀦𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀁 𑀅𑀧𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀯𑀢𑁆𑀢𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀲𑀸𑀤𑀺𑀬𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀤𑀺𑀬𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁦)

(77.)

181\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀤𑁆𑀯𑀸𑀭𑀁 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀯𑀸𑀢𑀽𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀕𑀦𑁆𑀥𑀜𑁆𑀘 𑀫𑀸𑀮𑀜𑁆𑀘 𑀆𑀤𑀸𑀬 𑀆𑀭𑀸𑀫𑀁 𑀆𑀕𑀫𑀁𑀲𑀼 𑀯𑀺𑀳𑀸𑀭𑀧𑁂𑀓𑁆𑀔𑀺𑀓𑀸𑀬𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀓𑀢𑁆𑀯𑀸, 𑀧𑀼𑀭𑀺𑀲𑀽𑀲𑀪𑁄 𑀯𑀢𑀸𑀬𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀕𑀦𑁆𑀥𑀜𑁆𑀘 𑀫𑀸𑀮𑀜𑁆𑀘 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀺𑀮𑀺𑀦𑁆𑀦𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁 𑀳𑁄𑀢𑀺—  𑀭𑀸𑀕𑁂𑀦, 𑀯𑀘𑁆𑀘𑁂𑀦, 𑀧𑀲𑁆𑀲𑀸𑀯𑁂𑀦, 𑀯𑀸𑀢𑁂𑀦, 𑀉𑀘𑁆𑀘𑀸𑀮𑀺𑀗𑁆𑀕𑀧𑀸𑀡𑀓𑀤𑀝𑁆𑀞𑁂𑀦𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄, 𑀬𑀁 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲𑁇 𑀅𑀭𑀳𑀁 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀯𑀸 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀦𑁆𑀢𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑀁 𑀲𑀁𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 (𑁫𑁧)

(78.)

182\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀸𑀭𑀼𑀓𑀘𑁆𑀙𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀅𑀳𑀁, 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺, 𑀪𑀸𑀭𑀼𑀓𑀘𑁆𑀙𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀮𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀏𑀯𑀫𑀸𑀳—  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦𑀸”𑀢𑀺𑁇 (𑁫𑁨)

183\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑀼𑀧𑀩𑁆𑀩𑀸 𑀦𑀸𑀫 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀫𑀼𑀥𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀳𑁄𑀢𑀺—  “𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀢𑀺 𑀲𑀸 𑀅𑀕𑁆𑀕𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀊𑀭𑀼𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀬 𑀖𑀝𑁆𑀝𑁂𑀳𑀺, 𑀏𑀯𑀁 𑀢𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀪𑀺𑀬𑀁 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀤𑀭𑀯𑀝𑁆𑀝𑀺𑀬𑀁 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀓𑀘𑁆𑀙𑀓𑁂 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀻𑀯𑀸𑀬𑀁 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀡𑁆𑀡𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑁂 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁂𑀲𑀯𑀝𑁆𑀝𑀺𑀬𑀁 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑀼𑀮𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀬 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀏𑀯𑀁 𑀢𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸 𑀅𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁩--𑁬𑁧)

(79.)

184\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀲𑀤𑁆𑀥𑀸 𑀦𑀸𑀫 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀫𑀼𑀥𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀳𑁄𑀢𑀺—  “𑀬𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀢𑀺 𑀲𑀸 𑀅𑀕𑁆𑀕𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀊𑀭𑀼𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀬 𑀖𑀝𑁆𑀝𑁂𑀳𑀺…𑀧𑁂…  𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀏𑀯𑀁 𑀢𑁂 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸 𑀅𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁨--𑁭𑁦)

(80.)

185\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀓𑀼𑀫𑀸𑀭𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀼𑀁…  𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀼𑀁…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀉𑀪𑁄 𑀲𑀸𑀤𑀺𑀬𑀺𑀁𑀲𑀼𑁇 𑀉𑀪𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀉𑀪𑁄 𑀦 𑀲𑀸𑀤𑀺𑀬𑀺𑀁𑀲𑀼𑁇 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 (𑁭𑁧--𑁭𑁬)

(81.)

186\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀓𑀼𑀫𑀸𑀭𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀲𑀺𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀼𑀁…  𑀧𑀡𑁆𑀟𑀓𑁂 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀼𑀁…  𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀤𑀺𑀬𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀸𑀤𑀺𑀬𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 (𑁭𑁭--𑁮𑁨)

187\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀓𑀼𑀫𑀸𑀭𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀉𑀪𑁄 𑀲𑀸𑀤𑀺𑀬𑀺𑀁𑀲𑀼𑁇 𑀉𑀪𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀉𑀪𑁄 𑀦 𑀲𑀸𑀤𑀺𑀬𑀺𑀁𑀲𑀼𑁇 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 (𑁮𑁩--𑁮𑁪)

(82.)

188\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀲𑀸—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀩𑁆𑀪𑀫𑀸”𑀢𑀺 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁄 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀧𑀭𑀺𑀧𑀢𑀺𑁇 𑀲𑀸 𑀉𑀩𑁆𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀲𑀸𑀤𑀺𑀬𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀤𑀺𑀬𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁫)

(83.)

189\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀫𑀺𑀕𑀧𑁄𑀢𑀓𑁄 𑀢𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀸𑀯𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀧𑀺𑀯𑀦𑁆𑀢𑁄 𑀫𑀼𑀔𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀤𑀺𑀬𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁮𑁬)
