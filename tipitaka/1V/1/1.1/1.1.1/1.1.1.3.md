# 𑁧𑁇𑁧𑁇𑁧𑁇𑁩 𑀲𑀦𑁆𑀣𑀢𑀪𑀸𑀡𑀯𑀸𑀭

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀲𑀦𑁆𑀣𑀢𑀪𑀸𑀡𑀯𑀸𑀭

(43.)

82. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸 𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼, 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧𑀺𑀁𑀲𑀼, 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀁 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀅𑀦𑀸𑀯𑀺𑀓𑀢𑁆𑀯𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦𑀧𑀺 𑀨𑀼𑀝𑁆𑀞𑀸 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦𑀧𑀺 𑀨𑀼𑀝𑁆𑀞𑀸 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦𑀧𑀺 𑀨𑀼𑀝𑁆𑀞𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀼𑀤𑁆𑀥𑀕𑀭𑀳𑀺𑀦𑁄 𑀦 𑀥𑀫𑁆𑀫𑀕𑀭𑀳𑀺𑀦𑁄 𑀦 𑀲𑀁𑀖𑀕𑀭𑀳𑀺𑀦𑁄; 𑀅𑀢𑁆𑀢𑀕𑀭𑀳𑀺𑀦𑁄 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑀜𑁆𑀜𑀕𑀭𑀳𑀺𑀦𑁄𑁇 𑀫𑀬𑀫𑁂𑀯𑀫𑁆𑀳𑀸 𑀅𑀮𑀓𑁆𑀔𑀺𑀓𑀸 𑀫𑀬𑀁 𑀅𑀧𑁆𑀧𑀧𑀼𑀜𑁆𑀜𑀸, 𑀬𑁂 𑀫𑀬𑀁 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀫𑁆𑀳𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀇𑀤𑀸𑀦𑀺 𑀘𑁂𑀧𑀺 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁, 𑀇𑀤𑀸𑀦𑀺𑀧𑀺 𑀫𑀬𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑀸 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀭𑀢𑁆𑀢𑀸𑀧𑀭𑀭𑀢𑁆𑀢𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀪𑀸𑀯𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸𑀦𑀁 𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇

83. “𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀚𑁆𑀚𑀻𑀦𑀁 𑀯𑀸 𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀯𑀸 𑀓𑀸𑀭𑀡𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀲𑀫𑀽𑀳𑀦𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

84. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀁 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀅𑀦𑀸𑀯𑀺𑀓𑀢𑁆𑀯𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺 𑀲𑁄 𑀆𑀕𑀢𑁄 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀆𑀯𑀺𑀓𑀢𑁆𑀯𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺 𑀲𑁄 𑀆𑀕𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(44.)

85. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀸𑀚𑀻𑀯𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀲𑀺𑀓𑁆𑀔𑀁 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀅𑀦𑀸𑀯𑀺𑀓𑀢𑁆𑀯𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀬𑀧𑀺, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑁄”**𑀢𑀺𑁇 (*𑁧*:**𑁧**)

(45.)

86. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄 𑀬𑀣𑀸𑀬𑀼𑀢𑁆𑀢𑁄 𑀬𑀣𑀸𑀚𑀘𑁆𑀘𑁄 𑀬𑀣𑀸𑀦𑀸𑀫𑁄 𑀬𑀣𑀸𑀕𑁄𑀢𑁆𑀢𑁄 𑀬𑀣𑀸𑀲𑀻𑀮𑁄 𑀬𑀣𑀸𑀯𑀺𑀳𑀸𑀭𑀻 𑀬𑀣𑀸𑀕𑁄𑀘𑀭𑁄 𑀣𑁂𑀭𑁄 𑀯𑀸 𑀦𑀯𑁄 𑀯𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑁄 𑀯𑀸𑁇 𑀏𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀬𑁄 𑀧𑀦𑀸”𑀢𑀺𑁇

87. **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀓𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀺𑀓𑁆𑀔𑀸𑀘𑀭𑀺𑀬𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀕𑀢𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀺𑀦𑁆𑀦𑀧𑀝𑀥𑀭𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀫𑀜𑁆𑀜𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀢𑀻𑀳𑀺 𑀲𑀭𑀡𑀕𑀫𑀦𑁂𑀳𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀪𑀤𑁆𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀸𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑁂𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀢𑀢𑁆𑀭 𑀬𑁆𑀯𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

88. **𑀲𑀺𑀓𑁆𑀔𑀸**𑀢𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀺𑀓𑁆𑀔𑀸—  𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸, 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸, 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑁇 𑀢𑀢𑁆𑀭 𑀬𑀸𑀬𑀁 𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸, 𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀢𑀺𑁇

89. **𑀲𑀸𑀚𑀻𑀯𑀁** 𑀦𑀸𑀫 𑀬𑀁 𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁, 𑀏𑀢𑀁 𑀲𑀸𑀚𑀻𑀯𑀁 𑀦𑀸𑀫𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀸𑀚𑀻𑀯𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺𑁇

90. **𑀲𑀺𑀓𑁆𑀔𑀁 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀅𑀦𑀸𑀯𑀺𑀓𑀢𑁆𑀯𑀸**𑀢𑀺 𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸; 𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸𑁇

91. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀸 𑀘𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀁 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑁄 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑁄 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀕𑀺𑀳𑀺𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 𑀉𑀧𑀸𑀲𑀓𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 𑀆𑀭𑀸𑀫𑀺𑀓𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 𑀅𑀲𑁆𑀲𑀫𑀡𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸𑁇

92. 𑀅𑀣 𑀯𑀸 𑀧𑀦 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀸 𑀘𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀁 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑁄 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑁄 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀕𑀺𑀳𑀺𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…𑀧𑁂…  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀺𑀓𑁆𑀔𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀯𑀺𑀦𑀬𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀆𑀘𑀭𑀺𑀬𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀫𑀸𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀫𑀸𑀦𑀸𑀘𑀭𑀺𑀬𑀓𑀁…  𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀕𑀺𑀳𑀻 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺…  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺…  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺…  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺…  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺…  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺…  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸𑁇

94. 𑀅𑀣 𑀯𑀸 𑀧𑀦 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀸 𑀘𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀁 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑁄 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑁄 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀕𑀺𑀳𑀺𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄 “𑀫𑀸𑀢𑀭𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀧𑀺𑀢𑀭𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀪𑀸𑀢𑀭𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀪𑀕𑀺𑀦𑀺𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀥𑀻𑀢𑀭𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀜𑀸𑀢𑀓𑁂 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀫𑀺𑀢𑁆𑀢𑁂 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀕𑀸𑀫𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀦𑀺𑀕𑀫𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀔𑁂𑀢𑁆𑀢𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀯𑀢𑁆𑀣𑀼𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀧𑀼𑀩𑁆𑀩𑁂 𑀳𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀓𑀻𑀍𑀇𑀢𑀁 𑀲𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸𑁇

95. 𑀅𑀣 𑀯𑀸 𑀧𑀦 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀸 𑀘𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀁 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑁄 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑁄 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀕𑀺𑀳𑀺𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄—  “𑀫𑀸𑀢𑀸 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀲𑀸 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀧𑀺𑀢𑀸 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀲𑁄 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀪𑀸𑀢𑀸 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀲𑁄 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀪𑀕𑀺𑀦𑀻 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀲𑀸 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀲𑁄 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀥𑀻𑀢𑀸 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀲𑀸 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀧𑀚𑀸𑀧𑀢𑀺 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀲𑀸 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀜𑀸𑀢𑀓𑀸 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀢𑁂 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺…  “𑀫𑀺𑀢𑁆𑀢𑀸 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺, 𑀢𑁂 𑀫𑀬𑀸 𑀧𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀅𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸𑁇

98. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀲𑀸𑀫𑀜𑁆𑀜𑀸 𑀘𑀯𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀪𑀸𑀯𑀁 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑁄 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑁄 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀕𑀺𑀳𑀺𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄…𑀧𑁂…  𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀪𑀸𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀫𑀸𑀦𑁄—  “𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀫𑀻”𑀢𑀺 𑀯𑀤𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀸𑁇

103. **𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑁄** 𑀦𑀸𑀫 𑀬𑁄 𑀲𑁄 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄 𑀕𑀸𑀫𑀥𑀫𑁆𑀫𑁄 𑀯𑀲𑀮𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀑𑀤𑀓𑀦𑁆𑀢𑀺𑀓𑀁 𑀭𑀳𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑀬𑀁𑀤𑁆𑀯𑀬𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺, 𑀏𑀲𑁄 𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀫𑁇

104. **𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺** 𑀦𑀸𑀫 𑀬𑁄 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀺𑀮𑀨𑀮𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀧𑀯𑁂𑀲𑁂𑀢𑀺, 𑀏𑀲𑁄 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺 𑀦𑀸𑀫𑁇

105. **𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀬𑀧𑀻**𑀢𑀺 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀧𑀺 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄, 𑀧𑀕𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  “𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀬𑀧𑀻”𑀢𑀺𑁇

106. **𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀻𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑁂𑀦 𑀲𑀭𑀻𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀼𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  “𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

107. **𑀅𑀲𑀁𑀯𑀸𑀲𑁄**𑀢𑀺 𑀲𑀁𑀯𑀸𑀲𑁄 𑀦𑀸𑀫 𑀏𑀓𑀓𑀫𑁆𑀫𑀁 𑀏𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀢𑀸—  𑀏𑀲𑁄 𑀲𑀁𑀯𑀸𑀲𑁄 𑀦𑀸𑀫𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  “𑀅𑀲𑀁𑀯𑀸𑀲𑁄”𑀢𑀺𑁇

(56.)

108. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀢𑁆𑀣𑀺𑀬𑁄—  𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀻, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀻, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀺𑀢𑁆𑀣𑀻𑁇 𑀢𑀬𑁄 𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀸—  𑀫𑀦𑀼𑀲𑁆𑀲𑀼𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑁄, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀼𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑁄, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀼𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑁄𑁇 𑀢𑀬𑁄 𑀧𑀡𑁆𑀟𑀓𑀸—  𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀡𑁆𑀟𑀓𑁄, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀡𑁆𑀟𑀓𑁄, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀧𑀡𑁆𑀟𑀓𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀭𑀺𑀲𑀸—  𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑁄, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑁄, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀧𑀼𑀭𑀺𑀲𑁄𑁇

109. 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑀬𑁄 𑀫𑀕𑁆𑀕𑁂 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲—  𑀯𑀘𑁆𑀘𑀫𑀕𑁆𑀕𑁂, 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑀕𑁆𑀕𑁂, 𑀫𑀼𑀔𑁂𑁇 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀺𑀬𑀸…𑀧𑁂…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑀬𑁄 𑀫𑀕𑁆𑀕𑁂 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲—  𑀯𑀘𑁆𑀘𑀫𑀕𑁆𑀕𑁂, 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑀕𑁆𑀕𑁂, 𑀫𑀼𑀔𑁂𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀼𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲…  𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀼𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀼𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲 𑀢𑀬𑁄 𑀫𑀕𑁆𑀕𑁂 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲—  𑀯𑀘𑁆𑀘𑀫𑀕𑁆𑀕𑁂, 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑀕𑁆𑀕𑁂, 𑀫𑀼𑀔𑁂𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀫𑀕𑁆𑀕𑁂 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲—  𑀯𑀘𑁆𑀘𑀫𑀕𑁆𑀕𑁂, 𑀫𑀼𑀔𑁂𑁇 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲…  𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲…  𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀫𑀕𑁆𑀕𑁂 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲—  𑀯𑀘𑁆𑀘𑀫𑀕𑁆𑀕𑁂, 𑀫𑀼𑀔𑁂𑁇

136. 𑀫𑀕𑁆𑀕𑁂𑀦 𑀫𑀕𑁆𑀕𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀫𑀕𑁆𑀕𑁂𑀦 𑀅𑀫𑀕𑁆𑀕𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀅𑀫𑀕𑁆𑀕𑁂𑀦 𑀫𑀕𑁆𑀕𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀅𑀫𑀕𑁆𑀕𑁂𑀦 𑀅𑀫𑀕𑁆𑀕𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

137. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀢𑁆𑀢𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑁆𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀉𑀪𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀦 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀤𑀽𑀲𑀓𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀢𑁆𑀢𑀲𑀸𑀫𑀡𑁂𑀭𑀫𑁆𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀉𑀪𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀦 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀤𑀽𑀲𑀓𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀲𑀼𑀢𑁆𑀢𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑁆𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀉𑀪𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀦 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀤𑀽𑀲𑀓𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀲𑀼𑀢𑁆𑀢𑀲𑀸𑀫𑀡𑁂𑀭𑀫𑁆𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺; 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀉𑀪𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀝𑀺𑀩𑀼𑀤𑁆𑀥𑁄 𑀦 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀤𑀽𑀲𑀓𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇

138. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀲𑁆𑀲, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

139. 𑀲𑀦𑁆𑀣𑀢𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
