# 𑁧𑁇𑁧𑁇𑁪𑁇𑁨 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

(223.)

713\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀺𑀫𑀸𑀦𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀥𑀺𑀫𑀸𑀦𑁂𑀦𑀸”𑀢𑀺𑁇 (𑁧)

714\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀢𑀺—  “𑀏𑀯𑀁 𑀫𑀁 𑀚𑀦𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑀁 𑀚𑀦𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀯𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨)

715\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺—  “𑀏𑀯𑀁 𑀫𑀁 𑀚𑀦𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑀁 𑀚𑀦𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀘𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩)

716\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀸 𑀲𑀩𑁆𑀩𑁂𑀯 𑀅𑀭𑀳𑀦𑁆𑀢𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀉𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪)

717\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀲𑀩𑁆𑀩𑁂𑀯 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀉𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫)

718\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀺𑀥𑀸𑀬 𑀘𑀗𑁆𑀓𑀫𑀢𑀺…  𑀧𑀡𑀺𑀥𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺…  𑀧𑀡𑀺𑀥𑀸𑀬 𑀦𑀺𑀲𑀻𑀤𑀢𑀺…  𑀧𑀡𑀺𑀥𑀸𑀬 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺—  “𑀏𑀯𑀁 𑀫𑀁 𑀚𑀦𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑀁 𑀚𑀦𑁄 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀺𑀥𑀸𑀬 𑀲𑁂𑀬𑁆𑀬𑀸 𑀓𑀧𑁆𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬--𑁯)

719\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺𑁇 𑀲𑁄𑀧𑀺 𑀏𑀯𑀫𑀸𑀳—  “𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀧𑀳𑀻𑀦𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦)

(224.)

720\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀳𑁄𑀕𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺𑁇 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀤𑀽 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀧𑀲𑀸𑀤𑁂𑀲𑀺—  “𑀫𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀪𑀡𑀺𑁇 𑀦𑀢𑁆𑀣𑁂𑀲𑁄 𑀢𑀼𑀬𑁆𑀳𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁧)

721\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀳𑁄𑀕𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺𑁇 𑀤𑁂𑀯𑀢𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀧𑀲𑀸𑀤𑁂𑀲𑀺—  “𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀪𑀡𑀺𑁇 𑀦𑀢𑁆𑀣𑁂𑀲𑁄 𑀢𑀼𑀬𑁆𑀳𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁨)

722\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀬𑁆𑀳𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀲𑀢𑀺 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀉𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁩)

723\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀁 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀧𑀝𑁆𑀞𑁂𑀲𑀺 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀘 𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑁂𑀢𑀺 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀉𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪)

(225.)

724\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄”𑀢𑀺? “𑀦𑀸𑀯𑀼𑀲𑁄, 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀬𑁂 𑀔𑁄 𑀢𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑁄𑁇 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁫)

725\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄”𑀢𑀺? “𑀆𑀭𑀸𑀥𑀦𑀻𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁄 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁂𑀦𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁬)

726\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀫𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀸𑀬𑀻”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀪𑀸𑀬𑀸𑀫𑀻”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁭)

727\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀫𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀸𑀬𑀻”𑀢𑀺𑁇 “𑀬𑁄 𑀦𑀽𑀦𑀸𑀯𑀼𑀲𑁄, 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀅𑀲𑁆𑀲 𑀲𑁄 𑀪𑀸𑀬𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁮)

728\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄”𑀢𑀺? “𑀆𑀭𑀸𑀥𑀦𑀻𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁄 𑀲𑀫𑁆𑀫𑀸𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁯)

729\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄”𑀢𑀺? “𑀆𑀭𑀸𑀥𑀦𑀻𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁄 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁂𑀦𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁦)

730\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄”𑀢𑀺? “𑀆𑀭𑀸𑀥𑀦𑀻𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁄 𑀬𑀼𑀢𑁆𑀢𑀬𑁄𑀕𑁂𑀦𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁧)

731\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀘𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀦𑁆”𑀢𑀺? “𑀦𑀸𑀯𑀼𑀲𑁄, 𑀲𑀓𑁆𑀓𑀸 𑀬𑁂𑀦 𑀯𑀸 𑀢𑁂𑀦 𑀯𑀸 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁨)

732\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀘𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀦𑁆”𑀢𑀺? “𑀦𑀸𑀯𑀼𑀲𑁄, 𑀲𑀓𑁆𑀓𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀦 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀉𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁩)

(226.)

733\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁄”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳 𑀦 𑀅𑀭𑀳𑀦𑁆𑀢𑁄𑁇 𑀅𑀬𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀅𑀫𑁆𑀳𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀲𑀸𑀤𑀪𑀜𑁆𑀜𑁂”𑀢𑀺𑁇 (𑁨𑁪)

734\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁄”𑀢𑀺…  “𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁄”𑀢𑀺…  “𑀢𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁄”𑀢𑀺…  “𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼, 𑀪𑁄𑀦𑁆𑀢𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁄”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳 𑀦 𑀅𑀭𑀳𑀦𑁆𑀢𑁄𑁇 𑀅𑀬𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀅𑀫𑁆𑀳𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀲𑀸𑀤𑀪𑀜𑁆𑀜𑁂”𑀢𑀺𑁇 (𑁨𑁫--𑁨𑁮)

735\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺𑁇 𑀲𑁄𑀧𑀺 𑀏𑀯𑀫𑀸𑀳—  “𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀲𑀯𑀸 𑀧𑀳𑀻𑀦𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁯)

736\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺𑁇 𑀲𑁄𑀧𑀺 𑀏𑀯𑀫𑀸𑀳—  “𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁦)

737\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺𑁇 𑀲𑁄𑀧𑀺 𑀏𑀯𑀫𑀸𑀳—  “𑀅𑀳𑀫𑁆𑀧𑀸𑀯𑀼𑀲𑁄, 𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁧)

738\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀜𑀸𑀢𑀓𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀸”𑀢𑀺𑁇 “𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸𑀤𑀺𑀲𑁄 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁨)

(227.)

739\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀜𑀸𑀢𑀓𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀸”𑀢𑀺𑁇 “𑀆𑀯𑀝𑀸 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁩)

740\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀜𑀸𑀢𑀓𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀪𑀺𑀭𑀫𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺? “𑀅𑀪𑀺𑀭𑀢𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀭𑀫𑀸𑀬 𑀅𑀪𑀺𑀭𑀢𑀺𑀬𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 “𑀬𑁂 𑀔𑁄 𑀢𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑁄𑁇 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁪)

741\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀢𑀺𑀓𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼—  “𑀬𑁄 𑀇𑀫𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀧𑀞𑀫𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀫𑀬𑀁 𑀅𑀭𑀳𑀸𑀢𑀺 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀫𑀁 𑀅𑀭𑀳𑀸𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑀽”𑀢𑀺, 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀧𑀞𑀫𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁫)

(228.)

742\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀮𑀓𑁆𑀔𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀸𑀫𑀸𑀯𑀼𑀲𑁄 𑀮𑀓𑁆𑀔𑀡, 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇

743\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑁆𑀯𑀸𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀓𑀫𑁆𑀫𑀸𑀬𑀸”𑀢𑀺? “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀮𑀓𑁆𑀔𑀡, 𑀏𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀳𑀲𑁆𑀲𑁇 𑀪𑀕𑀯𑀢𑁄 𑀫𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀏𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀸”𑀢𑀺𑁇

744\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀓𑁆𑀔𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑁆𑀯𑀸𑀓𑀸𑀲𑀺𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀓𑀫𑁆𑀫𑀸𑀬𑀸”𑀢𑀺? “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀝𑁆𑀞𑀺𑀓𑀲𑀗𑁆𑀔𑀮𑀺𑀓𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀁𑁇 𑀢𑀫𑁂𑀦𑀁 𑀕𑀺𑀚𑁆𑀛𑀸𑀧𑀺 𑀓𑀸𑀓𑀸𑀧𑀺 𑀓𑀼𑀮𑀮𑀸𑀧𑀺 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀨𑀸𑀲𑀼𑀍𑀅𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀳𑀺 𑀯𑀺𑀢𑀼𑀟𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀸 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀬𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀉𑀮𑁆𑀮𑀧𑀢𑀻”𑀢𑀺𑁇

745\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀘𑀓𑁆𑀔𑀼𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀜𑀸𑀡𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀁 𑀜𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀤𑀓𑁆𑀔𑀢𑀺 𑀯𑀸 𑀲𑀓𑁆𑀔𑀺𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀦 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀧𑀭𑁂 𑀘 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑁂 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀕𑁄𑀖𑀸𑀢𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀸𑀯𑀲𑁂𑀲𑁂𑀦 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑀘𑁆𑀘𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀳𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁬)

761\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀸 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀸, 𑀧𑀢𑁆𑀢𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄, 𑀓𑀸𑀬𑀩𑀦𑁆𑀥𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁, 𑀓𑀸𑀬𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄𑁇 𑀲𑁄 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀯𑀘𑀦𑁂 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇 (𑁫𑁨)

762\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁…  𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁…  𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀁…  𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀁 𑀯𑁂𑀳𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀁𑁇 𑀢𑀲𑁆𑀲𑀸 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀸 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀸, 𑀧𑀢𑁆𑀢𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄, 𑀓𑀸𑀬𑀩𑀦𑁆𑀥𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁, 𑀓𑀸𑀬𑁄𑀧𑀺 𑀆𑀤𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑁄𑁇 𑀲𑀸 𑀲𑀼𑀤𑀁 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀲𑀢𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀬𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁄𑀧𑀺 𑀦𑀸𑀫 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀉𑀮𑁆𑀮𑀧𑀢𑀻”𑀢𑀺𑁇

763\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀘𑀓𑁆𑀔𑀼𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀜𑀸𑀡𑀪𑀽𑀢𑀸 𑀯𑀢, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀭𑀽𑀧𑀁 𑀜𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀤𑀓𑁆𑀔𑀢𑀺 𑀯𑀸 𑀲𑀓𑁆𑀔𑀺𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀫𑁂 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀤𑀺𑀝𑁆𑀞𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀦 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑁂𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀧𑀭𑁂 𑀘 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀬𑁂 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀼𑀁 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀅𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀯𑀘𑀦𑁂 𑀧𑀸𑀧𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑀸 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀸𑀯𑀲𑁂𑀲𑁂𑀦 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑀘𑁆𑀘𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀳𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁩--𑁫𑁬)

764\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀢𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀧𑁄𑀤𑀸 𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑁄 𑀤𑀳𑁄 𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑁄 𑀲𑀻𑀢𑁄𑀤𑀓𑁄 𑀲𑀸𑀢𑁄𑀤𑀓𑁄 𑀲𑁂𑀢𑀓𑁄 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀢𑁆𑀣𑁄 𑀭𑀫𑀡𑀻𑀬𑁄 𑀧𑀳𑀽𑀢𑀫𑀘𑁆𑀙𑀓𑀘𑁆𑀙𑀧𑁄 𑀘𑀓𑁆𑀓𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀺—  ‘𑀬𑀢𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀧𑁄𑀤𑀸 𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑁄 𑀤𑀳𑁄 𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑁄 𑀲𑀻𑀢𑁄𑀤𑀓𑁄 𑀲𑀸𑀢𑁄𑀤𑀓𑁄 𑀲𑁂𑀢𑀓𑁄 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀢𑁆𑀣𑁄 𑀭𑀫𑀡𑀻𑀬𑁄 𑀧𑀳𑀽𑀢𑀫𑀘𑁆𑀙𑀓𑀘𑁆𑀙𑀧𑁄 𑀘𑀓𑁆𑀓𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀘 𑀧𑀦𑀸𑀬𑀁 𑀢𑀧𑁄𑀤𑀸 𑀓𑀼𑀣𑀺𑀢𑀸 𑀲𑀦𑁆𑀤𑀢𑀺𑁇 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀉𑀮𑁆𑀮𑀧𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀬𑀢𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀧𑁄𑀤𑀸 𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑁄 𑀤𑀳𑁄 𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑁄 𑀲𑀻𑀢𑁄𑀤𑀓𑁄 𑀲𑀸𑀢𑁄𑀤𑀓𑁄 𑀲𑁂𑀢𑀓𑁄 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀢𑁆𑀣𑁄 𑀭𑀫𑀡𑀻𑀬𑁄 𑀧𑀳𑀽𑀢𑀫𑀘𑁆𑀙𑀓𑀘𑁆𑀙𑀧𑁄 𑀘𑀓𑁆𑀓𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀧𑀤𑀼𑀫𑀸𑀦𑀺 𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀧𑁄𑀤𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀦𑀺𑀭𑀬𑀸𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀬𑀁 𑀢𑀧𑁄𑀤𑀸 𑀓𑀼𑀣𑀺𑀢𑀸 𑀲𑀦𑁆𑀤𑀢𑀺𑁇 𑀲𑀘𑁆𑀘𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀳𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁭)

765\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀧𑀪𑀕𑁆𑀕𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀭𑀸𑀚𑀸 𑀧𑀘𑁆𑀙𑀸 𑀲𑁂𑀦𑀁 𑀲𑀗𑁆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀬𑁄 𑀧𑀭𑀸𑀚𑁂𑀲𑀺𑁇 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀘 𑀦𑀦𑁆𑀤𑀻 𑀘𑀭𑀢𑀺—  “𑀭𑀜𑁆𑀜𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀧𑀪𑀕𑁆𑀕𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀭𑀸𑀚𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀳𑀺 𑀧𑀪𑀕𑁆𑀕𑁄”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀺—  ‘𑀭𑀸𑀚𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀳𑀺 𑀧𑀪𑀕𑁆𑀕𑁄’𑀢𑀺𑁇 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀘 𑀦𑀦𑁆𑀤𑀻 𑀘𑀭𑀢𑀺—  ‘𑀭𑀜𑁆𑀜𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀧𑀪𑀕𑁆𑀕𑀸’𑀢𑀺𑁇 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀉𑀮𑁆𑀮𑀧𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀧𑀞𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀚𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀳𑀺 𑀧𑀪𑀕𑁆𑀕𑁄𑁇 𑀅𑀣 𑀭𑀸𑀚𑀸 𑀧𑀘𑁆𑀙𑀸 𑀲𑁂𑀦𑀁 𑀲𑀗𑁆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀬𑁄 𑀧𑀭𑀸𑀚𑁂𑀲𑀺𑁇 𑀲𑀘𑁆𑀘𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀳𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁮)

(232.)

766\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀧𑁆𑀧𑀺𑀦𑀺𑀓𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀦𑀸𑀕𑀸𑀦𑀁 𑀑𑀕𑀬𑁆𑀳 𑀉𑀢𑁆𑀢𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑁄𑀜𑁆𑀘𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀦𑁆”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀦𑁂𑀜𑁆𑀚𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀲𑀤𑁆𑀤𑀁 𑀲𑁄𑀲𑁆𑀲𑀢𑀺𑁇 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀉𑀮𑁆𑀮𑀧𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀢𑁆𑀣𑁂𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀸𑀥𑀺 𑀲𑁄 𑀘 𑀔𑁄 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄𑁇 𑀲𑀘𑁆𑀘𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀳𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁯)

767\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀪𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀘 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀻”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀪𑀺𑀢𑁄 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀺—  ‘𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀘 𑀓𑀧𑁆𑀧𑀲𑀢𑀸𑀦𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀻’𑀢𑀺𑁇 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀪𑀺𑀢𑁄 𑀉𑀮𑁆𑀮𑀧𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀢𑁆𑀣𑁂𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀪𑀺𑀢𑀲𑁆𑀲𑁇 𑀲𑀸 𑀘 𑀔𑁄 𑀏𑀓𑀸𑀬𑁂𑀯 𑀚𑀸𑀢𑀺𑁇 𑀲𑀘𑁆𑀘𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀪𑀺𑀢𑁄 𑀆𑀳𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀪𑀺𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁦)

---

768\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇



(233.)

769\. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀮𑀪𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀸𑀲𑀁, 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑁄𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻𑀢𑀺𑁇

---

770\. 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

771\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



772\. _𑀫𑁂𑀣𑀼𑀦𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀜𑁆𑀘,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀼𑀢𑁆𑀢𑀭𑀺;_  
_𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺,_  
_𑀙𑁂𑀚𑁆𑀚𑀯𑀢𑁆𑀣𑀽 𑀅𑀲𑀁𑀲𑀬𑀸𑀢𑀺𑁇_  


---

773\. 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
