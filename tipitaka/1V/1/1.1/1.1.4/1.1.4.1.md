# 𑁧𑁇𑁧𑁇𑁪𑁇𑁧 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

705\. _𑀅𑀥𑀺𑀫𑀸𑀦𑁂 𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀧𑀺𑀡𑁆𑀟𑁄𑀧𑀚𑁆𑀛𑀸𑀭𑀺𑀬𑀸𑀧𑀣𑁄;_  
_𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀭𑀳𑁄𑀥𑀫𑁆𑀫𑀸,_  
_𑀯𑀺𑀳𑀸𑀭𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄𑁇_  


706\. _𑀦 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀯𑀺𑀭𑀺𑀬𑀫𑀣𑁄𑀧𑀺 𑀫𑀘𑁆𑀘𑀼𑀦𑁄,_  
_𑀪𑀸𑀬𑀸𑀯𑀼𑀲𑁄 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺 𑀲𑀫𑁆𑀫𑀸;_  
_𑀯𑀺𑀭𑀺𑀬𑁂𑀦 𑀬𑁄𑀕𑁂𑀦 𑀆𑀭𑀸𑀥𑀦𑀸𑀬,_  
_𑀅𑀣 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀸 𑀤𑀼𑀯𑁂𑁇_  


707\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀧𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀽𑀦𑀺,_  
_𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸 𑀢𑀬𑁄;_  
_𑀅𑀕𑀸𑀭𑀸𑀯𑀭𑀡𑀸 𑀓𑀸𑀫𑀸,_  
_𑀭𑀢𑀺 𑀘𑀸𑀧𑀺 𑀅𑀧𑀓𑁆𑀓𑀫𑀺𑁇_  


708\. _𑀅𑀝𑁆𑀞𑀺 𑀧𑁂𑀲𑀺 𑀉𑀪𑁄 𑀕𑀸𑀯𑀖𑀸𑀢𑀓𑀸,_  
_𑀧𑀺𑀡𑁆𑀟𑁄 𑀲𑀸𑀓𑀼𑀡𑀺𑀓𑁄 𑀦𑀺𑀘𑁆𑀙𑀯𑀺 𑀑𑀭𑀩𑁆𑀪𑀺;_  
_𑀅𑀲𑀺 𑀘 𑀲𑀽𑀓𑀭𑀺𑀓𑁄 𑀲𑀢𑁆𑀢𑀺 𑀫𑀸𑀕𑀯𑀺,_  
_𑀉𑀲𑀼 𑀘 𑀓𑀸𑀭𑀡𑀺𑀓𑁄 𑀲𑀽𑀘𑀺 𑀲𑀸𑀭𑀣𑀺𑁇_  


709\. _𑀬𑁄 𑀘 𑀲𑀺𑀩𑁆𑀩𑀻𑀬𑀢𑀺 𑀲𑀽𑀘𑀓𑁄 𑀳𑀺 𑀲𑁄,_  
_𑀅𑀡𑁆𑀟𑀪𑀸𑀭𑀺 𑀅𑀳𑀼 𑀕𑀸𑀫𑀓𑀽𑀝𑀓𑁄;_  
_𑀓𑀽𑀧𑁂 𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑁄 𑀳𑀺 𑀲𑁄 𑀧𑀸𑀭𑀤𑀸𑀭𑀺𑀓𑁄,_  
_𑀕𑀽𑀣𑀔𑀸𑀤𑀻 𑀅𑀳𑀼 𑀤𑀼𑀝𑁆𑀞𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


710\. _𑀦𑀺𑀘𑁆𑀙𑀯𑀺𑀢𑁆𑀣𑀻 𑀅𑀢𑀺𑀘𑀸𑀭𑀺𑀦𑀻 𑀅𑀳𑀼,_  
_𑀫𑀗𑁆𑀕𑀼𑀮𑀺𑀢𑁆𑀣𑀻 𑀅𑀳𑀼 𑀇𑀓𑁆𑀔𑀡𑀺𑀢𑁆𑀣𑀺𑀓𑀸;_  
_𑀑𑀓𑀺𑀮𑀺𑀦𑀻 𑀳𑀺 𑀲𑀧𑀢𑁆𑀢𑀗𑁆𑀕𑀸𑀭𑁄𑀓𑀺𑀭𑀺,_  
_𑀲𑀻𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀳𑀼 𑀘𑁄𑀭𑀖𑀸𑀢𑀓𑁄𑁇_  


711\. _𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸,_  
_𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀅𑀣 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀓𑀸;_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀩𑁆𑀩𑀚𑀁,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁𑀲𑀼 𑀢𑀸𑀯𑀤𑁂𑁇_  


712\. _𑀢𑀧𑁄𑀤𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀼𑀤𑁆𑀥𑀁,_  
_𑀦𑀸𑀕𑀸𑀦𑁄𑀕𑀸𑀳𑀦𑁂𑀦 𑀘;_  
_𑀲𑁄𑀪𑀺𑀢𑁄 𑀅𑀭𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀜𑁆𑀘𑀓𑀧𑁆𑀧𑀲𑀢𑀁 𑀲𑀭𑁂𑀢𑀺𑁇_
