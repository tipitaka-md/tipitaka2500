# 𑁧𑁇𑁧𑁇𑁨𑁇𑁨 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

(132.)

292\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀚𑀓𑀢𑁆𑀣𑀭𑀡𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀭𑀚𑀓𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀅𑀯𑀳𑀭𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧)

293\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀚𑀓𑀢𑁆𑀣𑀭𑀡𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀳𑀕𑁆𑀖𑀁 𑀤𑀼𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀘𑀺𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑁂”𑀢𑀺𑁇 (𑁨)

294\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀚𑀓𑀢𑁆𑀣𑀭𑀡𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀳𑀕𑁆𑀖𑀁 𑀤𑀼𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀫𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩)

295\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀚𑀓𑀢𑁆𑀣𑀭𑀡𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀳𑀕𑁆𑀖𑀁 𑀤𑀼𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀨𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪)

296\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀚𑀓𑀢𑁆𑀣𑀭𑀡𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀳𑀕𑁆𑀖𑀁 𑀤𑀼𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀞𑀸𑀦𑀸 𑀘𑀸𑀯𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁫)

(133.)

297\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀳𑀕𑁆𑀖𑀁 𑀉𑀢𑁆𑀢𑀭𑀢𑁆𑀣𑀭𑀡𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀫𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀨𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀞𑀸𑀦𑀸 𑀘𑀸𑀯𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁬--𑁯)

(134.)

298\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀯𑀸 𑀪𑀡𑁆𑀟𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀓𑀸𑀲𑀺—  “𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀯𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀁 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀁 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀁 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀁 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦--𑁧𑁩)

299\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀯𑀸 𑀪𑀡𑁆𑀟𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀓𑀸𑀲𑀺—  “𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀯𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀡𑁆𑀟𑀁 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪)

300\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀁 𑀳𑀭𑀦𑁆𑀢𑁄 𑀲𑀻𑀲𑁂 𑀪𑀸𑀭𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀫𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀨𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑀦𑁆𑀥𑀁 𑀑𑀭𑁄𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀔𑀦𑁆𑀥𑁂 𑀪𑀸𑀭𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀫𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀨𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀝𑀺𑀁 𑀑𑀭𑁄𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀓𑀝𑀺𑀬𑀸 𑀪𑀸𑀭𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀫𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀨𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑀢𑁆𑀣𑁂𑀦 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺…𑀧𑁂…  𑀳𑀢𑁆𑀣𑁂 𑀪𑀸𑀭𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀪𑀽𑀫𑀺𑀬𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀪𑀽𑀫𑀺𑀢𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁫--𑁨𑁫)

(135.)

301\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀫𑀸𑀬𑀺𑀤𑀁 𑀘𑀻𑀯𑀭𑀁 𑀦𑀲𑁆𑀲𑀻”𑀢𑀺, 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀲𑀺𑁇 𑀲𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀆𑀯𑀼𑀲𑁄, 𑀫𑀬𑁆𑀳𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑁂𑀦 𑀅𑀯𑀳𑀝𑀦𑁆”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀫𑀬𑀸 𑀅𑀯𑀳𑀝𑀦𑁆”𑀢𑀺𑁇 “𑀲𑁄 𑀢𑀁 𑀆𑀤𑀺𑀬𑀺, 𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀣𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀣𑁂”𑀢𑀺𑁇 (𑁨𑁬)

302\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀞𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸𑁇 𑀧𑀻𑀞𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸…  𑀳𑁂𑀝𑁆𑀞𑀸𑀧𑀻𑀞𑁂 𑀧𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀫𑀸𑀬𑀁 𑀧𑀢𑁆𑀢𑁄 𑀦𑀲𑁆𑀲𑀻”𑀢𑀺 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀲𑀺𑁇 𑀲𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀆𑀯𑀼𑀲𑁄, 𑀫𑀬𑁆𑀳𑀁 𑀧𑀢𑁆𑀢𑁄 𑀓𑁂𑀦 𑀅𑀯𑀳𑀝𑁄”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀫𑀬𑀸 𑀅𑀯𑀳𑀝𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀤𑀺𑀬𑀺, “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀣𑁂”𑀢𑀺𑁇 (𑁨𑁭--𑁨𑁯)

303\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑀢𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀫𑀸𑀬𑀺𑀤𑀁 𑀘𑀻𑀯𑀭𑀁 𑀦𑀲𑁆𑀲𑀻”𑀢𑀺 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀲𑀺𑁇 𑀲𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀅𑀬𑁆𑀬𑁂, 𑀫𑀬𑁆𑀳𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑁂𑀦 𑀅𑀯𑀳𑀝𑀦𑁆”𑀢𑀺? 𑀲𑀸 𑀏𑀯𑀫𑀸𑀳—  “𑀫𑀬𑀸 𑀅𑀯𑀳𑀝𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀸 𑀢𑀁 𑀆𑀤𑀺𑀬𑀺, 𑀅𑀲𑁆𑀲𑀫𑀡𑀻𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀣𑁂”𑀢𑀺𑁇 (𑁩𑁦)

(136.)

304\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀸𑀢𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸𑀬 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀲𑀸𑀝𑀓𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺, 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀅𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁧)

305\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀸𑀢𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸𑀬 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀯𑁂𑀞𑀦𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁨)

(137.)

306\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀲𑀸𑀦𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀦𑁆𑀦𑁂 𑀲𑀭𑀻𑀭𑁂 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀲𑀭𑀻𑀭𑁂 𑀧𑁂𑀢𑁄 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑁂𑀢𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀬𑁆𑀳𑀁 𑀲𑀸𑀝𑀓𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁄 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁 𑀉𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑀁 𑀣𑀓𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁 𑀢𑀢𑁆𑀣𑁂𑀯 𑀧𑀭𑀺𑀧𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀦𑁆𑀦𑁂 𑀲𑀭𑀻𑀭𑁂 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁩)

(138.)

307\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀼𑀲𑀁 𑀲𑀗𑁆𑀓𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁪)

(139.)

308\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀦𑀺𑀯𑀸𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀓𑀲𑀜𑁆𑀜𑀻 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁫)

(140.)

309\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑀸 𑀲𑀻𑀳𑀯𑀺𑀖𑀸𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀳𑀯𑀺𑀖𑀸𑀲𑁂”𑀢𑀺𑁇 (𑁩𑁬)

310\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑀸 𑀩𑁆𑀬𑀕𑁆𑀖𑀯𑀺𑀖𑀸𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸…  𑀤𑀻𑀧𑀺𑀯𑀺𑀖𑀸𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸…  𑀢𑀭𑀘𑁆𑀙𑀯𑀺𑀖𑀸𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸…  𑀓𑁄𑀓𑀯𑀺𑀖𑀸𑀲𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂”𑀢𑀺𑁇 (𑁩𑁭--𑁪𑁦)

(141.)

311\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀑𑀤𑀦𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂—  “𑀅𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀕𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺 𑀅𑀫𑀽𑀮𑀓𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪𑁧)

312\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀔𑀸𑀤𑀦𑀻𑀬𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀽𑀯𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀘𑁆𑀙𑀼𑀫𑁆𑀳𑀺 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂—  “𑀅𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀕𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺 𑀅𑀫𑀽𑀮𑀓𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪𑁨--𑁪𑁫)

(142.)

313\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀑𑀤𑀦𑀻𑀬𑀖𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀧𑀽𑀭𑀁 𑀑𑀤𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁪𑁬)

314\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀲𑀽𑀦𑀖𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀧𑀽𑀭𑀁 𑀫𑀁𑀲𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁪𑁭)

315\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀧𑀽𑀯𑀖𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀧𑀽𑀭𑀁 𑀧𑀽𑀯𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀧𑀢𑁆𑀢𑀧𑀽𑀭𑀸 𑀲𑀓𑁆𑀔𑀮𑀺𑀬𑁄 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀧𑀢𑁆𑀢𑀧𑀽𑀭𑁂 𑀫𑁄𑀤𑀓𑁂 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁪𑁮--𑁫𑁦)

(143.)

316\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀓𑀸𑀲𑀺—  “𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀯𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀁 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀁 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀢𑀁 𑀅𑀯𑀳𑀭𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀁 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁫𑁧--𑁫𑁪)

317\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀓𑀸𑀲𑀺—  “𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀯𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁫)

(144.)

318\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀻𑀞𑁂 𑀣𑀯𑀺𑀓𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀇𑀢𑁄 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀲𑀳 𑀧𑀻𑀞𑀓𑁂𑀦 𑀲𑀗𑁆𑀓𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁫𑁬)

319\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀺𑀲𑀺𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁫𑁭)

(145.)

320\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀻𑀯𑀭𑀯𑀁𑀲𑁂 𑀘𑀻𑀯𑀭𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁫𑁮)

321\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀸𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀅𑀯𑀳𑀭𑀺𑀢𑁆𑀯𑀸—  “𑀇𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀺𑀳𑀸𑀭𑀸 𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑀺…𑀧𑁂…  𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦𑀺𑀓𑁆𑀔𑀫𑀺 𑀯𑀸 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫𑁯)

(146.)

322\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀳𑀸𑀬𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀔𑀸𑀤𑀦𑀻𑀬𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂 𑀲𑀳𑀸𑀬𑀓𑀲𑁆𑀲 𑀪𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀲𑁄 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀕𑁆𑀕𑀸𑀳𑁂”𑀢𑀺𑁇 (𑁬𑁦)

(147.)

323\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑁆𑀲 𑀔𑀸𑀤𑀦𑀻𑀬𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀧𑀝𑀺𑀯𑀺𑀲𑀸 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀯𑀺𑀲𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀲𑁄 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀲𑀓𑀲𑀜𑁆𑀜𑀻 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀓𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁧)

324\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑁆𑀲 𑀔𑀸𑀤𑀦𑀻𑀬𑁂 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀯𑀺𑀲𑁄 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑀢𑁆𑀢𑀲𑀸𑀫𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀢𑀦𑁄 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀲𑁄 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀓𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁨)

(148.)

325\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀘𑁄𑀭𑀓𑀸 𑀅𑀫𑁆𑀩𑀁 𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀆𑀤𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀘𑁄𑀭𑀓𑁂 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀁𑀲𑀼𑁇 𑀘𑁄𑀭𑀓𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀧𑀮𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑀸 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺? “𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀫𑀬𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁩)

326\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀚𑀫𑁆𑀩𑀼𑀘𑁄𑀭𑀓𑀸…  𑀮𑀩𑀼𑀚𑀘𑁄𑀭𑀓𑀸…  𑀧𑀦𑀲𑀘𑁄𑀭𑀓𑀸…  𑀢𑀸𑀮𑀧𑀓𑁆𑀓𑀘𑁄𑀭𑀓𑀸…  𑀉𑀘𑁆𑀙𑀼𑀘𑁄𑀭𑀓𑀸…  𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑀘𑁄𑀭𑀓𑀸 𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑁂 𑀉𑀘𑁆𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀆𑀤𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀘𑁄𑀭𑀓𑁂 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀁𑀲𑀼𑁇 𑀘𑁄𑀭𑀓𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀧𑀮𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁪--𑁬𑁯)

327\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀘𑁄𑀭𑀓𑀸 𑀅𑀫𑁆𑀩𑀁 𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀆𑀤𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀘𑁄𑀭𑀓𑁂 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀁𑀲𑀼𑁇 𑀘𑁄𑀭𑀓𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀧𑀮𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁭𑁦)

328\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀚𑀫𑁆𑀩𑀼𑀘𑁄𑀭𑀓𑀸…  𑀮𑀩𑀼𑀚𑀘𑁄𑀭𑀓𑀸…  𑀧𑀦𑀲𑀘𑁄𑀭𑀓𑀸…  𑀢𑀸𑀮𑀧𑀓𑁆𑀓𑀘𑁄𑀭𑀓𑀸…  𑀉𑀘𑁆𑀙𑀼𑀘𑁄𑀭𑀓𑀸…  𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑀘𑁄𑀭𑀓𑀸 𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑁂 𑀉𑀘𑁆𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀆𑀤𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀘𑁄𑀭𑀓𑁂 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀁𑀲𑀼𑁇 𑀘𑁄𑀭𑀓𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀧𑀮𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁭𑁧--𑁭𑁬)

329\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀚𑀫𑁆𑀩𑀼𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀮𑀩𑀼𑀚𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀦𑀲𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀢𑀸𑀮𑀧𑀓𑁆𑀓𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀘𑁆𑀙𑀼𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁭𑁭--𑁮𑁩)

(149.)

330\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀧𑁆𑀨𑀸𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀑𑀘𑀺𑀢𑀁 𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀜𑁆𑀘𑀫𑀸𑀲𑀕𑁆𑀖𑀦𑀓𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁮𑁪)

331\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀧𑁆𑀨𑀸𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀼𑀧𑁆𑀨𑀁 𑀑𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀫𑀸𑀲𑀕𑁆𑀖𑀦𑀓𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁮𑁫)

(150.)

332\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀓𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀬𑁆𑀳𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀆𑀳𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀲𑁄 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀫𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁬)

333\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀓𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀫𑀬𑁆𑀳𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑀼𑀕𑀲𑀸𑀝𑀓𑀁 𑀆𑀳𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺, 𑀏𑀓𑀁 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀳𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁭)

334\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀓𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀬𑁆𑀳𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀫𑀻”𑀢𑀺𑁇 𑀲𑁄𑀧𑀺 𑀏𑀯𑀫𑀸𑀳—  “𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀍𑀳𑀓𑀁 𑀲𑀧𑁆𑀧𑀺𑀁 𑀢𑀼𑀮𑀁 𑀕𑀼𑀍𑀅𑀁 𑀤𑁄𑀡𑀁 𑀢𑀡𑁆𑀟𑀼𑀮𑀁 𑀆𑀳𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀲𑁄 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀫𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀘 𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑁂𑀳𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀯𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁮)

(151.)

335\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀕𑁆𑀖𑀁 𑀫𑀡𑀺𑀁 𑀆𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀗𑁆𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀣𑀯𑀺𑀓𑀸𑀬 𑀫𑀡𑀺𑀁 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀗𑁆𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮𑁯)

336\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀕𑁆𑀖𑀁 𑀫𑀡𑀺𑀁 𑀆𑀤𑀸𑀬 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀗𑁆𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀸𑀮𑀬𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀤𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀗𑁆𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀳𑀭 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀡𑁆𑀟𑀺𑀓𑀁; 𑀦𑀸𑀳𑀁 𑀅𑀓𑀮𑁆𑀮𑀓𑁄”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁦)

(152.)

337\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀺𑀲𑁂𑀦 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀗𑁆𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀕𑁆𑀖𑀁 𑀫𑀡𑀺𑀁 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀤𑀸𑀲𑀺—  “𑀇𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀡𑀺𑀁 𑀲𑀼𑀗𑁆𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀫𑀡𑀺𑀁 𑀲𑀼𑀗𑁆𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁯𑁧)

(153.)

338\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀲𑁂 𑀩𑀦𑁆𑀥𑀁 𑀲𑀽𑀓𑀭𑀁 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁨)

339\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀲𑁂 𑀩𑀦𑁆𑀥𑀁 𑀲𑀽𑀓𑀭𑀁—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁯𑁩)

340\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀲𑁂 𑀩𑀦𑁆𑀥𑀁 𑀫𑀺𑀕𑀁 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀼𑀜𑁆𑀘𑀺…  𑀧𑀸𑀲𑁂 𑀩𑀦𑁆𑀥𑀁 𑀫𑀺𑀕𑀁—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀫𑀼𑀜𑁆𑀘𑀺…  𑀓𑀼𑀫𑀺𑀦𑁂 𑀩𑀦𑁆𑀥𑁂 𑀫𑀘𑁆𑀙𑁂 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑁂𑀦 𑀫𑀼𑀜𑁆𑀘𑀺…  𑀓𑀼𑀫𑀺𑀦𑁂 𑀩𑀦𑁆𑀥𑁂 𑀫𑀘𑁆𑀙𑁂—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁯𑁪--𑁯𑁭)

341\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑁂 𑀪𑀡𑁆𑀟𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀇𑀢𑁄 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺, 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀯𑀝𑁆𑀝𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁯𑁮)

342\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀼𑀮𑀮𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀫𑀁𑀲𑀧𑁂𑀲𑀺𑀁—  “𑀲𑀸𑀫𑀺𑀓𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯𑁯)

343\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀼𑀮𑀮𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀫𑀁𑀲𑀧𑁂𑀲𑀺𑀁—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁦)

(154.)

344\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀍𑀉𑀫𑁆𑀧𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀅𑀘𑀺𑀭𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀑𑀲𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀩𑀦𑁆𑀥𑀦𑁂 𑀙𑀺𑀦𑁆𑀦𑁂 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀺𑀡𑁆𑀡𑀸𑀦𑀺 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀉𑀢𑁆𑀢𑀸𑀭𑁂𑀲𑀼𑀁𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦𑁧)

345\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀍𑀉𑀫𑁆𑀧𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀅𑀘𑀺𑀭𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀑𑀲𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀩𑀦𑁆𑀥𑀦𑁂 𑀙𑀺𑀦𑁆𑀦𑁂 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀺𑀡𑁆𑀡𑀸𑀦𑀺 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀸 𑀉𑀢𑁆𑀢𑀸𑀭𑁂𑀲𑀼𑀁𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁨)

346\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀭𑀼𑀓𑁆𑀔𑁂 𑀲𑀸𑀝𑀓𑀁 𑀆𑀮𑀕𑁆𑀕𑁂𑀢𑁆𑀯𑀸 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀕𑁄𑀧𑀸𑀮𑀓𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦𑁩)

347\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀤𑀺𑀁 𑀢𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀭𑀚𑀓𑀸𑀦𑀁 𑀳𑀢𑁆𑀣𑀢𑁄 𑀫𑀼𑀢𑁆𑀢𑀁 𑀲𑀸𑀝𑀓𑀁 𑀧𑀸𑀤𑁂 𑀮𑀕𑁆𑀕𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀲𑀸𑀫𑀺𑀓𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦𑁪)

348\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀤𑀺𑀁 𑀢𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀭𑀚𑀓𑀸𑀦𑀁 𑀳𑀢𑁆𑀣𑀢𑁄 𑀫𑀼𑀢𑁆𑀢𑀁 𑀲𑀸𑀝𑀓𑀁 𑀧𑀸𑀤𑁂 𑀮𑀕𑁆𑀕𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀧𑀼𑀭𑁂 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁫)

(155.)

349\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀧𑁆𑀧𑀺𑀓𑀼𑀫𑁆𑀪𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀣𑁄𑀓𑀁 𑀣𑁄𑀓𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦𑁬)

350\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀁𑀲𑀼—  “𑀪𑀡𑁆𑀟𑀁 𑀅𑀯𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀏𑀓𑁄 𑀪𑀡𑁆𑀟𑀁 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑁇 𑀬𑁄 𑀅𑀯𑀳𑀝𑁄 𑀲𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁭)

351\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀁 𑀅𑀯𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑁂𑀲𑀼𑀁𑁇 𑀢𑁂𑀳𑀺 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁂 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑀺𑀲𑁄 𑀦 𑀧𑀜𑁆𑀘𑀫𑀸𑀲𑀓𑁄 𑀧𑀽𑀭𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁮)

352\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀆𑀧𑀡𑀺𑀓𑀲𑁆𑀲 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑀼𑀝𑁆𑀞𑀺𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦𑁯)

353\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀆𑀧𑀡𑀺𑀓𑀲𑁆𑀲 𑀫𑀼𑀕𑁆𑀕𑀫𑀼𑀝𑁆𑀞𑀺𑀁…  𑀫𑀸𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀁…  𑀢𑀺𑀮𑀫𑀼𑀝𑁆𑀞𑀺𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁧𑁦--𑁧𑁧𑁨)

354\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀅𑀦𑁆𑀥𑀯𑀦𑁂 𑀘𑁄𑀭𑀓𑀸 𑀕𑀸𑀯𑀺𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀔𑀸𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀲𑀓𑀁 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀘𑁄𑀭𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁧𑁩)

355\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀅𑀦𑁆𑀥𑀯𑀦𑁂 𑀘𑁄𑀭𑀓𑀸 𑀲𑀽𑀓𑀭𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀔𑀸𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀲𑀓𑀁 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀘𑁄𑀭𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁧𑁪)

356\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀺𑀡𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀮𑀽𑀢𑀁 𑀢𑀺𑀡𑀁 𑀧𑀜𑁆𑀘𑀫𑀸𑀲𑀕𑁆𑀖𑀦𑀓𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁧𑁫)

357\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀺𑀡𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀺𑀡𑀁 𑀮𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀫𑀸𑀲𑀕𑁆𑀖𑀦𑀓𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁧𑁬)

(156.)

358\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀁 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑀸 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺? “𑀧𑀭𑀺𑀪𑁄𑀕𑀢𑁆𑀣𑀸𑀬 𑀫𑀬𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀪𑁄𑀕𑀢𑁆𑀣𑀸𑀬𑀸”𑀢𑀺𑁇 (𑁧𑁧𑁭)

359\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀲𑁆𑀲 𑀚𑀫𑁆𑀩𑀼𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀮𑀩𑀼𑀚𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀦𑀲𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀢𑀸𑀮𑀧𑀓𑁆𑀓𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀘𑁆𑀙𑀼𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑀁 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀸𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀪𑁄𑀕𑀢𑁆𑀣𑀸𑀬𑀸”𑀢𑀺𑁇 (𑁧𑁧𑁮--𑁧𑁨𑁩)

360\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀫𑁆𑀩𑀨𑀮𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀕𑁄𑀧𑁂𑀢𑀼𑀁 𑀇𑀫𑁂 𑀇𑀲𑁆𑀲𑀭𑀸, 𑀦𑀬𑀺𑀫𑁂 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀧𑀓𑀲𑁆𑀲 𑀤𑀸𑀦𑁂”𑀢𑀺𑁇 (𑁧𑁨𑁪)

361\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀚𑀫𑁆𑀩𑀼𑀧𑀸𑀮𑀓𑀸…  𑀮𑀩𑀼𑀚𑀧𑀸𑀮𑀓𑀸…  𑀧𑀦𑀲𑀧𑀸𑀮𑀓𑀸…  𑀢𑀸𑀮𑀧𑀓𑁆𑀓𑀧𑀸𑀮𑀓𑀸…  𑀉𑀘𑁆𑀙𑀼𑀧𑀸𑀮𑀓𑀸…  𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑀧𑀸𑀮𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀕𑁄𑀧𑁂𑀢𑀼𑀁 𑀇𑀫𑁂 𑀇𑀲𑁆𑀲𑀭𑀸, 𑀦𑀬𑀺𑀫𑁂 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀧𑀓𑀲𑁆𑀲 𑀤𑀸𑀦𑁂”𑀢𑀺𑁇 (𑁧𑁨𑁫--𑁧𑁩𑁦)

362\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀁 𑀢𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑀁 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀓𑀼𑀝𑁆𑀝𑀁 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀢𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀢𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁂”𑀢𑀺𑁇 (𑁧𑁩𑁧)

363\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀤𑀓𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀚𑀓𑀺𑀢𑀁 𑀢𑀺𑀡𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…  𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁩𑁨--𑁧𑁩𑁪)

364\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀚𑀓𑀺𑀢𑀁 𑀢𑀺𑀡𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀛𑀸𑀧𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁩𑁫)

365\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀫𑀜𑁆𑀘𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…  𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁩𑁬)

366\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀻𑀞𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀺𑀲𑀺𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀯𑀸𑀝𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀆𑀮𑁄𑀓𑀲𑀦𑁆𑀥𑀺𑀁…  𑀲𑀁𑀖𑀲𑁆𑀲 𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀁 𑀣𑁂𑀬𑁆𑀬𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀯𑀳𑀭𑀺…  𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁩𑁭--𑁧𑁪𑁨)

367\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀧𑀭𑀺𑀪𑁄𑀕𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀤𑀦𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀭𑀺𑀪𑁄𑀕𑀁 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀭𑀺𑀪𑁄𑀕𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪𑁩)

368\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑁄𑀲𑀣𑀕𑁆𑀕𑀫𑁆𑀧𑀺 𑀲𑀦𑁆𑀦𑀺𑀲𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀳𑀭𑀺𑀢𑀼𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀙𑀫𑀸𑀬𑀁 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺𑁇 𑀕𑀢𑁆𑀢𑀸𑀦𑀺𑀧𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺𑀧𑀺 𑀧𑀁𑀲𑀼𑀓𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑀁 𑀳𑀭𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁪𑁪)

369\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸—  “𑀅𑀬𑁆𑀬𑀸 𑀇𑀘𑁆𑀙𑀢𑀺 𑀢𑁂𑀓𑀝𑀼𑀮𑀬𑀸𑀕𑀼𑀁 𑀧𑀸𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀧𑀘𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀲𑀸 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀻𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂… 𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪𑁫)

370\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸—  “𑀅𑀬𑁆𑀬𑀸 𑀇𑀘𑁆𑀙𑀢𑀺 𑀫𑀥𑀼𑀕𑁄𑀍𑀅𑀓𑀁 𑀔𑀸𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀧𑀘𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀲𑀸 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀘𑁄𑀤𑁂𑀲𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀻𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪𑁬)

371\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀚𑁆𑀚𑀼𑀓𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀤𑁆𑀯𑁂 𑀤𑀸𑀭𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑁄 𑀘𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀕𑀳𑀧𑀢𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀚𑁆𑀚𑀼𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀑𑀓𑀸𑀲𑀁 𑀬𑁄 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀸𑀭𑀓𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺 𑀧𑀲𑀦𑁆𑀦𑁄 𑀢𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑁄 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺 𑀧𑀲𑀦𑁆𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑁆𑀚𑀼𑀓𑁄 𑀢𑀁 𑀑𑀓𑀸𑀲𑀁 𑀢𑀲𑁆𑀲 𑀤𑀸𑀭𑀓𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑁂𑀦 𑀓𑀼𑀝𑀼𑀫𑁆𑀩𑀜𑁆𑀘 𑀲𑀡𑁆𑀞𑀧𑁂𑀲𑀺 𑀤𑀸𑀦𑀜𑁆𑀘 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀺𑀢𑀼𑀦𑁄 𑀤𑀸𑀬𑀚𑁆𑀚𑁄—  𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑀸 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑁄 𑀯𑀸”𑀢𑀺? “𑀧𑀼𑀢𑁆𑀢𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀺𑀢𑀼𑀦𑁄 𑀤𑀸𑀬𑀚𑁆𑀚𑁄”𑀢𑀺𑁇 “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀅𑀚𑁆𑀚𑀼𑀓𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀫𑁂𑀣𑀼𑀦𑀓𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀻”𑀢𑀺𑁇 “𑀅𑀲𑁆𑀲𑀫𑀡𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑁆𑀚𑀼𑀓𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑁆𑀚𑀼𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑁂𑀳𑀺 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀚𑁆𑀚𑀼𑀓𑀲𑁆𑀲 𑀧𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀸𑀫𑀺𑀓𑁂𑀦 ‘𑀇𑀫𑀁 𑀑𑀓𑀸𑀲𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸𑀲𑀻’𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀺, 𑀓𑀺𑀁 𑀲𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀺𑀜𑁆𑀘𑀺 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀤𑀼𑀓𑁆𑀓𑀝𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀻”𑀢𑀺𑁇 “𑀅𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑁆𑀚𑀼𑀓𑁄 𑀲𑀸𑀫𑀺𑀓𑁂𑀦—  ‘𑀇𑀫𑀁 𑀑𑀓𑀸𑀲𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀸’𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀺; 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀚𑁆𑀚𑀼𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪𑁭)

(159.)

372\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀘𑁄𑀭𑁂𑀳𑀺 𑀉𑀧𑀤𑁆𑀤𑀼𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀘 𑀤𑀸𑀭𑀓𑀸 𑀦𑀻𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑁂 𑀤𑀸𑀭𑀓𑁂 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀆𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀲𑀸𑀤𑁂 𑀞𑀧𑁂𑀲𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑁂 𑀤𑀸𑀭𑀓𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀅𑀬𑁆𑀬𑀲𑁆𑀲𑀸𑀬𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄”𑀢𑀺, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁂 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀘𑁄𑀭𑁂𑀳𑀺 𑀦𑀻𑀢𑁂 𑀤𑀸𑀭𑀓𑁂 𑀆𑀦𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀤𑁆𑀥𑀺𑀫𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀲𑀬𑁂”𑀢𑀺𑁇 (𑁧𑁪𑁮)

(160.)

373\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀳𑀸𑀬𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀧𑀡𑁆𑀟𑀼𑀓𑁄 𑀘 𑀓𑀧𑀺𑀮𑁄 𑀘𑁇 𑀏𑀓𑁄 𑀕𑀸𑀫𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀏𑀓𑁄 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑀸𑀫𑀓𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀦𑀤𑀺𑀁 𑀢𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀽𑀓𑀭𑀺𑀓𑀸𑀦𑀁 𑀳𑀢𑁆𑀣𑀢𑁄 𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑁂𑀤𑀯𑀝𑁆𑀝𑀺 𑀧𑀸𑀤𑁂 𑀮𑀕𑁆𑀕𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀲𑀸𑀫𑀺𑀓𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑁄𑀤𑁂𑀲𑀼𑀁—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺 𑀢𑁆𑀯𑀦𑁆”𑀢𑀺𑁇 𑀢𑀁 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀁 𑀕𑁄𑀧𑀸𑀮𑀺𑀓𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀸”𑀢𑀺𑁇 𑀲𑁄—  “𑀧𑀓𑀢𑀺𑀬𑀸𑀧𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀫𑀡𑁄”𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀲𑀫𑀸𑀬𑁄𑀕𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪𑁯)

(161.)

374\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀕𑀮𑀸𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀍𑀳𑀺𑀓𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀬𑀸 𑀧𑀻𑀍𑀇𑀢𑁄 𑀆𑀧𑀡𑀺𑀓𑀲𑁆𑀲 𑀯𑁂𑀞𑀦𑀁 𑀅𑀯𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀤𑀍𑀳𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀢𑀬𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀢𑀦𑁆”𑀢𑀺? 𑀲𑁄 𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀆𑀳𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀖𑀸𑀧𑁂𑀲𑀺𑁇 𑀢𑀁 𑀅𑀕𑁆𑀖𑀸𑀧𑁂𑀦𑁆𑀢𑀁 𑀦 𑀧𑀜𑁆𑀘𑀫𑀸𑀲𑀓𑁂 𑀅𑀕𑁆𑀖𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀺𑀭𑀫𑀢𑀻𑀢𑀺𑁇 (𑁧𑁫𑁦)
