# 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀡𑁆𑀟

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀯𑁂𑀭𑀜𑁆𑀚𑀓𑀡𑁆𑀟

(1.)

2. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀯𑁂𑀭𑀜𑁆𑀚𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦𑀍𑀏𑀭𑀼𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀫𑀽𑀮𑁂 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄—  “𑀲𑀫𑀡𑁄 𑀔𑀮𑀼 𑀪𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦𑀍𑀏𑀭𑀼𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀫𑀽𑀮𑁂 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’, 𑀲𑁄 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀲𑀤𑁂𑀯𑀓𑀁 𑀲𑀫𑀸𑀭𑀓𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀁 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀧𑀚𑀁 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁; 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀢𑀺; 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

(2.)

3. 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫—  ‘𑀦 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀚𑀺𑀡𑁆𑀡𑁂 𑀯𑀼𑀟𑁆𑀠𑁂 𑀫𑀳𑀮𑁆𑀮𑀓𑁂 𑀅𑀤𑁆𑀥𑀕𑀢𑁂 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀢𑀬𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀢𑀣𑁂𑀯? 𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀚𑀺𑀡𑁆𑀡𑁂 𑀯𑀼𑀟𑁆𑀠𑁂 𑀫𑀳𑀮𑁆𑀮𑀓𑁂 𑀅𑀤𑁆𑀥𑀕𑀢𑁂 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺? 𑀢𑀬𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀫𑁂𑀯𑀸”𑀢𑀺𑁇

4. “𑀦𑀸𑀳𑀁 𑀢𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀬𑀫𑀳𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀬𑁆𑀬𑀁 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀁 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀬𑁆𑀬𑀁𑁇 𑀬𑀜𑁆𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀬𑁆𑀬, 𑀫𑀼𑀤𑁆𑀥𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

(3.)

5. “𑀅𑀭𑀲𑀭𑀽𑀧𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀭𑀲𑀭𑀽𑀧𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀭𑀽𑀧𑀭𑀲𑀸 𑀲𑀤𑁆𑀤𑀭𑀲𑀸 𑀕𑀦𑁆𑀥𑀭𑀲𑀸 𑀭𑀲𑀭𑀲𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀭𑀲𑀸, 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀭𑀲𑀭𑀽𑀧𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

(4.)

6. “𑀦𑀺𑀩𑁆𑀪𑁄𑀕𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦𑀺𑀩𑁆𑀪𑁄𑀕𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀭𑀽𑀧𑀪𑁄𑀕𑀸 𑀲𑀤𑁆𑀤𑀪𑁄𑀕𑀸 𑀕𑀦𑁆𑀥𑀪𑁄𑀕𑀸 𑀭𑀲𑀪𑁄𑀕𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀪𑁄𑀕𑀸, 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦𑀺𑀩𑁆𑀪𑁄𑀕𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

(5.)

7. “𑀅𑀓𑀺𑀭𑀺𑀬𑀯𑀸𑀤𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀓𑀺𑀭𑀺𑀬𑀯𑀸𑀤𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀓𑀺𑀭𑀺𑀬𑀁 𑀯𑀤𑀸𑀫𑀺 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀲𑁆𑀲 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀲𑁆𑀲 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀲𑁆𑀲𑁇 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀓𑀺𑀭𑀺𑀬𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀓𑀺𑀭𑀺𑀬𑀯𑀸𑀤𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

(6.)

8. “𑀉𑀘𑁆𑀙𑁂𑀤𑀯𑀸𑀤𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀉𑀘𑁆𑀙𑁂𑀤𑀯𑀸𑀤𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀉𑀘𑁆𑀙𑁂𑀤𑀁 𑀯𑀤𑀸𑀫𑀺 𑀭𑀸𑀕𑀲𑁆𑀲 𑀤𑁄𑀲𑀲𑁆𑀲 𑀫𑁄𑀳𑀲𑁆𑀲𑁇 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀘𑁆𑀙𑁂𑀤𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀉𑀘𑁆𑀙𑁂𑀤𑀯𑀸𑀤𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

(7.)

9. “𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀻 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀻 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦𑁇 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀻 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

(8.)

10. “𑀯𑁂𑀦𑀬𑀺𑀓𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀯𑁂𑀦𑀬𑀺𑀓𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀺𑀦𑀬𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀺 𑀭𑀸𑀕𑀲𑁆𑀲 𑀤𑁄𑀲𑀲𑁆𑀲 𑀫𑁄𑀳𑀲𑁆𑀲𑁇 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀬𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀯𑁂𑀦𑀬𑀺𑀓𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

(9.)

11. “𑀢𑀧𑀲𑁆𑀲𑀻 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀢𑀧𑀲𑁆𑀲𑀻 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀢𑀧𑀦𑀻𑀬𑀸𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀯𑀤𑀸𑀫𑀺, 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁𑁇 𑀬𑀲𑁆𑀲 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀧𑀦𑀻𑀬𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸 𑀢𑀫𑀳𑀁 𑀢𑀧𑀲𑁆𑀲𑀻𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀧𑀦𑀻𑀬𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀢𑀧𑀲𑁆𑀲𑀻 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

(10.)

12. “𑀅𑀧𑀕𑀩𑁆𑀪𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁆𑀯𑁂𑀲, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀧𑀕𑀩𑁆𑀪𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀬𑀲𑁆𑀲 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀆𑀬𑀢𑀺𑀁 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀸 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸 𑀢𑀫𑀳𑀁 𑀅𑀧𑀕𑀩𑁆𑀪𑁄𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀆𑀬𑀢𑀺𑀁 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀸 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀳𑀻𑀦𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑀸 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑀸 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀬𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀅𑀧𑀕𑀩𑁆𑀪𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀯𑀤𑁂𑀲𑀺𑁇

(11.)

13. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀺𑀬𑀸 𑀅𑀡𑁆𑀟𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀯𑀸 𑀤𑀲 𑀯𑀸 𑀤𑁆𑀯𑀸𑀤𑀲 𑀯𑀸𑁇 𑀢𑀸𑀦𑀲𑁆𑀲𑀼 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀸 𑀅𑀥𑀺𑀲𑀬𑀺𑀢𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀲𑁂𑀤𑀺𑀢𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺𑁇 𑀬𑁄 𑀦𑀼 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀘𑁆𑀙𑀸𑀧𑀓𑀸𑀦𑀁 𑀧𑀞𑀫𑀢𑀭𑀁 𑀧𑀸𑀤𑀦𑀔𑀲𑀺𑀔𑀸𑀬 𑀯𑀸 𑀫𑀼𑀔𑀢𑀼𑀡𑁆𑀟𑀓𑁂𑀦 𑀯𑀸 𑀅𑀡𑁆𑀟𑀓𑁄𑀲𑀁 𑀧𑀤𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀢𑁆𑀣𑀺𑀦𑀸 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀪𑀺𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀲𑁆𑀯𑀸𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀚𑁂𑀝𑁆𑀞𑁄 𑀯𑀸 𑀓𑀦𑀺𑀝𑁆𑀞𑁄 𑀯𑀸’”𑀢𑀺? “𑀚𑁂𑀝𑁆𑀞𑁄𑀢𑀺𑀲𑁆𑀲, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀯𑀘𑀦𑀻𑀬𑁄𑁇 𑀲𑁄 𑀳𑀺 𑀦𑁂𑀲𑀁 𑀚𑁂𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀕𑀢𑀸𑀬 𑀧𑀚𑀸𑀬 𑀅𑀡𑁆𑀟𑀪𑀽𑀢𑀸𑀬 𑀧𑀭𑀺𑀬𑁄𑀦𑀤𑁆𑀥𑀸𑀬 𑀅𑀯𑀺𑀚𑁆𑀚𑀡𑁆𑀟𑀓𑁄𑀲𑀁 𑀧𑀤𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑁄𑀯 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀲𑁆𑀯𑀸𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀚𑁂𑀝𑁆𑀞𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀮𑁄𑀓𑀲𑁆𑀲𑁇

14. 𑀆𑀭𑀤𑁆𑀥𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀅𑀲𑀮𑁆𑀮𑀻𑀦𑀁, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀢𑀺 𑀅𑀲𑀫𑁆𑀫𑀼𑀝𑁆𑀞𑀸, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀓𑀸𑀬𑁄 𑀅𑀲𑀸𑀭𑀤𑁆𑀥𑁄, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀓𑀕𑁆𑀕𑀁𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄, 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀲𑀺𑀁, 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇

(12.)

15. 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁂 𑀅𑀦𑀗𑁆𑀕𑀡𑁂 𑀯𑀺𑀕𑀢𑀽𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀫𑀼𑀤𑀼𑀪𑀽𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀞𑀺𑀢𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀲𑀺𑀁𑁇 𑀲𑁄 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀏𑀓𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀁 𑀤𑁆𑀯𑁂𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀢𑀺𑀲𑁆𑀲𑁄𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀘𑀢𑀲𑁆𑀲𑁄𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀧𑀜𑁆𑀘𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀤𑀲𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀯𑀻𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀢𑀺𑀁𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀬𑁄 𑀚𑀸𑀢𑀺𑀲𑀢𑀫𑁆𑀧𑀺, 𑀚𑀸𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺, 𑀅𑀦𑁂𑀓𑁂𑀧𑀺 𑀲𑀁𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂 𑀅𑀦𑁂𑀓𑁂𑀧𑀺 𑀯𑀺𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂 𑀅𑀦𑁂𑀓𑁂𑀧𑀺 𑀲𑀁𑀯𑀝𑁆𑀝𑀯𑀺𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂—  ‘𑀅𑀫𑀼𑀢𑁆𑀭𑀸𑀲𑀺𑀁 𑀏𑀯𑀁𑀦𑀸𑀫𑁄 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑁄 𑀏𑀯𑀁𑀯𑀡𑁆𑀡𑁄 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑁄 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀏𑀯𑀫𑀸𑀬𑀼𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁄; 𑀲𑁄 𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀤𑀧𑀸𑀤𑀺𑀁; 𑀢𑀢𑁆𑀭𑀸𑀧𑀸𑀲𑀺𑀁 𑀏𑀯𑀁𑀦𑀸𑀫𑁄 𑀏𑀯𑀁𑀕𑁄𑀢𑁆𑀢𑁄 𑀏𑀯𑀁𑀯𑀡𑁆𑀡𑁄 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑁄 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀏𑀯𑀫𑀸𑀬𑀼𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁄; 𑀲𑁄 𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀇𑀥𑀽𑀧𑀧𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀲𑀸𑀓𑀸𑀭𑀁 𑀲𑁅𑀤𑁆𑀤𑁂𑀲𑀁 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀞𑀫𑁂 𑀬𑀸𑀫𑁂 𑀧𑀞𑀫𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀥𑀺𑀕𑀢𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀯𑀺𑀳𑀢𑀸, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀢𑀫𑁄 𑀯𑀺𑀳𑀢𑁄, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄—  𑀬𑀣𑀸 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑀞𑀫𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀪𑀺𑀤𑀸 𑀅𑀳𑁄𑀲𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀘𑁆𑀙𑀸𑀧𑀓𑀲𑁆𑀲𑁂𑀯 𑀅𑀡𑁆𑀟𑀓𑁄𑀲𑀫𑁆𑀳𑀸𑁇

(13.)

16. 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁂 𑀅𑀦𑀗𑁆𑀕𑀡𑁂 𑀯𑀺𑀕𑀢𑀽𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀫𑀼𑀤𑀼𑀪𑀽𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀞𑀺𑀢𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑀜𑀸𑀡𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀲𑀺𑀁𑁇 𑀲𑁄 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂, 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑁂 𑀯𑀢 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀉𑀧𑀯𑀸𑀤𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸; 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇 𑀇𑀫𑁂 𑀯𑀸 𑀧𑀦 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀸𑀤𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁆𑀫𑀲𑀫𑀸𑀤𑀸𑀦𑀸; 𑀢𑁂 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀯𑀫𑀸𑀦𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀳𑀻𑀦𑁂 𑀧𑀡𑀻𑀢𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑁂, 𑀬𑀣𑀸𑀓𑀫𑁆𑀫𑀽𑀧𑀕𑁂 𑀲𑀢𑁆𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀤𑀼𑀢𑀺𑀬𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀥𑀺𑀕𑀢𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀯𑀺𑀳𑀢𑀸, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀢𑀫𑁄 𑀯𑀺𑀳𑀢𑁄, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄—  𑀬𑀣𑀸 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀪𑀺𑀤𑀸 𑀅𑀳𑁄𑀲𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀘𑁆𑀙𑀸𑀧𑀓𑀲𑁆𑀲𑁂𑀯 𑀅𑀡𑁆𑀟𑀓𑁄𑀲𑀫𑁆𑀳𑀸𑁇

(14.)

17. 𑀲𑁄 𑀏𑀯𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁂 𑀅𑀦𑀗𑁆𑀕𑀡𑁂 𑀯𑀺𑀕𑀢𑀽𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁂 𑀫𑀼𑀤𑀼𑀪𑀽𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁂 𑀞𑀺𑀢𑁂 𑀆𑀦𑁂𑀜𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀜𑀸𑀡𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀲𑀺𑀁𑁇 𑀲𑁄 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁; ‘𑀇𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁, ‘𑀅𑀬𑀁 𑀆𑀲𑀯𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀯𑀁 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀓𑀸𑀫𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀪𑀯𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣𑁇 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀢𑀢𑀺𑀬𑀸 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀥𑀺𑀕𑀢𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀯𑀺𑀳𑀢𑀸, 𑀯𑀺𑀚𑁆𑀚𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀢𑀫𑁄 𑀯𑀺𑀳𑀢𑁄, 𑀆𑀮𑁄𑀓𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄—  𑀬𑀣𑀸 𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀢𑀺𑀬𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀪𑀺𑀤𑀸 𑀅𑀳𑁄𑀲𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀘𑁆𑀙𑀸𑀧𑀓𑀲𑁆𑀲𑁂𑀯 𑀅𑀡𑁆𑀟𑀓𑁄𑀲𑀫𑁆𑀳𑀸”𑀢𑀺𑁇

(15.)

18. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀚𑁂𑀝𑁆𑀞𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄, 𑀲𑁂𑀝𑁆𑀞𑁄 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑁄𑀢𑀸 𑀕𑁄𑀢𑀫𑁂𑀦 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀁𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀘 𑀫𑁂 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑀸𑀬𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

19. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀭𑀜𑁆𑀚𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀤𑁆𑀯𑀻𑀳𑀺𑀢𑀺𑀓𑀸 𑀲𑁂𑀢𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀮𑀸𑀓𑀸𑀯𑀼𑀢𑁆𑀢𑀸 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀉𑀜𑁆𑀙𑁂𑀦 𑀧𑀕𑁆𑀕𑀳𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀸𑀧𑀣𑀓𑀸 𑀅𑀲𑁆𑀲𑀯𑀸𑀡𑀺𑀚𑀸 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀲𑀢𑁂𑀳𑀺 𑀯𑁂𑀭𑀜𑁆𑀚𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀅𑀲𑁆𑀲𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀢𑁆𑀣𑀧𑀢𑁆𑀣𑀧𑀼𑀮𑀓𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀯𑁂𑀭𑀜𑁆𑀚𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑀸 𑀅𑀲𑁆𑀲𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸𑀲𑀼 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀣𑀧𑀢𑁆𑀣𑀧𑀼𑀮𑀓𑀁 𑀆𑀭𑀸𑀫𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀼𑀓𑁆𑀔𑀮𑁂 𑀓𑁄𑀝𑁆𑀝𑁂𑀢𑁆𑀯𑀸 𑀓𑁄𑀝𑁆𑀝𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀦𑀸𑀦𑀦𑁆𑀤𑁄 𑀧𑀢𑁆𑀣𑀧𑀼𑀮𑀓𑀁 𑀲𑀺𑀮𑀸𑀬𑀁 𑀧𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀢𑀺𑁇 𑀢𑀁 𑀪𑀕𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇

20. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀤𑀼𑀓𑁆𑀔𑀮𑀲𑀤𑁆𑀤𑀁𑁇 𑀚𑀸𑀦𑀦𑁆𑀢𑀸𑀧𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀚𑀸𑀦𑀦𑁆𑀢𑀸𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀓𑀸𑀮𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀓𑀸𑀮𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁𑁇 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁂 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀁 𑀯𑀸 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫, 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸𑀢𑀺𑁇

21. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀤𑀼𑀓𑁆𑀔𑀮𑀲𑀤𑁆𑀤𑁄”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀆𑀦𑀦𑁆𑀤𑁇 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀳𑀺 𑀯𑀺𑀚𑀺𑀢𑀁𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸 𑀲𑀸𑀮𑀺𑀫𑀁𑀲𑁄𑀤𑀦𑀁 𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

(17.)

22. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀭𑀜𑁆𑀚𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀸 𑀤𑁆𑀯𑀻𑀳𑀺𑀢𑀺𑀓𑀸 𑀲𑁂𑀢𑀝𑁆𑀞𑀺𑀓𑀸 𑀲𑀮𑀸𑀓𑀸𑀯𑀼𑀢𑁆𑀢𑀸𑁇 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀉𑀜𑁆𑀙𑁂𑀦 𑀧𑀕𑁆𑀕𑀳𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀇𑀫𑀺𑀲𑁆𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀬𑀸 𑀳𑁂𑀝𑁆𑀞𑀺𑀫𑀢𑀮𑀁 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀔𑀼𑀤𑁆𑀤𑀫𑀥𑀼𑀁 𑀅𑀦𑀻𑀮𑀓𑀁; 𑀏𑀯𑀫𑀲𑁆𑀲𑀸𑀤𑀁𑁇 𑀲𑀸𑀥𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀣𑀯𑀺𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀧𑁆𑀧𑀝𑀓𑁄𑀚𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑁂, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀧𑀣𑀯𑀺𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀓𑀣𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? “𑀏𑀓𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀸𑀡𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀲𑁆𑀲𑀸𑀫𑀺—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀧𑀣𑀯𑀻𑁇 𑀬𑁂 𑀧𑀣𑀯𑀺𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀧𑀸𑀡𑀸 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀲𑀗𑁆𑀓𑀸𑀫𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀏𑀓𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀣𑀯𑀺𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀫𑀸 𑀢𑁂 𑀭𑀼𑀘𑁆𑀘𑀺 𑀧𑀣𑀯𑀺𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑀼𑀁𑁇 𑀯𑀺𑀧𑀮𑁆𑀮𑀸𑀲𑀫𑁆𑀧𑀺 𑀲𑀢𑁆𑀢𑀸 𑀧𑀝𑀺𑀮𑀪𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀩𑁆𑀩𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀉𑀢𑁆𑀢𑀭𑀓𑀼𑀭𑀼𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀫𑀸 𑀢𑁂 𑀭𑀼𑀘𑁆𑀘𑀺 𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀓𑀼𑀭𑀼𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀕𑀫𑀦𑀦𑁆”𑀢𑀺𑁇

23. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀓𑀢𑀫𑁂𑀲𑀸𑀦𑀁 𑀔𑁄 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀦 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀅𑀳𑁄𑀲𑀺; 𑀓𑀢𑀫𑁂𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺?

26. “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀘 𑀲𑀺𑀔𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀘 𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀦 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀪𑀕𑀯𑀸 𑀘, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸 𑀘 𑀲𑀺𑀔𑀻 𑀪𑀕𑀯𑀸 𑀘 𑀯𑁂𑀲𑁆𑀲𑀪𑀽 𑀓𑀺𑀮𑀸𑀲𑀼𑀦𑁄 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼𑀁𑁇 𑀅𑀧𑁆𑀧𑀓𑀜𑁆𑀘 𑀦𑁂𑀲𑀁 𑀅𑀳𑁄𑀲𑀺 𑀲𑀼𑀢𑁆𑀢𑀁 𑀕𑁂𑀬𑁆𑀬𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀕𑀸𑀣𑀸 𑀉𑀤𑀸𑀦𑀁 𑀇𑀢𑀺𑀯𑀼𑀢𑁆𑀢𑀓𑀁 𑀚𑀸𑀢𑀓𑀁 𑀅𑀩𑁆𑀪𑀼𑀢𑀥𑀫𑁆𑀫𑀁 𑀯𑁂𑀤𑀮𑁆𑀮𑀁𑁇 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁𑁇 𑀅𑀦𑀼𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁𑁇 𑀢𑁂𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑁂𑀦 𑀬𑁂 𑀢𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀲𑀸𑀯𑀓𑀸 𑀦𑀸𑀦𑀸𑀦𑀸𑀫𑀸 𑀦𑀸𑀦𑀸𑀕𑁄𑀢𑁆𑀢𑀸 𑀦𑀸𑀦𑀸𑀚𑀘𑁆𑀘𑀸 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀢𑁂 𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀜𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀲𑀼𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀦𑀸𑀦𑀸𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺 𑀨𑀮𑀓𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀯𑀸𑀢𑁄 𑀯𑀺𑀓𑀺𑀭𑀢𑀺 𑀯𑀺𑀥𑀫𑀢𑀺 𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀬𑀣𑀸 𑀢𑀁 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀢𑁆𑀢𑀸𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀢𑁂𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑁂𑀦 𑀬𑁂 𑀢𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀲𑀸𑀯𑀓𑀸 𑀦𑀸𑀦𑀸𑀦𑀸𑀫𑀸 𑀦𑀸𑀦𑀸𑀕𑁄𑀢𑁆𑀢𑀸 𑀦𑀸𑀦𑀸𑀚𑀘𑁆𑀘𑀸 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀢𑁂 𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀜𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀧𑁂𑀲𑀼𑀁𑁇

(22.)

35. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀬𑁂𑀳𑀺 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁂 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀯𑁂𑀭𑀜𑁆𑀚𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀅𑀧𑀮𑁄𑀓𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇

36. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀧𑀘𑁆𑀙𑀸𑀲𑀫𑀡𑁂𑀦 𑀬𑁂𑀦 𑀯𑁂𑀭𑀜𑁆𑀚𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀫𑁆𑀳 𑀢𑀬𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸, 𑀅𑀧𑀮𑁄𑀓𑁂𑀫 𑀢𑀁, 𑀇𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀘𑁆𑀘𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀢𑁆𑀣 𑀫𑀬𑀸 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸; 𑀅𑀧𑀺 𑀘 𑀬𑁄 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄 𑀲𑁄 𑀦 𑀤𑀺𑀦𑁆𑀦𑁄𑁇 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀁, 𑀦𑁄𑀧𑀺 𑀅𑀤𑀸𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸, 𑀢𑀁 𑀓𑀼𑀢𑁂𑀢𑁆𑀣 𑀮𑀩𑁆𑀪𑀸 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑀸 𑀖𑀭𑀸𑀯𑀸𑀲𑀸 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑀸𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀫𑁂 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑁂𑀭𑀜𑁆𑀚𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

37. 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀓𑁂 𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

(23.)

38. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀯𑁂𑀭𑀜𑁆𑀚𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀭𑀜𑁆𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑀺, 𑀏𑀓𑀫𑁂𑀓𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀓𑀫𑁂𑀓𑁂𑀦 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑁂𑀭𑀜𑁆𑀚𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

39. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑁂𑀭𑀜𑁆𑀚𑀸𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀕𑀫𑁆𑀫 𑀲𑁄𑀭𑁂𑀬𑁆𑀬𑀁 𑀲𑀗𑁆𑀓𑀲𑁆𑀲𑀁 𑀓𑀡𑁆𑀡𑀓𑀼𑀚𑁆𑀚𑀁 𑀬𑁂𑀦 𑀧𑀬𑀸𑀕𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀬𑀸𑀕𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑁂 𑀕𑀗𑁆𑀕𑀁 𑀦𑀤𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀦𑁆𑀢𑀺𑁇

---

40. 𑀯𑁂𑀭𑀜𑁆𑀚𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
