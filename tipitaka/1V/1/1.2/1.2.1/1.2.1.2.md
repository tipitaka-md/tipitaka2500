# 𑁧𑁇𑁨𑁇𑁧𑁇𑁨 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

(263.)

945\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦𑀸”𑀢𑀺𑁇 (𑁧)

946\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨)

947\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩)

948\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪)

949\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑁂𑀦 𑀦𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀓𑀺𑀁𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫)

950\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑁂𑀦 𑀦𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁬)

951\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑁂𑀦 𑀦𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭)

952\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀯𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑁂𑀲𑀚𑁆𑀚𑁂𑀦 𑀆𑀮𑀺𑀫𑁆𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮)

953\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀯𑀡𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀲𑀚𑁆𑀚𑁂𑀦 𑀆𑀮𑀺𑀫𑁆𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯--𑁧𑁦)

954\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀡𑁆𑀟𑀁 𑀓𑀡𑁆𑀟𑀼𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁧)

955\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀅𑀡𑁆𑀟𑀁 𑀓𑀡𑁆𑀟𑀼𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁨--𑁧𑁩)

(264.)

956\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀕𑁆𑀕𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪)

957\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁫--𑁧𑁬)

958\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁭)

959\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁮--𑁧𑁯)

960\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑁂 𑀉𑀤𑀭𑀯𑀝𑁆𑀝𑀺𑀁 𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁦)

961\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑁂 𑀉𑀤𑀭𑀯𑀝𑁆𑀝𑀺𑀁 𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁧--𑁨𑁨)

962\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀲𑁆𑀲 𑀧𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁩)

963\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀲𑁆𑀲 𑀧𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁪--𑁨𑁫)

(265.)

964\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀊𑀭𑀼𑀁 𑀖𑀝𑁆𑀝𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁬)

965\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀊𑀭𑀼𑀁 𑀖𑀝𑁆𑀝𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁭--𑁨𑁮)

966\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀫𑀡𑁂𑀭, 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁯)

967\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁦)

972\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀁 𑀣𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩𑁮--𑁩𑁯)

988\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀤𑀓𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀑𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁪)

989\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀉𑀤𑀓𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀑𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁫--𑁬𑁬)

990\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀲𑀬𑀦𑁂 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀖𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁭--𑁬𑁮)

991\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀅𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀖𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺…𑀧𑁂…  𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬𑁯--𑁭𑁦)
