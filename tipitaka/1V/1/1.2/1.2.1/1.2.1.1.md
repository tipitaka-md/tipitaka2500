# 𑁧𑁇𑁨𑁇𑁧𑁇𑁧 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

942\. _𑀲𑀼𑀧𑀺𑀦𑁄𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑁄,_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑀼𑀡𑁆𑀳𑁄𑀤𑀓𑁂𑀦 𑀘;_  
_𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀓𑀡𑁆𑀟𑀼𑀯𑀁 𑀫𑀕𑁆𑀕𑁄,_  
_𑀯𑀢𑁆𑀣𑀺 𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑀼𑀧𑀓𑁆𑀓𑀫𑁄𑁇_  


943\. _𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀘 𑀲𑀼𑀢𑁆𑀢𑁄 𑀘,_  
_𑀊𑀭𑀼 𑀫𑀼𑀝𑁆𑀞𑀺𑀦𑀸 𑀧𑀻𑀍𑀅𑀬𑀺;_  
_𑀆𑀓𑀸𑀲𑁂 𑀣𑀫𑁆𑀪𑀁 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀺,_  
_𑀙𑀺𑀤𑁆𑀤𑀁 𑀓𑀝𑁆𑀞𑁂𑀦 𑀖𑀝𑁆𑀝𑀬𑀺𑁇_  


944\. _𑀲𑁄𑀢𑁄 𑀉𑀤𑀜𑁆𑀚𑀮𑀁 𑀥𑀸𑀯𑀁,_  
_𑀧𑀼𑀧𑁆𑀨𑀸𑀯𑀮𑀺𑀬𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀁;_  
_𑀯𑀸𑀮𑀺𑀓𑀸 𑀓𑀤𑁆𑀤𑀫𑀼𑀲𑁆𑀲𑁂𑀓𑁄,_  
_𑀲𑀬𑀦𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_
