# 𑁧𑁇𑁨𑁇𑁪 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(290.)

1180\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀓𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀫𑀢𑀧𑀢𑀺𑀓𑀸 𑀅𑀪𑀺𑀭𑀽𑀧𑀸 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀢𑀲𑁆𑀲𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀇𑀢𑁆𑀣𑀻 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀇𑀢𑁆𑀣𑀻 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀤𑀸𑀬𑀺𑀦𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀅𑀢𑁆𑀣𑁄𑁇 𑀧𑀝𑀺𑀩𑀮𑀸 𑀫𑀬𑀁 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀤𑀸𑀢𑀼𑀁 𑀬𑀤𑀺𑀤𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀦𑁆”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1181\. “𑀦 𑀔𑁄 𑀢𑁂, 𑀪𑀕𑀺𑀦𑀺, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑀼𑀮𑁆𑀮𑀪𑀸 𑀬𑀤𑀺𑀤𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑁇 𑀅𑀧𑀺 𑀘 𑀬𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀢𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺? “𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑁄, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺? “𑀅𑀢𑁆𑀣𑁄, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺, 𑀑𑀯𑀭𑀓𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀝𑀓𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀫𑀜𑁆𑀘𑀓𑁂 𑀉𑀢𑁆𑀢𑀸𑀦𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀬𑁂𑀦 𑀲𑀸 𑀇𑀢𑁆𑀣𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸—  “𑀓𑁄 𑀇𑀫𑀁 𑀯𑀲𑀮𑀁 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀆𑀫𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀦𑀺𑀝𑁆𑀞𑀼𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1182\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀇𑀢𑁆𑀣𑀻 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁, 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁, 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁, 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀲𑀸𑀫𑀜𑁆𑀜𑀸 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀸𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑁄 𑀉𑀤𑀸𑀬𑀻 𑀫𑀁 𑀲𑀸𑀫𑀁 𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀁 𑀬𑀸𑀘𑀺𑀢𑁆𑀯𑀸, ‘𑀓𑁄 𑀇𑀫𑀁 𑀯𑀲𑀮𑀁 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀆𑀫𑀲𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑀺𑀝𑁆𑀞𑀼𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀺𑀁 𑀫𑁂 𑀧𑀸𑀧𑀓𑀁 𑀓𑀺𑀁 𑀫𑁂 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁, 𑀓𑀲𑁆𑀲𑀸𑀳𑀁 𑀓𑁂𑀦 𑀳𑀸𑀬𑀸𑀫𑀻”𑀢𑀺?

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1183\. 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑁄 𑀉𑀤𑀸𑀬𑀻 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀲𑀸𑀫𑀁 𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀁 𑀬𑀸𑀘𑀺𑀢𑁆𑀯𑀸, ‘𑀓𑁄 𑀇𑀫𑀁 𑀯𑀲𑀮𑀁 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀆𑀫𑀲𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑀺𑀝𑁆𑀞𑀼𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀺𑀁 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀧𑀸𑀧𑀓𑀁 𑀓𑀺𑀁 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁, 𑀓𑀲𑁆𑀲𑀸𑀬𑀁 𑀓𑁂𑀦 𑀳𑀸𑀬𑀢𑀻”𑀢𑀺?

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1184\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀸𑀲𑀁 𑀇𑀢𑁆𑀣𑀻𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀻𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀻𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

1185\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄 𑀦𑁄 𑀲𑀭𑀸𑀕𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄? 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(291.)

1186\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀑𑀢𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑁂𑀬𑁆𑀬—  ‘𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀬𑀸 𑀫𑀸𑀤𑀺𑀲𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁 𑀏𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀧𑀭𑀺𑀘𑀭𑁂𑀬𑁆𑀬𑀸𑀢𑀺 𑀫𑁂𑀣𑀼𑀦𑀼𑀧𑀲𑀁𑀳𑀺𑀢𑁂𑀦’, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”**𑀢𑀺𑁇 (*𑁪*:**𑁮**)

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(292.)

1187\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

1188\. **𑀑𑀢𑀺𑀡𑁆𑀡𑁄** 𑀦𑀸𑀫 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑁄𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

1189\. **𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑀦𑁆**𑀢𑀺 𑀭𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑀁, 𑀤𑀼𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑀁, 𑀫𑀽𑀍𑀳𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑀁𑁇 𑀅𑀧𑀺 𑀘 𑀭𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑀦𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1190\. **𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄** 𑀦𑀸𑀫 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀢𑁆𑀣𑀻, 𑀦 𑀬𑀓𑁆𑀔𑀻, 𑀦 𑀧𑁂𑀢𑀻, 𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀝𑀺𑀩𑀮𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀚𑀸𑀦𑀺𑀢𑀼𑀁𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1191\. **𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂**𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀅𑀯𑀺𑀤𑀽𑀭𑁂𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1192\. **𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀦𑁆**𑀢𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀓𑀸𑀫𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀳𑁂𑀢𑀼𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀁𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1193\. **𑀏𑀢𑀤𑀕𑁆𑀕𑀦𑁆**𑀢𑀺 𑀏𑀢𑀁 𑀅𑀕𑁆𑀕𑀁 𑀏𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑀁 𑀏𑀢𑀁 𑀫𑁄𑀓𑁆𑀔𑀁 𑀏𑀢𑀁 𑀉𑀢𑁆𑀢𑀫𑀁 𑀏𑀢𑀁 𑀧𑀯𑀭𑀁𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕
