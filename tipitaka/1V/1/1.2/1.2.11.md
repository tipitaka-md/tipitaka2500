# 𑁧𑁇𑁨𑁇𑁧𑁧 𑀪𑁂𑀤𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀪𑁂𑀤𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(417.)

1704. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀺 𑀘𑀓𑁆𑀓𑀪𑁂𑀤𑀸𑀬𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄, 𑀅𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀓𑀪𑁂𑀤𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀓𑁄𑀓𑀸𑀮𑀺𑀓𑁄 𑀓𑀝𑀫𑁄𑀤𑀓𑀢𑀺𑀲𑁆𑀲𑀓𑁄 𑀔𑀡𑁆𑀟𑀤𑁂𑀯𑀺𑀬𑀸 𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑀼𑀤𑁆𑀤𑀤𑀢𑁆𑀢𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀅𑀯𑀘𑀼𑀢𑁆𑀣𑁇 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄, 𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀙𑀦𑁆𑀤𑀜𑁆𑀘 𑀭𑀼𑀘𑀺𑀜𑁆𑀘 𑀆𑀤𑀸𑀬 𑀯𑁄𑀳𑀭𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺 𑀦𑁄 𑀪𑀸𑀲𑀢𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑀫𑀢𑀻”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸”𑀢𑀺𑁇

1705. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(418.)

1706. **“𑀢𑀲𑁆𑀲𑁂𑀯 𑀔𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸 𑀏𑀓𑁄 𑀯𑀸 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀬𑁄 𑀯𑀸𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀯𑀘𑀼𑀢𑁆𑀣𑁇 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀜𑁆𑀘 𑀭𑀼𑀘𑀺𑀜𑁆𑀘 𑀆𑀤𑀸𑀬 𑀯𑁄𑀳𑀭𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺 𑀦𑁄 𑀪𑀸𑀲𑀢𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑀫𑀢𑀻’𑀢𑀺, 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀅𑀯𑀘𑀼𑀢𑁆𑀣, 𑀦 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻, 𑀦 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻, 𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀭𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣, 𑀲𑀫𑁂𑀢𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀁𑀖𑁂𑀦, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀳𑀺 𑀲𑀁𑀖𑁄 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑁄 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑁄 𑀏𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 𑀢𑀣𑁂𑀯 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀜𑁆𑀘𑁂 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑀸 𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”**𑀢𑀺𑁇 (*𑁧𑁧*:**𑁧𑁫**)

(419.)

1707. **𑀢𑀲𑁆𑀲𑁂𑀯 𑀔𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇

1708. **𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑁄𑀦𑁆𑀢𑀻**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

1709. **𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸**𑀢𑀺 𑀬𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀬𑀁𑀔𑀦𑁆𑀢𑀺𑀓𑁄 𑀬𑀁𑀭𑀼𑀘𑀺𑀓𑁄 𑀢𑁂𑀧𑀺 𑀢𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀢𑀁𑀔𑀦𑁆𑀢𑀺𑀓𑀸 𑀢𑀁𑀭𑀼𑀘𑀺𑀓𑀸𑁇

1710. **𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸**𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀸𑀬 𑀧𑀓𑁆𑀔𑀸𑀬 𑀞𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

1711. **𑀏𑀓𑁄 𑀯𑀸 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀬𑁄 𑀯𑀸**𑀢𑀺 𑀏𑀓𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀤𑁆𑀯𑁂 𑀯𑀸 𑀢𑀬𑁄 𑀯𑀸𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀯𑀘𑀼𑀢𑁆𑀣, 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀦𑁆𑀤𑀜𑁆𑀘 𑀭𑀼𑀘𑀺𑀜𑁆𑀘 𑀆𑀤𑀸𑀬 𑀯𑁄𑀳𑀭𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺 𑀦𑁄 𑀪𑀸𑀲𑀢𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑁂𑀢𑀁 𑀔𑀫𑀢𑀻”𑀢𑀺𑁇

1712. **𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺 𑀬𑁂 𑀢𑁂 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑁇

1713. **𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺𑁇

1714. 𑀬𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀬𑁂 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀢𑁂𑀳𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀅𑀯𑀘𑀼𑀢𑁆𑀣𑁇 𑀦 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻, 𑀦 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻𑁇 𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀭𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣𑁇 𑀲𑀫𑁂𑀢𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀁𑀖𑁂𑀦𑁇 𑀲𑀫𑀕𑁆𑀕𑁄 𑀳𑀺 𑀲𑀁𑀖𑁄 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑁄 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑁄 𑀏𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀯𑀤𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑀫𑁆𑀧𑀺 𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀅𑀯𑀘𑀼𑀢𑁆𑀣𑁇 𑀦 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻, 𑀦 𑀘𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀦𑀬𑀯𑀸𑀤𑀻𑁇 𑀫𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀭𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀣𑁇 𑀲𑀫𑁂𑀢𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀁𑀖𑁂𑀦𑁇 𑀲𑀫𑀕𑁆𑀕𑁄 𑀳𑀺 𑀲𑀁𑀖𑁄 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑁄 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑁄 𑀏𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(420.)

1715. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀇𑀢𑁆𑀣𑀦𑀸𑀫𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸𑁇 𑀢𑁂 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀜𑁆𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑁂𑀬𑁆𑀬 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1716. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸𑁇 𑀢𑁂 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀜𑁆𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1717. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀯𑀕𑁆𑀕𑀯𑀸𑀤𑀓𑀸𑁇 𑀢𑁂 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀜𑁆𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1718. 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(421.)

1719. 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸, 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀢𑀬𑁄 𑀏𑀓𑀢𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸, 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺 𑀦 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇

1720. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄**𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺𑁇

(422.)

1721. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1722. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1723. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1724. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1725. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1726. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(423.)

1727. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀸𑀦𑀁, 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀸𑀦𑀁, 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁, 𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑀸𑀦𑀁, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀦𑁆𑀢𑀺𑁇

---

1728. 𑀪𑁂𑀤𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇
