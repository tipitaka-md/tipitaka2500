# 𑁧𑁇𑁨𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑀤𑀼𑀝𑁆𑀞𑀤𑁄𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀢𑀺𑀬𑀤𑀼𑀝𑁆𑀞𑀤𑁄𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(391.)

1626. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑁂𑀢𑁆𑀢𑀺𑀬𑀪𑀽𑀫𑀚𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀸 𑀧𑀩𑁆𑀩𑀢𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑀸 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀙𑀕𑀮𑀓𑀁 𑀅𑀚𑀺𑀓𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀁 𑀙𑀕𑀮𑀓𑀁 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀦𑀸𑀫 𑀓𑀭𑁄𑀫𑁇 𑀇𑀫𑀁 𑀅𑀚𑀺𑀓𑀁 𑀫𑁂𑀢𑁆𑀢𑀺𑀬𑀁 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀓𑀭𑁄𑀫𑁇 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀯𑁄𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀯𑁄𑀘𑀼𑀫𑁆𑀳𑀸𑁇 𑀇𑀤𑀸𑀦𑀺 𑀧𑀦 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑁂𑀢𑁆𑀢𑀺𑀬𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀙𑀕𑀮𑀓𑀁 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀦𑀸𑀫 𑀅𑀓𑀁𑀲𑀼𑁇 𑀢𑀁 𑀅𑀚𑀺𑀓𑀁 𑀫𑁂𑀢𑁆𑀢𑀺𑀬𑀁 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀓𑀁𑀲𑀼𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁—  “𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀢𑁂𑀦 𑀅𑀯𑁄𑀘𑀼𑀫𑁆𑀳𑀸𑁇 𑀇𑀤𑀸𑀦𑀺 𑀧𑀦 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑁂𑀢𑁆𑀢𑀺𑀬𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀸𑀯𑀼𑀲𑁄, 𑀏𑀯𑀁 𑀅𑀯𑀘𑀼𑀢𑁆𑀣𑁇 𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀤𑀩𑁆𑀩𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1627. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀭𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀤𑀩𑁆𑀩, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀢𑁆𑀢𑀸 𑀬𑀣𑀬𑀺𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀳𑀁𑀲𑀽”𑀢𑀺? “𑀬𑀣𑀸 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀭𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀤𑀩𑁆𑀩, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀢𑁆𑀢𑀸 𑀬𑀣𑀬𑀺𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀳𑀁𑀲𑀽”𑀢𑀺? “𑀬𑀣𑀸 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇 “𑀦 𑀔𑁄, 𑀤𑀩𑁆𑀩, 𑀤𑀩𑁆𑀩𑀸 𑀏𑀯𑀁 𑀦𑀺𑀩𑁆𑀩𑁂𑀞𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀘𑁂 𑀢𑀬𑀸 𑀓𑀢𑀁 𑀓𑀢𑀦𑁆𑀢𑀺 𑀯𑀤𑁂𑀳𑀺, 𑀲𑀘𑁂 𑀢𑀬𑀸 𑀅𑀓𑀢𑀁 𑀅𑀓𑀢𑀦𑁆𑀢𑀺 𑀯𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀬𑀢𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀢𑁄 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑀸, 𑀧𑀕𑁂𑀯 𑀚𑀸𑀕𑀭𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀣𑀸”𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

1628. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑁂𑀢𑁆𑀢𑀺𑀬𑀪𑀽𑀫𑀚𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀻𑀬𑀫𑀸𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁—  “𑀓𑀺𑀁 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑁂𑀲𑀁 𑀮𑁂𑀲𑀫𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀣𑀸”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀫𑁂𑀢𑁆𑀢𑀺𑀬𑀪𑀽𑀫𑀚𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀤𑀩𑁆𑀩𑀁 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑁂𑀲𑀁 𑀮𑁂𑀲𑀫𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1668. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀤𑀼𑀓𑁆𑀓𑀝𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀘𑁄𑀤𑀸𑀧𑁂𑀢𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺…𑀧𑁂…  𑀏𑀯𑀫𑁆𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺 𑀮𑁂𑀲𑁄 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀯𑀸𑀘𑀸𑀬, 𑀯𑀸𑀘𑀸𑀬 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1669. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑁂 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀤𑀼𑀓𑁆𑀓𑀝𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀘𑁄𑀤𑀸𑀧𑁂𑀢𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺…𑀧𑁂…  𑀏𑀯𑀫𑁆𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺 𑀮𑁂𑀲𑁄 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀯𑀸𑀘𑀸𑀬, 𑀯𑀸𑀘𑀸𑀬 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1670. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁…𑀧𑁂…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁…  𑀤𑀼𑀓𑁆𑀓𑀝𑀁…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑁂 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺…  𑀤𑀼𑀓𑁆𑀓𑀝𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀘𑁄𑀤𑀸𑀧𑁂𑀢𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑁄𑀲𑀺, 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑀲𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀉𑀧𑁄𑀲𑀣𑁄 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀸 𑀯𑀸 𑀲𑀁𑀖𑀓𑀫𑁆𑀫𑀁 𑀯𑀸”𑀢𑀺, 𑀏𑀯𑀫𑁆𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺 𑀮𑁂𑀲𑁄 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀯𑀸𑀘𑀸𑀬, 𑀯𑀸𑀘𑀸𑀬 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

(408.)

1671. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀢𑀣𑀸𑀲𑀜𑁆𑀜𑀻 𑀘𑁄𑀤𑁂𑀢𑀺 𑀯𑀸 𑀘𑁄𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
