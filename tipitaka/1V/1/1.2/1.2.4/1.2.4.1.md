# 𑁧𑁇𑁨𑁇𑁪𑁇𑁧 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑀉𑀤𑁆𑀤𑀸𑀦𑀕𑀸𑀣𑀸

1211\. _𑀓𑀣𑀁 𑀯𑀜𑁆𑀛𑀸 𑀮𑀪𑁂 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀧𑀺𑀬𑀸 𑀘 𑀲𑀼𑀪𑀕𑀸 𑀲𑀺𑀬𑀁;_  
_𑀓𑀺𑀁 𑀤𑀚𑁆𑀚𑀁 𑀓𑁂𑀦𑀼𑀧𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀁,_  
_𑀓𑀣𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀕𑁆𑀕𑀢𑀺𑀦𑁆𑀢𑀺𑁇_
