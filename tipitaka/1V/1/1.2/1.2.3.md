# 𑁧𑁇𑁨𑁇𑁩 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(283.)

1133\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁄𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀆𑀭𑀸𑀫𑀁 𑀆𑀕𑀫𑀁𑀲𑀼 𑀯𑀺𑀳𑀸𑀭𑀧𑁂𑀓𑁆𑀔𑀺𑀓𑀸𑀬𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀢𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑁂𑀓𑁆𑀔𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀲𑀁 𑀇𑀢𑁆𑀣𑀻𑀦𑀁 𑀯𑀘𑁆𑀘𑀫𑀕𑁆𑀕𑀁 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑀕𑁆𑀕𑀁 𑀆𑀤𑀺𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀫𑁆𑀧𑀺 𑀪𑀡𑀢𑀺 𑀅𑀯𑀡𑁆𑀡𑀫𑁆𑀧𑀺 𑀪𑀡𑀢𑀺 𑀬𑀸𑀘𑀢𑀺𑀧𑀺 𑀆𑀬𑀸𑀘𑀢𑀺𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀢𑀺𑀧𑀺 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺𑀧𑀺 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀺𑀧𑀺 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺𑀧𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺𑀧𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀙𑀺𑀦𑁆𑀦𑀺𑀓𑀸 𑀥𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀅𑀳𑀺𑀭𑀺𑀓𑀸𑀬𑁄 𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀤𑀸𑀬𑀺𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀉𑀳𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀮𑁆𑀮𑀧𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀬𑀸 𑀧𑀦 𑀢𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀳𑀺𑀭𑀺𑀫𑀦𑀸 𑀢𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺—  “𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀦𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁𑁇 𑀲𑀸𑀫𑀺𑀓𑁂𑀦𑀧𑀺 𑀫𑀬𑀁 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑀸 𑀦 𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫, 𑀓𑀺𑀁 𑀧𑀦𑀸𑀬𑁆𑀬𑁂𑀦 𑀉𑀤𑀸𑀬𑀺𑀦𑀸”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1134\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄 𑀦𑁄 𑀲𑀭𑀸𑀕𑀸𑀬…𑀧𑁂…  𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(284.)

1135\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀑𑀢𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀧𑀭𑀺𑀡𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀑𑀪𑀸𑀲𑁂𑀬𑁆𑀬 𑀬𑀣𑀸 𑀢𑀁 𑀬𑀼𑀯𑀸 𑀬𑀼𑀯𑀢𑀺𑀁 𑀫𑁂𑀣𑀼𑀦𑀼𑀧𑀲𑀁𑀳𑀺𑀢𑀸𑀳𑀺, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”**𑀢𑀺𑁇 (*𑁩*:**𑁭**)

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(285.)

1136\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1137\. **𑀑𑀢𑀺𑀡𑁆𑀡𑁄** 𑀦𑀸𑀫 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑁄𑁇
