# 𑁧𑁇𑁨𑁇𑁨 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(269.)

993. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁄, 𑀫𑀚𑁆𑀛𑁂𑀕𑀩𑁆𑀪𑁄, 𑀲𑀫𑀦𑁆𑀢𑀸𑀧𑀭𑀺𑀬𑀸𑀕𑀸𑀭𑁄, 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀫𑀜𑁆𑀘𑀧𑀻𑀞𑀁 𑀪𑀺𑀲𑀺𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀁, 𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀲𑀼𑀲𑀫𑁆𑀫𑀝𑁆𑀞𑀁𑁇 𑀩𑀳𑀽 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀧𑁂𑀓𑁆𑀔𑀓𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

994. 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁 𑀪𑁄𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀧𑁂𑀓𑁆𑀔𑀲𑁆𑀲𑀽”𑀢𑀺, 𑀅𑀯𑀸𑀧𑀼𑀭𑀡𑀁 𑀆𑀤𑀸𑀬 𑀖𑀝𑀺𑀓𑀁 𑀉𑀕𑁆𑀖𑀸𑀝𑁂𑀢𑁆𑀯𑀸 𑀓𑀯𑀸𑀝𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀲𑁄𑀧𑀺 𑀔𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀲𑀸𑀧𑀺 𑀔𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑀸𑀢𑀧𑀸𑀦𑁂 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑁄 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑀸𑀢𑀧𑀸𑀦𑁂 𑀣𑀓𑁂𑀦𑁆𑀢𑁄 𑀕𑀩𑁆𑀪𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀧𑀭𑀸𑀫𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀤𑀸𑀬𑀺𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑀯𑀸𑀘𑀁 𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑁂𑀲𑀺—  “𑀉𑀍𑀆𑀭𑀸 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀬𑁂 𑀇𑀫𑁂 𑀏𑀯𑀭𑀽𑀧𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀪𑀯𑀫𑁆𑀧𑀺 𑀉𑀤𑀸𑀬𑀻 𑀉𑀍𑀆𑀭𑁄 𑀬𑁄 𑀏𑀯𑀭𑀽𑀧𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

995. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀼𑀢𑁄 𑀢𑀲𑁆𑀲 𑀉𑀍𑀆𑀭𑀢𑁆𑀢𑀢𑀸𑁇 𑀬𑀣𑁂𑀯 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀧𑀭𑀸𑀫𑀲𑀺 𑀏𑀯𑀫𑁂𑀯 𑀫𑁂 𑀲𑀫𑀡𑁄 𑀉𑀤𑀸𑀬𑀻 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀧𑀭𑀸𑀫𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁, 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁, 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁, 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀲𑀸𑀫𑀜𑁆𑀜𑀸 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀸𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑁄 𑀉𑀤𑀸𑀬𑀻 𑀫𑀫 𑀪𑀭𑀺𑀬𑀸𑀬 𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀧𑀭𑀸𑀫𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦 𑀳𑀺 𑀲𑀓𑁆𑀓𑀸 𑀓𑀼𑀮𑀺𑀢𑁆𑀣𑀻𑀳𑀺 𑀓𑀼𑀮𑀥𑀻𑀢𑀸𑀳𑀺 𑀓𑀼𑀮𑀓𑀼𑀫𑀸𑀭𑀻𑀳𑀺 𑀓𑀼𑀮𑀲𑀼𑀡𑁆𑀳𑀸𑀳𑀺 𑀓𑀼𑀮𑀤𑀸𑀲𑀻𑀳𑀺 𑀆𑀭𑀸𑀫𑀁 𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀸 𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀲𑀘𑁂 𑀓𑀼𑀮𑀺𑀢𑁆𑀣𑀺𑀬𑁄 𑀓𑀼𑀮𑀥𑀻𑀢𑀭𑁄 𑀓𑀼𑀮𑀓𑀼𑀫𑀸𑀭𑀺𑀬𑁄 𑀓𑀼𑀮𑀲𑀼𑀡𑁆𑀳𑀸𑀬𑁄 𑀓𑀼𑀮𑀤𑀸𑀲𑀺𑀬𑁄 𑀆𑀭𑀸𑀫𑀁 𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀸𑀧𑀺 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀽𑀲𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

996. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇
