# 𑁧𑁇𑁨𑁇𑁧𑁩 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(431.)

1753\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸 𑀦𑀸𑀫 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀽𑁇 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀦𑁆𑀢𑀺—  𑀫𑀸𑀮𑀸𑀯𑀘𑁆𑀙𑀁 𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀭𑁄𑀧𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺𑀧𑀺 𑀲𑀺𑀜𑁆𑀘𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀑𑀘𑀺𑀦𑀦𑁆𑀢𑀺𑀧𑀺 𑀑𑀘𑀺𑀦𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀕𑀦𑁆𑀣𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀕𑀦𑁆𑀣𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀏𑀓𑀢𑁄𑀯𑀡𑁆𑀝𑀺𑀓𑀫𑀸𑀮𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀪𑀢𑁄𑀯𑀡𑁆𑀝𑀺𑀓𑀫𑀸𑀮𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀫𑀜𑁆𑀚𑀭𑀺𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀯𑀺𑀥𑀽𑀢𑀺𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀯𑀝𑀁𑀲𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀆𑀯𑁂𑀍𑀅𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀭𑀘𑁆𑀙𑀤𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀢𑁂 𑀓𑀼𑀮𑀺𑀢𑁆𑀣𑀻𑀦𑀁 𑀓𑀼𑀮𑀥𑀻𑀢𑀸𑀦𑀁 𑀓𑀼𑀮𑀓𑀼𑀫𑀸𑀭𑀻𑀦𑀁 𑀓𑀼𑀮𑀲𑀼𑀡𑁆𑀳𑀸𑀦𑀁 𑀓𑀼𑀮𑀤𑀸𑀲𑀻𑀦𑀁 𑀏𑀓𑀢𑁄𑀯𑀡𑁆𑀝𑀺𑀓𑀫𑀸𑀮𑀁 𑀳𑀭𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀪𑀢𑁄𑀯𑀡𑁆𑀝𑀺𑀓𑀫𑀸𑀮𑀁 𑀳𑀭𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀫𑀜𑁆𑀚𑀭𑀺𑀓𑀁 𑀳𑀭𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀯𑀺𑀥𑀽𑀢𑀺𑀓𑀁 𑀳𑀭𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀯𑀝𑀁𑀲𑀓𑀁 𑀳𑀭𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀆𑀯𑁂𑀍𑀅𑀁 𑀳𑀭𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀭𑀘𑁆𑀙𑀤𑀁 𑀳𑀭𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀢𑁂 𑀓𑀼𑀮𑀺𑀢𑁆𑀣𑀻𑀳𑀺 𑀓𑀼𑀮𑀥𑀻𑀢𑀸𑀳𑀺 𑀓𑀼𑀮𑀓𑀼𑀫𑀸𑀭𑀻𑀳𑀺 𑀓𑀼𑀮𑀲𑀼𑀡𑁆𑀳𑀸𑀳𑀺 𑀓𑀼𑀮𑀤𑀸𑀲𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑀪𑀸𑀚𑀦𑁂𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀏𑀓𑀣𑀸𑀮𑀓𑁂𑀧𑀺 𑀧𑀺𑀯𑀦𑁆𑀢𑀺, 𑀏𑀓𑀸𑀲𑀦𑁂𑀧𑀺 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺, 𑀏𑀓𑀫𑀜𑁆𑀘𑁂𑀧𑀺 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀸𑀧𑀺 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀏𑀓𑀧𑀸𑀯𑀼𑀭𑀡𑀸𑀧𑀺 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸𑀧𑀺 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀯𑀺𑀓𑀸𑀮𑁂𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀫𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀧𑀺𑀯𑀦𑁆𑀢𑀺, 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀫𑁆𑀧𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀧𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺, 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺, 𑀕𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀕𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺, 𑀕𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀕𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺, 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺, 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺, 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺, 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑀧𑀺 𑀮𑀸𑀲𑁂𑀦𑁆𑀢𑀺; 𑀅𑀝𑁆𑀞𑀧𑀤𑁂𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀤𑀲𑀧𑀤𑁂𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀆𑀓𑀸𑀲𑁂𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀧𑀭𑀺𑀳𑀸𑀭𑀧𑀣𑁂𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀲𑀦𑁆𑀢𑀺𑀓𑀸𑀬𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀔𑀮𑀺𑀓𑀸𑀬𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀖𑀝𑀺𑀓𑀸𑀬𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀲𑀮𑀸𑀓𑀳𑀢𑁆𑀣𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀅𑀓𑁆𑀔𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀧𑀗𑁆𑀕𑀘𑀻𑀭𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀯𑀗𑁆𑀓𑀓𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀫𑁄𑀓𑁆𑀔𑀘𑀺𑀓𑀸𑀬𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀘𑀺𑀗𑁆𑀕𑀼𑀮𑀓𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀧𑀢𑁆𑀢𑀸𑀍𑀳𑀓𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀭𑀣𑀓𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀥𑀦𑀼𑀓𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀅𑀓𑁆𑀔𑀭𑀺𑀓𑀸𑀬𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀫𑀦𑁂𑀲𑀺𑀓𑀸𑀬𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺, 𑀬𑀣𑀸𑀯𑀚𑁆𑀚𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑀺𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀭𑀣𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀥𑀦𑀼𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀣𑀭𑀼𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀲𑁆𑀲𑀧𑀺 𑀧𑀼𑀭𑀢𑁄 𑀥𑀸𑀯𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀲𑁆𑀲𑀧𑀺 𑀧𑀼𑀭𑀢𑁄 𑀥𑀸𑀯𑀦𑁆𑀢𑀺, 𑀭𑀣𑀲𑁆𑀲𑀧𑀺 𑀧𑀼𑀭𑀢𑁄 𑀥𑀸𑀯𑀦𑁆𑀢𑀺𑀧𑀺 𑀆𑀥𑀸𑀯𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀲𑁆𑀲𑁂𑀍𑀏𑀦𑁆𑀢𑀺𑀧𑀺, 𑀅𑀧𑁆𑀨𑁄𑀝𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀦𑀺𑀩𑁆𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑀧𑀺, 𑀫𑀼𑀝𑁆𑀞𑀻𑀳𑀺𑀧𑀺 𑀬𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺, 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂𑀧𑀺 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀘𑁆𑀘𑀓𑀺𑀁 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀇𑀥, 𑀪𑀕𑀺𑀦𑀺, 𑀦𑀘𑁆𑀘𑀲𑁆𑀲𑀽”𑀢𑀺, 𑀦𑀮𑀸𑀝𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀺𑀥𑀫𑁆𑀧𑀺 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀦𑁆𑀢𑀺𑁇

1754\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀲𑀻𑀲𑀼 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬, 𑀬𑁂𑀦 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂𑀦 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀆𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂𑀦 𑀧𑀲𑀸𑀭𑀺𑀢𑁂𑀦 𑀑𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀓𑁆𑀔𑀼 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑁆𑀯𑀸𑀬𑀁 𑀅𑀩𑀮𑀩𑀮𑁄 𑀯𑀺𑀬 𑀫𑀦𑁆𑀤𑀫𑀦𑁆𑀤𑁄 𑀯𑀺𑀬 𑀪𑀸𑀓𑀼𑀝𑀺𑀓𑀪𑀸𑀓𑀼𑀝𑀺𑀓𑁄 𑀯𑀺𑀬? 𑀓𑁄 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀕𑀢𑀲𑁆𑀲 𑀧𑀺𑀡𑁆𑀟𑀓𑀁 𑀤𑀲𑁆𑀲𑀢𑀺? 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀦 𑀅𑀬𑁆𑀬𑀸 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸 𑀲𑀡𑁆𑀳𑀸 𑀲𑀔𑀺𑀮𑀸 𑀲𑀼𑀔𑀲𑀫𑁆𑀪𑀸𑀲𑀸 𑀫𑀺𑀳𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸 𑀏𑀳𑀺𑀲𑁆𑀯𑀸𑀕𑀢𑀯𑀸𑀤𑀺𑀦𑁄 𑀅𑀩𑁆𑀪𑀸𑀓𑀼𑀝𑀺𑀓𑀸 𑀉𑀢𑁆𑀢𑀸𑀦𑀫𑀼𑀔𑀸 𑀧𑀼𑀩𑁆𑀩𑀪𑀸𑀲𑀺𑀦𑁄𑁇 𑀢𑁂𑀲𑀁 𑀔𑁄 𑀦𑀸𑀫 𑀧𑀺𑀡𑁆𑀟𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

1755\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀺𑀡𑁆𑀟𑁄 𑀮𑀩𑁆𑀪𑀢𑀻”𑀢𑀺? “𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀺𑀡𑁆𑀟𑁄 𑀮𑀩𑁆𑀪𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀖𑀭𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀖𑀭𑀁 𑀦𑁂𑀢𑁆𑀯𑀸 𑀪𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤, 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀳𑀺—  ‘𑀤𑀼𑀝𑁆𑀞𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁄𑁇 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸 𑀦𑀸𑀫 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀽𑁇 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀦𑁆𑀢𑀺—  𑀫𑀸𑀮𑀸𑀯𑀘𑁆𑀙𑀁 𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀭𑁄𑀧𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺𑀧𑀺 𑀲𑀺𑀜𑁆𑀘𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺…𑀧𑁂…  𑀯𑀺𑀯𑀺𑀥𑀫𑁆𑀧𑀺 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀤𑁆𑀥𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀧𑀲𑀦𑁆𑀦𑀸 𑀢𑁂𑀧𑀺 𑀏𑀢𑀭𑀳𑀺 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑁇 𑀬𑀸𑀦𑀺𑀧𑀺 𑀢𑀸𑀦𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀤𑀸𑀦𑀧𑀣𑀸𑀦𑀺 𑀢𑀸𑀦𑀺𑀧𑀺 𑀏𑀢𑀭𑀳𑀺 𑀉𑀧𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀺𑁇 𑀭𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺 𑀧𑁂𑀲𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀦𑀺𑀯𑀲𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀽𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬 𑀬𑀣𑀸𑀬𑀁 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁄 𑀲𑀡𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸’”𑀢𑀺𑁇

𑀓𑀻𑀍𑀇𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑀣𑀸𑀯𑀚𑁆𑀚𑁂𑀦𑀧𑀺 𑀓𑀻𑀍𑀇𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀭𑀣𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀥𑀦𑀼𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀣𑀭𑀼𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀲𑁆𑀲𑀧𑀺 𑀧𑀼𑀭𑀢𑁄 𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀲𑁆𑀲𑀧𑀺 𑀧𑀼𑀭𑀢𑁄 𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀭𑀣𑀲𑁆𑀲𑀧𑀺 𑀧𑀼𑀭𑀢𑁄 𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀆𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀲𑁆𑀲𑁂𑀍𑀏𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀅𑀧𑁆𑀨𑁄𑀝𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀦𑀺𑀩𑁆𑀩𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀫𑀼𑀝𑁆𑀞𑀻𑀳𑀺𑀧𑀺 𑀬𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀭𑀗𑁆𑀕𑀫𑀚𑁆𑀛𑁂𑀧𑀺 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀘𑁆𑀘𑀓𑀺𑀁 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀦𑁆𑀢𑀺 “𑀇𑀥, 𑀪𑀕𑀺𑀦𑀺, 𑀦𑀘𑁆𑀘𑀲𑁆𑀲𑀽”𑀢𑀺 𑀦𑀮𑀸𑀝𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀺𑀯𑀺𑀥𑀫𑁆𑀧𑀺 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺— 1758\. “𑀓𑀣𑀁 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫? 𑀘𑀡𑁆𑀟𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀨𑀭𑀼𑀲𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀸, 𑀩𑀳𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀕𑀘𑁆𑀙𑀣𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀞𑀫𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀘𑁄𑀤𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1760\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀫𑁂 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀸 𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑀸𑁇 𑀇𑀫𑁂𑀲𑀁 𑀧𑀸𑀧𑀓𑀸 𑀲𑀫𑀸𑀘𑀸𑀭𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀓𑀼𑀮𑀸𑀦𑀺 𑀘 𑀇𑀫𑁂𑀳𑀺 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀲𑀁𑀖𑁄 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀦 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀁—  “𑀦 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆”𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1763\. 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺—  “𑀦 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀓𑀻𑀝𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀲𑀁𑀖𑁂𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑀸 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀦 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀔𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀢𑀸 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀢𑀸 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀢𑀸 𑀪𑀬𑀕𑀸𑀫𑀺𑀢𑀸 𑀧𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑀧𑀺, 𑀯𑀺𑀩𑁆𑀪𑀫𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀲𑁆𑀲𑀚𑀺𑀧𑀼𑀦𑀩𑁆𑀩𑀲𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑁂𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑀸 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀔𑀫𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀓𑁆𑀓𑁄𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀢𑀸 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀢𑀸 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀢𑀸 𑀪𑀬𑀕𑀸𑀫𑀺𑀢𑀸 𑀧𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀻”𑀢𑀺𑁇

1765\. **“𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀦𑁂𑀯 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄 𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑁄𑁇 𑀢𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑀸 𑀲𑀫𑀸𑀘𑀸𑀭𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘, 𑀓𑀼𑀮𑀸𑀦𑀺 𑀘 𑀢𑁂𑀦 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄 𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑁄, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑁄 𑀧𑀸𑀧𑀓𑀸 𑀲𑀫𑀸𑀘𑀸𑀭𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘, 𑀓𑀼𑀮𑀸𑀦𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀢𑀸 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀧𑀓𑁆𑀓𑀫𑀢𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸𑁇 𑀅𑀮𑀁 𑀢𑁂 𑀇𑀥 𑀯𑀸𑀲𑁂𑀦𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀬𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀸𑀤𑀺𑀲𑀺𑀓𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀏𑀓𑀘𑁆𑀘𑀁 𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀅𑀯𑀘𑁇 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄 𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑁄 𑀧𑀸𑀧𑀓𑀸 𑀲𑀫𑀸𑀘𑀸𑀭𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀓𑀼𑀮𑀸𑀦𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀢𑀸 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀧𑀓𑁆𑀓𑀫𑀢𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸𑁇 𑀅𑀮𑀁 𑀢𑁂 𑀇𑀥 𑀯𑀸𑀲𑁂𑀦𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀢𑀣𑁂𑀯 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀜𑁆𑀘𑁂 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑁄 𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”**𑀢𑀺𑁇 (*𑁧𑁩*:**𑁧𑁭**)

1766\. **𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀦𑁂𑀯 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸**𑀢𑀺 𑀕𑀸𑀫𑁄𑀧𑀺 𑀦𑀺𑀕𑀫𑁄𑀧𑀺 𑀦𑀕𑀭𑀫𑁆𑀧𑀺 𑀕𑀸𑀫𑁄 𑀘𑁂𑀯 𑀦𑀺𑀕𑀫𑁄 𑀘𑁇

1767\. **𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀻**𑀢𑀺 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑁇

1768\. **𑀓𑀼𑀮𑀁** 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀼𑀮𑀸𑀦𑀺—  𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁, 𑀯𑁂𑀲𑁆𑀲𑀓𑀼𑀮𑀁, 𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑀁𑁇

1769\. **𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄**𑀢𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀤𑀽𑀲𑁂𑀢𑀺 𑀧𑀼𑀧𑁆𑀨𑁂𑀦 𑀯𑀸 𑀨𑀮𑁂𑀦 𑀯𑀸 𑀘𑀼𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀯𑀸 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀯𑁂𑀍𑀉𑀬𑀸 𑀯𑀸 𑀯𑁂𑀚𑁆𑀚𑀺𑀓𑀸𑀬 𑀯𑀸 𑀚𑀗𑁆𑀖𑀧𑁂𑀲𑀦𑀺𑀓𑁂𑀦 𑀯𑀸𑁇

1770\. **𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑁄**𑀢𑀺 𑀫𑀸𑀮𑀸𑀯𑀘𑁆𑀙𑀁 𑀭𑁄𑀧𑁂𑀢𑀺𑀧𑀺 𑀭𑁄𑀧𑀸𑀧𑁂𑀢𑀺𑀧𑀺, 𑀲𑀺𑀜𑁆𑀘𑀢𑀺𑀧𑀺 𑀲𑀺𑀜𑁆𑀘𑀸𑀧𑁂𑀢𑀺𑀧𑀺, 𑀑𑀘𑀺𑀦𑀸𑀢𑀺𑀧𑀺 𑀑𑀘𑀺𑀦𑀸𑀧𑁂𑀢𑀺𑀧𑀺, 𑀕𑀦𑁆𑀣𑁂𑀢𑀺𑀧𑀺 𑀕𑀦𑁆𑀣𑀸𑀧𑁂𑀢𑀺𑀧𑀺𑁇

1771\. **𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑀸**𑀢𑀺 𑀬𑁂 𑀲𑀁𑀫𑀼𑀔𑀸 𑀢𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀢𑀺𑀭𑁄𑀓𑁆𑀔𑀸 𑀢𑁂 𑀲𑀼𑀡𑀦𑁆𑀢𑀺𑁇

1772\. **𑀓𑀼𑀮𑀸𑀦𑀺 𑀘 𑀢𑁂𑀦 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀻**𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀤𑁆𑀥𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀆𑀕𑀫𑁆𑀫 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀲𑀦𑁆𑀦𑀸 𑀳𑀼𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

1773\. **𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑀸**𑀢𑀺 𑀬𑁂 𑀲𑀁𑀫𑀼𑀔𑀸 𑀢𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀢𑀺𑀭𑁄𑀓𑁆𑀔𑀸 𑀢𑁂 𑀲𑀼𑀡𑀦𑁆𑀢𑀺𑁇

1774\. **𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺 𑀬𑁄 𑀲𑁄 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑁇

1775\. **𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺𑁇 𑀬𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀬𑁂 𑀲𑀼𑀡𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄 𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑁄 𑀧𑀸𑀧𑀓𑀸 𑀲𑀫𑀸𑀘𑀸𑀭𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀓𑀼𑀮𑀸𑀦𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀢𑀸 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀧𑀓𑁆𑀓𑀫𑀢𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸𑁇 𑀅𑀮𑀁 𑀢𑁂 𑀇𑀥 𑀯𑀸𑀲𑁂𑀦𑀸”𑀢𑀺𑁇

1776\. 𑀏𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀪𑀬𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑁇 𑀢𑀸𑀤𑀺𑀲𑀺𑀓𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀏𑀓𑀘𑁆𑀘𑀁 𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1777\. **𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺 𑀬𑁄 𑀲𑁄 𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑁇

1778\. **𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺𑁇

1779\. 𑀬𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀬𑁂 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀢𑁂𑀳𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀅𑀯𑀘𑁇 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄 𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑁄 𑀧𑀸𑀧𑀓𑀸 𑀲𑀫𑀸𑀘𑀸𑀭𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀓𑀼𑀮𑀸𑀦𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀢𑀸 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀧𑀓𑁆𑀓𑀫𑀢𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸𑁇 𑀅𑀮𑀁 𑀢𑁂 𑀇𑀥 𑀯𑀸𑀲𑁂𑀦𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇

1780\. 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀯𑀤𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑀫𑁆𑀧𑀺 𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀅𑀯𑀘𑁇 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀦𑁄, 𑀦 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀬𑀕𑀸𑀫𑀺𑀦𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑁄 𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑁄 𑀧𑀸𑀧𑀓𑀸 𑀲𑀫𑀸𑀘𑀸𑀭𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀓𑀼𑀮𑀸𑀦𑀺 𑀘𑀸𑀬𑀲𑁆𑀫𑀢𑀸 𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀲𑀼𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀘𑁇 𑀧𑀓𑁆𑀓𑀫𑀢𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸𑁇 𑀅𑀮𑀁 𑀢𑁂 𑀇𑀥 𑀯𑀸𑀲𑁂𑀦𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇

1781\. “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(438.)

1782\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀢𑀸 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀢𑀸 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀢𑀸 𑀪𑀬𑀕𑀸𑀫𑀺𑀢𑀸 𑀧𑀸𑀧𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑁂𑀬𑁆𑀬 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1783\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺𑀢𑀸 𑀤𑁄𑀲𑀕𑀸𑀫𑀺𑀢𑀸 𑀫𑁄𑀳𑀕𑀸𑀫𑀺𑀢𑀸 𑀪𑀬𑀕𑀸𑀫𑀺𑀢𑀸 𑀧𑀸𑀧𑁂𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1784\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂… 𑁇

1785\. 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(439.)

1786\. 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸, 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑁆𑀢𑀺𑁇

1787\. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄**𑀢𑀺 𑀲𑀁𑀖𑁄𑀯 𑀢𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀺, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺, 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺, 𑀅𑀩𑁆𑀪𑁂𑀢𑀺; 𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸, 𑀦 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺, 𑀢𑀲𑁆𑀲𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀓𑀸𑀬𑀲𑁆𑀲 𑀦𑀸𑀫𑀓𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺𑁇

(440.)

1788\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1789\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1790\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1791\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1792\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1793\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(441.)

1794\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀲𑁆𑀲, 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1795\. 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑁂𑀭𑀲𑀫𑀁𑁇



(442.)

1796\. “𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀢𑁂𑀭𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀦𑀯 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸𑁇 𑀬𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀢𑀻𑀳𑀁 𑀚𑀸𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺 𑀢𑀸𑀯𑀢𑀻𑀳𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀓𑀸𑀫𑀸 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀭𑀺𑀯𑀼𑀢𑁆𑀣𑀧𑀭𑀺𑀯𑀸𑀲𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀢𑁆𑀢𑀭𑀺 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀢𑁆𑀣 𑀲𑀺𑀬𑀸 𑀯𑀻𑀲𑀢𑀺𑀕𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀢𑀢𑁆𑀣 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀩𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀏𑀓𑁂𑀦𑀧𑀺 𑀘𑁂 𑀊𑀦𑁄 𑀯𑀻𑀲𑀢𑀺𑀕𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀩𑁆𑀪𑀺𑀢𑁄, 𑀢𑁂 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀭𑀬𑁆𑀳𑀸, 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀸𑀫𑀻𑀘𑀺𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  ‘𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸’? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  ‘𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸’? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  ‘𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸’? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄𑁇 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻”𑀢𑀺𑁇

---

1797\. 𑀢𑁂𑀭𑀲𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

1798\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1799\. _𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁,_  
_𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀜𑁆𑀘;_  
_𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁆𑀢𑀁 𑀓𑀼𑀝𑀻 𑀘𑁂𑀯,_  
_𑀯𑀺𑀳𑀸𑀭𑁄 𑀘 𑀅𑀫𑀽𑀮𑀓𑀁𑁇_  


1800\. _𑀓𑀺𑀜𑁆𑀘𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀪𑁂𑀤𑁄 𑀘,_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸;_  
_𑀤𑀼𑀩𑁆𑀩𑀘𑀁 𑀓𑀼𑀮𑀤𑀽𑀲𑀜𑁆𑀘,_  
_𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀢𑁂𑀭𑀲𑀸𑀢𑀺𑁇_  


---

1801\. 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
