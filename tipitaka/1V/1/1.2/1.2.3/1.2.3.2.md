# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

(288.)

1169\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀦𑀯𑀭𑀢𑁆𑀢𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀧𑀸𑀭𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀮𑁄𑀳𑀺𑀢𑀁 𑀔𑁄 𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 “𑀆𑀫𑀸𑀬𑁆𑀬, 𑀦𑀯𑀭𑀢𑁆𑀢𑁄 𑀓𑀫𑁆𑀩𑀮𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺 “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧)

1170\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀔𑀭𑀓𑀫𑁆𑀩𑀮𑀁 𑀧𑀸𑀭𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀓𑁆𑀓𑀲𑀮𑁄𑀫𑀁 𑀔𑁄 𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 “𑀆𑀫𑀸𑀬𑁆𑀬, 𑀔𑀭𑀓𑀫𑁆𑀩𑀮𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨)

1171\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀦𑀯𑀸𑀯𑀼𑀢𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀧𑀸𑀭𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀓𑀺𑀡𑁆𑀡𑀮𑁄𑀫𑀁 𑀔𑁄 𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 “𑀆𑀫𑀸𑀬𑁆𑀬, 𑀦𑀯𑀸𑀯𑀼𑀢𑁄 𑀓𑀫𑁆𑀩𑀮𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁩)

1172\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀔𑀭𑀓𑀫𑁆𑀩𑀮𑀓𑀁 𑀧𑀸𑀭𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀔𑀭𑀮𑁄𑀫𑀁 𑀔𑁄 𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 “𑀆𑀫𑀸𑀬𑁆𑀬, 𑀔𑀭𑀓𑀫𑁆𑀩𑀮𑀓𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪)

1173\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀧𑀸𑀯𑀸𑀭𑀁 𑀧𑀸𑀭𑀼𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀻𑀖𑀮𑁄𑀫𑀁 𑀔𑁄 𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 “𑀆𑀫𑀸𑀬𑁆𑀬, 𑀧𑀸𑀯𑀸𑀭𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫)

(289.)

1174\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀔𑁂𑀢𑁆𑀢𑀁 𑀯𑀧𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀸𑀧𑀺𑀢𑀁 𑀔𑁄 𑀢𑁂, 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺? 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 “𑀆𑀫𑀸𑀬𑁆𑀬, 𑀦𑁄 𑀘 𑀔𑁄 𑀧𑀝𑀺𑀯𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬)

1175\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀁 𑀧𑀝𑀺𑀧𑀣𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺, 𑀪𑀕𑀺𑀦𑀺, 𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀲𑀻𑀤𑀢𑀻”𑀢𑀺? 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 “𑀆𑀫, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭)

1176\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀤𑁆𑀥𑀸𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀕𑀺𑀦𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀬𑀁 𑀲𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀤𑁂𑀲𑀺 𑀢𑀁 𑀦𑀸𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑁂𑀲𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺? “𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁮)

1177\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀤𑁆𑀥𑀸𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀕𑀺𑀦𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀬𑀁 𑀅𑀕𑁆𑀕𑀤𑀸𑀦𑀁 𑀢𑀁 𑀦𑀸𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑁂𑀲𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀕𑁆𑀕𑀤𑀸𑀦𑀦𑁆”𑀢𑀺? “𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁯)

1178\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀺𑀝𑁆𑀞, 𑀪𑀕𑀺𑀦𑀺, 𑀅𑀳𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀦𑀺𑀲𑀻𑀤, 𑀪𑀕𑀺𑀦𑀺, 𑀅𑀳𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀦𑀺𑀧𑀚𑁆𑀚, 𑀪𑀕𑀺𑀦𑀺, 𑀅𑀳𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦--𑁧𑁨)
