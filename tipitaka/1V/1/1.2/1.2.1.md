# 𑁧𑁇𑁨𑁇𑁧 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

774\. 𑀇𑀫𑁂 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀢𑁂𑀭𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸

  
𑀥𑀫𑁆𑀫𑀸 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇



* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(234.)

775\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀓𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀮𑀽𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁄 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁄𑁇 𑀅𑀤𑁆𑀤𑀲 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑀁 𑀓𑀺𑀲𑀁 𑀮𑀽𑀔𑀁 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀁 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀁 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀓, 𑀓𑀺𑀲𑁄 𑀮𑀽𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁄 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁄? 𑀓𑀘𑁆𑀘𑀺 𑀦𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀓, 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀓, 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑁇 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀬𑀤𑀸 𑀢𑁂 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺 𑀢𑀤𑀸 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀼𑀘𑀺𑀁 𑀫𑁄𑀘𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀧𑁆𑀧𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀆𑀯𑀼𑀲𑁄𑁇 𑀅𑀳𑀫𑁆𑀧𑀺 𑀏𑀯𑀁 𑀓𑀭𑁄𑀫𑀻”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

776\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀺 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧𑀺 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑀺𑁇 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀬𑀤𑀸 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺 𑀢𑀤𑀸 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀼𑀘𑀺𑀁 𑀫𑁄𑀘𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀡𑁆𑀡𑀯𑀸 𑀅𑀳𑁄𑀲𑀺 𑀧𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄 𑀧𑀲𑀦𑁆𑀦𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀧𑀼𑀩𑁆𑀩𑁂 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀓, 𑀓𑀺𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀮𑀽𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁄 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁄𑁇 𑀲𑁄 𑀤𑀸𑀦𑀺 𑀢𑁆𑀯𑀁 𑀏𑀢𑀭𑀳𑀺 𑀯𑀡𑁆𑀡𑀯𑀸 𑀧𑀻𑀡𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄 𑀧𑀲𑀦𑁆𑀦𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀓, 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀓𑀭𑁄𑀲𑀻”𑀢𑀺? “𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀓𑀭𑁄𑀫𑀺𑁇 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧𑀸𑀫𑀺 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑀸𑀫𑀺𑁇 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀲𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀦𑁆𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀬𑀤𑀸 𑀫𑁂 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀭𑀸𑀕𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺 𑀢𑀤𑀸 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀼𑀘𑀺𑀁 𑀫𑁄𑀘𑁂𑀫𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑀓, 𑀬𑁂𑀦𑁂𑀯 𑀳𑀢𑁆𑀣𑁂𑀦 𑀲𑀤𑁆𑀥𑀸𑀤𑁂𑀬𑁆𑀬𑀁 𑀪𑀼𑀜𑁆𑀚𑀲𑀺 𑀢𑁂𑀦𑁂𑀯 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀼𑀘𑀺𑀁 𑀫𑁄𑀘𑁂𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑁄 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀼𑀘𑀺𑀁 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

777\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀲𑁂𑀬𑁆𑀬𑀲𑀓, 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀼𑀘𑀺𑀁 𑀫𑁄𑀘𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀳𑀢𑁆𑀣𑁂𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀼𑀘𑀺𑀁 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄 𑀦𑁄 𑀲𑀭𑀸𑀕𑀸𑀬, 𑀯𑀺𑀲𑀜𑁆𑀜𑁄𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄 𑀦𑁄 𑀲𑀜𑁆𑀜𑁄𑀕𑀸𑀬, 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄 𑀦𑁄 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀸𑀬𑁇 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀬𑀸 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑀺𑀢𑁂 𑀲𑀭𑀸𑀕𑀸𑀬 𑀘𑁂𑀢𑁂𑀲𑁆𑀲𑀲𑀺, 𑀯𑀺𑀲𑀜𑁆𑀜𑁄𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑀺𑀢𑁂 𑀲𑀜𑁆𑀜𑁄𑀕𑀸𑀬 𑀘𑁂𑀢𑁂𑀲𑁆𑀲𑀲𑀺, 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑀺𑀢𑁂 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑀸𑀬 𑀘𑁂𑀢𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀭𑀸𑀕𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄, 𑀫𑀤𑀦𑀺𑀫𑁆𑀫𑀤𑀦𑀸𑀬 𑀧𑀺𑀧𑀸𑀲𑀯𑀺𑀦𑀬𑀸𑀬 𑀆𑀮𑀬𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸𑀬 𑀯𑀝𑁆𑀝𑀼𑀧𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄? 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀓𑀸𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀁 𑀅𑀓𑁆𑀔𑀸𑀢𑀁, 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀸, 𑀓𑀸𑀫𑀧𑀺𑀧𑀸𑀲𑀸𑀦𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑀬𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄, 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀦𑀁 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄, 𑀓𑀸𑀫𑀧𑀭𑀺𑀍𑀆𑀳𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄? 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬𑁇 𑀅𑀣 𑀔𑁆𑀯𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑀲𑀓𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀩𑁆𑀪𑀭𑀢𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

778\. **“𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”**𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

779\. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

780\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀻 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀻𑀦𑀁 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸𑀦𑀁 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄’𑀢𑀺𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀘𑁂𑀢𑁆𑀣 𑀘𑁂𑀢𑀦𑀸 𑀮𑀩𑁆𑀪𑀢𑀺𑁇 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀢𑁆𑀣𑁂𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀢𑀦𑀸; 𑀲𑀸 𑀘 𑀔𑁄 𑀅𑀩𑁆𑀩𑁄𑀳𑀸𑀭𑀺𑀓𑀸𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

---

926\. 𑀧𑀺𑀝𑁆𑀞𑀺𑀘𑀓𑁆𑀓𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀫𑀁 𑀕𑀫𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(260.)

927\. 𑀲𑀧𑁆𑀧𑀺𑀯𑀡𑁆𑀡𑀁 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀤𑀥𑀺𑀯𑀡𑁆𑀡𑀁 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

928\. 𑀦𑀻𑀮𑀁…𑀧𑁂…  𑀧𑀻𑀢𑀓𑀁…  𑀮𑁄𑀳𑀺𑀢𑀓𑀁…  𑀑𑀤𑀸𑀢𑀁…  𑀢𑀓𑁆𑀓𑀯𑀡𑁆𑀡𑀁…  𑀤𑀓𑀯𑀡𑁆𑀡𑀁…  𑀢𑁂𑀮𑀯𑀡𑁆𑀡𑀁…  𑀔𑀻𑀭𑀯𑀡𑁆𑀡𑀁 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀤𑀥𑀺𑀯𑀡𑁆𑀡𑀁 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

---

929\. 𑀧𑀺𑀝𑁆𑀞𑀺𑀘𑀓𑁆𑀓𑀲𑁆𑀲 𑀦𑀯𑀫𑀁 𑀕𑀫𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(261.)

930\. 𑀦𑀻𑀮𑀁 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀲𑀧𑁆𑀧𑀺𑀯𑀡𑁆𑀡𑀁 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

931\. 𑀧𑀻𑀢𑀓𑀁…𑀧𑁂…  𑀮𑁄𑀳𑀺𑀢𑀓𑀁…  𑀑𑀤𑀸𑀢𑀁…  𑀢𑀓𑁆𑀓𑀯𑀡𑁆𑀡𑀁…  𑀤𑀓𑀯𑀡𑁆𑀡𑀁…  𑀢𑁂𑀮𑀯𑀡𑁆𑀡𑀁…  𑀔𑀻𑀭𑀯𑀡𑁆𑀡𑀁…  𑀤𑀥𑀺𑀯𑀡𑁆𑀡𑀁 𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀲𑀧𑁆𑀧𑀺𑀯𑀡𑁆𑀡𑀁 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

---

932\. 𑀧𑀺𑀝𑁆𑀞𑀺𑀘𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑀲𑀫𑀁 𑀕𑀫𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀧𑀺𑀝𑁆𑀞𑀺𑀘𑀓𑁆𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(262.)

933\. 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

934\. 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

935\. 𑀘𑁂𑀢𑁂𑀢𑀺 𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

936\. 𑀘𑁂𑀢𑁂𑀢𑀺 𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

937\. 𑀦 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

938\. 𑀦 𑀘𑁂𑀢𑁂𑀢𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

939\. 𑀦 𑀘𑁂𑀢𑁂𑀢𑀺 𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

940\. 𑀦 𑀘𑁂𑀢𑁂𑀢𑀺 𑀦 𑀉𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

941\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦, 𑀦𑀫𑁄𑀘𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
