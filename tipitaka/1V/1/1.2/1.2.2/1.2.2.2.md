# 𑁧𑁇𑁨𑁇𑁨𑁇𑁨 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀼

(281.)

1113\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀸𑀢𑀼𑀬𑀸 𑀫𑀸𑀢𑀼𑀧𑁂𑀫𑁂𑀦 𑀆𑀫𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧)

1114\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀻𑀢𑀼𑀬𑀸 𑀥𑀻𑀢𑀼𑀧𑁂𑀫𑁂𑀦 𑀆𑀫𑀲𑀺…𑀧𑁂…  𑀪𑀕𑀺𑀦𑀺𑀬𑀸 𑀪𑀕𑀺𑀦𑀺𑀧𑁂𑀫𑁂𑀦 𑀆𑀫𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨--𑁩)

1115\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁪)

1116\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀓𑁆𑀔𑀺𑀦𑀺𑀬𑀸 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁫)

1117\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁬)

1118\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀢𑁆𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁭)

1119\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁮)

1120\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁯)

1121\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀸𑀭𑀼𑀥𑀻𑀢𑀮𑀺𑀓𑀸𑀬 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁦)

(282.)

1122\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀫𑁆𑀧𑀻𑀍𑀏𑀢𑁆𑀯𑀸 𑀩𑀸𑀳𑀸𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀬 𑀆𑀦𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀲𑀸𑀤𑀺𑀬𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀤𑀺𑀬𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁧)

1123\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀅𑀪𑀺𑀭𑀽𑀍𑀳𑀁 𑀲𑀗𑁆𑀓𑀫𑀁 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀲𑀜𑁆𑀘𑀸𑀮𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁨)

1124\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀁 𑀧𑀝𑀺𑀧𑀣𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀁𑀲𑀓𑀽𑀝𑁂𑀦 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁩)

1125\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀅𑀪𑀺𑀭𑀽𑀍𑀳𑀁 𑀭𑀼𑀓𑁆𑀔𑀁 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀲𑀜𑁆𑀘𑀸𑀮𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪)

1126\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀅𑀪𑀺𑀭𑀽𑀍𑀳𑀁 𑀦𑀸𑀯𑀁 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀲𑀜𑁆𑀘𑀸𑀮𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁫)

1127\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀕𑀳𑀺𑀢𑀁 𑀭𑀚𑁆𑀚𑀼𑀁 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁬)

1128\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀕𑀳𑀺𑀢𑀁 𑀤𑀡𑁆𑀟𑀁 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀆𑀯𑀺𑀜𑁆𑀙𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁭)

1129\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀇𑀢𑁆𑀣𑀺𑀁 𑀧𑀢𑁆𑀢𑁂𑀦 𑀧𑀡𑀸𑀫𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁮)

1130\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑀬𑀸 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀧𑀸𑀤𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁯)

1131\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀢𑁆𑀣𑀺𑀁 𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀯𑀸𑀬𑀫𑀺𑀢𑁆𑀯𑀸 𑀦 𑀙𑀼𑀧𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀅𑀳𑁄𑀲𑀺…𑀧𑁂…  “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁨𑁦)
