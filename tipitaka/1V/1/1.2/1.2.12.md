# 𑁧𑁇𑁨𑁇𑁧𑁨 𑀤𑀼𑀩𑁆𑀩𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀩𑁆𑀩𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(424.)

1729\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀙𑀦𑁆𑀦𑁄 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀸𑀯𑀼𑀲𑁄 𑀙𑀦𑁆𑀦, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀣? 𑀅𑀳𑀁 𑀔𑁄 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀤𑁂𑀬𑁆𑀬𑀁𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀅𑀪𑀺𑀲𑀫𑀺𑀢𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀫𑀳𑀸𑀯𑀸𑀢𑁄 𑀯𑀸𑀬𑀦𑁆𑀢𑁄 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀲𑀝𑀁 𑀏𑀓𑀢𑁄 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀦𑀤𑀻 𑀧𑀩𑁆𑀩𑀢𑁂𑀬𑁆𑀬𑀸 𑀲𑀗𑁆𑀔𑀲𑁂𑀯𑀸𑀮𑀧𑀡𑀓𑀁 𑀏𑀓𑀢𑁄 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸𑀦𑀸𑀫𑀸 𑀦𑀸𑀦𑀸𑀕𑁄𑀢𑁆𑀢𑀸 𑀦𑀸𑀦𑀸𑀚𑀘𑁆𑀘𑀸 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀏𑀓𑀢𑁄 𑀉𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑀸𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀣? 𑀅𑀳𑀁 𑀔𑁄 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀤𑁂𑀬𑁆𑀬𑀁𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀅𑀪𑀺𑀲𑀫𑀺𑀢𑁄”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀙𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1730\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀙𑀦𑁆𑀦𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀙𑀦𑁆𑀦, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑁄𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(425.)

1731\. **“𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀦𑁂𑀯 𑀤𑀼𑀩𑁆𑀩𑀘𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑁄𑀢𑀺—  ‘𑀫𑀸 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀯𑀘𑀼𑀢𑁆𑀣 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀯𑀸 𑀧𑀸𑀧𑀓𑀁 𑀯𑀸, 𑀅𑀳𑀫𑁆𑀧𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀯𑀸 𑀧𑀸𑀧𑀓𑀁 𑀯𑀸, 𑀯𑀺𑀭𑀫𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀫𑀫 𑀯𑀘𑀦𑀸𑀬𑀸’𑀢𑀺, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀯𑀘𑀦𑀻𑀬𑀫𑁂𑀯𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀭𑁄𑀢𑀼, 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀤𑁂𑀢𑀼 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀯𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦𑁇 𑀏𑀯𑀁 𑀲𑀁𑀯𑀤𑁆𑀥𑀸 𑀳𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀧𑀭𑀺𑀲𑀸 𑀬𑀤𑀺𑀤𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀯𑀘𑀦𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑁂𑀦𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀢𑀣𑁂𑀯 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀜𑁆𑀘𑁂 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑁄 𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”**𑀢𑀺𑁇 (*𑁧𑁨*:**𑁧𑁬**)

(426.)

1732\. **𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀦𑁂𑀯 𑀤𑀼𑀩𑁆𑀩𑀘𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀤𑀼𑀩𑁆𑀩𑀘𑁄 𑀳𑁄𑀢𑀺 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀓𑁆𑀔𑀫𑁄 𑀅𑀧𑁆𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇

1733\. **𑀉𑀤𑁆𑀤𑁂𑀲𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀽**𑀢𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇

1734\. **𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺𑁇

1735\. **𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁** 𑀦𑀸𑀫 𑀬𑀁 𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁, 𑀏𑀢𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀦𑀸𑀫𑁇

1736\. 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑁄𑀢𑀺—  “𑀫𑀸 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀯𑀘𑀼𑀢𑁆𑀣 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀯𑀸 𑀧𑀸𑀧𑀓𑀁 𑀯𑀸, 𑀅𑀳𑀫𑁆𑀧𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀯𑀸 𑀧𑀸𑀧𑀓𑀁 𑀯𑀸𑁇 𑀯𑀺𑀭𑀫𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀫𑀫 𑀯𑀘𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

1737\. **𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺 𑀬𑁄 𑀲𑁄 𑀤𑀼𑀩𑁆𑀩𑀘𑀚𑀸𑀢𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑁇

1738\. **𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺𑁇 𑀬𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀬𑁂 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀢𑁂𑀳𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀯𑀘𑀦𑀻𑀬𑀫𑁂𑀯 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀭𑁄𑀢𑀼𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀤𑁂𑀢𑀼 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀯𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦𑁇 𑀏𑀯𑀁 𑀲𑀁𑀯𑀤𑁆𑀥𑀸 𑀳𑀺 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀧𑀭𑀺𑀲𑀸 𑀬𑀤𑀺𑀤𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀯𑀘𑀦𑁂𑀦 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑁂𑀦𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀯𑀤𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑀫𑁆𑀧𑀺 𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀅𑀓𑀸𑀲𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑁂𑀦𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(427.)

1739\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑁂𑀬𑁆𑀬 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1740\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1741\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀘𑀦𑀻𑀬𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1742\. 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(428.)

1743\. 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸, 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑁆𑀢𑀺𑁇

1744\. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄**𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺𑁇

(429.)

1745\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1746\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1747\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

1748\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1749\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1750\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(430.)

1751\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀲𑁆𑀲, 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1752\. 𑀤𑀼𑀩𑁆𑀩𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁𑁇
