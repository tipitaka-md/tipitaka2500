# 𑁧𑁇𑁪𑁇𑁧𑁇𑁪 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀻𑀯𑀭𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(503.)

1941. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀉𑀤𑀸𑀬𑀻 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀪𑀢𑁆𑀢𑀯𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀁 𑀓𑀭𑁄𑀢𑀺𑁇

1942. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀢𑁄 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀲𑀸𑀧𑀺 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀉𑀧𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀺𑁇 𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙, 𑀪𑀕𑀺𑀦𑀺, 𑀉𑀤𑀓𑀁 𑀆𑀳𑀭, 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁 𑀥𑁄𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀆𑀳𑀭𑀬𑁆𑀬, 𑀅𑀳𑀫𑁂𑀯 𑀥𑁄𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀢𑀁 𑀅𑀲𑀼𑀘𑀺𑀁 𑀏𑀓𑀤𑁂𑀲𑀁 𑀫𑀼𑀔𑁂𑀦 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀏𑀓𑀤𑁂𑀲𑀁 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑁇 𑀲𑀸 𑀢𑁂𑀦 𑀕𑀩𑁆𑀪𑀁 𑀕𑀡𑁆𑀳𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀻 𑀅𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻, 𑀕𑀩𑁆𑀪𑀺𑀦𑀻”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀻”𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑁄 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1943. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀜𑀸𑀢𑀺𑀓𑀸 𑀢𑁂, 𑀉𑀤𑀸𑀬𑀺, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀯𑀸 𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀯𑀸𑁇 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(504.)

1944. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀭𑀚𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁪*:**𑁨𑁩**)

(505.)

1945. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1946. **𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀫𑀸𑀢𑀺𑀢𑁄 𑀯𑀸 𑀧𑀺𑀢𑀺𑀢𑁄 𑀯𑀸 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀲𑀫𑁆𑀩𑀤𑁆𑀥𑀸𑁇

1947. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻** 𑀦𑀸𑀫 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑁇

1948. **𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀲𑀓𑀺𑀁 𑀦𑀺𑀯𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀲𑀓𑀺𑀁 𑀧𑀸𑀭𑀼𑀢𑀫𑁆𑀧𑀺𑁇

1949. **𑀥𑁄𑀯𑀸𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺**, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀥𑁄𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 **𑀭𑀚𑀸𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺**, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀭𑀢𑁆𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 **𑀆𑀓𑁄𑀝𑁂𑀳𑀻𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺**, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀓𑀺𑀁 𑀧𑀸𑀡𑀺𑀧𑁆𑀧𑀳𑀸𑀭𑀁 𑀯𑀸 𑀫𑀼𑀕𑁆𑀕𑀭𑀧𑁆𑀧𑀳𑀸𑀭𑀁 𑀯𑀸 𑀤𑀺𑀦𑁆𑀦𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀥𑁄𑀯𑀸𑀧𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(506.)

1950. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇

1951. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇

1952. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀆𑀓𑁄𑀝𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇

1953. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑀻𑀤𑀦𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀭𑀡𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀏𑀓𑀢𑁄𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(507.)

1954. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀥𑁄𑀯𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸 𑀤𑀼𑀢𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀼𑀢𑁆𑀢𑀸 𑀥𑁄𑀯𑀢𑀺, 𑀅𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀘𑀻𑀯𑀭𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1955. 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
