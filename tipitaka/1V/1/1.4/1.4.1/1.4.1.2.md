# 𑁧𑁇𑁪𑁇𑁧𑁇𑁨 𑀉𑀤𑁄𑀲𑀺𑀢(𑀤𑀼𑀢𑀺𑀬𑀓𑀣𑀺𑀦)𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀻𑀯𑀭𑀯𑀕𑁆𑀕

### 𑀉𑀤𑁄𑀲𑀺𑀢(𑀤𑀼𑀢𑀺𑀬𑀓𑀣𑀺𑀦)𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(471.)

1883\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀘𑀺𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀓𑀡𑁆𑀡𑀓𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀑𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑁂𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀲𑁆𑀲𑀺𑀫𑀸𑀦𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀓𑀡𑁆𑀡𑀓𑀺𑀢𑀸𑀦𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1884\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(472.)

1885\. **“𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑀲𑁆𑀫𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀩𑁆𑀪𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀣𑀺𑀦𑁂 𑀏𑀓𑀭𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇

1886\. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(473.)

1887\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀜𑀸𑀢𑀓𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀪𑀤𑀦𑁆𑀢𑁄, 𑀫𑀬𑀁, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀜𑀸𑀢𑀓𑀸 𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀪𑀕𑀯𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀦𑀸𑀳𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀲𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀬𑀸𑀘𑀸𑀫𑀻’𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(474.)

1888\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1889\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸 𑀤𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1890\. 𑀤𑀺𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇

1891\. 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(475.)

1892\. **“𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑀲𑁆𑀫𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀩𑁆𑀪𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀣𑀺𑀦𑁂 𑀏𑀓𑀭𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀬𑁆𑀬, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨*:**𑁨𑁧**)

(476.)

1893\. **𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀘𑀻𑀯𑀭𑀲𑁆𑀫𑀺𑀦𑁆**𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀻𑀯𑀭𑀁 𑀓𑀢𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀦𑀝𑁆𑀞𑀁 𑀯𑀸 𑀯𑀺𑀦𑀝𑁆𑀞𑀁 𑀯𑀸 𑀤𑀟𑁆𑀠𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀯𑀸 𑀉𑀧𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸𑁇

1894\. **𑀉𑀩𑁆𑀪𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀣𑀺𑀦𑁂**𑀢𑀺 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀫𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀸𑀬 𑀫𑀸𑀢𑀺𑀓𑀸𑀬 𑀉𑀩𑁆𑀪𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀁𑀖𑁂𑀦 𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀸 𑀉𑀩𑁆𑀪𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

1895\. **𑀏𑀓𑀭𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀬𑀸 𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑁂𑀦 𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁂𑀦 𑀯𑀸𑁇

1896\. **𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸**𑀢𑀺 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁𑁇

1897\. **𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀲𑀳 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀭𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑀁 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁, 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

1898\. 𑀕𑀸𑀫𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑀁 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑀁𑁇 𑀉𑀤𑁄𑀲𑀺𑀢𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀅𑀝𑁆𑀝𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀫𑀸𑀍𑀑 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀧𑀸𑀲𑀸𑀤𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀳𑀫𑁆𑀫𑀺𑀬𑀁 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑀁 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑀁𑁇 𑀦𑀸𑀯𑀸 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑀸 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑀸𑁇 𑀲𑀢𑁆𑀣𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀔𑁂𑀢𑁆𑀢𑀁 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑀁 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑀁𑁇 𑀥𑀜𑁆𑀜𑀓𑀭𑀡𑀁 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑀁 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑀁𑁇 𑀆𑀭𑀸𑀫𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀯𑀺𑀳𑀸𑀭𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑀁 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑀁𑁇 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇

1899\. **𑀕𑀸𑀫𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀕𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀕𑀸𑀫𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀕𑀸𑀫𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀖𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀖𑀭𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1900\. 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀕𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀖𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀖𑀭𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀲𑀪𑀸𑀬𑁂 𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀪𑀸𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑀦 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀪𑀸𑀬𑁂 𑀯𑀸 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀪𑀸𑀬𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀪𑀸𑀬𑁂 𑀯𑀸 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀖𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀖𑀭𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1901\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀜𑁆𑀘, 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀅𑀦𑁆𑀢𑁄𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1902\. 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀜𑁆𑀘, 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1903\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀉𑀤𑁄𑀲𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘, 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀅𑀦𑁆𑀢𑁄𑀉𑀤𑁄𑀲𑀺𑀢𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀉𑀤𑁄𑀲𑀺𑀢𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1904\. 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀉𑀤𑁄𑀲𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘, 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(484.)

1905\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀅𑀝𑁆𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁆𑀢𑁄𑀅𑀝𑁆𑀝𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀅𑀝𑁆𑀝𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀅𑀝𑁆𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(485.)

1906\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀫𑀸𑀍𑀑 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁆𑀢𑁄𑀫𑀸𑀍𑀏 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀫𑀸𑀍𑀏 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀫𑀸𑀍𑀑 𑀳𑁄𑀢𑀺 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(486.)

1907\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀧𑀸𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁆𑀢𑁄𑀧𑀸𑀲𑀸𑀤𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀧𑀸𑀲𑀸𑀤𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀧𑀸𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(487.)

1908\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀳𑀫𑁆𑀫𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑁄𑀳𑀫𑁆𑀫𑀺𑀬𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀳𑀫𑁆𑀫𑀺𑀬𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀳𑀫𑁆𑀫𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀩𑁆𑀪𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(488.)

1909\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀦𑀸𑀯𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑁄𑀦𑀸𑀯𑀸𑀬 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀦𑀸𑀯𑀸𑀬 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀦𑀸𑀯𑀸 𑀳𑁄𑀢𑀺 𑀦𑀸𑀦𑀸𑀕𑀩𑁆𑀪𑀸 𑀦𑀸𑀦𑀸𑀑𑀯𑀭𑀓𑀸𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀑𑀯𑀭𑀓𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀑𑀯𑀭𑀓𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(489.)

1910\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀢𑁄 𑀯𑀸 𑀧𑀘𑁆𑀙𑀢𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀸, 𑀧𑀲𑁆𑀲𑀢𑁄 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀁 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(490.)

1911\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀔𑁂𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀔𑁂𑀢𑁆𑀢𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀔𑁂𑀢𑁆𑀢𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀔𑁂𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀔𑁂𑀢𑁆𑀢𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(491.)

1912\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀥𑀜𑁆𑀜𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀥𑀜𑁆𑀜𑀓𑀭𑀡𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀥𑀜𑁆𑀜𑀓𑀭𑀡𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀥𑀜𑁆𑀜𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀥𑀜𑁆𑀜𑀓𑀭𑀡𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(492.)

1913\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀆𑀭𑀸𑀫𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀆𑀭𑀸𑀫𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀆𑀭𑀸𑀫𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(493.)

1914\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 𑀅𑀦𑁆𑀢𑁄𑀯𑀺𑀳𑀸𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀤𑁆𑀯𑀸𑀭𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀯𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(494.)

1915\. 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀁 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀓𑀸𑀮𑁂 𑀲𑀫𑀦𑁆𑀢𑀸 𑀙𑀸𑀬𑀸 𑀨𑀭𑀢𑀺, 𑀅𑀦𑁆𑀢𑁄𑀙𑀸𑀬𑀸𑀬 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀙𑀸𑀬𑀸𑀬 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀲𑁆𑀲 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀸 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1916\. 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁄 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄 𑀦𑀸𑀫 𑀅𑀕𑀸𑀫𑀓𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀲𑀫𑀦𑁆𑀢𑀸 𑀲𑀢𑁆𑀢𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀽𑀧𑀘𑀸𑀭𑁄, 𑀢𑀢𑁄 𑀧𑀭𑀁 𑀦𑀸𑀦𑀽𑀧𑀘𑀸𑀭𑁄𑁇

(495.)

1917\. 𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁂 𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁂 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀼𑀤𑁆𑀥𑀝𑁂 𑀧𑀘𑁆𑀘𑀼𑀤𑁆𑀥𑀝𑀲𑀜𑁆𑀜𑀻…𑀧𑁂…  𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀲𑀜𑁆𑀜𑀻…  𑀅𑀦𑀝𑁆𑀞𑁂 𑀦𑀝𑁆𑀞𑀲𑀜𑁆𑀜𑀻…  𑀅𑀯𑀺𑀦𑀝𑁆𑀞𑁂 𑀯𑀺𑀦𑀝𑁆𑀞𑀲𑀜𑁆𑀜𑀻…  𑀅𑀤𑀟𑁆𑀠𑁂 𑀤𑀟𑁆𑀠𑀲𑀜𑁆𑀜𑀻 𑀅𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑁂 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

1918\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁂 𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁂 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(496.)

1919\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀦𑁆𑀢𑁄𑀅𑀭𑀼𑀡𑁂 𑀧𑀘𑁆𑀘𑀼𑀤𑁆𑀥𑀭𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀺, 𑀦𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺, 𑀟𑀬𑁆𑀳𑀢𑀺, 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀁 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
