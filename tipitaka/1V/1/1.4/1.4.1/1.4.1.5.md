# 𑁧𑁇𑁪𑁇𑁧𑁇𑁫 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀘𑀻𑀯𑀭𑀯𑀕𑁆𑀕

* 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

1956\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑁄𑀭𑀸 𑀓𑀢𑀓𑀫𑁆𑀫𑀸 𑀕𑀸𑀯𑀺𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀧𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀘𑁄𑀭𑀕𑀸𑀫𑀡𑀺𑀓𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑀪𑀸𑀢𑀼𑀓𑀸 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀞𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀦𑁆”𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀦 𑀫𑀕𑁆𑀕𑁂𑀦 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀘𑁄𑀭𑀕𑀸𑀫𑀡𑀺𑀓𑁄 𑀫𑀁𑀲𑁂 𑀧𑀓𑁆𑀓𑁂 𑀯𑀭𑀫𑀁𑀲𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀡𑁆𑀡𑀧𑀼𑀝𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀭𑀼𑀓𑁆𑀔𑁂 𑀆𑀮𑀕𑁆𑀕𑁂𑀢𑁆𑀯𑀸—  “𑀬𑁄 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀤𑀺𑀦𑁆𑀦𑀁𑀬𑁂𑀯 𑀳𑀭𑀢𑀽”𑀢𑀺, 𑀯𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀘𑁄𑀭𑀕𑀸𑀫𑀡𑀺𑀓𑀲𑁆𑀲 𑀇𑀫𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀫𑀁𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀁 𑀫𑀁𑀲𑀁 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑁂𑀦 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇

1957\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀑𑀳𑀺𑀬𑁆𑀬𑀓𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀳𑀸𑀭𑀧𑀸𑀮𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸”𑀢𑀺? “𑀧𑀯𑀺𑀝𑁆𑀞𑁄, 𑀪𑀕𑀺𑀦𑀺, 𑀪𑀕𑀯𑀸 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀇𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀁𑀲𑀁 𑀪𑀕𑀯𑀢𑁄 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑁄 𑀢𑀬𑀸, 𑀪𑀕𑀺𑀦𑀺, 𑀪𑀕𑀯𑀸 𑀫𑀁𑀲𑁂𑀦𑁇 𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀏𑀯𑀁 𑀅𑀳𑀫𑁆𑀧𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀫𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀸 𑀦𑀸𑀫 𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀸𑁇 𑀇𑀤𑀜𑁆𑀘 𑀫𑁂 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀘𑀻𑀯𑀭𑀁𑁇 𑀦𑀸𑀳𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀕𑀺𑀦𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀳𑀢𑁆𑀣𑀺𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀘𑁆𑀙𑁂 𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀪𑀕𑀯𑀢𑁄 𑀫𑀁𑀲𑀁 𑀤𑀢𑁆𑀯𑀸 𑀫𑀬𑀺 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁂 𑀲𑀚𑁆𑀚𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀤𑀸𑀬𑀺𑀦𑀸 𑀦𑀺𑀧𑁆𑀧𑀻𑀍𑀇𑀬𑀫𑀸𑀦𑀸 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁 𑀤𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑀬𑁄 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀳𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁄”𑀢𑀺? 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑁄 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1958\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀜𑀸𑀢𑀺𑀓𑀸 𑀢𑁂, 𑀉𑀤𑀸𑀬𑀺, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸𑁇 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀳𑀢𑁆𑀣𑀢𑁄 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(510.)

1961\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1962\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀻𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀻𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀁—  𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬, 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(511.)

1963\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀳𑀢𑁆𑀣𑀢𑁄 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫*:**𑁨𑁪**)

(512.)

1964\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1965\. **𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀫𑀸𑀢𑀺𑀢𑁄 𑀯𑀸 𑀧𑀺𑀢𑀺𑀢𑁄 𑀯𑀸 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀲𑀫𑁆𑀩𑀤𑁆𑀥𑀸𑁇

1966\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻** 𑀦𑀸𑀫 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑁇

1967\. **𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀙𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀼𑀧𑀕𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

1968\. **𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀸**𑀢𑀺 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁𑁇

1969\. 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀳𑀢𑁆𑀣𑀢𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(513.)

1970\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

1971\. 𑀏𑀓𑀢𑁄𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀳𑀢𑁆𑀣𑀢𑁄 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(514.)

1972\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀜𑀸𑀢𑀺𑀓𑀸𑀬, 𑀧𑀸𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀯𑀺𑀧𑀼𑀮𑀁, 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀢𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀘𑀻𑀯𑀭𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1973\. 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
