# 𑁧𑁇𑁪𑁇𑁧𑁇𑁬 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀻𑀯𑀭𑀯𑀕𑁆𑀕

### 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(515.)

1974. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀝𑁆𑀝𑁄 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀅𑀢𑁆𑀣𑁄𑁇 𑀧𑀝𑀺𑀩𑀮𑀸 𑀫𑀬𑀁 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀤𑀸𑀢𑀼𑀁 𑀬𑀤𑀺𑀤𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄𑀲𑀺, 𑀇𑀢𑁄 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀬 𑀏𑀓𑀲𑀸𑀝𑀓𑀁 𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯 𑀖𑀭𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇 𑀖𑀭𑀁 𑀕𑀢𑁄 𑀇𑀢𑁄 𑀯𑀸 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀧𑀳𑀺𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀇𑀢𑁄 𑀯𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀦𑁆”𑀢𑀺𑁇

1975. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄𑀲𑀺 𑀇𑀢𑁄 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀔𑁄 𑀪𑀦𑁆𑀢𑁂 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀬 𑀏𑀓𑀲𑀸𑀝𑀓𑀁 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀬𑀸𑀯 𑀖𑀭𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺, 𑀖𑀭𑀁 𑀕𑀢𑁄 𑀇𑀢𑁄 𑀯𑀸 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀧𑀳𑀺𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀇𑀢𑁄 𑀯𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀦𑁆𑀢𑀺𑁇

1976. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄𑀲𑀺, 𑀇𑀢𑁄 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀬 𑀏𑀓𑀲𑀸𑀝𑀓𑀁 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀬𑀸𑀯 𑀖𑀭𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺, 𑀖𑀭𑀁 𑀕𑀢𑁄 𑀇𑀢𑁄 𑀯𑀸 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀧𑀳𑀺𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀇𑀢𑁄 𑀯𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀢𑀭𑀦𑁆𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑀬𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁂𑀦 𑀧𑀯𑀸𑀭𑀺𑀢𑁂𑀦 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀦 𑀤𑁂𑀲𑀻”𑀢𑀺𑁇

1977. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀧𑁆𑀧𑀻𑀍𑀇𑀬𑀫𑀸𑀦𑁄 𑀏𑀓𑀁 𑀲𑀸𑀝𑀓𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁄, 𑀏𑀓𑀲𑀸𑀝𑀓𑁄 𑀆𑀕𑀘𑁆𑀙𑀲𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀫𑀳𑀺𑀘𑁆𑀙𑀸 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸𑁇 𑀦𑀬𑀺𑀫𑁂𑀲𑀁 𑀲𑀼𑀓𑀭𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀸𑀧𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀸𑀬 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑀸𑀬 𑀲𑀸𑀝𑀓𑀁 𑀕𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1978. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1979. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀜𑀸𑀢𑀓𑁄 𑀢𑁂, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀲𑁆𑀲 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸𑁇 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1980. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇

1982. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀓𑁂𑀢𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀘𑁄𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀼𑀁𑁇 𑀬𑀣𑀸𑀦𑀕𑁆𑀕𑀸𑀯 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀲𑀼𑀦𑁆𑀤𑀭𑀸 𑀔𑁄 𑀇𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀚𑀻𑀯𑀓𑀸 𑀬𑁂 𑀇𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀚𑀻𑀯𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀮𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀗𑁆𑀖𑀸𑀯𑀼𑀲𑁄 𑀉𑀧𑀸𑀮𑀺, 𑀇𑀫𑁂 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀸𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀧𑀸𑀮𑀺𑀦𑀸 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄𑁇 𑀤𑁂𑀣 𑀦𑁂𑀲𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀻”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀕𑁆𑀕𑀸 𑀆𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑀦𑀼 𑀦𑀸𑀫 𑀢𑀺𑀡𑁂𑀦 𑀯𑀸 𑀧𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀆𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

1983. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀘𑀻𑀯𑀭𑀲𑁆𑀲 𑀯𑀸 𑀦𑀝𑁆𑀞𑀘𑀻𑀯𑀭𑀲𑁆𑀲 𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀁 𑀆𑀯𑀸𑀲𑀁 𑀧𑀞𑀫𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀘𑁂 𑀢𑀢𑁆𑀣 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀪𑀽𑀫𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀘𑁆𑀙𑀯𑀺 𑀯𑀸, 𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑀼𑀁 ‘𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀑𑀤𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀦𑁄 𑀘𑁂 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀪𑀽𑀫𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀘𑁆𑀙𑀯𑀺 𑀯𑀸 𑀢𑀺𑀡𑁂𑀦 𑀯𑀸 𑀧𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀆𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁; 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀦𑀕𑁆𑀕𑁂𑀦 𑀆𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1991. **𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀘𑀻𑀯𑀭𑁄** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀯𑀸 𑀘𑁄𑀭𑁂𑀳𑀺 𑀯𑀸 𑀥𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀯𑀸, 𑀬𑁂𑀳𑀺 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

1992. **𑀦𑀝𑁆𑀞𑀘𑀻𑀯𑀭𑁄** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀯𑀸 𑀤𑀟𑁆𑀠𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀤𑀓𑁂𑀦 𑀯𑀸 𑀯𑀽𑀍𑀳𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀦𑁆𑀤𑀽𑀭𑁂𑀳𑀺 𑀯𑀸 𑀉𑀧𑀘𑀺𑀓𑀸𑀳𑀺 𑀯𑀸 𑀔𑀸𑀬𑀺𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀺𑀪𑁄𑀕𑀚𑀺𑀡𑁆𑀡𑀁 𑀯𑀸 𑀳𑁄𑀢𑀺𑁇

1993. 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀫𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀓𑀁, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀫𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀺𑀢𑀁, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(520.)

1994. 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀫𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀫𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁂 𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀫𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

1995. 𑀜𑀸𑀢𑀓𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀓𑁂 𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(521.)

1996. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀫𑀬𑁂, 𑀜𑀸𑀢𑀓𑀸𑀦𑀁, 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀥𑀦𑁂𑀦, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1997. 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
