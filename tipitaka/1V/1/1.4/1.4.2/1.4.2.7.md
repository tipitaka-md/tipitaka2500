# 𑁧𑁇𑁪𑁇𑁨𑁇𑁭 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀥𑁄𑀯𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀥𑁄𑀯𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(576.)

2189. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀭𑀚𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀦𑁆𑀢𑀺𑀬𑁄 𑀭𑀚𑀦𑁆𑀢𑀺𑀬𑁄 𑀯𑀺𑀚𑀝𑁂𑀦𑁆𑀢𑀺𑀬𑁄 𑀭𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀁 𑀅𑀥𑀺𑀲𑀻𑀮𑀁 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺 𑀕𑁄𑀢𑀫𑀻 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀕𑁄𑀢𑀫𑀺𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺, 𑀕𑁄𑀢𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑀺𑀬𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀓𑀼𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄𑁇 𑀅𑀬𑁆𑀬𑀸 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀭𑀚𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀦𑁆𑀢𑀺𑀬𑁄 𑀭𑀚𑀦𑁆𑀢𑀺𑀬𑁄 𑀯𑀺𑀚𑀝𑁂𑀦𑁆𑀢𑀺𑀬𑁄 𑀭𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀁 𑀅𑀥𑀺𑀲𑀻𑀮𑀁 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇

2190. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀕𑁄𑀢𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺 𑀕𑁄𑀢𑀫𑀻 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

2191. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀣𑀧𑀺 𑀭𑀚𑀸𑀧𑁂𑀣𑀧𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀣𑀧𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀜𑀸𑀢𑀺𑀓𑀸𑀬𑁄 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬𑁄”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬𑁄, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀯𑀸 𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀯𑀸𑁇 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀲𑁆𑀲𑀣𑀧𑀺 𑀭𑀚𑀸𑀧𑁂𑀲𑁆𑀲𑀣𑀧𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀲𑁆𑀲𑀣𑀧𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(577.)

2192. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀭𑀚𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁭*:**𑁩𑁬**)

(578.)

2193. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

2194. **𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀫𑀸𑀢𑀺𑀢𑁄 𑀯𑀸 𑀧𑀺𑀢𑀺𑀢𑁄 𑀯𑀸 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀲𑀫𑁆𑀩𑀤𑁆𑀥𑀸𑁇

2195. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻** 𑀦𑀸𑀫 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑁇

2196. 𑀥𑁄𑀯𑀸𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀥𑁄𑀢𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀭𑀚𑀸𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀭𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀯𑀺𑀚𑀝𑁂𑀳𑀻𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀯𑀺𑀚𑀝𑀺𑀢𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺…𑀧𑁂…  “𑀇𑀫𑀸𑀦𑀺 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀥𑁄𑀯𑀸𑀧𑀺𑀢𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀺𑁇 𑀇𑀫𑀸𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(579.)

2197. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇

2198. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇

2199. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀯𑀺𑀚𑀝𑀸𑀧𑁂𑀢𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀭𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇

(580.)

2200. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀏𑀓𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀜𑀸𑀢𑀺𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(581.)

2201. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀥𑁄𑀯𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸 𑀤𑀼𑀢𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀼𑀢𑁆𑀢𑀸 𑀥𑁄𑀯𑀢𑀺, 𑀅𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀁 𑀓𑀢𑀪𑀡𑁆𑀟𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2202. 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀥𑁄𑀯𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
