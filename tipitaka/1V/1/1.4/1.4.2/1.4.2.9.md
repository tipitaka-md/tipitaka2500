# 𑁧𑁇𑁪𑁇𑁨𑁇𑁯 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(587.)

2222\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇

2223\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2224\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(588.)

2225\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁯*:**𑁩𑁮**)

(589.)

2226\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

2227\. **𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁** 𑀦𑀸𑀫 𑀓𑀢𑀫𑁆𑀧𑀺 𑀅𑀓𑀢𑀫𑁆𑀧𑀺 𑀓𑀢𑀸𑀓𑀢𑀫𑁆𑀧𑀺𑁇 **𑀓𑀢𑀁** 𑀦𑀸𑀫 𑀲𑀻𑀲𑀽𑀧𑀕𑀁 𑀕𑀻𑀯𑀽𑀧𑀕𑀁 𑀳𑀢𑁆𑀣𑀽𑀧𑀕𑀁 𑀧𑀸𑀤𑀽𑀧𑀕𑀁 𑀓𑀝𑀽𑀧𑀕𑀁𑁇 **𑀅𑀓𑀢𑀁** 𑀦𑀸𑀫 𑀖𑀦𑀓𑀢𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀓𑀢𑀸𑀓𑀢𑀁** 𑀦𑀸𑀫 𑀢𑀤𑀼𑀪𑀬𑀁𑁇

2228\. **𑀭𑀽𑀧𑀺𑀬𑀁** 𑀦𑀸𑀫 𑀲𑀢𑁆𑀣𑀼𑀯𑀡𑁆𑀡𑁄 𑀓𑀳𑀸𑀧𑀡𑁄, 𑀮𑁄𑀳𑀫𑀸𑀲𑀓𑁄, 𑀤𑀸𑀭𑀼𑀫𑀸𑀲𑀓𑁄, 𑀚𑀢𑀼𑀫𑀸𑀲𑀓𑁄 𑀬𑁂 𑀯𑁄𑀳𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

2229\. **𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀓𑀢𑁂𑀦 𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀓𑀢𑁂𑀦 𑀅𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀓𑀢𑁂𑀦 𑀓𑀢𑀸𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀓𑀢𑁂𑀦 𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀓𑀢𑁂𑀦 𑀅𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀓𑀢𑁂𑀦 𑀓𑀢𑀸𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀓𑀢𑀸𑀓𑀢𑁂𑀦 𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀓𑀢𑀸𑀓𑀢𑁂𑀦 𑀅𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀓𑀢𑀸𑀓𑀢𑁂𑀦 𑀓𑀢𑀸𑀓𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  “𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀁𑁇 𑀇𑀤𑀁 𑀫𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀘𑁂 𑀢𑀢𑁆𑀣 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀯𑀸 𑀉𑀧𑀸𑀲𑀓𑁄 𑀯𑀸 𑀲𑁄 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀲𑁄 𑀪𑀡𑀢𑀺—  “𑀇𑀫𑀺𑀦𑀸 𑀓𑀺𑀁 𑀆𑀳𑀭𑀺𑀬𑁆𑀬𑀢𑀽”𑀢𑀺, 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀇𑀫𑀁 𑀯𑀸 𑀇𑀫𑀁 𑀯𑀸 𑀆𑀳𑀭𑀸”𑀢𑀺𑁇 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  𑀲𑀧𑁆𑀧𑀺 𑀯𑀸 𑀢𑁂𑀮𑀁 𑀯𑀸 𑀫𑀥𑀼 𑀯𑀸 𑀨𑀸𑀡𑀺𑀢𑀁 𑀯𑀸𑁇 𑀲𑀘𑁂 𑀲𑁄 𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀆𑀳𑀭𑀢𑀺, 𑀭𑀽𑀧𑀺𑀬𑀘𑁂𑀢𑀸𑀧𑀓𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸, 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀲𑁄 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀁 𑀙𑀟𑁆𑀟𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀲𑁄 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀽𑀧𑀺𑀬𑀙𑀟𑁆𑀟𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄—  𑀬𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀙𑀟𑁆𑀟𑀺𑀢𑀸𑀙𑀟𑁆𑀟𑀺𑀢𑀜𑁆𑀘 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀧𑀞𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀸𑀘𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

2230\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀭𑀽𑀧𑀺𑀬𑀙𑀟𑁆𑀟𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

2231\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀭𑀽𑀧𑀺𑀬𑀙𑀟𑁆𑀟𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀽𑀧𑀺𑀬𑀙𑀟𑁆𑀟𑀓𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

2233\. 𑀢𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(591.)

2234\. 𑀭𑀽𑀧𑀺𑀬𑁂 𑀭𑀽𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀭𑀽𑀧𑀺𑀬𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀭𑀽𑀧𑀺𑀬𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀭𑀽𑀧𑀺𑀬𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀭𑀽𑀧𑀺𑀬𑁂 𑀅𑀭𑀽𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀭𑀽𑀧𑀺𑀬𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀭𑀽𑀧𑀺𑀬𑁂 𑀭𑀽𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀭𑀽𑀧𑀺𑀬𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀭𑀽𑀧𑀺𑀬𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀭𑀽𑀧𑀺𑀬𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀭𑀽𑀧𑀺𑀬𑁂 𑀅𑀭𑀽𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀭𑀽𑀧𑀺𑀬𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2235\. 𑀅𑀭𑀽𑀧𑀺𑀬𑁂 𑀭𑀽𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀭𑀽𑀧𑀺𑀬𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀭𑀽𑀧𑀺𑀬𑁂 𑀅𑀭𑀽𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(592.)

2236\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
