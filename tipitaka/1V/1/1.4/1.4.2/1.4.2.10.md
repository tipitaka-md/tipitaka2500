# 𑁧𑁇𑁪𑁇𑁨𑁇𑁧𑁦 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(593.)

2238\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀝𑁆𑀝𑁄 𑀳𑁄𑀢𑀺 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀲𑁄 𑀧𑀝𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀭𑀢𑁆𑀢𑀁 𑀲𑀼𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀓𑀢𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀼𑀧𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀫𑀳𑀕𑁆𑀖𑀁 𑀧𑀝𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀦𑁆𑀤𑀭𑀸 𑀔𑁄 𑀢𑁆𑀬𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺; 𑀤𑁂𑀳𑀺 𑀫𑁂 𑀧𑀝𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀚𑀸𑀦𑀸𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 “𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀳𑀦𑁆𑀤𑀸𑀯𑀼𑀲𑁄”𑀢𑀺, 𑀅𑀤𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀢𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀭𑀸𑀫𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀼𑀦𑁆𑀤𑀭𑀸 𑀔𑁄 𑀢𑁆𑀬𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺; 𑀓𑀼𑀢𑁄 𑀢𑀬𑀸 𑀮𑀤𑁆𑀥𑀸”𑀢𑀺? “𑀢𑁂𑀦 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀝𑁂𑀦 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀸”𑀢𑀺𑁇 “𑀓𑀢𑀺𑀳𑀺𑀧𑀺 𑀢𑁆𑀬𑀸𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄𑀬𑁂𑀯 𑀢𑁂 𑀧𑀝𑁄 𑀯𑀭𑁄”𑀢𑀺𑁇

2239\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄—  “𑀲𑀘𑁆𑀘𑀁 𑀔𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀆𑀳𑀁𑀲𑀼—  ‘𑀓𑀢𑀺𑀳𑀺𑀧𑀺 𑀫𑁆𑀬𑀸𑀬𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁄𑀬𑁂𑀯 𑀫𑁂 𑀧𑀝𑁄 𑀯𑀭𑁄’”𑀢𑀺 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀳𑀦𑁆𑀤 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺; 𑀤𑁂𑀳𑀺 𑀫𑁂 𑀧𑀝𑀦𑁆”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑁄—  ‘𑀚𑀸𑀦𑀸𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄’𑀢𑀺𑁇 𑀦𑀸𑀳𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀕𑀺𑀳𑀻𑀧𑀺 𑀦𑀁 𑀕𑀺𑀳𑀺𑀲𑁆𑀲 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀲𑁆𑀲 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀓𑀺𑀁 𑀧𑀦 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑁆𑀲 𑀦 𑀤𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2240\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2241\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(594.)

2242\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁦*:**𑁩𑁯**)

(595.)

2243\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

2244\. **𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁** 𑀦𑀸𑀫 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀘𑀼𑀡𑁆𑀡𑀧𑀺𑀡𑁆𑀟𑁄𑀧𑀺 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀤𑀲𑀺𑀓𑀲𑀼𑀢𑁆𑀢𑀫𑁆𑀧𑀺𑁇

2245\. **𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀇𑀫𑀺𑀦𑀸 𑀇𑀫𑀁 𑀤𑁂𑀳𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀇𑀫𑀁 𑀆𑀳𑀭, 𑀇𑀫𑀺𑀦𑀸 𑀇𑀫𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀳𑀺, 𑀇𑀫𑀺𑀦𑀸 𑀇𑀫𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀳𑀻𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀸𑀘𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀬𑀢𑁄 𑀓𑀬𑀺𑀢𑀜𑁆𑀘 𑀳𑁄𑀢𑀺 𑀯𑀺𑀓𑁆𑀓𑀬𑀺𑀢𑀜𑁆𑀘 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀡𑁆𑀟𑀁 𑀧𑀭𑀳𑀢𑁆𑀣𑀕𑀢𑀁 𑀧𑀭𑀪𑀡𑁆𑀟𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀳𑀢𑁆𑀣𑀕𑀢𑀁, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀁𑁇 𑀇𑀤𑀁 𑀫𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(596.)

2246\. 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑁂 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀲𑀜𑁆𑀜𑀻, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑁂 𑀦𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀲𑀜𑁆𑀜𑀻, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2247\. 𑀦𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑁂 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑁂 𑀦𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(597.)

2248\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀕𑁆𑀖𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺, 𑀓𑀧𑁆𑀧𑀺𑀬𑀓𑀸𑀭𑀓𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀺, “𑀇𑀤𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀢𑁆𑀣𑀺, 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺 𑀪𑀡𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2249\. 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇

2250\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2251\. _𑀓𑁄𑀲𑀺𑀬𑀸 𑀲𑀼𑀤𑁆𑀥𑀤𑁆𑀯𑁂𑀪𑀸𑀕𑀸,_  
_𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀲𑀻𑀤𑀦𑀁;_  
_𑀤𑁆𑀯𑁂 𑀘 𑀮𑁄𑀫𑀸𑀦𑀺 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑁂,_  
_𑀉𑀪𑁄 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀸𑀢𑀺𑁇_
