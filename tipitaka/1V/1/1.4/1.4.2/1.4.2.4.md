# 𑁧𑁇𑁪𑁇𑁨𑁇𑁪 𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(557.)

2135\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀬𑀸𑀘𑀦𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  “𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀤𑁂𑀣𑁇 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑀸𑀘𑀦𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀤𑁂𑀣, 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀦 𑀲𑀓𑀺𑀁 𑀓𑀢𑀸𑀦𑀺 𑀲𑀦𑁆𑀣𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀧𑀺 𑀙𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀬𑁂𑀲𑀁 𑀦𑁄 𑀤𑀸𑀭𑀓𑀸 𑀉𑀳𑀤𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀫𑁆𑀫𑀺𑀳𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀦𑁆𑀤𑀽𑀭𑁂𑀳𑀺𑀧𑀺 𑀔𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀧𑀦 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀬𑀸𑀘𑀦𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  ‘𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀤𑁂𑀣, 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’”𑀢𑀺𑁇

2136\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑀸𑀘𑀦𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀤𑁂𑀣, 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’”𑀢𑀺𑁇

2137\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀬𑀸𑀘𑀦𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  ‘𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀤𑁂𑀣, 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀬𑀸𑀘𑀦𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀤𑁂𑀣, 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(558.)

2138\. **“𑀦𑀯𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀑𑀭𑁂𑀦 𑀘𑁂 𑀙𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀢𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀦𑀯𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇

2139\. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

2140\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀜𑀸𑀢𑀓𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀼𑀁—  “𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀪𑀤𑀦𑁆𑀢𑁄 𑀫𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀜𑀸𑀢𑀓𑀸 𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀪𑀕𑀯𑀢𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦𑀯𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀲𑀦𑁆𑀣𑀢𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀫𑀬𑁆𑀳𑀜𑁆𑀘 𑀯𑀺𑀦𑀸 𑀲𑀦𑁆𑀣𑀢𑀸 𑀦 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺𑁇 𑀦𑀸𑀳𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

2152\. **𑀢𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀤𑀢𑁆𑀯𑀸𑁇

2153\. **𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀦 𑀓𑀲𑁆𑀲𑀘𑀺 𑀤𑀢𑁆𑀯𑀸𑁇

2154\. **𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸**𑀢𑀺 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀅𑀜𑁆𑀜𑀁 𑀦𑀯𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀦𑁆𑀣𑀢𑀁 𑀊𑀦𑀓𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(563.)

2155\. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

(564.)

2156\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀯𑀺𑀢𑀸𑀦𑀁 𑀯𑀸 𑀪𑀽𑀫𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀁 𑀯𑀸 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
