# 𑁧𑁇𑁪𑁇𑁨𑁇𑁫 𑀦𑀺𑀲𑀻𑀤𑀦𑀲𑀦𑁆𑀣𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀲𑀻𑀤𑀦𑀲𑀦𑁆𑀣𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(565.)

2158. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁𑁇 𑀦𑀫𑁆𑀳𑀺 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑁆𑀲𑀼𑀥 𑀓𑁄𑀘𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀁𑀖𑁂𑀦 𑀓𑀢𑀺𑀓𑀸 𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  “𑀇𑀘𑁆𑀙𑀢𑀸𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁𑁇 𑀦 𑀪𑀕𑀯𑀸 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑁇 𑀬𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀲𑁄 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀤𑁂𑀲𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

2159. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀲𑁂𑀦𑁄 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑁄, 𑀲𑀧𑀭𑀺𑀲𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑁂𑀦𑀁 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺 𑀯𑁄, 𑀉𑀧𑀲𑁂𑀦, 𑀔𑀫𑀦𑀻𑀬𑀁 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀅𑀧𑁆𑀧𑀓𑀺𑀮𑀫𑀣𑁂𑀦 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀕𑀢𑀸”𑀢𑀺? “𑀔𑀫𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸, 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸𑁇 𑀅𑀧𑁆𑀧𑀓𑀺𑀮𑀫𑀣𑁂𑀦 𑀘 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀕𑀢𑀸”𑀢𑀺𑁇

2160. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑁂𑀦𑀲𑁆𑀲 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀦𑀸𑀧𑀸𑀦𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀻”𑀢𑀺? “𑀦 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀦𑀸𑀧𑀸𑀦𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄”𑀢𑀺? “𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄𑁇 𑀏𑀯𑀁 𑀅𑀳𑀫𑁆𑀧𑀺 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑁂𑀦𑀁 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸 𑀔𑁄 𑀢𑁆𑀬𑀸𑀬𑀁, 𑀉𑀧𑀲𑁂𑀦, 𑀧𑀭𑀺𑀲𑀸𑁇 𑀓𑀣𑀁 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀲𑁂𑀦, 𑀧𑀭𑀺𑀲𑀁 𑀯𑀺𑀦𑁂𑀲𑀻”𑀢𑀺? “𑀬𑁄 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀢𑀺 𑀢𑀫𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄𑁇 𑀲𑀘𑁂 𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄, 𑀏𑀯𑀸𑀳𑀁 𑀢𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀫𑁂 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀸𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀫𑀺, 𑀦𑁄 𑀘𑁂 𑀫𑁂 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀸𑀢𑀺 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀫𑀺𑁇 𑀬𑁄 𑀫𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀬𑀸𑀘𑀢𑀺 𑀢𑀫𑀳𑀁 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄𑁇 𑀲𑀘𑁂 𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄, 𑀏𑀯𑀸𑀳𑀁 𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀫𑁂 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀸𑀢𑀺 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀫𑀺, 𑀦𑁄 𑀘𑁂 𑀫𑁂 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀸𑀢𑀺 𑀦 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀫𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀭𑀺𑀲𑀁 𑀯𑀺𑀦𑁂𑀫𑀻”𑀢𑀺𑁇

2162. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀞𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  “𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑁂𑀦𑀁 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀤𑁂𑀲𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀲𑁂𑀦𑁄 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀧𑀭𑀺𑀲𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑁂𑀦𑀁 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀚𑀸𑀦𑀸𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀉𑀧𑀲𑁂𑀦, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀢𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 “𑀪𑀕𑀯𑀸𑀧𑀺 𑀫𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀫𑀸𑀳—  ‘𑀚𑀸𑀦𑀸𑀲𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀲𑁂𑀦, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀢𑀺𑀓𑀦𑁆’𑀢𑀺? ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀦𑀸𑀫𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀢𑀺𑀓𑀦𑁆’𑀢𑀺𑁇 ‘𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀔𑁄, 𑀉𑀧𑀲𑁂𑀦, 𑀲𑀁𑀖𑁂𑀦 𑀓𑀢𑀺𑀓𑀸 𑀓𑀢𑀸—  𑀇𑀘𑁆𑀙𑀢𑀸𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑀼𑀁𑁇 𑀦 𑀪𑀕𑀯𑀸 𑀓𑁂𑀦𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀏𑀓𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀦𑀻𑀳𑀸𑀭𑀓𑁂𑀦𑁇 𑀬𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀲𑁄 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀤𑁂𑀲𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 ‘𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀁𑀖𑁄 𑀲𑀓𑀸𑀬 𑀓𑀢𑀺𑀓𑀸𑀬, 𑀦 𑀫𑀬𑀁 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀯𑀸 𑀦 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀸𑀫, 𑀬𑀣𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀲𑀫𑀸𑀤𑀸𑀬 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸 𑀬𑀣𑀸𑀲𑀼𑀔𑀁 𑀫𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀽’”𑀢𑀺𑁇

2163. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀲𑀘𑁆𑀘𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀲𑁂𑀦𑁄 𑀆𑀳—  ‘𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀩𑁆𑀩𑀁, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀯𑀸 𑀦 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁, 𑀬𑀣𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀲𑀫𑀸𑀤𑀸𑀬 𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀦𑁆’”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸 𑀓𑀺𑀭 𑀪𑀕𑀯𑀢𑀸—  ‘𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸 𑀬𑀣𑀸𑀲𑀼𑀔𑀁 𑀫𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀽’”𑀢𑀺𑁇 𑀢𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀺𑀳𑁂𑀦𑁆𑀢𑀸 𑀲𑀦𑁆𑀣𑀢𑀸𑀦𑀺 𑀉𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀗𑁆𑀕𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀗𑁆𑀕𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀗𑁆𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀲𑀦𑁆𑀣𑀢𑀸𑀦𑀺 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀉𑀚𑁆𑀛𑀺𑀢𑀸𑀦𑀺𑁇 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀲𑁆𑀲𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑁆𑀣𑀢𑀸𑀦𑀺 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀉𑀚𑁆𑀛𑀺𑀢𑀸𑀦𑀻”𑀢𑀺?

2164. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬, 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑀢𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

2169. **𑀧𑀼𑀭𑀸𑀡𑀲𑀦𑁆𑀣𑀢𑀁** 𑀦𑀸𑀫 𑀲𑀓𑀺𑀁 𑀦𑀺𑀯𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀲𑀓𑀺𑀁 𑀧𑀸𑀭𑀼𑀢𑀫𑁆𑀧𑀺𑁇

2170. **𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀸 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀓𑀭𑀡𑀸𑀬𑀸**𑀢𑀺 𑀣𑀺𑀭𑀪𑀸𑀯𑀸𑀬 𑀯𑀝𑁆𑀝𑀁 𑀯𑀸 𑀘𑀢𑀼𑀭𑀲𑁆𑀲𑀁 𑀯𑀸 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀤𑁂𑀲𑁂 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑀩𑁆𑀩𑀁 𑀯𑀺𑀚𑀝𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇

2171. **𑀅𑀦𑀸𑀤𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀭𑀸𑀡𑀲𑀦𑁆𑀣𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀦𑁆**𑀢𑀺 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀸𑀡𑀲𑀦𑁆𑀣𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀁 𑀦𑀯𑀁 𑀦𑀺𑀲𑀻𑀤𑀦𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀲𑀻𑀤𑀦𑀲𑀦𑁆𑀣𑀢𑀁 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀸𑀡𑀲𑀦𑁆𑀣𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(569.)

2172. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2173. 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(570.)

2174. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀼𑀭𑀸𑀡𑀲𑀦𑁆𑀣𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀮𑀪𑀦𑁆𑀢𑁄 𑀣𑁄𑀓𑀢𑀭𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀮𑀪𑀦𑁆𑀢𑁄 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀯𑀺𑀢𑀸𑀦𑀁 𑀯𑀸 𑀪𑀽𑀫𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀁 𑀯𑀸 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
