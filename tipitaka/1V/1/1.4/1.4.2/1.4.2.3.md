# 𑁧𑁇𑁪𑁇𑁨𑁇𑁩 𑀤𑁆𑀯𑁂𑀪𑀸𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑁆𑀯𑁂𑀪𑀸𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(552.)

2121. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀢𑁂 𑀣𑁄𑀓𑀜𑁆𑀜𑁂𑀯 𑀑𑀤𑀸𑀢𑀁 𑀅𑀦𑁆𑀢𑁂 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑁂𑀯 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁄𑀓𑀜𑁆𑀜𑁂𑀯 𑀑𑀤𑀸𑀢𑀁 𑀅𑀦𑁆𑀢𑁂 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑁂𑀯 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2122. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑁄𑀓𑀜𑁆𑀜𑁂𑀯 𑀑𑀤𑀸𑀢𑀁 𑀅𑀦𑁆𑀢𑁂 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑁂𑀯 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀣𑁄𑀓𑀜𑁆𑀜𑁂𑀯 𑀑𑀤𑀸𑀢𑀁 𑀅𑀦𑁆𑀢𑁂 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑁂𑀯 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(553.)

2123. **“𑀦𑀯𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀸𑀕𑀸 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀸 𑀢𑀢𑀺𑀬𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁𑁇 𑀅𑀦𑀸𑀤𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑁆𑀯𑁂 𑀪𑀸𑀕𑁂 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀦𑀯𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁩*:**𑁩𑁨**)

(554.)

2124. **𑀦𑀯𑀁** 𑀦𑀸𑀫 𑀓𑀭𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2125. **𑀲𑀦𑁆𑀣𑀢𑀁** 𑀦𑀸𑀫 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀸𑀬𑀺𑀫𑀁𑁇

2126. **𑀓𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀦𑀸**𑀢𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀯𑀸𑁇

2127. **𑀤𑁆𑀯𑁂 𑀪𑀸𑀕𑀸 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀸**𑀢𑀺 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀢𑀼𑀮𑀸 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀸𑁇

2128. **𑀢𑀢𑀺𑀬𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀦𑁆**𑀢𑀺 𑀢𑀼𑀮𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁𑁇

2129. **𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀦𑁆**𑀢𑀺 𑀢𑀼𑀮𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁𑁇

2130. **𑀅𑀦𑀸𑀤𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑁆𑀯𑁂 𑀪𑀸𑀕𑁂 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀦𑁆**𑀢𑀺𑁇 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀢𑀼𑀮𑁂 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀢𑀼𑀮𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀢𑀼𑀮𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀦𑀯𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀦𑁆𑀣𑀢𑀁 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀢𑀼𑀮𑁂 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀢𑀼𑀮𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀢𑀼𑀮𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(555.)

2131. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2132. 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(556.)

2133. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀢𑀼𑀮𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀢𑀼𑀮𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀩𑀳𑀼𑀢𑀭𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀩𑀳𑀼𑀢𑀭𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀲𑀼𑀤𑁆𑀥𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀲𑀼𑀤𑁆𑀥𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀯𑀺𑀢𑀸𑀦𑀁 𑀯𑀸 𑀪𑀽𑀫𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀁 𑀯𑀸 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
