# 𑁧𑁇𑁪𑁇𑁩𑁇𑁨 𑀊𑀦𑀧𑀜𑁆𑀘𑀩𑀦𑁆𑀥𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀊𑀦𑀧𑀜𑁆𑀘𑀩𑀦𑁆𑀥𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(609.)

2273\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  “𑀬𑁂𑀲𑀁 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀅𑀳𑀁 𑀧𑀢𑁆𑀢𑁂𑀦𑀸”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑁂𑀲𑀁 𑀔𑀼𑀤𑁆𑀤𑀓𑀸 𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀫𑀳𑀦𑁆𑀢𑁂 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑁂𑀲𑀁 𑀫𑀳𑀦𑁆𑀢𑀸 𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀔𑀼𑀤𑁆𑀤𑀓𑁂 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀓𑁆𑀓𑀸𑀬𑀺𑀓𑀁 𑀪𑀡𑁆𑀟𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀦 𑀬𑀸𑀧𑁂𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀸𑀧𑀺𑀲𑁆𑀲 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀓𑁆𑀓𑀸𑀬𑀺𑀓𑀁 𑀪𑀡𑁆𑀟𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀦 𑀬𑀸𑀧𑁂𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀸𑀧𑀺𑀲𑁆𑀲 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2274\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2275\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀻𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀢𑁆𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

(610.)

2276\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀢𑁆𑀢𑁄 𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑁄 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑁂𑀲𑀼 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑀢𑁆𑀣𑁂𑀲𑀼 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸”𑀢𑀺𑁇

2277\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑀺𑀦𑁆𑀦𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(611.)

2278\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀦𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑀺𑀦𑁆𑀦𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀪𑀺𑀦𑁆𑀦𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀔𑀡𑁆𑀟𑁂𑀦 𑀯𑀺𑀮𑀺𑀔𑀺𑀢𑀫𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀣𑁂𑀯 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀓𑁆𑀓𑀸𑀬𑀺𑀓𑀁 𑀪𑀡𑁆𑀟𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀦 𑀬𑀸𑀧𑁂𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀸𑀧𑀺𑀲𑁆𑀲 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀣𑁂𑀯 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀓𑁆𑀓𑀸𑀬𑀺𑀓𑀁 𑀪𑀡𑁆𑀟𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀦 𑀬𑀸𑀧𑁂𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀸𑀧𑀺𑀲𑁆𑀲 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2279\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀪𑀺𑀦𑁆𑀦𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀔𑀡𑁆𑀟𑁂𑀦 𑀯𑀺𑀮𑀺𑀔𑀺𑀢𑀫𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2280\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀪𑀺𑀦𑁆𑀦𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀔𑀡𑁆𑀟𑁂𑀦 𑀯𑀺𑀮𑀺𑀔𑀺𑀢𑀫𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀪𑀺𑀦𑁆𑀦𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀔𑀡𑁆𑀟𑁂𑀦 𑀯𑀺𑀮𑀺𑀔𑀺𑀢𑀫𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(612.)

2281\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀊𑀦𑀧𑀜𑁆𑀘𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀦𑀯𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑁄 𑀧𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀭𑀺𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀘 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁄 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀤𑀸𑀢𑀩𑁆𑀩𑁄—  ‘𑀅𑀬𑀁 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀢𑁆𑀢𑁄 𑀬𑀸𑀯 𑀪𑁂𑀤𑀦𑀸𑀬 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀸𑀫𑀻𑀘𑀻”**𑀢𑀺𑁇 (*𑁨𑁨*:**𑁪𑁧**)

(613.)

2282\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

2283\. **𑀊𑀦𑀧𑀜𑁆𑀘𑀩𑀦𑁆𑀥𑀦𑁄** 𑀦𑀸𑀫 𑀧𑀢𑁆𑀢𑁄 𑀅𑀩𑀦𑁆𑀥𑀦𑁄 𑀯𑀸 𑀏𑀓𑀩𑀦𑁆𑀥𑀦𑁄 𑀯𑀸 𑀤𑁆𑀯𑀺𑀩𑀦𑁆𑀥𑀦𑁄 𑀯𑀸 𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑁄 𑀯𑀸 𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑁄 𑀯𑀸𑁇 𑀅𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁄 𑀦𑀸𑀫 𑀧𑀢𑁆𑀢𑁄 𑀬𑀲𑁆𑀲 𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀸 𑀭𑀸𑀚𑀺 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁄 𑀦𑀸𑀫 𑀧𑀢𑁆𑀢𑁄 𑀬𑀲𑁆𑀲 𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀸 𑀭𑀸𑀚𑀺 𑀳𑁄𑀢𑀺𑁇 **𑀦𑀯𑁄** 𑀦𑀸𑀫 𑀧𑀢𑁆𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2284\. **𑀘𑁂𑀢𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀅𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀦 𑀮𑀸𑀫𑀓𑁄 𑀧𑀢𑁆𑀢𑁄 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄—  “𑀫𑀳𑀕𑁆𑀖𑀁 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀮𑀸𑀫𑀓𑀁 𑀧𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀢𑀺—  “𑀫𑀳𑀕𑁆𑀖𑀁 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  “𑀅𑀬𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀢𑁆𑀢𑁄 𑀊𑀦𑀧𑀜𑁆𑀘𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄—  𑀬𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀕𑀸𑀳𑀺𑀢𑀸𑀕𑀸𑀳𑀺𑀢𑀜𑁆𑀘 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀧𑀞𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀸𑀘𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(614.)

2285\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

2286\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

2287\. 𑀲𑀫𑁆𑀫𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(615.)

2288\. 𑀢𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀢𑁆𑀢𑁄 𑀕𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀣𑁂𑀭𑁄 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀕𑀡𑁆𑀳𑀸𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀣𑁂𑀭𑁄 𑀧𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀣𑁂𑀭𑁄 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀣𑁂𑀭𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑁄 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀕𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀦 𑀘 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀤𑁆𑀤𑀬𑀢𑀸𑀬 𑀦 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀦 𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀧𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀦 𑀕𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀏𑀢𑁂𑀦𑁂𑀯 𑀉𑀧𑀸𑀬𑁂𑀦 𑀬𑀸𑀯 𑀲𑀁𑀖𑀦𑀯𑀓𑀸 𑀕𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀘 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀤𑀸𑀢𑀩𑁆𑀩𑁄—  “𑀅𑀬𑀁 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀧𑀢𑁆𑀢𑁄 𑀬𑀸𑀯 𑀪𑁂𑀤𑀦𑀸𑀬 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

2289\. 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑁄 𑀧𑀢𑁆𑀢𑁄 𑀦 𑀅𑀤𑁂𑀲𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀅𑀪𑁄𑀕𑁂𑀦 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀩𑁆𑀩𑁄—  “𑀓𑀣𑀸𑀬𑀁 𑀧𑀢𑁆𑀢𑁄 𑀦𑀲𑁆𑀲𑁂𑀬𑁆𑀬 𑀯𑀸 𑀯𑀺𑀦𑀲𑁆𑀲𑁂𑀬𑁆𑀬 𑀯𑀸 𑀪𑀺𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸”𑀢𑀺? 𑀲𑀘𑁂 𑀅𑀤𑁂𑀲𑁂 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀢𑀺 𑀅𑀪𑁄𑀕𑁂𑀦 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2290\. **𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀸𑀫𑀻𑀘𑀻**𑀢𑀺 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸𑁇

2294\. 𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀏𑀓𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁 𑀤𑁆𑀯𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2295\. 𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀏𑀓𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀤𑁆𑀯𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2301\. 𑀅𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀏𑀓𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀤𑁆𑀯𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2302\. 𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀏𑀓𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀤𑁆𑀯𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2303\. 𑀅𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀏𑀓𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀤𑁆𑀯𑀺𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2304\. 𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀅𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…𑀧𑁂…  𑀏𑀓𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀤𑁆𑀯𑀺𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀢𑀺𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁…  𑀘𑀢𑀼𑀩𑀦𑁆𑀥𑀦𑁄𑀓𑀸𑀲𑀁 𑀧𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

(617.)

2305\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀦𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑀲𑁆𑀲, 𑀪𑀺𑀦𑁆𑀦𑀧𑀢𑁆𑀢𑀲𑁆𑀲, 𑀜𑀸𑀢𑀓𑀸𑀦𑀁 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀥𑀦𑁂𑀦, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
