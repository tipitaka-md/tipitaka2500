# 𑁧𑁇𑁪𑁇𑁩𑁇𑁧𑁦 𑀧𑀭𑀺𑀡𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀭𑀺𑀡𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(657.)

2421\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀧𑀽𑀕𑀲𑁆𑀲 𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑀘𑀻𑀯𑀭𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺—  “𑀪𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀲𑁄 𑀧𑀽𑀕𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀧𑀽𑀕𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀤𑁂𑀣𑀸𑀯𑀼𑀲𑁄, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀇𑀫𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀦 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀲𑁆𑀲𑀸𑀫𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀘𑀻𑀯𑀭𑀪𑀺𑀓𑁆𑀔𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸”𑀢𑀺𑁇 “𑀩𑀳𑀽, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀸𑀬𑀓𑀸, 𑀩𑀳𑀽 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀸𑁇 𑀫𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀢𑀼𑀫𑁆𑀳𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀇𑀥 𑀯𑀺𑀳𑀭𑀸𑀫𑁇 𑀢𑀼𑀫𑁆𑀳𑁂 𑀘𑁂 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀦 𑀤𑀲𑁆𑀲𑀣, 𑀅𑀣 𑀓𑁄 𑀘𑀭𑀳𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑀲𑁆𑀲𑀢𑀺? 𑀤𑁂𑀣𑀸𑀯𑀼𑀲𑁄, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀇𑀫𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀽𑀕𑁄 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀦𑀺𑀧𑁆𑀧𑀻𑀍𑀇𑀬𑀫𑀸𑀦𑁄 𑀬𑀣𑀸𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑀘𑀻𑀯𑀭𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁, 𑀦 𑀘 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀑𑀡𑁄𑀚𑁂𑀣𑀸𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺𑁇 “𑀦𑀢𑁆𑀣𑀺, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀬𑀣𑀸𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀬𑁆𑀬𑀸 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀲𑀼𑀦𑁆”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑀁 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2422\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀦𑀁 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀚𑀸𑀦𑀁 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(658.)

2423\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀚𑀸𑀦𑀁 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩𑁦*:**𑁪𑁯**)

(659.)

2424\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

2425\. **𑀚𑀸𑀦𑀸𑀢𑀺** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀚𑀸𑀦𑀸𑀢𑀺 𑀅𑀜𑁆𑀜𑁂 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺 𑀲𑁄 𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺𑁇

2426\. **𑀲𑀁𑀖𑀺𑀓𑀁** 𑀦𑀸𑀫 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑁇

2427\. **𑀮𑀸𑀪𑁄** 𑀦𑀸𑀫 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑁇 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀘𑀼𑀡𑁆𑀡𑀧𑀺𑀡𑁆𑀟𑁄𑀧𑀺, 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀫𑁆𑀧𑀺, 𑀤𑀲𑀺𑀓𑀲𑀼𑀢𑁆𑀢𑀫𑁆𑀧𑀺𑁇

2428\. **𑀧𑀭𑀺𑀡𑀢𑀁** 𑀦𑀸𑀫 𑀤𑀲𑁆𑀲𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸𑀢𑀺 𑀯𑀸𑀘𑀸 𑀪𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

2429\. 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀦𑀁 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(660.)

2430\. 𑀧𑀭𑀺𑀡𑀢𑁂 𑀧𑀭𑀺𑀡𑀢𑀲𑀜𑁆𑀜𑀻 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2431\. 𑀧𑀭𑀺𑀡𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀭𑀺𑀡𑀢𑁂 𑀅𑀧𑀭𑀺𑀡𑀢𑀲𑀜𑁆𑀜𑀻 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀜𑁆𑀜𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀘𑁂𑀢𑀺𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀘𑁂𑀢𑀺𑀬𑀲𑁆𑀲 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀜𑁆𑀜𑀘𑁂𑀢𑀺𑀬𑀲𑁆𑀲 𑀯𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀧𑀭𑀺𑀡𑀢𑀁 𑀅𑀜𑁆𑀜𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀘𑁂𑀢𑀺𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀧𑀭𑀺𑀡𑀢𑁂 𑀧𑀭𑀺𑀡𑀢𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀧𑀭𑀺𑀡𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀧𑀭𑀺𑀡𑀢𑁂 𑀅𑀧𑀭𑀺𑀡𑀢𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(661.)

2432\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀓𑀢𑁆𑀣 𑀤𑁂𑀫𑀸𑀢𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀬𑀫𑀸𑀦𑁄 𑀬𑀢𑁆𑀣 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀪𑁄𑀕𑀁 𑀯𑀸 𑀮𑀪𑁂𑀬𑁆𑀬 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀯𑀸 𑀮𑀪𑁂𑀬𑁆𑀬 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑁄 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀢𑀺 𑀢𑀢𑁆𑀣 𑀤𑁂𑀣𑀸𑀢𑀺 𑀪𑀡𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2433\. 𑀧𑀭𑀺𑀡𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

2434\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2435\. _𑀤𑁆𑀯𑁂 𑀘 𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀁,_  
_𑀯𑀲𑁆𑀲𑀺𑀓𑀸 𑀤𑀸𑀦𑀧𑀜𑁆𑀘𑀫𑀁;_  
_𑀲𑀸𑀫𑀁 𑀯𑀸𑀬𑀸𑀧𑀦𑀘𑁆𑀘𑁂𑀓𑁄,_  
_𑀲𑀸𑀲𑀗𑁆𑀓𑀁 𑀲𑀁𑀖𑀺𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_  


(662.)

2436\. “𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀢𑀺𑀁𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  ‘𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸’? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  ‘𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸’? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  ‘𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸’? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻”𑀢𑀺𑁇

---

2437\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
