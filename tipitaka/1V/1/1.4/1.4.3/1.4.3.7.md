# 𑁧𑁇𑁪𑁇𑁩𑁇𑁭 𑀫𑀳𑀸𑀧𑁂𑀲𑀓𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀫𑀳𑀸𑀧𑁂𑀲𑀓𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(641.)

2362. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑀼𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑀲𑁆𑀲 𑀤𑁂𑀳𑀺, 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸𑀬𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧, 𑀆𑀕𑀢𑁄 𑀅𑀬𑁆𑀬𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀇𑀫𑀁 𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀳𑀸𑀧𑀼𑀜𑁆𑀜𑁄𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀑𑀓𑀸𑀲𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀲𑀼𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑀼𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑀲𑁆𑀲 𑀤𑁂𑀳𑀺, 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸𑀬𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧, 𑀆𑀕𑀢𑁄 𑀅𑀬𑁆𑀬𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀫𑀁 𑀲𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄”𑀢𑀺𑁇 𑀲𑁄𑀧𑀺 𑀔𑁄 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

2363. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀲𑁄 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀁 𑀫𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀯𑀺𑀬𑁆𑀬𑀢𑀺; 𑀆𑀬𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀳𑀺 𑀯𑀺𑀢𑁆𑀣𑀢𑀜𑁆𑀘𑁇 𑀅𑀧𑁆𑀧𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀻𑀢𑀜𑁆𑀘 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀬𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀮𑁂𑀔𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀢𑀘𑁆𑀙𑀺𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀢𑁂 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀤𑀁𑀲𑀼; 𑀇𑀫𑀺𑀦𑀸 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑀸 𑀆𑀬𑀢𑀁 𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀢𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀁 𑀯𑀸 𑀓𑀸𑀢𑀼𑀁𑁇 𑀲𑀓𑁆𑀓𑀸 𑀘 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀯𑀻𑀢𑀜𑁆𑀘 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀬𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀮𑁂𑀔𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀢𑀘𑁆𑀙𑀺𑀢𑀜𑁆𑀘 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀳𑀺 𑀯𑀺𑀢𑁆𑀣𑀢𑀜𑁆𑀘 𑀅𑀧𑁆𑀧𑀺𑀢𑀜𑁆𑀘𑁇 𑀦 𑀢𑁂𑀦 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2364. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀲𑀼𑀢𑁆𑀢𑀁 𑀢𑀦𑁆𑀢𑁂 𑀉𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸 𑀇𑀢𑁆𑀣𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀢𑁆𑀢𑁂𑀦, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀢𑁆𑀯𑀁 𑀅𑀬𑁆𑀬𑁄 𑀫𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑁄—  ‘𑀇𑀫𑀺𑀦𑀸 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀦𑀸𑀳𑀻’”𑀢𑀺𑁇 “𑀲𑀘𑁆𑀘𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑁄—  ‘𑀇𑀫𑀺𑀦𑀸 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀦𑀸𑀳𑀻’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀫𑀁 𑀅𑀬𑁆𑀬𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀳𑀺 𑀯𑀺𑀢𑁆𑀣𑀢𑀜𑁆𑀘 𑀅𑀧𑁆𑀧𑀺𑀢𑀜𑁆𑀘, 𑀦 𑀢𑁂𑀦 𑀲𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀇𑀢𑁆𑀣𑀻 𑀬𑀢𑁆𑀢𑀓𑀁𑀬𑁂𑀯 𑀲𑀼𑀢𑁆𑀢𑀁 𑀧𑀞𑀫𑀁 𑀅𑀤𑀸𑀲𑀺 𑀢𑀢𑁆𑀢𑀓𑀁 𑀧𑀘𑁆𑀙𑀸 𑀅𑀤𑀸𑀲𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀲𑁄 𑀓𑀺𑀭 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀯𑀸𑀲𑀢𑁄 𑀆𑀕𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀻𑀢𑀁 𑀢𑀁 𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺? “𑀆𑀫𑀸𑀬𑁆𑀬, 𑀯𑀻𑀢𑀁 𑀢𑀁 𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺𑁇 “𑀆𑀳𑀭, 𑀅𑀬𑁆𑀬𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀇𑀢𑁆𑀣𑀻 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀫𑀳𑀺𑀘𑁆𑀙𑀸 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸𑁇 𑀦𑀬𑀺𑀫𑁂 𑀲𑀼𑀓𑀭𑀸 𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀼𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀫𑀬𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2365. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀕𑀳𑀧𑀢𑀺𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2366. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀕𑀳𑀧𑀢𑀺𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀜𑀸𑀢𑀓𑁄 𑀢𑁂, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀲𑁆𑀲 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀦𑁆𑀢𑀁 𑀯𑀸 𑀅𑀲𑀦𑁆𑀢𑀁 𑀯𑀸𑁇 𑀢𑀢𑁆𑀣 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

2367. **“𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀦𑁂𑀯 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀯𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻 𑀯𑀸 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂𑀳𑀺 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸𑀬𑀸𑀧𑁂𑀬𑁆𑀬, 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀇𑀤𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀁 𑀫𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀯𑀺𑀬𑁆𑀬𑀢𑀺𑁇 𑀆𑀬𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀣 𑀯𑀺𑀢𑁆𑀣𑀢𑀜𑁆𑀘𑁇 𑀅𑀧𑁆𑀧𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀻𑀢𑀜𑁆𑀘 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀬𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀮𑁂𑀔𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀢𑀘𑁆𑀙𑀺𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀣𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀫𑀬𑀫𑁆𑀧𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀜𑁆𑀘𑀺𑀫𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀧𑀤𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀢𑁆𑀯𑀸 𑀓𑀺𑀜𑁆𑀘𑀺𑀫𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀧𑀤𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁭*:**𑁪𑁬**)

(643.)

2368. **𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀦𑁂𑀯 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀸**𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀼𑀓𑀸𑀫𑁄𑁇

2369. **𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄** 𑀦𑀸𑀫 𑀫𑀸𑀢𑀺𑀢𑁄 𑀯𑀸 𑀧𑀺𑀢𑀺𑀢𑁄 𑀯𑀸 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀲𑀫𑁆𑀩𑀤𑁆𑀥𑁄𑁇

2370. **𑀕𑀳𑀧𑀢𑀺** 𑀦𑀸𑀫 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇

2371. **𑀕𑀳𑀧𑀢𑀸𑀦𑀻** 𑀦𑀸𑀫 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇

2372. **𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂𑀳𑀻**𑀢𑀺 𑀧𑁂𑀲𑀓𑀸𑀭𑁂𑀳𑀺𑁇

2373. **𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀙𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀼𑀧𑀕𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

2374. **𑀯𑀸𑀬𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀯𑀺𑀦𑀸𑀧𑁂𑀢𑀺𑁇

2375. **𑀢𑀢𑁆𑀭 𑀘𑁂 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺 𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀬𑁆𑀬𑀢𑀺 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑁇

2376. **𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄**𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀯𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺—  “𑀓𑀻𑀤𑀺𑀲𑁂𑀦 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑁄, 𑀓𑀻𑀤𑀺𑀲𑀁 𑀢𑁂 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸𑀬𑀸𑀧𑁂𑀫𑀻”𑀢𑀺?

2377. **𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸**𑀢𑀺 𑀖𑀭𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸𑁇

2378. **𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸**𑀢𑀺—  “𑀇𑀤𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀁 𑀫𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀯𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀬𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀣 𑀯𑀺𑀢𑁆𑀣𑀢𑀜𑁆𑀘𑁇 𑀅𑀧𑁆𑀧𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀻𑀢𑀜𑁆𑀘 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀬𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀮𑁂𑀔𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀯𑀺𑀢𑀘𑁆𑀙𑀺𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀣𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀫𑀬𑀫𑁆𑀧𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀜𑁆𑀘𑀺𑀫𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀧𑀤𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇

2379. **𑀏𑀯𑀜𑁆𑀘 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀢𑁆𑀯𑀸 𑀓𑀺𑀜𑁆𑀘𑀺𑀫𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀧𑀤𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀻**𑀢𑀺𑁇 **𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄** 𑀦𑀸𑀫 𑀬𑀸𑀕𑀼𑀧𑀺 𑀪𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀘𑀼𑀡𑁆𑀡𑀧𑀺𑀡𑁆𑀟𑁄𑀧𑀺 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀤𑀲𑀺𑀓𑀲𑀼𑀢𑁆𑀢𑀫𑁆𑀧𑀺, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀪𑀡𑀢𑀺𑁇

2380. 𑀢𑀲𑁆𑀲 𑀯𑀘𑀦𑁂𑀦 𑀆𑀬𑀢𑀁 𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀢𑀁 𑀯𑀸 𑀅𑀧𑁆𑀧𑀺𑀢𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀦𑁆𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(644.)

2381. 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀕𑀳𑀧𑀢𑀺𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀕𑀳𑀧𑀢𑀺𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁂 𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀕𑀳𑀧𑀢𑀺𑀓𑀲𑁆𑀲 𑀢𑀦𑁆𑀢𑀯𑀸𑀬𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2382. 𑀜𑀸𑀢𑀓𑁂 𑀅𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀸𑀢𑀓𑁂 𑀜𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(645.)

2383. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀜𑀸𑀢𑀓𑀸𑀦𑀁, 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀥𑀦𑁂𑀦, 𑀫𑀳𑀕𑁆𑀖𑀁 𑀯𑀸𑀬𑀸𑀧𑁂𑀢𑀼𑀓𑀸𑀫𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀕𑁆𑀖𑀁 𑀯𑀸𑀬𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2384. 𑀫𑀳𑀸𑀧𑁂𑀲𑀓𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
