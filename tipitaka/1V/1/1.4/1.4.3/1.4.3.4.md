# 𑁧𑁇𑁪𑁇𑁩𑁇𑁪 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(626.)

2323\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸”𑀢𑀺, 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀺, 𑀚𑀺𑀡𑁆𑀡𑀸𑀬 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬 𑀦𑀕𑁆𑀕𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀚𑀺𑀡𑁆𑀡𑀸𑀬 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬 𑀦𑀕𑁆𑀕𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2324\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀣? 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀣? 𑀚𑀺𑀡𑁆𑀡𑀸𑀬 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬 𑀦𑀕𑁆𑀕𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀣, 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀲𑁆𑀲𑀣, 𑀚𑀺𑀡𑁆𑀡𑀸𑀬 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬 𑀦𑀕𑁆𑀕𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(627.)

2325\. **“‘𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆’𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀁**; **‘𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑁄** **𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆’𑀢𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀁**𑁇 **‘𑀑𑀭𑁂𑀦 𑀘𑁂 𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆’𑀢𑀺 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑁂𑀬𑁆𑀬, ‘𑀑𑀭𑁂𑀦𑀤𑁆𑀥𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆’𑀢𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁪*:**𑁪𑁩**)

(628.)

2326\. **“𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀦𑁆**𑀢𑀺𑁇 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀧𑀺 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑀺 𑀢𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  “𑀓𑀸𑀮𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬, 𑀲𑀫𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬, 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀤𑁂𑀣 𑀫𑁂 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁, 𑀆𑀳𑀭𑀣 𑀫𑁂 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁, 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀣 𑀫𑁂 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁, 𑀘𑁂𑀢𑀸𑀧𑁂𑀣 𑀫𑁂 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺𑁇

2327\. **“𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”𑀢𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀦𑁆**𑀢𑀺𑁇 𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀁𑁇

2328\. **“𑀑𑀭𑁂𑀦 𑀘𑁂 𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”**𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2329\. **“𑀑𑀭𑁂𑀦𑀤𑁆𑀥𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”**𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀧𑀭𑀺𑀬𑀺𑀝𑁆𑀞𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀓𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀤𑀳𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(629.)

2330\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀻 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀻 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2331\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀻 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀻 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2332\. 𑀲𑀢𑀺𑀬𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬 𑀦𑀕𑁆𑀕𑁄 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

2333\. 𑀊𑀦𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀲𑁂𑀲𑁂 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(630.)

2334\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  “𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”𑀢𑀺 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, “𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”𑀢𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺, “𑀊𑀦𑀓𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”𑀢𑀺 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, “𑀊𑀦𑀓𑀤𑁆𑀥𑀫𑀸𑀲𑁄 𑀲𑁂𑀲𑁄 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀦𑁆”𑀢𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺, 𑀧𑀭𑀺𑀬𑀺𑀝𑁆𑀞𑀸𑀬 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬 𑀯𑀲𑁆𑀲𑀁 𑀉𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑀬𑁆𑀬𑀢𑀺, 𑀦𑀺𑀯𑀢𑁆𑀣𑀸𑀬 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬 𑀯𑀲𑁆𑀲𑀁 𑀉𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑀬𑁆𑀬𑀢𑀺, 𑀥𑁄𑀯𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑀁; 𑀲𑀫𑀬𑁂 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀘𑀻𑀯𑀭𑀲𑁆𑀲, 𑀦𑀝𑁆𑀞𑀘𑀻𑀯𑀭𑀲𑁆𑀲, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2335\. 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
