# 𑁧𑁇𑁪𑁇𑁩𑁇𑁫 𑀘𑀻𑀯𑀭𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀘𑀻𑀯𑀭𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(631.)

2336\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀸𑀢𑀼𑀦𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀸𑀯𑀼𑀲𑁄, 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺; 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁄𑀫𑁆𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀸𑀯𑀼𑀲𑁄, 𑀅𑀳𑀁 𑀢𑁂 𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦 𑀤𑀸𑀦𑀸𑀳𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺 𑀤𑀸𑀦𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀬𑀫𑁆𑀧𑀺 𑀢𑁆𑀬𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑀁, 𑀫𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀸𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2337\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀸𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀸𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(632.)

2338\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀸𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁫*:**𑁪𑁪**)

(633.)

2339\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

2340\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑁇

2341\. **𑀲𑀸𑀫𑀦𑁆**𑀢𑀺 𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸𑁇

2342\. **𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀙𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁, 𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀼𑀧𑀕𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

2343\. **𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄**𑀢𑀺 𑀅𑀦𑀪𑀺𑀭𑀤𑁆𑀥𑁄 𑀆𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑀺𑀮𑀚𑀸𑀢𑁄𑁇

2344\. **𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2345\. **𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀓𑀺𑀁 𑀆𑀡𑀢𑁆𑀢𑁄 𑀩𑀳𑀼𑀓𑀫𑁆𑀧𑀺 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀸𑀫𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(634.)

2346\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2347\. 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(635.)

2348\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑁄 𑀯𑀸 𑀤𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀯𑀸 𑀯𑀺𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
