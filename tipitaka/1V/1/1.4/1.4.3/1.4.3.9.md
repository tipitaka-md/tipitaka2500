# 𑁧𑁇𑁪𑁇𑁩𑁇𑁯 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(652.)

2400\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀼𑀢𑁆𑀣𑀯𑀲𑁆𑀲𑀸 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑁆𑀢𑀺𑀓𑀘𑁄𑀭𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑀸”𑀢𑀺 𑀧𑀭𑀺𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑀦 𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

2401\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑀦 𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀼𑀦𑁆”𑀢𑀺 𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀟𑀬𑁆𑀳𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀦𑁆𑀤𑀽𑀭𑁂𑀳𑀺𑀧𑀺 𑀔𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀘𑁆𑀘𑁄𑀍𑀆 𑀳𑁄𑀦𑁆𑀢𑀺 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀘𑁆𑀘𑁄𑀍𑀆 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2402\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(653.)

2403\. **“𑀉𑀧𑀯𑀲𑁆𑀲𑀁 𑀔𑁄 𑀧𑀦 𑀓𑀢𑁆𑀢𑀺𑀓𑀧𑀼𑀡𑁆𑀡𑀫𑀁 𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀆𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀸𑀦𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬, 𑀲𑀺𑀬𑀸 𑀘 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀘𑁆𑀘𑀬𑁄 𑀢𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬𑁇 𑀙𑀸𑀭𑀢𑁆𑀢𑀧𑀭𑀫𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀢𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀢𑁄 𑀘𑁂 𑀉𑀢𑁆𑀢𑀭𑀺 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀬𑁆𑀬, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁯*:**𑁪𑁮**)

(654.)

2404\. **𑀉𑀧𑀯𑀲𑁆𑀲𑀁 𑀔𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀯𑀼𑀝𑁆𑀞𑀯𑀲𑁆𑀲𑀸𑀦𑀁𑁇

2405\. **𑀓𑀢𑁆𑀢𑀺𑀓𑀧𑀼𑀡𑁆𑀡𑀫𑀦𑁆**𑀢𑀺 𑀓𑀢𑁆𑀢𑀺𑀓𑀘𑀸𑀢𑀼𑀫𑀸𑀲𑀺𑀦𑀻 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2406\. **𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀆𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀻**𑀢𑀺 **𑀆𑀭𑀜𑁆𑀜𑀓𑀁** 𑀦𑀸𑀫 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀘𑀥𑀦𑀼𑀲𑀢𑀺𑀓𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

2407\. **𑀲𑀸𑀲𑀗𑁆𑀓𑀁** 𑀦𑀸𑀫 𑀆𑀭𑀸𑀫𑁂 𑀆𑀭𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑁂 𑀘𑁄𑀭𑀸𑀦𑀁 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀼𑀢𑁆𑀢𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀞𑀺𑀢𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀧𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇

2408\. **𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀁** 𑀦𑀸𑀫 𑀆𑀭𑀸𑀫𑁂 𑀆𑀭𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑁂 𑀘𑁄𑀭𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀳𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀆𑀓𑁄𑀝𑀺𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

2409\. **𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄**𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄𑁇

2410\. **𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄**𑀢𑀺 𑀇𑀘𑁆𑀙𑀫𑀸𑀦𑁄𑁇

2411\. **𑀢𑀺𑀡𑁆𑀡𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀦𑁆**𑀢𑀺 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁 𑀯𑀸𑁇

2412\. **𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀫𑀦𑁆𑀢𑀸 𑀕𑁄𑀘𑀭𑀕𑀸𑀫𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑁇

2413\. **𑀲𑀺𑀬𑀸 𑀘 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀘𑁆𑀘𑀬𑁄 𑀢𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬𑀸**𑀢𑀺 𑀲𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀺𑀬𑀸 𑀓𑀭𑀡𑀻𑀬𑀁𑁇

2414\. **𑀙𑀸𑀭𑀢𑁆𑀢𑀧𑀭𑀫𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀢𑁂𑀦 𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀦𑁆**𑀢𑀺 𑀙𑀸𑀭𑀢𑁆𑀢𑀧𑀭𑀫𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇

2415\. **𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸**𑀢𑀺 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁𑁇

2416\. **𑀢𑀢𑁄 𑀘𑁂 𑀉𑀢𑁆𑀢𑀭𑀺 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀢𑁆𑀢𑀫𑁂 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑀁, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁𑁇 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(655.)

2417\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀼𑀤𑁆𑀥𑀝𑁂 𑀧𑀘𑁆𑀘𑀼𑀤𑁆𑀥𑀝𑀲𑀜𑁆𑀜𑀻…𑀧𑁂…  𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀲𑀜𑁆𑀜𑀻…  𑀅𑀦𑀝𑁆𑀞𑁂 𑀦𑀝𑁆𑀞𑀲𑀜𑁆𑀜𑀻…  𑀅𑀯𑀺𑀦𑀝𑁆𑀞𑁂 𑀯𑀺𑀦𑀝𑁆𑀞𑀲𑀜𑁆𑀜𑀻…  𑀅𑀤𑀟𑁆𑀠𑁂 𑀤𑀟𑁆𑀠𑀲𑀜𑁆𑀜𑀻…  𑀅𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑁂 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2418\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(656.)

2419\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀢𑀺, 𑀊𑀦𑀓𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀢𑀺, 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦 𑀕𑀸𑀫𑀲𑀻𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀅𑀦𑁆𑀢𑁄𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀼𑀤𑁆𑀥𑀭𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀺, 𑀦𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺, 𑀟𑀬𑁆𑀳𑀢𑀺, 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀁 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2420\. 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀦𑀯𑀫𑀁𑁇
