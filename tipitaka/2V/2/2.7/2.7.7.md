# 𑁨𑁇𑁭𑁇𑁭 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀣

* 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓

---

3579\. 𑀇𑀫𑁂 𑀔𑁄 𑀧𑀦𑀸𑀬𑁆𑀬𑀸𑀬𑁄 𑀲𑀢𑁆𑀢 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀣𑀸

  
𑀥𑀫𑁆𑀫𑀸 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇



(1242.)

3580\. 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀲𑀫𑀣𑀸𑀬 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑀢𑀺𑁇

3581\. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄, 𑀅𑀬𑁆𑀬𑀸𑀬𑁄, 𑀲𑀢𑁆𑀢 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀣𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑁆𑀬𑀸𑀬𑁄 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑁆𑀬𑀸𑀬𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻𑀢𑀺𑁇

---

3582\. 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇



3583\. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀔𑁄, 𑀅𑀬𑁆𑀬𑀸𑀬𑁄, 𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀅𑀝𑁆𑀞 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑀢𑁆𑀢𑀭𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀢𑀺𑀁𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀙𑀲𑀝𑁆𑀞𑀺𑀲𑀢𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀅𑀝𑁆𑀞 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑁂𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑀢𑁆𑀢 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀣𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀏𑀢𑁆𑀢𑀓𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀸𑀕𑀢𑀁 𑀲𑀼𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑀸𑀳𑁂𑀯 𑀲𑀫𑀕𑁆𑀕𑀸𑀳𑀺 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸𑀳𑀺 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺𑁇

---

3584\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇

3585\. 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
