# 𑁨𑁇𑁧𑁇𑁫 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

2100\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀤𑁂𑀣 𑀫𑁂 𑀦𑀯𑀓𑀫𑁆𑀫𑀺𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀢𑀲𑁆𑀲𑁄 𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀦𑀦𑁆𑀤𑀸, 𑀦𑀦𑁆𑀤𑀯𑀢𑀻, 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀢𑀺𑁇 𑀢𑀸𑀲𑀼 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀭𑀼𑀡𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀅𑀪𑀺𑀭𑀽𑀧𑀸 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑀻 𑀤𑀓𑁆𑀔𑀸 𑀅𑀦𑀮𑀲𑀸, 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀍𑀳𑀲𑁆𑀲 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀼𑀦𑁄 𑀦𑀯𑀓𑀫𑁆𑀫𑀺𑀓𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀸𑀍𑀳𑀲𑁆𑀲 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀼𑀦𑁄 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  “𑀯𑀸𑀲𑀺𑀁 𑀤𑁂𑀣, 𑀧𑀭𑀲𑀼𑀁 𑀤𑁂𑀣, 𑀓𑀼𑀞𑀸𑀭𑀺𑀁 𑀤𑁂𑀣, 𑀓𑀼𑀤𑀸𑀮𑀁 𑀤𑁂𑀣, 𑀦𑀺𑀔𑀸𑀤𑀦𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀲𑀸𑀍𑀳𑁄𑀧𑀺 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑀁 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀓𑀢𑀸𑀓𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑀼𑀁𑁇 𑀢𑁂 𑀅𑀪𑀺𑀡𑁆𑀳𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

2101\. 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀤𑀽𑀲𑁂𑀢𑀼𑀁 𑀑𑀓𑀸𑀲𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀏𑀢𑀤𑁂𑀯𑀅𑀢𑁆𑀣𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑁄 “𑀏𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀬𑁆𑀬𑀸𑀬 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑀸”𑀢𑀺—  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺 “𑀏𑀢𑁆𑀢𑀓𑀸 𑀦𑀯𑀓𑀢𑀭𑀸”𑀢𑀺—  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺𑁇 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀑𑀓𑀸𑀲𑁂 𑀦𑀺𑀓𑀽𑀝𑁂 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺, 𑀬𑀣𑀸 𑀣𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁—  “𑀦𑀯𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸”𑀢𑀺; 𑀦𑀯𑀓𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁—  “𑀣𑁂𑀭𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑁂𑀢𑁆𑀯𑀸—  “𑀦 𑀩𑀳𑀼𑀓𑀢𑁄 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀅𑀓𑀸𑀲𑀺; 𑀫𑀁 𑀲𑁄 𑀤𑀽𑀲𑁂𑀢𑀼𑀓𑀸𑀫𑁄𑁇 𑀲𑀘𑀸𑀳𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀺𑀲𑁆𑀲𑀭𑁄 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙 𑀫𑁂 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀦𑀻𑀳𑀭𑁇 𑀬𑁄 𑀘𑁂 𑀫𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺, ‘𑀕𑀺𑀮𑀸𑀦𑀸’𑀢𑀺 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇

2102\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑁄—  “𑀓𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸? 𑀓𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀸𑀍𑀳𑀁 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀺𑀮𑀸𑀦𑀸𑀯𑀼𑀲𑁄; 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀦𑀻𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸—  “𑀬𑀫𑁆𑀧𑀸𑀳𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀅𑀓𑀸𑀲𑀺𑀁 𑀅𑀬𑁆𑀬𑀸𑀬 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀓𑀸𑀭𑀡𑀸”𑀢𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀆𑀡𑀸𑀧𑁂𑀢𑁆𑀯𑀸—  “𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀣𑀸”𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀩𑀳𑀸𑀭𑀸𑀫𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀸𑀍𑀳𑀁 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸𑀭𑀁 𑀧𑀢𑀺𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀻𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀸𑀍𑀳𑀁 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸𑀭𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀲𑀻𑀲𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀫𑀜𑁆𑀘𑀓𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀍𑀳𑁄 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀨𑀸𑀲𑀼, 𑀓𑀺𑀲𑁆𑀲 𑀦𑀺𑀧𑀦𑁆𑀦𑀸𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀜𑁆𑀳𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀬𑀸 𑀅𑀦𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀇𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇 “𑀓𑁆𑀬𑀸𑀳𑀁 𑀢𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀦 𑀇𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺? 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀑𑀓𑀸𑀲𑀁 𑀦 𑀮𑀪𑀸𑀫𑀺 𑀢𑀁 𑀤𑀽𑀲𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑀬 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇

2103\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀭𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑀸 𑀘𑀭𑀡𑀕𑀺𑀮𑀸𑀦𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀸𑀍𑀳𑀁 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸𑀭𑀁 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀁 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑀬 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀮𑀚𑁆𑀚𑀺𑀦𑀺𑀬𑁄 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀺𑀓𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑀸 𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀓𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑀸 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2104\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬𑁇 𑀅𑀣 𑀔𑁆𑀯𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀸𑀬 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀩𑁆𑀪𑀭𑀢𑀸𑀬 𑀤𑀼𑀧𑁆𑀧𑁄𑀲𑀢𑀸𑀬 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀬 𑀓𑁄𑀲𑀚𑁆𑀚𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀲𑀼𑀪𑀭𑀢𑀸𑀬 𑀲𑀼𑀧𑁄𑀲𑀢𑀸𑀬 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑁆𑀲 𑀥𑀼𑀢𑀲𑁆𑀲 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑁆𑀲 𑀅𑀧𑀘𑀬𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀤𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁 𑀢𑀤𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬, 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑀢𑀸𑀬, 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬, 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

2105\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀆𑀫𑀲𑀦𑀁 𑀯𑀸 𑀧𑀭𑀸𑀫𑀲𑀦𑀁 𑀯𑀸 𑀕𑀳𑀡𑀁 𑀯𑀸 𑀙𑀼𑀧𑀦𑀁 𑀯𑀸 𑀧𑀝𑀺𑀧𑀻𑀍𑀅𑀦𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑀸 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸”**𑀢𑀺𑁇 (*𑁫*:**𑁫**)

2106\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸 𑀬𑀣𑀸𑀬𑀼𑀢𑁆𑀢𑀸 𑀬𑀣𑀸𑀚𑀘𑁆𑀘𑀸 𑀬𑀣𑀸𑀦𑀸𑀫𑀸 𑀬𑀣𑀸𑀕𑁄𑀢𑁆𑀢𑀸 𑀬𑀣𑀸𑀲𑀻𑀮𑀸 𑀬𑀣𑀸𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑀻 𑀬𑀣𑀸𑀕𑁄𑀘𑀭𑀸 𑀣𑁂𑀭𑀸 𑀯𑀸 𑀦𑀯𑀸 𑀯𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀯𑀸, 𑀏𑀲𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀬𑀸 𑀧𑀦𑀸𑀢𑀺𑁇

2107\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀺𑀓𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀪𑀺𑀓𑁆𑀔𑀸𑀘𑀭𑀺𑀬𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀕𑀢𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀪𑀺𑀦𑁆𑀦𑀧𑀝𑀥𑀭𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀲𑀫𑀜𑁆𑀜𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀏𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀢𑀻𑀳𑀺 𑀲𑀭𑀡𑀕𑀫𑀦𑁂𑀳𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀪𑀤𑁆𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀲𑀸𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀲𑁂𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀅𑀲𑁂𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻; 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂𑀦 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑁇 𑀢𑀢𑁆𑀭 𑀬𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂𑀦 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸, 𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2108\. **𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸** 𑀦𑀸𑀫 𑀲𑀸𑀭𑀢𑁆𑀢𑀸 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀢𑀻 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸𑁇

2109\. **𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄** 𑀦𑀸𑀫 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑁄𑁇

2110\. **𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄** 𑀦𑀸𑀫 𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀬𑀓𑁆𑀔𑁄 𑀦 𑀧𑁂𑀢𑁄 𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀝𑀺𑀩𑀮𑁄 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇

2111\. **𑀅𑀥𑀓𑁆𑀔𑀓𑀦𑁆**𑀢𑀺 𑀳𑁂𑀝𑁆𑀞𑀓𑁆𑀔𑀓𑀁𑁇

2112\. **𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀦𑁆**𑀢𑀺 𑀉𑀧𑀭𑀺𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁𑁇

2113\. **𑀆𑀫𑀲𑀦𑀁** 𑀦𑀸𑀫 𑀆𑀫𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑀁𑁇

2114\. **𑀧𑀭𑀸𑀫𑀲𑀦𑀁** 𑀦𑀸𑀫 𑀇𑀢𑁄𑀘𑀺𑀢𑁄 𑀘 𑀲𑀜𑁆𑀘𑁄𑀧𑀦𑀁𑁇

2115\. **𑀕𑀳𑀡𑀁** 𑀦𑀸𑀫 𑀕𑀳𑀺𑀢𑀫𑀢𑁆𑀢𑀁𑁇

2116\. **𑀙𑀼𑀧𑀦𑀁** 𑀦𑀸𑀫 𑀨𑀼𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑀁𑁇

2117\. **𑀧𑀝𑀺𑀧𑀻𑀍𑀅𑀦𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀧𑁆𑀧𑀻𑀍𑀅𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺𑁇

2118\. **𑀅𑀬𑀫𑁆𑀧𑀻**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2119\. **𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀻𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑁂𑀦 𑀲𑀭𑀻𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀼𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀆𑀫𑀲𑀦𑀁 𑀯𑀸 𑀧𑀭𑀸𑀫𑀲𑀦𑀁 𑀯𑀸 𑀕𑀳𑀡𑀁 𑀯𑀸 𑀙𑀼𑀧𑀦𑀁 𑀯𑀸 𑀧𑀝𑀺𑀧𑀻𑀍𑀅𑀦𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀅𑀲𑁆𑀲𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀥𑀻𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

2120\. **𑀅𑀲𑀁𑀯𑀸𑀲𑀸**𑀢𑀺 **𑀲𑀁𑀯𑀸𑀲𑁄** 𑀦𑀸𑀫 𑀏𑀓𑀓𑀫𑁆𑀫𑀁 𑀏𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀢𑀸, 𑀏𑀲𑁄 𑀲𑀁𑀯𑀸𑀲𑁄 𑀦𑀸𑀫𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀢𑁆𑀣𑀺, 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑀸𑀢𑀺𑁇

(659.)

2121\. 𑀉𑀪𑀢𑁄𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁂 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2122\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2123\. 𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑀁 𑀅𑀥𑁄𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2124\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(660.)

2125\. 𑀏𑀓𑀢𑁄𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁂 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2126\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2127\. 𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑀁 𑀅𑀥𑁄𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2128\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(661.)

2129\. 𑀉𑀪𑀢𑁄𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁂 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑁂𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀲𑁆𑀲 𑀯𑀸 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2130\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2131\. 𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑀁 𑀅𑀥𑁄𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2132\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(662.)

2133\. 𑀏𑀓𑀢𑁄𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁂 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2134\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2135\. 𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑀁 𑀅𑀥𑁄𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2136\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀆𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(663.)

2137\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑀬𑀸, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀬, 𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---
