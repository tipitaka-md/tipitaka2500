# 𑁨𑁇𑁧𑁇𑁮 𑀅𑀝𑁆𑀞𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀅𑀝𑁆𑀞𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(674.)

2181. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀡𑁆𑀡𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺, 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀮𑁆𑀮𑀧𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀗𑁆𑀓𑁂𑀢𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲𑀧𑀺 𑀅𑀩𑁆𑀪𑀸𑀕𑀫𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺, 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀉𑀧𑀲𑀁𑀳𑀭𑀦𑁆𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬𑁇

2182. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀡𑁆𑀡𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀮𑁆𑀮𑀧𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀗𑁆𑀓𑁂𑀢𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲𑀧𑀺 𑀅𑀩𑁆𑀪𑀸𑀕𑀫𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀉𑀧𑀲𑀁𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬𑀸”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀡𑁆𑀡𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺, 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀮𑁆𑀮𑀧𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀗𑁆𑀓𑁂𑀢𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲𑀧𑀺 𑀅𑀩𑁆𑀪𑀸𑀕𑀫𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺, 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀉𑀧𑀲𑀁𑀳𑀭𑀦𑁆𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀡𑁆𑀡𑀕𑁆𑀕𑀳𑀡𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀮𑁆𑀮𑀧𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀗𑁆𑀓𑁂𑀢𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲𑀧𑀺 𑀅𑀩𑁆𑀪𑀸𑀕𑀫𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀓𑀸𑀬𑀫𑁆𑀧𑀺 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀉𑀧𑀲𑀁𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(675.)

2183. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑁆𑀕𑀳𑀡𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀡𑁆𑀡𑀕𑁆𑀕𑀳𑀡𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬, 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀲𑀗𑁆𑀓𑁂𑀢𑀁 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬**, **𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀯𑀸 𑀅𑀩𑁆𑀪𑀸𑀕𑀫𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬, 𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬, 𑀓𑀸𑀬𑀁 𑀯𑀸 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀉𑀧𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑀸 𑀅𑀝𑁆𑀞𑀯𑀢𑁆𑀣𑀼𑀓𑀸”**𑀢𑀺𑁇 (*𑁮*:**𑁮**)

(676.)

2184. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2185. **𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸** 𑀦𑀸𑀫 𑀲𑀸𑀭𑀢𑁆𑀢𑀸 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀢𑀻 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸𑁇

2186. **𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄** 𑀦𑀸𑀫 𑀲𑀸𑀭𑀢𑁆𑀢𑁄 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑁄𑁇

2187. **𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄** 𑀦𑀸𑀫 𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑁄, 𑀦 𑀬𑀓𑁆𑀔𑁄 𑀦 𑀧𑁂𑀢𑁄 𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀝𑀺𑀩𑀮𑁄 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇

2188. **𑀳𑀢𑁆𑀣𑀕𑁆𑀕𑀳𑀡𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀳𑀢𑁆𑀣𑁄 𑀦𑀸𑀫 𑀓𑀧𑁆𑀧𑀭𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀬𑀸𑀯 𑀅𑀕𑁆𑀕𑀦𑀔𑀸𑁇 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑀁 𑀅𑀥𑁄𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2189. **𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀡𑁆𑀡𑀕𑁆𑀕𑀳𑀡𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀦𑀺𑀯𑀢𑁆𑀣𑀁 𑀯𑀸 𑀧𑀸𑀭𑀼𑀢𑀁 𑀯𑀸 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2190. **𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2191. **𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀞𑀺𑀢𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2192. **𑀲𑀗𑁆𑀓𑁂𑀢𑀁 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀼𑀭𑀺𑀲𑁂𑀦—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀑𑀓𑀸𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀸”𑀢𑀺—  𑀯𑀼𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀧𑀤𑁂 𑀧𑀤𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2193. **𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀯𑀸 𑀅𑀩𑁆𑀪𑀸𑀕𑀫𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀅𑀩𑁆𑀪𑀸𑀕𑀫𑀦𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2194. **𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀑𑀓𑀸𑀲𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2195. **𑀓𑀸𑀬𑀁 𑀯𑀸 𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀉𑀧𑀲𑀁𑀳𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀞𑀺𑀢𑀸 𑀓𑀸𑀬𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

2196. **𑀅𑀬𑀫𑁆𑀧𑀻**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2197. **𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀢𑀸𑀮𑁄 𑀫𑀢𑁆𑀣𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀧𑀼𑀦 𑀯𑀺𑀭𑀼𑀍𑀳𑀺𑀬𑀸 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻 𑀅𑀲𑁆𑀲𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀥𑀻𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

2198. **𑀅𑀲𑀁𑀯𑀸𑀲𑀸**𑀢𑀺 **𑀲𑀁𑀯𑀸𑀲𑁄** 𑀦𑀸𑀫 𑀏𑀓𑀓𑀫𑁆𑀫𑀁 𑀏𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀢𑀸𑁇 𑀏𑀲𑁄 𑀲𑀁𑀯𑀸𑀲𑁄 𑀦𑀸𑀫𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑀸𑀢𑀺𑁇

(677.)

2199. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑀬𑀸, 𑀅𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀬, 𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2200. 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇



2201. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄, 𑀅𑀬𑁆𑀬𑀸𑀬𑁄, 𑀅𑀝𑁆𑀞 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀬𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀮𑀪𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀸𑀲𑀁, 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂 𑀢𑀣𑀸 𑀧𑀘𑁆𑀙𑀸, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑀸𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑁆𑀬𑀸𑀬𑁄 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑁆𑀬𑀸𑀬𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻𑀢𑀺𑁇

---

2202. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
