# 𑁨𑁇𑁧𑁇𑁬 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(664.)

2139\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀸𑀍𑀳𑁂𑀦 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀼𑀦𑀸 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀳𑁄𑀢𑀺𑁇 𑀬𑀸𑀯 𑀕𑀩𑁆𑀪𑁄 𑀢𑀭𑀼𑀡𑁄 𑀅𑀳𑁄𑀲𑀺 𑀢𑀸𑀯 𑀙𑀸𑀤𑁂𑀲𑀺𑁇 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂 𑀕𑀩𑁆𑀪𑁂 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀚𑀸𑀬𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀘𑀺𑀭𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀸 𑀯𑀺𑀚𑀸𑀢𑀸𑁇 𑀓𑀘𑁆𑀘𑀺 𑀦𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀬𑁂𑀯 𑀲𑀫𑀸𑀦𑀸 𑀕𑀩𑁆𑀪𑀺𑀦𑀻”𑀢𑀺? “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑀺 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑀻”𑀢𑀺? “𑀬𑁄 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀯𑀡𑁆𑀡𑁄 𑀫𑀬𑁆𑀳𑁂𑀲𑁄 𑀅𑀯𑀡𑁆𑀡𑁄, 𑀬𑀸 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀓𑀺𑀢𑁆𑀢𑀺 𑀫𑀬𑁆𑀳𑁂𑀲𑀸 𑀅𑀓𑀺𑀢𑁆𑀢𑀺, 𑀬𑁄 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀬𑀲𑁄 𑀫𑀬𑁆𑀳𑁂𑀲𑁄 𑀅𑀬𑀲𑁄, 𑀬𑁄 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀮𑀸𑀪𑁄 𑀫𑀬𑁆𑀳𑁂𑀲𑁄 𑀅𑀮𑀸𑀪𑁄𑁇 𑀓𑁆𑀬𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀯𑀡𑁆𑀡𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀬𑀲𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀮𑀸𑀪𑀁 𑀧𑀭𑁂𑀲𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺? 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2140\. 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀢𑀺 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(665.)

2141\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀬𑁆𑀬 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬, 𑀬𑀤𑀸 𑀘 𑀲𑀸 𑀞𑀺𑀢𑀸 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀘𑀼𑀢𑀸 𑀯𑀸 𑀦𑀸𑀲𑀺𑀢𑀸 𑀯𑀸 𑀅𑀯𑀲𑁆𑀲𑀝𑀸 𑀯𑀸, 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀧𑀼𑀩𑁆𑀩𑁂𑀯𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀁 𑀏𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀯𑀭𑀽𑀧𑀸 𑀘 𑀏𑀯𑀭𑀽𑀧𑀸 𑀘 𑀲𑀸 𑀪𑀕𑀺𑀦𑀻𑀢𑀺**, **𑀦𑁄 𑀘 𑀔𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀁 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸**”𑀢𑀺𑁇 (*𑁬*:**𑁬**)

(666.)

2142\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2143\. **𑀚𑀸𑀦𑀸𑀢𑀺** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂 𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺, 𑀲𑀸 𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺𑁇

2144\. **𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀦𑁆**𑀢𑀺 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁𑁇

2145\. **𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀲𑀬𑀁 𑀘𑁄𑀤𑁂𑀬𑁆𑀬𑁇

2146\. **𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬𑁇

2147\. **𑀬𑀤𑀸 𑀘 𑀲𑀸 𑀞𑀺𑀢𑀸 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀘𑀼𑀢𑀸 𑀯𑀸**𑀢𑀺 **𑀞𑀺𑀢𑀸** 𑀦𑀸𑀫 𑀲𑀮𑀺𑀗𑁆𑀕𑁂 𑀞𑀺𑀢𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀘𑀼𑀢𑀸** 𑀦𑀸𑀫 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀦𑀸𑀲𑀺𑀢𑀸** 𑀦𑀸𑀫 𑀲𑀬𑀁 𑀯𑀸 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀯𑀸 𑀦𑀸𑀲𑀺𑀢𑀸𑁇 **𑀅𑀯𑀲𑁆𑀲𑀝𑀸** 𑀦𑀸𑀫 𑀢𑀺𑀢𑁆𑀣𑀸𑀬𑀢𑀦𑀁 𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  “𑀧𑀼𑀩𑁆𑀩𑁂𑀯𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀁 𑀏𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀯𑀭𑀽𑀧𑀸 𑀘 𑀏𑀯𑀭𑀽𑀧𑀸 𑀘 𑀲𑀸 𑀪𑀕𑀺𑀦𑀻”𑀢𑀺𑁇

2148\. **𑀦𑁄 𑀘 𑀔𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀦𑁆**𑀢𑀺 𑀲𑀬𑀁 𑀯𑀸 𑀦 𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀁𑁇

2149\. **𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀦𑁆**𑀢𑀺 𑀦 𑀅𑀜𑁆𑀜𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀁𑁇

2150\. **𑀅𑀬𑀫𑁆𑀧𑀻**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2151\. **𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀡𑁆𑀟𑀼𑀧𑀮𑀸𑀲𑁄 𑀩𑀦𑁆𑀥𑀦𑀸 𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀳𑀭𑀺𑀢𑀢𑁆𑀣𑀸𑀬, 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀥𑀼𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀅𑀲𑁆𑀲𑀫𑀡𑀻 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀥𑀻𑀢𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

2152\. **𑀅𑀲𑀁𑀯𑀸𑀲𑀸**𑀢𑀺 **𑀲𑀁𑀯𑀸𑀲𑁄** 𑀦𑀸𑀫 𑀏𑀓𑀓𑀫𑁆𑀫𑀁 𑀏𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀢𑀸𑁇 𑀏𑀲𑁄 𑀲𑀁𑀯𑀸𑀲𑁄 𑀦𑀸𑀫𑁇 𑀲𑁄 𑀢𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀲𑀁𑀯𑀸𑀲𑀸𑀢𑀺𑁇

(667.)

2153\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀯𑀸 𑀓𑀮𑀳𑁄 𑀯𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀸 𑀯𑀺𑀯𑀸𑀤𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀦𑀸𑀭𑁄𑀘𑁂𑀢𑀺, “𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀯𑀸 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀦𑀸𑀭𑁄𑀘𑁂𑀢𑀺, “𑀅𑀬𑀁 𑀓𑀓𑁆𑀔𑀍𑀆 𑀨𑀭𑀼𑀲𑀸 𑀚𑀻𑀯𑀺𑀢𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀦𑀸𑀭𑁄𑀘𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻 𑀦𑀸𑀭𑁄𑀘𑁂𑀢𑀺, 𑀦𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀦𑀸𑀭𑁄𑀘𑁂𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀓𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦𑀸𑀢𑀺 𑀦𑀸𑀭𑁄𑀘𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬…𑀧𑁂…  𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
