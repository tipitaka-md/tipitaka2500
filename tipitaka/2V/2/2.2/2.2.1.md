# 𑁨𑁇𑁨𑁇𑁧 𑀉𑀲𑁆𑀲𑀬𑀯𑀸𑀤𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

2204\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀤𑁄𑀲𑀺𑀢𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀧𑀼𑀢𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀏𑀓𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄, 𑀏𑀓𑁄 𑀲𑀤𑁆𑀥𑁄 𑀧𑀲𑀦𑁆𑀦𑁄𑁇 𑀢𑁂 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀯𑀺𑀪𑀚𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀢𑀁 𑀲𑀤𑁆𑀥𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀤𑁄𑀲𑀺𑀢𑁄, 𑀢𑀁 𑀪𑀸𑀚𑁂𑀫𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑁄 𑀲𑀤𑁆𑀥𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 𑀢𑀁 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸𑀬𑁆𑀬𑁄, 𑀏𑀯𑀁 𑀅𑀯𑀘𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀺𑀢𑀼𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀢𑀁 𑀲𑀤𑁆𑀥𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀤𑁄𑀲𑀺𑀢𑁄, 𑀢𑀁 𑀪𑀸𑀚𑁂𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑀤𑁆𑀥𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 𑀢𑀁 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸𑀬𑁆𑀬𑁄, 𑀏𑀯𑀁 𑀅𑀯𑀘𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀺𑀢𑀼𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀢𑀁 𑀲𑀤𑁆𑀥𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀤𑁄𑀲𑀺𑀢𑁄, 𑀢𑀁 𑀪𑀸𑀚𑁂𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀲𑀤𑁆𑀥𑁄 𑀧𑀲𑀦𑁆𑀦𑁄—  “𑀲𑀘𑁂 𑀫𑀬𑁆𑀳𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺—  𑀢𑀁 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀸𑀚𑁂𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀤𑁄𑀲𑀺𑀢𑁄 𑀢𑁂𑀳𑀺 𑀪𑀸𑀚𑀻𑀬𑀫𑀸𑀦𑁄 𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀺𑀓𑁆𑀔𑀫𑀣𑀸𑀬𑁆𑀬𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀤𑁄𑀲𑀺𑀢𑁄”𑀢𑀺𑁇

2205\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸𑀬𑁆𑀬𑁄, 𑀏𑀯𑀁 𑀅𑀯𑀘, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀺𑀢𑀼𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 “𑀤𑀺𑀦𑁆𑀦𑁄 𑀦 𑀤𑀺𑀦𑁆𑀦𑁄”𑀢𑀺 𑀯𑁄𑀳𑀸𑀭𑀺𑀓𑁂 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀧𑀺 𑀦𑀸𑀬𑁆𑀬𑀸 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸 𑀲𑀓𑁆𑀔𑀺𑀁 𑀞𑀧𑀬𑀺𑀢𑁆𑀯𑀸 𑀤𑀸𑀦𑀁 𑀤𑀺𑀬𑁆𑀬𑀫𑀸𑀦𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸—  “𑀲𑀘𑁆𑀘𑀁 𑀔𑁄 𑀅𑀬𑁆𑀬𑀸 𑀆𑀳𑀸”𑀢𑀺 𑀢𑀁 𑀉𑀤𑁄𑀲𑀺𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀓𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀭𑀸𑀚𑀺𑀢𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀺𑀬𑁄 𑀇𑀫𑀸 𑀫𑀼𑀡𑁆𑀟𑀸 𑀩𑀦𑁆𑀥𑀓𑀺𑀦𑀺𑀬𑁄𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀤𑁄𑀲𑀺𑀢𑀁 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀤𑀡𑁆𑀟𑀸𑀧𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀡𑁆𑀟𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑀲𑁆𑀲 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀆𑀚𑀻𑀯𑀓𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀚𑀻𑀯𑀓𑁂 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀲𑀺—  “𑀏𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀣𑀸”𑀢𑀺𑁇

2206\. 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀩𑀦𑁆𑀥𑀸𑀧𑁂𑀲𑀼𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀧𑀞𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀤𑁄𑀲𑀺𑀢𑀁 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀼𑀁, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀤𑀡𑁆𑀟𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀢𑀺𑀬𑀁 𑀩𑀦𑁆𑀥𑀸𑀧𑁂𑀲𑀼𑀁𑁇 𑀇𑀤𑀸𑀦𑀺 𑀖𑀸𑀢𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2207\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀉𑀲𑁆𑀲𑀬𑀯𑀸𑀤𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀲𑁆𑀲𑀬𑀯𑀸𑀤𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀲𑁆𑀲𑀬𑀯𑀸𑀤𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(679.)

2208\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀲𑁆𑀲𑀬𑀯𑀸𑀤𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬 𑀕𑀳𑀧𑀢𑀺𑀦𑀸 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀤𑀸𑀲𑁂𑀦 𑀯𑀸 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑁂𑀦 𑀯𑀸 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀲𑀫𑀡𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀦𑀸𑀧𑀺, 𑀅𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”**𑀢𑀺𑁇 (*𑁧*:**𑁯**)

(680.)

2209\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2210\. **𑀉𑀲𑁆𑀲𑀬𑀯𑀸𑀤𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀅𑀟𑁆𑀟𑀓𑀸𑀭𑀺𑀓𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2211\. **𑀕𑀳𑀧𑀢𑀺** 𑀦𑀸𑀫 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇

2212\. **𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑁄** 𑀦𑀸𑀫 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀧𑀼𑀢𑁆𑀢𑀪𑀸𑀢𑀭𑁄𑁇

2213\. **𑀤𑀸𑀲𑁄** 𑀦𑀸𑀫 𑀅𑀦𑁆𑀢𑁄𑀚𑀸𑀢𑁄 𑀥𑀦𑀓𑁆𑀓𑀻𑀢𑁄 𑀓𑀭𑀫𑀭𑀸𑀦𑀻𑀢𑁄𑁇

2214\. **𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀪𑀝𑀓𑁄 𑀆𑀳𑀢𑀓𑁄𑁇

2215\. **𑀲𑀫𑀡𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀜𑁆𑀘 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀜𑁆𑀘 𑀲𑀸𑀫𑀡𑁂𑀭𑀜𑁆𑀘 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄𑁇

2216\. 𑀅𑀟𑁆𑀟𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀤𑀼𑀢𑀺𑀬𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀏𑀓𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀅𑀟𑁆𑀟𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2217\. **𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀦𑁆**𑀢𑀺 𑀲𑀳 𑀯𑀢𑁆𑀣𑀼𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬𑁇

2218\. **𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀦𑁆**𑀢𑀺 𑀲𑀁𑀖𑀫𑁆𑀳𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀻𑀬𑀢𑀺𑁇

2219\. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆**𑀢𑀺 𑀲𑀁𑀖𑁄𑀯 𑀢𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺 𑀅𑀩𑁆𑀪𑁂𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀦 𑀏𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀓𑀸𑀬𑀲𑁆𑀲 𑀦𑀸𑀫𑀓𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁𑁇 𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺𑁇

(681.)

2220\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀆𑀓𑀟𑁆𑀠𑀻𑀬𑀫𑀸𑀦𑀸 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀭𑀓𑁆𑀔𑀁 𑀬𑀸𑀘𑀢𑀺, 𑀅𑀦𑁄𑀤𑀺𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬…𑀧𑁂…  𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
