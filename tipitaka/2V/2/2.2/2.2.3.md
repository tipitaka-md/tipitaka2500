# 𑁨𑁇𑁨𑁇𑁩 𑀏𑀓𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀕𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀏𑀓𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀕𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(687.)

2245. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀸𑀬 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀡𑁆𑀟𑀺𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀓𑀁 𑀜𑀸𑀢𑀺𑀓𑀼𑀮𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀪𑀤𑁆𑀤𑀸 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀻 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀓𑀳𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸, 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀡𑁆𑀟𑀺𑀢𑁆𑀯𑀸 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀅𑀫𑁆𑀫𑀸, 𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀕𑀸𑀫𑀓𑁂 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀜𑀸𑀢𑀺𑀓𑀼𑀮𑀁𑁇 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀺𑀘𑀺𑀦𑀣𑀸”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀓𑀺𑀓𑀸 𑀆𑀕𑀢𑀸, 𑀓𑀘𑁆𑀘𑀺𑀲𑀺 𑀅𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀢𑀸”𑀢𑀺? “𑀅𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀢𑀸𑀫𑁆𑀳𑀺, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺𑁇

2246. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀁 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀁 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

2247. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”**𑀢𑀺𑁇

2248. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(688.)

2249. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀲𑀸𑀓𑁂𑀢𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀦𑀤𑀻 𑀢𑀭𑀺𑀢𑀩𑁆𑀩𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦𑀸𑀯𑀺𑀓𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀸𑀥𑀼 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀸𑀭𑁂𑀣𑀸”𑀢𑀺𑁇 “𑀦𑀸𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑀸 𑀉𑀪𑁄 𑀲𑀓𑀺𑀁 𑀢𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑁄 𑀏𑀓𑀁 𑀉𑀢𑁆𑀢𑀸𑀭𑁂𑀲𑀺𑁇 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑁄 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀁 𑀤𑀽𑀲𑁂𑀲𑀺𑁇 𑀅𑀦𑀼𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀁 𑀤𑀽𑀲𑁂𑀲𑀺𑁇 𑀢𑀸 𑀧𑀘𑁆𑀙𑀸 𑀲𑀫𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼—  “𑀓𑀘𑁆𑀘𑀺𑀲𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀢𑀸”𑀢𑀺? “𑀧𑀥𑀁𑀲𑀺𑀢𑀸𑀫𑁆𑀳𑀺, 𑀅𑀬𑁆𑀬𑁂𑁇 𑀢𑁆𑀯𑀁 𑀧𑀦, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀢𑀸”𑀢𑀺? “𑀧𑀥𑀁𑀲𑀺𑀢𑀸𑀫𑁆𑀳𑀺, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺𑁇

2250. 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀦𑀤𑀻𑀧𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀦𑀤𑀻𑀧𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀦𑀤𑀻𑀧𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

2251. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀯𑀸 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀏𑀓𑀸 𑀯𑀸 𑀦𑀤𑀻𑀧𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”**𑀢𑀺𑁇

2252. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(689.)

2258. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀕𑀡𑀫𑁆𑀳𑀸 𑀑𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀕𑀡𑀫𑁆𑀳𑀸 𑀑𑀳𑀻𑀬𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀕𑀡𑀫𑁆𑀳𑀸 𑀑𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(691.)

2259. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀸 𑀯𑀸 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀏𑀓𑀸 𑀯𑀸 𑀦𑀤𑀻𑀧𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀏𑀓𑀸 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀬𑁆𑀬, 𑀏𑀓𑀸 𑀯𑀸 𑀕𑀡𑀫𑁆𑀳𑀸 𑀑𑀳𑀻𑀬𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”**𑀢𑀺𑁇 (*𑁩*:**𑁧𑁧**)

(692.)

2260. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2261. **𑀏𑀓𑀸 𑀯𑀸 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀕𑀸𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀧𑀁 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2262. 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀕𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀘𑀸𑀭𑀁 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2263. **𑀏𑀓𑀸 𑀯𑀸 𑀦𑀤𑀻𑀧𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦𑀤𑀻 𑀦𑀸𑀫 𑀢𑀺𑀫𑀡𑁆𑀟𑀮𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀉𑀢𑁆𑀢𑀭𑀦𑁆𑀢𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁄 𑀢𑁂𑀫𑀺𑀬𑀢𑀺𑁇 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀉𑀢𑁆𑀢𑀭𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀉𑀢𑁆𑀢𑀭𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2264. **𑀏𑀓𑀸 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀳 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀯𑀺𑀚𑀳𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2265. **𑀏𑀓𑀸 𑀯𑀸 𑀕𑀡𑀫𑁆𑀳𑀸 𑀑𑀳𑀻𑀬𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀕𑀸𑀫𑀓𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀦𑀽𑀧𑀘𑀸𑀭𑀁 𑀯𑀸 𑀲𑀯𑀦𑀽𑀧𑀘𑀸𑀭𑀁 𑀯𑀸 𑀯𑀺𑀚𑀳𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀯𑀺𑀚𑀳𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2266. **𑀅𑀬𑀫𑁆𑀧𑀻**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2267. **𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀦𑁆**𑀢𑀺 𑀲𑀳 𑀯𑀢𑁆𑀣𑀼𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬𑁇

2268. **𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀦𑁆**𑀢𑀺 𑀲𑀁𑀖𑀫𑁆𑀳𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀻𑀬𑀢𑀺𑁇

2269. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄**𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺𑁇

(693.)

2270. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀸 𑀯𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀯𑀸 𑀧𑀓𑁆𑀔𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀸 𑀯𑀸, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2271. 𑀢𑀢𑀺𑀬𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
