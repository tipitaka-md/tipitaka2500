# 𑁨𑁇𑁨𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀪𑁄𑀚𑀦𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀢𑀺𑀬𑀪𑁄𑀚𑀦𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(704.)

2310\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀪𑀺𑀭𑀽𑀧𑀸 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀸𑀦𑀺 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀻 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀅𑀦𑀦𑁆𑀢𑀭𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀲𑀻”𑀢𑀺? “𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺𑁇 “𑀢𑁆𑀯𑀁 𑀧𑀦, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸, 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑁇 𑀇𑀗𑁆𑀖, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑀁 𑀢𑁂 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁂𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤 𑀯𑀸, 𑀪𑀼𑀜𑁆𑀚 𑀯𑀸”𑀢𑀺𑁇

2311\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀺—  ‘𑀓𑀺𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸, 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑁇 𑀇𑀗𑁆𑀖, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑀁 𑀢𑁂 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁂𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚 𑀯𑀸’”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀓𑀺𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑁇 𑀇𑀗𑁆𑀖, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑀁 𑀢𑁂 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁂𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚 𑀯𑀸’”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀺—  ‘𑀓𑀺𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑀸; 𑀇𑀗𑁆𑀖, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑀁 𑀢𑁂 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁂𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚 𑀯𑀸’𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(705.)

2312\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀓𑀺𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸, 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑁇 𑀇𑀗𑁆𑀖, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑀁 𑀢𑁂 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁂𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚 𑀯𑀸’𑀢𑀺, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”**𑀢𑀺𑁇 (*𑁬*:**𑁧𑁪**)

(706.)

2313\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2314\. **𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺—  “𑀓𑀺𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀯𑀸, 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑁇 𑀇𑀗𑁆𑀖, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑀁 𑀢𑁂 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁂𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚 𑀯𑀸”𑀢𑀺 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲𑀸 𑀯𑀘𑀦𑁂𑀦 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀪𑁄𑀚𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2315\. **𑀅𑀬𑀫𑁆𑀧𑀻**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2316\. **𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀦𑁆**𑀢𑀺 𑀲𑀳 𑀯𑀢𑁆𑀣𑀼𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬𑁇

2317\. **𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀦𑁆**𑀢𑀺 𑀲𑀁𑀖𑀫𑁆𑀳𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀻𑀬𑀢𑀺𑁇

2318\. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄**𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺𑁇

2319\. 𑀉𑀤𑀓𑀤𑀦𑁆𑀢𑀧𑁄𑀦𑀁 “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸”𑀢𑀺 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲𑀸 𑀯𑀘𑀦𑁂𑀦 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(707.)

2320\. 𑀏𑀓𑀢𑁄𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁂 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑁂𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀲𑁆𑀲 𑀯𑀸 𑀳𑀢𑁆𑀣𑀢𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 “𑀔𑀸𑀤 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚 𑀯𑀸”𑀢𑀺 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲𑀸 𑀯𑀘𑀦𑁂𑀦 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀪𑁄𑀚𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀉𑀤𑀓𑀤𑀦𑁆𑀢𑀧𑁄𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲𑀸 𑀯𑀘𑀦𑁂𑀦 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(708.)

2321\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  “𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄”𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑀻 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, “𑀓𑀼𑀧𑀺𑀢𑀸 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, “𑀓𑀼𑀮𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀢𑀸𑀬 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---
