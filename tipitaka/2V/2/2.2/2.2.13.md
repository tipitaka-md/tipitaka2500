# 𑁨𑁇𑁨𑁇𑁧𑁩 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀧𑀲𑀫𑀸𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(727.)

2394\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀸𑀘𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀤𑁆𑀤𑀸 𑀏𑀯𑀁𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀢𑀸 𑀲𑀁𑀖𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀆𑀳𑁇 𑀢𑀼𑀫𑁆𑀳𑀜𑁆𑀜𑁂𑀯 𑀲𑀁𑀖𑁄 𑀉𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀪𑀯𑁂𑀦 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀪𑀲𑁆𑀲𑀺𑀬𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸 𑀏𑀯𑀫𑀸𑀳—  ‘𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀔𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀸𑀘𑀸𑀭𑀸 𑀧𑀸𑀧𑀲𑀤𑁆𑀤𑀸 𑀧𑀸𑀧𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻’”𑀢𑀺𑁇

2395\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀺—  𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻𑀢𑀺”…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀸𑀘𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀤𑁆𑀤𑀸 𑀏𑀯𑀁𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀢𑀸 𑀲𑀁𑀖𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀆𑀳𑁇 𑀢𑀼𑀫𑁆𑀳𑀜𑁆𑀜𑁂𑀯 𑀲𑀁𑀖𑁄 𑀉𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀪𑀯𑁂𑀦 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀪𑀲𑁆𑀲𑀺𑀬𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸 𑀏𑀯𑀫𑀸𑀳—  𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀔𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀸𑀘𑀸𑀭𑀸 𑀧𑀸𑀧𑀲𑀤𑁆𑀤𑀸 𑀧𑀸𑀧𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀺—  𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

2396\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀸𑀘𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀤𑁆𑀤𑀸 𑀏𑀯𑀁𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀢𑀸 𑀲𑀁𑀖𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀆𑀳𑁇 𑀢𑀼𑀫𑁆𑀳𑀜𑁆𑀜𑁂𑀯 𑀲𑀁𑀖𑁄 𑀉𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀪𑀯𑁂𑀦 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀪𑀲𑁆𑀲𑀺𑀬𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸 𑀏𑀯𑀫𑀸𑀳—  𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀔𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀸𑀘𑀸𑀭𑀸 𑀧𑀸𑀧𑀲𑀤𑁆𑀤𑀸 𑀧𑀸𑀧𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻’𑀢𑀺𑁇 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀫𑀸, 𑀅𑀬𑁆𑀬𑁂, 𑀏𑀯𑀁 𑀅𑀯𑀘—  𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀸𑀘𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀤𑁆𑀤𑀸 𑀏𑀯𑀁𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀢𑀸 𑀲𑀁𑀖𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀆𑀳𑁇 𑀢𑀼𑀫𑁆𑀳𑀜𑁆𑀜𑁂𑀯 𑀲𑀁𑀖𑁄 𑀉𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀪𑀯𑁂𑀦 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀪𑀲𑁆𑀲𑀺𑀬𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸 𑀏𑀯𑀫𑀸𑀳—  𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀔𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀸𑀘𑀸𑀭𑀸 𑀧𑀸𑀧𑀲𑀤𑁆𑀤𑀸 𑀧𑀸𑀧𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 𑀢𑀣𑁂𑀯 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬, 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀜𑁆𑀘𑁂 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀻𑀬𑀫𑀸𑀦𑀸 𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁩*:**𑁨𑁧**)

2405\. **𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺 𑀬𑀸 𑀲𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑁇

2406\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀻**𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺𑁇

2407\. 𑀬𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀬𑀸 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀢𑀸𑀳𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀫𑀸𑀬𑁆𑀬𑁂, 𑀏𑀯𑀁 𑀅𑀯𑀘—  ‘𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻’”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀯𑀤𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑀫𑁆𑀧𑀺 𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀫𑀸𑀬𑁆𑀬𑁂, 𑀏𑀯𑀁 𑀅𑀯𑀘—  ‘𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻’”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀩𑀮𑀸𑀬 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(730.)

2408\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀸𑀘𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀤𑁆𑀤𑀸 𑀏𑀯𑀁𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀢𑀸 𑀲𑀁𑀖𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀆𑀳𑁇 𑀢𑀼𑀫𑁆𑀳𑀜𑁆𑀜𑁂𑀯 𑀲𑀁𑀖𑁄 𑀉𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀪𑀯𑁂𑀦 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀪𑀲𑁆𑀲𑀺𑀬𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸 𑀏𑀯𑀫𑀸𑀳—  𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀔𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀸𑀘𑀸𑀭𑀸 𑀧𑀸𑀧𑀲𑀤𑁆𑀤𑀸 𑀧𑀸𑀧𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑁂𑀬𑁆𑀬 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

2409\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑀯, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀳𑀭𑀣𑁇 𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀣𑁇 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑁂 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀸𑀘𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀤𑁆𑀤𑀸 𑀏𑀯𑀁𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀢𑀸 𑀲𑀁𑀖𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀆𑀳𑁇 𑀢𑀼𑀫𑁆𑀳𑀜𑁆𑀜𑁂𑀯 𑀲𑀁𑀖𑁄 𑀉𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀪𑀯𑁂𑀦 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀪𑀲𑁆𑀲𑀺𑀬𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀸 𑀏𑀯𑀫𑀸𑀳—  𑀪𑀕𑀺𑀦𑀺𑀬𑁄 𑀔𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀸𑀘𑀸𑀭𑀸 𑀧𑀸𑀧𑀲𑀤𑁆𑀤𑀸 𑀧𑀸𑀧𑀲𑀺𑀮𑁄𑀓𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑁂𑀲𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸𑁇 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑀣𑀸𑀬𑁆𑀬𑁂𑁇 𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀪𑀕𑀺𑀦𑀻𑀦𑀁 𑀲𑀁𑀖𑁄 𑀯𑀡𑁆𑀡𑁂𑀢𑀻”𑀢𑀺𑁇 𑀲𑀸 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀬𑀲𑁆𑀲𑀸 𑀅𑀬𑁆𑀬𑀸𑀬 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬, 𑀲𑀸 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲𑀸 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑀸 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

2410\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂… 𑁇

2411\. 𑀲𑀫𑀦𑀼𑀪𑀝𑁆𑀞𑀸 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀬𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

2412\. 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸, 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀬𑀸 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑁆𑀢𑀺𑁇

2413\. **𑀅𑀬𑀫𑁆𑀧𑀻**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2414\. **𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀦𑁆**𑀢𑀺 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀲𑀳𑀯𑀢𑁆𑀣𑀼𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑀸𑁇

2415\. **𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀦𑁆**𑀢𑀺 𑀲𑀁𑀖𑀫𑁆𑀳𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀻𑀬𑀢𑀺𑁇

2416\. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄**𑀢𑀺 𑀲𑀁𑀖𑁄𑀯 𑀢𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺, 𑀅𑀩𑁆𑀪𑁂𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀦 𑀏𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀓𑀸𑀬𑀲𑁆𑀲 𑀦𑀸𑀫𑀓𑀫𑁆𑀫𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁, 𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 “𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺𑁇

(731.)

2417\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2418\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(732.)

2419\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑀸, 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2420\. 𑀤𑀲𑀫𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
