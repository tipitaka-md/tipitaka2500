# 𑁨𑁇𑁨𑁇𑁪 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀑𑀲𑀸𑀭𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

### 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀑𑀲𑀸𑀭𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(694.)

2272\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀺𑀓𑀸 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀺𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀺𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀺𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀺𑀓𑀸𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀲𑁆𑀲𑀸 𑀓𑀫𑁆𑀫𑁂 𑀓𑀭𑀻𑀬𑀫𑀸𑀦𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀸𑀫𑀓𑀁 𑀅𑀕𑀫𑀸𑀲𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄—  “𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸”𑀢𑀺, 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀸𑀫𑀓𑁂 𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀻𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀘𑁆𑀙𑀺𑁇 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺 𑀦 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀞𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺 𑀦 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀦 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀫𑀬𑀺 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺 𑀦 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀞𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺 𑀦 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀦 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀆𑀧𑀼𑀘𑁆𑀙𑀻”𑀢𑀺? “𑀏𑀯𑀜𑁆𑀳𑁂𑀢𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀦𑀸𑀣𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀦𑀸𑀣𑀸”𑀢𑀺? “𑀇𑀫𑀸 𑀫𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄—  ‘𑀅𑀬𑀁 𑀅𑀦𑀸𑀣𑀸 𑀅𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀢𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀓𑀸𑀘𑀺 𑀧𑀝𑀺𑀯𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁𑀲𑀽”𑀢𑀺𑁇

2273\. 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀩𑀸𑀮𑀸 𑀏𑀢𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀏𑀢𑀸, 𑀦𑁂𑀯 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀫𑁆𑀫𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀯𑀸𑁇 𑀫𑀬𑀁 𑀔𑁄 𑀚𑀸𑀦𑀸𑀫 𑀓𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀫𑁆𑀧𑀺𑁇 𑀫𑀬𑀁 𑀔𑁄 𑀅𑀓𑀢𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫 𑀓𑀢𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑁄𑀧𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺, 𑀮𑀳𑀼𑀁 𑀮𑀳𑀼𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀑𑀲𑀸𑀭𑁂𑀲𑀺𑁇

2274\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑀁 𑀅𑀦𑀜𑁆𑀜𑀸𑀬 𑀕𑀡𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀑𑀲𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑀁 𑀅𑀦𑀜𑁆𑀜𑀸𑀬 𑀕𑀡𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀑𑀲𑀸𑀭𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑀁 𑀅𑀦𑀜𑁆𑀜𑀸𑀬 𑀕𑀡𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀑𑀲𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(695.)

2275\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑀁 𑀅𑀦𑀜𑁆𑀜𑀸𑀬 𑀕𑀡𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀑𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀦𑁆𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀦𑁆”**𑀢𑀺𑁇 (*𑁪*:**𑁧𑁨**)

(696.)

2276\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2277\. **𑀲𑀫𑀕𑁆𑀕𑁄** 𑀦𑀸𑀫 **𑀲𑀁𑀖𑁄** 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑁄 𑀲𑀫𑀸𑀦𑀲𑀻𑀫𑀸𑀬𑀁 𑀞𑀺𑀢𑁄𑁇

2278\. **𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸** 𑀦𑀸𑀫 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀯𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀯𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑁇

2279\. **𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦𑀸**𑀢𑀺 𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀬𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦𑁇

2280\. **𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦𑀸**𑀢𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑀦𑁇

2281\. **𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑀦𑁆**𑀢𑀺 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑁇

2282\. **𑀅𑀦𑀜𑁆𑀜𑀸𑀬 𑀕𑀡𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀦𑁆**𑀢𑀺 𑀕𑀡𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀅𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸𑁇

2283\. “𑀑𑀲𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀡𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀲𑀻𑀫𑀁 𑀯𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸, 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2284\. **𑀅𑀬𑀫𑁆𑀧𑀻**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬𑁄 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2285\. **𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀦𑁆**𑀢𑀺 𑀲𑀳 𑀯𑀢𑁆𑀣𑀼𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬𑁇

2286\. **𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀦𑁆**𑀢𑀺 𑀲𑀁𑀖𑀫𑁆𑀳𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀻𑀬𑀢𑀺𑁇

2287\. **𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄**𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀦𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺𑁇

(697.)

2288\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀑𑀲𑀸𑀭𑁂𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇

2289\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(698.)

2290\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑀁 𑀅𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀕𑀡𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀯𑀢𑁆𑀢𑁂 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀁 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀅𑀲𑀦𑁆𑀢𑁂 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀓𑀲𑀁𑀖𑁂 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2291\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
