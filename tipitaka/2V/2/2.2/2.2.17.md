# 𑁨𑁇𑁨𑁇𑁧𑁭 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟

* 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

2421\. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄, 𑀅𑀬𑁆𑀬𑀸𑀬𑁄, 𑀲𑀢𑁆𑀢𑀭𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀦𑀯 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸, 𑀅𑀝𑁆𑀞 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸𑁇 𑀬𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀧𑀓𑁆𑀔𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑀢𑁆𑀣 𑀲𑀺𑀬𑀸 𑀯𑀻𑀲𑀢𑀺𑀕𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀢𑀢𑁆𑀣 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀩𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀏𑀓𑀸𑀬𑀧𑀺 𑀘𑁂 𑀊𑀦𑁄 𑀯𑀻𑀲𑀢𑀺𑀕𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬𑁇 𑀲𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀦𑀩𑁆𑀪𑀺𑀢𑀸, 𑀢𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀸𑀭𑀬𑁆𑀳𑀸, 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀸𑀫𑀻𑀘𑀺𑁇

2422\. 𑀢𑀢𑁆𑀣𑀸𑀬𑁆𑀬𑀸𑀬𑁄 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑁆𑀬𑀸𑀬𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻𑀢𑀺𑁇

---

2423\. 𑀲𑀢𑁆𑀢𑀭𑀲𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀓𑀡𑁆𑀟𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
