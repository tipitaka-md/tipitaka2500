# 𑁨𑁇𑁩𑁇𑁧𑁇𑁨 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀪𑀸𑀚𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀪𑀸𑀚𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(738.)

2441\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀸𑀫𑀓𑀸𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀕𑀫𑀁𑀲𑀼 𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀤𑀼𑀘𑁆𑀘𑁄𑀍𑀆 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸𑁇 𑀉𑀧𑀸𑀲𑀓𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀇𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀤𑀼𑀘𑁆𑀘𑁄𑀍𑀆 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸, 𑀇𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 𑀅𑀤𑀁𑀲𑀼𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀓𑀣𑀺𑀦𑀁 𑀅𑀢𑁆𑀣𑀢𑀁 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀲𑀺𑁇 𑀉𑀧𑀸𑀲𑀓𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀧𑀬𑁆𑀬𑀸𑀳𑀺 𑀘𑀻𑀯𑀭𑀁 𑀮𑀤𑁆𑀥𑀦𑁆”𑀢𑀺? “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀁 𑀮𑀪𑀸𑀫𑁇 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸—  ‘𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀓𑀣𑀺𑀦𑀁 𑀅𑀢𑁆𑀣𑀢𑀁 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆’𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀲𑀻”𑀢𑀺𑁇 𑀉𑀧𑀸𑀲𑀓𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 ‘𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆’𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

2442\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 ‘𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆’𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 ‘𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆’𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 “𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(739.)

2443\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 ‘𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆’𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨*:**𑁨𑁭**)

(740.)

2444\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2445\. **𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀅𑀦𑀢𑁆𑀣𑀢𑁂 𑀓𑀣𑀺𑀦𑁂 𑀏𑀓𑀸𑀤𑀲𑀫𑀸𑀲𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀢𑁂 𑀓𑀣𑀺𑀦𑁂 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀓𑀸𑀮𑁂𑀧𑀺 𑀆𑀤𑀺𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀏𑀢𑀁 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 𑀦𑀸𑀫𑁇

2446\. 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 “𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 ‘𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆’𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁, 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺…𑀧𑁂…  𑀅𑀬𑁆𑀬𑀸𑀬 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺𑁇

(741.)

2447\. 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑁂 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸 “𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 “𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑁂 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸 “𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑁂 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑁂 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(742.)

2448\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀺, 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
