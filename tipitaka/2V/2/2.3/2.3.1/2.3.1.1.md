# 𑁨𑁇𑁩𑁇𑁧𑁇𑁧 𑀧𑀢𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀢𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

---

2424\. 𑀇𑀫𑁂 𑀔𑁄 𑀧𑀦𑀸𑀬𑁆𑀬𑀸𑀬𑁄 𑀢𑀺𑀁𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸

  
𑀥𑀫𑁆𑀫𑀸 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇



(733.)

2425\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀸𑀭𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀩𑀳𑀽 𑀧𑀢𑁆𑀢𑁂 𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀧𑀢𑁆𑀢𑀯𑀸𑀡𑀺𑀚𑁆𑀚𑀁 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀧𑀡𑀁 𑀯𑀸 𑀧𑀲𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2426\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀢𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀢𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀢𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(734.)

2427\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀢𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑁂𑀬𑁆𑀬**, **𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧*:**𑁨𑁬**)

(735.)

2428\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2429\. **𑀧𑀢𑁆𑀢𑁄** 𑀦𑀸𑀫 𑀤𑁆𑀯𑁂 𑀧𑀢𑁆𑀢𑀸—  𑀅𑀬𑁄𑀧𑀢𑁆𑀢𑁄, 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑁄𑁇 𑀢𑀬𑁄 𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀸—  𑀉𑀓𑁆𑀓𑀝𑁆𑀞𑁄 𑀧𑀢𑁆𑀢𑁄, 𑀫𑀚𑁆𑀛𑀺𑀫𑁄 𑀧𑀢𑁆𑀢𑁄, 𑀑𑀫𑀓𑁄 𑀧𑀢𑁆𑀢𑁄𑁇 **𑀉𑀓𑁆𑀓𑀝𑁆𑀞𑁄** **𑀦𑀸𑀫** **𑀧𑀢𑁆𑀢𑁄** 𑀅𑀟𑁆𑀠𑀸𑀍𑀳𑀓𑁄𑀤𑀦𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀘𑀢𑀼𑀪𑀸𑀕𑀁 𑀔𑀸𑀤𑀦𑀁 𑀢𑀤𑀼𑀧𑀺𑀬𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁𑁇 **𑀫𑀚𑁆𑀛𑀺𑀫𑁄 𑀦𑀸𑀫** **𑀧𑀢𑁆𑀢𑁄** 𑀦𑀸𑀍𑀇𑀓𑁄𑀤𑀦𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀘𑀢𑀼𑀪𑀸𑀕𑀁 𑀔𑀸𑀤𑀦𑀁 𑀢𑀤𑀼𑀧𑀺𑀬𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁𑁇 **𑀑𑀫𑀓𑁄 𑀦𑀸𑀫** **𑀧𑀢𑁆𑀢𑁄** 𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀘𑀢𑀼𑀪𑀸𑀕𑀁 𑀔𑀸𑀤𑀦𑀁 𑀢𑀤𑀼𑀧𑀺𑀬𑀁 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁𑁇 𑀢𑀢𑁄 𑀉𑀓𑁆𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑀢𑁆𑀢𑁄 𑀑𑀫𑀓𑁄 𑀅𑀧𑀢𑁆𑀢𑁄𑁇

2430\. **𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀯𑀺𑀓𑀧𑁆𑀧𑀺𑀢𑁄𑁇

2431\. **𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀲𑀳 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  “𑀅𑀬𑀁 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀧𑀢𑁆𑀢𑁄 𑀭𑀢𑁆𑀢𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄, 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀩𑀮𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—

2432\. “𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀧𑀢𑁆𑀢𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀫𑀁 𑀧𑀢𑁆𑀢𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀤𑀤𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

2433\. 𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  “𑀅𑀬𑀁 𑀫𑁂, 𑀅𑀬𑁆𑀬𑀸𑀬𑁄, 𑀧𑀢𑁆𑀢𑁄 𑀭𑀢𑁆𑀢𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄, 𑀇𑀫𑀸𑀳𑀁 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀩𑀮𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—

2434\. “𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑁂, 𑀅𑀬𑁆𑀬𑀸𑀬𑁄𑁇 𑀅𑀬𑀁 𑀧𑀢𑁆𑀢𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁄𑁇 𑀬𑀤𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀅𑀬𑁆𑀬𑀸𑀬𑁄, 𑀇𑀫𑀁 𑀧𑀢𑁆𑀢𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇

2435\. 𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑀸—  “𑀅𑀬𑀁 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀧𑀢𑁆𑀢𑁄 𑀭𑀢𑁆𑀢𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑁄𑁇 𑀇𑀫𑀸𑀳𑀁 𑀅𑀬𑁆𑀬𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—  “𑀇𑀫𑀁 𑀧𑀢𑁆𑀢𑀁 𑀅𑀬𑁆𑀬𑀸𑀬 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺𑁇

(736.)

2436\. 𑀭𑀢𑁆𑀢𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀭𑀢𑁆𑀢𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀭𑀢𑁆𑀢𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀯𑀺𑀓𑀧𑁆𑀧𑀺𑀢𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀦𑀝𑁆𑀞𑁂 𑀦𑀝𑁆𑀞𑀲𑀜𑁆𑀜𑀸…  𑀅𑀯𑀺𑀦𑀝𑁆𑀞𑁂 𑀯𑀺𑀦𑀝𑁆𑀞𑀲𑀜𑁆𑀜𑀸…  𑀅𑀪𑀺𑀦𑁆𑀦𑁂 𑀪𑀺𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸…  𑀅𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑁂 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2437\. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀢𑁆𑀢𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀭𑀢𑁆𑀢𑀸𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀭𑀢𑁆𑀢𑀸𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀭𑀢𑁆𑀢𑀸𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(737.)

2438\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀦𑁆𑀢𑁄𑀅𑀭𑀼𑀡𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀢𑀺, 𑀯𑀺𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀺, 𑀦𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺, 𑀪𑀺𑀚𑁆𑀚𑀢𑀺, 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀁 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

2439\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑀁 𑀦 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀧𑀢𑁆𑀢𑁄 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀸 𑀦 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
