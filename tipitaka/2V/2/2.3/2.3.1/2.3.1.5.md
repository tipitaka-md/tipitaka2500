# 𑁨𑁇𑁩𑁇𑁧𑁇𑁫 𑀅𑀜𑁆𑀜𑀘𑁂𑀢𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀅𑀜𑁆𑀜𑀘𑁂𑀢𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(753.)

2473\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀔𑀫𑀦𑀻𑀬𑀁 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀦𑁆”𑀢𑀺? “𑀦 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀦 𑀬𑀸𑀧𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀫𑀼𑀓𑀲𑁆𑀲, 𑀅𑀬𑁆𑀬𑁂, 𑀆𑀧𑀡𑀺𑀓𑀲𑁆𑀲 𑀖𑀭𑁂 𑀓𑀳𑀸𑀧𑀡𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁄 𑀬𑀁 𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑀁 𑀆𑀳𑀭𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑁂, 𑀅𑀫𑀼𑀓𑀲𑁆𑀲 𑀆𑀧𑀡𑀺𑀓𑀲𑁆𑀲 𑀖𑀭𑀸 𑀓𑀳𑀸𑀧𑀡𑀲𑁆𑀲 𑀢𑁂𑀮𑀁 𑀆𑀳𑀭𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀢𑀲𑁆𑀲 𑀆𑀧𑀡𑀺𑀓𑀲𑁆𑀲 𑀖𑀭𑀸 𑀓𑀳𑀸𑀧𑀡𑀲𑁆𑀲 𑀢𑁂𑀮𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀤𑀸𑀲𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀯𑀫𑀸𑀳—  “𑀦 𑀫𑁂, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑁂, 𑀢𑁂𑀮𑁂𑀦 𑀅𑀢𑁆𑀣𑁄, 𑀲𑀧𑁆𑀧𑀺𑀦𑀸 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀆𑀧𑀡𑀺𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀆𑀧𑀡𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦 𑀓𑀺𑀭, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀬𑁆𑀬𑀸𑀬 𑀢𑁂𑀮𑁂𑀦 𑀅𑀢𑁆𑀣𑁄, 𑀲𑀧𑁆𑀧𑀺𑀦𑀸 𑀅𑀢𑁆𑀣𑁄, 𑀳𑀦𑁆𑀤 𑀢𑁂 𑀢𑁂𑀮𑀁, 𑀲𑀧𑁆𑀧𑀺𑀁 𑀫𑁂 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀲𑀘𑁂 𑀫𑀬𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀺𑀓𑁆𑀓𑀻𑀢𑀁 𑀪𑀡𑁆𑀟𑀁 𑀧𑀼𑀦 𑀆𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫, 𑀓𑀤𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀡𑁆𑀟𑀁 𑀯𑀺𑀓𑁆𑀓𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑁂𑀮𑀲𑁆𑀲 𑀓𑀬𑁂𑀦 𑀢𑁂𑀮𑀁 𑀳𑀝𑀁, 𑀲𑀧𑁆𑀧𑀺𑀲𑁆𑀲 𑀓𑀬𑀁 𑀆𑀳𑀭, 𑀲𑀧𑁆𑀧𑀺𑀁 𑀳𑀭𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀭𑁄𑀤𑀦𑁆𑀢𑀻 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑁂, 𑀭𑁄𑀤𑀲𑀻”𑀢𑀺?

2474\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(754.)

2475\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫*:**𑁩𑁦**)

(755.)

2476\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2477\. **𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑁇

2478\. **𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁, 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀬𑁆𑀬𑀸𑀬 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(756.)

2479\. 𑀅𑀜𑁆𑀜𑁂 𑀅𑀜𑁆𑀜𑀲𑀜𑁆𑀜𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀜𑁆𑀜𑁂 𑀅𑀦𑀜𑁆𑀜𑀲𑀜𑁆𑀜𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2480\. 𑀅𑀦𑀜𑁆𑀜𑁂 𑀅𑀜𑁆𑀜𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀜𑁆𑀜𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀅𑀦𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀜𑁆𑀜𑁂 𑀅𑀦𑀜𑁆𑀜𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(757.)

2481\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀢𑀜𑁆𑀜𑁂𑀯 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑀜𑁆𑀘 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
