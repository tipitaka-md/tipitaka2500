# 𑁨𑁇𑁩𑁇𑁧𑁇𑁩 𑀘𑀻𑀯𑀭𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀘𑀻𑀯𑀭𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(743.)

2450\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀢𑀭𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀲𑀗𑁆𑀖𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀫𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀁, 𑀓𑀳𑀁 𑀢𑀁 𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑁂𑀲𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀳𑀦𑁆𑀤𑀸𑀬𑁆𑀬𑁂, 𑀢𑀼𑀬𑁆𑀳𑀁 𑀘𑀻𑀯𑀭𑀁, 𑀆𑀳𑀭 𑀫𑁂𑀢𑀁 𑀘𑀻𑀯𑀭𑀁, 𑀬𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀢𑀼𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑀁, 𑀬𑀁 𑀫𑀬𑁆𑀳𑀁 𑀫𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑀁, 𑀆𑀳𑀭 𑀫𑁂𑀢𑀁, 𑀲𑀓𑀁 𑀧𑀘𑁆𑀘𑀸𑀳𑀭𑀸”𑀢𑀺 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑁇

2451\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(744.)

2452\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀳𑀦𑁆𑀤𑀸𑀬𑁆𑀬𑁂, 𑀢𑀼𑀬𑁆𑀳𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀳𑀭, 𑀫𑁂𑀢𑀁 𑀘𑀻𑀯𑀭𑀁, 𑀬𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀢𑀼𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑀁, 𑀬𑀁 𑀫𑀬𑁆𑀳𑀁 𑀫𑀬𑁆𑀳𑀫𑁂𑀯𑁂𑀢𑀁, 𑀆𑀳𑀭 𑀫𑁂𑀢𑀁, 𑀲𑀓𑀁 𑀧𑀘𑁆𑀘𑀸𑀳𑀭𑀸’𑀢𑀺 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩*:**𑁨𑁮**)

(745.)

2453\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2454\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆**𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁𑁇

2455\. **𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀙𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀼𑀧𑀕𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

2456\. **𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀯𑀺𑀧𑀼𑀮𑀁, 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀁𑁇

2457\. **𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2458\. **𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀓𑀺𑀁 𑀆𑀡𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀓𑀫𑁆𑀧𑀺 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁, 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺…𑀧𑁂…  𑀅𑀬𑁆𑀬𑀸𑀬 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺𑁇

(746.)

2459\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2460\. 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(747.)

2461\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀸 𑀯𑀸 𑀤𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸 𑀯𑀸 𑀯𑀺𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑀻 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
