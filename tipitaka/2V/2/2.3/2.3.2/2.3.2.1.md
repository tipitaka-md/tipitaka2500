# 𑁨𑁇𑁩𑁇𑁨𑁇𑁧 𑀕𑀭𑀼𑀧𑀸𑀯𑀼𑀭𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀻𑀯𑀭𑀯𑀕𑁆𑀕

### 𑀕𑀭𑀼𑀧𑀸𑀯𑀼𑀭𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(783.)

2541. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀢𑀺 𑀪𑀸𑀡𑀺𑀓𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀝𑁆𑀝𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀲𑀻𑀢𑀓𑀸𑀮𑁂 𑀫𑀳𑀕𑁆𑀖𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀧𑀲𑁂𑀦𑀤𑀺𑀁 𑀓𑁄𑀲𑀮𑀁 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑁂𑀦 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺? “𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄𑀲𑀺, 𑀇𑀫𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀓𑀫𑁆𑀩𑀮𑀁 𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀫𑀳𑀺𑀘𑁆𑀙𑀸 𑀇𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀭𑀸𑀚𑀸𑀦𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2542. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀭𑀸𑀚𑀸𑀦𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀭𑀸𑀚𑀸𑀦𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(784.)

2543. **“𑀕𑀭𑀼𑀧𑀸𑀯𑀼𑀭𑀡𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀧𑀭𑀫𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀢𑁄 𑀘𑁂 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘𑁂𑀢𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁧*:**𑁩𑁬**)

(785.)

2544. **𑀕𑀭𑀼𑀧𑀸𑀯𑀼𑀭𑀡𑀁** 𑀦𑀸𑀫 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀻𑀢𑀓𑀸𑀮𑁂 𑀧𑀸𑀯𑀼𑀭𑀡𑀁𑁇

2545. **𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸**𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑁇

2546. **𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀧𑀭𑀫𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀦𑁆**𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀓𑀳𑀸𑀧𑀡𑀕𑁆𑀖𑀦𑀓𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁𑁇

2547. **𑀢𑀢𑁄 𑀘𑁂 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘𑁂𑀢𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀸 𑀕𑀡𑀲𑁆𑀲 𑀯𑀸 𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  “𑀇𑀤𑀁 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀕𑀭𑀼𑀧𑀸𑀯𑀼𑀭𑀡𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀧𑀭𑀫𑀁 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁, 𑀇𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀢𑀺…𑀧𑁂…  𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀬𑁆𑀬𑀸𑀬 𑀤𑀫𑁆𑀫𑀻𑀢𑀺𑁇

(786.)

2548. 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇

2549. 𑀊𑀦𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(787.)

2550. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀧𑀭𑀫𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀊𑀦𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀧𑀭𑀫𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀜𑀸𑀢𑀓𑀸𑀦𑀁, 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀥𑀦𑁂𑀦, 𑀫𑀳𑀕𑁆𑀖𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀼𑀓𑀸𑀫𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀕𑁆𑀖𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
