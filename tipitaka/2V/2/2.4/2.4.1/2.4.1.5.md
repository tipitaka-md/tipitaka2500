# 𑁨𑁇𑁪𑁇𑁧𑁇𑁫 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(810.)

2599\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺 𑀕𑁄𑀢𑀫𑀻 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀥𑁄𑀯𑀸𑀢𑁂 𑀅𑀝𑁆𑀞𑀸𑀲𑀺—  “𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑁄, 𑀪𑀕𑀯𑀸, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸—  “𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑀼 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀦𑁆”𑀢𑀺, 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀕𑁄𑀢𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺 𑀕𑁄𑀢𑀫𑀻 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀪𑀕𑀯𑀢𑀸 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸”𑀢𑀺 𑀅𑀢𑀺𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀫𑀼𑀢𑁆𑀢𑀓𑀭𑀡𑁂 𑀯𑀡𑀁 𑀅𑀓𑀸𑀲𑀺𑁇

2600\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀢𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑀺𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀆𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑀺𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀆𑀤𑀺𑀬𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑀺𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀆𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(811.)

2601\. **“𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀆𑀤𑀺𑀬𑀫𑀸𑀦𑀸𑀬 𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑀧𑀭𑀫𑀁 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫*:**𑁬𑁦**)

(812.)

2602\. **𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁** 𑀦𑀸𑀫 𑀫𑀼𑀢𑁆𑀢𑀓𑀭𑀡𑀲𑁆𑀲 𑀥𑁄𑀯𑀦𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2603\. **𑀆𑀤𑀺𑀬𑀫𑀸𑀦𑀸𑀬𑀸**𑀢𑀺 𑀥𑁄𑀯𑀦𑁆𑀢𑀺𑀬𑀸𑁇

2604\. **𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑀧𑀭𑀫𑀁 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀦𑁆**𑀢𑀺 𑀤𑁆𑀯𑀻𑀲𑀼 𑀅𑀗𑁆𑀕𑀼𑀮𑁂𑀲𑀼 𑀤𑁆𑀯𑁂 𑀧𑀩𑁆𑀩𑀧𑀭𑀫𑀸 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀸𑁇

2605\. **𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸**𑀢𑀺 𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀓𑁂𑀲𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(813.)

2606\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀸 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀸 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2607\. 𑀊𑀦𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑁂 𑀊𑀦𑀓𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(814.)

2608\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑀧𑀭𑀫𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀊𑀦𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑀧𑀭𑀫𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀆𑀩𑀸𑀥𑀧𑀘𑁆𑀘𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2609\. 𑀧𑀜𑁆𑀘𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
