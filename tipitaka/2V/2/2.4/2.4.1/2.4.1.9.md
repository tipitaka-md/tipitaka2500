# 𑁨𑁇𑁪𑁇𑁧𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑁅𑀘𑁆𑀘𑀸𑀭𑀙𑀟𑁆𑀟𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑁅𑀘𑁆𑀘𑀸𑀭𑀙𑀟𑁆𑀟𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(828.)

2649\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀬𑀯𑀔𑁂𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀲𑀗𑁆𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀯𑀺𑀖𑀸𑀲𑀫𑁆𑀧𑀺 𑀔𑁂𑀢𑁆𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀬𑀯𑀔𑁂𑀢𑁆𑀢𑀁 𑀤𑀽𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2650\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀲𑀗𑁆𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀯𑀺𑀖𑀸𑀲𑀫𑁆𑀧𑀺 𑀳𑀭𑀺𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀲𑀗𑁆𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀯𑀺𑀖𑀸𑀲𑀫𑁆𑀧𑀺 𑀳𑀭𑀺𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀲𑀗𑁆𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀯𑀺𑀖𑀸𑀲𑀫𑁆𑀧𑀺 𑀳𑀭𑀺𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(829.)

2651\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀁 𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀁 𑀯𑀸 𑀳𑀭𑀺𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬 𑀯𑀸 𑀙𑀟𑁆𑀟𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁯*:**𑁬𑁪**)

(830.)

2652\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2653\. **𑀉𑀘𑁆𑀘𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀕𑀽𑀣𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀧𑀲𑁆𑀲𑀸𑀯𑁄** 𑀦𑀸𑀫 𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2654\. **𑀲𑀗𑁆𑀓𑀸𑀭𑀁** 𑀦𑀸𑀫 𑀓𑀘𑀯𑀭𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2655\. **𑀯𑀺𑀖𑀸𑀲𑀁** 𑀦𑀸𑀫 𑀘𑀮𑀓𑀸𑀦𑀺 𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀘𑁆𑀙𑀺𑀝𑁆𑀞𑁄𑀤𑀓𑀁 𑀯𑀸𑁇

2656\. **𑀳𑀭𑀺𑀢𑀁** 𑀦𑀸𑀫 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀁 𑀅𑀧𑀭𑀡𑁆𑀡𑀁 𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀧𑀪𑁄𑀕𑀧𑀭𑀺𑀪𑁄𑀕𑀁 𑀭𑁄𑀧𑀺𑀫𑀁𑁇

2657\. **𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2658\. **𑀙𑀟𑁆𑀟𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀓𑀺𑀁 𑀆𑀡𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀓𑀫𑁆𑀧𑀺 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(831.)

2659\. 𑀳𑀭𑀺𑀢𑁂 𑀳𑀭𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑀺 𑀯𑀸 𑀙𑀟𑁆𑀟𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀳𑀭𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑀺 𑀯𑀸 𑀙𑀟𑁆𑀟𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀳𑀭𑀺𑀢𑁂 𑀅𑀳𑀭𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑀺 𑀯𑀸 𑀙𑀟𑁆𑀟𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2660\. 𑀅𑀳𑀭𑀺𑀢𑁂 𑀳𑀭𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀳𑀭𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀳𑀭𑀺𑀢𑁂 𑀅𑀳𑀭𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(832.)

2661\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀔𑁂𑀢𑁆𑀢𑀫𑀭𑀺𑀬𑀸𑀤𑁂 𑀙𑀟𑁆𑀟𑁂𑀢𑀺 𑀲𑀸𑀫𑀺𑀓𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2662\. 𑀦𑀯𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
