# 𑁨𑁇𑁪𑁇𑁧𑁇𑁬 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(815.)

2610\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀦𑀸𑀫 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀪𑀢𑁆𑀢𑀯𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀘 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀘 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀘𑀸𑀯𑀤𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀧𑀲𑀸𑀤𑁂𑀢𑀺—  “𑀫𑀸, 𑀪𑀕𑀺𑀦𑀺, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 “𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀁 𑀢𑁆𑀯𑀁 𑀏𑀯𑀜𑁆𑀘 𑀏𑀯𑀜𑁆𑀘 𑀓𑀭𑁄𑀲𑀺, 𑀇𑀤𑀸𑀦𑀺 𑀏𑀢𑁆𑀢𑀓𑀁 𑀦 𑀲𑀳𑀲𑀻”𑀢𑀺—  𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀫𑀢𑁆𑀣𑀓𑁂 𑀆𑀲𑀼𑀫𑁆𑀪𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀺𑁇

2611\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀳𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀳𑀸𑀭𑀁 𑀅𑀤𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀳𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(816.)

2612\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁬*:**𑁬𑁧**)

(817.)

2613\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2614\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑀸**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑁇

2615\. **𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲𑀸**𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀪𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲𑁇

2616\. **𑀧𑀸𑀦𑀻𑀬𑀁** 𑀦𑀸𑀫 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀧𑀸𑀦𑀻𑀬𑀁𑁇

2617\. **𑀯𑀺𑀥𑀽𑀧𑀦𑀁** 𑀦𑀸𑀫 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀩𑀻𑀚𑀦𑀻𑁇

2618\. **𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(818.)

2619\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2620\. 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(819.)

2621\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀤𑁂𑀢𑀺, 𑀤𑀸𑀧𑁂𑀢𑀺, 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2622\. 𑀙𑀝𑁆𑀞𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
