# 𑁨𑁇𑁪𑁇𑁧𑁇𑁪 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(806.)

2592. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀧𑀼𑀭𑀸𑀡𑀭𑀸𑀚𑁄𑀭𑁄𑀥𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀬𑀸 𑀧𑀻𑀍𑀇𑀢𑀸 𑀬𑁂𑀦 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀭𑀸𑀚𑀸 𑀔𑁄, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀘𑀺𑀭𑀸𑀘𑀺𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀣𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀥𑀸𑀭𑁂𑀣𑀸”𑀢𑀺? “𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑁂𑀦, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀏𑀢𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀥𑁄𑀯𑀺𑀢𑀼𑀁 𑀯𑀺𑀲𑁆𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀙𑀟𑁆𑀟𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀫𑀓𑁆𑀔𑀺𑀓𑀸𑀳𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀲𑁆𑀲𑀺𑀤𑀁 𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺? 𑀲𑀸 𑀏𑀯𑀫𑀸𑀳—  “𑀫𑀬𑁆𑀳𑀺𑀤𑀁 𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇

2593. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁 𑀆𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁 𑀆𑀤𑀺𑀬𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁 𑀆𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(807.)

2594. **“𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁪*:**𑁫𑁯**)

(808.)

2595. **𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁** 𑀦𑀸𑀫 𑀚𑀢𑀼𑀫𑀬𑀁 𑀓𑀝𑁆𑀞𑀫𑀬𑀁 𑀧𑀺𑀝𑁆𑀞𑀫𑀬𑀁 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀫𑀬𑀁𑁇

2596. **𑀆𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀉𑀧𑁆𑀧𑀮𑀧𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀫𑀼𑀢𑁆𑀢𑀓𑀭𑀡𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(809.)

2597. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀩𑀸𑀥𑀧𑀘𑁆𑀘𑀬𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2598. 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
