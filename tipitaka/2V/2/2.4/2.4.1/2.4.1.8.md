# 𑁨𑁇𑁪𑁇𑁧𑁇𑁮 𑀉𑀘𑁆𑀘𑀸𑀭𑀙𑀟𑁆𑀟𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

* 𑀉𑀘𑁆𑀘𑀸𑀭𑀙𑀟𑁆𑀟𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(824.)

2635\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀺𑀩𑁆𑀩𑀺𑀝𑁆𑀞𑀭𑀸𑀚𑀪𑀝𑁄 “𑀢𑀜𑁆𑀜𑁂𑀯 𑀪𑀝𑀧𑀣𑀁 𑀬𑀸𑀘𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀲𑀻𑀲𑀁 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀭𑀸𑀚𑀓𑀼𑀮𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀝𑀸𑀳𑁂 𑀯𑀘𑁆𑀘𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀻 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀫𑀢𑁆𑀣𑀓𑁂 𑀆𑀲𑀼𑀫𑁆𑀪𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀅𑀲𑁆𑀲𑀫𑀡𑀺𑀬𑁄 𑀇𑀫𑀸 𑀫𑀼𑀡𑁆𑀟𑀸 𑀩𑀦𑁆𑀥𑀓𑀺𑀦𑀺𑀬𑁄𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀕𑀽𑀣𑀓𑀝𑀸𑀳𑀁 𑀫𑀢𑁆𑀣𑀓𑁂 𑀆𑀲𑀼𑀫𑁆𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑀸𑀲𑀁 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀛𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀉𑀫𑁆𑀫𑀼𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀧𑀯𑀺𑀲𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀉𑀧𑀲𑁆𑀲𑀬𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀉𑀫𑁆𑀫𑀼𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀪𑁄, 𑀉𑀫𑁆𑀫𑀼𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀧𑀯𑀺𑀲𑀲𑀻”𑀢𑀺? “𑀇𑀫𑀸 𑀫𑀁, 𑀪𑁄, 𑀫𑀼𑀡𑁆𑀟𑀸 𑀩𑀦𑁆𑀥𑀓𑀺𑀦𑀺𑀬𑁄 𑀕𑀽𑀣𑀓𑀝𑀸𑀳𑀁 𑀫𑀢𑁆𑀣𑀓𑁂 𑀆𑀲𑀼𑀫𑁆𑀪𑀺𑀁𑀲𑀼𑁇 𑀇𑀫𑀸𑀲𑀁 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀛𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀕𑀘𑁆𑀙, 𑀪𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀫𑀗𑁆𑀕𑀮𑀁 𑀏𑀢𑀁𑁇 𑀲𑀳𑀲𑁆𑀲𑀁 𑀮𑀘𑁆𑀙𑀲𑀺 𑀢𑀜𑁆𑀘 𑀪𑀝𑀧𑀣𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀻𑀲𑀁 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀳𑀲𑁆𑀲𑀁 𑀅𑀮𑀢𑁆𑀣 𑀢𑀜𑁆𑀘 𑀪𑀝𑀧𑀣𑀁𑁇

2636\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(825.)

2637\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀁 𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀁 𑀯𑀸 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀯𑀸 𑀢𑀺𑀭𑁄𑀧𑀸𑀓𑀸𑀭𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬 𑀯𑀸 𑀙𑀟𑁆𑀟𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁮*:**𑁬𑁩**)

(826.)

2638\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2639\. **𑀉𑀘𑁆𑀘𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀕𑀽𑀣𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀧𑀲𑁆𑀲𑀸𑀯𑁄** 𑀦𑀸𑀫 𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2640\. **𑀲𑀗𑁆𑀓𑀸𑀭𑀁** 𑀦𑀸𑀫 𑀓𑀘𑀯𑀭𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2641\. **𑀯𑀺𑀖𑀸𑀲𑀁** 𑀦𑀸𑀫 𑀘𑀮𑀓𑀸𑀦𑀺 𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀓𑀸𑀦𑀺 𑀯𑀸 𑀉𑀘𑁆𑀙𑀺𑀝𑁆𑀞𑁄𑀤𑀓𑀁 𑀯𑀸𑁇

2642\. **𑀓𑀼𑀝𑁆𑀝𑁄** 𑀦𑀸𑀫 𑀢𑀬𑁄 𑀓𑀼𑀝𑁆𑀝𑀸—  𑀇𑀝𑁆𑀞𑀓𑀸𑀓𑀼𑀝𑁆𑀝𑁄, 𑀲𑀺𑀮𑀸𑀓𑀼𑀝𑁆𑀝𑁄, 𑀤𑀸𑀭𑀼𑀓𑀼𑀝𑁆𑀝𑁄𑁇

2643\. **𑀧𑀸𑀓𑀸𑀭𑁄** 𑀦𑀸𑀫 𑀢𑀬𑁄 𑀧𑀸𑀓𑀸𑀭𑀸—  𑀇𑀝𑁆𑀞𑀓𑀸𑀧𑀸𑀓𑀸𑀭𑁄, 𑀲𑀺𑀮𑀸𑀧𑀸𑀓𑀸𑀭𑁄, 𑀤𑀸𑀭𑀼𑀧𑀸𑀓𑀸𑀭𑁄𑁇

2644\. **𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂**𑀢𑀺 𑀓𑀼𑀝𑁆𑀝𑀲𑁆𑀲 𑀧𑀭𑀢𑁄𑁇 **𑀢𑀺𑀭𑁄𑀧𑀸𑀓𑀸𑀭𑁂**𑀢𑀺 𑀧𑀸𑀓𑀸𑀭𑀲𑁆𑀲 𑀧𑀭𑀢𑁄𑁇

2645\. **𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2646\. **𑀙𑀟𑁆𑀟𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀓𑀺𑀁 𑀆𑀡𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀓𑀫𑁆𑀧𑀺 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(827.)

2647\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀅𑀯𑀍𑀅𑀜𑁆𑀚𑁂 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2648\. 𑀅𑀝𑁆𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
