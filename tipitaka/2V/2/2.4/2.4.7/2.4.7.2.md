# 𑁨𑁇𑁪𑁇𑁭𑁇𑁨 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀕𑀩𑁆𑀪𑀺𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1072.)

3191\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀤𑁂𑀣𑀸𑀬𑁆𑀬𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀁, 𑀲𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀅𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

3192\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1073.)

3193\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁬𑁨*:**𑁧𑁧𑁭**)

(1074.)

3194\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3195\. **𑀧𑀸𑀬𑀦𑁆𑀢𑀻** 𑀦𑀸𑀫 𑀫𑀸𑀢𑀸 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀥𑀸𑀢𑀺 𑀯𑀸𑁇

3196\. **𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬𑁇

3197\. “𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀡𑀁 𑀯𑀸 𑀆𑀘𑀭𑀺𑀦𑀺𑀁 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀲𑀻𑀫𑀁 𑀯𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑁇 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀕𑀡𑀲𑁆𑀲 𑀘 𑀆𑀘𑀭𑀺𑀦𑀺𑀬𑀸 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(1075.)

3198\. 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(1076.)

3199\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀁 𑀅𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3200\. 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
