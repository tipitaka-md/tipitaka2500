# 𑁨𑁇𑁪𑁇𑁭𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀕𑀩𑁆𑀪𑀺𑀦𑀺𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1115.)

3285\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀢𑀺 𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀢𑀺𑁇 𑀲𑀸𑀫𑀺𑀓𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇

3286\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺, 𑀲𑀸𑀫𑀺𑀓𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀲𑀘𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀅𑀲𑁆𑀲, 𑀦 𑀘 𑀲𑀸𑀫𑀺𑀓𑁄 𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀸”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀢𑀺 𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀢𑀺, 𑀲𑀸𑀫𑀺𑀓𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺, 𑀲𑀸𑀫𑀺𑀓𑁄 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1116.)

3287\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀬𑁆𑀬 𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀬𑁆𑀬 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀙𑀧𑁆𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑀧𑀺, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁦*:**𑁧𑁨𑁫**)

(1117.)

3288\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3289\. **𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀻** 𑀦𑀸𑀫 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑀻 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑁆𑀯𑀸𑁇 **𑀦𑁂𑀯** **𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀲𑀬𑀁 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀬𑁆𑀬𑁇 **𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀬𑁆𑀬𑁇 “𑀦𑁂𑀯 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀙𑀧𑁆𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑀧𑀻”𑀢𑀺 𑀥𑀼𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1118.)

3290\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦 𑀮𑀪𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3291\. 𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀕𑀩𑁆𑀪𑀺𑀦𑀺𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇
