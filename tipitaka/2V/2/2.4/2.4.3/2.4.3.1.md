# 𑁨𑁇𑁪𑁇𑁩𑁇𑁧 𑀦𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀦𑀕𑁆𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(883.)

2785\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀘𑀺𑀭𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀯𑁂𑀲𑀺𑀬𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀕𑁆𑀕𑀸 𑀏𑀓𑀢𑀺𑀢𑁆𑀣𑁂 𑀦𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀯𑁂𑀲𑀺𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑁂𑀲𑀼𑀁—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀤𑀳𑀭𑀸𑀦𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀺𑀡𑁆𑀡𑁂𑀦, 𑀦𑀦𑀼 𑀦𑀸𑀫 𑀓𑀸𑀫𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑀤𑀸 𑀚𑀺𑀡𑁆𑀡𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀣 𑀢𑀤𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀣𑁇 𑀏𑀯𑀁 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀪𑁄 𑀅𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑁂𑀲𑀺𑀬𑀸𑀳𑀺 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀺𑀬𑀫𑀸𑀦𑀸 𑀫𑀗𑁆𑀓𑀽 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(884.)

2786\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀦𑀕𑁆𑀕𑀸 𑀦𑀳𑀸𑀬𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁧*:**𑁭𑁬**)

(885.)

2787\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2788\. **𑀦𑀕𑁆𑀕𑀸 𑀦𑀳𑀸𑀬𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀦𑀺𑀯𑀢𑁆𑀣𑀸 𑀯𑀸 𑀅𑀧𑀸𑀭𑀼𑀢𑀸 𑀯𑀸 𑀦𑀳𑀸𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀦𑀳𑀸𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(886.)

2789\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀘𑀻𑀯𑀭𑀺𑀓𑀸𑀬 𑀯𑀸 𑀦𑀝𑁆𑀞𑀘𑀻𑀯𑀭𑀺𑀓𑀸𑀬 𑀯𑀸, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2790\. 𑀧𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
