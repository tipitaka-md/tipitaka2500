# 𑁨𑁇𑁪𑁇𑁩𑁇𑁧𑁦 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(925.)

2872\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀫𑀳𑁂 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀣𑀺𑀦𑀁 𑀅𑀢𑁆𑀣𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀬𑀸𑀘𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀯𑁂 𑀓𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(926.)

2873\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀓𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

2874\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀓𑀣𑀺𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀉𑀤𑁆𑀥𑀸𑀭𑁄, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

2875\. 𑀉𑀩𑁆𑀪𑀢𑀁 𑀲𑀁𑀖𑁂𑀦 𑀓𑀣𑀺𑀦𑀁, 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(927.)

2876\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀬𑀸𑀘𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀘𑀻𑀯𑀭𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀦 𑀤𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2877\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(928.)

2878\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩𑁦*:**𑁮𑁫**)

(929.)

2879\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2880\. **𑀥𑀫𑁆𑀫𑀺𑀓𑁄** 𑀦𑀸𑀫 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄 𑀲𑀫𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑁆𑀥𑀭𑀢𑀺𑁇

2881\. **𑀧𑀝𑀺𑀩𑀸𑀳𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 “𑀓𑀣𑀁 𑀇𑀤𑀁 𑀓𑀣𑀺𑀦𑀁 𑀦 𑀉𑀤𑁆𑀥𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(930.)

2882\. 𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(931.)

2883\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---
