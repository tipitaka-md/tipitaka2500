# 𑁨𑁇𑁪𑁇𑁩𑁇𑁪 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(897.)

2815\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀘𑀺𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀓𑀡𑁆𑀡𑀓𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀑𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀲𑁆𑀲𑀺𑀫𑀸𑀦𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀓𑀡𑁆𑀡𑀓𑀺𑀢𑀸𑀦𑀻”𑀢𑀺?

2816\. 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(898.)

2817\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀜𑁆𑀘𑀸𑀳𑀺𑀓𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀘𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁪*:**𑁭𑁯**)

(899.)

2818\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2819\. **𑀧𑀜𑁆𑀘𑀸𑀳𑀺𑀓𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀘𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀧𑀜𑁆𑀘 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦𑁂𑀯 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺 𑀦 𑀧𑀸𑀭𑀼𑀧𑀢𑀺 𑀦 𑀑𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(900.)

2820\. 𑀧𑀜𑁆𑀘𑀸𑀳𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀜𑁆𑀘𑀸𑀳𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀜𑁆𑀘𑀸𑀳𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2821\. 𑀧𑀜𑁆𑀘𑀸𑀳𑀸𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀜𑁆𑀘𑀸𑀳𑀸𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀜𑁆𑀘𑀸𑀳𑀸𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(901.)

2822\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀧𑀜𑁆𑀘 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀺 𑀯𑀸 𑀧𑀸𑀭𑀼𑀧𑀢𑀺 𑀯𑀸 𑀑𑀢𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2823\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
