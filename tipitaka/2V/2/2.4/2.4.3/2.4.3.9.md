# 𑁨𑁇𑁪𑁇𑁩𑁇𑁯 𑀓𑀸𑀮𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀓𑀸𑀮𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(920.)

2861\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀫𑀬𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑁄𑀫, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀁𑀲𑀼𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀕𑀫𑁂𑀣, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀕𑀘𑁆𑀙𑀸𑀬𑁆𑀬𑁂, 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑁂𑀦 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑁂𑀣𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺𑁇 “𑀦 𑀫𑀬𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀓𑁆𑀓𑁄𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

2862\. 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀬 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀬 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀬 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(921.)

2863\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀬 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁯*:**𑁮𑁪**)

(922.)

2864\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2865\. **𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸** 𑀦𑀸𑀫 “𑀲𑀘𑁂 𑀫𑀬𑀁 𑀲𑀓𑁆𑀓𑁄𑀫, 𑀤𑀲𑁆𑀲𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺 𑀯𑀸𑀘𑀸 𑀪𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

2866\. **𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑁄** 𑀦𑀸𑀫 𑀅𑀦𑀢𑁆𑀣𑀢𑁂 𑀓𑀣𑀺𑀦𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀫𑀸𑀲𑁄, 𑀅𑀢𑁆𑀣𑀢𑁂 𑀓𑀣𑀺𑀦𑁂 𑀧𑀜𑁆𑀘 𑀫𑀸𑀲𑀸𑁇

2867\. **𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀦𑀢𑁆𑀣𑀢𑁂 𑀓𑀣𑀺𑀦𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀢𑁆𑀣𑀢𑁂 𑀓𑀣𑀺𑀦𑁂 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀤𑀺𑀯𑀲𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(923.)

2868\. 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁂 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁂 𑀅𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

2869\. 𑀅𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁂 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁂 𑀅𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(924.)

2870\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2871\. 𑀦𑀯𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
