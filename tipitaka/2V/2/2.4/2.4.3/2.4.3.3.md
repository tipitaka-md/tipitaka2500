# 𑁨𑁇𑁪𑁇𑁩𑁇𑁩 𑀘𑀻𑀯𑀭𑀲𑀺𑀩𑁆𑀩𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀘𑀻𑀯𑀭𑀲𑀺𑀩𑁆𑀩𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(892.)

2801\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀫𑀳𑀕𑁆𑀖𑁂 𑀘𑀻𑀯𑀭𑀤𑀼𑀲𑁆𑀲𑁂 𑀘𑀻𑀯𑀭𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀳𑁄𑀢𑀺 𑀤𑀼𑀲𑁆𑀲𑀺𑀩𑁆𑀩𑀺𑀢𑀁𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀦𑁆𑀤𑀭𑀁 𑀔𑁄 𑀇𑀤𑀁 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀘𑀻𑀯𑀭𑀤𑀼𑀲𑁆𑀲𑀁; 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀔𑁄 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀤𑀼𑀲𑁆𑀲𑀺𑀩𑁆𑀩𑀺𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀫𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? “𑀆𑀫𑀸𑀬𑁆𑀬𑁂, 𑀲𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀤𑀸𑀲𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀲𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺𑁇

2802\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(893.)

2803\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀺𑀦𑀻 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑁂𑀬𑁆𑀬 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁩*:**𑁭𑁮**)

(894.)

2804\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2805\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸𑁇

2806\. **𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀙𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁𑁇

2807\. **𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸𑁇 **𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑁇

2808\. **𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀺𑀦𑀻**𑀢𑀺 𑀅𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂𑁇

2809\. **𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀲𑀬𑀁 𑀲𑀺𑀩𑁆𑀩𑁂𑀬𑁆𑀬𑁇 **𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀬𑁆𑀬𑁇

2810\. **𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀸**𑀢𑀺 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀁𑁇 “𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀥𑀼𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(895.)

2811\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀺𑀦𑀻 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀺𑀦𑀻 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀺𑀦𑀻 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2812\. 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀺𑀦𑀻 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀺𑀦𑀻 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑀢𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(896.)

2813\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀦 𑀮𑀪𑀢𑀺, 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
