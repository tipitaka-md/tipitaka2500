# 𑁨𑁇𑁪𑁇𑁩𑁇𑁮 𑀘𑀻𑀯𑀭𑀤𑀸𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀘𑀻𑀯𑀭𑀤𑀸𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(916.)

2851\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀦𑀝𑀸𑀦𑀫𑁆𑀧𑀺 𑀦𑀝𑀓𑀸𑀦𑀫𑁆𑀧𑀺 𑀮𑀗𑁆𑀖𑀓𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑁄𑀓𑀚𑁆𑀛𑀸𑀬𑀺𑀓𑀸𑀦𑀫𑁆𑀧𑀺 𑀓𑀼𑀫𑁆𑀪𑀣𑀽𑀡𑀺𑀓𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀢𑀺—  “𑀫𑀬𑁆𑀳𑀁 𑀧𑀭𑀺𑀲𑀢𑀺 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀣𑀸”𑀢𑀺𑁇 𑀦𑀝𑀸𑀧𑀺 𑀦𑀝𑀓𑀸𑀧𑀺 𑀮𑀗𑁆𑀖𑀓𑀸𑀧𑀺 𑀲𑁄𑀓𑀚𑁆𑀛𑀸𑀬𑀺𑀓𑀸𑀧𑀺 𑀓𑀼𑀫𑁆𑀪𑀣𑀽𑀡𑀺𑀓𑀸𑀧𑀺 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀢𑀺 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀦𑁆𑀢𑀺—  “𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀸𑀡𑀺𑀓𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀝𑁆𑀝𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀸𑀢𑀼𑀁; 𑀤𑁂𑀣 𑀅𑀬𑁆𑀬𑀸𑀬, 𑀓𑀭𑁄𑀣 𑀅𑀬𑁆𑀬𑀸𑀬𑀸”𑀢𑀺𑁇

2852\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸, 𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(917.)

2853\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀬 𑀯𑀸 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁮*:**𑁮𑁩**)

(918.)

2854\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2855\. **𑀅𑀕𑀸𑀭𑀺𑀓𑁄** 𑀦𑀸𑀫 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇

2856\. **𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀲𑀸𑀫𑀡𑁂𑀭𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄𑁇

2857\. **𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀜𑁆𑀘 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀜𑁆𑀘 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑁇

2858\. **𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀓𑀧𑁆𑀧𑀓𑀢𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(919.)

2859\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀢𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑀁 𑀤𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2860\. 𑀅𑀝𑁆𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
