# 𑁨𑁇𑁪𑁇𑁩𑁇𑁭 𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(911.)

2842. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸, 𑀦 𑀢𑀸𑀯 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦 𑀢𑀸𑀯 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀻𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀼 𑀆𑀕𑀢𑀸𑀲𑀼 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑀸𑀧𑁂𑀲𑀺𑁇

2843. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(912.)

2844. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁** **𑀧𑀝𑀺𑀩𑀸𑀳𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁭*:**𑁮𑁨**)

(913.)

2845. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2846. **𑀥𑀫𑁆𑀫𑀺𑀓𑁄** 𑀦𑀸𑀫 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑁄 𑀲𑀫𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑁂𑀢𑀺𑁇

2847. **𑀧𑀝𑀺𑀩𑀸𑀳𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀓𑀣𑀁 𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀦 𑀪𑀸𑀚𑁂𑀬𑁆𑀬𑀸𑀢𑀺 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(914.)

2848. 𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(915.)

2849. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2850. 𑀲𑀢𑁆𑀢𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
