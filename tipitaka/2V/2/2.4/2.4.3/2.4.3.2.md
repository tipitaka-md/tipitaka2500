# 𑁨𑁇𑁪𑁇𑁩𑁇𑁨 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(887.)

2791\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄—  “𑀪𑀕𑀯𑀢𑀸 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸”𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑀼𑀁𑁇 𑀧𑀼𑀭𑀢𑁄𑀧𑀺 𑀧𑀘𑁆𑀙𑀢𑁄𑀧𑀺 𑀆𑀓𑀟𑁆𑀠𑀦𑁆𑀢𑀸 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀺𑁇

2792\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(888.)

2793\. **“𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀓𑀸𑀭𑀬𑀫𑀸𑀦𑀸𑀬 𑀧𑀫𑀸𑀡𑀺𑀓𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀭𑀺𑀤𑀁 𑀧𑀫𑀸𑀡𑀁—  𑀤𑀻𑀖𑀲𑁄 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄, 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑀸; 𑀢𑀺𑀭𑀺𑀬𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄𑁇 𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀙𑁂𑀤𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁨*:**𑁭𑁭**)

(889.)

2794\. **𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀬𑀸𑀬 𑀦𑀺𑀯𑀢𑁆𑀣𑀸 𑀦𑀳𑀸𑀬𑀢𑀺𑁇

2795\. **𑀓𑀸𑀭𑀬𑀫𑀸𑀦𑀸𑀬𑀸**𑀢𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑀸𑁇

2796\. 𑀧𑀫𑀸𑀡𑀺𑀓𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀭𑀺𑀤𑀁 𑀧𑀫𑀸𑀡𑀁—  𑀤𑀻𑀖𑀲𑁄 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄, 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑀸; 𑀢𑀺𑀭𑀺𑀬𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄𑁇 𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁𑁇

(890.)

2797\. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2798\. 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(891.)

2799\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀫𑀸𑀡𑀺𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀊𑀦𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀯𑀺𑀢𑀸𑀦𑀁 𑀯𑀸 𑀪𑀽𑀫𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀁 𑀯𑀸 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2800\. 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
