# 𑁨𑁇𑁪𑁇𑁩𑁇𑁬 𑀕𑀡𑀘𑀻𑀯𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀦𑀕𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀕𑀡𑀘𑀻𑀯𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(907.)

2833\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲, 𑀅𑀬𑁆𑀬𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀢𑀼𑀫𑁆𑀳𑁂 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑀸 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑀸”𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀖𑀭𑀁 𑀟𑀬𑁆𑀳𑀢𑀺𑁇 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀉𑀪𑀬𑁂𑀦𑀸𑀫𑁆𑀳 𑀧𑀭𑀺𑀩𑀸𑀳𑀺𑀭𑀸 𑀪𑁄𑀕𑁂𑀳𑀺 𑀘 𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀘𑀸”𑀢𑀺𑁇

2834\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀕𑀡𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀮𑀸𑀪𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀡𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀮𑀸𑀪𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀡𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀮𑀸𑀪𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(908.)

2835\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀡𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀮𑀸𑀪𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁬*:**𑁮𑁧**)

(909.)

2836\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2837\. **𑀕𑀡𑁄** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2838\. **𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀙𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀼𑀧𑀕𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

2839\. **𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 “𑀓𑀣𑀁 𑀇𑀫𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦 𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀯𑀸 𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(910.)

2840\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2841\. 𑀙𑀝𑁆𑀞𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
