# 𑁨𑁇𑁪𑁇𑁬𑁇𑁩 𑀕𑀡𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀕𑀡𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1033.)

3115\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀺𑀓𑀸 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀺𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀺𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀺𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀺𑀓𑀸𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀲𑁆𑀲𑀸 𑀓𑀫𑁆𑀫𑁂 𑀓𑀭𑀻𑀬𑀫𑀸𑀦𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀸𑀫𑀓𑀁 𑀅𑀕𑀫𑀸𑀲𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄—  “𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸”𑀢𑀺 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀸𑀫𑀓𑁂 𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀻𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀘𑁆𑀙𑀺𑁇 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺 𑀦 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀞𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺; 𑀦 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀦 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀫𑀬𑀺 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀯 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺 𑀦 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀞𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺; 𑀦 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀦 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀆𑀧𑀼𑀘𑁆𑀙𑀻”𑀢𑀺? “𑀏𑀯𑀜𑁆𑀳𑁂𑀢𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸 𑀢𑀁 𑀅𑀦𑀸𑀣𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀅𑀦𑀸𑀣𑀸”𑀢𑀺? “𑀇𑀫𑀸 𑀫𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄—  ‘𑀅𑀬𑀁 𑀅𑀦𑀸𑀣𑀸 𑀅𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀢𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀓𑀸𑀘𑀺 𑀧𑀢𑀺𑀯𑀢𑁆𑀢𑀸’𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁𑀲𑀽”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀩𑀸𑀮𑀸 𑀏𑀢𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀏𑀢𑀸 𑀦𑁂𑀢𑀸 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀫𑁆𑀫𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀯𑀸”𑀢𑀺, 𑀘𑀡𑁆𑀟𑀻𑀓𑀢𑀸 𑀕𑀡𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑁇

3116\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀘𑀡𑁆𑀟𑀻𑀓𑀢𑀸 𑀕𑀡𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑀡𑁆𑀟𑀻𑀓𑀢𑀸 𑀕𑀡𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑀡𑁆𑀟𑀻𑀓𑀢𑀸 𑀕𑀡𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1034.)

3117\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑀡𑁆𑀟𑀻𑀓𑀢𑀸 𑀕𑀡𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫𑁩*:**𑁧𑁦𑁮**)

(1035.)

3118\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3119\. **𑀘𑀡𑁆𑀟𑀻𑀓𑀢𑀸** 𑀦𑀸𑀫 𑀓𑁄𑀥𑀦𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

3120\. **𑀕𑀡𑁄** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

3121\. **𑀧𑀭𑀺𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 “𑀩𑀸𑀮𑀸 𑀏𑀢𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀏𑀢𑀸 𑀦𑁂𑀢𑀸 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀫𑁆𑀫𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀯𑀸”𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑀸 𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀸 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(1036.)

3122\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀢𑁆𑀣𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸𑀬, 𑀥𑀫𑁆𑀫𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸𑀬, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸𑀬, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3123\. 𑀢𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
