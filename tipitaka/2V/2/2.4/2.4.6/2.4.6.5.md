# 𑁨𑁇𑁪𑁇𑁬𑁇𑁫 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1042.)

3137\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀯𑀺𑀲𑀺𑀔𑀸𑀬 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑀸 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀓𑀼𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀪𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀜𑁆𑀜𑀸𑀧𑀺, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀑𑀓𑀸𑀲𑁂 𑀯𑀸𑀍𑀆 𑀲𑀼𑀦𑀔𑀸 𑀘𑀡𑁆𑀟𑁄 𑀩𑀮𑀺𑀩𑀤𑁆𑀤𑁄 𑀘𑀺𑀓𑁆𑀔𑀮𑁆𑀮𑁄 𑀑𑀓𑀸𑀲𑁄𑁇 𑀫𑀸 𑀔𑁄 𑀢𑀢𑁆𑀣 𑀅𑀕𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀲𑁆𑀲𑀸 𑀯𑀺𑀲𑀺𑀔𑀸𑀬 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑀸 𑀬𑁂𑀦 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀪𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲, 𑀅𑀬𑁆𑀬𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺?

3138\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀼𑀮𑀁 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀼𑀮𑀁 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀼𑀮𑀁 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1043.)

3139\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑀻 𑀅𑀲𑁆𑀲, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫𑁫*:**𑁧𑁧𑁦**)

(1044.)

3140\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3141\. **𑀓𑀼𑀮𑀁** 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀼𑀮𑀸𑀦𑀺—  𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁, 𑀯𑁂𑀲𑁆𑀲𑀓𑀼𑀮𑀁, 𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑀁𑁇

3142\. **𑀫𑀘𑁆𑀙𑀭𑀺𑀦𑀻 𑀅𑀲𑁆𑀲𑀸**𑀢𑀺 “𑀓𑀣𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀓𑀼𑀮𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀓𑀼𑀮𑀲𑁆𑀲 𑀯𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1045.)

3143\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀓𑀼𑀮𑀁 𑀦 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀦𑁆𑀢𑀻 𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀆𑀤𑀻𑀦𑀯𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3144\. 𑀧𑀜𑁆𑀘𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
