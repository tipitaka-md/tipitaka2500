# 𑁨𑁇𑁪𑁇𑁬𑁇𑁪 𑀧𑀯𑀸𑀭𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀧𑀯𑀸𑀭𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1037.)

3124\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀪𑁄𑀚𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀜𑀸𑀢𑀺𑀓𑀼𑀮𑀸𑀦𑀺 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀏𑀓𑀘𑁆𑀘𑀸 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼 𑀏𑀓𑀘𑁆𑀘𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀆𑀤𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀧𑀝𑀺𑀯𑀺𑀲𑁆𑀲𑀓𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀫𑀬𑀸, 𑀅𑀬𑁆𑀬𑀸, 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀸, 𑀏𑀣 𑀢𑀼𑀫𑁆𑀳𑁂𑀧𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁄, 𑀅𑀫𑁆𑀳𑁂 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀬𑀸𑀧𑀺 𑀢𑀬𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀢𑀸𑀧𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀖𑀭𑀸𑀦𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀏𑀓𑀘𑁆𑀘𑀸 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼 𑀏𑀓𑀘𑁆𑀘𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀆𑀤𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀖𑀭𑁂 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀸𑀳𑀁 𑀧𑀝𑀺𑀩𑀮𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

3125\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1038.)

3126\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀯𑀸 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀯𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀔𑀸𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫𑁪*:**𑁧𑁦𑁯**)

(1039.)

3127\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3128\. **𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀪𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀪𑁄𑀚𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸𑁇

3129\. **𑀧𑀯𑀸𑀭𑀺𑀢𑀸** 𑀦𑀸𑀫 𑀅𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀪𑁄𑀚𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀞𑀺𑀢𑀸 𑀅𑀪𑀺𑀳𑀭𑀢𑀺, 𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇

3130\. **𑀔𑀸𑀤𑀦𑀻𑀬𑀁** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀸𑀦𑀺—  𑀬𑀸𑀕𑀼𑀁 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀦𑀸𑀫𑁇

3131\. **𑀪𑁄𑀚𑀦𑀻𑀬𑀁** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀸𑀦𑀺—  𑀑𑀤𑀦𑁄, 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑁄, 𑀲𑀢𑁆𑀢𑀼, 𑀫𑀘𑁆𑀙𑁄, 𑀫𑀁𑀲𑀁𑁇

3132\. “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1040.)

3133\. 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁂 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁂 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

3134\. 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀆𑀳𑀸𑀭𑀢𑁆𑀣𑀸𑀬 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲…𑀧𑁂… 𑁇

(1041.)

3135\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑀸, 𑀬𑀸𑀕𑀼𑀁 𑀧𑀺𑀯𑀢𑀺, 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀬𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
