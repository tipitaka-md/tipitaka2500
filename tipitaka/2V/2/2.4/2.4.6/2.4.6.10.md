# 𑁨𑁇𑁪𑁇𑁬𑁇𑁧𑁦 𑀧𑀲𑀸𑀔𑁂𑀚𑀸𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀲𑀸𑀔𑁂𑀚𑀸𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1062.)

3173\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀲𑀸𑀔𑁂 𑀚𑀸𑀢𑀁 𑀕𑀡𑁆𑀟𑀁 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀪𑁂𑀤𑀸𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀤𑀽𑀲𑁂𑀢𑀼𑀁 𑀉𑀧𑀓𑁆𑀓𑀫𑀺𑁇 𑀲𑀸 𑀯𑀺𑀲𑁆𑀲𑀭𑀫𑀓𑀸𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀥𑀸𑀯𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀺𑀲𑁆𑀲𑀭𑀫𑀓𑀸𑀲𑀻”𑀢𑀺?

3174\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀲𑀸𑀔𑁂 𑀚𑀸𑀢𑀁 𑀕𑀡𑁆𑀟𑀁 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀪𑁂𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀲𑀸𑀔𑁂 𑀚𑀸𑀢𑀁 𑀕𑀡𑁆𑀟𑀁 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀪𑁂𑀤𑀸𑀧𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀲𑀸𑀔𑁂 𑀚𑀸𑀢𑀁 𑀕𑀡𑁆𑀟𑀁 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀪𑁂𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1063.)

3175\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀲𑀸𑀔𑁂 𑀚𑀸𑀢𑀁 𑀕𑀡𑁆𑀟𑀁 𑀯𑀸 𑀭𑀼𑀥𑀺𑀢𑀁 𑀯𑀸 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁 𑀯𑀸 𑀕𑀡𑀁 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀪𑁂𑀤𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀨𑀸𑀮𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀥𑁄𑀯𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀆𑀮𑀺𑀫𑁆𑀧𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀩𑀦𑁆𑀥𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀫𑁄𑀘𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁬𑁦*:**𑁧𑁧𑁫**)

(1064.)

3176\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3177\. **𑀧𑀲𑀸𑀔𑀁** 𑀦𑀸𑀫 𑀅𑀥𑁄𑀦𑀸𑀪𑀺 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁𑁇 **𑀚𑀸𑀢𑀦𑁆**𑀢𑀺 𑀢𑀢𑁆𑀣 𑀚𑀸𑀢𑀁𑁇 **𑀕𑀡𑁆𑀟𑁄** 𑀦𑀸𑀫 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀕𑀡𑁆𑀟𑁄𑁇 **𑀭𑀼𑀥𑀺𑀢𑀁** 𑀦𑀸𑀫 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀯𑀡𑀁𑁇 **𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑁇 **𑀲𑀁𑀖𑁄** 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀕𑀡𑁄** 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇 **𑀧𑀼𑀭𑀺𑀲𑁄** 𑀦𑀸𑀫 𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑁄, 𑀦 𑀬𑀓𑁆𑀔𑁄, 𑀦 𑀧𑁂𑀢𑁄, 𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄, 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀝𑀺𑀩𑀮𑁄 𑀤𑀽𑀲𑁂𑀢𑀼𑀁𑁇 **𑀲𑀤𑁆𑀥𑀺𑀦𑁆**𑀢𑀺 𑀏𑀓𑀢𑁄𑁇 **𑀏𑀓𑁂𑀦𑁂𑀓𑀸**𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑁄 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑁇

(1065.)

3178\. “𑀪𑀺𑀦𑁆𑀤𑀸”𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀪𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 “𑀨𑀸𑀮𑁂𑀳𑀻”𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀨𑀸𑀮𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 “𑀥𑁄𑀯𑀸”𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀥𑁄𑀯𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 “𑀆𑀮𑀺𑀫𑁆𑀧𑀸”𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀮𑀺𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 “𑀩𑀦𑁆𑀥𑀸𑀳𑀻”𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀩𑀤𑁆𑀥𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 “𑀫𑁄𑀘𑁂𑀳𑀻”𑀢𑀺 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀫𑀼𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1066.)

3179\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀪𑁂𑀤𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀨𑀸𑀮𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀆𑀮𑀺𑀫𑁆𑀧𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀫𑁄𑀘𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑀽 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀳𑁄𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3180\. 𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
