# 𑁨𑁇𑁪𑁇𑁬𑁇𑁬 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀸𑀯𑀸𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀸𑀯𑀸𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1046.)

3145\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀸𑀫𑀓𑀸𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀢𑁆𑀣𑀸𑀬𑁆𑀬𑀸𑀬𑁄 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸? 𑀓𑀘𑁆𑀘𑀺 𑀑𑀯𑀸𑀤𑁄 𑀇𑀤𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀦𑀢𑁆𑀣𑀬𑁆𑀬𑁂, 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽; 𑀓𑀼𑀢𑁄 𑀑𑀯𑀸𑀤𑁄 𑀇𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

3146\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁂 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁂 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁂 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1047.)

3147\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁂 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫𑁬*:**𑁧𑁧𑁧**)

(1048.)

3148\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3149\. **𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄** 𑀦𑀸𑀫 𑀆𑀯𑀸𑀲𑁄 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀳𑁄𑀢𑀺 𑀑𑀯𑀸𑀤𑀸𑀬 𑀯𑀸 𑀲𑀁𑀯𑀸𑀲𑀸𑀬 𑀯𑀸 𑀕𑀦𑁆𑀢𑀼𑀁𑁇 “𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀲𑀳 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1049.)

3150\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀯𑀲𑁆𑀲𑀼𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀯𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀸 𑀯𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀯𑀸 𑀧𑀓𑁆𑀔𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀸 𑀯𑀸, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3151\. 𑀙𑀝𑁆𑀞𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
