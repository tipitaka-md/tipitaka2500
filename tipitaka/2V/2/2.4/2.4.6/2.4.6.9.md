# 𑁨𑁇𑁪𑁇𑁬𑁇𑁯 𑀑𑀯𑀸𑀤𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀑𑀯𑀸𑀤𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1058.)

3167\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑀫𑁆𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀑𑀯𑀸𑀤𑀫𑁆𑀧𑀺 𑀦 𑀬𑀸𑀘𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑀫𑁆𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀑𑀯𑀸𑀤𑀫𑁆𑀧𑀺 𑀦 𑀬𑀸𑀘𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑀫𑁆𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀑𑀯𑀸𑀤𑀫𑁆𑀧𑀺 𑀦 𑀬𑀸𑀘𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑀫𑁆𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀑𑀯𑀸𑀤𑀫𑁆𑀧𑀺 𑀦 𑀬𑀸𑀘𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1059.)

3168\. **“𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀢𑁄 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀘𑁆𑀘𑀸𑀲𑀻𑀲𑀺𑀢𑀩𑁆𑀩𑀸—  𑀉𑀧𑁄𑀲𑀣𑀧𑀼𑀘𑁆𑀙𑀓𑀜𑁆𑀘 𑀑𑀯𑀸𑀤𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑀜𑁆𑀘𑁇 𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁫𑁯*:**𑁧𑁧𑁪**)

(1060.)

3169\. **𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀦𑁆**𑀢𑀺 𑀅𑀦𑀼𑀧𑁄𑀲𑀣𑀺𑀓𑀁𑁇 **𑀉𑀧𑁄𑀲𑀣𑁄** 𑀦𑀸𑀫 𑀤𑁆𑀯𑁂 𑀉𑀧𑁄𑀲𑀣𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀓𑁄 𑀘 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀓𑁄 𑀘𑁇

3170\. **𑀑𑀯𑀸𑀤𑁄** 𑀦𑀸𑀫 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀸𑁇 “𑀉𑀧𑁄𑀲𑀣𑀫𑁆𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀑𑀯𑀸𑀤𑀫𑁆𑀧𑀺 𑀦 𑀬𑀸𑀘𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀥𑀼𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1061.)

3171\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦 𑀮𑀪𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3172\. 𑀦𑀯𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
