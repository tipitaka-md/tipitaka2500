# 𑁨𑁇𑁪𑁇𑁯𑁇𑁧 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1178.)

3425. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑀺𑀬𑁄”𑀢𑀺𑁇

3426. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

3427. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇

3428. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(1179.)

3429. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀯𑀺𑀦𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1180.)

3430. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀕𑀺𑀮𑀸𑀦𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁮𑁪*:**𑁧𑁩𑁯**)

(1181.)

3431. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3432. **𑀅𑀕𑀺𑀮𑀸𑀦𑀸** 𑀦𑀸𑀫 𑀬𑀲𑁆𑀲𑀸 𑀯𑀺𑀦𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺𑁇

3433. **𑀕𑀺𑀮𑀸𑀦𑀸** 𑀦𑀸𑀫 𑀬𑀲𑁆𑀲𑀸 𑀯𑀺𑀦𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺𑁇

3434. **𑀙𑀢𑁆𑀢𑀁** 𑀦𑀸𑀫 𑀢𑀻𑀡𑀺 𑀙𑀢𑁆𑀢𑀸𑀦𑀺—  𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁, 𑀓𑀺𑀮𑀜𑁆𑀚𑀘𑁆𑀙𑀢𑁆𑀢𑀁, 𑀧𑀡𑁆𑀡𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀫𑀡𑁆𑀟𑀮𑀩𑀤𑁆𑀥𑀁 𑀲𑀮𑀸𑀓𑀩𑀤𑁆𑀥𑀁𑁇 **𑀥𑀸𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀓𑀺𑀫𑁆𑀧𑀺 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1182.)

3435. 𑀅𑀕𑀺𑀮𑀸𑀦𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑀜𑁆𑀜𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀕𑀺𑀮𑀸𑀦𑀸 𑀕𑀺𑀮𑀸𑀦𑀲𑀜𑁆𑀜𑀸 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

3436. 𑀙𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑁂𑀢𑀺 𑀦 𑀉𑀧𑀸𑀳𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀉𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀺 𑀦 𑀙𑀢𑁆𑀢𑀁, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀕𑀺𑀮𑀸𑀦𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀕𑀺𑀮𑀸𑀦𑀸 𑀕𑀺𑀮𑀸𑀦𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(1183.)

3437. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀭𑀸𑀫𑁂 𑀆𑀭𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑁂 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3438. 𑀧𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
