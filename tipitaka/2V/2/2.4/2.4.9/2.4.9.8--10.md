# 𑁨𑁇𑁪𑁇𑁯𑁇𑁮--𑁧𑁦 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦-𑀲𑀸𑀫𑀡𑁂𑀭𑀻-𑀕𑀺𑀳𑀺𑀦𑀺𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

* 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦-𑀲𑀸𑀫𑀡𑁂𑀭𑀻-𑀕𑀺𑀳𑀺𑀦𑀺𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1210.)

3490\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺…𑀧𑁂…  𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀸𑀭𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑀺𑀬𑁄”𑀢𑀺𑁇

3491\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1211.)

3492\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 (𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸…𑀧𑁂… ) 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (𑁯𑁧--*𑁯𑁩*:**𑁧𑁪𑁬--𑁧𑁪𑁮**)

(1212.)

3493\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3494\. **𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸** 𑀦𑀸𑀫 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑁇

3495\. **𑀲𑀸𑀫𑀡𑁂𑀭𑀻** 𑀦𑀸𑀫 𑀤𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀺𑀓𑀸𑁇

3496\. **𑀕𑀺𑀳𑀺𑀦𑀻** 𑀦𑀸𑀫 𑀅𑀕𑀸𑀭𑀺𑀦𑀻 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

3497\. **𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

3498\. **𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀲𑀫𑁆𑀩𑀸𑀳𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1213.)

3499\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3500\. 𑀅𑀝𑁆𑀞𑀫-𑀦𑀯𑀫-𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑀦𑀺𑁇
