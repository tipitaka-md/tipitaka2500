# 𑁨𑁇𑁪𑁇𑁯𑁇𑁧𑁧 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1214.)

3501\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1215.)

3502\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁯𑁪*:**𑁧𑁪𑁯**)

(1216.)

3503\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3504\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄𑁇 **𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸**𑀢𑀺 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸𑁇 **𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀙𑀫𑀸𑀬𑀧𑀺 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1217.)

3505\. 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

3506\. 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(1218.)

3507\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3508\. 𑀏𑀓𑀸𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
