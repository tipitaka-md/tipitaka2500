# 𑁨𑁇𑁪𑁇𑁯𑁇𑁧𑁩 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1224.)

3517\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀸 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀭𑀣𑀺𑀓𑀸𑀬 𑀯𑀸𑀢𑀫𑀡𑁆𑀟𑀮𑀺𑀓𑀸 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀬𑁄 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁𑀲𑀼𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑀺𑀁 𑀅𑀓𑀁𑀲𑀼—  “𑀲𑀼𑀦𑁆𑀤𑀭𑀸 𑀅𑀬𑁆𑀬𑀸𑀬 𑀣𑀦𑀼𑀤𑀭𑀸”𑀢𑀺𑁇 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀺𑀬𑀫𑀸𑀦𑀸 𑀫𑀗𑁆𑀓𑀼 𑀅𑀳𑁄𑀲𑀺𑁇

3518\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀸 𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀸 𑀕𑀸𑀫𑀁 𑀧𑀸𑀯𑀺𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀸 𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1225.)

3519\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀸 𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁯𑁬*:**𑁧𑁫𑁧**)

(1226.)

3520\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3521\. **𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀸**𑀢𑀺 𑀯𑀺𑀦𑀸 𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀁𑁇

3522\. **𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀁** 𑀦𑀸𑀫 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀦𑀸𑀪𑀺, 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀢𑁆𑀣𑀸𑀬𑁇

3523\. **𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀕𑀸𑀫𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀧𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀕𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀘𑀸𑀭𑀁 𑀑𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1227.)

3524\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀘𑀻𑀯𑀭𑀺𑀓𑀸𑀬, 𑀦𑀝𑁆𑀞𑀘𑀻𑀯𑀭𑀺𑀓𑀸𑀬, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑀬𑀸, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3525\. 𑀢𑁂𑀭𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕𑁄 𑀦𑀯𑀫𑁄𑁇
