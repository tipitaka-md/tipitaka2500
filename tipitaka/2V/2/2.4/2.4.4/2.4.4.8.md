# 𑁨𑁇𑁪𑁇𑁪𑁇𑁮 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(965.)

2963\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀥𑀼𑀢𑁆𑀢𑀸 𑀤𑀽𑀲𑁂𑀦𑁆𑀢𑀺𑁇

2964\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(966.)

2965\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩𑁮*:**𑁯𑁩**)

(967.)

2966\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2967\. **𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑁂**𑀢𑀺 𑀬𑀲𑁆𑀲 𑀯𑀺𑀚𑀺𑀢𑁂 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀭𑀝𑁆𑀞𑁂𑁇

2968\. **𑀲𑀸𑀲𑀗𑁆𑀓𑀁** 𑀦𑀸𑀫 𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑀕𑁆𑀕𑁂 𑀘𑁄𑀭𑀸𑀦𑀁 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀼𑀢𑁆𑀢𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀞𑀺𑀢𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀧𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇

2969\. **𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀁** 𑀦𑀸𑀫 𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑀕𑁆𑀕𑁂 𑀘𑁄𑀭𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀳𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀆𑀓𑁄𑀝𑀺𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

2970\. **𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀯𑀺𑀦𑀸 𑀲𑀢𑁆𑀣𑁂𑀦𑁇

2971\. **𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀫𑁆𑀧𑀸𑀢𑁂 𑀕𑀸𑀫𑁂 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀕𑀸𑀫𑀓𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑁂 𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(968.)

2972\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀢𑁆𑀣𑁂𑀦 𑀲𑀳 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀔𑁂𑀫𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2973\. 𑀅𑀝𑁆𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
