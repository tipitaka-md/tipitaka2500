# 𑁨𑁇𑁪𑁇𑁪𑁇𑁯 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

### 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(969.)

2974. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑀭𑀺𑀢𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀘 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀦𑁆𑀢𑀸, 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀸, 𑀩𑀳𑀽 𑀔𑀼𑀤𑁆𑀤𑀓𑁂 𑀧𑀸𑀡𑁂 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀸”𑀢𑀺𑁇

2975. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(970.)

2976. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩𑁯*:**𑁯𑁪**)

(971.)

2977. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2978. **𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀦𑁆**𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀅𑀯𑀲𑀺𑀢𑁆𑀯𑀸𑁇

2979. **𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀫𑁆𑀧𑀸𑀢𑁂 𑀕𑀸𑀫𑁂 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀕𑀸𑀫𑀓𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑁂 𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(972.)

2980. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀓𑁂𑀦𑀘𑀺 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2981. 𑀦𑀯𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
