# 𑁨𑁇𑁪𑁇𑁪𑁇𑁨 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀢𑀼𑀯𑀝𑁆𑀝𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀢𑀼𑀯𑀝𑁆𑀝𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(936.)

2892\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀸𑀭𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑀺𑀬𑁄”𑀢𑀺𑁇

2893\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(937.)

2894\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀬𑁆𑀬𑀼𑀁, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩𑁨*:**𑁮𑁭**)

(938.)

2895\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬𑁄 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

2896\. **𑀤𑁆𑀯𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀬𑁆𑀬𑀼𑀦𑁆**𑀢𑀺 𑀢𑀜𑁆𑀜𑁂𑀯 𑀅𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀢𑀜𑁆𑀜𑁂𑀯 𑀧𑀸𑀭𑀼𑀧𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(939.)

2897\. 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀲𑀜𑁆𑀜𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑁂 𑀦𑀸𑀦𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀲𑀜𑁆𑀜𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2898\. 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑁂 𑀦𑀸𑀦𑀸𑀧𑀸𑀯𑀼𑀭𑀡𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀸𑀦𑀢𑁆𑀣𑀭𑀡𑁂 𑀏𑀓𑀧𑀸𑀯𑀼𑀭𑀡𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀸𑀦𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑁂 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀸𑀦𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀸𑀦𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑁂 𑀦𑀸𑀦𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(940.)

2899\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀁, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀦𑁆𑀢𑀺𑁇

---

2900\. 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
