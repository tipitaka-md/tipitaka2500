# 𑁨𑁇𑁪𑁇𑁪𑁇𑁧𑁦 𑀘𑀸𑀭𑀺𑀓𑀦𑀧𑀓𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

* 𑀘𑀸𑀭𑀺𑀓𑀦𑀧𑀓𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(973.)

2982\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀢𑁆𑀣𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀳𑁂𑀫𑀦𑁆𑀢𑀁, 𑀢𑀢𑁆𑀣 𑀕𑀺𑀫𑁆𑀳𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀆𑀳𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀤𑀺𑀲𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀸, 𑀦 𑀇𑀫𑀸𑀲𑀁 𑀤𑀺𑀲𑀸 𑀧𑀓𑁆𑀔𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

2983\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬…𑀧𑁂…  𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(974.)

2984\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀦 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀙𑀧𑁆𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑀧𑀺, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁪𑁦*:**𑁯𑁫**)

(975.)

2985\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2986\. **𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸** 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑁇

2987\. “𑀘𑀸𑀭𑀺𑀓𑀁 𑀦 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀙𑀧𑁆𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀸𑀦𑀺𑀧𑀻”𑀢𑀺 𑀥𑀼𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(976.)

2988\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦 𑀮𑀪𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2989\. 𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
