# 𑁨𑁇𑁪𑁇𑁪𑁇𑁪 𑀦𑁅𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

* 𑀦𑁅𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(946.)

2912\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺𑁇

2913\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(947.)

2914\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩𑁪*:**𑁮𑁯**)

(948.)

2915\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2916\. **𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀸** 𑀦𑀸𑀫 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2917\. **𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀻** 𑀦𑀸𑀫 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑀻 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2918\. **𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀲𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑁇

2919\. **𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀬𑁆𑀬𑁇

2920\. “𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀥𑀼𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀺𑀁 𑀯𑀸 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀸 𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀢𑀺, 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(949.)

2921\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀦 𑀮𑀪𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2922\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
