# 𑁨𑁇𑁪𑁇𑁪𑁇𑁫 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(950.)

2923\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀸 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀻 𑀲𑀸𑀓𑁂𑀢𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀲𑀘𑁂 𑀫𑁂 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀤𑁂𑀬𑁆𑀬 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫𑀳𑀁 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑁆”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀯𑀫𑀸𑀳—  “𑀆𑀕𑀘𑁆𑀙𑀢𑀼, 𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀤𑁆𑀤𑀸 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀻 𑀲𑀸𑀓𑁂𑀢𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀤𑁆𑀤𑀸𑀬 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀬𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀢𑀺 𑀪𑀸𑀡𑀺𑀓𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀝𑁆𑀝𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀪𑀤𑁆𑀤𑀸𑀧𑀺 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀢𑀺 𑀪𑀸𑀡𑀺𑀓𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀝𑁆𑀝𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀸𑀢𑀼𑀁 𑀉𑀍𑀆𑀭𑀲𑀫𑁆𑀪𑀸𑀯𑀺𑀢𑀸𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  “𑀅𑀬𑁆𑀬𑀸 𑀪𑀤𑁆𑀤𑀸 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀻 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀸𑀡𑀺𑀓𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀝𑁆𑀝𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀸𑀢𑀼𑀁 𑀉𑀍𑀆𑀭𑀲𑀫𑁆𑀪𑀸𑀯𑀺𑀢𑀸”𑀢𑀺 𑀪𑀤𑁆𑀤𑀁 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀁 𑀧𑀞𑀫𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀲𑁆𑀲𑀸𑀧𑀓𑀢𑀸—  “𑀇𑀫𑀸 𑀓𑀺𑀭 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀬𑀸 𑀇𑀫𑀸 𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀻”𑀢𑀺 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀪𑀤𑁆𑀤𑀁 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀁 𑀉𑀧𑀲𑁆𑀲𑀬𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑁇

2924\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀅𑀬𑁆𑀬𑀸𑀬 𑀪𑀤𑁆𑀤𑀸𑀬 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀬𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀤𑁆𑀤𑀸𑀬 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀬𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀤𑁆𑀤𑀸𑀬 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀬𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(951.)

2925\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁩𑁫*:**𑁯𑁦**)

(952.)

2926\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2927\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸𑁇 **𑀉𑀧𑀲𑁆𑀲𑀬𑁄** 𑀦𑀸𑀫 𑀓𑀯𑀸𑀝𑀩𑀤𑁆𑀥𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀤𑀢𑁆𑀯𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸𑁇 **𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸**𑀢𑀺 𑀅𑀦𑀪𑀺𑀭𑀤𑁆𑀥𑀸 𑀆𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀔𑀺𑀮𑀚𑀸𑀢𑀸𑁇

2928\. **𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀕𑀩𑁆𑀪𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀫𑀼𑀔𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀫𑀼𑀔𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀏𑀓𑁂𑀦 𑀧𑀬𑁄𑀕𑁂𑀦 𑀩𑀳𑀼𑀓𑁂𑀧𑀺 𑀤𑁆𑀯𑀸𑀭𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2929\. **𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀲𑀓𑀺𑀁 𑀆𑀡𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀓𑁂𑀧𑀺 𑀤𑁆𑀯𑀸𑀭𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(953.)

2930\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2931\. 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀓𑀯𑀸𑀝𑀩𑀤𑁆𑀥𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀓𑀯𑀸𑀝𑀩𑀤𑁆𑀥𑀸 𑀯𑀸 𑀅𑀓𑀯𑀸𑀝𑀩𑀤𑁆𑀥𑀸 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(954.)

2932\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑀺𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀺𑀓𑀁 𑀓𑀮𑀳𑀓𑀸𑀭𑀺𑀓𑀁 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀺𑀓𑀁 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀺𑀓𑀁 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀺𑀓𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀺𑀁 𑀯𑀸 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑀺𑀁 𑀯𑀸 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺 𑀯𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
