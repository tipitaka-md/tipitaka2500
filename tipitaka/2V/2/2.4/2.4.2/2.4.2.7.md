# 𑁨𑁇𑁪𑁇𑁨𑁇𑁭 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀲𑀦𑁆𑀣𑀭𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀲𑀦𑁆𑀣𑀭𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(864.)

2747\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀬𑁄 𑀲𑀸𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀸𑀫𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀑𑀓𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀕𑀫𑁂𑀣, 𑀅𑀬𑁆𑀬𑁂, 𑀬𑀸𑀯 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀆𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄—  “𑀬𑀸𑀯 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀆𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀘𑁆𑀘𑀸 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼 𑀏𑀓𑀘𑁆𑀘𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀭𑀢𑁆𑀢𑀺𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀸 𑀇𑀫𑀸”𑀢𑀺? “𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄, 𑀅𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀣 𑀇𑀫𑀸 𑀫𑀼𑀡𑁆𑀟𑀸 𑀩𑀦𑁆𑀥𑀓𑀺𑀦𑀺𑀬𑁄”𑀢𑀺, 𑀖𑀭𑀢𑁄 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀸𑀧𑁂𑀲𑀺𑁇

2748\. 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑀺𑀓𑀸𑀮𑁂 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑀺𑀓𑀸𑀮𑁂 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀧𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑀺𑀓𑀸𑀮𑁂 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(865.)

2749\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑀺𑀓𑀸𑀮𑁂 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁭*:**𑁭𑁨**)

(866.)

2750\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2751\. **𑀯𑀺𑀓𑀸𑀮𑁄** 𑀦𑀸𑀫 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑁂 𑀲𑀽𑀭𑀺𑀬𑁂 𑀬𑀸𑀯 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑀸𑁇

2752\. **𑀓𑀼𑀮𑀁** 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀼𑀮𑀸𑀦𑀺—  𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁, 𑀯𑁂𑀲𑁆𑀲𑀓𑀼𑀮𑀁 𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑀁𑁇 **𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸**𑀢𑀺 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑁇

2753\. **𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸**𑀢𑀺 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀲𑀸𑀫𑀺𑀓𑁄 𑀤𑀸𑀢𑀼𑀁, 𑀢𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑁇

2754\. **𑀲𑁂𑀬𑁆𑀬𑀁** 𑀦𑀸𑀫 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑁄𑀧𑀺𑁇 **𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸𑁇 **𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑁇 **𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 **𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(867.)

2755\. 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2756\. 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(868.)

2757\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀧𑀼𑀘𑁆𑀙𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
