# 𑁨𑁇𑁪𑁇𑁨𑁇𑁯 𑀧𑀭𑀅𑀪𑀺𑀲𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀧𑀭𑀅𑀪𑀺𑀲𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(874.)

2770\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀡𑁆𑀟𑀓𑀁 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀧𑀸𑀬𑁆𑀬𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀡𑁆𑀟𑀓𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀅𑀳𑀫𑁂𑀯 𑀦𑀽𑀦 𑀘𑁄𑀭𑀻, 𑀅𑀳𑀫𑁂𑀯 𑀦𑀽𑀦 𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑀻, 𑀬𑀸 𑀅𑀬𑁆𑀬𑀸𑀬𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀡𑁆𑀟𑀓𑀁 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀢𑀸 𑀫𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀧𑀸𑀬𑁆𑀬𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀡𑁆𑀟𑀓𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀻’𑀢𑀺? 𑀲𑀘𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀡𑁆𑀟𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀫𑀺, 𑀅𑀲𑁆𑀲𑀫𑀡𑀻 𑀳𑁄𑀫𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀘𑀯𑀸𑀫𑀺, 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺; 𑀬𑀸 𑀧𑀦 𑀫𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀏𑀯𑀫𑀸𑀳 𑀲𑀸𑀧𑀺 𑀅𑀲𑁆𑀲𑀫𑀡𑀻 𑀳𑁄𑀢𑀼, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀘𑀯𑀢𑀼, 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀽”𑀢𑀺𑁇

2771\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀅𑀢𑁆𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀧𑀭𑀫𑁆𑀧𑀺 𑀦𑀺𑀭𑀬𑁂𑀦𑀧𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦𑀧𑀺 𑀅𑀪𑀺𑀲𑀧𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀧𑀭𑀫𑁆𑀧𑀺 𑀦𑀺𑀭𑀬𑁂𑀦𑀧𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦𑀧𑀺 𑀅𑀪𑀺𑀲𑀧𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀧𑀭𑀫𑁆𑀧𑀺 𑀦𑀺𑀭𑀬𑁂𑀦𑀧𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦𑀧𑀺 𑀅𑀪𑀺𑀲𑀧𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(875.)

2772\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀸 𑀧𑀭𑀁 𑀯𑀸 𑀦𑀺𑀭𑀬𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀲𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁯*:**𑁭𑁪**)

(876.)

2773\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2774\. **𑀅𑀢𑁆𑀢𑀸𑀦𑀦𑁆**𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑁇 **𑀧𑀭𑀦𑁆**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀦𑀺𑀭𑀬𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀲𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(877.)

2775\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑀬𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀲𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀦𑀺𑀭𑀬𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀲𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑀬𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀲𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2776\. 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀯𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁂𑀦 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑁄𑀪𑀕𑁆𑀕𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀲𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀪𑀺𑀲𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(878.)

2777\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀢𑁆𑀣𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸𑀬, 𑀥𑀫𑁆𑀫𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸𑀬, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸𑀬, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2778\. 𑀦𑀯𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
