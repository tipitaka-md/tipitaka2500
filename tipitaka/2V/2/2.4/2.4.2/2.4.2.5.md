# 𑁨𑁇𑁪𑁇𑁨𑁇𑁫 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀧𑀓𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀧𑀓𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(854.)

2721\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀓𑀼𑀮𑀽𑀧𑀺𑀓𑀸 𑀳𑁄𑀢𑀺 𑀦𑀺𑀘𑁆𑀘𑀪𑀢𑁆𑀢𑀺𑀓𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀤𑀸𑀲𑀻 𑀖𑀭𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀻 𑀢𑀁 𑀆𑀲𑀦𑀁 𑀪𑀸𑀚𑀦𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀬 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀆𑀲𑀦𑀁 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀳𑀁 𑀢𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀆𑀲𑀦𑀦𑁆”𑀢𑀺? “𑀦𑀸𑀳𑀁 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀲𑀦𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀤𑁂𑀣𑀸𑀬𑁆𑀬𑁂, 𑀢𑀁 𑀆𑀲𑀦𑀦𑁆”𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀘𑁆𑀘𑀪𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀖𑀭𑀁 𑀲𑁄𑀥𑁂𑀦𑁆𑀢𑀸 𑀢𑀁 𑀆𑀲𑀦𑀁 𑀪𑀸𑀚𑀦𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀬 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀔𑀫𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀘𑁆𑀘𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑀼𑀁𑁇

2722\. 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(855.)

2723\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁫*:**𑁭𑁦**)

(856.)

2724\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2725\. **𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁** 𑀦𑀸𑀫 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀬𑀸𑀯 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀸𑁇

2726\. **𑀓𑀼𑀮𑀁** 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀼𑀮𑀸𑀦𑀺—  𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁, 𑀯𑁂𑀲𑁆𑀲𑀓𑀼𑀮𑀁, 𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑀁𑁇 **𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸**𑀢𑀺 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑁇

2727\. **𑀆𑀲𑀦𑀁** 𑀦𑀸𑀫 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑁆𑀲 𑀑𑀓𑀸𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸**𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸𑁇

2728\. **𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀽 𑀢𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀓𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀉𑀧𑀘𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(857.)

2729\. 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2730\. 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑁆𑀲 𑀅𑀦𑁄𑀓𑀸𑀲𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(858.)
