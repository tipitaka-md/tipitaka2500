# 𑁨𑁇𑁪𑁇𑁨𑁇𑁬 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(859.)

2733\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺𑀧𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀳𑀺𑀭𑀻𑀬𑀫𑀸𑀦𑀸 𑀆𑀲𑀦𑁂 𑀦𑁂𑀯 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺 𑀦 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀻”𑀢𑀺𑁇

2734\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺𑀧𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(860.)

2735\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁬*:**𑁭𑁧**)

(861.)

2736\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2737\. **𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁** 𑀦𑀸𑀫 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁂 𑀬𑀸𑀯 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑁂 𑀲𑀽𑀭𑀺𑀬𑁂𑁇

2738\. **𑀓𑀼𑀮𑀁** 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀼𑀮𑀸𑀦𑀺—  𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁, 𑀯𑁂𑀲𑁆𑀲𑀓𑀼𑀮𑀁, 𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑀁𑁇 **𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸**𑀢𑀺 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑁇

2739\. **𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸**𑀢𑀺 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀲𑀸𑀫𑀺𑀓𑁄 𑀤𑀸𑀢𑀼𑀁, 𑀢𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑁇

2740\. **𑀆𑀲𑀦𑀁** 𑀦𑀸𑀫 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑁆𑀲 𑀑𑀓𑀸𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2741\. **𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2742\. **𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(862.)

2743\. 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀆𑀲𑀦𑁂 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2744\. 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑁆𑀲 𑀅𑀦𑁄𑀓𑀸𑀲𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(863.)
