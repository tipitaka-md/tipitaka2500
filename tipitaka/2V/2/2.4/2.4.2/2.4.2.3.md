# 𑁨𑁇𑁪𑁇𑁨𑁇𑁩 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑀲𑀮𑁆𑀮𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑀲𑀮𑁆𑀮𑀧𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(846.)

2695\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀸𑀬 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀜𑀸𑀢𑀓𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀕𑀸𑀫𑀓𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀕𑀫𑀸𑀲𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀑𑀓𑀸𑀲𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀼𑀁 𑀲𑀮𑁆𑀮𑀧𑀺𑀢𑀼𑀦𑁆”𑀢𑀺 𑀢𑁂𑀦𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑀧𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺𑀧𑀺𑁇

2696\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀺 𑀲𑀮𑁆𑀮𑀧𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑀧𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺𑀧𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀺 𑀲𑀮𑁆𑀮𑀧𑀺𑀲𑁆𑀲𑀢𑀺𑀧𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(847.)

2697\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀯𑀸 𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁧𑁩*:**𑁬𑁮**)

(848.)

2698\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2699\. **𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁄** 𑀦𑀸𑀫 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀓𑀼𑀝𑁆𑀝𑁂𑀦 𑀯𑀸 𑀓𑀯𑀸𑀝𑁂𑀦 𑀯𑀸 𑀓𑀺𑀮𑀜𑁆𑀚𑁂𑀦 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑁂𑀦 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑁂𑀦 𑀯𑀸 𑀣𑀫𑁆𑀪𑁂𑀦 𑀯𑀸 𑀓𑁄𑀢𑁆𑀣𑀍𑀇𑀬𑀸 𑀯𑀸, 𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

2700\. **𑀧𑀼𑀭𑀺𑀲𑁄** 𑀦𑀸𑀫 𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀼𑀭𑀺𑀲𑁄, 𑀦 𑀬𑀓𑁆𑀔𑁄 𑀦 𑀧𑁂𑀢𑁄 𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄, 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀝𑀺𑀩𑀮𑁄 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀼𑀁 𑀲𑀮𑁆𑀮𑀧𑀺𑀢𑀼𑀁𑁇

2701\. **𑀲𑀤𑁆𑀥𑀺𑀦𑁆**𑀢𑀺 𑀏𑀓𑀢𑁄𑁇 **𑀏𑀓𑁂𑀦𑁂𑀓𑀸**𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑁄 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑁇

2702\. **𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2703\. **𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀞𑀺𑀢𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

2704\. 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀬𑀓𑁆𑀔𑁂𑀦 𑀯𑀸 𑀧𑁂𑀢𑁂𑀦 𑀯𑀸 𑀧𑀡𑁆𑀟𑀓𑁂𑀦 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑁂𑀦 𑀯𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(849.)

2705\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀬𑁄 𑀓𑁄𑀘𑀺 𑀯𑀺𑀜𑁆𑀜𑀽 𑀤𑀼𑀢𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀭𑀳𑁄𑀧𑁂𑀓𑁆𑀔𑀸, 𑀅𑀜𑁆𑀜𑀯𑀺𑀳𑀺𑀢𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀸 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀯𑀸, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2706\. 𑀢𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
