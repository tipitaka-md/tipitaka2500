# 𑁨𑁇𑁪𑁇𑁨𑁇𑁧𑁦 𑀭𑁄𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀭𑁄𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(879.)

2779. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀡𑁆𑀟𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀢𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(880.)

2780. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁨𑁦*:**𑁭𑁫**)

(881.)

2781. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2782. **𑀅𑀢𑁆𑀢𑀸𑀦𑀦𑁆**𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑁇 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀯𑀥𑀢𑀺 𑀦 𑀭𑁄𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀭𑁄𑀤𑀢𑀺 𑀦 𑀯𑀥𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(882.)

2783. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑁂𑀦 𑀯𑀸 𑀨𑀼𑀝𑁆𑀞𑀸 𑀭𑁄𑀤𑀢𑀺 𑀦 𑀯𑀥𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

2784. 𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
