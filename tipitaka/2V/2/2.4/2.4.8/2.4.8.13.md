# 𑁨𑁇𑁪𑁇𑁮𑁇𑁧𑁩 𑀏𑀓𑀯𑀲𑁆𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀯𑀲𑁆𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1174.)

3417\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑀲𑁆𑀲𑀬𑁄 𑀢𑀣𑁂𑀯 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀣𑁂𑀯 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑀲𑁆𑀲𑀬𑁄 𑀢𑀣𑁂𑀯 𑀦 𑀲𑀫𑁆𑀫𑀢𑀻”𑀢𑀺𑁇

3418\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1175.)

3419\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁮𑁩*:**𑁧𑁩𑁮**)

(1176.)

3420\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3421\. **𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀦𑁆**𑀢𑀺 𑀏𑀓𑀁 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑀁𑁇 **𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀤𑁆𑀯𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬𑁇

3422\. “𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀡𑀁 𑀯𑀸 𑀆𑀘𑀭𑀺𑀦𑀺𑀁 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀲𑀻𑀫𑀁 𑀯𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑁇 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀕𑀡𑀲𑁆𑀲 𑀘 𑀆𑀘𑀭𑀺𑀦𑀺𑀬𑀸 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(1177.)

3423\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀏𑀓𑀦𑁆𑀢𑀭𑀺𑀓𑀁 𑀏𑀓𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3424\. 𑀢𑁂𑀭𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
