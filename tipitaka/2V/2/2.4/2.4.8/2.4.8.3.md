# 𑁨𑁇𑁪𑁇𑁮𑁇𑁩 𑀢𑀢𑀺𑀬𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

### 𑀢𑀢𑀺𑀬𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1130.)

3316. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀏𑀣, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸, 𑀇𑀫𑀁 𑀚𑀸𑀦𑀸𑀣, 𑀇𑀫𑀁 𑀤𑁂𑀣, 𑀇𑀫𑀁 𑀆𑀳𑀭𑀣, 𑀇𑀫𑀺𑀦𑀸 𑀅𑀢𑁆𑀣𑁄, 𑀇𑀫𑀁 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀓𑀭𑁄𑀣𑀸”𑀢𑀺𑁇 𑀢𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇

3317. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀅𑀬𑁆𑀬𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸 𑀲𑀁𑀖𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀬𑀸𑀘𑀸𑀫𑀻’𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀩𑀮𑀸𑀬 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(1131.)

3318. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀅𑀬𑁆𑀬𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸 𑀲𑀁𑀖𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

3319. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀅𑀬𑁆𑀬𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸 𑀲𑀁𑀖𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸 𑀅𑀬𑁆𑀬𑀸𑀬 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸 𑀤𑀸𑀦𑀁, 𑀲𑀸 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲𑀸 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑀸 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

3320. 𑀤𑀺𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸𑀬 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

3321. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀩𑁆𑀪𑀭𑀢𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1132.)

3322. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁩*:**𑁧𑁨𑁮**)

(1133.)

3323. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3324. **𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸** 𑀦𑀸𑀫 𑀧𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑁇 **𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀻**𑀢𑀺 𑀤𑁆𑀯𑁂 𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑀸𑀦𑀺𑁇 **𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸** 𑀦𑀸𑀫 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑁇 **𑀅𑀲𑀫𑁆𑀫𑀢𑀸** 𑀦𑀸𑀫 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀦 𑀤𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 **𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬𑁇

3325. “𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀡𑀁 𑀯𑀸 𑀆𑀘𑀭𑀺𑀦𑀺𑀁 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀲𑀻𑀫𑀁 𑀯𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑁇 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀕𑀡𑀲𑁆𑀲 𑀘 𑀆𑀘𑀭𑀺𑀦𑀺𑀬𑀸 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(1134.)

3326. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

3327. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(1135.)

3328. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3329. 𑀢𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
