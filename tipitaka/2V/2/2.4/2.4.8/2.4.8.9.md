# 𑁨𑁇𑁪𑁇𑁮𑁇𑁯 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1158.)

3384. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺𑁇

3385. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1159.)

3386. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁯*:**𑁧𑁩𑁪**)

(1160.)

3387. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3388. **𑀧𑀼𑀭𑀺𑀲𑁄** 𑀦𑀸𑀫 𑀧𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄𑁇 **𑀓𑀼𑀫𑀸𑀭𑀓𑁄** 𑀦𑀸𑀫 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄𑁇 **𑀲𑀁𑀲𑀝𑁆𑀞𑀸** 𑀦𑀸𑀫 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀦 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 **𑀘𑀡𑁆𑀟𑀻** 𑀦𑀸𑀫 𑀓𑁄𑀥𑀦𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

3389. **𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀸** 𑀦𑀸𑀫 𑀧𑀭𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀲𑁄𑀓𑀁 𑀆𑀯𑀺𑀲𑀢𑀺𑁇 **𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸** 𑀦𑀸𑀫 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑁇 **𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬𑁇

3390. “𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀡𑀁 𑀯𑀸 𑀆𑀘𑀭𑀺𑀦𑀺𑀁 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀲𑀻𑀫𑀁 𑀯𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑁇 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀕𑀡𑀲𑁆𑀲 𑀘 𑀆𑀘𑀭𑀺𑀦𑀺𑀬𑀸 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(1161.)

3391. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀻 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---
