# 𑁨𑁇𑁪𑁇𑁮𑁇𑁧 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1119.)

3292\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀸 𑀅𑀓𑁆𑀔𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀻𑀢𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑀲𑁆𑀲 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀬 𑀧𑀺𑀧𑀸𑀲𑀸𑀬 𑀟𑀁𑀲𑀫𑀓𑀲𑀯𑀸𑀢𑀸𑀢𑀧𑀲𑀭𑀻𑀲𑀧𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀦𑀁 𑀤𑀼𑀭𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀼𑀭𑀸𑀕𑀢𑀸𑀦𑀁 𑀯𑀘𑀦𑀧𑀣𑀸𑀦𑀁𑁇 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀁 𑀢𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀔𑀭𑀸𑀦𑀁 𑀓𑀝𑀼𑀓𑀸𑀦𑀁 𑀅𑀲𑀸𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀸𑀧𑀸𑀦𑀁 𑀧𑀸𑀡𑀳𑀭𑀸𑀦𑀁 𑀅𑀦𑀥𑀺𑀯𑀸𑀲𑀓𑀚𑀸𑀢𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

3293\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸 𑀅𑀓𑁆𑀔𑀫𑀸 𑀳𑁄𑀢𑀺 𑀲𑀻𑀢𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀸𑀡𑀳𑀭𑀸𑀦𑀁 𑀅𑀦𑀥𑀺𑀯𑀸𑀲𑀓𑀚𑀸𑀢𑀺𑀓𑀸 𑀳𑁄𑀢𑀺𑁇 𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸 𑀔𑀫𑀸 𑀳𑁄𑀢𑀺 𑀲𑀻𑀢𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀸𑀡𑀳𑀭𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀸𑀲𑀓𑀚𑀸𑀢𑀺𑀓𑀸 𑀳𑁄𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1120.)

3294\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁** **𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁧*:**𑁧𑁨𑁬**)

(1121.)

3295\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3296\. **𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸** 𑀦𑀸𑀫 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑁇 **𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 **𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬𑁇

3297\. “𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀡𑀁 𑀯𑀸 𑀆𑀘𑀭𑀺𑀦𑀺𑀁 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺, 𑀲𑀻𑀫𑀁 𑀯𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑁇 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀕𑀡𑀲𑁆𑀲 𑀘 𑀆𑀘𑀭𑀺𑀦𑀺𑀬𑀸 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(1122.)

3298\. 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀸𑀬 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(1123.)

3299\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑀜𑁆𑀜𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3300\. 𑀧𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
