# 𑁨𑁇𑁪𑁇𑁮𑁇𑁬 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

### 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1146.)

3354\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀧𑀭𑀺𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸—  “𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸”𑀢𑀺, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀦 𑀅𑀤𑀸𑀲𑀺𑁇 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 “𑀲𑀸𑀥𑀽”𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑁄 𑀅𑀜𑁆𑀜𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑁂𑀢𑀺𑁇 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀅𑀳𑀫𑁂𑀯 𑀦𑀽𑀦 𑀩𑀸𑀮𑀸, 𑀅𑀳𑀫𑁂𑀯 𑀦𑀽𑀦 𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑀻; 𑀬𑀁 𑀲𑀁𑀖𑁄 𑀅𑀜𑁆𑀜𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑁂𑀢𑀺, 𑀫𑀬𑁆𑀳𑀫𑁂𑀯 𑀦 𑀤𑁂𑀢𑀻”𑀢𑀺𑁇

3355\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈—  ‘𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 ‘𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  ‘𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1147.)

3356\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻—  ‘𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁬*:**𑁧𑁩𑁧**)

(1148.)

3357\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3358\. **𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸**𑀢𑀺 𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑀺𑀢𑁂𑀦𑁇 “𑀲𑀸𑀥𑀽”𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1149.)

3359\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀓𑀢𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀫𑁄𑀳𑀸 𑀪𑀬𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---
