# 𑁨𑁇𑁪𑁇𑁫𑁇𑁮 𑀆𑀯𑀲𑀣𑀯𑀺𑀳𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀆𑀯𑀲𑀣𑀯𑀺𑀳𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(1008.)

3061\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀯𑀲𑀣𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀆𑀯𑀲𑀣𑁄 𑀟𑀬𑁆𑀳𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀳𑀦𑁆𑀤𑀸𑀬𑁆𑀬𑁂, 𑀪𑀡𑁆𑀟𑀓𑀁 𑀦𑀻𑀳𑀭𑀸𑀫𑀸”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀦𑀻𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀦𑀝𑁆𑀞𑀁 𑀲𑀩𑁆𑀩𑀁 𑀅𑀫𑁆𑀳𑁂 𑀅𑀪𑀺𑀬𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀦𑀤𑁂𑀯 𑀢𑀁 𑀆𑀯𑀲𑀣𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀅𑀧𑀸𑀬𑁆𑀬𑁂, 𑀪𑀡𑁆𑀟𑀓𑀁 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? “𑀦 𑀫𑀬𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀦𑀻𑀳𑀭𑀺𑀫𑁆𑀳𑀸”𑀢𑀺𑁇 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀯𑀲𑀣𑁂 𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁂 𑀪𑀡𑁆𑀟𑀓𑀁 𑀦 𑀦𑀻𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

3062\. 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀆𑀯𑀲𑀣𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀯𑀲𑀣𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀯𑀲𑀣𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(1009.)

3063\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀯𑀲𑀣𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁪𑁮*:**𑁧𑁦𑁩**)

(1010.)

3064\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3065\. **𑀆𑀯𑀲𑀣𑁄** 𑀦𑀸𑀫 𑀓𑀯𑀸𑀝𑀩𑀤𑁆𑀥𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

3066\. **𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀯𑀸 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀯𑀸 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀯𑀲𑀣𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀧𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀯𑀲𑀣𑀲𑁆𑀲 𑀉𑀧𑀘𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(1011.)

3067\. 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀲𑀜𑁆𑀜𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

3068\. 𑀅𑀓𑀯𑀸𑀝𑀩𑀤𑁆𑀥𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀅𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(1012.)

3069\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂, 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀦 𑀮𑀪𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3070\. 𑀅𑀝𑁆𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
