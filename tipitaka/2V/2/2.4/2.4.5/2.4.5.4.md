# 𑁨𑁇𑁪𑁇𑁫𑁇𑁪 𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(990.)

3019. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(991.)

3020. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁪𑁪*:**𑁯𑁯**)

(992.)

3021. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3022. **𑀕𑀺𑀳𑀺𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁** 𑀦𑀸𑀫 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀬𑀸𑀕𑀼𑀁 𑀯𑀸 𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀧𑀘𑀢𑀺, 𑀲𑀸𑀝𑀓𑀁 𑀯𑀸 𑀯𑁂𑀞𑀦𑀁 𑀯𑀸 𑀥𑁄𑀯𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(993.)

3023. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀬𑀸𑀕𑀼𑀧𑀸𑀦𑁂, 𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑁂, 𑀘𑁂𑀢𑀺𑀬𑀧𑀽𑀚𑀸𑀬, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀓𑀭𑀲𑁆𑀲 𑀬𑀸𑀕𑀼𑀁 𑀯𑀸 𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀧𑀘𑀢𑀺, 𑀲𑀸𑀝𑀓𑀁 𑀯𑀸 𑀯𑁂𑀞𑀦𑀁 𑀯𑀸 𑀥𑁄𑀯𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3024. 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
