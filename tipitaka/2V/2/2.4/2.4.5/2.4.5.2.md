# 𑁨𑁇𑁪𑁇𑁫𑁇𑁨 𑀆𑀲𑀦𑁆𑀤𑀺𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀆𑀲𑀦𑁆𑀤𑀺𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(982.)

3002\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀲𑀦𑁆𑀤𑀺𑀫𑁆𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑁆𑀧𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀸𑀭𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀲𑀦𑁆𑀤𑀺𑀫𑁆𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑁆𑀧𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑀺𑀬𑁄”𑀢𑀺𑁇

3003\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀲𑀦𑁆𑀤𑀺𑀫𑁆𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑁆𑀧𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀲𑀦𑁆𑀤𑀺𑀫𑁆𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑁆𑀧𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀲𑀦𑁆𑀤𑀺𑀫𑁆𑀧𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑁆𑀧𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(983.)

3004\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀆𑀲𑀦𑁆𑀤𑀺𑀁 𑀯𑀸 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁪𑁨*:**𑁯𑁭**)

(984.)

3005\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

3006\. **𑀆𑀲𑀦𑁆𑀤𑀻** 𑀦𑀸𑀫 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

3007\. **𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁄** 𑀦𑀸𑀫 𑀆𑀳𑀭𑀺𑀫𑁂𑀳𑀺 𑀯𑀸𑀍𑀏𑀳𑀺 𑀓𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

3008\. **𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(985.)

3009\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀲𑀦𑁆𑀤𑀺𑀬𑀸 𑀧𑀸𑀤𑁂 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑁆𑀲 𑀯𑀸𑀍𑀏 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇

---

3010\. 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
