# 𑁨𑁇𑁪𑁇𑁫𑁇𑁧 𑀭𑀸𑀚𑀸𑀕𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀭𑀸𑀚𑀸𑀕𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(977.)

2990\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀉𑀬𑁆𑀬𑀸𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑁂 𑀧𑀝𑀺𑀪𑀸𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀩𑀳𑀽 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑀺𑀬𑁄”𑀢𑀺𑁇

2991\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑀸 𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀼—

(978.)

2992\. **“𑀬𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀭𑀸𑀚𑀸𑀕𑀸𑀭𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁 𑀯𑀸 𑀆𑀭𑀸𑀫𑀁 𑀯𑀸 𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀯𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁 𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁪𑁧*:**𑁯𑁬**)

(979.)

2993\. **𑀬𑀸 𑀧𑀦𑀸**𑀢𑀺 𑀬𑀸 𑀬𑀸𑀤𑀺𑀲𑀸…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀢𑀺𑁇

2994\. **𑀭𑀸𑀚𑀸𑀕𑀸𑀭𑀁** 𑀦𑀸𑀫 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀭𑀜𑁆𑀜𑁄 𑀓𑀻𑀍𑀇𑀢𑀼𑀁 𑀭𑀫𑀺𑀢𑀼𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

2995\. **𑀘𑀺𑀢𑁆𑀢𑀸𑀕𑀸𑀭𑀁** 𑀦𑀸𑀫 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀓𑀻𑀍𑀇𑀢𑀼𑀁 𑀭𑀫𑀺𑀢𑀼𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

2996\. **𑀆𑀭𑀸𑀫𑁄** 𑀦𑀸𑀫 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀓𑀻𑀍𑀇𑀢𑀼𑀁 𑀭𑀫𑀺𑀢𑀼𑀁 𑀓𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

2997\. **𑀉𑀬𑁆𑀬𑀸𑀦𑀁** 𑀦𑀸𑀫 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀓𑀻𑀍𑀇𑀢𑀼𑀁 𑀭𑀫𑀺𑀢𑀼𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

2998\. **𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻** 𑀦𑀸𑀫 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀓𑀻𑀍𑀇𑀢𑀼𑀁 𑀭𑀫𑀺𑀢𑀼𑀁 𑀓𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

(980.)

2999\. 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀤𑀲𑁆𑀲𑀦𑀽𑀧𑀘𑀸𑀭𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀏𑀓𑀫𑁂𑀓𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀤𑀲𑁆𑀲𑀦𑀽𑀧𑀘𑀸𑀭𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(981.)

3000\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀭𑀸𑀫𑁂 𑀞𑀺𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀯𑀸 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀢𑀺 𑀓𑀭𑀡𑀻𑀬𑁂 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇
