# 𑁧𑁇𑁬𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀓𑀡𑁆𑀟

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(557.)

1613\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀮𑁂𑀲𑀼 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀞𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  “𑀇𑀥 𑀲𑀽𑀧𑀁 𑀤𑁂𑀣, 𑀇𑀥 𑀑𑀤𑀦𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀘𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1614\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(558.)

1615\. **“𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀦𑁂𑀯 𑀓𑀼𑀮𑁂𑀲𑀼 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑁄𑀲𑀸𑀲𑀫𑀸𑀦𑀭𑀽𑀧𑀸 𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺—  ‘𑀇𑀥 𑀲𑀽𑀧𑀁 𑀤𑁂𑀣, 𑀇𑀥 𑀑𑀤𑀦𑀁 𑀤𑁂𑀣𑀸’𑀢𑀺, 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀧𑀲𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸—  ‘𑀅𑀧𑀲𑀓𑁆𑀓 𑀢𑀸𑀯, 𑀪𑀕𑀺𑀦𑀺, 𑀬𑀸𑀯 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀏𑀓𑀲𑁆𑀲 𑀘𑁂𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀧𑀝𑀺𑀪𑀸𑀲𑁂𑀬𑁆𑀬 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀧𑀲𑀸𑀤𑁂𑀢𑀼𑀁—  ‘𑀅𑀧𑀲𑀓𑁆𑀓 𑀢𑀸𑀯, 𑀪𑀕𑀺𑀦𑀺, 𑀬𑀸𑀯 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺—  ‘𑀕𑀸𑀭𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀫𑁆𑀳𑀸 𑀅𑀲𑀧𑁆𑀧𑀸𑀬𑀁 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁, 𑀢𑀁 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀫𑀸’”**𑀢𑀺𑁇 (*𑁨*:**𑁧𑁪𑁩**)

(559.)

1616\. **𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀦𑁂𑀯 𑀓𑀼𑀮𑁂𑀲𑀼 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻**𑀢𑀺 𑀓𑀼𑀮𑀁 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀼𑀮𑀸𑀦𑀺—  𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁, 𑀯𑁂𑀲𑁆𑀲𑀓𑀼𑀮𑀁, 𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑀁𑁇

1617\. **𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻**𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀪𑁄𑀚𑀦𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀪𑁄𑀚𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇

1618\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻** 𑀦𑀸𑀫 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑁇

1619\. **𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀻** 𑀦𑀸𑀫 𑀬𑀣𑀸𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀬𑀣𑀸𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀢𑀸 𑀬𑀣𑀸𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀢𑀸 𑀬𑀣𑀸𑀲𑀫𑀸𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑀢𑀸 𑀬𑀣𑀸𑀲𑀫𑀸𑀦𑀸𑀘𑀭𑀺𑀬𑀓𑀢𑀸—  “𑀇𑀥 𑀲𑀽𑀧𑀁 𑀤𑁂𑀣, 𑀇𑀥 𑀑𑀤𑀦𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀏𑀲𑀸 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀻 𑀦𑀸𑀫𑁇

1620\. **𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻**𑀢𑀺 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺𑁇

1621\. **𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻**𑀢𑀺 𑀬𑀸 𑀲𑀸 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑁇

1622\. 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀧𑀲𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸—  “𑀅𑀧𑀲𑀓𑁆𑀓 𑀢𑀸𑀯, 𑀪𑀕𑀺𑀦𑀺, 𑀬𑀸𑀯 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀓𑀲𑁆𑀲 𑀘𑁂𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀧𑀲𑀸𑀤𑀺𑀢𑁄—  “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇

(560.)

1623\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇

1624\. 𑀏𑀓𑀢𑁄𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(561.)

1625\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀧𑁂𑀢𑀺 𑀦 𑀤𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺 𑀦 𑀤𑀸𑀧𑁂𑀢𑀺, 𑀬𑀁 𑀦 𑀤𑀺𑀦𑁆𑀦𑀁 𑀢𑀁 𑀤𑀸𑀧𑁂𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦 𑀤𑀺𑀦𑁆𑀦𑀁 𑀢𑀢𑁆𑀣 𑀤𑀸𑀧𑁂𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀫𑀓𑀁 𑀤𑀸𑀧𑁂𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀯𑁄𑀲𑀸𑀲𑀢𑀺, 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀯𑁄𑀲𑀸𑀲𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀢𑁆𑀣, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
