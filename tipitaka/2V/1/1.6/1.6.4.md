# 𑁧𑁇𑁬𑁇𑁪 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀓𑀡𑁆𑀟

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(570.)

1656. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀓𑀺𑀬𑀤𑀸𑀲𑀓𑀸 𑀅𑀯𑀭𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀺𑀬𑁄 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀪𑀢𑁆𑀢𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀲𑀸𑀓𑀺𑀬𑀤𑀸𑀲𑀓𑀸—  “𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀺𑀬𑁄 𑀓𑀺𑀭 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀪𑀢𑁆𑀢𑀁 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀫𑀕𑁆𑀕𑁂 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀺𑀬𑁄 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀆𑀭𑀜𑁆𑀜𑀓𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀲𑀸𑀓𑀺𑀬𑀤𑀸𑀲𑀓𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀘 𑀤𑀽𑀲𑁂𑀲𑀼𑀜𑁆𑀘𑁇 𑀲𑀸𑀓𑀺𑀬𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀘𑁄𑀭𑁂 𑀲𑀪𑀡𑁆𑀟𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀤𑀦𑁆𑀢𑀸 𑀆𑀭𑀸𑀫𑁂 𑀘𑁄𑀭𑁂 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑁂 𑀦𑀸𑀭𑁄𑀘𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1657. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1658. **“𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀆𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀸𑀦𑀺, 𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁** **𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸—  ‘𑀕𑀸𑀭𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀅𑀲𑀧𑁆𑀧𑀸𑀬𑀁 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁 𑀢𑀁 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀫𑀻’”**𑀢𑀺𑁇

1659. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(571.)

1660. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀆𑀤𑀸𑀬 𑀆𑀭𑀜𑁆𑀜𑀓𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀪𑀼𑀜𑁆𑀚𑀣, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀢𑀼𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑁄 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺, 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑀼𑀁, 𑀙𑀺𑀦𑁆𑀦𑀪𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇

1661. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀆𑀭𑀜𑁆𑀜𑀓𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀢𑀼𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(572.)

1662. **“𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀆𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀸𑀦𑀺, 𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀔𑀸𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸—  ‘𑀕𑀸𑀭𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀅𑀲𑀧𑁆𑀧𑀸𑀬𑀁 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁, 𑀢𑀁 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀫𑀻’”**𑀢𑀺𑁇 (*𑁪*:**𑁧𑁪𑁫**)

(573.)

1663. **𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀆𑀭𑀜𑁆𑀜𑀓𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀻**𑀢𑀺 **𑀆𑀭𑀜𑁆𑀜𑀓𑀁** 𑀦𑀸𑀫 **𑀲𑁂𑀦𑀸𑀲𑀦𑀁** 𑀧𑀜𑁆𑀘𑀥𑀦𑀼𑀲𑀢𑀺𑀓𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

1664. **𑀲𑀸𑀲𑀗𑁆𑀓𑀁** 𑀦𑀸𑀫 𑀆𑀭𑀸𑀫𑁂 𑀆𑀭𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑁂 𑀘𑁄𑀭𑀸𑀦𑀁 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀼𑀢𑁆𑀢𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀞𑀺𑀢𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀧𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1665. **𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀁** 𑀦𑀸𑀫 𑀆𑀭𑀸𑀫𑁂 𑀆𑀭𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑁂 𑀘𑁄𑀭𑁂𑀳𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀳𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀆𑀓𑁄𑀝𑀺𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

1666. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1667. **𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀽**𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀲𑀼𑁇

1668. **𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁, 𑀏𑀢𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇 𑀆𑀭𑀸𑀫𑀁 𑀆𑀭𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁, 𑀏𑀢𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇

1669. 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀇𑀢𑁆𑀣𑀻 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀸 𑀆𑀭𑀸𑀫𑀁 𑀆𑀭𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀆𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀸𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀺 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁; 𑀲𑀘𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀦𑁆𑀢𑀺 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁; 𑀲𑀘𑁂—  “𑀳𑁄𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀳𑀭𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀪𑀡𑀢𑀺, 𑀘𑁄𑀭𑀸 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀇𑀥𑀽𑀧𑀘𑀭𑀦𑁆𑀢𑀺 𑀅𑀧𑀲𑀓𑁆𑀓𑀣𑀸”𑀢𑀺𑁇 𑀬𑀸𑀕𑀼𑀬𑀸 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀢𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑁄 𑀆𑀳𑀭𑀺𑀬𑁆𑀬𑀢𑀺, 𑀏𑀢𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀸𑀭𑁄 𑀆𑀳𑀭𑀺𑀬𑁆𑀬𑀢𑀺, 𑀏𑀢𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀸𑀭𑁄 𑀆𑀳𑀭𑀺𑀬𑁆𑀬𑀢𑀺, 𑀏𑀢𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇 𑀓𑀼𑀮𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀆𑀳𑀭𑀢𑀺, 𑀏𑀢𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇 𑀕𑀸𑀫𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀸𑀫𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀆𑀳𑀭𑀢𑀺, 𑀏𑀢𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇 𑀧𑀽𑀕𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀬𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀽𑀕𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀆𑀳𑀭𑀢𑀺, 𑀏𑀢𑀁 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀦𑀸𑀫𑁇

1670. **𑀔𑀸𑀤𑀦𑀻𑀬𑀁** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀸𑀦𑀺—  𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀦𑀸𑀫𑁇

1671. **𑀪𑁄𑀚𑀦𑀻𑀬𑀁** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀸𑀦𑀺—  𑀑𑀤𑀦𑁄, 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑁄, 𑀲𑀢𑁆𑀢𑀼, 𑀫𑀘𑁆𑀙𑁄, 𑀫𑀁𑀲𑀁𑁇

1672. **𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁄** 𑀦𑀸𑀫 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄𑀆𑀭𑀸𑀫𑁄𑁇 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀘𑀸𑀭𑁄𑁇

1673. **𑀅𑀕𑀺𑀮𑀸𑀦𑁄** 𑀦𑀸𑀫 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑀼𑀁𑁇

1674. **𑀕𑀺𑀮𑀸𑀦𑁄** 𑀦𑀸𑀫 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑀼𑀁𑁇

1675. 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀁 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇

(574.)

1676. 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀲𑀜𑁆𑀜𑀻 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀲𑀜𑁆𑀜𑀻 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇

1677. 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀆𑀳𑀸𑀭𑀢𑁆𑀣𑀸𑀬 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(575.)

1678. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂 𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀯𑀸 𑀲𑁂𑀲𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀩𑀳𑀸𑀭𑀸𑀫𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄𑀆𑀭𑀸𑀫𑁂 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀚𑀸𑀢𑀓𑀁 𑀫𑀽𑀮𑀁 𑀯𑀸 𑀢𑀘𑀁 𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀼𑀧𑁆𑀨𑀁 𑀯𑀸 𑀨𑀮𑀁 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀬𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1679. 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



1680. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻𑀢𑀺𑁇

---

1681. 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀓𑀡𑁆𑀟𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
