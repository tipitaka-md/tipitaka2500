# 𑁧𑁇𑁬𑁇𑁩 𑀢𑀢𑀺𑀬𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀓𑀡𑁆𑀟

### 𑀢𑀢𑀺𑀬𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(562.)

1627. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀓𑀼𑀮𑀁 𑀉𑀪𑀢𑁄𑀧𑀲𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀤𑁆𑀥𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀪𑁄𑀕𑁂𑀦 𑀳𑀸𑀬𑀢𑀺, 𑀬𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀤𑀸 𑀅𑀦𑀲𑀺𑀢𑀸 𑀅𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀇𑀫𑁂𑀲𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀤𑀸 𑀅𑀦𑀲𑀺𑀢𑀸 𑀅𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1628. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀁 𑀓𑀼𑀮𑀁 𑀲𑀤𑁆𑀥𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀪𑁄𑀕𑁂𑀦 𑀳𑀸𑀬𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(563.)

1629. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀓𑀼𑀮𑀁 𑀲𑀤𑁆𑀥𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀪𑁄𑀕𑁂𑀦 𑀳𑀸𑀬𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1630. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀓𑀼𑀮𑀁 𑀲𑀤𑁆𑀥𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀪𑁄𑀕𑁂𑀦 𑀳𑀸𑀬𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸 𑀤𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1631. 𑀤𑀺𑀦𑁆𑀦𑀸 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇

1632. 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1633. **“𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺, 𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸—  ‘𑀕𑀸𑀭𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀅𑀲𑀧𑁆𑀧𑀸𑀬𑀁 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁 𑀢𑀁 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀫𑀻’”**𑀢𑀺𑁇

1634. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(564.)

1635. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀉𑀲𑁆𑀲𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀪𑁄𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀫𑁆𑀧𑀺 𑀔𑁄 𑀓𑀼𑀮𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦𑀸𑀥𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀺—  “𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀢𑀼𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀚𑀻𑀯𑀺𑀢𑁂𑀦 𑀬𑀁 𑀅𑀬𑁆𑀬𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1636. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁂𑀦 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀢𑀼𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1637. **“𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺, 𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸**, **𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸—  ‘𑀕𑀸𑀭𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀅𑀲𑀧𑁆𑀧𑀸𑀬𑀁 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁, 𑀢𑀁 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀫𑀻’”**𑀢𑀺𑁇

1638. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(565.)

1639. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀪𑀼𑀜𑁆𑀚𑀣, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁂𑀦 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀢𑀼𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑁄 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺; 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑀼𑀁; 𑀙𑀺𑀦𑁆𑀦𑀪𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇

1640. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀢𑀼𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(566.)

1641. **“𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀺, 𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑁂𑀬𑁆𑀬 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸—  ‘𑀕𑀸𑀭𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀅𑀲𑀧𑁆𑀧𑀸𑀬𑀁 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁, 𑀢𑀁 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀫𑀻’”**𑀢𑀺𑁇 (*𑁩*:**𑁧𑁪𑁪**)

(567.)

1642. **𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀓𑀼𑀮𑀸𑀦𑀻**𑀢𑀺 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀁 𑀦𑀸𑀫 𑀓𑀼𑀮𑀁 𑀬𑀁 𑀓𑀼𑀮𑀁 𑀲𑀤𑁆𑀥𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺, 𑀪𑁄𑀕𑁂𑀦 𑀳𑀸𑀬𑀢𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀓𑀼𑀮𑀲𑁆𑀲 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀤𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇

1643. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1644. **𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀽**𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂𑀲𑀼 𑀓𑀼𑀮𑁂𑀲𑀼𑁇

1645. **𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄** 𑀦𑀸𑀫 𑀅𑀚𑁆𑀚𑀢𑀦𑀸𑀬 𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀯𑀸 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄, 𑀖𑀭𑀽𑀧𑀘𑀸𑀭𑀁 𑀑𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺, 𑀏𑀲𑁄 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀦𑀸𑀫𑁇

1646. **𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄** 𑀦𑀸𑀫 𑀅𑀚𑁆𑀚𑀢𑀦𑀸𑀬 𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄, 𑀖𑀭𑀽𑀧𑀘𑀸𑀭𑀁 𑀅𑀦𑁄𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺, 𑀏𑀲𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀦𑀸𑀫𑁇

1647. **𑀅𑀕𑀺𑀮𑀸𑀦𑁄** 𑀦𑀸𑀫 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑀼𑀁𑁇

1648. **𑀕𑀺𑀮𑀸𑀦𑁄** 𑀦𑀸𑀫 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑀼𑀁𑁇

1649. **𑀔𑀸𑀤𑀦𑀻𑀬𑀁** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀸𑀦𑀺—  𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀦𑀸𑀫𑁇

1650. **𑀪𑁄𑀚𑀦𑀻𑀬𑀁** 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀸𑀦𑀺—  𑀑𑀤𑀦𑁄, 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑁄, 𑀲𑀢𑁆𑀢𑀼, 𑀫𑀘𑁆𑀙𑁄, 𑀫𑀁𑀲𑀁𑁇

1651. 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇

(568.)

1652. 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀲𑀜𑁆𑀜𑀻 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄…𑀧𑁂…  𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂 𑀅𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀲𑀜𑁆𑀜𑀻 𑀅𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀢𑀺 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇

1653. 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀆𑀳𑀸𑀭𑀢𑁆𑀣𑀸𑀬 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁂 𑀅𑀲𑁂𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(569.)

1654. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀲𑁆𑀲 𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀯𑀸 𑀲𑁂𑀲𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀸 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀖𑀭𑀢𑁄 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦𑀺𑀘𑁆𑀘𑀪𑀢𑁆𑀢𑁂, 𑀲𑀮𑀸𑀓𑀪𑀢𑁆𑀢𑁂, 𑀧𑀓𑁆𑀔𑀺𑀓𑁂, 𑀉𑀧𑁄𑀲𑀣𑀺𑀓𑁂, 𑀧𑀸𑀝𑀺𑀧𑀤𑀺𑀓𑁂, 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁—  “𑀲𑀢𑀺 𑀧𑀘𑁆𑀘𑀬𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀸”𑀢𑀺 𑀤𑁂𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1655. 𑀢𑀢𑀺𑀬𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
