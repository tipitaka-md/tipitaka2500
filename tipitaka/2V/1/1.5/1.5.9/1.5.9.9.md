# 𑁧𑁇𑁫𑁇𑁯𑁇𑁯 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑀦𑀯𑀕𑁆𑀕

* 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(542.)

1576\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸”𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀧𑀼𑀭𑀢𑁄𑀧𑀺 𑀧𑀘𑁆𑀙𑀢𑁄𑀧𑀺 𑀆𑀓𑀟𑁆𑀠𑀦𑁆𑀢𑀸 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀺𑁇

1577\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(543.)

1578\. **“𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀦 𑀧𑀫𑀸𑀡𑀺𑀓𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀭𑀺𑀤𑀁 𑀧𑀫𑀸𑀡𑀁—  𑀤𑀻𑀖𑀲𑁄 𑀙 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄, 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑀸; 𑀢𑀺𑀭𑀺𑀬𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀬𑁆𑀬𑀸𑁇 𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑀬𑀢𑁄 𑀙𑁂𑀤𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁯𑁧*:**𑁧𑁪𑁦**)

(544.)

1579\. **𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸** 𑀦𑀸𑀫 𑀯𑀲𑁆𑀲𑀸𑀦𑀲𑁆𑀲 𑀘𑀢𑀼𑀫𑀸𑀲𑀢𑁆𑀣𑀸𑀬𑁇

1580\. **𑀓𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀦𑀸**𑀢𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀯𑀸𑁇 𑀧𑀫𑀸𑀡𑀺𑀓𑀸 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑁆𑀭𑀺𑀤𑀁 𑀧𑀫𑀸𑀡𑀁—  𑀤𑀻𑀖𑀲𑁄 𑀙 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄, 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑀸; 𑀢𑀺𑀭𑀺𑀬𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀬𑁆𑀬𑀸𑁇 𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁𑁇

(545.)

1581\. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1582\. 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(546.)

1583\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀫𑀸𑀡𑀺𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀊𑀦𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀯𑀺𑀢𑀸𑀦𑀁 𑀯𑀸 𑀪𑀽𑀫𑀢𑁆𑀣𑀭𑀡𑀁 𑀯𑀸 𑀲𑀸𑀡𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀁 𑀯𑀸 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1584\. 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀦𑀯𑀫𑀁𑁇
