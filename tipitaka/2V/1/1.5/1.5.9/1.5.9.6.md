# 𑁧𑁇𑁫𑁇𑁯𑁇𑁬 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑀦𑀯𑀕𑁆𑀕

* 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(526.)

1543\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀜𑁆𑀘𑀫𑁆𑀧𑀺 𑀧𑀻𑀞𑀫𑁆𑀧𑀺 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀸𑀭𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀜𑁆𑀘𑀫𑁆𑀧𑀺 𑀧𑀻𑀞𑀫𑁆𑀧𑀺 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇

1544\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀜𑁆𑀘𑀫𑁆𑀧𑀺 𑀧𑀻𑀞𑀫𑁆𑀧𑀺 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀜𑁆𑀘𑀫𑁆𑀧𑀺 𑀧𑀻𑀞𑀫𑁆𑀧𑀺 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀫𑀜𑁆𑀘𑀫𑁆𑀧𑀺 𑀧𑀻𑀞𑀫𑁆𑀧𑀺 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(527.)

1545\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀜𑁆𑀘𑀁 𑀯𑀸 𑀧𑀻𑀞𑀁 𑀯𑀸 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀬𑁆𑀬, 𑀉𑀤𑁆𑀤𑀸𑀮𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁮𑁮*:**𑁧𑁩𑁭**)

(528.)

1546\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1547\. **𑀫𑀜𑁆𑀘𑁄** 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀜𑁆𑀘𑀸—  𑀫𑀲𑀸𑀭𑀓𑁄, 𑀩𑀼𑀦𑁆𑀤𑀺𑀓𑀸𑀩𑀤𑁆𑀥𑁄, 𑀓𑀼𑀍𑀈𑀭𑀧𑀸𑀤𑀓𑁄, 𑀆𑀳𑀘𑁆𑀘𑀧𑀸𑀤𑀓𑁄𑁇

1548\. **𑀧𑀻𑀞𑀁** 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀻𑀞𑀸𑀦𑀺—  𑀫𑀲𑀸𑀭𑀓𑀁, 𑀩𑀼𑀦𑁆𑀤𑀺𑀓𑀸𑀩𑀤𑁆𑀥𑀁, 𑀓𑀼𑀍𑀈𑀭𑀧𑀸𑀤𑀓𑀁, 𑀆𑀳𑀘𑁆𑀘𑀧𑀸𑀤𑀓𑀁𑁇

1549\. **𑀢𑀽𑀮𑀁** 𑀦𑀸𑀫 𑀢𑀻𑀡𑀺 𑀢𑀽𑀮𑀸𑀦𑀺—  𑀭𑀼𑀓𑁆𑀔𑀢𑀽𑀮𑀁, 𑀮𑀢𑀸𑀢𑀽𑀮𑀁, 𑀧𑁄𑀝𑀓𑀺𑀢𑀽𑀮𑀁𑁇

1550\. **𑀓𑀸𑀭𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 𑀧𑀝𑀺𑀮𑀸𑀪𑁂𑀦 𑀉𑀤𑁆𑀤𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀁𑁇

(529.)

1551\. 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀭𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀁 𑀧𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1552\. 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(530.)

1553\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀆𑀬𑁄𑀕𑁂, 𑀓𑀸𑀬𑀩𑀦𑁆𑀥𑀦𑁂, 𑀅𑀁𑀲𑀩𑀤𑁆𑀥𑀓𑁂, 𑀧𑀢𑁆𑀢𑀢𑁆𑀣𑀯𑀺𑀓𑀸𑀬, 𑀧𑀭𑀺𑀲𑁆𑀲𑀸𑀯𑀦𑁂, 𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀢𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑁆𑀤𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1554\. 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
