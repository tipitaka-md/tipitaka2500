# 𑁧𑁇𑁫𑁇𑁯𑁇𑁨 𑀭𑀢𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑀦𑀯𑀕𑁆𑀕

* 𑀭𑀢𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(502.)

1476\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑀺𑀭𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀦𑀳𑀸𑀬𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀁 𑀣𑀯𑀺𑀓𑀁 𑀣𑀮𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀘𑀺𑀭𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀦𑀳𑀸𑀬𑀦𑁆𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀢𑀲𑁆𑀲𑀸𑀬𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀣𑀯𑀺𑀓𑀸, 𑀫𑀸 𑀇𑀥 𑀦𑀲𑁆𑀲𑀻”𑀢𑀺 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀢𑀼𑀭𑀺𑀢𑁄 𑀆𑀥𑀸𑀯𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀧𑀺 𑀫𑁂, 𑀪𑁄, 𑀣𑀯𑀺𑀓𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? “𑀳𑀦𑁆𑀤, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸”𑀢𑀺 𑀅𑀤𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀉𑀧𑀸𑀬𑁂𑀦 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀼𑀡𑁆𑀡𑀧𑀢𑁆𑀢𑀁 𑀦 𑀤𑀤𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀦 𑀫𑁂, 𑀪𑁄, 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺, 𑀲𑀳𑀲𑁆𑀲𑀁 𑀫𑁂”𑀢𑀺 𑀧𑀮𑀺𑀩𑀼𑀦𑁆𑀥𑁂𑀢𑁆𑀯𑀸 𑀫𑀼𑀜𑁆𑀘𑀺𑁇

1477\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀢𑀦𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀭𑀢𑀦𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀭𑀢𑀦𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1478\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀢𑀦𑀁 𑀯𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬 𑀯𑀸 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇

1479\. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(503.)

1480\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀉𑀲𑁆𑀲𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀅𑀮𑀗𑁆𑀓𑀢𑀧𑁆𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸 𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀯𑀺𑀲𑀸𑀔𑀸𑀧𑀺 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀅𑀮𑀗𑁆𑀓𑀢𑀧𑁆𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸 “𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀸𑀫𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸—  “𑀓𑁆𑀬𑀸𑀳𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸, 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺 𑀆𑀪𑀭𑀡𑀁 𑀑𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑁂𑀦 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀺𑀬𑀸 𑀅𑀤𑀸𑀲𑀺—  “𑀳𑀦𑁆𑀤, 𑀚𑁂, 𑀇𑀫𑀁 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀤𑀸𑀲𑀻 𑀢𑀁 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀯𑀺𑀲𑁆𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑀦𑀁 𑀯𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀼𑀁—  ‘𑀬𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1481\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀢𑀦𑀁 𑀯𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑀸, 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬 𑀯𑀸 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇

1483\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑀸𑀲𑀻𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀕𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀘 𑀕𑀳𑀧𑀢𑀺𑀦𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀻 𑀆𑀡𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺—  “𑀲𑀘𑁂 𑀪𑀤𑀦𑁆𑀢𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀪𑀢𑁆𑀢𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀸𑀲𑀻𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑀸 𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀕𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀼𑀁 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀁 𑀑𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀅𑀬𑁆𑀬𑀸 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀁 𑀯𑀺𑀲𑁆𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀲𑀘𑁂 𑀫𑀬𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫 𑀦𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀸𑀬𑀁 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀸”𑀢𑀺 𑀢𑀢𑁆𑀣𑁂𑀯 𑀅𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀇𑀥𑁂𑀯 𑀅𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

1484\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑀦𑀁 𑀯𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑁂 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀼𑀁—  ‘𑀬𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1485\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀢𑀦𑀁 𑀯𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑀸 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑀸 𑀯𑀸, 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬 𑀯𑀸 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀭𑀢𑀦𑀁 𑀯𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑁂 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀬𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀸𑀫𑀻𑀘𑀻”**𑀢𑀺𑁇 (*𑁮𑁪*:**𑁧𑁩𑁩**)

1486\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1487\. **𑀭𑀢𑀦𑀁** 𑀦𑀸𑀫 𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀡𑀺 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑁄 𑀲𑀗𑁆𑀔𑁄 𑀲𑀺𑀮𑀸 𑀧𑀯𑀸𑀮𑀁 𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑁄 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀁𑁇

1488\. **𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁** 𑀦𑀸𑀫 𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀧𑀪𑁄𑀕𑀧𑀭𑀺𑀪𑁄𑀕𑀁, 𑀏𑀢𑀁 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀦𑀸𑀫𑁇

1489\. **𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑀸 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑀸 𑀯𑀸**𑀢𑀺 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑀁𑁇

1490\. **𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁄** 𑀦𑀸𑀫 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄 𑀆𑀭𑀸𑀫𑁄, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀘𑀸𑀭𑁄𑁇

1491\. **𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑁄** 𑀦𑀸𑀫 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀯𑀲𑀣𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄 𑀆𑀯𑀲𑀣𑁄, 𑀅𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀘𑀸𑀭𑁄𑁇

1492\. **𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1493\. **𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀕𑀸𑀳𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1494\. **𑀭𑀢𑀦𑀁 𑀯𑀸 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑁂 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑀦𑁆**𑀢𑀺 𑀭𑀽𑀧𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀲𑀜𑁆𑀜𑀸𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  “𑀬𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀁 𑀦𑀝𑁆𑀞𑀁 𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑀢𑀽”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀢𑀢𑁆𑀣 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀲𑁄 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  “𑀆𑀯𑀼𑀲𑁄, 𑀓𑀻𑀤𑀺𑀲𑀁 𑀢𑁂 𑀪𑀡𑁆𑀟𑀦𑁆”𑀢𑀺? 𑀲𑀘𑁂 𑀭𑀽𑀧𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀤𑀸𑀢𑀩𑁆𑀩𑀁, 𑀦𑁄 𑀘𑁂 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 “𑀯𑀺𑀘𑀺𑀦𑀸𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁂𑀦 𑀬𑁂 𑀢𑀢𑁆𑀣 𑀳𑁄𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀢𑀺𑀭𑀽𑀧𑀸, 𑀢𑁂𑀲𑀁 𑀳𑀢𑁆𑀣𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀢𑀺𑀭𑀽𑀧𑀸, 𑀬𑁂 𑀢𑀢𑁆𑀣 𑀳𑁄𑀦𑁆𑀢𑀺 𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑀸, 𑀢𑁂𑀲𑀁 𑀳𑀢𑁆𑀣𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1495\. **𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀸𑀫𑀻𑀘𑀻**𑀢𑀺 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀢𑀸𑁇

(507.)

1496\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀭𑀢𑀦𑀁 𑀯𑀸 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀯𑀸 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀣𑁂 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀉𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀢𑀺—  “𑀬𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀭𑀢𑀦𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀢𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
