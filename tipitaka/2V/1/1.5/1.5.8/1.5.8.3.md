# 𑁧𑁇𑁫𑁇𑁮𑁇𑁩 𑀫𑁄𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀫𑁄𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(443.)

1329\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀺𑀢𑁆𑀯𑀸 “𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑁂𑀦 𑀆𑀧𑀦𑁆𑀦𑀸𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑀽”𑀢𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁 𑀚𑀸𑀦𑀸𑀫, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀓𑀺𑀭 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀢𑁆𑀢𑀸𑀕𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇

1330\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁 𑀚𑀸𑀦𑀸𑀫, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀓𑀺𑀭 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀢𑁆𑀢𑀸𑀕𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻’”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀣—  ‘𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁 𑀚𑀸𑀦𑀸𑀫, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀓𑀺𑀭 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀢𑁆𑀢𑀸𑀕𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻’”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀣—  ‘𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀫𑀬𑀁 𑀚𑀸𑀦𑀸𑀫, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀓𑀺𑀭 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀢𑁆𑀢𑀸𑀕𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻’𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(444.)

1331\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀅𑀳𑀁 𑀚𑀸𑀦𑀸𑀫𑀺, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀓𑀺𑀭 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀢𑁆𑀢𑀸𑀕𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻’𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀁 𑀇𑀫𑀺𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂, 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀪𑀺𑀬𑁆𑀬𑁄, 𑀦 𑀘 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑁂𑀦 𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀢𑁆𑀣𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀢𑁆𑀣 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀢𑀜𑁆𑀘 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘𑀲𑁆𑀲 𑀫𑁄𑀳𑁄 𑀆𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀢𑀲𑁆𑀲 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀮𑀸𑀪𑀸, 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀦 𑀲𑀸𑀥𑀼𑀓𑀁 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀲𑀻’𑀢𑀺𑁇 𑀇𑀤𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑁄𑀳𑀦𑀓𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁩*:**𑁧𑁨𑁨**)

(445.)

1332\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1333\. **𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀦𑁆**𑀢𑀺 𑀅𑀦𑀼𑀧𑁄𑀲𑀣𑀺𑀓𑀁𑁇

1334\. **𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂**𑀢𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑁂𑁇

1335\. **𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀺𑀢𑁆𑀯𑀸—  “𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑁂𑀦 𑀆𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑀽”𑀢𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀇𑀤𑀸𑀦𑁂𑀯 𑀔𑁄 𑀅𑀳𑀁 𑀚𑀸𑀦𑀸𑀫𑀺, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀓𑀺𑀭 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀢𑁆𑀢𑀸𑀕𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1336\. 𑀢𑀜𑁆𑀘𑁂 𑀫𑁄𑀳𑁂𑀢𑀼𑀓𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀁 𑀇𑀫𑀺𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂, 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀪𑀺𑀬𑁆𑀬𑁄, 𑀦 𑀘 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀜𑁆𑀜𑀸𑀡𑀓𑁂𑀦 𑀫𑀼𑀢𑁆𑀢𑀺 𑀅𑀢𑁆𑀣𑀺, 𑀬𑀜𑁆𑀘 𑀢𑀢𑁆𑀣 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄, 𑀢𑀜𑁆𑀘 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀘𑀲𑁆𑀲 𑀫𑁄𑀳𑁄 𑀆𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 “𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(446.)

1337\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀦 𑀲𑀸𑀥𑀼𑀓𑀁 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀳𑀁 𑀆𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1338\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀦 𑀲𑀸𑀥𑀼𑀓𑀁 𑀅𑀝𑁆𑀞𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀳𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀳𑀲𑁆𑀲 𑀆𑀭𑁄𑀧𑀦𑀸, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1339\. 𑀆𑀭𑁄𑀧𑀺𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑁄𑀳𑁄𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

1340\. 𑀅𑀦𑀸𑀭𑁄𑀧𑀺𑀢𑁂 𑀫𑁄𑀳𑁂 𑀫𑁄𑀳𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀆𑀭𑁄𑀧𑀺𑀢𑁂 𑀫𑁄𑀳𑁂 𑀫𑁄𑀳𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(447.)

1341\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀫𑁄𑀳𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀫𑁄𑀳𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀫𑁄𑀳𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1342\. 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(448.)

1343\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀦 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀊𑀦𑀓𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀦 𑀫𑁄𑀳𑁂𑀢𑀼𑀓𑀸𑀫𑀲𑁆𑀲, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1344\. 𑀫𑁄𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
