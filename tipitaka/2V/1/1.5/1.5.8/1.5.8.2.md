# 𑁧𑁇𑁫𑁇𑁮𑁇𑁨 𑀯𑀺𑀮𑁂𑀔𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀮𑁂𑀔𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(438.)

1318\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀦𑀬𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀸 𑀔𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀦𑀬𑀓𑀣𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀤𑀺𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀺𑀦𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀸𑀫𑀸”𑀢𑀺, 𑀢𑁂 𑀘 𑀩𑀳𑀽 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑀸 𑀘 𑀦𑀯𑀸 𑀘 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀘 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀺𑀦𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺𑁇

1319\. 𑀅𑀣 𑀔𑁄 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀏𑀢𑀭𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑀳𑀽 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑀸 𑀘 𑀦𑀯𑀸 𑀘 𑀫𑀚𑁆𑀛𑀺𑀫𑀸 𑀘 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀺𑀦𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺𑁇 𑀲𑀘𑁂 𑀇𑀫𑁂 𑀯𑀺𑀦𑀬𑁂 𑀧𑀓𑀢𑀜𑁆𑀜𑀼𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀫𑁆𑀳𑁂 𑀬𑁂𑀦𑀺𑀘𑁆𑀙𑀓𑀁 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁 𑀬𑀸𑀯𑀤𑀺𑀘𑁆𑀙𑀓𑀁 𑀆𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀓𑀺𑀁 𑀧𑀦𑀺𑀫𑁂𑀳𑀺 𑀔𑀼𑀤𑁆𑀤𑀸𑀦𑀼𑀔𑀼𑀤𑁆𑀤𑀓𑁂𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀳𑀺 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑁂𑀳𑀺, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀯𑀺𑀮𑁂𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1320\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(439.)

1321\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀓𑀺𑀁 𑀧𑀦𑀺𑀫𑁂𑀳𑀺 𑀔𑀼𑀤𑁆𑀤𑀸𑀦𑀼𑀔𑀼𑀤𑁆𑀤𑀓𑁂𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀳𑀺 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑁂𑀳𑀺, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀯𑀺𑀮𑁂𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀯𑀺𑀯𑀡𑁆𑀡𑀓𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁨*:**𑁧𑁨𑁧**)

(440.)

1322\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1323\. **𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂**𑀢𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑁂 𑀯𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑁂 𑀯𑀸 𑀲𑀚𑁆𑀛𑀸𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑁇

1324\. **𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀸**𑀢𑀺—  “𑀓𑀺𑀁 𑀧𑀦𑀺𑀫𑁂𑀳𑀺 𑀔𑀼𑀤𑁆𑀤𑀸𑀦𑀼𑀔𑀼𑀤𑁆𑀤𑀓𑁂𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀳𑀺 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑁂𑀳𑀺, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀯𑀺𑀮𑁂𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀬𑁂 𑀇𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀳𑁂𑀲𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀮𑁂𑀔𑀸 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀇𑀫𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑀸 𑀦 𑀳𑁄𑀢𑀺 𑀯𑀺𑀮𑁂𑀔𑀸 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀅𑀦𑀼𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀇𑀤𑀁 𑀯𑀭𑀁, 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀇𑀤𑀁 𑀯𑀭𑀁, 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀝𑀁 𑀇𑀤𑀁 𑀯𑀭𑀁, 𑀅𑀥𑀸𑀭𑀺𑀢𑀁 𑀇𑀤𑀁 𑀯𑀭𑀁, 𑀯𑀺𑀦𑀬𑁄 𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀢𑀼, 𑀇𑀫𑁂 𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀓𑀢𑀜𑁆𑀜𑀼𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀽”𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(441.)

1325\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1326\. 𑀅𑀜𑁆𑀜𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(442.)

1327\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀦 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀼𑀓𑀸𑀫𑁄, “𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁂 𑀯𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀯𑀸 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀲𑁆𑀲𑀼, 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀦𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺 𑀪𑀡𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
