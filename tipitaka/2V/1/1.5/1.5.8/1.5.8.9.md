# 𑁧𑁇𑁫𑁇𑁮𑁇𑁯 𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(474.)

1403. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀏𑀓𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀁𑀖𑁄—  “𑀅𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀓𑁄 𑀆𑀕𑀢𑁄, 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸”𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑁄 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀁𑀖𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀢𑁆𑀣𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀫𑁆𑀳𑀸—  ‘𑀢𑀼𑀬𑁆𑀳𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀘 𑀫𑀬𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫 ‘𑀢𑀼𑀬𑁆𑀳𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀦 𑀫𑀬𑀁 𑀙𑀦𑁆𑀤𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇

1404. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(475.)

1405. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁯*:**𑁧𑁨𑁮**)

(476.)

1406. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1407. **𑀥𑀫𑁆𑀫𑀺𑀓𑀁** 𑀦𑀸𑀫 **𑀓𑀫𑁆𑀫𑀁** 𑀅𑀧𑀮𑁄𑀓𑀦𑀓𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑀓𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀓𑀢𑀁, 𑀏𑀢𑀁 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀦𑀸𑀫 𑀓𑀫𑁆𑀫𑀁𑁇 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(477.)

1408. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(478.)

1409. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  “𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀯𑀕𑁆𑀕𑁂𑀦 𑀯𑀸 𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀦𑁆”𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1410. 𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀦𑀯𑀫𑀁𑁇
