# 𑁧𑁇𑁫𑁇𑁮𑁇𑁮 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(469.)

1391. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑁂𑀲𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀡𑁆𑀟𑀦𑁆𑀢𑀺𑁇 𑀧𑁂𑀲𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑁇 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀇𑀫𑁂𑀳𑀺 𑀲𑀳 𑀪𑀡𑁆𑀟𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀫𑁆𑀳𑁂 𑀅𑀮𑀚𑁆𑀚𑀺𑀯𑀸𑀤𑁂𑀦 𑀧𑀸𑀧𑁂𑀣𑀸”𑀢𑀺? “𑀓𑀳𑀁 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀲𑁆𑀲𑀼𑀢𑁆𑀣𑀸”𑀢𑀺? “𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀸”𑀢𑀺𑁇

1392. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸𑀦𑀁 𑀓𑀮𑀳𑀚𑀸𑀢𑀸𑀦𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸𑀦𑀁 𑀓𑀮𑀳𑀚𑀸𑀢𑀸𑀦𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸𑀦𑀁 𑀓𑀮𑀳𑀚𑀸𑀢𑀸𑀦𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(470.)

1393. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸𑀦𑀁 𑀓𑀮𑀳𑀚𑀸𑀢𑀸𑀦𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬—  ‘𑀬𑀁 𑀇𑀫𑁂 𑀪𑀡𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑁄𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀢𑀤𑁂𑀯 𑀧𑀘𑁆𑀘𑀬𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀜𑁆𑀜𑀁, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁮*:**𑁧𑁨𑁭**)

(471.)

1394. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1395. **𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀦𑁆**𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁𑁇

1396. **𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸𑀦𑀁 𑀓𑀮𑀳𑀚𑀸𑀢𑀸𑀦𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀦𑁆**𑀢𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀚𑀸𑀢𑀸𑀦𑀁𑁇

1397. **𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 “𑀇𑀫𑁂𑀲𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀝𑀺𑀲𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀫𑀗𑁆𑀓𑀽 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀘𑁆𑀙𑀢𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀢𑀼𑀭𑀺𑀢𑁄 𑀕𑀘𑁆𑀙𑀢𑀺 𑀲𑁄𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀧𑀼𑀭𑀢𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀑𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀲𑁄𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀞𑀺𑀢𑁄𑀓𑀸𑀲𑀁 𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑀁 𑀯𑀸 𑀦𑀺𑀧𑀦𑁆𑀦𑁄𑀓𑀸𑀲𑀁 𑀯𑀸 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀁 𑀉𑀓𑁆𑀓𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑀁, 𑀯𑀺𑀚𑀸𑀦𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦𑁄 𑀘𑁂 𑀉𑀓𑁆𑀓𑀸𑀲𑁂𑀬𑁆𑀬 𑀯𑀸 𑀯𑀺𑀚𑀸𑀦𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1398. **𑀏𑀢𑀤𑁂𑀯 𑀧𑀘𑁆𑀘𑀬𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀜𑁆𑀜𑀦𑁆**𑀢𑀺 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀓𑁄𑀘𑀺 𑀧𑀘𑁆𑀘𑀬𑁄 𑀳𑁄𑀢𑀺 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀼𑀁𑁇

(472.)

1399. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1400. 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(473.)

1401. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  “𑀇𑀫𑁂𑀲𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀑𑀭𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀺𑀭𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀽𑀧𑀲𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀫𑁄𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
