# 𑁧𑁇𑁫𑁇𑁮𑁇𑁧𑁧 𑀤𑀼𑀩𑁆𑀩𑀮𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀩𑁆𑀩𑀮𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(484.)

1420\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑀩𑁆𑀩𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑁂𑀦𑀸𑀲𑀦𑀜𑁆𑀘 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺𑁇 𑀲𑁄 𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀖𑀲𑁆𑀲 𑀏𑀓𑀁 𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀩𑁆𑀩𑀲𑁆𑀲 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀬𑀣𑀸𑀲𑀦𑁆𑀣𑀼𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1421\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁, 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(485.)

1422\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 ‘𑀬𑀣𑀸𑀲𑀦𑁆𑀣𑀼𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁮𑁧*:**𑁧𑁩𑁦**)

(486.)

1423\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1424\. **𑀲𑀫𑀕𑁆𑀕𑁄** 𑀦𑀸𑀫 𑀲𑀁𑀖𑁄 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑁄 𑀲𑀫𑀸𑀦𑀲𑀻𑀫𑀸𑀬𑀁 𑀞𑀺𑀢𑁄𑁇

1425\. **𑀘𑀻𑀯𑀭𑀁** 𑀦𑀸𑀫 𑀙𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀼𑀧𑀕𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁𑁇

1426\. **𑀤𑀢𑁆𑀯𑀸**𑀢𑀺 𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸𑁇

1427\. **𑀬𑀣𑀸𑀲𑀦𑁆𑀣𑀼𑀢𑀁** 𑀦𑀸𑀫 𑀬𑀣𑀸𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀬𑀣𑀸𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀢𑀸 𑀬𑀣𑀸𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀢𑀸 𑀬𑀣𑀸𑀲𑀫𑀸𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑀢𑀸 𑀬𑀣𑀸𑀲𑀫𑀸𑀦𑀸𑀘𑀭𑀺𑀬𑀓𑀢𑀸𑁇

1428\. **𑀲𑀁𑀖𑀺𑀓𑀁** 𑀦𑀸𑀫 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑁇

1429\. **𑀮𑀸𑀪𑁄** 𑀦𑀸𑀫 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀘𑀼𑀡𑁆𑀡𑀧𑀺𑀡𑁆𑀟𑁄𑀧𑀺, 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀫𑁆𑀧𑀺, 𑀤𑀲𑀺𑀓𑀲𑀼𑀢𑁆𑀢𑀫𑁆𑀧𑀺𑁇

1430\. **𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀲𑁆𑀲 𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑀲𑁆𑀲 𑀯𑀸 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑀲𑁆𑀲 𑀯𑀸 𑀬𑀸𑀕𑀼𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀨𑀮𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀔𑀚𑁆𑀚𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑁂 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(487.)

1431\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀘𑀻𑀯𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑁂 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀘𑀻𑀯𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑁂 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀘𑀻𑀯𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑁂 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1432\. 𑀅𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑁂 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀲𑁆𑀲 𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑀲𑁆𑀲 𑀯𑀸 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑀲𑁆𑀲 𑀯𑀸 𑀬𑀸𑀕𑀼𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀨𑀮𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀔𑀚𑁆𑀚𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑁂 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀲𑁆𑀲 𑀯𑀸 𑀅𑀲𑀫𑁆𑀫𑀢𑀲𑁆𑀲 𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑀲𑁆𑀲 𑀯𑀸 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑀲𑁆𑀲 𑀯𑀸 𑀬𑀸𑀕𑀼𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀨𑀮𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀔𑀚𑁆𑀚𑀪𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑁂 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(488.)

1433\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀧𑀓𑀢𑀺𑀬𑀸 𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀫𑁄𑀳𑀸 𑀪𑀬𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀁 “𑀓𑁆𑀯𑀢𑁆𑀣𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑁂𑀦 𑀮𑀤𑁆𑀥𑀸𑀧𑀺 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑁆𑀲𑀢𑀺 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀉𑀧𑀦𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
