# 𑁧𑁇𑁫𑁇𑁮𑁇𑁫 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(454.)

1356\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀲𑀢𑁆𑀢𑀭𑀲𑀯𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀧𑀳𑀸𑀭𑀲𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀸 𑀭𑁄𑀤𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀭𑁄𑀤𑀣𑀸”𑀢𑀺? “𑀇𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1357\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀲𑀢𑁆𑀢𑀭𑀲𑀯𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀲𑀢𑁆𑀢𑀭𑀲𑀯𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀲𑀢𑁆𑀢𑀭𑀲𑀯𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀺𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(455.)

1358\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁭𑁫*:**𑁧𑁨𑁪**)

(456.)

1359\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1360\. **𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑀸**𑀢𑀺 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑁇

1361\. **𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄**𑀢𑀺 𑀅𑀦𑀪𑀺𑀭𑀤𑁆𑀥𑁄 𑀆𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑀺𑀮𑀚𑀸𑀢𑁄𑁇

1362\. **𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀓𑀸𑀬𑀁 𑀯𑀸 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀯𑀸 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀉𑀧𑁆𑀧𑀮𑀧𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(457.)

1363\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1364\. 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(458.)

1365\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀓𑁂𑀦𑀘𑀺 𑀯𑀺𑀳𑁂𑀞𑀻𑀬𑀫𑀸𑀦𑁄 𑀫𑁄𑀓𑁆𑀔𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1366\. 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
