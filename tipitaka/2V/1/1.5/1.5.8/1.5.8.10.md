# 𑁧𑁇𑁫𑁇𑁮𑁇𑁧𑁦 𑀙𑀦𑁆𑀤𑀁𑀅𑀤𑀢𑁆𑀯𑀸𑀕𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀙𑀦𑁆𑀤𑀁𑀅𑀤𑀢𑁆𑀯𑀸𑀕𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(479.)

1411\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀏𑀓𑀲𑁆𑀲 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀁𑀖𑁄 “𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀜𑀢𑁆𑀢𑀺𑀁 𑀞𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼—  “𑀏𑀯𑀫𑁂𑀯𑀺𑀫𑁂 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀓𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀣𑀸”𑀢𑀺 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1412\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀣𑀸𑀬 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑁂 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀣𑀸𑀬 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀲𑀁𑀖𑁂 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀣𑀸𑀬 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(480.)

1413\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀣𑀸𑀬 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁮𑁦*:**𑁧𑁨𑁯**)

(481.)

1414\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1415\. **𑀲𑀁𑀖𑁂 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀣𑀸** 𑀦𑀸𑀫 𑀯𑀢𑁆𑀣𑀼 𑀯𑀸 𑀆𑀭𑁄𑀘𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀁, 𑀜𑀢𑁆𑀢𑀺 𑀯𑀸 𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸 𑀯𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸 𑀳𑁄𑀢𑀺𑁇

1416\. **𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸**𑀢𑀺—  “𑀓𑀣𑀁 𑀇𑀤𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀲𑁆𑀲 𑀯𑀕𑁆𑀕𑀁 𑀅𑀲𑁆𑀲 𑀦 𑀓𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀧𑀭𑀺𑀲𑀸𑀬 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀯𑀺𑀚𑀳𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(482.)

1417\. 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(483.)

1418\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀯𑀸 𑀓𑀮𑀳𑁄 𑀯𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀸 𑀯𑀺𑀯𑀸𑀤𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, “𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀯𑀸 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, “𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀯𑀕𑁆𑀕𑁂𑀦 𑀯𑀸 𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑁄 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀭𑁂𑀦 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑁂𑀦 𑀯𑀸 𑀧𑀻𑀍𑀇𑀢𑁄 𑀕𑀘𑁆𑀙𑀢𑀺, “𑀦 𑀓𑀫𑁆𑀫𑀁 𑀓𑁄𑀧𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀧𑀼𑀦 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1419\. 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀕𑀫𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀲𑀫𑀁𑁇
