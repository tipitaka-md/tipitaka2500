# 𑁧𑁇𑁫𑁇𑁭𑁇𑁯 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑁄𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑁄𑀕𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(423.)

1265. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑀁 𑀢𑀣𑀸𑀯𑀸𑀤𑀺𑀦𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀓𑀝𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑀧𑀺 𑀲𑀁𑀯𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀲𑀳𑀸𑀧𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺𑁇

1266. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑀁 𑀢𑀣𑀸𑀯𑀸𑀤𑀺𑀦𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀓𑀝𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀲𑀁𑀯𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀲𑀳𑀸𑀧𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀦𑀁 𑀢𑀣𑀸𑀯𑀸𑀤𑀺𑀦𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀓𑀝𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀣𑀸𑀧𑀺 𑀲𑀁𑀯𑀲𑀣𑀸𑀧𑀺 𑀲𑀳𑀸𑀧𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀚𑀸𑀦𑀁 𑀢𑀣𑀸𑀯𑀸𑀤𑀺𑀦𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀓𑀝𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀣𑀸𑀧𑀺 𑀲𑀁𑀯𑀲𑀺𑀲𑁆𑀲𑀣𑀸𑀧𑀺 𑀲𑀳𑀸𑀧𑀺 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(424.)

1267. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀚𑀸𑀦𑀁 𑀢𑀣𑀸𑀯𑀸𑀤𑀺𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀓𑀝𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸 𑀲𑀁𑀯𑀲𑁂𑀬𑁆𑀬 𑀯𑀸 𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁬𑁯*:**𑁧𑁧𑁮**)

(425.)

1268. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1269. **𑀚𑀸𑀦𑀸𑀢𑀺** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺𑁇

1270. **𑀢𑀣𑀸𑀯𑀸𑀤𑀺𑀦𑀸**𑀢𑀺—  “𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸, 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸”𑀢𑀺 𑀏𑀯𑀁 𑀯𑀸𑀤𑀺𑀦𑀸𑁇

1271. **𑀅𑀓𑀝𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑁄** 𑀦𑀸𑀫 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇

1272. **𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆**𑀢𑀺 𑀏𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁𑁇

1273. **𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀲𑀫𑁆𑀪𑁄𑀕𑁄 𑀦𑀸𑀫 𑀤𑁆𑀯𑁂 𑀲𑀫𑁆𑀪𑁄𑀕𑀸—  𑀆𑀫𑀺𑀲𑀲𑀫𑁆𑀪𑁄𑀕𑁄 𑀘 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀪𑁄𑀕𑁄 𑀘𑁇 **𑀆𑀫𑀺𑀲𑀲𑀫𑁆𑀪𑁄𑀕𑁄** 𑀦𑀸𑀫 𑀆𑀫𑀺𑀲𑀁 𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 **𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀪𑁄𑀕𑁄** 𑀦𑀸𑀫 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺 𑀯𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀤𑁂𑀦 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺 𑀯𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀤𑁂 𑀧𑀤𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀓𑁆𑀔𑀭𑀸𑀬 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺 𑀯𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀅𑀓𑁆𑀔𑀭𑀓𑁆𑀔𑀭𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1274. **𑀲𑀁𑀯𑀲𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀁 𑀯𑀸 𑀲𑀁𑀖𑀓𑀫𑁆𑀫𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1275. **𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀏𑀓𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂 𑀦𑀺𑀧𑀦𑁆𑀦𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀧𑀦𑁆𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀪𑁄 𑀯𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(426.)

1276. 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸 𑀲𑀁𑀯𑀲𑀢𑀺 𑀯𑀸 𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸 𑀲𑀁𑀯𑀲𑀢𑀺 𑀯𑀸 𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸 𑀲𑀁𑀯𑀲𑀢𑀺 𑀯𑀸 𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁂 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(427.)

1277. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀺𑀢𑁄𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁄𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
