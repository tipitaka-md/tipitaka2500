# 𑁧𑁇𑁫𑁇𑁭𑁇𑁧𑁦 𑀓𑀡𑁆𑀝𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑀯𑀕𑁆𑀕

### 𑀓𑀡𑁆𑀝𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(428.)

1279\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑀡𑁆𑀝𑀓𑀲𑁆𑀲 𑀦𑀸𑀫 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺—  “𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀡𑁆𑀝𑀓𑀲𑁆𑀲 𑀦𑀸𑀫 𑀓𑀺𑀭 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  “𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸, 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀓𑀡𑁆𑀝𑀓𑁄 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀡𑁆𑀝𑀓𑀁 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀡𑁆𑀝𑀓, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺? “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺𑁇

1280\. “𑀫𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀡𑁆𑀝𑀓, 𑀏𑀯𑀁 𑀅𑀯𑀘𑁇 𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺𑁇 𑀦 𑀳𑀺 𑀲𑀸𑀥𑀼 𑀪𑀕𑀯𑀢𑁄 𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁𑁇 𑀦 𑀳𑀺 𑀪𑀕𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦𑀸𑀯𑀼𑀲𑁄 𑀓𑀡𑁆𑀝𑀓, 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸𑁇 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑁇 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄…𑀧𑁂…  𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀓𑀡𑁆𑀝𑀓𑁄 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀢𑀣𑁂𑀯 𑀢𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀣𑀸𑀫𑀲𑀸 𑀧𑀭𑀸𑀫𑀸𑀲𑀸 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀯𑁄𑀳𑀭𑀢𑀺—  “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺𑁇

1281\. 𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀓𑀡𑁆𑀝𑀓𑀁 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀸𑀧𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀁, 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀡𑁆𑀝𑀓𑀁 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁂, 𑀓𑀡𑁆𑀝𑀓, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀸𑀧𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁—  ‘𑀢𑀣𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺? “𑀏𑀯𑀁 𑀩𑁆𑀬𑀸 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀬𑀣𑀸 𑀬𑁂𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀦𑀸𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬𑀸’”𑀢𑀺𑁇

1282\. “𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀫𑀬𑀸 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺? 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸, 𑀅𑀮𑀜𑁆𑀘 𑀧𑀦 𑀢𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀬? 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀝𑁆𑀞𑀺𑀓𑀗𑁆𑀓𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸…𑀧𑁂…  𑀲𑀧𑁆𑀧𑀲𑀺𑀭𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀬𑀸 𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀼𑀧𑀸𑀬𑀸𑀲𑀸, 𑀆𑀤𑀻𑀦𑀯𑁄 𑀏𑀢𑁆𑀣 𑀪𑀺𑀬𑁆𑀬𑁄𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀢𑁆𑀢𑀦𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀅𑀫𑁆𑀳𑁂 𑀘𑁂𑀯 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀲𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀜𑁆𑀘 𑀔𑀡𑀲𑀺 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀲𑀺𑁇 𑀢𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀓𑀡𑁆𑀝𑀓𑀁 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀦𑀸𑀲𑁂𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀓𑀡𑁆𑀝𑀓, 𑀦 𑀘𑁂𑀯 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸 𑀅𑀧𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀫𑁆𑀧𑀺 𑀘𑀜𑁆𑀜𑁂 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀸 𑀮𑀪𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀤𑀺𑀭𑀢𑁆𑀢𑀢𑀺𑀭𑀢𑁆𑀢𑀁 𑀲𑀳𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀸𑀧𑀺 𑀢𑁂 𑀦𑀢𑁆𑀣𑀺𑁇 𑀘𑀭 𑀧𑀺𑀭𑁂 𑀯𑀺𑀦𑀲𑁆𑀲𑀸’”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀁𑀖𑁄 𑀓𑀡𑁆𑀝𑀓𑀁 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀦𑀸𑀲𑁂𑀲𑀺𑁇

1289\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1290\. **𑀚𑀸𑀦𑀸𑀢𑀺** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺𑁇

1291\. **𑀢𑀣𑀸𑀦𑀸𑀲𑀺𑀢𑀦𑁆**𑀢𑀺 𑀏𑀯𑀁 𑀦𑀸𑀲𑀺𑀢𑀁𑁇

1292\. **𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑁄** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1293\. **𑀉𑀧𑀮𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀸 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀯𑀸 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀁 𑀯𑀸 𑀤𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1294\. **𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀼𑀡𑁆𑀡𑀁 𑀯𑀸 𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀯𑀸 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀁 𑀯𑀸 𑀫𑀼𑀔𑁄𑀤𑀓𑀁 𑀯𑀸 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1295\. **𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸**𑀢𑀺 𑀲𑀫𑁆𑀪𑁄𑀕𑁄 𑀦𑀸𑀫 𑀤𑁆𑀯𑁂 𑀲𑀫𑁆𑀪𑁄𑀕𑀸—  𑀆𑀫𑀺𑀲𑀲𑀫𑁆𑀪𑁄𑀕𑁄 𑀘 𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀪𑁄𑀕𑁄 𑀘𑁇 **𑀆𑀫𑀺𑀲𑀲𑀫𑁆𑀪𑁄𑀕𑁄** 𑀦𑀸𑀫 𑀆𑀫𑀺𑀲𑀁 𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 **𑀥𑀫𑁆𑀫𑀲𑀫𑁆𑀪𑁄𑀕𑁄** 𑀦𑀸𑀫 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺 𑀯𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀤𑁂𑀦 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺 𑀯𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀧𑀤𑁂 𑀧𑀤𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀓𑁆𑀔𑀭𑀸𑀬 𑀉𑀤𑁆𑀤𑀺𑀲𑀢𑀺 𑀯𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀢𑀺 𑀯𑀸, 𑀅𑀓𑁆𑀔𑀭𑀓𑁆𑀔𑀭𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

1296\. **𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀸**𑀢𑀺 𑀏𑀓𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀦𑀸𑀲𑀺𑀢𑀓𑁂 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑁂 𑀦𑀺𑀧𑀦𑁆𑀦𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀧𑀦𑁆𑀦𑁂 𑀦𑀸𑀲𑀺𑀢𑀓𑁄 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀪𑁄 𑀯𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀉𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(431.)

1297\. 𑀦𑀸𑀲𑀺𑀢𑀓𑁂 𑀦𑀸𑀲𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸 𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀦𑀸𑀲𑀺𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸 𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦𑀸𑀲𑀺𑀢𑀓𑁂 𑀅𑀦𑀸𑀲𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀻 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺 𑀯𑀸 𑀲𑀫𑁆𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀯𑀸 𑀲𑀳 𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀺𑀢𑀓𑁂 𑀦𑀸𑀲𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀲𑀺𑀢𑀓𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀸𑀲𑀺𑀢𑀓𑁂 𑀅𑀦𑀸𑀲𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀻, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(432.)

1298\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀦𑀸𑀲𑀺𑀢𑀓𑁄𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑁄𑀢𑀺 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

1299\. 𑀓𑀡𑁆𑀝𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇

1300\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1301\. _𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀯𑀥𑀲𑀧𑁆𑀧𑀸𑀡𑀁,_  
_𑀉𑀓𑁆𑀓𑁄𑀝𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀙𑀸𑀤𑀦𑀁;_  
_𑀊𑀦𑀯𑀻𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀜𑁆𑀘,_
