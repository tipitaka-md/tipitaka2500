# 𑁧𑁇𑁫𑁇𑁭𑁇𑁬 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑀯𑀕𑁆𑀕

### 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(407.)

1223\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀲𑀢𑁆𑀣𑁄 𑀭𑀸𑀚𑀕𑀳𑀸 𑀧𑀝𑀺𑀬𑀸𑀮𑁄𑀓𑀁 𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀫𑁆𑀧𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀫𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀗𑁆𑀓𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀧𑀚𑀸𑀦𑀸𑀣𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀺𑀬𑀸—  “𑀲𑀢𑁆𑀣𑁄 𑀓𑀺𑀭 𑀲𑀼𑀗𑁆𑀓𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀫𑀕𑁆𑀕𑁂 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀓𑀫𑁆𑀫𑀺𑀬𑀸 𑀢𑀁 𑀲𑀢𑁆𑀣𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀕𑀘𑁆𑀙𑀲𑀻”𑀢𑀺? 𑀧𑀮𑀺𑀩𑀼𑀦𑁆𑀥𑁂𑀢𑁆𑀯𑀸 𑀫𑀼𑀜𑁆𑀘𑀺𑀁𑀲𑀼𑁇

1224\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀚𑀸𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀚𑀸𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀚𑀸𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

(408.)

1225\. **“𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀚𑀸𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂𑀦** **𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀺, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”**𑀢𑀺𑁇 (*𑁬𑁬*:**𑁧𑁧𑁫**)

(409.)

1226\. **𑀬𑁄 𑀧𑀦𑀸**𑀢𑀺 𑀬𑁄 𑀬𑀸𑀤𑀺𑀲𑁄…𑀧𑁂…  **𑀪𑀺𑀓𑁆𑀔𑀽**𑀢𑀺…𑀧𑁂…  𑀅𑀬𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀢𑁆𑀣𑁂 𑀅𑀥𑀺𑀧𑁆𑀧𑁂𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺𑁇

1227\. **𑀚𑀸𑀦𑀸𑀢𑀺** 𑀦𑀸𑀫 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺𑁇

1228\. **𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁄** 𑀦𑀸𑀫 𑀘𑁄𑀭𑀸 𑀓𑀢𑀓𑀫𑁆𑀫𑀸 𑀯𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀓𑀢𑀓𑀫𑁆𑀫𑀸 𑀯𑀸, 𑀭𑀸𑀚𑀸𑀦𑀁 𑀯𑀸 𑀣𑁂𑀬𑁆𑀬𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀼𑀗𑁆𑀓𑀁 𑀯𑀸 𑀧𑀭𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇

1229\. **𑀲𑀤𑁆𑀥𑀺𑀦𑁆**𑀢𑀺 𑀏𑀓𑀢𑁄𑁇

1230\. **𑀲𑀁𑀯𑀺𑀥𑀸𑀬𑀸**𑀢𑀺—  “𑀕𑀘𑁆𑀙𑀸𑀫𑀸𑀯𑀼𑀲𑁄, 𑀕𑀘𑁆𑀙𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂; 𑀕𑀘𑁆𑀙𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀘𑁆𑀙𑀸𑀫𑀸𑀯𑀼𑀲𑁄, 𑀅𑀚𑁆𑀚 𑀯𑀸 𑀳𑀺𑀬𑁆𑀬𑁄 𑀯𑀸 𑀧𑀭𑁂 𑀯𑀸 𑀕𑀘𑁆𑀙𑀸𑀫𑀸”𑀢𑀺 𑀲𑀁𑀯𑀺𑀤𑀳𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1231\. **𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀻**𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑀫𑁆𑀧𑀸𑀢𑁂 𑀕𑀸𑀫𑁂 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀅𑀕𑀸𑀫𑀓𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑁂 𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇

(410.)

1232\. 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂 𑀅𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀯𑀺𑀤𑀳𑀢𑀺, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦 𑀲𑀁𑀯𑀺𑀤𑀳𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑁂 𑀅𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀜𑁆𑀜𑀻 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

(411.)

1233\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀁𑀯𑀺𑀤𑀳𑀦𑁆𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀲𑀁𑀯𑀺𑀤𑀳𑀢𑀺, 𑀯𑀺𑀲𑀗𑁆𑀓𑁂𑀢𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
