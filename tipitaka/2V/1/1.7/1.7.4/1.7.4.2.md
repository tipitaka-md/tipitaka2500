# 𑁧𑁇𑁭𑁇𑁪𑁇𑁨 𑀧𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀯𑀕𑁆𑀕

### 𑀧𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(607.)

1840\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀓𑀺𑀭𑀦𑁆𑀢𑁂𑀧𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀧𑀺 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1841\. **“𑀧𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁩𑁨*:**𑁧𑁭𑁭**)

1842\. 𑀧𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1843\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘…𑀧𑁂… 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1844\. 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
