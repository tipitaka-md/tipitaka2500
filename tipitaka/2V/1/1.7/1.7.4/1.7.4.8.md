# 𑁧𑁇𑁭𑁇𑁪𑁇𑁮 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀯𑀕𑁆𑀕

### 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀻𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(614.)

1874\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀻 𑀧𑀭𑁂𑀲𑀁 𑀧𑀢𑁆𑀢𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1875\. **“𑀦 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀻 𑀧𑀭𑁂𑀲𑀁 𑀧𑀢𑁆𑀢𑀁 𑀑𑀮𑁄𑀓𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁩𑁮*:**𑁧𑁮𑁩**)

1876\. 𑀦 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀺𑀦𑀸 𑀧𑀭𑁂𑀲𑀁 𑀧𑀢𑁆𑀢𑁄 𑀑𑀮𑁄𑀓𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀻 𑀧𑀭𑁂𑀲𑀁 𑀧𑀢𑁆𑀢𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1877\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, “𑀤𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀸 “𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀸 𑀑𑀮𑁄𑀓𑁂𑀢𑀺, 𑀦 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1878\. 𑀅𑀝𑁆𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
