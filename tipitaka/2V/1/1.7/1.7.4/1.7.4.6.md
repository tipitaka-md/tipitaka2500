# 𑁧𑁇𑁭𑁇𑁪𑁇𑁬 𑀑𑀤𑀦𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀯𑀕𑁆𑀕

* 𑀑𑀤𑀦𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(611.)

1860\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀽𑀧𑀫𑁆𑀧𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀫𑁆𑀧𑀺 𑀑𑀤𑀦𑁂𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬…𑀧𑁂… 𑁇

1861\. **“𑀦 𑀲𑀽𑀧𑀁 𑀯𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀯𑀸 𑀑𑀤𑀦𑁂𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁩𑁬*:**𑁧𑁮𑁧**)

1862\. 𑀦 𑀲𑀽𑀧𑀁 𑀯𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀯𑀸 𑀑𑀤𑀦𑁂𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀽𑀧𑀁 𑀯𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀯𑀸 𑀑𑀤𑀦𑁂𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1863\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀲𑀸𑀫𑀺𑀓𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀪𑀺𑀬𑁆𑀬𑁄𑀓𑀫𑁆𑀬𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1864\. 𑀙𑀝𑁆𑀞𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
