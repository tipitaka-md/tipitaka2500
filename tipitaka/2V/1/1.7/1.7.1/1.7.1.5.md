# 𑁧𑁇𑁭𑁇𑁧𑁇𑁫 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(580.)

1705\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀧𑀸𑀤𑀫𑁆𑀧𑀺 𑀓𑀻𑀍𑀆𑀧𑁂𑀦𑁆𑀢𑀸 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1706\. **“𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁫*:**𑁧𑁫𑁦**)

1707\. 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀢𑁆𑀣𑀁 𑀯𑀸 𑀧𑀸𑀤𑀁 𑀯𑀸 𑀓𑀻𑀍𑀆𑀧𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1708\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1709\. 𑀧𑀜𑁆𑀘𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
