# 𑁧𑁇𑁭𑁇𑁧𑁇𑁯 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

### 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(584.)

1725\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1726\. **“𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁯*:**𑁧𑁫𑁪**)

1727\. 𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀏𑀓𑀢𑁄 𑀯𑀸 𑀉𑀪𑀢𑁄 𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1728\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇
