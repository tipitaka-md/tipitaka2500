# 𑁧𑁇𑁭𑁇𑁨𑁇𑁧 𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(586.)

1735. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀸𑀳𑀲𑀺𑀢𑀁 𑀳𑀲𑀦𑁆𑀢𑀸 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1736. **“𑀦 𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀺𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁧𑁧*:**𑁧𑁫𑁬**)

1737. 𑀦 𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀺𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀳𑀸𑀳𑀲𑀺𑀢𑀁 𑀳𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1738. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀳𑀲𑀦𑀻𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀫𑀺𑀳𑀺𑀢𑀫𑀢𑁆𑀢𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---
