# 𑁧𑁇𑁭𑁇𑁭𑁇𑁧𑁪 𑀳𑀭𑀺𑀢𑁂𑀉𑀘𑁆𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕

### 𑀳𑀭𑀺𑀢𑁂𑀉𑀘𑁆𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(652.)

2077. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑀭𑀺𑀢𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

2078. **“𑀦 𑀳𑀭𑀺𑀢𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁭𑁪*:**𑁨𑁧𑁯**)

2079. 𑀦 𑀳𑀭𑀺𑀢𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀉𑀘𑁆𑀘𑀸𑀭𑁄 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑁄 𑀯𑀸 𑀔𑁂𑀍𑀑 𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀭𑀺𑀢𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2080. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑁂 𑀓𑀢𑁄 𑀳𑀭𑀺𑀢𑀁 𑀑𑀢𑁆𑀣𑀭𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2081. 𑀘𑀼𑀤𑁆𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
