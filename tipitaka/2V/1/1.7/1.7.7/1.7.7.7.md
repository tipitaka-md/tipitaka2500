# 𑁧𑁇𑁭𑁇𑁭𑁇𑁭 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕

### 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(644.)

2034\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀻𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

2035\. **“𑀦 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀻𑀲𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁬𑁭*:**𑁨𑁧𑁨**)

2036\. **𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀻𑀲𑁄** 𑀦𑀸𑀫 𑀲𑀲𑀻𑀲𑀁 𑀧𑀸𑀭𑀼𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

2037\. 𑀦 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀻𑀲𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀻𑀲𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2038\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀲𑀻𑀲𑀁 𑀯𑀺𑀯𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---
