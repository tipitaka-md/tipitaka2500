# 𑁧𑁇𑁭𑁇𑁭𑁇𑁧𑁫 𑀉𑀤𑀓𑁂𑀉𑀘𑁆𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕

* 𑀉𑀤𑀓𑁂𑀉𑀘𑁆𑀘𑀸𑀭𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(653.)

2082\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇

2083\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀣𑀸”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

2084\. **“𑀦 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇

2085\. 𑀏𑀯𑀜𑁆𑀘𑀺𑀤𑀁 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇

(654.)

2086\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀸𑀢𑀼𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

2087\. **“𑀦 𑀉𑀤𑀓𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁭𑁫*:**𑁨𑁨𑁦**)

2088\. 𑀦 𑀉𑀤𑀓𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀉𑀘𑁆𑀘𑀸𑀭𑁄 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑁄 𑀯𑀸 𑀔𑁂𑀍𑀑 𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2089\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀣𑀮𑁂 𑀓𑀢𑁄 𑀉𑀤𑀓𑀁 𑀑𑀢𑁆𑀣𑀭𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2090\. 𑀧𑀦𑁆𑀦𑀭𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇

  
( )



2091\. 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀲𑁂𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀢𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺—  “𑀓𑀘𑁆𑀘𑀺𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸”? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀢𑁆𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻𑀢𑀺𑁇

---

2092\. 𑀲𑁂𑀔𑀺𑀬𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇

  
𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
