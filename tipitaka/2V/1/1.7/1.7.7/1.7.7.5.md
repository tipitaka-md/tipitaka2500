# 𑁧𑁇𑁭𑁇𑁭𑁇𑁫 𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(642.)

2023. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

2024. **“𑀦 𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁬𑁫*:**𑁨𑁧𑁦**)

2025. 𑀦 𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀢𑁆𑀣𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2026. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2027. 𑀧𑀜𑁆𑀘𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
