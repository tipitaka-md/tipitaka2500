# 𑁧𑁇𑁭𑁇𑁫𑁇𑁭 𑀳𑀢𑁆𑀣𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀓𑀩𑀍𑀅𑀯𑀕𑁆𑀕

* 𑀳𑀢𑁆𑀣𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(623.)

1919\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑀢𑁆𑀣𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1920\. **“𑀦 𑀳𑀢𑁆𑀣𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁪𑁭*:**𑁧𑁯𑁨**)

1921\. 𑀦 𑀳𑀢𑁆𑀣𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀢𑁆𑀣𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1922\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀓𑀘𑀯𑀭𑀁 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑁄 𑀳𑀢𑁆𑀣𑀁 𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀸𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1923\. 𑀲𑀢𑁆𑀢𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
