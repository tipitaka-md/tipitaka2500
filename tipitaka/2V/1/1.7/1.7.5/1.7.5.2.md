# 𑁧𑁇𑁭𑁇𑁫𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀅𑀦𑀸𑀳𑀝𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀓𑀩𑀍𑀅𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀅𑀦𑀸𑀳𑀝𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(618.)

1894\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸 𑀲𑀩𑁆𑀩𑀁 𑀳𑀢𑁆𑀣𑀁 𑀫𑀼𑀔𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1895\. **“𑀦 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑁄 𑀲𑀩𑁆𑀩𑀁 𑀳𑀢𑁆𑀣𑀁 𑀫𑀼𑀔𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁪𑁨*:**𑁧𑁮𑁭**)

1896\. 𑀦 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑁂𑀦 𑀲𑀩𑁆𑀩𑁄 𑀳𑀢𑁆𑀣𑁄 𑀫𑀼𑀔𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑁄 𑀲𑀩𑁆𑀩𑀁 𑀳𑀢𑁆𑀣𑀁 𑀫𑀼𑀔𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1897\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘…𑀧𑁂… 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1898\. 𑀤𑀼𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
