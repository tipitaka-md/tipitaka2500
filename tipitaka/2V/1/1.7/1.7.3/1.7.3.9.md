# 𑁧𑁇𑁭𑁇𑁩𑁇𑁯 𑀲𑀫𑀲𑀽𑀧𑀓𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀔𑀫𑁆𑀪𑀓𑀢𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀲𑀽𑀧𑀓𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(604.)

1825\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀸 𑀲𑀽𑀧𑀜𑁆𑀜𑁂𑀯 𑀩𑀳𑀼𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1826\. **“𑀲𑀫𑀲𑀽𑀧𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁨𑁯*:**𑁧𑁭𑁪**)

1827\. **𑀲𑀽𑀧𑁄** 𑀦𑀸𑀫 𑀤𑁆𑀯𑁂 𑀲𑀽𑀧𑀸— 𑀫𑀼𑀕𑁆𑀕𑀲𑀽𑀧𑁄, 𑀫𑀸𑀲𑀲𑀽𑀧𑁄𑁇 𑀳𑀢𑁆𑀣𑀳𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀲𑀽𑀧𑀓𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀽𑀧𑀜𑁆𑀜𑁂𑀯 𑀩𑀳𑀼𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1828\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀭𑀲𑀭𑀲𑁂, 𑀜𑀸𑀢𑀓𑀸𑀦𑀁 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀥𑀦𑁂𑀦, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1829\. 𑀦𑀯𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
