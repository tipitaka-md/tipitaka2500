# 𑁧𑁇𑁭𑁇𑁬𑁇𑁩 𑀧𑀢𑁆𑀢𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕

* 𑀧𑀢𑁆𑀢𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(629.)

1950\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀢𑁆𑀢𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1951\. **“𑀦 𑀧𑀢𑁆𑀢𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁫𑁩*:**𑁧𑁯𑁮**)

1952\. 𑀦 𑀧𑀢𑁆𑀢𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀢𑁆𑀢𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1953\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, 𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑁂 𑀲𑁂𑀲𑁂 𑀏𑀓𑀢𑁄 𑀲𑀗𑁆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1954\. 𑀢𑀢𑀺𑀬𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
