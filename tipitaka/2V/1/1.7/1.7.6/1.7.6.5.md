# 𑁧𑁇𑁭𑁇𑁬𑁇𑁫 𑀲𑀸𑀫𑀺𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀫𑀺𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(631.)

1960\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀪𑀕𑁆𑀕𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀕𑀺𑀭𑁂 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀓𑀦𑀤𑁂 𑀧𑀸𑀲𑀸𑀤𑁂 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺— “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇

1961\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂… 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺— “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺…𑀧𑁂… “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂… 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂… 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1962\. **“𑀦 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁫𑁫*:**𑁨𑁦𑁦**)

1963\. 𑀦 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1964\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺— 𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘, 𑀅𑀲𑁆𑀲𑀢𑀺𑀬𑀸, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲, “𑀥𑁄𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀸 “𑀥𑁄𑀯𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀤𑀸𑀲𑀼, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1965\. 𑀧𑀜𑁆𑀘𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
