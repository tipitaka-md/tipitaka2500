# 𑁧𑁇𑁭𑁇𑁬𑁇𑁧𑁦 𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

### 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕

### 𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(637.)

1996. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺…𑀧𑁂… 𑁇

1997. **“𑀦 𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁬𑁦*:**𑁨𑁦𑁫**)

1998. **𑀆𑀯𑀼𑀥𑀁** 𑀦𑀸𑀫 𑀘𑀸𑀧𑁄 𑀓𑁄𑀤𑀡𑁆𑀟𑁄𑁇

1999. 𑀦 𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

2000. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

2001. 𑀤𑀲𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
