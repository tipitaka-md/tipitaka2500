# 𑁧𑁇𑁭𑁇𑁬𑁇𑁧 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀓𑀸𑀭𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

* 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀓𑀸𑀭𑀓𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤

(627.)

1939\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀦 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀬𑁄𑀧𑀸𑀦𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀓𑀸𑀭𑀓𑀁 𑀔𑀻𑀭𑀁 𑀧𑀺𑀯𑀦𑁆𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀦𑀝𑀧𑀼𑀩𑁆𑀩𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀫𑀸𑀳—  “𑀲𑀩𑁆𑀩𑁄𑀬𑀁 𑀫𑀜𑁆𑀜𑁂 𑀲𑀁𑀖𑁄 𑀲𑀻𑀢𑀻𑀓𑀢𑁄”𑀢𑀺𑁇

1940\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀯𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀁𑀖𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀯𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀲𑀁𑀖𑀁 𑀆𑀭𑀩𑁆𑀪 𑀤𑀯𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀸 𑀲𑀁𑀖𑀁 𑀯𑀸 𑀆𑀭𑀩𑁆𑀪 𑀤𑀯𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀩𑁆𑀪𑀭𑀢𑀸𑀬…𑀧𑁂…  𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀣—

1941\. **“𑀦 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀓𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀭𑀡𑀻𑀬𑀸”**𑀢𑀺𑁇 (*𑁫𑁧*:**𑁧𑁯𑁬**)

1942\. 𑀦 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀓𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀓𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1943\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺—  𑀅𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

---

1944\. 𑀧𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
