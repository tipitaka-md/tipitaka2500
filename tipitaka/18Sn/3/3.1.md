# 𑁩𑁇𑁧 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀲𑀼𑀢𑁆𑀢

(408.)

472. _“𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀣𑀸 𑀧𑀩𑁆𑀩𑀚𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀬𑀣𑀸 𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀲𑀫𑀭𑁄𑀘𑀬𑀺𑁇 (𑁧)_  


(409.)

473. _𑀲𑀫𑁆𑀩𑀸𑀥𑁄𑀬𑀁 𑀖𑀭𑀸𑀯𑀸𑀲𑁄,_  
_𑀭𑀚𑀲𑁆𑀲𑀸𑀬𑀢𑀦𑀁 𑀇𑀢𑀺;_  
_𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁄𑀯 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸,_  
_𑀇𑀢𑀺 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀧𑀩𑁆𑀩𑀚𑀺𑁇 (𑁨)_  


(410.)

474. _𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀓𑀸𑀬𑁂𑀦,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑀬𑀺;_  
_𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀆𑀚𑀻𑀯𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑀬𑀺𑁇 (𑁩)_  


(411.)

475. _𑀅𑀕𑀫𑀸 𑀭𑀸𑀚𑀕𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀫𑀕𑀥𑀸𑀦𑀁 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀁;_  
_𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀅𑀪𑀺𑀳𑀸𑀭𑁂𑀲𑀺,_  
_𑀆𑀓𑀺𑀡𑁆𑀡𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄”𑁇 (𑁪)_  


(412.)

476. _𑀢𑀫𑀤𑁆𑀤𑀲𑀸 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄,_  
_𑀧𑀸𑀲𑀸𑀤𑀲𑁆𑀫𑀺𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀮𑀓𑁆𑀔𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀇𑀫𑀫𑀢𑁆𑀣𑀁 𑀅𑀪𑀸𑀲𑀣𑁇 (𑁫)_  


(413.)

477. _“𑀇𑀫𑀁 𑀪𑁄𑀦𑁆𑀢𑁄 𑀦𑀺𑀲𑀸𑀫𑁂𑀣,_  
_𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀩𑁆𑀭𑀳𑀸 𑀲𑀼𑀘𑀺;_  
_𑀘𑀭𑀡𑁂𑀦 𑀘 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀬𑀼𑀕𑀫𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑁂𑀓𑁆𑀔𑀢𑀺𑁇 (𑁬)_  


(414.)

478. _𑀑𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀓𑁆𑀔𑀼 𑀲𑀢𑀺𑀫𑀸,_  
_𑀦𑀸𑀬𑀁 𑀦𑀻𑀘𑀓𑀼𑀮𑀸𑀫𑀺𑀯;_  
_𑀭𑀸𑀚𑀤𑀽𑀢𑀸𑀪𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀼,_  
_𑀓𑀼𑀳𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺”𑁇 (𑁭)_  


(415.)

479. _𑀢𑁂 𑀧𑁂𑀲𑀺𑀢𑀸 𑀭𑀸𑀚𑀤𑀽𑀢𑀸,_  
_𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀲𑀼𑀁;_  
_𑀓𑀼𑀳𑀺𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀓𑀢𑁆𑀣 𑀯𑀸𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 (𑁮)_  


(416.)

480. _𑀲𑀧𑀤𑀸𑀦𑀁 𑀘𑀭𑀫𑀸𑀦𑁄,_  
_𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁄 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑁄;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀢𑁆𑀢𑀁 𑀅𑀧𑀽𑀭𑁂𑀲𑀺,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄𑁇 (𑁯)_  


(417.)

481. _𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫 𑀦𑀕𑀭𑀸 𑀫𑀼𑀦𑀺;_  
_𑀧𑀡𑁆𑀟𑀯𑀁 𑀅𑀪𑀺𑀳𑀸𑀭𑁂𑀲𑀺,_  
_𑀏𑀢𑁆𑀣 𑀯𑀸𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 (𑁧𑁦)_  


(418.)

482. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀯𑀸𑀲𑀽𑀧𑀕𑀢𑀁,_  
_𑀢𑀬𑁄 𑀤𑀽𑀢𑀸 𑀉𑀧𑀸𑀯𑀺𑀲𑀼𑀁;_  
_𑀢𑁂𑀲𑀼 𑀏𑀓𑁄𑀯 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀭𑀸𑀚𑀺𑀦𑁄 𑀧𑀝𑀺𑀯𑁂𑀤𑀬𑀺𑁇 (𑁧𑁧)_  


(419.)

483. _“𑀏𑀲 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀧𑀡𑁆𑀟𑀯𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁆𑀣𑀢𑁄;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀩𑁆𑀬𑀕𑁆𑀖𑀼𑀲𑀪𑁄𑀯,_  
_𑀲𑀻𑀳𑁄𑀯 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑁂”𑁇 (𑁧𑁨)_  


(420.)

484. _𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀤𑀽𑀢𑀯𑀘𑀦𑀁,_  
_𑀪𑀤𑁆𑀤𑀬𑀸𑀦𑁂𑀦 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑁄 𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺,_  
_𑀬𑁂𑀦 𑀧𑀡𑁆𑀟𑀯𑀧𑀩𑁆𑀩𑀢𑁄𑁇 (𑁧𑁩)_  


(421.)

485. _𑀲 𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀬𑀸𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀦𑀸 𑀑𑀭𑀼𑀬𑁆𑀳 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀆𑀲𑀚𑁆𑀚 𑀦𑀁 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑁇 (𑁧𑁪)_  


(422.)

486. _𑀦𑀺𑀲𑀚𑁆𑀚 𑀭𑀸𑀚𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀺,_  
_𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀢𑀢𑁄;_  
_𑀓𑀣𑀁 𑀲𑁄 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀫𑀢𑁆𑀣𑀁 𑀅𑀪𑀸𑀲𑀣𑁇 (𑁧𑁫)_  


(423.)

487. _“𑀬𑀼𑀯𑀸 𑀘 𑀤𑀳𑀭𑁄 𑀘𑀸𑀲𑀺,_  
_𑀧𑀞𑀫𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀲𑀼𑀲𑀼;_  
_𑀯𑀡𑁆𑀡𑀸𑀭𑁄𑀳𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀚𑀸𑀢𑀺𑀫𑀸 𑀯𑀺𑀬 𑀔𑀢𑁆𑀢𑀺𑀬𑁄𑁇 (𑁧𑁬)_  


(424.)

488. _𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑁄 𑀅𑀦𑀻𑀓𑀕𑁆𑀕𑀁,_  
_𑀦𑀸𑀕𑀲𑀗𑁆𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀤𑀤𑀸𑀫𑀺 𑀪𑁄𑀕𑁂 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼,_  
_𑀚𑀸𑀢𑀺𑀁 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄”𑁇 (𑁧𑁭)_  


(425.)

489. _“𑀉𑀚𑀼𑀁 𑀚𑀦𑀧𑀤𑁄 𑀭𑀸𑀚,_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀢𑁄;_  
_𑀥𑀦𑀯𑀺𑀭𑀺𑀬𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀦𑀺𑀓𑁂𑀢𑀺𑀦𑁄𑁇 (𑁧𑁮)_  


(426.)

490. _𑀆𑀤𑀺𑀘𑁆𑀘𑀸 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀸𑀓𑀺𑀬𑀸 𑀦𑀸𑀫 𑀚𑀸𑀢𑀺𑀬𑀸;_  
_𑀢𑀫𑁆𑀳𑀸 𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀦 𑀓𑀸𑀫𑁂 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀁𑁇 (𑁧𑁯)_  


(427.)

491. _𑀓𑀸𑀫𑁂𑀲𑁆𑀯𑀸𑀤𑀻𑀦𑀯𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀤𑀝𑁆𑀞𑀼 𑀔𑁂𑀫𑀢𑁄;_  
_𑀧𑀥𑀸𑀦𑀸𑀬 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀏𑀢𑁆𑀣 𑀫𑁂 𑀭𑀜𑁆𑀚𑀢𑀻 𑀫𑀦𑁄”𑀢𑀺𑁇 (𑁨𑁦)_  


---

492. 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀲𑀼𑀢𑁆𑀢𑀁 𑀧𑀞𑀫𑀁𑁇
