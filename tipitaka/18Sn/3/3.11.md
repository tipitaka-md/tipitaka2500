# 𑁩𑁇𑁧𑁧 𑀦𑀸𑀮𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀦𑀸𑀮𑀓𑀲𑀼𑀢𑁆𑀢

(685.)

817\. _𑀆𑀦𑀦𑁆𑀤𑀚𑀸𑀢𑁂 𑀢𑀺𑀤𑀲𑀕𑀡𑁂 𑀧𑀢𑀻𑀢𑁂,_  
_𑀲𑀓𑁆𑀓𑀜𑁆𑀘 𑀇𑀦𑁆𑀤𑀁 𑀲𑀼𑀘𑀺𑀯𑀲𑀦𑁂 𑀘 𑀤𑁂𑀯𑁂;_  
_𑀤𑀼𑀲𑁆𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀭𑀺𑀯 𑀣𑁄𑀫𑀬𑀦𑁆𑀢𑁂,_  
_𑀅𑀲𑀺𑀢𑁄 𑀇𑀲𑀺 𑀅𑀤𑁆𑀤𑀲 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑁂𑁇 (𑁧)_  


(686.)

818\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀤𑁂𑀯𑁂 𑀫𑀼𑀤𑀺𑀢𑀫𑀦𑁂 𑀉𑀤𑀕𑁆𑀕𑁂,_  
_𑀘𑀺𑀢𑁆𑀢𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦 𑀇𑀤𑀫𑀯𑁄𑀘 𑀢𑀢𑁆𑀣;_  
_“𑀓𑀺𑀁 𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑁄 𑀅𑀢𑀺𑀭𑀺𑀯 𑀓𑀮𑁆𑀬𑀭𑀽𑀧𑁄,_  
_𑀤𑀼𑀲𑁆𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀭𑀫𑀬𑀣 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑁇 (𑁨)_  


(687.)

819\. _𑀬𑀤𑀸𑀧𑀺 𑀆𑀲𑀻 𑀅𑀲𑀼𑀭𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀫𑁄,_  
_𑀚𑀬𑁄 𑀲𑀼𑀭𑀸𑀦𑀁 𑀅𑀲𑀼𑀭𑀸 𑀧𑀭𑀸𑀚𑀺𑀢𑀸;_  
_𑀢𑀤𑀸𑀧𑀺 𑀦𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑁄,_  
_𑀓𑀺𑀫𑀩𑁆𑀪𑀼𑀢𑀁 𑀤𑀝𑁆𑀞𑀼 𑀫𑀭𑀽 𑀧𑀫𑁄𑀤𑀺𑀢𑀸𑁇 (𑁩)_  


(688.)

820\. _𑀲𑁂𑀍𑀏𑀦𑁆𑀢𑀺 𑀕𑀸𑀬𑀦𑁆𑀢𑀺 𑀘 𑀯𑀸𑀤𑀬𑀦𑁆𑀢𑀺 𑀘,_  
_𑀪𑀼𑀚𑀸𑀦𑀺 𑀨𑁄𑀝𑁂𑀦𑁆𑀢𑀺 𑀘 𑀦𑀘𑁆𑀘𑀬𑀦𑁆𑀢𑀺 𑀘;_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀫𑁂𑀭𑀼𑀫𑀼𑀤𑁆𑀥𑀯𑀸𑀲𑀺𑀦𑁂,_  
_𑀥𑀼𑀦𑀸𑀣 𑀫𑁂 𑀲𑀁𑀲𑀬𑀁 𑀔𑀺𑀧𑁆𑀧 𑀫𑀸𑀭𑀺𑀲𑀸”𑁇 (𑁪)_  


(689.)

821\. _“𑀲𑁄 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀭𑀢𑀦𑀯𑀭𑁄 𑀅𑀢𑀼𑀮𑁆𑀬𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂 𑀳𑀺𑀢𑀲𑀼𑀔𑀢𑁆𑀣𑀸𑀬 𑀚𑀸𑀢𑁄;_  
_𑀲𑀓𑁆𑀬𑀸𑀦 𑀕𑀸𑀫𑁂 𑀚𑀦𑀧𑀤𑁂 𑀮𑀼𑀫𑁆𑀩𑀺𑀦𑁂𑀬𑁆𑀬𑁂,_  
_𑀢𑁂𑀦𑀫𑁆𑀳 𑀢𑀼𑀝𑁆𑀞𑀸 𑀅𑀢𑀺𑀭𑀺𑀯 𑀓𑀮𑁆𑀬𑀭𑀽𑀧𑀸𑁇 (𑁫)_  


(690.)

822\. _𑀲𑁄 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀼𑀢𑁆𑀢𑀫𑁄 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄,_  
_𑀦𑀭𑀸𑀲𑀪𑁄 𑀲𑀩𑁆𑀩𑀧𑀚𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀯𑀢𑁆𑀢𑁂𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀓𑀫𑀺𑀲𑀺𑀯𑁆𑀳𑀬𑁂 𑀯𑀦𑁂,_  
_𑀦𑀤𑀁𑀯 𑀲𑀻𑀳𑁄 𑀩𑀮𑀯𑀸 𑀫𑀺𑀕𑀸𑀪𑀺𑀪𑀽”𑁇 (𑁬)_  


(691.)

823\. _𑀢𑀁 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀼𑀭𑀺𑀢𑀫𑀯𑀲𑀭𑀻 𑀲𑁄,_  
_𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑀲𑁆𑀲 𑀢𑀤 𑀪𑀯𑀦𑀁 𑀉𑀧𑀸𑀯𑀺𑀲𑀺;_  
_𑀦𑀺𑀲𑀚𑁆𑀚 𑀢𑀢𑁆𑀣 𑀇𑀤𑀫𑀯𑁄𑀘𑀸𑀲𑀺 𑀲𑀓𑁆𑀬𑁂,_  
_“𑀓𑀼𑀳𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀳𑀫𑀧𑀺 𑀤𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑁄”𑁇 (𑁭)_  


(692.)

824\. _𑀢𑀢𑁄 𑀓𑀼𑀫𑀸𑀭𑀁 𑀚𑀮𑀺𑀢𑀫𑀺𑀯 𑀲𑀼𑀯𑀡𑁆𑀡𑀁,_  
_𑀉𑀓𑁆𑀓𑀸𑀫𑀼𑀔𑁂𑀯 𑀲𑀼𑀓𑀼𑀲𑀮𑀲𑀫𑁆𑀧𑀳𑀝𑁆𑀞𑀁;_  
_𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀫𑀸𑀦𑀁 𑀲𑀺𑀭𑀺𑀬𑀸 𑀅𑀦𑁄𑀫𑀯𑀡𑁆𑀡𑀁,_  
_𑀤𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀲𑀺𑀢𑀯𑁆𑀳𑀬𑀲𑁆𑀲 𑀲𑀓𑁆𑀬𑀸𑁇 (𑁮)_  


(693.)

825\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀼𑀫𑀸𑀭𑀁 𑀲𑀺𑀔𑀺𑀫𑀺𑀯 𑀧𑀚𑁆𑀚𑀮𑀦𑁆𑀢𑀁,_  
_𑀢𑀸𑀭𑀸𑀲𑀪𑀁𑀯 𑀦𑀪𑀲𑀺𑀕𑀫𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁;_  
_𑀲𑀼𑀭𑀺𑀬𑀁 𑀢𑀧𑀦𑁆𑀢𑀁 𑀲𑀭𑀤𑀭𑀺𑀯𑀩𑁆𑀪𑀫𑀼𑀢𑁆𑀢𑀁,_  
_𑀆𑀦𑀦𑁆𑀤𑀚𑀸𑀢𑁄 𑀯𑀺𑀧𑀼𑀮𑀫𑀮𑀢𑁆𑀣 𑀧𑀻𑀢𑀺𑀁𑁇 (𑁯)_  


(694.)

826\. _𑀅𑀦𑁂𑀓𑀲𑀸𑀔𑀜𑁆𑀘 𑀲𑀳𑀲𑁆𑀲𑀫𑀡𑁆𑀟𑀮𑀁,_  
_𑀙𑀢𑁆𑀢𑀁 𑀫𑀭𑀽 𑀥𑀸𑀭𑀬𑀼𑀫𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂;_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀡𑁆𑀟𑀸 𑀯𑀻𑀢𑀺𑀧𑀢𑀦𑁆𑀢𑀺 𑀘𑀸𑀫𑀭𑀸,_  
_𑀦 𑀤𑀺𑀲𑁆𑀲𑀭𑁂 𑀘𑀸𑀫𑀭𑀙𑀢𑁆𑀢𑀕𑀸𑀳𑀓𑀸𑁇 (𑁧𑁦)_  


(695.)

827\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀚𑀝𑀻 𑀓𑀡𑁆𑀳𑀲𑀺𑀭𑀺𑀯𑁆𑀳𑀬𑁄 𑀇𑀲𑀺,_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑀺𑀓𑁆𑀔𑀁 𑀯𑀺𑀬 𑀧𑀡𑁆𑀟𑀼𑀓𑀫𑁆𑀩𑀮𑁂;_  
_𑀲𑁂𑀢𑀜𑁆𑀘 𑀙𑀢𑁆𑀢𑀁 𑀥𑀭𑀺𑀬𑀦𑁆𑀢 𑀫𑀼𑀤𑁆𑀥𑀦𑀺,_  
_𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑁇 (𑁧𑁧)_  


(696.)

828\. _𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀦 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁,_  
_𑀚𑀺𑀕𑀻𑀲𑀓𑁄 𑀮𑀓𑁆𑀔𑀡𑀫𑀦𑁆𑀢𑀧𑀸𑀭𑀕𑀽;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀺𑀭𑀫𑀩𑁆𑀪𑀼𑀤𑀻𑀭𑀬𑀺,_  
_“𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸𑀬𑀁 𑀤𑁆𑀯𑀺𑀧𑀤𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑁄”𑁇 (𑁧𑁨)_  


(697.)

829\. _𑀅𑀣𑀢𑁆𑀢𑀦𑁄 𑀕𑀫𑀦𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑁄,_  
_𑀅𑀓𑀮𑁆𑀬𑀭𑀽𑀧𑁄 𑀕𑀍𑀅𑀬𑀢𑀺 𑀅𑀲𑁆𑀲𑀼𑀓𑀸𑀦𑀺;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀓𑁆𑀬𑀸 𑀇𑀲𑀺𑀫𑀯𑁄𑀘𑀼𑀁 𑀭𑀼𑀤𑀦𑁆𑀢𑀁,_  
_“𑀦𑁄 𑀘𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄”𑁇 (𑁧𑁩)_  


(698.)

830\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀓𑁆𑀬𑁂 𑀇𑀲𑀺𑀫𑀯𑁄𑀘 𑀅𑀓𑀮𑁆𑀬𑁂,_  
_“𑀦𑀸𑀳𑀁 𑀓𑀼𑀫𑀸𑀭𑁂 𑀅𑀳𑀺𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺;_  
_𑀦 𑀘𑀸𑀧𑀺𑀫𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄,_  
_𑀦 𑀑𑀭𑀓𑀸𑀬𑀁 𑀅𑀥𑀺𑀫𑀸𑀦𑀲𑀸 𑀪𑀯𑀸𑀣𑁇 (𑁧𑁪)_  


(699.)

831\. _𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀬𑀕𑁆𑀕𑀁 𑀨𑀼𑀲𑀺𑀲𑁆𑀲𑀢𑀸𑀬𑀁 𑀓𑀼𑀫𑀸𑀭𑁄,_  
_𑀲𑁄 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀭𑀫𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀤𑀲𑁆𑀲𑀻;_  
_𑀯𑀢𑁆𑀢𑁂𑀲𑁆𑀲𑀢𑀸𑀬𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻,_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁𑁇 (𑁧𑁫)_  


(700.)

832\. _𑀫𑀫𑀜𑁆𑀘 𑀆𑀬𑀼 𑀦 𑀘𑀺𑀭𑀫𑀺𑀥𑀸𑀯𑀲𑁂𑀲𑁄,_  
_𑀅𑀣𑀦𑁆𑀢𑀭𑀸 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀮𑀓𑀺𑀭𑀺𑀬𑀸;_  
_𑀲𑁄𑀳𑀁 𑀦 𑀲𑁄𑀲𑁆𑀲𑀁 𑀅𑀲𑀫𑀥𑀼𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁,_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀅𑀝𑁆𑀝𑁄 𑀩𑁆𑀬𑀲𑀦𑀗𑁆𑀕𑀢𑁄 𑀅𑀖𑀸𑀯𑀻”𑁇 (𑁧𑁬)_  


(701.)

833\. _𑀲𑁄 𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀚𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀻𑀢𑀺𑀁,_  
_𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀫𑁆𑀳𑀸 𑀦𑀺𑀕𑁆𑀕𑀫𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻;_  
_𑀲𑁄 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑀁 𑀲𑀬𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀫𑀸𑀦𑁄,_  
_𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀅𑀲𑀫𑀥𑀼𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑁇 (𑁧𑁭)_  


(702.)

834\. _“𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀖𑁄𑀲𑀁 𑀬𑀤 𑀧𑀭𑀢𑁄 𑀲𑀼𑀡𑀸𑀲𑀺,_  
_𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀢𑁆𑀢𑁄 𑀯𑀺𑀯𑀭𑀢𑀺 𑀥𑀫𑁆𑀫𑀫𑀕𑁆𑀕𑀁;_  
_𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀢𑀢𑁆𑀣 𑀲𑀫𑀬𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄,_  
_𑀘𑀭𑀲𑁆𑀲𑀼 𑀢𑀲𑁆𑀫𑀺𑀁 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁”𑁇 (𑁧𑁮)_  


(703.)

835\. _𑀢𑁂𑀦𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑁄 𑀳𑀺𑀢𑀫𑀦𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀸,_  
_𑀅𑀦𑀸𑀕𑀢𑁂 𑀧𑀭𑀫𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀤𑀲𑁆𑀲𑀺𑀦𑀸;_  
_𑀲𑁄 𑀦𑀸𑀮𑀓𑁄 𑀉𑀧𑀘𑀺𑀢𑀧𑀼𑀜𑁆𑀜𑀲𑀜𑁆𑀘𑀬𑁄,_  
_𑀚𑀺𑀦𑀁 𑀧𑀢𑀺𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀯𑀲𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄𑁇 (𑁧𑁯)_  


(704.)

836\. _𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀖𑁄𑀲𑀁 𑀚𑀺𑀦𑀯𑀭𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀦𑁂,_  
_𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀤𑀺𑀲𑁆𑀯𑀸 𑀇𑀲𑀺𑀦𑀺𑀲𑀪𑀁 𑀧𑀲𑀦𑁆𑀦𑁄;_  
_𑀫𑁄𑀦𑁂𑀬𑁆𑀬𑀲𑁂𑀝𑁆𑀞𑀁 𑀫𑀼𑀦𑀺𑀧𑀯𑀭𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺,_  
_𑀲𑀫𑀸𑀕𑀢𑁂 𑀅𑀲𑀺𑀢𑀸𑀯𑁆𑀳𑀬𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑀢𑀺𑁇 (𑁨𑁦)_  


---

837\. 𑀯𑀢𑁆𑀣𑀼𑀕𑀸𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇



(705.)

838\. _“𑀅𑀜𑁆𑀜𑀸𑀢𑀫𑁂𑀢𑀁 𑀯𑀘𑀦𑀁,_  
_𑀅𑀲𑀺𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀢𑀁 𑀢𑀁 𑀕𑁄𑀢𑀫 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀼𑀁𑁇 (𑁨𑁧)_  


(706.)

839\. _𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀼𑀧𑁂𑀢𑀲𑁆𑀲,_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀘𑀭𑀺𑀬𑀁 𑀚𑀺𑀕𑀻𑀲𑀢𑁄;_  
_𑀫𑀼𑀦𑀺 𑀧𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑁂 𑀧𑀼𑀝𑁆𑀞𑁄,_  
_𑀫𑁄𑀦𑁂𑀬𑁆𑀬𑀁 𑀉𑀢𑁆𑀢𑀫𑀁 𑀧𑀤𑀁”𑁇 (𑁨𑁨)_  


(707.)

840\. _“𑀫𑁄𑀦𑁂𑀬𑁆𑀬𑀁 𑀢𑁂 𑀉𑀧𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀁, __(𑀇𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀁;_  
_𑀳𑀦𑁆𑀤 𑀢𑁂 𑀦𑀁 𑀧𑀯𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀲𑀦𑁆𑀣𑀫𑁆𑀪𑀲𑁆𑀲𑀼 𑀤𑀍𑀳𑁄 𑀪𑀯𑁇 (𑁨𑁩)_  


(708.)

841\. _𑀲𑀫𑀸𑀦𑀪𑀸𑀕𑀁 𑀓𑀼𑀩𑁆𑀩𑁂𑀣,_  
_𑀕𑀸𑀫𑁂 𑀅𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑀯𑀦𑁆𑀤𑀺𑀢𑀁;_  
_𑀫𑀦𑁄𑀧𑀤𑁄𑀲𑀁 𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬,_  
_𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀡𑁆𑀡𑀢𑁄 𑀘𑀭𑁂𑁇 (𑁨𑁪)_  


(709.)

842\. _𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀺,_  
_𑀤𑀸𑀬𑁂 𑀅𑀕𑁆𑀕𑀺𑀲𑀺𑀔𑀽𑀧𑀫𑀸;_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀫𑀼𑀦𑀺𑀁 𑀧𑀮𑁄𑀪𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑀸 𑀲𑀼 𑀢𑀁 𑀫𑀸 𑀧𑀮𑁄𑀪𑀬𑀼𑀁𑁇 (𑁨𑁫)_  


(710.)

843\. _𑀯𑀺𑀭𑀢𑁄 𑀫𑁂𑀣𑀼𑀦𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂 𑀧𑀭𑁄𑀧𑀭𑁂;_  
_𑀅𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀸𑀭𑀢𑁆𑀢𑁄,_  
_𑀧𑀸𑀡𑁂𑀲𑀼 𑀢𑀲𑀣𑀸𑀯𑀭𑁂𑁇 (𑁨𑁬)_  


(711.)

844\. _𑀬𑀣𑀸 𑀅𑀳𑀁 𑀢𑀣𑀸 𑀏𑀢𑁂,_  
_𑀬𑀣𑀸 𑀏𑀢𑁂 𑀢𑀣𑀸 𑀅𑀳𑀁;_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀫𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦 𑀳𑀦𑁂𑀬𑁆𑀬 𑀦 𑀖𑀸𑀢𑀬𑁂𑁇 (𑁨𑁭)_  


(712.)

845\. _𑀳𑀺𑀢𑁆𑀯𑀸 𑀇𑀘𑁆𑀙𑀜𑁆𑀘 𑀮𑁄𑀪𑀜𑁆𑀘,_  
_𑀬𑀢𑁆𑀣 𑀲𑀢𑁆𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄;_  
_𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀢𑀭𑁂𑀬𑁆𑀬 𑀦𑀭𑀓𑀁 𑀇𑀫𑀁𑁇 (𑁨𑁮)_  


(713.)

846\. _𑀊𑀦𑀽𑀤𑀭𑁄 𑀫𑀺𑀢𑀸𑀳𑀸𑀭𑁄,_  
_𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀅𑀮𑁄𑀮𑀼𑀧𑁄;_  
_𑀲𑀤𑀸 𑀇𑀘𑁆𑀙𑀸𑀬 𑀦𑀺𑀘𑁆𑀙𑀸𑀢𑁄,_  
_𑀅𑀦𑀺𑀘𑁆𑀙𑁄 𑀳𑁄𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇 (𑁨𑁯)_  


(714.)

847\. _𑀲 𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑀦𑀦𑁆𑀢𑀫𑀪𑀺𑀳𑀸𑀭𑀬𑁂;_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀲𑁆𑀫𑀺𑀁,_  
_𑀆𑀲𑀦𑀽𑀧𑀕𑀢𑁄 𑀫𑀼𑀦𑀺𑁇 (𑁩𑁦)_  


(715.)

848\. _𑀲 𑀛𑀸𑀦𑀧𑀲𑀼𑀢𑁄 𑀥𑀻𑀭𑁄,_  
_𑀯𑀦𑀦𑁆𑀢𑁂 𑀭𑀫𑀺𑀢𑁄 𑀲𑀺𑀬𑀸;_  
_𑀛𑀸𑀬𑁂𑀣 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀫𑀪𑀺𑀢𑁄𑀲𑀬𑀁𑁇 (𑁩𑁧)_  


(716.)

849\. _𑀢𑀢𑁄 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀕𑀸𑀫𑀦𑁆𑀢𑀫𑀪𑀺𑀳𑀸𑀭𑀬𑁂;_  
_𑀅𑀯𑁆𑀳𑀸𑀦𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑁂𑀬𑁆𑀬,_  
_𑀅𑀪𑀺𑀳𑀸𑀭𑀜𑁆𑀘 𑀕𑀸𑀫𑀢𑁄𑁇 (𑁩𑁨)_  


(717.)

850\. _𑀦 𑀫𑀼𑀦𑀻 𑀕𑀸𑀫𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀓𑀼𑀮𑁂𑀲𑀼 𑀲𑀳𑀲𑀸 𑀘𑀭𑁂;_  
_𑀖𑀸𑀲𑁂𑀲𑀦𑀁 𑀙𑀺𑀦𑁆𑀦𑀓𑀣𑁄,_  
_𑀦 𑀯𑀸𑀘𑀁 𑀧𑀬𑀼𑀢𑀁 𑀪𑀡𑁂𑁇 (𑁩𑁩)_  


(718.)

851\. _𑀅𑀮𑀢𑁆𑀣𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀸𑀥𑀼,_  
_𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀓𑀼𑀲𑀮𑀁 𑀇𑀢𑀺;_  
_𑀉𑀪𑀬𑁂𑀦𑁂𑀯 𑀲𑁄 𑀢𑀸𑀤𑀻,_  
_𑀭𑀼𑀓𑁆𑀔𑀁𑀯𑀼𑀧𑀦𑀺𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁩𑁪)_  


(719.)

852\. _𑀲 𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑁄,_  
_𑀅𑀫𑀽𑀕𑁄 𑀫𑀽𑀕𑀲𑀫𑁆𑀫𑀢𑁄;_  
_𑀅𑀧𑁆𑀧𑀁 𑀤𑀸𑀦𑀁 𑀦 𑀳𑀻𑀍𑀏𑀬𑁆𑀬,_  
_𑀤𑀸𑀢𑀸𑀭𑀁 𑀦𑀸𑀯𑀚𑀸𑀦𑀺𑀬𑀸𑁇 (𑁩𑁫)_  


(720.)

853\. _𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸 𑀳𑀺 𑀧𑀝𑀺𑀧𑀤𑀸,_  
_𑀲𑀫𑀡𑁂𑀦 𑀧𑀓𑀸𑀲𑀺𑀢𑀸;_  
_𑀦 𑀧𑀸𑀭𑀁 𑀤𑀺𑀕𑀼𑀡𑀁 𑀬𑀦𑁆𑀢𑀺,_  
_𑀦𑀬𑀺𑀤𑀁 𑀏𑀓𑀕𑀼𑀡𑀁 𑀫𑀼𑀢𑀁𑁇 (𑁩𑁬)_  


(721.)

854\. _𑀬𑀲𑁆𑀲 𑀘 𑀯𑀺𑀲𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀙𑀺𑀦𑁆𑀦𑀲𑁄𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀓𑀺𑀘𑁆𑀘𑀸𑀓𑀺𑀘𑁆𑀘𑀧𑁆𑀧𑀳𑀻𑀦𑀲𑁆𑀲,_  
_𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁩𑁭)_  


(722.)

855\. _𑀫𑁄𑀦𑁂𑀬𑁆𑀬𑀁 𑀢𑁂 𑀉𑀧𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀁,_  
_𑀔𑀼𑀭𑀥𑀸𑀭𑀽𑀧𑀫𑁄 𑀪𑀯𑁂;_  
_𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀢𑀸𑀮𑀼𑀫𑀸𑀳𑀘𑁆𑀘,_  
_𑀉𑀤𑀭𑁂 𑀲𑀜𑁆𑀜𑀢𑁄 𑀲𑀺𑀬𑀸𑁇 (𑁩𑁮)_  


(723.)

856\. _𑀅𑀮𑀻𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀘 𑀲𑀺𑀬𑀸,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀩𑀳𑀼 𑀘𑀺𑀦𑁆𑀢𑀬𑁂;_  
_𑀦𑀺𑀭𑀸𑀫𑀕𑀦𑁆𑀥𑁄 𑀅𑀲𑀺𑀢𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀸𑀬𑀡𑁄𑁇 (𑁩𑁯)_  


(724.)

857\. _𑀏𑀓𑀸𑀲𑀦𑀲𑁆𑀲 𑀲𑀺𑀓𑁆𑀔𑁂𑀣,_  
_𑀲𑀫𑀡𑀽𑀧𑀸𑀲𑀦𑀲𑁆𑀲 𑀘;_  
_𑀏𑀓𑀢𑁆𑀢𑀁 𑀫𑁄𑀦𑀫𑀓𑁆𑀔𑀸𑀢𑀁,_  
_𑀏𑀓𑁄 𑀘𑁂 𑀅𑀪𑀺𑀭𑀫𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀅𑀣 𑀪𑀸𑀳𑀺𑀲𑀺 𑀤𑀲𑀤𑀺𑀲𑀸𑁇 (𑁪𑁦)_  


(725.)

858\. _𑀲𑀼𑀢𑁆𑀯𑀸 𑀥𑀻𑀭𑀸𑀦𑀁 𑀦𑀺𑀖𑁄𑀲𑀁,_  
_𑀛𑀸𑀬𑀻𑀦𑀁 𑀓𑀸𑀫𑀘𑀸𑀕𑀺𑀦𑀁;_  
_𑀢𑀢𑁄 𑀳𑀺𑀭𑀺𑀜𑁆𑀘 𑀲𑀤𑁆𑀥𑀜𑁆𑀘,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀫𑀸𑀫𑀓𑁄𑁇 (𑁪𑁧)_  


(726.)

859\. _𑀢𑀁 𑀦𑀤𑀻𑀳𑀺 𑀯𑀺𑀚𑀸𑀦𑀸𑀣,_  
_𑀲𑁄𑀩𑁆𑀪𑁂𑀲𑀼 𑀧𑀤𑀭𑁂𑀲𑀼 𑀘;_  
_𑀲𑀡𑀦𑁆𑀢𑀸 𑀬𑀦𑁆𑀢𑀺 𑀓𑀼𑀲𑁄𑀩𑁆𑀪𑀸,_  
_𑀢𑀼𑀡𑁆𑀳𑀻 𑀬𑀦𑁆𑀢𑀺 𑀫𑀳𑁄𑀤𑀥𑀻𑁇 (𑁪𑁨)_  


(727.)

860\. _𑀬𑀤𑀽𑀦𑀓𑀁 𑀢𑀁 𑀲𑀡𑀢𑀺,_  
_𑀬𑀁 𑀧𑀽𑀭𑀁 𑀲𑀦𑁆𑀢𑀫𑁂𑀯 𑀢𑀁;_  
_𑀅𑀟𑁆𑀠𑀓𑀼𑀫𑁆𑀪𑀽𑀧𑀫𑁄 𑀩𑀸𑀮𑁄,_  
_𑀭𑀳𑀤𑁄 𑀧𑀽𑀭𑁄𑀯 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇 (𑁪𑁩)_  


(728.)

861\. _𑀬𑀁 𑀲𑀫𑀡𑁄 𑀩𑀳𑀼𑀁 𑀪𑀸𑀲𑀢𑀺,_  
_𑀉𑀧𑁂𑀢𑀁 𑀅𑀢𑁆𑀣𑀲𑀜𑁆𑀳𑀺𑀢𑀁;_  
_𑀚𑀸𑀦𑀁 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺,_  
_𑀚𑀸𑀦𑀁 𑀲𑁄 𑀩𑀳𑀼 𑀪𑀸𑀲𑀢𑀺𑁇 (𑁪𑁪)_  


(729.)

862\. _𑀬𑁄 𑀘 𑀚𑀸𑀦𑀁 𑀲𑀁𑀬𑀢𑀢𑁆𑀢𑁄,_  
_𑀚𑀸𑀦𑀁 𑀦 𑀩𑀳𑀼 𑀪𑀸𑀲𑀢𑀺;_  
_𑀲 𑀫𑀼𑀦𑀻 𑀫𑁄𑀦𑀫𑀭𑀳𑀢𑀺,_  
_𑀲 𑀫𑀼𑀦𑀻 𑀫𑁄𑀦𑀫𑀚𑁆𑀛𑀕𑀸”𑀢𑀺𑁇 (𑁪𑁫)_
