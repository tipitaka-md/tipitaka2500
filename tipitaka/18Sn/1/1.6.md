# 𑁧𑁇𑁬 𑀧𑀭𑀸𑀪𑀯𑀲𑀼𑀢𑁆𑀢

* 𑀉𑀭𑀕𑀯𑀕𑁆𑀕

* 𑀧𑀭𑀸𑀪𑀯𑀲𑀼𑀢𑁆𑀢

106\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

(91.)

107\. _“𑀧𑀭𑀸𑀪𑀯𑀦𑁆𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁,_  
_𑀫𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫 𑀕𑁄𑀢𑀫;_  
_𑀪𑀯𑀦𑁆𑀢𑀁 𑀧𑀼𑀝𑁆𑀞𑀼𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧)_  


(92.)

108\. _“𑀲𑀼𑀯𑀺𑀚𑀸𑀦𑁄 𑀪𑀯𑀁 𑀳𑁄𑀢𑀺,_  
_𑀲𑀼𑀯𑀺𑀚𑀸𑀦𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀪𑀯𑀁 𑀳𑁄𑀢𑀺,_  
_𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑁆𑀲𑀻 𑀧𑀭𑀸𑀪𑀯𑁄”𑁇 (𑁨)_  


(93.)

109\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀧𑀞𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀤𑀼𑀢𑀺𑀬𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁩)_  


(94.)

110\. _“𑀅𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀺𑀬𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀲𑀦𑁆𑀢𑁂 𑀦 𑀓𑀼𑀭𑀼𑀢𑁂 𑀧𑀺𑀬𑀁;_  
_𑀅𑀲𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁪)_  


(95.)

111\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀢𑀢𑀺𑀬𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁫)_  


(96.)

112\. _“𑀦𑀺𑀤𑁆𑀤𑀸𑀲𑀻𑀮𑀻 𑀲𑀪𑀸𑀲𑀻𑀮𑀻,_  
_𑀅𑀦𑀼𑀝𑁆𑀞𑀸𑀢𑀸 𑀘 𑀬𑁄 𑀦𑀭𑁄;_  
_𑀅𑀮𑀲𑁄 𑀓𑁄𑀥𑀧𑀜𑁆𑀜𑀸𑀡𑁄,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁬)_  


(97.)

113\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀢𑀢𑀺𑀬𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁭)_  


(98.)

114\. _“𑀬𑁄 𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀁 𑀯𑀸,_  
_𑀚𑀺𑀡𑁆𑀡𑀓𑀁 𑀕𑀢𑀬𑁄𑀩𑁆𑀩𑀦𑀁;_  
_𑀧𑀳𑀼 𑀲𑀦𑁆𑀢𑁄 𑀦 𑀪𑀭𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁮)_  


(99.)

115\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀧𑀜𑁆𑀘𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁯)_  


(100.)

116\. _“𑀬𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀲𑀫𑀡𑀁 𑀯𑀸,_  
_𑀅𑀜𑁆𑀜𑀁 𑀯𑀸𑀧𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀁;_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀯𑀜𑁆𑀘𑁂𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁦)_  


(101.)

117\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀧𑀜𑁆𑀘𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀙𑀝𑁆𑀞𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁧)_  


(102.)

118\. _“𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀲𑀳𑀺𑀭𑀜𑁆𑀜𑁄 𑀲𑀪𑁄𑀚𑀦𑁄;_  
_𑀏𑀓𑁄 𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀲𑀸𑀤𑀽𑀦𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁨)_  


(103.)

119\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀙𑀝𑁆𑀞𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀲𑀢𑁆𑀢𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁩)_  


(104.)

120\. _“𑀚𑀸𑀢𑀺𑀢𑁆𑀣𑀤𑁆𑀥𑁄 𑀥𑀦𑀢𑁆𑀣𑀤𑁆𑀥𑁄,_  
_𑀕𑁄𑀢𑁆𑀢𑀢𑁆𑀣𑀤𑁆𑀥𑁄 𑀘 𑀬𑁄 𑀦𑀭𑁄;_  
_𑀲𑀜𑁆𑀜𑀸𑀢𑀺𑀁 𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑁂𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁪)_  


(105.)

121\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀲𑀢𑁆𑀢𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀅𑀝𑁆𑀞𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁫)_  


(106.)

122\. _“𑀇𑀢𑁆𑀣𑀺𑀥𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀭𑀸𑀥𑀼𑀢𑁆𑀢𑁄,_  
_𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄 𑀘 𑀬𑁄 𑀦𑀭𑁄;_  
_𑀮𑀤𑁆𑀥𑀁 𑀮𑀤𑁆𑀥𑀁 𑀯𑀺𑀦𑀸𑀲𑁂𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁬)_  


(107.)

123\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀅𑀝𑁆𑀞𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀦𑀯𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁭)_  


(108.)

124\. _“𑀲𑁂𑀳𑀺 𑀤𑀸𑀭𑁂𑀳𑀺 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀲𑀼 𑀧𑀤𑀼𑀲𑁆𑀲𑀢𑀺;_  
_𑀤𑀼𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀤𑀸𑀭𑁂𑀲𑀼,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁮)_  


(109.)

125\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀦𑀯𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀤𑀲𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁧𑁯)_  


(110.)

126\. _“𑀅𑀢𑀻𑀢𑀬𑁄𑀩𑁆𑀩𑀦𑁄 𑀧𑁄𑀲𑁄,_  
_𑀆𑀦𑁂𑀢𑀺 𑀢𑀺𑀫𑁆𑀩𑀭𑀼𑀢𑁆𑀣𑀦𑀺𑀁;_  
_𑀢𑀲𑁆𑀲𑀸 𑀇𑀲𑁆𑀲𑀸 𑀦 𑀲𑀼𑀧𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁨𑁦)_  


(111.)

127\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀤𑀲𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀏𑀓𑀸𑀤𑀲𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁨𑁧)_  


(112.)

128\. _“𑀇𑀢𑁆𑀣𑀺𑀁 𑀲𑁄𑀡𑁆𑀟𑀺𑀁 𑀯𑀺𑀓𑀺𑀭𑀡𑀺𑀁,_  
_𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑀸𑀧𑀺 𑀢𑀸𑀤𑀺𑀲𑀁;_  
_𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀲𑁆𑀫𑀺𑀁 𑀞𑀧𑁂𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁨𑁨)_  


(113.)

129\. _“𑀇𑀢𑀺 𑀳𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀫,_  
_𑀏𑀓𑀸𑀤𑀲𑀫𑁄 𑀲𑁄 𑀧𑀭𑀸𑀪𑀯𑁄;_  
_𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁”𑁇 (𑁨𑁩)_  


(114.)

130\. _“𑀅𑀧𑁆𑀧𑀪𑁄𑀕𑁄 𑀫𑀳𑀸𑀢𑀡𑁆𑀳𑁄,_  
_𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀚𑀸𑀬𑀢𑁂 𑀓𑀼𑀮𑁂;_  
_𑀲𑁄 𑀘 𑀭𑀚𑁆𑀚𑀁 𑀧𑀢𑁆𑀣𑀬𑀢𑀺,_  
_𑀢𑀁 𑀧𑀭𑀸𑀪𑀯𑀢𑁄 𑀫𑀼𑀔𑀁𑁇 (𑁨𑁪)_  


(115.)

131\. _𑀏𑀢𑁂 𑀧𑀭𑀸𑀪𑀯𑁂 𑀮𑁄𑀓𑁂,_
