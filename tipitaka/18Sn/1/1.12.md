# 𑁧𑁇𑁧𑁨 𑀫𑀼𑀦𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀉𑀭𑀕𑀯𑀕𑁆𑀕

### 𑀫𑀼𑀦𑀺𑀲𑀼𑀢𑁆𑀢

(209.)

240\. _𑀲𑀦𑁆𑀣𑀯𑀸𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁,_  
_𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄;_  
_𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁,_  
_𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁𑁇 (𑁧)_  


(210.)

241\. _𑀬𑁄 𑀚𑀸𑀢𑀫𑀼𑀘𑁆𑀙𑀺𑀚𑁆𑀚 𑀦 𑀭𑁄𑀧𑀬𑁂𑀬𑁆𑀬,_  
_𑀚𑀸𑀬𑀦𑁆𑀢𑀫𑀲𑁆𑀲 𑀦𑀸𑀦𑀼𑀧𑁆𑀧𑀯𑁂𑀘𑁆𑀙𑁂;_  
_𑀢𑀫𑀸𑀳𑀼 𑀏𑀓𑀁 𑀫𑀼𑀦𑀺𑀦𑀁 𑀘𑀭𑀦𑁆𑀢𑀁,_  
_𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀲𑁄 𑀲𑀦𑁆𑀢𑀺𑀧𑀤𑀁 𑀫𑀳𑁂𑀲𑀺𑁇 (𑁨)_  


(211.)

242\. _𑀲𑀗𑁆𑀔𑀸𑀬 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀧𑀫𑀸𑀬 𑀩𑀻𑀚𑀁,_  
_𑀲𑀺𑀦𑁂𑀳𑀫𑀲𑁆𑀲 𑀦𑀸𑀦𑀼𑀧𑁆𑀧𑀯𑁂𑀘𑁆𑀙𑁂;_  
_𑀲 𑀯𑁂 𑀫𑀼𑀦𑀻 𑀚𑀸𑀢𑀺𑀔𑀬𑀦𑁆𑀢𑀤𑀲𑁆𑀲𑀻,_  
_𑀢𑀓𑁆𑀓𑀁 𑀧𑀳𑀸𑀬 𑀦 𑀉𑀧𑁂𑀢𑀺 𑀲𑀗𑁆𑀔𑀁𑁇 (𑁩)_  


(212.)

243\. _𑀅𑀜𑁆𑀜𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺,_  
_𑀅𑀦𑀺𑀓𑀸𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀫𑁆𑀧𑀺 𑀢𑁂𑀲𑀁;_  
_𑀲 𑀯𑁂 𑀫𑀼𑀦𑀻 𑀯𑀻𑀢𑀕𑁂𑀥𑁄 𑀅𑀕𑀺𑀤𑁆𑀥𑁄,_  
_𑀦𑀸𑀬𑀽𑀳𑀢𑀻 𑀧𑀸𑀭𑀕𑀢𑁄 𑀳𑀺 𑀳𑁄𑀢𑀺𑁇 (𑁪)_  


(213.)

244\. _𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀪𑀼𑀁 𑀲𑀩𑁆𑀩𑀯𑀺𑀤𑀼𑀁 𑀲𑀼𑀫𑁂𑀥𑀁,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀦𑀽𑀧𑀮𑀺𑀢𑁆𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀚𑀳𑀁 𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁫)_  


(214.)

245\. _𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀲𑀻𑀮𑀯𑀢𑀽𑀧𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀛𑀸𑀦𑀭𑀢𑀁 𑀲𑀢𑀻𑀫𑀁;_  
_𑀲𑀗𑁆𑀕𑀸 𑀧𑀫𑀼𑀢𑁆𑀢𑀁 𑀅𑀔𑀺𑀮𑀁 𑀅𑀦𑀸𑀲𑀯𑀁,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁬)_  


(215.)

246\. _𑀏𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀫𑀼𑀦𑀺𑀫𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀁,_  
_𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀲𑀁𑀲𑀸𑀲𑀼 𑀅𑀯𑁂𑀥𑀫𑀸𑀦𑀁;_  
_𑀲𑀻𑀳𑀁𑀯 𑀲𑀤𑁆𑀤𑁂𑀲𑀼 𑀅𑀲𑀦𑁆𑀢𑀲𑀦𑁆𑀢𑀁,_  
_𑀯𑀸𑀢𑀁𑀯 𑀚𑀸𑀮𑀫𑁆𑀳𑀺 𑀅𑀲𑀚𑁆𑀚𑀫𑀸𑀦𑀁;_  
_𑀧𑀤𑁆𑀫𑀁𑀯 𑀢𑁄𑀬𑁂𑀦 𑀅𑀮𑀺𑀧𑁆𑀧𑀫𑀸𑀦𑀁,_  
_𑀦𑁂𑀢𑀸𑀭𑀫𑀜𑁆𑀜𑁂𑀲𑀫𑀦𑀜𑁆𑀜𑀦𑁂𑀬𑁆𑀬𑀁;_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁭)_  


(216.)

247\. _𑀬𑁄 𑀑𑀕𑀳𑀡𑁂 𑀣𑀫𑁆𑀪𑁄𑀭𑀺𑀯𑀸𑀪𑀺𑀚𑀸𑀬𑀢𑀺,_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀭𑁂 𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀯𑀤𑀦𑁆𑀢𑀺;_  
_𑀢𑀁 𑀯𑀻𑀢𑀭𑀸𑀕𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁮)_  


(217.)

248\. _𑀬𑁄 𑀯𑁂 𑀞𑀺𑀢𑀢𑁆𑀢𑁄 𑀢𑀲𑀭𑀁𑀯 𑀉𑀚𑁆𑀚𑀼,_  
_𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀢𑀺 𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀧𑀸𑀧𑀓𑁂𑀳𑀺;_  
_𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑁄 𑀯𑀺𑀲𑀫𑀁 𑀲𑀫𑀜𑁆𑀘,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁯)_  


(218.)

249\. _𑀬𑁄 𑀲𑀜𑁆𑀜𑀢𑀢𑁆𑀢𑁄 𑀦 𑀓𑀭𑁄𑀢𑀺 𑀧𑀸𑀧𑀁,_  
_𑀤𑀳𑀭𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁄 𑀘 𑀫𑀼𑀦𑀺 𑀬𑀢𑀢𑁆𑀢𑁄;_  
_𑀅𑀭𑁄𑀲𑀦𑁂𑀬𑁆𑀬𑁄 𑀦 𑀲𑁄 𑀭𑁄𑀲𑁂𑀢𑀺 𑀓𑀜𑁆𑀘𑀺,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁧𑁦)_  


(219.)

250\. _𑀬𑀤𑀕𑁆𑀕𑀢𑁄 𑀫𑀚𑁆𑀛𑀢𑁄 𑀲𑁂𑀲𑀢𑁄 𑀯𑀸,_  
_𑀧𑀺𑀡𑁆𑀟𑀁 𑀮𑀪𑁂𑀣 𑀧𑀭𑀤𑀢𑁆𑀢𑀽𑀧𑀚𑀻𑀯𑀻;_  
_𑀦𑀸𑀮𑀁 𑀣𑀼𑀢𑀼𑀁 𑀦𑁄𑀧𑀺 𑀦𑀺𑀧𑀘𑁆𑀘𑀯𑀸𑀤𑀻,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁧𑁧)_  


(220.)

251\. _𑀫𑀼𑀦𑀺𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀭𑀢𑀁 𑀫𑁂𑀣𑀼𑀦𑀲𑁆𑀫𑀸,_  
_𑀬𑁄 𑀬𑁄𑀩𑁆𑀩𑀦𑁂 𑀦𑁄𑀧𑀦𑀺𑀩𑀚𑁆𑀛𑀢𑁂 𑀓𑁆𑀯𑀘𑀺;_  
_𑀫𑀤𑀧𑁆𑀧𑀫𑀸𑀤𑀸 𑀯𑀺𑀭𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁧𑁨)_  


(221.)

252\. _𑀅𑀜𑁆𑀜𑀸𑀬 𑀮𑁄𑀓𑀁 𑀧𑀭𑀫𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀁,_  
_𑀑𑀖𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀢𑀺𑀢𑀭𑀺𑀬 𑀢𑀸𑀤𑀺𑀁;_  
_𑀢𑀁 𑀙𑀺𑀦𑁆𑀦𑀕𑀦𑁆𑀣𑀁 𑀅𑀲𑀺𑀢𑀁 𑀅𑀦𑀸𑀲𑀯𑀁,_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀥𑀻𑀭𑀸 𑀫𑀼𑀦𑀺 𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇 (𑁧𑁩)_  


(222.)

253\. _𑀅𑀲𑀫𑀸 𑀉𑀪𑁄 𑀤𑀽𑀭𑀯𑀺𑀳𑀸𑀭𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄,_  
_𑀕𑀺𑀳𑀻 𑀤𑀸𑀭𑀧𑁄𑀲𑀻 𑀅𑀫𑀫𑁄 𑀘 𑀲𑀼𑀩𑁆𑀩𑀢𑁄;_  
_𑀧𑀭𑀧𑀸𑀡𑀭𑁄𑀥𑀸𑀬 𑀕𑀺𑀳𑀻 𑀅𑀲𑀜𑁆𑀜𑀢𑁄,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀫𑀼𑀦𑀻 𑀭𑀓𑁆𑀔𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑁂 𑀬𑀢𑁄𑁇 (𑁧𑁪)_  


(223.)

254\. _𑀲𑀺𑀔𑀻 𑀬𑀣𑀸 𑀦𑀻𑀮𑀕𑀻𑀯𑁄 𑀯𑀺𑀳𑀗𑁆𑀕𑀫𑁄,_  
_𑀳𑀁𑀲𑀲𑁆𑀲 𑀦𑁄𑀧𑁂𑀢𑀺 𑀚𑀯𑀁 𑀓𑀼𑀤𑀸𑀘𑀦𑀁;_  
_𑀏𑀯𑀁 𑀕𑀺𑀳𑀻 𑀦𑀸𑀦𑀼𑀓𑀭𑁄𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄,_  
_𑀫𑀼𑀦𑀺𑀦𑁄 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀦𑀫𑁆𑀳𑀺 𑀛𑀸𑀬𑀢𑁄𑀢𑀺𑁇 (𑁧𑁫)_  


---

255\. 𑀫𑀼𑀦𑀺𑀲𑀼𑀢𑁆𑀢𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁𑁇



---

256\. 𑀉𑀭𑀕𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

257\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



258\. _𑀉𑀭𑀕𑁄 𑀥𑀦𑀺𑀬𑁄 𑀘𑁂𑀯,_  
_𑀯𑀺𑀲𑀸𑀡𑀜𑁆𑀘 𑀢𑀣𑀸 𑀓𑀲𑀺;_  
_𑀘𑀼𑀦𑁆𑀤𑁄 𑀧𑀭𑀸𑀪𑀯𑁄 𑀘𑁂𑀯,_  
_𑀯𑀲𑀮𑁄 𑀫𑁂𑀢𑁆𑀢𑀪𑀸𑀯𑀦𑀸𑁇_  


259\. _𑀲𑀸𑀢𑀸𑀕𑀺𑀭𑁄 𑀆𑀍𑀅𑀯𑀓𑁄,_  
_𑀯𑀺𑀚𑀬𑁄 𑀘 𑀢𑀣𑀸 𑀫𑀼𑀦𑀺;_  
_𑀤𑁆𑀯𑀸𑀤𑀲𑁂𑀢𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀉𑀭𑀕𑀯𑀕𑁆𑀕𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
