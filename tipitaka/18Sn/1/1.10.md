# 𑁧𑁇𑁧𑁦 𑀆𑀍𑀅𑀯𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀉𑀭𑀕𑀯𑀕𑁆𑀕

### 𑀆𑀍𑀅𑀯𑀓𑀲𑀼𑀢𑁆𑀢

207\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀆𑀍𑀅𑀯𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀍𑀅𑀯𑀓𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀪𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀍𑀅𑀯𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀺𑀓𑁆𑀔𑀫, 𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑁇 “𑀧𑀯𑀺𑀲, 𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

208\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀍𑀅𑀯𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀺𑀓𑁆𑀔𑀫, 𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑁇 “𑀧𑀯𑀺𑀲, 𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

209\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀍𑀅𑀯𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀺𑀓𑁆𑀔𑀫, 𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 “𑀦 𑀔𑁆𑀯𑀸𑀳𑀁 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀁 𑀢𑁂 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀢𑀁 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇

210\. “𑀧𑀜𑁆𑀳𑀁 𑀢𑀁, 𑀲𑀫𑀡, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀲𑀘𑁂 𑀫𑁂 𑀦 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀢𑁂 𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀳𑀤𑀬𑀁 𑀯𑀸 𑀢𑁂 𑀨𑀸𑀮𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀸𑀤𑁂𑀲𑀼 𑀯𑀸 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀕𑀗𑁆𑀕𑀸𑀬 𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

211\. “𑀦 𑀔𑁆𑀯𑀸𑀳𑀁 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀬𑁄 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀸 𑀔𑀺𑀧𑁂𑀬𑁆𑀬 𑀳𑀤𑀬𑀁 𑀯𑀸 𑀨𑀸𑀮𑁂𑀬𑁆𑀬 𑀧𑀸𑀤𑁂𑀲𑀼 𑀯𑀸 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀕𑀗𑁆𑀕𑀸𑀬 𑀔𑀺𑀧𑁂𑀬𑁆𑀬𑁇 𑀅𑀧𑀺 𑀘 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀼𑀘𑁆𑀙 𑀬𑀤𑀸𑀓𑀗𑁆𑀔𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀍𑀅𑀯𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

(183.)

212\. _“𑀓𑀺𑀁 𑀲𑀽𑀥 𑀯𑀺𑀢𑁆𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀓𑀺𑀁 𑀲𑀼 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀁 𑀲𑀼𑀔𑀫𑀸𑀯𑀳𑀸𑀢𑀺;_  
_𑀓𑀺𑀁 𑀲𑀼 𑀳𑀯𑁂 𑀲𑀸𑀤𑀼𑀢𑀭𑀁 𑀭𑀲𑀸𑀦𑀁,_  
_𑀓𑀣𑀁𑀚𑀻𑀯𑀺𑀁 𑀚𑀻𑀯𑀺𑀢𑀫𑀸𑀳𑀼 𑀲𑁂𑀝𑁆𑀞𑀁”𑁇 (𑁧)_  


(184.)

213\. _“𑀲𑀤𑁆𑀥𑀻𑀥 𑀯𑀺𑀢𑁆𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁄 𑀲𑀼𑀔𑀫𑀸𑀯𑀳𑀸𑀢𑀺;_  
_𑀲𑀘𑁆𑀘𑀁 𑀳𑀯𑁂 𑀲𑀸𑀤𑀼𑀢𑀭𑀁 𑀭𑀲𑀸𑀦𑀁,_  
_𑀧𑀜𑁆𑀜𑀸𑀚𑀻𑀯𑀺𑀁 𑀚𑀻𑀯𑀺𑀢𑀫𑀸𑀳𑀼 𑀲𑁂𑀝𑁆𑀞𑀁”𑁇 (𑁨)_  


(185.)

214\. _“𑀓𑀣𑀁 𑀲𑀼 𑀢𑀭𑀢𑀺 𑀑𑀖𑀁,_  
_𑀓𑀣𑀁 𑀲𑀼 𑀢𑀭𑀢𑀺 𑀅𑀡𑁆𑀡𑀯𑀁;_  
_𑀓𑀣𑀁 𑀲𑀼 𑀤𑀼𑀓𑁆𑀔𑀫𑀘𑁆𑀘𑁂𑀢𑀺,_  
_𑀓𑀣𑀁 𑀲𑀼 𑀧𑀭𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀺”𑁇 (𑁩)_  


(186.)

215\. _“𑀲𑀤𑁆𑀥𑀸𑀬 𑀢𑀭𑀢𑀺 𑀑𑀖𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦 𑀅𑀡𑁆𑀡𑀯𑀁;_  
_𑀯𑀻𑀭𑀺𑀬𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀫𑀘𑁆𑀘𑁂𑀢𑀺,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀭𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀺”𑁇 (𑁪)_  


(187.)

216\. _“𑀓𑀣𑀁 𑀲𑀼 𑀮𑀪𑀢𑁂 𑀧𑀜𑁆𑀜𑀁,_  
_𑀓𑀣𑀁 𑀲𑀼 𑀯𑀺𑀦𑁆𑀤𑀢𑁂 𑀥𑀦𑀁;_  
_𑀓𑀣𑀁 𑀲𑀼 𑀓𑀺𑀢𑁆𑀢𑀺𑀁 𑀧𑀧𑁆𑀧𑁄𑀢𑀺,_  
_𑀓𑀣𑀁 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀕𑀦𑁆𑀣𑀢𑀺;_  
_𑀅𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸 𑀧𑀭𑀁 𑀮𑁄𑀓𑀁,_  
_𑀓𑀣𑀁 𑀧𑁂𑀘𑁆𑀘 𑀦 𑀲𑁄𑀘𑀢𑀺”𑁇 (𑁫)_  


(188.)

217\. _“𑀲𑀤𑁆𑀤𑀳𑀸𑀦𑁄 𑀅𑀭𑀳𑀢𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀼𑀲𑁆𑀲𑀽𑀲𑀁 𑀮𑀪𑀢𑁂 𑀧𑀜𑁆𑀜𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄𑁇 (𑁬)_  


(189.)

218\. _𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑀸𑀭𑀻 𑀥𑀼𑀭𑀯𑀸,_  
_𑀉𑀝𑁆𑀞𑀸𑀢𑀸 𑀯𑀺𑀦𑁆𑀤𑀢𑁂 𑀥𑀦𑀁;_  
_𑀲𑀘𑁆𑀘𑁂𑀦 𑀓𑀺𑀢𑁆𑀢𑀺𑀁 𑀧𑀧𑁆𑀧𑁄𑀢𑀺,_  
_𑀤𑀤𑀁 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀕𑀦𑁆𑀣𑀢𑀺𑁇 (𑁭)_  


(190.)

219\. _𑀬𑀲𑁆𑀲𑁂𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀥𑀫𑁆𑀫𑀸,_  
_𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀖𑀭𑀫𑁂𑀲𑀺𑀦𑁄;_  
_𑀲𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑁄 𑀥𑀺𑀢𑀺 𑀘𑀸𑀕𑁄,_  
_𑀲 𑀯𑁂 𑀧𑁂𑀘𑁆𑀘 𑀦 𑀲𑁄𑀘𑀢𑀺𑁇 (𑁮)_  


(191.)

220\. _𑀇𑀗𑁆𑀖 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀧𑀼𑀘𑁆𑀙𑀲𑁆𑀲𑀼,_  
_𑀧𑀼𑀣𑀽 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂;_  
_𑀬𑀤𑀺 𑀲𑀘𑁆𑀘𑀸 𑀤𑀫𑀸 𑀘𑀸𑀕𑀸,_  
_𑀔𑀦𑁆𑀢𑁆𑀬𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀥 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇 (𑁯)_  


(192.)

221\. _“𑀓𑀣𑀁 𑀦𑀼 𑀤𑀸𑀦𑀺 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁,_  
_𑀧𑀼𑀣𑀽 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂;_  
_𑀬𑁄𑀳𑀁 𑀅𑀚𑁆𑀚 𑀧𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀬𑁄 𑀅𑀢𑁆𑀣𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑁄𑁇 (𑁧𑁦)_  


(193.)

222\. _𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀢 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀸𑀲𑀸𑀬𑀸𑀍𑀅𑀯𑀺𑀫𑀸𑀕𑀫𑀸;_  
_𑀬𑁄𑀳𑀁 𑀅𑀚𑁆𑀚 𑀧𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀬𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇 (𑁧𑁧)_  


(194.)

223\. _𑀲𑁄 𑀅𑀳𑀁 𑀯𑀺𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀕𑀸𑀫𑀸 𑀕𑀸𑀫𑀁 𑀧𑀼𑀭𑀸 𑀧𑀼𑀭𑀁;_  
_𑀦𑀫𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁨)_  


---

224\. 𑀆𑀍𑀅𑀯𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀤𑀲𑀫𑀁𑁇
