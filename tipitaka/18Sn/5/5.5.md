# 𑁫𑁇𑁫 𑀥𑁄𑀢𑀓𑀫𑀸𑀡𑀯𑀧𑀼𑀘𑁆𑀙𑀸

* 𑀧𑀸𑀭𑀸𑀬𑀦𑀯𑀕𑁆𑀕

* 𑀥𑁄𑀢𑀓𑀫𑀸𑀡𑀯𑀧𑀼𑀘𑁆𑀙𑀸

(1068.)

1254\. _“𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑁂𑀢𑀁, __(𑀇𑀘𑁆𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀥𑁄𑀢𑀓𑁄)___  
_𑀯𑀸𑀘𑀸𑀪𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺 𑀫𑀳𑁂𑀲𑀺 𑀢𑀼𑀬𑁆𑀳𑀁;_  
_𑀢𑀯 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁,_  
_𑀲𑀺𑀓𑁆𑀔𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑀢𑁆𑀢𑀦𑁄”𑁇 (𑁧)_  


(1069.)

1255\. _“𑀢𑁂𑀦𑀳𑀸𑀢𑀧𑁆𑀧𑀁 𑀓𑀭𑁄𑀳𑀺, __(𑀥𑁄𑀢𑀓𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀇𑀥𑁂𑀯 𑀦𑀺𑀧𑀓𑁄 𑀲𑀢𑁄;_  
_𑀇𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁,_  
_𑀲𑀺𑀓𑁆𑀔𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑀢𑁆𑀢𑀦𑁄”𑁇 (𑁨)_  


(1070.)

1256\. _“𑀧𑀲𑁆𑀲𑀸𑀫𑀳𑀁 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂,_  
_𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀺𑀭𑀺𑀬𑀫𑀸𑀦𑀁;_  
_𑀢𑀁 𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀫𑀦𑁆𑀢𑀘𑀓𑁆𑀔𑀼,_  
_𑀧𑀫𑀼𑀜𑁆𑀘 𑀫𑀁 𑀲𑀓𑁆𑀓 𑀓𑀣𑀁𑀓𑀣𑀸𑀳𑀺”𑁇 (𑁩)_  


(1071.)

1257\. _“𑀦𑀸𑀳𑀁 𑀲𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀫𑁄𑀘𑀦𑀸𑀬,_  
_𑀓𑀣𑀁𑀓𑀣𑀺𑀁 𑀥𑁄𑀢𑀓 𑀓𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂;_  
_𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑁂𑀝𑁆𑀞𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑁄,_  
_𑀏𑀯𑀁 𑀢𑀼𑀯𑀁 𑀑𑀖𑀫𑀺𑀫𑀁 𑀢𑀭𑁂𑀲𑀺”𑁇 (𑁪)_  


(1072.)

1258\. _“𑀅𑀦𑀼𑀲𑀸𑀲 𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀓𑀭𑀼𑀡𑀸𑀬𑀫𑀸𑀦𑁄,_  
_𑀯𑀺𑀯𑁂𑀓𑀥𑀫𑁆𑀫𑀁 𑀬𑀫𑀳𑀁 𑀯𑀺𑀚𑀜𑁆𑀜𑀁;_  
_𑀬𑀣𑀸𑀳𑀁 𑀆𑀓𑀸𑀲𑁄𑀯 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄,_  
_𑀇𑀥𑁂𑀯 𑀲𑀦𑁆𑀢𑁄 𑀅𑀲𑀺𑀢𑁄 𑀘𑀭𑁂𑀬𑁆𑀬𑀁”𑁇 (𑁫)_  


(1073.)

1259\. _“𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑁂 𑀲𑀦𑁆𑀢𑀺𑀁, __(𑀥𑁄𑀢𑀓𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀤𑀺𑀝𑁆𑀞𑁂 𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀻𑀢𑀺𑀳𑀁;_  
_𑀬𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁄 𑀘𑀭𑀁,_  
_𑀢𑀭𑁂 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀁”𑁇 (𑁬)_  


(1074.)

1260\. _“𑀢𑀜𑁆𑀘𑀸𑀳𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀸𑀫𑀺,_  
_𑀫𑀳𑁂𑀲𑀺 𑀲𑀦𑁆𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀬𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁄 𑀘𑀭𑀁,_  
_𑀢𑀭𑁂 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀁”𑁇 (𑁭)_  


(1075.)

1261\. _“𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸𑀲𑀺, __(𑀥𑁄𑀢𑀓𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀜𑁆𑀘𑀸𑀧𑀺 𑀫𑀚𑁆𑀛𑁂;_  
_𑀏𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀕𑁄𑀢𑀺 𑀮𑁄𑀓𑁂,_  
_𑀪𑀯𑀸𑀪𑀯𑀸𑀬 𑀫𑀸𑀓𑀸𑀲𑀺 𑀢𑀡𑁆𑀳𑀦𑁆”𑀢𑀺𑁇 (𑁮)_  


---
