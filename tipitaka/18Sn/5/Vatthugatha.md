# 𑀯𑀢𑁆𑀣𑀼𑀕𑀸𑀣𑀸

* 𑀧𑀸𑀭𑀸𑀬𑀦𑀯𑀕𑁆𑀕

* 𑀯𑀢𑁆𑀣𑀼𑀕𑀸𑀣𑀸

(983.)

1164\. _𑀓𑁄𑀲𑀮𑀸𑀦𑀁 𑀧𑀼𑀭𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_𑀅𑀕𑀫𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑀁;_  
_𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀁 𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀫𑀦𑁆𑀢𑀧𑀸𑀭𑀕𑀽𑁇 (𑁧)_  


(984.)

1165\. _𑀲𑁄 𑀅𑀲𑁆𑀲𑀓𑀲𑁆𑀲 𑀯𑀺𑀲𑀬𑁂,_  
_𑀅𑀍𑀅𑀓𑀲𑁆𑀲 𑀲𑀫𑀸𑀲𑀦𑁂;_  
_𑀯𑀲𑀺 𑀕𑁄𑀥𑀸𑀯𑀭𑀻𑀓𑀽𑀮𑁂,_  
_𑀉𑀜𑁆𑀙𑁂𑀦 𑀘 𑀨𑀮𑁂𑀦 𑀘𑁇 (𑁨)_  


(985.)

1166\. _𑀢𑀲𑁆𑀲𑁂𑀯 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀕𑀸𑀫𑁄 𑀘 𑀯𑀺𑀧𑀼𑀮𑁄 𑀅𑀳𑀼;_  
_𑀢𑀢𑁄 𑀚𑀸𑀢𑁂𑀦 𑀆𑀬𑁂𑀦,_  
_𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀫𑀓𑀧𑁆𑀧𑀬𑀺𑁇 (𑁩)_  


(986.)

1167\. _𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀼𑀦 𑀧𑀸𑀯𑀺𑀲𑀺 𑀅𑀲𑁆𑀲𑀫𑀁;_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀝𑀺𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀺,_  
_𑀅𑀜𑁆𑀜𑁄 𑀆𑀕𑀜𑁆𑀙𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇 (𑁪)_  


(987.)

1168\. _𑀉𑀕𑁆𑀖𑀝𑁆𑀝𑀧𑀸𑀤𑁄 𑀢𑀲𑀺𑀢𑁄,_  
_𑀧𑀗𑁆𑀓𑀤𑀦𑁆𑀢𑁄 𑀭𑀚𑀲𑁆𑀲𑀺𑀭𑁄;_  
_𑀲𑁄 𑀘 𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀬𑀸𑀘𑀢𑀺𑁇 (𑁫)_  


(988.)

1169\. _𑀢𑀫𑁂𑀦𑀁 𑀩𑀸𑀯𑀭𑀻 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀆𑀲𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑀬𑀺;_  
_𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀼𑀲𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑁇 (𑁬)_  


(989.)

1170\. _“𑀬𑀁 𑀔𑁄 𑀫𑀫 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀯𑀺𑀲𑀚𑁆𑀚𑀺𑀢𑀁 𑀫𑀬𑀸;_  
_𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀳𑀺 𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀦𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺 𑀫𑁂”𑁇 (𑁭)_  


(990.)

1171\. _“𑀲𑀘𑁂 𑀫𑁂 𑀬𑀸𑀘𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀪𑀯𑀁 𑀦𑀸𑀦𑀼𑀧𑀤𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀫𑀼𑀤𑁆𑀥𑀸 𑀨𑀮𑀢𑀼 𑀲𑀢𑁆𑀢𑀥𑀸”𑁇 (𑁮)_  


(991.)

1172\. _𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀳𑀓𑁄,_  
_𑀪𑁂𑀭𑀯𑀁 𑀲𑁄 𑀅𑀓𑀺𑀢𑁆𑀢𑀬𑀺;_  
_𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀩𑀸𑀯𑀭𑀻 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀅𑀳𑀼𑁇 (𑁯)_  


(992.)

1173\. _𑀉𑀲𑁆𑀲𑀼𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑀸𑀳𑀸𑀭𑁄,_  
_𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄;_  
_𑀅𑀣𑁄𑀧𑀺 𑀏𑀯𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀛𑀸𑀦𑁂 𑀦 𑀭𑀫𑀢𑀻 𑀫𑀦𑁄𑁇 (𑁧𑁦)_  


(993.)

1174\. _𑀉𑀢𑁆𑀭𑀲𑁆𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀺𑀦𑀻;_  
_𑀩𑀸𑀯𑀭𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑁇 (𑁧𑁧)_  


(994.)

1175\. _“𑀦 𑀲𑁄 𑀫𑀼𑀤𑁆𑀥𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀓𑀼𑀳𑀓𑁄 𑀲𑁄 𑀥𑀦𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀫𑀼𑀤𑁆𑀥𑀧𑀸𑀢𑁂 𑀯𑀸,_  
_𑀜𑀸𑀡𑀁 𑀢𑀲𑁆𑀲 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇 (𑁧𑁨)_  


(995.)

1176\. _“𑀪𑁄𑀢𑀻 𑀘𑀭𑀳𑀺 𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀫 𑀯𑀘𑁄 𑀢𑀯”𑁇 (𑁧𑁩)_  


(996.)

1177\. _“𑀅𑀳𑀫𑁆𑀧𑁂𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀜𑀸𑀡𑀫𑁂𑀢𑁆𑀣 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑁂 𑀘,_  
_𑀚𑀺𑀦𑀸𑀦𑀁 𑀳𑁂𑀢𑁆𑀣 𑀤𑀲𑁆𑀲𑀦𑀁”𑁇 (𑁧𑁪)_  


(997.)

1178\. _“𑀅𑀣 𑀓𑁄 𑀘𑀭𑀳𑀺 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀧𑀣𑀯𑀺𑀫𑀡𑁆𑀟𑀮𑁂;_  
_𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀤𑁂𑀯𑀢𑁂”𑁇 (𑁧𑁫)_  


(998.)

1179\. _“𑀧𑀼𑀭𑀸 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀫𑁆𑀳𑀸,_  
_𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀅𑀧𑀘𑁆𑀘𑁄 𑀑𑀓𑁆𑀓𑀸𑀓𑀭𑀸𑀚𑀲𑁆𑀲,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀪𑀗𑁆𑀓𑀭𑁄𑁇 (𑁧𑁬)_  


(999.)

1180\. _𑀲𑁄 𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀬𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀉𑀧𑀥𑀺𑀓𑁆𑀔𑀬𑁂𑁇 (𑁧𑁭)_  


(1000.)

1181\. _𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀮𑁄𑀓𑁂,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀢𑀁 𑀢𑁆𑀯𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀧𑀼𑀘𑁆𑀙𑀲𑁆𑀲𑀼,_  
_𑀲𑁄 𑀢𑁂 𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺”𑁇 (𑁧𑁮)_  


(1001.)

1182\. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀯𑀘𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀉𑀤𑀕𑁆𑀕𑁄 𑀩𑀸𑀯𑀭𑀻 𑀅𑀳𑀼;_  
_𑀲𑁄𑀓𑀲𑁆𑀲 𑀢𑀦𑀼𑀓𑁄 𑀆𑀲𑀺,_  
_𑀧𑀻𑀢𑀺𑀜𑁆𑀘 𑀯𑀺𑀧𑀼𑀮𑀁 𑀮𑀪𑀺𑁇 (𑁧𑁯)_  


(1002.)

1183\. _𑀲𑁄 𑀩𑀸𑀯𑀭𑀻 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀉𑀤𑀕𑁆𑀕𑁄,_  
_𑀢𑀁 𑀤𑁂𑀯𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺 𑀯𑁂𑀤𑀚𑀸𑀢𑁄;_  
_“𑀓𑀢𑀫𑀫𑁆𑀳𑀺 𑀕𑀸𑀫𑁂 𑀦𑀺𑀕𑀫𑀫𑁆𑀳𑀺 𑀯𑀸 𑀧𑀦,_  
_𑀓𑀢𑀫𑀫𑁆𑀳𑀺 𑀯𑀸 𑀚𑀦𑀧𑀤𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑁄;_  
_𑀬𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀧𑀲𑁆𑀲𑁂𑀫𑀼,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑀁”𑁇 (𑁨𑁦)_  


(1003.)

1184\. _“𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀓𑁄𑀲𑀮𑀫𑀦𑁆𑀤𑀺𑀭𑁂 𑀚𑀺𑀦𑁄,_  
_𑀧𑀳𑀽𑀢𑀧𑀜𑁆𑀜𑁄 𑀯𑀭𑀪𑀽𑀭𑀺𑀫𑁂𑀥𑀲𑁄;_  
_𑀲𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀥𑀼𑀭𑁄 𑀅𑀦𑀸𑀲𑀯𑁄,_  
_𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀲𑁆𑀲 𑀯𑀺𑀤𑀽 𑀦𑀭𑀸𑀲𑀪𑁄”𑁇 (𑁨𑁧)_  


(1004.)

1185\. _𑀢𑀢𑁄 𑀆𑀫𑀦𑁆𑀢𑀬𑀻 𑀲𑀺𑀲𑁆𑀲𑁂,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀫𑀦𑁆𑀢𑀧𑀸𑀭𑀕𑁂;_  
_“𑀏𑀣 𑀫𑀸𑀡𑀯𑀸 𑀅𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀼𑀡𑀸𑀣 𑀯𑀘𑀦𑀁 𑀫𑀫𑁇 (𑁨𑁨)_  


(1005.)

1186\. _𑀬𑀲𑁆𑀲𑁂𑀲𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑁂,_  
_𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄;_  
_𑀲𑁆𑀯𑀸𑀚𑁆𑀚 𑀮𑁄𑀓𑀫𑁆𑀳𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀇𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁,_  
_𑀧𑀲𑁆𑀲𑀯𑁆𑀳𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑀁”𑁇 (𑁨𑁩)_  


(1006.)

1187\. _“𑀓𑀣𑀁 𑀘𑀭𑀳𑀺 𑀚𑀸𑀦𑁂𑀫𑀼,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀅𑀚𑀸𑀦𑀢𑀁 𑀦𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀬𑀣𑀸 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇 (𑁨𑁪)_  


(1007.)

1188\. _“𑀆𑀕𑀢𑀸𑀦𑀺 𑀳𑀺 𑀫𑀦𑁆𑀢𑁂𑀲𑀼,_  
_𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀸;_  
_𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀦𑀺 𑀘 𑀩𑁆𑀬𑀸𑀓𑁆𑀔𑀸𑀢𑀸,_  
_𑀲𑀫𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑁄𑁇 (𑁨𑁫)_  


(1008.)

1189\. _𑀬𑀲𑁆𑀲𑁂𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀕𑀢𑁆𑀢𑁂𑀲𑀼,_  
_𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀸;_  
_𑀤𑁆𑀯𑁂𑀬𑁂𑀯 𑀢𑀲𑁆𑀲 𑀕𑀢𑀺𑀬𑁄,_  
_𑀢𑀢𑀺𑀬𑀸 𑀳𑀺 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨𑁬)_  


(1009.)

1190\. _𑀲𑀘𑁂 𑀅𑀕𑀸𑀭𑀁 𑀆𑀯𑀲𑀢𑀺,_  
_𑀯𑀺𑀚𑁂𑀬𑁆𑀬 𑀧𑀣𑀯𑀺𑀁 𑀇𑀫𑀁;_  
_𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦,_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀺𑁇 (𑁨𑁭)_  


(1010.)

1191\. _𑀲𑀘𑁂 𑀘 𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑀢𑀺,_  
_𑀅𑀕𑀸𑀭𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀯𑀺𑀯𑀝𑁆𑀝𑀘𑁆𑀙𑀤𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀭𑀳𑀸 𑀪𑀯𑀢𑀺 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄𑁇 (𑁨𑁮)_  


(1011.)

1192\. _𑀚𑀸𑀢𑀺𑀁 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘 𑀮𑀓𑁆𑀔𑀡𑀁,_  
_𑀫𑀦𑁆𑀢𑁂 𑀲𑀺𑀲𑁆𑀲𑁂 𑀧𑀼𑀦𑀸𑀧𑀭𑁂;_  
_𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀫𑀦𑀲𑀸𑀬𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀣𑁇 (𑁨𑁯)_  


(1012.)

1193\. _𑀅𑀦𑀸𑀯𑀭𑀡𑀤𑀲𑁆𑀲𑀸𑀯𑀻,_  
_𑀬𑀤𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑀦𑀲𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀧𑀜𑁆𑀳𑁂,_  
_𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀲𑁆𑀲𑀚𑁂𑀲𑁆𑀲𑀢𑀺”𑁇 (𑁩𑁦)_  


(1013.)

1194\. _𑀩𑀸𑀯𑀭𑀺𑀲𑁆𑀲 𑀯𑀘𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀲𑁆𑀲𑀸 𑀲𑁄𑀍𑀅𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀅𑀚𑀺𑀢𑁄 𑀢𑀺𑀲𑁆𑀲𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑁄,_  
_𑀧𑀼𑀡𑁆𑀡𑀓𑁄 𑀅𑀣 𑀫𑁂𑀢𑁆𑀢𑀕𑀽𑁇 (𑁩𑁧)_  


(1014.)

1195\. _𑀥𑁄𑀢𑀓𑁄 𑀉𑀧𑀲𑀻𑀯𑁄 𑀘,_  
_𑀦𑀦𑁆𑀤𑁄 𑀘 𑀅𑀣 𑀳𑁂𑀫𑀓𑁄;_  
_𑀢𑁄𑀤𑁂𑀬𑁆𑀬-𑀓𑀧𑁆𑀧𑀸 𑀤𑀼𑀪𑀬𑁄,_  
_𑀚𑀢𑀼𑀓𑀡𑁆𑀡𑀻 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇 (𑁩𑁨)_  


(1015.)

1196\. _𑀪𑀤𑁆𑀭𑀸𑀯𑀼𑀥𑁄 𑀉𑀤𑀬𑁄 𑀘,_  
_𑀧𑁄𑀲𑀸𑀮𑁄 𑀘𑀸𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀫𑁄𑀖𑀭𑀸𑀚𑀸 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀧𑀺𑀗𑁆𑀕𑀺𑀬𑁄 𑀘 𑀫𑀳𑀸𑀇𑀲𑀺𑁇 (𑁩𑁩)_  


(1016.)

1197\. _𑀧𑀘𑁆𑀘𑁂𑀓𑀕𑀡𑀺𑀦𑁄 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑀯𑀸𑀲𑀦𑀯𑀸𑀲𑀺𑀢𑀸𑁇 (𑁩𑁪)_  


(1017.)

1198\. _𑀩𑀸𑀯𑀭𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀚𑀝𑀸𑀚𑀺𑀦𑀥𑀭𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀼𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑀸𑁇 (𑁩𑁫)_  


(1018.)

1199\. _𑀅𑀍𑀅𑀓𑀲𑁆𑀲 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀧𑀼𑀭𑀺𑀫𑀸𑀳𑀺𑀲𑁆𑀲𑀢𑀺𑀁 𑀢𑀤𑀸;_  
_𑀉𑀚𑁆𑀚𑁂𑀦𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀕𑁄𑀦𑀤𑁆𑀥𑀁,_  
_𑀯𑁂𑀤𑀺𑀲𑀁 𑀯𑀦𑀲𑀯𑁆𑀳𑀬𑀁𑁇 (𑁩𑁬)_  


(1019.)

1200\. _𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀸𑀓𑁂𑀢𑀁,_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀜𑁆𑀘 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑁂𑀢𑀩𑁆𑀬𑀁 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀁,_  
_𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀜𑁆𑀘 𑀫𑀦𑁆𑀤𑀺𑀭𑀁𑁇 (𑁩𑁭)_  


(1020.)

1201\. _𑀧𑀸𑀯𑀜𑁆𑀘 𑀪𑁄𑀕𑀦𑀕𑀭𑀁,_  
_𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀫𑀸𑀕𑀥𑀁 𑀧𑀼𑀭𑀁;_  
_𑀧𑀸𑀲𑀸𑀡𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀜𑁆𑀘,_  
_𑀭𑀫𑀡𑀻𑀬𑀁 𑀫𑀦𑁄𑀭𑀫𑀁𑁇 (𑁩𑁮)_  


(1021.)

1202\. _𑀢𑀲𑀺𑀢𑁄𑀯𑀼𑀤𑀓𑀁 𑀲𑀻𑀢𑀁,_  
_𑀫𑀳𑀸𑀮𑀸𑀪𑀁𑀯 𑀯𑀸𑀡𑀺𑀚𑁄;_  
_𑀙𑀸𑀬𑀁 𑀖𑀫𑁆𑀫𑀸𑀪𑀺𑀢𑀢𑁆𑀢𑁄𑀯,_  
_𑀢𑀼𑀭𑀺𑀢𑀸 𑀧𑀩𑁆𑀩𑀢𑀫𑀸𑀭𑀼𑀳𑀼𑀁𑁇 (𑁩𑁯)_  


(1022.)

1203\. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺,_  
_𑀲𑀻𑀳𑁄𑀯 𑀦𑀤𑀢𑀻 𑀯𑀦𑁂𑁇 (𑁪𑁦)_  


(1023.)

1204\. _𑀅𑀚𑀺𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀢𑀭𑀁𑀲𑀺𑀁𑀯 𑀪𑀸𑀡𑀼𑀫𑀁;_  
_𑀘𑀦𑁆𑀤𑀁 𑀬𑀣𑀸 𑀧𑀦𑁆𑀦𑀭𑀲𑁂,_  
_𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀉𑀧𑀸𑀕𑀢𑀁𑁇 (𑁪𑁧)_  


(1024.)

1205\. _𑀅𑀣𑀲𑁆𑀲 𑀕𑀢𑁆𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀧𑀭𑀺𑀧𑀽𑀭𑀜𑁆𑀘 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁;_  
_𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀳𑀝𑁆𑀞𑁄,_  
_𑀫𑀦𑁄𑀧𑀜𑁆𑀳𑁂 𑀅𑀧𑀼𑀘𑁆𑀙𑀣𑁇 (𑁪𑁨)_  


(1025.)

1206\. _“𑀆𑀤𑀺𑀲𑁆𑀲 𑀚𑀫𑁆𑀫𑀦𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀕𑁄𑀢𑁆𑀢𑀁 𑀩𑁆𑀭𑀽𑀳𑀺 𑀲𑀮𑀓𑁆𑀔𑀡𑀁;_  
_𑀫𑀦𑁆𑀢𑁂𑀲𑀼 𑀧𑀸𑀭𑀫𑀺𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀢𑀺 𑀯𑀸𑀘𑁂𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄”𑁇 (𑁪𑁩)_  


(1026.)

1207\. _“𑀯𑀻𑀲𑀁 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀆𑀬𑀼,_  
_𑀲𑁄 𑀘 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀩𑀸𑀯𑀭𑀻;_  
_𑀢𑀻𑀡𑀺𑀲𑁆𑀲 𑀮𑀓𑁆𑀔𑀡𑀸 𑀕𑀢𑁆𑀢𑁂,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽𑁇 (𑁪𑁪)_  


(1027.)

1208\. _𑀮𑀓𑁆𑀔𑀡𑁂 𑀇𑀢𑀺𑀳𑀸𑀲𑁂 𑀘,_  
_𑀲𑀦𑀺𑀖𑀡𑁆𑀟𑀼𑀲𑀓𑁂𑀝𑀼𑀪𑁂;_  
_𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺 𑀯𑀸𑀘𑁂𑀢𑀺,_  
_𑀲𑀥𑀫𑁆𑀫𑁂 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑁄”𑁇 (𑁪𑁫)_  


(1028.)

1209\. _“𑀮𑀓𑁆𑀔𑀡𑀸𑀦𑀁 𑀧𑀯𑀺𑀘𑀬𑀁,_  
_𑀩𑀸𑀯𑀭𑀺𑀲𑁆𑀲 𑀦𑀭𑀼𑀢𑁆𑀢𑀫;_  
_𑀓𑀗𑁆𑀔𑀘𑁆𑀙𑀺𑀤 𑀧𑀓𑀸𑀲𑁂𑀳𑀺,_  
_𑀫𑀸 𑀦𑁄 𑀓𑀗𑁆𑀔𑀸𑀬𑀺𑀢𑀁 𑀅𑀳𑀼”𑁇 (𑁪𑁬)_  


(1029.)

1210\. _“𑀫𑀼𑀔𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀙𑀸𑀤𑁂𑀢𑀺,_  
_𑀉𑀡𑁆𑀡𑀲𑁆𑀲 𑀪𑀫𑀼𑀓𑀦𑁆𑀢𑀭𑁂;_  
_𑀓𑁄𑀲𑁄𑀳𑀺𑀢𑀁 𑀯𑀢𑁆𑀣𑀕𑀼𑀬𑁆𑀳𑀁,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀫𑀸𑀡𑀯”𑁇 (𑁪𑁭)_  


(1030.)

1211\. _𑀧𑀼𑀘𑁆𑀙𑀜𑁆𑀳𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀲𑀼𑀡𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀳𑁂 𑀯𑀺𑀬𑀸𑀓𑀢𑁂;_  
_𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑁂𑀢𑀺 𑀚𑀦𑁄 𑀲𑀩𑁆𑀩𑁄,_  
_𑀯𑁂𑀤𑀚𑀸𑀢𑁄 𑀓𑀢𑀜𑁆𑀚𑀮𑀻𑁇 (𑁪𑁮)_  


(1031.)

1212\. _“𑀓𑁄 𑀦𑀼 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀸,_  
_𑀇𑀦𑁆𑀤𑁄 𑀯𑀸𑀧𑀺 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀺;_  
_𑀫𑀦𑀲𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀧𑀜𑁆𑀳𑁂,_  
_𑀓𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀪𑀸𑀲𑀢𑀺”𑁇 (𑁪𑁯)_  


(1032.)

1213\. _“𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀩𑀸𑀯𑀭𑀻 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺;_  
_𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺 𑀪𑀕𑀯𑀸,_  
_𑀓𑀗𑁆𑀔𑀁 𑀯𑀺𑀦𑀬 𑀦𑁄 𑀇𑀲𑁂”𑁇 (𑁫𑁦)_  


(1033.)

1214\. _“𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀫𑀼𑀤𑁆𑀥𑀸𑀢𑀺 𑀚𑀸𑀦𑀸𑀳𑀺,_  
_𑀯𑀺𑀚𑁆𑀚𑀸 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀺𑀦𑀻;_  
_𑀲𑀤𑁆𑀥𑀸𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀻𑀳𑀺,_  
_𑀙𑀦𑁆𑀤𑀯𑀺𑀭𑀺𑀬𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑀸”𑁇 (𑁫𑁧)_  


(1034.)

1215\. _𑀢𑀢𑁄 𑀯𑁂𑀤𑁂𑀦 𑀫𑀳𑀢𑀸,_  
_𑀲𑀦𑁆𑀣𑀫𑁆𑀪𑀺𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀡𑀯𑁄;_  
_𑀏𑀓𑀁𑀲𑀁 𑀅𑀚𑀺𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀧𑀢𑀺𑁇 (𑁫𑁨)_  


(1035.)

1216\. _“𑀩𑀸𑀯𑀭𑀻 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑁄𑀢𑁄,_  
_𑀲𑀳 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀫𑀸𑀭𑀺𑀲;_  
_𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀫”𑁇 (𑁫𑁩)_  


(1036.)

1217\. _“𑀲𑀼𑀔𑀺𑀢𑁄 𑀩𑀸𑀯𑀭𑀻 𑀳𑁄𑀢𑀼,_  
_𑀲𑀳 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀢𑁆𑀯𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀳𑀺,_  
_𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀸𑀳𑀺 𑀫𑀸𑀡𑀯𑁇 (𑁫𑁪)_  


(1037.)

1218\. _𑀩𑀸𑀯𑀭𑀺𑀲𑁆𑀲 𑀘 𑀢𑀼𑀬𑁆𑀳𑀁 𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀩𑁆𑀩𑀲𑀁𑀲𑀬𑀁;_  
_𑀓𑀢𑀸𑀯𑀓𑀸𑀲𑀸 𑀧𑀼𑀘𑁆𑀙𑀯𑁆𑀳𑁄,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀫𑀦𑀲𑀺𑀘𑁆𑀙𑀣”𑁇 (𑁫𑁫)_  


(1038.)

1219\. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀓𑀢𑁄𑀓𑀸𑀲𑁄,_  
_𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀜𑁆𑀚𑀮𑀻;_  
_𑀅𑀚𑀺𑀢𑁄 𑀧𑀞𑀫𑀁 𑀧𑀜𑁆𑀳𑀁,_  
_𑀢𑀢𑁆𑀣 𑀧𑀼𑀘𑁆𑀙𑀺 𑀢𑀣𑀸𑀕𑀢𑀁𑁇 (𑁫𑁬)_  


---

1220\. 𑀯𑀢𑁆𑀣𑀼𑀕𑀸𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
