# 𑁨𑁇𑁧𑁦 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(334.)

386. _𑀉𑀝𑁆𑀞𑀳𑀣 𑀦𑀺𑀲𑀻𑀤𑀣,_  
_𑀓𑁄 𑀅𑀢𑁆𑀣𑁄 𑀲𑀼𑀧𑀺𑀢𑁂𑀦 𑀯𑁄;_  
_𑀆𑀢𑀼𑀭𑀸𑀦𑀜𑁆𑀳𑀺 𑀓𑀸 𑀦𑀺𑀤𑁆𑀤𑀸,_  
_𑀲𑀮𑁆𑀮𑀯𑀺𑀤𑁆𑀥𑀸𑀦 𑀭𑀼𑀧𑁆𑀧𑀢𑀁𑁇 (𑁧)_  


(335.)

387. _𑀉𑀝𑁆𑀞𑀳𑀣 𑀦𑀺𑀲𑀻𑀤𑀣,_  
_𑀤𑀍𑀳𑀁 𑀲𑀺𑀓𑁆𑀔𑀣 𑀲𑀦𑁆𑀢𑀺𑀬𑀸;_  
_𑀫𑀸 𑀯𑁄 𑀧𑀫𑀢𑁆𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀫𑀘𑁆𑀘𑀼𑀭𑀸𑀚𑀸 𑀅𑀫𑁄𑀳𑀬𑀺𑀢𑁆𑀣 𑀯𑀲𑀸𑀦𑀼𑀕𑁂𑁇 (𑁨)_  


(336.)

388. _𑀬𑀸𑀬 𑀤𑁂𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘,_  
_𑀲𑀺𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀅𑀢𑁆𑀣𑀺𑀓𑀸;_  
_𑀢𑀭𑀣𑁂𑀢𑀁 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀁,_  
_𑀔𑀡𑁄 𑀯𑁄 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀔𑀡𑀸𑀢𑀻𑀢𑀸 𑀳𑀺 𑀲𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇 (𑁩)_  


(337.)

389. _𑀧𑀫𑀸𑀤𑁄 𑀭𑀚𑁄 𑀧𑀫𑀸𑀤𑁄,_  
_𑀧𑀫𑀸𑀤𑀸𑀦𑀼𑀧𑀢𑀺𑀢𑁄 𑀭𑀚𑁄;_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦 𑀯𑀺𑀚𑁆𑀚𑀸𑀬,_  
_𑀅𑀩𑁆𑀩𑀳𑁂 𑀲𑀮𑁆𑀮𑀫𑀢𑁆𑀢𑀦𑁄𑀢𑀺𑁇 (𑁪)_  


---

390. 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀁 𑀤𑀲𑀫𑀁𑁇
