# 𑁪𑁇𑁧𑁦 𑀧𑀼𑀭𑀸𑀪𑁂𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀭𑀸𑀪𑁂𑀤𑀲𑀼𑀢𑁆𑀢

(855.)

1025\. _“𑀓𑀣𑀁𑀤𑀲𑁆𑀲𑀻 𑀓𑀣𑀁𑀲𑀻𑀮𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀢𑀁 𑀫𑁂 𑀕𑁄𑀢𑀫 𑀧𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀉𑀢𑁆𑀢𑀫𑀁 𑀦𑀭𑀁”𑁇 (𑁧)_  


(856.)

1026\. _“𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑁄 𑀧𑀼𑀭𑀸 𑀪𑁂𑀤𑀸, __(𑀇𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀧𑀼𑀩𑁆𑀩𑀫𑀦𑁆𑀢𑀫𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄;_  
_𑀯𑁂𑀫𑀚𑁆𑀛𑁂 𑀦𑀼𑀧𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑁄,_  
_𑀢𑀲𑁆𑀲 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁𑁇 (𑁨)_  


(857.)

1027\. _𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀅𑀲𑀦𑁆𑀢𑀸𑀲𑀻,_  
_𑀅𑀯𑀺𑀓𑀢𑁆𑀣𑀻 𑀅𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁄;_  
_𑀫𑀦𑁆𑀢𑀪𑀸𑀡𑀻 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄,_  
_𑀲 𑀯𑁂 𑀯𑀸𑀘𑀸𑀬𑀢𑁄 𑀫𑀼𑀦𑀺𑁇 (𑁩)_  


(858.)

1028\. _𑀦𑀺𑀭𑀸𑀲𑀢𑁆𑀢𑀺 𑀅𑀦𑀸𑀕𑀢𑁂,_  
_𑀅𑀢𑀻𑀢𑀁 𑀦𑀸𑀦𑀼𑀲𑁄𑀘𑀢𑀺;_  
_𑀯𑀺𑀯𑁂𑀓𑀤𑀲𑁆𑀲𑀻 𑀨𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀤𑀺𑀝𑁆𑀞𑀻𑀲𑀼 𑀘 𑀦 𑀦𑀻𑀬𑀢𑀺𑁇 (𑁪)_  


(859.)

1029\. _𑀧𑀢𑀺𑀮𑀻𑀦𑁄 𑀅𑀓𑀼𑀳𑀓𑁄,_  
_𑀅𑀧𑀺𑀳𑀸𑀮𑀼 𑀅𑀫𑀘𑁆𑀙𑀭𑀻;_  
_𑀅𑀧𑁆𑀧𑀕𑀩𑁆𑀪𑁄 𑀅𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑁄,_  
_𑀧𑁂𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑁂 𑀘 𑀦𑁄 𑀬𑀼𑀢𑁄𑁇 (𑁫)_  


(860.)

1030\. _𑀲𑀸𑀢𑀺𑀬𑁂𑀲𑀼 𑀅𑀦𑀲𑁆𑀲𑀸𑀯𑀻,_  
_𑀅𑀢𑀺𑀫𑀸𑀦𑁂 𑀘 𑀦𑁄 𑀬𑀼𑀢𑁄;_  
_𑀲𑀡𑁆𑀳𑁄 𑀘 𑀧𑀝𑀺𑀪𑀸𑀦𑀯𑀸,_  
_𑀦 𑀲𑀤𑁆𑀥𑁄 𑀦 𑀯𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇 (𑁬)_  


(861.)

1031\. _𑀮𑀸𑀪𑀓𑀫𑁆𑀬𑀸 𑀦 𑀲𑀺𑀓𑁆𑀔𑀢𑀺,_  
_𑀅𑀮𑀸𑀪𑁂 𑀘 𑀦 𑀓𑀼𑀧𑁆𑀧𑀢𑀺;_  
_𑀅𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀘 𑀢𑀡𑁆𑀳𑀸𑀬,_  
_𑀭𑀲𑁂𑀲𑀼 𑀦𑀸𑀦𑀼𑀕𑀺𑀚𑁆𑀛𑀢𑀺𑁇 (𑁭)_  


(862.)

1032\. _𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀤𑀸 𑀲𑀢𑁄,_  
_𑀦 𑀮𑁄𑀓𑁂 𑀫𑀜𑁆𑀜𑀢𑁂 𑀲𑀫𑀁;_  
_𑀦 𑀯𑀺𑀲𑁂𑀲𑀻 𑀦 𑀦𑀻𑀘𑁂𑀬𑁆𑀬𑁄,_  
_𑀢𑀲𑁆𑀲 𑀦𑁄 𑀲𑀦𑁆𑀢𑀺 𑀉𑀲𑁆𑀲𑀤𑀸𑁇 (𑁮)_  


(863.)

1033\. _𑀬𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀬𑀦𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀜𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄;_  
_𑀪𑀯𑀸𑀬 𑀯𑀺𑀪𑀯𑀸𑀬 𑀯𑀸,_  
_𑀢𑀡𑁆𑀳𑀸 𑀬𑀲𑁆𑀲 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁯)_  


(864.)

1034\. _𑀢𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀉𑀧𑀲𑀦𑁆𑀢𑁄𑀢𑀺,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑀁;_  
_𑀕𑀦𑁆𑀣𑀸 𑀢𑀲𑁆𑀲 𑀦 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀅𑀢𑀭𑀻 𑀲𑁄 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀁𑁇 (𑁧𑁦)_  


(865.)

1035\. _𑀦 𑀢𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑀸 𑀧𑀲𑀯𑁄,_  
_𑀔𑁂𑀢𑁆𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀢𑁆𑀢𑀸 𑀯𑀸𑀧𑀺 𑀦𑀺𑀭𑀢𑁆𑀢𑀸 𑀯𑀸,_  
_𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺𑁇 (𑁧𑁧)_  


(866.)

1036\. _𑀬𑁂𑀦 𑀦𑀁 𑀯𑀚𑁆𑀚𑀼𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸,_  
_𑀅𑀣𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀢𑀁 𑀢𑀲𑁆𑀲 𑀅𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀯𑀸𑀤𑁂𑀲𑀼 𑀦𑁂𑀚𑀢𑀺𑁇 (𑁧𑁨)_  


(867.)

1037\. _𑀯𑀻𑀢𑀕𑁂𑀥𑁄 𑀅𑀫𑀘𑁆𑀙𑀭𑀻,_  
_𑀦 𑀉𑀲𑁆𑀲𑁂𑀲𑀼 𑀯𑀤𑀢𑁂 𑀫𑀼𑀦𑀺;_  
_𑀦 𑀲𑀫𑁂𑀲𑀼 𑀦 𑀑𑀫𑁂𑀲𑀼,_  
_𑀓𑀧𑁆𑀧𑀁 𑀦𑁂𑀢𑀺 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁄𑁇 (𑁧𑁩)_  


(868.)

1038\. _𑀬𑀲𑁆𑀲 𑀮𑁄𑀓𑁂 𑀲𑀓𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀅𑀲𑀢𑀸 𑀘 𑀦 𑀲𑁄𑀘𑀢𑀺;_  
_𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘 𑀦 𑀕𑀘𑁆𑀙𑀢𑀺,_  
_𑀲 𑀯𑁂 𑀲𑀦𑁆𑀢𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇 (𑁧𑁪)_  


---
