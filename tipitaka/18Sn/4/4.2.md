# 𑁪𑁇𑁨 𑀕𑀼𑀳𑀝𑁆𑀞𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀼𑀳𑀝𑁆𑀞𑀓𑀲𑀼𑀢𑁆𑀢

(779.)

941\. _𑀲𑀢𑁆𑀢𑁄 𑀕𑀼𑀳𑀸𑀬𑀁 𑀩𑀳𑀼𑀦𑀸𑀪𑀺𑀙𑀦𑁆𑀦𑁄,_  
_𑀢𑀺𑀝𑁆𑀞𑀁 𑀦𑀭𑁄 𑀫𑁄𑀳𑀦𑀲𑁆𑀫𑀺𑀁 𑀧𑀕𑀸𑀍𑀳𑁄;_  
_𑀤𑀽𑀭𑁂 𑀯𑀺𑀯𑁂𑀓𑀸 𑀳𑀺 𑀢𑀣𑀸𑀯𑀺𑀥𑁄 𑀲𑁄,_  
_𑀓𑀸𑀫𑀸 𑀳𑀺 𑀮𑁄𑀓𑁂 𑀦 𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀳𑀸𑀬𑀸𑁇 (𑁧)_  


(780.)

942\. _𑀇𑀘𑁆𑀙𑀸𑀦𑀺𑀤𑀸𑀦𑀸 𑀪𑀯𑀲𑀸𑀢𑀩𑀤𑁆𑀥𑀸,_  
_𑀢𑁂 𑀤𑀼𑀧𑁆𑀧𑀫𑀼𑀜𑁆𑀘𑀸 𑀦 𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑁄𑀓𑁆𑀔𑀸;_  
_𑀧𑀘𑁆𑀙𑀸 𑀧𑀼𑀭𑁂 𑀯𑀸𑀧𑀺 𑀅𑀧𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀸,_  
_𑀇𑀫𑁂 𑀯 𑀓𑀸𑀫𑁂 𑀧𑀼𑀭𑀺𑀫𑁂 𑀯 𑀚𑀧𑁆𑀧𑀁𑁇 (𑁨)_  


(781.)

943\. _𑀓𑀸𑀫𑁂𑀲𑀼 𑀕𑀺𑀤𑁆𑀥𑀸 𑀧𑀲𑀼𑀢𑀸 𑀧𑀫𑀽𑀍𑀳𑀸,_  
_𑀅𑀯𑀤𑀸𑀦𑀺𑀬𑀸 𑀢𑁂 𑀯𑀺𑀲𑀫𑁂 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀸;_  
_𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀦𑀻𑀢𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀬𑀦𑁆𑀢𑀺,_  
_𑀓𑀺𑀁𑀲𑀽 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫 𑀇𑀢𑁄 𑀘𑀼𑀢𑀸𑀲𑁂𑁇 (𑁩)_  


(782.)

944\. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑁂𑀣 𑀇𑀥𑁂𑀯 𑀚𑀦𑁆𑀢𑀼,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀚𑀜𑁆𑀜𑀸 𑀯𑀺𑀲𑀫𑀦𑁆𑀢𑀺 𑀮𑁄𑀓𑁂;_  
_𑀦 𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀽 𑀯𑀺𑀲𑀫𑀁 𑀘𑀭𑁂𑀬𑁆𑀬,_  
_𑀅𑀧𑁆𑀧𑀜𑁆𑀳𑀺𑀤𑀁 𑀚𑀻𑀯𑀺𑀢𑀫𑀸𑀳𑀼 𑀥𑀻𑀭𑀸𑁇 (𑁪)_  


(783.)

945\. _𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀮𑁄𑀓𑁂 𑀧𑀭𑀺𑀨𑀦𑁆𑀤𑀫𑀸𑀦𑀁,_  
_𑀧𑀚𑀁 𑀇𑀫𑀁 𑀢𑀡𑁆𑀳𑀕𑀢𑀁 𑀪𑀯𑁂𑀲𑀼;_  
_𑀳𑀻𑀦𑀸 𑀦𑀭𑀸 𑀫𑀘𑁆𑀘𑀼𑀫𑀼𑀔𑁂 𑀮𑀧𑀦𑁆𑀢𑀺,_  
_𑀅𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀸𑀲𑁂 𑀪𑀯𑀸𑀪𑀯𑁂𑀲𑀼𑁇 (𑁫)_  


(784.)

946\. _𑀫𑀫𑀸𑀬𑀺𑀢𑁂 𑀧𑀲𑁆𑀲𑀣 𑀨𑀦𑁆𑀤𑀫𑀸𑀦𑁂,_  
_𑀫𑀘𑁆𑀙𑁂𑀯 𑀅𑀧𑁆𑀧𑁄𑀤𑀓𑁂 𑀔𑀻𑀡𑀲𑁄𑀢𑁂;_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀫𑀫𑁄 𑀘𑀭𑁂𑀬𑁆𑀬,_  
_𑀪𑀯𑁂𑀲𑀼 𑀆𑀲𑀢𑁆𑀢𑀺𑀫𑀓𑀼𑀩𑁆𑀩𑀫𑀸𑀦𑁄𑁇 (𑁬)_  


(785.)

947\. _𑀉𑀪𑁄𑀲𑀼 𑀅𑀦𑁆𑀢𑁂𑀲𑀼 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀙𑀦𑁆𑀤𑀁,_  
_𑀨𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀦𑀸𑀦𑀼𑀕𑀺𑀤𑁆𑀥𑁄;_  
_𑀬𑀤𑀢𑁆𑀢𑀕𑀭𑀳𑀻 𑀢𑀤𑀓𑀼𑀩𑁆𑀩𑀫𑀸𑀦𑁄,_  
_𑀦 𑀮𑀺𑀧𑁆𑀧𑀢𑀻 𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑁂𑀲𑀼 𑀥𑀻𑀭𑁄𑁇 (𑁭)_  


(786.)

948\. _𑀲𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀜𑁆𑀜𑀸 𑀯𑀺𑀢𑀭𑁂𑀬𑁆𑀬 𑀑𑀖𑀁,_  
_𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀼 𑀫𑀼𑀦𑀺 𑀦𑁄𑀧𑀮𑀺𑀢𑁆𑀢𑁄;_  
_𑀅𑀩𑁆𑀩𑀽𑀍𑀳𑀲𑀮𑁆𑀮𑁄 𑀘𑀭𑀫𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄,_  
_𑀦𑀸𑀲𑀻𑀲𑀢𑀻 𑀮𑁄𑀓𑀫𑀺𑀫𑀁 𑀧𑀭𑀜𑁆𑀘𑀸𑀢𑀺𑁇 (𑁮)_  


---

949\. 𑀕𑀼𑀳𑀝𑁆𑀞𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
