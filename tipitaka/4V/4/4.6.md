# 𑁪𑁇𑁬 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸

* 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸

(205.)

855\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀧𑀯𑀸𑀍𑀑 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀅𑀯𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀢𑀺, 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀉𑀧𑀯𑀸𑀍𑀑 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀅𑀯𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀉𑀧𑀯𑀸𑀍𑀅𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀞𑀫𑀁 𑀉𑀧𑀯𑀸𑀍𑀑 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁄𑀤𑁂𑀢𑀩𑁆𑀩𑁄, 𑀘𑁄𑀤𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄, 𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

* 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸

(206.)

856\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀧𑀯𑀸𑀍𑀑 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀅𑀯𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀢𑀺, 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀉𑀧𑀯𑀸𑀍𑀅𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

* 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸

857\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀧𑀯𑀸𑀍𑀑 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀅𑀯𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀢𑀺, 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀧𑀯𑀸𑀍𑀅𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀉𑀧𑀯𑀸𑀍𑀅𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

* 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸

858\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂… 𑁇

* 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸

859\. 𑀓𑀢𑀁 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑀯𑀸𑀍𑀅𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇

* 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸

(207.)

860\. 𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀅𑀲𑀼𑀘𑀺 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀮𑀚𑁆𑀚𑀻 𑀘, 𑀲𑀸𑀦𑀼𑀯𑀸𑀤𑁄 𑀘, 𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀦, 𑀲𑀫𑀕𑁆𑀕𑁂𑀦—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀸𑀦𑀺𑁇
