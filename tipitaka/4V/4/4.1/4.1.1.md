# 𑁪𑁇𑁧𑁇𑁧 𑀓𑀡𑁆𑀳𑀧𑀓𑁆𑀔𑀦𑀯𑀓

* 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬

* 𑀓𑀡𑁆𑀳𑀧𑀓𑁆𑀔𑀦𑀯𑀓

(187.)

787\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁧)

788\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁨)

789\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀲𑀁𑀖𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁩)

790\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁪)

791\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁫)

792\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀲𑀁𑀖𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁬)

793\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀁𑀖𑁄 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁭)

794\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀁𑀖𑁄 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁮)

795\. 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀁𑀖𑁄 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀲𑀁𑀖𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦𑁇 (𑁯)

---

796\. 𑀓𑀡𑁆𑀳𑀧𑀓𑁆𑀔𑀦𑀯𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
