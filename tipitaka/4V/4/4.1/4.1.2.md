# 𑁪𑁇𑁧𑁇𑁨 𑀲𑀼𑀓𑁆𑀓𑀧𑀓𑁆𑀔𑀦𑀯𑀓

### 𑀲𑀫𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬

### 𑀲𑀼𑀓𑁆𑀓𑀧𑀓𑁆𑀔𑀦𑀯𑀓

(188.)

797\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁧)

798\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁨)

799\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀲𑀁𑀖𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁩)

800\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁪)

801\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁫)

802\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀲𑀁𑀖𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁬)

803\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀁𑀖𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁭)

804\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀁𑀖𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 (𑁮)

805\. 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀲𑀁𑀖𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀁 𑀲𑀁𑀖𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺 𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀅𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑀸”𑀢𑀺𑁇 (𑁯)

---

806\. 𑀲𑀼𑀓𑁆𑀓𑀧𑀓𑁆𑀔𑀦𑀯𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
