# 𑁧𑁦𑁇𑁧𑁇𑁨 𑀅𑀝𑁆𑀞𑀕𑀭𑀼𑀥𑀫𑁆𑀫

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀞𑀫𑀪𑀸𑀡𑀯𑀸𑀭

* 𑀅𑀝𑁆𑀞𑀕𑀭𑀼𑀥𑀫𑁆𑀫

(403.)

1969\. “𑀲𑀘𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺 𑀕𑁄𑀢𑀫𑀻 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁆𑀲𑀸𑁇 𑀳𑁄𑀢𑀼 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

1970\. 𑀯𑀲𑁆𑀲𑀲𑀢𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀢𑀤𑀳𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀪𑀺𑀯𑀸𑀤𑀦𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀫𑁆𑀫𑀁 𑀲𑀸𑀫𑀻𑀘𑀺𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁧)

1971\. 𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁂 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁨)

1972\. 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀢𑁄 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀘𑁆𑀘𑀸𑀲𑀻𑀲𑀺𑀢𑀩𑁆𑀩𑀸—  𑀉𑀧𑁄𑀲𑀣𑀧𑀼𑀘𑁆𑀙𑀓𑀜𑁆𑀘, 𑀑𑀯𑀸𑀤𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑀜𑁆𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁩)

1973\. 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁—  𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸, 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸, 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁪)

1974\. 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀧𑀓𑁆𑀔𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁫)

1975\. 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁬)

1976\. 𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀓𑁂𑀦𑀘𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑁄 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁭)

1977\. 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀑𑀯𑀝𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀯𑀘𑀦𑀧𑀣𑁄, 𑀅𑀦𑁄𑀯𑀝𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀼 𑀯𑀘𑀦𑀧𑀣𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇 (𑁮)

1978\. 𑀲𑀘𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺 𑀕𑁄𑀢𑀫𑀻 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁆𑀲𑀸 𑀳𑁄𑀢𑀼 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸”𑀢𑀺𑁇

1979\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑁂 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺 𑀕𑁄𑀢𑀫𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀕𑁄𑀢𑀫𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀕𑁄𑀢𑀫𑀺, 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀸𑀯 𑀢𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇

1980\. 𑀯𑀲𑁆𑀲𑀲𑀢𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀢𑀤𑀳𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀪𑀺𑀯𑀸𑀤𑀦𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀫𑁆𑀫𑀁 𑀲𑀸𑀫𑀻𑀘𑀺𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1981\. 𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁂 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1982\. 𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀢𑁄 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀘𑁆𑀘𑀸𑀲𑀻𑀲𑀺𑀢𑀩𑁆𑀩𑀸—  𑀉𑀧𑁄𑀲𑀣𑀧𑀼𑀘𑁆𑀙𑀓𑀜𑁆𑀘, 𑀑𑀯𑀸𑀤𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑀜𑁆𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1983\. 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁—  𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸, 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸, 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1984\. 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀧𑀓𑁆𑀔𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1985\. 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1986\. 𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀓𑁂𑀦𑀘𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑀩𑁆𑀩𑁄 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1987\. 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀑𑀯𑀝𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀯𑀘𑀦𑀧𑀣𑁄, 𑀅𑀦𑁄𑀯𑀝𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀼 𑀯𑀘𑀦𑀧𑀣𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀓𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁄𑁇

1988\. 𑀲𑀘𑁂 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀕𑁄𑀢𑀫𑀺, 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀸𑀯 𑀢𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸”𑀢𑀺𑁇

1989\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀢𑁆𑀣𑀻 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀸 𑀤𑀳𑀭𑁄, 𑀬𑀼𑀯𑀸, 𑀫𑀡𑁆𑀟𑀦𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀲𑀻𑀲𑀁𑀦𑀳𑀸𑀢𑁄 𑀉𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀁 𑀯𑀸 𑀯𑀲𑁆𑀲𑀺𑀓𑀫𑀸𑀮𑀁 𑀯𑀸 𑀅𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀓𑀫𑀸𑀮𑀁 𑀯𑀸 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀉𑀪𑁄𑀳𑀺 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑁂 𑀲𑀺𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀦𑀦𑁆𑀤 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑁂”𑀢𑀺𑁇

1990\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑀸 𑀕𑁄𑀢𑀫𑀺𑀬𑀸 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀸, 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀪𑀕𑀯𑀢𑁄 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸”𑀢𑀺𑁇

1991\. “𑀲𑀘𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀸𑀮𑀪𑀺𑀲𑁆𑀲 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀪𑀯𑀺𑀲𑁆𑀲, 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀦 𑀤𑀸𑀦𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑁂𑀯 𑀤𑀸𑀦𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄 𑀞𑀲𑁆𑀲𑀢𑀺𑁇

1992\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀩𑀳𑀼𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀧𑀼𑀭𑀺𑀲𑀓𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀲𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀬𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀘𑁄𑀭𑁂𑀳𑀺 𑀓𑀼𑀫𑁆𑀪𑀣𑁂𑀦𑀓𑁂𑀳𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀮𑀪𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀦 𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀳𑁄𑀢𑀺𑁇

1993\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀲𑀸𑀮𑀺𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀲𑁂𑀢𑀝𑁆𑀝𑀺𑀓𑀸 𑀦𑀸𑀫 𑀭𑁄𑀕𑀚𑀸𑀢𑀺 𑀦𑀺𑀧𑀢𑀢𑀺, 𑀏𑀯𑀁 𑀢𑀁 𑀲𑀸𑀮𑀺𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀦 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀮𑀪𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀦 𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀳𑁄𑀢𑀺𑁇

1994\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀘𑁆𑀙𑀼𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀸𑀫 𑀭𑁄𑀕𑀚𑀸𑀢𑀺 𑀦𑀺𑀧𑀢𑀢𑀺, 𑀏𑀯𑀁 𑀢𑀁 𑀉𑀘𑁆𑀙𑀼𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀦 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀮𑀪𑀢𑀺 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀦 𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀳𑁄𑀢𑀺𑁇

1995\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑁄 𑀢𑀍𑀆𑀓𑀲𑁆𑀲 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀆𑀍𑀇𑀁 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬 𑀬𑀸𑀯𑀤𑁂𑀯 𑀉𑀤𑀓𑀲𑁆𑀲 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀸𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀬𑀸 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑀸”𑀢𑀺𑁇

---

1996\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
