# 𑁭𑁇𑁩𑁇𑁨 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀳𑀸

### 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀢𑀢𑀺𑀬𑀪𑀸𑀡𑀯𑀸𑀭

### 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀳𑀸

(351.)

1593\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀭𑀸𑀚𑀻𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑀁𑀖𑀪𑁂𑀤𑁄? 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀘 𑀧𑀦 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀘𑀸”𑀢𑀺?

1594\. “𑀏𑀓𑀢𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀏𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀢𑁄 𑀤𑁆𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑀁𑀖𑀪𑁂𑀤𑁄𑁇 𑀏𑀓𑀢𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀤𑁆𑀯𑁂 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀢𑁄 𑀤𑁆𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑀁𑀖𑀪𑁂𑀤𑁄𑁇 𑀏𑀓𑀢𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀤𑁆𑀯𑁂 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀢𑁄 𑀢𑀬𑁄, 𑀙𑀝𑁆𑀞𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑀁𑀖𑀪𑁂𑀤𑁄𑁇 𑀏𑀓𑀢𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀢𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀢𑁄 𑀢𑀬𑁄, 𑀲𑀢𑁆𑀢𑀫𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑀁𑀖𑀪𑁂𑀤𑁄𑁇 𑀏𑀓𑀢𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀢𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄, 𑀅𑀝𑁆𑀞𑀫𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑀁𑀖𑀪𑁂𑀤𑁄𑁇 𑀏𑀓𑀢𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄, 𑀦𑀯𑀫𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀘𑁇 𑀦𑀯𑀦𑁆𑀦𑀁 𑀯𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀅𑀢𑀺𑀭𑁂𑀓𑀦𑀯𑀦𑁆𑀦𑀁 𑀯𑀸 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀘𑁇 𑀦 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀺, 𑀦 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀪𑁂𑀤𑀸𑀬 𑀧𑀭𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀓𑀢𑀢𑁆𑀢𑁄, 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑁄, 𑀲𑀫𑀸𑀦𑀲𑀻𑀫𑀸𑀬𑀁 𑀞𑀺𑀢𑁄, 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀻”𑀢𑀺𑁇

(352.)

1595\. “𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺?

1596\. “𑀇𑀥𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧)

1597\. 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁨)

1598\. 𑀅𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁩)

1599\. 𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁪)

1600\. 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁫)

1601\. 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁬)

1602\. 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁭)

1603\. 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁮)

1604\. 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁯)

1605\. 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁦)

1606\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁧)

1607\. 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁨)

1608\. 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁩)

1609\. 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁪)

1610\. 𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁫)

1611\. 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁬)

1612\. 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁭)

1613\. 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁮)

1614\. 𑀢𑁂 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀅𑀧𑀓𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀯𑀧𑀓𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀆𑀯𑁂𑀦𑀺𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀆𑀯𑁂𑀦𑀺𑀁 𑀧𑀯𑀸𑀭𑀡𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀆𑀯𑁂𑀦𑀺𑀁 𑀲𑀁𑀖𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑁄 𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

(353.)

1615\. “𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀲𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺?

1616\. “𑀇𑀥𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧)

1617\. 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁨)

1618\. 𑀅𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁩)

1619\. 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁪)

1620\. 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁫)

1621\. 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁬)

1622\. 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁭)

1623\. 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁮)

1624\. 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁯)

1625\. 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁦)

1626\. 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁧)

1627\. 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁨)

1628\. 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁩)

1629\. 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁪)

1630\. 𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁫)

1631\. 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁬)

1632\. 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁭)

1633\. 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁮)

1634\. 𑀢𑁂 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀦 𑀅𑀧𑀓𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀅𑀯𑀧𑀓𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀆𑀯𑁂𑀦𑀺𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀆𑀯𑁂𑀦𑀺𑀁 𑀧𑀯𑀸𑀭𑀡𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀆𑀯𑁂𑀦𑀺𑀁 𑀲𑀁𑀖𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑁄 𑀲𑀫𑀕𑁆𑀕𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

(354.)

1635\. “𑀲𑀫𑀕𑁆𑀕𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀓𑀺𑀁 𑀲𑁄 𑀧𑀲𑀯𑀢𑀻”𑀢𑀺? “𑀲𑀫𑀕𑁆𑀕𑀁 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑀺𑀓𑀁 𑀓𑀺𑀩𑁆𑀩𑀺𑀲𑀁 𑀧𑀲𑀯𑀢𑀺, 𑀓𑀧𑁆𑀧𑀁 𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀧𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇

1636\. _𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄,_  
_𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄;_  
_𑀯𑀕𑁆𑀕𑀭𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀧𑀥𑀁𑀲𑀢𑀺;_  
_𑀲𑀁𑀖𑀁 𑀲𑀫𑀕𑁆𑀕𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀁 𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀧𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


1637\. “𑀪𑀺𑀦𑁆𑀦𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀲𑀫𑀕𑁆𑀕𑀁 𑀓𑀢𑁆𑀯𑀸 𑀓𑀺𑀁 𑀲𑁄 𑀧𑀲𑀯𑀢𑀻”𑀢𑀺? “𑀪𑀺𑀦𑁆𑀦𑀁 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀁 𑀲𑀫𑀕𑁆𑀕𑀁 𑀓𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺, 𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀢𑀻𑀢𑀺𑁇

1638\. _𑀲𑀼𑀔𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑀸𑀫𑀕𑁆𑀕𑀻,_  
_𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀜𑁆𑀘 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑁄;_  
_𑀲𑀫𑀕𑁆𑀕𑀭𑀢𑁄 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀦 𑀥𑀁𑀲𑀢𑀺;_  
_𑀲𑀁𑀖𑀁 𑀲𑀫𑀕𑁆𑀕𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀢𑀻”𑀢𑀺𑁇_

1639\. “𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄”𑀢𑀺? “𑀲𑀺𑀬𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄”𑀢𑀺𑁇

1640\. “𑀲𑀺𑀬𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀦 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀦 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀦 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄”𑀢𑀺? “𑀲𑀺𑀬𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀦 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀦 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀦 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄”𑀢𑀺𑁇

1641\. “𑀓𑀢𑀫𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄”𑀢𑀺? “𑀇𑀥𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀔𑀦𑁆𑀢𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀭𑀼𑀘𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀪𑀸𑀯𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇

1642\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀔𑀦𑁆𑀢𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀭𑀼𑀘𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀪𑀸𑀯𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇

1643\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀔𑀦𑁆𑀢𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀭𑀼𑀘𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀪𑀸𑀯𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇

1644\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺 𑀪𑁂𑀤𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺 𑀪𑁂𑀤𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀪𑁂𑀤𑁂 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀪𑁂𑀤𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀪𑁂𑀤𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀔𑀦𑁆𑀢𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀭𑀼𑀘𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀪𑀸𑀯𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇

1645\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀪𑁂𑀤𑁂 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀪𑁂𑀤𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺…  𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀪𑁂𑀤𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑁄, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀔𑀦𑁆𑀢𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀭𑀼𑀘𑀺𑀁, 𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀪𑀸𑀯𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄”𑀢𑀺𑁇

1647\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑁂𑀤𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀁, 𑀅𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀔𑀦𑁆𑀢𑀺𑀁, 𑀅𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀭𑀼𑀘𑀺𑀁, 𑀅𑀯𑀺𑀦𑀺𑀥𑀸𑀬 𑀪𑀸𑀯𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀲𑀮𑀸𑀓𑀁 𑀕𑀸𑀳𑁂𑀢𑀺—  ‘𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀅𑀬𑀁 𑀯𑀺𑀦𑀬𑁄, 𑀇𑀤𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀇𑀫𑀁 𑀕𑀡𑁆𑀳𑀣, 𑀇𑀫𑀁 𑀭𑁄𑀘𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀦 𑀆𑀧𑀸𑀬𑀺𑀓𑁄, 𑀦 𑀦𑁂𑀭𑀬𑀺𑀓𑁄, 𑀦 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄, 𑀦 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄”𑀢𑀺𑁇

1649\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1650\. _𑀅𑀦𑀼𑀧𑀺𑀬𑁂 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸,_  
_𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑁄 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺;_  
_𑀓𑀲𑀸 𑀯𑀧𑀸 𑀅𑀪𑀺 𑀦𑀺𑀦𑁆𑀦𑁂,_  
_𑀦𑀺𑀤𑁆𑀥𑀸 𑀮𑀸𑀯𑁂 𑀘 𑀉𑀩𑁆𑀩𑀳𑁂𑁇_  


1651\. _𑀧𑀼𑀜𑁆𑀚𑀫𑀤𑁆𑀤𑀧𑀮𑀸𑀮𑀜𑁆𑀘,_  
_𑀪𑀼𑀲𑀑𑀨𑀼𑀡𑀦𑀻𑀳𑀭𑁂;_  
_𑀆𑀬𑀢𑀺𑀫𑁆𑀧𑀺 𑀦 𑀔𑀻𑀬𑀦𑁆𑀢𑀺,_  
_𑀧𑀺𑀢𑀭𑁄 𑀘 𑀧𑀺𑀢𑀸𑀫𑀳𑀸𑁇_  


1652\. _𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀘,_  
_𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀼 𑀓𑀺𑀫𑀺𑀮𑁄;_  
_𑀲𑀓𑁆𑀬𑀫𑀸𑀦𑁄 𑀘 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀁,_  
_𑀧𑀭𑀺𑀳𑀸𑀬𑀺 𑀓𑀓𑀼𑀥𑁂𑀦 𑀘𑁇_  


1653\. _𑀧𑀓𑀸𑀲𑁂𑀲𑀺 𑀧𑀺𑀢𑀼𑀦𑁄 𑀘,_  
_𑀧𑀼𑀭𑀺𑀲𑁂 𑀲𑀺𑀮𑀁 𑀦𑀸𑀍𑀆𑀕𑀺𑀭𑀺𑀁;_  
_𑀢𑀺𑀓𑀧𑀜𑁆𑀘𑀕𑀭𑀼𑀓𑁄 𑀔𑁄,_  
_𑀪𑀺𑀦𑁆𑀤𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑁂𑀦 𑀘;_  
_𑀢𑀬𑁄 𑀅𑀝𑁆𑀞 𑀧𑀼𑀦 𑀢𑀻𑀡𑀺,_  
_𑀭𑀸𑀚𑀺 𑀪𑁂𑀤𑀸 𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄𑀢𑀺𑁇_
