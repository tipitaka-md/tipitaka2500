# 𑁩𑁇𑁧𑁇𑁧𑁪 𑀧𑀓𑁆𑀔𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦

* 𑀧𑀓𑁆𑀔𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦

578\. 𑀲𑁄 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄𑀳𑀁 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄—

579\. 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀸𑀬𑀺𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄𑀳𑀁 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀁 𑀬𑀸𑀘𑀸𑀫𑀻’𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

580\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

581\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀧𑀓𑁆𑀔𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇
