# 𑁩𑁇𑁧𑁇𑁨 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀅𑀩𑁆𑀪𑀸𑀦

### 𑀲𑀫𑀼𑀘𑁆𑀘𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺

### 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀅𑀩𑁆𑀪𑀸𑀦

(100.)

505\. 𑀲𑁄 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄𑀳𑀁 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀩𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑁄—

506\. 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀸𑀬𑀺𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁, 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀬𑀸𑀘𑀸𑀫𑀺𑁇

507\. 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄𑀳𑀁 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀬𑀸𑀘𑀸𑀫𑀺𑁇

508\. 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄𑀳𑀁 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀬𑀸𑀘𑀸𑀫𑀻’𑀢𑀺𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(101.)

509\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

510\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀩𑁆𑀪𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

511\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀩𑁆𑀪𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

512\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀘𑀺𑀡𑁆𑀡𑀫𑀸𑀦𑀢𑁆𑀢𑁄 𑀲𑀁𑀖𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀩𑁆𑀪𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

513\. 𑀅𑀩𑁆𑀪𑀺𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀉𑀤𑀸𑀬𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇
