# 𑁩𑁇𑁧𑁇𑁧𑁮 𑀫𑀸𑀦𑀢𑁆𑀢𑀘𑀸𑀭𑀺𑀓𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀤𑀺

### 𑀲𑀫𑀼𑀘𑁆𑀘𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺

### 𑀫𑀸𑀦𑀢𑁆𑀢𑀘𑀸𑀭𑀺𑀓𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀤𑀺

(130.)

599\. 𑀲𑁄 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁…𑀧𑁂…  𑀲𑁄𑀳𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀁 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀁 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀉𑀤𑀸𑀬𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀢𑁆𑀯𑀸 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇

600\. 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇

601\. 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀤𑁂𑀢𑀺…𑀧𑁂… 𑁇

602\. 𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀁𑀖𑁂𑀦 𑀉𑀤𑀸𑀬𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀺𑀓𑀸𑀬 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀫𑀸𑀦𑀢𑁆𑀢𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇
