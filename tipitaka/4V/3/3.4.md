# 𑁩𑁇𑁪 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓

### 𑀲𑀫𑀼𑀘𑁆𑀘𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓

(165.)

691\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 (𑁧)

692\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀣𑀸𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 (𑁨)

693\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀣𑀸𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 (𑁩)

694\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀣𑀸𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 (𑁯)

695\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑁄…𑀧𑁂…  𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀘𑀭𑀦𑁆𑀢𑁄…𑀧𑁂…  (𑀬𑀣𑀸𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀢𑀣𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁) 𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑁄 𑀅𑀦𑁆𑀢𑀭𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺…𑀧𑁂…  𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄…𑀧𑁂…  𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑀭𑀺𑀫𑀸𑀡𑀸𑀬𑁄𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑀧𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀣𑀸𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 (𑁩𑁬)
