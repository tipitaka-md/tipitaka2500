# 𑁩𑁇𑁨𑁇𑁩 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲

### 𑀲𑀫𑀼𑀘𑁆𑀘𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀭𑀺𑀯𑀸𑀲

### 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲

(138.)

627\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆’”𑀢𑀺𑁇

628\. 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”’𑀢𑀺𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—

629\. 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸, 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸, 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸, 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸, 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀸𑀫𑀻’𑀢𑀺𑁇

630\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

631\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

632\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  𑀅𑀳𑀁 𑀔𑁄 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀺𑀁 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄𑀳𑀁 𑀲𑀁𑀖𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀁𑀖𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀮𑀚𑁆𑀚𑀻𑀥𑀫𑁆𑀫𑁄 𑀑𑀓𑁆𑀓𑀫𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑀲𑁆𑀲 𑀤𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

633\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂… 𑁇

634\. 𑀤𑀺𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀢𑀭𑀺𑀲𑁆𑀲𑀸𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀬 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀧𑀭𑀺𑀯𑀸𑀲𑁄𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀢𑀤𑀼𑀧𑀸𑀤𑀸𑀬 𑀤𑁆𑀯𑁂 𑀫𑀸𑀲𑀸 𑀧𑀭𑀺𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀸𑁇
