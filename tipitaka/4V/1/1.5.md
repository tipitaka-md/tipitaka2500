# 𑁧𑁇𑁫 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫

### 𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫

(46.)

213\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀙𑀦𑁆𑀦𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀙𑀦𑁆𑀦𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀇𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

### 𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫

214\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀇𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁? 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀙𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀼—  𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀲𑀁𑀖𑁂𑀦𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀧𑀞𑀫𑀁 𑀙𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑁄𑀤𑁂𑀢𑀩𑁆𑀩𑁄, 𑀘𑁄𑀤𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄, 𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

### 𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫

(47.)

215\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀙𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀙𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬—  𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀲𑀁𑀖𑁂𑀦𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

### 𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫

216\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀙𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇 𑀲𑀁𑀖𑁄 𑀙𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺—  𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀲𑀁𑀖𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀙𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀁—  𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀲𑀁𑀖𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

### 𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫

217\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀙𑀦𑁆𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇 𑀲𑀁𑀖𑁄 𑀙𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺—  𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀲𑀁𑀖𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀙𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀁—  𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀲𑀁𑀖𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

### 𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫

218\. 𑀓𑀢𑀁 𑀲𑀁𑀖𑁂𑀦 𑀙𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁—  𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀲𑀁𑀖𑁂𑀦𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇
