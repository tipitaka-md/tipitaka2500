# 𑁬𑁇𑁩𑁇𑁫 𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑀸𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀲𑁂𑀦𑀸𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀢𑀢𑀺𑀬𑀪𑀸𑀡𑀯𑀸𑀭

### 𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑀸𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(325.)

1430. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀍𑀅𑀯𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀁 𑀳𑁄𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑀁 𑀓𑀸𑀢𑀼𑀁; 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑀪𑀢𑁆𑀢𑀁 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀁 𑀲𑀮𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑀺𑀓𑀁 𑀧𑀸𑀝𑀺𑀧𑀤𑀺𑀓𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀪𑀢𑁆𑀢𑀁 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀁 𑀲𑀮𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀺𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑀺𑀓𑀁 𑀧𑀸𑀝𑀺𑀧𑀤𑀺𑀓𑀦𑁆”𑀢𑀺𑁇
