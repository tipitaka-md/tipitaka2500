# 𑁨𑁇𑁩 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

### 𑀬𑀼𑀕𑀦𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

(17.)

1416. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸”𑁇

### 𑀬𑀼𑀕𑀦𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

1417. **𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸**𑀢𑀺 𑀓𑁂𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸? 𑀩𑁄𑀥𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀅𑀦𑀼𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 (𑁧)

### 𑀬𑀼𑀕𑀦𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

1418. 𑀩𑀼𑀚𑁆𑀛𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀅𑀦𑀼𑀩𑀼𑀚𑁆𑀛𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 (𑁨)

### 𑀬𑀼𑀕𑀦𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

1419. 𑀩𑁄𑀥𑁂𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀅𑀦𑀼𑀩𑁄𑀥𑁂𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀧𑀝𑀺𑀩𑁄𑀥𑁂𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 𑀲𑀫𑁆𑀩𑁄𑀥𑁂𑀦𑁆𑀢𑀻𑀢𑀺—  𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 (𑁩)

### 𑀬𑀼𑀕𑀦𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

1420. 𑀩𑁄𑀥𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀅𑀦𑀼𑀩𑁄𑀥𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀧𑀝𑀺𑀩𑁄𑀥𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀲𑀫𑁆𑀩𑁄𑀥𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 (𑁪)

### 𑀬𑀼𑀕𑀦𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

1421. 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀅𑀦𑀼𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀧𑀝𑀺𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 (𑁫) (𑀧𑀜𑁆𑀘𑀫𑀘𑀢𑀼𑀓𑁆𑀓𑀁)

### 𑀬𑀼𑀕𑀦𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀓𑀣𑀸

1422. 𑀩𑀼𑀤𑁆𑀥𑀺𑀮𑀪𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀩𑀼𑀤𑁆𑀥𑀺𑀧𑀝𑀺𑀮𑀪𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀩𑀼𑀤𑁆𑀥𑀺𑀭𑁄𑀧𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀩𑀼𑀤𑁆𑀥𑀺𑀅𑀪𑀺𑀭𑁄𑀧𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀩𑀼𑀤𑁆𑀥𑀺𑀧𑀸𑀧𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀩𑀼𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀧𑀸𑀧𑀦𑀝𑁆𑀞𑁂𑀦 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸𑁇 (𑀙𑀓𑁆𑀓𑀁)
