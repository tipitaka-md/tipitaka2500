# 𑁧𑁇𑁧𑁇𑁭 𑀪𑀗𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀜𑀸𑀡𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀜𑀸𑀡𑀓𑀣𑀸

* 𑀪𑀗𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀜𑀸𑀡𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(51.)

366\. 𑀓𑀣𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀪𑀗𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁂 𑀧𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁂 𑀜𑀸𑀡𑀁? 𑀭𑀽𑀧𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀁 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇

367\. **𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀻**𑀢𑀺, 𑀓𑀣𑀁 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺? 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀦𑀺𑀘𑁆𑀘𑀢𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀲𑀼𑀔𑀢𑁄𑁇 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀅𑀢𑁆𑀢𑀢𑁄𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑁄 𑀦𑀦𑁆𑀤𑀢𑀺, 𑀯𑀺𑀭𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀭𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀭𑁄𑀥𑁂𑀢𑀺, 𑀦𑁄 𑀲𑀫𑀼𑀤𑁂𑀢𑀺𑁇 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀆𑀤𑀺𑀬𑀢𑀺𑁇

(52.)

368\. 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀦𑀦𑁆𑀤𑀺𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀯𑀺𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀦𑀺𑀭𑁄𑀥𑁂𑀦𑁆𑀢𑁄 𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀆𑀤𑀸𑀦𑀁 𑀧𑀚𑀳𑀢𑀺𑁇

369\. 𑀯𑁂𑀤𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸…  𑀘𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀁 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀻𑀢𑀺, 𑀓𑀣𑀁 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺? 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀦𑀺𑀘𑁆𑀘𑀢𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀲𑀼𑀔𑀢𑁄𑁇 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀅𑀢𑁆𑀢𑀢𑁄𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑁄 𑀦𑀦𑁆𑀤𑀢𑀺𑁇 𑀯𑀺𑀭𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀭𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀭𑁄𑀥𑁂𑀢𑀺, 𑀦𑁄 𑀲𑀫𑀼𑀤𑁂𑀢𑀺𑁇 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀆𑀤𑀺𑀬𑀢𑀺𑁇

370\. 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀲𑀜𑁆𑀜𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀦𑀦𑁆𑀤𑀺𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀯𑀺𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀭𑀸𑀕𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀦𑀺𑀭𑁄𑀥𑁂𑀦𑁆𑀢𑁄 𑀲𑀫𑀼𑀤𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀆𑀤𑀸𑀦𑀁 𑀧𑀚𑀳𑀢𑀺𑁇

371\. _𑀯𑀢𑁆𑀣𑀼𑀲𑀗𑁆𑀓𑀫𑀦𑀸 𑀘𑁂𑀯,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀯𑀺𑀯𑀝𑁆𑀝𑀦𑀸;_  
_𑀆𑀯𑀚𑁆𑀚𑀦𑀸 𑀩𑀮𑀜𑁆𑀘𑁂𑀯,_  
_𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑁇_  


372\. _𑀆𑀭𑀫𑁆𑀫𑀡𑀅𑀦𑁆𑀯𑀬𑁂𑀦,_  
_𑀉𑀪𑁄 𑀏𑀓𑀯𑀯𑀢𑁆𑀣𑀦𑀸;_  
_𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑀸,_  
_𑀯𑀬𑀮𑀓𑁆𑀔𑀡𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑁇_  


373\. _𑀆𑀭𑀫𑁆𑀫𑀡𑀜𑁆𑀘 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸,_  
_𑀪𑀗𑁆𑀕𑀜𑁆𑀘 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀼𑀜𑁆𑀜𑀢𑁄 𑀘 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑁇_  


374\. _𑀓𑀼𑀲𑀮𑁄 𑀢𑀻𑀲𑀼 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀲𑀼,_  
_𑀘𑀢𑀲𑁆𑀲𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀲𑀼;_  
_𑀢𑀬𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀓𑀼𑀲𑀮𑀢𑀸,_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀻𑀲𑀼 𑀦 𑀓𑀫𑁆𑀧𑀢𑀻𑀢𑀺𑁇_  


375\. 𑀢𑀁 𑀜𑀸𑀢𑀝𑁆𑀞𑁂𑀦 𑀜𑀸𑀡𑀁, 𑀧𑀚𑀸𑀦𑀦𑀝𑁆𑀞𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑁇 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  “𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸 𑀪𑀗𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑁂 𑀧𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁂 𑀜𑀸𑀡𑀁”𑁇

---

376\. 𑀪𑀗𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸𑀜𑀸𑀡𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇
