# 𑁬𑁇𑁧𑁇𑁮 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀺𑀮𑁂𑀲𑀯𑀕𑁆𑀕

### 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(329.)

1107\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸𑀬 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄𑁇 𑀬𑁄 𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸𑀬…  𑀬𑁄 𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸𑀬…  𑀬𑁄 𑀭𑀲𑀢𑀡𑁆𑀳𑀸𑀬…  𑀬𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸𑀬…  𑀬𑁄 𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸𑀬 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄…𑀧𑁂…  𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀽”𑀢𑀺𑁇
