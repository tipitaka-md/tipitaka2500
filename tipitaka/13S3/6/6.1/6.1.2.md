# 𑁬𑁇𑁧𑁇𑁨 𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

* 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀓𑀺𑀮𑁂𑀲𑀯𑀕𑁆𑀕

* 𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

(323.)

1095\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄𑁇 𑀬𑁄 𑀲𑀤𑁆𑀤𑁂𑀲𑀼…  𑀬𑁄 𑀕𑀦𑁆𑀥𑁂𑀲𑀼…  𑀬𑁄 𑀭𑀲𑁂𑀲𑀼…  𑀬𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀲𑀼…  𑀬𑁄 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀇𑀫𑁂𑀲𑀼 𑀙𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼 𑀘𑁂𑀢𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀦𑀺𑀦𑁆𑀦𑀜𑁆𑀘𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑀁 𑀔𑀸𑀬𑀢𑀺, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀽”𑀢𑀺𑁇
