# 𑁬𑁇𑁧𑁇𑁫 𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀓𑀺𑀮𑁂𑀲𑀯𑀕𑁆𑀕

### 𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀲𑀼𑀢𑁆𑀢

(326.)

1101\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄𑁇 𑀬𑁄 𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀬𑁄 𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀬𑁄 𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀬𑁄 𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀬𑁄 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑁄𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄…𑀧𑁂…  𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀽”𑀢𑀺𑁇
