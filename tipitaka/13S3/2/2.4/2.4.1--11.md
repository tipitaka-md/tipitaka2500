# 𑁨𑁇𑁪𑁇𑁧--𑁧𑁧 𑀫𑀸𑀭𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀏𑀓𑀸𑀤𑀲𑀓

### 𑀭𑀸𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀉𑀧𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀭𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀏𑀓𑀸𑀤𑀲𑀓

(194.)

892\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀭𑀸𑀥𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀫𑀸𑀭𑁄; 𑀢𑀢𑁆𑀭 𑀢𑁂 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑁄 𑀘, 𑀭𑀸𑀥, 𑀫𑀸𑀭𑁄? 𑀭𑀽𑀧𑀁 𑀔𑁄, 𑀭𑀸𑀥, 𑀫𑀸𑀭𑁄; 𑀢𑀢𑁆𑀭 𑀢𑁂 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀫𑀸𑀭𑁄; 𑀢𑀢𑁆𑀭 𑀢𑁂 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀫𑀸𑀭𑁄; 𑀢𑀢𑁆𑀭 𑀢𑁂 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 (𑁧)

(195.)

893\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀫𑀸𑀭𑀥𑀫𑁆𑀫𑁄; 𑀢𑀢𑁆𑀭 𑀢𑁂 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇 (𑁨)

(196.)

894\. “𑀬𑀁 𑀔𑁄, 𑀭𑀸𑀥, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁…𑀧𑁂… 𑁇 (𑁩)

(197.)

895\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀅𑀦𑀺𑀘𑁆𑀘𑀥𑀫𑁆𑀫𑁄…𑀧𑁂… 𑁇 (𑁪)

(198.)

896\. “𑀬𑀁 𑀔𑁄, 𑀭𑀸𑀥, 𑀤𑀼𑀓𑁆𑀔𑀁…𑀧𑁂… 𑁇 (𑁫)

(199.)

897\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁄…𑀧𑁂… 𑁇 (𑁬)

(200.)

898\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀅𑀦𑀢𑁆𑀢𑀸…𑀧𑁂… 𑁇 (𑁭)

(201.)

899\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀅𑀦𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁄…𑀧𑁂… 𑁇 (𑁮)

(202.)

900\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀔𑀬𑀥𑀫𑁆𑀫𑁄…𑀧𑁂… 𑁇 (𑁯)

(203.)

901\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀯𑀬𑀥𑀫𑁆𑀫𑁄…𑀧𑁂… 𑁇 (𑁧𑁦)

(204.)

902\. “𑀬𑁄 𑀔𑁄, 𑀭𑀸𑀥, 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑁄; 𑀢𑀢𑁆𑀭 𑀢𑁂 𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄, 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇 (𑁧𑁧)
