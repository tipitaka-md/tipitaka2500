# 𑁧𑁧𑁇𑁧𑁇𑁧 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀯𑀮𑀸𑀳𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑀮𑀸𑀳𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(550.)

1216\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑁂 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀸? 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀢𑀯𑀮𑀸𑀳𑀓𑀸 𑀤𑁂𑀯𑀸; 𑀲𑀦𑁆𑀢𑀺 𑀉𑀡𑁆𑀳𑀯𑀮𑀸𑀳𑀓𑀸 𑀤𑁂𑀯𑀸; 𑀲𑀦𑁆𑀢𑀺 𑀅𑀩𑁆𑀪𑀯𑀮𑀸𑀳𑀓𑀸 𑀤𑁂𑀯𑀸; 𑀲𑀦𑁆𑀢𑀺 𑀯𑀸𑀢𑀯𑀮𑀸𑀳𑀓𑀸 𑀤𑁂𑀯𑀸; 𑀲𑀦𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀯𑀮𑀸𑀳𑀓𑀸 𑀤𑁂𑀯𑀸—  𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀸’”𑀢𑀺𑁇
