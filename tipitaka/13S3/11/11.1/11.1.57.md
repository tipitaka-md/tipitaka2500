# 𑁧𑁧𑁇𑁧𑁇𑁫𑁭 𑀯𑀲𑁆𑀲𑀯𑀮𑀸𑀳𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀮𑀸𑀳𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑀮𑀸𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀯𑀲𑁆𑀲𑀯𑀮𑀸𑀳𑀓𑀲𑀼𑀢𑁆𑀢

(606.)

1231\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑁂𑀓𑀤𑀸 𑀤𑁂𑀯𑁄 𑀯𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀯𑀲𑁆𑀲𑀯𑀮𑀸𑀳𑀓𑀸 𑀦𑀸𑀫 𑀤𑁂𑀯𑀸𑁇 𑀢𑁂𑀲𑀁 𑀬𑀤𑀸 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀲𑀓𑀸𑀬 𑀭𑀢𑀺𑀬𑀸 𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀘𑁂𑀢𑁄𑀧𑀡𑀺𑀥𑀺𑀫𑀦𑁆𑀯𑀸𑀬 𑀤𑁂𑀯𑁄 𑀯𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦𑁂𑀓𑀤𑀸 𑀤𑁂𑀯𑁄 𑀯𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀁𑁇

---

1232\. 𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀯𑀮𑀸𑀳𑀓𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

1233\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1234\. _𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀲𑀼𑀘𑀭𑀺𑀢𑀜𑁆𑀘,_  
_𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀧𑀜𑁆𑀜𑀸𑀲𑀁;_  
_𑀲𑀻𑀢𑀁 𑀉𑀡𑁆𑀳𑀜𑁆𑀘 𑀅𑀩𑁆𑀪𑀜𑁆𑀘,_  
_𑀯𑀸𑀢𑀯𑀲𑁆𑀲𑀯𑀮𑀸𑀳𑀓𑀸𑀢𑀺𑁇_
