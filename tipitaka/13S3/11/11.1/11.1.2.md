# 𑁧𑁧𑁇𑁧𑁇𑁨 𑀲𑀼𑀘𑀭𑀺𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀮𑀸𑀳𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑀮𑀸𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀘𑀭𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(551.)

1218\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑁂𑀦 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀢𑀺, 𑀫𑀦𑀲𑀸 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺—  ‘𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀸 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀩𑀳𑀼𑀮𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀯𑀮𑀸𑀳𑀓𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
