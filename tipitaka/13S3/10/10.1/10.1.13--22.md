# 𑁧𑁦𑁇𑁧𑁇𑁧𑁩--𑁨𑁨 𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢𑀤𑀲𑀓

### 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀓𑀸𑀬𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀯𑀕𑁆𑀕

### 𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢𑀤𑀲𑀓

(450--459.)

1208\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑁂𑀦 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀢𑀺, 𑀫𑀦𑀲𑀸 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀸 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀩𑀳𑀼𑀮𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…  𑀯𑀢𑁆𑀣𑀁 𑀤𑁂𑀢𑀺…  𑀬𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…  𑀫𑀸𑀮𑀁 𑀤𑁂𑀢𑀺…  𑀕𑀦𑁆𑀥𑀁 𑀤𑁂𑀢𑀺…  𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀤𑁂𑀢𑀺…  𑀲𑁂𑀬𑁆𑀬𑀁 𑀤𑁂𑀢𑀺…  𑀆𑀯𑀲𑀣𑀁 𑀤𑁂𑀢𑀺…  𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁 𑀤𑁂𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀽𑀮𑀕𑀦𑁆𑀥𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
