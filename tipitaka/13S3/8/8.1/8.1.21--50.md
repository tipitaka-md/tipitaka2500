# 𑁮𑁇𑁧𑁇𑁨𑁧--𑁫𑁦 𑀚𑀮𑀸𑀩𑀼𑀚𑀸𑀤𑀺𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢𑀢𑁆𑀢𑀺𑀁𑀲𑀓

* 𑀦𑀸𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀸𑀕𑀯𑀕𑁆𑀕

* 𑀚𑀮𑀸𑀩𑀼𑀚𑀸𑀤𑀺𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢𑀢𑁆𑀢𑀺𑀁𑀲𑀓

(362--391.)

1178\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀚𑀮𑀸𑀩𑀼𑀚𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁…𑀧𑁂…  𑀲𑀁𑀲𑁂𑀤𑀚𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁…𑀧𑁂…  𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺?

1179\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑁂𑀦 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀻, 𑀫𑀦𑀲𑀸 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀻𑁇 𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺—  ‘𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀦𑀸𑀕𑀸 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀩𑀳𑀼𑀮𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁 𑀤𑁂𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

1180\. (𑀇𑀫𑀺𑀦𑀸 𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑁂𑀦 𑀤𑀲 𑀤𑀲 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀏𑀯𑀁 𑀘𑀢𑀽𑀲𑀼 𑀬𑁄𑀦𑀻𑀲𑀼 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀧𑀼𑀭𑀺𑀫𑁂𑀳𑀺 𑀧𑀦 𑀤𑀲𑀳𑀺 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀳 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀡𑀸𑀲𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸𑀢𑀺𑁇)

1182\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1183\. _𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀧𑀡𑀻𑀢𑀢𑀭𑀁,_  
_𑀘𑀢𑀼𑀭𑁄 𑀘 𑀉𑀧𑁄𑀲𑀣𑀸;_  
_𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀁 𑀘𑀢𑀼𑀭𑁄 𑀘,_  
_𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀸 𑀘 𑀢𑀸𑀮𑀻𑀲𑀁;_  
_𑀧𑀡𑁆𑀡𑀸𑀲 𑀧𑀺𑀡𑁆𑀟𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑀸,_  
_𑀦𑀸𑀕𑀫𑁆𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑀢𑀺𑁇_
