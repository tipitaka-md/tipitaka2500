# 𑁮𑁇𑁧𑁇𑁩 𑀉𑀧𑁄𑀲𑀣𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀸𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀸𑀕𑀯𑀕𑁆𑀕

### 𑀉𑀧𑁄𑀲𑀣𑀲𑀼𑀢𑁆𑀢

(344.)

1155\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀅𑀡𑁆𑀟𑀚𑀸 𑀦𑀸𑀕𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀦𑁆𑀢𑀺 𑀯𑁄𑀲𑁆𑀲𑀝𑁆𑀞𑀓𑀸𑀬𑀸 𑀘 𑀪𑀯𑀦𑁆𑀢𑀻”𑀢𑀺?

1156\. “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀡𑁆𑀟𑀚𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀬𑀁 𑀔𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀸𑀬𑁂𑀦 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀺𑀦𑁄 𑀅𑀳𑀼𑀫𑁆𑀳, 𑀯𑀸𑀘𑀸𑀬 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀺𑀦𑁄, 𑀫𑀦𑀲𑀸 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀺𑀦𑁄𑁇 𑀢𑁂 𑀫𑀬𑀁 𑀓𑀸𑀬𑁂𑀦 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀺𑀦𑁄, 𑀯𑀸𑀘𑀸𑀬 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀺𑀦𑁄, 𑀫𑀦𑀲𑀸 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀺𑀦𑁄, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀡𑁆𑀟𑀚𑀸𑀦𑀁 𑀦𑀸𑀕𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑁇 𑀲𑀘𑀚𑁆𑀚 𑀫𑀬𑀁 𑀓𑀸𑀬𑁂𑀦 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀯𑀸𑀘𑀸𑀬 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀫𑀦𑀲𑀸 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀫𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁 𑀏𑀢𑀭𑀳𑀺 𑀓𑀸𑀬𑁂𑀦 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀸𑀫, 𑀯𑀸𑀘𑀸𑀬 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀸𑀫, 𑀫𑀦𑀲𑀸 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀅𑀡𑁆𑀟𑀚𑀸 𑀦𑀸𑀕𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀦𑁆𑀢𑀺 𑀯𑁄𑀲𑁆𑀲𑀝𑁆𑀞𑀓𑀸𑀬𑀸 𑀘 𑀪𑀯𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
