### 𑀢𑀢𑀺𑀬𑁅𑀧𑁄𑀲𑀣𑀲𑀼𑀢𑁆𑀢

(346.)

1160\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀁𑀲𑁂𑀤𑀚𑀸 𑀦𑀸𑀕𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀦𑁆𑀢𑀺 𑀯𑁄𑀲𑁆𑀲𑀝𑁆𑀞𑀓𑀸𑀬𑀸 𑀘 𑀪𑀯𑀦𑁆𑀢𑀻”𑀢𑀺? (𑀲𑀩𑁆𑀩𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇) “𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀁𑀲𑁂𑀤𑀚𑀸 𑀦𑀸𑀕𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀦𑁆𑀢𑀺 𑀯𑁄𑀲𑁆𑀲𑀝𑁆𑀞𑀓𑀸𑀬𑀸 𑀘 𑀪𑀯𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
