# 𑁪𑁇𑁧𑁇𑁭 𑀭𑀽𑀧𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀭𑀽𑀧𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(308.)

1055\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀭𑀽𑀧𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀻 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀻; 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸…  𑀕𑀦𑁆𑀥𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸…  𑀭𑀲𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸…  𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀻 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀻𑁇 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑁂 𑀏𑀯𑀁 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀲𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀭𑀻…𑀧𑁂…  𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑁄’”𑀢𑀺𑁇
