# 𑁪𑁇𑁧𑁇𑁧 𑀘𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀘𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀘𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(302.)

1041. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀓𑁆𑀔𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀺; 𑀲𑁄𑀢𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀺; 𑀖𑀸𑀦𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀺; 𑀚𑀺𑀯𑁆𑀳𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀻 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀻; 𑀓𑀸𑀬𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀺; 𑀫𑀦𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀻 𑀅𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀻𑁇 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑁂 𑀏𑀯𑀁 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀭𑀻, 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀸𑀫𑀁, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀪𑀽𑀫𑀺𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑁄, 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀪𑀽𑀫𑀺𑀁; 𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀬𑀁 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀯𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁 𑀯𑀸 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬; 𑀅𑀪𑀩𑁆𑀩𑁄 𑀘 𑀢𑀸𑀯 𑀓𑀸𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀬𑀸𑀯 𑀦 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇

1042. 𑀬𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀯𑀁 𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑀢𑁆𑀢𑀲𑁄 𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀁 𑀔𑀫𑀦𑁆𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  ‘𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀭𑀻, 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀸𑀫𑀁, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀪𑀽𑀫𑀺𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑁄, 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀪𑀽𑀫𑀺𑀁; 𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀬𑀁 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀯𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁 𑀯𑀸 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬; 𑀅𑀪𑀩𑁆𑀩𑁄 𑀘 𑀢𑀸𑀯 𑀓𑀸𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀬𑀸𑀯 𑀦 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺’𑁇 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑁂 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺—  ‘𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑁄’”𑀢𑀺𑁇
