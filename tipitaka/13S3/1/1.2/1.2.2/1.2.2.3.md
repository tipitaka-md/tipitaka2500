# 𑁧𑁇𑁨𑁇𑁨𑁇𑁩 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀔𑀦𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀫𑀚𑁆𑀛𑀺𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(65.)

348. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦…𑀧𑁂…  𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀩𑀤𑁆𑀥𑁄 𑀫𑀸𑀭𑀲𑁆𑀲; 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄”𑀢𑀺𑁇 “𑀅𑀜𑁆𑀜𑀸𑀢𑀁, 𑀪𑀕𑀯𑀸, 𑀅𑀜𑁆𑀜𑀸𑀢𑀁, 𑀲𑀼𑀕𑀢𑀸”𑀢𑀺𑁇

349. “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀻”𑀢𑀺? “𑀭𑀽𑀧𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀩𑀤𑁆𑀥𑁄 𑀫𑀸𑀭𑀲𑁆𑀲; 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇 𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀩𑀤𑁆𑀥𑁄 𑀫𑀸𑀭𑀲𑁆𑀲; 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇

350. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀸𑀲𑀺𑁇 𑀭𑀽𑀧𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀩𑀤𑁆𑀥𑁄 𑀫𑀸𑀭𑀲𑁆𑀲; 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇 𑀯𑁂𑀤𑀦𑀁…  𑀲𑀜𑁆𑀜𑀁…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀩𑀤𑁆𑀥𑁄 𑀫𑀸𑀭𑀲𑁆𑀲; 𑀅𑀦𑀪𑀺𑀦𑀦𑁆𑀤𑀫𑀸𑀦𑁄 𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑀸𑀧𑀺𑀫𑀢𑁄𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀬𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄”𑀢𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇
