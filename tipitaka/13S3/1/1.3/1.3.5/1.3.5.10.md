# 𑁧𑁇𑁩𑁇𑁫𑁇𑁧𑁦 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀔𑀦𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀉𑀧𑀭𑀺𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀕𑁆𑀕

* 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(159.)

811\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼, 𑀬𑀫𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

812\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀭𑀽𑀧𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀯𑁂𑀤𑀦𑀸…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀓𑀮𑁆𑀮𑀁 𑀦𑀼 𑀢𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁…𑀧𑁂…  𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇

814\. 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇

815\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



816\. _𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀁 𑀏𑀢𑀁𑀫𑀫,_  
_𑀲𑁄𑀅𑀢𑁆𑀢𑀸 𑀦𑁄𑀘𑀫𑁂𑀲𑀺𑀬𑀸;_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀓𑁆𑀓𑀸𑀬𑀢𑁆𑀢𑀸𑀦𑀼 𑀤𑁆𑀯𑁂,_  
_𑀅𑀪𑀺𑀦𑀺𑀯𑁂𑀲𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦𑀸𑀢𑀺𑁇_  


---

817\. 𑀉𑀧𑀭𑀺𑀧𑀡𑁆𑀡𑀸𑀲𑀓𑁄 𑀲𑀫𑀢𑁆𑀢𑁄𑁇



---

818\. 𑀢𑀲𑁆𑀲 𑀉𑀧𑀭𑀺𑀧𑀡𑁆𑀡𑀸𑀲𑀓𑀲𑁆𑀲 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



819\. _𑀅𑀦𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸 𑀯𑀺𑀚𑁆𑀚𑀸,_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀍𑀅𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀜𑁆𑀘𑀫𑀁;_  
_𑀢𑀢𑀺𑀬𑁄 𑀧𑀡𑁆𑀡𑀸𑀲𑀓𑁄 𑀯𑀼𑀢𑁆𑀢𑁄,_  
_𑀦𑀺𑀧𑀸𑀢𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_  


---

820\. 𑀔𑀦𑁆𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
