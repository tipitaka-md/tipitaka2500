# 𑁧𑁇𑁩𑁇𑁪𑁇𑁪 𑀢𑀢𑀺𑀬𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢

(139.)

767\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁? 𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀺𑀘𑁆𑀘𑀁; 𑀢𑀢𑁆𑀭 𑀯𑁄 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
