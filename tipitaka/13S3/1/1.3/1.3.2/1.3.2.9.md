### 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(121.)

698\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣𑁇 𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀓𑀢𑀫𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀁? 𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑁄 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄; 𑀬𑁄 𑀢𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑁄, 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀉𑀧𑀸𑀤𑀸𑀦𑀁𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀇𑀤𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇
