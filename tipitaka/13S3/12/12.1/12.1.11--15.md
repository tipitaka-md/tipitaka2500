# 𑁧𑁨𑁇𑁧𑁇𑁧𑁧--𑁧𑁫 𑀭𑀽𑀧𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀧𑀜𑁆𑀘𑀓

* 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀭𑀽𑀧𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀧𑀜𑁆𑀘𑀓

(617--621.)

1248\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸…𑀧𑁂…  𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸…𑀧𑁂… 𑁇

1249\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀯𑁂𑀤𑀦𑀸𑀬 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸…𑀧𑁂… 𑁇

1250\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀲𑀜𑁆𑀜𑀸𑀬 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸…𑀧𑁂… 𑁇

1251\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸…𑀧𑁂… 𑁇

1252\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸…𑀧𑁂… 𑁇

---
