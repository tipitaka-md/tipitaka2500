* 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

1246\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸”𑀢𑀺? “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀤𑀲𑁆𑀲𑀦𑀸…𑀧𑁂…  𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀤𑀲𑁆𑀲𑀦𑀸…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀸𑀬…  𑀲𑀜𑁆𑀜𑀸𑀬…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀤𑀲𑁆𑀲𑀦𑀸…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀤𑀲𑁆𑀲𑀦𑀸…𑀧𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀤𑀲𑁆𑀲𑀦𑀸…𑀧𑁂… 𑁇
