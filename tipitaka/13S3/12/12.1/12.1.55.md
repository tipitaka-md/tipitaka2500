# 𑁧𑁨𑁇𑁧𑁇𑁫𑁫 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(661.)

1274\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸; 𑀏𑀯𑀫𑀺𑀫𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸, 𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸”𑀢𑀺𑁇

1277\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1278\. _𑀅𑀜𑁆𑀜𑀸𑀡𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑀸 𑀘𑁂𑀯,_  
_𑀅𑀦𑀪𑀺𑀲𑀫𑀬𑀸 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸;_  
_𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀅𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑀡𑀸,_  
_𑀅𑀦𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑁂𑀦 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀼𑀧𑀮𑀓𑁆𑀔𑀡𑀸;_  
_𑀅𑀲𑀫𑀧𑁂𑀓𑁆𑀔𑀡𑀸 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀼𑀧𑁂𑀓𑁆𑀔𑀡𑀸,_  
_𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇_  


---
