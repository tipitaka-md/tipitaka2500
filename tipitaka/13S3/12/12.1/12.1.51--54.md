# 𑁧𑁨𑁇𑁧𑁇𑁫𑁧--𑁫𑁪 𑀭𑀽𑀧𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀭𑀽𑀧𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸𑀤𑀺𑀲𑀼𑀢𑁆𑀢𑀘𑀢𑀼𑀓𑁆𑀓

(657--660.)

1269\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  𑀲𑀲𑁆𑀲𑀢𑁄 𑀮𑁄𑀓𑁄𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺 𑀯𑀸”𑀢𑀺? “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸, 𑀭𑀽𑀧𑀲𑀫𑀼𑀤𑀬𑁂 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸, 𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑁂 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸, 𑀭𑀽𑀧𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸…𑀧𑁂… 𑁇

1270\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀯𑁂𑀤𑀦𑀸𑀬 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀯𑁂𑀤𑀦𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸…𑀧𑁂… 𑁇

1271\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀲𑀜𑁆𑀜𑀸𑀬 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸…𑀧𑁂… 𑁇

1272\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀔𑁄, 𑀯𑀘𑁆𑀙, 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀓𑀫𑁆𑀫𑀸…𑀧𑁂… 𑁇
