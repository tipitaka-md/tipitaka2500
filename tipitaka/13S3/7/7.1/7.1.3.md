# 𑁭𑁇𑁧𑁇𑁩 𑀧𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(334.)

1124\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄…𑀧𑁂…  “𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺; 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀓𑀢𑀫𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀲𑀻”𑀢𑀺?

1125\. “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀫𑀺; 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺 𑀯𑀸 ‘𑀅𑀳𑀁 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄’𑀢𑀺 𑀯𑀸 ‘𑀅𑀳𑀁 𑀢𑀢𑀺𑀬𑀸 𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄’𑀢𑀺 𑀯𑀸”𑀢𑀺𑁇 𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀗𑁆𑀓𑀸𑀭𑀫𑀫𑀗𑁆𑀓𑀸𑀭𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀸 𑀲𑀼𑀲𑀫𑀽𑀳𑀢𑀸𑁇 𑀢𑀲𑁆𑀫𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀦 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “‘𑀅𑀳𑀁 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺 𑀯𑀸 ‘𑀅𑀳𑀁 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄’𑀢𑀺 𑀯𑀸 ‘𑀅𑀳𑀁 𑀢𑀢𑀺𑀬𑀸 𑀛𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄’𑀢𑀺 𑀯𑀸”𑀢𑀺𑁇
