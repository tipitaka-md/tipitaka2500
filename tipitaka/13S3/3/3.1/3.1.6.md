# 𑁩𑁇𑁧𑁇𑁬 𑀓𑀭𑁄𑀢𑁄𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀓𑀭𑁄𑀢𑁄𑀲𑀼𑀢𑁆𑀢

(211.)

933\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀓𑀺𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀭𑀬𑀢𑁄 𑀙𑀺𑀦𑁆𑀤𑀢𑁄 𑀙𑁂𑀤𑀸𑀧𑀬𑀢𑁄 𑀧𑀘𑀢𑁄 𑀧𑀸𑀘𑀸𑀧𑀬𑀢𑁄 𑀲𑁄𑀘𑀢𑁄 𑀲𑁄𑀘𑀸𑀧𑀬𑀢𑁄 𑀓𑀺𑀮𑀫𑀢𑁄 𑀓𑀺𑀮𑀫𑀸𑀧𑀬𑀢𑁄 𑀨𑀦𑁆𑀤𑀢𑁄 𑀨𑀦𑁆𑀤𑀸𑀧𑀬𑀢𑁄 𑀧𑀸𑀡𑀫𑀢𑀺𑀧𑀸𑀢𑀬𑀢𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑁄 𑀲𑀦𑁆𑀥𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑁄 𑀦𑀺𑀮𑁆𑀮𑁄𑀧𑀁 𑀳𑀭𑀢𑁄 𑀏𑀓𑀸𑀕𑀸𑀭𑀺𑀓𑀁 𑀓𑀭𑁄𑀢𑁄 𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀧𑀭𑀤𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑁄 𑀫𑀼𑀲𑀸 𑀪𑀡𑀢𑁄 𑀓𑀭𑁄𑀢𑁄 𑀦 𑀓𑀭𑀻𑀬𑀢𑀺 𑀧𑀸𑀧𑀁𑁇 𑀔𑀼𑀭𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑁂𑀦 𑀘𑁂𑀧𑀺 𑀘𑀓𑁆𑀓𑁂𑀦 𑀬𑁄 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀸𑀡𑁂 𑀏𑀓𑀫𑀁𑀲𑀔𑀮𑀁 𑀏𑀓𑀫𑀁𑀲𑀧𑀼𑀜𑁆𑀚𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀸𑀧𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀆𑀕𑀫𑁄𑁇 𑀤𑀓𑁆𑀔𑀺𑀡𑀜𑁆𑀘𑁂𑀧𑀺 𑀕𑀗𑁆𑀕𑀸𑀬 𑀢𑀻𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬; 𑀳𑀦𑀦𑁆𑀢𑁄 𑀖𑀸𑀢𑁂𑀦𑁆𑀢𑁄 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀙𑁂𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀧𑀘𑀦𑁆𑀢𑁄 𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀸𑀧𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀆𑀕𑀫𑁄𑁇 𑀉𑀢𑁆𑀢𑀭𑀜𑁆𑀘𑁂𑀧𑀺 𑀕𑀗𑁆𑀕𑀸𑀬 𑀢𑀻𑀭𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬; 𑀤𑀤𑀦𑁆𑀢𑁄 𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀬𑀚𑀦𑁆𑀢𑁄 𑀬𑀚𑀸𑀧𑁂𑀦𑁆𑀢𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀼𑀜𑁆𑀜𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄𑁇 𑀤𑀸𑀦𑁂𑀦 𑀤𑀫𑁂𑀦 𑀲𑀁𑀬𑀫𑁂𑀦 𑀲𑀘𑁆𑀘𑀯𑀚𑁆𑀚𑁂𑀦 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄’”𑀢𑀺𑁇 “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…

934\. “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀭𑀽𑀧𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀭𑀬𑀢𑁄…𑀧𑁂…  𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄’𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀭𑀬𑀢𑁄…𑀧𑁂…  𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄’𑀢𑀺𑁇

935\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀓𑀭𑁄𑀢𑁄…𑀧𑁂…  𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀯𑁂𑀤𑀦𑀸…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀭𑀬𑀢𑁄…𑀧𑁂…  𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀭𑀬𑀢𑁄…𑀧𑁂…  𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀆𑀕𑀫𑁄’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

936\. “𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀞𑀸𑀦𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑁂𑀧𑀺𑀲𑁆𑀲 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬𑀧𑀺𑀲𑁆𑀲 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑁄”𑀢𑀺𑁇
