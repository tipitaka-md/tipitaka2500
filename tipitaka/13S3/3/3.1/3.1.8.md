# 𑁩𑁇𑁧𑁇𑁮 𑀫𑀳𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀼𑀢𑁆𑀢

(213.)

944. “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀭𑀽𑀧𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀲𑀢𑁆𑀢𑀺𑀫𑁂 𑀓𑀸𑀬𑀸 𑀅𑀓𑀝𑀸, 𑀅𑀓𑀝𑀯𑀺𑀥𑀸…𑀧𑁂…  𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀮𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀲𑀢𑁆𑀢𑀺𑀫𑁂 𑀓𑀸𑀬𑀸 𑀅𑀓𑀝𑀸, 𑀅𑀓𑀝𑀯𑀺𑀥𑀸…𑀧𑁂…  𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀮𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇

945. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀢𑀺𑀫𑁂 𑀓𑀸𑀬𑀸 𑀅𑀓𑀝𑀸 𑀅𑀓𑀝𑀯𑀺𑀥𑀸…𑀧𑁂…  𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀮𑁂𑀦𑁆𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀲𑀢𑁆𑀢𑀺𑀫𑁂 𑀓𑀸𑀬𑀸 𑀅𑀓𑀝𑀸 𑀅𑀓𑀝𑀯𑀺𑀥𑀸…𑀧𑁂…  𑀦𑀺𑀩𑁆𑀩𑁂𑀞𑀺𑀬𑀫𑀸𑀦𑀸 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀮𑁂𑀦𑁆𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

946. “𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀞𑀸𑀦𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑁂𑀧𑀺𑀲𑁆𑀲 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬𑀧𑀺𑀲𑁆𑀲 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑁄”𑀢𑀺𑁇
