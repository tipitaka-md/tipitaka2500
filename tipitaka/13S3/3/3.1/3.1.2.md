# 𑁩𑁇𑁧𑁇𑁨 𑀏𑀢𑀁𑀫𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀏𑀢𑀁𑀫𑀫𑀲𑀼𑀢𑁆𑀢

(207.)

914\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀓𑀺𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀭𑀽𑀧𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’𑀢𑀺𑁇

915\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  “𑀯𑁂𑀤𑀦𑀸…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀫𑁆𑀧𑀺𑀤𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀺𑀢𑀁 𑀫𑀦𑀲𑀸 𑀢𑀫𑁆𑀧𑀺 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸”𑀢𑀺? “𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀁 𑀧𑀦𑀸𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀏𑀢𑀁 𑀫𑀫, 𑀏𑀲𑁄𑀳𑀫𑀲𑁆𑀫𑀺, 𑀏𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

916\. “𑀬𑀢𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀇𑀫𑁂𑀲𑀼 𑀘 𑀞𑀸𑀦𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑁂𑀧𑀺𑀲𑁆𑀲 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬𑀧𑀺𑀲𑁆𑀲 𑀓𑀗𑁆𑀔𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀦𑁄”𑀢𑀺𑁇
