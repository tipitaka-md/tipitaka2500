# 𑁩𑁇𑁨𑁇𑁧 𑀯𑀸𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀕𑀫𑀦𑀯𑀕𑁆𑀕

### 𑀯𑀸𑀢𑀲𑀼𑀢𑁆𑀢

(224.)

980\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀓𑀺𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀓𑀺𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀦 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀦 𑀦𑀚𑁆𑀚𑁄 𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀦 𑀕𑀩𑁆𑀪𑀺𑀦𑀺𑀬𑁄 𑀯𑀺𑀚𑀸𑀬𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑀸 𑀉𑀤𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀅𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀏𑀲𑀺𑀓𑀝𑁆𑀞𑀸𑀬𑀺𑀝𑁆𑀞𑀺𑀢𑀸’”𑀢𑀺? “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…

981\. “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀭𑀽𑀧𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀦 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀏𑀲𑀺𑀓𑀝𑁆𑀞𑀸𑀬𑀺𑀝𑁆𑀞𑀺𑀢𑀸’𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀦 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀏𑀲𑀺𑀓𑀝𑁆𑀞𑀸𑀬𑀺𑀝𑁆𑀞𑀺𑀢𑀸’𑀢𑀺𑁇

982\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  𑀦 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀏𑀲𑀺𑀓𑀝𑁆𑀞𑀸𑀬𑀺𑀝𑁆𑀞𑀺𑀢𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀦 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀏𑀲𑀺𑀓𑀝𑁆𑀞𑀸𑀬𑀺𑀝𑁆𑀞𑀺𑀢𑀸’”𑀢𑀺𑁇 “𑀯𑁂𑀤𑀦𑀸…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 ‘𑀦 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀏𑀲𑀺𑀓𑀝𑁆𑀞𑀸𑀬𑀺𑀝𑁆𑀞𑀺𑀢𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀦 𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀦 𑀦𑀚𑁆𑀚𑁄 𑀲𑀦𑁆𑀤𑀦𑁆𑀢𑀺, 𑀦 𑀕𑀩𑁆𑀪𑀺𑀦𑀺𑀬𑁄 𑀯𑀺𑀚𑀸𑀬𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀦𑁆𑀤𑀺𑀫𑀲𑀽𑀭𑀺𑀬𑀸 𑀉𑀤𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀅𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀏𑀲𑀺𑀓𑀝𑁆𑀞𑀸𑀬𑀺𑀝𑁆𑀞𑀺𑀢𑀸’”𑀢𑀺𑁇
