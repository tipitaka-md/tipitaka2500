# 𑁩𑁇𑁨𑁇𑁨𑁬 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀕𑀫𑀦𑀯𑀕𑁆𑀕

### 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻𑀲𑀼𑀢𑁆𑀢

(249.)

1006\. “𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸”𑀢𑀺? “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…

1007\. “𑀭𑀽𑀧𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑀺, 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀭𑀽𑀧𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑁇 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀢𑀺…  𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀢𑀺…  𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂 𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’𑀢𑀺𑁇

1008\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 “𑀯𑁂𑀤𑀦𑀸…  𑀲𑀜𑁆𑀜𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀯𑀸”𑀢𑀺? “𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀪𑀦𑁆𑀢𑁂” …𑀧𑁂…  𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬—  ‘𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀦𑀺𑀯𑀺𑀲𑁆𑀲 𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  ‘𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸’”𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀻𑀲𑀢𑀺𑀫𑀁𑁇

1009\. 𑀤𑀼𑀢𑀺𑀬𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑁄𑁇



---

1010\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1011\. _𑀯𑀸𑀢𑀁 𑀏𑀢𑀁 𑀫𑀫 𑀲𑁄,_  
_𑀅𑀢𑁆𑀢𑀸 𑀦𑁄 𑀘 𑀫𑁂 𑀲𑀺𑀬𑀸;_  
_𑀦𑀢𑁆𑀣𑀺 𑀓𑀭𑁄𑀢𑁄 𑀳𑁂𑀢𑀼 𑀘,_  
_𑀫𑀳𑀸𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀅𑀝𑁆𑀞𑀫𑀁𑁇_  


1012\. _𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀘𑁂𑀯,_  
_𑀅𑀦𑁆𑀢𑀸𑀦𑀦𑁆𑀢𑀯𑀸 𑀘 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀢𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀜𑁆𑀘,_  
_𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑁇_  


1013\. _𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀅𑀭𑀽𑀧𑀻 𑀘 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀭𑀽𑀧𑀻 𑀘 𑀅𑀭𑀽𑀧𑀻 𑀘 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺;_  
_𑀦𑁂𑀯 𑀭𑀽𑀧𑀻 𑀦𑀸𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺𑁇_  


1014\. _𑀏𑀓𑀦𑁆𑀢𑀤𑀼𑀓𑁆𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺;_  
_𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸𑀢𑀺;_  
_𑀇𑀫𑁂 𑀙𑀩𑁆𑀩𑀻𑀲𑀢𑀺 𑀲𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀼𑀢𑀺𑀬𑀯𑀸𑀭𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀸𑁇_
