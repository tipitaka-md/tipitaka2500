# 𑁯𑁇𑁧𑁇𑁧𑁭--𑁪𑁬 𑀚𑀮𑀸𑀩𑀼𑀚𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀼𑀧𑀡𑁆𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀼𑀧𑀡𑁆𑀡𑀯𑀕𑁆𑀕

### 𑀚𑀮𑀸𑀩𑀼𑀚𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(408--437.)

1195\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀚𑀮𑀸𑀩𑀼𑀚𑀸𑀦𑀁 𑀲𑀼𑀧𑀡𑁆𑀡𑀸𑀦𑀁…𑀧𑁂…  𑀲𑀁𑀲𑁂𑀤𑀚𑀸𑀦𑀁 𑀲𑀼𑀧𑀡𑁆𑀡𑀸𑀦𑀁…𑀧𑁂…  𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀼𑀧𑀡𑁆𑀡𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑁂𑀦 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀻, 𑀫𑀦𑀲𑀸 𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀻𑁇 𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺—  ‘𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀩𑀳𑀼𑀮𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀼𑀧𑀡𑁆𑀡𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁 𑀤𑁂𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀼𑀧𑀡𑁆𑀡𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀳𑁂𑀢𑀼, 𑀅𑀬𑀁 𑀧𑀘𑁆𑀘𑀬𑁄, 𑀬𑁂𑀦 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸𑀦𑀁 𑀲𑀼𑀧𑀡𑁆𑀡𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇 𑀙𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑀁𑁇

---

1196\. (𑀏𑀯𑀁 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀙𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀁 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇)

  
𑀲𑀼𑀧𑀡𑁆𑀡𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

1197\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1198\. _𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀳𑀭𑀦𑁆𑀢𑀺 𑀘𑁂𑀯,_  
_𑀤𑁆𑀯𑀬𑀓𑀸𑀭𑀻 𑀘 𑀘𑀢𑀼𑀭𑁄;_  
_𑀤𑀸𑀦𑀽𑀧𑀓𑀸𑀭𑀸 𑀢𑀸𑀮𑀻𑀲𑀁,_  
_𑀲𑀼𑀧𑀡𑁆𑀡𑁂 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑀢𑀺𑁇_
