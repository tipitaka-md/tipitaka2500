# 𑁯𑁇𑁧𑁇𑁨 𑀳𑀭𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀼𑀧𑀡𑁆𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀼𑀧𑀡𑁆𑀡𑀯𑀕𑁆𑀕

### 𑀳𑀭𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(393.)

1187. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀧𑀡𑁆𑀡𑀬𑁄𑀦𑀺𑀬𑁄𑁇 𑀓𑀢𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄? 𑀅𑀡𑁆𑀟𑀚𑀸…𑀧𑁂…  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀲𑀼𑀧𑀡𑁆𑀡𑀬𑁄𑀦𑀺𑀬𑁄𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀡𑁆𑀟𑀚𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀅𑀡𑁆𑀟𑀚𑁂𑀯 𑀦𑀸𑀕𑁂 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀚𑀮𑀸𑀩𑀼𑀚𑁂, 𑀦 𑀲𑀁𑀲𑁂𑀤𑀚𑁂, 𑀦 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁂𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀮𑀸𑀩𑀼𑀚𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀅𑀡𑁆𑀟𑀚𑁂 𑀘 𑀚𑀮𑀸𑀩𑀼𑀚𑁂 𑀘 𑀦𑀸𑀕𑁂 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀁𑀲𑁂𑀤𑀚𑁂, 𑀦 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁂𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀲𑁂𑀤𑀚𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀅𑀡𑁆𑀟𑀚𑁂 𑀘 𑀚𑀮𑀸𑀩𑀼𑀚𑁂 𑀘 𑀲𑀁𑀲𑁂𑀤𑀚𑁂 𑀘 𑀦𑀸𑀕𑁂 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁂𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀅𑀡𑁆𑀟𑀚𑁂 𑀘 𑀚𑀮𑀸𑀩𑀼𑀚𑁂 𑀘 𑀲𑀁𑀲𑁂𑀤𑀚𑁂 𑀘 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑁂 𑀘 𑀦𑀸𑀕𑁂 𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀲𑀼𑀧𑀡𑁆𑀡𑀬𑁄𑀦𑀺𑀬𑁄”𑀢𑀺𑁇
