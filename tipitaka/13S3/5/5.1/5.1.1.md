# 𑁫𑁇𑁧𑁇𑁧 𑀘𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀘𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(312.)

1068\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀲𑁄𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀬𑁄 𑀖𑀸𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺…  𑀬𑁄 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺…  𑀬𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺…  𑀬𑁄 𑀫𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑁇 𑀬𑁄 𑀲𑁄𑀢𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀬𑁄 𑀖𑀸𑀦𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄…  𑀬𑁄 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑁄…  𑀬𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄…  𑀬𑁄 𑀫𑀦𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇
