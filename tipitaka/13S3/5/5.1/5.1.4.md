# 𑁫𑁇𑁧𑁇𑁪 𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

(315.)

1074\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄…𑀧𑁂…  𑀬𑁄 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄…𑀧𑁂…  𑀬𑁄 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇
