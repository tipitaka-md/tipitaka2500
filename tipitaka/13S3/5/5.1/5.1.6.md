# 𑁫𑁇𑁧𑁇𑁬 𑀲𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀲𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

(317.)

1079\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀬 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄…𑀧𑁂…  𑀬𑁄 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸𑀬 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑁄…𑀧𑁂…  𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄…𑀧𑁂…  𑀬𑁄 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇
