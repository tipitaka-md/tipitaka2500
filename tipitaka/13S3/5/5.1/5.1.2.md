# 𑁫𑁇𑁧𑁇𑁨 𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

(313.)

1070. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀲𑀤𑁆𑀤𑀸𑀦𑀁…  𑀬𑁄 𑀕𑀦𑁆𑀥𑀸𑀦𑀁…  𑀬𑁄 𑀭𑀲𑀸𑀦𑀁…  𑀬𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀦𑀁…  𑀬𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀞𑀺𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀉𑀧𑁆𑀧𑀸𑀤𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀞𑀺𑀢𑀺, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄𑁇 𑀬𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀽𑀧𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑁇 𑀬𑁄 𑀲𑀤𑁆𑀤𑀸𑀦𑀁…  𑀬𑁄 𑀕𑀦𑁆𑀥𑀸𑀦𑀁…  𑀬𑁄 𑀭𑀲𑀸𑀦𑀁…  𑀬𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀦𑀁…  𑀬𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀯𑀽𑀧𑀲𑀫𑁄 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀭𑁄𑀕𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁄, 𑀚𑀭𑀸𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇
