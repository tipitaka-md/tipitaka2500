# 𑁧𑁩𑁇𑁧𑁇𑁩𑁫--𑁪𑁦 𑀓𑀮𑁆𑀮𑀺𑀢𑀫𑀽𑀮𑀓𑀆𑀭𑀫𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀙𑀓𑁆𑀓

### 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀮𑁆𑀮𑀺𑀢𑀫𑀽𑀮𑀓𑀆𑀭𑀫𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀙𑀓𑁆𑀓

(696--701.)

1326\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁…  “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄…  𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇
