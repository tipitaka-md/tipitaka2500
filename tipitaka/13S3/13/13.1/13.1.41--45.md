# 𑁧𑁩𑁇𑁧𑁇𑁪𑁧--𑁪𑁫 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑀽𑀮𑀓𑀕𑁄𑀘𑀭𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀧𑀜𑁆𑀘𑀓

### 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑀽𑀮𑀓𑀕𑁄𑀘𑀭𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀧𑀜𑁆𑀘𑀓

(702--706.)

1329\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁…  “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄…  𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇
