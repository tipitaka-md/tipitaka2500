# 𑁧𑁩𑁇𑁧𑁇𑁨𑁦--𑁨𑁭 𑀞𑀺𑀢𑀺𑀫𑀽𑀮𑀓𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀅𑀝𑁆𑀞𑀓

### 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀞𑀺𑀢𑀺𑀫𑀽𑀮𑀓𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀅𑀝𑁆𑀞𑀓

(681--688.)

1320. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀛𑀸𑀬𑀻𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀞𑀺𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀞𑀺𑀢𑀺𑀓𑀼𑀲𑀮𑁄𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀞𑀺𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀞𑀺𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇
