# 𑁧𑁩𑁇𑁧𑁇𑁪𑁬--𑁪𑁯 𑀕𑁄𑀘𑀭𑀫𑀽𑀮𑀓𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀕𑁄𑀘𑀭𑀫𑀽𑀮𑀓𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀘𑀢𑀼𑀓𑁆𑀓

(707.)

1332\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁…  “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄…  𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀘…  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀯𑀸 𑀔𑀻𑀭𑀁, 𑀔𑀻𑀭𑀫𑁆𑀳𑀸 𑀤𑀥𑀺, 𑀤𑀥𑀺𑀫𑁆𑀳𑀸 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀦𑀯𑀦𑀻𑀢𑀫𑁆𑀳𑀸 𑀲𑀧𑁆𑀧𑀺, 𑀲𑀧𑁆𑀧𑀺𑀫𑁆𑀳𑀸 𑀲𑀧𑁆𑀧𑀺𑀫𑀡𑁆𑀟𑁄 𑀢𑀢𑁆𑀭 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀬𑀻𑀦𑀁…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇 𑀙𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑀁𑁇

(708.)

1333\. “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻…𑀧𑁂… 𑁇 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀢𑁆𑀢𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑀁𑁇

(709.)

1334\. “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻…𑀧𑁂… 𑁇 𑀅𑀝𑁆𑀞𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑀁𑁇

(710.)

1335\. “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻…𑀧𑁂… 𑁇
