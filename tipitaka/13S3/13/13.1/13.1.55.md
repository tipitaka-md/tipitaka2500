# 𑁧𑁩𑁇𑁧𑁇𑁫𑁫 𑀲𑀸𑀢𑀘𑁆𑀘𑀫𑀽𑀮𑀓𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀢𑀘𑁆𑀘𑀫𑀽𑀮𑀓𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻𑀲𑀼𑀢𑁆𑀢

(716.)

1349. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀛𑀸𑀬𑀻𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻 𑀘 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀬𑀻𑀦𑀁 𑀅𑀕𑁆𑀕𑁄 𑀘 𑀲𑁂𑀝𑁆𑀞𑁄 𑀘 𑀫𑁄𑀓𑁆𑀔𑁄 𑀘 𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀯𑀸 𑀔𑀻𑀭𑀁, 𑀔𑀻𑀭𑀫𑁆𑀳𑀸 𑀤𑀥𑀺, 𑀤𑀥𑀺𑀫𑁆𑀳𑀸 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀦𑀯𑀦𑀻𑀢𑀫𑁆𑀳𑀸 𑀲𑀧𑁆𑀧𑀺, 𑀲𑀧𑁆𑀧𑀺𑀫𑁆𑀳𑀸 𑀲𑀧𑁆𑀧𑀺𑀫𑀡𑁆𑀟𑁄 𑀢𑀢𑁆𑀭 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻 𑀘 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀬𑀻𑀦𑀁 𑀅𑀕𑁆𑀕𑁄 𑀘 𑀲𑁂𑀝𑁆𑀞𑁄 𑀘 𑀫𑁄𑀓𑁆𑀔𑁄 𑀘 𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

1350. 𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀁𑁇



---

1351. (𑀬𑀣𑀸 𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇)



---

1352. 𑀔𑀦𑁆𑀥𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

1353. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1354. _𑀲𑀫𑀸𑀥𑀺 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺 𑀞𑀺𑀢𑀺 𑀘,_  
_𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁂𑀦 𑀘;_  
_𑀕𑁄𑀘𑀭𑀸 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑁄 𑀲𑀓𑁆𑀓𑀘𑁆𑀘,_  
_𑀲𑀸𑀢𑀘𑁆𑀘 𑀅𑀣𑁄𑀧𑀺 𑀲𑀧𑁆𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑁇_  


---

1355. 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇



---

1356. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1357. _𑀔𑀦𑁆𑀥 𑀭𑀸𑀥𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀤𑀺𑀝𑁆𑀞𑀺𑀑𑀓𑁆𑀓𑀦𑁆𑀢 𑀉𑀧𑁆𑀧𑀸𑀤𑀸;_  
_𑀓𑀺𑀮𑁂𑀲 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀦𑀸𑀕𑀸 𑀲𑀼𑀧𑀡𑁆𑀡 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸;_  
_𑀯𑀮𑀸𑀳 𑀯𑀘𑁆𑀙𑀛𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀔𑀦𑁆𑀥𑀯𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀢𑁂𑀭𑀲𑀸𑀢𑀺𑁇_  


---

1358. 𑀔𑀦𑁆𑀥𑀯𑀕𑁆𑀕𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
