# 𑁧𑁩𑁇𑁧𑁇𑁪 𑀲𑀫𑀸𑀥𑀺𑀫𑀽𑀮𑀓𑀓𑀮𑁆𑀮𑀺𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀫𑀸𑀥𑀺𑀫𑀽𑀮𑀓𑀓𑀮𑁆𑀮𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(665.)

1286. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀛𑀸𑀬𑀻𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑁄𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀬𑀁 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀬𑀻𑀦𑀁 𑀅𑀕𑁆𑀕𑁄 𑀘 𑀲𑁂𑀝𑁆𑀞𑁄 𑀘 𑀫𑁄𑀓𑁆𑀔𑁄 𑀘 𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀯𑀸 𑀔𑀻𑀭𑀁…𑀧𑁂…  𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇
