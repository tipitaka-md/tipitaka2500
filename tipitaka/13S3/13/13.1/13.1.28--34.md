# 𑁧𑁩𑁇𑁧𑁇𑁨𑁮--𑁩𑁪 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀫𑀽𑀮𑀓𑀓𑀮𑁆𑀮𑀺𑀢𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀲𑀢𑁆𑀢𑀓

### 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀫𑀽𑀮𑀓𑀓𑀮𑁆𑀮𑀺𑀢𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀲𑀢𑁆𑀢𑀓

(689--695.)

1323\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀛𑀸𑀬𑀻𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄…  𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀮𑁆𑀮𑀺𑀢𑀓𑀼𑀲𑀮𑁄 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇
