# 𑁧𑁩𑁇𑁧𑁇𑁫𑁦--𑁫𑁨 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀫𑀽𑀮𑀓𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀢𑀺𑀓

* 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀫𑀽𑀮𑀓𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀢𑀺𑀓

(711.)

1338\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁…  “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄…  𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀛𑀸𑀬𑀻…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇

---

1339\. 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀁𑁇



(712.)

1340\. “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻…𑀧𑁂… 𑁇

---

1341\. 𑀏𑀓𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀁𑁇



(713.)

1342\. “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻…𑀧𑁂… 𑁇
