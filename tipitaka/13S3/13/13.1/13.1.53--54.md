# 𑁧𑁩𑁇𑁧𑁇𑁫𑁩--𑁫𑁪 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀫𑀽𑀮𑀓𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻𑀲𑀼𑀢𑁆𑀢𑀤𑀼𑀓𑀸𑀤𑀺

* 𑀛𑀸𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀛𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀫𑀽𑀮𑀓𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻𑀲𑀼𑀢𑁆𑀢𑀤𑀼𑀓𑀸𑀤𑀺

(714.)

1345\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁…  “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻…  𑀦𑁂𑀯 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻…  𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀸𑀢𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀘𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁…𑀧𑁂…  𑀉𑀢𑁆𑀢𑀫𑁄 𑀘 𑀧𑀯𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇

1346\. 𑀢𑁂𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀁𑁇



(715.)

1347\. “𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀸𑀭𑀻…𑀧𑁂… 𑁇
