# 𑁩𑁇𑁩 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

* 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

(251.)

1918\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1919\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1920\. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1921\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
