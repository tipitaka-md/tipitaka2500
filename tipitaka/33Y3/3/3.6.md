# 𑁩𑁇𑁬 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

### 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

(254.)

1934. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1935. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑁄𑁇

1936. »  𑀬𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1937. 𑀬𑁂 𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀬𑁂 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

1938. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

1939. 𑀅𑀭𑀳𑀸 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀬𑁂 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑁇

---
