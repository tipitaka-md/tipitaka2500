# 𑁧𑁇𑁨𑁇𑁧𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(11.)

216\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

217\. «  𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀘𑀓𑁆𑀔𑀽𑀢𑀺?

218\. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀘𑀓𑁆𑀔𑀼𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇

219\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

220\. «  𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀲𑁄𑀢𑀦𑁆𑀢𑀺?

221\. 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑁄𑀢𑀁 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀁𑁇 𑀲𑁄𑀢𑀜𑁆𑀘 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀲𑁄𑀢𑀁 𑀦 𑀘 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁𑁇

222\. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

223\. «  𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀖𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

224\. »  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

225\. «  𑀦 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

226\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

227\. «  𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀦 𑀓𑀸𑀬𑁄𑀢𑀺?

228\. 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑁄𑁇 𑀓𑀸𑀬𑀜𑁆𑀘 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀓𑀸𑀬𑁄 𑀦 𑀘 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁𑁇

229\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

230\. «  𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀭𑀽𑀧𑀦𑁆𑀢𑀺?

231\. 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀁𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀭𑀽𑀧𑀁 𑀦 𑀘 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁𑁇

232\. »  𑀦 𑀲𑀤𑁆𑀤𑁄 𑀦 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

233\. «  𑀦 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀦 𑀲𑀤𑁆𑀤𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

234\. »  𑀦 𑀕𑀦𑁆𑀥𑁄 𑀦 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

235\. «  𑀦 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀦 𑀕𑀦𑁆𑀥𑁄𑀢𑀺?

236\. 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁄 𑀲𑀫𑀸𑀥𑀺𑀕𑀦𑁆𑀥𑁄 𑀧𑀜𑁆𑀜𑀸𑀕𑀦𑁆𑀥𑁄 𑀦 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑁄𑁇 𑀕𑀦𑁆𑀥𑀜𑁆𑀘 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀕𑀦𑁆𑀥𑁄 𑀦 𑀘 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁𑁇

237\. »  𑀦 𑀭𑀲𑁄 𑀦 𑀭𑀲𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

238\. «  𑀦 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀦 𑀭𑀲𑁄𑀢𑀺?

239\. 𑀅𑀢𑁆𑀣𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀭𑀲𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑁄 𑀦 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑁄𑁇 𑀭𑀲𑀜𑁆𑀘 𑀭𑀲𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀭𑀲𑁄 𑀦 𑀘 𑀭𑀲𑀸𑀬𑀢𑀦𑀁𑁇

240\. »  𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄 𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

241\. «  𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

242\. »  𑀦 𑀫𑀦𑁄 𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

243\. «  𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀫𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

244\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

245\. «  𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀥𑀫𑁆𑀫𑁄𑀢𑀺?

246\. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁, 𑀥𑀫𑁆𑀫𑁄𑁇 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇
