# 𑁧𑁇𑁨𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(10.)

185\. »  𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

186\. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼, 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

187\. «  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀽𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

188\. »  𑀲𑁄𑀢𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

189\. 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑁄𑀢𑀁 𑀲𑁄𑀢𑀁, 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁𑁇 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀲𑁄𑀢𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

190\. «  𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀲𑁄𑀢𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

191\. »  𑀖𑀸𑀦𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

192\. «  𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀖𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

193\. »  𑀚𑀺𑀯𑁆𑀳𑀸 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

194\. «  𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

195\. »  𑀓𑀸𑀬𑁄 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

196\. 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀓𑀸𑀬𑁄, 𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁𑁇 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑁄 𑀘𑁂𑀯 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

197\. «  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

198\. »  𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

199\. 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁𑁇 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀭𑀽𑀧𑀜𑁆𑀘𑁂𑀯 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

200\. «  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀭𑀽𑀧𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

201\. »  𑀲𑀤𑁆𑀤𑁄 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

202\. «  𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁 𑀲𑀤𑁆𑀤𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

203\. »  𑀕𑀦𑁆𑀥𑁄 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

204\. 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁄 𑀲𑀫𑀸𑀥𑀺𑀕𑀦𑁆𑀥𑁄 𑀧𑀜𑁆𑀜𑀸𑀕𑀦𑁆𑀥𑁄 𑀕𑀦𑁆𑀥𑁄, 𑀦 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁𑁇 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

205\. «  𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁 𑀕𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

206\. »  𑀭𑀲𑁄 𑀭𑀲𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

207\. 𑀅𑀢𑁆𑀣𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀭𑀲𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑁄 𑀭𑀲𑁄, 𑀦 𑀭𑀲𑀸𑀬𑀢𑀦𑀁𑁇 𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑁄 𑀘𑁂𑀯 𑀭𑀲𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

208\. «  𑀭𑀲𑀸𑀬𑀢𑀦𑀁 𑀭𑀲𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

209\. »  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

210\. «  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

211\. »  𑀫𑀦𑁄 𑀫𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

212\. «  𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀫𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

213\. »  𑀥𑀫𑁆𑀫𑁄 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

214\. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀥𑀫𑁆𑀫𑁄 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

215\. «  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
