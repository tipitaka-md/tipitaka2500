# 𑁧𑁇𑁨𑁇𑁨𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(13.)

265\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

266\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

267\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

268\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 …𑀧𑁂… 𑁇

269\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁧)

270\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

271\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

272\. »  𑀦 𑀖𑀸𑀦𑀁 𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…

273\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

274\. 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 …𑀧𑁂… 𑁇 (𑁨--𑁧𑁧)

275\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

276\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

277\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

278\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁧𑁨)

279\. (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑁂𑀦 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀆𑀫𑀦𑁆𑀢𑀸𑀢𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)
