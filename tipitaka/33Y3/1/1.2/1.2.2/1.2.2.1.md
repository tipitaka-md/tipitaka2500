# 𑁧𑁇𑁨𑁇𑁨𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(12.)

247\. »  𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

248\. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼, 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

249\. «  𑀆𑀬𑀢𑀦𑀸 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

250\. 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁𑁇

251\. »  𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

252\. 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀸𑀘𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼, 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

253\. «  𑀆𑀬𑀢𑀦𑀸 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀢𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

254\. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇

255\. »  𑀲𑁄𑀢𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁…𑀧𑁂… 𑁇

256\. »  𑀥𑀫𑁆𑀫𑁄 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

257\. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀥𑀫𑁆𑀫𑁄 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

258\. «  𑀆𑀬𑀢𑀦𑀸 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

259\. 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇

260\. »  𑀥𑀫𑁆𑀫𑁄 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

261\. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀥𑀫𑁆𑀫𑁄 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇

262\. «  𑀆𑀬𑀢𑀦𑀸 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀬𑀢𑀦𑀸 𑀫𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

263\. 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇

264\. (𑀏𑀓𑁂𑀓𑀧𑀤𑀫𑀽𑀮𑀓𑀁 𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀲𑀫𑁆𑀫𑁄𑀳𑀦𑁆𑀢𑁂𑀦𑁇)
