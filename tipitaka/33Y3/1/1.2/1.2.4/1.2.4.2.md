# 𑁧𑁇𑁨𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(17.)

318\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

319\. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀆𑀬𑀢𑀦𑀸𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀆𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀆𑀬𑀢𑀦𑀸𑁇

320\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

321\. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

322\. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀆𑀬𑀢𑀦𑀸𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀆𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀆𑀬𑀢𑀦𑀸𑁇

323\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁧)

324\. »  𑀦 𑀲𑁄𑀢𑀁 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

325\. 𑀲𑁄𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀦 𑀘 𑀆𑀬𑀢𑀦𑀸𑁇 (𑁨--𑁩)

326\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁪)

327\. »  𑀦 𑀓𑀸𑀬𑁄 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

328\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 …𑀧𑁂… 𑁇 𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 …𑀧𑁂… 𑁇 (𑁫--𑁧𑁧)

329\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

330\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

331\. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

332\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 …𑀧𑁂… 𑁇

333\. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑁧𑁨)

---

334\. (𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇)
