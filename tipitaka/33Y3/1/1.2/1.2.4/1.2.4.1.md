# 𑁧𑁇𑁨𑁇𑁪𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(16.)

304. »  𑀘𑀓𑁆𑀔𑀼 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

305. «  𑀆𑀬𑀢𑀦𑀸 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

306. 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁𑁇

307. »  𑀘𑀓𑁆𑀔𑀼 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

308. «  𑀆𑀬𑀢𑀦𑀸 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀆𑀬𑀢𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

309. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇 (𑁧)

310. »  𑀲𑁄𑀢𑀁 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

311. «  𑀆𑀬𑀢𑀦𑀸 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂… 𑁇 𑀆𑀬𑀢𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇 (𑁨)

312. »  𑀖𑀸𑀦𑀁 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

313. «  𑀆𑀬𑀢𑀦𑀸 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀆𑀬𑀢𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁…𑀧𑁂… 𑁇 (𑁩--𑁧𑁧)

314. »  𑀥𑀫𑁆𑀫𑁄 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

315. «  𑀆𑀬𑀢𑀦𑀸 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀆𑀬𑀢𑀦𑀸 𑀫𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

316. 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁𑁇 (𑁧𑁨)

---
