# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀸𑀬𑀢𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(15.)

288. »  𑀦 𑀘𑀓𑁆𑀔𑀼 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

289. 𑀘𑀓𑁆𑀔𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀼, 𑀆𑀬𑀢𑀦𑀸𑁇 𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀆𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀘 𑀆𑀬𑀢𑀦𑀸𑁇

290. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

291. »  𑀦 𑀲𑁄𑀢𑀁 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

292. 𑀲𑁄𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀦 𑀘 𑀆𑀬𑀢𑀦𑀸𑁇

293. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

294. »  𑀦 𑀓𑀸𑀬𑁄 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

295. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

296. »  𑀦 𑀭𑀽𑀧𑀁 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

297. 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀕𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀭𑀲𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀦 𑀘 𑀆𑀬𑀢𑀦𑀸𑁇

298. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

299. »  𑀦 𑀫𑀦𑁄 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

300. 𑀫𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀫𑀦𑁄, 𑀆𑀬𑀢𑀦𑀸𑁇 𑀫𑀦𑀜𑁆𑀘 𑀆𑀬𑀢𑀦𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀦𑁄 𑀦 𑀘 𑀆𑀬𑀢𑀦𑀸𑁇

301. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

302. »  𑀦 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

303. «  𑀦𑀸𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
