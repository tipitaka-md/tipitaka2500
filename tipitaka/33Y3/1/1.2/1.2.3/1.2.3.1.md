# 𑁧𑁇𑁨𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀸𑀬𑀢𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(14.)

280\. »  𑀘𑀓𑁆𑀔𑀼 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

281\. «  𑀆𑀬𑀢𑀦𑀸 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

282\. 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁𑁇

283\. »  𑀲𑁄𑀢𑀁 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

284\. 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂…  𑀖𑀸𑀦𑀁…  𑀚𑀺𑀯𑁆𑀳𑀸…  𑀓𑀸𑀬𑁄…  𑀭𑀽𑀧𑀁…  𑀲𑀤𑁆𑀤𑁄…  𑀕𑀦𑁆𑀥𑁄…  𑀭𑀲𑁄…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁄…  𑀫𑀦𑁄…

285\. »  𑀥𑀫𑁆𑀫𑁄 𑀆𑀬𑀢𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

286\. «  𑀆𑀬𑀢𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀦𑁆𑀢𑀺?

287\. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀆𑀬𑀢𑀦𑀜𑁆𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀆𑀬𑀢𑀦𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁𑁇
