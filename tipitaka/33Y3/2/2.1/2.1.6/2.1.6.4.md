# 𑁨𑁇𑁧𑁇𑁬𑁇𑁪 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

### 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

(113.)

1004\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1005\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

1006\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…  𑀫𑀦𑀸𑀬𑀢𑀦𑀁…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1007\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

(114.)

1008\. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…  𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1009\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇
