# 𑁨𑁇𑁩𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(199.)

1598. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1599. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1600. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…  𑀫𑀦𑀸𑀬𑀢𑀦𑀁…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1601. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑁄…𑀧𑁂… 𑁇

(200.)

1602. »  𑀬𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1603. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇
