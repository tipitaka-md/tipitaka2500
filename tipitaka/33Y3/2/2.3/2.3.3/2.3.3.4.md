# 𑁨𑁇𑁩𑁇𑁩𑁇𑁪 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

(220.)

1717\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1718\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1719\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1720\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1721\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1722\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1723\. 𑀬𑁂 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1724\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1725\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1726\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1727\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1728\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1729\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀓𑀁𑁇)

(221.)

1730\. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1731\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1732\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1733\. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1734\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1735\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(222.)

1736\. »  𑀬𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1737\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1738\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(223.)

1739\. »  𑀬𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1740\. 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1741\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
