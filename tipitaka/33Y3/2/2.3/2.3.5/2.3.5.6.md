# 𑁨𑁇𑁩𑁇𑁫𑁇𑁬 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

(244.)

1869\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1870\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1871\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1872\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1873\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1874\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

1875\. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1876\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1877\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1878\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1879\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1880\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀦𑀁 𑀢𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1881\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1882\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1883\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1884\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀓𑀁𑁇)

(245.)

1885\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1887\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1888\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1889\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1890\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1891\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1892\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1893\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀓𑀁𑁇)

1894\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1895\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1896\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

1897\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1898\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1900\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
