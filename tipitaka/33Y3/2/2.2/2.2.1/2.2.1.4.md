# 𑁨𑁇𑁨𑁇𑁧𑁇𑁪 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

(126.)

1092. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1093. 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀲𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀅𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1094. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1095. 𑀅𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑁄𑀢𑀓𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1096. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1097. 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀲𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1098. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1099. 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1100. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1101. 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀲𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1102. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1103. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1104. 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀲𑀘𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1105. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1106. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1107. 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1108. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀓𑀁𑁇)

(127.)

1109. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1110. 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀲𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1111. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1112. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1113. 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀲𑀘𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1114. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1115. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1116. 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1117. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀓𑀁𑁇)

(128.)

1118. »  𑀬𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1119. 𑀅𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1120. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1121. 𑀅𑀘𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1122. »  𑀬𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1123. 𑀅𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1124. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑀓𑀁𑁇)

(129.)

1125. »  𑀬𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1126. 𑀅𑀘𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1127. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
