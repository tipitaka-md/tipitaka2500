# 𑁨𑁇𑁨𑁇𑁫𑁇𑁪 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

(175.)

1412\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1413\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1414\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1415\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1416\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1417\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1418\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1419\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1420\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1421\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1422\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1423\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1424\. »  𑀬𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1425\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1426\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1427\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

(176.)

1428\. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1429\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1430\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1431\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀪𑀯𑀺𑀓𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

1432\. »  𑀬𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1433\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀖𑀸𑀦𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1434\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1435\. 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

(177.)

1436\. »  𑀬𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

1437\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀭𑀽𑀧𑀓𑀸𑀦𑀁 𑀘𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1438\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

1439\. 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇

(178.)

1440\. »  𑀬𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1441\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀦𑀸𑀬𑀢𑀦𑀁 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺? 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇
