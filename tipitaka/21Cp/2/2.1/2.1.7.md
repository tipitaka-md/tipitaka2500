# 𑁨𑁇𑁧𑁇𑁭 𑀫𑀸𑀢𑀗𑁆𑀕𑀘𑀭𑀺𑀬

* 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀯𑀕𑁆𑀕

* 𑀲𑀻𑀮𑀧𑀸𑀭𑀫𑀻

* 𑀫𑀸𑀢𑀗𑁆𑀕𑀘𑀭𑀺𑀬

(60.)

227\. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀚𑀝𑀺𑀮𑁄 𑀉𑀕𑁆𑀕𑀢𑀸𑀧𑀦𑁄;_  
_𑀫𑀸𑀢𑀗𑁆𑀕𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀲𑀻𑀮𑀯𑀸 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


(61.)

228\. _𑀅𑀳𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀏𑀓𑁄,_  
_𑀕𑀗𑁆𑀕𑀸𑀓𑀽𑀮𑁂 𑀯𑀲𑀸𑀫𑀼𑀪𑁄;_  
_𑀅𑀳𑀁 𑀯𑀲𑀸𑀫𑀺 𑀉𑀧𑀭𑀺,_  
_𑀳𑁂𑀝𑁆𑀞𑀸 𑀯𑀲𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


(62.)

229\. _𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀽𑀮𑀫𑁆𑀳𑀺,_  
_𑀉𑀤𑁆𑀥𑀁 𑀫𑁂 𑀅𑀲𑁆𑀲𑀫𑀤𑁆𑀤𑀲;_  
_𑀢𑀢𑁆𑀣 𑀫𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀪𑀺𑀲𑀧𑀺 𑀫𑀼𑀤𑁆𑀥𑀨𑀸𑀮𑀦𑀁𑁇_  


(63.)

230\. _𑀬𑀤𑀺𑀳𑀁 𑀢𑀲𑁆𑀲 𑀧𑀓𑀼𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀁,_  
_𑀬𑀤𑀺 𑀲𑀻𑀮𑀁 𑀦 𑀕𑁄𑀧𑀬𑁂;_  
_𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀲𑁆𑀲,_  
_𑀓𑀭𑁂𑀬𑁆𑀬𑀁 𑀙𑀸𑀭𑀺𑀓𑀁 𑀯𑀺𑀬𑁇_  


(64.)

231\. _𑀬𑀁 𑀲𑁄 𑀢𑀤𑀸 𑀫𑀁 𑀅𑀪𑀺𑀲𑀧𑀺,_  
_𑀓𑀼𑀧𑀺𑀢𑁄 𑀤𑀼𑀝𑁆𑀞𑀫𑀸𑀦𑀲𑁄;_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀫𑀢𑁆𑀣𑀓𑁂 𑀦𑀺𑀧𑀢𑀺,_  
_𑀬𑁄𑀕𑁂𑀦 𑀢𑀁 𑀧𑀫𑁄𑀘𑀬𑀺𑀁𑁇_  


(65.)

232\. _𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀺𑀁 𑀫𑀫 𑀲𑀻𑀮𑀁,_  
_𑀦𑀸𑀭𑀓𑁆𑀔𑀺𑀁 𑀫𑀫 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀲𑀻𑀮𑀯𑀸 𑀳𑀺 𑀢𑀤𑀸 𑀆𑀲𑀺𑀁,_  
_𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀓𑀸𑀭𑀡𑀸”𑀢𑀺𑁇_  


---

233\. 𑀫𑀸𑀢𑀗𑁆𑀕𑀘𑀭𑀺𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
