# 𑁨𑁇𑁧𑁇𑁪 𑀘𑀽𑀍𑀅𑀩𑁄𑀥𑀺𑀘𑀭𑀺𑀬

### 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀮𑀧𑀸𑀭𑀫𑀻

### 𑀘𑀽𑀍𑀅𑀩𑁄𑀥𑀺𑀘𑀭𑀺𑀬

(26.)

190. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀘𑀽𑀍𑀅𑀩𑁄𑀥𑀺 𑀲𑀼𑀲𑀻𑀮𑀯𑀸;_  
_𑀪𑀯𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀬𑀢𑁄,_  
_𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀁𑁇_  


(27.)

191. _𑀬𑀸 𑀫𑁂 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀆𑀲𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀓𑀦𑀓𑀲𑀦𑁆𑀦𑀺𑀪𑀸;_  
_𑀲𑀸𑀧𑀺 𑀯𑀝𑁆𑀝𑁂 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀸,_  
_𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑁇_  


(28.)

192. _𑀦𑀺𑀭𑀸𑀮𑀬𑀸 𑀙𑀺𑀦𑁆𑀦𑀩𑀦𑁆𑀥𑀽,_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀸 𑀓𑀼𑀮𑁂 𑀕𑀡𑁂;_  
_𑀘𑀭𑀦𑁆𑀢𑀸 𑀕𑀸𑀫𑀦𑀺𑀕𑀫𑀁,_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀫𑀼𑀧𑀸𑀕𑀫𑀼𑀁𑁇_  


(29.)

193. _𑀢𑀢𑁆𑀣 𑀯𑀲𑀸𑀫 𑀦𑀺𑀧𑀓𑀸,_  
_𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀓𑀼𑀮𑁂 𑀕𑀡𑁂;_  
_𑀦𑀺𑀭𑀸𑀓𑀼𑀮𑁂 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑁂,_  
_𑀭𑀸𑀚𑀼𑀬𑁆𑀬𑀸𑀦𑁂 𑀯𑀲𑀸𑀫𑀼𑀪𑁄𑁇_  


(30.)

194. _𑀉𑀬𑁆𑀬𑀸𑀦𑀤𑀲𑁆𑀲𑀦𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀭𑀸𑀚𑀸 𑀅𑀤𑁆𑀤𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁;_  
_𑀉𑀧𑀕𑀫𑁆𑀫 𑀫𑀫𑀁 𑀧𑀼𑀘𑁆𑀙𑀺,_  
_‘𑀢𑀼𑀬𑁆𑀳𑁂𑀲𑀸 𑀓𑀸 𑀓𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸’𑁇_  


(31.)

195. _𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁 𑀢𑀲𑁆𑀲,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁;_  
_‘𑀦 𑀫𑀬𑁆𑀳𑀁 𑀪𑀭𑀺𑀬𑀸 𑀏𑀲𑀸,_  
_𑀲𑀳𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑀲𑀸𑀲𑀦𑀻’𑁇_  


(32.)

196. _𑀢𑀺𑀲𑁆𑀲𑀸 𑀲𑀸𑀭𑀢𑁆𑀢𑀕𑀥𑀺𑀢𑁄,_  
_𑀕𑀸𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑀦 𑀘𑁂𑀝𑀓𑁂;_  
_𑀦𑀺𑀧𑁆𑀧𑀻𑀍𑀅𑀬𑀦𑁆𑀢𑁄 𑀩𑀮𑀲𑀸,_  
_𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀯𑁂𑀲𑀬𑀺𑁇_  


(33.)

197. _𑀑𑀤𑀧𑀢𑁆𑀢𑀓𑀺𑀬𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀲𑀳𑀚𑀸 𑀏𑀓𑀲𑀸𑀲𑀦𑀻;_  
_𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀦𑀬𑀦𑁆𑀢𑀺𑀬𑀸,_  
_𑀓𑁄𑀧𑁄 𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀣𑁇_  


(34.)

198. _𑀲𑀳 𑀓𑁄𑀧𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀓𑁄𑀧𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀺𑀁,_  
_𑀦𑀸𑀤𑀸𑀲𑀺𑀁 𑀯𑀟𑁆𑀠𑀺𑀢𑀽𑀧𑀭𑀺𑁇_  


(35.)

199. _𑀬𑀤𑀺 𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀓𑁄𑀘𑀺,_  
_𑀓𑁄𑀝𑁆𑀝𑁂𑀬𑁆𑀬 𑀢𑀺𑀡𑁆𑀳𑀲𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀦𑁂𑀯 𑀲𑀻𑀮𑀁 𑀧𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀁,_  
_𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀓𑀸𑀭𑀡𑀸𑁇_  


(36.)

200. _𑀦 𑀫𑁂𑀲𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀤𑁂𑀲𑁆𑀲𑀸,_  
_𑀦𑀧𑀺 𑀫𑁂 𑀩𑀮𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀲𑀻𑀮𑀸𑀦𑀼𑀭𑀓𑁆𑀔𑀺𑀲𑀦𑁆”𑀢𑀺𑁇_  


---

201. 𑀘𑀽𑀍𑀅𑀩𑁄𑀥𑀺𑀘𑀭𑀺𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
