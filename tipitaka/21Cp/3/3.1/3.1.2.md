# 𑁩𑁇𑁧𑁇𑁨 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀘𑀭𑀺𑀬

* 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀯𑀕𑁆𑀕

* 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀸𑀭𑀫𑀻

* 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀘𑀭𑀺𑀬

(7.)

276\. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀇𑀦𑁆𑀤𑀧𑀢𑁆𑀣𑁂 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀓𑀸𑀫𑀺𑀢𑁄 𑀤𑀬𑀺𑀢𑁄 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑁄𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄𑁇_  


(8.)

277\. _𑀲𑀻𑀮𑀯𑀸 𑀕𑀼𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀝𑀺𑀪𑀸𑀦𑀯𑀸;_  
_𑀯𑀼𑀟𑁆𑀠𑀸𑀧𑀘𑀸𑀬𑀻 𑀳𑀺𑀭𑀻𑀫𑀸,_  
_𑀲𑀗𑁆𑀕𑀳𑁂𑀲𑀼 𑀘 𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


(9.)

278\. _𑀢𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀧𑀢𑀺𑀓𑀭𑁄,_  
_𑀅𑀳𑁄𑀲𑀺 𑀓𑀼𑀳𑀓𑀢𑀸𑀧𑀲𑁄;_  
_𑀆𑀭𑀸𑀫𑀁 𑀫𑀸𑀮𑀸𑀯𑀘𑁆𑀙𑀜𑁆𑀘,_  
_𑀭𑁄𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀚𑀻𑀯𑀢𑀺𑁇_  


(10.)

279\. _𑀢𑀫𑀳𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀓𑀼𑀳𑀓𑀁,_  
_𑀣𑀼𑀲𑀭𑀸𑀲𑀺𑀁𑀯 𑀅𑀢𑀡𑁆𑀟𑀼𑀮𑀁;_  
_𑀤𑀼𑀫𑀁𑀯 𑀅𑀦𑁆𑀢𑁄 𑀲𑀼𑀲𑀺𑀭𑀁,_  
_𑀓𑀤𑀮𑀺𑀁𑀯 𑀅𑀲𑀸𑀭𑀓𑀁𑁇_  


(11.)

280\. _𑀦𑀢𑁆𑀣𑀺𑀫𑀲𑁆𑀲 𑀲𑀢𑀁 𑀥𑀫𑁆𑀫𑁄,_  
_𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀧𑀕𑀢𑁄 𑀅𑀬𑀁;_  
_𑀳𑀺𑀭𑀻𑀲𑀼𑀓𑁆𑀓𑀥𑀫𑁆𑀫𑀚𑀳𑀺𑀢𑁄,_  
_𑀚𑀻𑀯𑀺𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀡𑀸𑁇_  


(12.)

281\. _𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀳𑀼 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑁄,_  
_𑀅𑀝𑀯𑀻𑀳𑀺 𑀧𑀭𑀦𑁆𑀢𑀺𑀳𑀺;_  
_𑀢𑀁 𑀦𑀺𑀲𑁂𑀥𑁂𑀢𑀼𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄,_  
_𑀅𑀦𑀼𑀲𑀸𑀲𑀺 𑀧𑀺𑀢𑀸 𑀫𑀫𑀁𑁇_  


(13.)

282\. _‘𑀫𑀸 𑀧𑀫𑀚𑁆𑀚𑀺 𑀢𑀼𑀯𑀁 𑀢𑀸𑀢,_  
_𑀚𑀝𑀺𑀮𑀁 𑀉𑀕𑁆𑀕𑀢𑀸𑀧𑀦𑀁;_  
_𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀤𑀤𑁄 𑀳𑀺 𑀲𑁄’𑁇_  


(14.)

283\. _𑀢𑀫𑀳𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁;_  
_‘𑀓𑀘𑁆𑀘𑀺 𑀢𑁂 𑀕𑀳𑀧𑀢𑀺 𑀓𑀼𑀲𑀮𑀁,_  
_𑀓𑀺𑀁 𑀯𑀸 𑀢𑁂 𑀆𑀳𑀭𑀻𑀬𑀢𑀼’𑁇_  


(15.)

284\. _𑀢𑁂𑀦 𑀲𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀆𑀲𑀺,_  
_𑀓𑀼𑀳𑀓𑁄 𑀫𑀸𑀦𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄;_  
_‘𑀖𑀸𑀢𑀸𑀧𑁂𑀫𑀺 𑀢𑀼𑀯𑀁 𑀅𑀚𑁆𑀚,_  
_𑀭𑀝𑁆𑀞𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀸𑀫𑀺 𑀯𑀸’𑁇_  


(16.)

285\. _𑀦𑀺𑀲𑁂𑀥𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀭𑀸𑀚𑀸 𑀓𑀼𑀳𑀓𑀫𑀩𑁆𑀭𑀯𑀺;_  
_‘𑀓𑀘𑁆𑀘𑀺 𑀢𑁂 𑀪𑀦𑁆𑀢𑁂 𑀔𑀫𑀦𑀻𑀬𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀦𑁄 𑀢𑁂 𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑁄’𑁇_  


(17.)

286\. _𑀢𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀻 𑀧𑀸𑀧𑁄,_  
_𑀓𑀼𑀫𑀸𑀭𑁄 𑀬𑀣𑀸 𑀦𑀸𑀲𑀺𑀬𑁄;_  
_𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀆𑀡𑀸𑀧𑁂𑀲𑀺 𑀫𑀳𑀻𑀧𑀢𑀺𑁇_  


(18.)

287\. _‘𑀲𑀻𑀲𑀁 𑀢𑀢𑁆𑀣𑁂𑀯 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀘𑀢𑀼𑀔𑀡𑁆𑀟𑀺𑀓𑀁;_  
_𑀭𑀣𑀺𑀬𑀸 𑀭𑀣𑀺𑀬𑀁 𑀤𑀲𑁆𑀲𑁂𑀣,_  
_𑀲𑀸 𑀕𑀢𑀺 𑀚𑀝𑀺𑀮𑀳𑀻𑀍𑀇𑀢𑀸’𑁇_  


(19.)

288\. _𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀺𑀓𑀸 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀘𑀡𑁆𑀟𑀸 𑀮𑀼𑀤𑁆𑀤𑀸 𑀅𑀓𑀸𑀭𑀼𑀡𑀸;_  
_𑀫𑀸𑀢𑀼 𑀅𑀗𑁆𑀓𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀦𑀬𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(20.)

289\. _𑀢𑁂𑀲𑀸𑀳𑀁 𑀏𑀯𑀫𑀯𑀘𑀁,_  
_𑀩𑀦𑁆𑀥𑀢𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁;_  
_‘𑀭𑀜𑁆𑀜𑁄 𑀤𑀲𑁆𑀲𑁂𑀣 𑀫𑀁 𑀔𑀺𑀧𑁆𑀧𑀁,_  
_𑀭𑀸𑀚𑀓𑀺𑀭𑀺𑀬𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀺 𑀫𑁂’𑁇_  


(21.)

290\. _𑀢𑁂 𑀫𑀁 𑀭𑀜𑁆𑀜𑁄 𑀤𑀲𑁆𑀲𑀬𑀺𑀁𑀲𑀼,_  
_𑀧𑀸𑀧𑀲𑁆𑀲 𑀧𑀸𑀧𑀲𑁂𑀯𑀺𑀦𑁄;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀢𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺𑀁,_  
_𑀫𑀫𑀜𑁆𑀘 𑀯𑀲𑀫𑀸𑀦𑀬𑀺𑀁𑁇_  


(22.)

291\. _𑀲𑁄 𑀫𑀁 𑀢𑀢𑁆𑀣 𑀔𑀫𑀸𑀧𑁂𑀲𑀺,_  
_𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀫𑀤𑀸𑀲𑀺 𑀫𑁂;_  
_𑀲𑁄𑀳𑀁 𑀢𑀫𑀁 𑀤𑀸𑀮𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(23.)

292\. _𑀦 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀁 𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀁,_  
_𑀓𑀸𑀫𑀪𑁄𑀕𑁄 𑀦 𑀤𑁂𑀲𑁆𑀲𑀺𑀬𑁄;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀭𑀚𑁆𑀚𑀁 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

293\. 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀘𑀭𑀺𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
