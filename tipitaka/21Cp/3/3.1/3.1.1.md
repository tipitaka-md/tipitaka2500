# 𑁩𑁇𑁧𑁇𑁧 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀘𑀭𑀺𑀬

### 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀯𑀕𑁆𑀕

### 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀸𑀭𑀫𑀻

### 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀘𑀭𑀺𑀬

(1.)

269\. _“𑀬𑀤𑀸𑀳𑀁 𑀅𑀫𑀺𑀢𑀬𑀲𑁄,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑁄;_  
_𑀉𑀲𑁆𑀲𑀸𑀯𑀩𑀺𑀦𑁆𑀤𑀼𑀁 𑀲𑀽𑀭𑀺𑀬𑀸𑀢𑀧𑁂,_  
_𑀧𑀢𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀁𑀯𑀺𑀚𑀺𑀁𑁇_  


(2.)

270\. _𑀢𑀜𑁆𑀜𑁂𑀯𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀯𑁂𑀕𑀫𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑀬𑀺𑀁;_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽 𑀘 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀫𑀦𑀼𑀬𑀸𑀘𑀳𑀁𑁇_  


(3.)

271\. _𑀬𑀸𑀘𑀦𑁆𑀢𑀺 𑀫𑀁 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸,_  
_𑀲𑀦𑁂𑀕𑀫𑀸 𑀲𑀭𑀝𑁆𑀞𑀓𑀸;_  
_‘𑀅𑀚𑁆𑀚𑁂𑀯 𑀧𑀼𑀢𑁆𑀢 𑀧𑀝𑀺𑀧𑀚𑁆𑀚,_  
_𑀇𑀤𑁆𑀥𑀁 𑀨𑀻𑀢𑀁 𑀫𑀳𑀸𑀫𑀳𑀺𑀁’𑁇_  


(4.)

272\. _𑀲𑀭𑀸𑀚𑀓𑁂 𑀲𑀳𑁄𑀭𑁄𑀥𑁂,_  
_𑀲𑀦𑁂𑀕𑀫𑁂 𑀲𑀭𑀝𑁆𑀞𑀓𑁂;_  
_𑀓𑀭𑀼𑀡𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑁆𑀢𑁂,_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄𑀯 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀺𑀁𑁇_  


(5.)

273\. _𑀓𑁂𑀯𑀮𑀁 𑀧𑀣𑀯𑀺𑀁 𑀭𑀚𑁆𑀚𑀁,_  
_𑀜𑀸𑀢𑀺𑀧𑀭𑀺𑀚𑀦𑀁 𑀬𑀲𑀁;_  
_𑀘𑀚𑀫𑀸𑀦𑁄 𑀦 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀺𑀁,_  
_𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀓𑀸𑀭𑀡𑀸𑁇_  


(6.)

274\. _𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀸 𑀦 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀸,_  
_𑀦𑀧𑀺 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀁 𑀫𑀳𑀸𑀬𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀭𑀚𑁆𑀚𑀁 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀺𑀦𑁆”𑀢𑀺𑁇_  


---
