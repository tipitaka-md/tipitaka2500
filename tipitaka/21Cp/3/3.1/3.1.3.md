# 𑁩𑁇𑁧𑁇𑁩 𑀅𑀬𑁄𑀖𑀭𑀘𑀭𑀺𑀬

### 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀯𑀕𑁆𑀕

### 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀸𑀭𑀫𑀻

### 𑀅𑀬𑁄𑀖𑀭𑀘𑀭𑀺𑀬

(24.)

294. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀲𑁆𑀲 𑀅𑀢𑁆𑀭𑀚𑁄;_  
_𑀅𑀬𑁄𑀖𑀭𑀫𑁆𑀳𑀺 𑀲𑀁𑀯𑀟𑁆𑀠𑁄,_  
_𑀦𑀸𑀫𑁂𑀦𑀸𑀲𑀺 𑀅𑀬𑁄𑀖𑀭𑁄𑁇_  


(25.)

295. _‘𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑁄 𑀮𑀤𑁆𑀥𑁄,_  
_𑀲𑀁𑀧𑀻𑀍𑀏 𑀧𑀢𑀺𑀧𑁄𑀲𑀺𑀢𑁄;_  
_𑀅𑀚𑁆𑀚𑁂𑀯 𑀧𑀼𑀢𑁆𑀢 𑀧𑀝𑀺𑀧𑀚𑁆𑀚,_  
_𑀓𑁂𑀯𑀮𑀁 𑀯𑀲𑀼𑀥𑀁 𑀇𑀫𑀁’𑁇_  


(26.)

296. _𑀲𑀭𑀝𑁆𑀞𑀓𑀁 𑀲𑀦𑀺𑀕𑀫𑀁,_  
_𑀲𑀚𑀦𑀁 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯 𑀔𑀢𑁆𑀢𑀺𑀬𑀁;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(27.)

297. _‘𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀳𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀳𑀻𑀦𑀫𑀼𑀓𑁆𑀓𑀝𑁆𑀞𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀦𑀺𑀭𑀸𑀭𑀓𑁆𑀔𑀸 𑀲𑀓𑁂 𑀕𑁂𑀳𑁂,_  
_𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺 𑀲𑀓𑀜𑀸𑀢𑀺𑀪𑀺𑁇_  


(28.)

298. _𑀇𑀤𑀁 𑀮𑁄𑀓𑁂 𑀉𑀢𑁆𑀢𑀭𑀺𑀬𑀁,_  
_𑀲𑀁𑀧𑀻𑀍𑀏 𑀫𑀫 𑀧𑁄𑀲𑀦𑀁;_  
_𑀅𑀬𑁄𑀖𑀭𑀫𑁆𑀳𑀺 𑀲𑀁𑀯𑀟𑁆𑀠𑁄,_  
_𑀅𑀧𑁆𑀧𑀪𑁂 𑀘𑀦𑁆𑀤𑀲𑀽𑀭𑀺𑀬𑁂𑁇_  


(29.)

299. _𑀧𑀽𑀢𑀺𑀓𑀼𑀡𑀧𑀲𑀫𑁆𑀧𑀼𑀡𑁆𑀡𑀸,_  
_𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀢𑁄;_  
_𑀢𑀢𑁄 𑀖𑁄𑀭𑀢𑀭𑁂 𑀤𑀼𑀓𑁆𑀔𑁂,_  
_𑀧𑀼𑀦 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀬𑁄𑀖𑀭𑁂𑁇_  


(30.)

300. _𑀬𑀤𑀺𑀳𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀧𑀢𑁆𑀯𑀸,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀭𑀫𑀤𑀸𑀭𑀼𑀡𑀁;_  
_𑀭𑀚𑁆𑀚𑁂𑀲𑀼 𑀬𑀤𑀺 𑀭𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀧𑀸𑀧𑀸𑀦𑀁 𑀉𑀢𑁆𑀢𑀫𑁄 𑀲𑀺𑀬𑀁𑁇_  


(31.)

301. _𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀓𑀸𑀬𑁂𑀦,_  
_𑀭𑀚𑁆𑀚𑁂𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀺𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀁,_  
_𑀬𑀢𑁆𑀣 𑀫𑀁 𑀫𑀘𑁆𑀘𑀼 𑀦 𑀫𑀤𑁆𑀤𑀺𑀬𑁂’𑁇_  


(32.)

302. _𑀏𑀯𑀸𑀳𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀭𑀯𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀚𑀦𑁂;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀸𑀯𑀺𑀲𑀺𑀁 𑀓𑀸𑀦𑀦𑀁 𑀯𑀦𑀁𑁇_  


(33.)

303. _𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀸 𑀦 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀸,_  
_𑀦𑀧𑀺 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀁 𑀫𑀳𑀸𑀬𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀭𑀚𑁆𑀚𑀁 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

304. 𑀅𑀬𑁄𑀖𑀭𑀘𑀭𑀺𑀬𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
