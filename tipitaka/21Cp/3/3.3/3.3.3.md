# 𑁩𑁇𑁩𑁇𑁩 𑀯𑀝𑁆𑀝𑀧𑁄𑀢𑀓𑀘𑀭𑀺𑀬

* 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀘𑁆𑀘𑀧𑀸𑀭𑀫𑀻

* 𑀯𑀝𑁆𑀝𑀧𑁄𑀢𑀓𑀘𑀭𑀺𑀬

(72.)

348\. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀫𑀕𑀥𑁂 𑀯𑀝𑁆𑀝𑀧𑁄𑀢𑀓𑁄;_  
_𑀅𑀚𑀸𑀢𑀧𑀓𑁆𑀔𑁄 𑀢𑀭𑀼𑀡𑁄,_  
_𑀫𑀁𑀲𑀧𑁂𑀲𑀺 𑀓𑀼𑀮𑀸𑀯𑀓𑁂𑁇_  


(73.)

349\. _𑀫𑀼𑀔𑀢𑀼𑀡𑁆𑀟𑀓𑁂𑀦𑀸𑀳𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀸𑀢𑀸 𑀧𑁄𑀲𑀬𑀢𑀻 𑀫𑀫𑀁;_  
_𑀢𑀲𑁆𑀲𑀸 𑀨𑀲𑁆𑀲𑁂𑀦 𑀚𑀻𑀯𑀸𑀫𑀺,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀓𑀸𑀬𑀺𑀓𑀁 𑀩𑀮𑀁𑁇_  


(74.)

350\. _𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁂 𑀕𑀺𑀫𑁆𑀳𑀲𑀫𑀬𑁂,_  
_𑀤𑀯𑀟𑀸𑀳𑁄 𑀧𑀤𑀺𑀧𑁆𑀧𑀢𑀺;_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀢𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁,_  
_𑀧𑀸𑀯𑀓𑁄 𑀓𑀡𑁆𑀳𑀯𑀢𑁆𑀢𑀦𑀻𑁇_  


(75.)

351\. _𑀥𑀫𑀥𑀫𑀸𑀇𑀢𑀺 𑀏𑀯𑀁,_  
_𑀲𑀤𑁆𑀤𑀸𑀬𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀲𑀺𑀔𑀻;_  
_𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑁄,_  
_𑀅𑀕𑁆𑀕𑀺 𑀫𑀫𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(76.)

352\. _𑀅𑀕𑁆𑀕𑀺𑀯𑁂𑀕𑀪𑀬𑀸𑀢𑀻𑀢𑀸,_  
_𑀢𑀲𑀺𑀢𑀸 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀸 𑀫𑀫;_  
_𑀓𑀼𑀮𑀸𑀯𑀓𑁂 𑀫𑀁 𑀙𑀟𑁆𑀟𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀫𑁄𑀘𑀬𑀼𑀁𑁇_  


(77.)

353\. _𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑁂 𑀧𑀚𑀳𑀸𑀫𑀺,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀓𑀸𑀬𑀺𑀓𑀁 𑀩𑀮𑀁;_  
_𑀲𑁄𑀳𑀁 𑀅𑀕𑀢𑀺𑀓𑁄 𑀢𑀢𑁆𑀣,_  
_𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀳𑀁 𑀢𑀤𑀸𑁇_  


(78.)

354\. _‘𑀬𑁂𑀲𑀸𑀳𑀁 𑀉𑀧𑀥𑀸𑀯𑁂𑀬𑁆𑀬𑀁,_  
_𑀪𑀻𑀢𑁄 𑀢𑀲𑀺𑀢𑀯𑁂𑀥𑀺𑀢𑁄;_  
_𑀢𑁂 𑀫𑀁 𑀑𑀳𑀸𑀬 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀸,_  
_𑀓𑀣𑀁 𑀫𑁂 𑀅𑀚𑁆𑀚 𑀓𑀸𑀢𑀯𑁂𑁇_  


(79.)

355\. _𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀲𑀻𑀮𑀕𑀼𑀡𑁄,_  
_𑀲𑀘𑁆𑀘𑀁 𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀦𑀼𑀤𑁆𑀤𑀬𑀸;_  
_𑀢𑁂𑀦 𑀲𑀘𑁆𑀘𑁂𑀦 𑀓𑀸𑀳𑀸𑀫𑀺,_  
_𑀲𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(80.)

356\. _𑀆𑀯𑁂𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀩𑀮𑀁,_  
_𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀓𑁂 𑀚𑀺𑀦𑁂;_  
_𑀲𑀘𑁆𑀘𑀩𑀮𑀫𑀯𑀲𑁆𑀲𑀸𑀬,_  
_𑀲𑀘𑁆𑀘𑀓𑀺𑀭𑀺𑀬𑀫𑀓𑀸𑀲𑀳𑀁𑁇_  


(81.)

357\. _𑀲𑀦𑁆𑀢𑀺 𑀧𑀓𑁆𑀔𑀸 𑀅𑀧𑀢𑀦𑀸,_  
_𑀲𑀦𑁆𑀢𑀺 𑀧𑀸𑀤𑀸 𑀅𑀯𑀜𑁆𑀘𑀦𑀸;_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀸 𑀘 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀸,_  
_𑀚𑀸𑀢𑀯𑁂𑀤 𑀧𑀝𑀺𑀓𑁆𑀓𑀫’𑁇_  


(82.)

358\. _𑀲𑀳𑀲𑀘𑁆𑀘𑁂 𑀓𑀢𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀫𑀳𑀸𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀲𑀺𑀔𑀻;_  
_𑀯𑀚𑁆𑀚𑁂𑀲𑀺 𑀲𑁄𑀍𑀅𑀲𑀓𑀭𑀻𑀲𑀸𑀦𑀺,_  
_𑀉𑀤𑀓𑀁 𑀧𑀢𑁆𑀯𑀸 𑀬𑀣𑀸 𑀲𑀺𑀔𑀻;_  
_𑀲𑀘𑁆𑀘𑁂𑀦 𑀫𑁂 𑀲𑀫𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀏𑀲𑀸 𑀫𑁂 𑀲𑀘𑁆𑀘𑀧𑀸𑀭𑀫𑀻”𑀢𑀺𑁇_  


---
