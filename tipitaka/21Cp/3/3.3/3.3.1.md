# 𑁩𑁇𑁩𑁇𑁧 𑀓𑀧𑀺𑀭𑀸𑀚𑀘𑀭𑀺𑀬

### 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀘𑁆𑀘𑀧𑀸𑀭𑀫𑀻

### 𑀓𑀧𑀺𑀭𑀸𑀚𑀘𑀭𑀺𑀬

(67.)

341. _“𑀬𑀤𑀸 𑀅𑀳𑀁 𑀓𑀧𑀺 𑀆𑀲𑀺𑀁,_  
_𑀦𑀤𑀻𑀓𑀽𑀮𑁂 𑀤𑀭𑀻𑀲𑀬𑁂;_  
_𑀧𑀻𑀍𑀇𑀢𑁄 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑁂𑀦,_  
_𑀕𑀫𑀦𑀁 𑀦 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(68.)

342. _𑀬𑀫𑁆𑀳𑁄𑀓𑀸𑀲𑁂 𑀅𑀳𑀁 𑀞𑀢𑁆𑀯𑀸,_  
_𑀑𑀭𑀸 𑀧𑀸𑀭𑀁 𑀧𑀢𑀸𑀫𑀳𑀁;_  
_𑀢𑀢𑁆𑀣𑀘𑁆𑀙𑀺 𑀲𑀢𑁆𑀢𑀼 𑀯𑀥𑀓𑁄,_  
_𑀓𑀼𑀫𑁆𑀪𑀻𑀮𑁄 𑀮𑀼𑀤𑁆𑀤𑀤𑀲𑁆𑀲𑀦𑁄𑁇_  


(69.)

343. _𑀲𑁄 𑀫𑀁 𑀅𑀲𑀁𑀲𑀺 ‘𑀏𑀳𑀻’𑀢𑀺,_  
_‘𑀅𑀳𑀫𑁆𑀧𑁂𑀫𑀻’𑀢𑀺 𑀢𑀁 𑀯𑀢𑀺𑀁;_  
_𑀢𑀲𑁆𑀲 𑀫𑀢𑁆𑀣𑀓𑀫𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀧𑀭𑀓𑀽𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀳𑀺𑀁𑁇_  


(70.)

344. _𑀦 𑀢𑀲𑁆𑀲 𑀅𑀮𑀺𑀓𑀁 𑀪𑀡𑀺𑀢𑀁,_  
_𑀬𑀣𑀸 𑀯𑀸𑀘𑀁 𑀅𑀓𑀸𑀲𑀳𑀁;_  
_𑀲𑀘𑁆𑀘𑁂𑀦 𑀫𑁂 𑀲𑀫𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀏𑀲𑀸 𑀫𑁂 𑀲𑀘𑁆𑀘𑀧𑀸𑀭𑀫𑀻”𑀢𑀺𑁇_  


---

345. 𑀓𑀧𑀺𑀭𑀸𑀚𑀘𑀭𑀺𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
