# 𑁩𑁇𑁨𑁇𑁧 𑀢𑁂𑀫𑀺𑀬𑀘𑀭𑀺𑀬

### 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀧𑀸𑀭𑀫𑀻

### 𑀢𑁂𑀫𑀺𑀬𑀘𑀭𑀺𑀬

(48.)

321. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀲𑁆𑀲 𑀅𑀢𑁆𑀭𑀚𑁄;_  
_𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑁄𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀢𑁂𑀫𑀺𑀬𑁄𑀢𑀺 𑀯𑀤𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(49.)

322. _𑀲𑁄𑀍𑀅𑀲𑀺𑀢𑁆𑀣𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀧𑀼𑀫𑁄 𑀢𑀤𑀸;_  
_𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁄 𑀅𑀳𑀫𑁂𑀓𑀓𑁄𑁇_  


(50.)

323. _𑀓𑀺𑀘𑁆𑀙𑀸 𑀮𑀤𑁆𑀥𑀁 𑀧𑀺𑀬𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀁 𑀚𑀼𑀢𑀺𑀦𑁆𑀥𑀭𑀁;_  
_𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀬𑀦𑁂 𑀧𑁄𑀲𑁂𑀢𑀺 𑀫𑀁 𑀧𑀺𑀢𑀸𑁇_  


(51.)

324. _𑀦𑀺𑀤𑁆𑀤𑀸𑀬𑀫𑀸𑀦𑁄 𑀲𑀬𑀦𑀯𑀭𑁂,_  
_𑀧𑀩𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀙𑀢𑁆𑀢𑀁,_  
_𑀬𑁂𑀦𑀸𑀳𑀁 𑀦𑀺𑀭𑀬𑀁 𑀕𑀢𑁄𑁇_  


(52.)

325. _𑀲𑀳 𑀤𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀫𑁂 𑀙𑀢𑁆𑀢𑀁,_  
_𑀢𑀸𑀲𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀪𑁂𑀭𑀯𑁄;_  
_𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄,_  
_‘𑀓𑀣𑀸𑀳𑀁 𑀇𑀫𑀁 𑀫𑀼𑀜𑁆𑀘𑀺𑀲𑁆𑀲𑀁’𑁇_  


(53.)

326. _𑀧𑀼𑀩𑁆𑀩𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀺𑀦𑀻;_  
_𑀲𑀸 𑀫𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁,_  
_𑀢𑀻𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼 𑀬𑁄𑀚𑀬𑀺𑁇_  


(54.)

327. _‘𑀫𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀬𑀁 𑀯𑀺𑀪𑀸𑀯𑀬,_  
_𑀩𑀸𑀮𑀫𑀢𑁄 𑀪𑀯 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑁄 𑀢𑀁 𑀚𑀦𑁄 𑀑𑀘𑀺𑀦𑀸𑀬𑀢𑀼,_  
_𑀏𑀯𑀁 𑀢𑀯 𑀅𑀢𑁆𑀣𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(55.)

328. _𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑀸𑀬𑀳𑀁 𑀢𑀲𑁆𑀲𑀸,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁;_  
_‘𑀓𑀭𑁄𑀫𑀺 𑀢𑁂 𑀢𑀁 𑀯𑀘𑀦𑀁,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀪𑀡𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸𑀲𑀺 𑀫𑁂 𑀅𑀫𑁆𑀫,_  
_𑀳𑀺𑀢𑀓𑀸𑀫𑀸𑀲𑀺 𑀤𑁂𑀯𑀢𑁂’𑁇_  


(56.)

329. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀸𑀕𑀭𑁂𑀯 𑀣𑀮𑀁 𑀮𑀪𑀺𑀁;_  
_𑀳𑀝𑁆𑀞𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑁄,_  
_𑀢𑀬𑁄 𑀅𑀗𑁆𑀕𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀁𑁇_  


(57.)

330. _𑀫𑀽𑀕𑁄 𑀅𑀳𑁄𑀲𑀺𑀁 𑀩𑀥𑀺𑀭𑁄,_  
_𑀧𑀓𑁆𑀔𑁄 𑀕𑀢𑀺𑀯𑀺𑀯𑀚𑁆𑀚𑀺𑀢𑁄;_  
_𑀏𑀢𑁂 𑀅𑀗𑁆𑀕𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑁄𑀍𑀅𑀲𑀁 𑀯𑀲𑀺𑀁𑁇_  


(58.)

331. _𑀢𑀢𑁄 𑀫𑁂 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑁂 𑀘,_  
_𑀚𑀺𑀯𑁆𑀳𑀁 𑀲𑁄𑀢𑀜𑁆𑀘 𑀫𑀤𑁆𑀤𑀺𑀬;_  
_𑀅𑀦𑀽𑀦𑀢𑀁 𑀫𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_‘𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀻’𑀢𑀺 𑀦𑀺𑀦𑁆𑀤𑀺𑀲𑀼𑀁𑁇_  


(59.)

332. _𑀢𑀢𑁄 𑀚𑀸𑀦𑀧𑀤𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀏𑀓𑀫𑀦𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀙𑀟𑁆𑀟𑀦𑀁 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀲𑀼𑀁𑁇_  


(60.)

333. _𑀲𑁄𑀳𑀁 𑀢𑁂𑀲𑀁 𑀫𑀢𑀺𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀳𑀝𑁆𑀞𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑁄;_  
_‘𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀢𑀧𑁄𑀘𑀺𑀡𑁆𑀡𑁄,_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀲𑀫𑀺𑀚𑁆𑀛𑀣’𑁇_  


(61.)

334. _𑀦𑁆𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀮𑀺𑀫𑁆𑀧𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀯𑁂𑀞𑀦𑀁;_  
_𑀙𑀢𑁆𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀸𑀭𑁂𑀲𑀼𑀁 𑀧𑀼𑀭𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁𑁇_  


(62.)

335. _𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀕𑁆𑀕𑀢𑁂 𑀭𑀯𑀺𑀫𑀡𑁆𑀟𑀮𑁂;_  
_𑀭𑀣𑁂𑀦 𑀫𑀁 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀸𑀭𑀣𑀻 𑀯𑀦𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(63.)

336. _𑀏𑀓𑁄𑀓𑀸𑀲𑁂 𑀭𑀣𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀲𑀚𑁆𑀚𑀲𑁆𑀲𑀁 𑀳𑀢𑁆𑀣𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁄;_  
_𑀲𑀸𑀭𑀣𑀻 𑀔𑀡𑀢𑀻 𑀓𑀸𑀲𑀼𑀁,_  
_𑀦𑀺𑀔𑀸𑀢𑀼𑀁 𑀧𑀣𑀯𑀺𑀬𑀸 𑀫𑀫𑀁𑁇_  


(64.)

337. _𑀅𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀢𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑁄 𑀯𑀺𑀯𑀺𑀥𑀓𑀸𑀭𑀡𑀸;_  
_𑀦 𑀪𑀺𑀦𑁆𑀤𑀺𑀁 𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀓𑀸𑀭𑀡𑀸𑁇_  


(65.)

338. _𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀸 𑀦 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀸,_  
_𑀅𑀢𑁆𑀢𑀸 𑀫𑁂 𑀦 𑀘 𑀤𑁂𑀲𑁆𑀲𑀺𑀬𑁄;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀯𑀢𑀫𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀁𑁇_  


(66.)

339. _𑀏𑀢𑁂 𑀅𑀗𑁆𑀕𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑁄𑀍𑀅𑀲𑀁 𑀯𑀲𑀺𑀁;_  
_𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀫𑁂 𑀲𑀫𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀏𑀲𑀸 𑀫𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀧𑀸𑀭𑀫𑀻”𑀢𑀺𑁇_  


---

340. 𑀢𑁂𑀫𑀺𑀬𑀘𑀭𑀺𑀬𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
