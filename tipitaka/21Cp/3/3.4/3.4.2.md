# 𑁩𑁇𑁪𑁇𑁨 𑀏𑀓𑀭𑀸𑀚𑀘𑀭𑀺𑀬

### 𑀬𑀼𑀥𑀜𑁆𑀚𑀬𑀯𑀕𑁆𑀕

### 𑀫𑁂𑀢𑁆𑀢𑀸𑀧𑀸𑀭𑀫𑀻

### 𑀏𑀓𑀭𑀸𑀚𑀘𑀭𑀺𑀬

(114.)

395\. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀏𑀓𑀭𑀸𑀚𑀸𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀧𑀭𑀫𑀁 𑀲𑀻𑀮𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀧𑀲𑀸𑀲𑀸𑀫𑀺 𑀫𑀳𑀸𑀫𑀳𑀺𑀁𑁇_  


(115.)

396\. _𑀤𑀲 𑀓𑀼𑀲𑀮𑀓𑀫𑁆𑀫𑀧𑀣𑁂,_  
_𑀯𑀢𑁆𑀢𑀸𑀫𑀺 𑀅𑀦𑀯𑀲𑁂𑀲𑀢𑁄;_  
_𑀘𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀯𑀢𑁆𑀣𑀽𑀳𑀺,_  
_𑀲𑀗𑁆𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀫𑀳𑀸𑀚𑀦𑀁𑁇_  


(116.)

397\. _𑀏𑀯𑀁 𑀫𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀇𑀥 𑀮𑁄𑀓𑁂 𑀧𑀭𑀢𑁆𑀣 𑀘;_  
_𑀤𑀩𑁆𑀩𑀲𑁂𑀦𑁄 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀧𑀼𑀭𑀁 𑀫𑀫𑁇_  


(117.)

398\. _𑀭𑀸𑀚𑀽𑀧𑀚𑀻𑀯𑁂 𑀦𑀺𑀕𑀫𑁂,_  
_𑀲𑀩𑀮𑀝𑁆𑀞𑁂 𑀲𑀭𑀝𑁆𑀞𑀓𑁂;_  
_𑀲𑀩𑁆𑀩𑀁 𑀳𑀢𑁆𑀣𑀕𑀢𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀓𑀸𑀲𑀼𑀬𑀸 𑀦𑀺𑀔𑀡𑀻 𑀫𑀫𑀁𑁇_  


(118.)

399\. _𑀅𑀫𑀘𑁆𑀘𑀫𑀡𑁆𑀟𑀮𑀁 𑀭𑀚𑁆𑀚𑀁,_  
_𑀨𑀻𑀢𑀁 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀫𑀫;_  
_𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀕𑀳𑀺𑀢𑀁,_  
_𑀧𑀺𑀬𑀁 𑀧𑀼𑀢𑁆𑀢𑀁𑀯 𑀧𑀲𑁆𑀲𑀳𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀫𑁂 𑀲𑀫𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀏𑀲𑀸 𑀫𑁂 𑀫𑁂𑀢𑁆𑀢𑀸𑀧𑀸𑀭𑀫𑀻”𑀢𑀺𑁇_  


---

400\. 𑀏𑀓𑀭𑀸𑀚𑀘𑀭𑀺𑀬𑀁 𑀘𑀼𑀤𑁆𑀤𑀲𑀫𑀁𑁇
