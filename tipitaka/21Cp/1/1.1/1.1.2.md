# 𑁧𑁇𑁧𑁇𑁨 𑀲𑀗𑁆𑀔𑀘𑀭𑀺𑀬

* 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀤𑀸𑀦𑀧𑀸𑀭𑀫𑀻

* 𑀲𑀗𑁆𑀔𑀘𑀭𑀺𑀬

(11.)

13\. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀗𑁆𑀔𑀲𑀯𑁆𑀳𑀬𑁄;_  
_𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀢𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀧𑀝𑁆𑀝𑀦𑀁𑁇_  


(12.)

14\. _𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀁 𑀧𑀝𑀺𑀧𑀣𑁂,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁;_  
_𑀓𑀦𑁆𑀢𑀸𑀭𑀤𑁆𑀥𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁,_  
_𑀢𑀢𑁆𑀢𑀸𑀬 𑀓𑀞𑀺𑀦𑀪𑀽𑀫𑀺𑀬𑀸𑁇_  


(13.)

15\. _𑀢𑀫𑀳𑀁 𑀧𑀝𑀺𑀧𑀣𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀇𑀫𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀁;_  
_‘𑀇𑀤𑀁 𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀸𑀫𑀲𑁆𑀲 𑀚𑀦𑁆𑀢𑀼𑀦𑁄𑁇_  


(14.)

16\. _𑀬𑀣𑀸 𑀓𑀲𑁆𑀲𑀓𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀔𑁂𑀢𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀳𑀸𑀕𑀫𑀁;_  
_𑀢𑀢𑁆𑀣 𑀩𑀻𑀚𑀁 𑀦 𑀭𑁄𑀧𑁂𑀢𑀺,_  
_𑀦 𑀲𑁄 𑀥𑀜𑁆𑀜𑁂𑀦 𑀅𑀢𑁆𑀣𑀺𑀓𑁄𑁇_  


(15.)

17\. _𑀏𑀯𑀫𑁂𑀯𑀸𑀳𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀸𑀫𑁄,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀔𑁂𑀢𑁆𑀢𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀬𑀤𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀁 𑀦 𑀓𑀭𑁄𑀫𑀺,_  
_𑀦𑀸𑀳𑀁 𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀅𑀢𑁆𑀣𑀺𑀓𑁄𑁇_  


(16.)

18\. _𑀬𑀣𑀸 𑀅𑀫𑀘𑁆𑀘𑁄 𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀸𑀫𑁄,_  
_𑀭𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀚𑀦𑁂;_  
_𑀦 𑀤𑁂𑀢𑀺 𑀢𑁂𑀲𑀁 𑀥𑀦𑀥𑀜𑁆𑀜𑀁,_  
_𑀫𑀼𑀤𑁆𑀤𑀺𑀢𑁄 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺𑁇_  


(17.)

19\. _𑀏𑀯𑀫𑁂𑀯𑀸𑀳𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀸𑀫𑁄,_  
_𑀯𑀺𑀧𑀼𑀮𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀬𑀤𑀺 𑀢𑀲𑁆𑀲 𑀤𑀸𑀦𑀁 𑀦 𑀤𑀤𑀸𑀫𑀺,_  
_𑀧𑀭𑀺𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀼𑀜𑁆𑀜𑀢𑁄’𑁇_  


(18.)

20\. _𑀏𑀯𑀸𑀳𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀸𑀳𑀦𑀸;_  
_𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀦𑀺 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁𑁇_  


(19.)

21\. _𑀢𑁂𑀦𑁂𑀯𑀸𑀳𑀁 𑀲𑀢𑀕𑀼𑀡𑀢𑁄,_  
_𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄;_  
_𑀅𑀧𑀺 𑀘 𑀤𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑁄,_  
_𑀏𑀯𑀁 𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀦𑁆”𑀢𑀺𑁇_  


---

22\. 𑀲𑀗𑁆𑀔𑀘𑀭𑀺𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
