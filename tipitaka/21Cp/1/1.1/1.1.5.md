# 𑁧𑁇𑁧𑁇𑁫 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀘𑀭𑀺𑀬

### 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀤𑀸𑀦𑀧𑀸𑀭𑀫𑀻

### 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀘𑀭𑀺𑀬

(37.)

42. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀲𑀢𑁆𑀢𑀭𑀸𑀚𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑁄;_  
_𑀧𑀽𑀚𑀺𑀢𑁄 𑀦𑀭𑀤𑁂𑀯𑁂𑀳𑀺,_  
_𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


(38.)

43. _𑀢𑀤𑀸𑀳𑀁 𑀲𑀢𑁆𑀢𑀭𑀚𑁆𑀚𑁂𑀲𑀼,_  
_𑀬𑀁 𑀫𑁂 𑀆𑀲𑀺 𑀉𑀧𑀸𑀬𑀦𑀁;_  
_𑀢𑁂𑀦 𑀤𑁂𑀫𑀺 𑀫𑀳𑀸𑀤𑀸𑀦𑀁,_  
_𑀅𑀓𑁆𑀔𑁄𑀩𑁆𑀪𑀁 𑀲𑀸𑀕𑀭𑀽𑀧𑀫𑀁𑁇_  


(39.)

44. _𑀦 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀁 𑀥𑀦𑀁 𑀥𑀜𑁆𑀜𑀁,_  
_𑀦𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀦𑀺𑀘𑀬𑁄 𑀫𑀬𑀺;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀤𑁂𑀫𑀺 𑀯𑀭𑀁 𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

45. 𑀫𑀳𑀸𑀕𑁄𑀯𑀺𑀦𑁆𑀤𑀘𑀭𑀺𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
