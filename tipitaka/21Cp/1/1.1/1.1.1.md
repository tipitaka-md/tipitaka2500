# 𑁧𑁇𑁧𑁇𑁧 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀘𑀭𑀺𑀬

* 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀤𑀸𑀦𑀧𑀸𑀭𑀫𑀻

* 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀘𑀭𑀺𑀬

(1.)

2\. _“𑀓𑀧𑁆𑀧𑁂 𑀘 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂,_  
_𑀘𑀢𑀼𑀭𑁄 𑀘 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑁂;_  
_𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀬𑀁 𑀘𑀭𑀺𑀢𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀩𑁄𑀥𑀺𑀧𑀸𑀘𑀦𑀁𑁇_  


(2.)

3\. _𑀅𑀢𑀻𑀢𑀓𑀧𑁆𑀧𑁂 𑀘𑀭𑀺𑀢𑀁,_  
_𑀞𑀧𑀬𑀺𑀢𑁆𑀯𑀸 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀇𑀫𑀫𑁆𑀳𑀺 𑀓𑀧𑁆𑀧𑁂 𑀘𑀭𑀺𑀢𑀁,_  
_𑀧𑀯𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


(3.)

4\. _𑀬𑀤𑀸 𑀅𑀳𑀁 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀲𑀼𑀜𑁆𑀜𑁂 𑀯𑀺𑀧𑀺𑀦𑀓𑀸𑀦𑀦𑁂;_  
_𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀸𑀫𑀺,_  
_𑀅𑀓𑀺𑀢𑁆𑀢𑀺 𑀦𑀸𑀫 𑀢𑀸𑀧𑀲𑁄𑁇_  


(4.)

5\. _𑀢𑀤𑀸 𑀫𑀁 𑀢𑀧𑀢𑁂𑀚𑁂𑀦,_  
_𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑁄 𑀢𑀺𑀤𑀺𑀯𑀸𑀪𑀺𑀪𑀽;_  
_𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀡𑁆𑀡𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀬 𑀫𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(5.)

6\. _𑀧𑀯𑀦𑀸 𑀆𑀪𑀢𑀁 𑀧𑀡𑁆𑀡𑀁,_  
_𑀅𑀢𑁂𑀮𑀜𑁆𑀘 𑀅𑀮𑁄𑀡𑀺𑀓𑀁;_  
_𑀫𑀫 𑀤𑁆𑀯𑀸𑀭𑁂 𑀞𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀓𑀝𑀸𑀳𑁂𑀦 𑀆𑀓𑀺𑀭𑀺𑀁𑁇_  


(6.)

7\. _𑀢𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀧𑀡𑁆𑀡𑀁,_  
_𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀪𑀸𑀚𑀦𑀁;_  
_𑀧𑀼𑀦𑁂𑀲𑀦𑀁 𑀚𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀸𑀯𑀺𑀲𑀺𑀁 𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀓𑀁𑁇_  


(7.)

8\. _𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺,_  
_𑀉𑀧𑀕𑀜𑁆𑀙𑀺 𑀫𑀫𑀦𑁆𑀢𑀺𑀓𑀁;_  
_𑀅𑀓𑀫𑁆𑀧𑀺𑀢𑁄 𑀅𑀦𑁄𑀮𑀕𑁆𑀕𑁄,_  
_𑀏𑀯𑀫𑁂𑀯𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(8.)

9\. _𑀦 𑀫𑁂 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀭𑀻𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑀺𑀬𑀁;_  
_𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑁂𑀦 𑀭𑀢𑀺𑀬𑀸,_  
_𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀫𑀺 𑀢𑀁 𑀤𑀺𑀯𑀁𑁇_  


(9.)

10\. _𑀬𑀤𑀺 𑀫𑀸𑀲𑀫𑁆𑀧𑀺 𑀤𑁆𑀯𑁂𑀫𑀸𑀲𑀁,_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀁 𑀯𑀭𑀁 𑀮𑀪𑁂;_  
_𑀅𑀓𑀫𑁆𑀧𑀺𑀢𑁄 𑀅𑀦𑁄𑀮𑀻𑀦𑁄,_  
_𑀤𑀤𑁂𑀬𑁆𑀬𑀁 𑀤𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(10.)

11\. _𑀦 𑀢𑀲𑁆𑀲 𑀤𑀸𑀦𑀁 𑀤𑀤𑀫𑀸𑀦𑁄,_  
_𑀬𑀲𑀁 𑀮𑀸𑀪𑀜𑁆𑀘 𑀧𑀢𑁆𑀣𑀬𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄,_  
_𑀢𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀆𑀘𑀭𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

12\. 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀘𑀭𑀺𑀬𑀁 𑀧𑀞𑀫𑀁𑁇
