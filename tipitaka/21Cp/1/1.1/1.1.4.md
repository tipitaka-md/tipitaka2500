# 𑁧𑁇𑁧𑁇𑁪 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀘𑀭𑀺𑀬

* 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀤𑀸𑀦𑀧𑀸𑀭𑀫𑀻

* 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀘𑀭𑀺𑀬

(28.)

32\. _“𑀓𑀼𑀲𑀸𑀯𑀢𑀺𑀫𑁆𑀳𑀺 𑀦𑀕𑀭𑁂,_  
_𑀬𑀤𑀸 𑀆𑀲𑀺𑀁 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀦𑀸𑀫,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(29.)

33\. _𑀢𑀢𑁆𑀣𑀸𑀳𑀁 𑀤𑀺𑀯𑀲𑁂 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁,_  
_𑀖𑁄𑀲𑀸𑀧𑁂𑀫𑀺 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁;_  
_‘𑀓𑁄 𑀓𑀺𑀁 𑀇𑀘𑁆𑀙𑀢𑀺 𑀧𑀢𑁆𑀣𑁂𑀢𑀺,_  
_𑀓𑀲𑁆𑀲 𑀓𑀺𑀁 𑀤𑀻𑀬𑀢𑀽 𑀥𑀦𑀁𑁇_  


(30.)

34\. _𑀓𑁄 𑀙𑀸𑀢𑀓𑁄 𑀓𑁄 𑀢𑀲𑀺𑀢𑁄,_  
_𑀓𑁄 𑀫𑀸𑀮𑀁 𑀓𑁄 𑀯𑀺𑀮𑁂𑀧𑀦𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺,_  
_𑀓𑁄 𑀦𑀕𑁆𑀕𑁄 𑀧𑀭𑀺𑀤𑀳𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(31.)

35\. _𑀓𑁄 𑀧𑀣𑁂 𑀙𑀢𑁆𑀢𑀫𑀸𑀤𑁂𑀢𑀺,_  
_𑀓𑁄𑀧𑀸𑀳𑀦𑀸 𑀫𑀼𑀤𑀽 𑀲𑀼𑀪𑀸’;_  
_𑀇𑀢𑀺 𑀲𑀸𑀬𑀜𑁆𑀘 𑀧𑀸𑀢𑁄 𑀘,_  
_𑀖𑁄𑀲𑀸𑀧𑁂𑀫𑀺 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁𑁇_  


(32.)

36\. _𑀦 𑀢𑀁 𑀤𑀲𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼,_  
_𑀦𑀧𑀺 𑀞𑀸𑀦𑀲𑀢𑁂𑀲𑀼 𑀯𑀸;_  
_𑀅𑀦𑁂𑀓𑀲𑀢𑀞𑀸𑀦𑁂𑀲𑀼,_  
_𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀬𑀸𑀘𑀓𑁂 𑀥𑀦𑀁𑁇_  


(33.)

37\. _𑀤𑀺𑀯𑀸 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀬𑀤𑀺 𑀏𑀢𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁄;_  
_𑀮𑀤𑁆𑀥𑀸 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁 𑀪𑁄𑀕𑀁,_  
_𑀧𑀽𑀭𑀳𑀢𑁆𑀣𑁄𑀯 𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(34.)

38\. _𑀏𑀯𑀭𑀽𑀧𑀁 𑀫𑀳𑀸𑀤𑀸𑀦𑀁,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁;_  
_𑀦𑀧𑀸𑀳𑀁 𑀤𑁂𑀲𑁆𑀲𑀁 𑀥𑀦𑀁 𑀤𑀫𑁆𑀫𑀺,_  
_𑀦𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀦𑀺𑀘𑀬𑁄 𑀫𑀬𑀺𑁇_  


(35.)

39\. _𑀬𑀣𑀸𑀧𑀺 𑀆𑀢𑀼𑀭𑁄 𑀦𑀸𑀫,_  
_𑀭𑁄𑀕𑀢𑁄 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀥𑀦𑁂𑀦 𑀯𑁂𑀚𑁆𑀚𑀁 𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀭𑁄𑀕𑀢𑁄 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(36.)

40\. _𑀢𑀣𑁂𑀯𑀸𑀳𑀁 𑀚𑀸𑀦𑀫𑀸𑀦𑁄,_  
_𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀼𑀫𑀲𑁂𑀲𑀢𑁄;_  
_𑀊𑀦𑀫𑀦𑀁 𑀧𑀽𑀭𑀬𑀺𑀢𑀼𑀁,_  
_𑀤𑁂𑀫𑀺 𑀤𑀸𑀦𑀁 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁂;_  
_𑀦𑀺𑀭𑀸𑀮𑀬𑁄 𑀅𑀧𑀘𑁆𑀘𑀸𑀲𑁄,_  
_𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀦𑀼𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇_  


---

41\. 𑀫𑀳𑀸𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀘𑀭𑀺𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
