# 𑁧𑁇𑁧𑁇𑁯 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀘𑀭𑀺𑀬

### 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀤𑀸𑀦𑀧𑀸𑀭𑀫𑀻

### 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀘𑀭𑀺𑀬

(67.)

76. _“𑀬𑀸 𑀫𑁂 𑀅𑀳𑁄𑀲𑀺 𑀚𑀦𑀺𑀓𑀸,_  
_𑀨𑀼𑀲𑁆𑀲𑀢𑀻 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀸 𑀅𑀢𑀻𑀢𑀸𑀲𑀼 𑀚𑀸𑀢𑀻𑀲𑀼,_  
_𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀻 𑀧𑀺𑀬𑀸𑁇_  


(68.)

77. _𑀢𑀲𑁆𑀲𑀸 𑀆𑀬𑀼𑀓𑁆𑀔𑀬𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀺;_  
_‘𑀤𑀤𑀸𑀫𑀺 𑀢𑁂 𑀤𑀲 𑀯𑀭𑁂,_  
_𑀯𑀭𑀪𑀤𑁆𑀤𑁂 𑀬𑀤𑀺𑀘𑁆𑀙𑀲𑀺’𑁇_  


(69.)

78. _𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑀸 𑀘 𑀲𑀸 𑀤𑁂𑀯𑀻,_  
_𑀲𑀓𑁆𑀓𑀁 𑀧𑀼𑀦𑀺𑀤𑀫𑀩𑁆𑀭𑀯𑀺;_  
_‘𑀓𑀺𑀁 𑀦𑀼 𑀫𑁂 𑀅𑀧𑀭𑀸𑀥𑀢𑁆𑀣𑀺,_  
_𑀓𑀺𑀁 𑀦𑀼 𑀤𑁂𑀲𑁆𑀲𑀸 𑀅𑀳𑀁 𑀢𑀯;_  
_𑀭𑀫𑁆𑀫𑀸 𑀘𑀸𑀯𑁂𑀲𑀺 𑀫𑀁 𑀞𑀸𑀦𑀸,_  
_𑀯𑀸𑀢𑁄𑀯 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀁’𑁇_  


(70.)

79. _𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁄 𑀘 𑀲𑁄 𑀲𑀓𑁆𑀓𑁄,_  
_𑀧𑀼𑀦 𑀢𑀲𑁆𑀲𑀺𑀤𑀫𑀩𑁆𑀭𑀯𑀺;_  
_‘𑀦 𑀘𑁂𑀯 𑀢𑁂 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁,_  
_𑀦 𑀘 𑀫𑁂 𑀢𑁆𑀯𑀁𑀲𑀺 𑀅𑀧𑁆𑀧𑀺𑀬𑀸𑁇_  


(71.)

80. _𑀏𑀢𑁆𑀢𑀓𑀁𑀬𑁂𑀯 𑀢𑁂 𑀆𑀬𑀼,_  
_𑀘𑀯𑀦𑀓𑀸𑀮𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁂,_  
_𑀯𑀭𑁂 𑀤𑀲 𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂’𑁇_  


(72.)

81. _𑀲𑀓𑁆𑀓𑁂𑀦 𑀲𑀸 𑀤𑀺𑀦𑁆𑀦𑀯𑀭𑀸,_  
_𑀢𑀼𑀝𑁆𑀞𑀳𑀝𑁆𑀞𑀸 𑀧𑀫𑁄𑀤𑀺𑀢𑀸;_  
_𑀫𑀫𑀁 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀨𑀼𑀲𑁆𑀲𑀢𑀻 𑀤𑀲 𑀯𑀭𑁂 𑀯𑀭𑀻𑁇_  


(73.)

82. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀲𑀸 𑀨𑀼𑀲𑁆𑀲𑀢𑀻,_  
_𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀣;_  
_𑀚𑁂𑀢𑀼𑀢𑁆𑀢𑀭𑀫𑁆𑀳𑀺 𑀦𑀕𑀭𑁂,_  
_𑀲𑀜𑁆𑀚𑀬𑁂𑀦 𑀲𑀫𑀸𑀕𑀫𑀺𑁇_  


(74.)

83. _𑀬𑀤𑀸𑀳𑀁 𑀨𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀓𑀼𑀘𑁆𑀙𑀺𑀁,_  
_𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀧𑀺𑀬𑀫𑀸𑀢𑀼𑀬𑀸;_  
_𑀫𑀫 𑀢𑁂𑀚𑁂𑀦 𑀫𑁂 𑀫𑀸𑀢𑀸,_  
_𑀲𑀤𑀸 𑀤𑀸𑀦𑀭𑀢𑀸 𑀅𑀳𑀼𑁇_  


(75.)

84. _𑀅𑀥𑀦𑁂 𑀆𑀢𑀼𑀭𑁂 𑀚𑀺𑀡𑁆𑀡𑁂,_  
_𑀬𑀸𑀘𑀓𑁂 𑀅𑀤𑁆𑀥𑀺𑀓𑁂 𑀚𑀦𑁂;_  
_𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀔𑀻𑀡𑁂,_  
_𑀤𑁂𑀢𑀺 𑀤𑀸𑀦𑀁 𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑁂𑁇_  


(76.)

85. _𑀤𑀲 𑀫𑀸𑀲𑁂 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀭𑁄𑀦𑁆𑀢𑁂 𑀧𑀼𑀭𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀯𑁂𑀲𑁆𑀲𑀸𑀦𑀁 𑀯𑀻𑀣𑀺𑀬𑀸 𑀫𑀚𑁆𑀛𑁂,_  
_𑀚𑀦𑁂𑀲𑀺 𑀨𑀼𑀲𑁆𑀲𑀢𑀻 𑀫𑀫𑀁𑁇_  


(77.)

86. _𑀦 𑀫𑀬𑁆𑀳𑀁 𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀦𑀸𑀫𑀁,_  
_𑀦𑀧𑀺 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀲𑀫𑁆𑀪𑀯𑀁;_  
_𑀚𑀸𑀢𑁂𑀢𑁆𑀣 𑀯𑁂𑀲𑁆𑀲𑀯𑀻𑀣𑀺𑀬𑀸,_  
_𑀢𑀲𑁆𑀫𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀅𑀳𑀼𑁇_  


(78.)

87. _𑀬𑀤𑀸𑀳𑀁 𑀤𑀸𑀭𑀓𑁄 𑀳𑁄𑀫𑀺,_  
_𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀝𑁆𑀞𑀯𑀲𑁆𑀲𑀺𑀓𑁄;_  
_𑀢𑀤𑀸 𑀦𑀺𑀲𑀚𑁆𑀚 𑀧𑀸𑀲𑀸𑀤𑁂,_  
_𑀤𑀸𑀦𑀁 𑀤𑀸𑀢𑀼𑀁 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀁𑁇_  


(79.)

88. _‘𑀳𑀤𑀬𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑀁 𑀘𑀓𑁆𑀔𑀼𑀁,_  
_𑀫𑀁𑀲𑀫𑁆𑀧𑀺 𑀭𑀼𑀥𑀺𑀭𑀫𑁆𑀧𑀺 𑀘;_  
_𑀤𑀤𑁂𑀬𑁆𑀬𑀁 𑀓𑀸𑀬𑀁 𑀲𑀸𑀯𑁂𑀢𑁆𑀯𑀸,_  
_𑀬𑀤𑀺 𑀓𑁄𑀘𑀺 𑀬𑀸𑀘𑀬𑁂 𑀫𑀫𑀁’𑁇_  


(80.)

89. _𑀲𑀪𑀸𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀅𑀓𑀫𑁆𑀧𑀺𑀢𑀫𑀲𑀡𑁆𑀞𑀺𑀢𑀁;_  
_𑀅𑀓𑀫𑁆𑀧𑀺 𑀢𑀢𑁆𑀣 𑀧𑀣𑀯𑀻,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀯𑀦𑀯𑀝𑀁𑀲𑀓𑀸𑁇_  


(81.)

90. _𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂,_  
_𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀲𑁂 𑀉𑀧𑁄𑀲𑀣𑁂;_  
_𑀧𑀘𑁆𑀘𑀬𑀁 𑀦𑀸𑀕𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀤𑀸𑀦𑀁 𑀤𑀸𑀢𑀼𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(82.)

91. _𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀝𑁆𑀞𑀯𑀺𑀲𑀬𑀸,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀧𑀕𑀜𑁆𑀙𑀼 𑀫𑀁;_  
_𑀅𑀬𑀸𑀘𑀼𑀁 𑀫𑀁 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀁,_  
_𑀥𑀜𑁆𑀜𑀁 𑀫𑀗𑁆𑀕𑀮𑀲𑀫𑁆𑀫𑀢𑀁𑁇_  


(83.)

92. _‘𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑁄 𑀚𑀦𑀧𑀤𑁄,_  
_𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁄 𑀙𑀸𑀢𑀓𑁄 𑀫𑀳𑀸;_  
_𑀤𑀤𑀸𑀳𑀺 𑀧𑀯𑀭𑀁 𑀦𑀸𑀕𑀁,_  
_𑀲𑀩𑁆𑀩𑀲𑁂𑀢𑀁 𑀕𑀚𑀼𑀢𑁆𑀢𑀫𑀁’𑁇_  


(84.)

93. _𑀤𑀤𑀸𑀫𑀺 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑀸𑀫𑀺,_  
_𑀬𑀁 𑀫𑀁 𑀬𑀸𑀘𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀲𑀦𑁆𑀢𑀁 𑀦𑀧𑁆𑀧𑀢𑀺𑀕𑀽𑀳𑀸𑀫𑀺,_  
_𑀤𑀸𑀦𑁂 𑀫𑁂 𑀭𑀫𑀢𑁂 𑀫𑀦𑁄𑁇_  


(85.)

94. _𑀦 𑀫𑁂 𑀬𑀸𑀘𑀓𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀦𑀼𑀘𑁆𑀙𑀯𑁄;_  
_𑀫𑀸 𑀫𑁂 𑀪𑀺𑀚𑁆𑀚𑀺 𑀲𑀫𑀸𑀤𑀸𑀦𑀁,_  
_𑀤𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀺𑀧𑀼𑀮𑀁 𑀕𑀚𑀁𑁇_  


(86.)

95. _𑀦𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀸𑀬,_  
_𑀪𑀺𑀗𑁆𑀕𑀸𑀭𑁂 𑀭𑀢𑀦𑀸𑀫𑀬𑁂;_  
_𑀚𑀮𑀁 𑀳𑀢𑁆𑀣𑁂 𑀆𑀓𑀺𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀅𑀤𑀁 𑀕𑀚𑀁𑁇_  


(87.)

96. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀤𑀤𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑀲𑁂𑀢𑀁 𑀕𑀚𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀢𑀤𑀸𑀧𑀺 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀺,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀯𑀦𑀯𑀝𑀁𑀲𑀓𑀸𑁇_  


(88.)

97. _𑀢𑀲𑁆𑀲 𑀦𑀸𑀕𑀲𑁆𑀲 𑀤𑀸𑀦𑁂𑀦,_  
_𑀲𑀺𑀯𑀬𑁄 𑀓𑀼𑀤𑁆𑀥𑀸 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀼𑀁 𑀲𑀓𑀸 𑀭𑀝𑁆𑀞𑀸,_  
_‘𑀯𑀗𑁆𑀓𑀁 𑀕𑀘𑁆𑀙𑀢𑀼 𑀧𑀩𑁆𑀩𑀢𑀁’𑁇_  


(89.)

98. _𑀢𑁂𑀲𑀁 𑀦𑀺𑀘𑁆𑀙𑀼𑀪𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀅𑀓𑀫𑁆𑀧𑀺𑀢𑀫𑀲𑀡𑁆𑀞𑀺𑀢𑀁;_  
_𑀫𑀳𑀸𑀤𑀸𑀦𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀼𑀁,_  
_𑀏𑀓𑀁 𑀯𑀭𑀫𑀬𑀸𑀘𑀺𑀲𑀁𑁇_  


(90.)

99. _𑀬𑀸𑀘𑀺𑀢𑀸 𑀲𑀺𑀯𑀬𑁄 𑀲𑀩𑁆𑀩𑁂,_  
_𑀏𑀓𑀁 𑀯𑀭𑀫𑀤𑀁𑀲𑀼 𑀫𑁂;_  
_𑀲𑀸𑀯𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀡𑁆𑀡𑀪𑁂𑀭𑀺𑀁,_  
_𑀫𑀳𑀸𑀤𑀸𑀦𑀁 𑀤𑀤𑀸𑀫𑀳𑀁𑁇_  


(91.)

100. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀯𑀢𑁆𑀢𑀢𑀻 𑀲𑀤𑁆𑀤𑁄,_  
_𑀢𑀼𑀫𑀼𑀮𑁄 𑀪𑁂𑀭𑀯𑁄 𑀫𑀳𑀸;_  
_𑀤𑀸𑀦𑁂𑀦𑀺𑀫𑀁 𑀦𑀻𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀦 𑀤𑀸𑀦𑀁 𑀤𑀤𑀸𑀢𑀬𑀁𑁇_  


(92.)

101. _𑀳𑀢𑁆𑀣𑀺𑀁 𑀅𑀲𑁆𑀲𑁂 𑀭𑀣𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀤𑀸𑀲𑀺𑀁 𑀤𑀸𑀲𑀁 𑀕𑀯𑀁 𑀥𑀦𑀁;_  
_𑀫𑀳𑀸𑀤𑀸𑀦𑀁 𑀤𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀕𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀁 𑀢𑀤𑀸𑁇_  


(93.)

102. _𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦 𑀦𑀕𑀭𑀸,_  
_𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂;_  
_𑀢𑀤𑀸𑀧𑀺 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀺,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀯𑀦𑀯𑀝𑀁𑀲𑀓𑀸𑁇_  


(94.)

103. _𑀘𑀢𑀼𑀯𑀸𑀳𑀺𑀁 𑀭𑀣𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀞𑀢𑁆𑀯𑀸 𑀘𑀸𑀢𑀼𑀫𑁆𑀫𑀳𑀸𑀧𑀣𑁂;_  
_𑀏𑀓𑀸𑀓𑀺𑀬𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄,_  
_𑀫𑀤𑁆𑀤𑀺𑀤𑁂𑀯𑀺𑀁 𑀇𑀤𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(95.)

104. _‘𑀢𑁆𑀯𑀁 𑀫𑀤𑁆𑀤𑀺 𑀓𑀡𑁆𑀳𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺,_  
_𑀮𑀳𑀼𑀓𑀸 𑀏𑀲𑀸 𑀓𑀦𑀺𑀝𑁆𑀞𑀺𑀓𑀸;_  
_𑀅𑀳𑀁 𑀚𑀸𑀮𑀺𑀁 𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀢𑀺𑀓𑁄 𑀳𑀺 𑀲𑁄’𑁇_  


(96.)

105. _𑀧𑀤𑀼𑀫𑀁 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀁𑀯,_  
_𑀫𑀤𑁆𑀤𑀻 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀕𑁆𑀕𑀳𑀻;_  
_𑀅𑀳𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑀩𑀺𑀫𑁆𑀩𑀁𑀯,_  
_𑀚𑀸𑀮𑀺𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀕𑁆𑀕𑀳𑀺𑀁𑁇_  


(97.)

106. _𑀅𑀪𑀺𑀚𑀸𑀢𑀸 𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑀸,_  
_𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑀢𑀼𑀭𑁄 𑀚𑀦𑀸;_  
_𑀯𑀺𑀲𑀫𑀁 𑀲𑀫𑀁 𑀅𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀸,_  
_𑀯𑀗𑁆𑀓𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑀁𑁇_  


(98.)

107. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀦𑀼𑀚𑀸 𑀏𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑀼𑀫𑀕𑁆𑀕𑁂 𑀧𑀝𑀺𑀧𑁆𑀧𑀣𑁂;_  
_𑀫𑀕𑁆𑀕𑀦𑁆𑀢𑁂 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_‘𑀓𑀼𑀳𑀺𑀁 𑀯𑀗𑁆𑀓𑀦𑁆𑀢𑀧𑀩𑁆𑀩𑀢𑁄’𑁇_  


(99.)

108. _𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀫𑁆𑀳𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀭𑀼𑀡𑀁 𑀕𑀺𑀭𑀫𑀼𑀤𑀻𑀭𑀬𑀼𑀁;_  
_𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀺,_  
_𑀤𑀽𑀭𑁂 𑀯𑀗𑁆𑀓𑀦𑁆𑀢𑀧𑀩𑁆𑀩𑀢𑁄𑁇_  


(100.)

109. _𑀬𑀤𑀺 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀯𑀦𑁂,_  
_𑀤𑀸𑀭𑀓𑀸 𑀨𑀮𑀺𑀦𑁂 𑀤𑀼𑀫𑁂;_  
_𑀢𑁂𑀲𑀁 𑀨𑀮𑀸𑀦𑀁 𑀳𑁂𑀢𑀼𑀫𑁆𑀳𑀺,_  
_𑀉𑀧𑀭𑁄𑀤𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(101.)

110. _𑀭𑁄𑀤𑀦𑁆𑀢𑁂 𑀤𑀸𑀭𑀓𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀉𑀩𑁆𑀩𑀺𑀤𑁆𑀥𑀸 𑀯𑀺𑀧𑀼𑀮𑀸 𑀤𑀼𑀫𑀸;_  
_𑀲𑀬𑀫𑁂𑀯𑁄𑀡𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑁂𑁇_  


(102.)

111. _𑀇𑀤𑀁 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀲𑀸𑀳𑀼𑀓𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀲𑀺,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸𑁇_  


(103.)

112. _𑀅𑀘𑁆𑀙𑁂𑀭𑀁 𑀯𑀢 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀲𑁆𑀲 𑀢𑁂𑀚𑁂𑀦,_  
_𑀲𑀬𑀫𑁂𑀯𑁄𑀡𑀢𑀸 𑀤𑀼𑀫𑀸𑁇_  


(104.)

113. _𑀲𑀗𑁆𑀔𑀺𑀧𑀺𑀁𑀲𑀼 𑀧𑀣𑀁 𑀬𑀓𑁆𑀔𑀸,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀤𑀸𑀭𑀓𑁂;_  
_𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀤𑀺𑀯𑀲𑁂𑀦𑁂𑀯,_  
_𑀘𑁂𑀢𑀭𑀝𑁆𑀞𑀫𑀼𑀧𑀸𑀕𑀫𑀼𑀁𑁇_  


(105.)

114. _𑀲𑀝𑁆𑀞𑀺𑀭𑀸𑀚𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀤𑀸 𑀯𑀲𑀦𑁆𑀢𑀺 𑀫𑀸𑀢𑀼𑀮𑁂;_  
_𑀲𑀩𑁆𑀩𑁂 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀭𑁄𑀤𑀫𑀸𑀦𑀸 𑀉𑀧𑀸𑀕𑀫𑀼𑀁𑁇_  


(106.)

115. _𑀢𑀢𑁆𑀣 𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀲𑀮𑁆𑀮𑀸𑀧𑀁,_  
_𑀘𑁂𑀢𑁂𑀳𑀺 𑀘𑁂𑀢𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺;_  
_𑀢𑁂 𑀢𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀗𑁆𑀓𑀁 𑀅𑀕𑀫𑀼 𑀧𑀩𑁆𑀩𑀢𑀁𑁇_  


(107.)

116. _𑀆𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀁;_  
_𑀅𑀲𑁆𑀲𑀫𑀁 𑀲𑀼𑀓𑀢𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀁 𑀲𑀼𑀫𑀸𑀧𑀬𑁇_  


(108.)

117. _𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁄 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄;_  
_𑀅𑀲𑁆𑀲𑀫𑀁 𑀲𑀼𑀓𑀢𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀁 𑀲𑀼𑀫𑀸𑀧𑀬𑀺𑁇_  


(109.)

118. _𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀯𑀦𑀁,_  
_𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀦𑀺𑀭𑀸𑀓𑀼𑀮𑀁;_  
_𑀘𑀢𑀼𑀭𑁄 𑀚𑀦𑀸 𑀫𑀬𑀁 𑀢𑀢𑁆𑀣,_  
_𑀯𑀲𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂𑁇_  


(110.)

119. _𑀅𑀳𑀜𑁆𑀘 𑀫𑀤𑁆𑀤𑀺𑀤𑁂𑀯𑀻 𑀘,_  
_𑀚𑀸𑀮𑀻 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀸 𑀘𑀼𑀪𑁄;_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑁄𑀓𑀦𑀼𑀤𑀸,_  
_𑀯𑀲𑀸𑀫 𑀅𑀲𑁆𑀲𑀫𑁂 𑀢𑀤𑀸𑁇_  


(111.)

120. _𑀤𑀸𑀭𑀓𑁂 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄,_  
_𑀅𑀲𑀼𑀜𑁆𑀜𑁄 𑀳𑁄𑀫𑀺 𑀅𑀲𑁆𑀲𑀫𑁂;_  
_𑀫𑀤𑁆𑀤𑀻 𑀨𑀮𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑁄𑀲𑁂𑀢𑀺 𑀲𑀸 𑀢𑀬𑁄 𑀚𑀦𑁂𑁇_  


(112.)

121. _𑀧𑀯𑀦𑁂 𑀯𑀲𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀤𑁆𑀥𑀺𑀓𑁄 𑀫𑀁 𑀉𑀧𑀸𑀕𑀫𑀺;_  
_𑀆𑀬𑀸𑀘𑀺 𑀧𑀼𑀢𑁆𑀢𑀓𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀚𑀸𑀮𑀺𑀁 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀁 𑀘𑀼𑀪𑁄𑁇_  


(113.)

122. _𑀬𑀸𑀘𑀓𑀁 𑀉𑀧𑀕𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀸𑀲𑁄 𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀣;_  
_𑀉𑀪𑁄 𑀧𑀼𑀢𑁆𑀢𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀢𑀤𑀸𑁇_  


(114.)

123. _𑀲𑀓𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀘𑀚𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀚𑀽𑀚𑀓𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀬𑀤𑀸;_  
_𑀢𑀤𑀸𑀧𑀺 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀺,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀯𑀦𑀯𑀝𑀁𑀲𑀓𑀸𑁇_  


(115.)

124. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀓𑁆𑀓𑁄 𑀑𑀭𑀼𑀬𑁆𑀳,_  
_𑀳𑀼𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀦𑁆𑀦𑀺𑀪𑁄;_  
_𑀆𑀬𑀸𑀘𑀺 𑀫𑀁 𑀫𑀤𑁆𑀤𑀺𑀤𑁂𑀯𑀺𑀁,_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀺𑀁 𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀁𑁇_  


(116.)

125. _𑀫𑀤𑁆𑀤𑀺𑀁 𑀳𑀢𑁆𑀣𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀤𑀓𑀜𑁆𑀚𑀮𑀺 𑀧𑀽𑀭𑀺𑀬;_  
_𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀢𑀲𑁆𑀲 𑀫𑀤𑁆𑀤𑀺𑀁 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(117.)

126. _𑀫𑀤𑁆𑀤𑀺𑀬𑀸 𑀤𑀻𑀬𑀫𑀸𑀦𑀸𑀬,_  
_𑀕𑀕𑀦𑁂 𑀤𑁂𑀯𑀸 𑀧𑀫𑁄𑀤𑀺𑀢𑀸;_  
_𑀢𑀤𑀸𑀧𑀺 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀺,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀯𑀦𑀯𑀝𑀁𑀲𑀓𑀸𑁇_  


(118.)

127. _𑀚𑀸𑀮𑀺𑀁 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀁 𑀥𑀻𑀢𑀁,_  
_𑀫𑀤𑁆𑀤𑀺𑀤𑁂𑀯𑀺𑀁 𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀁;_  
_𑀘𑀚𑀫𑀸𑀦𑁄 𑀦 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀺𑀁,_  
_𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀓𑀸𑀭𑀡𑀸𑁇_  


(119.)

128. _𑀦 𑀫𑁂 𑀤𑁂𑀲𑁆𑀲𑀸 𑀉𑀪𑁄 𑀧𑀼𑀢𑁆𑀢𑀸,_  
_𑀫𑀤𑁆𑀤𑀺𑀤𑁂𑀯𑀻 𑀦 𑀤𑁂𑀲𑁆𑀲𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀺𑀬𑁂 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(120.)

129. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀼𑀲𑀫𑀸𑀕𑀫𑁂;_  
_𑀓𑀭𑀼𑀡𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑁆𑀢𑁂,_  
_𑀲𑀮𑁆𑀮𑀧𑀦𑁆𑀢𑁂 𑀲𑀼𑀔𑀁 𑀤𑀼𑀔𑀁𑁇_  


(121.)

130. _𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑁂𑀦 𑀕𑀭𑀼𑀦𑀸,_  
_𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺;_  
_𑀢𑀤𑀸𑀧𑀺 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀺,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀯𑀦𑀯𑀝𑀁𑀲𑀓𑀸𑁇_  


(122.)

131. _𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑀸,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀜𑀸𑀢𑀺𑀪𑀺;_  
_𑀧𑀯𑀺𑀲𑀸𑀫𑀺 𑀧𑀼𑀭𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀚𑁂𑀢𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(123.)

132. _𑀭𑀢𑀦𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀺𑀁𑀲𑀼,_  
_𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀧𑀯𑀲𑁆𑀲𑀣;_  
_𑀢𑀤𑀸𑀧𑀺 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀺,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀯𑀦𑀯𑀝𑀁𑀲𑀓𑀸𑁇_  


(124.)

133. _𑀅𑀘𑁂𑀢𑀦𑀸𑀬𑀁 𑀧𑀣𑀯𑀻,_  
_𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀔𑀁 𑀤𑀼𑀔𑀁;_  
_𑀲𑀸𑀧𑀺 𑀤𑀸𑀦𑀩𑀮𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀓𑀫𑁆𑀧𑀣𑀸”𑀢𑀺𑁇_  


---

134. 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀘𑀭𑀺𑀬𑀁 𑀦𑀯𑀫𑀁𑁇
