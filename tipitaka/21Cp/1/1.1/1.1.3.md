# 𑁧𑁇𑁧𑁇𑁩 𑀓𑀼𑀭𑀼𑀭𑀸𑀚𑀘𑀭𑀺𑀬

### 𑀅𑀓𑀺𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀤𑀸𑀦𑀧𑀸𑀭𑀫𑀻

### 𑀓𑀼𑀭𑀼𑀭𑀸𑀚𑀘𑀭𑀺𑀬

(20.)

23\. _“𑀧𑀼𑀦𑀸𑀧𑀭𑀁 𑀬𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀇𑀦𑁆𑀤𑀧𑀢𑁆𑀣𑁂 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀭𑀸𑀚𑀸 𑀥𑀦𑀜𑁆𑀘𑀬𑁄 𑀦𑀸𑀫,_  
_𑀓𑀼𑀲𑀮𑁂 𑀤𑀲𑀳𑀼𑀧𑀸𑀕𑀢𑁄𑁇_  


(21.)

24\. _𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀝𑁆𑀞𑀯𑀺𑀲𑀬𑀸,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀧𑀕𑀜𑁆𑀙𑀼 𑀫𑀁;_  
_𑀆𑀬𑀸𑀘𑀼𑀁 𑀫𑀁 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀁,_  
_𑀥𑀜𑁆𑀜𑀁 𑀫𑀗𑁆𑀕𑀮𑀲𑀫𑁆𑀫𑀢𑀁𑁇_  


(22.)

25\. _‘𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑁄 𑀚𑀦𑀧𑀤𑁄,_  
_𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁄 𑀙𑀸𑀢𑀓𑁄 𑀫𑀳𑀸;_  
_𑀤𑀤𑀸𑀳𑀺 𑀧𑀯𑀭𑀁 𑀦𑀸𑀕𑀁,_  
_𑀦𑀻𑀮𑀁 𑀅𑀜𑁆𑀚𑀦𑀲𑀯𑁆𑀳𑀬𑀁’𑁇_  


(23.)

26\. _‘𑀦 𑀫𑁂 𑀬𑀸𑀘𑀓𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀦𑀼𑀘𑁆𑀙𑀯𑁄;_  
_𑀫𑀸 𑀫𑁂 𑀪𑀺𑀚𑁆𑀚𑀺 𑀲𑀫𑀸𑀤𑀸𑀦𑀁,_  
_𑀤𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀺𑀧𑀼𑀮𑀁 𑀕𑀚𑀁’𑁇_  


(24.)

27\. _𑀦𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀸𑀬,_  
_𑀪𑀺𑀗𑁆𑀕𑀸𑀭𑁂 𑀭𑀢𑀦𑀸𑀫𑀬𑁂;_  
_𑀚𑀮𑀁 𑀳𑀢𑁆𑀣𑁂 𑀆𑀓𑀺𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀅𑀤𑀁 𑀕𑀚𑀁𑁇_  


(25.)

28\. _𑀢𑀲𑁆𑀲 𑀦𑀸𑀕𑁂 𑀧𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀳𑀺,_  
_𑀅𑀫𑀘𑁆𑀘𑀸 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀼𑀁;_  
_‘𑀓𑀺𑀁 𑀦𑀼 𑀢𑀼𑀬𑁆𑀳𑀁 𑀯𑀭𑀁 𑀦𑀸𑀕𑀁,_  
_𑀬𑀸𑀘𑀓𑀸𑀦𑀁 𑀧𑀤𑀲𑁆𑀲𑀲𑀺𑁇_  


(26.)

29\. _𑀥𑀜𑁆𑀜𑀁 𑀫𑀗𑁆𑀕𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀗𑁆𑀕𑀸𑀫𑀯𑀺𑀚𑀬𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀸𑀕𑁂 𑀧𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀳𑀺,_  
_𑀓𑀺𑀁 𑀢𑁂 𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(27.)

30\. _‘𑀭𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀫𑁂 𑀤𑀤𑁂 𑀲𑀩𑁆𑀩𑀁,_  
_𑀲𑀭𑀻𑀭𑀁 𑀤𑀚𑁆𑀚𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀺𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀁 𑀅𑀤𑀸𑀲𑀳𑀦𑁆’”𑀢𑀺𑁇_  


---

31\. 𑀓𑀼𑀭𑀼𑀭𑀸𑀚𑀘𑀭𑀺𑀬𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
