# 𑁨𑁇𑁬𑁇𑁫 𑀢𑀺𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀙𑀝𑁆𑀞𑀦𑀬 𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑁄𑀕𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀢𑀺𑀓

(257.)

334\. 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁧)

(258.)

335\. 𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀦𑁆𑀦𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(259.)

336\. 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁨)

(260.)

337\. 𑀯𑀺𑀧𑀸𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(261.)

338\. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸…  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(262.)

339\. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁩)

(263.)

340\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀙𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁪)

(264.)

341\. 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀦𑁆𑀦𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(265.)

342\. 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(266.)

343\. 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁬)

(267.)

344\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀦𑁆𑀦𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(268.)

345\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁭)

(269.)

346\. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸…  𑀲𑁂𑀓𑁆𑀔𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀳𑀕𑁆𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(270.)

347\. 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀡𑀻𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀙𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(271.)

348\. 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(272.)

349\. 𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀳𑀻𑀦𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁮--𑁧𑁬)

(273.)

350\. 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁧𑁭--𑁧𑁮)

(274.)

351\. 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(275.)

352\. 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁧𑁯--𑁨𑁧)

(276.)

353\. 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀸 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇 (𑁨𑁨)
