# 𑁨𑁇𑁧𑁇𑁧 𑀔𑀦𑁆𑀥

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀞𑀫𑀦𑀬 𑀲𑀗𑁆𑀕𑀳𑀸𑀲𑀗𑁆𑀕𑀳𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀔𑀦𑁆𑀥

(6.)

35. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 (𑁧)

(7.)

36. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 (𑁨)

(8.)

37. 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 (𑁩)

(9.)

38. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 (𑁪)

(10.)

39. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁄𑁇 (𑁫)

---

40. (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)



(11.)

41. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 (𑁧)

(12.)

42. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 (𑁨)

(13.)

43. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 (𑁩)

(14.)

44. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘…𑀧𑁂…  𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺, 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 (𑁪)

---

45. (𑀤𑀼𑀓𑀫𑀽𑀮𑀓𑀁𑁇)



(15.)

46. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 (𑁧)

(16.)

47. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘…𑀧𑁂…  𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 (𑁨)

(17.)

48. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘…𑀧𑁂…  𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺, 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 (𑁩)

---

49. (𑀢𑀺𑀓𑀫𑀽𑀮𑀓𑀁𑁇)



(18.)

50. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(19.)

51. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘…𑀧𑁂…  𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦, 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

---

52. (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)



(20.)

53. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(21.)

54. 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

---
