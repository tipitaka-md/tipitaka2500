# 𑁨𑁇𑁧𑁇𑁮𑁇𑁧𑁦 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀞𑀫𑀦𑀬 𑀲𑀗𑁆𑀕𑀳𑀸𑀲𑀗𑁆𑀕𑀳𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀤𑀼𑀓

* 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓

(135.)

181\. 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(136.)

182\. 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(137.)

183\. 𑀘𑀺𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(138.)

184\. 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(139.)

185\. 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀥𑀫𑁆𑀫𑀸 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(140.)

186\. 𑀅𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(141.)

187\. 𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(142.)

188\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀙𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀙𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀙𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(143.)

189\. 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑀸…  𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(144.)

190\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(145.)

191\. 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(146.)

192\. 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑀸…  𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(147.)

193\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀙𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀙𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀙𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(148.)

194\. 𑀩𑀸𑀳𑀺𑀭𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀙𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀙𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀙𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(149.)

195\. 𑀉𑀧𑀸𑀤𑀸 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(150.)

196\. 𑀦𑁄 𑀉𑀧𑀸𑀤𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀢𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀦𑀯𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦𑀯𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀦𑀯𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(151.)

197\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(152.)

198\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇
