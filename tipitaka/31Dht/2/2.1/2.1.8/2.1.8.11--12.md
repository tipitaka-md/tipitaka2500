# 𑁨𑁇𑁧𑁇𑁮𑁇𑁧𑁧--𑁧𑁨 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀞𑀫𑀦𑀬 𑀲𑀗𑁆𑀕𑀳𑀸𑀲𑀗𑁆𑀕𑀳𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀤𑀼𑀓

### 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

(153.)

199. 𑀉𑀧𑀸𑀤𑀸𑀦𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑁂𑀯 𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(154.)

200. 𑀦𑁄 𑀓𑀺𑀮𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(155.)

201. 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(156.)

202. 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(157.)

203. 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇
