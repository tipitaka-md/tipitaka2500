# 𑁨𑁇𑁧𑁇𑁭𑁇𑁧𑁮 𑀅𑀢𑀻𑀢𑀢𑁆𑀢𑀺𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀞𑀫𑀦𑀬 𑀲𑀗𑁆𑀕𑀳𑀸𑀲𑀗𑁆𑀕𑀳𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀢𑀺𑀓

* 𑀅𑀢𑀻𑀢𑀢𑁆𑀢𑀺𑀓

(104.)

150\. 𑀅𑀢𑀻𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀦𑀸𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(105.)

151\. 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀦 𑀓𑀸𑀳𑀺𑀘𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇
