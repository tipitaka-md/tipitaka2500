# 𑁨𑁇𑁧𑁇𑁭𑁇𑁩 𑀯𑀺𑀧𑀸𑀓𑀢𑁆𑀢𑀺𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀞𑀫𑀦𑀬 𑀲𑀗𑁆𑀕𑀳𑀸𑀲𑀗𑁆𑀕𑀳𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀢𑀺𑀓

### 𑀯𑀺𑀧𑀸𑀓𑀢𑁆𑀢𑀺𑀓

(81.)

127\. 𑀯𑀺𑀧𑀸𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(82.)

128\. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸…  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(83.)

129\. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀢𑁂𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇
