# 𑁨𑁇𑁧𑁩𑁇𑁨𑁇𑁧 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀢𑁂𑀭𑀲𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀤𑀼𑀓

* 𑀳𑁂𑀢𑀼𑀕𑁄𑀘𑁆𑀙𑀓

569\. 𑀲𑀳𑁂𑀢𑀼𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀳𑁂𑀢𑀼𑀓𑁂𑀳𑀺 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀺𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
