# 𑁨𑁇𑁫 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(193.)

250\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(194.)

251\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(195.)

252\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(196.)

253\. 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(197.)

254\. 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀇𑀢𑁆𑀣𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀼𑀭𑀺𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(198.)

255\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(199.)

256\. 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(200.)

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

264\. 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(208.)

265\. 𑀲𑁄𑀓𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀛𑀸𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀜𑁆𑀘𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀜𑁆𑀘𑀳𑀺 𑀩𑀮𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀢𑁆𑀢𑀳𑀺 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀭𑀺𑀬𑁂𑀦 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀫𑀕𑁆𑀕𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀨𑀲𑁆𑀲𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀯𑁂𑀤𑀦𑀸𑀬 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀜𑁆𑀜𑀸𑀬 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑁂𑀢𑀦𑀸𑀬 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(209.)

266\. 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇
