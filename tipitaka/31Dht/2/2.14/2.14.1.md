# 𑁨𑁇𑁧𑁪𑁇𑁧 𑀔𑀦𑁆𑀥𑀸𑀤𑀺

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀘𑀼𑀤𑁆𑀤𑀲𑀫𑀦𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀔𑀦𑁆𑀥𑀸𑀤𑀺

(456.)

577. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(457.)

578. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(458.)

579. 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀓𑁆𑀔𑀼𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(459.)

580. 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀥𑀸𑀢𑀼𑀬𑀸 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀔𑀦𑁆𑀥𑀢𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀤𑁆𑀯𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇 𑀓𑀢𑀺𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺, 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇
