# 𑁨𑁇𑁫𑁇𑁨𑁇𑁪--𑁧𑁦 𑀲𑀁𑀬𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀤𑀼𑀓

### 𑀲𑀁𑀬𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

(220.)

294\. 𑀲𑀁𑀬𑁄𑀚𑀦𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀕𑀦𑁆𑀣𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀑𑀖𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀬𑁄𑀕𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦𑀻𑀯𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀭𑀸𑀫𑀸𑀲𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀭𑀸𑀫𑀸𑀲𑁂𑀳𑀺 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀘 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(221.)

295\. 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀭𑀸𑀫𑀸𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(222.)

296\. 𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀩𑀸𑀳𑀺𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀉𑀧𑀸𑀤𑀸𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀲𑀢𑁆𑀢𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(223.)

297\. 𑀘𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀘𑀢𑀽𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(224.)

298\. 𑀘𑁂𑀢𑀲𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀳𑀪𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀘𑀺𑀢𑁆𑀢𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(225.)

299\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇
