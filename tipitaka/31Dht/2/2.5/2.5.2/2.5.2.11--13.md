# 𑁨𑁇𑁫𑁇𑁨𑁇𑁧𑁧--𑁧𑁩 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀜𑁆𑀘𑀫𑀦𑀬 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀤𑀼𑀓

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

(226.)

300\. 𑀉𑀧𑀸𑀤𑀸𑀦𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁂𑀳𑀺 𑀘 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂𑀳𑀺 𑀘 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀘𑁂𑀯 𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀘 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

(227.)

301\. 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂𑀳𑀺 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀯𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦𑀺𑀬𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀆𑀬𑀢𑀦𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸 𑀥𑀸𑀢𑀼𑀲𑀗𑁆𑀕𑀳𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸𑁇

302\. _𑀭𑀽𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼,_  
_𑀇𑀢𑁆𑀣𑀺𑀧𑀼𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁;_  
_𑀤𑁆𑀯𑁂 𑀪𑀯𑀸 𑀚𑀸𑀢𑀺 𑀚𑀭𑀸 𑀫𑀘𑁆𑀘𑀼𑀭𑀽𑀧𑀁,_  
_𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁𑁇_  


303\. _𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦-_  
_𑀲𑀳𑀪𑀼 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺;_  
_𑀩𑀸𑀳𑀺𑀭𑀁 𑀉𑀧𑀸𑀤𑀸 𑀤𑁆𑀯𑁂,_  
_𑀯𑀺𑀲𑀬𑁄 𑀏𑀲𑀦𑀬𑁄 𑀲𑀼𑀩𑀼𑀤𑁆𑀥𑁄𑁇 [𑁨𑁨]_  
_( )_
