# 𑁨𑁇𑁯𑁇𑁨𑁇𑁪--𑁯 𑀲𑀁𑀬𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀯𑀫𑀦𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀤𑀼𑀓

### 𑀲𑀁𑀬𑁄𑀚𑀦𑀕𑁄𑀘𑁆𑀙𑀓𑀸𑀤𑀺

(346.)

432\. 𑀲𑀁𑀬𑁄𑀚𑀦𑁂𑀳𑀺…  𑀕𑀦𑁆𑀣𑁂𑀳𑀺…  𑀑𑀖𑁂𑀳𑀺…  𑀬𑁄𑀕𑁂𑀳𑀺…  𑀦𑀻𑀯𑀭𑀡𑁂𑀳𑀺…  𑀧𑀭𑀸𑀫𑀸𑀲𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀭𑀸𑀫𑀸𑀲𑁂𑀳𑀺 𑀘𑁂𑀯 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀘 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀢𑀻𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(347.)

433\. 𑀧𑀭𑀸𑀫𑀸𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
