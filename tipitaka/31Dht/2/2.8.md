# 𑁨𑁇𑁮 𑀅𑀝𑁆𑀞𑀫𑀦𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀝𑁆𑀞𑀫𑀦𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

(317.)

400\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

(318.)

401\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…𑀧𑁂…  𑀲𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

402\. _𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼,_  
_𑀅𑀣 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁;_  
_𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀢𑀁,_  
_𑀤𑁆𑀯𑁂 𑀘 𑀢𑀺𑀓𑁂 𑀦 𑀮𑀩𑁆𑀪𑀭𑁂𑁇_  


403\. _𑀧𑀞𑀫𑀦𑁆𑀢𑀭𑁂 𑀲𑀢𑁆𑀢 𑀘,_  
_𑀕𑁄𑀘𑁆𑀙𑀓𑁂 𑀤𑀲 𑀅𑀧𑀭𑀦𑁆𑀢𑁂;_  
_𑀘𑀼𑀤𑁆𑀤𑀲 𑀙 𑀘 𑀫𑀢𑁆𑀣𑀓𑁂,_  
_𑀇𑀘𑁆𑀘𑁂𑀢𑁂 𑀲𑀢𑁆𑀢𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲 𑀥𑀫𑁆𑀫𑀸;_  
_𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑁂 𑀦 𑀮𑀩𑁆𑀪𑀦𑁆𑀢𑀺,_  
_𑀫𑁄𑀖𑀧𑀼𑀘𑁆𑀙𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇 [𑁪𑁭]_  


---

404\. 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
