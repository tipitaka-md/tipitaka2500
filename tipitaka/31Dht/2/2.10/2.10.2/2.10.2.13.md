# 𑁨𑁇𑁧𑁦𑁇𑁨𑁇𑁧𑁩 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀤𑀲𑀫𑀦𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀤𑀼𑀓

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

(401.)

491. 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀳𑁂𑀢𑀼𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(402.)

492. 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀯𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀦𑁆𑀦𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(403.)

493. 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀯𑀺𑀘𑀸𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(404.)

494. 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(405.)

495. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀧𑀦𑁆𑀦𑀭𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(406.)

496. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(407.)

497. 𑀦 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀙𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(408.)

498. 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀦𑀺𑀬𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀲𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀢𑀺𑀳𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀓𑀢𑀺𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸? 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

499. _𑀥𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀁 𑀥𑀫𑁆𑀫𑀥𑀸𑀢𑀼,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀲𑀍𑀆𑀬𑀢𑀦𑀁 𑀦𑀸𑀫𑀭𑀽𑀧𑀁,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀪𑀯𑀸𑁇_  


500. _𑀚𑀸𑀢𑀺 𑀚𑀭𑀸 𑀘 𑀫𑀭𑀡𑀁,_  
_𑀢𑀺𑀓𑁂𑀲𑁆𑀯𑁂𑀓𑀽𑀦𑀯𑀻𑀲𑀢𑀺;_  
_𑀕𑁄𑀘𑁆𑀙𑀓𑁂𑀲𑀼 𑀘 𑀧𑀜𑁆𑀜𑀸𑀲,_  
_𑀅𑀝𑁆𑀞 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑁂 𑀧𑀤𑀸𑁇_  


501. _𑀫𑀳𑀦𑁆𑀢𑀭𑁂 𑀧𑀦𑁆𑀦𑀭𑀲,_  
_𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀢𑀢𑁄 𑀧𑀭𑁂;_  
_𑀢𑁂𑀯𑀻𑀲 𑀧𑀤𑀲𑀢𑀁 𑀏𑀢𑀁,_  
_𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑁂 𑀦 𑀮𑀩𑁆𑀪𑀢𑀻𑀢𑀺𑁇 [𑁧𑁨𑁩]_  


---

502. 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲𑁄 𑀤𑀲𑀫𑁄𑁇
