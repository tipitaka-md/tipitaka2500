# 𑁨𑁇𑁧𑁦𑁇𑁨𑁇𑁩 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀤𑀲𑀫𑀦𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀤𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀤𑀼𑀓

* 𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓

(392.)

482\. 𑀆𑀲𑀯𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑁂𑀳𑀺 𑀘 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑁂𑀯 𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀘 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀓𑁂𑀦 𑀔𑀦𑁆𑀥𑁂𑀦 𑀤𑀲𑀳𑀸𑀬𑀢𑀦𑁂𑀳𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸; 𑀏𑀓𑁂𑀦𑀸𑀬𑀢𑀦𑁂𑀦 𑀏𑀓𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀓𑁂𑀳𑀺𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇

(393.)

483\. 𑀅𑀦𑀸𑀲𑀯𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸…  𑀆𑀲𑀯𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀅𑀦𑀸𑀲𑀯𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸…  𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀆𑀬𑀢𑀦𑁂𑀳𑀺 𑀙𑀳𑀺 𑀥𑀸𑀢𑀽𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑁇
