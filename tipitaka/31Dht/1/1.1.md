# 𑁧𑁇𑁧 𑀦𑀬𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀦𑀬𑀫𑀸𑀢𑀺𑀓𑀸

(1.)

2\. 𑀲𑀗𑁆𑀕𑀳𑁄 𑀅𑀲𑀗𑁆𑀕𑀳𑁄 (𑁧)

3\. 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 (𑁨)

4\. 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 (𑁩)

5\. 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 (𑁪)

6\. 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 (𑁫)

7\. 𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑁄 𑀯𑀺𑀧𑁆𑀧𑀬𑁄𑀕𑁄 (𑁬)

8\. 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 (𑁭)

9\. 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 (𑁮)

10\. 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 (𑁯)

11\. 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 (𑁧𑁦)

12\. 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 (𑁧𑁧)

13\. 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 (𑁧𑁨)

14\. 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 (𑁧𑁩)

15\. 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 𑀅𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁𑁇 (𑁧𑁪)
