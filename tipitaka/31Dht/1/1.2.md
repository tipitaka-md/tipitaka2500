# 𑁧𑁇𑁨 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀉𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀫𑀸𑀢𑀺𑀓𑀸

(2.)

16\. 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸 (𑁧)

17\. 𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺 (𑁨)

18\. 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀥𑀸𑀢𑀼𑀬𑁄 (𑁩)

19\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 (𑁪)

20\. 𑀩𑀸𑀯𑀻𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 (𑁫)

21\. 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑁄 (𑁬)

22\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 (𑁭)

23\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸 (𑁮)

24\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 (𑁯)

25\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀛𑀸𑀦𑀸𑀦𑀺 (𑁧𑁦)

26\. 𑀘𑀢𑀲𑁆𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬𑁄 (𑁧𑁧)

27\. 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 (𑁧𑁨)

28\. 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 (𑁧𑁩)

29\. 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 (𑁧𑁪)

30\. 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 (𑁧𑁫)

31\. 𑀨𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸 𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄𑁇 (𑁧𑁬--𑁨𑁨)
