# 𑁩𑁦 𑀕𑀦𑁆𑀣𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀕𑀦𑁆𑀣𑀕𑀦𑁆𑀣𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(44.)

141\. 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

142\. 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

143\. 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀦𑀕𑀦𑁆𑀣𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

144\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁄𑁇)
