# 𑀳𑁂𑀢𑀼

* 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀳𑁂𑀢𑀼

(5.)

12\. 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀁 𑀫𑁄𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁, 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀫𑀳𑀸𑀪𑀽𑀢𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑁆𑀯𑁂 𑀫𑀳𑀸𑀪𑀽𑀢𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀁𑁇 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀲𑀳𑀕𑀢𑀁 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀲𑀳𑀕𑀢𑀁 𑀫𑁄𑀳𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸 𑀔𑀦𑁆𑀥𑀸; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

13\. 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

14\. 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(6.)

15\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀙, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀘…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

16\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀁 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
