# 𑁨𑁯 𑀕𑀦𑁆𑀣𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀤𑀼𑀓

### 𑀕𑀦𑁆𑀣𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀤𑀼𑀓

(43.)

137\. 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

138\. 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄𑀕𑀦𑁆𑀣𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

139\. 𑀦𑀕𑀦𑁆𑀣𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀕𑀦𑁆𑀣𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀕𑀦𑁆𑀣𑁄 𑀘𑁂𑀯 𑀦𑀅𑀕𑀦𑁆𑀣𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

140\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁄𑁇)
