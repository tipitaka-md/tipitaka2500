# 𑁫𑁩 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀤𑀼𑀓

* 𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀤𑀼𑀓

(72.)

226\. 𑀦𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀺𑀮𑁂𑀲𑁄 𑀘𑁂𑀯 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

227\. 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

228\. 𑀦𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀺𑀮𑁂𑀲𑁄 𑀘𑁂𑀯 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

229\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇
