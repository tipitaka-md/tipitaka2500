# 𑁧𑁇𑁨 𑀤𑀼𑀓𑁅𑀤𑁆𑀤𑁂𑀲

* 𑀫𑀸𑀢𑀺𑀓𑀸

* 𑀤𑀼𑀓𑁅𑀤𑁆𑀤𑁂𑀲

(8.)

64\. **𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸—**

65\. 𑀓𑁄𑀥𑀦𑁄 𑀘, 𑀉𑀧𑀦𑀸𑀳𑀻 𑀘𑁇 (𑁧)

66\. 𑀫𑀓𑁆𑀔𑀻 𑀘, 𑀧𑀍𑀆𑀲𑀻 𑀘𑁇 (𑁨)

67\. 𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀘, 𑀫𑀘𑁆𑀙𑀭𑀻 𑀘𑁇 (𑁩)

68\. 𑀲𑀞𑁄 𑀘, 𑀫𑀸𑀬𑀸𑀯𑀻 𑀘𑁇 (𑁪)

69\. 𑀅𑀳𑀺𑀭𑀺𑀓𑁄 𑀘, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀘𑁇 (𑁫)

70\. 𑀤𑀼𑀩𑁆𑀩𑀘𑁄 𑀘, 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 (𑁬)

71\. 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁄 𑀘, 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘𑁇 (𑁭)

72\. 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺 𑀘, 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀘𑁇 (𑁮)

73\. 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘, 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘𑁇 (𑁯)

74\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀲𑀁𑀬𑁄𑀚𑀦𑁄 𑀘, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑁄 𑀘𑁇 (𑁧𑁦)

75\. 𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀘, 𑀅𑀦𑀼𑀧𑀦𑀸𑀳𑀻 𑀘𑁇 (𑁧𑁧)

76\. 𑀅𑀫𑀓𑁆𑀔𑀻 𑀘, 𑀅𑀧𑀍𑀆𑀲𑀻 𑀘𑁇 (𑁧𑁨)

77\. 𑀅𑀦𑀺𑀲𑁆𑀲𑀼𑀓𑀻 𑀘, 𑀅𑀫𑀘𑁆𑀙𑀭𑀻 𑀘𑁇 (𑁧𑁩)

78\. 𑀅𑀲𑀞𑁄 𑀘, 𑀅𑀫𑀸𑀬𑀸𑀯𑀻 𑀘𑁇 (𑁧𑁪)

79\. 𑀳𑀺𑀭𑀺𑀫𑀸 𑀘, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀘𑁇 (𑁧𑁫)

80\. 𑀲𑀼𑀯𑀘𑁄 𑀘, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀘𑁇 (𑁧𑁬)

81\. 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑁄 𑀘, 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘𑁇 (𑁧𑁭)

82\. 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀘, 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀘𑁇 (𑁧𑁮)

83\. 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀘, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀘𑁇 (𑁧𑁯)

84\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀼𑀮𑁆𑀮𑀪𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 (𑁨𑁦)

85\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀬𑀸𑁇 (𑁨𑁧)

86\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀼𑀢𑀧𑁆𑀧𑀬𑀸𑁇 (𑁨𑁨)

87\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺𑁇 (𑁨𑁩)

88\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺𑁇 (𑁨𑁪)

89\. 𑀳𑀻𑀦𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀘, 𑀧𑀡𑀻𑀢𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀘𑁇 (𑁨𑁫)

90\. 𑀢𑀺𑀢𑁆𑀢𑁄 𑀘, 𑀢𑀧𑁆𑀧𑁂𑀢𑀸 𑀘𑁇 (𑁨𑁬)
