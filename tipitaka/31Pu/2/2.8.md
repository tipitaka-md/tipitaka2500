# 𑁨𑁇𑁮 𑀅𑀝𑁆𑀞𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺

### 𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀅𑀝𑁆𑀞𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺

(207.)

534\. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀨𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸? 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀅𑀦𑀸𑀕𑀸𑀫𑀻, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀅𑀭𑀳𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄, 𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀨𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑁇
