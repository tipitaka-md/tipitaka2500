# 𑁨𑁇𑁧𑁇𑁧𑁨

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

(64.)

142\. “𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀉𑀪𑀢𑁄 𑀯𑀘𑀻𑀲𑀁𑀲𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀍𑀆𑀲𑁄 𑀘𑁂𑀢𑀲𑁄 𑀆𑀖𑀸𑀢𑁄 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀦𑀪𑀺𑀭𑀤𑁆𑀥𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀲𑁆𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  ‘𑀤𑀻𑀖𑀢𑁆𑀢𑀸𑀬 𑀔𑀭𑀢𑁆𑀢𑀸𑀬 𑀯𑀸𑀍𑀅𑀢𑁆𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀦 𑀨𑀸𑀲𑀼𑀁 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺’𑁇

143\. 𑀬𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀉𑀪𑀢𑁄 𑀯𑀘𑀻𑀲𑀁𑀲𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀍𑀆𑀲𑁄 𑀘𑁂𑀢𑀲𑁄 𑀆𑀖𑀸𑀢𑁄 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀦𑀪𑀺𑀭𑀤𑁆𑀥𑀺 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀼𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀲𑁆𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁—  ‘𑀦 𑀤𑀻𑀖𑀢𑁆𑀢𑀸𑀬 𑀔𑀭𑀢𑁆𑀢𑀸𑀬 𑀯𑀸𑀍𑀅𑀢𑁆𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀨𑀸𑀲𑀼𑀁 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’”𑀢𑀺𑁇
