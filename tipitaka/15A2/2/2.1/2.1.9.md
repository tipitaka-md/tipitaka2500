### 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

(61.)

137. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀓𑀺𑀁𑀧𑀼𑀭𑀺𑀲𑀸 𑀫𑀸𑀦𑀼𑀲𑀺𑀁 𑀯𑀸𑀘𑀁 𑀦 𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀫𑀸 𑀘 𑀫𑀼𑀲𑀸 𑀪𑀡𑀺𑀫𑁆𑀳𑀸, 𑀫𑀸 𑀘 𑀧𑀭𑀁 𑀅𑀪𑀽𑀢𑁂𑀦 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺𑀫𑁆𑀳𑀸𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀓𑀺𑀁𑀧𑀼𑀭𑀺𑀲𑀸 𑀫𑀸𑀦𑀼𑀲𑀺𑀁 𑀯𑀸𑀘𑀁 𑀦 𑀪𑀸𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
