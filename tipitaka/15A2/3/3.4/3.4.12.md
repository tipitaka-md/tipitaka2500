# 𑁩𑁇𑁪𑁇𑁧𑁨

### 𑀲𑀦𑁆𑀣𑀸𑀭𑀯𑀕𑁆𑀕

(163.)

255\. “𑀤𑁆𑀯𑁂𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀤𑁆𑀯𑁂? 𑀆𑀫𑀺𑀲𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀜𑁆𑀘𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀦𑀺𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀥𑀫𑁆𑀫𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀦𑁆”𑀢𑀺𑁇
