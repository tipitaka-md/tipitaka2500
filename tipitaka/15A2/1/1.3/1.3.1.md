# 𑁧𑁇𑁩𑁇𑁧

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀩𑀸𑀮𑀯𑀕𑁆𑀕

(22.)

54\. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀮𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀩𑀸𑀮𑀸𑀢𑀺𑁇

55\. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀧𑀡𑁆𑀟𑀺𑀢𑀸”𑀢𑀺𑁇
