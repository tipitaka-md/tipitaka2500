# 𑁧𑁇𑁩𑁇𑁬

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀩𑀸𑀮𑀯𑀕𑁆𑀕

(27.)

61\. “𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀕𑀢𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸—  𑀦𑀺𑀭𑀬𑁄 𑀯𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀯𑀸𑀢𑀺𑁇

62\. 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀕𑀢𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀕𑀢𑀺 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸—  𑀤𑁂𑀯𑀸 𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀸”𑀢𑀺𑁇
