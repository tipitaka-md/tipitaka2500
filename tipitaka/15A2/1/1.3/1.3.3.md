# 𑁧𑁇𑁩𑁇𑁩

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀩𑀸𑀮𑀯𑀕𑁆𑀕

(24.)

57\. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀬𑁄 𑀘 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀬𑁄 𑀘 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻𑀢𑀺𑁇

58\. 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀸𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀬𑁄 𑀘 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀬𑁄 𑀘 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀢𑀣𑀸𑀕𑀢𑀁 𑀦𑀸𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
