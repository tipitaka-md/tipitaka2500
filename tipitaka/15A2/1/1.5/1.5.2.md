# 𑁧𑁇𑁫𑁇𑁨

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀧𑀭𑀺𑀲𑀯𑀕𑁆𑀕

(44.)

108\. “𑀤𑁆𑀯𑁂𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀲𑀸𑁇 𑀓𑀢𑀫𑀸 𑀤𑁆𑀯𑁂? 𑀯𑀕𑁆𑀕𑀸 𑀘 𑀧𑀭𑀺𑀲𑀸 𑀲𑀫𑀕𑁆𑀕𑀸 𑀘 𑀧𑀭𑀺𑀲𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀕𑁆𑀕𑀸 𑀧𑀭𑀺𑀲𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀕𑁆𑀕𑀸 𑀧𑀭𑀺𑀲𑀸𑁇

109\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀧𑀭𑀺𑀲𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀔𑀻𑀭𑁄𑀤𑀓𑀻𑀪𑀽𑀢𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀘𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀧𑀭𑀺𑀲𑀸𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀧𑀭𑀺𑀲𑀸𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀲𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀫𑀕𑁆𑀕𑀸 𑀧𑀭𑀺𑀲𑀸”𑀢𑀺𑁇
