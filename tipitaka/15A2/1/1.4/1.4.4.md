# 𑁧𑁇𑁪𑁇𑁪

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀫𑀘𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

(36.)

76\. 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀮𑁄𑀓𑁂 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸, 𑀓𑀢𑁆𑀣 𑀘 𑀤𑀸𑀦𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀤𑁆𑀯𑁂 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀮𑁄𑀓𑁂 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸—  𑀲𑁂𑀔𑁄 𑀘 𑀅𑀲𑁂𑀔𑁄 𑀘𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀮𑁄𑀓𑁂 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸, 𑀏𑀢𑁆𑀣 𑀘 𑀤𑀸𑀦𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

77\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

78\. _“𑀲𑁂𑀔𑁄 𑀅𑀲𑁂𑀔𑁄 𑀘 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂,_  
_𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀁 𑀳𑁄𑀦𑁆𑀢𑀺;_  
_𑀢𑁂 𑀉𑀚𑁆𑀚𑀼𑀪𑀽𑀢𑀸 𑀓𑀸𑀬𑁂𑀦,_  
_𑀯𑀸𑀘𑀸𑀬 𑀉𑀤 𑀘𑁂𑀢𑀲𑀸;_  
_𑀔𑁂𑀢𑁆𑀢𑀁 𑀢𑀁 𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀏𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_
