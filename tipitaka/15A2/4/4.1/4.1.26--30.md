# 𑁪𑁇𑁧𑁇𑁨𑁬--𑁩𑁦

### 𑀧𑁂𑀬𑁆𑀬𑀸𑀮

### 𑀓𑁄𑀥𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(186.)

281. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀅𑀓𑁆𑀓𑁄𑀥𑁄 𑀘 𑀅𑀦𑀼𑀧𑀦𑀸𑀳𑁄 𑀘…  𑀅𑀫𑀓𑁆𑀔𑁄 𑀘 𑀅𑀧𑀍𑀆𑀲𑁄 𑀘…  𑀅𑀦𑀺𑀲𑁆𑀲𑀸 𑀘 𑀅𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀜𑁆𑀘…  𑀅𑀫𑀸𑀬𑀸 𑀘 𑀅𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀜𑁆𑀘…  𑀳𑀺𑀭𑀻 𑀘 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺”𑁇
