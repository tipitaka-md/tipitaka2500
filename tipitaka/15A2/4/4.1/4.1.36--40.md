# 𑁪𑁇𑁧𑁇𑁩𑁬--𑁪𑁦

### 𑀧𑁂𑀬𑁆𑀬𑀸𑀮

### 𑀓𑁄𑀥𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(188.)

283\. “𑀤𑁆𑀯𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀅𑀓𑁆𑀓𑁄𑀥𑁂𑀦 𑀘 𑀅𑀦𑀼𑀧𑀦𑀸𑀳𑁂𑀦 𑀘…  𑀅𑀫𑀓𑁆𑀔𑁂𑀦 𑀘 𑀅𑀧𑀍𑀆𑀲𑁂𑀦 𑀘…  𑀅𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀘 𑀅𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑁂𑀦 𑀘…  𑀅𑀫𑀸𑀬𑀸𑀬 𑀘 𑀅𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑁂𑀦 𑀘…  𑀳𑀺𑀭𑀺𑀬𑀸 𑀘 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑁂𑀦 𑀘𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑁇
