# 𑁩𑁫 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓𑀸𑀤𑀺, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓𑀸𑀤𑀺, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

(44.)

138\. 𑀦𑁅𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀦𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑀢𑁆𑀢, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀙, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀲𑀢𑁆𑀢𑁇

139\. 𑀦𑁅𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀙, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀙, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀙𑁇

140\. 𑀦𑀅𑀦𑀼𑀧𑀦𑁆𑀦𑁄 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

141\. 𑀦𑁅𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

142\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑀢𑁆𑀢, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀢𑀻𑀡𑀺, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀲𑀢𑁆𑀢…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

---

143\. (𑀅𑀢𑀻𑀢𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓𑀲𑀤𑀺𑀲𑀁𑁇)
