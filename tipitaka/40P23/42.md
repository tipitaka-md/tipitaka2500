# 𑁪𑁨 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀢𑁆𑀢𑀺𑀓, 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓

### 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀢𑁆𑀢𑀺𑀓, 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓

(51.)

162\. 𑀦𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄 𑀦𑁅𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀙, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀢𑀻𑀡𑀺, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀙, 𑀓𑀫𑁆𑀫𑁂 𑀆𑀳𑀸𑀭𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀢𑀻𑀡𑀺, 𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀙…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀙𑁇

163\. 𑀦𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀯𑀻𑀲, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀏𑀓𑀯𑀻𑀲…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀯𑀻𑀲𑁇

164\. 𑀦𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇
