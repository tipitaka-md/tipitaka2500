# 𑁧𑁫 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

* 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

(19.)

66\. 𑀦𑀓𑀼𑀲𑀮𑀁 𑀦𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀙…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀙𑁇

67\. 𑀦𑀓𑀼𑀲𑀮𑀁 𑀦𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

68\. 𑀦𑀓𑀼𑀲𑀮𑀁 𑀦𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀙…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀙𑁇
