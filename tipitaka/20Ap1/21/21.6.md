# 𑁨𑁧𑁇𑁬 𑀉𑀤𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀡𑀺𑀓𑀸𑀭𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀉𑀤𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(25.)

2784\. _“𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀁 𑀲𑀫𑀡𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀁;_  
_𑀖𑀝𑁂𑀦𑁄𑀤𑀓𑀫𑀸𑀤𑀸𑀬,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(26.)

2785\. _𑀦𑀺𑀫𑁆𑀫𑀮𑁄 𑀳𑁄𑀫𑀳𑀁 𑀅𑀚𑁆𑀚,_  
_𑀯𑀺𑀫𑀮𑁄 𑀔𑀻𑀡𑀲𑀁𑀲𑀬𑁄;_  
_𑀪𑀯𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀨𑀮𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑁂 𑀫𑀫𑁇_  


(27.)

2786\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀉𑀤𑀓𑀁 𑀬𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀓𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(28.)

2787\. _𑀏𑀓𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄𑀯 𑀯𑀺𑀫𑀮𑁄 𑀅𑀳𑀼;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(29.)

2788\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2789\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀓𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2790\. 𑀉𑀤𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
