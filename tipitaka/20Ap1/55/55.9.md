# 𑁫𑁫𑁇𑁯 𑀯𑀦𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀪𑀤𑁆𑀤𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀯𑀦𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(251.)

7601\. _“𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(252.)

7602\. _𑀢𑀤𑀸𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂;_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀁 𑀢𑀢𑁄 𑀘𑀼𑀢𑁄𑁇_  


(253.)

7603\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(254.)

7604\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁄𑀢𑁄 𑀆𑀲𑀳𑀁 𑀢𑀳𑀺𑀁;_  
_𑀯𑀲𑀢𑁂 𑀕𑀼𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀛𑀸𑀦𑀭𑀢𑁄 𑀲𑀤𑀸𑁇_  


(255.)

7605\. _𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀲𑀤𑀸 𑀧𑀫𑀼𑀤𑀺𑀢𑀸𑀦𑀦𑁄;_  
_𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀲𑀼 𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


(256.)

7606\. _𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀳𑀺𑀢𑀸𑀲𑀬𑁂;_  
_𑀯𑀺𑀲𑀝𑁆𑀞𑁄 𑀦𑀘𑀺𑀭𑁂𑀦𑀸𑀲𑀺𑀁,_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀼𑀕𑀢𑀲𑀸𑀯𑀓𑁂𑁇_  


(257.)

7607\. _𑀉𑀧𑁂𑀘𑁆𑀘 𑀧𑀸𑀤𑀫𑀽𑀮𑀫𑁆𑀳𑀺,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀢𑀤𑀸𑀲𑁆𑀲𑀫𑁂;_  
_𑀓𑀤𑀸𑀘𑀺 𑀲𑀸𑀫𑀺𑀲𑀁 𑀤𑁂𑀢𑀺,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺 𑀘𑁂𑀓𑀤𑀸𑁇_  


(258.)

7608\. _𑀢𑀤𑀸 𑀯𑀺𑀧𑀼𑀮𑀧𑁂𑀫𑁂𑀦,_  
_𑀉𑀧𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀢𑁆𑀭𑀚𑀁;_  
_𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀕𑀢𑁄 𑀲𑀕𑁆𑀕𑀁,_  
_𑀧𑀯𑀸𑀲𑁄 𑀲𑀖𑀭𑀁 𑀬𑀣𑀸𑁇_  


(259.)

7609\. _𑀲𑀕𑁆𑀕𑀸 𑀘𑀼𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁄 𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀼𑀦𑀸;_  
_𑀅𑀕𑀸𑀭𑀁 𑀙𑀟𑁆𑀟𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀩𑀳𑀼𑀲𑁄 𑀅𑀳𑀁𑁇_  


(260.)

7610\. _𑀲𑀫𑀡𑁄 𑀢𑀸𑀧𑀲𑁄 𑀯𑀺𑀧𑁆𑀧𑁄,_  
_𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁄 𑀢𑀣𑁂𑀯𑀳𑀁;_  
_𑀳𑀼𑀢𑁆𑀯𑀸 𑀯𑀲𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀅𑀦𑁂𑀓𑀲𑀢𑀲𑁄 𑀅𑀳𑀁𑁇_  


(261.)

7611\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀭𑀫𑁆𑀫𑁂 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀯𑁂;_  
_𑀯𑀘𑁆𑀙𑀕𑁄𑀢𑁆𑀢𑁄 𑀤𑀺𑀚𑁄 𑀢𑀲𑁆𑀲,_  
_𑀚𑀸𑀬𑀸𑀬 𑀅𑀳𑀫𑁄𑀓𑁆𑀓𑀫𑀺𑀁𑁇_  


(262.)

7612\. _𑀫𑀸𑀢𑀼 𑀫𑁂 𑀤𑁄𑀳𑀍𑀑 𑀆𑀲𑀺,_  
_𑀢𑀺𑀭𑁄𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑀲𑁆𑀲 𑀫𑁂;_  
_𑀚𑀸𑀬𑀫𑀸𑀦𑀲𑀫𑀻𑀧𑀫𑁆𑀳𑀺,_  
_𑀯𑀦𑀯𑀸𑀲𑀸𑀬 𑀦𑀺𑀘𑁆𑀙𑀬𑁄𑁇_  


(263.)

7613\. _𑀢𑀢𑁄 𑀫𑁂 𑀅𑀚𑀦𑀻 𑀫𑀸𑀢𑀸,_  
_𑀭𑀫𑀡𑀻𑀬𑁂 𑀯𑀦𑀦𑁆𑀢𑀭𑁂;_  
_𑀕𑀩𑁆𑀪𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀁 𑀫𑀁,_  
_𑀓𑀸𑀲𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀼𑀁𑁇_  


(264.)

7614\. _𑀢𑀢𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄,_  
_𑀚𑀸𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀤𑁆𑀥𑀚𑁄;_  
_𑀢𑀲𑁆𑀲 𑀫𑀺𑀢𑁆𑀢𑁄 𑀧𑀺𑀬𑁄 𑀆𑀲𑀺𑀁,_  
_𑀲𑀁𑀯𑀺𑀲𑀝𑁆𑀞𑁄 𑀲𑀼𑀫𑀸𑀦𑀺𑀬𑁄𑁇_  


(265.)

7615\. _𑀲𑀢𑁆𑀢𑀲𑀸𑀭𑁂𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂,_  
_𑀑𑀳𑀸𑀬 𑀯𑀺𑀧𑀼𑀮𑀁 𑀬𑀲𑀁;_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(266.)

7616\. _𑀯𑀦𑀸𑀮𑀬𑀁 𑀪𑀸𑀯𑀦𑀻𑀬𑀁,_  
_𑀓𑀲𑁆𑀲𑀧𑀁 𑀥𑀼𑀢𑀯𑀸𑀤𑀺𑀓𑀁;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀁,_  
_𑀉𑀧𑁂𑀲𑀺𑀁 𑀦𑀭𑀲𑀸𑀭𑀣𑀺𑀁𑁇_  


(267.)

7617\. _𑀲𑁄 𑀫𑁂 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀁 𑀲𑀫𑁆𑀧𑀓𑀸𑀲𑀬𑀁;_  
_𑀢𑀢𑁄𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀦𑀫𑁂𑀯 𑀧𑀼𑀦𑀸𑀕𑀫𑀁𑁇_  


(268.)

7618\. _𑀢𑀢𑁆𑀣𑀸𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀁,_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀅𑀨𑀲𑁆𑀲𑀬𑀺𑀁;_  
_𑀅𑀳𑁄 𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑁄𑀫𑁆𑀳𑀺,_  
_𑀲𑀼𑀫𑀺𑀢𑁆𑀢𑁂𑀦𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀢𑁄𑁇_  


(269.)

7619\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(270.)

7620\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(271.)

7621\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


7622\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑀦𑀯𑀘𑁆𑀙𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

7623\. 𑀯𑀦𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
