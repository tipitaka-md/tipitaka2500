# 𑁫𑁫𑁇𑁭 𑀅𑀪𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀤𑁆𑀤𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀪𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(195.)

7541\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(196.)

7542\. _𑀲𑀭𑀡𑀕𑀫𑀦𑁂 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀦𑀺𑀯𑁂𑀲𑁂𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑁄;_  
_𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀻𑀮𑁂 𑀦𑀺𑀯𑁂𑀲𑁂𑀲𑀺,_  
_𑀤𑀲𑀓𑀫𑁆𑀫𑀧𑀣𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(197.)

7543\. _𑀤𑁂𑀢𑀺 𑀓𑀲𑁆𑀲𑀘𑀺 𑀲𑁄 𑀯𑀻𑀭𑁄,_  
_𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻 𑀢𑀣𑀸 𑀅𑀝𑁆𑀞,_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀯𑁂𑀘𑁆𑀙𑀢𑀺𑁇_  


(198.)

7544\. _𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀬𑁄𑀚𑁂𑀲𑀺,_  
_𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀢𑁆𑀢𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀤𑁂𑀢𑀺 𑀓𑀲𑁆𑀲𑀘𑀺 𑀦𑀸𑀣𑁄 𑀲𑁄,_  
_𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑁇_  


(199.)

7545\. _𑀩𑁄𑀥𑀦𑁂𑀬𑁆𑀬𑀁 𑀧𑀚𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀫𑁆𑀧𑀺 𑀬𑁄𑀚𑀦𑀁;_  
_𑀔𑀡𑁂𑀦 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀦𑁂𑀢𑀺 𑀦𑀭𑀲𑀸𑀭𑀣𑀺𑁇_  


(200.)

7546\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀢𑁆𑀭𑀚𑁄;_  
_𑀧𑀸𑀭𑀕𑀽 𑀲𑀩𑁆𑀩𑀯𑁂𑀤𑀸𑀦𑀁,_  
_𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀲𑀫𑁆𑀫𑀢𑁄𑁇_  


(201.)

7547\. _𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀘 𑀓𑀼𑀲𑀮𑁄,_  
_𑀦𑀺𑀖𑀡𑁆𑀟𑀼𑀫𑁆𑀳𑀺 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀧𑀤𑀓𑁄 𑀓𑁂𑀝𑀼𑀪𑀯𑀺𑀤𑀽,_  
_𑀙𑀦𑁆𑀤𑁄𑀯𑀺𑀘𑀺𑀢𑀺𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


(202.)

7548\. _𑀚𑀗𑁆𑀖𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀺𑀘𑀭𑀁,_  
_𑀳𑀁𑀲𑀸𑀭𑀸𑀫𑀫𑀼𑀧𑁂𑀘𑁆𑀘𑀳𑀁;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀭𑀤𑀁 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀫𑀳𑀸𑀚𑀦𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁𑁇_  


(203.)

7549\. _𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀁 𑀯𑀺𑀭𑀚𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀫𑀢𑀻 𑀅𑀳𑀁;_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀓𑀮𑁆𑀬𑀸𑀡𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀯𑀺𑀫𑀮𑀁 𑀅𑀳𑀁𑁇_  


(204.)

7550\. _𑀩𑁆𑀬𑀸𑀳𑀢𑀁 𑀧𑀼𑀦𑀭𑀼𑀢𑁆𑀢𑀁 𑀯𑀸,_  
_𑀅𑀧𑀢𑁆𑀣𑀁 𑀯𑀸 𑀦𑀺𑀭𑀢𑁆𑀣𑀓𑀁;_  
_𑀦𑀸𑀤𑁆𑀤𑀲𑀁 𑀢𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀢𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁𑁇_  


(205.)

7551\. _𑀦𑀘𑀺𑀭𑁂𑀦𑁂𑀯 𑀓𑀸𑀮𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀦𑀺𑀧𑀼𑀡𑁄 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀕𑀼𑀡𑀺𑀲𑀫𑁆𑀫𑀢𑁄𑁇_  


(206.)

7552\. _𑀢𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀦𑁆𑀣𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸;_  
_𑀲𑀦𑁆𑀣𑀯𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀮𑁄𑀓𑀕𑁆𑀕𑀁,_  
_𑀤𑁂𑀲𑀬𑀺𑀲𑁆𑀲𑀁 𑀤𑀺𑀦𑁂 𑀤𑀺𑀦𑁂𑁇_  


(207.)

7553\. _‘𑀯𑀺𑀭𑀢𑁆𑀢𑁄𑀲𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀁𑀲𑀸𑀭𑁂 𑀲𑀪𑀬𑁂 𑀯𑀲𑀁;_  
_𑀓𑀭𑀼𑀡𑀸𑀬 𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺,_  
_𑀢𑀢𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀫𑀼𑀦𑀺𑁇_  


(208.)

7554\. _𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀯𑀬𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀦 𑀓𑀺𑀮𑁂𑀲𑀯𑀲𑁄 𑀅𑀳𑀼;_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀲𑁄 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑁄𑁇_  


(209.)

7555\. _𑀤𑀼𑀩𑁆𑀩𑀮𑀸𑀦𑀺 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀺,_  
_𑀬𑀲𑁆𑀲𑀸𑀲𑀬𑀕𑀢𑀸𑀦𑀺 𑀫𑁂;_  
_𑀜𑀸𑀡𑀕𑁆𑀕𑀺𑀧𑀭𑀺𑀤𑀟𑁆𑀠𑀸𑀦𑀺,_  
_𑀦 𑀔𑀻𑀬𑀺𑀁𑀲𑀼 𑀢𑀫𑀩𑁆𑀪𑀼𑀢𑀁𑁇_  


(210.)

7556\. _𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀲𑁆𑀲 𑀕𑀭𑀼,_  
_𑀮𑁄𑀓𑁄 𑀬𑀲𑁆𑀲 𑀢𑀣𑀸 𑀕𑀭𑀼;_  
_𑀢𑀣𑀸𑀧𑀺 𑀮𑁄𑀓𑀸𑀘𑀭𑀺𑀬𑁄,_  
_𑀮𑁄𑀓𑁄 𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑁄’𑁇_  


(211.)

7557\. _𑀏𑀯𑀫𑀸𑀤𑀻𑀳𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀓𑀺𑀢𑁆𑀢𑀬𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁;_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑀢𑁄 𑀲𑀕𑁆𑀕𑀁 𑀢𑀢𑁄 𑀘𑀼𑀢𑁄𑁇_  


(212.)

7558\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀫𑀪𑀺𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑀺𑀢𑁆𑀢𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(213.)

7559\. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀁,_  
_𑀧𑀸𑀤𑁂𑀲𑀺𑀁 𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀁;_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀸𑀭𑀚𑁆𑀚𑀁,_  
_𑀩𑀳𑀼𑀲𑁄𑀦𑀼𑀪𑀯𑀺𑀁 𑀅𑀳𑀁𑁇_  


(214.)

7560\. _𑀤𑀼𑀯𑁂 𑀪𑀯𑁂 𑀧𑀚𑀸𑀬𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑁂 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑁂;_  
_𑀅𑀜𑁆𑀜𑀁 𑀕𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑀺𑀢𑁆𑀢𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(215.)

7561\. _𑀤𑀼𑀯𑁂 𑀓𑀼𑀮𑁂 𑀧𑀚𑀸𑀬𑀸𑀫𑀺,_  
_𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀅𑀣 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂;_  
_𑀦𑀻𑀘𑁂 𑀓𑀼𑀮𑁂 𑀦 𑀚𑀸𑀬𑀸𑀫𑀺,_  
_𑀓𑀺𑀢𑁆𑀢𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(216.)

7562\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀭𑀜𑁆𑀜𑁄𑀳𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑀸𑀫𑁂𑀦 𑀘𑀸𑀪𑀬𑁄𑁇_  


(217.)

7563\. _𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀯𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀕𑀡𑁆𑀞𑁂𑀦 𑀯𑀺𑀫𑁄𑀳𑀺𑀢𑁄;_  
_𑀧𑁂𑀲𑀺𑀢𑁄 𑀦𑀸𑀝𑀧𑀼𑀢𑁆𑀢𑁂𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀫𑀼𑀧𑁂𑀘𑁆𑀘𑀳𑀁𑁇_  


(218.)

7564\. _𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀧𑀼𑀡𑀁 𑀧𑀜𑁆𑀳𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀦𑀘𑀺𑀭𑀁,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(219.)

7565\. _𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀯𑀭𑀁,_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑁄 𑀳𑁄𑀫𑀺 𑀲𑀩𑁆𑀩𑀤𑀸;_  
_𑀲𑀼𑀕𑀦𑁆𑀥𑀤𑁂𑀳𑀯𑀤𑀦𑁄,_  
_𑀆𑀲𑀺𑀁 𑀲𑀼𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄𑁇_  


(220.)

7566\. _𑀢𑀺𑀓𑁆𑀔𑀳𑀸𑀲𑀮𑀳𑀼𑀧𑀜𑁆𑀜𑁄,_  
_𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄 𑀢𑀣𑁂𑀯𑀳𑀁;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀧𑀝𑀺𑀪𑀸𑀦𑁄 𑀘,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸𑁇_  


(221.)

7567\. _𑀅𑀪𑀺𑀢𑁆𑀣𑀯𑀺𑀢𑁆𑀯𑀸 𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀸𑀳𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀲𑀫𑀁 𑀲𑀬𑀫𑁆𑀪𑀼𑀁;_  
_𑀦 𑀕𑀘𑁆𑀙𑀺 𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀅𑀧𑀸𑀬𑀪𑀽𑀫𑀺𑀁,_  
_𑀲𑀢𑀁𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀮𑁂𑀦 𑀢𑀲𑁆𑀲𑁇_  


(222.)

7568\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(223.)

7569\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(224.)

7570\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


7571\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀪𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
