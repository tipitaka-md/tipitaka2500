# 𑁫𑁫𑁇𑁧𑁦 𑀘𑀽𑀍𑀅𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀤𑁆𑀤𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(272.)

7624\. _“𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(273.)

7625\. _𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄;_  
_𑀩𑁆𑀬𑀸𑀫𑀧𑁆𑀧𑀪𑀸𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀭𑀁𑀲𑀺𑀚𑀸𑀮𑀲𑀫𑁄𑀢𑁆𑀣𑀝𑁄𑁇_  


(274.)

7626\. _𑀅𑀲𑁆𑀲𑀸𑀲𑁂𑀢𑀸 𑀬𑀣𑀸 𑀘𑀦𑁆𑀤𑁄,_  
_𑀲𑀽𑀭𑀺𑀬𑁄𑀯 𑀧𑀪𑀗𑁆𑀓𑀭𑁄;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀸 𑀬𑀣𑀸 𑀫𑁂𑀖𑁄,_  
_𑀲𑀸𑀕𑀭𑁄𑀯 𑀕𑀼𑀡𑀸𑀓𑀭𑁄𑁇_  


(275.)

7627\. _𑀥𑀭𑀡𑀻𑀭𑀺𑀯 𑀲𑀻𑀮𑁂𑀦,_  
_𑀳𑀺𑀫𑀯𑀸𑀯 𑀲𑀫𑀸𑀥𑀺𑀦𑀸;_  
_𑀆𑀓𑀸𑀲𑁄 𑀯𑀺𑀬 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀲𑀗𑁆𑀕𑁄 𑀅𑀦𑀺𑀮𑁄 𑀬𑀣𑀸𑁇_  


(276.)

7628\. _𑀢𑀤𑀸𑀳𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀓𑀼𑀮𑁂;_  
_𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀲𑀜𑁆𑀘𑀬𑁂𑁇_  


(277.)

7629\. _𑀫𑀳𑀢𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀉𑀧𑁂𑀘𑁆𑀘 𑀥𑀫𑁆𑀫𑀫𑀲𑁆𑀲𑁄𑀲𑀺𑀁,_  
_𑀅𑀫𑀢𑀁𑀯 𑀫𑀦𑁄𑀳𑀭𑀁𑁇_  


(278.)

7630\. _𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀮𑀓𑁆𑀔𑀡𑀥𑀭𑁄,_  
_𑀲𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑁄𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸;_  
_𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯 𑀨𑀼𑀮𑁆𑀮𑀺𑀢𑁄𑁇_  


(279.)

7631\. _𑀭𑀁𑀲𑀺𑀚𑀸𑀮𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄,_  
_𑀤𑀺𑀢𑁆𑀢𑁄𑀯 𑀓𑀦𑀓𑀸𑀘𑀮𑁄;_  
_𑀩𑁆𑀬𑀸𑀫𑀧𑁆𑀧𑀪𑀸𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀲𑀢𑀭𑀁𑀲𑀻 𑀤𑀺𑀯𑀸𑀓𑀭𑁄𑁇_  


(280.)

7632\. _𑀲𑁄𑀡𑁆𑀡𑀸𑀦𑀦𑁄 𑀚𑀺𑀦𑀯𑀭𑁄,_  
_𑀲𑀫𑀡𑀻𑀯 𑀲𑀺𑀮𑀼𑀘𑁆𑀘𑀬𑁄;_  
_𑀓𑀭𑀼𑀡𑀸 𑀧𑀼𑀡𑁆𑀡𑀳𑀤𑀬𑁄,_  
_𑀕𑀼𑀡𑁂𑀦 𑀯𑀺𑀬 𑀲𑀸𑀕𑀭𑁄𑁇_  


(281.)

7633\. _𑀮𑁄𑀓𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀓𑀺𑀢𑁆𑀢𑀺 𑀘,_  
_𑀲𑀺𑀦𑁂𑀭𑀽𑀯 𑀦𑀕𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀬𑀲𑀲𑀸 𑀯𑀺𑀢𑁆𑀣𑀢𑁄 𑀯𑀻𑀭𑁄,_  
_𑀆𑀓𑀸𑀲𑀲𑀤𑀺𑀲𑁄 𑀫𑀼𑀦𑀺𑁇_  


(282.)

7634\. _𑀅𑀲𑀗𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀢𑁆𑀣,_  
_𑀅𑀦𑀺𑀮𑁄 𑀯𑀺𑀬 𑀦𑀸𑀬𑀓𑁄;_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀁,_  
_𑀫𑀳𑀻𑀯 𑀫𑀼𑀦𑀺𑀲𑀢𑁆𑀢𑀫𑁄𑁇_  


(283.)

7635\. _𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑁄 𑀮𑁄𑀓𑁂𑀦,_  
_𑀢𑁄𑀬𑁂𑀦 𑀧𑀤𑀼𑀫𑀁 𑀬𑀣𑀸;_  
_𑀓𑀼𑀯𑀸𑀤𑀕𑀘𑁆𑀙𑀤𑀳𑀦𑁄,_  
_𑀅𑀕𑁆𑀕𑀺𑀔𑀦𑁆𑀥𑁄𑀯 𑀲𑁄𑀪𑀲𑀺𑁇_  


(284.)

7636\. _𑀅𑀕𑀥𑁄 𑀯𑀺𑀬 𑀲𑀩𑁆𑀩𑀢𑁆𑀣,_  
_𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀲𑀦𑀸𑀲𑀓𑁄;_  
_𑀕𑀦𑁆𑀥𑀫𑀸𑀤𑀦𑀲𑁂𑀮𑁄𑀯,_  
_𑀕𑀼𑀡𑀕𑀦𑁆𑀥𑀯𑀺𑀪𑀽𑀲𑀺𑀢𑁄𑁇_  


(285.)

7637\. _𑀕𑀼𑀡𑀸𑀦𑀁 𑀆𑀓𑀭𑁄 𑀯𑀻𑀭𑁄,_  
_𑀭𑀢𑀦𑀸𑀦𑀁𑀯 𑀲𑀸𑀕𑀭𑁄;_  
_𑀲𑀺𑀦𑁆𑀥𑀽𑀯 𑀯𑀦𑀭𑀸𑀚𑀻𑀦𑀁,_  
_𑀓𑀺𑀮𑁂𑀲𑀫𑀮𑀳𑀸𑀭𑀓𑁄𑁇_  


(286.)

7638\. _𑀯𑀺𑀚𑀬𑀻𑀯 𑀫𑀳𑀸𑀬𑁄𑀥𑁄,_  
_𑀫𑀸𑀭𑀲𑁂𑀦𑀸𑀯𑀫𑀤𑁆𑀤𑀦𑁄;_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻𑀯 𑀲𑁄 𑀭𑀸𑀚𑀸,_  
_𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀭𑀢𑀦𑀺𑀲𑁆𑀲𑀭𑁄𑁇_  


(287.)

7639\. _𑀫𑀳𑀸𑀪𑀺𑀲𑀓𑁆𑀓𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀤𑁄𑀲𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑁄;_  
_𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑁄 𑀬𑀣𑀸 𑀯𑁂𑀚𑁆𑀚𑁄,_  
_𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀡𑁆𑀟𑀯𑀺𑀨𑀸𑀮𑀓𑁄𑁇_  


(288.)

7640\. _𑀲𑁄 𑀢𑀤𑀸 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑁄,_  
_𑀲𑀦𑀭𑀸𑀫𑀭𑀲𑀓𑁆𑀓𑀢𑁄;_  
_𑀧𑀭𑀺𑀲𑀸𑀲𑀼 𑀦𑀭𑀸𑀤𑀺𑀘𑁆𑀘𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀬𑀢𑁂 𑀚𑀺𑀦𑁄𑁇_  


(289.)

7641\. _‘𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀪𑁄𑀕𑁄,_  
_𑀲𑀻𑀮𑁂𑀦 𑀲𑀼𑀕𑀢𑀽𑀧𑀕𑁄;_  
_𑀪𑀸𑀯𑀦𑀸𑀬 𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀺’,_  
_𑀇𑀘𑁆𑀘𑁂𑀯𑀫𑀦𑀼𑀲𑀸𑀲𑀣𑁇_  


(290.)

7642\. _𑀤𑁂𑀲𑀦𑀁 𑀢𑀁 𑀫𑀳𑀲𑁆𑀲𑀸𑀤𑀁,_  
_𑀆𑀤𑀺𑀫𑀚𑁆𑀛𑀦𑁆𑀢𑀲𑁄𑀪𑀦𑀁;_  
_𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀅𑀫𑀢𑀁𑀯 𑀫𑀳𑀸𑀭𑀲𑀁𑁇_  


(291.)

7643\. _𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀼𑀫𑀥𑀼𑀭𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂;_  
_𑀲𑀼𑀕𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀫𑀲𑁆𑀲𑀳𑀁𑁇_  


(292.)

7644\. _𑀫𑀼𑀦𑀺𑀦𑁄 𑀕𑀦𑁆𑀥𑀓𑀼𑀝𑀺𑀬𑀸,_  
_𑀑𑀧𑀼𑀜𑁆𑀚𑁂𑀲𑀺𑀁 𑀢𑀤𑀸 𑀫𑀳𑀺𑀁;_  
_𑀘𑀢𑀼𑀚𑁆𑀚𑀸𑀢𑁂𑀦 𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀫𑀸𑀲𑁂 𑀅𑀝𑁆𑀞 𑀤𑀺𑀦𑁂𑀲𑁆𑀯𑀳𑀁𑁇_  


(293.)

7645\. _𑀧𑀡𑀺𑀥𑀸𑀬 𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑁆𑀢𑀁,_  
_𑀲𑀭𑀻𑀭𑀯𑀺𑀲𑁆𑀲𑀕𑀦𑁆𑀥𑀺𑀦𑁄;_  
_𑀢𑀤𑀸 𑀚𑀺𑀦𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑀦𑀼𑀮𑀸𑀪𑀺𑀢𑀁𑁇_  


(294.)

7646\. _‘𑀬𑁄 𑀬𑀁 𑀕𑀦𑁆𑀥𑀓𑀼𑀝𑀺𑀪𑀽𑀫𑀺𑀁,_  
_𑀕𑀦𑁆𑀥𑁂𑀦𑁄𑀧𑀼𑀜𑁆𑀚𑀢𑁂 𑀲𑀓𑀺𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁𑁇_  


(295.)

7647\. _𑀲𑀼𑀕𑀦𑁆𑀥𑀤𑁂𑀳𑁄 𑀲𑀩𑁆𑀩𑀢𑁆𑀣,_  
_𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀬𑀁 𑀦𑀭𑁄;_  
_𑀕𑀼𑀡𑀕𑀦𑁆𑀥𑀬𑀼𑀢𑁆𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(296.)

7648\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(297.)

7649\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀚𑀸𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀓𑀼𑀮𑁂 𑀅𑀳𑀁;_  
_𑀕𑀩𑁆𑀪𑀁 𑀫𑁂 𑀯𑀲𑀢𑁄 𑀫𑀸𑀢𑀸,_  
_𑀤𑁂𑀳𑁂𑀦𑀸𑀲𑀺 𑀲𑀼𑀕𑀦𑁆𑀥𑀺𑀢𑀸𑁇_  


(298.)

7650\. _𑀬𑀤𑀸 𑀘 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀸,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀸𑀫𑀺 𑀢𑀤𑀸 𑀧𑀼𑀭𑀻;_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀲𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑁂𑀳𑀺,_  
_𑀯𑀸𑀲𑀺𑀢𑀸 𑀯𑀺𑀬 𑀯𑀸𑀬𑀣𑁇_  


(299.)

7651\. _𑀧𑀼𑀧𑁆𑀨𑀯𑀲𑁆𑀲𑀜𑁆𑀘 𑀲𑀼𑀭𑀪𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀁 𑀫𑀦𑁄𑀭𑀫𑀁;_  
_𑀥𑀽𑀧𑀸𑀦𑀺 𑀘 𑀫𑀳𑀕𑁆𑀖𑀸𑀦𑀺,_  
_𑀉𑀧𑀯𑀸𑀬𑀺𑀁𑀲𑀼 𑀢𑀸𑀯𑀤𑁂𑁇_  


(300.)

7652\. _𑀤𑁂𑀯𑀸 𑀘 𑀲𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑁂𑀳𑀺,_  
_𑀥𑀽𑀧𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺 𑀢𑀁 𑀖𑀭𑀁;_  
_𑀯𑀸𑀲𑀬𑀺𑀁𑀲𑀼 𑀲𑀼𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀚𑀸𑀢𑁄 𑀅𑀳𑀁 𑀖𑀭𑁂𑁇_  


(301.)

7653\. _𑀬𑀤𑀸 𑀘 𑀢𑀭𑀼𑀡𑁄 𑀪𑀤𑁆𑀤𑁄,_  
_𑀧𑀞𑀫𑁂 𑀬𑁄𑀩𑁆𑀩𑀦𑁂 𑀞𑀺𑀢𑁄;_  
_𑀢𑀤𑀸 𑀲𑁂𑀮𑀁 𑀲𑀧𑀭𑀺𑀲𑀁,_  
_𑀯𑀺𑀦𑁂𑀢𑁆𑀯𑀸 𑀦𑀭𑀲𑀸𑀭𑀣𑀺𑁇_  


(302.)

7654\. _𑀢𑁂𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀧𑀼𑀭𑀫𑀸𑀕𑀢𑁄;_  
_𑀢𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀁 𑀢𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁𑁇_  


(303.)

7655\. _𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀥𑀺𑀧𑀜𑁆𑀜𑀜𑁆𑀘,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀭𑁄 𑀥𑀫𑁆𑀫𑁂,_  
_𑀧𑀸𑀧𑀼𑀡𑀺𑀁 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(304.)

7656\. _𑀬𑀤𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀘𑀸𑀳𑀁,_  
_𑀬𑀤𑀸 𑀘 𑀅𑀭𑀳𑀸 𑀅𑀳𑀼𑀁;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀁 𑀬𑀤𑀸 𑀘𑀸𑀳𑀁,_  
_𑀕𑀦𑁆𑀥𑀯𑀲𑁆𑀲𑁄 𑀢𑀤𑀸 𑀅𑀳𑀼𑁇_  


(305.)

7657\. _𑀲𑀭𑀻𑀭𑀕𑀦𑁆𑀥𑁄 𑀘 𑀲𑀤𑀸𑀢𑀺𑀲𑁂𑀢𑀺 𑀫𑁂,_  
_𑀫𑀳𑀸𑀭𑀳𑀁 𑀘𑀦𑁆𑀤𑀦𑀘𑀫𑁆𑀧𑀓𑀼𑀧𑁆𑀧𑀮𑀁;_  
_𑀢𑀣𑁂𑀯 𑀕𑀦𑁆𑀥𑁂 𑀇𑀢𑀭𑁂 𑀘 𑀲𑀩𑁆𑀩𑀲𑁄,_  
_𑀧𑀲𑀬𑁆𑀳 𑀯𑀸𑀬𑀸𑀫𑀺 𑀢𑀢𑁄 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁𑁇_  


(306.)

7658\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(307.)

7659\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(308.)

7660\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁬𑁨𑁪𑁫)_  


7661\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀽𑀍𑀅𑀲𑀼𑀕𑀦𑁆𑀥𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

7662\. 𑀘𑀽𑀍𑀅𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

7663\. 𑀪𑀤𑁆𑀤𑀺𑀬𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁄𑁇



---

7664\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



7665\. _𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀭𑁂𑀯𑀢𑁄 𑀣𑁂𑀭𑁄,_  
_𑀫𑀳𑀸𑀮𑀸𑀪𑀻 𑀘 𑀲𑀻𑀯𑀮𑀺;_  
_𑀯𑀗𑁆𑀕𑀻𑀲𑁄 𑀦𑀦𑁆𑀤𑀓𑁄 𑀘𑁂𑀯,_  
_𑀓𑀸𑀍𑀉𑀤𑀸𑀬𑀻 𑀢𑀣𑀸𑀪𑀬𑁄𑁇_  


7666\. _𑀮𑁄𑀫𑀲𑁄 𑀯𑀦𑀯𑀘𑁆𑀙𑁄 𑀘,_  
_𑀲𑀼𑀕𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀤𑀲𑀫𑁄;_  
_𑀢𑀻𑀡𑀺 𑀕𑀸𑀣𑀸𑀲𑀢𑀸 𑀢𑀢𑁆𑀣,_  
_𑀲𑁄𑀍𑀅𑀲𑀸 𑀘 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_  


---

7667\. 𑀅𑀣 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



7668\. _𑀓𑀡𑀺𑀓𑀸𑀭𑀯𑁆𑀳𑀬𑁄 𑀯𑀕𑁆𑀕𑁄,_  
_𑀨𑀮𑀤𑁄 𑀢𑀺𑀡𑀤𑀸𑀬𑀓𑁄;_  
_𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀡𑀺𑀢𑀸 𑀘𑀺𑀫𑀸𑁇_  


7669\. _𑀦𑀯𑀕𑀸𑀣𑀸𑀲𑀢𑀸𑀦𑀻𑀳,_  
_𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀬𑁂𑀯 𑀘;_  
_𑀲𑀧𑀜𑁆𑀜𑀸𑀲𑀁 𑀧𑀜𑁆𑀘𑀲𑀢𑀁,_  
_𑀅𑀧𑀤𑀸𑀦𑀸 𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑁇_  


7670\. _𑀲𑀳 𑀉𑀤𑀸𑀦𑀕𑀸𑀣𑀸𑀳𑀺,_  
_𑀙𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑀫𑀸;_  
_𑀤𑁆𑀯𑁂𑀲𑀢𑀸𑀦𑀺 𑀘 𑀕𑀸𑀣𑀸𑀦𑀁,_  
_𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_
