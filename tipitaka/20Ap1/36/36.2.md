# 𑁩𑁬𑁇𑁨 𑀬𑀯𑀓𑀮𑀸𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀯𑀕𑁆𑀕

* 𑀬𑀯𑀓𑀮𑀸𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(6.)

3977\. _“𑀦𑀕𑀭𑁂 𑀅𑀭𑀼𑀡𑀯𑀢𑀺𑀬𑀸,_  
_𑀆𑀲𑀺𑀁 𑀬𑀯𑀲𑀺𑀓𑁄 𑀢𑀤𑀸;_  
_𑀧𑀦𑁆𑀣𑁂 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀬𑀯𑀓𑀮𑀸𑀧𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀁𑁇_  


(7.)

3978\. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀲𑀺𑀔𑀻 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀬𑀯𑀲𑀦𑁆𑀣𑀭𑁂𑁇_  


(8.)

3979\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀯𑀺𑀫𑀮𑀁,_  
_𑀫𑀳𑀸𑀛𑀸𑀬𑀺𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑀁;_  
_𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀦𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(9.)

3980\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀬𑀯𑀢𑁆𑀣𑀭𑁂 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(10.)

3981\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3982\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀯𑀓𑀮𑀸𑀧𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3983\. 𑀬𑀯𑀓𑀮𑀸𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
