# 𑁩𑁬𑁇𑁧𑁦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(37.)

4024\. _“𑀢𑀺𑀲𑁆𑀲𑁄 𑀦𑀸𑀫𑀸𑀲𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀺𑀳𑀸𑀲𑀺 𑀯𑀺𑀧𑀺𑀦𑁂 𑀢𑀤𑀸;_  
_𑀢𑀼𑀲𑀺𑀢𑀸 𑀳𑀺 𑀇𑀥𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(38.)

4025\. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀫𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀺𑀲𑁆𑀲𑀁 𑀦𑀸𑀫 𑀫𑀳𑀸𑀬𑀲𑀁;_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀼𑀲𑀺𑀢𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁𑁇_  


(39.)

4026\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(40.)

4027\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁩𑁧𑁭𑁩)_  


4028\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4029\. 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

4030\. 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀯𑀕𑁆𑀕𑁄 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁄𑁇



---

4031\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



4032\. _𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀻 𑀬𑀯𑀲𑀺𑀓𑁄,_  
_𑀓𑀺𑀁𑀲𑀼𑀓𑁄𑀭𑀡𑁆𑀟𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄;_  
_𑀆𑀮𑀫𑁆𑀩𑀦𑁄 𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀼,_  
_𑀲𑀼𑀧𑀼𑀝𑀻 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑁄;_  
_𑀲𑀭𑀡𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀘,_  
_𑀕𑀸𑀣𑀸 𑀢𑀸𑀮𑀻𑀲𑀫𑁂𑀯 𑀘𑁇_
