# 𑁩𑁬𑁇𑁧 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3970\. _“𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀦𑀦𑁂 𑀅𑀳𑀁;_  
_𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁𑁇_  


(2.)

3971\. _𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀁 𑀧𑀓𑀸𑀲𑁂𑀦𑁆𑀢𑀁,_  
_𑀉𑀤𑁆𑀥𑀭𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀫𑀥𑀼𑀭𑀁 𑀯𑀸𑀘𑀁,_  
_𑀓𑀭𑀯𑀻𑀓𑀭𑀼𑀤𑁄𑀧𑀫𑀁𑁇_  


(3.)

3972\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑀭𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀲𑀺𑀔𑀺𑀦𑁄 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼𑀦𑁄;_  
_𑀖𑁄𑀲𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(4.)

3973\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀲𑀸𑀤𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(5.)

3974\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3975\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3976\. 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
