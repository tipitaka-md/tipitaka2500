# 𑁭𑁇𑁩 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(13.)

1507\. _“𑀲𑀺𑀦𑁆𑀥𑀼𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀢𑀻𑀭𑁂,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀲𑀼𑀤𑁆𑀥𑀲𑁂𑀯𑀸𑀮𑀪𑀓𑁆𑀔𑁄𑀳𑀁,_  
_𑀧𑀸𑀧𑁂𑀲𑀼 𑀘 𑀲𑀼𑀲𑀜𑁆𑀜𑀢𑁄𑁇_  


(14.)

1508\. _𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂;_  
_𑀢𑀼𑀡𑁆𑀟𑁂𑀦 𑀲𑀸𑀮𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(15.)

1509\. _𑀬𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀢𑀣𑀸𑀕𑀢𑁂,_  
_𑀅𑀘𑀮𑀸 𑀲𑀼𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀲𑁄 𑀦 𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(16.)

1510\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀯𑀺𑀳𑀗𑁆𑀕𑀫𑁂𑀦 𑀲𑀦𑁆𑀢𑁂𑀦,_  
_𑀲𑀼𑀩𑀻𑀚𑀁 𑀭𑁄𑀧𑀺𑀢𑀁 𑀫𑀬𑀸𑁇_  


(17.)

1511\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(18.)

1512\. _𑀲𑀼𑀘𑀸𑀭𑀼𑀤𑀲𑁆𑀲𑀦𑀸 𑀦𑀸𑀫,_  
_𑀅𑀝𑁆𑀞𑁂𑀢𑁂 𑀏𑀓𑀦𑀸𑀫𑀓𑀸;_  
_𑀓𑀧𑁆𑀧𑁂 𑀲𑀢𑁆𑀢𑀭𑀲𑁂 𑀆𑀲𑀼𑀁,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(19.)

1513\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1514\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1515\. 𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
