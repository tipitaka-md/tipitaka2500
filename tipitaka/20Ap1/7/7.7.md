# 𑁭𑁇𑁭 𑀯𑀢𑁆𑀣𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀯𑀢𑁆𑀣𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(45.)

1547\. _“𑀧𑀓𑁆𑀔𑀺𑀚𑀸𑀢𑁄 𑀢𑀤𑀸 𑀆𑀲𑀺𑀁,_  
_𑀲𑀼𑀧𑀡𑁆𑀡𑁄 𑀕𑀭𑀼𑀍𑀆𑀥𑀺𑀧𑁄;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀕𑀦𑁆𑀥𑀫𑀸𑀤𑀦𑀁𑁇_  


(46.)

1548\. _𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀍𑀅𑀯𑀡𑁆𑀡𑀁,_  
_𑀫𑀸𑀡𑀯𑀓𑀁 𑀅𑀥𑀸𑀭𑀬𑀺𑀁;_  
_𑀏𑀓𑀁 𑀯𑀢𑁆𑀣𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(47.)

1549\. _𑀢𑀜𑁆𑀘 𑀤𑀼𑀲𑁆𑀲𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(48.)

1550\. _‘𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀤𑀸𑀦𑁂𑀦,_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀧𑀳𑀸𑀬 𑀕𑀭𑀼𑀍𑀅𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(49.)

1551\. _𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻 𑀢𑀼 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀯𑀢𑁆𑀣𑀤𑀸𑀦𑀁 𑀧𑀲𑀁𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(50.)

1552\. _𑀪𑀯𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑁂 𑀯𑀢𑁆𑀣𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀆𑀓𑀸𑀲𑁂 𑀙𑀤𑀦𑀁 𑀳𑁄𑀢𑀺,_  
_𑀯𑀢𑁆𑀣𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(51.)

1553\. _𑀅𑀭𑀼𑀡𑀯𑀸 𑀲𑀢𑁆𑀢 𑀚𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸;_  
_𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁆𑀳𑀺 𑀆𑀲𑀺𑀁𑀲𑀼,_  
_𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺 𑀫𑀦𑀼𑀚𑀸𑀥𑀺𑀧𑀸𑁇_  


(52.)

1554\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1555\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑀢𑁆𑀣𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1556\. 𑀯𑀢𑁆𑀣𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
