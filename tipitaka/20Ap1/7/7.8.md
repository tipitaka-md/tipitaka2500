# 𑁭𑁇𑁮 𑀅𑀫𑁆𑀩𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀫𑁆𑀩𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(53.)

1557. _“𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂;_  
_𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀅𑀨𑀭𑀺 𑀮𑁄𑀓𑁂,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑁇_  


(54.)

1558. _𑀓𑀧𑀺 𑀅𑀳𑀁 𑀢𑀤𑀸 𑀆𑀲𑀺𑀁,_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑁂 𑀦𑀕𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀁 𑀢𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁𑁇_  


(55.)

1559. _𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀅𑀫𑁆𑀩𑀸𑀲𑀼𑀁 𑀨𑀮𑀺𑀦𑁄 𑀢𑀤𑀸;_  
_𑀢𑀢𑁄 𑀧𑀓𑁆𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀫𑁆𑀩𑀁 𑀲𑀫𑀥𑀼𑀓𑀁 𑀅𑀤𑀁𑁇_  


(56.)

1560. _𑀢𑀁 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀇𑀫𑀺𑀦𑀸 𑀫𑀥𑀼𑀤𑀸𑀦𑁂𑀦,_  
_𑀅𑀫𑁆𑀩𑀤𑀸𑀦𑁂𑀦 𑀘𑀽𑀪𑀬𑀁𑁇_  


(57.)

1561. _𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀯𑀲𑁂𑀲𑁂𑀲𑀼 𑀓𑀧𑁆𑀧𑁂𑀲𑀼,_  
_𑀯𑁄𑀓𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀲𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(58.)

1562. _𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀧𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀺𑀬𑀸;_  
_𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀫𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(59.)

1563. _𑀤𑀫𑁂𑀦 𑀉𑀢𑁆𑀢𑀫𑁂𑀦𑀸𑀳𑀁,_  
_𑀤𑀫𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑀸;_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁𑁇_  


(60.)

1564. _𑀲𑀢𑁆𑀢𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀅𑀫𑁆𑀩𑀝𑁆𑀞𑀚𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀘𑀢𑀼𑀤𑁆𑀤𑀲 𑀢𑁂 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(61.)

1565. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1566. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀫𑁆𑀩𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1567. 𑀅𑀫𑁆𑀩𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
