# 𑁭𑁇𑁧𑁦 𑀧𑀼𑀧𑁆𑀨𑀘𑀗𑁆𑀓𑁄𑀝𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀧𑁆𑀨𑀘𑀗𑁆𑀓𑁄𑀝𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(68.)

1576. _“𑀅𑀪𑀻𑀢𑀭𑀽𑀧𑀁 𑀲𑀻𑀳𑀁𑀯,_  
_𑀕𑀭𑀼𑀍𑀅𑀕𑁆𑀕𑀁𑀯 𑀧𑀓𑁆𑀔𑀺𑀦𑀁;_  
_𑀩𑁆𑀬𑀕𑁆𑀖𑀽𑀲𑀪𑀁𑀯 𑀧𑀯𑀭𑀁,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀁𑀯 𑀓𑁂𑀲𑀭𑀺𑀁𑁇_  


(69.)

1577. _𑀲𑀺𑀔𑀺𑀁 𑀢𑀺𑀮𑁄𑀓𑀲𑀭𑀡𑀁,_  
_𑀅𑀦𑁂𑀚𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀲𑀫𑀡𑀸𑀦𑀕𑁆𑀕𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁𑁇_  


(70.)

1578. _𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀦𑁄𑀚𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀳 𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑁂𑀦𑁂𑀯,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀲𑀫𑁄𑀓𑀺𑀭𑀺𑀁𑁇_  


(71.)

1579. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤 𑀦𑀭𑀸𑀲𑀪;_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁𑁇_  


(72.)

1580. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(73.)

1581. _𑀲𑀫𑁆𑀧𑀼𑀡𑁆𑀡𑁂 𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀤𑁂𑀯𑀪𑀽𑀢𑀺𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀧𑀜𑁆𑀘𑀸𑀲𑀼𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(74.)

1582. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁧𑁪𑁧𑁨)_  


1583. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀧𑁆𑀨𑀘𑀗𑁆𑀓𑁄𑀝𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1584. 𑀧𑀼𑀧𑁆𑀨𑀘𑀗𑁆𑀓𑁄𑀝𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

1585. 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇



---

1586. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1587. _𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀻 𑀅𑀯𑁄𑀧𑀼𑀧𑁆𑀨𑀻,_  
_𑀲𑀧𑀘𑁆𑀘𑀸𑀕𑀫𑀦𑁂𑀦 𑀘;_  
_𑀧𑀭𑀧𑁆𑀧𑀲𑀸𑀤𑀻 𑀪𑀺𑀲𑀤𑁄,_  
_𑀲𑀼𑀘𑀺𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀤𑀸𑀬𑀓𑁄𑁇_  


1588. _𑀅𑀫𑁆𑀩𑀤𑀸𑀬𑀻 𑀘 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀼𑀧𑁆𑀨𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀻𑀧𑀺 𑀘;_  
_𑀕𑀸𑀣𑁂𑀓𑀲𑀢𑁆𑀢𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀕𑀡𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀪𑀺𑁇_
