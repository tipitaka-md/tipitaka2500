# 𑁭𑁇𑁯 𑀲𑀼𑀫𑀦𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀫𑀦𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(62.)

1568\. _“𑀲𑀼𑀫𑀦𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀫𑀸𑀮𑀸𑀓𑀸𑀭𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀮𑁄𑀓𑀸𑀳𑀼𑀢𑀺𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁𑁇_  


(63.)

1569\. _𑀉𑀪𑁄 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀲𑀼𑀫𑀦𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑁂𑀲𑀺𑀁,_  
_𑀲𑀺𑀔𑀺𑀦𑁄 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼𑀦𑁄𑁇_  


(64.)

1570\. _𑀇𑀫𑀸𑀬 𑀧𑀼𑀧𑁆𑀨𑀧𑀽𑀚𑀸𑀬,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(65.)

1571\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(66.)

1572\. _𑀙𑀩𑁆𑀩𑀻𑀲𑀢𑀺𑀫𑁆𑀳𑀺 𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀲𑀼𑀁 𑀫𑀳𑀸𑀬𑀲𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀭𑀸𑀚𑀸𑀦𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(67.)

1573\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1574\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀫𑀦𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1575\. 𑀲𑀼𑀫𑀦𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
