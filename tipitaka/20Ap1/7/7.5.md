# 𑁭𑁇𑁫 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(29.)

1527\. _“𑀯𑁂𑀲𑁆𑀲𑀪𑀽 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀇𑀲𑀻𑀦𑀁 𑀢𑀢𑀺𑀬𑁄 𑀅𑀳𑀼;_  
_𑀓𑀸𑀦𑀦𑀁 𑀯𑀦𑀫𑁄𑀕𑀬𑁆𑀳,_  
_𑀯𑀺𑀳𑀸𑀲𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(30.)

1528\. _𑀪𑀺𑀲𑀫𑀼𑀍𑀆𑀮𑀁 𑀕𑀡𑁆𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀕𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁;_  
_𑀢𑀜𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(31.)

1529\. _𑀓𑀭𑁂𑀦 𑀘 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀽𑀯𑀭𑀩𑀼𑀤𑁆𑀥𑀺𑀦𑀸;_  
_𑀲𑀼𑀔𑀸𑀳𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀫𑀁 𑀢𑁂𑀦 𑀓𑀼𑀢𑁄𑀢𑁆𑀢𑀭𑀺𑀁𑁇_  


(32.)

1530\. _𑀘𑀭𑀺𑀫𑁄 𑀯𑀢𑁆𑀢𑀢𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁂𑀦 𑀲𑀦𑁆𑀢𑁂𑀦,_  
_𑀓𑀼𑀲𑀮𑀁 𑀭𑁄𑀧𑀺𑀢𑀁 𑀫𑀬𑀸𑁇_  


(33.)

1531\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑀺𑀲𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(34.)

1532\. _𑀲𑀫𑁄𑀥𑀸𑀦𑀸 𑀘 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀲𑁄𑀍𑀅𑀲 𑀫𑀦𑀼𑀚𑀸𑀥𑀺𑀧𑀸;_  
_𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺 𑀘𑀼𑀤𑁆𑀤𑀲𑁂 𑀆𑀲𑀼𑀁,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(35.)

1533\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1534\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1535\. 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
