# 𑁭𑁇𑁬 𑀲𑀼𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀦𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(36.)

1536\. _“𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑀘𑀭𑁄 𑀆𑀲𑀺𑀁,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑁄𑀯 𑀓𑁂𑀲𑀭𑀻;_  
_𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀚𑀻𑀯𑀸𑀫𑀺 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂𑁇_  


(37.)

1537\. _𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻 𑀢𑀼 𑀪𑀕𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄;_  
_𑀫𑀫𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀧𑀩𑁆𑀩𑀢𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(38.)

1538\. _𑀧𑀲𑀤𑀜𑁆𑀘 𑀫𑀺𑀕𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀪𑀓𑁆𑀔𑀺𑀢𑀼𑀁 𑀲𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁;_  
_𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑁄 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(39.)

1539\. _𑀯𑀭𑀫𑀁𑀲𑀸𑀦𑀺 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀅𑀦𑀼𑀫𑁄𑀤𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀫𑀫𑀁 𑀢𑀤𑀸𑁇_  


(40.)

1540\. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑀁 𑀧𑀯𑀺𑀲𑀺𑀁 𑀅𑀳𑀁;_  
_𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(41.)

1541\. _𑀏𑀢𑁂𑀦 𑀫𑀁𑀲𑀤𑀸𑀦𑁂𑀦,_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀁 𑀅𑀳𑀁𑁇_  


(42.)

1542\. _𑀅𑀯𑀲𑁂𑀲𑁂𑀲𑀼 𑀓𑀧𑁆𑀧𑁂𑀲𑀼,_  
_𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀁 𑀫𑀬𑀸;_  
_𑀢𑁂𑀦𑁂𑀯 𑀫𑀁𑀲𑀤𑀸𑀦𑁂𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀭𑀡𑁂𑀦 𑀘𑁇_  


(43.)

1543\. _𑀅𑀝𑁆𑀞𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑁆𑀳𑀺 𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀅𑀝𑁆𑀞 𑀤𑀻𑀖𑀸𑀬𑀼𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀤𑀼𑀯𑁂 𑀯𑀭𑀼𑀡𑀦𑀸𑀫𑀓𑀸𑁇_  


(44.)

1544\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1545\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1546\. 𑀲𑀼𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
