# 𑁩𑁨𑁇𑁧𑁦 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(54.)

3577. _“𑀉𑀲𑀪𑀁 𑀧𑀯𑀭𑀁 𑀯𑀻𑀭𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀁 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀺𑀦𑀁;_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀁𑀯 𑀓𑁂𑀲𑀭𑀺𑀁𑁇_  


(55.)

3578. _𑀭𑀣𑀺𑀬𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀼𑀫𑀼𑀤𑀁 𑀫𑀸𑀮𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀲𑀫𑁄𑀓𑀺𑀭𑀺𑀁𑁇_  


(56.)

3579. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(57.)

3580. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁮𑁨𑁦)_  


3581. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3582. 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

3583. 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕𑁄 𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁄𑁇



---

3584. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3585. _𑀆𑀭𑀓𑁆𑀔𑀤𑁄 𑀪𑁄𑀚𑀦𑀤𑁄,_  
_𑀕𑀢𑀲𑀜𑁆𑀜𑀻 𑀧𑀤𑀼𑀫𑀺𑀬𑁄;_  
_𑀧𑀼𑀧𑁆𑀨𑀸𑀲𑀦𑀻 𑀲𑀦𑁆𑀣𑀯𑀺𑀓𑁄,_  
_𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀻 𑀢𑀺𑀭𑀁𑀲𑀺𑀬𑁄;_  
_𑀓𑀦𑁆𑀤𑀮𑀺𑀓𑁄 𑀓𑀼𑀫𑀼𑀤𑀻 𑀘,_  
_𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲 𑀕𑀸𑀣𑀓𑀸𑀢𑀺𑁇_
