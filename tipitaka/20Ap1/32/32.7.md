# 𑁩𑁨𑁇𑁭 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(34.)

3551\. _“𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀤𑁂𑀲𑁂𑀢𑀺 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁;_  
_𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀲𑀸𑀯𑀓𑁂𑀳𑀺,_  
_𑀯𑀲𑀢𑀺 𑀖𑀭𑀫𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(35.)

3552\. _𑀢𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀫𑀥𑀼𑀭𑀸𑀬,_  
_𑀲𑀗𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀖𑁄𑀲𑁄 𑀘 𑀯𑀺𑀧𑀼𑀮𑁄 𑀆𑀲𑀺,_  
_𑀆𑀲𑀻𑀲𑁄 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑁂𑁇_  


(36.)

3553\. _𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀲𑀤𑁆𑀤𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(37.)

3554\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(38.)

3555\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3556\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3557\. 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
