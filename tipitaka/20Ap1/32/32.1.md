# 𑁩𑁨𑁇𑁧 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3506\. _“𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀯𑀢𑀺 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑀸;_  
_𑀆𑀭𑀓𑁆𑀔𑁄 𑀘 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁄,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(2.)

3507\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀲𑁂𑀲𑁂𑀦,_  
_𑀧𑀢𑁆𑀢𑁄 𑀫𑁂 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑁄𑁇_  


(3.)

3508\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3509\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3510\. 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
