# 𑁩𑁨𑁇𑁨 𑀪𑁄𑀚𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀪𑁄𑀚𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(4.)

3511\. _“𑀲𑀼𑀚𑀸𑀢𑁄 𑀲𑀸𑀮𑀮𑀝𑁆𑀞𑀻𑀯,_  
_𑀲𑁄𑀪𑀜𑁆𑀚𑀦𑀫𑀺𑀯𑀼𑀕𑁆𑀕𑀢𑁄;_  
_𑀇𑀦𑁆𑀤𑀮𑀝𑁆𑀞𑀺𑀭𑀺𑀯𑀸𑀓𑀸𑀲𑁂,_  
_𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀲𑀤𑀸 𑀚𑀺𑀦𑁄𑁇_  


(5.)

3512\. _𑀢𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑀲𑁆𑀲,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀅𑀤𑀸𑀲𑀺 𑀪𑁄𑀚𑀦𑀫𑀳𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(6.)

3513\. _𑀢𑀁 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑀼𑀫𑁄𑀤𑀺,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀪𑀯𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀨𑀮𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀽 𑀢𑀯𑁇_  


(7.)

3514\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(8.)

3515\. _𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀅𑀫𑀺𑀢𑁆𑀢𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(9.)

3516\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3517\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑁄𑀚𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3518\. 𑀪𑁄𑀚𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
