# 𑁩𑁨𑁇𑁯 𑀓𑀦𑁆𑀤𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀓𑀦𑁆𑀤𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(47.)

3568\. _“𑀲𑀺𑀦𑁆𑀥𑀼𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀢𑀻𑀭𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀓𑀲𑁆𑀲𑀓𑁄 𑀢𑀤𑀸;_  
_𑀧𑀭𑀓𑀫𑁆𑀫𑀸𑀬𑀦𑁂 𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀭𑀪𑀢𑁆𑀢𑀁 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑁄𑁇_  


(48.)

3569\. _𑀲𑀺𑀦𑁆𑀥𑀼𑀁 𑀅𑀦𑀼𑀘𑀭𑀦𑁆𑀢𑁄𑀳𑀁,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀚𑀺𑀦𑀫𑀤𑁆𑀤𑀲𑀁;_  
_𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑀯,_  
_𑀲𑀢𑀧𑀢𑁆𑀢𑀁𑀯 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁𑁇_  


(49.)

3570\. _𑀲𑀢𑁆𑀢 𑀓𑀦𑁆𑀤𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀯𑀡𑁆𑀝𑁂 𑀙𑁂𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀫𑀢𑁆𑀣𑀓𑁂 𑀅𑀪𑀺𑀭𑁄𑀧𑁂𑀲𑀺𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑁄𑁇_  


(50.)

3571\. _𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀅𑀦𑀼𑀓𑀽𑀮𑁂 𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀢𑀺𑀥𑀸𑀧𑀪𑀺𑀦𑁆𑀦𑀫𑀸𑀢𑀗𑁆𑀕𑀁,_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀁𑀯 𑀤𑀼𑀭𑀸𑀲𑀤𑀁𑁇_  


(51.)

3572\. _𑀢𑀫𑀳𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀧𑀓𑀁 𑀪𑀸𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(52.)

3573\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(53.)

3574\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3575\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀦𑁆𑀤𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3576\. 𑀓𑀦𑁆𑀤𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
