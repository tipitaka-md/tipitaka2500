# 𑁩𑁨𑁇𑁩 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(10.)

3519. _“𑀆𑀓𑀸𑀲𑁂𑀯 𑀧𑀤𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀅𑀫𑁆𑀩𑀭𑁂 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂;_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀚𑀺𑀦𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀢𑀺𑀤𑀺𑀯𑀗𑁆𑀕𑀡𑁂𑁇_  


(11.)

3520. _𑀅𑀦𑀺𑀮𑁂𑀦𑁂𑀭𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀘𑀻𑀯𑀭𑀁;_  
_𑀯𑀺𑀢𑁆𑀢𑀺 𑀫𑀫𑀸𑀳𑀼 𑀢𑀸𑀯𑀤𑁂,_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀕𑀫𑀦𑀁 𑀫𑀼𑀦𑀺𑀁𑁇_  


(12.)

3521. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(13.)

3522. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3523. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3524. 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
