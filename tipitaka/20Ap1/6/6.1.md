# 𑁬𑁇𑁧 𑀯𑀺𑀥𑀽𑀧𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀩𑀻𑀚𑀦𑀺𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀥𑀽𑀧𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

1376\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀩𑀻𑀚𑀦𑀺𑀓𑀸 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(2.)

1377\. _𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀫𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀓𑁆𑀓𑀫𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(3.)

1378\. _𑀩𑀻𑀚𑀦𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀞𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(4.)

1379\. _‘𑀇𑀫𑀺𑀦𑀸 𑀩𑀻𑀚𑀦𑀺𑀤𑀸𑀦𑁂𑀦,_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦 𑀕𑀘𑁆𑀙𑀢𑀺’𑁇_  


(5.)

1380\. _𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄,_  
_𑀘𑁂𑀢𑁄𑀕𑀼𑀡𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑁄𑀳𑀁,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀅𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(6.)

1381\. _𑀲𑀝𑁆𑀞𑀺𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀩𑀻𑀚𑀫𑀸𑀦𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑁄𑀍𑀅𑀲𑀸𑀲𑀺𑀁𑀲𑀼 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(7.)

1382\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1383\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1384\. 𑀯𑀺𑀥𑀽𑀧𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
