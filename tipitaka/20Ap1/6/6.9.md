# 𑁬𑁇𑁯 𑀧𑀤𑀼𑀫𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀩𑀻𑀚𑀦𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀤𑀼𑀫𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(67.)

1458\. _“𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀁 𑀧𑀓𑀸𑀲𑁂𑀦𑁆𑀢𑁄,_  
_𑀯𑀭𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀓𑁄;_  
_𑀯𑀲𑁆𑀲𑀢𑁂 𑀅𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀺𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀚𑀦𑀁𑁇_  


(68.)

1459\. _𑀲𑀥𑀚𑀁 𑀧𑀤𑀼𑀫𑀁 𑀕𑀬𑁆𑀳,_  
_𑀅𑀟𑁆𑀠𑀓𑁄𑀲𑁂 𑀞𑀺𑀢𑁄 𑀅𑀳𑀁;_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀫𑀼𑀦𑀺𑀲𑁆𑀲,_  
_𑀧𑀳𑀝𑁆𑀞𑁄 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀫𑀫𑁆𑀩𑀭𑁂𑁇_  


(69.)

1460\. _𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀘 𑀧𑀤𑀼𑀫𑁂,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀆𑀲𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(70.)

1461\. _𑀓𑀭𑀲𑁂𑀝𑁆𑀞𑁂𑀦 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀚𑀮𑀚𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(71.)

1462\. _‘𑀬𑁂𑀦𑀺𑀤𑀁 𑀧𑀤𑀼𑀫𑀁 𑀔𑀺𑀢𑁆𑀢𑀁,_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀫𑁆𑀳𑀺 𑀯𑀺𑀦𑀸𑀬𑀓𑁂;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(72.)

1463\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀭𑀚𑁆𑀚𑀁 𑀲𑀢𑁆𑀢𑀲𑀢𑀁,_  
_𑀯𑀲𑀼𑀥𑀁 𑀆𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(73.)

1464\. _𑀢𑀢𑁆𑀣 𑀧𑀢𑁆𑀢𑀁 𑀕𑀡𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀆𑀓𑀸𑀲𑀢𑁄 𑀧𑀼𑀧𑁆𑀨𑀯𑀼𑀝𑁆𑀞𑀺,_  
_𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀻 𑀢𑀤𑀸𑁇_  


(74.)

1465\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(75.)

1466\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(76.)

1467\. _𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦 𑀓𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀸,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀚𑀸𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄𑀳𑀁,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀅𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(77.)

1468\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1469\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀤𑀼𑀫𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1470\. 𑀧𑀤𑀼𑀫𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
