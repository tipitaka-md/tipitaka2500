# 𑁬𑁇𑁧𑁦 𑀅𑀲𑀦𑀩𑁄𑀥𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀩𑀻𑀚𑀦𑀺𑀯𑀕𑁆𑀕

* 𑀅𑀲𑀦𑀩𑁄𑀥𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(78.)

1471\. _“𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑁄𑀳𑀁,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(79.)

1472\. _𑀢𑀺𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀭𑁄𑀧𑀬𑀺𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(80.)

1473\. _𑀅𑀲𑀦𑁄 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑁂𑀦,_  
_𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀧𑀸𑀤𑀧𑁄;_  
_𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁂 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀅𑀲𑀦𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(81.)

1474\. _𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀧𑀸𑀤𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀲𑀓𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(82.)

1475\. _𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀤𑀸 𑀲𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(83.)

1476\. _‘𑀬𑁂𑀦𑀸𑀬𑀁 𑀭𑁄𑀧𑀺𑀢𑀸 𑀩𑁄𑀥𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸 𑀘 𑀲𑀓𑁆𑀓𑀢𑀸;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(84.)

1477\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀘𑀢𑀼𑀲𑀝𑁆𑀞𑀺 𑀘𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀲𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(85.)

1478\. _𑀢𑀼𑀲𑀺𑀢𑀸 𑀳𑀺 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀤𑁆𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(86.)

1479\. _𑀧𑀥𑀸𑀦𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀲𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(87.)

1480\. _𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑀳𑀁,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(88.)

1481\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁄𑀥𑀺𑀁 𑀭𑁄𑀧𑁂𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑁄𑀥𑀺𑀭𑁄𑀧𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(89.)

1482\. _𑀘𑀢𑀼𑀲𑀢𑁆𑀢𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀤𑀡𑁆𑀟𑀲𑁂𑀦𑁄𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀢𑀤𑀸 𑀅𑀳𑀼𑀁𑁇_  


(90.)

1483\. _𑀢𑁂𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑁂𑀲𑀼𑀁 𑀫𑀳𑀻𑀧𑀢𑀻;_  
_𑀲𑀫𑀦𑁆𑀢𑀦𑁂𑀫𑀺𑀦𑀸𑀫𑁂𑀦,_  
_𑀭𑀸𑀚𑀸𑀦𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(91.)

1484\. _𑀧𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀼𑀡𑁆𑀡𑀓𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(92.)

1485\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁧𑁩𑁩𑁮)_  


1486\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑀦𑀩𑁄𑀥𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1487\. 𑀅𑀲𑀦𑀩𑁄𑀥𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

1488\. 𑀩𑀻𑀚𑀦𑀺𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇



---

1489\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1490\. _𑀩𑀻𑀚𑀦𑀻 𑀲𑀢𑀭𑀁𑀲𑀻 𑀘,_  
_𑀲𑀬𑀦𑁄𑀤𑀓𑀺𑀯𑀸𑀳𑀺𑀬𑁄;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁄 𑀧𑀤𑀻𑀧𑀜𑁆𑀘,_  
_𑀥𑀚𑁄 𑀧𑀤𑀼𑀫𑀧𑀽𑀚𑀓𑁄;_  
_𑀩𑁄𑀥𑀺 𑀘 𑀤𑀲𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄,_  
_𑀕𑀸𑀣𑀸 𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑀺 𑀢𑀣𑀸𑁇_
