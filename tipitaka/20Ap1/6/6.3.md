# 𑁬𑁇𑁩 𑀲𑀬𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑀻𑀚𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀲𑀬𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(20.)

1399\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄;_  
_𑀲𑀬𑀦𑀁 𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(21.)

1400\. _𑀢𑁂𑀦 𑀲𑀬𑀦𑀤𑀸𑀦𑁂𑀦,_  
_𑀲𑀼𑀔𑁂𑀢𑁆𑀢𑁂 𑀩𑀻𑀚𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀪𑁄𑀕𑀸 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀭𑁂 𑀢𑀲𑁆𑀲,_  
_𑀲𑀬𑀦𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(22.)

1401\. _𑀆𑀓𑀸𑀲𑁂 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀫𑀺,_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀧𑀣𑀯𑀺𑀁 𑀇𑀫𑀁;_  
_𑀧𑀸𑀡𑁂𑀲𑀼 𑀫𑁂 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁,_  
_𑀲𑀬𑀦𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(23.)

1402\. _𑀧𑀜𑁆𑀘𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀅𑀝𑁆𑀞 𑀆𑀲𑀼𑀁 𑀫𑀳𑀸𑀢𑁂𑀚𑀸;_  
_𑀘𑀢𑀼𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀘𑀢𑀼𑀭𑁄 𑀘 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(24.)

1403\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1404\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀬𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1405\. 𑀲𑀬𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
