# 𑁬𑁇𑁭 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀩𑀻𑀚𑀦𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(50.)

1437\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄;_  
_𑀲𑀤𑁆𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀉𑀚𑀼𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(51.)

1438\. _𑀧𑀤𑀻𑀧𑀤𑀸𑀦𑀁 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸𑀦 𑀩𑁄𑀥𑀺𑀬𑀁;_  
_𑀲𑀤𑁆𑀤𑀳𑀦𑁆𑀢𑁄 𑀧𑀤𑀻𑀧𑀸𑀦𑀺,_  
_𑀅𑀓𑀭𑀺𑀁 𑀢𑀸𑀯𑀤𑁂 𑀅𑀳𑀁𑁇_  


(52.)

1439\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀓𑁆𑀓𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(53.)

1440\. _𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀲𑁂𑀮𑀁,_  
_𑀲𑀫𑀢𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀼𑀪𑁄𑀫𑀳𑀁𑁇_  


(54.)

1441\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀯𑀲𑁂𑀲𑁂𑀦,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁;_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


(55.)

1442\. _𑀘𑀢𑀼𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀲𑀢𑀘𑀓𑁆𑀔𑀼𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀭𑀸𑀚𑀸𑀳𑁂𑀲𑀼𑀁 𑀫𑀳𑀸𑀢𑁂𑀚𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(56.)

1443\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1444\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1445\. 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
