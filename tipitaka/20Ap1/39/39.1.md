# 𑁩𑁯𑁇𑁧 𑀅𑀯𑀝𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

### 𑀅𑀯𑀝𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

4177. _“𑀲𑀢𑀭𑀁𑀲𑀺 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀯𑀺𑀯𑁂𑀓𑀓𑀸𑀫𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑁄𑀘𑀭𑀸𑀬𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑁇_  


(2.)

4178. _𑀨𑀮𑀳𑀢𑁆𑀣𑁄 𑀅𑀳𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀅𑀯𑀝𑀁 𑀅𑀤𑀤𑀺𑀁 𑀨𑀮𑀁𑁇_  


(3.)

4179. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀫𑀤𑀤𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(4.)

4180. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(5.)

4181. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(6.)

4182. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4183. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀯𑀝𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4184. 𑀅𑀯𑀝𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
