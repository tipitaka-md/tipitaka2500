# 𑁩𑁯𑁇𑁧𑁦𑁇𑁪

* 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(78.)

4272\. _𑀯𑁂𑀫𑀸𑀢𑀼𑀪𑀸𑀢𑀭𑀁 𑀧𑀼𑀩𑁆𑀩𑁂,_  
_𑀥𑀦𑀳𑁂𑀢𑀼 𑀳𑀦𑀺𑀁 𑀅𑀳𑀁;_  
_𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀁 𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀺𑀮𑀸𑀬 𑀘 𑀅𑀧𑀺𑀁𑀲𑀬𑀺𑀁𑁇_  


(79.)

4273\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀺𑀮𑀁 𑀔𑀺𑀧𑀺;_  
_𑀅𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑀁 𑀧𑀺𑀁𑀲𑀬𑀻 𑀧𑀸𑀤𑁂,_  
_𑀫𑀫 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀸𑁇_
