# 𑁩𑁯𑁇𑁧𑁦𑁇𑁫

### 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(80.)

4274. _𑀧𑀼𑀭𑁂𑀳𑀁 𑀤𑀸𑀭𑀓𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀓𑀻𑀍𑀅𑀫𑀸𑀦𑁄 𑀫𑀳𑀸𑀧𑀣𑁂;_  
_𑀧𑀘𑁆𑀘𑁂𑀓𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀫𑀕𑁆𑀕𑁂 𑀲𑀓𑀮𑀺𑀓𑀁 𑀔𑀺𑀧𑀺𑀁𑁇_  


(81.)

4275. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀇𑀥 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑁂 𑀪𑀯𑁂;_  
_𑀯𑀥𑀢𑁆𑀣𑀁 𑀫𑀁 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄,_  
_𑀅𑀪𑀺𑀫𑀸𑀭𑁂 𑀧𑀬𑁄𑀚𑀬𑀺𑁇_
