# 𑁩𑁯𑁇𑁧𑁦𑁇𑁮

### 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(86.)

4280. _𑀅𑀳𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀕𑀸𑀫𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀳𑀼𑀁 𑀓𑁂𑀯𑀝𑁆𑀝𑀤𑀸𑀭𑀓𑁄;_  
_𑀫𑀘𑁆𑀙𑀓𑁂 𑀖𑀸𑀢𑀺𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀚𑀦𑀬𑀺𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀓𑀁𑁇_  


(87.)

4281. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀲𑀻𑀲𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀳𑀽 𑀫𑀫;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀓𑁆𑀓𑀸 𑀘 𑀳𑀜𑁆𑀜𑀺𑀁𑀲𑀼,_  
_𑀬𑀤𑀸 𑀳𑀦𑀺 𑀯𑀺𑀝𑀝𑀽𑀪𑁄𑁇_
