# 𑁩𑁯𑁇𑁧𑁦𑁇𑁧𑁨

### 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(92.)

4286\. _𑀅𑀯𑀘𑀸𑀳𑀁 𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁄,_  
_𑀲𑀼𑀕𑀢𑀁 𑀓𑀲𑁆𑀲𑀧𑀁 𑀢𑀤𑀸;_  
_𑀓𑀼𑀢𑁄 𑀦𑀼 𑀩𑁄𑀥𑀺 𑀫𑀼𑀡𑁆𑀟𑀲𑁆𑀲,_  
_𑀩𑁄𑀥𑀺 𑀧𑀭𑀫𑀤𑀼𑀮𑁆𑀮𑀪𑀸𑁇_  


(93.)

4287\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀅𑀘𑀭𑀺𑀁 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀩𑀳𑀼𑀁;_  
_𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀼𑀭𑀼𑀯𑁂𑀮𑀸𑀬𑀁,_  
_𑀢𑀢𑁄 𑀩𑁄𑀥𑀺𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(94.)

4288\. _𑀦𑀸𑀳𑀁 𑀏𑀢𑁂𑀦 𑀫𑀕𑁆𑀕𑁂𑀦,_  
_𑀧𑀸𑀧𑀼𑀡𑀺𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑁂𑀦 𑀕𑀯𑁂𑀲𑀺𑀲𑁆𑀲𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀦 𑀯𑀸𑀭𑀺𑀢𑁄𑁇_  


(95.)

4289\. _𑀧𑀼𑀜𑁆𑀜𑀧𑀸𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑀚𑁆𑀚𑀺𑀢𑁄;_  
_𑀅𑀲𑁄𑀓𑁄 𑀅𑀦𑀼𑀧𑀸𑀬𑀸𑀲𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀫𑀦𑀸𑀲𑀯𑁄”𑁇_  


(96.)

4290\. _𑀏𑀯𑀁 𑀚𑀺𑀦𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀅𑀦𑁄𑀢𑀢𑁆𑀢𑁂 𑀫𑀳𑀸𑀲𑀭𑁂𑀢𑀺𑁇 (𑁩𑁩𑁬𑁭)_  


4291\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀼𑀩𑁆𑀩𑀘𑀭𑀺𑀢𑀁 𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀁 𑀦𑀸𑀫 𑀩𑀼𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4292\. 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀁 𑀦𑀸𑀫 𑀩𑀼𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

4293\. 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕𑁄 𑀏𑀓𑀽𑀦𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁄𑁇



---

4294\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



4295\. _𑀅𑀯𑀝𑀁 𑀮𑀩𑀼𑀚𑀜𑁆𑀘𑁂𑀯,_  
_𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀧𑀺𑀮𑀓𑁆𑀔𑀼 𑀘;_  
_𑀨𑀸𑀭𑀼 𑀯𑀮𑁆𑀮𑀻 𑀘 𑀓𑀤𑀮𑀻,_  
_𑀧𑀦𑀲𑁄 𑀓𑁄𑀝𑀺𑀯𑀻𑀲𑀓𑁄𑁇_  


4296\. _𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺 𑀘,_  
_𑀅𑀧𑀤𑀸𑀦𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀏𑀓𑀦𑀯𑀼𑀢𑀺,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_  


---

4297\. 𑀘𑀼𑀤𑁆𑀤𑀲𑀫𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
