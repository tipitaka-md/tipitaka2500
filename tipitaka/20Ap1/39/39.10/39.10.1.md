# 𑁩𑁯𑁇𑁧𑁦𑁇𑁧

### 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(67.)

4261\. _𑀫𑀼𑀦𑀸𑀍𑀇 𑀦𑀸𑀫𑀳𑀁 𑀥𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀜𑁆𑀜𑀸𑀲𑀼 𑀚𑀸𑀢𑀺𑀲𑀼;_  
_𑀧𑀘𑁆𑀘𑁂𑀓𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀼𑀭𑀪𑀺𑀁,_  
_𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺𑀁 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(68.)

4262\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀦𑀺𑀭𑀬𑁂 𑀲𑀁𑀲𑀭𑀺𑀁 𑀘𑀺𑀭𑀁;_  
_𑀩𑀳𑀽𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑁂𑀤𑁂𑀲𑀺 𑀯𑁂𑀤𑀦𑀁𑁇_  


(69.)

4263\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀯𑀲𑁂𑀲𑁂𑀦,_  
_𑀇𑀥 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑁂 𑀪𑀯𑁂;_  
_𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀫𑀬𑀸 𑀮𑀤𑁆𑀥𑀁,_  
_𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀸𑀬 𑀓𑀸𑀭𑀡𑀸𑁇_
