# 𑁩𑁯𑁇𑁧𑁦𑁇𑁭

* 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(84.)

4278\. _𑀭𑀸𑀚𑀸𑀳𑀁 𑀧𑀢𑁆𑀣𑀺𑀯𑁄 𑀆𑀲𑀺𑀁,_  
_𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀁 𑀳𑀦𑀺𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀺𑀲𑀁 𑀪𑀼𑀲𑀁𑁇_  


(85.)

4279\. _𑀓𑀫𑁆𑀫𑀼𑀦𑁄 𑀢𑀲𑁆𑀲 𑀲𑁂𑀲𑁂𑀦,_  
_𑀇𑀤𑀸𑀦𑀺 𑀲𑀓𑀮𑀁 𑀫𑀫;_  
_𑀧𑀸𑀤𑁂 𑀙𑀯𑀺𑀁 𑀧𑀓𑀧𑁆𑀧𑁂𑀲𑀺,_  
_𑀦 𑀳𑀺 𑀓𑀫𑁆𑀫𑀁 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺𑁇_
