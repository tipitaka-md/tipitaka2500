# 𑁩𑁯𑁇𑁧𑁦𑁇𑁯

* 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(88.)

4282\. _𑀨𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑀸𑀳𑀁 𑀧𑀸𑀯𑀘𑀦𑁂,_  
_𑀲𑀸𑀯𑀓𑁂 𑀧𑀭𑀺𑀪𑀸𑀲𑀬𑀺𑀁;_  
_𑀬𑀯𑀁 𑀔𑀸𑀤𑀣 𑀪𑀼𑀜𑁆𑀚𑀣,_  
_𑀫𑀸 𑀘 𑀪𑀼𑀜𑁆𑀚𑀣 𑀲𑀸𑀮𑀬𑁄𑁇_  


(89.)

4283\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀢𑁂𑀫𑀸𑀲𑀁 𑀔𑀸𑀤𑀺𑀢𑀁 𑀬𑀯𑀁;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀦,_  
_𑀯𑁂𑀭𑀜𑁆𑀚𑀸𑀬𑀁 𑀯𑀲𑀺𑀁 𑀢𑀤𑀸𑁇_
