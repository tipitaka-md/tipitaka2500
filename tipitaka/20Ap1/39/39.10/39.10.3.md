# 𑁩𑁯𑁇𑁧𑁦𑁇𑁩

### 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀺𑀮𑁄𑀢𑀺𑀓𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(73.)

4267\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀼𑀢𑀯𑀸 𑀆𑀲𑀺𑀁,_  
_𑀅𑀳𑀁 𑀲𑀓𑁆𑀓𑀢𑀧𑀽𑀚𑀺𑀢𑁄;_  
_𑀫𑀳𑀸𑀯𑀦𑁂 𑀧𑀜𑁆𑀘𑀲𑀢𑁂,_  
_𑀫𑀦𑁆𑀢𑁂 𑀯𑀸𑀘𑁂𑀫𑀺 𑀫𑀸𑀡𑀯𑁂𑁇_  


(74.)

4268\. _𑀢𑀢𑁆𑀣𑀸𑀕𑀢𑁄 𑀇𑀲𑀺 𑀪𑀻𑀫𑁄,_  
_𑀧𑀜𑁆𑀘𑀸𑀪𑀺𑀜𑁆𑀜𑁄 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄;_  
_𑀢𑀜𑁆𑀘𑀸𑀳𑀁 𑀆𑀕𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀺𑀁 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(75.)

4269\. _𑀢𑀢𑁄𑀳𑀁 𑀅𑀯𑀘𑀁 𑀲𑀺𑀲𑁆𑀲𑁂,_  
_𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀅𑀬𑀁 𑀇𑀲𑀺;_  
_𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀼 𑀫𑀸𑀡𑀯𑀸𑁇_  


(76.)

4270\. _𑀢𑀢𑁄 𑀫𑀸𑀡𑀯𑀓𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀓𑀼𑀮𑁂 𑀓𑀼𑀮𑁂;_  
_𑀫𑀳𑀸𑀚𑀦𑀲𑁆𑀲 𑀆𑀳𑀁𑀲𑀼,_  
_𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀅𑀬𑀁 𑀇𑀲𑀺𑁇_  


(77.)

4271\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀧𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸 𑀇𑀫𑁂;_  
_𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀮𑀪𑀼𑀁 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀸𑀬 𑀓𑀸𑀭𑀡𑀸𑁇_
