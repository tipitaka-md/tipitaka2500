# 𑁩𑁯𑁇𑁪 𑀧𑀺𑀮𑀓𑁆𑀔𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀓𑁆𑀔𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(21.)

4203\. _“𑀯𑀦𑀦𑁆𑀢𑀭𑁂 𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀁 𑀫𑀳𑀸𑀬𑀲𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀺𑀮𑀓𑁆𑀔𑀲𑁆𑀲𑀸𑀤𑀤𑀺𑀁 𑀨𑀮𑀁𑁇_  


(22.)

4204\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀨𑀮𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(23.)

4205\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(24.)

4206\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(25.)

4207\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4208\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀓𑁆𑀔𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4209\. 𑀧𑀺𑀮𑀓𑁆𑀔𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
