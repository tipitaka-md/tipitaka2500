# 𑁩𑁯𑁇𑁨 𑀮𑀩𑀼𑀚𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀅𑀯𑀝𑀨𑀮𑀯𑀕𑁆𑀕

### 𑀮𑀩𑀼𑀚𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(7.)

4185. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀆𑀲𑀺𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(8.)

4186. _𑀮𑀩𑀼𑀚𑀲𑁆𑀲 𑀨𑀮𑀁 𑀕𑀬𑁆𑀳,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁;_  
_𑀆𑀓𑀸𑀲𑁂 𑀞𑀺𑀢𑀓𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳𑀺 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(9.)

4187. _𑀯𑀺𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀚𑀦𑀦𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀸𑀯𑀳𑀁;_  
_𑀨𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(10.)

4188. _𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀺𑀁 𑀢𑀤𑀸 𑀧𑀻𑀢𑀺𑀁,_  
_𑀯𑀺𑀧𑀼𑀮𑀜𑁆𑀘 𑀲𑀼𑀔𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑁂𑀯 𑀭𑀢𑀦𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁𑁇_  


(11.)

4189. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(12.)

4190. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(13.)

4191. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(14.)

4192. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4193. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀩𑀼𑀚𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4194. 𑀮𑀩𑀼𑀚𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
