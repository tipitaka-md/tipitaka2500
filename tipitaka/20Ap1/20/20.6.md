# 𑁨𑁦𑁇𑁬 𑀓𑀺𑀲𑀮𑀬𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀓𑀺𑀲𑀮𑀬𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(28.)

2703. _“𑀦𑀕𑀭𑁂 𑀤𑁆𑀯𑀸𑀭𑀯𑀢𑀺𑀬𑀸,_  
_𑀫𑀸𑀮𑀸𑀯𑀘𑁆𑀙𑁄 𑀫𑀫𑀁 𑀅𑀳𑀼;_  
_𑀉𑀤𑀧𑀸𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣𑁂𑀯,_  
_𑀧𑀸𑀤𑀧𑀸𑀦𑀁 𑀯𑀺𑀭𑁄𑀳𑀦𑁄𑁇_  


(29.)

2704. _𑀲𑀩𑀮𑁂𑀦 𑀉𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑁄,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀫𑀫𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀕𑀘𑁆𑀙𑀢𑁂 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(30.)

2705. _𑀅𑀜𑁆𑀜𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀧𑀽𑀚𑀸𑀬𑁄𑀕𑁆𑀕𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀅𑀲𑁄𑀓𑀁 𑀧𑀮𑁆𑀮𑀯𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁 𑀅𑀳𑀁𑁇_  


(31.)

2706. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀢𑁂 𑀓𑀺𑀲𑀮𑀬𑀸,_  
_𑀕𑀘𑁆𑀙𑀢𑁄 𑀬𑀦𑁆𑀢𑀺 𑀧𑀘𑁆𑀙𑀢𑁄;_  
_𑀢𑀸𑀳𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀁𑀯𑀺𑀚𑀺𑀁,_  
_𑀅𑀳𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀼𑀍𑀆𑀭𑀢𑀸𑁇_  


(32.)

2707. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀮𑁆𑀮𑀯𑀁 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(33.)

2708. _𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀏𑀓𑀺𑀲𑁆𑀲𑀭𑁄 𑀅𑀳𑀼;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(34.)

2709. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2710. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀺𑀲𑀮𑀬𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2711. 𑀓𑀺𑀲𑀮𑀬𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
