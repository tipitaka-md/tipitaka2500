# 𑁨𑁦𑁇𑁧 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

2666\. _“𑀘𑀼𑀮𑁆𑀮𑀸𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀣𑀫𑁆𑀪𑀸 𑀲𑁄𑀯𑀡𑁆𑀡𑀬𑀸 𑀅𑀳𑀽;_  
_𑀤𑁂𑀯𑀮𑀝𑁆𑀞𑀺𑀧𑀝𑀺𑀪𑀸𑀕𑀁,_  
_𑀯𑀺𑀫𑀸𑀦𑀁 𑀫𑁂 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁𑁇_  


(2.)

2667\. _𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁,_  
_𑀲𑀺𑀔𑀺𑀦𑁄 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼𑀦𑁄𑁇_  


(3.)

2668\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(4.)

2669\. _𑀇𑀢𑁄 𑀯𑀻𑀲𑀢𑀺𑀫𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀦𑁆𑀤𑀢𑀺𑀢𑁆𑀢𑁄𑀢𑀺 𑀏𑀓𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(5.)

2670\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2671\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2672\. 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
