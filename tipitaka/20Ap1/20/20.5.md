# 𑁨𑁦𑁇𑁫 𑀅𑀗𑁆𑀓𑁄𑀮𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀗𑁆𑀓𑁄𑀮𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(23.)

2696\. _“𑀅𑀗𑁆𑀓𑁄𑀮𑀁 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀫𑀸𑀮𑀸𑀯𑀭𑀁 𑀲𑀓𑁄𑀲𑀓𑀁;_  
_𑀑𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀅𑀕𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(24.)

2697\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀧𑀢𑀺𑀮𑀻𑀦𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀕𑀼𑀳𑀸𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑁄𑀓𑀺𑀭𑀺𑀁𑁇_  


(25.)

2698\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(26.)

2699\. _𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑁆𑀳𑀺 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀆𑀲𑁂𑀓𑁄 𑀤𑁂𑀯𑀕𑀚𑁆𑀚𑀺𑀢𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(27.)

2700\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2701\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀗𑁆𑀓𑁄𑀮𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2702\. 𑀅𑀗𑁆𑀓𑁄𑀮𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
