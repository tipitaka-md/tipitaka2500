# 𑁨𑁦𑁇𑁪 𑀅𑀲𑁄𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀢𑀫𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀅𑀲𑁄𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(17.)

2688\. _“𑀢𑀺𑀯𑀭𑀸𑀬𑀁 𑀧𑀼𑀭𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀭𑀸𑀚𑀼𑀬𑁆𑀬𑀸𑀦𑀁 𑀅𑀳𑀼 𑀢𑀤𑀸;_  
_𑀉𑀬𑁆𑀬𑀸𑀦𑀧𑀸𑀮𑁄 𑀢𑀢𑁆𑀣𑀸𑀲𑀺𑀁,_  
_𑀭𑀜𑁆𑀜𑁄 𑀩𑀤𑁆𑀥𑀘𑀭𑁄 𑀅𑀳𑀁𑁇_  


(18.)

2689\. _𑀧𑀤𑀼𑀫𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀲𑀧𑁆𑀧𑀪𑁄 𑀅𑀳𑀼;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀫𑁆𑀳𑀺,_  
_𑀙𑀸𑀬𑀸 𑀦 𑀚𑀳𑀺 𑀢𑀁 𑀫𑀼𑀦𑀺𑀁𑁇_  


(19.)

2690\. _𑀅𑀲𑁄𑀓𑀁 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀺𑀡𑁆𑀟𑀺𑀪𑀸𑀭𑀁 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑁂𑀲𑀺𑀁,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀺𑀦𑁄𑁇_  


(20.)

2691\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(21.)

2692\. _𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑁄𑀍𑀅𑀲 𑀅𑀭𑀡𑀜𑁆𑀚𑀳𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(22.)

2693\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2694\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑁄𑀓𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2695\. 𑀅𑀲𑁄𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
