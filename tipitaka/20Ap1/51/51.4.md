# 𑁫𑁧𑁇𑁪 𑀅𑀯𑀝𑀨𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀡𑀺𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀅𑀯𑀝𑀨𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(57.)

6630\. _“𑀲𑀳𑀲𑁆𑀲𑀭𑀁𑀲𑀻 𑀪𑀕𑀯𑀸,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀯𑀺𑀯𑁂𑀓𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑁄𑀘𑀭𑀸𑀬𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑁇_  


(58.)

6631\. _𑀨𑀮𑀳𑀢𑁆𑀣𑁄 𑀅𑀳𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀅𑀯𑀝𑀁 𑀅𑀤𑀤𑀺𑀁 𑀨𑀮𑀁𑁇_  


(59.)

6632\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(60.)

6633\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(61.)

6634\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(62.)

6635\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6636\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀯𑀝𑀨𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6637\. 𑀅𑀯𑀝𑀨𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
