# 𑁫𑁧𑁇𑁧𑁦 𑀦𑀸𑀍𑀇𑀓𑁂𑀭𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀡𑀺𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀦𑀸𑀍𑀇𑀓𑁂𑀭𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(91.)

6676\. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(92.)

6677\. _𑀦𑀸𑀍𑀇𑀓𑁂𑀭𑀨𑀮𑀁 𑀕𑀬𑁆𑀳,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁;_  
_𑀆𑀓𑀸𑀲𑁂 𑀞𑀺𑀢𑀓𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀺 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(93.)

6678\. _𑀯𑀺𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀚𑀦𑀦𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀸𑀯𑀳𑁄;_  
_𑀨𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(94.)

6679\. _𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀺𑀁 𑀢𑀤𑀸 𑀧𑀻𑀢𑀺𑀁,_  
_𑀯𑀺𑀧𑀼𑀮𑀜𑁆𑀘 𑀲𑀼𑀔𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑁂𑀯 𑀭𑀢𑀦𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁𑁇_  


(95.)

6680\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(96.)

6681\. _𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀫𑁂,_  
_𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑁄 𑀅𑀳𑀁;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(97.)

6682\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(98.)

6683\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(99.)

6684\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁫𑁩𑁬𑁬)_  


6685\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀍𑀇𑀓𑁂𑀭𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6686\. 𑀦𑀸𑀍𑀇𑀓𑁂𑀭𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

6687\. 𑀓𑀡𑀺𑀓𑀸𑀭𑀯𑀕𑁆𑀕𑁄 𑀏𑀓𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁄𑁇



---

6688\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



6689\. _𑀓𑀡𑀺𑀓𑀸𑀭𑁂𑀓𑀧𑀢𑁆𑀢𑀸 𑀘,_  
_𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀻 𑀢𑀣𑀸𑀯𑀝𑀸;_  
_𑀧𑀸𑀤𑀜𑁆𑀘 𑀫𑀸𑀢𑀼𑀮𑀼𑀗𑁆𑀕𑀜𑁆𑀘,_  
_𑀅𑀚𑁂𑀮𑀻𑀫𑁄𑀤𑀫𑁂𑀯 𑀘𑁇_  


6690\. _𑀢𑀸𑀮𑀁 𑀢𑀣𑀸 𑀦𑀸𑀍𑀇𑀓𑁂𑀭𑀁,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀡𑀺𑀢𑀸 𑀯𑀺𑀳;_  
_𑀏𑀓𑀁 𑀕𑀸𑀣𑀸𑀲𑀢𑀁 𑀳𑁄𑀢𑀺,_  
_𑀊𑀦𑀸𑀥𑀺𑀓𑀯𑀺𑀯𑀚𑁆𑀚𑀺𑀢𑀁𑁇_
