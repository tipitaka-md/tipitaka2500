# 𑁨𑁇𑁭 𑀉𑀧𑀲𑁂𑀦𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀲𑁂𑀦𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(86.)

793\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑀫𑁆𑀳𑀺 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀁,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(87.)

794\. _𑀓𑀡𑀺𑀓𑀸𑀭𑀧𑀼𑀧𑁆𑀨𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀡𑁆𑀝𑁂 𑀙𑁂𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀮𑀗𑁆𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀙𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(88.)

795\. _𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀜𑁆𑀘 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀁 𑀲𑀼𑀪𑁄𑀚𑀦𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀦𑀯𑀫𑁂 𑀢𑀢𑁆𑀣,_  
_𑀲𑀫𑀡𑁂 𑀅𑀝𑁆𑀞 𑀪𑁄𑀚𑀬𑀺𑀁𑁇_  


(89.)

796\. _𑀅𑀦𑀼𑀫𑁄𑀤𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀇𑀫𑀺𑀦𑀸 𑀙𑀢𑁆𑀢𑀤𑀸𑀦𑁂𑀦,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀧𑀯𑁂𑀘𑁆𑀙𑀦𑀸𑁇_  


(90.)

797\. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀫𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀲𑀺;_  
_𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(91.)

798\. _𑀏𑀓𑀯𑀻𑀲𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(92.)

799\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(93.)

800\. _𑀲𑀸𑀲𑀦𑁂 𑀤𑀺𑀩𑁆𑀩𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄𑁇_  


(94.)

801\. _𑀉𑀧𑀲𑁂𑀦𑁄𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄;_  
_𑀲𑀫𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀢𑁆𑀢𑀸,_  
_𑀅𑀕𑁆𑀕𑀝𑁆𑀞𑀸𑀦𑁂 𑀞𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇_  


(95.)

802\. _𑀘𑀭𑀺𑀫𑀁 𑀯𑀢𑁆𑀢𑀢𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀁𑁇_  


(96.)

803\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


804\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀲𑁂𑀦𑁄 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

805\. 𑀉𑀧𑀲𑁂𑀦𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
