# 𑁨𑁇𑁧𑁦 𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(124.)

837\. _“𑀆𑀳𑀼𑀢𑀺𑀁 𑀬𑀺𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑁄𑀳𑀁,_  
_𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀢𑁆𑀯𑀸𑀦 𑀪𑁄𑀚𑀦𑀁;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀧𑀝𑀺𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀲𑀸𑀮𑁂 𑀫𑀸𑀍𑀅𑀓𑁂 𑀞𑀺𑀢𑁄𑁇_  


(125.)

838\. _𑀅𑀣𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀺𑀁 𑀫𑀳𑀸𑀬𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀯𑀺𑀦𑁂𑀢𑀸𑀭𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑀁𑁇_  


(126.)

839\. _𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑀁,_  
_𑀲𑀸𑀯𑀓𑁂𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁;_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑀫𑀺𑀯 𑀭𑁄𑀘𑀦𑁆𑀢𑀁,_  
_𑀭𑀣𑀺𑀬𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀓𑀁𑁇_  


(127.)

840\. _𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀫𑀦𑀲𑀸𑀯 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀲𑀺𑀁,_  
_‘𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀫𑀳𑀸𑀫𑀼𑀦𑀺’𑁇_  


(128.)

841\. _𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀫𑀫 𑀤𑁆𑀯𑀸𑀭𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(129.)

842\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀧𑀸𑀲𑀸𑀤𑀁 𑀅𑀪𑀺𑀭𑀽𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀁𑁇_  


(130.)

843\. _𑀤𑀦𑁆𑀢𑁄 𑀤𑀦𑁆𑀢𑀧𑀭𑀺𑀯𑀸𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑁄 𑀢𑀸𑀭𑀬𑀢𑀁 𑀯𑀭𑁄;_  
_𑀧𑀸𑀲𑀸𑀤𑀁 𑀅𑀪𑀺𑀭𑀽𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀯𑀭𑀸𑀲𑀦𑁂𑁇_  


(131.)

844\. _𑀬𑀁 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺 𑀲𑀓𑁂 𑀕𑁂𑀳𑁂,_  
_𑀆𑀫𑀺𑀲𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀢𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(132.)

845\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀯𑁂𑀤𑀚𑀸𑀢𑁄 𑀓𑀢𑀜𑁆𑀚𑀮𑀻;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀅𑀳𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀼𑀍𑀆𑀭𑀢𑀸𑁇_  


(133.)

846\. _𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀧𑀬𑀺𑀭𑀽𑀧𑀸𑀲𑀢𑀁,_  
_𑀪𑀼𑀜𑁆𑀚𑀁 𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀩𑀳𑀽;_  
_𑀢𑀼𑀬𑁆𑀳𑁂𑀯𑁂𑀲𑁄 𑀆𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀲𑀭𑀡𑀁 𑀢𑀁 𑀉𑀧𑁂𑀫𑀳𑀁𑁇_  


(134.)

847\. _𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻 𑀘 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(135.)

848\. _‘𑀬𑁄 𑀲𑁄 𑀲𑀁𑀖𑀁 𑀅𑀪𑁄𑀚𑁂𑀲𑀺,_  
_𑀉𑀚𑀼𑀪𑀽𑀢𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀢𑀣𑀸𑀕𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(136.)

849\. _𑀲𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀲𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀓𑀓𑀫𑁆𑀫𑀸𑀪𑀺𑀭𑀤𑁆𑀥𑁄 𑀲𑁄,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(137.)

850\. _𑀤𑀲 𑀅𑀝𑁆𑀞 𑀘𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀲𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀭𑀚𑁆𑀚𑀁 𑀧𑀜𑁆𑀘𑀲𑀢𑀁,_  
_𑀯𑀲𑀼𑀥𑀁 𑀆𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(138.)

851\. _𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀫𑁄𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀓𑀸𑀦𑀦𑀁 𑀩𑁆𑀬𑀕𑁆𑀖𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑀸𑁇_  


(139.)

852\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑀢𑁆𑀢𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(140.)

853\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁮𑁦𑁩)_  


854\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

855\. 𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

856\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



857\. _𑀲𑀻𑀳𑀸𑀲𑀦𑀻 𑀏𑀓𑀣𑀫𑁆𑀪𑀻,_  
_𑀦𑀦𑁆𑀤𑁄 𑀘 𑀘𑀽𑀍𑀅𑀧𑀦𑁆𑀣𑀓𑁄;_  
_𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀭𑀸𑀳𑀼𑀮𑁄 𑀘𑁂𑀯,_  
_𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑁄 𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑀓𑁄𑁇_  


858\. _𑀲𑁄𑀧𑀸𑀓𑁄 𑀫𑀗𑁆𑀕𑀮𑁄 𑀘𑁂𑀯,_  
_𑀤𑀲𑁂𑀯 𑀤𑀼𑀢𑀺𑀬𑁂 𑀯𑀕𑁆𑀕𑁂;_  
_𑀲𑀢𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀢𑀺𑀁𑀲 𑀘,_  
_𑀕𑀸𑀣𑀸 𑀘𑁂𑀢𑁆𑀣 𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑁇_  


---

859\. 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀬𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
