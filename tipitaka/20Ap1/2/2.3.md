# 𑁨𑁇𑁩 𑀦𑀦𑁆𑀤𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(27.)

726\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀯𑀢𑁆𑀣𑀁 𑀔𑁄𑀫𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(28.)

727\. _𑀢𑀁 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀭𑀦𑀸𑀫𑀓𑁄;_  
_‘𑀇𑀫𑀺𑀦𑀸 𑀯𑀢𑁆𑀣𑀤𑀸𑀦𑁂𑀦,_  
_𑀳𑁂𑀫𑀯𑀡𑁆𑀡𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(29.)

728\. _𑀤𑁆𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸,_  
_𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀳𑀺 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀓𑀦𑀺𑀝𑁆𑀞𑁄 𑀢𑁆𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(30.)

729\. _𑀭𑀸𑀕𑀭𑀢𑁆𑀢𑁄 𑀲𑀼𑀔𑀲𑀻𑀮𑁄,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀕𑁂𑀥𑀫𑀸𑀬𑀼𑀢𑁄;_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀢𑀤𑀸 𑀢𑁆𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(31.)

730\. _𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀢𑁆𑀯𑀁 𑀢𑀢𑁆𑀣,_  
_𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀲𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(32.)

731\. _𑀲𑀢𑁆𑀢 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀘𑀢𑀼𑀭𑁄 𑀘𑁂𑀍𑀅𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑀝𑁆𑀞𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀉𑀧𑀘𑁂𑀮𑀸 𑀘𑀢𑀼𑀚𑁆𑀚𑀦𑀸𑁇_  


(33.)

732\. _𑀧𑀜𑁆𑀘 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀘𑁂𑀍𑀆𑀯 𑀘𑀢𑀼𑀭𑁄 𑀚𑀦𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀢𑀼𑀤𑀻𑀧𑀫𑁆𑀳𑀺 𑀇𑀲𑁆𑀲𑀭𑀸𑁇_  


(34.)

733\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


734\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---
