# 𑁪𑁇𑁮 𑀆𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀼𑀡𑁆𑀟𑀥𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀆𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(94.)

1180\. _“𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀲𑀺𑀔𑀺𑀫𑁆𑀳𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁂;_  
_𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀣𑀽𑀧𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(95.)

1181\. _𑀯𑀟𑁆𑀠𑀓𑀻𑀳𑀺 𑀓𑀣𑀸𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀫𑀽𑀮𑀁 𑀤𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀆𑀬𑀸𑀕𑀁 𑀓𑀸𑀭𑀧𑁂𑀲𑀳𑀁𑁇_  


(96.)

1182\. _𑀅𑀝𑁆𑀞 𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀅𑀩𑁆𑀩𑁄𑀓𑀺𑀡𑁆𑀡𑀁 𑀯𑀲𑀺𑀁 𑀅𑀳𑀁;_  
_𑀅𑀯𑀲𑁂𑀲𑁂𑀲𑀼 𑀓𑀧𑁆𑀧𑁂𑀲𑀼,_  
_𑀯𑁄𑀓𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀲𑀭𑀺𑀁 𑀅𑀳𑀁𑁇_  


(97.)

1183\. _𑀓𑀸𑀬𑁂 𑀯𑀺𑀲𑀁 𑀦 𑀓𑀫𑀢𑀺,_  
_𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀦 𑀘 𑀳𑀦𑁆𑀢𑀺 𑀫𑁂;_  
_𑀉𑀤𑀓𑁂𑀳𑀁 𑀦 𑀫𑀺𑀬𑁆𑀬𑀸𑀫𑀺,_  
_𑀆𑀬𑀸𑀕𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(98.)

1184\. _𑀬𑀤𑀺𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀳𑀁 𑀯𑀲𑁆𑀲𑀁,_  
_𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀧𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀤𑁂𑀯𑀸𑀧𑀺 𑀫𑁂 𑀯𑀲𑀁 𑀏𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(99.)

1185\. _𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀦 𑀫𑀁 𑀓𑁂𑀘𑀸𑀯𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(100.)

1186\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀆𑀬𑀸𑀕𑀁 𑀬𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀆𑀬𑀸𑀕𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(101.)

1187\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1188\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1189\. 𑀆𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
