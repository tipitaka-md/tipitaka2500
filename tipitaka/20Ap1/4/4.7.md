# 𑁪𑁇𑁭 𑀮𑀲𑀼𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀡𑁆𑀟𑀥𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀮𑀲𑀼𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(89.)

1173\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀢𑀸𑀧𑀲𑁄 𑀆𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀮𑀲𑀼𑀡𑀁 𑀉𑀧𑀚𑀻𑀯𑀸𑀫𑀺,_  
_𑀮𑀲𑀼𑀡𑀁 𑀫𑀬𑁆𑀳𑀪𑁄𑀚𑀦𑀁𑁇_  


(90.)

1174\. _𑀔𑀸𑀭𑀺𑀬𑁄 𑀧𑀽𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀁𑀖𑀸𑀭𑀸𑀫𑀫𑀕𑀘𑁆𑀙𑀳𑀁;_  
_𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀮𑀲𑀼𑀡𑀁 𑀅𑀤𑀁𑁇_  


(91.)

1175\. _𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀦𑀭𑀕𑁆𑀕𑀲𑁆𑀲,_  
_𑀲𑀸𑀲𑀦𑁂 𑀦𑀺𑀭𑀢𑀲𑁆𑀲𑀳𑀁;_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀮𑀲𑀼𑀡𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀳𑀁𑁇_  


(92.)

1176\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀮𑀲𑀼𑀡𑀁 𑀬𑀫𑀤𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀮𑀲𑀼𑀡𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(93.)

1177\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1178\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀲𑀼𑀡𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1179\. 𑀮𑀲𑀼𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
