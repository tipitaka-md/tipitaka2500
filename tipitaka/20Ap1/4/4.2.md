# 𑁪𑁇𑁨 𑀲𑀸𑀕𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀡𑁆𑀟𑀥𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀕𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(17.)

1091\. _“𑀲𑁄𑀪𑀺𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀤𑀸;_  
_𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄 𑀲𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀆𑀭𑀸𑀫𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁𑁇_  


(18.)

1092\. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀆𑀭𑀸𑀫𑀤𑁆𑀯𑀸𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(19.)

1093\. _𑀢𑀫𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑀦𑁆𑀢𑀁 𑀤𑀦𑁆𑀢𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁;_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀦𑁆𑀣𑀯𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(20.)

1094\. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀧𑀸𑀤𑀧𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀫𑀳𑀺𑀬𑀸 𑀢𑁂 𑀯𑀺𑀭𑀽𑀳𑀭𑁂;_  
_𑀩𑀼𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀢𑀣𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀭𑀼𑀳𑀦𑁆𑀢𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂𑁇_  


(21.)

1095\. _𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄𑀲𑀺 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀫𑀳𑁂𑀲𑀺 𑀩𑀳𑀼𑀓𑁂 𑀚𑀦𑁂;_  
_𑀯𑀺𑀧𑀣𑀸 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀣𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀲𑁂 𑀢𑀼𑀯𑀁𑁇_  


(22.)

1096\. _𑀤𑀦𑁆𑀢𑁄 𑀤𑀦𑁆𑀢𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑁄,_  
_𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑁂𑀳𑀺 𑀘;_  
_𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀪𑀺𑁇_  


(23.)

1097\. _𑀅𑀮𑀗𑁆𑀓𑀢𑁄 𑀧𑀭𑀺𑀲𑀸𑀳𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀜𑀸𑀡𑁂𑀳𑀺 𑀲𑁄𑀪𑀢𑀺;_  
_𑀧𑀪𑀸 𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀢𑁂 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀲𑀽𑀭𑀺𑀬𑁄𑀤𑀬𑀦𑁂 𑀬𑀣𑀸𑁇_  


(24.)

1098\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀫𑀳𑁂𑀲𑀻 𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(25.)

1099\. _‘𑀬𑁄 𑀲𑁄 𑀳𑀸𑀲𑀁 𑀚𑀦𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀫𑀁 𑀓𑀺𑀢𑁆𑀢𑁂𑀲𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(26.)

1100\. _𑀢𑀼𑀲𑀺𑀢𑀸 𑀳𑀺 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀸𑀲𑀦𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(27.)

1101\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀮𑀪𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀸𑀕𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄’𑁇_  


(28.)

1102\. _𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀓𑀸𑀬𑁂𑀦,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑀬𑀺𑀁;_  
_𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀆𑀚𑀻𑀯𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑀬𑀺𑀁𑁇_  


(29.)

1103\. _𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀫𑀸𑀦𑁄𑀳𑀁,_  
_𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀽𑀲𑀼 𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(30.)

1104\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1105\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1106\. 𑀲𑀸𑀕𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
