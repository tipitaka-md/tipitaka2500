# 𑁪𑁇𑁬 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀡𑁆𑀟𑀥𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(84.)

1166. _“𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(85.)

1167. _𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁,_  
_𑀉𑀚𑀼𑀪𑀽𑀢𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀉𑀘𑁆𑀙𑀼𑀦𑀸 𑀫𑀡𑁆𑀟𑀧𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀪𑁄𑀚𑁂𑀲𑀺𑀁 𑀲𑀁𑀖𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(86.)

1168. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑁂 𑀅𑀪𑀺𑀪𑁄𑀫𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(87.)

1169. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀉𑀘𑁆𑀙𑀼𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(88.)

1170. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1171. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1172. 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
