# 𑁪𑁇𑁪 𑀓𑀸𑀍𑀉𑀤𑀸𑀬𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀡𑁆𑀟𑀥𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀍𑀉𑀤𑀸𑀬𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(48.)

1126\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀘𑀭𑀢𑁄 𑀘𑀸𑀭𑀺𑀓𑀁 𑀢𑀤𑀸𑁇_  


(49.)

1127\. _𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀁 𑀧𑀤𑀼𑀫𑀁 𑀕𑀬𑁆𑀳,_  
_𑀉𑀧𑁆𑀧𑀮𑀁 𑀫𑀮𑁆𑀮𑀺𑀓𑀜𑁆𑀘𑀳𑀁;_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(50.)

1128\. _𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀁 𑀲𑀼𑀪𑁄𑀚𑀦𑀁;_  
_𑀢𑀜𑁆𑀘 𑀧𑀼𑀧𑁆𑀨𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀚𑀦𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀤𑀲𑁆𑀲𑀬𑀺𑁇_  


(51.)

1129\. _𑀇𑀝𑁆𑀞𑀁 𑀓𑀦𑁆𑀢𑀁 𑀧𑀺𑀬𑀁 𑀮𑁄𑀓𑁂,_  
_𑀚𑀮𑀚𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀼𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀓𑀢𑀁 𑀢𑁂𑀦,_  
_𑀬𑁄 𑀫𑁂 𑀧𑀼𑀧𑁆𑀨𑀁 𑀅𑀤𑀸𑀲𑀺𑀤𑀁𑁇_  


(52.)

1130\. _𑀬𑁄 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑁂𑀲𑀺,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀜𑁆𑀘𑀤𑀸𑀲𑀺 𑀫𑁂;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(53.)

1131\. _‘𑀤𑀲 𑀅𑀝𑁆𑀞 𑀘𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀲𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀉𑀧𑁆𑀧𑀮𑀁 𑀧𑀤𑀼𑀫𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀫𑀮𑁆𑀮𑀺𑀓𑀜𑁆𑀘 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_  


(54.)

1132\. _𑀅𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀲𑀫𑀸𑀬𑀼𑀢𑀁;_  
_𑀆𑀓𑀸𑀲𑁂 𑀙𑀤𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀥𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(55.)

1133\. _𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀭𑀚𑁆𑀚𑀁 𑀧𑀜𑁆𑀘𑀲𑀢𑀁,_  
_𑀯𑀲𑀼𑀥𑀁 𑀆𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(56.)

1134\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(57.)

1135\. _𑀲𑀓𑀓𑀫𑁆𑀫𑀸𑀪𑀺𑀭𑀤𑁆𑀥𑁄 𑀲𑁄,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀲𑀓𑁆𑀬𑀸𑀦𑀁 𑀦𑀦𑁆𑀤𑀺𑀚𑀦𑀦𑁄,_  
_𑀜𑀸𑀢𑀺𑀩𑀦𑁆𑀥𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(58.)

1136\. _𑀲𑁄 𑀧𑀘𑁆𑀙𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄𑁇_  


(59.)

1137\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑀫𑀦𑀸𑀲𑀯𑀁;_  
_𑀕𑁄𑀢𑀫𑁄 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼 𑀢𑀁,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑁆𑀲𑀢𑀺𑁇_  


(60.)

1138\. _𑀧𑀥𑀸𑀦𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀲𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀉𑀤𑀸𑀬𑀻 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄’𑁇_  


(61.)

1139\. _𑀭𑀸𑀕𑁄 𑀤𑁄𑀲𑁄 𑀘 𑀫𑁄𑀳𑁄 𑀘,_  
_𑀫𑀸𑀦𑁄 𑀫𑀓𑁆𑀔𑁄 𑀘 𑀥𑀁𑀲𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(62.)

1140\. _𑀢𑁄𑀲𑀬𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀆𑀢𑀸𑀧𑀻 𑀦𑀺𑀧𑀓𑁄 𑀅𑀳𑀁;_  
_𑀧𑀲𑀸𑀤𑀺𑀢𑁄 𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑀺 𑀫𑀁𑁇_  


(63.)

1141\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1142\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀸𑀍𑀉𑀤𑀸𑀬𑀻 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1143\. 𑀓𑀸𑀍𑀉𑀤𑀸𑀬𑀺𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
