# 𑁧𑁪𑁇𑁮 𑀏𑀓𑀧𑀲𑀸𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑁄𑀪𑀺𑀢𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀧𑀲𑀸𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(55.)

2250\. _“𑀦𑀸𑀭𑀤𑁄 𑀇𑀢𑀺 𑀫𑁂 𑀦𑀸𑀫𑀁,_  
_𑀓𑁂𑀲𑀯𑁄 𑀇𑀢𑀺 𑀫𑀁 𑀯𑀺𑀤𑀽;_  
_𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀏𑀲𑀁,_  
_𑀅𑀕𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(56.)

2251\. _𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀅𑀲𑁆𑀲𑀸𑀲𑀬𑀦𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑁂 𑀲𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(57.)

2252\. _𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀭𑁂 𑀓𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀧𑀸𑀘𑀺𑀦𑀸𑀫𑀼𑀔𑁄𑁇_  


(58.)

2253\. _𑀲𑀢𑁆𑀢𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑀧𑀦𑁄 𑀦𑀸𑀫,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(59.)

2254\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2255\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀧𑀲𑀸𑀤𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2256\. 𑀏𑀓𑀧𑀲𑀸𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
