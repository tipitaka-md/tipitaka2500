# 𑁧𑁪𑁇𑁧𑁦 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑁄𑀪𑀺𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(66.)

2265. _“𑀧𑀸𑀭𑀸𑀯𑀢𑁄 𑀢𑀤𑀸 𑀆𑀲𑀺𑀁,_  
_𑀧𑀭𑀁 𑀅𑀦𑀼𑀧𑀭𑁄𑀥𑀓𑁄;_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑁂 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀫𑀺,_  
_𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀲𑀺𑀔𑀺𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(67.)

2266. _𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑀜𑁆𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑀁;_  
_𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄 𑀘 𑀫𑁂 𑀦𑀢𑁆𑀣𑀺,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(68.)

2267. _𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀫𑀸𑀤𑀸𑀬,_  
_𑀅𑀕𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄𑁇_  


(69.)

2268. _𑀢𑀢𑁄 𑀧𑀭𑀁 𑀉𑀧𑀸𑀤𑀸𑀬,_  
_𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑀁;_  
_𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(70.)

2269. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(71.)

2270. _𑀇𑀢𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀢𑀬𑁄 𑀆𑀲𑀼𑀁 𑀧𑀺𑀬𑀸𑀮𑀺𑀦𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(72.)

2271. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁧𑁯𑁩𑁫)_  


2272. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2273. 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

2274. 𑀲𑁄𑀪𑀺𑀢𑀯𑀕𑁆𑀕𑁄 𑀘𑀼𑀤𑁆𑀤𑀲𑀫𑁄𑁇



---

2275. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2276. _𑀲𑁄𑀪𑀺𑀢𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀘,_  
_𑀘𑀦𑁆𑀤𑀦𑁄 𑀧𑀼𑀧𑁆𑀨𑀙𑀤𑀦𑁄;_  
_𑀭𑀳𑁄 𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀻 𑀘,_  
_𑀅𑀢𑁆𑀣𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁂𑀦 𑀘𑁇_  


2277. _𑀏𑀓𑀧𑀲𑀸𑀤𑀻 𑀲𑀸𑀮𑀤𑀤𑁄,_  
_𑀤𑀲𑀫𑁄 𑀨𑀮𑀤𑀸𑀬𑀓𑁄;_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀲𑀢𑁆𑀢𑀢𑀺 𑀤𑁆𑀯𑁂 𑀘,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_
