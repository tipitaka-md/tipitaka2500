# 𑁧𑁪𑁇𑁩 𑀘𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑁄𑀪𑀺𑀢𑀯𑀕𑁆𑀕

* 𑀘𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(17.)

2202\. _“𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀸𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀓𑀺𑀦𑁆𑀦𑀭𑁄 𑀢𑀤𑀸;_  
_𑀧𑀼𑀧𑁆𑀨𑀪𑀓𑁆𑀔𑁄 𑀘𑀳𑀁 𑀆𑀲𑀺𑀁,_  
_𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺𑀯𑀲𑀦𑁄 𑀢𑀣𑀸𑁇_  


(18.)

2203\. _𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻 𑀢𑀼 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀯𑀺𑀧𑀺𑀦𑀕𑁆𑀕𑁂𑀦 𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺,_  
_𑀳𑀁𑀲𑀭𑀸𑀚𑀸𑀯 𑀅𑀫𑁆𑀩𑀭𑁂𑁇_  


(19.)

2204\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀢𑁂 𑀲𑀼𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄𑀲𑀺,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄𑁇_  


(20.)

2205\. _𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸𑀦 𑀆𑀓𑀸𑀲𑀸,_  
_𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑁄 𑀲𑀼𑀫𑁂𑀥𑀲𑁄;_  
_𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑁇_  


(21.)

2206\. _𑀯𑀺𑀮𑀻𑀦𑀁 𑀘𑀦𑁆𑀤𑀦𑀸𑀤𑀸𑀬,_  
_𑀅𑀕𑀫𑀸𑀲𑀺𑀁 𑀚𑀺𑀦𑀦𑁆𑀢𑀺𑀓𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(22.)

2207\. _𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀦𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(23.)

2208\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀘𑀦𑁆𑀤𑀦𑀁 𑀬𑀁 𑀅𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(24.)

2209\. _𑀘𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀇𑀢𑁄 𑀆𑀲𑀺𑀁𑀲𑀼 𑀢𑁂 𑀢𑀬𑁄;_  
_𑀭𑁄𑀳𑀡𑀻 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(25.)

2210\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2211\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2212\. 𑀘𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇

  
𑀅𑀝𑁆𑀞𑀫𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
