# 𑁧𑁪𑁇𑁫 𑀭𑀳𑁄𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑁄𑀪𑀺𑀢𑀯𑀕𑁆𑀕

### 𑀭𑀳𑁄𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(34.)

2223\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀯𑀲𑀪𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀩𑁆𑀩𑀢𑀧𑀸𑀤𑀫𑁆𑀳𑀺,_  
_𑀅𑀲𑁆𑀲𑀫𑁄 𑀆𑀲𑀺 𑀫𑀸𑀧𑀺𑀢𑁄𑁇_  


(35.)

2224\. _𑀢𑀻𑀡𑀺 𑀲𑀺𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀸𑀘𑁂𑀲𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀤𑀸;_  
_𑀲𑀁𑀳𑀭𑀺𑀢𑁆𑀯𑀸𑀦 𑀢𑁂 𑀲𑀺𑀲𑁆𑀲𑁂,_  
_𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑁇_  


(36.)

2225\. _𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀫𑀦𑁆𑀢𑀧𑀸𑀭𑀕𑀽;_  
_𑀩𑀼𑀤𑁆𑀥𑀯𑁂𑀤𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀜𑀸𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁𑁇_  


(37.)

2226\. _𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀲𑀻𑀤𑀺𑀁 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂;_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(38.)

2227\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀜𑀸𑀡𑀲𑀜𑁆𑀜𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(39.)

2228\. _𑀲𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀭𑀸𑀚𑀸 𑀲𑀺𑀭𑀺𑀥𑀭𑁄 𑀅𑀳𑀼;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(40.)

2229\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2230\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀭𑀳𑁄𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2231\. 𑀭𑀳𑁄𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
