# 𑁧𑁪𑁇𑁭 𑀅𑀢𑁆𑀣𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑁄𑀪𑀺𑀢𑀯𑀕𑁆𑀕

* 𑀅𑀢𑁆𑀣𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(47.)

2240\. _“𑀯𑀺𑀲𑀸𑀮𑀫𑀸𑀍𑀏 𑀆𑀲𑀻𑀦𑁄,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀁 𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁𑁇_  


(48.)

2241\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸 𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸,_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀓𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇_  


(49.)

2242\. _𑀜𑀸𑀡𑁂 𑀉𑀧𑀦𑀺𑀥𑀸 𑀬𑀲𑁆𑀲,_  
_𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀅𑀦𑀦𑁆𑀢𑀜𑀸𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀓𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇_  


(50.)

2243\. _𑀥𑀫𑁆𑀫𑀓𑀸𑀬𑀜𑁆𑀘 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀁,_  
_𑀓𑁂𑀯𑀮𑀁 𑀭𑀢𑀦𑀸𑀓𑀭𑀁;_  
_𑀯𑀺𑀓𑀧𑁆𑀧𑁂𑀢𑀼𑀁 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺,_  
_𑀓𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇_  


(51.)

2244\. _𑀇𑀫𑀸𑀳𑀺 𑀢𑀻𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺,_  
_𑀦𑀸𑀭𑀤𑁄𑀯𑁆𑀳𑀬𑀯𑀘𑁆𑀙𑀮𑁄;_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀁 𑀣𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁𑁇_  


(52.)

2245\. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀣𑀯𑀦𑁂𑀦 𑀘;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(53.)

2246\. _𑀇𑀢𑁄 𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀲𑀼𑀫𑀺𑀢𑁆𑀢𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(54.)

2247\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2248\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀢𑁆𑀣𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2249\. 𑀅𑀢𑁆𑀣𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
