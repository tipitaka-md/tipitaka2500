# 𑁪𑁨𑁇𑁭 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀯𑀕𑁆𑀕

### 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(142.)

5403. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀫𑀸𑀮𑀺𑀓𑁄 𑀢𑀤𑀸;_  
_𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀁 𑀧𑀽𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀕𑀫𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀡𑀁𑁇_  


(143.)

5404. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀫𑀳𑀢𑀸 𑀆𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(144.)

5405. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑀁,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀮𑁄𑀓𑀢𑀸𑀭𑀡𑀁;_  
_𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀅𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(145.)

5406. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(146.)

5407. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(147.)

5408. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(148.)

5409. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5410. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5411. 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
