# 𑁪𑁨𑁇𑁧 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀯𑀕𑁆𑀕

### 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

5249\. _“𑀲𑀼𑀫𑁂𑀥𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀕𑁆𑀕𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀫𑀼𑀦𑀺;_  
_𑀯𑀺𑀯𑁂𑀓𑀓𑀸𑀫𑁄 𑀮𑁄𑀓𑀕𑁆𑀕𑁄,_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(2.)

5250\. _𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀳𑀺𑀫𑀯𑀁,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(3.)

5251\. _𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑁄 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(4.)

5252\. _𑀔𑀸𑀭𑀺𑀪𑀸𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀦𑀫𑀚𑁆𑀛𑁄𑀕𑀳𑀺𑀁 𑀅𑀳𑀁;_  
_𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑀫𑀦𑀸𑀲𑀯𑀁𑁇_  


(5.)

5253\. _𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀅𑀲𑁆𑀲𑀫𑀁;_  
_𑀘𑀢𑀼𑀤𑀡𑁆𑀟𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀓𑀸𑀲𑀺𑀁 𑀫𑀡𑁆𑀟𑀧𑀁 𑀢𑀤𑀸𑁇_  


(6.)

5254\. _𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀡𑁆𑀟𑀧𑀁 𑀙𑀸𑀤𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀅𑀪𑀺𑀯𑀦𑁆𑀤𑀺𑀁 𑀢𑀣𑀸𑀕𑀢𑀁𑁇_  


(7.)

5255\. _𑀬𑀁 𑀯𑀤𑀦𑁆𑀢𑀺 𑀲𑀼𑀫𑁂𑀥𑁄𑀢𑀺,_  
_𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀁 𑀲𑀼𑀫𑁂𑀥𑀲𑀁;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(8.)

5256\. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀕𑀺𑀭𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸 𑀲𑀫𑀸𑀕𑀫𑀼𑀁;_  
_𑀅𑀲𑀁𑀲𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(9.)

5257\. _𑀲𑀼𑀫𑁂𑀥𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(10.)

5258\. _‘𑀬𑁄 𑀫𑁂 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀡𑁆𑀟𑀧𑀁,_  
_𑀥𑀸𑀭𑀬𑀻 𑀲𑀸𑀮𑀙𑀸𑀤𑀺𑀢𑀁;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(11.)

5259\. _𑀤𑁂𑀯𑀪𑀽𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸,_  
_𑀳𑁂𑀫𑀯𑀡𑁆𑀡𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀳𑀽𑀢𑀪𑁄𑀕𑁄 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(12.)

5260\. _𑀲𑀝𑁆𑀞𑀺 𑀦𑀸𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀘𑁆𑀙𑀸 𑀫𑀸𑀢𑀗𑁆𑀕𑀸,_  
_𑀳𑁂𑀫𑀓𑀧𑁆𑀧𑀦𑀯𑀸𑀲𑀲𑀸𑁇_  


(13.)

5261\. _𑀆𑀭𑀽𑀍𑀳𑀸 𑀕𑀸𑀫𑀡𑀻𑀬𑁂𑀳𑀺,_  
_𑀢𑁄𑀫𑀭𑀗𑁆𑀓𑀼𑀲𑀧𑀸𑀡𑀺𑀪𑀺;_  
_𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀆𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀭𑀁;_  
_𑀢𑁂𑀳𑀺 𑀦𑀸𑀕𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀬𑀁 𑀦𑀭𑁄𑁇_  


(14.)

5262\. _𑀲𑀝𑁆𑀞𑀺 𑀅𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑀸𑀯 𑀚𑀸𑀢𑀺𑀬𑀸,_  
_𑀲𑀺𑀦𑁆𑀥𑀯𑀸 𑀲𑀻𑀖𑀯𑀸𑀳𑀺𑀦𑁄𑁇_  


(15.)

5263\. _𑀆𑀭𑀽𑀍𑀳𑀸 𑀕𑀸𑀫𑀡𑀻𑀬𑁂𑀳𑀺,_  
_𑀇𑀮𑁆𑀮𑀺𑀬𑀸𑀘𑀸𑀧𑀥𑀸𑀭𑀺𑀪𑀺;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(16.)

5264\. _𑀲𑀝𑁆𑀞𑀺 𑀭𑀣𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀤𑀻𑀧𑀸 𑀅𑀣𑁄𑀧𑀺 𑀯𑁂𑀬𑀕𑁆𑀖𑀸,_  
_𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀸 𑀉𑀲𑁆𑀲𑀺𑀢𑀤𑁆𑀥𑀚𑀸𑁇_  


(17.)

5265\. _𑀆𑀭𑀽𑀍𑀳𑀸 𑀕𑀸𑀫𑀡𑀻𑀬𑁂𑀳𑀺,_  
_𑀘𑀸𑀧𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀯𑀫𑁆𑀫𑀺𑀪𑀺;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(18.)

5266\. _𑀲𑀝𑁆𑀞𑀺 𑀕𑀸𑀫𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀸𑀦𑀺,_  
_𑀲𑀼𑀲𑀫𑀺𑀤𑁆𑀥𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀲𑀤𑀸 𑀧𑀸𑀢𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(19.)

5267\. _𑀳𑀢𑁆𑀣𑀻 𑀅𑀲𑁆𑀲𑀸 𑀭𑀣𑀸 𑀧𑀢𑁆𑀢𑀻,_  
_𑀲𑁂𑀦𑀸 𑀘 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀻;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(20.)

5268\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(21.)

5269\. _𑀲𑀢𑀸𑀦𑀁 𑀢𑀻𑀡𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(22.)

5270\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(23.)

5271\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(24.)

5272\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑀫𑀼𑀧𑀸𑀤𑀸𑀬,_  
_𑀕𑀯𑁂𑀲𑀺𑀁 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁𑁇_  


(25.)

5273\. _𑀮𑀸𑀪𑀸 𑀫𑀬𑁆𑀳𑀁 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀫𑁂,_  
_𑀬𑀫𑀳𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀸𑀲𑀦𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(26.)

5274\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀢𑀯 𑀜𑀸𑀡𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁𑁇_  


(27.)

5275\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀫𑀺,_  
_𑀨𑀮𑀁 𑀫𑁂 𑀜𑀸𑀡𑀓𑀺𑀢𑁆𑀢𑀦𑁂𑁇_  


(28.)

5276\. _𑀇𑀤𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀘𑀭𑀺𑀫𑁄 𑀯𑀢𑁆𑀢𑀢𑁂 𑀪𑀯𑁄;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(29.)

5277\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(30.)

5278\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(31.)

5279\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5280\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5281\. 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
