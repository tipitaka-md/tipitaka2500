# 𑁪𑁨𑁇𑁬 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(124.)

5383\. _“𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀲𑀼𑀕𑀢𑁂,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀲𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(125.)

5384\. _𑀆𑀭𑁄𑀧𑀺𑀢𑁂 𑀘 𑀘𑀺𑀢𑀓𑁂,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀬𑀣𑀸𑀲𑀓𑁂𑀦 𑀣𑀸𑀫𑁂𑀦,_  
_𑀘𑀺𑀢𑀁 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(126.)

5385\. _𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀘𑀺𑀢𑀓𑀲𑁆𑀲,_  
_𑀤𑀻𑀧𑀁 𑀉𑀚𑁆𑀚𑀸𑀮𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀬𑀸𑀯 𑀉𑀤𑁂𑀢𑀺 𑀲𑀽𑀭𑀺𑀬𑁄,_  
_𑀤𑀻𑀧𑀁 𑀫𑁂 𑀢𑀸𑀯 𑀉𑀚𑁆𑀚𑀮𑀺𑁇_  


(127.)

5386\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(128.)

5387\. _𑀢𑀢𑁆𑀣 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀩𑁆𑀬𑀫𑁆𑀳𑀁,_  
_𑀏𑀓𑀤𑀻𑀧𑀻𑀢𑀺 𑀜𑀸𑀬𑀢𑀺;_  
_𑀤𑀻𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀩𑁆𑀬𑀫𑁆𑀳𑁂 𑀧𑀚𑁆𑀚𑀮𑀭𑁂 𑀫𑀫𑁇_  


(129.)

5388\. _𑀉𑀤𑀬𑀦𑁆𑀢𑁄𑀯 𑀲𑀽𑀭𑀺𑀬𑁄,_  
_𑀤𑁂𑀳𑁄 𑀫𑁂 𑀚𑁄𑀢𑀢𑁂 𑀲𑀤𑀸;_  
_𑀲𑀧𑁆𑀧𑀪𑀸𑀳𑀺 𑀲𑀭𑀻𑀭𑀲𑁆𑀲,_  
_𑀆𑀮𑁄𑀓𑁄 𑀳𑁄𑀢𑀺 𑀫𑁂 𑀲𑀤𑀸𑁇_  


(130.)

5389\. _𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀲𑁂𑀮𑀁,_  
_𑀲𑀫𑀢𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀳𑀁𑁇_  


(131.)

5390\. _𑀲𑀢𑁆𑀢𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀁 𑀅𑀳𑀁;_  
_𑀏𑀓𑀢𑀺𑀁𑀲𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(132.)

5391\. _𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(133.)

5392\. _𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀁 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀬𑀁;_  
_𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑀲𑁆𑀲𑀸𑀧𑀺,_  
_𑀅𑀓𑁆𑀔𑀺 𑀫𑁂 𑀦 𑀦𑀺𑀫𑀻𑀮𑀢𑀺𑁇_  


(134.)

5393\. _𑀚𑀸𑀢𑀺𑀬𑀸 𑀘𑀢𑀼𑀯𑀲𑁆𑀲𑁄𑀳𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀅𑀟𑁆𑀠𑀫𑀸𑀲𑁂 𑀅𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(135.)

5394\. _𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀲𑀺𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀓𑀺𑀮𑁂𑀲𑀸 𑀲𑀜𑁆𑀙𑀺𑀦𑁆𑀦𑀸,_  
_𑀏𑀓𑀤𑀻𑀧𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(136.)

5395\. _𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀲𑁂𑀮𑀁,_  
_𑀧𑀩𑁆𑀩𑀢𑀜𑁆𑀘𑀸𑀧𑀺 𑀓𑁂𑀯𑀮𑀁;_  
_𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀏𑀓𑀤𑀻𑀧𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(137.)

5396\. _𑀯𑀺𑀲𑀫𑀸 𑀫𑁂 𑀲𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀦𑀸𑀳𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀺𑀫𑀺𑀭𑀁,_  
_𑀏𑀓𑀤𑀻𑀧𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(138.)

5397\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀻𑀧𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀏𑀓𑀤𑀻𑀧𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(139.)

5398\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(140.)

5399\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(141.)

5400\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5401\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5402\. 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
