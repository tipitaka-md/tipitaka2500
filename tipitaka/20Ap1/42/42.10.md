# 𑁪𑁨𑁇𑁧𑁦 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀯𑀕𑁆𑀕

* 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(184.)

5451\. _“𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁂,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑁂 𑀧𑀸𑀯𑀘𑀦𑁂,_  
_𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑀲𑀓𑁆𑀓𑀢𑁂𑁇_  


(185.)

5452\. _𑀘𑀡𑁆𑀟𑀸𑀮𑁄 𑀆𑀲𑀳𑀁 𑀢𑀢𑁆𑀣,_  
_𑀆𑀲𑀦𑁆𑀤𑀺𑀧𑀻𑀞𑀓𑀸𑀭𑀓𑁄;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀚𑀻𑀯𑀸𑀫𑀺,_  
_𑀢𑁂𑀦 𑀧𑁄𑀲𑁂𑀫𑀺 𑀤𑀸𑀭𑀓𑁂𑁇_  


(186.)

5453\. _𑀆𑀲𑀦𑁆𑀤𑀺𑀁 𑀲𑀼𑀓𑀢𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺;_  
_𑀲𑀬𑀫𑁂𑀯𑀼𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(187.)

5454\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(188.)

5455\. _𑀤𑁂𑀯𑀮𑁄𑀓𑀕𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑁄𑀤𑀸𑀫𑀺 𑀢𑀺𑀤𑀲𑁂 𑀕𑀡𑁂;_  
_𑀲𑀬𑀦𑀸𑀦𑀺 𑀫𑀳𑀕𑁆𑀖𑀸𑀦𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(189.)

5456\. _𑀧𑀜𑁆𑀜𑀸𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(190.)

5457\. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀲𑀼𑀔𑀺𑀢𑁄 𑀬𑀲𑀯𑀸 𑀳𑁄𑀫𑀺,_  
_𑀫𑀜𑁆𑀘𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(191.)

5458\. _𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀏𑀫𑀺 𑀘𑁂 𑀫𑀸𑀦𑀼𑀲𑀁 𑀪𑀯𑀁;_  
_𑀫𑀳𑀸𑀭𑀳𑀸 𑀲𑀼𑀲𑀬𑀦𑀸,_  
_𑀲𑀬𑀫𑁂𑀯 𑀪𑀯𑀦𑁆𑀢𑀺 𑀫𑁂𑁇_  


(192.)

5459\. _𑀅𑀬𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀘𑀭𑀺𑀫𑁄 𑀯𑀢𑁆𑀢𑀢𑁂 𑀪𑀯𑁄;_  
_𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀲𑀬𑀦𑁂 𑀓𑀸𑀮𑁂,_  
_𑀲𑀬𑀦𑀁 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


(193.)

5460\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀜𑁆𑀘𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(194.)

5461\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(195.)

5462\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(196.)

5463\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁪𑁪𑁬𑁦)_  


5464\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5465\. 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

5466\. 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀯𑀕𑁆𑀕𑁄 𑀩𑀸𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁄𑁇



---

5467\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



5468\. _𑀪𑀤𑁆𑀤𑀸𑀮𑀻 𑀏𑀓𑀙𑀢𑁆𑀢𑁄 𑀘,_  
_𑀢𑀺𑀡𑀲𑀽𑀮𑁄 𑀘 𑀫𑀁𑀲𑀤𑁄;_  
_𑀦𑀸𑀕𑀧𑀮𑁆𑀮𑀯𑀺𑀓𑁄 𑀤𑀻𑀧𑀻,_  
_𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀺 𑀬𑀸𑀕𑀼𑀤𑀸𑀬𑀓𑁄𑁇_  


5469\. _𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀻 𑀫𑀜𑁆𑀘𑀤𑀤𑁄,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀡𑀺𑀢𑀸 𑀘𑀺𑀳;_  
_𑀤𑁆𑀯𑁂𑀲𑀢𑀸𑀦𑀺 𑀘 𑀕𑀸𑀣𑀸𑀦𑀁,_  
_𑀕𑀸𑀣𑀸 𑀘𑁂𑀓𑀸 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_
