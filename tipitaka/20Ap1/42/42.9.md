# 𑁪𑁨𑁇𑁯 𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(170.)

5435\. _“𑀯𑀦𑀘𑀸𑀭𑀻 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀲𑀢𑀢𑀁 𑀯𑀦𑀓𑀫𑁆𑀫𑀺𑀓𑁄;_  
_𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁𑁇_  


(171.)

5436\. _𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁;_  
_𑀯𑀦𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁𑁇_  


(172.)

5437\. _𑀧𑀭𑀓𑀫𑁆𑀫𑀸𑀬𑀦𑁂 𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀼𑀜𑁆𑀜𑀜𑁆𑀘 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀬𑀁 𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀪𑁄𑀚𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁 𑀫𑀼𑀦𑀺𑀁𑁇_  


(173.)

5438\. _𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀫𑀫 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺 𑀢𑀤𑀸 𑀫𑀼𑀦𑀺𑁇_  


(174.)

5439\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(175.)

5440\. _𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(176.)

5441\. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀲𑀼𑀔𑀺𑀢𑁄 𑀬𑀲𑀯𑀸 𑀳𑁄𑀫𑀺,_  
_𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(177.)

5442\. _𑀪𑀯𑀸𑀪𑀯𑁂 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄,_  
_𑀮𑀪𑀸𑀫𑀺 𑀅𑀫𑀺𑀢𑀁 𑀥𑀦𑀁;_  
_𑀪𑁄𑀕𑁂 𑀫𑁂 𑀊𑀦𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(178.)

5443\. _𑀦𑀤𑀻𑀲𑁄𑀢𑀧𑀝𑀺𑀪𑀸𑀕𑀸,_  
_𑀪𑁄𑀕𑀸 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀭𑁂 𑀫𑀫;_  
_𑀧𑀭𑀺𑀫𑁂𑀢𑀼𑀁 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺,_  
_𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(179.)

5444\. _𑀇𑀫𑀁 𑀔𑀸𑀤 𑀇𑀫𑀁 𑀪𑀼𑀜𑁆𑀚,_  
_𑀇𑀫𑀫𑁆𑀳𑀺 𑀲𑀬𑀦𑁂 𑀲𑀬;_  
_𑀢𑁂𑀦𑀸𑀳𑀁 𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀫𑀺,_  
_𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(180.)

5445\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(181.)

5446\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(182.)

5447\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(183.)

5448\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5449\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5450\. 𑀧𑀢𑁆𑀣𑁄𑀤𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
