# 𑁪𑁨𑁇𑁪 𑀫𑀥𑀼𑀫𑀁𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀤𑁆𑀤𑀸𑀮𑀺𑀯𑀕𑁆𑀕

### 𑀫𑀥𑀼𑀫𑀁𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(106.)

5361\. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀲𑀽𑀓𑀭𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀉𑀓𑁆𑀓𑁄𑀝𑀓𑀁 𑀭𑀦𑁆𑀥𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀥𑀼𑀫𑀁𑀲𑀫𑁆𑀳𑀺 𑀑𑀓𑀺𑀭𑀺𑀁𑁇_  


(107.)

5362\. _𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀁 𑀅𑀳𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀏𑀓𑀁 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀲𑀳𑀁;_  
_𑀧𑀽𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀧𑀢𑁆𑀢𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(108.)

5363\. _𑀬𑁄𑀢𑁆𑀣 𑀣𑁂𑀭𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀲𑀺 𑀫𑀫𑀁 𑀢𑀤𑀸;_  
_𑀇𑀫𑀺𑀦𑀸 𑀧𑀢𑁆𑀢𑀧𑀽𑀭𑁂𑀦,_  
_𑀮𑀪𑀲𑁆𑀲𑀼 𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀼𑀔𑀁𑁇_  


(109.)

5364\. _𑀤𑀼𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(110.)

5365\. _𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁;_  
_𑀢𑀢𑁆𑀣 𑀪𑀼𑀢𑁆𑀯𑀸 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘,_  
_𑀮𑀪𑀸𑀫𑀺 𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀼𑀔𑀁𑁇_  


(111.)

5366\. _𑀫𑀡𑁆𑀟𑀧𑁂 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀸𑀪𑀺𑀯𑀲𑁆𑀲𑁄 𑀫𑁂,_  
_𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(112.)

5367\. _𑀇𑀤𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀘𑀭𑀺𑀫𑁄 𑀯𑀢𑁆𑀢𑀢𑁂 𑀪𑀯𑁄;_  
_𑀇𑀥𑀸𑀧𑀺 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀁 𑀫𑁂,_  
_𑀯𑀲𑁆𑀲𑀢𑁂 𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀺𑀓𑀁𑁇_  


(113.)

5368\. _𑀢𑁂𑀦𑁂𑀯 𑀫𑀥𑀼𑀤𑀸𑀦𑁂𑀦,_  
_𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑁆𑀯𑀸 𑀪𑀯𑁂 𑀅𑀳𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(114.)

5369\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀥𑀼𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(115.)

5370\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(116.)

5371\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(117.)

5372\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5373\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀥𑀼𑀫𑀁𑀲𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5374\. 𑀫𑀥𑀼𑀫𑀁𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
