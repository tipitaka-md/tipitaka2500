# 𑁫𑁦𑁇𑁧𑁦 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀺𑀗𑁆𑀓𑀡𑀺𑀧𑀼𑀧𑁆𑀨𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(104.)

6552\. _“𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀯𑀺𑀧𑀺𑀦𑁂 𑀯𑀺𑀘𑀭𑀁 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀼𑀁𑁇_  


(105.)

6553\. _𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀫𑀸𑀤𑀸𑀬,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀻𑀭𑀲𑁆𑀲,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(106.)

6554\. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(107.)

6555\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(108.)

6556\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(109.)

6557\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁫𑁨𑁬𑁭)_  


6558\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6559\. 𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇

  
**𑀓𑀺𑀗𑁆𑀓𑀡𑀺𑀧𑀼𑀧𑁆𑀨𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁄𑁇**



---

6560\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



6561\. _𑀓𑀺𑀗𑁆𑀓𑀡𑀻 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀜𑁆𑀘,_  
_𑀓𑁄𑀭𑀡𑁆𑀟𑀫𑀣 𑀓𑀺𑀁𑀲𑀼𑀓𑀁;_  
_𑀉𑀧𑀟𑁆𑀠𑀤𑀼𑀲𑁆𑀲𑀻 𑀖𑀢𑀤𑁄,_  
_𑀉𑀤𑀓𑀁 𑀣𑀽𑀧𑀓𑀸𑀭𑀓𑁄𑁇_  


6562\. _𑀦𑀍𑀅𑀓𑀸𑀭𑀻 𑀘 𑀦𑀯𑀫𑁄,_  
_𑀧𑀺𑀬𑀸𑀮𑀨𑀮𑀤𑀸𑀬𑀓𑁄;_  
_𑀲𑀢𑀫𑁂𑀓𑀜𑁆𑀘 𑀕𑀸𑀣𑀸𑀦𑀁,_  
_𑀦𑀯𑀓𑀜𑁆𑀘 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_  


---

6563\. 𑀅𑀣 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



6564\. _𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕𑁄 𑀪𑀤𑁆𑀤𑀸𑀮𑀺,_  
_𑀲𑀓𑀺𑀁𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀓𑁄𑀧𑀺 𑀘;_  
_𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀻 𑀯𑀺𑀪𑀻𑀢𑀓𑀻,_  
_𑀚𑀕𑀢𑀻 𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄𑁇_  


6565\. _𑀦𑀍𑀆𑀕𑀸𑀭𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁,_  
_𑀓𑀺𑀗𑁆𑀓𑀡𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀢𑀣𑀸;_  
_𑀅𑀲𑀻𑀢𑀺 𑀤𑁆𑀯𑁂 𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀘𑀢𑀼𑀤𑁆𑀤𑀲𑀲𑀢𑀸𑀦𑀺 𑀘𑁇_  


---

6566\. 𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕𑀤𑀲𑀓𑀁𑁇



---

6567\. 𑀧𑀜𑁆𑀘𑀫𑀲𑀢𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
