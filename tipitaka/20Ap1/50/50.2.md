# 𑁫𑁦𑁇𑁨 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀺𑀗𑁆𑀓𑀡𑀺𑀧𑀼𑀧𑁆𑀨𑀯𑀕𑁆𑀕

* 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(8.)

6439\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀉𑀤𑀗𑁆𑀕𑀡𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁,_  
_𑀤𑀼𑀫𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀯𑀺𑀮𑀫𑁆𑀩𑀺𑀢𑀁𑁇_  


(9.)

6440\. _𑀢𑀻𑀡𑀺 𑀓𑀺𑀗𑁆𑀓𑀡𑀺𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀑𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀫𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(10.)

6441\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(11.)

6442\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀽𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀳𑀤𑁆𑀥𑀚𑀁𑁇_  


(12.)

6443\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(13.)

6444\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(14.)

6445\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6446\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6447\. 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
