# 𑁨𑁮𑁇𑁧𑁦 𑀢𑀭𑀡𑀻𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀯𑀡𑁆𑀡𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀯𑀕𑁆𑀕

* 𑀢𑀭𑀡𑀻𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(38.)

3278\. _“𑀫𑀳𑀸𑀧𑀣𑀫𑁆𑀳𑀺 𑀯𑀺𑀲𑀫𑁂,_  
_𑀲𑁂𑀢𑀼 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀫𑀬𑀸;_  
_𑀢𑀭𑀡𑀢𑁆𑀣𑀸𑀬 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(39.)

3279\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑁄 𑀲𑁂𑀢𑀼 𑀓𑀸𑀭𑀺𑀢𑁄 𑀫𑀬𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑁂𑀢𑀼𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(40.)

3280\. _𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀲𑀫𑁄𑀕𑀥𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(41.)

3281\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁬𑁧𑁮)_  


3282\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀭𑀡𑀻𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3283\. 𑀢𑀭𑀡𑀻𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇

  
**𑀲𑀼𑀯𑀡𑁆𑀡𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇**



---

3284\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3285\. _𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀢𑀺𑀮𑀫𑀼𑀝𑁆𑀞𑀺 𑀘,_  
_𑀘𑀗𑁆𑀓𑁄𑀝𑀩𑁆𑀪𑀜𑁆𑀚𑀦𑀜𑁆𑀚𑀮𑀻;_  
_𑀧𑁄𑀢𑁆𑀣𑀓𑁄 𑀘𑀺𑀢𑀫𑀸𑀮𑀼𑀯𑀸,_  
_𑀏𑀓𑀧𑀼𑀡𑁆𑀟𑀭𑀻 𑀲𑁂𑀢𑀼𑀦𑀸;_  
_𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀻𑀢𑀺𑁇_  


---

3286\. 𑀏𑀓𑀸𑀤𑀲𑀫𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
