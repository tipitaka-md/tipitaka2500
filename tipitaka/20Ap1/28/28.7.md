# 𑁨𑁮𑁇𑁭 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀯𑀡𑁆𑀡𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(27.)

3261. _“𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀸𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀅𑀦𑀼𑀲𑁄𑀢𑀁 𑀯𑀚𑀸𑀫𑀳𑀁;_  
_𑀲𑀢𑁆𑀢 𑀫𑀸𑀮𑀼𑀯𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀘𑀺𑀢𑀫𑀸𑀭𑁄𑀧𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(28.)

3262. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀺𑀢𑀓𑀁 𑀬𑀫𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀺𑀢𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(29.)

3263. _𑀲𑀢𑁆𑀢𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀝𑀺𑀚𑀕𑁆𑀕𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀢𑁆𑀢𑀸𑀲𑀼𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(30.)

3264. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3265. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3266. 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
