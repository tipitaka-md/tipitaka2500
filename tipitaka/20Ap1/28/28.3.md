# 𑁨𑁮𑁇𑁩 𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀯𑀡𑁆𑀡𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(10.)

3236\. _“𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀯𑀲𑀢𑀻 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂;_  
_𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀘𑀗𑁆𑀓𑁄𑀝𑀓 𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(11.)

3237\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄;_  
_𑀧𑀼𑀧𑁆𑀨𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀳𑀁𑁇_  


(12.)

3238\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀫𑀤𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(13.)

3239\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3240\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3241\. 𑀘𑀗𑁆𑀓𑁄𑀝𑀓𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
