# 𑁨𑁮𑁇𑁫 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀯𑀡𑁆𑀡𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(19.)

3249\. _“𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀦𑀺𑀬𑀢𑁂 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂;_  
_𑀯𑀼𑀢𑁆𑀣𑁄𑀓𑀸𑀲𑁄 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁄,_  
_𑀲𑀫𑀡𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(20.)

3250\. _𑀢𑀺𑀲𑁆𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲,_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀦𑁆𑀣𑀭𑀺𑀁 𑀧𑀼𑀧𑁆𑀨𑀲𑀦𑁆𑀣𑀭𑀁𑁇_  


(21.)

3251\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀭𑀺𑀁 𑀧𑀼𑀧𑁆𑀨𑀲𑀦𑁆𑀣𑀭𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀦𑁆𑀣𑀭𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(22.)

3252\. _𑀇𑀢𑁄 𑀘𑀼𑀤𑁆𑀤𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀫𑀦𑀼𑀚𑀸𑀥𑀺𑀧𑁄;_  
_𑀏𑀓𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀦𑀸𑀫,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(23.)

3253\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3254\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3255\. 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
