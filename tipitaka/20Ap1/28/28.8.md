# 𑁨𑁮𑁇𑁮 𑀆𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀯𑀡𑁆𑀡𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀆𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(31.)

3267\. _“𑀧𑀩𑁆𑀩𑀢𑁂 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀫𑀳𑀸𑀲𑀺𑀦𑁆𑀥𑀼 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀁 𑀯𑀻𑀢𑀭𑀸𑀕𑀁,_  
_𑀲𑀼𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀁𑁇_  


(32.)

3268\. _𑀧𑀭𑀫𑁄𑀧𑀲𑀫𑁂 𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀯𑀺𑀫𑁆𑀳𑀺𑀢𑀫𑀸𑀦𑀲𑁄;_  
_𑀆𑀮𑀼𑀯𑀁 𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(33.)

3269\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀆𑀮𑀼𑀯𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(34.)

3270\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3271\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3272\. 𑀆𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
