# 𑁧𑁭𑁇𑁩 𑀔𑀚𑁆𑀚𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀔𑀚𑁆𑀚𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(13.)

2449\. _“𑀢𑀺𑀲𑁆𑀲𑀲𑁆𑀲 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀨𑀮𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀦𑀸𑀍𑀇𑀓𑁂𑀭𑀜𑁆𑀘 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀔𑀚𑁆𑀚𑀓𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀫𑀢𑀁𑁇_  


(14.)

2450\. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀢𑀫𑀳𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀢𑀺𑀲𑁆𑀲𑀲𑁆𑀲 𑀢𑀼 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀫𑁄𑀤𑀸𑀫𑀳𑀁 𑀓𑀸𑀫𑀓𑀸𑀫𑀻,_  
_𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀁 𑀬𑀫𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(15.)

2451\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(16.)

2452\. _𑀇𑀢𑁄 𑀢𑁂𑀭𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀭𑀸𑀚𑀸 𑀇𑀦𑁆𑀤𑀲𑀫𑁄 𑀅𑀳𑀼;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(17.)

2453\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2454\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑀚𑁆𑀚𑀓𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2455\. 𑀔𑀚𑁆𑀚𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
