# 𑁧𑁭𑁇𑁧𑁦 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(50.)

2500\. _“𑀢𑀺𑀲𑁆𑀲𑁄 𑀦𑀸𑀫𑀸𑀲𑀺 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀧𑀯𑀺𑀲𑀢𑀺 𑀕𑀦𑁆𑀥𑀓𑀼𑀝𑀺𑀁,_  
_𑀯𑀺𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀫𑀼𑀦𑀺𑁇_  


(51.)

2501\. _𑀲𑀼𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀫𑀸𑀤𑀸𑀬,_  
_𑀅𑀕𑀫𑀸𑀲𑀺𑀁 𑀚𑀺𑀦𑀦𑁆𑀢𑀺𑀓𑀁;_  
_𑀅𑀧𑀲𑀤𑁆𑀤𑁄 𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂,_  
_𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(52.)

2502\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀕𑀦𑁆𑀥𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(53.)

2503\. _𑀤𑁆𑀯𑁂𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺𑀁 𑀲𑀬𑀫𑁆𑀧𑀪𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(54.)

2504\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁦𑁯𑁮)_  


2505\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2506\. 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

2507\. 𑀲𑀼𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀭𑀲𑀫𑁄𑁇



---

2508\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2509\. _𑀲𑀼𑀧𑀸𑀭𑀺𑀘𑀭𑀺 𑀓𑀡𑀯𑁂𑀭𑀻,_  
_𑀔𑀚𑁆𑀚𑀓𑁄 𑀤𑁂𑀲𑀧𑀽𑀚𑀓𑁄;_  
_𑀓𑀡𑀺𑀓𑀸𑀭𑁄 𑀲𑀧𑁆𑀧𑀺𑀤𑀤𑁄,_  
_𑀬𑀽𑀣𑀺𑀓𑁄 𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑁄;_  
_𑀫𑀸𑀍𑀑 𑀘 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀺𑀓𑁄,_  
_𑀘𑀢𑀼𑀧𑀜𑁆𑀜𑀸𑀲 𑀕𑀸𑀣𑀓𑀸𑀢𑀺𑁇_
