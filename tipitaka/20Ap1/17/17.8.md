# 𑁧𑁭𑁇𑁮 𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(38.)

2484. _“𑀢𑀺𑀯𑀭𑀸𑀬𑀁 𑀧𑀼𑀭𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀧𑀡𑁆𑀡𑀸𑀓𑀸𑀭𑀁 𑀮𑀪𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(39.)

2485. _𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸,_  
_𑀯𑀢𑁆𑀣𑀁 𑀳𑀢𑁆𑀣𑁂𑀦 𑀆𑀫𑀲𑀺;_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑁂𑀳𑀸𑀲𑀁 𑀦𑀪𑀫𑀼𑀕𑁆𑀕𑀫𑀺𑁇_  


(40.)

2486. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀕𑀘𑁆𑀙𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀤𑀼𑀲𑁆𑀲𑀸 𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀧𑀘𑁆𑀙𑀢𑁄;_  
_𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀲𑀺𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑁄 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


(41.)

2487. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀼𑀲𑁆𑀲𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(42.)

2488. _𑀲𑀢𑁆𑀢𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀢𑀤𑀸 𑀅𑀳𑀼𑀁;_  
_𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀫𑀦𑀼𑀚𑀺𑀦𑁆𑀤𑁄 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(43.)

2489. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2490. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2491. 𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
