# 𑁧𑁭𑁇𑁫 𑀓𑀡𑀺𑀓𑀸𑀭𑀙𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀓𑀡𑀺𑀓𑀸𑀭𑀙𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(23.)

2463\. _“𑀯𑁂𑀲𑁆𑀲𑀪𑀽 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀫𑀼𑀦𑀺,_  
_𑀑𑀕𑀸𑀳𑀬𑀺 𑀫𑀳𑀸𑀯𑀦𑀁𑁇_  


(24.)

2464\. _𑀓𑀡𑀺𑀓𑀸𑀭𑀁 𑀑𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸,_  
_𑀙𑀢𑁆𑀢𑀁 𑀓𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀧𑀼𑀧𑁆𑀨𑀘𑁆𑀙𑀤𑀦𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(25.)

2465\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(26.)

2466\. _𑀇𑀢𑁄 𑀯𑀻𑀲𑀢𑀺𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀲𑁄𑀡𑁆𑀡𑀸𑀪𑀸 𑀅𑀝𑁆𑀞 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(27.)

2467\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2468\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀡𑀺𑀓𑀸𑀭𑀙𑀢𑁆𑀢𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2469\. 𑀓𑀡𑀺𑀓𑀸𑀭𑀙𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
