# 𑁧𑁭𑁇𑁪 𑀤𑁂𑀲𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑁂𑀲𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(18.)

2456\. _“𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻 𑀢𑀼 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀯𑁂𑀳𑀸𑀲𑀁,_  
_𑀕𑀘𑁆𑀙𑀢𑁂 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(19.)

2457\. _𑀬𑀫𑁆𑀳𑀺 𑀤𑁂𑀲𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀢𑀸𑀳𑀁 𑀤𑁂𑀲𑀁 𑀅𑀧𑀽𑀚𑁂𑀲𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(20.)

2458\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀬𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑁂𑀲𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(21.)

2459\. _𑀏𑀓𑀸𑀤𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀕𑁄𑀲𑀼𑀚𑀸𑀢𑀲𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(22.)

2460\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2461\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑁂𑀲𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2462\. 𑀤𑁂𑀲𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
