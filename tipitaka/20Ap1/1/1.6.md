# 𑁧𑁇𑁬 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(421.)

434\. _“𑀲𑀼𑀫𑁂𑀥𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀳𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(422.)

435\. _𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀼𑀫𑁂𑀥𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀫𑀬𑀸𑀘𑀳𑀁𑁇_  


(423.)

436\. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪;_  
_𑀧𑀤𑀻𑀧𑀁 𑀢𑁂 𑀧𑀤𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀫𑁆𑀳𑀺 𑀛𑀸𑀬𑀢𑁄𑁇_  


(424.)

437\. _𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀲𑁄 𑀥𑀻𑀭𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄;_  
_𑀤𑀼𑀫𑁂𑀲𑀼 𑀯𑀺𑀦𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑀦𑁆𑀢𑀁 𑀬𑁄𑀚𑀺𑀬𑀳𑀁 𑀢𑀤𑀸𑁇_  


(425.)

438\. _𑀲𑀳𑀲𑁆𑀲𑀯𑀝𑁆𑀝𑀺𑀁 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼𑀦𑁄;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑀻𑀧𑀸 𑀯𑀽𑀧𑀲𑀫𑀺𑀁𑀲𑀼 𑀫𑁂𑁇_  


(426.)

439\. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀯𑀺𑀫𑀸𑀦𑀫𑀼𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(427.)

440\. _𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀤𑁂𑀯𑀢𑁆𑀢𑀁,_  
_𑀩𑁆𑀬𑀫𑁆𑀳𑀁 𑀆𑀲𑀺 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁;_  
_𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀧𑀚𑁆𑀚𑀮𑀢𑀺,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(428.)

441\. _𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀯𑀺𑀭𑁄𑀘𑁂𑀲𑀺𑀫𑀳𑀁 𑀢𑀤𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑁂 𑀅𑀪𑀺𑀪𑁄𑀫𑀺,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(429.)

442\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀦 𑀫𑀁 𑀓𑁂𑀘𑀻𑀢𑀺𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(430.)

443\. _𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀤𑀺𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀁 𑀢𑀤𑀸𑁇_  


(431.)

444\. _𑀲𑀳𑀲𑁆𑀲𑀮𑁄𑀓𑀁 𑀜𑀸𑀡𑁂𑀦,_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀲𑀦𑁂;_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(432.)

445\. _𑀲𑀼𑀫𑁂𑀥𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄;_  
_𑀢𑀲𑁆𑀲 𑀤𑀻𑀧𑁄 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁄,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(433.)

446\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


447\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

448\. 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
