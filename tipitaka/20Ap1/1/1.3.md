# 𑁧𑁇𑁩 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

146\. 𑀅𑀣 𑀣𑁂𑀭𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀼𑀡𑀸𑀣—

(141.)

147\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀮𑀫𑁆𑀩𑀓𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀅𑀲𑁆𑀲𑀫𑁄 𑀲𑀼𑀓𑀢𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀸 𑀲𑀼𑀫𑀸𑀧𑀺𑀢𑀸𑁇_  


(142.)

148\. _𑀉𑀢𑁆𑀢𑀸𑀦𑀓𑀽𑀮𑀸 𑀦𑀤𑀺𑀓𑀸,_  
_𑀲𑀼𑀧𑀢𑀺𑀢𑁆𑀣𑀸 𑀫𑀦𑁄𑀭𑀫𑀸;_  
_𑀲𑀼𑀲𑀼𑀤𑁆𑀥𑀧𑀼𑀮𑀺𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(143.)

149\. _𑀅𑀲𑀓𑁆𑀔𑀭𑀸 𑀅𑀧𑀩𑁆𑀪𑀸𑀭𑀸,_  
_𑀲𑀸𑀤𑀼 𑀅𑀧𑁆𑀧𑀝𑀺𑀕𑀦𑁆𑀥𑀺𑀓𑀸;_  
_𑀲𑀦𑁆𑀤𑀢𑀻 𑀦𑀤𑀺𑀓𑀸 𑀢𑀢𑁆𑀣,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀸 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(144.)

150\. _𑀓𑀼𑀫𑁆𑀪𑀻𑀮𑀸 𑀫𑀓𑀭𑀸 𑀘𑁂𑀢𑁆𑀣,_  
_𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀸 𑀘 𑀓𑀘𑁆𑀙𑀧𑀸;_  
_𑀘𑀭𑀦𑁆𑀢𑀺 𑀦𑀤𑀺𑀬𑀸 𑀢𑀢𑁆𑀣,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀸 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(145.)

151\. _𑀧𑀸𑀞𑀻𑀦𑀸 𑀧𑀸𑀯𑀼𑀲𑀸 𑀫𑀘𑁆𑀙𑀸,_  
_𑀩𑀮𑀚𑀸 𑀫𑀼𑀜𑁆𑀚𑀭𑁄𑀳𑀺𑀢𑀸;_  
_𑀯𑀕𑁆𑀕𑀍𑀆 𑀧𑀧𑀢𑀸𑀬𑀦𑁆𑀢𑀸,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀺 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(146.)

152\. _𑀉𑀪𑁄 𑀓𑀽𑀮𑁂𑀲𑀼 𑀦𑀤𑀺𑀬𑀸,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀦𑁄 𑀨𑀮𑀺𑀦𑁄 𑀤𑀼𑀫𑀸;_  
_𑀉𑀪𑀢𑁄 𑀅𑀪𑀺𑀮𑀫𑁆𑀩𑀦𑁆𑀢𑀸,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀺 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(147.)

153\. _𑀅𑀫𑁆𑀩𑀸 𑀲𑀸𑀮𑀸 𑀘 𑀢𑀺𑀮𑀓𑀸,_  
_𑀧𑀸𑀝𑀮𑀻 𑀲𑀺𑀦𑁆𑀤𑀼𑀯𑀸𑀭𑀓𑀸;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀸 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(148.)

154\. _𑀘𑀫𑁆𑀧𑀓𑀸 𑀲𑀍𑀅𑀮𑀸 𑀦𑀻𑀧𑀸,_  
_𑀦𑀸𑀕𑀧𑀼𑀦𑁆𑀦𑀸𑀕𑀓𑁂𑀢𑀓𑀸;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀸 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(149.)

155\. _𑀅𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀅𑀲𑁄𑀓𑀸 𑀘,_  
_𑀪𑀕𑀺𑀦𑀻𑀫𑀸𑀮𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀅𑀗𑁆𑀓𑁄𑀮𑀸 𑀩𑀺𑀫𑁆𑀩𑀺𑀚𑀸𑀮𑀸 𑀘,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(150.)

156\. _𑀓𑁂𑀢𑀓𑀸 𑀓𑀦𑁆𑀤𑀮𑀺 𑀘𑁂𑀯,_  
_𑀕𑁄𑀥𑀼𑀓𑀸 𑀢𑀺𑀡𑀲𑀽𑀮𑀺𑀓𑀸;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀁 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀸,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀺 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(151.)

157\. _𑀓𑀡𑀺𑀓𑀸𑀭𑀸 𑀓𑀡𑁆𑀡𑀺𑀓𑀸 𑀘,_  
_𑀅𑀲𑀦𑀸 𑀅𑀚𑁆𑀚𑀼𑀦𑀸 𑀩𑀳𑀽;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀁 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀸,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀺 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(152.)

158\. _𑀧𑀼𑀦𑁆𑀦𑀸𑀕𑀸 𑀕𑀺𑀭𑀺𑀧𑀼𑀦𑁆𑀦𑀸𑀕𑀸,_  
_𑀓𑁄𑀯𑀺𑀍𑀆𑀭𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀁 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀸,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀺 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(153.)

159\. _𑀉𑀤𑁆𑀥𑀸𑀮𑀓𑀸 𑀘 𑀓𑀼𑀝𑀚𑀸,_  
_𑀓𑀤𑀫𑁆𑀩𑀸 𑀯𑀓𑀼𑀮𑀸 𑀩𑀳𑀽;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀁 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀸,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀺 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(154.)

160\. _𑀆𑀍𑀅𑀓𑀸 𑀇𑀲𑀺𑀫𑀼𑀕𑁆𑀕𑀸 𑀘,_  
_𑀓𑀤𑀮𑀺𑀫𑀸𑀢𑀼𑀮𑀼𑀗𑁆𑀕𑀺𑀬𑁄;_  
_𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑁂𑀦 𑀲𑀁𑀯𑀟𑁆𑀠𑀸,_  
_𑀨𑀮𑀸𑀦𑀺 𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(155.)

161\. _𑀅𑀜𑁆𑀜𑁂 𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀺 𑀧𑀤𑀼𑀫𑀸,_  
_𑀅𑀜𑁆𑀜𑁂 𑀚𑀸𑀬𑀦𑁆𑀢𑀺 𑀓𑁂𑀲𑀭𑀻;_  
_𑀅𑀜𑁆𑀜𑁂 𑀑𑀧𑀼𑀧𑁆𑀨𑀸 𑀧𑀤𑀼𑀫𑀸,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀢𑀍𑀆𑀓𑁂 𑀢𑀤𑀸𑁇_  


(156.)

162\. _𑀕𑀩𑁆𑀪𑀁 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺 𑀧𑀤𑀼𑀫𑀸,_  
_𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀫𑀼𑀮𑀸𑀍𑀇𑀬𑁄;_  
_𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀫𑀸𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀲𑁄𑀪𑀦𑁆𑀢𑀺 𑀢𑀍𑀆𑀓𑁂 𑀢𑀤𑀸𑁇_  


(157.)

163\. _𑀦𑀬𑀺𑀢𑀸 𑀅𑀫𑁆𑀩𑀕𑀦𑁆𑀥𑀻 𑀘,_  
_𑀉𑀢𑁆𑀢𑀮𑀻 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑀸;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀸 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀢𑀍𑀆𑀓𑁂 𑀢𑀤𑀸𑁇_  


(158.)

164\. _𑀧𑀸𑀞𑀻𑀦𑀸 𑀧𑀸𑀯𑀼𑀲𑀸 𑀫𑀘𑁆𑀙𑀸,_  
_𑀩𑀮𑀚𑀸 𑀫𑀼𑀜𑁆𑀚𑀭𑁄𑀳𑀺𑀢𑀸;_  
_𑀲𑀗𑁆𑀕𑀼𑀮𑀸 𑀫𑀕𑁆𑀕𑀼𑀭𑀸 𑀘𑁂𑀯,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑀍𑀆𑀓𑁂 𑀢𑀤𑀸𑁇_  


(159.)

165\. _𑀓𑀼𑀫𑁆𑀪𑀻𑀮𑀸 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀢𑀦𑁆𑀢𑀺𑀕𑀸𑀳𑀸 𑀘 𑀭𑀓𑁆𑀔𑀲𑀸;_  
_𑀑𑀕𑀼𑀳𑀸 𑀅𑀚𑀕𑀭𑀸 𑀘,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑀍𑀆𑀓𑁂 𑀢𑀤𑀸𑁇_  


(160.)

166\. _𑀧𑀸𑀭𑁂𑀯𑀢𑀸 𑀭𑀯𑀺𑀳𑀁𑀲𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀸 𑀦𑀤𑀻𑀘𑀭𑀸;_  
_𑀓𑁄𑀓𑀺𑀮𑀸 𑀲𑀼𑀓𑀲𑀸𑀍𑀇𑀓𑀸,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(161.)

167\. _𑀓𑀼𑀓𑁆𑀓𑀼𑀢𑁆𑀣𑀓𑀸 𑀓𑀼𑀍𑀈𑀭𑀓𑀸,_  
_𑀯𑀦𑁂 𑀧𑁄𑀓𑁆𑀔𑀭𑀲𑀸𑀢𑀓𑀸;_  
_𑀤𑀺𑀦𑁆𑀤𑀺𑀪𑀸 𑀲𑀼𑀯𑀧𑁄𑀢𑀸 𑀘,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(162.)

168\. _𑀳𑀁𑀲𑀸 𑀓𑁄𑀜𑁆𑀘𑀸 𑀫𑀬𑀽𑀭𑀸 𑀘,_  
_𑀓𑁄𑀓𑀺𑀮𑀸 𑀢𑀫𑁆𑀩𑀘𑀽𑀍𑀅𑀓𑀸;_  
_𑀧𑀫𑁆𑀫𑀓𑀸 𑀚𑀻𑀯𑀁𑀚𑀻𑀯𑀸 𑀘,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(163.)

169\. _𑀓𑁄𑀲𑀺𑀓𑀸 𑀧𑁄𑀝𑁆𑀞𑀲𑀻𑀲𑀸 𑀘,_  
_𑀓𑀼𑀭𑀭𑀸 𑀲𑁂𑀦𑀓𑀸 𑀩𑀳𑀽;_  
_𑀫𑀳𑀸𑀓𑀸𑀍𑀆 𑀘 𑀲𑀓𑀼𑀡𑀸,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(164.)

170\. _𑀧𑀲𑀤𑀸 𑀘 𑀯𑀭𑀸𑀳𑀸 𑀘,_  
_𑀘𑀫𑀭𑀸 𑀕𑀡𑁆𑀟𑀓𑀸 𑀩𑀳𑀽;_  
_𑀭𑁄𑀳𑀺𑀘𑁆𑀘𑀸 𑀲𑀼𑀓𑀧𑁄𑀢𑀸 𑀘,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(165.)

171\. _𑀲𑀻𑀳𑀩𑁆𑀬𑀕𑁆𑀖𑀸 𑀘 𑀤𑀻𑀧𑀻 𑀘,_  
_𑀅𑀘𑁆𑀙𑀓𑁄𑀓𑀢𑀭𑀘𑁆𑀙𑀓𑀸;_  
_𑀢𑀺𑀥𑀸 𑀧𑀪𑀺𑀦𑁆𑀦𑀫𑀸𑀢𑀗𑁆𑀕𑀸,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(166.)

172\. _𑀓𑀺𑀦𑁆𑀦𑀭𑀸 𑀯𑀸𑀦𑀭𑀸 𑀘𑁂𑀯,_  
_𑀅𑀣𑁄𑀧𑀺 𑀯𑀦𑀓𑀫𑁆𑀫𑀺𑀓𑀸;_  
_𑀘𑁂𑀢𑀸 𑀘 𑀮𑀼𑀤𑁆𑀤𑀓𑀸 𑀘𑁂𑀯,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(167.)

173\. _𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀸𑀦𑀺 𑀧𑀺𑀬𑀸𑀮𑀸𑀦𑀺,_  
_𑀫𑀥𑀼𑀓𑁂 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀬𑁄;_  
_𑀥𑀼𑀯𑀁 𑀨𑀮𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(168.)

174\. _𑀓𑁄𑀲𑀫𑁆𑀩𑀸 𑀲𑀍𑀅𑀮𑀸 𑀦𑀺𑀫𑁆𑀩𑀸,_  
_𑀲𑀸𑀤𑀼𑀨𑀮𑀲𑀫𑀸𑀬𑀼𑀢𑀸;_  
_𑀥𑀼𑀯𑀁 𑀨𑀮𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(169.)

175\. _𑀳𑀭𑀻𑀢𑀓𑀸 𑀆𑀫𑀮𑀓𑀸,_  
_𑀅𑀫𑁆𑀩𑀚𑀫𑁆𑀩𑀼𑀯𑀺𑀪𑀻𑀢𑀓𑀸;_  
_𑀓𑁄𑀮𑀸 𑀪𑀮𑁆𑀮𑀸𑀢𑀓𑀸 𑀩𑀺𑀮𑁆𑀮𑀸,_  
_𑀨𑀮𑀸𑀦𑀺 𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(170.)

176\. _𑀆𑀮𑀼𑀯𑀸 𑀘 𑀓𑀍𑀅𑀫𑁆𑀩𑀸 𑀘,_  
_𑀩𑀺𑀍𑀆𑀮𑀻𑀢𑀓𑁆𑀓𑀍𑀆𑀦𑀺 𑀘;_  
_𑀚𑀻𑀯𑀓𑀸 𑀲𑀼𑀢𑀓𑀸 𑀘𑁂𑀯,_  
_𑀩𑀳𑀼𑀓𑀸 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(171.)

177\. _𑀅𑀲𑁆𑀲𑀫𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑀫𑁆𑀳𑀺,_  
_𑀢𑀍𑀆𑀓𑀸𑀲𑀼𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑀸 𑀲𑀻𑀢𑀚𑀮𑀸,_  
_𑀲𑀼𑀧𑀢𑀺𑀢𑁆𑀣𑀸 𑀫𑀦𑁄𑀭𑀫𑀸𑁇_  


(172.)

178\. _𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀲𑀫𑀸𑀬𑀼𑀢𑀸;_  
_𑀫𑀦𑁆𑀤𑀸𑀮𑀓𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑁄 𑀧𑀯𑀸𑀬𑀢𑀺𑁇_  


(173.)

179\. _𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁂 𑀨𑀮𑀺𑀢𑁂 𑀯𑀦𑁂;_  
_𑀲𑀼𑀓𑀢𑁂 𑀅𑀲𑁆𑀲𑀫𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀳𑀁 𑀢𑀤𑀸𑁇_  


(174.)

180\. _𑀲𑀻𑀮𑀯𑀸 𑀯𑀢𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑁄 𑀲𑀤𑀸;_  
_𑀧𑀜𑁆𑀘𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀼𑀭𑀼𑀘𑀺 𑀦𑀸𑀫 𑀢𑀸𑀧𑀲𑁄𑁇_  


(175.)

181\. _𑀘𑀢𑀼𑀯𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀺𑀲𑁆𑀲𑀸 𑀫𑀬𑁆𑀳𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀼𑀁;_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀢𑁂,_  
_𑀚𑀸𑀢𑀺𑀫𑀦𑁆𑀢𑁄 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(176.)

182\. _𑀮𑀓𑁆𑀔𑀡𑁂 𑀇𑀢𑀺𑀳𑀸𑀲𑁂 𑀘,_  
_𑀲𑀦𑀺𑀖𑀡𑁆𑀝𑀼𑀲𑀓𑁂𑀝𑀼𑀪𑁂;_  
_𑀧𑀤𑀓𑀸 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀸,_  
_𑀲𑀥𑀫𑁆𑀫𑁂 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸𑁇_  


(177.)

183\. _𑀉𑀧𑁆𑀧𑀸𑀢𑁂𑀲𑀼 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀲𑀼,_  
_𑀮𑀓𑁆𑀔𑀡𑁂𑀲𑀼 𑀘 𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀪𑀽𑀫𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸𑁇_  


(178.)

184\. _𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀦𑀺𑀧𑀓𑀸 𑀏𑀢𑁂,_  
_𑀅𑀧𑁆𑀧𑀸𑀳𑀸𑀭𑀸 𑀅𑀮𑁄𑀮𑀼𑀧𑀸;_  
_𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑁂𑀦 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(179.)

185\. _𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀁 𑀧𑀢𑁆𑀣𑀬𑀦𑁆𑀢𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(180.)

186\. _𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑁂 𑀕𑁄𑀘𑀭𑁂 𑀭𑀢𑀸;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀘𑀭𑀸 𑀥𑀻𑀭𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(181.)

187\. _𑀲𑀁𑀯𑀼𑀢𑀸 𑀙𑀲𑀼 𑀤𑁆𑀯𑀸𑀭𑁂𑀲𑀼,_  
_𑀅𑀦𑁂𑀚𑀸 𑀭𑀓𑁆𑀔𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸;_  
_𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀘 𑀢𑁂 𑀥𑀻𑀭𑀸,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(182.)

188\. _𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬,_  
_𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀦𑁂𑀦 𑀘;_  
_𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀦𑁆𑀢𑀺 𑀢𑁂 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(183.)

189\. _𑀭𑀚𑀦𑀻𑀬𑁂 𑀦 𑀭𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀤𑀼𑀲𑁆𑀲𑀦𑀻𑀬𑁂 𑀦 𑀤𑀼𑀲𑁆𑀲𑀭𑁂;_  
_𑀫𑁄𑀳𑀦𑀻𑀬𑁂 𑀦 𑀫𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑀺,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(184.)

190\. _𑀇𑀤𑁆𑀥𑀺𑀁 𑀯𑀻𑀫𑀁𑀲𑀫𑀸𑀦𑀸 𑀢𑁂,_  
_𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀦𑀺𑀘𑁆𑀘𑀓𑀸𑀮𑀺𑀓𑀁;_  
_𑀧𑀣𑀯𑀺𑀁 𑀢𑁂 𑀧𑀓𑀫𑁆𑀧𑁂𑀦𑁆𑀢𑀺,_  
_𑀲𑀸𑀭𑀫𑁆𑀪𑁂𑀦 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(185.)

191\. _𑀓𑀻𑀍𑀅𑀫𑀸𑀦𑀸 𑀘 𑀢𑁂 𑀲𑀺𑀲𑁆𑀲𑀸,_  
_𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺 𑀛𑀸𑀦𑀓𑀻𑀍𑀇𑀢𑀁;_  
_𑀚𑀫𑁆𑀩𑀼𑀢𑁄 𑀨𑀮𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(186.)

192\. _𑀅𑀜𑁆𑀜𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀕𑁄𑀬𑀸𑀦𑀁,_  
_𑀅𑀜𑁆𑀜𑁂 𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀤𑁂𑀳𑀓𑀁;_  
_𑀅𑀜𑁆𑀜𑁂 𑀘 𑀉𑀢𑁆𑀢𑀭𑀓𑀼𑀭𑀼𑀁,_  
_𑀏𑀲𑀦𑀸𑀬 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(187.)

193\. _𑀧𑀼𑀭𑀢𑁄 𑀧𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀔𑀸𑀭𑀺𑀁,_  
_𑀧𑀘𑁆𑀙𑀢𑁄 𑀘 𑀯𑀚𑀦𑁆𑀢𑀺 𑀢𑁂;_  
_𑀘𑀢𑀼𑀯𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀙𑀸𑀤𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀫𑁆𑀩𑀭𑀁𑁇_  


(188.)

194\. _𑀅𑀕𑁆𑀕𑀺𑀧𑀸𑀓𑀻 𑀅𑀦𑀕𑁆𑀕𑀻 𑀘,_  
_𑀤𑀦𑁆𑀢𑁄𑀤𑀼𑀓𑁆𑀔𑀮𑀺𑀓𑀸𑀧𑀺 𑀘;_  
_𑀅𑀲𑁆𑀫𑁂𑀦 𑀓𑁄𑀝𑁆𑀝𑀺𑀢𑀸 𑀓𑁂𑀘𑀺,_  
_𑀧𑀯𑀢𑁆𑀢𑀨𑀮𑀪𑁄𑀚𑀦𑀸𑁇_  


(189.)

195\. _𑀉𑀤𑀓𑁄𑀭𑁄𑀳𑀡𑀸 𑀓𑁂𑀘𑀺,_  
_𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑁄 𑀲𑀼𑀘𑀻𑀭𑀢𑀸;_  
_𑀢𑁄𑀬𑀸𑀪𑀺𑀲𑁂𑀘𑀦𑀓𑀭𑀸,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(190.)

196\. _𑀧𑀭𑀽𑀍𑀳𑀓𑀘𑁆𑀙𑀦𑀔𑀮𑁄𑀫𑀸,_  
_𑀧𑀗𑁆𑀓𑀤𑀦𑁆𑀢𑀸 𑀭𑀚𑀲𑁆𑀲𑀺𑀭𑀸;_  
_𑀕𑀦𑁆𑀥𑀺𑀢𑀸 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(191.)

197\. _𑀧𑀸𑀢𑁄𑀯 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸,_  
_𑀚𑀝𑀺𑀮𑀸 𑀉𑀕𑁆𑀕𑀢𑀸𑀧𑀦𑀸;_  
_𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀅𑀫𑁆𑀩𑀭𑁂 𑀢𑀤𑀸𑁇_  


(192.)

198\. _𑀏𑀢𑁂𑀲𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑁄 𑀧𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀅𑀚𑀺𑀦𑀘𑀫𑁆𑀫𑀲𑀤𑁆𑀤𑁂𑀦,_  
_𑀫𑀼𑀤𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀤𑁂𑀯𑀢𑀸𑁇_  


(193.)

199\. _𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀘𑀭𑀸 𑀇𑀲𑀻;_  
_𑀲𑀓𑁂 𑀩𑀮𑁂𑀦𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀸,_  
_𑀢𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(194.)

200\. _𑀧𑀣𑀯𑀻𑀓𑀫𑁆𑀧𑀓𑀸 𑀏𑀢𑁂,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀦𑀪𑀘𑀸𑀭𑀺𑀦𑁄;_  
_𑀉𑀕𑁆𑀕𑀢𑁂𑀚𑀸 𑀤𑀼𑀧𑁆𑀧𑀲𑀳𑀸,_  
_𑀲𑀸𑀕𑀭𑁄𑀯 𑀅𑀔𑁄𑀪𑀺𑀬𑀸𑁇_  


(195.)

201\. _𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀺𑀦𑁄 𑀓𑁂𑀘𑀺,_  
_𑀓𑁂𑀘𑀺 𑀦𑁂𑀲𑀚𑁆𑀚𑀺𑀓𑀸 𑀇𑀲𑀻;_  
_𑀧𑀯𑀢𑁆𑀢𑀪𑁄𑀚𑀦𑀸 𑀓𑁂𑀘𑀺,_  
_𑀫𑀫 𑀲𑀺𑀲𑁆𑀲𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(196.)

202\. _𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄 𑀏𑀢𑁂,_  
_𑀳𑀺𑀢𑁂𑀲𑀻 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀅𑀦𑀢𑁆𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀓𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀦 𑀢𑁂 𑀯𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺 𑀓𑀲𑁆𑀲𑀘𑀺𑁇_  


(197.)

203\. _𑀲𑀻𑀳𑀭𑀸𑀚𑀸𑀯𑀲𑀫𑁆𑀪𑀻𑀢𑀸,_  
_𑀕𑀚𑀭𑀸𑀚𑀸𑀯 𑀣𑀸𑀫𑀯𑀸;_  
_𑀤𑀼𑀭𑀸𑀲𑀤𑀸 𑀩𑁆𑀬𑀕𑁆𑀖𑀸𑀭𑀺𑀯,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(198.)

204\. _𑀯𑀺𑀚𑁆𑀚𑀸𑀥𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀘,_  
_𑀦𑀸𑀕𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀭𑀓𑁆𑀔𑀲𑀸;_  
_𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀸 𑀤𑀸𑀦𑀯𑀸 𑀕𑀭𑀼𑀍𑀆,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(199.)

205\. _𑀢𑁂 𑀚𑀝𑀸𑀔𑀸𑀭𑀺𑀪𑀭𑀺𑀢𑀸,_  
_𑀅𑀚𑀺𑀦𑀼𑀢𑁆𑀢𑀭𑀯𑀸𑀲𑀦𑀸;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀘𑀭𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(200.)

206\. _𑀲𑀤𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀸 𑀏𑀢𑁂,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀕𑀸𑀭𑀯𑀸;_  
_𑀘𑀢𑀼𑀩𑁆𑀩𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀔𑀺𑀧𑀺𑀢𑀲𑀤𑁆𑀤𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(201.)

207\. _𑀧𑀸𑀤𑁂 𑀧𑀸𑀤𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀸,_  
_𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀸 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑀸;_  
_𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫 𑀲𑀩𑁆𑀩𑁂𑀯,_  
_𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀭𑁂 𑀫𑀫𑀁𑁇_  


(202.)

208\. _𑀢𑁂𑀳𑀺 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀲𑀦𑁆𑀢𑁂𑀳𑀺 𑀘 𑀢𑀧𑀲𑁆𑀲𑀺𑀪𑀺;_  
_𑀯𑀲𑀸𑀫𑀺 𑀅𑀲𑁆𑀲𑀫𑁂 𑀢𑀢𑁆𑀣,_  
_𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑁄 𑀅𑀳𑀁𑁇_  


(203.)

209\. _𑀇𑀲𑀻𑀦𑀁 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀧𑀼𑀧𑁆𑀨𑀕𑀦𑁆𑀥𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀨𑀮𑀻𑀦𑀁 𑀨𑀮𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀕𑀦𑁆𑀥𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑁆𑀲𑀫𑁄𑁇_  


(204.)

210\. _𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀭𑀢𑀺 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀓𑁂 𑀲𑀺𑀲𑁆𑀲𑁂 𑀑𑀯𑀤𑀦𑁆𑀢𑁄,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀳𑀸𑀲𑀁 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(205.)

211\. _𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀨𑀮𑀸𑀦𑀜𑁆𑀘 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀸 𑀧𑀯𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑁄𑀪𑀬𑀦𑁆𑀢𑀸 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(206.)

212\. _𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀆𑀢𑀸𑀧𑀻 𑀦𑀺𑀧𑀓𑁄 𑀅𑀳𑀁;_  
_𑀔𑀸𑀭𑀺𑀪𑀸𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀳𑀺𑀁 𑀅𑀳𑀁𑁇_  


(207.)

213\. _𑀉𑀧𑁆𑀧𑀸𑀢𑁂 𑀲𑀼𑀧𑀺𑀦𑁂 𑀘𑀸𑀧𑀺,_  
_𑀮𑀓𑁆𑀔𑀡𑁂𑀲𑀼 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀁 𑀫𑀦𑁆𑀢𑀧𑀤𑀁,_  
_𑀥𑀸𑀭𑀬𑀸𑀫𑀺 𑀅𑀳𑀁 𑀢𑀤𑀸𑁇_  


(208.)

214\. _𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀯𑀺𑀯𑁂𑀓𑀓𑀸𑀫𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(209.)

215\. _𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀁,_  
_𑀅𑀕𑁆𑀕𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀫𑀼𑀦𑀺;_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(210.)

216\. _𑀢𑀫𑀤𑁆𑀤𑀲𑀸𑀳𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀁 𑀫𑀦𑁄𑀭𑀫𑀁;_  
_𑀇𑀦𑁆𑀤𑀻𑀯𑀭𑀁𑀯 𑀚𑀮𑀺𑀢𑀁,_  
_𑀆𑀤𑀺𑀢𑁆𑀢𑀁𑀯 𑀳𑀼𑀢𑀸𑀲𑀦𑀁𑁇_  


(211.)

217\. _𑀚𑀮𑀦𑁆𑀢𑀁 𑀤𑀻𑀧𑀭𑀼𑀓𑁆𑀔𑀁𑀯,_  
_𑀯𑀺𑀚𑁆𑀚𑀼𑀢𑀁 𑀕𑀕𑀦𑁂 𑀬𑀣𑀸;_  
_𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀁 𑀲𑀸𑀮𑀭𑀸𑀚𑀁𑀯,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(212.)

218\. _𑀅𑀬𑀁 𑀦𑀸𑀕𑁄 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀓𑀭𑁄 𑀫𑀼𑀦𑀺;_  
_𑀇𑀫𑀁 𑀤𑀲𑁆𑀲𑀦𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀭𑁂𑁇_  


(213.)

219\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀸𑀳𑀁 𑀤𑁂𑀯𑀤𑁂𑀯𑀁,_  
_𑀮𑀓𑁆𑀔𑀡𑀁 𑀉𑀧𑀥𑀸𑀭𑀬𑀺𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑀼 𑀔𑁄 𑀦 𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀳𑀦𑁆𑀤 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀁𑁇_  


(214.)

220\. _𑀲𑀳𑀲𑁆𑀲𑀸𑀭𑀸𑀦𑀺 𑀘𑀓𑁆𑀓𑀸𑀦𑀺,_  
_𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀘𑀭𑀡𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀮𑀓𑁆𑀔𑀡𑀸𑀦𑀺𑀲𑁆𑀲 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀝𑁆𑀞𑀁 𑀕𑀘𑁆𑀙𑀺𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑁇_  


(215.)

221\. _𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀣 𑀧𑀼𑀧𑁆𑀨𑁂 𑀲𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀅𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(216.)

222\. _𑀧𑀽𑀚𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑀫𑀦𑀸𑀲𑀯𑀁;_  
_𑀏𑀓𑀁𑀲𑀁 𑀅𑀚𑀺𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦𑀫𑀲𑁆𑀲𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(217.)

223\. _𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀢𑀁 𑀜𑀸𑀡𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(218.)

224\. _‘𑀲𑀫𑀼𑀤𑁆𑀥𑀭𑀲𑀺𑀫𑀁 𑀮𑁄𑀓𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀫𑀺𑀢𑁄𑀤𑀬;_  
_𑀢𑀯 𑀤𑀲𑁆𑀲𑀦𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀓𑀗𑁆𑀔𑀸𑀲𑁄𑀢𑀁 𑀢𑀭𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(219.)

225\. _𑀢𑀼𑀯𑀁 𑀲𑀢𑁆𑀣𑀸 𑀘 𑀓𑁂𑀢𑀼 𑀘,_  
_𑀥𑀚𑁄 𑀬𑀽𑀧𑁄 𑀘 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀧𑀭𑀸𑀬𑀡𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀘,_  
_𑀤𑀻𑀧𑁄 𑀘 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(220.)

226\. _𑀲𑀓𑁆𑀓𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁,_  
_𑀧𑀫𑁂𑀢𑀼𑀁 𑀆𑀍𑀳𑀓𑁂𑀦 𑀯𑀸;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀯 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼,_  
_𑀜𑀸𑀡𑀁 𑀲𑀓𑁆𑀓𑀸 𑀧𑀫𑁂𑀢𑀯𑁂𑁇_  


(221.)

227\. _𑀥𑀸𑀭𑁂𑀢𑀼𑀁 𑀧𑀣𑀯𑀺𑀁 𑀲𑀓𑁆𑀓𑀸,_  
_𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀼𑀮𑀫𑀡𑁆𑀟𑀮𑁂;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀯 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼,_  
_𑀜𑀸𑀡𑀁 𑀲𑀓𑁆𑀓𑀸 𑀥𑀭𑁂𑀢𑀯𑁂𑁇_  


(222.)

228\. _𑀆𑀓𑀸𑀲𑁄 𑀫𑀺𑀦𑀺𑀢𑀼𑀁 𑀲𑀓𑁆𑀓𑀸,_  
_𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀅𑀗𑁆𑀕𑀼𑀮𑁂𑀦 𑀯𑀸;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀯 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼,_  
_𑀜𑀸𑀡𑀁 𑀲𑀓𑁆𑀓𑀸 𑀧𑀫𑁂𑀢𑀯𑁂𑁇_  


(223.)

229\. _𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁,_  
_𑀧𑀣𑀯𑀻 𑀘𑀸𑀔𑀺𑀮𑀸𑀚𑀝𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀜𑀸𑀡𑀁 𑀉𑀧𑀸𑀤𑀸𑀬,_  
_𑀉𑀧𑀫𑀸𑀢𑁄 𑀦 𑀬𑀼𑀚𑁆𑀚𑀭𑁂𑁇_  


(224.)

230\. _𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀬𑁂𑀲𑀁 𑀧𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀅𑀦𑁆𑀢𑁄𑀚𑀸𑀮𑀕𑀢𑀸 𑀏𑀢𑁂,_  
_𑀢𑀯 𑀜𑀸𑀡𑀫𑁆𑀳𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑁇_  


(225.)

231\. _𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀧𑀢𑁆𑀢𑁄𑀲𑀺,_  
_𑀓𑁂𑀯𑀮𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀢𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼,_  
_𑀫𑀤𑁆𑀤𑀲𑀻 𑀧𑀭𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂’𑁇_  


(226.)

232\. _𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀣𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀭𑀼𑀘𑀺 𑀦𑀸𑀫 𑀢𑀸𑀧𑀲𑁄;_  
_𑀅𑀚𑀺𑀦𑀁 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀣𑀯𑀺𑀬𑀁 𑀦𑀺𑀲𑀻𑀤𑀺 𑀲𑁄𑁇_  


(227.)

233\. _‘𑀘𑀼𑀮𑁆𑀮𑀸𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀍𑀳𑁄 𑀫𑀳𑀡𑁆𑀡𑀯𑁂;_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀢𑀸𑀯𑀤𑁂𑀯,_  
_𑀕𑀺𑀭𑀺𑀭𑀸𑀚𑀸 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(228.)

234\. _𑀢𑀸𑀯 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀦𑁂𑀭𑀼,_  
_𑀆𑀬𑀢𑁄 𑀯𑀺𑀢𑁆𑀣𑀢𑁄 𑀘 𑀲𑁄;_  
_𑀘𑀼𑀡𑁆𑀡𑀺𑀢𑁄 𑀅𑀡𑀼𑀪𑁂𑀤𑁂𑀦,_  
_𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀲𑁄𑁇_  


(229.)

235\. _𑀮𑀓𑁆𑀔𑁂 𑀞𑀧𑀺𑀬𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀫𑀕𑀘𑁆𑀙𑀣;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀯 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼,_  
_𑀜𑀸𑀡𑀁 𑀲𑀓𑁆𑀓𑀸 𑀧𑀫𑁂𑀢𑀯𑁂𑁇_  


(230.)

236\. _𑀲𑀼𑀔𑀼𑀫𑀘𑁆𑀙𑀺𑀓𑁂𑀦 𑀚𑀸𑀮𑁂𑀦,_  
_𑀉𑀤𑀓𑀁 𑀬𑁄 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀧𑁂;_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀉𑀤𑀓𑁂 𑀧𑀸𑀡𑀸,_  
_𑀅𑀦𑁆𑀢𑁄𑀚𑀸𑀮𑀕𑀢𑀸 𑀲𑀺𑀬𑀼𑀁𑁇_  


(231.)

237\. _𑀢𑀣𑁂𑀯 𑀳𑀺 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀧𑀼𑀣𑀼𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸;_  
_𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀸,_  
_𑀧𑀭𑀸𑀫𑀸𑀲𑁂𑀦 𑀫𑁄𑀳𑀺𑀢𑀸𑁇_  


(232.)

238\. _𑀢𑀯 𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀜𑀸𑀡𑁂𑀦,_  
_𑀅𑀦𑀸𑀯𑀭𑀡𑀤𑀲𑁆𑀲𑀺𑀦𑀸;_  
_𑀅𑀦𑁆𑀢𑁄𑀚𑀸𑀮𑀕𑀢𑀸 𑀏𑀢𑁂,_  
_𑀜𑀸𑀡𑀁 𑀢𑁂 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀭𑁂’𑁇_  


(233.)

239\. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸,_  
_𑀤𑀺𑀲𑀁 𑀑𑀮𑁄𑀓𑀬𑀻 𑀚𑀺𑀦𑁄𑁇_  


(234.)

240\. _𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀦𑀺𑀲𑀪𑁄 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄;_  
_𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀪𑀺𑁇_  


(235.)

241\. _𑀔𑀻𑀡𑀸𑀲𑀯𑁂𑀳𑀺 𑀲𑀼𑀤𑁆𑀥𑁂𑀳𑀺,_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀳𑀺 𑀛𑀸𑀬𑀺𑀪𑀺;_  
_𑀘𑀺𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀉𑀧𑁂𑀲𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(236.)

242\. _𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀞𑀺𑀢𑀸 𑀢𑀢𑁆𑀣,_  
_𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀫𑀓𑀁𑀲𑀼 𑀢𑁂;_  
_𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸,_  
_𑀑𑀢𑀭𑀼𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(237.)

243\. _𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀻𑀢𑀁 𑀧𑀸𑀢𑀼𑀓𑀭𑀻 𑀚𑀺𑀦𑁄𑁇_  


(238.)

244\. _𑀯𑀭𑀼𑀡𑁄 𑀦𑀸𑀫𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀏𑀓𑀁𑀲𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀧𑀼𑀘𑁆𑀙𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(239.)

245\. _‘𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀳𑁂𑀢𑀼,_  
_𑀲𑀺𑀢𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀦 𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀳𑁂𑀢𑀽𑀳𑀺,_  
_𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀢𑁂’𑁇_  


(240.)

246\. _𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑀚𑁆𑀛𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀣𑁇_  


(241.)

247\. _‘𑀬𑁄 𑀫𑀁 𑀧𑀼𑀧𑁆𑀨𑁂𑀦 𑀧𑀽𑀚𑁂𑀲𑀺,_  
_𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀧𑀺 𑀅𑀦𑀼𑀢𑁆𑀣𑀯𑀺;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑁄𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄’𑁇_  


(242.)

248\. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀕𑀺𑀭𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀲𑁄𑀢𑀼𑀓𑀸𑀫𑀸 𑀢𑁂,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀫𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀼𑀁𑁇_  


(243.)

249\. _𑀤𑀲𑀲𑀼 𑀮𑁄𑀓𑀥𑀸𑀢𑀽𑀲𑀼,_  
_𑀤𑁂𑀯𑀓𑀸𑀬𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸;_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀲𑁄𑀢𑀼𑀓𑀸𑀫𑀸 𑀢𑁂,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀫𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀼𑀁𑁇_  


(244.)

250\. _‘𑀳𑀢𑁆𑀣𑀻 𑀅𑀲𑁆𑀲𑀸 𑀭𑀣𑀸 𑀧𑀢𑁆𑀢𑀻,_  
_𑀲𑁂𑀦𑀸 𑀘 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀻;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(245.)

251\. _𑀲𑀝𑁆𑀞𑀺𑀢𑀽𑀭𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀪𑁂𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(246.)

252\. _𑀲𑁄𑀍𑀅𑀲𑀺𑀢𑁆𑀣𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀸𑀪𑀭𑀡𑀸,_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀸𑁇_  


(247.)

253\. _𑀅𑀍𑀆𑀭𑀧𑀫𑁆𑀳𑀸 𑀳𑀲𑀼𑀮𑀸,_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀸 𑀢𑀦𑀼𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(248.)

254\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻,_  
_𑀭𑀸𑀚𑀸 𑀭𑀝𑁆𑀞𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(249.)

255\. _𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(250.)

256\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀲𑀸𑀭𑀺𑀬𑀸 𑀦𑀸𑀫,_  
_𑀥𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀼𑀘𑁆𑀙𑀺𑀦𑀸𑁇_  


(251.)

257\. _𑀫𑀸𑀢𑀼𑀬𑀸 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀬𑀁 𑀦𑀭𑁄;_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀢𑀺𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(252.)

258\. _𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀻 𑀙𑀟𑁆𑀟𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑀓𑀺𑀜𑁆𑀘𑀦𑁄;_  
_𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀧𑀤𑀁,_  
_𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀳𑀺𑀁 𑀇𑀫𑀁𑁇_  


(253.)

259\. _𑀅𑀧𑁆𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(254.)

260\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑁄𑁇_  


(255.)

261\. _𑀅𑀬𑀁 𑀪𑀸𑀕𑀻𑀭𑀣𑀻 𑀕𑀗𑁆𑀕𑀸,_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀸 𑀧𑀪𑀸𑀯𑀺𑀢𑀸;_  
_𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀫𑀧𑁆𑀧𑁂𑀢𑀺,_  
_𑀢𑀧𑁆𑀧𑀬𑀦𑁆𑀢𑀻 𑀫𑀳𑁄𑀤𑀥𑀺𑀁𑁇_  


(256.)

262\. _𑀢𑀣𑁂𑀯𑀸𑀬𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀲𑀓𑁂 𑀢𑀻𑀲𑀼 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀢𑀧𑁆𑀧𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑁂𑁇_  


(257.)

263\. _𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑀼𑀧𑀸𑀤𑀸𑀬,_  
_𑀲𑀸𑀕𑀭𑀜𑁆𑀘 𑀫𑀳𑁄𑀤𑀥𑀺𑀁;_  
_𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀬𑀁 𑀧𑀼𑀮𑀺𑀦𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(258.)

264\. _𑀢𑀫𑁆𑀧𑀺 𑀲𑀓𑁆𑀓𑀸 𑀅𑀲𑁂𑀲𑁂𑀦,_  
_𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁 𑀕𑀡𑀦𑀸 𑀬𑀣𑀸;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀧𑀜𑁆𑀜𑀸𑀬𑀦𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(259.)

265\. _𑀮𑀓𑁆𑀔𑁂 𑀞𑀧𑀺𑀬𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀔𑀻𑀬𑁂 𑀕𑀗𑁆𑀕𑀸𑀬 𑀯𑀸𑀮𑀼𑀓𑀸;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀧𑀜𑁆𑀜𑀸𑀬𑀦𑁆𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(260.)

266\. _𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀊𑀫𑀺𑀬𑁄,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀸;_  
_𑀢𑀣𑁂𑀯 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀧𑀜𑁆𑀜𑀸𑀬𑀦𑁆𑀢𑁄 𑀦 𑀳𑁂𑀲𑁆𑀲𑀢𑀺𑁇_  


(261.)

267\. _𑀆𑀭𑀸𑀥𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑁄𑀢𑀫𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑁄𑁇_  


(262.)

268\. _𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀸;_  
_𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀫𑀸,_  
_𑀯𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀯𑀼𑀝𑁆𑀞𑀺𑀬𑁄𑁇_  


(263.)

269\. _𑀲𑀩𑁆𑀩𑀫𑁂𑀢𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀕𑁆𑀕𑀝𑁆𑀞𑀸𑀦𑁂 𑀞𑀧𑁂𑀲𑁆𑀲𑀢𑀺’𑁇_  


(264.)

270\. _𑀅𑀳𑁄 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑁄𑁇_  


(265.)

271\. _𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁂 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀨𑀮𑀁 𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀫𑁂 𑀇𑀥;_  
_𑀲𑀼𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑀭𑀯𑁂𑀕𑁄𑀯,_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(266.)

272\. _𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁;_  
_𑀯𑀺𑀘𑀺𑀦𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀏𑀲𑀸𑀳𑀁 𑀲𑀁𑀲𑀭𑀺𑀁 𑀪𑀯𑁂𑁇_  


(267.)

273\. _𑀬𑀣𑀸𑀧𑀺 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀧𑀭𑀺𑀬𑁂𑀲𑁂𑀬𑁆𑀬 𑀑𑀲𑀥𑀁;_  
_𑀯𑀺𑀘𑀺𑀦𑁂𑀬𑁆𑀬 𑀯𑀦𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁄 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(268.)

274\. _𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁;_  
_𑀅𑀩𑁆𑀩𑁄𑀓𑀺𑀡𑁆𑀡𑀁 𑀧𑀜𑁆𑀘𑀲𑀢𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀇𑀲𑀺𑀧𑀩𑁆𑀩𑀚𑀁𑁇_  


(269.)

275\. _𑀚𑀝𑀸𑀪𑀸𑀭𑁂𑀦 𑀪𑀭𑀺𑀢𑁄,_  
_𑀅𑀚𑀺𑀦𑀼𑀢𑁆𑀢𑀭𑀦𑀺𑀯𑀸𑀲𑀦𑁄;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀁 𑀅𑀕𑀘𑁆𑀙𑀺𑀳𑀁𑁇_  


(270.)

276\. _𑀦𑀢𑁆𑀣𑀺 𑀩𑀸𑀳𑀺𑀭𑀓𑁂 𑀲𑀼𑀤𑁆𑀥𑀺,_  
_𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁;_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀩𑀼𑀤𑁆𑀥𑀺𑀫𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀲𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂𑁇_  


(271.)

277\. _𑀅𑀢𑁆𑀢𑀓𑀸𑀭𑀫𑀬𑀁 𑀏𑀢𑀁,_  
_𑀦𑀬𑀺𑀤𑀁 𑀇𑀢𑀺𑀳𑀻𑀢𑀺𑀳𑀁;_  
_𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀼𑀢𑀺𑀢𑁆𑀣𑁂 𑀲𑀜𑁆𑀘𑀭𑀺𑀁 𑀅𑀳𑀁𑁇_  


(272.)

278\. _𑀬𑀣𑀸 𑀲𑀸𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀧𑁄𑀲𑁄,_  
_𑀓𑀤𑀮𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸𑀦 𑀨𑀸𑀮𑀬𑁂;_  
_𑀦 𑀢𑀢𑁆𑀣 𑀲𑀸𑀭𑀁 𑀯𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬,_  
_𑀲𑀸𑀭𑁂𑀦 𑀭𑀺𑀢𑁆𑀢𑀓𑁄 𑀳𑀺 𑀲𑁄𑁇_  


(273.)

279\. _𑀢𑀣𑁂𑀯 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀮𑁄𑀓𑁂,_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀩𑀳𑀼𑀚𑁆𑀚𑀦𑀸;_  
_𑀅𑀲𑀗𑁆𑀔𑀢𑁂𑀦 𑀭𑀺𑀢𑁆𑀢𑀸𑀲𑁂,_  
_𑀲𑀸𑀭𑁂𑀦 𑀓𑀤𑀮𑀻 𑀬𑀣𑀸𑁇_  


(274.)

280\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀫𑀳𑀸𑀪𑁄𑀕𑀁 𑀙𑀟𑁆𑀟𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁”𑁇_  


---

281\. 𑀧𑀞𑀫𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇



(275.)

282\. _“𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀜𑁆𑀘𑀬𑁄 𑀦𑀸𑀫,_  
_𑀢𑀲𑁆𑀲 𑀫𑀽𑀮𑁂 𑀯𑀲𑀸𑀫𑀳𑀁𑁇_  


(276.)

283\. _𑀲𑀸𑀯𑀓𑁄 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀅𑀲𑁆𑀲𑀚𑀺 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀤𑀼𑀭𑀸𑀲𑀤𑁄 𑀉𑀕𑁆𑀕𑀢𑁂𑀚𑁄,_  
_𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀻 𑀢𑀤𑀸𑁇_  


(277.)

284\. _𑀢𑀫𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀁,_  
_𑀫𑀼𑀦𑀺𑀁 𑀫𑁄𑀦𑁂 𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀳𑀸𑀦𑀸𑀕𑀁,_  
_𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀁 𑀧𑀤𑀼𑀫𑀁 𑀬𑀣𑀸𑁇_  


(278.)

285\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀫𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺,_  
_𑀲𑀼𑀤𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑁆𑀥𑀫𑀸𑀦𑀲𑀁;_  
_𑀉𑀲𑀪𑀁 𑀧𑀯𑀭𑀁 𑀯𑀻𑀭𑀁,_  
_𑀅𑀭𑀳𑀸𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(279.)

286\. _𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁄 𑀇𑀭𑀺𑀬𑀢𑀺,_  
_𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑁄;_  
_𑀉𑀢𑁆𑀢𑀫𑁂 𑀤𑀫𑀣𑁂 𑀤𑀦𑁆𑀢𑁄,_  
_𑀅𑀫𑀢𑀤𑀲𑁆𑀲𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(280.)

287\. _𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀢𑀼𑀝𑁆𑀞𑀫𑀸𑀦𑀲𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀧𑀼𑀝𑁆𑀞𑁄 𑀓𑀣𑁂𑀲𑁆𑀲𑀢𑀺,_  
_𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀢𑀤𑀸𑁇_  


(281.)

288\. _𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀧𑀘𑁆𑀙𑀢𑁄 𑀅𑀕𑀫𑀸𑀲𑀳𑀁;_  
_𑀑𑀓𑀸𑀲𑀁 𑀧𑀝𑀺𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑁄,_  
_𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁𑁇_  


(282.)

289\. _𑀯𑀻𑀣𑀺𑀦𑁆𑀢𑀭𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀧𑀼𑀘𑁆𑀙𑀳𑀁;_  
_‘𑀓𑀣𑀁 𑀕𑁄𑀢𑁆𑀢𑁄𑀲𑀺 𑀢𑁆𑀯𑀁 𑀯𑀻𑀭,_  
_𑀓𑀲𑁆𑀲 𑀲𑀺𑀲𑁆𑀲𑁄𑀲𑀺 𑀫𑀸𑀭𑀺𑀲’𑁇_  


(283.)

290\. _𑀲𑁄 𑀫𑁂 𑀧𑀼𑀝𑁆𑀞𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀅𑀲𑀫𑁆𑀪𑀻𑀢𑁄𑀯 𑀓𑁂𑀲𑀭𑀻;_  
_‘𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀢𑀲𑁆𑀲 𑀲𑀺𑀲𑁆𑀲𑁄𑀫𑁆𑀳𑀺 𑀆𑀯𑀼𑀲𑁄’𑁇_  


(284.)

291\. _‘𑀓𑀻𑀤𑀺𑀲𑀁 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀅𑀦𑀼𑀚𑀸𑀢 𑀫𑀳𑀸𑀬𑀲;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀸𑀥𑀼 𑀫𑁂 𑀓𑀣𑀬𑀲𑁆𑀲𑀼 𑀪𑁄’𑁇_  


(285.)

292\. _𑀲𑁄 𑀫𑁂 𑀧𑀼𑀝𑁆𑀞𑁄 𑀓𑀣𑀻 𑀲𑀩𑁆𑀩𑀁,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀦𑀺𑀧𑀼𑀡𑀁 𑀧𑀤𑀁;_  
_𑀢𑀡𑁆𑀳𑀸𑀲𑀮𑁆𑀮𑀲𑁆𑀲 𑀳𑀦𑁆𑀢𑀸𑀭𑀁,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀦𑀽𑀤𑀦𑀁𑁇_  


(286.)

293\. _‘𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀼𑀧𑁆𑀧𑀪𑀯𑀸,_  
_𑀢𑁂𑀲𑀁 𑀳𑁂𑀢𑀼𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀆𑀳;_  
_𑀢𑁂𑀲𑀜𑁆𑀘 𑀬𑁄 𑀦𑀺𑀭𑁄𑀥𑁄,_  
_𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀫𑀳𑀸𑀲𑀫𑀡𑁄’𑁇_  


(287.)

294\. _𑀲𑁄𑀳𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁂 𑀧𑀜𑁆𑀳𑁂,_  
_𑀧𑀞𑀫𑀁 𑀨𑀮𑀫𑀚𑁆𑀛𑀕𑀁;_  
_𑀯𑀺𑀭𑀚𑁄 𑀯𑀺𑀫𑀮𑁄 𑀆𑀲𑀺𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁𑁇_  


(288.)

295\. _𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀫𑀼𑀦𑀺𑀦𑁄 𑀯𑀸𑀓𑁆𑀬𑀁,_  
_𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄,_  
_𑀇𑀫𑀁 𑀕𑀸𑀣𑀫𑀪𑀸𑀲𑀳𑀁𑁇_  


(289.)

296\. _‘𑀏𑀲𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀬𑀤𑀺 𑀢𑀸𑀯𑀤𑁂𑀯,_  
_𑀧𑀘𑁆𑀘𑀩𑁆𑀬𑀣 𑀧𑀤𑀫𑀲𑁄𑀓𑀁;_  
_𑀅𑀤𑀺𑀝𑁆𑀞𑀁 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀁,_  
_𑀩𑀳𑀼𑀓𑁂𑀳𑀺 𑀓𑀧𑁆𑀧𑀦𑀳𑀼𑀢𑁂𑀳𑀺’𑁇_  


(290.)

297\. _𑀲𑁆𑀯𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀼𑀢𑀺𑀢𑁆𑀣𑁂 𑀲𑀜𑁆𑀘𑀭𑀺𑀁 𑀅𑀳𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀓𑀸𑀮𑁄 𑀫𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇_  


(291.)

298\. _𑀢𑁄𑀲𑀺𑀢𑁄𑀳𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀦𑀸,_  
_𑀧𑀢𑁆𑀯𑀸𑀦 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁;_  
_𑀲𑀳𑀸𑀬𑀓𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀅𑀲𑁆𑀲𑀫𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁𑁇_  


(292.)

299\. _𑀤𑀽𑀭𑀢𑁄𑀯 𑀫𑀫𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀳𑀸𑀬𑁄 𑀫𑁂 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑁇_  


(293.)

300\. _‘𑀧𑀲𑀦𑁆𑀦𑀫𑀼𑀔𑀦𑁂𑀢𑁆𑀢𑁄𑀲𑀺,_  
_𑀫𑀼𑀦𑀺𑀪𑀸𑀯𑁄𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀫𑀢𑀸𑀥𑀺𑀕𑀢𑁄 𑀓𑀘𑁆𑀘𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑀘𑁆𑀘𑀼𑀢𑀁 𑀧𑀤𑀁𑁇_  


(294.)

301\. _𑀲𑀼𑀪𑀸𑀦𑀼𑀭𑀽𑀧𑁄 𑀆𑀬𑀸𑀲𑀺,_  
_𑀆𑀦𑁂𑀜𑁆𑀚𑀓𑀸𑀭𑀺𑀢𑁄 𑀯𑀺𑀬;_  
_𑀤𑀦𑁆𑀢𑁄𑀯 𑀤𑀦𑁆𑀢𑀤𑀫𑀣𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄𑀲𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡’𑁇_  


(295.)

302\. _‘𑀅𑀫𑀢𑀁 𑀫𑀬𑀸𑀥𑀺𑀕𑀢𑀁,_  
_𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀸𑀧𑀦𑀽𑀤𑀦𑀁;_  
_𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀢𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀲𑀺,_  
_𑀕𑀘𑁆𑀙𑀸𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁’𑁇_  


(296.)

303\. _𑀲𑀸𑀥𑀽𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀳𑀸𑀬𑁄 𑀫𑁂 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀳𑀢𑁆𑀣𑁂𑀦 𑀳𑀢𑁆𑀣𑀁 𑀕𑀡𑁆𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀉𑀧𑀕𑀫𑁆𑀫 𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(297.)

304\. _𑀉𑀪𑁄𑀧𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢 𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀯 𑀲𑀸𑀲𑀦𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀯𑀺𑀳𑀭𑀸𑀫 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(298.)

305\. _𑀓𑁄𑀮𑀺𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀅𑀳𑀁 𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀭𑀕𑁄;_  
_𑀉𑀪𑁄𑀯 𑀏𑀓𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀸𑀲𑀦𑀁 𑀲𑁄𑀪𑀬𑀸𑀫𑀲𑁂𑁇_  


(299.)

306\. _𑀅𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀓𑀼𑀢𑀺𑀢𑁆𑀣𑁂 𑀲𑀜𑁆𑀘𑀭𑀺𑀁 𑀅𑀳𑀁;_  
_𑀢𑀯 𑀤𑀲𑁆𑀲𑀦𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀽𑀭𑀺𑀢𑁄 𑀫𑀫𑁇_  


(300.)

307\. _𑀧𑀣𑀯𑀺𑀬𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀺 𑀲𑀫𑀬𑁂 𑀤𑀼𑀫𑀸;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀸 𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀺,_  
_𑀢𑁄𑀲𑁂𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(301.)

308\. _𑀢𑀣𑁂𑀯𑀸𑀳𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢 𑀫𑀳𑀸𑀬𑀲;_  
_𑀲𑀸𑀲𑀦𑁂 𑀢𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀲𑀫𑀬𑁂𑀲𑀸𑀫𑀺 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀼𑀁𑁇_  


(302.)

309\. _𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀁 𑀏𑀲𑀦𑁆𑀢𑁄,_  
_𑀪𑀯𑀲𑀁𑀲𑀸𑀭𑀫𑁄𑀘𑀦𑀁;_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀮𑀸𑀪𑁂𑀦,_  
_𑀢𑁄𑀲𑁂𑀫𑀺 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(303.)

310\. _𑀬𑀸𑀯𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀤𑀺𑀲𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀢𑀯 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼𑀫𑁇_  


(304.)

311\. _𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑀸 𑀘 𑀢𑁂 𑀲𑀺𑀲𑁆𑀲𑀸,_  
_𑀧𑀭𑀺𑀲𑀸 𑀘 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸;_  
_𑀉𑀢𑁆𑀢𑀫𑁂 𑀤𑀫𑀣𑁂 𑀤𑀦𑁆𑀢𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(305.)

312\. _𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀫𑀼𑀦𑀻 𑀫𑁄𑀦𑁂𑀬𑁆𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(306.)

313\. _𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀦𑀺𑀧𑀓𑀸 𑀥𑀻𑀭𑀸,_  
_𑀅𑀧𑁆𑀧𑀸𑀳𑀸𑀭𑀸 𑀅𑀮𑁄𑀮𑀼𑀧𑀸;_  
_𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑁂𑀦 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(307.)

314\. _𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀥𑀼𑀢𑀭𑀢𑀸,_  
_𑀛𑀸𑀬𑀺𑀦𑁄 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸;_  
_𑀯𑀺𑀯𑁂𑀓𑀸𑀪𑀺𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(308.)

315\. _𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀨𑀮𑀝𑁆𑀞𑀸 𑀘,_  
_𑀲𑁂𑀔𑀸 𑀨𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄;_  
_𑀆𑀲𑀻𑀲𑀓𑀸 𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(309.)

316\. _𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀘 𑀯𑀺𑀫𑀮𑀸,_  
_𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀬𑁂;_  
_𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀘 𑀅𑀭𑀳𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(310.)

317\. _𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀸,_  
_𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑀸;_  
_𑀲𑀸𑀯𑀓𑀸 𑀢𑁂 𑀩𑀳𑀽 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(311.)

318\. _𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀓𑀼𑀲𑀮𑀸,_  
_𑀲𑀫𑀸𑀥𑀺𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑀸;_  
_𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(312.)

319\. _𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀘,_  
_𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀁 𑀧𑀢𑁆𑀢𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(313.)

320\. _𑀏𑀤𑀺𑀲𑀸 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀢𑀯 𑀲𑀺𑀲𑁆𑀲𑀸 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸;_  
_𑀤𑀼𑀭𑀸𑀲𑀤𑀸 𑀉𑀕𑁆𑀕𑀢𑁂𑀚𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(314.)

321\. _𑀢𑁂𑀳𑀺 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀲𑀜𑁆𑀜𑀢𑁂𑀳𑀺 𑀢𑀧𑀲𑁆𑀲𑀺𑀪𑀺;_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀸𑀯𑀲𑀫𑁆𑀪𑀻𑀢𑁄,_  
_𑀉𑀍𑀉𑀭𑀸𑀚𑀸𑀯 𑀲𑁄𑀪𑀲𑀺𑁇_  


(315.)

322\. _𑀧𑀣𑀯𑀺𑀬𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀭𑀼𑀳𑀦𑁆𑀢𑀺 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀸;_  
_𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺,_  
_𑀨𑀮𑀜𑁆𑀘 𑀤𑀲𑁆𑀲𑀬𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(316.)

323\. _𑀧𑀣𑀯𑀻𑀲𑀤𑀺𑀲𑁄 𑀢𑁆𑀯𑀁𑀲𑀺,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢 𑀫𑀳𑀸𑀬𑀲;_  
_𑀲𑀸𑀲𑀦𑁂 𑀢𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀫𑀢𑀁 𑀨𑀮𑀁𑁇_  


(317.)

324\. _𑀲𑀺𑀦𑁆𑀥𑀼 𑀲𑀭𑀲𑁆𑀲𑀢𑀻 𑀘𑁂𑀯,_  
_𑀦𑀤𑀺𑀬𑁄 𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀺𑀓𑀸;_  
_𑀕𑀗𑁆𑀕𑀸 𑀘 𑀬𑀫𑀼𑀦𑀸 𑀘𑁂𑀯,_  
_𑀲𑀭𑀪𑀽 𑀘 𑀅𑀣𑁄 𑀫𑀳𑀻𑁇_  


(318.)

325\. _𑀏𑀢𑀸𑀲𑀁 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀲𑀸𑀕𑀭𑁄 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀢𑀺;_  
_𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀁 𑀦𑀸𑀫𑀁,_  
_𑀲𑀸𑀕𑀭𑁄𑀢𑁂𑀯 𑀜𑀸𑀬𑀢𑀺𑁇_  


(319.)

326\. _𑀢𑀣𑁂𑀯𑀺𑀫𑁂 𑀘𑀢𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀁 𑀦𑀸𑀫𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑀸𑀢𑀺 𑀜𑀸𑀬𑀭𑁂𑁇_  


(320.)

327\. _𑀬𑀣𑀸𑀧𑀺 𑀘𑀦𑁆𑀤𑁄 𑀯𑀺𑀫𑀮𑁄,_  
_𑀕𑀘𑁆𑀙𑀁 𑀆𑀓𑀸𑀲𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀢𑀸𑀭𑀕𑀡𑁂 𑀮𑁄𑀓𑁂,_  
_𑀆𑀪𑀸𑀬 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺𑁇_  


(321.)

328\. _𑀢𑀣𑁂𑀯 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑁂;_  
_𑀏𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀚𑀮𑀲𑀺 𑀲𑀩𑁆𑀩𑀤𑀸 𑀢𑀼𑀯𑀁𑁇_  


(322.)

329\. _𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀉𑀝𑁆𑀞𑀺𑀢𑀸 𑀊𑀫𑀻,_  
_𑀦 𑀯𑁂𑀮𑀫𑀢𑀺𑀯𑀢𑁆𑀢𑀭𑁂;_  
_𑀲𑀩𑁆𑀩𑀸 𑀯𑁂𑀮𑀁𑀯 𑀨𑀼𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑀜𑁆𑀘𑀼𑀡𑁆𑀡𑀸 𑀯𑀺𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀢𑀸𑁇_  


(323.)

330\. _𑀢𑀣𑁂𑀯 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀮𑁄𑀓𑁂,_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀩𑀳𑀼𑀚𑁆𑀚𑀦𑀸;_  
_𑀥𑀫𑁆𑀫𑀁 𑀯𑀸𑀤𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀢𑁂,_  
_𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀢𑀁 𑀫𑀼𑀦𑀺𑀁𑁇_  


(324.)

331\. _𑀲𑀘𑁂 𑀘 𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺,_  
_𑀧𑀝𑀺𑀯𑀸𑀤𑁂𑀳𑀺 𑀘𑀓𑁆𑀔𑀼𑀫;_  
_𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑀁 𑀉𑀧𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀲𑀜𑁆𑀘𑀼𑀡𑁆𑀡𑀸𑀯 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(325.)

332\. _𑀬𑀣𑀸𑀧𑀺 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀸,_  
_𑀓𑀼𑀫𑀼𑀤𑀸 𑀫𑀦𑁆𑀤𑀸𑀮𑀓𑀸 𑀩𑀳𑀽;_  
_𑀉𑀧𑀮𑀺𑀫𑁆𑀧𑀦𑁆𑀢𑀺 𑀢𑁄𑀬𑁂𑀦,_  
_𑀓𑀤𑁆𑀤𑀫𑀓𑀮𑀮𑁂𑀦 𑀘𑁇_  


(326.)

333\. _𑀢𑀣𑁂𑀯 𑀩𑀳𑀼𑀓𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀮𑁄𑀓𑁂 𑀚𑀸𑀢𑀸 𑀯𑀺𑀭𑀽𑀳𑀭𑁂;_  
_𑀅𑀝𑁆𑀝𑀺𑀢𑀸 𑀭𑀸𑀕𑀤𑁄𑀲𑁂𑀦,_  
_𑀓𑀤𑁆𑀤𑀫𑁂 𑀓𑀼𑀫𑀼𑀤𑀁 𑀬𑀣𑀸𑁇_  


(327.)

334\. _𑀬𑀣𑀸𑀧𑀺 𑀧𑀤𑀼𑀫𑀁 𑀚𑀮𑀚𑀁,_  
_𑀚𑀮𑀫𑀚𑁆𑀛𑁂 𑀯𑀺𑀭𑀽𑀳𑀢𑀺;_  
_𑀦 𑀲𑁄 𑀮𑀺𑀫𑁆𑀧𑀢𑀺 𑀢𑁄𑀬𑁂𑀦,_  
_𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑀺 𑀓𑁂𑀲𑀭𑀻𑁇_  


(328.)

335\. _𑀢𑀣𑁂𑀯 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀮𑁄𑀓𑁂 𑀚𑀸𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀦𑁄𑀧𑀮𑀺𑀫𑁆𑀧𑀲𑀺 𑀮𑁄𑀓𑁂𑀦,_  
_𑀢𑁄𑀬𑁂𑀦 𑀧𑀤𑀼𑀫𑀁 𑀬𑀣𑀸𑁇_  


(329.)

336\. _𑀬𑀣𑀸𑀧𑀺 𑀭𑀫𑁆𑀫𑀓𑁂 𑀫𑀸𑀲𑁂,_  
_𑀩𑀳𑀽 𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀺 𑀯𑀸𑀭𑀺𑀚𑀸;_  
_𑀦𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀢𑀁 𑀫𑀸𑀲𑀁,_  
_𑀲𑀫𑀬𑁄 𑀧𑀼𑀧𑁆𑀨𑀦𑀸𑀬 𑀲𑁄𑁇_  


(330.)

337\. _𑀢𑀣𑁂𑀯 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁄 𑀢𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀸𑀲𑀦𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀧𑀤𑀼𑀫𑀁 𑀯𑀸𑀭𑀺𑀚𑀁 𑀬𑀣𑀸𑁇_  


(331.)

338\. _𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁄 𑀲𑀸𑀮𑀭𑀸𑀚𑀸,_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀁 𑀧𑀯𑀸𑀬𑀢𑀺;_  
_𑀅𑀜𑁆𑀜𑀲𑀸𑀮𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯 𑀲𑁄𑀪𑀢𑀺𑁇_  


(332.)

339\. _𑀢𑀣𑁂𑀯 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀩𑀼𑀤𑁆𑀥𑀜𑀸𑀡𑁂𑀦 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯 𑀲𑁄𑀪𑀲𑀺𑁇_  


(333.)

340\. _𑀬𑀣𑀸𑀧𑀺 𑀲𑁂𑀮𑁄 𑀳𑀺𑀫𑀯𑀸,_  
_𑀑𑀲𑀥𑁄 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀦𑀸𑀕𑀸𑀦𑀁 𑀅𑀲𑀼𑀭𑀸𑀦𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀢𑀸𑀦𑀜𑁆𑀘 𑀆𑀮𑀬𑁄𑁇_  


(334.)

341\. _𑀢𑀣𑁂𑀯 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀑𑀲𑀥𑁄 𑀯𑀺𑀬 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀘,_  
_𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸𑁇_  


(335.)

342\. _𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀸 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀢𑀬𑀸 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁂𑀦 𑀢𑁂;_  
_𑀭𑀫𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀭𑀢𑀺𑀬𑀸,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑀯 𑀲𑀸𑀲𑀦𑁂𑁇_  


(336.)

343\. _𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀬𑀣𑀸 𑀲𑀻𑀳𑁄,_  
_𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫 𑀆𑀲𑀬𑀸;_  
_𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸𑀦𑀼𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀅𑀪𑀺𑀦𑀸𑀤𑀢𑀺𑁇_  


(337.)

344\. _𑀲𑀩𑁆𑀩𑁂 𑀫𑀺𑀕𑀸 𑀉𑀢𑁆𑀢𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀲𑁆𑀲 𑀕𑀚𑁆𑀚𑀢𑁄;_  
_𑀢𑀣𑀸 𑀳𑀺 𑀚𑀸𑀢𑀺𑀫𑀸 𑀏𑀲𑁄,_  
_𑀧𑀲𑀽 𑀢𑀸𑀲𑁂𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


(338.)

345\. _𑀕𑀚𑁆𑀚𑀢𑁄 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀯𑀲𑀼𑀥𑀸 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺;_  
_𑀩𑁄𑀥𑀦𑁂𑀬𑁆𑀬𑀸𑀯𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀢𑀲𑀦𑁆𑀢𑀺 𑀫𑀸𑀭𑀓𑀸𑀬𑀺𑀓𑀸𑁇_  


(339.)

346\. _𑀢𑀲𑀦𑁆𑀢𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀦𑀤𑀢𑁄 𑀢𑁂 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀓𑀸𑀓𑀸 𑀲𑁂𑀦𑀸𑀯 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀸,_  
_𑀫𑀺𑀕𑀭𑀜𑁆𑀜𑀸 𑀬𑀣𑀸 𑀫𑀺𑀕𑀸𑁇_  


(340.)

347\. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀕𑀡𑀺𑀦𑁄 𑀮𑁄𑀓𑁂,_  
_𑀲𑀢𑁆𑀣𑀸𑀭𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀭𑁂;_  
_𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀕𑀢𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀲𑀸𑀬 𑀢𑁂𑁇_  


(341.)

348\. _𑀦 𑀳𑁂𑀯𑀁 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀲𑀸𑀫𑀁 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀩𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑁂𑀯𑀮𑀁 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀁𑁇_  


(342.)

349\. _𑀆𑀲𑀬𑀸𑀦𑀼𑀲𑀬𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀁 𑀩𑀮𑀸𑀩𑀮𑀁;_  
_𑀪𑀩𑁆𑀩𑀸𑀪𑀩𑁆𑀩𑁂 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀳𑀸𑀫𑁂𑀖𑁄𑀯 𑀕𑀚𑁆𑀚𑀲𑀺𑁇_  


(343.)

350\. _𑀘𑀓𑁆𑀓𑀯𑀸𑀍𑀅𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀸,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀲𑀸 𑀪𑀯𑁂;_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀻 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀸,_  
_𑀯𑀺𑀫𑀢𑀺𑀘𑁆𑀙𑁂𑀤𑀦𑀸𑀬 𑀢𑀁𑁇_  


(344.)

351\. _𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀑𑀧𑀫𑁆𑀫𑀓𑀼𑀲𑀮𑁄 𑀫𑀼𑀦𑀺;_  
_𑀏𑀓𑀁 𑀧𑀜𑁆𑀳𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑁄𑀯,_  
_𑀯𑀺𑀫𑀢𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀲𑀺 𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(345.)

352\. _𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑀲𑀤𑀺𑀲𑁂𑀳𑁂𑀯,_  
_𑀯𑀲𑀼𑀥𑀸 𑀧𑀽𑀭𑀺𑀢𑀸 𑀪𑀯𑁂;_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑁂 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸,_  
_𑀓𑀺𑀢𑁆𑀢𑀬𑀼𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(346.)

353\. _𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑁂 𑀓𑀺𑀢𑁆𑀢𑀬𑀦𑁆𑀢𑀸,_  
_𑀦𑀸𑀦𑀸𑀯𑀡𑁆𑀡𑁂𑀳𑀺 𑀓𑀺𑀢𑁆𑀢𑀬𑀼𑀁;_  
_𑀧𑀭𑀺𑀫𑁂𑀢𑀼𑀁 𑀦 𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀢𑀣𑀸𑀕𑀢𑁄𑁇_  


(347.)

354\. _𑀬𑀣𑀸𑀲𑀓𑁂𑀦 𑀣𑀸𑀫𑁂𑀦,_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑁄 𑀳𑀺 𑀫𑀬𑀸 𑀚𑀺𑀦𑁄;_  
_𑀓𑀧𑁆𑀧𑀓𑁄𑀝𑀻𑀧𑀺 𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑀸,_  
_𑀏𑀯𑀫𑁂𑀯𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑀬𑀼𑀁𑁇_  


(348.)

355\. _𑀲𑀘𑁂 𑀳𑀺 𑀓𑁄𑀘𑀺 𑀤𑁂𑀯𑁄 𑀯𑀸,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀧𑀫𑁂𑀢𑀼𑀁 𑀧𑀭𑀺𑀓𑀧𑁆𑀧𑁂𑀬𑁆𑀬,_  
_𑀯𑀺𑀖𑀸𑀢𑀁𑀯 𑀮𑀪𑁂𑀬𑁆𑀬 𑀲𑁄𑁇_  


(349.)

356\. _𑀲𑀸𑀲𑀦𑁂 𑀢𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢 𑀫𑀳𑀸𑀬𑀲;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(350.)

357\. _𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀲𑀫𑁆𑀧𑀫𑀤𑁆𑀤𑀸𑀫𑀺,_  
_𑀯𑀢𑁆𑀢𑁂𑀫𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁;_  
_𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺 𑀅𑀚𑁆𑀚,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


(351.)

358\. _𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁂 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀨𑀮𑀁 𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀫𑁂 𑀇𑀥;_  
_𑀲𑀼𑀔𑀺𑀢𑁆𑀢𑁄 𑀲𑀭𑀯𑁂𑀕𑁄𑀯,_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀬𑀻 𑀫𑀫𑁇_  


(352.)

359\. _𑀬𑁄 𑀓𑁄𑀘𑀺 𑀫𑀦𑀼𑀚𑁄 𑀪𑀸𑀭𑀁,_  
_𑀥𑀸𑀭𑁂𑀬𑁆𑀬 𑀫𑀢𑁆𑀣𑀓𑁂 𑀲𑀤𑀸;_  
_𑀪𑀸𑀭𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀅𑀲𑁆𑀲,_  
_𑀪𑀸𑀭𑁂𑀳𑀺 𑀪𑀭𑀺𑀢𑁄 𑀢𑀣𑀸𑁇_  


(353.)

360\. _𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀢𑀻𑀳𑀕𑁆𑀕𑀻𑀳𑀺,_  
_𑀪𑀯𑁂𑀲𑀼 𑀲𑀁𑀲𑀭𑀺𑀁 𑀅𑀳𑀁;_  
_𑀪𑀭𑀺𑀢𑁄 𑀪𑀯𑀪𑀸𑀭𑁂𑀦,_  
_𑀕𑀺𑀭𑀺𑀁 𑀉𑀘𑁆𑀘𑀸𑀭𑀺𑀢𑁄 𑀬𑀣𑀸𑁇_  


(354.)

361\. _𑀑𑀭𑁄𑀧𑀺𑀢𑁄 𑀘 𑀫𑁂 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀸 𑀉𑀕𑁆𑀖𑀸𑀝𑀺𑀢𑀸 𑀫𑀬𑀸;_  
_𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


(355.)

362\. _𑀬𑀸𑀯𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁;_  
_𑀅𑀳𑀁 𑀅𑀕𑁆𑀕𑁄𑀫𑁆𑀳𑀺 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(356.)

363\. _𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀺 𑀲𑀼𑀓𑀼𑀲𑀮𑁄,_  
_𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑁄;_  
_𑀇𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀘𑀳𑀁 𑀅𑀚𑁆𑀚,_  
_𑀲𑀳𑀲𑁆𑀲𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑁂𑁇_  


(357.)

364\. _𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲,_  
_𑀯𑀲𑀻𑀪𑀽𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀓𑀣𑁂𑀲𑀺 𑀲𑀸𑀲𑀦𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀬𑀦𑀁 𑀫𑀫𑁇_  


(358.)

365\. _𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀫𑁂,_  
_𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑁄 𑀅𑀳𑀁;_  
_𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑁄𑁇_  


(359.)

366\. _𑀲𑀸𑀯𑀓𑁂𑀦 𑀳𑀺 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀲𑀩𑁆𑀩𑀫𑁂𑀯 𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(360.)

367\. _𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀼𑀲𑀮𑁄,_  
_𑀛𑀸𑀦𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦 𑀔𑀺𑀧𑁆𑀧𑀧𑀝𑀺𑀮𑀸𑀪𑀻;_  
_𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑁄,_  
_𑀲𑀸𑀯𑀓𑀕𑀼𑀡𑀧𑀸𑀭𑀫𑀺𑀕𑀢𑁄𑀲𑁆𑀫𑀺𑁇_  


(361.)

368\. _𑀲𑀸𑀯𑀓𑀕𑀼𑀡𑁂𑀦𑀧𑀺 𑀨𑀼𑀲𑁆𑀲𑁂𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑀪𑀸𑀭𑀯𑀸;_  
_𑀬𑀁 𑀲𑀤𑁆𑀥𑀸𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀲𑀤𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼𑁇_  


(362.)

369\. _𑀉𑀤𑁆𑀥𑀢𑀯𑀺𑀲𑁄𑀯 𑀲𑀧𑁆𑀧𑁄,_  
_𑀙𑀺𑀦𑁆𑀦𑀯𑀺𑀲𑀸𑀡𑁄𑀯 𑀉𑀲𑀪𑁄;_  
_𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀸𑀦𑀤𑀧𑁆𑀧𑁄𑀯,_  
_𑀉𑀧𑁂𑀫𑀺 𑀕𑀭𑀼𑀕𑀸𑀭𑀯𑁂𑀦 𑀕𑀡𑀁𑁇_  


(363.)

370\. _𑀬𑀤𑀺 𑀭𑀽𑀧𑀺𑀦𑀻 𑀪𑀯𑁂𑀬𑁆𑀬,_  
_𑀧𑀜𑁆𑀜𑀸 𑀫𑁂 𑀯𑀲𑀼𑀫𑀢𑀻𑀧𑀺 𑀦 𑀲𑀫𑁂𑀬𑁆𑀬;_  
_𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀨𑀮𑀫𑁂𑀢𑀁 𑀜𑀸𑀡𑀣𑀯𑀦𑀸𑀬𑁇_  


(364.)

371\. _𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀸;_  
_𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑁂𑀫𑀳𑀁 𑀲𑀫𑁆𑀫𑀸,_  
_𑀜𑀸𑀡𑀣𑀯𑀦𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(365.)

372\. _𑀫𑀸 𑀫𑁂 𑀓𑀤𑀸𑀘𑀺 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄,_  
_𑀓𑀼𑀲𑀻𑀢𑁄 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑁄;_  
_𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁄 𑀅𑀦𑀸𑀘𑀸𑀭𑁄,_  
_𑀲𑀫𑁂𑀢𑁄 𑀅𑀳𑀼 𑀓𑀢𑁆𑀣𑀘𑀺𑁇_  


(366.)

373\. _𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀘𑁂𑀢𑁄𑀲𑀫𑀣𑀸𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀅𑀧𑀺 𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀢𑀼𑁇_  


(367.)

374\. _𑀢𑀁 𑀯𑁄 𑀯𑀤𑀸𑀫𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀳𑁄𑀣 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸,_  
_𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑀸 𑀲𑀤𑀸𑁇_  


(368.)

375\. _𑀬𑀫𑀳𑀁 𑀧𑀞𑀫𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀺𑀭𑀚𑁄 𑀯𑀺𑀫𑀮𑁄 𑀅𑀳𑀼𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀥𑀻𑀭𑁄,_  
_𑀅𑀲𑁆𑀲𑀚𑀺 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄𑁇_  


(369.)

376\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀸𑀳𑀲𑀸 𑀅𑀚𑁆𑀚,_  
_𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀻 𑀅𑀳𑀼𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀸𑀭𑀫𑀺𑀁 𑀧𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(370.)

377\. _𑀬𑁄 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀆𑀲𑀺,_  
_𑀅𑀲𑁆𑀲𑀚𑀺 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄;_  
_𑀬𑀲𑁆𑀲𑀁 𑀤𑀺𑀲𑀸𑀬𑀁 𑀯𑀲𑀢𑀺,_  
_𑀉𑀲𑁆𑀲𑀻𑀲𑀫𑁆𑀳𑀺 𑀓𑀭𑁄𑀫𑀳𑀁𑁇_  


(371.)

378\. _𑀫𑀫 𑀓𑀫𑁆𑀫𑀁 𑀲𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀕𑁆𑀕𑀝𑁆𑀞𑀸𑀦𑁂 𑀞𑀧𑁂𑀲𑀺 𑀫𑀁𑁇_  


(372.)

379\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(373.)

380\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(374.)

381\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


382\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

383\. 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
