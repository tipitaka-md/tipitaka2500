# 𑁧𑁇𑁭 𑀧𑀼𑀡𑁆𑀡𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀡𑁆𑀡𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(434.)

449\. _“𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑀫𑁆𑀳𑀺 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(435.)

450\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀫𑀫 𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀲𑀺,_  
_𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(436.)

451\. _𑀢𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀓𑁆𑀓𑀫𑀺𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀫𑀼𑀔𑁄𑁇_  


(437.)

452\. _𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀪𑀸𑀲𑀬𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀺𑀲𑁆𑀲𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸,_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄;_  
_𑀲𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀼𑀁𑁇_  


(438.)

453\. _𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀤𑁂𑀲𑁂𑀫𑀺,_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀢𑀣𑁂𑀯𑀳𑀁;_  
_𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀦𑀬𑀜𑁆𑀜𑀽𑀳𑀁,_  
_𑀓𑀣𑀸𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(439.)

454\. _𑀇𑀢𑁄 𑀧𑀜𑁆𑀘𑀲𑀢𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀢𑀼𑀭𑁄 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀓𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀢𑀼𑀤𑀻𑀧𑀫𑁆𑀳𑀺 𑀇𑀲𑁆𑀲𑀭𑀸𑁇_  


(440.)

455\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


456\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

457\. 𑀧𑀼𑀡𑁆𑀡𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
