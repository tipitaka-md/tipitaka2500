# 𑁧𑁇𑁧𑁦 𑀧𑀺𑀡𑁆𑀟𑁄𑀮𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀡𑁆𑀟𑁄𑀮𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(613.)

635. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀧𑀼𑀭𑀢𑁄 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀘𑀺𑀢𑁆𑀢𑀓𑀽𑀝𑁂 𑀯𑀲𑀻 𑀢𑀤𑀸𑁇_  


(614.)

636. _𑀅𑀪𑀻𑀢𑀭𑀽𑀧𑁄 𑀢𑀢𑁆𑀣𑀸𑀲𑀺𑀁,_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑁄;_  
_𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀓𑁆𑀔𑀫𑁆𑀪𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀚𑁆𑀚𑀦𑀸𑁇_  


(615.)

637. _𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀁 𑀧𑀤𑀼𑀫𑀁 𑀕𑀬𑁆𑀳,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(616.)

638. _𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀁 𑀦𑀫𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀻𑀳𑀦𑀸𑀤𑀁 𑀦𑀤𑀺𑀁 𑀅𑀳𑀁𑁇_  


(617.)

639. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀲𑀓𑀸𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(618.)

640. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀕𑀺𑀭𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_‘𑀆𑀕𑀢𑁄 𑀯𑀤𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑁄𑀲𑁆𑀲𑀸𑀫 𑀢𑀁 𑀫𑀬𑀁’𑁇_  


(619.)

641. _𑀢𑁂𑀲𑀁 𑀳𑀸𑀲𑀧𑀭𑁂𑀢𑀸𑀦𑀁,_  
_𑀧𑀼𑀭𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀫𑀫 𑀲𑀤𑁆𑀤𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀲𑀺,_  
_𑀤𑀻𑀖𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(620.)

642. _𑀬𑁂𑀦𑀺𑀤𑀁 𑀧𑀤𑀼𑀫𑀁 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀲𑀻𑀳𑀦𑀸𑀤𑁄 𑀘 𑀦𑀸𑀤𑀺𑀢𑁄;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(621.)

643. _𑀇𑀢𑁄 𑀅𑀝𑁆𑀞𑀫𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀢𑀼𑀤𑀻𑀧𑀫𑁆𑀳𑀺 𑀇𑀲𑁆𑀲𑀭𑁄𑁇_  


(622.)

644. _𑀓𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁,_  
_𑀫𑀳𑀺𑀬𑀸 𑀘𑀢𑀼𑀲𑀝𑁆𑀞𑀺𑀬𑀸;_  
_𑀧𑀤𑀼𑀫𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(623.)

645. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(624.)

646. _𑀧𑀓𑀸𑀲𑀺𑀢𑁂 𑀧𑀸𑀯𑀘𑀦𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(625.)

647. _𑀧𑀥𑀸𑀦𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀲𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄𑁇_  


(626.)

648. _𑀯𑀺𑀚𑀦𑁂 𑀧𑀦𑁆𑀢𑀲𑁂𑀬𑁆𑀬𑀫𑁆𑀳𑀺,_  
_𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀲𑀫𑀸𑀓𑀼𑀮𑁂;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄𑁇_  


(627.)

649. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


650. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀡𑁆𑀟𑁄𑀮𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

651. 𑀧𑀺𑀡𑁆𑀟𑁄𑀮𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
