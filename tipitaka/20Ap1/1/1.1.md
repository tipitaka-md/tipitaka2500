# 𑁧𑁇𑁧 𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑀤𑀸𑀦

(1.)

2. _𑀢𑀣𑀸𑀕𑀢𑀁 𑀚𑁂𑀢𑀯𑀦𑁂 𑀯𑀲𑀦𑁆𑀢𑀁,_  
_𑀅𑀧𑀼𑀘𑁆𑀙𑀺 𑀯𑁂𑀤𑁂𑀳𑀫𑀼𑀦𑀻 𑀦𑀢𑀗𑁆𑀕𑁄;_  
_“𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀩𑀼𑀤𑁆𑀥𑀸 𑀓𑀺𑀭 𑀦𑀸𑀫 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑁂 𑀳𑁂𑀢𑀼𑀪𑀺 𑀓𑁂𑀳𑀺 𑀯𑀻𑀭”𑁇_  


(2.)

3. _𑀢𑀤𑀸𑀳 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀯𑀭𑁄 𑀫𑀳𑁂𑀲𑀻,_  
_𑀆𑀦𑀦𑁆𑀤𑀪𑀤𑁆𑀤𑀁 𑀫𑀥𑀼𑀭𑀲𑁆𑀲𑀭𑁂𑀦;_  
_“𑀬𑁂 𑀧𑀼𑀩𑁆𑀩𑀩𑀼𑀤𑁆𑀥𑁂𑀲𑀼 𑀓𑀢𑀸𑀥𑀺𑀓𑀸𑀭𑀸,_  
_𑀅𑀮𑀤𑁆𑀥𑀫𑁄𑀓𑁆𑀔𑀸 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂𑀲𑀼𑁇_  


(3.)

4. _𑀢𑁂𑀦𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀔𑁂𑀦 𑀥𑀻𑀭𑀸,_  
_𑀅𑀚𑁆𑀛𑀸𑀲𑀬𑁂𑀦𑀸𑀧𑀺 𑀫𑀳𑀸𑀩𑀮𑁂𑀦;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀢𑁂𑀚𑁂𑀦 𑀲𑀼𑀢𑀺𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑀸,_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀪𑀸𑀯𑀁 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺𑁇_  


(4.)

5. _𑀅𑀳𑀫𑁆𑀧𑀺 𑀧𑀼𑀩𑁆𑀩𑀩𑀼𑀤𑁆𑀥𑁂𑀲𑀼,_  
_𑀩𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀫𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀺𑀁;_  
_𑀫𑀦𑀲𑀸𑀬𑁂𑀯 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀸𑁇_  


(5.)

6. _𑀅𑀣 𑀩𑀼𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀸𑀦𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀲𑀼𑀤𑁆𑀥𑀫𑀸𑀦𑀲𑀸;_  
_𑀢𑀺𑀁𑀲𑀧𑀸𑀭𑀫𑀺𑀲𑀫𑁆𑀧𑀼𑀡𑁆𑀡𑀸,_  
_𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀸𑁇_  


(6.)

7. _𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀲𑀲𑀁𑀖𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀤𑀲𑀗𑁆𑀕𑀼𑀮𑀻 𑀦𑀫𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀭𑀲𑀸 𑀅𑀪𑀺𑀯𑀸𑀤𑀬𑀺𑀁𑁇_  


(7.)

8. _𑀬𑀸𑀯𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀔𑁂𑀢𑁆𑀢𑁂𑀲𑀼,_  
_𑀭𑀢𑀦𑀸 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀲𑀗𑁆𑀔𑀺𑀬𑀸;_  
_𑀆𑀓𑀸𑀲𑀝𑁆𑀞𑀸 𑀘 𑀪𑀽𑀫𑀝𑁆𑀞𑀸,_  
_𑀫𑀦𑀲𑀸 𑀲𑀩𑁆𑀩𑀫𑀸𑀳𑀭𑀺𑀁𑁇_  


(8.)

9. _𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀬𑀁,_  
_𑀧𑀸𑀲𑀸𑀤𑀁 𑀫𑀸𑀧𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀦𑁂𑀓𑀪𑀼𑀫𑁆𑀫𑀁 𑀭𑀢𑀦𑀸𑀫𑀬𑀁,_  
_𑀉𑀩𑁆𑀩𑀺𑀤𑁆𑀥𑀁 𑀦𑀪𑀫𑀼𑀕𑁆𑀕𑀢𑀁𑁇_  


(9.)

10. _𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀣𑀫𑁆𑀪𑀁 𑀲𑀼𑀓𑀢𑀁,_  
_𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀫𑀳𑀸𑀭𑀳𑀁;_  
_𑀓𑀦𑀓𑀫𑀬𑀲𑀗𑁆𑀖𑀸𑀝𑀁,_  
_𑀓𑁄𑀦𑁆𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑁂𑀳𑀺 𑀫𑀡𑁆𑀟𑀺𑀢𑀁𑁇_  


(10.)

11. _𑀧𑀞𑀫𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀪𑀽𑀫𑀺,_  
_𑀯𑀺𑀫𑀮𑀩𑁆𑀪𑀲𑀫𑀸 𑀲𑀼𑀪𑀸;_  
_𑀦𑀍𑀇𑀦𑀚𑀮𑀚𑀸𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀯𑀭𑀓𑀜𑁆𑀘𑀦𑀪𑀽𑀫𑀺𑀬𑀸𑁇_  


(11.)

12. _𑀧𑀯𑀸𑀍𑀅𑀁𑀲𑀸 𑀧𑀯𑀸𑀍𑀅𑀯𑀡𑁆𑀡𑀸,_  
_𑀓𑀸𑀘𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀸 𑀲𑀼𑀪𑀸;_  
_𑀇𑀦𑁆𑀤𑀕𑁄𑀧𑀓𑀯𑀡𑁆𑀡𑀸𑀪𑀸,_  
_𑀪𑀽𑀫𑀺 𑀑𑀪𑀸𑀲𑀢𑀻 𑀤𑀺𑀲𑀸𑁇_  


(12.)

13. _𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀸 𑀖𑀭𑀫𑀼𑀔𑀸,_  
_𑀦𑀺𑀬𑁆𑀬𑀽𑀳𑀸 𑀲𑀻𑀳𑀧𑀜𑁆𑀚𑀭𑀸;_  
_𑀘𑀢𑀼𑀭𑁄 𑀯𑁂𑀤𑀺𑀓𑀸 𑀚𑀸𑀮𑀸,_  
_𑀕𑀦𑁆𑀥𑀸𑀯𑁂𑀍𑀆 𑀫𑀦𑁄𑀭𑀫𑀸𑁇_  


(13.)

14. _𑀦𑀻𑀮𑀸 𑀧𑀻𑀢𑀸 𑀮𑁄𑀳𑀺𑀢𑀓𑀸,_  
_𑀑𑀤𑀸𑀢𑀸 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸;_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀯𑀭𑀽𑀧𑁂𑀢𑀸,_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀪𑀽𑀲𑀺𑀢𑀸𑁇_  


(14.)

15. _𑀑𑀮𑁄𑀓𑀫𑀬𑀸 𑀧𑀤𑀼𑀫𑀸,_  
_𑀯𑀸𑀍𑀅𑀯𑀺𑀳𑀗𑁆𑀕𑀲𑁄𑀪𑀺𑀢𑀸;_  
_𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀢𑀸𑀭𑀓𑀸𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀘𑀦𑁆𑀤𑀲𑀽𑀭𑁂𑀳𑀺 𑀫𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


(15.)

16. _𑀳𑁂𑀫𑀚𑀸𑀮𑁂𑀦 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀲𑁄𑀡𑁆𑀡𑀓𑀺𑀗𑁆𑀓𑀺𑀡𑀺𑀓𑀸𑀬𑀼𑀢𑀸;_  
_𑀯𑀸𑀢𑀯𑁂𑀕𑁂𑀦 𑀓𑀽𑀚𑀦𑁆𑀢𑀺,_  
_𑀲𑁄𑀡𑁆𑀡𑀫𑀸𑀮𑀸 𑀫𑀦𑁄𑀭𑀫𑀸𑁇_  


(16.)

17. _𑀫𑀜𑁆𑀚𑁂𑀝𑁆𑀞𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀁,_  
_𑀧𑀻𑀢𑀓𑀁 𑀳𑀭𑀺𑀧𑀺𑀜𑁆𑀚𑀭𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀻𑀢𑀁,_  
_𑀉𑀲𑁆𑀲𑀺𑀢𑀤𑁆𑀥𑀚𑀫𑀸𑀮𑀺𑀦𑀻𑁇_  


(17.)

18. _𑀦 𑀦𑀁 𑀩𑀳𑀽𑀦𑁂𑀓𑀲𑀢𑀸,_  
_𑀨𑀮𑀺𑀓𑀸 𑀭𑀚𑀢𑀸𑀫𑀬𑀸;_  
_𑀫𑀡𑀺𑀫𑀬𑀸 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀕𑀸,_  
_𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀫𑀬𑀸 𑀢𑀣𑀸;_  
_𑀦𑀸𑀦𑀸𑀲𑀬𑀦𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀲𑀡𑁆𑀳𑀓𑀸𑀲𑀺𑀓𑀲𑀦𑁆𑀣𑀢𑀸𑁇_  


(18.)

19. _𑀓𑀫𑁆𑀧𑀮𑀸 𑀤𑀼𑀓𑀽𑀮𑀸 𑀘𑀻𑀦𑀸,_  
_𑀧𑀢𑁆𑀢𑀼𑀡𑁆𑀡𑀸 𑀧𑀡𑁆𑀟𑀼𑀧𑀸𑀯𑀼𑀭𑀸;_  
_𑀯𑀺𑀯𑀺𑀥𑀢𑁆𑀣𑀭𑀡𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀫𑀦𑀲𑀸 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀳𑀁𑁇_  


(19.)

20. _𑀢𑀸𑀲𑀼 𑀢𑀸𑀲𑁆𑀯𑁂𑀯 𑀪𑀽𑀫𑀻𑀲𑀼,_  
_𑀭𑀢𑀦𑀓𑀽𑀝𑀮𑀗𑁆𑀓𑀢𑀁;_  
_𑀫𑀡𑀺𑀯𑁂𑀭𑁄𑀘𑀦𑀸 𑀉𑀓𑁆𑀓𑀸,_  
_𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑀸 𑀲𑀼𑀢𑀺𑀝𑁆𑀞𑀭𑁂𑁇_  


(20.)

21. _𑀲𑁄𑀪𑀦𑁆𑀢𑀺 𑀏𑀲𑀺𑀓𑀸 𑀣𑀫𑁆𑀪𑀸,_  
_𑀲𑀼𑀪𑀸 𑀓𑀜𑁆𑀘𑀦𑀢𑁄𑀭𑀡𑀸;_  
_𑀚𑀫𑁆𑀩𑁄𑀦𑀤𑀸 𑀲𑀸𑀭𑀫𑀬𑀸,_  
_𑀅𑀣𑁄 𑀭𑀚𑀢𑀫𑀬𑀸𑀧𑀺 𑀘𑁇_  


(21.)

22. _𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀥𑀻 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀸,_  
_𑀓𑀯𑀸𑀝𑀕𑁆𑀕𑀍𑀅𑀘𑀺𑀢𑁆𑀢𑀺𑀢𑀸;_  
_𑀉𑀪𑀢𑁄 𑀧𑀼𑀡𑁆𑀡𑀖𑀝𑀸𑀦𑁂𑀓𑀸,_  
_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀲𑀁𑀬𑀼𑀢𑀸𑁇_  


(22.)

23. _𑀅𑀢𑀻𑀢𑁂 𑀲𑀩𑁆𑀩𑀩𑀼𑀤𑁆𑀥𑁂 𑀘,_  
_𑀲𑀲𑀁𑀖𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀧𑀓𑀢𑀺𑀯𑀡𑁆𑀡𑀭𑀽𑀧𑁂𑀦,_  
_𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀲𑀲𑀸𑀯𑀓𑁂𑁇_  


(23.)

24. _𑀢𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀩𑀼𑀤𑁆𑀥𑀸 𑀲𑀲𑀸𑀯𑀓𑀸;_  
_𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑁂 𑀧𑀻𑀞𑁂,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀅𑀭𑀺𑀬𑀫𑀡𑁆𑀟𑀮𑀸𑁇_  


(24.)

25. _𑀬𑁂 𑀘 𑀏𑀢𑀭𑀳𑀺 𑀅𑀢𑁆𑀣𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀸 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸;_  
_𑀅𑀢𑀻𑀢𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀸 𑀘,_  
_𑀪𑀯𑀦𑀁 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀸𑀭𑀼𑀳𑀼𑀁𑁇_  


(25.)

26. _𑀧𑀘𑁆𑀘𑁂𑀓𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑁂𑀓𑀲𑀢𑁂,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁂;_  
_𑀅𑀢𑀻𑀢𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂 𑀘,_  
_𑀪𑀯𑀦𑀁 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀸𑀭𑀼𑀳𑀼𑀁𑁇_  


(26.)

27. _𑀓𑀧𑁆𑀧𑀭𑀼𑀓𑁆𑀔𑀸 𑀩𑀳𑀽 𑀅𑀢𑁆𑀣𑀺,_  
_𑀬𑁂 𑀤𑀺𑀩𑁆𑀩𑀸 𑀬𑁂 𑀘 𑀫𑀸𑀦𑀼𑀲𑀸;_  
_𑀲𑀩𑁆𑀩𑀁 𑀤𑀼𑀲𑁆𑀲𑀁 𑀲𑀫𑀸𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀫𑀺 𑀢𑀺𑀘𑀻𑀯𑀭𑀁𑁇_  


(27.)

28. _𑀔𑀚𑁆𑀚𑀁 𑀪𑁄𑀚𑁆𑀚𑀁 𑀲𑀸𑀬𑀦𑀻𑀬𑀁,_  
_𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀪𑁄𑀚𑀦𑀁;_  
_𑀫𑀡𑀺𑀫𑀬𑁂 𑀲𑀼𑀪𑁂 𑀧𑀢𑁆𑀢𑁂,_  
_𑀲𑀫𑁆𑀧𑀽𑀭𑁂𑀢𑁆𑀯𑀸 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(28.)

29. _𑀤𑀺𑀩𑁆𑀩𑀯𑀢𑁆𑀣𑀲𑀫𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀫𑀝𑁆𑀞𑀸 𑀘𑀻𑀯𑀭𑀲𑀁𑀬𑀼𑀢𑀸;_  
_𑀫𑀥𑀼𑀭𑀸 𑀲𑀓𑁆𑀔𑀭𑀸 𑀘𑁂𑀯,_  
_𑀢𑁂𑀮𑀸 𑀘 𑀫𑀥𑀼𑀨𑀸𑀡𑀺𑀢𑀸𑁇_  


(29.)

30. _𑀢𑀧𑁆𑀧𑀺𑀢𑀸 𑀧𑀭𑀫𑀦𑁆𑀦𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀅𑀭𑀺𑀬𑀫𑀡𑁆𑀟𑀮𑀸;_  
_𑀭𑀢𑀦𑀕𑀩𑁆𑀪𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑁂𑀲𑀭𑀻𑀯 𑀕𑀼𑀳𑀸𑀲𑀬𑀸𑁇_  


(30.)

31. _𑀫𑀳𑀸𑀭𑀳𑀫𑁆𑀳𑀺 𑀲𑀬𑀦𑁂,_  
_𑀲𑀻𑀳𑀲𑁂𑀬𑁆𑀬𑀫𑀓𑀧𑁆𑀧𑀬𑀼𑀁;_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀬,_  
_𑀲𑀬𑀦𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑀸𑀪𑀼𑀚𑀼𑀁𑁇_  


(31.)

32. _𑀕𑁄𑀘𑀭𑀁 𑀲𑀩𑁆𑀩𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀛𑀸𑀦𑀭𑀢𑀺𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸;_  
_𑀅𑀜𑁆𑀜𑁂 𑀥𑀫𑁆𑀫𑀸𑀦𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺,_  
_𑀅𑀜𑁆𑀜𑁂 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺 𑀇𑀤𑁆𑀥𑀺𑀬𑀸𑁇_  


(32.)

33. _𑀅𑀜𑁆𑀜𑁂 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀅𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺,_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀯𑀲𑀺𑀪𑀸𑀯𑀺𑀢𑀸;_  
_𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀸 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺,_  
_𑀅𑀜𑁆𑀜𑁂𑀦𑁂𑀓𑀲𑀳𑀲𑁆𑀲𑀺𑀬𑁄𑁇_  


(33.)

34. _𑀩𑀼𑀤𑁆𑀥𑀸𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀲𑀬𑀁 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀫𑀸𑀮𑀬𑀁;_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀦𑀺𑀧𑀼𑀡𑀁 𑀞𑀸𑀦𑀁,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀦𑀺𑀩𑀼𑀚𑁆𑀛𑀭𑁂𑁇_  


(34.)

35. _𑀲𑀸𑀯𑀓𑀸 𑀩𑀼𑀤𑁆𑀥𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀸𑀯𑀓𑁂;_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀜𑁆𑀘 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀜𑁆𑀜𑁄𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(35.)

36. _𑀩𑀼𑀤𑁆𑀥𑀸 𑀧𑀘𑁆𑀘𑁂𑀓𑀩𑀼𑀤𑁆𑀥𑀸 𑀘,_  
_𑀲𑀸𑀯𑀓𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸;_  
_𑀏𑀯𑀁 𑀲𑀓𑀸𑀬 𑀭𑀢𑀺𑀬𑀸,_  
_𑀧𑀸𑀲𑀸𑀤𑁂𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(36.)

37. _𑀙𑀢𑁆𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀼 𑀭𑀢𑀦𑀸,_  
_𑀓𑀜𑁆𑀘𑀦𑀸𑀯𑁂𑀍𑀅𑀧𑀦𑁆𑀢𑀺𑀓𑀸;_  
_𑀫𑀼𑀢𑁆𑀢𑀸𑀚𑀸𑀮𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀼 𑀫𑀢𑁆𑀣𑀓𑁂𑁇_  


(37.)

38. _𑀪𑀯𑀦𑁆𑀢𑀼 𑀘𑁂𑀍𑀅𑀯𑀺𑀢𑀸𑀦𑀸,_  
_𑀲𑁄𑀡𑁆𑀡𑀢𑀸𑀭𑀓𑀘𑀺𑀢𑁆𑀢𑀺𑀢𑀸;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀫𑀮𑁆𑀬𑀯𑀺𑀢𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀼 𑀫𑀢𑁆𑀣𑀓𑁂𑁇_  


(38.)

39. _𑀯𑀺𑀢𑀢𑀸 𑀫𑀮𑁆𑀬𑀤𑀸𑀫𑁂𑀳𑀺,_  
_𑀕𑀦𑁆𑀥𑀤𑀸𑀫𑁂𑀳𑀺 𑀲𑁄𑀪𑀺𑀢𑀸;_  
_𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀫𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀭𑀢𑀦𑀤𑀸𑀫𑀪𑀽𑀲𑀺𑀢𑀸𑁇_  


(39.)

40. _𑀧𑀼𑀧𑁆𑀨𑀸𑀪𑀺𑀓𑀺𑀡𑁆𑀡𑀸 𑀲𑀼𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀲𑀼𑀭𑀪𑀺𑀕𑀦𑁆𑀥𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀕𑀦𑁆𑀥𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀼𑀮𑀓𑀢𑀸,_  
_𑀳𑁂𑀫𑀘𑁆𑀙𑀤𑀦𑀙𑀸𑀤𑀺𑀢𑀸𑁇_  


(40.)

41. _𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀜𑁆𑀜𑁄,_  
_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀲𑀦𑁆𑀣𑀢𑀸;_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑀭𑀽𑀧𑀸 𑀔𑀸𑀬𑀦𑁆𑀢𑀼,_  
_𑀧𑀤𑁆𑀫𑀭𑁂𑀡𑀼𑀭𑀚𑀼𑀕𑁆𑀕𑀢𑀸𑁇_  


(41.)

42. _𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀼 𑀧𑀸𑀤𑀧𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀸𑀲𑀸𑀤𑀲𑁆𑀲 𑀲𑀫𑀦𑁆𑀢𑀢𑁄;_  
_𑀲𑀬𑀜𑁆𑀘 𑀧𑀼𑀧𑁆𑀨𑀸 𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸,_  
_𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀯𑀦𑀫𑁄𑀓𑀺𑀭𑀼𑀁𑁇_  


(42.)

43. _𑀲𑀺𑀔𑀺𑀦𑁄 𑀢𑀢𑁆𑀣 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀼,_  
_𑀤𑀺𑀩𑁆𑀩𑀳𑀁𑀲𑀸 𑀧𑀓𑀽𑀚𑀭𑁂;_  
_𑀓𑀭𑀯𑀻𑀓𑀸 𑀘 𑀕𑀸𑀬𑀦𑁆𑀢𑀼,_  
_𑀤𑀺𑀚𑀲𑀗𑁆𑀖𑀸 𑀲𑀫𑀦𑁆𑀢𑀢𑁄𑁇_  


(43.)

44. _𑀪𑁂𑀭𑀺𑀬𑁄 𑀲𑀩𑁆𑀩𑀸 𑀯𑀚𑁆𑀚𑀦𑁆𑀢𑀼,_  
_𑀯𑀻𑀡𑀸 𑀲𑀩𑁆𑀩𑀸 𑀭𑀲𑀦𑁆𑀢𑀼 𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑀸 𑀲𑀗𑁆𑀕𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀼,_  
_𑀧𑀸𑀲𑀸𑀤𑀲𑁆𑀲 𑀲𑀫𑀦𑁆𑀢𑀢𑁄𑁇_  


(44.)

45. _𑀬𑀸𑀯𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀍𑀏 𑀢𑀢𑁄 𑀧𑀭𑁂;_  
_𑀫𑀳𑀦𑁆𑀢𑀸 𑀚𑁄𑀢𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀭𑀢𑀦𑀸𑀫𑀬𑀸𑁇_  


(45.)

46. _𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀼 𑀲𑁄𑀡𑁆𑀡𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀸,_  
_𑀤𑀻𑀧𑀭𑀼𑀓𑁆𑀔𑀸 𑀚𑀮𑀦𑁆𑀢𑀼 𑀢𑁂;_  
_𑀪𑀯𑀦𑁆𑀢𑀼 𑀏𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑀸,_  
_𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀺𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑁇_  


(46.)

47. _𑀕𑀡𑀺𑀓𑀸 𑀮𑀸𑀲𑀺𑀓𑀸 𑀘𑁂𑀯,_  
_𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀼 𑀅𑀘𑁆𑀙𑀭𑀸𑀕𑀡𑀸;_  
_𑀦𑀸𑀦𑀸𑀭𑀗𑁆𑀕𑀸 𑀧𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀼,_  
_𑀧𑀸𑀲𑀸𑀤𑀲𑁆𑀲 𑀲𑀫𑀦𑁆𑀢𑀢𑁄𑁇_  


(47.)

48. _𑀤𑀼𑀫𑀕𑁆𑀕𑁂 𑀧𑀩𑁆𑀩𑀢𑀕𑁆𑀕𑁂 𑀯𑀸,_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀕𑀺𑀭𑀺𑀫𑀼𑀤𑁆𑀥𑀦𑀺;_  
_𑀉𑀲𑁆𑀲𑀸𑀧𑁂𑀫𑀺 𑀥𑀚𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀜𑁆𑀘𑀯𑀡𑁆𑀡𑀺𑀓𑀁𑁇_  


(48.)

49. _𑀦𑀭𑀸 𑀦𑀸𑀕𑀸 𑀘 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸 𑀉𑀧𑁂𑀦𑁆𑀢𑀼 𑀢𑁂;_  
_𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸,_  
_𑀧𑀸𑀲𑀸𑀤𑀁 𑀧𑀭𑀺𑀯𑀸𑀭𑀬𑀼𑀁𑁇_  


(49.)

50. _𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀺𑀭𑀺𑀬𑀁 𑀫𑀫;_  
_𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸 𑀫𑀦𑀲𑀸,_  
_𑀢𑀺𑀤𑀲𑁂 𑀲𑀼𑀓𑀢𑀁 𑀓𑀢𑀁𑁇_  


(50.)

51. _𑀬𑁂 𑀲𑀢𑁆𑀢𑀸 𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀬𑁂 𑀘 𑀲𑀢𑁆𑀢𑀸 𑀅𑀲𑀜𑁆𑀜𑀺𑀦𑁄;_  
_𑀓𑀢𑀁 𑀧𑀼𑀜𑁆𑀜𑀨𑀮𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀲𑀩𑁆𑀩𑁂 𑀪𑀸𑀕𑀻 𑀪𑀯𑀦𑁆𑀢𑀼 𑀢𑁂𑁇_  


(51.)

52. _𑀬𑁂𑀲𑀁 𑀓𑀢𑀁 𑀲𑀼𑀯𑀺𑀤𑀺𑀢𑀁,_  
_𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀜𑁆𑀜𑀨𑀮𑀁 𑀫𑀬𑀸;_  
_𑀬𑁂 𑀘 𑀢𑀢𑁆𑀣 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀯𑀸 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑁂𑀤𑀬𑀼𑀁𑁇_  


(52.)

53. _𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀫𑁆𑀳𑀺 𑀬𑁂 𑀲𑀢𑁆𑀢𑀸,_  
_𑀚𑀻𑀯𑀦𑁆𑀢𑀸𑀳𑀸𑀭𑀳𑁂𑀢𑀼𑀓𑀸;_  
_𑀫𑀦𑀼𑀜𑁆𑀜𑀁 𑀪𑁄𑀚𑀦𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀮𑀪𑀦𑁆𑀢𑀼 𑀫𑀫 𑀘𑁂𑀢𑀲𑀸𑁇_  


(53.)

54. _𑀫𑀦𑀲𑀸 𑀤𑀸𑀦𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀫𑀦𑀲𑀸 𑀧𑀲𑀸𑀤𑀫𑀸𑀯𑀳𑀺𑀁;_

_𑀭𑀢𑀦𑀜𑁆𑀘 𑀅𑀦𑀧𑁆𑀧𑀓𑀁;_  
_𑀯𑀺𑀯𑀺𑀥𑀸𑀦𑀺 𑀘 𑀯𑀢𑁆𑀣𑀸𑀦𑀺,_

(59.)

_𑀆𑀓𑀸𑀲𑁂 𑀉𑀤𑀓𑁂 𑀯𑀦𑁂;_

(61.)

62. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀧𑀩𑁆𑀩𑀢𑁂 𑀘𑁂𑀯,_

_𑀲𑀩𑁆𑀩𑁂 𑀬𑀸𑀦𑀸 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(62.)

63. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀧𑀩𑁆𑀩𑀢𑁂 𑀘𑁂𑀯,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀤𑀓𑁂 𑀯𑀦𑁂;_  
_𑀬𑀁 𑀬𑀁 𑀳𑀢𑁆𑀣𑀁 𑀧𑀲𑀸𑀭𑁂𑀫𑀺,_

(63.)

64. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀧𑀩𑁆𑀩𑀢𑁂 𑀘𑁂𑀯,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀤𑀓𑁂 𑀯𑀦𑁂;_  
_𑀬𑀁 𑀬𑀁 𑀳𑀢𑁆𑀣𑀁 𑀧𑀲𑀸𑀭𑁂𑀫𑀺,_  
_𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑀸 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(64.)

65. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀧𑀩𑁆𑀩𑀢𑁂 𑀘𑁂𑀯,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀤𑀓𑁂 𑀯𑀦𑁂;_  
_𑀬𑀁 𑀬𑀁 𑀳𑀢𑁆𑀣𑀁 𑀧𑀲𑀸𑀭𑁂𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑀸 𑀓𑀜𑁆𑀜𑀸 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(65.)

66. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀧𑀩𑁆𑀩𑀢𑁂 𑀘𑁂𑀯,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀤𑀓𑁂 𑀯𑀦𑁂;_  
_𑀬𑀁 𑀬𑀁 𑀳𑀢𑁆𑀣𑀁 𑀧𑀲𑀸𑀭𑁂𑀫𑀺,_  
_𑀫𑀥𑀼𑀲𑀓𑁆𑀔𑀭𑀸 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(66.)

67. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀧𑀩𑁆𑀩𑀢𑁂 𑀘𑁂𑀯,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀤𑀓𑁂 𑀯𑀦𑁂;_  
_𑀬𑀁 𑀬𑀁 𑀳𑀢𑁆𑀣𑀁 𑀧𑀲𑀸𑀭𑁂𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑁂 𑀔𑀚𑁆𑀚𑀸 𑀉𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀁𑁇_  


(67.)

68. _𑀅𑀥𑀦𑁂 𑀅𑀤𑁆𑀥𑀺𑀓𑀚𑀦𑁂,_  
_𑀬𑀸𑀘𑀓𑁂 𑀘 𑀧𑀣𑀸𑀯𑀺𑀦𑁄;_  
_𑀤𑀤𑀸𑀫𑀺𑀳𑀁 𑀤𑀸𑀦𑀯𑀭𑀁,_  
_𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀯𑀭𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇_

_𑀕𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑁄 𑀩𑀳𑀮𑀁 𑀕𑀺𑀭𑀺𑀁;_  
_𑀲𑀤𑁂𑀯𑀓𑀁 𑀳𑀸𑀲𑀬𑀦𑁆𑀢𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀪𑀯𑀸𑀫𑀳𑀁𑁇_  


(69.)

_𑀬𑀸𑀬𑀢𑁄 𑀦𑀢𑁆𑀣𑀺 𑀅𑀦𑁆𑀢𑀓𑀁;_  
_𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀤𑀺𑀲𑀸𑀪𑀸𑀕𑀫𑁆𑀳𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀔𑁂𑀢𑁆𑀢𑀸 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀸𑁇_  


(70.)

71. _𑀧𑀪𑀸 𑀧𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀬𑀫𑀓𑀸 𑀭𑀁𑀲𑀺𑀯𑀸𑀳𑀦𑀸;_  
_𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀭𑀁𑀲𑀺𑀚𑀸𑀮𑀁,_  
_𑀆𑀮𑁄𑀓𑁄 𑀯𑀺𑀧𑀼𑀮𑁄 𑀪𑀯𑁂𑁇_  


(71.)

_𑀲𑀩𑁆𑀩𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀼 𑀫𑀁 𑀚𑀦𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀫𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀼,_  
_𑀬𑀸𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑀦𑀺𑀯𑁂𑀲𑀦𑀁𑁇_  


(72.)

73. _𑀯𑀺𑀲𑀺𑀝𑁆𑀞𑀫𑀥𑀼𑀦𑀸𑀤𑁂𑀦,_  
_𑀅𑀫𑀢𑀪𑁂𑀭𑀺𑀫𑀸𑀳𑀦𑀺𑀁;_  
_𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀚𑀦𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑀥𑀼𑀭𑀁 𑀕𑀺𑀭𑀁𑁇_  


(73.)

_𑀬𑁂𑀢𑁆𑀣 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀪𑀯𑀦𑁆𑀢𑀼 𑀢𑁂𑁇_

75. _𑀤𑀢𑁆𑀯𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀓𑀁 𑀤𑀸𑀦𑀁,_

_𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_

_𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(76.)

77. _𑀓𑀢𑁆𑀯𑀸 𑀤𑀍𑀳𑀫𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀲𑀘𑁆𑀘𑀧𑀸𑀭𑀫𑀺 𑀧𑀽𑀭𑀺𑀬;_  
_𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(77.)

78. _𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑁂 𑀲𑀼𑀔𑁂 𑀤𑀼𑀓𑁆𑀔𑁂,_  
_𑀲𑀫𑁆𑀫𑀸𑀦𑁂 𑀘𑀸𑀯𑀫𑀸𑀦𑀦𑁂;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀫𑀓𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(78.)

79. _𑀓𑁄𑀲𑀚𑁆𑀚𑀁 𑀪𑀬𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀻𑀭𑀺𑀬𑀜𑁆𑀘𑀸𑀧𑀺 𑀔𑁂𑀫𑀢𑁄;_  
_𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸 𑀳𑁄𑀣,_  
_𑀏𑀲𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑁇_  


(79.)

_𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀔𑀺𑀮𑀸 𑀳𑁄𑀣,_  
_𑀏𑀲𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑁇_  


(80.)

81. _𑀧𑀫𑀸𑀤𑀁 𑀪𑀬𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀜𑁆𑀘 𑀔𑁂𑀫𑀢𑁄;_  
_𑀪𑀸𑀯𑁂𑀣𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀏𑀲𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑁇_  


(81.)

82. _𑀲𑀫𑀸𑀕𑀢𑀸 𑀩𑀳𑀽 𑀩𑀼𑀤𑁆𑀥𑀸,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑀸 𑀘 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁂 𑀘,_  
_𑀯𑀦𑁆𑀤𑀫𑀸𑀦𑀸 𑀦𑀫𑀲𑁆𑀲𑀣𑁇_  


(82.)

83. _𑀏𑀯𑀁 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑀸 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀸;_  
_𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑁂 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑁄”𑁇_  


84. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀩𑀼𑀤𑁆𑀥𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀪𑀸𑀯𑀬𑀫𑀸𑀦𑁄 𑀩𑀼𑀤𑁆𑀥𑀸𑀧𑀤𑀸𑀦𑀺𑀬𑀁 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---
