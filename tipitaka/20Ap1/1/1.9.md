# 𑁧𑁇𑁯 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(596.)

616\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀪𑀽𑀫𑀺𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀧𑀞𑀫𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀳𑀁𑁇_  


(597.)

617\. _𑀬𑀸𑀯𑀢𑀸 𑀩𑁄𑀥𑀺𑀬𑀸 𑀫𑀽𑀮𑁂,_  
_𑀬𑀓𑁆𑀔𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀚𑀮𑀻𑀓𑀢𑀸𑁇_  


(598.)

618\. _𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸 𑀢𑀼𑀝𑁆𑀞𑀫𑀦𑀸,_  
_𑀆𑀓𑀸𑀲𑁂 𑀲𑀜𑁆𑀘𑀭𑀦𑁆𑀢𑀺 𑀢𑁂;_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀬𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀢𑀫𑁄𑀦𑀼𑀤𑁄𑁇_  


(599.)

619\. _𑀢𑁂𑀲𑀁 𑀳𑀸𑀲𑀧𑀭𑁂𑀢𑀸𑀦𑀁,_  
_𑀫𑀳𑀸𑀦𑀸𑀤𑁄 𑀅𑀯𑀢𑁆𑀢𑀣;_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀬𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(600.)

620\. _𑀤𑁂𑀯𑀸𑀦𑀁 𑀕𑀺𑀭𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀸𑀘𑀸𑀲𑀪𑀺𑀫𑀼𑀤𑀻𑀭𑀺𑀳𑀁;_  
_𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀆𑀤𑀺𑀪𑀺𑀓𑁆𑀔𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(601.)

621\. _𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(602.)

622\. _𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀩𑁄𑀥𑀺𑀁 𑀅𑀚𑁆𑀛𑀕𑀫𑀁 𑀅𑀳𑀁;_  
_𑀇𑀤𑀁 𑀫𑁂 𑀧𑀞𑀫𑀁 𑀪𑀢𑁆𑀢𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀲𑁆𑀲 𑀬𑀸𑀧𑀦𑀁𑁇_  


(603.)

623\. _𑀢𑀼𑀲𑀺𑀢𑀸 𑀳𑀺 𑀇𑀥𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀬𑁄 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀁 𑀉𑀧𑀸𑀦𑀬𑀺;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑁄𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(604.)

624\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑁂 𑀅𑀪𑀺𑀪𑁄𑀢𑁆𑀯𑀸,_  
_𑀢𑀺𑀤𑀺𑀯𑀁 𑀆𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(605.)

625\. _𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀥𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻,_  
_𑀢𑀢𑁆𑀣 𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(606.)

626\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(607.)

627\. _𑀢𑀺𑀤𑀲𑀸 𑀲𑁄 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀕𑀸𑀭𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀙𑀩𑁆𑀩𑀲𑁆𑀲𑀸𑀦𑀺 𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(608.)

628\. _‘𑀢𑀢𑁄 𑀲𑀢𑁆𑀢𑀫𑀓𑁂 𑀯𑀲𑁆𑀲𑁂,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀘𑁆𑀘𑀁 𑀓𑀣𑁂𑀲𑁆𑀲𑀢𑀺;_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀧𑀞𑀫𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀳𑀺𑀢𑀺’𑁇_  


(609.)

629\. _𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂𑀦𑀸𑀦𑀼𑀧𑀩𑁆𑀩𑀚𑀺𑀁,_  
_𑀧𑀥𑀸𑀦𑀁 𑀲𑀼𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀦𑀢𑁆𑀣𑀸𑀬,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(610.)

630\. _𑀅𑀪𑀺𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀇𑀲𑀺𑀦𑀸𑀫𑁂 𑀫𑀺𑀕𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀅𑀫𑀢𑀪𑁂𑀭𑀺𑀫𑀸𑀳𑀦𑀺𑁇_  


(611.)

631\. _𑀲𑁄 𑀤𑀸𑀦𑀺 𑀧𑀢𑁆𑀢𑁄 𑀅𑀫𑀢𑀁,_  
_𑀲𑀦𑁆𑀢𑀺𑀧𑀤𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(612.)

632\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


633\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

634\. 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
