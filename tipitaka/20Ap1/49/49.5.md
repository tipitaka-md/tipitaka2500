# 𑁪𑁯𑁇𑁫 𑀘𑀦𑁆𑀤𑀦𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀯𑀕𑁆𑀕

### 𑀘𑀦𑁆𑀤𑀦𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(75.)

6268. _“𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀺𑀬𑀭𑀽𑀧𑁂 𑀫𑀦𑁄𑀭𑀫𑁂;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀬𑁄 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(76.)

6269. _𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀓𑀸𑀬𑁂𑀦,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑀬𑀺𑀁;_  
_𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀦𑀤𑀻𑀓𑀽𑀮𑁂 𑀯𑀲𑀸𑀫𑀳𑀁𑁇_  


(77.)

6270. _𑀏𑀓𑀓𑀁 𑀫𑀁 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑁄 𑀉𑀧𑀸𑀕𑀫𑀺;_  
_𑀦𑀸𑀳𑀁 𑀚𑀸𑀦𑀸𑀫𑀺 𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺,_  
_𑀅𑀓𑀸𑀲𑀺𑀁 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀁𑁇_  


(78.)

6271. _𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀁,_  
_𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀫𑀧𑀼𑀘𑁆𑀙𑀳𑀁;_  
_‘𑀤𑁂𑀯𑀢𑀸𑀦𑀼𑀲𑀺 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀅𑀤𑀼 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄𑁇_  


(79.)

6272. _𑀓𑁄 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀇𑀥𑀸𑀕𑀢𑁄;_  
_𑀯𑀺𑀭𑁄𑀘𑁂𑀲𑀺 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀉𑀤𑀬𑀁 𑀲𑀽𑀭𑀺𑀬𑁄 𑀬𑀣𑀸𑁇_  


(80.)

6273. _𑀲𑀳𑀲𑁆𑀲𑀸𑀭𑀸𑀦𑀺 𑀘𑀓𑁆𑀓𑀸𑀦𑀺,_  
_𑀧𑀸𑀤𑁂 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑀸𑀭𑀺𑀲;_  
_𑀓𑁄 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁;_  
_𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀧𑀯𑁂𑀤𑁂𑀳𑀺,_  
_𑀲𑀁𑀲𑀬𑀁 𑀅𑀧𑀦𑁂𑀳𑀺 𑀫𑁂𑁇_  


(81.)

6274. _𑀦𑀫𑁆𑀳𑀺 𑀤𑁂𑀯𑁄 𑀦 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀦𑀫𑁆𑀳𑀺 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀪𑀸𑀯𑁄 𑀘 𑀫𑁂 𑀦𑀢𑁆𑀣𑀺,_  
_𑀏𑀢𑁂𑀲𑀁 𑀉𑀢𑁆𑀢𑀫𑁄 𑀅𑀳𑀁𑁇_  


(82.)

6275. _𑀅𑀢𑀻𑀢𑁄 𑀯𑀺𑀲𑀬𑀁 𑀢𑁂𑀲𑀁,_  
_𑀤𑀸𑀮𑀬𑀺𑀁 𑀓𑀸𑀫𑀩𑀦𑁆𑀥𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁’𑁇_  


(83.)

6276. _𑀢𑀲𑁆𑀲 𑀯𑀸𑀘𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀳𑀁,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁;_  
_‘𑀬𑀤𑀺 𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽,_  
_𑀦𑀺𑀲𑀻𑀤 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(84.)

6277. _𑀢𑀫𑀳𑀁 𑀧𑀽𑀚𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀓𑀭𑁄 𑀢𑀼𑀯𑀁’;_  
_𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀘𑀫𑁆𑀫𑀁,_  
_𑀅𑀤𑀸𑀲𑀺 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(85.)

6278. _𑀦𑀺𑀲𑀻𑀤𑀺 𑀢𑀢𑁆𑀣 𑀪𑀕𑀯𑀸,_  
_𑀲𑀻𑀳𑁄𑀯 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑁂;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀩𑁆𑀩𑀢𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀅𑀫𑁆𑀩𑀲𑁆𑀲 𑀨𑀮𑀫𑀕𑁆𑀕𑀳𑀺𑀁𑁇_  


(86.)

6279. _𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀘𑀦𑁆𑀤𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀭𑀳𑀁;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(87.)

6280. _𑀨𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀫𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀘𑀦𑁆𑀤𑀦𑀁 𑀅𑀦𑀼𑀮𑀺𑀫𑁆𑀧𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(88.)

6281. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀯𑀺𑀧𑀼𑀮𑀸𑀬 𑀘 𑀧𑀻𑀢𑀺𑀬𑀸;_  
_𑀅𑀚𑀺𑀦𑀫𑁆𑀳𑀺 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(89.)

6282. _𑀫𑀫 𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀲𑀺,_  
_𑀳𑀸𑀲𑀬𑀦𑁆𑀢𑁄 𑀫𑀫𑀁 𑀢𑀤𑀸;_  
_‘𑀇𑀫𑀺𑀦𑀸 𑀨𑀮𑀤𑀸𑀦𑁂𑀦,_  
_𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑁂𑀳𑀺 𑀘𑀽𑀪𑀬𑀁𑁇_  


(90.)

6283. _𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀦𑀽𑀦𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(91.)

6284. _𑀙𑀩𑁆𑀩𑀻𑀲𑀢𑀺𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻,_  
_𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄𑁇_  


(92.)

6285. _𑀯𑁂𑀪𑀸𑀭𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁂𑀦 𑀫𑀸𑀧𑀺𑀢𑀁;_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄𑀯𑀡𑁆𑀡𑀁,_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀪𑀽𑀲𑀺𑀢𑀁𑁇_  


(93.)

6286. _𑀏𑀢𑁂𑀦𑁂𑀯 𑀉𑀧𑀸𑀬𑁂𑀦,_  
_𑀲𑀁𑀲𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀪𑀯𑁂;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀽𑀚𑀺𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑁂 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑁂𑁇_  


(94.)

6287. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀕𑀸𑀭𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀅𑀦𑀕𑀸𑀭𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀕𑀽 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(95.)

6288. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀫𑀫 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(96.)

6289. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(97.)

6290. _𑀢𑀼𑀲𑀺𑀢𑀢𑁄 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀁 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀬𑀁;_  
_𑀪𑁄𑀕𑁂 𑀫𑁂 𑀊𑀦𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀬𑀫𑁆𑀳𑀺 𑀕𑀩𑁆𑀪𑁂 𑀯𑀲𑀸𑀫𑀳𑀁𑁇_  


(98.)

6291. _𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑁂 𑀫𑀬𑀺,_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀁;_  
_𑀫𑀸𑀢𑀼𑀬𑀸 𑀫𑀫 𑀙𑀦𑁆𑀤𑁂𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀺 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(99.)

6292. _𑀚𑀸𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁂𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀑𑀭𑁄𑀧𑀺𑀢𑀫𑁆𑀳𑀺 𑀓𑁂𑀲𑀫𑁆𑀳𑀺,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(100.)

6293. _𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀑𑀭𑁂𑀦 𑀦𑀸𑀤𑁆𑀤𑀲𑀁 𑀅𑀳𑀁;_  
_𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀫𑀫 𑀓𑀫𑁆𑀫𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁𑁇_  


(101.)

6294. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀢𑀯 𑀲𑀸𑀲𑀦𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁𑁇_  


(102.)

6295. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(103.)

6296. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(104.)

6297. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(105.)

6298. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6299. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀦𑁆𑀤𑀦𑀫𑀸𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6300. 𑀘𑀦𑁆𑀤𑀦𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
