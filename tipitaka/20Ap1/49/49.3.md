# 𑁪𑁯𑁇𑁩 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(27.)

6215\. _“𑀑𑀕𑀬𑁆𑀳 𑀬𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁,_  
_𑀦𑀸𑀦𑀸𑀓𑀼𑀜𑁆𑀚𑀭𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀉𑀤𑁆𑀥𑀭𑀸𑀫𑀺 𑀪𑀺𑀲𑀁 𑀢𑀢𑁆𑀣,_  
_𑀖𑀸𑀲𑀳𑁂𑀢𑀼 𑀅𑀳𑀁 𑀢𑀤𑀸𑁇_  


(28.)

6216\. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑀯𑁆𑀳𑀬𑁄;_  
_𑀭𑀢𑁆𑀢𑀫𑁆𑀩𑀭𑀥𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀘𑁆𑀙𑀢𑀺 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(29.)

6217\. _𑀥𑀼𑀦𑀦𑁆𑀢𑁄 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺,_  
_𑀲𑀤𑁆𑀤𑀫𑀲𑁆𑀲𑁄𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀉𑀤𑁆𑀥𑀁 𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀫𑀸𑀦𑁄𑀳𑀁,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(30.)

6218\. _𑀢𑀢𑁆𑀣𑁂𑀯 𑀞𑀺𑀢𑀓𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀆𑀬𑀸𑀘𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_‘𑀫𑀥𑀼𑀁 𑀪𑀺𑀲𑁂𑀳𑀺 𑀲𑀯𑀢𑀺,_  
_𑀔𑀻𑀭𑀁 𑀲𑀧𑁆𑀧𑀺𑀁 𑀫𑀼𑀍𑀆𑀮𑀺𑀪𑀺𑁇_  


(31.)

6219\. _𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀘𑀓𑁆𑀔𑀼𑀫𑀸’;_  
_𑀢𑀢𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀑𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(32.)

6220\. _𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀺 𑀫𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀁,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀓𑀸 𑀫𑁂 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑀁𑁇_  


(33.)

6221\. _‘𑀲𑀼𑀔𑀻 𑀳𑁄𑀢𑀼 𑀫𑀳𑀸𑀧𑀼𑀜𑁆𑀜,_  
_𑀕𑀢𑀺 𑀢𑀼𑀬𑁆𑀳𑀁 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀼;_  
_𑀇𑀫𑀺𑀦𑀸 𑀪𑀺𑀲𑀤𑀸𑀦𑁂𑀦,_  
_𑀮𑀪𑀲𑁆𑀲𑀼 𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀼𑀔𑀁’𑁇_  


(34.)

6222\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀓𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀫𑀸𑀤𑀸𑀬 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀫𑁆𑀩𑀭𑁂𑀦𑀸𑀕𑀫𑀸 𑀚𑀺𑀦𑁄𑁇_  


(35.)

6223\. _𑀢𑀢𑁄 𑀪𑀺𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀕𑀘𑁆𑀙𑀺𑀁 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑀁;_  
_𑀪𑀺𑀲𑀁 𑀭𑀼𑀓𑁆𑀔𑁂 𑀮𑀕𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀫 𑀤𑀸𑀦𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁𑁇_  


(36.)

6224\. _𑀫𑀳𑀸𑀯𑀸𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀜𑁆𑀘𑀸𑀮𑁂𑀲𑀺 𑀯𑀦𑀁 𑀢𑀤𑀸;_  
_𑀆𑀓𑀸𑀲𑁄 𑀅𑀪𑀺𑀦𑀸𑀤𑀺𑀢𑁆𑀣,_  
_𑀅𑀲𑀦𑀺𑀬𑀸 𑀨𑀮𑀦𑁆𑀢𑀺𑀬𑀸𑁇_  


(37.)

6225\. _𑀢𑀢𑁄 𑀫𑁂 𑀅𑀲𑀦𑀻𑀧𑀸𑀢𑁄,_  
_𑀫𑀢𑁆𑀣𑀓𑁂 𑀦𑀺𑀧𑀢𑀻 𑀢𑀤𑀸;_  
_𑀲𑁄𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀓𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀼𑀁𑁇_  


(38.)

6226\. _𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀢𑀼𑀲𑀺𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀳𑀁;_  
_𑀓𑀍𑀏𑀯𑀭𑀁 𑀫𑁂 𑀧𑀢𑀺𑀢𑀁,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀁 𑀅𑀳𑀁𑁇_  


(39.)

6227\. _𑀙𑀍𑀅𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀪𑀺𑀲𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(40.)

6228\. _𑀫𑀦𑀼𑀲𑁆𑀲𑀬𑁄𑀦𑀺𑀫𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀫𑀳𑀁 𑀲𑀤𑀸;_  
_𑀪𑁄𑀕𑁂 𑀫𑁂 𑀊𑀦𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀪𑀺𑀲𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(41.)

6229\. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀢𑀓𑁄 𑀢𑁂𑀦,_  
_𑀤𑁂𑀯𑀤𑁂𑀯𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(42.)

6230\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀪𑀺𑀲𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑀺𑀲𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(43.)

6231\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(44.)

6232\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(45.)

6233\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6234\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6235\. 𑀪𑀺𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
