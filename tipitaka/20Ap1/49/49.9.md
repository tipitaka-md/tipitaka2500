# 𑁪𑁯𑁇𑁯 𑀥𑀫𑁆𑀫𑀭𑀼𑀘𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀭𑀼𑀘𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(164.)

6365\. _“𑀬𑀤𑀸 𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀼𑀫𑁂𑀥𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀻 𑀚𑀺𑀦𑁄;_  
_‘𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(165.)

6366\. _𑀇𑀫𑀲𑁆𑀲 𑀚𑀦𑀺𑀓𑀸 𑀫𑀸𑀢𑀸,_  
_𑀫𑀸𑀬𑀸 𑀦𑀸𑀫 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀺𑀢𑀸 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀦𑀸𑀫,_  
_𑀅𑀬𑀁 𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀕𑁄𑀢𑀫𑁄𑁇_  


(166.)

6367\. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀤𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀁;_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀫𑀽𑀮𑁂 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀩𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(167.)

6368\. _𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀓𑁄𑀮𑀺𑀢𑁄 𑀘,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀸𑀯𑀓𑀸;_  
_𑀆𑀦𑀦𑁆𑀤𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀲𑁆𑀲𑀢𑀺𑀫𑀁 𑀚𑀺𑀦𑀁𑁇_  


(168.)

6369\. _𑀔𑁂𑀫𑀸 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀍𑀅𑀯𑀓𑁄 𑀘𑁂𑀯,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀼𑀧𑀸𑀲𑀓𑀸𑁇_  


(169.)

6370\. _𑀔𑀼𑀚𑁆𑀚𑀼𑀢𑁆𑀢𑀭𑀸 𑀦𑀦𑁆𑀤𑀫𑀸𑀢𑀸,_  
_𑀅𑀕𑁆𑀕𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀼𑀧𑀸𑀲𑀺𑀓𑀸;_  
_𑀩𑁄𑀥𑀺 𑀇𑀫𑀲𑁆𑀲 𑀯𑀻𑀭𑀲𑁆𑀲,_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑁄𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺’𑁇_  


(170.)

6371\. _𑀇𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀯𑀘𑀦𑀁,_  
_𑀅𑀲𑀫𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀆𑀫𑁄𑀤𑀺𑀢𑀸 𑀦𑀭𑀫𑀭𑀽,_  
_𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀢𑀜𑁆𑀚𑀮𑀻𑁇_  


(171.)

6372\. _𑀢𑀤𑀸𑀳𑀁 𑀫𑀸𑀡𑀯𑁄 𑀆𑀲𑀺𑀁,_  
_𑀫𑁂𑀖𑁄 𑀦𑀸𑀫 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀲𑀼𑀫𑁂𑀥𑀲𑁆𑀲 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(172.)

6373\. _𑀲𑀁𑀯𑀺𑀲𑀝𑁆𑀞𑁄 𑀪𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀫𑁂𑀥𑁂 𑀓𑀭𑀼𑀡𑀸𑀲𑀬𑁂;_  
_𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀜𑁆𑀘 𑀢𑀁 𑀯𑀻𑀭𑀁,_  
_𑀲𑀳𑀸𑀯 𑀅𑀦𑀼𑀧𑀩𑁆𑀩𑀚𑀺𑀁𑁇_  


(173.)

6374\. _𑀲𑀁𑀯𑀼𑀢𑁄 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀁,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀘 𑀧𑀜𑁆𑀘𑀲𑀼;_  
_𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁄 𑀲𑀢𑁄 𑀯𑀻𑀭𑁄,_  
_𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀓𑁄𑁇_  


(174.)

6375\. _𑀏𑀯𑀁 𑀯𑀺𑀳𑀭𑀫𑀸𑀦𑁄𑀳𑀁,_  
_𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀓𑁂𑀦𑀘𑀺;_  
_𑀦𑀺𑀬𑁄𑀚𑀺𑀢𑁄 𑀅𑀦𑀸𑀘𑀸𑀭𑁂,_  
_𑀲𑀼𑀫𑀕𑁆𑀕𑀸 𑀧𑀭𑀺𑀥𑀁𑀲𑀺𑀢𑁄𑁇_  


(175.)

6376\. _𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀲𑀺𑀓𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀸𑀲𑀦𑀢𑁄 𑀅𑀧𑀓𑁆𑀓𑀫𑀺𑀁;_  
_𑀧𑀘𑁆𑀙𑀸 𑀢𑁂𑀦 𑀓𑀼𑀫𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀦𑀁𑁇_  


(176.)

6377\. _𑀅𑀓𑀭𑀺𑀁 𑀆𑀦𑀦𑁆𑀢𑀭𑀺𑀬𑀁,_  
_𑀖𑀸𑀢𑀬𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀫𑀸𑀦𑀲𑁄;_  
_𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀫𑀳𑀸𑀯𑀻𑀘𑀺𑀁,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀤𑀸𑀭𑀼𑀡𑀁𑁇_  


(177.)

6378\. _𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀕𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀁𑀲𑀭𑀺𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀘𑀺𑀭𑀁;_  
_𑀦 𑀧𑀼𑀦𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀻𑀭𑀁,_  
_𑀲𑀼𑀫𑁂𑀥𑀁 𑀦𑀭𑀧𑀼𑀗𑁆𑀕𑀯𑀁𑁇_  


(178.)

6379\. _𑀅𑀲𑁆𑀫𑀺𑀁 𑀓𑀧𑁆𑀧𑁂 𑀲𑀫𑀼𑀤𑁆𑀤𑀫𑁆𑀳𑀺,_  
_𑀫𑀘𑁆𑀙𑁄 𑀆𑀲𑀺𑀁 𑀢𑀺𑀫𑀺𑀗𑁆𑀕𑀮𑁄;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀳𑀁 𑀲𑀸𑀕𑀭𑁂 𑀦𑀸𑀯𑀁,_  
_𑀕𑁄𑀘𑀭𑀢𑁆𑀣𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(179.)

6380\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀁 𑀯𑀸𑀡𑀺𑀚𑀸 𑀪𑀻𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀼𑀁;_  
_𑀕𑁄𑀢𑀫𑁄𑀢𑀺 𑀫𑀳𑀸𑀖𑁄𑀲𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑁂𑀳𑀺 𑀉𑀤𑀻𑀭𑀺𑀢𑀁𑁇_  


(180.)

6381\. _𑀧𑀼𑀩𑁆𑀩𑀲𑀜𑁆𑀜𑀁 𑀲𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁;_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀓𑀼𑀮𑁂 𑀇𑀤𑁆𑀥𑁂,_  
_𑀚𑀸𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀚𑀸𑀢𑀺𑀬𑀁𑁇_  


(181.)

6382\. _𑀆𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀭𑀼𑀘𑀺 𑀦𑀸𑀫,_  
_𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀓𑁄;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀳𑀁 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑀁,_  
_𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑁄𑁇_  


(182.)

6383\. _𑀫𑀳𑀸𑀚𑁂𑀢𑀯𑀦𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀉𑀧𑁂𑀫𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁,_  
_𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀺𑀯𑀲𑀲𑁆𑀲 𑀘𑁇_  


(183.)

6384\. _𑀢𑀤𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀼𑀦𑀺 𑀆𑀳,_  
_𑀘𑀺𑀭𑀁 𑀥𑀫𑁆𑀫𑀭𑀼𑀘𑀻𑀢𑀺 𑀫𑀁;_  
_𑀢𑀢𑁄𑀳𑀁 𑀅𑀯𑀘𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀧𑀪𑀸𑀯𑀺𑀢𑀁𑁇_  


(184.)

6385\. _𑀲𑀼𑀘𑀺𑀭𑀁 𑀲𑀢𑀧𑀼𑀜𑁆𑀜𑀮𑀓𑁆𑀔𑀡𑀁,_  
_𑀧𑀢𑀺𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀧𑀘𑁆𑀘𑀬𑀁;_  
_𑀅𑀳𑀫𑀚𑁆𑀚𑀲𑀼𑀧𑁂𑀓𑁆𑀔𑀦𑀁 𑀯𑀢,_  
_𑀢𑀯 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀺𑀭𑀼𑀧𑀫𑀁 𑀯𑀺𑀕𑁆𑀕𑀳𑀁𑁇_  


(185.)

6386\. _𑀲𑀼𑀘𑀺𑀭𑀁 𑀯𑀺𑀳𑀢𑀢𑁆𑀢𑀫𑁄 𑀫𑀬𑀸,_  
_𑀲𑀼𑀘𑀺𑀭𑀓𑁆𑀔𑁂𑀦 𑀦𑀤𑀻 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸;_  
_𑀲𑀼𑀘𑀺𑀭𑀁 𑀅𑀫𑀮𑀁 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁,_  
_𑀦𑀬𑀦𑀁 𑀜𑀸𑀡𑀫𑀬𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(186.)

6387\. _𑀘𑀺𑀭𑀓𑀸𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀢𑁄 𑀢𑀬𑀸,_  
_𑀅𑀯𑀺𑀦𑀝𑁆𑀞𑁄 𑀧𑀼𑀦𑀭𑀦𑁆𑀢𑀭𑀁 𑀘𑀺𑀭𑀁;_  
_𑀧𑀼𑀦𑀭𑀚𑁆𑀚𑀲𑀫𑀸𑀕𑀢𑁄 𑀢𑀬𑀸,_  
_𑀦 𑀳𑀺 𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀢𑀸𑀦𑀺 𑀕𑁄𑀢𑀫𑁇_  


(187.)

6388\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(188.)

6389\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(189.)

6390\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6391\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀭𑀼𑀘𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
