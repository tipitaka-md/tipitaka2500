# 𑁩𑁧𑁇𑁯 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

### 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(43.)

3490\. _“𑀉𑀤𑁂𑀦𑁆𑀢𑀁 𑀲𑀢𑀭𑀁𑀲𑀺𑀁𑀯,_  
_𑀧𑀻𑀢𑀭𑀁𑀲𑀺𑀁𑀯 𑀪𑀸𑀡𑀼𑀫𑀁;_  
_𑀯𑀦𑀦𑁆𑀢𑀭𑀕𑀢𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(44.)

3491\. _𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀼𑀧𑀺𑀦𑀦𑁆𑀢𑁂𑀦,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀕𑀢𑀺𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(45.)

3492\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(46.)

3493\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3494\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3495\. 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
