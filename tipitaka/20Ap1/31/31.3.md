# 𑁩𑁧𑁇𑁩 𑀧𑀭𑀫𑀦𑁆𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

* 𑀧𑀭𑀫𑀦𑁆𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(9.)

3444\. _“𑀓𑀡𑀺𑀓𑀸𑀭𑀁𑀯 𑀚𑁄𑀢𑀦𑁆𑀢𑀁,_  
_𑀉𑀤𑀬𑀦𑁆𑀢𑀁𑀯 𑀪𑀸𑀡𑀼𑀫𑀁;_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(10.)

3445\. _𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀪𑀺𑀦𑁂𑀲𑀺𑀁 𑀲𑀓𑀁 𑀖𑀭𑀁;_  
_𑀅𑀪𑀺𑀦𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(11.)

3446\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀫𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀭𑀫𑀦𑁆𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(12.)

3447\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3448\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀭𑀫𑀦𑁆𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3449\. 𑀧𑀭𑀫𑀦𑁆𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
