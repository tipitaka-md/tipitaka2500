# 𑁩𑁧𑁇𑁫 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

### 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(20.)

3459\. _“𑀪𑀸𑀕𑀻𑀭𑀣𑀻𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀅𑀳𑁄𑀲𑀺 𑀅𑀲𑁆𑀲𑀫𑁄 𑀢𑀤𑀸;_  
_𑀢𑀫𑀳𑀁 𑀅𑀲𑁆𑀲𑀫𑀁 𑀕𑀘𑁆𑀙𑀺𑀁,_  
_𑀨𑀮𑀳𑀢𑁆𑀣𑁄 𑀅𑀧𑁂𑀓𑁆𑀔𑀯𑀸𑁇_  


(21.)

3460\. _𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀢𑀢𑁆𑀣 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀺𑀁,_  
_𑀧𑀻𑀢𑀭𑀁𑀲𑀺𑀁𑀯 𑀪𑀸𑀡𑀼𑀫𑀁;_  
_𑀬𑀁 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺 𑀨𑀮𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(22.)

3461\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(23.)

3462\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3463\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3464\. 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
