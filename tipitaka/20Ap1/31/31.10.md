# 𑁩𑁧𑁇𑁧𑁦 𑀧𑀩𑁆𑀪𑀸𑀭𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

### 𑀧𑀩𑁆𑀪𑀸𑀭𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(47.)

3496\. _“𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑁄 𑀲𑁄𑀥𑀺𑀢𑁄 𑀫𑀬𑀸;_  
_𑀖𑀝𑀓𑀜𑁆𑀘 𑀉𑀧𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀧𑀭𑀺𑀪𑁄𑀕𑀸𑀬 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(48.)

3497\. _𑀢𑀁 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑀡𑁆𑀟𑁄 𑀲𑀢𑀪𑁂𑀡𑁆𑀟𑀼,_  
_𑀥𑀚𑀸𑀮𑀼 𑀳𑀭𑀺𑀢𑀸𑀫𑀬𑁄𑁇_  


(49.)

3498\. _𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀬𑀽𑀧𑁄,_  
_𑀭𑀢𑀦𑀜𑁆𑀘 𑀅𑀦𑀧𑁆𑀧𑀓𑀁;_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀳𑀁𑁇_  


(50.)

3499\. _𑀇𑀢𑁄 𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀲𑀼𑀲𑀼𑀤𑁆𑀥𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(51.)

3500\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁭𑁬𑁩)_  


3501\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀪𑀸𑀭𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3502\. 𑀧𑀩𑁆𑀪𑀸𑀭𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

3503\. 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕𑁄 𑀏𑀓𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁄𑁇



---

3504\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3505\. _𑀓𑁂𑀲𑀭𑀁 𑀕𑀦𑁆𑀥𑀫𑀦𑁆𑀦𑀜𑁆𑀘,_  
_𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻 𑀨𑀮𑁂𑀦 𑀘;_  
_𑀧𑀲𑀸𑀤𑀸𑀭𑀸𑀫𑀤𑀸𑀬𑀻 𑀘,_  
_𑀮𑁂𑀧𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑁄;_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑀤𑁄 𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀏𑀓𑀧𑀜𑁆𑀜𑀸𑀲 𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇_
