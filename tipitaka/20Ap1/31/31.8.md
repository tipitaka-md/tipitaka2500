# 𑁩𑁧𑁇𑁮 𑀅𑀦𑀼𑀮𑁂𑀧𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀼𑀮𑁂𑀧𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(39.)

3484\. _“𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀸𑀯𑀓𑀁 𑀅𑀳𑀁;_  
_𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑀻𑀫𑀸𑀬 𑀉𑀧𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(40.)

3485\. _𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀦𑀯𑀓𑀫𑁆𑀫𑁂 𑀘,_  
_𑀅𑀦𑀼𑀮𑁂𑀧𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂𑁇_  


(41.)

3486\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀦𑀼𑀮𑁂𑀧𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(42.)

3487\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3488\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀮𑁂𑀧𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3489\. 𑀅𑀦𑀼𑀮𑁂𑀧𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
