# 𑁩𑁧𑁇𑁪 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(13.)

3450. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀫𑀳𑀸𑀩𑁄𑀥𑀺𑀫𑀳𑁄 𑀅𑀳𑀼;_  
_𑀭𑀼𑀓𑁆𑀔𑀝𑁆𑀞𑀲𑁆𑀲𑁂𑀯 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄𑁇_  


(14.)

3451. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀁 𑀧𑀓𑀸𑀲𑁂𑀢𑀺,_  
_𑀯𑀸𑀘𑀸𑀲𑀪𑀺𑀫𑀼𑀤𑀻𑀭𑀬𑀁𑁇_  


(15.)

3452. _𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀘 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀘 𑀤𑁂𑀲𑀬𑀁;_  
_𑀯𑀺𑀯𑀝𑁆𑀝𑀘𑁆𑀙𑀤𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀲𑀺 𑀫𑀳𑀸𑀚𑀦𑀁𑁇_  


(16.)

3453. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(17.)

3454. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀥𑀫𑁆𑀫𑀫𑀲𑀼𑀡𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀥𑀫𑁆𑀫𑀲𑀯𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(18.)

3455. _𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑀫𑁆𑀳𑀺 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_𑀲𑀼𑀢𑀯𑀸 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(19.)

3456. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3457. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3458. 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
