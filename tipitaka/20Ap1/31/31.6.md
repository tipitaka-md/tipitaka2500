# 𑁩𑁧𑁇𑁬 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

### 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(24.)

3465\. _“‘𑀦𑀫𑁄 𑀢𑁂 𑀩𑀼𑀤𑁆𑀥 𑀯𑀻𑀭𑀢𑁆𑀣𑀼,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄𑀲𑀺 𑀲𑀩𑁆𑀩𑀥𑀺;_  
_𑀩𑁆𑀬𑀲𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀭𑀡𑀁 𑀪𑀯’𑁇_  


(25.)

3466\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀮𑁄𑀓𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_‘𑀫𑀳𑁄𑀤𑀥𑀺𑀲𑀫𑁄 𑀲𑀁𑀖𑁄,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄𑁇_  


(26.)

3467\. _𑀢𑀢𑁆𑀣 𑀢𑁆𑀯𑀁 𑀯𑀺𑀭𑀚𑁂 𑀔𑁂𑀢𑁆𑀢𑁂,_  
_𑀅𑀦𑀦𑁆𑀢𑀨𑀮𑀤𑀸𑀬𑀓𑁂;_  
_𑀲𑀁𑀖𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀩𑀻𑀚𑀁 𑀯𑀸𑀧 𑀭𑁄𑀧𑀬’𑁇_  


(27.)

3468\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀫𑀫𑁂𑀯 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑁂𑀳𑀸𑀲𑀁 𑀦𑀪𑀫𑀼𑀕𑁆𑀕𑀫𑀺𑁇_  


(28.)

3469\. _𑀅𑀘𑀺𑀭𑀁 𑀕𑀢𑀫𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀫𑁆𑀳𑀺 𑀦𑀭𑀸𑀲𑀪𑁂;_  
_𑀫𑀭𑀡𑀁 𑀲𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀢𑀼𑀲𑀺𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(29.)

3470\. _𑀢𑀤𑀸𑀳𑀁 𑀯𑀺𑀭𑀚𑁂 𑀔𑁂𑀢𑁆𑀢𑁂,_  
_𑀅𑀦𑀦𑁆𑀢𑀨𑀮𑀤𑀸𑀬𑀓𑁂;_  
_𑀲𑀁𑀖𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀳𑀁𑁇_  


(30.)

3471\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀲𑀸𑀤𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀲𑀸𑀤𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(31.)

3472\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3473\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3474\. 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
