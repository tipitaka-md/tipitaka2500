# 𑁩𑁧𑁇𑁭 𑀆𑀭𑀸𑀫𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀓𑁂𑀲𑀭𑀯𑀕𑁆𑀕

### 𑀆𑀭𑀸𑀫𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(32.)

3475\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀆𑀭𑀸𑀫𑁄 𑀭𑁄𑀧𑀺𑀢𑁄 𑀫𑀬𑀸;_  
_𑀲𑀦𑁆𑀤𑀘𑁆𑀙𑀸𑀬𑁂𑀲𑀼 𑀭𑀼𑀓𑁆𑀔𑁂𑀲𑀼,_  
_𑀉𑀧𑀸𑀲𑀦𑁆𑀢𑁂𑀲𑀼 𑀧𑀓𑁆𑀔𑀺𑀲𑀼𑁇_  


(33.)

3476\. _𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀆𑀭𑀸𑀫𑀁 𑀅𑀪𑀺𑀦𑀸𑀫𑁂𑀲𑀺𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(34.)

3477\. _𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀨𑀮𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀢𑀢𑁄 𑀚𑀸𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁄𑀯,_  
_𑀢𑀁 𑀯𑀦𑀁 𑀧𑀭𑀺𑀡𑀸𑀫𑀬𑀺𑀁𑁇_  


(35.)

3478\. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀬𑀫𑀺𑀤𑀁 𑀤𑀸𑀲𑀺𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀪𑀯𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀺 𑀨𑀮𑀁 𑀫𑀫𑁇_  


(36.)

3479\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀆𑀭𑀸𑀫𑀫𑀤𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀆𑀭𑀸𑀫𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(37.)

3480\. _𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑁆𑀢𑀸𑀲𑀼𑀁 𑀫𑀼𑀤𑀼𑀲𑀻𑀢𑀮𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(38.)

3481\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3482\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀭𑀸𑀫𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3483\. 𑀆𑀭𑀸𑀫𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
