# 𑁮𑁇𑁩 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(9.)

1601\. _“𑀤𑀼𑀫𑀕𑁆𑀕𑁂 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀁,_  
_𑀮𑀕𑁆𑀕𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀢𑀢𑁄 𑀢𑀫𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀅𑀯𑀦𑁆𑀤𑀳𑀁𑁇_  


(10.)

1602\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(11.)

1603\. _𑀇𑀢𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀤𑀼𑀫𑀲𑀸𑀭𑁄𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(12.)

1604\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1605\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1606\. 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
