# 𑁮𑁇𑁫 𑀏𑀓𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(18.)

1614. _“𑀔𑀡𑁆𑀟𑁄 𑀦𑀸𑀫𑀸𑀲𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑁄;_  
_𑀏𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀸 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀮𑁄𑀓𑀸𑀳𑀼𑀢𑀺𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑁇_  


(19.)

1615. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤 𑀦𑀭𑀸𑀲𑀪;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(20.)

1616. _𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀯𑀭𑀼𑀡𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(21.)

1617. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1618. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1619. 𑀏𑀓𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
