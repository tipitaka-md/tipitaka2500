# 𑁮𑁇𑁧 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

1589\. _“𑀆𑀧𑀸𑀝𑀮𑀺𑀁 𑀅𑀳𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀉𑀚𑁆𑀛𑀺𑀢𑀁 𑀲𑀼𑀫𑀳𑀸𑀧𑀣𑁂;_  
_𑀣𑀽𑀧𑀫𑁆𑀳𑀺 𑀅𑀪𑀺𑀭𑁄𑀧𑁂𑀲𑀺𑀁,_  
_𑀲𑀺𑀔𑀺𑀦𑁄 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼𑀦𑁄𑁇_  


(2.)

1590\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀣𑀽𑀧𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(3.)

1591\. _𑀇𑀢𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀪𑀽𑀫𑀺𑀬𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(4.)

1592\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1593\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1594\. 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
