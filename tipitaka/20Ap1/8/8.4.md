# 𑁮𑁇𑁪 𑀪𑀺𑀲𑀸𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀲𑀸𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(13.)

1607\. _“𑀓𑀸𑀦𑀦𑀁 𑀯𑀦𑀫𑁄𑀕𑀬𑁆𑀳,_  
_𑀯𑀲𑀸𑀫𑀺 𑀯𑀺𑀧𑀺𑀦𑁂 𑀅𑀳𑀁;_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁𑁇_  


(14.)

1608\. _𑀪𑀺𑀲𑀸𑀮𑀼𑀯𑀜𑁆𑀘 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀉𑀤𑀓𑀁 𑀳𑀢𑁆𑀣𑀥𑁄𑀯𑀦𑀁;_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀺𑀭𑀲𑀸 𑀧𑀸𑀤𑁂,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(15.)

1609\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀪𑀺𑀲𑀸𑀮𑀼𑀯𑀫𑀤𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(16.)

1610\. _𑀇𑀢𑁄 𑀢𑀢𑀺𑀬𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀪𑀺𑀲𑀲𑀫𑁆𑀫𑀢𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(17.)

1611\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1612\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀺𑀲𑀸𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1613\. 𑀪𑀺𑀲𑀸𑀮𑀼𑀯𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇

  
𑀙𑀝𑁆𑀞𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
