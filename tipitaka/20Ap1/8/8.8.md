# 𑁮𑁇𑁮 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(36.)

1638. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀡𑁂;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑀁,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑀁𑁇_  


(37.)

1639. _𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀤𑀸 𑀆𑀲𑀺𑀁,_  
_𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄;_  
_𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀅𑀪𑀺𑀲𑀁𑀳𑀭𑀺𑀁𑁇_  


(38.)

1640. _𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺 𑀅𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑀁 𑀮𑁄𑀓𑀯𑀺𑀤𑀼𑀁 𑀦𑀸𑀣𑀁,_  
_𑀦𑀭𑀤𑁂𑀯𑀁 𑀦𑀫𑀲𑁆𑀲𑀳𑀁𑁇_  


(39.)

1641. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(40.)

1642. _𑀇𑀢𑁄 𑀢𑁂𑀲𑀝𑁆𑀞𑀺𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀅𑀪𑀺𑀲𑀫𑁆𑀫𑀢𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(41.)

1643. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1644. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1645. 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
