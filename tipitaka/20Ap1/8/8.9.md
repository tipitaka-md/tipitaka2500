# 𑁮𑁇𑁯 𑀞𑀺𑀢𑀜𑁆𑀚𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀞𑀺𑀢𑀜𑁆𑀚𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(42.)

1646\. _“𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀦𑀦𑁂 𑀅𑀳𑀁;_  
_𑀢𑀢𑁆𑀣 𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑀁𑁇_  


(43.)

1647\. _𑀢𑀢𑁆𑀣𑀸𑀳𑀁 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀧𑀸𑀘𑀺𑀦𑀸𑀫𑀼𑀔𑁄;_  
_𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀦𑀺𑀬𑀓𑁂 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂𑁇_  


(44.)

1648\. _𑀢𑀢𑁄 𑀫𑁂 𑀅𑀲𑀦𑀻𑀧𑀸𑀢𑁄,_  
_𑀫𑀢𑁆𑀣𑀓𑁂 𑀦𑀺𑀧𑀢𑀻 𑀢𑀤𑀸;_  
_𑀲𑁄𑀳𑀁 𑀫𑀭𑀡𑀓𑀸𑀮𑀫𑁆𑀳𑀺,_  
_𑀅𑀓𑀸𑀲𑀺𑀁 𑀧𑀼𑀦𑀭𑀜𑁆𑀚𑀮𑀺𑀁𑁇_  


(45.)

1649\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀅𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(46.)

1650\. _𑀘𑀢𑀼𑀧𑀡𑁆𑀡𑀸𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀫𑀺𑀕𑀓𑁂𑀢𑀼𑀲𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(47.)

1651\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1652\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀞𑀺𑀢𑀜𑁆𑀚𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1653\. 𑀞𑀺𑀢𑀜𑁆𑀚𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
