# 𑁮𑁇𑁭 𑀲𑀽𑀘𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀲𑀽𑀘𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(30.)

1630\. _“𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄𑁇_  


(31.)

1631\. _𑀢𑀲𑁆𑀲 𑀓𑀜𑁆𑀘𑀦𑀯𑀡𑁆𑀡𑀲𑁆𑀲,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀧𑀜𑁆𑀘 𑀲𑀽𑀘𑀻 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀲𑀺𑀩𑁆𑀩𑀦𑀢𑁆𑀣𑀸𑀬 𑀘𑀻𑀯𑀭𑀁𑁇_  


(32.)

1632\. _𑀢𑁂𑀦𑁂𑀯 𑀲𑀽𑀘𑀺𑀤𑀸𑀦𑁂𑀦,_  
_𑀦𑀺𑀧𑀼𑀡𑀢𑁆𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑀁;_  
_𑀢𑀺𑀓𑁆𑀔𑀁 𑀮𑀳𑀼𑀜𑁆𑀘 𑀨𑀸𑀲𑀼𑀜𑁆𑀘,_  
_𑀜𑀸𑀡𑀁 𑀫𑁂 𑀉𑀤𑀧𑀚𑁆𑀚𑀣𑁇_  


(33.)

1633\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(34.)

1634\. _𑀤𑁆𑀯𑀺𑀧𑀤𑀸𑀥𑀺𑀧𑀢𑀻 𑀦𑀸𑀫,_  
_𑀭𑀸𑀚𑀸𑀦𑁄 𑀘𑀢𑀼𑀭𑁄 𑀅𑀳𑀼𑀁;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(35.)

1635\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1636\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀽𑀘𑀺𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1637\. 𑀲𑀽𑀘𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
