# 𑁨𑁫𑁇𑁧𑁦 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀼𑀯𑀭𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(34.)

3085. _“𑀓𑀓𑀼𑀥𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀧𑀯𑀦𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀦𑀤𑀺𑀁𑁇_  


(35.)

3086. _𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀲𑀁𑀬𑀢𑀲𑁆𑀲𑀼𑀚𑀼𑀪𑀽𑀢𑀲𑁆𑀲,_  
_𑀧𑀲𑀦𑁆𑀦𑀫𑀸𑀦𑀲𑁄 𑀅𑀳𑀁𑁇_  


(36.)

3087. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(37.)

3088. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁪𑁯𑁪)_  


3089. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3090. 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

3091. 𑀢𑀼𑀯𑀭𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇



---

3092. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3093. _𑀢𑀼𑀯𑀭𑀦𑀸𑀕𑀦𑀍𑀇𑀦𑀸,_  
_𑀯𑀺𑀭𑀯𑀻 𑀓𑀼𑀝𑀺𑀥𑀽𑀧𑀓𑁄;_  
_𑀧𑀢𑁆𑀢𑁄 𑀥𑀸𑀢𑀼𑀲𑀢𑁆𑀢𑀮𑀺𑀬𑁄,_  
_𑀩𑀺𑀫𑁆𑀩𑀺 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑁂𑀦 𑀘;_  
_𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲𑀢𑀺 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_
