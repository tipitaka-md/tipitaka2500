# 𑁨𑁫𑁇𑁭 𑀥𑀸𑀢𑀼𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀼𑀯𑀭𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀥𑀸𑀢𑀼𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(23.)

3068\. _“𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀫𑁆𑀳𑀺 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀏𑀓𑀸 𑀥𑀸𑀢𑀼 𑀫𑀬𑀸 𑀮𑀤𑁆𑀥𑀸,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(24.)

3069\. _𑀢𑀸𑀳𑀁 𑀥𑀸𑀢𑀼𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑁄;_  
_𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁂 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀁𑀯 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(25.)

3070\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀥𑀸𑀢𑀼𑀁 𑀧𑀽𑀚𑀬𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀥𑀸𑀢𑀼𑀧𑀝𑁆𑀞𑀳𑀦𑁂 𑀨𑀮𑀁𑁇_  


(26.)

3071\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3072\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑀸𑀢𑀼𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3073\. 𑀥𑀸𑀢𑀼𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
