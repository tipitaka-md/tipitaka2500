# 𑁨𑁯𑁇𑁬 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(24.)

3320. _“𑀭𑀣𑀺𑀬𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀁 𑀫𑀳𑀸𑀦𑀸𑀕𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(25.)

3321. _𑀧𑀓𑁆𑀓𑁄𑀲𑀸𑀧𑀺𑀬 𑀢𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑁄;_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁄,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(26.)

3322. _𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀲𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀉𑀝𑁆𑀞𑀸𑀬 𑀆𑀲𑀦𑀸 𑀢𑀫𑁆𑀳𑀸,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀧𑀸𑀘𑀺𑀦𑀸𑀫𑀼𑀔𑁄𑁇_  


(27.)

3323. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀫𑀤𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(28.)

3324. _𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑀮𑀲𑁂𑀦𑀲𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(29.)

3325. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3326. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3327. 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
