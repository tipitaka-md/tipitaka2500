# 𑁨𑁯𑁇𑁧 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3287\. _“𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀺,_  
_𑀧𑀡𑁆𑀡𑀪𑁄𑀚𑀦𑀪𑁄𑀚𑀦𑁄;_  
_𑀉𑀧𑀯𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀫𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀉𑀧𑀸𑀕𑀘𑁆𑀙𑀺 𑀫𑀳𑀸𑀇𑀲𑀺𑁇_  


(2.)

3288\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑁄;_  
_𑀢𑀲𑁆𑀲 𑀧𑀡𑁆𑀡𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂𑁇_  


(3.)

3289\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀡𑁆𑀡𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀡𑁆𑀡𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(4.)

3290\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3291\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3292\. 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
