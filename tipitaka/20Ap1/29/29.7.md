# 𑁨𑁯𑁇𑁭 𑀅𑀧𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀧𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(30.)

3328\. _“𑀅𑀧𑀤𑀸𑀦𑀁 𑀲𑀼𑀕𑀢𑀸𑀦𑀁,_  
_𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀁𑀳𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀁;_  
_𑀧𑀸𑀤𑁂 𑀘 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(31.)

3329\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀧𑀤𑀸𑀦𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑀺𑀢𑁆𑀢𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(32.)

3330\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3331\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀧𑀤𑀸𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3332\. 𑀅𑀧𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
