# 𑁨𑁯𑁇𑁨 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(5.)

3293. _“𑀲𑀺𑀦𑁂𑀭𑀼𑀲𑀫𑀲𑀦𑁆𑀢𑁄𑀲𑁄,_  
_𑀥𑀭𑀡𑀻𑀲𑀫𑀲𑀸𑀤𑀺𑀲𑁄;_  
_𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸,_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀬 𑀫𑀫𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄𑁇_  


(6.)

3294. _𑀳𑀭𑀻𑀢𑀓𑀁 𑀆𑀫𑀮𑀓𑀁,_  
_𑀅𑀫𑁆𑀩𑀚𑀫𑁆𑀩𑀼𑀯𑀺𑀪𑀻𑀢𑀓𑀁;_  
_𑀓𑁄𑀮𑀁 𑀪𑀮𑁆𑀮𑀸𑀢𑀓𑀁 𑀩𑀺𑀮𑁆𑀮𑀁,_  
_𑀨𑀸𑀭𑀼𑀲𑀓𑀨𑀮𑀸𑀦𑀺 𑀘𑁇_  


(7.)

3295. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄;_  
_𑀢𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(8.)

3296. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(9.)

3297. _𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑀚𑁆𑀛𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(10.)

3298. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3299. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3300. 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
