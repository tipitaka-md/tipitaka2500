# 𑁨𑁯𑁇𑁮 𑀲𑀢𑁆𑀢𑀸𑀳𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀢𑁆𑀢𑀸𑀳𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(33.)

3333\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀁𑀖𑁄 𑀲𑀓𑁆𑀓𑀢𑀫𑀸𑀦𑀺𑀢𑁄;_  
_𑀩𑁆𑀬𑀲𑀦𑀁 𑀫𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀜𑀸𑀢𑀺𑀪𑁂𑀤𑁄 𑀧𑀼𑀭𑁂 𑀅𑀳𑀼𑁇_  


(34.)

3334\. _𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑁆𑀬𑀲𑀦𑀼𑀧𑀲𑀫𑀸𑀬𑀳𑀁;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀸𑀪𑀺𑀭𑀢𑁄 𑀢𑀢𑁆𑀣,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀫𑁆𑀬𑀢𑀸𑁇_  


(35.)

3335\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀫𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(36.)

3336\. _𑀲𑀢𑁆𑀢𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑁆𑀢 𑀆𑀲𑀼𑀁 𑀫𑀳𑀻𑀧𑀢𑀻;_  
_𑀲𑀼𑀦𑀺𑀓𑁆𑀔𑀫𑀸𑀢𑀺 𑀜𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(37.)

3337\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3338\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3339\. 𑀲𑀢𑁆𑀢𑀸𑀳𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
