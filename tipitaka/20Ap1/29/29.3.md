# 𑁨𑁯𑁇𑁩 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(11.)

3301\. _“𑀲𑀻𑀳𑀁 𑀬𑀣𑀸 𑀯𑀦𑀘𑀭𑀁,_  
_𑀦𑀺𑀲𑀪𑀸𑀚𑀸𑀦𑀺𑀬𑀁 𑀬𑀣𑀸;_  
_𑀓𑀓𑀼𑀥𑀁 𑀯𑀺𑀮𑀲𑀦𑁆𑀢𑀁𑀯,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(12.)

3302\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑀁,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀁;_  
_𑀅𑀓𑀸𑀲𑀺𑀁 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(13.)

3303\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑁂 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(14.)

3304\. _𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀚𑀦𑀸𑀥𑀺𑀧𑁄;_  
_𑀲𑀧𑀭𑀺𑀯𑀸𑀭𑁄𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(15.)

3305\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3306\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3307\. 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
