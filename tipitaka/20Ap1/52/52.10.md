# 𑁫𑁨𑁇𑁧𑁦 𑀧𑀤𑀼𑀫𑀥𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀨𑀮𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀤𑀼𑀫𑀥𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(78.)

6786. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀭𑁄𑀫𑀲𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀩𑀼𑀤𑁆𑀥𑁄𑀧𑀺 𑀲𑀫𑁆𑀪𑀯𑁄 𑀦𑀸𑀫,_  
_𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀯𑀲𑀻 𑀢𑀤𑀸𑁇_  


(79.)

6787. _𑀪𑀯𑀦𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀤𑀼𑀫𑀁 𑀥𑀸𑀭𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀏𑀓𑀸𑀳𑀁 𑀥𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀪𑀯𑀦𑀁 𑀧𑀼𑀦𑀭𑀸𑀕𑀫𑀺𑀁𑁇_  


(80.)

6788. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(81.)

6789. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(82.)

6790. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(83.)

6791. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁫𑁪𑁪𑁯)_  


6792. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀤𑀼𑀫𑀥𑀸𑀭𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6793. 𑀧𑀤𑀼𑀫𑀥𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

6794. 𑀨𑀮𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕𑁄 𑀤𑁆𑀯𑁂𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁄𑁇



---

6795. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



6796. _𑀓𑀼𑀭𑀜𑁆𑀘𑀺𑀬𑀁 𑀓𑀧𑀺𑀢𑁆𑀣𑀜𑁆𑀘,_  
_𑀓𑁄𑀲𑀫𑁆𑀩𑀫𑀣 𑀓𑁂𑀢𑀓𑀁;_  
_𑀦𑀸𑀕𑀧𑀼𑀧𑁆𑀨𑀚𑁆𑀚𑀼𑀦𑀜𑁆𑀘𑁂𑀯,_  
_𑀓𑀼𑀝𑀚𑀻 𑀖𑁄𑀲𑀲𑀜𑁆𑀜𑀓𑁄𑁇_  


6797. _𑀣𑁂𑀭𑁄 𑀘 𑀲𑀩𑁆𑀩𑀨𑀮𑀤𑁄,_  
_𑀢𑀣𑀸 𑀧𑀤𑀼𑀫𑀥𑀸𑀭𑀺𑀓𑁄;_  
_𑀅𑀲𑀻𑀢𑀺 𑀘𑁂𑀢𑁆𑀣 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀕𑀸𑀣𑀸 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_
