# 𑁪𑁬𑁇𑁭 𑀏𑀓𑀙𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀙𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(37.)

5927\. _“𑀅𑀗𑁆𑀕𑀸𑀭𑀚𑀸𑀢𑀸 𑀧𑀣𑀯𑀻,_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀍𑀆𑀦𑀼𑀕𑀢𑀸 𑀫𑀳𑀻;_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀪𑀕𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀫𑁆𑀳𑀺 𑀘𑀗𑁆𑀓𑀫𑀺𑁇_  


(38.)

5928\. _𑀧𑀡𑁆𑀟𑀭𑀁 𑀙𑀢𑁆𑀢𑀫𑀸𑀤𑀸𑀬,_  
_𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀳𑀁;_  
_𑀢𑀢𑁆𑀣 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀯𑀺𑀢𑁆𑀢𑀺 𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀣𑁇_  


(39.)

5929\. _𑀫𑀭𑀻𑀘𑀺𑀬𑁄𑀢𑁆𑀣𑀝𑀸 𑀪𑀽𑀫𑀺,_  
_𑀅𑀗𑁆𑀕𑀸𑀭𑀸𑀯 𑀫𑀳𑀻 𑀅𑀬𑀁;_  
_𑀉𑀧𑀳𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀯𑀸𑀢𑀸,_  
_𑀲𑀭𑀻𑀭𑀲𑁆𑀲𑀸𑀲𑀼𑀔𑁂𑀧𑀦𑀸𑁇_  


(40.)

5930\. _𑀲𑀻𑀢𑀁 𑀉𑀡𑁆𑀳𑀁 𑀯𑀺𑀳𑀦𑀦𑁆𑀢𑀁,_  
_𑀯𑀸𑀢𑀸𑀢𑀧𑀦𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳 𑀇𑀫𑀁 𑀙𑀢𑁆𑀢𑀁,_  
_𑀨𑀲𑁆𑀲𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀺𑀁𑁇_  


(41.)

5931\. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀺 𑀢𑀤𑀸 𑀚𑀺𑀦𑁄𑁇_  


(42.)

5932\. _𑀢𑀺𑀁𑀲 𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(43.)

5933\. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀲𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀓𑀢𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


(44.)

5934\. _𑀅𑀬𑀁 𑀫𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀸𑀢𑀺,_  
_𑀘𑀭𑀺𑀫𑁄 𑀯𑀢𑁆𑀢𑀢𑁂 𑀪𑀯𑁄;_  
_𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀫𑁂,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀁 𑀥𑀭𑀻𑀬𑀢𑀺𑁇_  


(45.)

5935\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀙𑀢𑁆𑀢𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀙𑀢𑁆𑀢𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(46.)

5936\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(47.)

5937\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(48.)

5938\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5939\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀙𑀢𑁆𑀢𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5940\. 𑀏𑀓𑀙𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
