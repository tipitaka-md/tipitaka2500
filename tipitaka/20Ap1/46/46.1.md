# 𑁪𑁬𑁇𑁧 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

5879\. _“𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀩𑁄𑀥𑀺𑀬𑀸 𑀧𑀸𑀤𑀧𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀚𑀕𑀢𑀺𑀁 𑀓𑀸𑀭𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(2.)

5880\. _𑀤𑀭𑀺𑀢𑁄 𑀧𑀩𑁆𑀩𑀢𑀢𑁄 𑀯𑀸,_  
_𑀭𑀼𑀓𑁆𑀔𑀢𑁄 𑀧𑀢𑀺𑀢𑁄 𑀅𑀳𑀁;_  
_𑀘𑀼𑀢𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀁 𑀯𑀺𑀦𑁆𑀤𑀸𑀫𑀺,_  
_𑀚𑀕𑀢𑀺𑀬𑀸 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(3.)

5881\. _𑀦 𑀫𑁂 𑀘𑁄𑀭𑀸 𑀯𑀺𑀳𑁂𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀫𑀺𑀢𑁆𑀢𑁂𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀫𑀺,_  
_𑀚𑀕𑀢𑀺𑀬𑀸 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(4.)

5882\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀽𑀚𑀺𑀢𑁄 𑀳𑁄𑀫𑀺,_  
_𑀚𑀕𑀢𑀺𑀬𑀸 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(5.)

5883\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀚𑀕𑀢𑀺𑀁 𑀓𑀸𑀭𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀚𑀕𑀢𑀺𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(6.)

5884\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(7.)

5885\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(8.)

5886\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5887\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5888\. 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
