# 𑁪𑁬𑁇𑁧𑁦 𑀕𑀦𑁆𑀥𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀕𑀦𑁆𑀥𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(63.)

5959\. _“𑀘𑀺𑀢𑀸𑀲𑀼 𑀓𑀼𑀭𑀼𑀫𑀸𑀦𑀸𑀲𑀼,_  
_𑀦𑀸𑀦𑀸𑀕𑀦𑁆𑀥𑁂 𑀲𑀫𑀸𑀳𑀝𑁂;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀕𑀦𑁆𑀥𑀫𑀼𑀝𑁆𑀞𑀺𑀫𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(64.)

5960\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀺𑀢𑀓𑀁 𑀬𑀫𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀺𑀢𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(65.)

5961\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(66.)

5962\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(67.)

5963\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁪𑁭𑁭𑁧)_  


5964\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀦𑁆𑀥𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5965\. 𑀕𑀦𑁆𑀥𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

5966\. 𑀚𑀕𑀢𑀺𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕𑁄 𑀙𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁄𑁇



---

5967\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



5968\. _𑀚𑀕𑀢𑀻 𑀫𑁄𑀭𑀳𑀢𑁆𑀣𑀻 𑀘,_  
_𑀆𑀲𑀦𑀻 𑀉𑀓𑁆𑀓𑀥𑀸𑀭𑀓𑁄;_  
_𑀅𑀓𑁆𑀓𑀫𑀺 𑀯𑀦𑀓𑁄𑀭𑀡𑁆𑀟𑀺,_  
_𑀙𑀢𑁆𑀢𑀤𑁄 𑀚𑀸𑀢𑀺𑀧𑀽𑀚𑀓𑁄𑁇_  


5969\. _𑀧𑀝𑁆𑀝𑀺𑀧𑀼𑀧𑁆𑀨𑀻 𑀘 𑀬𑁄 𑀣𑁂𑀭𑁄,_  
_𑀤𑀲𑀫𑁄 𑀕𑀦𑁆𑀥𑀧𑀽𑀚𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀲𑀝𑁆𑀞𑀺 𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_
