# 𑁧𑁫𑁇𑁨 𑀣𑀫𑁆𑀪𑀸𑀭𑁄𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀙𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀣𑀫𑁆𑀪𑀸𑀭𑁄𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(5.)

2284\. _“𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀻𑀦𑀭𑀸𑀲𑀪𑁂;_  
_𑀆𑀭𑁄𑀧𑁂𑀲𑀺𑀁 𑀥𑀚𑀢𑁆𑀣𑀫𑁆𑀪𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂𑁇_  


(6.)

2285\. _𑀦𑀺𑀲𑁆𑀲𑁂𑀡𑀺𑀁 𑀫𑀸𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀣𑀽𑀧𑀲𑁂𑀝𑁆𑀞𑀁 𑀲𑀫𑀸𑀭𑀼𑀳𑀺𑀁;_  
_𑀚𑀸𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀣𑀽𑀧𑀫𑁆𑀳𑀺 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(7.)

2286\. _𑀅𑀳𑁄 𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀳𑁄 𑀥𑀫𑁆𑀫𑀸,_  
_𑀅𑀳𑁄 𑀦𑁄 𑀲𑀢𑁆𑀣𑀼 𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀣𑀽𑀧𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(8.)

2287\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀣𑀽𑀧𑀲𑀻𑀔𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑁄𑀍𑀅𑀲𑀸𑀲𑀺𑀁𑀲𑀼 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(9.)

2288\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2289\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀣𑀫𑁆𑀪𑀸𑀭𑁄𑀧𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2290\. 𑀣𑀫𑁆𑀪𑀸𑀭𑁄𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
