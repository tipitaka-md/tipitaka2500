# 𑁧𑁫𑁇𑁩 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀙𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(10.)

2291\. _“𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀻𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀫𑀼𑀢𑁆𑀢𑀸𑀯𑁂𑀤𑀺𑀫𑀓𑀸𑀲𑀳𑀁𑁇_  


(11.)

2292\. _𑀫𑀡𑀻𑀳𑀺 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀓𑀸𑀲𑀺𑀁 𑀯𑁂𑀤𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀫𑀳𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(12.)

2293\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀫𑀡𑀻 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀆𑀓𑀸𑀲𑁂,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(13.)

2294\. _𑀲𑁄𑀍𑀅𑀲𑁂𑀢𑁄 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀫𑀡𑀺𑀧𑁆𑀧𑀪𑀸𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀲𑀺𑀁𑀲𑀼 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(14.)

2295\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2296\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2297\. 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
