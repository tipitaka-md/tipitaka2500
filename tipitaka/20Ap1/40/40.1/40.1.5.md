# 𑁪𑁦𑁇𑁧𑁇𑁫 𑀤𑀸𑀦𑀧𑀝𑀺𑀬𑀸𑀤𑀦

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀧𑀝𑀺𑀬𑀸𑀤𑀦

(55.)

4352. _𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑁂 𑀲𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁;_  
_‘𑀲𑀼𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀫𑀬𑀸 𑀮𑀤𑁆𑀥𑁄,_  
_𑀫𑀡𑀺 𑀚𑁄𑀢𑀺𑀭𑀲𑁄 𑀬𑀣𑀸𑁇_  


(56.)

4353. _𑀓𑁂𑀦 𑀢𑀁 𑀧𑀽𑀚𑀬𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀅𑀦𑀽𑀧𑀫𑁄;_  
_𑀅𑀢𑀼𑀮𑁄 𑀅𑀲𑀫𑁄 𑀥𑀻𑀭𑁄,_  
_𑀚𑀺𑀦𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇_  


(57.)

4354. _𑀢𑀣𑀸𑀲𑀫𑀲𑀫𑁄 𑀘𑁂𑀯,_  
_𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀅𑀥𑀺𑀓𑀸𑀭𑀜𑁆𑀳𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁 𑀫𑀬𑀸𑁇_  


(58.)

4355. _𑀦𑀸𑀦𑀸𑀧𑀼𑀧𑁆𑀨𑁂 𑀲𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀭𑁄𑀫 𑀧𑀼𑀧𑁆𑀨𑀫𑀡𑁆𑀟𑀧𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁 𑀏𑀢𑀁,_  
_𑀲𑀩𑁆𑀩𑀧𑀽𑀚𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(59.)

4356. _𑀉𑀧𑁆𑀧𑀮𑀁 𑀧𑀤𑀼𑀫𑀁 𑀯𑀸𑀧𑀺,_  
_𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀓𑀁;_  
_𑀘𑀫𑁆𑀧𑀓𑀁 𑀦𑀸𑀕𑀧𑀼𑀧𑁆𑀨𑀜𑁆𑀘,_  
_𑀫𑀡𑁆𑀟𑀧𑀁 𑀓𑀸𑀭𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(60.)

4357. _𑀲𑀢𑀸𑀲𑀦𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀙𑀢𑁆𑀢𑀘𑁆𑀙𑀸𑀬𑀸𑀬 𑀧𑀜𑁆𑀜𑀧𑀺𑀁;_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀆𑀲𑀦𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀅𑀥𑀺𑀓𑀁 𑀲𑀢𑀫𑀕𑁆𑀖𑀢𑀺𑁇_  


(61.)

4358. _𑀲𑀢𑀸𑀲𑀦𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀙𑀢𑁆𑀢𑀘𑁆𑀙𑀸𑀬𑀸𑀬 𑀧𑀜𑁆𑀜𑀧𑀺𑀁;_  
_𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀁,_  
_𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(62.)

4359. _𑀆𑀭𑁄𑀘𑀺𑀢𑀫𑁆𑀳𑀺 𑀓𑀸𑀮𑀫𑁆𑀳𑀺,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀯𑀲𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀫𑀼𑀧𑁂𑀲𑀺 𑀫𑁂𑁇_  


(63.)

4360. _𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀁 𑀉𑀧𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀧𑀼𑀧𑁆𑀨𑀫𑀡𑁆𑀟𑀧𑁂;_  
_𑀯𑀲𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(64.)

4361. _‘𑀙𑀢𑁆𑀢𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑀸𑀲𑀦𑀁;_  
_𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀦𑀯𑀚𑁆𑀚𑀜𑁆𑀘,_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳𑀸𑀳𑀺 𑀘𑀓𑁆𑀔𑀼𑀫’𑁇_  


(65.)

4362. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀫𑀫𑀁 𑀢𑀸𑀭𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_
