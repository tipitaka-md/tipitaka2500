# 𑁪𑁦𑁇𑁧𑁇𑁧 𑀘𑀺𑀦𑁆𑀢𑀦𑀸𑀓𑀸𑀭

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀘𑀺𑀦𑁆𑀢𑀦𑀸𑀓𑀸𑀭

(3.)

4300\. _‘𑀩𑀳𑀽 𑀫𑁂𑀥𑀺𑀕𑀢𑀸 𑀪𑁄𑀕𑀸,_  
_𑀨𑀻𑀢𑀁 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀫𑀫;_  
_𑀭𑀸𑀚𑀸𑀧𑀺 𑀲𑀦𑁆𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀲𑀺,_  
_𑀆𑀦𑀦𑁆𑀤𑁄 𑀧𑀣𑀯𑀺𑀲𑁆𑀲𑀭𑁄𑁇_  


(4.)

4301\. _𑀅𑀬𑀜𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀅𑀥𑀺𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀫𑀼𑀦𑀺;_  
_𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀫𑁂 𑀪𑁄𑀕𑀸,_  
_𑀤𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(5.)

4302\. _𑀧𑀤𑀼𑀫𑁂𑀦 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁂𑀦,_  
_𑀤𑀺𑀦𑁆𑀦𑀁 𑀤𑀸𑀦𑀯𑀭𑀁 𑀚𑀺𑀦𑁂;_  
_𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁂 𑀘 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂,_  
_𑀅𑀧𑀲𑁆𑀲𑁂𑀦𑀜𑁆𑀘𑀦𑀧𑁆𑀧𑀓𑀁𑁇_  


(6.)

4303\. _𑀅𑀳𑀫𑁆𑀧𑀺 𑀤𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀅𑀤𑀺𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀫𑀜𑁆𑀜𑁂𑀲𑀁,_  
_𑀪𑀯𑀺𑀲𑁆𑀲𑀁 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑁄𑁇_  


(7.)

4304\. _𑀘𑀺𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸𑀳𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑀁,_  
_𑀬𑀸𑀕𑁂 𑀬𑀲𑁆𑀲 𑀲𑀼𑀔𑀁𑀨𑀮𑀁;_  
_𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀤𑀸𑀦𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺𑀁,_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀧𑀽𑀭𑀡𑀁𑁇_  


(8.)

4305\. _𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀺 𑀤𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀅𑀤𑀺𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀫𑀜𑁆𑀜𑁂𑀲𑀁,_  
_𑀪𑀯𑀺𑀲𑁆𑀲𑀁 𑀆𑀤𑀺𑀓𑀫𑁆𑀫𑀺𑀓𑁄’𑁇_
