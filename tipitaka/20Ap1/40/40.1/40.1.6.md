# 𑁪𑁦𑁇𑁧𑁇𑁬 𑀤𑀸𑀦𑀓𑀣𑀸

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀓𑀣𑀸

(66.)

4363\. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲,_  
_𑀧𑀘𑁆𑀘𑁂𑀓𑀁 𑀧𑀢𑁆𑀢𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀚𑀳𑀺𑀁𑀲𑀼 𑀲𑀼𑀫𑁆𑀪𑀓𑀁 𑀧𑀢𑁆𑀢𑀁,_  
_𑀮𑁄𑀳𑀧𑀢𑁆𑀢𑀁 𑀅𑀥𑀸𑀭𑀬𑀼𑀁𑁇_  


(67.)

4364\. _𑀲𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀼𑀧𑁆𑀨𑀫𑀡𑁆𑀟𑀧𑁂;_  
_𑀩𑁄𑀥𑀬𑀦𑁆𑀢𑁄 𑀩𑀳𑀽 𑀲𑀢𑁆𑀢𑁂,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀬𑀺𑁇_  


(68.)

4365\. _𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀳𑁂𑀝𑁆𑀞𑀢𑁄 𑀧𑀼𑀧𑁆𑀨𑀫𑀡𑁆𑀟𑀧𑁂;_  
_𑀘𑀼𑀮𑁆𑀮𑀸𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀅𑀳𑀼𑁇_  


(69.)

4366\. _𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀧𑀢𑁆𑀢𑁂,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀙𑀢𑁆𑀢𑀘𑁆𑀙𑀸𑀬𑀸𑀬𑀫𑀸𑀲𑀻𑀦𑁄,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_
