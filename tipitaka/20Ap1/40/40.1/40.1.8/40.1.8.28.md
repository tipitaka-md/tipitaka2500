# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁨𑁮 𑀤𑀻𑀧𑀥𑀸𑀭𑀸𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀤𑀻𑀧𑀥𑀸𑀭𑀸𑀦𑀺𑀲𑀁𑀲

(160.)

4457. _𑀤𑀻𑀧𑀥𑀸𑀭𑁂 𑀚𑀺𑀦𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀻𑀡𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(161.)

4458. _𑀚𑀸𑀢𑀺𑀫𑀸 𑀅𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀧𑀜𑁆𑀜𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑀫𑁆𑀫𑀢𑁄;_  
_𑀇𑀫𑁂 𑀕𑀼𑀡𑁂 𑀧𑀝𑀺𑀮𑀪𑁂,_  
_𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀤𑀢𑁄 𑀅𑀳𑀁𑁇_
