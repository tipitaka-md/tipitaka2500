# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁩𑁩 𑀦𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺𑀲𑀁𑀲

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

* 𑀦𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺𑀲𑀁𑀲

(170.)

4467\. _𑀦𑀢𑁆𑀣𑀼𑀓𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(171.)

4468\. _𑀲𑀤𑁆𑀥𑀁 𑀲𑀻𑀮𑀁 𑀳𑀺𑀭𑀺𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀅𑀣 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀺𑀬𑀁 𑀕𑀼𑀡𑀁;_  
_𑀲𑀼𑀢𑀁 𑀘𑀸𑀕𑀜𑁆𑀘 𑀔𑀦𑁆𑀢𑀺𑀜𑁆𑀘,_  
_𑀧𑀜𑁆𑀜𑀁 𑀫𑁂 𑀅𑀝𑁆𑀞𑀫𑀁 𑀕𑀼𑀡𑀁𑁇_
