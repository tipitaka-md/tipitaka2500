# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁮 𑀯𑀺𑀥𑀽𑀧𑀦𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀸𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀯𑀺𑀥𑀽𑀧𑀦𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀸𑀦𑀺𑀲𑀁𑀲

(114.)

4411\. _𑀯𑀺𑀥𑀽𑀧𑀦𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑁂 𑀘 𑀲𑁄𑀪𑀦𑁂;_  
_𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(115.)

4412\. _𑀲𑀻𑀢𑀁 𑀉𑀡𑁆𑀳𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀤𑀭𑀣𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑀦𑁆𑀢𑀸𑀧𑀦𑀁 𑀫𑀫𑁇_  


(116.)

4413\. _𑀭𑀸𑀕𑀕𑁆𑀕𑀺 𑀤𑁄𑀲𑀫𑁄𑀳𑀕𑁆𑀕𑀺,_  
_𑀫𑀸𑀦𑀕𑁆𑀕𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀅𑀕𑁆𑀕𑀺 𑀘;_  
_𑀲𑀩𑁆𑀩𑀕𑁆𑀕𑀻 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀤𑀢𑁄 𑀫𑀫𑁇_
