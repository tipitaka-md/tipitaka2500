# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁩𑁮 𑀧𑀸𑀤𑀧𑀻𑀞𑀸𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀧𑀸𑀤𑀧𑀻𑀞𑀸𑀦𑀺𑀲𑀁𑀲

(189.)

4486. _𑀧𑀸𑀤𑀧𑀻𑀞𑁂 𑀚𑀺𑀦𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀤𑁆𑀯𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫;_  
_𑀮𑀪𑀸𑀫𑀺 𑀩𑀳𑀼𑀓𑁂 𑀬𑀸𑀦𑁂,_  
_𑀧𑀸𑀤𑀧𑀻𑀞𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(190.)

4487. _𑀤𑀸𑀲𑀻 𑀤𑀸𑀲𑀸 𑀘 𑀪𑀭𑀺𑀬𑀸,_  
_𑀬𑁂 𑀘𑀜𑁆𑀜𑁂 𑀅𑀦𑀼𑀚𑀻𑀯𑀺𑀦𑁄;_  
_𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀘𑀭𑀦𑁆𑀢𑁂 𑀫𑀁,_  
_𑀧𑀸𑀤𑀧𑀻𑀞𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_
