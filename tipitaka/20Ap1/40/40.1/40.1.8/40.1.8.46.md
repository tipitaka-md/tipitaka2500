# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁪𑁬 𑀥𑀽𑀧𑀸𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀥𑀽𑀧𑀸𑀦𑀺𑀲𑀁𑀲

(202.)

4499\. _𑀥𑀽𑀧𑀁 𑀤𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁂,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀤𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(203.)

4500\. _𑀲𑀼𑀕𑀦𑁆𑀥𑀤𑁂𑀳𑁄 𑀬𑀲𑀯𑀸,_  
_𑀲𑀻𑀖𑀧𑀜𑁆𑀜𑁄 𑀘 𑀓𑀺𑀢𑁆𑀢𑀺𑀫𑀸;_  
_𑀢𑀺𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑁄 𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑁄,_  
_𑀳𑀸𑀲𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀜𑀯𑀸𑁇_  


(204.)

4501\. _𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀯𑀸𑀳𑀲𑀸 𑀤𑀸𑀦𑀺,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀲𑀼𑀔𑀁 𑀲𑀺𑀯𑀁𑁇_
