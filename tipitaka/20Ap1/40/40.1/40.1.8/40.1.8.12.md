# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁧𑁨 𑀲𑀽𑀘𑀺𑀖𑀭𑀸𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀲𑀽𑀘𑀺𑀖𑀭𑀸𑀦𑀺𑀲𑀁𑀲

(123.)

4420. _𑀲𑀽𑀘𑀺𑀖𑀭𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀻𑀡𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(124.)

4421. _𑀘𑁂𑀢𑁄𑀲𑀼𑀔𑀁 𑀓𑀸𑀬𑀲𑀼𑀔𑀁,_  
_𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀚𑀁 𑀲𑀼𑀔𑀁;_  
_𑀇𑀫𑁂 𑀕𑀼𑀡𑁂 𑀧𑀝𑀺𑀮𑀪𑁂,_  
_𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀤𑀢𑁄 𑀅𑀳𑀁𑁇_
