# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁨𑁯 𑀢𑀼𑀫𑁆𑀩𑀓-𑀓𑀭𑀡𑁆𑀟

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

* 𑀢𑀼𑀫𑁆𑀩𑀓-𑀓𑀭𑀡𑁆𑀟

(162.)

4459\. _𑀢𑀼𑀫𑁆𑀩𑀓𑁂 𑀘 𑀓𑀭𑀡𑁆𑀟𑁂 𑀘,_  
_𑀤𑀢𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁂 𑀕𑀡𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀤𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(163.)

4460\. _𑀲𑀼𑀕𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀔𑀲𑀫𑀗𑁆𑀕𑀻,_  
_𑀫𑀳𑀸𑀬𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀺;_  
_𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀯𑀺𑀕𑀢𑁄 𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑁄,_  
_𑀲𑀩𑁆𑀩𑀻𑀢𑀺𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀺𑀢𑁄𑁇_  


(164.)

4461\. _𑀯𑀺𑀧𑀼𑀮𑁂 𑀘 𑀕𑀼𑀡𑁂 𑀮𑀸𑀪𑀻,_  
_𑀲𑀫𑀸𑀯 𑀘𑀮𑀦𑀸 𑀫𑀫;_  
_𑀲𑀼𑀯𑀺𑀯𑀚𑁆𑀚𑀺𑀢𑁅𑀩𑁆𑀩𑁂𑀕𑁄,_  
_𑀢𑀼𑀫𑁆𑀩𑀓𑁂 𑀘 𑀓𑀭𑀡𑁆𑀟𑀓𑁂𑁇_  


(165.)

4462\. _𑀮𑀪𑀸𑀫𑀺 𑀘𑀢𑀼𑀭𑁄 𑀯𑀡𑁆𑀡𑁂,_  
_𑀳𑀢𑁆𑀣𑀺𑀲𑁆𑀲𑀭𑀢𑀦𑀸𑀦𑀺 𑀘;_  
_𑀢𑀸𑀦𑀺 𑀫𑁂 𑀦 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀼𑀫𑁆𑀩𑀤𑀸𑀦𑁂 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_
