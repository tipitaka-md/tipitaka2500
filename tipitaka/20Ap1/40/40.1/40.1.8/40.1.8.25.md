# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁨𑁫 𑀓𑀼𑀜𑁆𑀘𑀺𑀓𑀸𑀖𑀭𑀸𑀦𑀺𑀲𑀁𑀲

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

* 𑀓𑀼𑀜𑁆𑀘𑀺𑀓𑀸𑀖𑀭𑀸𑀦𑀺𑀲𑀁𑀲

(155.)

4452\. _𑀓𑀼𑀜𑁆𑀘𑀺𑀓𑀸𑀦𑀁 𑀖𑀭𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀲𑀁𑀖𑁂 𑀕𑀡𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀤𑁆𑀯𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫;_  
_𑀅𑀧𑁆𑀧𑀓𑁄𑀥𑁄 𑀅𑀦𑀸𑀬𑀸𑀲𑁄,_  
_𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄 𑀪𑀯𑁂 𑀅𑀳𑀁𑁇_
