# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁧𑁧 𑀢𑁂𑀮𑀥𑀸𑀭𑀸𑀦𑀺𑀲𑀁𑀲

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

* 𑀢𑁂𑀮𑀥𑀸𑀭𑀸𑀦𑀺𑀲𑀁𑀲

(121.)

4418\. _𑀢𑁂𑀮𑀥𑀸𑀭𑁂 𑀲𑀼𑀕𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(122.)

4419\. _𑀲𑀼𑀘𑀸𑀭𑀼𑀭𑀽𑀧𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑁄,_  
_𑀲𑀼𑀲𑀫𑀼𑀕𑁆𑀕𑀢𑀫𑀸𑀦𑀲𑁄;_  
_𑀅𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀭𑀓𑁆𑀔𑁂𑀳𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑁄𑁇_
