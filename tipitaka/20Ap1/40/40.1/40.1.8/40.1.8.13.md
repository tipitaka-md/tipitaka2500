# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁧𑁩 𑀅𑀁𑀲𑀩𑀤𑁆𑀥𑀸𑀦𑀺𑀲𑀁𑀲

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

* 𑀅𑀁𑀲𑀩𑀤𑁆𑀥𑀸𑀦𑀺𑀲𑀁𑀲

(125.)

4422\. _𑀅𑀁𑀲𑀩𑀤𑁆𑀥𑁂 𑀚𑀺𑀦𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀻𑀡𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(126.)

4423\. _𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂 𑀕𑀸𑀥𑀁 𑀯𑀺𑀦𑁆𑀤𑀸𑀫𑀺,_  
_𑀲𑀭𑀸𑀫𑀺 𑀤𑀼𑀢𑀺𑀬𑀁 𑀪𑀯𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀘𑁆𑀙𑀯𑀻 𑀳𑁄𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀤𑀢𑁄 𑀅𑀳𑀁𑁇_
