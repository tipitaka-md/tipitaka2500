# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁬 𑀲𑀽𑀘𑀺𑀆𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀲𑀽𑀘𑀺𑀆𑀦𑀺𑀲𑀁𑀲

(109.)

4406\. _𑀲𑀽𑀘𑀺𑀬𑁄 𑀲𑀼𑀕𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(110.)

4407\. _𑀦 𑀲𑀁𑀲𑀬𑁄 𑀓𑀗𑁆𑀔𑀘𑁆𑀙𑁂𑀤𑁄,_  
_𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀘 𑀪𑁄𑀕𑀯𑀸;_  
_𑀢𑀺𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑁄 𑀲𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄 𑀪𑀯𑀸𑀪𑀯𑁂𑁇_  


(111.)

4408\. _𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀦𑀺𑀧𑀼𑀡𑀁 𑀞𑀸𑀦𑀁,_  
_𑀅𑀢𑁆𑀣𑀁 𑀜𑀸𑀡𑁂𑀦 𑀧𑀲𑁆𑀲𑀬𑀺𑀁;_  
_𑀯𑀚𑀺𑀭𑀕𑁆𑀕𑀲𑀫𑀁 𑀜𑀸𑀡𑀁,_  
_𑀳𑁄𑀢𑀺 𑀫𑁂 𑀢𑀫𑀖𑀸𑀢𑀦𑀁𑁇_
