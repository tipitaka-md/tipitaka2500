# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁪 𑀯𑀸𑀲𑀺𑀆𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀯𑀸𑀲𑀺𑀆𑀦𑀺𑀲𑀁𑀲

(103.)

4400\. _𑀘𑀺𑀢𑁆𑀢𑀩𑀦𑁆𑀥𑀦𑀲𑀫𑁆𑀩𑀤𑁆𑀥𑀸,_  
_𑀲𑀼𑀓𑀢𑀸 𑀯𑀸𑀲𑀺𑀬𑁄 𑀩𑀳𑀽;_  
_𑀤𑀢𑁆𑀯𑀸𑀦 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀘 𑀢𑀣𑁂𑀯𑀳𑀁𑁇_  


(104.)

4401\. _𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫;_  
_𑀲𑀽𑀭𑁄 𑀳𑁄𑀫𑀯𑀺𑀲𑀸𑀭𑀻 𑀘,_  
_𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑁂𑀲𑀼 𑀧𑀸𑀭𑀫𑀻𑁇_  


(105.)

4402\. _𑀥𑀺𑀢𑀺𑀯𑀻𑀭𑀺𑀬𑀯𑀸 𑀳𑁄𑀫𑀺,_  
_𑀧𑀕𑁆𑀕𑀳𑀻𑀢𑀫𑀦𑁄 𑀲𑀤𑀸;_  
_𑀓𑀺𑀮𑁂𑀲𑀘𑁆𑀙𑁂𑀤𑀦𑀁 𑀜𑀸𑀡𑀁,_  
_𑀲𑀼𑀔𑀼𑀫𑀁 𑀅𑀢𑀼𑀮𑀁 𑀲𑀼𑀘𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀝𑀺𑀮𑀪𑀸𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀤𑀢𑁄 𑀅𑀳𑀁𑁇_
