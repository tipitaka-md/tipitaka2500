# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁧𑁯 𑀉𑀧𑀸𑀳𑀦𑀸𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀉𑀧𑀸𑀳𑀦𑀸𑀦𑀺𑀲𑀁𑀲

(142.)

4439\. _𑀉𑀧𑀸𑀳𑀦𑁂 𑀚𑀺𑀦𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀻𑀡𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁂 𑀫𑀫𑁇_  


(143.)

4440\. _𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀬𑀸𑀦𑀁,_  
_𑀲𑀺𑀯𑀺𑀓𑀸 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀺𑀓𑀸;_  
_𑀲𑀝𑁆𑀞𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(144.)

4441\. _𑀫𑀡𑀺𑀫𑀬𑀸 𑀢𑀫𑁆𑀩𑀫𑀬𑀸,_  
_𑀲𑁄𑀡𑁆𑀡𑀭𑀚𑀢𑀧𑀸𑀤𑀼𑀓𑀸;_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀧𑀤𑀼𑀤𑁆𑀥𑀸𑀭𑁂,_  
_𑀪𑀯𑁂 𑀲𑀁𑀲𑀭𑀢𑁄 𑀫𑀫𑁇_  


(145.)

4442\. _𑀦𑀺𑀬𑀸𑀫𑀁 𑀲𑀢𑀺 𑀥𑀸𑀯𑀦𑁆𑀢𑀺,_  
_𑀆𑀕𑀼𑀆𑀘𑀸𑀭𑀲𑁄𑀥𑀦𑀁;_  
_𑀇𑀫𑁂 𑀕𑀼𑀡𑁂 𑀧𑀝𑀺𑀮𑀪𑁂,_  
_𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀤𑀢𑁄 𑀅𑀳𑀁𑁇_
