# 𑁪𑁦𑁇𑁧𑁇𑁮𑁇𑁧𑁫 𑀆𑀥𑀸𑀭𑀓𑀸𑀦𑀺𑀲𑀁𑀲

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀆𑀥𑀸𑀭𑀓𑀸𑀦𑀺𑀲𑀁𑀲

(130.)

4427\. _𑀆𑀥𑀸𑀭𑀓𑁂 𑀚𑀺𑀦𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀖𑁂 𑀕𑀡𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀜𑁆𑀘𑀯𑀡𑁆𑀡𑁂𑀳𑀺 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀅𑀘𑀮𑁄 𑀳𑁄𑀫𑀺 𑀓𑁂𑀦𑀘𑀺𑁇_  


(131.)

4428\. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑁂 𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀲𑀢𑀺𑀜𑀸𑀡𑀧𑁆𑀧𑀩𑁄𑀥𑀦𑀸;_  
_𑀥𑀢𑀸 𑀫𑁂 𑀦 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀪𑀯𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀸𑁇_
