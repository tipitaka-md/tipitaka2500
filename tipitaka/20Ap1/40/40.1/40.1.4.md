# 𑁪𑁦𑁇𑁧𑁇𑁪 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀓𑀣𑀸

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀓𑀣𑀸

(50.)

4347\. _𑀅𑀓𑁆𑀔𑀤𑀲𑁆𑀲𑁂𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀆𑀦𑀦𑁆𑀤𑀜𑁆𑀘𑀸𑀧𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀁;_  
_𑀢𑀼𑀝𑁆𑀞𑁄 𑀧𑀫𑀼𑀤𑀺𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀫𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑁇_  


(51.)

4348\. _𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑀫𑀦𑀸𑀲𑀯𑀁;_  
_𑀲𑀺𑀭𑀲𑀸 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(52.)

4349\. _‘𑀯𑀲𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀳𑀺 𑀘𑀓𑁆𑀔𑀼𑀫;_  
_𑀳𑀸𑀲𑀬𑀦𑁆𑀢𑁄 𑀫𑀫 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀫𑀼𑀧𑁂𑀳𑀺 𑀫𑁂’𑁇_  


(53.)

4350\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(54.)

4351\. _𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀪𑀺𑀯𑀸𑀤𑀺𑀬 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀳𑀝𑁆𑀞𑁄 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄𑀳𑀁,_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_
