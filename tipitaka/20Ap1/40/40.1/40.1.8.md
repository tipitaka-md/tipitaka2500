# 𑁪𑁦𑁇𑁧𑁇𑁮 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

(86.)

4383\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀨𑀮𑀁 𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀫𑁂 𑀇𑀥;_  
_𑀲𑀼𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑀭𑀯𑁂𑀕𑁄𑀯,_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀬𑀻 𑀫𑀫𑁇_  


### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

(87.)

4384\. _𑀅𑀳𑁄 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂;_  
_𑀬𑀢𑁆𑀣 𑀓𑀸𑀭𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁𑁇_  


### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑀣𑀸

(88.)

4385\. _𑀅𑀦𑀽𑀦𑀓𑀁 𑀤𑀸𑀦𑀯𑀭𑀁,_  
_𑀅𑀤𑀸𑀲𑀺 𑀬𑁄 𑀳𑀺 𑀫𑀸𑀡𑀯𑁄;_  
_𑀆𑀤𑀺𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑁄 𑀆𑀲𑀺,_  
_𑀢𑀲𑁆𑀲 𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_
