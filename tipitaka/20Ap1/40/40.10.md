# 𑁪𑁦𑁇𑁧𑁦 𑀅𑀚𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

* 𑀅𑀚𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(484.)

4800\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀁,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(485.)

4801\. _𑀦𑀸𑀳𑀁 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀦𑀧𑀺 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑁄𑀫𑀳𑀁;_  
_𑀫𑀫 𑀪𑀓𑁆𑀔𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀆𑀳𑀺𑀡𑁆𑀟𑀸𑀫𑀺 𑀯𑀦𑁂 𑀅𑀳𑀁𑁇_  


(486.)

4802\. _𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑁆𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑀁;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀯𑀺𑀢𑁆𑀢𑀺𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀁,_  
_𑀲𑀢𑁆𑀢𑁄 𑀓𑁄 𑀦𑀸𑀫𑀬𑀁 𑀪𑀯𑁂𑁇_  


(487.)

4803\. _𑀮𑀓𑁆𑀔𑀡𑀸𑀦𑀺 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸,_  
_𑀫𑀫 𑀯𑀺𑀚𑁆𑀚𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀲𑀼𑀢𑀜𑁆𑀳𑀺 𑀫𑁂𑀢𑀁 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑀦𑀁 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁𑁇_  


(488.)

4804\. _𑀢𑁂𑀲𑀁 𑀬𑀣𑀸 𑀢𑀁 𑀯𑀘𑀦𑀁,_  
_𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀁,_  
_𑀕𑀢𑀺𑀁 𑀫𑁂 𑀲𑁄𑀥𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(489.)

4805\. _𑀔𑀺𑀧𑁆𑀧𑀁 𑀅𑀲𑁆𑀲𑀫𑀫𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀫𑀥𑀼𑀢𑁂𑀮𑀁 𑀕𑀳𑀺𑀁 𑀅𑀳𑀁;_  
_𑀓𑁄𑀮𑀫𑁆𑀩𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑀁𑁇_  


(490.)

4806\. _𑀢𑀺𑀤𑀡𑁆𑀟𑀓𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀞𑀧𑁂𑀲𑀳𑀁;_  
_𑀧𑀤𑀻𑀧𑀁 𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀝𑁆𑀞𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀅𑀯𑀦𑁆𑀤𑀳𑀁𑁇_  


(491.)

4807\. _𑀲𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀢𑀢𑁄 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(492.)

4808\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀅𑀳𑀁;_  
_𑀤𑀻𑀧𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(493.)

4809\. _𑀲𑀩𑁆𑀩𑁂 𑀯𑀦𑀸 𑀕𑀦𑁆𑀥𑀫𑀬𑀸,_  
_𑀧𑀩𑁆𑀩𑀢𑁂 𑀕𑀦𑁆𑀥𑀫𑀸𑀤𑀦𑁂;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀆𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀆𑀕𑀘𑁆𑀙𑀼𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(494.)

4810\. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀧𑀼𑀧𑁆𑀨𑀕𑀦𑁆𑀥𑀸𑀲𑁂,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀆𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀼𑀁 𑀢𑀤𑀸𑁇_  


(495.)

4811\. _𑀬𑀸𑀯𑀢𑀸 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀦𑀸𑀕𑀸 𑀘 𑀕𑀭𑀼𑀍𑀆 𑀉𑀪𑁄;_  
_𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑁄𑀢𑀼𑀓𑀸𑀫𑀸 𑀢𑁂,_  
_𑀆𑀕𑀘𑁆𑀙𑀼𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(496.)

4812\. _𑀤𑁂𑀯𑀮𑁄 𑀦𑀸𑀫 𑀲𑀫𑀡𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑁄;_  
_𑀯𑀲𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(497.)

4813\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(498.)

4814\. _‘𑀬𑁄 𑀫𑁂 𑀤𑀻𑀧𑀁 𑀧𑀤𑀻𑀧𑁂𑀲𑀺,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(499.)

4815\. _𑀲𑀝𑁆𑀞𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


---

4816\. 𑀲𑁄𑀍𑀅𑀲𑀫𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇



(500.)

4817\. _𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀣𑀯𑀺𑀬𑀁 𑀲𑀢𑁆𑀢𑀲𑀢𑀁,_  
_𑀯𑀺𑀧𑀼𑀮𑀁 𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(501.)

4818\. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀇𑀫𑀺𑀦𑀸 𑀤𑀻𑀧𑀤𑀸𑀦𑁂𑀦,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(502.)

4819\. _𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀅𑀝𑁆𑀞𑀓𑁄𑀲𑀁,_  
_𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀬𑀁 𑀲𑀤𑀸;_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀘𑀯𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀦𑁆𑀢𑀲𑁆𑀲 𑀚𑀦𑁆𑀢𑀼𑀦𑁄𑁇_  


(503.)

4820\. _𑀤𑀺𑀯𑀸 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀧𑀤𑀻𑀧𑀁 𑀥𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀚𑀸𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄𑁇_  


(504.)

4821\. _𑀬𑀸𑀯𑀢𑀸 𑀦𑀕𑀭𑀁 𑀆𑀲𑀺,_  
_𑀢𑀸𑀯𑀢𑀸 𑀚𑁄𑀢𑀬𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁𑁇_  


(505.)

4822\. _𑀅𑀲𑁆𑀲𑁂𑀯 𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀝𑁆𑀞𑀤𑀻𑀧𑀨𑀮𑁂𑀦 𑀳𑀺;_  
_𑀦 𑀚𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀚𑀦𑁆𑀢𑀽,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(506.)

4823\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(507.)

4824\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄𑁇_  


(508.)

4825\. _𑀢𑁄𑀲𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑁄𑀢𑀫𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁;_  
_𑀅𑀚𑀺𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄’𑁇_  


(509.)

4826\. _𑀲𑀝𑁆𑀞𑀺 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀁 𑀅𑀳𑀁;_  
_𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀫𑁂 𑀤𑀻𑀧𑀲𑀢𑀁,_  
_𑀚𑁄𑀢𑀢𑁂 𑀦𑀺𑀘𑁆𑀘𑀓𑀸𑀮𑀺𑀓𑀁𑁇_  


(510.)

4827\. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸,_  
_𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀧𑀪𑀸 𑀫𑀫;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀲𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀳𑀸𑀲𑀁 𑀚𑀦𑁂𑀲𑀳𑀁𑁇_  


(511.)

4828\. _𑀢𑀼𑀲𑀺𑀢𑀸𑀳𑀁 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀑𑀓𑁆𑀓𑀫𑀺𑀁 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀬𑀁;_  
_𑀚𑀸𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀆𑀮𑁄𑀓𑁄 𑀯𑀺𑀧𑀼𑀮𑁄 𑀅𑀳𑀼𑁇_  


(512.)

4829\. _𑀅𑀕𑀸𑀭𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀩𑀸𑀯𑀭𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀲𑀺𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(513.)

4830\. _𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑁂 𑀯𑀲𑀦𑁆𑀢𑁄𑀳𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑀁𑁇_  


(514.)

4831\. _𑀤𑀦𑁆𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑀫𑁂𑀢𑀸𑀯𑀻,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀓𑀣𑀬𑀻 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀫𑁄𑀘𑀦𑀁𑁇_  


(515.)

4832\. _𑀢𑀁 𑀫𑁂 𑀆𑀕𑀫𑀦𑀁 𑀲𑀺𑀤𑁆𑀥𑀁,_  
_𑀢𑁄𑀲𑀺𑀢𑁄𑀳𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(516.)

4833\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀻𑀧𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(517.)

4834\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(518.)

4835\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(519.)

4836\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁩𑁮𑁮𑁬)_  


4837\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑀺𑀢𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4838\. 𑀅𑀚𑀺𑀢𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇

  
**𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁄𑁇**



---

4839\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



4840\. _𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀲𑁂𑀮𑁄 𑀘,_  
_𑀲𑀩𑁆𑀩𑀓𑀺𑀢𑁆𑀢𑀻 𑀫𑀥𑀼𑀦𑁆𑀤𑀤𑁄;_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀻 𑀩𑀸𑀓𑀼𑀮𑁄 𑀘,_  
_𑀕𑀺𑀭𑀺 𑀲𑀍𑀅𑀮𑀲𑀯𑁆𑀳𑀬𑁄𑁇_  


4841\. _𑀲𑀩𑁆𑀩𑀤𑁄 𑀅𑀚𑀺𑀢𑁄 𑀘𑁂𑀯,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀡𑀺𑀢𑀸 𑀇𑀳;_  
_𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀕𑀸𑀣𑀸𑀦𑀁,_  
_𑀯𑀻𑀲𑀢𑀺 𑀘 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀻𑀢𑀺𑁇_  


---

4842\. 𑀅𑀣 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



4843\. _𑀧𑀤𑀼𑀫𑀸𑀭𑀓𑁆𑀔𑀤𑁄 𑀘𑁂𑀯,_  
_𑀉𑀫𑀸 𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑁂𑀦 𑀘;_  
_𑀏𑀓𑀧𑀤𑁆𑀫 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀻,_  
_𑀫𑀦𑁆𑀤𑀸𑀭𑀁 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀓𑁄𑁇_  


4844\. _𑀅𑀯𑀝𑀜𑁆𑀘 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀺 𑀘,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀡𑀺𑀢𑀸 𑀇𑀳;_  
_𑀘𑀢𑀼𑀲𑀢𑁆𑀢𑀢𑀺 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀏𑀓𑀸𑀤𑀲 𑀲𑀢𑀸𑀦𑀺 𑀘𑁇_  


---

4845\. 𑀧𑀤𑀼𑀫𑀯𑀕𑁆𑀕𑀤𑀲𑀓𑀁𑁇



---

4846\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑀢𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
