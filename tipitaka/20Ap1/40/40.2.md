# 𑁪𑁦𑁇𑁨 𑀲𑁂𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀕𑁆𑀕

### 𑀲𑁂𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(208.)

4507. _“𑀦𑀕𑀭𑁂 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀸,_  
_𑀯𑀻𑀣𑀺𑀲𑀸𑀫𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀫𑀫 𑀜𑀸𑀢𑀻 𑀲𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(209.)

4508. _‘𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀆𑀲𑀺 𑀲𑁄 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄𑁇_  


(210.)

4509. _𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀕𑀫𑀸 𑀘𑁂𑀯,_  
_𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(211.)

4510. _𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(212.)

4511. _𑀉𑀕𑁆𑀕𑀸 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(213.)

4512. _𑀆𑀍𑀆𑀭𑀺𑀓𑀸 𑀓𑀧𑁆𑀧𑀓𑀸 𑀘,_  
_𑀦𑁆𑀳𑀸𑀧𑀓𑀸 𑀫𑀸𑀮𑀓𑀸𑀭𑀓𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(214.)

4513. _𑀭𑀚𑀓𑀸 𑀧𑁂𑀲𑀓𑀸𑀭𑀸 𑀘,_  
_𑀘𑀫𑁆𑀫𑀓𑀸𑀭𑀸 𑀘 𑀦𑁆𑀳𑀸𑀧𑀺𑀢𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(215.)

4514. _𑀉𑀲𑀼𑀓𑀸𑀭𑀸 𑀪𑀫𑀓𑀸𑀭𑀸,_  
_𑀘𑀫𑁆𑀫𑀓𑀸𑀭𑀸 𑀘 𑀢𑀘𑁆𑀙𑀓𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(216.)

4515. _𑀓𑀫𑁆𑀫𑀸𑀭𑀸 𑀲𑁄𑀡𑁆𑀡𑀓𑀸𑀭𑀸 𑀘,_  
_𑀢𑀺𑀧𑀼𑀮𑁄𑀳𑀓𑀭𑀸 𑀢𑀣𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(217.)

4516. _𑀪𑀢𑀓𑀸 𑀘𑁂𑀝𑀓𑀸 𑀘𑁂𑀯,_  
_𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀸 𑀩𑀳𑀽;_  
_𑀬𑀣𑀸𑀲𑀓𑁂𑀦 𑀣𑀸𑀫𑁂𑀦,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(218.)

4517. _𑀉𑀤𑀳𑀸𑀭𑀸 𑀓𑀝𑁆𑀞𑀳𑀸𑀭𑀸,_  
_𑀓𑀲𑁆𑀲𑀓𑀸 𑀢𑀺𑀡𑀳𑀸𑀭𑀓𑀸;_  
_𑀬𑀣𑀸𑀲𑀓𑁂𑀦 𑀣𑀸𑀫𑁂𑀦,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(219.)

4518. _𑀧𑀼𑀧𑁆𑀨𑀺𑀓𑀸 𑀫𑀸𑀮𑀺𑀓𑀸 𑀘𑁂𑀯,_  
_𑀧𑀡𑁆𑀡𑀺𑀓𑀸 𑀨𑀮𑀳𑀸𑀭𑀓𑀸;_  
_𑀬𑀣𑀸𑀲𑀓𑁂𑀦 𑀣𑀸𑀫𑁂𑀦,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(220.)

4519. _𑀕𑀡𑀺𑀓𑀸 𑀓𑀼𑀫𑁆𑀪𑀤𑀸𑀲𑀻 𑀘,_  
_𑀧𑀽𑀯𑀺𑀓𑀸 𑀫𑀘𑁆𑀙𑀺𑀓𑀸𑀧𑀺 𑀘;_  
_𑀬𑀣𑀸𑀲𑀓𑁂𑀦 𑀣𑀸𑀫𑁂𑀦,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(221.)

4520. _𑀏𑀣 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀕𑀡𑀁 𑀩𑀦𑁆𑀥𑀸𑀫 𑀏𑀓𑀢𑁄;_  
_𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂’𑁇_  


(222.)

4521. _𑀢𑁂 𑀫𑁂 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀯𑀘𑀦𑀁,_  
_𑀕𑀡𑀁 𑀩𑀦𑁆𑀥𑀺𑀁𑀲𑀼 𑀢𑀸𑀯𑀤𑁂;_  
_‘𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀁 𑀲𑀼𑀓𑀢𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀸𑀭𑀬𑀼𑀁’𑁇_  


(223.)

4522. _𑀦𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀲𑀸𑀮𑀁,_  
_𑀉𑀤𑀕𑁆𑀕𑁄 𑀢𑀼𑀝𑁆𑀞𑀫𑀸𑀦𑀲𑁄;_  
_𑀧𑀭𑁂𑀢𑁄 𑀢𑁂𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀳𑀺,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀫𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑁇_  


(224.)

4523. _𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(225.)

4524. _‘𑀇𑀫𑁂 𑀢𑀻𑀡𑀺 𑀲𑀢𑀸 𑀯𑀻𑀭,_  
_𑀧𑀼𑀭𑀺𑀲𑀸 𑀏𑀓𑀢𑁄 𑀕𑀡𑀸;_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀁 𑀲𑀼𑀓𑀢𑀁,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀢𑀼𑀯𑀁 𑀫𑀼𑀦𑀺𑁇_  


(226.)

4525. _𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄,_  
_𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺𑀢𑁆𑀯 𑀘𑀓𑁆𑀔𑀼𑀫𑀸’;_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀢𑀸𑀦𑀁 𑀧𑀼𑀭𑀢𑁄,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(227.)

4526. _‘𑀢𑀺𑀲𑀢𑀸𑀧𑀺 𑀘 𑀚𑁂𑀝𑁆𑀞𑁄 𑀘,_  
_𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀼 𑀏𑀓𑀢𑁄;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀜𑁆𑀳𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀩𑁆𑀩𑁂 𑀅𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀣𑁇_  


(228.)

4527. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀲𑀻𑀢𑀺𑀪𑀸𑀯𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀅𑀚𑀭𑀁 𑀅𑀫𑀢𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀨𑀲𑁆𑀲𑀬𑀺𑀲𑁆𑀲𑀣’𑁇_  


(229.)

4528. _𑀏𑀯𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽 𑀲𑀫𑀡𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀯𑁂𑀤𑀬𑀺𑀁𑁇_  


(230.)

4529. _𑀢𑀺𑀁𑀲 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑁂𑀯𑀸𑀥𑀺𑀧𑁄 𑀧𑀜𑁆𑀘𑀲𑀢𑀁,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(231.)

4530. _𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀫𑀳𑀸𑀤𑁂𑀯𑀸 𑀅𑀯𑀦𑁆𑀤𑀺𑀲𑀼𑀁𑁇_  


(232.)

4531. _𑀇𑀥 𑀫𑀸𑀦𑀼𑀲𑀓𑁂 𑀭𑀚𑁆𑀚𑀁,_  
_𑀧𑀭𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀩𑀦𑁆𑀥𑀯𑀸;_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


(233.)

4532. _𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺 𑀦𑀺𑀘𑀬𑁄,_  
_𑀢𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀲𑁂𑀮𑁄 𑀇𑀢𑀺 𑀫𑀫𑀁 𑀦𑀸𑀫𑀁,_  
_𑀙𑀍𑀅𑀗𑁆𑀕𑁂 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑁄𑁇_  


(234.)

4533. _𑀚𑀗𑁆𑀖𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀺𑀘𑀭𑀁,_  
_𑀲𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀚𑀝𑀸𑀪𑀸𑀭𑀺𑀓𑀪𑀭𑀺𑀢𑀁,_  
_𑀓𑁂𑀡𑀺𑀬𑀁 𑀦𑀸𑀫 𑀢𑀸𑀧𑀲𑀁𑁇_  


(235.)

4534. _𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸𑀳𑀼𑀢𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁;_

_𑀧𑀻𑀢𑀺 𑀫𑁂 𑀉𑀤𑀧𑀚𑁆𑀚𑀣𑁇_  


(262.)

4562. _𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀁 𑀅𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀁,_  
_𑀧𑀻𑀢𑀺 𑀫𑁂 𑀩𑀳𑀺 𑀦𑀺𑀘𑁆𑀙𑀭𑁂;_  
_𑀲𑁄𑀳𑀁 𑀧𑀻𑀢𑀺𑀫𑀦𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(263.)

4563. _‘𑀓𑀳𑀁 𑀦𑀼 𑀔𑁄 𑀲𑁄 𑀪𑀕𑀯𑀸,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀦𑀫𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀤𑀸𑀬𑀓𑀁𑁇_  


(264.)

4564. _𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀩𑀸𑀳𑀼𑀁,_  
_𑀯𑁂𑀤𑀚𑀸𑀢𑁄 𑀓𑀢𑀜𑁆𑀚𑀮𑀻;_  
_𑀆𑀘𑀺𑀓𑁆𑀔𑀺 𑀫𑁂 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀁,_  
_𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀯𑀺𑀦𑁄𑀤𑀦𑀁’𑁇_  


(265.)

4565. _‘𑀉𑀤𑁂𑀦𑁆𑀢𑀁𑀯 𑀫𑀳𑀸𑀫𑁂𑀖𑀁,_  
_𑀦𑀻𑀮𑀁 𑀅𑀜𑁆𑀚𑀦𑀲𑀦𑁆𑀦𑀺𑀪𑀁;_  
_𑀲𑀸𑀕𑀭𑀁 𑀯𑀺𑀬 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀁,_  
_𑀧𑀲𑁆𑀲𑀲𑁂𑀢𑀁 𑀫𑀳𑀸𑀯𑀦𑀁𑁇_  


(266.)

4566. _𑀏𑀢𑁆𑀣 𑀲𑁄 𑀯𑀲𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀤𑀦𑁆𑀢𑀤𑀫𑀓𑁄 𑀫𑀼𑀦𑀺;_  
_𑀯𑀺𑀦𑀬𑀦𑁆𑀢𑁄 𑀘 𑀯𑁂𑀦𑁂𑀬𑁆𑀬𑁂,_  
_𑀩𑁄𑀥𑁂𑀦𑁆𑀢𑁄 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑁂’𑁇_  


(267.)

4567. _𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄𑀯 𑀉𑀤𑀓𑀁,_  
_𑀪𑁄𑀚𑀦𑀁𑀯 𑀚𑀺𑀖𑀘𑁆𑀙𑀺𑀢𑁄;_  
_𑀕𑀸𑀯𑀻 𑀬𑀣𑀸 𑀯𑀘𑁆𑀙𑀕𑀺𑀤𑁆𑀥𑀸,_  
_𑀏𑀯𑀸𑀳𑀁 𑀯𑀺𑀘𑀺𑀦𑀺𑀁 𑀚𑀺𑀦𑀁𑁇_  


(268.)

4568. _𑀆𑀘𑀸𑀭𑁅𑀧𑀘𑀸𑀭𑀜𑁆𑀜𑀽,_  
_𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀲𑀁𑀯𑀭𑀁;_  
_𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑁂𑀫𑀺 𑀲𑀓𑁂 𑀲𑀺𑀲𑁆𑀲𑁂,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀚𑀺𑀦𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(269.)

4569. _‘𑀤𑀼𑀭𑀸𑀲𑀤𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑁄,_  
_𑀲𑀻𑀳𑀸𑀯 𑀏𑀓𑀘𑀸𑀭𑀺𑀦𑁄;_  
_𑀧𑀤𑁂 𑀧𑀤𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀸,_  
_𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀣 𑀫𑀸𑀡𑀯𑀸𑁇_  


(270.)

4570. _𑀆𑀲𑀻𑀯𑀺𑀲𑁄 𑀬𑀣𑀸 𑀖𑁄𑀭𑁄,_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀸𑀯 𑀓𑁂𑀲𑀭𑀻;_  
_𑀫𑀢𑁆𑀢𑁄𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄 𑀤𑀦𑁆𑀢𑀻,_  
_𑀏𑀯𑀁 𑀩𑀼𑀤𑁆𑀥𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸𑁇_  


(271.)

4571. _𑀉𑀓𑁆𑀓𑀸𑀲𑀺𑀢𑀜𑁆𑀘 𑀔𑀺𑀧𑀺𑀢𑀁,_  
_𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀬 𑀫𑀸𑀡𑀯𑀸;_  
_𑀧𑀤𑁂 𑀧𑀤𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀸,_  
_𑀉𑀧𑁂𑀣 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(272.)

4572. _𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀕𑀭𑀼𑀓𑀸,_  
_𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸;_  
_𑀤𑀼𑀭𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀸 𑀩𑀼𑀤𑁆𑀥𑀸,_  
_𑀕𑀭𑀽 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁂𑁇_  


(273.)

4573. _𑀬𑀤𑀸𑀳𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀬𑀸𑀫𑀺 𑀯𑀸;_  
_𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀸 𑀢𑀤𑀸 𑀳𑁄𑀣,_  
_𑀫𑀼𑀦𑀺𑀪𑀽𑀢𑀸𑀯 𑀢𑀺𑀝𑁆𑀞𑀣𑁇_  


(274.)

4574. _𑀬𑀁 𑀲𑁄 𑀤𑁂𑀲𑁂𑀢𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀔𑁂𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀢𑀫𑁂𑀯𑀢𑁆𑀣𑀁 𑀦𑀺𑀲𑀸𑀫𑁂𑀣,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀁 𑀲𑀼𑀔𑀁’𑁇_  


(275.)

4575. _𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁 𑀫𑀼𑀦𑀺𑀦𑀸 𑀅𑀳𑀁;_  
_𑀢𑀁 𑀓𑀣𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀮𑀓𑁆𑀔𑀡𑁂 𑀉𑀧𑀥𑀸𑀭𑀬𑀺𑀁𑁇_  


(276.)

4576. _𑀮𑀓𑁆𑀔𑀡𑁂 𑀤𑁆𑀯𑁂 𑀘 𑀓𑀗𑁆𑀔𑀸𑀫𑀺,_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀺𑀁𑀲𑀮𑀓𑁆𑀔𑀡𑁂;_  
_𑀓𑁄𑀲𑁄𑀳𑀺𑀢𑀯𑀢𑁆𑀣𑀕𑀼𑀬𑁆𑀳𑀁,_  
_𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀤𑀲𑁆𑀲𑀬𑀻 𑀫𑀼𑀦𑀺𑁇_  


(277.)

4577. _𑀚𑀺𑀯𑁆𑀳𑀁 𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀡𑁆𑀡𑀲𑁄𑀢𑁂 𑀘 𑀦𑀸𑀲𑀺𑀓𑁂;_  
_𑀧𑀝𑀺𑀫𑀲𑀺 𑀦𑀮𑀸𑀝𑀦𑁆𑀢𑀁,_  
_𑀓𑁂𑀯𑀮𑀁 𑀙𑀸𑀤𑀬𑀻 𑀚𑀺𑀦𑁄𑁇_  


(278.)

4578. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀮𑀓𑁆𑀔𑀡𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁂 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂;_  
_𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀦𑀺𑀝𑁆𑀞𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀳 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀁𑁇_  


(279.)

4579. _𑀲𑀢𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀲𑀳𑀺𑀢𑁄,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀅𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀲𑀩𑁆𑀩𑁂 𑀧𑀢𑁆𑀢𑀸𑀫𑁆𑀳 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀺𑀁𑁇_  


(280.)

4580. _𑀏𑀓𑀢𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂;_  
_𑀏𑀓𑀢𑁄 𑀲𑀁𑀲𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀏𑀓𑀢𑁄 𑀯𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑀬𑀼𑀁𑁇_  


(281.)

4581. _𑀕𑁄𑀧𑀸𑀦𑀲𑀺𑀬𑁄 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑁂 𑀯𑀲𑀺𑀁 𑀅𑀳𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(282.)

4582. _𑀤𑀺𑀲𑀸𑀲𑀼 𑀧𑀽𑀚𑀺𑀢𑁄 𑀳𑁄𑀫𑀺,_  
_𑀪𑁄𑀕𑀸 𑀘 𑀅𑀫𑀺𑀢𑀸 𑀫𑀫;_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀫𑀺 𑀲𑀩𑁆𑀩𑁂𑀲𑀁,_  
_𑀢𑀸𑀲𑁄 𑀫𑀫 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(283.)

4583. _𑀩𑁆𑀬𑀸𑀥𑀬𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀤𑀻𑀖𑀸𑀬𑀼𑀁 𑀧𑀸𑀮𑀬𑀸𑀫𑀺 𑀘;_  
_𑀲𑀼𑀔𑀼𑀫𑀘𑁆𑀙𑀯𑀺𑀓𑁄 𑀳𑁄𑀫𑀺,_  
_𑀆𑀯𑀸𑀲𑁂 𑀧𑀢𑁆𑀣𑀺𑀢𑁂 𑀯𑀲𑁂𑁇_  


(284.)

4584. _𑀅𑀝𑁆𑀞 𑀕𑁄𑀧𑀸𑀦𑀲𑀻 𑀤𑀢𑁆𑀯𑀸,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑁂 𑀯𑀲𑀺𑀁 𑀅𑀳𑀁;_  
_𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀭𑀳𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀏𑀢𑀁 𑀫𑁂 𑀅𑀧𑀭𑀝𑁆𑀞𑀫𑀁𑁇_  


(285.)

4585. _𑀲𑀩𑁆𑀩𑀯𑁄𑀲𑀺𑀢𑀯𑁄𑀲𑀸𑀦𑁄,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁄 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀅𑀝𑁆𑀞𑀕𑁄𑀧𑀸𑀦𑀲𑀻 𑀦𑀸𑀫,_  
_𑀢𑀯 𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(286.)

4586. _𑀧𑀜𑁆𑀘 𑀣𑀫𑁆𑀪𑀸𑀦𑀺 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀽𑀕𑀥𑀫𑁆𑀫𑁂 𑀯𑀲𑀺𑀁 𑀅𑀳𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀧𑀜𑁆𑀘 𑀳𑁂𑀢𑀽 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(287.)

4587. _𑀅𑀘𑀮𑁄 𑀳𑁄𑀫𑀺 𑀫𑁂𑀢𑁆𑀢𑀸𑀬,_  
_𑀅𑀦𑀽𑀦𑀗𑁆𑀕𑁄 𑀪𑀯𑀸𑀫𑀳𑀁;_  
_𑀆𑀤𑁂𑀬𑁆𑀬𑀯𑀘𑀦𑁄 𑀳𑁄𑀫𑀺,_  
_𑀦 𑀥𑀁𑀲𑁂𑀫𑀺 𑀬𑀣𑀸 𑀅𑀳𑀁𑁇_  


(288.)

4588. _𑀅𑀪𑀦𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀅𑀔𑀺𑀮𑁄 𑀳𑁄𑀫𑀺 𑀓𑀲𑁆𑀲𑀘𑀺;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀯𑀺𑀫𑀮𑁄 𑀳𑁄𑀫𑀺 𑀲𑀸𑀲𑀦𑁂𑁇_  


(289.)

4589. _𑀲𑀕𑀸𑀭𑀯𑁄 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁄 𑀅𑀦𑀸𑀲𑀯𑁄;_  
_𑀲𑀸𑀯𑀓𑁄 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀁 𑀯𑀦𑁆𑀤𑀢𑁂 𑀫𑀼𑀦𑀺𑁇_  


(290.)

4590. _𑀓𑀢𑁆𑀯𑀸 𑀲𑀼𑀓𑀢𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁,_  
_𑀲𑀸𑀮𑀸𑀬𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀳𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀧𑀜𑁆𑀘 𑀳𑁂𑀢𑀽 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(291.)

4591. _𑀉𑀘𑁆𑀘𑁂 𑀓𑀼𑀮𑁂 𑀧𑀚𑀸𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀳𑀸𑀪𑁄𑀕𑁄 𑀪𑀯𑀸𑀫𑀳𑀁;_  
_𑀲𑀩𑁆𑀩𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀳𑁄𑀫𑀺,_  
_𑀫𑀘𑁆𑀙𑁂𑀭𑀁 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(292.)

4592. _𑀕𑀫𑀦𑁂 𑀧𑀢𑁆𑀣𑀺𑀢𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁄 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺;_  
_𑀲𑀳 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀲𑁂𑀝𑁆𑀞𑁂𑀦,_  
_𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀫𑀫 𑀧𑀢𑁆𑀣𑀺𑀢𑀁𑁇_  


(293.)

4593. _𑀢𑁂𑀦 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀤𑀸𑀦𑁂𑀦,_  
_𑀢𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀯𑀺𑀦𑁄𑀤𑀬𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀣𑁂𑀭𑁄 𑀯𑀦𑁆𑀤𑀢𑀺 𑀢𑀁 𑀫𑀼𑀦𑀺𑁇_  


(294.)

4594. _𑀧𑀭𑀓𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀺 𑀲𑀸𑀥𑀬𑀺𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀧𑀸𑀯𑀺𑀲𑀺𑀁 𑀅𑀪𑀬𑀁 𑀧𑀼𑀭𑀁𑁇_  


(295.)

4595. _𑀧𑀭𑀺𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀲𑀸𑀮𑀫𑁆𑀳𑀺,_  
_𑀧𑀭𑀺𑀪𑁄𑀕𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀲𑁂𑀝𑁆𑀞𑀢𑁆𑀢𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀸𑀕𑀢𑁄𑁇_  


(296.)

4596. _𑀬𑁂 𑀓𑁂𑀘𑀺 𑀤𑀫𑀓𑀸 𑀮𑁄𑀓𑁂,_  
_𑀳𑀢𑁆𑀣𑀺𑀅𑀲𑁆𑀲𑁂 𑀤𑀫𑁂𑀦𑁆𑀢𑀺 𑀬𑁂;_  
_𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀭𑀡𑀸 𑀦𑀸𑀦𑀸,_  
_𑀤𑀸𑀭𑀼𑀡𑁂𑀦 𑀤𑀫𑁂𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(297.)

4597. _𑀦 𑀳𑁂𑀯𑀁 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀤𑀫𑁂𑀲𑀺 𑀦𑀭𑀦𑀸𑀭𑀺𑀬𑁄;_  
_𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦,_  
_𑀤𑀫𑁂𑀲𑀺 𑀉𑀢𑁆𑀢𑀫𑁂 𑀤𑀫𑁂𑁇_  


(298.)

4598. _𑀤𑀸𑀦𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑁂 𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀤𑁂𑀲𑀦𑀸𑀓𑀼𑀲𑀮𑁄 𑀫𑀼𑀦𑀺;_  
_𑀏𑀓𑀧𑀜𑁆𑀳𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑁄𑀯,_  
_𑀩𑁄𑀥𑁂𑀲𑀺 𑀢𑀺𑀲𑀢𑁂 𑀫𑀼𑀦𑀺𑁇_  


(299.)

4599. _𑀤𑀦𑁆𑀢𑀸 𑀫𑀬𑀁 𑀲𑀸𑀭𑀣𑀺𑀦𑀸,_  
_𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀸𑀲𑀯𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑀢𑁆𑀢𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀉𑀧𑀥𑀺𑀓𑁆𑀔𑀬𑁂𑁇_  


(300.)

4600. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀪𑀬𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀸𑀮𑀸𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(301.)

4601. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(302.)

4602. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(303.)

4603. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4604. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁂𑀮𑁄 𑀲𑀧𑀭𑀺𑀲𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4605. 𑀲𑁂𑀮𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
