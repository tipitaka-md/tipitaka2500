# 𑁨𑁭𑁇𑁧 𑀆𑀓𑀸𑀲𑀼𑀓𑁆𑀔𑀺𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕

### 𑀆𑀓𑀸𑀲𑀼𑀓𑁆𑀔𑀺𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3158\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀡𑁂;_  
_𑀚𑀮𑀚𑀕𑁆𑀕𑁂 𑀤𑀼𑀯𑁂 𑀕𑀬𑁆𑀳,_  
_𑀉𑀧𑀸𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(2.)

3159\. _𑀏𑀓𑀜𑁆𑀘 𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀸𑀤𑁂𑀲𑀼,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀁;_  
_𑀏𑀓𑀜𑁆𑀘 𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁 𑀅𑀳𑀁𑁇_  


(3.)

3160\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(4.)

3161\. _𑀇𑀢𑁄 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀓𑀭𑁄 𑀦𑀸𑀫,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(5.)

3162\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3163\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀓𑀸𑀲𑀼𑀓𑁆𑀔𑀺𑀧𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3164\. 𑀆𑀓𑀸𑀲𑀼𑀓𑁆𑀔𑀺𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
