# 𑁨𑁭𑁇𑁧𑁦 𑀢𑀭𑀡𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕

### 𑀢𑀭𑀡𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(39.)

3214\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀭𑀳𑁄;_  
_𑀦𑀤𑀻𑀢𑀻𑀭𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(40.)

3215\. _𑀦𑀸𑀯𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑁂 𑀢𑀢𑁆𑀣,_  
_𑀲𑀦𑁆𑀢𑀸𑀭𑀡𑀻 𑀫𑀳𑀡𑁆𑀡𑀯𑁂;_  
_𑀦𑀤𑀺𑀬𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀢𑀸𑀭𑁂𑀲𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(41.)

3216\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀢𑀸𑀭𑁂𑀲𑀺𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀢𑀭𑀡𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(42.)

3217\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁫𑁭𑁭)_  


3218\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀭𑀡𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3219\. 𑀢𑀭𑀡𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

3220\. 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇



---

3221\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3222\. _𑀉𑀓𑁆𑀔𑀺𑀧𑀻 𑀢𑁂𑀮𑀘𑀦𑁆𑀤𑀻 𑀘,_  
_𑀤𑀻𑀧𑀤𑁄 𑀘 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑁄;_  
_𑀫𑀘𑁆𑀙𑁄 𑀚𑀯𑁄 𑀲𑀍𑀅𑀮𑀤𑁄,_  
_𑀭𑀓𑁆𑀔𑀲𑁄 𑀢𑀭𑀡𑁄 𑀤𑀲;_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀘𑁂𑀢𑁆𑀣 𑀲𑀗𑁆𑀔𑀸𑀢𑀸,_  
_𑀢𑀸𑀮𑀻𑀲𑀁 𑀘𑁂𑀓𑀫𑁂𑀯 𑀘𑀸𑀢𑀺𑁇_
