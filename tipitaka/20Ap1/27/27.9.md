# 𑁨𑁭𑁇𑁯 𑀉𑀧𑀸𑀕𑀢𑀸𑀲𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀕𑀢𑀸𑀲𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(34.)

3207\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑁂𑀫𑀚𑁆𑀛𑁂,_  
_𑀲𑀭𑁄 𑀆𑀲𑀺 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀢𑀢𑁆𑀣𑀸𑀳𑀁 𑀭𑀓𑁆𑀔𑀲𑁄 𑀆𑀲𑀺𑀁,_  
_𑀳𑁂𑀞𑀲𑀻𑀮𑁄 𑀪𑀬𑀸𑀦𑀓𑁄𑁇_  


(35.)

3208\. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀫𑀫𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(36.)

3209\. _𑀉𑀧𑀸𑀕𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀤𑁂𑀯𑀤𑁂𑀯𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀆𑀲𑀬𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(37.)

3210\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀯𑀦𑁆𑀤𑀺𑀁 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑀦𑁆𑀤𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(38.)

3211\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3212\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀕𑀢𑀸𑀲𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3213\. 𑀉𑀧𑀸𑀕𑀢𑀸𑀲𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
