# 𑁨𑁭𑁇𑁫 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕

### 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(18.)

3183. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀭𑁄𑀫𑀲𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀢𑀫𑁆𑀳𑀺 𑀧𑀩𑁆𑀩𑀢𑀧𑀸𑀤𑀫𑁆𑀳𑀺,_  
_𑀲𑀫𑀡𑁄 𑀪𑀸𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄𑁇_  


(19.)

3184. _𑀩𑀺𑀍𑀆𑀮𑀺𑀬𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀫𑀡𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀅𑀦𑀼𑀫𑁄𑀤𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄𑁇_  


(20.)

3185. _𑀩𑀺𑀍𑀆𑀮𑀻 𑀢𑁂 𑀫𑀫 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀪𑀯𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀨𑀮𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀁 𑀢𑀯𑁇_  


(21.)

3186. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀩𑀺𑀍𑀆𑀮𑀺𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀺𑀍𑀆𑀮𑀺𑀬𑀸 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(22.)

3187. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3188. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3189. 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
