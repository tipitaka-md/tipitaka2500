# 𑁨𑁭𑁇𑁨 𑀢𑁂𑀮𑀫𑀓𑁆𑀔𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕

### 𑀢𑁂𑀮𑀫𑀓𑁆𑀔𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(6.)

3165\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀫𑁆𑀳𑀺 𑀪𑀕𑀯𑀢𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀫𑁆𑀳𑀺 𑀦𑀭𑀸𑀲𑀪𑁂;_  
_𑀩𑁄𑀥𑀺𑀬𑀸 𑀯𑁂𑀤𑀺𑀓𑀸𑀬𑀸𑀳𑀁,_  
_𑀢𑁂𑀮𑀁 𑀫𑀓𑁆𑀔𑁂𑀲𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(7.)

3166\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀢𑁂𑀮𑀁 𑀫𑀓𑁆𑀔𑀬𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀓𑁆𑀔𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(8.)

3167\. _𑀘𑀢𑀼𑀯𑀻𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀼𑀘𑁆𑀙𑀯𑀺 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(9.)

3168\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3169\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑁂𑀮𑀫𑀓𑁆𑀔𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3170\. 𑀢𑁂𑀮𑀫𑀓𑁆𑀔𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
