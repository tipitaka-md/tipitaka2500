# 𑁨𑁭𑁇𑁮 𑀲𑀍𑀅𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕

* 𑀲𑀍𑀅𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(30.)

3201\. _“𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀸𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀓𑀺𑀦𑁆𑀦𑀭𑁄 𑀢𑀤𑀸;_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀭𑀁𑀲𑀺𑀚𑀸𑀮𑀲𑀫𑀸𑀓𑀼𑀮𑀁𑁇_  


(31.)

3202\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀭𑀫𑀸𑀬 𑀘 𑀧𑀻𑀢𑀺𑀬𑀸;_  
_𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀲𑀍𑀅𑀮𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀑𑀓𑀺𑀭𑀺𑀁 𑀅𑀳𑀁𑁇_  


(32.)

3203\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(33.)

3204\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3205\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀍𑀅𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3206\. 𑀲𑀍𑀅𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
