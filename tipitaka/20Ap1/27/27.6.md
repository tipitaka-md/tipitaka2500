# 𑁨𑁭𑁇𑁬 𑀫𑀘𑁆𑀙𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀧𑀤𑀼𑀫𑀼𑀓𑁆𑀔𑀺𑀧𑀯𑀕𑁆𑀕

### 𑀫𑀘𑁆𑀙𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(23.)

3190\. _“𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀸𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀉𑀓𑁆𑀓𑀼𑀲𑁄 𑀆𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀫𑀘𑁆𑀙𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀫𑀼𑀦𑀺𑀦𑁄 𑀅𑀤𑀁𑁇_  


(24.)

3191\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀫𑀘𑁆𑀙𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀘𑁆𑀙𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(25.)

3192\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3193\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀘𑁆𑀙𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3194\. 𑀫𑀘𑁆𑀙𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
