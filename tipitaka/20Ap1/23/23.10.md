# 𑁨𑁩𑁇𑁧𑁦 𑀏𑀓𑀯𑀦𑁆𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀆𑀮𑀫𑁆𑀩𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀯𑀦𑁆𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(48.)

2964\. _“𑀉𑀲𑀪𑀁 𑀧𑀯𑀭𑀁 𑀯𑀻𑀭𑀁,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀁 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀺𑀦𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀫𑀯𑀦𑁆𑀤𑀳𑀁𑁇_  


(49.)

2965\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑀦𑁆𑀤𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(50.)

2966\. _𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀯𑀺𑀓𑀢𑀸𑀦𑀦𑁆𑀤𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(51.)

2967\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁪𑁧𑁯)_  


2968\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀯𑀦𑁆𑀤𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2969\. 𑀏𑀓𑀯𑀦𑁆𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

2970\. 𑀆𑀮𑀫𑁆𑀩𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇



---

2971\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2972\. _𑀆𑀮𑀫𑁆𑀩𑀡𑀜𑁆𑀘 𑀅𑀚𑀺𑀦𑀁,_  
_𑀫𑀁𑀲𑀤𑀸𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑁄;_  
_𑀅𑀩𑁆𑀬𑀸𑀥𑀺 𑀅𑀗𑁆𑀓𑁄𑀮𑀁 𑀲𑁄𑀡𑁆𑀡𑀁,_  
_𑀫𑀺𑀜𑁆𑀚𑀆𑀯𑁂𑀍𑀅𑀯𑀦𑁆𑀤𑀦𑀁;_  
_𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀡𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀪𑀺𑁇_
