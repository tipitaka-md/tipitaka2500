# 𑁨𑁩𑁇𑁪 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀮𑀫𑁆𑀩𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(19.)

2923\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀯𑁂𑀤𑀺 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑀸;_  
_𑀆𑀭𑀓𑁆𑀔𑁄 𑀘 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁄,_  
_𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(20.)

2924\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀲𑁂𑀲𑁂𑀦,_  
_𑀦 𑀧𑀲𑁆𑀲𑀺𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁;_  
_𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺 𑀉𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀢𑀸𑀲𑁄 𑀫𑀬𑁆𑀳𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(21.)

2925\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀯𑁂𑀤𑀺𑀁 𑀓𑀸𑀭𑀬𑀺𑀁 𑀧𑀼𑀭𑁂;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(22.)

2926\. _𑀇𑀢𑁄 𑀙𑀝𑁆𑀞𑀫𑁆𑀳𑀺 𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀅𑀧𑀲𑁆𑀲𑁂𑀦𑀲𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(23.)

2927\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2928\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2929\. 𑀆𑀭𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
