# 𑁨𑁩𑁇𑁩 𑀤𑁆𑀯𑁂𑀭𑀢𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀮𑀫𑁆𑀩𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀤𑁆𑀯𑁂𑀭𑀢𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(12.)

2914. _“𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀦𑀦𑁂 𑀅𑀳𑀁;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁𑁇_  


(13.)

2915. _𑀫𑀁𑀲𑀧𑁂𑀲𑀺 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀇𑀲𑁆𑀲𑀭𑀁 𑀓𑀸𑀭𑀬𑀸𑀫𑀳𑀁𑁇_  


(14.)

2916. _𑀇𑀫𑀺𑀦𑀸 𑀫𑀁𑀲𑀤𑀸𑀦𑁂𑀦,_  
_𑀭𑀢𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑁂 𑀫𑀫;_  
_𑀤𑀼𑀯𑁂𑀫𑁂 𑀭𑀢𑀦𑀸 𑀮𑁄𑀓𑁂,_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(15.)

2917. _𑀢𑁂𑀳𑀁 𑀲𑀩𑁆𑀩𑁂 𑀅𑀦𑀼𑀪𑁄𑀫𑀺,_  
_𑀫𑀁𑀲𑀤𑀸𑀦𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀕𑀢𑁆𑀢𑀜𑁆𑀘 𑀫𑀼𑀤𑀼𑀓𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀧𑀜𑁆𑀜𑀸 𑀦𑀺𑀧𑀼𑀡𑀯𑁂𑀤𑀦𑀻𑁇_  


(16.)

2918. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀫𑀁𑀲𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀁𑀲𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(17.)

2919. _𑀇𑀢𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀚𑀦𑀸𑀥𑀺𑀧𑁄;_  
_𑀫𑀳𑀸𑀭𑁄𑀳𑀺𑀢𑀦𑀸𑀫𑁄 𑀲𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(18.)

2920. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2921. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑁆𑀯𑁂𑀭𑀢𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2922. 𑀤𑁆𑀯𑁂𑀭𑀢𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇

  
𑀤𑀲𑀫𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
