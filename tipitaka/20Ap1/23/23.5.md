# 𑁨𑁩𑁇𑁫 𑀅𑀩𑁆𑀬𑀸𑀥𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀮𑀫𑁆𑀩𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀅𑀩𑁆𑀬𑀸𑀥𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(24.)

2930\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀅𑀕𑁆𑀕𑀺𑀲𑀸𑀮𑀁 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀩𑁆𑀬𑀸𑀥𑀺𑀓𑀸𑀦𑀜𑁆𑀘 𑀆𑀯𑀸𑀲𑀁,_  
_𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁𑁇_  


(25.)

2931\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑁄 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀩𑁆𑀬𑀸𑀥𑀸𑀳𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(26.)

2932\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀸𑀮𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀕𑁆𑀕𑀺𑀲𑀸𑀮𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(27.)

2933\. _𑀇𑀢𑁄 𑀘 𑀲𑀢𑁆𑀢𑀫𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄𑀲𑀺𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(28.)

2934\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2935\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀥𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2936\. 𑀅𑀩𑁆𑀬𑀸𑀥𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
