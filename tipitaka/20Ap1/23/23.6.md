# 𑁨𑁩𑁇𑁬 𑀅𑀗𑁆𑀓𑁄𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀆𑀮𑀫𑁆𑀩𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀅𑀗𑁆𑀓𑁄𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(29.)

2937. _“𑀦𑀸𑀭𑀤𑁄 𑀇𑀢𑀺 𑀫𑁂 𑀦𑀸𑀫𑀁,_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀢𑀺 𑀫𑀁 𑀯𑀺𑀤𑀽;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀫𑀡𑀸𑀦𑀕𑁆𑀕𑀁,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀤𑁂𑀯𑀲𑀓𑁆𑀓𑀢𑀁𑁇_  


(30.)

2938. _𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀥𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀅𑀗𑁆𑀓𑁄𑀮𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(31.)

2939. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(32.)

2940. _𑀘𑀢𑀼𑀲𑀢𑁆𑀢𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀭𑁄𑀫𑀲𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀆𑀫𑀼𑀓𑁆𑀓𑀫𑀸𑀮𑀸𑀪𑀭𑀡𑁄,_  
_𑀲𑀬𑁄𑀕𑁆𑀕𑀩𑀮𑀯𑀸𑀳𑀦𑁄𑁇_  


(33.)

2941. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2942. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀗𑁆𑀓𑁄𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2943. 𑀅𑀗𑁆𑀓𑁄𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
