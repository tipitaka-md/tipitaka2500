# 𑁫𑁩𑁇𑁩 𑀲𑀭𑀡𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀺𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀭𑀡𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(19.)

6820\. _“𑀆𑀭𑀼𑀳𑀺𑀫𑁆𑀳 𑀢𑀤𑀸 𑀦𑀸𑀯𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀸𑀚𑀻𑀯𑀺𑀓𑁄 𑀘𑀳𑀁;_  
_𑀦𑀸𑀯𑀸𑀬 𑀪𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸𑀬,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂 𑀲𑀭𑀡𑀁 𑀅𑀤𑀸𑁇_  


(20.)

6821\. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑁄 𑀫𑁂 𑀲𑀭𑀡𑀁 𑀅𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀭𑀡𑀕𑀫𑀦𑁂 𑀨𑀮𑀁𑁇_  


(21.)

6822\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(22.)

6823\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(23.)

6824\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6825\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀭𑀡𑀕𑀫𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6826\. 𑀲𑀭𑀡𑀕𑀫𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
