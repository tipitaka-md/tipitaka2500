# 𑁫𑁩𑁇𑁬 𑀤𑀡𑁆𑀟𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀢𑀺𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀤𑀡𑁆𑀟𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(36.)

6843\. _“𑀓𑀸𑀦𑀦𑀁 𑀯𑀦𑀫𑁄𑀕𑀬𑁆𑀳,_  
_𑀯𑁂𑀍𑀉𑀁 𑀙𑁂𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀆𑀮𑀫𑁆𑀩𑀡𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀤𑀤𑀺𑀁 𑀩𑀳𑀼𑀁𑁇_  


(37.)

6844\. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀲𑀼𑀩𑁆𑀩𑀢𑁂 𑀅𑀪𑀺𑀯𑀸𑀤𑀺𑀬;_  
_𑀆𑀮𑀫𑁆𑀩𑀤𑀡𑁆𑀟𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(38.)

6845\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀡𑁆𑀟𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀡𑁆𑀟𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(39.)

6846\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(40.)

6847\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(41.)

6848\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6849\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑀡𑁆𑀟𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6850\. 𑀤𑀡𑁆𑀟𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇

  
𑀢𑁂𑀯𑀻𑀲𑀢𑀺𑀫𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
