# 𑁫𑁩𑁇𑁧𑁦 𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀢𑀺𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(81.)

6896\. _“𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(82.)

6897\. _𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄;_  
_𑀩𑁆𑀬𑀸𑀫𑀧𑁆𑀧𑀪𑀸𑀧𑀭𑀺𑀯𑀼𑀢𑁄,_  
_𑀭𑀁𑀲𑀺𑀚𑀸𑀮𑀲𑀫𑁄𑀢𑁆𑀣𑀝𑁄𑁇_  


(83.)

6898\. _𑀅𑀲𑁆𑀲𑀸𑀲𑁂𑀢𑀸 𑀬𑀣𑀸 𑀘𑀦𑁆𑀤𑁄,_  
_𑀲𑀽𑀭𑀺𑀬𑁄𑀯 𑀧𑀪𑀗𑁆𑀓𑀭𑁄;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀸 𑀬𑀣𑀸 𑀫𑁂𑀖𑁄,_  
_𑀲𑀸𑀕𑀭𑁄𑀯 𑀕𑀼𑀡𑀸𑀓𑀭𑁄𑁇_  


(84.)

6899\. _𑀥𑀭𑀡𑀻𑀭𑀺𑀯 𑀲𑀻𑀮𑁂𑀦,_  
_𑀳𑀺𑀫𑀯𑀸𑀯 𑀲𑀫𑀸𑀥𑀺𑀦𑀸;_  
_𑀆𑀓𑀸𑀲𑁄 𑀯𑀺𑀬 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀲𑀗𑁆𑀕𑁄 𑀅𑀦𑀺𑀮𑁄 𑀬𑀣𑀸𑁇_  


(85.)

6900\. _𑀲 𑀓𑀤𑀸𑀘𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀧𑀭𑀺𑀲𑀸𑀲𑀼 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀲𑀫𑁆𑀧𑀓𑀸𑀲𑁂𑀢𑀺,_  
_𑀉𑀤𑁆𑀥𑀭𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀚𑀦𑀁𑁇_  


(86.)

6901\. _𑀢𑀤𑀸 𑀳𑀺 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁,_  
_𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀆𑀲𑀳𑀁 𑀥𑀦𑀥𑀜𑁆𑀜𑀲𑁆𑀲,_  
_𑀧𑀳𑀽𑀢𑀲𑁆𑀲 𑀩𑀳𑀽 𑀢𑀤𑀸𑁇_  


(87.)

6902\. _𑀚𑀗𑁆𑀖𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀺𑀘𑀭𑀁,_  
_𑀫𑀺𑀕𑀤𑀸𑀬𑀫𑀼𑀧𑁂𑀘𑁆𑀘𑀳𑀁;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀁 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁𑁇_  


(88.)

6903\. _𑀯𑀺𑀲𑀝𑁆𑀞𑀓𑀦𑁆𑀢𑀯𑀘𑀦𑀁,_  
_𑀓𑀭𑀯𑀻𑀓𑀲𑀫𑀲𑁆𑀲𑀭𑀁;_  
_𑀳𑀁𑀲𑀭𑀼𑀢𑁂𑀳𑀺 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁,_  
_𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀚𑀦𑀁𑁇_  


(89.)

6904\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑀁 𑀢𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀯 𑀫𑀥𑀼𑀭𑀁 𑀕𑀺𑀭𑀁;_  
_𑀧𑀳𑀸𑀬𑀦𑀧𑁆𑀧𑀓𑁂 𑀪𑁄𑀕𑁂,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(90.)

6905\. _𑀏𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀘𑀸𑀳𑀁,_  
_𑀦 𑀘𑀺𑀭𑁂𑀦 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄,_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀧𑀝𑀺𑀪𑀸𑀦𑀯𑀸𑁇_  


(91.)

6906\. _𑀫𑀳𑀸𑀧𑀭𑀺𑀲𑀫𑀚𑁆𑀛𑁂𑀳𑀁,_  
_𑀳𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁;_  
_𑀯𑀡𑁆𑀡𑀬𑀺𑀁 𑀳𑁂𑀫𑀯𑀡𑁆𑀡𑀲𑁆𑀲,_  
_𑀯𑀡𑁆𑀡𑀁 𑀯𑀡𑁆𑀡𑀯𑀺𑀲𑀸𑀭𑀤𑁄𑁇_  


(92.)

6907\. _‘𑀏𑀲 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀦𑀻𑀖𑁄 𑀙𑀺𑀦𑁆𑀦𑀲𑀁𑀲𑀬𑁄;_  
_𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀬𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄𑀧𑀥𑀺𑀲𑀗𑁆𑀔𑀬𑁂𑁇_  


(93.)

6908\. _𑀏𑀲 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀏𑀲 𑀲𑀻𑀳𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀓𑁆𑀓𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀓𑁄𑁇_  


(94.)

6909\. _𑀤𑀦𑁆𑀢𑁄 𑀤𑀫𑁂𑀢𑀸 𑀲𑀦𑁆𑀢𑁄 𑀘,_  
_𑀲𑀫𑁂𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀇𑀲𑀺;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀸 𑀘 𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑁄,_  
_𑀅𑀲𑁆𑀲𑀸𑀲𑁂𑀢𑀸 𑀫𑀳𑀸𑀚𑀦𑀁𑁇_  


(95.)

6910\. _𑀯𑀻𑀭𑁄 𑀲𑀽𑀭𑁄 𑀘 𑀯𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄,_  
_𑀧𑀜𑁆𑀜𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀯𑀲𑀻;_  
_𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀘 𑀲 𑀚𑀺𑀦𑁄,_  
_𑀅𑀧𑁆𑀧𑀕𑀩𑁆𑀩𑁄 𑀅𑀦𑀸𑀮𑀬𑁄𑁇_  


(96.)

6911\. _𑀅𑀦𑁂𑀜𑁆𑀚𑁄 𑀅𑀘𑀮𑁄 𑀥𑀻𑀫𑀸,_  
_𑀅𑀫𑁄𑀳𑁄 𑀅𑀲𑀫𑁄 𑀫𑀼𑀦𑀺;_  
_𑀥𑁄𑀭𑀬𑁆𑀳𑁄 𑀉𑀲𑀪𑁄 𑀦𑀸𑀕𑁄,_  
_𑀲𑀻𑀳𑁄 𑀲𑀓𑁆𑀓𑁄 𑀕𑀭𑀽𑀲𑀼𑀧𑀺𑁇_  


(97.)

6912\. _𑀯𑀺𑀭𑀸𑀕𑁄 𑀯𑀺𑀫𑀮𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸,_  
_𑀯𑀸𑀤𑀻 𑀲𑀽𑀭𑁄 𑀭𑀡𑀜𑁆𑀚𑀳𑁄;_  
_𑀅𑀔𑀺𑀮𑁄 𑀘 𑀯𑀺𑀲𑀮𑁆𑀮𑁄 𑀘,_  
_𑀅𑀲𑀫𑁄 𑀲𑀁𑀬𑀢𑁄 𑀲𑀼𑀘𑀺𑁇_  


(98.)

6913\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀫𑀡𑁄 𑀦𑀸𑀣𑁄,_  
_𑀪𑀺𑀲𑀓𑁆𑀓𑁄 𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑀓𑁄;_  
_𑀬𑁄𑀥𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀼𑀢𑀸𑀲𑀼𑀢𑁄,_  
_𑀅𑀘𑀮𑁄 𑀫𑀼𑀤𑀺𑀢𑁄 𑀲𑀺𑀢𑁄𑁇_  


(99.)

6914\. _𑀥𑀸𑀢𑀸 𑀥𑀢𑀸 𑀘 𑀲𑀦𑁆𑀢𑀺 𑀘,_  
_𑀓𑀢𑁆𑀢𑀸 𑀦𑁂𑀢𑀸 𑀧𑀓𑀸𑀲𑀺𑀢𑀸;_  
_𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀪𑁂𑀢𑁆𑀢𑀸 𑀘,_  
_𑀙𑁂𑀢𑁆𑀢𑀸 𑀲𑁄𑀢𑀸 𑀧𑀲𑀁𑀲𑀺𑀢𑀸𑁇_  


(100.)

6915\. _𑀅𑀔𑀺𑀮𑁄 𑀘 𑀯𑀺𑀲𑀮𑁆𑀮𑁄 𑀘,_  
_𑀅𑀦𑀻𑀖𑁄 𑀅𑀓𑀣𑀁𑀓𑀣𑀻;_  
_𑀅𑀦𑁂𑀚𑁄 𑀯𑀺𑀭𑀚𑁄 𑀓𑀢𑁆𑀢𑀸,_  
_𑀕𑀦𑁆𑀥𑀸 𑀯𑀢𑁆𑀢𑀸 𑀧𑀲𑀁𑀲𑀺𑀢𑀸𑁇_  


(101.)

6916\. _𑀢𑀸𑀭𑁂𑀢𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀭𑁂𑀢𑀸,_  
_𑀓𑀸𑀭𑁂𑀢𑀸 𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀺𑀢𑀸;_  
_𑀧𑀸𑀧𑁂𑀢𑀸 𑀲𑀳𑀺𑀢𑀸 𑀓𑀦𑁆𑀢𑀸,_  
_𑀳𑀦𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀻 𑀢𑀸𑀧𑀲𑁄𑁇_  


(102.)

6917\. _𑀲𑀫𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑀲𑀫𑁄,_  
_𑀅𑀲𑀳𑀸𑀬𑁄 𑀤𑀬𑀸𑀮𑀬𑁄;_  
_𑀅𑀘𑁆𑀙𑁂𑀭𑀲𑀢𑁆𑀢𑁄 𑀅𑀓𑀼𑀳𑁄,_  
_𑀓𑀢𑀸𑀯𑀻 𑀇𑀲𑀺𑀲𑀢𑁆𑀢𑀫𑁄𑁇_  


(103.)

6918\. _𑀦𑀺𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀓𑀗𑁆𑀔𑁄 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑁄,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀅𑀦𑀽𑀧𑀫𑁄;_  
_𑀲𑀩𑁆𑀩𑀯𑀸𑀓𑁆𑀬𑀧𑀣𑀸𑀢𑀻𑀢𑁄,_  
_𑀲𑀘𑁆𑀘𑀦𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀕𑀽 𑀚𑀺𑀦𑁄𑁇_  


(104.)

6919\. _𑀲𑀢𑁆𑀢𑀲𑀸𑀭𑀯𑀭𑁂 𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀲𑀸𑀤𑁄 𑀅𑀫𑀢𑀸𑀯𑀳𑁄;_  
_𑀢𑀲𑁆𑀫𑀸 𑀩𑀼𑀤𑁆𑀥𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘,_  
_𑀲𑀁𑀖𑁂 𑀲𑀤𑁆𑀥𑀸 𑀫𑀳𑀢𑁆𑀣𑀺𑀓𑀸’𑁇_  


(105.)

6920\. _𑀕𑀼𑀡𑁂𑀳𑀺 𑀏𑀯𑀫𑀸𑀤𑀻𑀳𑀺,_  
_𑀢𑀺𑀮𑁄𑀓𑀲𑀭𑀡𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀯𑀡𑁆𑀡𑁂𑀦𑁆𑀢𑁄 𑀧𑀭𑀺𑀲𑀸𑀫𑀚𑁆𑀛𑁂,_  
_𑀅𑀓𑀁 𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀅𑀳𑀁𑁇_  


(106.)

6921\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸𑀳𑀁 𑀢𑀼𑀲𑀺𑀢𑁂,_  
_𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀲𑀼𑀔𑀁;_  
_𑀢𑀢𑁄 𑀘𑀼𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀚𑀸𑀢𑁄 𑀳𑁄𑀫𑀺 𑀲𑀼𑀕𑀦𑁆𑀥𑀺𑀓𑁄𑁇_  


(107.)

6922\. _𑀦𑀺𑀲𑁆𑀲𑀸𑀲𑁄 𑀫𑀼𑀔𑀕𑀦𑁆𑀥𑁄 𑀘,_  
_𑀤𑁂𑀳𑀕𑀦𑁆𑀥𑁄 𑀢𑀣𑁂𑀯 𑀫𑁂;_  
_𑀲𑁂𑀤𑀕𑀦𑁆𑀥𑁄 𑀘 𑀲𑀢𑀢𑀁,_  
_𑀲𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑁄𑀯 𑀳𑁄𑀢𑀺 𑀫𑁂𑁇_  


(108.)

6923\. _𑀫𑀼𑀔𑀕𑀦𑁆𑀥𑁄 𑀲𑀤𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀘𑀫𑁆𑀧𑀓𑁄;_  
_𑀧𑀭𑀺𑀲𑀦𑁆𑀢𑁄 𑀲𑀤𑀸 𑀯𑀸𑀢𑀺,_  
_𑀲𑀭𑀻𑀭𑁄 𑀘 𑀢𑀣𑁂𑀯 𑀫𑁂𑁇_  


(109.)

6924\. _𑀕𑀼𑀡𑀢𑁆𑀣𑀯𑀲𑁆𑀲 𑀲𑀩𑁆𑀩𑀦𑁆𑀢𑀁,_  
_𑀨𑀮𑀁 𑀢𑀼 𑀧𑀭𑀫𑀩𑁆𑀪𑀼𑀢𑀁;_  
_𑀏𑀓𑀕𑁆𑀕𑀫𑀦𑀲𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀯𑀡𑁆𑀡𑀬𑀺𑀲𑁆𑀲𑀁 𑀲𑀼𑀡𑀸𑀣 𑀫𑁂𑁇_  


(110.)

6925\. _𑀕𑀼𑀡𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀢𑁆𑀯𑀸𑀦,_  
_𑀳𑀺𑀢𑀸𑀬 𑀘 𑀦 𑀲𑀤𑀺𑀲𑀁;_  
_𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀫𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣,_  
_𑀲𑀁𑀖𑁄 𑀯𑀻𑀭𑀲𑀫𑀸𑀬𑀼𑀢𑁄𑁇_  


(111.)

6926\. _𑀬𑀲𑀲𑁆𑀲𑀻 𑀲𑀼𑀔𑀺𑀢𑁄 𑀓𑀦𑁆𑀢𑁄,_  
_𑀚𑀼𑀢𑀺𑀫𑀸 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀦𑁄;_  
_𑀯𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀺𑀪𑀽𑀢𑁄 𑀘,_  
_𑀦𑀺𑀤𑁆𑀤𑁄𑀲𑁄 𑀧𑀜𑁆𑀜𑀯𑀸 𑀢𑀣𑀸𑁇_  


(112.)

6927\. _𑀔𑀻𑀡𑁂 𑀆𑀬𑀼𑀲𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀲𑀼𑀮𑀪𑀁 𑀩𑀼𑀤𑁆𑀥𑀪𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀢𑁂𑀲𑀁 𑀳𑁂𑀢𑀼𑀁 𑀧𑀯𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀢𑀁 𑀲𑀼𑀡𑀸𑀣 𑀬𑀣𑀸𑀢𑀣𑀁𑁇_  


(113.)

6928\. _𑀲𑀦𑁆𑀢𑀁 𑀬𑀲𑀁 𑀪𑀕𑀯𑀢𑁄,_  
_𑀯𑀺𑀥𑀺𑀦𑀸 𑀅𑀪𑀺𑀯𑀸𑀤𑀬𑀁;_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣𑀽𑀧𑀧𑀦𑁆𑀦𑁄𑀧𑀺,_  
_𑀬𑀲𑀲𑁆𑀲𑀻 𑀢𑁂𑀦 𑀳𑁄𑀫𑀳𑀁𑁇_  


(114.)

6929\. _𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀓𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀦𑁆𑀢𑀫𑀲𑀗𑁆𑀔𑀢𑀁;_  
_𑀯𑀡𑁆𑀡𑀬𑀁 𑀲𑀼𑀔𑀤𑁄 𑀆𑀲𑀺𑀁,_  
_𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀔𑀺𑀢𑁄 𑀢𑀢𑁄𑁇_  


(115.)

6930\. _𑀕𑀼𑀡𑀁 𑀯𑀤𑀦𑁆𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀻𑀢𑀺𑀲𑀫𑀸𑀬𑀼𑀢𑁄;_  
_𑀲𑀓𑀦𑁆𑀢𑀺𑀁 𑀧𑀭𑀓𑀦𑁆𑀢𑀺𑀜𑁆𑀘,_  
_𑀚𑀦𑀬𑀺𑀁 𑀢𑁂𑀦 𑀓𑀦𑁆𑀢𑀺𑀫𑀸𑁇_  


(116.)

6931\. _𑀚𑀺𑀦𑁄 𑀢𑁂 𑀢𑀺𑀢𑁆𑀣𑀺𑀓𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀓𑀼𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂;_  
_𑀕𑀼𑀡𑀁 𑀯𑀤𑀦𑁆𑀢𑁄 𑀚𑁄𑀢𑁂𑀲𑀺𑀁,_  
_𑀦𑀸𑀬𑀓𑀁 𑀚𑀼𑀢𑀺𑀫𑀸 𑀢𑀢𑁄𑁇_  


(117.)

6932\. _𑀧𑀺𑀬𑀓𑀸𑀭𑀻 𑀚𑀦𑀲𑁆𑀲𑀸𑀧𑀺,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀕𑀼𑀡𑀁 𑀯𑀤𑀁;_  
_𑀲𑀭𑀤𑁄𑀯 𑀲𑀲𑀗𑁆𑀓𑁄𑀳𑀁,_  
_𑀢𑁂𑀦𑀸𑀲𑀺𑀁 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀦𑁄𑁇_  


(118.)

6933\. _𑀬𑀣𑀸𑀲𑀢𑁆𑀢𑀺𑀯𑀲𑁂𑀦𑀸𑀳𑀁,_  
_𑀲𑀩𑁆𑀩𑀯𑀸𑀘𑀸𑀳𑀺 𑀲𑀦𑁆𑀣𑀯𑀺𑀁;_  
_𑀲𑀼𑀕𑀢𑀁 𑀢𑁂𑀦 𑀯𑀸𑀕𑀺𑀲𑁄,_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀧𑀝𑀺𑀪𑀸𑀦𑀯𑀸𑁇_  


(119.)

6934\. _𑀬𑁂 𑀩𑀸𑀮𑀸 𑀯𑀺𑀫𑀢𑀺𑀁 𑀧𑀢𑁆𑀢𑀸,_  
_𑀧𑀭𑀺𑀪𑁄𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁;_  
_𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀁 𑀢𑁂 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀦,_  
_𑀧𑀭𑀺𑀪𑀽𑀢𑁄 𑀦 𑀢𑁂𑀦𑀳𑀁𑁇_  


(120.)

6935\. _𑀩𑀼𑀤𑁆𑀥𑀯𑀡𑁆𑀡𑁂𑀦 𑀲𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀅𑀧𑀦𑁂𑀲𑀳𑀁;_  
_𑀦𑀺𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀫𑀦𑁄 𑀳𑁄𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸𑁇_  


(121.)

6936\. _𑀲𑁄𑀢𑀽𑀦𑀁 𑀯𑀼𑀤𑁆𑀥𑀺𑀫𑀚𑀦𑀺𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀤𑁂𑀲𑀓𑁄;_  
_𑀢𑁂𑀦𑀸𑀳𑀫𑀸𑀲𑀺𑀁 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀦𑀺𑀧𑀼𑀡𑀢𑁆𑀣𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑁄𑁇_  


(122.)

6937\. _𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀲𑀁𑀲𑀸𑀭𑀲𑀸𑀕𑀭𑁄;_  
_𑀲𑀺𑀔𑀻𑀯 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄,_  
_𑀧𑀸𑀧𑀼𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀺𑀁𑁇_  


(123.)

6938\. _𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀓𑀧𑁆𑀧𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑀫𑀳𑀁 𑀲𑀦𑁆𑀣𑀯𑀺𑀁 𑀚𑀺𑀦𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀯𑀡𑁆𑀡𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(124.)

6939\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(125.)

6940\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(126.)

6941\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁫𑁫𑁭𑁫)_  


6942\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀕𑀦𑁆𑀥𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6943\. 𑀲𑀼𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

6944\. 𑀢𑀺𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁄𑁇



---

6945\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



6946\. _𑀢𑀺𑀡𑀤𑁄 𑀫𑀜𑁆𑀘𑀤𑁄 𑀘𑁂𑀯,_  
_𑀲𑀭𑀡𑀩𑁆𑀪𑀜𑁆𑀚𑀦𑀧𑁆𑀧𑀤𑁄;_  
_𑀲𑀼𑀧𑀝𑁄 𑀤𑀡𑁆𑀟𑀤𑀸𑀬𑀻 𑀘,_  
_𑀦𑁂𑀮𑀧𑀽𑀚𑀻 𑀢𑀣𑁂𑀯 𑀘𑁇_  


6947\. _𑀩𑁄𑀥𑀺𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀓𑁄 𑀫𑀡𑁆𑀟𑁄,_  
_𑀲𑀼𑀕𑀦𑁆𑀥𑁄 𑀤𑀲𑀫𑁄𑀢𑀺 𑀘;_  
_𑀕𑀸𑀣𑀸𑀲𑀢𑀁 𑀲𑀢𑁂𑀯𑀻𑀲𑀁,_  
_𑀕𑀡𑀺𑀢𑀜𑁆𑀘𑁂𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑀲𑁄𑁇_
