# 𑁫𑁩𑁇𑁨 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀺𑀡𑀤𑀸𑀬𑀓𑀯𑀕𑁆𑀕

### 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(13.)

6812. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀏𑀓𑀫𑀜𑁆𑀘𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀲𑀧𑀸𑀡𑀺𑀦𑀸𑁇_  


(14.)

6813. _𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀬𑀸𑀦𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀬𑀸𑀦𑀁 𑀲𑀫𑀚𑁆𑀛𑀕𑀁;_  
_𑀢𑁂𑀦 𑀫𑀜𑁆𑀘𑀓𑀤𑀸𑀦𑁂𑀦,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(15.)

6814. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀫𑀜𑁆𑀘𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀜𑁆𑀘𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(16.)

6815. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(17.)

6816. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(18.)

6817. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6818. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6819. 𑀫𑀜𑁆𑀘𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
