# 𑁩𑁮𑁇𑁧𑁦 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(53.)

4166. _“𑀓𑀡𑀺𑀓𑀸𑀭𑀁𑀯 𑀚𑁄𑀢𑀦𑁆𑀢𑀁,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(54.)

4167. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀲𑀺𑀭𑁂 𑀓𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀨𑀮𑀁 𑀕𑀬𑁆𑀳,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(55.)

4168. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀫𑀤𑀤𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(56.)

4169. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(57.)

4170. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(58.)

4171. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁩𑁨𑁭𑁧)_  


4172. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4173. 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

4174. 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁄𑁇



---

4175. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



4176. _𑀩𑁄𑀥𑀺 𑀧𑀸𑀝𑀮𑀺 𑀉𑀧𑁆𑀧𑀮𑀻,_  
_𑀧𑀝𑁆𑀝𑀺 𑀘 𑀲𑀢𑁆𑀢𑀧𑀡𑁆𑀡𑀺𑀬𑁄;_  
_𑀕𑀦𑁆𑀥𑀫𑀼𑀝𑁆𑀞𑀺 𑀘 𑀘𑀺𑀢𑀓𑁄,_  
_𑀢𑀸𑀮𑀁 𑀲𑀼𑀫𑀦𑀤𑀸𑀫𑀓𑁄;_  
_𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀨𑀮𑀻 𑀘𑁂𑀯,_  
_𑀕𑀸𑀣𑀸 𑀏𑀓𑀽𑀦𑀲𑀝𑁆𑀞𑀺𑀓𑀸𑁇_
