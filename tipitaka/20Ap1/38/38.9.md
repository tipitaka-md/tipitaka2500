# 𑁩𑁮𑁇𑁯 𑀲𑀼𑀫𑀦𑀤𑀸𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀫𑀦𑀤𑀸𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(48.)

4159\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀦𑁆𑀳𑀸𑀢𑀓𑀲𑁆𑀲 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀫𑀦𑀤𑀸𑀫𑀁,_  
_𑀥𑀸𑀭𑀬𑀺𑀁 𑀧𑀼𑀭𑀢𑁄 𑀞𑀺𑀢𑁄𑁇_  


(49.)

4160\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀫𑀁 𑀥𑀸𑀭𑀬𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀼𑀫𑀦𑀥𑀸𑀭𑀡𑁂 𑀨𑀮𑀁𑁇_  


(50.)

4161\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(51.)

4162\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(52.)

4163\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4164\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀫𑀦𑀤𑀸𑀫𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4165\. 𑀲𑀼𑀫𑀦𑀤𑀸𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
