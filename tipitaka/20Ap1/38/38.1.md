# 𑁩𑁮𑁇𑁧 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

* 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

4096\. _“𑀧𑀸𑀝𑀮𑀺𑀁 𑀳𑀭𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀸𑀤𑀧𑀁 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀁;_  
_𑀏𑀓𑀁𑀲𑀁 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀧𑀸𑀝𑀮𑀺𑀁 𑀅𑀳𑀁𑁇_  


(2.)

4097\. _𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑀭𑀼𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀦𑀲𑀁;_  
_𑀅𑀦𑁆𑀢𑁄𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑀳𑀺𑀲𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀲𑀯𑀁𑁇_  


(3.)

4098\. _𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀮𑁄𑀓𑀫𑀳𑀺𑀢𑀁,_  
_𑀓𑀭𑀼𑀡𑀸𑀜𑀸𑀡𑀲𑀸𑀕𑀭𑀁;_  
_𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀯𑀺𑀬 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀧𑀸𑀝𑀮𑀺𑀁 𑀅𑀳𑀁𑁇_  


(4.)

4099\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀩𑁄𑀥𑀺𑀫𑀪𑀺𑀯𑀦𑁆𑀤𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑀦𑁆𑀤𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(5.)

4100\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4101\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4102\. 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
