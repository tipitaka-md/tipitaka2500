# 𑁩𑁮𑁇𑁮 𑀲𑀼𑀫𑀦𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀫𑀦𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(43.)

4152\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀲𑀼𑀫𑀦𑁂𑀳𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀥𑀸𑀭𑀬𑀸𑀫𑀺 𑀫𑀳𑀸𑀭𑀳𑀁𑁇_  


(44.)

4153\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(45.)

4154\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(46.)

4155\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(47.)

4156\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4157\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀫𑀦𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4158\. 𑀲𑀼𑀫𑀦𑀢𑀸𑀮𑀯𑀡𑁆𑀝𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
