# 𑁩𑁮𑁇𑁬 𑀕𑀦𑁆𑀥𑀫𑀼𑀝𑁆𑀞𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀩𑁄𑀥𑀺𑀯𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

### 𑀕𑀦𑁆𑀥𑀫𑀼𑀝𑁆𑀞𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(33.)

4138\. _“𑀘𑀺𑀢𑀓𑁂 𑀓𑀭𑀻𑀬𑀫𑀸𑀦𑁂,_  
_𑀦𑀸𑀦𑀸𑀕𑀦𑁆𑀥𑁂 𑀲𑀫𑀸𑀳𑀢𑁂;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀕𑀦𑁆𑀥𑀫𑀼𑀝𑁆𑀞𑀺𑀫𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(34.)

4139\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀺𑀢𑀓𑀁 𑀬𑀁 𑀅𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀺𑀢𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(35.)

4140\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(36.)

4141\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(37.)

4142\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4143\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀦𑁆𑀥𑀫𑀼𑀝𑁆𑀞𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4144\. 𑀕𑀦𑁆𑀥𑀫𑀼𑀝𑁆𑀞𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
