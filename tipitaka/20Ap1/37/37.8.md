# 𑁩𑁭𑁇𑁮 𑀏𑀓𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀫𑀦𑁆𑀤𑀸𑀭𑀯𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(27.)

4073\. _“𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀲𑀢𑀻 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂;_  
_𑀏𑀓𑀘𑀫𑁆𑀧𑀓𑀫𑀸𑀤𑀸𑀬,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(28.)

4074\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀘𑁆𑀘𑁂𑀓𑀫𑀼𑀦𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀉𑀪𑁄𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀧𑀽𑀚𑀬𑀺𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁𑁇_  


(29.)

4075\. _𑀧𑀜𑁆𑀘𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(30.)

4076\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4077\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4078\. 𑀏𑀓𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
