# 𑁩𑁭𑁇𑁭 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀫𑀦𑁆𑀤𑀸𑀭𑀯𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(24.)

4068. _“𑀅𑀦𑁄𑀫𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀗𑁆𑀕𑀸𑀓𑀽𑀮𑁂 𑀯𑀲𑀻 𑀢𑀤𑀸;_  
_𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀽𑀚𑀬𑀺𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁𑁇_  


(25.)

4069. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(26.)

4070. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4071. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4072. 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
