# 𑁧𑁩𑁇𑁧 𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕

### 𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

2054. _“𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀞𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(2.)

2055. _𑀲𑀻𑀳𑀁 𑀬𑀣𑀸 𑀯𑀦𑀘𑀭𑀁,_  
_𑀩𑁆𑀬𑀕𑁆𑀖𑀭𑀸𑀚𑀁𑀯 𑀦𑀺𑀢𑁆𑀢𑀲𑀁;_  
_𑀢𑀺𑀥𑀸𑀧𑀪𑀺𑀦𑁆𑀦𑀫𑀸𑀢𑀗𑁆𑀕𑀁,_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀁𑀯 𑀫𑀳𑁂𑀲𑀺𑀦𑀁𑁇_  


(3.)

2056. _𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁 𑀅𑀳𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀆𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑁄𑁇_  


(4.)

2057. _𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀼𑀧𑁆𑀨𑀘𑁆𑀙𑀤𑀦𑀸,_  
_𑀑𑀓𑀺𑀭𑀺𑀁𑀲𑀼 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(5.)

2058. _𑀢𑀢𑁄 𑀲𑀸 𑀧𑀼𑀧𑁆𑀨𑀓𑀜𑁆𑀘𑀼𑀓𑀸,_  
_𑀅𑀦𑁆𑀢𑁄𑀯𑀡𑁆𑀝𑀸 𑀩𑀳𑀺𑀫𑀼𑀔𑀸;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀙𑀤𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀢𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀣𑁇_  


(6.)

2059. _𑀢𑀜𑁆𑀘 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀲𑀺𑀁,_  
_𑀲𑀼𑀕𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂𑁇_  


(7.)

2060. _𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(8.)

2061. _𑀧𑀦𑁆𑀦𑀭𑀲𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀺;_  
_𑀯𑀻𑀢𑀫𑀮𑀸 𑀲𑀫𑀸𑀦𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(9.)

2062. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2063. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2064. 𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
