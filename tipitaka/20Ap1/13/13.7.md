# 𑁧𑁩𑁇𑁭 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕

### 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(75.)

2140. _“𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀯𑀲𑀸𑀫𑀺 𑀅𑀲𑁆𑀲𑀫𑁂 𑀅𑀳𑀁𑁇_  


(76.)

2141. _𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺,_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀨𑀮𑀸𑀦𑀺 𑀘;_  
_𑀧𑀼𑀝𑀓𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑀼𑀫𑀕𑁆𑀕𑁂 𑀮𑀕𑁆𑀕𑀺𑀢𑀁 𑀫𑀬𑀸𑁇_  


(77.)

2142. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀫𑀫𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀫𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(78.)

2143. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀨𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁;_  
_𑀯𑀺𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀚𑀦𑀦𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀸𑀯𑀳𑁄𑁇_  


(79.)

2144. _𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀣𑁇_  


(80.)

2145. _‘𑀇𑀫𑀺𑀦𑀸 𑀨𑀮𑀤𑀸𑀦𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀢𑀺’𑁇_  


(81.)

2146. _𑀢𑁂𑀦𑁂𑀯 𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦,_  
_𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁𑁇_  


(82.)

2147. _𑀇𑀢𑁄 𑀲𑀢𑁆𑀢𑀲𑀢𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺𑀁 𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(83.)

2148. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2149. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2150. 𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
