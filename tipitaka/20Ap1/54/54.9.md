# 𑁫𑁪𑁇𑁯 𑀭𑀸𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑀯𑀕𑁆𑀕

* 𑀭𑀸𑀥𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(296.)

7259\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀫𑀼𑀦𑀺;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(297.)

7260\. _𑀑𑀯𑀸𑀤𑀓𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑁄,_  
_𑀢𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀤𑁂𑀲𑀦𑀸𑀓𑀼𑀲𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(298.)

7261\. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀳𑀺𑀢𑁂𑀲𑀻 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀧𑀺𑁇_  


(299.)

7262\. _𑀏𑀯𑀁 𑀦𑀺𑀭𑀸𑀓𑀼𑀮𑀁 𑀆𑀲𑀺,_  
_𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀘;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺,_  
_𑀯𑀲𑀻𑀪𑀽𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀪𑀺𑁇_  


(300.)

7263\. _𑀭𑀢𑀦𑀸𑀦𑀝𑁆𑀞𑀧𑀜𑁆𑀜𑀸𑀲𑀁,_  
_𑀉𑀕𑁆𑀕𑀢𑁄 𑀲𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄𑁇_  


(301.)

7264\. _𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(302.)

7265\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀫𑀦𑁆𑀢𑀧𑀸𑀭𑀕𑀽;_  
_𑀉𑀧𑁂𑀘𑁆𑀘 𑀢𑀁 𑀦𑀭𑀯𑀭𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁𑁇_  


(303.)

7266\. _𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀧𑀭𑀺𑀲𑀸𑀲𑀼 𑀯𑀺𑀲𑀸𑀭𑀤𑀁;_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀬𑁆𑀬𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀯𑀺𑀦𑀸𑀬𑀓𑀁𑁇_  


(304.)

7267\. _𑀢𑀤𑀸𑀳𑀁 𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀲𑀁𑀖𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀦𑀺𑀧𑀘𑁆𑀘 𑀲𑀺𑀭𑀲𑀸 𑀧𑀸𑀤𑁂,_  
_𑀢𑀁 𑀞𑀸𑀦𑀁 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀺𑀁𑁇_  


(305.)

7268\. _𑀢𑀢𑁄 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀆𑀳,_  
_𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑀫𑀧𑁆𑀧𑀪𑁄;_  
_𑀲𑀭𑁂𑀦 𑀭𑀚𑀦𑀻𑀬𑁂𑀦,_  
_𑀓𑀺𑀮𑁂𑀲𑀫𑀮𑀳𑀸𑀭𑀺𑀦𑀸𑁇_  


(306.)

7269\. _‘𑀲𑀼𑀔𑀻 𑀪𑀯𑀲𑁆𑀲𑀼 𑀤𑀻𑀖𑀸𑀬𑀼,_  
_𑀲𑀺𑀚𑁆𑀛𑀢𑀼 𑀧𑀡𑀺𑀥𑀻 𑀢𑀯;_  
_𑀲𑀲𑀁𑀖𑁂 𑀫𑁂 𑀓𑀢𑀁 𑀓𑀸𑀭𑀁,_  
_𑀅𑀢𑀻𑀯 𑀯𑀺𑀧𑀼𑀮𑀁 𑀢𑀬𑀸𑁇_  


(307.)

7270\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(308.)

7271\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀭𑀸𑀥𑁄𑀢𑀺 𑀦𑀸𑀫𑀥𑁂𑀬𑁆𑀬𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄𑁇_  


(309.)

7272\. _𑀲 𑀢𑁂 𑀳𑁂𑀢𑀼𑀕𑀼𑀡𑁂 𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀬𑁆𑀬𑀓𑀸𑀦𑀕𑁆𑀕𑀁,_  
_𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀦𑀸𑀬𑀓𑁄’𑁇_  


(310.)

7273\. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀢𑀤𑀸 𑀚𑀺𑀦𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀲𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


(311.)

7274\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(312.)

7275\. _𑀲𑀢𑀸𑀦𑀁 𑀢𑀻𑀡𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(313.)

7276\. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑁄 𑀆𑀲𑀺𑀁,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸𑁇_  


(314.)

7277\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀚𑀸𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀓𑀼𑀮𑁂 𑀦𑀺𑀤𑁆𑀥𑁂,_  
_𑀯𑀺𑀓𑀮𑀘𑁆𑀙𑀸𑀤𑀦𑀸𑀲𑀦𑁂𑁇_  


(315.)

7278\. _𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀬𑀤𑀸 𑀚𑀺𑀡𑁆𑀡𑁄 𑀘 𑀯𑀼𑀤𑁆𑀥𑁄 𑀘,_  
_𑀢𑀤𑀸𑀭𑀸𑀫𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(316.)

7279\. _𑀧𑀩𑁆𑀩𑀚𑀢𑀺 𑀦 𑀫𑀁 𑀓𑁄𑀘𑀺,_  
_𑀚𑀺𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀮𑀣𑀸𑀫𑀓𑀁;_  
_𑀢𑁂𑀦 𑀤𑀻𑀦𑁄 𑀯𑀺𑀯𑀡𑁆𑀡𑀗𑁆𑀕𑁄,_  
_𑀲𑁄𑀓𑁄 𑀘𑀸𑀲𑀺𑀁 𑀢𑀤𑀸 𑀅𑀳𑀁𑁇_  


(317.)

7280\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀫𑀫𑀫𑀸𑀳 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_‘𑀓𑀺𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀢𑁆𑀢𑀲𑁄𑀓𑀝𑁆𑀝𑁄,_  
_𑀩𑁆𑀭𑀽𑀳𑀺 𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀚𑀁 𑀭𑀼𑀚𑀁’𑁇_  


(318.)

7281\. _‘𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀦 𑀮𑀪𑁂 𑀯𑀻𑀭,_  
_𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀢𑀯 𑀲𑀸𑀲𑀦𑁂;_  
_𑀢𑁂𑀦 𑀲𑁄𑀓𑁂𑀦 𑀤𑀻𑀦𑁄𑀲𑁆𑀫𑀺,_  
_𑀲𑀭𑀡𑀁 𑀳𑁄𑀳𑀺 𑀦𑀸𑀬𑀓’𑁇_  


(319.)

7282\. _𑀢𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀧𑀼𑀘𑁆𑀙𑀺 𑀫𑀼𑀦𑀺𑀲𑀢𑁆𑀢𑀫𑁄;_  
_‘𑀇𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀬𑁂,_  
_𑀲𑀭𑀦𑁆𑀢𑀺 𑀩𑁆𑀬𑀸𑀳𑀭𑀦𑁆𑀢𑀼 𑀢𑁂’𑁇_  


(320.)

7283\. _𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀤𑀸𑀯𑁄𑀘,_  
_‘𑀓𑀸𑀭𑀫𑀲𑁆𑀲 𑀲𑀭𑀸𑀫𑀳𑀁;_  
_𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀤𑀸𑀧𑁂𑀲𑀺,_  
_𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑁄 𑀫𑀫’𑁇_  


(321.)

7284\. _‘𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼 𑀓𑀢𑀜𑁆𑀜𑀽𑀲𑀺,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢 𑀇𑀫𑀁 𑀢𑀼𑀯𑀁;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀳𑀺 𑀤𑀺𑀚𑀁 𑀯𑀼𑀟𑁆𑀠𑀁,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀸𑀚𑀸𑀦𑀺𑀬𑁄 𑀅𑀬𑀁’𑁇_  


(322.)

7285\. _𑀢𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁,_  
_𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑁄𑀧𑀲𑀫𑁆𑀧𑀤𑀁;_  
_𑀦 𑀘𑀺𑀭𑁂𑀦𑁂𑀯 𑀓𑀸𑀮𑁂𑀦,_  
_𑀧𑀸𑀧𑀼𑀡𑀺𑀁 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(323.)

7286\. _𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀫𑀼𑀦𑀺𑀦𑁄 𑀯𑀸𑀓𑁆𑀬𑀁,_

_𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀬𑁆𑀬𑀓𑀸𑀦𑀕𑁆𑀕𑀁,_  
_𑀢𑀢𑁄 𑀫𑀁 𑀞𑀧𑀬𑀻 𑀚𑀺𑀦𑁄𑁇_  


(324.)

7287\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_

7289\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_
