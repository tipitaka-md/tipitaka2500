# 𑁫𑁪𑁇𑁭 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(221.)

7180. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀫𑀼𑀦𑀺;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(222.)

7181. _𑀑𑀯𑀸𑀤𑀓𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑁄,_  
_𑀢𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀤𑁂𑀲𑀦𑀸𑀓𑀼𑀲𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(223.)

7182. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀳𑀺𑀢𑁂𑀲𑀻 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀧𑀺𑁇_  


(224.)

7183. _𑀏𑀯𑀁 𑀦𑀺𑀭𑀸𑀓𑀼𑀮𑀁 𑀆𑀲𑀺,_  
_𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀘;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺,_  
_𑀯𑀲𑀻𑀪𑀽𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀪𑀺𑁇_  


(225.)

7184. _𑀭𑀢𑀦𑀸𑀦𑀝𑁆𑀞𑀧𑀜𑁆𑀜𑀸𑀲𑀁,_  
_𑀉𑀕𑁆𑀕𑀢𑁄 𑀲𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄𑁇_  


(226.)

7185. _𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(227.)

7186. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑁂𑀤𑀧𑀸𑀭𑀕𑀽;_  
_𑀉𑀧𑁂𑀘𑁆𑀘 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀕𑁆𑀕𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁𑁇_  


(228.)

7187. _𑀢𑀤𑀸 𑀲𑁄 𑀲𑀸𑀯𑀓𑀁 𑀯𑀻𑀭𑁄,_  
_𑀧𑀪𑀺𑀦𑁆𑀦𑀫𑀢𑀺𑀕𑁄𑀘𑀭𑀁;_  
_𑀅𑀢𑁆𑀣𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑁂 𑀘,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀘 𑀓𑁄𑀯𑀺𑀤𑀁𑁇_  


(229.)

7188. _𑀞𑀧𑁂𑀲𑀺 𑀏𑀢𑀤𑀕𑁆𑀕𑀫𑁆𑀳𑀺,_  
_𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑁄 𑀅𑀳𑀁;_  
_𑀲𑀲𑀸𑀯𑀓𑀁 𑀚𑀺𑀦𑀯𑀭𑀁,_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀪𑁄𑀚𑀬𑀺𑀁 𑀢𑀤𑀸𑁇_  


(230.)

7189. _𑀤𑀼𑀲𑁆𑀲𑁂𑀳𑀘𑁆𑀙𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀲𑀺𑀲𑁆𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀺𑀲𑀸𑀕𑀭𑀁;_  
_𑀦𑀺𑀧𑀘𑁆𑀘 𑀧𑀸𑀤𑀫𑀽𑀮𑀫𑁆𑀳𑀺,_  
_𑀢𑀁 𑀞𑀸𑀦𑀁 𑀧𑀢𑁆𑀣𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(231.)

7190. _𑀢𑀢𑁄 𑀅𑀯𑁄𑀘 𑀮𑁄𑀓𑀕𑁆𑀕𑁄,_  
_‘𑀧𑀲𑁆𑀲𑀣𑁂𑀢𑀁 𑀤𑀺𑀚𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀯𑀺𑀦𑀢𑀁 𑀧𑀸𑀤𑀫𑀽𑀮𑁂 𑀫𑁂,_  
_𑀓𑀫𑀮𑁄𑀤𑀭𑀲𑀧𑁆𑀧𑀪𑀁𑁇_  


(232.)

7191. _𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲,_  
_𑀞𑀸𑀦𑀁 𑀧𑀢𑁆𑀣𑀬𑀢𑁂 𑀅𑀬𑀁;_  
_𑀢𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬 𑀘𑀸𑀕𑁂𑀦,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑁂𑀦 𑀘𑁇_  


(233.)

7192. _𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀳𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑁂,_  
_𑀮𑀘𑁆𑀙𑀲𑁂 𑀢𑀁 𑀫𑀦𑁄𑀭𑀣𑀁𑁇_  


(234.)

7193. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(235.)

7194. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄’𑁇_  


(236.)

7195. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀢𑀤𑀸 𑀚𑀺𑀦𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀲𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


(237.)

7196. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(238.)

7197. _𑀲𑀢𑀸𑀦𑀁 𑀢𑀻𑀡𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(239.)

7198. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑁄 𑀆𑀲𑀺𑀁,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸𑁇_  


(240.)

7199. _𑀤𑀼𑀯𑁂 𑀪𑀯𑁂 𑀲𑀁𑀲𑀭𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑁂 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑁂;_  
_𑀅𑀜𑁆𑀜𑀁 𑀕𑀢𑀺𑀁 𑀦 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(241.)

7200. _𑀤𑀼𑀯𑁂 𑀓𑀼𑀮𑁂 𑀧𑀚𑀸𑀬𑀸𑀫𑀺,_  
_𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀅𑀣 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂;_  
_𑀦𑀻𑀘𑁂 𑀓𑀼𑀮𑁂 𑀦 𑀚𑀸𑀬𑀸𑀫𑀺,_  
_𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(242.)

7201. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀧𑁆𑀧𑀓𑀼𑀮𑁂,_  
_𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀫𑀳𑀤𑁆𑀥𑀦𑁂𑁇_  


(243.)

7202. _𑀫𑀸𑀢𑀸 𑀘𑀦𑁆𑀤𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀧𑀺𑀢𑀸 𑀫𑁂 𑀅𑀲𑁆𑀲𑀮𑀸𑀬𑀦𑁄;_  
_𑀬𑀤𑀸 𑀫𑁂 𑀧𑀺𑀢𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀺𑀦𑀬𑀻 𑀲𑀩𑁆𑀩𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸𑁇_  


(244.)

7203. _𑀢𑀤𑀸 𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁂,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀘𑀭𑀺𑀬𑁄,_  
_𑀉𑀧𑀚𑁆𑀛𑀸 𑀲𑀸𑀭𑀺𑀲𑀫𑁆𑀪𑀯𑁄𑁇_  


(245.)

7204. _𑀓𑁂𑀲𑁂𑀲𑀼 𑀙𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀲𑀼,_  
_𑀤𑀺𑀝𑁆𑀞𑀺 𑀙𑀺𑀦𑁆𑀦𑀸 𑀲𑀫𑀽𑀮𑀺𑀓𑀸;_  
_𑀦𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑁄 𑀘 𑀓𑀸𑀲𑀸𑀯𑀁,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(246.)

7205. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀘 𑀫𑁂 𑀫𑀢𑀺;_  
_𑀧𑀪𑀺𑀦𑁆𑀦𑀸 𑀢𑁂𑀦 𑀮𑁄𑀓𑀕𑁆𑀕𑁄,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑀺 𑀫𑀁𑁇_  


(247.)

7206. _𑀅𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺𑀁,_  
_𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄;_  
_𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀲𑀼 𑀢𑁂𑀦𑀸𑀳𑀁,_  
_𑀅𑀕𑁆𑀕𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(248.)

7207. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_

(249.)

7208. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(250.)

7209. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_
