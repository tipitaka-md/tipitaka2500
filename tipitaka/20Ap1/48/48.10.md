# 𑁪𑁮𑁇𑁧𑁦 𑀧𑀼𑀮𑀺𑀦𑀘𑀗𑁆𑀓𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀮𑀺𑀦𑀘𑀗𑁆𑀓𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(90.)

6173\. _“𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀦𑀦𑁂 𑀅𑀳𑀁;_  
_𑀯𑀸𑀢𑀫𑀺𑀕𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀘𑀗𑁆𑀓𑀫𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀳𑀁𑁇_  


(91.)

6174\. _𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂𑀦 𑀧𑀼𑀮𑀺𑀦𑀁 𑀕𑀬𑁆𑀳,_  
_𑀘𑀗𑁆𑀓𑀫𑁂 𑀑𑀓𑀺𑀭𑀺𑀁 𑀅𑀳𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀲𑀺𑀭𑀻𑀫𑀢𑁄𑁇_  


(92.)

6175\. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀼𑀮𑀺𑀦𑀁 𑀑𑀓𑀺𑀭𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀮𑀺𑀦𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(93.)

6176\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(94.)

6177\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(95.)

6178\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁪𑁯𑁩𑁮)_  


6179\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀮𑀺𑀦𑀘𑀗𑁆𑀓𑀫𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6180\. 𑀧𑀼𑀮𑀺𑀦𑀘𑀗𑁆𑀓𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

6181\. 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁄𑁇



---

6182\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



6183\. _𑀦𑀍𑀅𑀫𑀸𑀮𑀻 𑀫𑀡𑀺𑀤𑀤𑁄,_  
_𑀉𑀓𑁆𑀓𑀸𑀲𑀢𑀺𑀓𑀩𑀻𑀚𑀦𑀻;_  
_𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑁄 𑀘 𑀓𑀼𑀲𑀝𑁆𑀞𑁄 𑀘,_  
_𑀕𑀺𑀭𑀺𑀧𑀼𑀦𑁆𑀦𑀸𑀕𑀺𑀬𑁄𑀧𑀺 𑀘𑁇_  


6184\. _𑀯𑀮𑁆𑀮𑀺𑀓𑀸𑀭𑁄 𑀧𑀸𑀦𑀥𑀺𑀤𑁄,_  
_𑀅𑀣𑁄 𑀧𑀼𑀮𑀺𑀦𑀘𑀗𑁆𑀓𑀫𑁄;_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀧𑀜𑁆𑀘𑀦𑀯𑀼𑀢𑀺,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_
