# 𑁪𑁮𑁇𑁧 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀯𑀕𑁆𑀕

### 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

6066\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀯𑀺𑀧𑀺𑀦𑀕𑁆𑀕𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(2.)

6067\. _𑀦𑀍𑀅𑀫𑀸𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁄 𑀘 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑀫𑀦𑀸𑀲𑀯𑀁𑁇_  


(3.)

6068\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀦𑀍𑀅𑀫𑀸𑀮𑀫𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑀁𑁇_  


(4.)

6069\. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀫𑀸𑀮𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(5.)

6070\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(6.)

6071\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(7.)

6072\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6073\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6074\. 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
