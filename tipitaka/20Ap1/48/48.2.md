# 𑁪𑁮𑁇𑁨 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀯𑀕𑁆𑀕

* 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(8.)

6075\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀯𑀺𑀯𑁂𑀓𑀓𑀸𑀫𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀘𑁆𑀙𑀢𑁂 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(9.)

6076\. _𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀫𑀳𑀸𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁄 𑀅𑀳𑀼;_  
_𑀢𑀢𑁆𑀣 𑀫𑁂 𑀪𑀯𑀦𑀁 𑀆𑀲𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑀁𑁇_  


(10.)

6077\. _𑀪𑀯𑀦𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀇𑀦𑁆𑀤𑀻𑀯𑀭𑀁𑀯 𑀚𑀮𑀺𑀢𑀁,_  
_𑀆𑀤𑀺𑀢𑁆𑀢𑀁𑀯 𑀳𑀼𑀢𑀸𑀲𑀦𑀁𑁇_  


(11.)

6078\. _𑀯𑀺𑀘𑀺𑀦𑀁 𑀦𑀤𑁆𑀤𑀲𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀧𑀽𑀚𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦𑀸𑀬𑀓𑀁;_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(12.)

6079\. _𑀫𑀫 𑀲𑀻𑀲𑁂 𑀫𑀡𑀺𑀁 𑀕𑀬𑁆𑀳,_  
_𑀧𑀽𑀚𑀬𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀇𑀫𑀸𑀬 𑀫𑀡𑀺𑀧𑀽𑀚𑀸𑀬,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑁄𑀢𑀼 𑀪𑀤𑁆𑀤𑀓𑁄𑁇_  


(13.)

6080\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀣𑁇_  


(14.)

6081\. _‘𑀲𑁄 𑀢𑁂 𑀇𑀚𑁆𑀛𑀢𑀼 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀮𑀪𑀲𑁆𑀲𑀼 𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀼𑀔𑀁;_  
_𑀇𑀫𑀸𑀬 𑀫𑀡𑀺𑀧𑀽𑀚𑀸𑀬,_  
_𑀅𑀦𑀼𑀪𑁄𑀳𑀺 𑀫𑀳𑀸𑀬𑀲𑀁’𑁇_  


(15.)

6082\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀸,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀓𑁄;_  
_𑀅𑀕𑀫𑀸𑀲𑀺 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀬𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀡𑀻𑀳𑀺𑀢𑀁𑁇_  


(16.)

6083\. _𑀲𑀝𑁆𑀞𑀺𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀅𑀦𑁂𑀓𑀲𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(17.)

6084\. _𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀲𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀤𑁂𑀯𑀪𑀽𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄;_  
_𑀫𑀡𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀆𑀮𑁄𑀓𑀓𑀭𑀡𑁄 𑀫𑀫𑀁𑁇_  


(18.)

6085\. _𑀙𑀍𑀅𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀫𑁂 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀸;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀸𑀪𑀭𑀡𑀸,_  
_𑀆𑀫𑀼𑀓𑁆𑀓𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀸𑁇_  


(19.)

6086\. _𑀅𑀍𑀆𑀭𑀧𑀫𑁆𑀳𑀸 𑀳𑀲𑀼𑀮𑀸,_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀸 𑀢𑀦𑀼𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀫𑀡𑀺𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(20.)

6087\. _𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀸 𑀫𑀡𑀺𑀫𑀬𑀸,_  
_𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀕𑀫𑀬𑀸 𑀢𑀣𑀸;_  
_𑀪𑀡𑁆𑀟𑀸 𑀫𑁂 𑀲𑀼𑀓𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀬𑀤𑀺𑀘𑁆𑀙𑀲𑀺 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀦𑀸𑁇_  


(21.)

6088\. _𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀕𑀳𑀸𑀭𑀫𑁆𑀫𑀸,_  
_𑀲𑀬𑀦𑀜𑁆𑀘 𑀫𑀳𑀸𑀭𑀳𑀁;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(22.)

6089\. _𑀮𑀸𑀪𑀸 𑀢𑁂𑀲𑀁 𑀲𑀼𑀮𑀤𑁆𑀥𑀜𑁆𑀘,_  
_𑀬𑁂 𑀮𑀪𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀑𑀲𑀥𑀁 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(23.)

6090\. _𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀬𑁄𑀳𑀁 𑀅𑀤𑀓𑁆𑀔𑀺 𑀦𑀸𑀬𑀓𑀁;_  
_𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀸 𑀧𑀫𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁𑁇_  


(24.)

6091\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀤𑀺𑀯𑀲𑀜𑁆𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘,_  
_𑀆𑀮𑁄𑀓𑁄 𑀳𑁄𑀢𑀺 𑀫𑁂 𑀲𑀤𑀸𑁇_  


(25.)

6092\. _𑀢𑀸𑀬𑁂𑀯 𑀫𑀡𑀺𑀧𑀽𑀚𑀸𑀬,_  
_𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀜𑀸𑀡𑀸𑀮𑁄𑀓𑁄 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁𑁇_  


(26.)

6093\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀫𑀡𑀺𑀁 𑀅𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀡𑀺𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(27.)

6094\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(28.)

6095\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(29.)

6096\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6097\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6098\. 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
