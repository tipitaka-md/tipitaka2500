# 𑁪𑁮𑁇𑁩 𑀉𑀓𑁆𑀓𑀸𑀲𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀯𑀕𑁆𑀕

* 𑀉𑀓𑁆𑀓𑀸𑀲𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(30.)

6099\. _“𑀓𑁄𑀲𑀺𑀬𑁄 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸,_  
_𑀘𑀺𑀢𑁆𑀢𑀓𑀽𑀝𑁂 𑀯𑀲𑀻 𑀢𑀤𑀸;_  
_𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀯𑀺𑀯𑁂𑀓𑀸𑀪𑀺𑀭𑀢𑁄 𑀫𑀼𑀦𑀺𑁇_  


(31.)

6100\. _𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀁,_  
_𑀦𑀸𑀭𑀻𑀕𑀡𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀓𑁄𑀲𑀺𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀬𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀁𑁇_  


(32.)

6101\. _𑀉𑀓𑁆𑀓𑀸𑀲𑀢𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀞𑀢𑁆𑀯𑀸,_  
_𑀅𑀝𑁆𑀞𑀫𑁂𑀦 𑀅𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(33.)

6102\. _𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀓𑁄𑀲𑀺𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀁 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(34.)

6103\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀁 𑀢𑀼𑀲𑀺𑀢𑁂 𑀓𑀸𑀬𑁂,_  
_𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(35.)

6104\. _𑀤𑀺𑀯𑀲𑀜𑁆𑀘𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘,_  
_𑀆𑀮𑁄𑀓𑁄 𑀳𑁄𑀢𑀺 𑀫𑁂 𑀲𑀤𑀸;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀑𑀪𑀸𑀲𑁂𑀦 𑀨𑀭𑀸𑀫𑀳𑀁𑁇_  


(36.)

6105\. _𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻,_  
_𑀚𑀫𑁆𑀩𑀼𑀫𑀡𑁆𑀟𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑁄𑁇_  


(37.)

6106\. _𑀢𑀤𑀸 𑀫𑁂 𑀦𑀕𑀭𑀁 𑀆𑀲𑀺,_  
_𑀇𑀤𑁆𑀥𑀁 𑀨𑀻𑀢𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁;_  
_𑀢𑀺𑀁𑀲𑀬𑁄𑀚𑀦𑀫𑀸𑀬𑀸𑀫𑀁,_  
_𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀘 𑀯𑀻𑀲𑀢𑀺𑁇_  


(38.)

6107\. _𑀲𑁄𑀪𑀦𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁂𑀦 𑀫𑀸𑀧𑀺𑀢𑀁;_  
_𑀤𑀲𑀲𑀤𑁆𑀤𑀸𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀢𑀁,_  
_𑀲𑀫𑁆𑀫𑀢𑀸𑀍𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁𑁇_  


(39.)

6108\. _𑀦 𑀢𑀫𑁆𑀳𑀺 𑀦𑀕𑀭𑁂 𑀅𑀢𑁆𑀣𑀺,_  
_𑀯𑀮𑁆𑀮𑀺𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀫𑀢𑁆𑀢𑀺𑀓𑀸;_  
_𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀁𑀬𑁂𑀯,_  
_𑀚𑁄𑀢𑀢𑁂 𑀦𑀺𑀘𑁆𑀘𑀓𑀸𑀮𑀺𑀓𑀁𑁇_  


(40.)

6109\. _𑀘𑀢𑀼𑀧𑀸𑀓𑀸𑀭𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁,_  
_𑀢𑀬𑁄 𑀆𑀲𑀼𑀁 𑀫𑀡𑀺𑀫𑀬𑀸;_  
_𑀯𑁂𑀫𑀚𑁆𑀛𑁂 𑀢𑀸𑀮𑀧𑀦𑁆𑀢𑀻 𑀘,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁂𑀦 𑀫𑀸𑀧𑀺𑀢𑀸𑁇_  


(41.)

6110\. _𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀧𑁄𑀓𑁆𑀔𑀭𑀜𑁆𑀜𑁄,_  
_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀙𑀸𑀤𑀺𑀢𑀸;_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀦𑀸𑀦𑀸𑀕𑀦𑁆𑀥𑀲𑀫𑀻𑀭𑀺𑀢𑀸𑁇_  


(42.)

6111\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀉𑀓𑁆𑀓𑀁 𑀥𑀸𑀭𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀉𑀓𑁆𑀓𑀸𑀥𑀸𑀭𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(43.)

6112\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(44.)

6113\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(45.)

6114\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6115\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀓𑁆𑀓𑀸𑀲𑀢𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6116\. 𑀉𑀓𑁆𑀓𑀸𑀲𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
