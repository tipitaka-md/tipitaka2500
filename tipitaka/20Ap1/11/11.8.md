# 𑁧𑁧𑁇𑁮 𑀆𑀲𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀳𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕

### 𑀆𑀲𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀳𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(47.)

1912\. _“𑀓𑀸𑀦𑀦𑀁 𑀯𑀦𑀫𑁄𑀕𑀬𑁆𑀳,_  
_𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀦𑀺𑀭𑀸𑀓𑀼𑀮𑀁;_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(48.)

1913\. _𑀫𑀸𑀮𑀸𑀳𑀢𑁆𑀣𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(49.)

1914\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪;_  
_𑀲𑀦𑁆𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀫𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(50.)

1915\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(51.)

1916\. _𑀇𑀢𑁄 𑀲𑀢𑁆𑀢𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀲𑀦𑁆𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀓 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(52.)

1917\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1918\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀲𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀳𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1919\. 𑀆𑀲𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀳𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
