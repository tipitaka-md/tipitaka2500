# 𑁧𑁧𑁇𑁭 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕

* 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(36.)

1899\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀢𑀺𑀡𑀢𑁆𑀣𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(37.)

1900\. _𑀦𑀍𑀅𑀫𑀸𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀩𑀻𑀚𑀦𑀺𑀁 𑀅𑀳𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺𑀁,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(38.)

1901\. _𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽,_  
_𑀩𑀻𑀚𑀦𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀣𑁇_  


(39.)

1902\. _‘𑀬𑀣𑀸 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀺,_  
_𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀢𑀣𑁂𑀯 𑀢𑀺𑀯𑀺𑀥𑀕𑁆𑀕𑀻𑀳𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀢𑀯 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀼’𑁇_  


(40.)

1903\. _𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸 𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑀼𑀁,_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀯𑀦𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸;_  
_𑀲𑁄𑀲𑁆𑀲𑀸𑀫 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑀁,_  
_𑀳𑀸𑀲𑀬𑀦𑁆𑀢𑀜𑁆𑀘 𑀤𑀸𑀬𑀓𑀁𑁇_  


(41.)

1904\. _𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀸 𑀢𑀢𑁆𑀣,_  
_𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀤𑀸𑀬𑀓𑀁 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀦𑁆𑀢𑁄,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(42.)

1905\. _‘𑀇𑀫𑀺𑀦𑀸 𑀩𑀻𑀚𑀦𑀺𑀤𑀸𑀦𑁂𑀦,_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀲𑀼𑀩𑁆𑀩𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(43.)

1906\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀯𑀲𑁂𑀲𑁂𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀫𑀸𑀮𑀼𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(44.)

1907\. _𑀇𑀫𑀺𑀦𑀸 𑀩𑀻𑀚𑀦𑀺𑀤𑀸𑀦𑁂𑀦,_  
_𑀲𑀫𑁆𑀫𑀸𑀦𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀘;_  
_𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇_  


(45.)

1908\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀲𑀼𑀩𑁆𑀩𑀢𑀸 𑀅𑀝𑁆𑀞𑀢𑀺𑀁𑀲 𑀢𑁂;_  
_𑀏𑀓𑀽𑀦𑀢𑀺𑀁𑀲𑀲𑀳𑀲𑁆𑀲𑁂,_  
_𑀅𑀝𑁆𑀞 𑀫𑀸𑀮𑀼𑀢𑀦𑀸𑀫𑀓𑀸𑁇_  


(46.)

1909\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1910\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1911\. 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
