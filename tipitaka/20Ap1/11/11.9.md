# 𑁧𑁧𑁇𑁯 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕

### 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(53.)

1920\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀯𑀲𑀸𑀫𑀺 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂;_  
_𑀖𑀸𑀲𑁂𑀲𑀼 𑀕𑁂𑀥𑀫𑀸𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑁂𑀬𑁆𑀬𑀲𑀻𑀮𑁄 𑀘𑀳𑀁 𑀢𑀤𑀸𑁇_  


(54.)

1921\. _𑀔𑀡𑀦𑁆𑀢𑀸𑀮𑀼 𑀓𑀮𑀫𑁆𑀩𑀸𑀦𑀺,_  
_𑀩𑀺𑀍𑀆𑀮𑀺𑀢𑀓𑁆𑀓𑀮𑀸𑀦𑀺 𑀘;_  
_𑀓𑁄𑀮𑀁 𑀪𑀮𑁆𑀮𑀸𑀢𑀓𑀁 𑀩𑀺𑀮𑁆𑀮𑀁,_  
_𑀆𑀳𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀬𑀸𑀤𑀺𑀢𑀁𑁇_  


(55.)

1922\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(56.)

1923\. _𑀉𑀧𑀸𑀕𑀢𑀁 𑀫𑀳𑀸𑀦𑀸𑀕𑀁,_  
_𑀤𑁂𑀯𑀤𑁂𑀯𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀩𑀺𑀍𑀆𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀑𑀓𑀺𑀭𑀺𑀁 𑀅𑀳𑀁𑁇_  


(57.)

1924\. _𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀢𑁄𑀲𑀬𑀦𑁆𑀢𑁄 𑀫𑀫𑀁 𑀢𑀤𑀸;_  
_𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽,_  
_𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀣𑁇_  


(58.)

1925\. _‘𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀩𑀺𑀍𑀆𑀮𑀺𑀁 𑀫𑁂 𑀅𑀤𑀸 𑀢𑀼𑀯𑀁;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀲𑀺’𑁇_  


(59.)

1926\. _𑀘𑀭𑀺𑀫𑀁 𑀯𑀢𑁆𑀢𑀢𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(60.)

1927\. _𑀘𑀢𑀼𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀼𑀫𑁂𑀔𑀮𑀺𑀬𑀲𑀯𑁆𑀳𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(61.)

1928\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1929\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1930\. 𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
