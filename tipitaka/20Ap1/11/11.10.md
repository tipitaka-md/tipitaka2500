# 𑁧𑁧𑁇𑁧𑁦 𑀭𑁂𑀡𑀼𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕

### 𑀭𑁂𑀡𑀼𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(62.)

1931\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀢𑀭𑀁𑀲𑀺𑀁𑀯 𑀪𑀸𑀡𑀼𑀫𑀁;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀁 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀉𑀍𑀉𑀭𑀸𑀚𑀁𑀯 𑀧𑀽𑀭𑀺𑀢𑀁𑁇_  


(63.)

1932\. _𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁 𑀲𑀸𑀯𑀓𑁂𑀳𑀺,_  
_𑀲𑀸𑀕𑀭𑁂𑀳𑁂𑀯 𑀫𑁂𑀤𑀦𑀺𑀁;_  
_𑀦𑀸𑀕𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀭𑁂𑀡𑀽𑀳𑀺,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(64.)

1933\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀭𑁂𑀡𑀼𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(65.)

1934\. _𑀧𑀡𑁆𑀡𑀢𑀸𑀮𑀻𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀭𑁂𑀡𑀼 𑀦𑀸𑀫𑀸𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(66.)

1935\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁧𑁬𑁬𑁮)_  


1936\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀭𑁂𑀡𑀼𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1937\. 𑀭𑁂𑀡𑀼𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

1938\. 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕𑁄 𑀏𑀓𑀸𑀤𑀲𑀫𑁄𑁇



---

1939\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1940\. _𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀻 𑀜𑀸𑀡𑀲𑀜𑁆𑀜𑀻,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑁄 𑀧𑀤𑀧𑀽𑀚𑀓𑁄;_  
_𑀫𑀼𑀝𑁆𑀞𑀺𑀧𑀼𑀧𑁆𑀨𑀻 𑀉𑀤𑀓𑀤𑁄,_  
_𑀦𑀍𑀅𑀫𑀸𑀮𑀺 𑀉𑀧𑀝𑁆𑀞𑀓𑁄;_  
_𑀩𑀺𑀍𑀆𑀮𑀺𑀤𑀸𑀬𑀻 𑀭𑁂𑀡𑀼 𑀘,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀙 𑀘 𑀲𑀝𑁆𑀞𑀺 𑀘𑁇_
