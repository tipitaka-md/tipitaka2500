# 𑁧𑁧𑁇𑁩 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕

### 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(13.)

1868\. _“𑀢𑀺𑀯𑀭𑀸𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑀻𑀳𑀁,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀫𑀸𑀮𑀺𑀓𑁄 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀮𑁄𑀓𑀧𑀽𑀚𑀺𑀢𑀁𑁇_  


(14.)

1869\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀼𑀧𑁆𑀨𑀳𑀢𑁆𑀣𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸𑁇_  


(15.)

1870\. _𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀨𑀮𑀁 𑀇𑀝𑁆𑀞𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀓𑀢𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀮𑁆𑀮𑁂𑀳𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(16.)

1871\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(17.)

1872\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀼𑀧𑀸𑀤𑀸𑀬,_  
_𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀓𑀁;_  
_𑀧𑀜𑁆𑀘𑀭𑀸𑀚𑀲𑀢𑀸 𑀢𑀢𑁆𑀣,_  
_𑀦𑀚𑁆𑀚𑀲𑀫𑀲𑀦𑀸𑀫𑀓𑀸𑁇_  


(18.)

1873\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1874\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1875\. 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
