# 𑁧𑁧𑁇𑁧 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀺𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

1852. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀧𑀯𑀭𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀁,_  
_𑀯𑀦𑀸 𑀦𑀺𑀩𑁆𑀩𑀦𑀫𑀸𑀕𑀢𑀁𑁇_  


(2.)

1853. _𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀫𑀳𑀸𑀯𑀻𑀭𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(3.)

1854. _𑀧𑀤𑁂𑀦𑀸𑀦𑀼𑀧𑀤𑀸𑀬𑀦𑁆𑀢𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀉𑀍𑀆𑀭𑀸 𑀯𑀺𑀢𑁆𑀢𑀺 𑀫𑁂 𑀚𑀸𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑁂𑁇_  


(4.)

1855. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(5.)

1856. _𑀲𑀢𑁆𑀢𑀸𑀲𑀻𑀢𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀫𑀳𑀸𑀭𑁂𑀡𑀼 𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀢𑁆𑀢𑁂𑀢𑁂 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(6.)

1857. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1858. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1859. 𑀪𑀺𑀓𑁆𑀔𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
