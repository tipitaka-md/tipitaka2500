# 𑁩𑁦𑁇𑁪 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(17.)

3380. _“𑀅𑀦𑀼𑀕𑁆𑀕𑀢𑀫𑁆𑀳𑀺 𑀆𑀤𑀺𑀘𑁆𑀘𑁂,_  
_𑀧𑀦𑀸𑀤𑁄 𑀯𑀺𑀧𑀼𑀮𑁄 𑀅𑀳𑀼;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀮𑁄𑀓𑀫𑁆𑀳𑀺,_  
_𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(18.)

3381. _𑀖𑁄𑀲𑀫𑀲𑁆𑀲𑁄𑀲𑀳𑀁 𑀢𑀢𑁆𑀣,_  
_𑀦 𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀁 𑀚𑀺𑀦𑀁;_  
_𑀫𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁𑁇_  


(19.)

3382. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(20.)

3383. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3384. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3385. 𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
