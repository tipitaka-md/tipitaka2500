# 𑁩𑁦𑁇𑁧 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀯𑀕𑁆𑀕

* 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3358\. _“𑀅𑀚𑀺𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀤𑀸;_  
_𑀆𑀳𑀼𑀢𑀺𑀁 𑀬𑀺𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑁄𑀳𑀁,_  
_𑀦𑀸𑀦𑀸𑀧𑀼𑀧𑁆𑀨𑀁 𑀲𑀫𑀸𑀦𑀬𑀺𑀁𑁇_  


(2.)

3359\. _𑀚𑀮𑀦𑁆𑀢𑀁 𑀘𑀺𑀢𑀓𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀺𑀔𑀺𑀦𑁄 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼𑀦𑁄;_  
_𑀢𑀜𑁆𑀘 𑀧𑀼𑀧𑁆𑀨𑀁 𑀲𑀫𑀸𑀦𑁂𑀢𑁆𑀯𑀸,_  
_𑀘𑀺𑀢𑀓𑁂 𑀑𑀓𑀺𑀭𑀺𑀁 𑀅𑀳𑀁𑁇_  


(3.)

3360\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(4.)

3361\. _𑀲𑀢𑁆𑀢𑀯𑀻𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑁆𑀢𑀸𑀲𑀼𑀁 𑀫𑀦𑀼𑀚𑀸𑀥𑀺𑀧𑀸;_  
_𑀲𑀼𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀦𑀸𑀫𑀸 𑀢𑁂,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(5.)

3362\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3363\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3364\. 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
