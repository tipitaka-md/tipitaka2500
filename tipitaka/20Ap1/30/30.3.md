# 𑁩𑁦𑁇𑁩 𑀙𑀢𑁆𑀢𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀯𑀕𑁆𑀕

* 𑀙𑀢𑁆𑀢𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(11.)

3372\. _“𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀫 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀓𑀸𑀲𑀸𑀬𑀯𑀲𑀦𑁄 𑀢𑀤𑀸;_  
_𑀲𑁄 𑀘 𑀩𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀮𑁄𑀓𑀧𑀽𑀚𑀺𑀢𑁄𑁇_  


(12.)

3373\. _𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑁄 𑀲𑀓𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀅𑀕𑀫𑀁 𑀧𑀘𑁆𑀙𑀢𑁄 𑀅𑀳𑀁;_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀘𑀺𑀢𑀓𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁𑁇_  


(13.)

3374\. _𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀢𑀢𑁆𑀣,_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀘𑀺𑀢𑀓𑀁 𑀅𑀳𑀁;_  
_𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀆𑀭𑁄𑀧𑁂𑀲𑀺𑀁 𑀅𑀳𑀁 𑀢𑀤𑀸𑁇_  


(14.)

3375\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀙𑀢𑁆𑀢𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀙𑀢𑁆𑀢𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(15.)

3376\. _𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑁆𑀢 𑀆𑀲𑀼𑀁 𑀚𑀦𑀸𑀥𑀺𑀧𑀸;_  
_𑀫𑀳𑀸𑀭𑀳𑀲𑀦𑀸𑀫𑀸 𑀢𑁂,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(16.)

3377\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3378\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀙𑀢𑁆𑀢𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3379\. 𑀙𑀢𑁆𑀢𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
