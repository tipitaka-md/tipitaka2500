# 𑁩𑁦𑁇𑁭 𑀤𑁂𑀲𑀓𑀺𑀢𑁆𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀯𑀕𑁆𑀕

### 𑀤𑁂𑀲𑀓𑀺𑀢𑁆𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(31.)

3400. _“𑀉𑀧𑀲𑀸𑀮𑀓𑀦𑀸𑀫𑁄𑀳𑀁,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀤𑀸;_  
_𑀓𑀸𑀦𑀦𑀁 𑀯𑀦𑀫𑁄𑀕𑀸𑀍𑀳𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(32.)

3401. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀯𑀦𑁆𑀤𑀺𑀁 𑀧𑀸𑀤𑁂𑀲𑀼,_  
_𑀮𑁄𑀓𑀸𑀳𑀼𑀢𑀺𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀣𑁇_  


(33.)

3402. _𑀓𑀸𑀦𑀦𑀸 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀢𑀁 𑀤𑁂𑀲𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀳𑀁𑁇_  


(34.)

3403. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑁂𑀲𑀫𑀪𑀺𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑀺𑀢𑁆𑀢𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(35.)

3404. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3405. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑁂𑀲𑀓𑀺𑀢𑁆𑀢𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3406. 𑀤𑁂𑀲𑀓𑀺𑀢𑁆𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
