# 𑁩𑁦𑁇𑁨 𑀧𑀼𑀧𑁆𑀨𑀥𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀧𑁆𑀨𑀥𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(6.)

3365\. _“𑀯𑀸𑀓𑀘𑀻𑀭𑀥𑀭𑁄 𑀆𑀲𑀺𑀁,_  
_𑀅𑀚𑀺𑀦𑀼𑀢𑁆𑀢𑀭𑀯𑀸𑀲𑀦𑁄;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀧𑀜𑁆𑀘 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸,_  
_𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀓𑁄𑁇_  


(7.)

3366\. _𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀪𑀺𑀕𑀢𑀁 𑀫𑀫𑀁;_  
_𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀥𑀸𑀭𑁂𑀲𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(8.)

3367\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀥𑀸𑀭𑀡𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(9.)

3368\. _𑀲𑀢𑁆𑀢𑀸𑀲𑀻𑀢𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_𑀲𑀫𑀦𑁆𑀢𑀥𑀸𑀭𑀡𑁄 𑀦𑀸𑀫,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(10.)

3369\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3370\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀧𑁆𑀨𑀥𑀸𑀭𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3371\. 𑀧𑀼𑀧𑁆𑀨𑀥𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
