# 𑁩𑁦𑁇𑁧𑁦 𑀅𑀦𑀼𑀲𑀁𑀲𑀸𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀼𑀲𑀁𑀲𑀸𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(43.)

3418\. _“𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑀸𑀳𑀁,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀫𑀤𑁆𑀤𑀲𑀁 𑀚𑀺𑀦𑀁;_  
_𑀉𑀍𑀉𑀗𑁆𑀕𑀪𑀺𑀓𑁆𑀔𑀁 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(44.)

3419\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀦𑀼𑀲𑀁𑀲𑀸𑀯𑀬𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(45.)

3420\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀦𑀼𑀲𑀁𑀲𑀸𑀯𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀦𑀼𑀲𑀁𑀲𑀸𑀯𑀦𑀸 𑀨𑀮𑀁𑁇_  


(46.)

3421\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁭𑁧𑁨)_  


3422\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀲𑀁𑀲𑀸𑀯𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3423\. 𑀅𑀦𑀼𑀲𑀁𑀲𑀸𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇

  
**𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀯𑀕𑁆𑀕𑁄 𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁄𑁇**



---

3424\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3425\. _𑀘𑀺𑀢𑀓𑀁 𑀧𑀸𑀭𑀺𑀙𑀢𑁆𑀢𑁄 𑀘,_  
_𑀲𑀤𑁆𑀤𑀕𑁄𑀲𑀻𑀲𑀲𑀦𑁆𑀣𑀭𑀁;_  
_𑀧𑀸𑀤𑁄 𑀧𑀤𑁂𑀲𑀁 𑀲𑀭𑀡𑀁,_  
_𑀅𑀫𑁆𑀩𑁄 𑀲𑀁𑀲𑀸𑀯𑀓𑁄𑀧𑀺 𑀘;_  
_𑀅𑀝𑁆𑀞𑀢𑀸𑀮𑀻𑀲 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_  


---

3426\. 𑀅𑀣 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3427\. _𑀓𑀡𑀺𑀓𑀸𑀭𑁄 𑀳𑀢𑁆𑀣𑀺𑀤𑀤𑁄,_  
_𑀆𑀮𑀫𑁆𑀩𑀡𑀼𑀤𑀓𑀸𑀲𑀦𑀁;_  
_𑀢𑀼𑀯𑀭𑀁 𑀣𑁄𑀫𑀓𑁄 𑀘𑁂𑀯,_  
_𑀉𑀓𑁆𑀔𑁂𑀧𑀁 𑀲𑀻𑀲𑀼𑀧𑀥𑀸𑀦𑀁𑁇_  


3428\. _𑀧𑀡𑁆𑀡𑀤𑁄 𑀘𑀺𑀢𑀧𑀽𑀚𑀻 𑀘,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀘𑁂𑀯 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀲𑀢𑀸𑀦𑀻𑀳,_  
_𑀏𑀓𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁂𑀯 𑀘𑁇_  


3429\. _𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀲𑀢𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀤𑁆𑀯𑀸𑀲𑀢𑁆𑀢𑀢𑀺 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺;_  
_𑀢𑀺𑀲𑀢𑀁 𑀅𑀧𑀤𑀸𑀦𑀸𑀦𑀁,_  
_𑀕𑀡𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀪𑀺𑁇_  


---

3430\. 𑀓𑀡𑀺𑀓𑀸𑀭𑀯𑀕𑁆𑀕𑀤𑀲𑀓𑀁𑁇



---

3431\. 𑀢𑀢𑀺𑀬𑀲𑀢𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
