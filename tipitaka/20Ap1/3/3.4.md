# 𑁩𑁇𑁪 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀲𑀫𑀸𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀲𑀫𑀸𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(134.)

999\. _“𑀦𑀕𑀭𑁂 𑀘𑀦𑁆𑀤𑀯𑀢𑀺𑀬𑀸,_  
_𑀪𑀢𑀓𑁄 𑀆𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀧𑀭𑀓𑀫𑁆𑀫𑀸𑀬𑀦𑁂 𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀦 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(135.)

1000\. _𑀫𑀳𑀦𑁆𑀥𑀓𑀸𑀭𑀧𑀺𑀳𑀺𑀢𑀸,_  
_𑀢𑀺𑀯𑀺𑀥𑀕𑁆𑀕𑀻𑀳𑀺 𑀟𑀬𑁆𑀳𑀭𑁂;_  
_𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀉𑀧𑀸𑀬𑁂𑀦,_  
_𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄 𑀪𑀯𑁂 𑀅𑀳𑀁𑁇_  


(136.)

1001\. _𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄 𑀘 𑀫𑁂 𑀦𑀢𑁆𑀣𑀺,_  
_𑀯𑀭𑀸𑀓𑁄 𑀪𑀢𑀓𑁄 𑀅𑀳𑀁;_  
_𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀁,_  
_𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑀬𑀁𑁇_  


(137.)

1002\. _𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀦𑀺𑀲𑀪𑁄 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀫𑀳𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀕𑁆𑀕𑀳𑀺𑀁𑁇_  


(138.)

1003\. _𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸𑀦𑀺 𑀕𑁄𑀧𑀬𑀺𑀁𑁇_  


(139.)

1004\. _𑀫𑀘𑁆𑀘𑀼𑀓𑀸𑀮𑁂 𑀘 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀤𑁂𑀯𑀸 𑀅𑀲𑁆𑀲𑀸𑀲𑀬𑀦𑁆𑀢𑀺 𑀫𑀁;_  
_‘𑀭𑀣𑁄 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑁄 𑀢𑁂,_  
_𑀫𑀸𑀭𑀺𑀲𑀸𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁄’𑁇_  


(140.)

1005\. _𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑁂 𑀘𑀭𑀺𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑁂,_  
_𑀫𑀫 𑀲𑀻𑀮𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀁 𑀅𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(141.)

1006\. _𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀪𑀯𑀺𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(142.)

1007\. _𑀧𑀜𑁆𑀘𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(143.)

1008\. _𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀧𑀼𑀭𑁂 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀚𑀸𑀢𑁄,_  
_𑀫𑀳𑀸𑀲𑀸𑀮𑁂 𑀲𑀼𑀅𑀟𑁆𑀠𑀓𑁂𑁇_  


(144.)

1009\. _𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑁂 𑀓𑀸𑀮𑁂,_  
_𑀤𑀺𑀧𑁆𑀧𑀦𑁆𑀢𑁂 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂;_  
_𑀫𑀸𑀢𑀸 𑀘 𑀫𑁂 𑀧𑀺𑀢𑀸 𑀘𑁂𑀯,_  
_𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀕𑁆𑀕𑀳𑀼𑀁𑁇_  


(145.)

1010\. _𑀲𑀳 𑀲𑀼𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀲𑀻𑀮𑀁,_  
_𑀫𑀫 𑀲𑀻𑀮𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀏𑀓𑀸𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(146.)

1011\. _**𑀚𑀸𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁂𑀦**,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁;_  
_𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑀬𑀺 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀼𑀡𑀫𑀜𑁆𑀜𑀸𑀬 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(147.)

1012\. _𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸𑀦𑀺 𑀕𑁄𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀜𑁆𑀘𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀳𑀁;_  
_𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦 𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(148.)

1013\. _𑀲𑁆𑀯𑀸𑀳𑀁 𑀬𑀲𑀫𑀦𑀼𑀪𑀯𑀺𑀁,_  
_𑀢𑁂𑀲𑀁 𑀲𑀻𑀮𑀸𑀦 𑀯𑀸𑀳𑀲𑀸;_  
_𑀓𑀧𑁆𑀧𑀓𑁄𑀝𑀺𑀫𑁆𑀧𑀺 𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀓𑀺𑀢𑁆𑀢𑀬𑁂 𑀏𑀓𑀤𑁂𑀲𑀓𑀁𑁇_  


(149.)

1014\. _𑀧𑀜𑁆𑀘 𑀲𑀻𑀮𑀸𑀦𑀺 𑀕𑁄𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀬𑁄 𑀳𑁂𑀢𑀽 𑀮𑀪𑀸𑀫𑀳𑀁;_  
_𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑁄 𑀫𑀳𑀸𑀪𑁄𑀕𑁄,_  
_𑀢𑀺𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑁄 𑀪𑀯𑀸𑀫𑀳𑀁𑁇_  


(150.)

1015\. _𑀲𑀗𑁆𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄 𑀘 𑀲𑀩𑁆𑀩𑁂𑀲𑀁,_  
_𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑁄𑀭𑀺𑀲𑀁;_  
_𑀪𑀯𑀸𑀪𑀯𑁂 𑀲𑀁𑀲𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀏𑀢𑁂 𑀞𑀸𑀦𑁂 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(151.)

1016\. _𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑀲𑀻𑀮𑁂𑀲𑀼,_  
_𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀸 𑀚𑀺𑀦𑀲𑀸𑀯𑀓𑀸;_  
_𑀪𑀯𑁂𑀲𑀼 𑀬𑀤𑀺 𑀭𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀓𑀻𑀤𑀺𑀲𑁄 𑀪𑀯𑁂𑁇_  


(152.)

1017\. _𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀁 𑀫𑁂 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀁,_  
_𑀪𑀢𑀓𑁂𑀦 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀸;_  
_𑀢𑁂𑀦 𑀲𑀻𑀮𑁂𑀦𑀳𑀁 𑀅𑀚𑁆𑀚,_  
_𑀫𑁄𑀘𑀬𑀺𑀁 𑀲𑀩𑁆𑀩𑀩𑀦𑁆𑀥𑀦𑀸𑁇_  


(153.)

1018\. _𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀜𑁆𑀘 𑀲𑀻𑀮𑀸𑀦𑀺 𑀕𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀸𑀦𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(154.)

1019\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1020\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀲𑀫𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1021\. 𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑀲𑀫𑀸𑀤𑀸𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
