# 𑁩𑁇𑁮 𑀉𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀉𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(169.)

1042\. _“𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀸𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀲𑀕𑁄𑀘𑀭𑀧𑁆𑀧𑀲𑀼𑀢𑁄𑀳𑀁,_  
_𑀦𑀤𑀻𑀢𑀺𑀢𑁆𑀣𑀁 𑀅𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(170.)

1043\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀦𑀤𑀺𑀁 𑀢𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀦𑀤𑀻𑀢𑀺𑀢𑁆𑀣𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(171.)

1044\. _𑀉𑀧𑀸𑀕𑀢𑁂 𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂,_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀢𑁆𑀣𑀼𑀧𑀸𑀕𑀫𑀺𑀁;_  
_𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀇𑀫𑀁 𑀯𑀸𑀘𑀁 𑀉𑀤𑀻𑀭𑀬𑀺𑀁𑁇_  


(172.)

1045\. _‘𑀅𑀪𑀺𑀭𑀽𑀳 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀢𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀳𑀁 𑀢𑀼𑀯𑀁;_  
_𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀁 𑀯𑀺𑀲𑀬𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧 𑀫𑀳𑀸𑀫𑀼𑀦𑀺’𑁇_  


(173.)

1046\. _𑀫𑀫 𑀉𑀕𑁆𑀕𑀚𑁆𑀚𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀅𑀪𑀺𑀭𑀽𑀳𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀳𑀝𑁆𑀞𑁄 𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀢𑀸𑀭𑁂𑀲𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(174.)

1047\. _𑀦𑀤𑀺𑀬𑀸 𑀧𑀸𑀭𑀺𑀫𑁂 𑀢𑀻𑀭𑁂,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀅𑀲𑁆𑀲𑀸𑀲𑁂𑀲𑀺 𑀫𑀫𑀁 𑀢𑀢𑁆𑀣,_  
_𑀅𑀫𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(175.)

1048\. _𑀢𑀫𑁆𑀳𑀸 𑀓𑀸𑀬𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑀁 𑀆𑀕𑀘𑁆𑀙𑀳𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀪𑀯𑀺𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(176.)

1049\. _𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀲𑀳𑀁;_  
_𑀢𑀻𑀡𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻,_  
_𑀫𑀳𑀺𑀬𑀸 𑀇𑀲𑁆𑀲𑀭𑁄 𑀅𑀳𑀼𑀁𑁇_  


(177.)

1050\. _𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑀳𑀁,_  
_𑀦𑀺𑀧𑀓𑁄 𑀘 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑁄;_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(178.)

1051\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀢𑀸𑀭𑁂𑀲𑀺𑀁 𑀬𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀢𑀭𑀡𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(179.)

1052\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1053\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1054\. 𑀉𑀢𑁆𑀢𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
