# 𑁩𑁇𑁯 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(180.)

1055\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀡𑁂;_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀕𑁆𑀕𑀁,_  
_𑀦𑀭𑀯𑀭𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑀁𑁇_  


(181.)

1056\. _𑀅𑀤𑀦𑁆𑀢𑀤𑀫𑀦𑀁 𑀢𑀸𑀤𑀺𑀁,_  
_𑀫𑀳𑀸𑀯𑀸𑀤𑀺𑀁 𑀫𑀳𑀸𑀫𑀢𑀺𑀁;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀫𑀓𑀸𑀲𑀳𑀁𑁇_  


(182.)

1057\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀫𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(183.)

1058\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1059\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1060\. 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
