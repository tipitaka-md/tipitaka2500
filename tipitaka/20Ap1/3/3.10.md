# 𑁩𑁇𑁧𑁦 𑀔𑁄𑀫𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀔𑁄𑀫𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(184.)

1061\. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀯𑀸𑀡𑀺𑀚𑁄 𑀢𑀤𑀸;_  
_𑀢𑁂𑀦𑁂𑀯 𑀤𑀸𑀭𑀁 𑀧𑁄𑀲𑁂𑀫𑀺,_  
_𑀭𑁄𑀧𑁂𑀫𑀺 𑀩𑀻𑀚𑀲𑀫𑁆𑀧𑀤𑀁𑁇_  


(185.)

1062\. _𑀭𑀣𑀺𑀬𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀏𑀓𑀁 𑀔𑁄𑀫𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀓𑀼𑀲𑀮𑀢𑁆𑀣𑀸𑀬 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(186.)

1063\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀔𑁄𑀫𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀔𑁄𑀫𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(187.)

1064\. _𑀲𑀢𑁆𑀢𑀭𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀲𑀺𑀦𑁆𑀥𑀯𑀲𑀦𑁆𑀥𑀦𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀢𑀼𑀤𑀻𑀧𑀫𑁆𑀳𑀺 𑀇𑀲𑁆𑀲𑀭𑁄𑁇_  


(188.)

1065\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁯𑁯𑁧)_  


1066\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄𑀫𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1067\. 𑀔𑁄𑀫𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

1068\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1069\. _𑀲𑀼𑀪𑀽𑀢𑀺 𑀉𑀧𑀯𑀸𑀦𑁄 𑀘,_  
_𑀲𑀭𑀡𑁄 𑀲𑀻𑀮𑀕𑀸𑀳𑀓𑁄;_  
_𑀅𑀦𑁆𑀦𑀲𑀁𑀲𑀸𑀯𑀓𑁄 𑀥𑀽𑀧𑁄,_  
_𑀧𑀼𑀮𑀺𑀦𑁄 𑀉𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀘𑁇_  


1070\. _𑀅𑀜𑁆𑀚𑀮𑀻 𑀔𑁄𑀫𑀤𑀸𑀬𑀻 𑀘,_  
_𑀤𑀲𑁂𑀯 𑀢𑀢𑀺𑀬𑁂 𑀕𑀡𑁂;_  
_𑀧𑀜𑁆𑀘𑀸𑀲𑀻𑀢𑀺𑀲𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀲𑀩𑁆𑀩𑀧𑀺𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


---

1071\. 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

1072\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
