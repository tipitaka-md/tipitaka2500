# 𑁩𑁇𑁬 𑀥𑀽𑀧𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀥𑀽𑀧𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(161.)

1030\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀓𑀼𑀝𑀺𑀥𑀽𑀧𑀁 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(162.)

1031\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀧𑀺 𑀧𑀺𑀬𑁄 𑀳𑁄𑀫𑀺,_  
_𑀥𑀽𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(163.)

1032\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀥𑀽𑀧𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀥𑀽𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(164.)

1033\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1034\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑀽𑀧𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1035\. 𑀥𑀽𑀧𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
