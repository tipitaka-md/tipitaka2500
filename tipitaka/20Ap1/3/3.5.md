# 𑁩𑁇𑁫 𑀅𑀦𑁆𑀦𑀲𑀁𑀲𑀸𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀦𑁆𑀦𑀲𑀁𑀲𑀸𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(155.)

1022\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀡𑁂;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀁𑀓𑀸𑀲𑀁,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑀁𑁇_  


(156.)

1023\. _𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀁 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑁄𑀧𑀫𑀁;_  
_𑀅𑀮𑀢𑁆𑀣𑀁 𑀧𑀭𑀫𑀁 𑀧𑀻𑀢𑀺𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀦𑁆𑀢𑀁 𑀚𑀼𑀢𑀺𑀦𑁆𑀥𑀭𑀁𑁇_  


(157.)

1024\. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀦𑀸𑀫𑁂𑀢𑁆𑀯𑀸,_  
_𑀪𑁄𑀚𑀬𑀺𑀁 𑀢𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁;_  
_𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀮𑁄𑀓𑁂,_  
_𑀅𑀦𑀼𑀫𑁄𑀤𑀺 𑀫𑀫𑀁 𑀢𑀤𑀸𑁇_  


(158.)

1025\. _𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁂,_  
_𑀧𑀭𑀫𑀲𑁆𑀲𑀸𑀲𑀓𑀸𑀭𑀓𑁂;_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀳𑀁𑁇_  


(159.)

1026\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(160.)

1027\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1028\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑁆𑀦𑀲𑀁𑀲𑀸𑀯𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1029\. 𑀅𑀦𑁆𑀦𑀲𑀁𑀲𑀸𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
