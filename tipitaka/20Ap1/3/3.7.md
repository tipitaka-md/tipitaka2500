# 𑁩𑁇𑁭 𑀧𑀼𑀮𑀺𑀦𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀮𑀺𑀦𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(165.)

1036\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀩𑁄𑀥𑀺𑀬𑀸 𑀧𑀸𑀤𑀧𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀼𑀭𑀸𑀡𑀧𑀼𑀮𑀺𑀦𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀤𑁆𑀥𑀧𑀼𑀮𑀺𑀦𑀫𑀸𑀓𑀺𑀭𑀺𑀁𑁇_  


(166.)

1037\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀮𑀺𑀦𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀮𑀺𑀦𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(167.)

1038\. _𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺𑀁 𑀚𑀦𑀸𑀥𑀺𑀪𑀽;_  
_𑀫𑀳𑀸𑀧𑀼𑀮𑀺𑀦𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(168.)

1039\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1040\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀮𑀺𑀦𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1041\. 𑀧𑀼𑀮𑀺𑀦𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
