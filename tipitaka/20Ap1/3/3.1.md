# 𑁩𑁇𑁧 𑀲𑀼𑀪𑀽𑀢𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀪𑀽𑀢𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

860\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀦𑀺𑀲𑀪𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀅𑀲𑁆𑀲𑀫𑁄 𑀲𑀼𑀓𑀢𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀸 𑀲𑀼𑀫𑀸𑀧𑀺𑀢𑀸𑁇_  


(2.)

861\. _𑀓𑁄𑀲𑀺𑀬𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀚𑀝𑀺𑀮𑁄 𑀉𑀕𑁆𑀕𑀢𑀸𑀧𑀦𑁄;_  
_𑀏𑀓𑀸𑀓𑀺𑀬𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄,_  
_𑀯𑀲𑀸𑀫𑀺 𑀦𑀺𑀲𑀪𑁂 𑀢𑀤𑀸𑁇_  


(3.)

862\. _𑀨𑀮𑀁 𑀫𑀽𑀮𑀜𑁆𑀘 𑀧𑀡𑁆𑀡𑀜𑁆𑀘,_  
_𑀦 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀧𑀯𑀢𑁆𑀢𑀁𑀯 𑀲𑀼𑀧𑀸𑀢𑀸𑀳𑀁,_  
_𑀉𑀧𑀚𑀻𑀯𑀸𑀫𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(4.)

863\. _𑀦𑀸𑀳𑀁 𑀓𑁄𑀧𑁂𑀫𑀺 𑀆𑀚𑀻𑀯𑀁,_  
_𑀘𑀚𑀫𑀸𑀦𑁄𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀆𑀭𑀸𑀥𑁂𑀫𑀺 𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀯𑀺𑀯𑀚𑁆𑀚𑁂𑀫𑀺 𑀅𑀦𑁂𑀲𑀦𑀁𑁇_  


(5.)

864\. _𑀭𑀸𑀕𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀬𑀤𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑁂 𑀫𑀫;_  
_𑀲𑀬𑀁𑀯 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀏𑀓𑀕𑁆𑀕𑁄 𑀢𑀁 𑀤𑀫𑁂𑀫𑀳𑀁𑁇_  


(6.)

865\. _‘𑀭𑀚𑁆𑀚𑀲𑁂 𑀭𑀚𑁆𑀚𑀦𑀻𑀬𑁂 𑀘,_  
_𑀤𑀼𑀲𑁆𑀲𑀦𑀻𑀬𑁂 𑀘 𑀤𑀼𑀲𑁆𑀲𑀲𑁂;_  
_𑀫𑀼𑀬𑁆𑀳𑀲𑁂 𑀫𑁄𑀳𑀦𑀻𑀬𑁂 𑀘,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀲𑁆𑀲𑀼 𑀯𑀦𑀸 𑀢𑀼𑀯𑀁𑁇_  


(7.)

866\. _𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀬𑀁 𑀯𑀸𑀲𑁄,_  
_𑀦𑀺𑀫𑁆𑀫𑀮𑀸𑀦𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀁;_  
_𑀫𑀸 𑀔𑁄 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀤𑀽𑀲𑁂𑀲𑀺,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀲𑁆𑀲𑀼 𑀯𑀦𑀸 𑀢𑀼𑀯𑀁𑁇_  


(8.)

867\. _𑀅𑀕𑀸𑀭𑀺𑀓𑁄 𑀪𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀬𑀤𑀸 𑀧𑀼𑀢𑁆𑀢𑀁 𑀮𑀪𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀉𑀪𑁄𑀧𑀺 𑀫𑀸 𑀯𑀺𑀭𑀸𑀥𑁂𑀲𑀺,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀲𑁆𑀲𑀼 𑀯𑀦𑀸 𑀢𑀼𑀯𑀁𑁇_  


(9.)

868\. _𑀙𑀯𑀸𑀮𑀸𑀢𑀁 𑀬𑀣𑀸 𑀓𑀝𑁆𑀞𑀁,_  
_𑀦 𑀓𑁆𑀯𑀘𑀺 𑀓𑀺𑀘𑁆𑀘𑀓𑀸𑀭𑀓𑀁;_  
_𑀦𑁂𑀯 𑀕𑀸𑀫𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀸,_  
_𑀦 𑀳𑀺 𑀢𑀁 𑀓𑀝𑁆𑀞𑀲𑀫𑁆𑀫𑀢𑀁𑁇_  


(10.)

869\. _𑀙𑀯𑀸𑀮𑀸𑀢𑀽𑀧𑀫𑁄 𑀢𑁆𑀯𑀁𑀲𑀺,_  
_𑀦 𑀕𑀺𑀳𑀻 𑀦𑀸𑀧𑀺 𑀲𑀜𑁆𑀜𑀢𑁄;_  
_𑀉𑀪𑀢𑁄 𑀫𑀼𑀢𑁆𑀢𑀓𑁄 𑀅𑀚𑁆𑀚,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀲𑁆𑀲𑀼 𑀯𑀦𑀸 𑀢𑀼𑀯𑀁𑁇_  


(11.)

870\. _𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄 𑀢𑀯 𑀏𑀢𑀁,_  
_𑀓𑁄 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑁂 𑀇𑀤𑀁;_  
_𑀲𑀤𑁆𑀥𑀸𑀥𑀼𑀭𑀁 𑀯𑀳𑀺𑀲𑀺 𑀫𑁂,_  
_𑀓𑁄𑀲𑀚𑁆𑀚𑀩𑀳𑀼𑀮𑀸𑀬 𑀘𑁇_  


(12.)

871\. _𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀽,_  
_𑀅𑀲𑀼𑀘𑀺𑀁 𑀦𑀸𑀕𑀭𑀺𑀓𑁄 𑀬𑀣𑀸;_  
_𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸𑀦 𑀇𑀲𑀬𑁄,_  
_𑀘𑁄𑀤𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀤𑀸𑁇_  


(13.)

872\. _𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀯𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀸𑀲𑀦𑀁;_  
_𑀲𑀁𑀯𑀸𑀲𑀁 𑀅𑀮𑀪𑀦𑁆𑀢𑁄 𑀳𑀺,_  
_𑀓𑀣𑀁 𑀚𑀻𑀯𑀺𑀳𑀺𑀲𑀺 𑀢𑀼𑀯𑀁𑁇_  


(14.)

873\. _𑀢𑀺𑀥𑀸 𑀧𑀪𑀺𑀦𑁆𑀦𑀁 𑀫𑀸𑀢𑀗𑁆𑀕𑀁,_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀁 𑀲𑀝𑁆𑀞𑀺𑀳𑀸𑀬𑀦𑀁;_  
_𑀩𑀮𑀻 𑀦𑀸𑀕𑁄 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀬𑀽𑀣𑀸 𑀦𑀻𑀳𑀭𑀢𑁂 𑀕𑀚𑀁𑁇_  


(15.)

874\. _𑀬𑀽𑀣𑀸 𑀯𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀔𑀁 𑀲𑀸𑀢𑀁 𑀦 𑀯𑀺𑀦𑁆𑀤𑀢𑀺;_  
_𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀯𑀺𑀫𑀦𑁄 𑀳𑁄𑀢𑀺,_  
_𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑁄 𑀧𑀯𑁂𑀥𑀢𑀺𑁇_  


(16.)

875\. _𑀢𑀣𑁂𑀯 𑀚𑀝𑀺𑀮𑀸 𑀢𑀫𑁆𑀧𑀺,_  
_𑀦𑀻𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀼𑀫𑁆𑀫𑀢𑀺𑀁;_  
_𑀢𑁂𑀳𑀺 𑀢𑁆𑀯𑀁 𑀦𑀺𑀲𑁆𑀲𑀝𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀔𑀁 𑀲𑀸𑀢𑀁 𑀦 𑀮𑀘𑁆𑀙𑀲𑀺𑁇_  


(17.)

876\. _𑀤𑀺𑀯𑀸 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄;_  
_𑀟𑀬𑁆𑀳𑀲𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑁂𑀦,_  
_𑀕𑀚𑁄 𑀬𑀽𑀣𑀸𑀯 𑀦𑀺𑀲𑁆𑀲𑀝𑁄𑁇_  


(18.)

877\. _𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀬𑀣𑀸 𑀓𑀽𑀝𑀁,_  
_𑀦𑁂𑀯 𑀛𑀸𑀬𑀢𑀺 𑀓𑀢𑁆𑀣𑀘𑀺;_  
_𑀢𑀣𑀸 𑀲𑀻𑀮𑀯𑀺𑀳𑀻𑀦𑁄 𑀢𑁆𑀯𑀁,_  
_𑀦 𑀛𑀸𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀓𑀢𑁆𑀣𑀘𑀺𑁇_  


(19.)

878\. _𑀅𑀕𑀸𑀭𑀁 𑀯𑀲𑀫𑀸𑀦𑁄𑀧𑀺,_  
_𑀓𑀣𑀁 𑀚𑀻𑀯𑀺𑀳𑀺𑀲𑀺 𑀢𑀼𑀯𑀁;_  
_𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀦𑀢𑁆𑀣𑀺 𑀢𑁂 𑀦𑀺𑀳𑀺𑀢𑀁 𑀥𑀦𑀁𑁇_  


(20.)

879\. _𑀲𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑀢𑁆𑀢𑁂 𑀲𑁂𑀤𑀁 𑀧𑀫𑁄𑀘𑀬𑀁;_  
_𑀏𑀯𑀁 𑀚𑀻𑀯𑀺𑀳𑀺𑀲𑀺 𑀕𑁂𑀳𑁂,_  
_𑀲𑀸𑀥𑀼 𑀢𑁂 𑀢𑀁 𑀦 𑀭𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(21.)

880\. _𑀏𑀯𑀸𑀳𑀁 𑀢𑀢𑁆𑀣 𑀯𑀸𑀭𑁂𑀫𑀺,_  
_𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀕𑀢𑀁 𑀫𑀦𑀁;_  
_𑀦𑀸𑀦𑀸𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀸𑀧𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀯𑀸𑀭𑀬𑀺𑀁’𑁇_  


(22.)

881\. _𑀏𑀯𑀁 𑀫𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄;_  
_𑀢𑀺𑀁𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀺𑀧𑀺𑀦𑁂 𑀫𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀼𑀁𑁇_  


(23.)

882\. _𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀭𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀁 𑀕𑀯𑁂𑀲𑀓𑀁;_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(24.)

883\. _𑀢𑀺𑀫𑁆𑀩𑀭𑀽𑀲𑀓𑀯𑀡𑁆𑀡𑀸𑀪𑁄,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀅𑀦𑀽𑀧𑀫𑁄;_  
_𑀭𑀽𑀧𑁂𑀦𑀸𑀲𑀤𑀺𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀆𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀻 𑀢𑀤𑀸𑁇_  


(25.)

884\. _𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑁄 𑀲𑀸𑀮𑀭𑀸𑀚𑀸𑀯,_  
_𑀯𑀺𑀚𑁆𑀚𑀽𑀯𑀩𑁆𑀪𑀖𑀦𑀦𑁆𑀢𑀭𑁂;_  
_𑀜𑀸𑀡𑁂𑀦𑀸𑀲𑀤𑀺𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀆𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀻 𑀢𑀤𑀸𑁇_  


(26.)

885\. _𑀲𑀻𑀳𑀭𑀸𑀚𑀸𑀯𑀲𑀫𑁆𑀪𑀻𑀢𑁄,_  
_𑀕𑀚𑀭𑀸𑀚𑀸𑀯 𑀤𑀧𑁆𑀧𑀺𑀢𑁄;_  
_𑀮𑀸𑀲𑀺𑀢𑁄 𑀩𑁆𑀬𑀕𑁆𑀖𑀭𑀸𑀚𑀸𑀯,_  
_𑀆𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀻 𑀢𑀤𑀸𑁇_  


(27.)

886\. _𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑀯𑀡𑁆𑀡𑀸𑀪𑁄,_  
_𑀔𑀤𑀺𑀭𑀗𑁆𑀕𑀸𑀭𑀲𑀦𑁆𑀦𑀺𑀪𑁄;_  
_𑀫𑀡𑀺 𑀬𑀣𑀸 𑀚𑁄𑀢𑀺𑀭𑀲𑁄,_  
_𑀆𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀻 𑀢𑀤𑀸𑁇_  


(28.)

887\. _𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑁂𑀮𑀸𑀲𑀦𑀺𑀪𑁄,_  
_𑀧𑀼𑀡𑁆𑀡𑀫𑀸𑀬𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸;_  
_𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂𑀯 𑀲𑀽𑀭𑀺𑀬𑁄,_  
_𑀆𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀻 𑀢𑀤𑀸𑁇_  


(29.)

888\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀪𑁂 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀁,_  
_𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_‘𑀤𑁂𑀯𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑁄,_  
_𑀉𑀤𑀸𑀳𑀼 𑀫𑀦𑀼𑀚𑁄 𑀅𑀬𑀁𑁇_  


(30.)

889\. _𑀦 𑀫𑁂 𑀲𑀼𑀢𑁄 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀯𑀸,_  
_𑀫𑀳𑀺𑀬𑀸 𑀏𑀤𑀺𑀲𑁄 𑀦𑀭𑁄;_  
_𑀅𑀧𑀺 𑀫𑀦𑁆𑀢𑀧𑀤𑀁 𑀅𑀢𑁆𑀣𑀺,_  
_𑀅𑀬𑀁 𑀲𑀢𑁆𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(31.)

890\. _𑀏𑀯𑀸𑀳𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀦𑀸𑀦𑀸𑀧𑀼𑀧𑁆𑀨𑀜𑁆𑀘 𑀕𑀦𑁆𑀥𑀜𑁆𑀘,_  
_𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑀳𑀁 𑀢𑀤𑀸𑁇_  


(32.)

891\. _𑀧𑀼𑀧𑁆𑀨𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀸𑀥𑀼𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀭𑀫𑀁;_  
_𑀦𑀭𑀲𑀸𑀭𑀣𑀺𑀦𑀁 𑀅𑀕𑁆𑀕𑀁,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(33.)

892\. _‘𑀇𑀤𑀁 𑀫𑁂 𑀆𑀲𑀦𑀁 𑀯𑀻𑀭,_  
_𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀯𑀦𑀼𑀘𑁆𑀙𑀯𑀁;_  
_𑀳𑀸𑀲𑀬𑀦𑁆𑀢𑁄 𑀫𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀦𑀺𑀲𑀻𑀤 𑀓𑀼𑀲𑀼𑀫𑀸𑀲𑀦𑁂’𑁇_  


(34.)

893\. _𑀦𑀺𑀲𑀻𑀤𑀺 𑀢𑀢𑁆𑀣 𑀪𑀕𑀯𑀸,_  
_𑀅𑀲𑀫𑁆𑀪𑀻𑀢𑁄𑀯 𑀓𑁂𑀲𑀭𑀻;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀧𑀯𑀭𑁂 𑀓𑀼𑀲𑀼𑀫𑀸𑀲𑀦𑁂𑁇_  


(35.)

894\. _𑀦𑀫𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀅𑀳𑀁;_  
_𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀫𑀫 𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑁇_  


(36.)

895\. _‘𑀪𑀸𑀯𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀁,_  
_𑀪𑀸𑀯𑀦𑀸𑀦𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀇𑀫𑀁 𑀲𑀢𑀺𑀁 𑀪𑀸𑀯𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀽𑀭𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀫𑀸𑀦𑀲𑀁𑁇_  


(37.)

896\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀅𑀲𑀻𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻,_  
_𑀭𑀸𑀚𑀸 𑀭𑀝𑁆𑀞𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(38.)

897\. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀅𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀲𑀺 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀨𑀮𑀁𑁇_  


(39.)

898\. _𑀪𑀯𑀸𑀪𑀯𑁂 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄,_  
_𑀫𑀳𑀸𑀪𑁄𑀕𑀁 𑀮𑀪𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀪𑁄𑀕𑁂 𑀢𑁂 𑀊𑀦𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀨𑀮𑀁𑁇_  


(40.)

899\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(41.)

900\. _𑀅𑀲𑀻𑀢𑀺𑀓𑁄𑀝𑀺𑀁 𑀙𑀟𑁆𑀟𑁂𑀢𑁆𑀯𑀸,_  
_𑀤𑀸𑀲𑁂 𑀓𑀫𑁆𑀫𑀓𑀭𑁂 𑀩𑀳𑀽;_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀸𑀲𑀦𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(42.)

901\. _𑀆𑀭𑀸𑀥𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑁄𑀢𑀫𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁;_  
_𑀲𑀼𑀪𑀽𑀢𑀺 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄𑁇_  


(43.)

902\. _𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀕𑀼𑀡𑀫𑁆𑀳𑀺 𑀢𑀁;_  
_𑀢𑀣𑀸𑀭𑀡𑀯𑀺𑀳𑀸𑀭𑁂 𑀘,_  
_𑀤𑁆𑀯𑀻𑀲𑀼 𑀅𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑁆𑀲𑀢𑀺’𑁇_  


(44.)

903\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀓𑁄;_  
_𑀦𑀪𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀫𑀻 𑀯𑀻𑀭𑁄,_  
_𑀳𑀁𑀲𑀭𑀸𑀚𑀸𑀯 𑀅𑀫𑁆𑀩𑀭𑁂𑁇_  


(45.)

904\. _𑀲𑀸𑀲𑀺𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀣𑁂𑀦,_  
_𑀦𑀫𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀁;_  
_𑀲𑀤𑀸 𑀪𑀸𑀯𑁂𑀫𑀺 𑀫𑀼𑀤𑀺𑀢𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(46.)

905\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀁 𑀅𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(47.)

906\. _𑀅𑀲𑀻𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁𑁇_  


(48.)

907\. _𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀲𑀼𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀨𑀮𑀁𑁇_  


(49.)

908\. _𑀪𑀯𑀸𑀪𑀯𑁂 𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄,_  
_𑀫𑀳𑀸𑀪𑁄𑀕𑀁 𑀮𑀪𑀸𑀫𑀳𑀁;_  
_𑀪𑁄𑀕𑁂 𑀫𑁂 𑀊𑀦𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀨𑀮𑀁𑁇_  


(50.)

909\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀨𑀮𑀁𑁇_  


(51.)

910\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


911\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀪𑀽𑀢𑀺 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

912\. 𑀲𑀼𑀪𑀽𑀢𑀺𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
