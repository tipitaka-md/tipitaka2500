# 𑁫𑁬𑁇𑁩 𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀬𑀲𑀯𑀕𑁆𑀕

### 𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(35.)

7709\. _“𑀅𑀚𑀺𑀦𑀘𑀫𑁆𑀫𑀯𑀢𑁆𑀣𑁄𑀳𑀁,_  
_𑀔𑀸𑀭𑀺𑀪𑀸𑀭𑀥𑀭𑁄 𑀢𑀤𑀸;_  
_𑀔𑀸𑀭𑀺𑀓𑀁 𑀳𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑁄𑀮𑀁 𑀅𑀳𑀸𑀲𑀺 𑀅𑀲𑁆𑀲𑀫𑀁𑁇_  


(36.)

7710\. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀏𑀓𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀚𑀺𑀦𑁄;_  
_𑀫𑀫𑀲𑁆𑀲𑀫𑀁 𑀉𑀧𑀸𑀕𑀘𑁆𑀙𑀺,_  
_𑀚𑁄𑀢𑁂𑀦𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀺𑀓𑀁𑁇_  


(37.)

7711\. _𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀩𑁆𑀩𑀢𑀁;_  
_𑀉𑀪𑁄 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀓𑁄𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(38.)

7712\. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑁄𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(39.)

7713\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(40.)

7714\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(41.)

7715\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


7716\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

7717\. 𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
