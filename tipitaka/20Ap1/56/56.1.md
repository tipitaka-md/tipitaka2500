# 𑁫𑁬𑁇𑁧 𑀬𑀲𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀬𑀲𑀯𑀕𑁆𑀕

### 𑀬𑀲𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

7671. _“𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀑𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀪𑀯𑀦𑀁 𑀫𑁂 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁;_  
_𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀽𑀧𑀓𑀽𑀚𑀺𑀢𑀸𑁇_  


(2.)

7672. _𑀫𑀦𑁆𑀤𑀸𑀭𑀓𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀓𑁂𑀳𑀺 𑀘;_  
_𑀦𑀤𑀻 𑀘 𑀲𑀦𑁆𑀤𑀢𑁂 𑀢𑀢𑁆𑀣,_  
_𑀲𑀼𑀧𑀢𑀺𑀢𑁆𑀣𑀸 𑀫𑀦𑁄𑀭𑀫𑀸𑁇_  


(3.)

7673. _𑀫𑀘𑁆𑀙𑀓𑀘𑁆𑀙𑀧𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀚𑀲𑀫𑁄𑀢𑁆𑀣𑀝𑀸;_  
_𑀫𑀬𑀽𑀭𑀓𑁄𑀜𑁆𑀘𑀸𑀪𑀺𑀭𑀼𑀤𑀸,_  
_𑀓𑁄𑀓𑀺𑀮𑀸𑀤𑀻𑀳𑀺 𑀯𑀕𑁆𑀕𑀼𑀳𑀺𑁇_  


(4.)

7674. _𑀧𑀸𑀭𑁂𑀯𑀢𑀸 𑀭𑀯𑀺𑀳𑀁𑀲𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀸 𑀦𑀤𑀻𑀘𑀭𑀸;_  
_𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀸 𑀲𑀸𑀮𑀺𑀓𑀸 𑀘𑁂𑀢𑁆𑀣,_  
_𑀧𑀸𑀯𑀓𑀸 𑀚𑀻𑀯𑀁𑀚𑀻𑀯𑀓𑀸𑁇_  


(5.)

7675. _𑀳𑀁𑀲𑀸𑀓𑁄𑀜𑁆𑀘𑀸𑀪𑀺𑀦𑀤𑀺𑀢𑀸,_  
_𑀓𑁄𑀲𑀺𑀬𑀸 𑀧𑀺𑀗𑁆𑀕𑀮𑀸 𑀩𑀳𑀽;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀫𑀡𑀺𑀫𑀼𑀢𑁆𑀢𑀧𑀯𑀸𑀍𑀇𑀓𑀸𑁇_  


(6.)

7676. _𑀲𑀩𑁆𑀩𑁂 𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀸 𑀭𑀼𑀓𑁆𑀔𑀸,_  
_𑀦𑀸𑀦𑀸𑀔𑀦𑁆𑀥𑀲𑀫𑁂𑀭𑀺𑀢𑀸;_  
_𑀉𑀚𑁆𑀚𑁄𑀢𑁂𑀦𑁆𑀢𑀺 𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀪𑀯𑀦𑀁 𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀺𑀓𑀁𑁇_  


(7.)

7677. _𑀲𑀝𑁆𑀞𑀺𑀢𑀼𑀭𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑁄 𑀧𑀯𑀚𑁆𑀚𑀭𑁂;_  
_𑀲𑁄𑀍𑀅𑀲𑀺𑀢𑁆𑀣𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(8.)

7678. _𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫 𑀪𑀯𑀦𑀸,_  
_𑀲𑀼𑀫𑁂𑀥𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀯𑀦𑁆𑀤𑀬𑀺𑀁 𑀢𑀁 𑀫𑀳𑀸𑀬𑀲𑀁𑁇_  


(9.)

7679. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀲𑀁𑀖𑀁 𑀢𑀁 𑀦𑀺𑀫𑀦𑁆𑀢𑀬𑀺𑀁;_  
_𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀲𑁄 𑀥𑀻𑀭𑁄,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(10.)

7680. _𑀫𑀫 𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀲𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀪𑀯𑀦𑀁 𑀫𑁂 𑀉𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(11.)

7681. _𑀆𑀫𑀦𑁆𑀢𑀬𑀺𑀁 𑀧𑀭𑀺𑀚𑀦𑀁,_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀼𑀁 𑀢𑀤𑀸;_  
_‘𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀪𑀯𑀦𑀁 𑀆𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(12.)

7682. _𑀮𑀸𑀪𑀸 𑀅𑀫𑁆𑀳𑀁 𑀲𑀼𑀮𑀤𑁆𑀥𑀸 𑀦𑁄,_  
_𑀬𑁂 𑀯𑀲𑀸𑀫 𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀫𑀬𑀫𑁆𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀧𑀽𑀚𑀬𑀺𑀲𑁆𑀲𑀸𑀫 𑀲𑀢𑁆𑀣𑀼𑀦𑁄’𑁇_  


(13.)

7683. _𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀯𑀲𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀉𑀧𑁂𑀲𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(14.)

7684. _𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁂𑀳𑀺 𑀢𑀼𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀫𑀓𑀸𑀲𑀳𑀁;_  
_𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑁂 𑀧𑀻𑀞𑁂,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(15.)

7685. _𑀉𑀧𑀭𑀺𑀘𑁆𑀙𑀤𑀦𑀁 𑀆𑀲𑀺,_  
_𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀁 𑀢𑀤𑀸;_  
_𑀩𑀻𑀚𑀦𑀻𑀬𑁄 𑀧𑀯𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


(16.)

7686. _𑀧𑀳𑀽𑀢𑁂𑀦𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀅𑀢𑀧𑁆𑀧𑀬𑀺𑀁;_  
_𑀧𑀘𑁆𑀘𑁂𑀓𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀮𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(17.)

7687. _𑀬𑀁 𑀯𑀤𑁂𑀢𑀺 𑀲𑀼𑀫𑁂𑀥𑁄 𑀲𑁄,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(18.)

7688. _‘𑀬𑁄 𑀫𑀁 𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑁂 𑀇𑀫𑁂 𑀘 𑀢𑀧𑁆𑀧𑀬𑀺;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(19.)

7689. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸𑀬𑀁,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(20.)

7690. _𑀉𑀧𑀕𑀘𑁆𑀙𑀢𑀺 𑀬𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀁 𑀢𑀲𑁆𑀲,_  
_𑀙𑀤𑀦𑀁 𑀥𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(21.)

7691. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(22.)

7692. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄𑁇_  


(23.)

7693. _𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀻𑀳𑀦𑀸𑀤𑀁 𑀦𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀘𑀺𑀢𑀓𑁂 𑀙𑀢𑁆𑀢𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀳𑁂𑀝𑁆𑀞𑀸 𑀙𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀟𑀬𑁆𑀳𑀣’𑁇_  


(24.)

7694. _𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀫𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑀸;_  
_𑀫𑀡𑁆𑀟𑀧𑁂 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀸,_  
_𑀲𑀦𑁆𑀢𑀸𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(25.)

7695. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(26.)

7696. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(27.)

7697. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(28.)
