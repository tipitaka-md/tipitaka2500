# 𑁫𑁬𑁇𑁧𑁧 𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀬𑀲𑀯𑀕𑁆𑀕

* 𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(179.)

7869\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀯𑀭𑀦𑀸𑀕𑁄 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁄,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑁄 𑀉𑀭𑀽𑀍𑀳𑀯𑀸𑁇_  


(180.)

7870\. _𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑁄𑀧𑀲𑁄𑀪𑀺𑀢𑁄,_  
_𑀲𑀓𑀧𑁆𑀧𑀦𑁄 𑀲𑀳𑀢𑁆𑀣𑀺𑀧𑁄;_  
_𑀅𑀕𑁆𑀖𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀲𑀁𑀖𑀸𑀭𑀸𑀫𑀁 𑀅𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(181.)

7871\. _𑀘𑀢𑀼𑀧𑀜𑁆𑀜𑀸𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀸𑀲𑀸𑀤𑁂 𑀓𑀸𑀭𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀫𑀳𑁄𑀖𑀤𑀸𑀦𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀲𑀺𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(182.)

7872\. _𑀅𑀦𑀼𑀫𑁄𑀤𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀲𑀩𑁆𑀩𑁂 𑀚𑀦𑁂 𑀳𑀸𑀲𑀬𑀦𑁆𑀢𑁄,_  
_𑀤𑁂𑀲𑁂𑀲𑀺 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁𑁇_  


(183.)

7873\. _𑀢𑀁 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀓𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(184.)

7874\. _‘𑀘𑀢𑀼𑀧𑀜𑁆𑀜𑀸𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀸𑀲𑀸𑀤𑁂 𑀓𑀸𑀭𑀬𑀻 𑀅𑀬𑀁;_  
_𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀺𑀧𑀸𑀓𑀁,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(185.)

7875\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀭𑁂;_  
_𑀩𑁆𑀬𑀫𑁆𑀳𑀼𑀢𑁆𑀢𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀸 𑀘 𑀢𑁂𑁇_  


(186.)

7876\. _𑀧𑀜𑁆𑀜𑀸𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀝𑁆𑀞𑀧𑀜𑁆𑀜𑀸𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(187.)

7877\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(188.)

7878\. _𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀅𑀟𑁆𑀠𑁂 𑀓𑀼𑀮𑁂 𑀫𑀳𑀸𑀪𑁄𑀕𑁂,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(189.)

7879\. _𑀲𑁄 𑀧𑀘𑁆𑀙𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑁄𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀓𑁄𑁇_  


(190.)

7880\. _𑀧𑀥𑀸𑀦𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀲𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(191.)

7881\. _𑀉𑀝𑁆𑀞𑀸𑀬 𑀅𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀚𑀳𑀺𑀢𑀸 𑀪𑁄𑀕𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀔𑁂𑀍𑀅𑀧𑀺𑀡𑁆𑀟𑁂𑀯 𑀪𑁄𑀕𑀫𑁆𑀳𑀺,_  
_𑀧𑁂𑀫𑀁 𑀫𑀬𑁆𑀳𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(192.)

7882\. _𑀯𑀻𑀭𑀺𑀬𑀁 𑀫𑁂 𑀥𑀼𑀭𑀥𑁄𑀭𑀬𑁆𑀳𑀁,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸𑀥𑀺𑀯𑀸𑀳𑀦𑀁;_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(193.)

7883\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(194.)

7884\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(195.)

7885\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁬𑁪𑁪𑁦)_  


7886\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

7887\. 𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇



---

7888\. 𑀬𑀲𑀯𑀕𑁆𑀕𑁄 𑀙𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁄𑁇



---

7889\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



7890\. _𑀬𑀲𑁄 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑁄 𑀘,_  
_𑀕𑀬𑀸𑀓𑀺𑀫𑀺𑀮𑀯𑀚𑁆𑀚𑀺𑀦𑁄;_  
_𑀤𑀼𑀯𑁂 𑀉𑀢𑁆𑀢𑀭𑀸 𑀪𑀤𑁆𑀤𑀚𑀻,_  
_𑀲𑀺𑀯𑀓𑁄 𑀉𑀧𑀯𑀸𑀳𑀦𑁄;_  
_𑀭𑀝𑁆𑀞𑀧𑀸𑀮𑁄 𑀏𑀓𑀲𑀢𑀁,_  
_𑀕𑀸𑀣𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀦𑀯𑀼𑀢𑀺𑁇_  


---
