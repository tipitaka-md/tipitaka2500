# 𑁫𑁬𑁇𑁨 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀬𑀲𑀯𑀕𑁆𑀕

### 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(29.)

7701\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀁 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀸𑀭𑀢𑁄 𑀉𑀢𑁆𑀢𑀫𑀁 𑀬𑀲𑀁;_  
_𑀅𑀕𑁆𑀕𑀨𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(30.)

7702\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁𑁇_  


(31.)

7703\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀕𑁆𑀕𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(32.)

7704\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(33.)

7705\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(34.)

7706\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


7707\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

7708\. 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
