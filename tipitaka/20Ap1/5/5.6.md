# 𑁫𑁇𑁬 𑀧𑀤𑀼𑀫𑀘𑁆𑀙𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀧𑀸𑀮𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀤𑀼𑀫𑀘𑁆𑀙𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(83.)

1301\. _“𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀫𑁆𑀳𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁂;_  
_𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀧𑀤𑀼𑀫𑀁 𑀕𑀬𑁆𑀳,_  
_𑀘𑀺𑀢𑀫𑀸𑀭𑁄𑀧𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(84.)

1302\. _𑀆𑀭𑁄𑀧𑀺𑀢𑁂 𑀘 𑀘𑀺𑀢𑀓𑁂,_  
_𑀯𑁂𑀳𑀸𑀲𑀁 𑀦𑀪𑀫𑀼𑀕𑁆𑀕𑀫𑀺;_  
_𑀆𑀓𑀸𑀲𑁂 𑀙𑀤𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀘𑀺𑀢𑀓𑀫𑁆𑀳𑀺 𑀅𑀥𑀸𑀭𑀬𑀺𑁇_  


(85.)

1303\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(86.)

1304\. _𑀲𑀢𑁆𑀢𑀢𑀸𑀮𑀻𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀤𑀼𑀫𑀺𑀲𑁆𑀲𑀭𑀦𑀸𑀫𑀓𑁄;_  
_𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(87.)

1305\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1306\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀤𑀼𑀫𑀘𑁆𑀙𑀤𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1307\. 𑀧𑀤𑀼𑀫𑀘𑁆𑀙𑀤𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
