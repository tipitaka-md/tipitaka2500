# 𑁫𑁇𑁮 𑀘𑀗𑁆𑀓𑀫𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀧𑀸𑀮𑀺𑀯𑀕𑁆𑀕

* 𑀘𑀗𑁆𑀓𑀫𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(93.)

1315\. _“𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀇𑀝𑁆𑀞𑀓𑀸𑀳𑀺 𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀘𑀗𑁆𑀓𑀫𑀁 𑀓𑀸𑀭𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(94.)

1316\. _𑀉𑀘𑁆𑀘𑀢𑁄 𑀧𑀜𑁆𑀘𑀭𑀢𑀦𑀁,_  
_𑀘𑀗𑁆𑀓𑀫𑀁 𑀲𑀸𑀥𑀼𑀫𑀸𑀧𑀺𑀢𑀁;_  
_𑀆𑀬𑀸𑀫𑀢𑁄 𑀳𑀢𑁆𑀣𑀲𑀢𑀁,_  
_𑀪𑀸𑀯𑀦𑀻𑀬𑁆𑀬𑀁 𑀫𑀦𑁄𑀭𑀫𑀁𑁇_  


(95.)

1317\. _𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸,_  
_𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀼𑀮𑀺𑀦𑀁 𑀕𑀬𑁆𑀳,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(96.)

1318\. _‘𑀇𑀫𑀺𑀦𑀸 𑀧𑀼𑀮𑀺𑀦𑀤𑀸𑀦𑁂𑀦,_  
_𑀘𑀗𑁆𑀓𑀫𑀁 𑀲𑀼𑀓𑀢𑁂𑀦 𑀘;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀧𑀼𑀮𑀺𑀦𑀁 𑀅𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀢𑀺𑁇_  


(97.)

1319\. _𑀢𑀻𑀡𑀺 𑀓𑀧𑁆𑀧𑀸𑀦𑀺 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁,_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(98.)

1320\. _𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑀫𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀭𑀸𑀚𑀸 𑀭𑀝𑁆𑀞𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀘,_  
_𑀫𑀳𑀺𑀬𑀸 𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(99.)

1321\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀗𑁆𑀓𑀫𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(100.)

1322\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1323\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀗𑁆𑀓𑀫𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1324\. 𑀘𑀗𑁆𑀓𑀫𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
