# 𑁩𑁩𑁇𑁨 𑀧𑀼𑀮𑀺𑀦𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀮𑀺𑀦𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(7.)

3594\. _“𑀓𑀓𑀼𑀥𑀁 𑀯𑀺𑀮𑀲𑀦𑁆𑀢𑀁𑀯,_  
_𑀦𑀺𑀲𑀪𑀸𑀚𑀸𑀦𑀺𑀬𑀁 𑀬𑀣𑀸;_  
_𑀑𑀲𑀥𑀺𑀁𑀯 𑀯𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀁,_  
_𑀑𑀪𑀸𑀲𑀦𑁆𑀢𑀁 𑀦𑀭𑀸𑀲𑀪𑀁𑁇_  


(8.)

3595\. _𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁;_  
_𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀧𑀭𑀺𑀯𑀡𑁆𑀡𑁂𑀲𑀺𑀁,_  
_𑀲𑀓𑀓𑀫𑁆𑀫𑁂𑀦 𑀢𑁄𑀲𑀬𑀺𑀁𑁇_  


(9.)

3596\. _𑀲𑀼𑀲𑀼𑀤𑁆𑀥𑀁 𑀧𑀼𑀮𑀺𑀦𑀁 𑀕𑀬𑁆𑀳,_  
_𑀕𑀢𑀫𑀕𑁆𑀕𑁂 𑀲𑀫𑁄𑀓𑀺𑀭𑀺𑀁;_  
_𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂𑀦 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(10.)

3597\. _𑀢𑀢𑁄 𑀉𑀧𑀟𑁆𑀠𑀧𑀼𑀮𑀺𑀦𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑁂 𑀑𑀲𑀺𑀜𑁆𑀘𑀺𑀁,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(11.)

3598\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀼𑀮𑀺𑀦𑀁 𑀬𑀫𑀲𑀺𑀜𑁆𑀘𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀮𑀺𑀦𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(12.)

3599\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3600\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀼𑀮𑀺𑀦𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---
