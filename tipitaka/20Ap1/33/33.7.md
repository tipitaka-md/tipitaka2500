# 𑁩𑁩𑁇𑁭 𑀦𑀺𑀕𑁆𑀕𑀼𑀡𑁆𑀟𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀕𑁆𑀕𑀼𑀡𑁆𑀟𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(39.)

3636\. _“𑀬𑀤𑀸 𑀤𑁂𑀯𑁄 𑀤𑁂𑀯𑀓𑀸𑀬𑀸,_  
_𑀘𑀯𑀢𑁂 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀬𑀸;_  
_𑀢𑀬𑁄 𑀲𑀤𑁆𑀤𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀯𑀸𑀦𑀁 𑀅𑀦𑀼𑀫𑁄𑀤𑀢𑀁𑁇_  


(40.)

3637\. _‘𑀇𑀢𑁄 𑀪𑁄 𑀲𑀼𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀮𑀪 𑀲𑀤𑁆𑀥𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


(41.)

3638\. _𑀲𑀸 𑀢𑁂 𑀲𑀤𑁆𑀥𑀸 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀸𑀲𑁆𑀲,_  
_𑀫𑀽𑀮𑀚𑀸𑀢𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀲𑀁𑀳𑀻𑀭𑀸,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂𑁇_  


(42.)

3639\. _𑀓𑀸𑀬𑁂𑀦 𑀓𑀼𑀲𑀮𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀯𑀸𑀘𑀸𑀬 𑀓𑀼𑀲𑀮𑀁 𑀩𑀳𑀼𑀁;_  
_𑀫𑀦𑀲𑀸 𑀓𑀼𑀲𑀮𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀁 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁𑁇_  


(43.)

3640\. _𑀢𑀢𑁄 𑀑𑀧𑀥𑀺𑀓𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀓𑀢𑁆𑀯𑀸 𑀤𑀸𑀦𑁂𑀦 𑀢𑀁 𑀩𑀳𑀼𑀁;_  
_𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀫𑀘𑁆𑀘𑁂 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂 𑀦𑀺𑀯𑁂𑀲𑀬’𑁇_  


(44.)

3641\. _𑀇𑀫𑀸𑀬 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬,_  
_𑀤𑁂𑀯𑀸𑀤𑁂𑀯𑀁 𑀬𑀤𑀸 𑀯𑀺𑀤𑀽;_  
_𑀘𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀏𑀳𑀺 𑀤𑁂𑀯 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁𑁇_  


(45.)

3642\. _‘𑀲𑀁𑀯𑁂𑀕𑁄 𑀫𑁂 𑀢𑀤𑀸 𑀆𑀲𑀺,_  
_𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑁂 𑀲𑀫𑀸𑀕𑀢𑁂;_  
_𑀓𑀁𑀲𑀼 𑀦𑀸𑀫 𑀅𑀳𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀇𑀢𑁄 𑀘𑀼𑀢𑁄’𑁇_  


(46.)

3643\. _𑀫𑀫 𑀲𑀁𑀯𑁂𑀕𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀫𑀡𑁄 𑀪𑀸𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀫𑀫𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(47.)

3644\. _𑀲𑀼𑀫𑀦𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑀸𑀯𑀓𑁄;_  
_𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀯𑁂𑀚𑁂𑀲𑀺 𑀫𑀫𑀁 𑀢𑀤𑀸𑁇_  


---

3645\. 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇



(48.)

3646\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀢𑀁 𑀥𑀻𑀭𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(49.)

3647\. _𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀁 𑀲 𑀢𑀢𑁆𑀣𑁂𑀯,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀯𑀲𑀦𑁆𑀢𑁄 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀺,_  
_𑀧𑀼𑀦 𑀥𑀸𑀭𑁂𑀢𑀺 𑀫𑀸𑀢𑀼𑀬𑀸𑁇_  


(50.)

3648\. _𑀢𑀫𑁆𑀳𑀸 𑀓𑀸𑀬𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀺𑀤𑀲𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀳𑀁;_  
_𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑀳𑀁 𑀢𑀤𑀸𑁇_  


(51.)

3649\. _𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀁 𑀲𑀫𑁄𑀓𑁆𑀓𑀫𑀺𑀁;_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦 𑀓𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀸,_  
_𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀁 𑀅𑀚𑀸𑀦𑀳𑀁𑁇_  


(52.)

3650\. _𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑁄𑀯,_  
_𑀆𑀭𑀸𑀫𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑀁 𑀅𑀳𑀁;_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄𑁇_  


(53.)

3651\. _𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑁂 𑀧𑀸𑀯𑀘𑀦𑁂,_  
_𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀫𑁆𑀳𑀺 𑀲𑀸𑀲𑀦𑁂;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀸𑀲𑀦𑀓𑀭𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀯𑁄 𑀢𑀢𑁆𑀣 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(54.)

3652\. _𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀭𑀸𑀚𑀸 𑀢𑀢𑁆𑀣𑀸𑀲𑀺 𑀓𑁄𑀲𑀮𑁄;_  
_𑀭𑀣𑁂𑀦 𑀦𑀸𑀕𑀬𑀼𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁂𑀲𑀺 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(55.)

3653\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀦𑀸𑀕𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀫𑀬𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁𑁇_  


(56.)

3654\. _𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑁄𑀯,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀬𑁄 𑀲𑁄 𑀩𑀼𑀤𑁆𑀥𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀲𑀺,_  
_𑀆𑀦𑀦𑁆𑀤𑁄 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄𑁇_  


(57.)

3655\. _𑀕𑀢𑀺𑀫𑀸 𑀥𑀺𑀢𑀺𑀫𑀸 𑀘𑁂𑀯,_  
_𑀲𑀢𑀺𑀫𑀸 𑀘 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀭𑀜𑁆𑀜𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀲𑀺 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑁇_  


(58.)

3656\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀞𑀺𑀢𑀓𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(59.)

3657\. _𑀏𑀓𑀁𑀲𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀭𑁂 𑀓𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀁 𑀯𑀸𑀘𑀁 𑀉𑀤𑀻𑀭𑀬𑀺𑀁𑁇_  


(60.)

3658\. _‘𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀦𑀺𑀕𑁆𑀕𑀼𑀡𑁆𑀟𑀺𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑁂 𑀞𑀧𑁂𑀲𑀳𑀁𑁇_  


(61.)

3659\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪;_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁’𑁇_  


(62.)

3660\. _𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀫𑀦𑀼𑀚𑀸𑀥𑀺𑀧𑀸;_  
_𑀅𑀩𑁆𑀩𑀼𑀤𑀦𑀺𑀭𑀩𑁆𑀩𑀼𑀤𑀸𑀦𑀺,_  
_𑀅𑀝𑁆𑀞𑀝𑁆𑀞𑀸𑀲𑀺𑀁𑀲𑀼 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(63.)

3661\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3662\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀺𑀕𑁆𑀕𑀼𑀡𑁆𑀟𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
