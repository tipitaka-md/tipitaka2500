# 𑁩𑁩𑁇𑁧 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3586. _“𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀫𑀧𑀭𑀸𑀚𑀺𑀢𑀁;_  
_𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀉𑀧𑀯𑀺𑀝𑁆𑀞𑀁,_  
_𑀅𑀤𑁆𑀤𑀲𑀸𑀳𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(2.)

3587. _𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑀧𑀼𑀧𑁆𑀨𑀸 𑀏𑀓𑀲𑀻𑀲𑀸,_  
_𑀉𑀤𑁆𑀥𑀁𑀯𑀡𑁆𑀝𑀸 𑀅𑀥𑁄𑀫𑀼𑀔𑀸𑁇_  


(3.)

3588. _𑀲𑀼𑀘𑀺𑀢𑁆𑀢𑀸 𑀯𑀺𑀬 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁂,_  
_𑀆𑀓𑀸𑀲𑁂 𑀧𑀼𑀧𑁆𑀨𑀲𑀦𑁆𑀣𑀭𑀸;_  
_𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀢𑀼𑀲𑀺𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(4.)

3589. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(5.)

3590. _𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_𑀲𑀫𑀦𑁆𑀢𑀙𑀤𑀦𑁄 𑀦𑀸𑀫,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(6.)

3591. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3592. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3593. 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
