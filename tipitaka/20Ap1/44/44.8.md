# 𑁪𑁪𑁇𑁮 𑀳𑀭𑀻𑀢𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀯𑀕𑁆𑀕

* 𑀳𑀭𑀻𑀢𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(60.)

5737\. _“𑀳𑀭𑀻𑀢𑀓𑀁 𑀆𑀫𑀮𑀓𑀁,_  
_𑀅𑀫𑁆𑀩𑀚𑀫𑁆𑀩𑀼𑀯𑀺𑀪𑀻𑀢𑀓𑀁;_  
_𑀓𑁄𑀮𑀁 𑀪𑀮𑁆𑀮𑀸𑀢𑀓𑀁 𑀩𑀺𑀮𑁆𑀮𑀁,_  
_𑀲𑀬𑀫𑁂𑀯 𑀳𑀭𑀸𑀫𑀳𑀁𑁇_  


(61.)

5738\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀧𑀩𑁆𑀪𑀸𑀭𑀕𑀢𑀁,_  
_𑀛𑀸𑀬𑀺𑀁 𑀛𑀸𑀦𑀭𑀢𑀁 𑀫𑀼𑀦𑀺𑀁;_  
_𑀆𑀩𑀸𑀥𑁂𑀦 𑀆𑀧𑀻𑀍𑀏𑀦𑁆𑀢𑀁,_  
_𑀅𑀤𑀼𑀢𑀻𑀬𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁𑁇_  


(62.)

5739\. _𑀳𑀭𑀻𑀢𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀔𑀸𑀤𑀫𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑁂,_  
_𑀩𑁆𑀬𑀸𑀥𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(63.)

5740\. _𑀧𑀳𑀻𑀦𑀤𑀭𑀣𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀦𑀼𑀫𑁄𑀤𑀫𑀓𑀸𑀲𑀺 𑀫𑁂;_  
_‘𑀪𑁂𑀲𑀚𑁆𑀚𑀤𑀸𑀦𑁂𑀦𑀺𑀫𑀺𑀦𑀸,_  
_𑀩𑁆𑀬𑀸𑀥𑀺𑀯𑀽𑀧𑀲𑀫𑁂𑀦 𑀘𑁇_  


(64.)

5741\. _𑀤𑁂𑀯𑀪𑀽𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸,_  
_𑀚𑀸𑀢𑁄 𑀯𑀸 𑀅𑀜𑁆𑀜𑀚𑀸𑀢𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀢𑀼,_  
_𑀫𑀸 𑀘 𑀢𑁂 𑀩𑁆𑀬𑀸𑀥𑀺𑀫𑀸𑀕𑀫𑀸’𑁇_  


(65.)

5742\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀦𑀪𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀫𑀻 𑀥𑀻𑀭𑁄,_  
_𑀳𑀁𑀲𑀭𑀸𑀚𑀸𑀯 𑀅𑀫𑁆𑀩𑀭𑁂𑁇_  


(66.)

5743\. _𑀬𑀢𑁄 𑀳𑀭𑀻𑀢𑀓𑀁 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀇𑀫𑀁 𑀚𑀸𑀢𑀺𑀁 𑀉𑀧𑀸𑀤𑀸𑀬,_  
_𑀩𑁆𑀬𑀸𑀥𑀺 𑀫𑁂 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀣𑁇_  


(67.)

5744\. _𑀅𑀬𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀘𑀭𑀺𑀫𑁄 𑀯𑀢𑁆𑀢𑀢𑁂 𑀪𑀯𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(68.)

5745\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀪𑁂𑀲𑀚𑁆𑀚𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(69.)

5746\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(70.)

5747\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(71.)

5748\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5749\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀳𑀭𑀻𑀢𑀓𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5750\. 𑀳𑀭𑀻𑀢𑀓𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
