# 𑁪𑁪𑁇𑁧 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

5664\. _“𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(2.)

5665\. _𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑁄 𑀦𑀺𑀭𑀸𑀮𑀫𑁆𑀩𑁄,_  
_𑀆𑀓𑀸𑀲𑀲𑀫𑀫𑀸𑀦𑀲𑁄;_  
_𑀲𑀼𑀜𑁆𑀜𑀢𑀸𑀩𑀳𑀼𑀮𑁄 𑀢𑀸𑀤𑀻,_  
_𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀭𑀢𑁄 𑀯𑀲𑀻𑁇_  


(3.)

5666\. _𑀅𑀲𑀗𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀦𑀺𑀓𑁆𑀮𑁂𑀲𑁄,_  
_𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀓𑀼𑀮𑁂 𑀕𑀡𑁂;_  
_𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀯𑀻𑀭𑁄,_  
_𑀯𑀺𑀦𑀬𑁄𑀧𑀸𑀬𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


(4.)

5667\. _𑀉𑀬𑁆𑀬𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀓𑀺𑀘𑁆𑀘𑁂𑀲𑀼,_  
_𑀯𑀺𑀦𑀬𑀦𑁆𑀢𑁄 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀫𑀦𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀕𑀢𑀺𑀁 𑀧𑀗𑁆𑀓𑀯𑀺𑀲𑁄𑀲𑀦𑀁𑁇_  


(5.)

5668\. _𑀅𑀫𑀢𑀁 𑀧𑀭𑀫𑀲𑁆𑀲𑀸𑀤𑀁,_  
_𑀚𑀭𑀸𑀫𑀘𑁆𑀘𑀼𑀦𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀫𑀳𑀸𑀧𑀭𑀺𑀲𑀫𑀚𑁆𑀛𑁂 𑀲𑁄,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀮𑁄𑀓𑀢𑀸𑀭𑀓𑁄𑁇_  


(6.)

5669\. _𑀓𑀭𑀯𑀻𑀓𑀭𑀼𑀢𑁄 𑀦𑀸𑀣𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀖𑁄𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑁄;_  
_𑀉𑀤𑁆𑀥𑀭𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀤𑀼𑀕𑁆𑀕𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀦𑀝𑁆𑀞𑁂 𑀅𑀦𑀸𑀬𑀓𑁂𑁇_  


(7.)

5670\. _𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀯𑀺𑀭𑀚𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(8.)

5671\. _𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀤𑀸𑀧𑀸𑀳𑀁,_  
_𑀘𑀺𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑁄 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀁;_  
_𑀏𑀓𑀓𑁄𑀯 𑀯𑀦𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀯𑀲𑀺𑀁 𑀲𑀁𑀲𑀕𑁆𑀕𑀧𑀻𑀍𑀇𑀢𑁄𑁇_  


(9.)

5672\. _𑀲𑀓𑁆𑀓𑀸𑀬𑀯𑀽𑀧𑀓𑀸𑀲𑁄 𑀫𑁂,_  
_𑀳𑁂𑀢𑀼𑀪𑀽𑀢𑁄 𑀫𑀫𑀸𑀪𑀯𑀻;_  
_𑀫𑀦𑀲𑁄 𑀯𑀽𑀧𑀓𑀸𑀲𑀲𑁆𑀲,_  
_𑀲𑀁𑀲𑀕𑁆𑀕𑀪𑀬𑀤𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(10.)

5673\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(11.)

5674\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(12.)

5675\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5676\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5677\. 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
