# 𑁪𑁪𑁇𑁧𑁦 𑀅𑀫𑁆𑀩𑀨𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀫𑁆𑀩𑀨𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(81.)

5762\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀥𑀸𑀭𑀢𑁄 𑀉𑀢𑁆𑀢𑀫𑀁 𑀬𑀲𑀁𑁇_  


(82.)

5763\. _𑀅𑀕𑁆𑀕𑀨𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀯𑀻𑀭𑀲𑁆𑀲,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀅𑀳𑀁𑁇_  


(83.)

5764\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪;_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁𑁇_  


(84.)

5765\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀕𑁆𑀕𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(85.)

5766\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(86.)

5767\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(87.)

5768\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁪𑁬𑁨𑁪)_  


5769\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀫𑁆𑀩𑀨𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5770\. 𑀅𑀫𑁆𑀩𑀨𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

5771\. 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁄𑁇



---

5772\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



5773\. _𑀣𑁂𑀭𑁄 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀻 𑀘,_  
_𑀲𑀗𑁆𑀔𑀺𑀬𑁄 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀓𑁄;_  
_𑀣𑀯𑀺𑀓𑁄 𑀉𑀘𑁆𑀙𑀼𑀔𑀡𑁆𑀟𑀻 𑀘,_  
_𑀓𑀍𑀅𑀫𑁆𑀩𑀅𑀫𑁆𑀩𑀸𑀝𑀓𑀤𑁄𑁇_  


5774\. _𑀳𑀭𑀻𑀢𑀓𑀫𑁆𑀩𑀧𑀺𑀡𑁆𑀟𑀻 𑀘,_  
_𑀅𑀫𑁆𑀩𑀤𑁄 𑀤𑀲𑀫𑁄 𑀬𑀢𑀺;_  
_𑀙𑀍𑀅𑀲𑀻𑀢𑀺 𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_
