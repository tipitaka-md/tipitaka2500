# 𑁪𑁪𑁇𑁨 𑀏𑀓𑀲𑀗𑁆𑀔𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀲𑀗𑁆𑀔𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(13.)

5678\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀫𑀳𑀸𑀩𑁄𑀥𑀺𑀫𑀳𑁄 𑀅𑀳𑀼;_  
_𑀫𑀳𑀸𑀚𑀦𑀸 𑀲𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀺 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(14.)

5679\. _𑀦 𑀳𑀺 𑀢𑀁 𑀑𑀭𑀓𑀁 𑀫𑀜𑁆𑀜𑁂,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀬𑀲𑁆𑀲𑀸𑀬𑀁 𑀈𑀤𑀺𑀲𑀸 𑀩𑁄𑀥𑀺,_  
_𑀧𑀽𑀚𑀦𑀻𑀬𑀸 𑀘 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(15.)

5680\. _𑀢𑀢𑁄 𑀲𑀗𑁆𑀔𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑁄𑀥𑀺𑀭𑀼𑀓𑁆𑀔𑀫𑀼𑀧𑀝𑁆𑀞𑀳𑀺𑀁;_  
_𑀥𑀫𑀦𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀤𑀺𑀯𑀲𑀁,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(16.)

5681\. _𑀆𑀲𑀦𑁆𑀦𑀓𑁂 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑀁 𑀅𑀧𑀸𑀧𑀬𑀻;_  
_𑀓𑀍𑀏𑀯𑀭𑀁 𑀫𑁂 𑀧𑀢𑀺𑀢𑀁,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀸𑀫𑀳𑀁𑁇_  


(17.)

5682\. _𑀲𑀝𑁆𑀞𑀺𑀢𑀼𑀭𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀼𑀝𑁆𑀞𑀳𑀝𑁆𑀞𑀸 𑀧𑀫𑁄𑀤𑀺𑀢𑀸;_  
_𑀲𑀤𑀸 𑀫𑀬𑁆𑀳𑀁 𑀉𑀧𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(18.)

5683\. _𑀏𑀓𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺𑀁 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄;_  
_𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻,_  
_𑀚𑀫𑁆𑀩𑀼𑀫𑀡𑁆𑀟𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑁄𑁇_  


(19.)

5684\. _𑀢𑀢𑁄 𑀅𑀗𑁆𑀕𑀲𑀢𑀸 𑀢𑀼𑀭𑀺𑀬𑀸,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸;_  
_𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀲𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(20.)

5685\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑀲𑁆𑀲𑀸𑀧𑀺,_  
_𑀯𑀚𑁆𑀚𑀭𑁂 𑀪𑁂𑀭𑀺𑀬𑁄 𑀲𑀤𑀸𑁇_  


(21.)

5686\. _𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀅𑀦𑀼𑀪𑀼𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀲𑀺𑀯𑀁 𑀲𑀼𑀔𑁂𑀫𑀁 𑀅𑀫𑀢𑀁,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁𑁇_  


(22.)

5687\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(23.)

5688\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(24.)

5689\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(25.)

5690\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5691\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀲𑀗𑁆𑀔𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5692\. 𑀏𑀓𑀲𑀗𑁆𑀔𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
