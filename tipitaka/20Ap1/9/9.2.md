# 𑁯𑁇𑁨 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀢𑀺𑀫𑀺𑀭𑀯𑀕𑁆𑀕

* 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(10.)

1697\. _“𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑁄𑀳𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(11.)

1698\. _𑀲𑀢𑁆𑀢𑀦𑀗𑁆𑀕𑀮𑀓𑀻𑀧𑀼𑀧𑁆𑀨𑁂,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁 𑀅𑀳𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀺𑀲𑁆𑀲,_  
_𑀅𑀦𑀦𑁆𑀢𑀕𑀼𑀡𑀲𑀸𑀕𑀭𑀁𑁇_  


(12.)

1699\. _𑀲𑀼𑀕𑀢𑀸𑀦𑀼𑀕𑀢𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀳𑀝𑁆𑀞𑀫𑀸𑀦𑀲𑁄;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀜𑁆𑀘 𑀢𑀤𑀸𑀓𑀸𑀲𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(13.)

1700\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(14.)

1701\. _𑀇𑀢𑁄 𑀅𑀝𑁆𑀞𑀫𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀢𑀬𑁄 𑀅𑀕𑁆𑀕𑀺𑀲𑀺𑀔𑀸 𑀅𑀳𑀼;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(15.)

1702\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1703\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1704\. 𑀕𑀢𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
