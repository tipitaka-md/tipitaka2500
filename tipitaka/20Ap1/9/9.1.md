# 𑁯𑁇𑁧 𑀢𑀺𑀫𑀺𑀭𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀢𑀺𑀫𑀺𑀭𑀯𑀕𑁆𑀕

* 𑀢𑀺𑀫𑀺𑀭𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

1686\. _“𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀸𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀅𑀦𑀼𑀲𑁄𑀢𑀁 𑀯𑀚𑀸𑀫𑀳𑀁;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀲𑀫𑀡𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸𑀯𑀺𑀮𑀁𑁇_  


(2.)

1687\. _𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀢𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀺𑀡𑁆𑀡𑁄𑀬𑀁,_  
_𑀤𑀦𑁆𑀢𑁄𑀬𑀁 𑀤𑀫𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(3.)

1688\. _𑀅𑀲𑁆𑀲𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑁄,_  
_𑀲𑀦𑁆𑀢𑁄 𑀘 𑀲𑀫𑀬𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑁄𑀘𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀼𑀢𑁆𑀢𑁄 𑀘,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀲𑁆𑀲𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇_  


(4.)

1689\. _𑀏𑀯𑀸𑀳𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀺𑀫𑀺𑀭𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀫𑀢𑁆𑀣𑀓𑁂 𑀑𑀓𑀺𑀭𑀺𑀁 𑀅𑀳𑀁𑁇_  


(5.)

1690\. _𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀅𑀧𑀭𑀁 𑀤𑀺𑀲𑀁𑁇_  


(6.)

1691\. _𑀅𑀘𑀺𑀭𑀁 𑀕𑀢𑀫𑀢𑁆𑀢𑀁 𑀫𑀁,_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀯𑀺𑀳𑁂𑀞𑀬𑀺;_  
_𑀧𑀧𑀸𑀢𑀫𑀦𑀼𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄,_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀧𑀧𑀢𑀺𑀁 𑀅𑀳𑀁𑁇_  


(7.)

1692\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(8.)

1693\. _𑀙𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀳𑀺 𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀲𑀢𑁆𑀢𑁂𑀯𑀸𑀲𑀼𑀁 𑀫𑀳𑀸𑀬𑀲𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(9.)

1694\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1695\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀺𑀫𑀺𑀭𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1696\. 𑀢𑀺𑀫𑀺𑀭𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
