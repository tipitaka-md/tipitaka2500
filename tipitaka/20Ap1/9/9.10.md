# 𑁯𑁇𑁧𑁦 𑀧𑀤𑀼𑀫𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀢𑀺𑀫𑀺𑀭𑀯𑀕𑁆𑀕

### 𑀧𑀤𑀼𑀫𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(51.)

1754\. _“𑀧𑁄𑀓𑁆𑀔𑀭𑀯𑀦𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄,_  
_𑀪𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀧𑀤𑀼𑀫𑀸𑀦𑀺𑀳𑀁;_  
_𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲𑀁 𑀨𑀼𑀲𑁆𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑀁𑁇_  


(52.)

1755\. _𑀧𑀤𑀼𑀫𑀧𑀼𑀧𑁆𑀨𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁 𑀅𑀳𑀁;_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀲𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(53.)

1756\. _𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀓𑀸𑀬𑁂𑀦,_  
_𑀫𑀦𑀲𑀸 𑀲𑀁𑀯𑀼𑀢𑁂𑀦 𑀘;_  
_𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀆𑀚𑀻𑀯𑀁 𑀧𑀭𑀺𑀲𑁄𑀥𑀬𑀺𑀁𑁇_  


(54.)

1757\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(55.)

1758\. _𑀧𑀤𑀼𑀫𑀸𑀪𑀸𑀲𑀦𑀸𑀫𑀸 𑀘,_  
_𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀫𑀳𑀻𑀧𑀢𑀻;_  
_𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂𑀲𑀼 𑀓𑀧𑁆𑀧𑁂𑀲𑀼,_  
_𑀅𑀝𑁆𑀞𑀢𑀸𑀮𑀻𑀲𑀫𑀸𑀲𑀺𑀲𑀼𑀁𑁇_  


(56.)

1759\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁧𑁫𑁪𑁨)_  


1760\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀤𑀼𑀫𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1761\. 𑀧𑀤𑀼𑀫𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

1762\. 𑀢𑀺𑀫𑀺𑀭𑀯𑀕𑁆𑀕𑁄 𑀦𑀯𑀫𑁄𑁇



---

1763\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1764\. _𑀢𑀺𑀫𑀺𑀭𑀦𑀗𑁆𑀕𑀮𑀻𑀧𑀼𑀧𑁆𑀨,_  
_𑀦𑀺𑀧𑁆𑀧𑀦𑁆𑀦𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀅𑀥𑁄;_  
_𑀤𑁆𑀯𑁂 𑀭𑀁𑀲𑀺𑀲𑀜𑁆𑀜𑀻 𑀨𑀮𑀤𑁄,_  
_𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀻 𑀘 𑀲𑁂𑀘𑀓𑁄;_  
_𑀧𑀤𑁆𑀫𑀧𑀼𑀧𑁆𑀨𑀻 𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀙𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀲 𑀧𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇_
