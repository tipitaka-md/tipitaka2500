# 𑁯𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀭𑀁𑀲𑀺𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀢𑀺𑀫𑀺𑀭𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀭𑀁𑀲𑀺𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(35.)

1730\. _“𑀧𑀩𑁆𑀩𑀢𑁂 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀯𑀸𑀓𑀘𑀻𑀭𑀥𑀭𑁄 𑀅𑀳𑀁;_  
_𑀘𑀗𑁆𑀓𑀫𑀜𑁆𑀘 𑀲𑀫𑀸𑀭𑀽𑀍𑀳𑁄,_  
_𑀦𑀺𑀲𑀻𑀤𑀺𑀁 𑀧𑀸𑀘𑀺𑀦𑀸𑀫𑀼𑀔𑁄𑁇_  


(36.)

1731\. _𑀧𑀩𑁆𑀩𑀢𑁂 𑀲𑀼𑀕𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀨𑀼𑀲𑁆𑀲𑀁 𑀛𑀸𑀦𑀭𑀢𑀁 𑀢𑀤𑀸;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀭𑀁𑀲𑁆𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁𑁇_  


(37.)

1732\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀭𑀁𑀲𑀺𑀲𑀜𑁆𑀜𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(38.)

1733\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1734\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀭𑀁𑀲𑀺𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1735\. 𑀤𑀼𑀢𑀺𑀬𑀭𑀁𑀲𑀺𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
