# 𑁩𑁫𑁇𑁧𑁦 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(63.)

3961\. _“𑀭𑁄𑀫𑀲𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀤𑀻𑀓𑀽𑀮𑁂 𑀯𑀲𑀻 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀧𑀻𑀢𑀭𑀁𑀲𑀺𑀁𑀯 𑀪𑀸𑀡𑀼𑀫𑀁𑁇_  


(64.)

3962\. _𑀉𑀓𑁆𑀓𑀸𑀫𑀼𑀔𑀧𑀳𑀝𑁆𑀞𑀁𑀯,_  
_𑀔𑀤𑀺𑀭𑀗𑁆𑀕𑀸𑀭𑀲𑀦𑁆𑀦𑀺𑀪𑀁;_  
_𑀑𑀲𑀥𑀺𑀁𑀯 𑀯𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀁,_  
_𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀫𑀓𑀸𑀲𑀳𑀁𑁇_  


(65.)

3963\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀅𑀜𑁆𑀚𑀮𑀺𑀫𑀓𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀬𑀸 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(66.)

3964\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁩𑁧𑁩𑁩)_  


3965\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3966\. 𑀏𑀓𑀜𑁆𑀚𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

3967\. 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀢𑀺𑀁𑀲𑀢𑀺𑀫𑁄𑁇



---

3968\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3969\. _𑀧𑀤𑀼𑀫𑀻 𑀉𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀻,_  
_𑀥𑀚𑁄 𑀓𑀺𑀗𑁆𑀓𑀡𑀺𑀓𑀁 𑀦𑀍𑀅𑀁;_  
_𑀘𑀫𑁆𑀧𑀓𑁄 𑀧𑀤𑀼𑀫𑁄 𑀫𑀼𑀝𑁆𑀞𑀺,_  
_𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑁂𑀓𑀜𑁆𑀚𑀮𑀻 𑀢𑀣𑀸;_  
_𑀙 𑀘 𑀲𑀝𑁆𑀞𑀺 𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀕𑀡𑀺𑀢𑀸𑀬𑁄 𑀯𑀺𑀪𑀸𑀯𑀺𑀪𑀺𑁇_
