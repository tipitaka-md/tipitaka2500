# 𑁩𑁫𑁇𑁯 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(59.)

3955. _“𑀓𑀡𑀺𑀓𑀸𑀭𑀁𑀯 𑀚𑁄𑀢𑀦𑁆𑀢𑀁,_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑀫𑀦𑀸𑀲𑀯𑀁𑁇_  


(60.)

3956. _𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀁 𑀲𑀨𑀮𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀓𑁄𑀲𑀓𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(61.)

3957. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(62.)

3958. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3959. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3960. 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
