# 𑁩𑁫𑁇𑁧 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3881\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀪𑀯𑀸𑀪𑀯𑁂 𑀯𑀺𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(2.)

3882\. _𑀳𑀁𑀲𑀭𑀸𑀚𑀸 𑀢𑀤𑀸 𑀳𑁄𑀫𑀺,_  
_𑀤𑀺𑀚𑀸𑀦𑀁 𑀧𑀯𑀭𑁄 𑀅𑀳𑀁;_  
_𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑀁 𑀲𑀫𑁄𑀕𑀬𑁆𑀳,_  
_𑀓𑀻𑀍𑀆𑀫𑀺 𑀳𑀁𑀲𑀓𑀻𑀍𑀇𑀢𑀁𑁇_  


(3.)

3883\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑀲𑁆𑀲 𑀉𑀧𑀭𑀺,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀢𑀸𑀯𑀤𑁂 𑀚𑀺𑀦𑁄𑁇_  


(4.)

3884\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀳𑀁 𑀤𑁂𑀯𑀤𑁂𑀯𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀯𑀡𑁆𑀝𑁂 𑀙𑁂𑀢𑁆𑀯𑀸𑀦 𑀧𑀤𑀼𑀫𑀁,_  
_𑀲𑀢𑀧𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀭𑀫𑀁𑁇_  


(5.)

3885\. _𑀫𑀼𑀔𑀢𑀼𑀡𑁆𑀟𑁂𑀦 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸𑀦 𑀕𑀕𑀦𑁂,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀅𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(6.)

3886\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀅𑀓𑀸 𑀫𑁂 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑀁𑁇_  


(7.)

3887\. _‘𑀇𑀫𑀺𑀦𑀸 𑀏𑀓𑀧𑀤𑀼𑀫𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦 𑀕𑀘𑁆𑀙𑀲𑀺’𑁇_  


(8.)

3888\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀓𑁄;_  
_𑀫𑀫 𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀕𑀫𑀸 𑀬𑁂𑀦 𑀧𑀢𑁆𑀣𑀺𑀢𑀁𑁇_  


(9.)

3889\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(10.)

3890\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3891\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3892\. 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
