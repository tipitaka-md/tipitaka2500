# 𑁩𑁫𑁇𑁫 𑀦𑀍𑀆𑀕𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀦𑀍𑀆𑀕𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(31.)

3919\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀳𑀸𑀭𑀺𑀢𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀦𑀸𑀭𑀤𑁄 𑀦𑀸𑀫,_  
_𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀲𑀻 𑀢𑀤𑀸𑁇_  


(32.)

3920\. _𑀦𑀍𑀆𑀕𑀸𑀭𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀺𑀡𑁂𑀦 𑀙𑀸𑀤𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀘𑀗𑁆𑀓𑀫𑀁 𑀲𑁄𑀥𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(33.)

3921\. _𑀘𑀢𑀼𑀤𑁆𑀤𑀲𑀲𑀼 𑀓𑀧𑁆𑀧𑁂𑀲𑀼,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀁 𑀅𑀳𑀁;_  
_𑀘𑀢𑀼𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀅𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(34.)

3922\. _𑀘𑀢𑀼𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(35.)

3923\. _𑀉𑀩𑁆𑀩𑀺𑀤𑁆𑀥𑀁 𑀪𑀯𑀦𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀇𑀦𑁆𑀤𑀮𑀝𑁆𑀞𑀻𑀯 𑀉𑀕𑁆𑀕𑀢𑀁;_  
_𑀲𑀳𑀲𑁆𑀲𑀣𑀫𑁆𑀪𑀁 𑀅𑀢𑀼𑀮𑀁,_  
_𑀯𑀺𑀫𑀸𑀦𑀁 𑀲𑀧𑀪𑀲𑁆𑀲𑀭𑀁𑁇_  


(36.)

3924\. _𑀤𑁆𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀸𑀲𑀦𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀳𑀁𑁇_  


(37.)

3925\. _𑀧𑀥𑀸𑀦𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(38.)

3926\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3927\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀍𑀆𑀕𑀸𑀭𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3928\. 𑀦𑀍𑀆𑀕𑀸𑀭𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
