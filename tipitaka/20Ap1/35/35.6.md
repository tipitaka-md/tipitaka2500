# 𑁩𑁫𑁇𑁬 𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀏𑀓𑀧𑀤𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(39.)

3929\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀚𑀸𑀧𑀮𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄 𑀦𑀸𑀫,_  
_𑀯𑀺𑀳𑀸𑀲𑀺 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂𑁇_  


(40.)

3930\. _𑀧𑀼𑀧𑁆𑀨𑀁 𑀳𑁂𑀫𑀯𑀦𑁆𑀢𑀁 𑀕𑀬𑁆𑀳,_  
_𑀕𑀘𑁆𑀙𑀁 𑀯𑁂𑀳𑀸𑀬𑀲𑁂𑀦𑀳𑀁;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑀫𑀦𑀸𑀲𑀯𑀁𑁇_  


(41.)

3931\. _𑀲𑀢𑁆𑀢 𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀲𑀻𑀲𑁂 𑀓𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑁂𑀲𑀺𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(42.)

3932\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(43.)

3933\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3934\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3935\. 𑀘𑀫𑁆𑀧𑀓𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
