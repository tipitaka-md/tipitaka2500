# 𑁪𑁧𑁇𑁯 𑀚𑀢𑀼𑀓𑀡𑁆𑀡𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕

* 𑀚𑀢𑀼𑀓𑀡𑁆𑀡𑀺𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(276.)

5138\. _“𑀦𑀕𑀭𑁂 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀸,_  
_𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄 𑀓𑀸𑀫𑀕𑀼𑀡𑁂,_  
_𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀫𑀳𑀁 𑀢𑀤𑀸𑁇_  


(277.)

5139\. _𑀢𑀢𑁄 𑀧𑀸𑀲𑀸𑀤𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀫𑀳𑀸𑀪𑁄𑀕𑁂 𑀯𑀮𑀜𑁆𑀚𑀓𑁄;_  
_𑀢𑀢𑁆𑀣 𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺,_  
_𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀫𑀳𑀁 𑀢𑀤𑀸𑁇_  


(278.)

5140\. _𑀢𑀽𑀭𑀺𑀬𑀸 𑀆𑀳𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀲𑀫𑁆𑀫𑀢𑀸𑀍𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀸 𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀲𑀩𑁆𑀩𑀸,_  
_𑀳𑀭𑀦𑁆𑀢𑀺𑀬𑁂𑀯 𑀫𑁂 𑀫𑀦𑁄𑁇_  


(279.)

5141\. _𑀘𑁂𑀮𑀸𑀧𑀺𑀓𑀸 𑀯𑀸𑀫𑀦𑀺𑀓𑀸,_  
_𑀓𑀼𑀜𑁆𑀚𑀯𑀸𑀲𑀻 𑀢𑀺𑀫𑀚𑁆𑀛𑀺𑀓𑀸;_  
_𑀮𑀗𑁆𑀖𑀺𑀓𑀸 𑀲𑁄𑀓𑀚𑁆𑀛𑀸𑀬𑀻 𑀘,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(280.)

5142\. _𑀯𑁂𑀢𑀸𑀍𑀇𑀦𑁄 𑀓𑀼𑀫𑁆𑀪𑀣𑀽𑀦𑀻,_  
_𑀦𑀝𑀸 𑀘 𑀦𑀘𑁆𑀘𑀓𑀸 𑀩𑀳𑀽;_  
_𑀦𑀝𑀓𑀸 𑀦𑀸𑀝𑀓𑀸 𑀘𑁂𑀯,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(281.)

5143\. _𑀓𑀧𑁆𑀧𑀓𑀸 𑀦𑁆𑀳𑀸𑀧𑀓𑀸 𑀲𑀽𑀤𑀸,_  
_𑀫𑀸𑀮𑀸𑀓𑀸𑀭𑀸 𑀲𑀼𑀧𑀸𑀲𑀓𑀸;_  
_𑀚𑀮𑁆𑀮𑀸 𑀫𑀮𑁆𑀮𑀸 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(282.)

5144\. _𑀏𑀢𑁂𑀲𑀼 𑀓𑀻𑀍𑀅𑀫𑀸𑀦𑁂𑀲𑀼,_  
_𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁂 𑀓𑀢𑀼𑀧𑀸𑀲𑀦𑁂;_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀇𑀦𑁆𑀤𑁄𑀯 𑀢𑀺𑀤𑀲𑀗𑁆𑀕𑀡𑁂𑁇_  


(283.)

5145\. _𑀅𑀤𑁆𑀥𑀺𑀓𑀸 𑀧𑀣𑀺𑀓𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀬𑀸𑀘𑀓𑀸 𑀯𑀭𑀓𑀸 𑀩𑀳𑀽;_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀢𑁂 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀬𑀦𑁆𑀢𑀸 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(284.)

5146\. _𑀲𑀫𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀘𑁂𑀯,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸;_  
_𑀯𑀟𑁆𑀠𑀬𑀦𑁆𑀢𑀸 𑀫𑀫𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(285.)

5147\. _𑀧𑀝𑀕𑀸 𑀮𑀝𑀼𑀓𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀦𑀺𑀕𑀡𑁆𑀞𑀸 𑀧𑀼𑀧𑁆𑀨𑀲𑀸𑀝𑀓𑀸;_  
_𑀢𑁂𑀤𑀡𑁆𑀟𑀺𑀓𑀸 𑀏𑀓𑀲𑀺𑀔𑀸,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(286.)

5148\. _𑀆𑀚𑀻𑀯𑀓𑀸 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀸𑀯𑀻,_  
_𑀕𑁄𑀥𑀫𑁆𑀫𑀸 𑀤𑁂𑀯𑀥𑀫𑁆𑀫𑀺𑀓𑀸;_  
_𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀥𑀭𑀸 𑀏𑀢𑁂,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(287.)

5149\. _𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑁄𑀥𑀧𑀼𑀕𑁆𑀕𑀦𑀺𑀓𑀸 𑀩𑀳𑀽;_  
_𑀢𑀧𑀲𑁆𑀲𑀻 𑀯𑀦𑀘𑀸𑀭𑀻 𑀘,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(288.)

5150\. _𑀑𑀟𑁆𑀟𑀓𑀸 𑀤𑀫𑀺𑀍𑀆 𑀘𑁂𑀯,_  
_𑀲𑀸𑀓𑀼𑀍𑀆 𑀫𑀮𑀯𑀸𑀍𑀅𑀓𑀸;_  
_𑀲𑀯𑀭𑀸 𑀬𑁄𑀦𑀓𑀸 𑀘𑁂𑀯,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(289.)

5151\. _𑀅𑀦𑁆𑀥𑀓𑀸 𑀫𑀼𑀡𑁆𑀟𑀓𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀓𑁄𑀝𑀮𑀸 𑀳𑀦𑀼𑀯𑀺𑀦𑁆𑀤𑀓𑀸;_  
_𑀆𑀭𑀸𑀯𑀘𑀻𑀦𑀭𑀝𑁆𑀞𑀸 𑀘,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(290.)

5152\. _𑀅𑀮𑀲𑀦𑁆𑀤𑀓𑀸 𑀧𑀮𑁆𑀮𑀯𑀓𑀸,_  
_𑀥𑀫𑁆𑀫𑀭𑀸 𑀦𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀼𑀲𑀸;_  
_𑀕𑁂𑀳𑀺𑀓𑀸 𑀘𑁂𑀢𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(291.)

5153\. _𑀫𑀸𑀥𑀼𑀭𑀓𑀸 𑀓𑁄𑀲𑀮𑀓𑀸,_  
_𑀓𑀮𑀺𑀗𑁆𑀕𑀸 𑀳𑀢𑁆𑀣𑀺𑀧𑁄𑀭𑀺𑀓𑀸;_  
_𑀇𑀲𑀺𑀡𑁆𑀟𑀸 𑀫𑀓𑁆𑀓𑀮𑀸 𑀘𑁂𑀯,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(292.)

5154\. _𑀘𑁂𑀮𑀸𑀯𑀓𑀸 𑀆𑀭𑀩𑁆𑀪𑀸 𑀘,_  
_𑀑𑀖𑀼𑀍𑀳𑀸 𑀫𑁂𑀖𑀮𑀸 𑀩𑀳𑀽;_  
_𑀔𑀼𑀤𑁆𑀤𑀓𑀸 𑀲𑀼𑀤𑁆𑀤𑀓𑀸 𑀘𑁂𑀯,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(293.)

5155\. _𑀭𑁄𑀳𑀡𑀸 𑀲𑀺𑀦𑁆𑀥𑀯𑀸 𑀘𑁂𑀯,_  
_𑀘𑀺𑀢𑀓𑀸 𑀏𑀓𑀓𑀡𑁆𑀡𑀺𑀓𑀸;_  
_𑀲𑀼𑀭𑀝𑁆𑀞𑀸 𑀅𑀧𑀭𑀦𑁆𑀢𑀸 𑀘,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(294.)

5156\. _𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀸 𑀓𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀫𑀮𑁆𑀮𑀲𑁄𑀯𑀡𑁆𑀡𑀪𑀽𑀫𑀺𑀓𑀸;_  
_𑀯𑀚𑁆𑀚𑀻𑀢𑀗𑁆𑀕𑀸 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(295.)

5157\. _𑀦𑀍𑀅𑀓𑀸𑀭𑀸 𑀧𑁂𑀲𑀓𑀸𑀭𑀸,_  
_𑀘𑀫𑁆𑀫𑀓𑀸𑀭𑀸 𑀘 𑀢𑀘𑁆𑀙𑀓𑀸;_  
_𑀓𑀫𑁆𑀫𑀸𑀭𑀸 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀸 𑀘,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(296.)

5158\. _𑀫𑀡𑀺𑀓𑀸𑀭𑀸 𑀮𑁄𑀳𑀓𑀸𑀭𑀸,_  
_𑀲𑁄𑀡𑁆𑀡𑀓𑀸𑀭𑀸 𑀘 𑀤𑀼𑀲𑁆𑀲𑀺𑀓𑀸;_  
_𑀢𑀺𑀧𑀼𑀓𑀸𑀭𑀸 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(297.)

5159\. _𑀉𑀲𑀼𑀓𑀸𑀭𑀸 𑀪𑀫𑀓𑀸𑀭𑀸,_  
_𑀧𑁂𑀲𑀓𑀸𑀭𑀸 𑀘 𑀕𑀦𑁆𑀥𑀺𑀓𑀸;_  
_𑀭𑀚𑀓𑀸 𑀢𑀼𑀦𑁆𑀦𑀯𑀸𑀬𑀸 𑀘,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(298.)

5160\. _𑀢𑁂𑀮𑀺𑀓𑀸 𑀓𑀝𑁆𑀞𑀳𑀸𑀭𑀸 𑀘,_  
_𑀉𑀤𑀳𑀸𑀭𑀸 𑀘 𑀧𑁂𑀲𑁆𑀲𑀺𑀓𑀸;_  
_𑀲𑀽𑀧𑀺𑀓𑀸 𑀲𑀽𑀧𑀭𑀓𑁆𑀔𑀸 𑀘,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(299.)

5161\. _𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀩𑀦𑁆𑀥𑀺𑀓𑀸 𑀧𑀼𑀧𑁆𑀨𑀙𑀟𑁆𑀟𑀓𑀸;_  
_𑀳𑀢𑁆𑀣𑀸𑀭𑀼𑀳𑀸 𑀳𑀢𑁆𑀣𑀺𑀧𑀸𑀮𑀸,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀖𑀭𑀁𑁇_  


(300.)

5162\. _𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀜𑁆𑀜𑁄,_  
_𑀫𑀫𑀢𑁆𑀣𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀲𑀢𑁆𑀢𑀯𑀡𑁆𑀡𑁂𑀦 𑀭𑀢𑀦𑁂𑀦,_  
_𑀊𑀦𑀢𑁆𑀣𑀁 𑀧𑀽𑀭𑀬𑀸𑀫𑀳𑀁𑁇_  


(301.)

5163\. _𑀬𑁂 𑀫𑀬𑀸 𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀦𑀸𑀦𑀸𑀯𑀡𑁆𑀡𑀸 𑀩𑀳𑀽 𑀚𑀦𑀸;_  
_𑀢𑁂𑀲𑀸𑀳𑀁 𑀘𑀺𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀢𑀧𑁆𑀧𑀬𑀺𑀁 𑀭𑀢𑀦𑁂𑀦𑀳𑀁𑁇_  


(302.)

5164\. _𑀯𑀕𑁆𑀕𑀽𑀲𑀼 𑀪𑀸𑀲𑀫𑀸𑀦𑀸𑀲𑀼,_  
_𑀯𑀚𑁆𑀚𑀫𑀸𑀦𑀸𑀲𑀼 𑀪𑁂𑀭𑀺𑀲𑀼;_  
_𑀲𑀗𑁆𑀔𑁂𑀲𑀼 𑀥𑀫𑀬𑀦𑁆𑀢𑁂𑀲𑀼,_  
_𑀲𑀓𑀕𑁂𑀳𑁂 𑀭𑀫𑀸𑀫𑀳𑀁𑁇_  


(303.)

5165\. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀲𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸𑀲𑀯𑁂𑀳𑀺 𑀲𑁄𑁇_  


(304.)

5166\. _𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀺𑀢𑁄 𑀯𑀻𑀣𑀺𑀁,_  
_𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑁄 𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀤𑀻𑀧𑀭𑀼𑀓𑁆𑀔𑁄𑀯 𑀚𑁄𑀢𑀢𑀺𑁇_  


(305.)

5167\. _𑀯𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀪𑁂𑀭𑀺𑀬𑁄 𑀲𑀩𑁆𑀩𑀸,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁂;_  
_𑀧𑀪𑀸 𑀦𑀺𑀤𑁆𑀥𑀸𑀯𑀢𑁂 𑀢𑀲𑁆𑀲,_  
_𑀲𑀢𑀭𑀁𑀲𑀻𑀯 𑀉𑀕𑁆𑀕𑀢𑁄𑁇_  


(306.)

5168\. _𑀓𑀯𑀸𑀝𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀬𑀸𑀧𑀺,_  
_𑀧𑀯𑀺𑀝𑁆𑀞𑁂𑀦 𑀘 𑀭𑀲𑁆𑀫𑀺𑀦𑀸;_  
_𑀅𑀦𑁆𑀢𑁄𑀖𑀭𑁂𑀲𑀼 𑀯𑀺𑀧𑀼𑀮𑁄,_  
_𑀆𑀮𑁄𑀓𑁄 𑀆𑀲𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(307.)

5169\. _𑀧𑀪𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀧𑀸𑀭𑀺𑀲𑀚𑁆𑀚𑁂 𑀅𑀯𑁄𑀘𑀳𑀁;_  
_‘𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀇𑀫𑀁 𑀯𑀻𑀣𑀺𑀫𑀼𑀧𑀸𑀕𑀢𑁄’𑁇_  


(308.)

5170\. _𑀔𑀺𑀧𑁆𑀧𑀁 𑀑𑀭𑀼𑀬𑁆𑀳 𑀧𑀸𑀲𑀸𑀤𑀸,_  
_𑀅𑀕𑀫𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀡𑀁;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(309.)

5171\. _‘𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀢𑀼 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀲𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀲𑁄 𑀫𑀼𑀦𑀺’𑁇_  


(310.)

5172\. _𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀅𑀪𑀺𑀦𑁂𑀲𑀺𑀁 𑀲𑀓𑀁 𑀖𑀭𑀁;_  
_𑀢𑀢𑁆𑀣 𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦,_  
_𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀺𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁𑁇_  


(311.)

5173\. _𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀓𑀸𑀮𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀲𑀢𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀢𑀽𑀭𑀺𑀬𑁂𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑁇_  


(312.)

5174\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀅𑀦𑁆𑀢𑁄𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(313.)

5175\. _‘𑀬𑁄 𑀫𑀁 𑀢𑀽𑀭𑀺𑀬𑁂𑀳𑀼𑀧𑀝𑁆𑀞𑀸𑀲𑀺,_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀜𑁆𑀘𑀤𑀸𑀲𑀺 𑀫𑁂;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(314.)

5176\. _𑀧𑀳𑀽𑀢𑀪𑀓𑁆𑀔𑁄 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀳𑀺𑀭𑀜𑁆𑀜𑁄 𑀲𑀪𑁄𑀚𑀦𑁄;_  
_𑀘𑀢𑀼𑀤𑀻𑀧𑁂 𑀏𑀓𑀭𑀚𑁆𑀚𑀁,_  
_𑀓𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀬𑀁 𑀦𑀭𑁄𑁇_  


(315.)

5177\. _𑀧𑀜𑁆𑀘 𑀲𑀻𑀮𑁂 𑀲𑀫𑀸𑀤𑀸𑀬,_  
_𑀤𑀲𑀓𑀫𑁆𑀫𑀧𑀣𑁂 𑀢𑀢𑁄;_  
_𑀲𑀫𑀸𑀤𑀸𑀬 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀧𑀭𑀺𑀲𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(316.)

5178\. _𑀢𑀽𑀭𑀺𑀬𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀪𑁂𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀯𑀚𑁆𑀚𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀫𑀁 𑀦𑀺𑀘𑁆𑀘𑀁,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(317.)

5179\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀘𑀢𑀼𑀲𑀝𑁆𑀞𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(318.)

5180\. _𑀘𑀢𑀼𑀲𑀝𑁆𑀞𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(319.)

5181\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(320.)

5182\. _𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀅𑀦𑀽𑀦𑀪𑁄𑀕𑁄 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(321.)

5183\. _𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀪𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀳𑀺𑀁 𑀇𑀫𑀁𑁇_  


(322.)

5184\. _𑀲𑁄 𑀧𑀘𑁆𑀙𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀸𑀲𑀦𑁂𑀪𑀺𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(323.)

5185\. _𑀆𑀭𑀸𑀥𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑁄𑀢𑀫𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁;_  
_𑀓𑀺𑀮𑁂𑀲𑁂 𑀛𑀸𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀭𑀳𑀸𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(324.)

5186\. _𑀯𑀺𑀧𑀺𑀦𑁂 𑀩𑁆𑀬𑀕𑁆𑀖𑀭𑀸𑀚𑀸𑀯,_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀸𑀯 𑀓𑁂𑀲𑀭𑀻;_  
_𑀅𑀪𑀻𑀢𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀚𑁆𑀚,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


(325.)

5187\. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸,_  
_𑀤𑀮𑀺𑀤𑁆𑀤𑁂 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀫𑁆𑀳𑀺 𑀯𑀸;_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀁 𑀫𑁂 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(326.)

5188\. _𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(327.)

5189\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(328.)

5190\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(329.)

5191\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5192\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀚𑀢𑀼𑀓𑀡𑁆𑀡𑀺𑀢𑁆𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5193\. 𑀚𑀢𑀼𑀓𑀡𑁆𑀡𑀺𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
