# 𑁪𑁧𑁇𑁪𑁇𑁧 𑀲𑁂𑀢𑀼𑀤𑀸𑀦𑀆𑀦𑀺𑀲𑀁𑀲

### 𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕

### 𑀥𑁄𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑁂𑀢𑀼𑀤𑀸𑀦𑀆𑀦𑀺𑀲𑀁𑀲

(86.)

4938. _𑀤𑀭𑀺𑀢𑁄 𑀧𑀩𑁆𑀩𑀢𑀢𑁄 𑀯𑀸,_  
_𑀭𑀼𑀓𑁆𑀔𑀢𑁄 𑀧𑀢𑀺𑀢𑁄𑀧𑀺𑀬𑀁;_  
_𑀘𑀼𑀢𑁄𑀧𑀺 𑀮𑀘𑁆𑀙𑀢𑀻 𑀞𑀸𑀦𑀁,_  
_𑀲𑁂𑀢𑀼𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(87.)

4939. _𑀯𑀺𑀭𑀽𑀍𑀳𑀫𑀽𑀮𑀲𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀫𑀺𑀯 𑀫𑀸𑀮𑀼𑀢𑁄;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀦𑀧𑁆𑀧𑀲𑀳𑀦𑁆𑀢𑀺,_  
_𑀲𑁂𑀢𑀼𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(88.)

4940. _𑀦𑀸𑀲𑁆𑀲 𑀘𑁄𑀭𑀸 𑀧𑀲𑀳𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀢𑀭𑀺𑀲𑁆𑀲𑀢𑀸𑀫𑀺𑀢𑁆𑀢𑁂,_  
_𑀲𑁂𑀢𑀼𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(89.)

4941. _𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀕𑀢𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀓𑀞𑀺𑀦𑀸𑀢𑀧𑀢𑀸𑀧𑀺𑀢𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑁇_  


(90.)

4942. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁;_  
_𑀢𑀲𑁆𑀲 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(91.)

4943. _𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲𑀸 𑀯𑀸𑀢𑀚𑀯𑀸,_  
_𑀲𑀺𑀦𑁆𑀥𑀯𑀸 𑀲𑀻𑀖𑀯𑀸𑀳𑀦𑀸;_  
_𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑀁 𑀉𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑁂𑀢𑀼𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(92.)

4944. _𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀔𑀺𑀢𑁄𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀯𑁂𑀳𑀸𑀲𑀁 𑀫𑀦𑀼𑀚𑀲𑁆𑀲𑁂𑀯,_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(93.)

4945. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(94.)

4946. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(95.)

4947. _𑀅𑀳𑁄 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀓𑁂;_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀢𑁄𑀳𑀁 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(96.)

4948. _𑀧𑀥𑀸𑀦𑀁 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(97.)

4949. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(98.)

4950. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(99.)

4951. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


4952. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑁄𑀢𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

4953. 𑀥𑁄𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
