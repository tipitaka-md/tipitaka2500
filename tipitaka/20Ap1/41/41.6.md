# 𑁪𑁧𑁇𑁬 𑀦𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(161.)

5017\. _“𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀦𑀦𑁂 𑀅𑀳𑀁;_  
_𑀧𑀲𑀤𑀁 𑀫𑀺𑀕𑀫𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀳𑀁𑁇_  


(162.)

5018\. _𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀯𑀺𑀯𑁂𑀓𑀓𑀸𑀫𑁄 𑀲𑁄 𑀥𑀻𑀭𑁄,_  
_𑀯𑀦𑀫𑀚𑁆𑀛𑁄𑀕𑀳𑀻 𑀢𑀤𑀸𑁇_  


(163.)

5019\. _𑀘𑀢𑀼𑀤𑀡𑁆𑀟𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀘𑀢𑀼𑀝𑁆𑀞𑀸𑀦𑁂 𑀞𑀧𑁂𑀲𑀳𑀁;_  
_𑀫𑀡𑁆𑀟𑀧𑀁 𑀲𑀼𑀓𑀢𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀤𑁆𑀫𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺 𑀙𑀸𑀤𑀬𑀺𑀁𑁇_  


(164.)

5020\. _𑀫𑀡𑁆𑀟𑀧𑀁 𑀙𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑀬𑀺𑀁;_  
_𑀥𑀦𑀼𑀁 𑀢𑀢𑁆𑀣𑁂𑀯 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁆𑀧,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(165.)

5021\. _𑀦𑀘𑀺𑀭𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑁆𑀲,_  
_𑀩𑁆𑀬𑀸𑀥𑀺 𑀫𑁂 𑀉𑀤𑀧𑀚𑁆𑀚𑀣;_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀲𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(166.)

5022\. _𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀢𑀼𑀲𑀺𑀢𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁;_  
_𑀢𑀢𑁆𑀣 𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀁 𑀩𑁆𑀬𑀫𑁆𑀳𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀺 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(167.)

5023\. _𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀳𑀬𑀯𑀸𑀳𑀺𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀬𑀸𑀦𑀫𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀆𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀬𑀸𑀦𑀁,_  
_𑀕𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁𑁇_  


(168.)

5024\. _𑀢𑀢𑁄 𑀫𑁂 𑀦𑀺𑀬𑁆𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀤𑁂𑀯𑀪𑀽𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀫𑀡𑁆𑀟𑀧𑁄 𑀫𑁂 𑀥𑀭𑀻𑀬𑀢𑀺𑁇_  


(169.)

5025\. _𑀲𑀬𑀦𑁂𑀳𑀁 𑀢𑀼𑀯𑀝𑁆𑀝𑀸𑀫𑀺,_  
_𑀅𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀧𑀼𑀧𑁆𑀨𑀲𑀦𑁆𑀣𑀢𑁂;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀘 𑀧𑀤𑀼𑀫𑀸,_  
_𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀦𑀺𑀘𑁆𑀘𑀓𑀸𑀮𑀺𑀓𑀁𑁇_  


(170.)

5026\. _𑀫𑀭𑀻𑀘𑀺𑀓𑁂 𑀨𑀦𑁆𑀤𑀫𑀸𑀦𑁂,_  
_𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑁂 𑀘 𑀆𑀢𑀧𑁂;_  
_𑀦 𑀫𑀁 𑀢𑀸𑀧𑁂𑀢𑀺 𑀆𑀢𑀸𑀧𑁄,_  
_𑀫𑀡𑁆𑀟𑀧𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(171.)

5027\. _𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄,_  
_𑀅𑀧𑀸𑀬𑀸 𑀧𑀺𑀳𑀺𑀢𑀸 𑀫𑀫;_  
_𑀫𑀡𑁆𑀟𑀧𑁂 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀸,_  
_𑀲𑀦𑁆𑀢𑀸𑀧𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(172.)

5028\. _𑀫𑀳𑀻𑀲𑀜𑁆𑀜𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀬,_  
_𑀮𑁄𑀡𑀢𑁄𑀬𑀁 𑀢𑀭𑀸𑀫𑀳𑀁;_  
_𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(173.)

5029\. _𑀅𑀧𑀣𑀫𑁆𑀧𑀺 𑀧𑀣𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂;_  
_𑀅𑀳𑁄 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(174.)

5030\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀆𑀲𑀯𑀸 𑀫𑁂 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(175.)

5031\. _𑀚𑀳𑀺𑀢𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀚𑀸𑀢𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀑𑀭𑀲𑁄 𑀅𑀳𑀁;_  
_𑀤𑀸𑀬𑀸𑀤𑁄𑀫𑁆𑀳𑀺 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(176.)

5032\. _𑀆𑀭𑀸𑀥𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀲𑀼𑀕𑀢𑀁,_  
_𑀕𑁄𑀢𑀫𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁;_  
_𑀥𑀫𑁆𑀫𑀥𑀚𑁄 𑀥𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(177.)

5033\. _𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑁄𑀢𑀫𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀗𑁆𑀕𑀯𑀁;_  
_𑀧𑀸𑀭𑀗𑁆𑀕𑀫𑀦𑀺𑀬𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(178.)

5034\. _𑀅𑀚𑁆𑀛𑀺𑀝𑁆𑀞𑁄 𑀓𑀣𑀬𑀻 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀦𑀺𑀧𑀼𑀡𑀁 𑀧𑀤𑀁;_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁𑁇_  


(179.)

5035\. _𑀅𑀳𑁄 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀚𑀸𑀢𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(180.)

5036\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(181.)

5037\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(182.)

5038\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5039\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5040\. 𑀦𑀦𑁆𑀤𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
