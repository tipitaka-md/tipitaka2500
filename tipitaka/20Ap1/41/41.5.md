# 𑁪𑁧𑁇𑁫 𑀉𑀧𑀲𑀺𑀯𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀲𑀺𑀯𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(100.)

4954\. _“𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂,_  
_𑀅𑀦𑁄𑀫𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄;_  
_𑀅𑀲𑁆𑀲𑀫𑁄 𑀲𑀼𑀓𑀢𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀸 𑀲𑀼𑀫𑀸𑀧𑀺𑀢𑀸𑁇_  


(101.)

4955\. _𑀦𑀤𑀻 𑀘 𑀲𑀦𑁆𑀤𑀢𑀻 𑀢𑀢𑁆𑀣,_  
_𑀲𑀼𑀧𑀢𑀺𑀢𑁆𑀣𑀸 𑀫𑀦𑁄𑀭𑀫𑀸;_  
_𑀅𑀦𑀽𑀧𑀢𑀺𑀢𑁆𑀣𑁂 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀓𑀸 𑀩𑀳𑀽𑁇_  


(102.)

4956\. _𑀧𑀸𑀞𑀻𑀦𑀸 𑀧𑀸𑀯𑀼𑀲𑀸 𑀫𑀘𑁆𑀙𑀸,_  
_𑀩𑀮𑀚𑀸 𑀫𑀼𑀜𑁆𑀚𑀭𑁄𑀳𑀺𑀢𑀸;_  
_𑀫𑀘𑁆𑀙𑀓𑀘𑁆𑀙𑀧𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀦𑀤𑀺𑀓𑀸 𑀲𑀦𑁆𑀤𑀢𑁂 𑀢𑀤𑀸𑁇_  


(103.)

4957\. _𑀢𑀺𑀫𑀺𑀭𑀸 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀢𑀢𑁆𑀣,_  
_𑀅𑀲𑁄𑀓𑀸 𑀔𑀼𑀤𑁆𑀤𑀫𑀸𑀮𑀓𑀸;_  
_𑀧𑀼𑀦𑁆𑀦𑀸𑀕𑀸 𑀕𑀺𑀭𑀺𑀧𑀼𑀦𑁆𑀦𑀸𑀕𑀸,_  
_𑀲𑀫𑁆𑀧𑀯𑀦𑁆𑀢𑀺 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(104.)

4958\. _𑀓𑀼𑀝𑀚𑀸 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀢𑀢𑁆𑀣,_  
_𑀢𑀺𑀡𑀲𑀽𑀮𑀯𑀦𑀸𑀦𑀺 𑀘;_  
_𑀲𑀸𑀮𑀸 𑀘 𑀲𑀍𑀅𑀮𑀸 𑀢𑀢𑁆𑀣,_  
_𑀘𑀫𑁆𑀧𑀓𑀸 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀩𑀳𑀽𑁇_  


(105.)

4959\. _𑀅𑀚𑁆𑀚𑀼𑀦𑀸 𑀅𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀫𑀳𑀸𑀦𑀸𑀫𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀅𑀲𑀦𑀸 𑀫𑀥𑀼𑀕𑀦𑁆𑀥𑀻 𑀘,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀢𑁂 𑀫𑀫𑀲𑁆𑀲𑀫𑁂𑁇_  


(106.)

4960\. _𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀸 𑀧𑀸𑀝𑀮𑀺𑀓𑀸,_  
_𑀬𑀽𑀣𑀺𑀓𑀸 𑀘 𑀧𑀺𑀬𑀗𑁆𑀕𑀼𑀓𑀸;_  
_𑀩𑀺𑀫𑁆𑀩𑀺𑀚𑀸𑀮𑀓𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀅𑀟𑁆𑀠𑀬𑁄𑀚𑀦𑀁𑁇_  


(107.)

4961\. _𑀫𑀸𑀢𑀕𑁆𑀕𑀸𑀭𑀸 𑀲𑀢𑁆𑀢𑀮𑀺𑀬𑁄,_  
_𑀧𑀸𑀝𑀮𑀻 𑀲𑀺𑀦𑁆𑀤𑀼𑀯𑀸𑀭𑀓𑀸;_  
_𑀅𑀗𑁆𑀓𑁄𑀮𑀓𑀸 𑀩𑀳𑀽 𑀢𑀢𑁆𑀣,_  
_𑀢𑀸𑀮𑀓𑀼𑀝𑁆𑀞𑀺 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀲𑁂𑀮𑁂𑀬𑁆𑀬𑀓𑀸 𑀩𑀳𑀽 𑀢𑀢𑁆𑀣,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(108.)

4962\. _𑀏𑀢𑁂𑀲𑀼 𑀧𑀼𑀧𑁆𑀨𑀚𑀸𑀢𑁂𑀲𑀼,_  
_𑀲𑁄𑀪𑀦𑁆𑀢𑀺 𑀧𑀸𑀤𑀧𑀸 𑀩𑀳𑀽;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀢𑁂𑀦 𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀯𑀸𑀬𑀢𑁂 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁄𑁇_  


(109.)

4963\. _𑀳𑀭𑀻𑀢𑀓𑀸 𑀆𑀫𑀮𑀓𑀸,_  
_𑀅𑀫𑁆𑀩𑀚𑀫𑁆𑀩𑀼𑀯𑀺𑀪𑀻𑀢𑀓𑀸;_  
_𑀓𑁄𑀮𑀸 𑀪𑀮𑁆𑀮𑀸𑀢𑀓𑀸 𑀩𑀺𑀮𑁆𑀮𑀸,_  
_𑀨𑀸𑀭𑀼𑀲𑀓𑀨𑀮𑀸𑀦𑀺 𑀘𑁇_  


(110.)

4964\. _𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀸 𑀘 𑀧𑀺𑀬𑀸𑀮𑀸 𑀘,_  
_𑀫𑀥𑀼𑀓𑀸 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀬𑁄;_  
_𑀮𑀩𑀼𑀚𑀸 𑀧𑀦𑀲𑀸 𑀢𑀢𑁆𑀣,_  
_𑀓𑀤𑀮𑀻 𑀩𑀤𑀭𑀻𑀨𑀮𑀸𑁇_  


(111.)

4965\. _𑀅𑀫𑁆𑀩𑀸𑀝𑀓𑀸 𑀩𑀳𑀽 𑀢𑀢𑁆𑀣,_  
_𑀯𑀮𑁆𑀮𑀺𑀓𑀸𑀭𑀨𑀮𑀸𑀦𑀺 𑀘;_  
_𑀩𑀻𑀚𑀧𑀽𑀭𑀲𑀧𑀸𑀭𑀺𑀬𑁄,_  
_𑀨𑀮𑀺𑀢𑀸 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(112.)

4966\. _𑀆𑀍𑀅𑀓𑀸 𑀇𑀲𑀺𑀫𑀼𑀕𑁆𑀕𑀸 𑀘,_  
_𑀢𑀢𑁄 𑀫𑁄𑀤𑀨𑀮𑀸 𑀩𑀳𑀽;_  
_𑀅𑀯𑀝𑀸 𑀧𑀓𑁆𑀓𑀪𑀭𑀺𑀢𑀸,_  
_𑀧𑀺𑀮𑀓𑁆𑀔𑀼𑀤𑀼𑀫𑁆𑀩𑀭𑀸𑀦𑀺 𑀘𑁇_  


(113.)

4967\. _𑀧𑀺𑀧𑁆𑀨𑀮𑀻 𑀫𑀭𑀻𑀘𑀸 𑀢𑀢𑁆𑀣,_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸 𑀘 𑀓𑀧𑀺𑀢𑁆𑀣𑀦𑀸;_  
_𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀓𑀸 𑀩𑀳𑀯𑁄,_  
_𑀓𑀡𑁆𑀟𑀼𑀧𑀡𑁆𑀡𑀸 𑀘 𑀳𑀭𑀺𑀬𑁄𑁇_  


(114.)

4968\. _𑀏𑀢𑁂 𑀘𑀜𑁆𑀜𑁂 𑀘 𑀩𑀳𑀯𑁄,_  
_𑀨𑀮𑀺𑀢𑀸 𑀅𑀲𑁆𑀲𑀫𑁂 𑀫𑀫;_  
_𑀧𑀼𑀧𑁆𑀨𑀭𑀼𑀓𑁆𑀔𑀸𑀧𑀺 𑀩𑀳𑀯𑁄,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(115.)

4969\. _𑀆𑀮𑀼𑀯𑀸 𑀘 𑀓𑀍𑀅𑀫𑁆𑀩𑀸 𑀘,_  
_𑀩𑀺𑀍𑀆𑀮𑀻 𑀢𑀓𑁆𑀓𑀮𑀸𑀦𑀺 𑀘;_  
_𑀆𑀮𑀓𑀸 𑀢𑀸𑀮𑀓𑀸 𑀘𑁂𑀯,_  
_𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀲𑁆𑀲𑀫𑁂 𑀫𑀫𑁇_  


(116.)

4970\. _𑀅𑀲𑁆𑀲𑀫𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂 𑀫𑁂,_  
_𑀫𑀳𑀸𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁄 𑀅𑀳𑀼;_  
_𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑁄 𑀲𑀻𑀢𑀚𑀮𑁄,_  
_𑀲𑀼𑀧𑀢𑀺𑀢𑁆𑀣𑁄 𑀫𑀦𑁄𑀭𑀫𑁄𑁇_  


(117.)

4971\. _𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀸 𑀩𑀳𑀽 𑀢𑀢𑁆𑀣,_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀲𑀫𑀸𑀬𑀼𑀢𑀸;_  
_𑀫𑀦𑁆𑀤𑀸𑀮𑀓𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀦𑀸𑀦𑀸𑀕𑀦𑁆𑀥𑀲𑀫𑁂𑀭𑀺𑀢𑀸𑁇_  


(118.)

4972\. _𑀕𑀩𑁆𑀪𑀁 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺 𑀧𑀤𑀼𑀫𑀸,_  
_𑀅𑀜𑁆𑀜𑁂 𑀧𑀼𑀧𑁆𑀨𑀦𑁆𑀢𑀺 𑀓𑁂𑀲𑀭𑀻;_  
_𑀑𑀧𑀼𑀧𑁆𑀨𑀧𑀢𑁆𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀧𑀤𑀼𑀫𑀸𑀓𑀡𑁆𑀡𑀺𑀓𑀸 𑀩𑀳𑀽𑁇_  


(119.)

4973\. _𑀫𑀥𑀼 𑀪𑀺𑀲𑀫𑁆𑀳𑀸 𑀲𑀯𑀢𑀺,_  
_𑀔𑀻𑀭𑀁 𑀲𑀧𑁆𑀧𑀺 𑀫𑀼𑀮𑀸𑀍𑀇𑀪𑀺;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀢𑁂𑀦 𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀦𑀸𑀦𑀸𑀕𑀦𑁆𑀥𑀲𑀫𑁂𑀭𑀺𑀢𑀸𑁇_  


(120.)

4974\. _𑀓𑀼𑀫𑀼𑀤𑀸 𑀅𑀫𑁆𑀩𑀕𑀦𑁆𑀥𑀺 𑀘,_  
_𑀦𑀬𑀺𑀢𑀸 𑀤𑀺𑀲𑁆𑀲𑀭𑁂 𑀩𑀳𑀽;_  
_𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑀲𑁆𑀲𑀸𑀦𑀼𑀓𑀽𑀮𑀁,_  
_𑀓𑁂𑀢𑀓𑀸 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀩𑀳𑀽𑁇_  


(121.)

4975\. _𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀸 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀸 𑀘,_  
_𑀲𑁂𑀢𑀯𑀸𑀭𑀻 𑀲𑀼𑀕𑀦𑁆𑀥𑀺𑀓𑀸;_  
_𑀓𑀼𑀫𑁆𑀪𑀺𑀮𑀸 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀕𑀳𑀓𑀸 𑀢𑀢𑁆𑀣 𑀚𑀸𑀬𑀭𑁂𑁇_  


(122.)

4976\. _𑀉𑀕𑁆𑀕𑀸𑀳𑀓𑀸 𑀅𑀚𑀕𑀭𑀸,_  
_𑀢𑀢𑁆𑀣 𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁂 𑀩𑀳𑀽;_  
_𑀧𑀸𑀞𑀻𑀦𑀸 𑀧𑀸𑀯𑀼𑀲𑀸 𑀫𑀘𑁆𑀙𑀸,_  
_𑀩𑀮𑀚𑀸 𑀫𑀼𑀜𑁆𑀚𑀭𑁄𑀳𑀺𑀢𑀸𑁇_  


(123.)

4977\. _𑀫𑀘𑁆𑀙𑀓𑀘𑁆𑀙𑀧𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀅𑀣𑁄 𑀧𑀧𑀝𑀓𑀸𑀳𑀺 𑀘;_  
_𑀧𑀸𑀭𑁂𑀯𑀢𑀸 𑀭𑀯𑀺𑀳𑀁𑀲𑀸,_  
_𑀓𑀼𑀓𑀼𑀢𑁆𑀣𑀸 𑀘 𑀦𑀤𑀻𑀘𑀭𑀸𑁇_  


(124.)

4978\. _𑀤𑀺𑀦𑁆𑀤𑀺𑀪𑀸 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀸 𑀘,_  
_𑀧𑀫𑁆𑀧𑀓𑀸 𑀚𑀻𑀯𑀚𑀻𑀯𑀓𑀸;_  
_𑀓𑀮𑀦𑁆𑀤𑀓𑀸 𑀉𑀓𑁆𑀓𑀼𑀲𑀸 𑀘,_  
_𑀲𑁂𑀦𑀓𑀸 𑀉𑀤𑁆𑀥𑀭𑀸 𑀩𑀳𑀽𑁇_  


(125.)

4979\. _𑀓𑁄𑀝𑁆𑀞𑀓𑀸 𑀲𑀼𑀓𑀧𑁄𑀢𑀸 𑀘,_  
_𑀢𑀼𑀮𑀺𑀬𑀸 𑀘𑀫𑀭𑀸 𑀩𑀳𑀽;_  
_𑀓𑀸𑀭𑁂𑀦𑀺𑀬𑁄 𑀘 𑀢𑀺𑀮𑀓𑀸,_  
_𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀢𑀁 𑀲𑀭𑀁𑁇_  


(126.)

4980\. _𑀲𑀻𑀳𑀸 𑀩𑁆𑀬𑀕𑁆𑀖𑀸 𑀘 𑀤𑀻𑀧𑀻 𑀘,_  
_𑀅𑀘𑁆𑀙𑀓𑁄𑀓𑀢𑀭𑀘𑁆𑀙𑀓𑀸;_  
_𑀯𑀸𑀦𑀭𑀸 𑀓𑀺𑀦𑁆𑀦𑀭𑀸 𑀘𑁂𑀯,_  
_𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(127.)

4981\. _𑀢𑀸𑀦𑀺 𑀕𑀦𑁆𑀥𑀸𑀦𑀺 𑀖𑀸𑀬𑀦𑁆𑀢𑁄,_  
_𑀪𑀓𑁆𑀔𑀬𑀦𑁆𑀢𑁄 𑀨𑀮𑀸𑀦𑀳𑀁;_  
_𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑀁 𑀧𑀺𑀯𑀦𑁆𑀢𑁄 𑀘,_  
_𑀯𑀲𑀸𑀫𑀺 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(128.)

4982\. _𑀏𑀡𑀻𑀫𑀺𑀕𑀸 𑀯𑀭𑀸𑀳𑀸 𑀘,_  
_𑀧𑀲𑀤𑀸 𑀔𑀼𑀤𑁆𑀤𑀭𑀽𑀧𑀓𑀸;_  
_𑀅𑀕𑁆𑀕𑀺𑀓𑀸 𑀚𑁄𑀢𑀺𑀓𑀸 𑀘𑁂𑀯,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(129.)

4983\. _𑀳𑀁𑀲𑀸 𑀓𑁄𑀜𑁆𑀘𑀸 𑀫𑀬𑀽𑀭𑀸 𑀘,_  
_𑀲𑀸𑀮𑀺𑀓𑀸𑀧𑀺 𑀘 𑀓𑁄𑀓𑀺𑀮𑀸;_  
_𑀫𑀚𑁆𑀚𑀸𑀭𑀺𑀓𑀸 𑀩𑀳𑀽 𑀢𑀢𑁆𑀣,_  
_𑀓𑁄𑀲𑀺𑀓𑀸 𑀧𑁄𑀝𑁆𑀞𑀲𑀻𑀲𑀓𑀸𑁇_  


(130.)

4984\. _𑀧𑀺𑀲𑀸𑀘𑀸 𑀤𑀸𑀦𑀯𑀸 𑀘𑁂𑀯,_  
_𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀸 𑀭𑀓𑁆𑀔𑀲𑀸 𑀩𑀳𑀽;_  
_𑀕𑀭𑀼𑀍𑀆 𑀧𑀦𑁆𑀦𑀕𑀸 𑀘𑁂𑀯,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(131.)

4985\. _𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸 𑀇𑀲𑀬𑁄,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀓𑀫𑀡𑁆𑀟𑀮𑀼𑀥𑀭𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀅𑀚𑀺𑀦𑀼𑀢𑁆𑀢𑀭𑀯𑀸𑀲𑀦𑀸;_  
_𑀚𑀝𑀸𑀪𑀸𑀭𑀪𑀭𑀺𑀢𑀸𑀯,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(132.)

4986\. _𑀬𑀼𑀕𑀫𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀸,_  
_𑀦𑀺𑀧𑀓𑀸 𑀲𑀦𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑁂𑀦 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(133.)

4987\. _𑀯𑀸𑀓𑀘𑀻𑀭𑀁 𑀥𑀼𑀦𑀦𑁆𑀢𑀸 𑀢𑁂,_  
_𑀨𑁄𑀝𑁂𑀦𑁆𑀢𑀸𑀚𑀺𑀦𑀘𑀫𑁆𑀫𑀓𑀁;_  
_𑀲𑀩𑀮𑁂𑀳𑀺 𑀉𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀸,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀅𑀫𑁆𑀩𑀭𑁂 𑀢𑀤𑀸𑁇_  


(134.)

4988\. _𑀦 𑀢𑁂 𑀤𑀓𑀁 𑀆𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀓𑀝𑁆𑀞𑀁 𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀤𑀸𑀭𑀼𑀓𑀁;_  
_𑀲𑀬𑀜𑁆𑀘 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(135.)

4989\. _𑀮𑁄𑀳𑀤𑁄𑀡𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀦𑀫𑀚𑁆𑀛𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂;_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀸𑀯 𑀫𑀳𑀸𑀦𑀸𑀕𑀸,_  
_𑀅𑀲𑀫𑁆𑀪𑀻𑀢𑀸𑀯 𑀓𑁂𑀲𑀭𑀻𑁇_  


(136.)

4990\. _𑀅𑀜𑁆𑀜𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀕𑁄𑀬𑀸𑀦𑀁,_  
_𑀅𑀜𑁆𑀜𑁂 𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀤𑁂𑀳𑀓𑀁;_  
_𑀅𑀜𑁆𑀜𑁂 𑀘 𑀉𑀢𑁆𑀢𑀭𑀓𑀼𑀭𑀼𑀁,_  
_𑀲𑀓𑀁 𑀩𑀮𑀫𑀯𑀲𑁆𑀲𑀺𑀢𑀸𑁇_  


(137.)

4991\. _𑀢𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀏𑀓𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀉𑀕𑁆𑀕𑀢𑁂𑀚𑀸𑀦 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(138.)

4992\. _𑀅𑀚𑀺𑀦𑀘𑀫𑁆𑀫𑀲𑀤𑁆𑀤𑁂𑀦,_  
_𑀯𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑁂 𑀢𑀤𑀸;_  
_𑀏𑀤𑀺𑀲𑀸 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀲𑀺𑀲𑁆𑀲𑀸 𑀉𑀕𑁆𑀕𑀢𑀧𑀸 𑀫𑀫𑁇_  


(139.)

4993\. _𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀅𑀳𑀁 𑀢𑁂𑀳𑀺,_  
_𑀯𑀲𑀸𑀫𑀺 𑀫𑀫 𑀅𑀲𑁆𑀲𑀫𑁂;_  
_𑀢𑁄𑀲𑀺𑀢𑀸 𑀲𑀓𑀓𑀫𑁆𑀫𑁂𑀦,_  
_𑀯𑀺𑀦𑀻𑀢𑀸𑀧𑀺 𑀲𑀫𑀸𑀕𑀢𑀸𑁇_  


(140.)

4994\. _𑀆𑀭𑀸𑀥𑀬𑀺𑀁𑀲𑀼 𑀫𑀁 𑀏𑀢𑁂,_  
_𑀲𑀓𑀓𑀫𑁆𑀫𑀸𑀪𑀺𑀮𑀸𑀲𑀺𑀦𑁄;_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀘 𑀦𑀺𑀧𑀓𑀸,_  
_𑀅𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀲𑀼 𑀓𑁄𑀯𑀺𑀤𑀸𑁇_  


(141.)

4995\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀲𑀫𑀬𑀁 𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺 𑀯𑀺𑀦𑀸𑀬𑀓𑁄𑁇_  


(142.)

4996\. _𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀆𑀢𑀸𑀧𑀻 𑀦𑀺𑀧𑀓𑁄 𑀫𑀼𑀦𑀺;_  
_𑀧𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀬 𑀫𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(143.)

4997\. _𑀉𑀧𑀸𑀕𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀬𑀓𑀁;_  
_𑀢𑀺𑀡𑀲𑀦𑁆𑀣𑀭𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺 𑀑𑀓𑀺𑀭𑀺𑀁𑁇_  


(144.)

4998\. _𑀦𑀺𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀳𑀝𑁆𑀞𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑁄;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀩𑁆𑀩𑀢𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀅𑀕𑀍𑀉𑀁 𑀅𑀕𑁆𑀕𑀳𑀺𑀁 𑀅𑀳𑀁𑁇_  


(145.)

4999\. _𑀓𑀼𑀫𑁆𑀪𑀫𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀦𑀲𑀁 𑀤𑁂𑀯𑀕𑀦𑁆𑀥𑀺𑀓𑀁;_  
_𑀔𑀦𑁆𑀥𑁂 𑀆𑀭𑁄𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑀁𑁇_  


(146.)

5000\. _𑀨𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀕𑀍𑀉𑀁 𑀅𑀦𑀼𑀮𑀺𑀫𑁆𑀧𑀳𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀁 𑀅𑀯𑀦𑁆𑀤𑀺𑀳𑀁𑁇_  


(147.)

5001\. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀇𑀲𑀺𑀫𑀚𑁆𑀛𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(148.)

5002\. _‘𑀬𑁄 𑀫𑁂 𑀨𑀮𑀜𑁆𑀘 𑀅𑀕𑀍𑀉𑀁,_  
_𑀆𑀲𑀦𑀜𑁆𑀘 𑀅𑀤𑀸𑀲𑀺 𑀫𑁂;_  
_𑀢𑀫𑀳𑀁 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀡𑀸𑀣 𑀫𑀫 𑀪𑀸𑀲𑀢𑁄𑁇_  


(149.)

5003\. _𑀕𑀸𑀫𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑁂𑀲𑀼 𑀕𑀼𑀳𑀸𑀲𑀼 𑀯𑀸;_  
_𑀇𑀫𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑁄𑀚𑀦𑀁𑁇_  


(150.)

5004\. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀅𑀬𑀁 𑀦𑀭𑁄;_  
_𑀪𑁄𑀚𑀦𑁂𑀳𑀺 𑀘 𑀯𑀢𑁆𑀣𑁂𑀳𑀺,_  
_𑀧𑀭𑀺𑀲𑀁 𑀢𑀧𑁆𑀧𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(151.)

5005\. _𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀅𑀓𑁆𑀔𑁄𑀪𑀪𑁄𑀕𑁄 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀁𑀲𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑀬𑀁 𑀦𑀭𑁄𑁇_  


(152.)

5006\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀳𑀲𑁆𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(153.)

5007\. _𑀏𑀓𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(154.)

5008\. _𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(155.)

5009\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑁄,_  
_𑀑𑀭𑀲𑁄 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁄; ( )_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑀦𑀸𑀲𑀯𑁄’𑁇_  


(156.)

5010\. _𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑁄 𑀮𑀤𑁆𑀥𑁄 𑀫𑁂,_  
_𑀬𑁄𑀳𑀁 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀺𑀁 𑀦𑀸𑀬𑀓𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(157.)

5011\. _𑀕𑀸𑀫𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀧𑀩𑁆𑀪𑀸𑀭𑁂𑀲𑀼 𑀕𑀼𑀳𑀸𑀲𑀼 𑀯𑀸;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀪𑁄𑀚𑀦𑀁 𑀳𑁄𑀢𑀺 𑀫𑁂 𑀲𑀤𑀸𑁇_  


(158.)

5012\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(159.)

5013\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(160.)

5014\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5015\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀲𑀺𑀯𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---
