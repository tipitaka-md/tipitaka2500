# 𑁪𑁭𑁇𑁭 𑀮𑀩𑀼𑀚𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀸𑀮𑀓𑀼𑀲𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀮𑀩𑀼𑀚𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(32.)

6013\. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂𑁇_  


(33.)

6014\. _𑀮𑀩𑀼𑀚𑀁 𑀨𑀮𑀫𑀸𑀤𑀸𑀬,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁;_  
_𑀆𑀓𑀸𑀲𑁂𑀯 𑀞𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀺 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(34.)

6015\. _𑀯𑀺𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀚𑀦𑀦𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀸𑀯𑀳𑁄;_  
_𑀨𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(35.)

6016\. _𑀅𑀥𑀺𑀕𑀜𑁆𑀙𑀺𑀁 𑀢𑀤𑀸 𑀧𑀻𑀢𑀺𑀁,_  
_𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀼𑀔𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑁂𑀯 𑀭𑀢𑀦𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁𑁇_  


(36.)

6017\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(37.)

6018\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(38.)

6019\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(39.)

6020\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


6021\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀮𑀩𑀼𑀚𑀨𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

6022\. 𑀮𑀩𑀼𑀚𑀨𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
