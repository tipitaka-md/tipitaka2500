# 𑁪𑁭𑁇𑁩 𑀘𑀺𑀢𑀓𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀸𑀮𑀓𑀼𑀲𑀼𑀫𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀘𑀺𑀢𑀓𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(11.)

5984. _“𑀤𑀬𑁆𑀳𑀫𑀸𑀦𑁂 𑀲𑀭𑀻𑀭𑀫𑁆𑀳𑀺,_  
_𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀘𑀺𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(12.)

5985. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀺𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀬𑀺𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(13.)

5986. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(14.)

5987. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(15.)

5988. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5989. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀺𑀢𑀓𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5990. 𑀘𑀺𑀢𑀓𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
