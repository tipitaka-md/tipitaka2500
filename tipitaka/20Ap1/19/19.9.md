# 𑁧𑁯𑁇𑁯 𑀏𑀓𑀘𑀸𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀝𑀚𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀘𑀸𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(51.)

2647. _“𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀫𑀳𑀸𑀖𑁄𑀲𑁄 𑀢𑀤𑀸 𑀅𑀳𑀼;_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀘 𑀮𑁄𑀓𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀺,_  
_𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳 𑀲𑀭𑀸𑀕𑀺𑀦𑁄𑁇_  


(52.)

2648. _𑀢𑁂𑀲𑀁 𑀲𑀁𑀯𑁂𑀕𑀚𑀸𑀢𑀸𑀦𑀁,_  
_𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑀁;_  
_𑀲𑀩𑀮𑁂𑀦 𑀉𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑁄,_  
_𑀅𑀕𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(53.)

2649. _𑀫𑀦𑁆𑀤𑀸𑀭𑀯𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀗𑁆𑀕𑀻𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁;_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀓𑀸𑀮𑀫𑁆𑀳𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(54.)

2650. _𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀼,_  
_𑀅𑀘𑁆𑀙𑀭𑀸𑀬𑁄 𑀘 𑀫𑁂 𑀢𑀤𑀸;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(55.)

2651. _𑀲𑀝𑁆𑀞𑀺𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀇𑀢𑁄 𑀲𑁄𑀍𑀅𑀲 𑀢𑁂 𑀚𑀦𑀸;_  
_𑀫𑀳𑀸𑀫𑀮𑁆𑀮𑀚𑀦𑀸 𑀦𑀸𑀫,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(56.)

2652. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2653. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀘𑀸𑀭𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2654. 𑀏𑀓𑀘𑀸𑀭𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
