# 𑁧𑁯𑁇𑁨 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀝𑀚𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(6.)

2588. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀲𑀩𑁆𑀪𑀺 𑀯𑀡𑁆𑀡𑀺𑀢𑁄;_  
_𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑁄 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄,_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀩𑁆𑀩𑀢𑀦𑁆𑀢𑀭𑁂𑁇_  


(7.)

2589. _𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁂 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑁄,_  
_𑀤𑀓𑀚𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀁𑁇_  


(8.)

2590. _𑀉𑀪𑁄 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀉𑀧𑀸𑀕𑀘𑁆𑀙𑀺𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲𑀸𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(9.)

2591. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(10.)

2592. _𑀇𑀢𑁄 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀏𑀓𑁄 𑀆𑀲𑀺𑀁 𑀚𑀦𑀸𑀥𑀺𑀧𑁄;_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑀓𑀧𑁆𑀧𑁄 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(11.)

2593. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2594. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2595. 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
