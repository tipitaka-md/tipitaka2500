# 𑁧𑁯𑁇𑁭 𑀏𑀓𑀘𑀺𑀦𑁆𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀝𑀚𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀘𑀺𑀦𑁆𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(34.)

2626\. _“𑀬𑀤𑀸 𑀤𑁂𑀯𑁄 𑀤𑁂𑀯𑀓𑀸𑀬𑀸,_  
_𑀘𑀯𑀢𑁂 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀬𑀸;_  
_𑀢𑀬𑁄 𑀲𑀤𑁆𑀤𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀯𑀸𑀦𑀁 𑀅𑀦𑀼𑀫𑁄𑀤𑀢𑀁𑁇_  


(35.)

2627\. _‘𑀇𑀢𑁄 𑀪𑁄 𑀲𑀼𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀮𑀪 𑀲𑀤𑁆𑀥𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


(36.)

2628\. _𑀲𑀸 𑀢𑁂 𑀲𑀤𑁆𑀥𑀸 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀸𑀲𑁆𑀲,_  
_𑀫𑀽𑀮𑀚𑀸𑀢𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀲𑀁𑀳𑀻𑀭𑀸,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂𑁇_  


(37.)

2629\. _𑀓𑀸𑀬𑁂𑀦 𑀓𑀼𑀲𑀮𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀯𑀸𑀘𑀸𑀬 𑀓𑀼𑀲𑀮𑀁 𑀩𑀳𑀼𑀁;_  
_𑀫𑀦𑀲𑀸 𑀓𑀼𑀲𑀮𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀁 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁𑁇_  


(38.)

2630\. _𑀢𑀢𑁄 𑀑𑀧𑀥𑀺𑀓𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀓𑀢𑁆𑀯𑀸 𑀤𑀸𑀦𑁂𑀦 𑀢𑀁 𑀩𑀳𑀼𑀁;_  
_𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀫𑀘𑁆𑀘𑁂 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂 𑀦𑀺𑀯𑁂𑀲𑀬’𑁇_  


(39.)

2631\. _𑀇𑀫𑀸𑀬 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬,_  
_𑀤𑁂𑀯𑀸 𑀤𑁂𑀯𑀁 𑀬𑀤𑀸 𑀯𑀺𑀤𑀽;_  
_𑀘𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀏𑀳𑀺 𑀤𑁂𑀯 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁𑁇_  


(40.)

2632\. _𑀲𑀁𑀯𑁂𑀕𑁄 𑀫𑁂 𑀢𑀤𑀸 𑀆𑀲𑀺,_  
_𑀤𑁂𑀯𑀲𑀗𑁆𑀖𑁂 𑀲𑀫𑀸𑀕𑀢𑁂;_  
_𑀓𑀁 𑀲𑀼 𑀦𑀸𑀫 𑀅𑀳𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀇𑀢𑁄 𑀘𑀼𑀢𑁄𑁇_  


(41.)

2633\. _𑀫𑀫 𑀲𑀁𑀯𑁂𑀕𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀫𑀡𑁄 𑀪𑀸𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀫𑀫𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(42.)

2634\. _𑀲𑀼𑀫𑀦𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑀸𑀯𑀓𑁄;_  
_𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀯𑁂𑀚𑁂𑀲𑀺 𑀫𑀫𑀁 𑀢𑀤𑀸𑁇_  


(43.)

2635\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀢𑀁 𑀥𑀻𑀭𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(44.)

2636\. _𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀁 𑀲 𑀢𑀢𑁆𑀣𑁂𑀯,_  
_𑀲𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑁂𑀦 𑀘𑁄𑀤𑀺𑀢𑁄;_  
_𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(45.)

2637\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2638\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀘𑀺𑀦𑁆𑀢𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2639\. 𑀏𑀓𑀘𑀺𑀦𑁆𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
