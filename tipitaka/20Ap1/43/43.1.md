# 𑁪𑁩𑁇𑁧 𑀲𑀓𑀺𑀁𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀓𑀺𑀁𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀓𑀺𑀁𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

5470\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀧𑀸𑀝𑀮𑀺𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀯 𑀢𑀁 𑀧𑀸𑀤𑀧𑀕𑁆𑀕𑀁,_  
_𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁𑁇_  


(2.)

5471\. _𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑁄𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀩𑁄𑀥𑀺𑀁,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀁 𑀧𑀸𑀝𑀮𑀺𑀁 𑀅𑀳𑀁𑁇_  


(3.)

5472\. _𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀭𑁂 𑀓𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀦𑀫𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀢𑀁 𑀩𑁄𑀥𑀺𑀁,_  
_𑀕𑀜𑁆𑀙𑀺𑀁 𑀧𑀝𑀺𑀓𑀼𑀝𑀺𑀁 𑀅𑀳𑀁𑁇_  


(4.)

5473\. _𑀢𑀸𑀤𑀺𑀫𑀕𑁆𑀕𑁂𑀦 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀲𑀭𑀦𑁆𑀢𑁄 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀅𑀚𑀕𑀭𑁄 𑀫𑀁 𑀧𑀻𑀍𑀏𑀲𑀺,_  
_𑀖𑁄𑀭𑀭𑀽𑀧𑁄 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(5.)

5474\. _𑀆𑀲𑀦𑁆𑀦𑁂 𑀫𑁂 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀨𑀮𑁂𑀦 𑀢𑁄𑀲𑀬𑀻 𑀫𑀫𑀁;_  
_𑀓𑀍𑀏𑀯𑀭𑀁 𑀫𑁂 𑀕𑀺𑀮𑀢𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀸𑀫𑀳𑀁𑁇_  


(6.)

5475\. _𑀅𑀦𑀸𑀯𑀺𑀮𑀁 𑀫𑀫 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀲𑀤𑀸;_  
_𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑀦𑁆𑀢𑀸𑀧𑀦𑀁 𑀫𑀫𑁇_  


(7.)

5476\. _𑀓𑀼𑀝𑁆𑀞𑀁 𑀕𑀡𑁆𑀟𑁄 𑀓𑀺𑀮𑀸𑀲𑁄 𑀘,_  
_𑀅𑀧𑀫𑀸𑀭𑁄 𑀯𑀺𑀢𑀘𑁆𑀙𑀺𑀓𑀸;_  
_𑀤𑀤𑁆𑀤𑀼 𑀓𑀡𑁆𑀟𑀼 𑀘 𑀫𑁂 𑀦𑀢𑁆𑀣𑀺,_  
_𑀨𑀮𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀸𑀬𑀺𑀤𑀁𑁇_  


(8.)

5477\. _𑀲𑁄𑀓𑁄 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑁄 𑀘,_  
_𑀳𑀤𑀬𑁂 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀪𑀦𑁆𑀢𑀁 𑀉𑀚𑀼𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀨𑀮𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀸𑀬𑀺𑀤𑀁𑁇_  


(9.)

5478\. _𑀲𑀫𑀸𑀥𑀻𑀲𑀼 𑀦 𑀫𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀯𑀺𑀲𑀤𑀁 𑀳𑁄𑀢𑀺 𑀫𑀸𑀦𑀲𑀁;_  
_𑀬𑀁 𑀬𑀁 𑀲𑀫𑀸𑀥𑀺𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀲𑁄 𑀲𑁄 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀢𑁂 𑀫𑀫𑀁𑁇_  


(10.)

5479\. _𑀭𑀚𑀦𑀻𑀬𑁂 𑀦 𑀭𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀅𑀣𑁄 𑀤𑀼𑀲𑁆𑀲𑀦𑀺𑀬𑁂𑀲𑀼 𑀘;_  
_𑀫𑁄𑀳𑀦𑀻𑀬𑁂 𑀦 𑀫𑀼𑀬𑁆𑀳𑀸𑀫𑀺,_  
_𑀨𑀮𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀸𑀬𑀺𑀤𑀁𑁇_  


(11.)

5480\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀨𑀮𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀸𑀬𑀺𑀤𑀁𑁇_  


(12.)

5481\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(13.)

5482\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(14.)

5483\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5484\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀓𑀺𑀁𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5485\. 𑀲𑀓𑀺𑀁𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
