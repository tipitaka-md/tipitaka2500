# 𑁨𑁪𑁇𑁪 𑀓𑀺𑀮𑀜𑁆𑀚𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀺𑀮𑀜𑁆𑀚𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(14.)

2992\. _“𑀢𑀺𑀯𑀭𑀸𑀬𑀁 𑀧𑀼𑀭𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀦𑀍𑀅𑀓𑀸𑀭𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁂 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑁂,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀚𑀦𑀢𑀸 𑀢𑀳𑀺𑀁𑁇_  


(15.)

2993\. _𑀧𑀽𑀚𑀢𑁆𑀣𑀁 𑀮𑁄𑀓𑀦𑀸𑀣𑀲𑁆𑀲,_  
_𑀓𑀺𑀮𑀜𑁆𑀚𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺;_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀓𑀺𑀮𑀜𑁆𑀚𑀁 𑀅𑀤𑀤𑀺𑀁 𑀅𑀳𑀁𑁇_  


(16.)

2994\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑀺𑀮𑀜𑁆𑀚𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(17.)

2995\. _𑀲𑀢𑁆𑀢𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺𑀁 𑀚𑀮𑀤𑁆𑀥𑀭𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(18.)

2996\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2997\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀺𑀮𑀜𑁆𑀚𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2998\. 𑀓𑀺𑀮𑀜𑁆𑀚𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
