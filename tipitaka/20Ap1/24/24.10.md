# 𑁨𑁪𑁇𑁧𑁦 𑀯𑀸𑀲𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

### 𑀯𑀸𑀲𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(36.)

3026. _“𑀓𑀫𑁆𑀫𑀸𑀭𑁄𑀳𑀁 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀢𑀺𑀯𑀭𑀸𑀬𑀁 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀏𑀓𑀸 𑀯𑀸𑀲𑀺 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁𑁇_  


(37.)

3027. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀯𑀸𑀲𑀺𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑀸𑀲𑀺𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(38.)

3028. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁪𑁫𑁭)_  


3029. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑀸𑀲𑀺𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3030. 𑀯𑀸𑀲𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

3031. 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇



---

3032. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



3033. _𑀉𑀤𑀓𑀸𑀲𑀦𑀪𑀸𑀚𑀦𑀁,_  
_𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀻 𑀓𑀺𑀮𑀜𑁆𑀚𑀓𑁄;_  
_𑀯𑁂𑀤𑀺𑀓𑀸 𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑁄 𑀘,_  
_𑀧𑀺𑀬𑀸𑀮𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀤𑁄;_  
_𑀚𑀕𑀢𑀻 𑀯𑀸𑀲𑀺𑀤𑀸𑀢𑀸 𑀘,_  
_𑀕𑀸𑀣𑀸 𑀢𑀺𑀁𑀲 𑀘 𑀅𑀝𑁆𑀞 𑀘𑁇_
