# 𑁨𑁪𑁇𑁯 𑀚𑀕𑀢𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

### 𑀚𑀕𑀢𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(33.)

3021. _“𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀚𑀕𑀢𑀻 𑀓𑀸𑀭𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀣𑀽𑀧𑀫𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(34.)

3022. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀚𑀕𑀢𑀺𑀬𑀸 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(35.)

3023. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3024. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀚𑀕𑀢𑀺𑀓𑀸𑀭𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3025. 𑀚𑀕𑀢𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
