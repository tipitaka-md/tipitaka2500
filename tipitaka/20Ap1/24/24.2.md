# 𑁨𑁪𑁇𑁨 𑀪𑀸𑀚𑀦𑀧𑀸𑀮𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

### 𑀪𑀸𑀚𑀦𑀧𑀸𑀮𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(5.)

2979\. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀪𑀸𑀚𑀦𑀁 𑀅𑀦𑀼𑀧𑀸𑀮𑁂𑀲𑀺𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀢𑀸𑀯𑀤𑁂𑁇_  


(6.)

2980\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀪𑀸𑀚𑀦𑀁 𑀅𑀦𑀼𑀧𑀸𑀮𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑀸𑀚𑀦𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(7.)

2981\. _𑀢𑁂𑀧𑀜𑁆𑀜𑀸𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀦𑀦𑁆𑀢𑀚𑀸𑀮𑀺𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(8.)

2982\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2983\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀸𑀚𑀦𑀧𑀸𑀮𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2984\. 𑀪𑀸𑀚𑀦𑀧𑀸𑀮𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
