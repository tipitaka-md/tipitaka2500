# 𑁨𑁪𑁇𑁫 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

* 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(19.)

2999\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀩𑁄𑀥𑀺𑀬𑀸 𑀧𑀸𑀤𑀧𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀓𑀸𑀭𑁂𑀲𑀺𑀁 𑀯𑁂𑀤𑀺𑀓𑀁 𑀅𑀳𑀁𑁇_  


(20.)

3000\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀓𑀸𑀭𑁂𑀲𑀺𑀁 𑀯𑁂𑀤𑀺𑀓𑀁 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(21.)

3001\. _𑀇𑀢𑁄 𑀏𑀓𑀸𑀤𑀲𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲𑀫𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(22.)

3002\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3003\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3004\. 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
