# 𑁨𑁪𑁇𑁩 𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(9.)

2985\. _“𑀅𑀭𑀼𑀡𑀯𑀢𑀺𑀬𑀸 𑀦𑀕𑀭𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀧𑀽𑀧𑀺𑀓𑁄 𑀢𑀤𑀸;_  
_𑀫𑀫 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁,_  
_𑀲𑀺𑀔𑀺𑀦𑀁 𑀅𑀤𑁆𑀤𑀲𑀁 𑀚𑀺𑀦𑀁𑁇_  


(10.)

2986\. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀁 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀲𑀫𑁆𑀫𑀕𑁆𑀕𑀢𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(11.)

2987\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀤𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(12.)

2988\. _𑀇𑀢𑁄 𑀘𑀼𑀤𑁆𑀤𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀅𑀫𑀺𑀢𑀜𑁆𑀚𑀮𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(13.)

2989\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2990\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2991\. 𑀲𑀸𑀮𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
