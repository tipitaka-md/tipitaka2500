# 𑁨𑁪𑁇𑁬 𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

* 𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(23.)

3005\. _“𑀦𑀕𑀭𑁂 𑀅𑀭𑀼𑀡𑀯𑀢𑀺𑀬𑀸,_  
_𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀘𑁂𑀢𑀺𑀬𑁂 𑀤𑀼𑀲𑁆𑀲𑀪𑀡𑁆𑀟𑀸𑀦𑀺,_  
_𑀦𑀸𑀦𑀸𑀯𑀡𑁆𑀡𑀁 𑀭𑀚𑁂𑀲𑀳𑀁𑁇_  


(24.)

3006\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀯𑀡𑁆𑀡𑀁 𑀭𑀚𑀬𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑀡𑁆𑀡𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(25.)

3007\. _𑀇𑀢𑁄 𑀢𑁂𑀯𑀻𑀲𑀢𑀺𑀓𑀧𑁆𑀧𑁂,_  
_𑀯𑀡𑁆𑀡𑀲𑀫𑀲𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(26.)

3008\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3009\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3010\. 𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
