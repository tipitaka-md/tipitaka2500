# 𑁨𑁪𑁇𑁮 𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

* 𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(30.)

3016\. _“𑀲𑀓𑁂 𑀲𑀺𑀧𑁆𑀧𑁂 𑀅𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑁄,_  
_𑀅𑀕𑀫𑀁 𑀓𑀸𑀦𑀦𑀁 𑀅𑀳𑀁;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀬𑀦𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀁 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(31.)

3017\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(32.)

3018\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3019\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3020\. 𑀅𑀫𑁆𑀩𑀬𑀸𑀕𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
