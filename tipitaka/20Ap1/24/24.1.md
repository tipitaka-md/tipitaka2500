# 𑁨𑁪𑁇𑁧 𑀉𑀤𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀉𑀤𑀓𑀸𑀲𑀦𑀯𑀕𑁆𑀕

* 𑀉𑀤𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

2973\. _“𑀆𑀭𑀸𑀫𑀤𑁆𑀯𑀸𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀨𑀮𑀓𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀁 𑀅𑀳𑀁;_  
_𑀉𑀤𑀓𑀜𑁆𑀘 𑀉𑀧𑀝𑁆𑀞𑀸𑀲𑀺𑀁,_  
_𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(2.)

2974\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀆𑀲𑀦𑁂 𑀘𑁄𑀤𑀓𑁂 𑀨𑀮𑀁𑁇_  


(3.)

2975\. _𑀇𑀢𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀅𑀪𑀺𑀲𑀸𑀫𑀲𑀫𑀯𑁆𑀳𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(4.)

2976\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2977\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2978\. 𑀉𑀤𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
