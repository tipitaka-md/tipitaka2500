# 𑁧𑁦𑁇𑁫 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀥𑀸𑀯𑀕𑁆𑀕

* 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(24.)

1796\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀕𑀦𑁆𑀥𑀣𑀽𑀧𑀁 𑀅𑀓𑀸𑀲𑀳𑀁;_  
_𑀲𑀼𑀫𑀦𑁂𑀳𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁 𑀓𑀢𑀁𑁇_  


(25.)

1797\. _𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑀁;_  
_𑀇𑀦𑁆𑀤𑀻𑀯𑀭𑀁𑀯 𑀚𑀮𑀺𑀢𑀁,_  
_𑀆𑀤𑀺𑀢𑁆𑀢𑀁𑀯 𑀳𑀼𑀢𑀸𑀲𑀦𑀁𑁇_  


(26.)

1798\. _𑀩𑁆𑀬𑀕𑁆𑀖𑀽𑀲𑀪𑀁𑀯 𑀧𑀯𑀭𑀁,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀁𑀯 𑀓𑁂𑀲𑀭𑀺𑀁;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀲𑀫𑀡𑀸𑀦𑀕𑁆𑀕𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁𑁇_  


(27.)

1799\. _𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄;_  
_𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀁 𑀬𑀢𑁄 𑀅𑀤𑀁𑁇_  


(28.)

1800\. _𑀩𑀼𑀤𑁆𑀥𑁂 𑀓𑀢𑀲𑁆𑀲 𑀓𑀸𑀭𑀲𑁆𑀲,_  
_𑀨𑀮𑁂𑀦𑀸𑀳𑀁 𑀯𑀺𑀲𑁂𑀲𑀢𑁄;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(29.)

1801\. _𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀫𑁆𑀳𑀺 𑀏𑀓𑀽𑀦𑁂,_  
_𑀓𑀧𑁆𑀧𑁂 𑀆𑀲𑀺𑀁𑀲𑀼 𑀲𑁄𑀍𑀅𑀲;_  
_𑀤𑁂𑀯𑀕𑀦𑁆𑀥𑀲𑀦𑀸𑀫𑀸 𑀢𑁂,_  
_𑀭𑀸𑀚𑀸𑀦𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(30.)

1802\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1803\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1804\. 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
