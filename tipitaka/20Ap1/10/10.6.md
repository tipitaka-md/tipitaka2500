# 𑁧𑁦𑁇𑁬 𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀥𑀸𑀯𑀕𑁆𑀕

* 𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(31.)

1805\. _“𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀸𑀦𑀦𑁂 𑀅𑀳𑀁;_  
_𑀧𑀸𑀝𑀮𑀺𑀁 𑀳𑀭𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀢𑀻𑀡𑀺 𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺 𑀑𑀓𑀺𑀭𑀺𑀁𑁇_  


(32.)

1806\. _𑀧𑀢𑀺𑀢𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀕𑀡𑁆𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑀳𑀺 𑀙𑀟𑁆𑀟𑁂𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀅𑀦𑁆𑀢𑁄𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑀳𑀺𑀲𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑀸𑀲𑀯𑀁𑁇_  


(33.)

1807\. _𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀯𑀺𑀬 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀧𑀸𑀝𑀮𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(34.)

1808\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀩𑁄𑀥𑀺𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑁄𑀥𑀺𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(35.)

1809\. _𑀲𑀫𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸 𑀦𑀸𑀫,_  
_𑀢𑁂𑀭𑀲𑀸𑀲𑀺𑀁𑀲𑀼 𑀭𑀸𑀚𑀺𑀦𑁄;_  
_𑀇𑀢𑁄 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(36.)

1810\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1811\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1812\. 𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
