# 𑁧𑁦𑁇𑁧𑁦 𑀩𑀼𑀤𑁆𑀥𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀥𑀸𑀯𑀕𑁆𑀕

* 𑀩𑀼𑀤𑁆𑀥𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(56.)

1838\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀲𑀗𑁆𑀔𑀥𑀫𑁆𑀫𑀓𑁄;_  
_𑀦𑀺𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀬𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


(57.)

1839\. _𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀨𑀮𑀁 𑀧𑀲𑁆𑀲,_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀲𑀝𑁆𑀞𑀺𑀢𑀽𑀭𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸𑁇_  


(58.)

1840\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁 𑀫𑀳𑀸𑀇𑀲𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(59.)

1841\. _𑀘𑀢𑀼𑀯𑀻𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀫𑀳𑀸𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀲𑁄𑀍𑀅𑀲𑀸𑀲𑀺𑀁𑀲𑀼 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(60.)

1842\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁧𑁬𑁦𑁨)_  


1843\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀼𑀤𑁆𑀥𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1844\. 𑀩𑀼𑀤𑁆𑀥𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇

  
**𑀲𑀼𑀥𑀸𑀯𑀕𑁆𑀕𑁄 𑀤𑀲𑀫𑁄𑁇**



---

1845\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1846\. _𑀲𑀼𑀥𑀸 𑀲𑀼𑀘𑀺𑀦𑁆𑀢𑀺 𑀘𑁂𑀍𑀅𑀜𑁆𑀘,_  
_𑀲𑀽𑀘𑀻 𑀘 𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀺𑀬𑁄;_  
_𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄 𑀫𑀥𑀼𑀲𑁂𑀦𑀸,_  
_𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑁄 𑀘𑀼𑀧𑀝𑁆𑀞𑀓𑁄;_  
_𑀲𑀫𑀲𑀝𑁆𑀞𑀺 𑀘 𑀕𑀸𑀣𑀸𑀬𑁄,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀯𑀕𑁆𑀕𑁂 𑀧𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇_  


---

1847\. 𑀅𑀣 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1848\. _𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕𑁄 𑀳𑀺 𑀧𑀞𑀫𑁄,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀺 𑀲𑀼𑀪𑀽𑀢𑀺 𑀘;_  
_𑀓𑀼𑀡𑁆𑀟𑀥𑀸𑀦𑁄 𑀉𑀧𑀸𑀮𑀺 𑀘,_  
_𑀩𑀻𑀚𑀦𑀺𑀲𑀓𑀘𑀺𑀦𑁆𑀢𑀺 𑀘𑁇_  


1849\. _𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀢𑀺𑀫𑀺𑀭𑁄,_  
_𑀲𑀼𑀥𑀸𑀯𑀕𑁆𑀕𑁂𑀦 𑀢𑁂 𑀤𑀲;_  
_𑀘𑀢𑀼𑀤𑁆𑀤𑀲𑀲𑀢𑀸 𑀕𑀸𑀣𑀸,_  
_𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁂𑀯 𑀘𑁇_  


---

1850\. 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕𑀤𑀲𑀓𑀁𑁇



---

1851\. 𑀧𑀞𑀫𑀲𑀢𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
