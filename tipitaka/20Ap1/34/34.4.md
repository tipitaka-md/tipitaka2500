# 𑁩𑁪𑁇𑁪 𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(19.)

3739. _“𑀦𑀕𑀭𑁂 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀸,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀢𑀺𑀡𑀳𑀸𑀭𑀓𑁄;_  
_𑀢𑀺𑀡𑀳𑀸𑀭𑁂𑀦 𑀚𑀻𑀯𑀸𑀫𑀺,_  
_𑀢𑁂𑀦 𑀧𑁄𑀲𑁂𑀫𑀺 𑀤𑀸𑀭𑀓𑁂𑁇_  


(20.)

3740. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀢𑀫𑀦𑁆𑀥𑀓𑀸𑀭𑀁 𑀦𑀸𑀲𑁂𑀢𑁆𑀯𑀸,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(21.)

3741. _𑀲𑀓𑁂 𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_‘𑀩𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀓𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄 𑀘 𑀦𑀢𑁆𑀣𑀺 𑀫𑁂𑁇_  


(22.)

3742. _𑀇𑀤𑀁 𑀫𑁂 𑀲𑀸𑀝𑀓𑀁 𑀏𑀓𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀓𑁄𑀘𑀺 𑀤𑀸𑀬𑀓𑁄;_  
_𑀤𑀼𑀓𑁆𑀔𑁄 𑀦𑀺𑀭𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄,_  
_𑀭𑁄𑀧𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁’𑁇_  


(23.)

3743. _𑀏𑀯𑀸𑀳𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁;_  
_𑀏𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(24.)

3744. _𑀏𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀁 𑀤𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑀺𑀁 𑀲𑀫𑁆𑀧𑀯𑀢𑁆𑀢𑀬𑀺𑀁;_  
_𑀬𑀤𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑀼𑀯𑀁 𑀯𑀻𑀭,_  
_𑀢𑀸𑀭𑁂𑀳𑀺 𑀫𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(25.)

3745. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽,_  
_𑀆𑀳𑀼𑀢𑀻𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀫𑀫 𑀤𑀸𑀦𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀅𑀓𑀸 𑀫𑁂 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑀁𑁇_  


(26.)

3746. _‘𑀇𑀫𑀺𑀦𑀸 𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦 𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(27.)

3747. _𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(28.)

3748. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸,_  
_𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄 𑀢𑀼𑀯𑀁 𑀪𑀯𑁂;_  
_𑀭𑀽𑀧𑀯𑀸 𑀕𑀼𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀅𑀦𑀯𑀓𑁆𑀓𑀦𑁆𑀢𑀤𑁂𑀳𑀯𑀸;_  
_𑀅𑀓𑁆𑀔𑁄𑀪𑀁 𑀅𑀫𑀺𑀢𑀁 𑀤𑀼𑀲𑁆𑀲𑀁,_  
_𑀮𑀪𑀺𑀲𑁆𑀲𑀢𑀺 𑀬𑀤𑀺𑀘𑁆𑀙𑀓𑀁’𑁇_  


(29.)

3749. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀓𑁄;_  
_𑀦𑀪𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀫𑀻 𑀥𑀻𑀭𑁄,_  
_𑀳𑀁𑀲𑀭𑀸𑀚𑀸𑀯 𑀅𑀫𑁆𑀩𑀭𑁂𑁇_  


(30.)

3750. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀪𑁄𑀕𑁂 𑀫𑁂 𑀊𑀦𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(31.)

3751. _𑀧𑀤𑀼𑀤𑁆𑀥𑀸𑀭𑁂 𑀧𑀤𑀼𑀤𑁆𑀥𑀸𑀭𑁂,_  
_𑀤𑀼𑀲𑁆𑀲𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑁂 𑀫𑀫𑀁;_  
_𑀳𑁂𑀝𑁆𑀞𑀸 𑀤𑀼𑀲𑁆𑀲𑀫𑁆𑀳𑀺 𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺,_  
_𑀉𑀧𑀭𑀺 𑀙𑀤𑀦𑀁 𑀫𑀫𑁇_  


(32.)

3752. _𑀘𑀓𑁆𑀓𑀯𑀸𑀍𑀅𑀫𑀼𑀧𑀸𑀤𑀸𑀬,_  
_𑀲𑀓𑀸𑀦𑀦𑀁 𑀲𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀇𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀘𑀳𑀁 𑀅𑀚𑁆𑀚,_  
_𑀤𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀙𑀸𑀤𑀬𑁂𑀬𑁆𑀬𑀳𑀁𑁇_  


(33.)

3753. _𑀢𑁂𑀦𑁂𑀯 𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑁂𑀦,_  
_𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑁄 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀁𑀲𑀭𑀸𑀫𑀺 𑀪𑀯𑀸𑀪𑀯𑁂𑁇_  


(34.)

3754. _𑀯𑀺𑀧𑀸𑀓𑀁 𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑀲𑁆𑀲,_  
_𑀦𑀚𑁆𑀛𑀕𑀁 𑀓𑀢𑁆𑀣𑀘𑀺𑀓𑁆𑀔𑀬𑀁;_  
_𑀅𑀬𑀁 𑀫𑁂 𑀅𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀸𑀢𑀺,_  
_𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀺 𑀇𑀥𑀸𑀧𑀺 𑀫𑁂𑁇_  


(35.)

3755. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀼𑀲𑁆𑀲𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(36.)

3756. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑁄𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(37.)

3757. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3758. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3759. 𑀏𑀓𑀤𑀼𑀲𑁆𑀲𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
