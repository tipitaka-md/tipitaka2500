# 𑁩𑁪𑁇𑁧 𑀕𑀦𑁆𑀥𑀥𑀽𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑀯𑀕𑁆𑀕

* 𑀕𑀦𑁆𑀥𑀥𑀽𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3715\. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀕𑀦𑁆𑀥𑀥𑀽𑀧𑀁 𑀅𑀤𑀸𑀲𑀳𑀁;_  
_𑀲𑀼𑀫𑀦𑁂𑀳𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀜𑁆𑀘 𑀢𑀁𑁇_  


(2.)

3716\. _𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑀁;_  
_𑀇𑀦𑁆𑀤𑀻𑀯𑀭𑀁𑀯 𑀚𑀮𑀺𑀢𑀁,_  
_𑀆𑀤𑀺𑀢𑁆𑀢𑀁𑀯 𑀳𑀼𑀢𑀸𑀲𑀦𑀁𑁇_  


(3.)

3717\. _𑀩𑁆𑀬𑀕𑁆𑀖𑀼𑀲𑀪𑀁𑀯 𑀧𑀯𑀭𑀁,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀁𑀯 𑀓𑁂𑀲𑀭𑀺𑀁;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀲𑀫𑀡𑀸𑀦𑀕𑁆𑀕𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁𑁇_  


(4.)

3718\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀼𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑁄𑁇_  


(5.)

3719\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀕𑀦𑁆𑀥𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀕𑀦𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(6.)

3720\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3721\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀕𑀦𑁆𑀥𑀥𑀽𑀧𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3722\. 𑀕𑀦𑁆𑀥𑀥𑀽𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
