# 𑁩𑁪𑁇𑁨 𑀉𑀤𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀤𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(7.)

3723\. _“𑀲𑀼𑀯𑀡𑁆𑀡𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀅𑀦𑀺𑀮𑀜𑁆𑀚𑀲𑁂;_  
_𑀖𑀢𑀸𑀲𑀦𑀁𑀯 𑀚𑀮𑀺𑀢𑀁,_  
_𑀆𑀤𑀺𑀢𑁆𑀢𑀁𑀯 𑀳𑀼𑀢𑀸𑀲𑀦𑀁𑁇_  


(8.)

3724\. _𑀧𑀸𑀡𑀺𑀦𑀸 𑀉𑀤𑀓𑀁 𑀕𑀬𑁆𑀳,_  
_𑀆𑀓𑀸𑀲𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀁 𑀅𑀳𑀁;_  
_𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀇𑀲𑀺𑁇_  


(9.)

3725\. _𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀦𑀸𑀫𑀓𑁄;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀇𑀫𑀁 𑀕𑀸𑀣𑀫𑀪𑀸𑀲𑀣𑁇_  


(10.)

3726\. _‘𑀇𑀫𑀺𑀦𑀸 𑀤𑀓𑀤𑀸𑀦𑁂𑀦,_  
_𑀧𑀻𑀢𑀺𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑁂𑀦 𑀘;_  
_𑀓𑀧𑁆𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺,_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀢𑀺’𑁇_  


(11.)

3727\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪;_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁𑁇_  


(12.)

3728\. _𑀲𑀳𑀲𑁆𑀲𑀭𑀸𑀚𑀦𑀸𑀫𑁂𑀦,_  
_𑀢𑀬𑁄 𑀘 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀧𑀜𑁆𑀘𑀲𑀝𑁆𑀞𑀺𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑀸 𑀚𑀦𑀸𑀥𑀺𑀧𑀸𑁇_  


(13.)

3729\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3730\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀓𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3731\. 𑀉𑀤𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
