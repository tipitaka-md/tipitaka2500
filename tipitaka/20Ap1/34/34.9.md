# 𑁩𑁪𑁇𑁯 𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀕𑀦𑁆𑀥𑁄𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(111.)

3841. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀁 𑀧𑀓𑀸𑀲𑁂𑀦𑁆𑀢𑁄,_  
_𑀲𑀦𑁆𑀢𑀸𑀭𑁂𑀲𑀺 𑀩𑀳𑀼𑀁 𑀚𑀦𑀁𑁇_  


(112.)

3842. _𑀅𑀳𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦,_  
_𑀚𑀝𑀺𑀮𑁄 𑀉𑀕𑁆𑀕𑀢𑀸𑀧𑀦𑁄;_  
_𑀥𑀼𑀦𑀦𑁆𑀢𑁄 𑀯𑀸𑀓𑀘𑀻𑀭𑀸𑀦𑀺,_  
_𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀫𑁆𑀩𑀭𑁂 𑀢𑀤𑀸𑁇_  


(113.)

3843. _𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀉𑀧𑀭𑀺,_  
_𑀕𑀦𑁆𑀢𑀼𑀁 𑀦 𑀯𑀺𑀲𑀳𑀸𑀫𑀳𑀁;_  
_𑀧𑀓𑁆𑀔𑀻𑀯 𑀲𑁂𑀮𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀕𑀫𑀦𑀁 𑀦 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(114.)

3844. _𑀦 𑀫𑁂 𑀇𑀤𑀁 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁,_  
_𑀇𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀓𑁄𑀧𑀦𑀁;_  
_𑀤𑀓𑁂 𑀬𑀣𑀸 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸,_  
_𑀏𑀯𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀫𑁆𑀩𑀭𑁂𑁇_  


(115.)

3845. _𑀉𑀍𑀆𑀭𑀪𑀽𑀢𑁄 𑀫𑀦𑀼𑀚𑁄,_  
_𑀳𑁂𑀝𑁆𑀞𑀸𑀲𑀻𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀳𑀦𑁆𑀤 𑀫𑁂𑀦𑀁 𑀕𑀯𑁂𑀲𑀺𑀲𑁆𑀲𑀁,_  
_𑀅𑀧𑀺 𑀅𑀢𑁆𑀣𑀁 𑀮𑀪𑁂𑀬𑁆𑀬𑀳𑀁𑁇_  


(116.)

3846. _𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸,_  
_𑀲𑀤𑁆𑀤𑀫𑀲𑁆𑀲𑁄𑀲𑀺 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀢𑀫𑀳𑀁 𑀉𑀕𑁆𑀕𑀳𑀺𑀁 𑀢𑀤𑀸𑁇_  


(117.)

3847. _𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀫𑀼𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀅𑀕𑀫𑀸𑀲𑀺𑀁 𑀫𑀫𑀲𑁆𑀲𑀫𑀁;_  
_𑀬𑀸𑀯𑀢𑀸𑀬𑀼𑀁 𑀯𑀲𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(118.)

3848. _𑀘𑀭𑀺𑀫𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀢𑀁 𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀁 𑀲𑀭𑀺𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(119.)

3849. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀁 𑀅𑀳𑀁;_  
_𑀏𑀓𑀧𑀜𑁆𑀜𑀸𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(120.)

3850. _𑀏𑀓𑀲𑀢𑁆𑀢𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀜𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀧𑀤𑁂𑀲𑀭𑀚𑁆𑀚𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀕𑀡𑀦𑀸𑀢𑁄 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(121.)

3851. _𑀧𑀺𑀢𑀼𑀕𑁂𑀳𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀫𑀡𑁄 𑀪𑀸𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀕𑀸𑀣𑀸𑀬 𑀧𑀭𑀺𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄,_  
_𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀫𑀼𑀤𑀸𑀳𑀭𑀺𑁇_  


(122.)

3852. _𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺 𑀢𑀁 𑀲𑀜𑁆𑀜𑀁,_  
_𑀲𑀁𑀲𑀭𑀦𑁆𑀢𑁄 𑀪𑀯𑀸𑀪𑀯𑁂;_  
_𑀦 𑀓𑁄𑀝𑀺𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸𑀫𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀼𑀢𑀁 𑀧𑀤𑀁𑁇_  


(123.)

3853. _𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀯𑀢 𑀲𑀗𑁆𑀔𑀸𑀭𑀸,_  
_𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀬𑀥𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀲𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀲𑀼𑀔𑁄𑁇_  


(124.)

3854. _𑀲𑀳 𑀕𑀸𑀣𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀏𑀓𑀸𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(125.)

3855. _𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑁄𑀳𑀁,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁;_  
_𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑀬𑀺 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀼𑀡𑀫𑀜𑁆𑀜𑀸𑀬 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(126.)

3856. _𑀤𑀸𑀭𑀓𑁄𑀯 𑀅𑀳𑀁 𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀭𑀡𑀻𑀬𑀁 𑀲𑀫𑀸𑀧𑀬𑀺𑀁;_  
_𑀓𑀺𑀁 𑀫𑁂 𑀓𑀭𑀡𑀻𑀬𑀁 𑀅𑀚𑁆𑀚,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


(127.)

3857. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑁂 𑀨𑀮𑀁𑁇_  


(128.)

3858. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3859. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3860. 𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
