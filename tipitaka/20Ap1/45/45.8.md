# 𑁪𑁫𑁇𑁮 𑀧𑀸𑀤𑀧𑀻𑀞𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀯𑀺𑀪𑀻𑀢𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀸𑀤𑀧𑀻𑀞𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(50.)

5838\. _“𑀲𑀼𑀫𑁂𑀥𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀕𑁆𑀕𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀫𑀼𑀦𑀺;_  
_𑀢𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀲𑀢𑁆𑀢𑁂,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑁄 𑀫𑀳𑀸𑀬𑀲𑁄𑁇_  


(51.)

5839\. _𑀲𑀻𑀳𑀸𑀲𑀦𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸,_  
_𑀲𑀼𑀫𑁂𑀥𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀸𑀤𑀧𑀻𑀞𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(52.)

5840\. _𑀓𑀢𑁆𑀯𑀸𑀦 𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀼𑀔𑀧𑀸𑀓𑀁 𑀲𑀼𑀔𑀼𑀤𑁆𑀭𑀬𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(53.)

5841\. _𑀢𑀢𑁆𑀣 𑀫𑁂 𑀯𑀲𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄;_  
_𑀧𑀤𑀸𑀦𑀺 𑀉𑀤𑁆𑀥𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀲𑁄𑀡𑁆𑀡𑀧𑀻𑀞𑀸 𑀪𑀯𑀦𑁆𑀢𑀺 𑀫𑁂𑁇_  


(54.)

5842\. _𑀮𑀸𑀪𑀸 𑀢𑁂𑀲𑀁 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑁄,_  
_𑀬𑁂 𑀮𑀪𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁;_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀮𑀪𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀼𑀔𑀁𑁇_  


(55.)

5843\. _𑀫𑀬𑀸𑀧𑀺 𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑁆𑀚𑀁 𑀲𑀼𑀧𑁆𑀧𑀬𑁄𑀚𑀺𑀢𑀁;_  
_𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑁄𑀡𑁆𑀡𑀧𑀻𑀞𑀁 𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(56.)

5844\. _𑀬𑀁 𑀬𑀁 𑀤𑀺𑀲𑀁 𑀧𑀓𑁆𑀓𑀫𑀸𑀫𑀺,_  
_𑀓𑁂𑀦𑀘𑀺 𑀓𑀺𑀘𑁆𑀘𑀬𑁂𑀦𑀳𑀁;_  
_𑀲𑁄𑀡𑁆𑀡𑀧𑀻𑀞𑁂 𑀅𑀓𑁆𑀓𑀫𑀸𑀫𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(57.)

5845\. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀸𑀤𑀧𑀻𑀞𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(58.)

5846\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(59.)

5847\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(60.)

5848\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5849\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀸𑀤𑀧𑀻𑀞𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5850\. 𑀧𑀸𑀤𑀧𑀻𑀞𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
