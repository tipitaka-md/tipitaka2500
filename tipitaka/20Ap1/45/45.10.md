# 𑁪𑁫𑁇𑁧𑁦 𑀩𑁄𑀥𑀺𑀖𑀭𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀯𑀺𑀪𑀻𑀢𑀓𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀥𑀺𑀖𑀭𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(69.)

5861. _“𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀩𑁄𑀥𑀺𑀖𑀭𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(70.)

5862. _𑀢𑀼𑀲𑀺𑀢𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀺,_  
_𑀯𑀲𑀸𑀫𑀺 𑀭𑀢𑀦𑁂 𑀖𑀭𑁂;_  
_𑀦 𑀫𑁂 𑀲𑀻𑀢𑀁 𑀯𑀸 𑀉𑀡𑁆𑀳𑀁 𑀯𑀸,_  
_𑀯𑀸𑀢𑁄 𑀕𑀢𑁆𑀢𑁂 𑀦 𑀲𑀫𑁆𑀨𑀼𑀲𑁂𑁇_  


(71.)

5863. _𑀧𑀜𑁆𑀘𑀲𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑁄𑀲𑀳𑀁;_  
_𑀓𑀸𑀲𑀺𑀓𑀁 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁂𑀦 𑀫𑀸𑀧𑀺𑀢𑀁𑁇_  


(72.)

5864. _𑀤𑀲𑀬𑁄𑀚𑀦𑀆𑀬𑀸𑀫𑀁,_  
_𑀅𑀝𑁆𑀞𑀬𑁄𑀚𑀦𑀯𑀺𑀢𑁆𑀣𑀢𑀁;_  
_𑀦 𑀢𑀫𑁆𑀳𑀺 𑀦𑀕𑀭𑁂 𑀅𑀢𑁆𑀣𑀺,_  
_𑀓𑀝𑁆𑀞𑀁 𑀯𑀮𑁆𑀮𑀻 𑀘 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑁇_  


(73.)

5865. _𑀢𑀺𑀭𑀺𑀬𑀁 𑀬𑁄𑀚𑀦𑀁 𑀆𑀲𑀺,_  
_𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑀯𑀺𑀢𑁆𑀣𑀢𑀁;_  
_𑀫𑀗𑁆𑀕𑀮𑁄 𑀦𑀸𑀫 𑀧𑀸𑀲𑀸𑀤𑁄,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁂𑀦 𑀫𑀸𑀧𑀺𑀢𑁄𑁇_  


(74.)

5866. _𑀘𑀼𑀮𑁆𑀮𑀸𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀣𑀫𑁆𑀪𑀸 𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀸 𑀅𑀳𑀼𑀁;_  
_𑀫𑀡𑀺𑀫𑀬𑀸 𑀘 𑀦𑀺𑀬𑁆𑀬𑀽𑀳𑀸,_  
_𑀙𑀤𑀦𑀁 𑀭𑀽𑀧𑀺𑀬𑀁 𑀅𑀳𑀼𑁇_  


(75.)

5867. _𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀁 𑀖𑀭𑀁,_  
_𑀯𑀺𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑁂𑀦 𑀫𑀸𑀧𑀺𑀢𑀁;_  
_𑀅𑀚𑁆𑀛𑀸𑀯𑀼𑀢𑁆𑀣𑀁 𑀫𑀬𑀸 𑀏𑀢𑀁,_  
_𑀖𑀭𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(76.)

5868. _𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑀓𑁂 𑀪𑀯𑁂;_  
_𑀅𑀚𑁆𑀛𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀺𑀧𑀤𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


(77.)

5869. _𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀩𑁄𑀥𑀺𑀖𑀭𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀖𑀭𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(78.)

5870. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(79.)

5871. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(80.)

5872. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁪𑁭𑁦𑁪)_  


5873. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑁄𑀥𑀺𑀖𑀭𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5874. 𑀩𑁄𑀥𑀺𑀖𑀭𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

5875. 𑀯𑀺𑀪𑀻𑀢𑀓𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁄𑁇



---

5876. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



5877. _𑀯𑀺𑀪𑀻𑀢𑀓𑀻 𑀓𑁄𑀮𑀨𑀮𑀻,_  
_𑀩𑀺𑀮𑁆𑀮𑀪𑀮𑁆𑀮𑀸𑀢𑀓𑀧𑁆𑀧𑀤𑁄;_  
_𑀉𑀢𑁆𑀢𑀮𑀫𑁆𑀩𑀝𑀓𑀻 𑀘𑁂𑀯,_  
_𑀆𑀲𑀦𑀻 𑀧𑀸𑀤𑀧𑀻𑀞𑀓𑁄𑁇_  


5878. _𑀯𑁂𑀤𑀺𑀓𑁄 𑀩𑁄𑀥𑀺𑀖𑀭𑀺𑀓𑁄,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀡𑀺𑀢𑀸𑀧𑀺 𑀘;_  
_𑀏𑀓𑀽𑀦𑀸𑀲𑀻𑀢𑀺𑀓𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀯𑀕𑁆𑀕𑁂 𑀧𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇_
