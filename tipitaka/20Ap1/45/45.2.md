# 𑁪𑁫𑁇𑁨 𑀓𑁄𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀯𑀺𑀪𑀻𑀢𑀓𑀯𑀕𑁆𑀕

* 𑀓𑁄𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(8.)

5784\. _“𑀅𑀚𑀺𑀦𑁂𑀦 𑀦𑀺𑀯𑀢𑁆𑀣𑁄𑀳𑀁,_  
_𑀯𑀸𑀓𑀘𑀻𑀭𑀥𑀭𑁄 𑀢𑀤𑀸;_  
_𑀔𑀸𑀭𑀺𑀬𑀸 𑀧𑀽𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑁄𑀮𑀁𑀳𑀸𑀲𑀺𑀁 𑀫𑀫𑀲𑁆𑀲𑀫𑀁𑁇_  


(9.)

5785\. _𑀢𑀫𑁆𑀳𑀺 𑀓𑀸𑀮𑁂 𑀲𑀺𑀔𑀻 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀏𑀓𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀳𑀼;_  
_𑀫𑀫𑀲𑁆𑀲𑀫𑀁 𑀉𑀧𑀸𑀕𑀘𑁆𑀙𑀺,_  
_𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀺𑀓𑀁𑁇_  


(10.)

5786\. _𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀘 𑀲𑀼𑀩𑁆𑀩𑀢𑀁;_  
_𑀉𑀪𑁄 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀓𑁄𑀮𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀤𑀸𑀲𑀳𑀁𑁇_  


(11.)

5787\. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀨𑀮𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑁄𑀮𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(12.)

5788\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(13.)

5789\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(14.)

5790\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5791\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑁄𑀮𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5792\. 𑀓𑁄𑀮𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
