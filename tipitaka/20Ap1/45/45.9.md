# 𑁪𑁫𑁇𑁯 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀯𑀺𑀪𑀻𑀢𑀓𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(61.)

5851\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀩𑁄𑀥𑀺𑀬𑀸 𑀧𑀸𑀤𑀧𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀯𑁂𑀤𑀺𑀓𑀁 𑀲𑀼𑀓𑀢𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀲𑀓𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀺𑀁𑁇_  


(62.)

5852\. _𑀅𑀢𑁄𑀍𑀆𑀭𑀸𑀦𑀺 𑀪𑀡𑁆𑀟𑀸𑀦𑀺,_  
_𑀓𑀢𑀸𑀦𑀺 𑀅𑀓𑀢𑀸𑀦𑀺 𑀘;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀧𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(63.)

5853\. _𑀉𑀪𑀢𑁄 𑀩𑁆𑀬𑀽𑀍𑀳𑀲𑀗𑁆𑀕𑀸𑀫𑁂,_  
_𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀪𑀬𑀸𑀦𑀓𑁂;_  
_𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(64.)

5854\. _𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀩𑁆𑀬𑀫𑁆𑀳𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑁂 𑀲𑀼𑀪𑀁;_  
_𑀲𑀬𑀦𑀸𑀦𑀺 𑀫𑀳𑀕𑁆𑀖𑀸𑀦𑀺,_  
_𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(65.)

5855\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀯𑁂𑀤𑀺𑀓𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑁂𑀤𑀺𑀓𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(66.)

5856\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(67.)

5857\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(68.)

5858\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5859\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5860\. 𑀯𑁂𑀤𑀺𑀓𑀸𑀭𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
