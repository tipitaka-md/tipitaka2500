# 𑁪𑁫𑁇𑁭 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀯𑀺𑀪𑀻𑀢𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(42.)

5828\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑁄;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀫𑀤𑀸𑀲𑀳𑀁𑁇_  


(43.)

5829\. _𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀯𑀸,_  
_𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀯𑀲𑀸𑀫𑀳𑀁;_  
_𑀮𑀪𑀸𑀫𑀺 𑀯𑀺𑀧𑀼𑀮𑀁 𑀩𑁆𑀬𑀫𑁆𑀳𑀁,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(44.)

5830\. _𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀸 𑀭𑀽𑀧𑀺𑀫𑀬𑀸,_  
_𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀕𑀫𑀬𑀸 𑀩𑀳𑀽;_  
_𑀫𑀡𑀺𑀫𑀬𑀸 𑀘 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀲𑀤𑀸𑁇_  


(45.)

5831\. _𑀩𑁄𑀥𑀺𑀬𑀸 𑀆𑀲𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀚𑀮𑀚𑀼𑀢𑁆𑀢𑀫𑀦𑀸𑀫𑀺𑀦𑁄;_  
_𑀉𑀘𑁆𑀘𑁂 𑀓𑀼𑀮𑁂 𑀧𑀚𑀸𑀬𑀸𑀫𑀺,_  
_𑀅𑀳𑁄 𑀥𑀫𑁆𑀫𑀲𑀼𑀥𑀫𑁆𑀫𑀢𑀸𑁇_  


(46.)

5832\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀫𑀓𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(47.)

5833\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


(48.)

5834\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(49.)

5835\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


5836\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

5837\. 𑀲𑀻𑀳𑀸𑀲𑀦𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
