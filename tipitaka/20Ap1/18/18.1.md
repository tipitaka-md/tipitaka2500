# 𑁧𑁮𑁇𑁧 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀼𑀫𑀼𑀤𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

2510\. _“𑀧𑀩𑁆𑀩𑀢𑁂 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀫𑀳𑀸𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁄 𑀅𑀳𑀼;_  
_𑀢𑀢𑁆𑀣𑀚𑁄 𑀭𑀓𑁆𑀔𑀲𑁄 𑀆𑀲𑀺𑀁,_  
_𑀖𑁄𑀭𑀭𑀽𑀧𑁄 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(2.)

2511\. _𑀓𑀼𑀫𑀼𑀤𑀁 𑀧𑀼𑀧𑁆𑀨𑀢𑁂 𑀢𑀢𑁆𑀣,_  
_𑀘𑀓𑁆𑀓𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀚𑀸𑀬𑀭𑁂;_  
_𑀑𑀘𑀺𑀦𑀸𑀫𑀺 𑀘 𑀢𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀩𑀮𑀺𑀦𑁄 𑀲𑀫𑀺𑀢𑀺𑀁 𑀢𑀤𑀸𑁇_  


(3.)

2512\. _𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻 𑀢𑀼 𑀪𑀕𑀯𑀸,_  
_𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑁄 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀧𑀼𑀧𑁆𑀨𑀲𑀗𑁆𑀓𑁄𑀘𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(4.)

2513\. _𑀉𑀧𑀸𑀕𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑁂𑀯𑀤𑁂𑀯𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(5.)

2514\. _𑀬𑀸𑀯𑀢𑀸 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀦𑁆𑀢𑀸,_  
_𑀧𑀭𑀺𑀲𑀸 𑀲𑀸 𑀢𑀤𑀸 𑀅𑀳𑀼;_  
_𑀢𑀸𑀯𑀘𑁆𑀙𑀤𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀅𑀕𑀫𑀸𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑁄𑁇_  


(6.)

2515\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀢𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(7.)

2516\. _𑀇𑀢𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑁂𑀲𑀼𑀁 𑀚𑀦𑀸𑀥𑀺𑀧𑀸;_  
_𑀲𑀳𑀲𑁆𑀲𑀭𑀣𑀦𑀸𑀫𑀸 𑀢𑁂,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(8.)

2517\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2518\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2519\. 𑀓𑀼𑀫𑀼𑀤𑀫𑀸𑀮𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
