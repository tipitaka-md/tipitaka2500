# 𑁧𑁮𑁇𑁭 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀫𑀼𑀤𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(30.)

2551\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀫𑀼𑀦𑀺𑀦𑁄,_  
_𑀲𑀍𑀅𑀮𑁂 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀏𑀓𑀤𑀻𑀧𑀁 𑀅𑀤𑀸𑀲𑀳𑀁𑁇_  


(31.)

2552\. _𑀪𑀯𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂 𑀧𑀼𑀜𑁆𑀜𑀲𑀜𑁆𑀘𑀬𑁂;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(32.)

2553\. _𑀲𑁄𑀍𑀅𑀲𑁂 𑀓𑀧𑁆𑀧𑀲𑀳𑀲𑁆𑀲𑁂,_  
_𑀇𑀢𑁄 𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀚𑀦𑀸;_  
_𑀘𑀦𑁆𑀤𑀸𑀪𑀸 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(33.)

2554\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2555\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2556\. 𑀏𑀓𑀤𑀻𑀧𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇

  
𑀦𑀯𑀫𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇
