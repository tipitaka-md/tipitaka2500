# 𑁧𑁮𑁇𑁧𑁦 𑀲𑀁𑀖𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀼𑀫𑀼𑀤𑀯𑀕𑁆𑀕

* 𑀲𑀁𑀖𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(45.)

2572\. _“𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀫𑁆𑀳𑀺 𑀪𑀕𑀯𑀢𑀺,_  
_𑀅𑀳𑁄𑀲𑀸𑀭𑀸𑀫𑀺𑀓𑁄 𑀅𑀳𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀁 𑀲𑀁𑀖𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(46.)

2573\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(47.)

2574\. _𑀇𑀢𑁄 𑀢𑁂 𑀲𑀢𑁆𑀢𑀫𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑁆𑀢𑁂𑀯𑀸𑀲𑀼𑀁 𑀲𑀫𑁄𑀤𑀓𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(48.)

2575\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁨𑁧𑁪𑁬)_  


2576\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀁𑀖𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2577\. 𑀲𑀁𑀖𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

2578\. 𑀓𑀼𑀫𑀼𑀤𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀫𑁄𑁇



---

2579\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2580\. _𑀓𑀼𑀫𑀼𑀤𑁄 𑀅𑀣 𑀦𑀺𑀲𑁆𑀲𑁂𑀡𑀻,_  
_𑀭𑀢𑁆𑀢𑀺𑀓𑁄 𑀉𑀤𑀧𑀸𑀦𑀤𑁄;_  
_𑀲𑀻𑀳𑀸𑀲𑀦𑀻 𑀫𑀕𑁆𑀕𑀤𑀤𑁄,_  
_𑀏𑀓𑀤𑀻𑀧𑀻 𑀫𑀡𑀺𑀧𑁆𑀧𑀤𑁄;_  
_𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄,_  
_𑀏𑀓𑀧𑀜𑁆𑀜𑀸𑀲 𑀕𑀸𑀣𑀓𑀸𑀢𑀺𑁇_
