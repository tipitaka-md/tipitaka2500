# 𑁧𑁮𑁇𑁪 𑀉𑀤𑀧𑀸𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀼𑀫𑀼𑀤𑀯𑀕𑁆𑀕

* 𑀉𑀤𑀧𑀸𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(18.)

2533\. _“𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄,_  
_𑀉𑀤𑀧𑀸𑀦𑁄 𑀓𑀢𑁄 𑀫𑀬𑀸;_  
_𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀜𑁆𑀘 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀲𑀺𑀫𑀳𑀁 𑀢𑀤𑀸𑁇_  


(19.)

2534\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀉𑀤𑀧𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(20.)

2535\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2536\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀧𑀸𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2537\. 𑀉𑀤𑀧𑀸𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
