# 𑁧𑁮𑁇𑁮 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀫𑀼𑀤𑀯𑀕𑁆𑀕

### 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(34.)

2557\. _“𑀑𑀭𑁂𑀦 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀦𑀤𑀺𑀓𑀸 𑀲𑀫𑁆𑀧𑀯𑀢𑁆𑀢𑀣;_  
_𑀢𑀲𑁆𑀲𑀸 𑀘𑀸𑀦𑀼𑀧𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀯𑀲𑀢𑁂 𑀢𑀤𑀸𑁇_  


(35.)

2558\. _𑀫𑀡𑀺𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁,_  
_𑀲𑀸𑀥𑀼𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀦𑁄𑀭𑀫𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁𑁇_  


(36.)

2559\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀫𑀡𑀺𑀁 𑀅𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(37.)

2560\. _𑀇𑀢𑁄 𑀘 𑀤𑁆𑀯𑀸𑀤𑀲𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀢𑀭𑀁𑀲𑀻𑀲𑀦𑀸𑀫𑀓𑀸;_  
_𑀅𑀝𑁆𑀞 𑀢𑁂 𑀆𑀲𑀼𑀁 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(38.)

2561\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2562\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2563\. 𑀫𑀡𑀺𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
