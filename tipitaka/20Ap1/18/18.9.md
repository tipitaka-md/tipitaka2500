# 𑁧𑁮𑁇𑁯 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀫𑀼𑀤𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(39.)

2564\. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀯𑁂𑀚𑁆𑀚𑁄 𑀆𑀲𑀺𑀁 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀆𑀢𑀼𑀭𑀸𑀦𑀁 𑀲𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀁,_  
_𑀫𑀳𑀸𑀚𑀦𑀲𑀼𑀔𑀸𑀯𑀳𑁄𑁇_  


(40.)

2565\. _𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑀁 𑀲𑀫𑀡𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀪𑁂𑀲𑀚𑁆𑀚𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸𑁇_  


(41.)

2566\. _𑀅𑀭𑁄𑀕𑁄 𑀆𑀲𑀺 𑀢𑁂𑀦𑁂𑀯,_  
_𑀲𑀫𑀡𑁄 𑀲𑀁𑀯𑀼𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀅𑀲𑁄𑀓𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(42.)

2567\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀑𑀲𑀥𑀫𑀤𑀸𑀲𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑁆𑀲 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(43.)

2568\. _𑀇𑀢𑁄 𑀘 𑀅𑀝𑁆𑀞𑀫𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀩𑁆𑀩𑁄𑀲𑀥𑀲𑀦𑀸𑀫𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(44.)

2569\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2570\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2571\. 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
