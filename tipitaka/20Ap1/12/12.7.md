# 𑁧𑁨𑁇𑁭 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(57.)

2009. _“𑀬𑀤𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀮𑁄𑀓𑀕𑁆𑀕𑁄,_  
_𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀫𑁄𑀲𑁆𑀲𑀚𑀺;_  
_𑀧𑀣𑀯𑀻 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀺𑀢𑁆𑀣,_  
_𑀫𑁂𑀤𑀦𑀻 𑀚𑀮𑀫𑁂𑀔𑀮𑀸𑁇_  


(58.)

2010. _𑀑𑀢𑀢𑀁 𑀯𑀺𑀢𑁆𑀣𑀢𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀲𑀼𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀯𑀝𑀁𑀲𑀓𑀁;_  
_𑀪𑀯𑀦𑀫𑁆𑀧𑀺 𑀧𑀓𑀫𑁆𑀧𑀺𑀢𑁆𑀣,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀬𑁂𑁇_  


(59.)

2011. _𑀢𑀸𑀲𑁄 𑀫𑀬𑁆𑀳𑀁 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀪𑀯𑀦𑁂 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀺𑀢𑁂;_  
_𑀉𑀧𑁆𑀧𑀸𑀤𑁄 𑀦𑀼 𑀓𑀺𑀫𑀢𑁆𑀣𑀸𑀬,_  
_𑀆𑀮𑁄𑀓𑁄 𑀯𑀺𑀧𑀼𑀮𑁄 𑀅𑀳𑀼𑁇_  


(60.)

2012. _𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀇𑀥𑀸𑀕𑀫𑁆𑀫,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀲𑀺 𑀫𑀳𑀸𑀚𑀦𑀁;_  
_𑀧𑀸𑀡𑀪𑀽𑀢𑁂 𑀪𑀬𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀏𑀓𑀕𑁆𑀕𑀸 𑀳𑁄𑀣 𑀲𑀁𑀯𑀼𑀢𑀸𑁇_  


(61.)

2013. _𑀅𑀳𑁄 𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀳𑁄 𑀥𑀫𑁆𑀫𑀸,_  
_𑀅𑀳𑁄 𑀦𑁄 𑀲𑀢𑁆𑀣𑀼 𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀧𑀣𑀯𑀻 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺𑁇_  


(62.)

2014. _𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀁 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀁 𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀫𑁄𑀤𑀳𑀁;_  
_𑀅𑀯𑀲𑁂𑀲𑁂𑀲𑀼 𑀓𑀧𑁆𑀧𑁂𑀲𑀼,_  
_𑀓𑀼𑀲𑀮𑀁 𑀘𑀭𑀺𑀢𑀁 𑀫𑀬𑀸𑁇_  


(63.)

2015. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀲𑀜𑁆𑀜𑀫𑀮𑀪𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(64.)

2016. _𑀇𑀢𑁄 𑀘𑀼𑀤𑁆𑀤𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺𑀁 𑀧𑀢𑀸𑀧𑀯𑀸;_  
_𑀲𑀫𑀺𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(65.)

2017. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2018. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2019. 𑀩𑀼𑀤𑁆𑀥𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
