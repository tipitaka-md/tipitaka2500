# 𑁧𑁨𑁇𑁬 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀯𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(49.)

1999\. _“𑀯𑀲𑀸𑀫𑀺 𑀭𑀸𑀚𑀸𑀬𑀢𑀦𑁂,_  
_𑀲𑀸𑀫𑀘𑁆𑀘𑁄 𑀲𑀧𑀭𑀺𑀚𑁆𑀚𑀦𑁄;_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺,_  
_𑀲𑀺𑀔𑀺𑀦𑁄 𑀮𑁄𑀓𑀩𑀦𑁆𑀥𑀼𑀦𑁄𑁇_  


(50.)

2000\. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀘𑀺𑀢𑀓𑀁 𑀅𑀕𑀫𑀸𑀲𑀳𑀁;_  
_𑀢𑀽𑀭𑀺𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀕𑀦𑁆𑀥𑀫𑀸𑀮𑀁 𑀲𑀫𑁄𑀓𑀺𑀭𑀺𑀁𑁇_  


(51.)

2001\. _𑀘𑀺𑀢𑀫𑁆𑀳𑀺 𑀧𑀽𑀚𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀘𑀺𑀢𑀓𑀁 𑀅𑀳𑀁;_  
_𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀲𑀓𑀁 𑀪𑀯𑀦𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(52.)

2002\. _𑀪𑀯𑀦𑁂 𑀉𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑀳𑀁,_  
_𑀘𑀺𑀢𑀧𑀽𑀚𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪𑁇_  


(53.)

2003\. _𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁,_  
_𑀤𑁂𑀯𑁂𑀲𑀼 𑀫𑀸𑀦𑀼𑀲𑁂𑀲𑀼 𑀘;_  
_𑀧𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀬𑀧𑀭𑀸𑀚𑀬𑀁𑁇_  


(54.)

2004\. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀭𑁄𑀧𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀘𑀺𑀢𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(55.)

2005\. _𑀏𑀓𑀽𑀦𑀢𑀺𑀁𑀲𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀇𑀢𑁄 𑀲𑁄𑀍𑀅𑀲 𑀭𑀸𑀚𑀸𑀦𑁄;_  
_𑀉𑀕𑁆𑀕𑀢𑀸 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(56.)

2006\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2007\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2008\. 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
