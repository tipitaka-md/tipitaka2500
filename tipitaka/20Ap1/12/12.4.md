# 𑁧𑁨𑁇𑁪 𑀏𑀓𑀸𑀲𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀫𑀳𑀸𑀧𑀭𑀺𑀯𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀸𑀲𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(31.)

1977\. _“𑀯𑀭𑀼𑀡𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑀸 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀉𑀧𑀝𑁆𑀞𑀳𑁂𑀲𑀺𑀁 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀬𑁄𑀕𑁆𑀕𑀩𑀮𑀯𑀸𑀳𑀦𑁄𑁇_  


(32.)

1978\. _𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑀫𑁆𑀳𑀺,_  
_𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀻𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑀽𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀫𑀸𑀤𑀸𑀬,_  
_𑀅𑀕𑀫𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(33.)

1979\. _𑀯𑀸𑀤𑀺𑀢𑁂𑀦 𑀘 𑀦𑀘𑁆𑀘𑁂𑀦,_  
_𑀲𑀫𑁆𑀫𑀢𑀸𑀍𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀯𑀺𑀬 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(34.)

1980\. _𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀩𑁄𑀥𑀺𑀁,_  
_𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀧𑀸𑀤𑀧𑀁;_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀅𑀳𑀁𑁇_  


(35.)

1981\. _𑀲𑀓𑀓𑀫𑁆𑀫𑀸𑀪𑀺𑀭𑀤𑁆𑀥𑁄𑀳𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦,_  
_𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀳𑀁𑁇_  


(36.)

1982\. _𑀲𑀝𑁆𑀞𑀺𑀢𑀽𑀭𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀁 𑀲𑀤𑀸;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀘 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀁 𑀪𑀯𑀸𑀪𑀯𑁂𑁇_  


(37.)

1983\. _𑀢𑀺𑀯𑀺𑀥𑀕𑁆𑀕𑀻 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀥𑀸𑀭𑁂𑀫𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(38.)

1984\. _𑀲𑀼𑀩𑀸𑀳𑀽 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀘𑀢𑀼𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀲𑀼 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀧𑀜𑁆𑀘𑀓𑀧𑁆𑀧𑀲𑀢𑁂 𑀇𑀢𑁄𑁇_  


(39.)

1985\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1986\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀸𑀲𑀦𑀺𑀬𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1987\. 𑀏𑀓𑀸𑀲𑀦𑀺𑀬𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
