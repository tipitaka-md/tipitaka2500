# 𑁧𑁨𑁇𑁯 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀫𑀳𑀸𑀧𑀭𑀺𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(72.)

2028\. _“𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀫𑁆𑀳𑀺 𑀲𑀼𑀕𑀢𑁂,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸;_  
_𑀬𑀓𑁆𑀔𑀬𑁄𑀦𑀺𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺𑀁,_  
_𑀬𑀲𑀁 𑀧𑀢𑁆𑀢𑁄 𑀘𑀳𑀁 𑀢𑀤𑀸𑁇_  


(73.)

2029\. _𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀤𑀼𑀧𑁆𑀧𑀪𑀸𑀢𑀁 𑀤𑀼𑀭𑀼𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀬𑀁 𑀫𑁂 𑀪𑁄𑀕𑁂 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂,_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(74.)

2030\. _𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀸𑀕𑀭𑁄 𑀦𑀸𑀫 𑀲𑀸𑀯𑀓𑁄;_  
_𑀫𑀫𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(75.)

2031\. _𑀓𑀺𑀁 𑀦𑀼 𑀲𑁄𑀘𑀲𑀺 𑀫𑀸 𑀪𑀸𑀬𑀺,_  
_𑀘𑀭 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀫𑁂𑀥𑀲;_  
_𑀅𑀦𑀼𑀧𑁆𑀧𑀤𑀺𑀦𑁆𑀦𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀩𑀻𑀚𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


(76.)

2032\. _𑀬𑁄 𑀘𑁂 𑀧𑀽𑀚𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀥𑀸𑀢𑀼𑀁 𑀲𑀸𑀲𑀧𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲𑀸𑀧𑀺 𑀧𑀽𑀚𑀬𑁂𑁇_  


(77.)

2033\. _𑀲𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑀫𑁆𑀳𑀺,_  
_𑀲𑀫𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀫𑀳𑀕𑁆𑀕𑀢𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀣𑀽𑀧𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀽𑀚𑁂𑀳𑀺 𑀚𑀺𑀦𑀥𑀸𑀢𑀼𑀬𑁄𑁇_  


(78.)

2034\. _𑀲𑀸𑀕𑀭𑀲𑁆𑀲 𑀯𑀘𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀣𑀽𑀧𑀁 𑀅𑀓𑀸𑀲𑀳𑀁;_  
_𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁂 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀫𑀼𑀦𑀺𑀦𑁄 𑀣𑀽𑀧𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(79.)

2035\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞 𑀦𑀭𑀸𑀲𑀪;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(80.)

2036\. _𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀸 𑀘 𑀘𑀢𑁆𑀢𑀸𑀭𑁄,_  
_𑀲𑀢𑁆𑀢𑀓𑀧𑁆𑀧𑀲𑀢𑁂 𑀇𑀢𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(81.)

2037\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


2038\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

2039\. 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
