# 𑁨𑁬𑁇𑁮 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀣𑁄𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(28.)

3135\. _“𑀧𑀩𑁆𑀩𑀢𑁂 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀯𑀸𑀓𑀘𑀻𑀭𑀥𑀭𑁄 𑀅𑀳𑀁;_  
_𑀅𑀮𑁄𑀡𑀧𑀡𑁆𑀡𑀪𑀓𑁆𑀔𑁄𑀫𑁆𑀳𑀺,_  
_𑀦𑀺𑀬𑀫𑁂𑀲𑀼 𑀘 𑀲𑀁𑀯𑀼𑀢𑁄𑁇_  


(29.)

3136\. _𑀧𑀸𑀢𑀭𑀸𑀲𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑀺 𑀫𑀁;_  
_𑀢𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(30.)

3137\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀡𑁆𑀡𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀡𑁆𑀡𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(31.)

3138\. _𑀲𑀢𑁆𑀢𑀯𑀻𑀲𑀢𑀺𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀭𑀸𑀚𑀸 𑀆𑀲𑀺𑀁 𑀲𑀤𑀢𑁆𑀣𑀺𑀬𑁄;_  
_𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄𑁇_  


(32.)

3139\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3140\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀡𑁆𑀡𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---
