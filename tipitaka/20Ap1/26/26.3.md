# 𑁨𑁬𑁇𑁩 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀣𑁄𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(9.)

3106. _“𑀆𑀦𑀦𑁆𑀤𑁄 𑀦𑀸𑀫 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀬𑀫𑁆𑀪𑀽 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀫𑁆𑀳𑀺 𑀓𑀸𑀦𑀦𑁂𑁇_  


(10.)

3107. _𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀇𑀥𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀘𑀺𑀢𑀁 𑀓𑀢𑁆𑀯𑀸𑀦𑀳𑀁 𑀢𑀤𑀸;_  
_𑀲𑀭𑀻𑀭𑀁 𑀢𑀢𑁆𑀣 𑀛𑀸𑀧𑁂𑀲𑀺𑀁,_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑀜𑁆𑀘 𑀅𑀓𑀸𑀲𑀳𑀁𑁇_  


(11.)

3108. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(12.)

3109. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3110. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3111. 𑀘𑀺𑀢𑀓𑀧𑀽𑀚𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
