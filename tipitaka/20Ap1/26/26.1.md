# 𑁨𑁬𑁇𑁧 𑀣𑁄𑀫𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀣𑁄𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀣𑁄𑀫𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(1.)

3094\. _“𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀞𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑁄,_  
_𑀇𑀫𑀁 𑀯𑀸𑀘𑀁 𑀅𑀪𑀸𑀲𑀳𑀁𑁇_  


(2.)

3095\. _‘𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀩𑀳𑀼𑀚𑁆𑀚𑀦𑀁 𑀢𑀸𑀭𑀬𑀲𑀺,_  
_𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁’𑁇_  


(3.)

3096\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀯𑀸𑀘𑀫𑀪𑀡𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀣𑁄𑀫𑀦𑀸𑀬 𑀇𑀤𑀁 𑀨𑀮𑀁𑁇_  


(4.)

3097\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3098\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀣𑁄𑀫𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3099\. 𑀣𑁄𑀫𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀞𑀫𑀁𑁇
