# 𑁨𑁬𑁇𑁬 𑀉𑀧𑀸𑀳𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀣𑁄𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀳𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(20.)

3123\. _“𑀅𑀳𑁄𑀲𑀺𑀁 𑀘𑀦𑁆𑀤𑀦𑁄 𑀦𑀸𑀫,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀢𑁆𑀭𑀚𑁄 𑀢𑀤𑀸;_  
_𑀏𑀓𑁄𑀧𑀸𑀳𑀦𑁄 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑁄,_  
_𑀩𑁄𑀥𑀺𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀫𑁂 𑀢𑀼𑀯𑀁𑁇_  


(21.)

3124\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀸𑀦𑀥𑀺𑀁 𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀉𑀧𑀸𑀳𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(22.)

3125\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3126\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀳𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3127\. 𑀉𑀧𑀸𑀳𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
