# 𑁨𑁬𑁇𑁨 𑀏𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

* 𑀣𑁄𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(5.)

3100\. _“𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑀯𑀡𑁆𑀡𑀁,_  
_𑀲𑀪𑀭𑀺𑀬𑁄 𑀇𑀥𑀸𑀕𑀫𑀺𑀁;_  
_𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


(6.)

3101\. _𑀤𑁂𑀯𑀮𑁄 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀲𑀸𑀯𑀓𑁄;_  
_𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀸 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(7.)

3102\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀭𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(8.)

3103\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3104\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀏𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3105\. 𑀏𑀓𑀸𑀲𑀦𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
