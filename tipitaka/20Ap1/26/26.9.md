# 𑁨𑁬𑁇𑁯 𑀓𑀼𑀝𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

### 𑀣𑁄𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀝𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀅𑀧𑀤𑀸𑀦

(33.)

3142. _“𑀯𑀺𑀧𑀺𑀦𑀘𑀸𑀭𑀻 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀲𑀻 𑀢𑀤𑀸;_  
_𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀤𑀸𑀲𑀺𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁂𑁇_  


(34.)

3143. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀺𑀓𑀁 𑀅𑀤𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑀼𑀝𑀺𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(35.)

3144. _𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑁄𑀍𑀅𑀲𑀸𑀲𑀺𑀁𑀲𑀼 𑀭𑀸𑀚𑀸𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀻𑀢𑀺,_  
_𑀯𑀼𑀘𑁆𑀘𑀭𑁂 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(36.)

3145. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


3146. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀝𑀺𑀤𑀸𑀬𑀓𑁄 𑀣𑁂𑀭𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

3147. 𑀓𑀼𑀝𑀺𑀤𑀸𑀬𑀓𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
