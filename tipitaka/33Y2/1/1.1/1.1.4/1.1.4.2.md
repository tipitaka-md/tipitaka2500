# 𑁧𑁇𑁧𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁅𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀔𑀦𑁆𑀥𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(21.)

163\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

164\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸?

165\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

166\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸?

167\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

168\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸?

169\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

170\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

(22.)

171\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

172\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁?

173\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

174\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸?

175\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

176\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸?

177\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

178\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

(23.)

179\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

180\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁?

181\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

182\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸?

183\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

184\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸?

185\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

186\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

(24.)

187\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

188\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁?

189\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

190\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸?

191\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

192\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸?

193\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄?

194\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁?

(25.)

195\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

196\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁?

197\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

198\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸?

199\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

200\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸?

201\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄?

202\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸?
