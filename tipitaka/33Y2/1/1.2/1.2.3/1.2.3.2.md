# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀔𑀦𑁆𑀥𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(39.)

358\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

359\. 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀭𑀽𑀧𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

360\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

361\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

362\. 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑁂𑀤𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

363\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

364\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

365\. 𑀲𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

366\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

367\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

368\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

369\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

370\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

371\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
