# 𑁧𑁇𑁨𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀔𑀦𑁆𑀥𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫

(38.)

343\. »  𑀭𑀽𑀧𑀁 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

344\. «  𑀔𑀦𑁆𑀥𑀸 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

345\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

346\. »  𑀯𑁂𑀤𑀦𑀸 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

347\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

348\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

349\. »  𑀲𑀜𑁆𑀜𑀸 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

350\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

351\. 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

352\. »  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

353\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

354\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

355\. »  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

356\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

357\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇
