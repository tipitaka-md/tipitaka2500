# 𑁧𑁇𑁨𑁇𑁪𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀅𑀦𑀼𑀮𑁄𑀫

373\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

374\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

376\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

377\. 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

378\. »  𑀭𑀽𑀧𑀁 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

379\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

380\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

388\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

391\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

392\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

398\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

404\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

420\. »  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

422\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

423\. »  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

425\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇
