# 𑁧𑁇𑁨𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀔𑀦𑁆𑀥𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(45.)

433\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

434\. 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀭𑀽𑀧𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

435\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

436\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

437\. 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀭𑀽𑀧𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

438\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

439\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

440\. 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀭𑀽𑀧𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

441\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

442\. »  𑀦 𑀭𑀽𑀧𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

443\. 𑀭𑀽𑀧𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀭𑀽𑀧𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀭𑀽𑀧𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

444\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(46.)

445\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

446\. 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑁂𑀤𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

447\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

448\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

449\. 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑁂𑀤𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

450\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

451\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

452\. 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑁂𑀤𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

453\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

454\. »  𑀦 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

455\. 𑀯𑁂𑀤𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑁂𑀤𑀦𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

456\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(47.)

457\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

458\. 𑀲𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

459\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

460\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

461\. 𑀲𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

462\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

463\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

464\. 𑀲𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

465\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

466\. »  𑀦 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

467\. 𑀲𑀜𑁆𑀜𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀲𑀜𑁆𑀜𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

468\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(48.)

469\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

470\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

471\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

472\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

473\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

474\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

475\. »  𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

476\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(49.)

477\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

478\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

479\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

480\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

481\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

482\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

483\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

484\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

485\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

486\. »  𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

487\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀔𑀦𑁆𑀥𑀸𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀦 𑀘 𑀔𑀦𑁆𑀥𑀸𑁇

488\. «  𑀦 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
