# 𑁧𑁇𑁨𑁇𑁨𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫

(28.)

230\. »  𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

231\. 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀭𑀽𑀧𑀜𑁆𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

232\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

233\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

234\. 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

235\. »  𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

236\. 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀭𑀽𑀧𑀜𑁆𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

237\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

238\. 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

239\. »  𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

240\. 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀭𑀽𑀧𑀜𑁆𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

241\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

242\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

243\. »  𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

244\. 𑀧𑀺𑀬𑀭𑀽𑀧𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑀁 𑀭𑀽𑀧𑀁, 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀭𑀽𑀧𑀜𑁆𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

245\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

246\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(29.)

247\. »  𑀯𑁂𑀤𑀦𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

248\. «  𑀔𑀦𑁆𑀥𑀸 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

249\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

250\. »  𑀯𑁂𑀤𑀦𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

251\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

252\. 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

253\. »  𑀯𑁂𑀤𑀦𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

254\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

255\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

256\. »  𑀯𑁂𑀤𑀦𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

257\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

258\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(30.)

259\. »  𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

260\. 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

261\. «  𑀔𑀦𑁆𑀥𑀸 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

262\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

263\. »  𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

264\. 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

265\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

266\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

267\. »  𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

268\. 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

269\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

270\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

271\. »  𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

272\. 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀜𑀸, 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

273\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

274\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(31.)

275\. »  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

276\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

277\. «  𑀔𑀦𑁆𑀥𑀸 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

278\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

279\. »  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

280\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

281\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

282\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

283\. »  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

284\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

285\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

286\. 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

287\. »  𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

288\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇

289\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

290\. 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

(32.)

291\. »  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

292\. «  𑀔𑀦𑁆𑀥𑀸 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

293\. 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

294\. »  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

295\. «  𑀔𑀦𑁆𑀥𑀸 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

296\. 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

297\. »  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

298\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

299\. 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇

300\. »  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

301\. «  𑀔𑀦𑁆𑀥𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺?

302\. 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀔𑀦𑁆𑀥𑀸 𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇
