# 𑁨𑁇𑁧𑁇𑁪𑁇𑁫 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀢𑀻𑀢𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

(76.)

574\. »  𑀬𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

575\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

(77.)

576\. »  𑀬𑀢𑁆𑀣 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

577\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
