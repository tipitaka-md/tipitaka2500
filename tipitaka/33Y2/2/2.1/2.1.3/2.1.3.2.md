# 𑁨𑁇𑁧𑁇𑁩𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(63.)

533\. »  𑀬𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀢𑁆𑀣 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

534\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

535\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

536\. 𑀅𑀭𑀽𑀧𑁂 𑀢𑀢𑁆𑀣 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀢𑀢𑁆𑀣 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇
