# 𑁩𑁇𑁩 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

* 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

(208.)

971\. »  𑀬𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

972\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

973\. »  𑀬𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

974\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
