# 𑁩𑁇𑁨 𑀅𑀢𑀻𑀢𑀯𑀸𑀭

### 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

### 𑀅𑀢𑀻𑀢𑀯𑀸𑀭

(207.)

967. »  𑀬𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

968. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

969. »  𑀬𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

970. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣 𑀲𑁄 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
