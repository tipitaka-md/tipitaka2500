# 𑁧𑁦𑁇𑁨𑁧--𑁩𑁧

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀫𑀸𑀤𑀸𑀤𑀺𑀯𑀕𑁆𑀕

(118--128.)

231\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁄 𑀏𑀯𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸…𑀧𑁂…  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸…  𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸…  𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸…  𑀅𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄…  𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄…  𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁…  𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀁…  𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸…  𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸…  𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁𑁇 𑀅𑀦𑀼𑀬𑁄𑀕𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑁄𑀲𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
