# 𑁧𑁪𑁇𑁧𑁇𑁧--𑁧𑁦

### 𑀏𑀢𑀤𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

(188.)

271\. “𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀭𑀢𑁆𑀢𑀜𑁆𑀜𑀽𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄𑁇 (𑁧)

(189.)

272\. …  𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑁇 (𑁨)

(190.)

273\. …  𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄𑁇 (𑁩)

(191.)

274\. …  𑀥𑀼𑀢𑀯𑀸𑀤𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄𑁇 (𑁪)

(192.)

275\. …  𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄𑁇 (𑁫)

(193.)

276\. …  𑀉𑀘𑁆𑀘𑀸𑀓𑀼𑀮𑀺𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀓𑀸𑀍𑀇𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑁄𑁇 (𑁬)

(194.)

277\. …  𑀫𑀜𑁆𑀚𑀼𑀲𑁆𑀲𑀭𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀮𑀓𑀼𑀡𑁆𑀟𑀓 𑀪𑀤𑁆𑀤𑀺𑀬𑁄𑁇 (𑁭)

(195.)

278\. …  𑀲𑀻𑀳𑀦𑀸𑀤𑀺𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀺𑀡𑁆𑀟𑁄𑀮𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄𑁇 (𑁮)

(196.)

279\. …  𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀼𑀡𑁆𑀡𑁄 𑀫𑀦𑁆𑀢𑀸𑀡𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑁇 (𑁯)

(197.)

280\. …  𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀪𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄”𑀢𑀺𑁇 (𑁧𑁦)

---
