# 𑁧𑁪𑁇𑁪𑁇𑁧--𑁧𑁬

### 𑀏𑀢𑀤𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

(219.)

305. “𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 (𑁧)

(220.)

306. …  𑀲𑀢𑀺𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 (𑁨)

(221.)

307. …  𑀕𑀢𑀺𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 (𑁩)

(222.)

308. …  𑀥𑀺𑀢𑀺𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 (𑁪)

(223.)

309. …  𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀆𑀦𑀦𑁆𑀤𑁄𑁇 (𑁫)

(224.)

310. …  𑀫𑀳𑀸𑀧𑀭𑀺𑀲𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄𑁇 (𑁬)

(225.)

311. …  𑀓𑀼𑀮𑀧𑁆𑀧𑀲𑀸𑀤𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀓𑀸𑀍𑀉𑀤𑀸𑀬𑀻𑁇 (𑁭)

(226.)

312. …  𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀩𑀸𑀓𑀼𑀮𑁄𑁇 (𑁮)

(227.)

313. …  𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑁄𑀪𑀺𑀢𑁄𑁇 (𑁯)

(228.)

314. …  𑀯𑀺𑀦𑀬𑀥𑀭𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀉𑀧𑀸𑀮𑀺𑁇 (𑁧𑁦)

(229.)

315. …  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀦𑀦𑁆𑀤𑀓𑁄𑁇 (𑁧𑁧)

(230.)

316. …  𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀦𑀦𑁆𑀤𑁄𑁇 (𑁧𑁨)

(231.)

317. …  𑀪𑀺𑀓𑁆𑀔𑀼𑀑𑀯𑀸𑀤𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑁄𑁇 (𑁧𑁩)

(232.)

318. …  𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀸𑀕𑀢𑁄𑁇 (𑁧𑁪)

(233.)

319. …  𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀬𑁆𑀬𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀭𑀸𑀥𑁄𑁇 (𑁧𑁫)

(234.)

320. …  𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀥𑀭𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑁄𑀖𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 (𑁧𑁬)

---

321. 𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
