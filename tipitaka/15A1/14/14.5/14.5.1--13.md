# 𑁧𑁪𑁇𑁫𑁇𑁧--𑁧𑁩

### 𑀏𑀢𑀤𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀧𑀜𑁆𑀘𑀫𑀯𑀕𑁆𑀕

(235.)

322\. “𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀫 𑀲𑀸𑀯𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀭𑀢𑁆𑀢𑀜𑁆𑀜𑀽𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀧𑀚𑀸𑀧𑀢𑀺𑀕𑁄𑀢𑀫𑀻𑁇 (𑁧)

(236.)

323\. …  𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀔𑁂𑀫𑀸𑁇 (𑁨)

(237.)

324\. …  𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑀻𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸𑁇 (𑁩)

(238.)

325\. …  𑀯𑀺𑀦𑀬𑀥𑀭𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑁇 (𑁪)

(239.)

326\. …  𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸𑁇 (𑁫)

(240.)

327\. …  𑀛𑀸𑀬𑀻𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀦𑀦𑁆𑀤𑀸𑁇 (𑁬)

(241.)

328\. …  𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑁄𑀡𑀸𑁇 (𑁭)

(242.)

329\. …  𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀩𑀓𑀼𑀮𑀸𑁇 (𑁮)

(243.)

330\. …  𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀪𑀤𑁆𑀤𑀸 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀸𑁇 (𑁯)

(244.)

331\. …  𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀻𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀪𑀤𑁆𑀤𑀸 𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀻𑁇 (𑁧𑁦)

(245.)

332\. …  𑀫𑀳𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀢𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀪𑀤𑁆𑀤𑀓𑀘𑁆𑀘𑀸𑀦𑀸𑁇 (𑁧𑁧)

(246.)

333\. …  𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀥𑀭𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻𑁇 (𑁧𑁨)

(247.)

334\. …  𑀲𑀤𑁆𑀥𑀸𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀓𑀫𑀸𑀢𑀸”𑀢𑀺𑁇 (𑁧𑁩)
