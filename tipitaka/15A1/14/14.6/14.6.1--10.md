# 𑁧𑁪𑁇𑁬𑁇𑁧--𑁧𑁦

### 𑀏𑀢𑀤𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀙𑀝𑁆𑀞𑀯𑀕𑁆𑀕

(248.)

336\. “𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀧𑀞𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀢𑀧𑀼𑀲𑁆𑀲𑀪𑀮𑁆𑀮𑀺𑀓𑀸 𑀯𑀸𑀡𑀺𑀚𑀸𑁇 (𑁧)

(249.)

337\. …  𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀼𑀤𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄𑁇 (𑁨)

(250.)

338\. …  𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺 𑀫𑀘𑁆𑀙𑀺𑀓𑀸𑀲𑀡𑁆𑀟𑀺𑀓𑁄𑁇 (𑁩)

(251.)

339\. …  𑀘𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀧𑀭𑀺𑀲𑀁 𑀲𑀗𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀳𑀢𑁆𑀣𑀓𑁄 𑀆𑀍𑀅𑀯𑀓𑁄𑁇 (𑁪)

(252.)

340\. …  𑀧𑀡𑀻𑀢𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀦𑀸𑀫𑁄 𑀲𑀓𑁆𑀓𑁄𑁇 (𑁫)

(253.)

341\. …  𑀫𑀦𑀸𑀧𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀉𑀕𑁆𑀕𑁄 𑀕𑀳𑀧𑀢𑀺 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑁄𑁇 (𑁬)

(254.)

342\. …  𑀲𑀁𑀖𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀳𑀢𑁆𑀣𑀺𑀕𑀸𑀫𑀓𑁄 𑀉𑀕𑁆𑀕𑀢𑁄 𑀕𑀳𑀧𑀢𑀺𑁇 (𑁭)

(255.)

343\. …  𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀲𑀽𑀭𑀫𑁆𑀩𑀝𑁆𑀞𑁄𑁇 (𑁮)

(256.)

344\. …  𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄𑁇 (𑁯)

(257.)

345\. …  𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀓𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀦𑀓𑀼𑀮𑀧𑀺𑀢𑀸 𑀕𑀳𑀧𑀢𑀻”𑀢𑀺𑁇 (𑁧𑁦)

---
