# 𑁭𑁇𑁭

* 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀸𑀤𑀺𑀯𑀕𑁆𑀕

(67.)

141\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄𑁇 𑀬𑁄𑀦𑀺𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
