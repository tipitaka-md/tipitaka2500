# 𑁧𑁯𑁇𑁯

### 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀯𑀕𑁆𑀕

(571.)

460. “𑀏𑀓𑀥𑀫𑁆𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀓𑀸𑀬𑁄𑀧𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀧𑀺 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀓𑁂𑀯𑀮𑀸𑀧𑀺 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀲𑁆𑀫𑀺𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑁂? 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸𑁇 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁂 𑀪𑀸𑀯𑀺𑀢𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀓𑀸𑀬𑁄𑀧𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀧𑀺 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀓𑁂𑀯𑀮𑀸𑀧𑀺 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
