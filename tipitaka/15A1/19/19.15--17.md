# 𑁧𑁯𑁇𑁧𑁫--𑁧𑁭

* 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀯𑀕𑁆𑀕

(577--579.)

465\. “𑀏𑀓𑀥𑀫𑁆𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺…  𑀦𑀸𑀦𑀸𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺…  𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑀲𑁆𑀫𑀺𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑁂? 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸𑁇 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁂 𑀪𑀸𑀯𑀺𑀢𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺…  𑀦𑀸𑀦𑀸𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺…  𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
