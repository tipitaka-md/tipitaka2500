# 𑁧𑁯𑁇𑁧𑁦

* 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀯𑀕𑁆𑀕

(572.)

461\. “𑀏𑀓𑀥𑀫𑁆𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀲𑁆𑀫𑀺𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑁂? 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸𑁇 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁂 𑀪𑀸𑀯𑀺𑀢𑁂 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀻𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
