# 𑁧𑁯𑁇𑁨--𑁮

### 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀯𑀕𑁆𑀕

(564--570.)

459\. “𑀏𑀓𑀥𑀫𑁆𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀫𑀳𑀢𑁄 𑀲𑀁𑀯𑁂𑀕𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀫𑀳𑀢𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀫𑀳𑀢𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀲𑀢𑀺𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑀥𑀫𑁆𑀫𑁄? 𑀓𑀸𑀬𑀕𑀢𑀸 𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀫𑀳𑀢𑁄 𑀲𑀁𑀯𑁂𑀕𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀫𑀳𑀢𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀫𑀳𑀢𑁄 𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀲𑀢𑀺𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
