# 𑁧𑁯𑁇𑁧𑁮--𑁨𑁧

### 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀯𑀕𑁆𑀕

(580--583.)

466. “𑀏𑀓𑀥𑀫𑁆𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑀥𑀫𑁆𑀫𑁄? 𑀓𑀸𑀬𑀕𑀢𑀸 𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
