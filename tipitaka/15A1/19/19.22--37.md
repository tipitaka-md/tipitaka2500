# 𑁧𑁯𑁇𑁨𑁨--𑁩𑁭

### 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀯𑀕𑁆𑀕

(584--599.)

467\. “𑀏𑀓𑀥𑀫𑁆𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀜𑁆𑀜𑀸𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀼𑀣𑀼𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀯𑀺𑀧𑀼𑀮𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀲𑀸𑀫𑀦𑁆𑀢𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀜𑁆𑀜𑀸𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀲𑀻𑀖𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀮𑀳𑀼𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀢𑀺𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑀥𑀫𑁆𑀫𑁄? 𑀓𑀸𑀬𑀕𑀢𑀸 𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀜𑁆𑀜𑀸𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀼𑀣𑀼𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀯𑀺𑀧𑀼𑀮𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀅𑀲𑀸𑀫𑀦𑁆𑀢𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀧𑀜𑁆𑀜𑀸𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀲𑀻𑀖𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀮𑀳𑀼𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀳𑀸𑀲𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀚𑀯𑀦𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀢𑀺𑀓𑁆𑀔𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺…  𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀧𑀜𑁆𑀜𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
