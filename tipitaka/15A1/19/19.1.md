# 𑁧𑁯𑁇𑁧

(563.)

458\. “𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀘𑁂𑀢𑀲𑀸 𑀨𑀼𑀝𑁄 𑀅𑀦𑁆𑀢𑁄𑀕𑀥𑀸 𑀢𑀲𑁆𑀲 𑀓𑀼𑀦𑁆𑀦𑀤𑀺𑀬𑁄 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀲𑀫𑀼𑀤𑁆𑀤𑀗𑁆𑀕𑀫𑀸; 𑀏𑀯𑀫𑁂𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀓𑀸𑀬𑀕𑀢𑀸 𑀲𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀅𑀦𑁆𑀢𑁄𑀕𑀥𑀸 𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀯𑀺𑀚𑁆𑀚𑀸𑀪𑀸𑀕𑀺𑀬𑀸”𑀢𑀺𑁇
