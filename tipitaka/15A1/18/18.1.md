# 𑁧𑁮𑁇𑁧

* 𑀅𑀧𑀭𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀖𑀸𑀢𑀯𑀕𑁆𑀕

(382.)

440\. “𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀖𑀸𑀢𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀺𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀭𑁄 𑀑𑀯𑀸𑀤𑀧𑀢𑀺𑀓𑀭𑁄, 𑀅𑀫𑁄𑀖𑀁 𑀭𑀝𑁆𑀞𑀧𑀺𑀡𑁆𑀟𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺’𑁇 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀬𑁂 𑀦𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
