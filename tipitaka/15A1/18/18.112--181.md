# 𑁧𑁮𑁇𑁧𑁧𑁨--𑁧𑁮𑁧

* 𑀅𑀧𑀭𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀖𑀸𑀢𑀯𑀕𑁆𑀕

(493--562.)

456\. 𑀤𑀼𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀚𑁆𑀛𑀸𑀦𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀚𑁆𑀛𑀸𑀦𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀓𑀭𑀼𑀡𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀳𑀕𑀢𑀁…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀲𑀤𑁆𑀥𑀸𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀯𑀻𑀭𑀺𑀬𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀲𑀢𑀺𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺…  𑀧𑀜𑁆𑀜𑀸𑀩𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀺𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀭𑁄 𑀑𑀯𑀸𑀤𑀧𑀢𑀺𑀓𑀭𑁄, 𑀅𑀫𑁄𑀖𑀁 𑀭𑀝𑁆𑀞𑀧𑀺𑀡𑁆𑀟𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺’𑁇 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀬𑁂 𑀦𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
