# 𑁬𑁇𑁧𑁦

### 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀖𑀸𑀢𑀯𑀕𑁆𑀕

(60.)

126\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑁄𑀲𑀚𑁆𑀚𑀁𑁇 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
