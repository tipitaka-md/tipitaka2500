### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

(284.)

373\. “𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀲𑁆𑀲 𑀇𑀝𑁆𑀞𑁄 𑀓𑀦𑁆𑀢𑁄 𑀫𑀦𑀸𑀧𑁄 𑀯𑀺𑀧𑀸𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑁇 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀞𑀸𑀦𑀜𑁆𑀘 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀲𑁆𑀲 𑀅𑀦𑀺𑀝𑁆𑀞𑁄 𑀅𑀓𑀦𑁆𑀢𑁄 𑀅𑀫𑀦𑀸𑀧𑁄 𑀯𑀺𑀧𑀸𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑁇 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
