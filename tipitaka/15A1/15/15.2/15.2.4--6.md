# 𑁧𑁫𑁇𑁨𑁇𑁪--𑁬

### 𑀅𑀝𑁆𑀞𑀸𑀦𑀧𑀸𑀍𑀇

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

(281--283.)

372\. “𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀇𑀢𑁆𑀣𑀻 𑀲𑀓𑁆𑀓𑀢𑁆𑀢𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬…𑀧𑁂…  𑀫𑀸𑀭𑀢𑁆𑀢𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬…𑀧𑁂…  𑀩𑁆𑀭𑀳𑁆𑀫𑀢𑁆𑀢𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬𑁇 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀞𑀸𑀦𑀜𑁆𑀘 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀢𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬…𑀧𑁂…  𑀫𑀸𑀭𑀢𑁆𑀢𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬…𑀧𑁂…  𑀩𑁆𑀭𑀳𑁆𑀫𑀢𑁆𑀢𑀁 𑀓𑀸𑀭𑁂𑀬𑁆𑀬𑁇 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
