# 𑁧𑁫𑁇𑁨𑁇𑁧

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

(278.)

369\. “𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂 𑀭𑀸𑀚𑀸𑀦𑁄 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀞𑀸𑀦𑀜𑁆𑀘 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑁄 𑀭𑀸𑀚𑀸 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
