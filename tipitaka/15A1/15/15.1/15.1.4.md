### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

361. “𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀸𑀢𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑁇 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀞𑀸𑀦𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀫𑀸𑀢𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑁇 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
