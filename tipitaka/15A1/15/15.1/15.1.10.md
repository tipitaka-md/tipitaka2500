# 𑁧𑁫𑁇𑁧𑁇𑁧𑁦

### 𑀅𑀝𑁆𑀞𑀸𑀦𑀧𑀸𑀍𑀇

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

(277.)

367. “𑀅𑀝𑁆𑀞𑀸𑀦𑀫𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀞𑀸𑀦𑀜𑁆𑀘 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀏𑀓𑀺𑀲𑁆𑀲𑀸 𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀓𑁄𑀯 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

---

368. 𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇
