# 𑁫𑁇𑁩

* 𑀧𑀡𑀺𑀳𑀺𑀢𑀅𑀘𑁆𑀙𑀯𑀕𑁆𑀕

(43.)

91\. “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀫𑁆𑀳𑀺 𑀘𑁂 𑀅𑀬𑀁 𑀲𑀫𑀬𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂’𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀤𑀼𑀝𑁆𑀞𑀁𑁇 𑀘𑁂𑀢𑁄𑀧𑀤𑁄𑀲𑀳𑁂𑀢𑀼 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
