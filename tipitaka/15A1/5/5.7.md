# 𑁫𑁇𑁭

* 𑀧𑀡𑀺𑀳𑀺𑀢𑀅𑀘𑁆𑀙𑀯𑀕𑁆𑀕

(47.)

99\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑀺 𑀓𑀸𑀦𑀺𑀘𑀺 𑀭𑀼𑀓𑁆𑀔𑀚𑀸𑀢𑀸𑀦𑀁 𑀨𑀦𑁆𑀤𑀦𑁄 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺 𑀬𑀤𑀺𑀤𑀁 𑀫𑀼𑀤𑀼𑀢𑀸𑀬 𑀘𑁂𑀯 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸𑀬 𑀘𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀁 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀫𑀼𑀤𑀼 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀜𑁆𑀘 𑀬𑀣𑀬𑀺𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁𑁇 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀫𑀼𑀤𑀼 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀜𑁆𑀘 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
