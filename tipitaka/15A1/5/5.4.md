# 𑁫𑁇𑁪

* 𑀧𑀡𑀺𑀳𑀺𑀢𑀅𑀘𑁆𑀙𑀯𑀕𑁆𑀕

(44.)

93\. “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀫𑁆𑀳𑀺 𑀘𑁂 𑀅𑀬𑀁 𑀲𑀫𑀬𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂’𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀲𑀦𑁆𑀦𑀁𑁇 𑀘𑁂𑀢𑁄𑀧𑀲𑀸𑀤𑀳𑁂𑀢𑀼 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀲𑀢𑁆𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
