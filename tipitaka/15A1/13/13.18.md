# 𑁧𑁩𑁇𑁧𑁮

* 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

(187.)

269\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁄 𑀏𑀯𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀦𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑁇 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀦𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇
