# 𑁧𑁩𑁇𑁩

### 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

(172.)

265\. “𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄? 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀫𑀦𑀼𑀲𑁆𑀲𑁄”𑀢𑀺𑁇
