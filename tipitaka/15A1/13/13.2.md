# 𑁧𑁩𑁇𑁨

### 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

(171.)

264\. “𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑀲𑁆𑀲 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲? 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇
