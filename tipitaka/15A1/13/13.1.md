# 𑁧𑁩𑁇𑁧

* 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

(170.)

263\. “𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄? 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇
