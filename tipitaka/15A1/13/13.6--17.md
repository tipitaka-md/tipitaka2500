# 𑁧𑁩𑁇𑁬--𑁧𑁭

* 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

(175--186.)

268\. “𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀫𑀳𑀢𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀳𑀢𑁄 𑀆𑀮𑁄𑀓𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀳𑀢𑁄 𑀑𑀪𑀸𑀲𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀙𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀦𑀸𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑀲𑁆𑀲 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲? 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸 𑀫𑀳𑀢𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀳𑀢𑁄 𑀆𑀮𑁄𑀓𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀳𑀢𑁄 𑀑𑀪𑀸𑀲𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀙𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁂𑀓𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀦𑀸𑀥𑀸𑀢𑀼𑀧𑀝𑀺𑀯𑁂𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀚𑁆𑀚𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
