# 𑁨𑁇𑁮

### 𑀦𑀻𑀯𑀭𑀡𑀧𑁆𑀧𑀳𑀸𑀦𑀯𑀕𑁆𑀕

(18.)

38. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀯𑀸 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁 𑀧𑀳𑀻𑀬𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑀫𑁆𑀪𑀥𑀸𑀢𑀼 𑀦𑀺𑀓𑁆𑀓𑀫𑀥𑀸𑀢𑀼 𑀧𑀭𑀓𑁆𑀓𑀫𑀥𑀸𑀢𑀼𑁇 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀜𑁆𑀘𑁂𑀯 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀜𑁆𑀘 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁 𑀧𑀳𑀻𑀬𑀢𑀻”𑀢𑀺𑁇
