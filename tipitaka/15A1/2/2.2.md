# 𑁨𑁇𑁨

### 𑀦𑀻𑀯𑀭𑀡𑀧𑁆𑀧𑀳𑀸𑀦𑀯𑀕𑁆𑀕

(12.)

26\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀖𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁𑁇 𑀧𑀝𑀺𑀖𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘𑁂𑀯 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
