# 𑁨𑁇𑁬

* 𑀦𑀻𑀯𑀭𑀡𑀧𑁆𑀧𑀳𑀸𑀦𑀯𑀕𑁆𑀕

(16.)

34\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀻𑀬𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁𑁇 𑀅𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘𑁂𑀯 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀧𑀳𑀻𑀬𑀢𑀻”𑀢𑀺𑁇
