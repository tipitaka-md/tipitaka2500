# 𑁧𑁬𑁇𑁧𑁇𑁨--𑁧𑁦

### 𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑀸𑀍𑀇

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

(297.)

384. “𑀏𑀓𑀥𑀫𑁆𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀏𑀓𑀦𑁆𑀢𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑀥𑀫𑁆𑀫𑁄? 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺…  𑀲𑀻𑀮𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺…  𑀘𑀸𑀕𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺…  𑀤𑁂𑀯𑀢𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺…  𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺…  𑀫𑀭𑀡𑀲𑁆𑀲𑀢𑀺…  𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺…  𑀉𑀧𑀲𑀫𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀏𑀓𑀦𑁆𑀢𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀉𑀧𑀲𑀫𑀸𑀬 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
