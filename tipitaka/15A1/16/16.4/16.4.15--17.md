# 𑁧𑁬𑁇𑁪𑁇𑁧𑁫--𑁧𑁭

### 𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑀸𑀍𑀇

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

(336--338.)

427\. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀆𑀭𑀸𑀫𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀯𑀦𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀪𑀽𑀫𑀺𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑀤𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀁 𑀬𑀤𑀺𑀤𑀁 𑀉𑀓𑁆𑀓𑀽𑀮𑀯𑀺𑀓𑀽𑀮𑀁 𑀦𑀤𑀻𑀯𑀺𑀤𑀼𑀕𑁆𑀕𑀁 𑀔𑀸𑀡𑀼𑀓𑀡𑁆𑀝𑀓𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀩𑁆𑀩𑀢𑀯𑀺𑀲𑀫𑀁𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘𑀼𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀲𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘𑀼𑀢𑀸 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺”𑁇
