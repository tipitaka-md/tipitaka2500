# 𑁧𑁬𑁇𑁪𑁇𑁪𑁨--𑁪𑁪

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

(363--365.)

436\. …  𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀸 𑀘𑀼𑀢𑀸 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀲𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀬𑁂 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀸 𑀘𑀼𑀢𑀸 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺…  𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆”𑀢𑀺𑁇
