# 𑁧𑁬𑁇𑁪𑁇𑁨

### 𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑀸𑀍𑀇

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

(323.)

413\. …  𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀲𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀬𑁂 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺𑁇

414\. …  𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀲𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀬𑁂 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁂𑀲𑀼 𑀫𑀺𑀮𑀓𑁆𑀔𑁂𑀲𑀼𑁇
