# 𑁧𑁬𑁇𑁪𑁇𑁧𑁪

* 𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑀸𑀍𑀇

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀯𑀕𑁆𑀕

(335.)

426\. …  𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀬𑁂 𑀅𑀢𑁆𑀣𑀭𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀭𑀲𑀲𑁆𑀲 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀦𑁄; 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀲𑀢𑁆𑀢𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀬𑁂 𑀅𑀢𑁆𑀣𑀭𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀭𑀲𑀲𑁆𑀲 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑀲𑁆𑀲 𑀦 𑀮𑀸𑀪𑀺𑀦𑁄𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀢𑁆𑀣𑀭𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀭𑀲𑀲𑁆𑀲 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
