# 𑁧𑁬𑁇𑁨𑁇𑁧𑁦

* 𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑀸𑀍𑀇

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

(307.)

395\. “𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀬𑀜𑁆𑀘𑁂𑀯 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀬𑀜𑁆𑀘 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀬𑀜𑁆𑀘 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀬𑀸 𑀘 𑀘𑁂𑀢𑀦𑀸 𑀬𑀸 𑀘 𑀧𑀢𑁆𑀣𑀦𑀸 𑀬𑁄 𑀘 𑀧𑀡𑀺𑀥𑀺 𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀇𑀝𑁆𑀞𑀸𑀬 𑀓𑀦𑁆𑀢𑀸𑀬 𑀫𑀦𑀸𑀧𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀤𑁆𑀤𑀺𑀓𑀸𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀘𑁆𑀙𑀼𑀩𑀻𑀚𑀁 𑀯𑀸 𑀲𑀸𑀮𑀺𑀩𑀻𑀚𑀁 𑀯𑀸 𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀸𑀩𑀻𑀚𑀁 𑀯𑀸 𑀅𑀮𑁆𑀮𑀸𑀬 𑀧𑀣𑀯𑀺𑀬𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀬𑀜𑁆𑀘𑁂𑀯 𑀧𑀣𑀯𑀺𑀭𑀲𑀁 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺 𑀬𑀜𑁆𑀘 𑀆𑀧𑁄𑀭𑀲𑀁 𑀉𑀧𑀸𑀤𑀺𑀬𑀢𑀺 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀫𑀥𑀼𑀭𑀢𑁆𑀢𑀸𑀬 𑀲𑀸𑀢𑀢𑁆𑀢𑀸𑀬 𑀅𑀲𑁂𑀘𑀦𑀓𑀢𑁆𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀩𑀻𑀚𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀤𑁆𑀤𑀓𑀁𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀬𑀜𑁆𑀘𑁂𑀯 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀬𑀜𑁆𑀘 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀬𑀜𑁆𑀘 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀬𑀸 𑀘 𑀘𑁂𑀢𑀦𑀸 𑀬𑀸 𑀘 𑀧𑀢𑁆𑀣𑀦𑀸 𑀬𑁄 𑀘 𑀧𑀡𑀺𑀥𑀺 𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀇𑀝𑁆𑀞𑀸𑀬 𑀓𑀦𑁆𑀢𑀸𑀬 𑀫𑀦𑀸𑀧𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀤𑁆𑀤𑀺𑀓𑀸”𑀢𑀺𑁇
