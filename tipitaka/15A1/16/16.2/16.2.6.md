### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

(303.)

391. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀯𑀟𑁆𑀠𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄𑁇 𑀬𑁄𑀦𑀺𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀧𑀯𑀟𑁆𑀠𑀢𑀻”𑀢𑀺𑁇
