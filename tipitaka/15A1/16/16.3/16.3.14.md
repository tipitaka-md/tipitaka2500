# 𑁧𑁬𑁇𑁩𑁇𑁧𑁪

### 𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑀸𑀍𑀇

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

(321.)

410. “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀫𑁆𑀧𑀺 𑀫𑀼𑀢𑁆𑀢𑀁 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀳𑁄𑀢𑀺…  𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄𑀧𑀺 𑀔𑁂𑀍𑀑 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…  𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄𑀧𑀺 𑀧𑀼𑀩𑁆𑀩𑁄 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…  𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀫𑁆𑀧𑀺 𑀮𑁄𑀳𑀺𑀢𑀁 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀫𑁆𑀧𑀺 𑀪𑀯𑀁 𑀦 𑀯𑀡𑁆𑀡𑁂𑀫𑀺, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀖𑀸𑀢𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺”𑁇
