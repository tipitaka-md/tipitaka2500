# 𑁧𑁬𑁇𑁩𑁇𑁪

### 𑀏𑀓𑀥𑀫𑁆𑀫𑀧𑀸𑀍𑀇

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

(311.)

400\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁄 𑀏𑀯𑀁 𑀩𑀳𑀼𑀚𑀦𑀅𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀩𑀳𑀼𑀚𑀦𑀅𑀲𑀼𑀔𑀸𑀬, 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀓𑁆𑀔𑀮𑀺 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀤𑀻𑀫𑀼𑀔𑁂 𑀔𑀺𑀧𑁆𑀧𑀁 𑀉𑀟𑁆𑀟𑁂𑀬𑁆𑀬 𑀩𑀳𑀽𑀦𑀁 𑀫𑀘𑁆𑀙𑀸𑀦𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀅𑀦𑀬𑀸𑀬 𑀩𑁆𑀬𑀲𑀦𑀸𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀓𑁆𑀔𑀮𑀺 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀔𑀺𑀧𑁆𑀧𑀁 𑀫𑀜𑁆𑀜𑁂 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀩𑀳𑀽𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀅𑀦𑀬𑀸𑀬 𑀩𑁆𑀬𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇
