### 𑀅𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

(141.)

245\. “𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬, 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀩𑀳𑀼𑀜𑁆𑀘 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘𑀺𑀫𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀞𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
