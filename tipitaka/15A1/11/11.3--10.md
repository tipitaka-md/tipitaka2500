# 𑁧𑁧𑁇𑁩--𑁧𑁦

* 𑀅𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

(142--149.)

247\. “𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺; 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬, 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀩𑀳𑀼𑀜𑁆𑀘 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘𑀺𑀫𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀞𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
