(96.)

201\. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁄 𑀏𑀯𑀁 𑀫𑀳𑀢𑁄 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁𑁇 𑀅𑀦𑀼𑀬𑁄𑀕𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁, 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑁄 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀫𑀳𑀢𑁄 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇
