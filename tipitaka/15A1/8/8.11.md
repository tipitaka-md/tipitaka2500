# 𑁮𑁇𑁧𑁧

* 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀸𑀤𑀺𑀯𑀕𑁆𑀕

(81.)

170\. “𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀺𑀓𑀸 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀤𑁆𑀥𑀺 𑀬𑀤𑀺𑀤𑀁 𑀬𑀲𑁄𑀯𑀼𑀤𑁆𑀥𑀺𑁇 𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀤𑁆𑀥𑀻𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀧𑀜𑁆𑀜𑀸𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀯𑀤𑁆𑀥𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
