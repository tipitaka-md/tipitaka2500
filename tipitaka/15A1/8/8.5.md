# 𑁮𑁇𑁫

### 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀸𑀤𑀺𑀯𑀕𑁆𑀕

(75.)

158. “𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀥𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑁂𑀦 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄𑀦𑀺𑀲𑁄𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄𑁇 𑀬𑁄𑀦𑀺𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘𑁂𑀯 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀘 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀦𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
