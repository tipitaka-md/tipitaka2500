# 𑁪𑁇𑁨𑁇𑁪 𑀧𑀢𑀺𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀧𑀢𑀺𑀭𑀽𑀧𑀲𑀼𑀢𑁆𑀢

(150.)

838\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀏𑀓𑀲𑀸𑀮𑀸𑀬𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀺𑀬𑀸 𑀕𑀺𑀳𑀺𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺𑁇

839\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀫𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀬𑀁 𑀔𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀫𑀳𑀢𑀺𑀬𑀸 𑀕𑀺𑀳𑀺𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑁂𑀦 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀁 𑀯𑀺𑀘𑀓𑁆𑀔𑀼𑀓𑀫𑁆𑀫𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

840\. _“𑀦𑁂𑀢𑀁 𑀢𑀯 𑀧𑀢𑀺𑀭𑀽𑀧𑀁,_  
_𑀬𑀤𑀜𑁆𑀜𑀫𑀦𑀼𑀲𑀸𑀲𑀲𑀺;_  
_𑀅𑀦𑀼𑀭𑁄𑀥𑀯𑀺𑀭𑁄𑀥𑁂𑀲𑀼,_  
_𑀫𑀸 𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁄 𑀢𑀤𑀸𑀘𑀭𑀦𑁆”𑀢𑀺𑁇_  


841\. _“𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀬𑀤𑀜𑁆𑀜𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀺;_  
_𑀅𑀦𑀼𑀭𑁄𑀥𑀯𑀺𑀭𑁄𑀥𑁂𑀳𑀺,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀢𑀣𑀸𑀕𑀢𑁄”𑀢𑀺𑁇_  


842\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 “𑀚𑀸𑀦𑀸𑀢𑀺 𑀫𑀁 𑀪𑀕𑀯𑀸, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀫𑀁 𑀲𑀼𑀕𑀢𑁄”𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇
