# 𑁪𑁇𑁧𑁇𑁯 𑀧𑀞𑀫𑀆𑀬𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀆𑀬𑀼𑀲𑀼𑀢𑁆𑀢

(145.)

808\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

809\. “𑀅𑀧𑁆𑀧𑀫𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀆𑀬𑀼𑁇 𑀕𑀫𑀦𑀻𑀬𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄, 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁𑁇 𑀦𑀢𑁆𑀣𑀺 𑀚𑀸𑀢𑀲𑁆𑀲 𑀅𑀫𑀭𑀡𑀁𑁇 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀢𑀺, 𑀲𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀅𑀧𑁆𑀧𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄”𑀢𑀺𑁇

810\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

811\. _“𑀤𑀻𑀖𑀫𑀸𑀬𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀦 𑀦𑀁 𑀳𑀻𑀍𑀏 𑀲𑀼𑀧𑁄𑀭𑀺𑀲𑁄;_  
_𑀘𑀭𑁂𑀬𑁆𑀬 𑀔𑀻𑀭𑀫𑀢𑁆𑀢𑁄𑀯,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑀘𑁆𑀘𑀼𑀲𑁆𑀲 𑀆𑀕𑀫𑁄”𑀢𑀺𑁇_  


812\. _“𑀅𑀧𑁆𑀧𑀫𑀸𑀬𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀳𑀻𑀍𑀏𑀬𑁆𑀬 𑀦𑀁 𑀲𑀼𑀧𑁄𑀭𑀺𑀲𑁄;_  
_𑀘𑀭𑁂𑀬𑁆𑀬𑀸𑀤𑀺𑀢𑁆𑀢𑀲𑀻𑀲𑁄𑀯,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑀘𑁆𑀘𑀼𑀲𑁆𑀲 𑀦𑀸𑀕𑀫𑁄”𑀢𑀺𑁇_  


813\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄…𑀧𑁂…  𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇
