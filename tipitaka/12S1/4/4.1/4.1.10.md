# 𑁪𑁇𑁧𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀆𑀬𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀆𑀬𑀼𑀲𑀼𑀢𑁆𑀢

(146.)

814\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀏𑀢𑀤𑀯𑁄𑀘—

815\. “𑀅𑀧𑁆𑀧𑀫𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀆𑀬𑀼𑁇 𑀕𑀫𑀦𑀻𑀬𑁄 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄, 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀲𑀮𑀁, 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁𑁇 𑀦𑀢𑁆𑀣𑀺 𑀚𑀸𑀢𑀲𑁆𑀲 𑀅𑀫𑀭𑀡𑀁𑁇 𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀢𑀺, 𑀲𑁄 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀅𑀧𑁆𑀧𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄”𑀢𑀺𑁇

816\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

817\. _“𑀦𑀸𑀘𑁆𑀘𑀬𑀦𑁆𑀢𑀺 𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀸,_  
_𑀚𑀻𑀯𑀺𑀢𑀁 𑀦𑀽𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀆𑀬𑀼 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑀸𑀬𑀢𑀺 𑀫𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀦𑁂𑀫𑀻𑀯 𑀭𑀣𑀓𑀼𑀩𑁆𑀩𑀭𑀦𑁆”𑀢𑀺𑁇_  


818\. _“𑀅𑀘𑁆𑀘𑀬𑀦𑁆𑀢𑀺 𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀸,_  
_𑀚𑀻𑀯𑀺𑀢𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀆𑀬𑀼 𑀔𑀻𑀬𑀢𑀺 𑀫𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀓𑀼𑀦𑁆𑀦𑀤𑀻𑀦𑀁𑀯 𑀑𑀤𑀓𑀦𑁆”𑀢𑀺𑁇_  


819\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 “𑀚𑀸𑀦𑀸𑀢𑀺 𑀫𑀁 𑀪𑀕𑀯𑀸, 𑀚𑀸𑀦𑀸𑀢𑀺 𑀫𑀁 𑀲𑀼𑀕𑀢𑁄”𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇

---

820\. 𑀧𑀞𑀫𑁄 𑀯𑀕𑁆𑀕𑁄𑁇



---

821\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



822\. _𑀢𑀧𑁄𑀓𑀫𑁆𑀫𑀜𑁆𑀘 𑀦𑀸𑀕𑁄 𑀘,_  
_𑀲𑀼𑀪𑀁 𑀧𑀸𑀲𑁂𑀦 𑀢𑁂 𑀤𑀼𑀯𑁂;_  
_𑀲𑀧𑁆𑀧𑁄 𑀲𑀼𑀧𑀢𑀺 𑀦𑀦𑁆𑀤𑀦𑀁,_  
_𑀆𑀬𑀼𑀦𑀸 𑀅𑀧𑀭𑁂 𑀤𑀼𑀯𑁂𑀢𑀺𑁇_
