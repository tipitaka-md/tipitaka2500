# 𑁪𑁇𑁧𑁇𑁭 𑀲𑀼𑀧𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀸𑀭𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀧𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(143.)

800. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀳𑀼𑀤𑁂𑀯𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀻𑀳𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺 𑀧𑀸𑀤𑁂 𑀧𑀸𑀤𑀁 𑀅𑀘𑁆𑀘𑀸𑀥𑀸𑀬 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

801. _“𑀓𑀺𑀁 𑀲𑁄𑀧𑁆𑀧𑀲𑀺 𑀓𑀺𑀁 𑀦𑀼 𑀲𑁄𑀧𑁆𑀧𑀲𑀺,_  
_𑀓𑀺𑀫𑀺𑀤𑀁 𑀲𑁄𑀧𑁆𑀧𑀲𑀺 𑀤𑀼𑀩𑁆𑀪𑀕𑁄 𑀯𑀺𑀬;_  
_𑀲𑀼𑀜𑁆𑀜𑀫𑀕𑀸𑀭𑀦𑁆𑀢𑀺 𑀲𑁄𑀧𑁆𑀧𑀲𑀺,_  
_𑀓𑀺𑀫𑀺𑀤𑀁 𑀲𑁄𑀧𑁆𑀧𑀲𑀺 𑀲𑀽𑀭𑀺𑀬𑁂 𑀉𑀕𑁆𑀕𑀢𑁂”𑀢𑀺𑁇_  


802. _“𑀬𑀲𑁆𑀲 𑀚𑀸𑀮𑀺𑀦𑀻 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀢𑀡𑁆𑀳𑀸 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺 𑀦𑁂𑀢𑀯𑁂;_  
_𑀲𑀩𑁆𑀩𑀽𑀧𑀥𑀺𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑁄𑀧𑁆𑀧𑀢𑀺 𑀓𑀺𑀁 𑀢𑀯𑁂𑀢𑁆𑀣 𑀫𑀸𑀭𑀸”𑀢𑀺𑁇_  


803. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸…𑀧𑁂…  𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇
