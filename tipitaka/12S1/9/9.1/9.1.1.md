# 𑁯𑁇𑁧𑁇𑁧 𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀯𑀕𑁆𑀕

### 𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀢𑁆𑀢

(221.)

1378\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀕𑁂𑀳𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1379\. _“𑀯𑀺𑀯𑁂𑀓𑀓𑀸𑀫𑁄𑀲𑀺 𑀯𑀦𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄,_  
_𑀅𑀣 𑀢𑁂 𑀫𑀦𑁄 𑀦𑀺𑀘𑁆𑀙𑀭𑀢𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸;_  
_𑀚𑀦𑁄 𑀚𑀦𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀦𑀬𑀲𑁆𑀲𑀼 𑀙𑀦𑁆𑀤𑀁,_  
_𑀢𑀢𑁄 𑀲𑀼𑀔𑀻 𑀳𑁄𑀳𑀺𑀲𑀺 𑀯𑀻𑀢𑀭𑀸𑀕𑁄𑁇_  


1380\. _𑀅𑀭𑀢𑀺𑀁 𑀧𑀚𑀳𑀸𑀲𑀺 𑀲𑀢𑁄,_  
_𑀪𑀯𑀸𑀲𑀺 𑀲𑀢𑀁 𑀢𑀁 𑀲𑀸𑀭𑀬𑀸𑀫𑀲𑁂;_  
_𑀧𑀸𑀢𑀸𑀮𑀭𑀚𑁄 𑀳𑀺 𑀤𑀼𑀢𑁆𑀢𑀭𑁄,_  
_𑀫𑀸 𑀢𑀁 𑀓𑀸𑀫𑀭𑀚𑁄 𑀅𑀯𑀸𑀳𑀭𑀺𑁇_  


1381\. _𑀲𑀓𑀼𑀡𑁄 𑀬𑀣𑀸 𑀧𑀁𑀲𑀼𑀓𑀼𑀦𑁆𑀣𑀺𑀢𑁄,_  
_𑀯𑀺𑀥𑀼𑀦𑀁 𑀧𑀸𑀢𑀬𑀢𑀺 𑀲𑀺𑀢𑀁 𑀭𑀚𑀁;_  
_𑀏𑀯𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀥𑀸𑀦𑀯𑀸 𑀲𑀢𑀺𑀫𑀸,_  
_𑀯𑀺𑀥𑀼𑀦𑀁 𑀧𑀸𑀢𑀬𑀢𑀺 𑀲𑀺𑀢𑀁 𑀭𑀚𑀦𑁆”𑀢𑀺𑁇_  


1382\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀬 𑀤𑁂𑀯𑀢𑀸𑀬 𑀲𑀁𑀯𑁂𑀚𑀺𑀢𑁄 𑀲𑀁𑀯𑁂𑀕𑀫𑀸𑀧𑀸𑀤𑀻𑀢𑀺𑁇
