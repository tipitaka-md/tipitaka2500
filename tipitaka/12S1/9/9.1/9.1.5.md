# 𑁯𑁇𑁧𑁇𑁫 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀯𑀕𑁆𑀕

### 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(225.)

1399. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀕𑀺𑀳𑀺𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1400. _“𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀳𑀦𑀁 𑀧𑀲𑀓𑁆𑀓𑀺𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀳𑀤𑀬𑀲𑁆𑀫𑀺𑀁 𑀑𑀧𑀺𑀬;_  
_𑀛𑀸𑀬 𑀕𑁄𑀢𑀫 𑀫𑀸 𑀧𑀫𑀸𑀤𑁄,_  
_𑀓𑀺𑀁 𑀢𑁂 𑀩𑀺𑀍𑀇𑀩𑀺𑀍𑀇𑀓𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


1401. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀸𑀬 𑀤𑁂𑀯𑀢𑀸𑀬 𑀲𑀁𑀯𑁂𑀚𑀺𑀢𑁄 𑀲𑀁𑀯𑁂𑀕𑀫𑀸𑀧𑀸𑀤𑀻𑀢𑀺𑁇
