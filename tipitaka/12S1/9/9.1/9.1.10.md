# 𑁯𑁇𑁧𑁇𑁧𑁦 𑀲𑀚𑁆𑀛𑀸𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀚𑁆𑀛𑀸𑀬𑀲𑀼𑀢𑁆𑀢

(230.)

1421\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀁 𑀲𑀼𑀤𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀲𑀚𑁆𑀛𑀸𑀬𑀩𑀳𑀼𑀮𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑁄 𑀲𑀗𑁆𑀓𑀲𑀸𑀬𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑀁 𑀅𑀲𑀼𑀡𑀦𑁆𑀢𑀻 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1422\. _“𑀓𑀲𑁆𑀫𑀸 𑀢𑀼𑀯𑀁 𑀥𑀫𑁆𑀫𑀧𑀤𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀦𑀸𑀥𑀻𑀬𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀳𑀺 𑀲𑀁𑀯𑀲𑀦𑁆𑀢𑁄;_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀥𑀫𑁆𑀫𑀁 𑀮𑀪𑀢𑀺𑀧𑁆𑀧𑀲𑀸𑀤𑀁,_  
_𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀮𑀪𑀢𑀺𑀧𑁆𑀧𑀲𑀁𑀲𑀦𑁆”𑀢𑀺𑁇_  


1423\. _“𑀅𑀳𑀼 𑀧𑀼𑀭𑁂 𑀥𑀫𑁆𑀫𑀧𑀤𑁂𑀲𑀼 𑀙𑀦𑁆𑀤𑁄,_  
_𑀬𑀸𑀯 𑀯𑀺𑀭𑀸𑀕𑁂𑀦 𑀲𑀫𑀸𑀕𑀫𑀺𑀫𑁆𑀳;_  
_𑀬𑀢𑁄 𑀯𑀺𑀭𑀸𑀕𑁂𑀦 𑀲𑀫𑀸𑀕𑀫𑀺𑀫𑁆𑀳,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀺𑀝𑁆𑀞𑀁𑀯 𑀲𑀼𑀢𑀁 𑀫𑀼𑀢𑀁 𑀯𑀸;_  
_𑀅𑀜𑁆𑀜𑀸𑀬 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀦𑀫𑀸𑀳𑀼 𑀲𑀦𑁆𑀢𑁄”𑀢𑀺𑁇_
