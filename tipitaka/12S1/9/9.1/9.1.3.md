# 𑁯𑁇𑁧𑁇𑁩 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀯𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑀦𑀯𑀕𑁆𑀕

* 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(223.)

1390\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑁄 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑁄 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀙𑁂𑀢𑀁 𑀑𑀯𑀤𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1391\. _“𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑀘𑀭𑀁 𑀙𑁂𑀢𑀁,_  
_𑀅𑀧𑁆𑀧𑀧𑀜𑁆𑀜𑀁 𑀅𑀘𑁂𑀢𑀲𑀁;_  
_𑀅𑀓𑀸𑀮𑁂 𑀑𑀯𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀫𑀦𑁆𑀤𑁄𑀯 𑀧𑀝𑀺𑀪𑀸𑀢𑀺 𑀫𑀁𑁇_  


1392\. _𑀲𑀼𑀡𑀸𑀢𑀺 𑀦 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀆𑀮𑁄𑀓𑁂𑀢𑀺 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀫𑀺𑀁 𑀪𑀜𑁆𑀜𑀫𑀸𑀦𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀢𑁆𑀣𑀁 𑀩𑀸𑀮𑁄 𑀦 𑀩𑀼𑀚𑁆𑀛𑀢𑀺𑁇_  


1393\. _𑀲𑀘𑁂𑀧𑀺 𑀤𑀲 𑀧𑀚𑁆𑀚𑁄𑀢𑁂,_  
_𑀥𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀓𑀲𑁆𑀲𑀧;_  
_𑀦𑁂𑀯 𑀤𑀓𑁆𑀔𑀢𑀺 𑀭𑀽𑀧𑀸𑀦𑀺,_  
_𑀘𑀓𑁆𑀔𑀼 𑀳𑀺𑀲𑁆𑀲 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


1394\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑁄 𑀢𑀸𑀬 𑀤𑁂𑀯𑀢𑀸𑀬 𑀲𑀁𑀯𑁂𑀚𑀺𑀢𑁄 𑀲𑀁𑀯𑁂𑀕𑀫𑀸𑀧𑀸𑀤𑀻𑀢𑀺𑁇
