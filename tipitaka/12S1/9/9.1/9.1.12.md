# 𑁯𑁇𑁧𑁇𑁧𑁨 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀯𑀕𑁆𑀕

### 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(232.)

1428\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1429\. _“𑀞𑀺𑀢𑁂 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀓𑀸𑀮𑁂,_  
_𑀲𑀦𑁆𑀦𑀺𑀲𑀻𑀯𑁂𑀲𑀼 𑀧𑀓𑁆𑀔𑀺𑀲𑀼;_  
_𑀲𑀡𑀢𑁂𑀯 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑀁,_  
_𑀢𑀁 𑀪𑀬𑀁 𑀧𑀝𑀺𑀪𑀸𑀢𑀺 𑀫𑀁𑁇_  


1430\. _𑀞𑀺𑀢𑁂 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀓𑀸𑀮𑁂,_  
_𑀲𑀦𑁆𑀦𑀺𑀲𑀻𑀯𑁂𑀲𑀼 𑀧𑀓𑁆𑀔𑀺𑀲𑀼;_  
_𑀲𑀡𑀢𑁂𑀯 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑀁,_  
_𑀲𑀸 𑀭𑀢𑀺 𑀧𑀝𑀺𑀪𑀸𑀢𑀺 𑀫𑀦𑁆”𑀢𑀺𑁇_
