# 𑁯𑁇𑁧𑁇𑁧𑁪 𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀯𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀯𑀦𑀯𑀕𑁆𑀕

* 𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀦𑀲𑀼𑀢𑁆𑀢

(234.)

1437\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀼𑀫𑀁 𑀉𑀧𑀲𑀺𑀗𑁆𑀖𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1438\. _“𑀬𑀫𑁂𑀢𑀁 𑀯𑀸𑀭𑀺𑀚𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀉𑀧𑀲𑀺𑀗𑁆𑀖𑀲𑀺;_  
_𑀏𑀓𑀗𑁆𑀕𑀫𑁂𑀢𑀁 𑀣𑁂𑀬𑁆𑀬𑀸𑀦𑀁,_  
_𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀦𑁄𑀲𑀺 𑀫𑀸𑀭𑀺𑀲𑀸”𑀢𑀺𑁇_  


1439\. _“𑀦 𑀳𑀭𑀸𑀫𑀺 𑀦 𑀪𑀜𑁆𑀚𑀸𑀫𑀺,_  
_𑀆𑀭𑀸 𑀲𑀺𑀗𑁆𑀖𑀸𑀫𑀺 𑀯𑀸𑀭𑀺𑀚𑀁;_  
_𑀅𑀣 𑀓𑁂𑀦 𑀦𑀼 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀕𑀦𑁆𑀥𑀢𑁆𑀣𑁂𑀦𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1440\. _𑀬𑁆𑀯𑀸𑀬𑀁 𑀪𑀺𑀲𑀸𑀦𑀺 𑀔𑀦𑀢𑀺,_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀸𑀦𑀺 𑀪𑀜𑁆𑀚𑀢𑀺;_  
_𑀏𑀯𑀁 𑀆𑀓𑀺𑀡𑁆𑀡𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄,_  
_𑀓𑀲𑁆𑀫𑀸 𑀏𑀲𑁄 𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


1441\. _“𑀆𑀓𑀺𑀡𑁆𑀡𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀥𑀸𑀢𑀺𑀘𑁂𑀮𑀁𑀯 𑀫𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑁂 𑀯𑀘𑀦𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀢𑁆𑀯𑀜𑁆𑀘𑀸𑀭𑀳𑀸𑀫𑀺 𑀯𑀢𑁆𑀢𑀯𑁂𑁇_  


1442\. _𑀅𑀦𑀗𑁆𑀕𑀡𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑀼𑀘𑀺𑀕𑀯𑁂𑀲𑀺𑀦𑁄;_  
_𑀯𑀸𑀮𑀕𑁆𑀕𑀫𑀢𑁆𑀢𑀁 𑀧𑀸𑀧𑀲𑁆𑀲,_  
_𑀅𑀩𑁆𑀪𑀸𑀫𑀢𑁆𑀢𑀁𑀯 𑀔𑀸𑀬𑀢𑀻”𑀢𑀺𑁇_  


1443\. _“𑀅𑀤𑁆𑀥𑀸 𑀫𑀁 𑀬𑀓𑁆𑀔 𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀅𑀣𑁄 𑀫𑁂 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀲𑀺;_  
_𑀧𑀼𑀦𑀧𑀺 𑀬𑀓𑁆𑀔 𑀯𑀚𑁆𑀚𑀸𑀲𑀺,_  
_𑀬𑀤𑀸 𑀧𑀲𑁆𑀲𑀲𑀺 𑀏𑀤𑀺𑀲𑀦𑁆”𑀢𑀺𑁇_  


1444\. _“𑀦𑁂𑀯 𑀢𑀁 𑀉𑀧𑀚𑀻𑀯𑀸𑀫,_  
_𑀦𑀧𑀺 𑀢𑁂 𑀪𑀢𑀓𑀸𑀫𑁆𑀳𑀲𑁂;_  
_𑀢𑁆𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼 𑀚𑀸𑀦𑁂𑀬𑁆𑀬,_  
_𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀲𑀼𑀕𑁆𑀕𑀢𑀺𑀦𑁆”𑀢𑀺𑁇_  


1445\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀬 𑀤𑁂𑀯𑀢𑀸𑀬 𑀲𑀁𑀯𑁂𑀚𑀺𑀢𑁄 𑀲𑀁𑀯𑁂𑀕𑀫𑀸𑀧𑀸𑀤𑀻𑀢𑀺𑁇

1447\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1448\. _𑀯𑀺𑀯𑁂𑀓𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘,_  
_𑀓𑀲𑁆𑀲𑀧𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸;_  
_𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀘,_  
_𑀦𑀸𑀕𑀤𑀢𑁆𑀢𑀜𑁆𑀘 𑀓𑀼𑀮𑀖𑀭𑀡𑀻𑁇_  


1449\. _𑀯𑀚𑁆𑀚𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀯𑁂𑀲𑀸𑀮𑀻,_  
_𑀲𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀅𑀬𑁄𑀦𑀺𑀲𑁄;_  
_𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀸𑀮𑀫𑁆𑀳𑀺 𑀧𑀸𑀓𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬-_  
_𑀧𑀤𑀼𑀫𑀧𑀼𑀧𑁆𑀨𑁂𑀦 𑀘𑀼𑀤𑁆𑀤𑀲 𑀪𑀯𑁂𑀢𑀺𑁇_
