# 𑁯𑁇𑁧𑁇𑁮 𑀓𑀼𑀮𑀖𑀭𑀡𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀯𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀮𑀖𑀭𑀡𑀻𑀲𑀼𑀢𑁆𑀢

(228.)

1412. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑁂 𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀍𑀳𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀬𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑁂 𑀓𑀼𑀮𑀖𑀭𑀡𑀻, 𑀢𑀲𑁆𑀲𑀸 𑀯𑀡𑁆𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1413. _“𑀦𑀤𑀻𑀢𑀻𑀭𑁂𑀲𑀼 𑀲𑀡𑁆𑀞𑀸𑀦𑁂,_  
_𑀲𑀪𑀸𑀲𑀼 𑀭𑀣𑀺𑀬𑀸𑀲𑀼 𑀘;_  
_𑀚𑀦𑀸 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺,_  
_𑀫𑀜𑁆𑀘 𑀢𑀜𑁆𑀘 𑀓𑀺𑀫𑀦𑁆𑀢𑀭𑀦𑁆”𑀢𑀺𑁇_  


1414. _“𑀩𑀳𑀽𑀳𑀺 𑀲𑀤𑁆𑀤𑀸 𑀧𑀘𑁆𑀘𑀽𑀳𑀸,_  
_𑀔𑀫𑀺𑀢𑀩𑁆𑀩𑀸 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀸;_  
_𑀦 𑀢𑁂𑀦 𑀫𑀗𑁆𑀓𑀼 𑀳𑁄𑀢𑀩𑁆𑀩𑀁,_  
_𑀦 𑀳𑀺 𑀢𑁂𑀦 𑀓𑀺𑀮𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


1415. _𑀬𑁄 𑀘 𑀲𑀤𑁆𑀤𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀲𑀻,_  
_𑀯𑀦𑁂 𑀯𑀸𑀢𑀫𑀺𑀕𑁄 𑀬𑀣𑀸;_  
_𑀮𑀳𑀼𑀘𑀺𑀢𑁆𑀢𑁄𑀢𑀺 𑀢𑀁 𑀆𑀳𑀼,_  
_𑀦𑀸𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀢𑁂 𑀯𑀢𑀦𑁆”𑀢𑀺𑁇_
