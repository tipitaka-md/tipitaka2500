# 𑁧𑁦𑁇𑁧𑁇𑁬 𑀧𑀺𑀬𑀗𑁆𑀓𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀬𑀓𑁆𑀔𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀬𑀗𑁆𑀓𑀭𑀲𑀼𑀢𑁆𑀢

(240.)

1479\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀥𑀫𑁆𑀫𑀧𑀤𑀸𑀦𑀺 𑀪𑀸𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀺𑀬𑀗𑁆𑀓𑀭𑀫𑀸𑀢𑀸 𑀬𑀓𑁆𑀔𑀺𑀦𑀻 𑀧𑀼𑀢𑁆𑀢𑀓𑀁 𑀏𑀯𑀁 𑀢𑁄𑀲𑁂𑀲𑀺—

1480\. _“𑀫𑀸 𑀲𑀤𑁆𑀤𑀁 𑀓𑀭𑀺 𑀧𑀺𑀬𑀗𑁆𑀓𑀭,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀧𑀤𑀸𑀦𑀺 𑀪𑀸𑀲𑀢𑀺;_  
_𑀅𑀧𑀺 𑀘 𑀥𑀫𑁆𑀫𑀧𑀤𑀁 𑀯𑀺𑀚𑀸𑀦𑀺𑀬,_  
_𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑁂𑀫 𑀳𑀺𑀢𑀸𑀬 𑀦𑁄 𑀲𑀺𑀬𑀸𑁇_  


1481\. _𑀧𑀸𑀡𑁂𑀲𑀼 𑀘 𑀲𑀁𑀬𑀫𑀸𑀫𑀲𑁂,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀦 𑀪𑀡𑀸𑀫𑀲𑁂;_  
_𑀲𑀺𑀓𑁆𑀔𑁂𑀫 𑀲𑀼𑀲𑀻𑀮𑁆𑀬𑀫𑀢𑁆𑀢𑀦𑁄,_  
_𑀅𑀧𑀺 𑀫𑀼𑀘𑁆𑀘𑁂𑀫 𑀧𑀺𑀲𑀸𑀘𑀬𑁄𑀦𑀺𑀬𑀸”𑀢𑀺𑁇_
