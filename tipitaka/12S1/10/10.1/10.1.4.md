# 𑁧𑁦𑁇𑁧𑁇𑁪 𑀫𑀡𑀺𑀪𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀬𑀓𑁆𑀔𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑀓𑀯𑀕𑁆𑀕

* 𑀫𑀡𑀺𑀪𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(238.)

1465\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀫𑀕𑀥𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀡𑀺𑀫𑀸𑀮𑀺𑀓𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀫𑀡𑀺𑀪𑀤𑁆𑀤𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀪𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀡𑀺𑀪𑀤𑁆𑀤𑁄 𑀬𑀓𑁆𑀔𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1466\. _“𑀲𑀢𑀻𑀫𑀢𑁄 𑀲𑀤𑀸 𑀪𑀤𑁆𑀤𑀁,_  
_𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀫𑁂𑀥𑀢𑀺;_  
_𑀲𑀢𑀻𑀫𑀢𑁄 𑀲𑀼𑀯𑁂 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀯𑁂𑀭𑀸 𑀘 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


1467\. _“𑀲𑀢𑀻𑀫𑀢𑁄 𑀲𑀤𑀸 𑀪𑀤𑁆𑀤𑀁,_  
_𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀫𑁂𑀥𑀢𑀺;_  
_𑀲𑀢𑀻𑀫𑀢𑁄 𑀲𑀼𑀯𑁂 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀯𑁂𑀭𑀸 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1468\. _𑀬𑀲𑁆𑀲 𑀲𑀩𑁆𑀩𑀫𑀳𑁄𑀭𑀢𑁆𑀢𑀁,_  
_𑀅𑀳𑀺𑀁𑀲𑀸𑀬 𑀭𑀢𑁄 𑀫𑀦𑁄;_  
_𑀫𑁂𑀢𑁆𑀢𑀁𑀲𑁄 𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑁂𑀲𑀼,_  
_𑀯𑁂𑀭𑀁 𑀢𑀲𑁆𑀲 𑀦 𑀓𑁂𑀦𑀘𑀻”𑀢𑀺𑁇_
