# 𑁧𑁦𑁇𑁧𑁇𑁧 𑀇𑀦𑁆𑀤𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀬𑀓𑁆𑀔𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀇𑀦𑁆𑀤𑀓𑀯𑀕𑁆𑀕

* 𑀇𑀦𑁆𑀤𑀓𑀲𑀼𑀢𑁆𑀢

(235.)

1451\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀦𑁆𑀤𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂, 𑀇𑀦𑁆𑀤𑀓𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀪𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀇𑀦𑁆𑀤𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1452\. _“𑀭𑀽𑀧𑀁 𑀦 𑀚𑀻𑀯𑀦𑁆𑀢𑀺 𑀯𑀤𑀦𑁆𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀸,_  
_𑀓𑀣𑀁 𑀦𑁆𑀯𑀬𑀁 𑀯𑀺𑀦𑁆𑀤𑀢𑀺𑀫𑀁 𑀲𑀭𑀻𑀭𑀁;_  
_𑀓𑀼𑀢𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀻𑀬𑀓𑀧𑀺𑀡𑁆𑀟𑀫𑁂𑀢𑀺,_  
_𑀓𑀣𑀁 𑀦𑁆𑀯𑀬𑀁 𑀲𑀚𑁆𑀚𑀢𑀺 𑀕𑀩𑁆𑀪𑀭𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇_  


1453\. _“𑀧𑀞𑀫𑀁 𑀓𑀮𑀮𑀁 𑀳𑁄𑀢𑀺,_  
_𑀓𑀮𑀮𑀸 𑀳𑁄𑀢𑀺 𑀅𑀩𑁆𑀩𑀼𑀤𑀁;_  
_𑀅𑀩𑁆𑀩𑀼𑀤𑀸 𑀚𑀸𑀬𑀢𑁂 𑀧𑁂𑀲𑀺,_  
_𑀧𑁂𑀲𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀻 𑀖𑀦𑁄;_  
_𑀖𑀦𑀸 𑀧𑀲𑀸𑀔𑀸 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀓𑁂𑀲𑀸 𑀮𑁄𑀫𑀸 𑀦𑀔𑀸𑀧𑀺 𑀘𑁇_  


1454\. _𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀪𑀼𑀜𑁆𑀚𑀢𑀻 𑀫𑀸𑀢𑀸,_  
_𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀁;_  
_𑀢𑁂𑀦 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀬𑀸𑀧𑁂𑀢𑀺,_  
_𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑁄 𑀦𑀭𑁄”𑀢𑀺𑁇_
