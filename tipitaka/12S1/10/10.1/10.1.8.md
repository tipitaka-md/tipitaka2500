# 𑁧𑁦𑁇𑁧𑁇𑁮 𑀲𑀼𑀤𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀬𑀓𑁆𑀔𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀤𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(242.)

1492. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀻𑀢𑀯𑀦𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀭𑀸𑀚𑀕𑀳𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺—  “𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀺𑀭 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 𑀢𑀸𑀯𑀤𑁂𑀯 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣𑀲𑁆𑀲 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄 𑀅𑀚𑁆𑀚 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀲𑁆𑀯𑁂 𑀤𑀸𑀦𑀸𑀳𑀁 𑀓𑀸𑀮𑁂𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑁇 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀤𑀁 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺 𑀧𑀪𑀸𑀢𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀬𑁂𑀦 𑀲𑀺𑀯𑀣𑀺𑀓𑀤𑁆𑀯𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑁆𑀯𑀸𑀭𑀁 𑀯𑀺𑀯𑀭𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀕𑀭𑀫𑁆𑀳𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀮𑁄𑀓𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺, 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀢𑀢𑁄𑀯 𑀧𑀼𑀦 𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀺𑀯𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀺—

1493. _“𑀲𑀢𑀁 𑀳𑀢𑁆𑀣𑀻 𑀲𑀢𑀁 𑀅𑀲𑁆𑀲𑀸,_  
_𑀲𑀢𑀁 𑀅𑀲𑁆𑀲𑀢𑀭𑀻𑀭𑀣𑀸;_  
_𑀲𑀢𑀁 𑀓𑀜𑁆𑀜𑀸𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀫𑀼𑀓𑁆𑀓𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀸;_  
_𑀏𑀓𑀲𑁆𑀲 𑀧𑀤𑀯𑀻𑀢𑀺𑀳𑀸𑀭𑀲𑁆𑀲,_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


1494. _𑀅𑀪𑀺𑀓𑁆𑀓𑀫 𑀕𑀳𑀧𑀢𑀺,_  
_𑀅𑀪𑀺𑀓𑁆𑀓𑀫 𑀕𑀳𑀧𑀢𑀺;_  
_𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑀁 𑀢𑁂 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑀦𑁆”𑀢𑀺𑁇_  


1495. 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺, 𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀆𑀮𑁄𑀓𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺, 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀢𑀢𑁄𑀯 𑀧𑀼𑀦 𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀺𑀯𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀺—

1496. _“𑀲𑀢𑀁 𑀳𑀢𑁆𑀣𑀻 𑀲𑀢𑀁 𑀅𑀲𑁆𑀲𑀸,_  
_…𑀧𑁂…_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


1497. _𑀅𑀪𑀺𑀓𑁆𑀓𑀫 𑀕𑀳𑀧𑀢𑀺,_  
_𑀅𑀪𑀺𑀓𑁆𑀓𑀫 𑀕𑀳𑀧𑀢𑀺;_  
_𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑀁 𑀢𑁂 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑀦𑁆”𑀢𑀺𑁇_  


1498. 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺, 𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀆𑀮𑁄𑀓𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺, 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀢𑀢𑁄𑀯 𑀧𑀼𑀦 𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀺𑀯𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀲𑀤𑁆𑀤𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑀺—

1499. _“𑀲𑀢𑀁 𑀳𑀢𑁆𑀣𑀻 𑀲𑀢𑀁 𑀅𑀲𑁆𑀲𑀸,_  
_…𑀧𑁂…_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


1500. _𑀅𑀪𑀺𑀓𑁆𑀓𑀫 𑀕𑀳𑀧𑀢𑀺,_  
_𑀅𑀪𑀺𑀓𑁆𑀓𑀫 𑀕𑀳𑀧𑀢𑀺;_  
_𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑀁 𑀢𑁂 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀦𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑀦𑁆”𑀢𑀺𑁇_  


1501. 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺, 𑀆𑀮𑁄𑀓𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺, 𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀬𑁂𑀦 𑀲𑀻𑀢𑀯𑀦𑀁 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇

1502. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀘𑀗𑁆𑀓𑀫𑀸 𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺, 𑀲𑀼𑀤𑀢𑁆𑀢𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺, 𑀦𑀸𑀫𑁂𑀦 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀆𑀮𑀧𑀢𑀻𑀢𑀺, 𑀳𑀝𑁆𑀞𑁄 𑀉𑀤𑀕𑁆𑀕𑁄 𑀢𑀢𑁆𑀣𑁂𑀯 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀼𑀔𑀫𑀲𑀬𑀺𑀢𑁆𑀣𑀸”𑀢𑀺?

1503. _“𑀲𑀩𑁆𑀩𑀤𑀸 𑀯𑁂 𑀲𑀼𑀔𑀁 𑀲𑁂𑀢𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀬𑁄 𑀦 𑀮𑀺𑀫𑁆𑀧𑀢𑀺 𑀓𑀸𑀫𑁂𑀲𑀼,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑁇_  


1504. _𑀲𑀩𑁆𑀩𑀸 𑀆𑀲𑀢𑁆𑀢𑀺𑀬𑁄 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀳𑀤𑀬𑁂 𑀤𑀭𑀁;_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀁 𑀲𑁂𑀢𑀺,_  
_𑀲𑀦𑁆𑀢𑀺𑀁 𑀧𑀧𑁆𑀧𑀼𑀬𑁆𑀬 𑀘𑁂𑀢𑀲𑀸”𑀢𑀺𑁇_
