# 𑁧𑁦𑁇𑁧𑁇𑁫 𑀲𑀸𑀦𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀬𑀓𑁆𑀔𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀦𑀼𑀲𑀼𑀢𑁆𑀢

(239.)

1469. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬 𑀲𑀸𑀦𑀼 𑀦𑀸𑀫 𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑀓𑁆𑀔𑁂𑀦 𑀕𑀳𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀫𑀸𑀦𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

1470. _“𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀁 𑀧𑀜𑁆𑀘𑀤𑀲𑀺𑀁,_  
_𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀫𑀻;_  
_𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀧𑀓𑁆𑀔𑀜𑁆𑀘,_  
_𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑀁𑁇_  


1471. _𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀦𑁆𑀢𑀺,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀺 𑀬𑁂;_  
_𑀦 𑀢𑁂𑀳𑀺 𑀬𑀓𑁆𑀔𑀸 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺,_  
_𑀇𑀢𑀺 𑀫𑁂 𑀅𑀭𑀳𑀢𑀁 𑀲𑀼𑀢𑀁;_  
_𑀲𑀸 𑀤𑀸𑀦𑀺 𑀅𑀚𑁆𑀚 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀓𑁆𑀔𑀸 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺 𑀲𑀸𑀦𑀼𑀦𑀸”𑀢𑀺𑁇_  


1472. _“𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀁 𑀧𑀜𑁆𑀘𑀤𑀲𑀺𑀁,_  
_𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀫𑀻;_  
_𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀧𑀓𑁆𑀔𑀜𑁆𑀘,_  
_𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑀁;_  
_𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑀦𑁆𑀢𑀺,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀺 𑀬𑁂𑁇_  


1473. _𑀦 𑀢𑁂𑀳𑀺 𑀬𑀓𑁆𑀔𑀸 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺,_  
_𑀲𑀸𑀳𑀼 𑀢𑁂 𑀅𑀭𑀳𑀢𑀁 𑀲𑀼𑀢𑀁;_  
_𑀲𑀸𑀦𑀼𑀁 𑀧𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀚𑁆𑀚𑀸𑀲𑀺,_  
_𑀬𑀓𑁆𑀔𑀸𑀦𑀁 𑀯𑀘𑀦𑀁 𑀇𑀤𑀁;_  
_𑀫𑀸𑀓𑀸𑀲𑀺 𑀧𑀸𑀧𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀆𑀯𑀺 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀳𑁄𑁇_  


1474. _𑀲𑀘𑁂 𑀘 𑀧𑀸𑀧𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺 𑀓𑀭𑁄𑀲𑀺 𑀯𑀸;_  
_𑀦 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀢𑁆𑀬𑀢𑁆𑀣𑀺,_  
_𑀉𑀧𑁆𑀧𑀘𑁆𑀘𑀸𑀧𑀺 𑀧𑀮𑀸𑀬𑀢𑁄”𑀢𑀺𑁇_  


1475. _“𑀫𑀢𑀁 𑀯𑀸 𑀅𑀫𑁆𑀫 𑀭𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀯𑀸 𑀚𑀻𑀯𑀁 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀚𑀻𑀯𑀦𑁆𑀢𑀁 𑀅𑀫𑁆𑀫 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻,_  
_𑀓𑀲𑁆𑀫𑀸 𑀫𑀁 𑀅𑀫𑁆𑀫 𑀭𑁄𑀤𑀲𑀻”𑀢𑀺𑁇_  


1476. _“𑀫𑀢𑀁 𑀯𑀸 𑀧𑀼𑀢𑁆𑀢 𑀭𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀯𑀸 𑀚𑀻𑀯𑀁 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀬𑁄 𑀘 𑀓𑀸𑀫𑁂 𑀘𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀼𑀦𑀭𑀸𑀕𑀘𑁆𑀙𑀢𑁂 𑀇𑀥;_  
_𑀢𑀁 𑀯𑀸𑀧𑀺 𑀧𑀼𑀢𑁆𑀢 𑀭𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀦 𑀚𑀻𑀯𑀁 𑀫𑀢𑁄 𑀳𑀺 𑀲𑁄𑁇_  


1477. _𑀓𑀼𑀓𑁆𑀓𑀼𑀍𑀆 𑀉𑀩𑁆𑀪𑀢𑁄 𑀢𑀸𑀢,_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀍𑀅𑀁 𑀧𑀢𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀲𑀺;_  
_𑀦𑀭𑀓𑀸 𑀉𑀩𑁆𑀪𑀢𑁄 𑀢𑀸𑀢,_  
_𑀦𑀭𑀓𑀁 𑀧𑀢𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀲𑀺𑁇_  


1478. _𑀅𑀪𑀺𑀥𑀸𑀯𑀣 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀓𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀧𑀬𑀸𑀫𑀲𑁂;_  
_𑀆𑀤𑀺𑀢𑁆𑀢𑀸 𑀦𑀻𑀳𑀢𑀁 𑀪𑀡𑁆𑀟𑀁,_  
_𑀧𑀼𑀦 𑀟𑀬𑁆𑀳𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀲𑀻”𑀢𑀺𑁇_
