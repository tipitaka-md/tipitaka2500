# 𑁧𑁦𑁇𑁧𑁇𑁯 𑀧𑀞𑀫𑀲𑀼𑀓𑁆𑀓𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀬𑀓𑁆𑀔𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀇𑀦𑁆𑀤𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀲𑀼𑀓𑁆𑀓𑀸𑀲𑀼𑀢𑁆𑀢

(243.)

1505\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀓𑁆𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀫𑀳𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀭𑀺𑀯𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀓𑁆𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀬𑀓𑁆𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀭𑀣𑀺𑀓𑀸𑀬 𑀭𑀣𑀺𑀓𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

1506\. _“𑀓𑀺𑀁 𑀫𑁂 𑀓𑀢𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸,_  
_𑀫𑀥𑀼𑀧𑀻𑀢𑀸𑀯 𑀲𑁂𑀬𑀭𑁂;_  
_𑀬𑁂 𑀲𑀼𑀓𑁆𑀓𑀁 𑀦 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑀁 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁𑁇_  


1507\. _𑀢𑀜𑁆𑀘 𑀧𑀦 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀻𑀬𑀁,_  
_𑀅𑀲𑁂𑀘𑀦𑀓𑀫𑁄𑀚𑀯𑀁;_  
_𑀧𑀺𑀯𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀮𑀸𑀳𑀓𑀫𑀺𑀯 𑀧𑀦𑁆𑀣𑀕𑀽”𑀢𑀺𑁇_
