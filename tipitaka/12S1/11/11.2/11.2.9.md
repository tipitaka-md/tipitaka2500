# 𑁧𑁧𑁇𑁨𑁇𑁯 𑀲𑀢𑁆𑀣𑀸𑀭𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀢𑁆𑀣𑀸𑀭𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(265.)

1664\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀚𑁂𑀢𑀯𑀦𑁂𑁇 “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀸𑀢𑀮𑀺𑀁 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀬𑁄𑀚𑁂𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀫𑀸𑀢𑀮𑀺, 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑀁, 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀲𑀼𑀪𑀽𑀫𑀺𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑀁 𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀲𑀺—  ‘𑀬𑀼𑀢𑁆𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑁄 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑁄𑁇 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1665\. _‘𑀬𑀜𑁆𑀳𑀺 𑀤𑁂𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘,_  
_𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸𑀲𑀯;_  
_𑀅𑀣 𑀓𑁄 𑀦𑀸𑀫 𑀲𑁄 𑀬𑀓𑁆𑀔𑁄,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀓𑁆𑀓 𑀦𑀫𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇_  


1666\. _‘𑀬𑁄 𑀇𑀥 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀅𑀦𑁄𑀫𑀦𑀸𑀫𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁,_  
_𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀫𑀸𑀢𑀮𑀺𑁇_  


1667\. _𑀬𑁂𑀲𑀁 𑀭𑀸𑀕𑁄 𑀘 𑀤𑁄𑀲𑁄 𑀘,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀭𑀸𑀚𑀺𑀢𑀸;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀅𑀭𑀳𑀦𑁆𑀢𑁄,_  
_𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀫𑀸𑀢𑀮𑀺𑁇_  


1668\. _𑀬𑁂 𑀭𑀸𑀕𑀤𑁄𑀲𑀯𑀺𑀦𑀬𑀸,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸;_  
_𑀲𑁂𑀓𑁆𑀔𑀸 𑀅𑀧𑀘𑀬𑀸𑀭𑀸𑀫𑀸,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀭𑁂;_  
_𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀫𑀸𑀢𑀮𑀻’𑀢𑀺𑁇_  


1669\. _‘𑀲𑁂𑀝𑁆𑀞𑀸 𑀳𑀺 𑀓𑀺𑀭 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑁂 𑀢𑁆𑀯𑀁 𑀲𑀓𑁆𑀓 𑀦𑀫𑀲𑁆𑀲𑀲𑀺;_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑁂 𑀦𑀫𑀲𑁆𑀲𑀲𑀺 𑀯𑀸𑀲𑀯𑀸’𑀢𑀺𑁇_  


1670\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀫𑀖𑀯𑀸,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑀸 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀺;_  
_𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀫𑀼𑀔𑁄 𑀭𑀣𑀫𑀸𑀭𑀼𑀳𑀻”𑀢𑀺𑁇_
