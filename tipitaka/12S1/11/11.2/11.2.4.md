# 𑁧𑁧𑁇𑁨𑁇𑁪 𑀤𑀮𑀺𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀤𑀮𑀺𑀤𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(260.)

1640\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1641\. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑀮𑀺𑀤𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀧𑀡𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀭𑀸𑀓𑁄𑁇 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀲𑀤𑁆𑀥𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀲𑀼𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀘𑀸𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀧𑀜𑁆𑀜𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑁇 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀲𑀤𑁆𑀥𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀺 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁𑁇 𑀲𑁄 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀅𑀬𑀜𑁆𑀳𑀺 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑀮𑀺𑀤𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀧𑀡𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀭𑀸𑀓𑁄; 𑀲𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁𑁇 𑀲𑁄 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑀸’𑀢𑀺𑁇

1642\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀫𑀸 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀏𑀢𑀲𑁆𑀲 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀺𑀢𑁆𑀣𑁇 𑀏𑀲𑁄 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀲𑀤𑁆𑀥𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀲𑀼𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀘𑀸𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺, 𑀧𑀜𑁆𑀜𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑁇 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀲𑀤𑁆𑀥𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁𑁇 𑀲𑁄 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀅𑀦𑀼𑀦𑀬𑀫𑀸𑀦𑁄 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

1643\. _‘𑀬𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀸 𑀢𑀣𑀸𑀕𑀢𑁂,_  
_𑀅𑀘𑀮𑀸 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀲𑀻𑀮𑀜𑁆𑀘 𑀬𑀲𑁆𑀲 𑀓𑀮𑁆𑀬𑀸𑀡𑀁,_  
_𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑀁 𑀧𑀲𑀁𑀲𑀺𑀢𑀁𑁇_  


1644\. _𑀲𑀁𑀖𑁂 𑀧𑀲𑀸𑀤𑁄 𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀺,_  
_𑀉𑀚𑀼𑀪𑀽𑀢𑀜𑁆𑀘 𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀅𑀤𑀮𑀺𑀤𑁆𑀤𑁄𑀢𑀺 𑀢𑀁 𑀆𑀳𑀼,_  
_𑀅𑀫𑁄𑀖𑀁 𑀢𑀲𑁆𑀲 𑀚𑀻𑀯𑀺𑀢𑀁𑁇_  


1645\. _𑀢𑀲𑁆𑀫𑀸 𑀲𑀤𑁆𑀥𑀜𑁆𑀘 𑀲𑀻𑀮𑀜𑁆𑀘,_  
_𑀧𑀲𑀸𑀤𑀁 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑁂𑀣 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀲𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦 𑀲𑀸𑀲𑀦𑀦𑁆’”𑀢𑀺𑁇_
