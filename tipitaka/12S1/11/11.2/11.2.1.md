# 𑁧𑁧𑁇𑁨𑁇𑁧 𑀯𑀢𑀧𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀯𑀢𑀧𑀤𑀲𑀼𑀢𑁆𑀢

(257.)

1612\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁𑁇 “𑀲𑀓𑁆𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀯𑀢𑀧𑀤𑀸𑀦𑀺 𑀲𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀬𑁂𑀲𑀁 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀢𑁆𑀢𑀸 𑀲𑀓𑁆𑀓𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀢𑀁 𑀅𑀚𑁆𑀛𑀕𑀸𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀯𑀢𑀧𑀤𑀸𑀦𑀺? 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀓𑀼𑀮𑁂 𑀚𑁂𑀝𑁆𑀞𑀸𑀧𑀘𑀸𑀬𑀻 𑀅𑀲𑁆𑀲𑀁, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀲𑀡𑁆𑀳𑀯𑀸𑀘𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑁂𑀬𑁆𑀬𑀁 𑀫𑀼𑀢𑁆𑀢𑀘𑀸𑀕𑁄 𑀧𑀬𑀢𑀧𑀸𑀡𑀺 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀭𑀢𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑁄, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀲𑀘𑁆𑀘𑀯𑀸𑀘𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀅𑀲𑁆𑀲𑀁—  𑀲𑀘𑁂𑀧𑀺 𑀫𑁂 𑀓𑁄𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺𑁇 𑀲𑀓𑁆𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀇𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀯𑀢𑀧𑀤𑀸𑀦𑀺 𑀲𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀬𑁂𑀲𑀁 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀢𑁆𑀢𑀸 𑀲𑀓𑁆𑀓𑁄 𑀲𑀓𑁆𑀓𑀢𑁆𑀢𑀁 𑀅𑀚𑁆𑀛𑀕𑀸𑀢𑀺𑁇

1613\. _𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑀁 𑀚𑀦𑁆𑀢𑀼𑀁,_  
_𑀓𑀼𑀮𑁂 𑀚𑁂𑀝𑁆𑀞𑀸𑀧𑀘𑀸𑀬𑀺𑀦𑀁;_  
_𑀲𑀡𑁆𑀳𑀁 𑀲𑀔𑀺𑀮𑀲𑀫𑁆𑀪𑀸𑀲𑀁,_  
_𑀧𑁂𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑀧𑁆𑀧𑀳𑀸𑀬𑀺𑀦𑀁𑁇_  


1614\. _𑀫𑀘𑁆𑀙𑁂𑀭𑀯𑀺𑀦𑀬𑁂 𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀲𑀘𑁆𑀘𑀁 𑀓𑁄𑀥𑀸𑀪𑀺𑀪𑀼𑀁 𑀦𑀭𑀁;_  
_𑀢𑀁 𑀯𑁂 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸,_  
_𑀆𑀳𑀼 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀢𑀻”𑀢𑀺𑁇_
