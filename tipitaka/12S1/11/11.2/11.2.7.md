# 𑁧𑁧𑁇𑁨𑁇𑁭 𑀩𑀼𑀤𑁆𑀥𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀩𑀼𑀤𑁆𑀥𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(263.)

1653\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀚𑁂𑀢𑀯𑀦𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀘 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀘 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑁂𑀓𑀁 𑀤𑁆𑀯𑀸𑀭𑀩𑀸𑀳𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1654\. _“𑀉𑀝𑁆𑀞𑁂𑀳𑀺 𑀯𑀻𑀭 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫,_  
_𑀧𑀦𑁆𑀦𑀪𑀸𑀭 𑀅𑀦𑀡 𑀯𑀺𑀘𑀭 𑀮𑁄𑀓𑁂;_  
_𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀢𑁂 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁,_  
_𑀘𑀦𑁆𑀤𑁄 𑀬𑀣𑀸 𑀧𑀦𑁆𑀦𑀭𑀲𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀦𑁆”𑀢𑀺𑁇_  


1655\. “𑀦 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀏𑀯𑀁 𑀯𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀯𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀸—

1656\. _‘𑀉𑀝𑁆𑀞𑁂𑀳𑀺 𑀯𑀻𑀭 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫,_  
_𑀲𑀢𑁆𑀣𑀯𑀸𑀳 𑀅𑀦𑀡 𑀯𑀺𑀘𑀭 𑀮𑁄𑀓𑁂;_  
_𑀤𑁂𑀲𑀲𑁆𑀲𑀼 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁,_  
_𑀅𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’”𑀢𑀺𑁇_
