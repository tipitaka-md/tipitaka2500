# 𑁧𑁧𑁇𑁨𑁇𑁮 𑀕𑀳𑀝𑁆𑀞𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀕𑀳𑀝𑁆𑀞𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(264.)

1657\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁𑁇 𑀢𑀢𑁆𑀭…𑀧𑁂…  𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀸𑀢𑀮𑀺𑀁 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀬𑁄𑀚𑁂𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀫𑀸𑀢𑀮𑀺, 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑀁𑁇 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀲𑀼𑀪𑀽𑀫𑀺𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑀁 𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀲𑀺—  ‘𑀬𑀼𑀢𑁆𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑁄 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑁄𑁇 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀼𑀤𑀁 𑀧𑀼𑀣𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀦𑀫𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1658\. _‘𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀪𑀼𑀫𑁆𑀫𑀸 𑀘 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀢𑀺𑀤𑀲𑀸 𑀘 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀅𑀣 𑀓𑁄 𑀦𑀸𑀫 𑀲𑁄 𑀬𑀓𑁆𑀔𑁄,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀓𑁆𑀓 𑀦𑀫𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇_  


1659\. _‘𑀫𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸,_  
_𑀲𑀩𑁆𑀩𑁂 𑀪𑀼𑀫𑁆𑀫𑀸 𑀘 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀢𑀺𑀤𑀲𑀸 𑀘 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


1660\. _𑀅𑀳𑀜𑁆𑀘 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀘𑀺𑀭𑀭𑀢𑁆𑀢𑀲𑀫𑀸𑀳𑀺𑀢𑁂;_  
_𑀲𑀫𑁆𑀫𑀸𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀯𑀦𑁆𑀤𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀸𑀬𑀦𑁂𑁇_  


1661\. _𑀬𑁂 𑀕𑀳𑀝𑁆𑀞𑀸 𑀧𑀼𑀜𑁆𑀜𑀓𑀭𑀸,_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀉𑀧𑀸𑀲𑀓𑀸;_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀤𑀸𑀭𑀁 𑀧𑁄𑀲𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀫𑀸𑀢𑀮𑀻’𑀢𑀺𑁇_  


1662\. _‘𑀲𑁂𑀝𑁆𑀞𑀸 𑀳𑀺 𑀓𑀺𑀭 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑁂 𑀢𑁆𑀯𑀁 𑀲𑀓𑁆𑀓 𑀦𑀫𑀲𑁆𑀲𑀲𑀺;_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑁂 𑀦𑀫𑀲𑁆𑀲𑀲𑀺 𑀯𑀸𑀲𑀯𑀸’𑀢𑀺𑁇_  


1663\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀫𑀖𑀯𑀸,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑀸 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀺;_  
_𑀧𑀼𑀣𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀦𑀫𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀫𑀼𑀔𑁄 𑀭𑀣𑀫𑀸𑀭𑀼𑀳𑀻”𑀢𑀺𑁇_
