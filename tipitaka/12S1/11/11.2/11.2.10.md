# 𑁧𑁧𑁇𑁨𑁇𑁧𑁦 𑀲𑀁𑀖𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀁𑀖𑀯𑀦𑁆𑀤𑀦𑀸𑀲𑀼𑀢𑁆𑀢

(266.)

1671\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀚𑁂𑀢𑀯𑀦𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄…𑀧𑁂…  𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀸𑀢𑀮𑀺𑀁 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  ‘𑀬𑁄𑀚𑁂𑀳𑀺, 𑀲𑀫𑁆𑀫 𑀫𑀸𑀢𑀮𑀺, 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑀁, 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀲𑀼𑀪𑀽𑀫𑀺𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸, 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑀁 𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀲𑀺—  ‘𑀬𑀼𑀢𑁆𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑁄 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑁄, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀚𑀬𑀦𑁆𑀢𑀧𑀸𑀲𑀸𑀤𑀸 𑀑𑀭𑁄𑀳𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀼𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀦𑀫𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1672\. _‘𑀢𑀜𑁆𑀳𑀺 𑀏𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀧𑀽𑀢𑀺𑀤𑁂𑀳𑀲𑀬𑀸 𑀦𑀭𑀸;_  
_𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑀸 𑀓𑀼𑀡𑀧𑀫𑁆𑀳𑁂𑀢𑁂,_  
_𑀔𑀼𑀧𑁆𑀧𑀺𑀧𑀸𑀲𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


1673\. _𑀓𑀺𑀁 𑀦𑀼 𑀢𑁂𑀲𑀁 𑀧𑀺𑀳𑀬𑀲𑀺,_  
_𑀅𑀦𑀸𑀕𑀸𑀭𑀸𑀦 𑀯𑀸𑀲𑀯;_  
_𑀆𑀘𑀸𑀭𑀁 𑀇𑀲𑀺𑀦𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀫 𑀯𑀘𑁄 𑀢𑀯𑀸’𑀢𑀺𑁇_  


1674\. _‘𑀏𑀢𑀁 𑀢𑁂𑀲𑀁 𑀧𑀺𑀳𑀬𑀸𑀫𑀺,_  
_𑀅𑀦𑀸𑀕𑀸𑀭𑀸𑀦 𑀫𑀸𑀢𑀮𑀺;_  
_𑀬𑀫𑁆𑀳𑀸 𑀕𑀸𑀫𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀸 𑀯𑀚𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1675\. _𑀦 𑀢𑁂𑀲𑀁 𑀓𑁄𑀝𑁆𑀞𑁂 𑀑𑀧𑁂𑀦𑁆𑀢𑀺,_  
_𑀦 𑀓𑀼𑀫𑁆𑀪𑀺 𑀦 𑀓𑀍𑀑𑀧𑀺𑀬𑀁;_  
_𑀧𑀭𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀫𑁂𑀲𑀸𑀦𑀸,_  
_𑀢𑁂𑀦 𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑀼𑀩𑁆𑀩𑀢𑀸𑁇_  


1676\. _𑀲𑀼𑀫𑀦𑁆𑀢𑀫𑀦𑁆𑀢𑀺𑀦𑁄 𑀥𑀻𑀭𑀸,_  
_𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀲𑀫𑀜𑁆𑀘𑀭𑀸;_  
_𑀤𑁂𑀯𑀸 𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀅𑀲𑀼𑀭𑁂𑀳𑀺,_  
_𑀧𑀼𑀣𑀼 𑀫𑀘𑁆𑀘𑀸 𑀘 𑀫𑀸𑀢𑀮𑀺𑁇_  


1677\. _𑀅𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑀸 𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁂𑀲𑀼,_  
_𑀅𑀢𑁆𑀢𑀤𑀡𑁆𑀟𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸;_  
_𑀲𑀸𑀤𑀸𑀦𑁂𑀲𑀼 𑀅𑀦𑀸𑀤𑀸𑀦𑀸,_  
_𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀫𑀸𑀢𑀮𑀻’𑀢𑀺𑁇_  


1678\. _‘𑀲𑁂𑀝𑁆𑀞𑀸 𑀳𑀺 𑀓𑀺𑀭 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑁂 𑀢𑁆𑀯𑀁 𑀲𑀓𑁆𑀓 𑀦𑀫𑀲𑁆𑀲𑀲𑀺;_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑁂 𑀦𑀫𑀲𑁆𑀲𑀲𑀺 𑀯𑀸𑀲𑀯𑀸’”𑀢𑀺𑁇_  


1679\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀫𑀖𑀯𑀸,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑀸 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀺;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀦𑀫𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀫𑀼𑀔𑁄 𑀭𑀣𑀫𑀸𑀭𑀼𑀳𑀻𑀢𑀺𑁇_  


---

1680\. 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑁄𑁇



---

1681\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1682\. _𑀤𑁂𑀯𑀸 𑀧𑀦 𑀢𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀮𑀺𑀤𑁆𑀤𑀜𑁆𑀘 𑀭𑀸𑀫𑀡𑁂𑀬𑁆𑀬𑀓𑀁;_  
_𑀬𑀚𑀫𑀸𑀦𑀜𑁆𑀘 𑀯𑀦𑁆𑀤𑀦𑀸,_  
_𑀢𑀬𑁄 𑀲𑀓𑁆𑀓𑀦𑀫𑀲𑁆𑀲𑀦𑀸𑀢𑀺𑁇_
