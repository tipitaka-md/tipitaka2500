# 𑁧𑁧𑁇𑁧𑁇𑁮 𑀯𑁂𑀭𑁄𑀘𑀦𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀯𑁂𑀭𑁄𑀘𑀦𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(254.)

1595\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀚𑁂𑀢𑀯𑀦𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀘 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀭𑁄𑀘𑀦𑁄 𑀘 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑁂𑀓𑀁 𑀤𑁆𑀯𑀸𑀭𑀩𑀸𑀳𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀭𑁄𑀘𑀦𑁄 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1596\. _“𑀯𑀸𑀬𑀫𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀬𑀸𑀯 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀦𑀺𑀧𑁆𑀨𑀤𑀸;_  
_𑀦𑀺𑀧𑁆𑀨𑀦𑁆𑀦𑀲𑁄𑀪𑀦𑁄 𑀅𑀢𑁆𑀣𑁄,_  
_𑀯𑁂𑀭𑁄𑀘𑀦𑀯𑀘𑁄 𑀇𑀤𑀦𑁆”𑀢𑀺𑁇_  


1597\. _“𑀯𑀸𑀬𑀫𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀬𑀸𑀯 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀦𑀺𑀧𑁆𑀨𑀤𑀸;_  
_𑀦𑀺𑀧𑁆𑀨𑀦𑁆𑀦𑀲𑁄𑀪𑀦𑁄 𑀅𑀢𑁆𑀣𑁄,_  
_𑀔𑀦𑁆𑀢𑁆𑀬𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


1598\. _“𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀚𑀸𑀢𑀸,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀬𑀣𑀸𑀭𑀳𑀁;_  
_𑀲𑀁𑀬𑁄𑀕𑀧𑀭𑀫𑀸 𑀢𑁆𑀯𑁂𑀯,_  
_𑀲𑀫𑁆𑀪𑁄𑀕𑀸 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀦𑀺𑀧𑁆𑀨𑀦𑁆𑀦𑀲𑁄𑀪𑀦𑁄 𑀅𑀢𑁆𑀣𑁄,_  
_𑀯𑁂𑀭𑁄𑀘𑀦𑀯𑀘𑁄 𑀇𑀤𑀦𑁆”𑀢𑀺𑁇_  


1599\. _“𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑀸 𑀅𑀢𑁆𑀣𑀚𑀸𑀢𑀸,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀬𑀣𑀸𑀭𑀳𑀁;_  
_𑀲𑀁𑀬𑁄𑀕𑀧𑀭𑀫𑀸 𑀢𑁆𑀯𑁂𑀯,_  
_𑀲𑀫𑁆𑀪𑁄𑀕𑀸 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀦𑀺𑀧𑁆𑀨𑀦𑁆𑀦𑀲𑁄𑀪𑀦𑁄 𑀅𑀢𑁆𑀣𑁄,_  
_𑀔𑀦𑁆𑀢𑁆𑀬𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_
