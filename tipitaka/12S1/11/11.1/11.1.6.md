# 𑁧𑁧𑁇𑁧𑁇𑁬 𑀓𑀼𑀮𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀮𑀸𑀯𑀓𑀲𑀼𑀢𑁆𑀢

(252.)

1588\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁𑁇 “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀲𑀼𑀭𑀲𑀗𑁆𑀕𑀸𑀫𑁄 𑀲𑀫𑀼𑀧𑀩𑁆𑀬𑀽𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀅𑀲𑀼𑀭𑀸 𑀚𑀺𑀦𑀺𑀁𑀲𑀼, 𑀤𑁂𑀯𑀸 𑀧𑀭𑀸𑀚𑀺𑀦𑀺𑀁𑀲𑀼𑁇 𑀧𑀭𑀸𑀚𑀺𑀢𑀸 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸 𑀅𑀧𑀸𑀬𑀁𑀲𑁆𑀯𑁂𑀯 𑀉𑀢𑁆𑀢𑀭𑁂𑀦𑀫𑀼𑀔𑀸, 𑀅𑀪𑀺𑀬𑀁𑀲𑁆𑀯𑁂𑀯 𑀦𑁂 𑀅𑀲𑀼𑀭𑀸𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1589\. _‘𑀓𑀼𑀮𑀸𑀯𑀓𑀸 𑀫𑀸𑀢𑀮𑀺 𑀲𑀺𑀫𑁆𑀩𑀮𑀺𑀲𑁆𑀫𑀺𑀁,_  
_𑀈𑀲𑀸𑀫𑀼𑀔𑁂𑀦 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀲𑁆𑀲𑀼;_  
_𑀓𑀸𑀫𑀁 𑀘𑀚𑀸𑀫 𑀅𑀲𑀼𑀭𑁂𑀲𑀼 𑀧𑀸𑀡𑀁,_  
_𑀫𑀸𑀬𑀺𑀫𑁂 𑀤𑀺𑀚𑀸 𑀯𑀺𑀓𑀼𑀮𑀸𑀯𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀦𑁆’𑀢𑀺𑁇_  


1590\. ‘𑀏𑀯𑀁, 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀮𑀺 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑀁 𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀼𑀭𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑁄 𑀔𑁄 𑀤𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑁄 𑀆𑀚𑀜𑁆𑀜𑀭𑀣𑁄𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀤𑁂𑀯𑀸 𑀅𑀲𑀼𑀭𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀸𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺 𑀪𑀻𑀢𑀸 𑀅𑀲𑀼𑀭𑀧𑀼𑀭𑀫𑁂𑀯 𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀦 𑀚𑀬𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺𑁇
