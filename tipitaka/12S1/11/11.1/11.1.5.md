# 𑁧𑁧𑁇𑁧𑁇𑁫 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀚𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀚𑀬𑀲𑀼𑀢𑁆𑀢

(251.)

1574\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀲𑀼𑀭𑀲𑀗𑁆𑀕𑀸𑀫𑁄 𑀲𑀫𑀼𑀧𑀩𑁆𑀬𑀽𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀳𑁄𑀢𑀼, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑁂𑀦 𑀚𑀬𑁄’𑀢𑀺𑁇 ‘𑀳𑁄𑀢𑀼, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺, 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑁂𑀦 𑀚𑀬𑁄’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸 𑀘 𑀅𑀲𑀼𑀭𑀸 𑀘 𑀧𑀸𑀭𑀺𑀲𑀚𑁆𑀚𑁂 𑀞𑀧𑁂𑀲𑀼𑀁—  ‘𑀇𑀫𑁂 𑀦𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀪𑀡, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀕𑀸𑀣𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀢𑀼𑀫𑁆𑀳𑁂 𑀔𑁆𑀯𑁂𑀢𑁆𑀣, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀤𑁂𑀯𑀸𑁇 𑀪𑀡, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺, 𑀕𑀸𑀣𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1575\. _‘𑀪𑀺𑀬𑁆𑀬𑁄 𑀩𑀸𑀮𑀸 𑀧𑀪𑀺𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀦𑁄 𑀘𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀥𑀓𑁄;_  
_𑀢𑀲𑁆𑀫𑀸 𑀪𑀼𑀲𑁂𑀦 𑀤𑀡𑁆𑀟𑁂𑀦,_  
_𑀥𑀻𑀭𑁄 𑀩𑀸𑀮𑀁 𑀦𑀺𑀲𑁂𑀥𑀬𑁂’𑀢𑀺𑁇_  


1576\. 𑀪𑀸𑀲𑀺𑀢𑀸𑀬 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀦𑀸 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁂𑀦 𑀕𑀸𑀣𑀸𑀬 𑀅𑀲𑀼𑀭𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀤𑁂𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀪𑀡, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀕𑀸𑀣𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1577\. _‘𑀏𑀢𑀤𑁂𑀯 𑀅𑀳𑀁 𑀫𑀜𑁆𑀜𑁂,_  
_𑀩𑀸𑀮𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁂𑀥𑀦𑀁;_  
_𑀧𑀭𑀁 𑀲𑀗𑁆𑀓𑀼𑀧𑀺𑀢𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀬𑁄 𑀲𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀫𑀢𑀻’𑀢𑀺𑁇_  


1578\. 𑀪𑀸𑀲𑀺𑀢𑀸𑀬 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁂𑀦 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁂𑀦 𑀕𑀸𑀣𑀸𑀬, 𑀤𑁂𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀅𑀲𑀼𑀭𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀪𑀡, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺, 𑀕𑀸𑀣𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1579\. _‘𑀏𑀢𑀤𑁂𑀯 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀸𑀬,_  
_𑀯𑀚𑁆𑀚𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸𑀲𑀯;_  
_𑀬𑀤𑀸 𑀦𑀁 𑀫𑀜𑁆𑀜𑀢𑀺 𑀩𑀸𑀮𑁄,_  
_𑀪𑀬𑀸 𑀫𑁆𑀬𑀸𑀬𑀁 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀢𑀺;_  
_𑀅𑀚𑁆𑀛𑀸𑀭𑀼𑀳𑀢𑀺 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑁄,_  
_𑀕𑁄𑀯 𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀮𑀸𑀬𑀺𑀦𑀦𑁆’𑀢𑀺𑁇_  


1580\. 𑀪𑀸𑀲𑀺𑀢𑀸𑀬 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀦𑀸 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁂𑀦 𑀕𑀸𑀣𑀸𑀬 𑀅𑀲𑀼𑀭𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀤𑁂𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀪𑀡, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀕𑀸𑀣𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

1581\. _‘𑀓𑀸𑀫𑀁 𑀫𑀜𑁆𑀜𑀢𑀼 𑀯𑀸 𑀫𑀸 𑀯𑀸,_  
_𑀪𑀬𑀸 𑀫𑁆𑀬𑀸𑀬𑀁 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀢𑀺;_  
_𑀲𑀤𑀢𑁆𑀣𑀧𑀭𑀫𑀸 𑀅𑀢𑁆𑀣𑀸,_  
_𑀔𑀦𑁆𑀢𑁆𑀬𑀸 𑀪𑀺𑀬𑁆𑀬𑁄 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


1582\. _𑀬𑁄 𑀳𑀯𑁂 𑀩𑀮𑀯𑀸 𑀲𑀦𑁆𑀢𑁄,_  
_𑀤𑀼𑀩𑁆𑀩𑀮𑀲𑁆𑀲 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀢𑀺;_  
_𑀢𑀫𑀸𑀳𑀼 𑀧𑀭𑀫𑀁 𑀔𑀦𑁆𑀢𑀺𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀔𑀫𑀢𑀺 𑀤𑀼𑀩𑁆𑀩𑀮𑁄𑁇_  


1583\. _𑀅𑀩𑀮𑀁 𑀢𑀁 𑀩𑀮𑀁 𑀆𑀳𑀼,_  
_𑀬𑀲𑁆𑀲 𑀩𑀸𑀮𑀩𑀮𑀁 𑀩𑀮𑀁;_  
_𑀩𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀕𑀼𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀧𑀝𑀺𑀯𑀢𑁆𑀢𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


1584\. _𑀢𑀲𑁆𑀲𑁂𑀯 𑀢𑁂𑀦 𑀧𑀸𑀧𑀺𑀬𑁄,_  
_𑀬𑁄 𑀓𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀓𑀼𑀤𑁆𑀥𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑁄,_  
_𑀲𑀗𑁆𑀕𑀸𑀫𑀁 𑀚𑁂𑀢𑀺 𑀤𑀼𑀚𑁆𑀚𑀬𑀁𑁇_  


1585\. _𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀭𑀲𑁆𑀲 𑀘;_  
_𑀧𑀭𑀁 𑀲𑀗𑁆𑀓𑀼𑀧𑀺𑀢𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀬𑁄 𑀲𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


1586\. _𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀭𑀲𑁆𑀲 𑀘;_  
_𑀚𑀦𑀸 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑁄𑀢𑀺,_  
_𑀬𑁂 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑀸’𑀢𑀺𑁇_  


1587\. 𑀪𑀸𑀲𑀺𑀢𑀸𑀲𑀼 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁂𑀦 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁂𑀦 𑀕𑀸𑀣𑀸𑀲𑀼, 𑀤𑁂𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀅𑀲𑀼𑀭𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀦𑀜𑁆𑀘 𑀅𑀲𑀼𑀭𑀸𑀦𑀜𑁆𑀘 𑀧𑀸𑀭𑀺𑀲𑀚𑁆𑀚𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀦𑀸 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁂𑀦 𑀕𑀸𑀣𑀸𑀬𑁄𑁇 𑀢𑀸 𑀘 𑀔𑁄 𑀲𑀤𑀡𑁆𑀟𑀸𑀯𑀘𑀭𑀸 𑀲𑀲𑀢𑁆𑀣𑀸𑀯𑀘𑀭𑀸, 𑀇𑀢𑀺 𑀪𑀡𑁆𑀟𑀦𑀁 𑀇𑀢𑀺 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀇𑀢𑀺 𑀓𑀮𑀳𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀲𑀓𑁆𑀓𑁂𑀦 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁂𑀦 𑀕𑀸𑀣𑀸𑀬𑁄𑁇 𑀢𑀸 𑀘 𑀔𑁄 𑀅𑀤𑀡𑁆𑀟𑀸𑀯𑀘𑀭𑀸 𑀅𑀲𑀢𑁆𑀣𑀸𑀯𑀘𑀭𑀸, 𑀇𑀢𑀺 𑀅𑀪𑀡𑁆𑀟𑀦𑀁 𑀇𑀢𑀺 𑀅𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀇𑀢𑀺 𑀅𑀓𑀮𑀳𑁄𑁇 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑁂𑀦 𑀚𑀬𑁄’𑀢𑀺𑁇 𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑁂𑀦 𑀚𑀬𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺𑁇
