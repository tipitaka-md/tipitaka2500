# 𑁧𑁧𑁇𑁧𑁇𑁧𑁦 𑀲𑀫𑀼𑀤𑁆𑀤𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀓𑀲𑀼𑀢𑁆𑀢

(256.)

1603\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁𑁇 “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑀢𑀻𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀻𑀲𑀼 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁂𑀯𑀸𑀲𑀼𑀭𑀲𑀗𑁆𑀕𑀸𑀫𑁄 𑀲𑀫𑀼𑀧𑀩𑁆𑀬𑀽𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀲𑀁 𑀇𑀲𑀻𑀦𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀔𑁄 𑀤𑁂𑀯𑀸, 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀅𑀲𑀼𑀭𑀸𑁇 𑀲𑀺𑀬𑀸𑀧𑀺 𑀦𑁄 𑀅𑀲𑀼𑀭𑀢𑁄 𑀪𑀬𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀲𑀫𑁆𑀩𑀭𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀬𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀲𑀫𑀼𑀤𑁆𑀤𑀢𑀻𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀻𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸 𑀲𑀫𑁆𑀩𑀭𑀲𑁆𑀲 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀩𑀭𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺𑀁𑀲𑀼—

1604\. _‘𑀇𑀲𑀬𑁄 𑀲𑀫𑁆𑀩𑀭𑀁 𑀧𑀢𑁆𑀢𑀸,_  
_𑀬𑀸𑀘𑀦𑁆𑀢𑀺 𑀅𑀪𑀬𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀓𑀸𑀫𑀗𑁆𑀓𑀭𑁄 𑀳𑀺 𑀢𑁂 𑀤𑀸𑀢𑀼𑀁,_  
_𑀪𑀬𑀲𑁆𑀲 𑀅𑀪𑀬𑀲𑁆𑀲 𑀯𑀸’𑀢𑀺𑁇_  


1605\. _‘𑀇𑀲𑀻𑀦𑀁 𑀅𑀪𑀬𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀤𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀲𑀓𑁆𑀓𑀲𑁂𑀯𑀺𑀦𑀁;_  
_𑀅𑀪𑀬𑀁 𑀬𑀸𑀘𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀪𑀬𑀫𑁂𑀯 𑀤𑀤𑀸𑀫𑀺 𑀯𑁄’𑀢𑀺𑁇_  


1606\. _‘𑀅𑀪𑀬𑀁 𑀬𑀸𑀘𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀪𑀬𑀫𑁂𑀯 𑀤𑀤𑀸𑀲𑀺 𑀦𑁄;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫 𑀢𑁂 𑀏𑀢𑀁,_  
_𑀅𑀓𑁆𑀔𑀬𑀁 𑀳𑁄𑀢𑀼 𑀢𑁂 𑀪𑀬𑀁𑁇_  


1607\. _𑀬𑀸𑀤𑀺𑀲𑀁 𑀯𑀧𑀢𑁂 𑀩𑀻𑀚𑀁,_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀳𑀭𑀢𑁂 𑀨𑀮𑀁;_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀓𑀸𑀭𑀻 𑀓𑀮𑁆𑀬𑀸𑀡𑀁,_  
_𑀧𑀸𑀧𑀓𑀸𑀭𑀻 𑀘 𑀧𑀸𑀧𑀓𑀁;_  
_𑀧𑀯𑀼𑀢𑁆𑀢𑀁 𑀢𑀸𑀢 𑀢𑁂 𑀩𑀻𑀚𑀁,_  
_𑀨𑀮𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇_  


1608\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀩𑀭𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀅𑀪𑀺𑀲𑀧𑀺𑀢𑁆𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀲𑀫𑁆𑀩𑀭𑀲𑁆𑀲 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑀢𑀻𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀻𑀲𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑀭𑁄 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀢𑁂𑀳𑀺 𑀇𑀲𑀻𑀳𑀺 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀪𑀺𑀲𑀧𑀺𑀢𑁄 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀤𑀁 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀉𑀩𑁆𑀩𑀺𑀚𑁆𑀚𑀻”𑀢𑀺𑁇

1610\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1611\. _𑀲𑀼𑀯𑀻𑀭𑀁 𑀲𑀼𑀲𑀻𑀫𑀜𑁆𑀘𑁂𑀯,_  
_𑀥𑀚𑀕𑁆𑀕𑀁 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀚𑀬𑀜𑁆𑀘𑁂𑀯,_  
_𑀓𑀼𑀮𑀸𑀯𑀓𑀁 𑀦𑀤𑀼𑀩𑁆𑀪𑀺𑀬𑀁;_  
_𑀯𑁂𑀭𑁄𑀘𑀦 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄,_  
_𑀇𑀲𑀬𑁄 𑀅𑀭𑀜𑁆𑀜𑀓𑀜𑁆𑀘𑁂𑀯;_  
_𑀇𑀲𑀬𑁄 𑀘 𑀲𑀫𑀼𑀤𑁆𑀤𑀓𑀸𑀢𑀺𑁇_
