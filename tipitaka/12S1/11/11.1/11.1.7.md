# 𑁧𑁧𑁇𑁧𑁇𑁭 𑀦𑀤𑀼𑀩𑁆𑀪𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀦𑀤𑀼𑀩𑁆𑀪𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(253.)

1591\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁𑁇 “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀬𑁄𑀧𑀺 𑀫𑁂 𑀅𑀲𑁆𑀲 𑀲𑀼𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀳𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀬𑁂𑀦 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀢𑀺𑀝𑁆𑀞, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺, 𑀕𑀳𑀺𑀢𑁄𑀲𑀻’𑀢𑀺𑁇

1592\. ‘𑀬𑀤𑁂𑀯 𑀢𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀢𑀤𑁂𑀯 𑀢𑁆𑀯𑀁 𑀫𑀸 𑀧𑀚𑀳𑀸𑀲𑀻’𑀢𑀺𑁇

1593\. ‘𑀲𑀧𑀲𑁆𑀲𑀼 𑀘 𑀫𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺, 𑀅𑀤𑀼𑀩𑁆𑀪𑀸𑀬𑀸’𑀢𑀺𑁇

1594\. _‘𑀬𑀁 𑀫𑀼𑀲𑀸 𑀪𑀡𑀢𑁄 𑀧𑀸𑀧𑀁,_  
_𑀬𑀁 𑀧𑀸𑀧𑀁 𑀅𑀭𑀺𑀬𑀽𑀧𑀯𑀸𑀤𑀺𑀦𑁄;_  
_𑀫𑀺𑀢𑁆𑀢𑀤𑁆𑀤𑀼𑀦𑁄 𑀘 𑀬𑀁 𑀧𑀸𑀧𑀁,_  
_𑀬𑀁 𑀧𑀸𑀧𑀁 𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀦𑁄;_  
_𑀢𑀫𑁂𑀯 𑀧𑀸𑀧𑀁 𑀨𑀼𑀲𑀢𑀼,_  
_𑀬𑁄 𑀢𑁂 𑀤𑀼𑀩𑁆𑀪𑁂 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀻’”𑀢𑀺𑁇_
