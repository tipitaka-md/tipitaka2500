# 𑁧𑁧𑁇𑁧𑁇𑁯 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑁃𑀲𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑁃𑀲𑀺𑀲𑀼𑀢𑁆𑀢

(255.)

1600\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁𑁇 “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀅𑀭𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑁂 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀻𑀲𑀼 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀘 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀘 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀧𑀘𑀺𑀢𑁆𑀢𑀺 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀧𑀝𑀮𑀺𑀬𑁄 𑀉𑀧𑀸𑀳𑀦𑀸 𑀆𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀔𑀕𑁆𑀕𑀁 𑀑𑀮𑀕𑁆𑀕𑁂𑀢𑁆𑀯𑀸 𑀙𑀢𑁆𑀢𑁂𑀦 𑀥𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂𑀦 𑀅𑀕𑁆𑀕𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀅𑀲𑁆𑀲𑀫𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑀩𑁆𑀬𑀸𑀫𑀢𑁄 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀧𑀝𑀮𑀺𑀬𑁄 𑀉𑀧𑀸𑀳𑀦𑀸 𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀔𑀕𑁆𑀕𑀁 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀤𑀢𑁆𑀯𑀸 𑀙𑀢𑁆𑀢𑀁 𑀅𑀧𑀦𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑁂𑀦𑁂𑀯 𑀅𑀲𑁆𑀲𑀫𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀼𑀯𑀸𑀢𑀁 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀦𑀫𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂 𑀇𑀲𑀬𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺𑀁𑀲𑀼—

1601\. _‘𑀕𑀦𑁆𑀥𑁄 𑀇𑀲𑀻𑀦𑀁 𑀘𑀺𑀭𑀤𑀺𑀓𑁆𑀔𑀺𑀢𑀸𑀦𑀁,_  
_𑀓𑀸𑀬𑀸 𑀘𑀼𑀢𑁄 𑀕𑀘𑁆𑀙𑀢𑀺 𑀫𑀸𑀮𑀼𑀢𑁂𑀦;_  
_𑀇𑀢𑁄 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀳𑀲𑁆𑀲𑀦𑁂𑀢𑁆𑀢,_  
_𑀕𑀦𑁆𑀥𑁄 𑀇𑀲𑀻𑀦𑀁 𑀅𑀲𑀼𑀘𑀺 𑀤𑁂𑀯𑀭𑀸𑀚𑀸’𑀢𑀺𑁇_  


1602\. _‘𑀕𑀦𑁆𑀥𑁄 𑀇𑀲𑀻𑀦𑀁 𑀘𑀺𑀭𑀤𑀺𑀓𑁆𑀔𑀺𑀢𑀸𑀦𑀁,_  
_𑀓𑀸𑀬𑀸 𑀘𑀼𑀢𑁄 𑀕𑀘𑁆𑀙𑀢𑀼 𑀫𑀸𑀮𑀼𑀢𑁂𑀦;_  
_𑀲𑀼𑀘𑀺𑀢𑁆𑀭𑀧𑀼𑀧𑁆𑀨𑀁 𑀲𑀺𑀭𑀲𑁆𑀫𑀺𑀁𑀯 𑀫𑀸𑀮𑀁,_  
_𑀕𑀦𑁆𑀥𑀁 𑀏𑀢𑀁 𑀧𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀫 𑀪𑀦𑁆𑀢𑁂;_  
_𑀦 𑀳𑁂𑀢𑁆𑀣 𑀤𑁂𑀯𑀸 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀺𑀦𑁄’”𑀢𑀺𑁇_
