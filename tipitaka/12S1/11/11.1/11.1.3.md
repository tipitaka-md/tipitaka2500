# 𑁧𑁧𑁇𑁧𑁇𑁩 𑀥𑀚𑀕𑁆𑀕𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀥𑀚𑀕𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(249.)

1545\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1546\. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀲𑀼𑀭𑀲𑀗𑁆𑀕𑀸𑀫𑁄 𑀲𑀫𑀼𑀧𑀩𑁆𑀬𑀽𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

1547\. ‘𑀲𑀘𑁂, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑀕𑀢𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀫𑀫𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀫𑀫𑀜𑁆𑀳𑀺 𑀯𑁄 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1548\. 𑀦𑁄 𑀘𑁂 𑀫𑁂 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀸𑀣, 𑀅𑀣 𑀧𑀚𑀸𑀧𑀢𑀺𑀲𑁆𑀲 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀧𑀚𑀸𑀧𑀢𑀺𑀲𑁆𑀲 𑀳𑀺 𑀯𑁄 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1549\. 𑀦𑁄 𑀘𑁂 𑀧𑀚𑀸𑀧𑀢𑀺𑀲𑁆𑀲 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀸𑀣, 𑀅𑀣 𑀯𑀭𑀼𑀡𑀲𑁆𑀲 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀯𑀭𑀼𑀡𑀲𑁆𑀲 𑀳𑀺 𑀯𑁄 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1550\. 𑀦𑁄 𑀘𑁂 𑀯𑀭𑀼𑀡𑀲𑁆𑀲 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀸𑀣, 𑀅𑀣 𑀈𑀲𑀸𑀦𑀲𑁆𑀲 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀈𑀲𑀸𑀦𑀲𑁆𑀲 𑀳𑀺 𑀯𑁄 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

1551\. 𑀢𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀯𑀸 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁, 𑀧𑀚𑀸𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁, 𑀯𑀭𑀼𑀡𑀲𑁆𑀲 𑀯𑀸 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁, 𑀈𑀲𑀸𑀦𑀲𑁆𑀲 𑀯𑀸 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀥𑀚𑀕𑁆𑀕𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑀬𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑁂𑀣𑀸𑀧𑀺 𑀦𑁄𑀧𑀺 𑀧𑀳𑀻𑀬𑁂𑀣𑁇

1552\. 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀲𑀓𑁆𑀓𑁄 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀅𑀯𑀻𑀢𑀤𑁄𑀲𑁄 𑀅𑀯𑀻𑀢𑀫𑁄𑀳𑁄 𑀪𑀻𑀭𑀼 𑀙𑀫𑁆𑀪𑀻 𑀉𑀢𑁆𑀭𑀸𑀲𑀻 𑀧𑀮𑀸𑀬𑀻𑀢𑀺𑁇

1553\. 𑀅𑀳𑀜𑁆𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀲𑀘𑁂 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑀸𑀦𑀁 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑀸𑀦𑀁 𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑀸𑀦𑀁 𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀫𑀫𑁂𑀯 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣—  “𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀫𑀫𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1554\. 𑀦𑁄 𑀘𑁂 𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣, 𑀅𑀣 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻”𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1555\. 𑀦𑁄 𑀘𑁂 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣, 𑀅𑀣 𑀲𑀁𑀖𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣—  “𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀉𑀚𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀜𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀏𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀲𑀁𑀖𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀢𑀁 𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸 𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀯𑀸, 𑀲𑁄 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1556\. 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸𑀕𑀢𑁄 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀯𑀻𑀢𑀤𑁄𑀲𑁄 𑀯𑀻𑀢𑀫𑁄𑀳𑁄 𑀅𑀪𑀻𑀭𑀼 𑀅𑀘𑁆𑀙𑀫𑁆𑀪𑀻 𑀅𑀦𑀼𑀢𑁆𑀭𑀸𑀲𑀻 𑀅𑀧𑀮𑀸𑀬𑀻’”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

1557\. _“𑀅𑀭𑀜𑁆𑀜𑁂 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀸,_  
_𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀯𑁄;_  
_𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀣 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀪𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑀸𑀓 𑀦𑁄 𑀲𑀺𑀬𑀸𑁇_  


1558\. _𑀦𑁄 𑀘𑁂 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀦𑀭𑀸𑀲𑀪𑀁;_  
_𑀅𑀣 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀲𑀼𑀤𑁂𑀲𑀺𑀢𑀁𑁇_  


1559\. _𑀦𑁄 𑀘𑁂 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀁 𑀲𑀼𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀅𑀣 𑀲𑀁𑀖𑀁 𑀲𑀭𑁂𑀬𑁆𑀬𑀸𑀣,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


1560\. _𑀏𑀯𑀁 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀖𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀯𑁄;_  
_𑀪𑀬𑀁 𑀯𑀸 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀯𑀸,_  
_𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀦 𑀳𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_
