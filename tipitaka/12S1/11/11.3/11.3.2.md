# 𑁧𑁧𑁇𑁩𑁇𑁨 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀺𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(268.)

1686\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀚𑁂𑀢𑀯𑀦𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄…𑀧𑁂…  𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀬𑀓𑁆𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀑𑀓𑁄𑀝𑀺𑀫𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀑𑀓𑁄𑀝𑀺𑀫𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀬𑀓𑁆𑀔𑁄 𑀅𑀪𑀺𑀭𑀽𑀧𑀢𑀭𑁄 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀢𑀭𑁄 𑀘 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀢𑀭𑁄 𑀘𑁇

1687\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀬𑁂𑀦 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀇𑀥 𑀢𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀬𑀓𑁆𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀑𑀓𑁄𑀝𑀺𑀫𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀫𑀸𑀭𑀺𑀲, 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀑𑀓𑁄𑀝𑀺𑀫𑀓𑁄 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀤𑁂𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀬𑀓𑁆𑀔𑁄 𑀅𑀪𑀺𑀭𑀽𑀧𑀢𑀭𑁄 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀢𑀭𑁄 𑀘 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀢𑀭𑁄 𑀘𑀸𑀢𑀺𑁇 𑀲𑁄 𑀳𑀺 𑀦𑀽𑀦, 𑀫𑀸𑀭𑀺𑀲, 𑀓𑁄𑀥𑀪𑀓𑁆𑀔𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇

1688\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀲𑁄 𑀓𑁄𑀥𑀪𑀓𑁆𑀔𑁄 𑀬𑀓𑁆𑀔𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀧𑀣𑀯𑀺𑀬𑀁 𑀦𑀺𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀓𑁄𑀥𑀪𑀓𑁆𑀔𑁄 𑀬𑀓𑁆𑀔𑁄 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀦𑀸𑀫𑀁 𑀲𑀸𑀯𑁂𑀢𑀺—  ‘𑀲𑀓𑁆𑀓𑁄𑀳𑀁, 𑀫𑀸𑀭𑀺𑀲, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄, 𑀲𑀓𑁆𑀓𑁄𑀳𑀁, 𑀫𑀸𑀭𑀺𑀲, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄’𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀦𑀸𑀫𑀁 𑀲𑀸𑀯𑁂𑀲𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀬𑀓𑁆𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀢𑀭𑁄 𑀘𑁂𑀯 𑀅𑀳𑁄𑀲𑀺 𑀑𑀓𑁄𑀝𑀺𑀫𑀓𑀢𑀭𑁄 𑀘𑁇 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀢𑀭𑁄 𑀘𑁂𑀯 𑀳𑀼𑀢𑁆𑀯𑀸 𑀑𑀓𑁄𑀝𑀺𑀫𑀓𑀢𑀭𑁄 𑀘 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀓𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀅𑀦𑀼𑀦𑀬𑀫𑀸𑀦𑁄 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

1689\. _‘𑀦 𑀲𑀽𑀧𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺,_  
_𑀦𑀸𑀯𑀢𑁆𑀢𑁂𑀦 𑀲𑀼𑀯𑀸𑀦𑀬𑁄;_  
_𑀦 𑀯𑁄 𑀘𑀺𑀭𑀸𑀳𑀁 𑀓𑀼𑀚𑁆𑀛𑀸𑀫𑀺,_  
_𑀓𑁄𑀥𑁄 𑀫𑀬𑀺 𑀦𑀸𑀯𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


1690\. _𑀓𑀼𑀤𑁆𑀥𑀸𑀳𑀁 𑀦 𑀨𑀭𑀼𑀲𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀺𑀢𑁆𑀢𑀬𑁂;_  
_𑀲𑀦𑁆𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁 𑀅𑀢𑁆𑀣𑀫𑀢𑁆𑀢𑀦𑁄’”𑀢𑀺𑁇_
