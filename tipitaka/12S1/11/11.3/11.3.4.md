# 𑁧𑁧𑁇𑁩𑁇𑁪 𑀅𑀘𑁆𑀘𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀓𑁆𑀓𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀅𑀘𑁆𑀘𑀬𑀲𑀼𑀢𑁆𑀢

(270.)

1693\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁…𑀧𑁂…  𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑁆𑀧𑀬𑁄𑀚𑁂𑀲𑀼𑀁𑁇 𑀢𑀢𑁆𑀭𑁂𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀲𑀭𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀤𑁂𑀲𑁂𑀢𑀺; 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀧𑁆𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑁆𑀧𑀬𑁄𑀚𑁂𑀲𑀼𑀁, 𑀢𑀢𑁆𑀭𑁂𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀘𑀲𑀭𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀧𑁆𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀻”𑀢𑀺𑁇

1694\. “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀮𑀸𑁇 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀩𑀸𑀮𑀸𑁇 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀅𑀘𑁆𑀘𑀬𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀬𑁄 𑀘 𑀅𑀘𑁆𑀘𑀬𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇

1695\. 𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀲𑀪𑀸𑀬𑀁 𑀤𑁂𑀯𑁂 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂 𑀅𑀦𑀼𑀦𑀬𑀫𑀸𑀦𑁄 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1696\. _‘𑀓𑁄𑀥𑁄 𑀯𑁄 𑀯𑀲𑀫𑀸𑀬𑀸𑀢𑀼,_  
_𑀫𑀸 𑀘 𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀯𑁄 𑀚𑀭𑀸;_  
_𑀅𑀕𑀭𑀳𑀺𑀬𑀁 𑀫𑀸 𑀕𑀭𑀳𑀺𑀢𑁆𑀣,_  
_𑀫𑀸 𑀘 𑀪𑀸𑀲𑀺𑀢𑁆𑀣 𑀧𑁂𑀲𑀼𑀡𑀁;_  
_𑀅𑀣 𑀧𑀸𑀧𑀚𑀦𑀁 𑀓𑁄𑀥𑁄,_  
_𑀧𑀩𑁆𑀩𑀢𑁄𑀯𑀸𑀪𑀺𑀫𑀤𑁆𑀤𑀢𑀻’”𑀢𑀺𑁇_
