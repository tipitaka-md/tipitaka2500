# 𑁨𑁇𑁧𑁇𑁪 𑀫𑀸𑀕𑀥𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀸𑀕𑀥𑀲𑀼𑀢𑁆𑀢

(85.)

379\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀫𑀸𑀕𑀥𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

380\. _“𑀓𑀢𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀧𑀚𑁆𑀚𑁄𑀢𑀸,_  
_𑀬𑁂𑀳𑀺 𑀮𑁄𑀓𑁄 𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀪𑀯𑀦𑁆𑀢𑀁 𑀧𑀼𑀝𑁆𑀞𑀼𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇_  


381\. _“𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀚𑁆𑀚𑁄𑀢𑀸,_  
_𑀧𑀜𑁆𑀘𑀫𑁂𑀢𑁆𑀣 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀤𑀺𑀯𑀸 𑀢𑀧𑀢𑀺 𑀆𑀤𑀺𑀘𑁆𑀘𑁄,_  
_𑀭𑀢𑁆𑀢𑀺𑀫𑀸𑀪𑀸𑀢𑀺 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑁇_  


382\. _𑀅𑀣 𑀅𑀕𑁆𑀕𑀺 𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑀧𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀏𑀲𑀸 𑀆𑀪𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸”𑀢𑀺𑁇_
