# 𑁨𑁇𑁩𑁇𑁨 𑀔𑁂𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀔𑁂𑀫𑀲𑀼𑀢𑁆𑀢

(103.)

487. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀔𑁂𑀫𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

488. _“𑀘𑀭𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑀸 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑀸,_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑁂𑀦𑁂𑀯 𑀅𑀢𑁆𑀢𑀦𑀸;_  
_𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀧𑀸𑀧𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀬𑀁 𑀳𑁄𑀢𑀺 𑀓𑀝𑀼𑀓𑀧𑁆𑀨𑀮𑀁𑁇_  


489. _𑀦 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀲𑀸𑀥𑀼,_  
_𑀬𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀺;_  
_𑀬𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼𑀫𑀼𑀔𑁄 𑀭𑁄𑀤𑀁,_  
_𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺𑁇_  


490. _𑀢𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀲𑀸𑀥𑀼,_  
_𑀬𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀺;_  
_𑀬𑀲𑁆𑀲 𑀧𑀢𑀻𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺𑁇_  


491. _𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀢𑀁 𑀓𑀬𑀺𑀭𑀸,_  
_𑀬𑀁 𑀚𑀜𑁆𑀜𑀸 𑀳𑀺𑀢𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀦 𑀲𑀸𑀓𑀝𑀺𑀓𑀘𑀺𑀦𑁆𑀢𑀸𑀬,_  
_𑀫𑀦𑁆𑀢𑀸 𑀥𑀻𑀭𑁄 𑀧𑀭𑀓𑁆𑀓𑀫𑁂𑁇_  


492. _𑀬𑀣𑀸 𑀲𑀸𑀓𑀝𑀺𑀓𑁄 𑀫𑀝𑁆𑀞𑀁,_  
_𑀲𑀫𑀁 𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀧𑀣𑀁;_  
_𑀯𑀺𑀲𑀫𑀁 𑀫𑀕𑁆𑀕𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀅𑀓𑁆𑀔𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄𑀯 𑀛𑀸𑀬𑀢𑀺𑁇_  


493. _𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀅𑀥𑀫𑁆𑀫𑀫𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀬;_  
_𑀫𑀦𑁆𑀤𑁄 𑀫𑀘𑁆𑀘𑀼𑀫𑀼𑀔𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀅𑀓𑁆𑀔𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄𑀯 𑀛𑀸𑀬𑀢𑀻”𑀢𑀺𑁇_
