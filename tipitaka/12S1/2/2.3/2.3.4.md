# 𑁨𑁇𑁩𑁇𑁪 𑀖𑀝𑀻𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀖𑀝𑀻𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(105.)

508. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀖𑀝𑀻𑀓𑀸𑀭𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

509. _“𑀅𑀯𑀺𑀳𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑀸𑀲𑁂,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀲𑀢𑁆𑀢 𑀪𑀺𑀓𑁆𑀔𑀯𑁄;_  
_𑀭𑀸𑀕𑀤𑁄𑀲𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀢𑀺𑀡𑁆𑀡𑀸 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀦𑁆”𑀢𑀺𑁇_  


510. _“𑀓𑁂 𑀘 𑀢𑁂 𑀅𑀢𑀭𑀼𑀁 𑀧𑀗𑁆𑀓𑀁,_  
_𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀤𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀓𑁂 𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀬𑁄𑀕𑀁 𑀉𑀧𑀘𑁆𑀘𑀕𑀼𑀦𑁆”𑀢𑀺𑁇_  


511. _“𑀉𑀧𑀓𑁄 𑀧𑀮𑀕𑀡𑁆𑀟𑁄 𑀘,_  
_𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀸𑀢𑀺 𑀘 𑀢𑁂 𑀢𑀬𑁄;_  
_𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀔𑀡𑁆𑀟𑀤𑁂𑀯𑁄 𑀘,_  
_𑀩𑀸𑀳𑀼𑀭𑀕𑁆𑀕𑀺 𑀘 𑀲𑀗𑁆𑀕𑀺𑀬𑁄;_  
_𑀢𑁂 𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀬𑁄𑀕𑀁 𑀉𑀧𑀘𑁆𑀘𑀕𑀼𑀦𑁆”𑀢𑀺𑁇_  


512. _“𑀓𑀼𑀲𑀮𑀻 𑀪𑀸𑀲𑀲𑀻 𑀢𑁂𑀲𑀁,_  
_𑀫𑀸𑀭𑀧𑀸𑀲𑀧𑁆𑀧𑀳𑀸𑀬𑀺𑀦𑀁;_  
_𑀓𑀲𑁆𑀲 𑀢𑁂 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀘𑁆𑀙𑀺𑀤𑀼𑀁 𑀪𑀯𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


513. _“𑀦 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀪𑀕𑀯𑀢𑀸,_  
_𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀯 𑀲𑀸𑀲𑀦𑀸;_  
_𑀬𑀲𑁆𑀲 𑀢𑁂 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀘𑁆𑀙𑀺𑀤𑀼𑀁 𑀪𑀯𑀩𑀦𑁆𑀥𑀦𑀁𑁇_  


514. _𑀬𑀢𑁆𑀣 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘,_  
_𑀅𑀲𑁂𑀲𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀢𑀁 𑀢𑁂 𑀥𑀫𑁆𑀫𑀁 𑀇𑀥𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀘𑁆𑀙𑀺𑀤𑀼𑀁 𑀪𑀯𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


515. _“𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀪𑀸𑀲𑀲𑀻 𑀯𑀸𑀘𑀁,_  
_𑀤𑀼𑀩𑁆𑀩𑀺𑀚𑀸𑀦𑀁 𑀲𑀼𑀤𑀼𑀩𑁆𑀩𑀼𑀥𑀁;_  
_𑀓𑀲𑁆𑀲 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀲𑀺 𑀈𑀤𑀺𑀲𑀦𑁆”𑀢𑀺𑁇_  


516. _“𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀯𑁂𑀓𑀍𑀇𑀗𑁆𑀕𑁂 𑀖𑀝𑀻𑀓𑀭𑁄;_  
_𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑁄 𑀆𑀲𑀺𑀁,_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑁄𑁇_  


517. _𑀯𑀺𑀭𑀢𑁄 𑀫𑁂𑀣𑀼𑀦𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁄;_  
_𑀅𑀳𑀼𑀯𑀸 𑀢𑁂 𑀲𑀕𑀸𑀫𑁂𑀬𑁆𑀬𑁄,_  
_𑀅𑀳𑀼𑀯𑀸 𑀢𑁂 𑀧𑀼𑀭𑁂 𑀲𑀔𑀸𑁇_  


518. _𑀲𑁄𑀳𑀫𑁂𑀢𑁂 𑀧𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢 𑀪𑀺𑀓𑁆𑀔𑀯𑁄;_  
_𑀭𑀸𑀕𑀤𑁄𑀲𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁂,_  
_𑀢𑀺𑀡𑁆𑀡𑁂 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀦𑁆”𑀢𑀺𑁇_  


519. _“𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀤𑀸 𑀆𑀲𑀺,_  
_𑀬𑀣𑀸 𑀪𑀸𑀲𑀲𑀺 𑀪𑀕𑁆𑀕𑀯;_  
_𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺,_  
_𑀯𑁂𑀓𑀍𑀇𑀗𑁆𑀕𑁂 𑀖𑀝𑀻𑀓𑀭𑁄𑁇_  


520. _𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑁄 𑀆𑀲𑀺,_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑁄;_  
_𑀯𑀺𑀭𑀢𑁄 𑀫𑁂𑀣𑀼𑀦𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁄;_  
_𑀅𑀳𑀼𑀯𑀸 𑀫𑁂 𑀲𑀕𑀸𑀫𑁂𑀬𑁆𑀬𑁄,_  
_𑀅𑀳𑀼𑀯𑀸 𑀫𑁂 𑀧𑀼𑀭𑁂 𑀲𑀔𑀸”𑀢𑀺𑁇_  


521. _“𑀏𑀯𑀫𑁂𑀢𑀁 𑀧𑀼𑀭𑀸𑀡𑀸𑀦𑀁,_  
_𑀲𑀳𑀸𑀬𑀸𑀦𑀁 𑀅𑀳𑀼 𑀲𑀗𑁆𑀕𑀫𑁄;_  
_𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀲𑀭𑀻𑀭𑀦𑁆𑀢𑀺𑀫𑀥𑀸𑀭𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_
