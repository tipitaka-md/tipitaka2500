# 𑁨𑁇𑁩𑁇𑁮 𑀦𑀦𑁆𑀤𑀺𑀯𑀺𑀲𑀸𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀺𑀯𑀺𑀲𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(109.)

538\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀦𑀦𑁆𑀤𑀺𑀯𑀺𑀲𑀸𑀮𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

539\. _“𑀘𑀢𑀼𑀘𑀓𑁆𑀓𑀁 𑀦𑀯𑀤𑁆𑀯𑀸𑀭𑀁,_  
_𑀧𑀼𑀡𑁆𑀡𑀁 𑀮𑁄𑀪𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑀁;_  
_𑀧𑀗𑁆𑀓𑀚𑀸𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀓𑀣𑀁 𑀬𑀸𑀢𑁆𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


540\. _“𑀙𑁂𑀢𑁆𑀯𑀸 𑀦𑀤𑁆𑀥𑀺𑀁 𑀯𑀭𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀇𑀘𑁆𑀙𑀸𑀮𑁄𑀪𑀜𑁆𑀘 𑀧𑀸𑀧𑀓𑀁;_  
_𑀲𑀫𑀽𑀮𑀁 𑀢𑀡𑁆𑀳𑀫𑀩𑁆𑀩𑀼𑀬𑁆𑀳,_  
_𑀏𑀯𑀁 𑀬𑀸𑀢𑁆𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_
