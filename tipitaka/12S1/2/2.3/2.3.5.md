# 𑁨𑁇𑁩𑁇𑁫 𑀚𑀦𑁆𑀢𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀚𑀦𑁆𑀢𑀼𑀲𑀼𑀢𑁆𑀢

(106.)

522\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀧𑀲𑁆𑀲𑁂 𑀅𑀭𑀜𑁆𑀜𑀓𑀼𑀝𑀺𑀓𑀸𑀬 𑀉𑀤𑁆𑀥𑀢𑀸 𑀉𑀦𑁆𑀦𑀍𑀆 𑀘𑀧𑀮𑀸 𑀫𑀼𑀔𑀭𑀸 𑀯𑀺𑀓𑀺𑀡𑁆𑀡𑀯𑀸𑀘𑀸 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀦𑁄 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀸 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀸𑀓𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑁇

523\. 𑀅𑀣 𑀔𑁄 𑀚𑀦𑁆𑀢𑀼 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

524\. _“𑀲𑀼𑀔𑀚𑀻𑀯𑀺𑀦𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀼𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑁄𑀢𑀫𑀲𑀸𑀯𑀓𑀸;_  
_𑀅𑀦𑀺𑀘𑁆𑀙𑀸 𑀧𑀺𑀡𑁆𑀟𑀫𑁂𑀲𑀦𑀸,_  
_𑀅𑀦𑀺𑀘𑁆𑀙𑀸 𑀲𑀬𑀦𑀸𑀲𑀦𑀁;_  
_𑀮𑁄𑀓𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


525\. _𑀤𑀼𑀧𑁆𑀧𑁄𑀲𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀕𑀸𑀫𑁂 𑀕𑀸𑀫𑀡𑀺𑀓𑀸 𑀯𑀺𑀬;_  
_𑀪𑀼𑀢𑁆𑀯𑀸 𑀪𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀧𑀭𑀸𑀕𑀸𑀭𑁂𑀲𑀼 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸𑁇_  


526\. _𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀤𑀸𑀫𑀳𑀁;_  
_𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀸 𑀅𑀦𑀸𑀣𑀸 𑀢𑁂,_  
_𑀬𑀣𑀸 𑀧𑁂𑀢𑀸 𑀢𑀣𑁂𑀯 𑀢𑁂𑁇_  


527\. _𑀬𑁂 𑀔𑁄 𑀧𑀫𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀢𑁂 𑀫𑁂 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀁;_  
_𑀬𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀦𑀫𑁄 𑀢𑁂𑀲𑀁 𑀓𑀭𑁄𑀫𑀳𑀦𑁆”𑀢𑀺𑁇_
