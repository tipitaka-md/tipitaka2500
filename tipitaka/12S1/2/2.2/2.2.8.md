# 𑁨𑁇𑁨𑁇𑁮 𑀓𑀓𑀼𑀥𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀓𑀓𑀼𑀥𑀲𑀼𑀢𑁆𑀢

(99.)

453\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀓𑁂𑀢𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀚𑀦𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀓𑀓𑀼𑀥𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑁄 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀅𑀜𑁆𑀚𑀦𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀓𑀓𑀼𑀥𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀦𑁆𑀤𑀲𑀺, 𑀲𑀫𑀡𑀸”𑀢𑀺? “𑀓𑀺𑀁 𑀮𑀤𑁆𑀥𑀸, 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑀡, 𑀲𑁄𑀘𑀲𑀻”𑀢𑀺? “𑀓𑀺𑀁 𑀚𑀻𑀬𑀺𑀢𑁆𑀣, 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺, 𑀲𑀫𑀡, 𑀦𑁂𑀯 𑀦𑀦𑁆𑀤𑀲𑀺 𑀦 𑀘 𑀲𑁄𑀘𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇

454\. _“𑀓𑀘𑁆𑀘𑀺 𑀢𑁆𑀯𑀁 𑀅𑀦𑀖𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀓𑀘𑁆𑀘𑀺 𑀦𑀦𑁆𑀤𑀻 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀓𑀘𑁆𑀘𑀺 𑀢𑀁 𑀏𑀓𑀫𑀸𑀲𑀻𑀦𑀁,_  
_𑀅𑀭𑀢𑀻 𑀦𑀸𑀪𑀺𑀓𑀻𑀭𑀢𑀻”𑀢𑀺𑁇_  


455\. _“𑀅𑀦𑀖𑁄 𑀯𑁂 𑀅𑀳𑀁 𑀬𑀓𑁆𑀔,_  
_𑀅𑀣𑁄 𑀦𑀦𑁆𑀤𑀻 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀣𑁄 𑀫𑀁 𑀏𑀓𑀫𑀸𑀲𑀻𑀦𑀁,_  
_𑀅𑀭𑀢𑀻 𑀦𑀸𑀪𑀺𑀓𑀻𑀭𑀢𑀻”𑀢𑀺𑁇_  


456\. _“𑀓𑀣𑀁 𑀢𑁆𑀯𑀁 𑀅𑀦𑀖𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀓𑀣𑀁 𑀦𑀦𑁆𑀤𑀻 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀓𑀣𑀁 𑀢𑀁 𑀏𑀓𑀫𑀸𑀲𑀻𑀦𑀁,_  
_𑀅𑀭𑀢𑀻 𑀦𑀸𑀪𑀺𑀓𑀻𑀭𑀢𑀻”𑀢𑀺𑁇_  


457\. _“𑀅𑀖𑀚𑀸𑀢𑀲𑁆𑀲 𑀯𑁂 𑀦𑀦𑁆𑀤𑀻,_  
_𑀦𑀦𑁆𑀤𑀻𑀚𑀸𑀢𑀲𑁆𑀲 𑀯𑁂 𑀅𑀖𑀁;_  
_𑀅𑀦𑀦𑁆𑀤𑀻 𑀅𑀦𑀖𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀆𑀯𑀼𑀲𑁄”𑀢𑀺𑁇_  


458\. _“𑀘𑀺𑀭𑀲𑁆𑀲𑀁 𑀯𑀢 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀁;_  
_𑀅𑀦𑀦𑁆𑀤𑀺𑀁 𑀅𑀦𑀖𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀦𑁆”𑀢𑀺𑁇_
