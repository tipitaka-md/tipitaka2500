# 𑁨𑁇𑁨𑁇𑁫 𑀘𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀘𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

(96.)

442\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀘𑀦𑁆𑀤𑀦𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

443\. _“𑀓𑀣𑀁𑀲𑀼 𑀢𑀭𑀢𑀺 𑀑𑀖𑀁,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑁄;_  
_𑀅𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑁂 𑀅𑀦𑀸𑀮𑀫𑁆𑀩𑁂,_  
_𑀓𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀦 𑀲𑀻𑀤𑀢𑀻”𑀢𑀺𑁇_  


444\. _“𑀲𑀩𑁆𑀩𑀤𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀧𑀜𑁆𑀜𑀯𑀸 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄,_  
_𑀑𑀖𑀁 𑀢𑀭𑀢𑀺 𑀤𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


445\. _𑀯𑀺𑀭𑀢𑁄 𑀓𑀸𑀫𑀲𑀜𑁆𑀜𑀸𑀬,_  
_𑀭𑀽𑀧𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀢𑀺𑀕𑁄;_  
_𑀦𑀦𑁆𑀤𑀻𑀭𑀸𑀕𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑁄,_  
_𑀲𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀦 𑀲𑀻𑀤𑀢𑀻”𑀢𑀺𑁇_
