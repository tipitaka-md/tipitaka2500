# 𑁭𑁇𑁧𑁇𑁩 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀓𑀲𑀼𑀢𑁆𑀢

(189.)

1150\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄—  “𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑀕𑁄𑀢𑁆𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀓𑀺𑀭 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀚𑀺𑀢𑁄𑀲𑀺, 𑀲𑀫𑀡, 𑀚𑀺𑀢𑁄𑀲𑀺, 𑀲𑀫𑀡𑀸”𑀢𑀺𑁇

1151\. _“𑀚𑀬𑀁 𑀯𑁂 𑀫𑀜𑁆𑀜𑀢𑀺 𑀩𑀸𑀮𑁄,_  
_𑀯𑀸𑀘𑀸𑀬 𑀨𑀭𑀼𑀲𑀁 𑀪𑀡𑀁;_  
_𑀚𑀬𑀜𑁆𑀘𑁂𑀯𑀲𑁆𑀲 𑀢𑀁 𑀳𑁄𑀢𑀺,_  
_𑀬𑀸 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀸 𑀯𑀺𑀚𑀸𑀦𑀢𑁄𑁇_  


1152\. _𑀢𑀲𑁆𑀲𑁂𑀯 𑀢𑁂𑀦 𑀧𑀸𑀧𑀺𑀬𑁄,_  
_𑀬𑁄 𑀓𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀓𑀼𑀤𑁆𑀥𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑁄,_  
_𑀲𑀗𑁆𑀕𑀸𑀫𑀁 𑀚𑁂𑀢𑀺 𑀤𑀼𑀚𑁆𑀚𑀬𑀁𑁇_  


1153\. _𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀭𑀲𑁆𑀲 𑀘;_  
_𑀧𑀭𑀁 𑀲𑀗𑁆𑀓𑀼𑀧𑀺𑀢𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀬𑁄 𑀲𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


1154\. _𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀭𑀲𑁆𑀲 𑀘;_  
_𑀚𑀦𑀸 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑁄𑀢𑀺,_  
_𑀬𑁂 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀓𑁄𑀯𑀺𑀤𑀸”𑀢𑀺𑁇_  


1155\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇
