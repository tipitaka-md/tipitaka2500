# 𑁭𑁇𑁧𑁇𑁭 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(193.)

1168\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

1169\. _“𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑀼𑀚𑁆𑀛𑀢𑀺 𑀓𑁄𑀘𑀺,_  
_𑀮𑁄𑀓𑁂 𑀲𑀻𑀮𑀯𑀸𑀧𑀺 𑀢𑀧𑁄𑀓𑀭𑀁;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑁄 𑀲𑀼𑀚𑁆𑀛𑀢𑀺 𑀦 𑀅𑀜𑁆𑀜𑀸 𑀇𑀢𑀭𑀸 𑀧𑀚𑀸”𑀢𑀺𑁇_  


1170\. _“𑀩𑀳𑀼𑀫𑁆𑀧𑀺 𑀧𑀮𑀧𑀁 𑀚𑀧𑁆𑀧𑀁,_  
_𑀦 𑀚𑀘𑁆𑀘𑀸 𑀳𑁄𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀅𑀦𑁆𑀢𑁄𑀓𑀲𑀫𑁆𑀩𑀼 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁄,_  
_𑀓𑀼𑀳𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄𑁇_  


1171\. _𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑁂𑀲𑁆𑀲𑁄,_  
_𑀲𑀼𑀤𑁆𑀤𑁄 𑀘𑀡𑁆𑀟𑀸𑀮𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁄;_  
_𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄;_  
_𑀧𑀧𑁆𑀧𑁄𑀢𑀺 𑀧𑀭𑀫𑀁 𑀲𑀼𑀤𑁆𑀥𑀺𑀁,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸”𑀢𑀺𑁇_  


1172\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇
