# 𑁭𑁇𑁨𑁇𑁪 𑀫𑀳𑀸𑀲𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀉𑀧𑀸𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀲𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(200.)

1247\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁄 𑀮𑀽𑀔𑁄 𑀮𑀽𑀔𑀧𑀸𑀯𑀼𑀭𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀮𑀽𑀔𑁄 𑀮𑀽𑀔𑀧𑀸𑀯𑀼𑀭𑀡𑁄”𑀢𑀺? “𑀇𑀥 𑀫𑁂, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀢𑁆𑀢𑀸𑁇 𑀢𑁂 𑀫𑀁 𑀤𑀸𑀭𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀼𑀘𑁆𑀙 𑀖𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀲𑀪𑀸𑀬𑀁 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀘 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀪𑀸𑀲𑀲𑁆𑀲𑀼—

1248\. _‘𑀬𑁂𑀳𑀺 𑀚𑀸𑀢𑁂𑀳𑀺 𑀦𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀁,_  
_𑀬𑁂𑀲𑀜𑁆𑀘 𑀪𑀯𑀫𑀺𑀘𑁆𑀙𑀺𑀲𑀁;_  
_𑀢𑁂 𑀫𑀁 𑀤𑀸𑀭𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀼𑀘𑁆𑀙,_  
_𑀲𑀸𑀯 𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀲𑀽𑀓𑀭𑀁𑁇_  


1249\. _𑀅𑀲𑀦𑁆𑀢𑀸 𑀓𑀺𑀭 𑀫𑀁 𑀚𑀫𑁆𑀫𑀸,_  
_𑀢𑀸𑀢 𑀢𑀸𑀢𑀸𑀢𑀺 𑀪𑀸𑀲𑀭𑁂;_  
_𑀭𑀓𑁆𑀔𑀲𑀸 𑀧𑀼𑀢𑁆𑀢𑀭𑀽𑀧𑁂𑀦,_  
_𑀢𑁂 𑀚𑀳𑀦𑁆𑀢𑀺 𑀯𑀬𑁄𑀕𑀢𑀁𑁇_  


1250\. _𑀅𑀲𑁆𑀲𑁄𑀯 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀺𑀩𑁆𑀪𑁄𑀕𑁄,_  
_𑀔𑀸𑀤𑀦𑀸 𑀅𑀧𑀦𑀻𑀬𑀢𑀺;_  
_𑀩𑀸𑀮𑀓𑀸𑀦𑀁 𑀧𑀺𑀢𑀸 𑀣𑁂𑀭𑁄,_  
_𑀧𑀭𑀸𑀕𑀸𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀢𑀺𑁇_  


1251\. _𑀤𑀡𑁆𑀟𑁄𑀯 𑀓𑀺𑀭 𑀫𑁂 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑀜𑁆𑀘𑁂 𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀲𑁆𑀲𑀯𑀸;_  
_𑀘𑀡𑁆𑀟𑀫𑁆𑀧𑀺 𑀕𑁄𑀡𑀁 𑀯𑀸𑀭𑁂𑀢𑀺,_  
_𑀅𑀣𑁄 𑀘𑀡𑁆𑀟𑀫𑁆𑀧𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀁𑁇_  


1252\. _𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀧𑀼𑀭𑁂 𑀳𑁄𑀢𑀺,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀕𑀸𑀥𑀫𑁂𑀥𑀢𑀺;_  
_𑀤𑀡𑁆𑀟𑀲𑁆𑀲 𑀆𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀔𑀮𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀢𑀺𑀝𑁆𑀞𑀢𑀻’”𑀢𑀺𑁇_  


1253\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀲𑀪𑀸𑀬𑀁 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁂 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀘 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀅𑀪𑀸𑀲𑀺—

1254\. _“𑀬𑁂𑀳𑀺 𑀚𑀸𑀢𑁂𑀳𑀺 𑀦𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀁,_  
_𑀬𑁂𑀲𑀜𑁆𑀘 𑀪𑀯𑀫𑀺𑀘𑁆𑀙𑀺𑀲𑀁;_  
_𑀢𑁂 𑀫𑀁 𑀤𑀸𑀭𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀼𑀘𑁆𑀙,_  
_𑀲𑀸𑀯 𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀲𑀽𑀓𑀭𑀁𑁇_  


1255\. _𑀅𑀲𑀦𑁆𑀢𑀸 𑀓𑀺𑀭 𑀫𑀁 𑀚𑀫𑁆𑀫𑀸,_  
_𑀢𑀸𑀢 𑀢𑀸𑀢𑀸𑀢𑀺 𑀪𑀸𑀲𑀭𑁂;_  
_𑀭𑀓𑁆𑀔𑀲𑀸 𑀧𑀼𑀢𑁆𑀢𑀭𑀽𑀧𑁂𑀦,_  
_𑀢𑁂 𑀚𑀳𑀦𑁆𑀢𑀺 𑀯𑀬𑁄𑀕𑀢𑀁𑁇_  


1256\. _𑀅𑀲𑁆𑀲𑁄𑀯 𑀚𑀺𑀡𑁆𑀡𑁄 𑀦𑀺𑀩𑁆𑀪𑁄𑀕𑁄,_  
_𑀔𑀸𑀤𑀦𑀸 𑀅𑀧𑀦𑀻𑀬𑀢𑀺;_  
_𑀩𑀸𑀮𑀓𑀸𑀦𑀁 𑀧𑀺𑀢𑀸 𑀣𑁂𑀭𑁄,_  
_𑀧𑀭𑀸𑀕𑀸𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀢𑀺𑁇_  


1257\. _𑀤𑀡𑁆𑀟𑁄𑀯 𑀓𑀺𑀭 𑀫𑁂 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑀜𑁆𑀘𑁂 𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀲𑁆𑀲𑀯𑀸;_  
_𑀘𑀡𑁆𑀟𑀫𑁆𑀧𑀺 𑀕𑁄𑀡𑀁 𑀯𑀸𑀭𑁂𑀢𑀺,_  
_𑀅𑀣𑁄 𑀘𑀡𑁆𑀟𑀫𑁆𑀧𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀁𑁇_  


1258\. _𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀧𑀼𑀭𑁂 𑀳𑁄𑀢𑀺,_  
_𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀕𑀸𑀥𑀫𑁂𑀥𑀢𑀺;_  
_𑀤𑀡𑁆𑀟𑀲𑁆𑀲 𑀆𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀔𑀮𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑀺𑀢𑀺𑀝𑁆𑀞𑀢𑀻”𑀢𑀺𑁇_  


1259\. 𑀅𑀣 𑀔𑁄 𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀁 𑀧𑀼𑀢𑁆𑀢𑀸 𑀖𑀭𑀁 𑀦𑁂𑀢𑁆𑀯𑀸 𑀦𑁆𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑁂𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑁂𑀦 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁄 𑀏𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀦𑀸𑀫 𑀆𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀆𑀘𑀭𑀺𑀬𑀥𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀸𑀫𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀢𑀼 𑀫𑁂 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀆𑀘𑀭𑀺𑀬𑀥𑀦𑀦𑁆”𑀢𑀺𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
