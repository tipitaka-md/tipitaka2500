# 𑁫𑁇𑁧𑁇𑁬 𑀘𑀸𑀮𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀯𑀕𑁆𑀕

* 𑀘𑀸𑀮𑀸𑀲𑀼𑀢𑁆𑀢

(167.)

972\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀸𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀬𑁂𑀦 𑀘𑀸𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀮𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺, 𑀦 𑀭𑁄𑀘𑁂𑀲𑀻”𑀢𑀺? “𑀚𑀸𑀢𑀺𑀁 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀦 𑀭𑁄𑀘𑁂𑀫𑀻”𑀢𑀺𑁇

973\. _“𑀓𑀺𑀁 𑀦𑀼 𑀚𑀸𑀢𑀺𑀁 𑀦 𑀭𑁄𑀘𑁂𑀲𑀺,_  
_𑀚𑀸𑀢𑁄 𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀢𑀺;_  
_𑀓𑁄 𑀦𑀼 𑀢𑀁 𑀇𑀤𑀫𑀸𑀤𑀧𑀬𑀺,_  
_𑀚𑀸𑀢𑀺𑀁 𑀫𑀸 𑀭𑁄𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻”𑀢𑀺𑁇_  


974\. _“𑀚𑀸𑀢𑀲𑁆𑀲 𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺,_  
_𑀚𑀸𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺 𑀨𑀼𑀲𑁆𑀲𑀢𑀺;_  
_𑀩𑀦𑁆𑀥𑀁 𑀯𑀥𑀁 𑀧𑀭𑀺𑀓𑁆𑀮𑁂𑀲𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀚𑀸𑀢𑀺𑀁 𑀦 𑀭𑁄𑀘𑀬𑁂𑁇_  


975\. _𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀁;_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬,_  
_𑀲𑁄 𑀫𑀁 𑀲𑀘𑁆𑀘𑁂 𑀦𑀺𑀯𑁂𑀲𑀬𑀺𑁇_  


976\. _𑀬𑁂 𑀘 𑀭𑀽𑀧𑀽𑀧𑀕𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀬𑁂 𑀘 𑀅𑀭𑀽𑀧𑀝𑁆𑀞𑀸𑀬𑀺𑀦𑁄;_  
_𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀸,_  
_𑀆𑀕𑀦𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇_  


977\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 “𑀚𑀸𑀦𑀸𑀢𑀺 𑀫𑀁 𑀘𑀸𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻”𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇
