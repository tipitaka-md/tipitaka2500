# 𑁫𑁇𑁧𑁇𑁧 𑀆𑀍𑀅𑀯𑀺𑀓𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀯𑀕𑁆𑀕

### 𑀆𑀍𑀅𑀯𑀺𑀓𑀸𑀲𑀼𑀢𑁆𑀢

(162.)

939\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀍𑀅𑀯𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀅𑀦𑁆𑀥𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀯𑀺𑀯𑁂𑀓𑀢𑁆𑀣𑀺𑀦𑀻𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀆𑀍𑀅𑀯𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀯𑀺𑀯𑁂𑀓𑀫𑁆𑀳𑀸 𑀘𑀸𑀯𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀬𑁂𑀦 𑀆𑀍𑀅𑀯𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀍𑀅𑀯𑀺𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

940\. _“𑀦𑀢𑁆𑀣𑀺 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀮𑁄𑀓𑁂,_  
_𑀓𑀺𑀁 𑀯𑀺𑀯𑁂𑀓𑁂𑀦 𑀓𑀸𑀳𑀲𑀺;_  
_𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀓𑀸𑀫𑀭𑀢𑀺𑀬𑁄,_  
_𑀫𑀸𑀳𑀼 𑀧𑀘𑁆𑀙𑀸𑀦𑀼𑀢𑀸𑀧𑀺𑀦𑀻”𑀢𑀺𑁇_  


941\. 𑀅𑀣 𑀔𑁄 𑀆𑀍𑀅𑀯𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁄 𑀦𑀼 𑀔𑁆𑀯𑀸𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀯𑀸 𑀕𑀸𑀣𑀁 𑀪𑀸𑀲𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀍𑀅𑀯𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀸𑀭𑁄 𑀔𑁄 𑀅𑀬𑀁 𑀧𑀸𑀧𑀺𑀫𑀸 𑀫𑀫 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀯𑀺𑀯𑁂𑀓𑀫𑁆𑀳𑀸 𑀘𑀸𑀯𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀕𑀸𑀣𑀁 𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀍𑀅𑀯𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 “𑀫𑀸𑀭𑁄 𑀅𑀬𑀁 𑀧𑀸𑀧𑀺𑀫𑀸” 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀳𑀺 𑀧𑀘𑁆𑀘𑀪𑀸𑀲𑀺—

942\. _“𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀮𑁄𑀓𑁂,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀫𑁂 𑀲𑀼𑀨𑀼𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀧𑀫𑀢𑁆𑀢𑀩𑀦𑁆𑀥𑀼 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦 𑀢𑁆𑀯𑀁 𑀚𑀸𑀦𑀸𑀲𑀺 𑀢𑀁 𑀧𑀤𑀁𑁇_  


943\. _𑀲𑀢𑁆𑀢𑀺𑀲𑀽𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸,_  
_𑀔𑀦𑁆𑀥𑀸𑀲𑀁 𑀅𑀥𑀺𑀓𑀼𑀝𑁆𑀝𑀦𑀸;_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀓𑀸𑀫𑀭𑀢𑀺𑀁 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_𑀅𑀭𑀢𑀺 𑀫𑀬𑁆𑀳 𑀲𑀸 𑀅𑀳𑀽”𑀢𑀺𑁇_  


944\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 “𑀚𑀸𑀦𑀸𑀢𑀺 𑀫𑀁 𑀆𑀍𑀅𑀯𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻”𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀻 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇
