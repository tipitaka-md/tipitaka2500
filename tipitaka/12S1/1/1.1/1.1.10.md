# 𑁧𑁇𑁧𑁇𑁧𑁦 𑀅𑀭𑀜𑁆𑀜𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀅𑀭𑀜𑁆𑀜𑀲𑀼𑀢𑁆𑀢

(10.)

31\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

32\. _“𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀁;_  
_𑀏𑀓𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀓𑁂𑀦 𑀯𑀡𑁆𑀡𑁄 𑀧𑀲𑀻𑀤𑀢𑀻”𑀢𑀺𑁇_  


33\. _“𑀅𑀢𑀻𑀢𑀁 𑀦𑀸𑀦𑀼𑀲𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀦𑀧𑁆𑀧𑀚𑀧𑁆𑀧𑀦𑁆𑀢𑀺 𑀦𑀸𑀕𑀢𑀁;_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁄 𑀧𑀲𑀻𑀤𑀢𑀺𑁇_  


34\. _𑀅𑀦𑀸𑀕𑀢𑀧𑁆𑀧𑀚𑀧𑁆𑀧𑀸𑀬,_  
_𑀅𑀢𑀻𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀲𑁄𑀘𑀦𑀸;_  
_𑀏𑀢𑁂𑀦 𑀩𑀸𑀮𑀸 𑀲𑀼𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀍𑀑𑀯 𑀳𑀭𑀺𑀢𑁄 𑀮𑀼𑀢𑁄”𑀢𑀺𑁇_  


---

35\. 𑀦𑀍𑀅𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

36\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



37\. _𑀑𑀖𑀁 𑀦𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀧𑀦𑁂𑀬𑁆𑀬𑀁,_  
_𑀅𑀘𑁆𑀘𑁂𑀦𑁆𑀢𑀺 𑀓𑀢𑀺𑀙𑀺𑀦𑁆𑀤𑀺 𑀘;_  
_𑀚𑀸𑀕𑀭𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑀺𑀢𑀸,_  
_𑀲𑀼𑀲𑀫𑁆𑀫𑀼𑀝𑁆𑀞𑀸 𑀫𑀸𑀦𑀓𑀸𑀫𑀺𑀦𑀸;_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀤𑀲𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄,_  
_𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
