# 𑁧𑁇𑁧𑁇𑁬 𑀚𑀸𑀕𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀚𑀸𑀕𑀭𑀲𑀼𑀢𑁆𑀢

(6.)

19\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

20\. _“𑀓𑀢𑀺 𑀚𑀸𑀕𑀭𑀢𑀁 𑀲𑀼𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀺 𑀲𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀚𑀸𑀕𑀭𑀸;_  
_𑀓𑀢𑀺𑀪𑀺 𑀭𑀚𑀫𑀸𑀤𑁂𑀢𑀺,_  
_𑀓𑀢𑀺𑀪𑀺 𑀧𑀭𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_  


21\. _“𑀧𑀜𑁆𑀘 𑀚𑀸𑀕𑀭𑀢𑀁 𑀲𑀼𑀢𑁆𑀢𑀸,_  
_𑀧𑀜𑁆𑀘 𑀲𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀚𑀸𑀕𑀭𑀸;_  
_𑀧𑀜𑁆𑀘𑀪𑀺 𑀭𑀚𑀫𑀸𑀤𑁂𑀢𑀺,_  
_𑀧𑀜𑁆𑀘𑀪𑀺 𑀧𑀭𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_
