# 𑁧𑁇𑁨𑁇𑁫 𑀲𑀡𑀫𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

* 𑀲𑀡𑀫𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(15.)

51\. _“𑀞𑀺𑀢𑁂 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀓𑀸𑀮𑁂,_  
_𑀲𑀦𑁆𑀦𑀺𑀲𑀻𑀯𑁂𑀲𑀼 𑀧𑀓𑁆𑀔𑀺𑀲𑀼;_  
_𑀲𑀡𑀢𑁂𑀯 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑀁,_  
_𑀢𑀁 𑀪𑀬𑀁 𑀧𑀝𑀺𑀪𑀸𑀢𑀺 𑀫𑀦𑁆”𑀢𑀺𑁇_  


52\. _“𑀞𑀺𑀢𑁂 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀓𑀸𑀮𑁂,_  
_𑀲𑀦𑁆𑀦𑀺𑀲𑀻𑀯𑁂𑀲𑀼 𑀧𑀓𑁆𑀔𑀺𑀲𑀼;_  
_𑀲𑀡𑀢𑁂𑀯 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑀁,_  
_𑀲𑀸 𑀭𑀢𑀺 𑀧𑀝𑀺𑀪𑀸𑀢𑀺 𑀫𑀦𑁆”𑀢𑀺𑁇_
