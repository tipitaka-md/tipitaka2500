# 𑁧𑁇𑁨𑁇𑁯 𑀓𑀼𑀝𑀺𑀓𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀝𑀺𑀓𑀸𑀲𑀼𑀢𑁆𑀢

(19.)

60\. _“𑀓𑀘𑁆𑀘𑀺 𑀢𑁂 𑀓𑀼𑀝𑀺𑀓𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀓𑀘𑁆𑀘𑀺 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀮𑀸𑀯𑀓𑀸;_  
_𑀓𑀘𑁆𑀘𑀺 𑀲𑀦𑁆𑀢𑀸𑀦𑀓𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀓𑀘𑁆𑀘𑀺 𑀫𑀼𑀢𑁆𑀢𑁄𑀲𑀺 𑀩𑀦𑁆𑀥𑀦𑀸”𑀢𑀺𑁇_  


61\. _“𑀢𑀕𑁆𑀖 𑀫𑁂 𑀓𑀼𑀝𑀺𑀓𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀢𑀕𑁆𑀖 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀮𑀸𑀯𑀓𑀸;_  
_𑀢𑀕𑁆𑀖 𑀲𑀦𑁆𑀢𑀸𑀦𑀓𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀢𑀕𑁆𑀖 𑀫𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀩𑀦𑁆𑀥𑀦𑀸”𑀢𑀺𑁇_  


62\. _“𑀓𑀺𑀦𑁆𑀢𑀸𑀳𑀁 𑀓𑀼𑀝𑀺𑀓𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀓𑀺𑀁 𑀢𑁂 𑀩𑁆𑀭𑀽𑀫𑀺 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀓𑀺𑀁 𑀢𑁂 𑀲𑀦𑁆𑀢𑀸𑀦𑀓𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀓𑀺𑀦𑁆𑀢𑀸𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


63\. _“𑀫𑀸𑀢𑀭𑀁 𑀓𑀼𑀝𑀺𑀓𑀁 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_𑀪𑀭𑀺𑀬𑀁 𑀩𑁆𑀭𑀽𑀲𑀺 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀧𑀼𑀢𑁆𑀢𑁂 𑀲𑀦𑁆𑀢𑀸𑀦𑀓𑁂 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_𑀢𑀡𑁆𑀳𑀁 𑀫𑁂 𑀩𑁆𑀭𑀽𑀲𑀺 𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


64\. _“𑀲𑀸𑀳𑀼 𑀢𑁂 𑀓𑀼𑀝𑀺𑀓𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀸𑀳𑀼 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀮𑀸𑀯𑀓𑀸;_  
_𑀲𑀸𑀳𑀼 𑀲𑀦𑁆𑀢𑀸𑀦𑀓𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀸𑀳𑀼 𑀫𑀼𑀢𑁆𑀢𑁄𑀲𑀺 𑀩𑀦𑁆𑀥𑀦𑀸”𑀢𑀺𑁇_
