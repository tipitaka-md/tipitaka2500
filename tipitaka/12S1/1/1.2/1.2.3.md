# 𑁧𑁇𑁨𑁇𑁩 𑀦𑀢𑁆𑀣𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀦𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

### 𑀦𑀢𑁆𑀣𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀫𑀲𑀼𑀢𑁆𑀢

(13.)

46\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

47\. _“𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀢𑁆𑀢𑀲𑀫𑀁 𑀧𑁂𑀫𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀕𑁄𑀲𑀫𑀺𑀢𑀁 𑀥𑀦𑀁;_  
_𑀦𑀢𑁆𑀣𑀺 𑀲𑀽𑀭𑀺𑀬𑀲𑀫𑀸 𑀆𑀪𑀸,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑀧𑀭𑀫𑀸 𑀲𑀭𑀸”𑀢𑀺𑁇_  


48\. _“𑀦𑀢𑁆𑀣𑀺 𑀅𑀢𑁆𑀢𑀲𑀫𑀁 𑀧𑁂𑀫𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀥𑀜𑁆𑀜𑀲𑀫𑀁 𑀥𑀦𑀁;_  
_𑀦𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀸 𑀆𑀪𑀸,_  
_𑀯𑀼𑀝𑁆𑀞𑀺 𑀯𑁂 𑀧𑀭𑀫𑀸 𑀲𑀭𑀸”𑀢𑀺𑁇_
