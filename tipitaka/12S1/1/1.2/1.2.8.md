# 𑁧𑁇𑁨𑁇𑁮 𑀳𑀺𑀭𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

* 𑀳𑀺𑀭𑀻𑀲𑀼𑀢𑁆𑀢

(18.)

58\. _“𑀳𑀺𑀭𑀻𑀦𑀺𑀲𑁂𑀥𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀓𑁄𑀘𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀬𑁄 𑀦𑀺𑀦𑁆𑀤𑀁 𑀅𑀧𑀩𑁄𑀥𑀢𑀺,_  
_𑀅𑀲𑁆𑀲𑁄 𑀪𑀤𑁆𑀭𑁄 𑀓𑀲𑀸𑀫𑀺𑀯𑀸”𑀢𑀺𑁇_  


59\. _“𑀳𑀺𑀭𑀻𑀦𑀺𑀲𑁂𑀥𑀸 𑀢𑀦𑀼𑀬𑀸,_  
_𑀬𑁂 𑀘𑀭𑀦𑁆𑀢𑀺 𑀲𑀤𑀸 𑀲𑀢𑀸;_  
_𑀅𑀦𑁆𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀧𑁆𑀧𑀼𑀬𑁆𑀬,_  
_𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑀫𑁂 𑀲𑀫𑀦𑁆”𑀢𑀺𑁇_
