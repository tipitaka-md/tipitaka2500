# 𑁧𑁇𑁨𑁇𑁪 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

* 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(14.)

49\. _“𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀁 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑁄 𑀘𑀢𑀼𑀧𑁆𑀧𑀤𑀁;_  
_𑀓𑁄𑀫𑀸𑀭𑀻 𑀲𑁂𑀝𑁆𑀞𑀸 𑀪𑀭𑀺𑀬𑀸𑀦𑀁,_  
_𑀬𑁄 𑀘 𑀧𑀼𑀢𑁆𑀢𑀸𑀦 𑀧𑀼𑀩𑁆𑀩𑀚𑁄”𑀢𑀺𑁇_  


50\. _“𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀁 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑁄 𑀘𑀢𑀼𑀧𑁆𑀧𑀤𑀁;_  
_𑀲𑀼𑀲𑁆𑀲𑀽𑀲𑀸 𑀲𑁂𑀝𑁆𑀞𑀸 𑀪𑀭𑀺𑀬𑀸𑀦𑀁,_  
_𑀬𑁄 𑀘 𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀫𑀲𑁆𑀲𑀯𑁄”𑀢𑀺𑁇_
