# 𑁧𑁇𑁨𑁇𑁧 𑀦𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀦𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀦𑀲𑀼𑀢𑁆𑀢

(11.)

38\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

39\. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀢𑀸 𑀦𑀦𑁆𑀤𑀦𑁂 𑀯𑀦𑁂 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀗𑁆𑀖𑀧𑀭𑀺𑀯𑀼𑀢𑀸 𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀫𑀸𑀦𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

40\. _‘𑀦 𑀢𑁂 𑀲𑀼𑀔𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀦𑀦𑁆𑀤𑀦𑀁;_  
_𑀆𑀯𑀸𑀲𑀁 𑀦𑀭𑀤𑁂𑀯𑀸𑀦𑀁,_  
_𑀢𑀺𑀤𑀲𑀸𑀦𑀁 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀦𑁆’𑀢𑀺𑁇_  


41\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀢𑀁 𑀤𑁂𑀯𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀧𑀘𑁆𑀘𑀪𑀸𑀲𑀺—

42\. _‘𑀦 𑀢𑁆𑀯𑀁 𑀩𑀸𑀮𑁂 𑀧𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀬𑀣𑀸 𑀅𑀭𑀳𑀢𑀁 𑀯𑀘𑁄;_  
_𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑀸,_  
_𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀬𑀥𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀲𑀁 𑀯𑀽𑀧𑀲𑀫𑁄 𑀲𑀼𑀔𑁄’”𑀢𑀺𑁇_
