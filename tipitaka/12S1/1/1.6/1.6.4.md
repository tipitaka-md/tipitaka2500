# 𑁧𑁇𑁬𑁇𑁪 𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

* 𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

(54.)

294\. _“𑀓𑀺𑀁𑀲𑀼 𑀯𑀢𑁆𑀣𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀓𑀺𑀁𑀲𑀽𑀥 𑀧𑀭𑀫𑁄 𑀲𑀔𑀸;_  
_𑀓𑀺𑀁𑀲𑀼 𑀪𑀽𑀢𑀸 𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀧𑀸𑀡𑀸 𑀧𑀣𑀯𑀺𑀲𑁆𑀲𑀺𑀢𑀸”𑀢𑀺𑁇_  


295\. _“𑀧𑀼𑀢𑁆𑀢𑀸 𑀯𑀢𑁆𑀣𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀪𑀭𑀺𑀬𑀸 𑀘 𑀧𑀭𑀫𑁄 𑀲𑀔𑀸;_  
_𑀯𑀼𑀝𑁆𑀞𑀺𑀁 𑀪𑀽𑀢𑀸 𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀧𑀸𑀡𑀸 𑀧𑀣𑀯𑀺𑀲𑁆𑀲𑀺𑀢𑀸”𑀢𑀺𑁇_
