# 𑁧𑁇𑁬𑁇𑁯 𑀤𑀼𑀢𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(59.)

304\. _“𑀓𑀺𑀁𑀲𑀼 𑀤𑀼𑀢𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑁄𑀢𑀺,_  
_𑀓𑀺𑀁𑀲𑀼 𑀘𑁂𑀦𑀁 𑀧𑀲𑀸𑀲𑀢𑀺;_  
_𑀓𑀺𑀲𑁆𑀲 𑀘𑀸𑀪𑀺𑀭𑀢𑁄 𑀫𑀘𑁆𑀘𑁄,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


305\. _“𑀲𑀤𑁆𑀥𑀸 𑀤𑀼𑀢𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑁄𑀢𑀺,_  
_𑀧𑀜𑁆𑀜𑀸 𑀘𑁂𑀦𑀁 𑀧𑀲𑀸𑀲𑀢𑀺;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀪𑀺𑀭𑀢𑁄 𑀫𑀘𑁆𑀘𑁄,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_
