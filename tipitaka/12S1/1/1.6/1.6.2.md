# 𑁧𑁇𑁬𑁇𑁨 𑀅𑀚𑀭𑀲𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

* 𑀅𑀚𑀭𑀲𑀸𑀲𑀼𑀢𑁆𑀢

(52.)

290\. _“𑀓𑀺𑀁𑀲𑀼 𑀅𑀚𑀭𑀲𑀸 𑀲𑀸𑀥𑀼,_  
_𑀓𑀺𑀁𑀲𑀼 𑀲𑀸𑀥𑀼 𑀅𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀓𑀺𑀁𑀲𑀼 𑀦𑀭𑀸𑀦𑀁 𑀭𑀢𑀦𑀁,_  
_𑀓𑀺𑀁𑀲𑀼 𑀘𑁄𑀭𑁂𑀳𑁆𑀬𑀳𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇_  


291\. _“𑀲𑀻𑀮𑀁 𑀅𑀚𑀭𑀲𑀸 𑀲𑀸𑀥𑀼,_  
_𑀲𑀤𑁆𑀥𑀸 𑀲𑀸𑀥𑀼 𑀅𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀜𑀸 𑀦𑀭𑀸𑀦𑀁 𑀭𑀢𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀁 𑀘𑁄𑀭𑁂𑀳𑁆𑀬𑀳𑀸𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇_
