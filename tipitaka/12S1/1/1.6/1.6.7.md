# 𑁧𑁇𑁬𑁇𑁭 𑀢𑀢𑀺𑀬𑀚𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

* 𑀢𑀢𑀺𑀬𑀚𑀦𑀲𑀼𑀢𑁆𑀢

(57.)

300\. _“𑀓𑀺𑀁𑀲𑀼 𑀚𑀦𑁂𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑀁,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀯𑀺𑀥𑀸𑀯𑀢𑀺;_  
_𑀓𑀺𑀁𑀲𑀼 𑀲𑀁𑀲𑀸𑀭𑀫𑀸𑀧𑀸𑀤𑀺,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀧𑀭𑀸𑀬𑀦𑀦𑁆”𑀢𑀺𑁇_  


301\. _“𑀢𑀡𑁆𑀳𑀸 𑀚𑀦𑁂𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀯𑀺𑀥𑀸𑀯𑀢𑀺;_  
_𑀲𑀢𑁆𑀢𑁄 𑀲𑀁𑀲𑀸𑀭𑀫𑀸𑀧𑀸𑀤𑀺,_  
_𑀓𑀫𑁆𑀫𑀁 𑀢𑀲𑁆𑀲 𑀧𑀭𑀸𑀬𑀦𑀦𑁆”𑀢𑀺𑁇_
