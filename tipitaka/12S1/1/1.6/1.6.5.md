# 𑁧𑁇𑁬𑁇𑁫 𑀧𑀞𑀫𑀚𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀚𑀦𑀲𑀼𑀢𑁆𑀢

(55.)

296\. _“𑀓𑀺𑀁𑀲𑀼 𑀚𑀦𑁂𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑀁,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀯𑀺𑀥𑀸𑀯𑀢𑀺;_  
_𑀓𑀺𑀁𑀲𑀼 𑀲𑀁𑀲𑀸𑀭𑀫𑀸𑀧𑀸𑀤𑀺,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀦𑁆”𑀢𑀺𑁇_  


297\. _“𑀢𑀡𑁆𑀳𑀸 𑀚𑀦𑁂𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀯𑀺𑀥𑀸𑀯𑀢𑀺;_  
_𑀲𑀢𑁆𑀢𑁄 𑀲𑀁𑀲𑀸𑀭𑀫𑀸𑀧𑀸𑀤𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀦𑁆”𑀢𑀺𑁇_
