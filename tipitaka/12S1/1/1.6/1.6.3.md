# 𑁧𑁇𑁬𑁇𑁩 𑀫𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

### 𑀫𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(53.)

292\. _“𑀓𑀺𑀁𑀲𑀼 𑀧𑀯𑀲𑀢𑁄 𑀫𑀺𑀢𑁆𑀢𑀁,_  
_𑀓𑀺𑀁𑀲𑀼 𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀓𑁂 𑀖𑀭𑁂;_  
_𑀓𑀺𑀁 𑀫𑀺𑀢𑁆𑀢𑀁 𑀅𑀢𑁆𑀣𑀚𑀸𑀢𑀲𑁆𑀲,_  
_𑀓𑀺𑀁 𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀦𑁆”𑀢𑀺𑁇_  


293\. _“𑀲𑀢𑁆𑀣𑁄 𑀧𑀯𑀲𑀢𑁄 𑀫𑀺𑀢𑁆𑀢𑀁,_  
_𑀫𑀸𑀢𑀸 𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀓𑁂 𑀖𑀭𑁂;_  
_𑀲𑀳𑀸𑀬𑁄 𑀅𑀢𑁆𑀣𑀚𑀸𑀢𑀲𑁆𑀲,_  
_𑀳𑁄𑀢𑀺 𑀫𑀺𑀢𑁆𑀢𑀁 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁;_  
_𑀲𑀬𑀁𑀓𑀢𑀸𑀦𑀺 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺,_  
_𑀢𑀁 𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀦𑁆”𑀢𑀺𑁇_
