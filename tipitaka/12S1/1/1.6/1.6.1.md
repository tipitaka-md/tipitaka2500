# 𑁧𑁇𑁬𑁇𑁧 𑀚𑀭𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

### 𑀚𑀭𑀸𑀲𑀼𑀢𑁆𑀢

(51.)

288\. _“𑀓𑀺𑀁𑀲𑀼 𑀬𑀸𑀯 𑀚𑀭𑀸 𑀲𑀸𑀥𑀼,_  
_𑀓𑀺𑀁𑀲𑀼 𑀲𑀸𑀥𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀓𑀺𑀁𑀲𑀼 𑀦𑀭𑀸𑀦𑀁 𑀭𑀢𑀦𑀁,_  
_𑀓𑀺𑀁𑀲𑀼 𑀘𑁄𑀭𑁂𑀳𑀺 𑀤𑀽𑀳𑀭𑀦𑁆”𑀢𑀺𑁇_  


289\. _“𑀲𑀻𑀮𑀁 𑀬𑀸𑀯 𑀚𑀭𑀸 𑀲𑀸𑀥𑀼,_  
_𑀲𑀤𑁆𑀥𑀸 𑀲𑀸𑀥𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀧𑀜𑁆𑀜𑀸 𑀦𑀭𑀸𑀦𑀁 𑀭𑀢𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀁 𑀘𑁄𑀭𑁂𑀳𑀺 𑀤𑀽𑀳𑀭𑀦𑁆”𑀢𑀺𑁇_
