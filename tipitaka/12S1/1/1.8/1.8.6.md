# 𑁧𑁇𑁮𑁇𑁬 𑀦𑀚𑀻𑀭𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀙𑁂𑀢𑁆𑀯𑀸𑀯𑀕𑁆𑀕

### 𑀦𑀚𑀻𑀭𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(76.)

346\. _“𑀓𑀺𑀁 𑀚𑀻𑀭𑀢𑀺 𑀓𑀺𑀁 𑀦 𑀚𑀻𑀭𑀢𑀺,_  
_𑀓𑀺𑀁𑀲𑀼 𑀉𑀧𑁆𑀧𑀣𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀓𑀺𑀁𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑁄,_  
_𑀓𑀺𑀁𑀲𑀼 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀓𑁆𑀔𑀬𑁄;_  
_𑀓𑀺𑀁 𑀫𑀮𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲,_  
_𑀓𑀺𑀁 𑀲𑀺𑀦𑀸𑀦𑀫𑀦𑁄𑀤𑀓𑀁𑁇_  


347\. _𑀓𑀢𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀙𑀺𑀤𑁆𑀤𑀸𑀦𑀺,_  
_𑀬𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀢𑀁 𑀦 𑀢𑀺𑀝𑁆𑀞𑀢𑀺;_  
_𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀼𑀝𑁆𑀞𑀼𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇_  


348\. _“𑀭𑀽𑀧𑀁 𑀚𑀻𑀭𑀢𑀺 𑀫𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀦 𑀚𑀻𑀭𑀢𑀺;_  
_𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀣𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


349\. _𑀮𑁄𑀪𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑁄,_  
_𑀯𑀬𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀓𑁆𑀔𑀬𑁄;_  
_𑀇𑀢𑁆𑀣𑀻 𑀫𑀮𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲,_  
_𑀏𑀢𑁆𑀣𑀸𑀬𑀁 𑀲𑀚𑁆𑀚𑀢𑁂 𑀧𑀚𑀸;_  
_𑀢𑀧𑁄 𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀜𑁆𑀘,_  
_𑀢𑀁 𑀲𑀺𑀦𑀸𑀦𑀫𑀦𑁄𑀤𑀓𑀁𑁇_  


350\. _𑀙 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀙𑀺𑀤𑁆𑀤𑀸𑀦𑀺,_  
_𑀬𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀢𑀁 𑀦 𑀢𑀺𑀝𑁆𑀞𑀢𑀺;_  
_𑀆𑀮𑀲𑁆𑀬𑀜𑁆𑀘 𑀧𑀫𑀸𑀤𑁄 𑀘,_  
_𑀅𑀦𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀲𑀁𑀬𑀫𑁄;_  
_𑀦𑀺𑀤𑁆𑀤𑀸 𑀢𑀦𑁆𑀤𑀻 𑀘 𑀢𑁂 𑀙𑀺𑀤𑁆𑀤𑁂,_  
_𑀲𑀩𑁆𑀩𑀲𑁄 𑀢𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑀬𑁂”𑀢𑀺𑁇_
