# 𑁧𑁇𑁮𑁇𑁧𑁧 𑀅𑀭𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀙𑁂𑀢𑁆𑀯𑀸𑀯𑀕𑁆𑀕

* 𑀅𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(81.)

363\. _“𑀓𑁂𑀲𑀽𑀥 𑀅𑀭𑀡𑀸 𑀮𑁄𑀓𑁂,_  
_𑀓𑁂𑀲𑀁 𑀯𑀼𑀲𑀺𑀢𑀁 𑀦 𑀦𑀲𑁆𑀲𑀢𑀺;_  
_𑀓𑁂𑀥 𑀇𑀘𑁆𑀙𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀓𑁂𑀲𑀁 𑀪𑁄𑀚𑀺𑀲𑁆𑀲𑀺𑀬𑀁 𑀲𑀤𑀸𑁇_  


364\. _𑀓𑀺𑀁𑀲𑀼 𑀫𑀸𑀢𑀸 𑀧𑀺𑀢𑀸 𑀪𑀸𑀢𑀸,_  
_𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀦𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀓𑀺𑀁𑀲𑀼 𑀇𑀥 𑀚𑀸𑀢𑀺𑀳𑀻𑀦𑀁,_  
_𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇_  


365\. _“𑀲𑀫𑀡𑀻𑀥 𑀅𑀭𑀡𑀸 𑀮𑁄𑀓𑁂,_  
_𑀲𑀫𑀡𑀸𑀦𑀁 𑀯𑀼𑀲𑀺𑀢𑀁 𑀦 𑀦𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀫𑀡𑀸 𑀇𑀘𑁆𑀙𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑀡𑀸𑀦𑀁 𑀪𑁄𑀚𑀺𑀲𑁆𑀲𑀺𑀬𑀁 𑀲𑀤𑀸𑁇_  


366\. _𑀲𑀫𑀡𑀁 𑀫𑀸𑀢𑀸 𑀧𑀺𑀢𑀸 𑀪𑀸𑀢𑀸,_  
_𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀦𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀲𑀫𑀡𑀻𑀥 𑀚𑀸𑀢𑀺𑀳𑀻𑀦𑀁,_  
_𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇_  


---

367\. 𑀙𑁂𑀢𑁆𑀯𑀸𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇



---

368\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



369\. _𑀙𑁂𑀢𑁆𑀯𑀸 𑀭𑀣𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀯𑀼𑀝𑁆𑀞𑀺 𑀪𑀻𑀢𑀸 𑀦𑀚𑀻𑀭𑀢𑀺;_  
_𑀇𑀲𑁆𑀲𑀭𑀁 𑀓𑀸𑀫𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀁,_  
_𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀅𑀭𑀡𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_  


---

370\. 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇
