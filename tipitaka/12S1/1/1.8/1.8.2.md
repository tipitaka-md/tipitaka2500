# 𑁧𑁇𑁮𑁇𑁨 𑀭𑀣𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀙𑁂𑀢𑁆𑀯𑀸𑀯𑀕𑁆𑀕

### 𑀭𑀣𑀲𑀼𑀢𑁆𑀢

(72.)

337\. _“𑀓𑀺𑀁𑀲𑀼 𑀭𑀣𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀓𑀺𑀁𑀲𑀼 𑀧𑀜𑁆𑀜𑀸𑀡𑀫𑀕𑁆𑀕𑀺𑀦𑁄;_  
_𑀓𑀺𑀁𑀲𑀼 𑀭𑀝𑁆𑀞𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀓𑀺𑀁𑀲𑀼 𑀧𑀜𑁆𑀜𑀸𑀡𑀫𑀺𑀢𑁆𑀣𑀺𑀬𑀸”𑀢𑀺𑁇_  


338\. _“𑀥𑀚𑁄 𑀭𑀣𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀥𑀽𑀫𑁄 𑀧𑀜𑁆𑀜𑀸𑀡𑀫𑀕𑁆𑀕𑀺𑀦𑁄;_  
_𑀭𑀸𑀚𑀸 𑀭𑀝𑁆𑀞𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀪𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀡𑀫𑀺𑀢𑁆𑀣𑀺𑀬𑀸”𑀢𑀺𑁇_
