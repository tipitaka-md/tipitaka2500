# 𑁧𑁇𑁮𑁇𑁧𑁦 𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀙𑁂𑀢𑁆𑀯𑀸𑀯𑀕𑁆𑀕

* 𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑀼𑀢𑁆𑀢

(80.)

359\. _“𑀓𑀺𑀁𑀲𑀼 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀧𑀚𑁆𑀚𑁄𑀢𑁄,_  
_𑀓𑀺𑀁𑀲𑀼 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀚𑀸𑀕𑀭𑁄;_  
_𑀓𑀺𑀁𑀲𑀼 𑀓𑀫𑁆𑀫𑁂 𑀲𑀚𑀻𑀯𑀸𑀦𑀁,_  
_𑀓𑀺𑀫𑀲𑁆𑀲 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑁄𑁇_  


360\. _𑀓𑀺𑀁𑀲𑀼 𑀅𑀮𑀲𑀁 𑀅𑀦𑀮𑀲𑀜𑁆𑀘,_  
_𑀫𑀸𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀁𑀯 𑀧𑁄𑀲𑀢𑀺;_  
_𑀓𑀺𑀁 𑀪𑀽𑀢𑀸 𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀧𑀸𑀡𑀸 𑀧𑀣𑀯𑀺𑀲𑁆𑀲𑀺𑀢𑀸”𑀢𑀺𑁇_  


361\. _“𑀧𑀜𑁆𑀜𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀧𑀚𑁆𑀚𑁄𑀢𑁄,_  
_𑀲𑀢𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀚𑀸𑀕𑀭𑁄;_  
_𑀕𑀸𑀯𑁄 𑀓𑀫𑁆𑀫𑁂 𑀲𑀚𑀻𑀯𑀸𑀦𑀁,_  
_𑀲𑀻𑀢𑀲𑁆𑀲 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑁄𑁇_  


362\. _𑀯𑀼𑀝𑁆𑀞𑀺 𑀅𑀮𑀲𑀁 𑀅𑀦𑀮𑀲𑀜𑁆𑀘,_  
_𑀫𑀸𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀁𑀯 𑀧𑁄𑀲𑀢𑀺;_  
_𑀯𑀼𑀝𑁆𑀞𑀺𑀁 𑀪𑀽𑀢𑀸 𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀧𑀸𑀡𑀸 𑀧𑀣𑀯𑀺𑀲𑁆𑀲𑀺𑀢𑀸”𑀢𑀺𑁇_
