# 𑁧𑁇𑁮𑁇𑁫 𑀪𑀻𑀢𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀙𑁂𑀢𑁆𑀯𑀸𑀯𑀕𑁆𑀕

* 𑀪𑀻𑀢𑀸𑀲𑀼𑀢𑁆𑀢

(75.)

344\. _“𑀓𑀺𑀁𑀲𑀽𑀥 𑀪𑀻𑀢𑀸 𑀚𑀦𑀢𑀸 𑀅𑀦𑁂𑀓𑀸,_  
_𑀫𑀕𑁆𑀕𑁄 𑀘𑀦𑁂𑀓𑀸𑀬𑀢𑀦𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀢𑁄;_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀕𑁄𑀢𑀫 𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜,_  
_𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀞𑀺𑀢𑁄 𑀧𑀭𑀮𑁄𑀓𑀁 𑀦 𑀪𑀸𑀬𑁂”𑀢𑀺𑁇_  


345\. _“𑀯𑀸𑀘𑀁 𑀫𑀦𑀜𑁆𑀘 𑀧𑀡𑀺𑀥𑀸𑀬 𑀲𑀫𑁆𑀫𑀸,_  
_𑀓𑀸𑀬𑁂𑀦 𑀧𑀸𑀧𑀸𑀦𑀺 𑀅𑀓𑀼𑀩𑁆𑀩𑀫𑀸𑀦𑁄;_  
_𑀩𑀳𑁆𑀯𑀦𑁆𑀦𑀧𑀸𑀦𑀁 𑀖𑀭𑀫𑀸𑀯𑀲𑀦𑁆𑀢𑁄,_  
_𑀲𑀤𑁆𑀥𑁄 𑀫𑀼𑀤𑀽 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀻 𑀯𑀤𑀜𑁆𑀜𑀽;_  
_𑀏𑀢𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀞𑀺𑀢𑁄 𑀘𑀢𑀽𑀲𑀼,_  
_𑀥𑀫𑁆𑀫𑁂 𑀞𑀺𑀢𑁄 𑀧𑀭𑀮𑁄𑀓𑀁 𑀦 𑀪𑀸𑀬𑁂”𑀢𑀺𑁇_
