# 𑁧𑁇𑁪𑁇𑁧 𑀲𑀩𑁆𑀪𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀢𑀼𑀮𑁆𑀮𑀧𑀓𑀸𑀬𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀩𑁆𑀪𑀺𑀲𑀼𑀢𑁆𑀢

(31.)

127\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀢𑀼𑀮𑁆𑀮𑀧𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀏𑀓𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

128\. _“𑀲𑀩𑁆𑀪𑀺𑀭𑁂𑀯 𑀲𑀫𑀸𑀲𑁂𑀣,_  
_𑀲𑀩𑁆𑀪𑀺 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀲𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀧𑀸𑀧𑀺𑀬𑁄”𑀢𑀺𑁇_  


129\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

130\. _“𑀲𑀩𑁆𑀪𑀺𑀭𑁂𑀯 𑀲𑀫𑀸𑀲𑁂𑀣,_  
_𑀲𑀩𑁆𑀪𑀺 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀲𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀧𑀜𑁆𑀜𑀸 𑀮𑀩𑁆𑀪𑀢𑀺 𑀦𑀸𑀜𑁆𑀜𑀢𑁄”𑀢𑀺𑁇_  


131\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

132\. _“𑀲𑀩𑁆𑀪𑀺𑀭𑁂𑀯 𑀲𑀫𑀸𑀲𑁂𑀣,_  
_𑀲𑀩𑁆𑀪𑀺 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀲𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑁄𑀓𑀫𑀚𑁆𑀛𑁂 𑀦 𑀲𑁄𑀘𑀢𑀻”𑀢𑀺𑁇_  


133\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

134\. _“𑀲𑀩𑁆𑀪𑀺𑀭𑁂𑀯 𑀲𑀫𑀸𑀲𑁂𑀣,_  
_𑀲𑀩𑁆𑀪𑀺 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀲𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀜𑀸𑀢𑀺𑀫𑀚𑁆𑀛𑁂 𑀯𑀺𑀭𑁄𑀘𑀢𑀻”𑀢𑀺𑁇_  


135\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

136\. _“𑀲𑀩𑁆𑀪𑀺𑀭𑁂𑀯 𑀲𑀫𑀸𑀲𑁂𑀣,_  
_𑀲𑀩𑁆𑀪𑀺 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀲𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀼𑀕𑁆𑀕𑀢𑀺𑀦𑁆”𑀢𑀺𑁇_  


137\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

138\. _“𑀲𑀩𑁆𑀪𑀺𑀭𑁂𑀯 𑀲𑀫𑀸𑀲𑁂𑀣,_  
_𑀲𑀩𑁆𑀪𑀺 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀲𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀢𑁆𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀲𑀸𑀢𑀢𑀦𑁆”𑀢𑀺𑁇_  


139\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄, 𑀪𑀕𑀯𑀸, 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀦𑁆”𑀢𑀺?

140\. “𑀲𑀩𑁆𑀩𑀸𑀲𑀁 𑀯𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀅𑀧𑀺 𑀘 𑀫𑀫𑀧𑀺 𑀲𑀼𑀡𑀸𑀣—

141\. _𑀲𑀩𑁆𑀪𑀺𑀭𑁂𑀯 𑀲𑀫𑀸𑀲𑁂𑀣,_  
_𑀲𑀩𑁆𑀪𑀺 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀲𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


142\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀽𑀢𑀺𑁇
