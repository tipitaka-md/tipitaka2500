# 𑁧𑁇𑁪𑁇𑁩 𑀲𑀸𑀥𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀢𑀼𑀮𑁆𑀮𑀧𑀓𑀸𑀬𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀥𑀼𑀲𑀼𑀢𑁆𑀢

(33.)

161\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀢𑀼𑀮𑁆𑀮𑀧𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀏𑀓𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

162\. _“𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲 𑀤𑀸𑀦𑀁,_  
_𑀫𑀘𑁆𑀙𑁂𑀭𑀸 𑀘 𑀧𑀫𑀸𑀤𑀸 𑀘;_  
_𑀏𑀯𑀁 𑀤𑀸𑀦𑀁 𑀦 𑀤𑀻𑀬𑀢𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀁 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁂𑀦;_  
_𑀤𑁂𑀬𑁆𑀬𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀚𑀸𑀦𑀢𑀸”𑀢𑀺𑁇_  


163\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

164\. _“𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲 𑀤𑀸𑀦𑀁,_  
_𑀅𑀧𑀺 𑀘 𑀅𑀧𑁆𑀧𑀓𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁𑁇_  


165\. _𑀅𑀧𑁆𑀧𑀲𑁆𑀫𑁂𑀓𑁂 𑀧𑀯𑁂𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀩𑀳𑀼𑀦𑁂𑀓𑁂 𑀦 𑀤𑀺𑀘𑁆𑀙𑀭𑁂;_  
_𑀅𑀧𑁆𑀧𑀲𑁆𑀫𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀲𑀫𑀁 𑀫𑀺𑀢𑀸”𑀢𑀺𑁇_  


166\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

167\. _“𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲 𑀤𑀸𑀦𑀁,_  
_𑀅𑀧𑁆𑀧𑀓𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁;_  
_𑀅𑀧𑀺 𑀘 𑀲𑀤𑁆𑀥𑀸𑀬𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁,_  
_𑀤𑀸𑀦𑀜𑁆𑀘 𑀬𑀼𑀤𑁆𑀥𑀜𑁆𑀘 𑀲𑀫𑀸𑀦𑀫𑀸𑀳𑀼;_  
_𑀅𑀧𑁆𑀧𑀸𑀧𑀺 𑀲𑀦𑁆𑀢𑀸 𑀩𑀳𑀼𑀓𑁂 𑀚𑀺𑀦𑀦𑁆𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑁆𑀧𑀺 𑀘𑁂 𑀲𑀤𑁆𑀤𑀳𑀸𑀦𑁄 𑀤𑀤𑀸𑀢𑀺;_  
_𑀢𑁂𑀦𑁂𑀯 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀔𑀻 𑀧𑀭𑀢𑁆𑀣𑀸”𑀢𑀺𑁇_  


168\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

169\. _“𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲 𑀤𑀸𑀦𑀁,_  
_𑀅𑀧𑁆𑀧𑀓𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁;_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁,_  
_𑀅𑀧𑀺 𑀘 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁𑁇_  


170\. _𑀬𑁄 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀤𑀸𑀢𑀺 𑀤𑀸𑀦𑀁,_  
_𑀉𑀝𑁆𑀞𑀸𑀦𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀚𑀦𑁆𑀢𑀼;_  
_𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑁄 𑀯𑁂𑀢𑀭𑀡𑀺𑀁 𑀬𑀫𑀲𑁆𑀲,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀞𑀸𑀦𑀸𑀦𑀺 𑀉𑀧𑁂𑀢𑀺 𑀫𑀘𑁆𑀘𑁄”𑀢𑀺𑁇_  


171\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

172\. _“𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲 𑀤𑀸𑀦𑀁,_  
_𑀅𑀧𑁆𑀧𑀓𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁;_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁;_  
_𑀅𑀧𑀺 𑀘 𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀤𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁𑁇_  


173\. _𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀤𑀸𑀦𑀁 𑀲𑀼𑀕𑀢𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑀁,_  
_𑀬𑁂 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀇𑀥 𑀚𑀻𑀯𑀮𑁄𑀓𑁂;_  
_𑀏𑀢𑁂𑀲𑀼 𑀤𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀸𑀦𑀺,_  
_𑀩𑀻𑀚𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀬𑀣𑀸 𑀲𑀼𑀔𑁂𑀢𑁆𑀢𑁂”𑀢𑀺𑁇_  


174\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

175\. _“𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑀸𑀭𑀺𑀲 𑀤𑀸𑀦𑀁,_  
_𑀅𑀧𑁆𑀧𑀓𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁;_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁;_  
_𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀤𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀁,_  
_𑀅𑀧𑀺 𑀘 𑀧𑀸𑀡𑁂𑀲𑀼𑀧𑀺 𑀲𑀸𑀥𑀼 𑀲𑀁𑀬𑀫𑁄𑁇_  


176\. _𑀬𑁄 𑀧𑀸𑀡𑀪𑀽𑀢𑀸𑀦𑀺 𑀅𑀳𑁂𑀞𑀬𑀁 𑀘𑀭𑀁,_  
_𑀧𑀭𑀽𑀧𑀯𑀸𑀤𑀸 𑀦 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀁;_  
_𑀪𑀻𑀭𑀼𑀁 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺 𑀦 𑀳𑀺 𑀢𑀢𑁆𑀣 𑀲𑀽𑀭𑀁,_  
_𑀪𑀬𑀸 𑀳𑀺 𑀲𑀦𑁆𑀢𑁄 𑀦 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀦𑁆”𑀢𑀺𑁇_  


177\. 𑀅𑀣 𑀔𑁄 𑀅𑀧𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄, 𑀪𑀕𑀯𑀸, 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀦𑁆”𑀢𑀺?

178\. “𑀲𑀩𑁆𑀩𑀸𑀲𑀁 𑀯𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦, 𑀅𑀧𑀺 𑀘 𑀫𑀫𑀧𑀺 𑀲𑀼𑀡𑀸𑀣—

179\. _𑀲𑀤𑁆𑀥𑀸 𑀳𑀺 𑀤𑀸𑀦𑀁 𑀩𑀳𑀼𑀥𑀸 𑀧𑀲𑀢𑁆𑀣𑀁,_  
_𑀤𑀸𑀦𑀸 𑀘 𑀔𑁄 𑀥𑀫𑁆𑀫𑀧𑀤𑀁𑀯 𑀲𑁂𑀬𑁆𑀬𑁄;_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀘 𑀳𑀺 𑀧𑀼𑀩𑁆𑀩𑀢𑀭𑁂 𑀘 𑀲𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑁂𑀯𑀚𑁆𑀛𑀕𑀫𑀼𑀁 𑀲𑀧𑀜𑁆𑀜𑀸”𑀢𑀺𑁇_
