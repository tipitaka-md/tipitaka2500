# 𑁧𑁇𑁫𑁇𑁮 𑀚𑁂𑀢𑀯𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀆𑀤𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀚𑁂𑀢𑀯𑀦𑀲𑀼𑀢𑁆𑀢

(48.)

258. _“𑀇𑀤𑀜𑁆𑀳𑀺 𑀢𑀁 𑀚𑁂𑀢𑀯𑀦𑀁,_  
_𑀇𑀲𑀺𑀲𑀁𑀖𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀆𑀯𑀼𑀢𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑁂𑀦,_  
_𑀧𑀻𑀢𑀺𑀲𑀜𑁆𑀚𑀦𑀦𑀁 𑀫𑀫𑁇_  


259. _𑀓𑀫𑁆𑀫𑀁 𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀘,_  
_𑀲𑀻𑀮𑀁 𑀚𑀻𑀯𑀺𑀢𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀏𑀢𑁂𑀦 𑀫𑀘𑁆𑀘𑀸 𑀲𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀦 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀥𑀦𑁂𑀦 𑀯𑀸𑁇_  


260. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁 𑀅𑀢𑁆𑀣𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀬𑁄𑀦𑀺𑀲𑁄 𑀯𑀺𑀘𑀺𑀦𑁂 𑀥𑀫𑁆𑀫𑀁,_  
_𑀏𑀯𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀺𑁇_  


261. _𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀯 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀻𑀮𑁂𑀦 𑀉𑀧𑀲𑀫𑁂𑀦 𑀘;_  
_𑀬𑁄𑀧𑀺 𑀧𑀸𑀭𑀗𑁆𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀏𑀢𑀸𑀯𑀧𑀭𑀫𑁄 𑀲𑀺𑀬𑀸”𑀢𑀺𑁇_
