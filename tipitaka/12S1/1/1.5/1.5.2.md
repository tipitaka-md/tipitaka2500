# 𑁧𑁇𑁫𑁇𑁨 𑀓𑀺𑀁𑀤𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀆𑀤𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀓𑀺𑀁𑀤𑀤𑀲𑀼𑀢𑁆𑀢

(42.)

243\. _“𑀓𑀺𑀁𑀤𑀤𑁄 𑀩𑀮𑀤𑁄 𑀳𑁄𑀢𑀺,_  
_𑀓𑀺𑀁𑀤𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀡𑁆𑀡𑀤𑁄;_  
_𑀓𑀺𑀁𑀤𑀤𑁄 𑀲𑀼𑀔𑀤𑁄 𑀳𑁄𑀢𑀺,_  
_𑀓𑀺𑀁𑀤𑀤𑁄 𑀳𑁄𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀤𑁄;_  
_𑀓𑁄 𑀘 𑀲𑀩𑁆𑀩𑀤𑀤𑁄 𑀳𑁄𑀢𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄”𑀢𑀺𑁇_  


244\. _“𑀅𑀦𑁆𑀦𑀤𑁄 𑀩𑀮𑀤𑁄 𑀳𑁄𑀢𑀺,_  
_𑀯𑀢𑁆𑀣𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀡𑁆𑀡𑀤𑁄;_  
_𑀬𑀸𑀦𑀤𑁄 𑀲𑀼𑀔𑀤𑁄 𑀳𑁄𑀢𑀺,_  
_𑀤𑀻𑀧𑀤𑁄 𑀳𑁄𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀤𑁄𑁇_  


245\. _𑀲𑁄 𑀘 𑀲𑀩𑁆𑀩𑀤𑀤𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀤𑀤𑀸𑀢𑀺 𑀉𑀧𑀲𑁆𑀲𑀬𑀁;_  
_𑀅𑀫𑀢𑀁 𑀤𑀤𑁄 𑀘 𑀲𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀥𑀫𑁆𑀫𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_
