# 𑁧𑁇𑁫𑁇𑁭 𑀯𑀦𑀭𑁄𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀆𑀤𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀯𑀦𑀭𑁄𑀧𑀲𑀼𑀢𑁆𑀢

(47.)

255. _“𑀓𑁂𑀲𑀁 𑀤𑀺𑀯𑀸 𑀘 𑀭𑀢𑁆𑀢𑁄 𑀘,_  
_𑀲𑀤𑀸 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀯𑀟𑁆𑀠𑀢𑀺;_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀓𑁂 𑀚𑀦𑀸 𑀲𑀕𑁆𑀕𑀕𑀸𑀫𑀺𑀦𑁄”𑀢𑀺𑁇_  


256. _“𑀆𑀭𑀸𑀫𑀭𑁄𑀧𑀸 𑀯𑀦𑀭𑁄𑀧𑀸,_  
_𑀬𑁂 𑀚𑀦𑀸 𑀲𑁂𑀢𑀼𑀓𑀸𑀭𑀓𑀸;_  
_𑀧𑀧𑀜𑁆𑀘 𑀉𑀤𑀧𑀸𑀦𑀜𑁆𑀘,_  
_𑀬𑁂 𑀤𑀤𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑁆𑀲𑀬𑀁𑁇_  


257. _𑀢𑁂𑀲𑀁 𑀤𑀺𑀯𑀸 𑀘 𑀭𑀢𑁆𑀢𑁄 𑀘,_  
_𑀲𑀤𑀸 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀯𑀟𑁆𑀠𑀢𑀺;_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀢𑁂 𑀚𑀦𑀸 𑀲𑀕𑁆𑀕𑀕𑀸𑀫𑀺𑀦𑁄”𑀢𑀺𑁇_
