# 𑁧𑁇𑁫𑁇𑁩 𑀅𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀆𑀤𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

(43.)

246. _“𑀅𑀦𑁆𑀦𑀫𑁂𑀯𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺,_  
_𑀉𑀪𑀬𑁂 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑀸;_  
_𑀅𑀣 𑀓𑁄 𑀦𑀸𑀫 𑀲𑁄 𑀬𑀓𑁆𑀔𑁄,_  
_𑀬𑀁 𑀅𑀦𑁆𑀦𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀻”𑀢𑀺𑁇_  


247. _“𑀬𑁂 𑀦𑀁 𑀤𑀤𑀦𑁆𑀢𑀺 𑀲𑀤𑁆𑀥𑀸𑀬,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀢𑀫𑁂𑀯 𑀅𑀦𑁆𑀦𑀁 𑀪𑀚𑀢𑀺,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀫𑁆𑀳𑀺 𑀘𑁇_  


248. _𑀢𑀲𑁆𑀫𑀸 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀫𑀘𑁆𑀙𑁂𑀭𑀁,_  
_𑀤𑀚𑁆𑀚𑀸 𑀤𑀸𑀦𑀁 𑀫𑀮𑀸𑀪𑀺𑀪𑀽;_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀧𑀭𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_
