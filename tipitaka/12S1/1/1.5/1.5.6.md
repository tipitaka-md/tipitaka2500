# 𑁧𑁇𑁫𑁇𑁬 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀆𑀤𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀼𑀢𑁆𑀢

(46.)

251\. _“𑀅𑀘𑁆𑀙𑀭𑀸𑀕𑀡𑀲𑀗𑁆𑀖𑀼𑀝𑁆𑀞𑀁,_  
_𑀧𑀺𑀲𑀸𑀘𑀕𑀡𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀯𑀦𑀦𑁆𑀢𑀁 𑀫𑁄𑀳𑀦𑀁 𑀦𑀸𑀫,_  
_𑀓𑀣𑀁 𑀬𑀸𑀢𑁆𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


252\. _“𑀉𑀚𑀼𑀓𑁄 𑀦𑀸𑀫 𑀲𑁄 𑀫𑀕𑁆𑀕𑁄,_  
_𑀅𑀪𑀬𑀸 𑀦𑀸𑀫 𑀲𑀸 𑀤𑀺𑀲𑀸;_  
_𑀭𑀣𑁄 𑀅𑀓𑀽𑀚𑀦𑁄 𑀦𑀸𑀫,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑁂𑀳𑀺 𑀲𑀁𑀬𑀼𑀢𑁄𑁇_  


253\. _𑀳𑀺𑀭𑀻 𑀢𑀲𑁆𑀲 𑀅𑀧𑀸𑀮𑀫𑁆𑀩𑁄,_  
_𑀲𑀢𑁆𑀬𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀥𑀫𑁆𑀫𑀸𑀳𑀁 𑀲𑀸𑀭𑀣𑀺𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀭𑁂𑀚𑀯𑀁𑁇_  


254\. _𑀬𑀲𑁆𑀲 𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀬𑀸𑀦𑀁,_  
_𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀯𑀸;_  
_𑀲 𑀯𑁂 𑀏𑀢𑁂𑀦 𑀬𑀸𑀦𑁂𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲𑁂𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑀢𑀺𑁇_
