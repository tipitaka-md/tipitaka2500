# 𑁧𑁇𑁩𑁇𑁧𑁦 𑀏𑀡𑀺𑀚𑀗𑁆𑀖𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀏𑀡𑀺𑀚𑀗𑁆𑀖𑀲𑀼𑀢𑁆𑀢

(30.)

122. _“𑀏𑀡𑀺𑀚𑀗𑁆𑀖𑀁 𑀓𑀺𑀲𑀁 𑀯𑀻𑀭𑀁,_  
_𑀅𑀧𑁆𑀧𑀸𑀳𑀸𑀭𑀁 𑀅𑀮𑁄𑀮𑀼𑀧𑀁;_  
_𑀲𑀻𑀳𑀁𑀯𑁂𑀓𑀘𑀭𑀁 𑀦𑀸𑀕𑀁,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑀁;_  
_𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀣𑀁 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


123. _“𑀧𑀜𑁆𑀘 𑀓𑀸𑀫𑀕𑀼𑀡𑀸 𑀮𑁄𑀓𑁂,_  
_𑀫𑀦𑁄𑀙𑀝𑁆𑀞𑀸 𑀧𑀯𑁂𑀤𑀺𑀢𑀸;_  
_𑀏𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑀁 𑀯𑀺𑀭𑀸𑀚𑁂𑀢𑁆𑀯𑀸,_  
_𑀏𑀯𑀁 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


---

124. 𑀲𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

125. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



126. _𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀨𑀼𑀲𑀢𑀺 𑀘𑁂𑀯,_  
_𑀚𑀝𑀸 𑀫𑀦𑁄𑀦𑀺𑀯𑀸𑀭𑀡𑀸;_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀦 𑀧𑀚𑁆𑀚𑁄𑀢𑁄,_  
_𑀲𑀭𑀸 𑀫𑀳𑀤𑁆𑀥𑀦𑁂𑀦 𑀘;_  
_𑀘𑀢𑀼𑀘𑀓𑁆𑀓𑁂𑀦 𑀦𑀯𑀫𑀁,_  
_𑀏𑀡𑀺𑀚𑀗𑁆𑀖𑁂𑀦 𑀢𑁂 𑀤𑀲𑀸𑀢𑀺𑁇_
