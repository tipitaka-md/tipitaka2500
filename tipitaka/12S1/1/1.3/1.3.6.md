# 𑁧𑁇𑁩𑁇𑁬 𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑀼𑀢𑁆𑀢

(26.)

112. _“𑀓𑀢𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀧𑀚𑁆𑀚𑁄𑀢𑀸,_  
_𑀢𑁂𑀳𑀺 𑀮𑁄𑀓𑁄 𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀼𑀝𑁆𑀞𑀼𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀦𑁆”𑀢𑀺𑁇_  


113. _“𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀮𑁄𑀓𑁂 𑀧𑀚𑁆𑀚𑁄𑀢𑀸,_  
_𑀧𑀜𑁆𑀘𑀫𑁂𑀢𑁆𑀣 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀤𑀺𑀯𑀸 𑀢𑀧𑀢𑀺 𑀆𑀤𑀺𑀘𑁆𑀘𑁄,_  
_𑀭𑀢𑁆𑀢𑀺𑀫𑀸𑀪𑀸𑀢𑀺 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑁇_  


114. _𑀅𑀣 𑀅𑀕𑁆𑀕𑀺 𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑀧𑀢𑀁 𑀲𑁂𑀝𑁆𑀞𑁄,_  
_𑀏𑀲𑀸 𑀆𑀪𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸”𑀢𑀺𑁇_
