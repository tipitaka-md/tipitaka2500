# 𑁧𑁇𑁩𑁇𑁨 𑀨𑀼𑀲𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀨𑀼𑀲𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(22.)

99. _“𑀦𑀸𑀨𑀼𑀲𑀦𑁆𑀢𑀁 𑀨𑀼𑀲𑀢𑀺 𑀘,_  
_𑀨𑀼𑀲𑀦𑁆𑀢𑀜𑁆𑀘 𑀢𑀢𑁄 𑀨𑀼𑀲𑁂;_  
_𑀢𑀲𑁆𑀫𑀸 𑀨𑀼𑀲𑀦𑁆𑀢𑀁 𑀨𑀼𑀲𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀤𑀼𑀝𑁆𑀞𑀧𑀤𑁄𑀲𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_  


100. _“𑀬𑁄 𑀅𑀧𑁆𑀧𑀤𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀦𑀭𑀲𑁆𑀲 𑀤𑀼𑀲𑁆𑀲𑀢𑀺,_  
_𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲 𑀅𑀦𑀗𑁆𑀕𑀡𑀲𑁆𑀲;_  
_𑀢𑀫𑁂𑀯 𑀩𑀸𑀮𑀁 𑀧𑀘𑁆𑀘𑁂𑀢𑀺 𑀧𑀸𑀧𑀁,_  
_𑀲𑀼𑀔𑀼𑀫𑁄 𑀭𑀚𑁄 𑀧𑀝𑀺𑀯𑀸𑀢𑀁𑀯 𑀔𑀺𑀢𑁆𑀢𑁄”𑀢𑀺𑁇_
