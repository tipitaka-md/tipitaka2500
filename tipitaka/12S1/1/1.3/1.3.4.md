# 𑁧𑁇𑁩𑁇𑁪 𑀫𑀦𑁄𑀦𑀺𑀯𑀸𑀭𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀲𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀫𑀦𑁄𑀦𑀺𑀯𑀸𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(24.)

105. _“𑀬𑀢𑁄 𑀬𑀢𑁄 𑀫𑀦𑁄 𑀦𑀺𑀯𑀸𑀭𑀬𑁂,_  
_𑀦 𑀤𑀼𑀓𑁆𑀔𑀫𑁂𑀢𑀺 𑀦𑀁 𑀢𑀢𑁄 𑀢𑀢𑁄;_  
_𑀲 𑀲𑀩𑁆𑀩𑀢𑁄 𑀫𑀦𑁄 𑀦𑀺𑀯𑀸𑀭𑀬𑁂,_  
_𑀲 𑀲𑀩𑁆𑀩𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


106. _“𑀦 𑀲𑀩𑁆𑀩𑀢𑁄 𑀫𑀦𑁄 𑀦𑀺𑀯𑀸𑀭𑀬𑁂,_  
_𑀦 𑀫𑀦𑁄 𑀲𑀁𑀬𑀢𑀢𑁆𑀢𑀫𑀸𑀕𑀢𑀁;_  
_𑀬𑀢𑁄 𑀬𑀢𑁄 𑀘 𑀧𑀸𑀧𑀓𑀁,_  
_𑀢𑀢𑁄 𑀢𑀢𑁄 𑀫𑀦𑁄 𑀦𑀺𑀯𑀸𑀭𑀬𑁂”𑀢𑀺𑁇_
