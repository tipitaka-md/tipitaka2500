# 𑁧𑁇𑁩𑁇𑁧 𑀲𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀲𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(21.)

96\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

97\. _“𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀬 𑀑𑀫𑀝𑁆𑀞𑁄,_  
_𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄𑀯 𑀫𑀢𑁆𑀣𑀓𑁂;_  
_𑀓𑀸𑀫𑀭𑀸𑀕𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂”𑀢𑀺𑁇_  


98\. _“𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀬 𑀑𑀫𑀝𑁆𑀞𑁄,_  
_𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄𑀯 𑀫𑀢𑁆𑀣𑀓𑁂;_  
_𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂”𑀢𑀺𑁇_
