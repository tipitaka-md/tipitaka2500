# 𑁧𑁇𑁩𑁇𑁭 𑀲𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀲𑀢𑁆𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀲𑀭𑀲𑀼𑀢𑁆𑀢

(27.)

115\. _“𑀓𑀼𑀢𑁄 𑀲𑀭𑀸 𑀦𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀓𑀢𑁆𑀣 𑀯𑀝𑁆𑀝𑀁 𑀦 𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀓𑀢𑁆𑀣 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘,_  
_𑀅𑀲𑁂𑀲𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_  


116\. _“𑀬𑀢𑁆𑀣 𑀆𑀧𑁄 𑀘 𑀧𑀣𑀯𑀻,_  
_𑀢𑁂𑀚𑁄 𑀯𑀸𑀬𑁄 𑀦 𑀕𑀸𑀥𑀢𑀺;_  
_𑀅𑀢𑁄 𑀲𑀭𑀸 𑀦𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀏𑀢𑁆𑀣 𑀯𑀝𑁆𑀝𑀁 𑀦 𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀏𑀢𑁆𑀣 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘,_  
_𑀅𑀲𑁂𑀲𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_
