# 𑁧𑁇𑁭𑁇𑁪 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀲𑀁𑀬𑁄𑀚𑀦𑀲𑀼𑀢𑁆𑀢

(64.)

317\. _“𑀓𑀺𑀁𑀲𑀼 𑀲𑀁𑀬𑁄𑀚𑀦𑁄 𑀮𑁄𑀓𑁄,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀸𑀭𑀡𑀁;_  
_𑀓𑀺𑀲𑁆𑀲𑀲𑁆𑀲𑀼 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀦𑁂𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


318\. _“𑀦𑀦𑁆𑀤𑀻𑀲𑀁𑀬𑁄𑀚𑀦𑁄 𑀮𑁄𑀓𑁄,_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑁆𑀲 𑀯𑀺𑀘𑀸𑀭𑀡𑀁;_  
_𑀢𑀡𑁆𑀳𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀦𑁂𑀦,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_
