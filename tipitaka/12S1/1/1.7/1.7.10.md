# 𑁧𑁇𑁭𑁇𑁧𑁦 𑀮𑁄𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀮𑁄𑀓𑀲𑀼𑀢𑁆𑀢

(70.)

329\. _“𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀩𑁆𑀩𑀢𑀺 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀓𑀺𑀲𑁆𑀲 𑀮𑁄𑀓𑁄 𑀉𑀧𑀸𑀤𑀸𑀬,_  
_𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁄 𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇_  


330\. _“𑀙𑀲𑀼 𑀮𑁄𑀓𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀙𑀲𑀼 𑀓𑀼𑀩𑁆𑀩𑀢𑀺 𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀙𑀦𑁆𑀦𑀫𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀬,_  
_𑀙𑀲𑀼 𑀮𑁄𑀓𑁄 𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇_  


---

331\. 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇



---

332\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



333\. _𑀦𑀸𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀢𑀡𑁆𑀳𑀸 𑀘,_  
_𑀲𑀁𑀬𑁄𑀚𑀦𑀜𑁆𑀘 𑀩𑀦𑁆𑀥𑀦𑀸;_  
_𑀅𑀩𑁆𑀪𑀸𑀳𑀢𑀼𑀟𑁆𑀟𑀺𑀢𑁄 𑀧𑀺𑀳𑀺𑀢𑁄,_  
_𑀇𑀘𑁆𑀙𑀸 𑀮𑁄𑀓𑁂𑀦 𑀢𑁂 𑀤𑀲𑀸𑀢𑀺𑁇_
