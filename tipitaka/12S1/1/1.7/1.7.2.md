# 𑁧𑁇𑁭𑁇𑁨 𑀘𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀘𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(62.)

313. _“𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀻𑀬𑀢𑀺 𑀮𑁄𑀓𑁄,_  
_𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀧𑀭𑀺𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀓𑀺𑀲𑁆𑀲𑀲𑁆𑀲𑀼 𑀏𑀓𑀥𑀫𑁆𑀫𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀯𑀲𑀫𑀦𑁆𑀯𑀕𑀽”𑀢𑀺𑁇_  


314. _“𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀦𑀻𑀬𑀢𑀺 𑀮𑁄𑀓𑁄,_  
_𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀥𑀫𑁆𑀫𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀯𑀲𑀫𑀦𑁆𑀯𑀕𑀽”𑀢𑀺𑁇_
