# 𑁧𑁇𑁭𑁇𑁮 𑀧𑀺𑀳𑀺𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀳𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(68.)

325. _“𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀧𑀺𑀳𑀺𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀉𑀟𑁆𑀟𑀺𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄”𑀢𑀺𑁇_  


326. _“𑀫𑀘𑁆𑀘𑀼𑀦𑀸 𑀧𑀺𑀳𑀺𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀤𑀼𑀓𑁆𑀔𑁂 𑀮𑁄𑀓𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀢𑀡𑁆𑀳𑀸𑀬 𑀉𑀟𑁆𑀟𑀺𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀚𑀭𑀸𑀬 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄”𑀢𑀺𑁇_
