# 𑁧𑁇𑁭𑁇𑁯 𑀇𑀘𑁆𑀙𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀇𑀘𑁆𑀙𑀸𑀲𑀼𑀢𑁆𑀢

(69.)

327\. _“𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀩𑀚𑁆𑀛𑀢𑀻 𑀮𑁄𑀓𑁄,_  
_𑀓𑀺𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑀸𑀬 𑀫𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀓𑀺𑀲𑁆𑀲𑀲𑁆𑀲𑀼 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀦𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


328\. _“𑀇𑀘𑁆𑀙𑀸𑀬 𑀩𑀚𑁆𑀛𑀢𑀻 𑀮𑁄𑀓𑁄,_  
_𑀇𑀘𑁆𑀙𑀸𑀯𑀺𑀦𑀬𑀸𑀬 𑀫𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀇𑀘𑁆𑀙𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀦𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_
