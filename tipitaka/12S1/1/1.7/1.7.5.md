# 𑁧𑁇𑁭𑁇𑁫 𑀩𑀦𑁆𑀥𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀩𑀦𑁆𑀥𑀦𑀲𑀼𑀢𑁆𑀢

(65.)

319\. _“𑀓𑀺𑀁𑀲𑀼 𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑁄 𑀮𑁄𑀓𑁄,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀸𑀭𑀡𑀁;_  
_𑀓𑀺𑀲𑁆𑀲𑀲𑁆𑀲𑀼 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀦𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


320\. _“𑀦𑀦𑁆𑀤𑀻𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑁄 𑀮𑁄𑀓𑁄,_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑁆𑀲 𑀯𑀺𑀘𑀸𑀭𑀡𑀁;_  
_𑀢𑀡𑁆𑀳𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀦𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺 𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_
