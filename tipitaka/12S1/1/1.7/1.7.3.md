# 𑁧𑁇𑁭𑁇𑁩 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀤𑁂𑀯𑀢𑀸𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀅𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀢𑀡𑁆𑀳𑀸𑀲𑀼𑀢𑁆𑀢

(63.)

315\. _“𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀻𑀬𑀢𑀺 𑀮𑁄𑀓𑁄,_  
_𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀧𑀭𑀺𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀓𑀺𑀲𑁆𑀲𑀲𑁆𑀲𑀼 𑀏𑀓𑀥𑀫𑁆𑀫𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀯𑀲𑀫𑀦𑁆𑀯𑀕𑀽”𑀢𑀺𑁇_  


316\. _“𑀢𑀡𑁆𑀳𑀸𑀬 𑀦𑀻𑀬𑀢𑀺 𑀮𑁄𑀓𑁄,_  
_𑀢𑀡𑁆𑀳𑀸𑀬 𑀧𑀭𑀺𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀡𑁆𑀳𑀸𑀬 𑀏𑀓𑀥𑀫𑁆𑀫𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀯𑀲𑀫𑀦𑁆𑀯𑀕𑀽”𑀢𑀺𑁇_
