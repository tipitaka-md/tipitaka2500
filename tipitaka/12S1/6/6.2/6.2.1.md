# 𑁬𑁇𑁨𑁇𑁧 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(182.)

1092\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀧𑁆𑀧𑀺𑀦𑀻𑀢𑀻𑀭𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑁄 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀲𑀧𑁆𑀧𑀺𑀦𑀻𑀢𑀻𑀭𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀅𑀪𑀸𑀲𑀺—

1093\. _“𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀚𑀦𑁂𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑁂 𑀕𑁄𑀢𑁆𑀢𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀦𑁄;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑁂”𑀢𑀺𑁇_  


1094\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄𑁇 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀲𑀢𑁆𑀣𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀦𑀗𑁆𑀓𑀼𑀫𑀸𑀭𑁄 “𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀫𑁂 𑀲𑀢𑁆𑀣𑀸”𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇
