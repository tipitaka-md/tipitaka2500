# 𑁩𑁇𑁧𑁇𑁬 𑀅𑀧𑁆𑀧𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀓𑁄𑀲𑀮𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀧𑁆𑀧𑀓𑀲𑀼𑀢𑁆𑀢

(117.)

605\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀬𑁂 𑀉𑀍𑀆𑀭𑁂 𑀉𑀍𑀆𑀭𑁂 𑀪𑁄𑀕𑁂 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀧𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼 𑀕𑁂𑀥𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀬𑁂 𑀉𑀍𑀆𑀭𑁂 𑀉𑀍𑀆𑀭𑁂 𑀪𑁄𑀕𑁂 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀧𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼 𑀘 𑀕𑁂𑀥𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀘 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻’”𑀢𑀺𑁇

606\. “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚𑁇 𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀢𑁆𑀢𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀬𑁂 𑀉𑀍𑀆𑀭𑁂 𑀉𑀍𑀆𑀭𑁂 𑀪𑁄𑀕𑁂 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀧𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼 𑀕𑁂𑀥𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀏𑀢𑁂𑀯 𑀩𑀳𑀼𑀢𑀭𑀸 𑀲𑀢𑁆𑀢𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀬𑁂 𑀉𑀍𑀆𑀭𑁂 𑀉𑀍𑀆𑀭𑁂 𑀪𑁄𑀕𑁂 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘𑁂𑀯 𑀧𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀓𑀸𑀫𑁂𑀲𑀼 𑀘 𑀕𑁂𑀥𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀘 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘…𑀧𑁂…

607\. _“𑀲𑀸𑀭𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑀪𑁄𑀕𑁂𑀲𑀼,_  
_𑀕𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀅𑀢𑀺𑀲𑀸𑀭𑀁 𑀦 𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀫𑀺𑀕𑀸 𑀓𑀽𑀝𑀁𑀯 𑀑𑀟𑁆𑀟𑀺𑀢𑀁;_  
_𑀧𑀘𑁆𑀙𑀸𑀲𑀁 𑀓𑀝𑀼𑀓𑀁 𑀳𑁄𑀢𑀺,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑀺𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑁄”𑀢𑀺𑁇_
