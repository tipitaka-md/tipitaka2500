# 𑁩𑁇𑁧𑁇𑁭 𑀅𑀟𑁆𑀟𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀓𑁄𑀲𑀮𑀲𑀁𑀬𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

### 𑀅𑀟𑁆𑀟𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(118.)

608\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀟𑁆𑀟𑀓𑀭𑀡𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑁂𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑁂𑀧𑀺 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑁂𑀧𑀺 𑀅𑀟𑁆𑀠𑁂 𑀫𑀳𑀤𑁆𑀥𑀦𑁂 𑀫𑀳𑀸𑀪𑁄𑀕𑁂 𑀧𑀳𑀽𑀢𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑁂 𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑁂 𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑁂 𑀓𑀸𑀫𑀳𑁂𑀢𑀼 𑀓𑀸𑀫𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀦𑁆𑀢𑁂𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀮𑀁 𑀤𑀸𑀦𑀺 𑀫𑁂 𑀅𑀟𑁆𑀟𑀓𑀭𑀡𑁂𑀦, 𑀪𑀤𑁆𑀭𑀫𑀼𑀔𑁄 𑀤𑀸𑀦𑀺 𑀅𑀟𑁆𑀟𑀓𑀭𑀡𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇

609\. “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚𑁇 𑀬𑁂𑀧𑀺 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀅𑀟𑁆𑀠𑀸 𑀫𑀳𑀤𑁆𑀥𑀦𑀸 𑀫𑀳𑀸𑀪𑁄𑀕𑀸 𑀧𑀳𑀽𑀢𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀸 𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑀸 𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀸 𑀓𑀸𑀫𑀳𑁂𑀢𑀼 𑀓𑀸𑀫𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀸𑀫𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀦𑁆𑀢𑀺; 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘…𑀧𑁂…

610\. _“𑀲𑀸𑀭𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑀪𑁄𑀕𑁂𑀲𑀼,_  
_𑀕𑀺𑀤𑁆𑀥𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀅𑀢𑀺𑀲𑀸𑀭𑀁 𑀦 𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀫𑀘𑁆𑀙𑀸 𑀔𑀺𑀧𑁆𑀧𑀁𑀯 𑀑𑀟𑁆𑀟𑀺𑀢𑀁;_  
_𑀧𑀘𑁆𑀙𑀸𑀲𑀁 𑀓𑀝𑀼𑀓𑀁 𑀳𑁄𑀢𑀺,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑀺𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑁄”𑀢𑀺𑁇_
