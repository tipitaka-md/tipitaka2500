# 𑁩𑁇𑁧𑁇𑁯 𑀬𑀜𑁆𑀜𑀲𑀼𑀢𑁆𑀢

* 𑀓𑁄𑀲𑀮𑀲𑀁𑀬𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀬𑀜𑁆𑀜𑀲𑀼𑀢𑁆𑀢

(120.)

615\. 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀺𑀤𑀸𑀦𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀘 𑀉𑀲𑀪𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀯𑀘𑁆𑀙𑀢𑀭𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀯𑀘𑁆𑀙𑀢𑀭𑀺𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀅𑀚𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀉𑀭𑀩𑁆𑀪𑀲𑀢𑀸𑀦𑀺 𑀣𑀽𑀡𑀽𑀧𑀦𑀻𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀬𑀜𑁆𑀜𑀢𑁆𑀣𑀸𑀬𑁇 𑀬𑁂𑀧𑀺𑀲𑁆𑀲 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀤𑀸𑀲𑀸𑀢𑀺 𑀯𑀸 𑀧𑁂𑀲𑁆𑀲𑀸𑀢𑀺 𑀯𑀸 𑀓𑀫𑁆𑀫𑀓𑀭𑀸𑀢𑀺 𑀯𑀸, 𑀢𑁂𑀧𑀺 𑀤𑀡𑁆𑀟𑀢𑀚𑁆𑀚𑀺𑀢𑀸 𑀪𑀬𑀢𑀚𑁆𑀚𑀺𑀢𑀸 𑀅𑀲𑁆𑀲𑀼𑀫𑀼𑀔𑀸 𑀭𑀼𑀤𑀫𑀸𑀦𑀸 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇

616\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀘 𑀉𑀲𑀪𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀯𑀘𑁆𑀙𑀢𑀭𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀯𑀘𑁆𑀙𑀢𑀭𑀺𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀅𑀚𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀘 𑀉𑀭𑀩𑁆𑀪𑀲𑀢𑀸𑀦𑀺 𑀣𑀽𑀡𑀽𑀧𑀦𑀻𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀬𑀜𑁆𑀜𑀢𑁆𑀣𑀸𑀬𑁇 𑀬𑁂𑀧𑀺𑀲𑁆𑀲 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀤𑀸𑀲𑀸𑀢𑀺 𑀯𑀸 𑀧𑁂𑀲𑁆𑀲𑀸𑀢𑀺 𑀯𑀸 𑀓𑀫𑁆𑀫𑀓𑀭𑀸𑀢𑀺 𑀯𑀸, 𑀢𑁂𑀧𑀺 𑀤𑀡𑁆𑀟𑀢𑀚𑁆𑀚𑀺𑀢𑀸 𑀪𑀬𑀢𑀚𑁆𑀚𑀺𑀢𑀸 𑀅𑀲𑁆𑀲𑀼𑀫𑀼𑀔𑀸 𑀭𑀼𑀤𑀫𑀸𑀦𑀸 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

617\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

618\. _“𑀅𑀲𑁆𑀲𑀫𑁂𑀥𑀁 𑀧𑀼𑀭𑀺𑀲𑀫𑁂𑀥𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀧𑀸𑀲𑀁 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁;_  
_𑀦𑀺𑀭𑀕𑁆𑀕𑀍𑀅𑀁 𑀫𑀳𑀸𑀭𑀫𑁆𑀪𑀸,_  
_𑀦 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀸𑁇_  


619\. _𑀅𑀚𑁂𑀍𑀅𑀓𑀸 𑀘 𑀕𑀸𑀯𑁄 𑀘,_  
_𑀯𑀺𑀯𑀺𑀥𑀸 𑀬𑀢𑁆𑀣 𑀳𑀜𑁆𑀜𑀭𑁂;_  
_𑀦 𑀢𑀁 𑀲𑀫𑁆𑀫𑀕𑁆𑀕𑀢𑀸 𑀬𑀜𑁆𑀜𑀁,_  
_𑀉𑀧𑀬𑀦𑁆𑀢𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


620\. _𑀬𑁂 𑀘 𑀬𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑀸𑀭𑀫𑁆𑀪𑀸,_  
_𑀬𑀚𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀓𑀼𑀮𑀁 𑀲𑀤𑀸;_  
_𑀅𑀚𑁂𑀍𑀅𑀓𑀸 𑀘 𑀕𑀸𑀯𑁄 𑀘,_  
_𑀯𑀺𑀯𑀺𑀥𑀸 𑀦𑁂𑀢𑁆𑀣 𑀳𑀜𑁆𑀜𑀭𑁂;_  
_𑀏𑀢𑀁 𑀲𑀫𑁆𑀫𑀕𑁆𑀕𑀢𑀸 𑀬𑀜𑁆𑀜𑀁,_  
_𑀉𑀧𑀬𑀦𑁆𑀢𑀺 𑀫𑀳𑁂𑀲𑀺𑀦𑁄𑁇_  


621\. _𑀏𑀢𑀁 𑀬𑀚𑁂𑀣 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀏𑀲𑁄 𑀬𑀜𑁆𑀜𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑁄;_  
_𑀏𑀢𑀜𑁆𑀳𑀺 𑀬𑀚𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀲𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀧𑀸𑀧𑀺𑀬𑁄;_  
_𑀬𑀜𑁆𑀜𑁄 𑀘 𑀯𑀺𑀧𑀼𑀮𑁄 𑀳𑁄𑀢𑀺,_  
_𑀧𑀲𑀻𑀤𑀦𑁆𑀢𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸”𑀢𑀺𑁇_
