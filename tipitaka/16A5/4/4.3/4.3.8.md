# 𑁪𑁇𑁩𑁇𑁮 𑀭𑀸𑀚𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑀸𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀭𑀸𑀚𑀸𑀲𑀼𑀢𑁆𑀢

(178.)

1030. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀁 𑀧𑀳𑀸𑀬 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂𑁇 𑀫𑀬𑀸𑀧𑀺 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀯 𑀤𑀺𑀝𑁆𑀞𑀁 𑀦 𑀲𑀼𑀢𑀁—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀁 𑀧𑀳𑀸𑀬 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀢𑀣𑁂𑀯 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀇𑀢𑁆𑀣𑀺𑀁 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀻𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸’”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑀜𑁆𑀘 𑀲𑀼𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇

1031. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀳𑀸𑀬 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂𑁇 𑀫𑀬𑀸𑀧𑀺 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀯 𑀤𑀺𑀝𑁆𑀞𑀁 𑀦 𑀲𑀼𑀢𑀁—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀳𑀸𑀬 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀢𑀣𑁂𑀯 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀕𑀸𑀫𑀸 𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀸 𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑀻𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸’”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑀜𑁆𑀘 𑀲𑀼𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇

1034. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀳𑀸𑀬 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀻’”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂𑁇 𑀫𑀬𑀸𑀧𑀺 𑀔𑁄 𑀏𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑁂𑀯 𑀤𑀺𑀝𑁆𑀞𑀁 𑀦 𑀲𑀼𑀢𑀁—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀳𑀸𑀬 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀢𑀣𑁂𑀯 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀇𑀢𑁆𑀣𑀺𑀁 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺; 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀸 𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀸 𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑀺; 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀢𑁆𑀣𑀻𑀲𑀼 𑀧𑀭𑀓𑀼𑀫𑀸𑀭𑀻𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀺; 𑀅𑀬𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀧𑀪𑀜𑁆𑀚𑀻𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀳𑁂𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀺 𑀯𑀸 𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀺 𑀯𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀯𑀸 𑀬𑀣𑀸𑀧𑀘𑁆𑀘𑀬𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂𑀳𑀺 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸 𑀲𑀼𑀢𑀁 𑀯𑀸’”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀢𑀜𑁆𑀘 𑀲𑀼𑀬𑁆𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀘𑀸”𑀢𑀺𑁇
