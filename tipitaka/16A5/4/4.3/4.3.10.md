# 𑁪𑁇𑁩𑁇𑁧𑁦 𑀕𑀯𑁂𑀲𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀉𑀧𑀸𑀲𑀓𑀯𑀕𑁆𑀕

* 𑀕𑀯𑁂𑀲𑀻𑀲𑀼𑀢𑁆𑀢

(180.)

1059\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀢𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀫𑀳𑀦𑁆𑀢𑀁 𑀲𑀸𑀮𑀯𑀦𑀁; 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀢𑀁 𑀲𑀸𑀮𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀲𑀸𑀮𑀯𑀦𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑁆𑀯𑀸𑀓𑀸𑀲𑀺𑁇

1060\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀓𑀫𑁆𑀫𑀸𑀬? 𑀦 𑀅𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀸 𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀓𑀫𑁆𑀫𑀸𑀬? 𑀦 𑀅𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀸 𑀲𑀺𑀢𑀁 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1061\. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀦𑀕𑀭𑀁 𑀅𑀳𑁄𑀲𑀺 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀩𑀳𑀼𑀚𑀦𑀁 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀁𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀦𑀕𑀭𑀁 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀕𑀯𑁂𑀲𑀻 𑀦𑀸𑀫 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻𑁇 𑀕𑀯𑁂𑀲𑀺𑀦𑀸 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀧𑀝𑀺𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀺 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀻𑀮𑁂𑀲𑀼 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀺𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑁄 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸, 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻, 𑀇𑀫𑀸𑀦𑀺 𑀘 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄𑁇 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀲𑀫𑀲𑀫𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀁; 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀸𑀬𑀸’𑀢𑀺𑁇

1062\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀻 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑁂𑀦 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀁 𑀥𑀸𑀭𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑁆𑀬𑁄 𑀔𑁄 𑀕𑀯𑁂𑀲𑀻 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑁄 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸𑁇 𑀅𑀬𑁆𑀬𑁄 𑀳𑀺 𑀦𑀸𑀫 𑀕𑀯𑁂𑀲𑀻 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀺𑀫𑀗𑁆𑀕𑀁 𑀧𑀦 𑀫𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀬𑁂𑀦 𑀕𑀯𑁂𑀲𑀻 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀕𑀯𑁂𑀲𑀺𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀅𑀬𑁆𑀬𑁄 𑀕𑀯𑁂𑀲𑀻 𑀇𑀫𑀸𑀦𑀺𑀧𑀺 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄 𑀥𑀸𑀭𑁂𑀢𑀽’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀺𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑁄 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸, 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻, 𑀇𑀫𑀸𑀦𑀺𑀧𑀺 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑁂𑀲𑀼 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀺𑀦𑁄𑁇 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀲𑀫𑀲𑀫𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀁; 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀸𑀬𑀸’𑀢𑀺𑁇

1065\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀻 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆’𑀢𑀺𑁇 𑀅𑀮𑀢𑁆𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀻 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀢𑁆𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀅𑀘𑀺𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇

1066\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀬𑁆𑀬𑁄 𑀔𑁄 𑀕𑀯𑁂𑀲𑀻 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑁄 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸𑁇 𑀅𑀬𑁆𑀬𑁄 𑀳𑀺 𑀦𑀸𑀫 𑀕𑀯𑁂𑀲𑀻 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀺𑀫𑀗𑁆𑀕𑀁 𑀧𑀦 𑀫𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀬𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  ‘𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆’𑀢𑀺𑁇 𑀅𑀮𑀪𑀺𑀁𑀲𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑀸𑀦𑀺 𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀅𑀮𑀪𑀺𑀁𑀲𑀼 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇

1067\. 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀕𑀯𑁂𑀲𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀫𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀅𑀳𑁄 𑀯𑀢𑀺𑀫𑀸𑀦𑀺𑀧𑀺 𑀧𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀲𑁆𑀲𑀼 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀺𑀦𑁄’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑀸 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀸 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀁𑀲𑀼𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀺𑀁𑀲𑀼𑁇

1068\. 𑀇𑀢𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺 𑀕𑀯𑁂𑀲𑀻𑀧𑀫𑀼𑀔𑀸𑀦𑀺 𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺 𑀧𑀡𑀻𑀢𑀧𑀡𑀻𑀢𑀁 𑀯𑀸𑀬𑀫𑀫𑀸𑀦𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀲𑀘𑁆𑀙𑀸𑀓𑀁𑀲𑀼𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺 𑀧𑀡𑀻𑀢𑀧𑀡𑀻𑀢𑀁 𑀯𑀸𑀬𑀫𑀫𑀸𑀦𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
