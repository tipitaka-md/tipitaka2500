# 𑁪𑁇𑁩𑁇𑁬 𑀧𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑀸𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(176.)

1023\. 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1024\. “𑀢𑀼𑀫𑁆𑀳𑁂 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦𑁇 𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀢𑀸𑀯𑀢𑀓𑁂𑀦𑁂𑀯 𑀢𑀼𑀝𑁆𑀞𑀺 𑀓𑀭𑀡𑀻𑀬𑀸—  ‘𑀫𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀕𑀳𑀧𑀢𑀺, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀫𑀬𑀁 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

1025\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀬𑀸𑀯 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀢𑀼𑀫𑁆𑀳𑁂 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦𑁇 𑀦 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀢𑀸𑀯𑀢𑀓𑁂𑀦𑁂𑀯 𑀢𑀼𑀝𑁆𑀞𑀺 𑀓𑀭𑀡𑀻𑀬𑀸—  𑀫𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦𑀸𑀢𑀺𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀕𑀳𑀧𑀢𑀺, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  𑀓𑀺𑀦𑁆𑀢𑀺 𑀫𑀬𑀁 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀜𑁆𑀘𑀲𑁆𑀲 𑀞𑀸𑀦𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀇𑀫𑀸𑀦𑀺𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀞𑀸𑀦𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1026\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁇 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀫𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀧𑀜𑁆𑀘𑀲𑁆𑀲 𑀞𑀸𑀦𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀓𑀸𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁, 𑀢𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀫𑀬𑁂 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀧𑀻𑀢𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀇𑀫𑀸𑀦𑀺𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀞𑀸𑀦𑀸𑀦𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀳𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
