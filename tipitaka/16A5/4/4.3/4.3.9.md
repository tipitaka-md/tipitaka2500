# 𑁪𑁇𑁩𑁇𑁯 𑀕𑀺𑀳𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑀸𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀺𑀳𑀺𑀲𑀼𑀢𑁆𑀢

(179.)

1036. 𑀅𑀣 𑀔𑁄 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑀲𑀢𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀁 𑀓𑀜𑁆𑀘𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀣 𑀕𑀺𑀳𑀺𑀁 𑀑𑀤𑀸𑀢𑀯𑀲𑀦𑀁 𑀧𑀜𑁆𑀘𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀲𑀁𑀯𑀼𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀁 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀺𑀁 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀺𑀁, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑀢𑀺𑁇

1037. 𑀓𑀢𑀫𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀲𑀁𑀯𑀼𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀲𑀁𑀯𑀼𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

1038. 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻? 𑀇𑀥, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺, 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑀲𑁆𑀲 𑀧𑀞𑀫𑁄 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑁇

1039. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀥𑀫𑁆𑀫𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀅𑀓𑀸𑀮𑀺𑀓𑁄 𑀏𑀳𑀺𑀧𑀲𑁆𑀲𑀺𑀓𑁄 𑀑𑀧𑀦𑁂𑀬𑁆𑀬𑀺𑀓𑁄 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄 𑀯𑀺𑀜𑁆𑀜𑀽𑀳𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑁇

1040. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀯𑁂𑀘𑁆𑀘𑀧𑁆𑀧𑀲𑀸𑀤𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀉𑀚𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀜𑀸𑀬𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄, 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀏𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀅𑀬𑀫𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑁄 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑁇

1041. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀭𑀺𑀬𑀓𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀮𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀔𑀡𑁆𑀟𑁂𑀳𑀺 𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑁂𑀳𑀺 𑀅𑀲𑀩𑀮𑁂𑀳𑀺 𑀅𑀓𑀫𑁆𑀫𑀸𑀲𑁂𑀳𑀺 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑁂𑀳𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁂𑀳𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁂𑀳𑀺𑁇 𑀅𑀬𑀫𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑀦𑀸𑀬𑁇 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇

1042. 𑀬𑀁 𑀓𑀜𑁆𑀘𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀣 𑀕𑀺𑀳𑀺𑀁 𑀑𑀤𑀸𑀢𑀯𑀲𑀦𑀁—  𑀇𑀫𑁂𑀲𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼 𑀲𑀁𑀯𑀼𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀁, 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀁 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀺𑀁 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀺𑀁, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬—  ‘𑀔𑀻𑀡𑀦𑀺𑀭𑀬𑁄𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺 𑀔𑀻𑀡𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄 𑀔𑀻𑀡𑀸𑀧𑀸𑀬𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁄, 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄’𑀢𑀺𑁇

1043. _𑀦𑀺𑀭𑀬𑁂𑀲𑀼 𑀪𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀸𑀧𑀸𑀦𑀺 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑁂;_  
_𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀸𑀬,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑁂𑁇_  


1044. _𑀦 𑀳𑀺𑀁𑀲𑁂 𑀧𑀸𑀡𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀧𑀭𑀓𑁆𑀓𑀫𑁂;_  
_𑀫𑀼𑀲𑀸 𑀘 𑀦 𑀪𑀡𑁂 𑀚𑀸𑀦𑀁,_  
_𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀧𑀭𑀸𑀫𑀲𑁂𑁇_  


1045. _𑀲𑁂𑀳𑀺 𑀤𑀸𑀭𑁂𑀳𑀺 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀧𑀭𑀤𑀸𑀭𑀜𑁆𑀘 𑀆𑀭𑀫𑁂;_  
_𑀫𑁂𑀭𑀬𑀁 𑀯𑀸𑀭𑀼𑀡𑀺𑀁 𑀚𑀦𑁆𑀢𑀼,_  
_𑀦 𑀧𑀺𑀯𑁂 𑀘𑀺𑀢𑁆𑀢𑀫𑁄𑀳𑀦𑀺𑀁𑁇_  


1046. _𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑁂𑀬𑁆𑀬 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀥𑀫𑁆𑀫𑀜𑁆𑀘𑀸𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑀬𑁂;_  
_𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀁 𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀤𑁂𑀯𑀮𑁄𑀓𑀸𑀬 𑀪𑀸𑀯𑀬𑁂𑁇_  


1047. _𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁂 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁂,_  
_𑀧𑀼𑀜𑁆𑀜𑀢𑁆𑀣𑀲𑁆𑀲 𑀚𑀺𑀕𑀻𑀲𑀢𑁄;_  
_𑀲𑀦𑁆𑀢𑁂𑀲𑀼 𑀧𑀞𑀫𑀁 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀧𑀼𑀮𑀸 𑀳𑁄𑀢𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑁇_  


1048. _𑀲𑀦𑁆𑀢𑁄 𑀳𑀯𑁂 𑀧𑀯𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂;_  
_𑀇𑀢𑀺 𑀓𑀡𑁆𑀳𑀸𑀲𑀼 𑀲𑁂𑀢𑀸𑀲𑀼,_  
_𑀭𑁄𑀳𑀺𑀡𑀻𑀲𑀼 𑀳𑀭𑀻𑀲𑀼 𑀯𑀸𑁇_  


1049. _𑀓𑀫𑁆𑀫𑀸𑀲𑀸𑀲𑀼 𑀲𑀭𑀽𑀧𑀸𑀲𑀼,_  
_𑀕𑁄𑀲𑀼 𑀧𑀸𑀭𑁂𑀯𑀢𑀸𑀲𑀼 𑀯𑀸;_  
_𑀬𑀸𑀲𑀼 𑀓𑀸𑀲𑀼𑀘𑀺 𑀏𑀢𑀸𑀲𑀼,_  
_𑀤𑀦𑁆𑀢𑁄 𑀚𑀸𑀬𑀢𑀺 𑀧𑀼𑀗𑁆𑀕𑀯𑁄𑁇_  


1050. _𑀥𑁄𑀭𑀬𑁆𑀳𑁄 𑀩𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀚𑀯𑀦𑀺𑀓𑁆𑀓𑀫𑁄;_  
_𑀢𑀫𑁂𑀯 𑀪𑀸𑀭𑁂 𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀭𑁂𑁇_  


1051. _𑀏𑀯𑀫𑁂𑀯𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀓𑀺𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑀺 𑀚𑀸𑀢𑀺𑀬𑁂;_  
_𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀯𑁂𑀲𑁆𑀲𑁂,_  
_𑀲𑀼𑀤𑁆𑀤𑁂 𑀘𑀡𑁆𑀟𑀸𑀮𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑁂𑁇_  


1052. _𑀬𑀸𑀲𑀼 𑀓𑀸𑀲𑀼𑀘𑀺 𑀏𑀢𑀸𑀲𑀼,_  
_𑀤𑀦𑁆𑀢𑁄 𑀚𑀸𑀬𑀢𑀺 𑀲𑀼𑀩𑁆𑀩𑀢𑁄;_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀻 𑀳𑀺𑀭𑀻𑀫𑀦𑁄𑁇_  


1053. _𑀧𑀳𑀻𑀦𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀻;_  
_𑀧𑀦𑁆𑀦𑀪𑀸𑀭𑁄 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁄 𑀅𑀦𑀸𑀲𑀯𑁄𑁇_  


1054. _𑀧𑀸𑀭𑀕𑀽 𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀯𑀺𑀭𑀚𑁂 𑀔𑁂𑀢𑁆𑀢𑁂,_  
_𑀯𑀺𑀧𑀼𑀮𑀸 𑀳𑁄𑀢𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑁇_  


1055. _𑀩𑀸𑀮𑀸 𑀘 𑀅𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀸,_  
_𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑀸 𑀅𑀲𑁆𑀲𑀼𑀢𑀸𑀯𑀺𑀦𑁄;_  
_𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀤𑀤𑀦𑁆𑀢𑀺 𑀤𑀸𑀦𑀸𑀦𑀺,_  
_𑀦 𑀳𑀺 𑀲𑀦𑁆𑀢𑁂 𑀉𑀧𑀸𑀲𑀭𑁂𑁇_  


1056. _𑀬𑁂 𑀘 𑀲𑀦𑁆𑀢𑁂 𑀉𑀧𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁂 𑀥𑀻𑀭𑀲𑀫𑁆𑀫𑀢𑁂;_  
_𑀲𑀤𑁆𑀥𑀸 𑀘 𑀦𑁂𑀲𑀁 𑀲𑀼𑀕𑀢𑁂,_  
_𑀫𑀽𑀮𑀚𑀸𑀢𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇_  


1057. _𑀤𑁂𑀯𑀮𑁄𑀓𑀜𑁆𑀘 𑀢𑁂 𑀬𑀦𑁆𑀢𑀺,_  
_𑀓𑀼𑀮𑁂 𑀯𑀸 𑀇𑀥 𑀚𑀸𑀬𑀭𑁂;_  
_𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸”𑀢𑀺𑁇_  


---

1058. 𑀦𑀯𑀫𑀁𑁇
