# 𑁪𑁇𑁧𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀸𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀸𑀫𑀲𑀼𑀢𑁆𑀢

(152.)

877\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀼𑀡𑀦𑁆𑀢𑁄𑀧𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀅𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀬𑀸𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀣𑀁 𑀧𑀭𑀺𑀪𑁄𑀢𑀺, 𑀓𑀣𑀺𑀓𑀁 𑀧𑀭𑀺𑀪𑁄𑀢𑀺, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀪𑁄𑀢𑀺, 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺 𑀚𑀍𑀑 𑀏𑀍𑀅𑀫𑀽𑀕𑁄, 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀫𑀸𑀦𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀼𑀡𑀦𑁆𑀢𑁄𑀧𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀅𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀬𑀸𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀁𑁇

878\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀼𑀡𑀦𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀬𑀸𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀓𑀣𑀁 𑀧𑀭𑀺𑀪𑁄𑀢𑀺, 𑀦 𑀓𑀣𑀺𑀓𑀁 𑀧𑀭𑀺𑀪𑁄𑀢𑀺, 𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀪𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀚𑀍𑀑 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑁄, 𑀦 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀫𑀸𑀦𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀼𑀡𑀦𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀪𑀩𑁆𑀩𑁄 𑀦𑀺𑀬𑀸𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇
