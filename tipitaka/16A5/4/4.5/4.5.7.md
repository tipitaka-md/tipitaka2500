# 𑁪𑁇𑁫𑁇𑁭 𑀯𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀯𑀲𑁆𑀲𑀲𑀼𑀢𑁆𑀢

(197.)

1166\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸, 𑀬𑀁 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀉𑀧𑀭𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀓𑀸𑀲𑁂 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼 𑀧𑀓𑀼𑀧𑁆𑀧𑀢𑀺𑁇 𑀢𑁂𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂𑀖𑀸 𑀧𑀝𑀺𑀯𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑁄 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄, 𑀬𑀁 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑀢𑀺𑁇

1167\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀭𑀺 𑀆𑀓𑀸𑀲𑁂 𑀯𑀸𑀬𑁄𑀥𑀸𑀢𑀼 𑀧𑀓𑀼𑀧𑁆𑀧𑀢𑀺𑁇 𑀢𑁂𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑁂𑀖𑀸 𑀧𑀝𑀺𑀯𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄, 𑀬𑀁 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑀢𑀺𑁇

1168\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀳𑀼 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀧𑀸𑀡𑀺𑀦𑀸 𑀉𑀤𑀓𑀁 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀙𑀟𑁆𑀟𑁂𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑁄 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄, 𑀬𑀁 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑀢𑀺𑁇

1169\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀯𑀮𑀸𑀳𑀓𑀸 𑀤𑁂𑀯𑀸 𑀧𑀫𑀢𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄, 𑀬𑀁 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑀢𑀺𑁇

1170\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄, 𑀬𑀁 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀸, 𑀬𑀁 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀓𑀫𑀢𑀻”𑀢𑀺𑁇
