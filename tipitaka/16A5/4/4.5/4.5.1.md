# 𑁪𑁇𑁫𑁇𑁧 𑀲𑁄𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀡𑀲𑀼𑀢𑁆𑀢

(191.)

1098\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑁄𑀭𑀸𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀸 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀦𑀔𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁𑀬𑁂𑀯 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑀔𑀸 𑀲𑀼𑀦𑀔𑀺𑀁𑀬𑁂𑀯 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀲𑀼𑀦𑀔𑀺𑀁𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑁄 𑀧𑁄𑀭𑀸𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑁄 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀦𑀔𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼𑁇

1099\. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀉𑀢𑀼𑀦𑀺𑀁𑀬𑁂𑀯 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀦𑀼𑀢𑀼𑀦𑀺𑀁𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀉𑀢𑀼𑀦𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀼𑀢𑀼𑀦𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑀔𑀸 𑀲𑀼𑀦𑀔𑀺𑀁 𑀉𑀢𑀼𑀦𑀺𑀁𑀬𑁂𑀯 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀦𑀼𑀢𑀼𑀦𑀺𑀁𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑁄𑀭𑀸𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑁄 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀦𑀔𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼𑁇

1100\. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀦𑁂𑀯 𑀓𑀺𑀡𑀦𑁆𑀢𑀺 𑀦𑁄 𑀯𑀺𑀓𑁆𑀓𑀺𑀡𑀦𑁆𑀢𑀺, 𑀲𑀫𑁆𑀧𑀺𑀬𑁂𑀦𑁂𑀯 𑀲𑀁𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀸𑀬 𑀲𑀫𑁆𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀓𑀺𑀡𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀓𑁆𑀓𑀺𑀡𑀦𑁆𑀢𑀺𑀧𑀺, 𑀲𑀫𑁆𑀧𑀺𑀬𑁂𑀦𑀧𑀺 𑀲𑀁𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀸𑀬 𑀲𑀫𑁆𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑀔𑀸 𑀲𑀼𑀦𑀔𑀺𑀁 𑀦𑁂𑀯 𑀓𑀺𑀡𑀦𑁆𑀢𑀺 𑀦𑁄 𑀯𑀺𑀓𑁆𑀓𑀺𑀡𑀦𑁆𑀢𑀺, 𑀲𑀫𑁆𑀧𑀺𑀬𑁂𑀦𑁂𑀯 𑀲𑀁𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀸𑀬 𑀲𑀫𑁆𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑁄 𑀧𑁄𑀭𑀸𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑁄 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀦𑀔𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼𑁇

1101\. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀦 𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀥𑀦𑀲𑁆𑀲𑀧𑀺 𑀥𑀜𑁆𑀜𑀲𑁆𑀲𑀧𑀺 𑀭𑀚𑀢𑀲𑁆𑀲𑀧𑀺 𑀚𑀸𑀢𑀭𑀽𑀧𑀲𑁆𑀲𑀧𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀥𑀦𑀲𑁆𑀲𑀧𑀺 𑀥𑀜𑁆𑀜𑀲𑁆𑀲𑀧𑀺 𑀭𑀚𑀢𑀲𑁆𑀲𑀧𑀺 𑀚𑀸𑀢𑀭𑀽𑀧𑀲𑁆𑀲𑀧𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑀔𑀸 𑀦 𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀥𑀦𑀲𑁆𑀲𑀧𑀺 𑀥𑀜𑁆𑀜𑀲𑁆𑀲𑀧𑀺 𑀭𑀚𑀢𑀲𑁆𑀲𑀧𑀺 𑀚𑀸𑀢𑀭𑀽𑀧𑀲𑁆𑀲𑀧𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑁄𑀭𑀸𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑁄 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀦𑀔𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼𑁇

1102\. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀸𑀬𑀁 𑀲𑀸𑀬𑀫𑀸𑀲𑀸𑀬 𑀧𑀸𑀢𑁄 𑀧𑀸𑀢𑀭𑀸𑀲𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀉𑀤𑀭𑀸𑀯𑀤𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑀔𑀸 𑀲𑀸𑀬𑀁 𑀲𑀸𑀬𑀫𑀸𑀲𑀸𑀬 𑀧𑀸𑀢𑁄 𑀧𑀸𑀢𑀭𑀸𑀲𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀧𑁄𑀭𑀸𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑁄 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀦𑀔𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀧𑁄𑀭𑀸𑀡𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀫𑁆𑀫𑀸 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀦𑀔𑁂𑀲𑀼 𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀽”𑀢𑀺𑁇
