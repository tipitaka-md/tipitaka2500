# 𑁪𑁇𑁫𑁇𑁨 𑀤𑁄𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀤𑁄𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(192.)

1104. 𑀅𑀣 𑀔𑁄 𑀤𑁄𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀤𑁄𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

1105. “𑀲𑀼𑀢𑀁 𑀫𑁂𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫—  ‘𑀦 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀚𑀺𑀡𑁆𑀡𑁂 𑀯𑀼𑀟𑁆𑀠𑁂 𑀫𑀳𑀮𑁆𑀮𑀓𑁂 𑀅𑀤𑁆𑀥𑀕𑀢𑁂 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀻’𑀢𑀺𑁇 𑀢𑀬𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀢𑀣𑁂𑀯𑁇 𑀦 𑀳𑀺 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀚𑀺𑀡𑁆𑀡𑁂 𑀯𑀼𑀟𑁆𑀠𑁂 𑀫𑀳𑀮𑁆𑀮𑀓𑁂 𑀅𑀤𑁆𑀥𑀕𑀢𑁂 𑀯𑀬𑁄𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀺 𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸 𑀆𑀲𑀦𑁂𑀦 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺𑁇 𑀢𑀬𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀫𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀦𑁄, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀲𑀻”𑀢𑀺? “𑀬𑀜𑁆𑀳𑀺 𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀪𑀢𑁄 𑀲𑀼𑀚𑀸𑀢𑁄—  𑀫𑀸𑀢𑀺𑀢𑁄 𑀘 𑀧𑀺𑀢𑀺𑀢𑁄 𑀘, 𑀲𑀁𑀲𑀼𑀤𑁆𑀥𑀕𑀳𑀡𑀺𑀓𑁄, 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀧𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑁄 𑀚𑀸𑀢𑀺𑀯𑀸𑀤𑁂𑀦, 𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦𑀁 𑀧𑀸𑀭𑀕𑀽 𑀲𑀦𑀺𑀖𑀡𑁆𑀟𑀼𑀓𑁂𑀝𑀼𑀪𑀸𑀦𑀁 𑀲𑀸𑀓𑁆𑀔𑀭𑀧𑁆𑀧𑀪𑁂𑀤𑀸𑀦𑀁 𑀇𑀢𑀺𑀳𑀸𑀲𑀧𑀜𑁆𑀘𑀫𑀸𑀦𑀁, 𑀧𑀤𑀓𑁄 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁄 𑀮𑁄𑀓𑀸𑀬𑀢𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑁂𑀲𑀼 𑀅𑀦𑀯𑀬𑁄’𑀢𑀺, 𑀫𑀫𑁂𑀯 𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀪𑀢𑁄 𑀲𑀼𑀚𑀸𑀢𑁄—  𑀫𑀸𑀢𑀺𑀢𑁄 𑀘 𑀧𑀺𑀢𑀺𑀢𑁄 𑀘, 𑀲𑀁𑀲𑀼𑀤𑁆𑀥𑀕𑀳𑀡𑀺𑀓𑁄, 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀧𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑁄 𑀚𑀸𑀢𑀺𑀯𑀸𑀤𑁂𑀦, 𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀫𑀦𑁆𑀢𑀥𑀭𑁄, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦𑀁 𑀧𑀸𑀭𑀕𑀽 𑀲𑀦𑀺𑀖𑀡𑁆𑀟𑀼𑀓𑁂𑀝𑀼𑀪𑀸𑀦𑀁 𑀲𑀸𑀓𑁆𑀔𑀭𑀧𑁆𑀧𑀪𑁂𑀤𑀸𑀦𑀁 𑀇𑀢𑀺𑀳𑀸𑀲𑀧𑀜𑁆𑀘𑀫𑀸𑀦𑀁, 𑀧𑀤𑀓𑁄 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁄 𑀮𑁄𑀓𑀸𑀬𑀢𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑁂𑀲𑀼 𑀅𑀦𑀯𑀬𑁄”𑀢𑀺𑁇

1106. “𑀬𑁂 𑀔𑁄 𑀢𑁂, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀇𑀲𑀬𑁄 𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀯𑀢𑁆𑀢𑀸𑀭𑁄, 𑀬𑁂𑀲𑀫𑀺𑀤𑀁 𑀏𑀢𑀭𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀧𑁄𑀭𑀸𑀡𑀁 𑀫𑀦𑁆𑀢𑀧𑀤𑀁 𑀕𑀻𑀢𑀁 𑀧𑀯𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀺𑀳𑀺𑀢𑀁 𑀢𑀤𑀦𑀼𑀕𑀸𑀬𑀦𑁆𑀢𑀺 𑀢𑀤𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀪𑀸𑀲𑀺𑀢𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀲𑀚𑁆𑀛𑀸𑀬𑀺𑀢𑀫𑀦𑀼𑀲𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀯𑀸𑀘𑀺𑀢𑀫𑀦𑀼𑀯𑀸𑀘𑁂𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀅𑀝𑁆𑀞𑀓𑁄, 𑀯𑀸𑀫𑀓𑁄, 𑀯𑀸𑀫𑀤𑁂𑀯𑁄, 𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑀢𑁆𑀢𑁄, 𑀬𑀫𑀤𑀕𑁆𑀕𑀺, 𑀅𑀗𑁆𑀕𑀻𑀭𑀲𑁄, 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑁄, 𑀓𑀲𑁆𑀲𑀧𑁄, 𑀪𑀕𑀼; 𑀢𑁆𑀬𑀸𑀲𑁆𑀲𑀼𑀫𑁂 𑀧𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺—  𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑀫𑀁, 𑀤𑁂𑀯𑀲𑀫𑀁, 𑀫𑀭𑀺𑀬𑀸𑀤𑀁, 𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀫𑀭𑀺𑀬𑀸𑀤𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀘𑀡𑁆𑀟𑀸𑀮𑀁𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀫𑀁𑁇 𑀢𑁂𑀲𑀁 𑀢𑁆𑀯𑀁 𑀤𑁄𑀡, 𑀓𑀢𑀫𑁄”𑀢𑀺?

1107. “𑀦 𑀔𑁄 𑀫𑀬𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀧𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀚𑀸𑀦𑀸𑀫, 𑀅𑀣 𑀔𑁄 𑀫𑀬𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀢𑁆𑀯𑁂𑀯 𑀚𑀸𑀦𑀸𑀫𑁇 𑀲𑀸𑀥𑀼 𑀫𑁂 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼 𑀬𑀣𑀸 𑀅𑀳𑀁 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀲𑀼𑀡𑁄𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀤𑁄𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1108. “𑀓𑀣𑀜𑁆𑀘, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑀫𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀪𑀢𑁄 𑀲𑀼𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀫𑀸𑀢𑀺𑀢𑁄 𑀘 𑀧𑀺𑀢𑀺𑀢𑁄 𑀘, 𑀲𑀁𑀲𑀼𑀤𑁆𑀥𑀕𑀳𑀡𑀺𑀓𑁄, 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀧𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑁄 𑀚𑀸𑀢𑀺𑀯𑀸𑀤𑁂𑀦𑁇 𑀲𑁄 𑀅𑀝𑁆𑀞𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑁄𑀫𑀸𑀭𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺 𑀫𑀦𑁆𑀢𑁂 𑀅𑀥𑀻𑀬𑀫𑀸𑀦𑁄𑁇 𑀅𑀝𑁆𑀞𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑁄𑀫𑀸𑀭𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀦𑁆𑀢𑁂 𑀅𑀥𑀻𑀬𑀺𑀢𑁆𑀯𑀸 𑀆𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀆𑀘𑀭𑀺𑀬𑀥𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀦𑁂𑀯, 𑀦𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦𑁇

1116. 𑀓𑀣𑀜𑁆𑀘, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀘𑀡𑁆𑀟𑀸𑀮𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀪𑀢𑁄 𑀲𑀼𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀫𑀸𑀢𑀺𑀢𑁄 𑀘 𑀧𑀺𑀢𑀺𑀢𑁄 𑀘, 𑀲𑀁𑀲𑀼𑀤𑁆𑀥𑀕𑀳𑀡𑀺𑀓𑁄, 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀸 𑀧𑀺𑀢𑀸𑀫𑀳𑀬𑀼𑀕𑀸 𑀅𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀦𑀼𑀧𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑁄 𑀚𑀸𑀢𑀺𑀯𑀸𑀤𑁂𑀦𑁇 𑀲𑁄 𑀅𑀝𑁆𑀞𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑁄𑀫𑀸𑀭𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺 𑀫𑀦𑁆𑀢𑁂 𑀅𑀥𑀻𑀬𑀫𑀸𑀦𑁄𑁇 𑀅𑀝𑁆𑀞𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑁄𑀫𑀸𑀭𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀦𑁆𑀢𑁂 𑀅𑀥𑀻𑀬𑀺𑀢𑁆𑀯𑀸 𑀆𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀆𑀘𑀭𑀺𑀬𑀥𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀦𑀧𑀺 𑀅𑀥𑀫𑁆𑀫𑁂𑀦𑀧𑀺 𑀓𑀲𑀺𑀬𑀸𑀧𑀺 𑀯𑀡𑀺𑀚𑁆𑀚𑀸𑀬𑀧𑀺 𑀕𑁄𑀭𑀓𑁆𑀔𑁂𑀦𑀧𑀺 𑀇𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀦𑀧𑀺 𑀭𑀸𑀚𑀧𑁄𑀭𑀺𑀲𑁂𑀦𑀧𑀺 𑀲𑀺𑀧𑁆𑀧𑀜𑁆𑀜𑀢𑀭𑁂𑀦𑀧𑀺, 𑀓𑁂𑀯𑀮𑀫𑁆𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀸𑀘𑀭𑀺𑀬𑀸𑀬, 𑀓𑀧𑀸𑀮𑀁 𑀅𑀦𑀢𑀺𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇

1117. 𑀲𑁄 𑀆𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀆𑀘𑀭𑀺𑀬𑀥𑀦𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀤𑀸𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑁂𑀦𑀧𑀺 𑀅𑀥𑀫𑁆𑀫𑁂𑀦𑀧𑀺 𑀓𑀬𑁂𑀦𑀧𑀺 𑀯𑀺𑀓𑁆𑀓𑀬𑁂𑀦𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀫𑁆𑀧𑀺 𑀉𑀤𑀓𑀽𑀧𑀲𑁆𑀲𑀝𑁆𑀞𑀁𑁇 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀯𑁂𑀲𑁆𑀲𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀲𑀼𑀤𑁆𑀤𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀘𑀡𑁆𑀟𑀸𑀮𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀦𑁂𑀲𑀸𑀤𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀯𑁂𑀦𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀭𑀣𑀓𑀸𑀭𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀕𑀩𑁆𑀪𑀺𑀦𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀸𑀬𑀫𑀸𑀦𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀉𑀢𑀼𑀦𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺 𑀅𑀦𑀼𑀢𑀼𑀦𑀺𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀸 𑀳𑁄𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀓𑀸𑀫𑀢𑁆𑀣𑀸𑀧𑀺 𑀤𑀯𑀢𑁆𑀣𑀸𑀧𑀺 𑀭𑀢𑀢𑁆𑀣𑀸𑀧𑀺 𑀧𑀚𑀢𑁆𑀣𑀸𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀚𑀻𑀯𑀺𑀓𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀓𑀲𑁆𑀫𑀸 𑀪𑀯𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀫𑀸𑀦𑁄 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀚𑀻𑀯𑀺𑀓𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑁄, 𑀅𑀕𑁆𑀕𑀺 𑀲𑀼𑀘𑀺𑀫𑁆𑀧𑀺 𑀟𑀳𑀢𑀺 𑀅𑀲𑀼𑀘𑀺𑀫𑁆𑀧𑀺 𑀟𑀳𑀢𑀺, 𑀦 𑀘 𑀢𑁂𑀦 𑀅𑀕𑁆𑀕𑀺 𑀉𑀧𑀮𑀺𑀧𑁆𑀧𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑁄, 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀘𑁂𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀚𑀻𑀯𑀺𑀓𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀦 𑀘 𑀢𑁂𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀧𑀮𑀺𑀧𑁆𑀧𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀚𑀻𑀯𑀺𑀓𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀻’𑀢𑀺 𑀔𑁄, 𑀤𑁄𑀡, 𑀢𑀲𑁆𑀫𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀘𑀡𑁆𑀟𑀸𑀮𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀘𑀡𑁆𑀟𑀸𑀮𑁄 𑀳𑁄𑀢𑀺𑁇

1118. 𑀬𑁂 𑀔𑁄 𑀢𑁂, 𑀤𑁄𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀇𑀲𑀬𑁄 𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀯𑀢𑁆𑀢𑀸𑀭𑁄 𑀬𑁂𑀲𑀫𑀺𑀤𑀁 𑀏𑀢𑀭𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀧𑁄𑀭𑀸𑀡𑀁 𑀫𑀦𑁆𑀢𑀧𑀤𑀁 𑀕𑀻𑀢𑀁 𑀧𑀯𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑀻𑀳𑀺𑀢𑀁 𑀢𑀤𑀦𑀼𑀕𑀸𑀬𑀦𑁆𑀢𑀺 𑀢𑀤𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀪𑀸𑀲𑀺𑀢𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀲𑀚𑁆𑀛𑀸𑀬𑀺𑀢𑀫𑀦𑀼𑀲𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀯𑀸𑀘𑀺𑀢𑀫𑀦𑀼𑀯𑀸𑀘𑁂𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀅𑀝𑁆𑀞𑀓𑁄, 𑀯𑀸𑀫𑀓𑁄, 𑀯𑀸𑀫𑀤𑁂𑀯𑁄, 𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑀢𑁆𑀢𑁄, 𑀬𑀫𑀤𑀕𑁆𑀕𑀺, 𑀅𑀗𑁆𑀕𑀻𑀭𑀲𑁄, 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑁄, 𑀓𑀲𑁆𑀲𑀧𑁄, 𑀪𑀕𑀼; 𑀢𑁆𑀬𑀲𑁆𑀲𑀼𑀫𑁂 𑀧𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀺—  𑀩𑁆𑀭𑀳𑁆𑀫𑀲𑀫𑀁, 𑀤𑁂𑀯𑀲𑀫𑀁, 𑀫𑀭𑀺𑀬𑀸𑀤𑀁, 𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀫𑀭𑀺𑀬𑀸𑀤𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀘𑀡𑁆𑀟𑀸𑀮𑀁𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀫𑀁𑁇 𑀢𑁂𑀲𑀁 𑀢𑁆𑀯𑀁, 𑀤𑁄𑀡, 𑀓𑀢𑀫𑁄𑀢𑀺?

1119. 𑀏𑀯𑀁 𑀲𑀦𑁆𑀢𑁂 𑀫𑀬𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀘𑀡𑁆𑀟𑀸𑀮𑀫𑁆𑀧𑀺 𑀦 𑀧𑀽𑀭𑁂𑀫𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
