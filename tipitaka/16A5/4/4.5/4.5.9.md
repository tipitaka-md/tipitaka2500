# 𑁪𑁇𑁫𑁇𑁯 𑀓𑀼𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀮𑀲𑀼𑀢𑁆𑀢

(199.)

1174\. “𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀲𑀕𑁆𑀕𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

1175\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀆𑀲𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀓𑀼𑀮𑀻𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

1176\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀦𑁆𑀢𑀺, 𑀫𑀳𑁂𑀲𑀓𑁆𑀔𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

1177\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀬𑀣𑀸𑀲𑀢𑁆𑀢𑀺 𑀬𑀣𑀸𑀩𑀮𑀁 𑀲𑀁𑀯𑀺𑀪𑀚𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸𑀪𑁄𑀕𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇

1178\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀬𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀧𑀜𑁆𑀳𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀁 𑀓𑀼𑀮𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀓𑀼𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀇𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
