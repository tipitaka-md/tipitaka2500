# 𑁪𑁇𑁫𑁇𑁮 𑀯𑀸𑀘𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀯𑀸𑀘𑀸𑀲𑀼𑀢𑁆𑀢

(198.)

1172\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀸𑀘𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸, 𑀅𑀦𑀯𑀚𑁆𑀚𑀸 𑀘 𑀅𑀦𑀦𑀼𑀯𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀸𑀮𑁂𑀦 𑀘 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀘𑁆𑀘𑀸 𑀘 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀡𑁆𑀳𑀸 𑀘 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀸 𑀘 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀘 𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀯𑀸𑀘𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸, 𑀅𑀦𑀯𑀚𑁆𑀚𑀸 𑀘 𑀅𑀦𑀦𑀼𑀯𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀦𑁆”𑀢𑀺𑁇
