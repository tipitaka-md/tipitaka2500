# 𑁪𑁇𑁫𑁇𑁧𑁦 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(200.)

1180\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀸 𑀥𑀸𑀢𑀼𑀬𑁄𑁇 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀘? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀓𑀸𑀫𑁂𑀳𑀺; 𑀬𑁂 𑀘 𑀓𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀓𑀸𑀫𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁧)

1181\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂𑀦; 𑀬𑁂 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁨)

1182\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀳𑁂𑀲𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀯𑀺𑀳𑁂𑀲𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑀸𑀬; 𑀬𑁂 𑀘 𑀯𑀺𑀳𑁂𑀲𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁩)

1183\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀽𑀧𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀭𑀽𑀧𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀭𑀽𑀧𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑁂𑀳𑀺; 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁪)

1184\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀲𑀼𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀓𑁆𑀓𑀸𑀬𑁂𑀦; 𑀬𑁂 𑀘 𑀲𑀓𑁆𑀓𑀸𑀬𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁫)

1185\. 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀦𑀦𑁆𑀤𑀻𑀧𑀺 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀦𑁆𑀤𑀻𑀧𑀺 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀯𑀺𑀳𑁂𑀲𑀸𑀦𑀦𑁆𑀤𑀻𑀧𑀺 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀭𑀽𑀧𑀦𑀦𑁆𑀤𑀻𑀧𑀺 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀦𑁆𑀤𑀻𑀧𑀺 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀧𑀺 𑀅𑀦𑀦𑀼𑀲𑀬𑀸, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀧𑀺 𑀅𑀦𑀦𑀼𑀲𑀬𑀸, 𑀯𑀺𑀳𑁂𑀲𑀸𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀧𑀺 𑀅𑀦𑀦𑀼𑀲𑀬𑀸, 𑀭𑀽𑀧𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀧𑀺 𑀅𑀦𑀦𑀼𑀲𑀬𑀸, 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀦𑁆𑀤𑀺𑀬𑀸𑀧𑀺 𑀅𑀦𑀦𑀼𑀲𑀬𑀸𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀢𑀡𑁆𑀳𑀁, 𑀯𑀺𑀯𑀢𑁆𑀢𑀬𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀁, 𑀲𑀫𑁆𑀫𑀸 𑀫𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀸 𑀅𑀦𑁆𑀢𑀫𑀓𑀸𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀻𑀬𑀸 𑀥𑀸𑀢𑀼𑀬𑁄”𑀢𑀺𑁇
