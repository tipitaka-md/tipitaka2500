# 𑁪𑁇𑁨𑁇𑁧𑁦 𑀪𑀤𑁆𑀤𑀚𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀖𑀸𑀢𑀯𑀕𑁆𑀕

### 𑀪𑀤𑁆𑀤𑀚𑀺𑀲𑀼𑀢𑁆𑀢

(170.)

989\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀚𑀺 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀆𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑁇 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀪𑀤𑁆𑀤𑀚𑀺𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀪𑀤𑁆𑀤𑀚𑀺, 𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁, 𑀓𑀺𑀁 𑀲𑀯𑀦𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁, 𑀓𑀺𑀁 𑀲𑀼𑀔𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁, 𑀓𑀺𑀁 𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁, 𑀓𑀺𑀁 𑀪𑀯𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺?

990\. “𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀅𑀪𑀺𑀪𑀽 𑀅𑀦𑀪𑀺𑀪𑀽𑀢𑁄 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼𑀤𑀲𑁄 𑀯𑀲𑀯𑀢𑁆𑀢𑀻, 𑀬𑁄 𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀇𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀆𑀪𑀲𑁆𑀲𑀭𑀸 𑀦𑀸𑀫 𑀤𑁂𑀯𑀸 𑀲𑀼𑀔𑁂𑀦 𑀅𑀪𑀺𑀲𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀲𑀦𑁆𑀦𑀸𑁇 𑀢𑁂 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆’𑀢𑀺𑁇 𑀬𑁄 𑀢𑀁 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀇𑀤𑀁 𑀲𑀯𑀦𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀲𑀼𑀪𑀓𑀺𑀡𑁆𑀳𑀸 𑀦𑀸𑀫 𑀤𑁂𑀯𑀸𑁇 𑀢𑁂 𑀲𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀢𑀼𑀲𑀺𑀢𑀸 𑀲𑀼𑀔𑀁 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀺, 𑀇𑀤𑀁 𑀲𑀼𑀔𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸 𑀤𑁂𑀯𑀸, 𑀇𑀤𑀁 𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸 𑀤𑁂𑀯𑀸, 𑀇𑀤𑀁 𑀪𑀯𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀫𑁂𑀢𑀺 𑀔𑁄 𑀇𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀪𑀤𑁆𑀤𑀚𑀺𑀲𑁆𑀲, 𑀬𑀤𑀺𑀤𑀁 𑀩𑀳𑀼𑀦𑀸 𑀚𑀦𑁂𑀦𑀸”𑀢𑀺?

991\. “𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀆𑀦𑀦𑁆𑀤𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄𑁇 𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀆𑀦𑀦𑁆𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦𑀳𑀸𑀯𑀼𑀲𑁄 𑀪𑀤𑁆𑀤𑀚𑀺, 𑀲𑀼𑀡𑀸𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀚𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

992\. “𑀬𑀣𑀸 𑀧𑀲𑁆𑀲𑀢𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀇𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀬𑀣𑀸 𑀲𑀼𑀡𑀢𑁄 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀇𑀤𑀁 𑀲𑀯𑀦𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀬𑀣𑀸 𑀲𑀼𑀔𑀺𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀇𑀤𑀁 𑀲𑀼𑀔𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀬𑀣𑀸 𑀲𑀜𑁆𑀜𑀺𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀇𑀤𑀁 𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀁𑁇 𑀬𑀣𑀸 𑀪𑀽𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀇𑀤𑀁 𑀪𑀯𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇

995\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



996\. _𑀤𑁆𑀯𑁂 𑀆𑀖𑀸𑀢𑀯𑀺𑀦𑀬𑀸,_  
_𑀲𑀸𑀓𑀘𑁆𑀙𑀸 𑀲𑀸𑀚𑀻𑀯𑀢𑁄 𑀧𑀜𑁆𑀳𑀁;_  
_𑀧𑀼𑀘𑁆𑀙𑀸 𑀦𑀺𑀭𑁄𑀥𑁄 𑀘𑁄𑀤𑀦𑀸,_  
_𑀲𑀻𑀮𑀁 𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀪𑀤𑁆𑀤𑀚𑀻𑀢𑀺𑁇_
