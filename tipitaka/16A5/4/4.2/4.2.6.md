# 𑁪𑁇𑁨𑁇𑁬 𑀦𑀺𑀭𑁄𑀥𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀖𑀸𑀢𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀭𑁄𑀥𑀲𑀼𑀢𑁆𑀢

(166.)

951\. 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺…𑀧𑁂…  “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀁 𑀆𑀭𑀸𑀥𑁂𑀬𑁆𑀬, 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

952\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀦𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

953\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀁 𑀆𑀭𑀸𑀥𑁂𑀬𑁆𑀬, 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

954\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀦𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

955\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀫𑁆𑀧𑀺 𑀔𑁄 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺, 𑀦 𑀘 𑀫𑁂 𑀓𑁄𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀫𑁄𑀤𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀁 𑀆𑀭𑀸𑀥𑁂𑀬𑁆𑀬, 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

956\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀦𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

957\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀁 𑀆𑀭𑀸𑀥𑁂𑀬𑁆𑀬, 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

958\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀦𑀯𑀓𑀸𑀲𑁄 𑀬𑀁 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀦𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

959\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑁄𑀧𑀺 𑀔𑁄 𑀫𑁂 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀤𑀸𑀬𑀻 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀁 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺, 𑀦 𑀘 𑀫𑁂 𑀓𑁄𑀘𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀫𑁄𑀤𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

960\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀤𑀸𑀬𑀺𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀧𑀘𑁆𑀘𑁂𑀲𑀻”𑀢𑀺? “𑀬𑁂 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀫𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀼𑀬𑁆𑀳𑀁, 𑀉𑀤𑀸𑀬𑀺, 𑀩𑀸𑀮𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀡𑀺𑀢𑁂𑀦𑁇 𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀪𑀡𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇

961\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀢𑁆𑀣𑀺 𑀦𑀸𑀫, 𑀆𑀦𑀦𑁆𑀤, 𑀣𑁂𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑀺𑀳𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀣𑁇 𑀦 𑀳𑀺 𑀦𑀸𑀫, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀣𑁂𑀭𑀫𑁆𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀫𑁆𑀳𑀺 𑀯𑀺𑀳𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑀫𑁆𑀳𑀻”𑀢𑀺𑁇

962\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀜𑁆𑀜𑀁 𑀆𑀭𑀸𑀥𑁂𑀬𑁆𑀬, 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫𑁂𑀯 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀫𑀦𑁄𑀫𑀬𑀁 𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀧𑀺 𑀯𑀼𑀝𑁆𑀞𑀳𑁂𑀬𑁆𑀬𑀸𑀧𑀺—  𑀅𑀢𑁆𑀣𑁂𑀢𑀁 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

963\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀯𑀸𑀡𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀯𑀸𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥𑀸𑀯𑀼𑀲𑁄 𑀉𑀧𑀯𑀸𑀡, 𑀅𑀜𑁆𑀜𑁂 𑀣𑁂𑀭𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀬𑀁 𑀢𑁂𑀦 𑀦 𑀫𑀼𑀘𑁆𑀘𑀸𑀫𑁇 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀔𑁄, 𑀧𑀦𑁂𑀢𑀁 𑀆𑀯𑀼𑀲𑁄 𑀉𑀧𑀯𑀸𑀡, 𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀏𑀢𑀤𑁂𑀯 𑀆𑀭𑀩𑁆𑀪 𑀉𑀤𑀸𑀳𑀭𑁂𑀬𑁆𑀬 𑀬𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁𑀬𑁂𑀯𑁂𑀢𑁆𑀣 𑀉𑀧𑀯𑀸𑀡𑀁 𑀧𑀝𑀺𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀇𑀤𑀸𑀦𑁂𑀯 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀸𑀭𑀚𑁆𑀚𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀯𑀸𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

964\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀉𑀧𑀯𑀸𑀡, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑀸”𑀢𑀺? “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼; 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄 𑀧𑁄𑀭𑀺𑀬𑀸 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀮𑀸𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻; 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸…𑀧𑁂…  𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑀸”𑀢𑀺𑁇

965\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀉𑀧𑀯𑀸𑀡𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀯𑀸𑀡, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀇𑀫𑁂 𑀘𑁂, 𑀉𑀧𑀯𑀸𑀡, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀣𑁂𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀦 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀦 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀦 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁 𑀦 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀔𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑁂𑀦 𑀧𑀸𑀮𑀺𑀘𑁆𑀘𑁂𑀦 𑀯𑀮𑀺𑀢𑁆𑀢𑀘𑀢𑀸𑀬𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀉𑀧𑀯𑀸𑀡, 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀣𑁂𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
