# 𑁪𑁇𑁪𑁇𑁨 𑀘𑀻𑀯𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀘𑀻𑀯𑀭𑀲𑀼𑀢𑁆𑀢

(182.)

1076\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀫𑀦𑁆𑀤𑀢𑁆𑀢𑀸 𑀫𑁄𑀫𑀽𑀳𑀢𑁆𑀢𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀫𑀢𑁆𑀣𑀺𑀢𑀁𑀬𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸”𑀢𑀺𑁇
