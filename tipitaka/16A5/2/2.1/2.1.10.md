394\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀤𑀼𑀮𑁆𑀮𑀪𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀼𑀯𑀘𑁄, 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀕𑁆𑀕𑀸𑀳𑀻, 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻, 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁄, 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺𑁇
