# 𑁨𑁇𑁧𑁇𑁨 𑀅𑀓𑀼𑀲𑀮𑀭𑀸𑀲𑀺𑀲𑀼𑀢𑁆𑀢

(52.)

326\. “𑀅𑀓𑀼𑀲𑀮𑀭𑀸𑀲𑀻𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀤𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀓𑁂𑀯𑀮𑁄 𑀳𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀲𑀮𑀭𑀸𑀲𑀺 𑀬𑀤𑀺𑀤𑀁 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀭𑀸𑀲𑀻𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀤𑀫𑀸𑀦𑁄 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑁂 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑁄 𑀯𑀤𑁂𑀬𑁆𑀬𑁇 𑀓𑁂𑀯𑀮𑁄 𑀳𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑀼𑀲𑀮𑀭𑀸𑀲𑀺 𑀬𑀤𑀺𑀤𑀁 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸”𑀢𑀺𑁇
