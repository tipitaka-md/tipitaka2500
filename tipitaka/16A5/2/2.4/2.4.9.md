# 𑁨𑁇𑁪𑁇𑁯 𑀧𑀞𑀫𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀣𑁂𑀭𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

(89.)

556\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀢𑀸, 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑀢𑀸, 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑀢𑀸, 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀢𑀸, 𑀬𑀣𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇

557\. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀢𑀸, 𑀦 𑀪𑀲𑁆𑀲𑀸𑀭𑀸𑀫𑀢𑀸, 𑀦 𑀦𑀺𑀤𑁆𑀤𑀸𑀭𑀸𑀫𑀢𑀸, 𑀦 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀢𑀸, 𑀬𑀣𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑁂𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
