# 𑁨𑁇𑁪𑁇𑁧 𑀭𑀚𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀣𑁂𑀭𑀯𑀕𑁆𑀕

* 𑀭𑀚𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(81.)

532\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀭𑀚𑀦𑀻𑀬𑁂 𑀭𑀚𑁆𑀚𑀢𑀺, 𑀤𑀼𑀲𑁆𑀲𑀦𑀻𑀬𑁂 𑀤𑀼𑀲𑁆𑀲𑀢𑀺, 𑀫𑁄𑀳𑀦𑀻𑀬𑁂 𑀫𑀼𑀬𑁆𑀳𑀢𑀺, 𑀓𑀼𑀧𑁆𑀧𑀦𑀻𑀬𑁂 𑀓𑀼𑀧𑁆𑀧𑀢𑀺, 𑀫𑀤𑀦𑀻𑀬𑁂 𑀫𑀚𑁆𑀚𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇

533\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀭𑀚𑀦𑀻𑀬𑁂 𑀦 𑀭𑀚𑁆𑀚𑀢𑀺, 𑀤𑀼𑀲𑁆𑀲𑀦𑀻𑀬𑁂 𑀦 𑀤𑀼𑀲𑁆𑀲𑀢𑀺, 𑀫𑁄𑀳𑀦𑀻𑀬𑁂 𑀦 𑀫𑀼𑀬𑁆𑀳𑀢𑀺, 𑀓𑀼𑀧𑁆𑀧𑀦𑀻𑀬𑁂 𑀦 𑀓𑀼𑀧𑁆𑀧𑀢𑀺, 𑀫𑀤𑀦𑀻𑀬𑁂 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑀸”𑀢𑀺𑁇
