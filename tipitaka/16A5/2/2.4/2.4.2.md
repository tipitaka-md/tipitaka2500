# 𑁨𑁇𑁪𑁇𑁨 𑀯𑀻𑀢𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀣𑁂𑀭𑀯𑀕𑁆𑀕

* 𑀯𑀻𑀢𑀭𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(82.)

535\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀻𑀢𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀻𑀢𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀓𑁆𑀔𑀻 𑀘, 𑀧𑀍𑀆𑀲𑀻 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇

536\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀢𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀻𑀢𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀫𑀓𑁆𑀔𑀻 𑀘, 𑀅𑀧𑀍𑀆𑀲𑀻 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑀸”𑀢𑀺𑁇
