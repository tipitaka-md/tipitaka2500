# 𑁨𑁇𑁪𑁇𑁩 𑀓𑀼𑀳𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀣𑁂𑀭𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀳𑀓𑀲𑀼𑀢𑁆𑀢

(83.)

538\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀼𑀳𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀮𑀧𑀓𑁄 𑀘, 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀺𑀓𑁄 𑀘, 𑀦𑀺𑀧𑁆𑀧𑁂𑀲𑀺𑀓𑁄 𑀘, 𑀮𑀸𑀪𑁂𑀦 𑀘 𑀮𑀸𑀪𑀁 𑀦𑀺𑀚𑀺𑀕𑀻𑀲𑀺𑀢𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇

539\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀘 𑀓𑀼𑀳𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀮𑀧𑀓𑁄, 𑀦 𑀘 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀺𑀓𑁄, 𑀦 𑀘 𑀦𑀺𑀧𑁆𑀧𑁂𑀲𑀺𑀓𑁄, 𑀦 𑀘 𑀮𑀸𑀪𑁂𑀦 𑀮𑀸𑀪𑀁 𑀦𑀺𑀚𑀺𑀕𑀻𑀲𑀺𑀢𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑀸”𑀢𑀺𑁇
