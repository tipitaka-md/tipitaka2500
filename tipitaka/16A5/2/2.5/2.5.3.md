# 𑁨𑁇𑁫𑁇𑁩 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀓𑀓𑀼𑀥𑀯𑀕𑁆𑀕

* 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(93.)

577\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀸𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀫𑀦𑁆𑀤𑀢𑁆𑀢𑀸 𑀫𑁄𑀫𑀽𑀳𑀢𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺; 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺; 𑀉𑀫𑁆𑀫𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑁂𑀧𑀸 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺; 𑀅𑀥𑀺𑀫𑀸𑀦𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺; 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑀸𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀦𑀻”𑀢𑀺𑁇
