# 𑁨𑁇𑁨𑁇𑁧𑁦 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀜𑁆𑀜𑀸𑀯𑀕𑁆𑀕

### 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

(70.)

423\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑀼𑀪𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀓𑀸𑀬𑁂 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀆𑀳𑀸𑀭𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀻, 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂 𑀅𑀦𑀪𑀺𑀭𑀢𑀲𑀜𑁆𑀜𑀻, 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻, 𑀫𑀭𑀡𑀲𑀜𑁆𑀜𑀸 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
