# 𑁧𑁇𑁪𑁇𑁬 𑀓𑀸𑀮𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀼𑀫𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀮𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(36.)

187\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀮𑀤𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀲𑁆𑀲 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺; 𑀕𑀫𑀺𑀓𑀲𑁆𑀲 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺; 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺; 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺; 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀦𑀯𑀲𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀯𑀨𑀮𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀞𑀫𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀲𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀓𑀸𑀮𑀤𑀸𑀦𑀸𑀦𑀻𑀢𑀺𑁇

188\. _𑀓𑀸𑀮𑁂 𑀤𑀤𑀦𑁆𑀢𑀺 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀤𑀜𑁆𑀜𑀽 𑀯𑀻𑀢𑀫𑀘𑁆𑀙𑀭𑀸;_  
_𑀓𑀸𑀮𑁂𑀦 𑀤𑀺𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑁂𑀲𑀼,_  
_𑀉𑀚𑀼𑀪𑀽𑀢𑁂𑀲𑀼 𑀢𑀸𑀤𑀺𑀲𑀼𑁇_  


189\. _𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀫𑀦𑀸 𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀧𑀼𑀮𑀸 𑀳𑁄𑀢𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀸;_  
_𑀬𑁂 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀯𑀸;_  
_𑀦 𑀢𑁂𑀦 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀊𑀦𑀸,_  
_𑀢𑁂𑀧𑀺 𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑀸𑀕𑀺𑀦𑁄𑁇_  


190\. _𑀢𑀲𑁆𑀫𑀸 𑀤𑀤𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀬𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀧𑀭𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

191\. 𑀙𑀝𑁆𑀞𑀁𑁇
