# 𑁧𑁇𑁪𑁇𑁧𑁦 𑀫𑀳𑀸𑀲𑀸𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀼𑀫𑀦𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀲𑀸𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(40.)

211\. “𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀟𑁆𑀠𑀻𑀳𑀺 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀸𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀲𑀸𑀔𑀸𑀧𑀢𑁆𑀢𑀧𑀮𑀸𑀲𑁂𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺; 𑀢𑀘𑁂𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺; 𑀧𑀧𑀝𑀺𑀓𑀸𑀬 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺; 𑀨𑁂𑀕𑁆𑀕𑀼𑀦𑀸 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺; 𑀲𑀸𑀭𑁂𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺𑁇 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀩𑁆𑀩𑀢𑀭𑀸𑀚𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀇𑀫𑀸𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀟𑁆𑀠𑀻𑀳𑀺 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀥𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑁆𑀢𑁄𑀚𑀦𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀟𑁆𑀠𑀻𑀳𑀺 𑀯𑀟𑁆𑀠𑀢𑀺𑁇 𑀓𑀢𑀫𑀸𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀲𑀤𑁆𑀥𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺; 𑀲𑀻𑀮𑁂𑀦 𑀯𑀟𑁆𑀠𑀢𑀺; 𑀲𑀼𑀢𑁂𑀦 𑀯𑀟𑁆𑀠𑀢𑀺; 𑀘𑀸𑀕𑁂𑀦 𑀯𑀟𑁆𑀠𑀢𑀺; 𑀧𑀜𑁆𑀜𑀸𑀬 𑀯𑀟𑁆𑀠𑀢𑀺𑁇 𑀲𑀤𑁆𑀥𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑁆𑀢𑁄𑀚𑀦𑁄 𑀇𑀫𑀸𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀯𑀟𑁆𑀠𑀻𑀳𑀺 𑀯𑀟𑁆𑀠𑀢𑀻𑀢𑀺𑁇

212\. _𑀬𑀣𑀸 𑀳𑀺 𑀧𑀩𑁆𑀩𑀢𑁄 𑀲𑁂𑀮𑁄,_  
_𑀅𑀭𑀜𑁆𑀜𑀲𑁆𑀫𑀺𑀁 𑀩𑁆𑀭𑀳𑀸𑀯𑀦𑁂;_  
_𑀢𑀁 𑀭𑀼𑀓𑁆𑀔𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑁂 𑀢𑁂 𑀯𑀦𑀧𑁆𑀧𑀢𑀻𑁇_  


213\. _𑀢𑀣𑁂𑀯 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀤𑁆𑀥𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀇𑀫𑀁;_  
_𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀸 𑀘 𑀩𑀦𑁆𑀥𑀯𑀸;_  
_𑀅𑀫𑀘𑁆𑀘𑀸 𑀜𑀸𑀢𑀺𑀲𑀗𑁆𑀖𑀸 𑀘,_  
_𑀬𑁂 𑀘𑀲𑁆𑀲 𑀅𑀦𑀼𑀚𑀻𑀯𑀺𑀦𑁄𑁇_  


214\. _𑀢𑁆𑀬𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀁,_  
_𑀘𑀸𑀕𑀁 𑀲𑀼𑀘𑀭𑀺𑀢𑀸𑀦𑀺 𑀘;_  
_𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸𑀦𑀼𑀓𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀪𑀯𑀦𑁆𑀢𑀺 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑀸𑁇_  


215\. _𑀇𑀥 𑀥𑀫𑁆𑀫𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀕𑁆𑀕𑀁 𑀲𑀼𑀕𑀢𑀺𑀕𑀸𑀫𑀺𑀦𑀁;_  
_𑀦𑀦𑁆𑀤𑀺𑀦𑁄 𑀤𑁂𑀯𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀓𑀸𑀫𑀺𑀦𑁄”𑀢𑀺𑁇_

217\. 𑀲𑀼𑀫𑀦𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇

218\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



219\. _𑀲𑀼𑀫𑀦𑀸 𑀘𑀼𑀦𑁆𑀤𑀻 𑀉𑀕𑁆𑀕𑀳𑁄,_  
_𑀲𑀻𑀳𑁄 𑀤𑀸𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀓𑁄;_  
_𑀓𑀸𑀮𑀪𑁄𑀚𑀦𑀲𑀤𑁆𑀥𑀸 𑀘,_  
_𑀧𑀼𑀢𑁆𑀢𑀲𑀸𑀮𑁂𑀳𑀺 𑀢𑁂 𑀤𑀲𑀸𑀢𑀺𑁇_
