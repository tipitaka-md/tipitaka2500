# 𑁬𑁇𑁧𑁇𑁧𑁪 𑀅𑀧𑀭𑀧𑀞𑀫𑀛𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀙𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀧𑀭𑀧𑀞𑀫𑀛𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(264.)

1367\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀁 𑀅𑀓𑀢𑀯𑁂𑀤𑀺𑀢𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁𑁇

1368\. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀁 𑀅𑀓𑀢𑀯𑁂𑀤𑀺𑀢𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
