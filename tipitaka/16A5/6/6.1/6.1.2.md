# 𑁬𑁇𑁧𑁇𑁨 𑀦𑀺𑀲𑁆𑀲𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀺𑀲𑁆𑀲𑀬𑀲𑀼𑀢𑁆𑀢

(252.)

1353\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁂𑀔𑁂𑀦 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀲𑁂𑀔𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
