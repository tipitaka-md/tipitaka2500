# 𑁬𑁇𑁧𑁇𑁭--𑁧𑁩 𑀤𑀼𑀢𑀺𑀬𑀛𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀲𑀢𑁆𑀢𑀓

### 𑀙𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀛𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺𑀲𑀢𑁆𑀢𑀓

(257--263.)

1364\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁…  𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀅𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇

1365\. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀪𑀩𑁆𑀩𑁄 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…  𑀪𑀩𑁆𑀩𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁…  𑀪𑀩𑁆𑀩𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁…  𑀪𑀩𑁆𑀩𑁄 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁…  𑀪𑀩𑁆𑀩𑁄 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁…  𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀪𑀩𑁆𑀩𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
