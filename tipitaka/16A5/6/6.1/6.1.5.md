# 𑁬𑁇𑁧𑁇𑁫 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀧𑁆𑀧𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀙𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀯𑀕𑁆𑀕

### 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀧𑁆𑀧𑀳𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(255.)

1359\. “𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁? 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀺; 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑁆𑀲…  𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑁆𑀲…  𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑁆𑀲…  𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀯𑀼𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇
