# 𑁬𑁇𑁨𑁇𑁨--𑁧𑁪 𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺

### 𑀙𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀧𑁂𑀬𑁆𑀬𑀸𑀮

### 𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑀲𑀼𑀢𑁆𑀢𑀸𑀤𑀺

(273--285.)

1378\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂… 𑁇

1379\. 𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀸𑀳𑀸𑀧𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀸𑀳𑀸𑀧𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂… 𑁇

1380\. 𑀪𑀡𑁆𑀟𑀸𑀕𑀸𑀭𑀺𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀼𑀢𑁆𑀢𑀸𑀕𑀼𑀢𑁆𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…  𑀪𑀡𑁆𑀟𑀸𑀕𑀸𑀭𑀺𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀼𑀢𑁆𑀢𑀸𑀕𑀼𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺… 𑁇

1381\. 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…  𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄 …𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺… 𑁇

1382\. 𑀘𑀻𑀯𑀭𑀪𑀸𑀚𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀪𑀸𑀚𑀺𑀢𑀸𑀪𑀸𑀚𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…  𑀘𑀻𑀯𑀭𑀪𑀸𑀚𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀪𑀸𑀚𑀺𑀢𑀸𑀪𑀸𑀚𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺… 𑁇

1383\. 𑀬𑀸𑀕𑀼𑀪𑀸𑀚𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀬𑀸𑀕𑀼𑀪𑀸𑀚𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇

1384\. 𑀨𑀮𑀪𑀸𑀚𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀨𑀮𑀪𑀸𑀚𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇

1385\. 𑀔𑀚𑁆𑀚𑀓𑀪𑀸𑀚𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀪𑀸𑀚𑀺𑀢𑀸𑀪𑀸𑀚𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…  𑀔𑀚𑁆𑀚𑀓𑀪𑀸𑀚𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀪𑀸𑀚𑀺𑀢𑀸𑀪𑀸𑀚𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺… 𑁇

1386\. 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀸𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…  𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀸𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺… 𑁇

1387\. 𑀲𑀸𑀝𑀺𑀬𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…  𑀲𑀸𑀝𑀺𑀬𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺… 𑁇

1388\. 𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺…  𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺… 𑁇

1389\. 𑀆𑀭𑀸𑀫𑀺𑀓𑀧𑁂𑀲𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀆𑀭𑀸𑀫𑀺𑀓𑀧𑁂𑀲𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇

1390\. 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇

1391\. 𑀲𑀫𑁆𑀫𑀢𑁄 𑀦 𑀧𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀢𑁄 𑀧𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂… 𑁇

1392\. 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀩𑀸𑀮𑁄 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀯𑁂𑀤𑀺𑀢𑀩𑁆𑀩𑁄…  𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺…  𑀅𑀓𑁆𑀔𑀢𑀁 𑀅𑀦𑀼𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺…  𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂…  𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀧𑁂𑀲𑀺𑀢𑀸𑀧𑁂𑀲𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇
