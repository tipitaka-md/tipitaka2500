# 𑁬𑁇𑁩𑁇𑁧 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀙𑀝𑁆𑀞𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀧𑁂𑀬𑁆𑀬𑀸𑀮

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼𑀢𑁆𑀢

(286.)

1395\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀬𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇

1396\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇
