# 𑁫𑁇𑁩𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(222.)

1261\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀁 𑀅𑀦𑀯𑀢𑁆𑀣𑀘𑀸𑀭𑀺𑀓𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀦𑀥𑀺𑀕𑀢𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀥𑀺𑀕𑀢𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺, 𑀅𑀥𑀺𑀕𑀢𑁂𑀦𑁂𑀓𑀘𑁆𑀘𑁂𑀦 𑀅𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀕𑀸𑀍𑀳𑀁 𑀭𑁄𑀕𑀸𑀢𑀗𑁆𑀓𑀁 𑀨𑀼𑀲𑀢𑀺, 𑀦 𑀘 𑀫𑀺𑀢𑁆𑀢𑀯𑀸 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀁 𑀅𑀦𑀯𑀢𑁆𑀣𑀘𑀸𑀭𑀺𑀓𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄𑁇

1262\. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀫𑀯𑀢𑁆𑀣𑀘𑀸𑀭𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀦𑀥𑀺𑀕𑀢𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀥𑀺𑀕𑀢𑀸 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺, 𑀅𑀥𑀺𑀕𑀢𑁂𑀦𑁂𑀓𑀘𑁆𑀘𑁂𑀦 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀕𑀸𑀍𑀳𑀁 𑀭𑁄𑀕𑀸𑀢𑀗𑁆𑀓𑀁 𑀨𑀼𑀲𑀢𑀺, 𑀫𑀺𑀢𑁆𑀢𑀯𑀸 𑀘 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀫𑀯𑀢𑁆𑀣𑀘𑀸𑀭𑁂”𑀢𑀺𑁇
