# 𑁫𑁇𑁩𑁇𑁫 𑀧𑀞𑀫𑀓𑀼𑀮𑀽𑀧𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀓𑀼𑀮𑀽𑀧𑀓𑀲𑀼𑀢𑁆𑀢

(225.)

1270\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀓𑀼𑀮𑀽𑀧𑀓𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀦𑀸𑀫𑀦𑁆𑀢𑀘𑀸𑀭𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀺 𑀙𑀧𑁆𑀧𑀜𑁆𑀘𑀯𑀸𑀘𑀸𑀳𑀺 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀓𑀸𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀩𑀳𑀼𑀮𑁄 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀓𑀼𑀮𑀽𑀧𑀓𑁂”𑀢𑀺𑁇
