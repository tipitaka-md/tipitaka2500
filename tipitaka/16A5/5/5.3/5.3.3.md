# 𑁫𑁇𑁩𑁇𑁩 𑀅𑀢𑀺𑀦𑀺𑀯𑀸𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀅𑀢𑀺𑀦𑀺𑀯𑀸𑀲𑀲𑀼𑀢𑁆𑀢

(223.)

1264. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀢𑀺𑀦𑀺𑀯𑀸𑀲𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀩𑀳𑀼𑀪𑀡𑁆𑀟𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀪𑀡𑁆𑀟𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄, 𑀩𑀳𑀼𑀪𑁂𑀲𑀚𑁆𑀚𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼, 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀳𑀝𑁆𑀞𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀦 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀦, 𑀢𑀫𑁆𑀳𑀸 𑀘 𑀆𑀯𑀸𑀲𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁄 𑀲𑀸𑀧𑁂𑀓𑁆𑀔𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀢𑀺𑀦𑀺𑀯𑀸𑀲𑁂𑁇

1265. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀫𑀯𑀢𑁆𑀣𑀯𑀸𑀲𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀦 𑀩𑀳𑀼𑀪𑀡𑁆𑀟𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀩𑀳𑀼𑀪𑀡𑁆𑀟𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄, 𑀦 𑀩𑀳𑀼𑀪𑁂𑀲𑀚𑁆𑀚𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀩𑀳𑀼𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄, 𑀦 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄 𑀦 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼, 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀳𑀝𑁆𑀞𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀦 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀦, 𑀢𑀫𑁆𑀳𑀸 𑀘 𑀆𑀯𑀸𑀲𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀫𑀯𑀢𑁆𑀣𑀯𑀸𑀲𑁂”𑀢𑀺𑁇
