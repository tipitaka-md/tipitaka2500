# 𑁫𑁇𑁩𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀻𑀖𑀘𑀸𑀭𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑀲𑀼𑀢𑁆𑀢

(230.)

1283\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀓𑁄𑀥𑀦𑁄, 𑀉𑀧𑀦𑀸𑀳𑀻, 𑀖𑁄𑀭𑀯𑀺𑀲𑁄, 𑀤𑀼𑀚𑁆𑀚𑀺𑀯𑁆𑀳𑁄, 𑀫𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀪𑀻𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑁂𑁇

1284\. 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀺𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀸 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀓𑁄𑀥𑀦𑁄, 𑀉𑀧𑀦𑀸𑀳𑀻, 𑀖𑁄𑀭𑀯𑀺𑀲𑁄, 𑀤𑀼𑀚𑁆𑀚𑀺𑀯𑁆𑀳𑁄, 𑀫𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀪𑀻𑁇 𑀢𑀢𑁆𑀭𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀖𑁄𑀭𑀯𑀺𑀲𑀢𑀸—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀢𑀺𑀩𑁆𑀩𑀭𑀸𑀕𑁄𑁇 𑀢𑀢𑁆𑀭𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀤𑀼𑀚𑁆𑀚𑀺𑀯𑁆𑀳𑀢𑀸—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀺𑀲𑀼𑀡𑀯𑀸𑀘𑁄𑁇 𑀢𑀢𑁆𑀭𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀫𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀪𑀺𑀢𑀸—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀅𑀢𑀺𑀘𑀸𑀭𑀺𑀦𑀻𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂”𑀢𑀺𑁇
