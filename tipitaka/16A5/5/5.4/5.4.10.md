# 𑁫𑁇𑁪𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀯𑀸𑀲𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(240.)

1312\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇

1313\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀦 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀦 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀦 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺; 𑀦 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇
