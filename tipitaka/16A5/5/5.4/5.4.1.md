# 𑁫𑁇𑁪𑁇𑁧 𑀆𑀯𑀸𑀲𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀆𑀯𑀸𑀲𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀆𑀯𑀸𑀲𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(231.)

1289\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀆𑀓𑀧𑁆𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄; 𑀦 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀲𑀼𑀢𑀥𑀭𑁄; 𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑁂𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀭𑀸𑀫𑁄; 𑀦 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄; 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺 𑀚𑀍𑀑 𑀏𑀍𑀅𑀫𑀽𑀕𑁄𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀳𑁄𑀢𑀺𑁇

1290\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀓𑀧𑁆𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄; 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀢𑀥𑀭𑁄; 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑁂𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀭𑀸𑀫𑁄; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄; 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀚𑀍𑀑 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑁄𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
