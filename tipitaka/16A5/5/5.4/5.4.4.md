# 𑁫𑁇𑁪𑁇𑁪 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀯𑀸𑀲𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(234.)

1296\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀯𑀸𑀲𑀲𑁆𑀲 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼; 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸; 𑀔𑀡𑁆𑀟𑀨𑀼𑀮𑁆𑀮𑀁 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀭𑁄𑀢𑀺; 𑀫𑀳𑀸 𑀔𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀦𑀸𑀦𑀸𑀯𑁂𑀭𑀚𑁆𑀚𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀳𑀻𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺—  ‘𑀫𑀳𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀦𑀸𑀦𑀸𑀯𑁂𑀭𑀚𑁆𑀚𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀓𑀭𑁄𑀣 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺, 𑀲𑀫𑀬𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀸𑀢𑀼𑀦𑁆’𑀢𑀺; 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀯𑀸𑀲𑀲𑁆𑀲 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
