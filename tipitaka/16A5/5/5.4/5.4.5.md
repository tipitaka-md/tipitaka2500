# 𑁫𑁇𑁪𑁇𑁫 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀆𑀯𑀸𑀲𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀲𑀼𑀢𑁆𑀢

(235.)

1298\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀥𑀫𑁆𑀫𑀤𑀲𑁆𑀲𑀦𑁂 𑀦𑀺𑀯𑁂𑀲𑁂𑀢𑀺; 𑀕𑀺𑀮𑀸𑀦𑀓𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺—  ‘𑀅𑀭𑀳𑀕𑁆𑀕𑀢𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀣𑀸’𑀢𑀺; 𑀫𑀳𑀸 𑀔𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀦𑀸𑀦𑀸𑀯𑁂𑀭𑀚𑁆𑀚𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀳𑀻𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺—  ‘𑀫𑀳𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀦𑀸𑀦𑀸𑀯𑁂𑀭𑀚𑁆𑀚𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀓𑀭𑁄𑀣 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺, 𑀲𑀫𑀬𑁄 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀸𑀢𑀼𑀦𑁆’𑀢𑀺; 𑀬𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀪𑁄𑀚𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑀺 𑀮𑀽𑀔𑀁 𑀯𑀸 𑀧𑀡𑀻𑀢𑀁 𑀯𑀸 𑀢𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀲𑀤𑁆𑀥𑀸𑀤𑁂𑀬𑁆𑀬𑀁 𑀦 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀢𑀻”𑀢𑀺𑁇
