# 𑁫𑁇𑁪𑁇𑁭 𑀤𑀼𑀢𑀺𑀬𑀅𑀯𑀡𑁆𑀡𑀸𑀭𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀯𑀸𑀲𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀅𑀯𑀡𑁆𑀡𑀸𑀭𑀳𑀲𑀼𑀢𑁆𑀢

(237.)

1303\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀦𑀦𑀼𑀯𑀺𑀘𑁆𑀘 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀡𑁆𑀡𑀸𑀭𑀳𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺; 𑀅𑀦𑀦𑀼𑀯𑀺𑀘𑁆𑀘 𑀅𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀡𑁆𑀡𑀸𑀭𑀳𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺; 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺 𑀆𑀯𑀸𑀲𑀧𑀮𑀺𑀕𑁂𑀥𑀻; 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺 𑀓𑀼𑀮𑀧𑀮𑀺𑀕𑁂𑀥𑀻; 𑀲𑀤𑁆𑀥𑀸𑀤𑁂𑀬𑁆𑀬𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇

1304\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀦𑀼𑀯𑀺𑀘𑁆𑀘 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀡𑁆𑀡𑀸𑀭𑀳𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺; 𑀅𑀦𑀼𑀯𑀺𑀘𑁆𑀘 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀯𑀡𑁆𑀡𑀸𑀭𑀳𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺; 𑀦 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺 𑀦 𑀆𑀯𑀸𑀲𑀧𑀮𑀺𑀕𑁂𑀥𑀻; 𑀦 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀻 𑀳𑁄𑀢𑀺 𑀦 𑀓𑀼𑀮𑀧𑀮𑀺𑀕𑁂𑀥𑀻; 𑀲𑀤𑁆𑀥𑀸𑀤𑁂𑀬𑁆𑀬𑀁 𑀦 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇
