# 𑁫𑁇𑁧𑁇𑁭 𑀬𑀸𑀕𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀓𑀺𑀫𑀺𑀮𑀯𑀕𑁆𑀕

### 𑀬𑀸𑀕𑀼𑀲𑀼𑀢𑁆𑀢

(207.)

1208\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀬𑀸𑀕𑀼𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀔𑀼𑀤𑁆𑀤𑀁 𑀧𑀝𑀺𑀳𑀦𑀢𑀺, 𑀧𑀺𑀧𑀸𑀲𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁂𑀢𑀺, 𑀯𑀸𑀢𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑀺, 𑀯𑀢𑁆𑀣𑀺𑀁 𑀲𑁄𑀥𑁂𑀢𑀺, 𑀆𑀫𑀸𑀯𑀲𑁂𑀲𑀁 𑀧𑀸𑀘𑁂𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀬𑀸𑀕𑀼𑀬𑀸”𑀢𑀺𑁇
