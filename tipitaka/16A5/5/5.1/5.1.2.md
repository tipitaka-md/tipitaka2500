# 𑁫𑁇𑁧𑁇𑁨 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀓𑀺𑀫𑀺𑀮𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀲𑀼𑀢𑁆𑀢

(202.)

1194\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀲𑁆𑀲𑀼𑀢𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀲𑀼𑀢𑀁 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀧𑁂𑀢𑀺, 𑀓𑀗𑁆𑀔𑀁 𑀯𑀺𑀢𑀭𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀉𑀚𑀼𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑁂”𑀢𑀺𑁇
