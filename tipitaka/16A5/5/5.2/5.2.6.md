# 𑁫𑁇𑁨𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀅𑀓𑁆𑀓𑁄𑀲𑀓𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(216.)

1242\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄, 𑀮𑀼𑀤𑁆𑀤𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀘, 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸𑁇

1243\. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀔𑀦𑁆𑀢𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄, 𑀅𑀮𑀼𑀤𑁆𑀤𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀘, 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀔𑀦𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇
