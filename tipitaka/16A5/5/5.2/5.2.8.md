# 𑁫𑁇𑁨𑁇𑁮 𑀤𑀼𑀢𑀺𑀬𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀓𑁆𑀓𑁄𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(218.)

1248. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀦𑁆𑀢𑀺, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁 𑀅𑀓𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂𑁇

1249. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀧𑀲𑀻𑀤𑀦𑁆𑀢𑀺, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀜𑁆𑀘 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂”𑀢𑀺𑁇
