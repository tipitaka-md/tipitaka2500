# 𑁫𑁇𑁨𑁇𑁨 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀓𑁆𑀓𑁄𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀲𑀼𑀢𑁆𑀢

(212.)

1223\. “𑀬𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄, 𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀦𑀥𑀺𑀕𑀢𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀥𑀺𑀕𑀢𑀸 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺, 𑀧𑀸𑀧𑀓𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄, 𑀢𑀲𑁆𑀲 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸”𑀢𑀺𑁇
