# 𑁫𑁇𑁫𑁇𑁯 𑀲𑀺𑀯𑀣𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀯𑀕𑁆𑀕

### 𑀲𑀺𑀯𑀣𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(249.)

1336\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀺𑀯𑀣𑀺𑀓𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀲𑀼𑀘𑀺, 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀸, 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀸, 𑀯𑀸𑀍𑀆𑀦𑀁 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀆𑀯𑀸𑀲𑁄, 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀆𑀭𑁄𑀤𑀦𑀸—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀺𑀯𑀣𑀺𑀓𑀸𑀬𑁇

1337\. 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀺𑀫𑁂 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀺𑀯𑀣𑀺𑀓𑀽𑀧𑀫𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺; 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺; 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺𑀢𑀸𑀬 𑀯𑀤𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀯𑀣𑀺𑀓𑀸 𑀅𑀲𑀼𑀘𑀺; 𑀢𑀣𑀽𑀧𑀫𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

1338\. 𑀢𑀲𑁆𑀲 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀢𑀸𑀬 𑀯𑀤𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀯𑀣𑀺𑀓𑀸 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀸; 𑀢𑀣𑀽𑀧𑀫𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

1339\. 𑀢𑀫𑁂𑀦𑀁 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀧𑁂𑀲𑀮𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀆𑀭𑀓𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀯𑀣𑀺𑀓𑀸 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑀸; 𑀢𑀣𑀽𑀧𑀫𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

1340\. 𑀲𑁄 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀪𑀸𑀕𑁂𑀳𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀲𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀯𑀸𑀍𑀆𑀯𑀸𑀲𑀲𑁆𑀫𑀺𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀯𑀣𑀺𑀓𑀸 𑀯𑀸𑀍𑀆𑀦𑀁 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀆𑀯𑀸𑀲𑁄; 𑀢𑀣𑀽𑀧𑀫𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀯𑀤𑀸𑀫𑀺𑁇

1341\. 𑀢𑀫𑁂𑀦𑀁 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁, 𑀅𑀲𑀼𑀘𑀺𑀦𑀸 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀧𑁂𑀲𑀮𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀤𑀺𑀲𑁆𑀯𑀸 𑀔𑀻𑀬𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀬𑁂 𑀫𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑁂𑀳𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀇𑀤𑀫𑀲𑁆𑀲 𑀆𑀭𑁄𑀤𑀦𑀸𑀬 𑀯𑀤𑀸𑀫𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀯𑀣𑀺𑀓𑀸 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀆𑀭𑁄𑀤𑀦𑀸; 𑀢𑀣𑀽𑀧𑀫𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀲𑀺𑀯𑀣𑀺𑀓𑀽𑀧𑀫𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁂”𑀢𑀺𑁇
