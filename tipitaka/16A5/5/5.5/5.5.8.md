# 𑁫𑁇𑁫𑁇𑁮 𑀤𑀼𑀢𑀺𑀬𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(248.)

1334\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂…𑀧𑁂…  𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀢𑁆𑀢𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀉𑀧𑀯𑀤𑀢𑀺; 𑀅𑀦𑀼𑀯𑀺𑀘𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺; 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺; 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑁂”𑀢𑀺𑁇
