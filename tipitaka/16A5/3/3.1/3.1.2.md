# 𑁩𑁇𑁧𑁇𑁨 𑀉𑀲𑁆𑀲𑀗𑁆𑀓𑀺𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀉𑀲𑁆𑀲𑀗𑁆𑀓𑀺𑀢𑀲𑀼𑀢𑁆𑀢

(102.)

615\. “𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀲𑁆𑀲𑀗𑁆𑀓𑀺𑀢𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀧𑀺 𑀅𑀓𑀼𑀧𑁆𑀧𑀥𑀫𑁆𑀫𑁄𑀧𑀺𑁇

616\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑁂𑀲𑀺𑀬𑀸𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀥𑀯𑀸𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀣𑀼𑀮𑁆𑀮𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀡𑁆𑀟𑀓𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺𑁇

617\. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀲𑁆𑀲𑀗𑁆𑀓𑀺𑀢𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀧𑀺 𑀅𑀓𑀼𑀧𑁆𑀧𑀥𑀫𑁆𑀫𑁄𑀧𑀻”𑀢𑀺𑁇

---

618\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
