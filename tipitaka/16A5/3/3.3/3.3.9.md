# 𑁩𑁇𑁩𑁇𑁯 𑀧𑀭𑀺𑀓𑀼𑀧𑁆𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀕𑀺𑀮𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀧𑀭𑀺𑀓𑀼𑀧𑁆𑀧𑀲𑀼𑀢𑁆𑀢

(129.)

725. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸 𑀧𑀭𑀺𑀓𑀼𑀧𑁆𑀧𑀸 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀫𑀸𑀢𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀺𑀢𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀭𑀳𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀤𑀼𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀮𑁄𑀳𑀺𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀁𑀖𑁄 𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸 𑀧𑀭𑀺𑀓𑀼𑀧𑁆𑀧𑀸 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑀸”𑀢𑀺𑁇
