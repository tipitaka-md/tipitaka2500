# 𑁩𑁇𑁫𑁇𑁭 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀢𑀺𑀓𑀡𑁆𑀟𑀓𑀻𑀯𑀕𑁆𑀕

* 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(147.)

854\. “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑁂𑀢𑀺, 𑀅𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑁆𑀯𑀸 𑀤𑁂𑀢𑀺, 𑀅𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑁂𑀢𑀺, 𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑁂𑀢𑀺, 𑀅𑀦𑀸𑀕𑀫𑀦𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀤𑁂𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀸𑀦𑀺𑁇

855\. 𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀤𑁂𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀻𑀓𑀢𑁆𑀯𑀸 𑀤𑁂𑀢𑀺, 𑀲𑀳𑀢𑁆𑀣𑀸 𑀤𑁂𑀢𑀺, 𑀅𑀦𑀧𑀯𑀺𑀤𑁆𑀥𑀁 𑀤𑁂𑀢𑀺, 𑀆𑀕𑀫𑀦𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀤𑁂𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀸𑀦𑀻”𑀢𑀺𑁇

---
