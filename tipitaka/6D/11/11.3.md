# 𑁧𑁧𑁇𑁩 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬

### 𑀓𑁂𑀯𑀝𑁆𑀝𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬

(486.)

840. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁? 𑀇𑀥, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀺—  ‘𑀏𑀯𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀣, 𑀫𑀸 𑀏𑀯𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑀬𑀺𑀢𑁆𑀣, 𑀏𑀯𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀣, 𑀫𑀸 𑀏𑀯𑀁 𑀫𑀦𑀲𑀸𑀓𑀢𑁆𑀣, 𑀇𑀤𑀁 𑀧𑀚𑀳𑀣, 𑀇𑀤𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀣𑀸’𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁𑁇

841. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀇𑀥 𑀢𑀣𑀸𑀕𑀢𑁄 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀤𑀫𑁆𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀤𑀫𑁆𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀭𑀢𑀺 𑀅𑀪𑀺𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑁂𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀫𑁆𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀫𑁆𑀧𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁𑁇

842. 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀓𑁂𑀯𑀝𑁆𑀝, 𑀢𑀻𑀡𑀺 𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀸𑀦𑀺 𑀫𑀬𑀸 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑀺𑀢𑀸𑀦𑀺𑁇
