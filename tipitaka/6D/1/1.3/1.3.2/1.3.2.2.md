# 𑁧𑁇𑁩𑁇𑁨𑁇𑁨 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤

### 𑀩𑁆𑀭𑀳𑁆𑀫𑀚𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀺𑀝𑁆𑀞𑀺

### 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑀧𑁆𑀧𑀺𑀓

### 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤

(78.)

137\. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 **𑀅𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁** 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺𑁇 𑀢𑁂 𑀘 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀺𑀫𑀸𑀕𑀫𑁆𑀫 𑀓𑀺𑀫𑀸𑀭𑀩𑁆𑀪 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺?

(79.)

138\. ‘**𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀅𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧:*𑁩𑁫*)

139\. ‘**𑀅𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀅𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁨:*𑁩𑁬*)

140\. ‘**𑀭𑀽𑀧𑀻 𑀘 𑀅𑀭𑀽𑀧𑀻 𑀘 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇 (𑁩:*𑁩𑁭*)

141\. ‘**𑀦𑁂𑀯𑀭𑀽𑀧𑀻 𑀦𑀸𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁪:*𑁩𑁮*)

142\. ‘**𑀅𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁫:*𑁩𑁯*)

143\. ‘**𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁬:*𑁪𑁦*)

144\. ‘**𑀅𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁭:*𑁪𑁧*)

145\. ‘**𑀦𑁂𑀯𑀦𑁆𑀢𑀯𑀸 𑀦𑀸𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀅𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁮:*𑁪𑁨*)

(80.)

146\. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀅𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀳𑁂𑀯 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺, 𑀏𑀢𑁂𑀲𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀬𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑀁 𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑀸 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇
