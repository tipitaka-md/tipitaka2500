# 𑁧𑁇𑁩𑁇𑁨𑁇𑁩 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤

### 𑀩𑁆𑀭𑀳𑁆𑀫𑀚𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀺𑀝𑁆𑀞𑀺

### 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑀧𑁆𑀧𑀺𑀓

### 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤

(81.)

147\. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸, 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 **𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁** 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺𑁇 𑀢𑁂 𑀘 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀺𑀫𑀸𑀕𑀫𑁆𑀫 𑀓𑀺𑀫𑀸𑀭𑀩𑁆𑀪 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺?

(82.)

148\. ‘**𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧:*𑁪𑁩*)

149\. ‘**𑀅𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇 (𑁨:*𑁪𑁪*)

150\. ‘**𑀭𑀽𑀧𑀻 𑀘 𑀅𑀭𑀽𑀧𑀻 𑀘 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁩:*𑁪𑁫*)

151\. ‘**𑀦𑁂𑀯𑀭𑀽𑀧𑀻 𑀦𑀸𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁪:*𑁪𑁬*)

152\. ‘**𑀅𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁫:*𑁪𑁭*)

153\. ‘**𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁬:*𑁪𑁮*)

154\. ‘**𑀅𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁭:*𑁪𑁯*)

155\. ‘**𑀦𑁂𑀯𑀦𑁆𑀢𑀯𑀸 𑀦𑀸𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁮:*𑁫𑁦*)

(83.)

156\. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀳𑁂𑀯 𑀅𑀝𑁆𑀞𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺…𑀧𑁂…  𑀬𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑀁 𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑀸 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇
