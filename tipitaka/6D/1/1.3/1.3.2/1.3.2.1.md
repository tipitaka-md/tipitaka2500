# 𑁧𑁇𑁩𑁇𑁨𑁇𑁧 𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤

* 𑀩𑁆𑀭𑀳𑁆𑀫𑀚𑀸𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀺𑀝𑁆𑀞𑀺

* 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑀧𑁆𑀧𑀺𑀓

* 𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤

(75.)

119\. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 **𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁** 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺𑁇 𑀢𑁂 𑀘 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀺𑀫𑀸𑀕𑀫𑁆𑀫 𑀓𑀺𑀫𑀸𑀭𑀩𑁆𑀪 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺?

(76.)

120\. ‘**𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧:*𑁧𑁯*)

121\. ‘**𑀅𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁨:*𑁨𑁦*)

122\. ‘**𑀭𑀽𑀧𑀻 𑀘 𑀅𑀭𑀽𑀧𑀻 𑀘 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇 (𑁩:*𑁨𑁧*)

123\. ‘**𑀦𑁂𑀯𑀭𑀽𑀧𑀻 𑀦𑀸𑀭𑀽𑀧𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁪:*𑁨𑁨*)

124\. ‘**𑀅𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁫:*𑁨𑁩*)

125\. ‘**𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁬:*𑁨𑁪*)

126\. ‘**𑀅𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀦𑀦𑁆𑀢𑀯𑀸 𑀘 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁭:*𑁨𑁫*)

127\. ‘**𑀦𑁂𑀯𑀦𑁆𑀢𑀯𑀸 𑀦𑀸𑀦𑀦𑁆𑀢𑀯𑀸 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁮:*𑁨𑁬*)

128\. ‘**𑀏𑀓𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁯:*𑁨𑁭*)

129\. ‘**𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁧𑁦:*𑁨𑁮*)

130\. ‘**𑀧𑀭𑀺𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁧𑁧:*𑁨𑁯*)

131\. ‘**𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀲𑀜𑁆𑀜𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁧𑁨:*𑁩𑁦*)

132\. ‘**𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁧𑁩:*𑁩𑁧*)

133\. ‘**𑀏𑀓𑀦𑁆𑀢𑀤𑀼𑀓𑁆𑀔𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁧𑁪:*𑁩𑁨*)

134\. ‘**𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺… 𑁇 (𑁧𑁫:*𑁩𑁩*)

135\. ‘**𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸** 𑀳𑁄𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 **𑀲𑀜𑁆𑀜𑀻**’𑀢𑀺 𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺𑁇 (𑁧𑁬:*𑁩𑁪*)

(77.)

136\. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺𑁇 𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀸 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀺𑀓𑀸 𑀲𑀜𑁆𑀜𑀻𑀯𑀸𑀤𑀸 𑀉𑀤𑁆𑀥𑀫𑀸𑀖𑀸𑀢𑀦𑀁 𑀲𑀜𑁆𑀜𑀺𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀇𑀫𑁂𑀳𑁂𑀯 𑀲𑁄𑀍𑀅𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺, 𑀏𑀢𑁂𑀲𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀢𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀬𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑀁 𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑁆𑀫𑀸 𑀯𑀤𑀫𑀸𑀦𑀸 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇
