# 𑁧𑁇𑁩𑁇𑁨 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑀧𑁆𑀧𑀺𑀓

* 𑀩𑁆𑀭𑀳𑁆𑀫𑀚𑀸𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀺𑀝𑁆𑀞𑀺

* 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑀧𑁆𑀧𑀺𑀓

(74.)

118\. 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑀧𑁆𑀧𑀺𑀓𑀸 𑀅𑀧𑀭𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄, 𑀅𑀧𑀭𑀦𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀺 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀤𑀸𑀦𑀺 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺 𑀘𑀢𑀼𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸𑀬 𑀯𑀢𑁆𑀣𑀽𑀳𑀺𑁇 𑀢𑁂 𑀘 𑀪𑁄𑀦𑁆𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀓𑀺𑀫𑀸𑀕𑀫𑁆𑀫 𑀓𑀺𑀫𑀸𑀭𑀩𑁆𑀪 𑀅𑀧𑀭𑀦𑁆𑀢𑀓𑀧𑁆𑀧𑀺𑀓𑀸 𑀅𑀧𑀭𑀦𑁆𑀢𑀸𑀦𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀅𑀧𑀭𑀦𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀺 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀤𑀸𑀦𑀺 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺 𑀘𑀢𑀼𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸𑀬 𑀯𑀢𑁆𑀣𑀽𑀳𑀺?
