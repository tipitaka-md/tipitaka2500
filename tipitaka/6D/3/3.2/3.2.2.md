# 𑁩𑁇𑁨𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑁃𑀩𑁆𑀪𑀯𑀸𑀤

* 𑀅𑀫𑁆𑀩𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀫𑁆𑀩𑀝𑁆𑀞𑀫𑀸𑀡𑀯

* 𑀤𑀼𑀢𑀺𑀬𑁃𑀩𑁆𑀪𑀯𑀸𑀤

(265.)

359\. “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁂, 𑀅𑀫𑁆𑀩𑀝𑁆𑀞, 𑀲𑀓𑁆𑀬𑀸 𑀅𑀧𑀭𑀤𑁆𑀥𑀼𑀦𑁆”𑀢𑀺? “𑀏𑀓𑀫𑀺𑀤𑀸𑀳𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀲𑀫𑀬𑀁 𑀆𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀧𑁄𑀓𑁆𑀔𑀭𑀲𑀸𑀢𑀺𑀲𑁆𑀲 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀬𑁂𑀦 𑀲𑀓𑁆𑀬𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀕𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀓𑁆𑀬𑀸 𑀘𑁂𑀯 𑀲𑀓𑁆𑀬𑀓𑀼𑀫𑀸𑀭𑀸 𑀘 𑀲𑀦𑁆𑀥𑀸𑀕𑀸𑀭𑁂 𑀉𑀘𑁆𑀘𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀧𑀢𑁄𑀤𑀓𑁂𑀳𑀺 𑀲𑀜𑁆𑀚𑀕𑁆𑀖𑀦𑁆𑀢𑀸 𑀲𑀗𑁆𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀸, 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼 𑀫𑀫𑀜𑁆𑀜𑁂𑀯 𑀫𑀜𑁆𑀜𑁂 𑀅𑀦𑀼𑀚𑀕𑁆𑀖𑀦𑁆𑀢𑀸, 𑀦 𑀫𑀁 𑀓𑁄𑀘𑀺 𑀆𑀲𑀦𑁂𑀦𑀧𑀺 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀲𑀺𑁇 𑀢𑀬𑀺𑀤𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀦𑀘𑁆𑀙𑀦𑁆𑀦𑀁, 𑀢𑀬𑀺𑀤𑀁 𑀦𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁, 𑀬𑀤𑀺𑀫𑁂 𑀲𑀓𑁆𑀬𑀸 𑀇𑀩𑁆𑀪𑀸 𑀲𑀦𑁆𑀢𑀸 𑀇𑀩𑁆𑀪𑀸 𑀲𑀫𑀸𑀦𑀸 𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀲𑀓𑁆𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑀺, 𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀅𑀧𑀘𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀇𑀢𑀺𑀳 𑀅𑀫𑁆𑀩𑀝𑁆𑀞𑁄 𑀫𑀸𑀡𑀯𑁄 𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀲𑀓𑁆𑀬𑁂𑀲𑀼 𑀇𑀩𑁆𑀪𑀯𑀸𑀤𑀁 𑀦𑀺𑀧𑀸𑀢𑁂𑀲𑀺𑁇
