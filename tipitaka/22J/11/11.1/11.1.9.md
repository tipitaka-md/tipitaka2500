# 𑁧𑁧𑁇𑁧𑁇𑁯 𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀚𑀸𑀢𑀓

(108.)

2222\. _“𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀔𑀼𑀭𑀦𑀸𑀲𑀺𑀓𑀸;_  
_𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀓𑀢𑀫𑁄 𑀅𑀬𑀁”𑁇_  


(109.)

2223\. _“𑀓𑀼𑀭𑀼𑀓𑀘𑁆𑀙𑀸 𑀧𑀬𑀸𑀢𑀸𑀦𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀥𑀦𑁂𑀲𑀺𑀦𑀁;_  
_𑀦𑀸𑀯𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀦𑀝𑁆𑀞𑀸𑀬,_  
_𑀔𑀼𑀭𑀫𑀸𑀮𑀻𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(110.)

2224\. _“𑀬𑀣𑀸 𑀅𑀕𑁆𑀕𑀻𑀯 𑀲𑀼𑀭𑀺𑀬𑁄𑀯,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀧𑀝𑀺𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀓𑀢𑀫𑁄 𑀅𑀬𑀁”𑁇_  


(111.)

2225\. _“𑀓𑀼𑀭𑀼𑀓𑀘𑁆𑀙𑀸 𑀧𑀬𑀸𑀢𑀸𑀦𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀥𑀦𑁂𑀲𑀺𑀦𑀁;_  
_𑀦𑀸𑀯𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀦𑀝𑁆𑀞𑀸𑀬,_  
_𑀅𑀕𑁆𑀕𑀺𑀫𑀸𑀮𑀻𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(112.)

2226\. _“𑀬𑀣𑀸 𑀤𑀥𑀻𑀯 𑀔𑀻𑀭𑀁𑀯,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀧𑀝𑀺𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀓𑀢𑀫𑁄 𑀅𑀬𑀁”𑁇_  


(113.)

2227\. _“𑀓𑀼𑀭𑀼𑀓𑀘𑁆𑀙𑀸 𑀧𑀬𑀸𑀢𑀸𑀦𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀥𑀦𑁂𑀲𑀺𑀦𑀁;_  
_𑀦𑀸𑀯𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀦𑀝𑁆𑀞𑀸𑀬,_  
_𑀤𑀥𑀺𑀫𑀸𑀮𑀻𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(114.)

2228\. _“𑀬𑀣𑀸 𑀓𑀼𑀲𑁄𑀯 𑀲𑀲𑁆𑀲𑁄𑀯,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀧𑀝𑀺𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀓𑀢𑀫𑁄 𑀅𑀬𑀁”𑁇_  


(115.)

2229\. _“𑀓𑀼𑀭𑀼𑀓𑀘𑁆𑀙𑀸 𑀧𑀬𑀸𑀢𑀸𑀦𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀥𑀦𑁂𑀲𑀺𑀦𑀁;_  
_𑀦𑀸𑀯𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀦𑀝𑁆𑀞𑀸𑀬,_  
_𑀓𑀼𑀲𑀫𑀸𑀮𑀻𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(116.)

2230\. _“𑀬𑀣𑀸 𑀦𑀍𑀑𑀯 𑀯𑁂𑀍𑀊𑀯,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀧𑀝𑀺𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀓𑀢𑀫𑁄 𑀅𑀬𑀁”𑁇_  


(117.)

2231\. _“𑀓𑀼𑀭𑀼𑀓𑀘𑁆𑀙𑀸 𑀧𑀬𑀸𑀢𑀸𑀦𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀥𑀦𑁂𑀲𑀺𑀦𑀁;_  
_𑀦𑀸𑀯𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀦𑀝𑁆𑀞𑀸𑀬,_  
_𑀦𑀍𑀅𑀫𑀸𑀮𑀻𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(118.)

2232\. _“𑀫𑀳𑀩𑁆𑀪𑀬𑁄 𑀪𑀺𑀁𑀲𑀦𑀓𑁄,_  
_𑀲𑀤𑁆𑀤𑁄 𑀲𑀼𑀬𑁆𑀬𑀢𑀺𑀫𑀸𑀦𑀼𑀲𑁄;_  
_𑀬𑀣𑀸 𑀲𑁄𑀩𑁆𑀪𑁄 𑀧𑀧𑀸𑀢𑁄𑀯,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀧𑀝𑀺𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀓𑀢𑀫𑁄 𑀅𑀬𑀁”𑁇_  


(119.)

2233\. _“𑀓𑀼𑀭𑀼𑀓𑀘𑁆𑀙𑀸 𑀧𑀬𑀸𑀢𑀸𑀦𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀥𑀦𑁂𑀲𑀺𑀦𑀁;_  
_𑀦𑀸𑀯𑀸𑀬 𑀯𑀺𑀧𑁆𑀧𑀦𑀝𑁆𑀞𑀸𑀬,_  
_𑀩𑀍𑀅𑀯𑀸𑀫𑀼𑀔𑀻𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(120.)

2234\. _𑀬𑀢𑁄 𑀲𑀭𑀸𑀫𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀬𑀢𑁄 𑀧𑀢𑁆𑀢𑁄𑀲𑁆𑀫𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀢𑀁;_  
_𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘,_  
_𑀏𑀓𑀧𑀸𑀡𑀫𑁆𑀧𑀺 𑀳𑀺𑀁𑀲𑀺𑀢𑀁;_  
_𑀏𑀢𑁂𑀦 𑀲𑀘𑁆𑀘𑀯𑀚𑁆𑀚𑁂𑀦,_  
_𑀲𑁄𑀢𑁆𑀣𑀺𑀁 𑀦𑀸𑀯𑀸 𑀦𑀺𑀯𑀢𑁆𑀢𑀢𑀽”𑀢𑀺𑁇 (𑁧𑁬𑁪𑁭)_  


---

2235\. 𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇



---

2236\. 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

2237\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2238\. _𑀲𑀺𑀭𑀺𑀫𑀸𑀢𑀼𑀲𑀼𑀧𑁄𑀲𑀓𑀦𑀸𑀕𑀯𑀭𑁄,_  
_𑀧𑀼𑀦 𑀚𑀼𑀡𑁆𑀳𑀓 𑀥𑀫𑁆𑀫𑀫𑀼𑀤𑀬𑀯𑀭𑁄;_  
_𑀅𑀣 𑀧𑀸𑀦𑀺 𑀬𑀼𑀥𑀜𑁆𑀘𑀬𑀓𑁄 𑀘 𑀤𑀲-_  
_𑀭𑀣 𑀲𑀁𑀯𑀭 𑀧𑀸𑀭𑀕𑀢𑁂𑀦 𑀦𑀯𑀸𑀢𑀺𑁇_
