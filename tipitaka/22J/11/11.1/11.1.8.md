# 𑁧𑁧𑁇𑁧𑁇𑁮 𑀲𑀁𑀯𑀭𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀁𑀯𑀭𑀚𑀸𑀢𑀓

(97.)

2210\. _“𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀦𑁄 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀢𑀯 𑀲𑀻𑀮𑀁 𑀚𑀦𑀸𑀥𑀺𑀧𑁄;_  
_𑀇𑀫𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀧𑀽𑀚𑁂𑀦𑁆𑀢𑁄,_  
_𑀦 𑀢𑀁 𑀓𑁂𑀦𑀘𑀺 𑀫𑀜𑁆𑀜𑀣𑁇_  


(98.)

2211\. _𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁂 𑀦𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑁂,_  
_𑀅𑀤𑀼 𑀤𑁂𑀯𑁂 𑀤𑀺𑀯𑀗𑁆𑀕𑀢𑁂;_  
_𑀜𑀸𑀢𑀻 𑀢𑀁 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑀺𑀁𑀲𑀼,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁 𑀅𑀢𑁆𑀣𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


(99.)

2212\. _𑀓𑁂𑀦 𑀲𑀁𑀯𑀭 𑀯𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀜𑁆𑀚𑀸𑀢𑁂 𑀅𑀪𑀺𑀢𑀺𑀝𑁆𑀞𑀲𑀺;_  
_𑀓𑁂𑀦 𑀢𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀜𑀸𑀢𑀺𑀲𑀗𑁆𑀖𑀸 𑀲𑀫𑀸𑀕𑀢𑀸”𑁇_  


(100.)

2213\. _“𑀦 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀉𑀲𑀽𑀬𑀸𑀫𑀺,_  
_𑀲𑀫𑀡𑀸𑀦𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀁;_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀢𑁂 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀫𑀺 𑀢𑀸𑀤𑀺𑀦𑀁𑁇_  


(101.)

2214\. _𑀢𑁂 𑀫𑀁 𑀥𑀫𑁆𑀫𑀕𑀼𑀡𑁂 𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀲𑁆𑀲𑀽𑀲𑀫𑀦𑀼𑀲𑀽𑀬𑀓𑀁;_  
_𑀲𑀫𑀡𑀸 𑀫𑀦𑀼𑀲𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀇𑀲𑀻 𑀥𑀫𑁆𑀫𑀕𑀼𑀡𑁂 𑀭𑀢𑀸𑁇_  


(102.)

2215\. _𑀢𑁂𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀫𑀡𑀸𑀦𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀁;_  
_𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑀸𑀫𑀺,_  
_𑀥𑀫𑁆𑀫𑁂 𑀫𑁂 𑀦𑀺𑀭𑀢𑁄 𑀫𑀦𑁄𑁇_  


(103.)

2216\. _𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀢𑁂𑀲𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀩𑀦𑁆𑀥𑀸𑀫𑀺,_  
_𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀁 𑀪𑀢𑁆𑀢𑀯𑁂𑀢𑀦𑀁𑁇_  


(104.)

2217\. _𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀘 𑀫𑁂 𑀅𑀢𑁆𑀣𑀺,_  
_𑀫𑀦𑁆𑀢𑀺𑀦𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸;_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀁 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀩𑀳𑀼𑀫𑀁𑀲𑀲𑀼𑀭𑁄𑀤𑀓𑀁𑁇_  


(105.)

2218\. _𑀅𑀣𑁄𑀧𑀺 𑀯𑀸𑀡𑀺𑀚𑀸 𑀨𑀻𑀢𑀸,_  
_𑀦𑀸𑀦𑀸𑀭𑀝𑁆𑀞𑁂𑀳𑀺 𑀆𑀕𑀢𑀸;_  
_𑀢𑁂𑀲𑀼 𑀫𑁂 𑀯𑀺𑀳𑀺𑀢𑀸 𑀭𑀓𑁆𑀔𑀸,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀼𑀧𑁄𑀲𑀣”𑁇_  


(106.)

2219\. _“𑀥𑀫𑁆𑀫𑁂𑀦 𑀓𑀺𑀭 𑀜𑀸𑀢𑀻𑀦𑀁,_  
_𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀳𑀺 𑀲𑀁𑀯𑀭;_  
_𑀫𑁂𑀥𑀸𑀯𑀻 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀘𑀸𑀲𑀺,_  
_𑀅𑀣𑁄𑀧𑀺 𑀜𑀸𑀢𑀺𑀦𑀁 𑀳𑀺𑀢𑁄𑁇_  


(107.)

2220\. _𑀢𑀁 𑀢𑀁 𑀜𑀸𑀢𑀺𑀧𑀭𑀺𑀩𑁆𑀬𑀽𑀍𑀳𑀁,_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀫𑁄𑀘𑀺𑀢𑀁;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀦𑀧𑁆𑀧𑀲𑀳𑀦𑁆𑀢𑀺,_  
_𑀇𑀦𑁆𑀤𑀁𑀯 𑀅𑀲𑀼𑀭𑀸𑀥𑀺𑀧𑁄”𑀢𑀺𑁇_  


---

2221\. 𑀲𑀁𑀯𑀭𑀚𑀸𑀢𑀓𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
