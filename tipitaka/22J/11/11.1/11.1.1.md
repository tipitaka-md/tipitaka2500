# 𑁧𑁧𑁇𑁧𑁇𑁧 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀚𑀸𑀢𑀓

(1.)

2107. _“𑀢𑀲𑁆𑀲 𑀦𑀸𑀕𑀲𑁆𑀲 𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁂𑀦,_  
_𑀯𑀺𑀭𑀽𑀍𑀳𑀸 𑀲𑀮𑁆𑀮𑀓𑀻 𑀘 𑀓𑀼𑀝𑀚𑀸 𑀘;_  
_𑀓𑀼𑀭𑀼𑀯𑀺𑀦𑁆𑀤𑀓𑀭𑀯𑀻𑀭𑀸 𑀪𑀺𑀲𑀲𑀸𑀫𑀸 𑀘,_  
_𑀦𑀺𑀯𑀸𑀢𑁂 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀘 𑀓𑀡𑀺𑀓𑀸𑀭𑀸𑁇_  


(2.)

2108. _𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀸𑀬𑀼𑀭𑀸,_  
_𑀦𑀸𑀕𑀭𑀸𑀚𑀁 𑀪𑀭𑀦𑁆𑀢𑀺 𑀧𑀺𑀡𑁆𑀟𑁂𑀦;_  
_𑀬𑀢𑁆𑀣 𑀭𑀸𑀚𑀸 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑁄 𑀯𑀸,_  
_𑀓𑀯𑀘𑀫𑀪𑀺𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀅𑀙𑀫𑁆𑀪𑀺𑀢𑁄”𑁇_  


(3.)

2109. _“𑀕𑀡𑁆𑀳𑀸𑀳𑀺 𑀦𑀸𑀕 𑀓𑀩𑀍𑀅𑀁,_  
_𑀫𑀸 𑀦𑀸𑀕 𑀓𑀺𑀲𑀓𑁄 𑀪𑀯;_  
_𑀩𑀳𑀽𑀦𑀺 𑀭𑀸𑀚𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀺,_  
_𑀢𑀸𑀦𑀺 𑀦𑀸𑀕 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(4.)

2110. _“𑀲𑀸 𑀦𑀽𑀦𑀲𑀸 𑀓𑀧𑀡𑀺𑀓𑀸,_  
_𑀅𑀦𑁆𑀥𑀸 𑀅𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸;_  
_𑀔𑀸𑀡𑀼𑀁 𑀧𑀸𑀤𑁂𑀦 𑀖𑀝𑁆𑀝𑁂𑀢𑀺,_  
_𑀕𑀺𑀭𑀺𑀁 𑀘𑀡𑁆𑀟𑁄𑀭𑀡𑀁 𑀧𑀢𑀺”𑁇_  


(5.)

2111. _“𑀓𑀸 𑀦𑀼 𑀢𑁂 𑀲𑀸 𑀫𑀳𑀸𑀦𑀸𑀕,_  
_𑀅𑀦𑁆𑀥𑀸 𑀅𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸;_  
_𑀔𑀸𑀡𑀼𑀁 𑀧𑀸𑀤𑁂𑀦 𑀖𑀝𑁆𑀝𑁂𑀢𑀺,_  
_𑀕𑀺𑀭𑀺𑀁 𑀘𑀡𑁆𑀟𑁄𑀭𑀡𑀁 𑀧𑀢𑀺”𑁇_  


(6.)

2112. _“𑀫𑀸𑀢𑀸 𑀫𑁂 𑀲𑀸 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀅𑀦𑁆𑀥𑀸 𑀅𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸;_  
_𑀔𑀸𑀡𑀼𑀁 𑀧𑀸𑀤𑁂𑀦 𑀖𑀝𑁆𑀝𑁂𑀢𑀺,_  
_𑀕𑀺𑀭𑀺𑀁 𑀘𑀡𑁆𑀟𑁄𑀭𑀡𑀁 𑀧𑀢𑀺”𑁇_  


(7.)

2113. _“𑀫𑀼𑀜𑁆𑀘𑀣𑁂𑀢𑀁 𑀫𑀳𑀸𑀦𑀸𑀕𑀁,_  
_𑀬𑁄𑀬𑀁 𑀪𑀭𑀢𑀺 𑀫𑀸𑀢𑀭𑀁;_  
_𑀲𑀫𑁂𑀢𑀼 𑀫𑀸𑀢𑀭𑀸 𑀦𑀸𑀕𑁄,_  
_𑀲𑀳 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀜𑀸𑀢𑀺𑀪𑀺”𑁇_  


(8.)

2114. _“𑀫𑀼𑀢𑁆𑀢𑁄 𑀘 𑀩𑀦𑁆𑀥𑀦𑀸 𑀦𑀸𑀕𑁄,_  
_𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑀤𑀸𑀬 𑀓𑀼𑀜𑁆𑀚𑀭𑁄;_  
_𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀅𑀲𑁆𑀲𑀸𑀲𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀕𑀫𑀸 𑀬𑁂𑀦 𑀧𑀩𑁆𑀩𑀢𑁄𑁇_  


(9.)

2115. _𑀢𑀢𑁄 𑀲𑁄 𑀦𑀍𑀇𑀦𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀲𑀻𑀢𑀁 𑀓𑀼𑀜𑁆𑀚𑀭𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀲𑁄𑀡𑁆𑀟𑀸𑀬𑀽𑀤𑀓𑀫𑀸𑀳𑀢𑁆𑀯𑀸,_  
_𑀫𑀸𑀢𑀭𑀁 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑀣”𑁇_  


(10.)

2116. _“𑀓𑁄𑀬𑀁 𑀅𑀦𑀭𑀺𑀬𑁄 𑀤𑁂𑀯𑁄,_  
_𑀅𑀓𑀸𑀮𑁂𑀦𑀧𑀺 𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀕𑀢𑁄 𑀫𑁂 𑀅𑀢𑁆𑀭𑀚𑁄 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀬𑁄 𑀫𑀬𑁆𑀳𑀁 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁄”𑁇_  


(11.)

2117. _“𑀉𑀝𑁆𑀞𑁂𑀳𑀺 𑀅𑀫𑁆𑀫 𑀓𑀺𑀁 𑀲𑁂𑀲𑀺,_  
_𑀆𑀕𑀢𑁄 𑀢𑁆𑀬𑀸𑀳𑀫𑀢𑁆𑀭𑀚𑁄;_  
_𑀫𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑁂𑀦,_  
_𑀯𑁂𑀤𑁂𑀳𑁂𑀦 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀸”𑁇_  


(12.)

2118. _“𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀢𑀼 𑀲𑁄 𑀭𑀸𑀚𑀸,_  
_𑀓𑀸𑀲𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁄;_  
_𑀬𑁄 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀫𑁄𑀘𑁂𑀲𑀺,_  
_𑀲𑀤𑀸 𑀯𑀼𑀤𑁆𑀥𑀸𑀧𑀘𑀸𑀬𑀺𑀓𑀦𑁆”𑀢𑀺𑁇_  


---

2119. 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀚𑀸𑀢𑀓𑀁 𑀧𑀞𑀫𑀁𑁇
