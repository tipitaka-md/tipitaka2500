# 𑁧𑁧𑁇𑁧𑁇𑁫 𑀧𑀸𑀦𑀻𑀬𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀦𑀻𑀬𑀚𑀸𑀢𑀓

(59.)

2169\. _“𑀫𑀺𑀢𑁆𑀢𑁄 𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀦𑀻𑀬𑀁,_  
_𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑀁;_  
_𑀢𑁂𑀦 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀺𑀁,_  
_𑀢𑀁 𑀧𑀸𑀧𑀁 𑀧𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀫𑀸 𑀧𑀼𑀦 𑀅𑀓𑀭𑀁 𑀧𑀸𑀧𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁𑁇_  


(60.)

2170\. _𑀧𑀭𑀤𑀸𑀭𑀜𑁆𑀘 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀙𑀦𑁆𑀤𑁄 𑀫𑁂 𑀉𑀤𑀧𑀚𑁆𑀚𑀣;_  
_𑀢𑁂𑀦 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀺𑀁,_  
_𑀢𑀁 𑀧𑀸𑀧𑀁 𑀧𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀫𑀸 𑀧𑀼𑀦 𑀅𑀓𑀭𑀁 𑀧𑀸𑀧𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁𑁇_  


(61.)

2171\. _𑀧𑀺𑀢𑀭𑀁 𑀫𑁂 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀘𑁄𑀭𑀸 𑀅𑀕𑀡𑁆𑀳𑀼 𑀓𑀸𑀦𑀦𑁂;_  
_𑀢𑁂𑀲𑀸𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀚𑀸𑀦𑀁,_  
_𑀅𑀜𑁆𑀜𑀣𑀸 𑀦𑀁 𑀯𑀺𑀬𑀸𑀓𑀭𑀺𑀁𑁇_  


(62.)

2172\. _𑀢𑁂𑀦 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀺𑀁,_  
_𑀢𑀁 𑀧𑀸𑀧𑀁 𑀧𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀫𑀸 𑀧𑀼𑀦 𑀅𑀓𑀭𑀁 𑀧𑀸𑀧𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁𑁇_  


(63.)

2173\. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀫𑀓𑀭𑀼𑀁,_  
_𑀲𑁄𑀫𑀬𑀸𑀕𑁂 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁂;_  
_𑀢𑁂𑀲𑀸𑀳𑀁 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇_  


(64.)

2174\. _𑀢𑁂𑀦 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀺𑀁,_  
_𑀢𑀁 𑀧𑀸𑀧𑀁 𑀧𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀫𑀸 𑀧𑀼𑀦 𑀅𑀓𑀭𑀁 𑀧𑀸𑀧𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁𑁇_  


(65.)

2175\. _𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀸𑀥𑀼𑀓𑀸,_  
_𑀬𑁂 𑀚𑀦𑀸 𑀧𑀞𑀫𑀸𑀲𑀼 𑀦𑁄;_  
_𑀩𑀳𑀽𑀦𑀁 𑀢𑁂 𑀅𑀦𑀢𑁆𑀣𑀸𑀬,_  
_𑀫𑀚𑁆𑀚𑀧𑀸𑀦𑀫𑀓𑀧𑁆𑀧𑀬𑀼𑀁;_  
_𑀢𑁂𑀲𑀸𑀳𑀁 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇_  


(66.)

2176\. _𑀢𑁂𑀦 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀺𑀁,_  
_𑀢𑀁 𑀧𑀸𑀧𑀁 𑀧𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀫𑀸 𑀧𑀼𑀦 𑀅𑀓𑀭𑀁 𑀧𑀸𑀧𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁”𑁇_  


(67.)

2177\. _“𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀲𑀼𑀩𑀳𑀽 𑀓𑀸𑀫𑁂,_  
_𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑁂 𑀩𑀳𑀼𑀓𑀡𑁆𑀝𑀓𑁂;_  
_𑀬𑁂 𑀅𑀳𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑁄,_  
_𑀦𑀸𑀮𑀪𑀺𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀲𑀼𑀔𑀁”𑁇_  


(68.)

2178\. _“𑀫𑀳𑀲𑁆𑀲𑀸𑀤𑀸 𑀲𑀼𑀔𑀸 𑀓𑀸𑀫𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸 𑀧𑀭𑀁 𑀲𑀼𑀔𑀁;_  
_𑀬𑁂 𑀓𑀸𑀫𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺,_  
_𑀲𑀕𑁆𑀕𑀁 𑀢𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀭𑁂”𑁇_  


(69.)

2179\. _“𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀤𑀼𑀔𑀸 𑀓𑀸𑀫𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑀸 𑀧𑀭𑀁 𑀤𑀼𑀔𑀁;_  
_𑀬𑁂 𑀓𑀸𑀫𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀭𑀬𑀁 𑀢𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀭𑁂𑁇_  


(70.)

2180\. _𑀅𑀲𑀻 𑀬𑀣𑀸 𑀲𑀼𑀦𑀺𑀲𑀺𑀢𑁄,_  
_𑀦𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑁄𑀯 𑀲𑀼𑀧𑀸𑀬𑀺𑀓𑁄;_  
_𑀲𑀢𑁆𑀢𑀻𑀯 𑀉𑀭𑀲𑀺 𑀔𑀺𑀢𑁆𑀢𑀸,_  
_𑀓𑀸𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀢𑀭𑀸 𑀢𑀢𑁄𑁇_  


(71.)

2181\. _𑀅𑀗𑁆𑀕𑀸𑀭𑀸𑀦𑀁𑀯 𑀚𑀮𑀺𑀢𑀁,_  
_𑀓𑀸𑀲𑀼𑀁 𑀲𑀸𑀥𑀺𑀓𑀧𑁄𑀭𑀺𑀲𑀁;_  
_𑀨𑀸𑀮𑀁𑀯 𑀤𑀺𑀯𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀁,_  
_𑀓𑀸𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀢𑀭𑀸 𑀢𑀢𑁄𑁇_  


(72.)

2182\. _𑀯𑀺𑀲𑀁 𑀬𑀣𑀸 𑀳𑀮𑀸𑀳𑀮𑀁,_  
_𑀢𑁂𑀮𑀁 𑀧𑀓𑁆𑀓𑀼𑀣𑀺𑀢𑀁 𑀬𑀣𑀸;_  
_𑀢𑀫𑁆𑀩𑀮𑁄𑀳 𑀯𑀺𑀮𑀻𑀦𑀁𑀯,_  
_𑀓𑀸𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀢𑀭𑀸 𑀢𑀢𑁄”𑀢𑀺𑁇_  


---

2183\. 𑀧𑀸𑀦𑀻𑀬𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
