# 𑁧𑁧𑁇𑁧𑁇𑁨 𑀚𑀼𑀡𑁆𑀳𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀚𑀼𑀡𑁆𑀳𑀚𑀸𑀢𑀓

(13.)

2120. _“𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑀬𑁆𑀳𑀁 𑀯𑀘𑀦𑀁 𑀚𑀦𑀺𑀦𑁆𑀤,_  
_𑀅𑀢𑁆𑀣𑁂𑀦 𑀚𑀼𑀡𑁆𑀳𑀫𑁆𑀳𑀺 𑀇𑀥𑀸𑀦𑀼𑀧𑀢𑁆𑀢𑁄;_  
_𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀅𑀤𑁆𑀥𑀺𑀓𑁂 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁂,_  
_𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀫𑀸𑀳𑀼 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤𑀲𑁂𑀝𑁆𑀞”𑁇_  


(14.)

2121. _“𑀲𑀼𑀡𑁄𑀫𑀺 𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀯𑀤𑁂𑀳𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀬𑁂𑀦𑀸𑀲𑀺 𑀅𑀢𑁆𑀣𑁂𑀦 𑀇𑀥𑀸𑀦𑀼𑀧𑀢𑁆𑀢𑁄;_  
_𑀓𑀁 𑀯𑀸 𑀢𑁆𑀯𑀫𑀢𑁆𑀣𑀁 𑀫𑀬𑀺 𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄,_  
_𑀇𑀥𑀸𑀕𑀫𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀢𑀤𑀺𑀗𑁆𑀖 𑀩𑁆𑀭𑀽𑀳𑀺”𑁇_  


(15.)

2122. _“𑀤𑀤𑀸𑀳𑀺 𑀫𑁂 𑀕𑀸𑀫𑀯𑀭𑀸𑀦𑀺 𑀧𑀜𑁆𑀘,_  
_𑀤𑀸𑀲𑀻𑀲𑀢𑀁 𑀲𑀢𑁆𑀢 𑀕𑀯𑀁𑀲𑀢𑀸𑀦𑀺;_  
_𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀜𑁆𑀘 𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑀺𑀓𑁆𑀔𑁂,_  
_𑀪𑀭𑀺𑀬𑀸 𑀘 𑀫𑁂 𑀲𑀸𑀤𑀺𑀲𑀻 𑀤𑁆𑀯𑁂 𑀤𑀤𑀸𑀳𑀺𑁇_  


(16.)

2123. _𑀢𑀧𑁄 𑀦𑀼 𑀢𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀪𑀺𑀁𑀲𑀭𑀽𑀧𑁄,_  
_𑀫𑀦𑁆𑀢𑀸 𑀦𑀼 𑀢𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀘𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑀸;_  
_𑀬𑀓𑁆𑀔𑀸 𑀦𑀼 𑀢𑁂 𑀅𑀲𑁆𑀲𑀯𑀸 𑀲𑀦𑁆𑀢𑀺 𑀓𑁂𑀘𑀺,_  
_𑀅𑀢𑁆𑀣𑀁 𑀯𑀸 𑀫𑁂 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀲𑀺 𑀓𑀢𑁆𑀢𑀁𑁇_  


(17.)

2124. _𑀦 𑀫𑁂 𑀢𑀧𑁄 𑀅𑀢𑁆𑀣𑀺 𑀦 𑀘𑀸𑀧𑀺 𑀫𑀦𑁆𑀢𑀸,_  
_𑀬𑀓𑁆𑀔𑀸𑀧𑀺 𑀫𑁂 𑀅𑀲𑁆𑀲𑀯𑀸 𑀦𑀢𑁆𑀣𑀺 𑀓𑁂𑀘𑀺;_  
_𑀅𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀓𑀢𑁆𑀢𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀘 𑀔𑁄 𑀲𑀗𑁆𑀕𑀢𑀺𑀫𑀢𑁆𑀢𑀫𑀸𑀲𑀺𑁇_  


(18.)

2125. _𑀧𑀞𑀫𑀁 𑀇𑀤𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀚𑀸𑀦𑀢𑁄 𑀫𑁂,_  
_𑀦 𑀢𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀇𑀢𑁄 𑀧𑀼𑀭𑀢𑁆𑀣𑀸;_  
_𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀓𑀤𑀸 𑀓𑀼𑀳𑀺𑀁 𑀯𑀸 𑀅𑀳𑀼 𑀲𑀗𑁆𑀕𑀫𑁄 𑀦𑁄𑁇_  


(19.)

2126. _𑀕𑀦𑁆𑀥𑀸𑀭𑀭𑀸𑀚𑀲𑁆𑀲 𑀧𑀼𑀭𑀫𑁆𑀳𑀺 𑀭𑀫𑁆𑀫𑁂,_  
_𑀅𑀯𑀲𑀺𑀫𑁆𑀳𑀲𑁂 𑀢𑀓𑁆𑀓𑀲𑀻𑀮𑀸𑀬𑀁 𑀤𑁂𑀯;_  
_𑀢𑀢𑁆𑀣𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑁆𑀳𑀺 𑀢𑀺𑀫𑀻𑀲𑀺𑀓𑀸𑀬𑀁,_  
_𑀅𑀁𑀲𑁂𑀦 𑀅𑀁𑀲𑀁 𑀲𑀫𑀖𑀝𑁆𑀝𑀬𑀺𑀫𑁆𑀳𑁇_  


(20.)

2127. _𑀢𑁂 𑀢𑀢𑁆𑀣 𑀞𑀢𑁆𑀯𑀸𑀦 𑀉𑀪𑁄 𑀚𑀦𑀺𑀦𑁆𑀤,_  
_𑀲𑀸𑀭𑀸𑀡𑀺𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑀬𑀺𑀫𑁆𑀳 𑀢𑀢𑁆𑀣;_  
_𑀲𑀸𑀬𑁂𑀯 𑀦𑁄 𑀲𑀗𑁆𑀕𑀢𑀺𑀫𑀢𑁆𑀢𑀫𑀸𑀲𑀺,_  
_𑀢𑀢𑁄 𑀦 𑀧𑀘𑁆𑀙𑀸 𑀦 𑀧𑀼𑀭𑁂 𑀅𑀳𑁄𑀲𑀺𑁇_  


(21.)

2128. _𑀬𑀤𑀸 𑀓𑀤𑀸𑀘𑀺 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀲𑀫𑀸𑀕𑀫𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀳𑁄𑀢𑀺;_  
_𑀦 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀲𑀗𑁆𑀕𑀢𑀺𑀲𑀦𑁆𑀣𑀯𑀸𑀦𑀺,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀁 𑀯𑀸𑀧𑀺 𑀯𑀺𑀦𑀸𑀲𑀬𑀦𑁆𑀢𑀺𑁇_  


(22.)

2129. _𑀩𑀸𑀮𑀸𑀯 𑀔𑁄 𑀲𑀗𑁆𑀕𑀢𑀺𑀲𑀦𑁆𑀣𑀯𑀸𑀦𑀺,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀁 𑀯𑀸𑀧𑀺 𑀯𑀺𑀦𑀸𑀲𑀬𑀦𑁆𑀢𑀺;_  
_𑀩𑀳𑀼𑀫𑁆𑀧𑀺 𑀩𑀸𑀮𑁂𑀲𑀼 𑀓𑀢𑀁 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺,_  
_𑀢𑀣𑀸 𑀳𑀺 𑀩𑀸𑀮𑀸 𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀭𑀽𑀧𑀸𑁇_  


(23.)

2130. _𑀥𑀻𑀭𑀸 𑀘 𑀔𑁄 𑀲𑀗𑁆𑀕𑀢𑀺𑀲𑀦𑁆𑀣𑀯𑀸𑀦𑀺,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀁 𑀯𑀸𑀧𑀺 𑀦 𑀦𑀸𑀲𑀬𑀦𑁆𑀢𑀺;_  
_𑀅𑀧𑁆𑀧𑀫𑁆𑀧𑀺 𑀥𑀻𑀭𑁂𑀲𑀼 𑀓𑀢𑀁 𑀦 𑀦𑀲𑁆𑀲𑀢𑀺,_  
_𑀢𑀣𑀸 𑀳𑀺 𑀥𑀻𑀭𑀸 𑀲𑀼𑀓𑀢𑀜𑁆𑀜𑀼𑀭𑀽𑀧𑀸𑁇_  


(24.)

2131. _𑀤𑀤𑀸𑀫𑀺 𑀢𑁂 𑀕𑀸𑀫𑀯𑀭𑀸𑀦𑀺 𑀧𑀜𑁆𑀘,_  
_𑀤𑀸𑀲𑀻𑀲𑀢𑀁 𑀲𑀢𑁆𑀢 𑀕𑀯𑀁𑀲𑀢𑀸𑀦𑀺;_  
_𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀜𑁆𑀘 𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑀺𑀓𑁆𑀔𑁂,_  
_𑀪𑀭𑀺𑀬𑀸 𑀘 𑀢𑁂 𑀲𑀸𑀤𑀺𑀲𑀻 𑀤𑁆𑀯𑁂 𑀤𑀤𑀸𑀫𑀺𑁇_  


(25.)

2132. _𑀏𑀯𑀁 𑀲𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀫𑁂𑀘𑁆𑀘 𑀭𑀸𑀚,_  
_𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀭𑀸𑀚𑀸𑀭𑀺𑀯 𑀢𑀸𑀭𑀓𑀸𑀦𑀁;_  
_𑀆𑀧𑀽𑀭𑀢𑀻 𑀓𑀸𑀲𑀺𑀧𑀢𑀻 𑀢𑀣𑀸𑀳𑀁,_  
_𑀢𑀬𑀸𑀧𑀺 𑀫𑁂 𑀲𑀗𑁆𑀕𑀫𑁄 𑀅𑀚𑁆𑀚 𑀮𑀤𑁆𑀥𑁄”𑀢𑀺𑁇_  


---

2133. 𑀚𑀼𑀡𑁆𑀳𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
