# 𑁧𑁩𑁇𑁧𑁇𑁩 𑀚𑀯𑀦𑀳𑀁𑀲𑀚𑀸𑀢𑀓

### 𑀢𑁂𑀭𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀫𑁆𑀩𑀯𑀕𑁆𑀕

### 𑀚𑀯𑀦𑀳𑀁𑀲𑀚𑀸𑀢𑀓

(27.)

2412\. _“𑀇𑀥𑁂𑀯 𑀳𑀁𑀲 𑀦𑀺𑀧𑀢,_  
_𑀧𑀺𑀬𑀁 𑀫𑁂 𑀢𑀯 𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀇𑀲𑁆𑀲𑀭𑁄𑀲𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀬𑀫𑀺𑀥𑀢𑁆𑀣𑀺 𑀧𑀯𑁂𑀤𑀬”𑁇_  


(28.)

2413\. _“𑀲𑀯𑀦𑁂𑀦 𑀏𑀓𑀲𑁆𑀲 𑀧𑀺𑀬𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀦𑁂𑀓𑀲𑁆𑀲 𑀯𑀺𑀬𑁂𑀢𑀺 𑀙𑀦𑁆𑀤𑁄;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀘 𑀲𑀼𑀢𑁆𑀯𑀸 𑀘 𑀧𑀺𑀬𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀓𑀘𑁆𑀘𑀺𑀦𑁆𑀦𑀼 𑀫𑁂 𑀧𑀻𑀬𑀲𑀺 𑀤𑀲𑁆𑀲𑀦𑁂𑀦𑁇_  


(29.)

2414\. _𑀲𑀯𑀦𑁂𑀦 𑀧𑀺𑀬𑁄 𑀫𑁂𑀲𑀺,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀘𑀸𑀕𑀫𑁆𑀫 𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀏𑀯𑀁 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀦𑁄 𑀫𑁂,_  
_𑀯𑀲 𑀳𑀁𑀲 𑀫𑀫𑀦𑁆𑀢𑀺𑀓𑁂”𑁇_  


(30.)

2415\. _“𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫 𑀢𑀯𑀸𑀕𑀸𑀭𑁂,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑀓𑁆𑀓𑀢𑀧𑀽𑀚𑀺𑀢𑀸;_  
_𑀫𑀢𑁆𑀢𑁄 𑀘 𑀏𑀓𑀤𑀸 𑀯𑀚𑁆𑀚𑁂,_  
_‘𑀳𑀁𑀲𑀭𑀸𑀚𑀁 𑀧𑀘𑀦𑁆𑀢𑀼 𑀫𑁂’”𑁇_  


(31.)

2416\. _“𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀢𑀁 𑀫𑀚𑁆𑀚𑀧𑀸𑀦𑀁,_  
_𑀬𑀁 𑀫𑁂 𑀧𑀺𑀬𑀢𑀭𑀁 𑀢𑀬𑀸;_  
_𑀦 𑀘𑀸𑀧𑀺 𑀫𑀚𑁆𑀚𑀁 𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀸𑀯 𑀫𑁂 𑀯𑀘𑁆𑀙𑀲𑀻 𑀖𑀭𑁂”𑁇_  


(32.)

2417\. _“𑀲𑀼𑀯𑀺𑀚𑀸𑀦𑀁 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀸𑀦𑀁,_  
_𑀲𑀓𑀼𑀡𑀸𑀦𑀜𑁆𑀘 𑀯𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀲𑁆𑀲𑀺𑀢𑀁 𑀭𑀸𑀚,_  
_𑀤𑀼𑀩𑁆𑀩𑀺𑀚𑀸𑀦𑀢𑀭𑀁 𑀢𑀢𑁄𑁇_  


(33.)

2418\. _𑀅𑀧𑀺 𑀘𑁂 𑀫𑀜𑁆𑀜𑀢𑀻 𑀧𑁄𑀲𑁄,_  
_𑀜𑀸𑀢𑀺 𑀫𑀺𑀢𑁆𑀢𑁄 𑀲𑀔𑀸𑀢𑀺 𑀯𑀸;_  
_𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀼𑀫𑀦𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀧𑀘𑁆𑀙𑀸 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀢𑁂 𑀤𑀺𑀲𑁄𑁇_  


(34.)

2419\. _𑀬𑀲𑁆𑀫𑀺𑀁 𑀫𑀦𑁄 𑀦𑀺𑀯𑀺𑀲𑀢𑀺,_  
_𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀲𑀳𑀸𑀧𑀺 𑀲𑁄;_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑀧𑀺 𑀳𑀺 𑀲𑁄 𑀤𑀽𑀭𑁂,_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀦𑀸𑀯𑀺𑀲𑀢𑁂 𑀫𑀦𑁄𑁇_  


(35.)

2420\. _𑀅𑀦𑁆𑀢𑁄𑀧𑀺 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀧𑀸𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀅𑀦𑁆𑀢𑁄𑀧𑀺 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀧𑀸𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄𑁇_  


(36.)

2421\. _𑀲𑀁𑀯𑀲𑀦𑁆𑀢𑀸 𑀯𑀺𑀯𑀲𑀦𑁆𑀢𑀺,_  
_𑀬𑁂 𑀤𑀺𑀲𑀸 𑀢𑁂 𑀭𑀣𑁂𑀲𑀪;_  
_𑀆𑀭𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀦𑀲𑀸 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁇_  


(37.)

2422\. _𑀅𑀢𑀺𑀘𑀺𑀭𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀦,_  
_𑀧𑀺𑀬𑁄 𑀪𑀯𑀢𑀺 𑀅𑀧𑁆𑀧𑀺𑀬𑁄;_  
_𑀆𑀫𑀦𑁆𑀢 𑀔𑁄 𑀢𑀁 𑀕𑀘𑁆𑀙𑀸𑀫,_  
_𑀧𑀼𑀭𑀸 𑀢𑁂 𑀳𑁄𑀫 𑀅𑀧𑁆𑀧𑀺𑀬𑀸”𑁇_  


(38.)

2423\. _“𑀏𑀯𑀜𑁆𑀘𑁂 𑀬𑀸𑀘𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀦𑀸𑀯𑀩𑀼𑀚𑁆𑀛𑀲𑀺;_  
_𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸𑀦𑀁 𑀲𑀢𑀁,_  
_𑀯𑀘𑀦𑀁 𑀦 𑀓𑀭𑁄𑀲𑀺 𑀦𑁄;_  
_𑀏𑀯𑀁 𑀢𑀁 𑀅𑀪𑀺𑀬𑀸𑀘𑀸𑀫,_  
_𑀧𑀼𑀦 𑀓𑀬𑀺𑀭𑀸𑀲𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑀁”𑁇_  


(39.)

2424\. _“𑀏𑀯𑀜𑁆𑀘𑁂 𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑀁,_  
_𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀦 𑀳𑁂𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀼𑀬𑁆𑀳𑀜𑁆𑀘𑀸𑀧𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀫𑀬𑁆𑀳𑀜𑁆𑀘 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦;_  
_𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀧𑀲𑁆𑀲𑁂𑀫𑀼,_  
_𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀸𑀦𑀫𑀘𑁆𑀘𑀬𑁂”𑀢𑀺𑁇_  


---
