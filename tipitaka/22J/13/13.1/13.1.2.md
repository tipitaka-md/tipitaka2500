# 𑁧𑁩𑁇𑁧𑁇𑁨 𑀨𑀦𑁆𑀤𑀦𑀚𑀸𑀢𑀓

* 𑀢𑁂𑀭𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀫𑁆𑀩𑀯𑀕𑁆𑀕

* 𑀨𑀦𑁆𑀤𑀦𑀚𑀸𑀢𑀓

(14.)

2398\. _“𑀓𑀼𑀞𑀸𑀭𑀺𑀳𑀢𑁆𑀣𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀯𑀦𑀫𑁄𑀕𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀲𑀺;_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀫𑁂 𑀲𑀫𑁆𑀫 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀓𑀺𑀁 𑀤𑀸𑀭𑀼𑀁 𑀙𑁂𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀲𑀺”𑁇_  


(15.)

2399\. _“𑀇𑀲𑁆𑀲𑁄 𑀯𑀦𑀸𑀦𑀺 𑀘𑀭𑀲𑀺,_  
_𑀲𑀫𑀸𑀦𑀺 𑀯𑀺𑀲𑀫𑀸𑀦𑀺 𑀘;_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀫𑁂 𑀲𑀫𑁆𑀫 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀓𑀺𑀁 𑀤𑀸𑀭𑀼𑀁 𑀦𑁂𑀫𑀺𑀬𑀸 𑀤𑀍𑀳𑀁”𑁇_  


(16.)

2400\. _“𑀦𑁂𑀯 𑀲𑀸𑀮𑁄 𑀦 𑀔𑀤𑀺𑀭𑁄,_  
_𑀦𑀸𑀲𑁆𑀲𑀓𑀡𑁆𑀡𑁄 𑀓𑀼𑀢𑁄 𑀥𑀯𑁄;_  
_𑀭𑀼𑀓𑁆𑀔𑁄 𑀘 𑀨𑀦𑁆𑀤𑀦𑁄 𑀦𑀸𑀫,_  
_𑀢𑀁 𑀤𑀸𑀭𑀼𑀁 𑀦𑁂𑀫𑀺𑀬𑀸 𑀤𑀍𑀳𑀁”𑁇_  


(17.)

2401\. _“𑀓𑀻𑀤𑀺𑀲𑀸𑀦𑀺𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀔𑀦𑁆𑀥𑁄 𑀯𑀸 𑀧𑀦 𑀓𑀻𑀤𑀺𑀲𑁄;_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀫𑁂 𑀲𑀫𑁆𑀫 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀬𑀣𑀸 𑀚𑀸𑀦𑁂𑀫𑀼 𑀨𑀦𑁆𑀤𑀦𑀁”𑁇_  


(18.)

2402\. _“𑀬𑀲𑁆𑀲 𑀲𑀸𑀔𑀸 𑀧𑀮𑀫𑁆𑀩𑀦𑁆𑀢𑀺,_  
_𑀦𑀫𑀦𑁆𑀢𑀺 𑀦 𑀘 𑀪𑀜𑁆𑀚𑀭𑁂;_  
_𑀲𑁄 𑀭𑀼𑀓𑁆𑀔𑁄 𑀨𑀦𑁆𑀤𑀦𑁄 𑀦𑀸𑀫,_  
_𑀬𑀲𑁆𑀲 𑀫𑀽𑀮𑁂 𑀅𑀳𑀁 𑀞𑀺𑀢𑁄𑁇_  


(19.)

2403\. _𑀅𑀭𑀸𑀦𑀁 𑀘𑀓𑁆𑀓𑀦𑀸𑀪𑀻𑀦𑀁,_  
_𑀈𑀲𑀸𑀦𑁂𑀫𑀺𑀭𑀣𑀲𑁆𑀲 𑀘;_  
_𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀢𑁂 𑀓𑀫𑁆𑀫𑀦𑀺𑀬𑁄,_  
_𑀅𑀬𑀁 𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀨𑀦𑁆𑀤𑀦𑁄”𑁇_  


(20.)

2404\. _“𑀇𑀢𑀺 𑀨𑀦𑁆𑀤𑀦𑀭𑀼𑀓𑁆𑀔𑁄𑀧𑀺,_  
_𑀢𑀸𑀯𑀤𑁂 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀣;_  
_𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀅𑀢𑁆𑀣𑀺,_  
_𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


(21.)

2405\. _𑀇𑀲𑁆𑀲𑀲𑁆𑀲 𑀉𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑀫𑁆𑀳𑀸,_  
_𑀉𑀓𑁆𑀓𑀘𑁆𑀘 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀼𑀮𑀁;_  
_𑀢𑁂𑀦 𑀦𑁂𑀫𑀺𑀁 𑀧𑀲𑀸𑀭𑁂𑀲𑀺,_  
_𑀏𑀯𑀁 𑀤𑀍𑀳𑀢𑀭𑀁 𑀲𑀺𑀬𑀸𑁇_  


(22.)

2406\. _𑀇𑀢𑀺 𑀨𑀦𑁆𑀤𑀦𑀭𑀼𑀓𑁆𑀔𑁄𑀧𑀺,_  
_𑀯𑁂𑀭𑀁 𑀅𑀧𑁆𑀧𑁂𑀲𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀚𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀅𑀚𑀸𑀢𑀸𑀦𑀁,_  
_𑀇𑀲𑁆𑀲𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀸𑀯𑀳𑀺𑁇_  


(23.)

2407\. _𑀇𑀘𑁆𑀘𑁂𑀯𑀁 𑀨𑀦𑁆𑀤𑀦𑁄 𑀇𑀲𑁆𑀲𑀁,_  
_𑀇𑀲𑁆𑀲𑁄 𑀘 𑀧𑀦 𑀨𑀦𑁆𑀤𑀦𑀁;_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀯𑀺𑀯𑀸𑀤𑁂𑀦,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑀖𑀸𑀢𑀬𑀼𑀁𑁇_  


(24.)

2408\. _𑀏𑀯𑀫𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀯𑀺𑀯𑀸𑀤𑁄 𑀬𑀢𑁆𑀣 𑀚𑀸𑀬𑀢𑀺;_  
_𑀫𑀬𑀽𑀭𑀦𑀘𑁆𑀘𑀁 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀬𑀣𑀸 𑀢𑁂 𑀇𑀲𑁆𑀲𑀨𑀦𑁆𑀤𑀦𑀸𑁇_  


(25.)

2409\. _𑀢𑀁 𑀯𑁄 𑀯𑀤𑀸𑀫𑀺 𑀪𑀤𑁆𑀤𑀁 𑀯𑁄,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀲𑀫𑁆𑀫𑁄𑀤𑀣 𑀫𑀸 𑀯𑀺𑀯𑀤𑀣,_  
_𑀫𑀸 𑀳𑁄𑀣 𑀇𑀲𑁆𑀲𑀨𑀦𑁆𑀤𑀦𑀸𑁇_  


(26.)

2410\. _𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀫𑁂𑀯 𑀲𑀺𑀓𑁆𑀔𑁂𑀣,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀳𑁂𑀢𑀁 𑀧𑀲𑀁𑀲𑀺𑀢𑀁;_  
_𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀭𑀢𑁄 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀦 𑀥𑀁𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

2411\. 𑀨𑀦𑁆𑀤𑀦𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
