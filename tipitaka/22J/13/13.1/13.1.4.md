# 𑁧𑁩𑁇𑁧𑁇𑁪 𑀘𑀽𑀍𑀅𑀦𑀸𑀭𑀤𑀚𑀸𑀢𑀓

* 𑀢𑁂𑀭𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀫𑁆𑀩𑀯𑀕𑁆𑀕

* 𑀘𑀽𑀍𑀅𑀦𑀸𑀭𑀤𑀚𑀸𑀢𑀓

(40.)

2426\. _“𑀦 𑀢𑁂 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀪𑀺𑀦𑁆𑀦𑀸𑀦𑀺,_  
_𑀦 𑀢𑁂 𑀉𑀤𑀓𑀫𑀸𑀪𑀢𑀁;_  
_𑀅𑀕𑁆𑀕𑀻𑀧𑀺 𑀢𑁂 𑀦 𑀳𑀸𑀧𑀺𑀢𑁄,_  
_𑀓𑀺𑀁 𑀦𑀼 𑀫𑀦𑁆𑀤𑁄𑀯 𑀛𑀸𑀬𑀲𑀺”𑁇_  


(41.)

2427\. _“𑀦 𑀉𑀲𑁆𑀲𑀳𑁂 𑀯𑀦𑁂 𑀯𑀢𑁆𑀣𑀼𑀁,_  
_𑀓𑀲𑁆𑀲𑀧𑀸𑀫𑀦𑁆𑀢𑀬𑀸𑀫𑀺 𑀢𑀁;_  
_𑀤𑀼𑀓𑁆𑀔𑁄 𑀯𑀸𑀲𑁄 𑀅𑀭𑀜𑁆𑀜𑀲𑁆𑀫𑀺𑀁,_  
_𑀭𑀝𑁆𑀞𑀁 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀕𑀦𑁆𑀢𑀯𑁂𑁇_  


(42.)

2428\. _𑀬𑀣𑀸 𑀅𑀳𑀁 𑀇𑀢𑁄 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀚𑀦𑀧𑀤𑁂 𑀯𑀲𑀁;_  
_𑀆𑀘𑀸𑀭𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀁,_  
_𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑀸𑀲 𑀫𑀁”𑁇_  


(43.)

2429\. _“𑀲𑀘𑁂 𑀅𑀭𑀜𑁆𑀜𑀁 𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀦𑀫𑀽𑀮𑀨𑀮𑀸𑀦𑀺 𑀘;_  
_𑀭𑀝𑁆𑀞𑁂 𑀭𑁄𑀘𑀬𑀲𑁂 𑀯𑀸𑀲𑀁,_  
_𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀲𑀸𑀫𑁂𑀳𑀺 𑀫𑁂𑁇_  


(44.)

2430\. _𑀯𑀺𑀲𑀁 𑀫𑀸 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑁆𑀣𑁄,_  
_𑀧𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬;_  
_𑀧𑀗𑁆𑀓𑁂 𑀘 𑀫𑀸 𑀯𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀣𑁄,_  
_𑀬𑀢𑁆𑀢𑁄 𑀘𑀸𑀲𑀻𑀯𑀺𑀲𑁂 𑀘𑀭𑁂”𑁇_  


(45.)

2431\. _“𑀓𑀺𑀁 𑀦𑀼 𑀯𑀺𑀲𑀁 𑀧𑀧𑀸𑀢𑁄 𑀯𑀸,_  
_𑀧𑀗𑁆𑀓𑁄 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀁;_  
_𑀓𑀁 𑀢𑁆𑀯𑀁 𑀆𑀲𑀻𑀯𑀺𑀲𑀁 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄”𑁇_  


(46.)

2432\. _“𑀆𑀲𑀯𑁄 𑀢𑀸𑀢 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀼𑀭𑀸 𑀦𑀸𑀫 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀲𑀼𑀭𑀪𑀻 𑀯𑀕𑁆𑀕𑀼,_  
_𑀲𑀸𑀤𑀼 𑀔𑀼𑀤𑁆𑀤𑀭𑀲𑀽𑀧𑀫𑁄;_  
_𑀯𑀺𑀲𑀁 𑀢𑀤𑀸𑀳𑀼 𑀅𑀭𑀺𑀬𑀸 𑀲𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀸𑀭𑀤𑁇_  


(47.)

2433\. _𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀢𑀸𑀢 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀫𑀢𑁆𑀢𑀁 𑀧𑀫𑀣𑁂𑀦𑁆𑀢𑀺 𑀢𑀸;_  
_𑀳𑀭𑀦𑁆𑀢𑀺 𑀬𑀼𑀯𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀢𑀽𑀮𑀁 𑀪𑀝𑁆𑀞𑀁𑀯 𑀫𑀸𑀮𑀼𑀢𑁄;_  
_𑀧𑀧𑀸𑀢𑁄 𑀏𑀲𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀸𑀭𑀤𑁇_  


(48.)

2434\. _𑀮𑀸𑀪𑁄 𑀲𑀺𑀮𑁄𑀓𑁄 𑀲𑀓𑁆𑀓𑀸𑀭𑁄,_  
_𑀧𑀽𑀚𑀸 𑀧𑀭𑀓𑀼𑀮𑁂𑀲𑀼 𑀘;_  
_𑀧𑀗𑁆𑀓𑁄 𑀏𑀲𑁄 𑀘 𑀅𑀓𑁆𑀔𑀸𑀢𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀸𑀭𑀤𑁇_  


(49.)

2435\. _𑀲𑀲𑀢𑁆𑀣𑀸 𑀢𑀸𑀢 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀆𑀯𑀲𑀦𑁆𑀢𑀺 𑀫𑀳𑀺𑀁 𑀇𑀫𑀁;_  
_𑀢𑁂 𑀢𑀸𑀤𑀺𑀲𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁂,_  
_𑀫𑀳𑀦𑁆𑀢𑁂 𑀢𑀸𑀢 𑀦𑀸𑀭𑀤𑁇_  


(50.)

2436\. _𑀇𑀲𑁆𑀲𑀭𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀻𑀦𑀁,_  
_𑀦 𑀢𑁂𑀲𑀁 𑀧𑀸𑀤𑀢𑁄 𑀘𑀭𑁂;_  
_𑀆𑀲𑀻𑀯𑀺𑀲𑁄𑀢𑀺 𑀅𑀓𑁆𑀔𑀸𑀢𑁄,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀦𑀸𑀭𑀤𑁇_  


(51.)

2437\. _𑀪𑀢𑁆𑀢𑀢𑁆𑀣𑁄 𑀪𑀢𑁆𑀢𑀓𑀸𑀮𑁂 𑀘,_  
_𑀬𑀁 𑀕𑁂𑀳𑀫𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂;_  
_𑀬𑀤𑁂𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀁 𑀚𑀜𑁆𑀜𑀸,_  
_𑀢𑀢𑁆𑀣 𑀖𑀸𑀲𑁂𑀲𑀦𑀁 𑀘𑀭𑁂𑁇_  


(52.)

2438\. _𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀓𑀼𑀮𑀁,_  
_𑀧𑀸𑀦𑀢𑁆𑀣𑀁 𑀪𑁄𑀚𑀦𑀸𑀬 𑀯𑀸;_  
_𑀫𑀺𑀢𑀁 𑀔𑀸𑀤𑁂 𑀫𑀺𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑁂,_  
_𑀦 𑀘 𑀭𑀽𑀧𑁂 𑀫𑀦𑀁 𑀓𑀭𑁂𑁇_  


(53.)

2439\. _𑀕𑁄𑀝𑁆𑀞𑀁 𑀫𑀚𑁆𑀚𑀁 𑀓𑀺𑀭𑀸𑀝𑀜𑁆𑀘,_  
_𑀲𑀪𑀸 𑀦𑀺𑀓𑀺𑀭𑀡𑀸𑀦𑀺 𑀘;_  
_𑀆𑀭𑀓𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀳𑀺,_  
_𑀬𑀸𑀦𑀻𑀯 𑀯𑀺𑀲𑀫𑀁 𑀧𑀣𑀦𑁆”𑀢𑀺𑁇_  


---

2440\. 𑀘𑀽𑀍𑀅𑀦𑀸𑀭𑀤𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
