# 𑁧𑁩𑁇𑁧𑁇𑁧𑁦 𑀲𑀭𑀪𑀫𑀺𑀕𑀚𑀸𑀢𑀓

### 𑀢𑁂𑀭𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀫𑁆𑀩𑀯𑀕𑁆𑀕

### 𑀲𑀭𑀪𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(134.)

2526\. _“𑀆𑀲𑀻𑀲𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀬𑀣𑀸 𑀇𑀘𑁆𑀙𑀺𑀁 𑀢𑀣𑀸 𑀅𑀳𑀼𑁇_  


(135.)

2527\. _𑀆𑀲𑀻𑀲𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀉𑀤𑀓𑀸 𑀣𑀮𑀫𑀼𑀩𑁆𑀪𑀢𑀁𑁇_  


(136.)

2528\. _𑀯𑀸𑀬𑀫𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀬𑀣𑀸 𑀇𑀘𑁆𑀙𑀺𑀁 𑀢𑀣𑀸 𑀅𑀳𑀼𑁇_  


(137.)

2529\. _𑀯𑀸𑀬𑀫𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀉𑀤𑀓𑀸 𑀣𑀮𑀫𑀼𑀩𑁆𑀪𑀢𑀁𑁇_  


(138.)

2530\. _𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀦𑀻𑀢𑁄𑀧𑀺 𑀦𑀭𑁄 𑀲𑀧𑀜𑁆𑀜𑁄,_  
_𑀆𑀲𑀁 𑀦 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀲𑀼𑀔𑀸𑀕𑀫𑀸𑀬;_  
_𑀩𑀳𑀽 𑀳𑀺 𑀨𑀲𑁆𑀲𑀸 𑀅𑀳𑀺𑀢𑀸 𑀳𑀺𑀢𑀸 𑀘,_  
_𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀺𑀢𑀸 𑀫𑀘𑁆𑀘𑀫𑀼𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺𑁇_  


(139.)

2531\. _𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀫𑁆𑀧𑀺 𑀪𑀯𑀢𑀺,_  
_𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀫𑁆𑀧𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺;_  
_𑀦 𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀸 𑀪𑁄𑀕𑀸,_  
_𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀯𑀸”𑁇_  


(140.)

2532\. _“𑀲𑀭𑀪𑀁 𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑀲𑁆𑀫𑀺𑀁,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀅𑀦𑀼𑀲𑀭𑀻 𑀧𑀼𑀭𑁂;_  
_𑀅𑀮𑀻𑀦𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀼𑀯𑀁,_  
_𑀯𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀦𑀼𑀚𑀻𑀯𑀲𑀺𑁇_  


(141.)

2533\. _𑀬𑁄 𑀢𑀁 𑀯𑀺𑀤𑀼𑀕𑁆𑀕𑀸 𑀦𑀭𑀓𑀸 𑀲𑀫𑀼𑀤𑁆𑀥𑀭𑀺,_  
_𑀲𑀺𑀮𑀸𑀬 𑀬𑁄𑀕𑁆𑀕𑀁 𑀲𑀭𑀪𑁄 𑀓𑀭𑀺𑀢𑁆𑀯𑀸;_  
_𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀦𑀻𑀢𑀁 𑀫𑀘𑁆𑀘𑀼𑀫𑀼𑀔𑀸 𑀧𑀫𑁄𑀘𑀬𑀺,_  
_𑀅𑀮𑀻𑀦𑀘𑀺𑀢𑁆𑀢𑀁 𑀢 𑀫𑀺𑀕𑀁 𑀯𑀤𑁂𑀲𑀺”𑁇_  


(142.)

2534\. _“𑀓𑀺𑀁 𑀢𑁆𑀯𑀁 𑀦𑀼 𑀢𑀢𑁆𑀣𑁂𑀯 𑀢𑀤𑀸 𑀅𑀳𑁄𑀲𑀺,_  
_𑀉𑀤𑀸𑀳𑀼 𑀢𑁂 𑀓𑁄𑀘𑀺 𑀦𑀁 𑀏𑀢𑀤𑀓𑁆𑀔𑀸;_  
_𑀯𑀺𑀯𑀝𑁆𑀝𑀘𑁆𑀙𑀤𑁆𑀤𑁄 𑀦𑀼𑀲𑀺 𑀲𑀩𑁆𑀩𑀤𑀲𑁆𑀲𑀻,_  
_𑀜𑀸𑀡𑀁 𑀦𑀼 𑀢𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀪𑀺𑀁𑀲𑀭𑀽𑀧𑀁”𑁇_  


(143.)

2535\. _“𑀦 𑀘𑁂𑀯𑀳𑀁 𑀢𑀢𑁆𑀣 𑀢𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑀁,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀫𑁂 𑀓𑁄𑀘𑀺 𑀦𑀁 𑀏𑀢𑀤𑀓𑁆𑀔𑀸;_  
_𑀕𑀸𑀣𑀸𑀧𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀁,_  
_𑀅𑀢𑁆𑀣𑀁 𑀢𑀤𑀸𑀦𑁂𑀦𑁆𑀢𑀺 𑀚𑀦𑀺𑀦𑁆𑀤 𑀥𑀻𑀭𑀸”𑁇_  


(144.)

2536\. _“𑀆𑀤𑀸𑀬 𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀭𑀯𑀺𑀭𑀺𑀬𑀖𑀸𑀢𑀺𑀁,_  
_𑀘𑀸𑀧𑁂 𑀲𑀭𑀁 𑀓𑀺𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀲𑁂 𑀢𑀼𑀯𑀁;_  
_𑀦𑀼𑀦𑁆𑀦𑁄 𑀲𑀭𑁄 𑀲𑀭𑀪𑀁 𑀳𑀦𑁆𑀢𑀼 𑀔𑀺𑀧𑁆𑀧𑀁,_  
_𑀅𑀦𑁆𑀦𑀜𑁆𑀳𑀺 𑀏𑀢𑀁 𑀯𑀭𑀧𑀜𑁆𑀜 𑀭𑀜𑁆𑀜𑁄”𑁇_  


(145.)

2537\. _“𑀅𑀤𑁆𑀥𑀸 𑀧𑀚𑀸𑀦𑀸𑀫𑀺 𑀅𑀳𑀫𑁆𑀧𑀺 𑀏𑀢𑀁,_  
_𑀅𑀦𑁆𑀦𑀁 𑀫𑀺𑀕𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲;_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀜𑁆𑀘 𑀅𑀧𑀘𑀸𑀬𑀫𑀸𑀦𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀫𑀺𑀕𑀁 𑀲𑀭𑀪𑀁 𑀦𑁄 𑀳𑀦𑀸𑀫𑀺”𑁇_  


(146.)

2538\. _“𑀦𑁂𑀲𑁄 𑀫𑀺𑀕𑁄 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀅𑀲𑀼𑀭𑁂𑀲𑁄 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀏𑀢𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀺𑀦𑁆𑀤,_  
_𑀪𑀯𑀲𑁆𑀲𑀼 𑀅𑀫𑀭𑀸𑀥𑀺𑀧𑁄𑁇_  


(147.)

2539\. _𑀲𑀘𑁂 𑀘 𑀭𑀸𑀚𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀲𑁂 𑀢𑀼𑀯𑀁,_  
_𑀳𑀦𑁆𑀢𑀼𑀁 𑀫𑀺𑀕𑀁 𑀲𑀭𑀪𑀁 𑀲𑀳𑀸𑀬𑀓𑀁;_  
_𑀲𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑁄 𑀦𑀭𑀯𑀻𑀭𑀲𑁂𑀝𑁆𑀞,_  
_𑀕𑀦𑁆𑀢𑀸 𑀢𑀼𑀯𑀁 𑀯𑁂𑀢𑀭𑀡𑀺𑀁 𑀬𑀫𑀲𑁆𑀲”𑁇_  


(148.)

2540\. _“𑀓𑀸𑀫𑀁 𑀅𑀳𑀁 𑀚𑀸𑀦𑀧𑀤𑀸 𑀘 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀤𑀸𑀭𑀸 𑀘 𑀲𑀳𑀸𑀬𑀲𑀗𑁆𑀖𑀸;_  
_𑀕𑀘𑁆𑀙𑁂𑀫𑀼 𑀢𑀁 𑀯𑁂𑀢𑀭𑀡𑀺𑀁 𑀬𑀫𑀲𑁆𑀲,_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀳𑀜𑁆𑀜𑁄 𑀫𑀫 𑀧𑀸𑀡𑀤𑁄 𑀬𑁄𑁇_  


(149.)

2541\. _𑀅𑀬𑀁 𑀫𑀺𑀕𑁄 𑀓𑀺𑀘𑁆𑀙𑀕𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁,_  
_𑀏𑀓𑀲𑁆𑀲 𑀓𑀢𑁆𑀢𑀸 𑀯𑀺𑀯𑀦𑀲𑁆𑀫𑀺 𑀖𑁄𑀭𑁂;_  
_𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑀺𑀘𑁆𑀘𑀁 𑀲𑀭𑀦𑁆𑀢𑁄,_  
_𑀚𑀸𑀦𑀁 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀓𑀣𑀁 𑀳𑀦𑁂𑀬𑁆𑀬𑀁”𑁇_  


(150.)

2542\. _“𑀫𑀺𑀢𑁆𑀢𑀸𑀪𑀺𑀭𑀸𑀥𑀻 𑀘𑀺𑀭𑀫𑁂𑀯 𑀚𑀻𑀯,_  
_𑀭𑀚𑁆𑀚𑀁 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀕𑀼𑀡𑁂 𑀧𑀲𑀸𑀲;_  
_𑀦𑀸𑀭𑀻𑀕𑀡𑁂𑀳𑀺 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀬𑀦𑁆𑀢𑁄,_  
_𑀫𑁄𑀤𑀲𑁆𑀲𑀼 𑀭𑀝𑁆𑀞𑁂 𑀢𑀺𑀤𑀺𑀯𑁂𑀯 𑀯𑀸𑀲𑀯𑁄𑁇_  


(151.)

2543\. _𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀦𑀺𑀘𑁆𑀘𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀸𑀢𑀺𑀣𑀻 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀪𑀯𑀺𑀢𑁆𑀯𑀸;_  
_𑀤𑀢𑁆𑀯𑀸 𑀘 𑀪𑀼𑀢𑁆𑀯𑀸 𑀘 𑀬𑀣𑀸𑀦𑀼𑀪𑀸𑀯𑀁,_  
_𑀅𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑁄 𑀲𑀕𑁆𑀕𑀫𑀼𑀧𑁂𑀳𑀺 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁯𑁩𑁦)_  


---

2544\. 𑀲𑀭𑀪𑀫𑀺𑀕𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇



---

2545\. 𑀢𑁂𑀭𑀲𑀓𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

2546\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2547\. _𑀯𑀭𑀅𑀫𑁆𑀩 𑀓𑀼𑀞𑀸𑀭𑀺 𑀲𑀳𑀁𑀲𑀯𑀭𑁄,_  
_𑀅𑀣𑀭𑀜𑁆𑀜𑀲𑁆𑀫𑀺𑀁 𑀤𑀽𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑀓𑁄;_  
_𑀅𑀣 𑀩𑁄𑀥𑀺 𑀅𑀓𑀺𑀢𑁆𑀢𑀺 𑀲𑀼𑀢𑀓𑁆𑀓𑀭𑀺𑀦𑀸,_  
_𑀅𑀣 𑀭𑀼𑀭𑀼𑀫𑀺𑀕𑁂𑀦𑀧𑀭𑁄 𑀲𑀭𑀪𑁄𑀢𑀺𑁇_
