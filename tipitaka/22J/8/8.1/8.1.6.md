# 𑁮𑁇𑁧𑁇𑁬 𑀘𑁂𑀢𑀺𑀬𑀚𑀸𑀢𑀓

### 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀘𑁆𑀘𑀸𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀘𑁂𑀢𑀺𑀬𑀚𑀸𑀢𑀓

(45.)

1722\. _“𑀥𑀫𑁆𑀫𑁄 𑀳𑀯𑁂 𑀳𑀢𑁄 𑀳𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀳𑀢𑁄 𑀳𑀦𑁆𑀢𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀳𑀦𑁂,_  
_𑀫𑀸 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑁄 𑀳𑀢𑁄 𑀳𑀦𑀺𑁇 (𑁧)_  


(46.)

1723\. _𑀅𑀮𑀺𑀓𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀤𑁂𑀯𑀢𑀸;_  
_𑀧𑀽𑀢𑀺𑀓𑀜𑁆𑀘 𑀫𑀼𑀔𑀁 𑀯𑀸𑀢𑀺,_  
_𑀲𑀓𑀝𑁆𑀞𑀸𑀦𑀸 𑀘 𑀥𑀁𑀲𑀢𑀺;_  
_𑀬𑁄 𑀚𑀸𑀦𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑀁,_  
_𑀅𑀜𑁆𑀜𑀣𑀸 𑀦𑀁 𑀯𑀺𑀬𑀸𑀓𑀭𑁂𑁇_  


(47.)

1724\. _𑀲𑀘𑁂 𑀳𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺,_  
_𑀳𑁄𑀳𑀺 𑀭𑀸𑀚 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀫𑀼𑀲𑀸 𑀘𑁂 𑀪𑀸𑀲𑀲𑁂 𑀭𑀸𑀚,_  
_𑀪𑀽𑀫𑀺𑀬𑀁 𑀢𑀺𑀝𑁆𑀞 𑀘𑁂𑀢𑀺𑀬𑁇 (𑁨)_  


(48.)

1725\. _𑀅𑀓𑀸𑀮𑁂 𑀯𑀲𑁆𑀲𑀢𑀻 𑀢𑀲𑁆𑀲,_  
_𑀓𑀸𑀮𑁂 𑀢𑀲𑁆𑀲 𑀦 𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀬𑁄 𑀚𑀸𑀦𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑀁,_  
_𑀅𑀜𑁆𑀜𑀣𑀸 𑀦𑀁 𑀯𑀺𑀬𑀸𑀓𑀭𑁂𑁇_  


(49.)

1726\. _𑀲𑀘𑁂 𑀳𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺,_  
_𑀳𑁄𑀳𑀺 𑀭𑀸𑀚 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀫𑀼𑀲𑀸 𑀘𑁂 𑀪𑀸𑀲𑀲𑁂 𑀭𑀸𑀚,_  
_𑀪𑀽𑀫𑀺𑀁 𑀧𑀯𑀺𑀲 𑀘𑁂𑀢𑀺𑀬𑁇 (𑁩)_  


(50.)

1727\. _𑀚𑀺𑀯𑁆𑀳𑀸 𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑀺𑀥𑀸 𑀳𑁄𑀢𑀺,_  
_𑀉𑀭𑀕𑀲𑁆𑀲𑁂𑀯 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀬𑁄 𑀚𑀸𑀦𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑀁,_  
_𑀅𑀜𑁆𑀜𑀣𑀸 𑀦𑀁 𑀯𑀺𑀬𑀸𑀓𑀭𑁂𑁇_  


(51.)

1728\. _𑀲𑀘𑁂 𑀳𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺,_  
_𑀳𑁄𑀳𑀺 𑀭𑀸𑀚 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀫𑀼𑀲𑀸 𑀘𑁂 𑀪𑀸𑀲𑀲𑁂 𑀭𑀸𑀚,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀯𑀺𑀲 𑀘𑁂𑀢𑀺𑀬𑁇 (𑁪)_  


(52.)

1729\. _𑀚𑀺𑀯𑁆𑀳𑀸 𑀢𑀲𑁆𑀲 𑀦 𑀪𑀯𑀢𑀺,_  
_𑀫𑀘𑁆𑀙𑀲𑁆𑀲𑁂𑀯 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀬𑁄 𑀚𑀸𑀦𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑀁,_  
_𑀅𑀜𑁆𑀜𑀣𑀸 𑀦𑀁 𑀯𑀺𑀬𑀸𑀓𑀭𑁂𑁇_  


(53.)

1730\. _𑀲𑀘𑁂 𑀳𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺,_  
_𑀳𑁄𑀳𑀺 𑀭𑀸𑀚 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀫𑀼𑀲𑀸 𑀘𑁂 𑀪𑀸𑀲𑀲𑁂 𑀭𑀸𑀚,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀯𑀺𑀲 𑀘𑁂𑀢𑀺𑀬𑁇 (𑁫)_  


(54.)

1731\. _𑀣𑀺𑀬𑁄𑀯 𑀢𑀲𑁆𑀲 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀦 𑀧𑀼𑀫𑀸 𑀚𑀸𑀬𑀭𑁂 𑀓𑀼𑀮𑁂;_  
_𑀬𑁄 𑀚𑀸𑀦𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑀁,_  
_𑀅𑀜𑁆𑀜𑀣𑀸 𑀦𑀁 𑀯𑀺𑀬𑀸𑀓𑀭𑁂𑁇_  


(55.)

1732\. _𑀲𑀘𑁂 𑀳𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺,_  
_𑀳𑁄𑀳𑀺 𑀭𑀸𑀚 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀫𑀼𑀲𑀸 𑀘𑁂 𑀪𑀸𑀲𑀲𑁂 𑀭𑀸𑀚,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀯𑀺𑀲 𑀘𑁂𑀢𑀺𑀬𑁇 (𑁬)_  


(56.)

1733\. _𑀧𑀼𑀢𑁆𑀢𑀸 𑀢𑀲𑁆𑀲 𑀦 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁;_  
_𑀬𑁄 𑀚𑀸𑀦𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑀁,_  
_𑀅𑀜𑁆𑀜𑀣𑀸 𑀦𑀁 𑀯𑀺𑀬𑀸𑀓𑀭𑁂𑁇_  


(57.)

1734\. _𑀲𑀘𑁂 𑀳𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺,_  
_𑀳𑁄𑀳𑀺 𑀭𑀸𑀚 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀫𑀼𑀲𑀸 𑀘𑁂 𑀪𑀸𑀲𑀲𑁂 𑀭𑀸𑀚,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀧𑀯𑀺𑀲 𑀘𑁂𑀢𑀺𑀬”𑁇 (𑁭)_  


(58.)

1735\. _“𑀲 𑀭𑀸𑀚𑀸 𑀇𑀲𑀺𑀦𑀸 𑀲𑀢𑁆𑀢𑁄,_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀘𑀭𑁄 𑀧𑀼𑀭𑁂;_  
_𑀧𑀸𑀯𑁂𑀓𑁆𑀔𑀺 𑀧𑀣𑀯𑀺𑀁 𑀘𑁂𑀘𑁆𑀘𑁄,_  
_𑀳𑀻𑀦𑀢𑁆𑀢𑁄 𑀧𑀢𑁆𑀯 𑀧𑀭𑀺𑀬𑀸𑀬𑀁𑁇_  


(59.)

1736\. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀙𑀦𑁆𑀤𑀸𑀕𑀫𑀦𑀁,_  
_𑀦𑀧𑁆𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀅𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀕𑀺𑀭𑀁 𑀲𑀘𑁆𑀘𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀦𑁆𑀢𑀺”𑁇 (𑁮)_  


---
