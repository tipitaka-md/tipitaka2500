# 𑁮𑁇𑁧𑁇𑁩 𑀲𑀼𑀮𑀲𑀸𑀚𑀸𑀢𑀓

* 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀘𑁆𑀘𑀸𑀦𑀺𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀮𑀲𑀸𑀚𑀸𑀢𑀓

(18.)

1692\. _“𑀇𑀤𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀸𑀬𑀽𑀭𑀁,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀩𑀳𑀽;_  
_𑀲𑀩𑁆𑀩𑀁 𑀳𑀭𑀲𑁆𑀲𑀼 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀫𑀜𑁆𑀘 𑀤𑀸𑀲𑀻𑀢𑀺 𑀲𑀸𑀯𑀬”𑁇_  


(19.)

1693\. _“𑀑𑀭𑁄𑀧𑀬𑀲𑁆𑀲𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀺,_  
_𑀫𑀸 𑀩𑀸𑀍𑀳𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀲𑀺;_  
_𑀦 𑀘𑀸𑀳𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀥𑀦𑀫𑀸𑀪𑀢𑀁”𑁇_  


(20.)

1694\. _“𑀬𑀢𑁄 𑀲𑀭𑀸𑀫𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀬𑀢𑁄 𑀧𑀢𑁆𑀢𑀸𑀲𑁆𑀫𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀢𑀁;_  
_𑀦 𑀘𑀸𑀳𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀢𑀭𑀁 𑀢𑀬𑀸𑁇_  


(21.)

1695\. _𑀏𑀳𑀺 𑀢𑀁 𑀉𑀧𑀕𑀽𑀳𑀺𑀲𑁆𑀲𑀁,_  
_𑀓𑀭𑀺𑀲𑁆𑀲𑀜𑁆𑀘 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀦 𑀳𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦 𑀅𑀢𑁆𑀣𑀺,_  
_𑀫𑀫 𑀢𑀼𑀬𑁆𑀳𑀜𑁆𑀘 𑀲𑀗𑁆𑀕𑀫𑁄”𑁇_  


(22.)

1696\. _“𑀦 𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼,_  
_𑀧𑀼𑀭𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀇𑀢𑁆𑀣𑀻𑀧𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑀸𑁇_  


(23.)

1697\. _𑀦 𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼,_  
_𑀧𑀼𑀭𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀇𑀢𑁆𑀣𑀻𑀧𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀮𑀳𑀼𑀁 𑀅𑀢𑁆𑀣𑀁 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀺𑀓𑀸𑁇_  


(24.)

1698\. _𑀮𑀳𑀼𑀜𑁆𑀘 𑀯𑀢 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀘,_  
_𑀦𑀺𑀓𑀝𑁆𑀞𑁂 𑀲𑀫𑀘𑁂𑀢𑀬𑀺;_  
_𑀫𑀺𑀕𑀁 𑀧𑀼𑀡𑁆𑀡𑀸𑀬𑀢𑁂𑀦𑁂𑀯,_  
_𑀲𑀼𑀮𑀲𑀸 𑀲𑀢𑁆𑀢𑀼𑀓𑀁 𑀯𑀥𑀺𑁇_  


(25.)

1699\. _𑀬𑁄𑀥 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀦 𑀔𑀺𑀧𑁆𑀧𑀫𑀦𑀼𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀲𑁄 𑀳𑀜𑁆𑀜𑀢𑀺 𑀫𑀦𑁆𑀤𑀫𑀢𑀺,_  
_𑀘𑁄𑀭𑁄𑀯 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑁂𑁇_  


(26.)

1700\. _𑀬𑁄 𑀘 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀩𑁄𑀥𑀢𑀺;_  
_𑀫𑀼𑀘𑁆𑀘𑀢𑁂 𑀲𑀢𑁆𑀢𑀼𑀲𑀫𑁆𑀩𑀸𑀥𑀸,_  
_𑀲𑀼𑀮𑀲𑀸 𑀲𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑀺𑀯𑀸”𑀢𑀺𑁇_  


---
