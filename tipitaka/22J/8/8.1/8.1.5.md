# 𑁮𑁇𑁧𑁇𑁫 𑀕𑀗𑁆𑀕𑀫𑀸𑀮𑀚𑀸𑀢𑀓

### 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀘𑁆𑀘𑀸𑀦𑀺𑀯𑀕𑁆𑀕

### 𑀕𑀗𑁆𑀕𑀫𑀸𑀮𑀚𑀸𑀢𑀓

(36.)

1712\. _“𑀅𑀗𑁆𑀕𑀸𑀭𑀚𑀸𑀢𑀸 𑀧𑀣𑀯𑀻,_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀍𑀆𑀦𑀼𑀕𑀢𑀸 𑀫𑀳𑀻;_  
_𑀅𑀣 𑀕𑀸𑀬𑀲𑀺 𑀯𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀦 𑀢𑀁 𑀢𑀧𑀢𑀺 𑀆𑀢𑀧𑁄𑁇_  


(37.)

1713\. _𑀉𑀤𑁆𑀥𑀁 𑀢𑀧𑀢𑀺 𑀆𑀤𑀺𑀘𑁆𑀘𑁄,_  
_𑀅𑀥𑁄 𑀢𑀧𑀢𑀺 𑀯𑀸𑀮𑀼𑀓𑀸;_  
_𑀅𑀣 𑀕𑀸𑀬𑀲𑀺 𑀯𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀦 𑀢𑀁 𑀢𑀧𑀢𑀺 𑀆𑀢𑀧𑁄”𑁇_  


(38.)

1714\. _“𑀦 𑀫𑀁 𑀢𑀧𑀢𑀺 𑀆𑀢𑀧𑁄,_  
_𑀆𑀢𑀧𑀸 𑀢𑀧𑀬𑀦𑁆𑀢𑀺 𑀫𑀁;_  
_𑀅𑀢𑁆𑀣𑀸 𑀳𑀺 𑀯𑀺𑀯𑀺𑀥𑀸 𑀭𑀸𑀚,_  
_𑀢𑁂 𑀢𑀧𑀦𑁆𑀢𑀺 𑀦 𑀆𑀢𑀧𑁄”𑁇_  


(39.)

1715\. _“𑀅𑀤𑁆𑀤𑀲𑀁 𑀓𑀸𑀫 𑀢𑁂 𑀫𑀽𑀮𑀁,_  
_𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀓𑀸𑀫 𑀚𑀸𑀬𑀲𑀺;_  
_𑀦 𑀢𑀁 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀏𑀯𑀁 𑀓𑀸𑀫 𑀦 𑀳𑁂𑀳𑀺𑀲𑀺”𑁇_  


(40.)

1716\. _“𑀅𑀧𑁆𑀧𑀸𑀧𑀺 𑀓𑀸𑀫𑀸 𑀦 𑀅𑀮𑀁,_  
_𑀩𑀳𑀽𑀳𑀺𑀧𑀺 𑀦 𑀢𑀧𑁆𑀧𑀢𑀺;_  
_𑀅𑀳𑀳𑀸 𑀩𑀸𑀮𑀮𑀧𑀦𑀸,_  
_𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀣 𑀚𑀕𑁆𑀕𑀢𑁄”𑁇_  


(41.)

1717\. _“𑀅𑀧𑁆𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀨𑀮𑀁 𑀫𑀫𑁂𑀤𑀁,_  
_𑀉𑀤𑀬𑁄 𑀅𑀚𑁆𑀛𑀸𑀕𑀫𑀸 𑀫𑀳𑀢𑁆𑀢𑀧𑀢𑁆𑀢𑀁;_  
_𑀲𑀼𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑁄 𑀯𑀢 𑀫𑀸𑀡𑀯𑀲𑁆𑀲,_  
_𑀬𑁄 𑀧𑀩𑁆𑀩𑀚𑀻 𑀓𑀸𑀫𑀭𑀸𑀕𑀁 𑀧𑀳𑀸𑀬”𑁇_  


(42.)

1718\. _“𑀢𑀧𑀲𑀸 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁,_  
_𑀢𑀧𑀲𑀸 𑀦𑁆𑀳𑀸𑀧𑀺𑀢𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀪𑀸𑀯𑀁;_  
_𑀢𑀧𑀲𑀸 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀕𑀗𑁆𑀕𑀫𑀸𑀮,_  
_𑀦𑀸𑀫𑁂𑀦𑀸𑀮𑀧𑀲𑀚𑁆𑀚 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀁”𑁇_  


(43.)

1719\. _“𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀫𑁂𑀯 ‘𑀅𑀫𑁆𑀫’ 𑀧𑀲𑁆𑀲𑀣,_  
_𑀔𑀦𑁆𑀢𑀻𑀲𑁄𑀭𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄;_  
_𑀬𑁄 𑀲𑀩𑁆𑀩𑀚𑀦𑀲𑁆𑀲 𑀯𑀦𑁆𑀤𑀺𑀢𑁄𑀳𑀼,_  
_𑀢𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀲𑀭𑀸𑀚𑀺𑀓𑀸 𑀲𑀫𑀘𑁆𑀘𑀸”𑁇_  


(44.)

1720\. _“𑀫𑀸 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀯𑀘𑀼𑀢𑁆𑀣 𑀕𑀗𑁆𑀕𑀫𑀸𑀮𑀁,_  
_𑀫𑀼𑀦𑀺𑀦𑀁 𑀫𑁄𑀦𑀧𑀣𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁;_  
_𑀏𑀲𑁄 𑀳𑀺 𑀅𑀢𑀭𑀺 𑀅𑀡𑁆𑀡𑀯𑀁,_  
_𑀬𑀁 𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀻𑀢𑀲𑁄𑀓𑀸”𑀢𑀺𑁇_  


---

1721\. 𑀕𑀗𑁆𑀕𑀫𑀸𑀮𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
