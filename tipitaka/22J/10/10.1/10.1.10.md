# 𑁧𑁦𑁇𑁧𑁇𑁧𑁦 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀚𑀸𑀢𑀓

### 𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀚𑀸𑀢𑀓

(104.)

2021. _“𑀦𑀸𑀲𑁆𑀫𑀲𑁂 𑀓𑀢𑀧𑀸𑀧𑀫𑁆𑀳𑀺,_  
_𑀦𑀸𑀲𑁆𑀫𑀲𑁂 𑀅𑀮𑀺𑀓𑀯𑀸𑀤𑀺𑀦𑁂;_  
_𑀦𑀸𑀲𑁆𑀫𑀲𑁂 𑀅𑀢𑁆𑀢𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀅𑀢𑀺𑀲𑀦𑁆𑀢𑁂𑀧𑀺 𑀦𑀸𑀲𑁆𑀫𑀲𑁂𑁇_  


(105.)

2022. _𑀪𑀯𑀦𑁆𑀢𑀺 𑀳𑁂𑀓𑁂 𑀧𑀼𑀭𑀺𑀲𑀸,_  
_𑀕𑁄𑀧𑀺𑀧𑀸𑀲𑀺𑀓𑀚𑀸𑀢𑀺𑀓𑀸;_  
_𑀖𑀲𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀯𑀸𑀘𑀸𑀬 𑀦 𑀘 𑀓𑀫𑁆𑀫𑀼𑀦𑀸𑁇_  


(106.)

2023. _𑀲𑀼𑀓𑁆𑀔𑀜𑁆𑀚𑀮𑀺𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀸,_  
_𑀯𑀸𑀘𑀸𑀬 𑀧𑀮𑀺𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀸;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀨𑁂𑀕𑁆𑀕𑀽 𑀦𑀸𑀲𑀻𑀤𑁂,_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀦𑀢𑁆𑀣𑀺 𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑁇_  


(107.)

2024. _𑀦 𑀳𑀺 𑀅𑀜𑁆𑀜𑀜𑁆𑀜𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀇𑀢𑁆𑀣𑀻𑀦𑀁 𑀧𑀼𑀭𑀺𑀲𑀸𑀦 𑀯𑀸;_  
_𑀦𑀸𑀦𑀸𑀯𑀺𑀓𑀢𑁆𑀯𑀸 𑀲𑀁𑀲𑀕𑁆𑀕𑀁,_  
_𑀢𑀸𑀤𑀺𑀲𑀫𑁆𑀧𑀺 𑀘 𑀦𑀸𑀲𑁆𑀫𑀲𑁂𑁇_  


(108.)

2025. _𑀅𑀦𑀭𑀺𑀬𑀓𑀫𑁆𑀫𑀫𑁄𑀓𑁆𑀓𑀦𑁆𑀢𑀁,_  
_𑀅𑀣𑁂𑀢𑀁 𑀲𑀩𑁆𑀩𑀖𑀸𑀢𑀺𑀦𑀁;_  
_𑀦𑀺𑀲𑀺𑀢𑀁𑀯 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀢𑀸𑀤𑀺𑀲𑀫𑁆𑀧𑀺 𑀘 𑀦𑀸𑀲𑁆𑀫𑀲𑁂𑁇_  


(109.)

2026. _𑀫𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑁂𑀦𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂,_  
_𑀲𑀸𑀔𑀮𑁆𑀬𑁂𑀦 𑀅𑀘𑁂𑀢𑀲𑀸;_  
_𑀯𑀺𑀯𑀺𑀥𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀢𑀸𑀤𑀺𑀲𑀫𑁆𑀧𑀺 𑀘 𑀦𑀸𑀲𑁆𑀫𑀲𑁂𑁇_  


(110.)

2027. _𑀆𑀫𑀺𑀲𑀁 𑀯𑀸 𑀥𑀦𑀁 𑀯𑀸𑀧𑀺,_  
_𑀬𑀢𑁆𑀣 𑀧𑀲𑁆𑀲𑀢𑀺 𑀢𑀸𑀤𑀺𑀲𑁄;_  
_𑀤𑀼𑀩𑁆𑀪𑀺𑀁 𑀓𑀭𑁄𑀢𑀺 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑁄,_  
_𑀢𑀜𑁆𑀘 𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀕𑀘𑁆𑀙𑀢𑀺”𑁇_  


(111.)

2028. _“𑀫𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑁂𑀦 𑀩𑀳𑀯𑁄,_  
_𑀙𑀦𑁆𑀦𑀸 𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀯𑁄;_  
_𑀚𑀳𑁂 𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑁂 𑀳𑁂𑀢𑁂,_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀯𑀺𑀬 𑀲𑁂𑀦𑀓𑀁𑁇_  


(112.)

2029. _𑀬𑁄 𑀘 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀦 𑀔𑀺𑀧𑁆𑀧𑀫𑀦𑀼𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀯𑀲𑀫𑀦𑁆𑀯𑁂𑀢𑀺,_  
_𑀧𑀘𑁆𑀙𑀸 𑀘 𑀅𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀺𑁇_  


(113.)

2030. _𑀬𑁄 𑀘 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀩𑁄𑀥𑀢𑀺;_  
_𑀫𑀼𑀘𑁆𑀘𑀢𑁂 𑀲𑀢𑁆𑀢𑀼𑀲𑀫𑁆𑀩𑀸𑀥𑀸,_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀯𑀺𑀬 𑀲𑁂𑀦𑀓𑀸𑁇_  


(114.)

2031. _𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀓𑀽𑀝𑀫𑀺𑀯𑁄𑀟𑁆𑀟𑀺𑀢𑀁 𑀯𑀦𑁂,_  
_𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀦𑀺𑀘𑁆𑀘𑀯𑀺𑀥𑀁𑀲𑀓𑀸𑀭𑀺𑀦𑀁;_  
_𑀆𑀭𑀸 𑀯𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀦𑀭𑁄 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄,_  
_𑀲𑁂𑀦𑀁 𑀬𑀣𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀯𑀁𑀲𑀓𑀸𑀦𑀦𑁂”𑀢𑀺𑁇_  


---

2032. 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇
