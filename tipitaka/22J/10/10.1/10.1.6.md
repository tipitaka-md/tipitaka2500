# 𑁧𑁦𑁇𑁧𑁇𑁬 𑀓𑀡𑁆𑀳𑀤𑀻𑀧𑀸𑀬𑀦𑀚𑀸𑀢𑀓

### 𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀓𑀡𑁆𑀳𑀤𑀻𑀧𑀸𑀬𑀦𑀚𑀸𑀢𑀓

(62.)

1975\. _“𑀲𑀢𑁆𑀢𑀸𑀳𑀫𑁂𑀯𑀸𑀳𑀁 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀧𑀼𑀜𑁆𑀜𑀢𑁆𑀣𑀺𑀓𑁄 𑀆𑀘𑀭𑀺𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁;_  
_𑀅𑀣𑀸𑀧𑀭𑀁 𑀬𑀁 𑀘𑀭𑀺𑀢𑀁 𑀫𑀫𑁂𑀤𑀁,_  
_𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀸𑀲 𑀲𑀫𑀸𑀥𑀺𑀓𑀸𑀦𑀺;_  
_𑀅𑀓𑀸𑀫𑀓𑁄𑀯𑀸𑀧𑀺 𑀅𑀳𑀁 𑀘𑀭𑀸𑀫𑀺,_  
_𑀏𑀢𑁂𑀦 𑀲𑀘𑁆𑀘𑁂𑀦 𑀲𑀼𑀯𑀢𑁆𑀣𑀺 𑀳𑁄𑀢𑀼;_  
_𑀳𑀢𑀁 𑀯𑀺𑀲𑀁 𑀚𑀻𑀯𑀢𑀼 𑀬𑀜𑁆𑀜𑀤𑀢𑁆𑀢𑁄”𑁇_  


(63.)

1976\. _“𑀬𑀲𑁆𑀫𑀸 𑀤𑀸𑀦𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀁 𑀓𑀤𑀸𑀘𑀺,_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀳𑀁 𑀅𑀢𑀺𑀣𑀺𑀁 𑀯𑀸𑀲𑀓𑀸𑀮𑁂;_  
_𑀦 𑀘𑀸𑀧𑀺 𑀫𑁂 𑀅𑀧𑁆𑀧𑀺𑀬𑀢𑀁 𑀅𑀯𑁂𑀤𑀼𑀁,_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀘”𑁇_  
_𑀅𑀓𑀸𑀫𑀓𑁄𑀯𑀸𑀧𑀺 𑀅𑀳𑀁 𑀤𑀤𑀸𑀫𑀺,_  
_𑀏𑀢𑁂𑀦 𑀲𑀘𑁆𑀘𑁂𑀦 𑀲𑀼𑀯𑀢𑁆𑀣𑀺 𑀳𑁄𑀢𑀼;_  
_𑀳𑀢𑀁 𑀯𑀺𑀲𑀁 𑀚𑀻𑀯𑀢𑀼 𑀬𑀜𑁆𑀜𑀤𑀢𑁆𑀢𑁄𑁇_  


(64.)

1977\. _“𑀆𑀲𑀻𑀯𑀺𑀲𑁄 𑀢𑀸𑀢 𑀧𑀳𑀽𑀢𑀢𑁂𑀚𑁄,_  
_𑀬𑁄 𑀢𑀁 𑀅𑀤𑀁𑀲𑀻 𑀲𑀘𑀭𑀸 𑀉𑀤𑀺𑀘𑁆𑀘;_  
_𑀢𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀫𑁂 𑀅𑀧𑁆𑀧𑀺𑀬𑀢𑀸𑀬 𑀅𑀚𑁆𑀚,_  
_𑀧𑀺𑀢𑀭𑀺 𑀘 𑀢𑁂 𑀦𑀢𑁆𑀣𑀺 𑀓𑁄𑀘𑀺 𑀯𑀺𑀲𑁂𑀲𑁄;_  
_𑀏𑀢𑁂𑀦 𑀲𑀘𑁆𑀘𑁂𑀦 𑀲𑀼𑀯𑀢𑁆𑀣𑀺 𑀳𑁄𑀢𑀼,_  
_𑀳𑀢𑀁 𑀯𑀺𑀲𑀁 𑀚𑀻𑀯𑀢𑀼 𑀬𑀜𑁆𑀜𑀤𑀢𑁆𑀢𑁄”𑁇_  


(65.)

1978\. _“𑀲𑀦𑁆𑀢𑀸 𑀤𑀦𑁆𑀢𑀸𑀬𑁂𑀯 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺,_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀡𑁆𑀳𑀸 𑀦𑀢𑁆𑀣𑀸𑀓𑀸𑀫𑀭𑀽𑀧𑀸;_  
_𑀤𑀻𑀧𑀸𑀬𑀦 𑀓𑀺𑀲𑁆𑀲 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄,_  
_𑀅𑀓𑀸𑀫𑀓𑁄 𑀘𑀭𑀲𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁”𑁇_  


(66.)

1979\. _“𑀲𑀤𑁆𑀥𑀸𑀬 𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫 𑀧𑀼𑀦𑀁 𑀦𑀺𑀯𑀢𑁆𑀢𑁄,_  
_𑀲𑁄 𑀏𑀍𑀅𑀫𑀽𑀕𑁄𑀯 𑀩𑀸𑀮𑁄 𑀯𑀢𑀸𑀬𑀁;_  
_𑀏𑀢𑀲𑁆𑀲 𑀯𑀸𑀤𑀲𑁆𑀲 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄,_  
_𑀅𑀓𑀸𑀫𑀓𑁄 𑀘𑀭𑀸𑀫𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁;_  
_𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑀜𑁆𑀘 𑀲𑀢𑀜𑁆𑀘 𑀞𑀸𑀦𑀁,_  
_𑀏𑀯𑀫𑁆𑀧𑀳𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑀭𑁄 𑀪𑀯𑀸𑀫𑀺”𑁇_  


(67.)

1980\. _“𑀲𑀫𑀡𑁂 𑀢𑀼𑀯𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀅𑀤𑁆𑀥𑀺𑀓𑁂 𑀘,_  
_𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀬𑀸𑀲𑀺 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀁;_  
_𑀑𑀧𑀸𑀦𑀪𑀽𑀢𑀁𑀯 𑀖𑀭𑀁 𑀢𑀯 𑀬𑀺𑀤𑀁,_  
_𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦 𑀉𑀧𑁂𑀢𑀭𑀽𑀧𑀁;_  
_𑀅𑀣 𑀓𑀺𑀲𑁆𑀲 𑀯𑀸𑀤𑀲𑁆𑀲 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄,_  
_𑀅𑀓𑀸𑀫𑀓𑁄 𑀤𑀸𑀦𑀫𑀺𑀫𑀁 𑀤𑀤𑀸𑀲𑀺”𑁇_  


(68.)

1981\. _“𑀧𑀺𑀢𑀭𑁄 𑀘 𑀫𑁂 𑀆𑀲𑀼𑀁 𑀧𑀺𑀢𑀸𑀫𑀳𑀸 𑀘,_  
_𑀲𑀤𑁆𑀥𑀸 𑀅𑀳𑀼𑀁 𑀤𑀸𑀦𑀧𑀢𑀻 𑀯𑀤𑀜𑁆𑀜𑀽;_  
_𑀢𑀁 𑀓𑀼𑀮𑁆𑀮𑀯𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄,_  
_𑀫𑀸𑀳𑀁 𑀓𑀼𑀮𑁂 𑀅𑀦𑁆𑀢𑀺𑀫𑀕𑀦𑁆𑀥𑀦𑁄 𑀅𑀳𑀼𑀁;_  
_𑀏𑀢𑀲𑁆𑀲 𑀯𑀸𑀤𑀲𑁆𑀲 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄,_  
_𑀅𑀓𑀸𑀫𑀓𑁄 𑀤𑀸𑀦𑀫𑀺𑀫𑀁 𑀤𑀤𑀸𑀫𑀺”𑁇_  


(69.)

1982\. _“𑀤𑀳𑀭𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀁 𑀅𑀲𑀫𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀁,_  
_𑀬𑀁 𑀢𑀸𑀦𑀬𑀺𑀁 𑀜𑀸𑀢𑀺𑀓𑀼𑀮𑀸 𑀲𑀼𑀕𑀢𑁆𑀢𑁂;_  
_𑀦 𑀘𑀸𑀧𑀺 𑀫𑁂 𑀅𑀧𑁆𑀧𑀺𑀬𑀢𑀁 𑀅𑀯𑁂𑀤𑀺,_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀓𑀸𑀫𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀦𑁆𑀢𑀸;_  
_𑀅𑀣 𑀓𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦 𑀫𑀬𑀸 𑀢𑁂 𑀪𑁄𑀢𑀺,_  
_𑀲𑀁𑀯𑀸𑀲𑀥𑀫𑁆𑀫𑁄 𑀅𑀳𑀼 𑀏𑀯𑀭𑀽𑀧𑁄”𑁇_  


(70.)

1983\. _“𑀆𑀭𑀸 𑀤𑀽𑀭𑁂 𑀦𑀬𑀺𑀥 𑀓𑀤𑀸𑀘𑀺 𑀅𑀢𑁆𑀣𑀺,_  
_𑀧𑀭𑀫𑁆𑀧𑀭𑀸 𑀦𑀸𑀫 𑀓𑀼𑀮𑁂 𑀇𑀫𑀲𑁆𑀫𑀺𑀁;_  
_𑀢𑀁 𑀓𑀼𑀮𑁆𑀮𑀯𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀸,_  
_𑀫𑀸𑀳𑀁 𑀓𑀼𑀮𑁂 𑀅𑀦𑁆𑀢𑀺𑀫𑀕𑀦𑁆𑀥𑀺𑀦𑀻 𑀅𑀳𑀼𑀁;_  
_𑀏𑀢𑀲𑁆𑀲 𑀯𑀸𑀤𑀲𑁆𑀲 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑀸,_  
_𑀅𑀓𑀸𑀫𑀺𑀓𑀸 𑀧𑀤𑁆𑀥𑀘𑀭𑀸𑀫𑁆𑀳𑀺 𑀢𑀼𑀬𑁆𑀳𑀁”𑁇_  


(71.)

1984\. _“𑀫𑀡𑁆𑀟𑀩𑁆𑀬 𑀪𑀸𑀲𑀺𑀁 𑀬𑀫𑀪𑀸𑀲𑀦𑁂𑀬𑁆𑀬𑀁,_  
_𑀢𑀁 𑀔𑀫𑁆𑀬𑀢𑀁 𑀧𑀼𑀢𑁆𑀢𑀓𑀳𑁂𑀢𑀼 𑀫𑀚𑁆𑀚;_  
_𑀧𑀼𑀢𑁆𑀢𑀧𑁂𑀫𑀸 𑀦 𑀇𑀥 𑀧𑀭𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀲𑁄 𑀦𑁄 𑀅𑀬𑀁 𑀚𑀻𑀯𑀢𑀺 𑀬𑀜𑁆𑀜𑀤𑀢𑁆𑀢𑁄”𑀢𑀺𑁇_  


---

1985\. 𑀓𑀡𑁆𑀳𑀤𑀻𑀧𑀸𑀬𑀦𑀚𑀸𑀢𑀓𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
