# 𑁧𑁦𑁇𑁧𑁇𑁯 𑀫𑀳𑀸𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑀚𑀸𑀢𑀓

* 𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀘𑀢𑀼𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑀚𑀸𑀢𑀓

(92.)

2008\. _“𑀓𑀺𑀁 𑀢𑁂 𑀯𑀢𑀁 𑀓𑀺𑀁 𑀧𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀓𑀺𑀲𑁆𑀲 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄;_  
_𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀫𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀏𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀓𑀲𑁆𑀫𑀸 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂”𑁇_  


(93.)

2009\. _“𑀥𑀫𑁆𑀫𑀁 𑀘𑀭𑀸𑀫 𑀦 𑀫𑀼𑀲𑀸 𑀪𑀡𑀸𑀫,_  
_𑀧𑀸𑀧𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀸𑀫;_  
_𑀅𑀦𑀭𑀺𑀬𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀫𑀼 𑀲𑀩𑁆𑀩𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(94.)

2010\. _𑀲𑀼𑀡𑁄𑀫 𑀥𑀫𑁆𑀫𑀁 𑀅𑀲𑀢𑀁 𑀲𑀢𑀜𑁆𑀘,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀅𑀲𑀢𑀁 𑀭𑁄𑀘𑀬𑀸𑀫;_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑀦𑁆𑀢𑁂 𑀦 𑀚𑀳𑀸𑀫 𑀲𑀦𑁆𑀢𑁂,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(95.)

2011\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀤𑀸𑀦𑀸 𑀲𑀼𑀫𑀦𑀸 𑀪𑀯𑀸𑀫,_  
_𑀤𑀤𑀫𑁆𑀧𑀺 𑀯𑁂 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀪𑀯𑀸𑀫;_  
_𑀤𑀢𑁆𑀯𑀸𑀧𑀺 𑀯𑁂 𑀦𑀸𑀦𑀼𑀢𑀧𑁆𑀧𑀸𑀫 𑀧𑀘𑁆𑀙𑀸,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(96.)

2012\. _𑀲𑀫𑀡𑁂 𑀫𑀬𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀅𑀤𑁆𑀥𑀺𑀓𑁂 𑀘,_  
_𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁂 𑀬𑀸𑀘𑀦𑀓𑁂 𑀤𑀮𑀺𑀤𑁆𑀤𑁂;_  
_𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦 𑀅𑀪𑀺𑀢𑀧𑁆𑀧𑀬𑀸𑀫,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(97.)

2013\. _𑀫𑀬𑀜𑁆𑀘 𑀪𑀭𑀺𑀬𑀁 𑀦𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀫,_  
_𑀅𑀫𑁆𑀳𑁂 𑀘 𑀪𑀭𑀺𑀬𑀸 𑀦𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺;_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀸𑀳𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(98.)

2014\. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀫𑀸𑀫 𑀲𑀩𑁆𑀩𑁂,_  
_𑀮𑁄𑀓𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀸𑀫;_  
_𑀅𑀫𑀚𑁆𑀚𑀧𑀸 𑀦𑁄𑀧𑀺 𑀫𑀼𑀲𑀸 𑀪𑀡𑀸𑀫,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(99.)

2015\. _𑀏𑀢𑀸𑀲𑀼 𑀯𑁂 𑀚𑀸𑀬𑀭𑁂 𑀲𑀼𑀢𑁆𑀢𑀫𑀸𑀲𑀼,_  
_𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀳𑀽𑀢𑀧𑀜𑁆𑀜𑀸;_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀯𑁂𑀤𑀕𑀼𑀦𑁄 𑀘 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(100.)

2016\. _𑀫𑀸𑀢𑀸 𑀧𑀺𑀢𑀸 𑀘 𑀪𑀕𑀺𑀦𑀻 𑀪𑀸𑀢𑀭𑁄 𑀘,_  
_𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀤𑀸𑀭𑀸 𑀘 𑀫𑀬𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑁂;_  
_𑀥𑀫𑁆𑀫𑀁 𑀘𑀭𑀸𑀫 𑀧𑀭𑀮𑁄𑀓𑀳𑁂𑀢𑀼,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂𑁇_  


(101.)

2017\. _𑀤𑀸𑀲𑀸 𑀘 𑀤𑀸𑀲𑁆𑀬𑁄 𑀅𑀦𑀼𑀚𑀻𑀯𑀺𑀦𑁄 𑀘,_  
_𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑀸 𑀓𑀫𑁆𑀫𑀓𑀭𑀸 𑀘 𑀲𑀩𑁆𑀩𑁂;_  
_𑀥𑀫𑁆𑀫𑀁 𑀘𑀭𑀦𑁆𑀢𑀺 𑀧𑀭𑀮𑁄𑀓𑀳𑁂𑀢𑀼,_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀫𑁆𑀳𑀁 𑀤𑀳𑀭𑀸 𑀦 𑀫𑀺𑀬𑁆𑀬𑀭𑁂”𑁇_  


(102.)

2018\. _“𑀥𑀫𑁆𑀫𑁄 𑀳𑀯𑁂 𑀭𑀓𑁆𑀔𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁄 𑀲𑀼𑀔𑀫𑀸𑀯𑀳𑀸𑀢𑀺;_  
_𑀏𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑁄 𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑁂,_  
_𑀦 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻𑁇_  


(103.)

2019\. _𑀥𑀫𑁆𑀫𑁄 𑀳𑀯𑁂 𑀭𑀓𑁆𑀔𑀢𑀺 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀁,_  
_𑀙𑀢𑁆𑀢𑀁 𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀺𑀬 𑀯𑀲𑁆𑀲𑀓𑀸𑀮𑁂;_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀕𑀼𑀢𑁆𑀢𑁄 𑀫𑀫 𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑁄,_  
_𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀻𑀦𑀺 𑀲𑀼𑀔𑀻 𑀓𑀼𑀫𑀸𑀭𑁄”𑀢𑀺𑁇_  


---

2020\. 𑀫𑀳𑀸𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇
