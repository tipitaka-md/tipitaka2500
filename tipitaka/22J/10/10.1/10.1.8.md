# 𑁧𑁦𑁇𑁧𑁇𑁮 𑀢𑀓𑁆𑀓𑀮𑀚𑀸𑀢𑀓

### 𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀢𑀼𑀤𑁆𑀯𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀢𑀓𑁆𑀓𑀮𑀚𑀸𑀢𑀓

(82.)

1997\. _“𑀦 𑀢𑀓𑁆𑀓𑀮𑀸 𑀲𑀦𑁆𑀢𑀺 𑀦 𑀆𑀮𑀼𑀯𑀸𑀦𑀺,_  
_𑀦 𑀩𑀺𑀍𑀆𑀮𑀺𑀬𑁄 𑀦 𑀓𑀍𑀅𑀫𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀢;_  
_𑀏𑀓𑁄 𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺 𑀲𑀼𑀲𑀸𑀦𑀫𑀚𑁆𑀛𑁂,_  
_𑀓𑀺𑀫𑀢𑁆𑀣𑀺𑀓𑁄 𑀢𑀸𑀢 𑀔𑀡𑀸𑀲𑀺 𑀓𑀸𑀲𑀼𑀁”𑁇_  


(83.)

1998\. _“𑀧𑀺𑀢𑀸𑀫𑀳𑁄 𑀢𑀸𑀢 𑀲𑀼𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀢𑁂,_  
_𑀅𑀦𑁂𑀓𑀩𑁆𑀬𑀸𑀥𑀻𑀳𑀺 𑀤𑀼𑀔𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑁄;_  
_𑀢𑀫𑀚𑁆𑀚𑀳𑀁 𑀦𑀺𑀔𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑁄𑀩𑁆𑀪𑁂,_  
_𑀦 𑀳𑀺𑀲𑁆𑀲 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀭𑁄𑀘𑀬𑀸𑀫𑀺”𑁇_  


(84.)

1999\. _“𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀮𑀤𑁆𑀥 𑀧𑀸𑀧𑀓𑀁,_  
_𑀅𑀘𑁆𑀘𑀸𑀳𑀺𑀢𑀁 𑀓𑀫𑁆𑀫 𑀓𑀭𑁄𑀲𑀺 𑀮𑀼𑀤𑁆𑀤𑀁;_  
_𑀫𑀬𑀸𑀧𑀺 𑀢𑀸𑀢 𑀧𑀝𑀺𑀮𑀘𑁆𑀙𑀲𑁂 𑀢𑀼𑀯𑀁,_  
_𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀓𑀫𑁆𑀫 𑀚𑀭𑀽𑀧𑀦𑀻𑀢𑁄;_  
_𑀢𑀁 𑀓𑀼𑀮𑁆𑀮𑀯𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄,_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀁 𑀦𑀺𑀔𑀡𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑁄𑀩𑁆𑀪𑁂”𑁇_  


(85.)

2000\. _“𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀧𑀓𑀼𑀩𑁆𑀩𑀫𑀸𑀦𑁄,_  
_𑀆𑀲𑀚𑁆𑀚 𑀫𑀁 𑀢𑁆𑀯𑀁 𑀯𑀤𑀲𑁂 𑀓𑀼𑀫𑀸𑀭;_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀫𑀁 𑀑𑀭𑀲𑀓𑁄 𑀲𑀫𑀸𑀦𑁄,_  
_𑀅𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻 𑀫𑀫 𑀢𑁆𑀯𑀁𑀲𑀺 𑀧𑀼𑀢𑁆𑀢”𑁇_  


(86.)

2001\. _“𑀦 𑀢𑀸𑀳𑀁 𑀢𑀸𑀢 𑀅𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻,_  
_𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻 𑀢𑁂 𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑀸𑀢;_  
_𑀧𑀸𑀧𑀜𑁆𑀘 𑀢𑀁 𑀓𑀫𑁆𑀫 𑀧𑀓𑀼𑀩𑁆𑀩𑀫𑀸𑀦𑀁,_  
_𑀅𑀭𑀳𑀸𑀫𑀺 𑀦𑁄 𑀯𑀸𑀭𑀬𑀺𑀢𑀼𑀁 𑀢𑀢𑁄”𑁇_  


(87.)

2002\. _“𑀬𑁄 𑀫𑀸𑀢𑀭𑀁 𑀯𑀸 𑀧𑀺𑀢𑀭𑀁 𑀲𑀯𑀺𑀝𑁆𑀞,_  
_𑀅𑀤𑀽𑀲𑀓𑁂 𑀳𑀺𑀁𑀲𑀢𑀺 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀁,_  
_𑀅𑀲𑀁𑀲𑀬𑀁 𑀲𑁄 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑁂𑀢𑀺𑁇_  


(88.)

2003\. _𑀬𑁄 𑀫𑀸𑀢𑀭𑀁 𑀯𑀸 𑀧𑀺𑀢𑀭𑀁 𑀲𑀯𑀺𑀝𑁆𑀞,_  
_𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀳𑀸𑀢𑀺;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀁,_  
_𑀅𑀲𑀁𑀲𑀬𑀁 𑀲𑁄 𑀲𑀼𑀕𑀢𑀺𑀁 𑀉𑀧𑁂𑀢𑀺𑁇_  


(89.)

2004\. _𑀦 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀧𑀼𑀢𑁆𑀢 𑀅𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻,_  
_𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻 𑀫𑁂 𑀢𑁆𑀯𑀁𑀲𑀺 𑀧𑀼𑀢𑁆𑀢;_  
_𑀅𑀳𑀜𑁆𑀘 𑀢𑀁 𑀫𑀸𑀢𑀭𑀸 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄,_  
_𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀓𑀫𑁆𑀫 𑀓𑀭𑁄𑀫𑀺 𑀮𑀼𑀤𑁆𑀤𑀁”𑁇_  


(90.)

2005\. _“𑀬𑀸 𑀢𑁂 𑀲𑀸 𑀪𑀭𑀺𑀬𑀸 𑀅𑀦𑀭𑀺𑀬𑀭𑀽𑀧𑀸,_  
_𑀫𑀸𑀢𑀸 𑀫𑀫𑁂𑀲𑀸 𑀲𑀓𑀺𑀬𑀸 𑀚𑀦𑁂𑀢𑁆𑀢𑀺;_  
_𑀦𑀺𑀤𑁆𑀥𑀸𑀧𑀬𑁂 𑀢𑀜𑁆𑀘 𑀲𑀓𑀸 𑀅𑀕𑀸𑀭𑀸,_  
_𑀅𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀢𑁂 𑀲𑀸 𑀤𑀼𑀔𑀫𑀸𑀯𑀳𑁂𑀬𑁆𑀬𑁇_  


(91.)

2006\. _𑀬𑀸 𑀢𑁂 𑀲𑀸 𑀪𑀭𑀺𑀬𑀸 𑀅𑀦𑀭𑀺𑀬𑀭𑀽𑀧𑀸,_  
_𑀫𑀸𑀢𑀸 𑀫𑀫𑁂𑀲𑀸 𑀲𑀓𑀺𑀬𑀸 𑀚𑀦𑁂𑀢𑁆𑀢𑀺;_  
_𑀤𑀦𑁆𑀢𑀸 𑀓𑀭𑁂𑀡𑀽𑀯 𑀯𑀲𑀽𑀧𑀦𑀻𑀢𑀸,_  
_𑀲𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀧𑀼𑀦𑀭𑀸𑀯𑀚𑀸𑀢𑀽”𑀢𑀺𑁇_  


---

2007\. 𑀢𑀓𑁆𑀓𑀮𑀚𑀸𑀢𑀓𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
