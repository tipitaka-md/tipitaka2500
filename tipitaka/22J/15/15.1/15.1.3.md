# 𑁧𑁫𑁇𑁧𑁇𑁩 𑀲𑀺𑀯𑀺𑀚𑀸𑀢𑀓

### 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀸𑀢𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀺𑀯𑀺𑀚𑀸𑀢𑀓

(52.)

2906. _“𑀤𑀽𑀭𑁂 𑀅𑀧𑀲𑁆𑀲𑀁 𑀣𑁂𑀭𑁄𑀯,_  
_𑀘𑀓𑁆𑀔𑀼𑀁 𑀬𑀸𑀘𑀺𑀢𑀼𑀫𑀸𑀕𑀢𑁄;_  
_𑀏𑀓𑀦𑁂𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀘𑀓𑁆𑀔𑀼𑀁 𑀫𑁂 𑀤𑁂𑀳𑀺 𑀬𑀸𑀘𑀺𑀢𑁄”𑁇_  


(53.)

2907. _“𑀓𑁂𑀦𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑁄 𑀇𑀥 𑀫𑀸𑀕𑀢𑁄𑀲𑀺,_  
_𑀯𑀦𑀺𑀩𑁆𑀩𑀓 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑀸𑀦𑀺 𑀬𑀸𑀘𑀺𑀢𑀼𑀁;_  
_𑀲𑀼𑀤𑀼𑀘𑁆𑀘𑀚𑀁 𑀬𑀸𑀘𑀲𑀺 𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑀁,_  
_𑀬𑀫𑀸𑀳𑀼 𑀦𑁂𑀢𑁆𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀤𑀼𑀘𑁆𑀘𑀚𑀁”𑁇_  


(54.)

2908. _“𑀬𑀫𑀸𑀳𑀼 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀻𑀢𑀺,_  
_𑀫𑀖𑀯𑀸𑀢𑀺 𑀦𑀁 𑀆𑀳𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂;_  
_𑀢𑁂𑀦𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑁄 𑀇𑀥 𑀫𑀸𑀕𑀢𑁄𑀲𑁆𑀫𑀺,_  
_𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁄 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑀸𑀦𑀺 𑀬𑀸𑀘𑀺𑀢𑀼𑀁𑁇_  


(55.)

2909. _𑀯𑀦𑀺𑀩𑁆𑀩𑀢𑁄 𑀫𑀬𑁆𑀳 𑀯𑀦𑀺𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁,_  
_𑀤𑀤𑀸𑀳𑀺 𑀢𑁂 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑀸𑀦𑀺 𑀬𑀸𑀘𑀺𑀢𑁄;_  
_𑀤𑀤𑀸𑀳𑀺 𑀫𑁂 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁,_  
_𑀬𑀫𑀸𑀳𑀼 𑀦𑁂𑀢𑁆𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀤𑀼𑀘𑁆𑀘𑀚𑀁”𑁇_  


(56.)

2910. _“𑀬𑁂𑀦 𑀅𑀢𑁆𑀣𑁂𑀦 𑀆𑀕𑀘𑁆𑀙𑀺,_  
_𑀬𑀫𑀢𑁆𑀣𑀫𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀁;_  
_𑀢𑁂 𑀢𑁂 𑀇𑀚𑁆𑀛𑀦𑁆𑀢𑀼 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸,_  
_𑀮𑀪 𑀘𑀓𑁆𑀔𑀽𑀦𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁇_  


(57.)

2911. _𑀏𑀓𑀁 𑀢𑁂 𑀬𑀸𑀘𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀉𑀪𑀬𑀸𑀦𑀺 𑀤𑀤𑀸𑀫𑀳𑀁;_  
_𑀲 𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀕𑀘𑁆𑀙 𑀚𑀦𑀲𑁆𑀲 𑀧𑁂𑀓𑁆𑀔𑀢𑁄,_  
_𑀬𑀤𑀺𑀘𑁆𑀙𑀲𑁂 𑀢𑁆𑀯𑀁 𑀢𑀤𑀢𑁂 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀼”𑁇_  


(58.)

2912. _“𑀫𑀸 𑀦𑁄 𑀤𑁂𑀯 𑀅𑀤𑀸 𑀘𑀓𑁆𑀔𑀼𑀁,_  
_𑀫𑀸 𑀦𑁄 𑀲𑀩𑁆𑀩𑁂 𑀧𑀭𑀸𑀓𑀭𑀺;_  
_𑀥𑀦𑀁 𑀤𑁂𑀳𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀩𑀳𑀽𑁇_  


(59.)

2913. _𑀬𑀼𑀢𑁆𑀢𑁂 𑀤𑁂𑀯 𑀭𑀣𑁂 𑀤𑁂𑀳𑀺,_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑁂 𑀘𑀮𑀗𑁆𑀓𑀢𑁂;_  
_𑀦𑀸𑀕𑁂 𑀤𑁂𑀳𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀳𑁂𑀫𑀓𑀧𑁆𑀧𑀦𑀯𑀸𑀲𑀲𑁂𑁇_  


(60.)

2914. _𑀬𑀣𑀸 𑀢𑀁 𑀲𑀺𑀯𑀬𑁄 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀬𑁄𑀕𑁆𑀕𑀸 𑀲𑀭𑀣𑀸 𑀲𑀤𑀸;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀺𑀭𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀏𑀯𑀁 𑀤𑁂𑀳𑀺 𑀭𑀣𑁂𑀲𑀪”𑁇_  


(61.)

2915. _“𑀬𑁄 𑀯𑁂 𑀤𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀤𑀸𑀦𑁂 𑀓𑀼𑀭𑀼𑀢𑁂 𑀫𑀦𑁄;_  
_𑀪𑀽𑀫𑁆𑀬𑀁 𑀲𑁄 𑀧𑀢𑀺𑀢𑀁 𑀧𑀸𑀲𑀁,_  
_𑀕𑀻𑀯𑀸𑀬𑀁 𑀧𑀝𑀺𑀫𑀼𑀜𑁆𑀘𑀢𑀺𑁇_  


(62.)

2916. _𑀬𑁄 𑀯𑁂 𑀤𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀤𑀸𑀦𑁂 𑀓𑀼𑀭𑀼𑀢𑁂 𑀫𑀦𑁄;_  
_𑀧𑀸𑀧𑀸 𑀧𑀸𑀧𑀢𑀭𑁄 𑀳𑁄𑀢𑀺,_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁄 𑀬𑀫𑀲𑀸𑀥𑀦𑀁𑁇_  


(63.)

2917. _𑀬𑀜𑁆𑀳𑀺 𑀬𑀸𑀘𑁂 𑀢𑀜𑁆𑀳𑀺 𑀤𑀤𑁂,_  
_𑀬𑀁 𑀦 𑀬𑀸𑀘𑁂 𑀦 𑀢𑀁 𑀤𑀤𑁂;_  
_𑀲𑁆𑀯𑀸𑀳𑀁 𑀢𑀫𑁂𑀯 𑀤𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀁 𑀫𑀁 𑀬𑀸𑀘𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄”𑁇_  


(64.)

2918. _“𑀆𑀬𑀼𑀁 𑀦𑀼 𑀯𑀡𑁆𑀡𑀁 𑀦𑀼 𑀲𑀼𑀔𑀁 𑀩𑀮𑀁 𑀦𑀼,_  
_𑀓𑀺𑀁 𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄 𑀦𑀼 𑀚𑀦𑀺𑀦𑁆𑀤 𑀤𑁂𑀲𑀺;_  
_𑀓𑀣𑀜𑁆𑀳𑀺 𑀭𑀸𑀚𑀸 𑀲𑀺𑀯𑀺𑀦𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄,_  
_𑀘𑀓𑁆𑀔𑀽𑀦𑀺 𑀤𑀚𑁆𑀚𑀸 𑀧𑀭𑀮𑁄𑀓𑀳𑁂𑀢𑀼”𑁇_  


(65.)

2919. _“𑀦 𑀯𑀸𑀳𑀫𑁂𑀢𑀁 𑀬𑀲𑀲𑀸 𑀤𑀤𑀸𑀫𑀺,_  
_𑀦 𑀧𑀼𑀢𑁆𑀢𑀫𑀺𑀘𑁆𑀙𑁂 𑀦 𑀥𑀦𑀁 𑀦 𑀭𑀝𑁆𑀞𑀁;_  
_𑀲𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁄 𑀘𑀭𑀺𑀢𑁄 𑀧𑀼𑀭𑀸𑀡𑁄,_  
_𑀇𑀘𑁆𑀘𑁂𑀯 𑀤𑀸𑀦𑁂 𑀭𑀫𑀢𑁂 𑀫𑀦𑁄 𑀫𑀫”𑁇_  
_( )_  


(66.)

2920. _“𑀲𑀔𑀸 𑀘 𑀫𑀺𑀢𑁆𑀢𑁄 𑀘 𑀫𑀫𑀸𑀲𑀺 𑀲𑀻𑀯𑀺𑀓,_  
_𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄 𑀲𑀸𑀥𑀼 𑀓𑀭𑁄𑀳𑀺 𑀫𑁂 𑀯𑀘𑁄;_  
_𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸 𑀘𑀓𑁆𑀔𑀽𑀦𑀺 𑀫𑀫𑀁 𑀚𑀺𑀕𑀻𑀲𑀢𑁄,_  
_𑀳𑀢𑁆𑀣𑁂𑀲𑀼 𑀞𑀧𑁂𑀳𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀲𑁆𑀲”𑁇_  


(67.)

2921. _𑀘𑁄𑀤𑀺𑀢𑁄 𑀲𑀺𑀯𑀺𑀭𑀸𑀚𑁂𑀦,_  
_𑀲𑀻𑀯𑀺𑀓𑁄 𑀯𑀘𑀦𑀗𑁆𑀓𑀭𑁄;_  
_𑀭𑀜𑁆𑀜𑁄 𑀘𑀓𑁆𑀔𑀽𑀦𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀫𑀬𑀺;_  
_𑀲𑀘𑀓𑁆𑀔𑀼 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀆𑀲𑀺,_  
_𑀅𑀦𑁆𑀥𑁄 𑀭𑀸𑀚𑀸 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑁇_  


(68.)

2922. _𑀢𑀢𑁄 𑀲𑁄 𑀓𑀢𑀺𑀧𑀸𑀳𑀲𑁆𑀲,_  
_𑀉𑀧𑀭𑀽𑀍𑀳𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀲𑀼;_  
_𑀲𑀽𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑀬𑀻 𑀭𑀸𑀚𑀸,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁄𑁇_  


(69.)

2923. _“𑀬𑁄𑀚𑁂𑀳𑀺 𑀲𑀸𑀭𑀣𑀺 𑀬𑀸𑀦𑀁,_  
_𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀝𑀺𑀯𑁂𑀤𑀬;_  
_𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀕𑀘𑁆𑀙𑀸𑀫,_  
_𑀧𑁄𑀓𑁆𑀔𑀭𑀜𑁆𑀜𑁄 𑀯𑀦𑀸𑀦𑀺 𑀘”𑁇_  


(70.)

2924. _𑀲𑁄 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀢𑀻𑀭𑁂,_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀉𑀧𑀸𑀯𑀺𑀲𑀺;_  
_𑀢𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑀼,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑀸 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀺𑁇_  


(71.)

2925. _“𑀲𑀓𑁆𑀓𑁄𑀳𑀫𑀲𑁆𑀫𑀺 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀆𑀕𑀢𑁄𑀲𑁆𑀫𑀺 𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀯𑀭𑀁 𑀯𑀭𑀲𑁆𑀲𑀼 𑀭𑀸𑀚𑀻𑀲𑀺,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀫𑀦𑀲𑀺𑀘𑁆𑀙𑀲𑀺”𑁇_  


(72.)

2926. _“𑀧𑀳𑀽𑀢𑀁 𑀫𑁂 𑀥𑀦𑀁 𑀲𑀓𑁆𑀓,_  
_𑀩𑀮𑀁 𑀓𑁄𑀲𑁄 𑀘𑀦𑀧𑁆𑀧𑀓𑁄;_  
_𑀅𑀦𑁆𑀥𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄 𑀤𑀸𑀦𑀺,_  
_𑀫𑀭𑀡𑀜𑁆𑀜𑁂𑀯 𑀭𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(73.)

2927. _“𑀬𑀸𑀦𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀤𑁆𑀯𑀺𑀧𑀤𑀺𑀦𑁆𑀤,_  
_𑀢𑀸𑀦𑀺 𑀪𑀸𑀲𑀲𑁆𑀲𑀼 𑀔𑀢𑁆𑀢𑀺𑀬;_  
_𑀲𑀘𑁆𑀘𑀁 𑀢𑁂 𑀪𑀡𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀧𑀼𑀦 𑀘𑀓𑁆𑀔𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(74.)

2928. _“𑀬𑁂 𑀫𑀁 𑀬𑀸𑀘𑀺𑀢𑀼𑀫𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀦𑀸𑀕𑁄𑀢𑁆𑀢𑀸 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀸;_  
_𑀬𑁄𑀧𑀺 𑀫𑀁 𑀬𑀸𑀘𑀢𑁂 𑀢𑀢𑁆𑀣,_  
_𑀲𑁄𑀧𑀺 𑀫𑁂 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑁄;_  
_𑀏𑀢𑁂𑀦 𑀲𑀘𑁆𑀘𑀯𑀚𑁆𑀚𑁂𑀦,_  
_𑀘𑀓𑁆𑀔𑀼 𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀣𑁇_  


(75.)

2929. _𑀬𑀁 𑀫𑀁 𑀲𑁄 𑀬𑀸𑀘𑀺𑀢𑀼𑀁 𑀆𑀕𑀸,_  
_𑀤𑁂𑀳𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀢𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀽𑀦𑀺 𑀧𑀸𑀤𑀸𑀲𑀺𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀦𑀺𑀩𑁆𑀩𑀢𑁄𑁇_  


(76.)

2930. _𑀪𑀺𑀬𑁆𑀬𑁄 𑀫𑀁 𑀆𑀯𑀺𑀲𑀻 𑀧𑀻𑀢𑀺,_  
_𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀜𑁆𑀘𑀦𑀧𑁆𑀧𑀓𑀁;_  
_𑀏𑀢𑁂𑀦 𑀲𑀘𑁆𑀘𑀯𑀚𑁆𑀚𑁂𑀦,_  
_𑀤𑀼𑀢𑀺𑀬𑀁 𑀫𑁂 𑀉𑀧𑀧𑀚𑁆𑀚𑀣”𑁇_  


(77.)

2931. _“𑀥𑀫𑁆𑀫𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀸 𑀕𑀸𑀣𑀸,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦;_  
_𑀏𑀢𑀸𑀦𑀺 𑀢𑀯 𑀦𑁂𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀧𑀝𑀺𑀤𑀺𑀲𑁆𑀲𑀭𑁂𑁇_  


(78.)

2932. _𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀲𑁂𑀮𑀁,_  
_𑀲𑀫𑀢𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀼 𑀢𑁂”𑁇_  


(79.)

2933. _“𑀓𑁄 𑀦𑀻𑀥 𑀯𑀺𑀢𑁆𑀢𑀁 𑀦 𑀤𑀤𑁂𑀬𑁆𑀬 𑀬𑀸𑀘𑀺𑀢𑁄,_  
_𑀅𑀧𑀺 𑀯𑀺𑀲𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀧𑀺𑀬𑀫𑁆𑀧𑀺 𑀅𑀢𑁆𑀢𑀦𑁄;_  
_𑀢𑀤𑀺𑀗𑁆𑀖 𑀲𑀩𑁆𑀩𑁂 𑀲𑀺𑀯𑀬𑁄 𑀲𑀫𑀸𑀕𑀢𑀸,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑁂𑀢𑁆𑀢𑀸𑀦𑀺 𑀫𑀫𑀚𑁆𑀚 𑀧𑀲𑁆𑀲𑀣𑁇_  


(80.)

2934. _𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀲𑁂𑀮𑀁,_  
_𑀲𑀫𑀢𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀤𑀲𑁆𑀲𑀦𑀁 𑀅𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀺 𑀫𑁂𑁇_  


(81.)

2935. _𑀦 𑀘𑀸𑀕𑀫𑀢𑁆𑀢𑀸 𑀧𑀭𑀫𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀫𑀘𑁆𑀘𑀸𑀦𑀁 𑀇𑀥 𑀚𑀻𑀯𑀺𑀢𑁂;_  
_𑀤𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀦𑀼𑀲𑀁 𑀘𑀓𑁆𑀔𑀼𑀁,_  
_𑀮𑀤𑁆𑀥𑀁 𑀫𑁂 𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀫𑀸𑀦𑀼𑀲𑀁𑁇_  


(82.)

2936. _𑀏𑀢𑀫𑁆𑀧𑀺 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀺𑀯𑀬𑁄,_  
_𑀤𑁂𑀣 𑀤𑀸𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀣;_  
_𑀤𑀢𑁆𑀯𑀸 𑀘 𑀪𑀼𑀢𑁆𑀯𑀸 𑀘 𑀬𑀣𑀸𑀦𑀼𑀪𑀸𑀯𑀁,_  
_𑀅𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑀸 𑀲𑀕𑁆𑀕𑀫𑀼𑀧𑁂𑀣 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

2937. 𑀲𑀺𑀯𑀺𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
