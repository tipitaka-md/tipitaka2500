# 𑁧𑁫𑁇𑁧𑁇𑁯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀓

* 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

* 𑀫𑀸𑀢𑀗𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀓

(211.)

3071\. _“𑀓𑁄 𑀢𑀁 𑀳𑀺𑀁𑀲𑀢𑀺 𑀳𑁂𑀞𑁂𑀢𑀺,_  
_𑀓𑀺𑀁 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀲𑁄𑀘𑀲𑀺 𑀅𑀧𑁆𑀧𑀢𑀻𑀢𑁄;_  
_𑀓𑀲𑁆𑀲𑀚𑁆𑀚 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀭𑀼𑀤𑀦𑁆𑀢𑀼,_  
_𑀓𑁆𑀯𑀚𑁆𑀚 𑀲𑁂𑀢𑀼 𑀦𑀺𑀳𑀢𑁄 𑀧𑀣𑀩𑁆𑀬𑀸”𑁇_  


(212.)

3072\. _“𑀢𑀼𑀝𑁆𑀞𑁄𑀲𑁆𑀫𑀺 𑀤𑁂𑀯 𑀢𑀯 𑀤𑀲𑁆𑀲𑀦𑁂𑀦,_  
_𑀘𑀺𑀭𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀁 𑀪𑀽𑀫𑀺𑀧𑀸𑀮;_  
_𑀅𑀳𑀺𑀁𑀲𑀓𑁄 𑀭𑁂𑀡𑀼𑀫𑀦𑀼𑀧𑁆𑀧𑀯𑀺𑀲𑁆𑀲,_  
_𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀢𑁂 𑀳𑁂𑀞𑀬𑀺𑀢𑁄𑀲𑁆𑀫𑀺 𑀤𑁂𑀯”𑁇_  


(213.)

3073\. _“𑀆𑀬𑀦𑁆𑀢𑀼 𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑀸 𑀔𑀕𑁆𑀕𑀩𑀦𑁆𑀥𑀸,_  
_𑀓𑀸𑀲𑀸𑀯𑀺𑀬𑀸 𑀬𑀦𑁆𑀢𑀼 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀦𑁆𑀢𑀁;_  
_𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀁,_  
_𑀙𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀻𑀲𑀁 𑀯𑀭𑀫𑀸𑀳𑀭𑀦𑁆𑀢𑀼”𑁇_  


(214.)

3074\. _“𑀧𑁂𑀲𑀺𑀢𑀸 𑀭𑀸𑀚𑀺𑀦𑁄 𑀤𑀽𑀢𑀸,_  
_𑀓𑀼𑀫𑀸𑀭𑀁 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀼𑀁;_  
_𑀇𑀲𑁆𑀲𑀭𑁂𑀦 𑀯𑀺𑀢𑀺𑀡𑁆𑀡𑁄𑀲𑀺,_  
_𑀯𑀥𑀁 𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬”𑁇_  


(215.)

3075\. _“𑀲 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀤𑁂𑀯𑀬𑀦𑁆𑀢𑁄,_  
_𑀤𑀲𑀗𑁆𑀕𑀼𑀮𑀺𑀁 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸;_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀚𑀦𑀺𑀦𑁆𑀤 𑀤𑀝𑁆𑀞𑀼𑀁,_  
_𑀚𑀻𑀯𑀁 𑀫𑀁 𑀦𑁂𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀤𑀲𑁆𑀲𑀬𑁂𑀣”𑁇_  


(216.)

3076\. _𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀭𑀜𑁆𑀜𑁄 𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀤𑀲𑁆𑀲𑀬𑀼𑀁;_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀧𑀺𑀢𑀭𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀤𑀽𑀭𑀢𑁄𑀯𑀚𑁆𑀛𑀪𑀸𑀲𑀣𑁇_  


(217.)

3077\. _“𑀆𑀕𑀘𑁆𑀙𑀼𑀁 𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑀸 𑀔𑀕𑁆𑀕𑀩𑀦𑁆𑀥𑀸,_  
_𑀓𑀸𑀲𑀸𑀯𑀺𑀬𑀸 𑀳𑀦𑁆𑀢𑀼 𑀫𑀫𑀁 𑀚𑀦𑀺𑀦𑁆𑀤;_  
_𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀅𑀧𑀭𑀸𑀥𑁄 𑀓𑁄 𑀦𑀺𑀥 𑀫𑀫𑀚𑁆𑀚 𑀅𑀢𑁆𑀣𑀺”𑁇_  


(218.)

3078\. _“𑀲𑀸𑀬𑀜𑁆𑀘 𑀧𑀸𑀢𑁄 𑀉𑀤𑀓𑀁 𑀲𑀚𑀸𑀢𑀺,_  
_𑀅𑀕𑁆𑀕𑀺𑀁 𑀲𑀤𑀸 𑀧𑀸𑀭𑀺𑀘𑀭𑀢𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄;_  
_𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀲𑀁𑀬𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁,_  
_𑀓𑀲𑁆𑀫𑀸 𑀢𑀼𑀯𑀁 𑀩𑁆𑀭𑀽𑀲𑀺 𑀕𑀳𑀧𑁆𑀧𑀢𑀻𑀢𑀺”𑁇_  


(219.)

3079\. _“𑀢𑀸𑀮𑀸 𑀘 𑀫𑀽𑀮𑀸 𑀘 𑀨𑀮𑀸 𑀘 𑀤𑁂𑀯,_  
_𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀸 𑀯𑀺𑀯𑀺𑀥𑀸 𑀲𑀦𑁆𑀢𑀺𑀫𑀲𑁆𑀲;_  
_𑀢𑁂 𑀭𑀓𑁆𑀔𑀢𑀺 𑀕𑁄𑀧𑀬𑀢𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀅𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀕𑀳𑀧𑁆𑀧𑀢𑀻𑀢𑀺”𑁇_  


(220.)

3080\. _“𑀲𑀘𑁆𑀘𑀁 𑀔𑁄 𑀏𑀢𑀁 𑀯𑀤𑀲𑀺 𑀓𑀼𑀫𑀸𑀭,_  
_𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀸 𑀯𑀺𑀯𑀺𑀥𑀸 𑀲𑀦𑁆𑀢𑀺𑀫𑀲𑁆𑀲;_  
_𑀢𑁂 𑀭𑀓𑁆𑀔𑀢𑀺 𑀕𑁄𑀧𑀬𑀢𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄,_  
_𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀕𑀳𑀧𑀢𑀺 𑀢𑁂𑀦 𑀳𑁄𑀢𑀺”𑁇_  


(221.)

3081\. _“𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑀬𑁆𑀳𑀁 𑀧𑀭𑀺𑀲𑀸 𑀲𑀫𑀸𑀕𑀢𑀸,_  
_𑀲𑀦𑁂𑀕𑀫𑀸 𑀚𑀸𑀦𑀧𑀤𑀸 𑀘 𑀲𑀩𑁆𑀩𑁂;_  
_𑀩𑀸𑀮𑀸𑀬𑀁 𑀩𑀸𑀮𑀲𑁆𑀲 𑀯𑀘𑁄 𑀦𑀺𑀲𑀫𑁆𑀫,_  
_𑀅𑀳𑁂𑀢𑀼𑀦𑀸 𑀖𑀸𑀢𑀬𑀢𑁂 𑀫𑀁 𑀚𑀦𑀺𑀦𑁆𑀤𑁄𑁇_  


(222.)

3082\. _𑀤𑀍𑀳𑀲𑁆𑀫𑀺 𑀫𑀽𑀮𑁂 𑀯𑀺𑀲𑀝𑁂 𑀯𑀺𑀭𑀽𑀍𑀳𑁂,_  
_𑀤𑀼𑀦𑁆𑀦𑀺𑀓𑁆𑀓𑀬𑁄 𑀯𑁂𑀍𑀉 𑀧𑀲𑀸𑀔𑀚𑀸𑀢𑁄;_  
_𑀯𑀦𑁆𑀤𑀸𑀫𑀺 𑀧𑀸𑀤𑀸𑀦𑀺 𑀢𑀯 𑀚𑀦𑀺𑀦𑁆𑀤,_  
_𑀅𑀦𑀼𑀚𑀸𑀦 𑀫𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑁂𑀯”𑁇_  


(223.)

3083\. _“𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀪𑁄𑀕𑁂 𑀯𑀺𑀧𑀼𑀮𑁂 𑀓𑀼𑀫𑀸𑀭,_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀢𑁂 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀤𑀤𑀸𑀫𑀺;_  
_𑀅𑀚𑁆𑀚𑁂𑀯 𑀢𑁆𑀯𑀁 𑀓𑀼𑀭𑀽𑀦𑀁 𑀳𑁄𑀳𑀺 𑀭𑀸𑀚𑀸,_  
_𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀻 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑀸”𑁇_  


(224.)

3084\. _“𑀓𑀺𑀦𑁆𑀦𑀽𑀥 𑀤𑁂𑀯 𑀢𑀯𑀫𑀢𑁆𑀣𑀺 𑀪𑁄𑀕𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑁂𑀯𑀳𑀁 𑀤𑁂𑀯𑀮𑁄𑀓𑁂 𑀭𑀫𑀺𑀲𑁆𑀲𑀁;_  
_𑀭𑀽𑀧𑁂𑀳𑀺 𑀲𑀤𑁆𑀤𑁂𑀳𑀺 𑀅𑀣𑁄 𑀭𑀲𑁂𑀳𑀺,_  
_𑀕𑀦𑁆𑀥𑁂𑀳𑀺 𑀨𑀲𑁆𑀲𑁂𑀳𑀺 𑀫𑀦𑁄𑀭𑀫𑁂𑀳𑀺𑁇_  


(225.)

3085\. _𑀪𑀼𑀢𑁆𑀢𑀸 𑀘 𑀫𑁂 𑀪𑁄𑀕𑀸 𑀢𑀺𑀤𑀺𑀯𑀲𑁆𑀫𑀺𑀁 𑀤𑁂𑀯,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀘𑁆𑀙𑀭𑀸𑀦𑀁 𑀕𑀡𑁂𑀦;_  
_𑀢𑀼𑀯𑀜𑁆𑀘 𑀩𑀸𑀮𑀁 𑀧𑀭𑀦𑁂𑀬𑁆𑀬𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀦 𑀢𑀸𑀤𑀺𑀲𑁂 𑀭𑀸𑀚𑀓𑀼𑀮𑁂 𑀯𑀲𑁂𑀬𑁆𑀬𑀁”𑁇_  


(226.)

3086\. _“𑀲𑀘𑀸𑀳𑀁 𑀩𑀸𑀮𑁄 𑀧𑀭𑀦𑁂𑀬𑁆𑀬𑁄 𑀅𑀲𑁆𑀫𑀺,_  
_𑀏𑀓𑀸𑀧𑀭𑀸𑀥𑀁 𑀔𑀫 𑀧𑀼𑀢𑁆𑀢 𑀫𑀬𑁆𑀳𑀁;_  
_𑀧𑀼𑀦𑀧𑀺 𑀘𑁂 𑀏𑀤𑀺𑀲𑀓𑀁 𑀪𑀯𑁂𑀬𑁆𑀬,_  
_𑀬𑀣𑀸𑀫𑀢𑀺𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲 𑀓𑀭𑁄𑀳𑀺”𑁇_  


(227.)

3087\. _“𑀅𑀦𑀺𑀲𑀫𑁆𑀫 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀅𑀦𑀯𑀢𑁆𑀣𑀸𑀬 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀁;_  
_𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑁆𑀲𑁂𑀯 𑀯𑁂𑀪𑀗𑁆𑀕𑁄,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑁄𑀢𑀺 𑀧𑀸𑀧𑀓𑁄𑁇_  


(228.)

3088\. _𑀦𑀺𑀲𑀫𑁆𑀫 𑀘 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀣𑀸𑀬 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀁;_  
_𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑁆𑀲𑁂𑀯 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑁄𑀢𑀺 𑀪𑀤𑁆𑀭𑀓𑁄𑁇_  


(229.)

3089\. _𑀅𑀮𑀲𑁄 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀦 𑀲𑀸𑀥𑀼,_  
_𑀅𑀲𑀜𑁆𑀜𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀦 𑀲𑀸𑀥𑀼;_  
_𑀭𑀸𑀚𑀸 𑀦 𑀲𑀸𑀥𑀼 𑀅𑀦𑀺𑀲𑀫𑁆𑀫𑀓𑀸𑀭𑀻,_  
_𑀬𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀓𑁄𑀥𑀦𑁄 𑀢𑀁 𑀦 𑀲𑀸𑀥𑀼𑁇_  


(230.)

3090\. _𑀦𑀺𑀲𑀫𑁆𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀓𑀬𑀺𑀭𑀸,_  
_𑀦𑀸𑀦𑀺𑀲𑀫𑁆𑀫 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀦𑀺𑀲𑀫𑁆𑀫𑀓𑀸𑀭𑀺𑀦𑁄 𑀭𑀸𑀚,_  
_𑀬𑀲𑁄 𑀓𑀺𑀢𑁆𑀢𑀺 𑀘 𑀯𑀟𑁆𑀠𑀢𑀺𑁇_  


(231.)

3091\. _𑀦𑀺𑀲𑀫𑁆𑀫 𑀤𑀡𑁆𑀟𑀁 𑀧𑀡𑀬𑁂𑀬𑁆𑀬 𑀇𑀲𑁆𑀲𑀭𑁄,_  
_𑀯𑁂𑀕𑀸 𑀓𑀢𑀁 𑀢𑀧𑁆𑀧𑀢𑀺 𑀪𑀽𑀫𑀺𑀧𑀸𑀮;_  
_𑀲𑀫𑁆𑀫𑀸𑀧𑀡𑀻𑀥𑀻 𑀘 𑀦𑀭𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸,_  
_𑀅𑀦𑀸𑀦𑀼𑀢𑀧𑁆𑀧𑀸 𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑀺 𑀧𑀘𑁆𑀙𑀸𑁇_  


(232.)

3092\. _𑀅𑀦𑀸𑀦𑀼𑀢𑀧𑁆𑀧𑀸𑀦𑀺 𑀳𑀺 𑀬𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀪𑀚𑁆𑀚 𑀓𑀫𑁆𑀫𑀸𑀬𑀢𑀦𑀸𑀦𑀺 𑀮𑁄𑀓𑁂;_  
_𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀲𑀼𑀔𑀼𑀤𑁆𑀭𑀬𑀸𑀦𑀺,_  
_𑀪𑀯𑀦𑁆𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀫𑀢𑀸𑀦𑀺 𑀢𑀸𑀦𑀺𑁇_  


(233.)

3093\. _𑀆𑀕𑀘𑁆𑀙𑀼𑀁 𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑀸 𑀔𑀕𑁆𑀕𑀩𑀦𑁆𑀥𑀸,_  
_𑀓𑀸𑀲𑀸𑀯𑀺𑀬𑀸 𑀳𑀦𑁆𑀢𑀼 𑀫𑀫𑀁 𑀚𑀦𑀺𑀦𑁆𑀤;_  
_𑀫𑀸𑀢𑀼𑀜𑁆𑀘 𑀅𑀗𑁆𑀓𑀲𑁆𑀫𑀺𑀫𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄,_  
_𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁄 𑀲𑀳𑀲𑀸 𑀢𑁂𑀳𑀺 𑀤𑁂𑀯𑁇_  


(234.)

3094\. _𑀓𑀝𑀼𑀓𑀜𑁆𑀳𑀺 𑀲𑀫𑁆𑀩𑀸𑀥𑀁 𑀲𑀼𑀓𑀺𑀘𑁆𑀙𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀫𑀥𑀼𑀭𑀫𑁆𑀧𑀺 𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀮𑀤𑁆𑀥 𑀭𑀸𑀚;_  
_𑀓𑀺𑀘𑁆𑀙𑁂𑀦𑀳𑀁 𑀅𑀚𑁆𑀚 𑀯𑀥𑀸 𑀧𑀫𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀫𑁂𑀯𑀸𑀪𑀺𑀫𑀦𑁄𑀳𑀫𑀲𑁆𑀫𑀺”𑁇_  


(235.)

3095\. _“𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀯𑀸𑀬𑀁 𑀢𑀭𑀼𑀡𑁄 𑀲𑀼𑀥𑀫𑁆𑀫𑁂,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑁄 𑀓𑀼𑀫𑀸𑀭𑁄;_  
_𑀢𑀁 𑀬𑀸𑀘𑀫𑀸𑀦𑁄 𑀦 𑀮𑀪𑀸𑀫𑀺 𑀲𑁆𑀯𑀚𑁆𑀚,_  
_𑀅𑀭𑀳𑀲𑀺 𑀦𑀁 𑀬𑀸𑀘𑀺𑀢𑀯𑁂 𑀢𑀼𑀯𑀫𑁆𑀧𑀺”𑁇_  


(236.)

3096\. _“𑀭𑀫𑀲𑁆𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀸𑀘𑀭𑀺𑀬𑀸𑀬 𑀧𑀼𑀢𑁆𑀢,_  
_𑀦𑀺𑀲𑀫𑁆𑀫 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑀲𑁆𑀲𑀼;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀪𑀽𑀢𑁂𑀲𑀼 𑀦𑀺𑀥𑀸𑀬 𑀤𑀡𑁆𑀟𑀁,_  
_𑀅𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀫𑀼𑀧𑁂𑀳𑀺 𑀞𑀸𑀦𑀁”𑁇_  


(237.)

3097\. _“𑀅𑀘𑁆𑀙𑁂𑀭 𑀭𑀽𑀧𑀁 𑀯𑀢 𑀬𑀸𑀤𑀺𑀲𑀜𑁆𑀘,_  
_𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀫𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀬𑀲𑁂 𑀲𑀼𑀥𑀫𑁆𑀫𑁂;_  
_𑀬𑀸𑀘𑀲𑁆𑀲𑀼 𑀧𑀼𑀢𑁆𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸,_  
_𑀪𑀺𑀬𑁆𑀬𑁄𑀯 𑀉𑀲𑁆𑀲𑀸𑀳𑀬𑀲𑁂 𑀓𑀼𑀫𑀸𑀭𑀁”𑁇_  


(238.)

3098\. _“𑀬𑁂 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀅𑀦𑀯𑀚𑁆𑀚𑀪𑁄𑀕𑀺𑀦𑁄,_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀮𑁄𑀓𑀫𑀺𑀫𑀁 𑀘𑀭𑀦𑁆𑀢𑀺;_  
_𑀢𑀫𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀫𑀸𑀦𑀁,_  
_𑀦 𑀉𑀲𑁆𑀲𑀳𑁂 𑀯𑀸𑀭𑀬𑀺𑀢𑀼𑀁 𑀓𑀼𑀫𑀸𑀭𑀁”𑁇_  


(239.)

3099\. _“𑀅𑀤𑁆𑀥𑀸 𑀳𑀯𑁂 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀲𑀧𑀜𑁆𑀜𑀸,_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀬𑁂 𑀩𑀳𑀼𑀞𑀸𑀦𑀘𑀺𑀦𑁆𑀢𑀺𑀦𑁄;_  
_𑀬𑁂𑀲𑀸𑀬𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀺,_  
_𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀸 𑀯𑀻𑀢𑀲𑁄𑀓𑀸 𑀲𑀼𑀥𑀫𑁆𑀫𑀸”𑀢𑀺𑁇_  


---

3100\. 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇
