# 𑁧𑁫𑁇𑁧𑁇𑁫 𑀭𑁄𑀳𑀡𑀫𑀺𑀕𑀚𑀸𑀢𑀓

* 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

* 𑀫𑀸𑀢𑀗𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀭𑁄𑀳𑀡𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(104.)

2960\. _“𑀏𑀢𑁂 𑀬𑀽𑀣𑀸 𑀧𑀢𑀺𑀬𑀦𑁆𑀢𑀺,_  
_𑀪𑀻𑀢𑀸 𑀫𑀭𑀡𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀓;_  
_𑀕𑀘𑁆𑀙 𑀢𑀼𑀯𑀫𑁆𑀧𑀺 𑀫𑀸𑀓𑀗𑁆𑀔𑀺,_  
_𑀚𑀻𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀬𑀸 𑀲𑀳”𑁇_  


(105.)

2961\. _“𑀦𑀸𑀳𑀁 𑀭𑁄𑀳𑀡 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀳𑀤𑀬𑀁 𑀫𑁂 𑀅𑀯𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀦 𑀢𑀁 𑀅𑀳𑀁 𑀚𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀇𑀥 𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁”𑁇_  


(106.)

2962\. _“𑀢𑁂 𑀳𑀺 𑀦𑀽𑀦 𑀫𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑁆𑀥𑀸 𑀅𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀸;_  
_𑀕𑀘𑁆𑀙 𑀢𑀼𑀯𑀫𑁆𑀧𑀺 𑀫𑀸𑀓𑀗𑁆𑀔𑀺,_  
_𑀚𑀻𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀬𑀸 𑀲𑀳”𑁇_  


(107.)

2963\. _“𑀦𑀸𑀳𑀁 𑀭𑁄𑀳𑀡 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀳𑀤𑀬𑀁 𑀫𑁂 𑀅𑀯𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀦 𑀢𑀁 𑀩𑀤𑁆𑀥𑀁 𑀚𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀇𑀥 𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁”𑁇_  


(108.)

2964\. _“𑀕𑀘𑁆𑀙 𑀪𑀻𑀭𑀼 𑀧𑀮𑀸𑀬𑀲𑁆𑀲𑀼,_  
_𑀓𑀽𑀝𑁂 𑀩𑀤𑁆𑀥𑁄𑀲𑁆𑀫𑀺 𑀆𑀬𑀲𑁂;_  
_𑀕𑀘𑁆𑀙 𑀢𑀼𑀯𑀫𑁆𑀧𑀺 𑀫𑀸𑀓𑀗𑁆𑀔𑀺,_  
_𑀚𑀻𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀬𑀸 𑀲𑀳”𑁇_  


(109.)

2965\. _“𑀦𑀸𑀳𑀁 𑀭𑁄𑀳𑀡 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀳𑀤𑀬𑀁 𑀫𑁂 𑀅𑀯𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀦 𑀢𑀁 𑀅𑀳𑀁 𑀚𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀇𑀥 𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁”𑁇_  


(110.)

2966\. _“𑀢𑁂 𑀳𑀺 𑀦𑀽𑀦 𑀫𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑁆𑀥𑀸 𑀅𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀸;_  
_𑀕𑀘𑁆𑀙 𑀢𑀼𑀯𑀫𑁆𑀧𑀺 𑀫𑀸𑀓𑀗𑁆𑀔𑀺,_  
_𑀚𑀻𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀬𑀸 𑀲𑀳”𑁇_  


(111.)

2967\. _“𑀦𑀸𑀳𑀁 𑀭𑁄𑀳𑀡 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀳𑀤𑀬𑀁 𑀫𑁂 𑀅𑀯𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀦 𑀢𑀁 𑀩𑀤𑁆𑀥𑀁 𑀚𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀇𑀥 𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁”𑁇_  


(112.)

2968\. _“𑀅𑀬𑀁 𑀲𑁄 𑀮𑀼𑀤𑁆𑀤𑀓𑁄 𑀏𑀢𑀺,_  
_𑀮𑀼𑀤𑁆𑀤𑀭𑀽𑀧𑁄 𑀲𑀳𑀸𑀯𑀼𑀥𑁄;_  
_𑀬𑁄 𑀦𑁄 𑀯𑀥𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀚𑁆𑀚,_  
_𑀉𑀲𑀼𑀦𑀸 𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑀺”𑁇_  


(113.)

2969\. _“𑀲𑀸 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀧𑀮𑀸𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀪𑀬𑀝𑁆𑀝𑀸 𑀪𑀬𑀢𑀚𑁆𑀚𑀺𑀢𑀸;_  
_𑀲𑀼𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀅𑀓𑀭𑀸 𑀪𑀻𑀭𑀼,_  
_𑀫𑀭𑀡𑀸𑀬𑀽𑀧𑀦𑀺𑀯𑀢𑁆𑀢𑀣”𑁇_  


(114.)

2970\. _“𑀓𑀺𑀁 𑀦𑀼 𑀢𑁂𑀫𑁂 𑀫𑀺𑀕𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀩𑀤𑁆𑀥𑀁 𑀉𑀧𑀸𑀲𑀭𑁂;_  
_𑀦 𑀢𑀁 𑀘𑀚𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲𑀧𑀺 𑀓𑀸𑀭𑀡𑀸”𑁇_  


(115.)

2971\. _“𑀪𑀸𑀢𑀭𑁄 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑁂 𑀮𑀼𑀤𑁆𑀤,_  
_𑀲𑁄𑀤𑀭𑀺𑀬𑀸 𑀏𑀓𑀫𑀸𑀢𑀼𑀓𑀸;_  
_𑀦 𑀫𑀁 𑀘𑀚𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲𑀧𑀺 𑀓𑀸𑀭𑀡𑀸”𑁇_  


(116.)

2972\. _“𑀢𑁂 𑀳𑀺 𑀦𑀽𑀦 𑀫𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑁆𑀥𑀸 𑀅𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀸;_  
_𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀤𑁂𑀳𑀺,_  
_𑀪𑀸𑀢𑀭𑀁 𑀫𑀼𑀜𑁆𑀘 𑀮𑀼𑀤𑁆𑀤𑀓”𑁇_  


(117.)

2973\. _“𑀲𑁄 𑀯𑁄 𑀅𑀳𑀁 𑀧𑀫𑁄𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑀁 𑀫𑀺𑀕𑀁;_  
_𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀼 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀫𑀼𑀢𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀳𑀸𑀫𑀺𑀕𑀁”𑁇_  


(118.)

2974\. _“𑀏𑀯𑀁 𑀮𑀼𑀤𑁆𑀤𑀓 𑀦𑀦𑁆𑀤𑀲𑁆𑀲𑀼,_  
_𑀲𑀳 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀜𑀸𑀢𑀺𑀪𑀺;_  
_𑀬𑀣𑀸𑀳𑀫𑀚𑁆𑀚 𑀦𑀦𑁆𑀤𑀸𑀫𑀺,_  
_𑀫𑀼𑀢𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀳𑀸𑀫𑀺𑀕𑀁”𑁇_  


(119.)

2975\. _“𑀓𑀣𑀁 𑀢𑁆𑀯𑀁 𑀧𑀫𑁄𑀓𑁆𑀔𑁄 𑀆𑀲𑀺,_  
_𑀉𑀧𑀦𑀻𑀢𑀲𑁆𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑁂;_  
_𑀓𑀣𑀁 𑀧𑀼𑀢𑁆𑀢 𑀅𑀫𑁄𑀘𑁂𑀲𑀺,_  
_𑀓𑀽𑀝𑀧𑀸𑀲𑀫𑁆𑀳 𑀮𑀼𑀤𑁆𑀤𑀓𑁄”𑁇_  


(120.)

2976\. _“𑀪𑀡𑀁 𑀓𑀡𑁆𑀡𑀲𑀼𑀔𑀁 𑀯𑀸𑀘𑀁,_  
_𑀳𑀤𑀬𑀗𑁆𑀕𑀁 𑀳𑀤𑀬𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀓𑁄 𑀫𑀁 𑀅𑀫𑁄𑀘𑀬𑀺𑁇_  


(121.)

2977\. _𑀪𑀡𑀁 𑀓𑀡𑁆𑀡𑀲𑀼𑀔𑀁 𑀯𑀸𑀘𑀁,_  
_𑀳𑀤𑀬𑀗𑁆𑀕𑀁 𑀳𑀤𑀬𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺,_  
_𑀲𑀼𑀢𑀦𑀸 𑀫𑀁 𑀅𑀫𑁄𑀘𑀬𑀺𑁇_  


(122.)

2978\. _𑀲𑀼𑀢𑁆𑀯𑀸 𑀓𑀡𑁆𑀡𑀲𑀼𑀔𑀁 𑀯𑀸𑀘𑀁,_  
_𑀳𑀤𑀬𑀗𑁆𑀕𑀁 𑀳𑀤𑀬𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀮𑀼𑀤𑁆𑀤𑀓𑁄 𑀫𑀁 𑀅𑀫𑁄𑀘𑀬𑀺”𑁇_  


(123.)

2979\. _“𑀏𑀯𑀁 𑀆𑀦𑀦𑁆𑀤𑀺𑀢𑁄 𑀳𑁄𑀢𑀼,_  
_𑀲𑀳 𑀤𑀸𑀭𑁂𑀳𑀺 𑀮𑀼𑀤𑁆𑀤𑀓𑁄;_  
_𑀬𑀣𑀸 𑀫𑀬𑀚𑁆𑀚 𑀦𑀦𑁆𑀤𑀸𑀫,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀭𑁄𑀳𑀡𑀫𑀸𑀕𑀢𑀁”𑁇_  


(124.)

2980\. _“𑀦𑀦𑀼 𑀢𑁆𑀯𑀁 𑀅𑀯𑀘 𑀮𑀼𑀤𑁆𑀤,_  
_‘𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀆𑀳𑀭𑀺𑀁’;_  
_𑀅𑀣 𑀓𑁂𑀦 𑀦𑀼 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀦𑀸𑀳𑀭𑀺”𑁇_  


(125.)

2981\. _“𑀆𑀕𑀫𑀸 𑀘𑁂𑀯 𑀳𑀢𑁆𑀣𑀢𑁆𑀣𑀁,_  
_𑀓𑀽𑀝𑀧𑀸𑀲𑀜𑁆𑀘 𑀲𑁄 𑀫𑀺𑀕𑁄;_  
_𑀅𑀩𑀚𑁆𑀛𑀺 𑀢𑀁 𑀫𑀺𑀕𑀭𑀸𑀚𑀁,_  
_𑀢𑀜𑁆𑀘 𑀫𑀼𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀲𑀭𑁂𑁇_  


(126.)

2982\. _𑀢𑀲𑁆𑀲 𑀫𑁂 𑀅𑀳𑀼 𑀲𑀁𑀯𑁂𑀕𑁄,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑁄;_  
_𑀇𑀫𑀜𑁆𑀘𑀸𑀳𑀁 𑀫𑀺𑀕𑀁 𑀳𑀜𑁆𑀜𑁂,_  
_𑀅𑀚𑁆𑀚 𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁”𑁇_  


(127.)

2983\. _“𑀓𑀻𑀤𑀺𑀲𑀸 𑀢𑁂 𑀫𑀺𑀕𑀸 𑀮𑀼𑀤𑁆𑀤,_  
_𑀓𑀻𑀤𑀺𑀲𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀫𑀺𑀕𑀸;_  
_𑀓𑀣𑀁𑀯𑀡𑁆𑀡𑀸 𑀓𑀣𑀁𑀲𑀻𑀮𑀸,_  
_𑀩𑀸𑀍𑀳𑀁 𑀔𑁄 𑀦𑁂 𑀧𑀲𑀁𑀲𑀲𑀺”𑁇_  


(128.)

2984\. _“𑀑𑀤𑀸𑀢𑀲𑀺𑀗𑁆𑀕𑀸 𑀲𑀼𑀘𑀺𑀯𑀸𑀮𑀸,_  
_𑀚𑀸𑀢𑀭𑀽𑀧𑀢𑀘𑀽𑀧𑀫𑀸;_  
_𑀧𑀸𑀤𑀸 𑀮𑁄𑀳𑀺𑀢𑀓𑀸 𑀢𑁂𑀲𑀁,_  
_𑀅𑀜𑁆𑀚𑀺𑀢𑀓𑁆𑀔𑀸 𑀫𑀦𑁄𑀭𑀫𑀸”𑁇_  


(129.)

2985\. _“𑀏𑀤𑀺𑀲𑀸 𑀢𑁂 𑀫𑀺𑀕𑀸 𑀤𑁂𑀯,_  
_𑀏𑀤𑀺𑀲𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀫𑀺𑀕𑀸;_  
_𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑀸 𑀤𑁂𑀯,_  
_𑀦 𑀢𑁂 𑀲𑁄 𑀅𑀪𑀺𑀳𑀸𑀭𑀺𑀢𑀼𑀁”𑁇_  


(130.)

2986\. _“𑀤𑀫𑁆𑀫𑀺 𑀦𑀺𑀓𑁆𑀔𑀲𑀢𑀁 𑀮𑀼𑀤𑁆𑀤,_  
_𑀣𑀽𑀮𑀜𑁆𑀘 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀁;_  
_𑀘𑀢𑀼𑀲𑁆𑀲𑀤𑀜𑁆𑀘 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁,_  
_𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀲𑀭𑀺𑀦𑁆𑀦𑀺𑀪𑀁𑁇_  


(131.)

2987\. _𑀤𑁆𑀯𑁂 𑀘 𑀲𑀸𑀤𑀺𑀲𑀺𑀬𑁄 𑀪𑀭𑀺𑀬𑀸,_  
_𑀉𑀲𑀪𑀜𑁆𑀘 𑀕𑀯𑀁 𑀲𑀢𑀁;_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁,_  
_𑀩𑀳𑀼𑀓𑀸𑀭𑁄 𑀫𑁂𑀲𑀺 𑀮𑀼𑀤𑁆𑀤𑀓𑁇_  


(132.)

2988\. _𑀓𑀲𑀺𑀯𑀸𑀡𑀺𑀚𑁆𑀚𑀸 𑀇𑀡𑀤𑀸𑀦𑀁,_  
_𑀉𑀜𑁆𑀙𑀸𑀘𑀭𑀺𑀬𑀸 𑀘 𑀮𑀼𑀤𑁆𑀤𑀓;_  
_𑀏𑀢𑁂𑀦 𑀤𑀸𑀭𑀁 𑀧𑁄𑀲𑁂𑀳𑀺,_  
_𑀫𑀸 𑀧𑀸𑀧𑀁 𑀅𑀓𑀭𑀻 𑀧𑀼𑀦𑀸𑀢𑀺”𑁇_
