# 𑁭𑁇𑁨𑁇𑁧 𑀕𑀦𑁆𑀥𑀸𑀭𑀚𑀸𑀢𑀓

* 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀕𑀦𑁆𑀥𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀕𑀦𑁆𑀥𑀸𑀭𑀚𑀸𑀢𑀓

(76.)

1572\. _“𑀳𑀺𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸𑀦𑀺 𑀲𑁄𑀍𑀅𑀲;_  
_𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀸𑀦𑀺 𑀨𑀻𑀢𑀸𑀦𑀺,_  
_𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀁 𑀤𑀸𑀦𑀺 𑀓𑀼𑀩𑁆𑀩𑀲𑀺”𑁇_  


(77.)

1573\. _“𑀳𑀺𑀢𑁆𑀯𑀸 𑀕𑀦𑁆𑀥𑀸𑀭𑀯𑀺𑀲𑀬𑀁,_  
_𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀸𑀭𑀺𑀬𑀁;_  
_𑀧𑀲𑀸𑀲𑀦𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄,_  
_𑀇𑀥 𑀤𑀸𑀦𑀺 𑀧𑀲𑀸𑀲𑀲𑀺”𑁇_  


(78.)

1574\. _“𑀥𑀫𑁆𑀫𑀁 𑀪𑀡𑀸𑀫𑀺 𑀯𑁂𑀤𑁂𑀳,_  
_𑀅𑀥𑀫𑁆𑀫𑁄 𑀫𑁂 𑀦 𑀭𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀥𑀫𑁆𑀫𑀁 𑀫𑁂 𑀪𑀡𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀦 𑀧𑀸𑀧𑀫𑀼𑀧𑀮𑀺𑀫𑁆𑀧𑀢𑀺”𑁇_  


(79.)

1575\. _“𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀧𑀭𑁄 𑀮𑀪𑀢𑀺 𑀭𑀼𑀧𑁆𑀧𑀦𑀁;_  
_𑀫𑀳𑀢𑁆𑀣𑀺𑀬𑀫𑁆𑀧𑀺 𑀘𑁂 𑀯𑀸𑀘𑀁,_  
_𑀦 𑀢𑀁 𑀪𑀸𑀲𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄”𑁇_  


(80.)

1576\. _“𑀓𑀸𑀫𑀁 𑀭𑀼𑀧𑁆𑀧𑀢𑀼 𑀯𑀸 𑀫𑀸 𑀯𑀸,_  
_𑀪𑀼𑀲𑀁𑀯 𑀯𑀺𑀓𑀺𑀭𑀻𑀬𑀢𑀼;_  
_𑀥𑀫𑁆𑀫𑀁 𑀫𑁂 𑀪𑀡𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀦 𑀧𑀸𑀧𑀫𑀼𑀧𑀮𑀺𑀫𑁆𑀧𑀢𑀺”𑁇_  


(81.)

1577\. _“𑀦𑁄 𑀘𑁂 𑀅𑀲𑁆𑀲 𑀲𑀓𑀸 𑀩𑀼𑀤𑁆𑀥𑀺,_  
_𑀯𑀺𑀦𑀬𑁄 𑀯𑀸 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀯𑀦𑁂 𑀅𑀦𑁆𑀥𑀫𑀳𑀺𑀁𑀲𑁄𑀯,_  
_𑀘𑀭𑁂𑀬𑁆𑀬 𑀩𑀳𑀼𑀓𑁄 𑀚𑀦𑁄𑁇_  


(82.)

1578\. _𑀬𑀲𑁆𑀫𑀸 𑀘 𑀧𑀦𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂,_  
_𑀆𑀘𑁂𑀭𑀫𑁆𑀳𑀺 𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸;_  
_𑀢𑀲𑁆𑀫𑀸 𑀯𑀺𑀦𑀻𑀢𑀯𑀺𑀦𑀬𑀸,_  
_𑀘𑀭𑀦𑁆𑀢𑀺 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀸”𑀢𑀺𑁇_  


---
