# 𑁭𑁇𑁨𑁇𑁭 𑀓𑁄𑀝𑀲𑀺𑀫𑁆𑀩𑀮𑀺𑀚𑀸𑀢𑀓

* 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀕𑀦𑁆𑀥𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀓𑁄𑀝𑀲𑀺𑀫𑁆𑀩𑀮𑀺𑀚𑀸𑀢𑀓

(121.)

1623\. _“𑀅𑀳𑀁 𑀤𑀲𑀲𑀢𑀁𑀩𑁆𑀬𑀸𑀫𑀁,_  
_𑀉𑀭𑀕𑀫𑀸𑀤𑀸𑀬 𑀆𑀕𑀢𑁄;_  
_𑀢𑀜𑁆𑀘 𑀫𑀜𑁆𑀘 𑀫𑀳𑀸𑀓𑀸𑀬𑀁,_  
_𑀥𑀸𑀭𑀬𑀁 𑀦𑀧𑁆𑀧𑀯𑁂𑀥𑀲𑀺𑁇_  


(122.)

1624\. _𑀅𑀣𑀺𑀫𑀁 𑀔𑀼𑀤𑁆𑀤𑀓𑀁 𑀧𑀓𑁆𑀔𑀺𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑀁𑀲𑀢𑀭𑀁 𑀫𑀬𑀸;_  
_𑀥𑀸𑀭𑀬𑀁 𑀩𑁆𑀬𑀣𑀲𑀺 𑀪𑀻𑀢𑀸,_  
_𑀓𑀫𑀢𑁆𑀣𑀁 𑀓𑁄𑀝𑀲𑀺𑀫𑁆𑀩𑀮𑀺”𑁇_  


(123.)

1625\. _“𑀫𑀁𑀲𑀪𑀓𑁆𑀔𑁄 𑀢𑀼𑀯𑀁 𑀭𑀸𑀚,_  
_𑀨𑀮𑀪𑀓𑁆𑀔𑁄 𑀅𑀬𑀁 𑀤𑀺𑀚𑁄;_  
_𑀅𑀬𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀩𑀻𑀚𑀸𑀦𑀺,_  
_𑀧𑀺𑀮𑀓𑁆𑀔𑀼𑀤𑀼𑀫𑁆𑀩𑀭𑀸𑀦𑀺 𑀘;_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀘 𑀪𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸,_  
_𑀔𑀦𑁆𑀥𑁂 𑀫𑁂 𑀑𑀳𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(124.)

1626\. _𑀢𑁂 𑀭𑀼𑀓𑁆𑀔𑀸 𑀲𑀁𑀯𑀺𑀭𑀽𑀳𑀦𑁆𑀢𑀺,_  
_𑀫𑀫 𑀧𑀲𑁆𑀲𑁂 𑀦𑀺𑀯𑀸𑀢𑀚𑀸;_  
_𑀢𑁂 𑀫𑀁 𑀧𑀭𑀺𑀬𑁄𑀦𑀦𑁆𑀥𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀅𑀭𑀼𑀓𑁆𑀔𑀁 𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀭𑁂𑁇_  


(125.)

1627\. _𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀸 𑀲𑁂,_  
_𑀫𑀽𑀮𑀺𑀦𑁄 𑀔𑀦𑁆𑀥𑀺𑀦𑁄 𑀤𑀼𑀫𑀸;_  
_𑀇𑀫𑀺𑀦𑀸 𑀲𑀓𑀼𑀡𑀚𑀸𑀢𑁂𑀦,_  
_𑀩𑀻𑀚𑀫𑀸𑀳𑀭𑀺𑀢𑀸 𑀳𑀢𑀸𑁇_  


(126.)

1628\. _𑀅𑀚𑁆𑀛𑀸𑀭𑀽𑀳𑀸𑀪𑀺𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺,_  
_𑀩𑁆𑀭𑀳𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀯𑀦𑀧𑁆𑀧𑀢𑀺𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀭𑀸𑀚 𑀧𑀯𑁂𑀥𑀸𑀫𑀺,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁𑀦𑀸𑀕𑀢𑀁 𑀪𑀬𑀁”𑁇_  


(127.)

1629\. _“𑀲𑀗𑁆𑀓𑁂𑀬𑁆𑀬 𑀲𑀗𑁆𑀓𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺,_  
_𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸𑀦𑀸𑀕𑀢𑀁 𑀪𑀬𑀁;_  
_𑀅𑀦𑀸𑀕𑀢𑀪𑀬𑀸 𑀥𑀻𑀭𑁄,_  
_𑀉𑀪𑁄 𑀮𑁄𑀓𑁂 𑀅𑀯𑁂𑀓𑁆𑀔𑀢𑀻”𑀢𑀺𑁇_  


---

1630\. 𑀓𑁄𑀝𑀲𑀺𑀫𑁆𑀩𑀮𑀺𑀚𑀸𑀢𑀓𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
