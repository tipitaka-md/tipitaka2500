# 𑁭𑁇𑁨𑁇𑁧𑁦 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀺𑀚𑀸𑀢𑀓

### 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀕𑀦𑁆𑀥𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀺𑀚𑀸𑀢𑀓

(142.)

1647. _“𑀦 𑀓𑀺𑀭𑀢𑁆𑀣𑀺 𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑀼,_  
_𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀲𑀼 𑀅𑀧𑁆𑀧𑀺𑀓𑀸;_  
_𑀲𑀼𑀓𑁆𑀔𑀸𑀬 𑀅𑀮𑁄𑀡𑀺𑀓𑀸𑀬 𑀘,_  
_𑀧𑀲𑁆𑀲𑀨𑀮𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀺𑀬𑀸𑁇_  


(143.)

1648. _𑀳𑀢𑁆𑀣𑀺𑀕𑀯𑀲𑁆𑀲𑀸 𑀘𑀺𑀫𑁂 𑀩𑀳𑀽,_  
_𑀥𑀦𑀥𑀜𑁆𑀜𑀁 𑀧𑀣𑀯𑀻 𑀘 𑀓𑁂𑀯𑀮𑀸;_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀘𑀺𑀫𑀸 𑀅𑀘𑁆𑀙𑀭𑀽𑀧𑀫𑀸,_  
_𑀧𑀲𑁆𑀲 𑀨𑀮𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀺𑀬𑀸”𑁇_  


(144.)

1649. _“𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀭𑀸𑀚𑀓𑀼𑀜𑁆𑀚𑀭,_  
_𑀕𑀸𑀣𑀸 𑀪𑀸𑀲𑀲𑀺 𑀓𑁄𑀲𑀮𑀸𑀥𑀺𑀧;_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦,_  
_𑀩𑀸𑀍𑀳𑀁 𑀧𑀻𑀢𑀺𑀫𑀦𑁄 𑀧𑀪𑀸𑀲𑀲𑀺”𑁇_  


(145.)

1650. _“𑀇𑀫𑀲𑁆𑀫𑀺𑀜𑁆𑀜𑁂𑀯 𑀦𑀕𑀭𑁂,_  
_𑀓𑀼𑀮𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑁂 𑀅𑀳𑀼𑀁;_  
_𑀧𑀭𑀓𑀫𑁆𑀫𑀓𑀭𑁄 𑀆𑀲𑀺𑀁,_  
_𑀪𑀢𑀓𑁄 𑀲𑀻𑀮𑀲𑀁𑀯𑀼𑀢𑁄𑁇_  


(146.)

1651. _𑀓𑀫𑁆𑀫𑀸𑀬 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁄𑀳𑀁,_  
_𑀘𑀢𑀼𑀭𑁄 𑀲𑀫𑀡𑁂𑀤𑁆𑀤𑀲𑀁;_  
_𑀆𑀘𑀸𑀭𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑁂 𑀅𑀦𑀸𑀲𑀯𑁂𑁇_  


(147.)

1652. _𑀢𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀲𑀻𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀢𑁂;_  
_𑀅𑀤𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(148.)

1653. _𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲,_  
_𑀇𑀤𑀁 𑀫𑁂 𑀏𑀤𑀺𑀲𑀁 𑀨𑀮𑀁;_  
_𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀇𑀤𑀁 𑀭𑀚𑁆𑀚𑀁,_  
_𑀨𑀻𑀢𑀁 𑀥𑀭𑀡𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁”𑁇_  


(149.)

1654. _“𑀤𑀤𑀁 𑀪𑀼𑀜𑁆𑀚 𑀫𑀸 𑀘 𑀧𑀫𑀸𑀤𑁄,_  
_𑀘𑀓𑁆𑀓𑀁 𑀯𑀢𑁆𑀢𑀬 𑀓𑁄𑀲𑀮𑀸𑀥𑀺𑀧;_  
_𑀫𑀸 𑀭𑀸𑀚 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀅𑀳𑀼,_  
_𑀥𑀫𑁆𑀫𑀁 𑀧𑀸𑀮𑀬 𑀓𑁄𑀲𑀮𑀸𑀥𑀺𑀧”𑁇_  


(150.)

1655. _“𑀲𑁄𑀳𑀁 𑀢𑀤𑁂𑀯 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁,_  
_𑀯𑀝𑀼𑀫𑀁 𑀆𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑁄𑀪𑀦𑁂;_  
_𑀅𑀭𑀺𑀬𑀸𑀘𑀭𑀺𑀢𑀁 𑀲𑀼𑀓𑁄𑀲𑀮𑁂,_  
_𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀫𑁂 𑀫𑀦𑀸𑀧𑀸𑀯 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁𑁇_  


(151.)

1656. _𑀤𑁂𑀯𑀻 𑀯𑀺𑀬 𑀅𑀘𑁆𑀙𑀭𑀽𑀧𑀫𑀸,_  
_𑀫𑀚𑁆𑀛𑁂 𑀦𑀸𑀭𑀺𑀕𑀡𑀲𑁆𑀲 𑀲𑁄𑀪𑀲𑀺;_  
_𑀓𑀺𑀁 𑀓𑀫𑁆𑀫𑀫𑀓𑀸𑀲𑀺 𑀪𑀤𑁆𑀤𑀓𑀁,_  
_𑀓𑁂𑀦𑀸𑀲𑀺 𑀯𑀡𑁆𑀡𑀯𑀢𑀻 𑀲𑀼𑀓𑁄𑀲𑀮𑁂”𑁇_  


(152.)

1657. _“𑀅𑀫𑁆𑀩𑀝𑁆𑀞𑀓𑀼𑀮𑀲𑁆𑀲 𑀔𑀢𑁆𑀢𑀺𑀬,_  
_𑀤𑀸𑀲𑁆𑀬𑀸𑀳𑀁 𑀧𑀭𑀧𑁂𑀲𑀺𑀬𑀸 𑀅𑀳𑀼𑀁;_  
_𑀲𑀜𑁆𑀜𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀚𑀻𑀯𑀺𑀦𑀻,_  
_𑀲𑀻𑀮𑀯𑀢𑀻 𑀘 𑀅𑀧𑀸𑀧𑀤𑀲𑁆𑀲𑀦𑀸𑁇_  


(153.)

1658. _𑀉𑀤𑁆𑀥𑀝𑀪𑀢𑁆𑀢𑀁 𑀅𑀳𑀁 𑀢𑀤𑀸,_  
_𑀘𑀭𑀫𑀸𑀦𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀯𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸 𑀲𑀬𑀁 𑀅𑀳𑀁,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀨𑀮𑀁 𑀫𑀫𑁂𑀤𑀺𑀲𑀦𑁆”𑀢𑀺𑁇_  


---

1659. 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀺𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇
