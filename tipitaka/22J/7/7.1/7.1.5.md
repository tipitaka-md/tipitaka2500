# 𑁭𑁇𑁧𑁇𑁫 𑀤𑀩𑁆𑀪𑀧𑀼𑀧𑁆𑀨𑀚𑀸𑀢𑀓

### 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀓𑁆𑀓𑀼𑀯𑀕𑁆𑀕

### 𑀤𑀩𑁆𑀪𑀧𑀼𑀧𑁆𑀨𑀚𑀸𑀢𑀓

(29.)

1517\. _“𑀅𑀦𑀼𑀢𑀻𑀭𑀘𑀸𑀭𑀻 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀲𑀳𑀸𑀬𑀫𑀦𑀼𑀥𑀸𑀯 𑀫𑀁;_  
_𑀫𑀳𑀸 𑀫𑁂 𑀕𑀳𑀺𑀢𑁄 𑀫𑀘𑁆𑀙𑁄,_  
_𑀲𑁄 𑀫𑀁 𑀳𑀭𑀢𑀺 𑀯𑁂𑀕𑀲𑀸”𑁇_  


(30.)

1518\. _“𑀕𑀫𑁆𑀪𑀻𑀭𑀘𑀸𑀭𑀻 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀤𑀍𑀳𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺 𑀣𑀸𑀫𑀲𑀸;_  
_𑀅𑀳𑀁 𑀢𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀧𑀡𑁆𑀡𑁄 𑀉𑀭𑀕𑀸𑀫𑀺𑀯”𑁇_  


(31.)

1519\. _“𑀯𑀺𑀯𑀸𑀤𑁄 𑀦𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀤𑀩𑁆𑀪𑀧𑀼𑀧𑁆𑀨 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂;_  
_𑀲𑀫𑁂𑀳𑀺 𑀫𑁂𑀥𑀕𑀁 𑀲𑀫𑁆𑀫,_  
_𑀯𑀺𑀯𑀸𑀤𑁄 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀁”𑁇_  


(32.)

1520\. _“𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄𑀳𑀁 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀩𑀳𑀽 𑀅𑀟𑁆𑀟𑀸 𑀫𑁂 𑀢𑀻𑀭𑀺𑀢𑀸;_  
_𑀲𑀫𑁂𑀫𑀺 𑀫𑁂𑀥𑀕𑀁 𑀲𑀫𑁆𑀫,_  
_𑀯𑀺𑀯𑀸𑀤𑁄 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀁𑁇_  


(33.)

1521\. _𑀅𑀦𑀼𑀢𑀻𑀭𑀘𑀸𑀭𑀺 𑀦𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑀁,_  
_𑀲𑀻𑀲𑀁 𑀕𑀫𑁆𑀪𑀻𑀭𑀘𑀸𑀭𑀺𑀦𑁄;_  
_𑀅𑀘𑁆𑀘𑀸𑀬𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑁄 𑀔𑀡𑁆𑀟𑁄,_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀲𑁆𑀲 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(34.)

1522\. _“𑀘𑀺𑀭𑀫𑁆𑀧𑀺 𑀪𑀓𑁆𑀔𑁄 𑀅𑀪𑀯𑀺𑀲𑁆𑀲,_  
_𑀲𑀘𑁂 𑀦 𑀯𑀺𑀯𑀤𑁂𑀫𑀲𑁂;_  
_𑀅𑀲𑀻𑀲𑀓𑀁 𑀅𑀦𑀗𑁆𑀕𑀼𑀝𑁆𑀞𑀁,_  
_𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀳𑀭𑀢𑀺 𑀭𑁄𑀳𑀺𑀢𑀁”𑁇_  


(35.)

1523\. _“𑀬𑀣𑀸𑀧𑀺 𑀭𑀸𑀚𑀸 𑀦𑀦𑁆𑀤𑁂𑀬𑁆𑀬,_  
_𑀭𑀚𑁆𑀚𑀁 𑀮𑀤𑁆𑀥𑀸𑀦 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀏𑀯𑀸𑀳𑀫𑀚𑁆𑀚 𑀦𑀦𑁆𑀤𑀸𑀫𑀺,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀼𑀡𑁆𑀡𑀫𑀼𑀔𑀁 𑀧𑀢𑀺𑀁𑁇_  


(36.)

1524\. _𑀓𑀣𑀁 𑀦𑀼 𑀣𑀮𑀚𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀉𑀤𑀓𑁂 𑀫𑀘𑁆𑀙𑀁 𑀧𑀭𑀸𑀫𑀲𑀺;_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀫𑁂 𑀲𑀫𑁆𑀫 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀓𑀣𑀁 𑀅𑀥𑀺𑀕𑀢𑀁 𑀢𑀬𑀸”𑁇_  


(37.)

1525\. _“𑀯𑀺𑀯𑀸𑀤𑁂𑀦 𑀓𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀯𑀸𑀤𑁂𑀦 𑀥𑀦𑀓𑁆𑀔𑀬𑀸;_  
_𑀚𑀻𑀦𑀸 𑀉𑀤𑁆𑀤𑀸 𑀯𑀺𑀯𑀸𑀤𑁂𑀦,_  
_𑀪𑀼𑀜𑁆𑀚 𑀫𑀸𑀬𑀸𑀯𑀺 𑀭𑁄𑀳𑀺𑀢𑀁𑁇_  


(38.)

1526\. _𑀏𑀯𑀫𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀯𑀺𑀯𑀸𑀤𑁄 𑀬𑀢𑁆𑀣 𑀚𑀸𑀬𑀢𑀺;_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀁 𑀧𑀝𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺,_  
_𑀲𑁄 𑀳𑀺 𑀦𑁂𑀲𑀁 𑀯𑀺𑀦𑀸𑀬𑀓𑁄;_  
_𑀥𑀦𑀸𑀧𑀺 𑀢𑀢𑁆𑀣 𑀚𑀻𑀬𑀦𑁆𑀢𑀺,_  
_𑀭𑀸𑀚𑀓𑁄𑀲𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀻”𑀢𑀺𑁇_  


---

1527\. 𑀤𑀩𑁆𑀪𑀧𑀼𑀧𑁆𑀨𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
