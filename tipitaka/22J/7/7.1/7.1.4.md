# 𑁭𑁇𑁧𑁇𑁪 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓

### 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀓𑁆𑀓𑀼𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓

(22.)

1509. _“𑀢𑁂 𑀓𑀣𑀁 𑀦𑀼 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀯𑀼𑀤𑁆𑀥𑀸 𑀕𑀺𑀭𑀺𑀤𑀭𑀻𑀲𑀬𑀸;_  
_𑀅𑀳𑀁 𑀩𑀤𑁆𑀥𑁄𑀲𑁆𑀫𑀺 𑀧𑀸𑀲𑁂𑀦,_  
_𑀦𑀺𑀮𑀻𑀬𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀢𑁄”𑁇_  


(23.)

1510. _“𑀓𑀺𑀁 𑀕𑀺𑀚𑁆𑀛 𑀧𑀭𑀺𑀤𑁂𑀯𑀲𑀺,_  
_𑀓𑀸 𑀦𑀼 𑀢𑁂 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑀸;_  
_𑀦 𑀫𑁂 𑀲𑀼𑀢𑁄 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀯𑀸,_  
_𑀪𑀸𑀲𑀦𑁆𑀢𑁄 𑀫𑀸𑀦𑀼𑀲𑀺𑀁 𑀤𑀺𑀚𑁄”𑁇_  


(24.)

1511. _“𑀪𑀭𑀸𑀫𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀯𑀼𑀤𑁆𑀥𑁂 𑀕𑀺𑀭𑀺𑀤𑀭𑀻𑀲𑀬𑁂;_  
_𑀢𑁂 𑀓𑀣𑀁 𑀦𑀼 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀅𑀳𑀁 𑀯𑀲𑀁 𑀕𑀢𑁄 𑀢𑀯”𑁇_  


(25.)

1512. _“𑀬𑀦𑁆𑀦𑀼 𑀕𑀺𑀚𑁆𑀛𑁄 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀓𑀼𑀡𑀧𑀸𑀦𑀺 𑀅𑀯𑁂𑀓𑁆𑀔𑀢𑀺;_  
_𑀓𑀲𑁆𑀫𑀸 𑀚𑀸𑀮𑀜𑁆𑀘 𑀧𑀸𑀲𑀜𑁆𑀘,_  
_𑀆𑀲𑀚𑁆𑀚𑀸𑀧𑀺 𑀦 𑀩𑀼𑀚𑁆𑀛𑀲𑀺”𑁇_  


(26.)

1513. _“𑀬𑀤𑀸 𑀧𑀭𑀸𑀪𑀯𑁄 𑀳𑁄𑀢𑀺,_  
_𑀧𑁄𑀲𑁄 𑀚𑀻𑀯𑀺𑀢𑀲𑀗𑁆𑀔𑀬𑁂;_  
_𑀅𑀣 𑀚𑀸𑀮𑀜𑁆𑀘 𑀧𑀸𑀲𑀜𑁆𑀘,_  
_𑀆𑀲𑀚𑁆𑀚𑀸𑀧𑀺 𑀦 𑀩𑀼𑀚𑁆𑀛𑀢𑀺”𑁇_  


(27.)

1514. _“𑀪𑀭𑀲𑁆𑀲𑀼 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀯𑀼𑀤𑁆𑀥𑁂 𑀕𑀺𑀭𑀺𑀤𑀭𑀻𑀲𑀬𑁂;_  
_𑀫𑀬𑀸 𑀢𑁆𑀯𑀁 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄,_  
_𑀲𑁄𑀢𑁆𑀣𑀺𑀁 𑀧𑀲𑁆𑀲𑀸𑀳𑀺 𑀜𑀸𑀢𑀓𑁂”𑁇_  


(28.)

1515. _“𑀏𑀯𑀁 𑀮𑀼𑀤𑁆𑀤𑀓 𑀦𑀦𑁆𑀤𑀲𑁆𑀲𑀼,_  
_𑀲𑀳 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀜𑀸𑀢𑀺𑀪𑀺;_  
_𑀪𑀭𑀺𑀲𑁆𑀲𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀯𑀼𑀤𑁆𑀥𑁂 𑀕𑀺𑀭𑀺𑀤𑀭𑀻𑀲𑀬𑁂”𑀢𑀺𑁇_  


---

1516. 𑀫𑀸𑀢𑀼𑀧𑁄𑀲𑀓𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
