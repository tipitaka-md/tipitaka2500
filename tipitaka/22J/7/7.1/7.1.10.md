# 𑁭𑁇𑁧𑁇𑁧𑁦 𑀩𑀓𑀚𑀸𑀢𑀓

### 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀓𑁆𑀓𑀼𑀯𑀕𑁆𑀕

### 𑀩𑀓𑀚𑀸𑀢𑀓

(68.)

1561. _“𑀤𑁆𑀯𑀸𑀲𑀢𑁆𑀢𑀢𑀺 𑀕𑁄𑀢𑀫 𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀸,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀚𑀸𑀢𑀺𑀚𑀭𑀁 𑀅𑀢𑀻𑀢𑀸;_  
_𑀅𑀬𑀫𑀦𑁆𑀢𑀺𑀫𑀸 𑀯𑁂𑀤𑀕𑀽 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀢𑁆𑀢𑀺,_  
_𑀅𑀲𑁆𑀫𑀸𑀪𑀺𑀚𑀧𑁆𑀧𑀦𑁆𑀢𑀺 𑀚𑀦𑀸 𑀅𑀦𑁂𑀓𑀸”𑁇_  


(69.)

1562. _“𑀅𑀧𑁆𑀧𑀜𑁆𑀳𑀺 𑀏𑀢𑀁 𑀦 𑀳𑀺 𑀤𑀻𑀖𑀫𑀸𑀬𑀼,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀩𑀓 𑀫𑀜𑁆𑀜𑀲𑀺 𑀤𑀻𑀖𑀫𑀸𑀬𑀼𑀁;_  
_𑀲𑀢𑀁 𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀭𑀩𑁆𑀩𑀼𑀤𑀸𑀦𑀁,_  
_𑀆𑀬𑀼𑀁 𑀧𑀚𑀸𑀦𑀸𑀫𑀺 𑀢𑀯𑀸𑀳 𑀩𑁆𑀭𑀳𑁆𑀫𑁂”𑁇_  


(70.)

1563. _“𑀅𑀦𑀦𑁆𑀢𑀤𑀲𑁆𑀲𑀻 𑀪𑀕𑀯𑀸𑀳𑀫𑀲𑁆𑀫𑀺,_  
_𑀚𑀸𑀢𑀺𑀚𑁆𑀚𑀭𑀁 𑀲𑁄𑀓𑀫𑀼𑀧𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑁄;_  
_𑀓𑀺𑀁 𑀫𑁂 𑀧𑀼𑀭𑀸𑀡𑀁 𑀯𑀢𑀲𑀻𑀮𑀯𑀢𑁆𑀢𑀁,_  
_𑀆𑀘𑀺𑀓𑁆𑀔 𑀫𑁂 𑀢𑀁 𑀬𑀫𑀳𑀁 𑀯𑀺𑀚𑀜𑁆𑀜𑀁”𑁇_  


(71.)

1564. _“𑀬𑀁 𑀢𑁆𑀯𑀁 𑀅𑀧𑀸𑀬𑁂𑀲𑀺 𑀩𑀳𑀽 𑀫𑀦𑀼𑀲𑁆𑀲𑁂,_  
_𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁂 𑀖𑀫𑁆𑀫𑀦𑀺 𑀲𑀫𑁆𑀧𑀭𑁂𑀢𑁂;_  
_𑀢𑀁 𑀢𑁂 𑀧𑀼𑀭𑀸𑀡𑀁 𑀯𑀢𑀲𑀻𑀮𑀯𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄𑀯 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺𑁇_  


(72.)

1565. _𑀬𑀁 𑀏𑀡𑀺𑀓𑀽𑀮𑀲𑁆𑀫𑀺 𑀚𑀦𑀁 𑀕𑀳𑀻𑀢𑀁,_  
_𑀅𑀫𑁄𑀘𑀬𑀻 𑀕𑀬𑁆𑀳𑀓 𑀦𑀺𑀬𑁆𑀬𑀫𑀸𑀦𑀁;_  
_𑀢𑀁 𑀢𑁂 𑀧𑀼𑀭𑀸𑀡𑀁 𑀯𑀢𑀲𑀻𑀮𑀯𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄𑀯 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺𑁇_  


(73.)

1566. _𑀕𑀗𑁆𑀕𑀸𑀬 𑀲𑁄𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀳𑀻𑀢𑀦𑀸𑀯𑀁,_  
_𑀮𑀼𑀤𑁆𑀤𑁂𑀦 𑀦𑀸𑀕𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀧𑁆𑀧𑀸;_  
_𑀅𑀫𑁄𑀘𑀬𑀺 𑀢𑁆𑀯𑀁 𑀩𑀮𑀲𑀸 𑀧𑀲𑀬𑁆𑀳,_  
_𑀢𑀁 𑀢𑁂 𑀧𑀼𑀭𑀸𑀡𑀁 𑀯𑀢𑀲𑀻𑀮𑀯𑀢𑁆𑀢𑀁;_  
_𑀲𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄𑀯 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺𑁇_  


(74.)

1567. _𑀓𑀧𑁆𑀧𑁄 𑀘 𑀢𑁂 𑀩𑀤𑁆𑀥𑀘𑀭𑁄 𑀅𑀳𑁄𑀲𑀺𑀁,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑀁 𑀯𑀢𑀺𑀦𑀁 𑀅𑀫𑀜𑁆𑀜𑀁;_  
_𑀢𑀁 𑀢𑁂 𑀧𑀼𑀭𑀸𑀡𑀁 𑀯𑀢𑀲𑀻𑀮𑀯𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀢𑁆𑀢𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄𑀯 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀫𑀺”𑁇_  


(75.)

1568. _“𑀅𑀤𑁆𑀥𑀸 𑀧𑀚𑀸𑀦𑀸𑀲𑀺 𑀫𑀫𑁂𑀢𑀫𑀸𑀬𑀼𑀁,_  
_𑀅𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀚𑀸𑀦𑀸𑀲𑀺 𑀢𑀣𑀸 𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄;_  
_𑀢𑀣𑀸 𑀳𑀺 𑀢𑀸𑀬𑀁 𑀚𑀮𑀺𑀢𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀑𑀪𑀸𑀲𑀬𑀁 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀦𑁆”𑀢𑀺𑁇_  


---

1569. 𑀩𑀓𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀓𑀼𑀓𑁆𑀓𑀼𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

1570. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1571. _𑀯𑀭𑀓𑀡𑁆𑀡𑀺𑀓 𑀘𑀸𑀧𑀯𑀭𑁄 𑀲𑀼𑀢𑀦𑁄,_  
_𑀅𑀣 𑀕𑀺𑀚𑁆𑀛 𑀲𑀭𑁄𑀳𑀺𑀢𑀫𑀘𑁆𑀙𑀯𑀭𑁄;_  
_𑀧𑀼𑀦 𑀧𑀡𑁆𑀡𑀓 𑀲𑁂𑀦𑀓 𑀬𑀸𑀘𑀦𑀓𑁄,_  
_𑀅𑀣 𑀯𑁂𑀭𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀓𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
