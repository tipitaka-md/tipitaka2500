# 𑁧𑁇𑁧𑁨𑁇𑁧𑁦 𑀩𑀦𑁆𑀥𑀦𑀫𑁄𑀓𑁆𑀔𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀳𑀁𑀘𑀺𑀯𑀕𑁆𑀕

### 𑀩𑀦𑁆𑀥𑀦𑀫𑁄𑀓𑁆𑀔𑀚𑀸𑀢𑀓

(120.)

265\. _“𑀅𑀩𑀤𑁆𑀥𑀸 𑀢𑀢𑁆𑀣 𑀩𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀬𑀢𑁆𑀣 𑀩𑀸𑀮𑀸 𑀧𑀪𑀸𑀲𑀭𑁂;_  
_𑀩𑀤𑁆𑀥𑀸𑀧𑀺 𑀢𑀢𑁆𑀣 𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀬𑀢𑁆𑀣 𑀥𑀻𑀭𑀸 𑀧𑀪𑀸𑀲𑀭𑁂”𑀢𑀺𑁇_  


---

266\. 𑀩𑀦𑁆𑀥𑀦𑀫𑁄𑀓𑁆𑀔𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀳𑀁𑀘𑀺𑀯𑀕𑁆𑀕𑁄 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑁄𑁇



---

267\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



268\. _𑀅𑀣 𑀕𑀤𑁆𑀭𑀪 𑀲𑀢𑁆𑀢𑀼𑀯 𑀓𑀁𑀲𑀲𑀢𑀁,_  
_𑀩𑀳𑀼𑀘𑀺𑀦𑁆𑀢𑀺 𑀲𑀸𑀲𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑀓𑀭;_  
_𑀅𑀢𑀺𑀯𑁂𑀮 𑀯𑀺𑀲𑁂𑀲𑀫𑀦𑀸𑀘𑀭𑀺𑀬𑁄𑀯,_  
_𑀥𑀻𑀭𑀸𑀧𑀪𑀸𑀲𑀭𑀢𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
