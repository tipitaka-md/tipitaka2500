# 𑁧𑁇𑁧𑁨𑁇𑁬 𑀤𑀼𑀩𑁆𑀩𑀘𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀩𑁆𑀩𑀘𑀚𑀸𑀢𑀓

(116.)

257\. _𑀅𑀢𑀺𑀓𑀭𑀫𑀓𑀭𑀸𑀘𑀭𑀺𑀬,_  
_𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑁂𑀢𑀁 𑀦 𑀭𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀮𑀗𑁆𑀖𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀜𑁆𑀘𑀫𑀸𑀬𑀲𑀺 𑀆𑀯𑀼𑀢𑁄”𑀢𑀺𑁇_
