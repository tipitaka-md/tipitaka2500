# 𑁧𑁇𑁧𑁨𑁇𑁨 𑀅𑀫𑀭𑀸𑀤𑁂𑀯𑀻𑀧𑀜𑁆𑀳𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀳𑀁𑀘𑀺𑀯𑀕𑁆𑀕

* 𑀅𑀫𑀭𑀸𑀤𑁂𑀯𑀻𑀧𑀜𑁆𑀳𑀚𑀸𑀢𑀓

(112.)

249\. _“𑀬𑁂𑀦 𑀲𑀢𑁆𑀢𑀼𑀩𑀺𑀮𑀗𑁆𑀕𑀸 𑀘,_  
_𑀤𑀺𑀕𑀼𑀡𑀧𑀮𑀸𑀲𑁄 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁄;_  
_𑀬𑁂𑀦 𑀤𑀤𑀸𑀫𑀺 𑀢𑁂𑀦 𑀯𑀤𑀸𑀫𑀺,_  
_𑀬𑁂𑀦 𑀦 𑀤𑀤𑀸𑀫𑀺 𑀦 𑀢𑁂𑀦 𑀯𑀤𑀸𑀫𑀺;_  
_𑀏𑀲 𑀫𑀕𑁆𑀕𑁄 𑀬𑀯𑀫𑀚𑁆𑀛𑀓𑀲𑁆𑀲,_  
_𑀏𑀢𑀁 𑀙𑀦𑁆𑀦𑀧𑀣𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇_  


---

250\. 𑀅𑀫𑀭𑀸𑀤𑁂𑀯𑀻𑀧𑀜𑁆𑀳𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
