# 𑁧𑁇𑁨𑁇𑁧𑁦 𑀦𑀍𑀅𑀧𑀸𑀦𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀻𑀮𑀯𑀕𑁆𑀕

### 𑀦𑀍𑀅𑀧𑀸𑀦𑀚𑀸𑀢𑀓

(20.)

42\. _“𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀤𑀫𑀦𑀼𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑁄𑀢𑀭𑀺𑀢𑀁 𑀧𑀤𑀁;_  
_𑀦𑀍𑀏𑀦 𑀯𑀸𑀭𑀺𑀁 𑀧𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀦𑁂𑀯 𑀫𑀁 𑀢𑁆𑀯𑀁 𑀯𑀥𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺𑁇_  


---

43\. 𑀦𑀍𑀅𑀧𑀸𑀦𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀲𑀻𑀮𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



---

44\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



45\. _𑀅𑀣 𑀮𑀓𑁆𑀔𑀡 𑀲𑀸𑀔 𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀧𑀼𑀦,_  
_𑀦 𑀓𑀺𑀭𑀢𑁆𑀣𑀺 𑀭𑀲𑁂𑀳𑀺 𑀔𑀭𑀸𑀤𑀺𑀬𑀸;_  
_𑀅𑀢𑀺𑀪𑁄𑀢𑀺 𑀯𑀭 𑀫𑀸𑀮𑀼𑀢 𑀧𑀸𑀡,_  
_𑀫𑀼𑀘𑁆𑀘𑁂𑀦 𑀦𑀍𑀅𑀅𑀯𑁆𑀳𑀬𑀦𑁂𑀦 𑀪𑀯𑀦𑁆𑀢𑀺 𑀤𑀲𑀸𑀢𑀺𑁇_
