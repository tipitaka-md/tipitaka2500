# 𑁧𑁇𑁨𑁇𑁧 𑀮𑀓𑁆𑀔𑀡𑀫𑀺𑀕𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀻𑀮𑀯𑀕𑁆𑀕

### 𑀮𑀓𑁆𑀔𑀡𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(11.)

24\. _“𑀳𑁄𑀢𑀺 𑀲𑀻𑀮𑀯𑀢𑀁 𑀅𑀢𑁆𑀣𑁄,_  
_𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑀁;_  
_𑀮𑀓𑁆𑀔𑀡𑀁 𑀧𑀲𑁆𑀲 𑀆𑀬𑀦𑁆𑀢𑀁,_  
_𑀜𑀸𑀢𑀺𑀲𑀗𑁆𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀁;_  
_𑀅𑀣 𑀧𑀲𑁆𑀲𑀲𑀺𑀫𑀁 𑀓𑀸𑀍𑀅𑀁,_  
_𑀲𑀼𑀯𑀺𑀳𑀻𑀦𑀁𑀯 𑀜𑀸𑀢𑀺𑀪𑀻”𑀢𑀺𑁇_  


---
