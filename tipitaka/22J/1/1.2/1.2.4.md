# 𑁧𑁇𑁨𑁇𑁪 𑀯𑀸𑀢𑀫𑀺𑀕𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀻𑀮𑀯𑀕𑁆𑀕

### 𑀯𑀸𑀢𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(14.)

30\. _“𑀦 𑀓𑀺𑀭𑀢𑁆𑀣𑀺 𑀭𑀲𑁂𑀳𑀺 𑀧𑀸𑀧𑀺𑀬𑁄,_  
_𑀆𑀯𑀸𑀲𑁂𑀳𑀺 𑀯 𑀲𑀦𑁆𑀣𑀯𑁂𑀳𑀺 𑀯𑀸;_  
_𑀯𑀸𑀢𑀫𑀺𑀕𑀁 𑀕𑀳𑀦𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀁,_  
_𑀯𑀲𑀫𑀸𑀦𑁂𑀲𑀺 𑀭𑀲𑁂𑀳𑀺 𑀲𑀜𑁆𑀚𑀬𑁄”𑀢𑀺𑁇_  


---

31\. 𑀯𑀸𑀢𑀫𑀺𑀕𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
