# 𑁧𑁇𑁨𑁇𑁯 𑀆𑀬𑀸𑀘𑀺𑀢𑀪𑀢𑁆𑀢𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀻𑀮𑀯𑀕𑁆𑀕

### 𑀆𑀬𑀸𑀘𑀺𑀢𑀪𑀢𑁆𑀢𑀚𑀸𑀢𑀓

(19.)

40. _“𑀲𑀘𑁂 𑀫𑀼𑀘𑁆𑀘𑁂 𑀧𑁂𑀘𑁆𑀘 𑀫𑀼𑀘𑁆𑀘𑁂,_  
_𑀫𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀳𑀺 𑀩𑀚𑁆𑀛𑀢𑀺;_  
_𑀦 𑀳𑁂𑀯𑀁 𑀥𑀻𑀭𑀸 𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀫𑀼𑀢𑁆𑀢𑀺 𑀩𑀸𑀮𑀲𑁆𑀲 𑀩𑀦𑁆𑀥𑀦𑀦𑁆”𑀢𑀺𑁇_  


---
