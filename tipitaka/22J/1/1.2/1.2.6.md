# 𑁧𑁇𑁨𑁇𑁬 𑀢𑀺𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀫𑀺𑀕𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀻𑀮𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(16.)

34\. _“𑀫𑀺𑀕𑀁 𑀢𑀺𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀫𑀦𑁂𑀓𑀫𑀸𑀬𑀁,_  
_𑀅𑀝𑁆𑀞𑀓𑁆𑀔𑀼𑀭𑀁 𑀅𑀟𑁆𑀠𑀭𑀢𑁆𑀢𑀸𑀧𑀧𑀸𑀬𑀺𑀁;_  
_𑀏𑀓𑁂𑀦 𑀲𑁄𑀢𑁂𑀦 𑀙𑀫𑀸𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑁄,_  
_𑀙𑀳𑀺 𑀓𑀮𑀸𑀳𑀺𑀢𑀺𑀪𑁄𑀢𑀺 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑁄”𑀢𑀺𑁇_  


---

35\. 𑀢𑀺𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀫𑀺𑀕𑀚𑀸𑀢𑀓𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
