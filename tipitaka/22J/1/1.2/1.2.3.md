# 𑁧𑁇𑁨𑁇𑁩 𑀓𑀡𑁆𑀟𑀺𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀻𑀮𑀯𑀕𑁆𑀕

* 𑀓𑀡𑁆𑀟𑀺𑀚𑀸𑀢𑀓

(13.)

28\. _“𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀓𑀡𑁆𑀟𑀺𑀦𑀁 𑀲𑀮𑁆𑀮𑀁,_  
_𑀧𑀼𑀭𑀺𑀲𑀁 𑀕𑀸𑀍𑀳𑀯𑁂𑀥𑀺𑀦𑀁;_  
_𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀢𑀁 𑀚𑀦𑀧𑀤𑀁,_  
_𑀬𑀢𑁆𑀣𑀺𑀢𑁆𑀣𑀻 𑀧𑀭𑀺𑀡𑀸𑀬𑀺𑀓𑀸;_  
_𑀢𑁂 𑀘𑀸𑀧𑀺 𑀥𑀺𑀓𑁆𑀓𑀺𑀢𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀬𑁂 𑀇𑀢𑁆𑀣𑀻𑀦𑀁 𑀯𑀲𑀗𑁆𑀕𑀢𑀸”𑀢𑀺𑁇_
