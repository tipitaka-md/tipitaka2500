# 𑁧𑁇𑁬𑁇𑁬 𑀓𑀜𑁆𑀘𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀆𑀲𑀻𑀲𑀯𑀕𑁆𑀕

### 𑀓𑀜𑁆𑀘𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀚𑀸𑀢𑀓

(56.)

123\. _“𑀬𑁄 𑀧𑀳𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀧𑀳𑀝𑁆𑀞𑀫𑀦𑀲𑁄 𑀦𑀭𑁄;_  
_𑀪𑀸𑀯𑁂𑀢𑀺 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀧𑀸𑀧𑀼𑀡𑁂 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑀦𑁆”𑀢𑀺𑁇_  


---
