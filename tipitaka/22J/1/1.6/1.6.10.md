# 𑁧𑁇𑁬𑁇𑁧𑁦 𑀲𑀗𑁆𑀔𑀥𑀫𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀆𑀲𑀻𑀲𑀯𑀕𑁆𑀕

* 𑀲𑀗𑁆𑀔𑀥𑀫𑀚𑀸𑀢𑀓

(60.)

131\. _“𑀥𑀫𑁂 𑀥𑀫𑁂 𑀦𑀸𑀢𑀺𑀥𑀫𑁂,_  
_𑀅𑀢𑀺𑀥𑀦𑁆𑀢𑀜𑁆𑀳𑀺 𑀧𑀸𑀧𑀓𑀁;_  
_𑀥𑀦𑁆𑀢𑁂𑀦𑀸𑀥𑀺𑀕𑀢𑀸 𑀪𑁄𑀕𑀸,_  
_𑀢𑁂 𑀢𑀸𑀢𑁄 𑀯𑀺𑀥𑀫𑀻 𑀥𑀫𑀦𑁆”𑀢𑀺𑁇_  


---

132\. 𑀲𑀗𑁆𑀔𑀥𑀫𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀆𑀲𑀻𑀲𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇



---

133\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



134\. _𑀬𑀣𑀸 𑀇𑀘𑁆𑀙𑀺𑀁 𑀢𑀣𑀸𑀳𑀼𑀤𑀓𑀸 𑀣𑀮𑀸,_  
_𑀲𑀼𑀭 𑀲𑀸𑀤𑀼𑀨𑀮𑁄 𑀘 𑀅𑀮𑀻𑀦𑀫𑀦𑁄;_  
_𑀲𑀫𑁆𑀧𑀳𑀝𑁆𑀞𑀫𑀦𑁄 𑀘𑀢𑀼𑀭𑁄 𑀘 𑀢𑀬𑁄,_  
_𑀲𑀢𑀮𑀤𑁆𑀥𑀓 𑀪𑁄𑀕𑀥𑀦𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
