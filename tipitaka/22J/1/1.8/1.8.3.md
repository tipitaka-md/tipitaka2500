# 𑁧𑁇𑁮𑁇𑁩 𑀲𑀘𑁆𑀘𑀁𑀓𑀺𑀭𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀯𑀭𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀲𑀘𑁆𑀘𑀁𑀓𑀺𑀭𑀚𑀸𑀢𑀓

(73.)

161\. _“𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭𑁂𑀯𑀫𑀸𑀳𑀁𑀲𑀼,_  
_𑀦𑀭𑀸 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑀸 𑀇𑀥;_  
_𑀓𑀝𑁆𑀞𑀁 𑀦𑀺𑀧𑁆𑀮𑀯𑀺𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀦 𑀢𑁆𑀯𑁂𑀯𑁂𑀓𑀘𑁆𑀘𑀺𑀬𑁄 𑀦𑀭𑁄”𑀢𑀺𑁇_
