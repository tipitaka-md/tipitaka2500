# 𑁧𑁇𑁮𑁇𑁬 𑀅𑀲𑀁𑀓𑀺𑀬𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀯𑀭𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀅𑀲𑀁𑀓𑀺𑀬𑀚𑀸𑀢𑀓

(76.)

167. _“𑀅𑀲𑀗𑁆𑀓𑀺𑀬𑁄𑀫𑁆𑀳𑀺 𑀕𑀸𑀫𑀫𑁆𑀳𑀺,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀪𑀬𑀁;_  
_𑀉𑀚𑀼𑀁 𑀫𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀭𑀽𑀍𑀳𑁄,_  
_𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀓𑀭𑀼𑀡𑀸𑀬 𑀘𑀸”𑀢𑀺𑁇_
