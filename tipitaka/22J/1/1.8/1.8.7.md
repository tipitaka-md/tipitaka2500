# 𑁧𑁇𑁮𑁇𑁭 𑀫𑀳𑀸𑀲𑀼𑀧𑀺𑀦𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀯𑀭𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀲𑀼𑀧𑀺𑀦𑀚𑀸𑀢𑀓

(77.)

169\. _“𑀉𑀲𑀪𑀸 𑀭𑀼𑀓𑁆𑀔𑀸 𑀕𑀸𑀯𑀺𑀬𑁄 𑀕𑀯𑀸 𑀘,_  
_𑀅𑀲𑁆𑀲𑁄 𑀓𑀁𑀲𑁄 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀻 𑀘 𑀓𑀼𑀫𑁆𑀪𑁄;_  
_𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀘 𑀅𑀧𑀸𑀓𑀘𑀦𑁆𑀤𑀦𑀁,_  
_𑀮𑀸𑀩𑀽𑀦𑀺 𑀲𑀻𑀤𑀦𑁆𑀢𑀺 𑀲𑀺𑀮𑀸 𑀧𑁆𑀮𑀯𑀦𑁆”𑀢𑀺𑁇_  


170\. _𑀫𑀡𑁆𑀟𑀽𑀓𑀺𑀬𑁄 𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑁂 𑀕𑀺𑀮𑀦𑁆𑀢𑀺,_  
_𑀓𑀸𑀓𑀁 𑀲𑀼𑀧𑀡𑁆𑀡𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀬𑀦𑁆𑀢𑀺;_  
_𑀢𑀲𑀸 𑀯𑀓𑀸 𑀏𑀍𑀅𑀓𑀸𑀦𑀁 𑀪𑀬𑀸𑀳𑀺,_  
_𑀯𑀺𑀧𑀭𑀺𑀬𑀸𑀲𑁄 𑀯𑀢𑁆𑀢𑀢𑀺 𑀦𑀬𑀺𑀥 𑀫𑀢𑁆𑀣𑀻𑀢𑀺𑁇_  


---
