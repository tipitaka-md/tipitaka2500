# 𑁧𑁇𑁧𑁩𑁇𑁮 𑀩𑀺𑀍𑀆𑀭𑀯𑀢𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀲𑀦𑀸𑀍𑀇𑀯𑀕𑁆𑀕

### 𑀩𑀺𑀍𑀆𑀭𑀯𑀢𑀚𑀸𑀢𑀓

(128.)

283\. _“𑀬𑁄 𑀯𑁂 𑀥𑀫𑁆𑀫𑀁 𑀥𑀚𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀕𑀽𑀍𑀳𑁄 𑀧𑀸𑀧𑀫𑀸𑀘𑀭𑁂;_  
_𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀩𑀺𑀍𑀆𑀭𑀁 𑀦𑀸𑀫 𑀢𑀁 𑀯𑀢𑀦𑁆”𑀢𑀺𑁇_
