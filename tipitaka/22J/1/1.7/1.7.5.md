# 𑁧𑁇𑁭𑁇𑁫 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀚𑀸𑀢𑀓

(65.)

143. _“𑀬𑀣𑀸 𑀦𑀤𑀻 𑀘 𑀧𑀦𑁆𑀣𑁄 𑀘,_  
_𑀧𑀸𑀦𑀸𑀕𑀸𑀭𑀁 𑀲𑀪𑀸 𑀧𑀧𑀸;_  
_𑀏𑀯𑀁 𑀮𑁄𑀓𑀺𑀢𑁆𑀣𑀺𑀬𑁄 𑀦𑀸𑀫,_  
_𑀦𑀸𑀲𑀁 𑀓𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸”𑀢𑀺𑁇_
