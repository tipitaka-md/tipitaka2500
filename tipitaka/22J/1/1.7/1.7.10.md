# 𑁧𑁇𑁭𑁇𑁧𑁦 𑀓𑀼𑀤𑁆𑀤𑀸𑀮𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀇𑀢𑁆𑀣𑀺𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀤𑁆𑀤𑀸𑀮𑀚𑀸𑀢𑀓

(70.)

153\. _“𑀦 𑀢𑀁 𑀚𑀺𑀢𑀁 𑀲𑀸𑀥𑀼 𑀚𑀺𑀢𑀁,_  
_𑀬𑀁 𑀚𑀺𑀢𑀁 𑀅𑀯𑀚𑀻𑀬𑀢𑀺;_  
_𑀢𑀁 𑀔𑁄 𑀚𑀺𑀢𑀁 𑀲𑀸𑀥𑀼 𑀚𑀺𑀢𑀁,_  
_𑀬𑀁 𑀚𑀺𑀢𑀁 𑀦𑀸𑀯𑀚𑀻𑀬𑀢𑀻”𑀢𑀺𑁇_  


---

154\. 𑀓𑀼𑀤𑁆𑀤𑀸𑀮𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀇𑀢𑁆𑀣𑀺𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇



---

155\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



156\. _𑀲𑀺𑀔𑀻𑀲𑀩𑁆𑀩𑀖𑀲𑁄𑀧𑀺 𑀘 𑀯𑀻𑀡𑀯𑀭𑁄,_  
_𑀧𑀺𑀲𑀼𑀡𑀸 𑀫𑀺𑀢𑁆𑀢𑀪𑁂𑀤𑀺𑀓𑀸 𑀦𑀦𑁆𑀤𑀻 𑀦𑀤𑀻;_  
_𑀫𑀼𑀤𑀼𑀮𑀓𑁆𑀔𑀡 𑀲𑁄𑀤𑀭𑀺𑀬𑀸 𑀘 𑀫𑀦𑁄,_  
_𑀯𑀺𑀲 𑀲𑀸𑀥𑀼𑀚𑀺𑀢𑁂𑀦 𑀪𑀯𑀦𑁆𑀢𑀺 𑀤𑀲𑀸𑀢𑀺𑁇_
