# 𑁧𑁇𑁭𑁇𑁭 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀕𑁆𑀕

### 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀚𑀸𑀢𑀓

(67.)

147. _“𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂 𑀤𑁂𑀯 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀣𑁂 𑀥𑀸𑀯𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀢𑀺;_  
_𑀢𑀜𑁆𑀘 𑀤𑁂𑀲𑀁 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀢𑁄 𑀲𑁄𑀤𑀭𑀺𑀬𑀫𑀸𑀦𑀬𑁂𑀢𑀺𑁇_
