# 𑁧𑁇𑁭𑁇𑁮 𑀲𑀸𑀓𑁂𑀢𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀇𑀢𑁆𑀣𑀺𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀓𑁂𑀢𑀚𑀸𑀢𑀓

(68.)

149. _𑀬𑀲𑁆𑀫𑀺𑀁 𑀫𑀦𑁄 𑀦𑀺𑀯𑀺𑀲𑀢𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑀸𑀧𑀺 𑀧𑀲𑀻𑀤𑀢𑀺;_  
_𑀅𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀓𑁂 𑀧𑁄𑀲𑁂,_  
_𑀓𑀸𑀫𑀁 𑀢𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀯𑀺𑀲𑁆𑀲𑀲𑁂”𑀢𑀺𑁇_
