# 𑁧𑁇𑁫𑁇𑁬 𑀆𑀭𑀸𑀫𑀤𑀽𑀲𑀓𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀆𑀭𑀸𑀫𑀤𑀽𑀲𑀓𑀚𑀸𑀢𑀓

(46.)

100. _“𑀦 𑀯𑁂 𑀅𑀦𑀢𑁆𑀣𑀓𑀼𑀲𑀮𑁂𑀦,_  
_𑀅𑀢𑁆𑀣𑀘𑀭𑀺𑀬𑀸 𑀲𑀼𑀔𑀸𑀯𑀳𑀸;_  
_𑀳𑀸𑀧𑁂𑀢𑀺 𑀅𑀢𑁆𑀣𑀁 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑁄,_  
_𑀓𑀧𑀺 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀬𑀣𑀸”𑀢𑀺𑁇_
