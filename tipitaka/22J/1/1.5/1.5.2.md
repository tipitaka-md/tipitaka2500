# 𑁧𑁇𑁫𑁇𑁨 𑀓𑀧𑁄𑀢𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀓𑀧𑁄𑀢𑀚𑀸𑀢𑀓

(42.)

92\. _“𑀬𑁄 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀲𑁆𑀲 𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄,_  
_𑀑𑀯𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀦 𑀓𑀭𑁄𑀢𑀺 𑀲𑀸𑀲𑀦𑀁;_  
_𑀓𑀧𑁄𑀢𑀓𑀲𑁆𑀲 𑀯𑀘𑀦𑀁 𑀅𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀢𑁆𑀣𑀕𑀢𑁄𑀯 𑀲𑁂𑀢𑀻”𑀢𑀺𑁇_  


---

93\. 𑀓𑀧𑁄𑀢𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
