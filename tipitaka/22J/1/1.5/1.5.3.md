# 𑁧𑁇𑁫𑁇𑁩 𑀯𑁂𑀍𑀉𑀓𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀍𑀉𑀓𑀚𑀸𑀢𑀓

(43.)

94. _“𑀬𑁄 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀲𑁆𑀲 𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄,_  
_𑀑𑀯𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀦 𑀓𑀭𑁄𑀢𑀺 𑀲𑀸𑀲𑀦𑀁;_  
_𑀏𑀯𑀁 𑀲𑁄 𑀦𑀺𑀳𑀢𑁄 𑀲𑁂𑀢𑀺,_  
_𑀯𑁂𑀍𑀉𑀓𑀲𑁆𑀲 𑀬𑀣𑀸 𑀧𑀺𑀢𑀸”𑀢𑀺𑁇_
