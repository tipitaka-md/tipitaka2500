# 𑁧𑁇𑁫𑁇𑁪 𑀫𑀓𑀲𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀫𑀓𑀲𑀚𑀸𑀢𑀓

(44.)

96\. _“𑀲𑁂𑀬𑁆𑀬𑁄 𑀅𑀫𑀺𑀢𑁆𑀢𑁄 𑀫𑀢𑀺𑀬𑀸 𑀉𑀧𑁂𑀢𑁄,_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀫𑀺𑀢𑁆𑀢𑁄 𑀫𑀢𑀺𑀯𑀺𑀧𑁆𑀧𑀳𑀻𑀦𑁄;_  
_𑀫𑀓𑀲𑀁 𑀯𑀥𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀳𑀺 𑀏𑀍𑀅𑀫𑀽𑀕𑁄,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀺𑀢𑀼 𑀅𑀩𑁆𑀪𑀺𑀤𑀸 𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑀦𑁆”𑀢𑀺𑁇_  


---
