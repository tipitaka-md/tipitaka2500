# 𑁧𑁇𑁫𑁇𑁫 𑀭𑁄𑀳𑀺𑀡𑀺𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀯𑀕𑁆𑀕

### 𑀭𑁄𑀳𑀺𑀡𑀺𑀚𑀸𑀢𑀓

(45.)

98\. _“𑀲𑁂𑀬𑁆𑀬𑁄 𑀅𑀫𑀺𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀬𑀜𑁆𑀘𑁂 𑀩𑀸𑀮𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄;_  
_𑀧𑀲𑁆𑀲 𑀭𑁄𑀳𑀺𑀡𑀺𑀓𑀁 𑀚𑀫𑁆𑀫𑀺𑀁,_  
_𑀫𑀸𑀢𑀭𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑁄𑀘𑀢𑀻”𑀢𑀺𑁇_  


---
