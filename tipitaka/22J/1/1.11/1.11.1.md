# 𑁧𑁇𑁧𑁧𑁇𑁧 𑀧𑀭𑁄𑀲𑀢𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀭𑁄𑀲𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀭𑁄𑀲𑀢𑀚𑀸𑀢𑀓

(101.)

225. _“𑀧𑀭𑁄𑀲𑀢𑀜𑁆𑀘𑁂𑀧𑀺 𑀲𑀫𑀸𑀕𑀢𑀸𑀦𑀁,_  
_𑀛𑀸𑀬𑁂𑀬𑁆𑀬𑀼𑀁 𑀢𑁂 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀅𑀧𑀜𑁆𑀜𑀸;_  
_𑀏𑀓𑁄𑀯 𑀲𑁂𑀬𑁆𑀬𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀧𑀜𑁆𑀜𑁄,_  
_𑀬𑁄 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀅𑀢𑁆𑀣𑀦𑁆”𑀢𑀺𑁇_
