# 𑁧𑁇𑁧𑁧𑁇𑁪 𑀫𑀺𑀢𑁆𑀢𑀯𑀺𑀦𑁆𑀤𑀓𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀭𑁄𑀲𑀢𑀯𑀕𑁆𑀕

### 𑀫𑀺𑀢𑁆𑀢𑀯𑀺𑀦𑁆𑀤𑀓𑀚𑀸𑀢𑀓

(104.)

231\. _“𑀘𑀢𑀼𑀩𑁆𑀪𑀺 𑀅𑀝𑁆𑀞𑀚𑁆𑀛𑀕𑀫𑀸,_  
_𑀅𑀝𑁆𑀞𑀸𑀳𑀺𑀧𑀺 𑀘 𑀲𑁄𑀍𑀅𑀲;_  
_𑀲𑁄𑀍𑀅𑀲𑀸𑀳𑀺 𑀘 𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲,_  
_𑀅𑀢𑁆𑀭𑀺𑀘𑁆𑀙𑀁 𑀘𑀓𑁆𑀓𑀫𑀸𑀲𑀤𑁄;_  
_𑀇𑀘𑁆𑀙𑀸𑀳𑀢𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲,_  
_𑀘𑀓𑁆𑀓𑀁 𑀪𑀫𑀢𑀺 𑀫𑀢𑁆𑀣𑀓𑁂”𑀢𑀺𑁇_  


---
