# 𑁧𑁇𑁪𑁇𑁨 𑀦𑀘𑁆𑀘𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀮𑀸𑀯𑀓𑀯𑀕𑁆𑀕

* 𑀦𑀘𑁆𑀘𑀚𑀸𑀢𑀓

(32.)

70\. _“𑀭𑀼𑀤𑀁 𑀫𑀦𑀼𑀜𑁆𑀜𑀁 𑀭𑀼𑀘𑀺𑀭𑀸 𑀘 𑀧𑀺𑀝𑁆𑀞𑀺,_  
_𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀯𑀡𑁆𑀡𑀽𑀧𑀦𑀺𑀪𑀸 𑀘 𑀕𑀻𑀯𑀸;_  
_𑀩𑁆𑀬𑀸𑀫𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀧𑁂𑀔𑀼𑀡𑀸𑀦𑀺,_  
_𑀦𑀘𑁆𑀘𑁂𑀦 𑀢𑁂 𑀥𑀻𑀢𑀭𑀁 𑀦𑁄 𑀤𑀤𑀸𑀫𑀻”𑀢𑀺𑁇_
