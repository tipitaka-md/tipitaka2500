# 𑁧𑁇𑁩𑁇𑁨 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀚𑀸𑀢𑀓

(22.)

48\. _“𑀬𑁂 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀸 𑀭𑀸𑀚𑀓𑀼𑀮𑀫𑁆𑀳𑀺 𑀯𑀤𑁆𑀥𑀸,_  
_𑀓𑁄𑀮𑁂𑀬𑁆𑀬𑀓𑀸 𑀯𑀡𑁆𑀡𑀩𑀮𑀽𑀧𑀧𑀦𑁆𑀦𑀸;_  
_𑀢𑁂𑀫𑁂 𑀦 𑀯𑀚𑁆𑀛𑀸 𑀫𑀬𑀫𑀲𑁆𑀫 𑀯𑀚𑁆𑀛𑀸,_  
_𑀦𑀸𑀬𑀁 𑀲𑀖𑀘𑁆𑀘𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑀖𑀸𑀢𑀺𑀓𑀸𑀬𑀦𑁆”𑀢𑀺𑁇_
