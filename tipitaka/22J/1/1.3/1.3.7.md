# 𑁧𑁇𑁩𑁇𑁭 𑀅𑀪𑀺𑀡𑁆𑀳𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀅𑀪𑀺𑀡𑁆𑀳𑀚𑀸𑀢𑀓

(27.)

58\. _“𑀦𑀸𑀮𑀁 𑀓𑀩𑀍𑀅𑀁 𑀧𑀤𑀸𑀢𑀯𑁂,_  
_𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀁 𑀦 𑀓𑀼𑀲𑁂 𑀦 𑀖𑀁𑀲𑀺𑀢𑀼𑀁;_  
_𑀫𑀜𑁆𑀜𑀸𑀫𑀺 𑀅𑀪𑀺𑀡𑁆𑀳𑀤𑀲𑁆𑀲𑀦𑀸,_  
_𑀦𑀸𑀕𑁄 𑀲𑁆𑀦𑁂𑀳𑀫𑀓𑀸𑀲𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑁂”𑀢𑀺𑁇_  


---
