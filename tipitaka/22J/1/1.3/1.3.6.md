# 𑁧𑁇𑁩𑁇𑁬 𑀫𑀳𑀺𑀍𑀆𑀫𑀼𑀔𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀺𑀍𑀆𑀫𑀼𑀔𑀚𑀸𑀢𑀓

(26.)

56\. _“𑀧𑀼𑀭𑀸𑀡𑀘𑁄𑀭𑀸𑀦 𑀯𑀘𑁄 𑀦𑀺𑀲𑀫𑁆𑀫,_  
_𑀫𑀳𑀺𑀍𑀆𑀫𑀼𑀔𑁄 𑀧𑁄𑀣𑀬𑀫𑀦𑁆𑀯𑀘𑀸𑀭𑀻;_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀢𑀸𑀦𑀜𑁆𑀳𑀺 𑀯𑀘𑁄 𑀦𑀺𑀲𑀫𑁆𑀫,_  
_𑀕𑀚𑀼𑀢𑁆𑀢𑀫𑁄 𑀲𑀩𑁆𑀩𑀕𑀼𑀡𑁂𑀲𑀼 𑀅𑀝𑁆𑀞𑀸”𑀢𑀺𑁇_
