# 𑁧𑁇𑁩𑁇𑁧𑁦 𑀫𑀼𑀦𑀺𑀓𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀫𑀼𑀦𑀺𑀓𑀚𑀸𑀢𑀓

(30.)

64\. _“𑀫𑀸 𑀫𑀼𑀦𑀺𑀓𑀲𑁆𑀲 𑀧𑀺𑀳𑀬𑀺,_  
_𑀆𑀢𑀼𑀭𑀦𑁆𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀢𑀺;_  
_𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀪𑀼𑀲𑀁 𑀔𑀸𑀤,_  
_𑀏𑀢𑀁 𑀤𑀻𑀖𑀸𑀬𑀼𑀮𑀓𑁆𑀔𑀡𑀦𑁆”𑀢𑀺𑁇_  


---

65\. 𑀫𑀼𑀦𑀺𑀓𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

66\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



67\. _𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀕𑁄𑀚𑀯𑀭𑁄,_  
_𑀧𑀼𑀦 𑀯𑀸𑀍𑀅𑀯𑀲𑁆𑀲𑀲𑀺𑀭𑀺𑀯𑁆𑀳𑀬𑀦𑁄;_  
_𑀫𑀳𑀺𑀍𑀆𑀫𑀼𑀔𑀦𑀸𑀫𑀦𑀼𑀜𑁆𑀜𑀯𑀭𑁄,_  
_𑀯𑀳𑀢𑁂 𑀥𑀼𑀭 𑀫𑀼𑀦𑀺𑀓𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
