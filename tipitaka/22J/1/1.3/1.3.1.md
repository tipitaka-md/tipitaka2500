# 𑁧𑁇𑁩𑁇𑁧 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀫𑀺𑀕𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(21.)

46\. _“𑀜𑀸𑀢𑀫𑁂𑀢𑀁 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀲𑁆𑀲,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑁂𑀧𑀡𑁆𑀡𑀺 𑀲𑀺𑀬𑁆𑀬𑀲𑀺;_  
_𑀅𑀜𑁆𑀜𑀁 𑀲𑁂𑀧𑀡𑁆𑀡𑀺 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀦 𑀫𑁂 𑀢𑁂 𑀭𑀼𑀘𑁆𑀘𑀢𑁂 𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


---
