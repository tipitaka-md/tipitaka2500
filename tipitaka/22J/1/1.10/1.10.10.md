# 𑁧𑁇𑁧𑁦𑁇𑁧𑁦 𑀅𑀲𑀸𑀢𑀭𑀽𑀧𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀮𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀅𑀲𑀸𑀢𑀭𑀽𑀧𑀚𑀸𑀢𑀓

(100.)

220\. _“𑀅𑀲𑀸𑀢𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑁂𑀦,_  
_𑀧𑀺𑀬𑀭𑀽𑀧𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀬𑀁;_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀭𑀽𑀧𑁂𑀦,_  
_𑀧𑀫𑀢𑁆𑀢𑀫𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇_  


---

221\. 𑀅𑀲𑀸𑀢𑀭𑀽𑀧𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀮𑀺𑀢𑁆𑀢𑀯𑀕𑁆𑀕𑁄 𑀤𑀲𑀫𑁄𑁇



---

222\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



223\. _𑀕𑀺𑀮𑀫𑀓𑁆𑀔𑀓𑀼𑀢𑀽𑀳𑀮 𑀫𑀸𑀢𑀼𑀓𑀲𑁆𑀲𑀸,_  
_𑀫𑀼𑀦𑀺𑀦𑀸 𑀘 𑀅𑀦𑀺𑀘𑁆𑀘𑀢 𑀧𑀢𑁆𑀢𑀯𑀭𑀁;_  
_𑀥𑀦𑀧𑀸𑀮𑀺𑀯𑀭𑁄 𑀅𑀢𑀺𑀧𑀡𑁆𑀟𑀺𑀢𑀓𑁄,_  
_𑀲𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀅𑀲𑀸𑀢𑀤𑀲𑀸𑀢𑀺𑁇_  


---
