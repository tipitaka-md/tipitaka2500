# 𑁧𑁇𑁧𑁇𑁨 𑀯𑀡𑁆𑀡𑀼𑀧𑀣𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀧𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕

* 𑀯𑀡𑁆𑀡𑀼𑀧𑀣𑀚𑀸𑀢𑀓

(2.)

4\. _“𑀅𑀓𑀺𑀮𑀸𑀲𑀼𑀦𑁄 𑀯𑀡𑁆𑀡𑀼𑀧𑀣𑁂 𑀔𑀡𑀦𑁆𑀢𑀸,_  
_𑀉𑀤𑀗𑁆𑀕𑀡𑁂 𑀢𑀢𑁆𑀣 𑀧𑀧𑀁 𑀅𑀯𑀺𑀦𑁆𑀤𑀼𑀁;_  
_𑀏𑀯𑀁 𑀫𑀼𑀦𑀻 𑀯𑀺𑀭𑀺𑀬𑀩𑀮𑀽𑀧𑀧𑀦𑁆𑀦𑁄,_  
_𑀅𑀓𑀺𑀮𑀸𑀲𑀼 𑀯𑀺𑀦𑁆𑀤𑁂 𑀳𑀤𑀬𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀦𑁆”𑀢𑀺𑁇_  


---
