# 𑁧𑁇𑁧𑁇𑁮 𑀕𑀸𑀫𑀡𑀺𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀧𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀸𑀫𑀡𑀺𑀚𑀸𑀢𑀓

(8.)

16\. _“𑀅𑀧𑀺 𑀅𑀢𑀭𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀨𑀮𑀸𑀲𑀸𑀯 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀯𑀺𑀧𑀓𑁆𑀓𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁄𑀲𑁆𑀫𑀺,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀕𑀸𑀫𑀡𑀻”𑀢𑀺𑁇_  


---
