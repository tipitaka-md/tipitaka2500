# 𑁧𑁇𑁧𑁇𑁬 𑀤𑁂𑀯𑀥𑀫𑁆𑀫𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀧𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕

### 𑀤𑁂𑀯𑀥𑀫𑁆𑀫𑀚𑀸𑀢𑀓

(6.)

12. _“𑀳𑀺𑀭𑀺𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀼𑀓𑁆𑀓𑀥𑀫𑁆𑀫𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀲𑀦𑁆𑀢𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸 𑀮𑁄𑀓𑁂,_  
_𑀤𑁂𑀯𑀥𑀫𑁆𑀫𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀭𑁂”𑀢𑀺𑁇_
