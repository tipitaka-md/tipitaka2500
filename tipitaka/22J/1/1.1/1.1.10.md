# 𑁧𑁇𑁧𑁇𑁧𑁦 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀧𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀚𑀸𑀢𑀓

(10.)

20\. _“𑀬𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑁂 𑀦 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀘 𑀅𑀜𑁆𑀜𑁂 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺;_  
_𑀲 𑀯𑁂 𑀭𑀸𑀚 𑀲𑀼𑀔𑀁 𑀲𑁂𑀢𑀺,_  
_𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀯𑀸”𑀢𑀺𑁇_  


---

21\. 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀺𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀅𑀧𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

22\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



23\. _𑀯𑀭𑀸𑀧𑀡𑁆𑀡𑀓 𑀯𑀡𑁆𑀡𑀼𑀧𑀣 𑀲𑁂𑀭𑀺𑀯𑀭𑁄,_  
_𑀲𑀼𑀯𑀺𑀘𑀓𑁆𑀔𑀡 𑀢𑀡𑁆𑀟𑀼𑀮𑀦𑀸𑀍𑀇𑀓𑀲𑁆𑀲𑀸;_  
_𑀳𑀺𑀭𑀺 𑀧𑀼𑀢𑁆𑀢𑀯𑀭𑀼𑀢𑁆𑀢𑀕𑀸𑀫𑀡𑀺𑀦𑀸,_  
_𑀬𑁄 𑀘 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺 𑀢𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
