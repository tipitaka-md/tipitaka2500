# 𑁧𑁇𑁧𑁇𑁫 𑀢𑀡𑁆𑀟𑀼𑀮𑀦𑀸𑀍𑀇𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀧𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕

### 𑀢𑀡𑁆𑀟𑀼𑀮𑀦𑀸𑀍𑀇𑀚𑀸𑀢𑀓

(5.)

10. _“𑀓𑀺𑀫𑀕𑁆𑀖𑀢𑀺 𑀢𑀡𑁆𑀟𑀼𑀮𑀦𑀸𑀍𑀇𑀓𑀸𑀬𑀁,_  
_𑀅𑀲𑁆𑀲𑀸𑀦 𑀫𑀽𑀮𑀸𑀬 𑀯𑀤𑁂𑀳𑀺 𑀭𑀸𑀚;_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀁 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀦𑁆𑀢𑀁,_  
_𑀅𑀬𑀫𑀕𑁆𑀖𑀢𑀺 𑀢𑀡𑁆𑀟𑀼𑀮𑀦𑀸𑀍𑀇𑀓𑀸”𑀢𑀺𑁇_  


---

11. 𑀢𑀡𑁆𑀟𑀼𑀮𑀦𑀸𑀍𑀇𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
