# 𑁧𑁇𑁧𑁪𑁇𑁧𑁦 𑀓𑀸𑀓𑀚𑀸𑀢𑀓

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀓𑀸𑀓𑀚𑀸𑀢𑀓

(140.)

309\. _“𑀦𑀺𑀘𑁆𑀘𑀁 𑀉𑀩𑁆𑀩𑀺𑀕𑁆𑀕𑀳𑀤𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀯𑀺𑀳𑁂𑀲𑀓𑀸;_  
_𑀢𑀲𑁆𑀫𑀸 𑀦𑁂𑀲𑀁 𑀯𑀲𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀓𑀸𑀓𑀸𑀦𑀫𑁆𑀳𑀸𑀓 𑀜𑀸𑀢𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

310\. 𑀓𑀸𑀓𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀅𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀯𑀕𑁆𑀕𑁄 𑀘𑀼𑀤𑁆𑀤𑀲𑀫𑁄𑁇



---

311\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



312\. _𑀇𑀢𑀭𑀻𑀢𑀭 𑀭𑀓𑁆𑀔𑀲𑀺 𑀔𑁂𑀫𑀺𑀬𑁄 𑀘,_  
_𑀧𑀭𑁄𑀲𑀢𑀧𑀜𑁆𑀳𑁂𑀦 𑀆𑀪𑀲𑁆𑀲𑀭𑁄 𑀧𑀼𑀦;_  
_𑀅𑀣 𑀳𑀁𑀲𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀩𑀩𑁆𑀩𑀼𑀚𑀝𑀁,_  
_𑀧𑀝𑀦𑀝𑁆𑀞𑀓 𑀓𑀸𑀓𑀯𑀭𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
