# 𑁧𑁇𑁧𑁪𑁇𑁩 𑀖𑀢𑀸𑀲𑀦𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀖𑀢𑀸𑀲𑀦𑀚𑀸𑀢𑀓

(133.)

295. _“𑀔𑁂𑀫𑀁 𑀬𑀳𑀺𑀁 𑀢𑀢𑁆𑀣 𑀅𑀭𑀻 𑀉𑀤𑀻𑀭𑀺𑀢𑁄,_  
_𑀤𑀓𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂 𑀚𑀮𑀢𑁂 𑀖𑀢𑀸𑀲𑀦𑁄;_  
_𑀦 𑀅𑀚𑁆𑀚 𑀯𑀸𑀲𑁄 𑀫𑀳𑀺𑀬𑀸 𑀫𑀳𑀻𑀭𑀼𑀳𑁂,_  
_𑀤𑀺𑀲𑀸 𑀪𑀚𑀯𑁆𑀳𑁄 𑀲𑀭𑀡𑀸𑀚𑁆𑀚 𑀦𑁄 𑀪𑀬𑀦𑁆”𑀢𑀺𑁇_
