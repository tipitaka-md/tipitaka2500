# 𑁧𑁇𑁯𑁇𑁪 𑀅𑀢𑁆𑀣𑀲𑁆𑀲𑀤𑁆𑀯𑀸𑀭𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀧𑀸𑀬𑀺𑀫𑁆𑀳𑀯𑀕𑁆𑀕

### 𑀅𑀢𑁆𑀣𑀲𑁆𑀲𑀤𑁆𑀯𑀸𑀭𑀚𑀸𑀢𑀓

(84.)

186. _“𑀆𑀭𑁄𑀕𑁆𑀬𑀫𑀺𑀘𑁆𑀙𑁂 𑀧𑀭𑀫𑀜𑁆𑀘 𑀮𑀸𑀪𑀁,_  
_𑀲𑀻𑀮𑀜𑁆𑀘 𑀯𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀫𑀢𑀁 𑀲𑀼𑀢𑀜𑁆𑀘;_  
_𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀻 𑀘 𑀅𑀮𑀻𑀦𑀢𑀸 𑀘,_  
_𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀭𑀸 𑀧𑀫𑀼𑀔𑀸 𑀙𑀍𑀏𑀢𑁂”𑀢𑀺𑁇_  


---

187. 𑀅𑀢𑁆𑀣𑀲𑁆𑀲𑀤𑁆𑀯𑀸𑀭𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
