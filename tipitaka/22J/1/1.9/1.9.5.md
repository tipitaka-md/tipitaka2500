# 𑁧𑁇𑁯𑁇𑁫 𑀓𑀺𑀁𑀧𑀓𑁆𑀓𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀧𑀸𑀬𑀺𑀫𑁆𑀳𑀯𑀕𑁆𑀕

### 𑀓𑀺𑀁𑀧𑀓𑁆𑀓𑀚𑀸𑀢𑀓

(85.)

188. _“𑀆𑀬𑀢𑀺𑀁 𑀤𑁄𑀲𑀁 𑀦𑀸𑀜𑁆𑀜𑀸𑀬,_  
_𑀬𑁄 𑀓𑀸𑀫𑁂 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺;_  
_𑀯𑀺𑀧𑀸𑀓𑀦𑁆𑀢𑁂 𑀳𑀦𑀦𑁆𑀢𑀺 𑀦𑀁,_  
_𑀓𑀺𑀁𑀧𑀓𑁆𑀓𑀫𑀺𑀯 𑀪𑀓𑁆𑀔𑀺𑀢𑀦𑁆”𑀢𑀺𑁇_
