# 𑁧𑁇𑁯𑁇𑁧𑁦 𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀧𑀸𑀬𑀺𑀫𑁆𑀳𑀯𑀕𑁆𑀕

### 𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀚𑀸𑀢𑀓

(90.)

198\. _“𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀓𑀮𑁆𑀬𑀸𑀡𑁄,_  
_𑀓𑀢𑀢𑁆𑀣𑁄 𑀦𑀸𑀯𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀧𑀘𑁆𑀙𑀸 𑀓𑀺𑀘𑁆𑀘𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀓𑀢𑁆𑀢𑀸𑀭𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇_  


---

199\. 𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀅𑀧𑀸𑀬𑀺𑀫𑁆𑀳𑀯𑀕𑁆𑀕𑁄 𑀦𑀯𑀫𑁄𑁇



---

200\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



201\. _𑀅𑀧𑀸𑀬𑀺𑀫𑁆𑀳 𑀘 𑀤𑀽𑀪𑀓 𑀲𑀢𑁆𑀢𑀧𑀤𑀁,_  
_𑀙𑀍𑀅𑀤𑁆𑀯𑀭 𑀘 𑀆𑀬𑀢𑀺𑀦𑀸 𑀘 𑀧𑀼𑀦;_  
_𑀅𑀳𑀺𑀲𑀻𑀮𑀯 𑀫𑀗𑁆𑀕𑀮𑀺 𑀧𑀸𑀧𑀺𑀓𑀲𑁆𑀲𑀸,_  
_𑀲𑀢𑀁𑀦𑀺𑀓𑁆𑀔 𑀓𑀢𑀢𑁆𑀣𑀯𑀭𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
