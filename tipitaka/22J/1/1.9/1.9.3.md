# 𑁧𑁇𑁯𑁇𑁩 𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀧𑀸𑀬𑀺𑀫𑁆𑀳𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓

(83.)

184. _“𑀫𑀺𑀢𑁆𑀢𑁄 𑀳𑀯𑁂 𑀲𑀢𑁆𑀢𑀧𑀤𑁂𑀦 𑀳𑁄𑀢𑀺,_  
_𑀲𑀳𑀸𑀬𑁄 𑀧𑀦 𑀤𑁆𑀯𑀸𑀤𑀲𑀓𑁂𑀦 𑀳𑁄𑀢𑀺;_  
_𑀫𑀸𑀲𑀟𑁆𑀠𑀫𑀸𑀲𑁂𑀦 𑀘 𑀜𑀸𑀢𑀺 𑀳𑁄𑀢𑀺,_  
_𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀅𑀢𑁆𑀢𑀲𑀫𑁄𑀧𑀺 𑀳𑁄𑀢𑀺;_  
_𑀲𑁄𑀳𑀁 𑀓𑀣𑀁 𑀅𑀢𑁆𑀢𑀲𑀼𑀔𑀲𑁆𑀲 𑀳𑁂𑀢𑀼,_  
_𑀘𑀺𑀭𑀲𑀦𑁆𑀣𑀼𑀢𑀁 𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀁 𑀚𑀳𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

185. 𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁
