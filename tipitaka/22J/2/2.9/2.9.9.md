# 𑁨𑁇𑁯𑁇𑁯 𑀳𑀭𑀺𑀢𑀫𑀡𑁆𑀟𑀽𑀓𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀉𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

* 𑀳𑀭𑀺𑀢𑀫𑀡𑁆𑀟𑀽𑀓𑀚𑀸𑀢𑀓

(177.)

626\. _“𑀆𑀲𑀻𑀯𑀺𑀲𑀫𑁆𑀧𑀺 𑀫𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀧𑀯𑀺𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀺𑀦𑀸𑀫𑀼𑀔𑀁;_  
_𑀭𑀼𑀘𑁆𑀘𑀢𑁂 𑀳𑀭𑀺𑀢𑀸𑀫𑀸𑀢𑀸,_  
_𑀬𑀁 𑀫𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀺 𑀫𑀘𑁆𑀙𑀓𑀸”𑁇_  


(178.)

627\. _“𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀢𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀬𑀸𑀯𑀲𑁆𑀲 𑀉𑀧𑀓𑀧𑁆𑀧𑀢𑀺;_  
_𑀬𑀤𑀸 𑀘𑀜𑁆𑀜𑁂 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀦𑁆𑀢𑀺,_  
_𑀲𑁄 𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀢𑀻”𑀢𑀺𑁇_  


---
