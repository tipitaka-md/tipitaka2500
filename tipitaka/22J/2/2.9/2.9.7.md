# 𑁨𑁇𑁯𑁇𑁭 𑀲𑀸𑀓𑁂𑀢𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀓𑁂𑀢𑀚𑀸𑀢𑀓

(173.)

620\. _“𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀳𑁂𑀢𑀼,_  
_𑀏𑀓𑀘𑁆𑀘𑁂 𑀇𑀥 𑀧𑀼𑀕𑁆𑀕𑀮𑁂;_  
_𑀅𑀢𑀻𑀯 𑀳𑀤𑀬𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑀸𑀧𑀺 𑀧𑀲𑀻𑀤𑀢𑀺”𑁇_  


(174.)

621\. _“𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑀦𑁆𑀦𑀺𑀯𑀸𑀲𑁂𑀦,_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀳𑀺𑀢𑁂𑀦 𑀯𑀸;_  
_𑀏𑀯𑀁 𑀢𑀁 𑀚𑀸𑀬𑀢𑁂 𑀧𑁂𑀫𑀁,_  
_𑀉𑀧𑁆𑀧𑀮𑀁𑀯 𑀬𑀣𑁄𑀤𑀓𑁂”𑀢𑀺𑁇_
