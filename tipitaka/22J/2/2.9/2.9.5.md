# 𑁨𑁇𑁯𑁇𑁫 𑀯𑀘𑁆𑀙𑀦𑀔𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀯𑀘𑁆𑀙𑀦𑀔𑀚𑀸𑀢𑀓

(169.)

614\. _“𑀲𑀼𑀔𑀸 𑀖𑀭𑀸 𑀯𑀘𑁆𑀙𑀦𑀔,_  
_𑀲𑀳𑀺𑀭𑀜𑁆𑀜𑀸 𑀲𑀪𑁄𑀚𑀦𑀸;_  
_𑀬𑀢𑁆𑀣 𑀪𑀼𑀢𑁆𑀯𑀸 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘,_  
_𑀲𑀬𑁂𑀬𑁆𑀬𑀸𑀣 𑀅𑀦𑀼𑀲𑁆𑀲𑀼𑀓𑁄”𑁇_  


(170.)

615\. _“𑀖𑀭𑀸 𑀦𑀸𑀦𑀻𑀳𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀖𑀭𑀸 𑀦𑀸𑀪𑀡𑀢𑁄 𑀫𑀼𑀲𑀸;_  
_𑀖𑀭𑀸 𑀦𑀸𑀤𑀺𑀦𑁆𑀦𑀤𑀡𑁆𑀟𑀲𑁆𑀲,_  
_𑀧𑀭𑁂𑀲𑀁 𑀅𑀦𑀺𑀓𑀼𑀩𑁆𑀩𑀢𑁄;_  
_𑀏𑀯𑀁 𑀙𑀺𑀤𑁆𑀤𑀁 𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀁,_  
_𑀓𑁄 𑀖𑀭𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_
