# 𑁨𑁇𑁯𑁇𑁬 𑀩𑀓𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀩𑀓𑀚𑀸𑀢𑀓

(171.)

617. _“𑀪𑀤𑁆𑀤𑀓𑁄 𑀯𑀢𑀬𑀁 𑀧𑀓𑁆𑀔𑀻,_  
_𑀤𑀺𑀚𑁄 𑀓𑀼𑀫𑀼𑀤𑀲𑀦𑁆𑀦𑀺𑀪𑁄;_  
_𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑁂𑀳𑀺 𑀧𑀓𑁆𑀔𑁂𑀳𑀺,_  
_𑀫𑀦𑁆𑀤𑀫𑀦𑁆𑀤𑁄𑀯 𑀛𑀸𑀬𑀢𑀺”𑁇_  


(172.)

618. _“𑀦𑀸𑀲𑁆𑀲 𑀲𑀻𑀮𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀣,_  
_𑀅𑀦𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑀁𑀲𑀣;_  
_𑀅𑀫𑁆𑀳𑁂 𑀤𑀺𑀚𑁄 𑀦 𑀧𑀸𑀮𑁂𑀢𑀺,_  
_𑀢𑁂𑀦 𑀧𑀓𑁆𑀔𑀻 𑀦 𑀨𑀦𑁆𑀤𑀢𑀻”𑀢𑀺𑁇_
