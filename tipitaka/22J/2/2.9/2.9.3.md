# 𑁨𑁇𑁯𑁇𑁩 𑀯𑀺𑀓𑀡𑁆𑀡𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀉𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀓𑀡𑁆𑀡𑀚𑀸𑀢𑀓

(165.)

608\. _“𑀓𑀸𑀫𑀁 𑀬𑀳𑀺𑀁 𑀇𑀘𑁆𑀙𑀲𑀺 𑀢𑁂𑀦 𑀕𑀘𑁆𑀙,_  
_𑀯𑀺𑀤𑁆𑀥𑁄𑀲𑀺 𑀫𑀫𑁆𑀫𑀫𑁆𑀳𑀺 𑀯𑀺𑀓𑀡𑁆𑀡𑀓𑁂𑀦;_  
_𑀳𑀢𑁄𑀲𑀺 𑀪𑀢𑁆𑀢𑁂𑀦 𑀲𑀼𑀯𑀸𑀤𑀺𑀢𑁂𑀦,_  
_𑀮𑁄𑀮𑁄 𑀘 𑀫𑀘𑁆𑀙𑁂 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀫𑀸𑀦𑁄”𑁇_  


(166.)

609\. _“𑀏𑀯𑀫𑁆𑀧𑀺 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀁 𑀑𑀧𑀢𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀻 𑀘𑀺𑀢𑁆𑀢𑀯𑀲𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀻;_  
_𑀲𑁄 𑀳𑀜𑁆𑀜𑀢𑀻 𑀜𑀸𑀢𑀺𑀲𑀔𑀸𑀦 𑀫𑀚𑁆𑀛𑁂,_  
_𑀫𑀘𑁆𑀙𑀸𑀦𑀼𑀕𑁄 𑀲𑁄𑀭𑀺𑀯 𑀲𑀼𑀁𑀲𑀼𑀫𑀸𑀭𑁄”𑀢𑀺𑁇_
