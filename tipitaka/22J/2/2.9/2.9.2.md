# 𑁨𑁇𑁯𑁇𑁨 𑀯𑀻𑀡𑀸𑀕𑀼𑀡𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀉𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

* 𑀯𑀻𑀡𑀸𑀕𑀼𑀡𑀚𑀸𑀢𑀓

(163.)

605\. _“𑀏𑀓𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑁄 𑀬𑀫𑀢𑁆𑀣𑁄,_  
_𑀩𑀸𑀮𑁄 𑀅𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑁄;_  
_𑀦 𑀳𑀺 𑀔𑀼𑀚𑁆𑀚𑁂𑀦 𑀯𑀸𑀫𑁂𑀦,_  
_𑀪𑁄𑀢𑀺 𑀲𑀗𑁆𑀕𑀦𑁆𑀢𑀼𑀫𑀭𑀳𑀲𑀺”𑁇_  


(164.)

606\. _“𑀧𑀼𑀭𑀺𑀲𑀽𑀲𑀪𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑀸,_  
_𑀅𑀳𑀁 𑀔𑀼𑀚𑁆𑀚𑀫𑀓𑀸𑀫𑀬𑀺𑀁;_  
_𑀲𑁄𑀬𑀁 𑀲𑀗𑁆𑀓𑀼𑀝𑀺𑀢𑁄 𑀲𑁂𑀢𑀺,_  
_𑀙𑀺𑀦𑁆𑀦𑀢𑀦𑁆𑀢𑀺 𑀬𑀣𑀸 𑀯𑀺𑀡𑀸”𑀢𑀺𑁇_  


---
