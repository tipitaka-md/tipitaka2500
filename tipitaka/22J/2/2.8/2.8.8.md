# 𑁨𑁇𑁮𑁇𑁮 𑀓𑀸𑀫𑀦𑀻𑀢𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀸𑀲𑀸𑀯𑀯𑀕𑁆𑀕

* 𑀓𑀸𑀫𑀦𑀻𑀢𑀚𑀸𑀢𑀓

(155.)

591\. _“𑀢𑀬𑁄 𑀕𑀺𑀭𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑀁 𑀓𑀸𑀫𑀬𑀸𑀫𑀺,_  
_𑀧𑀜𑁆𑀘𑀸𑀮𑀸 𑀓𑀼𑀭𑀼𑀬𑁄 𑀓𑁂𑀓𑀓𑁂 𑀘;_  
_𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀓𑀸𑀫𑀬𑀸𑀫𑀺,_  
_𑀢𑀺𑀓𑀺𑀘𑁆𑀙 𑀫𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀓𑀸𑀫𑀦𑀻𑀢𑀁”𑁇_  


(156.)

592\. _“𑀓𑀡𑁆𑀳𑀸𑀳𑀺𑀤𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀳𑁂𑀓𑁂,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀯𑀺𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀦 𑀓𑀸𑀫𑀦𑀻𑀢𑀲𑁆𑀲 𑀓𑀭𑁄𑀢𑀺 𑀓𑁄𑀘𑀺,_  
_𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀳𑀺 𑀓𑀸 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸”𑀢𑀺𑁇_  


---
