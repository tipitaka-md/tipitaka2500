# 𑁨𑁇𑁮𑁇𑁪 𑀓𑀼𑀫𑁆𑀪𑀺𑀮𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀸𑀲𑀸𑀯𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀫𑁆𑀪𑀺𑀮𑀚𑀸𑀢𑀓

(147.)

579\. _“𑀬𑀲𑁆𑀲𑁂𑀢𑁂 𑀘𑀢𑀼𑀭𑁄 𑀥𑀫𑁆𑀫𑀸,_  
_𑀯𑀸𑀦𑀭𑀺𑀦𑁆𑀤 𑀬𑀣𑀸 𑀢𑀯;_  
_𑀲𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑁄 𑀥𑀺𑀢𑀺 𑀘𑀸𑀕𑁄,_  
_𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑁄 𑀅𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺𑁇_  


(148.)

580\. _𑀬𑀲𑁆𑀲 𑀘𑁂𑀢𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀕𑀼𑀡𑀸 𑀧𑀭𑀫𑀪𑀤𑁆𑀤𑀓𑀸;_  
_𑀲𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑁄 𑀥𑀺𑀢𑀺 𑀘𑀸𑀕𑁄,_  
_𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑁄 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇_  


---
