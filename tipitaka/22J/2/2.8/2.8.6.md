# 𑁨𑁇𑁮𑁇𑁬 𑀓𑁄𑀲𑀺𑀬𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀸𑀲𑀸𑀯𑀯𑀕𑁆𑀕

* 𑀓𑁄𑀲𑀺𑀬𑀚𑀸𑀢𑀓

(151.)

585\. _“𑀓𑀸𑀮𑁂 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑀸 𑀲𑀸𑀥𑀼,_  
_𑀦𑀸𑀓𑀸𑀮𑁂 𑀲𑀸𑀥𑀼 𑀦𑀺𑀓𑁆𑀔𑀫𑁄;_  
_𑀅𑀓𑀸𑀮𑁂𑀦 𑀳𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀏𑀓𑀓𑀫𑁆𑀧𑀺 𑀩𑀳𑀼𑀚𑁆𑀚𑀦𑁄;_  
_𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀢𑁆𑀣𑀁 𑀚𑁄𑀢𑁂𑀢𑀺,_  
_𑀥𑀗𑁆𑀓𑀲𑁂𑀦𑀸𑀯 𑀓𑁄𑀲𑀺𑀬𑀁𑁇_  


(152.)

586\. _𑀥𑀻𑀭𑁄 𑀘 𑀯𑀺𑀥𑀺𑀯𑀺𑀥𑀸𑀦𑀜𑁆𑀜𑀽,_  
_𑀧𑀭𑁂𑀲𑀁 𑀯𑀺𑀯𑀭𑀸𑀦𑀼𑀕𑀽;_  
_𑀲𑀩𑁆𑀩𑀸𑀫𑀺𑀢𑁆𑀢𑁂 𑀯𑀲𑀻𑀓𑀢𑁆𑀯𑀸,_  
_𑀓𑁄𑀲𑀺𑀬𑁄𑀯 𑀲𑀼𑀔𑀻 𑀲𑀺𑀬𑀸”𑀢𑀺𑁇_
