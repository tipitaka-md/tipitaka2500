# 𑁨𑁇𑁮𑁇𑁯 𑀧𑀮𑀸𑀬𑀺𑀢𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀸𑀲𑀸𑀯𑀯𑀕𑁆𑀕

### 𑀧𑀮𑀸𑀬𑀺𑀢𑀚𑀸𑀢𑀓

(157.)

594. _“𑀕𑀚𑀕𑁆𑀕𑀫𑁂𑀖𑁂𑀳𑀺 𑀳𑀬𑀕𑁆𑀕𑀫𑀸𑀮𑀺𑀪𑀺,_  
_𑀭𑀣𑀽𑀫𑀺𑀚𑀸𑀢𑁂𑀳𑀺 𑀲𑀭𑀸𑀪𑀺𑀯𑀲𑁆𑀲𑁂𑀪𑀺;_  
_𑀣𑀭𑀼𑀕𑁆𑀕𑀳𑀸𑀯𑀝𑁆𑀝𑀤𑀍𑀳𑀧𑁆𑀧𑀳𑀸𑀭𑀺𑀪𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑀸 𑀢𑀓𑁆𑀓𑀲𑀺𑀮𑀸 𑀲𑀫𑀦𑁆𑀢𑀢𑁄𑁇_  


(158.)

595. _𑀅𑀪𑀺𑀥𑀸𑀯𑀣 𑀘𑀽𑀧𑀥𑀸𑀯𑀣 𑀘,_  
_𑀯𑀺𑀯𑀺𑀥𑀸 𑀯𑀺𑀦𑀸𑀤𑀺𑀢𑀸 𑀯𑀤𑀦𑁆𑀢𑀺𑀪𑀺;_  
_𑀯𑀢𑁆𑀢𑀢𑀚𑁆𑀚 𑀢𑀼𑀫𑀼𑀮𑁄 𑀖𑁄𑀲𑁄 𑀬𑀣𑀸,_  
_𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑀢𑀸 𑀚𑀮𑀥𑀭𑀲𑁆𑀲 𑀕𑀚𑁆𑀚𑀢𑁄”𑀢𑀺𑁇_  


---
