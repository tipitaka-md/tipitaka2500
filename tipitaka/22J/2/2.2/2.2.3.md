# 𑁨𑁇𑁨𑁇𑁩 𑀲𑀼𑀲𑀻𑀫𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀲𑀻𑀫𑀚𑀸𑀢𑀓

(25.)

380\. _“𑀓𑀸𑀍𑀆 𑀫𑀺𑀕𑀸 𑀲𑁂𑀢𑀤𑀦𑁆𑀢𑀸 𑀢𑀯𑀻𑀫𑁂,_  
_𑀧𑀭𑁄𑀲𑀢𑀁 𑀳𑁂𑀫𑀚𑀸𑀮𑀸𑀪𑀺𑀙𑀦𑁆𑀦𑀸;_  
_𑀢𑁂 𑀢𑁂 𑀤𑀤𑀸𑀫𑀻𑀢𑀺 𑀲𑀼𑀲𑀻𑀫 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀧𑀺𑀢𑀸𑀫𑀳𑀸𑀦𑀁”𑁇_  


(26.)

381\. _“𑀓𑀸𑀍𑀆 𑀫𑀺𑀕𑀸 𑀲𑁂𑀢𑀤𑀦𑁆𑀢𑀸 𑀫𑀫𑀻𑀫𑁂,_  
_𑀧𑀭𑁄𑀲𑀢𑀁 𑀳𑁂𑀫𑀚𑀸𑀮𑀸𑀪𑀺𑀙𑀦𑁆𑀦𑀸;_  
_𑀢𑁂 𑀢𑁂 𑀤𑀤𑀸𑀫𑀻𑀢𑀺 𑀯𑀤𑀸𑀫𑀺 𑀫𑀸𑀡𑀯,_  
_𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀧𑀺𑀢𑀸𑀫𑀳𑀸𑀦𑀦𑁆”𑀢𑀺𑁇_  


---
