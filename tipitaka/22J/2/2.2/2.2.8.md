# 𑁨𑁇𑁨𑁇𑁮 𑀲𑀓𑀼𑀡𑀕𑁆𑀖𑀺𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀲𑀓𑀼𑀡𑀕𑁆𑀖𑀺𑀚𑀸𑀢𑀓

(35.)

395\. _“𑀲𑁂𑀦𑁄 𑀩𑀮𑀲𑀸 𑀧𑀢𑀫𑀸𑀦𑁄,_  
_𑀮𑀸𑀧𑀁 𑀕𑁄𑀘𑀭𑀞𑀸𑀬𑀺𑀦𑀁;_  
_𑀲𑀳𑀲𑀸 𑀅𑀚𑁆𑀛𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑀯,_  
_𑀫𑀭𑀡𑀁 𑀢𑁂𑀦𑀼𑀧𑀸𑀕𑀫𑀺𑁇_  


(36.)

396\. _𑀲𑁄𑀳𑀁 𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑁂 𑀕𑁄𑀘𑀭𑁂 𑀭𑀢𑁄;_  
_𑀅𑀧𑁂𑀢𑀲𑀢𑁆𑀢𑀼 𑀫𑁄𑀤𑀸𑀫𑀺,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁 𑀅𑀢𑁆𑀣𑀫𑀢𑁆𑀢𑀦𑁄”𑀢𑀺𑁇_  


---
