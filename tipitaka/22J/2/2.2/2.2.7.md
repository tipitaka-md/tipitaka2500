# 𑁨𑁇𑁨𑁇𑁭 𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀚𑀸𑀢𑀓

(33.)

392\. _“𑀅𑀪𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀦 𑀳𑀺 𑀪𑀼𑀢𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀲𑀺;_  
_𑀪𑀼𑀢𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀲𑁆𑀲𑀼,_  
_𑀫𑀸 𑀢𑀁 𑀓𑀸𑀮𑁄 𑀉𑀧𑀘𑁆𑀘𑀕𑀸𑁇_  


(34.)

393\. _𑀓𑀸𑀮𑀁 𑀯𑁄𑀳𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀙𑀦𑁆𑀦𑁄 𑀓𑀸𑀮𑁄 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀅𑀪𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀫𑀸 𑀫𑀁 𑀓𑀸𑀮𑁄 𑀉𑀧𑀘𑁆𑀘𑀕𑀸”𑀢𑀺𑁇_  


---
