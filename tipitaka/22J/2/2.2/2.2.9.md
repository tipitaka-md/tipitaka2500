# 𑁨𑁇𑁨𑁇𑁯 𑀅𑀭𑀓𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

* 𑀅𑀭𑀓𑀚𑀸𑀢𑀓

(37.)

398\. _“𑀬𑁄 𑀯𑁂 𑀫𑁂𑀢𑁆𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀢𑀺;_  
_𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀘 𑀢𑀺𑀭𑀺𑀬𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀲𑀩𑁆𑀩𑀲𑁄𑁇_  


(38.)

399\. _𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁;_  
_𑀬𑀁 𑀧𑀫𑀸𑀡𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀦 𑀢𑀁 𑀢𑀢𑁆𑀭𑀸𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_
