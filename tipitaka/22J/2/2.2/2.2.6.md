# 𑁨𑁇𑁨𑁇𑁬 𑀉𑀧𑀲𑀸𑀍𑀅𑀓𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀲𑀸𑀍𑀅𑀓𑀚𑀸𑀢𑀓

(31.)

389\. _“𑀉𑀧𑀲𑀸𑀍𑀅𑀓𑀦𑀸𑀫𑀸𑀦𑀺,_  
_𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀘𑀢𑀼𑀤𑁆𑀤𑀲;_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀤𑀟𑁆𑀠𑀸𑀦𑀺,_  
_𑀦𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀅𑀦𑀸𑀫𑀢𑀁𑁇_  


(32.)

390\. _𑀬𑀫𑁆𑀳𑀺 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁄 𑀘,_  
_𑀅𑀳𑀺𑀁𑀲𑀸 𑀲𑀁𑀬𑀫𑁄 𑀤𑀫𑁄;_  
_𑀏𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀲𑁂𑀯𑀦𑁆𑀢𑀺,_  
_𑀏𑀢𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑀸𑀫𑀢𑀦𑁆”𑀢𑀺𑁇_
