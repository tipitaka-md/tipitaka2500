# 𑁨𑁇𑁨𑁇𑁫 𑀦𑀓𑀼𑀮𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀦𑀓𑀼𑀮𑀚𑀸𑀢𑀓

(29.)

386\. _“𑀲𑀦𑁆𑀥𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀫𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀅𑀡𑁆𑀟𑀚𑁂𑀦 𑀚𑀮𑀸𑀩𑀼𑀚;_  
_𑀯𑀺𑀯𑀭𑀺𑀬 𑀤𑀸𑀞𑀁 𑀲𑁂𑀲𑀺,_  
_𑀓𑀼𑀢𑁄 𑀢𑁂 𑀪𑀬𑀫𑀸𑀕𑀢𑀁”𑁇_  


(30.)

387\. _“𑀲𑀗𑁆𑀓𑁂𑀣𑁂𑀯 𑀅𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦 𑀯𑀺𑀲𑁆𑀲𑀲𑁂;_  
_𑀅𑀪𑀬𑀸 𑀪𑀬𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀅𑀧𑀺 𑀫𑀽𑀮𑀸𑀦𑀺 𑀓𑀦𑁆𑀢𑀢𑀻”𑀢𑀺𑁇_
