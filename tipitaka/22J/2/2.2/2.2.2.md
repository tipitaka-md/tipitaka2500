# 𑁨𑁇𑁨𑁇𑁨 𑀲𑀦𑁆𑀣𑀯𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

* 𑀲𑀦𑁆𑀣𑀯𑀚𑀸𑀢𑀓

(23.)

377\. _“𑀦 𑀲𑀦𑁆𑀣𑀯𑀲𑁆𑀫𑀸 𑀧𑀭𑀫𑀢𑁆𑀣𑀺 𑀧𑀸𑀧𑀺𑀬𑁄,_  
_𑀬𑁄 𑀲𑀦𑁆𑀣𑀯𑁄 𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀳𑁄𑀢𑀺;_  
_𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑁄 𑀲𑀧𑁆𑀧𑀺𑀦𑀸 𑀧𑀸𑀬𑀲𑁂𑀦,_  
_𑀓𑀺𑀘𑁆𑀙𑀸𑀓𑀢𑀁 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀺𑀁 𑀅𑀤𑀬𑁆𑀳𑀺𑁇_  


(24.)

378\. _𑀦 𑀲𑀦𑁆𑀣𑀯𑀲𑁆𑀫𑀸 𑀧𑀭𑀫𑀢𑁆𑀣𑀺 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑁄 𑀲𑀦𑁆𑀣𑀯𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀳𑁄𑀢𑀺;_  
_𑀲𑀻𑀳𑀲𑁆𑀲 𑀩𑁆𑀬𑀕𑁆𑀖𑀲𑁆𑀲 𑀘 𑀤𑀻𑀧𑀺𑀦𑁄 𑀘,_  
_𑀲𑀸𑀫𑀸 𑀫𑀼𑀔𑀁 𑀮𑁂𑀳𑀢𑀺 𑀲𑀦𑁆𑀣𑀯𑁂𑀦𑀸”𑀢𑀺𑁇_  


---
