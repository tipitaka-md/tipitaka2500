# 𑁨𑁇𑁨𑁇𑁧 𑀇𑀦𑁆𑀤𑀲𑀫𑀸𑀦𑀕𑁄𑀢𑁆𑀢𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀇𑀦𑁆𑀤𑀲𑀫𑀸𑀦𑀕𑁄𑀢𑁆𑀢𑀚𑀸𑀢𑀓

(21.)

374\. _“𑀦 𑀲𑀦𑁆𑀣𑀯𑀁 𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀓𑀬𑀺𑀭𑀸,_  
_𑀅𑀭𑀺𑀬𑁄 𑀅𑀦𑀭𑀺𑀬𑁂𑀦 𑀧𑀚𑀸𑀦𑀫𑀢𑁆𑀣𑀁;_  
_𑀘𑀺𑀭𑀸𑀦𑀼𑀯𑀼𑀢𑁆𑀣𑁄𑀧𑀺 𑀓𑀭𑁄𑀢𑀺 𑀧𑀸𑀧𑀁,_  
_𑀕𑀚𑁄 𑀬𑀣𑀸 𑀇𑀦𑁆𑀤𑀲𑀫𑀸𑀦𑀕𑁄𑀢𑁆𑀢𑀁𑁇_  


(22.)

375\. _𑀬𑀁 𑀢𑁆𑀯𑁂𑀯 𑀚𑀜𑁆𑀜𑀸 𑀲𑀤𑀺𑀲𑁄 𑀫𑀫𑀦𑁆𑀢𑀺,_  
_𑀲𑀻𑀮𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀢𑁂𑀦 𑀘𑀸𑀧𑀺;_  
_𑀢𑁂𑀦𑁂𑀯 𑀫𑁂𑀢𑁆𑀢𑀺𑀁 𑀓𑀬𑀺𑀭𑀸𑀣 𑀲𑀤𑁆𑀥𑀺𑀁,_  
_𑀲𑀼𑀔𑁄 𑀳𑀯𑁂 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀗𑁆𑀕𑀫𑁄”𑀢𑀺𑁇_  


---

376\. 𑀇𑀦𑁆𑀤𑀲𑀫𑀸𑀦𑀕𑁄𑀢𑁆𑀢𑀚𑀸𑀢𑀓𑀁 𑀧𑀞𑀫𑀁𑁇
