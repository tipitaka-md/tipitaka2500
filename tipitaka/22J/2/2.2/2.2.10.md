# 𑁨𑁇𑁨𑁇𑁧𑁦 𑀓𑀓𑀡𑁆𑀝𑀓𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀓𑀓𑀡𑁆𑀝𑀓𑀚𑀸𑀢𑀓

(39.)

401\. _“𑀦𑀸𑀬𑀁 𑀧𑀼𑀭𑁂 𑀉𑀡𑁆𑀡𑀫𑀢𑀺,_  
_𑀢𑁄𑀭𑀡𑀕𑁆𑀕𑁂 𑀓𑀓𑀡𑁆𑀝𑀓𑁄;_  
_𑀫𑀳𑁄𑀲𑀥 𑀯𑀺𑀚𑀸𑀦𑀸𑀳𑀺,_  
_𑀓𑁂𑀦 𑀣𑀤𑁆𑀥𑁄 𑀓𑀓𑀡𑁆𑀝𑀓𑁄”𑁇_  


(40.)

402\. _“𑀅𑀮𑀤𑁆𑀥𑀧𑀼𑀩𑁆𑀩𑀁 𑀮𑀤𑁆𑀥𑀸𑀦,_  
_𑀅𑀟𑁆𑀠𑀫𑀸𑀲𑀁 𑀓𑀓𑀡𑁆𑀝𑀓𑁄;_  
_𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺 𑀭𑀸𑀚𑀸𑀦𑀁,_  
_𑀯𑁂𑀤𑁂𑀳𑀁 𑀫𑀺𑀣𑀺𑀮𑀕𑁆𑀕𑀳𑀦𑁆”𑀢𑀺𑁇_  


---

403\. 𑀓𑀓𑀡𑁆𑀝𑀓𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



---

404\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



405\. _𑀅𑀣 𑀇𑀦𑁆𑀤𑀲𑀫𑀸𑀦 𑀲𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀺,_  
_𑀲𑀼𑀲𑀺𑀫𑀼𑀢𑁆𑀢𑀫 𑀕𑀺𑀚𑁆𑀛 𑀚𑀮𑀸𑀩𑀼𑀚𑀓𑁄;_  
_𑀉𑀧𑀲𑀸𑀍𑀅𑀓 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀮𑀸𑀧𑀯𑀭𑁄,_  
_𑀅𑀣 𑀫𑁂𑀢𑁆𑀢𑀯𑀭𑁄 𑀤𑀲𑀧𑀼𑀡𑁆𑀡𑀫𑀢𑀻𑀢𑀺𑁇_
