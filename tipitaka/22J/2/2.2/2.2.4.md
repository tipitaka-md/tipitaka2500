# 𑁨𑁇𑁨𑁇𑁪 𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓

(27.)

383\. _“𑀬𑀦𑁆𑀦𑀼 𑀕𑀺𑀚𑁆𑀛𑁄 𑀬𑁄𑀚𑀦𑀲𑀢𑀁,_  
_𑀓𑀼𑀡𑀧𑀸𑀦𑀺 𑀅𑀯𑁂𑀓𑁆𑀔𑀢𑀺;_  
_𑀓𑀲𑁆𑀫𑀸 𑀚𑀸𑀮𑀜𑁆𑀘 𑀧𑀸𑀲𑀜𑁆𑀘,_  
_𑀆𑀲𑀚𑁆𑀚𑀸𑀧𑀺 𑀦 𑀩𑀼𑀚𑁆𑀛𑀲𑀺”𑁇_  


(28.)

384\. _“𑀬𑀤𑀸 𑀧𑀭𑀸𑀪𑀯𑁄 𑀳𑁄𑀢𑀺,_  
_𑀧𑁄𑀲𑁄 𑀚𑀻𑀯𑀺𑀢𑀲𑀗𑁆𑀔𑀬𑁂;_  
_𑀅𑀣 𑀚𑀸𑀮𑀜𑁆𑀘 𑀧𑀸𑀲𑀜𑁆𑀘,_  
_𑀆𑀲𑀚𑁆𑀚𑀸𑀧𑀺 𑀦 𑀩𑀼𑀚𑁆𑀛𑀢𑀻”𑀢𑀺𑁇_
