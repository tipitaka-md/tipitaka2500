# 𑁨𑁇𑁩𑁇𑁧𑁦 𑀤𑀼𑀤𑁆𑀤𑀤𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀤𑁆𑀤𑀤𑀚𑀸𑀢𑀓

(59.)

433. _“𑀤𑀼𑀤𑁆𑀤𑀤𑀁 𑀤𑀤𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀓𑀫𑁆𑀫 𑀓𑀼𑀩𑁆𑀩𑀢𑀁;_  
_𑀅𑀲𑀦𑁆𑀢𑁄 𑀦𑀸𑀦𑀼𑀓𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺,_  
_𑀲𑀢𑀁 𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀭𑀦𑁆𑀦𑀬𑁄”𑁇_  


(60.)

434. _“𑀢𑀲𑁆𑀫𑀸 𑀲𑀢𑀜𑁆𑀘 𑀅𑀲𑀢𑀁,_  
_𑀦𑀸𑀦𑀸 𑀳𑁄𑀢𑀺 𑀇𑀢𑁄 𑀕𑀢𑀺;_  
_𑀅𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀬𑀁 𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑀦𑁆𑀢𑁄 𑀲𑀕𑁆𑀕𑀧𑀭𑀸𑀬𑀡𑀸𑀢𑀺𑁇_  


---

435. 𑀤𑀼𑀤𑁆𑀤𑀤𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

436. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



437. _𑀲𑀼𑀲𑀫𑀜𑁆𑀜𑀫𑀺𑀕𑀸𑀥𑀺𑀪𑀽 𑀫𑀸𑀡𑀯𑀓𑁄,_  
_𑀯𑀸𑀭𑀺𑀧𑀳𑀽𑀢𑀭𑀽𑀧𑀸𑀤𑀺𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀸;_  
_𑀲𑀓𑀍𑀆𑀬𑀲𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀧𑀗𑁆𑀓 𑀧𑀼𑀦,_  
_𑀲𑀢𑀥𑀫𑁆𑀫 𑀲𑀼𑀤𑀼𑀤𑁆𑀤𑀤𑀓𑁂𑀦 𑀤𑀲𑀸”𑀢𑀺𑁇_
