# 𑁨𑁇𑁩𑁇𑁨 𑀤𑀤𑁆𑀤𑀭𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀤𑀤𑁆𑀤𑀭𑀚𑀸𑀢𑀓

(43.)

409\. _“𑀓𑁄 𑀦𑀼 𑀲𑀤𑁆𑀤𑁂𑀦 𑀫𑀳𑀢𑀸,_  
_𑀅𑀪𑀺𑀦𑀸𑀤𑁂𑀢𑀺 𑀤𑀤𑁆𑀤𑀭𑀁;_  
_𑀢𑀁 𑀲𑀻𑀳𑀸 𑀦𑀧𑁆𑀧𑀝𑀺𑀦𑀤𑀦𑁆𑀢𑀺,_  
_𑀓𑁄 𑀦𑀸𑀫𑁂𑀲𑁄 𑀫𑀺𑀕𑀸𑀥𑀺𑀪𑀽”𑁇_  


(44.)

410\. _“𑀅𑀥𑀫𑁄 𑀫𑀺𑀕𑀚𑀸𑀢𑀸𑀦𑀁,_  
_𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀢𑀸𑀢 𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀚𑀸𑀢𑀺𑀫𑀲𑁆𑀲 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀸,_  
_𑀢𑀼𑀡𑁆𑀳𑀻 𑀲𑀻𑀳𑀸 𑀲𑀫𑀘𑁆𑀙𑀭𑁂”𑀢𑀺𑁇_
