# 𑁨𑁇𑁩𑁇𑁩 𑀫𑀓𑁆𑀓𑀝𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀫𑀓𑁆𑀓𑀝𑀚𑀸𑀢𑀓

(45.)

412. _“𑀢𑀸𑀢 𑀫𑀸𑀡𑀯𑀓𑁄 𑀏𑀲𑁄,_  
_𑀢𑀸𑀮𑀫𑀽𑀮𑀁 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑁄;_  
_𑀅𑀕𑀸𑀭𑀓𑀜𑁆𑀘𑀺𑀤𑀁 𑀅𑀢𑁆𑀣𑀺,_  
_𑀳𑀦𑁆𑀤 𑀤𑁂𑀫𑀲𑁆𑀲𑀕𑀸𑀭𑀓𑀁”𑁇_  


(46.)

413. _“𑀫𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁 𑀢𑀸𑀢 𑀧𑀓𑁆𑀓𑁄𑀲𑀺,_  
_𑀤𑀽𑀲𑁂𑀬𑁆𑀬 𑀦𑁄 𑀅𑀕𑀸𑀭𑀓𑀁;_  
_𑀦𑁂𑀢𑀸𑀤𑀺𑀲𑀁 𑀫𑀼𑀔𑀁 𑀳𑁄𑀢𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀲𑀼𑀲𑀻𑀮𑀺𑀦𑁄”𑀢𑀺𑁇_  


---
