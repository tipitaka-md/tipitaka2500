# 𑁨𑁇𑁩𑁇𑁮 𑀓𑀘𑁆𑀙𑀧𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀓𑀘𑁆𑀙𑀧𑀚𑀸𑀢𑀓

(55.)

427\. _“𑀚𑀦𑀺𑀢𑁆𑀢𑀁 𑀫𑁂 𑀪𑀯𑀺𑀢𑁆𑀢𑀁 𑀫𑁂,_  
_𑀇𑀢𑀺 𑀧𑀗𑁆𑀓𑁂 𑀅𑀯𑀲𑁆𑀲𑀬𑀺𑀁;_  
_𑀢𑀁 𑀫𑀁 𑀧𑀗𑁆𑀓𑁄 𑀅𑀚𑁆𑀛𑀪𑀯𑀺,_  
_𑀬𑀣𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑀓𑀁 𑀢𑀣𑀸;_  
_𑀢𑀁 𑀢𑀁 𑀯𑀤𑀸𑀫𑀺 𑀪𑀕𑁆𑀕𑀯,_  
_𑀲𑀼𑀡𑁄𑀳𑀺 𑀯𑀘𑀦𑀁 𑀫𑀫”𑁇_  


(56.)

428\. _“𑀕𑀸𑀫𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀜𑁆𑀜𑁂,_  
_𑀲𑀼𑀔𑀁 𑀬𑀢𑁆𑀭𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀢𑀁 𑀚𑀦𑀺𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀧𑀚𑀸𑀦𑀢𑁄;_  
_𑀬𑀫𑁆𑀳𑀺 𑀚𑀻𑀯𑁂 𑀢𑀫𑁆𑀳𑀺 𑀕𑀘𑁆𑀙𑁂,_  
_𑀦 𑀦𑀺𑀓𑁂𑀢𑀳𑀢𑁄 𑀲𑀺𑀬𑀸”𑀢𑀺𑁇_
