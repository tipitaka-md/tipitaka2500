# 𑁨𑁇𑁩𑁇𑁫 𑀆𑀤𑀺𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀆𑀤𑀺𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀚𑀸𑀢𑀓

(49.)

418\. _“𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀓𑀺𑀭 𑀪𑀽𑀢𑁂𑀲𑀼,_  
_𑀲𑀦𑁆𑀢𑀺 𑀲𑀻𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀧𑀲𑁆𑀲 𑀲𑀸𑀔𑀸𑀫𑀺𑀕𑀁 𑀚𑀫𑁆𑀫𑀁,_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑀫𑀼𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀺”𑁇_  


(50.)

419\. _“𑀦𑀸𑀲𑁆𑀲 𑀲𑀻𑀮𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀣,_  
_𑀅𑀦𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑀁𑀲𑀣;_  
_𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀉𑀳𑀦𑁆𑀦𑀁,_  
_𑀤𑁆𑀯𑁂 𑀘 𑀪𑀺𑀦𑁆𑀦𑀸 𑀓𑀫𑀡𑁆𑀟𑀮𑀽”𑀢𑀺𑁇_  


---
