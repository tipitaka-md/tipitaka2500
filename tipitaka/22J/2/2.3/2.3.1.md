# 𑁨𑁇𑁩𑁇𑁧 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀚𑀸𑀢𑀓

(41.)

406\. _“𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀬𑀤𑀸 𑀚𑀦𑀺𑀦𑁆𑀤,_  
_𑀮𑁄𑀓𑁂 𑀲𑀫𑀜𑁆𑀜𑀁 𑀅𑀦𑀼𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀦 𑀳𑀺𑀬𑁆𑀬𑁂𑀣 𑀦𑀭𑁄 𑀲𑀧𑀜𑁆𑀜𑁄,_  
_𑀳𑀺𑀭𑀺𑀬𑀸𑀧𑀺 𑀲𑀦𑁆𑀢𑁄 𑀥𑀼𑀭𑀫𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺𑁇_  


(42.)

407\. _𑀲𑀸𑀬𑀁 𑀲𑀫𑀜𑁆𑀜𑀸 𑀇𑀥 𑀫𑀚𑁆𑀚 𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀚𑀦𑀺𑀦𑁆𑀤 𑀮𑁄𑀓𑁂;_  
_𑀢𑀸𑀳𑀁 𑀲𑀫𑁂𑀓𑁆𑀔𑀁 𑀇𑀥 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀁,_  
_𑀦 𑀳𑀺 𑀫𑀢𑁆𑀣𑀺 𑀙𑀦𑁆𑀤𑁄 𑀇𑀥 𑀓𑀸𑀫𑀪𑁄𑀕𑁂”𑀢𑀺𑁇_  


---
