# 𑁨𑁇𑁩𑁇𑁭 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀚𑀸𑀢𑀓

(53.)

424\. _“𑀥𑀦𑀼𑀳𑀢𑁆𑀣𑀓𑀮𑀸𑀧𑁂𑀳𑀺,_  
_𑀦𑁂𑀢𑁆𑀢𑀺𑀁 𑀲𑀯𑀭𑀥𑀸𑀭𑀺𑀪𑀺;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀫𑁆𑀳,_  
_𑀓𑀣𑀁 𑀫𑁄𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(54.)

425\. _“𑀅𑀧𑁆𑀧𑁂𑀯 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀅𑀢𑁆𑀣𑁄 𑀚𑀸𑀬𑁂𑀣 𑀓𑁄𑀘𑀺 𑀦𑀁;_  
_𑀅𑀢𑁆𑀣𑀺 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁,_  
_𑀔𑀚𑁆𑀚𑀣𑀜𑁆𑀜𑁂𑀯 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀦𑁆”𑀢𑀺𑁇_  


---

426\. 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀚𑀸𑀢𑀓𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
