# 𑁨𑁇𑁬𑁇𑁧𑁦 𑀓𑀦𑁆𑀤𑀕𑀮𑀓𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

### 𑀓𑀦𑁆𑀤𑀕𑀮𑀓𑀚𑀸𑀢𑀓

(119.)

533\. _“𑀅𑀫𑁆𑀪𑁄 𑀓𑁄 𑀦𑀸𑀫 𑀬𑀁 𑀭𑀼𑀓𑁆𑀔𑁄,_  
_𑀲𑀺𑀦𑁆𑀦𑀧𑀢𑁆𑀢𑁄 𑀲𑀓𑀡𑁆𑀝𑀓𑁄;_  
_𑀬𑀢𑁆𑀣 𑀏𑀓𑀧𑁆𑀧𑀳𑀸𑀭𑁂𑀦,_  
_𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑀁 𑀯𑀺𑀪𑀺𑀚𑁆𑀚𑀺𑀢𑀁”𑁇_  


(120.)

534\. _“𑀅𑀘𑀸𑀭𑀺 𑀯𑀢𑀸𑀬𑀁 𑀯𑀺𑀢𑀼𑀤𑀁 𑀯𑀦𑀸𑀦𑀺,_  
_𑀓𑀝𑁆𑀞𑀗𑁆𑀕𑀭𑀼𑀓𑁆𑀔𑁂𑀲𑀼 𑀅𑀲𑀸𑀭𑀓𑁂𑀲𑀼;_  
_𑀅𑀣𑀸𑀲𑀤𑀸 𑀔𑀤𑀺𑀭𑀁 𑀚𑀸𑀢𑀲𑀸𑀭𑀁,_  
_𑀬𑀢𑁆𑀣𑀩𑁆𑀪𑀺𑀤𑀸 𑀕𑀭𑀼𑀍𑀑 𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑀦𑁆”𑀢𑀺𑁇_  


---

535\. 𑀓𑀦𑁆𑀤𑀕𑀮𑀓𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇



---

536\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



537\. _𑀤𑀍𑀳𑀩𑀦𑁆𑀥𑀦 𑀳𑀁𑀲𑀯𑀭𑁄 𑀘 𑀧𑀼𑀦,_  
_𑀯𑀺𑀭𑀽𑀧𑀓𑁆𑀔 𑀲𑀯𑀺𑀝𑁆𑀞𑀓 𑀫𑀘𑁆𑀙𑀯𑀭𑁄;_  
_𑀲𑀓𑀼𑀭𑀼𑀗𑁆𑀕 𑀲𑀅𑀲𑁆𑀲𑀓 𑀅𑀫𑁆𑀩𑀯𑀭𑁄,_  
_𑀧𑀼𑀦 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀓𑁄 𑀕𑀭𑀼𑀍𑀏𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
