# 𑁨𑁇𑁬𑁇𑁬 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀫𑀺𑀕𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(111.)

521\. _“𑀇𑀗𑁆𑀖 𑀯𑀝𑁆𑀝𑀫𑀬𑀁 𑀧𑀸𑀲𑀁,_  
_𑀙𑀺𑀦𑁆𑀤 𑀤𑀦𑁆𑀢𑁂𑀳𑀺 𑀓𑀘𑁆𑀙𑀧;_  
_𑀅𑀳𑀁 𑀢𑀣𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀣𑀸 𑀦𑁂𑀳𑀺𑀢𑀺 𑀮𑀼𑀤𑁆𑀤𑀓𑁄”𑁇_  


(112.)

522\. _“𑀓𑀘𑁆𑀙𑀧𑁄 𑀧𑀸𑀯𑀺𑀲𑀻 𑀯𑀸𑀭𑀺𑀁,_  
_𑀓𑀼𑀭𑀼𑀗𑁆𑀕𑁄 𑀧𑀸𑀯𑀺𑀲𑀻 𑀯𑀦𑀁;_  
_𑀲𑀢𑀧𑀢𑁆𑀢𑁄 𑀤𑀼𑀫𑀕𑁆𑀕𑀫𑁆𑀳𑀸,_  
_𑀤𑀽𑀭𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑀸𑀦𑀬𑀻”𑀢𑀺𑁇_
