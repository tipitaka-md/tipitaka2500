# 𑁨𑁇𑁬𑁇𑁮 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀚𑀸𑀢𑀓

(115.)

527\. _“𑀅𑀮𑀁 𑀫𑁂𑀢𑁂𑀳𑀺 𑀅𑀫𑁆𑀩𑁂𑀳𑀺,_  
_𑀚𑀫𑁆𑀩𑀽𑀳𑀺 𑀧𑀦𑀲𑁂𑀳𑀺 𑀘;_  
_𑀬𑀸𑀦𑀺 𑀧𑀸𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲,_  
_𑀯𑀭𑀁 𑀫𑀬𑁆𑀳𑀁 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑁄𑁇_  


(116.)

528\. _𑀫𑀳𑀢𑀻 𑀯𑀢 𑀢𑁂 𑀩𑁄𑀦𑁆𑀤𑀺,_  
_𑀦 𑀘 𑀧𑀜𑁆𑀜𑀸 𑀢𑀤𑀽𑀧𑀺𑀓𑀸;_  
_𑀲𑀼𑀲𑀼𑀫𑀸𑀭 𑀯𑀜𑁆𑀘𑀺𑀢𑁄 𑀫𑁂𑀲𑀺,_  
_𑀕𑀘𑁆𑀙 𑀤𑀸𑀦𑀺 𑀬𑀣𑀸𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇_
