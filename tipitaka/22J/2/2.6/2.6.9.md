# 𑁨𑁇𑁬𑁇𑁯 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀚𑀸𑀢𑀓

(117.)

530\. _“𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸 𑀯𑀦𑁂 𑀭𑀼𑀓𑁆𑀔𑀸,_  
_𑀅𑀲𑁆𑀲𑀓𑀡𑁆𑀡𑀸 𑀯𑀺𑀪𑀻𑀝𑀓𑀸;_  
_𑀦 𑀢𑀸𑀦𑀺 𑀏𑀯𑀁 𑀲𑀓𑁆𑀓𑀦𑁆𑀢𑀺,_  
_𑀬𑀣𑀸 𑀢𑁆𑀯𑀁 𑀭𑀼𑀓𑁆𑀔 𑀲𑀓𑁆𑀓𑀲𑀺”𑁇_  


(118.)

531\. _“𑀧𑀼𑀭𑀸𑀡𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀅𑀬𑀁,_  
_𑀪𑁂𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀚𑀭𑀫𑀸𑀕𑀢𑁄;_  
_𑀓𑀼𑀲𑀮𑁄 𑀯𑀸𑀍𑀅𑀧𑀸𑀲𑀸𑀦𑀁,_  
_𑀅𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀪𑀸𑀲𑀢𑀻”𑀢𑀺𑁇_
