# 𑁨𑁇𑁬𑁇𑁨 𑀓𑁂𑀍𑀇𑀲𑀻𑀮𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

* 𑀓𑁂𑀍𑀇𑀲𑀻𑀮𑀚𑀸𑀢𑀓

(103.)

505\. _“𑀳𑀁𑀲𑀸 𑀓𑁄𑀜𑁆𑀘𑀸 𑀫𑀬𑀽𑀭𑀸 𑀘,_  
_𑀳𑀢𑁆𑀣𑀬𑁄 𑀧𑀲𑀤𑀸 𑀫𑀺𑀕𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀻𑀳𑀲𑁆𑀲 𑀪𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀦𑀢𑁆𑀣𑀺 𑀓𑀸𑀬𑀲𑁆𑀫𑀺 𑀢𑀼𑀮𑁆𑀬𑀢𑀸𑁇_  


(104.)

506\. _𑀏𑀯𑀫𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀤𑀳𑀭𑁄 𑀘𑁂𑀧𑀺 𑀧𑀜𑁆𑀜𑀯𑀸;_  
_𑀲𑁄 𑀳𑀺 𑀢𑀢𑁆𑀣 𑀫𑀳𑀸 𑀳𑁄𑀢𑀺,_  
_𑀦𑁂𑀯 𑀩𑀸𑀮𑁄 𑀲𑀭𑀻𑀭𑀯𑀸”𑀢𑀺𑁇_
