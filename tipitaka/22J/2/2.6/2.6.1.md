# 𑁨𑁇𑁬𑁇𑁧 𑀩𑀦𑁆𑀥𑀦𑀸𑀕𑀸𑀭𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

### 𑀩𑀦𑁆𑀥𑀦𑀸𑀕𑀸𑀭𑀚𑀸𑀢𑀓

(101.)

502. _“𑀦 𑀢𑀁 𑀤𑀍𑀳𑀁 𑀩𑀦𑁆𑀥𑀦𑀫𑀸𑀳𑀼 𑀥𑀻𑀭𑀸,_  
_𑀬𑀤𑀸𑀬𑀲𑀁 𑀤𑀸𑀭𑀼𑀚𑀧𑀩𑁆𑀩𑀚𑀜𑁆𑀘;_  
_𑀲𑀸𑀭𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀸 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑁂𑀲𑀼,_  
_𑀧𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀤𑀸𑀭𑁂𑀲𑀼 𑀘 𑀬𑀸 𑀅𑀧𑁂𑀓𑁆𑀔𑀸𑁇_  


(102.)

503. _𑀏𑀢𑀁 𑀤𑀍𑀳𑀁 𑀩𑀦𑁆𑀥𑀦𑀫𑀸𑀳𑀼 𑀥𑀻𑀭𑀸,_  
_𑀑𑀳𑀸𑀭𑀺𑀦𑀁 𑀲𑀻𑀣𑀺𑀮𑀁 𑀤𑀼𑀧𑁆𑀧𑀫𑀼𑀜𑁆𑀘𑀁;_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀙𑁂𑀢𑁆𑀯𑀸𑀦 𑀯𑀚𑀦𑁆𑀢𑀺 𑀥𑀻𑀭𑀸,_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀁 𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇_  


---
