# 𑁨𑁇𑁬𑁇𑁪 𑀯𑀻𑀭𑀓𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

* 𑀯𑀻𑀭𑀓𑀚𑀸𑀢𑀓

(107.)

515\. _“𑀅𑀧𑀺 𑀯𑀻𑀭𑀓 𑀧𑀲𑁆𑀲𑁂𑀲𑀺,_  
_𑀲𑀓𑀼𑀡𑀁 𑀫𑀜𑁆𑀚𑀼𑀪𑀸𑀡𑀓𑀁;_  
_𑀫𑀬𑀽𑀭𑀕𑀻𑀯𑀲𑀗𑁆𑀓𑀸𑀲𑀁,_  
_𑀧𑀢𑀺𑀁 𑀫𑀬𑁆𑀳𑀁 𑀲𑀯𑀺𑀝𑁆𑀞𑀓𑀁”𑁇_  


(108.)

516\. _“𑀉𑀤𑀓𑀣𑀮𑀘𑀭𑀲𑁆𑀲 𑀧𑀓𑁆𑀔𑀺𑀦𑁄,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀆𑀫𑀓𑀫𑀘𑁆𑀙𑀪𑁄𑀚𑀺𑀦𑁄;_  
_𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀓𑀭𑀁 𑀲𑀯𑀺𑀝𑁆𑀞𑀓𑁄,_  
_𑀲𑁂𑀯𑀸𑀮𑁂 𑀧𑀮𑀺𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑁄 𑀫𑀢𑁄”𑀢𑀺𑁇_
