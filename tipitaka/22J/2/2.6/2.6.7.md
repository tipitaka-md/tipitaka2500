# 𑁨𑁇𑁬𑁇𑁭 𑀅𑀲𑁆𑀲𑀓𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀦𑀢𑀁𑀤𑀍𑀳𑀯𑀕𑁆𑀕

* 𑀅𑀲𑁆𑀲𑀓𑀚𑀸𑀢𑀓

(113.)

524\. _“𑀅𑀬𑀫𑀲𑁆𑀲𑀓𑀭𑀸𑀚𑁂𑀦,_  
_𑀤𑁂𑀲𑁄 𑀯𑀺𑀘𑀭𑀺𑀢𑁄 𑀫𑀬𑀸;_  
_𑀅𑀦𑀼𑀓𑀸𑀫𑀬𑀓𑀸𑀫𑁂𑀦,_  
_𑀧𑀺𑀬𑁂𑀦 𑀧𑀢𑀺𑀦𑀸 𑀲𑀳𑁇_  


(114.)

525\. _𑀦𑀯𑁂𑀦 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑁂𑀦,_  
_𑀧𑁄𑀭𑀸𑀡𑀁 𑀅𑀧𑀺𑀥𑀻𑀬𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀓𑀭𑀜𑁆𑀜𑀸𑀯,_  
_𑀓𑀻𑀝𑁄 𑀧𑀺𑀬𑀢𑀭𑁄 𑀫𑀫𑀸”𑀢𑀺𑁇_
