# 𑁨𑁇𑁧𑁇𑁫 𑀪𑀕𑁆𑀕𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀍𑀳𑀯𑀕𑁆𑀕

### 𑀪𑀕𑁆𑀕𑀚𑀸𑀢𑀓

(9.)

352. _“𑀚𑀻𑀯 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀪𑀕𑁆𑀕,_  
_𑀅𑀧𑀭𑀸𑀦𑀺 𑀘 𑀯𑀻𑀲𑀢𑀺𑀁;_  
_𑀫𑀸 𑀫𑀁 𑀧𑀺𑀲𑀸𑀘𑀸 𑀔𑀸𑀤𑀦𑁆𑀢𑀼,_  
_𑀚𑀻𑀯 𑀢𑁆𑀯𑀁 𑀲𑀭𑀤𑁄 𑀲𑀢𑀁”𑁇_  


(10.)

353. _“𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀚𑀻𑀯,_  
_𑀅𑀧𑀭𑀸𑀦𑀺 𑀘 𑀯𑀻𑀲𑀢𑀺𑀁;_  
_𑀯𑀺𑀲𑀁 𑀧𑀺𑀲𑀸𑀘𑀸 𑀔𑀸𑀤𑀦𑁆𑀢𑀼,_  
_𑀚𑀻𑀯 𑀢𑁆𑀯𑀁 𑀲𑀭𑀤𑁄 𑀲𑀢𑀦𑁆”𑀢𑀺𑁇_
