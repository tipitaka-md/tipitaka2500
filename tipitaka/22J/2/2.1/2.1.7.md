# 𑁨𑁇𑁧𑁇𑁭 𑀕𑀼𑀡𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀍𑀳𑀯𑀕𑁆𑀕

### 𑀕𑀼𑀡𑀚𑀸𑀢𑀓

(13.)

358\. _“𑀬𑁂𑀦 𑀓𑀸𑀫𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀺,_  
_𑀥𑀫𑁆𑀫𑁄 𑀩𑀮𑀯𑀢𑀁 𑀫𑀺𑀕𑀻;_  
_𑀉𑀦𑁆𑀦𑀤𑀦𑁆𑀢𑀻 𑀯𑀺𑀚𑀸𑀦𑀸𑀳𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(14.)

359\. _“𑀅𑀧𑀺 𑀘𑁂𑀧𑀺 𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀫𑀺𑀢𑁆𑀢𑁄,_  
_𑀫𑀺𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀢𑀺𑀝𑁆𑀞𑀢𑀺;_  
_𑀲𑁄 𑀜𑀸𑀢𑀓𑁄 𑀘 𑀩𑀦𑁆𑀥𑀼 𑀘,_  
_𑀲𑁄 𑀫𑀺𑀢𑁆𑀢𑁄 𑀲𑁄 𑀘 𑀫𑁂 𑀲𑀔𑀸;_  
_𑀤𑀸𑀞𑀺𑀦𑀺 𑀫𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀺𑀢𑁆𑀣𑁄,_  
_𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀫𑀫 𑀧𑀸𑀡𑀤𑁄”𑀢𑀺𑁇_  


---
