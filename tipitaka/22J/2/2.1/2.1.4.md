# 𑁨𑁇𑁧𑁇𑁪 𑀉𑀭𑀕𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀤𑀍𑀳𑀯𑀕𑁆𑀕

### 𑀉𑀭𑀕𑀚𑀸𑀢𑀓

(7.)

349. _“𑀇𑀥𑀽𑀭𑀕𑀸𑀦𑀁 𑀧𑀯𑀭𑁄 𑀧𑀯𑀺𑀝𑁆𑀞𑁄,_  
_𑀲𑁂𑀮𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑁂𑀦 𑀧𑀫𑁄𑀓𑁆𑀔𑀫𑀺𑀘𑁆𑀙𑀁;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀘 𑀯𑀡𑁆𑀡𑀁 𑀅𑀧𑀘𑀸𑀬𑀫𑀸𑀦𑁄,_  
_𑀩𑀼𑀪𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀦𑁄 𑀯𑀺𑀢𑀭𑀸𑀫𑀺 𑀪𑁄𑀢𑁆𑀢𑀼𑀁”𑁇_  


(8.)

350. _“𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀕𑀼𑀢𑁆𑀢𑁄 𑀘𑀺𑀭𑀫𑁂𑀯 𑀚𑀻𑀯,_  
_𑀤𑀺𑀩𑁆𑀬𑀸 𑀘 𑀢𑁂 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀼 𑀪𑀓𑁆𑀔𑀸;_  
_𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀡𑁆𑀡𑀁 𑀅𑀧𑀘𑀸𑀬𑀫𑀸𑀦𑁄,_  
_𑀩𑀼𑀪𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀦𑁄 𑀯𑀺𑀢𑀭𑀸𑀲𑀺 𑀪𑁄𑀢𑁆𑀢𑀼𑀦𑁆”𑀢𑀺𑁇_  


---

351. 𑀉𑀭𑀕𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
