# 𑁨𑁇𑁧𑁇𑁮 𑀲𑀼𑀳𑀦𑀼𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀤𑀍𑀳𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀳𑀦𑀼𑀚𑀸𑀢𑀓

(15.)

361\. _“𑀦𑀬𑀺𑀤𑀁 𑀯𑀺𑀲𑀫𑀲𑀻𑀮𑁂𑀦,_  
_𑀲𑁄𑀡𑁂𑀦 𑀲𑀼𑀳𑀦𑀽 𑀲𑀳;_  
_𑀲𑀼𑀳𑀦𑀽𑀧𑀺 𑀢𑀸𑀤𑀺𑀲𑁄𑀬𑁂𑀯,_  
_𑀬𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀲𑀕𑁄𑀘𑀭𑁄𑁇_  


(16.)

362\. _𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀺𑀦𑀸 𑀧𑀕𑀩𑁆𑀪𑁂𑀦,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑀦𑁆𑀤𑀸𑀦𑀔𑀸𑀤𑀺𑀦𑀸;_  
_𑀲𑀫𑁂𑀢𑀺 𑀧𑀸𑀧𑀁 𑀧𑀸𑀧𑁂𑀦,_  
_𑀲𑀫𑁂𑀢𑀺 𑀅𑀲𑀢𑀸 𑀅𑀲𑀦𑁆”𑀢𑀺𑁇_
