# 𑁨𑁇𑁧𑁦𑁇𑁧𑁦 𑀓𑀧𑀺𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀓𑀧𑀺𑀚𑀸𑀢𑀓

(200.)

662\. _“𑀅𑀬𑀁 𑀇𑀲𑀻 𑀉𑀧𑀲𑀫𑀲𑀁𑀬𑀫𑁂 𑀭𑀢𑁄,_  
_𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀲𑀺𑀲𑀺𑀭𑀪𑀬𑁂𑀦 𑀅𑀝𑁆𑀝𑀺𑀢𑁄;_  
_𑀳𑀦𑁆𑀤 𑀅𑀬𑀁 𑀧𑀯𑀺𑀲𑀢𑀼𑀫𑀁 𑀅𑀕𑀸𑀭𑀓𑀁,_  
_𑀯𑀺𑀦𑁂𑀢𑀼 𑀲𑀻𑀢𑀁 𑀤𑀭𑀣𑀜𑁆𑀘 𑀓𑁂𑀯𑀮𑀁”𑁇_  


(201.)

663\. _“𑀦𑀸𑀬𑀁 𑀇𑀲𑀻 𑀉𑀧𑀲𑀫𑀲𑀁𑀬𑀫𑁂 𑀭𑀢𑁄,_  
_𑀓𑀧𑀻 𑀅𑀬𑀁 𑀤𑀼𑀫𑀯𑀭𑀲𑀸𑀔𑀕𑁄𑀘𑀭𑁄;_  
_𑀲𑁄 𑀤𑀽𑀲𑀓𑁄 𑀭𑁄𑀲𑀓𑁄 𑀘𑀸𑀧𑀺 𑀚𑀫𑁆𑀫𑁄,_  
_𑀲𑀘𑁂𑀯𑀚𑁂𑀫𑀫𑁆𑀧𑀺 𑀤𑀽𑀲𑁂𑀬𑁆𑀬𑀕𑀸𑀭𑀦𑁆”𑀢𑀺𑁇 (𑁩𑁫𑁧)_  


---

664\. 𑀓𑀧𑀺𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀯𑀕𑁆𑀕𑁄 𑀤𑀲𑀫𑁄𑁇



---

665\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



666\. _𑀅𑀣 𑀭𑀸𑀚𑀸 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀯𑀭𑁄 𑀲𑀼𑀦𑀔𑁄,_  
_𑀢𑀣𑀸 𑀓𑁄𑀲𑀺𑀬 𑀇𑀘𑁆𑀙𑀢𑀺 𑀓𑀸𑀮𑀖𑀲𑁄;_  
_𑀅𑀣 𑀤𑀸𑀦𑀯𑀭𑁄𑀝𑁆𑀞𑀧𑀺 𑀲𑀸𑀭𑀣𑀺𑀦𑀸,_  
_𑀧𑀼𑀦𑀫𑁆𑀩𑀯𑀦𑀜𑁆𑀘 𑀲𑀺𑀲𑀺𑀭𑀓𑀧𑀺 𑀤𑀲𑀸𑀢𑀺𑁇_  


---

667\. 𑀅𑀣 𑀯𑀕𑁆𑀕𑀼𑀤𑁆𑀤𑀸𑀦𑀁



668\. _𑀤𑀍𑀳𑀜𑁆𑀘 𑀯𑀕𑁆𑀕𑀁 𑀅𑀧𑀭𑁂𑀦 𑀲𑀦𑁆𑀣𑀯𑀁,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀕𑁆𑀕𑀸𑀲𑀤𑀺𑀲𑁄 𑀘 𑀭𑀽𑀳𑀓𑀁;_  
_𑀦𑀢𑀁𑀤𑀍𑀳 𑀩𑀻𑀭𑀡𑀣𑀫𑁆𑀪𑀓𑀁 𑀧𑀼𑀦,_  
_𑀓𑀸𑀲𑀸𑀯𑀼𑀧𑀸𑀳𑀦 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀓𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_  


---

669\. 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
