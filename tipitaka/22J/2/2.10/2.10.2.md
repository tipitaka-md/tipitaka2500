# 𑁨𑁇𑁧𑁦𑁇𑁨 𑀲𑀼𑀦𑀔𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀦𑀔𑀚𑀸𑀢𑀓

(184.)

638\. _“𑀩𑀸𑀮𑁄 𑀯𑀢𑀸𑀬𑀁 𑀲𑀼𑀦𑀔𑁄,_  
_𑀬𑁄 𑀯𑀭𑀢𑁆𑀢𑀁 𑀦 𑀔𑀸𑀤𑀢𑀺;_  
_𑀩𑀦𑁆𑀥𑀦𑀸 𑀘 𑀧𑀫𑀼𑀜𑁆𑀘𑁂𑀬𑁆𑀬,_  
_𑀅𑀲𑀺𑀢𑁄 𑀘 𑀖𑀭𑀁 𑀯𑀚𑁂”𑁇_  


(185.)

639\. _“𑀅𑀝𑁆𑀞𑀺𑀢𑀁 𑀫𑁂 𑀫𑀦𑀲𑁆𑀫𑀺𑀁 𑀫𑁂,_  
_𑀅𑀣𑁄 𑀫𑁂 𑀳𑀤𑀬𑁂 𑀓𑀢𑀁;_  
_𑀓𑀸𑀮𑀜𑁆𑀘 𑀧𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺,_  
_𑀬𑀸𑀯 𑀧𑀲𑁆𑀲𑀼𑀧𑀢𑀽 𑀚𑀦𑁄”𑀢𑀺𑁇_
