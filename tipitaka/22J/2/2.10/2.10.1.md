# 𑁨𑁇𑁧𑁦𑁇𑁧 𑀲𑀩𑁆𑀩𑀤𑀸𑀞𑀺𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀲𑀩𑁆𑀩𑀤𑀸𑀞𑀺𑀚𑀸𑀢𑀓

(182.)

635\. _“𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀫𑀸𑀦𑀣𑀤𑁆𑀥𑁄 𑀘,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀧𑀸𑀧𑀼𑀡𑀺 𑀫𑀳𑀢𑀺𑀁 𑀪𑀽𑀫𑀺𑀁,_  
_𑀭𑀸𑀚𑀸𑀲𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑀞𑀺𑀦𑀁𑁇_  


(183.)

636\. _𑀏𑀯𑀫𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀬𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀯𑀸𑀭𑀯𑀸;_  
_𑀲𑁄 𑀳𑀺 𑀢𑀢𑁆𑀣 𑀫𑀳𑀸 𑀳𑁄𑀢𑀺,_  
_𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀯𑀺𑀬 𑀤𑀸𑀞𑀺𑀦𑀦𑁆”𑀢𑀺𑁇_
