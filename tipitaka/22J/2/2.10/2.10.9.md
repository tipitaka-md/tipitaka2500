# 𑁨𑁇𑁧𑁦𑁇𑁯 𑀲𑀸𑀮𑀓𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀮𑀓𑀚𑀸𑀢𑀓

(198.)

659\. _“𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺,_  
_𑀢𑁆𑀯𑀜𑁆𑀘 𑀦𑁄 𑀳𑁂𑀲𑁆𑀲𑀲𑀺 𑀇𑀲𑁆𑀲𑀭𑁄 𑀓𑀼𑀮𑁂;_  
_𑀑𑀭𑁄𑀳 𑀤𑀼𑀫𑀲𑁆𑀫𑀸 𑀲𑀸𑀮𑀓,_  
_𑀏𑀳𑀺 𑀤𑀸𑀦𑀺 𑀖𑀭𑀓𑀁 𑀯𑀚𑁂𑀫𑀲𑁂”𑁇_  


(199.)

660\. _“𑀦𑀦𑀼 𑀫𑀁 𑀲𑀼𑀳𑀤𑀬𑁄𑀢𑀺 𑀫𑀜𑁆𑀜𑀲𑀺,_  
_𑀬𑀜𑁆𑀘 𑀫𑀁 𑀳𑀦𑀲𑀺 𑀯𑁂𑀍𑀉𑀬𑀝𑁆𑀞𑀺𑀬𑀸;_  
_𑀧𑀓𑁆𑀓𑀫𑁆𑀩𑀯𑀦𑁂 𑀭𑀫𑀸𑀫𑀲𑁂,_  
_𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁 𑀖𑀭𑀓𑀁 𑀬𑀣𑀸𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇_  


---
