# 𑁨𑁇𑁧𑁦𑁇𑁭 𑀧𑀸𑀤𑀜𑁆𑀚𑀮𑀻𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀸𑀤𑀜𑁆𑀚𑀮𑀻𑀚𑀸𑀢𑀓

(194.)

653\. _“𑀅𑀤𑁆𑀥𑀸 𑀧𑀸𑀤𑀜𑁆𑀚𑀮𑀻 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺;_  
_𑀢𑀣𑀸 𑀳𑀺 𑀑𑀝𑁆𑀞𑀁 𑀪𑀜𑁆𑀚𑀢𑀺,_  
_𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀦𑀽𑀦 𑀧𑀲𑁆𑀲𑀢𑀺”𑁇_  


(195.)

654\. _“𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑀁 𑀯𑀸,_  
_𑀅𑀢𑁆𑀣𑀸𑀦𑀢𑁆𑀣𑀜𑁆𑀘 𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀑𑀝𑁆𑀞𑀦𑀺𑀩𑁆𑀪𑁄𑀕𑀸,_  
_𑀦𑀸𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀦𑁆”𑀢𑀺𑁇_  


---
