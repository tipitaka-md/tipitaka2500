# 𑁨𑁇𑁭𑁇𑁫 𑀓𑀘𑁆𑀙𑀧𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀩𑀻𑀭𑀡𑀣𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀓𑀘𑁆𑀙𑀧𑀚𑀸𑀢𑀓

(129.)

550\. _“𑀅𑀯𑀥𑀻 𑀯𑀢 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀓𑀘𑁆𑀙𑀧𑁄 𑀩𑁆𑀬𑀸𑀳𑀭𑀁 𑀕𑀺𑀭𑀁;_  
_𑀲𑀼𑀕𑁆𑀕𑀳𑀻𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀝𑁆𑀞𑀲𑁆𑀫𑀺𑀁,_  
_𑀯𑀸𑀘𑀸𑀬 𑀲𑀓𑀺𑀬𑀸𑀯𑀥𑀺𑁇_  


(130.)

551\. _𑀏𑀢𑀫𑁆𑀧𑀺 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀭𑀯𑀻𑀭𑀺𑀬𑀲𑁂𑀝𑁆𑀞,_  
_𑀯𑀸𑀘𑀁 𑀧𑀫𑀼𑀜𑁆𑀘𑁂 𑀓𑀼𑀲𑀮𑀁 𑀦𑀸𑀢𑀺𑀯𑁂𑀮𑀁;_  
_𑀧𑀲𑁆𑀲𑀲𑀺 𑀩𑀳𑀼𑀪𑀸𑀡𑁂𑀦,_  
_𑀓𑀘𑁆𑀙𑀧𑀁 𑀩𑁆𑀬𑀲𑀦𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇_  


---
