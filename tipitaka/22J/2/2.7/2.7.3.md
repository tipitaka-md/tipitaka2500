# 𑁨𑁇𑁭𑁇𑁩 𑀪𑀭𑀼𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀩𑀻𑀭𑀡𑀣𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀪𑀭𑀼𑀚𑀸𑀢𑀓

(125.)

544\. _“𑀇𑀲𑀻𑀦𑀫𑀦𑁆𑀢𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀪𑀭𑀼𑀭𑀸𑀚𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀲𑀳 𑀭𑀝𑁆𑀞𑁂𑀳𑀺,_  
_𑀲𑀭𑀸𑀚𑀸 𑀯𑀺𑀪𑀯𑀗𑁆𑀕𑀢𑁄𑁇_  


(126.)

545\. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀙𑀦𑁆𑀤𑀸𑀕𑀫𑀦𑀁,_  
_𑀦𑀧𑁆𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀅𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀕𑀺𑀭𑀁 𑀲𑀘𑁆𑀘𑀼𑀧𑀲𑀁𑀳𑀺𑀢𑀦𑁆”𑀢𑀺𑁇_  


---
