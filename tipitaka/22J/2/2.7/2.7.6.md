# 𑁨𑁇𑁭𑁇𑁬 𑀫𑀘𑁆𑀙𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀩𑀻𑀭𑀡𑀣𑀫𑁆𑀪𑀯𑀕𑁆𑀕

* 𑀫𑀘𑁆𑀙𑀚𑀸𑀢𑀓

(131.)

553\. _“𑀦 𑀫𑀸𑀬𑀫𑀕𑁆𑀕𑀺 𑀢𑀧𑀢𑀺,_  
_𑀦 𑀲𑀽𑀮𑁄 𑀲𑀸𑀥𑀼𑀢𑀘𑁆𑀙𑀺𑀢𑁄;_  
_𑀬𑀜𑁆𑀘 𑀫𑀁 𑀫𑀜𑁆𑀜𑀢𑁂 𑀫𑀘𑁆𑀙𑀻,_  
_𑀅𑀜𑁆𑀜𑀁 𑀲𑁄 𑀭𑀢𑀺𑀬𑀸 𑀕𑀢𑁄𑁇_  


(132.)

554\. _𑀲𑁄 𑀫𑀁 𑀤𑀳𑀢𑀺 𑀭𑀸𑀕𑀕𑁆𑀕𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀽𑀧𑀢𑀧𑁂𑀢𑀺 𑀫𑀁;_  
_𑀚𑀸𑀮𑀺𑀦𑁄 𑀫𑀼𑀜𑁆𑀘𑀣𑀸𑀬𑀺𑀭𑀸 𑀫𑀁,_  
_𑀦 𑀓𑀸𑀫𑁂 𑀳𑀜𑁆𑀜𑀢𑁂 𑀓𑁆𑀯𑀘𑀻”𑀢𑀺𑁇_
