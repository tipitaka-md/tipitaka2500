# 𑁨𑁇𑁫𑁇𑁨 𑀲𑀺𑀭𑀺𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀺𑀭𑀺𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓

(83.)

473. _“𑀇𑀢𑁆𑀣𑀻 𑀲𑀺𑀬𑀸 𑀭𑀽𑀧𑀯𑀢𑀻,_  
_𑀲𑀸 𑀘 𑀲𑀻𑀮𑀯𑀢𑀻 𑀲𑀺𑀬𑀸;_  
_𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀁 𑀦 𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_𑀲𑀤𑁆𑀤𑀳𑀸𑀲𑀺 𑀫𑀳𑁄𑀲𑀥”𑁇_  


(84.)

474. _“𑀲𑀤𑁆𑀤𑀳𑀸𑀫𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀼𑀩𑁆𑀪𑀕𑁄 𑀲𑀺𑀬𑀸;_  
_𑀲𑀺𑀭𑀻 𑀘 𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀻 𑀘,_  
_𑀦 𑀲𑀫𑁂𑀦𑁆𑀢𑀺 𑀓𑀼𑀤𑀸𑀘𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

475. 𑀲𑀺𑀭𑀺𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
