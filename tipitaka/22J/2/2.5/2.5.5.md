# 𑁨𑁇𑁫𑁇𑁫 𑀧𑀩𑁆𑀩𑀢𑀽𑀧𑀢𑁆𑀣𑀭𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀩𑁆𑀩𑀢𑀽𑀧𑀢𑁆𑀣𑀭𑀚𑀸𑀢𑀓

(89.)

482\. _“𑀧𑀩𑁆𑀩𑀢𑀽𑀧𑀢𑁆𑀣𑀭𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀚𑀸𑀢𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀲𑀺𑀯𑀸;_  
_𑀢𑀁 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀅𑀧𑀸𑀧𑀸𑀬𑀺,_  
_𑀚𑀸𑀦𑀁 𑀲𑀻𑀳𑁂𑀦 𑀭𑀓𑁆𑀔𑀺𑀢𑀁”𑁇_  


(90.)

483\. _“𑀧𑀺𑀯𑀦𑁆𑀢𑀺 𑀘𑁂 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀲𑀸𑀧𑀤𑀸𑀦𑀺 𑀫𑀳𑀸𑀦𑀤𑀺𑀁;_  
_𑀦 𑀢𑁂𑀦 𑀅𑀦𑀤𑀻 𑀳𑁄𑀢𑀺,_  
_𑀔𑀫𑀲𑁆𑀲𑀼 𑀬𑀤𑀺 𑀢𑁂 𑀧𑀺𑀬𑀸”𑀢𑀺𑁇_  


---
