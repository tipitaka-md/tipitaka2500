# 𑁨𑁇𑁫𑁇𑁯 𑀕𑀳𑀧𑀢𑀺𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀕𑀳𑀧𑀢𑀺𑀚𑀸𑀢𑀓

(97.)

494\. _“𑀉𑀪𑀬𑀁 𑀫𑁂 𑀦 𑀔𑀫𑀢𑀺,_  
_𑀉𑀪𑀬𑀁 𑀫𑁂 𑀦 𑀭𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀬𑀸𑀘𑀸𑀬𑀁 𑀓𑁄𑀝𑁆𑀞𑀫𑁄𑀢𑀺𑀡𑁆𑀡𑀸,_  
_𑀦𑀸𑀤𑁆𑀤𑀲𑀁 𑀇𑀢𑀺 𑀪𑀸𑀲𑀢𑀺𑁇_  


(98.)

495\. _𑀢𑀁 𑀢𑀁 𑀕𑀸𑀫𑀧𑀢𑀺 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀓𑀤𑀭𑁂 𑀅𑀧𑁆𑀧𑀲𑁆𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑁂;_  
_𑀤𑁆𑀯𑁂 𑀫𑀸𑀲𑁂 𑀲𑀗𑁆𑀕𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀫𑀁𑀲𑀁 𑀚𑀭𑀕𑁆𑀕𑀯𑀁 𑀓𑀺𑀲𑀁;_  
_𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀓𑀸𑀮𑁂 𑀘𑁄𑀤𑁂𑀲𑀺,_  
_𑀢𑀫𑁆𑀧𑀺 𑀫𑀬𑁆𑀳𑀁 𑀦 𑀭𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


---
