# 𑁨𑁇𑁫𑁇𑁮 𑀭𑀸𑀥𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

* 𑀭𑀸𑀥𑀚𑀸𑀢𑀓

(95.)

491\. _“𑀧𑀯𑀸𑀲𑀸 𑀆𑀕𑀢𑁄 𑀢𑀸𑀢,_  
_𑀇𑀤𑀸𑀦𑀺 𑀦𑀘𑀺𑀭𑀸𑀕𑀢𑁄;_  
_𑀓𑀘𑁆𑀘𑀺𑀦𑁆𑀦𑀼 𑀢𑀸𑀢 𑀢𑁂 𑀫𑀸𑀢𑀸,_  
_𑀦 𑀅𑀜𑁆𑀜𑀫𑀼𑀧𑀲𑁂𑀯𑀢𑀺”𑁇_  


(96.)

492\. _“𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀲𑀼𑀪𑀡𑀁,_  
_𑀕𑀺𑀭𑀁 𑀲𑀘𑁆𑀘𑀼𑀧𑀲𑀁𑀳𑀺𑀢𑀁;_  
_𑀲𑀬𑁂𑀣 𑀧𑁄𑀝𑁆𑀞𑀧𑀸𑀤𑁄𑀯,_  
_𑀫𑀼𑀫𑁆𑀫𑀼𑀭𑁂 𑀉𑀧𑀓𑀽𑀣𑀺𑀢𑁄”𑀢𑀺𑁇_
