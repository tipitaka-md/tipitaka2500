# 𑁨𑁇𑁫𑁇𑁪 𑀫𑀡𑀺𑀘𑁄𑀭𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀫𑀡𑀺𑀘𑁄𑀭𑀚𑀸𑀢𑀓

(87.)

479\. _“𑀦 𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀯𑀸 𑀧𑀯𑀲𑀦𑁆𑀢𑀺 𑀦𑀽𑀦,_  
_𑀦 𑀳𑀺 𑀦𑀽𑀦 𑀲𑀦𑁆𑀢𑀺 𑀇𑀥 𑀮𑁄𑀓𑀧𑀸𑀮𑀸;_  
_𑀲𑀳𑀲𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀸𑀦𑀫𑀲𑀜𑁆𑀜𑀢𑀸𑀦𑀁,_  
_𑀦 𑀳𑀺 𑀦𑀽𑀦 𑀲𑀦𑁆𑀢𑀻 𑀧𑀝𑀺𑀲𑁂𑀥𑀺𑀢𑀸𑀭𑁄”𑁇_  


(88.)

480\. _“𑀅𑀓𑀸𑀮𑁂 𑀯𑀲𑁆𑀲𑀢𑀻 𑀢𑀲𑁆𑀲,_  
_𑀓𑀸𑀮𑁂 𑀢𑀲𑁆𑀲 𑀦 𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀕𑁆𑀕𑀸 𑀘 𑀘𑀯𑀢𑀺 𑀞𑀸𑀦𑀸,_  
_𑀦𑀦𑀼 𑀲𑁄 𑀢𑀸𑀯𑀢𑀸 𑀳𑀢𑁄”𑀢𑀺𑁇_  


---
