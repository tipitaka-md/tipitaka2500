# 𑁨𑁇𑁫𑁇𑁬 𑀯𑀮𑀸𑀳𑀓𑀲𑁆𑀲𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀯𑀮𑀸𑀳𑀓𑀲𑁆𑀲𑀚𑀸𑀢𑀓

(91.)

485\. _“𑀬𑁂 𑀦 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀑𑀯𑀸𑀤𑀁,_  
_𑀦𑀭𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀩𑁆𑀬𑀲𑀦𑀁 𑀢𑁂 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀭𑀓𑁆𑀔𑀲𑀻𑀳𑀺𑀯 𑀯𑀸𑀡𑀺𑀚𑀸𑁇_  


(92.)

486\. _𑀬𑁂 𑀘 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀑𑀯𑀸𑀤𑀁,_  
_𑀦𑀭𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀲𑁄𑀢𑁆𑀣𑀺𑀁 𑀧𑀸𑀭𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀯𑀮𑀸𑀳𑁂𑀦𑁂𑀯 𑀯𑀸𑀡𑀺𑀚𑀸”𑀢𑀺𑁇_
