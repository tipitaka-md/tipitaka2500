# 𑁨𑁇𑁫𑁇𑁭 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀺𑀢𑁆𑀢𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀺𑀢𑁆𑀢𑀚𑀸𑀢𑀓

(93.)

488\. _“𑀦 𑀦𑀁 𑀉𑀫𑁆𑀳𑀬𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀦 𑀘 𑀦𑀁 𑀧𑀝𑀺𑀦𑀦𑁆𑀤𑀢𑀺;_  
_𑀘𑀓𑁆𑀔𑀽𑀦𑀺 𑀘𑀲𑁆𑀲 𑀦 𑀤𑀤𑀸𑀢𑀺,_  
_𑀧𑀝𑀺𑀮𑁄𑀫𑀜𑁆𑀘 𑀯𑀢𑁆𑀢𑀢𑀺𑁇_  


(94.)

489\. _𑀏𑀢𑁂 𑀪𑀯𑀦𑁆𑀢𑀺 𑀆𑀓𑀸𑀭𑀸,_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀬𑁂𑀳𑀺 𑀅𑀫𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄”𑀢𑀺𑁇_  


---

490\. 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀺𑀢𑁆𑀢𑀚𑀸𑀢𑀓𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
