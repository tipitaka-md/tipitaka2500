# 𑁨𑁇𑁫𑁇𑁧 𑀭𑀼𑀳𑀓𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

* 𑀭𑀼𑀳𑀓𑀚𑀸𑀢𑀓

(81.)

470\. _“𑀅𑀧𑀺 𑀭𑀼𑀳𑀓 𑀙𑀺𑀦𑁆𑀦𑀸𑀧𑀺,_  
_𑀚𑀺𑀬𑀸 𑀲𑀦𑁆𑀥𑀻𑀬𑀢𑁂 𑀧𑀼𑀦;_  
_𑀲𑀦𑁆𑀥𑀻𑀬𑀲𑁆𑀲𑀼 𑀧𑀼𑀭𑀸𑀡𑀺𑀬𑀸,_  
_𑀫𑀸 𑀓𑁄𑀥𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀫𑀺”𑁇_  


(82.)

471\. _“𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀲𑀼 𑀯𑀸𑀓𑁂𑀲𑀼,_  
_𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀲𑀼 𑀓𑀸𑀭𑀺𑀲𑀼;_  
_𑀅𑀜𑁆𑀜𑀁 𑀚𑀺𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀅𑀮𑀜𑁆𑀜𑁂𑀯 𑀧𑀼𑀭𑀸𑀡𑀺𑀬𑀸”𑀢𑀺𑁇_
