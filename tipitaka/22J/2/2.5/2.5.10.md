# 𑁨𑁇𑁫𑁇𑁧𑁦 𑀲𑀸𑀥𑀼𑀲𑀻𑀮𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀥𑀼𑀲𑀻𑀮𑀚𑀸𑀢𑀓

(99.)

497\. _“𑀲𑀭𑀻𑀭𑀤𑀩𑁆𑀬𑀁 𑀯𑀼𑀟𑁆𑀠𑀩𑁆𑀬𑀁,_  
_𑀲𑁄𑀚𑀘𑁆𑀘𑀁 𑀲𑀸𑀥𑀼𑀲𑀻𑀮𑀺𑀬𑀁;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀢𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀁 𑀦𑀼 𑀢𑁂𑀲𑀁 𑀯𑀦𑀺𑀫𑁆𑀳𑀲𑁂”𑁇_  


(100.)

498\. _“𑀅𑀢𑁆𑀣𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀭𑀻𑀭𑀲𑁆𑀫𑀺𑀁,_  
_𑀯𑀼𑀟𑁆𑀠𑀩𑁆𑀬𑀲𑁆𑀲 𑀦𑀫𑁄 𑀓𑀭𑁂;_  
_𑀅𑀢𑁆𑀣𑁄 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀚𑀸𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀻𑀮𑀁 𑀅𑀲𑁆𑀫𑀸𑀓 𑀭𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_  


---

499\. 𑀲𑀸𑀥𑀼𑀲𑀻𑀮𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

500\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



501\. _𑀅𑀧𑀺𑀭𑀼𑀳𑀓 𑀭𑀽𑀧𑀯𑀢𑀻 𑀫𑀼𑀲𑀮𑁄,_  
_𑀧𑀯𑀲𑀦𑁆𑀢𑀺 𑀲𑀧𑀜𑁆𑀘𑀫𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻;_  
_𑀅𑀣 𑀫𑀼𑀢𑁆𑀢𑀺𑀫𑀯𑀸𑀡𑀺𑀚 𑀉𑀫𑁆𑀳𑀬𑀢𑁂,_  
_𑀘𑀺𑀭𑀆𑀕𑀢 𑀓𑁄𑀝𑁆𑀞 𑀲𑀭𑀻𑀭 𑀤𑀲𑀸𑀢𑀺𑁇_
