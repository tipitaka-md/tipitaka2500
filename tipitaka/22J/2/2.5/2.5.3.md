# 𑁨𑁇𑁫𑁇𑁩 𑀘𑀽𑀍𑀅𑀧𑀤𑀼𑀫𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀭𑀼𑀳𑀓𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀧𑀤𑀼𑀫𑀚𑀸𑀢𑀓

(85.)

476\. _“𑀅𑀬𑀫𑁂𑀯 𑀲𑀸 𑀅𑀳𑀫𑀧𑀺 𑀲𑁄 𑀅𑀦𑀜𑁆𑀜𑁄,_  
_𑀅𑀬𑀫𑁂𑀯 𑀲𑁄 𑀳𑀢𑁆𑀣𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀦𑀜𑁆𑀜𑁄;_  
_𑀬𑀫𑀸𑀳 ‘𑀓𑁄𑀫𑀸𑀭𑀧𑀢𑀻 𑀫𑀫𑀦𑁆’𑀢𑀺,_  
_𑀯𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀺𑀬𑁄 𑀦𑀢𑁆𑀣𑀺 𑀇𑀢𑁆𑀣𑀻𑀲𑀼 𑀲𑀘𑁆𑀘𑀁𑁇_  


(86.)

477\. _𑀇𑀫𑀜𑁆𑀘 𑀚𑀫𑁆𑀫𑀁 𑀫𑀼𑀲𑀮𑁂𑀦 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀮𑀼𑀤𑁆𑀤𑀁 𑀙𑀯𑀁 𑀧𑀭𑀤𑀸𑀭𑀽𑀧𑀲𑁂𑀯𑀺𑀁;_  
_𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀘 𑀦𑀁 𑀧𑀸𑀧𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀸𑀬,_  
_𑀚𑀻𑀯𑀦𑁆𑀢𑀺𑀬𑀸 𑀙𑀺𑀦𑁆𑀤𑀣 𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀦𑁆”𑀢𑀺𑁇_  


---

478\. 𑀘𑀽𑀍𑀅𑀧𑀤𑀼𑀫𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
