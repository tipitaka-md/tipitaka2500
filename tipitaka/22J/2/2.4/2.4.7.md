# 𑁨𑁇𑁪𑁇𑁭 𑀘𑀢𑀼𑀫𑀝𑁆𑀞𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀘𑀢𑀼𑀫𑀝𑁆𑀞𑀚𑀸𑀢𑀓

(73.)

456\. _“𑀉𑀘𑁆𑀘𑁂 𑀯𑀺𑀝𑀪𑀺𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀫𑀦𑁆𑀢𑀬𑀯𑁆𑀳𑁄 𑀭𑀳𑁄𑀕𑀢𑀸;_  
_𑀦𑀻𑀘𑁂 𑀑𑀭𑀼𑀬𑁆𑀳 𑀫𑀦𑁆𑀢𑀯𑁆𑀳𑁄,_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀸𑀧𑀺 𑀲𑁄𑀲𑁆𑀲𑀢𑀺𑁇_  


(74.)

457\. _𑀬𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑁄 𑀲𑀼𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀤𑁂𑀯𑁄 𑀤𑁂𑀯𑁂𑀦 𑀫𑀦𑁆𑀢𑀬𑁂;_  
_𑀓𑀺𑀁 𑀢𑁂𑀢𑁆𑀣 𑀘𑀢𑀼𑀫𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀩𑀺𑀮𑀁 𑀧𑀯𑀺𑀲 𑀚𑀫𑁆𑀩𑀼𑀓𑀸”𑀢𑀺𑁇_  


---
