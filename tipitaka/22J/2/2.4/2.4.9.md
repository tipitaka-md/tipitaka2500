# 𑁨𑁇𑁪𑁇𑁯 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑀚𑀸𑀢𑀓

(77.)

462\. _“𑀦𑁂𑀢𑀁 𑀲𑀻𑀳𑀲𑁆𑀲 𑀦𑀤𑀺𑀢𑀁,_  
_𑀦 𑀩𑁆𑀬𑀕𑁆𑀖𑀲𑁆𑀲 𑀦 𑀤𑀻𑀧𑀺𑀦𑁄;_  
_𑀧𑀸𑀭𑀼𑀢𑁄 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑁂𑀦,_  
_𑀚𑀫𑁆𑀫𑁄 𑀦𑀤𑀢𑀺 𑀕𑀤𑁆𑀭𑀪𑁄”𑁇_  


(78.)

463\. _“𑀘𑀺𑀭𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑀁 𑀔𑀸𑀤𑁂𑀬𑁆𑀬,_  
_𑀕𑀤𑁆𑀭𑀪𑁄 𑀳𑀭𑀺𑀢𑀁 𑀬𑀯𑀁;_  
_𑀧𑀸𑀭𑀼𑀢𑁄 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑁂𑀦,_  
_𑀭𑀯𑀫𑀸𑀦𑁄𑀯 𑀤𑀽𑀲𑀬𑀻”𑀢𑀺𑁇_
