# 𑁨𑁇𑁪𑁇𑁩 𑀯𑀸𑀮𑁄𑀤𑀓𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀯𑀸𑀮𑁄𑀤𑀓𑀚𑀸𑀢𑀓

(65.)

444\. _“𑀯𑀸𑀮𑁄𑀤𑀓𑀁 𑀅𑀧𑁆𑀧𑀭𑀲𑀁 𑀦𑀺𑀳𑀻𑀦𑀁,_  
_𑀧𑀺𑀢𑁆𑀯𑀸 𑀫𑀤𑁄 𑀚𑀸𑀬𑀢𑀺 𑀕𑀤𑁆𑀭𑀪𑀸𑀦𑀁;_  
_𑀇𑀫𑀜𑁆𑀘 𑀧𑀺𑀢𑁆𑀯𑀸𑀦 𑀭𑀲𑀁 𑀧𑀡𑀻𑀢𑀁,_  
_𑀫𑀤𑁄 𑀦 𑀲𑀜𑁆𑀚𑀸𑀬𑀢𑀺 𑀲𑀺𑀦𑁆𑀥𑀯𑀸𑀦𑀁”𑁇_  


(66.)

445\. _“𑀅𑀧𑁆𑀧𑀁 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸𑀦 𑀦𑀺𑀳𑀻𑀦𑀚𑀘𑁆𑀘𑁄,_  
_𑀲𑁄 𑀫𑀚𑁆𑀚𑀢𑀻 𑀢𑁂𑀦 𑀚𑀦𑀺𑀦𑁆𑀤 𑀧𑀼𑀝𑁆𑀞𑁄;_  
_𑀥𑁄𑀭𑀬𑁆𑀳𑀲𑀻𑀮𑀻 𑀘 𑀓𑀼𑀮𑀫𑁆𑀳𑀺 𑀚𑀸𑀢𑁄,_  
_𑀦 𑀫𑀚𑁆𑀚𑀢𑀻 𑀅𑀕𑁆𑀕𑀭𑀲𑀁 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸”𑀢𑀺𑁇_  


---
