# 𑁨𑁇𑁪𑁇𑁮 𑀲𑀻𑀳𑀓𑁄𑀢𑁆𑀣𑀼𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀳𑀓𑁄𑀢𑁆𑀣𑀼𑀚𑀸𑀢𑀓

(75.)

459\. _“𑀲𑀻𑀳𑀗𑁆𑀕𑀼𑀮𑀻 𑀲𑀻𑀳𑀦𑀔𑁄,_  
_𑀲𑀻𑀳𑀧𑀸𑀤𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀲𑁄 𑀲𑀻𑀳𑁄 𑀲𑀻𑀳𑀲𑀁𑀖𑀫𑁆𑀳𑀺,_  
_𑀏𑀓𑁄 𑀦𑀤𑀢𑀺 𑀅𑀜𑁆𑀜𑀣𑀸”𑁇_  


(76.)

460\. _“𑀫𑀸 𑀢𑁆𑀯𑀁 𑀦𑀤𑀺 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢,_  
_𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑁄 𑀯𑀦𑁂 𑀯𑀲;_  
_𑀲𑀭𑁂𑀦 𑀔𑁄 𑀢𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀦 𑀳𑀺 𑀢𑁂 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑁄 𑀲𑀭𑁄”𑀢𑀺𑁇_  


---

461\. 𑀲𑀻𑀳𑀓𑁄𑀢𑁆𑀣𑀼𑀚𑀸𑀢𑀓𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
