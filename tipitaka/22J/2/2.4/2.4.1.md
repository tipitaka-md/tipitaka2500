# 𑁨𑁇𑁪𑁇𑁧 𑀅𑀲𑀤𑀺𑀲𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀅𑀲𑀤𑀺𑀲𑀚𑀸𑀢𑀓

(61.)

438\. _𑀥𑀦𑀼𑀕𑁆𑀕𑀳𑁄 𑀅𑀲𑀤𑀺𑀲𑁄,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀳𑀩𑁆𑀩𑀮𑁄;_  
_𑀤𑀽𑀭𑁂𑀧𑀸𑀢𑀻 𑀅𑀓𑁆𑀔𑀡𑀯𑁂𑀥𑀻,_  
_𑀫𑀳𑀸𑀓𑀸𑀬𑀧𑁆𑀧𑀤𑀸𑀮𑀦𑁄𑁇_  


(62.)

439\. _𑀲𑀩𑁆𑀩𑀸𑀫𑀺𑀢𑁆𑀢𑁂 𑀭𑀡𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦 𑀘 𑀓𑀜𑁆𑀘𑀺 𑀯𑀺𑀳𑁂𑀞𑀬𑀺;_  
_𑀪𑀸𑀢𑀭𑀁 𑀲𑁄𑀢𑁆𑀣𑀺𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀁𑀬𑀫𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀸𑀕𑀫𑀻𑀢𑀺𑁇_
