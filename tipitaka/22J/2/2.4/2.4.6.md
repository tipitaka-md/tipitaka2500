# 𑁨𑁇𑁪𑁇𑁬 𑀤𑀥𑀺𑀯𑀸𑀳𑀦𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀤𑀥𑀺𑀯𑀸𑀳𑀦𑀚𑀸𑀢𑀓

(71.)

453\. _“𑀯𑀡𑁆𑀡𑀕𑀦𑁆𑀥𑀭𑀲𑀽𑀧𑁂𑀢𑁄,_  
_𑀅𑀫𑁆𑀩𑁄𑀬𑀁 𑀅𑀳𑀼𑀯𑀸 𑀧𑀼𑀭𑁂;_  
_𑀢𑀫𑁂𑀯 𑀧𑀽𑀚𑀁 𑀮𑀪𑀫𑀸𑀦𑁄,_  
_𑀓𑁂𑀦𑀫𑁆𑀩𑁄 𑀓𑀝𑀼𑀓𑀧𑁆𑀨𑀮𑁄”𑁇_  


(72.)

454\. _“𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀧𑀭𑀺𑀯𑀸𑀭𑁄,_  
_𑀅𑀫𑁆𑀩𑁄 𑀢𑁂 𑀤𑀥𑀺𑀯𑀸𑀳𑀦;_  
_𑀫𑀽𑀮𑀁 𑀫𑀽𑀮𑁂𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑀁,_  
_𑀲𑀸𑀔𑀸 𑀲𑀸𑀔𑀸 𑀦𑀺𑀲𑁂𑀯𑀭𑁂;_  
_𑀅𑀲𑀸𑀢𑀲𑀦𑁆𑀦𑀺𑀯𑀸𑀲𑁂𑀦,_  
_𑀢𑁂𑀦𑀫𑁆𑀩𑁄 𑀓𑀝𑀼𑀓𑀧𑁆𑀨𑀮𑁄”𑀢𑀺𑁇_  


---
