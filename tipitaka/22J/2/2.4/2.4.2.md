# 𑁨𑁇𑁪𑁇𑁨 𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑀚𑀸𑀢𑀓

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

### 𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑀚𑀸𑀢𑀓

(63.)

441\. _“𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑁄 𑀲𑀽𑀭𑁄,_  
_𑀩𑀮𑀯𑀸 𑀇𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀓𑀺𑀁 𑀦𑀼 𑀢𑁄𑀭𑀡𑀫𑀸𑀲𑀚𑁆𑀚,_  
_𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀲𑀺 𑀓𑀼𑀜𑁆𑀚𑀭𑁇_  


(64.)

442\. _𑀑𑀫𑀤𑁆𑀤 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀮𑀺𑀖𑀁,_  
_𑀏𑀲𑀺𑀓𑀸𑀦𑀺 𑀘 𑀅𑀩𑁆𑀩𑀳;_  
_𑀢𑁄𑀭𑀡𑀸𑀦𑀺 𑀘 𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀯𑀺𑀲 𑀓𑀼𑀜𑁆𑀚𑀭𑀸”𑀢𑀺𑁇_  


---

443\. 𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
