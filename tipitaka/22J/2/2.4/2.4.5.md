# 𑁨𑁇𑁪𑁇𑁫 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀚𑀸𑀢𑀓

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀲𑀤𑀺𑀲𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀚𑀸𑀢𑀓

(69.)

450\. _“𑀬𑀣𑁄𑀤𑀓𑁂 𑀆𑀯𑀺𑀮𑁂 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂,_  
_𑀦 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑀺𑀧𑁆𑀧𑀺𑀓𑀲𑀫𑁆𑀩𑀼𑀓𑀜𑁆𑀘;_  
_𑀲𑀓𑁆𑀔𑀭𑀁 𑀯𑀸𑀮𑀼𑀓𑀁 𑀫𑀘𑁆𑀙𑀕𑀼𑀫𑁆𑀩𑀁,_  
_𑀏𑀯𑀁 𑀆𑀯𑀺𑀮𑀫𑁆𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑁂;_  
_𑀦 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀢𑁆𑀢𑀤𑀢𑁆𑀣𑀁 𑀧𑀭𑀢𑁆𑀣𑀁𑁇_  


(70.)

451\. _𑀬𑀣𑁄𑀤𑀓𑁂 𑀅𑀘𑁆𑀙𑁂 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂,_  
_𑀲𑁄 𑀧𑀲𑁆𑀲𑀢𑀺 𑀲𑀺𑀧𑁆𑀧𑀺𑀓𑀲𑀫𑁆𑀩𑀼𑀓𑀜𑁆𑀘;_  
_𑀲𑀓𑁆𑀔𑀭𑀁 𑀯𑀸𑀮𑀼𑀓𑀁 𑀫𑀘𑁆𑀙𑀕𑀼𑀫𑁆𑀩𑀁,_  
_𑀏𑀯𑀁 𑀅𑀦𑀸𑀯𑀺𑀮𑀫𑁆𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑁂;_  
_𑀲𑁄 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀢𑁆𑀢𑀤𑀢𑁆𑀣𑀁 𑀧𑀭𑀢𑁆𑀣𑀦𑁆”𑀢𑀺𑁇_  


---
