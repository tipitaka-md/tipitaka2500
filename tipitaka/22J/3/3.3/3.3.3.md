# 𑁩𑁇𑁩𑁇𑁩 𑀓𑀘𑁆𑀙𑀧𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀤𑀧𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀓𑀘𑁆𑀙𑀧𑀚𑀸𑀢𑀓

(67.)

762. _“𑀓𑁄 𑀦𑀼 𑀉𑀤𑁆𑀥𑀺𑀢𑀪𑀢𑁆𑀢𑁄𑀯,_  
_𑀧𑀽𑀭𑀳𑀢𑁆𑀣𑁄𑀯 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀓𑀳𑀁 𑀦𑀼 𑀪𑀺𑀓𑁆𑀔𑀁 𑀅𑀘𑀭𑀺,_  
_𑀓𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺”𑁇_  


(68.)

763. _“𑀅𑀳𑀁 𑀓𑀧𑀺𑀲𑁆𑀫𑀺 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑁄,_  
_𑀅𑀦𑀸𑀫𑀸𑀲𑀸𑀦𑀺 𑀆𑀫𑀲𑀺𑀁;_  
_𑀢𑁆𑀯𑀁 𑀫𑀁 𑀫𑁄𑀘𑀬 𑀪𑀤𑁆𑀤𑀁 𑀢𑁂,_  
_𑀫𑀼𑀢𑁆𑀢𑁄 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀧𑀩𑁆𑀩𑀢𑀁”𑁇_  


(69.)

764. _“𑀓𑀘𑁆𑀙𑀧𑀸 𑀓𑀲𑁆𑀲𑀧𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀓𑁆𑀓𑀝𑀸;_  
_𑀫𑀼𑀜𑁆𑀘 𑀓𑀲𑁆𑀲𑀧 𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀁,_  
_𑀓𑀢𑀁 𑀫𑁂𑀣𑀼𑀦𑀓𑀁 𑀢𑀬𑀸”𑀢𑀺𑁇_  


---
