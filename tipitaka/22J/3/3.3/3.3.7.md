# 𑁩𑁇𑁩𑁇𑁭 𑀭𑁄𑀫𑀓𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀤𑀧𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀭𑁄𑀫𑀓𑀚𑀸𑀢𑀓

(79.)

778. _“𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀸𑀲 𑀲𑀫𑀸𑀥𑀺𑀓𑀸𑀦𑀺,_  
_𑀯𑀲𑀺𑀫𑁆𑀳 𑀲𑁂𑀮𑀲𑁆𑀲 𑀕𑀼𑀳𑀸𑀬 𑀭𑁄𑀫𑀓;_  
_𑀅𑀲𑀗𑁆𑀓𑀫𑀸𑀦𑀸 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀢𑁆𑀢𑀸,_  
_𑀳𑀢𑁆𑀣𑀢𑁆𑀢𑀫𑀸𑀬𑀦𑁆𑀢𑀺 𑀫𑀫𑀡𑁆𑀟𑀚𑀸 𑀧𑀼𑀭𑁂𑁇_  


(80.)

779. _𑀢𑁂 𑀤𑀸𑀦𑀺 𑀯𑀓𑁆𑀓𑀗𑁆𑀕 𑀓𑀺𑀫𑀢𑁆𑀣𑀫𑀼𑀲𑁆𑀲𑀼𑀓𑀸,_  
_𑀪𑀚𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀁 𑀕𑀺𑀭𑀺𑀓𑀦𑁆𑀤𑀭𑀁 𑀤𑀺𑀚𑀸;_  
_𑀦 𑀦𑀽𑀦 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀫𑀫𑀁 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂,_  
_𑀘𑀺𑀭𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑀸 𑀅𑀣 𑀯𑀸 𑀦 𑀢𑁂 𑀇𑀫𑁂”𑁇_  


(81.)

780. _“𑀚𑀸𑀦𑀸𑀫 𑀢𑀁 𑀦 𑀫𑀬𑀁 𑀲𑀫𑁆𑀧𑀫𑀽𑀍𑀳𑀸,_  
_𑀲𑁄𑀬𑁂𑀯 𑀢𑁆𑀯𑀁 𑀢𑁂 𑀫𑀬𑀫𑀲𑁆𑀫 𑀦𑀸𑀜𑁆𑀜𑁂;_  
_𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀢𑁂 𑀅𑀲𑁆𑀫𑀺𑀁 𑀚𑀦𑁂 𑀧𑀤𑀼𑀝𑁆𑀞𑀁,_  
_𑀆𑀚𑀻𑀯𑀺𑀓𑀸 𑀢𑁂𑀦 𑀢𑀫𑀼𑀢𑁆𑀢𑀲𑀸𑀫𑀸”𑀢𑀺𑁇_  


---

781. 𑀭𑁄𑀫𑀓𑀚𑀸𑀢𑀓𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
