# 𑁩𑁇𑁩𑁇𑁨 𑀩𑁆𑀬𑀕𑁆𑀖𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀤𑀧𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀬𑀕𑁆𑀖𑀚𑀸𑀢𑀓

(64.)

758\. _“𑀬𑁂𑀦 𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀲𑀕𑁆𑀕𑀸,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑁄 𑀯𑀺𑀳𑀺𑀬𑁆𑀬𑀢𑀺;_  
_𑀧𑀼𑀩𑁆𑀩𑁂𑀯𑀚𑁆𑀛𑀸𑀪𑀯𑀁 𑀢𑀲𑁆𑀲,_  
_𑀭𑀼𑀓𑁆𑀔𑁂 𑀅𑀓𑁆𑀔𑀻𑀯 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


(65.)

759\. _𑀬𑁂𑀦 𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀲𑀕𑁆𑀕𑀸,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑁄 𑀧𑀯𑀟𑁆𑀠𑀢𑀺;_  
_𑀓𑀭𑁂𑀬𑁆𑀬𑀢𑁆𑀢𑀲𑀫𑀁 𑀯𑀼𑀢𑁆𑀢𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑀓𑀺𑀘𑁆𑀘𑁂𑀲𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄”𑁇_  


(66.)

760\. _“𑀏𑀣 𑀩𑁆𑀬𑀕𑁆𑀖𑀸 𑀦𑀺𑀯𑀢𑁆𑀢𑀯𑁆𑀳𑁄,_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁂𑀣 𑀫𑀳𑀸𑀯𑀦𑀁;_  
_𑀫𑀸 𑀯𑀦𑀁 𑀙𑀺𑀦𑁆𑀤𑀺 𑀦𑀺𑀩𑁆𑀬𑀕𑁆𑀖𑀁,_  
_𑀩𑁆𑀬𑀕𑁆𑀖𑀸 𑀫𑀸𑀳𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀦𑀸”𑀢𑀺𑁇_  


---
