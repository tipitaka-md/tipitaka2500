# 𑁩𑁇𑁩𑁇𑁫 𑀭𑀼𑀘𑀺𑀭𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀤𑀧𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀭𑀼𑀘𑀺𑀭𑀚𑀸𑀢𑀓

(73.)

770\. _“𑀓𑀸𑀬𑀁 𑀩𑀮𑀸𑀓𑀸 𑀭𑀼𑀘𑀺𑀭𑀸,_  
_𑀓𑀸𑀓𑀦𑀻𑀍𑀅𑀲𑁆𑀫𑀺𑀫𑀘𑁆𑀙𑀢𑀺;_  
_𑀘𑀡𑁆𑀟𑁄 𑀓𑀸𑀓𑁄 𑀲𑀔𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀬𑀲𑁆𑀲 𑀘𑁂𑀢𑀁 𑀓𑀼𑀮𑀸𑀯𑀓𑀁”𑁇_  


(74.)

771\. _“𑀦𑀦𑀼 𑀫𑀁 𑀲𑀫𑁆𑀫 𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀤𑀺𑀚 𑀲𑀸𑀫𑀸𑀓𑀪𑁄𑀚𑀦;_  
_𑀅𑀓𑀢𑁆𑀯𑀸 𑀯𑀘𑀦𑀁 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀧𑀲𑁆𑀲 𑀮𑀽𑀦𑁄𑀲𑁆𑀫𑀺 𑀆𑀕𑀢𑁄”𑁇_  


(75.)

772\. _“𑀧𑀼𑀦𑀧𑀸𑀧𑀚𑁆𑀚𑀲𑀻 𑀲𑀫𑁆𑀫,_  
_𑀲𑀻𑀮𑀜𑁆𑀳𑀺 𑀢𑀯 𑀢𑀸𑀤𑀺𑀲𑀁;_  
_𑀦 𑀳𑀺 𑀫𑀸𑀦𑀼𑀲𑀓𑀸 𑀪𑁄𑀕𑀸,_  
_𑀲𑀼𑀪𑀼𑀜𑁆𑀚𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀓𑁆𑀔𑀺𑀦𑀸”𑀢𑀺𑁇_  


---
