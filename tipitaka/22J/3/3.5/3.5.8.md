# 𑁩𑁇𑁫𑁇𑁮 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀚𑀸𑀢𑀓

(142.)

866\. _“𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀸 𑀘𑀺𑀫𑁂 𑀧𑀓𑁆𑀓𑀸,_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸 𑀘 𑀓𑀧𑀺𑀢𑁆𑀣𑀦𑀸;_  
_𑀏𑀳𑀺 𑀦𑀺𑀓𑁆𑀔𑀫 𑀪𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼,_  
_𑀓𑀺𑀁 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀬 𑀫𑀺𑀬𑁆𑀬𑀲𑀺”𑁇_  


(143.)

867\. _“𑀏𑀯𑀁 𑀲𑁄 𑀲𑀼𑀳𑀺𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀯𑀼𑀟𑁆𑀠𑀫𑀧𑀘𑀸𑀬𑀢𑀺;_  
_𑀬𑀣𑀸𑀳𑀫𑀚𑁆𑀚 𑀲𑀼𑀳𑀺𑀢𑁄,_  
_𑀤𑀼𑀫𑀧𑀓𑁆𑀓𑀸𑀦𑀺 𑀫𑀸𑀲𑀺𑀢𑁄”𑁇_  


(144.)

868\. _“𑀬𑀁 𑀯𑀦𑁂𑀚𑁄 𑀯𑀦𑁂𑀚𑀲𑁆𑀲,_  
_𑀯𑀜𑁆𑀘𑁂𑀬𑁆𑀬 𑀓𑀧𑀺𑀦𑁄 𑀓𑀧𑀺;_  
_𑀤𑀳𑀭𑁄 𑀓𑀧𑀺 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬,_  
_𑀦 𑀳𑀺 𑀚𑀺𑀡𑁆𑀡𑁄 𑀚𑀭𑀸𑀓𑀧𑀻”𑀢𑀺𑁇_  


---

869\. 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀚𑀸𑀢𑀓𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
