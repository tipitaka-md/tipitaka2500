# 𑁩𑁇𑁫𑁇𑁩 𑀓𑀸𑀬𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀬𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀚𑀸𑀢𑀓

(127.)

846\. _“𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀫𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀩𑁆𑀬𑀸𑀥𑀺𑀦𑀸,_  
_𑀭𑁄𑀕𑁂𑀦 𑀩𑀸𑀍𑀳𑀁 𑀤𑀼𑀔𑀺𑀢𑀲𑁆𑀲 𑀭𑀼𑀧𑁆𑀧𑀢𑁄;_  
_𑀧𑀭𑀺𑀲𑀼𑀲𑁆𑀲𑀢𑀺 𑀔𑀺𑀧𑁆𑀧𑀫𑀺𑀤𑀁 𑀓𑀍𑀏𑀯𑀭𑀁,_  
_𑀧𑀼𑀧𑁆𑀨𑀁 𑀬𑀣𑀸 𑀧𑀁𑀲𑀼𑀦𑀺 𑀆𑀢𑀧𑁂 𑀓𑀢𑀁𑁇_  


(128.)

847\. _𑀅𑀚𑀜𑁆𑀜𑀁 𑀚𑀜𑁆𑀜𑀲𑀗𑁆𑀔𑀸𑀢𑀁,_  
_𑀅𑀲𑀼𑀘𑀺𑀁 𑀲𑀼𑀘𑀺𑀲𑀫𑁆𑀫𑀢𑀁;_  
_𑀦𑀸𑀦𑀸𑀓𑀼𑀡𑀧𑀧𑀭𑀺𑀧𑀽𑀭𑀁,_  
_𑀚𑀜𑁆𑀜𑀭𑀽𑀧𑀁 𑀅𑀧𑀲𑁆𑀲𑀢𑁄𑁇_  


(129.)

848\. _𑀥𑀺𑀭𑀢𑁆𑀣𑀼𑀫𑀁 𑀆𑀢𑀼𑀭𑀁 𑀧𑀽𑀢𑀺𑀓𑀸𑀬𑀁,_  
_𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀺𑀬𑀁 𑀅𑀲𑁆𑀲𑀼𑀘𑀺𑀁 𑀩𑁆𑀬𑀸𑀥𑀺𑀥𑀫𑁆𑀫𑀁;_  
_𑀬𑀢𑁆𑀣𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀧𑀚𑀸,_  
_𑀳𑀸𑀧𑁂𑀦𑁆𑀢𑀺 𑀫𑀕𑁆𑀕𑀁 𑀲𑀼𑀕𑀢𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇_  


---

849\. 𑀓𑀸𑀬𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
