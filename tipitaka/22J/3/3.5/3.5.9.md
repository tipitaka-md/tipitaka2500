# 𑁩𑁇𑁫𑁇𑁯 𑀓𑁄𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀓𑁄𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀚𑀸𑀢𑀓

(145.)

870\. _“𑀧𑀼𑀭𑁂 𑀢𑀼𑀯𑀁 𑀲𑀻𑀮𑀯𑀢𑀁 𑀲𑀓𑀸𑀲𑁂,_  
_𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑀓𑀁 𑀓𑀻𑀍𑀅𑀲𑀺 𑀅𑀲𑁆𑀲𑀫𑀫𑁆𑀳𑀺;_  
_𑀓𑀭𑁄𑀳𑀭𑁂 𑀫𑀓𑁆𑀓𑀝𑀺𑀬𑀸𑀦𑀺 𑀫𑀓𑁆𑀓𑀝,_  
_𑀦 𑀢𑀁 𑀫𑀬𑀁 𑀲𑀻𑀮𑀯𑀢𑀁 𑀭𑀫𑀸𑀫”𑁇_  


(146.)

871\. _“𑀲𑀼𑀢𑀸 𑀳𑀺 𑀫𑀬𑁆𑀳𑀁 𑀧𑀭𑀫𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺,_  
_𑀓𑁄𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲;_  
_𑀫𑀸 𑀤𑀸𑀦𑀺 𑀫𑀁 𑀫𑀜𑁆𑀜𑀺 𑀢𑀼𑀯𑀁 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂,_  
_𑀛𑀸𑀦𑀸𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀆𑀯𑀼𑀲𑁄”𑁇_  


(147.)

872\. _“𑀲𑀘𑁂𑀧𑀺 𑀲𑁂𑀮𑀲𑁆𑀫𑀺 𑀯𑀧𑁂𑀬𑁆𑀬 𑀩𑀻𑀚𑀁,_  
_𑀤𑁂𑀯𑁄 𑀘 𑀯𑀲𑁆𑀲𑁂 𑀦 𑀳𑀺 𑀢𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑁂;_  
_𑀲𑀼𑀢𑀸 𑀳𑀺 𑀢𑁂 𑀲𑀸 𑀧𑀭𑀫𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺,_  
_𑀆𑀭𑀸 𑀢𑀼𑀯𑀁 𑀫𑀓𑁆𑀓𑀝 𑀛𑀸𑀦𑀪𑀽𑀫𑀺𑀬𑀸”𑀢𑀺𑁇_  


---
