# 𑁩𑁇𑁧𑁇𑁧 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀭𑀸𑀕𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀯𑀕𑁆𑀕

### 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀭𑀸𑀕𑀚𑀸𑀢𑀓

(1.)

670. _“𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀭𑀸𑀕𑀥𑁄𑀢𑁂𑀦,_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑀦𑀺𑀲𑀺𑀢𑁂𑀦 𑀘;_  
_𑀦𑀸𑀮𑀗𑁆𑀓𑀢𑁂𑀦 𑀪𑀤𑁆𑀭𑁂𑀦,_  
_𑀉𑀲𑀼𑀓𑀸𑀭𑀸𑀓𑀢𑁂𑀦 𑀘𑁇_  


(2.)

671. _𑀦 𑀓𑀡𑁆𑀡𑀸𑀬𑀢𑀫𑀼𑀢𑁆𑀢𑁂𑀦,_  
_𑀦𑀸𑀧𑀺 𑀫𑁄𑀭𑀽𑀧𑀲𑁂𑀯𑀺𑀦𑀸;_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀳𑀤𑀬𑁂 𑀯𑀺𑀤𑁆𑀥𑁄,_  
_𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀧𑀭𑀺𑀤𑀸𑀳𑀺𑀦𑀸𑁇_  


(3.)

672. _𑀆𑀯𑁂𑀥𑀜𑁆𑀘 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀢𑁄 𑀭𑀼𑀳𑀺𑀭𑀫𑀲𑁆𑀲𑀯𑁂;_  
_𑀬𑀸𑀯 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀲𑀬𑀁 𑀫𑁂 𑀤𑀼𑀓𑁆𑀔𑀫𑀸𑀪𑀢𑀦𑁆”𑀢𑀺𑁇_  


---

673. 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀭𑀸𑀕𑀚𑀸𑀢𑀓𑀁 𑀧𑀞𑀫𑀁𑁇
