# 𑁩𑁇𑁧𑁇𑁩 𑀫𑀡𑀺𑀓𑀡𑁆𑀞𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀯𑀕𑁆𑀕

### 𑀫𑀡𑀺𑀓𑀡𑁆𑀞𑀚𑀸𑀢𑀓

(7.)

678\. _“𑀫𑀫𑀦𑁆𑀦𑀧𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀉𑀍𑀆𑀭𑀁,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀫𑀲𑁆𑀲 𑀫𑀡𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼;_  
_𑀢𑀁 𑀢𑁂 𑀦 𑀤𑀲𑁆𑀲𑀁 𑀅𑀢𑀺𑀬𑀸𑀘𑀓𑁄𑀲𑀺,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀢𑁂 𑀅𑀲𑁆𑀲𑀫𑀫𑀸𑀕𑀫𑀺𑀲𑁆𑀲𑀁𑁇_  


(8.)

679\. _𑀲𑀼𑀲𑀽 𑀬𑀣𑀸 𑀲𑀓𑁆𑀔𑀭𑀥𑁄𑀢𑀧𑀸𑀡𑀻,_  
_𑀢𑀸𑀲𑁂𑀲𑀺 𑀫𑀁 𑀲𑁂𑀮𑀁 𑀬𑀸𑀘𑀫𑀸𑀦𑁄;_  
_𑀢𑀁 𑀢𑁂 𑀦 𑀤𑀲𑁆𑀲𑀁 𑀅𑀢𑀺𑀬𑀸𑀘𑀓𑁄𑀲𑀺,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀢𑁂 𑀅𑀲𑁆𑀲𑀫𑀫𑀸𑀕𑀫𑀺𑀲𑁆𑀲𑀁”𑁇_  


(9.)

680\. _“𑀦 𑀢𑀁 𑀬𑀸𑀘𑁂 𑀬𑀲𑁆𑀲 𑀧𑀺𑀬𑀁 𑀚𑀺𑀕𑀻𑀲𑁂,_  
_𑀤𑁂𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀬𑀸𑀘𑀦𑀸𑀬;_  
_𑀦𑀸𑀕𑁄 𑀫𑀡𑀺𑀁 𑀬𑀸𑀘𑀺𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀦,_  
_𑀅𑀤𑀲𑁆𑀲𑀦𑀁𑀬𑁂𑀯 𑀢𑀤𑀚𑁆𑀛𑀕𑀫𑀸”𑀢𑀺𑁇_  


---

681\. 𑀫𑀡𑀺𑀓𑀡𑁆𑀞𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
