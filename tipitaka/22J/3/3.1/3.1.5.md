# 𑁩𑁇𑁧𑁇𑁫 𑀲𑀼𑀓𑀚𑀸𑀢𑀓

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀓𑀚𑀸𑀢𑀓

(13.)

686\. _“𑀬𑀸𑀯 𑀲𑁄 𑀫𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀲𑀺,_  
_𑀪𑁄𑀚𑀦𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀗𑁆𑀕𑀫𑁄;_  
_𑀢𑀸𑀯 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀸𑀧𑀸𑀤𑀺,_  
_𑀫𑀸𑀢𑀭𑀜𑁆𑀘 𑀅𑀧𑁄𑀲𑀬𑀺𑁇_  


(14.)

687\. _𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀩𑀳𑀼𑀢𑀭𑀁,_  
_𑀪𑁄𑀚𑀦𑀁 𑀅𑀚𑁆𑀛𑀯𑀸𑀳𑀭𑀺;_  
_𑀢𑀢𑁄 𑀢𑀢𑁆𑀣𑁂𑀯 𑀲𑀁𑀲𑀻𑀤𑀺,_  
_𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀳𑀺 𑀲𑁄 𑀅𑀳𑀼𑁇_  


(15.)

688\. _𑀢𑀲𑁆𑀫𑀸 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀲𑀸𑀥𑀼,_  
_𑀪𑁄𑀚𑀦𑀲𑁆𑀫𑀺𑀁 𑀅𑀕𑀺𑀤𑁆𑀥𑀢𑀸;_  
_𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀳𑀺 𑀲𑀻𑀤𑀦𑁆𑀢𑀺,_  
_𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘 𑀦 𑀲𑀻𑀤𑀭𑁂”𑀢𑀺𑁇_  


---
