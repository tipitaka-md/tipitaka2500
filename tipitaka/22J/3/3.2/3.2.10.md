# 𑁩𑁇𑁨𑁇𑁧𑁦 𑀉𑀮𑀽𑀓𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀤𑀼𑀫𑀯𑀕𑁆𑀕

### 𑀉𑀮𑀽𑀓𑀚𑀸𑀢𑀓

(58.)

748. _“𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀓𑀺𑀭 𑀜𑀸𑀢𑀻𑀳𑀺,_  
_𑀓𑁄𑀲𑀺𑀬𑁄 𑀇𑀲𑁆𑀲𑀭𑁄 𑀓𑀢𑁄;_  
_𑀲𑀘𑁂 𑀜𑀸𑀢𑀻𑀳𑀺 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄,_  
_𑀪𑀡𑁂𑀬𑁆𑀬𑀸𑀳𑀁 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀁”𑁇_  


(59.)

749. _“𑀪𑀡 𑀲𑀫𑁆𑀫 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄,_  
_𑀅𑀢𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀓𑁂𑀯𑀮𑀁;_  
_𑀲𑀦𑁆𑀢𑀺 𑀳𑀺 𑀤𑀳𑀭𑀸 𑀧𑀓𑁆𑀔𑀻,_  
_𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀦𑁆𑀥𑀭𑀸”𑁇_  


(60.)

750. _“𑀦 𑀫𑁂 𑀭𑀼𑀘𑁆𑀘𑀢𑀺 𑀪𑀤𑁆𑀤𑀁 𑀯𑁄,_  
_𑀉𑀮𑀽𑀓𑀲𑁆𑀲𑀸𑀪𑀺𑀲𑁂𑀘𑀦𑀁;_  
_𑀅𑀓𑁆𑀓𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀫𑀼𑀔𑀁 𑀧𑀲𑁆𑀲,_  
_𑀓𑀣𑀁 𑀓𑀼𑀤𑁆𑀥𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

751. 𑀉𑀮𑀽𑀓𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀧𑀤𑀼𑀫𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



---

752. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



753. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀫 𑀦𑀸𑀕𑀲𑀺𑀭𑀺𑀯𑁆𑀳𑀬𑀦𑁄,_  
_𑀲-𑀫𑀳𑀡𑁆𑀡𑀯 𑀬𑀽𑀧 𑀔𑀼𑀭𑀧𑁆𑀧𑀯𑀭𑁄;_  
_𑀅𑀣 𑀪𑀤𑁆𑀤𑀮𑀻 𑀓𑀼𑀜𑁆𑀚𑀭 𑀭𑀼𑀓𑁆𑀔 𑀧𑀼𑀦,_  
_𑀔𑀭𑀯𑀸𑀘 𑀉𑀮𑀽𑀓𑀯𑀭𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
