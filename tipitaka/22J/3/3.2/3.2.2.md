# 𑁩𑁇𑁨𑁇𑁨 𑀫𑀼𑀤𑀼𑀧𑀸𑀡𑀺𑀚𑀸𑀢𑀓

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀤𑀼𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀼𑀤𑀼𑀧𑀸𑀡𑀺𑀚𑀸𑀢𑀓

(34.)

716\. _“𑀧𑀸𑀡𑀺 𑀘𑁂 𑀫𑀼𑀤𑀼𑀓𑁄 𑀘𑀲𑁆𑀲,_  
_𑀦𑀸𑀕𑁄 𑀘𑀲𑁆𑀲 𑀲𑀼𑀓𑀸𑀭𑀺𑀢𑁄;_  
_𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄 𑀘 𑀯𑀲𑁆𑀲𑁂𑀬𑁆𑀬,_  
_𑀅𑀣 𑀦𑀽𑀦 𑀢𑀤𑀸 𑀲𑀺𑀬𑀸”𑁇_  


(35.)

717\. _“𑀅𑀦𑀮𑀸 𑀫𑀼𑀤𑀼𑀲𑀫𑁆𑀪𑀸𑀲𑀸,_  
_𑀤𑀼𑀧𑁆𑀧𑀽𑀭𑀸 𑀢𑀸 𑀦𑀤𑀻𑀲𑀫𑀸;_  
_𑀲𑀻𑀤𑀦𑁆𑀢𑀺 𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀭𑀓𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑁂𑁇_  


(36.)

718\. _𑀬𑀁 𑀏𑀢𑀸 𑀉𑀧𑀲𑁂𑀯𑀦𑁆𑀢𑀺,_  
_𑀙𑀦𑁆𑀤𑀲𑀸 𑀯𑀸 𑀥𑀦𑁂𑀦 𑀯𑀸;_  
_𑀚𑀸𑀢𑀯𑁂𑀤𑁄𑀯 𑀲𑀁 𑀞𑀸𑀦𑀁,_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀅𑀦𑀼𑀤𑀳𑀦𑁆𑀢𑀺 𑀦𑀦𑁆”𑀢𑀺𑁇_  


---
