# 𑁩𑁇𑁨𑁇𑁪 𑀫𑀳𑀸𑀧𑀦𑀸𑀤𑀚𑀸𑀢𑀓

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀤𑀼𑀫𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀧𑀦𑀸𑀤𑀚𑀸𑀢𑀓

(40.)

724\. _“𑀧𑀦𑀸𑀤𑁄 𑀦𑀸𑀫 𑀲𑁄 𑀭𑀸𑀚𑀸,_  
_𑀬𑀲𑁆𑀲 𑀬𑀽𑀧𑁄 𑀲𑀼𑀯𑀡𑁆𑀡𑀬𑁄;_  
_𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑁄𑀍𑀅𑀲𑀼𑀩𑁆𑀩𑁂𑀥𑁄,_  
_𑀉𑀤𑁆𑀥𑀫𑀸𑀳𑀼 𑀲𑀳𑀲𑁆𑀲𑀥𑀸𑁇_  


(41.)

725\. _𑀲𑀳𑀲𑁆𑀲𑀓𑀡𑁆𑀟𑁄 𑀲𑀢𑀕𑁂𑀡𑁆𑀟𑀼,_  
_𑀥𑀚𑀸𑀮𑀼 𑀳𑀭𑀺𑀢𑀸𑀫𑀬𑁄;_  
_𑀅𑀦𑀘𑁆𑀘𑀼𑀁 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸,_  
_𑀙 𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢𑀥𑀸𑁇_  


(42.)

726\. _𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀤𑀸 𑀆𑀲𑀺,_  
_𑀬𑀣𑀸 𑀪𑀸𑀲𑀲𑀺 𑀪𑀤𑁆𑀤𑀚𑀺;_  
_𑀲𑀓𑁆𑀓𑁄 𑀅𑀳𑀁 𑀢𑀤𑀸 𑀆𑀲𑀺𑀁,_  
_𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀓𑀭𑁄 𑀢𑀯𑀸”𑀢𑀺𑁇_  


---
