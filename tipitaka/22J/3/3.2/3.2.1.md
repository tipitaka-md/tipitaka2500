# 𑁩𑁇𑁨𑁇𑁧 𑀧𑀤𑀼𑀫𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀤𑀼𑀫𑀯𑀕𑁆𑀕

### 𑀧𑀤𑀼𑀫𑀚𑀸𑀢𑀓

(31.)

712. _“𑀬𑀣𑀸 𑀓𑁂𑀲𑀸 𑀘 𑀫𑀲𑁆𑀲𑀽 𑀘,_  
_𑀙𑀺𑀦𑁆𑀦𑀁 𑀙𑀺𑀦𑁆𑀦𑀁 𑀯𑀺𑀭𑀽𑀳𑀢𑀺;_  
_𑀏𑀯𑀁 𑀭𑀽𑀳𑀢𑀼 𑀢𑁂 𑀦𑀸𑀲𑀸,_  
_𑀧𑀤𑀼𑀫𑀁 𑀤𑁂𑀳𑀺 𑀬𑀸𑀘𑀺𑀢𑁄”𑁇_  


(32.)

713. _“𑀬𑀣𑀸 𑀲𑀸𑀭𑀤𑀺𑀓𑀁 𑀩𑀻𑀚𑀁,_  
_𑀔𑁂𑀢𑁆𑀢𑁂 𑀯𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑀽𑀳𑀢𑀺;_  
_𑀏𑀯𑀁 𑀭𑀽𑀳𑀢𑀼 𑀢𑁂 𑀦𑀸𑀲𑀸,_  
_𑀧𑀤𑀼𑀫𑀁 𑀤𑁂𑀳𑀺 𑀬𑀸𑀘𑀺𑀢𑁄”𑁇_  


(33.)

714. _“𑀉𑀪𑁄𑀧𑀺 𑀧𑀮𑀧𑀦𑁆𑀢𑁂𑀢𑁂,_  
_𑀅𑀧𑀺 𑀧𑀤𑁆𑀫𑀸𑀦𑀺 𑀤𑀲𑁆𑀲𑀢𑀺;_  
_𑀯𑀚𑁆𑀚𑀼𑀁 𑀯𑀸 𑀢𑁂 𑀦 𑀯𑀸 𑀯𑀚𑁆𑀚𑀼𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀦𑀸𑀲𑀸𑀬 𑀭𑀽𑀳𑀦𑀸;_  
_𑀤𑁂𑀳𑀺 𑀲𑀫𑁆𑀫 𑀧𑀤𑀼𑀫𑀸𑀦𑀺,_  
_𑀅𑀳𑀁 𑀬𑀸𑀘𑀸𑀫𑀺 𑀬𑀸𑀘𑀺𑀢𑁄”𑀢𑀺𑁇_  


---
