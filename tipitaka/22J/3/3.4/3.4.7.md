# 𑁩𑁇𑁪𑁇𑁭 𑀮𑀸𑀪𑀕𑀭𑀳𑀚𑀸𑀢𑀓

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀮𑀸𑀪𑀕𑀭𑀳𑀚𑀸𑀢𑀓

(109.)

820\. _“𑀦𑀸𑀦𑀼𑀫𑁆𑀫𑀢𑁆𑀢𑁄 𑀦𑀸𑀧𑀺𑀲𑀼𑀡𑁄,_  
_𑀦𑀸𑀦𑀝𑁄 𑀦𑀸𑀓𑀼𑀢𑀽𑀳𑀮𑁄;_  
_𑀫𑀽𑀍𑀳𑁂𑀲𑀼 𑀮𑀪𑀢𑁂 𑀮𑀸𑀪𑀁,_  
_𑀏𑀲𑀸 𑀢𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻”𑁇_  


(110.)

821\. _“𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀢𑀁 𑀬𑀲𑀮𑀸𑀪𑀁,_  
_𑀥𑀦𑀮𑀸𑀪𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀺 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁂𑀦,_  
_𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀡𑁂𑀦 𑀯𑀸𑁇_  


(111.)

822\. _𑀅𑀧𑀺 𑀘𑁂 𑀧𑀢𑁆𑀢𑀫𑀸𑀤𑀸𑀬,_  
_𑀅𑀦𑀕𑀸𑀭𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂;_  
_𑀏𑀲𑀸𑀯 𑀚𑀻𑀯𑀺𑀓𑀸 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑀸 𑀘𑀸𑀥𑀫𑁆𑀫𑁂𑀦 𑀏𑀲𑀦𑀸”𑀢𑀺𑁇_  


---

823\. 𑀮𑀸𑀪𑀕𑀭𑀳𑀚𑀸𑀢𑀓𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
