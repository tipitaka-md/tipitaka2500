# 𑁩𑁇𑁪𑁇𑁩 𑀯𑀟𑁆𑀠𑀓𑀻𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

### 𑀯𑀟𑁆𑀠𑀓𑀻𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓

(97.)

804. _“𑀯𑀭𑀁 𑀯𑀭𑀁 𑀢𑁆𑀯𑀁 𑀦𑀺𑀳𑀦𑀁 𑀧𑀼𑀭𑁂 𑀘𑀭𑀺,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀲𑀽𑀓𑀭𑁂;_  
_𑀲𑁄 𑀤𑀸𑀦𑀺 𑀏𑀓𑁄 𑀩𑁆𑀬𑀧𑀕𑀫𑁆𑀫 𑀛𑀸𑀬𑀲𑀺,_  
_𑀩𑀮𑀁 𑀦𑀼 𑀢𑁂 𑀩𑁆𑀬𑀕𑁆𑀖 𑀦 𑀘𑀚𑁆𑀚 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(98.)

805. _“𑀇𑀫𑁂 𑀲𑀼𑀤𑀁 𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁 𑀧𑀼𑀭𑁂,_  
_𑀪𑀬𑀝𑁆𑀝𑀺𑀢𑀸 𑀮𑁂𑀡𑀕𑀯𑁂𑀲𑀺𑀦𑁄 𑀧𑀼𑀣𑀽;_  
_𑀢𑁂 𑀤𑀸𑀦𑀺 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀯𑀲𑀦𑁆𑀢𑀺 𑀏𑀓𑀢𑁄,_  
_𑀬𑀢𑁆𑀣𑀝𑁆𑀞𑀺𑀢𑀸 𑀤𑀼𑀧𑁆𑀧𑀲𑀳𑀚𑁆𑀚𑀫𑁂 𑀫𑀬𑀸”𑁇_  


(99.)

806. _“𑀦𑀫𑀢𑁆𑀣𑀼 𑀲𑀁𑀖𑀸𑀦 𑀲𑀫𑀸𑀕𑀢𑀸𑀦𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀬𑀁 𑀲𑀔𑁆𑀬 𑀯𑀤𑀸𑀫𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑀁;_  
_𑀩𑁆𑀬𑀕𑁆𑀖𑀁 𑀫𑀺𑀕𑀸 𑀬𑀢𑁆𑀣 𑀚𑀺𑀦𑀺𑀁𑀲𑀼 𑀤𑀸𑀞𑀺𑀦𑁄,_  
_𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀤𑀸𑀞𑀩𑀮𑁂𑀲𑀼 𑀫𑀼𑀘𑁆𑀘𑀭𑁂”𑀢𑀺𑁇_  


---

807. 𑀯𑀟𑁆𑀠𑀓𑀻𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
