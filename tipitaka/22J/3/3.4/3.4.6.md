# 𑁩𑁇𑁪𑁇𑁬 𑀲𑀸𑀮𑀽𑀓𑀚𑀸𑀢𑀓

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀮𑀽𑀓𑀚𑀸𑀢𑀓

(106.)

816\. _“𑀫𑀸 𑀲𑀸𑀮𑀽𑀓𑀲𑁆𑀲 𑀧𑀺𑀳𑀬𑀺,_  
_𑀆𑀢𑀼𑀭𑀦𑁆𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀢𑀺;_  
_𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀪𑀼𑀲𑀁 𑀔𑀸𑀤,_  
_𑀏𑀢𑀁 𑀤𑀻𑀖𑀸𑀬𑀼𑀮𑀓𑁆𑀔𑀡𑀁𑁇_  


(107.)

817\. _𑀇𑀤𑀸𑀦𑀺 𑀲𑁄 𑀇𑀥𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀢𑀺𑀣𑀻 𑀬𑀼𑀢𑁆𑀢𑀲𑁂𑀯𑀓𑁄;_  
_𑀅𑀣 𑀤𑀓𑁆𑀔𑀲𑀺 𑀲𑀸𑀮𑀽𑀓𑀁,_  
_𑀲𑀬𑀦𑁆𑀢𑀁 𑀫𑀼𑀲𑀮𑀼𑀢𑁆𑀢𑀭𑀁”𑁇_  


(108.)

818\. _𑀯𑀺𑀓𑀦𑁆𑀢𑀁 𑀲𑀽𑀓𑀭𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀬𑀦𑁆𑀢𑀁 𑀫𑀼𑀲𑀮𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀚𑀭𑀕𑁆𑀕𑀯𑀸 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁,_  
_𑀯𑀭𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀼𑀲𑀸𑀫𑀺𑀯𑀸𑀢𑀺𑁇_  


---
