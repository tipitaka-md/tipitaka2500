# 𑁩𑁇𑁪𑁇𑁨 𑀲𑁂𑀬𑁆𑀬𑀚𑀸𑀢𑀓

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀲𑁂𑀬𑁆𑀬𑀚𑀸𑀢𑀓

(94.)

800\. _“𑀲𑁂𑀬𑁆𑀬𑀁𑀲𑁄 𑀲𑁂𑀬𑁆𑀬𑀲𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀲𑁂𑀬𑁆𑀬𑀫𑀼𑀧𑀲𑁂𑀯𑀢𑀺;_  
_𑀏𑀓𑁂𑀦 𑀲𑀦𑁆𑀥𑀺𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀢𑀁 𑀯𑀚𑁆𑀛𑁂 𑀅𑀫𑁄𑀘𑀬𑀺𑀁𑁇_  


(95.)

801\. _𑀢𑀲𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑁂𑀦 𑀮𑁄𑀓𑁂𑀦,_  
_𑀲𑀦𑁆𑀥𑀺𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀏𑀓𑀢𑁄;_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑀕𑁆𑀕𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_𑀇𑀤𑀁 𑀲𑀼𑀡𑀸𑀣 𑀓𑀸𑀲𑀺𑀬𑀸”𑁇_  


(96.)

802\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀓𑀁𑀲𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀕𑁆𑀕𑀳𑁄;_  
_𑀥𑀦𑀼𑀁 𑀓𑀡𑁆𑀟𑀜𑁆𑀘 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁆𑀧,_  
_𑀲𑀁𑀬𑀫𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀸𑀕𑀫𑀻𑀢𑀺𑁇_  


---
