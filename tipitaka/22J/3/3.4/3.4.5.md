# 𑁩𑁇𑁪𑁇𑁫 𑀫𑀡𑀺𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

### 𑀫𑀡𑀺𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓

(103.)

812\. _“𑀤𑀭𑀺𑀬𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀺𑀁𑀲𑀫𑀢𑁆𑀢𑀸 𑀯𑀲𑀸𑀫𑀲𑁂;_  
_𑀳𑀜𑁆𑀜𑀸𑀫 𑀫𑀡𑀺𑀦𑁄 𑀆𑀪𑀁,_  
_𑀇𑀢𑀺 𑀦𑁄 𑀫𑀦𑁆𑀢𑀦𑀁 𑀅𑀳𑀼𑁇_  


(104.)

813\. _𑀬𑀸𑀯𑀢𑀸 𑀫𑀡𑀺𑀁 𑀖𑀁𑀲𑀸𑀫,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀯𑁄𑀤𑀸𑀬𑀢𑁂 𑀫𑀡𑀺;_  
_𑀇𑀤𑀜𑁆𑀘 𑀤𑀸𑀦𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀺𑀁 𑀓𑀺𑀘𑁆𑀘𑀁 𑀇𑀥 𑀫𑀜𑁆𑀜𑀲𑀺”𑁇_  


(105.)

814\. _“𑀅𑀬𑀁 𑀫𑀡𑀺 𑀯𑁂𑀍𑀊𑀭𑀺𑀬𑁄,_  
_𑀅𑀓𑀸𑀘𑁄 𑀯𑀺𑀫𑀮𑁄 𑀲𑀼𑀪𑁄;_  
_𑀦𑀸𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑀸 𑀲𑀺𑀭𑀺𑀁 𑀳𑀦𑁆𑀢𑀼𑀁,_  
_𑀅𑀧𑀓𑁆𑀓𑀫𑀣 𑀲𑀽𑀓𑀭𑀸”𑀢𑀺𑁇_  


---
