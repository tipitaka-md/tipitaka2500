# 𑁯𑁇𑁧𑁇𑁩 𑀫𑀳𑀸𑀲𑀼𑀯𑀚𑀸𑀢𑀓

* 𑀦𑀯𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀕𑀺𑀚𑁆𑀛𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀲𑀼𑀯𑀚𑀸𑀢𑀓

(20.)

1802\. _“𑀤𑀼𑀫𑁄 𑀬𑀤𑀸 𑀳𑁄𑀢𑀺 𑀨𑀮𑀽𑀧𑀧𑀦𑁆𑀦𑁄,_  
_𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀁 𑀯𑀺𑀳𑀗𑁆𑀕𑀫𑀸 𑀲𑀫𑁆𑀧𑀢𑀦𑁆𑀢𑀸;_  
_𑀔𑀻𑀡𑀦𑁆𑀢𑀺 𑀜𑀢𑁆𑀯𑀸𑀦 𑀤𑀼𑀫𑀁 𑀨𑀮𑀘𑁆𑀘𑀬𑁂,_  
_𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁 𑀬𑀦𑁆𑀢𑀺 𑀢𑀢𑁄 𑀯𑀺𑀳𑀗𑁆𑀕𑀫𑀸𑁇_  


(21.)

1803\. _𑀘𑀭 𑀘𑀸𑀭𑀺𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀢𑀼𑀡𑁆𑀟 𑀫𑀸𑀫𑀭𑀺,_  
_𑀓𑀺𑀁 𑀢𑁆𑀯𑀁 𑀲𑀼𑀯 𑀲𑀼𑀓𑁆𑀔𑀤𑀼𑀫𑀫𑁆𑀳𑀺 𑀛𑀸𑀬𑀲𑀺;_  
_𑀢𑀤𑀺𑀗𑁆𑀖 𑀫𑀁 𑀩𑁆𑀭𑀽𑀳𑀺 𑀯𑀲𑀦𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀪,_  
_𑀓𑀲𑁆𑀫𑀸 𑀲𑀼𑀯 𑀲𑀼𑀓𑁆𑀔𑀤𑀼𑀫𑀁 𑀦 𑀭𑀺𑀜𑁆𑀘𑀲𑀺”𑁇_  


(22.)

1804\. _“𑀬𑁂 𑀯𑁂 𑀲𑀔𑀻𑀦𑀁 𑀲𑀔𑀸𑀭𑁄 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀧𑀸𑀡𑀘𑁆𑀘𑀬𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀼𑀔𑁂𑀲𑀼 𑀳𑀁𑀲;_  
_𑀔𑀻𑀡𑀁 𑀅𑀔𑀻𑀡𑀫𑁆𑀧𑀺 𑀦 𑀢𑀁 𑀚𑀳𑀦𑁆𑀢𑀺,_  
_𑀲𑀦𑁆𑀢𑁄 𑀲𑀢𑀁 𑀥𑀫𑁆𑀫𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀸𑁇_  


(23.)

1805\. _𑀲𑁄𑀳𑀁 𑀲𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑀲𑁆𑀫𑀺 𑀳𑀁𑀲,_  
_𑀜𑀸𑀢𑀻 𑀘 𑀫𑁂 𑀳𑁄𑀢𑀺 𑀲𑀔𑀸 𑀘 𑀭𑀼𑀓𑁆𑀔𑁄;_  
_𑀢𑀁 𑀦𑀼𑀲𑁆𑀲𑀳𑁂 𑀚𑀻𑀯𑀺𑀓𑀢𑁆𑀣𑁄 𑀧𑀳𑀸𑀢𑀼𑀁,_  
_𑀔𑀻𑀡𑀦𑁆𑀢𑀺 𑀜𑀢𑁆𑀯𑀸𑀦 𑀦 𑀳𑁂𑀲 𑀥𑀫𑁆𑀫𑁄”𑁇_  


(24.)

1806\. _“𑀲𑀸𑀥𑀼 𑀲𑀓𑁆𑀔𑀺 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺,_  
_𑀫𑁂𑀢𑁆𑀢𑀺 𑀲𑀁𑀲𑀢𑀺 𑀲𑀦𑁆𑀣𑀯𑁄;_  
_𑀲𑀘𑁂𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀲𑀺,_  
_𑀧𑀸𑀲𑀁𑀲𑁄𑀲𑀺 𑀯𑀺𑀚𑀸𑀦𑀢𑀁𑁇_  


(25.)

1807\. _𑀲𑁄 𑀢𑁂 𑀲𑀼𑀯 𑀯𑀭𑀁 𑀤𑀫𑁆𑀫𑀺,_  
_𑀧𑀢𑁆𑀢𑀬𑀸𑀦 𑀯𑀺𑀳𑀗𑁆𑀕𑀫;_  
_𑀯𑀭𑀁 𑀯𑀭𑀲𑁆𑀲𑀼 𑀯𑀓𑁆𑀓𑀗𑁆𑀕,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀫𑀦𑀲𑀺𑀘𑁆𑀙𑀲𑀺”𑁇_  


(26.)

1808\. _“𑀯𑀭𑀜𑁆𑀘 𑀫𑁂 𑀳𑀁𑀲 𑀪𑀯𑀁 𑀤𑀤𑁂𑀬𑁆𑀬,_  
_𑀅𑀬𑀜𑁆𑀘 𑀭𑀼𑀓𑁆𑀔𑁄 𑀧𑀼𑀦𑀭𑀸𑀬𑀼𑀁 𑀮𑀪𑁂𑀣;_  
_𑀲𑁄 𑀲𑀸𑀔𑀯𑀸 𑀨𑀮𑀺𑀫𑀸 𑀲𑀁𑀯𑀺𑀭𑀽𑀍𑀳𑁄,_  
_𑀫𑀥𑀼𑀢𑁆𑀣𑀺𑀓𑁄 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑁄𑀪𑀫𑀸𑀦𑁄”𑁇_  


(27.)

1809\. _“𑀢𑀁 𑀧𑀲𑁆𑀲 𑀲𑀫𑁆𑀫 𑀨𑀮𑀺𑀫𑀁 𑀉𑀍𑀆𑀭𑀁,_  
_𑀲𑀳𑀸𑀯 𑀢𑁂 𑀳𑁄𑀢𑀼 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑁂𑀦;_  
_𑀲𑁄 𑀲𑀸𑀔𑀯𑀸 𑀨𑀮𑀺𑀫𑀸 𑀲𑀁𑀯𑀺𑀭𑀽𑀍𑀳𑁄,_  
_𑀫𑀥𑀼𑀢𑁆𑀣𑀺𑀓𑁄 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑁄𑀪𑀫𑀸𑀦𑁄”𑁇_  


(28.)

1810\. _“𑀏𑀯𑀁 𑀲𑀓𑁆𑀓 𑀲𑀼𑀔𑀻 𑀳𑁄𑀳𑀺,_  
_𑀲𑀳 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀜𑀸𑀢𑀺𑀪𑀺;_  
_𑀬𑀣𑀸𑀳𑀫𑀚𑁆𑀚 𑀲𑀼𑀔𑀺𑀢𑁄,_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀨𑀮𑀁 𑀤𑀼𑀫𑀁”𑁇_  


(29.)

1811\. _“𑀲𑀼𑀯𑀲𑁆𑀲 𑀘 𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀲𑀨𑀮𑀁 𑀤𑀼𑀫𑀁;_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀲𑀳 𑀪𑀭𑀺𑀬𑀸𑀬,_  
_𑀤𑁂𑀯𑀸𑀦𑀁 𑀦𑀦𑁆𑀤𑀦𑀁 𑀯𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

1812\. 𑀫𑀳𑀸𑀲𑀼𑀯𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
