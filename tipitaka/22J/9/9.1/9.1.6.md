# 𑁯𑁇𑁧𑁇𑁬 𑀧𑀤𑀓𑀼𑀲𑀮𑀫𑀸𑀡𑀯𑀚𑀸𑀢𑀓

### 𑀦𑀯𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀕𑀺𑀚𑁆𑀛𑀯𑀕𑁆𑀕

### 𑀧𑀤𑀓𑀼𑀲𑀮𑀫𑀸𑀡𑀯𑀚𑀸𑀢𑀓

(49.)

1834. _“𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀓𑀣𑀺𑀁,_  
_𑀕𑀗𑁆𑀕𑀸 𑀯𑀳𑀢𑀺 𑀧𑀸𑀝𑀮𑀺𑀁;_  
_𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑀓 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀏𑀓𑀁 𑀫𑁂 𑀤𑁂𑀳𑀺 𑀕𑀸𑀣𑀓𑀁”𑁇_  


(50.)

1835. _“𑀬𑁂𑀦 𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁,_  
_𑀬𑁂𑀦 𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺 𑀆𑀢𑀼𑀭𑀁;_  
_𑀢𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(51.)

1836. _“𑀬𑀢𑁆𑀣 𑀩𑀻𑀚𑀸𑀦𑀺 𑀭𑀽𑀳𑀦𑁆𑀢𑀺,_  
_𑀲𑀢𑁆𑀢𑀸 𑀬𑀢𑁆𑀣 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀲𑀸 𑀫𑁂 𑀲𑀻𑀲𑀁 𑀦𑀺𑀧𑀻𑀍𑀏𑀢𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(52.)

1837. _“𑀬𑁂𑀦 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀲𑀻𑀢𑀁 𑀬𑁂𑀦 𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀺;_  
_𑀲𑁄 𑀫𑀁 𑀟𑀳𑀢𑀺 𑀕𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(53.)

1838. _“𑀬𑁂𑀦 𑀪𑀼𑀢𑁆𑀢𑁂𑀦 𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀣𑀽 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑁄 𑀫𑀁 𑀪𑀼𑀢𑁆𑀢𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂𑀢𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(54.)

1839. _“𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂,_  
_𑀯𑀸𑀢𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀲𑁄 𑀫𑀁 𑀪𑀜𑁆𑀚𑀢𑀺 𑀕𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(55.)

1840. _“𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀚𑀕𑀢𑀺𑀭𑀼𑀳𑀁,_  
_𑀲𑁆𑀯𑀸𑀬𑀁 𑀅𑀕𑁆𑀕𑀺𑀁 𑀧𑀫𑀼𑀜𑁆𑀘𑀢𑀺;_  
_𑀤𑀺𑀲𑀸 𑀪𑀚𑀣 𑀯𑀓𑁆𑀓𑀗𑁆𑀕𑀸,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(56.)

1841. _“𑀬𑀫𑀸𑀦𑀬𑀺𑀁 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁,_  
_𑀫𑀸𑀮𑀺𑀦𑀺𑀁 𑀘𑀦𑁆𑀤𑀦𑀼𑀲𑁆𑀲𑀤𑀁;_  
_𑀲𑀸 𑀫𑀁 𑀖𑀭𑀸 𑀦𑀺𑀘𑁆𑀙𑀼𑀪𑀢𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(57.)

1842. _“𑀬𑁂𑀦 𑀚𑀸𑀢𑁂𑀦 𑀦𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀁,_  
_𑀬𑀲𑁆𑀲 𑀘 𑀪𑀯𑀫𑀺𑀘𑁆𑀙𑀺𑀲𑀁;_  
_𑀲𑁄 𑀫𑀁 𑀖𑀭𑀸 𑀦𑀺𑀘𑁆𑀙𑀼𑀪𑀢𑀺,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀁”𑁇_  


(58.)

1843. _“𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑁂 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀬𑀢𑁄𑀤𑀓𑀁 𑀢𑀤𑀸𑀤𑀺𑀢𑁆𑀢𑀁,_  
_𑀬𑀢𑁄 𑀔𑁂𑀫𑀁 𑀢𑀢𑁄 𑀪𑀬𑀁𑁇_  


(59.)

1844. _𑀭𑀸𑀚𑀸 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀢𑁂 𑀭𑀝𑁆𑀞𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀘 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑁄;_  
_𑀅𑀢𑁆𑀢𑀕𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣,_  
_𑀚𑀸𑀢𑀁 𑀲𑀭𑀡𑀢𑁄 𑀪𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

1845. 𑀧𑀤𑀓𑀼𑀲𑀮𑀫𑀸𑀡𑀯𑀚𑀸𑀢𑀓𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
