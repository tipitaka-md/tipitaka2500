# 𑁯𑁇𑁧𑁇𑁧 𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓

* 𑀦𑀯𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀕𑀺𑀚𑁆𑀛𑀯𑀕𑁆𑀕

* 𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓

(1.)

1781\. _“𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀼𑀧𑀣𑁄 𑀦𑀸𑀫,_  
_𑀕𑀺𑀚𑁆𑀛𑀧𑀦𑁆𑀣𑁄 𑀲𑀦𑀦𑁆𑀢𑀦𑁄;_  
_𑀢𑀢𑁆𑀭𑀸𑀲𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀕𑀺𑀚𑁆𑀛𑁄 𑀧𑁄𑀲𑁂𑀲𑀺 𑀚𑀺𑀡𑁆𑀡𑀓𑁂;_  
_𑀢𑁂𑀲𑀁 𑀅𑀚𑀕𑀭𑀫𑁂𑀤𑀁,_  
_𑀅𑀘𑁆𑀘𑀳𑀸𑀲𑀺 𑀩𑀳𑀼𑀢𑁆𑀢𑀲𑁄𑁇_  


(2.)

1782\. _𑀧𑀺𑀢𑀸 𑀘 𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀯𑀘,_  
_𑀚𑀸𑀦𑀁 𑀉𑀘𑁆𑀘𑀁 𑀧𑀧𑀸𑀢𑀺𑀦𑀁;_  
_𑀲𑀼𑀧𑀢𑁆𑀢𑀁 𑀣𑀸𑀫𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀢𑁂𑀚𑀲𑁆𑀲𑀺𑀁 𑀤𑀽𑀭𑀕𑀸𑀫𑀺𑀦𑀁𑁇_  


(3.)

1783\. _𑀧𑀭𑀺𑀧𑁆𑀮𑀯𑀦𑁆𑀢𑀁 𑀧𑀣𑀯𑀺𑀁,_  
_𑀬𑀤𑀸 𑀢𑀸𑀢 𑀯𑀺𑀚𑀸𑀦𑀳𑀺;_  
_𑀲𑀸𑀕𑀭𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁,_  
_𑀘𑀓𑁆𑀓𑀁𑀯 𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀁;_  
_𑀢𑀢𑁄 𑀢𑀸𑀢 𑀦𑀺𑀯𑀢𑁆𑀢𑀲𑁆𑀲𑀼,_  
_𑀫𑀸𑀲𑁆𑀲𑀼 𑀏𑀢𑁆𑀢𑁄 𑀧𑀭𑀁 𑀕𑀫𑀺𑁇_  


(4.)

1784\. _𑀉𑀤𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀯𑁂𑀕𑁂𑀦,_  
_𑀩𑀮𑀻 𑀧𑀓𑁆𑀔𑀻 𑀤𑀺𑀚𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀑𑀮𑁄𑀓𑀬𑀦𑁆𑀢𑁄 𑀯𑀓𑁆𑀓𑀗𑁆𑀕𑁄,_  
_𑀧𑀩𑁆𑀩𑀢𑀸𑀦𑀺 𑀯𑀦𑀸𑀦𑀺 𑀘𑁇_  


(5.)

1785\. _𑀅𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀣𑀯𑀺𑀁 𑀕𑀺𑀚𑁆𑀛𑁄,_  
_𑀬𑀣𑀸𑀲𑀸𑀲𑀺 𑀧𑀺𑀢𑀼𑀲𑁆𑀲𑀼𑀢𑀁;_  
_𑀲𑀸𑀕𑀭𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁,_  
_𑀘𑀓𑁆𑀓𑀁𑀯 𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀁𑁇_  


(6.)

1786\. _𑀢𑀜𑁆𑀘 𑀲𑁄 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀧𑀭𑀫𑁂𑀯𑀘𑁆𑀘𑀯𑀢𑁆𑀢𑀣;_  
_𑀢𑀜𑁆𑀘 𑀯𑀸𑀢𑀲𑀺𑀔𑀸 𑀢𑀺𑀓𑁆𑀔𑀸,_  
_𑀅𑀘𑁆𑀘𑀳𑀸𑀲𑀺 𑀩𑀮𑀺𑀁 𑀤𑀺𑀚𑀁𑁇_  


(7.)

1787\. _𑀦𑀸𑀲𑀓𑁆𑀔𑀸𑀢𑀺𑀕𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀧𑀼𑀦𑀤𑁂𑀯 𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀼𑀁;_  
_𑀤𑀺𑀚𑁄 𑀩𑁆𑀬𑀲𑀦𑀫𑀸𑀧𑀸𑀤𑀺,_  
_𑀯𑁂𑀭𑀫𑁆𑀪𑀸𑀦𑀁 𑀯𑀲𑀁 𑀕𑀢𑁄𑁇_  


(8.)

1788\. _𑀢𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀤𑀸𑀭𑀸 𑀘,_  
_𑀬𑁂 𑀘𑀜𑁆𑀜𑁂 𑀅𑀦𑀼𑀚𑀻𑀯𑀺𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑁂 𑀩𑁆𑀬𑀲𑀦𑀫𑀸𑀧𑀸𑀤𑀼𑀁,_  
_𑀅𑀦𑁄𑀯𑀸𑀤𑀓𑀭𑁂 𑀤𑀺𑀚𑁂𑁇_  


(9.)

1789\. _𑀏𑀯𑀫𑁆𑀧𑀺 𑀇𑀥 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁,_  
_𑀬𑁄 𑀯𑀸𑀓𑁆𑀬𑀁 𑀦𑀸𑀯𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀅𑀢𑀺𑀲𑀻𑀫𑀘𑀭𑁄 𑀤𑀺𑀢𑁆𑀢𑁄,_  
_𑀕𑀺𑀚𑁆𑀛𑁄𑀯𑀸𑀢𑀻𑀢𑀲𑀸𑀲𑀦𑁄;_  
_𑀲 𑀯𑁂 𑀩𑁆𑀬𑀲𑀦𑀁 𑀧𑀧𑁆𑀧𑁄𑀢𑀺,_  
_𑀅𑀓𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

1790\. 𑀕𑀺𑀚𑁆𑀛𑀚𑀸𑀢𑀓𑀁 𑀧𑀞𑀫𑀁𑁇
