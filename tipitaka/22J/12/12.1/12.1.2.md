# 𑁧𑁨𑁇𑁧𑁇𑁨 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑀚𑀸𑀢𑀓

### 𑀤𑁆𑀯𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀽𑀍𑀅𑀓𑀼𑀡𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑀚𑀸𑀢𑀓

(13.)

2252\. _“𑀓𑀸 𑀢𑁆𑀯𑀁 𑀲𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀯𑀢𑁆𑀣𑁂𑀳𑀺,_  
_𑀅𑀖𑁂 𑀯𑁂𑀳𑀸𑀬𑀲𑀁 𑀞𑀺𑀢𑀸;_  
_𑀓𑁂𑀦 𑀢𑁆𑀬𑀸𑀲𑁆𑀲𑀽𑀦𑀺 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀓𑀼𑀢𑁄 𑀢𑀁 𑀪𑀬𑀫𑀸𑀕𑀢𑀁”𑁇_  


(14.)

2253\. _“𑀢𑀯𑁂𑀯 𑀤𑁂𑀯 𑀯𑀺𑀚𑀺𑀢𑁂,_  
_𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑁄𑀢𑀺 𑀫𑀁 𑀯𑀺𑀤𑀽;_  
_𑀲𑀝𑁆𑀞𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀺𑀝𑁆𑀞𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑀲𑁆𑀲 𑀫𑁂𑁇_  


(15.)

2254\. _𑀓𑀸𑀭𑀬𑀦𑁆𑀢𑀸 𑀦𑀕𑀭𑀸𑀦𑀺,_  
_𑀅𑀕𑀸𑀭𑁂 𑀘 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀯𑀺𑀯𑀺𑀥𑁂 𑀘𑀸𑀧𑀺 𑀧𑀸𑀲𑀸𑀤𑁂,_  
_𑀦 𑀫𑀁 𑀢𑁂 𑀅𑀘𑁆𑀘𑀫𑀜𑁆𑀜𑀺𑀲𑀼𑀁;_  
_𑀬𑀣𑁂𑀯 𑀫𑀁 𑀢𑁂 𑀧𑀽𑀚𑁂𑀲𑀼𑀁,_  
_𑀢𑀣𑁂𑀯 𑀢𑁆𑀯𑀫𑁆𑀧𑀺 𑀧𑀽𑀚𑀬”𑁇_  


(16.)

2255\. _“𑀢𑀁 𑀇𑀯𑀸𑀳𑀁 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀣𑀽𑀮𑀁 𑀓𑀸𑀬𑁂𑀦 𑀢𑁂 𑀤𑀼𑀫𑀁;_  
_𑀆𑀭𑁄𑀳𑀧𑀭𑀺𑀡𑀸𑀳𑁂𑀦,_  
_𑀅𑀪𑀺𑀭𑀽𑀧𑁄𑀲𑀺 𑀚𑀸𑀢𑀺𑀬𑀸𑁇_  


(17.)

2256\. _𑀧𑀸𑀲𑀸𑀤𑀁 𑀓𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀏𑀓𑀢𑁆𑀣𑀫𑁆𑀪𑀁 𑀫𑀦𑁄𑀭𑀫𑀁;_  
_𑀢𑀢𑁆𑀣 𑀢𑀁 𑀉𑀧𑀦𑁂𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀘𑀺𑀭𑀁 𑀢𑁂 𑀬𑀓𑁆𑀔 𑀚𑀻𑀯𑀺𑀢𑀁”𑁇_  


(18.)

2257\. _“𑀏𑀯𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀤𑀧𑀸𑀤𑀺,_  
_𑀲𑀭𑀻𑀭𑁂𑀦 𑀯𑀺𑀦𑀸𑀪𑀸𑀯𑁄;_  
_𑀧𑀼𑀣𑀼𑀲𑁄 𑀫𑀁 𑀯𑀺𑀓𑀦𑁆𑀢𑀺𑀢𑁆𑀯𑀸,_  
_𑀔𑀡𑁆𑀟𑀲𑁄 𑀅𑀯𑀓𑀦𑁆𑀢𑀣𑁇_  


(19.)

2258\. _𑀅𑀕𑁆𑀕𑁂 𑀘 𑀙𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀘,_  
_𑀧𑀘𑁆𑀙𑀸 𑀫𑀽𑀮𑀫𑁆𑀳𑀺 𑀙𑀺𑀦𑁆𑀤𑀣;_  
_𑀏𑀯𑀁 𑀫𑁂 𑀙𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀫𑀭𑀡𑀁 𑀲𑀺𑀬𑀸”𑁇_  


(20.)

2259\. _“𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀁 𑀬𑀣𑀸 𑀙𑀺𑀦𑁆𑀤𑁂,_  
_𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀜𑁆𑀘 𑀚𑀻𑀯𑀢𑁄;_  
_𑀢𑀢𑁄 𑀧𑀘𑁆𑀙𑀸 𑀲𑀺𑀭𑁄 𑀙𑀺𑀦𑁆𑀤𑁂,_  
_𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀫𑀭𑀡𑀁 𑀲𑀺𑀬𑀸𑁇_  


(21.)

2260\. _𑀲𑀼𑀔𑀁 𑀦𑀼 𑀔𑀡𑁆𑀟𑀲𑁄 𑀙𑀺𑀦𑁆𑀦𑀁,_  
_𑀪𑀤𑁆𑀤𑀲𑀸𑀮 𑀯𑀦𑀧𑁆𑀧𑀢𑀺;_  
_𑀓𑀺𑀁𑀳𑁂𑀢𑀼 𑀓𑀺𑀁 𑀉𑀧𑀸𑀤𑀸𑀬,_  
_𑀔𑀡𑁆𑀟𑀲𑁄 𑀙𑀺𑀦𑁆𑀦𑀫𑀺𑀘𑁆𑀙𑀲𑀺”𑁇_  


(22.)

2261\. _“𑀬𑀜𑁆𑀘 𑀳𑁂𑀢𑀼𑀫𑀼𑀧𑀸𑀤𑀸𑀬,_  
_𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁;_  
_𑀔𑀡𑁆𑀟𑀲𑁄 𑀙𑀺𑀦𑁆𑀦𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


(23.)

2262\. _𑀜𑀸𑀢𑀻 𑀫𑁂 𑀲𑀼𑀔𑀲𑀁𑀯𑀤𑁆𑀥𑀸,_  
_𑀫𑀫 𑀧𑀲𑁆𑀲𑁂 𑀦𑀺𑀯𑀸𑀢𑀚𑀸;_  
_𑀢𑁂𑀧𑀺𑀳𑀁 𑀉𑀧𑀳𑀺𑀁𑀲𑁂𑀬𑁆𑀬𑀁,_  
_𑀧𑀭𑁂𑀲𑀁 𑀅𑀲𑀼𑀔𑁄𑀘𑀺𑀢𑀁”𑁇_  


(24.)

2263\. _“𑀘𑁂𑀢𑁂𑀬𑁆𑀬𑀭𑀽𑀧𑀁 𑀘𑁂𑀢𑁂𑀲𑀺,_  
_𑀪𑀤𑁆𑀤𑀲𑀸𑀮 𑀯𑀦𑀧𑁆𑀧𑀢𑀺;_  
_𑀳𑀺𑀢𑀓𑀸𑀫𑁄𑀲𑀺 𑀜𑀸𑀢𑀻𑀦𑀁,_  
_𑀅𑀪𑀬𑀁 𑀲𑀫𑁆𑀫 𑀤𑀫𑁆𑀫𑀺 𑀢𑁂”𑀢𑀺𑁇_  


---

2264\. 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
