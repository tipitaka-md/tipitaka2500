# 𑁧𑁨𑁇𑁧𑁇𑁪 𑀓𑀸𑀫𑀚𑀸𑀢𑀓

### 𑀤𑁆𑀯𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀽𑀍𑀅𑀓𑀼𑀡𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀫𑀚𑀸𑀢𑀓

(37.)

2278. _“𑀓𑀸𑀫𑀁 𑀓𑀸𑀫𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀢𑀲𑁆𑀲 𑀘𑁂 𑀢𑀁 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀅𑀤𑁆𑀥𑀸 𑀧𑀻𑀢𑀺𑀫𑀦𑁄 𑀳𑁄𑀢𑀺,_  
_𑀮𑀤𑁆𑀥𑀸 𑀫𑀘𑁆𑀘𑁄 𑀬𑀤𑀺𑀘𑁆𑀙𑀢𑀺𑁇_  


(38.)

2279. _𑀓𑀸𑀫𑀁 𑀓𑀸𑀫𑀬𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀢𑀲𑁆𑀲 𑀘𑁂 𑀢𑀁 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀢𑀢𑁄 𑀦𑀁 𑀅𑀧𑀭𑀁 𑀓𑀸𑀫𑁂,_  
_𑀖𑀫𑁆𑀫𑁂 𑀢𑀡𑁆𑀳𑀁𑀯 𑀯𑀺𑀦𑁆𑀤𑀢𑀺𑁇_  


(39.)

2280. _𑀕𑀯𑀁𑀯 𑀲𑀺𑀗𑁆𑀕𑀺𑀦𑁄 𑀲𑀺𑀗𑁆𑀕𑀁,_  
_𑀯𑀟𑁆𑀠𑀫𑀸𑀦𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀢𑀺;_  
_𑀏𑀯𑀁 𑀫𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲,_  
_𑀩𑀸𑀮𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑀸𑀦𑀢𑁄;_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀢𑀡𑁆𑀳𑀸 𑀧𑀺𑀧𑀸𑀲𑀸 𑀘,_  
_𑀯𑀟𑁆𑀠𑀫𑀸𑀦𑀲𑁆𑀲 𑀯𑀟𑁆𑀠𑀢𑀺𑁇_  


(40.)

2281. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀲𑀸𑀮𑀺𑀬𑀯𑀓𑀁,_  
_𑀕𑀯𑀸𑀲𑁆𑀲𑀁 𑀤𑀸𑀲𑀧𑁄𑀭𑀺𑀲𑀁;_  
_𑀤𑀢𑁆𑀯𑀸 𑀘 𑀦𑀸𑀮𑀫𑁂𑀓𑀲𑁆𑀲,_  
_𑀇𑀢𑀺 𑀯𑀺𑀤𑁆𑀯𑀸 𑀲𑀫𑀁 𑀘𑀭𑁂𑁇_  


(41.)

2282. _𑀭𑀸𑀚𑀸 𑀧𑀲𑀬𑁆𑀳 𑀧𑀣𑀯𑀺𑀁 𑀯𑀺𑀚𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀲𑀸𑀕𑀭𑀦𑁆𑀢𑀁 𑀫𑀳𑀺𑀫𑀸𑀯𑀲𑀦𑁆𑀢𑁄;_  
_𑀑𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀅𑀢𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑁄,_  
_𑀧𑀸𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲𑀧𑀺 𑀧𑀢𑁆𑀣𑀬𑁂𑀣𑁇_  


(42.)

2283. _𑀬𑀸𑀯 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀁 𑀓𑀸𑀫𑁂,_  
_𑀫𑀦𑀲𑀸 𑀢𑀺𑀢𑁆𑀢𑀺 𑀦𑀸𑀚𑁆𑀛𑀕𑀸;_  
_𑀢𑀢𑁄 𑀦𑀺𑀯𑀢𑁆𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀢𑁂 𑀯𑁂 𑀲𑀼𑀢𑀺𑀢𑁆𑀢𑀸 𑀬𑁂 𑀧𑀜𑁆𑀜𑀸𑀬 𑀢𑀺𑀢𑁆𑀢𑀸𑁇_  


(43.)

2284. _𑀧𑀜𑁆𑀜𑀸𑀬 𑀢𑀺𑀢𑁆𑀢𑀺𑀦𑀁 𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀦 𑀲𑁄 𑀓𑀸𑀫𑁂𑀳𑀺 𑀢𑀧𑁆𑀧𑀢𑀺;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀢𑀺𑀢𑁆𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁,_  
_𑀢𑀡𑁆𑀳𑀸 𑀦 𑀓𑀼𑀭𑀼𑀢𑁂 𑀯𑀲𑀁𑁇_  


(44.)

2285. _𑀅𑀧𑀘𑀺𑀦𑁂𑀣𑁂𑀯 𑀓𑀸𑀫𑀸𑀦𑀁,_  
_𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀅𑀮𑁄𑀮𑀼𑀧𑁄;_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑀫𑀢𑁆𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀲𑁄 𑀓𑀸𑀫𑁂𑀳𑀺 𑀢𑀧𑁆𑀧𑀢𑀺𑁇_  


(45.)

2286. _𑀭𑀣𑀓𑀸𑀭𑁄𑀯 𑀘𑀫𑁆𑀫𑀲𑁆𑀲,_  
_𑀧𑀭𑀺𑀓𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀳𑀦𑀁;_  
_𑀬𑀁 𑀬𑀁 𑀘𑀚𑀢𑀺 𑀓𑀸𑀫𑀸𑀦𑀁,_  
_𑀢𑀁 𑀢𑀁 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀢𑁂 𑀲𑀼𑀔𑀁;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀘𑁂 𑀲𑀼𑀔𑀫𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_𑀲𑀩𑁆𑀩𑁂 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑁂”𑁇_  


(46.)

2287. _“𑀅𑀝𑁆𑀞 𑀢𑁂 𑀪𑀸𑀲𑀺𑀢𑀸 𑀕𑀸𑀣𑀸,_  
_𑀲𑀩𑁆𑀩𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀳𑀲𑁆𑀲𑀺𑀬𑁄;_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀲𑀸𑀥𑁂𑀢𑀁 𑀢𑀯 𑀪𑀸𑀲𑀺𑀢𑀁”𑁇_  


(47.)

2288. _“𑀦 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀲𑀢𑁂𑀳𑀺 𑀦𑀳𑀼𑀢𑁂𑀳𑀺 𑀯𑀸;_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀪𑀸𑀲𑀢𑁄 𑀕𑀸𑀣𑀁,_  
_𑀓𑀸𑀫𑁂 𑀫𑁂 𑀦 𑀭𑀢𑁄 𑀫𑀦𑁄”𑁇_  


(48.)

2289. _“𑀪𑀤𑁆𑀭𑀓𑁄 𑀯𑀢𑀸𑀬𑀁 𑀫𑀸𑀡𑀯𑀓𑁄,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀫𑀼𑀦𑀺;_  
_𑀬𑁄 𑀇𑀫𑀁 𑀢𑀡𑁆𑀳𑀁 𑀤𑀼𑀓𑁆𑀔𑀚𑀦𑀦𑀺𑀁,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄”𑀢𑀺𑁇_  


---

2290. 𑀓𑀸𑀫𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
