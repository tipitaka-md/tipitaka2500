# 𑁧𑁨𑁇𑁧𑁇𑁬 𑀫𑀳𑀸𑀓𑀡𑁆𑀳𑀚𑀸𑀢𑀓

### 𑀤𑁆𑀯𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀽𑀍𑀅𑀓𑀼𑀡𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀓𑀡𑁆𑀳𑀚𑀸𑀢𑀓

(61.)

2304\. _“𑀓𑀡𑁆𑀳𑁄 𑀓𑀡𑁆𑀳𑁄 𑀘 𑀖𑁄𑀭𑁄 𑀘,_  
_𑀲𑀼𑀓𑁆𑀓𑀤𑀸𑀞𑁄 𑀧𑀪𑀸𑀲𑀯𑀸;_  
_𑀩𑀤𑁆𑀥𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀭𑀚𑁆𑀚𑀽𑀳𑀺,_  
_𑀓𑀺𑀁 𑀭𑀯𑀺 𑀲𑀼𑀦𑀔𑁄 𑀢𑀯”𑁇_  


(62.)

2305\. _“𑀦𑀸𑀬𑀁 𑀫𑀺𑀕𑀸𑀦𑀫𑀢𑁆𑀣𑀸𑀬,_  
_𑀉𑀲𑀻𑀦𑀓 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀦𑀬𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺”𑁇_  


(63.)

2306\. _“𑀧𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀸 𑀲𑀫𑀡𑀓𑀸,_  
_𑀫𑀼𑀡𑁆𑀟𑀸 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀦𑀗𑁆𑀕𑀮𑁂𑀳𑀺 𑀓𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(64.)

2307\. _𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀺𑀬𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸,_  
_𑀫𑀼𑀡𑁆𑀟𑀸 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀬𑀤𑀸 𑀮𑁄𑀓𑁂 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(65.)

2308\. _𑀤𑀻𑀖𑁄𑀢𑁆𑀢𑀭𑁄𑀝𑁆𑀞𑀸 𑀚𑀝𑀺𑀮𑀸,_  
_𑀧𑀗𑁆𑀓𑀤𑀦𑁆𑀢𑀸 𑀭𑀚𑀲𑁆𑀲𑀺𑀭𑀸;_  
_𑀇𑀡𑀁 𑀘𑁄𑀤𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(66.)

2309\. _𑀅𑀥𑀺𑀘𑁆𑀘 𑀯𑁂𑀤𑁂 𑀲𑀸𑀯𑀺𑀢𑁆𑀢𑀺𑀁,_  
_𑀬𑀜𑁆𑀜𑀢𑀦𑁆𑀢𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀪𑀢𑀺𑀓𑀸𑀬 𑀬𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(67.)

2310\. _𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀚𑀺𑀡𑁆𑀡𑀓𑀁 𑀕𑀢𑀬𑁄𑀩𑁆𑀩𑀦𑀁;_  
_𑀧𑀳𑀽 𑀲𑀦𑁆𑀢𑁄 𑀦 𑀪𑀭𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(68.)

2311\. _𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀚𑀺𑀡𑁆𑀡𑀓𑀁 𑀕𑀢𑀬𑁄𑀩𑁆𑀩𑀦𑀁;_  
_𑀩𑀸𑀮𑀸 𑀢𑀼𑀫𑁆𑀳𑁂𑀢𑀺 𑀯𑀓𑁆𑀔𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(69.)

2312\. _𑀆𑀘𑀭𑀺𑀬𑀪𑀭𑀺𑀬𑀁 𑀲𑀔𑀺𑀁,_  
_𑀫𑀸𑀢𑀼𑀮𑀸𑀦𑀺𑀁 𑀧𑀺𑀢𑀼𑀘𑁆𑀙𑀓𑀺𑀁;_  
_𑀬𑀤𑀸 𑀮𑁄𑀓𑁂 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(70.)

2313\. _𑀅𑀲𑀺𑀘𑀫𑁆𑀫𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀔𑀕𑁆𑀕𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀧𑀦𑁆𑀣𑀖𑀸𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(71.)

2314\. _𑀲𑀼𑀓𑁆𑀓𑀘𑁆𑀙𑀯𑀻 𑀯𑁂𑀥𑀯𑁂𑀭𑀸,_  
_𑀣𑀽𑀮𑀩𑀸𑀳𑀽 𑀅𑀧𑀸𑀢𑀼𑀪𑀸;_  
_𑀫𑀺𑀢𑁆𑀢𑀪𑁂𑀤𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀺𑁇_  


(72.)

2315\. _𑀫𑀸𑀬𑀸𑀯𑀺𑀦𑁄 𑀦𑁂𑀓𑀢𑀺𑀓𑀸,_  
_𑀅𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀘𑀺𑀦𑁆𑀢𑀓𑀸;_  
_𑀬𑀤𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀤𑀸 𑀓𑀡𑁆𑀳𑁄 𑀧𑀫𑁄𑀓𑁆𑀔𑀢𑀻”𑀢𑀺𑁇_  


---
