# 𑁧𑁪𑁇𑁧𑁇𑁧𑁦 𑀫𑀳𑀸𑀯𑀸𑀡𑀺𑀚𑀚𑀸𑀢𑀓

* 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀸𑀮𑀺𑀓𑁂𑀤𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀯𑀸𑀡𑀺𑀚𑀚𑀸𑀢𑀓

(180.)

2736\. _𑀯𑀸𑀡𑀺𑀚𑀸 𑀲𑀫𑀺𑀢𑀺𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦𑀸𑀦𑀸𑀭𑀝𑁆𑀞𑀢𑁄 𑀆𑀕𑀢𑀸;_  
_𑀥𑀦𑀸𑀳𑀭𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼,_  
_𑀏𑀓𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀕𑀸𑀫𑀡𑀺𑀁𑁇_  


(181.)

2737\. _𑀢𑁂 𑀢𑀁 𑀓𑀦𑁆𑀢𑀸𑀭𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀅𑀧𑁆𑀧𑀪𑀓𑁆𑀔𑀁 𑀅𑀦𑁄𑀤𑀓𑀁;_  
_𑀫𑀳𑀸𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀼𑀁,_  
_𑀲𑀻𑀢𑀘𑁆𑀙𑀸𑀬𑀁 𑀫𑀦𑁄𑀭𑀫𑀁𑁇_  


(182.)

2738\. _𑀢𑁂 𑀘 𑀢𑀢𑁆𑀣 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀢𑀲𑁆𑀲 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀙𑀸𑀬𑀬𑀸;_  
_𑀯𑀸𑀡𑀺𑀚𑀸 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁,_  
_𑀩𑀸𑀮𑀸 𑀫𑁄𑀳𑁂𑀦 𑀧𑀸𑀭𑀼𑀢𑀸𑁇_  


(183.)

2739\. _𑀅𑀮𑁆𑀮𑀸𑀬𑀢𑁂 𑀅𑀬𑀁 𑀭𑀼𑀓𑁆𑀔𑁄,_  
_𑀅𑀧𑀺 𑀯𑀸𑀭𑀻𑀯 𑀲𑀦𑁆𑀤𑀢𑀺;_  
_𑀇𑀗𑁆𑀖𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀫𑀁 𑀲𑀸𑀔𑀁,_  
_𑀫𑀬𑀁 𑀙𑀺𑀦𑁆𑀤𑀸𑀫 𑀯𑀸𑀡𑀺𑀚𑀸𑁇_  


(184.)

2740\. _𑀲𑀸 𑀘 𑀙𑀺𑀦𑁆𑀦𑀸𑀯 𑀧𑀕𑁆𑀖𑀭𑀺,_  
_𑀅𑀘𑁆𑀙𑀁 𑀯𑀸𑀭𑀺𑀁 𑀅𑀦𑀸𑀯𑀺𑀮𑀁;_  
_𑀢𑁂 𑀢𑀢𑁆𑀣 𑀦𑁆𑀳𑀢𑁆𑀯𑀸 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀢𑀺𑀘𑁆𑀙𑀺𑀁𑀲𑀼 𑀯𑀸𑀡𑀺𑀚𑀸𑁇_  


(185.)

2741\. _𑀤𑀼𑀢𑀺𑀬𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁,_  
_𑀩𑀸𑀮𑀸 𑀫𑁄𑀳𑁂𑀦 𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀇𑀗𑁆𑀖𑀲𑁆𑀲 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀲𑀸𑀔𑀁,_  
_𑀫𑀬𑀁 𑀙𑀺𑀦𑁆𑀤𑀸𑀫 𑀯𑀸𑀡𑀺𑀚𑀸𑁇_  


(186.)

2742\. _𑀲𑀸 𑀘 𑀙𑀺𑀦𑁆𑀦𑀸𑀯 𑀧𑀕𑁆𑀖𑀭𑀺,_  
_𑀲𑀸𑀮𑀺𑀫𑀁𑀲𑁄𑀤𑀦𑀁 𑀩𑀳𑀼𑀁;_  
_𑀅𑀧𑁆𑀧𑁄𑀤𑀯𑀡𑁆𑀡𑁂 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑁂,_  
_𑀲𑀺𑀗𑁆𑀕𑀺𑀁 𑀯𑀺𑀤𑀮𑀲𑀽𑀧𑀺𑀬𑁄𑁇_  


(187.)

2743\. _𑀢𑁂 𑀢𑀢𑁆𑀣 𑀪𑀼𑀢𑁆𑀯𑀸 𑀔𑀸𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀢𑀺𑀘𑁆𑀙𑀺𑀁𑀲𑀼 𑀯𑀸𑀡𑀺𑀚𑀸;_  
_𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁,_  
_𑀩𑀸𑀮𑀸 𑀫𑁄𑀳𑁂𑀦 𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀇𑀗𑁆𑀖𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀲𑀸𑀔𑀁,_  
_𑀫𑀬𑀁 𑀙𑀺𑀦𑁆𑀤𑀸𑀫 𑀯𑀸𑀡𑀺𑀚𑀸𑁇_  


(188.)

2744\. _𑀲𑀸 𑀘 𑀙𑀺𑀦𑁆𑀦𑀸𑀯 𑀧𑀕𑁆𑀖𑀭𑀺,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀭𑀯𑀢𑁆𑀣𑀸𑀪𑀭𑀡𑀸,_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀸𑁇_  


(189.)

2745\. _𑀅𑀧𑀺 𑀲𑀼 𑀯𑀸𑀡𑀺𑀚𑀸 𑀏𑀓𑀸,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀧𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀁𑀲𑀼,_  
_𑀢𑀲𑁆𑀲 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀙𑀸𑀬𑀬𑀸𑁇_  


(190.)

2746\. _𑀢𑁂 𑀢𑀸𑀳𑀺 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀢𑀺𑀘𑁆𑀙𑀺𑀁𑀲𑀼 𑀯𑀸𑀡𑀺𑀚𑀸;_  
_𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁,_  
_𑀩𑀸𑀮𑀸 𑀫𑁄𑀳𑁂𑀦 𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀇𑀗𑁆𑀖𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀁 𑀲𑀸𑀔𑀁,_  
_𑀫𑀬𑀁 𑀙𑀺𑀦𑁆𑀤𑀸𑀫 𑀯𑀸𑀡𑀺𑀚𑀸𑁇_  


(191.)

2747\. _𑀲𑀸 𑀘 𑀙𑀺𑀦𑁆𑀦𑀸𑀯 𑀧𑀕𑁆𑀖𑀭𑀺,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀩𑀳𑀽;_  
_𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀭𑀽𑀧𑀜𑁆𑀘,_  
_𑀓𑀼𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀝𑀺𑀬𑀸𑀦𑀺 𑀘𑁇_  


(192.)

2748\. _𑀓𑀸𑀲𑀺𑀓𑀸𑀦𑀺 𑀘 𑀯𑀢𑁆𑀣𑀸𑀦𑀺,_  
_𑀉𑀤𑁆𑀤𑀺𑀬𑀸𑀦𑀺 𑀘 𑀓𑀫𑁆𑀩𑀮𑀸;_  
_𑀢𑁂 𑀢𑀢𑁆𑀣 𑀪𑀸𑀭𑁂 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀢𑀺𑀘𑁆𑀙𑀺𑀁𑀲𑀼 𑀯𑀸𑀡𑀺𑀚𑀸𑁇_  


(193.)

2749\. _𑀧𑀜𑁆𑀘𑀫𑀁 𑀲𑀫𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀼𑀁,_  
_𑀩𑀸𑀮𑀸 𑀫𑁄𑀳𑁂𑀦 𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀇𑀗𑁆𑀖𑀲𑁆𑀲 𑀫𑀽𑀮𑀁 𑀙𑀺𑀦𑁆𑀤𑀸𑀫,_  
_𑀅𑀧𑀺 𑀪𑀺𑀬𑁆𑀬𑁄 𑀮𑀪𑀸𑀫𑀲𑁂𑁇_  


(194.)

2750\. _𑀅𑀣𑀼𑀝𑁆𑀞𑀳𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄,_  
_𑀬𑀸𑀘𑀫𑀸𑀦𑁄 𑀓𑀢𑀜𑁆𑀚𑀮𑀻;_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁄 𑀓𑀺𑀁 𑀧𑀭𑀚𑁆𑀛𑀢𑀺,_  
_𑀯𑀸𑀡𑀺𑀚𑀸 𑀪𑀤𑁆𑀤𑀫𑀢𑁆𑀣𑀼 𑀢𑁂𑁇_  


(195.)

2751\. _𑀯𑀸𑀭𑀺𑀤𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀸𑀔𑀸,_  
_𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀜𑁆𑀘 𑀤𑀓𑁆𑀔𑀺𑀡𑀸;_  
_𑀦𑀸𑀭𑀺𑀤𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀲𑀸𑀔𑀸,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑁂 𑀘 𑀉𑀢𑁆𑀢𑀭𑀸;_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁄 𑀓𑀺𑀁 𑀧𑀭𑀚𑁆𑀛𑀢𑀺,_  
_𑀯𑀸𑀡𑀺𑀚𑀸 𑀪𑀤𑁆𑀤𑀫𑀢𑁆𑀣𑀼 𑀢𑁂𑁇_  


(196.)

2752\. _𑀬𑀲𑁆𑀲 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀙𑀸𑀬𑀸𑀬,_  
_𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀲𑀬𑁂𑀬𑁆𑀬 𑀯𑀸;_  
_𑀦 𑀢𑀲𑁆𑀲 𑀲𑀸𑀔𑀁 𑀪𑀜𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀫𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀪𑁄 𑀳𑀺 𑀧𑀸𑀧𑀓𑁄𑁇_  


(197.)

2753\. _𑀢𑁂 𑀘 𑀢𑀲𑁆𑀲𑀸𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀏𑀓𑀲𑁆𑀲 𑀯𑀘𑀦𑀁 𑀩𑀳𑀽;_  
_𑀦𑀺𑀲𑀺𑀢𑀸𑀳𑀺 𑀓𑀼𑀞𑀸𑀭𑀻𑀳𑀺,_  
_𑀫𑀽𑀮𑀢𑁄 𑀦𑀁 𑀉𑀧𑀓𑁆𑀓𑀫𑀼𑀁𑁇_  


(198.)

2754\. _𑀢𑀢𑁄 𑀦𑀸𑀕𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀁𑀲𑀼,_  
_𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀸 𑀧𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺;_  
_𑀥𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀦𑀁 𑀢𑀺𑀲𑀢𑀸,_  
_𑀙𑀲𑀳𑀲𑁆𑀲𑀸 𑀘 𑀯𑀫𑁆𑀫𑀺𑀦𑁄𑁇_  


(199.)

2755\. _“𑀏𑀢𑁂 𑀳𑀦𑀣 𑀩𑀦𑁆𑀥𑀣,_  
_𑀫𑀸 𑀯𑁄 𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀣 𑀚𑀻𑀯𑀺𑀢𑀁;_  
_𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀁𑀯,_  
_𑀲𑀩𑁆𑀩𑁂 𑀪𑀲𑁆𑀫𑀁 𑀓𑀭𑁄𑀣 𑀦𑁂”𑁇_  


(200.)

2756\. _“𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁 𑀅𑀢𑁆𑀣𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀮𑁄𑀪𑀲𑁆𑀲 𑀦 𑀯𑀲𑀁 𑀕𑀘𑁆𑀙𑁂,_  
_𑀳𑀦𑁂𑀬𑁆𑀬𑀸𑀭𑀺𑀲𑀓𑀁 𑀫𑀦𑀁𑁇_  


(201.)

2757\. _𑀏𑀯𑀫𑀸𑀤𑀻𑀦𑀯𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀢𑀡𑁆𑀳𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀪𑀯𑀁;_  
_𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑁄 𑀅𑀦𑀸𑀤𑀸𑀦𑁄,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂”𑀢𑀺𑁇_  


---

2758\. 𑀫𑀳𑀸𑀯𑀸𑀡𑀺𑀚𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇
