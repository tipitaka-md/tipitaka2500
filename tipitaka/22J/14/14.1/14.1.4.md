# 𑁧𑁪𑁇𑁧𑁇𑁪 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀚𑀸𑀢𑀓

* 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀸𑀮𑀺𑀓𑁂𑀤𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀚𑀸𑀢𑀓

(62.)

2612\. _“𑀔𑀭𑀸𑀚𑀺𑀦𑀸 𑀚𑀝𑀺𑀮𑀸 𑀧𑀗𑁆𑀓𑀤𑀦𑁆𑀢𑀸,_  
_𑀤𑀼𑀫𑁆𑀫𑀓𑁆𑀔𑀭𑀽𑀧𑀸 𑀬𑁂 𑀫𑀦𑁆𑀢𑀁 𑀚𑀧𑁆𑀧𑀦𑁆𑀢𑀺;_  
_𑀓𑀘𑁆𑀘𑀺𑀦𑁆𑀦𑀼 𑀢𑁂 𑀫𑀸𑀦𑀼𑀲𑀓𑁂 𑀧𑀬𑁄𑀕𑁂,_  
_𑀇𑀤𑀁 𑀯𑀺𑀤𑀽 𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑀸 𑀅𑀧𑀸𑀬𑀸”𑁇_  


(63.)

2613\. _“𑀧𑀸𑀧𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁂𑀣 𑀭𑀸𑀚,_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀘𑁂 𑀦 𑀘𑀭𑁂𑀬𑁆𑀬 𑀥𑀫𑁆𑀫𑀁;_  
_𑀲𑀳𑀲𑁆𑀲𑀯𑁂𑀤𑁄𑀧𑀺 𑀦 𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘,_  
_𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑁂 𑀘𑀭𑀡𑀁 𑀅𑀧𑀢𑁆𑀯𑀸”𑁇_  


(64.)

2614\. _“𑀲𑀳𑀲𑁆𑀲𑀯𑁂𑀤𑁄𑀧𑀺 𑀦 𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘,_  
_𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑁂 𑀘𑀭𑀡𑀁 𑀅𑀧𑀢𑁆𑀯𑀸;_  
_𑀫𑀜𑁆𑀜𑀸𑀫𑀺 𑀯𑁂𑀤𑀸 𑀅𑀨𑀮𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀲𑀲𑀁𑀬𑀫𑀁 𑀘𑀭𑀡𑀜𑁆𑀜𑁂𑀯 𑀲𑀘𑁆𑀘𑀁”𑁇_  


(65.)

2615\. _“𑀦 𑀳𑁂𑀯 𑀯𑁂𑀤𑀸 𑀅𑀨𑀮𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀲𑀲𑀁𑀬𑀫𑀁 𑀘𑀭𑀡𑀜𑁆𑀜𑁂𑀯 𑀲𑀘𑁆𑀘𑀁;_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀳𑀺 𑀧𑀧𑁆𑀧𑁄𑀢𑀺 𑀅𑀥𑀺𑀘𑁆𑀘 𑀯𑁂𑀤𑁂,_  
_𑀲𑀦𑁆𑀢𑀺𑀁 𑀧𑀼𑀡𑀸𑀢𑀺 𑀘𑀭𑀡𑁂𑀦 𑀤𑀦𑁆𑀢𑁄”𑁇_  


(66.)

2616\. _“𑀪𑀘𑁆𑀘𑀸 𑀫𑀸𑀢𑀸 𑀧𑀺𑀢𑀸 𑀩𑀦𑁆𑀥𑀽,_  
_𑀬𑁂𑀦 𑀚𑀸𑀢𑁄 𑀲𑀬𑁂𑀯 𑀲𑁄;_  
_𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑁄 𑀅𑀳𑀁 𑀪𑁄𑀢𑁄,_  
_𑀲𑁄𑀢𑁆𑀢𑀺𑀬𑀸𑀓𑀼𑀮𑀯𑀁𑀲𑀓𑁄”𑁇_  


(67.)

2617\. _“𑀓𑀣𑀁 𑀪𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀳𑁄𑀢𑀺,_  
_𑀓𑀣𑀁 𑀪𑀯𑀢𑀺 𑀓𑁂𑀯𑀮𑀻;_  
_𑀓𑀣𑀜𑁆𑀘 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄 𑀓𑀺𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(68.)

2618\. _“𑀦𑀺𑀭𑀁𑀓𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀫𑀸𑀤𑀸𑀬 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄,_  
_𑀆𑀧𑁄 𑀲𑀺𑀜𑁆𑀘𑀁 𑀬𑀚𑀁 𑀉𑀲𑁆𑀲𑁂𑀢𑀺 𑀬𑀽𑀧𑀁;_  
_𑀏𑀯𑀁𑀓𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀔𑁂𑀫𑀻,_  
_𑀥𑀫𑁆𑀫𑁂 𑀞𑀺𑀢𑀁 𑀢𑁂𑀦 𑀅𑀫𑀸𑀧𑀬𑀺𑀁𑀲𑀼”𑁇_  


(69.)

2619\. _“𑀦 𑀲𑀼𑀤𑁆𑀥𑀺 𑀲𑁂𑀘𑀦𑁂𑀦𑀢𑁆𑀣𑀺,_  
_𑀦𑀸𑀧𑀺 𑀓𑁂𑀯𑀮𑀻 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀦 𑀔𑀦𑁆𑀢𑀻 𑀦𑀸𑀧𑀺 𑀲𑁄𑀭𑀘𑁆𑀘𑀁,_  
_𑀦𑀸𑀧𑀺 𑀲𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑁇_  


(70.)

2620\. _“𑀓𑀣𑀁 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀳𑁄𑀢𑀺,_  
_𑀓𑀣𑀁 𑀪𑀯𑀢𑀺 𑀓𑁂𑀯𑀮𑀻;_  
_𑀓𑀣𑀜𑁆𑀘 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄 𑀓𑀺𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(71.)

2621\. _“𑀅𑀔𑁂𑀢𑁆𑀢𑀩𑀦𑁆𑀥𑀽 𑀅𑀫𑀫𑁄 𑀦𑀺𑀭𑀸𑀲𑁄,_  
_𑀦𑀺𑀮𑁆𑀮𑁄𑀪𑀧𑀸𑀧𑁄 𑀪𑀯𑀮𑁄𑀪𑀔𑀻𑀡𑁄;_  
_𑀏𑀯𑀁𑀓𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀔𑁂𑀫𑀻,_  
_𑀥𑀫𑁆𑀫𑁂 𑀞𑀺𑀢𑀁 𑀢𑁂𑀦 𑀅𑀫𑀸𑀧𑀬𑀺𑀁𑀲𑀼”𑁇_  


(72.)

2622\. _“𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑁂𑀲𑁆𑀲𑀸,_  
_𑀲𑀼𑀤𑁆𑀤𑀸 𑀘𑀡𑁆𑀟𑀸𑀮𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀭𑀢𑀸 𑀤𑀦𑁆𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀸𑀦𑀁,_  
_𑀅𑀢𑁆𑀣𑀺 𑀲𑁂𑀬𑁆𑀬𑁄𑀣 𑀧𑀸𑀧𑀺𑀬𑁄”𑁇_  


(73.)

2623\. _“𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑁂𑀲𑁆𑀲𑀸,_  
_𑀲𑀼𑀤𑁆𑀤𑀸 𑀘𑀡𑁆𑀟𑀸𑀮𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀭𑀢𑀸 𑀤𑀦𑁆𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀸𑀦𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀲𑁂𑀬𑁆𑀬𑁄𑀣 𑀧𑀸𑀧𑀺𑀬𑁄”𑁇_  


(74.)

2624\. _“𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀯𑁂𑀲𑁆𑀲𑀸,_  
_𑀲𑀼𑀤𑁆𑀤𑀸 𑀘𑀡𑁆𑀟𑀸𑀮𑀧𑀼𑀓𑁆𑀓𑀼𑀲𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀭𑀢𑀸 𑀤𑀦𑁆𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸𑁇_  


(75.)

2625\. _𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀸𑀦𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀲𑁂𑀬𑁆𑀬𑁄𑀣 𑀧𑀸𑀧𑀺𑀬𑁄;_  
_𑀧𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀲𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁,_  
_𑀲𑁄𑀢𑁆𑀢𑀺𑀬𑀸𑀓𑀼𑀮𑀯𑀁𑀲𑀢𑀁”𑁇_  


(76.)

2626\. _“𑀦𑀸𑀦𑀸𑀭𑀢𑁆𑀢𑁂𑀳𑀺 𑀯𑀢𑁆𑀣𑁂𑀳𑀺,_  
_𑀯𑀺𑀫𑀸𑀦𑀁 𑀪𑀯𑀢𑀺 𑀙𑀸𑀤𑀺𑀢𑀁;_  
_𑀦 𑀢𑁂𑀲𑀁 𑀙𑀸𑀬𑀸 𑀯𑀢𑁆𑀣𑀸𑀦𑀁,_  
_𑀲𑁄 𑀭𑀸𑀕𑁄 𑀅𑀦𑀼𑀧𑀚𑁆𑀚𑀣𑁇_  


(77.)

2627\. _𑀏𑀯𑀫𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀬𑀤𑀸 𑀲𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀫𑀸𑀡𑀯𑀸;_  
_𑀢𑁂 𑀲𑀚𑀸𑀢𑀺𑀁 𑀧𑀫𑀼𑀜𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀩𑁆𑀩𑀢𑀸”𑀢𑀺𑁇_  


---

2628\. 𑀉𑀤𑁆𑀤𑀸𑀮𑀓𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
