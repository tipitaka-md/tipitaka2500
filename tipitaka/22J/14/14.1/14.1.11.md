# 𑁧𑁪𑁇𑁧𑁇𑁧𑁧 𑀲𑀸𑀥𑀺𑀦𑀚𑀸𑀢𑀓

### 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀸𑀮𑀺𑀓𑁂𑀤𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀥𑀺𑀦𑀚𑀸𑀢𑀓

(202.)

2759. _“𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀯𑀢 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑁄;_  
_𑀤𑀺𑀩𑁆𑀩𑁄 𑀭𑀣𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑀼,_  
_𑀯𑁂𑀤𑁂𑀳𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄”𑁇_  


(203.)

2760. _𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄,_  
_𑀫𑀸𑀢𑀮𑀺 𑀤𑁂𑀯𑀲𑀸𑀭𑀣𑀺;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀣 𑀭𑀸𑀚𑀸𑀦𑀁,_  
_𑀯𑁂𑀤𑁂𑀳𑀁 𑀫𑀺𑀣𑀺𑀮𑀕𑁆𑀕𑀳𑀁𑁇_  


(204.)

2761. _“𑀏𑀳𑀺𑀫𑀁 𑀭𑀣𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀭𑀸𑀚𑀲𑁂𑀝𑁆𑀞 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀤𑁂𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑀸 𑀢𑁂,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑁃𑀦𑁆𑀤𑀓𑀸;_  
_𑀲𑀭𑀫𑀸𑀦𑀸 𑀳𑀺 𑀢𑁂 𑀤𑁂𑀯𑀸,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀲𑀫𑀘𑁆𑀙𑀭𑁂”𑁇_  


(205.)

2762. _𑀢𑀢𑁄 𑀘 𑀭𑀸𑀚𑀸 𑀲𑀸𑀥𑀺𑀦𑁄,_  
_𑀯𑁂𑀤𑁂𑀳𑁄 𑀫𑀺𑀣𑀺𑀮𑀕𑁆𑀕𑀳𑁄;_  
_𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀅𑀕𑀸 𑀤𑁂𑀯𑀸𑀦 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀁 𑀤𑁂𑀯𑀸 𑀧𑀝𑀺𑀦𑀦𑁆𑀤𑀺𑀁𑀲𑀼,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀫𑀸𑀕𑀢𑀁𑁇_  


(206.)

2763. _“𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀅𑀣𑁄 𑀢𑁂 𑀅𑀤𑀼𑀭𑀸𑀕𑀢𑀁;_  
_𑀦𑀺𑀲𑀻𑀤 𑀤𑀸𑀦𑀺 𑀭𑀸𑀚𑀻𑀲𑀺,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑁇_  


(207.)

2764. _𑀲𑀓𑁆𑀓𑁄𑀧𑀺 𑀧𑀝𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀣,_  
_𑀯𑁂𑀤𑁂𑀳𑀁 𑀫𑀺𑀣𑀺𑀮𑀕𑁆𑀕𑀳𑀁;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀣 𑀓𑀸𑀫𑁂𑀳𑀺,_  
_𑀆𑀲𑀦𑁂𑀦 𑀘 𑀯𑀸𑀲𑀯𑁄𑁇_  


(208.)

2765. _“𑀲𑀸𑀥𑀼 𑀔𑁄𑀲𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀆𑀯𑀸𑀲𑀁 𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀦𑀁;_  
_𑀯𑀲 𑀤𑁂𑀯𑁂𑀲𑀼 𑀭𑀸𑀚𑀻𑀲𑀺,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀲𑀼;_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀪𑀼𑀜𑁆𑀚 𑀓𑀸𑀫𑁂 𑀅𑀫𑀸𑀦𑀼𑀲𑁂”𑁇_  


(209.)

2766. _“𑀅𑀳𑀁 𑀧𑀼𑀭𑁂 𑀲𑀕𑁆𑀕𑀕𑀢𑁄 𑀭𑀫𑀸𑀫𑀺,_  
_𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀘 𑀯𑀸𑀤𑀺𑀢𑁂𑀳𑀺;_  
_𑀲𑁄 𑀤𑀸𑀦𑀺 𑀅𑀚𑁆𑀚 𑀦 𑀭𑀫𑀸𑀫𑀺 𑀲𑀕𑁆𑀕𑁂,_  
_𑀆𑀬𑀼𑀁 𑀦𑀼 𑀔𑀻𑀡𑁄 𑀫𑀭𑀡𑀁 𑀦𑀼 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀉𑀤𑀸𑀳𑀼 𑀫𑀽𑀍𑀳𑁄𑀲𑁆𑀫𑀺 𑀚𑀦𑀺𑀦𑁆𑀤𑀲𑁂𑀝𑁆𑀞”𑁇_  


(210.)

2767. _“𑀦 𑀢𑀸𑀬𑀼 𑀔𑀻𑀡𑀁 𑀫𑀭𑀡𑀜𑁆𑀘 𑀤𑀽𑀭𑁂,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀫𑀽𑀍𑀳𑁄 𑀦𑀭𑀯𑀻𑀭𑀲𑁂𑀝𑁆𑀞;_  
_𑀢𑀼𑀬𑁆𑀳𑀜𑁆𑀘 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀺,_  
_𑀬𑁂𑀲𑀁 𑀯𑀺𑀧𑀸𑀓𑀁 𑀇𑀥 𑀯𑁂𑀤𑀬𑀺𑀢𑁆𑀣𑁄𑁇_  


(211.)

2768. _𑀯𑀲 𑀤𑁂𑀯𑀸𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀭𑀸𑀚𑀲𑁂𑀝𑁆𑀞 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀪𑀼𑀜𑁆𑀚 𑀓𑀸𑀫𑁂 𑀅𑀫𑀸𑀦𑀼𑀲𑁂”𑁇_  


(212.)

2769. _“𑀬𑀣𑀸 𑀬𑀸𑀘𑀺𑀢𑀓𑀁 𑀬𑀸𑀦𑀁,_  
_𑀬𑀣𑀸 𑀬𑀸𑀘𑀺𑀢𑀓𑀁 𑀥𑀦𑀁;_  
_𑀏𑀯𑀁𑀲𑀫𑁆𑀧𑀤𑀫𑁂𑀯𑁂𑀢𑀁,_  
_𑀬𑀁 𑀧𑀭𑀢𑁄 𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


(213.)

2770. _𑀦 𑀘𑀸𑀳𑀫𑁂𑀢𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀬𑀁 𑀧𑀭𑀢𑁄 𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸;_  
_𑀲𑀬𑀁𑀓𑀢𑀸𑀦𑀺 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀆𑀯𑁂𑀡𑀺𑀓𑀁 𑀥𑀦𑀁𑁇_  


(214.)

2771. _𑀲𑁄𑀳𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀓𑀸𑀳𑀸𑀫𑀺 𑀓𑀼𑀲𑀮𑀁 𑀩𑀳𑀼𑀁;_  
_𑀤𑀸𑀦𑁂𑀦 𑀲𑀫𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀲𑀁𑀬𑀫𑁂𑀦 𑀤𑀫𑁂𑀦 𑀘;_  
_𑀬𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀦 𑀘 𑀧𑀘𑁆𑀙𑀸𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀺𑁇_  


(215.)

2772. _𑀇𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀔𑁂𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀇𑀫𑀁 𑀦𑀺𑀓𑁆𑀔𑀁 𑀲𑀼𑀓𑀼𑀡𑁆𑀟𑀮𑀁;_  
_𑀇𑀫𑀸 𑀢𑀸 𑀳𑀭𑀺𑀢𑀸𑀦𑀽𑀧𑀸,_  
_𑀇𑀫𑀸 𑀦𑀚𑁆𑀚𑁄 𑀲𑀯𑀦𑁆𑀢𑀺𑀬𑁄𑁇_  


(216.)

2773. _𑀇𑀫𑀸 𑀢𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀭𑀫𑁆𑀫𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀧𑀓𑀽𑀚𑀺𑀢𑀸;_  
_𑀫𑀦𑁆𑀤𑀸𑀮𑀓𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀓𑁂𑀳𑀺 𑀘;_  
_𑀬𑀲𑁆𑀲𑀺𑀫𑀸𑀦𑀺 𑀫𑀫𑀸𑀬𑀺𑀁𑀲𑀼,_  
_𑀓𑀺𑀁 𑀦𑀼 𑀢𑁂 𑀤𑀺𑀲𑀢𑀁 𑀕𑀢𑀸𑁇_  


(217.)

2774. _𑀢𑀸𑀦𑀻𑀥 𑀔𑁂𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑁄 𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑁄,_  
_𑀢𑁂𑀬𑁂𑀯 𑀆𑀭𑀸𑀫𑀯𑀦𑀽𑀧𑀘𑀸𑀭𑀸;_  
_𑀢𑀫𑁂𑀯 𑀫𑀬𑁆𑀳𑀁 𑀚𑀦𑀢𑀁 𑀅𑀧𑀲𑁆𑀲𑀢𑁄,_  
_𑀲𑀼𑀜𑁆𑀜𑀁𑀯 𑀫𑁂 𑀦𑀸𑀭𑀤 𑀔𑀸𑀬𑀢𑁂 𑀤𑀺𑀲𑀸”𑁇_  


(218.)

2775. _“𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸 𑀯𑀺𑀫𑀸𑀦𑀸𑀦𑀺,_  
_𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀸 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸;_  
_𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀤𑁂𑀯𑀭𑀸𑀚𑀲𑁆𑀲,_  
_𑀢𑀺𑀤𑀲𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑁇_  


(219.)

2776. _𑀯𑀼𑀢𑁆𑀣𑀁 𑀫𑁂 𑀪𑀯𑀦𑀁 𑀤𑀺𑀩𑁆𑀬𑀁,_  
_𑀪𑀼𑀢𑁆𑀢𑀸 𑀓𑀸𑀫𑀸 𑀅𑀫𑀸𑀦𑀼𑀲𑀸;_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀲𑀼𑁇_  


(220.)

2777. _𑀲𑁄𑀳𑀁 𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀬𑀫𑁆𑀳𑀺 𑀇𑀥𑀸𑀕𑀢𑁄;_  
_𑀥𑀫𑁆𑀫𑀫𑁂𑀯 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀦𑀸𑀳𑀁 𑀭𑀚𑁆𑀚𑁂𑀦 𑀅𑀢𑁆𑀣𑀺𑀓𑁄𑁇_  


(221.)

2778. _𑀅𑀤𑀡𑁆𑀟𑀸𑀯𑀘𑀭𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀢𑀁 𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀁,_  
_𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀼𑀩𑁆𑀩𑀢𑀸”𑀢𑀺𑁇_  


---

2779. 𑀲𑀸𑀥𑀺𑀦𑀚𑀸𑀢𑀓𑀁 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇
