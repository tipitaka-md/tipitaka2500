# 𑁧𑁪𑁇𑁧𑁇𑁯 𑀢𑀘𑁆𑀙𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓

### 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀸𑀮𑀺𑀓𑁂𑀤𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀢𑀘𑁆𑀙𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓

(160.)

2715\. _“𑀬𑀤𑁂𑀲𑀫𑀸𑀦𑀸 𑀯𑀺𑀘𑀭𑀺𑀫𑁆𑀳,_  
_𑀧𑀩𑁆𑀩𑀢𑀸𑀦𑀺 𑀯𑀦𑀸𑀦𑀺 𑀘;_  
_𑀅𑀦𑁆𑀯𑁂𑀲𑀁 𑀯𑀺𑀘𑀭𑀺𑀁 𑀜𑀸𑀢𑀻,_  
_𑀢𑁂𑀫𑁂 𑀅𑀥𑀺𑀕𑀢𑀸 𑀫𑀬𑀸𑁇_  


(161.)

2716\. _𑀩𑀳𑀼𑀜𑁆𑀘𑀺𑀤𑀁 𑀫𑀽𑀮𑀨𑀮𑀁,_  
_𑀪𑀓𑁆𑀔𑁄 𑀘𑀸𑀬𑀁 𑀅𑀦𑀧𑁆𑀧𑀓𑁄;_  
_𑀭𑀫𑁆𑀫𑀸 𑀘𑀺𑀫𑀸 𑀕𑀺𑀭𑀻𑀦𑀚𑁆𑀚𑁄,_  
_𑀨𑀸𑀲𑀼𑀯𑀸𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(162.)

2717\. _𑀇𑀥𑁂𑀯𑀸𑀳𑀁 𑀯𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀳 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀜𑀸𑀢𑀺𑀪𑀺;_  
_𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀦𑀺𑀭𑀸𑀲𑀗𑁆𑀓𑀻,_  
_𑀅𑀲𑁄𑀓𑁄 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑁄”𑁇_  


(163.)

2718\. _“𑀅𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀮𑁂𑀡𑀁 𑀧𑀭𑀺𑀬𑁂𑀲,_  
_𑀲𑀢𑁆𑀢𑀼 𑀦𑁄 𑀇𑀥 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑁄 𑀢𑀘𑁆𑀙 𑀲𑀽𑀓𑀭𑁂 𑀳𑀦𑁆𑀢𑀺,_  
_𑀇𑀥𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀭𑀁 𑀯𑀭𑀁”𑁇_  


(164.)

2719\. _“𑀓𑁄 𑀦𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀇𑀥 𑀲𑀢𑁆𑀢𑀼,_  
_𑀓𑁄 𑀜𑀸𑀢𑀻 𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑁂;_  
_𑀤𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑁂 𑀧𑀥𑀁𑀲𑁂𑀢𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀣 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸”𑁇_  


(165.)

2720\. _“𑀉𑀤𑁆𑀥𑀕𑁆𑀕𑀭𑀸𑀚𑀻 𑀫𑀺𑀕𑀭𑀸𑀚𑀸,_  
_𑀩𑀮𑀻 𑀤𑀸𑀞𑀸𑀯𑀼𑀥𑁄 𑀫𑀺𑀕𑁄;_  
_𑀲𑁄 𑀢𑀘𑁆𑀙 𑀲𑀽𑀓𑀭𑁂 𑀳𑀦𑁆𑀢𑀺,_  
_𑀇𑀥𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀭𑀁 𑀯𑀭𑀁”𑁇_  


(166.)

2721\. _“𑀦 𑀦𑁄 𑀤𑀸𑀞𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀩𑀮𑀁 𑀓𑀸𑀬𑁂 𑀲𑀫𑁄𑀳𑀺𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀕𑁆𑀕𑀸 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀯𑀲𑀁 𑀓𑀸𑀳𑀸𑀫 𑀏𑀓𑀓𑀁”𑁇_  


(167.)

2722\. _“𑀳𑀤𑀬𑀗𑁆𑀕𑀫𑀁 𑀓𑀡𑁆𑀡𑀲𑀼𑀔𑀁,_  
_𑀯𑀸𑀘𑀁 𑀪𑀸𑀲𑀲𑀺 𑀢𑀘𑁆𑀙𑀓;_  
_𑀬𑁄𑀧𑀺 𑀬𑀼𑀤𑁆𑀥𑁂 𑀧𑀮𑀸𑀬𑁂𑀬𑁆𑀬,_  
_𑀢𑀫𑁆𑀧𑀺 𑀧𑀘𑁆𑀙𑀸 𑀳𑀦𑀸𑀫𑀲𑁂”𑁇_  


(168.)

2723\. _“𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀢𑁄 𑀦𑀼 𑀅𑀚𑁆𑀚,_  
_𑀅𑀪𑀬𑀁 𑀦𑀼 𑀢𑁂 𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑁂𑀲𑀼 𑀤𑀺𑀦𑁆𑀦𑀁;_  
_𑀤𑀸𑀞𑀸 𑀦𑀼 𑀢𑁂 𑀫𑀺𑀕𑀯𑀥𑀸𑀬 𑀦 𑀲𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀲𑀁𑀖𑀧𑀢𑁆𑀢𑁄 𑀓𑀧𑀡𑁄𑀯 𑀛𑀸𑀬𑀲𑀺”𑁇_  


(169.)

2724\. _“𑀦 𑀫𑁂 𑀤𑀸𑀞𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀩𑀮𑀁 𑀓𑀸𑀬𑁂 𑀲𑀫𑁄𑀳𑀺𑀢𑀁;_  
_𑀜𑀸𑀢𑀻 𑀘 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀏𑀓𑀢𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀘 𑀛𑀸𑀬𑀸𑀫𑀺 𑀯𑀦𑀫𑁆𑀳𑀺 𑀏𑀓𑀓𑁄𑁇_  


(170.)

2725\. _𑀇𑀫𑀲𑁆𑀲𑀼𑀤𑀁 𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁 𑀧𑀼𑀭𑁂,_  
_𑀪𑀬𑀝𑁆𑀝𑀺𑀢𑀸 𑀮𑁂𑀡𑀕𑀯𑁂𑀲𑀺𑀦𑁄 𑀧𑀼𑀣𑀼;_  
_𑀢𑁂 𑀤𑀸𑀦𑀺 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀯𑀲𑀦𑁆𑀢𑀺 𑀏𑀓𑀢𑁄,_  
_𑀬𑀢𑁆𑀣𑀝𑁆𑀞𑀺𑀢𑀸 𑀤𑀼𑀧𑁆𑀧𑀲𑀳𑀚𑁆𑀚 𑀢𑁂 𑀫𑀬𑀸𑁇_  


(171.)

2726\. _𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀳𑀺𑀢𑀸 𑀏𑀓𑀯𑀸𑀤𑀺𑀦𑁄;_  
_𑀢𑁂 𑀫𑀁 𑀲𑀫𑀕𑁆𑀕𑀸 𑀳𑀺𑀁𑀲𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀦𑁂𑀲𑀁 𑀦 𑀧𑀢𑁆𑀣𑀬𑁂”𑁇_  


(172.)

2727\. _“𑀏𑀓𑁄𑀯 𑀇𑀦𑁆𑀤𑁄 𑀅𑀲𑀼𑀭𑁂 𑀚𑀺𑀦𑀸𑀢𑀺,_  
_𑀏𑀓𑁄𑀯 𑀲𑁂𑀦𑁄 𑀳𑀦𑁆𑀢𑀺 𑀤𑀺𑀚𑁂 𑀧𑀲𑀬𑁆𑀳;_  
_𑀏𑀓𑁄𑀯 𑀩𑁆𑀬𑀕𑁆𑀖𑁄 𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑀧𑀢𑁆𑀢𑁄,_  
_𑀯𑀭𑀁 𑀯𑀭𑀁 𑀳𑀦𑁆𑀢𑀺 𑀩𑀮𑀜𑁆𑀳𑀺 𑀢𑀸𑀤𑀺𑀲𑀁”𑁇_  


(173.)

2728\. _“𑀦 𑀳𑁂𑀯 𑀇𑀦𑁆𑀤𑁄 𑀦 𑀲𑁂𑀦𑁄,_  
_𑀦𑀧𑀺 𑀩𑁆𑀬𑀕𑁆𑀖𑁄 𑀫𑀺𑀕𑀸𑀥𑀺𑀧𑁄;_  
_𑀲𑀫𑀕𑁆𑀕𑁂 𑀲𑀳𑀺𑀢𑁂 𑀜𑀸𑀢𑀻,_  
_𑀦 𑀩𑁆𑀬𑀕𑁆𑀖𑁂 𑀓𑀼𑀭𑀼𑀢𑁂 𑀯𑀲𑁂”𑁇_  


(174.)

2729\. _“𑀓𑀼𑀫𑁆𑀪𑀻𑀮𑀓𑀸 𑀲𑀓𑀼𑀡𑀓𑀸,_  
_𑀲𑀁𑀖𑀺𑀦𑁄 𑀕𑀡𑀘𑀸𑀭𑀺𑀦𑁄;_  
_𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀏𑀓𑀚𑁆𑀛𑀁,_  
_𑀉𑀧𑁆𑀧𑀢𑀦𑁆𑀢𑀺 𑀟𑀬𑀦𑁆𑀢𑀺 𑀘𑁇_  


(175.)

2730\. _𑀢𑁂𑀲𑀜𑁆𑀘 𑀟𑀬𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀏𑀓𑁂𑀢𑁆𑀣 𑀅𑀧𑀲𑀓𑁆𑀓𑀢𑀺;_  
_𑀢𑀜𑁆𑀘 𑀲𑁂𑀦𑁄 𑀦𑀺𑀢𑀸𑀍𑀏𑀢𑀺,_  
_𑀯𑁂𑀬𑁆𑀬𑀕𑁆𑀖𑀺𑀬𑁂𑀯 𑀲𑀸 𑀕𑀢𑀺”𑁇_  


(176.)

2731\. _“𑀉𑀲𑁆𑀲𑀸𑀳𑀺𑀢𑁄 𑀚𑀝𑀺𑀮𑁂𑀦,_  
_𑀮𑀼𑀤𑁆𑀤𑁂𑀦𑀸𑀫𑀺𑀲𑀘𑀓𑁆𑀔𑀼𑀦𑀸;_  
_𑀤𑀸𑀞𑀻 𑀤𑀸𑀞𑀻𑀲𑀼 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀺,_  
_𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂”𑁇_  


(177.)

2732\. _“𑀲𑀸𑀥𑀼 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀜𑀸𑀢𑀻,_  
_𑀅𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀸 𑀅𑀭𑀜𑁆𑀜𑀚𑀸;_  
_𑀲𑀽𑀓𑀭𑁂𑀳𑀺 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺,_  
_𑀩𑁆𑀬𑀕𑁆𑀖𑁄 𑀏𑀓𑀸𑀬𑀦𑁂 𑀳𑀢𑁄”𑁇_  


(178.)

2733\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀜𑁆𑀘𑁂𑀯 𑀩𑁆𑀬𑀕𑁆𑀖𑀜𑁆𑀘,_  
_𑀉𑀪𑁄 𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀽𑀓𑀭𑀸𑁇_  
_𑀆𑀦𑀦𑁆𑀤𑀺𑀦𑁄 𑀧𑀫𑀼𑀤𑀺𑀢𑀸,_  
_𑀫𑀳𑀸𑀦𑀸𑀤𑀁 𑀧𑀦𑀸𑀤𑀺𑀲𑀼𑀁𑁇_  


(179.)

2734\. _𑀢𑁂 𑀲𑀼 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀫𑀽𑀮𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀽𑀓𑀭𑀸 𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀢𑀘𑁆𑀙𑀓𑀁 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑀺𑀁𑀲𑀼,_  
_“𑀢𑁆𑀯𑀁 𑀦𑁄 𑀭𑀸𑀚𑀸𑀲𑀺 𑀇𑀲𑁆𑀲𑀭𑁄”𑀢𑀺𑁇_  


---

2735\. 𑀢𑀘𑁆𑀙𑀲𑀽𑀓𑀭𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇
