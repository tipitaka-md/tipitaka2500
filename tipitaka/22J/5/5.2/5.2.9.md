# 𑁫𑁇𑁨𑁇𑁯 𑀫𑀺𑀢𑁆𑀢𑀯𑀺𑀦𑁆𑀤𑀓𑀚𑀸𑀢𑀓

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀯𑀡𑁆𑀡𑀸𑀭𑁄𑀳𑀯𑀕𑁆𑀕

### 𑀫𑀺𑀢𑁆𑀢𑀯𑀺𑀦𑁆𑀤𑀓𑀚𑀸𑀢𑀓

(100.)

1265. _“𑀓𑁆𑀬𑀸𑀳𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀓𑀭𑀁,_  
_𑀓𑀺𑀁 𑀧𑀸𑀧𑀁 𑀧𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀬𑀁 𑀫𑁂 𑀲𑀺𑀭𑀲𑁆𑀫𑀺𑀁 𑀑𑀳𑀘𑁆𑀘,_  
_𑀘𑀓𑁆𑀓𑀁 𑀪𑀫𑀢𑀺 𑀫𑀢𑁆𑀣𑀓𑁂”𑁇_  


(101.)

1266. _“𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀭𑀫𑀡𑀓𑀁,_  
_𑀲𑀤𑀸𑀫𑀢𑁆𑀢𑀜𑁆𑀘 𑀤𑀽𑀪𑀓𑀁;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀢𑁆𑀢𑀭𑀜𑁆𑀘 𑀧𑀸𑀲𑀸𑀤𑀁,_  
_𑀓𑁂𑀦𑀢𑁆𑀣𑁂𑀦 𑀇𑀥𑀸𑀕𑀢𑁄”𑁇_  


(102.)

1267. _“𑀇𑀢𑁄 𑀩𑀳𑀼𑀢𑀭𑀸 𑀪𑁄𑀕𑀸,_  
_𑀅𑀢𑁆𑀭 𑀫𑀜𑁆𑀜𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀭𑁂;_  
_𑀇𑀢𑀺 𑀏𑀢𑀸𑀬 𑀲𑀜𑁆𑀜𑀸𑀬,_  
_𑀧𑀲𑁆𑀲 𑀫𑀁 𑀩𑁆𑀬𑀲𑀦𑀁 𑀕𑀢𑀁”𑁇_  


(103.)

1268. _“𑀘𑀢𑀼𑀩𑁆𑀪𑀺 𑀅𑀝𑁆𑀞𑀚𑁆𑀛𑀕𑀫𑀸,_  
_𑀅𑀝𑁆𑀞𑀸𑀳𑀺𑀧𑀺 𑀘 𑀲𑁄𑀍𑀅𑀲;_  
_𑀲𑁄𑀍𑀅𑀲𑀸𑀳𑀺 𑀘 𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲,_  
_𑀅𑀢𑁆𑀭𑀺𑀘𑁆𑀙𑀁 𑀘𑀓𑁆𑀓𑀫𑀸𑀲𑀤𑁄;_  
_𑀇𑀘𑁆𑀙𑀸𑀳𑀢𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲,_  
_𑀘𑀓𑁆𑀓𑀁 𑀪𑀫𑀢𑀺 𑀫𑀢𑁆𑀣𑀓𑁂𑁇_  


(104.)

1269. _𑀉𑀧𑀭𑀺𑀯𑀺𑀲𑀸𑀮𑀸 𑀤𑀼𑀧𑁆𑀧𑀽𑀭𑀸,_  
_𑀇𑀘𑁆𑀙𑀸 𑀯𑀺𑀲𑀝𑀕𑀸𑀫𑀺𑀦𑀻;_  
_𑀬𑁂 𑀘 𑀢𑀁 𑀅𑀦𑀼𑀕𑀺𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀘𑀓𑁆𑀓𑀥𑀸𑀭𑀺𑀦𑁄”𑀢𑀺𑁇_  


---

1270. 𑀫𑀺𑀢𑁆𑀢𑀯𑀺𑀦𑁆𑀤𑀓𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇
