# 𑁫𑁇𑁨𑁇𑁮 𑀢𑀘𑀲𑀸𑀭𑀚𑀸𑀢𑀓

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀯𑀡𑁆𑀡𑀸𑀭𑁄𑀳𑀯𑀕𑁆𑀕

### 𑀢𑀘𑀲𑀸𑀭𑀚𑀸𑀢𑀓

(95.)

1259. _“𑀅𑀫𑀺𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀢𑁆𑀣𑀕𑀢𑀸,_  
_𑀢𑀘𑀲𑀸𑀭𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑀸𑀢𑁆𑀣,_  
_𑀓𑀲𑁆𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦 𑀲𑁄𑀘𑀣”𑁇_  


(96.)

1260. _“𑀦 𑀲𑁄𑀘𑀦𑀸𑀬 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑀸𑀬,_  
_𑀅𑀢𑁆𑀣𑁄𑀯 𑀮𑀩𑁆𑀪𑁄 𑀅𑀧𑀺 𑀅𑀧𑁆𑀧𑀓𑁄𑀧𑀺;_  
_𑀲𑁄𑀘𑀦𑁆𑀢𑀫𑁂𑀦𑀁 𑀤𑀼𑀔𑀺𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀪𑀯𑀦𑁆𑀢𑀺𑁇_  


(97.)

1261. _𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀆𑀧𑀤𑀸𑀲𑀼,_  
_𑀦 𑀯𑁂𑀥𑀢𑀻 𑀅𑀢𑁆𑀣𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀜𑁆𑀜𑀽;_  
_𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸𑀲𑁆𑀲 𑀤𑀼𑀔𑀺𑀢𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀼𑀔𑀁 𑀅𑀯𑀺𑀓𑀸𑀭𑀁 𑀧𑀼𑀭𑀸𑀡𑀁𑁇_  


(98.)

1262. _𑀚𑀧𑁆𑀧𑁂𑀦 𑀫𑀦𑁆𑀢𑁂𑀦 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑁂𑀦,_  
_𑀅𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑁂𑀦 𑀧𑀯𑁂𑀡𑀺𑀬𑀸 𑀯𑀸;_  
_𑀬𑀣𑀸 𑀬𑀣𑀸 𑀬𑀢𑁆𑀣 𑀮𑀪𑁂𑀣 𑀅𑀢𑁆𑀣𑀁,_  
_𑀢𑀣𑀸 𑀢𑀣𑀸 𑀢𑀢𑁆𑀣 𑀧𑀭𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑁇_  


(99.)

1263. _𑀬𑀢𑁄 𑀘 𑀚𑀸𑀦𑁂𑀬𑁆𑀬 𑀅𑀮𑀩𑁆𑀪𑀦𑁂𑀬𑁆𑀬𑁄,_  
_𑀫𑀬𑀸 𑀯 𑀅𑀜𑁆𑀜𑁂𑀦 𑀯𑀸 𑀏𑀲 𑀅𑀢𑁆𑀣𑁄;_  
_𑀅𑀲𑁄𑀘𑀫𑀸𑀦𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀬𑁂𑀬𑁆𑀬,_  
_𑀓𑀫𑁆𑀫𑀁 𑀤𑀍𑀳𑀁 𑀓𑀺𑀦𑁆𑀢𑀺 𑀓𑀭𑁄𑀫𑀺 𑀤𑀸𑀦𑀻”𑀢𑀺𑁇_  


---
