# 𑁫𑁇𑁨𑁇𑁬 𑀕𑀼𑀫𑁆𑀩𑀺𑀬𑀚𑀸𑀢𑀓

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀯𑀡𑁆𑀡𑀸𑀭𑁄𑀳𑀯𑀕𑁆𑀕

### 𑀕𑀼𑀫𑁆𑀩𑀺𑀬𑀚𑀸𑀢𑀓

(85.)

1247\. _“𑀫𑀥𑀼𑀯𑀡𑁆𑀡𑀁 𑀫𑀥𑀼𑀭𑀲𑀁,_  
_𑀫𑀥𑀼𑀕𑀦𑁆𑀥𑀁 𑀯𑀺𑀲𑀁 𑀅𑀳𑀼;_  
_𑀕𑀼𑀫𑁆𑀩𑀺𑀬𑁄 𑀖𑀸𑀲𑀫𑁂𑀲𑀸𑀦𑁄,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀑𑀤𑀳𑀻 𑀯𑀺𑀲𑀁𑁇_  


(86.)

1248\. _𑀫𑀥𑀼 𑀇𑀢𑀺 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑀸,_  
_𑀬𑁂 𑀢𑀁 𑀯𑀺𑀲𑀫𑀔𑀸𑀤𑀺𑀲𑀼𑀁;_  
_𑀢𑁂𑀲𑀁 𑀢𑀁 𑀓𑀝𑀼𑀓𑀁 𑀆𑀲𑀺,_  
_𑀫𑀭𑀡𑀁 𑀢𑁂𑀦𑀼𑀧𑀸𑀕𑀫𑀼𑀁𑁇_  


(87.)

1249\. _𑀬𑁂 𑀘 𑀔𑁄 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬,_  
_𑀯𑀺𑀲𑀁 𑀢𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀼𑀁;_  
_𑀢𑁂 𑀆𑀢𑀼𑀭𑁂𑀲𑀼 𑀲𑀼𑀔𑀺𑀢𑀸,_  
_𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁂𑀲𑀼 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸𑁇_  


(88.)

1250\. _𑀏𑀯𑀫𑁂𑀯 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀯𑀺𑀲𑀁 𑀓𑀸𑀫𑀸 𑀲𑀫𑁄𑀳𑀺𑀢𑀸;_  
_𑀆𑀫𑀺𑀲𑀁 𑀩𑀦𑁆𑀥𑀦𑀜𑁆𑀘𑁂𑀢𑀁,_  
_𑀫𑀘𑁆𑀘𑀼𑀯𑁂𑀲𑁄 𑀕𑀼𑀳𑀸𑀲𑀬𑁄𑁇_  


(89.)

1251\. _𑀏𑀯𑀫𑁂𑀯 𑀇𑀫𑁂 𑀓𑀸𑀫𑁂,_  
_𑀆𑀢𑀼𑀭𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀓𑁂;_  
_𑀬𑁂 𑀲𑀤𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑀺,_  
_𑀲𑀗𑁆𑀕𑀁 𑀮𑁄𑀓𑁂 𑀉𑀧𑀘𑁆𑀘𑀕𑀼𑀦𑁆”𑀢𑀺𑁇_  


---

1252\. 𑀕𑀼𑀫𑁆𑀩𑀺𑀬𑀚𑀸𑀢𑀓𑀁 𑀙𑀝𑁆𑀞𑀁𑁇
