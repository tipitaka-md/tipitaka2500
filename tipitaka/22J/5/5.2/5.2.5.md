# 𑁫𑁇𑁨𑁇𑁫 𑀅𑀳𑀺𑀢𑀼𑀡𑁆𑀟𑀺𑀓𑀚𑀸𑀢𑀓

* 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀯𑀡𑁆𑀡𑀸𑀭𑁄𑀳𑀯𑀕𑁆𑀕

* 𑀅𑀳𑀺𑀢𑀼𑀡𑁆𑀟𑀺𑀓𑀚𑀸𑀢𑀓

(80.)

1241\. _“𑀥𑀼𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺 𑀲𑀫𑁆𑀫 𑀲𑀼𑀫𑀼𑀔,_  
_𑀚𑀽𑀢𑁂 𑀅𑀓𑁆𑀔𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀳𑀭𑁂𑀳𑀺 𑀅𑀫𑁆𑀩𑀧𑀓𑁆𑀓𑀸𑀦𑀺,_  
_𑀯𑀻𑀭𑀺𑀬𑀁 𑀢𑁂 𑀪𑀓𑁆𑀔𑀬𑀸𑀫𑀲𑁂”𑁇_  


(81.)

1242\. _“𑀅𑀮𑀺𑀓𑀁 𑀯𑀢 𑀫𑀁 𑀲𑀫𑁆𑀫,_  
_𑀅𑀪𑀽𑀢𑁂𑀦 𑀧𑀲𑀁𑀲𑀲𑀺;_  
_𑀓𑁄 𑀢𑁂 𑀲𑀼𑀢𑁄 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑁄 𑀯𑀸,_  
_𑀲𑀼𑀫𑀼𑀔𑁄 𑀦𑀸𑀫 𑀫𑀓𑁆𑀓𑀝𑁄𑁇_  


(82.)

1243\. _𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀫𑁂 𑀢𑀁 𑀫𑀦𑀲𑀺,_  
_𑀬𑀁 𑀫𑀁 𑀢𑁆𑀯𑀁 𑀅𑀳𑀺𑀢𑀼𑀡𑁆𑀟𑀺𑀓;_  
_𑀥𑀜𑁆𑀜𑀸𑀧𑀡𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀢𑁆𑀢𑁄 𑀙𑀸𑀢𑀁 𑀳𑀦𑀸𑀲𑀺 𑀫𑀁𑁇_  


(83.)

1244\. _𑀢𑀸𑀳𑀁 𑀲𑀭𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁂𑀬𑁆𑀬𑀁,_  
_𑀅𑀧𑀺 𑀭𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀓𑀸𑀭𑀬𑁂;_  
_𑀦𑁂𑀯𑀸𑀳𑀁 𑀬𑀸𑀘𑀺𑀢𑁄 𑀤𑀚𑁆𑀚𑀁,_  
_𑀢𑀣𑀸 𑀳𑀺 𑀪𑀬𑀢𑀚𑁆𑀚𑀺𑀢𑁄𑁇_  


(84.)

1245\. _𑀬𑀜𑁆𑀘 𑀚𑀜𑁆𑀜𑀸 𑀓𑀼𑀮𑁂 𑀚𑀸𑀢𑀁,_  
_𑀕𑀩𑁆𑀪𑁂 𑀢𑀺𑀢𑁆𑀢𑀁 𑀅𑀫𑀘𑁆𑀙𑀭𑀺𑀁;_  
_𑀢𑁂𑀦 𑀲𑀔𑀺𑀜𑁆𑀘 𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀥𑀻𑀭𑁄 𑀲𑀦𑁆𑀥𑀸𑀢𑀼𑀫𑀭𑀳𑀢𑀻”𑀢𑀺𑁇_  


---

1246\. 𑀅𑀳𑀺𑀢𑀼𑀡𑁆𑀟𑀺𑀓𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
