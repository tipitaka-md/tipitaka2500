# 𑁫𑁇𑁧𑁇𑁨 𑀲𑀼𑀚𑀸𑀢𑀚𑀸𑀢𑀓

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀚𑀸𑀢𑀚𑀸𑀢𑀓

(6.)

1152\. _“𑀓𑀺𑀁 𑀦𑀼 𑀲𑀦𑁆𑀢𑀭𑀫𑀸𑀦𑁄𑀯,_  
_𑀮𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀳𑀭𑀺𑀢𑀁 𑀢𑀺𑀡𑀁;_  
_𑀔𑀸𑀤 𑀔𑀸𑀤𑀸𑀢𑀺 𑀮𑀧𑀲𑀺,_  
_𑀕𑀢𑀲𑀢𑁆𑀢𑀁 𑀚𑀭𑀕𑁆𑀕𑀯𑀁𑁇_  


(7.)

1153\. _𑀦 𑀳𑀺 𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦,_  
_𑀫𑀢𑁄 𑀕𑁄𑀡𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀳𑁂;_  
_𑀢𑁆𑀯𑀜𑁆𑀘 𑀢𑀼𑀘𑁆𑀙𑀁 𑀯𑀺𑀮𑀧𑀲𑀺,_  
_𑀬𑀣𑀸 𑀢𑀁 𑀤𑀼𑀫𑁆𑀫𑀢𑀻 𑀢𑀣𑀸”𑁇_  


(8.)

1154\. _“𑀢𑀣𑁂𑀯 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀲𑀻𑀲𑀁,_  
_𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀸 𑀘 𑀯𑀸𑀮𑀥𑀺;_  
_𑀲𑁄𑀢𑀸 𑀢𑀣𑁂𑀯 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀫𑀜𑁆𑀜𑁂 𑀕𑁄𑀡𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀳𑁂𑁇_  


(9.)

1155\. _𑀦𑁂𑀯𑀬𑁆𑀬𑀓𑀲𑁆𑀲 𑀲𑀻𑀲𑀜𑁆𑀘,_  
_𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀸 𑀘 𑀤𑀺𑀲𑁆𑀲𑀭𑁂;_  
_𑀭𑀼𑀤𑀁 𑀫𑀢𑁆𑀢𑀺𑀓𑀣𑀽𑀧𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀦𑀼 𑀢𑁆𑀯𑀜𑁆𑀜𑁂𑀯 𑀤𑀼𑀫𑁆𑀫𑀢𑀺”𑁇_  


(10.)

1156\. _“𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀯𑀢 𑀫𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀖𑀢𑀲𑀺𑀢𑁆𑀢𑀁𑀯 𑀧𑀸𑀯𑀓𑀁;_  
_𑀯𑀸𑀭𑀺𑀦𑀸 𑀯𑀺𑀬 𑀑𑀲𑀺𑀜𑁆𑀘𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀬𑁂 𑀤𑀭𑀁𑁇_  


(11.)

1157\. _𑀅𑀩𑁆𑀩𑀳𑀻 𑀯𑀢 𑀫𑁂 𑀲𑀮𑁆𑀮𑀁,_  
_𑀬𑀫𑀸𑀲𑀺 𑀳𑀤𑀬𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀬𑁄 𑀫𑁂 𑀲𑁄𑀓𑀧𑀭𑁂𑀢𑀲𑁆𑀲,_  
_𑀧𑀺𑀢𑀼𑀲𑁄𑀓𑀁 𑀅𑀧𑀸𑀦𑀼𑀤𑀺𑁇_  


(12.)

1158\. _𑀲𑁄𑀳𑀁 𑀅𑀩𑁆𑀩𑀽𑀍𑀳𑀲𑀮𑁆𑀮𑁄𑀲𑁆𑀫𑀺,_  
_𑀯𑀻𑀢𑀲𑁄𑀓𑁄 𑀅𑀦𑀸𑀯𑀺𑀮𑁄;_  
_𑀦 𑀲𑁄𑀘𑀸𑀫𑀺 𑀦 𑀭𑁄𑀤𑀸𑀫𑀺,_  
_𑀢𑀯 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀡𑀯𑁇_  


(13.)

1159\. _𑀏𑀯𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀸,_  
_𑀬𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑀸;_  
_𑀯𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑀺 𑀲𑁄𑀓𑀫𑁆𑀳𑀸,_  
_𑀲𑀼𑀚𑀸𑀢𑁄 𑀧𑀺𑀢𑀭𑀁 𑀬𑀣𑀸”𑀢𑀺𑁇_  


---

1160\. 𑀲𑀼𑀚𑀸𑀢𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
