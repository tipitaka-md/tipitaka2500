# 𑁫𑁇𑁧𑁇𑁧𑁦 𑀲𑀼𑀬𑁄𑀦𑀦𑁆𑀤𑀻𑀚𑀸𑀢𑀓

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀬𑁄𑀦𑀦𑁆𑀤𑀻𑀚𑀸𑀢𑀓

(55.)

1209. _“𑀯𑀸𑀢𑀺 𑀕𑀦𑁆𑀥𑁄 𑀢𑀺𑀫𑀺𑀭𑀸𑀦𑀁,_  
_𑀓𑀼𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀘 𑀖𑁄𑀲𑀯𑀸;_  
_𑀤𑀽𑀭𑁂 𑀇𑀢𑁄 𑀲𑀼𑀬𑁄𑀦𑀦𑁆𑀤𑀻,_  
_𑀢𑀫𑁆𑀩 𑀓𑀸𑀫𑀸 𑀢𑀼𑀤𑀦𑁆𑀢𑀺 𑀫𑀁”𑁇_  


(56.)

1210. _“𑀓𑀣𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀫𑀢𑀭𑀺,_  
_𑀓𑀣𑀁 𑀅𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀲𑁂𑀤𑀼𑀫𑀁;_  
_𑀓𑀣𑀁 𑀢𑀲𑁆𑀲𑀸 𑀘 𑀢𑀼𑀬𑁆𑀳𑀜𑁆𑀘,_  
_𑀅𑀳𑀼 𑀲𑀕𑁆𑀕𑀲𑀫𑀸𑀕𑀫𑁄”𑁇_  


(57.)

1211. _“𑀓𑀼𑀭𑀼𑀓𑀘𑁆𑀙𑀸 𑀧𑀬𑀸𑀢𑀸𑀦𑀁,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀥𑀦𑁂𑀲𑀺𑀦𑀁;_  
_𑀫𑀓𑀭𑁂𑀳𑀺 𑀅𑀪𑀺𑀤𑀸 𑀦𑀸𑀯𑀸,_  
_𑀨𑀮𑀓𑁂𑀦𑀸𑀳𑀫𑀧𑁆𑀮𑀯𑀺𑀁𑁇_  


(58.)

1212. _𑀲𑀸 𑀫𑀁 𑀲𑀡𑁆𑀳𑁂𑀦 𑀫𑀼𑀤𑀼𑀦𑀸,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀘𑀦𑁆𑀤𑀦𑀕𑀦𑁆𑀥𑀺𑀦𑀻;_  
_𑀅𑀗𑁆𑀕𑁂𑀦 𑀉𑀤𑁆𑀥𑀭𑀻 𑀪𑀤𑁆𑀤𑀸,_  
_𑀫𑀸𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀁𑀯 𑀑𑀭𑀲𑀁𑁇_  


(59.)

1213. _𑀲𑀸 𑀫𑀁 𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦,_  
_𑀯𑀢𑁆𑀣𑁂𑀦 𑀲𑀬𑀦𑁂𑀦 𑀘;_  
_𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀘 𑀫𑀦𑁆𑀤𑀓𑁆𑀔𑀻,_  
_𑀏𑀯𑀁 𑀢𑀫𑁆𑀩 𑀯𑀺𑀚𑀸𑀦𑀳𑀻”𑀢𑀺𑁇_  


---

1214. 𑀲𑀼𑀬𑁄𑀦𑀦𑁆𑀤𑀻𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

1215. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1216. _𑀅𑀣 𑀚𑀺𑀦𑀯𑀭𑁄 𑀳𑀭𑀺𑀢𑀁 𑀢𑀺𑀡𑀓𑁄,_  
_𑀅𑀣 𑀪𑀺𑀦𑁆𑀦𑀧𑁆𑀮𑀯𑁄 𑀉𑀭𑀕𑁄𑀯 𑀖𑀝𑁄;_  
_𑀤𑀭𑀺𑀬𑀸 𑀧𑀼𑀦 𑀓𑀼𑀜𑁆𑀚𑀭 𑀪𑀽𑀦𑀳𑀢𑀸,_  
_𑀫𑀺𑀕𑀫𑀼𑀢𑁆𑀢𑀫𑀲𑀕𑁆𑀕𑀯𑀭𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
