# 𑁫𑁇𑁧𑁇𑁩 𑀯𑁂𑀦𑀲𑀸𑀔𑀚𑀸𑀢𑀓

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀦𑀲𑀸𑀔𑀚𑀸𑀢𑀓

(14.)

1161\. _“𑀦𑀬𑀺𑀤𑀁 𑀦𑀺𑀘𑁆𑀘𑀁 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢,_  
_𑀔𑁂𑀫𑀁 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀲𑀼𑀔𑀢𑀸 𑀘 𑀓𑀸𑀬𑁂;_  
_𑀅𑀢𑁆𑀣𑀘𑁆𑀘𑀬𑁂 𑀫𑀸 𑀅𑀳𑀼 𑀲𑀫𑁆𑀧𑀫𑀽𑀍𑀳𑁄,_  
_𑀪𑀺𑀦𑁆𑀦𑀧𑁆𑀮𑀯𑁄 𑀲𑀸𑀕𑀭𑀲𑁆𑀲𑁂𑀯 𑀫𑀚𑁆𑀛𑁂𑁇_  


(15.)

1162\. _𑀬𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀢𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀓𑀸𑀭𑀻 𑀓𑀮𑁆𑀬𑀸𑀡𑀁,_  
_𑀧𑀸𑀧𑀓𑀸𑀭𑀻 𑀘 𑀧𑀸𑀧𑀓𑀁;_  
_𑀬𑀸𑀤𑀺𑀲𑀁 𑀯𑀧𑀢𑁂 𑀩𑀻𑀚𑀁,_  
_𑀢𑀸𑀤𑀺𑀲𑀁 𑀳𑀭𑀢𑁂 𑀨𑀮𑀁”𑁇_  


(16.)

1163\. _“𑀇𑀤𑀁 𑀢𑀤𑀸𑀘𑀭𑀺𑀬𑀯𑀘𑁄,_  
_𑀧𑀸𑀭𑀸𑀲𑀭𑀺𑀬𑁄 𑀬𑀤𑀩𑁆𑀭𑀯𑀺;_  
_𑀫𑀸 𑀲𑀼 𑀢𑁆𑀯𑀁 𑀅𑀓𑀭𑀺 𑀧𑀸𑀧𑀁,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀧𑀘𑁆𑀙𑀸 𑀓𑀢𑀁 𑀢𑀧𑁂𑁇_  


(17.)

1164\. _𑀅𑀬𑀫𑁂𑀯 𑀲𑁄 𑀧𑀺𑀗𑁆𑀕𑀺𑀬 𑀯𑁂𑀦𑀲𑀸𑀔𑁄,_  
_𑀬𑀫𑁆𑀳𑀺 𑀖𑀸𑀢𑀬𑀺𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀲𑀳𑀲𑁆𑀲𑀁;_  
_𑀅𑀮𑀗𑁆𑀓𑀢𑁂 𑀘𑀦𑁆𑀤𑀦𑀲𑀸𑀭𑀸𑀦𑀼𑀮𑀺𑀢𑁆𑀢𑁂,_  
_𑀢𑀫𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀢𑀁 𑀫𑀫𑀁𑁇_  


(18.)

1165\. _𑀲𑀸𑀫𑀸 𑀘 𑀔𑁄 𑀘𑀦𑁆𑀤𑀦𑀮𑀺𑀢𑁆𑀢𑀕𑀢𑁆𑀢𑀸,_  
_𑀮𑀝𑁆𑀞𑀻𑀯 𑀲𑁄𑀪𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲 𑀉𑀕𑁆𑀕𑀢𑀸;_  
_𑀅𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀉𑀩𑁆𑀩𑀭𑀺𑀁,_  
_𑀢𑀁 𑀫𑁂 𑀇𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

1166\. 𑀯𑁂𑀦𑀲𑀸𑀔𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
