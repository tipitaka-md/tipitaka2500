# 𑁫𑁇𑁧𑁇𑁪 𑀉𑀭𑀕𑀚𑀸𑀢𑀓

* 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

* 𑀉𑀭𑀕𑀚𑀸𑀢𑀓

(19.)

1167\. _“𑀉𑀭𑀕𑁄𑀯 𑀢𑀘𑀁 𑀚𑀺𑀡𑁆𑀡𑀁,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀕𑀘𑁆𑀙𑀢𑀺 𑀲𑀁 𑀢𑀦𑀼𑀁;_  
_𑀏𑀯𑀁 𑀲𑀭𑀻𑀭𑁂 𑀦𑀺𑀩𑁆𑀪𑁄𑀕𑁂,_  
_𑀧𑁂𑀢𑁂 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀲𑀢𑀺𑁇_  


(20.)

1168\. _𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀢𑀁 𑀦 𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀕𑀢𑁄 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀕𑀢𑀺”𑁇_  


(21.)

1169\. _“𑀅𑀦𑀯𑁆𑀳𑀺𑀢𑁄 𑀢𑀢𑁄 𑀆𑀕𑀸,_  
_𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄 𑀇𑀢𑁄 𑀕𑀢𑁄;_  
_𑀬𑀣𑀸𑀕𑀢𑁄 𑀢𑀣𑀸 𑀕𑀢𑁄,_  
_𑀢𑀢𑁆𑀣 𑀓𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀦𑀸𑁇_  


(22.)

1170\. _𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀢𑀁 𑀦 𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀕𑀢𑁄 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀕𑀢𑀺”𑁇_  


(23.)

1171\. _“𑀲𑀘𑁂 𑀭𑁄𑀤𑁂 𑀓𑀺𑀲𑀸 𑀅𑀲𑁆𑀲𑀁,_  
_𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀓𑀺𑀁 𑀨𑀮𑀁 𑀲𑀺𑀬𑀸;_  
_𑀜𑀸𑀢𑀺𑀫𑀺𑀢𑁆𑀢𑀲𑀼𑀳𑀚𑁆𑀚𑀸𑀦𑀁,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀦𑁄 𑀅𑀭𑀢𑀻 𑀲𑀺𑀬𑀸𑁇_  


(24.)

1172\. _𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀢𑀁 𑀦 𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀕𑀢𑁄 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀕𑀢𑀺”𑁇_  


(25.)

1173\. _“𑀬𑀣𑀸𑀧𑀺 𑀤𑀸𑀭𑀓𑁄 𑀘𑀦𑁆𑀤𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀫𑀦𑀼𑀭𑁄𑀤𑀢𑀺;_  
_𑀏𑀯𑀁𑀲𑀫𑁆𑀧𑀤𑀫𑁂𑀯𑁂𑀢𑀁,_  
_𑀬𑁄 𑀧𑁂𑀢𑀫𑀦𑀼𑀲𑁄𑀘𑀢𑀺𑁇_  


(26.)

1174\. _𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀢𑀁 𑀦 𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀕𑀢𑁄 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀕𑀢𑀺”𑁇_  


(27.)

1175\. _“𑀬𑀣𑀸𑀧𑀺 𑀉𑀤𑀓𑀓𑀼𑀫𑁆𑀪𑁄,_  
_𑀪𑀺𑀦𑁆𑀦𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀬𑁄;_  
_𑀏𑀯𑀁𑀲𑀫𑁆𑀧𑀤𑀫𑁂𑀯𑁂𑀢𑀁,_  
_𑀬𑁄 𑀧𑁂𑀢𑀫𑀦𑀼𑀲𑁄𑀘𑀢𑀺𑁇_  


(28.)

1176\. _𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀢𑀁 𑀦 𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀕𑀢𑁄 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀬𑀸 𑀕𑀢𑀻”𑀢𑀺𑁇_  


---
