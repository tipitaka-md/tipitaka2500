# 𑁫𑁇𑁧𑁇𑁯 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀺𑀕𑀚𑀸𑀢𑀓

* 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀺𑀕𑀚𑀸𑀢𑀓

(50.)

1203\. _“𑀯𑀺𑀓𑁆𑀓𑀫 𑀭𑁂 𑀳𑀭𑀺𑀧𑀸𑀤,_  
_𑀯𑀺𑀓𑁆𑀓𑀫 𑀭𑁂 𑀫𑀳𑀸𑀫𑀺𑀕;_  
_𑀙𑀺𑀦𑁆𑀤 𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀓𑀁 𑀧𑀸𑀲𑀁,_  
_𑀦𑀸𑀳𑀁 𑀏𑀓𑀸 𑀯𑀦𑁂 𑀭𑀫𑁂”𑁇_  


(51.)

1204\. _“𑀯𑀺𑀓𑁆𑀓𑀫𑀸𑀫𑀺 𑀦 𑀧𑀸𑀭𑁂𑀫𑀺,_  
_𑀪𑀽𑀫𑀺𑀁 𑀲𑀼𑀫𑁆𑀪𑀸𑀫𑀺 𑀯𑁂𑀕𑀲𑀸;_  
_𑀤𑀍𑀳𑁄 𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀓𑁄 𑀧𑀸𑀲𑁄,_  
_𑀧𑀸𑀤𑀁 𑀫𑁂 𑀧𑀭𑀺𑀓𑀦𑁆𑀢𑀢𑀺”𑁇_  


(52.)

1205\. _“𑀅𑀢𑁆𑀣𑀭𑀲𑁆𑀲𑀼 𑀧𑀮𑀸𑀲𑀸𑀦𑀺,_  
_𑀅𑀲𑀺𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀳 𑀮𑀼𑀤𑁆𑀤𑀓;_  
_𑀧𑀞𑀫𑀁 𑀫𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀳𑀦 𑀧𑀘𑁆𑀙𑀸 𑀫𑀳𑀸𑀫𑀺𑀕𑀁”𑁇_  


(53.)

1206\. _“𑀦 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸,_  
_𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑀁 𑀫𑀸𑀦𑀼𑀲𑀺𑀁 𑀫𑀺𑀕𑀺𑀁;_  
_𑀢𑁆𑀯𑀜𑁆𑀘 𑀪𑀤𑁆𑀤𑁂 𑀲𑀼𑀔𑀻 𑀳𑁄𑀳𑀺,_  
_𑀏𑀲𑁄 𑀘𑀸𑀧𑀺 𑀫𑀳𑀸𑀫𑀺𑀕𑁄”𑁇_  


(54.)

1207\. _“𑀏𑀯𑀁 𑀮𑀼𑀤𑁆𑀤𑀓 𑀦𑀦𑁆𑀤𑀲𑁆𑀲𑀼,_  
_𑀲𑀳 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀜𑀸𑀢𑀺𑀪𑀺;_  
_𑀬𑀣𑀸𑀳𑀫𑀚𑁆𑀚 𑀦𑀦𑁆𑀤𑀸𑀫𑀺,_  
_𑀫𑀼𑀢𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀳𑀸𑀫𑀺𑀕𑀦𑁆”𑀢𑀺𑁇_  


---

1208\. 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀺𑀕𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇
