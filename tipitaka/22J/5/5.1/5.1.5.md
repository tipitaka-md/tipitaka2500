# 𑁫𑁇𑁧𑁇𑁫 𑀖𑀝𑀚𑀸𑀢𑀓

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

### 𑀖𑀝𑀚𑀸𑀢𑀓

(29.)

1178. _“𑀅𑀜𑁆𑀜𑁂 𑀲𑁄𑀘𑀦𑁆𑀢𑀺 𑀭𑁄𑀤𑀦𑁆𑀢𑀺,_  
_𑀅𑀜𑁆𑀜𑁂 𑀅𑀲𑁆𑀲𑀼𑀫𑀼𑀔𑀸 𑀚𑀦𑀸;_  
_𑀧𑀲𑀦𑁆𑀦𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄𑀲𑀺,_  
_𑀓𑀲𑁆𑀫𑀸 𑀖𑀝 𑀦 𑀲𑁄𑀘𑀲𑀺”𑁇_  


(30.)

1179. _“𑀦𑀸𑀩𑁆𑀪𑀢𑀻𑀢𑀳𑀭𑁄 𑀲𑁄𑀓𑁄,_  
_𑀦𑀸𑀦𑀸𑀕𑀢𑀲𑀼𑀔𑀸𑀯𑀳𑁄;_  
_𑀢𑀲𑁆𑀫𑀸 𑀥𑀗𑁆𑀓 𑀦 𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀦𑀢𑁆𑀣𑀺 𑀲𑁄𑀓𑁂 𑀤𑀼𑀢𑀻𑀬𑀢𑀸𑁇_  


(31.)

1180. _𑀲𑁄𑀘𑀁 𑀧𑀡𑁆𑀟𑀼 𑀓𑀺𑀲𑁄 𑀳𑁄𑀢𑀺,_  
_𑀪𑀢𑁆𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀦 𑀭𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀲𑀮𑁆𑀮𑀯𑀺𑀤𑁆𑀥𑀲𑁆𑀲 𑀭𑀼𑀧𑁆𑀧𑀢𑁄𑁇_  


(32.)

1181. _𑀕𑀸𑀫𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀦𑀺𑀦𑁆𑀦𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀣𑀮𑁂;_  
_𑀞𑀺𑀢𑀁 𑀫𑀁 𑀦𑀸𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺,_  
_𑀏𑀯𑀁 𑀤𑀺𑀝𑁆𑀞𑀧𑀤𑁄 𑀅𑀳𑀁𑁇_  


(33.)

1182. _𑀬𑀲𑁆𑀲𑀢𑁆𑀢𑀸 𑀦𑀸𑀮𑀫𑁂𑀓𑁄𑀯,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀭𑀲𑀸𑀳𑀭𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀧𑀺 𑀧𑀣𑀯𑀻 𑀢𑀲𑁆𑀲,_  
_𑀦 𑀲𑀼𑀔𑀁 𑀆𑀯𑀳𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

1183. 𑀖𑀝𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
