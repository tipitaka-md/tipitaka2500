# 𑁬𑁇𑁧𑁇𑁪 𑀦𑁂𑀭𑀼𑀚𑀸𑀢𑀓

* 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀯𑀸𑀭𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑁂𑀭𑀼𑀚𑀸𑀢𑀓

(20.)

1340\. _“𑀓𑀸𑀓𑁄𑀮𑀸 𑀓𑀸𑀓𑀲𑀗𑁆𑀖𑀸 𑀘,_  
_𑀫𑀬𑀜𑁆𑀘 𑀧𑀢𑀢𑀁 𑀯𑀭𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀲𑀤𑀺𑀲𑀸 𑀳𑁄𑀫,_  
_𑀇𑀫𑀁 𑀆𑀕𑀫𑁆𑀫 𑀧𑀩𑁆𑀩𑀢𑀁𑁇_  


(21.)

1341\. _𑀇𑀥 𑀲𑀻𑀳𑀸 𑀘 𑀩𑁆𑀬𑀕𑁆𑀖𑀸 𑀘,_  
_𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀸 𑀘 𑀫𑀺𑀕𑀸𑀥𑀫𑀸;_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀲𑀤𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀅𑀬𑀁 𑀓𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄”𑁇_  


(22.)

1342\. _“𑀇𑀫𑀁 𑀦𑁂𑀭𑀽𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀩𑁆𑀩𑀢𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀇𑀥 𑀯𑀡𑁆𑀡𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀯𑀲𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑁄”𑁇_  


(23.)

1343\. _“𑀅𑀫𑀸𑀦𑀦𑀸 𑀬𑀢𑁆𑀣 𑀲𑀺𑀬𑀸,_  
_𑀲𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀸 𑀯𑀺𑀫𑀸𑀦𑀦𑀸;_  
_𑀳𑀻𑀦𑀲𑀫𑁆𑀫𑀸𑀦𑀦𑀸 𑀯𑀸𑀧𑀺,_  
_𑀦 𑀢𑀢𑁆𑀣 𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(24.)

1344\. _𑀬𑀢𑁆𑀣𑀸𑀮𑀲𑁄 𑀘 𑀤𑀓𑁆𑀔𑁄 𑀘,_  
_𑀲𑀽𑀭𑁄 𑀪𑀻𑀭𑀼 𑀘 𑀧𑀽𑀚𑀺𑀬𑀸;_  
_𑀦 𑀢𑀢𑁆𑀣 𑀲𑀦𑁆𑀢𑁄 𑀯𑀲𑀦𑁆𑀢𑀺,_  
_𑀅𑀯𑀺𑀲𑁂𑀲𑀓𑀭𑁂 𑀦𑀭𑁂𑁇_  


(25.)

1345\. _𑀦𑀸𑀬𑀁 𑀦𑁂𑀭𑀼 𑀯𑀺𑀪𑀚𑀢𑀺,_  
_𑀳𑀻𑀦𑁅𑀓𑁆𑀓𑀝𑁆𑀞𑀫𑀚𑁆𑀛𑀺𑀫𑁂;_  
_𑀅𑀯𑀺𑀲𑁂𑀲𑀓𑀭𑁄 𑀦𑁂𑀭𑀼,_  
_𑀳𑀦𑁆𑀤 𑀦𑁂𑀭𑀼𑀁 𑀚𑀳𑀸𑀫𑀲𑁂”𑀢𑀺𑁇_  


---
