# 𑁬𑁇𑁧𑁇𑁭 𑀲𑀺𑀭𑀺𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀯𑀸𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀺𑀭𑀺𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓

(40.)

1363\. _“𑀓𑀸 𑀦𑀼 𑀓𑀸𑀍𑀏𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀓𑀸 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀥𑀻𑀢𑀸,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(41.)

1364\. _“𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲𑀳𑀁 𑀥𑀻𑀢𑀸,_  
_𑀯𑀺𑀭𑀽𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀘𑀡𑁆𑀟𑀺𑀬𑀸;_  
_𑀅𑀳𑀁 𑀓𑀸𑀍𑀈 𑀅𑀮𑀓𑁆𑀔𑀺𑀓𑀸,_  
_𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀻𑀢𑀺 𑀫𑀁 𑀯𑀺𑀤𑀽;_  
_𑀑𑀓𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀢𑁄 𑀤𑁂𑀳𑀺,_  
_𑀯𑀲𑁂𑀫𑀼 𑀢𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑁇_  


(42.)

1365\. _“𑀓𑀺𑀁𑀲𑀻𑀮𑁂 𑀓𑀺𑀁𑀲𑀫𑀸𑀘𑀸𑀭𑁂,_  
_𑀧𑀼𑀭𑀺𑀲𑁂 𑀦𑀺𑀯𑀺𑀲𑀲𑁂 𑀢𑀼𑀯𑀁;_  
_𑀧𑀼𑀝𑁆𑀞𑀸 𑀫𑁂 𑀓𑀸𑀍𑀇 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(43.)

1366\. _“𑀫𑀓𑁆𑀔𑀻 𑀧𑀍𑀆𑀲𑀻 𑀲𑀸𑀭𑀫𑁆𑀪𑀻,_  
_𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀫𑀘𑁆𑀙𑀭𑀻 𑀲𑀞𑁄;_  
_𑀲𑁄 𑀫𑀬𑁆𑀳𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀦𑁆𑀢𑁄,_  
_𑀮𑀤𑁆𑀥𑀁 𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺𑁇_  


(44.)

1367\. _𑀓𑁄𑀥𑀦𑁄 𑀉𑀧𑀦𑀸𑀳𑀻 𑀘,_  
_𑀧𑀺𑀲𑀼𑀡𑁄 𑀘 𑀯𑀺𑀪𑁂𑀤𑀓𑁄;_  
_𑀓𑀡𑁆𑀟𑀓𑀯𑀸𑀘𑁄 𑀨𑀭𑀼𑀲𑁄,_  
_𑀲𑁄 𑀫𑁂 𑀓𑀦𑁆𑀢𑀢𑀭𑁄 𑀢𑀢𑁄𑁇_  


(45.)

1368\. _𑀅𑀚𑁆𑀚 𑀲𑀼𑀯𑁂𑀢𑀺 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀲𑀤𑀢𑁆𑀣𑀁 𑀦𑀸𑀯𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀑𑀯𑀚𑁆𑀚𑀫𑀸𑀦𑁄 𑀓𑀼𑀧𑁆𑀧𑀢𑀺,_  
_𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑁄 𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺𑁇_  


(46.)

1369\. _𑀤𑀯𑀧𑁆𑀧𑀮𑀼𑀤𑁆𑀥𑁄 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀲𑀩𑁆𑀩𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀥𑀁𑀲𑀢𑀺;_  
_𑀲𑁄 𑀫𑀬𑁆𑀳𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀦𑁆𑀢𑁄,_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀳𑁄𑀫𑀺 𑀅𑀦𑀸𑀫𑀬𑀸”𑁇_  


(47.)

1370\. _“𑀅𑀧𑁂𑀳𑀺 𑀏𑀢𑁆𑀢𑁄 𑀢𑁆𑀯𑀁 𑀓𑀸𑀍𑀇,_  
_𑀦𑁂𑀢𑀁 𑀅𑀫𑁆𑀳𑁂𑀲𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀜𑁆𑀜𑀁 𑀚𑀦𑀧𑀤𑀁 𑀕𑀘𑁆𑀙,_  
_𑀦𑀺𑀕𑀫𑁂 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑁄”𑁇_  


(48.)

1371\. _“𑀅𑀳𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀦𑁂𑀢𑀁 𑀢𑀼𑀫𑁆𑀳𑁂𑀲𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀦𑁆𑀢𑀺 𑀮𑁄𑀓𑁂 𑀅𑀮𑀓𑁆𑀔𑀺𑀓𑀸,_  
_𑀲𑀁𑀖𑀭𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀁 𑀥𑀦𑀁;_  
_𑀅𑀳𑀁 𑀤𑁂𑀯𑁄 𑀘 𑀫𑁂 𑀪𑀸𑀢𑀸,_  
_𑀉𑀪𑁄 𑀦𑀁 𑀯𑀺𑀥𑀫𑀸𑀫𑀲𑁂”𑁇_  


(49.)

1372\. _“𑀓𑀸 𑀦𑀼 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀲𑀼𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀓𑀸 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀥𑀻𑀢𑀸,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(50.)

1373\. _“𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲𑀳𑀁 𑀥𑀻𑀢𑀸,_  
_𑀥𑀢𑀭𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀺𑀭𑀻𑀫𑀢𑁄;_  
_𑀅𑀳𑀁 𑀲𑀺𑀭𑀻 𑀘 𑀮𑀓𑁆𑀔𑀻 𑀘,_  
_𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜𑀸𑀢𑀺 𑀫𑀁 𑀯𑀺𑀤𑀽;_  
_𑀑𑀓𑀸𑀲𑀁 𑀬𑀸𑀘𑀺𑀢𑁄 𑀤𑁂𑀳𑀺,_  
_𑀯𑀲𑁂𑀫𑀼 𑀢𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑁇_  


(51.)

1374\. _“𑀓𑀺𑀁𑀲𑀻𑀮𑁂 𑀓𑀺𑀁𑀲𑀫𑀸𑀘𑀸𑀭𑁂,_  
_𑀧𑀼𑀭𑀺𑀲𑁂 𑀦𑀺𑀯𑀺𑀲𑀲𑁂 𑀢𑀼𑀯𑀁;_  
_𑀧𑀼𑀝𑁆𑀞𑀸 𑀫𑁂 𑀮𑀓𑁆𑀔𑀺 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(52.)

1375\. _“𑀬𑁄 𑀘𑀸𑀧𑀺 𑀲𑀻𑀢𑁂 𑀅𑀣 𑀯𑀸𑀧𑀺 𑀉𑀡𑁆𑀳𑁂,_  
_𑀯𑀸𑀢𑀸𑀢𑀧𑁂 𑀟𑀁𑀲𑀲𑀭𑀻𑀲𑀧𑁂 𑀘;_  
_𑀔𑀼𑀥𑀁 𑀧𑀺𑀧𑀸𑀲𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀲𑀩𑁆𑀩𑀁,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀬𑁄 𑀲𑀢𑀢𑀁 𑀦𑀺𑀬𑀼𑀢𑁆𑀢𑁄𑁇_  


(53.)

1376\. _𑀓𑀸𑀮𑀸𑀕𑀢𑀜𑁆𑀘 𑀦 𑀳𑀸𑀧𑁂𑀢𑀺 𑀅𑀢𑁆𑀣𑀁,_  
_𑀲𑁄 𑀫𑁂 𑀫𑀦𑀸𑀧𑁄 𑀦𑀺𑀯𑀺𑀲𑁂 𑀘 𑀢𑀫𑁆𑀳𑀺;_  
_𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀫𑀺𑀢𑁆𑀢𑀯𑀸 𑀘𑀸𑀕𑀯𑀸 𑀘,_  
_𑀲𑀻𑀮𑀽𑀧𑀧𑀦𑁆𑀦𑁄 𑀅𑀲𑀞𑁄𑀚𑀼𑀪𑀽𑀢𑁄𑁇_  


(54.)

1377\. _𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀔𑀺𑀮𑁄 𑀲𑀡𑁆𑀳𑀯𑀸𑀘𑁄,_  
_𑀫𑀳𑀢𑁆𑀢𑀧𑀢𑁆𑀢𑁄𑀧𑀺 𑀦𑀺𑀯𑀸𑀢𑀯𑀼𑀢𑁆𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀺𑀁𑀳𑀁 𑀧𑁄𑀲𑁂 𑀯𑀺𑀧𑀼𑀮𑀸 𑀪𑀯𑀸𑀫𑀺,_  
_𑀊𑀫𑀺 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀬𑀣𑀸𑀧𑀺 𑀯𑀡𑁆𑀡𑀁𑁇_  


(55.)

1378\. _𑀬𑁄 𑀘𑀸𑀧𑀺 𑀫𑀺𑀢𑁆𑀢𑁂 𑀅𑀣 𑀯𑀸 𑀅𑀫𑀺𑀢𑁆𑀢𑁂,_  
_𑀲𑁂𑀝𑁆𑀞𑁂 𑀲𑀭𑀺𑀓𑁆𑀔𑁂 𑀅𑀣 𑀯𑀸𑀧𑀺 𑀳𑀻𑀦𑁂;_  
_𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀅𑀣 𑀯𑀸 𑀅𑀦𑀢𑁆𑀣𑀁,_  
_𑀆𑀯𑀻 𑀭𑀳𑁄 𑀲𑀗𑁆𑀕𑀳𑀫𑁂𑀯 𑀯𑀢𑁆𑀢𑁂𑁇_  


(56.)

1379\. _𑀯𑀸𑀘𑀁 𑀦 𑀯𑀚𑁆𑀚𑀸 𑀨𑀭𑀼𑀲𑀁 𑀓𑀤𑀸𑀘𑀺,_  
_𑀫𑀢𑀲𑁆𑀲 𑀚𑀻𑀯𑀲𑁆𑀲 𑀘 𑀢𑀲𑁆𑀲 𑀳𑁄𑀫𑀺;_  
_𑀏𑀢𑁂𑀲𑀁 𑀬𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀮𑀪𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀦𑁆𑀢𑀸 𑀲𑀺𑀭𑀻 𑀫𑀚𑁆𑀚𑀢𑀺 𑀅𑀧𑁆𑀧𑀧𑀜𑁆𑀜𑁄;_  
_𑀢𑀁 𑀤𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑀁 𑀯𑀺𑀲𑀫𑀁 𑀘𑀭𑀦𑁆𑀢𑀁,_  
_𑀓𑀭𑀻𑀲𑀞𑀸𑀦𑀁𑀯 𑀯𑀺𑀯𑀚𑁆𑀚𑀬𑀸𑀫𑀺𑁇_  


(57.)

1380\. _𑀅𑀢𑁆𑀢𑀦𑀸 𑀓𑀼𑀭𑀼𑀢𑁂 𑀮𑀓𑁆𑀔𑀺𑀁,_  
_𑀅𑀮𑀓𑁆𑀔𑀺𑀁 𑀓𑀼𑀭𑀼𑀢𑀢𑁆𑀢𑀦𑀸;_  
_𑀦 𑀳𑀺 𑀮𑀓𑁆𑀔𑀺𑀁 𑀅𑀮𑀓𑁆𑀔𑀺𑀁 𑀯𑀸,_  
_𑀅𑀜𑁆𑀜𑁄 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀓𑀸𑀭𑀓𑁄”𑀢𑀺𑁇_  


---

1381\. 𑀲𑀺𑀭𑀺𑀓𑀸𑀍𑀅𑀓𑀡𑁆𑀡𑀺𑀚𑀸𑀢𑀓𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
