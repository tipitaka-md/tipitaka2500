# 𑁬𑁇𑁧𑁇𑁧 𑀅𑀯𑀸𑀭𑀺𑀬𑀚𑀸𑀢𑀓

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀯𑀸𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀅𑀯𑀸𑀭𑀺𑀬𑀚𑀸𑀢𑀓

(1.)

1318\. _“𑀫𑀸𑀲𑀼 𑀓𑀼𑀚𑁆𑀛 𑀪𑀽𑀫𑀺𑀧𑀢𑀺,_  
_𑀫𑀸𑀲𑀼 𑀓𑀼𑀚𑁆𑀛 𑀭𑀣𑁂𑀲𑀪;_  
_𑀓𑀼𑀤𑁆𑀥𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑁄,_  
_𑀭𑀸𑀚𑀸 𑀭𑀝𑁆𑀞𑀲𑁆𑀲 𑀧𑀽𑀚𑀺𑀢𑁄𑁇_  


(2.)

1319\. _𑀕𑀸𑀫𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀦𑀺𑀦𑁆𑀦𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀣𑀮𑁂;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀦𑀼𑀲𑀸𑀲𑀸𑀫𑀺,_  
_𑀫𑀸𑀲𑀼 𑀓𑀼𑀚𑁆𑀛 𑀭𑀣𑁂𑀲𑀪”𑁇_  


(3.)

1320\. _“𑀅𑀯𑀸𑀭𑀺𑀬𑀧𑀺𑀢𑀸 𑀦𑀸𑀫,_  
_𑀅𑀳𑀼 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀸𑀯𑀺𑀓𑁄;_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀚𑀦𑀁 𑀢𑀸𑀭𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀘𑁆𑀙𑀸 𑀬𑀸𑀘𑀢𑀺 𑀯𑁂𑀢𑀦𑀁;_  
_𑀢𑁂𑀦𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀳𑁄𑀢𑀺,_  
_𑀦 𑀘 𑀪𑁄𑀕𑁂𑀳𑀺 𑀯𑀟𑁆𑀠𑀢𑀺”𑁇_  


(4.)

1321\. _“𑀅𑀢𑀺𑀡𑁆𑀡𑀁𑀬𑁂𑀯 𑀬𑀸𑀘𑀲𑁆𑀲𑀼,_  
_𑀅𑀧𑀸𑀭𑀁 𑀢𑀸𑀢 𑀦𑀸𑀯𑀺𑀓;_  
_𑀅𑀜𑁆𑀜𑁄 𑀳𑀺 𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀫𑀦𑁄,_  
_𑀅𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺 𑀧𑀸𑀭𑁂𑀲𑀺𑀦𑁄𑁇_  


(5.)

1322\. _𑀕𑀸𑀫𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸𑀭𑀜𑁆𑀜𑁂,_  
_𑀦𑀺𑀦𑁆𑀦𑁂 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀣𑀮𑁂;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀦𑀼𑀲𑀸𑀲𑀸𑀫𑀺,_  
_𑀫𑀸𑀲𑀼 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀦𑀸𑀯𑀺𑀓”𑁇_  


(6.)

1323\. _“𑀬𑀸𑀬𑁂𑀯𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀬𑀸,_  
_𑀭𑀸𑀚𑀸 𑀕𑀸𑀫𑀯𑀭𑀁 𑀅𑀤𑀸;_  
_𑀢𑀸𑀬𑁂𑀯𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀬𑀸,_  
_𑀦𑀸𑀯𑀺𑀓𑁄 𑀧𑀳𑀭𑀻 𑀫𑀼𑀔𑀁𑁇_  


(7.)

1324\. _𑀪𑀢𑁆𑀢𑀁 𑀪𑀺𑀦𑁆𑀦𑀁 𑀳𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀕𑀩𑁆𑀪𑁄 𑀘 𑀧𑀢𑀺𑀢𑁄 𑀙𑀫𑀸;_  
_𑀫𑀺𑀕𑁄𑀯 𑀚𑀸𑀢𑀭𑀽𑀧𑁂𑀦,_  
_𑀦 𑀢𑁂𑀦𑀢𑁆𑀣𑀁 𑀅𑀩𑀦𑁆𑀥𑀺 𑀲𑀽”𑀢𑀺𑁇_  


---
