# 𑁬𑁇𑁧𑁇𑁫 𑀆𑀲𑀗𑁆𑀓𑀚𑀸𑀢𑀓

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀯𑀸𑀭𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀆𑀲𑀗𑁆𑀓𑀚𑀸𑀢𑀓

(26.)

1347. _“𑀆𑀲𑀸𑀯𑀢𑀻 𑀦𑀸𑀫 𑀮𑀢𑀸,_  
_𑀚𑀸𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀦𑁂;_  
_𑀢𑀲𑁆𑀲𑀸 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑁂𑀦,_  
_𑀏𑀓𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑁂 𑀨𑀮𑀁𑁇_  


(27.)

1348. _𑀢𑀁 𑀤𑁂𑀯𑀸 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀸𑀯 𑀤𑀽𑀭𑀨𑀮𑀁 𑀲𑀢𑀺𑀁;_  
_𑀆𑀲𑀻𑀲𑁂𑀯 𑀢𑀼𑀯𑀁 𑀭𑀸𑀚,_  
_𑀆𑀲𑀸 𑀨𑀮𑀯𑀢𑀻 𑀲𑀼𑀔𑀸𑁇_  


(28.)

1349. _𑀆𑀲𑀻𑀲𑀢𑁂𑀯 𑀲𑁄 𑀧𑀓𑁆𑀔𑀻,_  
_𑀆𑀲𑀻𑀲𑀢𑁂𑀯 𑀲𑁄 𑀤𑀺𑀚𑁄;_  
_𑀢𑀲𑁆𑀲 𑀘𑀸𑀲𑀸 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀺,_  
_𑀢𑀸𑀯 𑀤𑀽𑀭𑀕𑀢𑀸 𑀲𑀢𑀻;_  
_𑀆𑀲𑀻𑀲𑁂𑀯 𑀢𑀼𑀯𑀁 𑀭𑀸𑀚,_  
_𑀆𑀲𑀸 𑀨𑀮𑀯𑀢𑀻 𑀲𑀼𑀔𑀸”𑁇_  


(29.)

1350. _“𑀲𑀫𑁆𑀧𑁂𑀲𑀺 𑀔𑁄 𑀫𑀁 𑀯𑀸𑀘𑀸𑀬,_  
_𑀦 𑀘 𑀲𑀫𑁆𑀧𑁂𑀲𑀺 𑀓𑀫𑁆𑀫𑀼𑀦𑀸;_  
_𑀫𑀸𑀮𑀸 𑀲𑁂𑀭𑁂𑀬𑁆𑀬𑀓𑀲𑁆𑀲𑁂𑀯,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑀸 𑀅𑀕𑀦𑁆𑀥𑀺𑀓𑀸𑁇_  


(30.)

1351. _𑀅𑀨𑀮𑀁 𑀫𑀥𑀼𑀭𑀁 𑀯𑀸𑀘𑀁,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑁂𑀲𑀼 𑀧𑀓𑀼𑀩𑁆𑀩𑀢𑀺;_  
_𑀅𑀤𑀤𑀁 𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑀁 𑀪𑁄𑀕𑀁,_  
_𑀲𑀦𑁆𑀥𑀺 𑀢𑁂𑀦𑀲𑁆𑀲 𑀚𑀻𑀭𑀢𑀺𑁇_  


(31.)

1352. _𑀬𑀜𑁆𑀳𑀺 𑀓𑀬𑀺𑀭𑀸 𑀢𑀜𑁆𑀳𑀺 𑀯𑀤𑁂,_  
_𑀬𑀁 𑀦 𑀓𑀬𑀺𑀭𑀸 𑀦 𑀢𑀁 𑀯𑀤𑁂;_  
_𑀅𑀓𑀭𑁄𑀦𑁆𑀢𑀁 𑀪𑀸𑀲𑀫𑀸𑀦𑀁,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


(32.)

1353. _𑀩𑀮𑀜𑁆𑀘 𑀯𑀢 𑀫𑁂 𑀔𑀻𑀡𑀁,_  
_𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀗𑁆𑀓𑁂 𑀧𑀸𑀡𑀽𑀧𑀭𑁄𑀥𑀸𑀬,_  
_𑀳𑀦𑁆𑀤 𑀤𑀸𑀦𑀺 𑀯𑀚𑀸𑀫𑀳𑀁”𑁇_  


(33.)

1354. _“𑀏𑀢𑀤𑁂𑀯 𑀳𑀺 𑀫𑁂 𑀦𑀸𑀫𑀁,_  
_𑀬𑀁 𑀦𑀸𑀫𑀲𑁆𑀫𑀺 𑀭𑀣𑁂𑀲𑀪;_  
_𑀆𑀕𑀫𑁂𑀳𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀧𑀺𑀢𑀭𑀁 𑀆𑀫𑀦𑁆𑀢𑀬𑀸𑀫𑀳𑀦𑁆”𑀢𑀺𑁇_  


---

1355. 𑀆𑀲𑀗𑁆𑀓𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
