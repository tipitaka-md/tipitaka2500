# 𑁪𑁇𑁧𑁇𑁬 𑀲𑀼𑀚𑀸𑀢𑀸𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀚𑀸𑀢𑀸𑀚𑀸𑀢𑀓

(21.)

908\. _“𑀓𑀺𑀫𑀡𑁆𑀟𑀓𑀸 𑀇𑀫𑁂 𑀤𑁂𑀯,_  
_𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀓𑀁𑀲𑀫𑀮𑁆𑀮𑀓𑁂;_  
_𑀉𑀧𑀮𑁄𑀳𑀺𑀢𑀓𑀸 𑀯𑀕𑁆𑀕𑀽,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄”𑁇_  


(22.)

909\. _“𑀬𑀸𑀦𑀺 𑀧𑀼𑀭𑁂 𑀢𑀼𑀯𑀁 𑀤𑁂𑀯𑀺,_  
_𑀪𑀡𑁆𑀟𑀼 𑀦𑀦𑁆𑀢𑀓𑀯𑀸𑀲𑀺𑀦𑀻;_  
_𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑀳𑀢𑁆𑀣𑀸 𑀧𑀘𑀺𑀦𑀸𑀲𑀺,_  
_𑀢𑀲𑁆𑀲𑀸 𑀢𑁂 𑀓𑁄𑀮𑀺𑀬𑀁 𑀨𑀮𑀁𑁇_  


(23.)

910\. _𑀉𑀟𑁆𑀟𑀬𑁆𑀳𑀢𑁂 𑀦 𑀭𑀫𑀢𑀺,_  
_𑀪𑁄𑀕𑀸 𑀯𑀺𑀧𑁆𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑀁;_  
_𑀢𑀢𑁆𑀣𑁂𑀯𑀺𑀫𑀁 𑀧𑀝𑀺𑀦𑁂𑀣,_  
_𑀬𑀢𑁆𑀣 𑀓𑁄𑀮𑀁 𑀧𑀘𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(24.)

911\. _“𑀳𑁄𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀇𑀤𑁆𑀥𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀬 𑀦𑀸𑀭𑀺𑀬𑀸;_  
_𑀔𑀫 𑀤𑁂𑀯 𑀲𑀼𑀚𑀸𑀢𑀸𑀬,_  
_𑀫𑀸𑀲𑁆𑀲𑀸 𑀓𑀼𑀚𑁆𑀛 𑀭𑀣𑁂𑀲𑀪𑀸”𑀢𑀺𑁇_  


---
