# 𑁪𑁇𑁧𑁇𑁭 𑀧𑀮𑀸𑀲𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀧𑀮𑀸𑀲𑀚𑀸𑀢𑀓

(25.)

913. _“𑀅𑀘𑁂𑀢𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀅𑀲𑁆𑀲𑀼𑀡𑀦𑁆𑀢𑀁,_  
_𑀚𑀸𑀦𑁄 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀫𑀺𑀫𑀁 𑀧𑀮𑀸𑀲𑀁;_  
_𑀆𑀭𑀤𑁆𑀥𑀯𑀺𑀭𑀺𑀬𑁄 𑀥𑀼𑀯𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄,_  
_𑀲𑀼𑀔𑀲𑁂𑀬𑁆𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑀲𑀺 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼”𑁇_  


(26.)

914. _“𑀤𑀽𑀭𑁂 𑀲𑀼𑀢𑁄 𑀘𑁂𑀯 𑀩𑁆𑀭𑀳𑀸 𑀘 𑀭𑀼𑀓𑁆𑀔𑁄,_  
_𑀤𑁂𑀲𑁂 𑀞𑀺𑀢𑁄 𑀪𑀽𑀢𑀦𑀺𑀯𑀸𑀲𑀭𑀽𑀧𑁄;_  
_𑀢𑀲𑁆𑀫𑀸 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺 𑀇𑀫𑀁 𑀧𑀮𑀸𑀲𑀁,_  
_𑀬𑁂 𑀘𑁂𑀢𑁆𑀣 𑀪𑀽𑀢𑀸 𑀢𑁂 𑀥𑀦𑀲𑁆𑀲 𑀳𑁂𑀢𑀼”𑁇_  


(27.)

915. _“𑀲𑁄 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀣𑀸𑀦𑀼𑀪𑀸𑀯𑀁,_  
_𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀧𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑁄;_  
_𑀓𑀣𑀜𑁆𑀳𑀺 𑀆𑀕𑀫𑁆𑀫 𑀲𑀢𑀁 𑀲𑀓𑀸𑀲𑁂,_  
_𑀫𑁄𑀖𑀸𑀦𑀺 𑀢𑁂 𑀅𑀲𑁆𑀲𑀼 𑀧𑀭𑀺𑀨𑀦𑁆𑀤𑀺𑀢𑀸𑀦𑀺𑁇_  


(28.)

916. _𑀬𑁄 𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀭𑁄 𑀧𑀺𑀮𑀓𑁆𑀔𑁄,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄 𑀧𑀼𑀩𑁆𑀩𑀬𑀜𑁆𑀜𑁄 𑀉𑀍𑀆𑀭𑁄;_  
_𑀢𑀲𑁆𑀲𑁂𑀲 𑀫𑀽𑀮𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀥𑀺 𑀦𑀺𑀔𑀸𑀢𑁄,_  
_𑀅𑀤𑀸𑀬𑀸𑀤𑁄 𑀕𑀘𑁆𑀙 𑀢𑀁 𑀉𑀤𑁆𑀥𑀭𑀸𑀳𑀻”𑀢𑀺𑁇_  


---
