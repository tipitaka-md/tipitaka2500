# 𑁪𑁇𑁧𑁇𑁫 𑀲𑀻𑀮𑀯𑀻𑀫𑀁𑀲𑀦𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀮𑀯𑀻𑀫𑀁𑀲𑀦𑀚𑀸𑀢𑀓

(17.)

903\. _“𑀦𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀭𑀳𑁄 𑀦𑀸𑀫,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀼𑀩𑁆𑀩𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀦𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀢𑀁 𑀩𑀸𑀮𑁄 𑀫𑀜𑁆𑀜𑀢𑀻 𑀭𑀳𑁄𑁇_  


(18.)

904\. _𑀅𑀳𑀁 𑀭𑀳𑁄 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀯𑀸𑀧𑀺 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀬𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀁 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀅𑀲𑀼𑀜𑁆𑀜𑀁 𑀳𑁄𑀢𑀺 𑀢𑀁 𑀫𑀬𑀸”𑁇_  


(19.)

905\. _“𑀤𑀼𑀚𑁆𑀚𑀘𑁆𑀘𑁄 𑀘 𑀲𑀼𑀚𑀘𑁆𑀘𑁄 𑀘,_  
_𑀦𑀦𑁆𑀤𑁄 𑀘 𑀲𑀼𑀔𑀯𑀟𑁆𑀠𑀺𑀢𑁄;_  
_𑀯𑁂𑀚𑁆𑀚𑁄 𑀘 𑀅𑀤𑁆𑀥𑀼𑀯𑀲𑀻𑀮𑁄 𑀘,_  
_𑀢𑁂 𑀥𑀫𑁆𑀫𑀁 𑀚𑀳𑀼 𑀫𑀢𑁆𑀣𑀺𑀓𑀸𑁇_  


(20.)

906\. _𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀘 𑀓𑀣𑀁 𑀚𑀳𑁂,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀬𑁄 𑀥𑀫𑁆𑀫𑀫𑀦𑀼𑀧𑀸𑀮𑁂𑀢𑀺,_  
_𑀥𑀺𑀢𑀺𑀫𑀸 𑀲𑀘𑁆𑀘𑀦𑀺𑀓𑁆𑀓𑀫𑁄”𑀢𑀺𑁇_  


---

907\. 𑀲𑀻𑀮𑀯𑀻𑀫𑀁𑀲𑀦𑀚𑀸𑀢𑀓𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
