# 𑁪𑁇𑁧𑁇𑁧𑁦 𑀲𑁂𑀬𑁆𑀬𑀚𑀸𑀢𑀓

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀲𑁂𑀬𑁆𑀬𑀚𑀸𑀢𑀓

(37.)

928\. _“𑀲𑀲𑀫𑀼𑀤𑁆𑀤𑀧𑀭𑀺𑀬𑀸𑀬𑀁,_  
_𑀫𑀳𑀺𑀁 𑀲𑀸𑀕𑀭𑀓𑀼𑀡𑁆𑀟𑀮𑀁;_  
_𑀦 𑀇𑀘𑁆𑀙𑁂 𑀲𑀳 𑀦𑀺𑀦𑁆𑀤𑀸𑀬,_  
_𑀏𑀯𑀁 𑀲𑁂𑀬𑁆𑀬 𑀯𑀺𑀚𑀸𑀦𑀳𑀺𑁇_  


(38.)

929\. _𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀢𑀁 𑀬𑀲𑀮𑀸𑀪𑀁,_  
_𑀥𑀦𑀮𑀸𑀪𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀺 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑁂𑀦,_  
_𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀡𑁂𑀦 𑀯𑀸𑁇_  


(39.)

930\. _𑀅𑀧𑀺 𑀘𑁂 𑀧𑀢𑁆𑀢𑀫𑀸𑀤𑀸𑀬,_  
_𑀅𑀦𑀕𑀸𑀭𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂;_  
_𑀲𑀸𑀬𑁂𑀯 𑀚𑀻𑀯𑀺𑀓𑀸 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑀸 𑀘𑀸𑀥𑀫𑁆𑀫𑁂𑀦 𑀏𑀲𑀦𑀸𑁇_  


(40.)

931\. _𑀅𑀧𑀺 𑀘𑁂 𑀧𑀢𑁆𑀢𑀫𑀸𑀤𑀸𑀬,_  
_𑀅𑀦𑀕𑀸𑀭𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂;_  
_𑀅𑀜𑁆𑀜𑀁 𑀅𑀳𑀺𑀁𑀲𑀬𑀁 𑀮𑁄𑀓𑁂,_  
_𑀅𑀧𑀺 𑀭𑀚𑁆𑀚𑁂𑀦 𑀢𑀁 𑀯𑀭𑀦𑁆”𑀢𑀺𑁇_  


---

932\. 𑀲𑁂𑀬𑁆𑀬𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

933\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



934\. _𑀯𑀺𑀯𑀭𑀜𑁆𑀘 𑀅𑀤𑁂𑀬𑁆𑀬 𑀲𑀫𑀺𑀤𑁆𑀥𑀯𑀭𑀁,_  
_𑀅𑀣 𑀤𑀤𑁆𑀤𑀭 𑀧𑀸𑀧𑀫𑀳𑀸𑀢𑀺𑀭𑀳𑁄;_  
_𑀅𑀣 𑀓𑁄𑀮𑀺 𑀧𑀮𑀸𑀲𑀯𑀭𑀜𑁆𑀘 𑀓𑀭,_  
_𑀘𑀭𑀺𑀫𑀁 𑀲𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀭𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
