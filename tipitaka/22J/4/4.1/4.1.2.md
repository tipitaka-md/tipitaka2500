# 𑁪𑁇𑁧𑁇𑁨 𑀫𑀳𑀸𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀚𑀸𑀢𑀓

(5.)

888. _“𑀅𑀤𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀤𑀤𑀁 𑀤𑀸𑀦𑀁,_  
_𑀤𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀦𑀧𑁆𑀧𑀯𑁂𑀘𑁆𑀙𑀢𑀺;_  
_𑀆𑀧𑀸𑀲𑀼 𑀩𑁆𑀬𑀲𑀦𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀳𑀸𑀬𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(6.)

889. _𑀦𑀸𑀤𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀤𑀤𑀁 𑀤𑀸𑀦𑀁,_  
_𑀤𑁂𑀬𑁆𑀬𑁂𑀲𑀼 𑀬𑁄 𑀧𑀯𑁂𑀘𑁆𑀙𑀢𑀺;_  
_𑀆𑀧𑀸𑀲𑀼 𑀩𑁆𑀬𑀲𑀦𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀳𑀸𑀬𑀫𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(7.)

890. _𑀲𑀜𑁆𑀜𑁄𑀕𑀲𑀫𑁆𑀪𑁄𑀕𑀯𑀺𑀲𑁂𑀲𑀤𑀲𑁆𑀲𑀦𑀁,_  
_𑀅𑀦𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀞𑁂𑀲𑀼 𑀦𑀲𑁆𑀲𑀢𑀺;_  
_𑀓𑀢𑀜𑁆𑀘 𑀅𑀭𑀺𑀬𑁂𑀲𑀼 𑀘 𑀅𑀚𑁆𑀚𑀯𑁂𑀲𑀼,_  
_𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀅𑀡𑀼𑀫𑁆𑀧𑀺 𑀢𑀸𑀤𑀺𑀲𑀼𑁇_  


(8.)

891. _𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀓𑀮𑁆𑀬𑀸𑀡𑁄,_  
_𑀅𑀓𑀸 𑀮𑁄𑀓𑁂 𑀲𑀼𑀤𑀼𑀓𑁆𑀓𑀭𑀁;_  
_𑀧𑀘𑁆𑀙𑀸 𑀓𑀬𑀺𑀭𑀸 𑀦 𑀯𑀸 𑀓𑀬𑀺𑀭𑀸,_  
_𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀁 𑀧𑀽𑀚𑀦𑀸𑀭𑀳𑁄”𑀢𑀺𑁇_  


---

892. 𑀫𑀳𑀸𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
