# 𑁪𑁇𑁧𑁇𑁮 𑀲𑀓𑀼𑀡𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀓𑀼𑀡𑀚𑀸𑀢𑀓

(29.)

918. _“𑀅𑀓𑀭𑀫𑁆𑀳𑀲 𑀢𑁂 𑀓𑀺𑀘𑁆𑀘𑀁,_  
_𑀬𑀁 𑀩𑀮𑀁 𑀅𑀳𑀼𑀯𑀫𑁆𑀳𑀲𑁂;_  
_𑀫𑀺𑀕𑀭𑀸𑀚 𑀦𑀫𑁄 𑀢𑁆𑀬𑀢𑁆𑀣𑀼,_  
_𑀅𑀧𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀮𑀪𑀸𑀫𑀲𑁂”𑁇_  


(30.)

919. _“𑀫𑀫 𑀮𑁄𑀳𑀺𑀢𑀪𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀮𑀼𑀤𑁆𑀤𑀸𑀦𑀺 𑀓𑀼𑀩𑁆𑀩𑀢𑁄;_  
_𑀤𑀦𑁆𑀢𑀦𑁆𑀢𑀭𑀕𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀢𑀁 𑀩𑀳𑀼𑀁 𑀬𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀲𑀺”𑁇_  


(31.)

920. _“𑀅𑀓𑀢𑀜𑁆𑀜𑀼𑀫𑀓𑀢𑁆𑀢𑀸𑀭𑀁,_  
_𑀓𑀢𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀸𑀭𑀓𑀁;_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀦𑀺𑀭𑀢𑁆𑀣𑀸 𑀢𑀲𑁆𑀲 𑀲𑁂𑀯𑀦𑀸𑁇_  


(32.)

921. _𑀬𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑀘𑀺𑀡𑁆𑀡𑁂𑀦,_  
_𑀫𑀺𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁄 𑀦 𑀮𑀩𑁆𑀪𑀢𑀺;_  
_𑀅𑀦𑀼𑀲𑀽𑀬𑀫𑀦𑀓𑁆𑀓𑁄𑀲𑀁,_  
_𑀲𑀡𑀺𑀓𑀁 𑀢𑀫𑁆𑀳𑀸 𑀅𑀧𑀓𑁆𑀓𑀫𑁂”𑀢𑀺𑁇_  


---

922. 𑀲𑀓𑀼𑀡𑀚𑀸𑀢𑀓𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
