# 𑁪𑁇𑁪𑁇𑁩 𑀧𑀓𑁆𑀓𑀕𑁄𑀥𑀚𑀸𑀢𑀓

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑁄𑀓𑀺𑀮𑀯𑀕𑁆𑀕

* 𑀧𑀓𑁆𑀓𑀕𑁄𑀥𑀚𑀸𑀢𑀓

(129.)

1049\. _“𑀢𑀤𑁂𑀯 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁄,_  
_𑀯𑀦𑀫𑀚𑁆𑀛𑁂 𑀭𑀣𑁂𑀲𑀪;_  
_𑀬𑀲𑁆𑀲 𑀢𑁂 𑀔𑀕𑁆𑀕𑀩𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀲𑁆𑀲 𑀢𑀺𑀭𑀻𑀝𑀺𑀦𑁄;_  
_𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀤𑀼𑀫𑀲𑀸𑀔𑀸𑀬,_  
_𑀧𑀓𑁆𑀓𑀸 𑀕𑁄𑀥𑀸 𑀧𑀮𑀸𑀬𑀣”𑁇_  


(130.)

1050\. _“𑀦𑀫𑁂 𑀦𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀚𑁂 𑀪𑀚𑀦𑁆𑀢𑀁,_  
_𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀓𑀼𑀩𑁆𑀩𑀲𑁆𑀲 𑀓𑀭𑁂𑀬𑁆𑀬 𑀓𑀺𑀘𑁆𑀘𑀁;_  
_𑀦𑀸𑀦𑀢𑁆𑀣𑀓𑀸𑀫𑀲𑁆𑀲 𑀓𑀭𑁂𑀬𑁆𑀬 𑀅𑀢𑁆𑀣𑀁,_  
_𑀅𑀲𑀫𑁆𑀪𑀚𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀦 𑀲𑀫𑁆𑀪𑀚𑁂𑀬𑁆𑀬𑁇_  


(131.)

1051\. _𑀘𑀚𑁂 𑀘𑀚𑀦𑁆𑀢𑀁 𑀯𑀦𑀣𑀁 𑀦 𑀓𑀬𑀺𑀭𑀸,_  
_𑀅𑀧𑁂𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀦 𑀲𑀫𑁆𑀪𑀚𑁂𑀬𑁆𑀬;_  
_𑀤𑀺𑀚𑁄 𑀤𑀼𑀫𑀁 𑀔𑀻𑀡𑀨𑀮𑀦𑁆𑀢𑀺 𑀜𑀢𑁆𑀯𑀸,_  
_𑀅𑀜𑁆𑀜𑀁 𑀲𑀫𑁂𑀓𑁆𑀔𑁂𑀬𑁆𑀬 𑀫𑀳𑀸 𑀳𑀺 𑀮𑁄𑀓𑁄”𑁇_  


(132.)

1052\. _“𑀲𑁄 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀣𑀸𑀦𑀼𑀪𑀸𑀯𑀁,_  
_𑀓𑀢𑀜𑁆𑀜𑀼𑀢𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀧𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀢𑁂 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀤𑀤𑀸𑀫𑀺,_  
_𑀬𑀲𑁆𑀲𑀺𑀘𑁆𑀙𑀲𑀻 𑀢𑀲𑁆𑀲 𑀢𑀼𑀯𑀁 𑀤𑀤𑀸𑀫𑀻”𑀢𑀺𑁇_  


---
