# 𑁪𑁇𑁪𑁇𑁯 𑀩𑀸𑀯𑁂𑀭𑀼𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑁄𑀓𑀺𑀮𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀯𑁂𑀭𑀼𑀚𑀸𑀢𑀓

(153.)

1079. _“𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀫𑁄𑀭𑀲𑁆𑀲,_  
_𑀲𑀺𑀔𑀺𑀦𑁄 𑀫𑀜𑁆𑀚𑀼𑀪𑀸𑀡𑀺𑀦𑁄;_  
_𑀓𑀸𑀓𑀁 𑀢𑀢𑁆𑀣 𑀅𑀧𑀽𑀚𑁂𑀲𑀼𑀁,_  
_𑀫𑀁𑀲𑁂𑀦 𑀘 𑀨𑀮𑁂𑀦 𑀘𑁇_  


(154.)

1080. _𑀬𑀤𑀸 𑀘 𑀲𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀫𑁄𑀭𑁄 𑀩𑀸𑀯𑁂𑀭𑀼𑀫𑀸𑀕𑀫𑀸;_  
_𑀅𑀣 𑀮𑀸𑀪𑁄 𑀘 𑀲𑀓𑁆𑀓𑀸𑀭𑁄,_  
_𑀯𑀸𑀬𑀲𑀲𑁆𑀲 𑀅𑀳𑀸𑀬𑀣𑁇_  


(155.)

1081. _𑀬𑀸𑀯 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀧𑀪𑀗𑁆𑀓𑀭𑁄;_  
_𑀢𑀸𑀯 𑀅𑀜𑁆𑀜𑁂 𑀅𑀧𑀽𑀚𑁂𑀲𑀼𑀁,_  
_𑀧𑀼𑀣𑀽 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑁇_  


(156.)

1082. _𑀬𑀤𑀸 𑀘 𑀲𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀁 𑀅𑀤𑁂𑀲𑀬𑀺;_  
_𑀅𑀣 𑀮𑀸𑀪𑁄 𑀘 𑀲𑀓𑁆𑀓𑀸𑀭𑁄,_  
_𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀅𑀳𑀸𑀬𑀣𑀸”𑀢𑀺𑁇_  


---
