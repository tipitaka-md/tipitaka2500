# 𑁪𑁇𑁫𑁇𑁯 𑀲𑀦𑁆𑀥𑀺𑀪𑁂𑀤𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀽𑀍𑀅𑀓𑀼𑀡𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀲𑀦𑁆𑀥𑀺𑀪𑁂𑀤𑀚𑀸𑀢𑀓

(193.)

1131. _“𑀦𑁂𑀯 𑀇𑀢𑁆𑀣𑀻𑀲𑀼 𑀲𑀸𑀫𑀜𑁆𑀜𑀁,_  
_𑀦𑀸𑀧𑀺 𑀪𑀓𑁆𑀔𑁂𑀲𑀼 𑀲𑀸𑀭𑀣𑀺;_  
_𑀅𑀣𑀲𑁆𑀲 𑀲𑀦𑁆𑀥𑀺𑀪𑁂𑀤𑀲𑁆𑀲,_  
_𑀧𑀲𑁆𑀲 𑀬𑀸𑀯 𑀲𑀼𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀁𑁇_  


(194.)

1132. _𑀅𑀲𑀺 𑀢𑀺𑀓𑁆𑀔𑁄𑀯 𑀫𑀁𑀲𑀫𑁆𑀳𑀺,_  
_𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀬𑀢𑁆𑀣𑀽𑀲𑀪𑀜𑁆𑀘 𑀲𑀻𑀳𑀜𑁆𑀘,_  
_𑀪𑀓𑁆𑀔𑀬𑀦𑁆𑀢𑀺 𑀫𑀺𑀕𑀸𑀥𑀫𑀸𑁇_  


(195.)

1133. _𑀇𑀫𑀁 𑀲𑁄 𑀲𑀬𑀦𑀁 𑀲𑁂𑀢𑀺,_  
_𑀬𑀫𑀺𑀫𑀁 𑀧𑀲𑁆𑀲𑀲𑀺 𑀲𑀸𑀭𑀣𑀺;_  
_𑀬𑁄 𑀯𑀸𑀘𑀁 𑀲𑀦𑁆𑀥𑀺𑀪𑁂𑀤𑀲𑁆𑀲,_  
_𑀧𑀺𑀲𑀼𑀡𑀲𑁆𑀲 𑀦𑀺𑀩𑁄𑀥𑀢𑀺𑁇_  


(196.)

1134. _𑀢𑁂 𑀚𑀦𑀸 𑀲𑀼𑀔𑀫𑁂𑀥𑀦𑁆𑀢𑀺,_  
_𑀦𑀭𑀸 𑀲𑀕𑁆𑀕𑀕𑀢𑀸𑀭𑀺𑀯;_  
_𑀬𑁂 𑀯𑀸𑀘𑀁 𑀲𑀦𑁆𑀥𑀺𑀪𑁂𑀤𑀲𑁆𑀲,_  
_𑀦𑀸𑀯𑀩𑁄𑀥𑀦𑁆𑀢𑀺 𑀲𑀸𑀭𑀣𑀻”𑀢𑀺𑁇_  


---

1135. 𑀲𑀦𑁆𑀥𑀺𑀪𑁂𑀤𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇
