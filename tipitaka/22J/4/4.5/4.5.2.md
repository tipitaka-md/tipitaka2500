# 𑁪𑁇𑁫𑁇𑁨 𑀯𑀸𑀦𑀭𑀚𑀸𑀢𑀓

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀘𑀽𑀍𑀅𑀓𑀼𑀡𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀯𑀸𑀦𑀭𑀚𑀸𑀢𑀓

(165.)

1096\. _“𑀅𑀲𑀓𑁆𑀔𑀺𑀁 𑀯𑀢 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀉𑀤𑁆𑀥𑀸𑀢𑀼𑀁 𑀉𑀤𑀓𑀸 𑀣𑀮𑀁;_  
_𑀦 𑀤𑀸𑀦𑀸𑀳𑀁 𑀧𑀼𑀦 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀯𑀲𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀯𑀸𑀭𑀺𑀚𑁇_  


(166.)

1097\. _𑀅𑀮𑀫𑁂𑀢𑁂𑀳𑀺 𑀅𑀫𑁆𑀩𑁂𑀳𑀺,_  
_𑀚𑀫𑁆𑀩𑀽𑀳𑀺 𑀧𑀦𑀲𑁂𑀳𑀺 𑀘;_  
_𑀬𑀸𑀦𑀺 𑀧𑀸𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲,_  
_𑀯𑀭𑀁 𑀫𑀬𑁆𑀳𑀁 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑁄𑁇_  


(167.)

1098\. _𑀬𑁄 𑀘 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀦 𑀔𑀺𑀧𑁆𑀧𑀫𑀦𑀼𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀯𑀲𑀫𑀦𑁆𑀯𑁂𑀢𑀺,_  
_𑀧𑀘𑁆𑀙𑀸 𑀘 𑀅𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀺𑁇_  


(168.)

1099\. _𑀬𑁄 𑀘 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀩𑁄𑀥𑀢𑀺;_  
_𑀫𑀼𑀘𑁆𑀘𑀢𑁂 𑀲𑀢𑁆𑀢𑀼𑀲𑀫𑁆𑀩𑀸𑀥𑀸,_  
_𑀦 𑀘 𑀧𑀘𑁆𑀙𑀸𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇_  


---
