# 𑁪𑁇𑁫𑁇𑁪 𑀅𑀫𑁆𑀩𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀽𑀍𑀅𑀓𑀼𑀡𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀅𑀫𑁆𑀩𑀚𑀸𑀢𑀓

(173.)

1106\. _“𑀬𑁄 𑀦𑀻𑀮𑀺𑀬𑀁 𑀫𑀡𑁆𑀟𑀬𑀢𑀺,_  
_𑀲𑀡𑁆𑀟𑀸𑀲𑁂𑀦 𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀺;_  
_𑀢𑀲𑁆𑀲 𑀲𑀸 𑀯𑀲𑀫𑀦𑁆𑀯𑁂𑀢𑀼,_  
_𑀬𑀸 𑀢𑁂 𑀅𑀫𑁆𑀩𑁂 𑀅𑀯𑀸𑀳𑀭𑀺”𑁇_  


(174.)

1107\. _“𑀯𑀻𑀲𑀁 𑀯𑀸 𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀁 𑀯𑀸,_  
_𑀊𑀦𑀢𑀺𑀁𑀲𑀁 𑀯 𑀚𑀸𑀢𑀺𑀬𑀸;_  
_𑀢𑀸𑀤𑀺𑀲𑀸 𑀧𑀢𑀺 𑀫𑀸 𑀮𑀤𑁆𑀥𑀸,_  
_𑀬𑀸 𑀢𑁂 𑀅𑀫𑁆𑀩𑁂 𑀅𑀯𑀸𑀳𑀭𑀺”𑁇_  


(175.)

1108\. _“𑀤𑀻𑀖𑀁 𑀕𑀘𑁆𑀙𑀢𑀼 𑀅𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀏𑀓𑀺𑀓𑀸 𑀅𑀪𑀺𑀲𑀸𑀭𑀺𑀓𑀸;_  
_𑀲𑀗𑁆𑀓𑁂𑀢𑁂 𑀧𑀢𑀺 𑀫𑀸 𑀅𑀤𑁆𑀤,_  
_𑀬𑀸 𑀢𑁂 𑀅𑀫𑁆𑀩𑁂 𑀅𑀯𑀸𑀳𑀭𑀺”𑁇_  


(176.)

1109\. _“𑀅𑀮𑀗𑁆𑀓𑀢𑀸 𑀲𑀼𑀯𑀲𑀦𑀸,_  
_𑀫𑀸𑀮𑀺𑀦𑀻 𑀘𑀦𑁆𑀤𑀦𑀼𑀲𑁆𑀲𑀤𑀸;_  
_𑀏𑀓𑀺𑀓𑀸 𑀲𑀬𑀦𑁂 𑀲𑁂𑀢𑀼,_  
_𑀬𑀸 𑀢𑁂 𑀅𑀫𑁆𑀩𑁂 𑀅𑀯𑀸𑀳𑀭𑀻”𑀢𑀺𑁇_  


---
