# 𑁪𑁇𑁩𑁇𑁮 𑀅𑀦𑀦𑀼𑀲𑁄𑀘𑀺𑀬𑀚𑀸𑀢𑀓

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀝𑀺𑀤𑀽𑀲𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀦𑀼𑀲𑁄𑀘𑀺𑀬𑀚𑀸𑀢𑀓

(109.)

1022\. _“𑀩𑀳𑀽𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻 𑀪𑁄𑀢𑀻,_  
_𑀢𑁂𑀳𑀺 𑀫𑁂 𑀓𑀺𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀢𑀁 𑀦 𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀧𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀺𑀮𑁆𑀮𑀳𑀸𑀲𑀺𑀦𑀺𑀁𑁇_  


(110.)

1023\. _𑀢𑀁 𑀢𑀜𑁆𑀘𑁂 𑀅𑀦𑀼𑀲𑁄𑀘𑁂𑀬𑁆𑀬,_  
_𑀬𑀁 𑀬𑀁 𑀢𑀲𑁆𑀲 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀫𑀦𑀼𑀲𑁄𑀘𑁂𑀬𑁆𑀬,_  
_𑀲𑀤𑀸 𑀫𑀘𑁆𑀘𑀼𑀯𑀲𑀁 𑀧𑀢𑀁𑁇_  


(111.)

1024\. _𑀦 𑀳𑁂𑀯 𑀞𑀺𑀢𑀁 𑀦𑀸𑀲𑀻𑀦𑀁,_  
_𑀦 𑀲𑀬𑀸𑀦𑀁 𑀦 𑀧𑀤𑁆𑀥𑀕𑀼𑀁;_  
_𑀬𑀸𑀯 𑀩𑁆𑀬𑀸𑀢𑀺 𑀦𑀺𑀫𑀺𑀲𑀢𑀺,_  
_𑀢𑀢𑁆𑀭𑀸𑀧𑀺 𑀭𑀲𑀢𑀻 𑀯𑀬𑁄𑁇_  


(112.)

1025\. _𑀢𑀢𑁆𑀣𑀢𑁆𑀢𑀦𑀺 𑀯𑀢𑀧𑁆𑀧𑀤𑁆𑀥𑁂,_  
_𑀯𑀺𑀦𑀸𑀪𑀸𑀯𑁂 𑀅𑀲𑀁𑀲𑀬𑁂;_  
_𑀪𑀽𑀢𑀁 𑀲𑁂𑀲𑀁 𑀤𑀬𑀺𑀢𑀩𑁆𑀩𑀁,_  
_𑀯𑀻𑀢𑀁 𑀅𑀦𑀦𑀼𑀲𑁄𑀘𑀺𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

1026\. 𑀅𑀦𑀦𑀼𑀲𑁄𑀘𑀺𑀬𑀚𑀸𑀢𑀓𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
