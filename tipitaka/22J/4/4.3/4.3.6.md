# 𑁪𑁇𑁩𑁇𑁬 𑀓𑀓𑁆𑀓𑀸𑀭𑀼𑀚𑀸𑀢𑀓

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀝𑀺𑀤𑀽𑀲𑀓𑀯𑀕𑁆𑀕

* 𑀓𑀓𑁆𑀓𑀸𑀭𑀼𑀚𑀸𑀢𑀓

(101.)

1012\. _“𑀓𑀸𑀬𑁂𑀦 𑀬𑁄 𑀦𑀸𑀯𑀳𑀭𑁂,_  
_𑀯𑀸𑀘𑀸𑀬 𑀦 𑀫𑀼𑀲𑀸 𑀪𑀡𑁂;_  
_𑀬𑀲𑁄 𑀮𑀤𑁆𑀥𑀸 𑀦 𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀲 𑀯𑁂 𑀓𑀓𑁆𑀓𑀸𑀭𑀼𑀫𑀭𑀳𑀢𑀺”𑁇_  


(102.)

1013\. _“𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀢𑁆𑀢𑀫𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀦 𑀦𑀺𑀓𑀢𑁆𑀬𑀸 𑀥𑀦𑀁 𑀳𑀭𑁂;_  
_𑀪𑁄𑀕𑁂 𑀮𑀤𑁆𑀥𑀸 𑀦 𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀲 𑀯𑁂 𑀓𑀓𑁆𑀓𑀸𑀭𑀼𑀫𑀭𑀳𑀢𑀺”𑁇_  


(103.)

1014\. _“𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀳𑀸𑀮𑀺𑀤𑁆𑀤𑀁,_  
_𑀲𑀤𑁆𑀥𑀸 𑀘 𑀅𑀯𑀺𑀭𑀸𑀕𑀺𑀦𑀻;_  
_𑀏𑀓𑁄 𑀲𑀸𑀤𑀼𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀲 𑀯𑁂 𑀓𑀓𑁆𑀓𑀸𑀭𑀼𑀫𑀭𑀳𑀢𑀺”𑁇_  


(104.)

1015\. _“𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀯𑀸 𑀢𑀺𑀭𑁄𑀓𑁆𑀔𑀸 𑀯𑀸,_  
_𑀬𑁄 𑀲𑀦𑁆𑀢𑁂 𑀦 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺;_  
_𑀬𑀣𑀸𑀯𑀸𑀤𑀻 𑀢𑀣𑀸𑀓𑀸𑀭𑀻,_  
_𑀲 𑀯𑁂 𑀓𑀓𑁆𑀓𑀸𑀭𑀼𑀫𑀭𑀳𑀢𑀻”𑀢𑀺𑁇_  


---
