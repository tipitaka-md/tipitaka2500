# 𑁪𑁇𑁩𑁇𑁧𑁦 𑀲𑀻𑀮𑀯𑀻𑀫𑀁𑀲𑀚𑀸𑀢𑀓

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀝𑀺𑀤𑀽𑀲𑀓𑀯𑀕𑁆𑀕

* 𑀲𑀻𑀮𑀯𑀻𑀫𑀁𑀲𑀚𑀸𑀢𑀓

(117.)

1032\. _“𑀲𑀻𑀮𑀁 𑀓𑀺𑀭𑁂𑀯 𑀓𑀮𑁆𑀬𑀸𑀡𑀁,_  
_𑀲𑀻𑀮𑀁 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀧𑀲𑁆𑀲 𑀖𑁄𑀭𑀯𑀺𑀲𑁄 𑀦𑀸𑀕𑁄,_  
_𑀲𑀻𑀮𑀯𑀸𑀢𑀺 𑀦 𑀳𑀜𑁆𑀜𑀢𑀺”𑁇_  


(118.)

1033\. _“𑀬𑀸𑀯𑀤𑁂𑀯𑀲𑁆𑀲𑀳𑀽 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀢𑀸𑀯𑀤𑁂𑀯 𑀅𑀔𑀸𑀤𑀺𑀲𑀼𑀁;_  
_𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀓𑀼𑀮𑀮𑀸 𑀮𑁄𑀓𑁂,_  
_𑀦 𑀳𑀺𑀁𑀲𑀦𑁆𑀢𑀺 𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑀁𑁇_  


(119.)

1034\. _𑀲𑀼𑀔𑀁 𑀦𑀺𑀭𑀸𑀲𑀸 𑀲𑀼𑀧𑀢𑀺,_  
_𑀆𑀲𑀸 𑀨𑀮𑀯𑀢𑀻 𑀲𑀼𑀔𑀸;_  
_𑀆𑀲𑀁 𑀦𑀺𑀭𑀸𑀲𑀁 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀼𑀔𑀁 𑀲𑀼𑀧𑀢𑀺 𑀧𑀺𑀗𑁆𑀕𑀮𑀸𑁇_  


(120.)

1035\. _𑀦 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀫𑁆𑀳𑀺 𑀘;_  
_𑀦 𑀧𑀭𑀁 𑀦𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀯𑀺𑀳𑀺𑀁𑀲𑀢𑀺 𑀲𑀫𑀸𑀳𑀺𑀢𑁄”𑀢𑀺𑁇_  


---

1036\. 𑀲𑀻𑀮𑀯𑀻𑀫𑀁𑀲𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇

  
𑀓𑀼𑀝𑀺𑀤𑀽𑀲𑀓𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

1037\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1038\. _𑀲𑀫𑀦𑀼𑀲𑁆𑀲 𑀲𑀤𑀼𑀤𑁆𑀤𑀼𑀪 𑀬𑀸𑀘𑀦𑀓𑁄,_  
_𑀅𑀣 𑀫𑁂𑀡𑁆𑀟𑀯𑀭𑀼𑀢𑁆𑀢𑀫 𑀕𑁄𑀥𑀯𑀭𑁄;_  
_𑀅𑀣 𑀓𑀸𑀬𑀲𑀓𑁂𑀧𑀼𑀓 𑀪𑁄𑀢𑀻𑀯𑀭𑁄,_  
_𑀅𑀣 𑀭𑀸𑀥𑀲𑀼𑀲𑀻𑀮𑀯𑀭𑁂𑀦 𑀤𑀲𑀸𑀢𑀺𑁇_
