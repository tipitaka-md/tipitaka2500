# 𑁪𑁇𑁩𑁇𑁩 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀓𑀼𑀝𑀺𑀤𑀽𑀲𑀓𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀚𑀸𑀢𑀓

(89.)

997. _“𑀤𑁆𑀯𑀬𑀁 𑀬𑀸𑀘𑀦𑀓𑁄 𑀭𑀸𑀚,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀅𑀮𑀸𑀪𑀁 𑀥𑀦𑀮𑀸𑀪𑀁 𑀯𑀸,_  
_𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀸 𑀳𑀺 𑀬𑀸𑀘𑀦𑀸𑁇_  


(90.)

998. _𑀬𑀸𑀘𑀦𑀁 𑀭𑁄𑀤𑀦𑀁 𑀆𑀳𑀼,_  
_𑀧𑀜𑁆𑀘𑀸𑀮𑀸𑀦𑀁 𑀭𑀣𑁂𑀲𑀪;_  
_𑀬𑁄 𑀬𑀸𑀘𑀦𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀺,_  
_𑀢𑀫𑀸𑀳𑀼 𑀧𑀝𑀺𑀭𑁄𑀤𑀦𑀁𑁇_  


(91.)

999. _𑀫𑀸 𑀫𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀭𑁄𑀤𑀦𑁆𑀢𑀁,_  
_𑀧𑀜𑁆𑀘𑀸𑀮𑀸 𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀢𑀼𑀯𑀁 𑀯𑀸 𑀧𑀝𑀺𑀭𑁄𑀤𑀦𑁆𑀢𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀭𑀳𑁄”𑁇_  


(92.)

1000. _“𑀤𑀤𑀸𑀫𑀺 𑀢𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀭𑁄𑀳𑀺𑀡𑀻𑀦𑀁,_  
_𑀕𑀯𑀁 𑀲𑀳𑀲𑁆𑀲𑀁 𑀲𑀳 𑀧𑀼𑀗𑁆𑀕𑀯𑁂𑀦;_  
_𑀅𑀭𑀺𑀬𑁄 𑀳𑀺 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀓𑀣𑀁 𑀦 𑀤𑀚𑁆𑀚𑀸,_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀕𑀸𑀣𑀸 𑀢𑀯 𑀥𑀫𑁆𑀫𑀬𑀼𑀢𑁆𑀢𑀸”𑀢𑀺𑁇_  


---

1001. 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
