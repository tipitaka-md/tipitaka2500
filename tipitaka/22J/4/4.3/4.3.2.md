# 𑁪𑁇𑁩𑁇𑁨 𑀤𑀼𑀤𑁆𑀤𑀼𑀪𑀚𑀸𑀢𑀓

* 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀝𑀺𑀤𑀽𑀲𑀓𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀤𑁆𑀤𑀼𑀪𑀚𑀸𑀢𑀓

(85.)

992\. _“𑀤𑀼𑀤𑁆𑀤𑀼𑀪𑀸𑀬𑀢𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀬𑀲𑁆𑀫𑀺𑀁 𑀤𑁂𑀲𑁂 𑀯𑀲𑀸𑀫𑀳𑀁;_  
_𑀅𑀳𑀫𑁆𑀧𑁂𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀓𑀺𑀫𑁂𑀢𑀁 𑀤𑀼𑀤𑁆𑀤𑀼𑀪𑀸𑀬𑀢𑀺”𑁇_  


(86.)

993\. _“𑀩𑁂𑀮𑀼𑀯𑀁 𑀧𑀢𑀺𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀤𑀼𑀤𑁆𑀤𑀼𑀪𑀦𑁆𑀢𑀺 𑀲𑀲𑁄 𑀚𑀯𑀺;_  
_𑀲𑀲𑀲𑁆𑀲 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀸 𑀫𑀺𑀕𑀯𑀸𑀳𑀺𑀦𑀻𑁇_  


(87.)

994\. _𑀅𑀧𑁆𑀧𑀢𑁆𑀯𑀸 𑀧𑀤𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀧𑀭𑀖𑁄𑀲𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄;_  
_𑀧𑀦𑀸𑀤𑀧𑀭𑀫𑀸 𑀩𑀸𑀮𑀸,_  
_𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀭𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇_  


(88.)

995\. _𑀬𑁂 𑀘 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀧𑀜𑁆𑀜𑀸𑀬𑀽𑀧𑀲𑀫𑁂 𑀭𑀢𑀸;_  
_𑀆𑀭𑀓𑀸 𑀯𑀺𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀦 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀭𑀧𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇_  


---

996\. 𑀤𑀼𑀤𑁆𑀤𑀼𑀪𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
