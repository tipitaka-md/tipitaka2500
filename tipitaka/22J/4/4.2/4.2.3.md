# 𑁪𑁇𑁨𑁇𑁩 𑀔𑀦𑁆𑀢𑀻𑀯𑀸𑀤𑀻𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀔𑀦𑁆𑀢𑀻𑀯𑀸𑀤𑀻𑀚𑀸𑀢𑀓

(49.)

945\. _“𑀬𑁄 𑀢𑁂 𑀳𑀢𑁆𑀣𑁂 𑀘 𑀧𑀸𑀤𑁂 𑀘,_  
_𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀜𑁆𑀘 𑀙𑁂𑀤𑀬𑀺;_  
_𑀢𑀲𑁆𑀲 𑀓𑀼𑀚𑁆𑀛 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀫𑀸 𑀭𑀝𑁆𑀞𑀁 𑀯𑀺𑀦𑀲𑀸 𑀇𑀤𑀁”𑁇_  


(50.)

946\. _“𑀬𑁄 𑀫𑁂 𑀳𑀢𑁆𑀣𑁂 𑀘 𑀧𑀸𑀤𑁂 𑀘,_  
_𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀜𑁆𑀘 𑀙𑁂𑀤𑀬𑀺;_  
_𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀢𑀼 𑀲𑁄 𑀭𑀸𑀚𑀸,_  
_𑀦 𑀳𑀺 𑀓𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀫𑀸𑀤𑀺𑀲𑀸”𑁇_  


(51.)

947\. _𑀅𑀳𑀽 𑀅𑀢𑀻𑀢𑀫𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀲𑀫𑀡𑁄 𑀔𑀦𑁆𑀢𑀺𑀤𑀻𑀧𑀦𑁄;_  
_𑀢𑀁 𑀔𑀦𑁆𑀢𑀺𑀬𑀸𑀬𑁂𑀯 𑀞𑀺𑀢𑀁,_  
_𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀅𑀙𑁂𑀤𑀬𑀺𑁇_  


(52.)

948\. _𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀨𑀭𑀼𑀲𑀲𑁆𑀲,_  
_𑀯𑀺𑀧𑀸𑀓𑁄 𑀓𑀝𑀼𑀓𑁄 𑀅𑀳𑀼;_  
_𑀬𑀁 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀯𑁂𑀤𑁂𑀲𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄𑀢𑀺𑁇_  


---

949\. 𑀔𑀦𑁆𑀢𑀻𑀯𑀸𑀤𑀻𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
