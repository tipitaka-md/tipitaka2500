# 𑁪𑁇𑁨𑁇𑁯 𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀚𑀸𑀢𑀓

(73.)

975. _“𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀚𑀻𑀯𑀸𑀫𑀺,_  
_𑀮𑀪𑀸𑀫𑀺 𑀘𑁂𑀯 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁;_  
_𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑁂 𑀘 𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺,_  
_𑀓𑀸 𑀦𑀼 𑀪𑀦𑁆𑀢𑁂 𑀕𑀢𑀻 𑀫𑀫”𑁇_  


(74.)

976. _“𑀫𑀦𑁄 𑀘𑁂 𑀢𑁂 𑀦𑀧𑁆𑀧𑀡𑀫𑀢𑀺,_  
_𑀧𑀓𑁆𑀔𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀼𑀦𑁄;_  
_𑀅𑀩𑁆𑀬𑀸𑀯𑀝𑀲𑁆𑀲 𑀪𑀤𑁆𑀭𑀲𑁆𑀲,_  
_𑀦 𑀧𑀸𑀧𑀫𑀼𑀧𑀮𑀺𑀫𑁆𑀧𑀢𑀺”𑁇_  


(75.)

977. _“𑀜𑀸𑀢𑀓𑁄 𑀦𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑀢𑀺,_  
_𑀩𑀳𑀼 𑀆𑀕𑀘𑁆𑀙𑀢𑁂 𑀚𑀦𑁄;_  
_𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀫𑁆𑀫𑀁 𑀨𑀼𑀲𑀢𑀺,_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀫𑁂 𑀲𑀗𑁆𑀓𑀢𑁂 𑀫𑀦𑁄”𑁇_  


(76.)

978. _“𑀦 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀫𑁆𑀫𑀁 𑀨𑀼𑀲𑀢𑀺,_  
_𑀫𑀦𑁄 𑀘𑁂 𑀦𑀧𑁆𑀧𑀤𑀼𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀲𑁆𑀲 𑀪𑀤𑁆𑀭𑀲𑁆𑀲,_  
_𑀦 𑀧𑀸𑀧𑀫𑀼𑀧𑀮𑀺𑀫𑁆𑀧𑀢𑀻”𑀢𑀺𑁇_  


---

979. 𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀚𑀸𑀢𑀓𑀁 𑀦𑀯𑀫𑀁𑁇
