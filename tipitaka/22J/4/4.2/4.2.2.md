# 𑁪𑁇𑁨𑁇𑁨 𑀓𑀲𑁆𑀲𑀧𑀫𑀦𑁆𑀤𑀺𑀬𑀚𑀸𑀢𑀓

### 𑀘𑀢𑀼𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀓𑀲𑁆𑀲𑀧𑀫𑀦𑁆𑀤𑀺𑀬𑀚𑀸𑀢𑀓

(45.)

940\. _“𑀅𑀧𑀺 𑀓𑀲𑁆𑀲𑀧 𑀫𑀦𑁆𑀤𑀺𑀬𑀸,_  
_𑀬𑀼𑀯𑀸 𑀲𑀧𑀢𑀺 𑀳𑀦𑁆𑀢𑀺 𑀯𑀸;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀔𑀫𑀢𑁂 𑀥𑀻𑀭𑁄,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀢𑀁 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀢𑀺𑁇_  


(46.)

941\. _𑀲𑀘𑁂𑀧𑀺 𑀲𑀦𑁆𑀢𑁄 𑀯𑀺𑀯𑀤𑀦𑁆𑀢𑀺,_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀦𑁆𑀥𑀻𑀬𑀭𑁂 𑀧𑀼𑀦;_  
_𑀩𑀸𑀮𑀸 𑀧𑀢𑁆𑀢𑀸𑀯 𑀪𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀦 𑀢𑁂 𑀲𑀫𑀣𑀫𑀚𑁆𑀛𑀕𑀽𑁇_  


(47.)

942\. _𑀏𑀢𑁂 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑀫𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑀦𑁆𑀥𑀺 𑀢𑁂𑀲𑀁 𑀦 𑀚𑀻𑀭𑀢𑀺;_  
_𑀬𑁄 𑀘𑀸𑀥𑀺𑀧𑀦𑁆𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀬𑁄 𑀘 𑀚𑀸𑀦𑀸𑀢𑀺 𑀤𑁂𑀲𑀦𑀁𑁇_  


(48.)

943\. _𑀏𑀲𑁄 𑀳𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑀭𑁄,_  
_𑀪𑀸𑀭𑀯𑀳𑁄 𑀥𑀼𑀭𑀤𑁆𑀥𑀭𑁄;_  
_𑀬𑁄 𑀧𑀭𑁂𑀲𑀸𑀥𑀺𑀧𑀦𑁆𑀦𑀸𑀦𑀁,_  
_𑀲𑀬𑀁 𑀲𑀦𑁆𑀥𑀸𑀢𑀼𑀫𑀭𑀳𑀢𑀻”𑀢𑀺𑁇_  


---

944\. 𑀓𑀲𑁆𑀲𑀧𑀫𑀦𑁆𑀤𑀺𑀬𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
