# 𑁧𑁬𑁇𑁧𑁇𑁭 𑀤𑀓𑀭𑀓𑁆𑀔𑀲𑀚𑀸𑀢𑀓

* 𑀢𑀺𑀁𑀲𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀺𑀁𑀙𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀤𑀓𑀭𑀓𑁆𑀔𑀲𑀚𑀸𑀢𑀓

(224.)

3485\. _“𑀲𑀘𑁂 𑀯𑁄 𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀉𑀤𑀓𑀡𑁆𑀡𑀯𑁂;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀩𑀮𑀺𑀫𑁂𑀲𑀸𑀦𑁄,_  
_𑀦𑀸𑀯𑀁 𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬 𑀭𑀓𑁆𑀔𑀲𑁄;_  
_𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀁 𑀓𑀣𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀫𑀼𑀜𑁆𑀘𑁂𑀲𑀺 𑀤𑀓𑀭𑀓𑁆𑀔𑀲𑀸”𑁇_  


(225.)

3486\. _“𑀫𑀸𑀢𑀭𑀁 𑀧𑀞𑀫𑀁 𑀤𑀚𑁆𑀚𑀁,_  
_𑀪𑀭𑀺𑀬𑀁 𑀤𑀢𑁆𑀯𑀸𑀦 𑀪𑀸𑀢𑀭𑀁;_  
_𑀢𑀢𑁄 𑀲𑀳𑀸𑀬𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀚𑁆𑀚𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁;_  
_𑀙𑀝𑁆𑀞𑀸𑀳𑀁 𑀤𑀚𑁆𑀚𑀫𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀦𑁂𑀯 𑀤𑀚𑁆𑀚𑀁 𑀫𑀳𑁄𑀲𑀥𑀁”𑁇_  


(226.)

3487\. _“𑀧𑁄𑀲𑁂𑀢𑀸 𑀢𑁂 𑀚𑀦𑁂𑀢𑁆𑀢𑀻 𑀘,_  
_𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸;_  
_𑀙𑀩𑁆𑀪𑀻 𑀢𑀬𑀺 𑀧𑀤𑀼𑀲𑁆𑀲𑀢𑀺,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀦𑀻;_  
_𑀅𑀜𑁆𑀜𑀁 𑀉𑀧𑀦𑀺𑀲𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀯𑀥𑀸 𑀢𑀁 𑀧𑀭𑀺𑀫𑁄𑀘𑀬𑀺𑁇_  


(227.)

3488\. _𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀺𑀁 𑀧𑀸𑀡𑀤𑀤𑀺𑀁,_  
_𑀑𑀭𑀲𑀁 𑀕𑀩𑁆𑀪𑀥𑀸𑀭𑀺𑀦𑀺𑀁;_  
_𑀫𑀸𑀢𑀭𑀁 𑀓𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀲𑀺 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(228.)

3489\. _“𑀤𑀳𑀭𑀸 𑀯𑀺𑀬𑀮𑀗𑁆𑀓𑀸𑀭𑀁,_  
_𑀥𑀸𑀭𑁂𑀢𑀺 𑀅𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀦𑀁;_  
_𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑁂 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑁂,_  
_𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀧𑀚𑀕𑁆𑀖𑀢𑀺𑁇_  


(229.)

3490\. _𑀅𑀣𑁄𑀧𑀺 𑀧𑀝𑀺𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀲𑀬𑀁 𑀤𑀽𑀢𑀸𑀦𑀺 𑀲𑀸𑀲𑀢𑀺;_  
_𑀫𑀸𑀢𑀭𑀁 𑀢𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀳𑀁 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(230.)

3491\. _“𑀇𑀢𑁆𑀣𑀺𑀕𑀼𑀫𑁆𑀩𑀲𑁆𑀲 𑀧𑀯𑀭𑀸,_  
_𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀁 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑀻;_  
_𑀅𑀦𑀼𑀕𑁆𑀕𑀢𑀸 𑀲𑀻𑀮𑀯𑀢𑀻,_  
_𑀙𑀸𑀬𑀸𑀯 𑀅𑀦𑀧𑀸𑀬𑀺𑀦𑀻𑁇_  


(231.)

3492\. _𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑀸 𑀧𑀼𑀜𑁆𑀜𑀯𑀢𑀻,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀤𑀲𑁆𑀲𑀺𑀦𑀻;_  
_𑀉𑀩𑁆𑀩𑀭𑀺𑀁 𑀓𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀲𑀺 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(232.)

3493\. _“𑀔𑀺𑀟𑁆𑀟𑀸𑀭𑀢𑀺𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀁,_  
_𑀅𑀦𑀢𑁆𑀣𑀯𑀲𑀫𑀸𑀕𑀢𑀁;_  
_𑀲𑀸 𑀫𑀁 𑀲𑀓𑀸𑀦 𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀅𑀬𑀸𑀘𑀁 𑀬𑀸𑀘𑀢𑁂 𑀥𑀦𑀁𑁇_  


(233.)

3494\. _𑀲𑁄𑀳𑀁 𑀤𑀤𑀸𑀫𑀺 𑀲𑀸𑀭𑀢𑁆𑀢𑁄,_  
_𑀩𑀳𑀼𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀁 𑀥𑀦𑀁;_  
_𑀲𑀼𑀤𑀼𑀘𑁆𑀘𑀚𑀁 𑀘𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀘𑁆𑀙𑀸 𑀲𑁄𑀘𑀸𑀫𑀺 𑀤𑀼𑀫𑁆𑀫𑀦𑁄;_  
_𑀉𑀩𑁆𑀩𑀭𑀺𑀁 𑀢𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀳𑀁 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(234.)

3495\. _“𑀬𑁂𑀦𑁄𑀘𑀺𑀢𑀸 𑀚𑀦𑀧𑀤𑀸,_  
_𑀆𑀦𑀻𑀢𑀸 𑀘 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀆𑀪𑀢𑀁 𑀧𑀭𑀭𑀚𑁆𑀚𑁂𑀪𑀺,_  
_𑀅𑀪𑀺𑀝𑁆𑀞𑀸𑀬 𑀩𑀳𑀼𑀁 𑀥𑀦𑀁𑁇_  


(235.)

3496\. _𑀥𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀦𑀁 𑀧𑀯𑀭𑀁,_  
_𑀲𑀽𑀭𑀁 𑀢𑀺𑀔𑀺𑀡𑀫𑀦𑁆𑀢𑀺𑀦𑀁;_  
_𑀪𑀸𑀢𑀭𑀁 𑀓𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀲𑀺 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(236.)

3497\. _“𑀬𑁂𑀦𑁄𑀘𑀺𑀢𑀸 𑀚𑀦𑀧𑀤𑀸,_  
_𑀆𑀦𑀻𑀢𑀸 𑀘 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁;_  
_𑀆𑀪𑀢𑀁 𑀧𑀭𑀭𑀚𑁆𑀚𑁂𑀪𑀺,_  
_𑀅𑀪𑀺𑀝𑁆𑀞𑀸𑀬 𑀩𑀳𑀼𑀁 𑀥𑀦𑀁𑁇_  


(237.)

3498\. _𑀥𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀦𑀁 𑀧𑀯𑀭𑁄,_  
_𑀲𑀽𑀭𑁄 𑀢𑀺𑀔𑀺𑀡𑀫𑀦𑁆𑀢𑀺 𑀘;_  
_𑀫𑀬𑀸𑀬𑀁 𑀲𑀼𑀔𑀺𑀢𑁄 𑀭𑀸𑀚𑀸,_  
_𑀅𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺 𑀤𑀸𑀭𑀓𑁄𑁇_  


(238.)

3499\. _𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀫𑁆𑀧𑀺 𑀫𑁂 𑀅𑀬𑁆𑀬𑁂,_  
_𑀦 𑀲𑁄 𑀏𑀢𑀺 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀪𑀸𑀢𑀭𑀁 𑀢𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀳𑀁 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(239.)

3500\. _“𑀏𑀓𑀭𑀢𑁆𑀢𑁂𑀦 𑀉𑀪𑀬𑁄,_  
_𑀢𑁆𑀯𑀜𑁆𑀘𑁂𑀯 𑀥𑀦𑀼𑀲𑁂𑀔 𑀘;_  
_𑀉𑀪𑁄 𑀚𑀸𑀢𑁂𑀢𑁆𑀣 𑀧𑀜𑁆𑀘𑀸𑀮𑀸,_  
_𑀲𑀳𑀸𑀬𑀸 𑀲𑀼𑀲𑀫𑀸𑀯𑀬𑀸𑁇_  


(240.)

3501\. _𑀘𑀭𑀺𑀬𑀸 𑀢𑀁 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀣𑁄,_  
_𑀏𑀓𑀤𑀼𑀓𑁆𑀔𑀲𑀼𑀔𑁄 𑀢𑀯;_  
_𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑁂 𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑀓𑀺𑀘𑁆𑀘𑁂𑀲𑀼 𑀩𑁆𑀬𑀸𑀯𑀝𑁄;_  
_𑀲𑀳𑀸𑀬𑀁 𑀓𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀲𑀺 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(241.)

3502\. _“𑀘𑀭𑀺𑀬𑀸 𑀫𑀁 𑀅𑀬𑀁 𑀅𑀬𑁆𑀬𑁂,_  
_𑀧𑀚𑀕𑁆𑀖𑀺𑀢𑁆𑀣𑁄 𑀫𑀬𑀸 𑀲𑀳;_  
_𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀧𑀚𑀕𑁆𑀖𑀢𑀺𑁇_  


(242.)

3503\. _𑀉𑀩𑁆𑀩𑀭𑀺𑀬𑀸𑀧𑀺𑀳𑀁 𑀅𑀬𑁆𑀬𑁂,_  
_𑀫𑀦𑁆𑀢𑀬𑀸𑀫𑀺 𑀭𑀳𑁄𑀕𑀢𑁄;_  
_𑀅𑀦𑀸𑀫𑀦𑁆𑀢𑁄 𑀧𑀯𑀺𑀲𑀢𑀺,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀤𑀺𑀢𑁄𑁇_  


(243.)

3504\. _𑀮𑀤𑁆𑀥𑀤𑁆𑀯𑀸𑀭𑁄 𑀓𑀢𑁄𑀓𑀸𑀲𑁄,_  
_𑀅𑀳𑀺𑀭𑀺𑀓𑀁 𑀅𑀦𑀸𑀤𑀭𑀁;_  
_𑀲𑀳𑀸𑀬𑀁 𑀢𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀳𑀁 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(244.)

3505\. _“𑀓𑀼𑀲𑀮𑁄 𑀲𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀭𑀼𑀢𑀜𑁆𑀜𑀽 𑀆𑀕𑀢𑀸𑀕𑀫𑁄;_  
_𑀉𑀧𑁆𑀧𑀸𑀢𑁂 𑀲𑀼𑀧𑀺𑀦𑁂 𑀬𑀼𑀢𑁆𑀢𑁄,_  
_𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑁂 𑀘 𑀧𑀯𑁂𑀲𑀦𑁂𑁇_  


(245.)

3506\. _𑀧𑀝𑁆𑀞𑁄 𑀪𑀽𑀫𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀧𑀤𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀓𑁂𑀦 𑀤𑁄𑀲𑁂𑀦,_  
_𑀤𑀚𑁆𑀚𑀸𑀲𑀺 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(246.)

3507\. _“𑀧𑀭𑀺𑀲𑀸𑀬𑀫𑁆𑀧𑀺 𑀫𑁂 𑀅𑀬𑁆𑀬𑁂,_  
_𑀉𑀫𑁆𑀫𑀻𑀮𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀺𑀓𑁆𑀔𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀅𑀘𑁆𑀘𑀪𑀫𑀼𑀁 𑀮𑀼𑀤𑁆𑀤𑀁,_  
_𑀤𑀚𑁆𑀚𑀸𑀳𑀁 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(247.)

3508\. _“𑀲𑀲𑀫𑀼𑀤𑁆𑀤𑀧𑀭𑀺𑀬𑀸𑀬𑀁,_  
_𑀫𑀳𑀺𑀁 𑀲𑀸𑀕𑀭𑀓𑀼𑀡𑁆𑀟𑀮𑀁;_  
_𑀯𑀲𑀼𑀦𑁆𑀥𑀭𑀁 𑀆𑀯𑀲𑀲𑀺,_  
_𑀅𑀫𑀘𑁆𑀘𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄𑁇_  


(248.)

3509\. _𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀫𑀳𑀸𑀭𑀝𑁆𑀞𑁄,_  
_𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀫𑀳𑀩𑁆𑀩𑀮𑁄;_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀏𑀓𑀭𑀸𑀚𑀸𑀲𑀺,_  
_𑀬𑀲𑁄 𑀢𑁂 𑀯𑀺𑀧𑀼𑀮𑀁 𑀕𑀢𑁄𑁇_  


(249.)

3510\. _𑀲𑁄𑀍𑀅𑀲𑀺𑀢𑁆𑀣𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀸;_  
_𑀦𑀸𑀦𑀸𑀚𑀦𑀧𑀤𑀸 𑀦𑀸𑀭𑀻,_  
_𑀤𑁂𑀯𑀓𑀜𑁆𑀜𑀽𑀧𑀫𑀸 𑀲𑀼𑀪𑀸𑁇_  


(250.)

3511\. _𑀏𑀯𑀁 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀦𑀁;_  
_𑀲𑀼𑀔𑀺𑀢𑀸𑀦𑀁 𑀧𑀺𑀬𑀁 𑀤𑀻𑀖𑀁,_  
_𑀚𑀻𑀯𑀺𑀢𑀁 𑀆𑀳𑀼 𑀔𑀢𑁆𑀢𑀺𑀬𑁇_  


(251.)

3512\. _𑀅𑀣 𑀢𑁆𑀯𑀁 𑀓𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀓𑁂𑀦 𑀯𑀸 𑀧𑀦 𑀳𑁂𑀢𑀼𑀦𑀸;_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄,_  
_𑀧𑀸𑀡𑀁 𑀘𑀚𑀲𑀺 𑀤𑀼𑀘𑁆𑀘𑀚𑀁”𑁇_  


(252.)

3513\. _“𑀬𑀢𑁄𑀧𑀺 𑀆𑀕𑀢𑁄 𑀅𑀬𑁆𑀬𑁂,_  
_𑀫𑀫 𑀳𑀢𑁆𑀣𑀁 𑀫𑀳𑁄𑀲𑀥𑁄;_  
_𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀥𑀻𑀭𑀲𑁆𑀲,_  
_𑀅𑀦𑀼𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇_  


(253.)

3514\. _𑀲𑀘𑁂 𑀘 𑀓𑀺𑀲𑁆𑀫𑀺𑀘𑀺 𑀓𑀸𑀮𑁂,_  
_𑀫𑀭𑀡𑀁 𑀫𑁂 𑀧𑀼𑀭𑁂 𑀲𑀺𑀬𑀸;_  
_𑀲𑁄 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀧𑀧𑀼𑀢𑁆𑀢𑁂 𑀘,_  
_𑀲𑀼𑀔𑀸𑀧𑁂𑀬𑁆𑀬 𑀫𑀳𑁄𑀲𑀥𑁄𑁇_  


(254.)

3515\. _𑀅𑀦𑀸𑀕𑀢𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀩𑁆𑀩𑀫𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀦𑀸𑀧𑀭𑀸𑀥𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀁,_  
_𑀦 𑀤𑀚𑁆𑀚𑀁 𑀤𑀓𑀭𑀓𑁆𑀔𑀺𑀦𑁄”𑁇_  


(255.)

3516\. _“𑀇𑀤𑀁 𑀲𑀼𑀡𑀸𑀣 𑀧𑀜𑁆𑀘𑀸𑀮𑀸,_  
_𑀘𑀽𑀍𑀅𑀦𑁂𑀬𑁆𑀬𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁;_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄,_  
_𑀧𑀸𑀡𑀁 𑀘𑀚𑀢𑀺 𑀤𑀼𑀘𑁆𑀘𑀚𑀁𑁇_  


(256.)

3517\. _𑀫𑀸𑀢𑀼 𑀪𑀭𑀺𑀬𑀸𑀬 𑀪𑀸𑀢𑀼𑀘𑁆𑀘,_  
_𑀲𑀔𑀺𑀦𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀘;_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀘𑀸𑀧𑀺 𑀧𑀜𑁆𑀘𑀸𑀮𑁄,_  
_𑀙𑀦𑁆𑀦𑀁 𑀘𑀚𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀁𑁇_  


(257.)

3518\. _𑀏𑀯𑀁 𑀫𑀳𑀢𑁆𑀣𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸,_  
_𑀦𑀺𑀧𑀼𑀡𑀸 𑀲𑀸𑀥𑀼𑀘𑀺𑀦𑁆𑀢𑀺𑀦𑀻;_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀳𑀺𑀢𑀢𑁆𑀣𑀸𑀬,_  
_𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀲𑀼𑀔𑀸𑀬 𑀘𑀸”𑀢𑀺𑁇_  


---

3519\. 𑀤𑀓𑀭𑀓𑁆𑀔𑀲𑀚𑀸𑀢𑀓𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
