# 𑁧𑁇𑁪𑁇𑁫 𑀳𑀢𑁆𑀣𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀤𑁂𑀯𑀤𑀽𑀢𑀯𑀕𑁆𑀕

* 𑀳𑀢𑁆𑀣𑀓𑀲𑀼𑀢𑁆𑀢

(35.)

199\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀆𑀍𑀅𑀯𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑁄𑀫𑀕𑁆𑀕𑁂 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑁂 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀳𑀢𑁆𑀣𑀓𑁄 𑀆𑀍𑀅𑀯𑀓𑁄 𑀚𑀗𑁆𑀖𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀦𑀼𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀺𑀘𑀭𑀫𑀸𑀦𑁄 𑀅𑀤𑁆𑀤𑀲 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀫𑀕𑁆𑀕𑁂 𑀲𑀺𑀁𑀲𑀧𑀸𑀯𑀦𑁂 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀳𑀢𑁆𑀣𑀓𑁄 𑀆𑀍𑀅𑀯𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀪𑀕𑀯𑀸, 𑀲𑀼𑀔𑀫𑀲𑀬𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀓𑀼𑀫𑀸𑀭, 𑀲𑀼𑀔𑀫𑀲𑀬𑀺𑀢𑁆𑀣𑀁𑁇 𑀬𑁂 𑀘 𑀧𑀦 𑀮𑁄𑀓𑁂 𑀲𑀼𑀔𑀁 𑀲𑁂𑀦𑁆𑀢𑀺, 𑀅𑀳𑀁 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄”𑀢𑀺𑁇

200\. “𑀲𑀻𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸 𑀭𑀢𑁆𑀢𑀺, 𑀅𑀦𑁆𑀢𑀭𑀝𑁆𑀞𑀓𑁄 𑀳𑀺𑀫𑀧𑀸𑀢𑀲𑀫𑀬𑁄, 𑀔𑀭𑀸 𑀕𑁄𑀓𑀡𑁆𑀝𑀓𑀳𑀢𑀸 𑀪𑀽𑀫𑀺, 𑀢𑀦𑀼𑀓𑁄 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁄, 𑀯𑀺𑀭𑀍𑀆𑀦𑀺 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀸𑀦𑀺, 𑀲𑀻𑀢𑀸𑀦𑀺 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺, 𑀲𑀻𑀢𑁄 𑀘 𑀯𑁂𑀭𑀫𑁆𑀪𑁄 𑀯𑀸𑀢𑁄 𑀯𑀸𑀬𑀢𑀺𑁇 𑀅𑀣 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀸 𑀏𑀯𑀫𑀸𑀳—  ‘𑀏𑀯𑀁, 𑀓𑀼𑀫𑀸𑀭, 𑀲𑀼𑀔𑀫𑀲𑀬𑀺𑀢𑁆𑀣𑀁𑁇 𑀬𑁂 𑀘 𑀧𑀦 𑀮𑁄𑀓𑁂 𑀲𑀼𑀔𑀁 𑀲𑁂𑀦𑁆𑀢𑀺, 𑀅𑀳𑀁 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄’”𑀢𑀺𑁇

201\. “𑀢𑁂𑀦 𑀳𑀺, 𑀓𑀼𑀫𑀸𑀭, 𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀢𑁂 𑀔𑀫𑁂𑀬𑁆𑀬 𑀢𑀣𑀸 𑀦𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀓𑀼𑀫𑀸𑀭, 𑀇𑀥𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀁 𑀉𑀮𑁆𑀮𑀺𑀢𑁆𑀢𑀸𑀯𑀮𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀯𑀸𑀢𑀁 𑀨𑀼𑀲𑀺𑀢𑀕𑁆𑀕𑀍𑀅𑀁 𑀧𑀺𑀳𑀺𑀢𑀯𑀸𑀢𑀧𑀸𑀦𑀁𑁇 𑀢𑀢𑁆𑀭𑀲𑁆𑀲 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁄 𑀕𑁄𑀦𑀓𑀢𑁆𑀣𑀢𑁄 𑀧𑀝𑀺𑀓𑀢𑁆𑀣𑀢𑁄 𑀧𑀝𑀮𑀺𑀓𑀢𑁆𑀣𑀢𑁄 𑀓𑀤𑀮𑀺𑀫𑀺𑀕𑀧𑀯𑀭𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀭𑀡𑁄 𑀲𑁅𑀢𑁆𑀢𑀭𑀘𑁆𑀙𑀤𑁄 𑀉𑀪𑀢𑁄 𑀮𑁄𑀳𑀺𑀢𑀓𑀽𑀧𑀥𑀸𑀦𑁄; 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁄 𑀘𑁂𑀢𑁆𑀣 𑀛𑀸𑀬𑁂𑀬𑁆𑀬; 𑀘𑀢𑀲𑁆𑀲𑁄 𑀘 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑀸𑀫𑀦𑀸𑀧𑁂𑀦 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀸 𑀅𑀲𑁆𑀲𑀼𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀓𑀼𑀫𑀸𑀭, 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀲𑁄 𑀲𑀬𑁂𑀬𑁆𑀬 𑀦𑁄 𑀯𑀸? 𑀓𑀣𑀁 𑀯𑀸 𑀢𑁂 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀼𑀔𑀁 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀬𑁂𑀬𑁆𑀬𑁇 𑀬𑁂 𑀘 𑀧𑀦 𑀮𑁄𑀓𑁂 𑀲𑀼𑀔𑀁 𑀲𑁂𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄”𑀢𑀺𑁇

202\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀓𑀼𑀫𑀸𑀭, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀭𑀸𑀕𑀚𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸 𑀓𑀸𑀬𑀺𑀓𑀸 𑀯𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀯𑀸 𑀬𑁂𑀳𑀺 𑀲𑁄 𑀭𑀸𑀕𑀚𑁂𑀳𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑁂𑀳𑀺 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀬𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇

203\. “𑀬𑁂𑀳𑀺 𑀔𑁄 𑀲𑁄, 𑀓𑀼𑀫𑀸𑀭, 𑀕𑀳𑀧𑀢𑀺 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑀸 𑀭𑀸𑀕𑀚𑁂𑀳𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑁂𑀳𑀺 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀬𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀭𑀸𑀕𑁄 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑁄 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄𑁇 𑀢𑀲𑁆𑀫𑀸𑀳𑀁 𑀲𑀼𑀔𑀫𑀲𑀬𑀺𑀢𑁆𑀣𑀁𑁇

204\. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀓𑀼𑀫𑀸𑀭, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀤𑁄𑀲𑀚𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸…𑀧𑁂…  𑀫𑁄𑀳𑀚𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸 𑀓𑀸𑀬𑀺𑀓𑀸 𑀯𑀸 𑀘𑁂𑀢𑀲𑀺𑀓𑀸 𑀯𑀸 𑀬𑁂𑀳𑀺 𑀲𑁄 𑀫𑁄𑀳𑀚𑁂𑀳𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑁂𑀳𑀺 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀬𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇

205\. “𑀬𑁂 𑀳𑀺 𑀔𑁄 𑀲𑁄, 𑀓𑀼𑀫𑀸𑀭, 𑀕𑀳𑀧𑀢𑀺 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑀸 𑀫𑁄𑀳𑀚𑁂𑀳𑀺 𑀧𑀭𑀺𑀍𑀆𑀳𑁂𑀳𑀺 𑀧𑀭𑀺𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀬𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀫𑁄𑀳𑁄 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀳𑀻𑀦𑁄 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄𑁇 𑀢𑀲𑁆𑀫𑀸𑀳𑀁 𑀲𑀼𑀔𑀫𑀲𑀬𑀺𑀢𑁆𑀣𑀦𑁆𑀢𑀺𑁇

206\. _𑀲𑀩𑁆𑀩𑀤𑀸 𑀯𑁂 𑀲𑀼𑀔𑀁 𑀲𑁂𑀢𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄;_  
_𑀬𑁄 𑀦 𑀮𑀺𑀫𑁆𑀧𑀢𑀺 𑀓𑀸𑀫𑁂𑀲𑀼,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑁄 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑁇_  


207\. _𑀲𑀩𑁆𑀩𑀸 𑀆𑀲𑀢𑁆𑀢𑀺𑀬𑁄 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀳𑀤𑀬𑁂 𑀤𑀭𑀁;_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀁 𑀲𑁂𑀢𑀺,_  
_𑀲𑀦𑁆𑀢𑀺𑀁 𑀧𑀧𑁆𑀧𑀼𑀬𑁆𑀬 𑀘𑁂𑀢𑀲𑁄”𑀢𑀺𑁇_
