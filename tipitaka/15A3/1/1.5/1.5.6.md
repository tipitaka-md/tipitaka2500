# 𑁧𑁇𑁫𑁇𑁬 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(46.)

286\. “𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀓𑀸𑀬𑁂𑀦, 𑀯𑀸𑀘𑀸𑀬, 𑀫𑀦𑀲𑀸𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
