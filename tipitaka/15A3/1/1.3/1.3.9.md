# 𑁧𑁇𑁩𑁇𑁯 𑀅𑀦𑁆𑀥𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀧𑀼𑀕𑁆𑀕𑀮𑀯𑀕𑁆𑀕

* 𑀅𑀦𑁆𑀥𑀲𑀼𑀢𑁆𑀢

(29.)

142\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀅𑀦𑁆𑀥𑁄, 𑀏𑀓𑀘𑀓𑁆𑀔𑀼, 𑀤𑁆𑀯𑀺𑀘𑀓𑁆𑀔𑀼𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑁆𑀥𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀁 𑀯𑀸 𑀪𑁄𑀕𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀅𑀥𑀺𑀕𑀢𑀁 𑀯𑀸 𑀪𑁄𑀕𑀁 𑀨𑀸𑀢𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬; 𑀢𑀣𑀸𑀭𑀽𑀧𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀲𑀸𑀯𑀚𑁆𑀚𑀸𑀦𑀯𑀚𑁆𑀚𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀳𑀻𑀦𑀧𑁆𑀧𑀡𑀻𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀦𑁆𑀥𑁄𑁇

143\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀏𑀓𑀘𑀓𑁆𑀔𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀁 𑀯𑀸 𑀪𑁄𑀕𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀅𑀥𑀺𑀕𑀢𑀁 𑀯𑀸 𑀪𑁄𑀕𑀁 𑀨𑀸𑀢𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬; 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀧𑀦𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼 𑀦 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀲𑀸𑀯𑀚𑁆𑀚𑀸𑀦𑀯𑀚𑁆𑀚𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀳𑀻𑀦𑀧𑁆𑀧𑀡𑀻𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀏𑀓𑀘𑀓𑁆𑀔𑀼𑁇

144\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁆𑀯𑀺𑀘𑀓𑁆𑀔𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀘𑀓𑁆𑀔𑀼 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀁 𑀯𑀸 𑀪𑁄𑀕𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀅𑀥𑀺𑀕𑀢𑀁 𑀯𑀸 𑀪𑁄𑀕𑀁 𑀨𑀸𑀢𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬; 𑀢𑀣𑀸𑀭𑀽𑀧𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀘𑀓𑁆𑀔𑀼 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸𑀭𑀽𑀧𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬; 𑀲𑀸𑀯𑀚𑁆𑀚𑀸𑀦𑀯𑀚𑁆𑀚𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀳𑀻𑀦𑀧𑁆𑀧𑀡𑀻𑀢𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑁂 𑀥𑀫𑁆𑀫𑁂 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑁆𑀯𑀺𑀘𑀓𑁆𑀔𑀼𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺𑁇

145\. _𑀦 𑀘𑁂𑀯 𑀪𑁄𑀕𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀸,_  
_𑀦 𑀘 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀼𑀩𑁆𑀩𑀢𑀺;_  
_𑀉𑀪𑀬𑀢𑁆𑀣 𑀓𑀮𑀺𑀕𑁆𑀕𑀸𑀳𑁄,_  
_𑀅𑀦𑁆𑀥𑀲𑁆𑀲 𑀳𑀢𑀘𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


146\. _𑀅𑀣𑀸𑀧𑀭𑀸𑀬𑀁 𑀅𑀓𑁆𑀔𑀸𑀢𑁄,_  
_𑀏𑀓𑀘𑀓𑁆𑀔𑀼 𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀥𑀫𑁆𑀫𑀸𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀞𑁄𑀲𑁄,_  
_𑀪𑁄𑀕𑀸𑀦𑀺 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺𑁇_  


147\. _𑀣𑁂𑀬𑁆𑀬𑁂𑀦 𑀓𑀽𑀝𑀓𑀫𑁆𑀫𑁂𑀦,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺 𑀲𑀗𑁆𑀖𑀸𑀢𑀼𑀁,_  
_𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀘 𑀫𑀸𑀦𑀯𑁄;_  
_𑀇𑀢𑁄 𑀲𑁄 𑀦𑀺𑀭𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀏𑀓𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀺𑁇_  


148\. _𑀤𑁆𑀯𑀺𑀘𑀓𑁆𑀔𑀼 𑀧𑀦 𑀅𑀓𑁆𑀔𑀸𑀢𑁄,_  
_𑀲𑁂𑀝𑁆𑀞𑁄 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑁂𑀳𑀺 𑀪𑁄𑀕𑁂𑀳𑀺,_  
_𑀉𑀝𑁆𑀞𑀸𑀦𑀸𑀥𑀺𑀕𑀢𑀁 𑀥𑀦𑀁𑁇_  


149\. _𑀤𑀤𑀸𑀢𑀺 𑀲𑁂𑀝𑁆𑀞𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀅𑀩𑁆𑀬𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑁄 𑀦𑀭𑁄;_  
_𑀉𑀧𑁂𑀢𑀺 𑀪𑀤𑁆𑀤𑀓𑀁 𑀞𑀸𑀦𑀁,_  
_𑀬𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀦 𑀲𑁄𑀘𑀢𑀺𑁇_  


150\. _𑀅𑀦𑁆𑀥𑀜𑁆𑀘 𑀏𑀓𑀘𑀓𑁆𑀔𑀼𑀜𑁆𑀘,_  
_𑀆𑀭𑀓𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑁂;_  
_𑀤𑁆𑀯𑀺𑀘𑀓𑁆𑀔𑀼𑀁 𑀧𑀦 𑀲𑁂𑀯𑁂𑀣,_  
_𑀲𑁂𑀝𑁆𑀞𑀁 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀦𑁆”𑀢𑀺𑁇_
