# 𑁩𑁇𑁩𑁇𑁫 𑀳𑀢𑁆𑀣𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀳𑀢𑁆𑀣𑀓𑀲𑀼𑀢𑁆𑀢

(128.)

982\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀳𑀢𑁆𑀣𑀓𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑁄 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸—  “𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀞𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀑𑀲𑀻𑀤𑀢𑀺𑀫𑁂𑀯 𑀲𑀁𑀲𑀻𑀤𑀢𑀺𑀫𑁂𑀯, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀲𑀡𑁆𑀞𑀸𑀢𑀼𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀧𑁆𑀧𑀺 𑀯𑀸 𑀢𑁂𑀮𑀁 𑀯𑀸 𑀯𑀸𑀮𑀼𑀓𑀸𑀬 𑀆𑀲𑀺𑀢𑁆𑀢𑀁 𑀑𑀲𑀻𑀤𑀢𑀺𑀫𑁂𑀯 𑀲𑀁𑀲𑀻𑀤𑀢𑀺𑀫𑁂𑀯, 𑀦 𑀲𑀡𑁆𑀞𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀳𑀢𑁆𑀣𑀓𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀞𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀑𑀲𑀻𑀤𑀢𑀺𑀫𑁂𑀯 𑀲𑀁𑀲𑀻𑀤𑀢𑀺𑀫𑁂𑀯, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀲𑀡𑁆𑀞𑀸𑀢𑀼𑀁𑁇

983\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀳𑀢𑁆𑀣𑀓𑀁 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀑𑀍𑀆𑀭𑀺𑀓𑀁, 𑀳𑀢𑁆𑀣𑀓, 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺, 𑀔𑁄 𑀳𑀢𑁆𑀣𑀓𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀳𑀢𑁆𑀣𑀓𑀁 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

984\. “𑀬𑁂 𑀢𑁂, 𑀳𑀢𑁆𑀣𑀓, 𑀥𑀫𑁆𑀫𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀧𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀢𑀭𑀳𑀺 𑀧𑀯𑀢𑁆𑀢𑀺𑀦𑁄”𑀢𑀺? “𑀬𑁂 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀧𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀢𑁂 𑀘 𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀢𑀭𑀳𑀺 𑀧𑀯𑀢𑁆𑀢𑀺𑀦𑁄; 𑀬𑁂 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀦𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀺𑀦𑁄 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀢𑁂 𑀘 𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀏𑀢𑀭𑀳𑀺 𑀧𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀏𑀢𑀭𑀳𑀺 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀳𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁂𑀳𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺𑁇 𑀤𑀽𑀭𑀢𑁄𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀳𑀢𑁆𑀣𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 ‘𑀥𑀫𑁆𑀫𑀁 𑀲𑁄𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑀺𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀢𑀺𑀡𑁆𑀡𑀁? 𑀪𑀕𑀯𑀢𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀅𑀢𑀺𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄; 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀲𑁆𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑀺𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄; 𑀲𑀁𑀖𑀲𑁆𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀢𑀺𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑀺𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑀢𑀺𑁇

985\. _𑀦𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲,_  
_𑀢𑀺𑀢𑁆𑀢𑀺𑀫𑀚𑁆𑀛𑀕𑀸 𑀓𑀼𑀤𑀸𑀘𑀦𑀁;_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑀲𑁆𑀲 𑀘𑁇_  


986\. _𑀅𑀥𑀺𑀲𑀻𑀮𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑁄,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑀯𑀦𑁂 𑀭𑀢𑁄;_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑀺𑀢𑁆𑀢𑁄,_  
_𑀳𑀢𑁆𑀣𑀓𑁄 𑀅𑀯𑀺𑀳𑀁 𑀕𑀢𑁄”𑀢𑀺𑁇_  


---

987\. 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
