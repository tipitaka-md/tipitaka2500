# 𑁩𑁇𑁩𑁇𑁯 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀲𑀼𑀢𑁆𑀢

(132.)

1005\. “𑀢𑀻𑀡𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀺 𑀆𑀯𑀳𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀯𑀺𑀯𑀝𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺? 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁄 𑀆𑀯𑀳𑀢𑀺, 𑀦𑁄 𑀯𑀺𑀯𑀝𑁄; 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁆𑀢𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸 𑀆𑀯𑀳𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀯𑀺𑀯𑀝𑀸; 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸 𑀆𑀯𑀳𑀢𑀺, 𑀦𑁄 𑀯𑀺𑀯𑀝𑀸𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀡𑀺 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀺 𑀆𑀯𑀳𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀯𑀺𑀯𑀝𑀸𑀦𑀺𑁇

1006\. 𑀢𑀻𑀡𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀯𑀝𑀸𑀦𑀺 𑀯𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀻𑀡𑀺? 𑀘𑀦𑁆𑀤𑀫𑀡𑁆𑀟𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀯𑀝𑀁 𑀯𑀺𑀭𑁄𑀘𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁; 𑀲𑀽𑀭𑀺𑀬𑀫𑀡𑁆𑀟𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀯𑀝𑀁 𑀯𑀺𑀭𑁄𑀘𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁; 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀯𑀝𑁄 𑀯𑀺𑀭𑁄𑀘𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁄𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀡𑀺 𑀯𑀺𑀯𑀝𑀸𑀦𑀺 𑀯𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀦𑀻”𑀢𑀺𑁇
