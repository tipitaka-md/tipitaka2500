# 𑁩𑁇𑁫𑁇𑁩 𑀯𑀺𑀲𑀫𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀫𑀗𑁆𑀕𑀮𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀲𑀫𑀲𑀼𑀢𑁆𑀢

(149.)

1069\. “𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀯𑀺𑀲𑀫𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀯𑀺𑀲𑀫𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀯𑀺𑀲𑀫𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇

1070\. 𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀫𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀲𑀫𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀲𑀫𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇
