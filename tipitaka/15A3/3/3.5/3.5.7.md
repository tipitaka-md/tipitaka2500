# 𑁩𑁇𑁫𑁇𑁭 𑀢𑀢𑀺𑀬𑀔𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑀗𑁆𑀕𑀮𑀯𑀕𑁆𑀕

### 𑀢𑀢𑀺𑀬𑀔𑀢𑀲𑀼𑀢𑁆𑀢

(153.)

1081. “𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀯𑀺𑀲𑀫𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀯𑀺𑀲𑀫𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀯𑀺𑀲𑀫𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦…𑀧𑁂… 𑁇

1082. 𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑀫𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀲𑀫𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀲𑀫𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦…𑀧𑁂… 𑁇
