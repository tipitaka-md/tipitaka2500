# 𑁩𑁇𑁫𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀔𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀫𑀗𑁆𑀕𑀮𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀔𑀢𑀲𑀼𑀢𑁆𑀢

(152.)

1078\. “𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑀸𑀯𑀚𑁆𑀚𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀲𑀸𑀯𑀚𑁆𑀚𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀲𑀸𑀯𑀚𑁆𑀚𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦…𑀧𑁂… 𑁇

1079\. 𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀅𑀦𑀯𑀚𑁆𑀚𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀅𑀦𑀯𑀚𑁆𑀚𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀅𑀦𑀯𑀚𑁆𑀚𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦…𑀧𑁂… 𑁇
