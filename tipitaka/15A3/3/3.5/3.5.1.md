# 𑁩𑁇𑁫𑁇𑁧 𑀅𑀓𑀼𑀲𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑀗𑁆𑀕𑀮𑀯𑀕𑁆𑀕

### 𑀅𑀓𑀼𑀲𑀮𑀲𑀼𑀢𑁆𑀢

(147.)

1063. “𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀅𑀓𑀼𑀲𑀮𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀅𑀓𑀼𑀲𑀮𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀅𑀓𑀼𑀲𑀮𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇

1064. 𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀓𑀼𑀲𑀮𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑁂𑀦, 𑀓𑀼𑀲𑀮𑁂𑀦 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑁂𑀦, 𑀓𑀼𑀲𑀮𑁂𑀦 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑁂𑀦—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇
