# 𑁩𑁇𑁨𑁇𑁧 𑀆𑀧𑀸𑀬𑀺𑀓𑀲𑀼𑀢𑁆𑀢

908\. “𑀢𑀬𑁄𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸 𑀇𑀤𑀫𑀧𑁆𑀧𑀳𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄? 𑀬𑁄 𑀘 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀧𑀝𑀺𑀜𑁆𑀜𑁄, 𑀬𑁄 𑀘 𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺, 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺—  ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑁂𑀲𑀼 𑀤𑁄𑀲𑁄’𑀢𑀺, 𑀲𑁄 𑀢𑀸𑀬 𑀓𑀸𑀫𑁂𑀲𑀼 𑀧𑀸𑀢𑀩𑁆𑀬𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀬𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸 𑀇𑀤𑀫𑀧𑁆𑀧𑀳𑀸𑀬𑀸”𑀢𑀺𑁇
