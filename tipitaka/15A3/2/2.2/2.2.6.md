# 𑁨𑁇𑁨𑁇𑁬 𑀲𑀸𑀍𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀍𑀳𑀲𑀼𑀢𑁆𑀢

(67.)

526. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑀓𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑁂 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀍𑀳𑁄 𑀘 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸 𑀲𑀸𑀡𑁄 𑀘 𑀲𑁂𑀔𑀼𑀦𑀺𑀬𑀦𑀢𑁆𑀢𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀓𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀲𑀸𑀍𑀳𑀁 𑀫𑀺𑀕𑀸𑀭𑀦𑀢𑁆𑀢𑀸𑀭𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑀓𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

527. “𑀏𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀫𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑁂𑀦, 𑀫𑀸 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀬, 𑀫𑀸 𑀇𑀢𑀺𑀓𑀺𑀭𑀸𑀬, 𑀫𑀸 𑀧𑀺𑀝𑀓𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑁂𑀦, 𑀫𑀸 𑀢𑀓𑁆𑀓𑀳𑁂𑀢𑀼, 𑀫𑀸 𑀦𑀬𑀳𑁂𑀢𑀼, 𑀫𑀸 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀦, 𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸, 𑀫𑀸 𑀪𑀩𑁆𑀩𑀭𑀽𑀧𑀢𑀸𑀬, 𑀫𑀸 𑀲𑀫𑀡𑁄 𑀦𑁄 𑀕𑀭𑀽𑀢𑀺𑁇 𑀬𑀤𑀸 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀣 ‘𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀯𑀚𑁆𑀚𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀼𑀕𑀭𑀳𑀺𑀢𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀅𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀸𑀣𑁇

528. 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀪𑁄”𑀢𑀺?

529. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

530. “𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀢𑀺 𑀔𑁄 𑀅𑀳𑀁, 𑀲𑀸𑀍𑀳𑀸, 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀮𑀼𑀤𑁆𑀥𑁄 𑀔𑁄 𑀅𑀬𑀁, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀧𑀸𑀡𑀫𑁆𑀧𑀺 𑀳𑀦𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀧𑀭𑀤𑀸𑀭𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑀼𑀲𑀸𑀧𑀺 𑀪𑀡𑀢𑀺, 𑀧𑀭𑀫𑁆𑀧𑀺 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀬𑀁 𑀲 𑀳𑁄𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

531. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

532. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁄𑀲𑁄”𑀢𑀺?

533. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

534. “𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄𑀢𑀺 𑀔𑁄 𑀅𑀳𑀁, 𑀲𑀸𑀍𑀳𑀸, 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀤𑀼𑀝𑁆𑀞𑁄 𑀔𑁄 𑀅𑀬𑀁, 𑀲𑀸𑀍𑀳𑀸, 𑀩𑁆𑀬𑀸𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀸𑀡𑀫𑁆𑀧𑀺 𑀳𑀦𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀧𑀭𑀤𑀸𑀭𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑀼𑀲𑀸𑀧𑀺 𑀪𑀡𑀢𑀺, 𑀧𑀭𑀫𑁆𑀧𑀺 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀬𑀁 𑀲 𑀳𑁄𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

535. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

536. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀢𑁆𑀣𑀺 𑀫𑁄𑀳𑁄”𑀢𑀺?

537. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

538. “𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀢𑀺 𑀔𑁄 𑀅𑀳𑀁, 𑀲𑀸𑀍𑀳𑀸, 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀫𑀽𑀍𑀳𑁄 𑀔𑁄 𑀅𑀬𑀁, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀕𑀢𑁄 𑀧𑀸𑀡𑀫𑁆𑀧𑀺 𑀳𑀦𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀧𑀭𑀤𑀸𑀭𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑀼𑀲𑀸𑀧𑀺 𑀪𑀡𑀢𑀺, 𑀧𑀭𑀫𑁆𑀧𑀺 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀬𑀁 𑀲 𑀳𑁄𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

539. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇

540. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣, 𑀲𑀸𑀍𑀳𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸 𑀯𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀯𑀸”𑀢𑀺?

541. “𑀅𑀓𑀼𑀲𑀮𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇

542. “𑀲𑀸𑀯𑀚𑁆𑀚𑀸 𑀯𑀸 𑀅𑀦𑀯𑀚𑁆𑀚𑀸 𑀯𑀸”𑀢𑀺?

543. “𑀲𑀸𑀯𑀚𑁆𑀚𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇

544. “𑀯𑀺𑀜𑁆𑀜𑀼𑀕𑀭𑀳𑀺𑀢𑀸 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑀸 𑀯𑀸”𑀢𑀺?

545. “𑀯𑀺𑀜𑁆𑀜𑀼𑀕𑀭𑀳𑀺𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂”𑁇

546. “𑀲𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀯𑀸? 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺?

547. “𑀲𑀫𑀢𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻𑀢𑀺𑁇 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

548. “𑀇𑀢𑀺 𑀔𑁄, 𑀲𑀸𑀍𑀳𑀸, 𑀬𑀁 𑀢𑀁 𑀅𑀯𑁄𑀘𑀼𑀫𑁆𑀳𑀸—  ‘𑀏𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀫𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑁂𑀦, 𑀫𑀸 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀬, 𑀫𑀸 𑀇𑀢𑀺𑀓𑀺𑀭𑀸𑀬, 𑀫𑀸 𑀧𑀺𑀝𑀓𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑁂𑀦, 𑀫𑀸 𑀢𑀓𑁆𑀓𑀳𑁂𑀢𑀼, 𑀫𑀸 𑀦𑀬𑀳𑁂𑀢𑀼, 𑀫𑀸 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀦, 𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸, 𑀫𑀸 𑀪𑀩𑁆𑀩𑀭𑀽𑀧𑀢𑀸𑀬, 𑀫𑀸 𑀲𑀫𑀡𑁄 𑀦𑁄 𑀕𑀭𑀽𑀢𑀺𑁇 𑀬𑀤𑀸 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀣—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀸𑀯𑀚𑁆𑀚𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀼𑀕𑀭𑀳𑀺𑀢𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻𑀢𑀺, 𑀅𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀧𑀚𑀳𑁂𑀬𑁆𑀬𑀸𑀣𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇

568. “𑀲𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀯𑀸? 𑀓𑀣𑀁 𑀯𑀸 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺?

569. “𑀲𑀫𑀢𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻𑀢𑀺𑁇 𑀏𑀯𑀁 𑀦𑁄 𑀏𑀢𑁆𑀣 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

570. “𑀇𑀢𑀺 𑀔𑁄, 𑀲𑀸𑀍𑀳𑀸, 𑀬𑀁 𑀢𑀁 𑀅𑀯𑁄𑀘𑀼𑀫𑁆𑀳𑀸—  ‘𑀏𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀫𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀯𑁂𑀦, 𑀫𑀸 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀬, 𑀫𑀸 𑀇𑀢𑀺𑀓𑀺𑀭𑀸𑀬, 𑀫𑀸 𑀧𑀺𑀝𑀓𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑁂𑀦, 𑀫𑀸 𑀢𑀓𑁆𑀓𑀳𑁂𑀢𑀼, 𑀫𑀸 𑀦𑀬𑀳𑁂𑀢𑀼, 𑀫𑀸 𑀆𑀓𑀸𑀭𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀦, 𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀚𑁆𑀛𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀬𑀸, 𑀫𑀸 𑀪𑀩𑁆𑀩𑀭𑀽𑀧𑀢𑀸𑀬, 𑀫𑀸 𑀲𑀫𑀡𑁄 𑀦𑁄 𑀕𑀭𑀽𑀢𑀺𑁇 𑀬𑀤𑀸 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀣—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀯𑀚𑁆𑀚𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑁆𑀧𑀲𑀢𑁆𑀣𑀸, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻𑀢𑀺, 𑀅𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀲𑀸𑀍𑀳𑀸, 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇

571. 𑀲 𑀔𑁄 𑀲𑁄, 𑀲𑀸𑀍𑀳𑀸, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀏𑀯𑀁 𑀯𑀺𑀕𑀢𑀸𑀪𑀺𑀚𑁆𑀛𑁄 𑀯𑀺𑀕𑀢𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸…𑀧𑁂…  𑀓𑀭𑀼𑀡𑀸…  𑀫𑀼𑀤𑀺𑀢𑀸…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀏𑀓𑀁 𑀤𑀺𑀲𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀣𑀸 𑀤𑀼𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀢𑀢𑀺𑀬𑀁, 𑀢𑀣𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁, 𑀇𑀢𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀛𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀇𑀤𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀳𑀻𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀡𑀻𑀢𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀇𑀫𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀕𑀢𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀺 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀦𑁆’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀚𑀸𑀦𑀢𑁄 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀢𑁄 𑀓𑀸𑀫𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀯𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀯𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺; 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺𑁇 ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

572. 𑀲𑁄 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀅𑀳𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀮𑁄𑀪𑁄, 𑀢𑀤𑀳𑀼 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑁄 𑀏𑀢𑀭𑀳𑀺 𑀦𑀢𑁆𑀣𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁; 𑀅𑀳𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀤𑁄𑀲𑁄…𑀧𑁂…  𑀅𑀳𑀼 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑁄𑀳𑁄, 𑀢𑀤𑀳𑀼 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑁄 𑀏𑀢𑀭𑀳𑀺 𑀦𑀢𑁆𑀣𑀺, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀦𑀺𑀘𑁆𑀙𑀸𑀢𑁄 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑁄 𑀲𑀼𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀪𑀽𑀢𑁂𑀦 𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇
