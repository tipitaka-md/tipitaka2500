# 𑁨𑁇𑁪𑁇𑁯 𑀧𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀲𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀬𑀲𑀼𑀢𑁆𑀢

(90.)

762\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸, 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸, 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑁇

763\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀲𑀻𑀮𑀲𑀺𑀓𑁆𑀔𑀸𑁇

764\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑁇

765\. 𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁆𑀔𑀸? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁆𑀔𑀸𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀺𑀓𑁆𑀔𑀸”𑀢𑀺𑁇
