# 𑁨𑁇𑁪𑁇𑁫 𑀲𑁂𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀲𑁂𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

(86.)

740. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

741. “‘𑀲𑁂𑀔𑁄, 𑀲𑁂𑀔𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁂𑀔𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀺𑀓𑁆𑀔𑀢𑀻𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀢𑀲𑁆𑀫𑀸 𑀲𑁂𑀔𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀜𑁆𑀘 𑀲𑀺𑀓𑁆𑀔𑀢𑀺? 𑀅𑀥𑀺𑀲𑀻𑀮𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺, 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇 𑀲𑀺𑀓𑁆𑀔𑀢𑀻𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀢𑀲𑁆𑀫𑀸 𑀲𑁂𑀔𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇

742. _𑀲𑁂𑀔𑀲𑁆𑀲 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀉𑀚𑀼𑀫𑀕𑁆𑀕𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄;_  
_𑀔𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀞𑀫𑀁 𑀜𑀸𑀡𑀁,_  
_𑀢𑀢𑁄 𑀅𑀜𑁆𑀜𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑁇_  


743. _𑀢𑀢𑁄 𑀅𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀜𑀸𑀡𑀁 𑀯𑁂 𑀳𑁄𑀢𑀺 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀫𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀻𑀢𑀺,_  
_𑀪𑀯𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁂”𑀢𑀺𑁇_
