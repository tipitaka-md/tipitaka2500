# 𑁨𑁇𑁩𑁇𑁬 𑀧𑀞𑀫𑀪𑀯𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀆𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀯𑀲𑀼𑀢𑁆𑀢

(77.)

690\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀯𑁄, 𑀪𑀯𑁄𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀯𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺?

691\. “𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀯𑁂𑀧𑀓𑁆𑀓𑀜𑁆𑀘, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀁 𑀦𑀸𑀪𑀯𑀺𑀲𑁆𑀲, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀓𑀸𑀫𑀪𑀯𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁂𑀢𑁆𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀩𑀻𑀚𑀁, 𑀢𑀡𑁆𑀳𑀸 𑀲𑁆𑀦𑁂𑀳𑁄𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀳𑀻𑀦𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀏𑀯𑀁 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 ( )

692\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀯𑁂𑀧𑀓𑁆𑀓𑀜𑁆𑀘, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀁 𑀦𑀸𑀪𑀯𑀺𑀲𑁆𑀲, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀭𑀽𑀧𑀪𑀯𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁂𑀢𑁆𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀩𑀻𑀚𑀁, 𑀢𑀡𑁆𑀳𑀸 𑀲𑁆𑀦𑁂𑀳𑁄𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀏𑀯𑀁 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 ( )

693\. 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀯𑁂𑀧𑀓𑁆𑀓𑀜𑁆𑀘, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀁 𑀦𑀸𑀪𑀯𑀺𑀲𑁆𑀲, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀭𑀽𑀧𑀪𑀯𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀣𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁂𑀢𑁆𑀢𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀩𑀻𑀚𑀁, 𑀢𑀡𑁆𑀳𑀸 𑀲𑁆𑀦𑁂𑀳𑁄𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀡𑁆𑀳𑀸𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀡𑀻𑀢𑀸𑀬 𑀥𑀸𑀢𑀼𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀏𑀯𑀁 𑀆𑀬𑀢𑀺𑀁 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀺 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀯𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
