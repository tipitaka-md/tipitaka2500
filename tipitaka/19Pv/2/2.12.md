# 𑁨𑁇𑁧𑁨 𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑀧𑁂𑀢𑀺𑀯𑀢𑁆𑀣𑀼

* 𑀉𑀩𑁆𑀩𑀭𑀺𑀯𑀕𑁆𑀕

* 𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑀧𑁂𑀢𑀺𑀯𑀢𑁆𑀣𑀼

(348.)

376\. _“𑀲𑁄𑀡𑁆𑀡𑀲𑁄𑀧𑀸𑀦𑀨𑀮𑀓𑀸,_  
_𑀲𑁄𑀡𑁆𑀡𑀯𑀸𑀮𑀼𑀓𑀲𑀦𑁆𑀣𑀢𑀸;_  
_𑀢𑀢𑁆𑀣 𑀲𑁄𑀕𑀦𑁆𑀥𑀺𑀬𑀸 𑀯𑀕𑁆𑀕𑀽,_  
_𑀲𑀼𑀘𑀺𑀕𑀦𑁆𑀥𑀸 𑀫𑀦𑁄𑀭𑀫𑀸𑁇_  


(349.)

377\. _𑀦𑀸𑀦𑀸𑀭𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀦𑀸𑀦𑀸𑀕𑀦𑁆𑀥𑀲𑀫𑁂𑀭𑀺𑀢𑀸;_  
_𑀦𑀸𑀦𑀸𑀧𑀤𑀼𑀫𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀧𑀼𑀡𑁆𑀟𑀭𑀻𑀓𑀲𑀫𑁄𑀢𑀢𑀸𑁇_  


(350.)

378\. _𑀲𑀼𑀭𑀪𑀺𑀁 𑀲𑀫𑁆𑀧𑀯𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀫𑀦𑀼𑀜𑁆𑀜𑀸 𑀫𑀸𑀮𑀼𑀢𑁂𑀭𑀺𑀢𑀸;_  
_𑀳𑀁𑀲𑀓𑁄𑀜𑁆𑀘𑀸𑀪𑀺𑀭𑀼𑀤𑀸 𑀘,_  
_𑀘𑀓𑁆𑀓𑀯𑀓𑁆𑀓𑀸𑀪𑀺𑀓𑀽𑀚𑀺𑀢𑀸𑁇_  


(351.)

379\. _𑀦𑀸𑀦𑀸𑀤𑀺𑀚𑀕𑀡𑀸𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀦𑀸𑀦𑀸𑀲𑀭𑀕𑀡𑀸𑀬𑀼𑀢𑀸;_  
_𑀦𑀸𑀦𑀸𑀨𑀮𑀥𑀭𑀸 𑀭𑀼𑀓𑁆𑀔𑀸,_  
_𑀦𑀸𑀦𑀸𑀧𑀼𑀧𑁆𑀨𑀥𑀭𑀸 𑀯𑀦𑀸𑁇_  


(352.)

380\. _𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀈𑀤𑀺𑀲𑀁,_  
_𑀦𑀕𑀭𑀁 𑀬𑀸𑀤𑀺𑀲𑀁 𑀇𑀤𑀁;_  
_𑀧𑀸𑀲𑀸𑀤𑀸 𑀩𑀳𑀼𑀓𑀸 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑀭𑀽𑀧𑀺𑀬𑀸𑀫𑀬𑀸;_  
_𑀤𑀤𑁆𑀤𑀮𑁆𑀮𑀫𑀸𑀦𑀸 𑀆𑀪𑁂𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀘𑀢𑀼𑀭𑁄 𑀤𑀺𑀲𑀸𑁇_  


(353.)

381\. _𑀧𑀜𑁆𑀘 𑀤𑀸𑀲𑀺𑀲𑀢𑀸 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀬𑀸 𑀢𑁂𑀫𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀓𑀸;_  
_𑀢𑀸 𑀓𑀫𑁆𑀩𑀼𑀓𑀸𑀬𑀽𑀭𑀥𑀭𑀸,_  
_𑀓𑀜𑁆𑀘𑀦𑀸𑀯𑁂𑀍𑀅𑀪𑀽𑀲𑀺𑀢𑀸𑁇_  


(354.)

382\. _𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀸 𑀩𑀳𑀼𑀓𑀸 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑀭𑀽𑀧𑀺𑀬𑀸𑀫𑀬𑀸;_  
_𑀓𑀤𑀮𑀺𑀫𑀺𑀕𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀲𑀚𑁆𑀚𑀸 𑀕𑁄𑀦𑀓𑀲𑀦𑁆𑀣𑀢𑀸𑁇_  


(355.)

383\. _𑀬𑀢𑁆𑀣 𑀢𑁆𑀯𑀁 𑀯𑀸𑀲𑀽𑀧𑀕𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀦𑀻;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀸𑀬𑀟𑁆𑀠𑀭𑀢𑁆𑀢𑀸𑀬,_  
_𑀢𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬 𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(356.)

384\. _𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑁄𑀓𑁆𑀔𑀭𑀜𑁆𑀜𑀸 𑀲𑀫𑀦𑁆𑀢𑀢𑁄;_  
_𑀢𑀲𑁆𑀲𑀸 𑀢𑀻𑀭𑁂 𑀢𑀼𑀯𑀁 𑀞𑀸𑀲𑀺,_  
_𑀳𑀭𑀺𑀢𑁂 𑀲𑀤𑁆𑀤𑀮𑁂 𑀲𑀼𑀪𑁂𑁇_  


(357.)

385\. _𑀢𑀢𑁄 𑀢𑁂 𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑁄 𑀲𑀼𑀦𑀔𑁄,_  
_𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀔𑀸𑀤𑀢𑀺;_  
_𑀬𑀤𑀸 𑀘 𑀔𑀸𑀬𑀺𑀢𑀸 𑀆𑀲𑀺,_  
_𑀅𑀝𑁆𑀞𑀺𑀲𑀗𑁆𑀔𑀮𑀺𑀓𑀸 𑀓𑀢𑀸;_  
_𑀑𑀕𑀸𑀳𑀲𑀺 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁,_  
_𑀳𑁄𑀢𑀺 𑀓𑀸𑀬𑁄 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂𑁇_  


(358.)

386\. _𑀢𑀢𑁄 𑀢𑁆𑀯𑀁 𑀅𑀗𑁆𑀕𑀧𑀘𑁆𑀘𑀗𑁆𑀕𑀻,_  
_𑀲𑀼𑀘𑀸𑀭𑀼 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀯𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀭𑀼𑀧𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀬𑀸𑀲𑀺 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(359.)

387\. _𑀓𑀺𑀁 𑀦𑀼 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬,_  
_𑀫𑀦𑀲𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀢𑀁;_  
_𑀓𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑁄 𑀲𑀼𑀦𑀔𑁄 𑀢𑀯;_  
_𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀔𑀸𑀤𑀢𑀻”𑀢𑀺𑁇_  


(360.)

388\. _“𑀓𑀺𑀫𑀺𑀮𑀸𑀬𑀁 𑀕𑀳𑀧𑀢𑀺,_  
_𑀲𑀤𑁆𑀥𑁄 𑀆𑀲𑀺 𑀉𑀧𑀸𑀲𑀓𑁄;_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀪𑀭𑀺𑀬𑀸 𑀆𑀲𑀺𑀁,_  
_𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀅𑀢𑀺𑀘𑀸𑀭𑀺𑀦𑀻𑁇_  


(361.)

389\. _𑀲𑁄 𑀫𑀁 𑀅𑀢𑀺𑀘𑀭𑀫𑀸𑀦𑀸𑀬,_  
_𑀲𑀸𑀫𑀺𑀓𑁄 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀺;_  
_‘𑀦𑁂𑀢𑀁 𑀙𑀦𑁆𑀦𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀅𑀢𑀺𑀘𑀭𑀸𑀲𑀺 𑀫𑀁’𑁇_  


(362.)

390\. _𑀲𑀸𑀳𑀁 𑀖𑁄𑀭𑀜𑁆𑀘 𑀲𑀧𑀣𑀁,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀜𑁆𑀘 𑀪𑀸𑀲𑀺𑀲𑀁;_  
_‘𑀦𑀸𑀳𑀁 𑀢𑀁 𑀅𑀢𑀺𑀘𑀭𑀸𑀫𑀺,_  
_𑀓𑀸𑀬𑁂𑀦 𑀉𑀤 𑀘𑁂𑀢𑀲𑀸𑁇_  


(363.)

391\. _𑀲𑀘𑀸𑀳𑀁 𑀢𑀁 𑀅𑀢𑀺𑀘𑀭𑀸𑀫𑀺,_  
_𑀓𑀸𑀬𑁂𑀦 𑀉𑀤 𑀘𑁂𑀢𑀲𑀸;_  
_𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑁄𑀬𑀁 𑀲𑀼𑀦𑀔𑁄,_  
_𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀔𑀸𑀤𑀢𑀼’𑁇_  


(364.)

392\. _𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀁,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀲𑁆𑀲 𑀘𑀽𑀪𑀬𑀁;_  
_𑀲𑀢𑁆𑀢𑁂𑀯 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺,_  
_𑀅𑀦𑀼𑀪𑀽𑀢𑀁 𑀬𑀢𑁄 𑀳𑀺 𑀫𑁂;_  
_𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑁄 𑀘 𑀲𑀼𑀦𑀔𑁄,_  
_𑀅𑀗𑁆𑀕𑀫𑀗𑁆𑀕𑀸𑀦𑀺 𑀔𑀸𑀤𑀢𑀺𑁇_  


(365.)

393\. _𑀢𑁆𑀯𑀜𑁆𑀘 𑀤𑁂𑀯 𑀩𑀳𑀼𑀓𑀸𑀭𑁄,_  
_𑀅𑀢𑁆𑀣𑀸𑀬 𑀫𑁂 𑀇𑀥𑀸𑀕𑀢𑁄;_  
_𑀲𑀼𑀫𑀼𑀢𑁆𑀢𑀸𑀳𑀁 𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑀲𑁆𑀲,_  
_𑀅𑀲𑁄𑀓𑀸 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑀸𑁇_  


(366.)

394\. _𑀢𑀸𑀳𑀁 𑀤𑁂𑀯 𑀦𑀫𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀸𑀘𑀸𑀫𑀺 𑀧𑀜𑁆𑀚𑀮𑀻𑀓𑀢𑀸;_  
_𑀪𑀼𑀜𑁆𑀚 𑀅𑀫𑀸𑀦𑀼𑀲𑁂 𑀓𑀸𑀫𑁂,_  
_𑀭𑀫 𑀤𑁂𑀯 𑀫𑀬𑀸 𑀲𑀳𑀸”𑀢𑀺𑁇_  


(367.)

395\. _“𑀪𑀼𑀢𑁆𑀢𑀸 𑀅𑀫𑀸𑀦𑀼𑀲𑀸 𑀓𑀸𑀫𑀸,_  
_𑀭𑀫𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀢𑀬𑀸 𑀲𑀳;_  
_𑀢𑀸𑀳𑀁 𑀲𑀼𑀪𑀕𑁂 𑀬𑀸𑀘𑀸𑀫𑀺,_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀦𑀬𑀸𑀳𑀺 𑀫𑀦𑁆”𑀢𑀺𑁇_  


---

396\. 𑀓𑀡𑁆𑀡𑀫𑀼𑀡𑁆𑀟𑀧𑁂𑀢𑀺𑀯𑀢𑁆𑀣𑀼 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁𑁇
