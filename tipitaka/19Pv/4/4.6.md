# 𑁪𑁇𑁬 𑀓𑀼𑀫𑀸𑀭𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀫𑀸𑀭𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

(746.)

799. _“𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁,_  
_𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀢𑁄;_  
_𑀢𑀢𑁆𑀣 𑀆𑀲𑀼𑀁 𑀤𑁆𑀯𑁂 𑀓𑀼𑀫𑀸𑀭𑀸,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁𑁇_  


(747.)

800. _𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑁂𑀲𑀼,_  
_𑀓𑀸𑀫𑀲𑁆𑀲𑀸𑀤𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀦𑁄;_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀔𑁂 𑀕𑀺𑀤𑁆𑀥𑀸,_  
_𑀦 𑀢𑁂 𑀧𑀲𑁆𑀲𑀺𑀁𑀲𑀼𑀦𑀸𑀕𑀢𑀁𑁇_  


(748.)

801. _𑀢𑁂 𑀘𑀼𑀢𑀸 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀸,_  
_𑀧𑀭𑀮𑁄𑀓𑀁 𑀇𑀢𑁄 𑀕𑀢𑀸;_  
_𑀢𑁂𑀥 𑀖𑁄𑀲𑁂𑀦𑁆𑀢𑁆𑀬𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


(749.)

802. _𑀩𑀳𑀽𑀲𑀼 𑀯𑀢 𑀲𑀦𑁆𑀢𑁂𑀲𑀼,_  
_𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁂 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁂;_  
_𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀫𑁆𑀳𑀸 𑀘 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀓𑀸𑀢𑀼𑀁 𑀲𑀼𑀔𑀸𑀯𑀳𑀁𑁇_  


(750.)

803. _𑀓𑀺𑀁 𑀢𑀢𑁄 𑀧𑀸𑀧𑀓𑀁 𑀅𑀲𑁆𑀲,_  
_𑀬𑀁 𑀦𑁄 𑀭𑀸𑀚𑀓𑀼𑀮𑀸 𑀘𑀼𑀢𑀸;_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁,_  
_𑀔𑀼𑀧𑁆𑀧𑀺𑀧𑀸𑀲𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(751.)

804. _𑀲𑀸𑀫𑀺𑀦𑁄 𑀇𑀥 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀲𑀸𑀫𑀺𑀦𑁄 𑀢𑀳𑀺𑀁;_  
_𑀪𑀫𑀦𑁆𑀢𑀺 𑀔𑀼𑀧𑁆𑀧𑀺𑀧𑀸𑀲𑀸𑀬,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀦𑁆𑀦𑀢𑁄𑀦𑀢𑀸𑁇_  


(752.)

805. _𑀏𑀢𑀫𑀸𑀤𑀻𑀦𑀯𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀇𑀲𑁆𑀲𑀭𑀫𑀤𑀲𑀫𑁆𑀪𑀯𑀁;_  
_𑀧𑀳𑀸𑀬 𑀇𑀲𑁆𑀲𑀭𑀫𑀤𑀁,_  
_𑀪𑀯𑁂 𑀲𑀕𑁆𑀕𑀕𑀢𑁄 𑀦𑀭𑁄;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀕𑁆𑀕𑀁 𑀲𑁄 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


---

806. 𑀓𑀼𑀫𑀸𑀭𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼 𑀙𑀝𑁆𑀞𑀁𑁇
