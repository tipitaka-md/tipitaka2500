# 𑁪𑁇𑁧 𑀅𑀫𑁆𑀩𑀲𑀓𑁆𑀓𑀭𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀅𑀫𑁆𑀩𑀲𑀓𑁆𑀓𑀭𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

(517.)

564\. _𑀯𑁂𑀲𑀸𑀮𑀻 𑀦𑀸𑀫 𑀦𑀕𑀭𑀢𑁆𑀣𑀺 𑀯𑀚𑁆𑀚𑀻𑀦𑀁,_  
_𑀢𑀢𑁆𑀣 𑀅𑀳𑀼 𑀮𑀺𑀘𑁆𑀙𑀯𑀺 𑀅𑀫𑁆𑀩𑀲𑀓𑁆𑀓𑀭𑁄;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀧𑁂𑀢𑀁 𑀦𑀕𑀭𑀲𑁆𑀲 𑀩𑀸𑀳𑀺𑀭𑀁,_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀣 𑀢𑀁 𑀓𑀸𑀭𑀡𑀢𑁆𑀣𑀺𑀓𑁄𑁇_  


(518.)

565\. _“𑀲𑁂𑀬𑁆𑀬𑀸 𑀦𑀺𑀲𑀚𑁆𑀚𑀸 𑀦𑀬𑀺𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺,_  
_𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑁄 𑀦𑀢𑁆𑀣𑀺 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁄 𑀘;_  
_𑀅𑀲𑀺𑀢𑀧𑀻𑀢𑀔𑀸𑀬𑀺𑀢𑀯𑀢𑁆𑀣𑀪𑁄𑀕𑀸,_  
_𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀓𑀸 𑀲𑀸𑀧𑀺 𑀇𑀫𑀲𑁆𑀲 𑀦𑀢𑁆𑀣𑀺𑁇_  


(519.)

566\. _𑀬𑁂 𑀜𑀸𑀢𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑀸 𑀲𑀼𑀳𑀚𑁆𑀚𑀸,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑀸 𑀬𑀲𑁆𑀲 𑀅𑀳𑁂𑀲𑀼𑀁 𑀧𑀼𑀩𑁆𑀩𑁂;_  
_𑀤𑀝𑁆𑀞𑀼𑀫𑁆𑀧𑀺 𑀢𑁂 𑀤𑀸𑀦𑀺 𑀦 𑀢𑀁 𑀮𑀪𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀭𑀸𑀚𑀺𑀢𑀢𑁆𑀢𑁄 𑀳𑀺 𑀚𑀦𑁂𑀦 𑀢𑁂𑀦𑁇_  


(520.)

567\. _𑀦 𑀑𑀕𑁆𑀕𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀯𑀦𑁆𑀢𑀺 𑀫𑀺𑀢𑁆𑀢𑀸,_  
_𑀚𑀳𑀦𑁆𑀢𑀺 𑀫𑀺𑀢𑁆𑀢𑀸 𑀯𑀺𑀓𑀮𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸;_  
_𑀅𑀢𑁆𑀣𑀜𑁆𑀘 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀬𑀦𑁆𑀢𑀺,_  
_𑀩𑀳𑀽 𑀫𑀺𑀢𑁆𑀢𑀸 𑀉𑀕𑁆𑀕𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀳𑁄𑀦𑁆𑀢𑀺𑁇_  


(521.)

568\. _𑀦𑀺𑀳𑀻𑀦𑀢𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀪𑁄𑀕𑁂𑀳𑀺 𑀓𑀺𑀘𑁆𑀙𑁄,_  
_𑀲𑀫𑁆𑀫𑀓𑁆𑀔𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀭𑀺𑀪𑀺𑀦𑁆𑀦𑀕𑀢𑁆𑀢𑁄;_  
_𑀉𑀲𑁆𑀲𑀸𑀯𑀩𑀺𑀦𑁆𑀤𑀽𑀯 𑀧𑀮𑀺𑀫𑁆𑀧𑀫𑀸𑀦𑁄,_  
_𑀅𑀚𑁆𑀚 𑀲𑀼𑀯𑁂 𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲𑀽𑀧𑀭𑁄𑀥𑁄𑁇_  


(522.)

569\. _𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀉𑀢𑁆𑀢𑀫𑀓𑀺𑀘𑁆𑀙𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀉𑀢𑁆𑀢𑀸𑀲𑀺𑀢𑀁 𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀽𑀮𑁂;_  
_‘𑀅𑀣 𑀢𑁆𑀯𑀁 𑀓𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀤𑁂𑀲𑀺 𑀬𑀓𑁆𑀔,_  
_𑀚𑀻𑀯 𑀪𑁄 𑀚𑀻𑀯𑀺𑀢𑀫𑁂𑀯 𑀲𑁂𑀬𑁆𑀬𑁄’”𑀢𑀺𑁇_  


(523.)

570\. _“𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑁄 𑀏𑀲 𑀅𑀳𑁄𑀲𑀺 𑀫𑀬𑁆𑀳𑀁,_  
_𑀅𑀳𑀁 𑀲𑀭𑀸𑀫𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀘 𑀫𑁂 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀫𑀳𑁄𑀲𑀺 𑀭𑀸𑀚,_  
_𑀫𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀭𑀬𑀁 𑀧𑀢𑀸𑀬𑀁𑁇_  


(524.)

571\. _𑀇𑀢𑁄 𑀘𑀼𑀢𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀺 𑀏𑀲 𑀧𑁄𑀲𑁄,_  
_𑀲𑀢𑁆𑀢𑀼𑀲𑁆𑀲𑀤𑀁 𑀦𑀺𑀭𑀬𑀁 𑀖𑁄𑀭𑀭𑀽𑀧𑀁;_  
_𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀻,_  
_𑀫𑀳𑀸𑀪𑀺𑀢𑀸𑀧𑀁 𑀓𑀝𑀼𑀓𑀁 𑀪𑀬𑀸𑀦𑀓𑀁𑁇_  


(525.)

572\. _𑀅𑀦𑁂𑀓𑀪𑀸𑀕𑁂𑀦 𑀕𑀼𑀡𑁂𑀦 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀅𑀬𑀫𑁂𑀯 𑀲𑀽𑀮𑁄 𑀦𑀺𑀭𑀬𑁂𑀦 𑀢𑁂𑀦;_  
_𑀏𑀓𑀦𑁆𑀢𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀝𑀼𑀓𑀁 𑀪𑀬𑀸𑀦𑀓𑀁,_  
_𑀏𑀓𑀦𑁆𑀢𑀢𑀺𑀩𑁆𑀩𑀁 𑀦𑀺𑀭𑀬𑀁 𑀧𑀢𑀸𑀬𑀁𑁇_  


(526.)

573\. _𑀇𑀤𑀜𑁆𑀘 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑀘𑀦𑀁 𑀫𑀫𑁂𑀲𑁄,_  
_𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀦𑀻𑀢𑁄 𑀯𑀺𑀚𑀳𑁂𑀬𑁆𑀬 𑀧𑀸𑀡𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀅𑀳𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦 𑀪𑀡𑀸𑀫𑀺,_  
_𑀫𑀸 𑀫𑁂 𑀓𑀢𑁄 𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲𑀽𑀧𑀭𑁄𑀥𑁄”𑁇_  


(527.)

574\. _“𑀅𑀜𑁆𑀜𑀸𑀢𑁄 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄,_  
_𑀅𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀇𑀘𑁆𑀙𑀸𑀫𑀲𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁 𑀢𑀼𑀯𑀁;_  
_𑀑𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀲𑀘𑁂 𑀦𑁄 𑀓𑀭𑁄𑀲𑀺,_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫 𑀢𑀁 𑀦𑁄 𑀦 𑀘 𑀓𑀼𑀚𑁆𑀛𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇_  


(528.)

575\. _“𑀅𑀤𑁆𑀥𑀸 𑀧𑀝𑀺𑀜𑁆𑀜𑀸 𑀫𑁂 𑀢𑀤𑀸 𑀅𑀳𑀼,_  
_𑀦𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀲𑁆𑀲 𑀳𑁄𑀢𑀺;_  
_𑀅𑀓𑀸𑀫𑀸 𑀲𑀤𑁆𑀥𑁂𑀬𑁆𑀬𑀯𑀘𑁄𑀢𑀺 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀼𑀘𑁆𑀙𑀲𑁆𑀲𑀼 𑀫𑀁 𑀓𑀸𑀫𑀁 𑀬𑀣𑀸 𑀯𑀺𑀲𑀬𑁆𑀳𑀦𑁆”𑀢𑀺𑁇_  


(529.)

576\. _“𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀳𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀢𑀸𑀳𑀁 𑀅𑀪𑀺𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀁;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀯 𑀢𑀁 𑀦𑁄𑀧𑀺 𑀘𑁂 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀁,_  
_𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀫𑁂 𑀬𑀓𑁆𑀔 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇_  


(530.)

577\. _“𑀲𑀘𑁆𑀘𑀧𑁆𑀧𑀝𑀺𑀜𑁆𑀜𑀸 𑀢𑀯 𑀫𑁂𑀲𑀸 𑀳𑁄𑀢𑀼,_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀥𑀫𑁆𑀫𑀁 𑀮𑀪 𑀲𑀼𑀧𑁆𑀧𑀲𑀸𑀤𑀁;_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀣𑀺𑀓𑁄 𑀦𑁄 𑀘 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀬𑀁 𑀢𑁂 𑀲𑀼𑀢𑀁 𑀅𑀲𑀼𑀢𑀜𑁆𑀘𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀁;_  
_𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀅𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀁 𑀬𑀣𑀸 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺”𑁇_  


(531.)

578\. _“𑀲𑁂𑀢𑁂𑀦 𑀅𑀲𑁆𑀲𑁂𑀦 𑀅𑀮𑀗𑁆𑀓𑀢𑁂𑀦,_  
_𑀉𑀧𑀬𑀸𑀲𑀺 𑀲𑀽𑀮𑀸𑀯𑀼𑀢𑀓𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀬𑀸𑀦𑀁 𑀇𑀤𑀁 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑁂𑀬𑁆𑀬𑀁,_  
_𑀓𑀺𑀲𑁆𑀲𑁂𑀢𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄”𑀢𑀺𑁇_  


(532.)

579\. _“𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀸 𑀦𑀕𑀭𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂,_  
_𑀘𑀺𑀓𑁆𑀔𑀮𑁆𑀮𑀫𑀕𑁆𑀕𑁂 𑀦𑀭𑀓𑀁 𑀅𑀳𑁄𑀲𑀺;_  
_𑀕𑁄𑀲𑀻𑀲𑀫𑁂𑀓𑀸𑀳𑀁 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑁂𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀭𑀓𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀁𑁇_  


(533.)

580\. _𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀸𑀤𑀸𑀦𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀫𑀬𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑁂 𑀘 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀫𑁆𑀳𑀸;_  
_𑀬𑀸𑀦𑀁 𑀇𑀤𑀁 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑁂𑀬𑁆𑀬𑀁,_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄”𑀢𑀺𑁇_  


(534.)

581\. _“𑀯𑀡𑁆𑀡𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀢𑀺,_  
_𑀕𑀦𑁆𑀥𑁄 𑀘 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀧𑀯𑀸𑀬𑀢𑀺;_  
_𑀬𑀓𑁆𑀔𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀦𑀕𑁆𑀕𑁄 𑀘𑀸𑀲𑀺 𑀓𑀺𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄”𑀢𑀺𑁇_  


(535.)

582\. _“𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀦𑀺𑀘𑁆𑀘𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀡𑁆𑀳𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀚𑀦𑀁 𑀉𑀧𑁂𑀫𑀺;_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄,_  
_𑀤𑀺𑀩𑁆𑀩𑁄 𑀫𑁂 𑀯𑀡𑁆𑀡𑁄 𑀲𑀢𑀢𑀁 𑀧𑀪𑀸𑀲𑀢𑀺𑁇_  


(536.)

583\. _𑀬𑀲𑀜𑁆𑀘 𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀞𑀺𑀢𑀸𑀦𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀦𑁆𑀢𑁂𑀫𑀺 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄,_  
_𑀤𑀺𑀩𑁆𑀩𑁄 𑀫𑁂 𑀕𑀦𑁆𑀥𑁄 𑀲𑀢𑀢𑀁 𑀧𑀯𑀸𑀬𑀢𑀺𑁇_  


(537.)

584\. _𑀲𑀳𑀸𑀬𑀸𑀦𑀁 𑀢𑀺𑀢𑁆𑀣𑀲𑁆𑀫𑀺𑀁 𑀦𑁆𑀳𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀣𑀮𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀤𑀳𑀺𑀲𑁆𑀲 𑀤𑀼𑀲𑁆𑀲𑀁;_  
_𑀔𑀺𑀟𑁆𑀟𑀢𑁆𑀣𑀺𑀓𑁄 𑀦𑁄 𑀘 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀦𑀕𑁆𑀕𑁄 𑀓𑀲𑀺𑀭𑀸 𑀘 𑀯𑀼𑀢𑁆𑀢𑀻”𑀢𑀺𑁇_  


(538.)

585\. _“𑀬𑁄 𑀓𑀻𑀍𑀅𑀫𑀸𑀦𑁄 𑀧𑀓𑀭𑁄𑀢𑀺 𑀧𑀸𑀧𑀁,_  
_𑀢𑀲𑁆𑀲𑁂𑀤𑀺𑀲𑀁 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀫𑀸𑀳𑀼;_  
_𑀅𑀓𑀻𑀍𑀅𑀫𑀸𑀦𑁄 𑀧𑀦 𑀬𑁄 𑀓𑀭𑁄𑀢𑀺,_  
_𑀓𑀺𑀁 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀫𑀸𑀳𑀽”𑀢𑀺𑁇_  


(539.)

586\. _“𑀬𑁂 𑀤𑀼𑀝𑁆𑀞𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀦𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸,_  
_𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀘 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀸;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀁,_  
_𑀅𑀲𑀁𑀲𑀬𑀁 𑀢𑁂 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀺𑁇_  


(540.)

587\. _𑀅𑀧𑀭𑁂 𑀧𑀦 𑀲𑀼𑀕𑀢𑀺𑀫𑀸𑀲𑀫𑀸𑀦𑀸,_  
_𑀤𑀸𑀦𑁂 𑀭𑀢𑀸 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀢𑁆𑀢𑀪𑀸𑀯𑀸;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀁,_  
_𑀅𑀲𑀁𑀲𑀬𑀁 𑀢𑁂 𑀲𑀼𑀕𑀢𑀺𑀁 𑀉𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇_  


(541.)

588\. _“𑀢𑀁 𑀓𑀺𑀦𑁆𑀢𑀺 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀫𑀳𑀁 𑀅𑀯𑁂𑀘𑁆𑀘,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄;_  
_𑀓𑀺𑀁 𑀯𑀸𑀳𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀁,_  
_𑀓𑁄 𑀯𑀸𑀧𑀺 𑀫𑀁 𑀲𑀤𑁆𑀤𑀳𑀸𑀧𑁂𑀬𑁆𑀬 𑀏𑀢𑀦𑁆”𑀢𑀺𑁇_  


(542.)

589\. _“𑀤𑀺𑀲𑁆𑀯𑀸 𑀘 𑀲𑀼𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀲𑀤𑁆𑀤𑀳𑀲𑁆𑀲𑀼,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄;_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑁂 𑀉𑀪𑀬𑁂 𑀅𑀲𑀦𑁆𑀢𑁂,_  
_𑀲𑀺𑀬𑀸 𑀦𑀼 𑀲𑀢𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀸 𑀯𑀸𑁇_  


(543.)

590\. _𑀦𑁄 𑀘𑁂𑀢𑁆𑀣 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁 𑀫𑀘𑁆𑀘𑀸,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀸𑀦𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂;_  
_𑀦𑀸𑀳𑁂𑀲𑀼𑀁 𑀲𑀢𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀸 𑀯𑀸,_  
_𑀳𑀻𑀦𑀸 𑀧𑀡𑀻𑀢𑀸 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂𑁇_  


(544.)

591\. _𑀬𑀲𑁆𑀫𑀸 𑀘 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀫𑀘𑁆𑀘𑀸,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀸𑀦𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀲𑀢𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀸 𑀤𑀼𑀕𑁆𑀕𑀢𑀸 𑀯𑀸,_  
_𑀳𑀻𑀦𑀸 𑀧𑀡𑀻𑀢𑀸 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂𑁇_  


(545.)

592\. _𑀤𑁆𑀯𑀬𑀚𑁆𑀚 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀫𑀸𑀳𑀼,_  
_𑀲𑀼𑀔𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀯𑁂𑀤𑀦𑀻𑀬𑀁;_  
_𑀢𑀸 𑀤𑁂𑀯𑀢𑀸𑀬𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀦𑁆𑀢𑀺,_  
_𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑀸 𑀤𑁆𑀯𑀬𑀢𑀁 𑀅𑀧𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(546.)

593\. _𑀦 𑀫𑀢𑁆𑀣𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀲𑀬𑀁𑀓𑀢𑀸𑀦𑀺,_  
_𑀤𑀢𑁆𑀯𑀸𑀧𑀺 𑀫𑁂 𑀦𑀢𑁆𑀣𑀺 𑀬𑁄 𑀆𑀤𑀺𑀲𑁂𑀬𑁆𑀬;_  
_𑀅𑀘𑁆𑀙𑀸𑀤𑀦𑀁 𑀲𑀬𑀦𑀫𑀣𑀦𑁆𑀦𑀧𑀸𑀦𑀁,_  
_𑀢𑁂𑀦𑀫𑁆𑀳𑀺 𑀦𑀕𑁆𑀕𑁄 𑀓𑀲𑀺𑀭𑀸 𑀘 𑀯𑀼𑀢𑁆𑀢𑀻”𑀢𑀺𑁇_  


(547.)

594\. _“𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄 𑀓𑀸𑀭𑀡𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀬𑀓𑁆𑀔,_  
_𑀅𑀘𑁆𑀙𑀸𑀤𑀦𑀁 𑀬𑁂𑀦 𑀢𑀼𑀯𑀁 𑀮𑀪𑁂𑀣;_  
_𑀆𑀘𑀺𑀓𑁆𑀔 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀬𑀤𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼,_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀺𑀓𑀁 𑀳𑁂𑀢𑀼𑀯𑀘𑁄 𑀲𑀼𑀡𑁄𑀫𑀸”𑀢𑀺𑁇_  


(548.)

595\. _“𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁄 𑀦𑀸𑀫 𑀇𑀥𑀢𑁆𑀣𑀺 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀛𑀸𑀬𑀻 𑀲𑀼𑀲𑀻𑀮𑁄 𑀅𑀭𑀳𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄;_  
_𑀕𑀼𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄 𑀲𑀁𑀯𑀼𑀢𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁄,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑁄 𑀉𑀢𑁆𑀢𑀫𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀢𑁆𑀢𑁄𑁇_  


(549.)

596\. _𑀲𑀔𑀺𑀮𑁄 𑀯𑀤𑀜𑁆𑀜𑀽 𑀲𑀼𑀯𑀘𑁄 𑀲𑀼𑀫𑀼𑀔𑁄,_  
_𑀲𑁆𑀯𑀸𑀕𑀫𑁄 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀫𑀼𑀢𑁆𑀢𑀓𑁄 𑀘;_  
_𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲 𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀭𑀡𑀯𑀺𑀳𑀸𑀭𑀻,_  
_𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀜𑁆𑀘 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄𑁇_  


(550.)

597\. _𑀲𑀦𑁆𑀢𑁄 𑀯𑀺𑀥𑀽𑀫𑁄 𑀅𑀦𑀻𑀖𑁄 𑀦𑀺𑀭𑀸𑀲𑁄,_  
_𑀫𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀲𑀮𑁆𑀮𑁄 𑀅𑀫𑀫𑁄 𑀅𑀯𑀗𑁆𑀓𑁄;_  
_𑀦𑀺𑀭𑀽𑀧𑀥𑀻 𑀲𑀩𑁆𑀩𑀧𑀧𑀜𑁆𑀘𑀔𑀻𑀡𑁄,_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀚𑀼𑀢𑀺𑀫𑀸𑁇_  


(551.)

598\. _𑀅𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸𑀧𑀺 𑀦 𑀘 𑀲𑀼𑀚𑀸𑀦𑁄,_  
_𑀫𑀼𑀦𑀻𑀢𑀺 𑀦𑀁 𑀯𑀚𑁆𑀚𑀺𑀲𑀼 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑀺;_  
_𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀢𑀁 𑀬𑀓𑁆𑀔𑀪𑀽𑀢𑀸 𑀅𑀦𑁂𑀚𑀁,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑁂𑁇_  


(552.)

599\. _𑀢𑀲𑁆𑀲 𑀢𑀼𑀯𑀁 𑀏𑀓𑀬𑀼𑀕𑀁 𑀤𑀼𑀯𑁂 𑀯𑀸,_  
_𑀫𑀫𑀼𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀘𑁂 𑀤𑀤𑁂𑀣;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀻𑀢𑀸𑀦𑀺 𑀘 𑀢𑀸𑀦𑀺 𑀅𑀲𑁆𑀲𑀼,_  
_𑀫𑀫𑀜𑁆𑀘 𑀧𑀲𑁆𑀲𑁂𑀣 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀤𑀼𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇_  


(553.)

600\. _“𑀓𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀲𑀫𑀡𑀁 𑀯𑀲𑀦𑁆𑀢𑀁,_  
_𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀧𑀲𑁆𑀲𑁂𑀫𑀼 𑀫𑀬𑀁 𑀇𑀤𑀸𑀦𑀺;_  
_𑀬𑁄 𑀫𑀚𑁆𑀚 𑀓𑀗𑁆𑀔𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀜𑁆𑀘,_  
_𑀤𑀺𑀝𑁆𑀞𑀻𑀯𑀺𑀲𑀽𑀓𑀸𑀦𑀺 𑀯𑀺𑀦𑁄𑀤𑀬𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇_  


(554.)

601\. _“𑀏𑀲𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀓𑀧𑀺𑀦𑀘𑁆𑀘𑀦𑀸𑀬𑀁,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀩𑀳𑀽𑀳𑀺;_  
_𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀪𑀸𑀲𑀢𑀺 𑀲𑀘𑁆𑀘𑀦𑀸𑀫𑁄,_  
_𑀲𑀓𑀲𑁆𑀫𑀺𑀫𑀸𑀘𑁂𑀭𑀓𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄”𑀢𑀺𑁇_  


(555.)

602\. _“𑀢𑀣𑀸𑀳𑀁 𑀓𑀲𑁆𑀲𑀸𑀫𑀺 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀇𑀤𑀸𑀦𑀺,_  
_𑀅𑀘𑁆𑀙𑀸𑀤𑀬𑀺𑀲𑁆𑀲𑀁 𑀲𑀫𑀡𑀁 𑀬𑀼𑀕𑁂𑀦;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀘 𑀢𑀸𑀦𑀺 𑀅𑀲𑁆𑀲𑀼,_  
_𑀢𑀼𑀯𑀜𑁆𑀘 𑀧𑀲𑁆𑀲𑁂𑀫𑀼 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀤𑀼𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇_  


(556.)

603\. _“𑀫𑀸 𑀅𑀓𑁆𑀔𑀡𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀁 𑀉𑀧𑀸𑀕𑀫𑀺,_  
_𑀲𑀸𑀥𑀼 𑀯𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀺 𑀦𑁂𑀲 𑀥𑀫𑁆𑀫𑁄;_  
_𑀢𑀢𑁄 𑀘 𑀓𑀸𑀮𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀧𑀲𑁆𑀲𑀸𑀳𑀺 𑀭𑀳𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀦𑁆”𑀢𑀺𑁇_  


(557.)

604\. _𑀢𑀣𑀸𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀅𑀕𑀫𑀸𑀲𑀺 𑀢𑀢𑁆𑀣,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄 𑀤𑀸𑀲𑀕𑀡𑁂𑀦 𑀮𑀺𑀘𑁆𑀙𑀯𑀺;_  
_𑀲𑁄 𑀢𑀁 𑀦𑀕𑀭𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑀸𑀲𑀽𑀧𑀕𑀘𑁆𑀙𑀺𑀢𑁆𑀣 𑀲𑀓𑁂 𑀦𑀺𑀯𑁂𑀲𑀦𑁂𑁇_  


(558.)

605\. _𑀢𑀢𑁄 𑀘 𑀓𑀸𑀮𑁂 𑀕𑀺𑀳𑀺𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀺 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦𑁆𑀳𑀢𑁆𑀯𑀸 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘 𑀔𑀡𑀁 𑀮𑀪𑀺𑀢𑁆𑀯𑀸;_  
_𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀧𑁂𑀍𑀆𑀢𑁄 𑀘 𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞,_  
_𑀕𑀸𑀳𑀸𑀧𑀬𑀻 𑀤𑀸𑀲𑀕𑀡𑁂𑀦 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑁇_  


(559.)

606\. _𑀲𑁄 𑀢𑀁 𑀧𑀤𑁂𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀢𑀁 𑀅𑀤𑁆𑀤𑀲 𑀲𑀫𑀡𑀁 𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁;_  
_𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀕𑁄𑀘𑀭𑀢𑁄 𑀦𑀺𑀯𑀢𑁆𑀢𑀁,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇_  


(560.)

607\. _𑀢𑀫𑁂𑀦𑀫𑀯𑁄𑀘 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀜𑁆𑀘 𑀧𑀼𑀘𑁆𑀙𑀺;_  
_“𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀺𑀳𑀁 𑀪𑀤𑀦𑁆𑀢𑁂,_  
_𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀫𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀺 𑀅𑀫𑁆𑀩𑀲𑀓𑁆𑀓𑀭𑁄𑁇_  


(561.)

608\. _𑀇𑀫𑀸𑀦𑀺 𑀫𑁂 𑀅𑀝𑁆𑀞 𑀬𑀼𑀕𑀸 𑀲𑀼𑀪𑀸𑀦𑀺,_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳 𑀪𑀦𑁆𑀢𑁂 𑀧𑀤𑀤𑀸𑀫𑀺 𑀢𑀼𑀬𑁆𑀳𑀁;_  
_𑀢𑁂𑀦𑁂𑀯 𑀅𑀢𑁆𑀣𑁂𑀦 𑀇𑀥𑀸𑀕𑀢𑁄𑀲𑁆𑀫𑀺,_  
_𑀬𑀣𑀸 𑀅𑀳𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇_  


(562.)

609\. _“𑀤𑀽𑀭𑀢𑁄𑀯 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀘,_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀦𑁆𑀢𑀺;_  
_𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀦𑀺𑀯𑁂𑀲𑀦𑁂,_  
_𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀬𑁄 𑀘𑀸𑀧𑀺 𑀯𑀺𑀤𑀸𑀮𑀬𑀦𑁆𑀢𑀺𑁇_  


(563.)

610\. _𑀅𑀣𑀸𑀧𑀭𑁂 𑀧𑀸𑀤𑀓𑀼𑀞𑀸𑀭𑀺𑀓𑀸𑀳𑀺,_  
_𑀅𑀯𑀁𑀲𑀺𑀭𑀸 𑀲𑀫𑀡𑀸 𑀧𑀸𑀢𑀬𑀦𑁆𑀢𑀺;_  
_𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀯𑀺𑀳𑁂𑀲𑀁,_  
_𑀢𑀬𑀸 𑀓𑀢𑀁 𑀲𑀫𑀡𑀸 𑀧𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺𑁇_  


(564.)

611\. _𑀢𑀺𑀡𑁂𑀦 𑀢𑁂𑀮𑀫𑁆𑀧𑀺 𑀦 𑀢𑁆𑀯𑀁 𑀅𑀤𑀸𑀲𑀺,_  
_𑀫𑀼𑀍𑀳𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀫𑁆𑀧𑀺 𑀦 𑀧𑀸𑀯𑀤𑀸𑀲𑀺;_  
_𑀅𑀦𑁆𑀥𑀲𑁆𑀲 𑀤𑀡𑁆𑀟𑀁 𑀲𑀬𑀫𑀸𑀤𑀺𑀬𑀸𑀲𑀺,_  
_𑀏𑀢𑀸𑀤𑀺𑀲𑁄 𑀓𑀤𑀭𑀺𑀬𑁄 𑀅𑀲𑀁𑀯𑀼𑀢𑁄 𑀢𑀼𑀯𑀁;_  
_𑀅𑀣 𑀢𑁆𑀯𑀁 𑀓𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦 𑀓𑀺𑀫𑁂𑀯 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀲𑀳 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀁 𑀓𑀭𑁄𑀲𑀻”𑀢𑀺𑁇_  


(565.)

612\. _“𑀧𑀘𑁆𑀘𑁂𑀫𑀺 𑀪𑀦𑁆𑀢𑁂 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀯𑀤𑁂𑀲𑀺,_  
_𑀯𑀺𑀳𑁂𑀲𑀬𑀺𑀁 𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀘;_  
_𑀔𑀺𑀟𑁆𑀟𑀢𑁆𑀣𑀺𑀓𑁄 𑀦𑁄 𑀘 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀫𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀫𑁂𑀯 𑀪𑀦𑁆𑀢𑁂𑁇_  


(566.)

613\. _𑀔𑀺𑀟𑁆𑀟𑀸𑀬 𑀬𑀓𑁆𑀔𑁄 𑀧𑀲𑀯𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀧𑀁,_  
_𑀯𑁂𑀤𑁂𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀲𑀫𑀢𑁆𑀢𑀪𑁄𑀕𑀻;_  
_𑀤𑀳𑀭𑁄 𑀬𑀼𑀯𑀸 𑀦𑀕𑁆𑀕𑀦𑀺𑀬𑀲𑁆𑀲 𑀪𑀸𑀕𑀻,_  
_𑀓𑀺𑀁 𑀲𑀼 𑀢𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑀭𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇_  


(567.)

614\. _𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀁𑀯𑁂𑀕𑀫𑀮𑀢𑁆𑀣𑀁 𑀪𑀦𑁆𑀢𑁂,_  
_𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀸𑀧𑀺 𑀤𑀤𑀸𑀫𑀺 𑀤𑀸𑀦𑀁;_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳 𑀪𑀦𑁆𑀢𑁂 𑀯𑀢𑁆𑀣𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞,_  
_𑀬𑀓𑁆𑀔𑀲𑁆𑀲𑀺𑀫𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬𑁄”𑀢𑀺𑁇_  


(568.)

615\. _“𑀅𑀤𑁆𑀥𑀸 𑀳𑀺 𑀤𑀸𑀦𑀁 𑀩𑀳𑀼𑀥𑀸 𑀧𑀲𑀢𑁆𑀣𑀁,_  
_𑀤𑀤𑀢𑁄 𑀘 𑀢𑁂 𑀅𑀓𑁆𑀔𑀬𑀥𑀫𑁆𑀫𑀫𑀢𑁆𑀣𑀼;_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀢𑁂 𑀯𑀢𑁆𑀣𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞,_  
_𑀬𑀓𑁆𑀔𑀲𑁆𑀲𑀺𑀫𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬𑁄”𑀢𑀺𑁇_  


(569.)

616\. _𑀢𑀢𑁄 𑀳𑀺 𑀲𑁄 𑀆𑀘𑀫𑀬𑀺𑀢𑁆𑀯𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀺,_  
_𑀣𑁂𑀭𑀲𑁆𑀲 𑀤𑀢𑁆𑀯𑀸𑀦 𑀬𑀼𑀕𑀸𑀦𑀺 𑀅𑀝𑁆𑀞;_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀘 𑀢𑀸𑀦𑀺 𑀅𑀲𑁆𑀲𑀼,_  
_𑀬𑀓𑁆𑀔𑀜𑁆𑀘 𑀧𑀲𑁆𑀲𑁂𑀣 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀤𑀼𑀲𑁆𑀲𑀁𑁇_  


(570.)

617\. _𑀢𑀫𑀤𑁆𑀤𑀲𑀸 𑀘𑀦𑁆𑀤𑀦𑀲𑀸𑀭𑀮𑀺𑀢𑁆𑀢𑀁,_  
_𑀆𑀚𑀜𑁆𑀜𑀫𑀸𑀭𑀽𑀍𑀳𑀫𑀼𑀍𑀆𑀭𑀯𑀡𑁆𑀡𑀁;_  
_𑀅𑀮𑀗𑁆𑀓𑀢𑀁 𑀲𑀸𑀥𑀼𑀦𑀺𑀯𑀢𑁆𑀣𑀤𑀼𑀲𑁆𑀲𑀁,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑀁 𑀬𑀓𑁆𑀔𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑀁𑁇_  


(571.)

618\. _𑀲𑁄 𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀉𑀤𑀕𑁆𑀕𑁄,_  
_𑀧𑀳𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄 𑀘 𑀲𑀼𑀪𑀕𑁆𑀕𑀭𑀽𑀧𑁄;_  
_𑀓𑀫𑁆𑀫𑀜𑁆𑀘 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀳𑀸𑀯𑀺𑀧𑀸𑀓𑀁,_  
_𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀁 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸𑁇_  


(572.)

619\. _𑀢𑀫𑁂𑀦𑀫𑀯𑁄𑀘 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_“𑀤𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀸𑀦𑀁 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁;_  
_𑀦 𑀘𑀸𑀧𑀺 𑀫𑁂 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀤𑁂𑀬𑁆𑀬𑀫𑀢𑁆𑀣𑀺,_  
_𑀢𑀼𑀯𑀜𑁆𑀘 𑀫𑁂 𑀬𑀓𑁆𑀔 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄”𑀢𑀺𑁇_  


(573.)

620\. _“𑀢𑀼𑀯𑀜𑁆𑀘 𑀫𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀺 𑀏𑀓𑀤𑁂𑀲𑀁,_  
_𑀅𑀤𑀸𑀲𑀺 𑀤𑀸𑀦𑀸𑀦𑀺 𑀅𑀫𑁄𑀖𑀫𑁂𑀢𑀁;_  
_𑀲𑁆𑀯𑀸𑀳𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀬𑀸𑀯 𑀲𑀓𑁆𑀔𑀺𑀁,_  
_𑀅𑀫𑀸𑀦𑀼𑀲𑁄 𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀦𑁆”𑀢𑀺𑁇_  


(574.)

621\. _“𑀕𑀢𑀻 𑀘 𑀩𑀦𑁆𑀥𑀽 𑀘 𑀧𑀭𑀸𑀬𑀡𑀜𑁆𑀘,_  
_𑀫𑀺𑀢𑁆𑀢𑁄 𑀫𑀫𑀸𑀲𑀺 𑀅𑀣 𑀤𑁂𑀯𑀢𑀸 𑀫𑁂;_  
_𑀬𑀸𑀘𑀸𑀫𑀺 𑀢𑀁 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀪𑀯𑀺𑀢𑁆𑀯𑀸,_  
_𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀬𑀓𑁆𑀔 𑀧𑀼𑀦𑀸𑀧𑀺 𑀤𑀝𑁆𑀞𑀼𑀦𑁆”𑀢𑀺𑁇_  


(575.)

622\. _“𑀲𑀘𑁂 𑀢𑀼𑀯𑀁 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺,_  
_𑀓𑀤𑀭𑀺𑀬𑀭𑀽𑀧𑁄 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀢𑁆𑀯𑀁 𑀦𑁂𑀯 𑀫𑀁 𑀮𑀘𑁆𑀙𑀲𑀺 𑀤𑀲𑁆𑀲𑀦𑀸𑀬,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀘 𑀢𑀁 𑀦𑁄𑀧𑀺 𑀘 𑀆𑀮𑀧𑀺𑀲𑁆𑀲𑀁𑁇_  


(576.)

623\. _𑀲𑀘𑁂 𑀧𑀦 𑀢𑁆𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺 𑀥𑀫𑁆𑀫𑀕𑀸𑀭𑀯𑁄,_  
_𑀤𑀸𑀦𑁂 𑀭𑀢𑁄 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀢𑁆𑀢𑀪𑀸𑀯𑁄;_  
_𑀑𑀧𑀸𑀦𑀪𑀽𑀢𑁄 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁,_  
_𑀏𑀯𑀁 𑀫𑀫𑀁 𑀮𑀘𑁆𑀙𑀲𑀺 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇_  


(577.)

624\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀘 𑀢𑀁 𑀆𑀮𑀧𑀺𑀲𑁆𑀲𑀁 𑀪𑀤𑀦𑁆𑀢𑁂,_  
_𑀇𑀫𑀜𑁆𑀘 𑀲𑀽𑀮𑀢𑁄 𑀮𑀳𑀼𑀁 𑀧𑀫𑀼𑀜𑁆𑀘;_  
_𑀬𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀅𑀓𑀭𑀺𑀫𑁆𑀳 𑀲𑀓𑁆𑀔𑀺𑀁,_  
_𑀫𑀜𑁆𑀜𑀸𑀫𑀺 𑀲𑀽𑀮𑀸𑀯𑀼𑀢𑀓𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸𑁇_  


(578.)

625\. _𑀢𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀅𑀓𑀭𑀺𑀫𑁆𑀳 𑀲𑀓𑁆𑀔𑀺𑀁,_  
_𑀅𑀬𑀜𑁆𑀘 𑀲𑀽𑀮𑀢𑁄 𑀮𑀳𑀼𑀁 𑀧𑀫𑀼𑀢𑁆𑀢𑁄;_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀺 𑀲𑀫𑀸𑀘𑀭𑀦𑁆𑀢𑁄,_  
_𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬 𑀲𑁄 𑀦𑀺𑀭𑀬𑀸 𑀘 𑀢𑀫𑁆𑀳𑀸;_  
_𑀓𑀫𑁆𑀫𑀁 𑀲𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑁂𑀤𑀦𑀻𑀬𑀁𑁇_  


(579.)

626\. _𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑀜𑁆𑀘 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀢𑁂𑀦𑁂𑀯 𑀲𑀳 𑀲𑀁𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂;_  
_𑀲𑀬𑀁 𑀫𑀼𑀔𑁂𑀦𑀽𑀧𑀦𑀺𑀲𑀚𑁆𑀚 𑀧𑀼𑀘𑁆𑀙,_  
_𑀲𑁄 𑀢𑁂 𑀅𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁𑁇_  


(580.)

627\. _𑀢𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀼𑀘𑁆𑀙𑀲𑁆𑀲𑀼 𑀅𑀜𑁆𑀜𑀢𑁆𑀣𑀺𑀓𑁄 𑀦𑁄 𑀘 𑀧𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀲𑁄 𑀢𑁂 𑀲𑀼𑀢𑀁 𑀅𑀲𑀼𑀢𑀜𑁆𑀘𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀅𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀺 𑀬𑀣𑀸 𑀧𑀚𑀸𑀦𑀦𑁆”𑀢𑀺𑁇_  


(581.)

628\. _𑀲𑁄 𑀢𑀢𑁆𑀣 𑀭𑀳𑀲𑁆𑀲𑀁 𑀲𑀫𑀼𑀮𑁆𑀮𑀧𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀓𑁆𑀔𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑀦 𑀅𑀫𑀸𑀦𑀼𑀲𑁂𑀦;_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀲𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀦𑀁 𑀲𑀓𑀸𑀲𑀁,_  
_𑀅𑀣 𑀩𑁆𑀭𑀯𑀺 𑀧𑀭𑀺𑀲𑀁 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇_  


(582.)

629\. _“𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀫𑀫 𑀏𑀓𑀯𑀸𑀓𑁆𑀬𑀁,_  
_𑀯𑀭𑀁 𑀯𑀭𑀺𑀲𑁆𑀲𑀁 𑀮𑀪𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀢𑁆𑀣𑀁;_  
_𑀲𑀽𑀮𑀸𑀯𑀼𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀮𑀼𑀤𑁆𑀤𑀓𑀫𑁆𑀫𑁄,_  
_𑀧𑀡𑀺𑀳𑀺𑀢𑀤𑀡𑁆𑀟𑁄 𑀅𑀦𑀼𑀲𑀢𑁆𑀢𑀭𑀽𑀧𑁄𑁇_  


(583.)

630\. _𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀯𑀻𑀲𑀢𑀺𑀭𑀢𑁆𑀢𑀺𑀫𑀢𑁆𑀢𑀸,_  
_𑀬𑀢𑁄 𑀆𑀯𑀼𑀢𑁄 𑀦𑁂𑀯 𑀚𑀻𑀯𑀢𑀺 𑀦 𑀫𑀢𑁄;_  
_𑀢𑀸𑀳𑀁 𑀫𑁄𑀘𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀸𑀦𑀺,_  
_𑀬𑀣𑀸𑀫𑀢𑀺𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀼 𑀲𑀁𑀖𑁄”𑀢𑀺𑁇_  


(584.)

631\. _“𑀏𑀢𑀜𑁆𑀘 𑀅𑀜𑁆𑀜𑀜𑁆𑀘 𑀮𑀳𑀼𑀁 𑀧𑀫𑀼𑀜𑁆𑀘,_  
_𑀓𑁄 𑀢𑀁 𑀯𑀤𑁂𑀣 𑀢𑀣𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀁;_  
_𑀬𑀣𑀸 𑀧𑀚𑀸𑀦𑀸𑀲𑀺 𑀢𑀣𑀸 𑀓𑀭𑁄𑀳𑀺,_  
_𑀬𑀣𑀸𑀫𑀢𑀺𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑀁𑀖𑁄”𑀢𑀺𑁇_  


(585.)

632\. _𑀲𑁄 𑀢𑀁 𑀧𑀤𑁂𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀽𑀮𑀸𑀯𑀼𑀢𑀁 𑀫𑁄𑀘𑀬𑀺 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯;_  
_“𑀫𑀸 𑀪𑀸𑀬𑀺 𑀲𑀫𑁆𑀫𑀸”𑀢𑀺 𑀘 𑀢𑀁 𑀅𑀯𑁄𑀘,_  
_𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀸𑀦𑀜𑁆𑀘 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑀺𑁇_  


(586.)

633\. _“𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑀜𑁆𑀘 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀢𑁂𑀦𑁂𑀯 𑀲𑀳 𑀲𑀁𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂;_  
_𑀲𑀬𑀁 𑀫𑀼𑀔𑁂𑀦𑀽𑀧𑀦𑀺𑀲𑀚𑁆𑀚 𑀮𑀺𑀘𑁆𑀙𑀯𑀺,_  
_𑀢𑀣𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀣 𑀦𑀁 𑀓𑀸𑀭𑀡𑀢𑁆𑀣𑀺𑀓𑁄𑁇_  


(587.)

634\. _𑀲𑀽𑀮𑀸𑀯𑀼𑀢𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀮𑀼𑀤𑁆𑀤𑀓𑀫𑁆𑀫𑁄,_  
_𑀧𑀡𑀺𑀳𑀺𑀢𑀤𑀡𑁆𑀟𑁄 𑀅𑀦𑀼𑀲𑀢𑁆𑀢𑀭𑀽𑀧𑁄;_  
_𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀯𑀻𑀲𑀢𑀺𑀭𑀢𑁆𑀢𑀺𑀫𑀢𑁆𑀢𑀸,_  
_𑀬𑀢𑁄 𑀆𑀯𑀼𑀢𑁄 𑀦𑁂𑀯 𑀚𑀻𑀯𑀢𑀺 𑀦 𑀫𑀢𑁄𑁇_  


(588.)

635\. _𑀲𑁄 𑀫𑁄𑀘𑀺𑀢𑁄 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀬𑀸 𑀇𑀤𑀸𑀦𑀺,_  
_𑀏𑀢𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀯𑀘𑁄 𑀳𑀺 𑀪𑀦𑁆𑀢𑁂;_  
_𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄 𑀓𑀸𑀭𑀡𑀁 𑀓𑀺𑀜𑁆𑀘𑀺𑀤𑁂𑀯,_  
_𑀬𑁂𑀦 𑀲𑁄 𑀦𑀺𑀭𑀬𑀁 𑀦𑁄 𑀯𑀚𑁂𑀬𑁆𑀬𑁇_  


(589.)

636\. _𑀆𑀘𑀺𑀓𑁆𑀔 𑀪𑀦𑁆𑀢𑁂 𑀬𑀤𑀺 𑀅𑀢𑁆𑀣𑀺 𑀳𑁂𑀢𑀼,_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀺𑀓𑀁 𑀳𑁂𑀢𑀼𑀯𑀘𑁄 𑀲𑀼𑀡𑁄𑀫;_  
_𑀦 𑀢𑁂𑀲𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀸𑀲𑀫𑀢𑁆𑀣𑀺,_  
_𑀅𑀯𑁂𑀤𑀬𑀺𑀢𑁆𑀯𑀸 𑀇𑀥 𑀩𑁆𑀬𑀦𑁆𑀢𑀺𑀪𑀸𑀯𑁄”𑀢𑀺𑁇_  


(590.)

637\. _“𑀲𑀘𑁂 𑀲 𑀥𑀫𑁆𑀫𑀸𑀦𑀺 𑀲𑀫𑀸𑀘𑀭𑁂𑀬𑁆𑀬,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄;_  
_𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬 𑀲𑁄 𑀦𑀺𑀭𑀬𑀸 𑀘 𑀢𑀫𑁆𑀳𑀸,_  
_𑀓𑀫𑁆𑀫𑀁 𑀲𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑁂𑀤𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇_  


(591.)

638\. _“𑀅𑀜𑁆𑀜𑀸𑀢𑁄 𑀏𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄,_  
_𑀫𑀫𑀫𑁆𑀧𑀺 𑀤𑀸𑀦𑀺 𑀅𑀦𑀼𑀓𑀫𑁆𑀧 𑀪𑀦𑁆𑀢𑁂;_  
_𑀅𑀦𑀼𑀲𑀸𑀲 𑀫𑀁 𑀑𑀯𑀤 𑀪𑀽𑀭𑀺𑀧𑀜𑁆𑀜,_  
_𑀬𑀣𑀸 𑀅𑀳𑀁 𑀦𑁄 𑀦𑀺𑀭𑀬𑀁 𑀯𑀚𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇_  


(592.)

639\. _“𑀅𑀚𑁆𑀚𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀉𑀧𑁂𑀳𑀺,_  
_𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑀁𑀖𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀢𑀣𑁂𑀯 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀧𑀤𑀸𑀦𑀺 𑀧𑀜𑁆𑀘,_  
_𑀅𑀔𑀡𑁆𑀟𑀨𑀼𑀮𑁆𑀮𑀸𑀦𑀺 𑀲𑀫𑀸𑀤𑀺𑀬𑀲𑁆𑀲𑀼𑁇_  


(593.)

640\. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀫𑀲𑁆𑀲𑀼 𑀔𑀺𑀧𑁆𑀧𑀁,_  
_𑀮𑁄𑀓𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀲𑁆𑀲𑀼;_  
_𑀅𑀫𑀚𑁆𑀚𑀧𑁄 𑀫𑀸 𑀘 𑀫𑀼𑀲𑀸 𑀅𑀪𑀸𑀡𑀻,_  
_𑀲𑀓𑁂𑀦 𑀤𑀸𑀭𑁂𑀦 𑀘 𑀳𑁄𑀳𑀺 𑀢𑀼𑀝𑁆𑀞𑁄;_  
_𑀇𑀫𑀜𑁆𑀘 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀯𑀭𑁂𑀦𑀼𑀧𑁂𑀢𑀁,_  
_𑀲𑀫𑀸𑀤𑀺𑀬𑀸𑀳𑀺 𑀓𑀼𑀲𑀮𑀁 𑀲𑀼𑀔𑀼𑀤𑁆𑀭𑀬𑀁𑁇_  


(594.)

641\. _𑀘𑀻𑀯𑀭𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀧𑀘𑁆𑀘𑀬𑀁 𑀲𑀬𑀦𑀸𑀲𑀦𑀁;_  
_𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁,_  
_𑀯𑀢𑁆𑀣𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀘;_  
_𑀤𑀤𑀸𑀳𑀺 𑀉𑀚𑀼𑀪𑀽𑀢𑁂𑀲𑀼,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(595.)

642\. _𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑁂 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂;_  
_𑀢𑀧𑁆𑀧𑁂𑀳𑀺 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦,_  
_𑀲𑀤𑀸 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀯𑀟𑁆𑀠𑀢𑀺𑁇_  


(596.)

643\. _𑀏𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀺 𑀲𑀫𑀸𑀘𑀭𑀦𑁆𑀢𑁄,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄;_  
_𑀫𑀼𑀜𑁆𑀘 𑀢𑀼𑀯𑀁 𑀦𑀺𑀭𑀬𑀸 𑀘 𑀢𑀫𑁆𑀳𑀸,_  
_𑀓𑀫𑁆𑀫𑀁 𑀲𑀺𑀬𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑁂𑀤𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇_  


(597.)

644\. _“𑀅𑀚𑁆𑀚𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀉𑀧𑁂𑀫𑀺,_  
_𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑀁𑀖𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀢𑀣𑁂𑀯 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀧𑀤𑀸𑀦𑀺 𑀧𑀜𑁆𑀘,_  
_𑀅𑀔𑀡𑁆𑀟𑀨𑀼𑀮𑁆𑀮𑀸𑀦𑀺 𑀲𑀫𑀸𑀤𑀺𑀬𑀸𑀫𑀺𑁇_  


(598.)

645\. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑀺𑀭𑀫𑀸𑀫𑀺 𑀔𑀺𑀧𑁆𑀧𑀁,_  
_𑀮𑁄𑀓𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑀸𑀫𑀺;_  
_𑀅𑀫𑀚𑁆𑀚𑀧𑁄 𑀦𑁄 𑀘 𑀫𑀼𑀲𑀸 𑀪𑀡𑀸𑀫𑀺,_  
_𑀲𑀓𑁂𑀦 𑀤𑀸𑀭𑁂𑀦 𑀘 𑀳𑁄𑀫𑀺 𑀢𑀼𑀝𑁆𑀞𑁄;_  
_𑀇𑀫𑀜𑁆𑀘 𑀅𑀭𑀺𑀬𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀯𑀭𑁂𑀦𑀼𑀧𑁂𑀢𑀁,_  
_𑀲𑀫𑀸𑀤𑀺𑀬𑀸𑀫𑀺 𑀓𑀼𑀲𑀮𑀁 𑀲𑀼𑀔𑀼𑀤𑁆𑀭𑀬𑀁𑁇_  


(599.)

646\. _𑀘𑀻𑀯𑀭𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀧𑀘𑁆𑀘𑀬𑀁 𑀲𑀬𑀦𑀸𑀲𑀦𑀁;_  
_𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁,_  
_𑀯𑀢𑁆𑀣𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀘𑁇_  


(600.)

647\. _𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑁂 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂;_  
_𑀤𑀤𑀸𑀫𑀺 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑁄”𑀢𑀺𑁇_  


(601.)

648\. _𑀏𑀢𑀸𑀤𑀺𑀲𑁄 𑀮𑀺𑀘𑁆𑀙𑀯𑀺 𑀅𑀫𑁆𑀩𑀲𑀓𑁆𑀓𑀭𑁄,_  
_𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁄;_  
_𑀲𑀤𑁆𑀥𑁄 𑀫𑀼𑀤𑀽 𑀓𑀸𑀭𑀓𑀭𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀁𑀖𑀜𑁆𑀘 𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀢𑀤𑀸 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑁇_  


(602.)

649\. _𑀲𑀽𑀮𑀸𑀯𑀼𑀢𑁄 𑀘 𑀅𑀭𑁄𑀕𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑁂𑀭𑀻 𑀲𑀼𑀔𑀻 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀉𑀧𑀸𑀕𑀫𑀺;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀜𑁆𑀘 𑀆𑀕𑀫𑁆𑀫 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑀼𑀢𑁆𑀢𑀫𑀁,_  
_𑀉𑀪𑁄𑀧𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺 𑀅𑀚𑁆𑀛𑀕𑀼𑀁𑁇_  


(603.)

650\. _𑀏𑀢𑀸𑀤𑀺𑀲𑀸 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦 𑀲𑁂𑀯𑀦𑀸,_  
_𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀲𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀢𑀁;_  
_𑀲𑀽𑀮𑀸𑀯𑀼𑀢𑁄 𑀅𑀕𑁆𑀕𑀨𑀮𑀁 𑀅𑀨𑀲𑁆𑀲𑀬𑀺,_  
_𑀨𑀮𑀁 𑀓𑀦𑀺𑀝𑁆𑀞𑀁 𑀧𑀦 𑀅𑀫𑁆𑀩𑀲𑀓𑁆𑀓𑀭𑁄𑀢𑀺𑁇_  


---

651\. 𑀅𑀫𑁆𑀩𑀲𑀓𑁆𑀓𑀭𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼 𑀧𑀞𑀫𑀁𑁇
