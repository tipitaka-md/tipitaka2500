### 𑀅𑀓𑁆𑀔𑀭𑀼𑀓𑁆𑀔𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

(800.)

860\. _“𑀬𑀁 𑀤𑀤𑀸𑀢𑀺 𑀦 𑀢𑀁 𑀳𑁄𑀢𑀺,_  
_𑀤𑁂𑀣𑁂𑀯 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀉𑀪𑀬𑀁 𑀢𑀭𑀢𑀺;_  
_𑀉𑀪𑀬𑀁 𑀢𑁂𑀦 𑀤𑀸𑀦𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺,_  
_𑀚𑀸𑀕𑀭𑀣 𑀫𑀸𑀧𑀫𑀚𑁆𑀚𑀣𑀸”𑀢𑀺𑁇_
