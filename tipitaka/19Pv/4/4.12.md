# 𑁪𑁇𑁧𑁨 𑀅𑀫𑁆𑀩𑀯𑀦𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀫𑁆𑀩𑀯𑀦𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

(796.)

855. _“𑀅𑀬𑀜𑁆𑀘 𑀢𑁂 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀲𑀼𑀭𑀫𑁆𑀫𑀸,_  
_𑀲𑀫𑀸 𑀲𑀼𑀢𑀺𑀢𑁆𑀣𑀸 𑀘 𑀫𑀳𑁄𑀤𑀓𑀸 𑀘;_  
_𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀪𑀫𑀭𑀕𑀡𑀸𑀦𑀼𑀓𑀺𑀡𑁆𑀡𑀸,_  
_𑀓𑀣𑀁 𑀢𑀬𑀸 𑀮𑀤𑁆𑀥𑀸 𑀅𑀬𑀁 𑀫𑀦𑀼𑀜𑁆𑀜𑀸𑁇_  


(797.)

856. _𑀇𑀤𑀜𑁆𑀘 𑀢𑁂 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀲𑀼𑀭𑀫𑁆𑀫𑀁,_  
_𑀲𑀩𑁆𑀩𑁄𑀢𑀼𑀓𑀁 𑀥𑀸𑀭𑀬𑀢𑁂 𑀨𑀮𑀸𑀦𑀺;_  
_𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀪𑀫𑀭𑀕𑀡𑀸𑀦𑀼𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀓𑀣𑀁 𑀢𑀬𑀸 𑀮𑀤𑁆𑀥𑀫𑀺𑀤𑀁 𑀯𑀺𑀫𑀸𑀦𑀁”𑁇_  


(798.)

857. _“𑀅𑀫𑁆𑀩𑀧𑀓𑁆𑀓𑀁 𑀤𑀓𑀁 𑀬𑀸𑀕𑀼,_  
_𑀲𑀻𑀢𑀘𑁆𑀙𑀸𑀬𑀸 𑀫𑀦𑁄𑀭𑀫𑀸;_  
_𑀥𑀻𑀢𑀸𑀬 𑀤𑀺𑀦𑁆𑀦𑀤𑀸𑀦𑁂𑀦,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀇𑀥 𑀮𑀩𑁆𑀪𑀢𑀺”𑁇_  


(799.)

858. _“𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀁 𑀓𑀫𑁆𑀫𑀁 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀣,_  
_𑀤𑀸𑀦𑀲𑁆𑀲 𑀤𑀫𑀲𑁆𑀲 𑀲𑀁𑀬𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀁;_  
_𑀤𑀸𑀲𑀻 𑀅𑀳𑀁 𑀅𑀬𑁆𑀬𑀓𑀼𑀮𑁂𑀲𑀼 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀡𑀺𑀲𑀸 𑀳𑁄𑀫𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑀸”𑀢𑀺𑁇_  


---

859. 𑀅𑀫𑁆𑀩𑀯𑀦𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀁𑁇
