# 𑁩𑁇𑁮 𑀤𑀼𑀢𑀺𑀬𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑀓𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑀓𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼

(488.)

529\. _“𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑁂 𑀘 𑀧𑀸𑀲𑀸𑀤𑁂,_  
_𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂 𑀕𑁄𑀦𑀓𑀢𑁆𑀣𑀢𑁂;_  
_𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀢𑀼𑀭𑀺𑀬𑁂𑀦,_  
_𑀭𑀫𑀲𑀺 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀤𑀺𑀢𑁂𑁇_  


(489.)

530\. _𑀢𑀢𑁄 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀲𑀽𑀭𑀺𑀬𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀧𑀢𑀺;_  
_𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑁄 𑀲𑀼𑀲𑀸𑀦𑀲𑁆𑀫𑀺𑀁,_  
_𑀩𑀳𑀼𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺𑁇_  


(490.)

531\. _𑀓𑀺𑀁 𑀦𑀼 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬,_  
_𑀫𑀦𑀲𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀢𑀁;_  
_𑀓𑀺𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀲𑀺”𑁇_  


(491.)

532\. _“𑀅𑀳𑀁 𑀭𑀸𑀚𑀕𑀳𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀭𑀫𑀡𑀻𑀬𑁂 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑁂;_  
_𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀮𑀼𑀤𑁆𑀤𑁄 𑀘𑀸𑀲𑀺𑀫𑀲𑀜𑁆𑀜𑀢𑁄𑁇_  


(492.)

533\. _𑀢𑀲𑁆𑀲 𑀫𑁂 𑀲𑀳𑀸𑀬𑁄 𑀲𑀼𑀳𑀤𑀬𑁄,_  
_𑀲𑀤𑁆𑀥𑁄 𑀆𑀲𑀺 𑀉𑀧𑀸𑀲𑀓𑁄;_  
_𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀼𑀧𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀆𑀲𑀺 𑀕𑁄𑀢𑀫𑀲𑀸𑀯𑀓𑁄;_  
_𑀲𑁄𑀧𑀺 𑀫𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀯𑀸𑀭𑁂𑀲𑀺 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁𑁇_  


(493.)

534\. _‘𑀫𑀸𑀓𑀸𑀲𑀺 𑀧𑀸𑀧𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀫𑀸 𑀢𑀸𑀢 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀅𑀕𑀸;_  
_𑀲𑀘𑁂 𑀇𑀘𑁆𑀙𑀲𑀺 𑀧𑁂𑀘𑁆𑀘 𑀲𑀼𑀔𑀁,_  
_𑀯𑀺𑀭𑀫 𑀧𑀸𑀡𑀯𑀥𑀸 𑀅𑀲𑀁𑀬𑀫𑀸’𑁇_  


(494.)

535\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀔𑀓𑀸𑀫𑀲𑁆𑀲 𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄;_  
_𑀦𑀸𑀓𑀸𑀲𑀺𑀁 𑀲𑀓𑀮𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁,_  
_𑀘𑀺𑀭𑀧𑀸𑀧𑀸𑀪𑀺𑀭𑀢𑁄 𑀅𑀩𑀼𑀤𑁆𑀥𑀺𑀫𑀸𑁇_  


(495.)

536\. _𑀲𑁄 𑀫𑀁 𑀧𑀼𑀦 𑀪𑀽𑀭𑀺𑀲𑀼𑀫𑁂𑀥𑀲𑁄,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀲𑀁𑀬𑀫𑁂 𑀦𑀺𑀯𑁂𑀲𑀬𑀺;_  
_‘𑀲𑀘𑁂 𑀤𑀺𑀯𑀸 𑀳𑀦𑀲𑀺 𑀧𑀸𑀡𑀺𑀦𑁄,_  
_𑀅𑀣 𑀢𑁂 𑀭𑀢𑁆𑀢𑀺𑀁 𑀪𑀯𑀢𑀼 𑀲𑀁𑀬𑀫𑁄’𑁇_  


(496.)

537\. _𑀲𑁆𑀯𑀸𑀳𑀁 𑀤𑀺𑀯𑀸 𑀳𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀡𑀺𑀦𑁄,_  
_𑀯𑀺𑀭𑀢𑁄 𑀭𑀢𑁆𑀢𑀺𑀫𑀳𑁄𑀲𑀺 𑀲𑀜𑁆𑀜𑀢𑁄;_  
_𑀭𑀢𑁆𑀢𑀸𑀳𑀁 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀫𑀺,_  
_𑀤𑀺𑀯𑀸 𑀔𑀚𑁆𑀚𑀸𑀫𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑁄𑁇_  


(497.)

538\. _𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀲𑁆𑀲,_  
_𑀅𑀦𑀼𑀪𑁄𑀫𑀺 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀫𑀸𑀦𑀼𑀲𑀺𑀁;_  
_𑀤𑀺𑀯𑀸 𑀧𑀝𑀺𑀳𑀢𑀸𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀸,_  
_𑀉𑀧𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀲𑀫𑀦𑁆𑀢𑀸 𑀔𑀸𑀤𑀺𑀢𑀼𑀁𑁇_  


(498.)

539\. _𑀬𑁂 𑀘 𑀢𑁂 𑀲𑀢𑀢𑀸𑀦𑀼𑀬𑁄𑀕𑀺𑀦𑁄,_  
_𑀥𑀼𑀯𑀁 𑀧𑀬𑀼𑀢𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂;_  
_𑀫𑀜𑁆𑀜𑀸𑀫𑀺 𑀢𑁂 𑀅𑀫𑀢𑀫𑁂𑀯 𑀓𑁂𑀯𑀮𑀁,_  
_𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀤𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀦𑁆”𑀢𑀺𑁇_  


---

540\. 𑀤𑀼𑀢𑀺𑀬𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑀓𑀧𑁂𑀢𑀯𑀢𑁆𑀣𑀼 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
