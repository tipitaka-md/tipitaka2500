# 𑁨𑁦𑁇𑁮 𑀧𑀜𑁆𑀘𑀫𑀳𑀮𑁆𑀮𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀕𑁆𑀕𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀫𑀳𑀮𑁆𑀮𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(283.)

305\. _𑀯𑀦𑀁 𑀙𑀺𑀦𑁆𑀤𑀣 𑀫𑀸 𑀭𑀼𑀓𑁆𑀔𑀁,_  
_𑀯𑀦𑀢𑁄 𑀚𑀸𑀬𑀢𑁂 𑀪𑀬𑀁;_  
_𑀙𑁂𑀢𑁆𑀯𑀸 𑀯𑀦𑀜𑁆𑀘 𑀯𑀦𑀣𑀜𑁆𑀘,_  
_𑀦𑀺𑀩𑁆𑀩𑀦𑀸 𑀳𑁄𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄𑁇_  


(284.)

306\. _𑀬𑀸𑀯 𑀳𑀺 𑀯𑀦𑀣𑁄 𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺,_  
_𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁄𑀧𑀺 𑀦𑀭𑀲𑁆𑀲 𑀦𑀸𑀭𑀺𑀲𑀼;_  
_𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀫𑀦𑁄𑀯 𑀢𑀸𑀯 𑀲𑁄,_  
_𑀯𑀘𑁆𑀙𑁄 𑀔𑀻𑀭𑀧𑀓𑁄𑀯 𑀫𑀸𑀢𑀭𑀺𑁇_
