# 𑁨𑁦𑁇𑁯 𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀸𑀭𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(285.)

307\. _𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀤 𑀲𑀺𑀦𑁂𑀳𑀫𑀢𑁆𑀢𑀦𑁄,_  
_𑀓𑀼𑀫𑀼𑀤𑀁 𑀲𑀸𑀭𑀤𑀺𑀓𑀁𑀯 𑀧𑀸𑀡𑀺𑀦𑀸;_  
_𑀲𑀦𑁆𑀢𑀺𑀫𑀕𑁆𑀕𑀫𑁂𑀯 𑀩𑁆𑀭𑀽𑀳𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀼𑀕𑀢𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁𑁇_
