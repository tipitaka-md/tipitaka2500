# 𑁨𑁦𑁇𑁭 𑀧𑁄𑀝𑁆𑀞𑀺𑀮𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀧𑁄𑀝𑁆𑀞𑀺𑀮𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(282.)

304\. _𑀬𑁄𑀕𑀸 𑀯𑁂 𑀚𑀸𑀬𑀢𑀻 𑀪𑀽𑀭𑀺,_  
_𑀅𑀬𑁄𑀕𑀸 𑀪𑀽𑀭𑀺𑀲𑀗𑁆𑀔𑀬𑁄;_  
_𑀏𑀢𑀁 𑀤𑁆𑀯𑁂𑀥𑀸𑀧𑀣𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀪𑀯𑀸𑀬 𑀯𑀺𑀪𑀯𑀸𑀬 𑀘;_  
_𑀢𑀣𑀸𑀢𑁆𑀢𑀸𑀦𑀁 𑀦𑀺𑀯𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀬𑀣𑀸 𑀪𑀽𑀭𑀺 𑀧𑀯𑀟𑁆𑀠𑀢𑀺𑁇_
