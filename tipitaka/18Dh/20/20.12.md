# 𑁨𑁦𑁇𑁧𑁨 𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑀯𑀢𑁆𑀣𑀼

(288.)

310. _𑀦 𑀲𑀦𑁆𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑀸 𑀢𑀸𑀡𑀸𑀬,_  
_𑀦 𑀧𑀺𑀢𑀸 𑀦𑀸𑀧𑀺 𑀩𑀦𑁆𑀥𑀯𑀸;_  
_𑀅𑀦𑁆𑀢𑀓𑁂𑀦𑀸𑀥𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀦𑀢𑁆𑀣𑀺 𑀜𑀸𑀢𑀻𑀲𑀼 𑀢𑀸𑀡𑀢𑀸𑁇_  


(289.)

311. _𑀏𑀢𑀫𑀢𑁆𑀣𑀯𑀲𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀲𑀻𑀮𑀲𑀁𑀯𑀼𑀢𑁄;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀫𑀦𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀯𑀺𑀲𑁄𑀥𑀬𑁂𑁇_  


---
