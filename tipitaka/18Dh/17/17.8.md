# 𑁧𑁭𑁇𑁮 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀯𑀢𑁆𑀣𑀼

### 𑀓𑁄𑀥𑀯𑀕𑁆𑀕

### 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀯𑀢𑁆𑀣𑀼

(231.)

250. _𑀓𑀸𑀬𑀧𑁆𑀧𑀓𑁄𑀧𑀁 𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬,_  
_𑀓𑀸𑀬𑁂𑀦 𑀲𑀁𑀯𑀼𑀢𑁄 𑀲𑀺𑀬𑀸;_  
_𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀸𑀬𑁂𑀦 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑁂𑁇_  


(232.)

251. _𑀯𑀘𑀻𑀧𑀓𑁄𑀧𑀁 𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬,_  
_𑀯𑀸𑀘𑀸𑀬 𑀲𑀁𑀯𑀼𑀢𑁄 𑀲𑀺𑀬𑀸;_  
_𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑀸𑀘𑀸𑀬 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑁂𑁇_  


(233.)

252. _𑀫𑀦𑁄𑀧𑀓𑁄𑀧𑀁 𑀭𑀓𑁆𑀔𑁂𑀬𑁆𑀬,_  
_𑀫𑀦𑀲𑀸 𑀲𑀁𑀯𑀼𑀢𑁄 𑀲𑀺𑀬𑀸;_  
_𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀦𑀲𑀸 𑀲𑀼𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑁂𑁇_  


(234.)

253. _𑀓𑀸𑀬𑁂𑀦 𑀲𑀁𑀯𑀼𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀅𑀣𑁄 𑀯𑀸𑀘𑀸𑀬 𑀲𑀁𑀯𑀼𑀢𑀸;_  
_𑀫𑀦𑀲𑀸 𑀲𑀁𑀯𑀼𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀢𑁂 𑀯𑁂 𑀲𑀼𑀧𑀭𑀺𑀲𑀁𑀯𑀼𑀢𑀸𑁇_  


---

254. 𑀓𑁄𑀥𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀭𑀲𑀫𑁄𑁇
