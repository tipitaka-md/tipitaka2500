# 𑁧𑁦𑁇𑁭 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀤𑀡𑁆𑀟𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(137.)

147\. _𑀬𑁄 𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀤𑀡𑁆𑀟𑁂𑀲𑀼,_  
_𑀅𑀧𑁆𑀧𑀤𑀼𑀝𑁆𑀞𑁂𑀲𑀼 𑀤𑀼𑀲𑁆𑀲𑀢𑀺;_  
_𑀤𑀲𑀦𑁆𑀦𑀫𑀜𑁆𑀜𑀢𑀭𑀁 𑀞𑀸𑀦𑀁,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(138.)

148\. _𑀯𑁂𑀤𑀦𑀁 𑀨𑀭𑀼𑀲𑀁 𑀚𑀸𑀦𑀺𑀁,_  
_𑀲𑀭𑀻𑀭𑀲𑁆𑀲 𑀯 𑀪𑁂𑀤𑀦𑀁;_  
_𑀕𑀭𑀼𑀓𑀁 𑀯𑀸𑀧𑀺 𑀆𑀩𑀸𑀥𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑁂𑀧𑀁 𑀯 𑀧𑀸𑀧𑀼𑀡𑁂𑁇_  


(139.)

149\. _𑀭𑀸𑀚𑀢𑁄 𑀯𑀸 𑀉𑀧𑀲𑀕𑁆𑀕𑀁,_  
_𑀅𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀁 𑀯 𑀤𑀸𑀭𑀼𑀡𑀁;_  
_𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀯 𑀜𑀸𑀢𑀻𑀦𑀁,_  
_𑀪𑁄𑀕𑀸𑀦𑀁 𑀯 𑀧𑀪𑀗𑁆𑀕𑀼𑀭𑀁𑁇_  


(140.)

150\. _𑀅𑀣 𑀯𑀸𑀲𑁆𑀲 𑀅𑀕𑀸𑀭𑀸𑀦𑀺,_  
_𑀅𑀕𑁆𑀕𑀺 𑀟𑀳𑀢𑀺 𑀧𑀸𑀯𑀓𑁄;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀦𑀺𑀭𑀬𑀁 𑀲𑁄𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇_
