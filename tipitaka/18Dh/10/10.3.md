# 𑁧𑁦𑁇𑁩 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀓𑀼𑀫𑀸𑀭𑀓𑀯𑀢𑁆𑀣𑀼

* 𑀤𑀡𑁆𑀟𑀯𑀕𑁆𑀕

* 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀓𑀼𑀫𑀸𑀭𑀓𑀯𑀢𑁆𑀣𑀼

(131.)

141\. _𑀲𑀼𑀔𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀬𑁄 𑀤𑀡𑁆𑀟𑁂𑀦 𑀯𑀺𑀳𑀺𑀁𑀲𑀢𑀺;_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀲𑀼𑀔𑀫𑁂𑀲𑀸𑀦𑁄,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑁄 𑀦 𑀮𑀪𑀢𑁂 𑀲𑀼𑀔𑀁𑁇_  


(132.)

142\. _𑀲𑀼𑀔𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀬𑁄 𑀤𑀡𑁆𑀟𑁂𑀦 𑀦 𑀳𑀺𑀁𑀲𑀢𑀺;_
