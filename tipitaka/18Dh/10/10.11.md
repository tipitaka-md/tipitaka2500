# 𑁧𑁦𑁇𑁧𑁧 𑀲𑀼𑀔𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀤𑀡𑁆𑀟𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀔𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(145.)

155\. _𑀉𑀤𑀓𑀜𑁆𑀳𑀺 𑀦𑀬𑀦𑁆𑀢𑀺 𑀦𑁂𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀉𑀲𑀼𑀓𑀸𑀭𑀸 𑀦𑀫𑀬𑀦𑁆𑀢𑀺 𑀢𑁂𑀚𑀦𑀁;_  
_𑀤𑀸𑀭𑀼𑀁 𑀦𑀫𑀬𑀦𑁆𑀢𑀺 𑀢𑀘𑁆𑀙𑀓𑀸,_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀫𑀬𑀦𑁆𑀢𑀺 𑀲𑀼𑀩𑁆𑀩𑀢𑀸𑁇_
