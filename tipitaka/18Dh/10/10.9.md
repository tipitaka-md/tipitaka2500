# 𑁧𑁦𑁇𑁯 𑀲𑀦𑁆𑀢𑀢𑀺𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀤𑀡𑁆𑀟𑀯𑀕𑁆𑀕

### 𑀲𑀦𑁆𑀢𑀢𑀺𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(142.)

152\. _𑀅𑀮𑀗𑁆𑀓𑀢𑁄 𑀘𑁂𑀧𑀺 𑀲𑀫𑀁 𑀘𑀭𑁂𑀬𑁆𑀬,_  
_𑀲𑀦𑁆𑀢𑁄 𑀤𑀦𑁆𑀢𑁄 𑀦𑀺𑀬𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻;_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀪𑀽𑀢𑁂𑀲𑀼 𑀦𑀺𑀥𑀸𑀬 𑀤𑀡𑁆𑀟𑀁,_  
_𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑁄 𑀲𑀫𑀡𑁄 𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑁇_
