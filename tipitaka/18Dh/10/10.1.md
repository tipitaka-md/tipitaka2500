# 𑁧𑁦𑁇𑁧 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀤𑀡𑁆𑀟𑀯𑀕𑁆𑀕

### 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(129.)

139. _𑀲𑀩𑁆𑀩𑁂 𑀢𑀲𑀦𑁆𑀢𑀺 𑀤𑀡𑁆𑀟𑀲𑁆𑀲,_  
_𑀲𑀩𑁆𑀩𑁂 𑀪𑀸𑀬𑀦𑁆𑀢𑀺 𑀫𑀘𑁆𑀘𑀼𑀦𑁄;_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀫𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀦 𑀳𑀦𑁂𑀬𑁆𑀬 𑀦 𑀖𑀸𑀢𑀬𑁂𑁇_
