# 𑁧𑁨𑁇𑁯 𑀘𑀽𑀍𑀅𑀓𑀸𑀮𑁅𑀧𑀸𑀲𑀓𑀯𑀢𑁆𑀣𑀼

* 𑀅𑀢𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀘𑀽𑀍𑀅𑀓𑀸𑀮𑁅𑀧𑀸𑀲𑀓𑀯𑀢𑁆𑀣𑀼

(165.)

177\. _𑀅𑀢𑁆𑀢𑀦𑀸 𑀳𑀺 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁,_  
_𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀗𑁆𑀓𑀺𑀮𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀓𑀢𑀁 𑀧𑀸𑀧𑀁,_  
_𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀲𑀼𑀤𑁆𑀥𑀻 𑀅𑀲𑀼𑀤𑁆𑀥𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁,_  
_𑀦𑀸𑀜𑁆𑀜𑁄 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀲𑁄𑀥𑀬𑁂𑁇_
