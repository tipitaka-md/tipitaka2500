# 𑁧𑁨𑁇𑁧𑁦 𑀅𑀢𑁆𑀢𑀤𑀢𑁆𑀣𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀅𑀢𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀢𑁆𑀢𑀤𑀢𑁆𑀣𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(166.)

178\. _𑀅𑀢𑁆𑀢𑀤𑀢𑁆𑀣𑀁 𑀧𑀭𑀢𑁆𑀣𑁂𑀦,_  
_𑀩𑀳𑀼𑀦𑀸𑀧𑀺 𑀦 𑀳𑀸𑀧𑀬𑁂;_  
_𑀅𑀢𑁆𑀢𑀤𑀢𑁆𑀣𑀫𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀤𑀢𑁆𑀣𑀧𑀲𑀼𑀢𑁄 𑀲𑀺𑀬𑀸𑁇_
