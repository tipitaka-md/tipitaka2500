# 𑁧𑁬𑁇𑁪 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀯𑀢𑁆𑀣𑀼

(214.)

232\. _𑀭𑀢𑀺𑀬𑀸 𑀚𑀸𑀬𑀢𑀻 𑀲𑁄𑀓𑁄,_

_𑀦𑀢𑁆𑀣𑀺 𑀲𑁄𑀓𑁄 𑀓𑀼𑀢𑁄 𑀪𑀬𑀁𑁇_
