# 𑁧𑁬𑁇𑁯 𑀦𑀦𑁆𑀤𑀺𑀬𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀺𑀬𑀯𑀢𑁆𑀣𑀼

(219.)

237\. _𑀘𑀺𑀭𑀧𑁆𑀧𑀯𑀸𑀲𑀺𑀁 𑀧𑀼𑀭𑀺𑀲𑀁,_  
_𑀤𑀽𑀭𑀢𑁄 𑀲𑁄𑀢𑁆𑀣𑀺𑀫𑀸𑀕𑀢𑀁;_  
_𑀜𑀸𑀢𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀳𑀚𑁆𑀚𑀸 𑀘,_  
_𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀆𑀕𑀢𑀁𑁇_  


(220.)

238\. _𑀢𑀣𑁂𑀯 𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑀫𑁆𑀧𑀺,_  
_𑀅𑀲𑁆𑀫𑀸 𑀮𑁄𑀓𑀸 𑀧𑀭𑀁 𑀕𑀢𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀧𑀝𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺,_  
_𑀧𑀺𑀬𑀁 𑀜𑀸𑀢𑀻𑀯 𑀆𑀕𑀢𑀁𑁇_  


---
