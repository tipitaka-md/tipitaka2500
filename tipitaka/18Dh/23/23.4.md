# 𑁨𑁩𑁇𑁪 𑀧𑀲𑁂𑀦𑀤𑀺𑀓𑁄𑀲𑀮𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀲𑁂𑀦𑀤𑀺𑀓𑁄𑀲𑀮𑀯𑀢𑁆𑀣𑀼

(325.)

350\. _𑀫𑀺𑀤𑁆𑀥𑀻 𑀬𑀤𑀸 𑀳𑁄𑀢𑀺 𑀫𑀳𑀕𑁆𑀖𑀲𑁄 𑀘,_  
_𑀦𑀺𑀤𑁆𑀤𑀸𑀬𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀲𑀸𑀬𑀻;_  
_𑀫𑀳𑀸𑀯𑀭𑀸𑀳𑁄𑀯 𑀦𑀺𑀯𑀸𑀧𑀧𑀼𑀝𑁆𑀞𑁄,_  
_𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀕𑀩𑁆𑀪𑀫𑀼𑀧𑁂𑀢𑀺 𑀫𑀦𑁆𑀤𑁄𑁇_
