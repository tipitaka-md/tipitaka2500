# 𑁨𑁩𑁇𑁧 𑀅𑀢𑁆𑀢𑀤𑀦𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀦𑀸𑀕𑀯𑀕𑁆𑀕

### 𑀅𑀢𑁆𑀢𑀤𑀦𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(320.)

345. _𑀅𑀳𑀁 𑀦𑀸𑀕𑁄𑀯 𑀲𑀗𑁆𑀕𑀸𑀫𑁂,_  
_𑀘𑀸𑀧𑀢𑁄 𑀧𑀢𑀺𑀢𑀁 𑀲𑀭𑀁;_  
_𑀅𑀢𑀺𑀯𑀸𑀓𑁆𑀬𑀁 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀁,_  
_𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀳𑀺 𑀩𑀳𑀼𑀚𑁆𑀚𑀦𑁄𑁇_  


(321.)

346. _𑀤𑀦𑁆𑀢𑀁 𑀦𑀬𑀦𑁆𑀢𑀺 𑀲𑀫𑀺𑀢𑀺𑀁,_  
_𑀤𑀦𑁆𑀢𑀁 𑀭𑀸𑀚𑀸𑀪𑀺𑀭𑀽𑀳𑀢𑀺;_  
_𑀤𑀦𑁆𑀢𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀬𑁄𑀢𑀺𑀯𑀸𑀓𑁆𑀬𑀁 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀢𑀺𑁇_  


(322.)

347. _𑀯𑀭𑀫𑀲𑁆𑀲𑀢𑀭𑀸 𑀤𑀦𑁆𑀢𑀸,_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑀸 𑀘 𑀲𑀺𑀦𑁆𑀥𑀯𑀸;_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀸 𑀘 𑀫𑀳𑀸𑀦𑀸𑀕𑀸,_  
_𑀅𑀢𑁆𑀢𑀤𑀦𑁆𑀢𑁄 𑀢𑀢𑁄 𑀯𑀭𑀁𑁇_
