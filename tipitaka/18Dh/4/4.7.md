# 𑁪𑁇𑁭 𑀙𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀉𑀧𑀸𑀲𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀼𑀧𑁆𑀨𑀯𑀕𑁆𑀕

### 𑀙𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀉𑀧𑀸𑀲𑀓𑀯𑀢𑁆𑀣𑀼

(51.)

55\. _𑀬𑀣𑀸𑀧𑀺 𑀭𑀼𑀘𑀺𑀭𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑀁 𑀅𑀕𑀦𑁆𑀥𑀓𑀁;_  
_𑀏𑀯𑀁 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸 𑀯𑀸𑀘𑀸,_  
_𑀅𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀅𑀓𑀼𑀩𑁆𑀩𑀢𑁄𑁇_  


(52.)

56\. _𑀬𑀣𑀸𑀧𑀺 𑀭𑀼𑀘𑀺𑀭𑀁 𑀧𑀼𑀧𑁆𑀨𑀁,_  
_𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑀁 𑀲𑀕𑀦𑁆𑀥𑀓𑀁;_  
_𑀏𑀯𑀁 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸 𑀯𑀸𑀘𑀸,_  
_𑀲𑀨𑀮𑀸 𑀳𑁄𑀢𑀺 𑀓𑀼𑀩𑁆𑀩𑀢𑁄𑁇_
