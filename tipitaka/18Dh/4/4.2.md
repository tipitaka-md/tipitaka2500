# 𑁪𑁇𑁨 𑀫𑀭𑀻𑀘𑀺𑀓𑀫𑁆𑀫𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀼𑀧𑁆𑀨𑀯𑀕𑁆𑀕

### 𑀫𑀭𑀻𑀘𑀺𑀓𑀫𑁆𑀫𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(46.)

50\. _𑀨𑁂𑀡𑀽𑀧𑀫𑀁 𑀓𑀸𑀬𑀫𑀺𑀫𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀫𑀭𑀻𑀘𑀺𑀥𑀫𑁆𑀫𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀥𑀸𑀦𑁄;_  
_𑀙𑁂𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀧𑀼𑀧𑁆𑀨𑀓𑀸𑀦𑀺,_  
_𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀫𑀘𑁆𑀘𑀼𑀭𑀸𑀚𑀲𑁆𑀲 𑀕𑀘𑁆𑀙𑁂𑁇_
