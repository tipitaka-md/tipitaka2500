# 𑁨𑁨𑁇𑁭 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

* 𑀦𑀺𑀭𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(315.)

339\. _𑀦𑀕𑀭𑀁 𑀬𑀣𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀕𑀼𑀢𑁆𑀢𑀁 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀁;_  
_𑀏𑀯𑀁 𑀕𑁄𑀧𑁂𑀣 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀔𑀡𑁄 𑀯𑁄 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀔𑀡𑀸𑀢𑀻𑀢𑀸 𑀳𑀺 𑀲𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_
