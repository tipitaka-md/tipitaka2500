# 𑁨𑁨𑁇𑁮 𑀦𑀺𑀕𑀡𑁆𑀞𑀯𑀢𑁆𑀣𑀼

### 𑀦𑀺𑀭𑀬𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀕𑀡𑁆𑀞𑀯𑀢𑁆𑀣𑀼

(316.)

340\. _𑀅𑀮𑀚𑁆𑀚𑀺𑀢𑀸𑀬𑁂 𑀮𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀮𑀚𑁆𑀚𑀺𑀢𑀸𑀬𑁂 𑀦 𑀮𑀚𑁆𑀚𑀭𑁂;_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑀸𑀤𑀸𑀦𑀸,_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁𑁇_  


(317.)

341\. _𑀅𑀪𑀬𑁂 𑀪𑀬𑀤𑀲𑁆𑀲𑀺𑀦𑁄,_  
_𑀪𑀬𑁂 𑀘𑀸𑀪𑀬𑀤𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑀸𑀤𑀸𑀦𑀸,_  
_𑀲𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁𑁇_
