# 𑁨𑁨𑁇𑁫 𑀤𑀼𑀩𑁆𑀩𑀘𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀦𑀺𑀭𑀬𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀩𑁆𑀩𑀘𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(311.)

335. _𑀓𑀼𑀲𑁄 𑀬𑀣𑀸 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁄,_  
_𑀳𑀢𑁆𑀣𑀫𑁂𑀯𑀸𑀦𑀼𑀓𑀦𑁆𑀢𑀢𑀺;_  
_𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀤𑀼𑀧𑁆𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀁,_  
_𑀦𑀺𑀭𑀬𑀸𑀬𑀼𑀧𑀓𑀟𑁆𑀠𑀢𑀺𑁇_  


(312.)

336. _𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀺𑀣𑀺𑀮𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀬𑀁 𑀯𑀢𑀁;_  
_𑀲𑀗𑁆𑀓𑀲𑁆𑀲𑀭𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀦 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(313.)

337. _𑀓𑀬𑀺𑀭𑀸 𑀘𑁂 𑀓𑀬𑀺𑀭𑀸𑀣𑁂𑀦𑀁,_  
_𑀤𑀍𑀳𑀫𑁂𑀦𑀁 𑀧𑀭𑀓𑁆𑀓𑀫𑁂;_  
_𑀲𑀺𑀣𑀺𑀮𑁄 𑀳𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑁄,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀆𑀓𑀺𑀭𑀢𑁂 𑀭𑀚𑀁𑁇_
