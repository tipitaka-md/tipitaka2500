# 𑁨𑁨𑁇𑁧 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀯𑀢𑁆𑀣𑀼

### 𑀦𑀺𑀭𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀯𑀢𑁆𑀣𑀼

(306.)

330\. _𑀅𑀪𑀽𑀢𑀯𑀸𑀤𑀻 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑁂𑀢𑀺,_  
_𑀬𑁄 𑀯𑀸𑀧𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦 𑀓𑀭𑁄𑀫𑀺 𑀘𑀸𑀳;_  
_𑀉𑀪𑁄𑀧𑀺 𑀢𑁂 𑀧𑁂𑀘𑁆𑀘 𑀲𑀫𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀳𑀻𑀦𑀓𑀫𑁆𑀫𑀸 𑀫𑀦𑀼𑀚𑀸 𑀧𑀭𑀢𑁆𑀣𑁇_
