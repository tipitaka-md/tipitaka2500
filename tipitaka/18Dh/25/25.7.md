# 𑁨𑁫𑁇𑁭 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(368.)

395. _𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀺𑀳𑀸𑀭𑀻 𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀲𑀦𑁆𑀦𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂;_  
_𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂 𑀧𑀤𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀲𑀗𑁆𑀔𑀸𑀭𑀽𑀧𑀲𑀫𑀁 𑀲𑀼𑀔𑀁𑁇_  


(369.)

396. _𑀲𑀺𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑀁 𑀦𑀸𑀯𑀁,_  
_𑀲𑀺𑀢𑁆𑀢𑀸 𑀢𑁂 𑀮𑀳𑀼𑀫𑁂𑀲𑁆𑀲𑀢𑀺;_  
_𑀙𑁂𑀢𑁆𑀯𑀸 𑀭𑀸𑀕𑀜𑁆𑀘 𑀤𑁄𑀲𑀜𑁆𑀘,_  
_𑀢𑀢𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑁂𑀳𑀺𑀲𑀺𑁇_  


(370.)

397. _𑀧𑀜𑁆𑀘 𑀙𑀺𑀦𑁆𑀤𑁂 𑀧𑀜𑁆𑀘 𑀚𑀳𑁂,_  
_𑀧𑀜𑁆𑀘 𑀘𑀼𑀢𑁆𑀢𑀭𑀺 𑀪𑀸𑀯𑀬𑁂;_  
_𑀧𑀜𑁆𑀘 𑀲𑀗𑁆𑀕𑀸𑀢𑀺𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_“𑀑𑀖𑀢𑀺𑀡𑁆𑀡𑁄”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(371.)

398. _𑀛𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀸 𑀧𑀫𑀸𑀤𑁄,_  
_𑀫𑀸 𑀢𑁂 𑀓𑀸𑀫𑀕𑀼𑀡𑁂 𑀭𑀫𑁂𑀲𑁆𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁;_  
_𑀫𑀸 𑀮𑁄𑀳𑀕𑀼𑀍𑀅𑀁 𑀕𑀺𑀮𑀻 𑀧𑀫𑀢𑁆𑀢𑁄,_  
_𑀫𑀸 𑀓𑀦𑁆𑀤𑀺 “𑀤𑀼𑀓𑁆𑀔𑀫𑀺𑀤𑀦𑁆”𑀢𑀺 𑀤𑀬𑁆𑀳𑀫𑀸𑀦𑁄𑁇_  


(372.)

399. _𑀦𑀢𑁆𑀣𑀺 𑀛𑀸𑀦𑀁 𑀅𑀧𑀜𑁆𑀜𑀲𑁆𑀲,_  
_𑀧𑀜𑁆𑀜𑀸 𑀦𑀢𑁆𑀣𑀺 𑀅𑀛𑀸𑀬𑀢𑁄;_  
_𑀬𑀫𑁆𑀳𑀺 𑀛𑀸𑀦𑀜𑁆𑀘 𑀧𑀜𑁆𑀜𑀸 𑀘,_  
_𑀲 𑀯𑁂 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(373.)

400. _𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀅𑀫𑀸𑀦𑀼𑀲𑀻 𑀭𑀢𑀺 𑀳𑁄𑀢𑀺,_  
_𑀲𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑁄𑁇_  


(374.)

401. _𑀬𑀢𑁄 𑀬𑀢𑁄 𑀲𑀫𑁆𑀫𑀲𑀢𑀺,_  
_𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁;_  
_𑀮𑀪𑀢𑀻 𑀧𑀻𑀢𑀺𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁,_  
_𑀅𑀫𑀢𑀁 𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀢𑀁𑁇_  


(375.)

402. _𑀢𑀢𑁆𑀭𑀸𑀬𑀫𑀸𑀤𑀺 𑀪𑀯𑀢𑀺,_  
_𑀇𑀥 𑀧𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀕𑀼𑀢𑁆𑀢𑀺 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺,_  
_𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀘 𑀲𑀁𑀯𑀭𑁄𑁇_  


(376.)

403. _𑀫𑀺𑀢𑁆𑀢𑁂 𑀪𑀚𑀲𑁆𑀲𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑁂,_  
_𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁂 𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑁂;_  
_𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑀯𑀼𑀢𑁆𑀬𑀲𑁆𑀲,_  
_𑀆𑀘𑀸𑀭𑀓𑀼𑀲𑀮𑁄 𑀲𑀺𑀬𑀸;_  
_𑀢𑀢𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀩𑀳𑀼𑀮𑁄,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_
