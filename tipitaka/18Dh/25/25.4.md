# 𑁨𑁫𑁇𑁪 𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(364.)

391\. _𑀥𑀫𑁆𑀫𑀸𑀭𑀸𑀫𑁄 𑀥𑀫𑁆𑀫𑀭𑀢𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀬𑀁;_  
_𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺𑁇_
