# 𑁨𑁬𑁇𑁬 𑀅𑀜𑁆𑀜𑀢𑀭𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀯𑀢𑁆𑀣𑀼

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀅𑀜𑁆𑀜𑀢𑀭𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀯𑀢𑁆𑀣𑀼

(388.)

416\. _𑀩𑀸𑀳𑀺𑀢𑀧𑀸𑀧𑁄𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄,_  
_𑀲𑀫𑀘𑀭𑀺𑀬𑀸 𑀲𑀫𑀡𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀫𑀢𑁆𑀢𑀦𑁄 𑀫𑀮𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 “𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
