# 𑁨𑁬𑁇𑁨𑁫 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(408.)

436\. _𑀅𑀓𑀓𑁆𑀓𑀲𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀁,_
