# 𑁨𑁬𑁇𑁯 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(392.)

420. _𑀬𑀫𑁆𑀳𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀚𑀸𑀦𑁂𑀬𑁆𑀬,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑁂𑀬𑁆𑀬,_  
_𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑁆𑀢𑀁𑀯 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_
