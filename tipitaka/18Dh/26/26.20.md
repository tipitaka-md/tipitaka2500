# 𑁨𑁬𑁇𑁨𑁦 𑀔𑁂𑀫𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀯𑀢𑁆𑀣𑀼

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀔𑁂𑀫𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀯𑀢𑁆𑀣𑀼

(403.)

431\. _𑀕𑀫𑁆𑀪𑀻𑀭𑀧𑀜𑁆𑀜𑀁 𑀫𑁂𑀥𑀸𑀯𑀺𑀁,_  
_𑀫𑀕𑁆𑀕𑀸𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀓𑁄𑀯𑀺𑀤𑀁;_
