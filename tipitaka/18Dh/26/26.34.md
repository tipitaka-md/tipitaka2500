# 𑁨𑁬𑁇𑁩𑁪 𑀚𑁄𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀚𑁄𑀢𑀺𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(417.)

445\. _𑀬𑁄𑀥 𑀢𑀡𑁆𑀳𑀁 𑀧𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀦𑀸𑀕𑀸𑀭𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂;_  
_𑀢𑀡𑁆𑀳𑀸𑀪𑀯𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁,_  
_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_
