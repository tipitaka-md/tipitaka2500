# 𑁨𑁬𑁇𑁨𑁪 𑀫𑀳𑀸𑀧𑀦𑁆𑀣𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀧𑀦𑁆𑀣𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(407.)

435\. _𑀬𑀲𑁆𑀲 𑀭𑀸𑀕𑁄 𑀘 𑀤𑁄𑀲𑁄 𑀘,_  
_𑀫𑀸𑀦𑁄 𑀫𑀓𑁆𑀔𑁄 𑀘 𑀧𑀸𑀢𑀺𑀢𑁄;_

_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_
