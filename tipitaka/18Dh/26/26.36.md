# 𑁨𑁬𑁇𑁩𑁬 𑀦𑀝𑀧𑀼𑀢𑁆𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀦𑀝𑀧𑀼𑀢𑁆𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(419.)

447\. _𑀳𑀺𑀢𑁆𑀯𑀸 𑀭𑀢𑀺𑀜𑁆𑀘 𑀅𑀭𑀢𑀺𑀜𑁆𑀘,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀁 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀸𑀪𑀺𑀪𑀼𑀁 𑀯𑀻𑀭𑀁,_  
_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_
