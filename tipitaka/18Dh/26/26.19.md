# 𑁨𑁬𑁇𑁧𑁯 𑀅𑀜𑁆𑀜𑀢𑀭𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀢𑁆𑀣𑀼

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

* 𑀅𑀜𑁆𑀜𑀢𑀭𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀢𑁆𑀣𑀼

(402.)

430\. _𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀇𑀥𑁂𑀯 𑀔𑀬𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀧𑀦𑁆𑀦𑀪𑀸𑀭𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_
