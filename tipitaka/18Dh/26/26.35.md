# 𑁨𑁬𑁇𑁩𑁫 𑀦𑀝𑀧𑀼𑀢𑁆𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀕𑁆𑀕

### 𑀦𑀝𑀧𑀼𑀢𑁆𑀢𑀓𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(418.)

446\. _𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀓𑀁 𑀬𑁄𑀕𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀁 𑀬𑁄𑀕𑀁 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀲𑀩𑁆𑀩𑀬𑁄𑀕𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁𑁇_
