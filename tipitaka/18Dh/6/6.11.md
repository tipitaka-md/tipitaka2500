# 𑁬𑁇𑁧𑁧 𑀧𑀜𑁆𑀘𑀲𑀢𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑀕𑁆𑀕

### 𑀧𑀜𑁆𑀘𑀲𑀢𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(87.)

93\. _𑀓𑀡𑁆𑀳𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑁆𑀧𑀳𑀸𑀬,_  
_𑀲𑀼𑀓𑁆𑀓𑀁 𑀪𑀸𑀯𑁂𑀣 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀑𑀓𑀸 𑀅𑀦𑁄𑀓𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀯𑀺𑀯𑁂𑀓𑁂 𑀬𑀢𑁆𑀣 𑀤𑀽𑀭𑀫𑀁𑁇_  


(88.)

94\. _𑀢𑀢𑁆𑀭𑀸𑀪𑀺𑀭𑀢𑀺𑀫𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂 𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑁄;_  
_𑀧𑀭𑀺𑀬𑁄𑀤𑀧𑁂𑀬𑁆𑀬 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀮𑁂𑀲𑁂𑀳𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


(89.)

95\. _𑀬𑁂𑀲𑀁 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀬𑀗𑁆𑀕𑁂𑀲𑀼,_  
_𑀲𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁;_  
_𑀆𑀤𑀸𑀦𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂,_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀬𑁂 𑀭𑀢𑀸;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀸 𑀚𑀼𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀢𑁂 𑀮𑁄𑀓𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸𑁇_  


---

96\. 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
