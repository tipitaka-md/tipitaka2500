# 𑁬𑁇𑁫 𑀧𑀡𑁆𑀟𑀺𑀢𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑀕𑁆𑀕

* 𑀧𑀡𑁆𑀟𑀺𑀢𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(80.)

86\. _𑀉𑀤𑀓𑀜𑁆𑀳𑀺 𑀦𑀬𑀦𑁆𑀢𑀺 𑀦𑁂𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀉𑀲𑀼𑀓𑀸𑀭𑀸 𑀦𑀫𑀬𑀦𑁆𑀢𑀺 𑀢𑁂𑀚𑀦𑀁;_  
_𑀤𑀸𑀭𑀼𑀁 𑀦𑀫𑀬𑀦𑁆𑀢𑀺 𑀢𑀘𑁆𑀙𑀓𑀸,_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀫𑀬𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇_
