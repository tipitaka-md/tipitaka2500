# 𑁬𑁇𑁧𑁦 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀡𑁆𑀟𑀺𑀢𑀯𑀕𑁆𑀕

* 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(85.)

91\. _𑀅𑀧𑁆𑀧𑀓𑀸 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀬𑁂 𑀚𑀦𑀸 𑀧𑀸𑀭𑀕𑀸𑀫𑀺𑀦𑁄;_  
_𑀅𑀣𑀸𑀬𑀁 𑀇𑀢𑀭𑀸 𑀧𑀚𑀸,_  
_𑀢𑀻𑀭𑀫𑁂𑀯𑀸𑀦𑀼𑀥𑀸𑀯𑀢𑀺𑁇_  


(86.)

92\. _𑀬𑁂 𑀘 𑀔𑁄 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑁂,_  
_𑀥𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀢𑁂 𑀚𑀦𑀸 𑀧𑀸𑀭𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀘𑁆𑀘𑀼𑀥𑁂𑀬𑁆𑀬𑀁 𑀲𑀼𑀤𑀼𑀢𑁆𑀢𑀭𑀁𑁇_
