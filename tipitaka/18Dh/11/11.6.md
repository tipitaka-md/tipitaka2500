# 𑁧𑁧𑁇𑁬 𑀫𑀮𑁆𑀮𑀺𑀓𑀸𑀤𑁂𑀯𑀻𑀯𑀢𑁆𑀣𑀼

### 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

### 𑀫𑀮𑁆𑀮𑀺𑀓𑀸𑀤𑁂𑀯𑀻𑀯𑀢𑁆𑀣𑀼

(151.)

162. _𑀚𑀻𑀭𑀦𑁆𑀢𑀺 𑀯𑁂 𑀭𑀸𑀚𑀭𑀣𑀸 𑀲𑀼𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀅𑀣𑁄 𑀲𑀭𑀻𑀭𑀫𑁆𑀧𑀺 𑀚𑀭𑀁 𑀉𑀧𑁂𑀢𑀺;_  
_𑀲𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀚𑀭𑀁 𑀉𑀧𑁂𑀢𑀺,_  
_𑀲𑀦𑁆𑀢𑁄 𑀳𑀯𑁂 𑀲𑀩𑁆𑀪𑀺 𑀧𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀺𑁇_
