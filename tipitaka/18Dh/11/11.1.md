# 𑁧𑁧𑁇𑁧 𑀯𑀺𑀲𑀸𑀔𑀸𑀬𑀲𑀳𑀸𑀬𑀺𑀓𑀸𑀦𑀁𑀯𑀢𑁆𑀣𑀼

### 𑀚𑀭𑀸𑀯𑀕𑁆𑀕

### 𑀯𑀺𑀲𑀸𑀔𑀸𑀬𑀲𑀳𑀸𑀬𑀺𑀓𑀸𑀦𑀁𑀯𑀢𑁆𑀣𑀼

(146.)

157\. _𑀓𑁄 𑀦𑀼 𑀳𑀸𑀲𑁄 𑀓𑀺𑀫𑀸𑀦𑀦𑁆𑀤𑁄,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁂 𑀲𑀢𑀺;_  
_𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂𑀦 𑀑𑀦𑀤𑁆𑀥𑀸,_  
_𑀧𑀤𑀻𑀧𑀁 𑀦 𑀕𑀯𑁂𑀲𑀣𑁇_
