# 𑁮𑁇𑁭 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀣𑁂𑀭𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀢𑁆𑀣𑀼

* 𑀲𑀳𑀲𑁆𑀲𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀣𑁂𑀭𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀢𑁆𑀣𑀼

(108.)

116\. _𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀬𑀺𑀝𑁆𑀞𑀁 𑀯 𑀳𑀼𑀢𑀁 𑀯 𑀮𑁄𑀓𑁂,_  
_𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑀁 𑀬𑀚𑁂𑀣 𑀧𑀼𑀜𑁆𑀜𑀧𑁂𑀓𑁆𑀔𑁄;_  
_𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀢𑀁 𑀦 𑀘𑀢𑀼𑀪𑀸𑀕𑀫𑁂𑀢𑀺,_  
_𑀅𑀪𑀺𑀯𑀸𑀤𑀦𑀸 𑀉𑀚𑁆𑀚𑀼𑀕𑀢𑁂𑀲𑀼 𑀲𑁂𑀬𑁆𑀬𑁄𑁇_
