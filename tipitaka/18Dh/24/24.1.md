# 𑁨𑁪𑁇𑁧 𑀓𑀧𑀺𑀮𑀫𑀘𑁆𑀙𑀯𑀢𑁆𑀣𑀼

### 𑀢𑀡𑁆𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀓𑀧𑀺𑀮𑀫𑀘𑁆𑀙𑀯𑀢𑁆𑀣𑀼

(334.)

360\. _𑀫𑀦𑀼𑀚𑀲𑁆𑀲 𑀧𑀫𑀢𑁆𑀢𑀘𑀸𑀭𑀺𑀦𑁄,_  
_𑀢𑀡𑁆𑀳𑀸 𑀯𑀟𑁆𑀠𑀢𑀺 𑀫𑀸𑀮𑀼𑀯𑀸 𑀯𑀺𑀬;_  
_𑀲𑁄 𑀧𑁆𑀮𑀯𑀢𑀻 𑀳𑀼𑀭𑀸 𑀳𑀼𑀭𑀁,_  
_𑀨𑀮𑀫𑀺𑀘𑁆𑀙𑀁𑀯 𑀯𑀦𑀲𑁆𑀫𑀺 𑀯𑀸𑀦𑀭𑁄𑁇_  


(335.)

361\. _𑀬𑀁 𑀏𑀲𑀸 𑀲𑀳𑀢𑁂 𑀚𑀫𑁆𑀫𑀻,_  
_𑀢𑀡𑁆𑀳𑀸 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀸;_  
_𑀲𑁄𑀓𑀸 𑀢𑀲𑁆𑀲 𑀧𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺,_  
_𑀅𑀪𑀺𑀯𑀝𑁆𑀞𑀁𑀯 𑀩𑀻𑀭𑀡𑀁𑁇_  


(336.)

362\. _𑀬𑁄 𑀘𑁂𑀢𑀁 𑀲𑀳𑀢𑁂 𑀚𑀫𑁆𑀫𑀺𑀁,_  
_𑀢𑀡𑁆𑀳𑀁 𑀮𑁄𑀓𑁂 𑀤𑀼𑀭𑀘𑁆𑀘𑀬𑀁;_  
_𑀲𑁄𑀓𑀸 𑀢𑀫𑁆𑀳𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺,_  
_𑀉𑀤𑀩𑀺𑀦𑁆𑀤𑀼𑀯 𑀧𑁄𑀓𑁆𑀔𑀭𑀸𑁇_  


(337.)

363\. _𑀢𑀁 𑀯𑁄 𑀯𑀤𑀸𑀫𑀺 𑀪𑀤𑁆𑀤𑀁 𑀯𑁄,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀢𑀡𑁆𑀳𑀸𑀬 𑀫𑀽𑀮𑀁 𑀔𑀡𑀣,_  
_𑀉𑀲𑀻𑀭𑀢𑁆𑀣𑁄𑀯 𑀩𑀻𑀭𑀡𑀁;_  
_𑀫𑀸 𑀯𑁄 𑀦𑀍𑀅𑀁𑀯 𑀲𑁄𑀢𑁄𑀯,_  
_𑀫𑀸𑀭𑁄 𑀪𑀜𑁆𑀚𑀺 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁𑁇_
