# 𑁨𑁪𑁇𑁯 𑀉𑀧𑀓𑀸𑀚𑀻𑀯𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀢𑀡𑁆𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀓𑀸𑀚𑀻𑀯𑀓𑀯𑀢𑁆𑀣𑀼

(353.)

379\. _𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀪𑀽 𑀲𑀩𑁆𑀩𑀯𑀺𑀤𑀽𑀳𑀫𑀲𑁆𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀦𑀽𑀧𑀮𑀺𑀢𑁆𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀚𑀳𑁄 𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄,_  
_𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀓𑀫𑀼𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀁𑁇_
