# 𑁨𑁪𑁇𑁮 𑀫𑀸𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀢𑀡𑁆𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀫𑀸𑀭𑀯𑀢𑁆𑀣𑀼

(351.)

377\. _𑀦𑀺𑀝𑁆𑀞𑀗𑁆𑀕𑀢𑁄 𑀅𑀲𑀦𑁆𑀢𑀸𑀲𑀻,_  
_𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑁄 𑀅𑀦𑀗𑁆𑀕𑀡𑁄;_  
_𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺 𑀪𑀯𑀲𑀮𑁆𑀮𑀸𑀦𑀺,_  
_𑀅𑀦𑁆𑀢𑀺𑀫𑁄𑀬𑀁 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑁄𑁇_  


(352.)

378\. _𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑁄 𑀅𑀦𑀸𑀤𑀸𑀦𑁄,_  
_𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀤𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀅𑀓𑁆𑀔𑀭𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀁,_  
_𑀚𑀜𑁆𑀜𑀸 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀸𑀦𑀺 𑀘;_  
_𑀲 𑀯𑁂 “𑀅𑀦𑁆𑀢𑀺𑀫𑀲𑀸𑀭𑀻𑀭𑁄,_  
_𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑁄”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
