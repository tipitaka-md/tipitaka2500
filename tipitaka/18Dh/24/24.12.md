# 𑁨𑁪𑁇𑁧𑁨 𑀅𑀗𑁆𑀓𑀼𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀢𑀡𑁆𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀗𑁆𑀓𑀼𑀭𑀯𑀢𑁆𑀣𑀼

(356.)

382. _𑀢𑀺𑀡𑀤𑁄𑀲𑀸𑀦𑀺 𑀔𑁂𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀭𑀸𑀕𑀤𑁄𑀲𑀸 𑀅𑀬𑀁 𑀧𑀚𑀸;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀯𑀻𑀢𑀭𑀸𑀕𑁂𑀲𑀼,_  
_𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(357.)

383. _𑀢𑀺𑀡𑀤𑁄𑀲𑀸𑀦𑀺 𑀔𑁂𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀤𑁄𑀲𑀤𑁄𑀲𑀸 𑀅𑀬𑀁 𑀧𑀚𑀸;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀯𑀻𑀢𑀤𑁄𑀲𑁂𑀲𑀼,_  
_𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(358.)

384. _𑀢𑀺𑀡𑀤𑁄𑀲𑀸𑀦𑀺 𑀔𑁂𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀫𑁄𑀳𑀤𑁄𑀲𑀸 𑀅𑀬𑀁 𑀧𑀚𑀸;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀯𑀻𑀢𑀫𑁄𑀳𑁂𑀲𑀼,_  
_𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


(359.)

385. _𑀢𑀺𑀡𑀤𑁄𑀲𑀸𑀦𑀺 𑀔𑁂𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀇𑀘𑁆𑀙𑀸𑀤𑁄𑀲𑀸 𑀅𑀬𑀁 𑀧𑀚𑀸;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀯𑀺𑀕𑀢𑀺𑀘𑁆𑀙𑁂𑀲𑀼,_  
_𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀁𑁇_  


---

386. 𑀢𑀡𑁆𑀳𑀸𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇
