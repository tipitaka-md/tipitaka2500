# 𑁫𑁇𑁧𑁫 𑀯𑀦𑀯𑀸𑀲𑀺𑀢𑀺𑀲𑁆𑀲𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀩𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀯𑀦𑀯𑀸𑀲𑀺𑀢𑀺𑀲𑁆𑀲𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(75.)

80. _𑀅𑀜𑁆𑀜𑀸 𑀳𑀺 𑀮𑀸𑀪𑀽𑀧𑀦𑀺𑀲𑀸,_  
_𑀅𑀜𑁆𑀜𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀕𑀸𑀫𑀺𑀦𑀻;_  
_𑀏𑀯𑀫𑁂𑀢𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀯𑀓𑁄;_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑁂𑀬𑁆𑀬,_  
_𑀯𑀺𑀯𑁂𑀓𑀫𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑀬𑁂𑁇_  


---

81. 𑀩𑀸𑀮𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇
