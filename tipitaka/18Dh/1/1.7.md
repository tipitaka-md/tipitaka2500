# 𑁧𑁇𑁭 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

* 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(9.)

10\. _𑀅𑀦𑀺𑀓𑁆𑀓𑀲𑀸𑀯𑁄 𑀓𑀸𑀲𑀸𑀯𑀁,_  
_𑀬𑁄 𑀯𑀢𑁆𑀣𑀁 𑀧𑀭𑀺𑀤𑀳𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀧𑁂𑀢𑁄 𑀤𑀫𑀲𑀘𑁆𑀘𑁂𑀦,_  
_𑀦 𑀲𑁄 𑀓𑀸𑀲𑀸𑀯𑀫𑀭𑀳𑀢𑀺𑁇_  


(10.)

11\. _𑀬𑁄 𑀘 𑀯𑀦𑁆𑀢𑀓𑀲𑀸𑀯𑀲𑁆𑀲,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁄;_  
_𑀉𑀧𑁂𑀢𑁄 𑀤𑀫𑀲𑀘𑁆𑀘𑁂𑀦,_  
_𑀲 𑀯𑁂 𑀓𑀸𑀲𑀸𑀯𑀫𑀭𑀳𑀢𑀺𑁇_
