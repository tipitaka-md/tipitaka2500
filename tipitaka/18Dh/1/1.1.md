# 𑁧𑁇𑁧 𑀘𑀓𑁆𑀔𑀼𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀘𑀓𑁆𑀔𑀼𑀧𑀸𑀮𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(1.)

2\. _𑀫𑀦𑁄𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀫𑀦𑁄𑀲𑁂𑀝𑁆𑀞𑀸 𑀫𑀦𑁄𑀫𑀬𑀸;_  
_𑀫𑀦𑀲𑀸 𑀘𑁂 𑀧𑀤𑀼𑀝𑁆𑀞𑁂𑀦,_  
_𑀪𑀸𑀲𑀢𑀺 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺 𑀯𑀸;_  
_𑀢𑀢𑁄 𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀦𑁆𑀯𑁂𑀢𑀺,_  
_𑀘𑀓𑁆𑀓𑀁𑀯 𑀯𑀳𑀢𑁄 𑀧𑀤𑀁𑁇_
