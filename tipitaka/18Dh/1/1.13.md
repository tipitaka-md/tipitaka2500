### 𑀲𑀼𑀫𑀦𑀤𑁂𑀯𑀻𑀯𑀢𑁆𑀣𑀼

(18.)

19\. _𑀇𑀥 𑀦𑀦𑁆𑀤𑀢𑀺 𑀧𑁂𑀘𑁆𑀘 𑀦𑀦𑁆𑀤𑀢𑀺,_  
_𑀓𑀢𑀧𑀼𑀜𑁆𑀜𑁄 𑀉𑀪𑀬𑀢𑁆𑀣 𑀦𑀦𑁆𑀤𑀢𑀺;_  
_“𑀧𑀼𑀜𑁆𑀜𑀁 𑀫𑁂 𑀓𑀢𑀦𑁆”𑀢𑀺 𑀦𑀦𑁆𑀤𑀢𑀺,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀦𑀦𑁆𑀤𑀢𑀺 𑀲𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀕𑀢𑁄𑁇_
