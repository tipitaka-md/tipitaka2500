# 𑁧𑁇𑁧𑁨 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(17.)

18. _𑀇𑀥 𑀢𑀧𑁆𑀧𑀢𑀺 𑀧𑁂𑀘𑁆𑀘 𑀢𑀧𑁆𑀧𑀢𑀺,_  
_𑀧𑀸𑀧𑀓𑀸𑀭𑀻 𑀉𑀪𑀬𑀢𑁆𑀣 𑀢𑀧𑁆𑀧𑀢𑀺;_  
_“𑀧𑀸𑀧𑀁 𑀫𑁂 𑀓𑀢𑀦𑁆”𑀢𑀺 𑀢𑀧𑁆𑀧𑀢𑀺,_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀢𑀧𑁆𑀧𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀕𑀢𑁄𑁇_
