# 𑁧𑁇𑁮 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(11.)

12\. _𑀅𑀲𑀸𑀭𑁂 𑀲𑀸𑀭𑀫𑀢𑀺𑀦𑁄,_  
_𑀲𑀸𑀭𑁂 𑀘𑀸𑀲𑀸𑀭𑀤𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀢𑁂 𑀲𑀸𑀭𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑀸𑁇_  


(12.)

13\. _𑀲𑀸𑀭𑀜𑁆𑀘 𑀲𑀸𑀭𑀢𑁄 𑀜𑀢𑁆𑀯𑀸,_  
_𑀅𑀲𑀸𑀭𑀜𑁆𑀘 𑀅𑀲𑀸𑀭𑀢𑁄;_  
_𑀢𑁂 𑀲𑀸𑀭𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀕𑁄𑀘𑀭𑀸𑁇_
