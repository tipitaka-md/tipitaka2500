# 𑁧𑁇𑁧𑁪 𑀤𑁆𑀯𑁂𑀲𑀳𑀸𑀬𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀤𑁆𑀯𑁂𑀲𑀳𑀸𑀬𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(19.)

20. _𑀩𑀳𑀼𑀫𑁆𑀧𑀺 𑀘𑁂 𑀲𑀁𑀳𑀺𑀢 𑀪𑀸𑀲𑀫𑀸𑀦𑁄,_  
_𑀦 𑀢𑀓𑁆𑀓𑀭𑁄 𑀳𑁄𑀢𑀺 𑀦𑀭𑁄 𑀧𑀫𑀢𑁆𑀢𑁄;_  
_𑀕𑁄𑀧𑁄𑀯 𑀕𑀸𑀯𑁄 𑀕𑀡𑀬𑀁 𑀧𑀭𑁂𑀲𑀁,_  
_𑀦 𑀪𑀸𑀕𑀯𑀸 𑀲𑀸𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇_  


(20.)

21. _𑀅𑀧𑁆𑀧𑀫𑁆𑀧𑀺 𑀘𑁂 𑀲𑀁𑀳𑀺𑀢 𑀪𑀸𑀲𑀫𑀸𑀦𑁄,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻;_  
_𑀭𑀸𑀕𑀜𑁆𑀘 𑀤𑁄𑀲𑀜𑁆𑀘 𑀧𑀳𑀸𑀬 𑀫𑁄𑀳𑀁,_  
_𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀬𑀸𑀦𑁄 𑀇𑀥 𑀯𑀸 𑀳𑀼𑀭𑀁 𑀯𑀸,_  
_𑀲 𑀪𑀸𑀕𑀯𑀸 𑀲𑀸𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇_  


---

22. 𑀬𑀫𑀓𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇
