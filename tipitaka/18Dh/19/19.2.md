# 𑁧𑁯𑁇𑁨 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀯𑀢𑁆𑀣𑀼

### 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀯𑀕𑁆𑀕

### 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀯𑀢𑁆𑀣𑀼

(258.)

279\. _𑀦 𑀢𑁂𑀦 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑀸𑀯𑀢𑀸 𑀩𑀳𑀼 𑀪𑀸𑀲𑀢𑀺;_  
_𑀔𑁂𑀫𑀻 𑀅𑀯𑁂𑀭𑀻 𑀅𑀪𑀬𑁄,_  
_“𑀧𑀡𑁆𑀟𑀺𑀢𑁄”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
