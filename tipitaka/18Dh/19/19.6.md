# 𑁧𑁯𑁇𑁬 𑀳𑀢𑁆𑀣𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀯𑀕𑁆𑀕

### 𑀳𑀢𑁆𑀣𑀓𑀯𑀢𑁆𑀣𑀼

(264.)

285. _𑀦 𑀫𑀼𑀡𑁆𑀟𑀓𑁂𑀦 𑀲𑀫𑀡𑁄,_  
_𑀅𑀩𑁆𑀩𑀢𑁄 𑀅𑀮𑀺𑀓𑀁 𑀪𑀡𑀁;_  
_𑀇𑀘𑁆𑀙𑀸𑀮𑁄𑀪𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀫𑀡𑁄 𑀓𑀺𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(265.)

286. _𑀬𑁄 𑀘 𑀲𑀫𑁂𑀢𑀺 𑀧𑀸𑀧𑀸𑀦𑀺,_  
_𑀅𑀡𑀼𑀁𑀣𑀽𑀮𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀲𑀫𑀺𑀢𑀢𑁆𑀢𑀸 𑀳𑀺 𑀧𑀸𑀧𑀸𑀦𑀁,_  
_“𑀲𑀫𑀡𑁄”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
