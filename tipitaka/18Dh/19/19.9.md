# 𑁧𑁯𑁇𑁯 𑀩𑀸𑀮𑀺𑀲𑀺𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀩𑀸𑀮𑀺𑀲𑀺𑀓𑀯𑀢𑁆𑀣𑀼

(270.)

291\. _𑀦 𑀢𑁂𑀦 𑀅𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀺,_

_“𑀅𑀭𑀺𑀬𑁄”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
