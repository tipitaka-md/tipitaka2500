# 𑁧𑁯𑁇𑁩 𑀏𑀓𑀼𑀤𑀸𑀦𑀔𑀻𑀡𑀸𑀲𑀯𑀯𑀢𑁆𑀣𑀼

### 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀼𑀤𑀸𑀦𑀔𑀻𑀡𑀸𑀲𑀯𑀯𑀢𑁆𑀣𑀼

(259.)

280. _𑀦 𑀢𑀸𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑁄,_  
_𑀬𑀸𑀯𑀢𑀸 𑀩𑀳𑀼 𑀪𑀸𑀲𑀢𑀺;_  
_𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀫𑁆𑀧𑀺 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑀢𑀺;_  
_𑀲 𑀯𑁂 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑀺𑁇_
