# 𑁧𑁯𑁇𑁫 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀯𑀕𑁆𑀕

### 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(262.)

283\. _𑀦 𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑀫𑀢𑁆𑀢𑁂𑀦,_  
_𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀯𑀸;_  
_𑀲𑀸𑀥𑀼𑀭𑀽𑀧𑁄 𑀦𑀭𑁄 𑀳𑁄𑀢𑀺,_  
_𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀫𑀘𑁆𑀙𑀭𑀻 𑀲𑀞𑁄𑁇_  


(263.)

284\. _𑀬𑀲𑁆𑀲 𑀘𑁂𑀢𑀁 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁,_  
_𑀫𑀽𑀮𑀖𑀘𑁆𑀘𑀁 𑀲𑀫𑀽𑀳𑀢𑀁;_  
_𑀲 𑀯𑀦𑁆𑀢𑀤𑁄𑀲𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_“𑀲𑀸𑀥𑀼𑀭𑀽𑀧𑁄”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
