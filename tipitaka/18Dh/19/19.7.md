# 𑁧𑁯𑁇𑁭 𑀅𑀜𑁆𑀜𑀢𑀭𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀢𑁆𑀣𑀼

* 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀯𑀕𑁆𑀕

* 𑀅𑀜𑁆𑀜𑀢𑀭𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀢𑁆𑀣𑀼

(266.)

287\. _𑀦 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑀸𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀢𑁂 𑀧𑀭𑁂;_  
_𑀯𑀺𑀲𑁆𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀫𑀸𑀤𑀸𑀬,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀳𑁄𑀢𑀺 𑀦 𑀢𑀸𑀯𑀢𑀸𑁇_  


(267.)

288\. _𑀬𑁄𑀥 𑀧𑀼𑀜𑁆𑀜𑀜𑁆𑀘 𑀧𑀸𑀧𑀜𑁆𑀘,_  
_𑀩𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸;_  
_𑀲𑀗𑁆𑀔𑀸𑀬 𑀮𑁄𑀓𑁂 𑀘𑀭𑀢𑀺,_  
_𑀲 𑀯𑁂 “𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_
