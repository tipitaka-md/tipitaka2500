# 𑁧𑁫𑁇𑁮 𑀲𑀓𑁆𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀲𑀼𑀔𑀯𑀕𑁆𑀕

### 𑀲𑀓𑁆𑀓𑀯𑀢𑁆𑀣𑀼

(206.)

222. _𑀲𑀸𑀳𑀼 𑀤𑀲𑁆𑀲𑀦𑀫𑀭𑀺𑀬𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀦𑀺𑀯𑀸𑀲𑁄 𑀲𑀤𑀸 𑀲𑀼𑀔𑁄;_  
_𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀩𑀸𑀮𑀸𑀦𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀫𑁂𑀯 𑀲𑀼𑀔𑀻 𑀲𑀺𑀬𑀸𑁇_  


(207.)

223. _𑀩𑀸𑀮𑀲𑀗𑁆𑀕𑀢𑀘𑀸𑀭𑀻 𑀳𑀺,_  
_𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦 𑀲𑁄𑀘𑀢𑀺;_  
_𑀤𑀼𑀓𑁆𑀔𑁄 𑀩𑀸𑀮𑁂𑀳𑀺 𑀲𑀁𑀯𑀸𑀲𑁄,_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑁂𑀦𑁂𑀯 𑀲𑀩𑁆𑀩𑀤𑀸;_  
_𑀥𑀻𑀭𑁄 𑀘 𑀲𑀼𑀔𑀲𑀁𑀯𑀸𑀲𑁄,_  
_𑀜𑀸𑀢𑀻𑀦𑀁𑀯 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(208.)

224. 𑀢𑀲𑁆𑀫𑀸 𑀳𑀺—

225. _𑀥𑀻𑀭𑀜𑁆𑀘 𑀧𑀜𑁆𑀜𑀜𑁆𑀘 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀜𑁆𑀘,_  
_𑀥𑁄𑀭𑀬𑁆𑀳𑀲𑀻𑀮𑀁 𑀯𑀢𑀯𑀦𑁆𑀢𑀫𑀭𑀺𑀬𑀁;_  
_𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀁 𑀲𑀼𑀫𑁂𑀥𑀁,_  
_𑀪𑀚𑁂𑀣 𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀧𑀣𑀁𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑁇_  


---

226. 𑀲𑀼𑀔𑀯𑀕𑁆𑀕𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑀫𑁄𑁇
