# 𑁯𑁇𑁧𑁨 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀲𑀓𑁆𑀬𑀯𑀢𑁆𑀣𑀼

* 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀲𑀓𑁆𑀬𑀯𑀢𑁆𑀣𑀼

(128.)

137\. _𑀦 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀦 𑀲𑀫𑀼𑀤𑁆𑀤𑀫𑀚𑁆𑀛𑁂,_  
_𑀦 𑀧𑀩𑁆𑀩𑀢𑀸𑀦𑀁 𑀯𑀺𑀯𑀭𑀁 𑀧𑀯𑀺𑀲𑁆𑀲;_  
_𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀻 𑀲𑁄 𑀚𑀕𑀢𑀺𑀧𑁆𑀧𑀤𑁂𑀲𑁄,_  
_𑀬𑀢𑁆𑀣𑀝𑁆𑀞𑀺𑀢𑀁 𑀦𑀧𑁆𑀧𑀲𑀳𑁂𑀬𑁆𑀬 𑀫𑀘𑁆𑀘𑀼𑁇_
