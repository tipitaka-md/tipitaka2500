# 𑁯𑁇𑁯 𑀓𑁄𑀓𑀲𑀼𑀦𑀔𑀮𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀸𑀧𑀯𑀕𑁆𑀕

### 𑀓𑁄𑀓𑀲𑀼𑀦𑀔𑀮𑀼𑀤𑁆𑀤𑀓𑀯𑀢𑁆𑀣𑀼

(125.)

134\. _𑀬𑁄 𑀅𑀧𑁆𑀧𑀤𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀦𑀭𑀲𑁆𑀲 𑀤𑀼𑀲𑁆𑀲𑀢𑀺,_  
_𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲 𑀅𑀦𑀗𑁆𑀕𑀡𑀲𑁆𑀲;_  
_𑀢𑀫𑁂𑀯 𑀩𑀸𑀮𑀁 𑀧𑀘𑁆𑀘𑁂𑀢𑀺 𑀧𑀸𑀧𑀁,_  
_𑀲𑀼𑀔𑀼𑀫𑁄 𑀭𑀚𑁄 𑀧𑀝𑀺𑀯𑀸𑀢𑀁𑀯 𑀔𑀺𑀢𑁆𑀢𑁄𑁇_
