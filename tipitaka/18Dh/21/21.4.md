# 𑁨𑁧𑁇𑁪 𑀮𑀓𑀼𑀡𑁆𑀟𑀓𑀪𑀤𑁆𑀤𑀺𑀬𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕

* 𑀮𑀓𑀼𑀡𑁆𑀟𑀓𑀪𑀤𑁆𑀤𑀺𑀬𑀯𑀢𑁆𑀣𑀼

(294.)

317\. _𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀭𑀸𑀚𑀸𑀦𑁄 𑀤𑁆𑀯𑁂 𑀘 𑀔𑀢𑁆𑀢𑀺𑀬𑁂;_  
_𑀭𑀝𑁆𑀞𑀁 𑀲𑀸𑀦𑀼𑀘𑀭𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀻𑀖𑁄 𑀬𑀸𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


(295.)

318\. _𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀭𑀸𑀚𑀸𑀦𑁄 𑀤𑁆𑀯𑁂 𑀘 𑀲𑁄𑀢𑁆𑀣𑀺𑀬𑁂;_  
_𑀯𑁂𑀬𑀕𑁆𑀖𑀧𑀜𑁆𑀘𑀫𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀻𑀖𑁄 𑀬𑀸𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_
