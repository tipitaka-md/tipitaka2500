# 𑁨𑁧𑁇𑁯 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀧𑀓𑀺𑀡𑁆𑀡𑀓𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑀺𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(305.)

328\. _𑀏𑀓𑀸𑀲𑀦𑀁 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀁,_  
_𑀏𑀓𑁄 𑀘𑀭𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑁄;_  
_𑀏𑀓𑁄 𑀤𑀫𑀬𑀫𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀯𑀦𑀦𑁆𑀢𑁂 𑀭𑀫𑀺𑀢𑁄 𑀲𑀺𑀬𑀸𑁇_  


---
