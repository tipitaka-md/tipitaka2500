# 𑁧𑁪𑁇𑁫 𑀅𑀦𑀪𑀺𑀭𑀢𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀪𑀺𑀭𑀢𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(186.)

200\. _𑀦 𑀓𑀳𑀸𑀧𑀡𑀯𑀲𑁆𑀲𑁂𑀦,_  
_𑀢𑀺𑀢𑁆𑀢𑀺 𑀓𑀸𑀫𑁂𑀲𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀸𑀤𑀸 𑀤𑀼𑀔𑀸 𑀓𑀸𑀫𑀸,_  
_𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


(187.)

201\. _𑀅𑀧𑀺 𑀤𑀺𑀩𑁆𑀩𑁂𑀲𑀼 𑀓𑀸𑀫𑁂𑀲𑀼,_  
_𑀭𑀢𑀺𑀁 𑀲𑁄 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑀭𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑁄𑁇_
