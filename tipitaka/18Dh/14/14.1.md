# 𑁧𑁪𑁇𑁧 𑀫𑀸𑀭𑀥𑀻𑀢𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀫𑀸𑀭𑀥𑀻𑀢𑀭𑀯𑀢𑁆𑀣𑀼

(179.)

193\. _𑀬𑀲𑁆𑀲 𑀚𑀺𑀢𑀁 𑀦𑀸𑀯𑀚𑀻𑀬𑀢𑀺,_  
_𑀚𑀺𑀢𑀁 𑀬𑀲𑁆𑀲 𑀦𑁄𑀬𑀸𑀢𑀺 𑀓𑁄𑀘𑀺 𑀮𑁄𑀓𑁂;_  
_𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀫𑀦𑀦𑁆𑀢𑀕𑁄𑀘𑀭𑀁,_  
_𑀅𑀧𑀤𑀁 𑀓𑁂𑀦 𑀧𑀤𑁂𑀦 𑀦𑁂𑀲𑁆𑀲𑀣𑁇_  


(180.)

194\. _𑀬𑀲𑁆𑀲 𑀚𑀸𑀮𑀺𑀦𑀻 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀢𑀡𑁆𑀳𑀸 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺 𑀦𑁂𑀢𑀯𑁂;_  
_𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀫𑀦𑀦𑁆𑀢𑀕𑁄𑀘𑀭𑀁,_  
_𑀅𑀧𑀤𑀁 𑀓𑁂𑀦 𑀧𑀤𑁂𑀦 𑀦𑁂𑀲𑁆𑀲𑀣𑁇_
