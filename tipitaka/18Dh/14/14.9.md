# 𑁧𑁪𑁇𑁯 𑀓𑀲𑁆𑀲𑀧𑀤𑀲𑀩𑀮𑀲𑁆𑀲𑀲𑀼𑀯𑀡𑁆𑀡𑀘𑁂𑀢𑀺𑀬𑀯𑀢𑁆𑀣𑀼

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀓𑀲𑁆𑀲𑀧𑀤𑀲𑀩𑀮𑀲𑁆𑀲𑀲𑀼𑀯𑀡𑁆𑀡𑀘𑁂𑀢𑀺𑀬𑀯𑀢𑁆𑀣𑀼

(195.)

209\. _𑀧𑀽𑀚𑀸𑀭𑀳𑁂 𑀧𑀽𑀚𑀬𑀢𑁄,_  
_𑀩𑀼𑀤𑁆𑀥𑁂 𑀬𑀤𑀺 𑀯 𑀲𑀸𑀯𑀓𑁂;_  
_𑀧𑀧𑀜𑁆𑀘𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂,_  
_𑀢𑀺𑀡𑁆𑀡𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁆𑀤𑀯𑁂𑁇_  


(196.)

210\. _𑀢𑁂 𑀢𑀸𑀤𑀺𑀲𑁂 𑀧𑀽𑀚𑀬𑀢𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑁂;_  
_𑀦 𑀲𑀓𑁆𑀓𑀸 𑀧𑀼𑀜𑁆𑀜𑀁 𑀲𑀗𑁆𑀔𑀸𑀢𑀼𑀁,_  
_𑀇𑀫𑁂𑀢𑁆𑀢𑀫𑀧𑀺 𑀓𑁂𑀦𑀘𑀺𑁇_
