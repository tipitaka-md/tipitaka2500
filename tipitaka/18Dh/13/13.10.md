# 𑁧𑁩𑁇𑁧𑁦 𑀅𑀲𑀤𑀺𑀲𑀤𑀸𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀮𑁄𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀲𑀤𑀺𑀲𑀤𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(177.)

190\. _𑀦 𑀯𑁂 𑀓𑀤𑀭𑀺𑀬𑀸 𑀤𑁂𑀯𑀮𑁄𑀓𑀁 𑀯𑀚𑀦𑁆𑀢𑀺,_  
_𑀩𑀸𑀮𑀸 𑀳𑀯𑁂 𑀦𑀧𑁆𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺 𑀤𑀸𑀦𑀁;_  
_𑀥𑀻𑀭𑁄 𑀘 𑀤𑀸𑀦𑀁 𑀅𑀦𑀼𑀫𑁄𑀤𑀫𑀸𑀦𑁄,_  
_𑀢𑁂𑀦𑁂𑀯 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀔𑀻 𑀧𑀭𑀢𑁆𑀣𑁇_
