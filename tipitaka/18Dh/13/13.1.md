# 𑁧𑁩𑁇𑁧 𑀤𑀳𑀭𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀮𑁄𑀓𑀯𑀕𑁆𑀕

### 𑀤𑀳𑀭𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(167.)

180\. _𑀳𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀲𑁂𑀯𑁂𑀬𑁆𑀬,_  
_𑀧𑀫𑀸𑀤𑁂𑀦 𑀦 𑀲𑀁𑀯𑀲𑁂;_  
_𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀦 𑀲𑁂𑀯𑁂𑀬𑁆𑀬,_  
_𑀦 𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀯𑀟𑁆𑀠𑀦𑁄𑁇_
