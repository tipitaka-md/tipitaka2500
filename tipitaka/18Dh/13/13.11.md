### 𑀮𑁄𑀓𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑀸𑀮𑀯𑀢𑁆𑀣𑀼

(178.)

191. _𑀧𑀣𑀩𑁆𑀬𑀸 𑀏𑀓𑀭𑀚𑁆𑀚𑁂𑀦,_  
_𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀕𑀫𑀦𑁂𑀦 𑀯𑀸;_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀸𑀥𑀺𑀧𑀘𑁆𑀘𑁂𑀦,_  
_𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀯𑀭𑀁𑁇_
