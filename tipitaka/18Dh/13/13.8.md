# 𑁧𑁩𑁇𑁮 𑀢𑀺𑀁𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

* 𑀢𑀺𑀁𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(175.)

188\. _𑀳𑀁𑀲𑀸𑀤𑀺𑀘𑁆𑀘𑀧𑀣𑁂 𑀬𑀦𑁆𑀢𑀺,_  
_𑀆𑀓𑀸𑀲𑁂 𑀬𑀦𑁆𑀢𑀺 𑀇𑀤𑁆𑀥𑀺𑀬𑀸;_  
_𑀦𑀻𑀬𑀦𑁆𑀢𑀺 𑀥𑀻𑀭𑀸 𑀮𑁄𑀓𑀫𑁆𑀳𑀸,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀁𑁇_
