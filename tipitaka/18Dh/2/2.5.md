# 𑁨𑁇𑁫 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕

* 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(28.)

30\. _𑀧𑀫𑀸𑀤𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦,_  
_𑀬𑀤𑀸 𑀦𑀼𑀤𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀅𑀲𑁄𑀓𑁄 𑀲𑁄𑀓𑀺𑀦𑀺𑀁 𑀧𑀚𑀁;_  
_𑀧𑀩𑁆𑀩𑀢𑀝𑁆𑀞𑁄𑀯 𑀪𑀽𑀫𑀝𑁆𑀞𑁂,_  
_𑀥𑀻𑀭𑁄 𑀩𑀸𑀮𑁂 𑀅𑀯𑁂𑀓𑁆𑀔𑀢𑀺𑁇_
