# 𑁨𑁇𑁯 𑀦𑀺𑀕𑀫𑀯𑀸𑀲𑀺𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀕𑀫𑀯𑀸𑀲𑀺𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(32.)

34\. _𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀭𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀫𑀸𑀤𑁂 𑀪𑀬𑀤𑀲𑁆𑀲𑀺 𑀯𑀸;_
