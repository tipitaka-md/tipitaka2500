# 𑁨𑁇𑁪 𑀩𑀸𑀮𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀲𑀁𑀖𑀼𑀝𑁆𑀞𑀯𑀢𑁆𑀣𑀼

### 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀮𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀲𑀁𑀖𑀼𑀝𑁆𑀞𑀯𑀢𑁆𑀣𑀼

(26.)

28. _𑀧𑀫𑀸𑀤𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀩𑀸𑀮𑀸 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑀺𑀦𑁄 𑀚𑀦𑀸;_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀜𑁆𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀥𑀦𑀁 𑀲𑁂𑀝𑁆𑀞𑀁𑀯 𑀭𑀓𑁆𑀔𑀢𑀺𑁇_  


(27.)

29. _𑀫𑀸 𑀧𑀫𑀸𑀤𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑁂𑀣,_  
_𑀫𑀸 𑀓𑀸𑀫𑀭𑀢𑀺𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀳𑀺 𑀛𑀸𑀬𑀦𑁆𑀢𑁄,_  
_𑀧𑀧𑁆𑀧𑁄𑀢𑀺 𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀼𑀔𑀁𑁇_
