# 𑁧𑁮𑁇𑁬 𑀘𑀽𑀍𑀅𑀲𑀸𑀭𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀮𑀯𑀕𑁆𑀕

### 𑀘𑀽𑀍𑀅𑀲𑀸𑀭𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀢𑁆𑀣𑀼

(244.)

264. _𑀲𑀼𑀚𑀻𑀯𑀁 𑀅𑀳𑀺𑀭𑀺𑀓𑁂𑀦,_  
_𑀓𑀸𑀓𑀲𑀽𑀭𑁂𑀦 𑀥𑀁𑀲𑀺𑀦𑀸;_  
_𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀺𑀦𑀸 𑀧𑀕𑀩𑁆𑀪𑁂𑀦,_  
_𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀁𑁇_  


(245.)

265. _𑀳𑀺𑀭𑀻𑀫𑀢𑀸 𑀘 𑀤𑀼𑀚𑁆𑀚𑀻𑀯𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀲𑀼𑀘𑀺𑀕𑀯𑁂𑀲𑀺𑀦𑀸;_  
_𑀅𑀮𑀻𑀦𑁂𑀦𑀸𑀧𑁆𑀧𑀕𑀩𑁆𑀪𑁂𑀦,_  
_𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁂𑀦 𑀧𑀲𑁆𑀲𑀢𑀸𑁇_
