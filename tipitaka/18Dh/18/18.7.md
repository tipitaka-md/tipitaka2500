# 𑁧𑁮𑁇𑁭 𑀧𑀜𑁆𑀘𑁅𑀧𑀸𑀲𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀮𑀯𑀕𑁆𑀕

### 𑀧𑀜𑁆𑀘𑁅𑀧𑀸𑀲𑀓𑀯𑀢𑁆𑀣𑀼

(246.)

266. _𑀬𑁄 𑀧𑀸𑀡𑀫𑀢𑀺𑀧𑀸𑀢𑁂𑀢𑀺,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀜𑁆𑀘 𑀪𑀸𑀲𑀢𑀺;_  
_𑀮𑁄𑀓𑁂 𑀅𑀤𑀺𑀦𑁆𑀦𑀫𑀸𑀤𑀺𑀬𑀢𑀺,_  
_𑀧𑀭𑀤𑀸𑀭𑀜𑁆𑀘 𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(247.)

267. _𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀧𑀸𑀦𑀜𑁆𑀘,_  
_𑀬𑁄 𑀦𑀭𑁄 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀢𑀺;_  
_𑀇𑀥𑁂𑀯 𑀫𑁂𑀲𑁄 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑀽𑀮𑀁 𑀔𑀡𑀢𑀺 𑀅𑀢𑁆𑀢𑀦𑁄𑁇_  


(248.)

268. _𑀏𑀯𑀁 𑀪𑁄 𑀧𑀼𑀭𑀺𑀲 𑀚𑀸𑀦𑀸𑀳𑀺,_  
_𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑀜𑁆𑀜𑀢𑀸;_  
_𑀫𑀸 𑀢𑀁 𑀮𑁄𑀪𑁄 𑀅𑀥𑀫𑁆𑀫𑁄 𑀘,_  
_𑀘𑀺𑀭𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀭𑀦𑁆𑀥𑀬𑀼𑀁𑁇_
