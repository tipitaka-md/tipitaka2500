# 𑁧𑁮𑁇𑁮 𑀢𑀺𑀲𑁆𑀲𑀤𑀳𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀮𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀲𑁆𑀲𑀤𑀳𑀭𑀯𑀢𑁆𑀣𑀼

(249.)

269. _𑀤𑀤𑀸𑀢𑀺 𑀯𑁂 𑀬𑀣𑀸𑀲𑀤𑁆𑀥𑀁,_  
_𑀬𑀣𑀸𑀧𑀲𑀸𑀤𑀦𑀁 𑀚𑀦𑁄;_  
_𑀢𑀢𑁆𑀣 𑀬𑁄 𑀘 𑀫𑀗𑁆𑀓𑀼 𑀪𑀯𑀢𑀺,_  
_𑀧𑀭𑁂𑀲𑀁 𑀧𑀸𑀦𑀪𑁄𑀚𑀦𑁂;_  
_𑀦 𑀲𑁄 𑀤𑀺𑀯𑀸 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀸,_  
_𑀲𑀫𑀸𑀥𑀺𑀫𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(250.)

270. _𑀬𑀲𑁆𑀲 𑀘𑁂𑀢𑀁 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁,_  
_𑀫𑀽𑀮𑀖𑀘𑁆𑀘𑀁 𑀲𑀫𑀽𑀳𑀢𑀁;_  
_𑀲 𑀯𑁂 𑀤𑀺𑀯𑀸 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀸,_  
_𑀲𑀫𑀸𑀥𑀺𑀫𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_
