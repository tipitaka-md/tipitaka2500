# 𑁧𑁮𑁇𑁧𑁦 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁂𑀝𑁆𑀞𑀺𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀮𑀯𑀕𑁆𑀕

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁂𑀝𑁆𑀞𑀺𑀯𑀢𑁆𑀣𑀼

(252.)

272\. _𑀲𑀼𑀤𑀲𑁆𑀲𑀁 𑀯𑀚𑁆𑀚𑀫𑀜𑁆𑀜𑁂𑀲𑀁,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀦 𑀤𑀼𑀤𑁆𑀤𑀲𑀁;_  
_𑀧𑀭𑁂𑀲𑀁 𑀳𑀺 𑀲𑁄 𑀯𑀚𑁆𑀚𑀸𑀦𑀺,_  
_𑀑𑀧𑀼𑀦𑀸𑀢𑀺 𑀬𑀣𑀸 𑀪𑀼𑀲𑀁;_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀦 𑀙𑀸𑀤𑁂𑀢𑀺,_  
_𑀓𑀮𑀺𑀁𑀯 𑀓𑀺𑀢𑀯𑀸 𑀲𑀞𑁄𑁇_
