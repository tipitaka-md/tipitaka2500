# 𑁧𑁮𑁇𑁫 𑀅𑀜𑁆𑀜𑀢𑀭𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀮𑀯𑀕𑁆𑀕

### 𑀅𑀜𑁆𑀜𑀢𑀭𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(242.)

262. _𑀫𑀮𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁,_  
_𑀫𑀘𑁆𑀙𑁂𑀭𑀁 𑀤𑀤𑀢𑁄 𑀫𑀮𑀁;_  
_𑀫𑀮𑀸 𑀯𑁂 𑀧𑀸𑀧𑀓𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀫𑁆𑀳𑀺 𑀘𑁇_  


(243.)

263. _𑀢𑀢𑁄 𑀫𑀮𑀸 𑀫𑀮𑀢𑀭𑀁,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀭𑀫𑀁 𑀫𑀮𑀁;_  
_𑀏𑀢𑀁 𑀫𑀮𑀁 𑀧𑀳𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀫𑁆𑀫𑀮𑀸 𑀳𑁄𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄𑁇_
