# 𑁭𑁇𑁪 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(93.)

100. _𑀬𑀲𑁆𑀲𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀆𑀳𑀸𑀭𑁂 𑀘 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄;_  
_𑀲𑀼𑀜𑁆𑀜𑀢𑁄 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀘,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀬𑀲𑁆𑀲 𑀕𑁄𑀘𑀭𑁄;_  
_𑀆𑀓𑀸𑀲𑁂𑀯 𑀲𑀓𑀼𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀧𑀤𑀁 𑀢𑀲𑁆𑀲 𑀤𑀼𑀭𑀦𑁆𑀦𑀬𑀁𑁇_
