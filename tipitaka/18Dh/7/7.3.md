# 𑁭𑁇𑁩 𑀩𑁂𑀮𑀝𑁆𑀞𑀲𑀻𑀲𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀕𑁆𑀕

* 𑀩𑁂𑀮𑀝𑁆𑀞𑀲𑀻𑀲𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(92.)

99\. _𑀬𑁂𑀲𑀁 𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀬𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀢𑀪𑁄𑀚𑀦𑀸;_  
_𑀲𑀼𑀜𑁆𑀜𑀢𑁄 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀘,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄 𑀬𑁂𑀲𑀁 𑀕𑁄𑀘𑀭𑁄;_  
_𑀆𑀓𑀸𑀲𑁂𑀯 𑀲𑀓𑀼𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀕𑀢𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀭𑀦𑁆𑀦𑀬𑀸𑁇_
