# 𑁭𑁇𑁨 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀢𑁆𑀣𑁂𑀭𑀯𑀢𑁆𑀣𑀼

(91.)

98\. _𑀉𑀬𑁆𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀲𑀢𑀻𑀫𑀦𑁆𑀢𑁄,_  
_𑀦 𑀦𑀺𑀓𑁂𑀢𑁂 𑀭𑀫𑀦𑁆𑀢𑀺 𑀢𑁂;_  
_𑀳𑀁𑀲𑀸𑀯 𑀧𑀮𑁆𑀮𑀮𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀑𑀓𑀫𑁄𑀓𑀁 𑀚𑀳𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_
