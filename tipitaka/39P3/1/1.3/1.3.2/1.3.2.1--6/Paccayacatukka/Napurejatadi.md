# 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀸𑀤𑀺

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀢𑀺𑀓

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀯𑀺𑀧𑀸𑀓𑀢𑁆𑀢𑀺𑀓

### 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀧𑀤

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀸𑀤𑀺

(125.)

218\. 𑀳𑁂𑀢𑀼𑀁 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑁂𑀢𑀼 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀯…  𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀯…  𑀦𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀯𑁇

(126.)

219\. 𑀳𑁂𑀢𑀼𑀁 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)

220\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)

221\. 𑀳𑁂𑀢𑀼𑀁 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀦𑀳𑁂𑀢𑀼𑀁 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(127.)

222\. 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯, 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀦𑀯, 𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀦𑀯, 𑀦𑀆𑀲𑁂𑀯𑀦𑁂 𑀦𑀯, 𑀦𑀓𑀫𑁆𑀫𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀯𑀺𑀧𑀸𑀓𑁂 𑀦𑀯, 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁, 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀁𑁇)

223\. 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁, 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀁𑁇)

224\. 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁, 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀁𑁇)

225\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁄𑀧𑀺 𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭𑁄𑀧𑀺 𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀸𑀭𑁄𑀧𑀺 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭𑁄𑀧𑀺 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭𑁄𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇)
