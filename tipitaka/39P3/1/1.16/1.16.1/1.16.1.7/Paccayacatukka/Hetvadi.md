# 𑀳𑁂𑀢𑁆𑀯𑀸𑀤𑀺

* 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀢𑀺𑀓

* 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

* 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀧𑀤

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀳𑁂𑀢𑁆𑀯𑀸𑀤𑀺

(457.)

1000\. 𑀳𑁂𑀢𑀼 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

1001\. 𑀳𑁂𑀢𑀼 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺…  𑀢𑀻𑀡𑀺𑁇

1002\. 𑀦𑀳𑁂𑀢𑀼 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸𑀥𑀺𑀧𑀢𑀺…  𑀢𑀻𑀡𑀺𑁇

1003\. 𑀳𑁂𑀢𑀼 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(458.)

1004\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀙, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀦𑀯, 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀦𑀯, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀦𑀯, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀦𑀯, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀦𑀯, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀦𑀯, 𑀆𑀲𑁂𑀯𑀦𑁂 𑀦𑀯, 𑀓𑀫𑁆𑀫𑁂 𑀢𑀻𑀡𑀺, 𑀆𑀳𑀸𑀭𑁂 𑀢𑀻𑀡𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀦𑀯, 𑀛𑀸𑀦𑁂 𑀢𑀻𑀡𑀺, 𑀫𑀕𑁆𑀕𑁂 𑀦𑀯, 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀦𑀯, 𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀦𑀯, 𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀦𑀯, 𑀯𑀺𑀕𑀢𑁂 𑀦𑀯, 𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)
