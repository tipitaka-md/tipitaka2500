# 𑀳𑁂𑀢𑁆𑀯𑀸𑀤𑀺

### 𑀘𑀽𑀍𑀅𑀦𑁆𑀢𑀭𑀤𑀼𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀤𑀼𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀧𑀤

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀳𑁂𑀢𑁆𑀯𑀸𑀤𑀺

(11.)

1485\. 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

1486\. 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

1487\. 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

1488\. 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

1489\. 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄𑁇 (𑁧) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1490\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀏𑀓𑀁, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀤𑁆𑀯𑁂, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀏𑀓𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀏𑀓𑀁, (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁), 𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)
