# 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀤𑀼𑀓𑀸𑀤𑀺, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

(55.)

2499. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑁆𑀧𑀺 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀁𑁇)

(56.)

2500. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀫𑁆𑀧𑀺 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑁂 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑁂 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀦𑀛𑀸𑀦𑁂 𑀙, 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑁂 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀛𑀸𑀦𑁂 𑀙, 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁂 𑀦𑀯𑁇)
