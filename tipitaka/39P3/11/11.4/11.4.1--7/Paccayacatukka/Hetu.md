# 𑀳𑁂𑀢𑀼

* 𑀉𑀧𑀸𑀤𑀸𑀦𑀕𑁄𑀘𑁆𑀙𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

* 𑀉𑀧𑀸𑀤𑀸𑀦𑁅𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀤𑀼𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀳𑁂𑀢𑀼

(21.)

2279\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2280\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2281\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁂𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

(22.)

2282\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

2283\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

2284\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀘𑁂𑀯 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(23.)

2285\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀇𑀫𑀁 𑀕𑀫𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀦𑀤𑀼𑀓𑀅𑀓𑀼𑀲𑀮𑀲𑀤𑀺𑀲𑀁𑁇 𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑀫𑁆𑀧𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)

(24.)

2286\. 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀉𑀧𑀸𑀤𑀸𑀦𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(25.)

2287\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2288\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁂𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)
