# 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

* 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀤𑀼𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

(20.)

2346\. 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2347\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2348\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁂𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

(21.)

2349\. 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

2350\. 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

2351\. 𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀘𑁂𑀯 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(22.)

2352\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁, 𑀓𑀺𑀮𑁂𑀲𑀤𑀼𑀓𑀅𑀓𑀼𑀲𑀮𑀲𑀤𑀺𑀲𑀁𑁇 𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁂𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)

(23.)

2353\. 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2354\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

2355\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁂𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)
