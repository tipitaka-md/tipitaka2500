# 𑁭𑁇𑁪𑁇𑁯 𑀯𑀭𑀸𑀳𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀉𑀧𑀘𑀺𑀓𑀸𑀯𑀕𑁆𑀕

### 𑀯𑀭𑀸𑀳𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(263.)

1873. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀯𑀭𑀸𑀳𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀭𑀸𑀳𑁄 𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀓𑀞𑀺𑀢𑁂 𑀕𑀺𑀫𑁆𑀳𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂 𑀉𑀤𑀓𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀤𑁄𑀲𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂 𑀆𑀮𑀼𑀍𑀇𑀢𑀔𑀮𑀺𑀢𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑁂 𑀲𑀻𑀢𑀮𑀸𑀫𑀢𑀧𑀡𑀻𑀢𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀭𑀸𑀳𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1874. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀭𑀸𑀳𑁄 𑀘𑀺𑀓𑁆𑀔𑀮𑁆𑀮𑀫𑀼𑀤𑀓𑀫𑀼𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑀺𑀓𑀸𑀬 𑀧𑀣𑀯𑀺𑀁 𑀔𑀡𑀺𑀢𑁆𑀯𑀸 𑀤𑁄𑀡𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀤𑁄𑀡𑀺𑀓𑀸𑀬 𑀲𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀫𑀸𑀦𑀲𑁂 𑀓𑀸𑀬𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀦𑁆𑀢𑀭𑀕𑀢𑁂𑀦 𑀲𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀭𑀸𑀳𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑁄𑀮𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁂𑀦—

1875. _‘𑀓𑀸𑀬𑁂 𑀲𑀪𑀸𑀯𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑁄;_  
_𑀏𑀓𑀸𑀓𑀺𑀬𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄,_  
_𑀲𑁂𑀢𑀺 𑀆𑀭𑀫𑁆𑀫𑀡𑀦𑁆𑀢𑀭𑁂’”𑀢𑀺𑁇_  


---

1876. 𑀯𑀭𑀸𑀳𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀦𑀯𑀫𑁄𑁇
