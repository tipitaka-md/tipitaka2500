# 𑁭𑁇𑁪𑁇𑁮 𑀕𑁄𑀭𑀽𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀉𑀧𑀘𑀺𑀓𑀸𑀯𑀕𑁆𑀕

* 𑀕𑁄𑀭𑀽𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(262.)

1866\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀕𑁄𑀭𑀽𑀧𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑁄 𑀲𑀓𑀁 𑀕𑁂𑀳𑀁 𑀦 𑀯𑀺𑀚𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀓𑁄 𑀓𑀸𑀬𑁄 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑁄 ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀼𑀘𑁆𑀙𑀸𑀤𑀦𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀦𑀪𑁂𑀤𑀦𑀯𑀺𑀓𑀺𑀭𑀡𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑀦𑀥𑀫𑁆𑀫𑁄 𑀅𑀬𑀁 𑀓𑀸𑀬𑁄’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1867\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑁄 𑀆𑀤𑀺𑀦𑁆𑀦𑀥𑀼𑀭𑁄 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀥𑀼𑀭𑀁 𑀯𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀤𑀺𑀦𑁆𑀦𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀬𑀸𑀯 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀆𑀧𑀸𑀡𑀓𑁄𑀝𑀺𑀓𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1868\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑁄 𑀙𑀦𑁆𑀤𑁂𑀦 𑀖𑀸𑀬𑀫𑀸𑀦𑁄 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀺𑀯𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺 𑀙𑀦𑁆𑀤𑁂𑀦 𑀧𑁂𑀫𑁂𑀦 𑀧𑀲𑀸𑀤𑁂𑀦 𑀖𑀸𑀬𑀫𑀸𑀦𑁂𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1869\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑁄 𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺 𑀯𑀸𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀯𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀣𑁂𑀭𑀦𑀯𑀫𑀚𑁆𑀛𑀺𑀫𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀫𑁆𑀧𑀺 𑀕𑀺𑀳𑀺𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲𑀸𑀧𑀺 𑀑𑀯𑀸𑀤𑀸𑀦𑀼𑀲𑀸𑀲𑀦𑀻 𑀲𑀺𑀭𑀲𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀭𑀽𑀧𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1870\. _‘𑀢𑀤𑀳𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑁄;_  
_𑀲𑁄𑀧𑀺 𑀫𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀺 𑀫𑀢𑁆𑀣𑀓𑁂𑁇_  


1871\. _𑀢𑀺𑀩𑁆𑀩𑀁 𑀙𑀦𑁆𑀤𑀜𑁆𑀘 𑀧𑁂𑀫𑀜𑁆𑀘,_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀧𑁂;_  
_𑀞𑀧𑁂𑀬𑁆𑀬𑀸𑀘𑀭𑀺𑀬𑀝𑁆𑀞𑀸𑀦𑁂,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀦𑀁 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀦𑁆’”𑀢𑀺𑁇_  


---

1872\. 𑀕𑁄𑀭𑀽𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
