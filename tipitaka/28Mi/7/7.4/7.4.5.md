# 𑁭𑁇𑁪𑁇𑁫 𑀦𑀓𑀼𑀮𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀉𑀧𑀘𑀺𑀓𑀸𑀯𑀕𑁆𑀕

* 𑀦𑀓𑀼𑀮𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(259.)

1849\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀦𑀓𑀼𑀮𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀓𑀼𑀮𑁄 𑀉𑀭𑀕𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀪𑁂𑀲𑀚𑁆𑀚𑁂𑀦 𑀓𑀸𑀬𑀁 𑀧𑀭𑀺𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀉𑀭𑀕𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀢𑀺 𑀕𑀳𑁂𑀢𑀼𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑁄𑀥𑀸𑀖𑀸𑀢𑀩𑀳𑀼𑀮𑀁 𑀓𑀮𑀳𑀯𑀺𑀕𑁆𑀕𑀳𑀯𑀺𑀯𑀸𑀤𑀯𑀺𑀭𑁄𑀥𑀸𑀪𑀺𑀪𑀽𑀢𑀁 𑀮𑁄𑀓𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑀦 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑁂𑀲𑀚𑁆𑀚𑁂𑀦 𑀫𑀸𑀦𑀲𑀁 𑀅𑀦𑀼𑀮𑀺𑀫𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀓𑀼𑀮𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1850\. _‘𑀢𑀲𑁆𑀫𑀸 𑀲𑀓𑀁 𑀧𑀭𑁂𑀲𑀫𑁆𑀧𑀺,_  
_𑀓𑀸𑀢𑀩𑁆𑀩𑀸 𑀫𑁂𑀢𑁆𑀢𑀪𑀸𑀯𑀦𑀸;_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀨𑀭𑀺𑀢𑀩𑁆𑀩𑀁,_  
_𑀏𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦 𑀲𑀸𑀲𑀦𑀦𑁆’”𑀢𑀺𑁇_  


---

1851\. 𑀦𑀓𑀼𑀮𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇
