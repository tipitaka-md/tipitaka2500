# 𑁭𑁇𑁪𑁇𑁨 𑀩𑀺𑀍𑀆𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀉𑀧𑀘𑀺𑀓𑀸𑀯𑀕𑁆𑀕

* 𑀩𑀺𑀍𑀆𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(256.)

1839\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀩𑀺𑀍𑀆𑀭𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀺𑀍𑀆𑀭𑁄 𑀕𑀼𑀳𑀸𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀲𑀺𑀭𑀕𑀢𑁄𑀧𑀺 𑀳𑀫𑁆𑀫𑀺𑀬𑀦𑁆𑀢𑀭𑀕𑀢𑁄𑀧𑀺 𑀉𑀦𑁆𑀤𑀽𑀭𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀕𑀸𑀫𑀕𑀢𑁂𑀦𑀸𑀧𑀺 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁂𑀦𑀸𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁂𑀦𑀸𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁂𑀦𑀸𑀧𑀺 𑀲𑀢𑀢𑀁 𑀲𑀫𑀺𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁂𑀦 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀪𑁄𑀚𑀦𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀺𑀍𑀆𑀭𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1840\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀺𑀍𑀆𑀭𑁄 𑀆𑀲𑀦𑁆𑀦𑁂𑀬𑁂𑀯 𑀕𑁄𑀘𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀇𑀫𑁂𑀲𑀼𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀺𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀇𑀢𑀺 𑀭𑀽𑀧𑀁 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀬 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀺𑀍𑀆𑀭𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1841\. _‘𑀦 𑀇𑀢𑁄 𑀤𑀽𑀭𑁂 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁,_  
_𑀪𑀯𑀕𑁆𑀕𑀁 𑀓𑀺𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀫𑁆𑀳𑀺 𑀯𑁄𑀳𑀸𑀭𑁂,_  
_𑀲𑀓𑁂 𑀓𑀸𑀬𑀫𑁆𑀳𑀺 𑀯𑀺𑀦𑁆𑀤𑀣𑀸’”𑀢𑀺𑁇_  


---

1842\. 𑀩𑀺𑀍𑀆𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
