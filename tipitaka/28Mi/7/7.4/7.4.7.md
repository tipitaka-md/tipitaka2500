# 𑁭𑁇𑁪𑁇𑁭 𑀫𑀺𑀕𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀉𑀧𑀘𑀺𑀓𑀸𑀯𑀕𑁆𑀕

### 𑀫𑀺𑀕𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(261.)

1859\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀫𑀺𑀕𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀕𑁄 𑀤𑀺𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑁂 𑀘𑀭𑀢𑀺, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀤𑀺𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀕𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1860\. 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀧𑀭𑀺𑀬𑀸𑀬𑁂—

1861\. ‘𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀬𑀸 𑀢𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀻𑀢𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸 𑀅𑀦𑁆𑀢𑀭𑀝𑁆𑀞𑀓𑀸 𑀳𑀺𑀫𑀧𑀸𑀢𑀲𑀫𑀬𑀸, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀯𑀺𑀳𑀭𑀸𑀫𑀺, 𑀤𑀺𑀯𑀸 𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀕𑀺𑀫𑁆𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀫𑀸𑀲𑁂 𑀤𑀺𑀯𑀸 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀯𑀺𑀳𑀭𑀸𑀫𑀺, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂’𑀢𑀺𑁇

1862\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀕𑁄 𑀲𑀢𑁆𑀢𑀺𑀫𑁆𑀳𑀺 𑀯𑀸 𑀲𑀭𑁂 𑀯𑀸 𑀑𑀧𑀢𑀦𑁆𑀢𑁂 𑀯𑀜𑁆𑀘𑁂𑀢𑀺 𑀧𑀮𑀸𑀬𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀫𑀼𑀧𑀦𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀺𑀮𑁂𑀲𑁂𑀲𑀼 𑀑𑀧𑀢𑀦𑁆𑀢𑁂𑀲𑀼 𑀯𑀜𑁆𑀘𑀬𑀺𑀢𑀩𑁆𑀩𑀁 𑀧𑀮𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀫𑀼𑀧𑀦𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀕𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1863\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀕𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑁂𑀦 𑀯𑀸 𑀢𑁂𑀦 𑀯𑀸 𑀧𑀮𑀸𑀬𑀢𑀺 ‘𑀫𑀸 𑀫𑀁 𑀢𑁂 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀽’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀪𑀡𑁆𑀟𑀦𑀓𑀮𑀳𑀯𑀺𑀕𑁆𑀕𑀳𑀯𑀺𑀯𑀸𑀤𑀲𑀻𑀮𑁂 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁂 𑀓𑀼𑀲𑀻𑀢𑁂 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑁂𑀦 𑀯𑀸 𑀢𑁂𑀦 𑀯𑀸 𑀧𑀮𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀫𑀸 𑀫𑀁 𑀢𑁂 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼, 𑀅𑀳𑀜𑁆𑀘 𑀢𑁂 𑀫𑀸 𑀅𑀤𑁆𑀤𑀲𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀕𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1864\. _‘𑀫𑀸 𑀫𑁂 𑀓𑀤𑀸𑀘𑀺 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄,_  
_𑀓𑀼𑀲𑀻𑀢𑁄 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑁄;_  
_𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁄 𑀅𑀦𑀸𑀘𑀸𑀭𑁄,_  
_𑀲𑀫𑁆𑀫𑀢𑁄 𑀅𑀳𑀼 𑀓𑀢𑁆𑀣𑀘𑀻’”𑀢𑀺𑁇_  


---

1865\. 𑀫𑀺𑀕𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇
