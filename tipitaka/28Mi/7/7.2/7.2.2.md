# 𑁭𑁇𑁨𑁇𑁨 𑀧𑀤𑀼𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀧𑀤𑀼𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(236.)

1711\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀤𑀼𑀫𑀁 𑀉𑀤𑀓𑁂 𑀚𑀸𑀢𑀁 𑀉𑀤𑀓𑁂 𑀲𑀁𑀯𑀤𑁆𑀥𑀁 𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑀁 𑀉𑀤𑀓𑁂𑀦; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀼𑀮𑁂 𑀕𑀡𑁂 𑀮𑀸𑀪𑁂 𑀬𑀲𑁂 𑀲𑀓𑁆𑀓𑀸𑀭𑁂 𑀲𑀫𑁆𑀫𑀸𑀦𑀦𑀸𑀬 𑀧𑀭𑀺𑀪𑁄𑀕𑀧𑀘𑁆𑀘𑀬𑁂𑀲𑀼 𑀘 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1712\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀤𑀼𑀫𑀁 𑀉𑀤𑀓𑀸 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑁆𑀫 𑀞𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀁 𑀅𑀪𑀺𑀪𑀯𑀺𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑁆𑀫 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀥𑀫𑁆𑀫𑁂 𑀞𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1713\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀤𑀼𑀫𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀅𑀦𑀺𑀮𑁂𑀦 𑀏𑀭𑀺𑀢𑀁 𑀘𑀮𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀲𑀼𑀧𑀺 𑀓𑀺𑀮𑁂𑀲𑁂𑀲𑀼 𑀲𑀁𑀬𑀫𑁄 𑀓𑀭𑀡𑀻𑀬𑁄, 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 ‘𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀽’”𑀢𑀺𑁇

---
