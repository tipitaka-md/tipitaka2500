# 𑁭𑁇𑁨𑁇𑁫 𑀦𑀸𑀯𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀦𑀸𑀯𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(239.)

1725\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀦𑀸𑀯𑀸𑀬 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸 𑀩𑀳𑀼𑀯𑀺𑀥𑀤𑀸𑀭𑀼𑀲𑀗𑁆𑀖𑀸𑀝𑀲𑀫𑀯𑀸𑀬𑁂𑀦 𑀩𑀳𑀼𑀫𑁆𑀧𑀺 𑀚𑀦𑀁 𑀢𑀸𑀭𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀘𑀸𑀭𑀲𑀻𑀮𑀕𑀼𑀡𑀯𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀯𑀢𑁆𑀢𑀩𑀳𑀼𑀯𑀺𑀥𑀥𑀫𑁆𑀫𑀲𑀗𑁆𑀖𑀸𑀝𑀲𑀫𑀯𑀸𑀬𑁂𑀦 𑀲𑀤𑁂𑀯𑀓𑁄 𑀮𑁄𑀓𑁄 𑀢𑀸𑀭𑀬𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸𑀬 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1726\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸 𑀩𑀳𑀼𑀯𑀺𑀥𑀊𑀫𑀺𑀢𑁆𑀣𑀦𑀺𑀢𑀯𑁂𑀕𑀯𑀺𑀲𑀝𑀫𑀸𑀯𑀝𑁆𑀝𑀯𑁂𑀕𑀁 𑀲𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀩𑀳𑀼𑀯𑀺𑀥𑀓𑀺𑀮𑁂𑀲𑀊𑀫𑀺𑀯𑁂𑀕𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀬𑀲𑀲𑀺𑀮𑁄𑀓𑀧𑀽𑀚𑀦𑀯𑀦𑁆𑀤𑀦𑀸 𑀧𑀭𑀓𑀼𑀮𑁂𑀲𑀼 𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀲𑀁𑀲𑀸𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀸𑀦𑀦𑀯𑀺𑀫𑀸𑀦𑀦𑀩𑀳𑀼𑀯𑀺𑀥𑀤𑁄𑀲𑀊𑀫𑀺𑀯𑁂𑀕𑀜𑁆𑀘 𑀲𑀳𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸𑀬 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1727\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸 𑀅𑀧𑀭𑀺𑀫𑀺𑀢𑀫𑀦𑀦𑁆𑀢𑀫𑀧𑀸𑀭𑀫𑀓𑁆𑀔𑁄𑀪𑀺𑀢𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀖𑁄𑀲𑁂 𑀢𑀺𑀫𑀺𑀢𑀺𑀫𑀺𑀗𑁆𑀕𑀮𑀫𑀓𑀭𑀫𑀘𑁆𑀙𑀕𑀡𑀸𑀓𑀼𑀮𑁂 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀢𑀺𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀤𑁆𑀯𑀸𑀤𑀲𑀸𑀓𑀸𑀭𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀲𑀫𑀬𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁂 𑀫𑀸𑀦𑀲𑀁 𑀲𑀜𑁆𑀘𑀸𑀭𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸𑀬 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂 𑀲𑀘𑁆𑀘𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑁂—

1728\. ‘𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀦𑁆𑀢𑀸 𑀘 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, “𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀸𑀣, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀸𑀣, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀸𑀣, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀸𑀣𑀸’”𑀢𑀺𑁇
