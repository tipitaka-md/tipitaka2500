# 𑁭𑁇𑁨𑁇𑁬 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(240.)

1730\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀁 𑀩𑀳𑀼𑀊𑀫𑀺𑀚𑀸𑀮𑀸𑀓𑀼𑀮𑀯𑀺𑀓𑁆𑀔𑁄𑀪𑀺𑀢𑀲𑀮𑀺𑀮𑀢𑀮𑁂 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀦𑀸𑀯𑀁 𑀮𑀕𑁆𑀕𑁂𑀢𑀺 𑀞𑀧𑁂𑀢𑀺, 𑀦 𑀤𑁂𑀢𑀺 𑀤𑀺𑀲𑀸𑀯𑀺𑀤𑀺𑀲𑀁 𑀳𑀭𑀺𑀢𑀼𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀽𑀫𑀺𑀚𑀸𑀮𑁂 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀫𑁆𑀧𑀳𑀸𑀭𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀮𑀕𑁆𑀕𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑀁 𑀤𑀺𑀲𑀸𑀯𑀺𑀤𑀺𑀲𑀁 𑀳𑀭𑀺𑀢𑀼𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1731\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀁 𑀦 𑀧𑁆𑀮𑀯𑀢𑀺 𑀯𑀺𑀲𑀻𑀤𑀢𑀺, 𑀳𑀢𑁆𑀣𑀲𑀢𑁂𑀧𑀺 𑀉𑀤𑀓𑁂 𑀦𑀸𑀯𑀁 𑀮𑀕𑁆𑀕𑁂𑀢𑀺 𑀞𑀸𑀦𑀫𑀼𑀧𑀦𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀮𑀸𑀪𑀬𑀲𑀲𑀓𑁆𑀓𑀸𑀭𑀫𑀸𑀦𑀦𑀯𑀦𑁆𑀤𑀦𑀧𑀽𑀚𑀦𑀅𑀧𑀘𑀺𑀢𑀻𑀲𑀼 𑀮𑀸𑀪𑀕𑁆𑀕𑀬𑀲𑀕𑁆𑀕𑁂𑀧𑀺 𑀦 𑀧𑁆𑀮𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀲𑀭𑀻𑀭𑀬𑀸𑀧𑀦𑀫𑀢𑁆𑀢𑀓𑁂 𑀬𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1732\. _‘𑀬𑀣𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀮𑀕𑁆𑀕𑀦𑀓𑀁,_  
_𑀦 𑀧𑁆𑀮𑀯𑀢𑀺 𑀯𑀺𑀲𑀻𑀤𑀢𑀺;_  
_𑀢𑀣𑁂𑀯 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑁂,_  
_𑀫𑀸 𑀧𑁆𑀮𑀯𑀣 𑀯𑀺𑀲𑀻𑀤𑀣𑀸’”𑀢𑀺𑁇_  


---

1733\. 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
