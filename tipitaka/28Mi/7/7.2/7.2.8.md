# 𑁭𑁇𑁨𑁇𑁮 𑀦𑀺𑀬𑀸𑀫𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀬𑀸𑀫𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(242.)

1736\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀦𑀺𑀬𑀸𑀫𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀬𑀸𑀫𑀓𑁄 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀲𑀢𑀢𑀁 𑀲𑀫𑀺𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀬𑀢𑁆𑀢𑀧𑁆𑀧𑀬𑀢𑁆𑀢𑁄 𑀦𑀸𑀯𑀁 𑀲𑀸𑀭𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀬𑀸𑀫𑀬𑀫𑀸𑀦𑁂𑀦 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀲𑀢𑀢𑀁 𑀲𑀫𑀺𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁂𑀦 𑀬𑀢𑁆𑀢𑀧𑁆𑀧𑀬𑀢𑁆𑀢𑁂𑀦 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀬𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀬𑀸𑀫𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀥𑀫𑁆𑀫𑀧𑀤𑁂—

1737\. _‘𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀭𑀢𑀸 𑀳𑁄𑀣,_  
_𑀲𑀘𑀺𑀢𑁆𑀢𑀫𑀦𑀼𑀭𑀓𑁆𑀔𑀣;_  
_𑀤𑀼𑀕𑁆𑀕𑀸 𑀉𑀤𑁆𑀥𑀭𑀣𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀧𑀗𑁆𑀓𑁂 𑀲𑀦𑁆𑀦𑁄𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄’𑀢𑀺𑁇_  


1738\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀬𑀸𑀫𑀓𑀲𑁆𑀲 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀯𑀸 𑀧𑀸𑀧𑀓𑀁 𑀯𑀸, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑀁 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀲𑀸𑀯𑀚𑁆𑀚𑀸𑀦𑀯𑀚𑁆𑀚𑀁 𑀳𑀻𑀦𑀧𑁆𑀧𑀡𑀻𑀢𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀁 𑀯𑀺𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀬𑀸𑀫𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1739\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀬𑀸𑀫𑀓𑁄 𑀬𑀦𑁆𑀢𑁂 𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀁 𑀤𑁂𑀢𑀺 ‘𑀫𑀸 𑀓𑁄𑀘𑀺 𑀬𑀦𑁆𑀢𑀁 𑀆𑀫𑀲𑀺𑀢𑁆𑀣𑀸’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀁𑀯𑀭𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸 ‘𑀫𑀸 𑀓𑀺𑀜𑁆𑀘𑀺 𑀧𑀸𑀧𑀓𑀁 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑀻’𑀢𑀺, 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀬𑀸𑀫𑀓𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂—  ‘𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬𑀸𑀣, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀦𑁆’”𑀢𑀺𑁇

---

1740\. 𑀦𑀺𑀬𑀸𑀫𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
