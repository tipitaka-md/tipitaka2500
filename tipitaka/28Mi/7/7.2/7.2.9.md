# 𑁭𑁇𑁨𑁇𑁯 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(243.)

1741. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑁄 𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀢𑀺 ‘𑀪𑀢𑀓𑁄 𑀅𑀳𑀁 𑀇𑀫𑀸𑀬 𑀦𑀸𑀯𑀸𑀬 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀺, 𑀇𑀫𑀸𑀬𑀸𑀳𑀁 𑀦𑀸𑀯𑀸𑀬 𑀯𑀸𑀳𑀲𑀸 𑀪𑀢𑁆𑀢𑀯𑁂𑀢𑀦𑀁 𑀮𑀪𑀸𑀫𑀺, 𑀦 𑀫𑁂 𑀧𑀫𑀸𑀤𑁄 𑀓𑀭𑀡𑀻𑀬𑁄, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦 𑀫𑁂 𑀅𑀬𑀁 𑀦𑀸𑀯𑀸 𑀯𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑀸’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀇𑀫𑀁 𑀔𑁄 𑀅𑀳𑀁 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀪𑀽𑀢𑀺𑀓𑀁 𑀓𑀸𑀬𑀁 𑀲𑀫𑁆𑀫𑀲𑀦𑁆𑀢𑁄 𑀲𑀢𑀢𑀁 𑀲𑀫𑀺𑀢𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀩𑁆𑀬𑀸𑀥𑀺𑀫𑀭𑀡𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁄 𑀫𑁂 𑀓𑀭𑀡𑀻𑀬𑁄’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1742. _‘𑀓𑀸𑀬𑀁 𑀇𑀫𑀁 𑀲𑀫𑁆𑀫𑀲𑀣,_  
_𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀣 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁;_  
_𑀓𑀸𑀬𑁂 𑀲𑀪𑀸𑀯𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀣𑀸’”𑀢𑀺𑁇_  


---

1743. 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀦𑀯𑀫𑁄𑁇
