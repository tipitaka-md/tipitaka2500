# 𑁭𑁇𑁨𑁇𑁧𑁦 𑀲𑀫𑀼𑀤𑁆𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(244.)

1744\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀫𑀢𑁂𑀦 𑀓𑀼𑀡𑀧𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦 𑀲𑀁𑀯𑀲𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀫𑀸𑀦𑀤𑀺𑀝𑁆𑀞𑀺𑀫𑀓𑁆𑀔𑀧𑀍𑀆𑀲𑁃𑀲𑁆𑀲𑀸𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀫𑀸𑀬𑀸𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀓𑀼𑀝𑀺𑀮𑀯𑀺𑀲𑀫𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀓𑀺𑀮𑁂𑀲𑀫𑀮𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦 𑀲𑀁𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1745\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀫𑀼𑀢𑁆𑀢𑀸𑀫𑀡𑀺𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀲𑀗𑁆𑀔𑀲𑀺𑀮𑀸𑀧𑀯𑀸𑀍𑀅𑀨𑀮𑀺𑀓𑀫𑀡𑀺𑀯𑀺𑀯𑀺𑀥𑀭𑀢𑀦𑀦𑀺𑀘𑀬𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀧𑀺𑀤𑀳𑀢𑀺, 𑀦 𑀩𑀳𑀺 𑀯𑀺𑀓𑀺𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀫𑀕𑁆𑀕𑀨𑀮𑀛𑀸𑀦𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀯𑀺𑀯𑀺𑀥𑀕𑀼𑀡𑀭𑀢𑀦𑀸𑀦𑀺 𑀅𑀥𑀺𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀺𑀤𑀳𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀩𑀳𑀺 𑀦𑀻𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1746\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀫𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀫𑀳𑀸𑀪𑀽𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀲𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀁 𑀥𑀼𑀢𑀯𑀸𑀤𑀁 𑀲𑀮𑁆𑀮𑁂𑀔𑀯𑀼𑀢𑁆𑀢𑀺𑀁 𑀆𑀘𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀚𑁆𑀚𑀺𑀁 𑀧𑁂𑀲𑀮𑀁 𑀕𑀭𑀼𑀁 𑀪𑀸𑀯𑀦𑀻𑀬𑀁 𑀯𑀢𑁆𑀢𑀸𑀭𑀁 𑀯𑀘𑀦𑀓𑁆𑀔𑀫𑀁 𑀘𑁄𑀤𑀓𑀁 𑀧𑀸𑀧𑀕𑀭𑀳𑀺𑀁 𑀑𑀯𑀸𑀤𑀓𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀓𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑀁 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑀁 𑀲𑀫𑀸𑀤𑀧𑀓𑀁 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑀁 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1747\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀦𑀯𑀲𑀮𑀺𑀮𑀲𑀫𑁆𑀧𑀼𑀡𑁆𑀡𑀸𑀳𑀺 𑀕𑀗𑁆𑀕𑀸𑀬𑀫𑀼𑀦𑀸𑀅𑀘𑀺𑀭𑀯𑀢𑀻𑀲𑀭𑀪𑀽𑀫𑀳𑀻𑀆𑀤𑀻𑀳𑀺 𑀦𑀤𑀻𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀲𑀮𑀺𑀮𑀥𑀸𑀭𑀸𑀳𑀺 𑀘 𑀧𑀽𑀭𑀺𑀢𑁄𑀧𑀺 𑀲𑀓𑀁 𑀯𑁂𑀮𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀯𑀦𑁆𑀤𑀦𑀫𑀸𑀦𑀦𑀧𑀽𑀚𑀦𑀓𑀸𑀭𑀡𑀸 𑀚𑀻𑀯𑀺𑀢𑀳𑁂𑀢𑀼𑀧𑀺 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀦 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1748\. ‘𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀞𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑁂𑀮𑀁 𑀦𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀁 𑀫𑀬𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀢𑀁 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸 𑀚𑀻𑀯𑀺𑀢𑀳𑁂𑀢𑀼𑀧𑀺 𑀦𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀻’𑀢𑀺𑁇

1749\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀲𑀩𑁆𑀩𑀲𑀯𑀦𑁆𑀢𑀻𑀳𑀺 𑀕𑀗𑁆𑀕𑀸𑀬𑀫𑀼𑀦𑀸𑀅𑀘𑀺𑀭𑀯𑀢𑀻𑀲𑀭𑀪𑀽𑀫𑀳𑀻𑀳𑀺 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀉𑀤𑀓𑀥𑀸𑀭𑀸𑀳𑀺𑀧𑀺 𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀲𑀯𑀦𑀥𑀸𑀭𑀡𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑀕𑀸𑀍𑀳𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀯𑀺𑀕𑁆𑀕𑀳𑀧𑀤𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀧𑀤𑀲𑀦𑁆𑀥𑀺 𑀧𑀤𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀦𑀯𑀗𑁆𑀕𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀯𑀭𑀁 𑀲𑀼𑀡𑀦𑁆𑀢𑁂𑀦𑀸𑀧𑀺 𑀦 𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀼𑀢𑀲𑁄𑀫𑀚𑀸𑀢𑀓𑁂—

1750\. _‘𑀅𑀕𑁆𑀕𑀺 𑀬𑀣𑀸 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀁 𑀤𑀳𑀦𑁆𑀢𑁄,_  
_𑀦 𑀢𑀧𑁆𑀧𑀢𑀺 𑀲𑀸𑀕𑀭𑁄 𑀯𑀸 𑀦𑀤𑀻𑀳𑀺;_  
_𑀏𑀯𑀫𑁆𑀧𑀺 𑀘𑁂 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀭𑀸𑀚𑀲𑁂𑀝𑁆𑀞,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀦 𑀢𑀧𑁆𑀧𑀦𑁆𑀢𑀺 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑁂𑀦𑀸’”𑀢𑀺𑁇_  


---

1751\. 𑀲𑀫𑀼𑀤𑁆𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀲𑀫𑁄𑁇



---

1752\. 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



---

1753\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1754\. _𑀮𑀸𑀩𑀼𑀮𑀢𑀸 𑀘 𑀧𑀤𑀼𑀫𑀁,_  
_𑀩𑀻𑀚𑀁 𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸;_  
_𑀦𑀸𑀯𑀸 𑀘 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀁,_  
_𑀓𑀽𑀧𑁄 𑀦𑀺𑀬𑀸𑀫𑀓𑁄 𑀢𑀣𑀸;_  
_𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑁄 𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀘,_  
_𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
