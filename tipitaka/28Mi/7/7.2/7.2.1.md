# 𑁭𑁇𑁨𑁇𑁧 𑀮𑀸𑀩𑀼𑀮𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀮𑀸𑀩𑀼𑀮𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(235.)

1707\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀮𑀸𑀩𑀼𑀮𑀢𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑀸𑀩𑀼𑀮𑀢𑀸 𑀢𑀺𑀡𑁂 𑀯𑀸 𑀓𑀝𑁆𑀞𑁂 𑀯𑀸 𑀮𑀢𑀸𑀬 𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀸𑀳𑀺 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑀽𑀧𑀭𑀺 𑀯𑀟𑁆𑀠𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀭𑀳𑀢𑁆𑀢𑁂 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀢𑀼𑀓𑀸𑀫𑁂𑀦 𑀫𑀦𑀲𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑁂 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑀸𑀩𑀼𑀮𑀢𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1708\. _‘𑀬𑀣𑀸 𑀮𑀸𑀩𑀼𑀮𑀢𑀸 𑀦𑀸𑀫,_  
_𑀢𑀺𑀡𑁂 𑀓𑀝𑁆𑀞𑁂 𑀮𑀢𑀸𑀬 𑀯𑀸;_  
_𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀸𑀳𑀺,_  
_𑀢𑀢𑁄 𑀯𑀟𑁆𑀠𑀢𑀺 𑀉𑀧𑁆𑀧𑀭𑀺𑁇_  


1709\. _𑀢𑀣𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑁂𑀦,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀓𑀸𑀫𑀺𑀦𑀸;_  
_𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸,_  
_𑀯𑀟𑁆𑀠𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀲𑁂𑀓𑁆𑀔𑀨𑀮𑁂’”𑀢𑀺𑁇_  


---
