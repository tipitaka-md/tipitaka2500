# 𑁭𑁇𑁨𑁇𑁩 𑀩𑀻𑀚𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀩𑀻𑀚𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(237.)

1715. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀩𑀻𑀚𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀻𑀚𑀁 𑀅𑀧𑁆𑀧𑀓𑀫𑁆𑀧𑀺 𑀲𑀫𑀸𑀦𑀁 𑀪𑀤𑁆𑀤𑀓𑁂 𑀔𑁂𑀢𑁆𑀢𑁂 𑀯𑀼𑀢𑁆𑀢𑀁 𑀤𑁂𑀯𑁂 𑀲𑀫𑁆𑀫𑀸 𑀥𑀸𑀭𑀁 𑀧𑀯𑁂𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀲𑀼𑀩𑀳𑀽𑀦𑀺 𑀨𑀮𑀸𑀦𑀺 𑀅𑀦𑀼𑀤𑀲𑁆𑀲𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑀣𑀸 𑀧𑀝𑀺𑀧𑀸𑀤𑀺𑀢𑀁 𑀲𑀻𑀮𑀁 𑀓𑁂𑀯𑀮𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀫𑀦𑀼𑀤𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀻𑀚𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1716. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀻𑀚𑀁 𑀲𑀼𑀧𑀭𑀺𑀲𑁄𑀥𑀺𑀢𑁂 𑀔𑁂𑀢𑁆𑀢𑁂 𑀭𑁄𑀧𑀺𑀢𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀁𑀯𑀺𑀭𑀽𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀫𑀸𑀦𑀲𑀁 𑀲𑀼𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂 𑀧𑀭𑀺𑀲𑁄𑀥𑀺𑀢𑀁 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀔𑁂𑀢𑁆𑀢𑀯𑀭𑁂 𑀔𑀺𑀢𑁆𑀢𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀯𑀺𑀭𑀽𑀳𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀻𑀚𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁂𑀦—

1717. _‘𑀬𑀣𑀸𑀧𑀺 𑀔𑁂𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂,_  
_𑀩𑀻𑀚𑀜𑁆𑀘𑀲𑁆𑀲 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀯𑀺𑀧𑀼𑀮𑀁 𑀢𑀲𑁆𑀲 𑀨𑀮𑀁 𑀳𑁄𑀢𑀺,_  
_𑀅𑀧𑀺 𑀢𑁄𑀲𑁂𑀢𑀺 𑀓𑀲𑁆𑀲𑀓𑀁𑁇_  


1718. _𑀢𑀣𑁂𑀯 𑀬𑁄𑀕𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀔𑁂𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀯𑀺𑀭𑀽𑀳𑀢𑀻’”𑀢𑀺𑁇_  


---

1719. 𑀩𑀻𑀚𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
