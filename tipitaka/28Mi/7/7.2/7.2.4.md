# 𑁭𑁇𑁨𑁇𑁪 𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(238.)

1720\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸 𑀦𑀸𑀫 𑀅𑀦𑁆𑀢𑁄𑀧𑀣𑀯𑀺𑀬𑀁𑀬𑁂𑀯 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀢𑀺 𑀳𑀢𑁆𑀣𑀲𑀢𑀫𑁆𑀧𑀺 𑀪𑀺𑀬𑁆𑀬𑁄𑀧𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬𑁄 𑀓𑁂𑀯𑀮𑀜𑁆𑀘 𑀲𑀫𑀡𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂𑀬𑁂𑀯 𑀧𑀭𑀺𑀧𑀽𑀭𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀭𑀸𑀳𑀼𑀮𑁂𑀦—

1721\. _‘𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸 𑀦𑀸𑀫,_  
_𑀧𑀸𑀤𑀧𑁄 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑁄;_  
_𑀅𑀦𑁆𑀢𑁄𑀧𑀣𑀯𑀺𑀬𑀁𑀬𑁂𑀯,_  
_𑀲𑀢𑀳𑀢𑁆𑀣𑁄𑀧𑀺 𑀯𑀟𑁆𑀠𑀢𑀺𑁇_  


1722\. _𑀬𑀣𑀸 𑀓𑀸𑀮𑀫𑁆𑀳𑀺 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀧𑀭𑀺𑀧𑀸𑀓𑁂𑀦 𑀲𑁄 𑀤𑀼𑀫𑁄;_  
_𑀉𑀕𑁆𑀕𑀜𑁆𑀙𑀺𑀢𑁆𑀯𑀸𑀦 𑀏𑀓𑀸𑀳𑀁,_  
_𑀲𑀢𑀳𑀢𑁆𑀣𑁄𑀧𑀺 𑀯𑀟𑁆𑀠𑀢𑀺𑁇_  


1723\. _𑀏𑀯𑀫𑁂𑀯𑀸𑀳𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸 𑀯𑀺𑀬;_  
_𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑁂 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂,_  
_𑀥𑀫𑁆𑀫𑀢𑁄 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀬𑀺𑀦𑁆’”𑀢𑀺𑁇_  


---

1724\. 𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
