# 𑁭𑁇𑁭𑁇𑁨 𑀓𑀸𑀍𑀆𑀬𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀍𑀆𑀬𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(286.)

1989\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀓𑀸𑀍𑀆𑀬𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀍𑀆𑀬𑀲𑁄 𑀲𑀼𑀧𑀻𑀢𑁄 𑀯𑀫𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑁄 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀫𑀸𑀦𑀲𑀁 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀅𑀧𑀻𑀢𑀁 𑀯𑀫𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀍𑀆𑀬𑀲𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1990\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀍𑀆𑀬𑀲𑁄 𑀲𑀓𑀺𑀁 𑀧𑀻𑀢𑀁 𑀉𑀤𑀓𑀁 𑀦 𑀯𑀫𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑁄 𑀲𑀓𑀺𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀧𑀲𑀸𑀤𑁄, 𑀦 𑀧𑀼𑀦 𑀲𑁄 𑀯𑀫𑀺𑀢𑀩𑁆𑀩𑁄 ‘𑀉𑀍𑀆𑀭𑁄 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁄’𑀢𑀺𑁇 ‘𑀭𑀽𑀧𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀁, 𑀯𑁂𑀤𑀦𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀅𑀦𑀺𑀘𑁆𑀘𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀲𑀓𑀺𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀜𑀸𑀡𑀁, 𑀦 𑀧𑀼𑀦 𑀢𑀁 𑀯𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀍𑀆𑀬𑀲𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1991\. _‘𑀤𑀲𑁆𑀲𑀦𑀫𑁆𑀳𑀺 𑀧𑀭𑀺𑀲𑁄𑀥𑀺𑀢𑁄 𑀦𑀭𑁄,_  
_𑀅𑀭𑀺𑀬𑀥𑀫𑁆𑀫𑁂 𑀦𑀺𑀬𑀢𑁄 𑀯𑀺𑀲𑁂𑀲𑀕𑀽;_  
_𑀦𑀧𑁆𑀧𑀯𑁂𑀥𑀢𑀺 𑀅𑀦𑁂𑀓𑀪𑀸𑀕𑀲𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑁄 𑀘 𑀫𑀼𑀔𑀪𑀸𑀯𑀫𑁂𑀯 𑀲𑁄’”𑀢𑀺𑁇_  


---

1992\. 𑀓𑀸𑀍𑀆𑀬𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
