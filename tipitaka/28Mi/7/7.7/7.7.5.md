# 𑁭𑁇𑁭𑁇𑁫 𑀅𑀕𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀅𑀕𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(289.)

2004. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀅𑀕𑀤𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑀤𑁂 𑀓𑀺𑀫𑀻 𑀦 𑀲𑀡𑁆𑀞𑀳𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀫𑀸𑀦𑀲𑁂 𑀓𑀺𑀮𑁂𑀲𑀸 𑀦 𑀲𑀡𑁆𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑀤𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

2005. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑀤𑁄 𑀤𑀝𑁆𑀞𑀨𑀼𑀝𑁆𑀞𑀤𑀺𑀝𑁆𑀞𑀅𑀲𑀺𑀢𑀧𑀻𑀢𑀔𑀸𑀬𑀺𑀢𑀲𑀸𑀬𑀺𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀯𑀺𑀲𑀁 𑀧𑀝𑀺𑀳𑀦𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀫𑀸𑀦𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀁 𑀲𑀩𑁆𑀩𑀁 𑀧𑀝𑀺𑀳𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑀤𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

2006. _‘𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀲𑀪𑀸𑀯𑀢𑁆𑀣𑀁,_  
_𑀤𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑁂𑀦 𑀬𑁄𑀕𑀺𑀦𑀸;_  
_𑀅𑀕𑀤𑁂𑀦𑁂𑀯 𑀳𑁄𑀢𑀩𑁆𑀩𑀁,_  
_𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀲𑀦𑀸𑀲𑀦𑁂’”𑀢𑀺𑁇_  


---

2007. 𑀅𑀕𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇
