# 𑁭𑁇𑁭𑁇𑁬 𑀪𑁄𑀚𑀦𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀪𑁄𑀚𑀦𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(290.)

2008\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀁 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀼𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

2009\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀁 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑀮𑀁 𑀯𑀟𑁆𑀠𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀼𑀜𑁆𑀜𑀯𑀟𑁆𑀠𑀺𑀬𑀸 𑀯𑀟𑁆𑀠𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

2010\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀁 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀸𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂𑀦—

2011\. _‘𑀲𑀁𑀬𑀫𑁂𑀦 𑀦𑀺𑀬𑀫𑁂𑀦,_  
_𑀲𑀻𑀮𑁂𑀦 𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀧𑀢𑁆𑀣𑀺𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀲𑁆𑀲 𑀬𑁄𑀕𑀺𑀦𑀸’”𑀢𑀺𑁇_  


---

2012\. 𑀪𑁄𑀚𑀦𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
