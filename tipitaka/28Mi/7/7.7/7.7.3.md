# 𑁭𑁇𑁭𑁇𑁩 𑀙𑀢𑁆𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕

### 𑀙𑀢𑁆𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(287.)

1993\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀙𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀙𑀢𑁆𑀢𑀁 𑀉𑀧𑀭𑀺 𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁 𑀉𑀧𑀭𑀺 𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀘𑀭𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀙𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1994\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀙𑀢𑁆𑀢𑀁 𑀫𑀼𑀤𑁆𑀥𑀦𑀼𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑀁 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀼𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀙𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1995\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀙𑀢𑁆𑀢𑀁 𑀯𑀸𑀢𑀸𑀢𑀧𑀫𑁂𑀖𑀯𑀼𑀝𑁆𑀞𑀺𑀬𑁄 𑀧𑀝𑀺𑀳𑀦𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀦𑀸𑀦𑀸𑀯𑀺𑀥𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀼𑀣𑀼𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀫𑀢𑀯𑀸𑀢-𑀢𑀺𑀯𑀺𑀥𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀓𑀺𑀮𑁂𑀲𑀯𑀼𑀝𑁆𑀞𑀺𑀬𑁄 𑀧𑀝𑀺𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀙𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1996\. _‘𑀬𑀣𑀸𑀧𑀺 𑀙𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑀼𑀮𑀁,_  
_𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀁 𑀣𑀺𑀭𑀲𑀁𑀳𑀺𑀢𑀁;_  
_𑀯𑀸𑀢𑀸𑀢𑀧𑀁 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺,_  
_𑀫𑀳𑀢𑀻 𑀫𑁂𑀖𑀯𑀼𑀝𑁆𑀞𑀺𑀬𑁄𑁇_  


1997\. _𑀢𑀣𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑁄𑀧𑀺,_  
_𑀲𑀻𑀮𑀙𑀢𑁆𑀢𑀥𑀭𑁄 𑀲𑀼𑀘𑀺;_  
_𑀓𑀺𑀮𑁂𑀲𑀯𑀼𑀝𑁆𑀞𑀺𑀁 𑀯𑀸𑀭𑁂𑀢𑀺,_  
_𑀲𑀦𑁆𑀢𑀸𑀧𑀢𑀺𑀯𑀺𑀥𑀕𑁆𑀕𑀬𑁄’”𑀢𑀺𑁇_  


---

1998\. 𑀙𑀢𑁆𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
