# 𑀫𑀸𑀢𑀺𑀓𑀸

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀫𑀸𑀢𑀺𑀓𑀸

(224.)

1542. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀺𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻”𑀢𑀺?

1543. “𑀇𑀥, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀼𑀓𑀸𑀫𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸—

1544. 𑀕𑀤𑁆𑀭𑀪𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1545. 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1546. 𑀓𑀮𑀦𑁆𑀤𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1547. 𑀤𑀻𑀧𑀺𑀦𑀺𑀬𑀸 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1548. 𑀤𑀻𑀧𑀺𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1549. 𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1550. 𑀯𑀁𑀲𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1551. 𑀘𑀸𑀧𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1552. 𑀯𑀸𑀬𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1553. 𑀫𑀓𑁆𑀓𑀝𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

---

1554. 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



1555. 𑀮𑀸𑀩𑀼𑀮𑀢𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1556. 𑀧𑀤𑀼𑀫𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1557. 𑀩𑀻𑀚𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1558. 𑀲𑀸𑀮𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1559. 𑀦𑀸𑀯𑀸𑀬 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1560. 𑀦𑀸𑀯𑀸𑀮𑀕𑁆𑀕𑀦𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1561. 𑀓𑀽𑀧𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1562. 𑀦𑀺𑀬𑀸𑀫𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1563. 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1564. 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

---

1565. 𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



1566. 𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1567. 𑀆𑀧𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1568. 𑀢𑁂𑀚𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1569. 𑀯𑀸𑀬𑀼𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1570. 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1571. 𑀆𑀓𑀸𑀲𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1572. 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1573. 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1574. 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1575. 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

---

1576. 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



1577. 𑀉𑀧𑀘𑀺𑀓𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1578. 𑀩𑀺𑀍𑀆𑀭𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1579. 𑀉𑀦𑁆𑀤𑀽𑀭𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1580. 𑀯𑀺𑀘𑁆𑀙𑀺𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1581. 𑀦𑀓𑀼𑀮𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1582. 𑀚𑀭𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1583. 𑀫𑀺𑀕𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1584. 𑀕𑁄𑀭𑀽𑀧𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1585. 𑀯𑀭𑀸𑀳𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1586. 𑀳𑀢𑁆𑀣𑀺𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

---

1587. 𑀉𑀧𑀘𑀺𑀓𑀸𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇



1588. 𑀲𑀻𑀳𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1589. 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1590. 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀸𑀬 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1591. 𑀖𑀭𑀓𑀧𑁄𑀢𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1592. 𑀉𑀮𑀽𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1593. 𑀲𑀢𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1594. 𑀯𑀕𑁆𑀕𑀼𑀮𑀺𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1595. 𑀚𑀮𑀽𑀓𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1596. 𑀲𑀧𑁆𑀧𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1597. 𑀅𑀚𑀕𑀭𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

---

1598. 𑀲𑀻𑀳𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



1599. 𑀧𑀦𑁆𑀣𑀫𑀓𑁆𑀓𑀝𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1600. 𑀣𑀦𑀲𑀺𑀢𑀤𑀸𑀭𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1601. 𑀘𑀺𑀢𑁆𑀢𑀓𑀥𑀭𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1602. 𑀧𑀯𑀦𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1603. 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1604. 𑀫𑁂𑀖𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1605. 𑀫𑀡𑀺𑀭𑀢𑀦𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1606. 𑀫𑀸𑀕𑀯𑀺𑀓𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1607. 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1608. 𑀢𑀘𑁆𑀙𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

---

1609. 𑀫𑀓𑁆𑀓𑀝𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇



1610. 𑀓𑀼𑀫𑁆𑀪𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1611. 𑀓𑀸𑀍𑀆𑀬𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1612. 𑀙𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1613. 𑀔𑁂𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1614. 𑀅𑀕𑀤𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1615. 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1616. 𑀇𑀲𑁆𑀲𑀸𑀲𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

---

1617. 𑀓𑀼𑀫𑁆𑀪𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇



1618. 𑀭𑀜𑁆𑀜𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1619. 𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1620. 𑀦𑀺𑀲𑀤𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1621. 𑀧𑀤𑀻𑀧𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1622. 𑀫𑀬𑀽𑀭𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1623. 𑀢𑀼𑀭𑀗𑁆𑀕𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1624. 𑀲𑁄𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1625. 𑀇𑀦𑁆𑀤𑀔𑀻𑀮𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1626. 𑀢𑀼𑀮𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1627. 𑀔𑀕𑁆𑀕𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1628. 𑀫𑀘𑁆𑀙𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1629. 𑀇𑀡𑀕𑁆𑀕𑀸𑀳𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1630. 𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1631. 𑀫𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1632. 𑀦𑀤𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1633. 𑀉𑀲𑀪𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1634. 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1635. 𑀲𑀼𑀗𑁆𑀓𑀲𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1636. 𑀘𑁄𑀭𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1637. 𑀲𑀓𑀼𑀡𑀕𑁆𑀖𑀺𑀬𑀸 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1638. 𑀲𑀼𑀦𑀔𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1639. 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1640. 𑀕𑀩𑁆𑀪𑀺𑀦𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1641. 𑀘𑀫𑀭𑀺𑀬𑀸 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1642. 𑀓𑀺𑀓𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1643. 𑀓𑀧𑁄𑀢𑀺𑀓𑀸𑀬 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1644. 𑀏𑀓𑀦𑀬𑀦𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1645. 𑀓𑀲𑁆𑀲𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1646. 𑀚𑀫𑁆𑀩𑀼𑀓𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀺𑀬𑀸 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1647. 𑀘𑀗𑁆𑀕𑀯𑀸𑀭𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1648. 𑀤𑀩𑁆𑀩𑀺𑀬𑀸 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1649. 𑀇𑀡𑀲𑀸𑀥𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1650. 𑀅𑀦𑀼𑀯𑀺𑀘𑀺𑀦𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1651. 𑀲𑀸𑀭𑀣𑀺𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1652. 𑀪𑁄𑀚𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇

1653. 𑀢𑀼𑀦𑁆𑀦𑀯𑀸𑀬𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1654. 𑀦𑀸𑀯𑀺𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1655. 𑀪𑀫𑀭𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺𑁇
