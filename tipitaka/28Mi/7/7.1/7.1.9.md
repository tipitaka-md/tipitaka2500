# 𑁭𑁇𑁧𑁇𑁯 𑀯𑀸𑀬𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀯𑀸𑀬𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(233.)

1696\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀯𑀸𑀬𑀲𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀲𑁄 𑀆𑀲𑀗𑁆𑀓𑀺𑀢𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀺𑀢𑁄 𑀬𑀢𑁆𑀢𑀧𑁆𑀧𑀬𑀢𑁆𑀢𑁄 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀲𑀗𑁆𑀓𑀺𑀢𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀺𑀢𑁂𑀦 𑀬𑀢𑁆𑀢𑀧𑀬𑀢𑁆𑀢𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀲𑀁𑀯𑀼𑀢𑁂𑀳𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀳𑀺 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀲𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1697\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀲𑁄 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀪𑁄𑀚𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀜𑀸𑀢𑀻𑀳𑀺 𑀲𑀁𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑁂 𑀢𑁂 𑀮𑀸𑀪𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀸 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀧𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀳𑀺 𑀮𑀸𑀪𑁂𑀳𑀺 𑀧𑀝𑀺𑀯𑀺𑀪𑀢𑁆𑀢𑀪𑁄𑀕𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀲𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1698\. _‘𑀲𑀘𑁂 𑀫𑁂 𑀉𑀧𑀦𑀸𑀫𑁂𑀦𑁆𑀢𑀺,_  
_𑀬𑀣𑀸𑀮𑀤𑁆𑀥𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀁𑀯𑀺𑀪𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁄 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺 𑀪𑁄𑀚𑀦𑀦𑁆’”𑀢𑀺𑁇_  


---

1699\. 𑀯𑀸𑀬𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀦𑀯𑀫𑁄𑁇
