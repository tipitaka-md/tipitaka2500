# 𑁭𑁇𑁧𑁇𑁭 𑀯𑀁𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀯𑀁𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(231.)

1690\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀯𑀁𑀲𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀁𑀲𑁄 𑀬𑀢𑁆𑀣 𑀯𑀸𑀢𑁄, 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑀺, 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀣𑀫𑀦𑀼𑀥𑀸𑀯𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀪𑀸𑀲𑀺𑀢𑀁 𑀦𑀯𑀗𑁆𑀕𑀁 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁, 𑀢𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀦𑀯𑀚𑁆𑀚𑁂 𑀞𑀢𑁆𑀯𑀸 𑀲𑀫𑀡𑀥𑀫𑁆𑀫𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀁𑀲𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀭𑀸𑀳𑀼𑀮𑁂𑀦—

1691\. _‘𑀦𑀯𑀗𑁆𑀕𑀁 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑀁,_  
_𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑀩𑁆𑀩𑀤𑀸;_  
_𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀦𑀯𑀚𑁆𑀚𑀲𑁆𑀫𑀺𑀁,_  
_𑀞𑀢𑁆𑀯𑀸𑀧𑀸𑀬𑀁 𑀲𑀫𑀼𑀢𑁆𑀢𑀭𑀺𑀦𑁆’”𑀢𑀺𑁇_  


---
