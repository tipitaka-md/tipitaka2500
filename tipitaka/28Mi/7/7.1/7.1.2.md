# 𑁭𑁇𑁧𑁇𑁨 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(226.)

1660\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀓𑀸𑀮𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀸𑀮𑁂𑀦 𑀲𑀫𑀬𑁂𑀦𑁂𑀯 𑀘𑁂𑀢𑀺𑀬𑀗𑁆𑀕𑀡𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀭𑀻𑀭𑀁 𑀧𑀝𑀺𑀚𑀕𑁆𑀕𑀺𑀢𑁆𑀯𑀸 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀘𑁂𑀢𑀺𑀬𑀁 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀟𑁆𑀠𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1661\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀓𑀸𑀮𑁂𑀦 𑀲𑀫𑀬𑁂𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀸𑀮𑁂𑀦 𑀲𑀫𑀬𑁂𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀘𑁂𑀢𑀺𑀬𑀗𑁆𑀕𑀡𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀭𑀻𑀭𑀁 𑀧𑀝𑀺𑀚𑀕𑁆𑀕𑀺𑀢𑁆𑀯𑀸 𑀘𑁂𑀢𑀺𑀬𑀁 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1662\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀧𑀣𑀯𑀺𑀁 𑀔𑀡𑀺𑀢𑁆𑀯𑀸 𑀔𑀡𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑁄𑀳𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀦𑁂𑀯 𑀤𑀯𑀸𑀬 𑀦 𑀫𑀤𑀸𑀬 𑀦 𑀫𑀡𑁆𑀟𑀦𑀸𑀬 𑀦 𑀯𑀺𑀪𑀽𑀲𑀦𑀸𑀬, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀇𑀫𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀞𑀺𑀢𑀺𑀬𑀸 𑀬𑀸𑀧𑀦𑀸𑀬 𑀯𑀺𑀳𑀺𑀁𑀲𑀽𑀧𑀭𑀢𑀺𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬, 𑀇𑀢𑀺 𑀧𑀼𑀭𑀸𑀡𑀜𑁆𑀘 𑀯𑁂𑀤𑀦𑀁 𑀧𑀝𑀺𑀳𑀗𑁆𑀔𑀸𑀫𑀺 𑀦𑀯𑀜𑁆𑀘 𑀯𑁂𑀤𑀦𑀁 𑀦 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑀸𑀢𑁆𑀭𑀸 𑀘 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑀯𑀚𑁆𑀚𑀢𑀸 𑀘 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀘𑀸’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1663\. _‘𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀧𑀼𑀢𑁆𑀢𑀫𑀁𑀲𑀁𑀯,_  
_𑀅𑀓𑁆𑀔𑀲𑁆𑀲𑀩𑁆𑀪𑀜𑁆𑀚𑀦𑀁 𑀬𑀣𑀸;_  
_𑀏𑀯𑀁 𑀆𑀳𑀭𑀺 𑀆𑀳𑀸𑀭𑀁,_  
_𑀬𑀸𑀧𑀦𑀢𑁆𑀣𑀫𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄’𑀢𑀺𑁇_  


1664\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀲𑀘𑀓𑁆𑀔𑀼𑀓𑁄𑀧𑀺 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑁆𑀥𑁄 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀦𑀦𑁆𑀥𑁂𑀦𑁂𑀯 𑀅𑀦𑁆𑀥𑁂𑀦 𑀯𑀺𑀬 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀭𑀜𑁆𑀜𑁂𑀧𑀺 𑀕𑁄𑀘𑀭𑀕𑀸𑀫𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀭𑀚𑀦𑀻𑀬𑁂𑀲𑀼 𑀭𑀽𑀧𑀲𑀤𑁆𑀤𑀕𑀦𑁆𑀥𑀭𑀲𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀦𑁆𑀥𑁂𑀦 𑀩𑀥𑀺𑀭𑁂𑀦 𑀫𑀽𑀕𑁂𑀦 𑀯𑀺𑀬 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦𑀸𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁂𑀦—

1665\. _‘𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑀲𑁆𑀲 𑀬𑀣𑀸 𑀅𑀦𑁆𑀥𑁄,_  
_𑀲𑁄𑀢𑀯𑀸 𑀩𑀥𑀺𑀭𑁄 𑀬𑀣𑀸;_  
_𑀧𑀜𑁆𑀜𑀯𑀸𑀲𑁆𑀲 𑀬𑀣𑀸 𑀫𑀽𑀕𑁄,_  
_𑀩𑀮𑀯𑀸 𑀤𑀼𑀩𑁆𑀩𑀮𑁄𑀭𑀺𑀯;_  
_𑀅𑀢𑁆𑀢𑀅𑀢𑁆𑀣𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀲𑀬𑁂𑀣 𑀫𑀢𑀲𑀸𑀬𑀺𑀓𑀦𑁆’𑀢𑀺𑁇_  


1666\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄 𑀮𑁂𑀟𑁆𑀟𑀼𑀤𑀡𑁆𑀟𑀮𑀕𑀼𑀍𑀅𑀫𑀼𑀕𑁆𑀕𑀭𑁂𑀳𑀺 𑀧𑀭𑀺𑀧𑀸𑀢𑀺𑀬𑀦𑁆𑀢𑁄𑀧𑀺 𑀲𑀓𑀁 𑀕𑁂𑀳𑀁 𑀦 𑀯𑀺𑀚𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀘𑀻𑀯𑀭𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀦𑀯𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀯𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀯𑀢𑁆𑀢𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑁄, 𑀲𑀓𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑁄 𑀕𑁂𑀳𑀁, 𑀬𑀤𑀺𑀤𑀁 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 ‘𑀓𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀕𑁄𑀘𑀭𑁄 𑀲𑀓𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑁄 𑀯𑀺𑀲𑀬𑁄? 𑀬𑀤𑀺𑀤𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸’𑀢𑀺𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸𑀧𑀺—

1667\. _‘𑀬𑀣𑀸 𑀲𑀼𑀤𑀦𑁆𑀢𑁄 𑀫𑀸𑀢𑀗𑁆𑀕𑁄,_  
_𑀲𑀓𑀁 𑀲𑁄𑀡𑁆𑀟𑀁 𑀦 𑀫𑀤𑁆𑀤𑀢𑀺;_  
_𑀪𑀓𑁆𑀔𑀸𑀪𑀓𑁆𑀔𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀯𑀼𑀢𑁆𑀢𑀺𑀓𑀧𑁆𑀧𑀦𑀁𑁇_  


1668\. _𑀢𑀣𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑁂𑀦,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀧𑀦;_  
_𑀚𑀺𑀦𑀯𑀘𑀦𑀁 𑀦 𑀫𑀤𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁,_  
_𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀦𑁆’”𑀢𑀺𑁇_  


---

1669\. 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
