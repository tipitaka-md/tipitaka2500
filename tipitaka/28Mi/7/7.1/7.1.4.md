# 𑁭𑁇𑁧𑁇𑁪 𑀤𑀻𑀧𑀺𑀦𑀺𑀬𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀤𑀻𑀧𑀺𑀦𑀺𑀬𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(228.)

1673\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀤𑀻𑀧𑀺𑀦𑀺𑀬𑀸 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀻𑀧𑀺𑀦𑀻 𑀲𑀓𑀺𑀁𑀬𑁂𑀯 𑀕𑀩𑁆𑀪𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀦 𑀧𑀼𑀦𑀧𑁆𑀧𑀼𑀦𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀉𑀧𑁂𑀢𑀺? 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀬𑀢𑀺𑀁 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀁 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀁 𑀘𑀼𑀢𑀺𑀁 𑀪𑁂𑀤𑀁 𑀔𑀬𑀁 𑀯𑀺𑀦𑀸𑀲𑀁 𑀲𑀁𑀲𑀸𑀭𑀪𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀲𑀫𑀁 𑀲𑀫𑁆𑀧𑀻𑀍𑀇𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 ‘𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁂 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀻𑀧𑀺𑀦𑀺𑀬𑀸 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀼𑀢𑁆𑀢𑀦𑀺𑀧𑀸𑀢𑁂 𑀥𑀦𑀺𑀬𑀕𑁄𑀧𑀸𑀮𑀓𑀲𑀼𑀢𑁆𑀢𑁂—

1674\. _‘𑀉𑀲𑀪𑁄𑀭𑀺𑀯 𑀙𑁂𑀢𑁆𑀯 𑀩𑀦𑁆𑀥𑀦𑀸𑀦𑀺,_  
_𑀦𑀸𑀕𑁄 𑀧𑀽𑀢𑀺𑀮𑀢𑀁𑀯 𑀤𑀸𑀮𑀬𑀺𑀢𑁆𑀯𑀸;_  
_𑀦𑀸𑀳𑀁 𑀧𑀼𑀦𑀼𑀧𑁂𑀲𑁆𑀲𑀁 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀁,_  
_𑀅𑀣 𑀘𑁂 𑀧𑀢𑁆𑀣𑀬𑀲𑀻 𑀧𑀯𑀲𑁆𑀲 𑀤𑁂𑀯𑀸’”𑀢𑀺𑁇_  


---

1675\. 𑀤𑀻𑀧𑀺𑀦𑀺𑀬𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
