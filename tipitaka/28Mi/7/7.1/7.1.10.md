# 𑁭𑁇𑁧𑁇𑁧𑁦 𑀫𑀓𑁆𑀓𑀝𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀫𑀓𑁆𑀓𑀝𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(234.)

1700\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀫𑀓𑁆𑀓𑀝𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀓𑁆𑀓𑀝𑁄 𑀯𑀸𑀲𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀢𑀣𑀸𑀭𑀽𑀧𑁂 𑀑𑀓𑀸𑀲𑁂 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑁂 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂 𑀲𑀩𑁆𑀩𑀝𑁆𑀞𑀓𑀲𑀸𑀔𑁂 𑀪𑀻𑀭𑀼𑀢𑁆𑀢𑀸𑀡𑁂 𑀯𑀸𑀲𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀮𑀚𑁆𑀚𑀺𑀁 𑀧𑁂𑀲𑀮𑀁 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀁 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀁 𑀥𑀫𑁆𑀫𑀥𑀭𑀁 𑀯𑀺𑀦𑀬𑀥𑀭𑀁 𑀧𑀺𑀬𑀁 𑀕𑀭𑀼𑀪𑀸𑀯𑀦𑀻𑀬𑀁 𑀯𑀢𑁆𑀢𑀸𑀭𑀁 𑀯𑀘𑀦𑀓𑁆𑀔𑀫𑀁 𑀑𑀯𑀸𑀤𑀓𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑀁 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑀁 𑀲𑀫𑀸𑀤𑀧𑀓𑀁 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑀁 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀘𑀭𑀺𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀓𑁆𑀓𑀝𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1701\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀓𑁆𑀓𑀝𑁄 𑀭𑀼𑀓𑁆𑀔𑁂𑀬𑁂𑀯 𑀘𑀭𑀢𑀺 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀬𑀤𑀺 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀸𑀲𑀫𑀦𑀼𑀪𑀯𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀯𑀦𑀸𑀪𑀺𑀫𑀼𑀔𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀧𑀯𑀦𑁂𑀬𑁂𑀯 𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀲𑀬𑀦𑀁 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀫𑀦𑀼𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀓𑁆𑀓𑀝𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1702\. _‘𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁄𑀧𑀺 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁄,_  
_𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀲𑀬𑀦𑁂𑀦 𑀯𑀸;_  
_𑀧𑀯𑀦𑁂 𑀲𑁄𑀪𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀯𑀦𑀦𑁆𑀢𑀁𑀯 𑀯𑀡𑁆𑀡𑀺𑀢𑀦𑁆’”𑀢𑀺𑁇_  


---

1703\. 𑀫𑀓𑁆𑀓𑀝𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀲𑀫𑁄𑁇



---

1704\. 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

1705\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1706\. _𑀕𑀤𑁆𑀭𑀪𑁄 𑀘𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑁄,_  
_𑀓𑀮𑀦𑁆𑀤𑁄 𑀤𑀻𑀧𑀺𑀦𑀺 𑀤𑀻𑀧𑀺𑀓𑁄;_  
_𑀓𑀼𑀫𑁆𑀫𑁄 𑀯𑀁𑀲𑁄 𑀘 𑀘𑀸𑀧𑁄 𑀘,_  
_𑀯𑀸𑀬𑀲𑁄 𑀅𑀣 𑀫𑀓𑁆𑀓𑀝𑁄𑀢𑀺𑁇_
