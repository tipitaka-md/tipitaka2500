# 𑁭𑁇𑁧𑁇𑁮 𑀘𑀸𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀘𑀸𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(232.)

1693\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀘𑀸𑀧𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀸𑀧𑁄 𑀲𑀼𑀢𑀘𑁆𑀙𑀺𑀢𑁄 𑀦𑀫𑀺𑀢𑁄 𑀬𑀸𑀯𑀕𑁆𑀕𑀫𑀽𑀮𑀁 𑀲𑀫𑀓𑀫𑁂𑀯 𑀅𑀦𑀼𑀦𑀫𑀢𑀺 𑀦𑀧𑁆𑀧𑀝𑀺𑀢𑁆𑀣𑀫𑁆𑀪𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀣𑁂𑀭𑀦𑀯𑀫𑀚𑁆𑀛𑀺𑀫𑀲𑀫𑀓𑁂𑀲𑀼 𑀅𑀦𑀼𑀦𑀫𑀺𑀢𑀩𑁆𑀩𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀨𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀸𑀧𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀯𑀺𑀥𑀼𑀭 (𑀧𑀼𑀡𑁆𑀡𑀓) 𑀚𑀸𑀢𑀓𑁂—

1694\. _‘𑀘𑀸𑀧𑁄𑀯𑀽𑀦𑀼𑀤𑀭𑁄 𑀥𑀻𑀭𑁄,_  
_𑀯𑀁𑀲𑁄 𑀯𑀸𑀧𑀺 𑀧𑀓𑀫𑁆𑀧𑀬𑁂;_  
_𑀧𑀝𑀺𑀮𑁄𑀫𑀁 𑀦 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂’”𑀢𑀺𑁇_  


---
