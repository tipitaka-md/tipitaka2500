# 𑁭𑁇𑁧𑁇𑁬 𑀓𑀼𑀫𑁆𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

### 𑀓𑀼𑀫𑁆𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(230.)

1682\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑁄 𑀉𑀤𑀓𑀘𑀭𑁄 𑀉𑀤𑀓𑁂𑀬𑁂𑀯 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀪𑀽𑀢𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀳𑀕𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀧𑀼𑀮𑁂𑀦 𑀫𑀳𑀕𑁆𑀕𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁂𑀦 𑀅𑀯𑁂𑀭𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁂𑀦 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀮𑁄𑀓𑀁 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1683\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑁄 𑀉𑀤𑀓𑁂 𑀉𑀧𑁆𑀧𑀺𑀮𑀯𑀦𑁆𑀢𑁄 𑀲𑀻𑀲𑀁 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑀤𑀺 𑀓𑁄𑀘𑀺 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀢𑀺 𑀕𑀸𑀍𑀳𑀫𑁄𑀕𑀸𑀳𑀢𑀺 ‘𑀫𑀸 𑀫𑀁 𑀢𑁂 𑀧𑀼𑀦 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀺𑀮𑁂𑀲𑁂𑀲𑀼 𑀑𑀧𑀢𑀦𑁆𑀢𑁂𑀲𑀼 𑀆𑀭𑀫𑁆𑀫𑀡𑀲𑀭𑁂 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀕𑀸𑀍𑀳𑀫𑁄𑀕𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀫𑀸 𑀫𑀁 𑀓𑀺𑀮𑁂𑀲𑀸 𑀧𑀼𑀦 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1684\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑁄 𑀉𑀤𑀓𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀁 𑀑𑀢𑀸𑀧𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀦𑀺𑀲𑀚𑁆𑀚𑀝𑁆𑀞𑀸𑀦𑀲𑀬𑀦𑀘𑀗𑁆𑀓𑀫𑀢𑁄 𑀫𑀸𑀦𑀲𑀁 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑁂 𑀫𑀸𑀦𑀲𑀁 𑀑𑀢𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1685\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑁄 𑀧𑀣𑀯𑀺𑀁 𑀔𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀜𑁆𑀜𑀁 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀓𑀸𑀦𑀦𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀓𑀦𑁆𑀤𑀭𑀁 𑀕𑀺𑀭𑀺𑀕𑀼𑀳𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀫𑁄𑀕𑀸𑀳𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂 𑀬𑁂𑀯 𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀉𑀧𑀲𑁂𑀦𑁂𑀦 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑁂𑀦—

1686\. _‘𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁,_  
_𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀲𑁂𑀯𑁂 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀓𑀸𑀭𑀡𑀸’𑀢𑀺𑁇_  


1687\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑁄 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑀤𑀺 𑀓𑀜𑁆𑀘𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀯𑀸, 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀲𑁄𑀡𑁆𑀟𑀺𑀧𑀜𑁆𑀘𑀫𑀸𑀦𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀲𑀓𑁂 𑀓𑀧𑀸𑀮𑁂 𑀦𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑁄 𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀓𑀸𑀬𑀫𑀦𑀼𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀭𑀽𑀧𑀲𑀤𑁆𑀤𑀕𑀦𑁆𑀥𑀭𑀲𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀆𑀧𑀢𑀦𑁆𑀢𑁂𑀲𑀼 𑀙𑀲𑀼 𑀤𑁆𑀯𑀸𑀭𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑀓𑀯𑀸𑀝𑀁 𑀅𑀦𑀼𑀕𑁆𑀖𑀸𑀝𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀲𑀁 𑀲𑀫𑁄𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀯𑀭𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀢𑁂𑀦 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁂𑀦 𑀯𑀺𑀳𑀸𑀢𑀩𑁆𑀩𑀁 𑀲𑀫𑀡𑀥𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀫𑀸𑀦𑁂𑀦𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂 𑀓𑀼𑀫𑁆𑀫𑀽𑀧𑀫𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁂—

1688\. _‘𑀓𑀼𑀫𑁆𑀫𑁄𑀯 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀲𑀓𑁂 𑀓𑀧𑀸𑀮𑁂,_  
_𑀲𑀫𑁄𑀤𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀦𑁄𑀯𑀺𑀢𑀓𑁆𑀓𑁂;_  
_𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀅𑀜𑁆𑀜𑀫𑀳𑁂𑀞𑀬𑀸𑀦𑁄,_  
_𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀦𑀽𑀧𑀯𑀤𑁂𑀬𑁆𑀬 𑀓𑀜𑁆𑀘𑀻’”𑀢𑀺𑁇_  


---

1689\. 𑀓𑀼𑀫𑁆𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
