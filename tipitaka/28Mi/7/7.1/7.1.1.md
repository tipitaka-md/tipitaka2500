# 𑁭𑁇𑁧𑁇𑁧 𑀕𑀤𑁆𑀭𑀪𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀕𑀤𑁆𑀭𑀪𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(225.)

1657\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀕𑀤𑁆𑀭𑀪𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀤𑁆𑀭𑀪𑁄 𑀦𑀸𑀫 𑀲𑀗𑁆𑀓𑀸𑀭𑀓𑀽𑀝𑁂𑀧𑀺 𑀘𑀢𑀼𑀓𑁆𑀓𑁂𑀧𑀺 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀧𑀺 𑀕𑀸𑀫𑀤𑁆𑀯𑀸𑀭𑁂𑀧𑀺 𑀣𑀼𑀲𑀭𑀸𑀲𑀺𑀫𑁆𑀳𑀺𑀧𑀺 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀲𑀬𑀢𑀺, 𑀦 𑀲𑀬𑀦𑀩𑀳𑀼𑀮𑁄 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀢𑀺𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑁂𑀧𑀺 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑁂𑀧𑀺 𑀓𑀝𑁆𑀞𑀫𑀜𑁆𑀘𑀓𑁂𑀧𑀺 𑀙𑀫𑀸𑀬𑀧𑀺 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀘𑀫𑁆𑀫𑀔𑀡𑁆𑀟𑀁 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀲𑀬𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀲𑀬𑀦𑀩𑀳𑀼𑀮𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀤𑁆𑀭𑀪𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 ‘𑀓𑀮𑀺𑀗𑁆𑀕𑀭𑀽𑀧𑀥𑀸𑀦𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀥𑀸𑀦𑀲𑁆𑀫𑀺𑀦𑁆’𑀢𑀺𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸𑀧𑀺—

1658\. _‘𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀚𑀡𑁆𑀡𑀼𑀓𑁂 𑀦𑀸𑀪𑀺𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄’”𑀢𑀺𑁇_  


---

1659\. 𑀕𑀤𑁆𑀭𑀪𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀞𑀫𑁄𑁇
