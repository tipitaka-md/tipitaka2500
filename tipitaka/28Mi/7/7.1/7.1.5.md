# 𑁭𑁇𑁧𑁇𑁫 𑀤𑀻𑀧𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀕𑀤𑁆𑀭𑀪𑀯𑀕𑁆𑀕

* 𑀤𑀻𑀧𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(229.)

1676\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀤𑀻𑀧𑀺𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀻𑀧𑀺𑀓𑁄 𑀅𑀭𑀜𑁆𑀜𑁂 𑀢𑀺𑀡𑀕𑀳𑀦𑀁 𑀯𑀸 𑀯𑀦𑀕𑀳𑀦𑀁 𑀯𑀸 𑀧𑀩𑁆𑀩𑀢𑀕𑀳𑀦𑀁 𑀯𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦𑀺𑀮𑀻𑀬𑀺𑀢𑁆𑀯𑀸 𑀫𑀺𑀕𑁂 𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀯𑀺𑀯𑁂𑀓𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀭𑀜𑁆𑀜𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀓𑀦𑁆𑀤𑀭𑀁 𑀕𑀺𑀭𑀺𑀕𑀼𑀳𑀁 𑀲𑀼𑀲𑀸𑀦𑀁 𑀯𑀦𑀧𑀢𑁆𑀣𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀁 𑀧𑀮𑀸𑀮𑀧𑀼𑀜𑁆𑀚𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁 𑀯𑀺𑀚𑀦𑀯𑀸𑀢𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀳𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀁; 𑀯𑀺𑀯𑁂𑀓𑀁 𑀲𑁂𑀯𑀫𑀸𑀦𑁄 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀻 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀘 𑀯𑀲𑀺𑀪𑀸𑀯𑀁 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀻𑀧𑀺𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁂𑀳𑀺—

1677\. _‘𑀬𑀣𑀸𑀧𑀺 𑀤𑀻𑀧𑀺𑀓𑁄 𑀦𑀸𑀫,_  
_𑀦𑀺𑀮𑀻𑀬𑀺𑀢𑁆𑀯𑀸 𑀕𑀡𑁆𑀳𑀢𑁂 𑀫𑀺𑀕𑁂;_  
_𑀢𑀣𑁂𑀯𑀸𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀬𑀼𑀢𑁆𑀢𑀬𑁄𑀕𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑁄;_  
_𑀅𑀭𑀜𑁆𑀜𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀨𑀮𑀫𑀼𑀢𑁆𑀢𑀫𑀦𑁆’𑀢𑀺𑁇_  


1678\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀻𑀧𑀺𑀓𑁄 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀧𑀲𑀼𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀸𑀫𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀧𑀢𑀺𑀢𑀁 𑀦 𑀪𑀓𑁆𑀔𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀯𑁂𑀍𑀉𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀧𑀢𑁆𑀢𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀧𑀼𑀧𑁆𑀨𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀨𑀮𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀺𑀦𑀸𑀦𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀘𑀼𑀡𑁆𑀡𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀫𑀼𑀔𑁄𑀤𑀓𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀘𑀸𑀢𑀼𑀓𑀫𑁆𑀬𑀢𑀸𑀬 𑀯𑀸 𑀫𑀼𑀕𑁆𑀕𑀲𑀼𑀧𑁆𑀬𑀢𑀸𑀬 𑀯𑀸 𑀧𑀸𑀭𑀺𑀪𑀝𑁆𑀬𑀢𑀸𑀬 𑀯𑀸 𑀚𑀗𑁆𑀖𑀧𑁂𑀲𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑁂𑀚𑁆𑀚𑀓𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀤𑀽𑀢𑀓𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀧𑀳𑀺𑀡𑀕𑀫𑀦𑁂𑀦 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀝𑀺𑀧𑀺𑀡𑁆𑀟𑁂𑀦 𑀯𑀸 𑀤𑀸𑀦𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑁂𑀦 𑀯𑀸 𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀯𑀸 𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀯𑀸 𑀅𑀗𑁆𑀕𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀚𑀻𑀯𑁂𑀦 𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑀺𑀢𑀁 𑀪𑁄𑀚𑀦𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀯𑀸𑀫𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀧𑀢𑀺𑀢𑀁 𑀧𑀲𑀼𑀁 𑀯𑀺𑀬 𑀤𑀻𑀧𑀺𑀓𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀻𑀧𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1679\. _‘𑀯𑀘𑀻𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑀸,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀫𑀥𑀼𑀧𑀸𑀬𑀲𑀁;_  
_𑀲𑀘𑁂 𑀪𑀼𑀢𑁆𑀢𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀳𑀁,_  
_𑀲𑀸𑀚𑀻𑀯𑁄 𑀕𑀭𑀳𑀺𑀢𑁄 𑀫𑀫𑁇_  


1680\. _𑀬𑀤𑀺𑀧𑀺 𑀫𑁂 𑀅𑀦𑁆𑀢𑀕𑀼𑀡𑀁,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀻 𑀘𑀭𑁂;_  
_𑀦𑁂𑀯 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀫𑀸𑀚𑀻𑀯𑀁,_  
_𑀘𑀚𑀫𑀸𑀦𑁄𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀦𑁆’”𑀢𑀺𑁇_  


---

1681\. 𑀤𑀻𑀧𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇
