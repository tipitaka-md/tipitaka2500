# 𑁭𑁇𑁬𑁇𑁧 𑀧𑀦𑁆𑀣𑀫𑀓𑁆𑀓𑀝𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀦𑁆𑀣𑀫𑀓𑁆𑀓𑀝𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(275.)

1935\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀧𑀦𑁆𑀣𑀫𑀓𑁆𑀓𑀝𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀦𑁆𑀣𑀫𑀓𑁆𑀓𑀝𑀓𑁄 𑀧𑀦𑁆𑀣𑁂 𑀫𑀓𑁆𑀓𑀝𑀚𑀸𑀮𑀯𑀺𑀢𑀸𑀦𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑀤𑀺 𑀢𑀢𑁆𑀣 𑀚𑀸𑀮𑀓𑁂 𑀮𑀕𑁆𑀕𑀢𑀺 𑀓𑀺𑀫𑀺 𑀯𑀸 𑀫𑀓𑁆𑀔𑀺𑀓𑀸 𑀯𑀸 𑀧𑀝𑀗𑁆𑀕𑁄 𑀯𑀸, 𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀓𑁆𑀔𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀙𑀲𑀼 𑀤𑁆𑀯𑀸𑀭𑁂𑀲𑀼 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀚𑀸𑀮𑀯𑀺𑀢𑀸𑀦𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑀤𑀺 𑀢𑀢𑁆𑀣 𑀓𑀺𑀮𑁂𑀲𑀫𑀓𑁆𑀔𑀺𑀓𑀸 𑀩𑀚𑁆𑀛𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀖𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀦𑁆𑀣𑀫𑀓𑁆𑀓𑀝𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁂𑀦—

1936\. _‘𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀬𑀫𑁂 𑀙𑀲𑀼 𑀤𑁆𑀯𑀸𑀭𑁂𑀲𑀼,_  
_𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀓𑀺𑀮𑁂𑀲𑀸 𑀢𑀢𑁆𑀣 𑀮𑀕𑁆𑀕𑀸 𑀘𑁂,_  
_𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑀸 𑀢𑁂 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑀸’”𑀢𑀺𑁇_  


---

1937\. 𑀧𑀦𑁆𑀣𑀫𑀓𑁆𑀓𑀝𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀞𑀫𑁄𑁇
