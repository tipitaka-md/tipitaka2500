# 𑁭𑁇𑁬𑁇𑁪 𑀧𑀯𑀦𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀯𑀦𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(278.)

1944\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀧𑀯𑀦𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀁 𑀦𑀸𑀫 𑀅𑀲𑀼𑀘𑀺𑀚𑀦𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀭𑁂𑀲𑀁 𑀅𑀧𑀭𑀤𑁆𑀥𑀁 𑀔𑀮𑀺𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦 𑀯𑀺𑀯𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1945\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀁 𑀲𑀼𑀜𑁆𑀜𑀁 𑀧𑀘𑀼𑀭𑀚𑀦𑁂𑀳𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀫𑀸𑀦𑀤𑀺𑀝𑁆𑀞𑀺𑀚𑀸𑀮𑁂𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀘 𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀲𑀼𑀜𑁆𑀜𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1946\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀁 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀚𑀦𑀲𑀫𑁆𑀩𑀸𑀥𑀭𑀳𑀺𑀢𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀳𑀺 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀦𑀭𑀺𑀬𑁂𑀳𑀺 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1947\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀁 𑀲𑀦𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀦𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂𑀦 𑀧𑀳𑀻𑀦𑀫𑀸𑀦𑁂𑀦 𑀧𑀳𑀻𑀦𑀫𑀓𑁆𑀔𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1948\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀁 𑀅𑀭𑀺𑀬𑀚𑀦𑀲𑀁𑀲𑁂𑀯𑀺𑀢𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀭𑀺𑀬𑀚𑀦𑀲𑀁𑀲𑁂𑀯𑀺𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀯𑀦𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂—

1949\. _‘𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀅𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁂𑀳𑀺 𑀛𑀸𑀬𑀺𑀪𑀺;_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁂𑀳𑀺,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁂𑀳𑀺 𑀲𑀳𑀸𑀯𑀲𑁂’”𑀢𑀺𑁇_  


---

1950\. 𑀧𑀯𑀦𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
