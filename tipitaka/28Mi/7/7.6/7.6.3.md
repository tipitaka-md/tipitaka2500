# 𑁭𑁇𑁬𑁇𑁩 𑀘𑀺𑀢𑁆𑀢𑀓𑀥𑀭𑀓𑀼𑀫𑁆𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

### 𑀘𑀺𑀢𑁆𑀢𑀓𑀥𑀭𑀓𑀼𑀫𑁆𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(277.)

1941\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀘𑀺𑀢𑁆𑀢𑀓𑀥𑀭𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀓𑀥𑀭𑀓𑀼𑀫𑁆𑀫𑁄 𑀉𑀤𑀓𑀪𑀬𑀸 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀘𑀭𑀢𑀺, 𑀢𑀸𑀬 𑀘 𑀧𑀦 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀦𑀸𑀬 𑀆𑀬𑀼𑀦𑀸 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀫𑀸𑀤𑁂 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂 𑀕𑀼𑀡𑀯𑀺𑀲𑁂𑀲𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑀸𑁇 𑀢𑀸𑀬 𑀘 𑀧𑀦 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀢𑀸𑀬 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀸, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀉𑀧𑁂𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀓𑀥𑀭𑀓𑀼𑀫𑁆𑀫𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀥𑀫𑁆𑀫𑀧𑀤𑁂—

1942\. _‘𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀭𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀧𑀫𑀸𑀤𑁂 𑀪𑀬𑀤𑀲𑁆𑀲𑀺 𑀯𑀸;_  
_𑀅𑀪𑀩𑁆𑀩𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲𑁂𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂’”𑀢𑀺𑁇_  


---

1943\. 𑀘𑀺𑀢𑁆𑀢𑀓𑀥𑀭𑀓𑀼𑀫𑁆𑀫𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
