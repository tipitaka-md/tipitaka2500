# 𑁭𑁇𑁬𑁇𑁧𑁦 𑀢𑀘𑁆𑀙𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

### 𑀢𑀘𑁆𑀙𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(284.)

1978\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀢𑀘𑁆𑀙𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀘𑁆𑀙𑀓𑁄 𑀓𑀸𑀍𑀅𑀲𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑁆𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀁 𑀢𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀫𑀦𑀼𑀮𑁄𑀫𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀧𑀣𑀯𑀺𑀬𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀤𑁆𑀥𑀸𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀯𑀸𑀲𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀢𑀘𑁆𑀙𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀘𑁆𑀙𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1979\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀘𑁆𑀙𑀓𑁄 𑀨𑁂𑀕𑁆𑀕𑀼𑀁 𑀅𑀧𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀭𑀫𑀸𑀤𑀺𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀲𑁆𑀲𑀢𑀁 𑀉𑀘𑁆𑀙𑁂𑀤𑀁 𑀢𑀁 𑀚𑀻𑀯𑀁 𑀢𑀁 𑀲𑀭𑀻𑀭𑀁 𑀅𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀁 𑀅𑀜𑁆𑀜𑀁 𑀲𑀭𑀻𑀭𑀁 𑀢𑀤𑀼𑀢𑁆𑀢𑀫𑀁 𑀅𑀜𑁆𑀜𑀤𑀼𑀢𑁆𑀢𑀫𑀁 𑀅𑀓𑀢𑀫𑀪𑀩𑁆𑀩𑀁 𑀅𑀧𑀼𑀭𑀺𑀲𑀓𑀸𑀭𑀁 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀁 𑀲𑀢𑁆𑀢𑀯𑀺𑀦𑀸𑀲𑀁 𑀦𑀯𑀲𑀢𑁆𑀢𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀁 𑀲𑀗𑁆𑀔𑀸𑀭𑀲𑀲𑁆𑀲𑀢𑀪𑀸𑀯𑀁 𑀬𑁄 𑀓𑀭𑁄𑀢𑀺, 𑀲𑁄 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑁄 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑁂𑀢𑀺, 𑀓𑀫𑁆𑀫𑀨𑀮𑀤𑀲𑁆𑀲𑀦𑀸 𑀘 𑀓𑀺𑀭𑀺𑀬𑀨𑀮𑀤𑀺𑀝𑁆𑀞𑀺 𑀘 𑀇𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀘𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀯𑀺𑀯𑀸𑀤𑀧𑀣𑀸𑀦𑀺 𑀅𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀲𑀪𑀸𑀯𑀁 𑀧𑀭𑀫𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀦𑀺𑀭𑀻𑀳𑀦𑀺𑀚𑁆𑀚𑀻𑀯𑀢𑀁 𑀅𑀘𑁆𑀘𑀦𑁆𑀢𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀘𑁆𑀙𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀼𑀢𑁆𑀢𑀦𑀺𑀧𑀸𑀢𑁂—

1980\. _‘𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑀁 𑀦𑀺𑀤𑁆𑀥𑀫𑀣,_  
_𑀓𑀲𑀫𑁆𑀩𑀼𑀁 𑀅𑀧𑀓𑀲𑁆𑀲𑀣;_  
_𑀢𑀢𑁄 𑀧𑀮𑀸𑀧𑁂 𑀯𑀸𑀳𑁂𑀣,_  
_𑀅𑀲𑁆𑀲𑀫𑀡𑁂 𑀲𑀫𑀡𑀫𑀸𑀦𑀺𑀦𑁂𑁇_  


1981\. _𑀦𑀺𑀤𑁆𑀥𑀫𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁂,_  
_𑀧𑀸𑀧𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑁂;_  
_𑀲𑀼𑀤𑁆𑀥𑀸 𑀲𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀲𑀁𑀯𑀸𑀲𑀁,_  
_𑀓𑀧𑁆𑀧𑀬𑀯𑁆𑀳𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑀸;_  
_𑀢𑀢𑁄 𑀲𑀫𑀕𑁆𑀕𑀸 𑀦𑀺𑀧𑀓𑀸,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀣𑀸’”𑀢𑀺𑁇_  


---

1982\. 𑀢𑀘𑁆𑀙𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀲𑀫𑁄𑁇



---

1983\. 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇



---

1984\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1985\. _𑀫𑀓𑁆𑀓𑀝𑁄 𑀤𑀸𑀭𑀓𑁄 𑀓𑀼𑀫𑁆𑀫𑁄,_  
_𑀯𑀦𑀁 𑀭𑀼𑀓𑁆𑀔𑁄 𑀘 𑀧𑀜𑁆𑀘𑀫𑁄;_  
_𑀫𑁂𑀖𑁄 𑀫𑀡𑀺 𑀫𑀸𑀕𑀯𑀺𑀓𑁄,_  
_𑀩𑀸𑀍𑀇𑀲𑀻 𑀢𑀘𑁆𑀙𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_
