# 𑁭𑁇𑁬𑁇𑁮 𑀫𑀸𑀕𑀯𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

### 𑀫𑀸𑀕𑀯𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(282.)

1968\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀫𑀸𑀕𑀯𑀺𑀓𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑁄 𑀅𑀧𑁆𑀧𑀫𑀺𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀺𑀤𑁆𑀥𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1969\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑁄 𑀫𑀺𑀕𑁂𑀲𑀼𑀬𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀭𑀫𑁆𑀫𑀡𑁂𑀲𑀼𑀬𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1970\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑁄 𑀓𑀸𑀮𑀁 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀚𑀸𑀦𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀓𑀸𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄 ‘𑀅𑀬𑀁 𑀓𑀸𑀮𑁄 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲, 𑀅𑀬𑀁 𑀓𑀸𑀮𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑀸𑀬𑀸’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1971\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑁄 𑀫𑀺𑀕𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀳𑀸𑀲𑀫𑀪𑀺𑀚𑀦𑁂𑀢𑀺 ‘𑀇𑀫𑀁 𑀮𑀘𑁆𑀙𑀸𑀫𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀅𑀪𑀺𑀭𑀫𑀺𑀢𑀩𑁆𑀩𑀁, 𑀳𑀸𑀲𑀫𑀪𑀺𑀚𑀦𑁂𑀢𑀩𑁆𑀩𑀁 ‘𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀺𑀲𑁂𑀲𑀫𑀥𑀺𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀕𑀯𑀺𑀓𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀫𑁄𑀖𑀭𑀸𑀚𑁂𑀦—

1972\. _‘𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀮𑀪𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸;_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀳𑀸𑀲𑁄 𑀚𑀦𑁂𑀢𑀩𑁆𑀩𑁄,_  
_𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀦𑁆’”𑀢𑀺𑁇_  


---

1973\. 𑀫𑀸𑀕𑀯𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
