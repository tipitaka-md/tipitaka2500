# 𑁭𑁇𑁬𑁇𑁨 𑀣𑀦𑀲𑁆𑀲𑀺𑀢𑀤𑀸𑀭𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

### 𑀣𑀦𑀲𑁆𑀲𑀺𑀢𑀤𑀸𑀭𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(276.)

1938. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀣𑀦𑀲𑁆𑀲𑀺𑀢𑀤𑀸𑀭𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑀦𑀲𑁆𑀲𑀺𑀢𑀤𑀸𑀭𑀓𑁄 𑀲𑀤𑀢𑁆𑀣𑁂 𑀮𑀕𑁆𑀕𑀢𑀺, 𑀔𑀻𑀭𑀢𑁆𑀣𑀺𑀓𑁄 𑀭𑁄𑀤𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀤𑀢𑁆𑀣𑁂 𑀮𑀕𑁆𑀕𑀺𑀢𑀩𑁆𑀩𑀁, 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀜𑀸𑀡𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀉𑀤𑁆𑀤𑁂𑀲𑁂 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀬 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀬𑁄𑀕𑁂 𑀧𑀯𑀺𑀯𑁂𑀓𑁂 𑀕𑀭𑀼𑀲𑀁𑀯𑀸𑀲𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀲𑁂𑀯𑀦𑁂𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑀦𑀲𑁆𑀲𑀺𑀢𑀤𑀸𑀭𑀓𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀤𑀻𑀖𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁂—

1939. ‘𑀇𑀗𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀸𑀭𑀢𑁆𑀣𑁂 𑀖𑀝𑀣, 𑀲𑀸𑀭𑀢𑁆𑀣𑁂 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀣, 𑀲𑀸𑀭𑀢𑁆𑀣𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀣𑀸’”𑀢𑀺𑁇

---

1940. 𑀣𑀦𑀲𑁆𑀲𑀺𑀢𑀤𑀸𑀭𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
