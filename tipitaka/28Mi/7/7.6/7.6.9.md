# 𑁭𑁇𑁬𑁇𑁯 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(283.)

1974. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑁄 𑀩𑀍𑀇𑀲𑁂𑀦 𑀫𑀘𑁆𑀙𑁂 𑀉𑀤𑁆𑀥𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1975. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑁄 𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀼𑀮𑀁 𑀮𑀸𑀪𑀫𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀭𑀺𑀢𑁆𑀢𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀫𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀫𑀢𑁆𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀺𑀢𑁆𑀯𑀸 𑀬𑁄𑀕𑀻 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁄 𑀯𑀺𑀧𑀼𑀮𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀭𑀸𑀳𑀼𑀮𑁂𑀦—

1976. _‘𑀲𑀼𑀜𑁆𑀜𑀢𑀜𑁆𑀘𑀸𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀜𑁆𑀘𑀸𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀁;_  
_𑀘𑀢𑀼𑀭𑁄 𑀨𑀮𑁂 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸,_  
_𑀘𑀚𑀺𑀢𑁆𑀯𑀸 𑀮𑁄𑀓𑀸𑀫𑀺𑀲𑀁 𑀮𑀪𑁂’”𑀢𑀺𑁇_  


---

1977. 𑀩𑀸𑀍𑀇𑀲𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀦𑀯𑀫𑁄𑁇
