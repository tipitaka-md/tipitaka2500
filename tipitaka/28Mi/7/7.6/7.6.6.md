# 𑁭𑁇𑁬𑁇𑁬 𑀫𑁂𑀖𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

* 𑀫𑁂𑀖𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(280.)

1956\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀫𑁂𑀖𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀓𑀺𑀮𑁂𑀲𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1957\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑁄 𑀧𑀣𑀯𑀺𑀬𑀸 𑀉𑀡𑁆𑀳𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀤𑁂𑀯𑀓𑁄 𑀮𑁄𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1958\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑁄 𑀲𑀩𑁆𑀩𑀩𑀻𑀚𑀸𑀦𑀺 𑀯𑀺𑀭𑀼𑀳𑀸𑀧𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀲𑀤𑁆𑀥𑀸𑀩𑀻𑀚𑀁 𑀢𑀻𑀲𑀼 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑀁, 𑀤𑀺𑀩𑁆𑀩𑀫𑀸𑀦𑀼𑀲𑀺𑀓𑀸𑀲𑀼 𑀲𑀼𑀔𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀬𑀸𑀯𑀧𑀭𑀫𑀢𑁆𑀣𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀼𑀔𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1959\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑁄 𑀉𑀢𑀼𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀭𑀡𑀺𑀢𑀮𑀭𑀼𑀳𑁂 𑀢𑀺𑀡𑀭𑀼𑀓𑁆𑀔𑀮𑀢𑀸𑀕𑀼𑀫𑁆𑀩𑀑𑀲𑀥𑀺𑀯𑀦𑀧𑁆𑀧𑀢𑀬𑁄 𑀧𑀭𑀺𑀭𑀓𑁆𑀔𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀢𑁂𑀦 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀲𑀫𑀡𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀭𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑁄, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀫𑀽𑀮𑀓𑀸 𑀲𑀩𑁆𑀩𑁂 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1960\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑁄 𑀯𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀦𑀤𑀺𑀢𑀍𑀆𑀓𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑁄 𑀓𑀦𑁆𑀤𑀭𑀧𑀤𑀭𑀲𑀭𑀲𑁄𑀩𑁆𑀪𑁅𑀤𑀧𑀸𑀦𑀸𑀦𑀺 𑀘 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺 𑀉𑀤𑀓𑀥𑀸𑀭𑀸𑀳𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀕𑀫𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀫𑁂𑀖𑀫𑀪𑀺𑀯𑀲𑁆𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀥𑀺𑀕𑀫𑀓𑀸𑀫𑀸𑀦𑀁 𑀫𑀸𑀦𑀲𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀖𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1961\. _‘𑀩𑁄𑀥𑀦𑁂𑀬𑁆𑀬𑀁 𑀚𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀧𑀺 𑀬𑁄𑀚𑀦𑁂;_  
_𑀔𑀡𑁂𑀦 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑁄𑀥𑁂𑀢𑀺 𑀢𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀻’”𑀢𑀺𑁇_  


---

1962\. 𑀫𑁂𑀖𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
