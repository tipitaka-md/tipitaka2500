# 𑁭𑁇𑁬𑁇𑁫 𑀭𑀼𑀓𑁆𑀔𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀫𑀓𑁆𑀓𑀝𑀓𑀯𑀕𑁆𑀕

### 𑀭𑀼𑀓𑁆𑀔𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(279.)

1951. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀼𑀓𑁆𑀔𑁄 𑀦𑀸𑀫 𑀧𑀼𑀧𑁆𑀨𑀨𑀮𑀥𑀭𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀼𑀧𑁆𑀨𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀥𑀸𑀭𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1952. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀼𑀓𑁆𑀔𑁄 𑀉𑀧𑀕𑀢𑀸𑀦𑀫𑀦𑀼𑀧𑁆𑀧𑀯𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀚𑀦𑀸𑀦𑀁 𑀙𑀸𑀬𑀁 𑀤𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀉𑀧𑀕𑀢𑀸𑀦𑀫𑀦𑀼𑀧𑁆𑀧𑀯𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀫𑀺𑀲𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀸𑀭𑁂𑀦 𑀯𑀸 𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1953. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀼𑀓𑁆𑀔𑁄 𑀙𑀸𑀬𑀸𑀯𑁂𑀫𑀢𑁆𑀢𑀁 𑀦 𑀓𑀭𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀸, 𑀘𑁄𑀭𑀯𑀥𑀓𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑁂𑀲𑀼𑀧𑀺 𑀅𑀢𑁆𑀢𑀦𑀺𑀧𑀺 𑀲𑀫𑀲𑀫𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸, ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢𑀸 𑀅𑀯𑁂𑀭𑀸 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀸 𑀅𑀦𑀻𑀖𑀸 𑀲𑀼𑀔𑀻 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1954. _‘𑀯𑀥𑀓𑁂 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀘𑁄𑀭𑁂 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀫𑀸𑀮𑀓𑁂;_  
_𑀥𑀦𑀧𑀸𑀮𑁂 𑀭𑀸𑀳𑀼𑀮𑁂 𑀘,_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀫𑀓𑁄 𑀫𑀼𑀦𑀻’”𑀢𑀺𑁇_  


---

1955. 𑀭𑀼𑀓𑁆𑀔𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇
