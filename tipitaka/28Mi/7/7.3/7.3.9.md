# 𑁭𑁇𑁩𑁇𑁯 𑀲𑀓𑁆𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

### 𑀲𑀓𑁆𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(253.)

1820. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑁄 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀏𑀓𑀦𑁆𑀢𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀸𑀪𑀺𑀭𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1821. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀳𑀸𑀲𑀫𑀪𑀺𑀚𑀦𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀮𑀻𑀦𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑀁 𑀲𑀦𑁆𑀢𑀁 𑀫𑀸𑀦𑀲𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀳𑀸𑀲𑀫𑀪𑀺𑀚𑀦𑁂𑀢𑀩𑁆𑀩𑀁, 𑀉𑀝𑁆𑀞𑀳𑀺𑀢𑀩𑁆𑀩𑀁 𑀖𑀝𑀺𑀢𑀩𑁆𑀩𑀁 𑀯𑀸𑀬𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1822. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀦 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀼𑀪𑀽𑀢𑀺𑀦𑀸—

1823. _‘𑀲𑀸𑀲𑀦𑁂 𑀢𑁂 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀬𑀢𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑀁;_  
_𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀫𑀸𑀦𑀲𑀁 𑀓𑀸𑀫𑀲𑀁𑀳𑀺𑀢𑀦𑁆’”𑀢𑀺𑁇_  


---

1824. 𑀲𑀓𑁆𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀦𑀯𑀫𑁄𑁇
