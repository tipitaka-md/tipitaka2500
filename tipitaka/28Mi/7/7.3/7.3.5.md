# 𑁭𑁇𑁩𑁇𑁫 𑀧𑀩𑁆𑀩𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

* 𑀧𑀩𑁆𑀩𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(249.)

1784\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑁄 𑀅𑀘𑀮𑁄 𑀅𑀓𑀫𑁆𑀧𑀺𑀢𑁄 𑀅𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀻; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀫𑁆𑀫𑀸𑀦𑀦𑁂 𑀯𑀺𑀫𑀸𑀦𑀦𑁂 𑀲𑀓𑁆𑀓𑀸𑀭𑁂 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂 𑀕𑀭𑀼𑀓𑀸𑀭𑁂 𑀅𑀕𑀭𑀼𑀓𑀸𑀭𑁂 𑀬𑀲𑁂 𑀅𑀬𑀲𑁂 𑀦𑀺𑀦𑁆𑀤𑀸𑀬 𑀧𑀲𑀁𑀲𑀸𑀬 𑀲𑀼𑀔𑁂 𑀤𑀼𑀓𑁆𑀔𑁂 𑀇𑀝𑁆𑀞𑀸𑀦𑀺𑀝𑁆𑀞𑁂𑀲𑀼 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀭𑀽𑀧𑀲𑀤𑁆𑀤𑀕𑀦𑁆𑀥𑀭𑀲𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀭𑀚𑀦𑀻𑀬𑁂𑀲𑀼 𑀦 𑀭𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁, 𑀤𑀼𑀲𑁆𑀲𑀦𑀻𑀬𑁂𑀲𑀼 𑀦 𑀤𑀼𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑀁, 𑀫𑀼𑀬𑁆𑀳𑀦𑀻𑀬𑁂𑀲𑀼 𑀦 𑀫𑀼𑀬𑁆𑀳𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀓𑀫𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑀁 𑀦 𑀘𑀮𑀺𑀢𑀩𑁆𑀩𑀁, 𑀧𑀩𑁆𑀩𑀢𑁂𑀦 𑀯𑀺𑀬 𑀅𑀘𑀮𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1785\. _‘𑀲𑁂𑀮𑁄 𑀬𑀣𑀸 𑀏𑀓𑀖𑀦𑁄,_  
_𑀯𑀸𑀢𑁂𑀦 𑀦 𑀲𑀫𑀻𑀭𑀢𑀺;_  
_𑀏𑀯𑀁 𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀲𑀁𑀲𑀸𑀲𑀼,_  
_𑀦 𑀲𑀫𑀺𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸’𑀢𑀺𑁇_  


1786\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑁄 𑀣𑀤𑁆𑀥𑁄 𑀦 𑀓𑁂𑀦𑀘𑀺 𑀲𑀁𑀲𑀝𑁆𑀞𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀣𑀤𑁆𑀥𑁂𑀦 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀓𑁂𑀦𑀘𑀺 𑀲𑀁𑀲𑀕𑁆𑀕𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1787\. _‘𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀕𑀳𑀝𑁆𑀞𑁂𑀳𑀺,_  
_𑀅𑀦𑀸𑀕𑀸𑀭𑁂𑀳𑀺 𑀘𑀽𑀪𑀬𑀁;_  
_𑀅𑀦𑁄𑀓𑀲𑀸𑀭𑀺𑀫𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀁,_  
_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀦𑁆’𑀢𑀺𑁇_  


1788\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑁂 𑀩𑀻𑀚𑀁 𑀦 𑀯𑀺𑀭𑀽𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀓𑀫𑀸𑀦𑀲𑁂 𑀓𑀺𑀮𑁂𑀲𑀸 𑀦 𑀯𑀺𑀭𑀽𑀳𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀼𑀪𑀽𑀢𑀺𑀦𑀸—

1789\. _‘𑀭𑀸𑀕𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀬𑀤𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑁂 𑀫𑀫;_  
_𑀲𑀬𑀁𑀯 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀏𑀓𑀕𑁆𑀕𑁄 𑀢𑀁 𑀤𑀫𑁂𑀫𑀳𑀁𑁇_  


1790\. _𑀭𑀚𑁆𑀚𑀲𑁂 𑀭𑀚𑀦𑀻𑀬𑁂 𑀘,_  
_𑀤𑀼𑀲𑁆𑀲𑀦𑀻𑀬𑁂 𑀘 𑀤𑀼𑀲𑁆𑀲𑀲𑁂;_  
_𑀫𑀼𑀬𑁆𑀳𑀲𑁂 𑀫𑁄𑀳𑀦𑀻𑀬𑁂 𑀘,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀲𑁆𑀲𑀼 𑀯𑀦𑀸 𑀢𑀼𑀯𑀁𑁇_  


1791\. _𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀬𑀁 𑀯𑀸𑀲𑁄,_  
_𑀦𑀺𑀫𑁆𑀫𑀮𑀸𑀦𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀁;_  
_𑀫𑀸 𑀔𑁄 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀤𑀽𑀲𑁂𑀲𑀺,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀲𑁆𑀲𑀼 𑀯𑀦𑀸 𑀢𑀼𑀯𑀦𑁆’𑀢𑀺𑁇_  


1792\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑁄 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀜𑀸𑀡𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1793\. _‘𑀧𑀫𑀸𑀤𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂𑀦,_  
_𑀬𑀤𑀸 𑀦𑀼𑀤𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀲𑀸𑀤𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀅𑀲𑁄𑀓𑁄 𑀲𑁄𑀓𑀺𑀦𑀺𑀁 𑀧𑀚𑀁;_  
_𑀧𑀩𑁆𑀩𑀢𑀝𑁆𑀞𑁄𑀯 𑀪𑀽𑀫𑀝𑁆𑀞𑁂,_  
_𑀥𑀻𑀭𑁄 𑀩𑀸𑀮𑁂 𑀅𑀯𑁂𑀓𑁆𑀔𑀢𑀻’𑀢𑀺𑁇_  


1794\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑁄 𑀅𑀦𑀼𑀦𑁆𑀦𑀢𑁄 𑀅𑀦𑁄𑀦𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀉𑀦𑁆𑀦𑀢𑀸𑀯𑀦𑀢𑀺 𑀦 𑀓𑀭𑀡𑀻𑀬𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀩𑁆𑀩𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬 𑀘𑀽𑀍𑀅𑀲𑀼𑀪𑀤𑁆𑀤𑀸𑀬 𑀲𑀓𑀲𑀫𑀡𑁂 𑀧𑀭𑀺𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑀸𑀬—

1795\. _‘𑀮𑀸𑀪𑁂𑀦 𑀉𑀦𑁆𑀦𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀅𑀮𑀸𑀪𑁂𑀦 𑀘 𑀑𑀦𑀢𑁄;_  
_𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑁂𑀦 𑀏𑀓𑀝𑁆𑀞𑀸,_  
_𑀢𑀸𑀤𑀺𑀲𑀸 𑀲𑀫𑀡𑀸 𑀫𑀫𑀸’”𑀢𑀺𑁇_  


---

1796\. 𑀧𑀩𑁆𑀩𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇
