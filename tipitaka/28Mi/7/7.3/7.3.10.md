# 𑁭𑁇𑁩𑁇𑁧𑁦 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

### 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(254.)

1825. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀘𑀢𑀽𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀚𑀦𑀁 𑀲𑀗𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀘𑀢𑀲𑁆𑀲𑀦𑁆𑀦𑀁 𑀧𑀭𑀺𑀲𑀸𑀦𑀁 𑀫𑀸𑀦𑀲𑀁 𑀲𑀗𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1826. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀯𑀺𑀚𑀺𑀢𑁂 𑀘𑁄𑀭𑀸 𑀦 𑀉𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀸𑀫𑀭𑀸𑀕𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀦 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1827. _‘𑀯𑀺𑀢𑀓𑁆𑀓𑀽𑀧𑀲𑀫𑁂 𑀘 𑀬𑁄 𑀭𑀢𑁄,_  
_𑀅𑀲𑀼𑀪𑀁 𑀪𑀸𑀯𑀬𑀢𑁂 𑀲𑀤𑀸 𑀲𑀢𑁄;_  
_𑀏𑀲 𑀔𑁄 𑀩𑁆𑀬𑀦𑁆𑀢𑀺 𑀓𑀸𑀳𑀺𑀢𑀺,_  
_𑀏𑀲 𑀙𑁂𑀘𑁆𑀙𑀢𑀺 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑀦𑁆’𑀢𑀺𑁇_  


1828. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀤𑀺𑀯𑀲𑁂 𑀤𑀺𑀯𑀲𑁂 𑀲𑀫𑀼𑀤𑁆𑀤𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀁 𑀅𑀦𑀼𑀬𑀸𑀬𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀓𑀸𑀦𑀺 𑀯𑀺𑀘𑀺𑀦𑀫𑀸𑀦𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀁 𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀤𑀺𑀯𑀲𑁂 𑀤𑀺𑀯𑀲𑁂 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑁂 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀅𑀦𑀼𑀧𑀯𑀚𑁆𑀚𑀲𑁆𑀲 𑀤𑀺𑀯𑀲𑁄 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀻’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀅𑀗𑁆𑀕𑀼𑀢𑁆𑀢𑀭𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂—

1829. ‘𑀓𑀣𑀫𑁆𑀪𑀽𑀢𑀲𑁆𑀲 𑀫𑁂 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇

1830. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀸𑀭𑀓𑁆𑀔𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸𑀦𑀁 𑀩𑀸𑀳𑀺𑀭𑀸𑀦𑀁 𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁 𑀆𑀭𑀓𑁆𑀔𑀸𑀬 𑀲𑀢𑀺𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑁄 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1831. ‘𑀲𑀢𑀺𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀅𑀓𑀼𑀲𑀮𑀁 𑀧𑀚𑀳𑀢𑀺 𑀓𑀼𑀲𑀮𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀅𑀦𑀯𑀚𑁆𑀚𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀲𑀼𑀤𑁆𑀥𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀻’”𑀢𑀺𑁇

---

1832. 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀲𑀫𑁄𑁇



---

1833. 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

1834. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1835. _𑀧𑀣𑀯𑀻 𑀆𑀧𑁄 𑀘 𑀢𑁂𑀚𑁄 𑀘,_  
_𑀯𑀸𑀬𑁄 𑀘 𑀧𑀩𑁆𑀩𑀢𑁂𑀦 𑀘;_  
_𑀆𑀓𑀸𑀲𑁄 𑀘𑀦𑁆𑀤𑀲𑀽𑀭𑀺𑀬𑁄 𑀘,_  
_𑀲𑀓𑁆𑀓𑁄 𑀘 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑀸𑀢𑀺𑁇_
