# 𑁭𑁇𑁩𑁇𑁨 𑀆𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

* 𑀆𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(246.)

1763\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀆𑀧𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑁄 𑀲𑀼𑀲𑀡𑁆𑀞𑀺𑀢𑀫𑀓𑀫𑁆𑀧𑀺𑀢𑀫𑀮𑀼𑀍𑀇𑀢𑀲𑀪𑀸𑀯𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀼𑀳𑀦𑀮𑀧𑀦𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑁆𑀧𑁂𑀲𑀺𑀓𑀢𑀁 𑀅𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀲𑀼𑀲𑀡𑁆𑀞𑀺𑀢𑀫𑀓𑀫𑁆𑀧𑀺𑀢𑀫𑀮𑀼𑀍𑀇𑀢𑀲𑀪𑀸𑀯𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀘𑀸𑀭𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1764\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑁄 𑀲𑀻𑀢𑀮𑀲𑀪𑀸𑀯𑀲𑀡𑁆𑀞𑀺𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑁂𑀲𑀼 𑀔𑀦𑁆𑀢𑀺𑀫𑁂𑀢𑁆𑀢𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1765\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑁄 𑀅𑀲𑀼𑀘𑀺𑀁 𑀲𑀼𑀘𑀺𑀁 𑀓𑀭𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀕𑀸𑀫𑁂 𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀸 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀆𑀘𑀭𑀺𑀬𑁂 𑀆𑀘𑀭𑀺𑀬𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀦𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀓𑀸𑀭𑀺𑀦𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1766\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑁄 𑀩𑀳𑀼𑀚𑀦𑀧𑀢𑁆𑀣𑀺𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑁂𑀦 𑀲𑀢𑀢𑀁 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀫𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1767\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑁄 𑀦 𑀓𑀲𑁆𑀲𑀘𑀺 𑀅𑀳𑀺𑀢𑀫𑀼𑀧𑀤𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀭𑀪𑀡𑁆𑀟𑀦𑀓𑀮𑀳𑀯𑀺𑀕𑁆𑀕𑀳𑀯𑀺𑀯𑀸𑀤𑀭𑀺𑀢𑁆𑀢𑀚𑁆𑀛𑀸𑀦𑀅𑀭𑀢𑀺𑀚𑀦𑀦𑀁 𑀓𑀸𑀬𑀯𑀘𑀻𑀘𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀧𑀸𑀧𑀓𑀁 𑀦 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸, 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀓𑀡𑁆𑀳𑀚𑀸𑀢𑀓𑁂—

1768\. _‘𑀯𑀭𑀜𑁆𑀘𑁂 𑀫𑁂 𑀅𑀤𑁄 𑀲𑀓𑁆𑀓,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀫𑀺𑀲𑁆𑀲𑀭;_  
_𑀦 𑀫𑀦𑁄 𑀯𑀸 𑀲𑀭𑀻𑀭𑀁 𑀯𑀸,_  
_𑀫𑀁-𑀓𑀢𑁂 𑀲𑀓𑁆𑀓 𑀓𑀲𑁆𑀲𑀘𑀺;_  
_𑀓𑀤𑀸𑀘𑀺 𑀉𑀧𑀳𑀜𑁆𑀜𑁂𑀣,_  
_𑀏𑀢𑀁 𑀲𑀓𑁆𑀓 𑀯𑀭𑀁 𑀯𑀭𑁂’”𑀢𑀺𑁇_  


---

1769\. 𑀆𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
