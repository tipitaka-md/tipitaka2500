# 𑁭𑁇𑁩𑁇𑁪 𑀯𑀸𑀬𑀼𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

* 𑀯𑀸𑀬𑀼𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(248.)

1777\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀯𑀸𑀬𑀼𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼 𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀯𑀦𑀲𑀡𑁆𑀟𑀦𑁆𑀢𑀭𑀁 𑀅𑀪𑀺𑀯𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀯𑀭𑀓𑀼𑀲𑀼𑀫𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀯𑀦𑀦𑁆𑀢𑀭𑁂 𑀭𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1778\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀧𑀸𑀤𑀧𑀕𑀡𑁂 𑀫𑀣𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀯𑀦𑀦𑁆𑀢𑀭𑀕𑀢𑁂𑀦 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑁂𑀦 𑀓𑀺𑀮𑁂𑀲𑀸 𑀫𑀣𑀬𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1779\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼 𑀆𑀓𑀸𑀲𑁂 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀫𑀸𑀦𑀲𑀁 𑀲𑀜𑁆𑀘𑀸𑀭𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1780\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼 𑀕𑀦𑁆𑀥𑀁 𑀅𑀦𑀼𑀪𑀯𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀢𑁆𑀢𑀦𑁄 𑀲𑀻𑀮𑀯𑀭𑀲𑀼𑀭𑀪𑀺𑀕𑀦𑁆𑀥𑁄 𑀅𑀦𑀼𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1781\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼 𑀦𑀺𑀭𑀸𑀮𑀬𑁄 𑀅𑀦𑀺𑀓𑁂𑀢𑀯𑀸𑀲𑀻; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀦𑀺𑀭𑀸𑀮𑀬𑀫𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑁂𑀦 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀬𑀼𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀼𑀢𑁆𑀢𑀦𑀺𑀧𑀸𑀢𑁂—

1782\. _‘𑀲𑀦𑁆𑀣𑀯𑀸𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁,_  
_𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄;_  
_𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁,_  
_𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀦𑁆’”𑀢𑀺𑁇_  


---

1783\. 𑀯𑀸𑀬𑀼𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
