# 𑁭𑁇𑁩𑁇𑁭 𑀘𑀦𑁆𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

### 𑀘𑀦𑁆𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(251.)

1804\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑁄 𑀲𑀼𑀓𑁆𑀓𑀧𑀓𑁆𑀔𑁂 𑀉𑀤𑀬𑀦𑁆𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀟𑁆𑀠𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀘𑀸𑀭𑀲𑀻𑀮𑀕𑀼𑀡𑀯𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀕𑀫𑀸𑀥𑀺𑀕𑀫𑁂 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑁂 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸𑀬 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑀬 𑀚𑀸𑀕𑀭𑀺𑀬𑀸𑀦𑀼𑀬𑁄𑀕𑁂 𑀉𑀢𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀯𑀟𑁆𑀠𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1805\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑁄 𑀉𑀍𑀆𑀭𑀸𑀥𑀺𑀧𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀉𑀍𑀆𑀭𑁂𑀦 𑀙𑀦𑁆𑀤𑀸𑀥𑀺𑀧𑀢𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1806\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑁄 𑀦𑀺𑀲𑀸𑀬 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1807\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑁄 𑀯𑀺𑀫𑀸𑀦𑀓𑁂𑀢𑀼; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀻𑀮𑀓𑁂𑀢𑀼𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1808\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑁄 𑀆𑀬𑀸𑀘𑀺𑀢𑀧𑀢𑁆𑀣𑀺𑀢𑁄 𑀉𑀤𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀬𑀸𑀘𑀺𑀢𑀧𑀢𑁆𑀣𑀺𑀢𑁂𑀦 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂 ‘𑀘𑀦𑁆𑀤𑀽𑀧𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀣, 𑀅𑀧𑀓𑀲𑁆𑀲𑁂𑀯 𑀓𑀸𑀬𑀁 𑀅𑀧𑀓𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀘𑁆𑀘𑀦𑀯𑀓𑀸 𑀓𑀼𑀮𑁂𑀲𑀼 𑀅𑀧𑁆𑀧𑀕𑀩𑁆𑀪𑀸’”𑀢𑀺𑁇

---

1809\. 𑀘𑀦𑁆𑀤𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇
