# 𑁭𑁇𑁩𑁇𑁮 𑀲𑀽𑀭𑀺𑀬𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

### 𑀲𑀽𑀭𑀺𑀬𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(252.)

1810. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀲𑀩𑁆𑀩𑀁 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀲𑁄𑀲𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑀸 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀧𑀭𑀺𑀲𑁄𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1811. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀢𑀫𑀦𑁆𑀥𑀓𑀸𑀭𑀁 𑀯𑀺𑀥𑀫𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀁 𑀭𑀸𑀕𑀢𑀫𑀁 𑀤𑁄𑀲𑀢𑀫𑀁 𑀫𑁄𑀳𑀢𑀫𑀁 𑀫𑀸𑀦𑀢𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀢𑀫𑀁 𑀓𑀺𑀮𑁂𑀲𑀢𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀢𑀫𑀁 𑀯𑀺𑀥𑀫𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1812. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1813. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀭𑀁𑀲𑀺𑀫𑀸𑀮𑀻; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑀸𑀮𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1814. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑀁 𑀲𑀦𑁆𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀆𑀘𑀸𑀭𑀲𑀻𑀮𑀕𑀼𑀡𑀯𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀛𑀸𑀦𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀩𑀮𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑁃𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀳𑀺 𑀲𑀤𑁂𑀯𑀓𑁄 𑀮𑁄𑀓𑁄 𑀲𑀦𑁆𑀢𑀸𑀧𑀬𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1815. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀭𑀸𑀳𑀼𑀪𑀬𑀸 𑀪𑀻𑀢𑁄 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀯𑀺𑀲𑀫𑀓𑀦𑁆𑀢𑀸𑀭𑀯𑀺𑀧𑀸𑀓𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀓𑀺𑀮𑁂𑀲𑀚𑀸𑀮𑀚𑀝𑀺𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀗𑁆𑀖𑀸𑀝𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑁂 𑀓𑀼𑀧𑀣𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑁂 𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂 𑀲𑀢𑁆𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀳𑀢𑀸 𑀲𑀁𑀯𑁂𑀕𑀪𑀬𑁂𑀦 𑀫𑀸𑀦𑀲𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀙𑀝𑁆𑀞𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1816. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀓𑁂 𑀤𑀲𑁆𑀲𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀩𑀮𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑁃𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀮𑁄𑀓𑀺𑀬𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀥𑀫𑁆𑀫𑀸 𑀤𑀲𑁆𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀯𑀗𑁆𑀕𑀻𑀲𑁂𑀦—

1817. _‘𑀬𑀣𑀸𑀧𑀺 𑀲𑀽𑀭𑀺𑀬𑁄 𑀉𑀤𑀬𑀦𑁆𑀢𑁄,_  
_𑀭𑀽𑀧𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀲𑀼𑀘𑀺𑀜𑁆𑀘 𑀅𑀲𑀼𑀘𑀺𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀜𑁆𑀘𑀸𑀧𑀺 𑀧𑀸𑀧𑀓𑀁𑁇_  


1818. _𑀢𑀣𑀸 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀥𑀭𑁄,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀺𑀳𑀺𑀢𑀁 𑀚𑀦𑀁;_  
_𑀧𑀣𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀯𑀺𑀯𑀺𑀥𑀁,_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑁄𑀯𑀼𑀤𑀬𑀁 𑀬𑀣𑀸’”𑀢𑀺𑁇_  


---

1819. 𑀲𑀽𑀭𑀺𑀬𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
