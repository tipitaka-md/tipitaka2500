# 𑁭𑁇𑁩𑁇𑁧 𑀧𑀣𑀯𑀻𑀅𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

### 𑀧𑀣𑀯𑀻𑀅𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(245.)

1755. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀻 𑀇𑀝𑁆𑀞𑀸𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀺 𑀓𑀧𑁆𑀧𑀽𑀭𑀸𑀕𑀭𑀼𑀢𑀕𑀭𑀘𑀦𑁆𑀤𑀦𑀓𑀼𑀗𑁆𑀓𑀼𑀫𑀸𑀤𑀻𑀦𑀺 𑀆𑀓𑀺𑀭𑀦𑁆𑀢𑁂𑀧𑀺 𑀧𑀺𑀢𑁆𑀢𑀲𑁂𑀫𑁆𑀳𑀧𑀼𑀩𑁆𑀩𑀭𑀼𑀳𑀺𑀭𑀲𑁂𑀤𑀫𑁂𑀤𑀔𑁂𑀍𑀅𑀲𑀺𑀗𑁆𑀖𑀸𑀡𑀺𑀓𑀮𑀲𑀺𑀓𑀫𑀼𑀢𑁆𑀢𑀓𑀭𑀻𑀲𑀸𑀤𑀻𑀦𑀺 𑀆𑀓𑀺𑀭𑀦𑁆𑀢𑁂𑀧𑀺 𑀢𑀸𑀤𑀺𑀲𑀸𑀬𑁂𑀯; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀇𑀝𑁆𑀞𑀸𑀦𑀺𑀝𑁆𑀞𑁂 𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑁂 𑀬𑀲𑀸𑀬𑀲𑁂 𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀲𑀁𑀲𑀸𑀬 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑁂 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀢𑀸𑀤𑀺𑀦𑀸𑀬𑁂𑀯 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1756. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀻 𑀫𑀡𑁆𑀟𑀦𑀯𑀺𑀪𑀽𑀲𑀦𑀸𑀧𑀕𑀢𑀸 𑀲𑀓𑀕𑀦𑁆𑀥𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀯𑀺𑀪𑀽𑀲𑀦𑀸𑀧𑀕𑀢𑁂𑀦 𑀲𑀓𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀺𑀬𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1757. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀻 𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀸 𑀅𑀔𑀡𑁆𑀟𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀸 𑀅𑀲𑀼𑀲𑀺𑀭𑀸 𑀩𑀳𑀮𑀸 𑀖𑀦𑀸 𑀯𑀺𑀢𑁆𑀣𑀺𑀡𑁆𑀡𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀫𑀔𑀡𑁆𑀟𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀫𑀲𑀼𑀲𑀺𑀭𑀩𑀳𑀮𑀖𑀦𑀯𑀺𑀢𑁆𑀣𑀺𑀡𑁆𑀡𑀲𑀻𑀮𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀺𑀬𑀸 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1758. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀻 𑀕𑀸𑀫𑀦𑀺𑀕𑀫𑀦𑀕𑀭𑀚𑀦𑀧𑀤𑀭𑀼𑀓𑁆𑀔𑀧𑀩𑁆𑀩𑀢𑀦𑀤𑀻𑀢𑀍𑀆𑀓𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀫𑀺𑀕𑀧𑀓𑁆𑀔𑀺𑀫𑀦𑀼𑀚𑀦𑀭𑀦𑀸𑀭𑀺𑀕𑀡𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻𑀧𑀺 𑀅𑀓𑀺𑀮𑀸𑀲𑀼 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀑𑀯𑀤𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀲𑀼 𑀅𑀓𑀺𑀮𑀸𑀲𑀼𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀺𑀬𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1759. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀻 𑀅𑀦𑀼𑀦𑀬𑀧𑁆𑀧𑀝𑀺𑀖𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀦𑀼𑀦𑀬𑀧𑁆𑀧𑀝𑀺𑀖𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀣𑀯𑀺𑀲𑀫𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀣𑀯𑀺𑀬𑀸 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬 𑀘𑀽𑀍𑀅𑀲𑀼𑀪𑀤𑁆𑀤𑀸𑀬 𑀲𑀓𑀲𑀫𑀡𑁂 𑀧𑀭𑀺𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑀸𑀬—

1760. _‘𑀏𑀓𑀜𑁆𑀘𑁂 𑀩𑀸𑀳𑀁 𑀯𑀸𑀲𑀺𑀬𑀸,_  
_𑀢𑀘𑁆𑀙𑁂 𑀓𑀼𑀧𑀺𑀢𑀫𑀸𑀦𑀲𑀸;_  
_𑀏𑀓𑀜𑁆𑀘𑁂 𑀩𑀸𑀳𑀁 𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀆𑀮𑀺𑀫𑁆𑀧𑁂𑀬𑁆𑀬 𑀧𑀫𑁄𑀤𑀺𑀢𑀸𑁇_  


1761. _𑀅𑀫𑀼𑀲𑁆𑀫𑀺𑀁 𑀧𑀝𑀺𑀖𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀭𑀸𑀕𑁄 𑀅𑀲𑁆𑀫𑀺𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀧𑀣𑀯𑀻𑀲𑀫𑀘𑀺𑀢𑁆𑀢𑀸 𑀢𑁂,_  
_𑀢𑀸𑀤𑀺𑀲𑀸 𑀲𑀫𑀡𑀸 𑀫𑀫𑀸’”𑀢𑀺𑁇_  


---

1762. 𑀧𑀣𑀯𑀻𑀅𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀞𑀫𑁄𑁇
