# 𑁭𑁇𑁩𑁇𑁩 𑀢𑁂𑀚𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

### 𑀢𑁂𑀚𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(247.)

1770\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀢𑁂𑀚𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑁄 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀁 𑀟𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑁂 𑀢𑁂 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸 𑀯𑀸 𑀩𑀸𑀳𑀺𑀭𑀸 𑀯𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀇𑀝𑁆𑀞𑀸𑀦𑀺𑀝𑁆𑀞𑀸𑀭𑀫𑁆𑀫𑀡𑀸𑀦𑀼𑀪𑀯𑀦𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀜𑀸𑀡𑀕𑁆𑀕𑀺𑀦𑀸 𑀟𑀳𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1771\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑁄 𑀦𑀺𑀤𑁆𑀤𑀬𑁄 𑀅𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑁂𑀲𑀼 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀸 𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1772\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑁄 𑀲𑀻𑀢𑀁 𑀧𑀝𑀺𑀳𑀦𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀯𑀻𑀭𑀺𑀬𑀲𑀦𑁆𑀢𑀸𑀧𑀢𑁂𑀚𑀁 𑀅𑀪𑀺𑀚𑀦𑁂𑀢𑁆𑀯𑀸 𑀓𑀺𑀮𑁂𑀲𑀸 𑀧𑀝𑀺𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1773\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑁄 𑀅𑀦𑀼𑀦𑀬𑀧𑁆𑀧𑀝𑀺𑀖𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀉𑀡𑁆𑀳𑀫𑀪𑀺𑀚𑀦𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀦𑀼𑀦𑀬𑀧𑁆𑀧𑀝𑀺𑀖𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁂𑀦 𑀢𑁂𑀚𑁄𑀲𑀫𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1774\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑁄 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀁 𑀯𑀺𑀥𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀮𑁄𑀓𑀁 𑀤𑀲𑁆𑀲𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀥𑀓𑀸𑀭𑀁 𑀯𑀺𑀥𑀫𑀺𑀢𑁆𑀯𑀸 𑀜𑀸𑀡𑀸𑀮𑁄𑀓𑀁 𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀚𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀓𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀭𑀸𑀳𑀼𑀮𑀁 𑀑𑀯𑀤𑀦𑁆𑀢𑁂𑀦—

1775\. ‘𑀢𑁂𑀚𑁄𑀲𑀫𑀁, 𑀭𑀸𑀳𑀼𑀮, 𑀪𑀸𑀯𑀦𑀁 𑀪𑀸𑀯𑁂𑀳𑀺𑁇 𑀢𑁂𑀚𑁄𑀲𑀫𑀁 𑀳𑀺 𑀢𑁂, 𑀭𑀸𑀳𑀼𑀮, 𑀪𑀸𑀯𑀦𑀁 𑀪𑀸𑀯𑀬𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑀦𑀸𑀧𑀸𑀫𑀦𑀸𑀧𑀸 𑀨𑀲𑁆𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀻’”𑀢𑀺𑁇

---

1776\. 𑀢𑁂𑀚𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
