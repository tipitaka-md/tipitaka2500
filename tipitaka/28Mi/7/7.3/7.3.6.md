# 𑁭𑁇𑁩𑁇𑁬 𑀆𑀓𑀸𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀧𑀣𑀯𑀻𑀯𑀕𑁆𑀕

### 𑀆𑀓𑀸𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(250.)

1797. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀆𑀓𑀸𑀲𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑁄 𑀲𑀩𑁆𑀩𑀲𑁄 𑀅𑀕𑀬𑁆𑀳𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀲𑁄 𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺 𑀅𑀕𑀬𑁆𑀳𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1798. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑁄 𑀇𑀲𑀺𑀢𑀸𑀧𑀲𑀪𑀽𑀢𑀤𑀺𑀚𑀕𑀡𑀸𑀦𑀼𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀦𑀢𑁆𑀢𑀸’𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀫𑀸𑀦𑀲𑀁 𑀲𑀜𑁆𑀘𑀸𑀭𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1799. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑁄 𑀲𑀦𑁆𑀢𑀸𑀲𑀦𑀻𑀬𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀩𑁆𑀩𑀪𑀯𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀻𑀲𑀼 𑀫𑀸𑀦𑀲𑀁 𑀉𑀩𑁆𑀩𑁂𑀚𑀬𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀲𑁆𑀲𑀸𑀤𑁄 𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1800. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑁄 𑀅𑀦𑀦𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑁄 𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀦𑀦𑁆𑀢𑀲𑀻𑀮𑁂𑀦 𑀅𑀧𑀭𑀺𑀫𑀺𑀢𑀜𑀸𑀡𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1801. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑁄 𑀅𑀮𑀕𑁆𑀕𑁄 𑀅𑀲𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀧𑀮𑀺𑀩𑀼𑀤𑁆𑀥𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀼𑀮𑁂 𑀕𑀡𑁂 𑀮𑀸𑀪𑁂 𑀆𑀯𑀸𑀲𑁂 𑀧𑀮𑀺𑀩𑁄𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑁂 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑁂𑀲𑀼 𑀘 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀮𑀕𑁆𑀕𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀦𑀸𑀲𑀢𑁆𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀅𑀧𑀮𑀺𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀓𑀸𑀲𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀓𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀭𑀸𑀳𑀼𑀮𑀁 𑀑𑀯𑀤𑀦𑁆𑀢𑁂𑀦—

1802. ‘𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀭𑀸𑀳𑀼𑀮, 𑀆𑀓𑀸𑀲𑁄 𑀦 𑀓𑀢𑁆𑀣𑀘𑀺 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀭𑀸𑀳𑀼𑀮, 𑀆𑀓𑀸𑀲𑀲𑀫𑀁 𑀪𑀸𑀯𑀦𑀁 𑀪𑀸𑀯𑁂𑀳𑀺𑁇 𑀆𑀓𑀸𑀲𑀲𑀫𑀁 𑀳𑀺 𑀢𑁂, 𑀭𑀸𑀳𑀼𑀮, 𑀪𑀸𑀯𑀦𑀁 𑀪𑀸𑀯𑀬𑀢𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀫𑀦𑀸𑀧𑀸𑀫𑀦𑀸𑀧𑀸 𑀨𑀲𑁆𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀻’”𑀢𑀺𑁇

---

1803. 𑀆𑀓𑀸𑀲𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
