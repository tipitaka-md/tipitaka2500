# 𑁭𑁇𑁫𑁇𑁯 𑀲𑀧𑁆𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

* 𑀲𑀧𑁆𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(273.)

1923\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀲𑀧𑁆𑀧𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀧𑁆𑀧𑁄 𑀉𑀭𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑀲𑁆𑀲 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀜𑀸𑀬𑁂 𑀘𑀭𑀢𑀺, 𑀯𑀺𑀮𑀓𑁆𑀔𑀡𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺, 𑀲𑀮𑀓𑁆𑀔𑀡𑀁 𑀪𑀸𑀯𑁂𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀧𑁆𑀧𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1924\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀧𑁆𑀧𑁄 𑀘𑀭𑀫𑀸𑀦𑁄 𑀑𑀲𑀥𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑁄 𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑁂𑀦 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀧𑁆𑀧𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1925\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀧𑁆𑀧𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀢𑀧𑁆𑀧𑀢𑀺 𑀲𑁄𑀘𑀢𑀺 𑀘𑀺𑀦𑁆𑀢𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂, 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑁆𑀯𑀸 𑀅𑀭𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀧𑁆𑀧𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑁄𑀘𑀺𑀢𑀩𑁆𑀩𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀧𑀫𑀸𑀤𑁂𑀦 𑀫𑁂 𑀤𑀺𑀯𑀲𑁄 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑀺𑀢𑁄, 𑀦 𑀲𑁄 𑀧𑀼𑀦 𑀲𑀓𑁆𑀓𑀸 𑀮𑀤𑁆𑀥𑀼𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀧𑁆𑀧𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀮𑁆𑀮𑀸𑀝𑀺𑀬𑀚𑀸𑀢𑀓𑁂 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀓𑀺𑀦𑁆𑀦𑀭𑀸𑀦𑀁—

1926\. _‘𑀫𑀬𑁂𑀓𑀭𑀢𑁆𑀢𑀁 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀫𑁆𑀳 𑀮𑀼𑀤𑁆𑀤,_  
_𑀅𑀓𑀸𑀫𑀓𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀲𑀭𑀦𑁆𑀢𑀸;_  
_𑀢𑀫𑁂𑀓𑀭𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑀸,_  
_𑀲𑁄𑀘𑀸𑀫𑀲𑀸 𑀭𑀢𑁆𑀢𑀺 𑀧𑀼𑀦 𑀦 𑀳𑁂𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇_  


---

1927\. 𑀲𑀧𑁆𑀧𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀦𑀯𑀫𑁄𑁇
