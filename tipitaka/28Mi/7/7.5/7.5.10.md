# 𑁭𑁇𑁫𑁇𑁧𑁦 𑀅𑀚𑀕𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

* 𑀅𑀚𑀕𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(274.)

1928\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀅𑀚𑀕𑀭𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑀕𑀭𑁄 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀓𑀸𑀬𑁄 𑀩𑀳𑀽𑀧𑀺 𑀤𑀺𑀯𑀲𑁂 𑀊𑀦𑀽𑀤𑀭𑁄 𑀤𑀻𑀦𑀢𑀭𑁄 𑀓𑀼𑀘𑁆𑀙𑀺𑀧𑀽𑀭𑀁 𑀆𑀳𑀸𑀭𑀁 𑀦 𑀮𑀪𑀢𑀺, 𑀅𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄 𑀬𑁂𑀯 𑀬𑀸𑀯𑀤𑁂𑀯 𑀲𑀭𑀻𑀭𑀬𑀸𑀧𑀦𑀫𑀢𑁆𑀢𑀓𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑁄 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀸𑀘𑀭𑀺𑀬𑀧𑁆𑀧𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀭𑀧𑀺𑀡𑁆𑀟𑀫𑀼𑀧𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀤𑀺𑀦𑁆𑀦𑀧𑁆𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀺𑀲𑁆𑀲 𑀲𑀬𑀗𑁆𑀕𑀸𑀳𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀲𑁆𑀲 𑀤𑀼𑀮𑁆𑀮𑀪𑀁 𑀉𑀤𑀭𑀧𑀭𑀺𑀧𑀽𑀭𑀁 𑀆𑀳𑀸𑀭𑀁, 𑀅𑀧𑀺 𑀘 𑀅𑀢𑁆𑀣𑀯𑀲𑀺𑀓𑁂𑀦 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀜𑁆𑀘 𑀆𑀮𑁄𑀧𑁂 𑀅𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀉𑀤𑀓𑁂𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑀕𑀭𑀲𑁆𑀲 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸—

1929\. _‘𑀅𑀮𑁆𑀮𑀁 𑀲𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄,_  
_𑀦 𑀩𑀸𑀍𑀳𑀁 𑀲𑀼𑀳𑀺𑀢𑁄 𑀲𑀺𑀬𑀸;_  
_𑀊𑀦𑀽𑀤𑀭𑁄 𑀫𑀺𑀢𑀸𑀳𑀸𑀭𑁄,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂𑁇_  


1930\. _𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀜𑁆𑀘 𑀆𑀮𑁄𑀧𑁂,_  
_𑀅𑀪𑀼𑀢𑁆𑀯𑀸 𑀉𑀤𑀓𑀁 𑀧𑀺𑀯𑁂;_  
_𑀅𑀮𑀁 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀸𑀭𑀸𑀬,_  
_𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄’”𑀢𑀺𑁇_  


---

1931\. 𑀅𑀚𑀕𑀭𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀲𑀫𑁄𑁇



---

1932\. 𑀲𑀻𑀳𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

1933\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1934\. _𑀓𑁂𑀲𑀭𑀻 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑁄 𑀘,_  
_𑀧𑁂𑀡𑀸𑀳𑀺 𑀖𑀭𑀓𑀧𑁄𑀢𑀓𑁄;_  
_𑀉𑀮𑀽𑀓𑁄 𑀲𑀢𑀧𑀢𑁆𑀢𑁄 𑀘,_  
_𑀯𑀕𑁆𑀕𑀼𑀮𑀺 𑀘 𑀚𑀮𑀽𑀧𑀺𑀓𑀸;_  
_𑀲𑀧𑁆𑀧𑁄 𑀅𑀚𑀕𑀭𑁄 𑀘𑁂𑀯,_  
_𑀯𑀕𑁆𑀕𑁄 𑀢𑁂𑀦 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀻𑀢𑀺𑁇_
