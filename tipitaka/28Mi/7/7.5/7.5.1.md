# 𑁭𑁇𑁫𑁇𑁧 𑀲𑀻𑀳𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

### 𑀲𑀻𑀳𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(265.)

1887. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀲𑀻𑀳𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑁄 𑀦𑀸𑀫 𑀲𑁂𑀢𑀯𑀺𑀫𑀮𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀧𑀡𑁆𑀟𑀭𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑁂𑀢𑀯𑀺𑀫𑀮𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀧𑀡𑁆𑀟𑀭𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀩𑁆𑀬𑀧𑀕𑀢𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1888. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑁄 𑀘𑀢𑀼𑀘𑀭𑀡𑁄 𑀯𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀘𑀸𑀭𑀻; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀘𑀢𑀼𑀭𑀺𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀘𑀭𑀡𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1889. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑁄 𑀅𑀪𑀺𑀭𑀽𑀧𑀭𑀼𑀘𑀺𑀭𑀓𑁂𑀲𑀭𑀻; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀪𑀺𑀭𑀽𑀧𑀭𑀼𑀘𑀺𑀭𑀲𑀻𑀮𑀓𑁂𑀲𑀭𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1890. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑁄 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑁂𑀧𑀺 𑀦 𑀓𑀲𑁆𑀲𑀘𑀺 𑀑𑀦𑀫𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑁂𑀧𑀺 𑀦 𑀓𑀲𑁆𑀲𑀘𑀺 𑀑𑀦𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1891. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑁄 𑀲𑀧𑀤𑀸𑀦𑀪𑀓𑁆𑀔𑁄 𑀬𑀲𑁆𑀫𑀺𑀁 𑀑𑀓𑀸𑀲𑁂 𑀦𑀺𑀧𑀢𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀪𑀓𑁆𑀔𑀬𑀢𑀺, 𑀦 𑀯𑀭𑀫𑀁𑀲𑀁 𑀯𑀺𑀘𑀺𑀦𑀸𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀧𑀤𑀸𑀦𑀪𑀓𑁆𑀔𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀓𑀼𑀮𑀸𑀦𑀺 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀧𑀼𑀩𑁆𑀩𑀕𑁂𑀳𑀁 𑀳𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀪𑁄𑀚𑀦𑀁 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑀩𑁆𑀩𑀁, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀑𑀓𑀸𑀲𑁂 𑀓𑀩𑀍𑀅𑀁 𑀆𑀤𑀻𑀬𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀑𑀓𑀸𑀲𑁂 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀭𑀻𑀭𑀬𑀸𑀧𑀦𑀢𑁆𑀣𑀁, 𑀦 𑀯𑀭𑀪𑁄𑀚𑀦𑀁 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1892. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑁄 𑀅𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀪𑀓𑁆𑀔𑁄, 𑀲𑀓𑀺𑀁 𑀕𑁄𑀘𑀭𑀁 𑀪𑀓𑁆𑀔𑀬𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀦 𑀢𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀓𑀸𑀭𑀧𑀭𑀺𑀪𑁄𑀕𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑀲𑁆𑀲 𑀙𑀝𑁆𑀞𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1893. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑁄 𑀪𑁄𑀚𑀦𑀁 𑀅𑀮𑀤𑁆𑀥𑀸 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺, 𑀮𑀤𑁆𑀥𑀸𑀧𑀺 𑀪𑁄𑀚𑀦𑀁 𑀅𑀕𑀥𑀺𑀢𑁄 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀦𑁆𑀦𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀪𑁄𑀚𑀦𑀁 𑀅𑀮𑀤𑁆𑀥𑀸 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑀁, 𑀮𑀤𑁆𑀥𑀸𑀧𑀺 𑀪𑁄𑀚𑀦𑀁 𑀅𑀕𑀥𑀺𑀢𑁂𑀦 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁂𑀦 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀦𑁆𑀦𑁂𑀦 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑁂𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀳𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂 𑀣𑁂𑀭𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀧𑀭𑀺𑀓𑀺𑀢𑁆𑀢𑀬𑀫𑀸𑀦𑁂𑀦—

1894. ‘𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀦, 𑀇𑀢𑀭𑀻𑀢𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀳𑁂𑀢𑀼 𑀅𑀦𑁂𑀲𑀦𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀮𑀤𑁆𑀥𑀸 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀦 𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀢𑀺, 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀅𑀕𑀥𑀺𑀢𑁄 𑀅𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀦𑁆𑀦𑁄 𑀆𑀤𑀻𑀦𑀯𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀧𑀜𑁆𑀜𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀻’”𑀢𑀺𑁇

---

1895. 𑀲𑀻𑀳𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀞𑀫𑁄𑁇
