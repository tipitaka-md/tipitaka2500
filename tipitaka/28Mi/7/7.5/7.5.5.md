# 𑁭𑁇𑁫𑁇𑁫 𑀉𑀮𑀽𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

### 𑀉𑀮𑀽𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(269.)

1908\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀉𑀮𑀽𑀓𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀮𑀽𑀓𑁄 𑀓𑀸𑀓𑁂𑀳𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁄, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀓𑀸𑀓𑀲𑀗𑁆𑀖𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀩𑀳𑀽𑀧𑀺 𑀓𑀸𑀓𑁂 𑀳𑀦𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀅𑀜𑁆𑀜𑀸𑀡𑁂𑀦 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀏𑀓𑁂𑀦 𑀭𑀳𑁄 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀲𑀫𑁆𑀧𑀫𑀤𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁, 𑀫𑀽𑀮𑀢𑁄 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀮𑀽𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1909\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀮𑀽𑀓𑁄 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀭𑀸𑀫𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀭𑀢𑁂𑀦𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀮𑀽𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂—

1910\. ‘𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸𑀭𑀸𑀫𑁄 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀭𑀢𑁄 “𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, “𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀻’”𑀢𑀺𑁇

---

1911\. 𑀉𑀮𑀽𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇
