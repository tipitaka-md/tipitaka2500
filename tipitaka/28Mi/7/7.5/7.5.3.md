# 𑁭𑁇𑁫𑁇𑁩 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

* 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(267.)

1901\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀸𑀬 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀸 𑀲𑀓𑀧𑀢𑀺𑀫𑁆𑀳𑀺 𑀉𑀲𑀽𑀬𑀸𑀬 𑀙𑀸𑀧𑀓𑁂 𑀦 𑀧𑁄𑀲𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀲𑀓𑀫𑀦𑁂 𑀓𑀺𑀮𑁂𑀲𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀉𑀲𑀽𑀬𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁, 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑁆𑀫𑀸𑀲𑀁𑀯𑀭𑀲𑀼𑀲𑀺𑀭𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀫𑀦𑁄𑀤𑁆𑀯𑀸𑀭𑁂 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀸𑀬 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1902\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀸 𑀧𑀯𑀦𑁂 𑀤𑀺𑀯𑀲𑀁 𑀕𑁄𑀘𑀭𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀬𑀁 𑀧𑀓𑁆𑀔𑀺𑀕𑀡𑀁 𑀉𑀧𑁂𑀢𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀕𑀼𑀢𑁆𑀢𑀺𑀬𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀏𑀓𑀓𑁂𑀦 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸, 𑀢𑀢𑁆𑀭 𑀭𑀢𑀺𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁂𑀦 𑀉𑀧𑀯𑀸𑀤𑀪𑀬𑀧𑀭𑀺𑀭𑀓𑁆𑀔𑀡𑀸𑀬 𑀲𑀁𑀖𑀁 𑀑𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀭𑀓𑁆𑀔𑀺𑀢𑁂𑀦 𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀸𑀬 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺𑀦𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂—

1903\. _‘𑀲𑁂𑀯𑁂𑀣 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺,_  
_𑀘𑀭𑁂𑀬𑁆𑀬 𑀲𑀁𑀬𑁄𑀚𑀦𑀯𑀺𑀧𑁆𑀧𑀫𑁄𑀓𑁆𑀔𑀸;_  
_𑀲𑀘𑁂 𑀭𑀢𑀺𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀢𑀢𑁆𑀣,_  
_𑀲𑀁𑀖𑁂 𑀯𑀲𑁂 𑀭𑀓𑁆𑀔𑀺𑀢𑀢𑁆𑀢𑁄 𑀲𑀢𑀻𑀫𑀸’”𑀢𑀺𑁇_  


---

1904\. 𑀧𑁂𑀡𑀸𑀳𑀺𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
