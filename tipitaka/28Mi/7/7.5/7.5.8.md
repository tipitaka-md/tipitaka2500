# 𑁭𑁇𑁫𑁇𑁮 𑀚𑀮𑀽𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

* 𑀚𑀮𑀽𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(272.)

1920\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀚𑀮𑀽𑀓𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀮𑀽𑀓𑀸 𑀬𑀢𑁆𑀣 𑀅𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀤𑀍𑀳𑀁 𑀅𑀮𑁆𑀮𑀻𑀬𑀺𑀢𑁆𑀯𑀸 𑀭𑀼𑀳𑀺𑀭𑀁 𑀧𑀺𑀯𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑀲𑁆𑀫𑀺𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀮𑁆𑀮𑀻𑀬𑀢𑀺, 𑀢𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀯𑀡𑁆𑀡𑀢𑁄 𑀘 𑀲𑀡𑁆𑀞𑀸𑀦𑀢𑁄 𑀘 𑀤𑀺𑀲𑀢𑁄 𑀘 𑀑𑀓𑀸𑀲𑀢𑁄 𑀘 𑀧𑀭𑀺𑀘𑁆𑀙𑁂𑀤𑀢𑁄 𑀘 𑀮𑀺𑀗𑁆𑀕𑀢𑁄 𑀘 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀤𑀍𑀳𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑁂𑀦𑁂𑀯𑀸𑀭𑀫𑁆𑀫𑀡𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑀫𑀲𑁂𑀘𑀦𑀓𑀁 𑀧𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀮𑀽𑀓𑀸𑀬 𑀏𑀓𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁂𑀦—

1921\. _‘𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬;_  
_𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀲𑀫𑀲𑁂𑀘𑀦𑀦𑁆’”𑀢𑀺𑁇_  


---

1922\. 𑀚𑀮𑀽𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
