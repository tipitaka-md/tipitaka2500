# 𑁭𑁇𑁫𑁇𑁭 𑀯𑀕𑁆𑀕𑀼𑀮𑀺𑀗𑁆𑀕𑀧𑀜𑁆𑀳

* 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

* 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

* 𑀯𑀕𑁆𑀕𑀼𑀮𑀺𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(271.)

1915\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀯𑀕𑁆𑀕𑀼𑀮𑀺𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀤𑁆𑀯𑁂 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀕𑁆𑀕𑀼𑀮𑀺 𑀕𑁂𑀳𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀢𑀺, 𑀦 𑀢𑀢𑁆𑀣 𑀧𑀮𑀺𑀩𑀼𑀤𑁆𑀥𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀧𑀤𑀸𑀦𑀁 𑀯𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀮𑀸𑀪𑁂𑀦 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀢𑀢𑁆𑀣 𑀧𑀮𑀺𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀕𑁆𑀕𑀼𑀮𑀺𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1916\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀕𑁆𑀕𑀼𑀮𑀺 𑀧𑀭𑀕𑁂𑀳𑁂 𑀯𑀲𑀫𑀸𑀦𑁄 𑀦 𑀢𑁂𑀲𑀁 𑀧𑀭𑀺𑀳𑀸𑀦𑀺𑀁 𑀓𑀭𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀺𑀬𑀸𑀘𑀦𑀸𑀬 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀢𑀸𑀬 𑀯𑀸 𑀓𑀸𑀬𑀤𑁄𑀲𑀩𑀳𑀼𑀮𑀢𑀸𑀬 𑀯𑀸 𑀅𑀢𑀺𑀪𑀸𑀡𑀺𑀢𑀸𑀬 𑀯𑀸 𑀲𑀫𑀸𑀦𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀢𑀸𑀬 𑀯𑀸 𑀦 𑀢𑁂𑀲𑀁 𑀓𑁄𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 𑀓𑀭𑀡𑀻𑀬𑁄, 𑀦𑀧𑀺 𑀢𑁂𑀲𑀁 𑀫𑀽𑀮𑀓𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀳𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁, 𑀲𑀩𑁆𑀩𑀣𑀸 𑀯𑀟𑁆𑀠𑀺𑀬𑁂𑀯 𑀇𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀕𑁆𑀕𑀼𑀮𑀺𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀤𑀻𑀖𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑁂 𑀮𑀓𑁆𑀔𑀡𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁂—

1917\. _‘𑀲𑀼𑀤𑁆𑀥𑀸𑀬 𑀲𑀻𑀮𑁂𑀦 𑀲𑀼𑀢𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀺𑀬𑀸,_  
_𑀘𑀸𑀕𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀩𑀳𑀽𑀳𑀺 𑀲𑀸𑀥𑀼𑀳𑀺;_  
_𑀥𑀦𑁂𑀦 𑀥𑀜𑁆𑀜𑁂𑀦 𑀘 𑀔𑁂𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼𑀦𑀸,_  
_𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀤𑀸𑀭𑁂𑀳𑀺 𑀘𑀢𑀼𑀧𑁆𑀧𑀤𑁂𑀳𑀺 𑀘𑁇_  


1918\. _𑀜𑀸𑀢𑀻𑀳𑀺 𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀘 𑀩𑀦𑁆𑀥𑀯𑁂𑀳𑀺,_  
_𑀩𑀮𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦 𑀲𑀼𑀔𑁂𑀦 𑀘𑀽𑀪𑀬𑀁;_  
_𑀓𑀣𑀁 𑀦 𑀳𑀸𑀬𑁂𑀬𑁆𑀬𑀼𑀁 𑀧𑀭𑁂𑀢𑀺 𑀇𑀘𑁆𑀙𑀢𑀺,_  
_𑀅𑀢𑁆𑀣𑀲𑀫𑀺𑀤𑁆𑀥𑀺𑀜𑁆𑀘 𑀧𑀦𑀸𑀪𑀺𑀓𑀗𑁆𑀔𑀢𑀻’”𑀢𑀺𑁇_  


---

1919\. 𑀯𑀕𑁆𑀕𑀼𑀮𑀺𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇
