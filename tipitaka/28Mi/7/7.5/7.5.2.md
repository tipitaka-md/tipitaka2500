# 𑁭𑁇𑁫𑁇𑁨 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀑𑀧𑀫𑁆𑀫𑀓𑀣𑀸𑀧𑀜𑁆𑀳

### 𑀲𑀻𑀳𑀯𑀕𑁆𑀕

### 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(266.)

1896\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀲𑁆𑀲 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀢𑀻𑀡𑀺 𑀅𑀗𑁆𑀕𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀻𑀢𑀺”? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑁄 𑀬𑀸𑀯 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀁 𑀦 𑀯𑀺𑀚𑀳𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑀸𑀯 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀦 𑀯𑀺𑀚𑀳𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀲𑁆𑀲 𑀧𑀞𑀫𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1897\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑁄 𑀲𑁂𑀯𑀸𑀮𑀧𑀡𑀓𑀪𑀓𑁆𑀔𑁄, 𑀢𑁂𑀦 𑀘 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀢𑀸𑀬 𑀘 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀬𑀸 𑀩𑀮𑁂𑀦 𑀘 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀬𑀣𑀸𑀮𑀸𑀪𑀲𑀦𑁆𑀢𑁄𑀲𑁄 𑀓𑀭𑀡𑀻𑀬𑁄, 𑀬𑀣𑀸𑀮𑀸𑀪𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀻 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁄 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺 𑀲𑀻𑀮𑁂𑀦, 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺 𑀲𑀫𑀸𑀥𑀺𑀦𑀸, 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀬, 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸, 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦, 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺 𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1898\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑁄 𑀧𑀸𑀡𑁂 𑀦 𑀯𑀺𑀳𑁂𑀞𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀕𑀺𑀦𑀸 𑀬𑁄𑀕𑀸𑀯𑀘𑀭𑁂𑀦 𑀦𑀺𑀳𑀺𑀢𑀤𑀡𑁆𑀟𑁂𑀦 𑀦𑀺𑀳𑀺𑀢𑀲𑀢𑁆𑀣𑁂𑀦 𑀮𑀚𑁆𑀚𑀺𑀦𑀸 𑀤𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀦 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀪𑀽𑀢𑀳𑀺𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑀸 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀇𑀤𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑀁 𑀅𑀗𑁆𑀕𑀁 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀚𑀸𑀢𑀓𑁂—

1899\. _‘𑀬𑁄 𑀦 𑀳𑀦𑁆𑀢𑀺 𑀦 𑀖𑀸𑀢𑁂𑀢𑀺,_  
_𑀦 𑀚𑀺𑀦𑀸𑀢𑀺 𑀦 𑀚𑀸𑀧𑀬𑁂;_  
_𑀫𑁂𑀢𑁆𑀢𑀁𑀲𑁄 𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑁂𑀲𑀼,_  
_𑀯𑁂𑀭𑀁 𑀢𑀲𑁆𑀲 𑀦 𑀓𑁂𑀦𑀘𑀻’”𑀢𑀺𑁇_  


---

1900\. 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
