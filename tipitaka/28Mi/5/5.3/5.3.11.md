# 𑁫𑁇𑁩𑁇𑁧𑁧 𑀅𑀳𑀺𑀁𑀲𑀸𑀦𑀺𑀕𑁆𑀕𑀳𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀧𑀡𑀸𑀫𑀺𑀢𑀯𑀕𑁆𑀕

* 𑀅𑀳𑀺𑀁𑀲𑀸𑀦𑀺𑀕𑁆𑀕𑀳𑀧𑀜𑁆𑀳

(148.)

843\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀅𑀳𑀺𑀁𑀲𑀸𑀬 𑀘𑀭 𑀮𑁄𑀓𑁂, 𑀧𑀺𑀬𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀫𑀁𑀫𑀺𑀯𑀸’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀁, 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀧𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀦𑁆’𑀢𑀺𑁇 𑀦𑀺𑀕𑁆𑀕𑀳𑁄 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀳𑀢𑁆𑀣𑀘𑁆𑀙𑁂𑀤𑁄 𑀧𑀸𑀤𑀘𑁆𑀙𑁂𑀤𑁄 𑀯𑀥𑁄 𑀩𑀦𑁆𑀥𑀦𑀁 𑀓𑀸𑀭𑀡𑀸 𑀫𑀸𑀭𑀡𑀁 𑀲𑀦𑁆𑀢𑀢𑀺𑀯𑀺𑀓𑁄𑀧𑀦𑀁, 𑀦 𑀏𑀢𑀁 𑀯𑀘𑀦𑀁 𑀪𑀕𑀯𑀢𑁄 𑀬𑀼𑀢𑁆𑀢𑀁, 𑀦 𑀘 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀢𑀺 𑀏𑀢𑀁 𑀯𑀘𑀦𑀁 𑀯𑀢𑁆𑀢𑀼𑀁𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀅𑀳𑀺𑀁𑀲𑀸𑀬 𑀘𑀭 𑀮𑁄𑀓𑁂, 𑀧𑀺𑀬𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀫𑀁𑀫𑀺𑀯𑀸’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀁, 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀧𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀁, 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀧𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀦𑁆’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀅𑀳𑀺𑀁𑀲𑀸𑀬 𑀘𑀭 𑀮𑁄𑀓𑁂, 𑀧𑀺𑀬𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀫𑀁𑀫𑀺𑀯𑀸’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

844\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀅𑀳𑀺𑀁𑀲𑀸𑀬 𑀘𑀭 𑀮𑁄𑀓𑁂, 𑀧𑀺𑀬𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀫𑀁𑀫𑀺𑀯𑀸’𑀢𑀺, 𑀪𑀡𑀺𑀢𑀜𑁆𑀘—  ‘𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀁, 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀧𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀦𑁆’𑀢𑀺𑁇 ‘𑀅𑀳𑀺𑀁𑀲𑀸𑀬 𑀘𑀭 𑀮𑁄𑀓𑁂, 𑀧𑀺𑀬𑁄 𑀳𑁄𑀳𑀺𑀲𑀺 𑀫𑀁𑀫𑀺𑀯𑀸’𑀢𑀺 𑀲𑀩𑁆𑀩𑁂𑀲𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀅𑀦𑀼𑀫𑀢𑀁 𑀏𑀢𑀁, 𑀏𑀲𑀸 𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺, 𑀏𑀲𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀥𑀫𑁆𑀫𑁄 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀳𑀺𑀁𑀲𑀸𑀮𑀓𑁆𑀔𑀡𑁄, 𑀲𑀪𑀸𑀯𑀯𑀘𑀦𑀁 𑀏𑀢𑀁𑁇 𑀬𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀆𑀳—  ‘𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀁, 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑁂 𑀧𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑀦𑁆’𑀢𑀺, 𑀪𑀸𑀲𑀸 𑀏𑀲𑀸, 𑀉𑀤𑁆𑀥𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀮𑀻𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄, 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁄 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄, 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀦𑀭𑀺𑀬𑁄 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄 𑀅𑀭𑀺𑀬𑁄 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀘𑁄𑀭𑁄 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄, 𑀅𑀘𑁄𑀭𑁄 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

845\. “𑀳𑁄𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀤𑀸𑀦𑀺 𑀢𑁆𑀯𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀢𑁄𑀲𑀺 𑀫𑀫 𑀯𑀺𑀲𑀬𑀁, 𑀬𑀫𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺, 𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀉𑀧𑀕𑀢𑁄𑁇 𑀘𑁄𑀭𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁂𑀦 𑀓𑀣𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺? “𑀘𑁄𑀭𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁂𑀦 𑀏𑀯𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄, 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑀻𑀬𑁄 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄, 𑀤𑀡𑁆𑀟𑀦𑀻𑀬𑁄 𑀤𑀡𑁆𑀟𑁂𑀢𑀩𑁆𑀩𑁄, 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄, 𑀩𑀦𑁆𑀥𑀦𑀻𑀬𑁄 𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑁄, 𑀖𑀸𑀢𑀦𑀻𑀬𑁄 𑀖𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 “𑀬𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀘𑁄𑀭𑀸𑀦𑀁 𑀖𑀸𑀢𑀦𑀁, 𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀅𑀦𑀼𑀫𑀢𑀦𑁆”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀘𑁄𑀭𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀻𑀬𑁄 𑀅𑀦𑀼𑀫𑀢𑁄 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀦𑁆”𑀢𑀺? “𑀬𑁄 𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀖𑀸𑀢𑀻𑀬𑀢𑀺, 𑀦 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀅𑀦𑀼𑀫𑀢𑀺𑀬𑀸 𑀖𑀸𑀢𑀻𑀬𑀢𑀺, 𑀲𑀬𑀁𑀓𑀢𑁂𑀦 𑀲𑁄 𑀖𑀸𑀢𑀻𑀬𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀦𑀼𑀲𑀸𑀲𑀻𑀬𑀢𑀺, 𑀲𑀓𑁆𑀓𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀁 𑀅𑀓𑀸𑀭𑀓𑀁 𑀅𑀦𑀧𑀭𑀸𑀥𑀁 𑀯𑀻𑀣𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀖𑀸𑀢𑀬𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦 𑀲𑀓𑁆𑀓𑀸, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺? “𑀅𑀓𑀸𑀭𑀓𑀢𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦 𑀘𑁄𑀭𑁄 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀅𑀦𑀼𑀫𑀢𑀺𑀬𑀸 𑀳𑀜𑁆𑀜𑀢𑀺, 𑀲𑀬𑀁𑀓𑀢𑁂𑀦 𑀲𑁄 𑀳𑀜𑁆𑀜𑀢𑀺, 𑀓𑀺𑀁 𑀧𑀦𑁂𑀢𑁆𑀣 𑀅𑀦𑀼𑀲𑀸𑀲𑀓𑁄 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑁄𑀲𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺 𑀲𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
