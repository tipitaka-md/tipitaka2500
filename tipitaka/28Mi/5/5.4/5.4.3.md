# 𑁫𑁇𑁪𑁇𑁩 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀕𑀭𑀼𑀮𑀳𑀼𑀪𑀸𑀯𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀕𑀭𑀼𑀮𑀳𑀼𑀪𑀸𑀯𑀧𑀜𑁆𑀳

(152.)

869. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀏𑀓𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑁂𑀲𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀦𑁆’𑀢𑀺𑁇 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁄 𑀧𑀦𑁂𑀢𑁆𑀣 𑀯𑀺𑀲𑁂𑀲𑁄, 𑀓𑀺𑀁𑀓𑀸𑀭𑀡𑀁, 𑀬𑀜𑁆𑀘𑁂𑀓𑁂𑀦 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀉𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀜𑁆𑀘𑁂𑀓𑁂𑀦 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀲𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄 𑀳𑁄𑀢𑀺? 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀏𑀓𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑁂𑀲𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀏𑀓𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑁂𑀲𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀦𑁆’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

870. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀪𑀡𑀺𑀢𑀜𑁆𑀘—  ‘𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁂 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀏𑀓𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑁂𑀲𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀦𑁆’𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀯𑀢𑁆𑀣𑀼𑀯𑀲𑁂𑀦 𑀕𑀭𑀼𑀓𑀮𑀳𑀼𑀓𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀥 𑀓𑁄𑀘𑀺 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀭𑀲𑁆𑀲 𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑀳𑀸𑀭𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀓𑀺𑀁 𑀤𑀡𑁆𑀟𑀁 𑀥𑀸𑀭𑁂𑀣𑀸”𑀢𑀺? “𑀬𑀤𑀺 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀳—  ‘𑀦𑀓𑁆𑀔𑀫𑀸𑀫𑀻’𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀫𑀬𑀁 𑀅𑀓𑁆𑀔𑀫𑀫𑀸𑀦𑁂 𑀓𑀳𑀸𑀧𑀡𑀁 𑀳𑀭𑀸𑀧𑁂𑀫𑀸”𑀢𑀺𑁇 “𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁄𑀬𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀢𑀯 𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑀳𑀸𑀭𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲 𑀧𑀦 𑀓𑁄 𑀤𑀡𑁆𑀟𑁄”𑀢𑀺? “𑀳𑀢𑁆𑀣𑀫𑁆𑀧𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀙𑁂𑀤𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀫, 𑀧𑀸𑀤𑀫𑁆𑀧𑀺 𑀙𑁂𑀤𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀫, 𑀬𑀸𑀯 𑀲𑀻𑀲𑀁 𑀓𑀍𑀈𑀭𑀘𑁆𑀙𑁂𑀚𑁆𑀚𑀁 𑀙𑁂𑀤𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀫, 𑀲𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀢𑀁 𑀕𑁂𑀳𑀁 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀫, 𑀉𑀪𑀢𑁄𑀧𑀓𑁆𑀔𑁂 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀁 𑀓𑀼𑀮𑀁 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀓𑁄 𑀧𑀦𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀺𑀲𑁂𑀲𑁄, 𑀓𑀺𑀁𑀓𑀸𑀭𑀡𑀁, 𑀬𑀁 𑀏𑀓𑀲𑁆𑀲 𑀧𑀸𑀡𑀺𑀧𑁆𑀧𑀳𑀸𑀭𑁂 𑀲𑀼𑀔𑀼𑀫𑁄 𑀓𑀳𑀸𑀧𑀡𑁄 𑀤𑀡𑁆𑀟𑁄, 𑀬𑀁 𑀢𑀯 𑀧𑀸𑀡𑀺𑀧𑁆𑀧𑀳𑀸𑀭𑁂 𑀳𑀢𑁆𑀣𑀘𑁆𑀙𑁂𑀚𑁆𑀚𑀁 𑀧𑀸𑀤𑀘𑁆𑀙𑁂𑀚𑁆𑀚𑀁 𑀬𑀸𑀯 𑀓𑀍𑀈𑀭𑀘𑁆𑀙𑁂𑀚𑁆𑀚𑀁 𑀲𑀩𑁆𑀩𑀕𑁂𑀳𑀸𑀤𑀸𑀦𑀁 𑀉𑀪𑀢𑁄𑀧𑀓𑁆𑀔𑁂 𑀬𑀸𑀯 𑀲𑀢𑁆𑀢𑀫𑀓𑀼𑀮𑀸 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄”𑀢𑀺? “𑀫𑀦𑀼𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀯𑀢𑁆𑀣𑀼𑀯𑀲𑁂𑀦 𑀕𑀭𑀼𑀓𑀮𑀳𑀼𑀓𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
