# 𑁫𑁇𑁪𑁇𑁪 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀢𑀸𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀢𑀸𑀧𑀜𑁆𑀳

(153.)

872. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀢𑀸𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁂 ‘𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀦𑀺𑀬𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀩𑁄𑀥𑀺 𑀦𑀺𑀬𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀕𑁆𑀕𑀲𑀸𑀯𑀓𑀸 𑀦𑀺𑀬𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑀺𑀬𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀦𑀺𑀬𑀢𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀢𑀼𑀲𑀺𑀢𑁂 𑀓𑀸𑀬𑁂 𑀞𑀺𑀢𑁄 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀯𑀺𑀮𑁄𑀓𑀦𑀸𑀦𑀺 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀓𑀸𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀤𑀻𑀧𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀤𑁂𑀲𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀓𑀼𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀚𑀦𑁂𑀢𑁆𑀢𑀺𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀆𑀬𑀼𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀫𑀸𑀲𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀻’𑀢𑀺𑁇 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂 𑀜𑀸𑀡𑁂 𑀩𑀼𑀚𑁆𑀛𑀦𑀁 𑀦𑀢𑁆𑀣𑀺, 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂 𑀜𑀸𑀡𑁂 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀦𑀺𑀫𑁂𑀲𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀺 𑀆𑀕𑀫𑁂𑀢𑀼𑀁, 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀫𑀸𑀦𑀲𑀁𑁇 𑀓𑀲𑁆𑀫𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀓𑀸𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺 ‘𑀓𑀫𑁆𑀳𑀺 𑀓𑀸𑀮𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂 𑀜𑀸𑀡𑁂 𑀩𑀼𑀚𑁆𑀛𑀦𑀁 𑀦𑀢𑁆𑀣𑀺, 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂 𑀜𑀸𑀡𑁂 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀦𑀺𑀫𑁂𑀲𑀦𑁆𑀢𑀭𑀫𑁆𑀧𑀺 𑀆𑀕𑀫𑁂𑀢𑀼𑀁, 𑀓𑀲𑁆𑀫𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀓𑀼𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺 ‘𑀓𑀫𑁆𑀳𑀺 𑀓𑀼𑀮𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀦𑀺𑀬𑀢𑀸, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀓𑀼𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀓𑀼𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀦𑀺𑀬𑀢𑀸’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

873. “𑀦𑀺𑀬𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄, 𑀓𑀼𑀮𑀜𑁆𑀘 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺𑁇 𑀓𑀺𑀦𑁆𑀢𑀺 𑀧𑀦 𑀓𑀼𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺 ‘𑀬𑁂 𑀫𑁂 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄, 𑀢𑁂 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀤𑀸𑀳𑀼 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀓𑀼𑀮𑀁 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑀺𑁇

874. 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀅𑀦𑀸𑀕𑀢𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁? 𑀯𑀸𑀡𑀺𑀚𑀲𑁆𑀲, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀯𑀺𑀓𑁆𑀓𑀬𑀪𑀡𑁆𑀟𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺, 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀡𑁆𑀟𑀸𑀬 𑀅𑀦𑀸𑀕𑀢𑁄 𑀫𑀕𑁆𑀕𑁄 𑀑𑀮𑁄𑀓𑁂𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀸𑀓𑀝𑀺𑀓𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀅𑀦𑀸𑀕𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀺𑀬𑀸𑀫𑀓𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀅𑀦𑀸𑀕𑀢𑀁 𑀢𑀻𑀭𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀦𑀸𑀯𑀸 𑀧𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸 𑀳𑁄𑀢𑀺, 𑀪𑀺𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀆𑀬𑀼𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀆𑀢𑀼𑀭𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀲𑁂𑀢𑀼𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀣𑀺𑀭𑀸𑀣𑀺𑀭𑀪𑀸𑀯𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀅𑀦𑀸𑀕𑀢𑀁 𑀓𑀸𑀮𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀓𑀼𑀮𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺 ‘𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀼𑀮𑀁 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀁 𑀯𑀸’𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀅𑀦𑀸𑀕𑀢𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

---

875. 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀢𑀸𑀧𑀜𑁆𑀳𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
