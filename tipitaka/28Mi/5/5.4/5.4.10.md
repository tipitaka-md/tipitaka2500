# 𑁫𑁇𑁪𑁇𑁧𑁦 𑀩𑀼𑀤𑁆𑀥𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀪𑀸𑀯𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

### 𑀩𑀼𑀤𑁆𑀥𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀪𑀸𑀯𑀧𑀜𑁆𑀳

(159.)

934. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀣—  ‘𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑁆𑀧𑀫𑀼𑀔𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂 𑀧𑀡𑀸𑀫𑀺𑀢𑁂 𑀘𑀸𑀢𑀼𑀫𑁂𑀬𑁆𑀬𑀓𑀸 𑀘 𑀲𑀓𑁆𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀘 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀩𑀻𑀚𑀽𑀧𑀫𑀜𑁆𑀘 𑀯𑀘𑁆𑀙𑀢𑀭𑀼𑀡𑀽𑀧𑀫𑀜𑁆𑀘 𑀉𑀧𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀲𑀼𑀁 𑀔𑀫𑀸𑀧𑁂𑀲𑀼𑀁 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀓𑀁𑀲𑀽’𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀜𑁆𑀜𑀸𑀢𑀸 𑀢𑀸 𑀉𑀧𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲, 𑀬𑀸𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀉𑀧𑀫𑀸𑀳𑀺 𑀑𑀭𑀢𑁄 𑀔𑀫𑀺𑀢𑁄 𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀕𑀢𑁄? 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀢𑀸 𑀉𑀧𑀫𑀸 𑀅𑀜𑁆𑀜𑀸𑀢𑀸, 𑀢𑁂𑀦 𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽, 𑀬𑀤𑀺 𑀜𑀸𑀢𑀸, 𑀢𑁂𑀦 𑀳𑀺 𑀑𑀓𑀲𑁆𑀲 𑀧𑀲𑀬𑁆𑀳 𑀯𑀻𑀫𑀁𑀲𑀸𑀧𑁂𑀓𑁆𑀔𑁄 𑀧𑀡𑀸𑀫𑁂𑀲𑀺, 𑀢𑁂𑀦 𑀳𑀺 𑀢𑀲𑁆𑀲 𑀅𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀢𑀸 𑀲𑀫𑁆𑀪𑀯𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

935. “𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄, 𑀢𑀸𑀳𑀺 𑀘 𑀉𑀧𑀫𑀸𑀳𑀺 𑀪𑀕𑀯𑀸 𑀧𑀲𑀦𑁆𑀦𑁄 𑀑𑀭𑀢𑁄 𑀔𑀫𑀺𑀢𑁄 𑀉𑀧𑀲𑀦𑁆𑀢𑁄 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀕𑀢𑁄𑁇 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀸𑀫𑀻, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄, 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂𑀳𑁂𑀯 𑀢𑁂 𑀑𑀧𑀫𑁆𑀫𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀆𑀭𑀸𑀥𑁂𑀲𑀼𑀁 𑀢𑁄𑀲𑁂𑀲𑀼𑀁 𑀧𑀲𑀸𑀤𑁂𑀲𑀼𑀁, 𑀢𑁂𑀲𑀜𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀺𑁇

936. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀢𑁆𑀣𑀻 𑀲𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀓𑁂𑀦𑁂𑀯 𑀥𑀦𑁂𑀦 𑀲𑀸𑀫𑀺𑀓𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺 𑀢𑁄𑀲𑁂𑀢𑀺 𑀧𑀲𑀸𑀤𑁂𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀲𑀸𑀫𑀺𑀓𑁄 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀸𑀢𑀼𑀫𑁂𑀬𑁆𑀬𑀓𑀸 𑀘 𑀲𑀓𑁆𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀘 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂𑀳𑁂𑀯 𑀑𑀧𑀫𑁆𑀫𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀆𑀭𑀸𑀥𑁂𑀲𑀼𑀁 𑀢𑁄𑀲𑁂𑀲𑀼𑀁 𑀧𑀲𑀸𑀤𑁂𑀲𑀼𑀁, 𑀢𑁂𑀲𑀜𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀺𑁇

937. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀧𑁆𑀧𑀓𑁄 𑀭𑀜𑁆𑀜𑁄 𑀲𑀦𑁆𑀢𑀓𑁂𑀦𑁂𑀯 𑀲𑀼𑀯𑀡𑁆𑀡𑀨𑀡𑀓𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑀁 𑀧𑀲𑀸𑀥𑀬𑀫𑀸𑀦𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺 𑀢𑁄𑀲𑁂𑀢𑀺 𑀧𑀲𑀸𑀤𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘 𑀭𑀸𑀚𑀸 𑀧𑀲𑀦𑁆𑀦𑁄 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀢𑀺, 𑀬𑀣𑀺𑀘𑁆𑀙𑀺𑀢𑀫𑀦𑀼𑀧𑁆𑀧𑀤𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀸𑀢𑀼𑀫𑁂𑀬𑁆𑀬𑀓𑀸 𑀘 𑀲𑀓𑁆𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀘 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂𑀳𑁂𑀯 𑀑𑀧𑀫𑁆𑀫𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀆𑀭𑀸𑀥𑁂𑀲𑀼𑀁 𑀢𑁄𑀲𑁂𑀲𑀼𑀁 𑀧𑀲𑀸𑀤𑁂𑀲𑀼𑀁, 𑀢𑁂𑀲𑀜𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀺𑁇

938. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸𑀪𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀲𑁆𑀲 𑀉𑀧𑀦𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺 𑀢𑁄𑀲𑁂𑀢𑀺 𑀧𑀲𑀸𑀤𑁂𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀸𑀢𑀼𑀫𑁂𑀬𑁆𑀬𑀓𑀸 𑀘 𑀲𑀓𑁆𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀘 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂𑀳𑁂𑀯 𑀑𑀧𑀫𑁆𑀫𑁂𑀳𑀺 𑀢𑀣𑀸𑀕𑀢𑀁 𑀆𑀭𑀸𑀥𑁂𑀲𑀼𑀁 𑀢𑁄𑀲𑁂𑀲𑀼𑀁 𑀧𑀲𑀸𑀤𑁂𑀲𑀼𑀁, 𑀢𑁂𑀲𑀜𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 ‘𑀲𑀸𑀥𑀽’𑀢𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

---

939. 𑀩𑀼𑀤𑁆𑀥𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀪𑀸𑀯𑀧𑀜𑁆𑀳𑁄 𑀤𑀲𑀫𑁄𑁇

940. 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
