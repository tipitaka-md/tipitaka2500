# 𑁫𑁇𑁪𑁇𑁧 𑀇𑀤𑁆𑀥𑀺𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀇𑀤𑁆𑀥𑀺𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀧𑀜𑁆𑀳

(150.)

854\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀓𑀺𑀭 𑀲𑁄 𑀮𑀕𑀼𑀍𑀏𑀳𑀺 𑀧𑀭𑀺𑀧𑁄𑀣𑀺𑀢𑁄 𑀪𑀺𑀦𑁆𑀦𑀲𑀻𑀲𑁄 𑀲𑀜𑁆𑀘𑀼𑀡𑁆𑀡𑀺𑀢𑀝𑁆𑀞𑀺𑀫𑀁𑀲𑀥𑀫𑀦𑀺𑀙𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀕𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀣𑁂𑀭𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀓𑁄𑀝𑀺𑀁 𑀕𑀢𑁄, 𑀢𑁂𑀦 𑀳𑀺 𑀮𑀕𑀼𑀍𑀏𑀳𑀺 𑀧𑁄𑀣𑀺𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀮𑀕𑀼𑀍𑀏𑀳𑀺 𑀧𑀭𑀺𑀧𑁄𑀣𑀺𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀢𑁂𑀦 𑀳𑀺 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀓𑁄𑀝𑀺𑀁 𑀕𑀢𑁄𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀓𑀺𑀁 𑀦 𑀲𑀫𑀢𑁆𑀣𑁄 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀉𑀧𑀖𑀸𑀢𑀁 𑀅𑀧𑀦𑀬𑀺𑀢𑀼𑀁, 𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲𑀧𑀺 𑀮𑁄𑀓𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀭𑀡𑀁 𑀪𑀯𑀺𑀢𑀼𑀁 𑀅𑀭𑀳𑁄𑀢𑀺? 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

855\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄’𑀢𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀮𑀕𑀼𑀍𑀅𑀳𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀓𑀫𑁆𑀫𑀸𑀥𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑁂𑀦𑀸”𑀢𑀺𑁇

856\. “𑀦𑀦𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀤𑁆𑀥𑀺𑀫𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀲𑀬𑁄𑀧𑀺 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁄𑀧𑀺 𑀤𑁆𑀯𑁂 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀸, 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑁂𑀦 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀁 𑀅𑀧𑀦𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀣𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁂𑀘𑀺 𑀨𑀮𑀓𑀸𑀫𑀸 𑀓𑀧𑀺𑀢𑁆𑀣𑁂𑀦 𑀓𑀧𑀺𑀢𑁆𑀣𑀁 𑀧𑁄𑀣𑁂𑀦𑁆𑀢𑀺, 𑀅𑀫𑁆𑀩𑁂𑀦 𑀅𑀫𑁆𑀩𑀁 𑀧𑁄𑀣𑁂𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑁂𑀦 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀁 𑀧𑁄𑀣𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑀦𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀸𑀦𑀫𑁆𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀁 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀺𑀬𑀸 𑀭𑀸𑀚𑀸𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀫𑀚𑀘𑁆𑀘𑀸, 𑀲𑀫𑀚𑀘𑁆𑀘𑀸𑀦𑀫𑁆𑀧𑀺 𑀢𑁂𑀲𑀁 𑀏𑀓𑁄 𑀲𑀩𑁆𑀩𑁂 𑀅𑀪𑀺𑀪𑀯𑀺𑀢𑁆𑀯𑀸 𑀆𑀡𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀲𑀁 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁𑀬𑁂𑀯 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁 𑀬𑁂𑀯 𑀲𑀩𑁆𑀩𑁂 𑀅𑀪𑀺𑀪𑀯𑀺𑀬 𑀆𑀡𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺, 𑀓𑀫𑁆𑀫𑀸𑀥𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀲𑁆𑀲 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀑𑀓𑀸𑀲𑀁 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺𑁇

857\. 𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑀺𑀤𑁂𑀯 𑀧𑀓𑀭𑀡𑁂 𑀅𑀧𑀭𑀚𑁆𑀛𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀯𑀸 𑀧𑀺𑀢𑀸 𑀯𑀸 𑀪𑀕𑀺𑀦𑀻 𑀯𑀸 𑀪𑀸𑀢𑀭𑁄 𑀯𑀸 𑀲𑀔𑀻 𑀯𑀸 𑀲𑀳𑀸𑀬𑀓𑀸 𑀯𑀸 𑀢𑀸𑀬𑀦𑁆𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸𑀬𑁂𑀯 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀪𑀯𑀺𑀬 𑀆𑀡𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀁? 𑀅𑀧𑀭𑀸𑀥𑀺𑀓𑀢𑀸𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀲𑀁 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁𑀬𑁂𑀯 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁𑀬𑁂𑀯 𑀲𑀩𑁆𑀩𑁂 𑀅𑀪𑀺𑀪𑀯𑀺𑀬 𑀆𑀡𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺, 𑀓𑀫𑁆𑀫𑀸𑀥𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀲𑁆𑀲 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀑𑀓𑀸𑀲𑀁 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺𑁇

858\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀺𑀬𑀸 𑀤𑀯𑀟𑀸𑀳𑁂 𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑁂 𑀖𑀝𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀉𑀤𑀓𑀁 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀼𑀁, 𑀅𑀣 𑀔𑁄 𑀅𑀕𑁆𑀕𑀺𑀬𑁂𑀯 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀪𑀯𑀺𑀬 𑀆𑀡𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺𑁇 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀁? 𑀩𑀮𑀯𑀢𑀸 𑀢𑁂𑀚𑀲𑁆𑀲𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀲𑀁 𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁 𑀬𑁂𑀯 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀁𑀬𑁂𑀯 𑀲𑀩𑁆𑀩𑁂 𑀅𑀪𑀺𑀪𑀯𑀺𑀬 𑀆𑀡𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺, 𑀓𑀫𑁆𑀫𑀸𑀥𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀲𑁆𑀲 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑀺𑀭𑀺𑀬𑀸 𑀑𑀓𑀸𑀲𑀁 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀥𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀲𑁆𑀲 𑀮𑀕𑀼𑀍𑀏𑀳𑀺 𑀧𑁄𑀣𑀺𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀸𑀭𑁄 𑀦𑀸𑀳𑁄𑀲𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

---

859\. 𑀇𑀤𑁆𑀥𑀺𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑀧𑀜𑁆𑀳𑁄 𑀧𑀞𑀫𑁄𑁇
