# 𑁫𑁇𑁪𑁇𑁫 𑀅𑀢𑁆𑀢𑀦𑀺𑀧𑀸𑀢𑀦𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀅𑀢𑁆𑀢𑀦𑀺𑀧𑀸𑀢𑀦𑀧𑀜𑁆𑀳

(154.)

876\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀁, 𑀬𑁄 𑀧𑀸𑀢𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀬𑀫𑀸𑀦𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀩𑁆𑀬𑀸𑀥𑀺𑀦𑁄 𑀫𑀭𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀬𑁄 𑀳𑀺 𑀓𑁄𑀘𑀺 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀩𑁆𑀬𑀸𑀥𑀺𑀫𑀭𑀡𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀢𑀁 𑀧𑀭𑀫𑀸𑀬 𑀧𑀲𑀁𑀲𑀸𑀬 𑀧𑀲𑀁𑀲𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀁, 𑀬𑁄 𑀧𑀸𑀢𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀩𑁆𑀬𑀸𑀥𑀺𑀦𑁄 𑀫𑀭𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀩𑁆𑀬𑀸𑀥𑀺𑀦𑁄 𑀫𑀭𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀁, 𑀬𑁄 𑀧𑀸𑀢𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

877\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀁, 𑀬𑁄 𑀧𑀸𑀢𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺 𑀪𑀕𑀯𑀢𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀬𑀫𑀸𑀦𑁂𑀦 𑀘 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀩𑁆𑀬𑀸𑀥𑀺𑀦𑁄 𑀫𑀭𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀥𑀫𑁆𑀫𑁄 𑀤𑁂𑀲𑀺𑀢𑁄, 𑀢𑀢𑁆𑀣 𑀧𑀦 𑀓𑀸𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀺, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀓𑀸𑀭𑀡𑁂𑀦 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀘𑀸”𑀢𑀺𑁇

878\. “𑀓𑀺𑀁 𑀧𑀦𑁂𑀢𑁆𑀣, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀓𑀸𑀭𑀡𑁂𑀦 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀘𑀸”𑀢𑀺? “𑀲𑀻𑀮𑀯𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀕𑀤𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀲𑀯𑀺𑀦𑀸𑀲𑀦𑁂, 𑀑𑀲𑀥𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀮𑁂𑀲𑀩𑁆𑀬𑀸𑀥𑀺𑀯𑀽𑀧𑀲𑀫𑁂, 𑀉𑀤𑀓𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀺𑀮𑁂𑀲𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀸𑀧𑀳𑀭𑀡𑁂, 𑀫𑀡𑀺𑀭𑀢𑀦𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀩𑁆𑀩𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀤𑀸𑀦𑁂, 𑀦𑀸𑀯𑀸𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀢𑀼𑀭𑁄𑀖𑀧𑀸𑀭𑀕𑀫𑀦𑁂, 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀚𑀸𑀢𑀺𑀓𑀦𑁆𑀢𑀸𑀭𑀢𑀸𑀭𑀡𑁂, 𑀯𑀸𑀢𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑀺𑀯𑀺𑀥𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑁂, 𑀫𑀳𑀸𑀫𑁂𑀖𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀫𑀸𑀦𑀲𑀧𑀭𑀺𑀧𑀽𑀭𑀡𑁂, 𑀆𑀘𑀭𑀺𑀬𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀦𑁂, 𑀲𑀼𑀤𑁂𑀲𑀓𑀲𑀫𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀔𑁂𑀫𑀧𑀣𑀫𑀸𑀘𑀺𑀓𑁆𑀔𑀡𑁂𑁇 𑀏𑀯𑀭𑀽𑀧𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀳𑀼𑀕𑀼𑀡𑁄 𑀅𑀦𑁂𑀓𑀕𑀼𑀡𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀕𑀼𑀡𑁄 𑀕𑀼𑀡𑀭𑀸𑀲𑀺 𑀕𑀼𑀡𑀧𑀼𑀜𑁆𑀚𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀟𑁆𑀠𑀺𑀓𑀭𑁄 𑀲𑀻𑀮𑀯𑀸 ‘𑀫𑀸 𑀯𑀺𑀦𑀲𑁆𑀲𑀻’𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀪𑀕𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺 ‘𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀸𑀢𑁂𑀢𑀩𑁆𑀩𑀁, 𑀬𑁄 𑀧𑀸𑀢𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀪𑀕𑀯𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁂𑀦 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀯𑀺𑀘𑀺𑀢𑁆𑀭𑀓𑀣𑀺𑀓𑁂𑀦 𑀧𑀸𑀬𑀸𑀲𑀺𑀭𑀸𑀚𑀜𑁆𑀜𑀲𑁆𑀲 𑀧𑀭𑀮𑁄𑀓𑀁 𑀤𑀻𑀧𑀬𑀫𑀸𑀦𑁂𑀦 ‘𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄, 𑀭𑀸𑀚𑀜𑁆𑀜, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀭𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀦𑁆𑀢𑀺, 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀘 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆’𑀢𑀺𑁇

879\. 𑀓𑁂𑀦 𑀧𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺? 𑀚𑀸𑀢𑀺𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀚𑀭𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀩𑁆𑀬𑀸𑀥𑀺𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑁄𑀓𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀧𑀭𑀺𑀤𑁂𑀯𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀉𑀧𑀸𑀬𑀸𑀲𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀅𑀧𑁆𑀧𑀺𑀬𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀬𑁄𑀕𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀧𑀺𑀬𑁂𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑁄𑀕𑁄𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑁄, 𑀫𑀸𑀢𑀼𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀧𑀺𑀢𑀼𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑀸𑀢𑀼𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑀕𑀺𑀦𑀺𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀧𑀼𑀢𑁆𑀢𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀸𑀭𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀸𑀲𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀜𑀸𑀢𑀺𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀭𑀸𑀚𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀘𑁄𑀭𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀯𑁂𑀭𑀺𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀕𑁆𑀕𑀺𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀉𑀤𑀓𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀊𑀫𑀺𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀆𑀯𑀝𑁆𑀝𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀓𑀼𑀫𑁆𑀪𑀻𑀮𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀼𑀲𑀼𑀓𑀸𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀸𑀤𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀧𑀭𑀸𑀦𑀼𑀯𑀸𑀤𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀡𑁆𑀟𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀧𑀭𑀺𑀲𑀸𑀲𑀸𑀭𑀚𑁆𑀚𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀆𑀚𑀻𑀯𑀓𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀫𑀭𑀡𑀪𑀬𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀯𑁂𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀸𑀍𑀅𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀓𑀲𑀸𑀳𑀺 𑀢𑀸𑀍𑀅𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀤𑁆𑀥𑀤𑀡𑁆𑀟𑀓𑁂𑀳𑀺 𑀢𑀸𑀍𑀅𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀳𑀢𑁆𑀣𑀘𑁆𑀙𑁂𑀤𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀧𑀸𑀤𑀘𑁆𑀙𑁂𑀤𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀘𑁆𑀙𑁂𑀤𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀓𑀡𑁆𑀡𑀘𑁆𑀙𑁂𑀤𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀦𑀸𑀲𑀘𑁆𑀙𑁂𑀤𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀘𑁆𑀙𑁂𑀤𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀩𑀺𑀮𑀗𑁆𑀕𑀣𑀸𑀮𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀗𑁆𑀔𑀫𑀼𑀡𑁆𑀟𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀭𑀸𑀳𑀼𑀫𑀼𑀔𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀚𑁄𑀢𑀺𑀫𑀸𑀮𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀳𑀢𑁆𑀣𑀧𑀚𑁆𑀚𑁄𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀏𑀭𑀓𑀯𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀘𑀻𑀭𑀓𑀯𑀸𑀲𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀏𑀡𑁂𑀬𑁆𑀬𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀩𑀍𑀇𑀲𑀫𑀁𑀲𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀓𑀳𑀸𑀧𑀡𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀔𑀸𑀭𑀸𑀧𑀢𑀘𑁆𑀙𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀧𑀮𑀺𑀖𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀧𑀮𑀸𑀮𑀧𑀻𑀞𑀓𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀢𑀢𑁆𑀢𑁂𑀦 𑀢𑁂𑀮𑁂𑀦 𑀑𑀲𑀺𑀜𑁆𑀘𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀼𑀦𑀔𑁂𑀳𑀺 𑀔𑀸𑀤𑀸𑀧𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀚𑀻𑀯𑀲𑀽𑀮𑀸𑀭𑁄𑀧𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀲𑀺𑀦𑀸 𑀲𑀻𑀲𑀘𑁆𑀙𑁂𑀤𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀳𑀼𑀯𑀺𑀥𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀯𑀺𑀥𑀸𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺 𑀲𑀁𑀲𑀸𑀭𑀕𑀢𑁄 𑀅𑀦𑀼𑀪𑀯𑀢𑀺𑁇

880\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀳𑀺𑀫𑀯𑀦𑁆𑀢𑀧𑀩𑁆𑀩𑀢𑁂 𑀅𑀪𑀺𑀯𑀼𑀝𑁆𑀞𑀁 𑀉𑀤𑀓𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀔𑀭𑀫𑀭𑀼𑀫𑁆𑀩𑀆𑀯𑀝𑁆𑀝𑀕𑀕𑁆𑀕𑀮𑀓𑀊𑀫𑀺𑀓𑀯𑀗𑁆𑀓𑀘𑀤𑀺𑀓-𑀆𑀯𑀭𑀡𑀦𑀻𑀯𑀭𑀡𑀫𑀽𑀮𑀓𑀲𑀸𑀔𑀸𑀲𑀼 𑀧𑀭𑀺𑀬𑁄𑀢𑁆𑀣𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀸𑀦𑀺 𑀅𑀦𑁂𑀓𑀯𑀺𑀥𑀸𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺 𑀲𑀁𑀲𑀸𑀭𑀕𑀢𑁄 𑀅𑀦𑀼𑀪𑀯𑀢𑀺𑁇 𑀧𑀯𑀢𑁆𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀁𑁇 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀲𑁆𑀲 𑀕𑀼𑀡𑀁 𑀧𑀯𑀢𑁆𑀢𑁂 𑀘 𑀪𑀬𑀁 𑀤𑀻𑀧𑀬𑀫𑀸𑀦𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀸 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀩𑁆𑀬𑀸𑀥𑀺𑀫𑀭𑀡𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺, 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀼𑀦𑀺𑀩𑁆𑀩𑁂𑀞𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑁄, 𑀲𑀼𑀓𑀣𑀺𑀢𑀁 𑀓𑀸𑀭𑀡𑀁, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
