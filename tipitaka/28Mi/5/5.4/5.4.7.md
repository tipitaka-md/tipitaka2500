# 𑁫𑁇𑁪𑁇𑁭 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀲𑀫𑀯𑀺𑀲𑀫𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀲𑀫𑀯𑀺𑀲𑀫𑀧𑀜𑁆𑀳

(156.)

890\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦 𑀓𑀼𑀲𑀮𑀓𑀸𑀭𑀺𑀲𑁆𑀲𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀓𑀸𑀭𑀺𑀲𑁆𑀲𑀧𑀺 𑀯𑀺𑀧𑀸𑀓𑁄 𑀲𑀫𑀲𑀫𑁄, 𑀉𑀤𑀸𑀳𑀼 𑀓𑁄𑀘𑀺 𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀢𑁆𑀣𑀻”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀯𑀺𑀲𑁂𑀲𑁄, 𑀓𑀼𑀲𑀮𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀔𑀯𑀺𑀧𑀸𑀓𑀁 𑀲𑀕𑁆𑀕𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁, 𑀅𑀓𑀼𑀲𑀮𑀁 𑀤𑀼𑀓𑁆𑀔𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀦𑁆”𑀢𑀺𑁇

891\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀏𑀓𑀦𑁆𑀢𑀓𑀡𑁆𑀳𑁄, 𑀏𑀓𑀦𑁆𑀢𑀓𑀡𑁆𑀳𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀓𑁆𑀓𑁄, 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀓𑁆𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀪𑀯𑁂 𑀪𑀯𑁂 𑀬𑀲𑁂𑀦 𑀘 𑀧𑀓𑁆𑀔𑁂𑀦 𑀘 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑀲𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀤𑀸𑀘𑀺 𑀅𑀥𑀺𑀓𑀢𑀭𑁄 𑀯𑀸𑁇 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀦𑀕𑀭𑁂 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀙𑀯𑀓𑀘𑀡𑁆𑀟𑀸𑀮𑁄 𑀅𑀳𑁄𑀲𑀺 𑀯𑀺𑀚𑁆𑀚𑀸𑀥𑀭𑁄, 𑀯𑀺𑀚𑁆𑀚𑀁 𑀧𑀭𑀺𑀚𑀧𑁆𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀓𑀸𑀮𑁂 𑀅𑀫𑁆𑀩𑀨𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀲𑀺, 𑀏𑀢𑁆𑀣 𑀢𑀸𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀢𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀦𑀺𑀳𑀻𑀦𑁄 𑀬𑀲𑁂𑀦 𑀘 𑀦𑀺𑀳𑀻𑀦𑁄𑁇

892\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺 𑀫𑀳𑀸 𑀫𑀳𑀻𑀧𑀢𑀺 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀗𑁆𑀕𑀻, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲𑀽𑀧𑀪𑁄𑀕𑁄 𑀅𑀳𑁄𑀲𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀲𑀩𑁆𑀩𑀮𑀓𑁆𑀔𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄, 𑀢𑀲𑁆𑀲 𑀘𑀸𑀭𑀼𑀕𑀢𑀺𑀯𑀺𑀮𑀸𑀲𑀁 𑀅𑀲𑀳𑀫𑀸𑀦𑁄 𑀭𑀸𑀚𑀸 𑀯𑀥𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀳𑀢𑁆𑀣𑀸𑀘𑀭𑀺𑀬𑀁 𑀏𑀯𑀫𑀯𑁄𑀘 ‘𑀅𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄 𑀢𑁂, 𑀆𑀘𑀭𑀺𑀬, 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄, 𑀢𑀲𑁆𑀲 𑀆𑀓𑀸𑀲𑀕𑀫𑀦𑀁 𑀦𑀸𑀫 𑀓𑀸𑀭𑀡𑀁 𑀓𑀭𑁄𑀳𑀻’𑀢𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀢𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀦𑀺𑀳𑀻𑀦𑁄 𑀮𑀸𑀫𑀓𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄𑁇

893\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀯𑀦𑁂 𑀦𑀝𑁆𑀞𑀸𑀬𑀺𑀓𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀧𑀣𑀯𑀻 𑀦𑀸𑀫 𑀫𑀓𑁆𑀓𑀝𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀏𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀯𑀺𑀲𑁂𑀲𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀲𑁆𑀲 𑀘 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀲𑁆𑀲 𑀘, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀢𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀦𑀺𑀳𑀻𑀦𑁄𑁇

894\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑁄𑀡𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀦𑁂𑀲𑀸𑀤𑁄 𑀩𑀮𑀯𑀸 𑀩𑀮𑀯𑀢𑀭𑁄 𑀦𑀸𑀕𑀩𑀮𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀙𑀤𑁆𑀤𑀦𑁆𑀢𑁄 𑀦𑀸𑀫 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀲𑁄 𑀮𑀼𑀤𑁆𑀤𑀓𑁄 𑀢𑀁 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀁 𑀖𑀸𑀢𑁂𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀯 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

895\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀯𑀦𑀘𑀭𑀓𑁄 𑀅𑀦𑀺𑀓𑁂𑀢𑀯𑀸𑀲𑀻, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀲𑀓𑀼𑀡𑁄 𑀅𑀳𑁄𑀲𑀺 𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑁄 𑀫𑀦𑁆𑀢𑀚𑁆𑀛𑀸𑀬𑀻, 𑀢𑀤𑀸𑀧𑀺 𑀲𑁄 𑀯𑀦𑀘𑀭𑀓𑁄 𑀢𑀁 𑀲𑀓𑀼𑀡𑀁 𑀖𑀸𑀢𑁂𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀯 𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

896\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 ( ) 𑀓𑀮𑀸𑀩𑀼 𑀦𑀸𑀫 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀸𑀧𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀔𑀦𑁆𑀢𑀺𑀯𑀸𑀤𑀻, 𑀢𑀤𑀸 𑀲𑁄 𑀭𑀸𑀚𑀸 𑀢𑀲𑁆𑀲 𑀢𑀸𑀧𑀲𑀲𑁆𑀲 𑀓𑀼𑀤𑁆𑀥𑁄 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑁂 𑀯𑀁𑀲𑀓𑀍𑀈𑀭𑁂 𑀯𑀺𑀬 𑀙𑁂𑀤𑀸𑀧𑁂𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀅𑀥𑀺𑀓𑀢𑀭𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀘 𑀬𑀲𑁂𑀦 𑀘𑁇

897\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀯𑀦𑀘𑀭𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀦𑀦𑁆𑀤𑀺𑀬𑁄 𑀦𑀸𑀫 𑀯𑀸𑀦𑀭𑀺𑀦𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸𑀧𑀺 𑀲𑁄 𑀯𑀦𑀘𑀭𑁄 𑀢𑀁 𑀯𑀸𑀦𑀭𑀺𑀦𑁆𑀤𑀁 𑀖𑀸𑀢𑁂𑀲𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀫𑀸𑀢𑀭𑀸 𑀓𑀦𑀺𑀝𑁆𑀞𑀪𑀸𑀢𑀺𑀓𑁂𑀦 𑀘, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀅𑀥𑀺𑀓𑀢𑀭𑁄 𑀚𑀸𑀢𑀺𑀬𑀸𑁇

898\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀅𑀘𑁂𑀮𑀓𑁄 𑀓𑀸𑀭𑀫𑁆𑀪𑀺𑀬𑁄 𑀦𑀸𑀫, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀧𑀡𑁆𑀟𑀭𑀓𑁄 𑀦𑀸𑀫 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀅𑀥𑀺𑀓𑀢𑀭𑁄 𑀚𑀸𑀢𑀺𑀬𑀸𑁇

899\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀯𑀦𑁂 𑀚𑀝𑀺𑀮𑀓𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀘𑁆𑀙𑀓𑁄 𑀦𑀸𑀫 𑀫𑀳𑀸𑀲𑀽𑀓𑀭𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

900\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀘𑁂𑀢𑀻𑀲𑀼 𑀲𑀽𑀭𑀧𑀭𑀺𑀘𑀭𑁄 𑀦𑀸𑀫 𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺 𑀉𑀧𑀭𑀺 𑀧𑀼𑀭𑀺𑀲𑀫𑀢𑁆𑀢𑁂 𑀕𑀕𑀦𑁂 𑀯𑁂𑀳𑀸𑀲𑀗𑁆𑀕𑀫𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀓𑀧𑀺𑀮𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀅𑀥𑀺𑀓𑀢𑀭𑁄 𑀚𑀸𑀢𑀺𑀬𑀸 𑀘 𑀬𑀲𑁂𑀦 𑀘𑁇

901\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑀸𑀫𑁄 𑀦𑀸𑀫, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀭𑀼𑀭𑀼 𑀦𑀸𑀫 𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀬𑁂𑀯 𑀚𑀸𑀢𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

902\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀮𑀼𑀤𑁆𑀤𑀓𑁄 𑀧𑀯𑀦𑀘𑀭𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄 𑀮𑀼𑀤𑁆𑀤𑀓𑁄 𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑀦𑁆𑀢𑁂 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀳𑀭𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀬𑁂𑀯 𑀬𑁄𑀦𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

903\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑁄 𑀅𑀳𑁄𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀥𑀫𑁆𑀫𑁄, 𑀲𑁄 𑀬𑀸𑀯𑀢𑀸 𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀧𑀤𑁂𑀲𑀭𑀸𑀚𑀸𑀦𑁄 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁂 𑀅𑀓𑀸𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀯𑀺𑀥𑀼𑀭𑁄 𑀦𑀸𑀫 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀬𑀲𑁂𑀦 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

904\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀳𑀼𑀢𑁆𑀯𑀸 𑀮𑀝𑀼𑀓𑀺𑀓𑀸𑀬 𑀲𑀓𑀼𑀡𑀺𑀓𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑀓𑁂 𑀖𑀸𑀢𑁂𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀅𑀳𑁄𑀲𑀺 𑀬𑀽𑀣𑀧𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀢𑀸𑀯 𑀉𑀪𑁄𑀧𑀺 𑀢𑁂 𑀲𑀫𑀲𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

905\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀬𑀓𑁆𑀔𑁄 𑀅𑀳𑁄𑀲𑀺 𑀅𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀫, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺 𑀬𑀓𑁆𑀔𑁄 𑀅𑀳𑁄𑀲𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀫, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀉𑀪𑁄𑀧𑀺 𑀲𑀫𑀲𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

906\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀯𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀓𑀼𑀮𑀲𑀢𑀸𑀦𑀁 𑀇𑀲𑁆𑀲𑀭𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺 𑀦𑀸𑀯𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀓𑀼𑀮𑀲𑀢𑀸𑀦𑀁 𑀇𑀲𑁆𑀲𑀭𑁄, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀉𑀪𑁄𑀧𑀺 𑀲𑀫𑀲𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

907\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀁 𑀇𑀲𑁆𑀲𑀭𑁄, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺 𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀁 𑀇𑀲𑁆𑀲𑀭𑁄, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀉𑀪𑁄𑀧𑀺 𑀲𑀫𑀲𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

908\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀸𑀔𑁄 𑀦𑀸𑀫 𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁄 𑀦𑀸𑀫 𑀫𑀺𑀕𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀉𑀪𑁄𑀧𑀺 𑀲𑀫𑀲𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

909\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀸𑀔𑁄 𑀦𑀸𑀫 𑀲𑁂𑀦𑀸𑀧𑀢𑀺 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑁄 𑀦𑀸𑀫 𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀉𑀪𑁄𑀧𑀺 𑀲𑀫𑀲𑀫𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

910\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀔𑀡𑁆𑀟𑀳𑀸𑀮𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀘𑀦𑁆𑀤𑁄 𑀦𑀸𑀫 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀲𑁄 𑀔𑀡𑁆𑀟𑀳𑀸𑀮𑁄𑀬𑁂𑀯 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

911\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀫 𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀧𑀤𑀼𑀫𑁄 𑀦𑀸𑀫 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀲𑁄 𑀭𑀸𑀚𑀸 𑀲𑀓𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑁄𑀭𑀧𑀧𑀸𑀢𑁂 𑀔𑀺𑀧𑀸𑀧𑁂𑀲𑀺, 𑀬𑀢𑁄 𑀓𑀼𑀢𑁄𑀘𑀺 𑀧𑀺𑀢𑀸𑀯 𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀥𑀺𑀓𑀢𑀭𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀲𑀺𑀝𑁆𑀞𑁄𑀢𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

912\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀬𑀤𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀧𑀢𑀸𑀧𑁄 𑀦𑀸𑀫 𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑁄 𑀦𑀸𑀫 𑀓𑀼𑀫𑀸𑀭𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀤𑀸 𑀲𑁄 𑀭𑀸𑀚𑀸 𑀲𑀓𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑁂 𑀲𑀻𑀲𑀜𑁆𑀘 𑀙𑁂𑀤𑀸𑀧𑁂𑀲𑀺, 𑀢𑀢𑁆𑀣𑀧𑀺 𑀢𑀸𑀯 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀬𑁂𑀯 𑀉𑀢𑁆𑀢𑀭𑁄 𑀅𑀥𑀺𑀓𑀢𑀭𑁄𑁇

913\. 𑀅𑀚𑁆𑀚𑁂𑀢𑀭𑀳𑀺 𑀉𑀪𑁄𑀧𑀺 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄, 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀮𑀬𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀁 𑀫𑀬𑀸 𑀪𑀡𑀺𑀢𑀁, 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀣𑀁 𑀉𑀤𑀸𑀳𑀼 𑀯𑀺𑀢𑀣𑀦𑁆”𑀢𑀺?

914\. “𑀬𑀁 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀳𑀼𑀯𑀺𑀥𑀁 𑀓𑀸𑀭𑀡𑀁 𑀑𑀲𑀸𑀭𑁂𑀲𑀺, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀢𑀣𑁂𑀯, 𑀦𑁄 𑀅𑀜𑁆𑀜𑀣𑀸”𑀢𑀺𑁇 “𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀡𑁆𑀳𑁄𑀧𑀺 𑀲𑀼𑀓𑁆𑀓𑁄𑀧𑀺 𑀲𑀫𑀲𑀫𑀕𑀢𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀲𑀫𑀲𑀫𑀯𑀺𑀧𑀸𑀓𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀲𑀫𑀲𑀫𑀯𑀺𑀧𑀸𑀓𑀁 𑀳𑁄𑀢𑀺, 𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀚𑀦𑁂𑀳𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁄, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦𑁂𑀯 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁄𑁇 𑀬𑁄 𑀢𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑁄, 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀪𑀯𑁂 𑀧𑀘𑁆𑀘𑀢𑀺 𑀨𑀮𑀁 𑀤𑁂𑀢𑀺𑁇 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑁂 𑀞𑀺𑀢𑁄 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀆𑀭𑀓𑁆𑀔𑀁 𑀤𑁂𑀢𑀺, 𑀲𑁂𑀢𑀼𑀁 𑀲𑀪𑀁 𑀧𑀼𑀜𑁆𑀜𑀲𑀸𑀮𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀓𑀧𑀡𑀤𑁆𑀥𑀺𑀓𑀯𑀡𑀺𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀦𑀸𑀣𑀸𑀦𑀸𑀣𑀸𑀦𑀁 𑀬𑀣𑀸𑀧𑀡𑀺𑀳𑀺𑀢𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑁄 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀪𑀯𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀝𑀺𑀮𑀪𑀢𑀺𑁇 𑀓𑀲𑁆𑀲𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑀸 𑀯𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀦𑀸 𑀤𑀸𑀦𑁂𑀦 𑀤𑀫𑁂𑀦 𑀲𑀁𑀬𑀫𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

915\. 𑀬𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀯𑀁 𑀯𑀤𑁂𑀲𑀺 ‘𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀘 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀘 𑀏𑀓𑀢𑁄 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀲𑁄 𑀦 𑀚𑀸𑀢𑀺𑀲𑀢𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀫𑀸𑀕𑀫𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀦 𑀚𑀸𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦, 𑀦 𑀚𑀸𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦, 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀩𑀳𑀽𑀦𑀁 𑀅𑀳𑁄𑀭𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀫𑀸𑀕𑀫𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀬𑀁 𑀧𑀦𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀓𑀸𑀡𑀓𑀘𑁆𑀙𑀧𑁄𑀧𑀫𑀁 𑀉𑀧𑀤𑀲𑁆𑀲𑀺𑀢𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀢𑀣𑀽𑀧𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀫𑁂𑀲𑀁 𑀲𑀫𑀸𑀕𑀫𑀁 𑀥𑀸𑀭𑁂𑀳𑀺𑁇

916\. 𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁂𑀦𑁂𑀯 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑀸𑀕𑀫𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀣𑁂𑀭𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑁂𑀓𑁂𑀲𑀼 𑀚𑀸𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀲𑀼 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀫𑀳𑀸𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀘𑀽𑀍𑀅𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀪𑀸𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀫𑀺𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇

917\. 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑁂𑀓𑁂𑀲𑀼 𑀚𑀸𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀲𑀼 𑀣𑁂𑀭𑀲𑁆𑀲 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀫𑀳𑀸𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀘𑀽𑀍𑀅𑀧𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀪𑀸𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀪𑀸𑀕𑀺𑀦𑁂𑀬𑁆𑀬𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀫𑀺𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑀩𑁆𑀩𑁂𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀲𑀁𑀲𑀸𑀭𑀲𑁄𑀢𑀫𑀦𑀼𑀕𑀢𑀸 𑀲𑀁𑀲𑀸𑀭𑀲𑁄𑀢𑁂𑀦 𑀯𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀬𑁂𑀳𑀺𑀧𑀺 𑀧𑀺𑀬𑁂𑀳𑀺𑀧𑀺 𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀲𑁄𑀢𑁂𑀦 𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑀁 𑀲𑀼𑀘𑀺𑀅𑀲𑀼𑀘𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀓𑁂𑀦 𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀲𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀸 𑀲𑀁𑀲𑀸𑀭𑀲𑁄𑀢𑀫𑀦𑀼𑀕𑀢𑀸 𑀲𑀁𑀲𑀸𑀭𑀲𑁄𑀢𑁂𑀦 𑀯𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀬𑁂𑀳𑀺𑀧𑀺 𑀧𑀺𑀬𑁂𑀳𑀺𑀧𑀺 𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀓𑁆𑀔𑁄 𑀲𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑁂 𑀅𑀥𑀫𑁆𑀫𑁂 𑀦𑀺𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀬𑁄 𑀲𑀝𑁆𑀞𑀺 𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀫𑀳𑀸𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀺, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀓𑁆𑀔𑁄 𑀲𑀫𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑁂 𑀥𑀫𑁆𑀫𑁂 𑀦𑀺𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑀸𑀲𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀬𑁄 𑀲𑀝𑁆𑀞𑀺 𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀕𑁆𑀕𑁂 𑀫𑁄𑀤𑀺 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀲𑀫𑀗𑁆𑀕𑀻, 𑀅𑀧𑀺 𑀘, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀪𑀯𑁂 𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀦𑀸𑀲𑀸𑀤𑀦𑀻𑀬𑀫𑀸𑀲𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀣𑀯𑀺𑀁 𑀧𑀸𑀯𑀺𑀲𑀺, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀩𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀉𑀧𑀥𑀺𑀲𑀗𑁆𑀔𑀬𑁂”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
