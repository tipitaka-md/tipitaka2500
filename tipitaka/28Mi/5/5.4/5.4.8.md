# 𑁫𑁇𑁪𑁇𑁮 𑀅𑀫𑀭𑀸𑀤𑁂𑀯𑀻𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀅𑀫𑀭𑀸𑀤𑁂𑀯𑀻𑀧𑀜𑁆𑀳

(157.)

919\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—

920\. _‘𑀲𑀘𑁂 𑀮𑀪𑁂𑀣 𑀔𑀡𑀁 𑀯𑀸 𑀭𑀳𑁄 𑀯𑀸,_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑀓𑀁 𑀯𑀸𑀧𑀺 𑀮𑀪𑁂𑀣 𑀢𑀸𑀤𑀺𑀲𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀯 𑀇𑀢𑁆𑀣𑀻 𑀓𑀬𑀺𑀭𑀼𑀁 𑀦𑀼 𑀧𑀸𑀧𑀁,_  
_𑀅𑀜𑁆𑀜𑀁 𑀅𑀮𑀤𑁆𑀥𑀸 𑀧𑀻𑀞𑀲𑀧𑁆𑀧𑀺𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀦𑁆’𑀢𑀺𑁇_  


921\. 𑀧𑀼𑀦 𑀘 𑀓𑀣𑀻𑀬𑀢𑀺 ‘𑀫𑀳𑁄𑀲𑀥𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸 𑀅𑀫𑀭𑀸 𑀦𑀸𑀫 𑀇𑀢𑁆𑀣𑀻 𑀕𑀸𑀫𑀓𑁂 𑀞𑀧𑀺𑀢𑀸 𑀧𑀯𑀼𑀢𑁆𑀣𑀧𑀢𑀺𑀓𑀸 𑀭𑀳𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸 𑀭𑀸𑀚𑀧𑁆𑀧𑀝𑀺𑀲𑀫𑀁 𑀲𑀸𑀫𑀺𑀓𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑀻𑀬𑀫𑀸𑀦𑀸 𑀧𑀸𑀧𑀁 𑀦𑀸𑀓𑀸𑀲𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀲𑀘𑁂…𑀧𑁂…  𑀲𑀤𑁆𑀥𑀺𑀦𑁆’𑀢𑀺 𑀢𑁂𑀦 𑀳𑀺 ‘𑀫𑀳𑁄𑀲𑀥𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸…𑀧𑁂…  𑀦𑀸𑀓𑀸𑀲𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀫𑀳𑁄𑀲𑀥𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸…𑀧𑁂…  𑀦𑀸𑀓𑀸𑀲𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀲𑀘𑁂…𑀧𑁂…  𑀲𑀤𑁆𑀥𑀺𑀦𑁆’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

922\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀲𑀘𑁂…𑀧𑁂…  𑀲𑀤𑁆𑀥𑀺𑀦𑁆’𑀢𑀺𑁇 𑀓𑀣𑀻𑀬𑀢𑀺 𑀘 ‘𑀫𑀳𑁄𑀲𑀥𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸…𑀧𑁂…  𑀦𑀸𑀓𑀸𑀲𑀻’𑀢𑀺𑁇 𑀓𑀭𑁂𑀬𑁆𑀬 𑀲𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀢𑁆𑀣𑀻 𑀲𑀳𑀲𑁆𑀲𑀁 𑀮𑀪𑀫𑀸𑀦𑀸 𑀢𑀸𑀤𑀺𑀲𑁂𑀦 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁, 𑀦 𑀲𑀸 𑀓𑀭𑁂𑀬𑁆𑀬 𑀲𑀘𑁂 𑀔𑀡𑀁 𑀯𑀸 𑀭𑀳𑁄 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀓𑀁 𑀯𑀸𑀧𑀺 𑀢𑀸𑀤𑀺𑀲𑀁 𑀮𑀪𑁂𑀬𑁆𑀬, 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀻 𑀲𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀫𑀭𑀸 𑀇𑀢𑁆𑀣𑀻 𑀦 𑀅𑀤𑁆𑀤𑀲 𑀔𑀡𑀁 𑀯𑀸 𑀭𑀳𑁄 𑀯𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀓𑀁 𑀯𑀸𑀧𑀺 𑀢𑀸𑀤𑀺𑀲𑀁𑁇

923\. 𑀇𑀥 𑀮𑁄𑀓𑁂 𑀕𑀭𑀳𑀪𑀬𑀸 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺, 𑀧𑀭𑀮𑁄𑀓𑁂 𑀦𑀺𑀭𑀬𑀪𑀬𑀸 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺, 𑀓𑀝𑀼𑀓𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀸𑀧𑀦𑁆𑀢𑀺 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺, 𑀧𑀺𑀬𑀁 𑀅𑀫𑀼𑀜𑁆𑀘𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺, 𑀲𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀕𑀭𑀼𑀓𑀢𑀸𑀬 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀧𑀘𑀸𑀬𑀦𑁆𑀢𑀻 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺, 𑀅𑀦𑀭𑀺𑀬𑀁 𑀕𑀭𑀳𑀦𑁆𑀢𑀻 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺, 𑀓𑀺𑀭𑀺𑀬𑀁 𑀅𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑁂𑀳𑀺 𑀩𑀳𑀽𑀳𑀺 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀔𑀡𑀁 𑀦 𑀧𑀲𑁆𑀲𑀺𑁇

924\. 𑀭𑀳𑁄𑀧𑀺 𑀲𑀸 𑀮𑁄𑀓𑁂 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻 𑀧𑀸𑀧𑀁 𑀦𑀸𑀓𑀸𑀲𑀺𑁇 𑀲𑀘𑁂 𑀲𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀭𑀳𑁄 𑀮𑀪𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀭𑀳𑁄 𑀦 𑀮𑀪𑁂𑀬𑁆𑀬𑁇 𑀲𑀘𑁂 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀭𑀳𑁄 𑀮𑀪𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀤𑀽𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺 𑀭𑀳𑁄 𑀦 𑀮𑀪𑁂𑀬𑁆𑀬𑁇 𑀲𑀘𑁂 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀤𑀽𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀳𑀺 𑀭𑀳𑁄 𑀮𑀪𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀤𑀽𑀦𑀻𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀭𑀳𑁄 𑀦 𑀮𑀪𑁂𑀬𑁆𑀬𑁇 𑀲𑀘𑁂 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀤𑀽𑀦𑀻𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀭𑀳𑁄 𑀮𑀪𑁂𑀬𑁆𑀬, 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀧𑀸𑀧𑁂𑀳𑀺 𑀭𑀳𑁄 𑀦 𑀮𑀪𑁂𑀬𑁆𑀬𑁇 𑀲𑀘𑁂 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀧𑀸𑀧𑁂𑀳𑀺 𑀭𑀳𑁄 𑀮𑀪𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀭𑀳𑁄 𑀦 𑀮𑀪𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀭𑀽𑀧𑁂𑀳𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑁂𑀳𑀺 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀭𑀳𑁄 𑀅𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀧𑀁 𑀦𑀸𑀓𑀸𑀲𑀺𑁇

925\. 𑀦𑀺𑀫𑀦𑁆𑀢𑀓𑀫𑁆𑀧𑀺 𑀲𑀸 𑀮𑁄𑀓𑁂 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀤𑀺𑀲𑀁 𑀅𑀮𑀪𑀦𑁆𑀢𑀻 𑀧𑀸𑀧𑀁 𑀦𑀸𑀓𑀸𑀲𑀺𑁇 𑀫𑀳𑁄𑀲𑀥𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀢𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀢𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄? 𑀫𑀳𑁄𑀲𑀥𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑁄 𑀳𑀺𑀭𑀺𑀫𑀸 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀲𑀧𑀓𑁆𑀔𑁄 𑀫𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑀫𑁄 𑀲𑀻𑀮𑀯𑀸 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀻 𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀅𑀦𑀢𑀺𑀫𑀸𑀦𑀻 𑀅𑀦𑀼𑀲𑀽𑀬𑀓𑁄 𑀯𑀻𑀭𑀺𑀬𑀯𑀸 𑀆𑀬𑀽𑀳𑀓𑁄 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑁄 𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀻 𑀲𑀔𑀺𑀮𑁄 𑀦𑀺𑀯𑀸𑀢𑀯𑀼𑀢𑁆𑀢𑀺 𑀲𑀡𑁆𑀳𑁄 𑀅𑀲𑀞𑁄 𑀅𑀫𑀸𑀬𑀸𑀯𑀻 𑀅𑀢𑀺𑀩𑀼𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀫𑀸 𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑀺𑀢𑁂𑀲𑀻 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸𑀦𑀁 𑀧𑀢𑁆𑀣𑀺𑀢𑁄 𑀲𑀩𑁆𑀩𑀚𑀦𑀲𑁆𑀲 𑀥𑀦𑀯𑀸 𑀬𑀲𑀯𑀸𑁇 𑀫𑀳𑁄𑀲𑀥𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀢𑀺𑀬𑀸 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀦𑀺𑀫𑀦𑁆𑀢𑀓𑀁 𑀅𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀧𑀁 𑀦𑀸𑀓𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

---
