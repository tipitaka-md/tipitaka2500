# 𑁫𑁇𑁪𑁇𑁬 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀯𑀕𑁆𑀕

* 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀦𑀺𑀲𑀁𑀲𑀧𑀜𑁆𑀳

(155.)

882\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀫𑁂𑀢𑁆𑀢𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀸𑀬 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀬 𑀬𑀸𑀦𑀻𑀓𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀬 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀘𑀺𑀢𑀸𑀬 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑀬 𑀏𑀓𑀸𑀤𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑁂 𑀏𑀓𑀸𑀤𑀲? 𑀲𑀼𑀔𑀁 𑀲𑀼𑀧𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦 𑀧𑀸𑀧𑀓𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀦𑀸𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀺 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀺, 𑀢𑀼𑀯𑀝𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺, 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀦𑁆𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀽𑀧𑀕𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀲𑀸𑀫𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀺𑀳𑀸𑀭𑀻 𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁂𑀦 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀧𑀯𑀦𑁂 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑁄 𑀧𑀻𑀍𑀇𑀬𑀓𑁆𑀔𑁂𑀦 𑀭𑀜𑁆𑀜𑀸 𑀯𑀺𑀤𑁆𑀥𑁄 𑀯𑀺𑀲𑀧𑀻𑀢𑁂𑀦 𑀲𑀮𑁆𑀮𑁂𑀦 𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀢𑀺𑀢𑁄’𑀢𑀺𑁇

883\. 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀽𑀧𑀕𑁄 𑀳𑁄𑀢𑀻’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀲𑀸𑀫𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀺𑀳𑀸𑀭𑀻 𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁂𑀦 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀧𑀯𑀦𑁂 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑁄 𑀧𑀻𑀍𑀇𑀬𑀓𑁆𑀔𑁂𑀦 𑀭𑀜𑁆𑀜𑀸 𑀯𑀺𑀤𑁆𑀥𑁄 𑀯𑀺𑀲𑀧𑀻𑀢𑁂𑀦 𑀲𑀮𑁆𑀮𑁂𑀦 𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀢𑀺𑀢𑁄’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀲𑀸𑀫𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀺𑀳𑀸𑀭𑀻 𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁂𑀦 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀧𑀯𑀦𑁂 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑁄 𑀧𑀻𑀍𑀇𑀬𑀓𑁆𑀔𑁂𑀦 𑀭𑀜𑁆𑀜𑀸 𑀯𑀺𑀤𑁆𑀥𑁄 𑀯𑀺𑀲𑀧𑀻𑀢𑁂𑀦 𑀲𑀮𑁆𑀮𑁂𑀦 𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀢𑀺𑀢𑁄, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀫𑁂𑀢𑁆𑀢𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀻’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀲𑀼𑀦𑀺𑀧𑀼𑀡𑁄 𑀧𑀭𑀺𑀲𑀡𑁆𑀳𑁄 𑀲𑀼𑀔𑀼𑀫𑁄 𑀕𑀫𑁆𑀪𑀻𑀭𑁄, 𑀅𑀧𑀺 𑀲𑀼𑀦𑀺𑀧𑀼𑀡𑀸𑀦𑀁 𑀫𑀦𑀼𑀚𑀸𑀦𑀁 𑀕𑀢𑁆𑀢𑁂 𑀲𑁂𑀤𑀁 𑀫𑁄𑀘𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀯𑀺𑀚𑀝𑁂𑀳𑀺 𑀢𑀁 𑀫𑀳𑀸𑀚𑀝𑀸𑀚𑀝𑀺𑀢𑀁, 𑀅𑀦𑀸𑀕𑀢𑀸𑀦𑀁 𑀚𑀺𑀦𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀤𑁂𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

884\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀻’𑀢𑀺𑁇 𑀲𑀸𑀫𑁄 𑀘 𑀓𑀼𑀫𑀸𑀭𑁄 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀺𑀳𑀸𑀭𑀻 𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑁂𑀦 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀧𑀯𑀦𑁂 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑁄 𑀧𑀻𑀍𑀇𑀬𑀓𑁆𑀔𑁂𑀦 𑀭𑀜𑁆𑀜𑀸 𑀯𑀺𑀤𑁆𑀥𑁄 𑀯𑀺𑀲𑀧𑀻𑀢𑁂𑀦 𑀲𑀮𑁆𑀮𑁂𑀦 𑀢𑀢𑁆𑀣𑁂𑀯 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀢𑀺𑀢𑁄, 𑀢𑀢𑁆𑀣 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀺𑁇 𑀓𑀢𑀫𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀁? 𑀦𑁂𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸, 𑀲𑀸𑀫𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑀸𑀭𑁄 𑀖𑀝𑀁 𑀉𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑀡𑁂 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀫𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇

885\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀔𑀡𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑁂𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑀡𑁂 𑀅𑀕𑁆𑀕𑀺 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀑𑀓𑀸𑀲𑀁 𑀮𑀪𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸𑁇 𑀇𑀥, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀲𑀽𑀭𑁄 𑀅𑀪𑁂𑀚𑁆𑀚𑀓𑀯𑀘𑀚𑀸𑀮𑀺𑀓𑀁 𑀲𑀦𑁆𑀦𑀬𑁆𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀁 𑀑𑀢𑀭𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲 𑀲𑀭𑀸 𑀔𑀺𑀢𑁆𑀢𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑀦𑁆𑀢𑀺 𑀯𑀺𑀓𑀺𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀑𑀓𑀸𑀲𑀁 𑀮𑀪𑀦𑁆𑀢𑀺, 𑀦𑁂𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀲𑀽𑀭𑀲𑁆𑀲, 𑀅𑀪𑁂𑀚𑁆𑀚𑀓𑀯𑀘𑀚𑀸𑀮𑀺𑀓𑀸𑀬𑁂𑀲𑁄 𑀕𑀼𑀡𑁄, 𑀬𑀲𑁆𑀲 𑀲𑀭𑀸 𑀔𑀺𑀢𑁆𑀢𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑀦𑁆𑀢𑀺 𑀯𑀺𑀓𑀺𑀭𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑁂𑀢𑁂 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸𑁇

886\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀔𑀡𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑁂𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑀡𑁂 𑀅𑀕𑁆𑀕𑀺 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀑𑀓𑀸𑀲𑀁 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺, 𑀦𑁂𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸𑁇 𑀇𑀥 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀩𑁆𑀩𑀁 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀁 𑀫𑀽𑀮𑀁 𑀳𑀢𑁆𑀣𑁂 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀬𑀸𑀯 𑀢𑀁 𑀫𑀽𑀮𑀁 𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀸𑀯 𑀦 𑀅𑀜𑁆𑀜𑁄 𑀓𑁄𑀘𑀺 𑀧𑀓𑀢𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑁂𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑁄 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲, 𑀫𑀽𑀮𑀲𑁆𑀲𑁂𑀲𑁄 𑀕𑀼𑀡𑁄 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀦𑀲𑁆𑀲, 𑀬𑀁 𑀲𑁄 𑀧𑀓𑀢𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑁂 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑁂𑀢𑁂 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸𑁇

887\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀔𑀡𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑁂𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑀡𑁂 𑀅𑀕𑁆𑀕𑀺 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀑𑀓𑀸𑀲𑀁 𑀮𑀪𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸𑁇 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀲𑀼𑀓𑀢𑀁 𑀫𑀳𑀸𑀮𑁂𑀡𑀫𑀦𑀼𑀧𑁆𑀧𑀯𑀺𑀝𑁆𑀞𑀁 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀢𑁂𑀫𑀬𑀺𑀢𑀼𑀁, 𑀦𑁂𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑁄 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲, 𑀫𑀳𑀸𑀮𑁂𑀡𑀲𑁆𑀲𑁂𑀲𑁄 𑀕𑀼𑀡𑁄, 𑀬𑀁 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀦 𑀢𑀁 𑀢𑁂𑀫𑁂𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑁂𑀢𑁂 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸𑁇

888\. 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀔𑀡𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑁂𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑀡𑁂 𑀅𑀕𑁆𑀕𑀺 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀳𑀺𑀢𑀓𑀸𑀫𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀢𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀅𑀳𑀺𑀢𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀦𑁂𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸𑀬𑁂𑀢𑁂 𑀕𑀼𑀡𑀸”𑀢𑀺𑁇 “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀩𑁆𑀩𑀧𑀸𑀧𑀦𑀺𑀯𑀸𑀭𑀡𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸”𑀢𑀺𑁇 “𑀲𑀩𑁆𑀩𑀓𑀼𑀲𑀮𑀕𑀼𑀡𑀸𑀯𑀳𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸 𑀳𑀺𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀅𑀳𑀺𑀢𑀸𑀦𑀫𑁆𑀧𑀺, 𑀬𑁂 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀩𑀤𑁆𑀥𑀸, 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀪𑀸𑀯𑀦𑀸 𑀲𑀁𑀯𑀺𑀪𑀚𑀺𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
