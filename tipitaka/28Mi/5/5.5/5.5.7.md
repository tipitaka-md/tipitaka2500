# 𑁫𑁇𑁫𑁇𑁭 𑀖𑀝𑀺𑀓𑀸𑀭𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

* 𑀖𑀝𑀺𑀓𑀸𑀭𑀧𑀜𑁆𑀳

(166.)

991\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀖𑀝𑀺𑀓𑀸𑀭𑀲𑁆𑀲 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀲𑁆𑀲 𑀆𑀯𑁂𑀲𑀦𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑁂𑀫𑀸𑀲𑀁 𑀆𑀓𑀸𑀲𑀘𑁆𑀙𑀤𑀦𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺, 𑀦 𑀤𑁂𑀯𑁄𑀢𑀺𑀯𑀲𑁆𑀲𑀻’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀼𑀝𑀺 𑀑𑀯𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀓𑀺𑀲𑁆𑀲 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀏𑀯𑀫𑀼𑀲𑁆𑀲𑀦𑁆𑀦𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀲𑁆𑀲 𑀓𑀼𑀝𑀺 𑀑𑀯𑀲𑁆𑀲𑀢𑀺, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀦𑀸𑀫 𑀲𑁄 𑀆𑀦𑀼𑀪𑀸𑀯𑁄 𑀇𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄? 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀖𑀝𑀺𑀓𑀸𑀭𑀲𑁆𑀲 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀲𑁆𑀲 𑀆𑀯𑁂𑀲𑀦𑀁 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀁 𑀆𑀓𑀸𑀲𑀘𑁆𑀙𑀤𑀦𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀼𑀝𑀺 𑀑𑀯𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀼𑀝𑀺 𑀑𑀯𑀲𑁆𑀲𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀖𑀝𑀺𑀓𑀸𑀭𑀲𑁆𑀲 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀲𑁆𑀲 𑀆𑀯𑁂𑀲𑀦𑀁 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀓𑀁 𑀅𑀳𑁄𑀲𑀺 𑀆𑀓𑀸𑀲𑀘𑁆𑀙𑀤𑀦𑀦𑁆’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

992\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀖𑀝𑀺𑀓𑀸𑀭𑀲𑁆𑀲 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀲𑁆𑀲 𑀆𑀯𑁂𑀲𑀦𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑁂𑀫𑀸𑀲𑀁 𑀆𑀓𑀸𑀲𑀘𑁆𑀙𑀤𑀦𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺, 𑀦 𑀤𑁂𑀯𑁄𑀢𑀺𑀯𑀲𑁆𑀲𑀻’𑀢𑀺𑁇 𑀪𑀡𑀺𑀢𑀜𑁆𑀘—  ‘𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀼𑀝𑀺 𑀑𑀯𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀖𑀝𑀺𑀓𑀸𑀭𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁄 𑀲𑀻𑀮𑀯𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁄 𑀉𑀲𑁆𑀲𑀦𑁆𑀦𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁄 𑀅𑀦𑁆𑀥𑁂 𑀚𑀺𑀡𑁆𑀡𑁂 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀧𑁄𑀲𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀅𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀬𑁂𑀯𑀲𑁆𑀲 𑀖𑀭𑁂 𑀢𑀺𑀡𑀁 𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀼𑀝𑀺𑀁 𑀙𑀸𑀤𑁂𑀲𑀼𑀁, 𑀲𑁄 𑀢𑁂𑀦 𑀢𑀺𑀡𑀳𑀭𑀡𑁂𑀦 𑀅𑀓𑀫𑁆𑀧𑀺𑀢𑀁 𑀅𑀲𑀜𑁆𑀘𑀮𑀺𑀢𑀁 𑀲𑀼𑀲𑀡𑁆𑀞𑀺𑀢𑀁 𑀯𑀺𑀧𑀼𑀮𑀫𑀲𑀫𑀁 𑀧𑀻𑀢𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀪𑀺𑀬𑁆𑀬𑁄 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀜𑁆𑀘 𑀅𑀢𑀼𑀮𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀲𑀺 ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀫𑁄 𑀲𑀼𑀯𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁄’𑀢𑀺, 𑀢𑁂𑀦 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁄𑁇 𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀢𑀸𑀯𑀢𑀓𑁂𑀦 𑀯𑀺𑀓𑀸𑀭𑁂𑀦 𑀘𑀮𑀢𑀺𑁇

993\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀺𑀦𑁂𑀭𑀼 𑀕𑀺𑀭𑀺𑀭𑀸𑀚𑀸 𑀅𑀦𑁂𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀯𑀸𑀢𑀲𑀫𑁆𑀧𑀳𑀸𑀭𑁂𑀦𑀧𑀺 𑀦 𑀓𑀫𑁆𑀧𑀢𑀺 𑀦 𑀘𑀮𑀢𑀺, 𑀫𑀳𑁄𑀤𑀥𑀺 𑀯𑀭𑀧𑁆𑀧𑀯𑀭𑀲𑀸𑀕𑀭𑁄 𑀅𑀦𑁂𑀓𑀲𑀢𑀦𑀳𑀼𑀢𑀫𑀳𑀸𑀕𑀗𑁆𑀕𑀸𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺𑀧𑀺 𑀦 𑀧𑀽𑀭𑀢𑀺 𑀦 𑀯𑀺𑀓𑀸𑀭𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀦 𑀢𑀸𑀯𑀢𑀓𑁂𑀦 𑀯𑀺𑀓𑀸𑀭𑁂𑀦 𑀘𑀮𑀢𑀺𑁇

994\. 𑀬𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀼𑀝𑀺 𑀑𑀯𑀲𑁆𑀲𑀢𑀺, 𑀢𑀁 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬𑁇 𑀤𑁆𑀯𑁂𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑀸 𑀲𑀬𑀁 𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁 𑀧𑀘𑁆𑀘𑀬𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺, ‘𑀅𑀬𑀁 𑀅𑀕𑁆𑀕𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀲𑀢𑁆𑀣𑀸’𑀢𑀺 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀩𑁆𑀩𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀢𑁄 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺, 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀯𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀻𑀢𑀺 ‘𑀫𑀸 𑀅𑀜𑁆𑀜𑁂 𑀉𑀧𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑀸 𑀲𑀬𑀁 𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁 𑀧𑀘𑁆𑀘𑀬𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑁄 𑀯𑀸 𑀢𑀁 𑀓𑀼𑀝𑀺𑀁 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀁 𑀓𑀭𑁂𑀬𑁆𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀸 𑀲𑀬𑀁 𑀯𑀸, 𑀲𑀸𑀯𑀚𑁆𑀚𑀁 𑀪𑀯𑁂𑀬𑁆𑀬 𑀢𑀁𑀬𑁂𑀯 𑀓𑀭𑀡𑀁 𑀲𑀤𑁄𑀲𑀁 𑀲𑀦𑀺𑀕𑁆𑀕𑀳𑀁, 𑀇𑀫𑁂 𑀯𑀺𑀪𑀽𑀢𑀁 𑀓𑀢𑁆𑀯𑀸 𑀮𑁄𑀓𑀁 𑀲𑀫𑁆𑀫𑁄𑀳𑁂𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀓𑀢𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀁 𑀓𑀭𑀡𑀁 𑀯𑀚𑁆𑀚𑀦𑀻𑀬𑀁𑁇 𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀁 𑀬𑀸𑀘𑀦𑁆𑀢𑀺, 𑀢𑀸𑀬 𑀅𑀯𑀢𑁆𑀣𑀼𑀬𑀸𑀘𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀬𑀸 𑀪𑀯𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
