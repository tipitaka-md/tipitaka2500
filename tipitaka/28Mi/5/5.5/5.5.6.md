# 𑁫𑁇𑁫𑁇𑁬 𑀙𑀤𑁆𑀤𑀦𑁆𑀢𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑀸𑀭𑀩𑁆𑀪𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

* 𑀙𑀤𑁆𑀤𑀦𑁆𑀢𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑀸𑀭𑀩𑁆𑀪𑀧𑀜𑁆𑀳

(165.)

984\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀙𑀤𑁆𑀤𑀦𑁆𑀢𑁄 𑀦𑀸𑀕𑀭𑀸𑀚𑀸—

985\. _‘𑀯𑀥𑀺𑀲𑁆𑀲𑀫𑁂𑀢𑀦𑁆𑀢𑀺 𑀧𑀭𑀸𑀫𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀸𑀲𑀸𑀯𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺 𑀥𑀚𑀁 𑀇𑀲𑀻𑀦𑀁;_  
_𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀼𑀤𑀧𑀸𑀤𑀺 𑀲𑀜𑁆𑀜𑀸,_  
_𑀅𑀭𑀳𑀤𑁆𑀥𑀚𑁄 𑀲𑀩𑁆𑀪𑀺 𑀅𑀯𑀚𑁆𑀛𑀭𑀽𑀧𑁄’𑀢𑀺𑁇_  


986\. 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑀫𑀸𑀡𑀯𑁄 𑀲𑀫𑀸𑀦𑁄 𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀡𑁆𑀟𑀓𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀡𑀓𑀯𑀸𑀤𑁂𑀦 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀓𑀸𑀲𑀸𑀯𑀁 𑀅𑀪𑀺𑀧𑀽𑀚𑀬𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁂𑀦 𑀫𑀸𑀡𑀯𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀫𑀼𑀡𑁆𑀟𑀓𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀡𑀓𑀯𑀸𑀤𑁂𑀦 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑁄 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑁄’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁂𑀦 𑀫𑀸𑀡𑀯𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀫𑀼𑀡𑁆𑀟𑀓𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀡𑀓𑀯𑀸𑀤𑁂𑀦 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑁄 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑁄, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀙𑀤𑁆𑀤𑀦𑁆𑀢𑁂𑀦 𑀦𑀸𑀕𑀭𑀸𑀚𑁂𑀦 𑀓𑀸𑀲𑀸𑀯𑀁 𑀧𑀽𑀚𑀺𑀢𑀦𑁆’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁂𑀦 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀓𑀓𑁆𑀔𑀍𑀅𑀔𑀭𑀓𑀝𑀼𑀓𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀫𑀸𑀦𑁂𑀦 𑀮𑀼𑀤𑁆𑀤𑀓𑁂𑀦 𑀦𑀺𑀯𑀢𑁆𑀣𑀁 𑀓𑀸𑀲𑀸𑀯𑀁 𑀧𑀽𑀚𑀺𑀢𑀁, 𑀓𑀺𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀜𑀸𑀡𑁄 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀸𑀬 𑀩𑁄𑀥𑀺𑀬𑀸 𑀓𑀲𑁆𑀲𑀧𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑀲𑀩𑀮𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁 𑀉𑀤𑀺𑀢𑁄𑀤𑀺𑀢𑀁 𑀚𑀮𑀺𑀢𑀩𑁆𑀬𑀸𑀫𑁄𑀪𑀸𑀲𑀁 𑀧𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑀁 𑀧𑀯𑀭𑀭𑀼𑀘𑀺𑀭𑀓𑀸𑀲𑀺𑀓𑀓𑀸𑀲𑀸𑀯𑀫𑀪𑀺𑀧𑀸𑀭𑀼𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦 𑀧𑀽𑀚𑀬𑀺? 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄𑀢𑀺𑁇

987\. 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀙𑀤𑁆𑀤𑀦𑁆𑀢𑁄 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 ‘𑀯𑀥𑀺𑀲𑁆𑀲𑀫𑁂𑀢𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀯𑀚𑁆𑀛𑀭𑀽𑀧𑁄’𑀢𑀺𑁇 𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁂𑀦 𑀘 𑀫𑀸𑀡𑀯𑁂𑀦 𑀓𑀲𑁆𑀲𑀧𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀫𑀼𑀡𑁆𑀟𑀓𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀡𑀓𑀯𑀸𑀤𑁂𑀦 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑀼𑀝𑁆𑀞𑁄 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑁄, 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀚𑀸𑀢𑀺𑀯𑀲𑁂𑀦 𑀓𑀼𑀮𑀯𑀲𑁂𑀦𑁇 𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀸𑀡𑀯𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁂 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂 𑀓𑀼𑀮𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀪𑀕𑀺𑀦𑀺𑀪𑀸𑀢𑀭𑁄 𑀤𑀸𑀲𑀺𑀤𑀸𑀲𑀘𑁂𑀝𑀓𑀧𑀭𑀺𑀯𑀸𑀭𑀓𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑁂𑀯𑀢𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀕𑀭𑀼𑀓𑀸, 𑀢𑁂 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯 𑀉𑀢𑁆𑀢𑀫𑀸 𑀧𑀯𑀭𑀸’𑀢𑀺 𑀅𑀯𑀲𑁂𑀲𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀕𑀭𑀳𑀦𑁆𑀢𑀺 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁄 𑀫𑀸𑀡𑀯𑁄 𑀖𑀝𑀺𑀓𑀸𑀭𑁂𑀦 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑁂𑀦 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀧𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑁄 𑀏𑀯𑀫𑀸𑀳—  ‘𑀓𑀺𑀁 𑀧𑀦 𑀢𑁂𑀦 𑀫𑀼𑀡𑁆𑀟𑀓𑁂𑀦 𑀲𑀫𑀡𑀓𑁂𑀦 𑀤𑀺𑀝𑁆𑀞𑁂𑀦𑀸’𑀢𑀺𑁇

988\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀫𑀢𑀁 𑀯𑀺𑀲𑀫𑀸𑀲𑀚𑁆𑀚 𑀢𑀺𑀢𑁆𑀢𑀓𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸 𑀘 𑀲𑀻𑀢𑁄𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀺𑀫𑀸𑀲𑀚𑁆𑀚 𑀉𑀡𑁆𑀳𑀁 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁄 𑀫𑀸𑀡𑀯𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁂 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂 𑀓𑀼𑀮𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄, 𑀲𑁄 𑀓𑀼𑀮𑀯𑀲𑁂𑀦 𑀅𑀦𑁆𑀥𑁄 𑀳𑀼𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀓𑁆𑀓𑁄𑀲𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑁇

989\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀮𑀺𑀢𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑁄 𑀉𑀤𑀓𑀫𑀸𑀲𑀚𑁆𑀚 𑀉𑀧𑀳𑀢𑀧𑁆𑀧𑀪𑀸𑀢𑁂𑀚𑁄 𑀲𑀻𑀢𑀮𑁄 𑀓𑀸𑀍𑀅𑀓𑁄 𑀪𑀯𑀢𑀺 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀦𑀺𑀕𑁆𑀕𑀼𑀡𑁆𑀟𑀺𑀨𑀮𑀲𑀤𑀺𑀲𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑁄𑀢𑀺𑀧𑀸𑀮𑁄 𑀫𑀸𑀡𑀯𑁄 𑀧𑀼𑀜𑁆𑀜𑀯𑀸 𑀲𑀤𑁆𑀥𑁄 𑀜𑀸𑀡𑀯𑀺𑀧𑀼𑀮𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁂 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂 𑀓𑀼𑀮𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄, 𑀲𑁄 𑀓𑀼𑀮𑀯𑀲𑁂𑀦 𑀅𑀦𑁆𑀥𑁄 𑀳𑀼𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀓𑁆𑀓𑁄𑀲𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀺, 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀘 𑀩𑀼𑀤𑁆𑀥𑀕𑀼𑀡𑀫𑀜𑁆𑀜𑀸𑀬 𑀘𑁂𑀝𑀓𑀪𑀽𑀢𑁄 𑀯𑀺𑀬 𑀅𑀳𑁄𑀲𑀺, 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀘 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀽𑀧𑀕𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
