# 𑁫𑁇𑁫𑁇𑁩 𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀩𑀼𑀤𑁆𑀥𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀧𑀜𑁆𑀳

(162.)

958. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀲𑀤𑀸 𑀧𑀬𑀢𑀧𑀸𑀡𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀤𑁂𑀳𑀥𑀭𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀪𑀺𑀲𑀓𑁆𑀓𑁄 𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑁄’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀏𑀢𑀤𑀕𑁆𑀕𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀸𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀩𑀸𑀓𑀼𑀮𑁄’𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀘 𑀲𑀭𑀻𑀭𑁂 𑀩𑀳𑀼𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀆𑀩𑀸𑀥𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀏𑀢𑀤𑀕𑁆𑀕𑀁…𑀧𑁂…  𑀩𑀸𑀓𑀼𑀮𑁄’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀣𑁂𑀭𑁄 𑀩𑀸𑀓𑀼𑀮𑁄 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑁄, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺…𑀧𑁂…  𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑁄’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

959. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺…𑀧𑁂…  𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑁄’𑀢𑀺, 𑀪𑀡𑀺𑀢𑀜𑁆𑀘—  ‘𑀏𑀢𑀤𑀕𑁆𑀕𑀁…𑀧𑁂…  𑀩𑀸𑀓𑀼𑀮𑁄’𑀢𑀺, 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀩𑀸𑀳𑀺𑀭𑀸𑀦𑀁 𑀆𑀕𑀫𑀸𑀦𑀁 𑀅𑀥𑀺𑀕𑀫𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀻𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀺 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀢𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀁𑁇

960. 𑀲𑀦𑁆𑀢𑀺 𑀔𑁄 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀺𑀓𑀸, 𑀢𑁂 𑀞𑀸𑀦𑁂𑀦 𑀘𑀗𑁆𑀓𑀫𑁂𑀦 𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀦𑁆𑀢𑀺, 𑀪𑀕𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀞𑀸𑀦𑁂𑀦 𑀘𑀗𑁆𑀓𑀫𑁂𑀦 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀲𑀬𑀦𑁂𑀦 𑀤𑀺𑀯𑀸𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀢𑀺, 𑀬𑁂 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀺𑀓𑀸, 𑀢𑁂 𑀢𑁂𑀦 𑀅𑀗𑁆𑀕𑁂𑀦 𑀅𑀢𑀺𑀭𑁂𑀓𑀸𑁇

961. 𑀲𑀦𑁆𑀢𑀺 𑀔𑁄 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀏𑀓𑀸𑀲𑀦𑀺𑀓𑀸, 𑀢𑁂 𑀚𑀻𑀯𑀺𑀢𑀳𑁂𑀢𑀼𑀧𑀺 𑀤𑀼𑀢𑀺𑀬𑀁 𑀪𑁄𑀚𑀦𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀕𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀯 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀬𑁂 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀓𑀸𑀲𑀦𑀺𑀓𑀸, 𑀢𑁂 𑀢𑁂𑀦 𑀅𑀗𑁆𑀕𑁂𑀦 𑀅𑀢𑀺𑀭𑁂𑀓𑀸, 𑀅𑀦𑁂𑀓𑀯𑀺𑀥𑀸𑀦𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀸𑀦𑀺 𑀓𑀸𑀭𑀡𑀸𑀦𑀺 𑀢𑁂𑀲𑀁 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀢𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀡𑀺𑀢𑀸𑀦𑀺𑁇 𑀪𑀕𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀤𑀲𑀳𑀺 𑀘 𑀩𑀮𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀙𑀳𑀺 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑁂𑀳𑀺 𑀜𑀸𑀡𑁂𑀳𑀺, 𑀓𑁂𑀯𑀮𑁂 𑀘 𑀩𑀼𑀤𑁆𑀥𑀯𑀺𑀲𑀬𑁂 𑀢𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀡𑀺𑀢𑀁—  ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺…𑀧𑁂…  𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑁄’𑀢𑀺𑁇

962. 𑀇𑀥, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀏𑀓𑁄 𑀚𑀸𑀢𑀺𑀫𑀸 𑀳𑁄𑀢𑀺, 𑀏𑀓𑁄 𑀥𑀦𑀯𑀸, 𑀏𑀓𑁄 𑀯𑀺𑀚𑁆𑀚𑀯𑀸, 𑀏𑀓𑁄 𑀲𑀺𑀧𑁆𑀧𑀯𑀸, 𑀏𑀓𑁄 𑀲𑀽𑀭𑁄, 𑀏𑀓𑁄 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄, 𑀲𑀩𑁆𑀩𑁂𑀧𑁂𑀢𑁂 𑀅𑀪𑀺𑀪𑀯𑀺𑀬 𑀭𑀸𑀚𑀸𑀬𑁂𑀯 𑀢𑁂𑀲𑀁 𑀉𑀢𑁆𑀢𑀫𑁄 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀸 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑁄 𑀚𑁂𑀝𑁆𑀞𑁄 𑀲𑁂𑀝𑁆𑀞𑁄𑁇

963. 𑀬𑀁 𑀧𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀩𑀸𑀓𑀼𑀮𑁄 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀯𑀲𑁂𑀦, 𑀲𑁄 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑁄𑀫𑀤𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀉𑀤𑀭𑀯𑀸𑀢𑀸𑀩𑀸𑀥𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀘 𑀪𑀕𑀯𑀢𑁄 𑀅𑀝𑁆𑀞𑀲𑀝𑁆𑀞𑀺𑀬𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀢𑀺𑀡𑀧𑀼𑀧𑁆𑀨𑀓𑀭𑁄𑀕𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀲𑀬𑀁 𑀢𑀸𑀧𑀲𑁄 𑀲𑀫𑀸𑀦𑁄 𑀦𑀸𑀦𑀸𑀪𑁂𑀲𑀚𑁆𑀚𑁂𑀳𑀺 𑀢𑀁 𑀩𑁆𑀬𑀸𑀥𑀺𑀁 𑀅𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀢𑀁 𑀧𑀢𑁆𑀢𑁄, 𑀪𑀡𑀺𑀢𑁄 𑀘 ‘𑀏𑀢𑀤𑀕𑁆𑀕𑀁…𑀧𑁂…  𑀩𑀸𑀓𑀼𑀮𑁄’𑀢𑀺𑁇

964. 𑀪𑀕𑀯𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁆𑀬𑀸𑀥𑀺𑀫𑁆𑀳𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁂𑀧𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁂𑀧𑀺 𑀥𑀼𑀢𑀗𑁆𑀕𑀁 𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑁂𑀧𑀺 𑀅𑀦𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁂𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑀺𑀲𑁄 𑀓𑁄𑀘𑀺 𑀲𑀢𑁆𑀢𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑀮𑀜𑁆𑀙𑀓𑁂—

965. ‘𑀬𑀸𑀯𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀧𑀤𑀸 𑀯𑀸 𑀤𑁆𑀯𑀺𑀧𑀤𑀸 𑀯𑀸 𑀘𑀢𑀼𑀧𑁆𑀧𑀤𑀸 𑀯𑀸 𑀩𑀳𑀼𑀧𑁆𑀧𑀤𑀸 𑀯𑀸 𑀭𑀽𑀧𑀺𑀦𑁄 𑀯𑀸 𑀅𑀭𑀽𑀧𑀺𑀦𑁄 𑀯𑀸 𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀯𑀸 𑀅𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀯𑀸 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀻𑀦𑀸𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀯𑀸, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀢𑁂𑀲𑀁 𑀅𑀕𑁆𑀕𑀫𑀓𑁆𑀔𑀸𑀬𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄’”𑀢𑀺𑁇
