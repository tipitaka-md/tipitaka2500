# 𑁫𑁇𑁫𑁇𑁧𑁧 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀸𑀘𑀭𑀺𑀬𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀸𑀘𑀭𑀺𑀬𑀧𑀜𑁆𑀳

(170.)

1043. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—

1044. _‘𑀦 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺𑁇_  


1045. 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀫𑁂 𑀲𑀫𑀸𑀦𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀁 𑀫𑀁 𑀲𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀫𑀲𑀫𑀁 𑀞𑀧𑁂𑀲𑀺, 𑀉𑀍𑀆𑀭𑀸𑀬 𑀘 𑀫𑀁 𑀧𑀽𑀚𑀸𑀬 𑀧𑀽𑀚𑁂𑀲𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀦 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀺, 𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀫𑁂 𑀲𑀫𑀸𑀦𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀁 𑀫𑀁 𑀲𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀫𑀲𑀫𑀁 𑀞𑀧𑁂𑀲𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀫𑁂 𑀲𑀫𑀸𑀦𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀁 𑀫𑀁 𑀲𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀫𑀲𑀫𑀁 𑀞𑀧𑁂𑀲𑀻’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀦 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀺, 𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

1046. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦

1047. _‘𑀦 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄’𑀢𑀺𑁇_  


1048. 𑀪𑀡𑀺𑀢𑀜𑁆𑀘—  ‘𑀇𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀫𑁂 𑀲𑀫𑀸𑀦𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀁 𑀫𑀁 𑀲𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀫𑀲𑀫𑀁 𑀞𑀧𑁂𑀲𑀺, 𑀉𑀍𑀆𑀭𑀸𑀬 𑀘 𑀫𑀁 𑀧𑀽𑀚𑀸𑀬 𑀧𑀽𑀚𑁂𑀲𑀻’𑀢𑀺𑁇

1049. 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀯𑀘𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀯 𑀲𑀢𑁄 𑀆𑀘𑀭𑀺𑀬𑀪𑀸𑀯𑀁 𑀲𑀦𑁆𑀥𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀁𑁇

1050. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀢𑁄 𑀆𑀘𑀭𑀺𑀬𑀸, 𑀬𑁂𑀳𑀺 𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑁄 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀤𑀺𑀯𑀲𑀁 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀲𑀺𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀬𑁂 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀝𑁆𑀞 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀚𑀸𑀢𑀫𑀢𑁆𑀢𑁂 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂 𑀮𑀓𑁆𑀔𑀡𑀸𑀦𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀺𑀁𑀲𑀼, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀸𑀫𑁄 𑀥𑀚𑁄 𑀮𑀓𑁆𑀔𑀡𑁄 𑀫𑀦𑁆𑀢𑀻 𑀬𑀜𑁆𑀜𑁄 𑀲𑀼𑀬𑀸𑀫𑁄 𑀲𑀼𑀪𑁄𑀚𑁄 𑀲𑀼𑀤𑀢𑁆𑀢𑁄𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀲𑁄𑀢𑁆𑀣𑀺𑀁 𑀧𑀯𑁂𑀤𑀬𑀺𑀢𑁆𑀯𑀸 𑀭𑀓𑁆𑀔𑀸𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼, 𑀢𑁂 𑀘 𑀧𑀞𑀫𑀁 𑀆𑀘𑀭𑀺𑀬𑀸𑁇

1051. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀭𑀸𑀚𑀸 𑀬𑀁 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀪𑀺𑀚𑀸𑀢𑀁 𑀉𑀤𑀺𑀘𑁆𑀘𑀚𑀸𑀢𑀺𑀫𑀦𑁆𑀢𑀁 𑀧𑀤𑀓𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀙𑀍𑀅𑀗𑁆𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀩𑁆𑀩𑀫𑀺𑀢𑁆𑀢𑀁 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀉𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀲𑁄𑀯𑀡𑁆𑀡𑁂𑀦 𑀪𑀺𑀗𑁆𑀕𑀸𑀭𑁂𑀦 𑀉𑀤𑀓𑀁 𑀑𑀡𑁄𑀚𑁂𑀢𑁆𑀯𑀸 ‘𑀇𑀫𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑁂𑀳𑀻’𑀢𑀺 𑀅𑀤𑀸𑀲𑀺, 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀘𑀭𑀺𑀬𑁄𑁇

1052. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀸 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀲𑀺, 𑀬𑀲𑁆𑀲𑀸 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑁄 𑀉𑀩𑁆𑀩𑀺𑀕𑁆𑀕𑁄 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀔𑀡𑁂 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑀺, 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀆𑀘𑀭𑀺𑀬𑁄𑁇

1053. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀍𑀆𑀭𑁄 𑀓𑀸𑀮𑀸𑀫𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀺, 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀆𑀘𑀭𑀺𑀬𑁄𑁇

1054. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑁄 𑀭𑀸𑀫𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀺, 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀆𑀘𑀭𑀺𑀬𑁄𑁇

1055. 𑀇𑀫𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀢𑁄 𑀧𑀜𑁆𑀘 𑀆𑀘𑀭𑀺𑀬𑀸𑁇 𑀢𑁂 𑀘 𑀧𑀦 𑀆𑀘𑀭𑀺𑀬𑀸 𑀮𑁄𑀓𑀺𑀬𑁂 𑀥𑀫𑁆𑀫𑁂𑁇 𑀇𑀫𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁂 𑀥𑀫𑁆𑀫𑁂 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸𑀬 𑀦𑀢𑁆𑀣𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑀓𑁄, 𑀲𑀬𑀫𑁆𑀪𑀽, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀸𑀘𑀭𑀺𑀬𑀓𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀓𑀸𑀭𑀡𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—

1056. _‘𑀦 𑀫𑁂 𑀆𑀘𑀭𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀤𑀺𑀲𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑁂 𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄’”𑀢𑀺𑁇_  


1057. “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

1058. 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀸𑀘𑀭𑀺𑀬𑀧𑀜𑁆𑀳𑁄 𑀏𑀓𑀸𑀤𑀲𑀫𑁄𑁇
