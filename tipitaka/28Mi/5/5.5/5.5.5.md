# 𑁫𑁇𑁫𑁇𑁫 𑀩𑀼𑀤𑁆𑀥𑀅𑀯𑀺𑀳𑁂𑀞𑀓𑀧𑀜𑁆𑀳

* 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

* 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

* 𑀩𑀼𑀤𑁆𑀥𑀅𑀯𑀺𑀳𑁂𑀞𑀓𑀧𑀜𑁆𑀳

(164.)

976\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀺𑀳𑁂𑀞𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀮𑁄𑀫𑀲𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀇𑀲𑀺 𑀲𑀫𑀸𑀦𑁄 𑀅𑀦𑁂𑀓𑀲𑀢𑁂 𑀧𑀸𑀡𑁂 𑀖𑀸𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀺𑀳𑁂𑀞𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀮𑁄𑀫𑀲𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀇𑀲𑀺𑀦𑀸 𑀅𑀦𑁂𑀓𑀲𑀢𑁂 𑀧𑀸𑀡𑁂 𑀖𑀸𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑀢𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 ‘𑀮𑁄𑀫𑀲𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀇𑀲𑀺𑀦𑀸 𑀅𑀦𑁂𑀓𑀲𑀢𑁂 𑀧𑀸𑀡𑁂 𑀖𑀸𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑀢𑀁’, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀺𑀳𑁂𑀞𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

977\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀳𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀅𑀯𑀺𑀳𑁂𑀞𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑀦𑁆’𑀢𑀺, ‘𑀮𑁄𑀫𑀲𑀓𑀲𑁆𑀲𑀧𑁂𑀦 𑀇𑀲𑀺𑀦𑀸 𑀅𑀦𑁂𑀓𑀲𑀢𑁂 𑀧𑀸𑀡𑁂 𑀖𑀸𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑀢𑀁’, 𑀢𑀜𑁆𑀘 𑀧𑀦 𑀭𑀸𑀕𑀯𑀲𑁂𑀦 𑀯𑀺𑀲𑀜𑁆𑀜𑀺𑀦𑀸, 𑀦𑁄 𑀲𑀘𑁂𑀢𑀦𑁂𑀦𑀸”𑀢𑀺𑁇

978\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀸𑀡𑀁 𑀳𑀦𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀭𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑀯𑀲𑁂𑀦 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑁄 𑀤𑁄𑀲𑀯𑀲𑁂𑀦 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀫𑀽𑀍𑀳𑁄 𑀫𑁄𑀳𑀯𑀲𑁂𑀦 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀫𑀸𑀦𑀻 𑀫𑀸𑀦𑀯𑀲𑁂𑀦 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀮𑀼𑀤𑁆𑀥𑁄 𑀮𑁄𑀪𑀯𑀲𑁂𑀦 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑁄 𑀚𑀻𑀯𑀺𑀓𑀢𑁆𑀣𑀸𑀬 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀳𑀲𑁆𑀲𑀯𑀲𑁂𑀦 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀭𑀸𑀚𑀸 𑀯𑀺𑀦𑀬𑀦𑀯𑀲𑁂𑀦 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀸𑀡𑀁 𑀳𑀦𑀦𑁆𑀢𑀺𑁇 𑀧𑀸𑀓𑀢𑀺𑀓𑀁𑀬𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀓𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀸𑀓𑀢𑀺𑀓𑀁 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀓𑀢𑀁, 𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀧𑀓𑀢𑀺𑀪𑀸𑀯𑁂𑀦 𑀑𑀦𑀫𑁂𑀬𑁆𑀬 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑀢𑀼𑀁, 𑀦 𑀬𑀺𑀫𑀁 𑀕𑀸𑀣𑀁 𑀪𑀡𑁂𑀬𑁆𑀬—

979\. _‘𑀲𑀲𑀫𑀼𑀤𑁆𑀤𑀧𑀭𑀺𑀬𑀸𑀬𑀁,_  
_𑀫𑀳𑀺𑀁 𑀲𑀸𑀕𑀭𑀓𑀼𑀡𑁆𑀟𑀮𑀁;_  
_𑀦 𑀇𑀘𑁆𑀙𑁂 𑀲𑀳 𑀦𑀺𑀦𑁆𑀤𑀸𑀬,_  
_𑀏𑀯𑀁 𑀲𑁂𑀬𑁆𑀳 𑀯𑀺𑀚𑀸𑀦𑀳𑀻’𑀢𑀺𑁇_  


980\. 𑀏𑀯𑀁𑀯𑀸𑀤𑀻, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀲𑀳 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀘𑀦𑁆𑀤𑀯𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀲𑀜𑁆𑀜𑀻 𑀅𑀳𑁄𑀲𑀺 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀭𑀢𑁆𑀢𑁄 𑀯𑀺𑀲𑀜𑁆𑀜𑀺𑀪𑀽𑀢𑁄 𑀆𑀓𑀼𑀮𑀸𑀓𑀼𑀮𑁄 𑀢𑀼𑀭𑀺𑀢𑀢𑀼𑀭𑀺𑀢𑁄 𑀢𑁂𑀦 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀪𑀦𑁆𑀢𑀮𑀼𑀍𑀇𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀧𑀲𑀼𑀖𑀸𑀢𑀕𑀮𑀭𑀼𑀳𑀺𑀭𑀲𑀜𑁆𑀘𑀬𑀁 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑁇

981\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑁄 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀚𑀮𑀺𑀢𑀫𑁆𑀧𑀺 𑀚𑀸𑀢𑀯𑁂𑀤𑀁 𑀅𑀓𑁆𑀓𑀫𑀢𑀺, 𑀓𑀼𑀧𑀺𑀢𑀫𑁆𑀧𑀺 𑀆𑀲𑀻𑀯𑀺𑀲𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀳𑀢𑁆𑀣𑀺𑀁 𑀉𑀧𑁂𑀢𑀺, 𑀲𑀫𑀼𑀤𑁆𑀤𑀫𑁆𑀧𑀺 𑀅𑀢𑀻𑀭𑀤𑀲𑁆𑀲𑀺𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺, 𑀘𑀦𑁆𑀤𑀦𑀺𑀓𑀫𑁆𑀧𑀺 𑀑𑀍𑀇𑀕𑀮𑁆𑀮𑀫𑁆𑀧𑀺 𑀑𑀫𑀤𑁆𑀤𑀢𑀺, 𑀓𑀡𑁆𑀝𑀓𑀸𑀥𑀸𑀦𑀫𑁆𑀧𑀺 𑀅𑀪𑀺𑀭𑀼𑀳𑀢𑀺, 𑀧𑀧𑀸𑀢𑁂𑀧𑀺 𑀧𑀢𑀢𑀺, 𑀅𑀲𑀼𑀘𑀺𑀫𑁆𑀧𑀺 𑀪𑀓𑁆𑀔𑁂𑀢𑀺, 𑀦𑀕𑁆𑀕𑁄𑀧𑀺 𑀭𑀣𑀺𑀬𑀸 𑀘𑀭𑀢𑀺, 𑀅𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀁 𑀅𑀓𑀺𑀭𑀺𑀬𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀲𑀳 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀘𑀦𑁆𑀤𑀯𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀲𑀜𑁆𑀜𑀻 𑀅𑀳𑁄𑀲𑀺 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀭𑀢𑁆𑀢𑁄 𑀯𑀺𑀲𑀜𑁆𑀜𑀺𑀪𑀽𑀢𑁄 𑀆𑀓𑀼𑀮𑀸𑀓𑀼𑀮𑁄 𑀢𑀼𑀭𑀺𑀢𑀢𑀼𑀭𑀺𑀢𑁄, 𑀢𑁂𑀦 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀪𑀦𑁆𑀢𑀮𑀼𑀍𑀇𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀧𑀲𑀼𑀖𑀸𑀢𑀕𑀮𑀭𑀼𑀳𑀺𑀭𑀲𑀜𑁆𑀘𑀬𑀁 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑁇

982\. 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁂𑀧𑀺 𑀦 𑀫𑀳𑀸𑀲𑀸𑀯𑀚𑁆𑀚𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁂 𑀯𑀺𑀧𑀸𑀓𑁂𑀦𑀧𑀺 𑀦𑁄 𑀢𑀣𑀸𑁇 𑀇𑀥, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑁄 𑀯𑀚𑁆𑀛𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀢𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀓𑀺𑀁 𑀤𑀡𑁆𑀟𑀁 𑀥𑀸𑀭𑁂𑀣𑀸”𑀢𑀺? “𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀤𑀡𑁆𑀟𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀁 𑀫𑀬𑀁 𑀧𑁄𑀣𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀻𑀳𑀭𑀸𑀧𑁂𑀫, 𑀏𑀲𑁄𑀯 𑀢𑀲𑁆𑀲 𑀤𑀡𑁆𑀟𑁄”𑀢𑀺𑁇 “𑀇𑀢𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀅𑀧𑀭𑀸𑀥𑁂 𑀤𑀡𑁆𑀟𑁄𑀧𑀺 𑀦 𑀪𑀯𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀓𑀢𑁂𑀧𑀺 𑀦 𑀤𑁄𑀲𑁄 𑀪𑀯𑀢𑀺 𑀲𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀫𑀲𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀲𑀺 𑀲𑀳 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀘𑀦𑁆𑀤𑀯𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀲𑀜𑁆𑀜𑀻 𑀅𑀳𑁄𑀲𑀺 𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀭𑀢𑁆𑀢𑁄 𑀯𑀺𑀲𑀜𑁆𑀜𑀺𑀪𑀽𑀢𑁄 𑀯𑀺𑀲𑀝𑀧𑀬𑀸𑀢𑁄 𑀆𑀓𑀼𑀮𑀸𑀓𑀼𑀮𑁄 𑀢𑀼𑀭𑀺𑀢𑀢𑀼𑀭𑀺𑀢𑁄, 𑀢𑁂𑀦 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀪𑀦𑁆𑀢𑀮𑀼𑀍𑀇𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀧𑀲𑀼𑀖𑀸𑀢𑀕𑀮𑀭𑀼𑀳𑀺𑀭𑀲𑀜𑁆𑀘𑀬𑀁 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑁇 𑀬𑀤𑀸 𑀘 𑀧𑀦 𑀧𑀓𑀢𑀺𑀘𑀺𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀲𑁆𑀲𑀢𑀺, 𑀢𑀤𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀘𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀬𑁄 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀽𑀧𑀕𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
