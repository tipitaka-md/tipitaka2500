# 𑁫𑁇𑁫𑁇𑁮 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀭𑀸𑀚𑀯𑀸𑀤𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀭𑀸𑀚𑀯𑀸𑀤𑀧𑀜𑁆𑀳

(167.)

996\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀭𑀸𑀚𑀸𑀳𑀫𑀲𑁆𑀫𑀺 𑀲𑁂𑀮𑀸’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀭𑀸𑀚𑀸𑀳𑀫𑀲𑁆𑀫𑀺 𑀲𑁂𑀮𑀸’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀭𑀸𑀚𑀸𑀳𑀫𑀲𑁆𑀫𑀺 𑀲𑁂𑀮𑀸’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀔𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀸 𑀳𑀺 𑀪𑀯𑁂𑀬𑁆𑀬 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀏𑀓𑀸𑀬 𑀚𑀸𑀢𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀯𑀡𑁆𑀡𑀸 𑀦𑀸𑀫, 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

997\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀅𑀳𑀫𑀲𑁆𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄’𑀢𑀺, 𑀧𑀼𑀦 𑀘 𑀪𑀡𑀺𑀢𑀁—  ‘𑀭𑀸𑀚𑀸𑀳𑀫𑀲𑁆𑀫𑀺 𑀲𑁂𑀮𑀸’𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀺, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀘 𑀭𑀸𑀚𑀸 𑀘 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

998\. “𑀓𑀺𑀁 𑀧𑀦 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀘 𑀭𑀸𑀚𑀸 𑀘 𑀳𑁄𑀢𑀺”? “𑀲𑀩𑁆𑀩𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑀸𑀳𑀺𑀢𑀸 𑀧𑀳𑀻𑀦𑀸 𑀅𑀧𑀕𑀢𑀸 𑀩𑁆𑀬𑀧𑀕𑀢𑀸 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀔𑀻𑀡𑀸 𑀔𑀬𑀁 𑀧𑀢𑁆𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀉𑀧𑀲𑀦𑁆𑀢𑀸, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

999\. 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀸𑀫 𑀲𑀁𑀲𑀬𑀫𑀦𑁂𑀓𑀁𑀲𑀁 𑀯𑀺𑀫𑀢𑀺𑀧𑀣𑀁 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁄, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀁𑀲𑀬𑀫𑀦𑁂𑀓𑀁𑀲𑀁 𑀯𑀺𑀫𑀢𑀺𑀧𑀣𑀁 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁄, 𑀢𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1000\. 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀸𑀫 𑀲𑀩𑁆𑀩𑀪𑀯𑀕𑀢𑀺𑀬𑁄𑀦𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁄 𑀫𑀮𑀭𑀚𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀲𑀳𑀸𑀬𑁄, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑀪𑀯𑀕𑀢𑀺𑀬𑁄𑀦𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁄 𑀫𑀮𑀭𑀚𑀕𑀢𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀅𑀲𑀳𑀸𑀬𑁄, 𑀢𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1001\. 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀸𑀫 𑀅𑀕𑁆𑀕𑀲𑁂𑀝𑁆𑀞𑀯𑀭𑀧𑀯𑀭𑀤𑀺𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀩𑀳𑀼𑀮𑁄, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑁆𑀕𑀲𑁂𑀝𑁆𑀞𑀯𑀭𑀧𑀯𑀭𑀤𑀺𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀩𑀳𑀼𑀮𑁄, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1002\. 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀸𑀫 𑀅𑀚𑁆𑀛𑀬𑀦𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑀤𑀸𑀦𑀧𑁆𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀤𑀫𑀲𑀁𑀬𑀫𑀦𑀺𑀬𑀫𑀧𑀼𑀩𑁆𑀩𑀫𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺𑀧𑀯𑁂𑀡𑀺𑀯𑀁𑀲𑀥𑀭𑀡𑁄, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑁆𑀛𑀬𑀦𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑀤𑀸𑀦𑀧𑁆𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑀤𑀫𑀲𑀁𑀬𑀫𑀦𑀺𑀬𑀫𑀧𑀼𑀩𑁆𑀩𑀚𑀺𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺𑀧𑀯𑁂𑀡𑀺𑀯𑀁𑀲𑀥𑀭𑀡𑁄, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1003\. 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀳𑀸𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀚𑁆𑀛𑀸𑀦𑀛𑀸𑀬𑀻; 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁆𑀭𑀳𑀸𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀚𑁆𑀛𑀸𑀦𑀛𑀸𑀬𑀻, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1004\. 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀦𑀸𑀫 𑀲𑀩𑁆𑀩𑀪𑀯𑀸𑀪𑀯𑀕𑀢𑀻𑀲𑀼 𑀅𑀪𑀺𑀚𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀫𑀦𑀼𑀘𑀭𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑀪𑀯𑀸𑀪𑀯𑀕𑀢𑀻𑀲𑀼 𑀅𑀪𑀺𑀚𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑀫𑀦𑀼𑀘𑀭𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇

1005\. 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀦𑁂𑀢𑀁 𑀦𑀸𑀫𑀁 𑀫𑀸𑀢𑀭𑀸 𑀓𑀢𑀁, 𑀦 𑀧𑀺𑀢𑀭𑀸 𑀓𑀢𑀁, 𑀦 𑀪𑀸𑀢𑀭𑀸 𑀓𑀢𑀁, 𑀦 𑀪𑀕𑀺𑀦𑀺𑀬𑀸 𑀓𑀢𑀁, 𑀦 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑁂𑀳𑀺 𑀓𑀢𑀁, 𑀦 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑁂𑀳𑀺 𑀓𑀢𑀁, 𑀦 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀳𑀺 𑀓𑀢𑀁, 𑀦 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀓𑀢𑀁, 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀓𑀫𑁂𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀦𑀸𑀫𑀁 𑀩𑁄𑀥𑀺𑀬𑀸𑀬𑁂𑀯 𑀫𑀽𑀮𑁂 𑀫𑀸𑀭𑀲𑁂𑀦𑀁 𑀯𑀺𑀥𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀧𑀸𑀧𑀓𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂 𑀩𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀳 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀲𑁆𑀲 𑀧𑀝𑀺𑀮𑀸𑀪𑀸 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀫𑀢𑁆𑀢𑁂 𑀲𑀘𑁆𑀙𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀬𑀤𑀺𑀤𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀢𑀺, 𑀢𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’”𑀢𑀺𑁇

1006\. “𑀓𑁂𑀦 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀭𑀸𑀚𑀸’”𑀢𑀺? “𑀭𑀸𑀚𑀸 𑀦𑀸𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀢𑀺 𑀮𑁄𑀓𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀺, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀲𑀤𑁂𑀯𑀓𑀁 𑀮𑁄𑀓𑀁 𑀲𑀫𑀸𑀭𑀓𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀁 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀧𑀚𑀁 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀭𑀸𑀚𑀸’𑀢𑀺𑁇

1007\. 𑀭𑀸𑀚𑀸 𑀦𑀸𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑀚𑀦𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀅𑀪𑀺𑀪𑀯𑀺𑀢𑁆𑀯𑀸 𑀦𑀦𑁆𑀤𑀬𑀦𑁆𑀢𑁄 𑀜𑀸𑀢𑀺𑀲𑀗𑁆𑀖𑀁, 𑀲𑁄𑀘𑀬𑀦𑁆𑀢𑁄 𑀅𑀫𑀺𑀢𑁆𑀢𑀲𑀁𑀖𑀁, 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀬𑀲𑀲𑀺𑀭𑀺𑀳𑀭𑀁 𑀣𑀺𑀭𑀲𑀸𑀭𑀤𑀡𑁆𑀟𑀁 𑀅𑀦𑀽𑀦𑀲𑀢𑀲𑀮𑀸𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀁 𑀉𑀲𑁆𑀲𑀸𑀧𑁂𑀢𑀺 𑀧𑀡𑁆𑀟𑀭𑀯𑀺𑀫𑀮𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁄𑀘𑀬𑀦𑁆𑀢𑁄 𑀫𑀸𑀭𑀲𑁂𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁, 𑀦𑀦𑁆𑀤𑀬𑀦𑁆𑀢𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂 𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀺𑀬𑀸 𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀬𑀲𑀲𑀺𑀭𑀺𑀳𑀭𑀁 𑀔𑀦𑁆𑀢𑀺𑀣𑀺𑀭𑀲𑀸𑀭𑀤𑀡𑁆𑀟𑀁 𑀜𑀸𑀡𑀯𑀭𑀲𑀢𑀲𑀮𑀸𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀁 𑀉𑀲𑁆𑀲𑀸𑀧𑁂𑀢𑀺 𑀅𑀕𑁆𑀕𑀯𑀭𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀡𑁆𑀟𑀭𑀯𑀺𑀫𑀮𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀭𑀸𑀚𑀸’𑀢𑀺𑁇

1008\. 𑀭𑀸𑀚𑀸 𑀦𑀸𑀫 𑀉𑀧𑀕𑀢𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀚𑀦𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀫𑀪𑀺𑀯𑀦𑁆𑀤𑀦𑀻𑀬𑁄 𑀪𑀯𑀢𑀺, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀧𑀕𑀢𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀫𑀪𑀺𑀯𑀦𑁆𑀤𑀦𑀻𑀬𑁄, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀭𑀸𑀚𑀸’𑀢𑀺𑁇

1009\. 𑀭𑀸𑀚𑀸 𑀦𑀸𑀫 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀆𑀭𑀸𑀥𑀓𑀲𑁆𑀲 𑀧𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀯𑀭𑀺𑀢𑀁 𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂𑀦 𑀢𑀧𑁆𑀧𑀬𑀢𑀺, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀫𑀦𑀲𑀸 𑀆𑀭𑀸𑀥𑀓𑀲𑁆𑀲 𑀧𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀯𑀭𑀺𑀢𑀁 𑀯𑀭𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀧𑀭𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀢𑁆𑀯𑀸 𑀅𑀲𑁂𑀲𑀓𑀸𑀫𑀯𑀭𑁂𑀦 𑀘 𑀢𑀧𑁆𑀧𑀬𑀢𑀺, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀭𑀸𑀚𑀸’𑀢𑀺𑁇

1010\. 𑀭𑀸𑀚𑀸 𑀦𑀸𑀫 𑀆𑀡𑀁 𑀯𑀻𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀁 𑀯𑀺𑀕𑀭𑀳𑀢𑀺 𑀛𑀸𑀧𑁂𑀢𑀺 𑀥𑀁𑀲𑁂𑀢𑀺, 𑀪𑀕𑀯𑀢𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀸𑀲𑀦𑀯𑀭𑁂 𑀆𑀡𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁄 𑀅𑀮𑀚𑁆𑀚𑀻 𑀫𑀗𑁆𑀓𑀼𑀪𑀸𑀯𑁂𑀦 𑀑𑀜𑁆𑀜𑀸𑀢𑁄 𑀳𑀻𑀍𑀇𑀢𑁄 𑀕𑀭𑀳𑀺𑀢𑁄 𑀪𑀯𑀺𑀢𑁆𑀯𑀸 𑀯𑀚𑁆𑀚𑀢𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀯𑀭𑀫𑁆𑀳𑀸, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀭𑀸𑀚𑀸’𑀢𑀺𑁇

1011\. 𑀭𑀸𑀚𑀸 𑀦𑀸𑀫 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀭𑀸𑀚𑀽𑀦𑀁 𑀧𑀯𑁂𑀡𑀺𑀫𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑀥𑀫𑁆𑀫𑀫𑀦𑀼𑀤𑀻𑀧𑀬𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑀬𑀫𑀸𑀦𑁄 𑀧𑀺𑀳𑀬𑀺𑀢𑁄 𑀧𑀺𑀬𑁄 𑀧𑀢𑁆𑀣𑀺𑀢𑁄 𑀪𑀯𑀢𑀺 𑀚𑀦𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁, 𑀘𑀺𑀭𑀁 𑀭𑀸𑀚𑀓𑀼𑀮𑀯𑀁𑀲𑀁 𑀞𑀧𑀬𑀢𑀺 𑀥𑀫𑁆𑀫𑀕𑀼𑀡𑀩𑀮𑁂𑀦, 𑀪𑀕𑀯𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀲𑀬𑀫𑁆𑀪𑀽𑀦𑀁 𑀧𑀯𑁂𑀡𑀺𑀫𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑀥𑀫𑁆𑀫𑀫𑀦𑀼𑀤𑀻𑀧𑀬𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀮𑁄𑀓𑀫𑀦𑀼𑀲𑀸𑀲𑀫𑀸𑀦𑁄 𑀧𑀺𑀳𑀬𑀺𑀢𑁄 𑀧𑀺𑀬𑁄 𑀧𑀢𑁆𑀣𑀺𑀢𑁄 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀘𑀺𑀭𑀁 𑀲𑀸𑀲𑀦𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺 𑀥𑀫𑁆𑀫𑀕𑀼𑀡𑀩𑀮𑁂𑀦, 𑀢𑁂𑀦𑀸𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀭𑀸𑀚𑀸’𑀢𑀺𑁇 𑀏𑀯𑀫𑀦𑁂𑀓𑀯𑀺𑀥𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑀧𑀺 𑀪𑀯𑁂𑀬𑁆𑀬 𑀭𑀸𑀚𑀸𑀧𑀺 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀲𑀼𑀦𑀺𑀧𑀼𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀫𑁆𑀧𑀺 𑀦𑁄 𑀦𑀁 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬, 𑀓𑀺𑀁 𑀅𑀢𑀺𑀩𑀳𑀼𑀁 𑀪𑀡𑀺𑀢𑁂𑀦, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
