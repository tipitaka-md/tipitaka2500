# 𑁫𑁇𑁫𑁇𑁧 𑀲𑀦𑁆𑀣𑀯𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀲𑀦𑁆𑀣𑀯𑀧𑀜𑁆𑀳

(160.)

942\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—

943\. _‘𑀲𑀦𑁆𑀣𑀯𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁,_  
_𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄;_  
_𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁,_  
_𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀦𑁆’𑀢𑀺𑁇_  


944\. 𑀧𑀼𑀦 𑀘 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀯𑀺𑀳𑀸𑀭𑁂 𑀓𑀸𑀭𑀬𑁂 𑀭𑀫𑁆𑀫𑁂, 𑀯𑀸𑀲𑀬𑁂𑀢𑁆𑀣 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀲𑀦𑁆𑀣𑀯𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁, 𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄𑁇 𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁, 𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀦𑁆’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀯𑀺𑀳𑀸𑀭𑁂 𑀓𑀸𑀭𑀬𑁂 𑀭𑀫𑁆𑀫𑁂, 𑀯𑀸𑀲𑀬𑁂𑀢𑁆𑀣 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀯𑀺𑀳𑀸𑀭𑁂 𑀓𑀸𑀭𑀬𑁂 𑀭𑀫𑁆𑀫𑁂, 𑀯𑀸𑀲𑀬𑁂𑀢𑁆𑀣 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀲𑀦𑁆𑀣𑀯𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁, 𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄𑁇 𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁, 𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀦𑁆’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

945\. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀲𑀦𑁆𑀣𑀯𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁, 𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄𑁇 𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁, 𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀦𑁆’𑀢𑀺𑁇 𑀪𑀡𑀺𑀢𑀜𑁆𑀘—  ‘𑀯𑀺𑀳𑀸𑀭𑁂 𑀓𑀸𑀭𑀬𑁂 𑀭𑀫𑁆𑀫𑁂, 𑀯𑀸𑀲𑀬𑁂𑀢𑁆𑀣 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂’𑀢𑀺𑁇 𑀬𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀲𑀦𑁆𑀣𑀯𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁, 𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄𑁇 𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁, 𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀦𑁆’𑀢𑀺, 𑀢𑀁 𑀲𑀪𑀸𑀯𑀯𑀘𑀦𑀁 𑀅𑀲𑁂𑀲𑀯𑀘𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑁂𑀲𑀯𑀘𑀦𑀁 𑀦𑀺𑀧𑁆𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀘𑀦𑀁 𑀲𑀫𑀡𑀸𑀦𑀼𑀘𑁆𑀙𑀯𑀁 𑀲𑀫𑀡𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀁 𑀲𑀫𑀡𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀲𑀫𑀡𑀸𑀭𑀳𑀁 𑀲𑀫𑀡𑀕𑁄𑀘𑀭𑀁 𑀲𑀫𑀡𑀧𑁆𑀧𑀝𑀺𑀧𑀤𑀸 𑀲𑀫𑀡𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀭𑀜𑁆𑀜𑀓𑁄 𑀫𑀺𑀕𑁄 𑀅𑀭𑀜𑁆𑀜𑁂 𑀧𑀯𑀦𑁂 𑀘𑀭𑀫𑀸𑀦𑁄 𑀦𑀺𑀭𑀸𑀮𑀬𑁄 𑀅𑀦𑀺𑀓𑁂𑀢𑁄 𑀬𑀣𑀺𑀘𑁆𑀙𑀓𑀁 𑀲𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 ‘𑀲𑀦𑁆𑀣𑀯𑀢𑁄 𑀪𑀬𑀁 𑀚𑀸𑀢𑀁, 𑀦𑀺𑀓𑁂𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀭𑀚𑁄𑁇 𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁, 𑀏𑀢𑀁 𑀯𑁂 𑀫𑀼𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀦𑁆’𑀢𑀺 𑀘𑀺𑀦𑁆𑀢𑁂𑀢𑀩𑁆𑀩𑀁𑁇

946\. 𑀬𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀯𑀺𑀳𑀸𑀭𑁂 𑀓𑀸𑀭𑀬𑁂 𑀭𑀫𑁆𑀫𑁂, 𑀯𑀸𑀲𑀬𑁂𑀢𑁆𑀣 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂’𑀢𑀺, 𑀢𑀁 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀯𑀺𑀳𑀸𑀭𑀤𑀸𑀦𑀁 𑀦𑀸𑀫 𑀲𑀩𑁆𑀩𑀩𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀯𑀡𑁆𑀡𑀺𑀢𑀁 𑀅𑀦𑀼𑀫𑀢𑀁 𑀣𑁄𑀫𑀺𑀢𑀁 𑀧𑀲𑀢𑁆𑀣𑀁, 𑀢𑀁 𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀸 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀸𑀯 𑀧𑀞𑀫𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀯𑀺𑀳𑀸𑀭𑀤𑀸𑀦𑁂𑁇

947\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑀲𑀗𑁆𑀓𑁂𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀼𑀮𑀪𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑀸𑀦𑀁, 𑀅𑀦𑀺𑀓𑁂𑀢𑁂 𑀤𑀼𑀤𑁆𑀤𑀲𑁆𑀲𑀦𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀯𑀺𑀳𑀸𑀭𑀤𑀸𑀦𑁂𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀯𑀺𑀳𑀸𑀭𑁂 𑀓𑀸𑀭𑀬𑁂 𑀭𑀫𑁆𑀫𑁂, 𑀯𑀸𑀲𑀬𑁂𑀢𑁆𑀣 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂’𑀢𑀺, 𑀦 𑀢𑀢𑁆𑀣 𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀆𑀮𑀬𑁄 𑀓𑀭𑀡𑀻𑀬𑁄 𑀦𑀺𑀓𑁂𑀢𑁂”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
