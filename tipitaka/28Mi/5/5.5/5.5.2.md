# 𑁫𑁇𑁫𑁇𑁨 𑀉𑀤𑀭𑀲𑀁𑀬𑀢𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀉𑀤𑀭𑀲𑀁𑀬𑀢𑀧𑀜𑁆𑀳

(161.)

949. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—

950. _‘𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀲𑀺𑀬𑀸’𑀢𑀺𑁇_  


951. 𑀧𑀼𑀦 𑀘 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦𑀼𑀤𑀸𑀬𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀤𑀸 𑀇𑀫𑀺𑀦𑀸 𑀧𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑀢𑀺𑀢𑁆𑀣𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺, 𑀪𑀺𑀬𑁆𑀬𑁄𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀲𑀺𑀬𑀸’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦𑀼𑀤𑀸𑀬𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀤𑀸 𑀇𑀫𑀺𑀦𑀸 𑀧𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑀢𑀺𑀢𑁆𑀣𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺, 𑀪𑀺𑀬𑁆𑀬𑁄𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀻’𑀢𑀺 𑀬𑀁 𑀯𑀘𑀦𑀁, 𑀢𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀬𑀤𑀺 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀡𑀺𑀢𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦𑀼𑀤𑀸𑀬𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀤𑀸 𑀇𑀫𑀺𑀦𑀸 𑀧𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑀢𑀺𑀢𑁆𑀣𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺, 𑀪𑀺𑀬𑁆𑀬𑁄𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀻’𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺 ‘𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀲𑀺𑀬𑀸’𑀢𑀺 𑀢𑀫𑁆𑀧𑀺 𑀯𑀘𑀦𑀁 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀉𑀪𑀢𑁄 𑀓𑁄𑀝𑀺𑀓𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀯𑀸𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑁄 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

952. “𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀲𑀺𑀬𑀸’𑀢𑀺, 𑀪𑀡𑀺𑀢𑀜𑁆𑀘—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦𑀼𑀤𑀸𑀬𑀺, 𑀅𑀧𑁆𑀧𑁂𑀓𑀤𑀸 𑀇𑀫𑀺𑀦𑀸 𑀧𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑀢𑀺𑀢𑁆𑀣𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺, 𑀪𑀺𑀬𑁆𑀬𑁄𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀲𑀺𑀬𑀸’𑀢𑀺, 𑀢𑀁 𑀲𑀪𑀸𑀯𑀯𑀘𑀦𑀁 𑀅𑀲𑁂𑀲𑀯𑀘𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑁂𑀲𑀯𑀘𑀦𑀁 𑀦𑀺𑀧𑁆𑀧𑀭𑀺𑀬𑀸𑀬𑀯𑀘𑀦𑀁 𑀪𑀽𑀢𑀯𑀘𑀦𑀁 𑀢𑀘𑁆𑀙𑀯𑀘𑀦𑀁 𑀬𑀸𑀣𑀸𑀯𑀯𑀘𑀦𑀁 𑀅𑀯𑀺𑀧𑀭𑀻𑀢𑀯𑀘𑀦𑀁 𑀇𑀲𑀺𑀯𑀘𑀦𑀁 𑀫𑀼𑀦𑀺𑀯𑀘𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑀘𑀦𑀁 𑀧𑀘𑁆𑀘𑁂𑀓𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑀁 𑀚𑀺𑀦𑀯𑀘𑀦𑀁 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀯𑀘𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀘𑀦𑀁𑁇

953. 𑀉𑀤𑀭𑁂 𑀅𑀲𑀁𑀬𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀸𑀡𑀫𑁆𑀧𑀺 𑀳𑀦𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀧𑀭𑀤𑀸𑀭𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑀼𑀲𑀸𑀧𑀺 𑀪𑀡𑀢𑀺, 𑀫𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀧𑀺𑀯𑀢𑀺, 𑀫𑀸𑀢𑀭𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀧𑀺𑀢𑀭𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀅𑀭𑀳𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀲𑀁𑀖𑀫𑁆𑀧𑀺 𑀪𑀺𑀦𑁆𑀤𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀮𑁄𑀳𑀺𑀢𑀫𑁆𑀧𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀦𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀉𑀤𑀭𑁂 𑀅𑀲𑀁𑀬𑀢𑁄 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑀺𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀆𑀬𑀽𑀳𑀺𑁇 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀜𑁆𑀜𑀸𑀦𑀺𑀧𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀸𑀦𑀺 𑀓𑀸𑀭𑀡𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀲𑀺𑀬𑀸’𑀢𑀺𑁇

954. 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀸𑀪𑀺𑀲𑀫𑀬𑀁 𑀅𑀪𑀺𑀲𑀫𑁂𑀢𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀨𑀮𑀸𑀦𑀺 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺, 𑀘𑀢𑀽𑀲𑀼 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀲𑀼 𑀅𑀝𑁆𑀞𑀲𑀼 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀲𑀼 𑀙𑀲𑀼 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀯𑀲𑀻𑀪𑀸𑀯𑀁 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀺, 𑀓𑁂𑀯𑀮𑀜𑁆𑀘 𑀲𑀫𑀡𑀥𑀫𑁆𑀫𑀁 𑀧𑀽𑀭𑁂𑀢𑀺𑁇 𑀦𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀓𑀧𑁄𑀢𑀓𑁄 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸 𑀬𑀸𑀯 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀪𑀯𑀦𑀁 𑀓𑀫𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀫𑀼𑀧𑀦𑁂𑀲𑀺, 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀜𑁆𑀜𑀸𑀦𑀺𑀧𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀸𑀦𑀺 𑀓𑀸𑀭𑀡𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀉𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀢𑁄 𑀲𑀺𑀬𑀸’𑀢𑀺𑁇

955. 𑀬𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀪𑀡𑀺𑀢𑀁—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀧𑀦𑀼𑀤𑀸𑀬𑀺 𑀅𑀧𑁆𑀧𑁂𑀓𑀤𑀸 𑀇𑀫𑀺𑀦𑀸 𑀧𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑀢𑀺𑀢𑁆𑀣𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺, 𑀪𑀺𑀬𑁆𑀬𑁄𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀻’𑀢𑀺, 𑀢𑀁 𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑁂𑀦 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀓𑀺𑀭𑀺𑀬𑁂𑀦 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑁂𑀦 𑀯𑀼𑀲𑀺𑀢𑀯𑁄𑀲𑀸𑀦𑁂𑀦 𑀦𑀺𑀭𑀸𑀯𑀭𑀡𑁂𑀦 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀦𑀸 𑀲𑀬𑀫𑁆𑀪𑀼𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀪𑀡𑀺𑀢𑀁𑁇

956. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀭𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀯𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀆𑀢𑀼𑀭𑀲𑁆𑀲 𑀲𑀧𑁆𑀧𑀸𑀬𑀓𑀺𑀭𑀺𑀬𑀸 𑀇𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑀺𑀮𑁂𑀲𑀲𑁆𑀲 𑀅𑀤𑀺𑀝𑁆𑀞𑀲𑀘𑁆𑀘𑀲𑁆𑀲 𑀉𑀤𑀭𑁂 𑀲𑀁𑀬𑀫𑁄 𑀓𑀭𑀡𑀻𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀡𑀺𑀭𑀢𑀦𑀲𑁆𑀲 𑀲𑀧𑁆𑀧𑀪𑀸𑀲𑀲𑁆𑀲 𑀚𑀸𑀢𑀺𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀪𑀺𑀚𑀸𑀢𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀫𑀚𑁆𑀚𑀦𑀦𑀺𑀖𑀁𑀲𑀦𑀧𑀭𑀺𑀲𑁄𑀥𑀦𑁂𑀦 𑀓𑀭𑀡𑀻𑀬𑀁 𑀦 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑀯𑀺𑀲𑀬𑁂 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀲𑁆𑀲 𑀓𑀺𑀭𑀺𑀬𑀸𑀓𑀭𑀡𑁂𑀲𑀼 𑀆𑀯𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
