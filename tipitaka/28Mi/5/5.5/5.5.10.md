# 𑁫𑁇𑁫𑁇𑁧𑁦 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀧𑀜𑁆𑀳

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀳

### 𑀲𑀦𑁆𑀣𑀯𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀧𑀜𑁆𑀳

(169.)

1033. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀘𑀢𑀽𑀳𑀺 𑀘 𑀅𑀲𑀗𑁆𑀔𑁆𑀬𑁂𑀬𑁆𑀬𑁂𑀳𑀺 𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀘 𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑀺𑀢𑀁 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑁆𑀥𑀭𑀡𑀸𑀬𑀸’𑀢𑀺𑁇 𑀧𑀼𑀦 𑀘 ‘𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀢𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀫𑀺, 𑀦𑁄 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬𑀸’𑀢𑀺𑁇

1034. 𑀬𑀣𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀲𑁆𑀲𑀸𑀲𑁄 𑀯𑀸 𑀇𑀲𑁆𑀲𑀸𑀲𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀻 𑀯𑀸 𑀩𑀳𑀼𑀓𑁂 𑀤𑀺𑀯𑀲𑁂 𑀲𑀗𑁆𑀕𑀸𑀫𑀢𑁆𑀣𑀸𑀬 𑀉𑀧𑀸𑀲𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂 𑀫𑀳𑀸𑀬𑀼𑀤𑁆𑀥𑁂 𑀑𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀘𑀢𑀽𑀳𑀺 𑀘 𑀅𑀲𑀗𑁆𑀔𑁆𑀬𑁂𑀬𑁆𑀬𑁂𑀳𑀺 𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀘 𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑁆𑀥𑀭𑀡𑀸𑀬 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬 𑀑𑀲𑀓𑁆𑀓𑀺𑀢𑀁𑁇

1035. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀫𑀮𑁆𑀮𑁄 𑀯𑀸 𑀫𑀮𑁆𑀮𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀻 𑀯𑀸 𑀩𑀳𑀼𑀓𑁂 𑀤𑀺𑀯𑀲𑁂 𑀦𑀺𑀩𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂 𑀫𑀮𑁆𑀮𑀬𑀼𑀤𑁆𑀥𑁂 𑀑𑀲𑀓𑁆𑀓𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀘𑀢𑀽𑀳𑀺 𑀘 𑀅𑀲𑀗𑁆𑀔𑁆𑀬𑁂𑀬𑁆𑀬𑁂𑀳𑀺 𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀘 𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀢𑁄 𑀚𑀦𑀓𑀸𑀬𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑁆𑀥𑀭𑀡𑀸𑀬 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬 𑀑𑀲𑀓𑁆𑀓𑀺𑀢𑀁𑁇

1041. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺 𑀭𑀸𑀚𑀸 𑀯𑀸 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀯𑀸 𑀬𑀲𑁆𑀲 𑀑𑀦𑀫𑀢𑀺 𑀅𑀧𑀘𑀺𑀢𑀺𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀩𑀮𑀯𑀢𑀭𑀲𑁆𑀲 𑀢𑀲𑁆𑀲 𑀑𑀦𑀫𑀦𑁂𑀦 𑀅𑀯𑀲𑁂𑀲𑀸 𑀚𑀦𑀢𑀸 𑀑𑀦𑀫𑀢𑀺 𑀅𑀧𑀘𑀺𑀢𑀺𑀁 𑀓𑀭𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀑𑀦𑀫𑀺𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀲𑀤𑁂𑀯𑀓𑁄 𑀮𑁄𑀓𑁄 𑀑𑀦𑀫𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀽𑀚𑀺𑀢𑀧𑀽𑀚𑀓𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀓𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀲𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀆𑀬𑀸𑀘𑀢𑀺 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬, 𑀢𑁂𑀦 𑀘 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸 𑀆𑀬𑀸𑀘𑀺𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀼𑀦𑀺𑀩𑁆𑀩𑁂𑀞𑀺𑀢𑁄 𑀧𑀜𑁆𑀳𑁄, 𑀅𑀢𑀺𑀪𑀤𑁆𑀭𑀓𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
