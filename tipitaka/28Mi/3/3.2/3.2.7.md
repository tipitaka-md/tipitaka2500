# 𑁩𑁇𑁨𑁇𑁭 𑀣𑁂𑀭𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀦𑀸𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀦𑀧𑀜𑁆𑀳

### 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

### 𑀅𑀤𑁆𑀥𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀣𑁂𑀭𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀦𑀸𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀦𑀧𑀜𑁆𑀳

(46.)

215\. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑁆𑀯𑀁 𑀧𑀦 𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? “𑀅𑀮𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀺𑀁 𑀢𑁂 𑀢𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂𑀦, 𑀦𑀦𑀼 𑀫𑀬𑀸 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀅𑀓𑁆𑀔𑀸𑀢𑀁 ‘𑀲𑀘𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀘𑁂 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇

216\. “𑀑𑀧𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀜𑁆𑀜𑁄 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀭𑀸𑀚𑀸 𑀢𑀼𑀝𑁆𑀞𑁄 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀥𑀺𑀓𑀸𑀭𑁂𑀦 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄 𑀲𑀫𑀗𑁆𑀕𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀘𑀭𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀘𑁂 𑀚𑀦𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬 ‘𑀦 𑀫𑁂 𑀭𑀸𑀚𑀸 𑀓𑀺𑀜𑁆𑀘𑀺 𑀧𑀝𑀺𑀓𑀭𑁄𑀢𑀻’𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀭𑀻 𑀪𑀯𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀺𑀁 𑀢𑁂 𑀢𑁂𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂𑀦, 𑀦𑀦𑀼 𑀫𑀬𑀸 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀅𑀓𑁆𑀔𑀸𑀢𑀁 ‘𑀲𑀘𑁂 𑀲𑁅𑀧𑀸𑀤𑀸𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀘𑁂 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦 𑀧𑀝𑀺𑀲𑀦𑁆𑀤𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇

217\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇
