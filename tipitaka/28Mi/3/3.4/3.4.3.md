# 𑁩𑁇𑁪𑁇𑁩 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀓𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀧𑀜𑁆𑀳

### 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

### 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀓𑀫𑁆𑀫𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀧𑀜𑁆𑀳

(65.)

323. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀸𑀦𑀺, 𑀓𑀺𑀁 𑀦𑀼 𑀢𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸𑀦𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦𑀸”𑀢𑀺? “𑀦𑀸𑀦𑀸𑀓𑀫𑁆𑀫𑁂𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸𑀦𑀺, 𑀦 𑀏𑀓𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦𑀸”𑀢𑀺𑁇

324. “𑀑𑀧𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀲𑁆𑀫𑀺𑀁 𑀔𑁂𑀢𑁆𑀢𑁂 𑀦𑀸𑀦𑀸𑀩𑀻𑀚𑀸𑀦𑀺 𑀯𑀧𑁆𑀧𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑁂𑀲𑀁 𑀦𑀸𑀦𑀸𑀩𑀻𑀚𑀸𑀦𑀁 𑀦𑀸𑀦𑀸𑀨𑀮𑀸𑀦𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀸𑀦𑀺 𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸𑀦𑀺, 𑀦 𑀏𑀓𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦𑀸”𑀢𑀺𑁇

325. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇

---
