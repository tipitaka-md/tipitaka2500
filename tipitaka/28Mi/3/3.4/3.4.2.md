# 𑁩𑁇𑁪𑁇𑁨 𑀦𑀸𑀕𑀲𑁂𑀦𑀧𑀜𑁆𑀳

* 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

* 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀯𑀕𑁆𑀕

* 𑀦𑀸𑀕𑀲𑁂𑀦𑀧𑀜𑁆𑀳

(64.)

319\. 𑀣𑁂𑀭𑁄 𑀆𑀳—  “𑀮𑁄𑀡𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀩𑁆𑀩𑀁 𑀮𑁄𑀡𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑀁 𑀮𑁄𑀡𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺”𑁇

320\. “𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀩𑁆𑀩𑀁 𑀮𑁄𑀡𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀯𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑀁 𑀲𑀓𑀝𑁂𑀳𑀺 𑀩𑀮𑀻𑀩𑀤𑁆𑀤𑀸 𑀆𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦𑀦𑀼 𑀮𑁄𑀡𑀫𑁂𑀯 𑀆𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀦 𑀲𑀓𑁆𑀓𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀡𑀫𑁂𑀯 𑀆𑀳𑀭𑀺𑀢𑀼𑀁 𑀏𑀓𑀢𑁄𑀪𑀸𑀯𑀕𑀢𑀸 𑀏𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀕𑁄𑀘𑀭𑀦𑀸𑀦𑀢𑁆𑀢𑀕𑀢𑀸 𑀮𑁄𑀡𑀁 𑀕𑀭𑀼𑀪𑀸𑀯𑁄 𑀘𑀸𑀢𑀺𑁇 𑀲𑀓𑁆𑀓𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀡𑀁 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑀬𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑀸”𑀢𑀺𑁇 “𑀦 𑀲𑀓𑁆𑀓𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀡𑀁 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑀬𑀺𑀢𑀼𑀁, 𑀕𑀭𑀼𑀪𑀸𑀯𑁄 𑀢𑀼𑀮𑀸𑀬 𑀢𑀼𑀮𑀺𑀬𑀢𑀻”𑀢𑀺𑁇

321\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇
