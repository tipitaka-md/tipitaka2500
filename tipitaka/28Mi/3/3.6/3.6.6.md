# 𑁩𑁇𑁬𑁇𑁬 𑀅𑀲𑁆𑀲𑀼𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀜𑁆𑀳

### 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

### 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀲𑁆𑀲𑀼𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀜𑁆𑀳

(88.)

415\. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁄 𑀘 𑀫𑀸𑀢𑀭𑀺 𑀫𑀢𑀸𑀬 𑀭𑁄𑀤𑀢𑀺, 𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀧𑁂𑀫𑁂𑀦 𑀭𑁄𑀤𑀢𑀺, 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀢𑁂𑀲𑀁 𑀭𑁄𑀤𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼 𑀪𑁂𑀲𑀚𑁆𑀚𑀁, 𑀓𑀲𑁆𑀲 𑀦 𑀪𑁂𑀲𑀚𑁆𑀚𑀦𑁆”𑀢𑀺? “𑀏𑀓𑀲𑁆𑀲 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀲𑁆𑀲𑀼 𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑁂𑀳𑀺 𑀲𑀫𑀮𑀁 𑀉𑀡𑁆𑀳𑀁, 𑀏𑀓𑀲𑁆𑀲 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀦 𑀯𑀺𑀫𑀮𑀁 𑀲𑀻𑀢𑀮𑀁𑁇 𑀬𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀢𑀮𑀁, 𑀢𑀁 𑀪𑁂𑀲𑀚𑁆𑀚𑀁, 𑀬𑀁 𑀉𑀡𑁆𑀳𑀁, 𑀢𑀁 𑀦 𑀪𑁂𑀲𑀚𑁆𑀚𑀦𑁆”𑀢𑀺𑁇

416\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇

---
