# 𑁩𑁇𑁬𑁇𑁭 𑀲𑀭𑀸𑀕𑀯𑀻𑀢𑀭𑀸𑀕𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀧𑀜𑁆𑀳

* 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

* 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀲𑀭𑀸𑀕𑀯𑀻𑀢𑀭𑀸𑀕𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀧𑀜𑁆𑀳

(89.)

418\. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀺𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀲𑀭𑀸𑀕𑀲𑁆𑀲 𑀘 𑀯𑀻𑀢𑀭𑀸𑀕𑀲𑁆𑀲 𑀘𑀸”𑀢𑀺? “𑀏𑀓𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑁄, 𑀏𑀓𑁄 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑁄”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀏𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑁄 𑀅𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑁄 𑀦𑀸𑀫𑀸”𑀢𑀺? “𑀏𑀓𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺𑀓𑁄, 𑀏𑀓𑁄 𑀅𑀦𑀢𑁆𑀣𑀺𑀓𑁄”𑀢𑀺𑁇 “𑀧𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀬𑁄 𑀘 𑀲𑀭𑀸𑀕𑁄, 𑀬𑁄 𑀘 𑀯𑀻𑀢𑀭𑀸𑀕𑁄, 𑀲𑀩𑁆𑀩𑁄𑀧𑁂𑀲𑁄 𑀲𑁄𑀪𑀦𑀁𑀬𑁂𑀯 𑀇𑀘𑁆𑀙𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸, 𑀦 𑀓𑁄𑀘𑀺 𑀧𑀸𑀧𑀓𑀁 𑀇𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇 “𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀲𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀭𑀲𑀭𑀸𑀕𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀧𑀦 𑀭𑀲𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀭𑀲𑀭𑀸𑀕𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻”𑀢𑀺𑁇

419\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇

---
