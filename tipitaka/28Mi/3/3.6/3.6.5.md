# 𑁩𑁇𑁬𑁇𑁫 𑀪𑀕𑀯𑀢𑁄𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑀜𑁆𑀳

* 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

* 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀪𑀕𑀯𑀢𑁄𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑀜𑁆𑀳

(87.)

412\. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀸”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀸”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸, 𑀉𑀤𑀸𑀳𑀼 𑀦𑀢𑁆𑀣𑀻”𑀢𑀺? “𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀸 𑀩𑁄𑀥𑀺𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀲𑀳 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑁂𑀦, 𑀦𑀢𑁆𑀣𑀺 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀤𑀺𑀦𑁆𑀦𑀸, 𑀬𑀣𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇

413\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇

---
