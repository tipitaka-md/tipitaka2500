# 𑁩𑁇𑁩𑁇𑁧𑁧 𑀘𑁂𑀢𑀦𑀸𑀮𑀓𑁆𑀔𑀡𑀧𑀜𑁆𑀳

### 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

### 𑀯𑀺𑀘𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀘𑁂𑀢𑀦𑀸𑀮𑀓𑁆𑀔𑀡𑀧𑀜𑁆𑀳

(59.)

296\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀺𑀁𑀮𑀓𑁆𑀔𑀡𑀸 𑀘𑁂𑀢𑀦𑀸”𑀢𑀺? “𑀘𑁂𑀢𑀬𑀺𑀢𑀮𑀓𑁆𑀔𑀡𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑁂𑀢𑀦𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀡𑀮𑀓𑁆𑀔𑀡𑀸 𑀘𑀸”𑀢𑀺𑁇

297\. “𑀑𑀧𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀳𑀻”𑀢𑀺𑁇 “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀺𑀲𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀧𑀺𑀯𑁂𑀬𑁆𑀬, 𑀧𑀭𑁂 𑀘 𑀧𑀸𑀬𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀧𑀭𑁂𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀦𑀸𑀬 𑀘𑁂𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀬𑁂𑀧𑀺 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀢𑁂𑀧𑀺 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

298\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺𑀤𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀧𑁆𑀧𑀺𑀦𑀯𑀦𑀻𑀢𑀢𑁂𑀮𑀫𑀥𑀼𑀨𑀸𑀡𑀺𑀢𑀁 𑀏𑀓𑀚𑁆𑀛𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀧𑀺𑀯𑁂𑀬𑁆𑀬, 𑀧𑀭𑁂 𑀘 𑀧𑀸𑀬𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀲𑀼𑀔𑀺𑀢𑁄 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀧𑀭𑁂𑀧𑀺 𑀲𑀼𑀔𑀺𑀢𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀦𑀸𑀬 𑀘𑁂𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁂𑀧𑀺 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀢𑁂𑀧𑀺 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑁂𑀢𑀬𑀺𑀢𑀮𑀓𑁆𑀔𑀡𑀸 𑀘𑁂𑀢𑀦𑀸 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀡𑀮𑀓𑁆𑀔𑀡𑀸 𑀘𑀸”𑀢𑀺𑁇

299\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇
