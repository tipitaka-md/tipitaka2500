# 𑁩𑁇𑁩𑁇𑁧 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀽𑀮𑀧𑀜𑁆𑀳

### 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

### 𑀯𑀺𑀘𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀽𑀮𑀧𑀜𑁆𑀳

(49.)

228\. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀢𑀻𑀢𑀲𑁆𑀲 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑁆𑀲 𑀓𑀺𑀁 𑀫𑀽𑀮𑀁, 𑀅𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑁆𑀲 𑀓𑀺𑀁 𑀫𑀽𑀮𑀁, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑁆𑀲 𑀓𑀺𑀁 𑀫𑀽𑀮𑀦𑁆”𑀢𑀺? “𑀅𑀢𑀻𑀢𑀲𑁆𑀲 𑀘, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀘 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀘 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀫𑀽𑀮𑀁𑁇 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁, 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀫𑀸 𑀓𑁄𑀝𑀺 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀻”𑀢𑀺𑁇

229\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇
