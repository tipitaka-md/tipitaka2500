### 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀧𑀜𑁆𑀳

(76.)

369. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀥𑀫𑁆𑀫𑁄 𑀢𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄”𑀢𑀺𑁇 “𑀩𑀼𑀤𑁆𑀥𑀦𑁂𑀢𑁆𑀢𑀺𑀬𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀼𑀤𑁆𑀥𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀲𑀸𑀯𑀓𑁂𑀳𑀺 𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

370. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇
