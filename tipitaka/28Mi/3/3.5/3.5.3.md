# 𑁩𑁇𑁫𑁇𑁩 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀪𑀸𑀯𑀚𑀸𑀦𑀦𑀧𑀜𑁆𑀳

### 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀪𑀸𑀯𑀚𑀸𑀦𑀦𑀧𑀜𑁆𑀳

(75.)

366\. 𑀭𑀸𑀚𑀸 𑀆𑀳—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀓𑁆𑀓𑀸 𑀚𑀸𑀦𑀺𑀢𑀼𑀁 ‘𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄’”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑁆𑀓𑀸 𑀚𑀸𑀦𑀺𑀢𑀼𑀁 ‘𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄’”𑀢𑀺𑁇 “𑀓𑀣𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀓𑁆𑀓𑀸 𑀚𑀸𑀦𑀺𑀢𑀼𑀁 ‘𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄’”𑀢𑀺? “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑁄 𑀦𑀸𑀫 𑀮𑁂𑀔𑀸𑀘𑀭𑀺𑀬𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀸𑀦𑀺 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀲𑁆𑀲 𑀓𑀣𑀁 𑀲𑁄 𑀜𑀸𑀬𑀢𑀻”𑀢𑀺𑁇 “𑀮𑁂𑀔𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑁄 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀲𑀺𑀢𑁄”𑀢𑀺𑁇

367\. “𑀓𑀮𑁆𑀮𑁄𑀲𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦𑀸”𑀢𑀺𑁇

---

368\. 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀪𑀸𑀯𑀚𑀸𑀦𑀦𑀧𑀜𑁆𑀳𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
