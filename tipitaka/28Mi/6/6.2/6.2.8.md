# 𑁬𑁇𑁨𑁇𑁮 𑀅𑀦𑀯𑀲𑁂𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀯𑀲𑁂𑀲𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀧𑀜𑁆𑀳

(187.)

1198. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁂 𑀢𑁂 𑀅𑀳𑁂𑀲𑀼𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀸𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀆𑀘𑀭𑀺𑀬𑀸 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀦𑀸𑀭𑀤𑁄 𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀭𑀻 𑀅𑀗𑁆𑀕𑀺𑀭𑀲𑁄 𑀓𑀧𑀺𑀮𑁄 𑀓𑀡𑁆𑀟𑀭𑀕𑁆𑀕𑀺 𑀲𑀸𑀫𑁄 𑀅𑀢𑀼𑀮𑁄 𑀧𑀼𑀩𑁆𑀩𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄, 𑀲𑀩𑁆𑀩𑁂𑀧𑁂𑀢𑁂 𑀆𑀘𑀭𑀺𑀬𑀸 𑀲𑀓𑀺𑀁𑀬𑁂𑀯 𑀭𑁄𑀕𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀦𑀺𑀤𑀸𑀦𑀜𑁆𑀘 𑀲𑀪𑀸𑀯𑀜𑁆𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀜𑁆𑀘 𑀓𑀺𑀭𑀺𑀬𑀜𑁆𑀘 𑀲𑀺𑀤𑁆𑀥𑀸𑀲𑀺𑀤𑁆𑀥𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑀯𑀲𑁂𑀲𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 ‘𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀬𑁂 𑀏𑀢𑁆𑀢𑀓𑀸 𑀭𑁄𑀕𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺 𑀏𑀓𑀧𑁆𑀧𑀳𑀸𑀭𑁂𑀦 𑀓𑀮𑀸𑀧𑀕𑁆𑀕𑀸𑀳𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀢𑀁 𑀩𑀦𑁆𑀥𑀺𑀁𑀲𑀼, 𑀅𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀦𑁄 𑀏𑀢𑁂 𑀲𑀩𑁆𑀩𑁂, 𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽 𑀲𑀫𑀸𑀦𑁄 𑀅𑀦𑀸𑀕𑀢𑀁 𑀓𑀺𑀭𑀺𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀜𑀸𑀡𑁂𑀦 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 ‘𑀏𑀢𑁆𑀢𑀓𑁂 𑀦𑀸𑀫 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀏𑀢𑁆𑀢𑀓𑀁 𑀦𑀸𑀫 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀧𑀭𑀺𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀯𑀲𑁂𑀲𑀢𑁄 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀦 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀬𑀲𑁂 𑀧𑀸𑀓𑀝𑁂 𑀤𑁄𑀲𑁂 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑁂 𑀧𑀼𑀣𑀼𑀕𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀢𑀲𑁆𑀫𑀺𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀓𑀸𑀮𑁂 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀻”𑀢𑀺?

1199. “𑀜𑀸𑀢𑀫𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 ‘𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀇𑀫𑁂𑀲𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀲𑀸𑀥𑀺𑀓𑀁 𑀤𑀺𑀬𑀟𑁆𑀠𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀲𑀢𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀅𑀳𑁄𑀲𑀺 ‘𑀲𑀘𑁂 𑀔𑁄 𑀅𑀳𑀁 𑀲𑀸𑀥𑀺𑀓𑀁 𑀤𑀺𑀬𑀟𑁆𑀠𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀲𑀢𑀁 𑀏𑀓𑀧𑁆𑀧𑀳𑀸𑀭𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀫𑀳𑀸𑀚𑀦𑁄 𑀲𑀦𑁆𑀢𑀸𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 “𑀩𑀳𑀼𑀓𑀁 𑀇𑀥 𑀭𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁, 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀯𑀢 𑀪𑁄 𑀲𑀫𑀡𑀲𑁆𑀲 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀓𑀸𑀫𑀸𑀧𑀺 𑀦 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀘𑀦𑀜𑁆𑀘 𑀫𑁂 𑀦 𑀲𑀤𑁆𑀤𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑀤𑁆𑀤𑀳𑀦𑁆𑀢𑀸 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀅𑀧𑀸𑀬𑀕𑀸𑀫𑀺𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀓𑀝𑁂 𑀤𑁄𑀲𑁂 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇

1200. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁, 𑀬𑀸𑀯 𑀫𑀳𑀦𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀁, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀼𑀦𑀺𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑁄 𑀏𑀲𑁄 𑀅𑀢𑁆𑀣𑁄 𑀢𑀣𑀸𑀕𑀢𑁂𑀦, ‘𑀩𑀳𑀼𑀓𑀁 𑀇𑀥 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺 𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀦𑁆𑀢𑀸𑀲𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀏𑀓𑁄𑀧𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀦 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
