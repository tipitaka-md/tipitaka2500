# 𑁬𑁇𑁨𑁇𑁩 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕

### 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑀧𑀜𑁆𑀳

(182.)

1174\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀳𑀢𑁄 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑁄”𑀢𑀺? “𑀯𑀺𑀕𑀢𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀦𑁆𑀢𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑁄”𑀢𑀺𑁇 “𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀳𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺? “𑀓𑀼𑀝𑀺𑀓𑀸𑀭𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁆𑀢𑁂, 𑀯𑀺𑀓𑀸𑀮𑁂 𑀓𑀸𑀮𑀲𑀜𑁆𑀜𑀸𑀬, 𑀧𑀯𑀸𑀭𑀺𑀢𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑀲𑀜𑁆𑀜𑀸𑀬, 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑁂 𑀅𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀬𑀸”𑀢𑀺𑁇

1175\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀬𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀅𑀚𑀸𑀦𑀦𑁂𑀦 𑀯𑀸’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀳𑀢𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀁 𑀅𑀭𑀳𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇

1176\. “𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀭𑀳𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀘 𑀅𑀭𑀳𑀢𑁄 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁, 𑀢𑁂𑀦 𑀳𑀺 𑀅𑀢𑁆𑀣𑀺 𑀅𑀭𑀳𑀢𑁄 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑁄”𑀢𑀺? “𑀦𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀢𑁄 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀅𑀭𑀳𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

1177\. “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀫𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀺, 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀦𑁆”𑀢𑀺? “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀺𑀮𑁂𑀲𑀸 𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀁 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀚𑁆𑀚𑀜𑁆𑀘𑀸𑀢𑀺𑁇 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀁? 𑀤𑀲 𑀅𑀓𑀼𑀲𑀮𑀓𑀫𑁆𑀫𑀧𑀣𑀸, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀁𑁇 𑀓𑀢𑀫𑀁 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀚𑁆𑀚𑀁? 𑀬𑀁 𑀮𑁄𑀓𑁂 𑀅𑀢𑁆𑀣𑀺 𑀲𑀫𑀡𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁, 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀦𑀯𑀚𑁆𑀚𑀁𑁇 𑀢𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺 ‘𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑀦𑁆’𑀢𑀺𑁇 𑀯𑀺𑀓𑀸𑀮𑀪𑁄𑀚𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀓𑀲𑁆𑀲 𑀅𑀦𑀯𑀚𑁆𑀚𑀁, 𑀢𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀯𑀚𑁆𑀚𑀁𑁇 𑀪𑀽𑀢𑀕𑀸𑀫𑀯𑀺𑀓𑁄𑀧𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀓𑀲𑁆𑀲 𑀅𑀦𑀯𑀚𑁆𑀚𑀁, 𑀢𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀯𑀚𑁆𑀚𑀁𑁇 𑀉𑀤𑀓𑁂 𑀳𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀮𑁄𑀓𑀲𑁆𑀲 𑀅𑀦𑀯𑀚𑁆𑀚𑀁, 𑀢𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀯𑀚𑁆𑀚𑀁𑁇 𑀇𑀢𑀺 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀯𑀚𑁆𑀚𑀸𑀦𑀺, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀚𑁆𑀚𑀁𑁇 𑀮𑁄𑀓𑀯𑀚𑁆𑀚𑀁 𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀘𑀭𑀺𑀢𑀼𑀁𑁇

1178\. 𑀬𑀁 𑀓𑀺𑀮𑁂𑀲𑀁 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀚𑁆𑀚𑀁, 𑀢𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀅𑀯𑀺𑀲𑀬𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁄 𑀲𑀩𑁆𑀩𑀁 𑀚𑀸𑀦𑀺𑀢𑀼𑀁, 𑀦 𑀳𑀺 𑀢𑀲𑁆𑀲 𑀩𑀮𑀁 𑀅𑀢𑁆𑀣𑀺 𑀲𑀩𑁆𑀩𑀁 𑀚𑀸𑀦𑀺𑀢𑀼𑀁𑁇 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀢𑁄 𑀇𑀢𑁆𑀣𑀺𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀦𑀸𑀫𑀫𑁆𑀧𑀺 𑀕𑁄𑀢𑁆𑀢𑀫𑁆𑀧𑀺, 𑀫𑀕𑁆𑀕𑁄𑀧𑀺 𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀬𑀸 𑀅𑀦𑀜𑁆𑀜𑀸𑀢𑁄; 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁𑀬𑁂𑀯, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀅𑀭𑀳𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬; 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑁄 𑀅𑀭𑀳𑀸 𑀲𑀓𑀯𑀺𑀲𑀬𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬; 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁄𑀯 𑀲𑀩𑁆𑀩𑀁 𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

---

1179\. 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑀧𑀜𑁆𑀳𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
