# 𑁬𑁇𑁨𑁇𑁧 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀧𑀜𑁆𑀳

(180.)

1160\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀳𑀭𑀣 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄’𑀢𑀺, 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀦𑁆”𑀢𑀺? “𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀁, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀁, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀁, 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀁 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀦𑁆”𑀢𑀺𑁇

1161\. “𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀁, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀁 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀁, 𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀇𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀼𑀢𑁆𑀢𑀁 𑀕𑁂𑀬𑁆𑀬𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀕𑀸𑀣𑀁 𑀉𑀤𑀸𑀦𑀁 𑀇𑀢𑀺𑀯𑀼𑀢𑁆𑀢𑀓𑀁 𑀚𑀸𑀢𑀓𑀁 𑀅𑀩𑁆𑀪𑀼𑀢𑀥𑀫𑁆𑀫𑀁 𑀯𑁂𑀤𑀮𑁆𑀮𑀁, 𑀦𑀯𑀓𑀫𑁆𑀫𑁂𑀦 𑀧𑀮𑀺𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀤𑀸𑀦𑁂𑀦 𑀘 𑀧𑀽𑀚𑀸𑀬 𑀘, 𑀦𑀦𑀼 𑀢𑁂 𑀚𑀺𑀦𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺?

1162\. “𑀬𑁂 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀼𑀢𑁆𑀢𑀁 𑀕𑁂𑀬𑁆𑀬𑀁 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀕𑀸𑀣𑀁 𑀉𑀤𑀸𑀦𑀁 𑀇𑀢𑀺𑀯𑀼𑀢𑁆𑀢𑀓𑀁 𑀚𑀸𑀢𑀓𑀁 𑀅𑀩𑁆𑀪𑀼𑀢𑀥𑀫𑁆𑀫𑀁 𑀯𑁂𑀤𑀮𑁆𑀮𑀁, 𑀦𑀯𑀓𑀫𑁆𑀫𑁂𑀦 𑀧𑀮𑀺𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀤𑀸𑀦𑁂𑀦 𑀘 𑀧𑀽𑀚𑀸𑀬 𑀘, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀪𑀸𑀯𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀲𑀺𑀢𑀯𑀸𑀲𑀦𑀸, 𑀢𑁂 𑀏𑀓𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑀡𑁂𑀦 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑀸, 𑀢𑁂 𑀇𑀫𑁂𑀳𑀺 𑀧𑀬𑁄𑀕𑁂𑀳𑀺 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

1163\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀔𑁂𑀢𑁆𑀢𑁂 𑀩𑀻𑀚𑀁 𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀬𑀣𑀸𑀩𑀮𑀯𑀻𑀭𑀺𑀬𑁂𑀦 𑀯𑀺𑀦𑀸 𑀧𑀸𑀓𑀸𑀭𑀯𑀢𑀺𑀬𑀸 𑀥𑀜𑁆𑀜𑀁 𑀉𑀤𑁆𑀥𑀭𑁂𑀬𑁆𑀬, 𑀏𑀓𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀔𑁂𑀢𑁆𑀢𑁂 𑀩𑀻𑀚𑀁 𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀦𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀲𑀸𑀔𑀜𑁆𑀘 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀯𑀢𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀁 𑀉𑀤𑁆𑀥𑀭𑁂𑀬𑁆𑀬𑁇 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀯𑀢𑀺𑀧𑀸𑀓𑀸𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸, 𑀲𑀸 𑀥𑀜𑁆𑀜𑀢𑁆𑀣𑀸𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂 𑀢𑁂 𑀲𑀪𑀸𑀯𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀲𑀺𑀢𑀯𑀸𑀲𑀦𑀸, 𑀢𑁂 𑀏𑀓𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑀡𑁂𑀦 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀯𑀺𑀦𑀸 𑀯𑀢𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀺𑀬 𑀥𑀜𑁆𑀜𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑀸, 𑀢𑁂 𑀇𑀫𑁂𑀳𑀺 𑀧𑀬𑁄𑀕𑁂𑀳𑀺 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀯𑀢𑀺𑀧𑀸𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀺𑀬 𑀥𑀜𑁆𑀜𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇

1164\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀳𑀢𑀺𑀫𑀳𑀦𑁆𑀢𑁂 𑀅𑀫𑁆𑀩𑀭𑀼𑀓𑁆𑀔𑀫𑀢𑁆𑀣𑀓𑁂 𑀨𑀮𑀧𑀺𑀡𑁆𑀟𑀺 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀅𑀣 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀨𑀮𑀁 𑀳𑀭𑁂𑀬𑁆𑀬, 𑀬𑁄 𑀧𑀦 𑀢𑀢𑁆𑀣 𑀅𑀦𑀺𑀤𑁆𑀥𑀺𑀫𑀸, 𑀲𑁄 𑀓𑀝𑁆𑀞𑀜𑁆𑀘 𑀯𑀮𑁆𑀮𑀺𑀜𑁆𑀘 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑁆𑀲𑁂𑀡𑀺𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬 𑀢𑀁 𑀭𑀼𑀓𑁆𑀔𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀨𑀮𑀁 𑀳𑀭𑁂𑀬𑁆𑀬𑁇 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑁂𑀡𑀺𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸, 𑀲𑀸 𑀨𑀮𑀢𑁆𑀣𑀸𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂 𑀢𑁂 𑀲𑀪𑀸𑀯𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀲𑀺𑀢𑀯𑀸𑀲𑀦𑀸, 𑀢𑁂 𑀏𑀓𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑀡𑁂𑀦 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀯𑀺𑀬 𑀭𑀼𑀓𑁆𑀔𑀨𑀮𑀁 𑀳𑀭𑀦𑁆𑀢𑁄𑁇 𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑀸, 𑀢𑁂 𑀇𑀫𑀺𑀦𑀸 𑀧𑀬𑁄𑀕𑁂𑀦 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀪𑀺𑀲𑀫𑁂𑀦𑁆𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑁂𑀡𑀺𑀬𑀸 𑀯𑀺𑀬 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀼𑀓𑁆𑀔𑀨𑀮𑀁 𑀳𑀭𑀦𑁆𑀢𑁄𑁇

1165\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀢𑁆𑀣𑀓𑀭𑀡𑀺𑀓𑁄 𑀏𑀓𑀓𑁄𑀬𑁂𑀯 𑀲𑀸𑀫𑀺𑀓𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀣𑀁 𑀲𑀸𑀥𑁂𑀢𑀺𑁇 𑀏𑀓𑁄 𑀥𑀦𑀯𑀸 𑀥𑀦𑀯𑀲𑁂𑀦 𑀧𑀭𑀺𑀲𑀁 𑀯𑀟𑁆𑀠𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀅𑀢𑁆𑀣𑀁 𑀲𑀸𑀥𑁂𑀢𑀺𑁇 𑀬𑀸 𑀢𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀲𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸, 𑀲𑀸 𑀅𑀢𑁆𑀣𑀢𑁆𑀣𑀸𑀬𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂 𑀢𑁂 𑀲𑀪𑀸𑀯𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀸𑀲𑀺𑀢𑀯𑀸𑀲𑀦𑀸, 𑀢𑁂 𑀏𑀓𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑀡𑁂𑀦 𑀙𑀲𑀼 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀲𑀼 𑀯𑀲𑀺𑀪𑀸𑀯𑀁 𑀧𑀸𑀧𑀼𑀡𑀦𑁆𑀢𑀺, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀺𑀬 𑀏𑀓𑀓𑁄 𑀅𑀢𑁆𑀣𑀲𑀺𑀤𑁆𑀥𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄𑁇 𑀬𑁂 𑀧𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀸𑀭𑀚𑀓𑁆𑀔𑀸, 𑀢𑁂 𑀇𑀫𑁂𑀳𑀺 𑀧𑀬𑁄𑀕𑁂𑀳𑀺 𑀲𑀸𑀫𑀜𑁆𑀜𑀢𑁆𑀣𑀫𑀪𑀺𑀲𑀸𑀥𑁂𑀦𑁆𑀢𑀺, 𑀧𑀭𑀺𑀲𑀸𑀬 𑀯𑀺𑀬 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀢𑁆𑀣𑀲𑀺𑀤𑁆𑀥𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄𑁇

1166\. 𑀉𑀤𑁆𑀤𑁂𑀲𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀳𑀼𑀓𑀸𑀭𑁄, 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀸, 𑀦𑀯𑀓𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀁, 𑀤𑀸𑀦𑀫𑁆𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀁, 𑀧𑀽𑀚𑀸𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀸 𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀭𑀸𑀚𑀽𑀧𑀲𑁂𑀯𑀻 𑀓𑀢𑀸𑀯𑀻 𑀅𑀫𑀘𑁆𑀘𑀪𑀝𑀩𑀮𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀧𑀸𑀭𑀺𑀲𑀚𑁆𑀚𑀚𑀦𑁂𑀳𑀺, 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀓𑀭𑀡𑀻𑀬𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀉𑀧𑀓𑀸𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑁆𑀤𑁂𑀲𑁄𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑁄, 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀸, 𑀦𑀯𑀓𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀁, 𑀤𑀸𑀦𑀫𑁆𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀁, 𑀧𑀽𑀚𑀸𑀧𑀺 𑀩𑀳𑀼𑀓𑀸𑀭𑀸 𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀓𑀭𑀡𑀻𑀬𑁂𑀲𑀼𑁇 𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀅𑀪𑀺𑀚𑀸𑀢𑀺𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀓𑀭𑀡𑀻𑀬𑀁 𑀦 𑀪𑀯𑁂𑀬𑁆𑀬𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀯𑀦𑁂𑀦 𑀓𑀭𑀡𑀻𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀣𑁂𑀭𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑀭𑀺𑀫𑀺𑀢𑀫𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀓𑀧𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀉𑀧𑀘𑀺𑀢𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀓𑁄𑀝𑀺𑀁 𑀕𑀢𑁄, 𑀲𑁄𑀧𑀺 𑀯𑀺𑀦𑀸 𑀲𑀯𑀦𑁂𑀦 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑀼𑀁, 𑀢𑀲𑁆𑀫𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀳𑀼𑀓𑀸𑀭𑀁 𑀲𑀯𑀦𑀁, 𑀢𑀣𑀸 𑀉𑀤𑁆𑀤𑁂𑀲𑁄𑀧𑀺 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀧𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀉𑀤𑁆𑀤𑁂𑀲𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀧𑀺 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸 𑀲𑀗𑁆𑀔𑀢𑀸”𑀢𑀺𑁇 “𑀲𑀼𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑀺𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀧𑀜𑁆𑀳𑁄, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
