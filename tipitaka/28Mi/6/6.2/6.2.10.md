# 𑁬𑁇𑁨𑁇𑁧𑁦 𑀓𑀞𑀺𑀦𑀢𑀧𑀦𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕

### 𑀓𑀞𑀺𑀦𑀢𑀧𑀦𑀧𑀜𑁆𑀳

(189.)

1204\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀫𑀦𑁆𑀢𑁂 𑀲𑀽𑀭𑀺𑀬𑁄 𑀓𑀞𑀺𑀦𑀁 𑀢𑀧𑀢𑀺, 𑀦𑁄 𑀢𑀣𑀸 𑀕𑀺𑀫𑁆𑀳𑁂”𑀢𑀺? “𑀕𑀺𑀫𑁆𑀳𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑀼𑀧𑀳𑀢𑀁 𑀳𑁄𑀢𑀺 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁, 𑀯𑀸𑀢𑀓𑁆𑀔𑀼𑀪𑀺𑀢𑀸 𑀭𑁂𑀡𑀽 𑀕𑀕𑀦𑀸𑀦𑀼𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀓𑀸𑀲𑁂𑀧𑀺 𑀅𑀩𑁆𑀪𑀸 𑀲𑀼𑀩𑀳𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸𑀯𑀸𑀢𑁄 𑀘 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀯𑀸𑀬𑀢𑀺, 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀸 𑀲𑀫𑀸𑀬𑀼𑀢𑀸 𑀲𑀽𑀭𑀺𑀬𑀭𑀁𑀲𑀺𑀬𑁄 𑀧𑀺𑀤𑀳𑀦𑁆𑀢𑀺, 𑀢𑁂𑀦 𑀕𑀺𑀫𑁆𑀳𑁂 𑀲𑀽𑀭𑀺𑀬𑁄 𑀫𑀦𑁆𑀤𑀁 𑀢𑀧𑀢𑀺𑁇

1205\. 𑀳𑁂𑀫𑀦𑁆𑀢𑁂 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀳𑁂𑀝𑁆𑀞𑀸 𑀧𑀣𑀯𑀻 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀉𑀧𑀭𑀺 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑀲𑀦𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁, 𑀭𑁂𑀡𑀼 𑀘 𑀲𑀦𑁆𑀢𑀲𑀦𑁆𑀢𑀁 𑀕𑀕𑀦𑁂 𑀘𑀭𑀢𑀺, 𑀯𑀺𑀕𑀢𑀯𑀮𑀸𑀳𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀆𑀓𑀸𑀲𑁄, 𑀯𑀸𑀢𑁄 𑀘 𑀫𑀦𑁆𑀤𑀫𑀦𑁆𑀤𑀁 𑀯𑀸𑀬𑀢𑀺, 𑀏𑀢𑁂𑀲𑀁 𑀉𑀧𑀭𑀢𑀺𑀬𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀽𑀭𑀺𑀬𑀭𑀁𑀲𑀺𑀬𑁄, 𑀉𑀧𑀖𑀸𑀢𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀢𑀸𑀧𑁄 𑀅𑀢𑀺𑀯𑀺𑀬 𑀢𑀧𑀢𑀺𑁇 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀲𑀽𑀭𑀺𑀬𑁄 𑀳𑁂𑀫𑀦𑁆𑀢𑁂 𑀓𑀞𑀺𑀦𑀁 𑀢𑀧𑀢𑀺, 𑀦𑁄 𑀢𑀣𑀸 𑀕𑀺𑀫𑁆𑀳𑁂”𑀢𑀺𑁇 “𑀲𑀩𑁆𑀩𑀻𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀽𑀭𑀺𑀬𑁄 𑀓𑀞𑀺𑀦𑀁 𑀢𑀧𑀢𑀺, 𑀫𑁂𑀖𑀸𑀤𑀺𑀲𑀳𑀕𑀢𑁄 𑀓𑀞𑀺𑀦𑀁 𑀦 𑀢𑀧𑀢𑀻”𑀢𑀺𑁇
