# 𑁬𑁇𑁨𑁇𑁭 𑀬𑀓𑁆𑀔𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕

### 𑀬𑀓𑁆𑀔𑀧𑀜𑁆𑀳

(186.)

1196\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀬𑀓𑁆𑀔𑀸 𑀦𑀸𑀫𑀸”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀬𑀓𑁆𑀔𑀸 𑀦𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀘𑀯𑀦𑁆𑀢𑀺 𑀧𑀦 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀓𑁆𑀔𑀸 𑀢𑀫𑁆𑀳𑀸 𑀬𑁄𑀦𑀺𑀬𑀸”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀯𑀦𑁆𑀢𑀺 𑀢𑁂 𑀬𑀓𑁆𑀔𑀸 𑀢𑀫𑁆𑀳𑀸 𑀬𑁄𑀦𑀺𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑁂𑀲𑀁 𑀫𑀢𑀸𑀦𑀁 𑀬𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑀭𑀻𑀭𑀁 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀼𑀡𑀧𑀕𑀦𑁆𑀥𑁄𑀧𑀺 𑀦 𑀯𑀸𑀬𑀢𑀻”𑀢𑀺? “𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀢𑀸𑀦𑀁 𑀬𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑀭𑀻𑀭𑀁, 𑀓𑀼𑀡𑀧𑀕𑀦𑁆𑀥𑁄𑀧𑀺 𑀢𑁂𑀲𑀁 𑀯𑀸𑀬𑀢𑀺, 𑀫𑀢𑀸𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑀭𑀻𑀭𑀁 𑀓𑀻𑀝𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀺𑀫𑀺𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀺𑀧𑀺𑀮𑁆𑀮𑀺𑀓𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀝𑀗𑁆𑀕𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀳𑀺𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀘𑁆𑀙𑀺𑀓𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀢𑀧𑀤𑀺𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀤𑀺𑀚𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀺𑀕𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀸 𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀓𑁄 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀜𑁆𑀜𑁄 𑀇𑀤𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀯𑀺𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀯𑀸𑀤𑀺𑀲𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀺𑀫𑀢𑀸”𑀢𑀺𑁇
