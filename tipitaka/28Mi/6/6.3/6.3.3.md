# 𑁬𑁇𑁩𑁇𑁩 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀩𑀮𑀯𑀢𑀭𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀩𑀮𑀯𑀢𑀭𑀧𑀜𑁆𑀳

(192.)

1263\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁 𑀓𑀼𑀲𑀮𑀁 𑀯𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀯𑀸”𑀢𑀺? “𑀓𑀼𑀲𑀮𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀦𑁄 𑀢𑀣𑀸 𑀅𑀓𑀼𑀲𑀮𑀦𑁆”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀺 ‘𑀓𑀼𑀲𑀮𑀁 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀦𑁄 𑀢𑀣𑀸 𑀅𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺, 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀥 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀺𑀦𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀺𑀦𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀺𑀦𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄 𑀕𑀸𑀫𑀖𑀸𑀢𑀺𑀓𑀸 𑀧𑀦𑁆𑀣𑀤𑀽𑀲𑀓𑀸 𑀦𑁂𑀓𑀢𑀺𑀓𑀸 𑀯𑀜𑁆𑀘𑀦𑀺𑀓𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀢𑀸𑀯𑀢𑀓𑁂𑀦 𑀧𑀸𑀧𑁂𑀦 𑀮𑀪𑀦𑁆𑀢𑀺 𑀳𑀢𑁆𑀣𑀘𑁆𑀙𑁂𑀤𑀁 𑀧𑀸𑀤𑀘𑁆𑀙𑁂𑀤𑀁 𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀘𑁆𑀙𑁂𑀤𑀁 𑀓𑀡𑁆𑀡𑀘𑁆𑀙𑁂𑀤𑀁 𑀦𑀸𑀲𑀘𑁆𑀙𑁂𑀤𑀁 𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀘𑁆𑀙𑁂𑀤𑀁 𑀩𑀺𑀮𑀗𑁆𑀕𑀣𑀸𑀮𑀺𑀓𑀁 𑀲𑀗𑁆𑀔𑀫𑀼𑀡𑁆𑀟𑀺𑀓𑀁 𑀭𑀸𑀳𑀼𑀫𑀼𑀔𑀁 𑀚𑁄𑀢𑀺𑀫𑀸𑀮𑀺𑀓𑀁 𑀳𑀢𑁆𑀣𑀧𑀚𑁆𑀚𑁄𑀢𑀺𑀓𑀁 𑀏𑀭𑀓𑀯𑀢𑁆𑀢𑀺𑀓𑀁 𑀘𑀻𑀭𑀓𑀯𑀸𑀲𑀺𑀓𑀁 𑀏𑀡𑁂𑀬𑁆𑀬𑀓𑀁 𑀩𑀍𑀇𑀲𑀫𑀁𑀲𑀺𑀓𑀁 𑀓𑀳𑀸𑀧𑀡𑀺𑀓𑀁 𑀔𑀸𑀭𑀸𑀧𑀢𑀘𑁆𑀙𑀺𑀓𑀁 𑀧𑀮𑀺𑀖𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀓𑀁 𑀧𑀮𑀸𑀮𑀧𑀻𑀞𑀓𑀁 𑀢𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀢𑁂𑀮𑁂𑀦 𑀑𑀲𑀺𑀜𑁆𑀘𑀦𑀁 𑀲𑀼𑀦𑀔𑁂𑀳𑀺𑀧𑀺 𑀔𑀸𑀤𑀸𑀧𑀦𑀁 𑀚𑀻𑀯𑀲𑀽𑀮𑀸𑀭𑁄𑀧𑀦𑀁 𑀅𑀲𑀺𑀦𑀸𑀧𑀺 𑀲𑀻𑀲𑀘𑁆𑀙𑁂𑀤𑀁, 𑀓𑁂𑀘𑀺 𑀭𑀢𑁆𑀢𑀺𑀁 𑀧𑀸𑀧𑀁 𑀓𑀢𑁆𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀬𑁂𑀯 𑀯𑀺𑀧𑀸𑀓𑀁 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀭𑀢𑁆𑀢𑀺𑀁 𑀓𑀢𑁆𑀯𑀸 𑀤𑀺𑀯𑀸𑀬𑁂𑀯 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀤𑀺𑀯𑀸 𑀓𑀢𑁆𑀯𑀸 𑀤𑀺𑀯𑀸𑀬𑁂𑀯 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀤𑀺𑀯𑀸 𑀓𑀢𑁆𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀬𑁂𑀯 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀤𑁆𑀯𑁂 𑀢𑀬𑁄 𑀤𑀺𑀯𑀲𑁂 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁂 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀯𑀺𑀧𑀸𑀓𑀁 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁄𑀘𑀺 𑀏𑀓𑀲𑁆𑀲 𑀯𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀯𑀸 𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑀸 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀸 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀸 𑀤𑀲𑀦𑁆𑀦𑀁 𑀯𑀸 𑀲𑀢𑀲𑁆𑀲 𑀯𑀸 𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲 𑀯𑀸 𑀲𑀧𑀭𑀺𑀯𑀸𑀭𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀪𑁄𑀕𑀁 𑀯𑀸 𑀬𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀅𑀦𑀼𑀪𑀯𑀺𑀢𑀸 𑀲𑀻𑀮𑁂𑀦 𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑁂𑀦 𑀯𑀸”𑀢𑀺?

1264\. “𑀅𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀸 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀢𑁂𑀦𑁂𑀯 𑀲𑀭𑀻𑀭𑀤𑁂𑀳𑁂𑀦 𑀢𑀺𑀤𑀲𑀧𑀼𑀭𑁂 𑀲𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸”𑀢𑀺𑁇 “𑀓𑁄 𑀘 𑀓𑁄 𑀘, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺? “𑀫𑀦𑁆𑀥𑀸𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀭𑀸𑀚𑀸, 𑀦𑀺𑀫𑀺 𑀭𑀸𑀚𑀸, 𑀲𑀸𑀥𑀻𑀦𑁄 𑀭𑀸𑀚𑀸, 𑀕𑀼𑀢𑁆𑀢𑀺𑀮𑁄 𑀘 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

1265\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀦𑁂𑀓𑁂𑀳𑀺 𑀢𑀁 𑀪𑀯𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺 𑀅𑀦𑁆𑀢𑀭𑀺𑀢𑀁, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑁂𑀢𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀭𑁄𑀓𑁆𑀔𑀁, 𑀬𑀤𑀺 𑀲𑀫𑀢𑁆𑀣𑁄𑀲𑀺 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀓𑁂 𑀪𑀯𑁂 𑀪𑀕𑀯𑀢𑁄 𑀥𑀭𑀫𑀸𑀦𑀓𑀸𑀮𑁂 𑀓𑀣𑁂𑀳𑀻”𑀢𑀺? “𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀓𑁂𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀯𑁂 𑀧𑀼𑀡𑁆𑀡𑀓𑁄 𑀤𑀸𑀲𑁄 𑀣𑁂𑀭𑀲𑁆𑀲 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑁄𑀚𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀢𑀤𑀳𑁂𑀯 𑀲𑁂𑀝𑁆𑀞𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀕𑀢𑁄, 𑀲𑁄 𑀏𑀢𑀭𑀳𑀺 𑀧𑀼𑀡𑁆𑀡𑀓𑁄 𑀲𑁂𑀝𑁆𑀞𑀻𑀢𑀺 𑀧𑀜𑁆𑀜𑀸𑀬𑀺, 𑀕𑁄𑀧𑀸𑀮𑀫𑀸𑀢𑀸 𑀤𑁂𑀯𑀻 𑀅𑀢𑁆𑀢𑀦𑁄 𑀓𑁂𑀲𑁂 𑀯𑀺𑀓𑁆𑀓𑀺𑀡𑀺𑀢𑁆𑀯𑀸 𑀮𑀤𑁆𑀥𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀓𑀳𑀸𑀧𑀡𑁂𑀳𑀺 𑀣𑁂𑀭𑀲𑁆𑀲 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑀲𑁆𑀲 𑀅𑀢𑁆𑀢𑀝𑁆𑀞𑀫𑀓𑀲𑁆𑀲 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀤𑀢𑁆𑀯𑀸 𑀢𑀤𑀳𑁂𑀯 𑀭𑀜𑁆𑀜𑁄 𑀘𑀦𑁆𑀤𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀫𑀳𑁂𑀲𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀸𑁇 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀕𑀺𑀮𑀸𑀦𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀊𑀭𑀼𑀫𑀁𑀲𑁂𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀻𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀤𑀺𑀯𑀲𑁂𑀬𑁂𑀯 𑀭𑀽𑀍𑀳𑀯𑀡𑀸 𑀲𑀜𑁆𑀙𑀯𑀻 𑀅𑀭𑁄𑀕𑀸 𑀚𑀸𑀢𑀸𑁇 𑀫𑀮𑁆𑀮𑀺𑀓𑀸 𑀤𑁂𑀯𑀻 𑀪𑀕𑀯𑀢𑁄 𑀆𑀪𑀺𑀤𑁄𑀲𑀺𑀓𑀁 𑀓𑀼𑀫𑁆𑀫𑀸𑀲𑀧𑀺𑀡𑁆𑀟𑀁 𑀤𑀢𑁆𑀯𑀸 𑀢𑀤𑀳𑁂𑀯 𑀭𑀜𑁆𑀜𑁄 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀫𑀳𑁂𑀲𑀻 𑀚𑀸𑀢𑀸𑁇 𑀲𑀼𑀫𑀦𑁄 𑀫𑀸𑀮𑀸𑀓𑀸𑀭𑁄 𑀅𑀝𑁆𑀞𑀳𑀺 𑀲𑀼𑀫𑀦𑀧𑀼𑀧𑁆𑀨𑀫𑀼𑀝𑁆𑀞𑀻𑀳𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀤𑀺𑀯𑀲𑀁𑀬𑁂𑀯 𑀫𑀳𑀸𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀢𑁆𑀢𑁄𑁇 𑀏𑀓𑀲𑀸𑀝𑀓𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀉𑀢𑁆𑀢𑀭𑀲𑀸𑀝𑀓𑁂𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀽𑀚𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀤𑀺𑀯𑀲𑀁 𑀬𑁂𑀯 𑀲𑀩𑁆𑀩𑀝𑁆𑀞𑀓𑀁 𑀮𑀪𑀺, 𑀲𑀩𑁆𑀩𑁂𑀧𑁂𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀪𑁄𑀕𑀜𑁆𑀘 𑀬𑀲𑀜𑁆𑀘 𑀅𑀦𑀼𑀪𑀯𑀺𑀁𑀲𑀽”𑀢𑀺𑁇

1266\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀙 𑀚𑀦𑁂𑀬𑁂𑀯 𑀅𑀤𑁆𑀤𑀲𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀓𑀼𑀲𑀮𑀁𑀬𑁂𑀯 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀦𑁄 𑀢𑀣𑀸 𑀓𑀼𑀲𑀮𑀁𑁇 𑀅𑀳𑀜𑁆𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀓𑀤𑀺𑀯𑀲𑀁𑀬𑁂𑀯 𑀤𑀲𑀧𑀺 𑀧𑀼𑀭𑀺𑀲𑁂 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀲𑀽𑀮𑁂𑀲𑀼 𑀆𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑁂, 𑀯𑀻𑀲𑀫𑁆𑀧𑀺 𑀢𑀺𑀁𑀲𑀫𑁆𑀧𑀺 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀫𑁆𑀧𑀺 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀺 𑀧𑀼𑀭𑀺𑀲𑀲𑀢𑀫𑁆𑀧𑀺 𑀧𑀼𑀭𑀺𑀲𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀲𑀽𑀮𑁂𑀲𑀼 𑀆𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑁂𑁇 𑀦𑀦𑁆𑀤𑀓𑀼𑀮𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑁄 𑀦𑀸𑀫 𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑁂𑀦 𑀘 𑀭𑀜𑁆𑀜𑀸 𑀘𑀦𑁆𑀤𑀕𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀗𑁆𑀕𑀸𑀫𑁄 𑀲𑀫𑀼𑀧𑀩𑁆𑀬𑀽𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀗𑁆𑀕𑀸𑀫𑁂 𑀉𑀪𑀢𑁄 𑀩𑀮𑀓𑀸𑀬𑁂 𑀅𑀲𑀻𑀢𑀺𑀓𑀩𑀦𑁆𑀥𑀭𑀽𑀧𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀏𑀓𑀲𑁆𑀫𑀺𑀁 𑀓𑀺𑀭 𑀲𑀻𑀲𑀓𑀩𑀦𑁆𑀥𑁂 𑀧𑀭𑀺𑀧𑀸𑀢𑁂 𑀏𑀓𑀁 𑀓𑀩𑀦𑁆𑀥𑀭𑀽𑀧𑀁 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂𑀧𑁂𑀢𑁂 𑀧𑀸𑀧𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑀸𑁇 𑀇𑀫𑀺𑀦𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀪𑀡𑀸𑀫𑀺 𑀅𑀓𑀼𑀲𑀮𑀁𑀬𑁂𑀯 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀦𑁄 𑀢𑀣𑀸 𑀓𑀼𑀲𑀮𑀦𑁆𑀢𑀺𑁇

1267\. 𑀲𑀼𑀬𑁆𑀬𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂 𑀓𑁄𑀲𑀮𑁂𑀦 𑀭𑀜𑁆𑀜𑀸 𑀅𑀲𑀤𑀺𑀲𑀤𑀸𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀬𑁆𑀬𑀢𑀻”𑀢𑀺𑁇 “𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸 𑀢𑀁 𑀅𑀲𑀤𑀺𑀲𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀜𑁆𑀘𑀺 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀪𑁄𑀕𑀁 𑀯𑀸 𑀬𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀧𑀝𑀺𑀮𑀪𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸𑀧𑀺 𑀦 𑀮𑀪𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀜𑁆𑀘𑀺 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀪𑁄𑀕𑀁 𑀯𑀸 𑀬𑀲𑀁 𑀯𑀸 𑀲𑀼𑀔𑀁 𑀯𑀸, 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀓𑀼𑀲𑀮𑀁𑀬𑁂𑀯 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀦𑁄 𑀢𑀣𑀸 𑀓𑀼𑀲𑀮𑀦𑁆”𑀢𑀺𑁇

1268\. “𑀧𑀭𑀺𑀢𑁆𑀢𑀢𑁆𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀓𑀼𑀲𑀮𑀁 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀡𑀫𑀢𑀺, 𑀯𑀺𑀧𑀼𑀮𑀢𑁆𑀢𑀸 𑀓𑀼𑀲𑀮𑀁 𑀤𑀻𑀖𑁂𑀦 𑀓𑀸𑀮𑁂𑀦 𑀧𑀭𑀺𑀡𑀫𑀢𑀺, 𑀉𑀧𑀫𑀸𑀬𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀢𑀁 𑀉𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀧𑀭𑀦𑁆𑀢𑁂 𑀚𑀦𑀧𑀤𑁂 𑀓𑀼𑀫𑀼𑀤𑀪𑀡𑁆𑀟𑀺𑀓𑀸 𑀦𑀸𑀫 𑀥𑀜𑁆𑀜𑀚𑀸𑀢𑀺 𑀫𑀸𑀲𑀮𑀽𑀦𑀸 𑀅𑀦𑁆𑀢𑁄𑀕𑁂𑀳𑀕𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀸𑀮𑀬𑁄 𑀙𑀧𑁆𑀧𑀜𑁆𑀘𑀫𑀸𑀲𑁂𑀳𑀺 𑀧𑀭𑀺𑀡𑀫𑀦𑁆𑀢𑀺, 𑀓𑀺𑀁 𑀧𑀦𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑁆𑀢𑀭𑀁 𑀓𑁄 𑀯𑀺𑀲𑁂𑀲𑁄 𑀓𑀼𑀫𑀼𑀤𑀪𑀡𑁆𑀟𑀺𑀓𑀸𑀬 𑀘 𑀲𑀸𑀮𑀻𑀦𑀜𑁆𑀘𑀸”𑀢𑀺? “𑀧𑀭𑀺𑀢𑁆𑀢𑀢𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀫𑀼𑀤𑀪𑀡𑁆𑀟𑀺𑀓𑀸𑀬, 𑀯𑀺𑀧𑀼𑀮𑀢𑁆𑀢𑀸 𑀘 𑀲𑀸𑀮𑀻𑀦𑀁𑁇 𑀲𑀸𑀮𑀬𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀭𑀸𑀚𑀸𑀭𑀳𑀸 𑀭𑀸𑀚𑀪𑁄𑀚𑀦𑀁, 𑀓𑀼𑀫𑀼𑀤𑀪𑀡𑁆𑀟𑀺𑀓𑀸 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀸𑀦𑀁 𑀪𑁄𑀚𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀭𑀺𑀢𑁆𑀢𑀢𑁆𑀢𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀡𑀫𑀢𑀺, 𑀯𑀺𑀧𑀼𑀮𑀢𑁆𑀢𑀸 𑀓𑀼𑀲𑀮𑀁 𑀤𑀻𑀖𑁂𑀦 𑀓𑀸𑀮𑁂𑀦 𑀧𑀭𑀺𑀡𑀫𑀢𑀻”𑀢𑀺𑁇

1269\. “𑀬𑀁 𑀢𑀢𑁆𑀣, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀡𑀫𑀢𑀺, 𑀢𑀁 𑀦𑀸𑀫 𑀮𑁄𑀓𑁂 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀢𑀲𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀩𑀮𑀯𑀢𑀭𑀁, 𑀦𑁄 𑀢𑀣𑀸 𑀓𑀼𑀲𑀮𑀁𑁇 𑀬𑀣𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀬𑁄𑀥𑁄 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀬𑀼𑀤𑁆𑀥𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑀢𑁆𑀢𑀼𑀁 𑀉𑀧𑀓𑀘𑁆𑀙𑀓𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀓𑀟𑁆𑀠𑀺𑀢𑁆𑀯𑀸 𑀔𑀺𑀧𑁆𑀧𑀢𑀭𑀁 𑀲𑀸𑀫𑀺𑀦𑁄 𑀉𑀧𑀦𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀬𑁄𑀥𑁄 𑀮𑁄𑀓𑁂 𑀲𑀫𑀢𑁆𑀣𑁄 𑀲𑀽𑀭𑁄 𑀦𑀸𑀫𑁇 𑀬𑁄 𑀘 𑀪𑀺𑀲𑀓𑁆𑀓𑁄 𑀔𑀺𑀧𑁆𑀧𑀁 𑀲𑀮𑁆𑀮𑀁 𑀉𑀤𑁆𑀥𑀭𑀢𑀺 𑀭𑁄𑀕𑀫𑀧𑀦𑁂𑀢𑀺, 𑀲𑁄 𑀪𑀺𑀲𑀓𑁆𑀓𑁄 𑀙𑁂𑀓𑁄 𑀦𑀸𑀫𑁇 𑀬𑁄 𑀕𑀡𑀓𑁄 𑀲𑀻𑀖𑀲𑀻𑀖𑀁 𑀕𑀡𑁂𑀢𑁆𑀯𑀸 𑀔𑀺𑀧𑁆𑀧𑀁 𑀤𑀲𑁆𑀲𑀬𑀢𑀺, 𑀲𑁄 𑀕𑀡𑀓𑁄 𑀙𑁂𑀓𑁄 𑀦𑀸𑀫𑁇 𑀬𑁄 𑀫𑀮𑁆𑀮𑁄 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀫𑀮𑁆𑀮𑀁 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀸𑀦𑀓𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀲𑁄 𑀫𑀮𑁆𑀮𑁄 𑀲𑀫𑀢𑁆𑀣𑁄 𑀲𑀽𑀭𑁄 𑀦𑀸𑀫𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀁 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀡𑀫𑀢𑀺 𑀓𑀼𑀲𑀮𑀁 𑀯𑀸 𑀅𑀓𑀼𑀲𑀮𑀁 𑀯𑀸, 𑀢𑀁 𑀮𑁄𑀓𑁂 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀦𑁆”𑀢𑀺𑁇

1270\. “𑀉𑀪𑀬𑀫𑁆𑀧𑀺 𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀫𑁆𑀫𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀯𑁂𑀤𑀦𑀻𑀬𑀫𑁂𑀯, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀅𑀓𑀼𑀲𑀮𑀁 𑀲𑀸𑀯𑀚𑁆𑀚𑀢𑀸𑀬 𑀔𑀡𑁂𑀦 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀦𑀻𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀓𑁂𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀔𑀢𑁆𑀢𑀺𑀬𑁂𑀳𑀺 𑀞𑀧𑀺𑀢𑁄 𑀏𑀲𑁄 𑀦𑀺𑀬𑀫𑁄 ‘𑀬𑁄 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺, 𑀲𑁄 𑀤𑀡𑁆𑀟𑀸𑀭𑀳𑁄…  𑀬𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀢𑀺…  𑀬𑁄 𑀧𑀭𑀤𑀸𑀭𑀁 𑀕𑀘𑁆𑀙𑀢𑀺…  𑀬𑁄 𑀫𑀼𑀲𑀸 𑀪𑀡𑀢𑀺…  𑀬𑁄 𑀕𑀸𑀫𑀁 𑀖𑀸𑀢𑁂𑀢𑀺…  𑀬𑁄 𑀧𑀦𑁆𑀣𑀁 𑀤𑀽𑀲𑁂𑀢𑀺…  𑀬𑁄 𑀦𑀺𑀓𑀢𑀺𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀬𑁄 𑀯𑀜𑁆𑀘𑀦𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀲𑁄 𑀤𑀡𑁆𑀟𑀸𑀭𑀳𑁄 𑀯𑀥𑀺𑀢𑀩𑁆𑀩𑁄 𑀙𑁂𑀢𑁆𑀢𑀩𑁆𑀩𑁄 𑀪𑁂𑀢𑁆𑀢𑀩𑁆𑀩𑁄 𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀢𑀁 𑀢𑁂 𑀉𑀧𑀸𑀤𑀸𑀬 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀤𑀡𑁆𑀟𑁂𑀦𑁆𑀢𑀺 𑀯𑀥𑁂𑀦𑁆𑀢𑀺 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀪𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺 𑀳𑀦𑀦𑁆𑀢𑀺 𑀘, 𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀳𑀺𑀘𑀺 𑀞𑀧𑀺𑀢𑁄 𑀦𑀺𑀬𑀫𑁄 ‘𑀬𑁄 𑀤𑀸𑀦𑀁 𑀯𑀸 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁 𑀯𑀸 𑀭𑀓𑁆𑀔𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀯𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀥𑀦𑀁 𑀯𑀸 𑀬𑀲𑀁 𑀯𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺; 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀥𑀦𑀁 𑀯𑀸 𑀬𑀲𑀁 𑀯𑀸 𑀤𑁂𑀦𑁆𑀢𑀺, 𑀘𑁄𑀭𑀲𑁆𑀲 𑀓𑀢𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀥𑀩𑀦𑁆𑀥𑀦𑀁 𑀯𑀺𑀬𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀥𑀦𑀁 𑀯𑀸 𑀬𑀲𑀁 𑀯𑀸 𑀤𑀤𑁂𑀬𑁆𑀬𑀼𑀁, 𑀓𑀼𑀲𑀮𑀫𑁆𑀧𑀺 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀦𑀻𑀬𑀁 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀸𑀬𑀓𑁂 𑀦 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺 ‘𑀥𑀦𑀁 𑀯𑀸 𑀬𑀲𑀁 𑀯𑀸 𑀤𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀁 𑀦 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀦𑀻𑀬𑀁𑁇 𑀇𑀫𑀺𑀦𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀓𑀼𑀲𑀮𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀯𑁂𑀤𑀦𑀻𑀬𑀁, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁂𑀯 𑀲𑁄 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀩𑀮𑀯𑀢𑀭𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀯𑀸𑀤𑀺𑀲𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁂𑀦 𑀯𑀺𑀦𑀸 𑀦𑁂𑀲𑁄 𑀧𑀜𑁆𑀳𑁄 𑀲𑀼𑀦𑀺𑀩𑁆𑀩𑁂𑀞𑀺𑀬𑁄, 𑀮𑁄𑀓𑀺𑀓𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀺𑀢𑀦𑁆”𑀢𑀺𑁇
