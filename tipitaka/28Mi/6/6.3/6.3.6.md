# 𑁬𑁇𑁩𑁇𑁬 𑀅𑀓𑀸𑀮𑀫𑀭𑀡𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

### 𑀅𑀓𑀸𑀮𑀫𑀭𑀡𑀧𑀜𑁆𑀳

(195.)

1294. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁂 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀓𑀸𑀮𑁂𑀬𑁂𑀯 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀓𑀸𑀮𑁂𑀧𑀺 𑀫𑀭𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀮𑁂𑀧𑀺 𑀫𑀭𑀡𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀸𑀮𑁂𑀧𑀺 𑀫𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

1295. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁂 𑀓𑀸𑀮𑁂 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀓𑁂 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀬𑀸 𑀅𑀫𑁆𑀩𑀭𑀼𑀓𑁆𑀔𑀸 𑀯𑀸 𑀚𑀫𑁆𑀩𑀼𑀭𑀼𑀓𑁆𑀔𑀸 𑀯𑀸, 𑀅𑀜𑁆𑀜𑀲𑁆𑀫𑀸 𑀯𑀸 𑀧𑀦 𑀨𑀮𑀭𑀼𑀓𑁆𑀔𑀸 𑀨𑀮𑀸𑀦𑀺 𑀧𑀢𑀦𑁆𑀢𑀸𑀦𑀺 𑀆𑀫𑀸𑀦𑀺 𑀘 𑀧𑀓𑁆𑀓𑀸𑀦𑀺 𑀘𑀸”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀨𑀮𑀸𑀦𑀺 𑀭𑀼𑀓𑁆𑀔𑀢𑁄 𑀧𑀢𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀓𑀸𑀮𑁂𑀬𑁂𑀯 𑀧𑀢𑀦𑁆𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀓𑀸𑀮𑁂𑀧𑀻”𑀢𑀺? “𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀨𑀮𑀸𑀦𑀺 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀸𑀦𑀺 𑀯𑀺𑀮𑀻𑀦𑀸𑀦𑀺 𑀧𑀢𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀓𑀸𑀮𑁂 𑀧𑀢𑀦𑁆𑀢𑀺𑁇 𑀬𑀸𑀦𑀺 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀅𑀯𑀲𑁂𑀲𑀸𑀦𑀺 𑀨𑀮𑀸𑀦𑀺 𑀢𑁂𑀲𑀼 𑀓𑀸𑀦𑀺𑀘𑀺 𑀓𑀺𑀫𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑀦𑀺 𑀧𑀢𑀦𑁆𑀢𑀺, 𑀓𑀸𑀦𑀺𑀘𑀺 𑀮𑀕𑀼𑀍𑀅𑀳𑀢𑀸𑀦𑀺 𑀧𑀢𑀦𑁆𑀢𑀺, 𑀓𑀸𑀦𑀺𑀘𑀺 𑀯𑀸𑀢𑀧𑁆𑀧𑀳𑀢𑀸𑀦𑀺 𑀧𑀢𑀦𑁆𑀢𑀺, 𑀓𑀸𑀦𑀺𑀘𑀺 𑀅𑀦𑁆𑀢𑁄𑀧𑀽𑀢𑀺𑀓𑀸𑀦𑀺 𑀳𑀼𑀢𑁆𑀯𑀸 𑀧𑀢𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀅𑀓𑀸𑀮𑁂 𑀧𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂 𑀢𑁂 𑀚𑀭𑀸𑀯𑁂𑀕𑀳𑀢𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂𑀬𑁂𑀯 𑀓𑀸𑀮𑁂 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀅𑀯𑀲𑁂𑀲𑀸 𑀓𑁂𑀘𑀺 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀍𑀳𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀕𑀢𑀺𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀍𑀳𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀓𑀺𑀭𑀺𑀬𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀍𑀳𑀸 𑀫𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1296. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁂 𑀢𑁂 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀍𑀳𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀬𑁂𑀧𑀺 𑀢𑁂 𑀕𑀢𑀺𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀍𑀳𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀬𑁂𑀧𑀺 𑀢𑁂 𑀓𑀺𑀭𑀺𑀬𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀍𑀳𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀬𑁂𑀧𑀺 𑀢𑁂 𑀚𑀭𑀸𑀯𑁂𑀕𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀍𑀳𑀸 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀓𑀸𑀮𑁂𑀬𑁂𑀯 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀬𑁄𑀧𑀺 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑁄 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀮𑁄, 𑀓𑀸𑀮𑁂𑀬𑁂𑀯 𑀲𑁄 𑀫𑀭𑀢𑀺𑁇 𑀬𑁄𑀧𑀺 𑀯𑀺𑀚𑀸𑀢𑀖𑀭𑁂 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀮𑁄, 𑀲𑁄𑀧𑀺 𑀓𑀸𑀮𑁂𑀬𑁂𑀯 𑀫𑀭𑀢𑀺𑁇 𑀬𑁄𑀧𑀺 𑀫𑀸𑀲𑀺𑀓𑁄 𑀫𑀭𑀢𑀺…𑀧𑁂…  𑀬𑁄𑀧𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀺𑀓𑁄 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀮𑁄, 𑀓𑀸𑀮𑁂𑀬𑁂𑀯 𑀲𑁄 𑀫𑀭𑀢𑀺, 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀁 𑀦𑀸𑀫 𑀦 𑀳𑁄𑀢𑀺, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀫𑀭𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀓𑀸𑀮𑁂𑀬𑁂𑀯 𑀫𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1297. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀚𑀺𑀖𑀘𑁆𑀙𑀺𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀉𑀧𑀳𑀢𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑁄 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺; 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀧𑀭𑀺𑀲𑀼𑀓𑁆𑀔𑀳𑀤𑀬𑁄 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺; 𑀅𑀳𑀺𑀦𑀸 𑀤𑀝𑁆𑀞𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀺𑀲𑀯𑁂𑀕𑀸𑀪𑀺𑀳𑀢𑁄 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀓𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺; 𑀯𑀺𑀲𑀫𑀸𑀲𑀺𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀟𑀬𑁆𑀳𑀦𑁆𑀢𑁂𑀲𑀼 𑀅𑀗𑁆𑀕𑀧𑀘𑁆𑀘𑀗𑁆𑀕𑁂𑀲𑀼 𑀅𑀕𑀤𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺; 𑀅𑀕𑁆𑀕𑀺𑀕𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀛𑀸𑀬𑀫𑀸𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺; 𑀉𑀤𑀓𑀕𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀢𑀺𑀝𑁆𑀞𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺; 𑀲𑀢𑁆𑀢𑀺𑀳𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀪𑀺𑀲𑀓𑁆𑀓𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺; 𑀇𑀫𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀢𑁆𑀢 𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀧𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀆𑀬𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀭𑀸𑀧𑀸𑀳𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀁𑀲𑁂𑀦 𑀯𑀤𑀸𑀫𑀺𑁇

1302. 𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀧𑀭𑁂 𑀅𑀳𑀺𑀦𑀸 𑀟𑀁𑀲𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀢𑀺, 𑀲𑁄 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀚𑀕𑀭𑀫𑀼𑀔𑁂𑀦𑁂𑀯 𑀅𑀚𑀕𑀭𑀫𑀼𑀔𑀁 𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑀫𑀼𑀔𑁂𑀦𑁂𑀯 𑀓𑀡𑁆𑀳𑀲𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀳𑀺 𑀔𑀸𑀬𑀺𑀢𑀔𑀸𑀬𑀺𑀢𑁄 𑀅𑀳𑀻𑀳𑀺 𑀤𑀝𑁆𑀞𑁄𑀬𑁂𑀯 𑀫𑀭𑀢𑀺 𑀤𑀳𑀭𑁄𑀧𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄𑀧𑀺 𑀫𑀳𑀮𑁆𑀮𑀓𑁄𑀧𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀬𑀺𑀓𑀫𑀭𑀡𑀁𑁇

1303. 𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀧𑀭𑁂 𑀯𑀺𑀲𑀁 𑀤𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀢𑀺, 𑀲𑁄 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀟𑀬𑁆𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀅𑀗𑁆𑀕𑀧𑀘𑁆𑀘𑀗𑁆𑀕𑁂𑀳𑀺 𑀪𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀦 𑀲𑀭𑀻𑀭𑁂𑀦 𑀓𑀼𑀡𑀧𑀕𑀦𑁆𑀥𑀁 𑀯𑀸𑀬𑀦𑁆𑀢𑁄 𑀯𑀺𑀲𑁂𑀦𑁂𑀯 𑀫𑀭𑀢𑀺 𑀤𑀳𑀭𑁄𑀧𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄𑀧𑀺 𑀫𑀳𑀮𑁆𑀮𑀓𑁄𑀧𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀬𑀺𑀓𑀫𑀭𑀡𑀁𑁇

1304. 𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀧𑀭𑁂 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀫𑀸𑀭𑁂𑀢𑀺, 𑀲𑁄 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀗𑁆𑀕𑀸𑀭𑀧𑀩𑁆𑀩𑀢𑁂𑀦𑁂𑀯 𑀅𑀗𑁆𑀕𑀸𑀭𑀧𑀩𑁆𑀩𑀢𑀁 𑀬𑀫𑀯𑀺𑀲𑀬𑁂𑀦𑁂𑀯 𑀬𑀫𑀯𑀺𑀲𑀬𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀤𑀟𑁆𑀠𑀯𑀺𑀤𑀟𑁆𑀠𑀕𑀢𑁆𑀢𑁄 𑀅𑀕𑁆𑀕𑀺𑀦𑀸𑀬𑁂𑀯 𑀫𑀭𑀢𑀺 𑀤𑀳𑀭𑁄𑀧𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄𑀧𑀺 𑀫𑀳𑀮𑁆𑀮𑀓𑁄𑀧𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀬𑀺𑀓𑀫𑀭𑀡𑀁𑁇

1305. 𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀧𑀭𑁂 𑀉𑀤𑀓𑁂𑀦 𑀫𑀸𑀭𑁂𑀢𑀺, 𑀲𑁄 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀳𑀢𑀯𑀺𑀮𑀼𑀢𑁆𑀢𑀪𑀕𑁆𑀕𑀤𑀼𑀩𑁆𑀩𑀮𑀕𑀢𑁆𑀢𑁄 𑀔𑀼𑀩𑁆𑀪𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀉𑀤𑀓𑁂𑀦𑁂𑀯 𑀫𑀭𑀢𑀺 𑀤𑀳𑀭𑁄𑀧𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄𑀧𑀺 𑀫𑀳𑀮𑁆𑀮𑀓𑁄𑀧𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀬𑀺𑀓𑀫𑀭𑀡𑀁𑁇

1306. 𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀧𑀭𑁂 𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀸𑀭𑁂𑀢𑀺, 𑀲𑁄 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀺𑀦𑁆𑀦𑀪𑀺𑀦𑁆𑀦𑀓𑁄𑀝𑁆𑀝𑀺𑀢𑀯𑀺𑀓𑁄𑀝𑁆𑀝𑀺𑀢𑁄 𑀲𑀢𑁆𑀢𑀺𑀫𑀼𑀔𑀲𑀫𑀸𑀳𑀢𑁄 𑀲𑀢𑁆𑀢𑀺𑀬𑀸𑀬𑁂𑀯 𑀫𑀭𑀢𑀺 𑀤𑀳𑀭𑁄𑀧𑀺 𑀫𑀚𑁆𑀛𑀺𑀫𑁄𑀧𑀺 𑀫𑀳𑀮𑁆𑀮𑀓𑁄𑀧𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀬𑀺𑀓𑀫𑀭𑀡𑀁”𑁇

1307. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀁 𑀅𑀢𑀺𑀤𑀺𑀲𑀸𑀢𑀺”𑁇 “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀆𑀤𑀺𑀦𑁆𑀦𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀪𑀓𑁆𑀔𑁄 𑀉𑀧𑀸𑀤𑀸𑀦𑀲𑀗𑁆𑀔𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺, 𑀲𑁄 𑀅𑀕𑁆𑀕𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀅𑀦𑀻𑀢𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀤𑁆𑀤𑀯𑁄 𑀲𑀫𑀬𑁂 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀦𑀸𑀫𑀸’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀩𑀳𑀽𑀦𑀺 𑀤𑀺𑀯𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀚𑀻𑀯𑀺𑀢𑁆𑀯𑀸 𑀚𑀭𑀸𑀚𑀺𑀡𑁆𑀡𑁄 𑀆𑀬𑀼𑀓𑁆𑀔𑀬𑀸 𑀅𑀦𑀻𑀢𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀤𑁆𑀤𑀯𑁄 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀲𑀫𑀬𑁂 𑀫𑀭𑀡𑀫𑀼𑀧𑀕𑀢𑁄’𑀢𑀺𑁇

1308. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀆𑀤𑀺𑀦𑁆𑀦𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑁄 𑀅𑀲𑁆𑀲, 𑀢𑀁 𑀅𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑁂𑀬𑁂𑀯 𑀢𑀺𑀡𑀓𑀝𑁆𑀞𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑁂 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀪𑀺𑀧𑁆𑀧𑀯𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀬𑁆𑀬, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀫𑀬𑁂 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀦𑀸𑀫 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀧𑀼𑀭𑀺𑀫𑀓𑁂𑀦 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀲𑀫𑀕𑀢𑀺𑀓𑁄 𑀦𑀸𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂𑀖𑁂𑀦 𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀲𑁄 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀲𑀫𑀬𑁂 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀭𑁄𑀕𑁂𑀦 𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀯𑀸𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀧𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑁂𑀫𑁆𑀳𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀺𑀓𑁂𑀦 𑀯𑀸 𑀉𑀢𑀼𑀧𑀭𑀺𑀡𑀸𑀫𑀚𑁂𑀦 𑀯𑀸 𑀯𑀺𑀲𑀫𑀧𑀭𑀺𑀳𑀸𑀭𑀚𑁂𑀦 𑀯𑀸 𑀑𑀧𑀓𑁆𑀓𑀫𑀺𑀓𑁂𑀦 𑀯𑀸 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀬 𑀯𑀸 𑀧𑀺𑀧𑀸𑀲𑀸𑀬 𑀯𑀸 𑀲𑀧𑁆𑀧𑀤𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀯𑀺𑀲𑀫𑀸𑀲𑀺𑀢𑁂𑀦 𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀯𑀸 𑀉𑀤𑀓𑁂𑀦 𑀯𑀸 𑀲𑀢𑁆𑀢𑀺𑀯𑁂𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀯𑀸 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺𑁇 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀺𑁇

1309. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀕𑀦𑁂 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀯𑀮𑀸𑀳𑀓𑁄 𑀉𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀦𑁆𑀦𑀜𑁆𑀘 𑀣𑀮𑀜𑁆𑀘 𑀧𑀭𑀺𑀧𑀽𑀭𑀬𑀦𑁆𑀢𑁄 𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀫𑁂𑀖𑁄 𑀅𑀦𑀻𑀢𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀤𑁆𑀤𑀯𑁄 𑀯𑀲𑁆𑀲𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑁆𑀯𑀸 𑀚𑀭𑀸𑀚𑀺𑀡𑁆𑀡𑁄 𑀆𑀬𑀼𑀓𑁆𑀔𑀬𑀸 𑀅𑀦𑀻𑀢𑀺𑀓𑁄 𑀅𑀦𑀼𑀧𑀤𑁆𑀤𑀯𑁄 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 ‘𑀲𑀫𑀬𑁂 𑀫𑀭𑀡𑀫𑀼𑀧𑀕𑀢𑁄’𑀢𑀺𑁇

1319. 𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀁 𑀧𑀦 𑀢𑀬𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀢𑀭𑀼𑀡𑀲𑀲𑁆𑀲𑀁 𑀓𑀺𑀫𑀬𑁄 𑀉𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀽𑀮𑀁 𑀦𑀸𑀲𑁂𑀦𑁆𑀢𑀻’”𑀢𑀺? “𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑁂𑀯 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀲𑁆𑀲𑀁 𑀓𑀸𑀮𑁂 𑀦𑀝𑁆𑀞𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀓𑀸𑀮𑁂 𑀦𑀝𑁆𑀞𑀦𑁆”𑀢𑀺? “𑀅𑀓𑀸𑀮𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀤𑀺 𑀔𑁄 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀲𑁆𑀲𑀁 𑀓𑀺𑀫𑀬𑁄 𑀦 𑀔𑀸𑀤𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀲𑁆𑀲𑀼𑀤𑁆𑀥𑀭𑀡𑀲𑀫𑀬𑀁 𑀧𑀸𑀧𑀼𑀡𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑁂𑀦 𑀲𑀲𑁆𑀲𑀁 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀭𑀼𑀧𑀖𑀸𑀢𑀁 𑀲𑀲𑁆𑀲𑀁 𑀲𑀲𑁆𑀲𑀼𑀤𑁆𑀥𑀭𑀡𑀲𑀫𑀬𑀁 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀭𑁄𑀕𑁂𑀦 𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀯𑀸𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀺𑀯𑁂𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀯𑀸 𑀫𑀭𑀢𑀺𑁇 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀻𑀢𑀺𑁇

1320. 𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀁 𑀧𑀦 𑀢𑀬𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀲𑀲𑁆𑀲𑁂 𑀨𑀮𑀪𑀸𑀭𑀦𑀫𑀺𑀢𑁂 𑀫𑀜𑁆𑀚𑀭𑀺𑀢𑀧𑀢𑁆𑀢𑁂 𑀓𑀭𑀓𑀯𑀲𑁆𑀲𑀁 𑀦𑀸𑀫 𑀯𑀲𑁆𑀲𑀚𑀸𑀢𑀺 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀸𑀲𑁂𑀢𑀺 𑀅𑀨𑀮𑀁 𑀓𑀭𑁄𑀢𑀻’”𑀢𑀺? “𑀲𑀼𑀢𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑁂𑀯 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 “𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀲𑁆𑀲𑀁 𑀓𑀸𑀮𑁂 𑀦𑀝𑁆𑀞𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀓𑀸𑀮𑁂 𑀦𑀝𑁆𑀞𑀦𑁆”𑀢𑀺? “𑀅𑀓𑀸𑀮𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀤𑀺 𑀔𑁄 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀲𑁆𑀲𑀁 𑀓𑀭𑀓𑀯𑀲𑁆𑀲𑀁 𑀦 𑀯𑀲𑁆𑀲𑁂𑀬𑁆𑀬 𑀲𑀲𑁆𑀲𑀼𑀤𑁆𑀥𑀭𑀡𑀲𑀫𑀬𑀁 𑀧𑀸𑀧𑀼𑀡𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑁂𑀦 𑀲𑀲𑁆𑀲𑀁 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀭𑀼𑀧𑀖𑀸𑀢𑀁 𑀲𑀲𑁆𑀲𑀁 𑀲𑀲𑁆𑀲𑀼𑀤𑁆𑀥𑀭𑀡𑀲𑀫𑀬𑀁 𑀧𑀸𑀧𑀼𑀡𑀸𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺, 𑀲𑁄 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀭𑁄𑀕𑁂𑀦 𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀯𑀸𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀧𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑁂𑀫𑁆𑀳𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀯𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀺𑀓𑁂𑀦 𑀯𑀸 𑀉𑀢𑀼𑀧𑀭𑀺𑀡𑀸𑀫𑀚𑁂𑀦 𑀯𑀸 𑀯𑀺𑀲𑀫𑀧𑀭𑀺𑀳𑀸𑀭𑀚𑁂𑀦 𑀯𑀸 𑀑𑀧𑀓𑁆𑀓𑀫𑀺𑀓𑁂𑀦 𑀯𑀸 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀬 𑀯𑀸 𑀧𑀺𑀧𑀸𑀲𑀸𑀬 𑀯𑀸 𑀲𑀧𑁆𑀧𑀤𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀯𑀺𑀲𑀫𑀸𑀲𑀺𑀢𑁂𑀦 𑀯𑀸 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀯𑀸 𑀉𑀤𑀓𑁂𑀦 𑀯𑀸 𑀲𑀢𑁆𑀢𑀺𑀯𑁂𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀯𑀸 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀢𑀺𑁇 𑀬𑀤𑀺 𑀧𑀦 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀭𑁄𑀕𑁂𑀦 𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑁄 𑀦 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀲𑀫𑀬𑁂𑀯 𑀫𑀭𑀡𑀁 𑀧𑀸𑀧𑀼𑀡𑁂𑀬𑁆𑀬𑁇 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀻”𑀢𑀺𑁇

1321. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀼𑀤𑀲𑁆𑀲𑀺𑀢𑀁 𑀓𑀸𑀭𑀡𑀁, 𑀲𑀼𑀤𑀲𑁆𑀲𑀺𑀢𑀁 𑀑𑀧𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀲𑁆𑀲 𑀧𑀭𑀺𑀤𑀻𑀧𑀦𑀸𑀬, ‘𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀦𑁆’𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀢𑀁 𑀧𑀸𑀓𑀝𑀁 𑀓𑀢𑀁 𑀯𑀺𑀪𑀽𑀢𑀁 𑀓𑀢𑀁, 𑀅𑀘𑀺𑀢𑁆𑀢𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀫𑀦𑀼𑀚𑁄 𑀏𑀓𑀫𑁂𑀓𑁂𑀦𑀧𑀺 𑀢𑀸𑀯 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀦𑀺𑀝𑁆𑀞𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀦𑁆’𑀢𑀺, 𑀓𑀺𑀁 𑀧𑀦 𑀫𑀦𑀼𑀚𑁄 𑀲𑀘𑁂𑀢𑀦𑁄? 𑀧𑀞𑀫𑁄𑀧𑀫𑁆𑀫𑁂𑀦𑁂𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀅𑀓𑀸𑀮𑁂 𑀫𑀭𑀡𑀦𑁆’𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀅𑀧𑀭𑀸𑀧𑀭𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀳𑀦𑀁 𑀲𑁄𑀢𑀼𑀓𑀸𑀫𑁄 𑀦 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺𑀦𑁆”𑀢𑀺𑁇
