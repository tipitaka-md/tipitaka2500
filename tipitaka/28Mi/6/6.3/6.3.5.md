# 𑁬𑁇𑁩𑁇𑁫 𑀲𑀼𑀧𑀺𑀦𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀧𑀺𑀦𑀧𑀜𑁆𑀳

(194.)

1284\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀦𑀭𑀦𑀸𑀭𑀺𑀬𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑁆𑀧𑀺 𑀧𑀸𑀧𑀓𑀫𑁆𑀧𑀺, 𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀅𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀓𑀢𑀧𑀼𑀩𑁆𑀩𑀫𑁆𑀧𑀺 𑀅𑀓𑀢𑀧𑀼𑀩𑁆𑀩𑀫𑁆𑀧𑀺, 𑀔𑁂𑀫𑀫𑁆𑀧𑀺 𑀲𑀪𑀬𑀫𑁆𑀧𑀺, 𑀤𑀽𑀭𑁂𑀧𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑀧𑀺, 𑀩𑀳𑀼𑀯𑀺𑀥𑀸𑀦𑀺𑀧𑀺 𑀅𑀦𑁂𑀓𑀯𑀡𑁆𑀡𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀦𑀸𑀫, 𑀓𑁄 𑀘𑁂𑀢𑀁 𑀧𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀫𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀧𑀺𑀦𑀁 𑀦𑀸𑀫, 𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀸𑀢𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀙𑀬𑀺𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑀸𑀢𑀺𑀓𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀧𑀺𑀢𑁆𑀢𑀺𑀓𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑁂𑀫𑁆𑀳𑀺𑀓𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑁂𑀯𑀢𑀽𑀧𑀲𑀁𑀳𑀸𑀭𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑀫𑀼𑀤𑀸𑀘𑀺𑀡𑁆𑀡𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀢𑀢𑁆𑀭, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀁 𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀢𑀁𑀬𑁂𑀯 𑀲𑀘𑁆𑀘𑀁, 𑀅𑀯𑀲𑁂𑀲𑀁 𑀫𑀺𑀘𑁆𑀙𑀸”𑀢𑀺𑁇

1285\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀓𑀺𑀁 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀘𑀺𑀦𑀸𑀢𑀺, 𑀢𑀁 𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀸𑀢𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀜𑁆𑀜𑁄 𑀯𑀸 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀢𑀻”𑀢𑀺? “𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀘𑀺𑀦𑀸𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁄 𑀓𑁄𑀘𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀢𑀁𑀬𑁂𑀯 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀸𑀢𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀤𑀸𑀲𑁄 𑀦 𑀲𑀬𑀁 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀙𑀸𑀬𑀁 𑀯𑀺𑀘𑀺𑀦𑀸𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁄 𑀓𑁄𑀘𑀺 𑀙𑀸𑀬𑀁 𑀆𑀦𑁂𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀲𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀬𑀢𑁄 𑀓𑀼𑀢𑁄𑀘𑀺 𑀙𑀸𑀬𑀸 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀤𑀸𑀲𑀲𑁆𑀲 𑀆𑀧𑀸𑀢𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀘𑀺𑀦𑀸𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁄 𑀓𑁄𑀘𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀅𑀣 𑀔𑁄 𑀬𑀢𑁄 𑀓𑀼𑀢𑁄𑀘𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀸𑀢𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇

1286\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 ‘𑀏𑀯𑀁 𑀦𑀸𑀫 𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁂𑀫𑀁 𑀯𑀸 𑀪𑀬𑀁 𑀯𑀸’𑀢𑀺? 𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 ‘𑀏𑀯𑀁𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁂𑀫𑀁 𑀯𑀸 𑀪𑀬𑀁 𑀯𑀸’𑀢𑀺, 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀧𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀢𑀢𑁄 𑀢𑁂 𑀅𑀢𑁆𑀣𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1287\. “𑀇𑀗𑁆𑀖, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀸𑀭𑀡𑀁 𑀫𑁂 𑀤𑀲𑁆𑀲𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀭𑀻𑀭𑁂 𑀢𑀺𑀮𑀓𑀸 𑀧𑀻𑀍𑀅𑀓𑀸 𑀤𑀤𑁆𑀤𑀽𑀦𑀺 𑀉𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀺 𑀮𑀸𑀪𑀸𑀬 𑀯𑀸 𑀅𑀮𑀸𑀪𑀸𑀬 𑀯𑀸, 𑀬𑀲𑀸𑀬 𑀯𑀸 𑀅𑀬𑀲𑀸𑀬 𑀯𑀸, 𑀦𑀺𑀦𑁆𑀤𑀸𑀬 𑀯𑀸 𑀧𑀲𑀁𑀲𑀸𑀬 𑀯𑀸, 𑀲𑀼𑀔𑀸𑀬 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑀸, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀻𑀍𑀅𑀓𑀸 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 ‘𑀇𑀫𑀁 𑀦𑀸𑀫 𑀫𑀬𑀁 𑀅𑀢𑁆𑀣𑀁 𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀸’”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀤𑀺𑀲𑁂 𑀢𑀸 𑀑𑀓𑀸𑀲𑁂 𑀧𑀻𑀍𑀅𑀓𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀢𑀸 𑀧𑀻𑀍𑀅𑀓𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑁂𑀫𑀺𑀢𑁆𑀢𑀓𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀺 ‘𑀏𑀯𑀁 𑀦𑀸𑀫 𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑀁 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 ‘𑀏𑀯𑀁 𑀦𑀸𑀫 𑀯𑀺𑀧𑀸𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑁂𑀫𑀁 𑀯𑀸 𑀪𑀬𑀁 𑀯𑀸’𑀢𑀺, 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀧𑀦 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀓𑀣𑁂𑀢𑀺, 𑀢𑀢𑁄 𑀢𑁂 𑀅𑀢𑁆𑀣𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1288\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄 𑀦𑀺𑀤𑁆𑀤𑀸𑀬𑀦𑁆𑀢𑁄, 𑀉𑀤𑀸𑀳𑀼 𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀬𑁄 𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀲𑁄 𑀦𑀺𑀤𑁆𑀤𑀸𑀬𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀫𑀺𑀤𑁆𑀥𑁂 𑀅𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂 𑀪𑀯𑀗𑁆𑀕𑁂 𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑀸𑀭𑀽𑀍𑀳𑀲𑁆𑀲, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀯𑀗𑁆𑀕𑀕𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀪𑀯𑀗𑁆𑀕𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀼𑀧𑀺𑀦𑁄 𑀦 𑀳𑁄𑀢𑀺, 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇

1289\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀺𑀫𑀺𑀭𑁂 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀅𑀧𑁆𑀧𑀪𑀸𑀲𑁂 𑀲𑀼𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀧𑀺 𑀆𑀤𑀸𑀲𑁂 𑀙𑀸𑀬𑀸 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑀸𑀭𑀽𑀍𑀳𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀪𑀯𑀗𑁆𑀕𑀕𑀢𑁂 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁂𑀧𑀺 𑀲𑀭𑀻𑀭𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀼𑀧𑀺𑀦𑀁 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀤𑀸𑀲𑁄, 𑀏𑀯𑀁 𑀲𑀭𑀻𑀭𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁; 𑀬𑀣𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁄, 𑀏𑀯𑀁 𑀫𑀺𑀤𑁆𑀥𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁; 𑀬𑀣𑀸 𑀆𑀮𑁄𑀓𑁄, 𑀏𑀯𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇

1290\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀺𑀓𑁄𑀢𑁆𑀣𑀝𑀲𑁆𑀲 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀪𑀸 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀸𑀬𑁂𑀯 𑀲𑀽𑀭𑀺𑀬𑀭𑀲𑁆𑀫𑀺 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀸𑀬 𑀲𑀽𑀭𑀺𑀬𑀭𑀲𑁆𑀫𑀺𑀬𑀸 𑀆𑀮𑁄𑀓𑁄 𑀦 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑀸𑀭𑀽𑀍𑀳𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀯𑀗𑁆𑀕𑀕𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀪𑀯𑀗𑁆𑀕𑀕𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀲𑀼𑀧𑀺𑀦𑀁 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀽𑀭𑀺𑀬𑁄, 𑀏𑀯𑀁 𑀲𑀭𑀻𑀭𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁; 𑀬𑀣𑀸 𑀫𑀳𑀺𑀓𑁄𑀢𑁆𑀣𑀭𑀡𑀁, 𑀏𑀯𑀁 𑀫𑀺𑀤𑁆𑀥𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁; 𑀬𑀣𑀸 𑀲𑀽𑀭𑀺𑀬𑀭𑀲𑁆𑀫𑀺, 𑀏𑀯𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇

1291\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀦𑁆𑀢𑁂𑀧𑀺 𑀲𑀭𑀻𑀭𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑀸𑀭𑀽𑀍𑀳𑀲𑁆𑀲 𑀪𑀯𑀗𑁆𑀕𑀕𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑁂𑀧𑀺 𑀲𑀭𑀻𑀭𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑁂𑀧𑀺 𑀲𑀭𑀻𑀭𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑀲𑁆𑀲, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀮𑁄𑀮𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀝𑀁 𑀧𑀸𑀓𑀝𑀁 𑀅𑀦𑀺𑀩𑀤𑁆𑀥𑀁, 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀸𑀢𑀁 𑀦 𑀉𑀧𑁂𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑀺𑀯𑀝𑀁 𑀧𑀸𑀓𑀝𑀁 𑀅𑀓𑀺𑀭𑀺𑀬𑀁 𑀅𑀭𑀳𑀲𑁆𑀲𑀁 𑀭𑀳𑀲𑁆𑀲𑀓𑀸𑀫𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀺𑀩𑁆𑀩𑁄 𑀅𑀢𑁆𑀣𑁄 𑀆𑀧𑀸𑀢𑀁 𑀦 𑀉𑀧𑁂𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀦 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀺𑀦𑁆𑀦𑀸𑀚𑀻𑀯𑀁 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀁 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀁 𑀓𑀼𑀲𑀻𑀢𑀁 𑀳𑀻𑀦𑀯𑀻𑀭𑀺𑀬𑀁 𑀓𑀼𑀲𑀮𑀸 𑀩𑁄𑀥𑀺𑀧𑀓𑁆𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀆𑀧𑀸𑀢𑀁 𑀦 𑀉𑀧𑁂𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀺𑀩𑁆𑀩𑁄 𑀅𑀢𑁆𑀣𑁄 𑀆𑀧𑀸𑀢𑀁 𑀦 𑀉𑀧𑁂𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀚𑀸𑀕𑀭𑀦𑁆𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀦 𑀧𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1292\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀤𑁆𑀥𑀲𑁆𑀲 𑀆𑀤𑀺𑀫𑀚𑁆𑀛𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀺𑀤𑁆𑀥𑀲𑁆𑀲 𑀆𑀤𑀺𑀫𑀚𑁆𑀛𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀓𑀢𑀫𑀁 𑀆𑀤𑀺, 𑀓𑀢𑀫𑀁 𑀫𑀚𑁆𑀛𑀁, 𑀓𑀢𑀫𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀦𑁆”𑀢𑀺? “𑀬𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀑𑀦𑀸𑀳𑁄 𑀧𑀭𑀺𑀬𑁄𑀦𑀸𑀳𑁄 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀫𑀦𑁆𑀤𑀢𑀸 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑀢𑀸 𑀓𑀸𑀬𑀲𑁆𑀲, 𑀅𑀬𑀁 𑀫𑀺𑀤𑁆𑀥𑀲𑁆𑀲 𑀆𑀤𑀺; 𑀬𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀧𑀺𑀦𑀺𑀤𑁆𑀤𑀸𑀧𑀭𑁂𑀢𑁄 𑀯𑁄𑀓𑀺𑀡𑁆𑀡𑀓𑀁 𑀚𑀕𑁆𑀕𑀢𑀺, 𑀇𑀤𑀁 𑀫𑀺𑀤𑁆𑀥𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑀁; 𑀪𑀯𑀗𑁆𑀕𑀕𑀢𑀺 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁𑁇 𑀫𑀚𑁆𑀛𑀽𑀧𑀕𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀧𑀺𑀦𑀺𑀤𑁆𑀤𑀸𑀧𑀭𑁂𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺 𑀬𑀢𑀘𑀸𑀭𑀻 𑀲𑀫𑀸𑀳𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀞𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀅𑀘𑀮𑀩𑀼𑀤𑁆𑀥𑀺 𑀧𑀳𑀻𑀦𑀓𑁄𑀢𑀽𑀳𑀮𑀲𑀤𑁆𑀤𑀁 𑀯𑀦𑀫𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀔𑀼𑀫𑀁 𑀅𑀢𑁆𑀣𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀢𑀺, 𑀦 𑀘 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀫𑀺𑀤𑁆𑀥𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀔𑀼𑀫𑀁 𑀅𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀸𑀕𑀭𑁄 𑀦 𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄, 𑀫𑀚𑁆𑀛𑀽𑀧𑀕𑀢𑁄 𑀓𑀧𑀺𑀦𑀺𑀤𑁆𑀤𑀸𑀧𑀭𑁂𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀢𑀽𑀳𑀮𑀲𑀤𑁆𑀤𑁄, 𑀏𑀯𑀁 𑀚𑀸𑀕𑀭𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁; 𑀬𑀣𑀸 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀯𑀦𑀁, 𑀏𑀯𑀁 𑀓𑀧𑀺𑀦𑀺𑀤𑁆𑀤𑀸𑀧𑀭𑁂𑀢𑁄 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑁄; 𑀬𑀣𑀸 𑀲𑁄 𑀓𑁄𑀢𑀽𑀳𑀮𑀲𑀤𑁆𑀤𑀁 𑀑𑀳𑀸𑀬 𑀫𑀺𑀤𑁆𑀥𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑀢𑁆𑀢𑀪𑀽𑀢𑁄 𑀲𑀼𑀔𑀼𑀫𑀁 𑀅𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀢𑀺, 𑀏𑀯𑀁 𑀚𑀸𑀕𑀭𑁄 𑀦 𑀫𑀺𑀤𑁆𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀓𑀧𑀺𑀦𑀺𑀤𑁆𑀤𑀸𑀧𑀭𑁂𑀢𑁄 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
