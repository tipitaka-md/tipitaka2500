# 𑁬𑁇𑁩𑁇𑁧𑁦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀭𑀽𑀧𑀲𑀡𑁆𑀞𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

### 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀭𑀽𑀧𑀲𑀡𑁆𑀞𑀸𑀦𑀧𑀜𑁆𑀳

(199.)

1342\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀲𑀓𑁆𑀓𑀸 𑀧𑀦 𑀢𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀯𑀸 𑀯𑀬𑀁 𑀯𑀸 𑀧𑀫𑀸𑀡𑀁 𑀯𑀸 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀓𑀸𑀭𑀡𑁂𑀦 𑀯𑀸 𑀳𑁂𑀢𑀼𑀦𑀸 𑀯𑀸 𑀦𑀬𑁂𑀦 𑀯𑀸 𑀉𑀧𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀅𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀯𑀸 𑀯𑀬𑀁 𑀯𑀸 𑀧𑀫𑀸𑀡𑀁 𑀯𑀸 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀓𑀸𑀭𑀡𑁂𑀦 𑀯𑀸 𑀳𑁂𑀢𑀼𑀦𑀸 𑀯𑀸 𑀦𑀬𑁂𑀦 𑀯𑀸 𑀉𑀧𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀢𑀫𑁆𑀧𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀺, 𑀬𑀁 𑀅𑀢𑁆𑀣𑀺𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀯𑀸 𑀯𑀬𑀁 𑀯𑀸 𑀧𑀫𑀸𑀡𑀁 𑀯𑀸 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀓𑀸𑀭𑀡𑁂𑀦 𑀯𑀸 𑀳𑁂𑀢𑀼𑀦𑀸 𑀯𑀸 𑀦𑀬𑁂𑀦 𑀯𑀸 𑀅𑀧𑀜𑁆𑀜𑀸𑀧𑀦𑀁, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀫𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀳𑁄𑀢𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀦𑀸𑀫𑀸”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑁂𑀲𑁄 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄”𑀢𑀺𑁇 “𑀲𑀘𑁂 𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑁄𑀘𑀺 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀓𑀺𑀢𑁆𑀢𑀓𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁, 𑀓𑀢𑀺 𑀧𑀦 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸, 𑀬𑁂 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀁 𑀧𑀼𑀝𑁆𑀞𑁄 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁂 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀘𑀺 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬 ‘𑀓𑀺𑀢𑁆𑀢𑀓𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁, 𑀓𑀢𑀺 𑀧𑀦 𑀢𑁂 𑀲𑀢𑁆𑀢𑀸, 𑀬𑁂 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀢𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀁 ‘𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀁 𑀢𑁆𑀯𑀁, 𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀧𑀼𑀘𑁆𑀙𑀲𑀺, 𑀦𑁂𑀲𑀸 𑀧𑀼𑀘𑁆𑀙𑀸 𑀓𑁂𑀦𑀘𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸, 𑀞𑀧𑀦𑀻𑀬𑁄 𑀏𑀲𑁄 𑀧𑀜𑁆𑀳𑁄𑁇 𑀅𑀯𑀺𑀪𑀢𑁆𑀢𑁄 𑀮𑁄𑀓𑀓𑁆𑀔𑀸𑀬𑀺𑀓𑁂𑀳𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄, 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀫𑀺𑀦𑀺𑀢𑀼𑀁 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸 𑀬𑁂 𑀢𑀢𑁆𑀣 𑀯𑀸𑀲𑀫𑀼𑀧𑀕𑀢𑀸𑀢𑀺 𑀏𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀯𑀘𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑀦𑁆’”𑀢𑀺𑁇

1343\. “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺𑀥𑀫𑁆𑀫𑁂 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀏𑀯𑀁 𑀧𑀝𑀺𑀯𑀘𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀦𑀦𑀼 𑀯𑀺𑀕𑀡𑁂𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀏𑀢𑁆𑀢𑀓𑀁 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁, 𑀏𑀢𑁆𑀢𑀓𑀸 𑀘 𑀲𑀢𑁆𑀢𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀻’”𑀢𑀺? “𑀦 𑀲𑀓𑁆𑀓𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀯𑀺𑀲𑀬𑁄 𑀏𑀲𑁄 𑀧𑀜𑁆𑀳𑁄”𑀢𑀺𑁇

1344\. “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺𑀥𑀫𑁆𑀫𑁂𑀬𑁂𑀯 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀕𑀡𑁂𑀢𑀼𑀁 𑀲𑀢𑁆𑀢𑀸 𑀯𑀸 𑀬𑁂 𑀢𑀢𑁆𑀣 𑀯𑀸𑀲𑀫𑀼𑀧𑀕𑀢𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑁂𑀯 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀯𑀸 𑀯𑀬𑀁 𑀯𑀸 𑀧𑀫𑀸𑀡𑀁 𑀯𑀸 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀓𑀸𑀭𑀡𑁂𑀦 𑀯𑀸 𑀳𑁂𑀢𑀼𑀦𑀸 𑀯𑀸 𑀦𑀬𑁂𑀦 𑀯𑀸 𑀉𑀧𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑀼𑀁, 𑀯𑀺𑀕𑀡𑁂𑀬𑁆𑀬, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀘𑁂𑀢𑁄𑀯𑀲𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀉𑀤𑀓𑀁 𑀢𑀢𑁆𑀭𑀸𑀲𑀬𑁂 𑀘 𑀲𑀢𑁆𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀲𑁄 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀘𑁂𑀢𑁄𑀯𑀲𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀓𑀼𑀡𑁂𑀬𑁆𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀁 𑀯𑀸 𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀯𑀸 𑀯𑀬𑀁 𑀯𑀸 𑀧𑀫𑀸𑀡𑀁 𑀯𑀸 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀓𑀸𑀭𑀡𑁂𑀦 𑀯𑀸 𑀳𑁂𑀢𑀼𑀦𑀸 𑀯𑀸 𑀦𑀬𑁂𑀦 𑀯𑀸 𑀉𑀧𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑀼𑀁𑁇

1357\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, ‘𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀕𑀼𑀡𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑀸’𑀢𑀺 𑀬𑀁 𑀯𑀤𑁂𑀲𑀺, 𑀓𑀢𑀫𑁂 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀕𑀼𑀡𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑀸”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀁 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑀁𑁇 𑀅𑀬𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀧𑀞𑀫𑁄 𑀕𑀼𑀡𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀁 𑀅𑀘𑀮𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀘𑀮𑀁𑁇 𑀅𑀬𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀤𑀼𑀢𑀺𑀬𑁄 𑀕𑀼𑀡𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀁 𑀤𑀼𑀭𑀥𑀺𑀭𑁄𑀳𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀤𑀼𑀭𑀥𑀺𑀭𑁄𑀳𑀁 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁𑁇 𑀅𑀬𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀢𑀢𑀺𑀬𑁄 𑀕𑀼𑀡𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀁 𑀲𑀩𑁆𑀩𑀩𑀻𑀚𑀸𑀦𑀁 𑀅𑀯𑀺𑀭𑀽𑀳𑀦𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑀸𑀦𑀁 𑀅𑀯𑀺𑀭𑀽𑀳𑀦𑀁𑁇 𑀅𑀬𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀕𑀼𑀡𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀁 𑀅𑀦𑀼𑀦𑀬𑀧𑁆𑀧𑀝𑀺𑀖𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀦𑀬𑀧𑁆𑀧𑀝𑀺𑀖𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀁𑁇 𑀅𑀬𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀫𑁄 𑀕𑀼𑀡𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀺𑀭𑀺𑀲𑀺𑀔𑀭𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀕𑀼𑀡𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀯𑀺𑀝𑁆𑀞𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
