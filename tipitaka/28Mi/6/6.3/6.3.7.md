# 𑁬𑁇𑁩𑁇𑁭 𑀘𑁂𑀢𑀺𑀬𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀘𑁂𑀢𑀺𑀬𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀧𑀜𑁆𑀳

(196.)

1323\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁𑀬𑁂𑀯 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀳𑁄𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀢𑀫𑁂𑀲𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁄𑀢𑀺, 𑀓𑀢𑀫𑁂𑀲𑀁 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁? 𑀇𑀥, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁄𑀯 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺 ‘𑀏𑀯𑀁𑀦𑀸𑀫 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀽’𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀲𑁂𑀦 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀁 𑀅𑀭𑀳𑀢𑁄 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀲𑁂𑀦 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺𑁇

1324\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑁂𑀯𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 ‘𑀇𑀫𑀺𑀦𑀸 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑁂𑀦 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀘𑁆𑀘𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘 𑀧𑀲𑀦𑁆𑀦𑀸 𑀓𑀼𑀲𑀮𑁂𑀦 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀏𑀯𑀁 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀲𑁂𑀦 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺𑁇

1325\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀢𑁆𑀣𑀻 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀯𑀸 𑀲𑀤𑁆𑀥𑁄 𑀧𑀲𑀦𑁆𑀦𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀩𑀼𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀬𑁄𑀦𑀺𑀲𑁄 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀕𑀦𑁆𑀥𑀁 𑀯𑀸 𑀫𑀸𑀮𑀁 𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀁 𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀘𑁂𑀢𑀺𑀬𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀢𑀺 ‘𑀏𑀯𑀁𑀦𑀸𑀫 𑀳𑁄𑀢𑀽’𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀲𑁂𑀦 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀲𑁂𑀦 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺𑁇

1326\. 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀲𑁂𑀦 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀳𑁄𑀢𑀺𑁇

1327\. 𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀲𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲𑀧𑀺 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀲𑁆𑀲 𑀘𑁂𑀢𑁄𑀯𑀲𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀺𑀬𑁂 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀦 𑀳𑁄𑀢𑀺, 𑀅𑀲𑀢𑀺𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑁂 𑀘𑀭𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀼𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀑𑀓𑀧𑁆𑀧𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀝𑁆𑀞𑀁 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁 𑀲𑀤𑁆𑀤𑀳𑀺𑀢𑀩𑁆𑀩𑀁 ‘𑀲𑀼𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀅𑀬𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀢𑁆𑀢𑁄’”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

---
