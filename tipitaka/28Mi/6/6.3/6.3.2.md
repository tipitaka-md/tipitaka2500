# 𑁬𑁇𑁩𑁇𑁨 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀧𑀜𑁆𑀳

(191.)

1244\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀸 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀕𑁄𑀢𑀫𑁂𑀦𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸 𑀓𑀢𑀸”𑀢𑀺? “𑀦𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸, 𑀕𑁄𑀢𑀫𑁂𑀦𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸 𑀓𑀢𑀸”𑀢𑀺𑁇

1245\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀤𑀺 𑀏𑀯𑀁 𑀅𑀬𑀼𑀢𑁆𑀢𑀁, 𑀬𑀁 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀘𑀢𑀽𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀞𑀸𑀦𑁂𑀳𑀺 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀓𑀼𑀮𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀧𑀥𑀸𑀦𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀆𑀬𑀼𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀧𑀫𑀸𑀡𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀢𑀽𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀳𑀺 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀭𑀽𑀧𑁂 𑀲𑀻𑀮𑁂 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀺 𑀧𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑁂 𑀘𑀢𑀼𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑁂 𑀤𑀲𑀢𑀣𑀸𑀕𑀢𑀩𑀮𑁂 𑀙𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀜𑀸𑀡𑁂 𑀘𑀼𑀤𑁆𑀤𑀲𑀩𑀼𑀤𑁆𑀥𑀜𑀸𑀡𑁂 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑁂 𑀓𑁂𑀯𑀮𑁂 𑀘 𑀩𑀼𑀤𑁆𑀥𑀕𑀼𑀡𑁂 𑀦𑀢𑁆𑀣𑀺 𑀯𑁂𑀫𑀢𑁆𑀢𑀢𑀸, 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀲𑀫𑀸”𑀢𑀺𑁇

1246\. “𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀲𑀫𑀸, 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦𑁂𑀯 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸 𑀓𑀢𑀸”𑀢𑀺? “𑀅𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀜𑀸𑀡𑁂 𑀅𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀸𑀬 𑀩𑁄𑀥𑀺𑀬𑀸 𑀕𑁄𑀢𑀫𑁄 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀫𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀅𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀁 𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑀬𑀫𑀸𑀦𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸 𑀓𑀢𑀸”𑀢𑀺𑁇

1247\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀅𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂 𑀜𑀸𑀡𑁂 𑀅𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀸𑀬 𑀩𑁄𑀥𑀺𑀬𑀸 𑀫𑀳𑀸𑀪𑀺𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑀁 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄, 𑀦𑀦𑀼 𑀦𑀸𑀫 𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑁂 𑀜𑀸𑀡𑁂 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺?

1248\. “𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀺𑀧𑀭𑀻𑀢𑀁 𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀻 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀲𑁆𑀲 𑀅𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺, 𑀅𑀭𑀢𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀫𑀸𑀭𑀓𑀸𑀬𑀺𑀓𑁄 𑀤𑁂𑀯𑀧𑀼𑀢𑁆𑀢𑁄 ‘𑀅𑀬𑀁 𑀔𑁄 𑀓𑀸𑀮𑁄 𑀅𑀭𑀢𑀺𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀦𑁄𑀤𑀦𑀸𑀬𑀸’𑀢𑀺 𑀯𑁂𑀳𑀸𑀲𑁂 𑀞𑀢𑁆𑀯𑀸 𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺—

1249\. ‘𑀫𑀸𑀭𑀺𑀲, 𑀫𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀇𑀢𑁄 𑀢𑁂 𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀤𑀺𑀩𑁆𑀩𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀧𑀸𑀢𑀼𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀳𑀲𑁆𑀲𑀸𑀭𑀁 𑀲𑀦𑁂𑀫𑀺𑀓𑀁 𑀲𑀦𑀸𑀪𑀺𑀓𑀁 𑀲𑀩𑁆𑀩𑀸𑀓𑀸𑀭𑀧𑀭𑀺𑀧𑀽𑀭𑀁, 𑀧𑀣𑀯𑀺𑀕𑀢𑀸𑀦𑀺 𑀘 𑀢𑁂 𑀭𑀢𑀦𑀸𑀦𑀺 𑀆𑀓𑀸𑀲𑀝𑁆𑀞𑀸𑀦𑀺 𑀘 𑀲𑀬𑀫𑁂𑀯 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀤𑁆𑀯𑀺𑀲𑀳𑀲𑁆𑀲𑀧𑀭𑀺𑀢𑁆𑀢𑀤𑀻𑀧𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑀼 𑀘𑀢𑀽𑀲𑀼 𑀫𑀳𑀸𑀤𑀻𑀧𑁂𑀲𑀼 𑀏𑀓𑀫𑀼𑀔𑁂𑀦 𑀆𑀡𑀸 𑀧𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀜𑁆𑀘 𑀢𑁂 𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀽𑀭𑀸 𑀯𑀻𑀭𑀗𑁆𑀕𑀭𑀽𑀧𑀸 𑀧𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑀸, 𑀢𑁂𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀘𑀢𑀼𑀤𑁆𑀤𑀻𑀧𑀫𑀦𑀼𑀲𑀸𑀲𑀺𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇

1250\. 𑀬𑀣𑀸 𑀦𑀸𑀫 𑀤𑀺𑀯𑀲𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀁 𑀅𑀬𑁄𑀲𑀽𑀮𑀁 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀉𑀧𑀟𑀳𑀦𑁆𑀢𑀁 𑀓𑀡𑁆𑀡𑀲𑁄𑀢𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀓𑀡𑁆𑀡𑀲𑁄𑀢𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀣, 𑀇𑀢𑀺 𑀲𑁄 𑀧𑀓𑀢𑀺𑀬𑀸𑀯 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀤𑁂𑀯𑀢𑀸𑀬 𑀯𑀘𑀦𑁂𑀦 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀉𑀩𑁆𑀩𑀺𑀚𑁆𑀚𑀺 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀺 𑀲𑀁𑀯𑁂𑀕𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇

1251\. 𑀬𑀣𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀚𑀮𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑁂𑀦 𑀓𑀝𑁆𑀞𑁂𑀦 𑀉𑀧𑀟𑀳𑀺𑀢𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀚𑀮𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀧𑀓𑀢𑀺𑀬𑀸𑀯 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀤𑁂𑀯𑀢𑀸𑀬 𑀯𑀘𑀦𑁂𑀦 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀉𑀩𑁆𑀩𑀺𑀚𑁆𑀚𑀺 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀺 𑀲𑀁𑀯𑁂𑀕𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑁇

1252\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀧𑀣𑀯𑀻 𑀧𑀓𑀢𑀺𑀢𑀺𑀦𑁆𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀳𑀭𑀺𑀢𑀲𑀤𑁆𑀤𑀮𑀸 𑀆𑀲𑀺𑀢𑁆𑀢𑁄𑀤𑀓𑀸 𑀘𑀺𑀓𑁆𑀔𑀮𑁆𑀮𑀚𑀸𑀢𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀫𑀳𑀸𑀫𑁂𑀖𑁂 𑀅𑀪𑀺𑀯𑀼𑀝𑁆𑀞𑁂 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀘𑀺𑀓𑁆𑀔𑀮𑁆𑀮𑀢𑀭𑀸 𑀅𑀲𑁆𑀲; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀧𑀓𑀢𑀺𑀬𑀸𑀯 𑀉𑀓𑁆𑀓𑀡𑁆𑀞𑀺𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀤𑁂𑀯𑀢𑀸𑀬 𑀯𑀘𑀦𑁂𑀦 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀉𑀩𑁆𑀩𑀺𑀚𑁆𑀚𑀺 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀺 𑀲𑀁𑀯𑁂𑀕𑀫𑀸𑀧𑀚𑁆𑀚𑀻”𑀢𑀺𑁇

1253\. “𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀬𑀤𑀺 𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀤𑀺𑀩𑁆𑀩𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑀺𑀩𑁆𑀩𑁂 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑁂 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀺𑀩𑁆𑀩𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀅𑀧𑀺 𑀘 𑀧𑀮𑁄𑀪𑀦𑀢𑁆𑀣𑀸𑀬 𑀢𑀸𑀬 𑀤𑁂𑀯𑀢𑀸𑀬 𑀫𑀼𑀲𑀸 𑀪𑀡𑀺𑀢𑀁, 𑀬𑀤𑀺𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀤𑀺𑀩𑁆𑀩𑀁 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀦 𑀦𑀺𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑁇 𑀓𑀺𑀁𑀓𑀸𑀭𑀡𑀁? ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆’𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑀍𑀳𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺, ‘𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀦𑀢𑁆𑀢𑀸’𑀢𑀺 𑀤𑀍𑀳𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺, 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀬𑀁 𑀧𑀢𑁆𑀢𑁄𑁇

1254\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑁄𑀢𑀢𑁆𑀢𑀤𑀳𑀢𑁄 𑀉𑀤𑀓𑀁 𑀕𑀗𑁆𑀕𑀁 𑀦𑀤𑀺𑀁 𑀧𑀯𑀺𑀲𑀢𑀺, 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀧𑀯𑀺𑀲𑀢𑀺, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀢𑁄 𑀧𑀸𑀢𑀸𑀮𑀫𑀼𑀔𑀁 𑀧𑀯𑀺𑀲𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀁 𑀉𑀤𑀓𑀁 𑀧𑀸𑀢𑀸𑀮𑀫𑀼𑀔𑀕𑀢𑀁 𑀧𑀝𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀢𑁄 𑀕𑀗𑁆𑀕𑀁 𑀦𑀤𑀺𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬, 𑀕𑀗𑁆𑀕𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀧𑀼𑀦 𑀅𑀦𑁄𑀢𑀢𑁆𑀢𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁂𑀦 𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀘𑀢𑀼𑀭𑁄 𑀘 𑀅𑀲𑀗𑁆𑀔𑁆𑀬𑁂𑀬𑁆𑀬𑁂 𑀓𑀼𑀲𑀮𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑀺𑀢𑀁 𑀇𑀫𑀲𑁆𑀲 𑀪𑀯𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸, 𑀲𑁄𑀬𑀁 𑀅𑀦𑁆𑀢𑀺𑀫𑀪𑀯𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀧𑀓𑁆𑀓𑀁 𑀩𑁄𑀥𑀺𑀜𑀸𑀡𑀁 𑀙𑀳𑀺 𑀯𑀲𑁆𑀲𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀽 𑀮𑁄𑀓𑁂 𑀅𑀕𑁆𑀕𑀧𑀼𑀕𑁆𑀕𑀮𑁄, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀓𑀸𑀭𑀡𑀸 𑀧𑀝𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇

1259\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀏𑀯𑀫𑀸𑀳 ‘𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀬 𑀓𑀝𑀼𑀓𑀸𑀬 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸𑀬 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁, 𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄 𑀅𑀜𑁆𑀜𑁄 𑀫𑀕𑁆𑀕𑁄 𑀩𑁄𑀥𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀁 𑀆𑀭𑀩𑁆𑀪 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺?

1260\. “𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀺 𑀔𑁄 𑀧𑀦𑀺𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀭𑀡𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂𑀳𑀺 𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀺? 𑀓𑁄𑀥𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑁂𑀦 𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬, 𑀉𑀧𑀦𑀸𑀳𑁄…𑀧𑁂…  𑀫𑀓𑁆𑀔𑁄…  𑀧𑀍𑀆𑀲𑁄…  𑀇𑀲𑁆𑀲𑀸…  𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁…  𑀫𑀸𑀬𑀸…  𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀁…  𑀣𑀫𑁆𑀪𑁄…  𑀲𑀸𑀭𑀫𑁆𑀪𑁄…  𑀫𑀸𑀦𑁄…  𑀅𑀢𑀺𑀫𑀸𑀦𑁄…  𑀫𑀤𑁄…  𑀧𑀫𑀸𑀤𑁄…  𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀁…  𑀢𑀦𑁆𑀤𑀺…  𑀆𑀮𑀲𑁆𑀬𑀁…  𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸…  𑀭𑀽𑀧𑀸…  𑀲𑀤𑁆𑀤𑀸…  𑀕𑀦𑁆𑀥𑀸…  𑀭𑀲𑀸…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸…  𑀔𑀼𑀤𑀸𑀧𑀺𑀧𑀸𑀲𑀸…  𑀅𑀭𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑁂𑀦 𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀜𑁆𑀘𑀯𑀻𑀲𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀭𑀡𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂𑀳𑀺 𑀤𑀼𑀩𑁆𑀩𑀮𑀻𑀓𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑁇

1261\. 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀔𑀼𑀤𑀸𑀧𑀺𑀧𑀸𑀲𑀸 𑀓𑀸𑀬𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀺𑀁𑀲𑀼, 𑀓𑀸𑀬𑁂 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸𑀬𑁇 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀘𑀢𑀼𑀭𑁄 𑀘 𑀅𑀲𑀗𑁆𑀔𑁆𑀬𑁂𑀬𑁆𑀬𑁂 𑀓𑀧𑁆𑀧𑁂 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁𑀬𑁂𑀯 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀁 𑀅𑀦𑁆𑀯𑁂𑀲𑀺 𑀢𑀸𑀲𑀼 𑀢𑀸𑀲𑀼 𑀚𑀸𑀢𑀻𑀲𑀼, 𑀓𑀺𑀁 𑀧𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀅𑀪𑀺𑀲𑀫𑀬𑀚𑀸𑀢𑀺𑀬𑀁 𑀫𑀕𑁆𑀕𑀁 𑀆𑀭𑀩𑁆𑀪 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑁄 𑀳𑁂𑀲𑁆𑀲𑀢𑀺? 𑀅𑀧𑀺 𑀘, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀜𑁆𑀜𑀸𑀫𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 ‘𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄 𑀅𑀜𑁆𑀜𑁄 𑀫𑀕𑁆𑀕𑁄 𑀩𑁄𑀥𑀸𑀬𑀸’𑀢𑀺𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀏𑀓𑀫𑀸𑀲𑀺𑀓𑁄 𑀲𑀫𑀸𑀦𑁄 𑀧𑀺𑀢𑀼 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁂 𑀲𑀻𑀢𑀸𑀬 𑀚𑀫𑁆𑀩𑀼𑀘𑁆𑀙𑀸𑀬𑀸𑀬 𑀲𑀺𑀭𑀺𑀲𑀬𑀦𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀘𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀺, 𑀜𑀸𑀡𑀁 𑀧𑀭𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑁄 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺𑁇
