# 𑁬𑁇𑁩𑁇𑁧𑁨 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑀧𑀜𑁆𑀳

(201.)

1372\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀢𑁆𑀣𑀺 𑀲𑁄 𑀧𑀤𑁂𑀲𑁄 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀉𑀤𑁆𑀥𑀁 𑀯𑀸 𑀅𑀥𑁄 𑀯𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀯𑀸, 𑀬𑀢𑁆𑀣 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑀦𑁆”𑀢𑀺? “𑀦𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁄 𑀧𑀤𑁂𑀲𑁄 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀯𑀸 𑀤𑀺𑀲𑀸𑀬 𑀉𑀤𑁆𑀥𑀁 𑀯𑀸 𑀅𑀥𑁄 𑀯𑀸 𑀢𑀺𑀭𑀺𑀬𑀁 𑀯𑀸, 𑀬𑀢𑁆𑀣 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑀦𑁆”𑀢𑀺𑁇

1373\. “𑀬𑀤𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦𑀢𑁆𑀣𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑁄𑀓𑀸𑀲𑁄, 𑀢𑁂𑀦 𑀳𑀺 𑀦𑀢𑁆𑀣𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁? 𑀬𑁂𑀲𑀜𑁆𑀘 𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀁, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀸, 𑀓𑀸𑀭𑀡𑀁 𑀢𑀢𑁆𑀣 𑀯𑀓𑁆𑀔𑀸𑀫𑀺𑁇 𑀬𑀣𑀸, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀫𑀳𑀺𑀬𑀸 𑀥𑀜𑁆𑀜𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀢𑁆𑀣𑀺, 𑀕𑀦𑁆𑀥𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀼𑀧𑁆𑀨𑀁 𑀅𑀢𑁆𑀣𑀺, 𑀧𑀼𑀧𑁆𑀨𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀕𑀼𑀫𑁆𑀩𑁄 𑀅𑀢𑁆𑀣𑀺, 𑀨𑀮𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀼𑀓𑁆𑀔𑁄 𑀅𑀢𑁆𑀣𑀺, 𑀭𑀢𑀦𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀓𑀭𑁄 𑀅𑀢𑁆𑀣𑀺, 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀬𑀁 𑀬𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀁 𑀢𑀁 𑀳𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀤𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺, 𑀢𑀲𑁆𑀲 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀸𑀦𑁄𑀓𑀸𑀲𑁄𑀧𑀺 𑀇𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄, 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀸𑀦𑁄𑀓𑀸𑀲𑁄 𑀦𑀢𑁆𑀣𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀦𑀢𑁆𑀣𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀫𑀺, 𑀬𑁂𑀲𑀜𑁆𑀘 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀁, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸 𑀫𑀺𑀘𑁆𑀙𑀸”𑀢𑀺𑁇

1374\. “𑀦𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑁄𑀓𑀸𑀲𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀘𑁂𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 𑀬𑀣𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀅𑀕𑁆𑀕𑀺 𑀦𑀸𑀫, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑁄𑀓𑀸𑀲𑁄, 𑀤𑁆𑀯𑁂 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀲𑀗𑁆𑀖𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑁄 𑀅𑀕𑁆𑀕𑀺𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑁄𑀓𑀸𑀲𑁄, 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇

1375\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢 𑀭𑀢𑀦𑀸𑀦𑀺 𑀦𑀸𑀫𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀇𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀁𑁇 𑀦 𑀘 𑀢𑁂𑀲𑀁 𑀭𑀢𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑁄𑀓𑀸𑀲𑁄 𑀅𑀢𑁆𑀣𑀺, 𑀔𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲 𑀧𑀦 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀩𑀮𑁂𑀦 𑀢𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑁄𑀓𑀸𑀲𑁄, 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻”𑀢𑀺𑁇

1376\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀳𑀺𑀢𑁄𑀓𑀸𑀲𑁄 𑀫𑀸 𑀳𑁄𑀢𑀼, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀦 𑀢𑀁 𑀞𑀸𑀦𑀁, 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀁 𑀞𑀸𑀦𑀁, 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻”𑀢𑀺𑁇

1377\. “𑀓𑀢𑀫𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀁 𑀞𑀸𑀦𑀁, 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀻”𑀢𑀺? “𑀲𑀻𑀮𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀞𑀸𑀦𑀁, 𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀲𑀓𑁆𑀓𑀬𑀯𑀦𑁂𑀧𑀺 𑀘𑀻𑀦𑀯𑀺𑀮𑀸𑀢𑁂𑀧𑀺 𑀅𑀮𑀲𑀦𑁆𑀤𑁂𑀧𑀺 𑀦𑀺𑀕𑀼𑀫𑁆𑀩𑁂𑀧𑀺 𑀓𑀸𑀲𑀺𑀓𑁄𑀲𑀮𑁂𑀧𑀺 𑀓𑀲𑁆𑀫𑀻𑀭𑁂𑀧𑀺 𑀕𑀦𑁆𑀥𑀸𑀭𑁂𑀧𑀺 𑀦𑀕𑀫𑀼𑀤𑁆𑀥𑀦𑀺𑀧𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂𑀧𑀺 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺𑀧𑀺 𑀞𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀓𑀬𑀯𑀦𑁂𑀧𑀺 𑀘𑀻𑀦𑀯𑀺𑀮𑀸𑀢𑁂𑀧𑀺 𑀅𑀮𑀲𑀦𑁆𑀤𑁂𑀧𑀺 𑀦𑀺𑀕𑀼𑀫𑁆𑀩𑁂𑀧𑀺 𑀓𑀸𑀲𑀺𑀓𑁄𑀲𑀮𑁂𑀧𑀺 𑀓𑀲𑁆𑀫𑀻𑀭𑁂𑀧𑀺 𑀕𑀦𑁆𑀥𑀸𑀭𑁂𑀧𑀺 𑀦𑀕𑀫𑀼𑀤𑁆𑀥𑀦𑀺𑀧𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑁂𑀧𑀺 𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺𑀧𑀺 𑀞𑀺𑀢𑁄 𑀆𑀓𑀸𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀲𑀓𑀬𑀯𑀦𑁂𑀧𑀺…𑀧𑁂…  𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺𑀧𑀺 𑀞𑀺𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀭𑁄𑀢𑀺𑁇

1378\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀓𑀬𑀯𑀦𑁂𑀧𑀺…𑀧𑁂…  𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺𑀧𑀺 𑀞𑀺𑀢𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑀤𑀺𑀲𑀸 𑀅𑀢𑁆𑀣𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀬𑁄𑀦𑀺𑀲𑁄 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑀬𑀯𑀦𑁂𑀧𑀺…𑀧𑁂…  𑀬𑀢𑁆𑀣 𑀓𑀢𑁆𑀣𑀘𑀺𑀧𑀺 𑀞𑀺𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀤𑁂𑀲𑀺𑀢𑀁 𑀢𑀬𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀤𑁂𑀲𑀺𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸, 𑀧𑀭𑀺𑀓𑁆𑀔𑀢𑀸 𑀲𑀻𑀮𑀕𑀼𑀡𑀸, 𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺, 𑀉𑀲𑁆𑀲𑀸𑀧𑀺𑀢𑁄 𑀥𑀫𑁆𑀫𑀤𑁆𑀥𑀚𑁄, 𑀲𑀡𑁆𑀞𑀧𑀺𑀢𑀸 𑀥𑀫𑁆𑀫𑀦𑁂𑀢𑁆𑀢𑀺, 𑀅𑀯𑀜𑁆𑀛𑁄 𑀲𑀼𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀸𑀧𑀬𑁄𑀕𑁄, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀕𑀡𑀺𑀯𑀭𑀧𑀯𑀭 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
