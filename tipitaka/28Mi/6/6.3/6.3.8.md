# 𑁬𑁇𑁩𑁇𑁮 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

### 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑀧𑀜𑁆𑀳

(197.)

1329. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁂 𑀢𑁂 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀲𑀩𑁆𑀩𑁂𑀲𑀁𑀬𑁂𑀯 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀓𑀲𑁆𑀲𑀘𑀺 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀓𑀲𑁆𑀲𑀘𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀳𑁄𑀢𑀺, 𑀓𑀲𑁆𑀲𑀘𑀺 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁄𑀢𑀺, 𑀓𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀲𑁆𑀲, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀺, 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀽𑀧𑀧𑀦𑁆𑀦𑀲𑁆𑀲…  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀲𑁆𑀲…  𑀓𑀼𑀳𑀓𑀲𑁆𑀲…  𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀲𑁆𑀲…  𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀲𑁆𑀲…  𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀲𑁆𑀲…  𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀲𑁆𑀲…  𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀲𑁆𑀲…  𑀢𑁂𑀭𑀲𑀦𑁆𑀦𑀁 𑀕𑀭𑀼𑀓𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲…  𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀺…  𑀬𑁄𑀧𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑀳𑀭𑀓𑁄 𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑁄, 𑀢𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁄𑀍𑀅𑀲𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸𑀦𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1330. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑁂 𑀢𑁂 𑀧𑀦𑁆𑀦𑀭𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀭𑀼𑀤𑁆𑀥𑀸𑀬𑁂𑀯, 𑀢𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀳𑁄𑀢𑀼 𑀯𑀸 𑀫𑀸 𑀯𑀸 𑀳𑁄𑀢𑀼, 𑀅𑀣 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑀳𑀭𑀓𑀲𑁆𑀲 𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀺? 𑀏𑀢𑁆𑀣 𑀢𑀸𑀯 𑀧𑀜𑁆𑀳𑁄 𑀪𑀯𑀢𑀺 ‘𑀦𑀦𑀼 𑀦𑀸𑀫 𑀤𑀳𑀭𑀓𑀲𑁆𑀲 𑀦 𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀤𑁄𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀫𑁄𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀫𑀸𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺, 𑀦 𑀅𑀭𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀦 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀫𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀺𑀮𑁂𑀲𑁂𑀳𑀺, 𑀲𑁄 𑀦𑀸𑀫 𑀤𑀳𑀭𑀓𑁄 𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘 𑀅𑀭𑀳𑀢𑀺 𑀘 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀏𑀓𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑁂𑀦 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀦𑁆’”𑀢𑀺𑁇

1331. “𑀢𑀜𑁆𑀜𑁂𑀯𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦𑀸𑀳𑀁 𑀓𑀸𑀭𑀡𑁂𑀦 𑀪𑀡𑀸𑀫𑀺 ‘𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑁄 𑀭𑀚𑀦𑀻𑀬𑁂 𑀭𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀤𑀼𑀲𑁆𑀲𑀦𑀻𑀬𑁂 𑀤𑀼𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀫𑁄𑀳𑀦𑀻𑀬𑁂 𑀫𑀼𑀬𑁆𑀳𑁂𑀬𑁆𑀬, 𑀫𑀤𑀦𑀻𑀬𑁂 𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀯𑀺𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀭𑀢𑀺𑀜𑁆𑀘 𑀅𑀭𑀢𑀺𑀜𑁆𑀘 𑀯𑀺𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀬𑁆𑀬, 𑀪𑀯𑁂𑀬𑁆𑀬 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄, 𑀅𑀧𑀺 𑀘, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀩𑀮𑀁 𑀤𑀼𑀩𑁆𑀩𑀮𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀁 𑀣𑁄𑀓𑀁 𑀫𑀦𑁆𑀤𑀁 𑀅𑀯𑀺𑀪𑀽𑀢𑀁, 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼 𑀕𑀭𑀼𑀓𑀸 𑀪𑀸𑀭𑀺𑀓𑀸 𑀯𑀺𑀧𑀼𑀮𑀸 𑀫𑀳𑀢𑀻𑁇 𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀦 𑀤𑀼𑀩𑁆𑀩𑀮𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑁂𑀦 𑀫𑀦𑁆𑀤𑁂𑀦 𑀅𑀯𑀺𑀪𑀽𑀢𑁂𑀦 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀁 𑀪𑀸𑀭𑀺𑀓𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀫𑀳𑀢𑀺𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁𑁇

1335. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀢𑀼𑀭𑁄 𑀓𑀺𑀲𑁄 𑀅𑀡𑀼𑀧𑀭𑀺𑀫𑀺𑀢𑀓𑀸𑀬𑁄 𑀲𑀸𑀮𑀓𑀓𑀺𑀫𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀁 𑀢𑀺𑀥𑀸 𑀧𑀪𑀺𑀦𑁆𑀦𑀁 𑀦𑀯𑀸𑀬𑀢𑀁 𑀢𑀺𑀯𑀺𑀢𑁆𑀣𑀢𑀁 𑀤𑀲𑀧𑀭𑀺𑀡𑀸𑀳𑀁 𑀅𑀝𑁆𑀞𑀭𑀢𑀦𑀺𑀓𑀁 𑀲𑀓𑀝𑁆𑀞𑀸𑀦𑀫𑀼𑀧𑀕𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀕𑀺𑀮𑀺𑀢𑀼𑀁 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑁂𑀬𑁆𑀬, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀸𑀮𑀓𑀓𑀺𑀫𑀺 𑀲𑀓𑁆𑀓𑀼𑀡𑁂𑀬𑁆𑀬 𑀢𑀁 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀁 𑀕𑀺𑀮𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺? “𑀧𑀭𑀺𑀢𑁆𑀢𑀢𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀮𑀓𑀓𑀺𑀫𑀺𑀲𑁆𑀲, 𑀫𑀳𑀦𑁆𑀢𑀢𑁆𑀢𑀸 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀩𑀮𑀁 𑀤𑀼𑀩𑁆𑀩𑀮𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀅𑀧𑁆𑀧𑀁 𑀣𑁄𑀓𑀁 𑀫𑀦𑁆𑀤𑀁 𑀅𑀯𑀺𑀪𑀽𑀢𑀁, 𑀫𑀳𑀢𑀻 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀤𑀼𑀩𑁆𑀩𑀮𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑁂𑀦 𑀫𑀦𑁆𑀤𑁂𑀦 𑀅𑀯𑀺𑀪𑀽𑀢𑁂𑀦 𑀦 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀫𑀳𑀢𑀺𑀁 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀁 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁, 𑀢𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀊𑀦𑀓𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
