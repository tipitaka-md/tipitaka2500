# 𑁬𑁇𑁩𑁇𑁯 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀯𑀕𑁆𑀕

* 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀜𑁆𑀳

(198.)

1337\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀺𑀁 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀦𑁆”𑀢𑀺? “𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀅𑀫𑀺𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇

1338\. “𑀦 𑀫𑀬𑀁 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀯𑀘𑀦𑀁 𑀲𑀤𑁆𑀤𑀳𑀸𑀫 ‘𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆’𑀢𑀺, 𑀏𑀯𑀫𑁂𑀢𑁆𑀣 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀧𑀘𑁆𑀘𑁂𑀫 ‘𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀦𑁆’𑀢𑀺, 𑀓𑀸𑀭𑀡𑀜𑁆𑀘𑁂𑀢𑁆𑀣 𑀉𑀧𑀮𑀪𑀸𑀫 ‘𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀦𑁆’𑀢𑀺𑁇 𑀓𑀢𑀫𑀁 𑀏𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀁? 𑀬𑁂 𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂𑀲𑀁 𑀤𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀬𑀲𑁆𑀲 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀆𑀢𑀸𑀧𑁄 𑀧𑀭𑀺𑀢𑀸𑀧𑁄 𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀲𑀬𑀦𑀸𑀳𑀸𑀭𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁄 𑀫𑀺𑀤𑁆𑀥𑀲𑁆𑀲 𑀘 𑀉𑀧𑀭𑁄𑀥𑁄 𑀆𑀬𑀢𑀦𑀸𑀦𑀜𑁆𑀘 𑀧𑀝𑀺𑀧𑀻𑀍𑀅𑀦𑀁 𑀥𑀦𑀥𑀜𑁆𑀜𑀧𑀺𑀬𑀜𑀸𑀢𑀺𑀫𑀺𑀢𑁆𑀢𑀧𑁆𑀧𑀚𑀳𑀦𑀁𑁇 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀮𑁄𑀓𑁂 𑀲𑀼𑀔𑀺𑀢𑀸 𑀲𑀼𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸, 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀆𑀬𑀢𑀦𑁂 𑀭𑀫𑁂𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀳𑁂𑀦𑁆𑀢𑀺, 𑀫𑀦𑀸𑀧𑀺𑀓𑀫𑀦𑀸𑀧𑀺𑀓𑀩𑀳𑀼𑀯𑀺𑀥𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀭𑀽𑀧𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀁 𑀭𑀫𑁂𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀳𑁂𑀦𑁆𑀢𑀺, 𑀫𑀦𑀸𑀧𑀺𑀓𑀫𑀦𑀸𑀧𑀺𑀓𑀕𑀻𑀢𑀯𑀸𑀤𑀺𑀢𑀩𑀳𑀼𑀯𑀺𑀥𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑁂𑀦 𑀲𑁄𑀢𑀁 𑀭𑀫𑁂𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀳𑁂𑀦𑁆𑀢𑀺, 𑀫𑀦𑀸𑀧𑀺𑀓𑀫𑀦𑀸𑀧𑀺𑀓𑀧𑀼𑀧𑁆𑀨𑀨𑀮𑀧𑀢𑁆𑀢𑀢𑀘𑀫𑀽𑀮𑀲𑀸𑀭𑀩𑀳𑀼𑀯𑀺𑀥𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀕𑀦𑁆𑀥𑁂𑀦 𑀖𑀸𑀦𑀁 𑀭𑀫𑁂𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀳𑁂𑀦𑁆𑀢𑀺, 𑀫𑀦𑀸𑀧𑀺𑀓𑀫𑀦𑀸𑀧𑀺𑀓𑀔𑀚𑁆𑀚𑀪𑁄𑀚𑁆𑀚𑀮𑁂𑀬𑁆𑀬𑀧𑁂𑀬𑁆𑀬𑀲𑀸𑀬𑀦𑀻𑀬𑀩𑀳𑀼𑀯𑀺𑀥𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀭𑀲𑁂𑀦 𑀚𑀺𑀯𑁆𑀳𑀁 𑀭𑀫𑁂𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀳𑁂𑀦𑁆𑀢𑀺, 𑀫𑀦𑀸𑀧𑀺𑀓𑀫𑀦𑀸𑀧𑀺𑀓𑀲𑀡𑁆𑀳𑀲𑀼𑀔𑀼𑀫𑀫𑀼𑀤𑀼𑀫𑀤𑁆𑀤𑀯𑀩𑀳𑀼𑀯𑀺𑀥𑀲𑀼𑀪𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀨𑀲𑁆𑀲𑁂𑀦 𑀓𑀸𑀬𑀁 𑀭𑀫𑁂𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀳𑁂𑀦𑁆𑀢𑀺, 𑀫𑀦𑀸𑀧𑀺𑀓𑀫𑀦𑀸𑀧𑀺𑀓𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀓𑀲𑀼𑀪𑀸𑀲𑀼𑀪𑀩𑀳𑀼𑀯𑀺𑀥𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑁂𑀦 𑀫𑀦𑀁 𑀭𑀫𑁂𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀽𑀳𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀼𑀫𑁆𑀳𑁂 𑀢𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑁄𑀢𑀖𑀸𑀦𑀚𑀺𑀯𑁆𑀳𑀸𑀓𑀸𑀬𑀫𑀦𑁄𑀩𑁆𑀭𑀽𑀳𑀦𑀁 𑀳𑀦𑀣 𑀉𑀧𑀳𑀦𑀣, 𑀙𑀺𑀦𑁆𑀤𑀣 𑀉𑀧𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀣, 𑀭𑀼𑀦𑁆𑀥𑀣 𑀉𑀧𑀭𑀼𑀦𑁆𑀥𑀣𑁇 𑀢𑁂𑀦 𑀓𑀸𑀬𑁄𑀧𑀺 𑀧𑀭𑀺𑀢𑀧𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀧𑀭𑀺𑀢𑀧𑀢𑀺, 𑀓𑀸𑀬𑁂 𑀧𑀭𑀺𑀢𑀢𑁆𑀢𑁂 𑀓𑀸𑀬𑀺𑀓𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀺𑀬𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀭𑀺𑀢𑀢𑁆𑀢𑁂 𑀘𑁂𑀢𑀲𑀺𑀓𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀦𑀦𑀼 𑀫𑀸𑀕𑀡𑁆𑀟𑀺𑀬𑁄𑀧𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀭𑀳𑀫𑀸𑀦𑁄 𑀏𑀯𑀫𑀸𑀳 ‘𑀪𑀽𑀦𑀳𑀼 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄’𑀢𑀺𑁇 𑀇𑀤𑀫𑁂𑀢𑁆𑀣 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦𑀸𑀳𑀁 𑀓𑀸𑀭𑀡𑁂𑀦 𑀩𑁆𑀭𑀽𑀫𑀺 ‘𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀦𑁆’”𑀢𑀺𑁇

1340\. 𑀅𑀧𑀭𑀫𑁆𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀓𑀸𑀭𑀡𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀦 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀲𑀺𑀧𑁆𑀧𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀁 𑀦𑀸𑀫𑀸”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀺 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀲𑀺𑀧𑁆𑀧𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀦𑁆”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀘𑀭𑀺𑀬𑀸 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑀦𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀉𑀤𑀓𑀸𑀳𑀭𑀡𑀖𑀭𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀫𑀼𑀔𑁄𑀤𑀓𑀸𑀦𑀼𑀧𑁆𑀧𑀤𑀸𑀦𑁂𑀦 𑀉𑀘𑁆𑀙𑀺𑀝𑁆𑀞𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀡𑁅𑀘𑁆𑀙𑀸𑀤𑀦𑀦𑀳𑀸𑀧𑀦𑀧𑀸𑀤𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀓𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀦𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑁂𑀬𑁆𑀬𑀸𑀬 𑀯𑀺𑀲𑀫𑀪𑁄𑀚𑀦𑁂𑀦 𑀓𑀸𑀬𑀁 𑀆𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀦𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀁 𑀦𑀸𑀫, 𑀲𑀺𑀧𑁆𑀧𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀸𑀬 𑀧𑀼𑀩𑁆𑀩𑀪𑀸𑀕𑁄 𑀏𑀲𑁄, 𑀤𑀼𑀓𑁆𑀔𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀆𑀘𑀭𑀺𑀬𑀸 𑀲𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀅𑀫𑀺𑀲𑁆𑀲𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀢𑀁 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀦 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀁𑁇 𑀬𑁂 𑀧𑀦 𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀓𑀸𑀬𑀜𑁆𑀘 𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀆𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀲𑀬𑀦𑀸𑀳𑀸𑀭𑀁 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀫𑀺𑀤𑁆𑀥𑀁 𑀉𑀧𑀭𑀼𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀢𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀧𑀻𑀍𑀏𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀜𑁆𑀘 𑀚𑀻𑀯𑀺𑀢𑀜𑁆𑀘 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀺𑀢𑁆𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺, 𑀆𑀘𑀭𑀺𑀬𑀸 𑀯𑀺𑀬 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀔𑀁𑁇 𑀏𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀦𑁆𑀢𑀲𑀼𑀔𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁, 𑀦 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀫𑀺𑀲𑁆𑀲𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
