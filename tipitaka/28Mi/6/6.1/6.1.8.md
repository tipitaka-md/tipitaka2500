# 𑁬𑁇𑁧𑁇𑁮 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀧𑀜𑁆𑀳

(178.)

1137\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀕𑀺𑀳𑀺𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀘 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀘 𑀓𑁄 𑀯𑀺𑀲𑁂𑀲𑁄, 𑀓𑀺𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀉𑀪𑁄𑀧𑁂𑀢𑁂 𑀲𑀫𑀲𑀫𑀕𑀢𑀺𑀓𑀸, 𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀲𑀫𑀲𑀫𑁄 𑀯𑀺𑀧𑀸𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀓𑀺𑀜𑁆𑀘𑀺 𑀦𑀸𑀦𑀸𑀓𑀸𑀭𑀡𑀁 𑀅𑀢𑁆𑀣𑀻”𑀢𑀺?

1138\. “𑀤𑀲𑀬𑀺𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑀼𑀡𑀸 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀕𑀺𑀳𑀺𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀢𑁄 𑀯𑀺𑀲𑁂𑀲𑁂𑀦 𑀅𑀢𑀺𑀭𑁂𑀓𑀸, 𑀤𑀲𑀳𑀺 𑀘 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇

1139\. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀕𑀼𑀡𑀸 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀕𑀺𑀳𑀺𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀢𑁄 𑀯𑀺𑀲𑁂𑀲𑁂𑀦 𑀅𑀢𑀺𑀭𑁂𑀓𑀸? 𑀇𑀥, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀲𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂 𑀲𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀁𑀖𑁂 𑀲𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀲𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀤𑁆𑀤𑁂𑀲𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀬 𑀯𑀸𑀬𑀫𑀢𑀺, 𑀲𑀯𑀦𑀩𑀳𑀼𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀪𑀺𑀦𑁆𑀦𑀲𑀻𑀮𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀭𑀺𑀲𑀕𑀢𑁄 𑀆𑀓𑀧𑁆𑀧𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀺, 𑀕𑀭𑀳𑀪𑀬𑀸 𑀓𑀸𑀬𑀺𑀓𑀁 𑀯𑀸𑀘𑀲𑀺𑀓𑀁 𑀭𑀓𑁆𑀔𑀢𑀺, 𑀧𑀥𑀸𑀦𑀸𑀪𑀺𑀫𑀼𑀔𑀜𑁆𑀘𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀸𑀫𑀜𑁆𑀜𑀁 𑀉𑀧𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀭𑁄𑀦𑁆𑀢𑁄𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀸𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀆𑀘𑀭𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀢𑁆𑀣𑀻 𑀲𑀧𑀢𑀺𑀓𑀸 𑀦𑀺𑀮𑀻𑀬𑀺𑀢𑁆𑀯𑀸 𑀭𑀳𑀲𑁆𑀲𑁂𑀦𑁂𑀯 𑀧𑀸𑀧𑀫𑀸𑀘𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀭𑁄𑀦𑁆𑀢𑁄𑀧𑀺 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀸𑀧𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀆𑀘𑀭𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀲 𑀕𑀼𑀡𑀸 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀕𑀺𑀳𑀺𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀢𑁄 𑀯𑀺𑀲𑁂𑀲𑁂𑀦 𑀅𑀢𑀺𑀭𑁂𑀓𑀸𑁇

1140\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺? 𑀅𑀦𑀯𑀚𑁆𑀚𑀓𑀯𑀘𑀥𑀸𑀭𑀡𑀢𑀸𑀬𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀇𑀲𑀺𑀲𑀸𑀫𑀜𑁆𑀜𑀪𑀡𑁆𑀟𑀼𑀮𑀺𑀗𑁆𑀕𑀥𑀸𑀭𑀡𑀢𑁄𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀲𑀁𑀖𑀲𑀫𑀬𑀫𑀦𑀼𑀧𑁆𑀧𑀯𑀺𑀝𑁆𑀞𑀢𑀸𑀬𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑀲𑀁𑀖𑀲𑀭𑀡𑀕𑀢𑀢𑀸𑀬𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀧𑀥𑀸𑀦𑀸𑀲𑀬𑀦𑀺𑀓𑁂𑀢𑀯𑀸𑀲𑀺𑀢𑀸𑀬𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀥𑀭𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀢𑁄𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀧𑀯𑀭𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀢𑁄𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀤𑀻𑀧𑀕𑀢𑀺𑀧𑀭𑀸𑀬𑀡𑀢𑀸𑀬𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, ‘𑀅𑀕𑁆𑀕𑁄 𑀩𑀼𑀤𑁆𑀥𑁄’𑀢𑀺 𑀏𑀓𑀦𑁆𑀢𑁅𑀚𑀼𑀤𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑀬𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀲𑀫𑀸𑀤𑀸𑀦𑀢𑁄𑀧𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀲𑀳𑀺 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇

1141\. 𑀲𑀼𑀯𑀺𑀧𑀦𑁆𑀦𑁄𑀧𑀺 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀲𑀼𑀩𑀳𑀮𑀫𑁆𑀧𑀺 𑀓𑀮𑀮𑀓𑀤𑁆𑀤𑀫𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀅𑀧𑀦𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀯𑀺𑀧𑀦𑁆𑀦𑁄𑀧𑀺 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇

1142\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑀁 𑀲𑀼𑀓𑀼𑀥𑀺𑀢𑀫𑁆𑀧𑀺 𑀧𑀚𑁆𑀚𑀮𑀦𑁆𑀢𑀁 𑀫𑀳𑀦𑁆𑀢𑀁 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀯𑀺𑀧𑀦𑁆𑀦𑁄𑀧𑀺 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇

1143\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁄𑀚𑀦𑀁 𑀯𑀺𑀭𑀲𑀫𑁆𑀧𑀺 𑀔𑀼𑀤𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀅𑀧𑀦𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀯𑀺𑀧𑀦𑁆𑀦𑁄𑀧𑀺 𑀲𑀫𑀡𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀤𑀸𑀬𑀓𑀸𑀦𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺𑁇

1144\. 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀫𑀚𑁆𑀛𑀺𑀫𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑀮𑀜𑁆𑀙𑀓𑁂 𑀤𑀓𑁆𑀔𑀺𑀡𑀯𑀺𑀪𑀗𑁆𑀕𑁂 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂—

1145\. _‘𑀬𑁄 𑀲𑀻𑀮𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁂𑀲𑀼 𑀤𑀤𑀸𑀢𑀺 𑀤𑀸𑀦𑀁,_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀮𑀤𑁆𑀥𑀁 𑀲𑀼𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄;_  
_𑀅𑀪𑀺𑀲𑀤𑁆𑀤𑀳𑀁 𑀓𑀫𑁆𑀫𑀨𑀮𑀁 𑀉𑀍𑀆𑀭𑀁,_  
_𑀲𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀤𑀸𑀬𑀓𑀢𑁄 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀢𑀻’”𑀢𑀺𑁇_  


1146\. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀸𑀯𑀢𑀓𑀁 𑀫𑀬𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳, 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀑𑀧𑀫𑁆𑀫𑁂𑀳𑀺 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀯𑀺𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑁄 𑀅𑀫𑀢𑀫𑀥𑀼𑀭𑀁 𑀲𑀯𑀦𑀽𑀧𑀕𑀁 𑀅𑀓𑀸𑀲𑀺𑁇 𑀬𑀣𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀽𑀤𑁄 𑀯𑀸 𑀲𑀽𑀤𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀻 𑀯𑀸 𑀢𑀸𑀯𑀢𑀓𑀁 𑀫𑀁𑀲𑀁 𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀯𑀺𑀥𑁂𑀳𑀺 𑀲𑀫𑁆𑀪𑀸𑀭𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀽𑀧𑀪𑁄𑀕𑀁 𑀓𑀭𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀸𑀯𑀢𑀓𑀁 𑀫𑀬𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳, 𑀢𑀁 𑀢𑁆𑀯𑀁 𑀑𑀧𑀫𑁆𑀫𑁂𑀳𑀺 𑀓𑀸𑀭𑀡𑁂𑀳𑀺 𑀯𑀺𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀅𑀫𑀢𑀫𑀥𑀼𑀭𑀁 𑀲𑀯𑀦𑀽𑀧𑀕𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺𑁇
