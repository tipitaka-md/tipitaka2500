# 𑁬𑁇𑁧𑁇𑁭 𑀅𑀪𑀺𑀲𑀫𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀓𑀭𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀅𑀪𑀺𑀲𑀫𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀓𑀭𑀧𑀜𑁆𑀳

(177.)

1127\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀥 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀕𑀺𑀳𑀻 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬, 𑀅𑀢𑁆𑀢𑀦𑀸𑀧𑀺 𑀲𑁄 𑀦 𑀚𑀸𑀦𑁂𑀬𑁆𑀬 ‘𑀕𑀺𑀳𑀺𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀲𑁆𑀫𑀻’𑀢𑀺, 𑀦𑀧𑀺 𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑁄 𑀓𑁄𑀘𑀺 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬 ‘𑀕𑀺𑀳𑀺𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀲𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀘 𑀢𑀣𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀓𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀭𑀡𑁂𑀦𑀸”𑀢𑀺? “𑀬𑁄 𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑀸𑀬, 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀢𑀲𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀪𑀯𑀢𑀻”𑀢𑀺𑁇

1128\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑁂 𑀲𑀢𑀺 𑀆𑀯𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀆𑀯𑀝𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀻’𑀢𑀺𑁇 𑀇𑀫𑀲𑁆𑀲 𑀧𑀦 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀚𑀸𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀲𑀫𑁂𑀦 𑀯𑀺𑀲𑀫𑁂𑀦𑁂𑀲𑁄 𑀧𑀜𑁆𑀳𑁄 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀘𑀺𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀚𑁆𑀚𑁂𑀣𑀸”𑀢𑀺𑁇

1129\. “𑀭𑀼𑀳𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀓𑀝𑁆𑀞𑁂 𑀲𑀼𑀓𑀮𑀮𑁂 𑀫𑀡𑁆𑀟𑀔𑁂𑀢𑁆𑀢𑁂 𑀲𑀸𑀭𑀤𑀁 𑀲𑀼𑀔𑀲𑀬𑀺𑀢𑀁 𑀩𑀻𑀚𑀦𑁆”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀜𑁆𑀜𑁂𑀯 𑀩𑀻𑀚𑀁 𑀖𑀦𑀲𑁂𑀮𑀲𑀺𑀮𑀸𑀢𑀮𑁂 𑀭𑀼𑀳𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀜𑁆𑀜𑁂𑀯 𑀩𑀻𑀚𑀁 𑀓𑀮𑀮𑁂 𑀭𑀼𑀳𑀢𑀺, 𑀓𑀺𑀲𑁆𑀲 𑀖𑀦𑀲𑁂𑀮𑁂 𑀦 𑀭𑀼𑀳𑀢𑀻”𑀢𑀺? “𑀦𑀢𑁆𑀣𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀲𑁆𑀲 𑀩𑀻𑀚𑀲𑁆𑀲 𑀭𑀼𑀳𑀦𑀸𑀬 𑀖𑀦𑀲𑁂𑀮𑁂 𑀳𑁂𑀢𑀼, 𑀅𑀳𑁂𑀢𑀼𑀦𑀸 𑀩𑀻𑀚𑀁 𑀦 𑀭𑀼𑀳𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂𑀦 𑀳𑁂𑀢𑀼𑀦𑀸 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀅𑀳𑁂𑀢𑀼𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀺𑁇

1130\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀡𑁆𑀟𑀮𑁂𑀟𑁆𑀟𑀼𑀮𑀕𑀼𑀍𑀅𑀫𑀼𑀕𑁆𑀕𑀭𑀸 𑀧𑀣𑀯𑀺𑀬𑀸 𑀞𑀸𑀦𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀬𑁂𑀯 𑀤𑀡𑁆𑀟𑀮𑁂𑀟𑁆𑀟𑀼𑀮𑀕𑀼𑀍𑀅𑀫𑀼𑀕𑁆𑀕𑀭𑀸 𑀕𑀕𑀦𑁂 𑀞𑀸𑀦𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑁂𑀬𑁂𑀯 𑀤𑀡𑁆𑀟𑀮𑁂𑀟𑁆𑀟𑀼𑀮𑀕𑀼𑀍𑀅𑀫𑀼𑀕𑁆𑀕𑀭𑀸 𑀧𑀣𑀯𑀺𑀬𑀸 𑀞𑀸𑀦𑀫𑀼𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀕𑀕𑀦𑁂 𑀦 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀦𑀢𑁆𑀣𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑁂𑀲𑀁 𑀤𑀡𑁆𑀟𑀮𑁂𑀟𑁆𑀟𑀼𑀮𑀕𑀼𑀍𑀅𑀫𑀼𑀕𑁆𑀕𑀭𑀸𑀦𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑀸𑀬 𑀆𑀓𑀸𑀲𑁂 𑀳𑁂𑀢𑀼, 𑀅𑀳𑁂𑀢𑀼𑀦𑀸 𑀦 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀲𑁆𑀲 𑀢𑁂𑀦 𑀤𑁄𑀲𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑀬𑀳𑁂𑀢𑀼 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁂 𑀅𑀳𑁂𑀢𑀼𑀦𑀸 𑀅𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

1131\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑀮𑁂 𑀅𑀕𑁆𑀕𑀺 𑀚𑀮𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁄𑀬𑁂𑀯 𑀅𑀕𑁆𑀕𑀺 𑀉𑀤𑀓𑁂 𑀚𑀮𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀲𑁄𑀬𑁂𑀯 𑀅𑀕𑁆𑀕𑀺 𑀣𑀮𑁂 𑀚𑀮𑀢𑀺, 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀉𑀤𑀓𑁂 𑀦 𑀚𑀮𑀢𑀻”𑀢𑀺? “𑀦𑀢𑁆𑀣𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀕𑁆𑀕𑀺𑀲𑁆𑀲 𑀚𑀮𑀦𑀸𑀬 𑀉𑀤𑀓𑁂 𑀳𑁂𑀢𑀼, 𑀅𑀳𑁂𑀢𑀼𑀦𑀸 𑀦 𑀚𑀮𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀲𑁆𑀲 𑀢𑁂𑀦 𑀤𑁄𑀲𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑀬𑀳𑁂𑀢𑀼 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁂 𑀅𑀳𑁂𑀢𑀼𑀦𑀸 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄 𑀦 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1132\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀧𑀼𑀦𑀧𑁂𑀢𑀁 𑀅𑀢𑁆𑀣𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀳𑀺, 𑀦 𑀫𑁂 𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀪𑀯𑀢𑀺, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀲𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑁂 𑀆𑀯𑀭𑀡𑀁 𑀳𑁄𑀢𑀻𑀢𑀺, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀫𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀺𑀲𑀁 𑀳𑀮𑀸𑀳𑀮𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦 𑀔𑀸𑀬𑀺𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀳𑀭𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀓𑀭𑀁 𑀳𑁄𑀢𑀺𑁇

1133\. 𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑁆𑀕𑀺 𑀅𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀁 𑀟𑀳𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀓𑀭𑀁 𑀳𑁄𑀢𑀺𑁇

1134\. 𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀁 𑀆𑀲𑀻𑀯𑀺𑀲𑁄 𑀟𑀁𑀲𑀺𑀢𑁆𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀁 𑀳𑀭𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀓𑀭𑀁 𑀳𑁄𑀢𑀺𑁇

1135\. 𑀦𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀮𑀺𑀗𑁆𑀕𑀭𑀸𑀚𑀸 𑀲𑀫𑀡𑀓𑁄𑀮𑀜𑁆𑀜𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑁄 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀫𑀪𑀺𑀭𑀼𑀬𑁆𑀳 𑀓𑀼𑀮𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁄𑀧𑀺 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀩𑁄𑀥𑀺𑀫𑀡𑁆𑀟𑀲𑁆𑀲 𑀉𑀧𑀭𑀺𑀢𑁄 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁂𑀦𑀧𑀺 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁 𑀅𑀪𑀺𑀲𑀫𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀓𑀭𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀚𑀺𑀦𑀪𑀸𑀲𑀺𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀸𑀭𑀡𑀁 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑀼𑀁, 𑀏𑀲𑁄𑀯𑁂𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇

---

1136\. 𑀅𑀪𑀺𑀲𑀫𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀓𑀭𑀧𑀜𑁆𑀳𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇
