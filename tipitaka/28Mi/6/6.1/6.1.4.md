# 𑁬𑁇𑁧𑁇𑁪 𑀧𑀝𑀺𑀧𑀤𑀸𑀤𑁄𑀲𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀧𑀝𑀺𑀧𑀤𑀸𑀤𑁄𑀲𑀧𑀜𑁆𑀳

(174.)

1093\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀦𑁂𑀢𑀸𑀤𑀺𑀲𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀆𑀭𑀫𑁆𑀪𑁄 𑀅𑀳𑁄𑀲𑀺 𑀦𑀺𑀓𑁆𑀓𑀫𑁄 𑀓𑀺𑀮𑁂𑀲𑀬𑀼𑀤𑁆𑀥𑀁 𑀫𑀘𑁆𑀘𑀼𑀲𑁂𑀦𑀁 𑀯𑀺𑀥𑀫𑀦𑀁 𑀆𑀳𑀸𑀭𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁄 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸, 𑀏𑀯𑀭𑀽𑀧𑁂 𑀧𑀭𑀓𑁆𑀓𑀫𑁂 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀲𑁆𑀲𑀸𑀤𑀁 𑀅𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀢𑀫𑁂𑀯 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀯𑁄𑀘 ‘𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀬 𑀓𑀝𑀼𑀓𑀸𑀬 𑀤𑀼𑀓𑁆𑀓𑀭𑀓𑀸𑀭𑀺𑀓𑀸𑀬 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑀁, 𑀲𑀺𑀬𑀸 𑀦𑀼 𑀔𑁄 𑀅𑀜𑁆𑀜𑁄 𑀫𑀕𑁆𑀕𑁄 𑀩𑁄𑀥𑀸𑀬𑀸’𑀢𑀺, 𑀢𑀢𑁄 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑁂𑀦 𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀢𑁆𑀢𑁄, 𑀧𑀼𑀦 𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀸𑀯𑀓𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺𑁇

1094\. _‘𑀆𑀭𑀫𑁆𑀪𑀣 𑀦𑀺𑀓𑁆𑀔𑀫𑀣,_  
_𑀬𑀼𑀜𑁆𑀚𑀣 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂;_  
_𑀥𑀼𑀦𑀸𑀣 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀲𑁂𑀦𑀁,_  
_𑀦𑀍𑀆𑀕𑀸𑀭𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑁄’𑀢𑀺𑁇_  


1095\. 𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑀸 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀦𑁄 𑀯𑀺𑀭𑀢𑁆𑀢𑀭𑀽𑀧𑁄, 𑀢𑀢𑁆𑀣 𑀲𑀸𑀯𑀓𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀻”𑀢𑀺?

1096\. “𑀢𑀤𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀢𑀭𑀳𑀺𑀧𑀺 𑀲𑀸𑀬𑁂𑀯 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀢𑀁𑀬𑁂𑀯 𑀧𑀝𑀺𑀧𑀤𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀢𑁆𑀢𑁄𑁇 𑀅𑀧𑀺 𑀘, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀅𑀢𑀺𑀯𑀻𑀭𑀺𑀬𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦𑀺𑀭𑀯𑀲𑁂𑀲𑀢𑁄 𑀆𑀳𑀸𑀭𑀁 𑀉𑀧𑀭𑀼𑀦𑁆𑀥𑀺𑁇 𑀢𑀲𑁆𑀲 𑀆𑀳𑀸𑀭𑀽𑀧𑀭𑁄𑀥𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀤𑀼𑀩𑁆𑀩𑀮𑁆𑀬𑁂𑀦 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑀼𑀁, 𑀲𑁄 𑀫𑀢𑁆𑀢𑀫𑀢𑁆𑀢𑀁 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑀸𑀳𑀸𑀭𑀁 𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀢𑀸𑀬𑁂𑀯 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑁇 𑀲𑀸𑀬𑁂𑀯, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀝𑀺𑀧𑀤𑀸 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬𑁇

1097\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀳𑀸𑀭𑁄 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑁄, 𑀆𑀳𑀸𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀢𑁆𑀢𑀸 𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀸𑀬𑁂𑀯 𑀧𑀝𑀺𑀧𑀤𑀸 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀧𑁆𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀦𑁂𑀲𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑁄𑀲𑁄 𑀆𑀭𑀫𑁆𑀪𑀲𑁆𑀲, 𑀦 𑀦𑀺𑀓𑁆𑀓𑀫𑀲𑁆𑀲, 𑀦 𑀓𑀺𑀮𑁂𑀲𑀬𑀼𑀤𑁆𑀥𑀲𑁆𑀲, 𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀧𑀸𑀧𑀼𑀡𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀁, 𑀅𑀣 𑀔𑁄 𑀆𑀳𑀸𑀭𑀽𑀧𑀭𑁄𑀥𑀲𑁆𑀲𑁂𑀯𑁂𑀲𑁄 𑀤𑁄𑀲𑁄, 𑀲𑀤𑀸 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸 𑀬𑁂𑀯𑁂𑀲𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑁇

1098\. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑀺𑀯𑁂𑀕𑁂𑀦 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀢𑁂𑀦 𑀲𑁄 𑀧𑀓𑁆𑀔𑀳𑀢𑁄 𑀯𑀸 𑀪𑀯𑁂𑀬𑁆𑀬 𑀧𑀻𑀞𑀲𑀧𑁆𑀧𑀻 𑀯𑀸 𑀅𑀲𑀜𑁆𑀘𑀭𑁄 𑀧𑀣𑀯𑀺𑀢𑀮𑁂𑁇 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀸𑀧𑀣𑀯𑀺𑀬𑀸 𑀤𑁄𑀲𑁄 𑀅𑀢𑁆𑀣𑀺, 𑀬𑁂𑀦 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀓𑁆𑀔𑀳𑀢𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂; 𑀲𑀤𑀸 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀧𑀣𑀯𑀻, 𑀓𑀼𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀤𑁄𑀲𑁄? 𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲𑁂𑀯𑁂𑀲𑁄 𑀤𑁄𑀲𑁄, 𑀬𑁂𑀦 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀓𑁆𑀔𑀳𑀢𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑁂𑀲𑁄 𑀤𑁄𑀲𑁄 𑀆𑀭𑀫𑁆𑀪𑀲𑁆𑀲, 𑀦 𑀦𑀺𑀓𑁆𑀓𑀫𑀲𑁆𑀲, 𑀦 𑀓𑀺𑀮𑁂𑀲𑀬𑀼𑀤𑁆𑀥𑀲𑁆𑀲, 𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀧𑀸𑀧𑀼𑀡𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀁, 𑀅𑀣 𑀔𑁄 𑀆𑀳𑀸𑀭𑀽𑀧𑀭𑁄𑀥𑀲𑁆𑀲𑁂𑀯𑁂𑀲𑁄 𑀤𑁄𑀲𑁄 𑀲𑀤𑀸 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸 𑀬𑁂𑀯𑁂𑀲𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑁇

1099\. 𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀲𑀸𑀝𑀓𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀬𑁆𑀬, 𑀦 𑀲𑁄 𑀢𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀬𑁆𑀬, 𑀦𑁂𑀲𑁄 𑀤𑁄𑀲𑁄 𑀉𑀤𑀓𑀲𑁆𑀲, 𑀲𑀤𑀸 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀉𑀤𑀓𑀁𑁇 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲𑁂𑀯𑁂𑀲𑁄 𑀤𑁄𑀲𑁄𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑁂𑀲𑁄 𑀤𑁄𑀲𑁄 𑀆𑀭𑀫𑁆𑀪𑀲𑁆𑀲, 𑀦 𑀦𑀺𑀓𑁆𑀓𑀫𑀲𑁆𑀲, 𑀦 𑀓𑀺𑀮𑁂𑀲𑀬𑀼𑀤𑁆𑀥𑀲𑁆𑀲, 𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀦 𑀧𑀸𑀧𑀼𑀡𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀢𑀜𑀸𑀡𑀁, 𑀅𑀣 𑀔𑁄 𑀆𑀳𑀸𑀭𑀽𑀧𑀭𑁄𑀥𑀲𑁆𑀲𑁂𑀯𑁂𑀲𑁄 𑀤𑁄𑀲𑁄, 𑀲𑀤𑀸 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸 𑀬𑁂𑀯𑁂𑀲𑀸 𑀧𑀝𑀺𑀧𑀤𑀸, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀢𑀸𑀬𑁂𑀯 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀲𑀸𑀯𑀓𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺, 𑀏𑀯𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀤𑀸 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀸 𑀅𑀦𑀯𑀚𑁆𑀚𑀸 𑀲𑀸 𑀧𑀝𑀺𑀧𑀤𑀸”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
