# 𑁬𑁇𑁧𑁇𑁯 𑀉𑀤𑀓𑀲𑀢𑁆𑀢𑀚𑀻𑀯𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀉𑀤𑀓𑀲𑀢𑁆𑀢𑀚𑀻𑀯𑀧𑀜𑁆𑀳

(179.)

1148. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀫𑀁 𑀉𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀺𑀫𑁆𑀳𑀺 𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑀁 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀁, 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀉𑀤𑀓𑀁 𑀚𑀻𑀯𑀢𑀺, 𑀓𑀺𑀁 𑀓𑀻𑀍𑀅𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀅𑀜𑁆𑀜𑁂𑀦 𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀚𑀻𑀯𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸, 𑀅𑀧𑀺 𑀘, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀉𑀤𑀓𑀁 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀦𑁆”𑀢𑀺𑁇

1149. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀉𑀤𑀓𑀁 𑀚𑀻𑀯𑀢𑀻𑀢𑀺 𑀲𑀻𑀢𑁄𑀤𑀓𑀁 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀉𑀤𑀓𑀁 𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀓𑀢𑀺𑀓𑀯𑁂𑀓𑀢𑀺𑀓𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀕𑀭𑀳𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀯𑀦𑁆𑀢𑀺 ‘𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀻’𑀢𑀺, 𑀢𑀁 𑀢𑁂𑀲𑀁 𑀕𑀭𑀳𑀁 𑀧𑀭𑀺𑀪𑀯𑀁 𑀯𑀺𑀦𑁄𑀤𑁂𑀳𑀺 𑀅𑀧𑀦𑁂𑀳𑀺 𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀚𑀻𑀯𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸, 𑀅𑀧𑀺 𑀘, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀉𑀤𑀓𑀁 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀁𑁇

1150. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀲𑁄𑀩𑁆𑀪𑀲𑀭𑀲𑀭𑀺𑀢𑀤𑀳𑀢𑀍𑀆𑀓𑀓𑀦𑁆𑀤𑀭𑀧𑀤𑀭𑁅𑀤𑀧𑀸𑀦𑀦𑀺𑀦𑁆𑀦𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀕𑀢𑀁 𑀯𑀸𑀢𑀸𑀢𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀢𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀢𑁆𑀣 𑀉𑀤𑀓𑀁 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀦𑁆”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀚𑀻𑀯𑁂𑀬𑁆𑀬, 𑀢𑀢𑁆𑀣𑀸𑀧𑀺 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑁂𑀬𑁆𑀬, 𑀇𑀫𑀺𑀦𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀚𑀸𑀦𑀸𑀳𑀺 ‘𑀦𑀢𑁆𑀣𑀺 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸, 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀉𑀤𑀓𑀁 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀦𑁆’𑀢𑀺𑁇

1151. 𑀅𑀧𑀭𑀫𑁆𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀓𑀸𑀭𑀡𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 ‘𑀦𑀢𑁆𑀣𑀺 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸, 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑀸 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀢𑀡𑁆𑀟𑀼𑀮𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀺𑀲𑁆𑀲𑀺𑀢𑀁 𑀪𑀸𑀚𑀦𑀕𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑀺𑀳𑀺𑀢𑀁 𑀉𑀤𑁆𑀥𑀦𑁂 𑀅𑀞𑀧𑀺𑀢𑀁, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀢𑁆𑀣 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀘𑀮𑀁 𑀳𑁄𑀢𑀺 𑀲𑀦𑁆𑀢𑀲𑀦𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀢𑀁𑀬𑁂𑀯 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀪𑀸𑀚𑀦𑀕𑀢𑀁 𑀅𑀕𑁆𑀕𑀺𑀁 𑀉𑀚𑁆𑀚𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀉𑀤𑁆𑀥𑀦𑁂 𑀞𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀢𑁆𑀣 𑀉𑀤𑀓𑀁 𑀅𑀘𑀮𑀁 𑀳𑁄𑀢𑀺 𑀲𑀦𑁆𑀢𑀲𑀦𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀮𑀢𑀺 𑀔𑀼𑀩𑁆𑀪𑀢𑀺 𑀮𑀼𑀍𑀅𑀢𑀺 𑀆𑀯𑀺𑀮𑀢𑀺 𑀊𑀫𑀺𑀚𑀸𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀤𑀺𑀲𑀸𑀯𑀺𑀤𑀺𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀢𑀺 𑀧𑀢𑀭𑀢𑀺 𑀨𑁂𑀡𑀫𑀸𑀮𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀸𑀓𑀢𑀺𑀓𑀁 𑀉𑀤𑀓𑀁 𑀦 𑀘𑀮𑀢𑀺 𑀲𑀦𑁆𑀢𑀲𑀦𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀅𑀕𑁆𑀕𑀺𑀕𑀢𑀁 𑀘𑀮𑀢𑀺 𑀔𑀼𑀩𑁆𑀪𑀢𑀺 𑀮𑀼𑀍𑀅𑀢𑀺 𑀆𑀯𑀺𑀮𑀢𑀺 𑀊𑀫𑀺𑀚𑀸𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀤𑀺𑀲𑀸𑀯𑀺𑀤𑀺𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀢𑀺 𑀧𑀢𑀭𑀢𑀺 𑀨𑁂𑀡𑀫𑀸𑀮𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀧𑀸𑀓𑀢𑀺𑀓𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀤𑀓𑀁 𑀦 𑀘𑀮𑀢𑀺, 𑀅𑀕𑁆𑀕𑀺𑀓𑀢𑀁 𑀧𑀦 𑀉𑀤𑀓𑀁 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺 𑀩𑀳𑀼𑀯𑀺𑀥𑀦𑁆”𑀢𑀺𑁇 “𑀇𑀫𑀺𑀦𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀚𑀸𑀦𑀸𑀳𑀺 ‘𑀦𑀢𑁆𑀣𑀺 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸, 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻’𑀢𑀺𑁇

1154. 𑀦𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁂𑀭𑀺𑀧𑁄𑀓𑁆𑀔𑀭𑀁 𑀲𑀼𑀓𑁆𑀔𑀁 𑀲𑀼𑀓𑁆𑀔𑁂𑀦 𑀕𑁄𑀘𑀫𑁆𑀫𑁂𑀦 𑀑𑀦𑀦𑁆𑀥𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀅𑀧𑀺 𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁂𑀭𑀺𑀬𑀸 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸 𑀅𑀢𑁆𑀣𑀻”𑀢𑀺𑁇 “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑁂𑀭𑀻 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻”𑀢𑀺? “𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀯𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀯𑀸 𑀢𑀚𑁆𑀚𑁂𑀦 𑀯𑀸𑀬𑀸𑀫𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀯𑀸 𑀢𑀚𑁆𑀚𑁂𑀦 𑀯𑀸𑀬𑀸𑀫𑁂𑀦 𑀪𑁂𑀭𑀻 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺𑁇 𑀇𑀫𑀺𑀦𑀸𑀧𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀚𑀸𑀦𑀸𑀳𑀺 ‘𑀦𑀢𑁆𑀣𑀺 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸, 𑀅𑀕𑁆𑀕𑀺𑀲𑀦𑁆𑀢𑀸𑀧𑀯𑁂𑀕𑀲𑁆𑀲 𑀫𑀳𑀦𑁆𑀢𑀢𑀸𑀬 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻’𑀢𑀺𑁇

1155. 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺 𑀢𑀸𑀯, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀯 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀢𑁆𑀣𑀺, 𑀏𑀯𑀫𑁂𑀲𑁄 𑀧𑀜𑁆𑀳𑁄 𑀲𑀼𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑁂𑀳𑀺𑀧𑀺 𑀪𑀸𑀚𑀦𑁂𑀳𑀺 𑀉𑀤𑀓𑀁 𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀏𑀓𑀘𑁆𑀘𑁂𑀳𑀺𑀬𑁂𑀯 𑀪𑀸𑀚𑀦𑁂𑀳𑀺 𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀩𑁆𑀩𑁂𑀳𑀺𑀧𑀺 𑀪𑀸𑀚𑀦𑁂𑀳𑀺 𑀉𑀤𑀓𑀁 𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂𑀳𑀺𑀬𑁂𑀯 𑀪𑀸𑀚𑀦𑁂𑀳𑀺 𑀉𑀤𑀓𑀁 𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀚𑀳𑀺𑀢𑁄𑀲𑀺 𑀲𑀓𑀲𑀫𑀬𑀁, 𑀧𑀘𑁆𑀘𑀸𑀕𑀢𑁄𑀲𑀺 𑀫𑀫 𑀯𑀺𑀲𑀬𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸𑁇 𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀩𑁆𑀩𑁂𑀳𑀺𑀧𑀺 𑀪𑀸𑀚𑀦𑁂𑀳𑀺 𑀉𑀤𑀓𑀁 𑀢𑀧𑁆𑀧𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑁂𑀬𑁆𑀬, 𑀬𑀼𑀢𑁆𑀢𑀫𑀺𑀤𑀁 ‘𑀉𑀤𑀓𑀁 𑀚𑀻𑀯𑀢𑀻’𑀢𑀺 𑀯𑀢𑁆𑀢𑀼𑀁𑁇 𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀤𑁆𑀯𑀬𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺, 𑀢𑀁 𑀚𑀻𑀯𑀢𑀺, 𑀬𑀁 𑀦 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺, 𑀢𑀁 𑀦 𑀚𑀻𑀯𑀢𑀻𑀢𑀺𑁇 𑀬𑀤𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀉𑀤𑀓𑀁 𑀚𑀻𑀯𑁂𑀬𑁆𑀬, 𑀫𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀦𑁆𑀦𑀓𑀸𑀬𑀸𑀦𑀁 𑀧𑀪𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑁄𑀡𑁆𑀟𑀸𑀬 𑀉𑀲𑁆𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀔𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀘𑁆𑀙𑀺𑀁 𑀧𑀯𑁂𑀲𑀬𑀦𑁆𑀢𑀸𑀦𑀁, 𑀢𑀫𑁆𑀧𑀺 𑀉𑀤𑀓𑀁 𑀢𑁂𑀲𑀁 𑀤𑀦𑁆𑀢𑀦𑁆𑀢𑀭𑁂 𑀘𑀺𑀧𑁆𑀧𑀺𑀬𑀫𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑁂𑀬𑁆𑀬𑁇 𑀳𑀢𑁆𑀣𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀫𑀳𑀸𑀦𑀸𑀯𑀸 𑀕𑀭𑀼𑀓𑀸 𑀪𑀸𑀭𑀺𑀓𑀸 𑀅𑀦𑁂𑀓𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀪𑀸𑀭𑀧𑀭𑀺𑀧𑀽𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀺, 𑀢𑀸𑀳𑀺𑀧𑀺 𑀘𑀺𑀧𑁆𑀧𑀺𑀬𑀫𑀸𑀦𑀁 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑁂𑀬𑁆𑀬𑁇 𑀫𑀳𑀢𑀺𑀫𑀳𑀦𑁆𑀢𑀸𑀧𑀺 𑀫𑀘𑁆𑀙𑀸 𑀅𑀦𑁂𑀓𑀲𑀢𑀬𑁄𑀚𑀦𑀺𑀓𑀓𑀸𑀬𑀸 𑀢𑀺𑀫𑀻 𑀢𑀺𑀫𑀺𑀗𑁆𑀕𑀮𑀸 𑀢𑀺𑀫𑀺𑀭𑀧𑀺𑀗𑁆𑀕𑀮𑀸 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑁂 𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀦𑀺𑀯𑀸𑀲𑀝𑁆𑀞𑀸𑀦𑀢𑀸𑀬 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀸 𑀫𑀳𑀸𑀉𑀤𑀓𑀥𑀸𑀭𑀸 𑀆𑀘𑀫𑀦𑁆𑀢𑀺 𑀥𑀫𑀦𑁆𑀢𑀺 𑀘, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀢𑀁 𑀤𑀦𑁆𑀢𑀦𑁆𑀢𑀭𑁂𑀧𑀺 𑀉𑀤𑀭𑀦𑁆𑀢𑀭𑁂𑀧𑀺 𑀘𑀺𑀧𑁆𑀧𑀺𑀬𑀫𑀸𑀦𑀁 𑀉𑀤𑀓𑀁 𑀲𑀤𑁆𑀤𑀸𑀬𑁂𑀬𑁆𑀬𑁇 𑀬𑀲𑁆𑀫𑀸 𑀘 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀯𑀭𑀽𑀧𑁂𑀳𑀺 𑀏𑀯𑀭𑀽𑀧𑁂𑀳𑀺 𑀫𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀻𑀍𑀅𑀦𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀻𑀍𑀇𑀢𑀁 𑀉𑀤𑀓𑀁 𑀦 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀺 𑀢𑀲𑁆𑀫𑀸𑀧𑀺 𑀦𑀢𑁆𑀣𑀺 𑀉𑀤𑀓𑁂 𑀚𑀻𑀯𑁄 𑀯𑀸 𑀲𑀢𑁆𑀢𑁄 𑀯𑀸𑀢𑀺, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀥𑀸𑀭𑁂𑀳𑀻”𑀢𑀺𑁇

1156. “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀤𑁄𑀲𑀸𑀕𑀢𑁄 𑀧𑀜𑁆𑀳𑁄 𑀅𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀸𑀬 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀪𑀢𑁆𑀢𑁄, 𑀬𑀣𑀸 𑀦𑀸𑀫, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀫𑀳𑀕𑁆𑀖𑀁 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀙𑁂𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀁 𑀓𑀼𑀲𑀮𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀁 𑀫𑀡𑀺𑀓𑀸𑀭𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀓𑀺𑀢𑁆𑀢𑀺𑀁 𑀮𑀪𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑀦𑀁 𑀧𑀲𑀁𑀲𑀁, 𑀫𑀼𑀢𑁆𑀢𑀸𑀭𑀢𑀦𑀁 𑀯𑀸 𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀭𑀢𑀦𑀁 𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀺𑀓𑀁, 𑀮𑁄𑀳𑀺𑀢𑀘𑀦𑁆𑀤𑀦𑀁 𑀯𑀸 𑀕𑀦𑁆𑀥𑀺𑀓𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀓𑀺𑀢𑁆𑀢𑀺𑀁 𑀮𑀪𑁂𑀬𑁆𑀬 𑀣𑁄𑀫𑀦𑀁 𑀧𑀲𑀁𑀲𑀁𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀤𑁄𑀲𑀸𑀕𑀢𑁄 𑀧𑀜𑁆𑀳𑁄 𑀅𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀸𑀬 𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀪𑀢𑁆𑀢𑁄, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀢𑀣𑀸 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀫𑀻”𑀢𑀺𑁇
