# 𑁬𑁇𑁧𑁇𑁬 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑁂𑀤𑀦𑀸𑀯𑁂𑀤𑀺𑀬𑀦𑀧𑀜𑁆𑀳

* 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

* 𑀩𑀼𑀤𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀅𑀭𑀳𑀦𑁆𑀢𑀯𑁂𑀤𑀦𑀸𑀯𑁂𑀤𑀺𑀬𑀦𑀧𑀜𑁆𑀳

(176.)

1120\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀼𑀫𑁆𑀳𑁂 𑀪𑀡𑀣—  ‘𑀅𑀭𑀳𑀸 𑀏𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺 𑀓𑀸𑀬𑀺𑀓𑀁, 𑀦 𑀘𑁂𑀢𑀲𑀺𑀓𑀦𑁆’𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀭𑀳𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀬𑀁 𑀓𑀸𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀅𑀭𑀳𑀸 𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑁄 𑀅𑀲𑁆𑀲𑀸𑀫𑀻 𑀅𑀯𑀲𑀯𑀢𑁆𑀢𑀻”𑀢𑀺? “𑀆𑀫, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀦 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀼𑀢𑁆𑀢𑀫𑁂𑀢𑀁, 𑀬𑀁 𑀲𑁄 𑀲𑀓𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂 𑀓𑀸𑀬𑁂 𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑁆𑀲𑀸𑀫𑀻 𑀅𑀯𑀲𑀯𑀢𑁆𑀢𑀻; 𑀲𑀓𑀼𑀡𑁄𑀧𑀺 𑀢𑀸𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑁆𑀫𑀺𑀁 𑀓𑀼𑀮𑀸𑀯𑀓𑁂 𑀧𑀝𑀺𑀯𑀲𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀲𑁄 𑀇𑀲𑁆𑀲𑀭𑁄 𑀳𑁄𑀢𑀺 𑀲𑀸𑀫𑀻 𑀯𑀲𑀯𑀢𑁆𑀢𑀻”𑀢𑀺𑁇

1121\. “𑀤𑀲𑀬𑀺𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀬𑀸𑀦𑀼𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀯𑁂 𑀪𑀯𑁂 𑀓𑀸𑀬𑀁 𑀅𑀦𑀼𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀤𑀲? 𑀲𑀻𑀢𑀁 𑀉𑀡𑁆𑀳𑀁 𑀚𑀺𑀖𑀘𑁆𑀙𑀸 𑀧𑀺𑀧𑀸𑀲𑀸 𑀉𑀘𑁆𑀘𑀸𑀭𑁄 𑀧𑀲𑁆𑀲𑀸𑀯𑁄 𑀫𑀺𑀤𑁆𑀥𑀁 𑀚𑀭𑀸 𑀩𑁆𑀬𑀸𑀥𑀺 𑀫𑀭𑀡𑀁𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀲 𑀓𑀸𑀬𑀸𑀦𑀼𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀪𑀯𑁂 𑀪𑀯𑁂 𑀓𑀸𑀬𑀁 𑀅𑀦𑀼𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀅𑀭𑀳𑀸 𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑁄 𑀅𑀲𑁆𑀲𑀸𑀫𑀻 𑀅𑀯𑀲𑀯𑀢𑁆𑀢𑀻”𑀢𑀺𑁇

1122\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀭𑀳𑀢𑁄 𑀓𑀸𑀬𑁂 𑀆𑀡𑀸 𑀦𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀢𑀺 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀯𑀸, 𑀢𑀢𑁆𑀣 𑀫𑁂 𑀓𑀸𑀭𑀡𑀁 𑀩𑁆𑀭𑀽𑀳𑀻”𑀢𑀺? “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀧𑀣𑀯𑀺𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀢𑁆𑀢𑀸, 𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀧𑀣𑀯𑀺𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀺𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀲𑀁 𑀧𑀣𑀯𑀺𑀬𑀸 𑀆𑀡𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀺 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀯𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀯𑀢𑁆𑀢𑀢𑀺, 𑀦 𑀘 𑀅𑀭𑀳𑀢𑁄 𑀓𑀸𑀬𑁂 𑀆𑀡𑀸 𑀧𑀯𑀢𑁆𑀢𑀢𑀺 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀯𑀸”𑀢𑀺𑁇

1123\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀺𑀓𑀫𑁆𑀧𑀺 𑀘𑁂𑀢𑀲𑀺𑀓𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺? “𑀅𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀺𑀓𑀫𑁆𑀧𑀺 𑀘𑁂𑀢𑀲𑀺𑀓𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀕𑁄𑀡𑁄 𑀙𑀸𑀢𑁄 𑀧𑀭𑀺𑀢𑀲𑀺𑀢𑁄 𑀅𑀩𑀮𑀤𑀼𑀩𑁆𑀩𑀮𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑀢𑀺𑀡𑁂𑀲𑀼 𑀯𑀸 𑀮𑀢𑀸𑀬 𑀯𑀸 𑀉𑀧𑀦𑀺𑀩𑀤𑁆𑀥𑁄 𑀅𑀲𑁆𑀲, 𑀬𑀤𑀸 𑀲𑁄 𑀕𑁄𑀡𑁄 𑀧𑀭𑀺𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀢𑀤𑀸 𑀲𑀳 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀪𑀸𑀯𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁂𑀤𑀦𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀓𑁄𑀧𑁂𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀓𑀼𑀧𑀺𑀢𑀁 𑀓𑀸𑀬𑀁 𑀆𑀪𑀼𑀚𑀢𑀺 𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀅𑀪𑀸𑀯𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀢𑀲𑀢𑀺 𑀭𑀯𑀢𑀺 𑀪𑁂𑀭𑀯𑀭𑀸𑀯𑀫𑀪𑀺𑀭𑀯𑀢𑀺, 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀬𑀺𑀓𑀫𑁆𑀧𑀺 𑀘𑁂𑀢𑀲𑀺𑀓𑀫𑁆𑀧𑀺 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀻”𑀢𑀺𑁇

1124\. “𑀓𑀺𑀁 𑀧𑀦 𑀢𑀁 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀭𑀳𑀸 𑀏𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺 𑀓𑀸𑀬𑀺𑀓𑀁, 𑀦 𑀘𑁂𑀢𑀲𑀺𑀓𑀦𑁆”𑀢𑀺? “𑀅𑀭𑀳𑀢𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀤𑀦𑁆𑀢𑀁 𑀲𑀼𑀤𑀦𑁆𑀢𑀁 𑀅𑀲𑁆𑀲𑀯𑀁 𑀯𑀘𑀦𑀓𑀭𑀁, 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆’𑀢𑀺 𑀤𑀍𑀳𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀣𑀫𑁆𑀪𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀣𑀫𑁆𑀪𑁂 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀦 𑀯𑁂𑀥𑀢𑀺 𑀦 𑀘𑀮𑀢𑀺, 𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁, 𑀢𑀲𑁆𑀲 𑀯𑁂𑀤𑀦𑀸𑀯𑀺𑀓𑀸𑀭𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑁂𑀦 𑀓𑀸𑀬𑁄 𑀆𑀪𑀼𑀚𑀢𑀺 𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀢𑀺, 𑀇𑀤𑀫𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀭𑀳𑀸 𑀏𑀓𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑀬𑀢𑀺 𑀓𑀸𑀬𑀺𑀓𑀁, 𑀦 𑀘𑁂𑀢𑀲𑀺𑀓𑀦𑁆”𑀢𑀺𑁇

1125\. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀁 𑀦𑀸𑀫 𑀮𑁄𑀓𑁂 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀬𑀁 𑀓𑀸𑀬𑁂 𑀘𑀮𑀫𑀸𑀦𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀘𑀮𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀫𑁂 𑀓𑀸𑀭𑀡𑀁 𑀩𑁆𑀭𑀽𑀳𑀻”𑀢𑀺𑁇 “𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀳𑀢𑀺𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑁂 𑀔𑀦𑁆𑀥𑀲𑀸𑀔𑀸𑀧𑀮𑀸𑀲𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑀺𑀮𑀩𑀮𑀲𑀫𑀸𑀳𑀢𑁂 𑀲𑀸𑀔𑀸 𑀘𑀮𑀢𑀺, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑀲𑁆𑀲 𑀔𑀦𑁆𑀥𑁄𑀧𑀺 𑀘𑀮𑀢𑀻”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀨𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀦𑀺𑀘𑁆𑀘𑀦𑁆’𑀢𑀺 𑀤𑀍𑀳𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀲𑀫𑀸𑀥𑀺𑀣𑀫𑁆𑀪𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀣𑀫𑁆𑀪𑁂 𑀉𑀧𑀦𑀺𑀩𑀦𑁆𑀥𑀦𑀁 𑀦 𑀯𑁂𑀥𑀢𑀺 𑀦 𑀘𑀮𑀢𑀺, 𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁, 𑀢𑀲𑁆𑀲 𑀯𑁂𑀤𑀦𑀸𑀯𑀺𑀓𑀸𑀭𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑁂𑀦 𑀓𑀸𑀬𑁄 𑀆𑀪𑀼𑀚𑀢𑀺 𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀦 𑀢𑀲𑁆𑀲 𑀦 𑀯𑁂𑀥𑀢𑀺 𑀦 𑀘𑀮𑀢𑀺 𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀬 𑀫𑀳𑀸𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦 𑀫𑁂 𑀏𑀯𑀭𑀽𑀧𑁄 𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀧𑀤𑀻𑀧𑁄 𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
