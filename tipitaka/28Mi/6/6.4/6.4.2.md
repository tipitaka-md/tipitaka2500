# 𑁬𑁇𑁪𑁇𑁨 𑀥𑀼𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀥𑀼𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳

(211.)

1477. _“𑀧𑀲𑁆𑀲𑀢𑀸𑀭𑀜𑁆𑀜𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽,_  
_𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀍𑀳𑁂 𑀥𑀼𑀢𑁂 𑀕𑀼𑀡𑁂;_  
_𑀧𑀼𑀦 𑀧𑀲𑁆𑀲𑀢𑀺 𑀕𑀺𑀳𑀻 𑀭𑀸𑀚𑀸,_  
_𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑁂 𑀞𑀺𑀢𑁂𑁇_  


1478. _𑀉𑀪𑁄𑀧𑀺 𑀢𑁂 𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀲𑀁𑀲𑀬𑁄 𑀫𑀳𑀸;_  
_𑀩𑀼𑀚𑁆𑀛𑁂𑀬𑁆𑀬 𑀘𑁂 𑀕𑀺𑀳𑀻 𑀥𑀫𑁆𑀫𑁂,_  
_𑀥𑀼𑀢𑀗𑁆𑀕𑀁 𑀦𑀺𑀧𑁆𑀨𑀮𑀁 𑀲𑀺𑀬𑀸𑁇_  


1479. _𑀧𑀭𑀯𑀸𑀤𑀺𑀯𑀸𑀤𑀫𑀣𑀦𑀁,_  
_𑀦𑀺𑀧𑀼𑀡𑀁 𑀧𑀺𑀝𑀓𑀢𑁆𑀢𑀬𑁂;_  
_𑀳𑀦𑁆𑀤 𑀧𑀼𑀘𑁆𑀙𑁂 𑀓𑀣𑀺𑀲𑁂𑀝𑁆𑀞𑀁,_  
_𑀲𑁄 𑀫𑁂 𑀓𑀗𑁆𑀔𑀁 𑀯𑀺𑀦𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


1480. 𑀅𑀣 𑀔𑁄 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑁄 𑀭𑀸𑀚𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑁂𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀲𑁂𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑁄 𑀭𑀸𑀚𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀲𑁂𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀢𑁆𑀣𑀺 𑀓𑁄𑀘𑀺 𑀕𑀺𑀳𑀻 𑀅𑀕𑀸𑀭𑀺𑀓𑁄 𑀓𑀸𑀫𑀪𑁄𑀕𑀻 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀬𑀦𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀦𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀓𑀘𑀦𑁆𑀤𑀦𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑁄 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑁄 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁄 𑀫𑀡𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀓𑀜𑁆𑀘𑀦𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀫𑁄𑀍𑀇𑀩𑀤𑁆𑀥𑁄 𑀬𑁂𑀦 𑀲𑀦𑁆𑀢𑀁 𑀧𑀭𑀫𑀢𑁆𑀣𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀦𑁆”𑀢𑀺?

1481. “𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀓𑀜𑁆𑀜𑁂𑀯 𑀲𑀢𑀁 𑀦 𑀤𑁆𑀯𑁂 𑀲𑀢𑀸𑀦𑀺 𑀦 𑀢𑀻𑀡𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀜𑁆𑀘 𑀲𑀢𑀸𑀦𑀺 𑀦 𑀲𑀳𑀲𑁆𑀲𑀁 𑀦 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀦 𑀓𑁄𑀝𑀺𑀲𑀢𑀁 𑀦 𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀁 𑀦 𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀲𑀦𑁆𑀦𑀁 𑀯𑀻𑀲𑀢𑀺𑀬𑀸 𑀲𑀢𑀲𑁆𑀲 𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀪𑀺𑀲𑀫𑀬𑁄, 𑀓𑀢𑀫𑁂𑀦 𑀢𑁂 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀦𑀼𑀬𑁄𑀕𑀁 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺𑁇

1482. “𑀢𑁆𑀯𑀫𑁂𑀯𑁂𑀢𑀁 𑀩𑁆𑀭𑀽𑀳𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀢𑁂𑀦 𑀯𑀸 𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀯𑀸 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀯𑀸 𑀓𑁄𑀝𑀺𑀬𑀸 𑀯𑀸 𑀓𑁄𑀝𑀺𑀲𑀢𑁂𑀦 𑀯𑀸 𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀯𑀸 𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀯𑀸, 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀦𑀯𑀗𑁆𑀕𑁂 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀺𑀢𑀸𑀘𑀸𑀭𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀥𑀼𑀢𑀯𑀭𑀗𑁆𑀕𑀕𑀼𑀡𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀓𑀣𑀸, 𑀢𑀸 𑀲𑀩𑁆𑀩𑀸 𑀇𑀥 𑀲𑀫𑁄𑀲𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀦𑀺𑀦𑁆𑀦𑀼𑀦𑁆𑀦𑀢𑀲𑀫𑀯𑀺𑀲𑀫𑀣𑀮𑀸𑀣𑀮𑀤𑁂𑀲𑀪𑀸𑀕𑁂 𑀅𑀪𑀺𑀯𑀼𑀝𑁆𑀞𑀁 𑀉𑀤𑀓𑀁, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀢𑀢𑁄 𑀯𑀺𑀦𑀺𑀕𑀍𑀇𑀢𑁆𑀯𑀸 𑀫𑀳𑁄𑀤𑀥𑀺𑀁 𑀲𑀸𑀕𑀭𑀁 𑀲𑀫𑁄𑀲𑀭𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑁆𑀧𑀸𑀤𑀓𑁂 𑀲𑀢𑀺 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀦𑀯𑀗𑁆𑀕𑁂 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑁂 𑀲𑀮𑁆𑀮𑁂𑀔𑀺𑀢𑀸𑀘𑀸𑀭𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀥𑀼𑀢𑀗𑁆𑀕𑀕𑀼𑀡𑀥𑀭𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀓𑀣𑀸, 𑀢𑀸 𑀲𑀩𑁆𑀩𑀸 𑀇𑀥 𑀲𑀫𑁄𑀲𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

1483. 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑁂𑀢𑁆𑀣, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀭𑀺𑀩𑁆𑀬𑀢𑁆𑀢𑀢𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀓𑀸𑀭𑀡𑀧𑀭𑀺𑀤𑀻𑀧𑀦𑀁 𑀲𑀫𑁄𑀲𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑁂𑀦𑁂𑀲𑁄 𑀅𑀢𑁆𑀣𑁄 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑁄 𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀲𑀫𑀸𑀦𑀻𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀼𑀲𑀮𑁄 𑀮𑁂𑀔𑀸𑀘𑀭𑀺𑀬𑁄 𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑁄 𑀮𑁂𑀔𑀁 𑀑𑀲𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑀢𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀓𑀸𑀭𑀡𑀧𑀭𑀺𑀤𑀻𑀧𑀦𑁂𑀦 𑀮𑁂𑀔𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺, 𑀏𑀯𑀁 𑀲𑀸 𑀮𑁂𑀔𑀸 𑀲𑀫𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀸 𑀅𑀦𑀽𑀦𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑁂𑀢𑁆𑀣 𑀧𑀭𑀺𑀩𑁆𑀬𑀢𑁆𑀢𑀢𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀓𑀸𑀭𑀡𑀧𑀭𑀺𑀤𑀻𑀧𑀦𑀁 𑀲𑀫𑁄𑀲𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑁂𑀦𑁂𑀲𑁄 𑀅𑀢𑁆𑀣𑁄 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑁄 𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀲𑀫𑀸𑀦𑀻𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1525. 𑀬𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀓𑀼𑀳𑀓𑁄 𑀮𑀼𑀤𑁆𑀥𑁄 𑀑𑀤𑀭𑀺𑀓𑁄 𑀮𑀸𑀪𑀓𑀸𑀫𑁄 𑀬𑀲𑀓𑀸𑀫𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀓𑀸𑀫𑁄 𑀅𑀬𑀼𑀢𑁆𑀢𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑁄 𑀅𑀦𑀭𑀳𑁄 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑁄 𑀥𑀼𑀢𑀗𑁆𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺, 𑀲𑁄 𑀤𑀺𑀕𑀼𑀡𑀁 𑀤𑀡𑁆𑀟𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑀩𑁆𑀩𑀕𑀼𑀡𑀖𑀸𑀢𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀳𑀻𑀍𑀅𑀦𑀁 𑀔𑀻𑀍𑀅𑀦𑀁 𑀕𑀭𑀳𑀦𑀁 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀦𑀁 𑀔𑀺𑀧𑀦𑀁 𑀅𑀲𑀫𑁆𑀪𑁄𑀕𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁 𑀦𑀺𑀘𑁆𑀙𑀼𑀪𑀦𑀁 𑀧𑀯𑀸𑀳𑀦𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀁 𑀧𑀝𑀺𑀮𑀪𑀢𑀺, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁂𑀧𑀺 𑀲𑀢𑀬𑁄𑀚𑀦𑀺𑀓𑁂 𑀅𑀯𑀻𑀘𑀺𑀫𑀳𑀸𑀦𑀺𑀭𑀬𑁂 𑀉𑀡𑁆𑀳𑀓𑀞𑀺𑀢𑀢𑀢𑁆𑀢𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀅𑀘𑁆𑀘𑀺𑀚𑀸𑀮𑀸𑀫𑀸𑀮𑀓𑁂 𑀅𑀦𑁂𑀓𑀯𑀲𑁆𑀲𑀓𑁄𑀝𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀉𑀤𑁆𑀥𑀫𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀨𑁂𑀡𑀼𑀤𑁆𑀤𑁂𑀳𑀓𑀁 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁 𑀧𑀘𑁆𑀘𑀢𑀺, 𑀢𑀢𑁄 𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀓𑀺𑀲𑀨𑀭𑀼𑀲𑀓𑀸𑀍𑀅𑀗𑁆𑀕𑀧𑀘𑁆𑀘𑀗𑁆𑀕𑁄 𑀲𑀽𑀦𑀼𑀤𑁆𑀥𑀼𑀫𑀸𑀢𑀲𑀼𑀲𑀺𑀭𑀼𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑁄 𑀙𑀸𑀢𑁄 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄 𑀯𑀺𑀲𑀫𑀪𑀻𑀫𑀭𑀽𑀧𑀯𑀡𑁆𑀡𑁄 𑀪𑀕𑁆𑀕𑀓𑀡𑁆𑀡𑀲𑁄𑀢𑁄 𑀉𑀫𑁆𑀫𑀻𑀮𑀺𑀢𑀦𑀺𑀫𑀻𑀮𑀺𑀢𑀦𑁂𑀢𑁆𑀢𑀦𑀬𑀦𑁄 𑀅𑀭𑀼𑀕𑀢𑁆𑀢𑀧𑀓𑁆𑀓𑀕𑀢𑁆𑀢𑁄 𑀧𑀼𑀮𑀯𑀸𑀓𑀺𑀡𑁆𑀡𑀲𑀩𑁆𑀩𑀓𑀸𑀬𑁄 𑀯𑀸𑀢𑀫𑀼𑀔𑁂 𑀚𑀮𑀫𑀸𑀦𑁄 𑀯𑀺𑀬 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀅𑀦𑁆𑀢𑁄 𑀚𑀮𑀫𑀸𑀦𑁄 𑀧𑀚𑁆𑀚𑀮𑀫𑀸𑀦𑁄 𑀅𑀢𑀸𑀡𑁄 𑀅𑀲𑀭𑀡𑁄 𑀆𑀭𑀼𑀡𑁆𑀡𑀭𑀼𑀡𑁆𑀡𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑀭𑀯𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀫𑀸𑀦𑁄 𑀦𑀺𑀚𑁆𑀛𑀸𑀫𑀢𑀡𑁆𑀳𑀺𑀓𑁄 𑀲𑀫𑀡𑀫𑀳𑀸𑀧𑁂𑀢𑁄 𑀳𑀼𑀢𑁆𑀯𑀸 𑀆𑀳𑀺𑀡𑁆𑀟𑀫𑀸𑀦𑁄 𑀫𑀳𑀺𑀬𑀸 𑀅𑀝𑁆𑀝𑀲𑁆𑀲𑀭𑀁 𑀓𑀭𑁄𑀢𑀺𑁇

1535. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀤𑀼𑀫𑀁 𑀅𑀪𑀺𑀯𑀼𑀤𑁆𑀥𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁅𑀤𑀺𑀘𑁆𑀘𑀚𑀸𑀢𑀺𑀧𑁆𑀧𑀪𑀯𑀁 𑀲𑀺𑀦𑀺𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑀼 𑀮𑁄𑀪𑀦𑀻𑀬𑀁 𑀲𑀼𑀕𑀦𑁆𑀥𑀁 𑀧𑀺𑀬𑀁 𑀧𑀢𑁆𑀣𑀺𑀢𑀁 𑀧𑀲𑀢𑁆𑀣𑀁 𑀚𑀮𑀓𑀤𑁆𑀤𑀫𑀫𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑀁 𑀅𑀡𑀼𑀧𑀢𑁆𑀢𑀓𑁂𑀲𑀭𑀓𑀡𑁆𑀡𑀺𑀓𑀸𑀪𑀺𑀫𑀡𑁆𑀟𑀺𑀢𑀁 𑀪𑀫𑀭𑀕𑀡𑀲𑁂𑀯𑀺𑀢𑀁 𑀲𑀻𑀢𑀮𑀲𑀮𑀺𑀮𑀲𑀁𑀯𑀤𑁆𑀥𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀇𑀫𑁂𑀳𑀺 𑀢𑁂𑀭𑀲𑀳𑀺 𑀥𑀼𑀢𑀕𑀼𑀡𑁂𑀳𑀺 𑀧𑀼𑀩𑁆𑀩𑁂 𑀆𑀲𑁂𑀯𑀺𑀢𑁂𑀳𑀺 𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑁂𑀳𑀺 𑀘𑀺𑀡𑁆𑀡𑁂𑀳𑀺 𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁂𑀳𑀺 𑀘𑀭𑀺𑀢𑁂𑀳𑀺 𑀉𑀧𑀘𑀭𑀺𑀢𑁂𑀳𑀺 𑀧𑀭𑀺𑀧𑀽𑀭𑀺𑀢𑁂𑀳𑀺 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀢𑀺𑀁𑀲𑀕𑀼𑀡𑀯𑀭𑁂𑀳𑀺 𑀲𑀫𑀼𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

(222.)

1536. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀺𑀁𑀲𑀕𑀼𑀡𑀯𑀭𑁂𑀳𑀺? 𑀲𑀺𑀦𑀺𑀤𑁆𑀥𑀫𑀼𑀤𑀼𑀫𑀤𑁆𑀤𑀯𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀖𑀸𑀢𑀺𑀢𑀳𑀢𑀯𑀺𑀳𑀢𑀓𑀺𑀮𑁂𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀳𑀢𑀦𑀺𑀳𑀢𑀫𑀸𑀦𑀤𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀘𑀮𑀤𑀍𑀳𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀦𑀺𑀩𑁆𑀩𑁂𑀫𑀢𑀺𑀓𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀧𑀻𑀡𑀺𑀢𑀧𑀳𑀝𑁆𑀞𑀮𑁄𑀪𑀦𑀻𑀬𑀲𑀦𑁆𑀢𑀲𑀼𑀔𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀻𑀮𑀯𑀭𑀧𑀯𑀭𑀅𑀲𑀫𑀲𑀼𑀘𑀺𑀕𑀦𑁆𑀥𑀧𑀭𑀺𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀅𑀭𑀺𑀬𑀯𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀢𑁆𑀣𑀺𑀢𑁄, 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀯𑀦𑁆𑀤𑀺𑀢𑀧𑀽𑀚𑀺𑀢𑁄, 𑀩𑀼𑀥𑀯𑀺𑀩𑀼𑀥𑀧𑀡𑁆𑀟𑀺𑀢𑀚𑀦𑀸𑀦𑀁 𑀣𑀼𑀢𑀣𑀯𑀺𑀢𑀣𑁄𑀫𑀺𑀢𑀧𑀲𑀢𑁆𑀣𑁄, 𑀇𑀥 𑀯𑀸 𑀳𑀼𑀭𑀁 𑀯𑀸 𑀮𑁄𑀓𑁂𑀦 𑀅𑀦𑀼𑀧𑀮𑀺𑀢𑁆𑀢𑁄, 𑀅𑀧𑁆𑀧𑀣𑁄𑀓𑀯𑀚𑁆𑀚𑁂𑀧𑀺 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻, 𑀯𑀺𑀧𑀼𑀮𑀯𑀭𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀫𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀨𑀮𑀯𑀭𑀢𑁆𑀣𑀲𑀸𑀥𑀦𑁄, 𑀆𑀬𑀸𑀘𑀺𑀢𑀯𑀺𑀧𑀼𑀮𑀧𑀡𑀻𑀢𑀧𑀘𑁆𑀘𑀬𑀪𑀸𑀕𑀻, 𑀅𑀦𑀺𑀓𑁂𑀢𑀲𑀬𑀦𑁄, 𑀛𑀸𑀦𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀢𑀧𑁆𑀧𑀯𑀭𑀯𑀺𑀳𑀸𑀭𑀻, 𑀯𑀺𑀚𑀝𑀺𑀢𑀓𑀺𑀮𑁂𑀲𑀚𑀸𑀮𑀯𑀢𑁆𑀣𑀼, 𑀪𑀺𑀦𑁆𑀦𑀪𑀕𑁆𑀕𑀲𑀗𑁆𑀓𑀼𑀝𑀺𑀢𑀲𑀜𑁆𑀙𑀺𑀦𑁆𑀦𑀕𑀢𑀺𑀦𑀻𑀯𑀭𑀡𑁄, 𑀅𑀓𑀼𑀧𑁆𑀧𑀥𑀫𑁆𑀫𑁄, 𑀅𑀪𑀺𑀦𑀻𑀢𑀯𑀸𑀲𑁄, 𑀅𑀦𑀯𑀚𑁆𑀚𑀪𑁄𑀕𑀻, 𑀕𑀢𑀺𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄, 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀲𑀩𑁆𑀩𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀚𑁆𑀛𑁄𑀲𑀺𑀢𑀢𑁆𑀣𑁄, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁄, 𑀅𑀘𑀮𑀤𑀍𑀳𑀪𑀻𑀭𑀼𑀢𑁆𑀢𑀸𑀡𑀫𑀼𑀧𑀕𑀢𑁄, 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀼𑀲𑀬𑁄, 𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀓𑁆𑀔𑀬𑀁 𑀧𑀢𑁆𑀢𑁄, 𑀲𑀦𑁆𑀢𑀲𑀼𑀔𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀺𑀳𑀸𑀭𑀩𑀳𑀼𑀮𑁄, 𑀲𑀩𑁆𑀩𑀲𑀫𑀡𑀕𑀼𑀡𑀲𑀫𑀼𑀧𑁂𑀢𑁄, 𑀇𑀫𑁂𑀳𑀺 𑀢𑀺𑀁𑀲𑀕𑀼𑀡𑀯𑀭𑁂𑀳𑀺 𑀲𑀫𑀼𑀧𑁂𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

(223.)

1537. 𑀦𑀦𑀼, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀣𑁂𑀭𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀺𑀮𑁄𑀓𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀕𑁆𑀕𑀧𑀼𑀭𑀺𑀲𑁄 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑀲𑀩𑀮𑀁 𑀮𑁄𑀓𑀸𑀘𑀭𑀺𑀬𑀁, 𑀲𑁄𑀧𑀺 𑀅𑀧𑀭𑀺𑀫𑀺𑀢𑀫𑀲𑀗𑁆𑀔𑁆𑀬𑁂𑀬𑁆𑀬𑀓𑀧𑁆𑀧𑁂 𑀲𑀫𑀸𑀘𑀺𑀢𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀓𑀼𑀮𑀻𑀦𑁄 𑀫𑀦𑀸𑀧𑀺𑀓𑀁 𑀓𑀸𑀫𑀭𑀢𑀺𑀁 𑀅𑀦𑁂𑀓𑀲𑀢𑀲𑀗𑁆𑀔𑀁 𑀥𑀦𑀯𑀭𑀜𑁆𑀘 𑀑𑀳𑀸𑀬 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀇𑀫𑁂𑀳𑀺 𑀢𑁂𑀭𑀲𑀳𑀺 𑀥𑀼𑀢𑀕𑀼𑀡𑁂𑀳𑀺 𑀓𑀸𑀬𑀯𑀘𑀻𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀫𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀚𑁂𑀢𑀭𑀳𑀺 𑀅𑀦𑀦𑁆𑀢𑀕𑀼𑀡𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀕𑁄𑀢𑀫𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀲𑀦𑀯𑀭𑁂 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀫𑀦𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀓𑁄 𑀚𑀸𑀢𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 𑀏𑀓𑀗𑁆𑀕𑀼𑀢𑁆𑀢𑀭𑀦𑀺𑀓𑀸𑀬𑀯𑀭𑀮𑀜𑁆𑀙𑀓𑁂—

1538. ‘𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀁 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑀫𑁆𑀧𑀺 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀬𑁄 𑀏𑀯𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀦𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺, 𑀬𑀣𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑀺𑀢𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀦𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇

1539. “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀦𑀯𑀗𑁆𑀕𑀁 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑀁, 𑀬𑀸 𑀘 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀸 𑀓𑀺𑀭𑀺𑀬𑀸, 𑀬𑀸 𑀘 𑀮𑁄𑀓𑁂 𑀅𑀥𑀺𑀕𑀫𑀯𑀺𑀧𑀼𑀮𑀯𑀭𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀢𑁂𑀭𑀲𑀲𑀼 𑀥𑀼𑀢𑀕𑀼𑀡𑁂𑀲𑀼 𑀲𑀫𑁄𑀥𑀸𑀦𑁄𑀧𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

---

1540. 𑀥𑀼𑀢𑀗𑁆𑀕𑀧𑀜𑁆𑀳𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
