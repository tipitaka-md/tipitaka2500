# 𑁬𑁇𑁪𑁇𑁧 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳

(202.)

1382. 𑀅𑀣 𑀔𑁄 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑁄 𑀭𑀸𑀚𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑁂𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀲𑁂𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑁄 𑀭𑀸𑀚𑀸 𑀜𑀸𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑁄𑀢𑀼𑀓𑀸𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀜𑀸𑀡𑀸𑀮𑁄𑀓𑀁 𑀤𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑁄 𑀅𑀜𑁆𑀜𑀸𑀡𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀜𑀸𑀡𑀸𑀮𑁄𑀓𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀥𑀓𑀸𑀭𑀁 𑀦𑀸𑀲𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀥𑀺𑀢𑀺𑀜𑁆𑀘 𑀉𑀲𑁆𑀲𑀸𑀳𑀜𑁆𑀘 𑀲𑀢𑀺𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀲𑁂𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀺𑀁 𑀧𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀢𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑁄”𑀢𑀺𑁇 “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁂 𑀆𑀘𑀭𑀺𑀬𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑀺𑀝𑁆𑀞𑁄”𑀢𑀺? “𑀦 𑀳𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦 𑀓𑀺𑀭 𑀢𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀦𑀸𑀧𑀺 𑀓𑀺𑀭 𑀢𑁂 𑀆𑀘𑀭𑀺𑀬𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀤𑀺𑀝𑁆𑀞𑁄𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀦𑀢𑁆𑀣𑀺 𑀩𑀼𑀤𑁆𑀥𑁄, 𑀦 𑀳𑁂𑀢𑁆𑀣 𑀩𑀼𑀤𑁆𑀥𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀻”𑀢𑀺𑁇

1383. “𑀅𑀢𑁆𑀣𑀺 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀬𑁂 𑀢𑁂 𑀢𑀯 𑀔𑀢𑁆𑀢𑀺𑀬𑀯𑀁𑀲𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸”𑀢𑀺? “𑀆𑀫, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀓𑁄 𑀲𑀁𑀲𑀬𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀬𑁂 𑀫𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑀯𑀁𑀲𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀫𑀸”𑀢𑀺𑁇 “𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀸 𑀢𑀬𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀬𑁂 𑀧𑀦 𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀸 𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑁄 𑀅𑀓𑁆𑀔𑀤𑀲𑁆𑀲𑀸 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸, 𑀢𑁂𑀳𑀺 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀸”𑀢𑀺? “𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀬𑀤𑀺 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀤𑀺𑀝𑁆𑀞𑀸, 𑀦𑀸𑀧𑀺 𑀓𑀺𑀭 𑀢𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀓𑁂𑀳𑀺 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀸, 𑀢𑁂𑀦 𑀳𑀺 𑀦𑀢𑁆𑀣𑀺 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀦 𑀳𑁂𑀢𑁆𑀣 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1384. “𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀅𑀦𑀼𑀪𑀽𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀪𑁄𑀕𑀪𑀡𑁆𑀟𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀉𑀡𑁆𑀳𑀻𑀲𑀁 𑀧𑀸𑀤𑀼𑀓𑀸 𑀯𑀸𑀮𑀩𑀻𑀚𑀦𑀻 𑀔𑀕𑁆𑀕𑀭𑀢𑀦𑀁 𑀫𑀳𑀸𑀭𑀳𑀸𑀦𑀺 𑀘 𑀲𑀬𑀦𑀸𑀦𑀺𑁇 𑀬𑁂𑀳𑀺 𑀫𑀬𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀸𑀫 ‘𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀩𑁆𑀩𑀓𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀸’”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀫𑀬𑀫𑁆𑀧𑁂𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀸𑀫𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀁 𑀓𑀸𑀭𑀡𑀁, 𑀬𑁂𑀦 𑀫𑀬𑀁 𑀓𑀸𑀭𑀡𑁂𑀦 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫 𑀲𑀤𑁆𑀤𑀳𑁂𑀬𑁆𑀬𑀸𑀫 ‘𑀅𑀢𑁆𑀣𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀓𑀸𑀭𑀡𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀦𑀼𑀪𑀽𑀢𑀸𑀦𑀺 𑀧𑀭𑀺𑀪𑁄𑀕𑀪𑀡𑁆𑀟𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀬𑁂𑀳𑀺 𑀲𑀤𑁂𑀯𑀓𑁄 𑀮𑁄𑀓𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 ‘𑀅𑀢𑁆𑀣𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀇𑀫𑀺𑀦𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀇𑀫𑀺𑀦𑀸 𑀳𑁂𑀢𑀼𑀦𑀸 𑀇𑀫𑀺𑀦𑀸 𑀦𑀬𑁂𑀦 𑀇𑀫𑀺𑀦𑀸 𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑁄 ‘𑀅𑀢𑁆𑀣𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇

1385. _‘𑀩𑀳𑀽 𑀚𑀦𑁂 𑀢𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄 𑀉𑀧𑀥𑀺𑀓𑁆𑀔𑀬𑁂;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀅𑀢𑁆𑀣𑀺 𑀲𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑁄’”𑀢𑀺𑁇_

1388. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀧𑁆𑀨𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀆𑀭𑀫𑁆𑀫𑀡𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀸 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸 𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸 𑀆𑀤𑀻𑀦𑀯𑀲𑀜𑁆𑀜𑀸 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀸 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂 𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀲𑀜𑁆𑀜𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂𑀲𑀼 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸 𑀆𑀦𑀸𑀧𑀸𑀦𑀲𑁆𑀲𑀢𑀺 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀦𑀻𑀮𑀓𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑀼𑀩𑁆𑀩𑀓𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀓𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀓𑁆𑀔𑀸𑀬𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀸 𑀳𑀢𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑀜𑁆𑀜𑀸 𑀮𑁄𑀳𑀺𑀢𑀓𑀲𑀜𑁆𑀜𑀸 𑀧𑀼𑀍𑀅𑀯𑀓𑀲𑀜𑁆𑀜𑀸 𑀅𑀝𑁆𑀞𑀺𑀓𑀲𑀜𑁆𑀜𑀸 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀜𑁆𑀜𑀸 𑀓𑀭𑀼𑀡𑀸𑀲𑀜𑁆𑀜𑀸 𑀫𑀼𑀤𑀺𑀢𑀸𑀲𑀜𑁆𑀜𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀜𑁆𑀜𑀸 𑀫𑀭𑀡𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺, 𑀇𑀫𑁂 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀚𑀭𑀸𑀫𑀭𑀡𑀸 𑀫𑀼𑀘𑁆𑀘𑀺𑀢𑀼𑀓𑀸𑀫𑁄, 𑀲𑁄 𑀢𑁂𑀲𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀢𑁂𑀦 𑀆𑀭𑀫𑁆𑀫𑀡𑁂𑀦 𑀭𑀸𑀕𑀸 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀤𑁄𑀲𑀸 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑁄𑀳𑀸 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀸𑀦𑀢𑁄 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀲𑀁𑀲𑀸𑀭𑀁 𑀢𑀭𑀢𑀺, 𑀢𑀡𑁆𑀳𑀸𑀲𑁄𑀢𑀁 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺, 𑀢𑀺𑀯𑀺𑀥𑀁 𑀫𑀮𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑁂 𑀉𑀧𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀫𑀮𑀁 𑀯𑀺𑀭𑀚𑀁 𑀲𑀼𑀤𑁆𑀥𑀁 𑀧𑀡𑁆𑀟𑀭𑀁 𑀅𑀚𑀸𑀢𑀺𑀁 𑀅𑀚𑀭𑀁 𑀅𑀫𑀭𑀁 𑀲𑀼𑀔𑀁 𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀁 𑀅𑀪𑀬𑀁 𑀦𑀕𑀭𑀼𑀢𑁆𑀢𑀫𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀦𑀕𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑁄𑀘𑁂𑀢𑀺, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀧𑁆𑀨𑀸𑀧𑀡𑀦𑁆’𑀢𑀺𑁇

1389. _‘𑀓𑀫𑁆𑀫𑀫𑀽𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀧𑀡𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀣;_  
_𑀆𑀭𑀫𑁆𑀫𑀡𑀁 𑀓𑀺𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁄 𑀫𑀼𑀘𑁆𑀘𑀣 𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸’”𑀢𑀺𑁇_  


(204.)

1390. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀕𑀦𑁆𑀥𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺 𑀔𑁄 𑀧𑀦, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀲𑀻𑀮𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀬𑁄 𑀅𑀓𑁆𑀔𑀸𑀢𑀸, 𑀬𑁂𑀦 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁂𑀦 𑀅𑀦𑀼𑀮𑀺𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀤𑁂𑀯𑀓𑀁 𑀮𑁄𑀓𑀁 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁂𑀦 𑀥𑀽𑀧𑁂𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀥𑀽𑀧𑁂𑀦𑁆𑀢𑀺, 𑀤𑀺𑀲𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀤𑀺𑀲𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀯𑀸𑀢𑀫𑁆𑀧𑀺 𑀧𑀝𑀺𑀯𑀸𑀢𑀫𑁆𑀧𑀺 𑀯𑀸𑀬𑀦𑁆𑀢𑀺 𑀅𑀢𑀺𑀯𑀸𑀬𑀦𑁆𑀢𑀺, 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀢𑀸 𑀲𑀻𑀮𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀬𑁄? 𑀲𑀭𑀡𑀲𑀻𑀮𑀁 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀲𑀻𑀮𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀻𑀮𑀁 𑀤𑀲𑀗𑁆𑀕𑀲𑀻𑀮𑀁 𑀧𑀜𑁆𑀘𑀼𑀤𑁆𑀤𑁂𑀲𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀻𑀮𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀕𑀦𑁆𑀥𑀸𑀧𑀡𑀦𑁆’𑀢𑀺𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦—

1391. _‘𑀦 𑀧𑀼𑀧𑁆𑀨𑀕𑀦𑁆𑀥𑁄 𑀧𑀝𑀺𑀯𑀸𑀢𑀫𑁂𑀢𑀺,_

1392. _𑀘𑀦𑁆𑀤𑀦𑀁 𑀢𑀕𑀭𑀁 𑀯𑀸𑀧𑀺,_

1394. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀨𑀮𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀨𑀮𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀁 𑀲𑀼𑀜𑁆𑀜𑀢𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀬𑀁 𑀨𑀮𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑁄 𑀓𑀫𑁆𑀫𑀫𑀽𑀮𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀣𑀺𑀢𑀁 𑀨𑀮𑀁 𑀓𑀺𑀡𑀸𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁, 𑀬𑀤𑀺 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁, 𑀬𑀤𑀺 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀁, 𑀬𑀤𑀺 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀁, 𑀬𑀤𑀺 𑀲𑀼𑀜𑁆𑀜𑀢𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁, 𑀬𑀤𑀺 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁, 𑀬𑀤𑀺 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀲𑁆𑀲𑀘𑀺 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀥𑀼𑀯𑀨𑀮𑁄 𑀅𑀫𑁆𑀩𑁄 𑀪𑀯𑁂𑀬𑁆𑀬, 𑀲𑁄 𑀦 𑀢𑀸𑀯 𑀢𑀢𑁄 𑀨𑀮𑀸𑀦𑀺 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀬𑀸𑀯 𑀓𑀬𑀺𑀓𑀸 𑀦 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀧𑀦 𑀓𑀬𑀺𑀓𑁂 𑀫𑀽𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀺 ‘𑀅𑀫𑁆𑀪𑁄 𑀧𑀼𑀭𑀺𑀲, 𑀏𑀲𑁄 𑀔𑁄 𑀥𑀼𑀯𑀨𑀮𑁄 𑀅𑀫𑁆𑀩𑁄, 𑀢𑀢𑁄 𑀬𑀁 𑀇𑀘𑁆𑀙𑀲𑀺, 𑀏𑀢𑁆𑀢𑀓𑀁 𑀨𑀮𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺 𑀲𑀮𑀸𑀝𑀼𑀓𑀁 𑀯𑀸 𑀤𑁄𑀯𑀺𑀮𑀁 𑀯𑀸 𑀓𑁂𑀲𑀺𑀓𑀁 𑀯𑀸 𑀆𑀫𑀁 𑀯𑀸 𑀧𑀓𑁆𑀓𑀁 𑀯𑀸’𑀢𑀺, 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀢𑁆𑀢𑀦𑀸 𑀤𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁂𑀦 𑀬𑀤𑀺 𑀲𑀮𑀸𑀝𑀼𑀓𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑀮𑀸𑀝𑀼𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀬𑀤𑀺 𑀤𑁄𑀯𑀺𑀮𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀤𑁄𑀯𑀺𑀮𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀬𑀤𑀺 𑀓𑁂𑀲𑀺𑀓𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀓𑁂𑀲𑀺𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀬𑀤𑀺 𑀆𑀫𑀓𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀆𑀫𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀬𑀤𑀺 𑀧𑀓𑁆𑀓𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀧𑀓𑁆𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄 𑀬𑀁 𑀨𑀮𑀁 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑁄 𑀓𑀫𑁆𑀫𑀫𑀽𑀮𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀣𑀺𑀢𑀁 𑀨𑀮𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀬𑀤𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀁…𑀧𑁂…  𑀬𑀤𑀺 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀨𑀮𑀸𑀧𑀡𑀦𑁆’𑀢𑀺𑁇

1396. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀕𑀤𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀅𑀕𑀤𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑀦𑀺, 𑀬𑁂𑀳𑀺 𑀅𑀕𑀤𑁂𑀳𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑀤𑁂𑀯𑀓𑀁 𑀮𑁄𑀓𑀁 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀲𑀢𑁄 𑀧𑀭𑀺𑀫𑁄𑀘𑁂𑀢𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀅𑀕𑀤𑀸𑀦𑀺? 𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀺 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁, 𑀢𑀢𑁆𑀣 𑀬𑁂 𑀓𑁂𑀘𑀺 𑀅𑀜𑁆𑀜𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀚𑀸𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀚𑀭𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀫𑀭𑀡𑀸 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀅𑀕𑀤𑀸𑀧𑀡𑀦𑁆’𑀢𑀺𑁇

1398. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀑𑀲𑀥𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀑𑀲𑀥𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑀦𑀺, 𑀬𑁂𑀳𑀺 𑀑𑀲𑀥𑁂𑀳𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀏𑀢𑁂𑀳𑀺 𑀑𑀲𑀥𑁂𑀳𑀺 𑀪𑀕𑀯𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀢𑀺, 𑀮𑁄𑀪𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀤𑁄𑀲𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀫𑁄𑀳𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀫𑀸𑀦𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀅𑀳𑀺𑀭𑀺𑀓𑀸𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀲𑀩𑁆𑀩𑀓𑀺𑀮𑁂𑀲𑀯𑀫𑀦𑀁 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀑𑀲𑀥𑀸𑀧𑀡𑀦𑁆’𑀢𑀺𑁇

1400. _𑀥𑀫𑁆𑀫𑁄𑀲𑀥𑀁 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_

1401. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀫𑀢𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀅𑀫𑀢𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀁, 𑀬𑁂𑀦 𑀅𑀫𑀢𑁂𑀦 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑀤𑁂𑀯𑀓𑀁 𑀮𑁄𑀓𑀁 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑀺, 𑀬𑁂𑀦 𑀅𑀫𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀺𑀢𑁆𑀢𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀩𑁆𑀬𑀸𑀥𑀺𑀫𑀭𑀡𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼𑁇 𑀓𑀢𑀫𑀁 𑀢𑀁 𑀅𑀫𑀢𑀁? 𑀬𑀤𑀺𑀤𑀁 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑁇 𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑁂𑀢𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀯𑀸𑀢𑀺𑀤𑁂𑀯𑁂𑀦 ‘𑀅𑀫𑀢𑀁 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀬𑁂 𑀓𑀸𑀬𑀕𑀢𑀸𑀲𑀢𑀺𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀅𑀫𑀢𑀸𑀧𑀡𑀦𑁆’𑀢𑀺𑁇

1402. _‘𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑀁 𑀚𑀦𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_

1403. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀭𑀢𑀦𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀭𑀢𑀦𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀓𑁆𑀔𑀸𑀢𑀸𑀦𑀺, 𑀬𑁂𑀳𑀺 𑀭𑀢𑀦𑁂𑀳𑀺 𑀯𑀺𑀪𑀽𑀲𑀺𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀤𑁂𑀯𑀓𑀁 𑀮𑁄𑀓𑀁 𑀯𑀺𑀭𑁄𑀘𑀦𑁆𑀢𑀺 𑀑𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀪𑀸𑀲𑁂𑀦𑁆𑀢𑀺 𑀚𑀮𑀦𑁆𑀢𑀺 𑀧𑀚𑁆𑀚𑀮𑀦𑁆𑀢𑀺 𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀆𑀮𑁄𑀓𑀁 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺? 𑀲𑀻𑀮𑀭𑀢𑀦𑀁 𑀲𑀫𑀸𑀥𑀺𑀭𑀢𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀢𑀦𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀭𑀢𑀦𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀭𑀢𑀦𑀁 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀭𑀢𑀦𑀁𑁇

1404. 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀻𑀮𑀭𑀢𑀦𑀁? 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀻𑀮𑀁 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀲𑀁𑀯𑀭𑀲𑀻𑀮𑀁 𑀆𑀚𑀻𑀯𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀲𑀻𑀮𑀁 𑀧𑀘𑁆𑀘𑀬𑀲𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀲𑀻𑀮𑀁 𑀘𑀽𑀍𑀅𑀲𑀻𑀮𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀲𑀻𑀮𑀁 𑀫𑀳𑀸𑀲𑀻𑀮𑀁 𑀫𑀕𑁆𑀕𑀲𑀻𑀮𑀁 𑀨𑀮𑀲𑀻𑀮𑀁𑁇 𑀲𑀻𑀮𑀭𑀢𑀦𑁂𑀦 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀺𑀪𑀽𑀲𑀺𑀢𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀤𑁂𑀯𑀓𑁄 𑀮𑁄𑀓𑁄 𑀲𑀫𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁄 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀧𑀚𑀸 𑀧𑀺𑀳𑀬𑀢𑀺 𑀧𑀢𑁆𑀣𑁂𑀢𑀺, 𑀲𑀻𑀮𑀭𑀢𑀦𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀤𑀺𑀲𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀤𑀺𑀲𑀫𑁆𑀧𑀺 𑀉𑀤𑁆𑀥𑀫𑁆𑀧𑀺 𑀅𑀥𑁄𑀧𑀺 𑀢𑀺𑀭𑀺𑀬𑀫𑁆𑀧𑀺 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀅𑀢𑀺𑀯𑀺𑀭𑁄𑀘𑀢𑀺, 𑀳𑁂𑀝𑁆𑀞𑀢𑁄 𑀅𑀯𑀻𑀘𑀺𑀁 𑀉𑀧𑀭𑀺𑀢𑁄 𑀪𑀯𑀕𑁆𑀕𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀏𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂 𑀲𑀩𑁆𑀩𑀭𑀢𑀦𑀸𑀦𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀪𑀯𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀻𑀮𑀭𑀢𑀦𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑁄 𑀭𑀢𑀦𑀸𑀧𑀡𑁂 𑀧𑀲𑀸𑀭𑀺𑀢𑀸𑀦𑀺, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀲𑀻𑀮𑀭𑀢𑀦𑀦𑁆’𑀢𑀺𑁇

_𑀭𑀢𑀦𑀁 𑀯𑁄 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀣𑀸’𑀢𑀺𑁇_  


1406. 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀸𑀥𑀺𑀭𑀢𑀦𑀁? 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀲𑀼𑀜𑁆𑀜𑀢𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁄 𑀲𑀫𑀸𑀥𑀺𑁇 𑀲𑀫𑀸𑀥𑀺𑀭𑀢𑀦𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑁂 𑀢𑁂 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀸𑀦𑀼𑀤𑁆𑀥𑀘𑁆𑀘𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀓𑀺𑀮𑁂𑀲𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀯𑀺𑀯𑀺𑀥𑀸𑀦𑀺 𑀘 𑀓𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀦𑀺, 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀸𑀥𑀺𑀁 𑀆𑀲𑀚𑁆𑀚 𑀯𑀺𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀥𑀫𑀦𑁆𑀢𑀺 𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑀦𑁆𑀢𑀺 𑀦 𑀲𑀡𑁆𑀞𑀦𑁆𑀢𑀺 𑀦 𑀉𑀧𑀮𑀺𑀫𑁆𑀧𑀦𑁆𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀯𑀸𑀭𑀺 𑀧𑁄𑀓𑁆𑀔𑀭𑀧𑀢𑁆𑀢𑁂 𑀯𑀺𑀓𑀺𑀭𑀢𑀺 𑀯𑀺𑀥𑀫𑀢𑀺 𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑀢𑀺 𑀦 𑀲𑀡𑁆𑀞𑀸𑀢𑀺 𑀦 𑀉𑀧𑀮𑀺𑀫𑁆𑀧𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀸 𑀧𑀤𑀼𑀫𑀲𑁆𑀲𑁇 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑀸𑀥𑀺𑀭𑀢𑀦𑀁 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑁂 𑀢𑁂 𑀓𑀸𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀸𑀦𑀼𑀤𑁆𑀥𑀘𑁆𑀘𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀓𑀺𑀮𑁂𑀲𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀯𑀺𑀯𑀺𑀥𑀸𑀦𑀺 𑀘 𑀓𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑀸𑀦𑀺, 𑀢𑁂 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀸𑀥𑀺𑀁 𑀆𑀲𑀚𑁆𑀚 𑀯𑀺𑀓𑀺𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀥𑀫𑀦𑁆𑀢𑀺 𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑀦𑁆𑀢𑀺 𑀦 𑀲𑀡𑁆𑀞𑀦𑁆𑀢𑀺 𑀦 𑀉𑀧𑀮𑀺𑀫𑁆𑀧𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀸𑀥𑀺𑀭𑀢𑀦𑀦𑁆’𑀢𑀺, 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀫𑀸𑀥𑀺𑀭𑀢𑀦𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑁄 𑀭𑀢𑀦𑀸𑀧𑀡𑁂 𑀧𑀲𑀸𑀭𑀺𑀢𑀸𑀦𑀺𑁇

1407. _‘𑀲𑀫𑀸𑀥𑀺𑀭𑀢𑀦𑀫𑀸𑀮𑀲𑁆𑀲,_  
_𑀓𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑀸 𑀦 𑀚𑀸𑀬𑀭𑁂;_  
_𑀦 𑀘 𑀯𑀺𑀓𑁆𑀔𑀺𑀧𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀏𑀢𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀣𑀸’𑀢𑀺𑁇_  


1408. 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀁? 𑀬𑀸𑀬, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 ‘𑀇𑀤𑀁 𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀇𑀤𑀁 𑀅𑀓𑀼𑀲𑀮𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀇𑀤𑀁 𑀲𑀸𑀯𑀚𑁆𑀚𑀁, 𑀇𑀤𑀁 𑀅𑀦𑀯𑀚𑁆𑀚𑀁, 𑀇𑀤𑀁 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀁 𑀦 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀁, 𑀇𑀤𑀁 𑀳𑀻𑀦𑀁, 𑀇𑀤𑀁 𑀧𑀡𑀻𑀢𑀁, 𑀇𑀤𑀁 𑀓𑀡𑁆𑀳𑀁, 𑀇𑀤𑀁 𑀲𑀼𑀓𑁆𑀓𑀁, 𑀇𑀤𑀁 𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀕𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺, ‘𑀅𑀬𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀭𑀢𑀦𑀦𑁆’𑀢𑀺𑁇

_𑀔𑀺𑀧𑁆𑀧𑀁 𑀨𑀲𑁆𑀲𑁂𑀢𑀺 𑀅𑀫𑀢𑀁,_  
_𑀦 𑀘 𑀲𑁄 𑀭𑁄𑀘𑀢𑁂 𑀪𑀯𑁂’𑀢𑀺𑁇_  


1410. 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀢𑀦𑀁? 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀢𑀦𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀧𑀢𑁆𑀢𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀢𑀦𑀁 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑁄’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀫𑀼𑀢𑁆𑀢𑀸𑀓𑀮𑀸𑀧𑀫𑀡𑀺𑀓𑀮𑀸𑀧𑀧𑀯𑀸𑀍𑀅𑀓𑀮𑀸𑀧𑀸𑀪𑀭𑀡𑀧𑁆𑀧𑀝𑀺𑀫𑀡𑁆𑀟𑀺𑀢𑁄 𑀅𑀕𑀮𑀼𑀢𑀕𑀭𑀢𑀸𑀮𑀻𑀲𑀓𑀮𑁄𑀳𑀺𑀢𑀘𑀦𑁆𑀤𑀦𑀸𑀦𑀼𑀮𑀺𑀢𑁆𑀢𑀕𑀢𑁆𑀢𑁄 𑀦𑀸𑀕𑀧𑀼𑀦𑁆𑀦𑀸𑀕𑀲𑀸𑀮𑀲𑀮𑀍𑀅𑀘𑀫𑁆𑀧𑀓𑀬𑀽𑀣𑀺𑀓𑀸𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀓𑀧𑀸𑀝𑀮𑀼𑀧𑁆𑀧𑀮𑀯𑀲𑁆𑀲𑀺𑀓𑀫𑀮𑁆𑀮𑀺𑀓𑀸𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑁂𑀲𑀚𑀦𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀅𑀢𑀺𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀑𑀪𑀸𑀲𑀢𑀺 𑀧𑀪𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀧𑀪𑀸𑀲𑀢𑀺 𑀚𑀮𑀢𑀺 𑀧𑀚𑁆𑀚𑀮𑀢𑀺 𑀅𑀪𑀺𑀪𑀯𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑀢𑀺 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀭𑀢𑀦𑀸𑀪𑀭𑀡𑁂𑀳𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀧𑀢𑁆𑀢𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀢𑀦𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑁄 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀅𑀢𑀺𑀯𑀺𑀭𑁄𑀘𑀢𑀺 𑀑𑀪𑀸𑀲𑀢𑀺 𑀧𑀪𑀸𑀲𑀢𑀺 𑀲𑀫𑁆𑀧𑀪𑀸𑀲𑀢𑀺 𑀚𑀮𑀢𑀺 𑀧𑀚𑁆𑀚𑀮𑀢𑀺 𑀅𑀪𑀺𑀪𑀯𑀢𑀺 𑀅𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑀢𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀅𑀕𑁆𑀕𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀏𑀢𑀁 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀦𑀁 𑀲𑀩𑁆𑀩𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀦𑀸𑀦𑀁, 𑀬𑀤𑀺𑀤𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀦𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀢𑀦𑀦𑁆’𑀢𑀺𑁇

1411. _‘𑀫𑀡𑀺𑀫𑀸𑀮𑀸𑀥𑀭𑀁 𑀕𑁂𑀳,_  
_𑀚𑀦𑁄 𑀲𑀸𑀫𑀺𑀁 𑀉𑀤𑀺𑀓𑁆𑀔𑀢𑀺;_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀭𑀢𑀦𑀫𑀸𑀮𑀦𑁆𑀢𑀼,_  
_𑀉𑀤𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑀸’𑀢𑀺𑁇_  


1412. 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀭𑀢𑀦𑀁? 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀡𑀜𑀸𑀡𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀭𑀢𑀦𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀫𑀕𑁆𑀕𑀨𑀮𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀓𑀺𑀮𑁂𑀲𑀸𑀯𑀲𑀺𑀝𑁆𑀞𑀓𑀺𑀮𑁂𑀲𑁂 𑀘 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀢𑀺𑁇

1413. _‘𑀬𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀩𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺,_  
_𑀅𑀭𑀺𑀬𑀸 𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑀢𑀁;_  
_𑀢𑀁 𑀜𑀸𑀡𑀭𑀢𑀦𑀁 𑀮𑀤𑁆𑀥𑀼𑀁,_  
_𑀯𑀸𑀬𑀫𑁂𑀣 𑀚𑀺𑀦𑁄𑀭𑀲𑀸’𑀢𑀺𑁇_  


1414. 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀭𑀢𑀦𑀁? 𑀘𑀢𑀲𑁆𑀲𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬𑁄 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀢𑀽𑀳𑀺 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀭𑀢𑀦𑁂𑀳𑀺 𑀲𑀫𑀮𑀗𑁆𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀁 𑀬𑀁 𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺, 𑀬𑀤𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀁, 𑀬𑀤𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀁, 𑀬𑀤𑀺 𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀁, 𑀬𑀤𑀺 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀁, 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀅𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑁄 𑀅𑀪𑀻𑀭𑀼 𑀅𑀘𑁆𑀙𑀫𑁆𑀪𑀻 𑀅𑀦𑀼𑀢𑁆𑀭𑀸𑀲𑀻 𑀯𑀺𑀕𑀢𑀮𑁄𑀫𑀳𑀁𑀲𑁄 𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺𑁇

1415. 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀬𑁄𑀥𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀲𑀽𑀭𑁄 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀧𑀜𑁆𑀘𑀸𑀯𑀼𑀥𑁄 𑀅𑀘𑁆𑀙𑀫𑁆𑀪𑀺𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀁 𑀑𑀢𑀭𑀢𑀺, ‘𑀲𑀘𑁂 𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀤𑀽𑀭𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀲𑀼𑀦𑀸 𑀧𑀸𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁄 𑀑𑀭𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁄 𑀑𑀭𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀡𑀬𑁂𑀦 𑀧𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀉𑀧𑀕𑀢𑀁 𑀲𑀦𑁆𑀢𑀁 𑀫𑀡𑁆𑀟𑀮𑀕𑁆𑀕𑁂𑀦 𑀤𑁆𑀯𑀺𑀥𑀸 𑀙𑀺𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀓𑀸𑀬𑀽𑀧𑀕𑀢𑀁 𑀙𑀼𑀭𑀺𑀓𑀸𑀬 𑀯𑀺𑀦𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀘𑀢𑀼𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀭𑀢𑀦𑀫𑀡𑁆𑀟𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀘𑁆𑀙𑀫𑁆𑀪𑀺𑀢𑁄 𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺𑁇

1416. ‘𑀬𑁄 𑀓𑁄𑀘𑀺 𑀫𑀁 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑁂 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀓𑀸𑀭𑀡𑁂𑀦 𑀓𑀸𑀭𑀡𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀳𑁂𑀢𑀼𑀦𑀸 𑀳𑁂𑀢𑀼𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀬𑁂𑀦 𑀦𑀬𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀯𑀺𑀫𑀢𑀺𑀁 𑀯𑀺𑀯𑁂𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑁄𑀲𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀦𑁇

1417. 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑁂 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀫𑀢𑁂𑀦 𑀅𑀫𑀢𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀲𑀗𑁆𑀔𑀢𑁂𑀦 𑀅𑀲𑀗𑁆𑀔𑀢𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀼𑀜𑁆𑀜𑀢𑁂𑀦 𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀦 𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀡𑀺𑀳𑀺𑀢𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀦𑁂𑀚𑁂𑀦 𑀅𑀦𑁂𑀚𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀯𑀺𑀫𑀢𑀺𑀁 𑀯𑀺𑀯𑁂𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑁄𑀲𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀦𑁇

1418. 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀫𑀁 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑁂 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀤𑁂𑀦 𑀧𑀤𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀦𑀼𑀧𑀤𑁂𑀦 𑀅𑀦𑀼𑀧𑀤𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀓𑁆𑀔𑀭𑁂𑀦 𑀅𑀓𑁆𑀔𑀭𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀦𑁆𑀥𑀺𑀬𑀸 𑀲𑀦𑁆𑀥𑀺𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀦 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁂𑀦 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀯𑀡𑁆𑀡𑁂𑀦 𑀯𑀡𑁆𑀡𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀭𑁂𑀦 𑀲𑀭𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀯𑁄𑀳𑀸𑀭𑁂𑀦 𑀯𑁄𑀳𑀸𑀭𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀯𑀺𑀫𑀢𑀺𑀁 𑀯𑀺𑀯𑁂𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑁄𑀲𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀦𑁇

1419. 𑀬𑁄 𑀓𑁄𑀘𑀺 𑀫𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑁂 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀦 𑀧𑀝𑀺𑀪𑀸𑀦𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀑𑀧𑀫𑁆𑀫𑁂𑀦 𑀑𑀧𑀫𑁆𑀫𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀮𑀓𑁆𑀔𑀡𑁂𑀦 𑀮𑀓𑁆𑀔𑀡𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀭𑀲𑁂𑀦 𑀭𑀲𑀁 𑀓𑀣𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀯𑀺𑀫𑀢𑀺𑀁 𑀯𑀺𑀯𑁂𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑁄𑀲𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀦𑀸’𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀭𑀢𑀦𑀦𑁆’𑀢𑀺𑁇

1420. _‘𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀓𑀺𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀜𑀸𑀡𑁂𑀦 𑀨𑀲𑁆𑀲𑀬𑁂𑀬𑁆𑀬 𑀬𑁄;_  
_𑀅𑀘𑁆𑀙𑀫𑁆𑀪𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑁆𑀩𑀺𑀕𑁆𑀕𑁄,_  
_𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁂’𑀢𑀺𑁇_  


1421. 𑀓𑀢𑀫𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀭𑀢𑀦𑀁? 𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀢𑁆𑀢𑀳𑀺 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀭𑀢𑀦𑁂𑀳𑀺 𑀧𑀝𑀺𑀫𑀡𑁆𑀟𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀩𑀁 𑀢𑀫𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀲𑀤𑁂𑀯𑀓𑀁 𑀮𑁄𑀓𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑀺 𑀧𑀪𑀸𑀲𑁂𑀢𑀺 𑀆𑀮𑁄𑀓𑀁 𑀚𑀦𑁂𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀭𑀢𑀦𑀦𑁆’𑀢𑀺𑁇

1422. _‘𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀭𑀢𑀦𑀫𑀸𑀮𑀲𑁆𑀲,_  
_𑀉𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑀸;_  
_𑀓𑀫𑁆𑀫𑁂𑀦 𑀢𑀁 𑀓𑀺𑀡𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀭𑀢𑀦𑀁 𑀯𑁄 𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀣𑀸’”𑀢𑀺𑁇_  


(210.)

1423. “𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀓𑀢𑀫𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀩𑁆𑀩𑀸𑀧𑀡𑀦𑁆”𑀢𑀺? “𑀲𑀩𑁆𑀩𑀸𑀧𑀡𑀁 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀦𑀯𑀗𑁆𑀕𑀁 𑀩𑀼𑀤𑁆𑀥𑀯𑀘𑀦𑀁 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀭𑀺𑀪𑁄𑀕𑀺𑀓𑀸𑀦𑀺 𑀘𑁂𑀢𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀖𑀭𑀢𑀦𑀜𑁆𑀘𑁇 𑀲𑀩𑁆𑀩𑀸𑀧𑀡𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀢𑀺𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀪𑁄𑀕𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀆𑀬𑀼𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀆𑀭𑁄𑀕𑁆𑀬𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀯𑀡𑁆𑀡𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀫𑀸𑀦𑀼𑀲𑀺𑀓𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀤𑀺𑀩𑁆𑀩𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸, 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺 𑀧𑀲𑀸𑀭𑀺𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀬𑁂 𑀢𑀁 𑀢𑀁 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀓𑀫𑁆𑀫𑀫𑀽𑀮𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀣𑀺𑀢𑀧𑀢𑁆𑀣𑀺𑀢𑀁 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀓𑀺𑀡𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀲𑀻𑀮𑀲𑀫𑀸𑀤𑀸𑀦𑁂𑀦 𑀓𑀺𑀡𑀦𑁆𑀢𑀺, 𑀓𑁂𑀘𑀺 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑁂𑀦 𑀓𑀺𑀡𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀓𑀫𑁆𑀫𑀫𑀽𑀮𑁂𑀦 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀝𑀺𑀮𑀪𑀦𑁆𑀢𑀺𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀧𑀡𑀺𑀓𑀲𑁆𑀲 𑀆𑀧𑀡𑁂 𑀢𑀺𑀮𑀫𑀼𑀕𑁆𑀕𑀫𑀸𑀲𑁂 𑀧𑀭𑀺𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑀼𑀕𑁆𑀕𑀫𑀸𑀲𑁂𑀦 𑀅𑀧𑁆𑀧𑀓𑁂𑀦𑀧𑀺 𑀫𑀽𑀮𑁂𑀦 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀩𑁆𑀩𑀸𑀧𑀡𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀓𑀫𑁆𑀫𑀫𑀽𑀮𑁂𑀦 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀧𑀝𑀺𑀮𑀪𑀦𑁆𑀢𑀺𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀫𑀳𑀸𑀭𑀸𑀚, ‘𑀪𑀕𑀯𑀢𑁄 𑀲𑀩𑁆𑀩𑀸𑀧𑀡𑀦𑁆’𑀢𑀺𑁇

1424. _‘𑀆𑀬𑀼 𑀅𑀭𑁄𑀕𑀢𑀸 𑀯𑀡𑁆𑀡𑀁,_  
_𑀲𑀕𑁆𑀕𑀁 𑀉𑀘𑁆𑀘𑀸𑀓𑀼𑀮𑀻𑀦𑀢𑀸;_  
_𑀅𑀲𑀗𑁆𑀔𑀢𑀜𑁆𑀘 𑀅𑀫𑀢𑀁,_  
_𑀅𑀢𑁆𑀣𑀺 𑀲𑀩𑁆𑀩𑀸𑀧𑀡𑁂 𑀚𑀺𑀦𑁂𑁇_  


1425. _𑀅𑀧𑁆𑀧𑁂𑀦 𑀩𑀳𑀼𑀓𑁂𑀦𑀸𑀧𑀺,_  
_𑀓𑀫𑁆𑀫𑀫𑀽𑀮𑁂𑀦 𑀕𑀬𑁆𑀳𑀢𑀺;_  
_𑀓𑀺𑀡𑀺𑀢𑁆𑀯𑀸 𑀲𑀤𑁆𑀥𑀸𑀫𑀽𑀮𑁂𑀦,_  
_𑀲𑀫𑀺𑀤𑁆𑀥𑀸 𑀳𑁄𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄’𑀢𑀺𑁇_  


1426. 𑀪𑀕𑀯𑀢𑁄 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀏𑀯𑀭𑀽𑀧𑀸 𑀚𑀦𑀸 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀺, 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑀸 𑀯𑁂𑀦𑀬𑀺𑀓𑀸 𑀆𑀪𑀺𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑀸 𑀚𑀸𑀢𑀓𑀪𑀸𑀡𑀓𑀸 𑀤𑀻𑀖𑀪𑀸𑀡𑀓𑀸 𑀫𑀚𑁆𑀛𑀺𑀫𑀪𑀸𑀡𑀓𑀸 𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀪𑀸𑀡𑀓𑀸 𑀅𑀗𑁆𑀕𑀼𑀢𑁆𑀢𑀭𑀪𑀸𑀡𑀓𑀸 𑀔𑀼𑀤𑁆𑀤𑀓𑀪𑀸𑀡𑀓𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑀸 𑀲𑀤𑀢𑁆𑀣𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀺𑀓𑀸 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀺𑀓𑀸 𑀧𑀮𑀸𑀮𑀧𑀼𑀜𑁆𑀚𑀺𑀓𑀸 𑀲𑁄𑀲𑀸𑀦𑀺𑀓𑀸 𑀦𑁂𑀲𑀚𑁆𑀚𑀺𑀓𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀓𑀸 𑀨𑀮𑀝𑁆𑀞𑀸 𑀲𑁂𑀓𑁆𑀔𑀸 𑀨𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑁄 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀗𑁆𑀕𑀢𑀸 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑁃𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁃𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀩𑀮𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀫𑀕𑁆𑀕𑀯𑀭𑀛𑀸𑀦𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀭𑀽𑀧𑀸𑀭𑀽𑀧𑀲𑀦𑁆𑀢𑀲𑀼𑀔𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀸, 𑀢𑁂𑀳𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺 𑀆𑀓𑀼𑀮𑀁 𑀲𑀫𑀸𑀓𑀼𑀮𑀁 𑀆𑀓𑀺𑀡𑁆𑀡𑀁 𑀲𑀫𑀸𑀓𑀺𑀡𑁆𑀡𑀁 𑀦𑀍𑀅𑀯𑀦𑀲𑀭𑀯𑀦𑀫𑀺𑀯 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀯𑀢𑀻𑀳—

1427. _‘𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀯𑀻𑀢𑀤𑁄𑀲𑀸,_  
_𑀯𑀻𑀢𑀫𑁄𑀳𑀸 𑀅𑀦𑀸𑀲𑀯𑀸;_  
_𑀯𑀻𑀢𑀢𑀡𑁆𑀳𑀸 𑀅𑀦𑀸𑀤𑀸𑀦𑀸,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1428. _𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸 𑀥𑀼𑀢𑀥𑀭𑀸,_  
_𑀛𑀸𑀬𑀺𑀦𑁄 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸;_  
_𑀯𑀺𑀯𑁂𑀓𑀸𑀪𑀺𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1429. _𑀦𑁂𑀲𑀚𑁆𑀚𑀺𑀓𑀸 𑀲𑀦𑁆𑀣𑀢𑀺𑀓𑀸,_  
_𑀅𑀣𑁄𑀧𑀺 𑀞𑀸𑀦𑀘𑀗𑁆𑀓𑀫𑀸;_  
_𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀥𑀭𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1430. _𑀢𑀺𑀘𑀻𑀯𑀭𑀥𑀭𑀸 𑀲𑀦𑁆𑀢𑀸,_  
_𑀘𑀫𑁆𑀫𑀔𑀡𑁆𑀟𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀸;_  
_𑀭𑀢𑀸 𑀏𑀓𑀸𑀲𑀦𑁂 𑀯𑀺𑀜𑁆𑀜𑀽,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1431. _𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀦𑀺𑀧𑀓𑀸 𑀥𑀻𑀭𑀸,_  
_𑀅𑀧𑁆𑀧𑀸𑀳𑀸𑀭𑀸 𑀅𑀮𑁄𑀮𑀼𑀧𑀸;_  
_𑀮𑀸𑀪𑀸𑀮𑀸𑀪𑁂𑀦 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1432. _𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀁 𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑀸,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1433. _𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀨𑀮𑀝𑁆𑀞𑀸 𑀘,_  
_𑀲𑁂𑀓𑁆𑀔𑀸 𑀨𑀮𑀲𑀫𑀗𑁆𑀕𑀺𑀦𑁄;_  
_𑀆𑀲𑀻𑀲𑀓𑀸 𑀉𑀢𑁆𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1434. _𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸 𑀘 𑀯𑀺𑀫𑀮𑀸,_  
_𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑁄 𑀘 𑀬𑁂;_  
_𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀘 𑀅𑀭𑀳𑀦𑁆𑀢𑁄,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1435. _𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀸,_  
_𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑀸;_

_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1436. _𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀓𑀼𑀲𑀮𑀸,_  
_𑀲𑀫𑀸𑀥𑀺𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑀸;_  
_𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1437. _𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑁂 𑀕𑁄𑀘𑀭𑁂 𑀭𑀢𑀸;_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀫𑁆𑀳𑀺 𑀘𑀭𑀡𑀸,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_

_𑀲𑀼𑀤𑀦𑁆𑀢𑀸 𑀉𑀢𑁆𑀢𑀫𑁂 𑀤𑀫𑁆𑀫𑁂,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


1439. _𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑀸 𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀘,_  
_𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀭𑀫𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺 𑀢𑁂’𑀢𑀺𑁇_  


1440. 𑀬𑁂 𑀔𑁄 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑀭𑀺𑀫𑀺𑀢𑀜𑀸𑀡𑀯𑀭𑀥𑀭𑀸 𑀅𑀲𑀗𑁆𑀕𑀸 𑀅𑀢𑀼𑀮𑀕𑀼𑀡𑀸 𑀅𑀢𑀼𑀮𑀬𑀲𑀸 𑀅𑀢𑀼𑀮𑀩𑀮𑀸 𑀅𑀢𑀼𑀮𑀢𑁂𑀚𑀸 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀸𑀦𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 **‘𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑁄’**𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

1441. 𑀬𑁂 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀤𑁆𑀥𑀺𑀫𑀦𑁆𑀢𑁄 𑀅𑀥𑀺𑀕𑀢𑀧𑁆𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀧𑀢𑁆𑀢𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀸 𑀕𑀕𑀦𑀘𑀭𑀸 𑀤𑀼𑀭𑀸𑀲𑀤𑀸 𑀤𑀼𑀧𑁆𑀧𑀲𑀳𑀸 𑀅𑀦𑀸𑀮𑀫𑁆𑀩𑀘𑀭𑀸 𑀲𑀲𑀸𑀕𑀭𑀫𑀳𑀺𑀥𑀭𑀧𑀣𑀯𑀺𑀓𑀫𑁆𑀧𑀓𑀸 𑀘𑀦𑁆𑀤𑀲𑀽𑀭𑀺𑀬𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀓𑀸 𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀸𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀸𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀓𑀼𑀲𑀮𑀸 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 **‘𑀧𑀼𑀭𑁄𑀳𑀺𑀢𑀸’**𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

1442. 𑀬𑁂 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀼𑀢𑀗𑁆𑀕𑀫𑀦𑀼𑀕𑀢𑀸 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀫𑀦𑁂𑀲𑀦𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀓𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀲𑀧𑀤𑀸𑀦𑀘𑀸𑀭𑀺𑀦𑁄 𑀪𑀫𑀭𑀸𑀯 𑀕𑀦𑁆𑀥𑀫𑀦𑀼𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀓𑀸𑀦𑀦𑀁, 𑀓𑀸𑀬𑁂 𑀘 𑀚𑀻𑀯𑀺𑀢𑁂 𑀘 𑀦𑀺𑀭𑀧𑁂𑀓𑁆𑀔𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀥𑀼𑀢𑀗𑁆𑀕𑀕𑀼𑀡𑁂 𑀅𑀕𑁆𑀕𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 **‘𑀅𑀓𑁆𑀔𑀤𑀲𑁆𑀲𑀸’**𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

1443. 𑀬𑁂 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸 𑀯𑀺𑀫𑀮𑀸 𑀦𑀺𑀓𑁆𑀓𑀺𑀮𑁂𑀲𑀸 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑀓𑀼𑀲𑀮𑀸 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀫𑁆𑀳𑀺 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 **‘𑀦𑀕𑀭𑀚𑁄𑀢𑀓𑀸’**𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

1444. 𑀬𑁂 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀆𑀕𑀢𑀸𑀕𑀫𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀯𑀺𑀦𑀬𑀥𑀭𑀸 𑀫𑀸𑀢𑀺𑀓𑀸𑀥𑀭𑀸 𑀲𑀺𑀣𑀺𑀮𑀥𑀦𑀺𑀢𑀤𑀻𑀖𑀭𑀲𑁆𑀲𑀕𑀭𑀼𑀓𑀮𑀳𑀼𑀓𑀓𑁆𑀔𑀭𑀧𑀭𑀺𑀘𑁆𑀙𑁂𑀤𑀓𑀼𑀲𑀮𑀸 𑀦𑀯𑀗𑁆𑀕𑀲𑀸𑀲𑀦𑀥𑀭𑀸, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 **‘𑀥𑀫𑁆𑀫𑀭𑀓𑁆𑀔𑀸’**𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

1445. 𑀬𑁂 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀦𑀬𑀜𑁆𑀜𑀽 𑀯𑀺𑀦𑀬𑀓𑁄𑀯𑀺𑀤𑀸 𑀞𑀸𑀦𑀸𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀕𑀭𑀼𑀓𑀮𑀳𑀼𑀓𑀲𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀤𑁂𑀲𑀦𑀸𑀦𑀺𑀕𑁆𑀕𑀳𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀑𑀲𑀸𑀭𑀡𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀓𑀼𑀲𑀮𑀸 𑀯𑀺𑀦𑀬𑁂 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑀸, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀦𑀕𑀭𑁂 **‘𑀭𑀽𑀧𑀭𑀓𑁆𑀔𑀸’**𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑁇

_𑀊𑀫𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀸𑀕𑀭𑁂;_  
_𑀬𑀣𑀸𑀬𑀁 𑀤𑀺𑀲𑁆𑀲𑀢𑁂 𑀊𑀫𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑁄 𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


1458. _𑀢𑀣𑀸 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑁄𑀓𑀦𑀼𑀤𑀁,_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀫𑀧𑀭𑀸𑀚𑀺𑀢𑀁;_  
_𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀪𑀯𑀲𑀁𑀲𑀸𑀭𑀫𑁄𑀘𑀦𑀁𑁇_  


1459. _𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀊𑀫𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀬𑀣𑀸 𑀥𑀫𑁆𑀫𑀽𑀫𑀺𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑁄,_  
_𑀅𑀕𑁆𑀕𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


1460. _𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑀁 𑀕𑀺𑀭𑀺𑀁;_  
_𑀬𑀣𑀸 𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑁄 𑀏𑀲𑁄,_  
_𑀳𑀺𑀫𑀯𑀸 𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


1461. _𑀢𑀣𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀕𑀺𑀭𑀺𑀁,_  
_𑀲𑀻𑀢𑀻𑀪𑀽𑀢𑀁 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁;_  
_𑀅𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀢𑀁 𑀪𑀕𑀯𑀢𑁄,_  
_𑀅𑀘𑀮𑀁 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇_  


1462. _𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀥𑀫𑁆𑀫𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀢𑀣𑀸 𑀳𑀺 𑀲𑁄 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀅𑀕𑁆𑀕𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


1463. _𑀬𑀣𑀸𑀧𑀺 𑀕𑀚𑀭𑀸𑀚𑀲𑁆𑀲,_  
_𑀧𑀤𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀸𑀦𑀼𑀲𑀸;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀸 𑀏𑀲𑁄 𑀕𑀚𑁄 𑀇𑀢𑀺𑁇_  


1464. _𑀢𑀣𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑀦𑀸𑀕𑀲𑁆𑀲,_  
_𑀧𑀤𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀯𑀺𑀪𑀸𑀯𑀺𑀦𑁄;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀉𑀍𑀆𑀭𑁄 𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


1465. _𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀪𑀻𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀓𑀼𑀫𑁆𑀫𑀺𑀕𑁂;_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀲𑁆𑀲 𑀲𑀤𑁆𑀤𑁂𑀦,_  
_𑀪𑀻𑀢𑀸𑀫𑁂 𑀓𑀼𑀫𑁆𑀫𑀺𑀕𑀸 𑀇𑀢𑀺𑁇_  


1466. _𑀢𑀣𑁂𑀯 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀺𑀢𑁆𑀣𑀢𑁂 𑀪𑀻𑀢𑀫𑀸𑀦𑀲𑁂;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑁂𑀦 𑀕𑀚𑁆𑀚𑀺𑀢𑀁𑁇_  


1467. _𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀁 𑀧𑀣𑀯𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑀭𑀺𑀢𑀧𑀢𑁆𑀢𑀁 𑀫𑀳𑁄𑀤𑀺𑀓𑀁;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀸𑀫𑁂𑀖𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀁𑁇_  


1468. _𑀢𑀣𑁂𑀯𑀺𑀫𑀁 𑀚𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀆𑀫𑁄𑀤𑀺𑀢𑀧𑀫𑁄𑀤𑀺𑀢𑀁;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀥𑀫𑁆𑀫𑀫𑁂𑀖𑁂𑀦 𑀢𑀧𑁆𑀧𑀺𑀢𑀁𑁇_  


1469. _𑀮𑀕𑁆𑀕𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀼𑀲𑀁 𑀧𑀗𑁆𑀓𑀁,_  
_𑀓𑀮𑀮𑀤𑁆𑀤𑀕𑀢𑀁 𑀫𑀳𑀺𑀁;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀯𑀸𑀭𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀫𑀳𑀸 𑀕𑀢𑁄𑁇_  


1470. _𑀢𑀣𑁂𑀯𑀺𑀫𑀁 𑀚𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀭𑀚𑀧𑀗𑁆𑀓𑀲𑀫𑁄𑀳𑀺𑀢𑀁;_  
_𑀯𑀳𑀺𑀢𑀁 𑀥𑀫𑁆𑀫𑀦𑀤𑀺𑀬𑀸,_  
_𑀯𑀺𑀲𑀝𑁆𑀞𑀁 𑀥𑀫𑁆𑀫𑀲𑀸𑀕𑀭𑁂𑁇_  


1471. _𑀥𑀫𑁆𑀫𑀸𑀫𑀢𑀕𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀤𑁂𑀯𑀓𑀫𑀺𑀫𑀁 𑀫𑀳𑀺𑀁;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀥𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀫𑀳𑀸 𑀕𑀢𑁄𑁇_  


1472. _𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀖𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀕𑀦𑁆𑀥𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀬𑀣𑀸𑀬𑀁 𑀯𑀸𑀬𑀢𑁂 𑀕𑀦𑁆𑀥𑁄,_  
_𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀤𑀼𑀫𑀸𑁇_  


1473. _𑀢𑀣𑁂𑀯𑀸𑀬𑀁 𑀲𑀻𑀮𑀕𑀦𑁆𑀥𑁄,_  
_𑀧𑀯𑀸𑀬𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀜𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀅𑀢𑁆𑀣𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄’𑀢𑀺𑁇_  


1474. 𑀏𑀯𑀭𑀽𑀧𑁂𑀦 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀓𑀸𑀭𑀡𑀲𑀢𑁂𑀦 𑀓𑀸𑀭𑀡𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀳𑁂𑀢𑀼𑀲𑀢𑁂𑀦 𑀳𑁂𑀢𑀼𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀦𑀬𑀲𑀢𑁂𑀦 𑀦𑀬𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀑𑀧𑀫𑁆𑀫𑀲𑀢𑁂𑀦 𑀑𑀧𑀫𑁆𑀫𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀲𑀓𑁆𑀓𑀸 𑀩𑀼𑀤𑁆𑀥𑀩𑀮𑀁 𑀉𑀧𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑀼𑀁𑁇 𑀬𑀣𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀤𑀓𑁆𑀔𑁄 𑀫𑀸𑀮𑀸𑀓𑀸𑀭𑁄 𑀦𑀸𑀦𑀸𑀧𑀼𑀧𑁆𑀨𑀭𑀸𑀲𑀺𑀫𑁆𑀳𑀸 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀧𑀼𑀭𑀺𑀲𑀓𑀸𑀭𑁂𑀦 𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀸𑀮𑀸𑀕𑀼𑀡𑀭𑀸𑀲𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀧𑀼𑀧𑁆𑀨𑀭𑀸𑀲𑀺 𑀯𑀺𑀬 𑀅𑀦𑀦𑁆𑀢𑀕𑀼𑀡𑁄 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀕𑀼𑀡𑁄, 𑀅𑀳𑀫𑁂𑀢𑀭𑀳𑀺 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂 𑀫𑀸𑀮𑀸𑀓𑀸𑀭𑁄 𑀯𑀺𑀬 𑀧𑀼𑀧𑁆𑀨𑀕𑀦𑁆𑀣𑀓𑁄 𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀆𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑁂𑀦𑀧𑀺 𑀫𑀬𑁆𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀺𑀩𑀮𑁂𑀦𑀧𑀺 𑀅𑀲𑀗𑁆𑀔𑁆𑀬𑁂𑀬𑁆𑀬𑁂𑀦𑀧𑀺 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀩𑀮𑀁 𑀤𑀻𑀧𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑁆𑀯𑀁 𑀧𑀦𑁂𑀢𑁆𑀣 𑀙𑀦𑁆𑀤𑀁 𑀚𑀦𑁂𑀳𑀺 𑀲𑀯𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

1475. “𑀤𑀼𑀓𑁆𑀓𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀏𑀯𑀭𑀽𑀧𑁂𑀦 𑀓𑀸𑀭𑀡𑁂𑀦 𑀅𑀦𑀼𑀫𑀸𑀦𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀩𑀮𑀁 𑀉𑀧𑀤𑀲𑁆𑀲𑀬𑀺𑀢𑀼𑀁, 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑀲𑁆𑀫𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀭𑀫𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀳𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑁂𑀦𑀸”𑀢𑀺𑁇

1476. 𑀅𑀦𑀼𑀫𑀸𑀦𑀧𑀜𑁆𑀳𑁄 𑀧𑀞𑀫𑁄𑁇
