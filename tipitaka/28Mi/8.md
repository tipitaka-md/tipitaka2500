# 𑁮 𑀦𑀺𑀕𑀫𑀦

### 𑀦𑀺𑀕𑀫𑀦

(292.)

2024\. 𑀇𑀢𑀺 𑀙𑀲𑀼 𑀓𑀡𑁆𑀟𑁂𑀲𑀼 𑀩𑀸𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑀧𑀢𑀺𑀫𑀡𑁆𑀟𑀺𑀢𑁂𑀲𑀼 𑀤𑁆𑀯𑀸𑀲𑀝𑁆𑀞𑀺𑀅𑀥𑀺𑀓𑀸 𑀤𑁆𑀯𑁂𑀲𑀢𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀧𑁄𑀢𑁆𑀣𑀓𑁂 𑀆𑀕𑀢𑀸 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳𑀸 𑀲𑀫𑀢𑁆𑀢𑀸, 𑀅𑀦𑀸𑀕𑀢𑀸 𑀘 𑀧𑀦 𑀤𑁆𑀯𑀸𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀢𑀸 𑀘 𑀅𑀦𑀸𑀕𑀢𑀸 𑀘 𑀲𑀩𑁆𑀩𑀸 𑀲𑀫𑁄𑀥𑀸𑀦𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀳𑀺 𑀅𑀥𑀺𑀓𑀸 𑀢𑀺𑀲𑀢𑀧𑀜𑁆𑀳𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑀸𑀯 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳𑀸𑀢𑀺 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

2025\. 𑀭𑀜𑁆𑀜𑁄 𑀘 𑀣𑁂𑀭𑀲𑁆𑀲 𑀘 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀺𑀲𑀚𑁆𑀚𑀦𑀸𑀯𑀲𑀸𑀦𑁂 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀬𑁄𑀚𑀦𑀩𑀳𑀮𑀸 𑀉𑀤𑀓𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀬𑀁 𑀫𑀳𑀸𑀧𑀣𑀯𑀻 𑀙𑀥𑀸 𑀓𑀫𑁆𑀧𑀺𑀢𑁆𑀣, 𑀯𑀺𑀚𑁆𑀚𑀼𑀮𑁆𑀮𑀢𑀸 𑀦𑀺𑀘𑁆𑀙𑀭𑀺𑀁𑀲𑀼, 𑀤𑁂𑀯𑀢𑀸 𑀤𑀺𑀩𑁆𑀩𑀧𑀼𑀧𑁆𑀨𑀯𑀲𑁆𑀲𑀁 𑀧𑀯𑀲𑁆𑀲𑀺𑀁𑀲𑀼, 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀸𑀥𑀼𑀓𑀸𑀭𑀫𑀤𑀸𑀲𑀺, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀓𑀼𑀘𑁆𑀙𑀺𑀬𑀁 𑀫𑁂𑀖𑀢𑁆𑀣𑀦𑀺𑀢𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑁄 𑀯𑀺𑀬 𑀫𑀳𑀸𑀖𑁄𑀲𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀇𑀢𑀺 𑀲𑁄 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑁄 𑀭𑀸𑀚𑀸 𑀘 𑀑𑀭𑁄𑀥𑀕𑀡𑀸 𑀘 𑀲𑀺𑀭𑀲𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀯𑀦𑁆𑀤𑀺𑀁𑀲𑀼𑁇

(293.)

2026\. 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑁄 𑀭𑀸𑀚𑀸 𑀅𑀢𑀺𑀯𑀺𑀬 𑀧𑀫𑀼𑀤𑀺𑀢𑀳𑀤𑀬𑁄 𑀲𑀼𑀫𑀣𑀺𑀢𑀫𑀸𑀦𑀳𑀤𑀬𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂 𑀲𑀸𑀭𑀫𑀢𑀺𑀦𑁄 𑀭𑀢𑀦𑀢𑁆𑀢𑀬𑁂 𑀲𑀼𑀦𑀺𑀓𑁆𑀓𑀗𑁆𑀔𑁄 𑀦𑀺𑀕𑁆𑀕𑀼𑀫𑁆𑀩𑁄 𑀦𑀺𑀢𑁆𑀣𑀤𑁆𑀥𑁄 𑀳𑀼𑀢𑁆𑀯𑀸 𑀣𑁂𑀭𑀲𑁆𑀲 𑀕𑀼𑀡𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀲𑀼 𑀧𑀝𑀺𑀧𑀤𑀸𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑁂𑀲𑀼 𑀘 𑀅𑀢𑀺𑀯𑀺𑀬 𑀧𑀲𑀦𑁆𑀦𑁄 𑀯𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁄 𑀦𑀺𑀭𑀸𑀮𑀬𑁄 𑀦𑀺𑀳𑀢𑀫𑀸𑀦𑀢𑁆𑀣𑀫𑁆𑀪𑁄 𑀉𑀤𑁆𑀥𑀝𑀤𑀸𑀞𑁄 𑀯𑀺𑀬 𑀪𑀼𑀚𑀕𑀺𑀦𑁆𑀤𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀩𑀼𑀤𑁆𑀥𑀯𑀺𑀲𑀬𑁄 𑀧𑀜𑁆𑀳𑁄 𑀢𑀬𑀸 𑀯𑀺𑀲𑀚𑁆𑀚𑀺𑀢𑁄, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀢𑁆𑀣𑁂𑀭𑀁 𑀅𑀜𑁆𑀜𑁄 𑀢𑀬𑀸 𑀲𑀤𑀺𑀲𑁄 𑀧𑀜𑁆𑀳𑀯𑀺𑀲𑀚𑁆𑀚𑀦𑁂 𑀦𑀢𑁆𑀣𑀺, 𑀔𑀫𑀣, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀫𑀫 𑀅𑀘𑁆𑀘𑀬𑀁, 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀦𑀸𑀕𑀲𑁂𑀦, 𑀥𑀸𑀭𑁂𑀣 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

(294.)

2027\. 𑀢𑀤𑀸 𑀭𑀸𑀚𑀸 𑀲𑀳 𑀩𑀮𑀓𑀸𑀬𑁂𑀳𑀺 𑀦𑀸𑀕𑀲𑁂𑀦𑀢𑁆𑀣𑁂𑀭𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀁 𑀦𑀸𑀫 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀣𑁂𑀭𑀲𑁆𑀲 𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀳𑀺 𑀧𑀘𑁆𑀘𑀬𑁂𑀳𑀺 𑀦𑀸𑀕𑀲𑁂𑀦𑀁 𑀓𑁄𑀝𑀺𑀲𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀭𑀺𑀘𑀭𑀺, 𑀧𑀼𑀦𑀧𑀺 𑀣𑁂𑀭𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀬 𑀧𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁 𑀯𑀟𑁆𑀠𑁂𑀢𑁆𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀺, 𑀢𑁂𑀦 𑀯𑀼𑀢𑁆𑀢𑀁—

2028\. _“𑀧𑀜𑁆𑀜𑀸 𑀧𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀓𑀢𑀸 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸;_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀯𑀺𑀫𑀢𑀺𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀲𑀦𑁆𑀢𑀺𑀁 𑀧𑀧𑁆𑀧𑁄𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


2029\. _𑀬𑀲𑁆𑀫𑀺𑀁 𑀔𑀦𑁆𑀥𑁂 𑀞𑀺𑀢𑀸 𑀧𑀜𑁆𑀜𑀸,_  
_𑀲𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀦𑀽𑀦𑀓𑀸;_  
_𑀧𑀽𑀚𑀸 𑀯𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑀸𑀥𑀸𑀭𑁄,_  
_𑀅𑀕𑁆𑀕𑁄 𑀲𑁂𑀝𑁆𑀞𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁 𑀳𑀺𑀢𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑀁𑀪𑀺𑀧𑀽𑀚𑁂𑀬𑁆𑀬,_  
_𑀘𑁂𑀢𑀺𑀬𑀁 𑀯𑀺𑀬 𑀲𑀸𑀤𑀭𑁄”𑀢𑀺𑁇_  


2030\. _𑀮𑀗𑁆𑀓𑀸𑀬𑀁 𑀤𑁄𑀡𑀺𑀦𑀕𑀭𑁂,_  
_𑀯𑀲𑀢𑀸 𑀤𑁄𑀡𑀺𑀦𑀸𑀫𑀺𑀦𑀸;_  
_𑀫𑀳𑀸𑀣𑁂𑀭𑁂𑀦 𑀮𑁂𑀔𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀝𑁆𑀞𑀧𑀺𑀢𑀁 𑀬𑀣𑀸𑀲𑀼𑀢𑀁;_  
_𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀭𑀸𑀚𑀧𑀜𑁆𑀳𑁄 𑀘,_  
_𑀦𑀸𑀕𑀲𑁂𑀦𑀯𑀺𑀲𑀚𑁆𑀚𑀦𑀁;_  
_𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑁄 𑀳𑀺 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀦𑀸𑀕𑀲𑁂𑀦𑁄 𑀲𑀼𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


2031\. _𑀇𑀫𑀺𑀦𑀸 𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑁂𑀦,_  
_𑀇𑀢𑁄 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀼𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑀁𑀦𑀸𑀕𑀢𑁂 𑀧𑀲𑁆𑀲𑁂,_  
_𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑀁 𑀥𑀫𑁆𑀫𑀫𑀼𑀢𑁆𑀢𑀫𑀦𑁆𑀢𑀺𑁇_  


---

2032\. 𑀫𑀺𑀮𑀺𑀦𑁆𑀤𑀧𑀜𑁆𑀳𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
