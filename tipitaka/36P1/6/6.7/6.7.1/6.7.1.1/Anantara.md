# 𑀅𑀦𑀦𑁆𑀢𑀭

### 𑀅𑀦𑀦𑁆𑀢𑀭

(13.)

2847. 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2848. 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2849. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀕𑁄𑀢𑁆𑀭𑀪𑀼𑀲𑁆𑀲…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀯𑁄𑀤𑀸𑀦𑀲𑁆𑀲…  𑀆𑀯𑀚𑁆𑀚𑀦𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2850. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀯𑀚𑁆𑀚𑀦𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2851. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀕𑁄𑀢𑁆𑀭𑀪𑀼 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…  𑀯𑁄𑀤𑀸𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…  𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸…  𑀦𑀺𑀭𑁄𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2852. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑀺𑀫𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀦𑀁 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀫𑀕𑁆𑀕𑁄 𑀨𑀮𑀲𑁆𑀲…  𑀨𑀮𑀁 𑀨𑀮𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2853. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀨𑀮𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)
