### 𑀯𑀺𑀧𑀸𑀓

(21.)

2878\. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀯𑀺𑀧𑀸𑀓𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀔𑀦𑁆𑀥𑀸 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2879\. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀯𑀺𑀧𑀸𑀓𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂… 𑁇 (𑁧)

2880\. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀯𑀺𑀧𑀸𑀓𑀸 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2881\. 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀘 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀯𑀺𑀧𑀸𑀓𑁄 𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂… 𑁇 (𑁩)
