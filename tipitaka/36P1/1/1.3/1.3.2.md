# 𑁧𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀯𑀸𑀭

### 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀧𑀘𑁆𑀘𑀦𑀻𑀬

(43.)

90. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇

91. (𑀬𑀣𑀸 𑀅𑀦𑀼𑀮𑁄𑀫𑁂 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁄 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀢𑁄, 𑀏𑀯𑀁 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑁂𑀧𑀺 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁄 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇)

(44.)

92. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑁅𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀆𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑁃𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀛𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀫𑀕𑁆𑀕𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑁄𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑁄𑀦𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑁄𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑁄𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

(45.)

93. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸… 𑁇

94. (𑀬𑀣𑀸 𑀅𑀦𑀼𑀮𑁄𑀫𑁂 𑀏𑀓𑁂𑀓𑀲𑁆𑀲 𑀧𑀤𑀲𑁆𑀲 𑀏𑀓𑀫𑀽𑀮𑀓𑀁, 𑀤𑀼𑀫𑀽𑀮𑀓𑀁, 𑀢𑀺𑀫𑀽𑀮𑀓𑀁, 𑀘𑀢𑀼𑀫𑀽𑀮𑀓𑀁, 𑀬𑀸𑀯 𑀢𑁂𑀯𑀻𑀲𑀢𑀺𑀫𑀽𑀮𑀓𑀁 𑀏𑀯𑀁 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑁂𑀧𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)

95. _𑀢𑀺𑀓𑀜𑁆𑀘 𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀭𑀁 𑀤𑀼𑀓𑀼𑀢𑁆𑀢𑀫𑀁,_  
_𑀤𑀼𑀓𑀁 𑀢𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀢𑀺𑀓𑀁 𑀤𑀼𑀓𑀜𑁆𑀘;_  
_𑀢𑀺𑀓𑀁 𑀢𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀤𑀼𑀓𑀁 𑀤𑀼𑀓𑀜𑁆𑀘,_  
_𑀙 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀫𑁆𑀳𑀺 𑀦𑀬𑀸 𑀲𑀼𑀕𑀫𑁆𑀪𑀻𑀭𑀸𑀢𑀺𑁇 (𑁨)_
