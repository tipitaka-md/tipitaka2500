# 𑁧𑁇𑁩𑁇𑁧𑁇𑁩 𑀳𑁂𑀢𑀼𑀫𑀽𑀮𑀓

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀯𑀸𑀭

### 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀳𑁂𑀢𑀼𑀫𑀽𑀮𑀓

(36.)

74\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑁇

(37.)

75\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…  𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

---

76\. 𑀤𑀼𑀫𑀽𑀮𑀓𑀁𑁇



(38.)

77\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

---

78\. 𑀢𑀺𑀫𑀽𑀮𑀓𑀁𑁇



(39.)

79\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

---

80\. 𑀘𑀢𑀼𑀫𑀽𑀮𑀓𑀁𑁇



81\. (𑀧𑀜𑁆𑀘𑀫𑀽𑀮𑀓𑀸𑀤𑀺𑀓𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸𑁇 𑀏𑀓𑀫𑀽𑀮𑀓𑀁, 𑀤𑀼𑀫𑀽𑀮𑀓𑀁, 𑀢𑀺𑀫𑀽𑀮𑀓𑀁, 𑀘𑀢𑀼𑀫𑀽𑀮𑀓𑀁, 𑀧𑀜𑁆𑀘𑀫𑀽𑀮𑀓𑀁, 𑀲𑀩𑁆𑀩𑀫𑀽𑀮𑀓𑀁 𑀅𑀲𑀫𑁆𑀫𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑁂𑀦 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)

---
