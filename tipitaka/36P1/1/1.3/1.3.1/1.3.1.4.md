# 𑁧𑁇𑁩𑁇𑁧𑁇𑁪 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑀽𑀮𑀓𑀸𑀤𑀺

### 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀯𑀸𑀭

### 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀆𑀭𑀫𑁆𑀫𑀡𑀫𑀽𑀮𑀓𑀸𑀤𑀺

(40.)

83\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

84\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…  𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…  𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

---

85\. 𑀤𑀼𑀫𑀽𑀮𑀓𑀁𑁇



(41.)

86\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

(42.)

87\. 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸…  𑀅𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀯𑀺𑀕𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

88\. (𑀏𑀓𑁂𑀓𑀲𑁆𑀲 𑀧𑀤𑀲𑁆𑀲 𑀏𑀓𑀫𑀽𑀮𑀓𑀁, 𑀤𑀼𑀫𑀽𑀮𑀓𑀁, 𑀢𑀺𑀫𑀽𑀮𑀓𑀁, 𑀘𑀢𑀼𑀫𑀽𑀮𑀓𑀁, 𑀧𑀜𑁆𑀘𑀫𑀽𑀮𑀓𑀁, 𑀲𑀩𑁆𑀩𑀫𑀽𑀮𑀓𑀁 𑀅𑀲𑀫𑁆𑀫𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑁂𑀦 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)

89\. _𑀢𑀺𑀓𑀜𑁆𑀘 𑀧𑀝𑁆𑀞𑀸𑀦𑀯𑀭𑀁 𑀤𑀼𑀓𑀼𑀢𑁆𑀢𑀫𑀁,_  
_𑀤𑀼𑀓𑀁 𑀢𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀢𑀺𑀓𑀁 𑀤𑀼𑀓𑀜𑁆𑀘;_  
_𑀢𑀺𑀓𑀁 𑀢𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀤𑀼𑀓𑀁 𑀤𑀼𑀓𑀜𑁆𑀘,_  
_𑀙 𑀅𑀦𑀼𑀮𑁄𑀫𑀫𑁆𑀳𑀺 𑀦𑀬𑀸 𑀲𑀼𑀕𑀫𑁆𑀪𑀻𑀭𑀸𑀢𑀺𑁇 (𑁧)_
