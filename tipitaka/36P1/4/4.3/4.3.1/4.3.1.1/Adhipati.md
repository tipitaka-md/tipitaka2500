# 𑀅𑀥𑀺𑀧𑀢𑀺

### 𑀅𑀥𑀺𑀧𑀢𑀺

(62.)

2189. 𑀯𑀺𑀧𑀸𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀦𑀢𑁆𑀣𑀺𑁇)

---

2190. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇



2191. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀫𑀳𑀸𑀪𑀽𑀢𑀸…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀁, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁧)

2192. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

2193. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁩)

2194. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑁄 𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁪)

2195. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁫)

(63.)

2196. 𑀯𑀺𑀧𑀸𑀓𑀜𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁧)

2197. 𑀯𑀺𑀧𑀸𑀓𑀜𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁨)

2198. 𑀯𑀺𑀧𑀸𑀓𑀜𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑁄 𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁩)

(64.)

2199. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁧)

2200. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁨)

2201. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁩)
