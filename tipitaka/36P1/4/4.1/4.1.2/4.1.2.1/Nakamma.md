### 𑀦𑀓𑀫𑁆𑀫

(39.)

2121\. 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀢𑀦𑀸𑁇

2122\. 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸—  𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁂𑀯𑀯𑀺𑀧𑀸𑀓𑀦𑀯𑀺𑀧𑀸𑀓𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀢𑀦𑀸; 𑀩𑀸𑀳𑀺𑀭𑀁…  𑀆𑀳𑀸𑀭𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀉𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀬𑁄 𑀫𑀳𑀸𑀪𑀽𑀢𑀸…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀁𑁇
