# 𑀳𑁂𑀢𑀼

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

### 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

### 𑀳𑁂𑀢𑀼

(108.)

421. 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀼𑀲𑀮𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁩)

(109.)

422. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸𑁇 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀓𑀼𑀲𑀮𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁𑁇 (𑁩)

(110.)

423. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀺𑀧𑀸𑀓𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀁; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀯𑀺𑀧𑀸𑀓𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀁, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀁, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀁; 𑀔𑀦𑁆𑀥𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀢𑁆𑀣𑀼, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀔𑀦𑁆𑀥𑀸; 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀫𑀳𑀸𑀪𑀽𑀢𑀸, 𑀢𑀬𑁄 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁, 𑀤𑁆𑀯𑁂 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀫𑀳𑀸𑀪𑀽𑀢𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀁; 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁧)

424. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

425. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀔𑀦𑁆𑀥𑀸𑁇 (𑁩)

426. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁪)

427. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀔𑀦𑁆𑀥𑀸, 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁫)

(111.)

428. 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸𑁇 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀼𑀲𑀮𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀓𑀼𑀲𑀮𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁩)

(112.)

429. 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸𑁇 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀓𑀼𑀲𑀮𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸—  𑀅𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸, 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄, 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸, 𑀅𑀓𑀼𑀲𑀮𑁂 𑀔𑀦𑁆𑀥𑁂 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑁂 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀭𑀽𑀧𑀁𑁇 (𑁩)
