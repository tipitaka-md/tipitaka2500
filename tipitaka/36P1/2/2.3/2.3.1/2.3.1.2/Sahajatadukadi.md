### 𑀲𑀳𑀚𑀸𑀢𑀤𑀼𑀓𑀸𑀤𑀺

(139.)

508. 𑀲𑀳𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀧𑀘𑁆𑀘𑀬𑀸…  𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑀸…  𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑀸…  𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸…𑀧𑁂… 𑁇

509. 𑀆𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑀢𑁆𑀢, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀢𑁆𑀢, 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀢𑁆𑀢, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀲𑀢𑁆𑀢, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀲𑀢𑁆𑀢, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀲𑀢𑁆𑀢, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀓𑀫𑁆𑀫𑁂 𑀲𑀢𑁆𑀢, 𑀆𑀳𑀸𑀭𑁂 𑀲𑀢𑁆𑀢, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀲𑀢𑁆𑀢, 𑀛𑀸𑀦𑁂 𑀲𑀢𑁆𑀢, 𑀫𑀕𑁆𑀕𑁂 𑀲𑀢𑁆𑀢, 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀯𑀺𑀕𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀅𑀯𑀺𑀕𑀢𑁂 𑀲𑀢𑁆𑀢…𑀧𑁂… 𑁇
