# 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢

* 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

* 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢

(209.)

705\. 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀼𑀲𑀮𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑁂 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸; 𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

706\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇

707\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇
