### 𑀦𑀯𑀺𑀧𑀸𑀓

(219.)

721\. 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄…𑀧𑁂…  𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀯𑀺𑀧𑀸𑀓𑀧𑀘𑁆𑀘𑀬𑀸—  𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀏𑀓𑀁 𑀔𑀦𑁆𑀥𑀁𑁇

722\. (𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭𑁂 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀯𑀺𑀪𑀗𑁆𑀕𑁂 𑀦𑀓𑀫𑁆𑀫𑁂 𑀘 𑀦𑀯𑀺𑀧𑀸𑀓𑁂 𑀘 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺 𑀦𑀢𑁆𑀣𑀺; 𑀅𑀯𑀲𑁂𑀲𑁂𑀲𑀼 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀅𑀢𑁆𑀣𑀺𑁇)
