(435.)

1365. 𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑁂𑀭𑀲, 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀢𑁂𑀭𑀲, 𑀦𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀲𑀳𑀚𑀸𑀢𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀲𑀢𑁆𑀢, 𑀦𑁅𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀦𑀯, 𑀦𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀆𑀲𑁂𑀯𑀦𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀓𑀫𑁆𑀫𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀆𑀳𑀸𑀭𑁂 𑀢𑁂𑀭𑀲, 𑀦𑁃𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀛𑀸𑀦𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀫𑀕𑁆𑀕𑁂 𑀢𑁂𑀭𑀲, 𑀦𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀢𑀻𑀡𑀺, 𑀦𑁄𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑁂𑀭𑀲, 𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀢𑁂𑀭𑀲𑁇 (𑁨𑁧)
