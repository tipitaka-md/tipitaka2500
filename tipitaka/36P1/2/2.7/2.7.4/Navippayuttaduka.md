* 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(513.)

1769\. 𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀢𑁆𑀢, 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀢𑁆𑀢, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀢𑀻𑀡𑀺, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀢𑀻𑀡𑀺, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀦𑀯, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀢𑀻𑀡𑀺, 𑀆𑀲𑁂𑀯𑀦𑁂 𑀢𑀻𑀡𑀺, 𑀓𑀫𑁆𑀫𑁂 𑀧𑀜𑁆𑀘, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁, 𑀆𑀳𑀸𑀭𑁂 𑀢𑀻𑀡𑀺, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂 𑀢𑀻𑀡𑀺, 𑀛𑀸𑀦𑁂 𑀢𑀻𑀡𑀺, 𑀫𑀕𑁆𑀕𑁂 𑀢𑀻𑀡𑀺, 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀢𑀻𑀡𑀺, 𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀯𑀺𑀕𑀢𑁂 𑀲𑀢𑁆𑀢, 𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘…𑀧𑁂… 𑁇
