# 𑀓𑀫𑁆𑀫

### 𑀓𑀫𑁆𑀫

(291.)

942\. 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

943\. 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

944\. 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

945\. 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

946\. 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀸, 𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀅𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀦𑀸𑀦𑀸𑀓𑁆𑀔𑀡𑀺𑀓𑀸**—  𑀅𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀯𑀺𑀧𑀸𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

947\. 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

948\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀯𑀺𑀧𑀸𑀓𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀓𑀺𑀭𑀺𑀬𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁, 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀯𑀺𑀧𑀸𑀓𑀸𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘𑁂𑀢𑀦𑀸 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀘𑁂𑀢𑀦𑀸 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)
