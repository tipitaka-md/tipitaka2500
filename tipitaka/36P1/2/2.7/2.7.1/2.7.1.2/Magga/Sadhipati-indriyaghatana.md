(370.)

1123. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀳𑀚𑀸𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑁇 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀳𑀚𑀸𑀢 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜 𑀦𑀺𑀲𑁆𑀲𑀬 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀢𑀻𑀡𑀺𑁇 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀳𑀚𑀸𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀢𑀻𑀡𑀺𑁇 [𑀅𑀯𑀺𑀧𑀸𑀓—  𑁩]

1124. 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀳𑀚𑀸𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀯𑀺𑀧𑀸𑀓 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀏𑀓𑀁𑁇 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀳𑀚𑀸𑀢 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜 𑀦𑀺𑀲𑁆𑀲𑀬 𑀯𑀺𑀧𑀸𑀓 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀏𑀓𑀁𑁇 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀲𑀳𑀚𑀸𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀯𑀺𑀧𑀸𑀓 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀺 𑀅𑀯𑀺𑀕𑀢𑀦𑁆𑀢𑀺 𑀏𑀓𑀁𑁇 [𑀲𑀯𑀺𑀧𑀸𑀓—  𑁩]
