1153\. 𑀅𑀯𑀺𑀕𑀢 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀢𑁂𑀭𑀲𑁇 𑀅𑀯𑀺𑀕𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀢𑁂𑀭𑀲𑁇 𑀅𑀯𑀺𑀕𑀢 𑀅𑀥𑀺𑀧𑀢𑀺 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀅𑀝𑁆𑀞𑁇 𑀅𑀯𑀺𑀕𑀢𑀸𑀥𑀺𑀧𑀢𑀺 𑀦𑀺𑀲𑁆𑀲𑀬 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀅𑀝𑁆𑀞𑁇 𑀅𑀯𑀺𑀕𑀢 𑀆𑀳𑀸𑀭 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀲𑀢𑁆𑀢𑁇 𑀅𑀯𑀺𑀕𑀢 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀲𑀢𑁆𑀢𑁇 𑀅𑀯𑀺𑀕𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀲𑀢𑁆𑀢𑁇 𑀅𑀯𑀺𑀕𑀢 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑁇 𑀅𑀯𑀺𑀕𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀧𑀜𑁆𑀘𑁇 𑀅𑀯𑀺𑀕𑀢 𑀅𑀥𑀺𑀧𑀢𑀺 𑀦𑀺𑀲𑁆𑀲𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑁇 𑀅𑀯𑀺𑀕𑀢 𑀦𑀺𑀲𑁆𑀲𑀬 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬 𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀢𑀻𑀡𑀺𑁇
