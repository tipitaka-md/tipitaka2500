# 𑀅𑀢𑁆𑀣𑀺

2740\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄; 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑁆𑀔𑀡𑁂 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀝𑀢𑁆𑀢𑀸 𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀔𑀦𑁆𑀥𑀸 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

2741\. **𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀲𑁂𑀓𑁆𑀔𑀸 𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀯𑀸 𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁂 𑀯𑀺𑀧𑀸𑀓𑁄 𑀢𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑀁…  𑀖𑀸𑀦𑀁…  𑀚𑀺𑀯𑁆𑀳𑀁…  𑀓𑀸𑀬𑀁…  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀭𑀽𑀧𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁂 𑀯𑀺𑀧𑀸𑀓𑁄 𑀢𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

2742\. **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 **𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄** 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀭𑀽𑀧𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁**—  𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2743\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑀁…  𑀖𑀸𑀦𑀁…  𑀚𑀺𑀯𑁆𑀳𑀁…  𑀓𑀸𑀬𑀁…  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀭𑀽𑀧𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀯𑀢𑁆𑀣𑀼 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 **𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄** 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2744\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀯𑀢𑁆𑀣𑀼 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2745\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 **𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄** 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁪)

2746\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀸𑀦𑀁…𑀧𑁂…  𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀩𑀸𑀳𑀺𑀭𑀁…  𑀆𑀳𑀸𑀭𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁…  𑀉𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀏𑀓𑀁 𑀫𑀳𑀸𑀪𑀽𑀢𑀁…𑀧𑁂… 𑁇 **𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀭𑀽𑀧𑁂…  𑀲𑀤𑁆𑀤𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 **𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄** 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2747\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀭𑀽𑀧𑁂…  𑀲𑀤𑁆𑀤𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁂 𑀯𑀺𑀧𑀸𑀓𑁄 𑀢𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 **𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄** 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2748\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 **𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄** 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2749\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2750\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2751\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2752\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀏𑀓𑀲𑁆𑀲 𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁪)

2753\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀏𑀓𑀲𑁆𑀲 𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀜𑁆𑀘 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀏𑀓𑀲𑁆𑀲 𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁫)

2754\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁬)

2755\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀏𑀓𑀲𑁆𑀲 𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀲𑁆𑀲 𑀇𑀫𑀲𑁆𑀲 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁭)

2756\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀭𑀽𑀧𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2757\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀏𑀓𑁄 𑀔𑀦𑁆𑀥𑁄 𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀢𑀬𑁄 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀏𑀓𑀲𑁆𑀲 𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀤𑁆𑀯𑁂 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2758\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2759\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2760\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2761\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 **𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀭𑀽𑀧𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2762\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀲𑀳𑀚𑀸𑀢𑀁, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁𑁇 **𑀲𑀳𑀚𑀸𑀢𑀸**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀫𑀳𑀸𑀪𑀽𑀢𑀸 𑀘 𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁…𑀧𑁂…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2763\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  **𑀆𑀳𑀸𑀭𑀁…** 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀘 𑀓𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2764\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀁, 𑀆𑀳𑀸𑀭𑀁, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 **𑀧𑀘𑁆𑀙𑀸𑀚𑀸𑀢𑀸**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀔𑀦𑁆𑀥𑀸 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀓𑀩𑀍𑀈𑀓𑀸𑀭𑁄 𑀆𑀳𑀸𑀭𑁄 𑀘 𑀭𑀽𑀧𑀚𑀻𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀜𑁆𑀘 𑀓𑀝𑀢𑁆𑀢𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)
