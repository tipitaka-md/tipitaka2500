### 𑀧𑀼𑀭𑁂𑀚𑀸𑀢

(64.)

2662\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀲𑁂𑀓𑁆𑀔𑀸 𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀯𑀸 𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁂 𑀯𑀺𑀧𑀸𑀓𑁄 𑀢𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑀁…  𑀖𑀸𑀦𑀁…  𑀚𑀺𑀯𑁆𑀳𑀁…  𑀓𑀸𑀬𑀁…  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀭𑀽𑀧𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁂 𑀯𑀺𑀧𑀸𑀓𑁄 𑀢𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁…  𑀭𑀲𑀸𑀬𑀢𑀦𑀁…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2663\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄𑀢𑀁…𑀧𑁂…  𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 **𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀯𑀢𑁆𑀣𑀼 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2664\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀯𑀢𑁆𑀣𑀼 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(65.)

2665\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀭𑀽𑀧𑁂…  𑀲𑀤𑁆𑀤𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀡𑀸𑀢𑀺𑁇 (𑁧)

2666\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁂 𑀭𑀽𑀧𑁂…  𑀲𑀤𑁆𑀤𑁂…  𑀕𑀦𑁆𑀥𑁂…  𑀭𑀲𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑀺, 𑀅𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀺 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺; 𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀭𑀸𑀕𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑁂 𑀦𑀺𑀭𑀼𑀤𑁆𑀥𑁂 𑀯𑀺𑀧𑀸𑀓𑁄 𑀢𑀤𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲…𑀧𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2667\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  **𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2668\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  **𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁, 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀁**𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀜𑁆𑀘 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀦𑀁 𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)
