# 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

* 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀢𑁆𑀢𑀺𑀓

* 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀯𑀺𑀪𑀗𑁆𑀕𑀯𑀸𑀭

* 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬

(61.)

2653\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲…𑀧𑁂…  𑀉𑀢𑀼 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲…𑀧𑁂…  𑀪𑁄𑀚𑀦𑀁 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲…𑀧𑁂…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼…  𑀪𑁄𑀚𑀦𑀁 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2654\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀛𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀅𑀪𑀺𑀜𑁆𑀜𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼…  𑀪𑁄𑀚𑀦𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬…  𑀲𑀻𑀮𑀲𑁆𑀲…  𑀲𑀼𑀢𑀲𑁆𑀲…  𑀘𑀸𑀕𑀲𑁆𑀲…  𑀧𑀜𑁆𑀜𑀸𑀬…  𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀫𑀸𑀦𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2655\. 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀸𑀬𑀺𑀓𑀁 𑀲𑀼𑀔𑀁…  𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁…  𑀉𑀢𑀼…  𑀪𑁄𑀚𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…  𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

(62.)

2656\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀛𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀅𑀪𑀺𑀜𑁆𑀜𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀻𑀮𑀁…  𑀲𑀼𑀢𑀁…  𑀘𑀸𑀕𑀁…  𑀧𑀜𑁆𑀜𑀁…  𑀭𑀸𑀕𑀁…  𑀤𑁄𑀲𑀁…  𑀫𑁄𑀳𑀁…  𑀫𑀸𑀦𑀁…  𑀤𑀺𑀝𑁆𑀞𑀺𑀁…  𑀧𑀢𑁆𑀣𑀦𑀁…  𑀉𑀢𑀼𑀁…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀫𑀸𑀦𑀁 𑀚𑀧𑁆𑀧𑁂𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀸𑀡𑀁 𑀳𑀦𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀸…  𑀲𑀻𑀮𑀁…  𑀲𑀼𑀢𑀁…  𑀘𑀸𑀕𑁄…  𑀧𑀜𑁆𑀜𑀸…  𑀭𑀸𑀕𑁄…  𑀤𑁄𑀲𑁄…  𑀫𑁄𑀳𑁄…  𑀫𑀸𑀦𑁄…  𑀤𑀺𑀝𑁆𑀞𑀺…  𑀧𑀢𑁆𑀣𑀦𑀸…  𑀉𑀢𑀼…  𑀪𑁄𑀚𑀦𑀁…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁…  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸𑀬 𑀲𑀤𑁆𑀥𑀸𑀬…  𑀲𑀻𑀮𑀲𑁆𑀲…  𑀲𑀼𑀢𑀲𑁆𑀲…  𑀘𑀸𑀕𑀲𑁆𑀲…  𑀧𑀜𑁆𑀜𑀸𑀬…  𑀭𑀸𑀕𑀲𑁆𑀲…  𑀤𑁄𑀲𑀲𑁆𑀲…  𑀫𑁄𑀳𑀲𑁆𑀲…  𑀫𑀸𑀦𑀲𑁆𑀲…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…  𑀧𑀢𑁆𑀣𑀦𑀸𑀬 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀞𑀫𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀧𑀞𑀫𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲…𑀧𑁂…  𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲…𑀧𑁂…  𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀲𑁆𑀲…𑀧𑁂…  𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀲𑁆𑀲…𑀧𑁂…  𑀦𑀺𑀬𑀢𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀦𑀺𑀬𑀢𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸…𑀧𑁂… 𑁇 (𑁧)

2657\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀤𑁆𑀥𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀭𑀺𑀬𑀺𑀝𑁆𑀞𑀺𑀫𑀽𑀮𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀁 𑀲𑀻𑀮𑀁…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀢𑀸𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀭𑀺𑀬𑀺𑀝𑁆𑀞𑀺𑀫𑀽𑀮𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲, 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀓𑀼𑀲𑀮𑀸𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑀸𑀓𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2658\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀧𑀞𑀫𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀧𑀞𑀫𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

2659\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀧𑀞𑀫𑁄 𑀫𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀤𑀼𑀢𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀢𑀢𑀺𑀬𑁄 𑀫𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲…𑀧𑁂…  𑀫𑀕𑁆𑀕𑁄 𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

2660\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀅𑀦𑀦𑁆𑀢𑀭𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀨𑀮𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺 𑀓𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

2661\. 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀺𑀬𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄—  𑀆𑀭𑀫𑁆𑀫𑀡𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄, 𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄…𑀧𑁂… 𑁇 **𑀧𑀓𑀢𑀽𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁄**—  𑀅𑀭𑀺𑀬𑀸 𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀢𑁄 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀫𑀕𑁆𑀕𑁄 𑀅𑀭𑀺𑀬𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬 𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬 𑀧𑀝𑀺𑀪𑀸𑀦𑀧𑁆𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸𑀬 𑀞𑀸𑀦𑀸𑀝𑁆𑀞𑀸𑀦𑀓𑁄𑀲𑀮𑁆𑀮𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)
