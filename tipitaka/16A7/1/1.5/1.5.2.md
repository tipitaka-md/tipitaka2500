# 𑁧𑁇𑁫𑁇𑁨 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(45.)

235\. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀸𑀥𑀺𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑁇 𑀬𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀳𑀺 𑀲𑀢𑁆𑀢𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀓𑀕𑁆𑀕𑀢𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀢𑀸, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺 𑀲𑁅𑀧𑀦𑀺𑀲𑁄 𑀇𑀢𑀺𑀧𑀺 𑀲𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄 𑀇𑀢𑀺𑀧𑀻”𑀢𑀺𑁇
