# 𑁧𑁇𑁪𑁇𑁩 𑀧𑀞𑀫𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑁂𑀯𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸𑀲𑀼𑀢𑁆𑀢

(34.)

184. “𑀇𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀤𑁂𑀯𑀢𑀸…𑀧𑁂…  𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀢𑁆𑀣𑀼𑀕𑀸𑀭𑀯𑀢𑀸, 𑀥𑀫𑁆𑀫𑀕𑀸𑀭𑀯𑀢𑀸, 𑀲𑀁𑀖𑀕𑀸𑀭𑀯𑀢𑀸, 𑀲𑀺𑀓𑁆𑀔𑀸𑀕𑀸𑀭𑀯𑀢𑀸, 𑀲𑀫𑀸𑀥𑀺𑀕𑀸𑀭𑀯𑀢𑀸, 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀻𑀢𑀺𑁇

185. _𑀲𑀢𑁆𑀣𑀼𑀕𑀭𑀼 𑀥𑀫𑁆𑀫𑀕𑀭𑀼,_  
_𑀲𑀁𑀖𑁂 𑀘 𑀢𑀺𑀩𑁆𑀩𑀕𑀸𑀭𑀯𑁄;_  
_𑀲𑀫𑀸𑀥𑀺𑀕𑀭𑀼 𑀆𑀢𑀸𑀧𑀻,_  
_𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀢𑀺𑀩𑁆𑀩𑀕𑀸𑀭𑀯𑁄𑁇_  


186. _𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀯𑀘𑁄,_  
_𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀕𑀸𑀭𑀯𑁄;_  
_𑀅𑀪𑀩𑁆𑀩𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲𑁂𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑀢𑀺𑁇_  


---

187. 𑀢𑀢𑀺𑀬𑀁𑁇
