# 𑁧𑁇𑁨𑁇𑁪 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀦𑀼𑀲𑀬𑀯𑀕𑁆𑀕

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀼𑀢𑁆𑀢

(14.)

66\. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄, 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄, 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀻, 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀲𑀤𑁆𑀥𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀭𑀻, 𑀲𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀭𑀻𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
