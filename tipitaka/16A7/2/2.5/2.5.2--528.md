# 𑁨𑁇𑁫𑁇𑁨--𑁫𑁨𑁮

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕

(96--622.)

658\. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀔𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀭𑀸𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀭𑁄𑀥𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂… 𑁇 (𑁨--𑁮)

659\. 𑀲𑁄𑀢𑀲𑁆𑀫𑀺𑀁…𑀧𑁂…  𑀖𑀸𑀦𑀲𑁆𑀫𑀺𑀁…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬…  𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁…  𑀫𑀦𑀲𑁆𑀫𑀺𑀁…𑀧𑁂… 𑁇 (𑁯--𑁪𑁮)

660\. 𑀭𑀽𑀧𑁂𑀲𑀼…𑀧𑁂…  𑀲𑀤𑁆𑀤𑁂𑀲𑀼…  𑀕𑀦𑁆𑀥𑁂𑀲𑀼…  𑀭𑀲𑁂𑀲𑀼…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀲𑀼…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼…𑀧𑁂… 𑁇 (𑁪𑁯--𑁯𑁬)

661\. 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂…𑀧𑁂…  𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂…  𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂…𑀧𑁂… 𑁇 (𑁯𑁭--𑁧𑁪𑁪)

662\. 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂…𑀧𑁂…  𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂…𑀧𑁂… 𑁇 (𑁧𑁪𑁫--𑁧𑁯𑁨)

663\. 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂…  𑀲𑁄𑀢𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀖𑀸𑀦𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀚𑀺𑀯𑁆𑀳𑀸𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀓𑀸𑀬𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…  𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀚𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬…𑀧𑁂… 𑁇 (𑁧𑁯𑁩--𑁨𑁪𑁦)

664\. 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀜𑀸𑀬…  𑀕𑀦𑁆𑀥𑀲𑀜𑁆𑀜𑀸𑀬…  𑀭𑀲𑀲𑀜𑁆𑀜𑀸𑀬…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀜𑀸𑀬…  𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀸𑀬…𑀧𑁂… 𑁇 (𑁨𑁪𑁧--𑁨𑁮𑁮)

665\. 𑀭𑀽𑀧𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬…  𑀕𑀦𑁆𑀥𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬…  𑀭𑀲𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬…  𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸𑀬…𑀧𑁂… 𑁇 (𑁨𑁮𑁯--𑁩𑁩𑁬)

666\. 𑀭𑀽𑀧𑀢𑀡𑁆𑀳𑀸𑀬…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀢𑀡𑁆𑀳𑀸𑀬…  𑀕𑀦𑁆𑀥𑀢𑀡𑁆𑀳𑀸𑀬…  𑀭𑀲𑀢𑀡𑁆𑀳𑀸𑀬…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀢𑀡𑁆𑀳𑀸𑀬…  𑀥𑀫𑁆𑀫𑀢𑀡𑁆𑀳𑀸𑀬…𑀧𑁂… 𑁇 (𑁩𑁩𑁭--𑁩𑁮𑁪)

667\. 𑀭𑀽𑀧𑀯𑀺𑀢𑀓𑁆𑀓𑁂…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀯𑀺𑀢𑀓𑁆𑀓𑁂…  𑀕𑀦𑁆𑀥𑀯𑀺𑀢𑀓𑁆𑀓𑁂…  𑀭𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀯𑀺𑀢𑀓𑁆𑀓𑁂…  𑀥𑀫𑁆𑀫𑀯𑀺𑀢𑀓𑁆𑀓𑁂…𑀧𑁂… 𑁇 (𑁩𑁮𑁫--𑁪𑁩𑁨)

668\. 𑀭𑀽𑀧𑀯𑀺𑀘𑀸𑀭𑁂…𑀧𑁂…  𑀲𑀤𑁆𑀤𑀯𑀺𑀘𑀸𑀭𑁂…  𑀕𑀦𑁆𑀥𑀯𑀺𑀘𑀸𑀭𑁂…  𑀭𑀲𑀯𑀺𑀘𑀸𑀭𑁂…  𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀯𑀺𑀘𑀸𑀭𑁂…  𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀸𑀭𑁂…𑀧𑁂… 𑁇 (𑁪𑁩𑁩--𑁪𑁮𑁦)

669\. 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑁂…𑀧𑁂…  𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁂…  𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂…  𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁂…  𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁂…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀅𑀦𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀔𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀯𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀯𑀺𑀭𑀸𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀦𑀺𑀭𑁄𑀥𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…  𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁪𑁮𑁧--𑁫𑁨𑁮)

670\. _“𑀙𑀤𑁆𑀯𑀸𑀭𑀸𑀭𑀫𑁆𑀫𑀡𑁂𑀲𑁆𑀯𑁂𑀢𑁆𑀣,_  
_𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀲𑀼 𑀘 𑀨𑀲𑁆𑀲𑁂𑀲𑀼;_  
_𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀘 𑀤𑁆𑀯𑀸𑀭𑀲𑁆𑀲,_  
_𑀲𑀼𑀢𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑀼𑀁 𑀅𑀝𑁆𑀞𑁇_  


671\. _𑀲𑀜𑁆𑀜𑀸 𑀲𑀜𑁆𑀘𑁂𑀢𑀦𑀸 𑀢𑀡𑁆𑀳𑀸,_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀘𑀸𑀭𑁂 𑀘;_  
_𑀕𑁄𑀘𑀭𑀲𑁆𑀲 𑀯𑀺𑀲𑀼𑀁 𑀅𑀝𑁆𑀞,_  
_𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀘 𑀧𑀘𑁆𑀘𑁂𑀓𑁂𑁇_  


672\. _𑀲𑁄𑀍𑀅𑀲𑀲𑁆𑀯𑁂𑀢𑁆𑀣 𑀫𑀽𑀮𑁂𑀲𑀼,_  
_𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀦𑀢𑁆𑀢𑀸;_  
_𑀔𑀬𑀸 𑀯𑀬𑀸 𑀯𑀺𑀭𑀸𑀕𑀸 𑀘,_  
_𑀦𑀺𑀭𑁄𑀥𑀸 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀸𑁇_  


673\. _𑀓𑀫𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀦𑀸,_  
_𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸𑀦 𑀯𑀺𑀲𑀼𑀁 𑀯𑀺𑀲𑀼𑀁;_  
_𑀲𑀫𑁆𑀧𑀺𑀡𑁆𑀟𑀺𑀢𑁂𑀲𑀼 𑀲𑀩𑁆𑀩𑁂𑀲𑀼,_  
_𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀘 𑀲𑀢𑀸𑀦𑀺 𑀘;_  
_𑀅𑀝𑁆𑀞𑀯𑀻𑀲𑀢𑀺 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁂 𑀘 𑀯𑀕𑁆𑀕𑀺𑀓𑁂”𑁇_  


---

674\. 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀯𑀕𑁆𑀕𑁄 𑀤𑀲𑀫𑁄𑁇
