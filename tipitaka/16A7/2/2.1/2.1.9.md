# 𑁨𑁇𑁧𑁇𑁯 𑀫𑁂𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀯𑀕𑁆𑀕

### 𑀫𑁂𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(62.)

401\. “𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀪𑀸𑀬𑀺𑀢𑁆𑀣𑁇 𑀲𑀼𑀔𑀲𑁆𑀲𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀬𑀤𑀺𑀤𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺𑁇 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀓𑀢𑀸𑀦𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀇𑀝𑁆𑀞𑀁 𑀓𑀦𑁆𑀢𑀁 𑀫𑀦𑀸𑀧𑀁 𑀯𑀺𑀧𑀸𑀓𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑀽𑀢𑀁𑁇 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀲𑀺𑀁𑁇 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢 𑀲𑀁𑀯𑀝𑁆𑀝𑀯𑀺𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂 𑀦𑀬𑀺𑀫𑀁 𑀮𑁄𑀓𑀁 𑀧𑀼𑀦𑀸𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀲𑀁𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑁂 𑀲𑀼𑀤𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑁄𑀓𑁂 𑀆𑀪𑀲𑁆𑀲𑀭𑀽𑀧𑀕𑁄 𑀳𑁄𑀫𑀺, 𑀯𑀺𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑁂 𑀮𑁄𑀓𑁂 𑀲𑀼𑀜𑁆𑀜𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀺𑀫𑀸𑀦𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺𑁇

402\. 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀳𑁄𑀫𑀺 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀅𑀪𑀺𑀪𑀽 𑀅𑀦𑀪𑀺𑀪𑀽𑀢𑁄 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼𑀤𑀲𑁄 𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑁇 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀅𑀳𑁄𑀲𑀺𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄; 𑀅𑀦𑁂𑀓𑀲𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑀭𑀸𑀚𑀸 𑀘𑀸𑀢𑀼𑀭𑀦𑁆𑀢𑁄 𑀯𑀺𑀚𑀺𑀢𑀸𑀯𑀻 𑀚𑀦𑀧𑀤𑀢𑁆𑀣𑀸𑀯𑀭𑀺𑀬𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀢𑁆𑀢𑀭𑀢𑀦𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀭𑀢𑀦𑀸𑀦𑀺 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀓𑁆𑀓𑀭𑀢𑀦𑀁, 𑀳𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁, 𑀅𑀲𑁆𑀲𑀭𑀢𑀦𑀁, 𑀫𑀡𑀺𑀭𑀢𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀭𑀢𑀦𑀁, 𑀕𑀳𑀧𑀢𑀺𑀭𑀢𑀦𑀁, 𑀧𑀭𑀺𑀡𑀸𑀬𑀓𑀭𑀢𑀦𑀫𑁂𑀯 𑀲𑀢𑁆𑀢𑀫𑀁𑁇 𑀧𑀭𑁄𑀲𑀳𑀲𑁆𑀲𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀲𑀽𑀭𑀸 𑀯𑀻𑀭𑀗𑁆𑀕𑀭𑀽𑀧𑀸 𑀧𑀭𑀲𑁂𑀦𑀧𑁆𑀧𑀫𑀤𑁆𑀤𑀦𑀸𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀧𑀣𑀯𑀺𑀁 𑀲𑀸𑀕𑀭𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀪𑀺𑀯𑀺𑀚𑀺𑀬 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀺𑀦𑁆𑀢𑀺𑁇

403\. _𑀧𑀲𑁆𑀲 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀓𑀁,_  
_𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀲𑀼𑀔𑁂𑀲𑀺𑀦𑁄;_  
_𑀫𑁂𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀯𑁄;_  
_𑀲𑀢𑁆𑀢𑀲𑀁𑀯𑀝𑁆𑀝𑀯𑀺𑀯𑀝𑁆𑀝𑀓𑀧𑁆𑀧𑁂,_  
_𑀦𑀬𑀺𑀫𑀁 𑀮𑁄𑀓𑀁 𑀧𑀼𑀦𑀸𑀕𑀫𑀺𑀁𑁇_  


404\. _𑀲𑀁𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑁂 𑀮𑁄𑀓𑀫𑁆𑀳𑀺,_  
_𑀳𑁄𑀫𑀺 𑀆𑀪𑀲𑁆𑀲𑀭𑀽𑀧𑀕𑁄;_  
_𑀯𑀺𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑁂 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀼𑀜𑁆𑀜𑀩𑁆𑀭𑀳𑁆𑀫𑀽𑀧𑀕𑁄 𑀅𑀳𑀼𑀁𑁇_  


405\. _𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑀸,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀢𑀤𑀸 𑀅𑀳𑀼𑀁;_  
_𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄,_  
_𑀤𑁂𑀯𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


406\. _𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻 𑀅𑀳𑀼𑀁 𑀭𑀸𑀚𑀸,_  
_𑀚𑀫𑁆𑀩𑀼𑀫𑀡𑁆𑀟𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑁄;_  
_𑀫𑀼𑀤𑁆𑀥𑀸𑀯𑀲𑀺𑀢𑁆𑀢𑁄 𑀔𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀥𑀺𑀧𑀢𑀻 𑀅𑀳𑀼𑀁𑁇_  


407\. _𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦,_  
_𑀯𑀺𑀚𑁂𑀬𑁆𑀬 𑀧𑀣𑀯𑀺𑀁 𑀇𑀫𑀁;_  
_𑀅𑀲𑀸𑀳𑀲𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦,_  
_𑀲𑀫𑁂𑀦 𑀫𑀦𑀼𑀲𑀸𑀲𑀺 𑀢𑀁𑁇_  


408\. _𑀥𑀫𑁆𑀫𑁂𑀦 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀧𑀣𑀯𑀺𑀫𑀡𑁆𑀟𑀮𑁂;_  
_𑀫𑀳𑀤𑁆𑀥𑀦𑁂 𑀫𑀳𑀸𑀪𑁄𑀕𑁂,_  
_𑀅𑀟𑁆𑀠𑁂 𑀅𑀚𑀸𑀬𑀺𑀳𑀁 𑀓𑀼𑀮𑁂𑁇_  


409\. _𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀭𑀢𑀦𑁂𑀳𑀺 𑀘 𑀲𑀢𑁆𑀢𑀳𑀺;_  
_𑀩𑀼𑀤𑁆𑀥𑀸 𑀲𑀗𑁆𑀕𑀸𑀳𑀓𑀸 𑀮𑁄𑀓𑁂,_  
_𑀢𑁂𑀳𑀺 𑀏𑀢𑀁 𑀲𑀼𑀤𑁂𑀲𑀺𑀢𑀁𑁇_  


410\. _𑀏𑀲𑁄 𑀳𑁂𑀢𑀼 𑀫𑀳𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀧𑀣𑀩𑁆𑀬𑁄 𑀫𑁂 𑀦 𑀯𑀺𑀧𑀚𑁆𑀚𑀢𑀺;_  
_𑀧𑀳𑀽𑀢𑀯𑀺𑀢𑁆𑀢𑀽𑀧𑀓𑀭𑀡𑁄,_  
_𑀭𑀸𑀚𑀸 𑀳𑁄𑀢𑀺 𑀧𑀢𑀸𑀧𑀯𑀸𑁇_  


411\. _𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀬𑀲𑀯𑀸 𑀳𑁄𑀢𑀺,_  
_𑀚𑀫𑁆𑀩𑀼𑀫𑀡𑁆𑀟𑀲𑁆𑀲 𑀇𑀲𑁆𑀲𑀭𑁄;_  
_𑀓𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀧𑁆𑀧𑀲𑀻𑀤𑁂𑀬𑁆𑀬,_  
_𑀅𑀧𑀺 𑀓𑀡𑁆𑀳𑀸𑀪𑀺𑀚𑀸𑀢𑀺𑀬𑁄𑁇_  


412\. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑁂𑀦,_  
_𑀫𑀳𑀢𑁆𑀢𑀫𑀪𑀺𑀓𑀗𑁆𑀔𑀢𑀸;_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄 𑀕𑀭𑀼𑀓𑀸𑀢𑀩𑁆𑀩𑁄,_  
_𑀲𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

413\. 𑀦𑀯𑀫𑀁𑁇
