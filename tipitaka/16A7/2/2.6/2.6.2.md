# 𑁨𑁇𑁬𑁇𑁨

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀭𑀸𑀕𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(624.)

676\. “𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸, 𑀆𑀤𑀻𑀦𑀯𑀲𑀜𑁆𑀜𑀸, 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀸—  𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
