# 𑁨𑁇𑁬𑁇𑁧

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀭𑀸𑀕𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(623.)

675\. “𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄…𑀧𑁂…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄—  𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
