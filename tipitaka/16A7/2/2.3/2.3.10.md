(84.)

627. “𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀣𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀲𑀫𑀣𑀸𑀬 𑀯𑀽𑀧𑀲𑀫𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀁, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀣𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀲𑀫𑀣𑀸𑀬 𑀯𑀽𑀧𑀲𑀫𑀸𑀬𑀸”𑀢𑀺𑁇
