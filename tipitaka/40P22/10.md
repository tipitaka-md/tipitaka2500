# 𑁧𑁦 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀤𑀼𑀓

(30.)

69\. 𑀦𑀓𑀼𑀲𑀮𑀁 𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

70\. 𑀦𑀓𑀼𑀲𑀮𑀁 𑀦𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀘…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇
