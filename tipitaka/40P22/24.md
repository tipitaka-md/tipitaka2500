# 𑁨𑁪 𑀅𑀢𑀻𑀢𑀢𑁆𑀢𑀺𑀓𑀤𑁆𑀯𑀬, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

### 𑀅𑀢𑀻𑀢𑀢𑁆𑀢𑀺𑀓𑀤𑁆𑀯𑀬, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

(53.)

125\. 𑀦𑀅𑀢𑀻𑀢𑁄 𑀦𑀦𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀦𑀅𑀦𑀸𑀕𑀢𑁄 𑀦𑀦𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀦𑀅𑀢𑀻𑀢𑁄 𑀦𑀦𑀳𑁂𑀢𑀼 𑀘 𑀦𑀅𑀦𑀸𑀕𑀢𑁄 𑀦𑀦𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

126\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀙, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀙…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

(54.)

127\. 𑀦𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

128\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢𑀭𑀲, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑀢𑁆𑀢𑀭𑀲, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀆𑀲𑁂𑀯𑀦𑁂 𑀦𑀯…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀲𑀢𑁆𑀢𑀭𑀲𑁇
