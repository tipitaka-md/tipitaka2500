# 𑁧𑁧𑁇𑁭 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸

* 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢

## 1235\. 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸

(357.)

1236\. 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇

1237\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀩𑀳𑀼𑀓𑀸𑀭𑀸?** 𑀲𑀢𑁆𑀢 𑀅𑀭𑀺𑀬𑀥𑀦𑀸𑀦𑀺—  𑀲𑀤𑁆𑀥𑀸𑀥𑀦𑀁, 𑀲𑀻𑀮𑀥𑀦𑀁, 𑀳𑀺𑀭𑀺𑀥𑀦𑀁, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀥𑀦𑀁, 𑀲𑀼𑀢𑀥𑀦𑀁, 𑀘𑀸𑀕𑀥𑀦𑀁, 𑀧𑀜𑁆𑀜𑀸𑀥𑀦𑀁𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸𑁇 (𑁧)

1238\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀲𑀢𑁆𑀢 𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸—  𑀲𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀥𑀫𑁆𑀫𑀯𑀺𑀘𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀯𑀻𑀭𑀺𑀬𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀻𑀢𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀲𑀫𑀸𑀥𑀺𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀫𑁆𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑁄𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁨)

1239\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀲𑀢𑁆𑀢 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑁄—  𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘 𑀤𑁂𑀯𑀸 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀺𑀓𑀸𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 (𑁩𑁇𑁧)

1240\. 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀏𑀓𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀤𑁂𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸 𑀧𑀞𑀫𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀸𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 (𑁩𑁇𑁨)

1241\. 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀏𑀓𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀤𑁂𑀯𑀸 𑀆𑀪𑀲𑁆𑀲𑀭𑀸𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 (𑁩𑁇𑁩)

1242\. 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀏𑀓𑀢𑁆𑀢𑀓𑀸𑀬𑀸 𑀏𑀓𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀺𑀦𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀤𑁂𑀯𑀸 𑀲𑀼𑀪𑀓𑀺𑀡𑁆𑀳𑀸𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 (𑁩𑁇𑁪)

1243\. 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸…𑀧𑁂…  ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 (𑁩𑁇𑁫)

1244\. 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 (𑁩𑁇𑁬)

1245\. 𑀲𑀦𑁆𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀽𑀧𑀕𑀸𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑀻 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀝𑁆𑀞𑀺𑀢𑀺𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁩𑁇𑁭)

1246\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀲𑀢𑁆𑀢𑀸𑀦𑀼𑀲𑀬𑀸—  𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄, 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄, 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄, 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄, 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁪)

1247\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀲𑀢𑁆𑀢 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀳𑀺𑀭𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀼𑀲𑀻𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁫)

1248\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀲𑀢𑁆𑀢 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀳𑀺𑀭𑀺𑀫𑀸 𑀳𑁄𑀢𑀺, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁬)

1249\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸?** 𑀲𑀢𑁆𑀢 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀥𑀫𑁆𑀫𑀸—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀜𑁆𑀜𑀽 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑀽 𑀘 𑀅𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀽 𑀘 𑀓𑀸𑀮𑀜𑁆𑀜𑀽 𑀘 𑀧𑀭𑀺𑀲𑀜𑁆𑀜𑀽 𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀜𑁆𑀜𑀽 𑀘𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸𑁇 (𑁭)

1250\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀲𑀢𑁆𑀢 𑀲𑀜𑁆𑀜𑀸—  𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸, 𑀆𑀤𑀻𑀦𑀯𑀲𑀜𑁆𑀜𑀸, 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀸𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

1251\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀲𑀢𑁆𑀢 𑀦𑀺𑀤𑁆𑀤𑀲𑀯𑀢𑁆𑀣𑀽𑀦𑀺—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀫𑀸𑀤𑀸𑀦𑁂 𑀢𑀺𑀩𑁆𑀩𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀫𑀸𑀤𑀸𑀦𑁂 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄𑁇 𑀥𑀫𑁆𑀫𑀦𑀺𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀢𑀺𑀩𑁆𑀩𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀦𑀺𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄𑁇 𑀇𑀘𑁆𑀙𑀸𑀯𑀺𑀦𑀬𑁂 𑀢𑀺𑀩𑁆𑀩𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀇𑀘𑁆𑀙𑀸𑀯𑀺𑀦𑀬𑁂 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄𑁇 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑁂 𑀢𑀺𑀩𑁆𑀩𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑁂 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄𑁇 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁂 𑀢𑀺𑀩𑁆𑀩𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑁂 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄𑁇 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂 𑀢𑀺𑀩𑁆𑀩𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄𑁇 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀝𑀺𑀯𑁂𑀥𑁂 𑀢𑀺𑀩𑁆𑀩𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀝𑀺𑀯𑁂𑀥𑁂 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁯)

1252\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 **𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀲𑀢𑁆𑀢 𑀔𑀻𑀡𑀸𑀲𑀯𑀩𑀮𑀸𑀦𑀺—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀲𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑁄 𑀲𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀁 𑀩𑀮𑀁 𑀆𑀕𑀫𑁆𑀫 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺𑁇 (𑁧𑁦𑁇𑁧)

1253\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀗𑁆𑀕𑀸𑀭𑀓𑀸𑀲𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄…𑀧𑁂…  ‘𑀔𑀻𑀡𑀸 𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺𑁇 (𑁧𑁦𑁇𑁨)

1254\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀦𑁆𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑁂𑀓𑀧𑁄𑀡𑀁 𑀯𑀺𑀯𑁂𑀓𑀧𑀩𑁆𑀪𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀝𑁆𑀞𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀸𑀪𑀺𑀭𑀢𑀁 𑀩𑁆𑀬𑀦𑁆𑀢𑀻𑀪𑀽𑀢𑀁 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀲𑀯𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄…𑀧𑁂…  ‘𑀔𑀻𑀡𑀸 𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺𑁇 (𑁧𑁦𑁇𑁩)

1255\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀸𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄…𑀧𑁂…  ‘𑀔𑀻𑀡𑀸 𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺𑁇 (𑁧𑁦𑁇𑁪)

1256\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀸𑀦𑀺𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄…𑀧𑁂…  ‘𑀔𑀻𑀡𑀸 𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺𑁇 (𑁧𑁦𑁇𑁫)

1257\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀸𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄…𑀧𑁂…  ‘𑀔𑀻𑀡𑀸 𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺𑁇 (𑁧𑁦𑁇𑁬)

1258\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄, 𑀇𑀤𑀫𑁆𑀧𑀺 𑀔𑀻𑀡𑀸𑀲𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀩𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀁 𑀩𑀮𑀁 𑀆𑀕𑀫𑁆𑀫 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀫𑁂 𑀆𑀲𑀯𑀸’𑀢𑀺𑁇 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁧𑁦𑁇𑁭)

1259\. 𑀇𑀢𑀺𑀫𑁂 𑀲𑀢𑁆𑀢𑀢𑀺 𑀥𑀫𑁆𑀫𑀸 𑀪𑀽𑀢𑀸 𑀢𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀅𑀯𑀺𑀢𑀣𑀸 𑀅𑀦𑀜𑁆𑀜𑀣𑀸 𑀲𑀫𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸𑁇

---
