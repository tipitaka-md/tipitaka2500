# 𑁧𑁧𑁇𑁫 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸

* 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢

## 1186\. 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸

(355.)

1187\. 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇

1188\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀩𑀳𑀼𑀓𑀸𑀭𑀸?** 𑀧𑀜𑁆𑀘 𑀧𑀥𑀸𑀦𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀁—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑁄 𑀲𑀫𑀯𑁂𑀧𑀸𑀓𑀺𑀦𑀺𑀬𑀸 𑀕𑀳𑀡𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀦𑀸𑀢𑀺𑀲𑀻𑀢𑀸𑀬 𑀦𑀸𑀘𑁆𑀘𑀼𑀡𑁆𑀳𑀸𑀬 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑀬 𑀧𑀥𑀸𑀦𑀓𑁆𑀔𑀫𑀸𑀬𑁇 𑀅𑀲𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀫𑀸𑀬𑀸𑀯𑀻 𑀬𑀣𑀸𑀪𑀽𑀢𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀯𑀻𑀓𑀢𑁆𑀢𑀸 𑀲𑀢𑁆𑀣𑀭𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀲𑀼 𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼𑁇 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀤𑀬𑀢𑁆𑀣𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸𑁇 (𑁧)

1189\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  𑀧𑀻𑀢𑀺𑀨𑀭𑀡𑀢𑀸, 𑀲𑀼𑀔𑀨𑀭𑀡𑀢𑀸, 𑀘𑁂𑀢𑁄𑀨𑀭𑀡𑀢𑀸, 𑀆𑀮𑁄𑀓𑀨𑀭𑀡𑀢𑀸, 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀡𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁨)

1190\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸—  𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑁂𑀤𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀜𑁆𑀜𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀗𑁆𑀔𑀸𑀭𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁩)

1191\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀧𑀜𑁆𑀘 𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀺—  𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁪)

1192\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀧𑀜𑁆𑀘 𑀘𑁂𑀢𑁄𑀔𑀺𑀮𑀸—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀭𑀺 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀦 𑀲𑀫𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀬𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀭𑀺 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀦 𑀲𑀫𑁆𑀧𑀲𑀻𑀤𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀘𑁂𑀢𑁄𑀔𑀺𑀮𑁄𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺…𑀧𑁂…  𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑀺𑀮𑀚𑀸𑀢𑁄, 𑀬𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑀺𑀮𑀚𑀸𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀘𑁂𑀢𑁄𑀔𑀺𑀮𑁄𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁫)

1193\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺—  𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁬)

1194\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸?** 𑀧𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀺𑀬𑀸 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀓𑀸𑀫𑁂𑀳𑀺𑁇 𑀬𑁂 𑀘 𑀓𑀸𑀫𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺𑁇 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀓𑀸𑀫𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁭𑁇𑁧)

1195\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁂𑀦𑁇 𑀬𑁂 𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺𑁇 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁭𑁇𑁨)

1196\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀳𑁂𑀲𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀯𑀺𑀳𑁂𑀲𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑀸𑀬𑁇 𑀬𑁂 𑀘 𑀯𑀺𑀳𑁂𑀲𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺𑁇 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁭𑁇𑁩)

1197\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀽𑀧𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀭𑀽𑀧𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀭𑀽𑀧𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑁂𑀳𑀺𑁇 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺𑁇 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁭𑁇𑁪)

1198\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀓𑁆𑀓𑀸𑀬𑁂𑀦𑁇 𑀬𑁂 𑀘 𑀲𑀓𑁆𑀓𑀸𑀬𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺𑁇 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸𑁇 (𑁭𑁇𑁫)

1199\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀧𑀜𑁆𑀘 𑀜𑀸𑀡𑀺𑀓𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺—  ‘𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑀼𑀔𑁄 𑀘𑁂𑀯 𑀆𑀬𑀢𑀺𑀜𑁆𑀘 𑀲𑀼𑀔𑀯𑀺𑀧𑀸𑀓𑁄’𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑀬𑁂𑀯 𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 ‘𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀅𑀭𑀺𑀬𑁄 𑀦𑀺𑀭𑀸𑀫𑀺𑀲𑁄’𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀜𑁆𑀜𑁂𑀯 𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 ‘𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀅𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁂𑀯𑀺𑀢𑁄’𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑀬𑁂𑀯 𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 ‘𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀲𑀦𑁆𑀢𑁄 𑀧𑀡𑀻𑀢𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀮𑀤𑁆𑀥𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀸𑀥𑀺𑀕𑀢𑁄, 𑀦 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀦𑀺𑀕𑁆𑀕𑀬𑁆𑀳𑀯𑀸𑀭𑀺𑀢𑀕𑀢𑁄’𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑀬𑁂𑀯 𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 ‘𑀲𑁄 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀢𑁄𑀯 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀺 𑀲𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀳𑀸𑀫𑀻’𑀢𑀺 𑀧𑀘𑁆𑀘𑀢𑁆𑀢𑀁𑀬𑁂𑀯 𑀜𑀸𑀡𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

1200\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀧𑀜𑁆𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀸𑀦𑀺—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁧)

1201\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑁂𑀲𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀬𑀣𑀸 𑀬𑀣𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑁂𑀲𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁨)

1202\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻, 𑀦𑀸𑀧𑀺 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑁂𑀲𑀁 𑀤𑁂𑀲𑁂𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀚𑁆𑀛𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀚𑁆𑀛𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁩)

1203\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻, 𑀦𑀸𑀧𑀺 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑁂𑀲𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀚𑁆𑀛𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀲𑀸 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀢𑀺𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀲𑀸 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀢𑀺 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁪)

1204\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻, 𑀦𑀸𑀧𑀺 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑁂𑀲𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀚𑁆𑀛𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀦𑀸𑀧𑀺 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀲𑀸 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀢𑀺; 𑀅𑀧𑀺 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀁 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀁 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀁 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀁 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁 𑀧𑀜𑁆𑀜𑀸𑀬 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁯𑁇𑁫)

1205\. 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 **𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀸—  𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁧𑁦)

1206\. 𑀇𑀢𑀺 𑀇𑀫𑁂 𑀧𑀜𑁆𑀜𑀸𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀽𑀢𑀸 𑀢𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀅𑀯𑀺𑀢𑀣𑀸 𑀅𑀦𑀜𑁆𑀜𑀣𑀸 𑀲𑀫𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸𑁇
