# 𑁧𑁧𑁇𑁩 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸

* 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢

## 1160\. 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸

(353.)

1161\. 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸, 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸…𑀧𑁂…  𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇

1162\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀩𑀳𑀼𑀓𑀸𑀭𑀸?** 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑁂𑀯𑁄, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀁, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸𑁇 (𑁧)

1163\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀢𑀬𑁄 𑀲𑀫𑀸𑀥𑀻—  𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀲𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀥𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀅𑀯𑀺𑀘𑀸𑀭𑁄 𑀲𑀫𑀸𑀥𑀺𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁨)

1164\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑁂𑀤𑀦𑀸—  𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀤𑀼𑀓𑁆𑀔𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀸 𑀯𑁂𑀤𑀦𑀸𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁩)

1165\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀢𑀺𑀲𑁆𑀲𑁄 𑀢𑀡𑁆𑀳𑀸—  𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸, 𑀪𑀯𑀢𑀡𑁆𑀳𑀸, 𑀯𑀺𑀪𑀯𑀢𑀡𑁆𑀳𑀸𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁪)

1166\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀤𑁄𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁫)

1167\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀢𑀻𑀡𑀺 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀅𑀫𑁄𑀳𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁬)

1168\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸?** 𑀢𑀺𑀲𑁆𑀲𑁄 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀺𑀬𑀸 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀓𑀸𑀫𑀸𑀦𑀫𑁂𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀤𑀺𑀤𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁, 𑀭𑀽𑀧𑀸𑀦𑀫𑁂𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀬𑀤𑀺𑀤𑀁 𑀅𑀭𑀽𑀧𑀁, 𑀬𑀁 𑀔𑁄 𑀧𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀪𑀽𑀢𑀁 𑀲𑀗𑁆𑀔𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀦𑀺𑀭𑁄𑀥𑁄 𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸𑁇 (𑁭)

1169\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀢𑀻𑀡𑀺 𑀜𑀸𑀡𑀸𑀦𑀺—  𑀅𑀢𑀻𑀢𑀁𑀲𑁂 𑀜𑀸𑀡𑀁, 𑀅𑀦𑀸𑀕𑀢𑀁𑀲𑁂 𑀜𑀸𑀡𑀁, 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑀲𑁂 𑀜𑀸𑀡𑀁𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

1170\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀢𑀺𑀲𑁆𑀲𑁄 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀓𑀸𑀫𑀥𑀸𑀢𑀼, 𑀭𑀽𑀧𑀥𑀸𑀢𑀼, 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁯)

1171\. 𑀓𑀢𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 **𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸—  𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀲𑁆𑀲𑀢𑀺𑀜𑀸𑀡𑀁 𑀯𑀺𑀚𑁆𑀚𑀸, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀘𑀼𑀢𑀽𑀧𑀧𑀸𑀢𑁂 𑀜𑀸𑀡𑀁 𑀯𑀺𑀚𑁆𑀚𑀸, 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑁂 𑀜𑀸𑀡𑀁 𑀯𑀺𑀚𑁆𑀚𑀸𑁇 𑀇𑀫𑁂 𑀢𑀬𑁄 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁧𑁦)

1172\. 𑀇𑀢𑀺 𑀇𑀫𑁂 𑀢𑀺𑀁𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀽𑀢𑀸 𑀢𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀅𑀯𑀺𑀢𑀣𑀸 𑀅𑀦𑀜𑁆𑀜𑀣𑀸 𑀲𑀫𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸𑁇
