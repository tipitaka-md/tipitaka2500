# 𑁧𑁧𑁇𑁨 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸

* 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢

## 1147\. 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸

(352.)

1148\. 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸, 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇

1149\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀩𑀳𑀼𑀓𑀸𑀭𑀸?** 𑀲𑀢𑀺 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸𑁇 (𑁧)

1150\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁨)

1151\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁩)

1152\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀘𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁪)

1153\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀘 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀘𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁫)

1154\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀘𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁬)

1155\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸?** 𑀬𑁄 𑀘 𑀳𑁂𑀢𑀼 𑀬𑁄 𑀘 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀸𑀬, 𑀬𑁄 𑀘 𑀳𑁂𑀢𑀼 𑀬𑁄 𑀘 𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸𑁇 (𑁭)

1156\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀤𑁆𑀯𑁂 𑀜𑀸𑀡𑀸𑀦𑀺—  𑀔𑀬𑁂 𑀜𑀸𑀡𑀁, 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁂 𑀜𑀸𑀡𑀁𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

1157\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀤𑁆𑀯𑁂 𑀥𑀸𑀢𑀼𑀬𑁄—  𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼 𑀅𑀲𑀗𑁆𑀔𑀢𑀸 𑀘 𑀥𑀸𑀢𑀼𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁯)

1158\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 **𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀘𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁧𑁦)

1159\. 𑀇𑀢𑀺 𑀇𑀫𑁂 𑀯𑀻𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀸 𑀪𑀽𑀢𑀸 𑀢𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀅𑀯𑀺𑀢𑀣𑀸 𑀅𑀦𑀜𑁆𑀜𑀣𑀸 𑀲𑀫𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸𑁇
