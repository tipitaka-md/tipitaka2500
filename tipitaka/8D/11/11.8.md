# 𑁧𑁧𑁇𑁮 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸

* 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢

## 1261\. 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸

(358.)

1262\. 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸…𑀧𑁂…  𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇

1263\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀩𑀳𑀼𑀓𑀸𑀭𑀸?** 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀁 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑁂𑀫𑀜𑁆𑀘 𑀕𑀸𑀭𑀯𑁄 𑀘𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀳𑁂𑀢𑀼 𑀧𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧𑁇𑁧)

1264\. 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀁 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑁂𑀫𑀜𑁆𑀘 𑀕𑀸𑀭𑀯𑁄 𑀘𑁇 𑀢𑁂 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺 𑀧𑀭𑀺𑀧𑀜𑁆𑀳𑀢𑀺—  ‘𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀣𑀁? 𑀇𑀫𑀲𑁆𑀲 𑀓𑁄 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀯𑀺𑀯𑀝𑀜𑁆𑀘𑁂𑀯 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺, 𑀅𑀦𑀼𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀢𑀜𑁆𑀘 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀺 , 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀘 𑀓𑀗𑁆𑀔𑀸𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬, 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧𑁇𑁨)

1265\. 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀬𑁂𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺—  𑀓𑀸𑀬𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀘𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼 𑀢𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧𑁇𑁩)

1266\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀳𑁂𑀢𑀼 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧𑁇𑁪)

1267\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀢𑀥𑀭𑁄 𑀲𑀼𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄𑁇 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀲𑀸𑀢𑁆𑀣𑀸 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀥𑀸𑀢𑀸 𑀯𑀘𑀲𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀳𑁂𑀢𑀼 𑀧𑀜𑁆𑀘𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧𑁇𑁫)

1268\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀳𑁂𑀢𑀼 𑀙𑀝𑁆𑀞𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧𑁇𑁬)

1269\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀺𑀫𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀫𑁂𑀦 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄𑁇 𑀘𑀺𑀭𑀓𑀢𑀫𑁆𑀧𑀺 𑀘𑀺𑀭𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑀺 𑀲𑀭𑀺𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀳𑁂𑀢𑀼 𑀲𑀢𑁆𑀢𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧𑁇𑁭)

1270\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀘𑀲𑀼 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼, 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺—  ‘𑀇𑀢𑀺 𑀭𑀽𑀧𑀁 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸𑀬 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀳𑁂𑀢𑀼 𑀅𑀝𑁆𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸𑁇 (𑁧𑁇𑁮)

1271\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁨)

1272\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀅𑀝𑁆𑀞 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀸—  𑀮𑀸𑀪𑁄 𑀘, 𑀅𑀮𑀸𑀪𑁄 𑀘, 𑀬𑀲𑁄 𑀘, 𑀅𑀬𑀲𑁄 𑀘, 𑀦𑀺𑀦𑁆𑀤𑀸 𑀘, 𑀧𑀲𑀁𑀲𑀸 𑀘, 𑀲𑀼𑀔𑀜𑁆𑀘, 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁩)

1273\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀅𑀝𑁆𑀞 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸—  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁪)

1274\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀅𑀝𑁆𑀞 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫𑁇𑁧)

1275\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑀁, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫𑁇𑁨)

1276\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀕𑁆𑀕𑁄 𑀔𑁄 𑀫𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫𑁇𑁩)

1277\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀫𑀕𑁆𑀕𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁, 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫𑁇𑁪)

1278\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫𑁇𑁫)

1279\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀕𑀭𑀼𑀓𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀫𑀸𑀲𑀸𑀘𑀺𑀢𑀁 𑀫𑀜𑁆𑀜𑁂, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀙𑀝𑁆𑀞𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫𑁇𑁬)

1280\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄, 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀧𑁆𑀧𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫𑁇𑁭)

1281\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑀸𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁫𑁇𑁮)

1282\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀅𑀝𑁆𑀞 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀦 𑀦 𑀲𑀼𑀓𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺𑀓𑀸𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬𑁇𑁧)

1283\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑀁, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺𑀓𑀸𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬𑁇𑁨)

1284\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀕𑁆𑀕𑁄 𑀔𑁄 𑀫𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑀦 𑀦 𑀲𑀼𑀓𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺𑀓𑀸𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬𑁇𑁩)

1285\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀫𑀕𑁆𑀕𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁, 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺𑀓𑀸𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬𑁇𑁪)

1286\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀮𑀳𑀼𑀓𑁄 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬𑁇𑁫)

1287\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀩𑀮𑀯𑀸 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀙𑀝𑁆𑀞𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬𑁇𑁬)

1288\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀫𑁂 𑀆𑀩𑀸𑀥𑁄 𑀧𑀯𑀟𑁆𑀠𑁂𑀬𑁆𑀬, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬𑁇𑁭)

1289\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸, 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀫𑁂 𑀆𑀩𑀸𑀥𑁄 𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀲𑁂𑀲𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁬𑁇𑁮)

1290\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸?** 𑀅𑀝𑁆𑀞 𑀅𑀓𑁆𑀔𑀡𑀸 𑀅𑀲𑀫𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀥𑀫𑁆𑀫𑁄 𑀘 𑀤𑁂𑀲𑀺𑀬𑀢𑀺 𑀑𑀧𑀲𑀫𑀺𑀓𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀓𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀻 𑀲𑀼𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁄𑁇 𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭𑁇𑁧)

1291\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀥𑀫𑁆𑀫𑁄 𑀘 𑀤𑁂𑀲𑀺𑀬𑀢𑀺 𑀑𑀧𑀲𑀫𑀺𑀓𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀓𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀻 𑀲𑀼𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁄𑁇 𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭𑁇𑁨)

1292\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁…𑀧𑁂…  𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭𑁇𑁩)

1293\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀁 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭𑁇𑁪)

1294\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁…𑀧𑁂…  𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺 𑀫𑀺𑀮𑀓𑁆𑀔𑁂𑀲𑀼 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁂𑀲𑀼, 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀕𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭𑁇𑁫)

1295\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀯𑀺𑀧𑀭𑀻𑀢𑀤𑀲𑁆𑀲𑀦𑁄—  ‘𑀦𑀢𑁆𑀣𑀺 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀬𑀺𑀝𑁆𑀞𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀳𑀼𑀢𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀅𑀬𑀁 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀫𑀸𑀢𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀧𑀺𑀢𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀫𑁆𑀫𑀕𑁆𑀕𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀬𑁂 𑀇𑀫𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀧𑀭𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭𑁇𑁬)

1296\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀚𑀍𑀑 𑀏𑀍𑀅𑀫𑀽𑀕𑁄, 𑀦𑀧𑁆𑀧𑀝𑀺𑀩𑀮𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀫𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭𑁇𑁭)

1297\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀜𑁆𑀜𑀯𑀸 𑀅𑀚𑀍𑀑 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑁄, 𑀧𑀝𑀺𑀩𑀮𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀫𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸𑁇 (𑁭𑁇𑁮)

1298\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸—  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑁇 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑁇 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑁇 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲𑁇 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑀢𑀺𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑁇 𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑁇 𑀧𑀜𑁆𑀜𑀯𑀢𑁄 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀲𑁆𑀲𑁇 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑀸𑀢𑀺 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

1299\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀅𑀝𑁆𑀞 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀸𑀦𑀺—  𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁧)

1300\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁨)

1301\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁩)

1302\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁪)

1303\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀦𑀻𑀮𑀸𑀦𑀺 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀁 𑀦𑀻𑀮𑀁 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀁 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀦𑀻𑀮𑀁 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀁 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀦𑀻𑀮𑀸𑀦𑀺 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁫)

1304\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀻𑀢𑀸𑀦𑀺 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀓𑀡𑀺𑀓𑀸𑀭𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀻𑀢𑀁 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀁 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀧𑀻𑀢𑀁 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀁 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀻𑀢𑀸𑀦𑀺 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀙𑀝𑁆𑀞𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁬)

1305\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑀧𑀼𑀧𑁆𑀨𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁯𑁇𑁭)

1306\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀑𑀤𑀸𑀢𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀑𑀲𑀥𑀺𑀢𑀸𑀭𑀓𑀸 𑀑𑀤𑀸𑀢𑀸 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀑𑀤𑀸𑀢𑀁 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀁 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀑𑀤𑀸𑀢𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁯𑁇𑁮)

1307\. 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 **𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀅𑀝𑁆𑀞 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸—  𑀭𑀽𑀧𑀻 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦𑁇𑁧)

1308\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦𑁇𑁨)

1309\. 𑀲𑀼𑀪𑀦𑁆𑀢𑁂𑀯 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦𑁇𑁩)

1310\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦𑁇𑁪)

1311\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦𑁇𑁫)

1312\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦𑁇𑁬)

1313\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁦𑁇𑁭)

1314\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁧𑁦𑁇𑁮)

1315\. 𑀇𑀢𑀺 𑀇𑀫𑁂 𑀅𑀲𑀻𑀢𑀺 𑀥𑀫𑁆𑀫𑀸 𑀪𑀽𑀢𑀸 𑀢𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀅𑀯𑀺𑀢𑀣𑀸 𑀅𑀦𑀜𑁆𑀜𑀣𑀸 𑀲𑀫𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸𑁇
