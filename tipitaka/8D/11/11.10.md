# 𑁧𑁧𑁇𑁧𑁦 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸

### 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢

## 1337. 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸

(360.)

1338. 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸…𑀧𑁂…  𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇

1339. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 **𑀩𑀳𑀼𑀓𑀸𑀭𑀸?** 𑀤𑀲 𑀦𑀸𑀣𑀓𑀭𑀡𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁧)

1340. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄…𑀧𑁂…  𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁨)

1341. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁩)

1342. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀔𑀫𑁄 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁪)

1343. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀢𑀢𑁆𑀣 𑀤𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑁄 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁫)

1344. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀺𑀬𑀲𑀫𑀼𑀤𑀸𑀳𑀸𑀭𑁄 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁂 𑀉𑀍𑀆𑀭𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁄𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀉𑀍𑀆𑀭𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁬)

1345. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀳𑀺 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁭)

1346. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁮)

1347. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀺𑀫𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀫𑁂𑀦 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀘𑀺𑀭𑀓𑀢𑀫𑁆𑀧𑀺 𑀘𑀺𑀭𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑀺 𑀲𑀭𑀺𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁯)

1348. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀤𑀬𑀢𑁆𑀣𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀭𑀺𑀬𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼…𑀧𑁂…  𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀓𑀸𑀭𑀸𑁇 (𑁧𑁇𑁧𑁦)

1349. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 **𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀤𑀲 𑀓𑀲𑀺𑀡𑀸𑀬𑀢𑀦𑀸𑀦𑀺—  𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺 𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀯𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀆𑀧𑁄𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀚𑁄𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀯𑀸𑀬𑁄𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀦𑀻𑀮𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀧𑀻𑀢𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀮𑁄𑀳𑀺𑀢𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀑𑀤𑀸𑀢𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀆𑀓𑀸𑀲𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺 𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀯𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁨)

1350. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 **𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀤𑀲𑀸𑀬𑀢𑀦𑀸𑀦𑀺—  𑀘𑀓𑁆𑀔𑀸𑀬𑀢𑀦𑀁, 𑀭𑀽𑀧𑀸𑀬𑀢𑀦𑀁, 𑀲𑁄𑀢𑀸𑀬𑀢𑀦𑀁, 𑀲𑀤𑁆𑀤𑀸𑀬𑀢𑀦𑀁, 𑀖𑀸𑀦𑀸𑀬𑀢𑀦𑀁, 𑀕𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀁, 𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀢𑀦𑀁, 𑀭𑀲𑀸𑀬𑀢𑀦𑀁, 𑀓𑀸𑀬𑀸𑀬𑀢𑀦𑀁, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸𑀬𑀢𑀦𑀁𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁩)

1352. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 **𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸?** 𑀤𑀲 𑀅𑀓𑀼𑀲𑀮𑀓𑀫𑁆𑀫𑀧𑀣𑀸—  𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄, 𑀧𑀺𑀲𑀼𑀡𑀸 𑀯𑀸𑀘𑀸, 𑀨𑀭𑀼𑀲𑀸 𑀯𑀸𑀘𑀸, 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑁄, 𑀅𑀪𑀺𑀚𑁆𑀛𑀸, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀳𑀸𑀦𑀪𑀸𑀕𑀺𑀬𑀸𑁇 (𑁫)

1357. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀸𑀭𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀸𑀭𑀓𑁆𑀔𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀸𑀭𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭𑁇𑁩)

1358. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀭𑀸𑀧𑀲𑁆𑀲𑁂𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀭𑀸𑀧𑀲𑁆𑀲𑁂𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭𑁇𑁪)

1359. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀼𑀦𑁆𑀦𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀘𑁆𑀘𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀼𑀣𑀼𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀼𑀣𑀼𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀦𑀼𑀦𑁆𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀡𑀼𑀦𑁆𑀦𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀦𑁆𑀢𑀸𑀦𑀺 𑀫𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀦𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀼𑀦𑁆𑀦𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀘𑁆𑀘𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭𑁇𑁫)

1360. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀯𑀬𑀲𑀝𑁆𑀞𑁂𑀲𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑁂𑀲𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀪𑀯𑁂𑀲𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀲𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀯𑀬𑀲𑀝𑁆𑀞𑁂𑀲𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭𑁇𑁬)

1361. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀸𑀯𑀺𑀮𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀸𑀯𑀺𑀮𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭𑁇𑁭)

1362. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭𑁇𑁮)

1363. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭𑁇𑁯)

1364. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀭𑀸𑀕𑁄 𑀫𑁂 𑀧𑀳𑀻𑀦𑁄 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀤𑁄𑀲𑁄 𑀫𑁂 𑀧𑀳𑀻𑀦𑁄…𑀧𑁂…  𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀫𑁄𑀳𑁄 𑀫𑁂 𑀧𑀳𑀻𑀦𑁄…𑀧𑁂…  𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀸𑁇 (𑁭𑁇𑁧𑁦)

1365. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 **𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸?** 𑀤𑀲 𑀲𑀜𑁆𑀜𑀸—  𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸, 𑀫𑀭𑀡𑀲𑀜𑁆𑀜𑀸, 𑀆𑀳𑀸𑀭𑁂𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀸, 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂𑀅𑀦𑀪𑀺𑀭𑀢𑀺𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀸, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀸𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸𑁇 (𑁮)

1366. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 **𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸?** 𑀤𑀲 𑀦𑀺𑀚𑁆𑀚𑀭𑀯𑀢𑁆𑀣𑀽𑀦𑀺—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀢𑀺𑁇 𑀬𑁂 𑀘 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑁂𑀓𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸…  𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄…  𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁄…  𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄…  𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺…  𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀜𑀸𑀡𑀁 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑁆𑀲 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀢𑀺𑁇 𑀬𑁂 𑀘 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀅𑀦𑁂𑀓𑁂 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀘𑀲𑁆𑀲 𑀦𑀺𑀚𑁆𑀚𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑁇 (𑁯)

1367. 𑀓𑀢𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 **𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸?** 𑀤𑀲 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀥𑀫𑁆𑀫𑀸—  𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑀁 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀁, 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 (𑁧𑁦)

1368. 𑀇𑀢𑀺 𑀇𑀫𑁂 𑀲𑀢𑀥𑀫𑁆𑀫𑀸 𑀪𑀽𑀢𑀸 𑀢𑀘𑁆𑀙𑀸 𑀢𑀣𑀸 𑀅𑀯𑀺𑀢𑀣𑀸 𑀅𑀦𑀜𑁆𑀜𑀣𑀸 𑀲𑀫𑁆𑀫𑀸 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

1369. 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀏𑀓𑀸𑀤𑀲𑀫𑀁𑁇



---

1370. 𑀧𑀸𑀣𑀺𑀓𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

1371. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1372. _𑀧𑀸𑀣𑀺𑀓𑁄 𑀘 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀁,_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀜𑁆𑀜𑀓𑀁;_  
_𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀧𑀸𑀲𑀸𑀤𑀁,_  
_𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀁𑁇_  


1373. _𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀸𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀓𑀁,_  
_𑀲𑀗𑁆𑀕𑀻𑀢𑀺 𑀘 𑀤𑀲𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀏𑀓𑀸𑀤𑀲𑀳𑀺 𑀲𑀼𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀧𑀸𑀣𑀺𑀓𑀯𑀕𑁆𑀕𑁄𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


---

1374. 𑀧𑀸𑀣𑀺𑀓𑀯𑀕𑁆𑀕𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇



---

1375. 𑀢𑀻𑀳𑀺 𑀯𑀕𑁆𑀕𑁂𑀳𑀺 𑀧𑀝𑀺𑀫𑀡𑁆𑀟𑀺𑀢𑁄 𑀲𑀓𑀮𑁄



---

1376. 𑀤𑀻𑀖𑀦𑀺𑀓𑀸𑀬𑁄 𑀲𑀫𑀢𑁆𑀢𑁄𑁇
