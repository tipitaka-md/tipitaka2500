# 𑁨𑁇𑁬 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁𑀧𑀚𑁆𑀛𑀸𑀬𑀦

* 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀺𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁𑀧𑀚𑁆𑀛𑀸𑀬𑀦

(78.)

174\. 𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦 𑀢𑁂, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀫𑁆𑀬𑀢𑀸 𑀦𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀏𑀯𑀫𑀸𑀳𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀏𑀯 𑀯𑁄 𑀆𑀘𑀭𑀺𑀬𑁄, 𑀲𑁄 𑀏𑀯 𑀯𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀼𑁇

175\. 𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦 𑀢𑁂, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀉𑀤𑁆𑀤𑁂𑀲𑀸 𑀦𑁄 𑀘𑀸𑀯𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀏𑀯𑀫𑀸𑀳𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀏𑀯 𑀯𑁄 𑀉𑀤𑁆𑀤𑁂𑀲𑁄 𑀲𑁄 𑀏𑀯 𑀯𑁄 𑀉𑀤𑁆𑀤𑁂𑀲𑁄 𑀳𑁄𑀢𑀼𑁇

176\. 𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦 𑀢𑁂, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀆𑀚𑀻𑀯𑀸 𑀦𑁄 𑀘𑀸𑀯𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀏𑀯𑀫𑀸𑀳𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀏𑀯 𑀯𑁄 𑀆𑀚𑀻𑀯𑁄, 𑀲𑁄 𑀏𑀯 𑀯𑁄 𑀆𑀚𑀻𑀯𑁄 𑀳𑁄𑀢𑀼𑁇

177\. 𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦 𑀢𑁂, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀬𑁂 𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀅𑀓𑀼𑀲𑀮𑀲𑀗𑁆𑀔𑀸𑀢𑀸 𑀲𑀸𑀘𑀭𑀺𑀬𑀓𑀸𑀦𑀁, 𑀢𑁂𑀲𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀏𑀯𑀫𑀸𑀳𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀯 𑀯𑁄 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀼 𑀅𑀓𑀼𑀲𑀮𑀲𑀗𑁆𑀔𑀸𑀢𑀸 𑀘 𑀲𑀸𑀘𑀭𑀺𑀬𑀓𑀸𑀦𑀁𑁇

178\. 𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦 𑀢𑁂, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀫𑀲𑁆𑀲—  ‘𑀬𑁂 𑀦𑁄 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸 𑀓𑀼𑀲𑀮𑀲𑀗𑁆𑀔𑀸𑀢𑀸 𑀲𑀸𑀘𑀭𑀺𑀬𑀓𑀸𑀦𑀁, 𑀢𑁂𑀳𑀺 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄 𑀏𑀯𑀫𑀸𑀳𑀸’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀏𑀯𑀁 𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀁𑁇 𑀓𑀼𑀲𑀮𑀸 𑀘𑁂𑀯 𑀯𑁄 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀼 𑀓𑀼𑀲𑀮𑀲𑀗𑁆𑀔𑀸𑀢𑀸 𑀘 𑀲𑀸𑀘𑀭𑀺𑀬𑀓𑀸𑀦𑀁𑁇

179\. 𑀇𑀢𑀺 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀦𑁂𑀯 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀫𑁆𑀬𑀢𑀸 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺, 𑀦𑀧𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑀸 𑀘𑀸𑀯𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺, 𑀦𑀧𑀺 𑀆𑀚𑀻𑀯𑀸 𑀘𑀸𑀯𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺, 𑀦𑀧𑀺 𑀬𑁂 𑀯𑁄 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀅𑀓𑀼𑀲𑀮𑀲𑀗𑁆𑀔𑀸𑀢𑀸 𑀲𑀸𑀘𑀭𑀺𑀬𑀓𑀸𑀦𑀁, 𑀢𑁂𑀲𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺, 𑀦𑀧𑀺 𑀬𑁂 𑀯𑁄 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸 𑀓𑀼𑀲𑀮𑀲𑀗𑁆𑀔𑀸𑀢𑀸 𑀲𑀸𑀘𑀭𑀺𑀬𑀓𑀸𑀦𑀁, 𑀢𑁂𑀳𑀺 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺𑁇 𑀲𑀦𑁆𑀢𑀺 𑀘 𑀔𑁄, 𑀦𑀺𑀕𑁆𑀭𑁄𑀥, 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀧𑁄𑀦𑁄𑀩𑁆𑀪𑀯𑀺𑀓𑀸 𑀲𑀤𑀭𑀸 𑀤𑀼𑀓𑁆𑀔𑀯𑀺𑀧𑀸𑀓𑀸 𑀆𑀬𑀢𑀺𑀁 𑀚𑀸𑀢𑀺𑀚𑀭𑀸𑀫𑀭𑀡𑀺𑀬𑀸, 𑀬𑁂𑀲𑀸𑀳𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀺𑁇 𑀬𑀣𑀸𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑁄 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀳𑀻𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀯𑁄𑀤𑀸𑀦𑀻𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀢𑁆𑀢𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀣𑀸”𑀢𑀺𑁇

(79.)

180\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀢𑁂 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑀸 𑀧𑀢𑁆𑀢𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀅𑀥𑁄𑀫𑀼𑀔𑀸 𑀧𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀦𑀸 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀩𑁆𑀩𑁂𑀧𑀺𑀫𑁂 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀨𑀼𑀝𑁆𑀞𑀸 𑀧𑀸𑀧𑀺𑀫𑀢𑀸𑁇 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀏𑀓𑀲𑁆𑀲𑀧𑀺 𑀦 𑀏𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺—  ‘𑀳𑀦𑁆𑀤 𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀸𑀡𑀢𑁆𑀣𑀫𑁆𑀧𑀺 𑀲𑀫𑀡𑁂 𑀕𑁄𑀢𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫, 𑀓𑀺𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀳𑁄’”𑀢𑀺?

181\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀺𑀓𑀸𑀬 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀭𑀸𑀫𑁂 𑀲𑀻𑀳𑀦𑀸𑀤𑀁 𑀦𑀤𑀺𑀢𑁆𑀯𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀲𑀦𑁆𑀥𑀸𑀦𑁄 𑀧𑀦 𑀕𑀳𑀧𑀢𑀺 𑀢𑀸𑀯𑀤𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀸𑀯𑀺𑀲𑀻𑀢𑀺𑁇

---

182\. 𑀉𑀤𑀼𑀫𑁆𑀩𑀭𑀺𑀓𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
