# 𑁬𑁇𑁧𑁩 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀝𑁆𑀞𑀸𑀦

### 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀝𑁆𑀞𑀸𑀦

(189.)

380\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆’𑀢𑀺? 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸—  “𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”’𑀢𑀺𑁇

381\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀺𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆’𑀢𑀺? 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁—  “𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”’𑀢𑀺𑁇

382\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀺𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆’𑀢𑀺? 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀔𑁄 𑀏𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸—  “𑀳𑁄𑀢𑀺 𑀘 𑀦 𑀘 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”’𑀢𑀺𑁇

383\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀺𑀁 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆’𑀢𑀺? 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀏𑀢𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁—  “𑀦𑁂𑀯 𑀳𑁄𑀢𑀺 𑀦 𑀦 𑀳𑁄𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”’𑀢𑀺𑁇

384\. 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀯𑀺𑀚𑁆𑀚𑀢𑀺, 𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀓𑀲𑁆𑀫𑀸 𑀧𑀦𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑀡𑁂𑀦 𑀕𑁄𑀢𑀫𑁂𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀦𑁆’𑀢𑀺? 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄, 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀦 𑀳𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁 𑀦 𑀥𑀫𑁆𑀫𑀲𑀁𑀳𑀺𑀢𑀁 𑀦 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀓𑀁 𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀦 𑀯𑀺𑀭𑀸𑀕𑀸𑀬 𑀦 𑀦𑀺𑀭𑁄𑀥𑀸𑀬 𑀦 𑀉𑀧𑀲𑀫𑀸𑀬 𑀦 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀦 𑀲𑀫𑁆𑀩𑁄𑀥𑀸𑀬 𑀦 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀦𑁆’𑀢𑀺𑁇
