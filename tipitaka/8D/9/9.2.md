# 𑁯𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀡𑀯𑀸𑀭

### 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀡𑀯𑀸𑀭

(285.)

707\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀫𑀳𑀢𑀺𑀬𑀸 𑀘 𑀬𑀓𑁆𑀔𑀲𑁂𑀦𑀸𑀬 𑀫𑀳𑀢𑀺𑀬𑀸 𑀘 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀲𑁂𑀦𑀸𑀬 𑀫𑀳𑀢𑀺𑀬𑀸 𑀘 𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀲𑁂𑀦𑀸𑀬 𑀫𑀳𑀢𑀺𑀬𑀸 𑀘 𑀦𑀸𑀕𑀲𑁂𑀦𑀸𑀬 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀁 𑀭𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀁 𑀕𑀼𑀫𑁆𑀩𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀁 𑀑𑀯𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀁 𑀧𑀩𑁆𑀩𑀢𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀓𑁆𑀔𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀲𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇

(286.)

708\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀲𑀦𑁆𑀢𑀺 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀍𑀆𑀭𑀸 𑀬𑀓𑁆𑀔𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸…𑀧𑁂…  𑀲𑀦𑁆𑀢𑀺 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀻𑀘𑀸 𑀬𑀓𑁆𑀔𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀲𑀦𑁆𑀦𑀸𑁇 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀓𑁆𑀔𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀬𑁂𑀯 𑀪𑀕𑀯𑀢𑁄𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀪𑀕𑀯𑀸 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺…  𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺𑁇 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀓𑁆𑀔𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸𑀬𑁂𑀯 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸…  𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀫𑀦𑀸𑀧𑀁𑁇 𑀲𑀦𑁆𑀢𑀺 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀸𑀦𑀺 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀸𑀦𑀺 𑀯𑀺𑀚𑀦𑀯𑀸𑀢𑀸𑀦𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀳𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀓𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀸𑀦𑀺𑁇 𑀢𑀢𑁆𑀣 𑀲𑀦𑁆𑀢𑀺 𑀉𑀍𑀆𑀭𑀸 𑀬𑀓𑁆𑀔𑀸 𑀦𑀺𑀯𑀸𑀲𑀺𑀦𑁄, 𑀬𑁂 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀯𑀘𑀦𑁂 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸, 𑀢𑁂𑀲𑀁 𑀧𑀲𑀸𑀤𑀸𑀬 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀁 𑀭𑀓𑁆𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀕𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀭𑀓𑁆𑀔𑀸𑀬 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀬 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸’𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀫𑁂 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀁 𑀭𑀓𑁆𑀔𑀁 𑀅𑀪𑀸𑀲𑀺—

(287.)

709\. _‘𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲 𑀘 𑀦𑀫𑀢𑁆𑀣𑀼,_  
_𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀺𑀭𑀻𑀫𑀢𑁄;_  
_𑀲𑀺𑀔𑀺𑀲𑁆𑀲𑀧𑀺 𑀘 𑀦𑀫𑀢𑁆𑀣𑀼,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀦𑁄𑁇_  


710\. _𑀯𑁂𑀲𑁆𑀲𑀪𑀼𑀲𑁆𑀲 𑀘 𑀦𑀫𑀢𑁆𑀣𑀼,_  
_𑀦𑁆𑀳𑀸𑀢𑀓𑀲𑁆𑀲 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀦𑀫𑀢𑁆𑀣𑀼 𑀓𑀓𑀼𑀲𑀦𑁆𑀥𑀲𑁆𑀲,_  
_𑀫𑀸𑀭𑀲𑁂𑀦𑀸𑀧𑀫𑀤𑁆𑀤𑀺𑀦𑁄𑁇_  


711\. _𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑀲𑁆𑀲 𑀦𑀫𑀢𑁆𑀣𑀼,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀼𑀲𑀻𑀫𑀢𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀘 𑀦𑀫𑀢𑁆𑀣𑀼,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀩𑁆𑀩𑀥𑀺𑁇_  


712\. _𑀅𑀗𑁆𑀕𑀻𑀭𑀲𑀲𑁆𑀲 𑀦𑀫𑀢𑁆𑀣𑀼,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀺𑀭𑀻𑀫𑀢𑁄;_  
_𑀬𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀸𑀧𑀦𑀽𑀤𑀦𑀁𑁇_  


713\. _𑀬𑁂 𑀘𑀸𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸 𑀮𑁄𑀓𑁂,_  
_𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀲𑀼𑀁;_  
_𑀢𑁂 𑀚𑀦𑀸 𑀅𑀧𑀺𑀲𑀼𑀡𑀸𑀣,_  
_𑀫𑀳𑀦𑁆𑀢𑀸 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀸𑁇_  


714\. _𑀳𑀺𑀢𑀁 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀬𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀕𑁄𑀢𑀫𑀁;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁𑁇_  


(288.)

715\. _𑀬𑀢𑁄 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺 𑀲𑀽𑀭𑀺𑀬𑁄,_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑁄 𑀫𑀡𑁆𑀟𑀮𑀻 𑀫𑀳𑀸;_  
_𑀬𑀲𑁆𑀲 𑀘𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀲𑀁𑀯𑀭𑀻𑀧𑀺 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀬𑀲𑁆𑀲 𑀘𑀼𑀕𑁆𑀕𑀢𑁂 𑀲𑀽𑀭𑀺𑀬𑁂,_  
_“𑀤𑀺𑀯𑀲𑁄”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


716\. _𑀭𑀳𑀤𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀕𑀫𑁆𑀪𑀻𑀭𑁄,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀲𑀭𑀺𑀢𑁄𑀤𑀓𑁄;_  
_𑀏𑀯𑀁 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_“𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀲𑀭𑀺𑀢𑁄𑀤𑀓𑁄”𑁇_  


717\. _𑀇𑀢𑁄 “𑀲𑀸 𑀧𑀼𑀭𑀺𑀫𑀸 𑀤𑀺𑀲𑀸”,_  
_𑀇𑀢𑀺 𑀦𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀻 𑀚𑀦𑁄;_  
_𑀬𑀁 𑀤𑀺𑀲𑀁 𑀅𑀪𑀺𑀧𑀸𑀮𑁂𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺 𑀲𑁄𑁇_  


718\. _𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_“𑀥𑀢𑀭𑀝𑁆𑀞𑁄”𑀢𑀺 𑀦𑀸𑀫𑀲𑁄;_  
_𑀭𑀫𑀢𑀻 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑁂𑀳𑀺,_  
_𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁂𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


719\. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀏𑀓𑀦𑀸𑀫𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀅𑀲𑀻𑀢𑀺 𑀤𑀲 𑀏𑀓𑁄 𑀘,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


720\. _𑀢𑁂 𑀘𑀸𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀁;_  
_𑀤𑀽𑀭𑀢𑁄𑀯 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁𑁇_  


721\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀓𑀼𑀲𑀮𑁂𑀦 𑀲𑀫𑁂𑀓𑁆𑀔𑀲𑀺,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀧𑀺 𑀢𑀁 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺;_  
_𑀲𑀼𑀢𑀁 𑀦𑁂𑀢𑀁 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫𑀲𑁂𑁇_  


722\. _“𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀣 𑀕𑁄𑀢𑀫𑀁,_  
_𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀁;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀁”𑁇_  


(289.)

723\. _𑀬𑁂𑀦 𑀧𑁂𑀢𑀸 𑀧𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺,_  
_𑀧𑀺𑀲𑀼𑀡𑀸 𑀧𑀺𑀝𑁆𑀞𑀺𑀫𑀁𑀲𑀺𑀓𑀸;_  
_𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀺𑀦𑁄 𑀮𑀼𑀤𑁆𑀤𑀸,_  
_𑀘𑁄𑀭𑀸 𑀦𑁂𑀓𑀢𑀺𑀓𑀸 𑀚𑀦𑀸𑁇_  


724\. _𑀇𑀢𑁄 “𑀲𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀤𑀺𑀲𑀸”,_  
_𑀇𑀢𑀺 𑀦𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀻 𑀚𑀦𑁄;_  
_𑀬𑀁 𑀤𑀺𑀲𑀁 𑀅𑀪𑀺𑀧𑀸𑀮𑁂𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺 𑀲𑁄𑁇_  


725\. _𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑀸𑀦𑀁 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_“𑀯𑀺𑀭𑀽𑀍𑀳𑁄” 𑀇𑀢𑀺 𑀦𑀸𑀫𑀲𑁄;_  
_𑀭𑀫𑀢𑀻 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑁂𑀳𑀺,_  
_𑀓𑀼𑀫𑁆𑀪𑀡𑁆𑀟𑁂𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


726\. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀏𑀓𑀦𑀸𑀫𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀅𑀲𑀻𑀢𑀺 𑀤𑀲 𑀏𑀓𑁄 𑀘,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


727\. _𑀢𑁂 𑀘𑀸𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀁;_  
_𑀤𑀽𑀭𑀢𑁄𑀯 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁𑁇_  


728\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀓𑀼𑀲𑀮𑁂𑀦 𑀲𑀫𑁂𑀓𑁆𑀔𑀲𑀺,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀧𑀺 𑀢𑀁 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺;_  
_𑀲𑀼𑀢𑀁 𑀦𑁂𑀢𑀁 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫𑀲𑁂𑁇_  


729\. _“𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀣 𑀕𑁄𑀢𑀫𑀁,_  
_𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀁;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀁”𑁇_  


(290.)

730\. _𑀬𑀢𑁆𑀣 𑀘𑁄𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺 𑀲𑀽𑀭𑀺𑀬𑁄,_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑁄 𑀫𑀡𑁆𑀟𑀮𑀻 𑀫𑀳𑀸;_  
_𑀬𑀲𑁆𑀲 𑀘𑁄𑀕𑁆𑀕𑀘𑁆𑀙𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀤𑀺𑀯𑀲𑁄𑀧𑀺 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀬𑀲𑁆𑀲 𑀘𑁄𑀕𑁆𑀕𑀢𑁂 𑀲𑀽𑀭𑀺𑀬𑁂,_  
_“𑀲𑀁𑀯𑀭𑀻”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


731\. _𑀭𑀳𑀤𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀕𑀫𑁆𑀪𑀻𑀭𑁄,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀲𑀭𑀺𑀢𑁄𑀤𑀓𑁄;_  
_𑀏𑀯𑀁 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀲𑀭𑀺𑀢𑁄𑀤𑀓𑁄𑁇_  


732\. _𑀇𑀢𑁄 “𑀲𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀤𑀺𑀲𑀸”,_  
_𑀇𑀢𑀺 𑀦𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀻 𑀚𑀦𑁄;_  
_𑀬𑀁 𑀤𑀺𑀲𑀁 𑀅𑀪𑀺𑀧𑀸𑀮𑁂𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺 𑀲𑁄𑁇_  


733\. _𑀦𑀸𑀕𑀸𑀦𑀜𑁆𑀘 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_“𑀯𑀺𑀭𑀽𑀧𑀓𑁆𑀔𑁄”𑀢𑀺 𑀦𑀸𑀫𑀲𑁄;_  
_𑀭𑀫𑀢𑀻 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑁂𑀳𑀺,_  
_𑀦𑀸𑀕𑁂𑀳𑁂𑀯 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


734\. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀏𑀓𑀦𑀸𑀫𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀅𑀲𑀻𑀢𑀺 𑀤𑀲 𑀏𑀓𑁄 𑀘,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


735\. _𑀢𑁂 𑀘𑀸𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀁;_  
_𑀤𑀽𑀭𑀢𑁄𑀯 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁𑁇_  


736\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀓𑀼𑀲𑀮𑁂𑀦 𑀲𑀫𑁂𑀓𑁆𑀔𑀲𑀺,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀧𑀺 𑀢𑀁 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺;_  
_𑀲𑀼𑀢𑀁 𑀦𑁂𑀢𑀁 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫𑀲𑁂𑁇_  


737\. _“𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀣 𑀕𑁄𑀢𑀫𑀁,_  
_𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀁;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀁”𑁇_  


(291.)

738\. _𑀬𑁂𑀦 𑀉𑀢𑁆𑀢𑀭𑀓𑀼𑀭𑀼𑀯𑁆𑀳𑁄,_  
_𑀫𑀳𑀸𑀦𑁂𑀭𑀼 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑁄;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀢𑁆𑀣 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀅𑀫𑀫𑀸 𑀅𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀸𑁇_  


739\. _𑀦 𑀢𑁂 𑀩𑀻𑀚𑀁 𑀧𑀯𑀧𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀧𑀺 𑀦𑀻𑀬𑀦𑁆𑀢𑀺 𑀦𑀗𑁆𑀕𑀮𑀸;_  
_𑀅𑀓𑀝𑁆𑀞𑀧𑀸𑀓𑀺𑀫𑀁 𑀲𑀸𑀮𑀺𑀁,_  
_𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀼𑀲𑀸𑁇_  


740\. _𑀅𑀓𑀡𑀁 𑀅𑀣𑀼𑀲𑀁 𑀲𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀼𑀕𑀦𑁆𑀥𑀁 𑀢𑀡𑁆𑀟𑀼𑀮𑀧𑁆𑀨𑀮𑀁;_  
_𑀢𑀼𑀡𑁆𑀟𑀺𑀓𑀻𑀭𑁂 𑀧𑀘𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀢𑁄 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺 𑀪𑁄𑀚𑀦𑀁𑁇_  


741\. _𑀕𑀸𑀯𑀺𑀁 𑀏𑀓𑀔𑀼𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁;_  
_𑀧𑀲𑀼𑀁 𑀏𑀓𑀔𑀼𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁𑁇_  


742\. _𑀇𑀢𑁆𑀣𑀺𑀁 𑀯𑀸 𑀯𑀸𑀳𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁;_  
_𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑀸𑀳𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁𑁇_  


743\. _𑀓𑀼𑀫𑀸𑀭𑀺𑀁 𑀯𑀸𑀳𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁;_  
_𑀓𑀼𑀫𑀸𑀭𑀁 𑀯𑀸𑀳𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀺 𑀤𑀺𑀲𑁄𑀤𑀺𑀲𑀁𑁇_  


744\. _𑀢𑁂 𑀬𑀸𑀦𑁂 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑀸 𑀤𑀺𑀲𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑀸𑀬𑀦𑁆𑀢𑀺;_  
_𑀧𑀘𑀸𑀭𑀸 𑀢𑀲𑁆𑀲 𑀭𑀸𑀚𑀺𑀦𑁄𑁇_  


745\. _𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀬𑀸𑀦𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀁 𑀬𑀸𑀦𑀁 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀧𑀸𑀲𑀸𑀤𑀸 𑀲𑀺𑀯𑀺𑀓𑀸 𑀘𑁂𑀯,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


746\. _𑀢𑀲𑁆𑀲 𑀘 𑀦𑀕𑀭𑀸 𑀅𑀳𑀼,_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀲𑀼𑀫𑀸𑀧𑀺𑀢𑀸;_  
_𑀆𑀝𑀸𑀦𑀸𑀝𑀸 𑀓𑀼𑀲𑀺𑀦𑀸𑀝𑀸 𑀧𑀭𑀓𑀼𑀲𑀺𑀦𑀸𑀝𑀸,_  
_𑀦𑀸𑀝𑀲𑀼𑀭𑀺𑀬𑀸 𑀧𑀭𑀓𑀼𑀲𑀺𑀝𑀦𑀸𑀝𑀸𑁇_  


747\. _𑀉𑀢𑁆𑀢𑀭𑁂𑀦 𑀓𑀲𑀺𑀯𑀦𑁆𑀢𑁄,_  
_𑀚𑀦𑁄𑀖𑀫𑀧𑀭𑁂𑀦 𑀘;_  
_𑀦𑀯𑀦𑀯𑀼𑀢𑀺𑀬𑁄 𑀅𑀫𑁆𑀩𑀭𑀅𑀫𑁆𑀩𑀭𑀯𑀢𑀺𑀬𑁄,_  
_𑀆𑀍𑀅𑀓𑀫𑀦𑁆𑀤𑀸 𑀦𑀸𑀫 𑀭𑀸𑀚𑀥𑀸𑀦𑀻𑁇_  


748\. _𑀓𑀼𑀯𑁂𑀭𑀲𑁆𑀲 𑀔𑁄 𑀧𑀦 𑀫𑀸𑀭𑀺𑀲,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀲𑁆𑀲 𑀯𑀺𑀲𑀸𑀡𑀸 𑀦𑀸𑀫 𑀭𑀸𑀚𑀥𑀸𑀦𑀻;_  
_𑀢𑀲𑁆𑀫𑀸 𑀓𑀼𑀯𑁂𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_“𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


749\. _𑀧𑀘𑁆𑀘𑁂𑀲𑀦𑁆𑀢𑁄 𑀧𑀓𑀸𑀲𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑀢𑁄𑀮𑀸 𑀢𑀢𑁆𑀢𑀮𑀸 𑀢𑀢𑁄𑀢𑀮𑀸;_  
_𑀑𑀚𑀲𑀺 𑀢𑁂𑀚𑀲𑀺 𑀢𑀢𑁄𑀚𑀲𑀻,_  
_𑀲𑀽𑀭𑁄 𑀭𑀸𑀚𑀸 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀦𑁂𑀫𑀺𑁇_  


750\. _𑀭𑀳𑀤𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀥𑀭𑀡𑀻 𑀦𑀸𑀫,_  
_𑀬𑀢𑁄 𑀫𑁂𑀖𑀸 𑀧𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑀺;_  
_𑀯𑀲𑁆𑀲𑀸 𑀬𑀢𑁄 𑀧𑀢𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑀪𑀸𑀧𑀺 𑀢𑀢𑁆𑀣 𑀲𑀸𑀮𑀯𑀢𑀻 𑀦𑀸𑀫𑁇_  


751\. _𑀬𑀢𑁆𑀣 𑀬𑀓𑁆𑀔𑀸 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀢𑁆𑀣 𑀦𑀺𑀘𑁆𑀘𑀨𑀮𑀸 𑀭𑀼𑀓𑁆𑀔𑀸;_  
_𑀦𑀸𑀦𑀸 𑀤𑀺𑀚𑀕𑀡𑀸 𑀬𑀼𑀢𑀸,_  
_𑀫𑀬𑀽𑀭𑀓𑁄𑀜𑁆𑀘𑀸𑀪𑀺𑀭𑀼𑀤𑀸;_  
_𑀓𑁄𑀓𑀺𑀮𑀸𑀤𑀻𑀳𑀺 𑀯𑀕𑁆𑀕𑀼𑀳𑀺𑁇_  


752\. _𑀚𑀻𑀯𑀜𑁆𑀚𑀻𑀯𑀓𑀲𑀤𑁆𑀤𑁂𑀢𑁆𑀣,_  
_𑀅𑀣𑁄 𑀑𑀝𑁆𑀞𑀯𑀘𑀺𑀢𑁆𑀢𑀓𑀸;_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀓𑀸 𑀓𑀼𑀍𑀈𑀭𑀓𑀸,_  
_𑀯𑀦𑁂 𑀧𑁄𑀓𑁆𑀔𑀭𑀲𑀸𑀢𑀓𑀸𑁇_  


753\. _𑀲𑀼𑀓𑀲𑀸𑀍𑀇𑀓 𑀲𑀤𑁆𑀤𑁂𑀢𑁆𑀣,_  
_𑀤𑀡𑁆𑀟𑀫𑀸𑀡𑀯𑀓𑀸𑀦𑀺 𑀘;_  
_𑀲𑁄𑀪𑀢𑀺 𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀁 𑀲𑀸,_  
_𑀓𑀼𑀯𑁂𑀭𑀦𑀍𑀇𑀦𑀻 𑀲𑀤𑀸𑁇_  


754\. _𑀇𑀢𑁄 “𑀲𑀸 𑀉𑀢𑁆𑀢𑀭𑀸 𑀤𑀺𑀲𑀸”,_  
_𑀇𑀢𑀺 𑀦𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀢𑀻 𑀚𑀦𑁄;_  
_𑀬𑀁 𑀤𑀺𑀲𑀁 𑀅𑀪𑀺𑀧𑀸𑀮𑁂𑀢𑀺,_  
_𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑀲𑀲𑁆𑀲𑀺 𑀲𑁄𑁇_  


755\. _𑀬𑀓𑁆𑀔𑀸𑀦𑀜𑁆𑀘 𑀅𑀥𑀺𑀧𑀢𑀺,_  
_“𑀓𑀼𑀯𑁂𑀭𑁄” 𑀇𑀢𑀺 𑀦𑀸𑀫𑀲𑁄;_  
_𑀭𑀫𑀢𑀻 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑁂𑀳𑀺,_  
_𑀬𑀓𑁆𑀔𑁂𑀳𑁂𑀯 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


756\. _𑀧𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀢𑀲𑁆𑀲 𑀩𑀳𑀯𑁄,_  
_𑀏𑀓𑀦𑀸𑀫𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀅𑀲𑀻𑀢𑀺 𑀤𑀲 𑀏𑀓𑁄 𑀘,_  
_𑀇𑀦𑁆𑀤𑀦𑀸𑀫𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


757\. _𑀢𑁂 𑀘𑀸𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀁;_  
_𑀤𑀽𑀭𑀢𑁄𑀯 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀳𑀦𑁆𑀢𑀁 𑀯𑀻𑀢𑀲𑀸𑀭𑀤𑀁𑁇_  


758\. _𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫;_  
_𑀓𑀼𑀲𑀮𑁂𑀦 𑀲𑀫𑁂𑀓𑁆𑀔𑀲𑀺,_  
_𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀧𑀺 𑀢𑀁 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀺;_  
_𑀲𑀼𑀢𑀁 𑀦𑁂𑀢𑀁 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀫𑀲𑁂𑁇_  


759\. _“𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀣 𑀕𑁄𑀢𑀫𑀁,_  
_𑀚𑀺𑀦𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀁;_  
_𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀁 𑀯𑀦𑁆𑀤𑀸𑀫 𑀕𑁄𑀢𑀫𑀦𑁆”𑀢𑀺𑁇_  


(292.)

_𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄 𑀧𑀚𑀸𑀧𑀢𑀺;_  
_𑀘𑀦𑁆𑀤𑀦𑁄 𑀓𑀸𑀫𑀲𑁂𑀝𑁆𑀞𑁄 𑀘,_  
_𑀓𑀺𑀦𑁆𑀦𑀼𑀖𑀡𑁆𑀟𑀼 𑀦𑀺𑀖𑀡𑁆𑀟𑀼 𑀘𑁇_  


764\. _𑀧𑀦𑀸𑀤𑁄 𑀑𑀧𑀫𑀜𑁆𑀜𑁄 𑀘,_  
_𑀤𑁂𑀯𑀲𑀽𑀢𑁄 𑀘 𑀫𑀸𑀢𑀮𑀺;_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑁂𑀦𑁄 𑀘 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀦𑀍𑀑 𑀭𑀸𑀚𑀸 𑀚𑀦𑁂𑀲𑀪𑁄𑁇_  


765\. _𑀲𑀸𑀢𑀸𑀕𑀺𑀭𑁄 𑀳𑁂𑀫𑀯𑀢𑁄,_  
_𑀧𑀼𑀡𑁆𑀡𑀓𑁄 𑀓𑀭𑀢𑀺𑀬𑁄 𑀕𑀼𑀍𑀑;_  
_𑀲𑀺𑀯𑀓𑁄 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑁄 𑀘,_  
_𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑀢𑁆𑀢𑁄 𑀬𑀼𑀕𑀦𑁆𑀥𑀭𑁄𑁇_  


766\. _𑀕𑁄𑀧𑀸𑀮𑁄 𑀲𑀼𑀧𑁆𑀧𑀭𑁄𑀥𑁄 𑀘,_  
_𑀳𑀺𑀭𑀺 𑀦𑁂𑀢𑁆𑀢𑀺 𑀘 𑀫𑀦𑁆𑀤𑀺𑀬𑁄;_  
_𑀧𑀜𑁆𑀘𑀸𑀮𑀘𑀡𑁆𑀟𑁄 𑀆𑀍𑀅𑀯𑀓𑁄,_  
_𑀧𑀚𑁆𑀚𑀼𑀦𑁆𑀦𑁄 𑀲𑀼𑀫𑀦𑁄 𑀲𑀼𑀫𑀼𑀔𑁄;_  
_𑀤𑀥𑀺𑀫𑀼𑀔𑁄 𑀫𑀡𑀺 𑀫𑀸𑀡𑀺𑀯𑀭𑁄 𑀤𑀻𑀖𑁄,_  
_𑀅𑀣𑁄 𑀲𑁂𑀭𑀻𑀲𑀓𑁄 𑀲𑀳𑁇_  


767\. 𑀇𑀫𑁂𑀲𑀁 𑀬𑀓𑁆𑀔𑀸𑀦𑀁 𑀫𑀳𑀸𑀬𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑁂𑀦𑀸𑀧𑀢𑀻𑀦𑀁 𑀫𑀳𑀸𑀲𑁂𑀦𑀸𑀧𑀢𑀻𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁 𑀯𑀺𑀓𑁆𑀓𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀯𑀺𑀭𑀯𑀺𑀢𑀩𑁆𑀩𑀁—  “𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀆𑀯𑀺𑀲𑀢𑀺, 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀳𑁂𑀞𑁂𑀢𑀺, 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀯𑀺𑀳𑁂𑀞𑁂𑀢𑀺, 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀳𑀺𑀁𑀲𑀢𑀺, 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀯𑀺𑀳𑀺𑀁𑀲𑀢𑀺, 𑀅𑀬𑀁 𑀬𑀓𑁆𑀔𑁄 𑀦 𑀫𑀼𑀜𑁆𑀘𑀢𑀻”𑀢𑀺𑁇

768\. 𑀅𑀬𑀁 𑀔𑁄, 𑀫𑀸𑀭𑀺𑀲, 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀸 𑀭𑀓𑁆𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀕𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀭𑀓𑁆𑀔𑀸𑀬 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀬 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀳𑀦𑁆𑀤 𑀘 𑀤𑀸𑀦𑀺 𑀫𑀬𑀁, 𑀫𑀸𑀭𑀺𑀲, 𑀕𑀘𑁆𑀙𑀸𑀫, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑀸 𑀫𑀬𑀁 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑀸’𑀢𑀺𑁇 ‘𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄, 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀣𑀸’𑀢𑀺𑁇

(294.)

769\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀧𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀓𑁆𑀔𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀲𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇

(295.)

770\. 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀁 𑀭𑀓𑁆𑀔𑀁𑁇 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀸𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀁 𑀭𑀓𑁆𑀔𑀁𑁇 𑀥𑀸𑀭𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀁 𑀭𑀓𑁆𑀔𑀁𑁇 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀸 𑀭𑀓𑁆𑀔𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀕𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀭𑀓𑁆𑀔𑀸𑀬 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀬 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇

771\. 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇

---

772\. 𑀆𑀝𑀸𑀦𑀸𑀝𑀺𑀬𑀲𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀦𑀯𑀫𑀁𑁇
