# 𑁧𑁦𑁇𑁨 𑀤𑀼𑀓

* 𑀲𑀗𑁆𑀕𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀓

(304.)

783\. 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀯𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁, 𑀬𑀣𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀥𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁, 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂?

784\. 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘𑁇 (𑁧)

785\. 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀪𑀯𑀢𑀡𑁆𑀳𑀸 𑀘𑁇 (𑁨)

786\. 𑀪𑀯𑀤𑀺𑀝𑁆𑀞𑀺 𑀘 𑀯𑀺𑀪𑀯𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑁇 (𑁩)

787\. 𑀅𑀳𑀺𑀭𑀺𑀓𑀜𑁆𑀘 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘𑁇 (𑁪)

788\. 𑀳𑀺𑀭𑀻 𑀘 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀜𑁆𑀘𑁇 (𑁫)

789\. 𑀤𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀘 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀘𑁇 (𑁬)

790\. 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀢𑀸 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀘𑁇 (𑁭)

791\. 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸 𑀘𑁇 (𑁮)

792\. 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑀢𑀸 𑀘 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸 𑀘𑁇 (𑁯)

793\. 𑀥𑀸𑀢𑀼𑀓𑀼𑀲𑀮𑀢𑀸 𑀘 𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀓𑀼𑀲𑀮𑀢𑀸 𑀘𑁇 (𑁧𑁦)

794\. 𑀆𑀬𑀢𑀦𑀓𑀼𑀲𑀮𑀢𑀸 𑀘 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀼𑀲𑀮𑀢𑀸 𑀘𑁇 (𑁧𑁧)

795\. 𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸 𑀘 𑀅𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀢𑀸 𑀘𑁇 (𑁧𑁨)

796\. 𑀅𑀚𑁆𑀚𑀯𑀜𑁆𑀘 𑀮𑀚𑁆𑀚𑀯𑀜𑁆𑀘𑁇 (𑁧𑁩)

797\. 𑀔𑀦𑁆𑀢𑀺 𑀘 𑀲𑁄𑀭𑀘𑁆𑀘𑀜𑁆𑀘𑁇 (𑁧𑁪)

798\. 𑀲𑀸𑀔𑀮𑁆𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀸𑀭𑁄 𑀘𑁇 (𑁧𑁫)

799\. 𑀅𑀯𑀺𑀳𑀺𑀁𑀲𑀸 𑀘 𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀜𑁆𑀘𑁇 (𑁧𑁬)

800\. 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀅𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘𑁇 (𑁧𑁭)

801\. 𑀲𑀢𑀺 𑀘 𑀲𑀫𑁆𑀧𑀚𑀜𑁆𑀜𑀜𑁆𑀘𑁇 (𑁧𑁮)

802\. 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀅𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸 𑀘 𑀪𑁄𑀚𑀦𑁂 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀘𑁇 (𑁧𑁯)

803\. 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑀤𑁆𑀯𑀸𑀭𑀢𑀸 𑀘 𑀪𑁄𑀚𑀦𑁂 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸 𑀘𑁇 (𑁨𑁦)

804\. 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀩𑀮𑀜𑁆𑀘 𑀪𑀸𑀯𑀦𑀸𑀩𑀮𑀜𑁆𑀘𑁇 (𑁨𑁧)

805\. 𑀲𑀢𑀺𑀩𑀮𑀜𑁆𑀘 𑀲𑀫𑀸𑀥𑀺𑀩𑀮𑀜𑁆𑀘𑁇 (𑁨𑁨)

806\. 𑀲𑀫𑀣𑁄 𑀘 𑀯𑀺𑀧𑀲𑁆𑀲𑀦𑀸 𑀘𑁇 (𑁨𑁩)

807\. 𑀲𑀫𑀣𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀕𑁆𑀕𑀳𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘𑁇 (𑁨𑁪)

808\. 𑀧𑀕𑁆𑀕𑀳𑁄 𑀘 𑀅𑀯𑀺𑀓𑁆𑀔𑁂𑀧𑁄 𑀘𑁇 (𑁨𑁫)

809\. 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀘𑁇 (𑁨𑁬)

810\. 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸 𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀤𑀸 𑀘𑁇 (𑁨𑁭)

811\. 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺 𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺 𑀘𑁇 (𑁨𑁮)

812\. 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺 𑀔𑁄 𑀧𑀦 𑀬𑀣𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀘 𑀧𑀥𑀸𑀦𑀁𑁇 (𑁨𑁯)

813\. 𑀲𑀁𑀯𑁂𑀕𑁄 𑀘 𑀲𑀁𑀯𑁂𑀚𑀦𑀻𑀬𑁂𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀲𑁆𑀲 𑀘 𑀬𑁄𑀦𑀺𑀲𑁄 𑀧𑀥𑀸𑀦𑀁𑁇 (𑁩𑁦)

814\. 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀘 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀺𑀢𑀸 𑀘 𑀧𑀥𑀸𑀦𑀲𑁆𑀫𑀺𑀁𑁇 (𑁩𑁧)

815\. 𑀯𑀺𑀚𑁆𑀚𑀸 𑀘 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀘𑁇 (𑁩𑁨)

816\. 𑀔𑀬𑁂𑀜𑀸𑀡𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑁂𑀜𑀸𑀡𑀁𑁇 (𑁩𑁩)

817\. 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀯𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁, 𑀬𑀣𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀥𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁, 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇
