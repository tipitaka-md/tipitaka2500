# 𑁧𑁦𑁇𑁧𑁦 𑀤𑀲𑀓

### 𑀲𑀗𑁆𑀕𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀲𑀓

(345.)

1102. 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂 𑀤𑀲?

1103. 𑀤𑀲 **𑀦𑀸𑀣𑀓𑀭𑀡𑀸 𑀥𑀫𑁆𑀫𑀸𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺𑁇 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄, 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁧)

1104. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀢𑀥𑀭𑁄 𑀲𑀼𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄𑁇 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀲𑀸𑀢𑁆𑀣𑀸 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀥𑀸𑀢𑀸 𑀯𑀘𑀲𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁨)

1105. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁩)

1106. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺 𑀲𑁄𑀯𑀘𑀲𑁆𑀲𑀓𑀭𑀡𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀔𑀫𑁄 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀘𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁪)

1107. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀓𑀺𑀁𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺, 𑀢𑀢𑁆𑀣 𑀤𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑁄 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁…𑀧𑁂…  𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁫)

1108. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀺𑀬𑀲𑀫𑀼𑀤𑀸𑀳𑀸𑀭𑁄, 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁂 𑀉𑀍𑀆𑀭𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁄𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀉𑀍𑀆𑀭𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑁄𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁬)

1109. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀳𑀺 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀳𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁭)

1110. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁮)

1111. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀺𑀫𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀫𑁂𑀦 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀘𑀺𑀭𑀓𑀢𑀫𑁆𑀧𑀺 𑀘𑀺𑀭𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑀺 𑀲𑀭𑀺𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑁇 𑀬𑀁𑀧𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀺𑀫𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀭𑀺𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀦𑀸𑀣𑀓𑀭𑀡𑁄𑁇 (𑁧𑁇𑁯)

1113. 𑀤𑀲 **𑀓𑀲𑀺𑀡𑀸𑀬𑀢𑀦𑀸𑀦𑀺𑁇** 𑀧𑀣𑀯𑀻𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀯𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 𑀆𑀧𑁄𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂…  𑀢𑁂𑀚𑁄𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀯𑀸𑀬𑁄𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀦𑀻𑀮𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀧𑀻𑀢𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀮𑁄𑀳𑀺𑀢𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀑𑀤𑀸𑀢𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀆𑀓𑀸𑀲𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺…  𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑀲𑀺𑀡𑀫𑁂𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀢𑀺𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀯𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁𑁇 (𑁨)

1118. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀍𑀅𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑁂𑀯 𑀲𑀼𑀫𑀦𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀤𑀼𑀫𑁆𑀫𑀦𑁄, 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑁇 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑀁 𑀲𑀼𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀫𑀦𑀲𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀬 𑀦𑁂𑀯 𑀲𑀼𑀫𑀦𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀤𑀼𑀫𑁆𑀫𑀦𑁄, 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀙𑀍𑀅𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁨)

1119. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀸𑀭𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀸𑀭𑀓𑁆𑀔𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀸𑀭𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁩)

1120. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀭𑀸𑀧𑀲𑁆𑀲𑁂𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑀺, 𑀲𑀗𑁆𑀔𑀸𑀬𑁂𑀓𑀁 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀢𑀼𑀭𑀸𑀧𑀲𑁆𑀲𑁂𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁪)

1121. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀼𑀦𑁆𑀦𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀘𑁆𑀘𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀼𑀣𑀼𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀼𑀣𑀼𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀘𑁆𑀘𑀸𑀦𑀺, 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀦𑀼𑀦𑁆𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀡𑀼𑀦𑁆𑀦𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀦𑁆𑀢𑀸𑀦𑀺 𑀫𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀳𑀻𑀦𑀸𑀦𑀺 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀦𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀼𑀦𑁆𑀦𑀧𑀘𑁆𑀘𑁂𑀓𑀲𑀘𑁆𑀘𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁫)

1122. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀯𑀬𑀲𑀝𑁆𑀞𑁂𑀲𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑁂𑀲𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀪𑀯𑁂𑀲𑀦𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀲𑀦𑀸 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀸𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀯𑀬𑀲𑀝𑁆𑀞𑁂𑀲𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁬)

1123. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀸𑀯𑀺𑀮𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀸𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀳𑀺𑀁𑀲𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀧𑀳𑀻𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀸𑀯𑀺𑀮𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁭)

1124. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀧𑀳𑀸𑀦𑀸 𑀧𑀼𑀩𑁆𑀩𑁂𑀯 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀢𑀺𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁮)

1125. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀸𑀕𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀤𑁄𑀲𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀫𑁄𑀳𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁯)

1126. 𑀓𑀣𑀜𑁆𑀘𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀭𑀸𑀕𑁄 𑀫𑁂 𑀧𑀳𑀻𑀦𑁄 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀤𑁄𑀲𑁄 𑀫𑁂 𑀧𑀳𑀻𑀦𑁄 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 ‘𑀫𑁄𑀳𑁄 𑀫𑁂 𑀧𑀳𑀻𑀦𑁄 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀫𑀽𑀮𑁄 𑀢𑀸𑀮𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑁄 𑀅𑀦𑀪𑀸𑀯𑀁𑀓𑀢𑁄 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀸𑀤𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀧𑀜𑁆𑀜𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫𑁇𑁧𑁦)

1127. 𑀤𑀲 **𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀥𑀫𑁆𑀫𑀸—** 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑀁 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀁, 𑀅𑀲𑁂𑀓𑁆𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑁇 (𑁬)

1128. 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁 𑀦 𑀯𑀺𑀯𑀤𑀺𑀢𑀩𑁆𑀩𑀁, 𑀬𑀣𑀬𑀺𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀤𑁆𑀥𑀦𑀺𑀬𑀁 𑀅𑀲𑁆𑀲 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁, 𑀢𑀤𑀲𑁆𑀲 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

(349.)

1129. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀗𑁆𑀕𑀻𑀢𑀺𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀅𑀪𑀸𑀲𑀻”𑀢𑀺𑁇

1130. 𑀇𑀤𑀫𑀯𑁄𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄, 𑀲𑀫𑀦𑀼𑀜𑁆𑀜𑁄 𑀲𑀢𑁆𑀣𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀼𑀦𑁆𑀢𑀺𑁇
