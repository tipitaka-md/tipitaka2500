# 𑁧𑁦𑁇𑁫 𑀧𑀜𑁆𑀘𑀓

* 𑀲𑀗𑁆𑀕𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀜𑁆𑀘𑀓

(315.)

943\. 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸𑁇 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘?

944\. **𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇** 𑀭𑀽𑀧𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑁂𑀤𑀦𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 (𑁧)

945\. **𑀧𑀜𑁆𑀘𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇** 𑀭𑀽𑀧𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑁂𑀤𑀦𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀜𑁆𑀜𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀗𑁆𑀔𑀸𑀭𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀼𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 (𑁨)

946\. 𑀧𑀜𑁆𑀘 **𑀓𑀸𑀫𑀕𑀼𑀡𑀸𑁇** 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀽𑀧𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸, 𑀲𑁄𑀢𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀲𑀤𑁆𑀤𑀸…  𑀖𑀸𑀦𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀕𑀦𑁆𑀥𑀸…  𑀚𑀺𑀯𑁆𑀳𑀸𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀭𑀲𑀸…  𑀓𑀸𑀬𑀯𑀺𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀇𑀝𑁆𑀞𑀸 𑀓𑀦𑁆𑀢𑀸 𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀭𑀽𑀧𑀸 𑀓𑀸𑀫𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀸 𑀭𑀚𑀦𑀻𑀬𑀸𑁇 (𑁩)

947\. 𑀧𑀜𑁆𑀘 **𑀕𑀢𑀺𑀬𑁄—** 𑀦𑀺𑀭𑀬𑁄, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺, 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸, 𑀤𑁂𑀯𑀸𑁇 (𑁪)

948\. 𑀧𑀜𑁆𑀘 **𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀸𑀦𑀺—** 𑀆𑀯𑀸𑀲𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀓𑀼𑀮𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀮𑀸𑀪𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀯𑀡𑁆𑀡𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀥𑀫𑁆𑀫𑀫𑀘𑁆𑀙𑀭𑀺𑀬𑀁𑁇 (𑁫)

949\. 𑀧𑀜𑁆𑀘 **𑀦𑀻𑀯𑀭𑀡𑀸𑀦𑀺—** 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀦𑀻𑀯𑀭𑀡𑀁, 𑀣𑀺𑀦𑀫𑀺𑀤𑁆𑀥𑀦𑀻𑀯𑀭𑀡𑀁, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀦𑀻𑀯𑀭𑀡𑀁, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀻𑀯𑀭𑀡𑀁𑁇 (𑁬)

950\. 𑀧𑀜𑁆𑀘 **𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺—** 𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺, 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸, 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑁄, 𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑁄, 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑁄𑁇 (𑁭)

951\. 𑀧𑀜𑁆𑀘 **𑀉𑀤𑁆𑀥𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺—** 𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀅𑀭𑀽𑀧𑀭𑀸𑀕𑁄, 𑀫𑀸𑀦𑁄, 𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀁, 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑁇 (𑁮)

952\. 𑀧𑀜𑁆𑀘 **𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺—** 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀻𑁇 (𑁯)

(316.)

953\. 𑀧𑀜𑁆𑀘 **𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀺𑁇** 𑀅𑀪𑀩𑁆𑀩𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀧𑀸𑀡𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀼𑀁𑁇 𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑀼𑀁𑁇 𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑀼𑀁𑁇 𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀼𑀁𑁇 𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑀻𑀡𑀸𑀲𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀓𑀸𑀭𑀓𑀁 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀼𑀩𑁆𑀩𑁂 𑀆𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑁄𑁇 (𑁧𑁦)

954\. 𑀧𑀜𑁆𑀘 **𑀩𑁆𑀬𑀲𑀦𑀸𑀦𑀺—** 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑀁, 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀁, 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀁, 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑀁, 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑀁𑁇 𑀦𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑀳𑁂𑀢𑀼 𑀯𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁧𑁧)

955\. 𑀧𑀜𑁆𑀘 **𑀲𑀫𑁆𑀧𑀤𑀸—** 𑀜𑀸𑀢𑀺𑀲𑀫𑁆𑀧𑀤𑀸, 𑀪𑁄𑀕𑀲𑀫𑁆𑀧𑀤𑀸, 𑀆𑀭𑁄𑀕𑁆𑀬𑀲𑀫𑁆𑀧𑀤𑀸, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑁇 𑀦𑀸𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀜𑀸𑀢𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀳𑁂𑀢𑀼 𑀯𑀸 𑀪𑁄𑀕𑀲𑀫𑁆𑀧𑀤𑀸𑀳𑁂𑀢𑀼 𑀯𑀸 𑀆𑀭𑁄𑀕𑁆𑀬𑀲𑀫𑁆𑀧𑀤𑀸𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀳𑁂𑀢𑀼 𑀯𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀢𑁆𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑀳𑁂𑀢𑀼 𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 (𑁧𑁨)

956\. 𑀧𑀜𑁆𑀘 **𑀆𑀤𑀻𑀦𑀯𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀧𑀫𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀫𑀳𑀢𑀺𑀁 𑀪𑁄𑀕𑀚𑀸𑀦𑀺𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀬𑀜𑁆𑀜𑀤𑁂𑀯 𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀬𑀤𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀁, 𑀅𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑁄, 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀆𑀤𑀻𑀦𑀯𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 (𑁧𑁩)

957\. 𑀧𑀜𑁆𑀘 **𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀫𑀳𑀦𑁆𑀢𑀁 𑀪𑁄𑀕𑀓𑁆𑀔𑀦𑁆𑀥𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀬𑀜𑁆𑀜𑀤𑁂𑀯 𑀧𑀭𑀺𑀲𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀬𑀤𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀁 𑀬𑀤𑀺 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀁, 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀢𑀺 𑀅𑀫𑀗𑁆𑀓𑀼𑀪𑀽𑀢𑁄, 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀸 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀆𑀦𑀺𑀲𑀁𑀲𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬𑁇 (𑁧𑁪)

959\. 𑀧𑀜𑁆𑀘 **𑀧𑀥𑀸𑀦𑀺𑀬𑀗𑁆𑀕𑀸𑀦𑀺𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀁—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄, 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑁄, 𑀲𑀫𑀯𑁂𑀧𑀸𑀓𑀺𑀦𑀺𑀬𑀸 𑀕𑀳𑀡𑀺𑀬𑀸 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀦𑀸𑀢𑀺𑀲𑀻𑀢𑀸𑀬 𑀦𑀸𑀘𑁆𑀘𑀼𑀡𑁆𑀳𑀸𑀬 𑀫𑀚𑁆𑀛𑀺𑀫𑀸𑀬 𑀧𑀥𑀸𑀦𑀓𑁆𑀔𑀫𑀸𑀬𑁇 𑀅𑀲𑀞𑁄 𑀳𑁄𑀢𑀺 𑀅𑀫𑀸𑀬𑀸𑀯𑀻, 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀆𑀯𑀺𑀓𑀢𑁆𑀢𑀸 𑀲𑀢𑁆𑀣𑀭𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀲𑀼 𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼𑁇 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀤𑀬𑀢𑁆𑀣𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀸𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸𑁇 (𑁧𑁬)

(318.)

960\. 𑀧𑀜𑁆𑀘 **𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀸—** 𑀅𑀯𑀺𑀳𑀸, 𑀅𑀢𑀧𑁆𑀧𑀸, 𑀲𑀼𑀤𑀲𑁆𑀲𑀸, 𑀲𑀼𑀤𑀲𑁆𑀲𑀻, 𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑀸𑁇 (𑁧𑁭)

961\. 𑀧𑀜𑁆𑀘 **𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑁄—** 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻, 𑀉𑀧𑀳𑀘𑁆𑀘𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻, 𑀅𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻, 𑀲𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀻, 𑀉𑀤𑁆𑀥𑀁𑀲𑁄𑀢𑁄𑀅𑀓𑀦𑀺𑀝𑁆𑀞𑀕𑀸𑀫𑀻𑁇 (𑁧𑁮)

(319.)

962\. 𑀧𑀜𑁆𑀘 **𑀘𑁂𑀢𑁄𑀔𑀺𑀮𑀸𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀭𑀺 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀦 𑀲𑀫𑁆𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀬𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀭𑀺 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺 𑀦𑀸𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀦 𑀲𑀫𑁆𑀧𑀲𑀻𑀤𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬, 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬, 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀘𑁂𑀢𑁄𑀔𑀺𑀮𑁄𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀥𑀫𑁆𑀫𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑁂 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺…  𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀢𑀺…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑀺𑀮𑀚𑀸𑀢𑁄𑁇 𑀬𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀳𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀔𑀺𑀮𑀚𑀸𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬, 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬, 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀘𑁂𑀢𑁄𑀔𑀺𑀮𑁄𑁇 (𑁧𑁯)

(320.)

963\. 𑀧𑀜𑁆𑀘 **𑀘𑁂𑀢𑀲𑁄𑀯𑀺𑀦𑀺𑀩𑀦𑁆𑀥𑀸𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀕𑀢𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄 𑀅𑀯𑀺𑀕𑀢𑀧𑀺𑀧𑀸𑀲𑁄 𑀅𑀯𑀺𑀕𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀅𑀯𑀺𑀕𑀢𑀢𑀡𑁆𑀳𑁄𑁇 𑀬𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀫𑁂𑀲𑀼 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀕𑀢𑀘𑁆𑀙𑀦𑁆𑀤𑁄 𑀅𑀯𑀺𑀕𑀢𑀧𑁂𑀫𑁄 𑀅𑀯𑀺𑀕𑀢𑀧𑀺𑀧𑀸𑀲𑁄 𑀅𑀯𑀺𑀕𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀅𑀯𑀺𑀕𑀢𑀢𑀡𑁆𑀳𑁄, 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀦𑀺𑀩𑀦𑁆𑀥𑁄𑁇 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀬𑁂 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀭𑀽𑀧𑁂 𑀅𑀯𑀻𑀢𑀭𑀸𑀕𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀉𑀤𑀭𑀸𑀯𑀤𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀲𑀼𑀔𑀁 𑀧𑀲𑁆𑀲𑀲𑀼𑀔𑀁 𑀫𑀺𑀤𑁆𑀥𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺—  ‘𑀇𑀫𑀺𑀦𑀸𑀳𑀁 𑀲𑀻𑀮𑁂𑀦 𑀯𑀸 𑀯𑀢𑁂𑀦 𑀯𑀸 𑀢𑀧𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑁂𑀯𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸’𑀢𑀺𑁇 𑀬𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺—  ‘𑀇𑀫𑀺𑀦𑀸𑀳𑀁 𑀲𑀻𑀮𑁂𑀦 𑀯𑀸 𑀯𑀢𑁂𑀦 𑀯𑀸 𑀢𑀧𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑁂𑀯𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸’𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀦𑀫𑀢𑀺 𑀆𑀢𑀧𑁆𑀧𑀸𑀬 𑀅𑀦𑀼𑀬𑁄𑀕𑀸𑀬 𑀲𑀸𑀢𑀘𑁆𑀘𑀸𑀬 𑀧𑀥𑀸𑀦𑀸𑀬𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀘𑁂𑀢𑀲𑁄 𑀯𑀺𑀦𑀺𑀩𑀦𑁆𑀥𑁄𑁇 (𑁨𑁦)

964\. **𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺—** 𑀘𑀓𑁆𑀔𑀼𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀖𑀸𑀦𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀚𑀺𑀯𑁆𑀳𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀓𑀸𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 (𑁨𑁧)

965\. 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 **𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺—** 𑀲𑀼𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀤𑀼𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀉𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 (𑁨𑁨)

966\. 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 **𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺—** 𑀲𑀤𑁆𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀯𑀻𑀭𑀺𑀬𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀲𑀫𑀸𑀥𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁, 𑀧𑀜𑁆𑀜𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁𑁇 (𑁨𑁩)

(321.)

969\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀳𑁂𑀲𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀯𑀺𑀳𑁂𑀲𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑀸𑀬𑁇 𑀬𑁂 𑀘 𑀯𑀺𑀳𑁂𑀲𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀯𑀺𑀳𑁂𑀲𑀸𑀬 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁨𑁪𑁇𑁩)

970\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀽𑀧𑁂 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀭𑀽𑀧𑁂𑀲𑀼 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀅𑀭𑀽𑀧𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀅𑀭𑀽𑀧𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑁂𑀳𑀺𑁇 𑀬𑁂 𑀘 𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀭𑀽𑀧𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁨𑁪𑁇𑁪)

971\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀁 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺 𑀦 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀦 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀫𑀦𑀲𑀺𑀓𑀭𑁄𑀢𑁄 𑀲𑀓𑁆𑀓𑀸𑀬𑀦𑀺𑀭𑁄𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑀢𑀁 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀁 𑀲𑀼𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀼𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑀓𑁆𑀓𑀸𑀬𑁂𑀦𑁇 𑀬𑁂 𑀘 𑀲𑀓𑁆𑀓𑀸𑀬𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀢𑁂𑀳𑀺, 𑀦 𑀲𑁄 𑀢𑀁 𑀯𑁂𑀤𑀦𑀁 𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀢𑀁 𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁𑁇 (𑁨𑁪𑁇𑁫)

(322.)

972\. 𑀧𑀜𑁆𑀘 **𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀸𑀦𑀺𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 (𑁨𑁫𑁇𑁧)

973\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀲𑀢𑁆𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻, 𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀧𑀭𑁂𑀲𑀁 𑀤𑁂𑀲𑁂𑀢𑀺…𑀧𑁂…  𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑀚𑁆𑀛𑀸𑀬𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀧𑀺 𑀘 𑀔𑁄 𑀬𑀣𑀸𑀲𑀼𑀢𑀁 𑀬𑀣𑀸𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀘𑁂𑀢𑀲𑀸 𑀅𑀦𑀼𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀢𑀺 𑀅𑀦𑀼𑀯𑀺𑀘𑀸𑀭𑁂𑀢𑀺 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀢𑀺…𑀧𑁂…  𑀅𑀧𑀺 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀁 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀁 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑁇 𑀬𑀣𑀸 𑀬𑀣𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀁 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀁 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁 𑀧𑀜𑁆𑀜𑀸𑀬 𑀢𑀣𑀸 𑀢𑀣𑀸 𑀲𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑁂 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘 𑀳𑁄𑀢𑀺 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻 𑀘𑁇 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑁄 𑀧𑀸𑀫𑁄𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀲𑁆𑀲 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑁂𑀢𑀺, 𑀲𑀼𑀔𑀺𑀦𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀸𑀬𑀢𑀦𑀁𑁇 (𑁨𑁫𑁇𑁨--𑁫)

974\. 𑀧𑀜𑁆𑀘 **𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀧𑀭𑀺𑀧𑀸𑀘𑀦𑀻𑀬𑀸 𑀲𑀜𑁆𑀜𑀸—** 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀸, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀸𑁇 (𑁨𑁬)

975\. 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸; 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇
