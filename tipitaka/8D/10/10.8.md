# 𑁧𑁦𑁇𑁮 𑀅𑀝𑁆𑀞𑀓

* 𑀲𑀗𑁆𑀕𑀻𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀅𑀝𑁆𑀞𑀓

(333.)

1034\. 𑀅𑀢𑁆𑀣𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸; 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞?

1035\. 𑀅𑀝𑁆𑀞 **𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸—** 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑁄 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑁇 (𑁧)

1036\. 𑀅𑀝𑁆𑀞 **𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀸—** 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 (𑁨)

1037\. 𑀅𑀝𑁆𑀞 **𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸—** 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀅𑀦𑀸𑀕𑀸𑀫𑀻, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀅𑀭𑀳𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄𑁇 (𑁩)

(334.)

1038\. 𑀅𑀝𑁆𑀞 **𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

1039\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑀁, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

1040\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀕𑁆𑀕𑁄 𑀔𑁄 𑀫𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

1041\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀫𑀕𑁆𑀕𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁, 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

1042\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

1043\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀕𑀭𑀼𑀓𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀫𑀸𑀲𑀸𑀘𑀺𑀢𑀁 𑀫𑀜𑁆𑀜𑁂, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀙𑀝𑁆𑀞𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

1044\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄; 𑀅𑀢𑁆𑀣𑀺 𑀓𑀧𑁆𑀧𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇

1045\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁪)

1046\. 𑀅𑀝𑁆𑀞 **𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇** 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀦 𑀦 𑀲𑀼𑀓𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬, 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸, 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇

1047\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑀁, 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇

1049\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀫𑀕𑁆𑀕𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁, 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇

1050\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀮𑀳𑀼𑀓𑁄 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇

1051\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀩𑀮𑀯𑀸 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀙𑀝𑁆𑀞𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇

1052\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄, 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀫𑁂 𑀆𑀩𑀸𑀥𑁄 𑀧𑀯𑀟𑁆𑀠𑁂𑀬𑁆𑀬, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…  𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇

1053\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸, 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀫𑁂 𑀆𑀩𑀸𑀥𑁄 𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫)

1055\. 𑀅𑀝𑁆𑀞 **𑀤𑀸𑀦𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑁄**𑁇 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀁 𑀬𑀸𑀦𑀁 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀸𑀯𑀲𑀣𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁𑁇 𑀲𑁄 𑀬𑀁 𑀤𑁂𑀢𑀺 𑀢𑀁 𑀧𑀘𑁆𑀘𑀸𑀲𑀻𑀲𑀢𑀺𑁇 𑀲𑁄 𑀧𑀲𑁆𑀲𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀁 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀁 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀁 𑀯𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀁 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑀁 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀫𑀸𑀦𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀯𑀸 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀳𑀢𑀺, 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺, 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑀻𑀦𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺 𑀅𑀪𑀸𑀯𑀺𑀢𑀁 𑀢𑀢𑁆𑀭𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀯𑀤𑀸𑀫𑀺 𑀦𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲𑁇 𑀇𑀚𑁆𑀛𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀢𑁄 𑀘𑁂𑀢𑁄𑀧𑀡𑀺𑀥𑀺 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀸𑁇 (𑁭𑁇𑁧)

1057\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁…𑀧𑁂…  𑀲𑁂𑀬𑁆𑀬𑀸𑀯𑀲𑀣𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁𑁇 𑀲𑁄 𑀬𑀁 𑀤𑁂𑀢𑀺 𑀢𑀁 𑀧𑀘𑁆𑀘𑀸𑀲𑀻𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺—  ‘𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀬𑀸𑀫𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀢𑀼𑀲𑀺𑀢𑀸 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀻 𑀤𑁂𑀯𑀸…𑀧𑁂…  𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀤𑁂𑀯𑀸 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀩𑀳𑀼𑀮𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀧𑀭𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀯𑀲𑀯𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀳𑀢𑀺, 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺, 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑀻𑀦𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺 𑀅𑀪𑀸𑀯𑀺𑀢𑀁 𑀢𑀢𑁆𑀭𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀯𑀤𑀸𑀫𑀺 𑀦𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲𑁇 𑀇𑀚𑁆𑀛𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀢𑁄 𑀘𑁂𑀢𑁄𑀧𑀡𑀺𑀥𑀺 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀢𑀸𑁇 (𑁭𑁇𑁩--𑁭)

1058\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀁 𑀬𑀸𑀦𑀁 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀸𑀯𑀲𑀣𑀧𑀤𑀻𑀧𑁂𑀬𑁆𑀬𑀁𑁇 𑀲𑁄 𑀬𑀁 𑀤𑁂𑀢𑀺 𑀢𑀁 𑀧𑀘𑁆𑀘𑀸𑀲𑀻𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀼𑀢𑀁 𑀳𑁄𑀢𑀺—  ‘𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀸 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀯𑀡𑁆𑀡𑀯𑀦𑁆𑀢𑁄 𑀲𑀼𑀔𑀩𑀳𑀼𑀮𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢𑀸𑀳𑀁 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀳𑀢𑀺, 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺, 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀳𑀻𑀦𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺 𑀅𑀪𑀸𑀯𑀺𑀢𑀁 𑀢𑀢𑁆𑀭𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀲𑀻𑀮𑀯𑀢𑁄 𑀯𑀤𑀸𑀫𑀺 𑀦𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀲𑁆𑀲; 𑀯𑀻𑀢𑀭𑀸𑀕𑀲𑁆𑀲 𑀦𑁄 𑀲𑀭𑀸𑀕𑀲𑁆𑀲𑁇 𑀇𑀚𑁆𑀛𑀢𑀸𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀢𑁄 𑀘𑁂𑀢𑁄𑀧𑀡𑀺𑀥𑀺 𑀯𑀻𑀢𑀭𑀸𑀕𑀢𑁆𑀢𑀸𑁇 (𑁭𑁇𑁮)

1059\. 𑀅𑀝𑁆𑀞 **𑀧𑀭𑀺𑀲𑀸—** 𑀔𑀢𑁆𑀢𑀺𑀬𑀧𑀭𑀺𑀲𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀲𑀸, 𑀕𑀳𑀧𑀢𑀺𑀧𑀭𑀺𑀲𑀸, 𑀲𑀫𑀡𑀧𑀭𑀺𑀲𑀸, 𑀘𑀸𑀢𑀼𑀫𑀳𑀸𑀭𑀸𑀚𑀺𑀓𑀧𑀭𑀺𑀲𑀸, 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀧𑀭𑀺𑀲𑀸, 𑀫𑀸𑀭𑀧𑀭𑀺𑀲𑀸, 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀭𑀺𑀲𑀸𑁇 (𑁮)

1061\. 𑀅𑀝𑁆𑀞 **𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀸𑀦𑀺𑁇** 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀞𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇

1062\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺—  𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇

1063\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀢𑀢𑀺𑀬𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇

1064\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇

1065\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀦𑀻𑀮𑀸𑀦𑀺 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀉𑀫𑀸𑀧𑀼𑀧𑁆𑀨𑀁 𑀦𑀻𑀮𑀁 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀁 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀦𑀻𑀮𑀁 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀁 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀦𑀻𑀮𑀸𑀦𑀺 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇

1066\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀻𑀢𑀸𑀦𑀺 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀓𑀡𑀺𑀓𑀸𑀭𑀧𑀼𑀧𑁆𑀨𑀁 𑀧𑀻𑀢𑀁 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀁 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀧𑀻𑀢𑀁 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀁 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀻𑀢𑀸𑀦𑀺 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀧𑀻𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀙𑀝𑁆𑀞𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇

1067\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀦𑁆𑀥𑀼𑀚𑀻𑀯𑀓𑀧𑀼𑀧𑁆𑀨𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇

1068\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀑𑀤𑀸𑀢𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀑𑀲𑀥𑀺𑀢𑀸𑀭𑀓𑀸 𑀑𑀤𑀸𑀢𑀸 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸 𑀯𑀸 𑀧𑀦 𑀢𑀁 𑀯𑀢𑁆𑀣𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀝𑁆𑀞𑀁 𑀑𑀤𑀸𑀢𑀁 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀁 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀏𑀓𑁄 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀑𑀤𑀸𑀢𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, ‘𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀤𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀅𑀪𑀺𑀪𑀸𑀬𑀢𑀦𑀁𑁇 (𑁧𑁦)

1069\. 𑀅𑀝𑁆𑀞 **𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑁇** 𑀭𑀽𑀧𑀻 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1070\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1071\. 𑀲𑀼𑀪𑀦𑁆𑀢𑁂𑀯 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1072\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1073\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1074\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1075\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇

1076\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧𑁧)

1077\. 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀲𑀫𑁆𑀫𑀤𑀓𑁆𑀔𑀸𑀢𑀸; 𑀢𑀢𑁆𑀣 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀲𑀗𑁆𑀕𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇
