# 𑁫𑁇𑁧𑁇𑁬 𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀤𑁂𑀲𑀦𑀸

### 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀻𑀳𑀦𑀸𑀤

### 𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀤𑁂𑀲𑀦𑀸

(150.)

310. 𑀅𑀧𑀭𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀤𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁, 𑀬𑀣𑀸 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀻𑀲𑀼𑁇 𑀲𑀢𑁆𑀢𑀺𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑁇 𑀉𑀪𑀢𑁄𑀪𑀸𑀕𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀓𑀸𑀬𑀲𑀓𑁆𑀔𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀲𑀸𑀭𑀻 𑀲𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑀸𑀭𑀻𑁇 𑀏𑀢𑀤𑀸𑀦𑀼𑀢𑁆𑀢𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀻𑀲𑀼𑁇
