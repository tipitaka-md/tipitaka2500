# 𑁮𑁇𑁨 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑀸

### 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑀸

(245.)

580. 𑀓𑀢𑀫𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑁄, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑁄 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑁄𑁇 𑀇𑀫𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀓𑀫𑁆𑀫𑀓𑀺𑀮𑁂𑀲𑀸 𑀧𑀳𑀻𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

581. _“𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁,_  
_𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀘 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀧𑀭𑀤𑀸𑀭𑀕𑀫𑀦𑀜𑁆𑀘𑁂𑀯,_  
_𑀦𑀧𑁆𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸”𑀢𑀺𑁇_
