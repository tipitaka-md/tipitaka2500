# 𑁮𑁇𑁯 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑀬 𑀙 𑀆𑀤𑀻𑀦𑀯𑀸

### 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑀬 𑀙 𑀆𑀤𑀻𑀦𑀯𑀸

(252.)

590. 𑀙 𑀔𑁄𑀫𑁂, 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀬𑁄𑀕𑁂𑁇 𑀬𑁂 𑀥𑀼𑀢𑁆𑀢𑀸, 𑀬𑁂 𑀲𑁄𑀡𑁆𑀟𑀸, 𑀬𑁂 𑀧𑀺𑀧𑀸𑀲𑀸, 𑀬𑁂 𑀦𑁂𑀓𑀢𑀺𑀓𑀸, 𑀬𑁂 𑀯𑀜𑁆𑀘𑀦𑀺𑀓𑀸, 𑀬𑁂 𑀲𑀸𑀳𑀲𑀺𑀓𑀸𑁇 𑀢𑁆𑀬𑀸𑀲𑁆𑀲 𑀫𑀺𑀢𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑀳𑀸𑀬𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢, 𑀙 𑀆𑀤𑀻𑀦𑀯𑀸 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀬𑁄𑀕𑁂𑁇
