# 𑁮𑁇𑁩 𑀘𑀢𑀼𑀞𑀸𑀦𑀁

### 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀞𑀸𑀦𑀁

(246.)

582. “𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀦 𑀓𑀭𑁄𑀢𑀺? 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀬𑀢𑁄 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀦𑁂𑀯 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; 𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀦 𑀓𑀭𑁄𑀢𑀻”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀅𑀣𑀸𑀧𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘 𑀲𑀢𑁆𑀣𑀸—

583. _“𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀪𑀬𑀸 𑀫𑁄𑀳𑀸,_  
_𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀅𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀦𑀺𑀳𑀻𑀬𑀢𑀺 𑀬𑀲𑁄 𑀢𑀲𑁆𑀲,_  
_𑀓𑀸𑀍𑀅𑀧𑀓𑁆𑀔𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑁇_  


584. _𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀪𑀬𑀸 𑀫𑁄𑀳𑀸,_  
_𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀆𑀧𑀽𑀭𑀢𑀺 𑀬𑀲𑁄 𑀢𑀲𑁆𑀲,_  
_𑀲𑀼𑀓𑁆𑀓𑀧𑀓𑁆𑀔𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸”𑀢𑀺𑁇_
