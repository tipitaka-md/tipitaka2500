# 𑁪𑁇𑁬 𑀅𑀓𑀝𑁆𑀞𑀧𑀸𑀓𑀲𑀸𑀮𑀺𑀧𑀸𑀢𑀼𑀪𑀸𑀯

### 𑀅𑀕𑁆𑀕𑀜𑁆𑀜𑀲𑀼𑀢𑁆𑀢

### 𑀅𑀓𑀝𑁆𑀞𑀧𑀸𑀓𑀲𑀸𑀮𑀺𑀧𑀸𑀢𑀼𑀪𑀸𑀯

(125.)

261\. 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀤𑀸𑀮𑀢𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸𑀬 𑀅𑀓𑀝𑁆𑀞𑀧𑀸𑀓𑁄 𑀲𑀸𑀮𑀺 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺 𑀅𑀓𑀡𑁄 𑀅𑀣𑀼𑀲𑁄 𑀲𑀼𑀤𑁆𑀥𑁄 𑀲𑀼𑀕𑀦𑁆𑀥𑁄 𑀢𑀡𑁆𑀟𑀼𑀮𑀧𑁆𑀨𑀮𑁄𑁇 𑀬𑀁 𑀢𑀁 𑀲𑀸𑀬𑀁 𑀲𑀸𑀬𑀫𑀸𑀲𑀸𑀬 𑀆𑀳𑀭𑀦𑁆𑀢𑀺, 𑀧𑀸𑀢𑁄 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑀽𑀍𑀳𑀁𑁇 𑀬𑀁 𑀢𑀁 𑀧𑀸𑀢𑁄 𑀧𑀸𑀢𑀭𑀸𑀲𑀸𑀬 𑀆𑀳𑀭𑀦𑁆𑀢𑀺, 𑀲𑀸𑀬𑀁 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀁 𑀧𑀝𑀺𑀯𑀺𑀭𑀽𑀍𑀳𑀁; 𑀦𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀲𑀢𑁆𑀢𑀸 𑀅𑀓𑀝𑁆𑀞𑀧𑀸𑀓𑀁 𑀲𑀸𑀮𑀺𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸 𑀢𑀫𑁆𑀪𑀓𑁆𑀔𑀸 𑀢𑀤𑀸𑀳𑀸𑀭𑀸 𑀘𑀺𑀭𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇
