# 𑁫 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀳𑁂𑀢𑀼𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀳𑁂𑀢𑀼𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓

(13.)

36. 𑀓𑀼𑀲𑀮𑀁 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀳𑁂𑀢𑀼𑀜𑁆𑀘𑁂𑀯 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

37. (𑀅𑀓𑀼𑀲𑀮𑀁 𑀢𑀻𑀡𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀁, 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀢𑀻𑀡𑀺 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇)

38. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

(14.)

39. 𑀓𑀼𑀲𑀮𑀁 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑀳𑁂𑀢𑀼 𑀘 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑀳𑁂𑀢𑀼𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁄 𑀘𑁂𑀯 𑀦𑀦𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

40. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…  𑀢𑀻𑀡𑀺𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁…  𑀢𑀻𑀡𑀺𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇
