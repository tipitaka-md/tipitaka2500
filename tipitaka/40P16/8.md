# 𑁮 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀤𑀼𑀓

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀤𑀼𑀓

(18.)

50\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀦𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀦𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀲𑁆𑀲…𑀧𑁂… 𑁇 (𑀙 𑀧𑀜𑁆𑀳𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇) 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀙, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀙, 𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀅𑀯𑀺𑀕𑀢𑁂 𑀙𑁇

51\. 𑀓𑀼𑀲𑀮𑀁 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀓𑀼𑀲𑀮𑁂𑀦 𑀢𑀻𑀡𑀺𑀬𑁂𑀯𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁂𑀦 𑀢𑀻𑀡𑀺𑀬𑁂𑀯𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇
