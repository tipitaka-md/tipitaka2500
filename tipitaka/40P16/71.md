# 𑁭𑁧 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀢𑁆𑀢𑀺𑀓, 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑀤𑀼𑀓

### 𑀲𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀢𑁆𑀢𑀺𑀓, 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽𑀤𑀼𑀓

(117.)

283\. 𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀅𑀧𑁆𑀧𑀝𑀺𑀖𑁄 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀲𑀳𑀪𑀽 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

284\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀙, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀙𑁇
