# 𑁧𑁧 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀮𑁄𑀓𑀺𑀬𑀤𑀼𑀓

(21.)

56\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑀮𑁄𑀓𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀺𑀬𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀺𑀬𑁄 𑀘 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑀮𑁄𑀓𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀺𑀬𑁄 𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀺𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇

57\. 𑀓𑀼𑀲𑀮𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

58\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑀮𑁄𑀓𑀼𑀢𑁆𑀢𑀭𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

59\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀙…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀙𑁇
