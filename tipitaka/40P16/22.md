# 𑁨𑁨 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀤𑀼𑀓𑀸𑀤𑀺

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀤𑀼𑀓𑀸𑀤𑀺

(42.)

119\. 𑀓𑀼𑀲𑀮𑀁 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…  (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑁂𑀭𑀲, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑁂𑀭𑀲𑁇

120\. 𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…  (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑁂𑀭𑀲, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑁂𑀭𑀲𑁇 (𑀏𑀢𑁂 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸, 𑀤𑀼𑀓𑀢𑁆𑀢𑀬𑀁 𑀘𑀺𑀢𑁆𑀢𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀁𑁇)
