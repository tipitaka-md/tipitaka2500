# 𑁧𑁇𑁪𑁇𑁯 𑀧𑀞𑀫𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑀸𑀮𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(39.)

284\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “‘𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄’𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀇𑀥𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀇𑀫𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀅𑀯𑀓𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀧𑀓𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀯𑁂𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀆𑀯𑁂𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀁𑀖𑁄 𑀪𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

285\. “𑀲𑀫𑀕𑁆𑀕𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀓𑀺𑀁 𑀲𑁄 𑀧𑀲𑀯𑀢𑀻”𑀢𑀺? “𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑀺𑀓𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀺𑀩𑁆𑀩𑀺𑀲𑀁 𑀧𑀲𑀯𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑀺𑀓𑀁 𑀓𑀺𑀩𑁆𑀩𑀺𑀲𑀦𑁆”𑀢𑀺? “𑀓𑀧𑁆𑀧𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀧𑀘𑁆𑀘𑀢𑀻𑀢𑀺—

286\. _𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄,_  
_𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄;_  
_𑀯𑀕𑁆𑀕𑀭𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑁄,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀧𑀥𑀁𑀲𑀢𑀺;_  
_𑀲𑀁𑀖𑀁 𑀲𑀫𑀕𑁆𑀕𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀧𑁆𑀧𑀁 𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀧𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇_
