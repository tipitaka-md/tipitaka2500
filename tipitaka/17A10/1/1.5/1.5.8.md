# 𑁧𑁇𑁫𑁇𑁮 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀓𑁆𑀓𑁄𑀲𑀯𑀕𑁆𑀕

### 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑀼𑀢𑁆𑀢

(48.)

336\. “𑀤𑀲𑀬𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑀲?

337\. ‘𑀯𑁂𑀯𑀡𑁆𑀡𑀺𑀬𑀫𑁆𑀳𑀺 𑀅𑀚𑁆𑀛𑀼𑀧𑀕𑀢𑁄’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁧)

338\. ‘𑀧𑀭𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀸 𑀫𑁂 𑀚𑀻𑀯𑀺𑀓𑀸’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁨)

339\. ‘𑀅𑀜𑁆𑀜𑁄 𑀫𑁂 𑀆𑀓𑀧𑁆𑀧𑁄 𑀓𑀭𑀡𑀻𑀬𑁄’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁩)

340\. ‘𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀫𑁂 𑀅𑀢𑁆𑀢𑀸 𑀲𑀻𑀮𑀢𑁄 𑀦 𑀉𑀧𑀯𑀤𑀢𑀻’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁪)

341\. ‘𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀫𑀁 𑀅𑀦𑀼𑀯𑀺𑀘𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀽 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀻𑀮𑀢𑁄 𑀦 𑀉𑀧𑀯𑀤𑀦𑁆𑀢𑀻’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁫)

342\. ‘𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀫𑁂 𑀧𑀺𑀬𑁂𑀳𑀺 𑀫𑀦𑀸𑀧𑁂𑀳𑀺 𑀦𑀸𑀦𑀸𑀪𑀸𑀯𑁄 𑀯𑀺𑀦𑀸𑀪𑀸𑀯𑁄’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁬)

343\. ‘𑀓𑀫𑁆𑀫𑀲𑁆𑀲𑀓𑁄𑀫𑁆𑀳𑀺 𑀓𑀫𑁆𑀫𑀤𑀸𑀬𑀸𑀤𑁄 𑀓𑀫𑁆𑀫𑀬𑁄𑀦𑀺 𑀓𑀫𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀭𑀡𑁄, 𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀯𑀸 𑀧𑀸𑀧𑀓𑀁 𑀯𑀸 𑀢𑀲𑁆𑀲 𑀤𑀸𑀬𑀸𑀤𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁭)

344\. ‘𑀓𑀣𑀁𑀪𑀽𑀢𑀲𑁆𑀲 𑀫𑁂 𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀸 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁮)

345\. ‘𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂 𑀅𑀪𑀺𑀭𑀫𑀸𑀫𑀻’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁯)

346\. ‘𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀫𑁂 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄 𑀅𑀮𑀫𑀭𑀺𑀬𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀥𑀺𑀕𑀢𑁄, 𑀬𑁂𑀦𑀸𑀳𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀓𑀸𑀮𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀧𑀼𑀝𑁆𑀞𑁄 𑀦 𑀫𑀗𑁆𑀓𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁧𑁦)

347\. 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇

---
