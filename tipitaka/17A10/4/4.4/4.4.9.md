# 𑁪𑁇𑁪𑁇𑁯 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

(197.)

1379\. “𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀸𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑀜𑁆𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄𑁇

1380\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇
