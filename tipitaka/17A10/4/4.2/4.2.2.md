# 𑁪𑁇𑁨𑁇𑁨 𑀅𑀭𑀺𑀬𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀭𑀺𑀬𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻𑀲𑀼𑀢𑁆𑀢

(168.)

1173. “𑀅𑀭𑀺𑀬𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀺𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1174. “𑀓𑀢𑀫𑀸 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1175. … ‘𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1176. … ‘𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1177. … ‘𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1178. … ‘𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1179. … ‘𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1180. … ‘𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1181. … ‘𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1182. … ‘𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄…𑀧𑁂…  𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1183. … ‘𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻”𑀢𑀺𑁇
