# 𑁪𑁇𑁨𑁇𑁧 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻𑀲𑀼𑀢𑁆𑀢

(167.)

1156. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑁄 𑀦𑀯𑀁 𑀔𑁄𑀫𑀬𑀼𑀕𑀁 𑀦𑀺𑀯𑀢𑁆𑀣𑁄 𑀅𑀮𑁆𑀮𑀓𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀁 𑀆𑀤𑀸𑀬 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

1157. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑀁 𑀦𑀯𑀁 𑀔𑁄𑀫𑀬𑀼𑀕𑀁 𑀦𑀺𑀯𑀢𑁆𑀣𑀁 𑀅𑀮𑁆𑀮𑀓𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀁 𑀆𑀤𑀸𑀬 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑁄 𑀦𑀯𑀁 𑀔𑁄𑀫𑀬𑀼𑀕𑀁 𑀦𑀺𑀯𑀢𑁆𑀣𑁄 𑀅𑀮𑁆𑀮𑀓𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀁 𑀆𑀤𑀸𑀬 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄? 𑀓𑀺𑀁 𑀦𑁆𑀯𑀚𑁆𑀚 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀲𑁆𑀲𑀸”𑀢𑀺? “𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀚𑁆𑀚 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1158. “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀇𑀥, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀻𑀲𑀁𑀦𑁆𑀳𑀸𑀢𑀸 𑀦𑀯𑀁 𑀔𑁄𑀫𑀬𑀼𑀕𑀁 𑀦𑀺𑀯𑀢𑁆𑀣𑀸 𑀅𑀮𑁆𑀮𑁂𑀦 𑀕𑁄𑀫𑀬𑁂𑀦 𑀧𑀣𑀯𑀺𑀁 𑀑𑀧𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀳𑀭𑀺𑀢𑁂𑀳𑀺 𑀓𑀼𑀲𑁂𑀳𑀺 𑀧𑀢𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀯𑁂𑀮𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀁 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀁 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀅𑀕𑁆𑀕𑀺𑀁 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀸𑀫 𑀪𑀯𑀦𑁆𑀢𑀁, 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀸𑀫 𑀪𑀯𑀦𑁆𑀢𑀦𑁆’𑀢𑀺𑁇 𑀩𑀳𑀼𑀓𑁂𑀦 𑀘 𑀲𑀧𑁆𑀧𑀺𑀢𑁂𑀮𑀦𑀯𑀦𑀻𑀢𑁂𑀦 𑀅𑀕𑁆𑀕𑀺𑀁 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀘 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1159. “𑀅𑀜𑁆𑀜𑀣𑀸 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀣𑀸 𑀘 𑀧𑀦 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 “𑀬𑀣𑀸 𑀓𑀣𑀁 𑀧𑀦, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀺? 𑀲𑀸𑀥𑀼 𑀫𑁂 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼 𑀬𑀣𑀸 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1160. “𑀢𑁂𑀦 𑀳𑀺, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀲𑀼𑀡𑀸𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑁄”𑀢𑀺 𑀔𑁄 𑀚𑀸𑀡𑀼𑀲𑁆𑀲𑁄𑀡𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

1161. “𑀇𑀥, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1162. … ‘𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1163. … ‘𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1164. … ‘𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1165. … ‘𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀧𑀺𑀲𑀼𑀡𑀁 𑀯𑀸𑀘𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1166. … ‘𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀨𑀭𑀼𑀲𑀁 𑀯𑀸𑀘𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1167. … ‘𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1168. … ‘𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀅𑀪𑀺𑀚𑁆𑀛𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀬 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1169. … ‘𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀲𑁆𑀲 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇

1170. … ‘𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀔𑁄 𑀧𑀸𑀧𑀓𑁄 𑀯𑀺𑀧𑀸𑀓𑁄—  𑀤𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀜𑁆𑀘𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀬 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀚𑀳𑀢𑀺; 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1171. “𑀅𑀜𑁆𑀜𑀣𑀸 𑀔𑁄, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀜𑁆𑀜𑀣𑀸 𑀘 𑀧𑀦 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑀺𑀲𑁆𑀲𑀸, 𑀪𑁄 𑀕𑁄𑀢𑀫, 𑀅𑀭𑀺𑀬𑀲𑁆𑀲 𑀯𑀺𑀦𑀬𑁂 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀺𑀬𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀡𑀻 𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑁄 𑀕𑁄𑀢𑀫…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀯𑀁 𑀕𑁄𑀢𑀫𑁄 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇
