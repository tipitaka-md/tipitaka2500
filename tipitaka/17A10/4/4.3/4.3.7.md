# 𑁪𑁇𑁩𑁇𑁭 𑀯𑀚𑁆𑀚𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀸𑀥𑀼𑀯𑀕𑁆𑀕

### 𑀯𑀚𑁆𑀚𑀲𑀼𑀢𑁆𑀢

(184.)

1339. “𑀲𑀸𑀯𑀚𑁆𑀚𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀦𑀯𑀚𑁆𑀚𑀜𑁆𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀥𑀫𑁆𑀫𑁄𑁇

1340. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀚𑁆𑀚𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀯𑀚𑁆𑀚𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇
