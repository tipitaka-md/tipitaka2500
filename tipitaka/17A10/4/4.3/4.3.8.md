# 𑁪𑁇𑁩𑁇𑁮 𑀢𑀧𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀢𑀼𑀢𑁆𑀣𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀸𑀥𑀼𑀯𑀕𑁆𑀕

### 𑀢𑀧𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(185.)

1342\. “𑀢𑀧𑀦𑀻𑀬𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀢𑀧𑀦𑀻𑀬𑀜𑁆𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄𑁇

1343\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇
