# 𑁫𑁇𑁩𑁇𑁨

* 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀭𑀸𑀕𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(238.)

1528\. “𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑀲? 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀆𑀳𑀸𑀭𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀸, 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂 𑀅𑀦𑀪𑀺𑀭𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀝𑁆𑀞𑀺𑀓𑀲𑀜𑁆𑀜𑀸, 𑀧𑀼𑀍𑀅𑀯𑀓𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀦𑀻𑀮𑀓𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀧𑀼𑀩𑁆𑀩𑀓𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀓𑀲𑀜𑁆𑀜𑀸, 𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀢𑀓𑀲𑀜𑁆𑀜𑀸—  𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
