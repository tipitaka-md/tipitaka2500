# 𑁫𑁇𑁩𑁇𑁧

### 𑀧𑀜𑁆𑀘𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀭𑀸𑀕𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(237.)

1527\. “𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑀲? 𑀅𑀲𑀼𑀪𑀲𑀜𑁆𑀜𑀸, 𑀫𑀭𑀡𑀲𑀜𑁆𑀜𑀸, 𑀆𑀳𑀸𑀭𑁂 𑀧𑀝𑀺𑀓𑀽𑀮𑀲𑀜𑁆𑀜𑀸, 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑁂 𑀅𑀦𑀪𑀺𑀭𑀢𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀺𑀘𑁆𑀘𑀲𑀜𑁆𑀜𑀸, 𑀅𑀦𑀺𑀘𑁆𑀘𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀜𑁆𑀜𑀸, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀅𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸, 𑀧𑀳𑀸𑀦𑀲𑀜𑁆𑀜𑀸, 𑀯𑀺𑀭𑀸𑀕𑀲𑀜𑁆𑀜𑀸, 𑀦𑀺𑀭𑁄𑀥𑀲𑀜𑁆𑀜𑀸—  𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
