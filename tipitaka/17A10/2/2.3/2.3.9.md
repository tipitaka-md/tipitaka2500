# 𑁨𑁇𑁩𑁇𑁯 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀆𑀓𑀗𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

(79.)

603\. “𑀤𑀲𑀬𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀤𑀲? ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀅𑀘𑀭𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; ‘𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺, ‘𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀅𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀻’𑀢𑀺…𑀧𑁂…  ‘𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺; 𑀅𑀝𑁆𑀞𑀸𑀦𑁂 𑀘 𑀓𑀼𑀧𑁆𑀧𑀢𑀺—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀲 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀻”𑀢𑀺𑁇
