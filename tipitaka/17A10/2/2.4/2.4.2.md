# 𑁨𑁇𑁪𑁇𑁨 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀣𑁂𑀭𑀯𑀕𑁆𑀕

### 𑀆𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(82.)

613\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

614\. “𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧)

615\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨)

616\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁩)

617\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀤𑀼𑀩𑁆𑀩𑀘𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁪)

618\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁫)

619\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀓𑀼𑀲𑀻𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁬)

620\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁭)

621\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁮)

622\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁯)

623\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧𑁦)

624\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀫𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇

625\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀲𑀤𑁆𑀥𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧)

626\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀲𑀻𑀮𑀯𑀸 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨)

627\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀲𑀼𑀢𑀥𑀭𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁩)

628\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀲𑀼𑀯𑀘𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁪)

629\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁫)

630\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁬)

631\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁭)

632\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁮)

633\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁯)

634\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀲𑀫𑀸𑀦𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧𑁦)

635\. 𑀲𑁄 𑀯𑀢𑀸𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀫𑁂𑀳𑀺 𑀤𑀲𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

---

636\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
