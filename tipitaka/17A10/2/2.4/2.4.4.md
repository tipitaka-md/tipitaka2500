# 𑁨𑁇𑁪𑁇𑁪 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀣𑁂𑀭𑀯𑀕𑁆𑀕

### 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀲𑀼𑀢𑁆𑀢

(84.)

642\. 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀯𑀼𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀯𑁂”𑀢𑀺𑁇 “𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

643\. “𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  ‘𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑁄 𑀯𑀸 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀓𑀼𑀲𑀮𑁄 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀓𑀼𑀲𑀮𑁄 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀢𑀺 𑀲𑀫𑀦𑀼𑀕𑁆𑀕𑀸𑀳𑀢𑀺 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀢𑀺𑁇 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑁂𑀦 𑀯𑀸 𑀛𑀸𑀬𑀺𑀦𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑁂𑀦 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀓𑀼𑀲𑀮𑁂𑀦 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀓𑀼𑀲𑀮𑁂𑀦 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀕𑁆𑀕𑀸𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑁄 𑀇𑀭𑀻𑀡𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀘𑀺𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀦𑀬𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

644\. 𑀢𑀫𑁂𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑁄 𑀯𑀸 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀓𑀼𑀲𑀮𑁄 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀓𑀼𑀲𑀮𑁄 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀢𑀺—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀢𑀺—  𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀫𑀻’𑀢𑀺?

645\. 𑀢𑀫𑁂𑀦𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑁄 𑀯𑀸 𑀛𑀸𑀬𑀻 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀓𑀼𑀲𑀮𑁄 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀬𑀓𑀼𑀲𑀮𑁄 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀧𑀚𑀸𑀦𑀸𑀢𑀺—

646\. ‘𑀓𑁄𑀥𑀦𑁄 𑀔𑁄 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀓𑁄𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀓𑁄𑀥𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁧)

647\. 𑀉𑀧𑀦𑀸𑀳𑀻 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀉𑀧𑀦𑀸𑀳𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀉𑀧𑀦𑀸𑀳𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁨)

648\. 𑀫𑀓𑁆𑀔𑀻 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀫𑀓𑁆𑀔𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀫𑀓𑁆𑀔𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁩)

649\. 𑀧𑀍𑀆𑀲𑀻 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀧𑀍𑀆𑀲𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀧𑀍𑀆𑀲𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁪)

650\. 𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀇𑀲𑁆𑀲𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀲𑁆𑀲𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁫)

651\. 𑀫𑀘𑁆𑀙𑀭𑀻 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀫𑀘𑁆𑀙𑁂𑀭𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀫𑀘𑁆𑀙𑁂𑀭𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁬)

652\. 𑀲𑀞𑁄 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁭)

653\. 𑀫𑀸𑀬𑀸𑀯𑀻 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀫𑀸𑀬𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀫𑀸𑀬𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁮)

654\. 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸; 𑀇𑀘𑁆𑀙𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀩𑀳𑀼𑀮𑀁 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀘𑁆𑀙𑀸𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁𑁇 (𑁯)

655\. 𑀲𑀢𑀺 𑀔𑁄 𑀧𑀦 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑀺 𑀓𑀭𑀡𑀻𑀬𑁂 𑀑𑀭𑀫𑀢𑁆𑀢𑀓𑁂𑀦 𑀯𑀺𑀲𑁂𑀲𑀸𑀥𑀺𑀕𑀫𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀸 𑀯𑁄𑀲𑀸𑀦𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀦𑁆𑀢𑀭𑀸 𑀯𑁄𑀲𑀸𑀦𑀕𑀫𑀦𑀁 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀭𑀺𑀳𑀸𑀦𑀫𑁂𑀢𑀁’𑁇 (𑁧𑁦)

656\. 𑀲𑁄 𑀯𑀢𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑁂 𑀅𑀧𑁆𑀧𑀳𑀸𑀬 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀦𑁂𑀢𑀁 𑀞𑀸𑀦𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀯𑀢𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 ‘𑀇𑀫𑁂 𑀤𑀲 𑀥𑀫𑁆𑀫𑁂 𑀧𑀳𑀸𑀬 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀯𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀭𑀽𑀍𑀳𑀺𑀁 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀞𑀸𑀦𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇
