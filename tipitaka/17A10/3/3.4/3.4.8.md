# 𑁩𑁇𑁪𑁇𑁮 𑀢𑀧𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀸𑀥𑀼𑀯𑀕𑁆𑀕

* 𑀢𑀧𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(141.)

1120\. “𑀢𑀧𑀦𑀻𑀬𑀜𑁆𑀘 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀢𑀧𑀦𑀻𑀬𑀜𑁆𑀘𑁇 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣…𑀧𑁂…  𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄? 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀧𑀦𑀻𑀬𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇
