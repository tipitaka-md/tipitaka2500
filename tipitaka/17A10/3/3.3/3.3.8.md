* 𑀅𑀝𑁆𑀞𑀫𑀲𑀼𑀢𑁆𑀢

(130.)

1096\. “𑀤𑀲𑀬𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀤𑁄𑀲𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑁄𑀳𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀼𑀕𑀢𑀯𑀺𑀦𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑀲? 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀲 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀭𑀸𑀕𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀤𑁄𑀲𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑁄𑀳𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀼𑀕𑀢𑀯𑀺𑀦𑀬𑀸”𑀢𑀺𑁇
