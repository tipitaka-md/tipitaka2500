# 𑁩𑁇𑁧𑁇𑁧 𑀲𑀫𑀡𑀲𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀫𑀡𑀲𑀜𑁆𑀜𑀸𑀯𑀕𑁆𑀕

* 𑀲𑀫𑀡𑀲𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

(101.)

925\. “𑀢𑀺𑀲𑁆𑀲𑁄 𑀇𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄? 𑀯𑁂𑀯𑀡𑁆𑀡𑀺𑀬𑀫𑁆𑀳𑀺 𑀅𑀚𑁆𑀛𑀼𑀧𑀕𑀢𑁄, 𑀧𑀭𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀸 𑀫𑁂 𑀚𑀻𑀯𑀺𑀓𑀸, 𑀅𑀜𑁆𑀜𑁄 𑀫𑁂 𑀆𑀓𑀧𑁆𑀧𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑀢𑀺—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀫𑀡𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺𑁇

926\. 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢? 𑀲𑀦𑁆𑀢𑀢𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺 𑀲𑀦𑁆𑀢𑀢𑀯𑀼𑀢𑁆𑀢𑀺 𑀲𑀻𑀮𑁂𑀲𑀼, 𑀅𑀦𑀪𑀺𑀚𑁆𑀛𑀸𑀮𑀼 𑀳𑁄𑀢𑀺, 𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀦𑀢𑀺𑀫𑀸𑀦𑀻 𑀳𑁄𑀢𑀺, 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, “𑀇𑀤𑀫𑀢𑁆𑀣𑀦𑁆”𑀢𑀺𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀲𑀼, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀫𑀡𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀥𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
