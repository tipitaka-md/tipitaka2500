# 𑁩𑁇𑁧𑁇𑁧𑁨 𑀤𑀼𑀢𑀺𑀬𑀅𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

### 𑀢𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀫𑀡𑀲𑀜𑁆𑀜𑀸𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀅𑀲𑁂𑀔𑀲𑀼𑀢𑁆𑀢

(112.)

981\. “𑀤𑀲𑀬𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁂𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀢𑀫𑁂 𑀤𑀲? 𑀅𑀲𑁂𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀲𑁂𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄, 𑀅𑀲𑁂𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀅𑀲𑁂𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄, 𑀅𑀲𑁂𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑁄, 𑀅𑀲𑁂𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑁄, 𑀅𑀲𑁂𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺, 𑀅𑀲𑁂𑀔𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺, 𑀅𑀲𑁂𑀔𑀁 𑀲𑀫𑁆𑀫𑀸𑀜𑀸𑀡𑀁, 𑀅𑀲𑁂𑀔𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀲 𑀅𑀲𑁂𑀔𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀸”𑀢𑀺𑁇

984\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



985\. _𑀲𑀜𑁆𑀜𑀸 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀁,_  
_𑀩𑀻𑀚𑀁 𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀺𑀚𑁆𑀚𑀭𑀁;_  
_𑀥𑁄𑀯𑀦𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀫𑀦𑀁,_  
_𑀦𑀺𑀤𑁆𑀥𑀫𑀦𑀁 𑀤𑁆𑀯𑁂 𑀅𑀲𑁂𑀔𑀸𑀢𑀺𑁇_
