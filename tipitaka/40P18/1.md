# 𑁧 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

(1.)

2\. 𑀳𑁂𑀢𑀼𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀦𑀳𑁂𑀢𑀼𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

3\. 𑀳𑁂𑀢𑀼𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀦𑀳𑁂𑀢𑀼𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

4\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀧𑀜𑁆𑀘…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

(2.)

5\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀲𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀦𑀳𑁂𑀢𑀼 𑀦𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

6\. 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁, 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀢𑀻𑀡𑀺, 𑀦𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺…𑀧𑁂…  𑀦𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀦𑁄𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

### 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓

7\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀁𑁇)
