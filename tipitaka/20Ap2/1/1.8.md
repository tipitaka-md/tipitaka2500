# 𑁧𑁇𑁮 𑀲𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀫𑁂𑀥𑀸𑀯𑀕𑁆𑀕

### 𑀲𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(71.)

87. _“𑀦𑀕𑀭𑁂 𑀅𑀭𑀼𑀡𑀯𑀢𑀺𑀬𑀸,_  
_𑀅𑀭𑀼𑀡𑁄 𑀦𑀸𑀫 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀢𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀅𑀳𑀼𑀁 𑀪𑀭𑀺𑀬𑀸,_  
_𑀯𑀸𑀭𑀺𑀢𑀁 𑀯𑀸𑀭𑀬𑀸𑀫𑀳𑀁𑁇_  


(72.)

88. _𑀲𑀢𑁆𑀢 𑀫𑀸𑀮𑀸 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑁆𑀧𑀮𑀸 𑀤𑁂𑀯𑀕𑀦𑁆𑀥𑀺𑀓𑀸;_  
_𑀦𑀺𑀲𑀚𑁆𑀚 𑀧𑀸𑀲𑀸𑀤𑀯𑀭𑁂,_  
_𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(73.)

89. _‘𑀓𑀺𑀁 𑀫𑁂 𑀇𑀫𑀸𑀳𑀺 𑀫𑀸𑀮𑀸𑀳𑀺,_  
_𑀲𑀺𑀭𑀲𑀸𑀭𑁄𑀧𑀺𑀢𑀸𑀳𑀺 𑀫𑁂;_  
_𑀯𑀭𑀁 𑀫𑁂 𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀜𑀸𑀡𑀫𑁆𑀳𑀺 𑀅𑀪𑀺𑀭𑁄𑀧𑀺𑀢𑀁’𑁇_  


(74.)

90. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀻,_  
_𑀤𑁆𑀯𑀸𑀭𑀸𑀲𑀦𑁆𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀳𑀁;_  
_‘𑀬𑀤𑀸 𑀏𑀳𑀺𑀢𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀧𑀽𑀚𑀬𑀺𑀲𑁆𑀲𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑀁’𑁇_  


(75.)

91. _𑀓𑀓𑀼𑀥𑁄 𑀯𑀺𑀮𑀲𑀦𑁆𑀢𑁄𑀯,_  
_𑀫𑀺𑀕𑀭𑀸𑀚𑀸𑀯 𑀓𑁂𑀲𑀭𑀻;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀳𑀺𑀢𑁄,_  
_𑀆𑀕𑀘𑁆𑀙𑀺 𑀯𑀻𑀣𑀺𑀬𑀸 𑀚𑀺𑀦𑁄𑁇_  


(76.)

92. _𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀭𑀁𑀲𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀳𑀝𑁆𑀞𑀸 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑀸;_  
_𑀤𑁆𑀯𑀸𑀭𑀁 𑀅𑀯𑀸𑀧𑀼𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀫𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(77.)

93. _𑀲𑀢𑁆𑀢 𑀉𑀧𑁆𑀧𑀮𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀸𑀦𑀺 𑀅𑀫𑁆𑀩𑀭𑁂;_  
_𑀙𑀤𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀫𑀢𑁆𑀣𑀓𑁂 𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(78.)

94. _𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀯𑁂𑀤𑀚𑀸𑀢𑀸 𑀓𑀢𑀜𑁆𑀚𑀮𑀻;_  
_𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(79.)

95. _𑀫𑀳𑀸𑀦𑁂𑀮𑀲𑁆𑀲 𑀙𑀸𑀤𑀦𑀁,_  
_𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀫𑀫 𑀫𑀼𑀤𑁆𑀥𑀦𑀺;_  
_𑀤𑀺𑀩𑁆𑀩𑀕𑀦𑁆𑀥𑀁 𑀧𑀯𑀸𑀬𑀸𑀫𑀺,_  
_𑀲𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀮𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(80.)

96. _𑀓𑀤𑀸𑀘𑀺 𑀦𑀻𑀬𑀫𑀸𑀦𑀸𑀬,_  
_𑀜𑀸𑀢𑀺𑀲𑀗𑁆𑀖𑁂𑀦 𑀫𑁂 𑀢𑀤𑀸;_  
_𑀬𑀸𑀯𑀢𑀸 𑀧𑀭𑀺𑀲𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀫𑀳𑀸𑀦𑁂𑀮𑀁 𑀥𑀭𑀻𑀬𑀢𑀺𑁇_  


(81.)

97. _𑀲𑀢𑁆𑀢𑀢𑀺𑀤𑁂𑀯𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀇𑀲𑁆𑀲𑀭𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀲𑀭𑀸𑀫𑀺 𑀪𑀯𑀸𑀪𑀯𑁂𑁇_  


(82.)

98. _𑀢𑁂𑀲𑀝𑁆𑀞𑀺 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀫𑀫𑀦𑀼𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀆𑀤𑁂𑀬𑁆𑀬𑀯𑀘𑀦𑀸 𑀅𑀳𑀼𑀁𑁇_  


(83.)

99. _𑀉𑀧𑁆𑀧𑀮𑀲𑁆𑀲𑁂𑀯 𑀫𑁂 𑀯𑀡𑁆𑀡𑁄,_  
_𑀕𑀦𑁆𑀥𑁄 𑀘𑁂𑀯 𑀧𑀯𑀸𑀬𑀢𑀺;_  
_𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀺𑀬𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(84.)

100. _𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑁂𑀲𑀼 𑀓𑀼𑀲𑀮𑀸,_  
_𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀪𑀸𑀯𑀦𑀸𑀭𑀢𑀸;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(85.)

101. _𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑀸,_  
_𑀲𑀫𑀸𑀥𑀺𑀛𑀸𑀦𑀕𑁄𑀘𑀭𑀸;_  
_𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(86.)

102. _𑀯𑀻𑀭𑀺𑀬𑀁 𑀫𑁂 𑀥𑀼𑀭𑀥𑁄𑀭𑀬𑁆𑀳𑀁,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸𑀥𑀺𑀯𑀸𑀳𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(87.)

103. _𑀏𑀓𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(88.)

104. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(89.)

105. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(90.)

106. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


107. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀲𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

108. 𑀲𑀢𑁆𑀢𑀼𑀧𑁆𑀧𑀮𑀫𑀸𑀮𑀺𑀓𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
