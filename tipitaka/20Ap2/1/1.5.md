# 𑁧𑁇𑁫 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀫𑁂𑀥𑀸𑀯𑀕𑁆𑀕

### 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(37.)

47. _“𑀘𑀦𑁆𑀤𑀪𑀸𑀕𑀸𑀦𑀤𑀻𑀢𑀻𑀭𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀓𑀺𑀦𑁆𑀦𑀭𑀻 𑀢𑀤𑀸;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑀁𑁇_  


(38.)

48. _𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀯𑁂𑀤𑀚𑀸𑀢𑀸 𑀓𑀢𑀜𑁆𑀚𑀮𑀻;_  
_𑀦𑀍𑀅𑀫𑀸𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀬𑀫𑁆𑀪𑀼𑀁 𑀅𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁𑁇_  


(39.)

49. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀓𑀺𑀦𑁆𑀦𑀭𑀻𑀤𑁂𑀳𑀁,_  
_𑀅𑀕𑀘𑁆𑀙𑀺𑀁 𑀢𑀺𑀤𑀲𑀗𑁆𑀕𑀢𑀺𑀁𑁇_  


(40.)

50. _𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀤𑁂𑀯𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀤𑀲𑀦𑁆𑀦𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀁𑀯𑁂𑀚𑁂𑀢𑁆𑀯𑀸𑀦 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(41.)

51. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(42.)

52. _𑀘𑀢𑀼𑀦𑁆𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀧𑀼𑀧𑁆𑀨𑀫𑀪𑀺𑀧𑀽𑀚𑀬𑀺𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀼𑀧𑁆𑀨𑀧𑀽𑀚𑀸𑀬𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(43.)

53. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(44.)

54. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(45.)

55. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


56. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀓𑀸 𑀣𑁂𑀭𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

57. 𑀦𑀍𑀅𑀫𑀸𑀮𑀺𑀓𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
