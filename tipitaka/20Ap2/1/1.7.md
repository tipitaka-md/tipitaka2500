# 𑁧𑁇𑁭 𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀲𑀼𑀫𑁂𑀥𑀸𑀯𑀕𑁆𑀕

### 𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(60.)

74\. _“𑀧𑀺𑀡𑁆𑀟𑀘𑀸𑀭𑀁 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀢𑀺𑀲𑁆𑀲𑀦𑀸𑀫𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀧𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀤𑀸𑀲𑀳𑀁𑁇_  


(61.)

75\. _𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀯𑀻𑀣𑀺𑀬𑀸 𑀲𑀡𑁆𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀅𑀓𑀸 𑀫𑁂 𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑀁𑁇_  


(62.)

76\. _‘𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀤𑁂𑀯𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(63.)

77\. _𑀧𑀜𑁆𑀜𑀸𑀲𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀫𑀦𑀲𑀸 𑀧𑀢𑁆𑀣𑀺𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀧𑀝𑀺𑀮𑀘𑁆𑀙𑀲𑀺 𑀲𑀩𑁆𑀩𑀤𑀸𑁇_  


(64.)

78\. _𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑀓𑀺𑀜𑁆𑀘𑀦𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑁂 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀲𑀺𑀦𑀸𑀲𑀯𑀸’𑁇_  


(65.)

79\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀦𑀪𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀫𑀻 𑀯𑀻𑀭𑁄,_  
_𑀳𑀁𑀲𑀭𑀸𑀚𑀸𑀯 𑀅𑀫𑁆𑀩𑀭𑁂𑁇_  


(66.)

80\. _𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑁂 𑀤𑀸𑀦𑀯𑀭𑀁,_  
_𑀲𑀼𑀬𑀺𑀝𑁆𑀞𑀸 𑀬𑀸𑀕𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀤𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀢𑀸𑀳𑀁 𑀅𑀘𑀮𑀁 𑀧𑀤𑀁𑁇_  


(67.)

81\. _𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀸𑀦𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(68.)

82\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(69.)

83\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(70.)

84\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


85\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

86\. 𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀸𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
