# 𑁧𑁇𑁧𑁦 𑀉𑀤𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

* 𑀲𑀼𑀫𑁂𑀥𑀸𑀯𑀕𑁆𑀕

* 𑀉𑀤𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(116.)

136\. _“𑀦𑀕𑀭𑁂 𑀩𑀦𑁆𑀥𑀼𑀫𑀢𑀺𑀬𑀸,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀉𑀤𑀳𑀸𑀭𑀺𑀓𑀸;_  
_𑀉𑀤𑀳𑀸𑀭𑁂𑀦 𑀚𑀻𑀯𑀸𑀫𑀺,_  
_𑀢𑁂𑀦 𑀧𑁄𑀲𑁂𑀫𑀺 𑀤𑀸𑀭𑀓𑁂𑁇_  


(117.)

137\. _𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄 𑀘 𑀫𑁂 𑀦𑀢𑁆𑀣𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂;_  
_𑀓𑁄𑀝𑁆𑀞𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀉𑀤𑀓𑀁 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑀳𑀁𑁇_  


(118.)

138\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁;_  
_𑀢𑀢𑁆𑀣 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀩𑁆𑀬𑀫𑁆𑀳𑀁,_  
_𑀉𑀤𑀳𑀸𑀭𑁂𑀦 𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀁𑁇_  


(119.)

139\. _𑀅𑀘𑁆𑀙𑀭𑀸𑀦𑀁 𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲,_  
_𑀅𑀳𑀜𑁆𑀳𑀺 𑀧𑀯𑀭𑀸 𑀢𑀤𑀸;_  
_𑀤𑀲𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀢𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀅𑀪𑀺𑀪𑁄𑀫𑀺 𑀲𑀤𑀸 𑀅𑀳𑀁𑁇_  


(120.)

140\. _𑀧𑀜𑁆𑀜𑀸𑀲𑀁 𑀤𑁂𑀯𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀯𑀻𑀲𑀢𑀺𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(121.)

141\. _𑀤𑀼𑀯𑁂 𑀪𑀯𑁂 𑀲𑀁𑀲𑀭𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑁂 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑁂;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀓𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(122.)

142\. _𑀧𑀩𑁆𑀩𑀢𑀕𑁆𑀕𑁂 𑀤𑀼𑀫𑀕𑁆𑀕𑁂 𑀯𑀸,_  
_𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀘 𑀪𑀽𑀫𑀺𑀬𑀁;_  
_𑀬𑀤𑀸 𑀉𑀤𑀓𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀝𑀺𑀮𑀪𑀸𑀫𑀳𑀁𑁇_  


(123.)

143\. _𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀓𑀸 𑀤𑀺𑀲𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑀸 𑀓𑀼𑀣𑀺𑀢𑀸𑀧𑀺 𑀘;_  
_𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀧𑀯𑀲𑁆𑀲𑀢𑀺𑁇_  


(124.)

144\. _𑀓𑀤𑀸𑀘𑀺 𑀦𑀻𑀬𑀫𑀸𑀦𑀸𑀬,_  
_𑀜𑀸𑀢𑀺𑀲𑀗𑁆𑀖𑁂𑀦 𑀫𑁂 𑀢𑀤𑀸;_  
_𑀬𑀤𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀯𑀲𑁆𑀲𑀁,_  
_𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀅𑀚𑀸𑀬𑀣𑁇_  


(125.)

145\. _𑀉𑀡𑁆𑀳𑀁 𑀯𑀸 𑀧𑀭𑀺𑀍𑀆𑀳𑁄 𑀯𑀸,_  
_𑀲𑀭𑀻𑀭𑁂 𑀫𑁂 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀓𑀸𑀬𑁂 𑀘 𑀫𑁂 𑀭𑀚𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀤𑀓𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(126.)

146\. _𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀫𑀦𑀲𑀸 𑀅𑀚𑁆𑀚,_  
_𑀅𑀧𑁂𑀢𑀫𑀦𑀧𑀸𑀧𑀺𑀓𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(127.)

147\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀓𑀁 𑀅𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀓𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(128.)

148\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(129.)

149\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(130.)

150\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


151\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀉𑀤𑀓𑀤𑀸𑀬𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

152\. 𑀉𑀤𑀓𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

153\. 𑀲𑀼𑀫𑁂𑀥𑀸𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇



---

154\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



155\. _𑀲𑀼𑀫𑁂𑀥𑀸 𑀫𑁂𑀔𑀮𑀸𑀤𑀸𑀬𑀻,_  
_𑀫𑀡𑁆𑀟𑀧𑀁 𑀲𑀗𑁆𑀓𑀫𑀁 𑀤𑀤𑀸;_  
_𑀦𑀍𑀅𑀫𑀸𑀮𑀻 𑀧𑀺𑀡𑁆𑀟𑀤𑀤𑀸,_  
_𑀓𑀝𑀘𑁆𑀙𑀼 𑀉𑀧𑁆𑀧𑀮𑀧𑁆𑀧𑀤𑀸𑁇_  


156\. _𑀤𑀻𑀧𑀤𑀸 𑀤𑀓𑀤𑀸 𑀘𑁂𑀯,_  
_𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀡𑀺𑀢𑀸 𑀇𑀳;_  
_𑀏𑀓𑀕𑀸𑀣𑀸𑀲𑀢𑀜𑁆𑀘𑁂𑀯,_  
_𑀢𑀺𑀁𑀲𑀢𑀺 𑀘 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_
