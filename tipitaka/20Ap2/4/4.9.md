# 𑁪𑁇𑁯 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(204.)

1424. _“𑀬𑁄 𑀭𑀁𑀲𑀺𑀨𑀼𑀲𑀺𑀢𑀸𑀯𑁂𑀍𑀑,_  
_𑀨𑀼𑀲𑁆𑀲𑁄 𑀦𑀸𑀫 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀪𑀕𑀺𑀦𑀻 𑀆𑀲𑀺𑀁,_  
_𑀅𑀚𑀸𑀬𑀺𑀁 𑀔𑀢𑁆𑀢𑀺𑀬𑁂 𑀓𑀼𑀮𑁂𑁇_  


(205.)

1425. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀳𑀁,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀫𑀳𑀸𑀤𑀸𑀦𑀁 𑀤𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀣𑀬𑀺𑀁 𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇_  


(206.)

1426. _𑀏𑀓𑀢𑁆𑀢𑀺𑀁𑀲𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀺𑀔𑀻 𑀮𑁄𑀓𑀕𑁆𑀕𑀦𑀸𑀬𑀓𑁄;_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑁄,_  
_𑀢𑀺𑀮𑁄𑀓𑀲𑀭𑀡𑁄 𑀚𑀺𑀦𑁄𑁇_  


(207.)

1427. _𑀢𑀤𑀸𑀭𑀼𑀡𑀧𑀼𑀭𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀜𑁆𑀜𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀓𑀼𑀧𑀺𑀢𑀸,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀅𑀪𑀺𑀲𑀸𑀧𑀬𑀺𑀁𑁇_  


(208.)

1428. _‘𑀯𑁂𑀲𑀺𑀓𑀸𑀯 𑀅𑀦𑀸𑀘𑀸𑀭𑀸’,_  
_𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑀤𑀽𑀲𑀺𑀓𑀸;_  
_𑀏𑀯𑀁 𑀅𑀓𑁆𑀓𑁄𑀲𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑁂𑀦 𑀧𑀸𑀧𑁂𑀦 𑀓𑀫𑁆𑀫𑀼𑀦𑀸𑁇_  


(209.)

1429. _𑀤𑀸𑀭𑀼𑀡𑀁 𑀦𑀺𑀭𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀫𑀳𑀸𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸;_  
_𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑀸 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀻𑁇_  


(210.)

1430. _𑀤𑀲𑀚𑀸𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀕𑀡𑀺𑀓𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀢𑀫𑁆𑀳𑀸 𑀧𑀸𑀧𑀸 𑀦 𑀫𑀼𑀘𑁆𑀘𑀺𑀲𑁆𑀲𑀁,_  
_𑀪𑀼𑀢𑁆𑀯𑀸 𑀤𑀼𑀝𑁆𑀞𑀯𑀺𑀲𑀁 𑀬𑀣𑀸𑁇_  


(211.)

1431. _𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀫𑀲𑁂𑀯𑀺𑀲𑁆𑀲𑀁,_  
_𑀓𑀲𑁆𑀲𑀧𑁂 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂;_  
_𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀅𑀚𑀸𑀬𑀺𑀁 𑀢𑀺𑀤𑀲𑁂 𑀧𑀼𑀭𑁂𑁇_  


(212.)

1432. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸;_  
_𑀅𑀫𑁆𑀩𑀲𑀸𑀔𑀦𑁆𑀢𑀭𑁂 𑀚𑀸𑀢𑀸,_  
_𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀻𑀢𑀺 𑀢𑁂𑀦𑀳𑀁𑁇_  


(213.)

1433. _𑀧𑀭𑀺𑀯𑀼𑀢𑀸 𑀧𑀸𑀡𑀓𑁄𑀝𑀻𑀳𑀺,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂;_  
_𑀧𑀢𑁆𑀢𑀸𑀳𑀁 𑀅𑀘𑀮𑀁 𑀞𑀸𑀦𑀁,_  
_𑀥𑀻𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀑𑀭𑀲𑀸𑁇_  


(214.)

1434. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫𑀺,_  
_𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸;_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲,_  
_𑀯𑀲𑀻 𑀳𑁄𑀫𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(215.)

1435. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(216.)

1436. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀜𑀸𑀡𑀁 𑀫𑁂 𑀯𑀺𑀫𑀮𑀁 𑀲𑀼𑀤𑁆𑀥𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸𑁇_  


(217.)

1437. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(218.)

1438. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(219.)

1439. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1440. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1441. 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀦𑀯𑀫𑀁𑁇
