# 𑁪𑁇𑁭 𑀅𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(168.)

1384\. _“𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(169.)

1385\. _𑀢𑀤𑀸𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀲𑁆𑀲 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂;_  
_𑀲𑀁𑀯𑀼𑀢𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀫𑁆𑀳𑀺,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀘 𑀧𑀜𑁆𑀘𑀲𑀼𑁇_  


(170.)

1386\. _𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀦𑀻 𑀘 𑀅𑀲𑀦𑁂,_  
_𑀬𑀼𑀢𑁆𑀢𑀸 𑀚𑀸𑀕𑀭𑀺𑀬𑁂𑀧𑀺 𑀘;_  
_𑀯𑀲𑀦𑁆𑀢𑀻 𑀬𑀼𑀢𑁆𑀢𑀬𑁄𑀕𑀸𑀳𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀯𑀺𑀕𑀢𑀸𑀲𑀯𑀁𑁇_  


(171.)

1387\. _𑀅𑀓𑁆𑀓𑁄𑀲𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑀸𑀳𑀁,_  
_𑀕𑀡𑀺𑀓𑁂𑀢𑀺 𑀪𑀡𑀺𑀁 𑀢𑀤𑀸;_  
_𑀢𑁂𑀦 𑀧𑀸𑀧𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀅𑀧𑀘𑁆𑀘𑀺𑀲𑀁𑁇_  


(172.)

1388\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀯𑀲𑁂𑀲𑁂𑀦,_  
_𑀅𑀚𑀸𑀬𑀺𑀁 𑀕𑀡𑀺𑀓𑀸𑀓𑀼𑀮𑁂;_  
_𑀩𑀳𑀼𑀲𑁄𑀯 𑀧𑀭𑀸𑀥𑀻𑀦𑀸,_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀘 𑀚𑀸𑀢𑀺𑀬𑀁𑁇_  


(173.)

1389\. _𑀓𑀸𑀲𑀻𑀲𑀼 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑀚𑀸,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀩𑀮𑁂𑀦𑀳𑀁;_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀯𑀺𑀬 𑀤𑁂𑀯𑁂𑀲𑀼,_  
_𑀅𑀳𑁄𑀲𑀺𑀁 𑀭𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


(174.)

1390\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀁 𑀫𑀁,_  
_𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀕𑀡𑀺𑀓𑀢𑁆𑀢𑁂 𑀦𑀺𑀯𑁂𑀲𑁂𑀲𑀼𑀁,_  
_𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑀩𑀮𑁂𑀦 𑀫𑁂𑁇_  


(175.)

1391\. _𑀲𑀸𑀳𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀧𑀼𑀩𑁆𑀩𑀯𑀸𑀲𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(176.)

1392\. _𑀢𑀤𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀢𑁆𑀣𑀸𑀬,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀚𑀺𑀦𑀲𑀦𑁆𑀢𑀺𑀓𑀁;_  
_𑀫𑀕𑁆𑀕𑁂 𑀥𑀼𑀢𑁆𑀢𑁂 𑀞𑀺𑀢𑁂 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀮𑀪𑀺𑀁 𑀤𑀽𑀢𑁄𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇_  


(177.)

1393\. _𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀸𑀧𑀁 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀲𑀸𑀭𑀫𑀼𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀸,_  
_𑀕𑀡𑀺𑀓𑀢𑁆𑀢𑀜𑁆𑀘 𑀔𑁂𑀧𑀺𑀢𑀁𑁇_  


(178.)

1394\. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲,_  
_𑀯𑀲𑀻 𑀳𑁄𑀫𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(179.)

1395\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(180.)

1396\. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀜𑀸𑀡𑀁 𑀫𑀫 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(181.)

1397\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(182.)

1398\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(183.)

1399\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1400\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀅𑀟𑁆𑀠𑀓𑀸𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1401\. 𑀅𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁𑁇
