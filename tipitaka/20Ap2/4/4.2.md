# 𑁪𑁇𑁨 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀅𑀧𑀤𑀸𑀦

### 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀯𑀕𑁆𑀕

### 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀅𑀧𑀤𑀸𑀦

(13.)

1219. _“𑀘𑀼𑀮𑁆𑀮𑀸𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀜𑁆𑀜𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀸,_  
_𑀧𑀼𑀭𑁂 𑀢𑀼𑀬𑁆𑀳𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(14.)

1220. _𑀯𑁂𑀲𑁆𑀲𑀲𑀼𑀤𑁆𑀤𑀓𑀼𑀮𑁂 𑀚𑀸𑀢𑀸,_  
_𑀤𑁂𑀯𑀸 𑀦𑀸𑀕𑀸 𑀘 𑀓𑀺𑀦𑁆𑀦𑀭𑀸;_  
_𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀻𑀧𑀸 𑀩𑀳𑀽 𑀓𑀜𑁆𑀜𑀸,_  
_𑀧𑀼𑀭𑁂 𑀢𑀼𑀬𑁆𑀳𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(15.)

1221. _𑀓𑀸𑀘𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀩𑁆𑀩𑀤𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑁄 𑀩𑀳𑀽;_  
_𑀤𑁂𑀯𑀸 𑀘 𑀓𑀺𑀦𑁆𑀦𑀭𑀸 𑀦𑀸𑀕𑀸,_  
_𑀨𑀼𑀲𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀕𑀢𑁂𑁇_  


(16.)

1222. _𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸 𑀬𑀲𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀧𑀢𑁆𑀯𑀸𑀦 𑀲𑀩𑁆𑀩𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀢𑀼𑀫𑁆𑀳𑀁 𑀧𑀲𑀸𑀤𑀁 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸,_  
_𑀩𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀕𑀢𑁂𑁇_  


(17.)

1223. _𑀅𑀫𑁆𑀳𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀥𑀻𑀢𑀸 𑀢𑀼,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀜𑁆𑀜𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀧𑁂𑀓𑁆𑀔𑀢𑁄 𑀦𑁄 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀫 𑀘𑀓𑁆𑀔𑀼𑀫𑁇_  


(18.)

1224. _𑀉𑀧𑀸𑀳𑀢𑀸 𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀫𑀽𑀮𑀢𑀡𑁆𑀳𑀸 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀲𑀬𑀸,_  
_𑀧𑀼𑀜𑁆𑀜𑀲𑀗𑁆𑀔𑀸𑀭𑀤𑀸𑀮𑀺𑀢𑀸𑁇_  


(19.)

1225. _𑀲𑀫𑀸𑀥𑀺𑀕𑁄𑀘𑀭𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀲𑀻 𑀓𑀢𑀸;_  
_𑀛𑀸𑀦𑁂𑀦 𑀥𑀫𑁆𑀫𑀭𑀢𑀺𑀬𑀸,_  
_𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫 𑀦𑁄 𑀲𑀤𑀸𑁇_  


(20.)

1226. _𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀘,_  
_𑀲𑀗𑁆𑀔𑀸𑀭𑀸𑀧𑀺 𑀘 𑀔𑁂𑀧𑀺𑀢𑀸;_  
_𑀲𑀼𑀤𑀼𑀤𑁆𑀤𑀲𑀁 𑀧𑀤𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀣 𑀦𑀸𑀬𑀓”𑁇_  


(21.)

1227. _“𑀉𑀧𑀓𑀸𑀭𑀸 𑀫𑀫𑀁 𑀢𑀼𑀫𑁆𑀳𑁂,_  
_𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀓𑀢𑀸𑀯𑀺𑀦𑁄;_  
_𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀲𑀁𑀲𑀬𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀺𑀁”𑁇_  


(22.)

1228. _𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀦𑀺𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀓𑀢𑁆𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀁;_  
_𑀓𑀸𑀘𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀆𑀮𑁄𑀓𑀁,_  
_𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀫𑀣𑀸𑀧𑀭𑀸𑁇_  


(23.)

1229. _𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀘𑀦𑁆𑀤𑀲𑀽𑀭𑀺𑀬𑁂,_  
_𑀲𑀸𑀕𑀭𑀜𑁆𑀘 𑀲𑀫𑀘𑁆𑀙𑀓𑀁;_  
_𑀲𑀺𑀦𑁂𑀭𑀼𑀁 𑀧𑀭𑀺𑀪𑀡𑁆𑀟𑀜𑁆𑀘,_  
_𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀸𑀭𑀺𑀙𑀢𑁆𑀢𑀓𑀁𑁇_  


(24.)

1230. _𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀜𑁆𑀘 𑀪𑀯𑀦𑀁,_  
_𑀬𑀸𑀫𑀁 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺 𑀇𑀤𑁆𑀥𑀺𑀬𑀸;_  
_𑀢𑀼𑀲𑀺𑀢𑀁 𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁂 𑀤𑁂𑀯𑁂,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀻 𑀫𑀳𑀺𑀲𑁆𑀲𑀭𑁂𑁇_  


(25.)

1231. _𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀦𑁄 𑀓𑀸𑀘𑀺 𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀺,_  
_𑀘𑀗𑁆𑀓𑀫𑀜𑁆𑀘 𑀫𑀳𑀸𑀭𑀳𑀁;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀯𑀡𑁆𑀡𑀜𑁆𑀘 𑀫𑀸𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀲𑀼𑀜𑁆𑀜𑀢𑀁𑁇_  


(26.)

1232. _𑀦𑀸𑀦𑀸𑀯𑀺𑀓𑀼𑀩𑁆𑀩𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀇𑀤𑁆𑀥𑀺𑀁 𑀤𑀲𑁆𑀲𑀺𑀬 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀤𑀲𑁆𑀲𑀬𑀺𑀁𑀲𑀼 𑀩𑀮𑀁 𑀲𑀩𑁆𑀩𑀸,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(27.)

1233. _“𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲,_  
_𑀯𑀲𑀻 𑀳𑁄𑀫 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(28.)

1234. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(29.)

1235. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀜𑀸𑀡𑀁 𑀅𑀫𑁆𑀳𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(30.)

1236. _𑀧𑀼𑀩𑁆𑀩𑀸𑀦𑀁 𑀮𑁄𑀓𑀦𑀸𑀣𑀸𑀦𑀁,_  
_𑀲𑀗𑁆𑀕𑀫𑀁 𑀦𑁄 𑀦𑀺𑀤𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀩𑀳𑀼𑀁 𑀅𑀫𑁆𑀳𑀁,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(31.)

1237. _𑀬𑀁 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀼𑀲𑀮𑀁 𑀲𑀭 𑀢𑀁 𑀫𑀼𑀦𑁂;_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀼𑀧𑀘𑀺𑀢𑀸𑀦𑀺 𑀦𑁄𑁇_  


(32.)

1238. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀧𑀼𑀭𑀁 𑀳𑀁𑀲𑀯𑀢𑀻 𑀦𑀸𑀫,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀮𑀸𑀲𑀬𑀁𑁇_  


(33.)

1239. _𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀸,_  
_𑀕𑀗𑁆𑀕𑀸 𑀲𑀦𑁆𑀤𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸;_  
_𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀦𑀤𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽,_  
_𑀕𑀫𑀦𑀁 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(34.)

1240. _𑀤𑀺𑀯𑀲𑀁 𑀤𑁆𑀯𑁂 𑀢𑀬𑁄 𑀘𑁂𑀯,_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀸𑀲𑀺𑀓𑀁 𑀢𑀢𑁄;_  
_𑀘𑀢𑀼𑀫𑀸𑀲𑀫𑁆𑀧𑀺 𑀲𑀫𑁆𑀧𑀼𑀡𑁆𑀡𑀁,_  
_𑀕𑀫𑀦𑀁 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(35.)

1241. _𑀢𑀤𑀸 𑀅𑀳𑀼 𑀲𑀢𑁆𑀢𑀲𑀸𑀭𑁄,_  
_𑀚𑀝𑀺𑀮𑁄 𑀦𑀸𑀫 𑀭𑀝𑁆𑀞𑀺𑀓𑁄;_  
_𑀑𑀭𑀼𑀤𑁆𑀥𑁂 𑀪𑀺𑀓𑁆𑀔𑀯𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑁂𑀢𑀼𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀓𑀸𑀭𑀬𑀺𑁇_  


(36.)

1242. _𑀢𑀤𑀸 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀲𑁂𑀢𑀼𑀁 𑀕𑀗𑁆𑀕𑀸𑀬 𑀓𑀸𑀭𑀬𑀺;_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀑𑀭𑀺𑀫𑁂 𑀢𑀻𑀭𑁂,_  
_𑀯𑀺𑀳𑀸𑀭𑀜𑁆𑀘 𑀅𑀓𑀸𑀭𑀬𑀺𑁇_  


(37.)

1243. _𑀇𑀢𑁆𑀣𑀺𑀬𑁄 𑀧𑀼𑀭𑀺𑀲𑀸 𑀘𑁂𑀯,_  
_𑀉𑀘𑁆𑀘𑀦𑀻𑀘𑀓𑀼𑀮𑀸𑀦𑀺 𑀘;_  
_𑀢𑀲𑁆𑀲 𑀲𑁂𑀢𑀼𑀁 𑀯𑀺𑀳𑀸𑀭𑀜𑁆𑀘,_  
_𑀲𑀫𑀪𑀸𑀕𑀁 𑀅𑀓𑀁𑀲𑀼 𑀢𑁂𑁇_  


(38.)

1244. _𑀅𑀫𑁆𑀳𑁂 𑀅𑀜𑁆𑀜𑁂 𑀘 𑀫𑀸𑀦𑀼𑀚𑀸,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑀸,_  
_𑀦𑀕𑀭𑁂 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀘𑁇_  


(39.)

1245. _𑀇𑀢𑁆𑀣𑀻 𑀧𑀼𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀩𑀳𑀽 𑀘𑁂𑀯 𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑀸;_  
_𑀲𑁂𑀢𑀼𑀦𑁄 𑀘 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲,_  
_𑀯𑀸𑀮𑀼𑀓𑀸 𑀆𑀓𑀺𑀭𑀺𑀁𑀲𑀼 𑀢𑁂𑁇_  


(40.)

1246. _𑀯𑀻𑀣𑀺𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀓𑀤𑀮𑀻𑀧𑀼𑀡𑁆𑀡𑀖𑀝𑁂 𑀥𑀚𑁂;_  
_𑀥𑀽𑀧𑀁 𑀘𑀼𑀡𑁆𑀡𑀜𑁆𑀘 𑀫𑀸𑀮𑀜𑁆𑀘,_  
_𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀲𑀢𑁆𑀣𑀼𑀦𑁄𑁇_  


(41.)

1247. _𑀲𑁂𑀢𑀼𑀯𑀺𑀳𑀸𑀭𑁂 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀸𑀬𑀓𑀁;_  
_𑀫𑀳𑀸𑀤𑀸𑀦𑀁 𑀤𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀺𑀁𑁇_  


(42.)

1248. _𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀢𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀅𑀦𑀼𑀫𑁄𑀤𑀦𑀻𑀬𑀁𑀓𑀸𑀲𑀺,_  
_𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀫𑀳𑀸𑀫𑀼𑀦𑀺𑁇_  


(43.)

1249. _‘𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂,_  
_𑀓𑀧𑁆𑀧𑁄 𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀪𑀤𑁆𑀤𑀓𑁄;_  
_𑀪𑀯𑀸𑀪𑀯𑁂𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀸𑀧𑀼𑀡𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑁄𑀥𑀺𑀬𑀁𑁇_  


(44.)

1250. _𑀓𑀸𑀘𑀺 𑀳𑀢𑁆𑀣𑀧𑀭𑀺𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀢𑀸𑀯𑀻 𑀦𑀭𑀦𑀸𑀭𑀺𑀬𑁄;_  
_𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀳𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑁂,_  
_𑀲𑀩𑁆𑀩𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸’𑁇_  


(45.)

1251. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀯𑀺𑀧𑀸𑀓𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀤𑁂𑀯𑀪𑀯𑀦𑀁,_  
_𑀢𑀼𑀬𑁆𑀳𑀁 𑀢𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀓𑀸𑁇_  


(46.)

1252. _𑀤𑀺𑀩𑁆𑀩𑀲𑀼𑀔𑀁 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁,_  
_𑀫𑀸𑀦𑀼𑀲𑀜𑁆𑀘 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁;_  
_𑀢𑀼𑀬𑁆𑀳𑀁 𑀢𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀫,_  
_𑀲𑀁𑀲𑀭𑀺𑀫𑁆𑀳 𑀪𑀯𑀸𑀪𑀯𑁂𑁇_  


(47.)

1253. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀤𑀁;_  
_𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑀻 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀅𑀣𑁄 𑀤𑁂𑀯𑀧𑀼𑀭𑁂 𑀯𑀭𑁂𑁇_  


(48.)

1254. _𑀭𑀽𑀧𑀪𑁄𑀕𑀬𑀲𑁂 𑀘𑁂𑀯,_  
_𑀅𑀣𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀲𑀓𑁆𑀓𑀢𑀁;_  
_𑀮𑀪𑀸𑀫 𑀲𑀢𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀲𑀼𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀤𑀁𑁇_  


(49.)

1255. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀚𑀸𑀢𑀸𑀫𑁆𑀳 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀓𑀼𑀮𑁂;_  
_𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀸,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀦𑀺𑀯𑁂𑀲𑀦𑁂𑁇_  


(50.)

1256. _𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀫𑁆𑀧𑀺 𑀧𑀣𑀯𑀺𑀁,_  
_𑀦 𑀧𑀲𑁆𑀲𑀸𑀫 𑀦 𑀮𑀗𑁆𑀓𑀢𑀁;_  
_𑀘𑀺𑀓𑁆𑀔𑀮𑁆𑀮𑀪𑀽𑀫𑀺𑀫𑀲𑀼𑀘𑀺𑀁,_  
_𑀦 𑀧𑀲𑁆𑀲𑀸𑀫 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(51.)

1257. _𑀅𑀕𑀸𑀭𑀁 𑀯𑀲𑀦𑁆𑀢𑁂 𑀅𑀫𑁆𑀳𑁂,_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑀁 𑀲𑀩𑁆𑀩𑀓𑀸𑀮𑀺𑀓𑀁;_  
_𑀉𑀧𑀦𑁂𑀦𑁆𑀢𑀺 𑀲𑀤𑀸 𑀲𑀩𑁆𑀩𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀨𑀮𑁂𑀦 𑀦𑁄𑁇_  


(52.)

1258. _𑀅𑀕𑀸𑀭𑀁 𑀧𑀚𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑀁𑀲𑀸𑀭𑀧𑀣𑀦𑀺𑀢𑁆𑀣𑀺𑀡𑁆𑀡𑀸,_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀪𑀯𑀸𑀫𑀲𑁂𑁇_  


(53.)

1259. _𑀘𑀻𑀯𑀭𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀧𑀘𑁆𑀘𑀬𑀁 𑀲𑀬𑀦𑀸𑀲𑀦𑀁;_  
_𑀉𑀧𑀦𑁂𑀦𑁆𑀢𑀺 𑀲𑀤𑀸 𑀅𑀫𑁆𑀳𑁂,_  
_𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀢𑀢𑁄 𑀢𑀢𑁄𑁇_  


(54.)

1260. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀅𑀫𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(55.)

1261. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀦𑁄 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(56.)

1262. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


1263. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀜𑁆𑀜𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1264. 𑀘𑀢𑀼𑀭𑀸𑀲𑀻𑀢𑀺𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
