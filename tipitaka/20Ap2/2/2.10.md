# 𑁨𑁇𑁧𑁦 𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(468.)

644\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(469.)

645\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀚𑀸𑀢𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑁂 𑀅𑀳𑀼𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀧𑀚𑁆𑀚𑁄𑀢𑁂,_  
_𑀫𑀳𑀸𑀲𑀼𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(470.)

646\. _𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁;_  
_𑀢𑀢𑁄 𑀚𑀸𑀢𑀧𑀲𑀸𑀤𑀸𑀳𑀁,_  
_𑀉𑀧𑁂𑀲𑀺𑀁 𑀲𑀭𑀡𑀁 𑀚𑀺𑀦𑀁𑁇_  


(471.)

647\. _𑀢𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀸𑀭𑀻𑀦𑀁,_  
_𑀅𑀕𑁆𑀕𑀁 𑀯𑀡𑁆𑀡𑁂𑀲𑀺 𑀦𑀸𑀬𑀓𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀮𑀚𑁆𑀚𑀺𑀦𑀺𑀁 𑀢𑀸𑀤𑀺𑀁,_  
_𑀓𑀧𑁆𑀧𑀸𑀓𑀧𑁆𑀧𑀯𑀺𑀲𑀸𑀭𑀤𑀁𑁇_  


(472.)

648\. _𑀢𑀤𑀸 𑀫𑀼𑀤𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀳𑀁,_  
_𑀢𑀁 𑀞𑀸𑀦𑀫𑀪𑀺𑀓𑀗𑁆𑀔𑀺𑀦𑀻;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀤𑀲𑀩𑀮𑀁,_  
_𑀲𑀲𑀁𑀖𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(473.)

649\. _𑀪𑁄𑀚𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀢𑁆𑀢𑀸𑀳𑀁,_  
_𑀤𑀤𑀺𑀢𑁆𑀯𑀸𑀯 𑀢𑀺𑀘𑀻𑀯𑀭𑀁;_  
_𑀦𑀺𑀧𑀘𑁆𑀘 𑀲𑀺𑀭𑀲𑀸 𑀧𑀸𑀤𑁂,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(474.)

650\. _‘𑀬𑀸 𑀢𑀬𑀸 𑀯𑀡𑁆𑀡𑀺𑀢𑀸 𑀯𑀻𑀭,_  
_𑀇𑀢𑁄 𑀅𑀝𑁆𑀞𑀫𑀓𑁂 𑀫𑀼𑀦𑀺;_  
_𑀢𑀸𑀤𑀺𑀲𑀸𑀳𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀤𑀺 𑀲𑀺𑀚𑁆𑀛𑀢𑀺 𑀦𑀸𑀬𑀓’𑁇_  


(475.)

651\. _𑀢𑀤𑀸 𑀅𑀯𑁄𑀘 𑀫𑀁 𑀲𑀢𑁆𑀣𑀸,_  
_‘𑀪𑀤𑁆𑀤𑁂 𑀫𑀸 𑀪𑀸𑀬𑀺 𑀅𑀲𑁆𑀲𑀲;_  
_𑀅𑀦𑀸𑀕𑀢𑀫𑁆𑀳𑀺 𑀅𑀤𑁆𑀥𑀸𑀦𑁂,_  
_𑀮𑀘𑁆𑀙𑀲𑁂 𑀢𑀁 𑀫𑀦𑁄𑀭𑀣𑀁𑁇_  


(476.)

652\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(477.)

653\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑀸,_  
_𑀑𑀭𑀲𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑀢𑀺 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀺𑀓𑀸’𑁇_  


(478.)

654\. _𑀢𑀤𑀸𑀳𑀁 𑀫𑀼𑀤𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀢𑀤𑀸 𑀚𑀺𑀦𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀲𑀲𑀁𑀖𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(479.)

655\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(480.)

656\. _𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(481.)

657\. _𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀫𑀳𑁂𑀲𑀺𑀲𑁆𑀲,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀦𑀭𑀺𑀲𑁆𑀲𑀭𑁄;_  
_𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀓𑀺𑀓𑀻 𑀦𑀸𑀫,_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(482.)

658\. _𑀢𑀲𑁆𑀲𑀸𑀲𑀺𑀁 𑀢𑀢𑀺𑀬𑀸 𑀥𑀻𑀢𑀸,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀕𑁆𑀕𑀲𑁆𑀲,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀲𑀫𑀭𑁄𑀘𑀬𑀺𑀁𑁇_  


(483.)

659\. _𑀅𑀦𑀼𑀚𑀸𑀦𑀺 𑀦 𑀦𑁄 𑀢𑀸𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑁂𑀯 𑀢𑀤𑀸 𑀫𑀬𑀁;_  
_𑀯𑀻𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀺𑀘𑀭𑀺𑀫𑁆𑀳 𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑀸𑁇_  


(484.)

660\. _𑀓𑁄𑀫𑀸𑀭𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑁄𑀧𑀝𑁆𑀞𑀸𑀦𑀦𑀺𑀭𑀢𑀸,_  
_𑀫𑀼𑀤𑀺𑀢𑀸 𑀲𑀢𑁆𑀢𑀥𑀻𑀢𑀭𑁄𑁇_  


(485.)

661\. _𑀲𑀫𑀡𑀻 𑀲𑀫𑀡𑀕𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀤𑀸𑀬𑀺𑀓𑀸;_  
_𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀲𑀼𑀥𑀫𑁆𑀫𑀸 𑀘,_  
_𑀲𑀢𑁆𑀢𑀫𑀻 𑀲𑀁𑀖𑀤𑀸𑀬𑀺𑀓𑀸𑁇_  


(486.)

662\. _𑀅𑀳𑀁 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀔𑁂𑀫𑀸 𑀪𑀤𑁆𑀤𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻;_  
_𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀲𑀸𑀔𑀸 𑀳𑁄𑀢𑀺 𑀲𑀢𑁆𑀢𑀫𑀻𑁇_  


(487.)

663\. _𑀢𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀼𑀓𑀢𑁂𑀳𑀺,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(488.)

664\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀚𑀸𑀢𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑁂 𑀅𑀳𑀁;_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀼𑀭𑀯𑀭𑁂,_  
_𑀇𑀤𑁆𑀥𑁂 𑀨𑀻𑀢𑁂 𑀫𑀳𑀤𑁆𑀥𑀦𑁂𑁇_  


(489.)

665\. _𑀬𑀤𑀸 𑀘 𑀬𑁄𑀩𑁆𑀩𑀦𑀽𑀧𑁂𑀢𑀸,_  
_𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀲𑀕𑀸 𑀅𑀳𑀁;_  
_𑀦𑀭𑀁 𑀚𑀸𑀭𑀧𑀢𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀢𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(490.)

666\. _𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀧𑀲𑀽𑀢𑀸𑀳𑀁,_  
_𑀤𑀼𑀢𑀺𑀬𑁄 𑀓𑀼𑀘𑁆𑀙𑀺𑀬𑀸 𑀫𑀫𑀁;_  
_𑀢𑀤𑀸𑀳𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀑𑀓𑁆𑀔𑀸𑀫𑀻𑀢𑀺 𑀲𑀼𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀸𑁇_  


(491.)

667\. _𑀦𑀸𑀭𑁄𑀘𑁂𑀲𑀺𑀁 𑀧𑀢𑀺𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀢𑀤𑀸 𑀢𑀫𑁆𑀳𑀺 𑀧𑀯𑀸𑀲𑀺𑀢𑁂;_  
_𑀏𑀓𑀺𑀓𑀸 𑀦𑀺𑀕𑁆𑀕𑀢𑀸 𑀕𑁂𑀳𑀸,_  
_𑀕𑀦𑁆𑀢𑀼𑀁 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(492.)

668\. _𑀢𑀢𑁄 𑀫𑁂 𑀲𑀸𑀫𑀺 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑀺 𑀧𑀣𑁂 𑀫𑀫𑀁;_  
_𑀢𑀤𑀸 𑀫𑁂 𑀓𑀫𑁆𑀫𑀚𑀸 𑀯𑀸𑀢𑀸,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀢𑀺𑀤𑀸𑀭𑀼𑀡𑀸𑁇_  


(493.)

669\. _𑀉𑀝𑁆𑀞𑀺𑀢𑁄 𑀘 𑀫𑀳𑀸𑀫𑁂𑀖𑁄,_  
_𑀧𑀲𑀽𑀢𑀺𑀲𑀫𑀬𑁂 𑀫𑀫;_  
_𑀤𑀩𑁆𑀩𑀢𑁆𑀣𑀸𑀬 𑀢𑀤𑀸 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀲𑀸𑀫𑀺 𑀲𑀧𑁆𑀧𑁂𑀦 𑀫𑀸𑀭𑀺𑀢𑁄𑁇_  


(494.)

670\. _𑀢𑀤𑀸 𑀯𑀺𑀚𑀸𑀢𑀤𑀼𑀓𑁆𑀔𑁂𑀦,_  
_𑀅𑀦𑀸𑀣𑀸 𑀓𑀧𑀡𑀸 𑀅𑀳𑀁;_  
_𑀓𑀼𑀦𑁆𑀦𑀤𑀺𑀁 𑀧𑀽𑀭𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀲𑀓𑀼𑀮𑀸𑀮𑀬𑀁𑁇_  


(495.)

671\. _𑀩𑀸𑀮𑀁 𑀆𑀤𑀸𑀬 𑀅𑀢𑀭𑀺𑀁,_  
_𑀧𑀸𑀭𑀓𑀽𑀮𑁂 𑀘 𑀏𑀓𑀓𑀁;_  
_𑀲𑀸𑀬𑁂𑀢𑁆𑀯𑀸 𑀩𑀸𑀮𑀓𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀇𑀢𑀭𑀁 𑀢𑀭𑀡𑀸𑀬𑀳𑀁𑁇_  


(496.)

672\. _𑀦𑀺𑀯𑀢𑁆𑀢𑀸 𑀉𑀓𑁆𑀓𑀼𑀲𑁄 𑀳𑀸𑀲𑀺,_  
_𑀢𑀭𑀼𑀡𑀁 𑀯𑀺𑀮𑀧𑀦𑁆𑀢𑀓𑀁;_  
_𑀇𑀢𑀭𑀜𑁆𑀘 𑀯𑀳𑀻 𑀲𑁄𑀢𑁄,_  
_𑀲𑀸𑀳𑀁 𑀲𑁄𑀓𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(497.)

673\. _𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀦𑀕𑀭𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀲𑀚𑀦𑁂 𑀫𑀢𑁂;_  
_𑀢𑀤𑀸 𑀅𑀯𑁄𑀘𑀁 𑀲𑁄𑀓𑀝𑁆𑀝𑀸,_  
_𑀫𑀳𑀸𑀲𑁄𑀓𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(498.)

674\. _‘𑀉𑀪𑁄 𑀧𑀼𑀢𑁆𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸,_  
_𑀧𑀦𑁆𑀣𑁂 𑀫𑀬𑁆𑀳𑀁 𑀧𑀢𑀻 𑀫𑀢𑁄;_  
_𑀫𑀸𑀢𑀸 𑀧𑀺𑀢𑀸 𑀘 𑀪𑀸𑀢𑀸 𑀘,_  
_𑀏𑀓𑀘𑀺𑀢𑀫𑁆𑀳𑀺 𑀟𑀬𑁆𑀳𑀭𑁂’𑁇_  


(499.)

675\. _𑀢𑀤𑀸 𑀓𑀺𑀲𑀸 𑀘 𑀧𑀡𑁆𑀟𑀼 𑀘,_  
_𑀅𑀦𑀸𑀣𑀸 𑀤𑀻𑀦𑀫𑀸𑀦𑀲𑀸;_  
_𑀇𑀢𑁄 𑀢𑀢𑁄 𑀪𑀫𑀦𑁆𑀢𑀻𑀳𑀁,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀦𑀭𑀲𑀸𑀭𑀣𑀺𑀁𑁇_  


(500.)

676\. _𑀢𑀢𑁄 𑀅𑀯𑁄𑀘 𑀫𑀁 𑀲𑀢𑁆𑀣𑀸,_  
_‘𑀧𑀼𑀢𑁆𑀢𑁂 𑀫𑀸 𑀲𑁄𑀘𑀺 𑀅𑀲𑁆𑀲𑀲;_  
_𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂 𑀕𑀯𑁂𑀲𑀲𑁆𑀲𑀼,_  
_𑀓𑀺𑀁 𑀦𑀺𑀭𑀢𑁆𑀣𑀁 𑀯𑀺𑀳𑀜𑁆𑀜𑀲𑀺𑁇_  


(501.)

677\. _𑀦 𑀲𑀦𑁆𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑀸 𑀢𑀸𑀡𑀸𑀬,_  
_𑀦 𑀜𑀸𑀢𑀻 𑀦𑀧𑀺 𑀩𑀦𑁆𑀥𑀯𑀸;_  
_𑀅𑀦𑁆𑀢𑀓𑁂𑀦𑀸𑀥𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀦𑀢𑁆𑀣𑀺 𑀜𑀸𑀢𑀻𑀲𑀼 𑀢𑀸𑀡𑀢𑀸’𑁇_  


(502.)

678\. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀦𑀺𑀦𑁄 𑀯𑀸𑀓𑁆𑀬𑀁,_  
_𑀧𑀞𑀫𑀁 𑀨𑀮𑀫𑀚𑁆𑀛𑀕𑀁;_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀦𑀘𑀺𑀭𑀁,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(503.)

679\. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀓𑀸𑁇_  


(504.)

680\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀯𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀲𑀺𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀮𑀸𑁇_  


(505.)

681\. _𑀢𑀢𑁄𑀳𑀁 𑀯𑀺𑀦𑀬𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀲𑀩𑁆𑀩𑀤𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀉𑀕𑁆𑀕𑀳𑀺𑀁 𑀲𑀩𑁆𑀩𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀁,_  
_𑀩𑁆𑀬𑀸𑀳𑀭𑀺𑀜𑁆𑀘 𑀬𑀣𑀸𑀢𑀣𑀁𑁇_  


(506.)

682\. _𑀚𑀺𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀼𑀡𑁂 𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑀺 𑀫𑀁;_  
_‘𑀅𑀕𑁆𑀕𑀸 𑀯𑀺𑀦𑀬𑀥𑀸𑀭𑀻𑀦𑀁,_  
_𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑀯 𑀏𑀓𑀺𑀓𑀸’𑁇_  


(507.)

683\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(508.)

684\. _𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄𑁇_  


(509.)

685\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(510.)

686\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(511.)

687\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇 (𑁬𑁪𑁧)_  


688\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀧𑀝𑀸𑀘𑀸𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

689\. 𑀧𑀝𑀸𑀘𑀸𑀭𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

690\. 𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀺𑀓𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



---

691\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



692\. _𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀺𑀓𑀸 𑀘𑁂𑀯,_  
_𑀲𑀍𑀅𑀮𑀸 𑀘𑀸𑀣 𑀫𑁄𑀤𑀓𑀸;_  
_𑀏𑀓𑀸𑀲𑀦𑀸 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀸,_  
_𑀦𑀍𑀅𑀫𑀸𑀮𑀻 𑀘 𑀕𑁄𑀢𑀫𑀻𑁇_  


693\. _𑀔𑁂𑀫𑀸 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀧𑀝𑀸𑀘𑀸𑀭𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻;_  
_𑀕𑀸𑀣𑀸 𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑁂𑀯,_  
_𑀦𑀯 𑀘𑀸𑀧𑀺 𑀢𑀤𑀼𑀢𑁆𑀢𑀭𑀺𑁇_
