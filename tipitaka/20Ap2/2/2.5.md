# 𑁨𑁇𑁫 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

* 𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(61.)

225\. _“𑀦𑀕𑀭𑁂 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀸,_  
_𑀘𑀸𑀭𑀺𑀓𑀻 𑀆𑀲𑀳𑀁 𑀢𑀤𑀸;_  
_𑀆𑀭𑀸𑀫𑁂𑀦 𑀘 𑀆𑀭𑀸𑀫𑀁,_  
_𑀘𑀭𑀸𑀫𑀺 𑀓𑀼𑀲𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑁇_  


(62.)

226\. _𑀓𑀸𑀍𑀅𑀧𑀓𑁆𑀔𑀫𑁆𑀳𑀺 𑀤𑀺𑀯𑀲𑁂,_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀩𑁄𑀥𑀺𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀳𑀁𑁇_  


(63.)

227\. _𑀕𑀭𑀼𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀭𑁂 𑀓𑀢𑁆𑀯𑀸𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀁;_  
_𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀁 𑀧𑀯𑁂𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀏𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀺 𑀢𑀸𑀯𑀤𑁂𑁇_  


(64.)

228\. _‘𑀬𑀤𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀫𑀺𑀢𑀕𑀼𑀡𑁄,_  
_𑀅𑀲𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀤𑀲𑁆𑀲𑁂𑀢𑀼 𑀧𑀸𑀝𑀺𑀳𑀻𑀭𑀁 𑀫𑁂,_  
_𑀩𑁄𑀥𑀺 𑀑𑀪𑀸𑀲𑀢𑀼 𑀅𑀬𑀁’𑁇_  


(65.)

229\. _𑀲𑀳 𑀆𑀯𑀚𑁆𑀚𑀺𑀢𑁂 𑀫𑀬𑁆𑀳𑀁,_  
_𑀩𑁄𑀥𑀺 𑀧𑀚𑁆𑀚𑀮𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀲𑀩𑁆𑀩𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀸 𑀆𑀲𑀺,_  
_𑀤𑀺𑀲𑀸 𑀲𑀩𑁆𑀩𑀸 𑀯𑀺𑀭𑁄𑀘𑀢𑀺𑁇_  


(66.)

230\. _𑀲𑀢𑁆𑀢𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀢𑀢𑁆𑀣,_  
_𑀩𑁄𑀥𑀺𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀳𑀁;_  
_𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀧𑀢𑁆𑀢𑁂,_  
_𑀤𑀻𑀧𑀧𑀽𑀚𑀁 𑀅𑀓𑀸𑀲𑀳𑀁𑁇_  


(67.)

231\. _𑀆𑀲𑀦𑀁 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀜𑁆𑀘 𑀤𑀻𑀧𑀸𑀦𑀺 𑀧𑀚𑁆𑀚𑀮𑀼𑀁;_  
_𑀬𑀸𑀯 𑀉𑀤𑁂𑀢𑀺 𑀲𑀽𑀭𑀺𑀬𑁄,_  
_𑀤𑀻𑀧𑀸 𑀫𑁂 𑀧𑀚𑁆𑀚𑀮𑀼𑀁 𑀢𑀤𑀸𑁇_  


(68.)

232\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(69.)

233\. _𑀢𑀢𑁆𑀣 𑀫𑁂 𑀲𑀼𑀓𑀢𑀁 𑀩𑁆𑀬𑀫𑁆𑀳𑀁,_  
_𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀲𑀝𑁆𑀞𑀺𑀬𑁄𑀚𑀦𑀫𑀼𑀩𑁆𑀩𑁂𑀥𑀁,_  
_𑀢𑀺𑀁𑀲𑀬𑁄𑀚𑀦𑀯𑀺𑀢𑁆𑀣𑀢𑀁𑁇_  


(70.)

234\. _𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀸𑀦𑀺 𑀤𑀻𑀧𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂 𑀚𑀮𑀺𑀁𑀲𑀼 𑀫𑁂;_  
_𑀬𑀸𑀯𑀢𑀸 𑀤𑁂𑀯𑀪𑀯𑀦𑀁,_  
_𑀤𑀻𑀧𑀸𑀮𑁄𑀓𑁂𑀦 𑀚𑁄𑀢𑀢𑀺𑁇_  


(71.)

235\. _𑀧𑀭𑀫𑁆𑀫𑀼𑀔𑀸 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀬𑀤𑀺 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁;_  
_𑀉𑀤𑁆𑀥𑀁 𑀅𑀥𑁄 𑀘 𑀢𑀺𑀭𑀺𑀬𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀘𑀓𑁆𑀔𑀼𑀦𑀸𑁇_  


(72.)

236\. _𑀬𑀸𑀯𑀢𑀸 𑀅𑀪𑀺𑀓𑀗𑁆𑀔𑀸𑀫𑀺,_  
_𑀤𑀝𑁆𑀞𑀼𑀁 𑀲𑀼𑀕𑀢𑀤𑀼𑀕𑁆𑀕𑀢𑁂;_  
_𑀢𑀢𑁆𑀣 𑀆𑀯𑀭𑀡𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀭𑀼𑀓𑁆𑀔𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀢𑁂𑀲𑀼 𑀯𑀸𑁇_  


(73.)

237\. _𑀅𑀲𑀻𑀢𑀺 𑀤𑁂𑀯𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀲𑀢𑀸𑀦𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀻𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(74.)

238\. _𑀬𑀁 𑀬𑀁 𑀬𑁄𑀦𑀼𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀢𑁆𑀢𑀁 𑀅𑀣 𑀫𑀸𑀦𑀼𑀲𑀁;_  
_𑀤𑀻𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂 𑀚𑀮𑀦𑁆𑀢𑀺 𑀫𑁂𑁇_  


(75.)

239\. _𑀤𑁂𑀯𑀮𑁄𑀓𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀁 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀬𑀁;_  
_𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀕𑀢𑀸 𑀲𑀦𑁆𑀢𑀻,_  
_𑀅𑀓𑁆𑀔𑀺 𑀫𑁂 𑀦 𑀦𑀺𑀫𑀻𑀮𑀢𑀺𑁇_  


(76.)

240\. _𑀤𑀻𑀧𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑀫𑁆𑀫𑀲𑀫𑀗𑁆𑀕𑀺𑀢𑀸;_  
_𑀚𑀮𑀦𑁆𑀢𑀺 𑀲𑀽𑀢𑀺𑀓𑀸𑀕𑁂𑀳𑁂,_  
_𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀸𑀦𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(77.)

241\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀫𑀸𑀦𑀲𑀁 𑀯𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑀬𑀺𑀁;_  
_𑀅𑀚𑀭𑀸𑀫𑀢𑀁 𑀲𑀻𑀢𑀺𑀪𑀸𑀯𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀨𑀲𑁆𑀲𑀬𑀺𑀁 𑀅𑀳𑀁𑁇_  


(78.)

242\. _𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁;_  
_𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑀬𑀻 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀕𑀼𑀡𑀫𑀜𑁆𑀜𑀸𑀬 𑀕𑁄𑀢𑀫𑁄𑁇_  


(79.)

243\. _𑀫𑀡𑁆𑀟𑀧𑁂 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀸,_  
_𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂 𑀯𑀲𑀦𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀤𑀸 𑀧𑀚𑁆𑀚𑀮𑀢𑁂 𑀤𑀻𑀧𑀁,_  
_𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀸𑀦𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(80.)

244\. _𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀫𑁂,_  
_𑀲𑀫𑀸𑀥𑀺𑀓𑀼𑀲𑀮𑀸 𑀅𑀳𑀁;_  
_𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀸𑀦𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(81.)

245\. _𑀲𑀩𑁆𑀩𑀯𑁄𑀲𑀺𑀢𑀯𑁄𑀲𑀸𑀦𑀸,_  
_𑀓𑀢𑀓𑀺𑀘𑁆𑀘𑀸 𑀅𑀦𑀸𑀲𑀯𑀸;_  
_𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀸 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀫𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑁇_  


(82.)

246\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀤𑀻𑀧𑀫𑀤𑀤𑀺𑀁 𑀢𑀤𑀸;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀸𑀦𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(83.)

247\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(84.)

248\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(85.)

249\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


250\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀤𑀸𑀬𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

251\. 𑀧𑀜𑁆𑀘𑀤𑀻𑀧𑀤𑀸𑀬𑀺𑀓𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
