# 𑁨𑁇𑁮 𑀔𑁂𑀫𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

* 𑀏𑀓𑀽𑀧𑁄𑀲𑀣𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀔𑁂𑀫𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(289.)

460\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(290.)

461\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀚𑀸𑀢𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑁂 𑀅𑀳𑀼𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀧𑀚𑁆𑀚𑁄𑀢𑁂,_  
_𑀫𑀳𑀸𑀲𑀼𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(291.)

462\. _𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁;_  
_𑀢𑀢𑁄 𑀚𑀸𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑀸𑀳𑀁,_  
_𑀉𑀧𑁂𑀫𑀺 𑀲𑀭𑀡𑀁 𑀚𑀺𑀦𑀁𑁇_  


(292.)

463\. _𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀜𑁆𑀘𑀸𑀳𑀁,_  
_𑀆𑀬𑀸𑀘𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀸𑀬𑀓𑀁;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀁,_  
_𑀪𑁄𑀚𑀬𑀺𑀁 𑀲𑀳𑀲𑀸𑀯𑀓𑀁𑁇_  


(293.)

464\. _𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀘 𑀲𑀢𑁆𑀢𑀸𑀳𑁂,_  
_𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀏𑀢𑀤𑀕𑁆𑀕𑀫𑁆𑀳𑀺,_  
_𑀞𑀧𑁂𑀲𑀺 𑀦𑀭𑀲𑀸𑀭𑀣𑀺𑁇_  


(294.)

465\. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀧𑀼𑀦𑁄 𑀢𑀲𑁆𑀲 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀞𑀸𑀦𑀁,_  
_𑀧𑀡𑀺𑀧𑀘𑁆𑀘 𑀧𑀡𑀻𑀤𑀳𑀺𑀁𑁇_  


(295.)

466\. _𑀢𑀢𑁄 𑀫𑀫 𑀚𑀺𑀦𑁄 𑀆𑀳,_  
_‘𑀲𑀺𑀚𑁆𑀛𑀢𑀁 𑀧𑀡𑀺𑀥𑀻 𑀢𑀯;_  
_𑀲𑀲𑀁𑀖𑁂 𑀫𑁂 𑀓𑀢𑀁 𑀓𑀸𑀭𑀁,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀨𑀮𑀁 𑀢𑀬𑀸𑁇_  


(296.)

467\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(297.)

468\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑀸,_  
_𑀑𑀭𑀲𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀔𑁂𑀫𑀸 𑀦𑀸𑀫 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺’𑁇_  


(298.)

469\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀽𑀧𑀕𑀸 𑀅𑀳𑀁𑁇_  


(299.)

470\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀬𑀸𑀫𑀫𑀕𑀁,_  
_𑀢𑀢𑁄𑀳𑀁 𑀢𑀼𑀲𑀺𑀢𑀁 𑀕𑀢𑀸;_  
_𑀢𑀢𑁄 𑀘 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀺𑀁,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀧𑀼𑀭𑀁 𑀢𑀢𑁄𑁇_  


(300.)

471\. _𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣𑀽𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸;_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣𑁂𑀯 𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(301.)

472\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑁂,_  
_𑀭𑀸𑀚𑀽𑀦𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑀁;_  
_𑀫𑀡𑁆𑀟𑀮𑀻𑀦𑀜𑁆𑀘 𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(302.)

473\. _𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑁂𑀯𑁂𑀲𑀼 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀘;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀦𑁂𑀓𑀓𑀧𑁆𑀧𑁂𑀲𑀼 𑀲𑀁𑀲𑀭𑀺𑀁𑁇_  


(303.)

474\. _𑀏𑀓𑀦𑀯𑀼𑀢𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀯𑀺𑀧𑀲𑁆𑀲𑀻 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀘𑀸𑀭𑀼𑀤𑀲𑁆𑀲𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀯𑀺𑀧𑀲𑁆𑀲𑀓𑁄𑁇_  


(304.)

475\. _𑀢𑀫𑀳𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁,_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀭𑀲𑀸𑀭𑀣𑀺𑀁;_  
_𑀥𑀫𑁆𑀫𑀁 𑀪𑀡𑀺𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(305.)

476\. _𑀤𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀲𑁆𑀲 𑀯𑀻𑀭𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀬𑀼𑀢𑁆𑀢𑀬𑁄𑀕𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸𑁇_  


(306.)

477\. _𑀧𑀘𑁆𑀘𑀬𑀸𑀓𑀸𑀭𑀓𑀼𑀲𑀮𑀸,_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀯𑀺𑀲𑀸𑀭𑀤𑀸;_  
_𑀦𑀺𑀧𑀼𑀡𑀸 𑀘𑀺𑀢𑁆𑀢𑀓𑀣𑀺𑀓𑀸,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀓𑀸𑁇_  


(307.)

478\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸𑀳𑀁 𑀢𑀼𑀲𑀺𑀢𑀁,_  
_𑀉𑀧𑀧𑀦𑁆𑀦𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀻;_  
_𑀅𑀪𑀺𑀪𑁄𑀫𑀺 𑀢𑀳𑀺𑀁 𑀅𑀜𑁆𑀜𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀨𑀮𑁂𑀦𑀳𑀁𑁇_  


(308.)

479\. _𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣𑀽𑀧𑀧𑀦𑁆𑀦𑀸𑀳𑀁,_  
_𑀫𑀳𑀸𑀪𑁄𑀕𑀸 𑀫𑀳𑀤𑁆𑀥𑀦𑀸;_  
_𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑀻 𑀭𑀽𑀧𑀯𑀢𑀻,_  
_𑀯𑀺𑀦𑀻𑀢𑀧𑀭𑀺𑀲𑀸𑀧𑀺 𑀘𑁇_  


(309.)

480\. _𑀪𑀯𑀸𑀫𑀺 𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦,_  
_𑀬𑁄𑀕𑁂𑀦 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂;_  
_𑀲𑀩𑁆𑀩𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀫𑀬𑁆𑀳𑀁,_  
_𑀲𑀼𑀮𑀪𑀸 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀸𑁇_  


(310.)

481\. _𑀬𑁄𑀧𑀺 𑀫𑁂 𑀪𑀯𑀢𑁂 𑀪𑀢𑁆𑀢𑀸,_  
_𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀕𑀢𑀸𑀬𑀧𑀺;_  
_𑀯𑀺𑀫𑀸𑀦𑁂𑀢𑀺 𑀦 𑀫𑀁 𑀓𑁄𑀘𑀺,_  
_𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀩𑀮𑁂𑀦 𑀫𑁂𑁇_  


(311.)

482\. _𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀦𑀸𑀫𑁂𑀦 𑀓𑁄𑀡𑀸𑀕𑀫𑀦𑁄,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(312.)

483\. _𑀢𑀤𑀸 𑀳𑀺 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁,_  
_𑀲𑀼𑀲𑀫𑀺𑀤𑁆𑀥𑀓𑀼𑀮𑀧𑁆𑀧𑀚𑀸;_  
_𑀥𑀦𑀜𑁆𑀚𑀸𑀦𑀻 𑀲𑀼𑀫𑁂𑀥𑀸 𑀘,_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀘 𑀢𑀬𑁄 𑀚𑀦𑀸𑁇_  


(313.)

484\. _𑀲𑀁𑀖𑀸𑀭𑀸𑀫𑀫𑀤𑀸𑀲𑀺𑀫𑁆𑀳,_  
_𑀤𑀸𑀦𑀲𑀳𑀸𑀬𑀺𑀓𑀸 𑀧𑀼𑀭𑁂;_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀘 𑀯𑀺𑀳𑀸𑀭𑀫𑁆𑀧𑀺,_  
_𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀓𑀸𑀭𑀺𑀓𑀸 𑀫𑀬𑀁𑁇_  


(314.)

485\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀫𑀬𑀁 𑀲𑀩𑁆𑀩𑀸,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀽𑀧𑀕𑀸 𑀅𑀳𑀼𑀁;_  
_𑀬𑀲𑀲𑀸 𑀅𑀕𑁆𑀕𑀢𑀁 𑀧𑀢𑁆𑀢𑀸,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀢𑀣𑁂𑀯 𑀘𑁇_  


(315.)

486\. _𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀓𑀧𑁆𑀧𑀫𑁆𑀳𑀺,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(316.)

487\. _𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀫𑀳𑁂𑀲𑀺𑀲𑁆𑀲,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀦𑀭𑀺𑀲𑁆𑀲𑀭𑁄;_  
_𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀓𑀺𑀓𑀻 𑀦𑀸𑀫,_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(317.)

488\. _𑀢𑀲𑁆𑀲𑀸𑀲𑀺𑀁 𑀚𑁂𑀝𑁆𑀞𑀺𑀓𑀸 𑀥𑀻𑀢𑀸,_  
_𑀲𑀫𑀡𑀻 𑀇𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀕𑁆𑀕𑀲𑁆𑀲,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀲𑀫𑀭𑁄𑀘𑀬𑀺𑀁𑁇_  


(318.)

489\. _𑀅𑀦𑀼𑀚𑀸𑀦𑀺 𑀦 𑀦𑁄 𑀢𑀸𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑁂𑀯 𑀢𑀤𑀸 𑀫𑀬𑀁;_  
_𑀯𑀻𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀺𑀘𑀭𑀺𑀫𑁆𑀳 𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑀸𑁇_  


(319.)

490\. _𑀓𑁄𑀫𑀸𑀭𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑁄𑀧𑀝𑁆𑀞𑀸𑀦𑀦𑀺𑀭𑀢𑀸,_  
_𑀫𑀼𑀤𑀺𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀻𑀢𑀭𑁄𑁇_  


(320.)

491\. _𑀲𑀫𑀡𑀻 𑀲𑀫𑀡𑀕𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀤𑀸𑀬𑀺𑀓𑀸;_  
_𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀲𑀼𑀥𑀫𑁆𑀫𑀸 𑀘,_  
_𑀲𑀢𑁆𑀢𑀫𑀻 𑀲𑀁𑀖𑀤𑀸𑀬𑀺𑀓𑀸𑁇_  


(321.)

492\. _𑀅𑀳𑀁 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀧𑀝𑀸𑀘𑀸𑀭𑀸 𑀘 𑀓𑀼𑀡𑁆𑀟𑀮𑀸;_  
_𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀲𑀸𑀔𑀸 𑀳𑁄𑀢𑀺 𑀲𑀢𑁆𑀢𑀫𑀻𑁇_  


(322.)

493\. _𑀓𑀤𑀸𑀘𑀺 𑀲𑁄 𑀦𑀭𑀸𑀤𑀺𑀘𑁆𑀘𑁄,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑀁;_  
_𑀫𑀳𑀸𑀦𑀺𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(323.)

494\. _𑀢𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀼𑀓𑀢𑁂𑀳𑀺,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(324.)

495\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀲𑀸𑀓𑀮𑀸𑀬 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀭𑀜𑁆𑀜𑁄 𑀫𑀤𑁆𑀤𑀲𑁆𑀲 𑀥𑀻𑀢𑀸𑀫𑁆𑀳𑀺,_  
_𑀫𑀦𑀸𑀧𑀸 𑀤𑀬𑀺𑀢𑀸 𑀧𑀺𑀬𑀸𑁇_  


(325.)

496\. _𑀲𑀳 𑀫𑁂 𑀚𑀸𑀢𑀫𑀢𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀔𑁂𑀫𑀁 𑀢𑀫𑁆𑀳𑀺 𑀧𑀼𑀭𑁂 𑀅𑀳𑀼;_  
_𑀢𑀢𑁄 𑀔𑁂𑀫𑀸𑀢𑀺 𑀦𑀸𑀫𑀁 𑀫𑁂,_  
_𑀕𑀼𑀡𑀢𑁄 𑀉𑀧𑀧𑀚𑁆𑀚𑀣𑁇_  


(326.)

497\. _𑀬𑀤𑀸𑀳𑀁 𑀬𑁄𑀩𑁆𑀩𑀦𑀁 𑀧𑀢𑁆𑀢𑀸,_  
_𑀭𑀽𑀧𑀮𑀸𑀯𑀜𑁆𑀜𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀢𑀤𑀸 𑀅𑀤𑀸𑀲𑀺 𑀫𑀁 𑀢𑀸𑀢𑁄,_  
_𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀭𑀸𑀚𑀺𑀦𑁄𑁇_  


(327.)

498\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀆𑀲𑀺𑀁,_  
_𑀭𑀽𑀧𑀓𑁂 𑀮𑀸𑀬𑀦𑁂 𑀭𑀢𑀸;_  
_𑀭𑀽𑀧𑀸𑀦𑀁 𑀤𑁄𑀲𑀯𑀸𑀤𑀻𑀢𑀺,_  
_𑀦 𑀉𑀧𑁂𑀲𑀺𑀁 𑀫𑀳𑀸𑀤𑀬𑀁𑁇_  


(328.)

499\. _𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑀤𑀸 𑀭𑀸𑀚𑀸,_  
_𑀫𑀫𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀩𑀼𑀤𑁆𑀥𑀺𑀬𑀸;_  
_𑀯𑀡𑁆𑀡𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑁂𑀍𑀉𑀯𑀦𑀁,_  
_𑀕𑀸𑀬𑀓𑁂 𑀕𑀸𑀧𑀬𑀻 𑀫𑀫𑀁𑁇_  


(329.)

500\. _𑀭𑀫𑁆𑀫𑀁 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀬𑁂𑀦,_  
_𑀦 𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀕𑀢𑀸𑀮𑀬𑀁;_  
_𑀦 𑀢𑁂𑀦 𑀦𑀦𑁆𑀤𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀁,_  
_𑀇𑀢𑀺 𑀫𑀜𑁆𑀜𑀸𑀫𑀲𑁂 𑀫𑀬𑀁𑁇_  


(330.)

501\. _𑀬𑁂𑀦 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀁,_  
_𑀦𑀭𑀦𑀦𑁆𑀤𑀦𑀦𑀦𑁆𑀤𑀦𑀁;_  
_𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑀁 𑀦𑀦𑁆𑀤𑀦𑀁 𑀢𑁂𑀦,_  
_𑀅𑀫𑀭𑀺𑀦𑁆𑀤𑀲𑀼𑀦𑀦𑁆𑀤𑀦𑀁𑁇_  


(331.)

502\. _𑀯𑀺𑀳𑀸𑀬 𑀦𑀦𑁆𑀤𑀦𑀁 𑀤𑁂𑀯𑀸,_  
_𑀑𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀻𑀢𑀮𑀁;_  
_𑀭𑀫𑁆𑀫𑀁 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀦 𑀢𑀧𑁆𑀧𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀺𑀫𑁆𑀳𑀺𑀢𑀸𑁇_  


(332.)

503\. _𑀭𑀸𑀚𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀪𑀽𑀲𑀺𑀢𑀁;_  
_𑀓𑁄 𑀯𑀢𑁆𑀢𑀸 𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑁂𑀲𑀁,_  
_𑀯𑀦𑀲𑁆𑀲 𑀕𑀼𑀡𑀲𑀜𑁆𑀘𑀬𑀁𑁇_  


(333.)

504\. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑀦𑀲𑀫𑀺𑀤𑁆𑀥𑀁,_  
_𑀫𑀫 𑀲𑁄𑀢𑀫𑀦𑁄𑀳𑀭𑀁;_  
_𑀤𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑀸 𑀢𑀫𑀼𑀬𑁆𑀬𑀸𑀦𑀁,_  
_𑀭𑀜𑁆𑀜𑁄 𑀆𑀭𑁄𑀘𑀬𑀺𑀁 𑀢𑀤𑀸𑁇_  


(334.)

505\. _𑀫𑀳𑀢𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦,_  
_𑀢𑀤𑀸 𑀘 𑀲𑁄 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_𑀫𑀁 𑀧𑁂𑀲𑁂𑀲𑀺 𑀢𑀫𑀼𑀬𑁆𑀬𑀸𑀦𑀁,_  
_𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀲𑀫𑀼𑀲𑁆𑀲𑀼𑀓𑀁𑁇_  


(335.)

506\. _‘𑀕𑀘𑁆𑀙 𑀧𑀲𑁆𑀲 𑀫𑀳𑀸𑀪𑁄𑀕𑁂,_  
_𑀯𑀦𑀁 𑀦𑁂𑀢𑁆𑀢𑀭𑀲𑀸𑀬𑀦𑀁;_  
_𑀬𑀁 𑀲𑀤𑀸 𑀪𑀸𑀢𑀺 𑀲𑀺𑀭𑀺𑀬𑀸,_  
_𑀲𑀼𑀕𑀢𑀸𑀪𑀸𑀦𑀼𑀭𑀜𑁆𑀚𑀺𑀢𑀁’𑁇_  


(336.)

507\. _‘𑀬𑀤𑀸 𑀘 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀫𑀼𑀦𑀺,_  
_𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀧𑀯𑀺𑀝𑁆𑀞𑁄𑀳𑀁 𑀢𑀤𑀸𑀬𑁂𑀯,_  
_𑀯𑀦𑀁 𑀤𑀝𑁆𑀞𑀼𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁’𑁇_  


(337.)

508\. _𑀢𑀤𑀸 𑀢𑀁 𑀨𑀼𑀮𑁆𑀮𑀯𑀺𑀧𑀺𑀦𑀁,_  
_𑀦𑀸𑀦𑀸𑀪𑀫𑀭𑀓𑀽𑀚𑀺𑀢𑀁;_  
_𑀓𑁄𑀓𑀺𑀮𑀸𑀕𑀻𑀢𑀲𑀳𑀺𑀢𑀁,_  
_𑀫𑀬𑀽𑀭𑀕𑀡𑀦𑀘𑁆𑀘𑀺𑀢𑀁𑁇_  


(338.)

509\. _𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀫𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀦𑀸𑀦𑀸𑀘𑀗𑁆𑀓𑀫𑀪𑀽𑀲𑀺𑀢𑀁;_  
_𑀓𑀼𑀝𑀺𑀫𑀡𑁆𑀟𑀧𑀲𑀁𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀬𑁄𑀕𑀻𑀯𑀭𑀯𑀺𑀭𑀸𑀚𑀺𑀢𑀁𑁇_  


(339.)

510\. _𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀻 𑀅𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀁,_  
_‘𑀲𑀨𑀮𑀁 𑀦𑀬𑀦𑀁 𑀫𑀫’;_  
_𑀢𑀢𑁆𑀣𑀸𑀧𑀺 𑀢𑀭𑀼𑀡𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_𑀬𑀼𑀢𑁆𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀁𑁇_  


(340.)

511\. _‘𑀈𑀤𑀺𑀲𑁂 𑀯𑀺𑀧𑀺𑀦𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀞𑀺𑀢𑁄𑀬𑀁 𑀦𑀯𑀬𑁄𑀩𑁆𑀩𑀦𑁂;_  
_𑀯𑀲𑀦𑁆𑀢𑀫𑀺𑀯 𑀓𑀦𑁆𑀢𑁂𑀦,_  
_𑀭𑀽𑀧𑁂𑀦 𑀘 𑀲𑀫𑀦𑁆𑀯𑀺𑀢𑁄𑁇_  


(341.)

512\. _𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀫𑁆𑀳𑀺,_  
_𑀫𑀼𑀡𑁆𑀟𑁄 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀸𑀭𑀼𑀢𑁄;_  
_𑀛𑀸𑀬𑀢𑁂 𑀯𑀢𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀬𑀚𑀁 𑀭𑀢𑀺𑀁𑁇_  


(342.)

513\. _𑀦𑀦𑀼 𑀦𑀸𑀫 𑀕𑀳𑀝𑁆𑀞𑁂𑀦,_  
_𑀓𑀸𑀫𑀁 𑀪𑀼𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀲𑀼𑀔𑀁;_  
_𑀧𑀘𑁆𑀙𑀸 𑀚𑀺𑀡𑁆𑀡𑁂𑀦 𑀥𑀫𑁆𑀫𑁄𑀬𑀁,_  
_𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑁄 𑀲𑀼𑀪𑀤𑁆𑀤𑀓𑁄’𑁇_  


(343.)

514\. _‘𑀲𑀼𑀜𑁆𑀜𑀓𑀦𑁆’𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸𑀦,_  
_‘𑀕𑀦𑁆𑀥𑀕𑁂𑀳𑀁 𑀚𑀺𑀦𑀸𑀮𑀬𑀁’;_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀁,_  
_𑀉𑀤𑀬𑀦𑁆𑀢𑀁𑀯 𑀪𑀸𑀓𑀭𑀁𑁇_  


(344.)

515\. _𑀏𑀓𑀓𑀁 𑀲𑀼𑀔𑀫𑀸𑀲𑀻𑀦𑀁,_  
_𑀩𑀻𑀚𑀫𑀸𑀦𑀁 𑀯𑀭𑀺𑀢𑁆𑀣𑀺𑀬𑀸;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑁂𑀯𑀁 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀺𑀁,_  
_‘𑀦𑀸𑀬𑀁 𑀮𑀽𑀔𑁄 𑀦𑀭𑀸𑀲𑀪𑁄𑁇_  


(345.)

516\. _𑀲𑀸 𑀓𑀜𑁆𑀜𑀸 𑀓𑀦𑀓𑀸𑀪𑀸𑀲𑀸,_  
_𑀧𑀤𑀼𑀫𑀸𑀦𑀦𑀮𑁄𑀘𑀦𑀸;_  
_𑀩𑀺𑀫𑁆𑀩𑁄𑀝𑁆𑀞𑀻 𑀓𑀼𑀦𑁆𑀤𑀤𑀲𑀦𑀸,_  
_𑀫𑀦𑁄𑀦𑁂𑀢𑁆𑀢𑀭𑀲𑀸𑀬𑀦𑀸𑁇_  


(346.)

517\. _𑀳𑁂𑀫𑀤𑁄𑀮𑀸𑀪𑀲𑀯𑀦𑀸,_  
_𑀓𑀮𑀺𑀓𑀸𑀓𑀸𑀭𑀲𑀼𑀢𑁆𑀣𑀦𑀻;_  
_𑀯𑁂𑀤𑀺𑀫𑀚𑁆𑀛𑀸𑀯 𑀲𑀼𑀲𑁆𑀲𑁄𑀡𑀻,_  
_𑀭𑀫𑁆𑀪𑁄𑀭𑀼 𑀘𑀸𑀭𑀼𑀪𑀽𑀲𑀦𑀸𑁇_  


(347.)

518\. _𑀭𑀢𑁆𑀢𑀁𑀲𑀓𑀼𑀧𑀲𑀫𑁆𑀩𑁆𑀬𑀸𑀦𑀸,_  
_𑀦𑀻𑀮𑀫𑀝𑁆𑀞𑀦𑀺𑀯𑀸𑀲𑀦𑀸;_  
_𑀅𑀢𑀧𑁆𑀧𑀦𑁂𑀬𑁆𑀬𑀭𑀽𑀧𑁂𑀦,_  
_𑀳𑀸𑀲𑀪𑀸𑀯𑀲𑀫𑀦𑁆𑀯𑀺𑀢𑀸’𑁇_  


(348.)

519\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀢𑀫𑁂𑀯𑀁 𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀺𑀁,_  
_‘𑀅𑀳𑁄𑀬𑀫𑀪𑀺𑀭𑀽𑀧𑀺𑀦𑀻;_  
_𑀦 𑀫𑀬𑀸𑀦𑁂𑀦 𑀦𑁂𑀢𑁆𑀢𑁂𑀦,_  
_𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀸 𑀓𑀼𑀤𑀸𑀘𑀦𑀁’𑁇_  


(349.)

520\. _𑀢𑀢𑁄 𑀚𑀭𑀸𑀪𑀺𑀪𑀽𑀢𑀸 𑀲𑀸,_  
_𑀯𑀺𑀯𑀡𑁆𑀡𑀸 𑀯𑀺𑀓𑀢𑀸𑀦𑀦𑀸;_  
_𑀪𑀺𑀦𑁆𑀦𑀤𑀦𑁆𑀢𑀸 𑀲𑁂𑀢𑀲𑀺𑀭𑀸,_  
_𑀲𑀮𑀸𑀮𑀸 𑀯𑀤𑀦𑀸𑀲𑀼𑀘𑀺𑁇_  


(350.)

521\. _𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀓𑀡𑁆𑀡𑀸 𑀲𑁂𑀢𑀓𑁆𑀔𑀻,_  
_𑀮𑀫𑁆𑀩𑀸𑀲𑀼𑀪𑀧𑀬𑁄𑀥𑀭𑀸;_  
_𑀯𑀮𑀺𑀯𑀺𑀢𑀢𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀻,_  
_𑀲𑀺𑀭𑀸𑀯𑀺𑀢𑀢𑀤𑁂𑀳𑀺𑀦𑀻𑁇_  


(351.)

522\. _𑀦𑀢𑀗𑁆𑀕𑀸 𑀤𑀡𑁆𑀟𑀤𑀼𑀢𑀺𑀬𑀸,_  
_𑀉𑀧𑁆𑀨𑀸𑀲𑀼𑀮𑀺𑀓𑀢𑀸 𑀓𑀺𑀲𑀸;_  
_𑀧𑀯𑁂𑀥𑀫𑀸𑀦𑀸 𑀧𑀢𑀺𑀢𑀸,_  
_𑀦𑀺𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑀻 𑀫𑀼𑀳𑀼𑀁 𑀫𑀼𑀳𑀼𑀁𑁇_  


(352.)

523\. _𑀢𑀢𑁄 𑀫𑁂 𑀆𑀲𑀺 𑀲𑀁𑀯𑁂𑀕𑁄,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑁄;_  
_‘𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀭𑀽𑀧𑀁 𑀅𑀲𑀼𑀘𑀺𑀁,_  
_𑀭𑀫𑀦𑁆𑀢𑁂 𑀬𑀢𑁆𑀣 𑀩𑀸𑀮𑀺𑀲𑀸’𑁇_  


(353.)

524\. _𑀢𑀤𑀸 𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑀁;_  
_𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀕𑀢𑁄,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(354.)

525\. _‘𑀆𑀢𑀼𑀭𑀁 𑀅𑀲𑀼𑀘𑀺𑀁 𑀧𑀽𑀢𑀺𑀁,_  
_𑀧𑀲𑁆𑀲 𑀔𑁂𑀫𑁂 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀁;_  
_𑀉𑀕𑁆𑀖𑀭𑀦𑁆𑀢𑀁 𑀧𑀕𑁆𑀖𑀭𑀦𑁆𑀢𑀁,_  
_𑀩𑀸𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀁𑁇_  


(355.)

526\. _𑀅𑀲𑀼𑀪𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀳𑀺,_  
_𑀏𑀓𑀕𑁆𑀕𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀲𑀢𑀺 𑀓𑀸𑀬𑀕𑀢𑀸 𑀢𑁆𑀬𑀢𑁆𑀣𑀼,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸 𑀩𑀳𑀼𑀮𑀸 𑀪𑀯𑁇_  


(356.)

527\. _𑀬𑀣𑀸 𑀇𑀤𑀁 𑀢𑀣𑀸 𑀏𑀢𑀁,_  
_𑀬𑀣𑀸 𑀏𑀢𑀁 𑀢𑀣𑀸 𑀇𑀤𑀁;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀜𑁆𑀘 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀘,_  
_𑀓𑀸𑀬𑁂 𑀙𑀦𑁆𑀤𑀁 𑀯𑀺𑀭𑀸𑀚𑀬𑁇_  


(357.)

528\. _𑀅𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀸𑀯𑁂𑀳𑀺,_  
_𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀫𑀼𑀚𑁆𑀚𑀳;_  
_𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀬𑀸,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀸 𑀘𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(358.)

529\. _𑀬𑁂 𑀭𑀸𑀕𑀭𑀢𑁆𑀢𑀸𑀦𑀼𑀧𑀢𑀦𑁆𑀢𑀺 𑀲𑁄𑀢𑀁,_  
_𑀲𑀬𑀁 𑀓𑀢𑀁 𑀫𑀓𑁆𑀓𑀝𑀓𑁄𑀯 𑀚𑀸𑀮𑀁;_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀙𑁂𑀢𑁆𑀯𑀸𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀺𑀦𑁄 𑀓𑀸𑀫𑀲𑀼𑀔𑀁 𑀧𑀳𑀸𑀬’𑁇_  


(359.)

530\. _𑀢𑀢𑁄 𑀓𑀮𑁆𑀮𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀁,_  
_𑀜𑀢𑁆𑀯𑀸𑀦 𑀦𑀭𑀲𑀸𑀭𑀣𑀺;_  
_𑀫𑀳𑀸𑀦𑀺𑀤𑀸𑀦𑀁 𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀯𑀺𑀦𑀬𑀸𑀬 𑀫𑁂𑁇_  


(360.)

531\. _𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀲𑁂𑀝𑁆𑀞𑀁 𑀢𑀁,_  
_𑀧𑀼𑀩𑁆𑀩𑀲𑀜𑁆𑀜𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀢𑀢𑁆𑀣 𑀞𑀺𑀢𑀸𑀯𑀳𑀁 𑀲𑀦𑁆𑀢𑀻,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀯𑀺𑀲𑁄𑀥𑀬𑀺𑀁𑁇_  


(361.)

532\. _𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑁂𑀲𑀺𑀲𑁆𑀲,_  
_𑀧𑀸𑀤𑀫𑀽𑀮𑀫𑁆𑀳𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀅𑀘𑁆𑀘𑀬𑀁 𑀤𑁂𑀲𑀦𑀢𑁆𑀣𑀸𑀬,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(362.)

533\. _‘𑀦𑀫𑁄 𑀢𑁂 𑀲𑀩𑁆𑀩𑀤𑀲𑁆𑀲𑀸𑀯𑀻,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀓𑀭𑀼𑀡𑀸𑀓𑀭;_  
_𑀦𑀫𑁄 𑀢𑁂 𑀢𑀺𑀡𑁆𑀡𑀲𑀁𑀲𑀸𑀭,_  
_𑀦𑀫𑁄 𑀢𑁂 𑀅𑀫𑀢𑀁𑀤𑀤𑁇_  


(363.)

534\. _𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀳𑀦𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀸,_  
_𑀓𑀸𑀫𑀭𑀸𑀕𑀯𑀺𑀫𑁄𑀳𑀺𑀢𑀸;_  
_𑀢𑀬𑀸 𑀲𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀬𑁂𑀦,_  
_𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀦𑀬𑁂 𑀭𑀢𑀸𑁇_  


(364.)

535\. _𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀯𑀺𑀪𑁄𑀕𑀸,_  
_𑀢𑀸𑀤𑀺𑀲𑀸𑀦𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀁;_  
_𑀅𑀦𑀼𑀪𑁄𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀲𑀢𑁆𑀢𑀸 𑀲𑀁𑀲𑀸𑀭𑀲𑀸𑀕𑀭𑁂𑁇_  


(365.)

536\. _𑀬𑀤𑀸𑀳𑀁 𑀮𑁄𑀓𑀲𑀭𑀡𑀁,_  
_𑀅𑀭𑀡𑀁 𑀅𑀭𑀡𑀦𑁆𑀢𑀕𑀼𑀁;_  
_𑀦𑀸𑀤𑁆𑀤𑀲𑀸𑀫𑀺 𑀅𑀤𑀽𑀭𑀝𑁆𑀞𑀁,_  
_𑀤𑁂𑀲𑀬𑀸𑀫𑀺 𑀢𑀫𑀘𑁆𑀘𑀬𑀁𑁇_  


(366.)

537\. _𑀫𑀳𑀸𑀳𑀺𑀢𑀁 𑀯𑀭𑀤𑀤𑀁,_  
_𑀅𑀳𑀺𑀢𑁄𑀢𑀺 𑀯𑀺𑀲𑀗𑁆𑀓𑀺𑀢𑀸;_  
_𑀦𑁄𑀧𑁂𑀲𑀺𑀁 𑀭𑀽𑀧𑀦𑀺𑀭𑀢𑀸,_  
_𑀤𑁂𑀲𑀬𑀸𑀫𑀺 𑀢𑀫𑀘𑁆𑀘𑀬𑀁’𑁇_  


(367.)

538\. _𑀢𑀤𑀸 𑀫𑀥𑀼𑀭𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑁄,_  
_𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀚𑀺𑀦𑁄;_  
_𑀅𑀯𑁄𑀘 𑀢𑀺𑀝𑁆𑀞 𑀔𑁂𑀫𑁂𑀢𑀺,_  
_𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑁄 𑀅𑀫𑀢𑁂𑀦 𑀫𑀁𑁇_  


(368.)

539\. _𑀢𑀤𑀸 𑀧𑀡𑀫𑁆𑀬 𑀲𑀺𑀭𑀲𑀸,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀭𑀧𑀢𑀺𑀁,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(369.)

540\. _‘𑀅𑀳𑁄 𑀲𑀫𑁆𑀫𑀸 𑀉𑀧𑀸𑀬𑁄 𑀢𑁂,_  
_𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑁄𑀬𑀫𑀭𑀺𑀦𑁆𑀤𑀫;_  
_𑀯𑀦𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑀸𑀬,_  
_𑀤𑀺𑀝𑁆𑀞𑁄 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀢𑁄 𑀫𑀼𑀦𑀺𑁇_  


(370.)

541\. _𑀬𑀤𑀺 𑀢𑁂 𑀭𑀼𑀘𑁆𑀘𑀢𑁂 𑀭𑀸𑀚,_  
_𑀲𑀸𑀲𑀦𑁂 𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄;_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀭𑀽𑀧𑁂𑀳𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀦𑀸 𑀫𑀼𑀦𑀺𑀯𑀸𑀡𑀺𑀦𑀸’𑁇_  


---

542\. 𑀤𑀼𑀢𑀺𑀬𑀁 𑀪𑀸𑀡𑀯𑀸𑀭𑀁𑁇



(371.)

543\. _𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀤𑀸𑀳 𑀲 𑀫𑀳𑀻𑀧𑀢𑀺;_  
_‘𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺 𑀢𑁂 𑀪𑀤𑁆𑀤𑁂,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀢𑀯 𑀲𑀺𑀚𑁆𑀛𑀢𑀼’𑁇_  


(372.)

544\. _𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀤𑀸 𑀘𑀸𑀳𑀁,_  
_𑀅𑀤𑁆𑀥𑀫𑀸𑀲𑁂 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁂;_  
_𑀤𑀻𑀧𑁄𑀤𑀬𑀜𑁆𑀘 𑀪𑁂𑀤𑀜𑁆𑀘,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀫𑀸𑀦𑀲𑀸𑁇_  


(373.)

545\. _𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀦𑀸 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀔𑀸𑀭𑁂,_  
_𑀧𑀘𑁆𑀘𑀬𑀸𑀓𑀸𑀭𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀘𑀢𑀼𑀭𑁄𑀖𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(374.)

546\. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀆𑀲𑀺𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲,_  
_𑀯𑀲𑀻 𑀘𑀸𑀧𑀺 𑀪𑀯𑀸𑀫𑀳𑀁𑁇_  


(375.)

547\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(376.)

548\. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀫𑀫 𑀜𑀸𑀡𑀁,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑁂𑁇_  


(377.)

549\. _𑀓𑀼𑀲𑀮𑀸𑀳𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀻𑀲𑀼,_  
_𑀓𑀣𑀸𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀲𑀸𑀭𑀤𑀸;_  
_𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑀦𑀬𑀜𑁆𑀜𑀽 𑀘,_  
_𑀯𑀲𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀫𑁆𑀳𑀺 𑀲𑀸𑀲𑀦𑁂𑁇_  


(378.)

550\. _𑀢𑀢𑁄 𑀢𑁄𑀭𑀡𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀭𑀜𑁆𑀜𑀸 𑀓𑁄𑀲𑀮𑀲𑀸𑀫𑀺𑀦𑀸;_  
_𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀦𑀺𑀧𑀼𑀡𑁂 𑀧𑀜𑁆𑀳𑁂,_  
_𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀬𑀣𑀸𑀢𑀣𑀁𑁇_  


(379.)

551\. _𑀢𑀤𑀸 𑀲 𑀭𑀸𑀚𑀸 𑀲𑀼𑀕𑀢𑀁,_  
_𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫 𑀧𑀼𑀘𑁆𑀙𑀣;_  
_𑀢𑀣𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀬𑀣𑀸 𑀢𑁂 𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀫𑀬𑀸𑁇_  


(380.)

552\. _𑀚𑀺𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀼𑀡𑁂 𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑀺 𑀫𑀁;_  
_𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸𑀦𑀫𑀕𑁆𑀕𑀸𑀢𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(381.)

553\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(382.)

554\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(383.)

555\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


556\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀔𑁂𑀫𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

557\. 𑀔𑁂𑀫𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
