# 𑁩𑁇𑁨 𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀻𑀯𑀕𑁆𑀕

### 𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(55.)

750\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(56.)

751\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀚𑀸𑀢𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑁂 𑀓𑀼𑀮𑁂;_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀦𑀭𑀯𑀭𑀁,_  
_𑀲𑀭𑀡𑀁 𑀲𑀫𑀼𑀧𑀸𑀕𑀫𑀺𑀁𑁇_  


(57.)

752\. _𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀢𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁,_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀁;_  
_𑀫𑀥𑀼𑀭𑀁 𑀧𑀭𑀫𑀲𑁆𑀲𑀸𑀤𑀁,_  
_𑀯𑀝𑁆𑀝𑀲𑀦𑁆𑀢𑀺𑀲𑀼𑀔𑀸𑀯𑀳𑀁𑁇_  


(58.)

753\. _𑀢𑀤𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀯𑀻𑀭𑁄,_  
_𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀺𑀦𑀺𑀁;_  
_𑀞𑀧𑁂𑀦𑁆𑀢𑁄 𑀏𑀢𑀤𑀕𑁆𑀕𑀫𑁆𑀳𑀺,_  
_𑀯𑀡𑁆𑀡𑀬𑀻 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(59.)

754\. _𑀚𑀦𑁂𑀢𑁆𑀯𑀸𑀦𑀧𑁆𑀧𑀓𑀁 𑀧𑀻𑀢𑀺𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀕𑀼𑀡𑁂;_  
_𑀓𑀸𑀭𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀬𑀣𑀸𑀲𑀢𑁆𑀢𑀺 𑀬𑀣𑀸𑀩𑀮𑀁𑁇_  


(60.)

755\. _𑀦𑀺𑀧𑀘𑁆𑀘 𑀫𑀼𑀦𑀺𑀯𑀭𑀁 𑀢𑀁,_  
_𑀢𑀁 𑀞𑀸𑀦𑀫𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀺𑀁;_  
_𑀢𑀤𑀸𑀦𑀼𑀫𑁄𑀤𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀞𑀸𑀦𑀮𑀸𑀪𑀸𑀬 𑀦𑀸𑀬𑀓𑁄𑁇_  


(61.)

756\. _‘𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(62.)

757\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑀸,_  
_𑀑𑀭𑀲𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀺𑀓𑀸’𑁇_  


(63.)

758\. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀢𑀤𑀸 𑀚𑀺𑀦𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀧𑀘𑁆𑀘𑀬𑁂𑀳𑀺 𑀯𑀺𑀦𑀸𑀬𑀓𑀁𑁇_  


(64.)

759\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(65.)

760\. _𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(66.)

761\. _𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀫𑀳𑁂𑀲𑀺𑀲𑁆𑀲,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀦𑀭𑀺𑀲𑁆𑀲𑀭𑁄;_  
_𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀓𑀺𑀓𑀻 𑀦𑀸𑀫,_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(67.)

762\. _𑀧𑀜𑁆𑀘𑀫𑀻 𑀢𑀲𑁆𑀲 𑀥𑀻𑀢𑀸𑀲𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑀸 𑀦𑀸𑀫𑁂𑀦 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀕𑁆𑀕𑀲𑁆𑀲,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀲𑀫𑀭𑁄𑀘𑀬𑀺𑀁𑁇_  


(68.)

763\. _𑀅𑀦𑀼𑀚𑀸𑀦𑀺 𑀦 𑀦𑁄 𑀢𑀸𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑁂𑀯 𑀢𑀤𑀸 𑀫𑀬𑀁;_  
_𑀯𑀻𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀺𑀘𑀭𑀺𑀫𑁆𑀳 𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑀸𑁇_  


(69.)

764\. _𑀓𑁄𑀫𑀸𑀭𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑁄𑀧𑀝𑁆𑀞𑀸𑀦𑀦𑀺𑀭𑀢𑀸,_  
_𑀫𑀼𑀤𑀺𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀻𑀢𑀭𑁄𑁇_  


(70.)

765\. _𑀲𑀫𑀡𑀻 𑀲𑀫𑀡𑀕𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀤𑀸𑀬𑀺𑀓𑀸;_  
_𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀲𑀼𑀥𑀫𑁆𑀫𑀸 𑀘,_  
_𑀲𑀢𑁆𑀢𑀫𑀻 𑀲𑀁𑀖𑀤𑀸𑀬𑀺𑀓𑀸𑁇_  


(71.)

766\. _𑀔𑁂𑀫𑀸 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀧𑀝𑀸𑀘𑀸𑀭𑀸 𑀘 𑀓𑀼𑀡𑁆𑀟𑀮𑀸;_  
_𑀅𑀳𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸 𑀘,_  
_𑀯𑀺𑀲𑀸𑀔𑀸 𑀳𑁄𑀢𑀺 𑀲𑀢𑁆𑀢𑀫𑀻𑁇_  


(72.)

767\. _𑀢𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀼𑀓𑀢𑁂𑀳𑀺,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(73.)

768\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀚𑀸𑀢𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑁂 𑀅𑀳𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀢𑁂 𑀅𑀥𑀦𑁂 𑀦𑀝𑁆𑀞𑁂,_  
_𑀕𑀢𑀸 𑀘 𑀲𑀥𑀦𑀁 𑀓𑀼𑀮𑀁𑁇_  


(74.)

769\. _𑀧𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑁂𑀲𑀸 𑀫𑁂,_  
_𑀤𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀥𑀦𑀸 𑀇𑀢𑀺;_  
_𑀬𑀤𑀸 𑀘 𑀧𑀲𑀽𑀢𑀸 𑀆𑀲𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀤𑀬𑀺𑀢𑀸 𑀢𑀤𑀸𑁇_  


(75.)

770\. _𑀬𑀤𑀸 𑀲𑁄 𑀢𑀭𑀼𑀡𑁄 𑀪𑀤𑁆𑀤𑁄,_  
_𑀓𑁄𑀫𑀮𑀓𑁄 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑁄;_  
_𑀲𑀧𑀸𑀡𑀫𑀺𑀯 𑀓𑀦𑁆𑀢𑁄 𑀫𑁂,_  
_𑀢𑀤𑀸 𑀬𑀫𑀯𑀲𑀁 𑀕𑀢𑁄𑁇_  


(76.)

771\. _𑀲𑁄𑀓𑀝𑁆𑀝𑀸𑀤𑀻𑀦𑀯𑀤𑀦𑀸,_  
_𑀅𑀲𑁆𑀲𑀼𑀦𑁂𑀢𑁆𑀢𑀸 𑀭𑀼𑀤𑀫𑁆𑀫𑀼𑀔𑀸;_  
_𑀫𑀢𑀁 𑀓𑀼𑀡𑀧𑀫𑀸𑀤𑀸𑀬,_  
_𑀯𑀺𑀮𑀧𑀦𑁆𑀢𑀻 𑀕𑀫𑀸𑀫𑀳𑀁𑁇_  


(77.)

772\. _𑀢𑀤𑀸 𑀏𑀓𑁂𑀦 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀸,_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸𑀪𑀺𑀲𑀓𑁆𑀓𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀅𑀯𑁄𑀘𑀁 ‘𑀤𑁂𑀳𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀁,_  
_𑀧𑀼𑀢𑁆𑀢𑀲𑀜𑁆𑀚𑀻𑀯𑀦𑀦𑁆𑀢𑀺 𑀪𑁄’𑁇_  


(78.)

773\. _‘𑀦 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑁂 𑀫𑀢𑀸 𑀬𑀲𑁆𑀫𑀺𑀁,_  
_𑀕𑁂𑀳𑁂 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀓𑀁 𑀢𑀢𑁄;_  
_𑀆𑀳𑀭𑀸’𑀢𑀺 𑀚𑀺𑀦𑁄 𑀆𑀳,_  
_𑀯𑀺𑀦𑀬𑁄𑀧𑀸𑀬𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


(79.)

774\. _𑀢𑀤𑀸 𑀕𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁,_  
_𑀦 𑀮𑀪𑀺𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀖𑀭𑀁;_  
_𑀓𑀼𑀢𑁄 𑀲𑀺𑀤𑁆𑀥𑀢𑁆𑀣𑀓𑀁 𑀢𑀲𑁆𑀫𑀸,_  
_𑀢𑀢𑁄 𑀮𑀤𑁆𑀥𑀸 𑀲𑀢𑀺𑀁 𑀅𑀳𑀁𑁇_  


(80.)

775\. _𑀓𑀼𑀡𑀧𑀁 𑀙𑀟𑁆𑀟𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀧𑁂𑀲𑀺𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀤𑀽𑀭𑀢𑁄𑀯 𑀫𑀫𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀯𑁄𑀘 𑀫𑀥𑀼𑀭𑀲𑁆𑀲𑀭𑁄𑁇_  


(81.)

776\. _‘𑀬𑁄 𑀘 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀚𑀻𑀯𑁂,_  
_𑀅𑀧𑀲𑁆𑀲𑀁 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁;_  
_𑀏𑀓𑀸𑀳𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀧𑀲𑁆𑀲𑀢𑁄 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀁𑁇_  


(82.)

777\. _𑀦 𑀕𑀸𑀫𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀕𑀫𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁄,_  
_𑀦 𑀘𑀸𑀧𑀺𑀬𑀁 𑀏𑀓𑀓𑀼𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁄;_  
_𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲 𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲,_  
_𑀏𑀲𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀬𑀤𑀺𑀤𑀁 𑀅𑀦𑀺𑀘𑁆𑀘𑀢𑀸’𑁇_  


(83.)

778\. _𑀲𑀸𑀳𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦𑀺𑀫𑀸 𑀕𑀸𑀣𑀸,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀯𑀺𑀲𑁄𑀥𑀬𑀺𑀁;_  
_𑀢𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(84.)

779\. _𑀢𑀣𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀲𑀦𑁆𑀢𑀻,_  
_𑀬𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂;_  
_𑀦 𑀘𑀺𑀭𑁂𑀦𑁂𑀯 𑀓𑀸𑀮𑁂𑀦,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(85.)

780\. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀓𑀸𑁇_  


(86.)

781\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀯𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀲𑀺𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀮𑀸𑁇_  


(87.)

782\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(88.)

783\. _𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄𑁇_  


(89.)

784\. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀜𑀸𑀡𑀁 𑀫𑁂 𑀯𑀺𑀫𑀮𑀁 𑀲𑀼𑀤𑁆𑀥𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸𑁇_  


(90.)

785\. _𑀲𑀗𑁆𑀓𑀸𑀭𑀓𑀽𑀝𑀸 𑀆𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀲𑀸𑀦𑀸 𑀭𑀣𑀺𑀬𑀸𑀧𑀺 𑀘;_  
_𑀢𑀢𑁄 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀓𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀮𑀽𑀔𑀁 𑀥𑀸𑀭𑁂𑀫𑀺 𑀘𑀻𑀯𑀭𑀁𑁇_  


(91.)

786\. _𑀚𑀺𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀼𑀡𑁂 𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑁂;_  
_𑀞𑀧𑁂𑀲𑀺 𑀏𑀢𑀤𑀕𑁆𑀕𑀫𑁆𑀳𑀺,_  
_𑀧𑀭𑀺𑀲𑀸𑀲𑀼 𑀯𑀺𑀦𑀸𑀬𑀓𑁄𑁇_  


(92.)

787\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(93.)

788\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(94.)

789\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


790\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

791\. 𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
