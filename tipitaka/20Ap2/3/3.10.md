# 𑁩𑁇𑁧𑁦 𑀬𑀲𑁄𑀥𑀭𑀸𑀧𑀫𑀼𑀔𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀻𑀯𑀕𑁆𑀕

* 𑀬𑀲𑁄𑀥𑀭𑀸𑀧𑀫𑀼𑀔𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀅𑀧𑀤𑀸𑀦

(426.)

1139\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀓𑁆𑀬𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀬𑀲𑁄𑀥𑀭𑀸𑀧𑀫𑀼𑀔𑀸𑀦𑀺,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀼𑀁𑁇_  


(427.)

1140\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸;_  
_𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀻 𑀫𑀼𑀦𑀺𑀦𑁄 𑀧𑀸𑀤𑁂,_  
_𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺 𑀬𑀣𑀸𑀩𑀮𑀁𑁇_  


(428.)

1141\. _“𑀚𑀸𑀢𑀺 𑀔𑀻𑀡𑀸 𑀚𑀭𑀸 𑀩𑁆𑀬𑀸𑀥𑀺,_  
_𑀫𑀭𑀡𑀜𑁆𑀘 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀅𑀦𑀸𑀲𑀯𑀁 𑀧𑀤𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀅𑀫𑀢𑀁 𑀬𑀸𑀫 𑀦𑀸𑀬𑀓𑁇_  


(429.)

1142\. _𑀔𑀮𑀺𑀢𑀜𑁆𑀘𑁂 𑀧𑀼𑀭𑁂 𑀅𑀢𑁆𑀣𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀫𑁆𑀧𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀅𑀧𑀭𑀸𑀥𑀫𑀚𑀸𑀦𑀦𑁆𑀢𑀻,_  
_𑀔𑀫 𑀅𑀫𑁆𑀳𑀁 𑀯𑀺𑀦𑀸𑀬𑀓”𑁇_  


(430.)

1143\. _“𑀇𑀤𑁆𑀥𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀦𑀺𑀤𑀲𑁆𑀲𑁂𑀣,_  
_𑀫𑀫 𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀓𑀸;_  
_𑀧𑀭𑀺𑀲𑀸𑀦𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀸𑀲𑀁,_  
_𑀓𑀗𑁆𑀔𑀁 𑀙𑀺𑀦𑁆𑀤𑀣 𑀬𑀸𑀯𑀢𑀸”𑁇_  


(431.)

1144\. _“𑀬𑀲𑁄𑀥𑀭𑀸 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀫𑀦𑀸𑀧𑀸 𑀧𑀺𑀬𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀲𑀩𑁆𑀩𑀸 𑀢𑀼𑀬𑁆𑀳𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀚𑀸𑀧𑀢𑀺𑁇_  


(432.)

1145\. _𑀣𑀻𑀦𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀦𑀯𑀼𑀢𑀻𑀦𑀁 𑀙𑀤𑀼𑀢𑁆𑀢𑀭𑀺;_  
_𑀅𑀕𑀸𑀭𑁂 𑀢𑁂 𑀫𑀬𑀁 𑀯𑀻𑀭,_  
_𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑀸 𑀲𑀩𑁆𑀩𑀸 𑀇𑀲𑁆𑀲𑀭𑀸𑁇_  


(433.)

1146\. _𑀭𑀽𑀧𑀸𑀘𑀸𑀭𑀕𑀼𑀡𑀽𑀧𑁂𑀢𑀸,_  
_𑀬𑁄𑀩𑁆𑀩𑀦𑀝𑁆𑀞𑀸 𑀧𑀺𑀬𑀁𑀯𑀤𑀸;_  
_𑀲𑀩𑁆𑀩𑀸 𑀦𑁄 𑀅𑀧𑀘𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀯𑀢𑀸 𑀯𑀺𑀬 𑀫𑀸𑀦𑀼𑀲𑀸𑁇_  


(434.)

1147\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸 𑀲𑀸𑀓𑀺𑀬𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀬𑀲𑁄𑀥𑀭𑀸𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑀸 𑀇𑀲𑁆𑀲𑀭𑀸 𑀢𑀤𑀸𑁇_  


(435.)

1148\. _𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫,_  
_𑀲𑀡𑁆𑀞𑀺𑀢𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀭𑀽𑀧𑁂𑀦 𑀲𑀤𑀺𑀲𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑀺”𑁇_  


(436.)

1149\. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀇𑀤𑁆𑀥𑀺𑀁 𑀤𑀲𑁆𑀲𑀁𑀲𑀼 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀦𑁂𑀓𑀸 𑀦𑀸𑀦𑀸𑀯𑀺𑀥𑀸𑀓𑀸𑀭𑀸,_  
_𑀫𑀳𑀸𑀇𑀤𑁆𑀥𑀻𑀧𑀺 𑀤𑀲𑁆𑀲𑀬𑀼𑀁𑁇_  


(437.)

1150\. _𑀘𑀓𑁆𑀓𑀯𑀸𑀍𑀅𑀲𑀫𑀁 𑀓𑀸𑀬𑀁,_  
_𑀲𑀻𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀢𑁄 𑀓𑀼𑀭𑀼;_  
_𑀉𑀪𑁄 𑀧𑀓𑁆𑀔𑀸 𑀤𑀼𑀯𑁂 𑀤𑀻𑀧𑀸,_  
_𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑀁 𑀲𑀭𑀻𑀭𑀢𑁄𑁇_  


(438.)

1151\. _𑀤𑀓𑁆𑀔𑀺𑀡𑀜𑁆𑀘 𑀲𑀭𑀁 𑀧𑀺𑀜𑁆𑀙𑀁,_  
_𑀦𑀸𑀦𑀸𑀲𑀸𑀔𑀸 𑀢𑀼 𑀧𑀢𑁆𑀢𑀓𑀸;_  
_𑀘𑀦𑁆𑀤𑀜𑁆𑀘 𑀲𑀽𑀭𑀺𑀬𑀜𑁆𑀘𑀓𑁆𑀔𑀺,_  
_𑀫𑁂𑀭𑀼𑀧𑀩𑁆𑀩𑀢𑀢𑁄 𑀲𑀺𑀔𑀁𑁇_  


(439.)

1152\. _𑀘𑀓𑁆𑀓𑀯𑀸𑀍𑀅𑀕𑀺𑀭𑀺𑀁 𑀢𑀼𑀡𑁆𑀟𑀁,_  
_𑀚𑀫𑁆𑀩𑀼𑀭𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀽𑀮𑀓𑀁;_  
_𑀩𑀻𑀚𑀫𑀸𑀦𑀸 𑀉𑀧𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀻 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(440.)

1153\. _𑀳𑀢𑁆𑀣𑀺𑀯𑀡𑁆𑀡𑀁 𑀢𑀣𑁂𑀯𑀲𑁆𑀲𑀁,_  
_𑀧𑀩𑁆𑀩𑀢𑀁 𑀚𑀮𑀥𑀺𑀁 𑀢𑀣𑀸;_  
_𑀘𑀦𑁆𑀤𑀜𑁆𑀘 𑀲𑀽𑀭𑀺𑀬𑀁 𑀫𑁂𑀭𑀼𑀁,_  
_𑀲𑀓𑁆𑀓𑀯𑀡𑁆𑀡𑀜𑁆𑀘 𑀤𑀲𑁆𑀲𑀬𑀼𑀁𑁇_  


(441.)

1154\. _“𑀬𑀲𑁄𑀥𑀭𑀸 𑀫𑀬𑀁 𑀯𑀻𑀭,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀫 𑀘𑀓𑁆𑀔𑀼𑀫;_  
_𑀢𑀯 𑀘𑀺𑀭𑀧𑀪𑀸𑀯𑁂𑀦,_  
_𑀦𑀺𑀧𑁆𑀨𑀦𑁆𑀦𑀸 𑀦𑀭𑀦𑀸𑀬𑀓𑁇_  


(442.)

1155\. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀬𑀜𑀸𑀡𑀲𑁆𑀲,_  
_𑀯𑀲𑀻 𑀳𑁄𑀫 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(443.)

1156\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑀸𑀲𑀯𑀸 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀸,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑁇_  


(444.)

1157\. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀜𑀸𑀡𑀁 𑀅𑀫𑁆𑀳𑀁 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀢𑀯 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(445.)

1158\. _𑀧𑀼𑀩𑁆𑀩𑀸𑀦𑀁 𑀮𑁄𑀓𑀦𑀸𑀣𑀸𑀦𑀁,_  
_𑀲𑀗𑁆𑀕𑀫𑀁 𑀦𑁄 𑀦𑀺𑀤𑀲𑁆𑀲𑀺𑀢𑀁;_  
_𑀅𑀥𑀺𑀓𑀸𑀭𑀸 𑀩𑀳𑀽 𑀅𑀫𑁆𑀳𑀁,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(446.)

1159\. _𑀬𑀁 𑀅𑀫𑁆𑀳𑀁 𑀧𑀽𑀭𑀺𑀢𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀼𑀲𑀮𑀁 𑀲𑀭𑀲𑁂 𑀫𑀼𑀦𑁂;_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀼𑀧𑀘𑀺𑀢𑀸𑀦𑀺 𑀦𑁄𑁇_  


(447.)

1160\. _𑀅𑀪𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦𑁂 𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀸𑀭𑀬𑀺𑀫𑁆𑀳 𑀅𑀦𑀸𑀘𑀭𑀁;_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀘𑀢𑁆𑀢𑀸𑀦𑀺 𑀚𑀻𑀯𑀺𑀢𑀸𑀦𑀺 𑀦𑁄𑁇_  


(448.)

1161\. _𑀦𑁂𑀓𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀪𑀭𑀺𑀬𑀢𑁆𑀣𑀸𑀬𑀤𑀸𑀲𑀺 𑀦𑁄;_  
_𑀦 𑀢𑀢𑁆𑀣 𑀯𑀺𑀫𑀦𑀸 𑀳𑁄𑀫,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(449.)

1162\. _𑀦𑁂𑀓𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀉𑀧𑀓𑀸𑀭𑀸𑀬𑀤𑀸𑀲𑀺 𑀦𑁄;_  
_𑀦 𑀢𑀢𑁆𑀣 𑀯𑀺𑀫𑀦𑀸 𑀳𑁄𑀫,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(450.)

1163\. _𑀦𑁂𑀓𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀪𑁄𑀚𑀦𑀢𑁆𑀣𑀸𑀬𑀤𑀸𑀲𑀺 𑀦𑁄;_  
_𑀦 𑀢𑀢𑁆𑀣 𑀯𑀺𑀫𑀦𑀸 𑀳𑁄𑀫,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(451.)

1164\. _𑀦𑁂𑀓𑀓𑁄𑀝𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀚𑀻𑀯𑀺𑀢𑀸𑀦𑀺 𑀘𑀚𑀺𑀫𑁆𑀳𑀲𑁂;_  
_𑀪𑀬𑀫𑁄𑀓𑁆𑀔𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀚𑀻𑀯𑀺𑀢𑀸𑀦𑀺 𑀘𑀚𑀺𑀫𑁆𑀳𑀲𑁂𑁇_  


(452.)

1165\. _𑀅𑀗𑁆𑀕𑀕𑀢𑁂 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑁂,_  
_𑀯𑀢𑁆𑀣𑁂 𑀦𑀸𑀦𑀸𑀯𑀺𑀥𑁂 𑀩𑀳𑀽;_  
_𑀇𑀢𑁆𑀣𑀺𑀪𑀡𑁆𑀟𑁂 𑀦 𑀕𑀽𑀳𑀸𑀫,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(453.)

1166\. _𑀥𑀦𑀥𑀜𑁆𑀜𑀧𑀭𑀺𑀘𑁆𑀘𑀸𑀕𑀁,_  
_𑀕𑀸𑀫𑀸𑀦𑀺 𑀦𑀺𑀕𑀫𑀸𑀦𑀺 𑀘;_  
_𑀔𑁂𑀢𑁆𑀢𑀁 𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀥𑀻𑀢𑀸 𑀘,_  
_𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀸 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(454.)

1167\. _𑀳𑀢𑁆𑀣𑀻 𑀅𑀲𑁆𑀲𑀸 𑀕𑀯𑀸 𑀘𑀸𑀧𑀺,_  
_𑀤𑀸𑀲𑀺𑀬𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀓𑀸;_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(455.)

1168\. _𑀬𑀁 𑀅𑀫𑁆𑀳𑁂 𑀧𑀝𑀺𑀫𑀦𑁆𑀢𑁂𑀲𑀺,_  
_‘𑀤𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀸𑀫 𑀬𑀸𑀘𑀓𑁂’;_  
_𑀯𑀺𑀫𑀦𑀁 𑀦𑁄 𑀦 𑀧𑀲𑁆𑀲𑀸𑀫,_  
_𑀤𑀤𑀢𑁄 𑀤𑀸𑀦𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(456.)

1169\. _𑀦𑀸𑀦𑀸𑀯𑀺𑀥𑀁 𑀩𑀳𑀼𑀁 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀲𑀁𑀲𑀸𑀭𑁂 𑀘 𑀩𑀳𑀼𑀩𑁆𑀩𑀺𑀥𑁂;_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀅𑀦𑀼𑀪𑀼𑀢𑁆𑀢𑀁 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(457.)

1170\. _𑀲𑀼𑀔𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀼𑀫𑁄𑀤𑀸𑀫,_  
_𑀦 𑀘 𑀤𑀼𑀓𑁆𑀔𑁂𑀲𑀼 𑀤𑀼𑀫𑁆𑀫𑀦𑀸;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀢𑀼𑀮𑀺𑀢𑀸 𑀳𑁄𑀫,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(458.)

1171\. _𑀅𑀦𑀼𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀪𑀺𑀦𑀻𑀳𑀭𑀺;_  
_𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸 𑀲𑀼𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀧𑀢𑁆𑀢𑁄 𑀩𑁄𑀥𑀺𑀁 𑀫𑀳𑀸𑀫𑀼𑀦𑁂𑁇_  


(459.)

1172\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑁂𑀯𑀜𑁆𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀕𑁄𑀢𑀫𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀮𑁄𑀓𑀦𑀸𑀣𑀸𑀦𑀁,_  
_𑀲𑀗𑁆𑀕𑀫𑀁 𑀢𑁂𑀳𑀺 𑀦𑁄 𑀩𑀳𑀽𑁇_  


(460.)

1173\. _𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀩𑀳𑀼𑀁 𑀅𑀫𑁆𑀳𑁂,_  
_𑀢𑀼𑀬𑁆𑀳𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑀸𑀫𑀼𑀦𑁂;_  
_𑀕𑀯𑁂𑀲𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑁂,_  
_𑀫𑀬𑀁 𑀢𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀓𑀸𑁇_  


(461.)

1174\. _𑀓𑀧𑁆𑀧𑁂 𑀘 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂,_  
_𑀘𑀢𑀼𑀭𑁄 𑀘 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑁂;_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑁄 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄𑁇_  


(462.)

1175\. _𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀤𑁂𑀲𑀯𑀺𑀲𑀬𑁂,_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀁;_  
_𑀢𑀲𑁆𑀲 𑀆𑀕𑀫𑀦𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀲𑁄𑀥𑁂𑀦𑁆𑀢𑀺 𑀢𑀼𑀝𑁆𑀞𑀫𑀸𑀦𑀲𑀸𑁇_  


(463.)

1176\. _𑀢𑁂𑀦 𑀓𑀸𑀮𑁂𑀦 𑀲𑁄 𑀆𑀲𑀺,_  
_𑀲𑀼𑀫𑁂𑀥𑁄 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀲𑀺,_  
_𑀆𑀬𑀢𑁄 𑀲𑀩𑁆𑀩𑀤𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


(464.)

1177\. _𑀢𑁂𑀦 𑀓𑀸𑀮𑁂𑀦 𑀅𑀳𑀼𑀫𑁆𑀳,_  
_𑀲𑀩𑁆𑀩𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀫𑁆𑀪𑀯𑀸;_  
_𑀣𑀮𑀽𑀤𑀚𑀸𑀦𑀺 𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀆𑀳𑀭𑀺𑀫𑁆𑀳 𑀲𑀫𑀸𑀕𑀫𑀁𑁇_  


(465.)

1178\. _𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑁄 𑀲𑀫𑀬𑁂 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀤𑀻𑀧𑀗𑁆𑀓𑀭𑁄 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀇𑀲𑀺𑀫𑀼𑀕𑁆𑀕𑀢𑀫𑀸𑀦𑀲𑀁𑁇_  


(466.)

1179\. _𑀘𑀮𑀢𑀻 𑀭𑀯𑀢𑀻 𑀧𑀼𑀣𑀯𑀻,_  
_𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁂;_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀺𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁂,_  
_𑀇𑀲𑀺𑀫𑀼𑀕𑁆𑀕𑀢𑀫𑀸𑀦𑀲𑀁𑁇_  


(467.)

1180\. _𑀤𑁂𑀯𑀓𑀜𑁆𑀜𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘,_  
_𑀫𑀬𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀤𑁂𑀯𑀓𑀸;_  
_𑀦𑀸𑀦𑀸𑀧𑀽𑀚𑀦𑀻𑀬𑀁 𑀪𑀡𑁆𑀟𑀁,_  
_𑀧𑀽𑀚𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀢𑁆𑀣𑀬𑀼𑀁𑁇_  


(468.)

1181\. _𑀢𑁂𑀲𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀸𑀲𑀺,_  
_𑀚𑁄𑀢𑀺𑀤𑀻𑀧 𑀲𑀦𑀸𑀫𑀓𑁄;_  
_‘𑀅𑀚𑁆𑀚 𑀬𑁂 𑀧𑀢𑁆𑀣𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀺,_  
_𑀢𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸’𑁇_  


(469.)

1182\. _𑀅𑀧𑀭𑀺𑀫𑁂𑀬𑁆𑀬𑁂 𑀇𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀬𑀁 𑀦𑁄 𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀬𑀸𑀓𑀭𑀺;_  
_𑀢𑀁 𑀯𑀸𑀘𑀫𑀦𑀼𑀫𑁄𑀤𑁂𑀦𑁆𑀢𑀸,_  
_𑀏𑀯𑀁𑀓𑀸𑀭𑀻 𑀅𑀳𑀼𑀫𑁆𑀳 𑀦𑁄𑁇_  


(470.)

1183\. _𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀼𑀓𑀢𑀲𑁆𑀲,_  
_𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑀬𑀼𑀁;_  
_𑀤𑁂𑀯𑀫𑀸𑀦𑀼𑀲𑀺𑀓𑀁 𑀬𑁄𑀦𑀺𑀁,_  
_𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸 𑀅𑀲𑀗𑁆𑀔𑀺𑀬𑀁𑁇_  


(471.)

1184\. _𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑁂𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑁂𑀯𑁂𑀲𑀼 𑀫𑀸𑀦𑀼𑀲𑁂𑀲𑀼 𑀘;_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀚𑀸𑀢𑀸𑀫𑁆𑀳 𑀲𑀸𑀓𑀺𑀬𑁂 𑀓𑀼𑀮𑁂𑁇_  


(472.)

1185\. _𑀭𑀽𑀧𑀯𑀢𑀻 𑀪𑁄𑀕𑀯𑀢𑀻,_  
_𑀬𑀲𑀲𑀻𑀮𑀯𑀢𑀻 𑀢𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀤𑀸 𑀳𑁄𑀫,_  
_𑀓𑀼𑀮𑁂𑀲𑀼 𑀅𑀪𑀺𑀲𑀓𑁆𑀓𑀢𑀸𑁇_  


(473.)

1186\. _𑀮𑀸𑀪𑀁 𑀲𑀺𑀮𑁄𑀓𑀁 𑀲𑀓𑁆𑀓𑀸𑀭𑀁,_  
_𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀲𑀫𑀸𑀕𑀫𑀁;_  
_𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀯𑀲𑀸𑀫 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑀸𑁇_  


(474.)

1187\. _𑀯𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑁂𑀢𑀁 𑀪𑀕𑀯𑀢𑀸,_  
_‘𑀭𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀢𑀤𑀸;_  
_𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀧𑀼𑀭𑁂 𑀯𑀻𑀭,_  
_𑀉𑀧𑀓𑀸𑀭𑀜𑁆𑀘 𑀦𑀺𑀤𑁆𑀤𑀺𑀲𑀺𑁇_  


(475.)

1188\. _𑀉𑀧𑀓𑀸𑀭𑀸 𑀘 𑀬𑀸 𑀦𑀸𑀭𑀻,_  
_𑀬𑀸 𑀘 𑀦𑀸𑀭𑀻 𑀲𑀼𑀔𑁂 𑀤𑀼𑀔𑁂;_  
_𑀅𑀢𑁆𑀣𑀓𑁆𑀔𑀸𑀬𑀻 𑀘 𑀬𑀸 𑀦𑀸𑀭𑀻,_  
_𑀬𑀸 𑀘 𑀦𑀸𑀭𑀻𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸𑁇_  


(476.)

1189\. _𑀥𑀫𑁆𑀫𑀁 𑀘𑀭𑁂 𑀲𑀼𑀘𑀭𑀺𑀢𑀁,_  
_𑀦 𑀦𑀁 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀘𑀭𑁂;_  
_𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀲𑀼𑀔𑀁 𑀲𑁂𑀢𑀺,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀫𑁆𑀳𑀺 𑀘’𑁇_  


(477.)

1190\. _𑀅𑀕𑀸𑀭𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀫𑁆𑀳𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀅𑀟𑁆𑀠𑀫𑀸𑀲𑁂 𑀅𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀘𑀢𑀼𑀲𑀘𑁆𑀘𑀁 𑀨𑀼𑀲𑀺𑀫𑁆𑀳 𑀦𑁄𑁇_  


(478.)

1191\. _𑀘𑀻𑀯𑀭𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀧𑀘𑁆𑀘𑀬𑀁 𑀲𑀬𑀦𑀸𑀲𑀦𑀁;_  
_𑀉𑀧𑀦𑁂𑀦𑁆𑀢𑀺 𑀩𑀳𑀽 𑀅𑀫𑁆𑀳𑁂,_  
_𑀲𑀸𑀕𑀭𑀲𑁆𑀲𑁂𑀯 𑀊𑀫𑀺𑀬𑁄𑁇_  


(479.)

1192\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀅𑀫𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑀻𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(480.)

1193\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀦𑁄 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(481.)

1194\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(482.)

1195\. _𑀏𑀯𑀁 𑀩𑀳𑀼𑀯𑀺𑀥𑀁 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻 𑀘 𑀩𑀳𑀼𑀩𑁆𑀩𑀺𑀥𑀸;_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀪𑀸𑀯𑀁 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀮𑀪𑀸𑀫 𑀲𑀩𑁆𑀩𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


(483.)

1196\. _𑀬𑀸 𑀤𑀤𑀦𑁆𑀢𑀺 𑀲𑀓𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀢𑁆𑀣𑀸𑀬 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀲𑀳𑀸𑀬𑀲𑀫𑁆𑀧𑀤𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀤𑀫𑀲𑀗𑁆𑀔𑀢𑀁𑁇_  


(484.)

1197\. _𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁 𑀅𑀢𑀻𑀢𑀜𑁆𑀘,_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀦𑀸𑀕𑀢𑀁;_  
_𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀫𑁆𑀧𑀺 𑀦𑁄 𑀔𑀻𑀡𑀁,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀫 𑀘𑀓𑁆𑀔𑀼𑀫”𑁇_  


(485.)

1198\. _“𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀬 𑀯𑀤𑀦𑁆𑀢𑀻𑀦𑀁,_  
_𑀓𑀺𑀁 𑀯𑁄 𑀯𑀓𑁆𑀔𑀸𑀫 𑀉𑀢𑁆𑀢𑀭𑀺;_  
_𑀲𑀦𑁆𑀢𑀲𑀗𑁆𑀔𑀢𑀤𑁄𑀲𑀜𑁆𑀳𑀺,_  
_𑀧𑀧𑁆𑀧𑁄𑀣 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁”𑁇 (𑁧𑁧𑁨𑁬)_  


1199\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀬𑀲𑁄𑀥𑀭𑀸𑀧𑀫𑀼𑀔𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

1200\. 𑀬𑀲𑁄𑀥𑀭𑀸𑀧𑀫𑀼𑀔𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀲𑀳𑀲𑁆𑀲𑀸𑀧𑀤𑀸𑀦𑀁 𑀤𑀲𑀫𑀁𑁇



---

1201\. 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀻𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

1202\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1203\. _𑀓𑀼𑀡𑁆𑀟𑀮𑀸 𑀕𑁄𑀢𑀫𑀻 𑀘𑁂𑀯,_  
_𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸 𑀘 𑀲𑀓𑀼𑀮𑀸;_  
_𑀯𑀭𑀦𑀦𑁆𑀤𑀸 𑀘 𑀲𑁄𑀡𑀸 𑀘,_  
_𑀓𑀸𑀧𑀺𑀮𑀸𑀦𑀻 𑀬𑀲𑁄𑀥𑀭𑀸𑁇_  


1204\. _𑀤𑀲𑀲𑀳𑀲𑁆𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻,_  
_𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀲𑀳𑀲𑁆𑀲𑀓𑀸;_  
_𑀕𑀸𑀣𑀸𑀲𑀢𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺,_  
_𑀙 𑀘 𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑁂𑀯 𑀘𑁇_
