# 𑁩𑁇𑁧 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀻𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(1.)

694\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(2.)

695\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀚𑀸𑀢𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑁂 𑀅𑀳𑀼𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀧𑀚𑁆𑀚𑁄𑀢𑁂,_  
_𑀫𑀳𑀸𑀲𑀼𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(3.)

696\. _𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁;_  
_𑀢𑀢𑁄 𑀚𑀸𑀢𑀧𑁆𑀧𑀲𑀸𑀤𑀸𑀳𑀁,_  
_𑀉𑀧𑁂𑀲𑀺𑀁 𑀲𑀭𑀡𑀁 𑀚𑀺𑀦𑀁𑁇_  


(4.)

697\. _𑀢𑀤𑀸 𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀦𑀸𑀫𑀓𑁄;_  
_𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀦𑀫𑀕𑁆𑀕𑀦𑁆𑀢𑀺,_  
_𑀞𑀧𑁂𑀲𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀲𑀼𑀪𑀁𑁇_  


(5.)

698\. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀫𑀳𑁂𑀲𑀺𑀦𑁄;_  
_𑀦𑀺𑀧𑀘𑁆𑀘 𑀲𑀺𑀭𑀲𑀸 𑀧𑀸𑀤𑁂,_  
_𑀢𑀁 𑀞𑀸𑀦𑀫𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀺𑀁𑁇_  


(6.)

699\. _𑀅𑀦𑀼𑀫𑁄𑀤𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_‘𑀪𑀤𑁆𑀤𑁂 𑀬𑀁 𑀢𑁂𑀪𑀺𑀧𑀢𑁆𑀣𑀺𑀢𑀁;_  
_𑀲𑀫𑀺𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀲𑀼𑀔𑀺𑀦𑀻 𑀳𑁄𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸𑁇_  


(7.)

700\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(8.)

701\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑀸,_  
_𑀑𑀭𑀲𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀪𑀤𑁆𑀤𑀸𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀸𑀢𑀺,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀺𑀓𑀸’𑁇_  


(9.)

702\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(10.)

703\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀬𑀸𑀫𑀫𑀕𑀁,_  
_𑀢𑀢𑁄𑀳𑀁 𑀢𑀼𑀲𑀺𑀢𑀁 𑀕𑀢𑀸;_  
_𑀢𑀢𑁄 𑀘 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀺𑀁,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀧𑀼𑀭𑀁 𑀢𑀢𑁄𑁇_  


(11.)

704\. _𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣𑀽𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸;_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣𑁂𑀯 𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(12.)

705\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼,_  
_𑀭𑀸𑀚𑀽𑀦𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑀁;_  
_𑀫𑀡𑁆𑀟𑀮𑀻𑀦𑀜𑁆𑀘 𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(13.)

706\. _𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑁂𑀯𑁂𑀲𑀼 𑀫𑀸𑀦𑀼𑀲𑁂𑀲𑀼 𑀘;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀦𑁂𑀓𑀓𑀧𑁆𑀧𑁂𑀲𑀼 𑀲𑀁𑀲𑀭𑀺𑀁𑁇_  


(14.)

707\. _𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(15.)

708\. _𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀫𑀳𑁂𑀲𑀺𑀲𑁆𑀲,_  
_𑀢𑀤𑀸 𑀆𑀲𑀺 𑀦𑀭𑀺𑀲𑁆𑀲𑀭𑁄;_  
_𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀓𑀺𑀓𑀻 𑀦𑀸𑀫,_  
_𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂𑁇_  


(16.)

709\. _𑀢𑀲𑁆𑀲 𑀥𑀻𑀢𑀸 𑀘𑀢𑀼𑀢𑁆𑀣𑀸𑀲𑀺𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀤𑀸𑀬𑀻𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀚𑀺𑀦𑀕𑁆𑀕𑀲𑁆𑀲,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀲𑀫𑀭𑁄𑀘𑀬𑀺𑀁𑁇_  


(17.)

710\. _𑀅𑀦𑀼𑀚𑀸𑀦𑀺 𑀦 𑀦𑁄 𑀢𑀸𑀢𑁄,_  
_𑀅𑀕𑀸𑀭𑁂𑀯 𑀢𑀤𑀸 𑀫𑀬𑀁;_  
_𑀯𑀻𑀲𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀯𑀺𑀘𑀭𑀺𑀫𑁆𑀳 𑀅𑀢𑀦𑁆𑀤𑀺𑀢𑀸𑁇_  


(18.)

711\. _𑀓𑁄𑀫𑀸𑀭𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸;_  
_𑀩𑀼𑀤𑁆𑀥𑁄𑀧𑀝𑁆𑀞𑀸𑀦𑀦𑀺𑀭𑀢𑀸,_  
_𑀫𑀼𑀤𑀺𑀢𑀸 𑀲𑀢𑁆𑀢 𑀥𑀻𑀢𑀭𑁄𑁇_  


(19.)

712\. _𑀲𑀫𑀡𑀻 𑀲𑀫𑀡𑀕𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀤𑀸𑀬𑀺𑀓𑀸;_  
_𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀲𑀼𑀥𑀫𑁆𑀫𑀸 𑀘,_  
_𑀲𑀢𑁆𑀢𑀫𑀻 𑀲𑀁𑀖𑀤𑀸𑀬𑀺𑀓𑀸𑁇_  


(20.)

713\. _𑀔𑁂𑀫𑀸 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀘,_  
_𑀧𑀝𑀸𑀘𑀸𑀭𑀸 𑀅𑀳𑀁 𑀢𑀤𑀸;_  
_𑀓𑀺𑀲𑀸𑀕𑁄𑀢𑀫𑀻 𑀥𑀫𑁆𑀫𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀯𑀺𑀲𑀸𑀔𑀸 𑀳𑁄𑀢𑀺 𑀲𑀢𑁆𑀢𑀫𑀻𑁇_  


(21.)

714\. _𑀢𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀼𑀓𑀢𑁂𑀳𑀺,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(22.)

715\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑁂;_  
_𑀚𑀸𑀢𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑁂 𑀨𑀻𑀢𑁂,_  
_𑀬𑀤𑀸𑀳𑀁 𑀬𑁄𑀩𑁆𑀩𑀦𑁂 𑀞𑀺𑀢𑀸𑁇_  


(23.)

716\. _𑀘𑁄𑀭𑀁 𑀯𑀥𑀢𑁆𑀣𑀁 𑀦𑀻𑀬𑀦𑁆𑀢𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀭𑀢𑁆𑀢𑀸 𑀢𑀳𑀺𑀁 𑀅𑀳𑀁;_  
_𑀧𑀺𑀢𑀸 𑀫𑁂 𑀢𑀁 𑀲𑀳𑀲𑁆𑀲𑁂𑀦,_  
_𑀫𑁄𑀘𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀸 𑀢𑀢𑁄𑁇_  


(24.)

717\. _𑀅𑀤𑀸𑀲𑀺 𑀢𑀲𑁆𑀲 𑀫𑀁 𑀢𑀸𑀢𑁄,_  
_𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀫𑀦𑀁 𑀫𑀫;_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀫𑀸𑀲𑀺𑀁 𑀯𑀺𑀲𑀝𑁆𑀞𑀸,_  
_𑀅𑀢𑀻𑀯 𑀤𑀬𑀺𑀢𑀸 𑀳𑀺𑀢𑀸𑁇_  


(25.)

718\. _𑀲𑁄 𑀫𑁂 𑀪𑀽𑀲𑀦𑀮𑁄𑀪𑁂𑀦,_  
_𑀩𑀮𑀺𑀫𑀚𑁆𑀛𑀸𑀲𑀬𑁄 𑀤𑀺𑀲𑁄;_  
_𑀘𑁄𑀭𑀧𑁆𑀧𑀧𑀸𑀢𑀁 𑀦𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀢𑀁 𑀘𑁂𑀢𑀬𑀻 𑀯𑀥𑀁𑁇_  


(26.)

719\. _𑀢𑀤𑀸𑀳𑀁 𑀧𑀡𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀢𑁆𑀢𑀼𑀓𑀁 𑀲𑀼𑀓𑀢𑀜𑁆𑀚𑀮𑀻;_  
_𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀻 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀸𑀡𑀁,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑀁𑁇_  


(27.)

720\. _‘𑀇𑀤𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑁂𑀬𑀽𑀭𑀁,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀩𑀳𑀽;_  
_𑀲𑀩𑁆𑀩𑀁 𑀳𑀭𑀲𑁆𑀲𑀼 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀫𑀜𑁆𑀘 𑀤𑀸𑀲𑀻𑀢𑀺 𑀲𑀸𑀯𑀬’𑁇_  


(28.)

721\. _‘𑀑𑀭𑁄𑀧𑀬𑀲𑁆𑀲𑀼 𑀓𑀮𑁆𑀬𑀸𑀡𑀻,_  
_𑀫𑀸 𑀩𑀸𑀍𑀳𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀲𑀺;_  
_𑀦 𑀘𑀸𑀳𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀥𑀦𑀫𑀸𑀪𑀢𑀁’𑁇_  


(29.)

722\. _‘𑀬𑀢𑁄 𑀲𑀭𑀸𑀫𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀬𑀢𑁄 𑀧𑀢𑁆𑀢𑁄𑀲𑁆𑀫𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀢𑀁;_  
_𑀦 𑀘𑀸𑀳𑀁 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀅𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀢𑀭𑀁 𑀢𑀬𑀸𑁇_  


(30.)

723\. _𑀏𑀳𑀺 𑀢𑀁 𑀉𑀧𑀕𑀽𑀳𑀺𑀲𑁆𑀲𑀁,_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀢𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀦 𑀘 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑁄 𑀅𑀢𑁆𑀣𑀺,_  
_𑀫𑀫 𑀢𑀼𑀬𑁆𑀳𑀜𑁆𑀘 𑀲𑀗𑁆𑀕𑀫𑁄’𑁇_  


(31.)

724\. _‘𑀦 𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼,_  
_𑀧𑀼𑀭𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀇𑀢𑁆𑀣𑀻𑀧𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑀸𑁇_  


(32.)

725\. _𑀦 𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼,_  
_𑀧𑀼𑀭𑀺𑀲𑁄 𑀳𑁄𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀇𑀢𑁆𑀣𑀻𑀧𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸 𑀳𑁄𑀢𑀺,_  
_𑀮𑀳𑀼𑀁 𑀅𑀢𑁆𑀣𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑀺𑀓𑀸’𑁇_  


(33.)

726\. _𑀮𑀳𑀼𑀜𑁆𑀘 𑀯𑀢 𑀔𑀺𑀧𑁆𑀧𑀜𑁆𑀘,_  
_𑀦𑀺𑀓𑀝𑁆𑀞𑁂 𑀲𑀫𑀘𑁂𑀢𑀬𑀺𑀁;_  
_𑀫𑀺𑀕𑀁 𑀉𑀡𑁆𑀡𑀸 𑀬𑀣𑀸 𑀏𑀯𑀁,_  
_𑀢𑀤𑀸𑀳𑀁 𑀲𑀢𑁆𑀢𑀼𑀓𑀁 𑀯𑀥𑀺𑀁𑁇_  


(34.)

727\. _𑀬𑁄 𑀘 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀦 𑀔𑀺𑀧𑁆𑀧𑀫𑀦𑀼𑀩𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀲𑁄 𑀳𑀜𑁆𑀜𑀢𑁂 𑀫𑀦𑁆𑀤𑀫𑀢𑀺,_  
_𑀘𑁄𑀭𑁄𑀯 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑁂𑁇_  


(35.)

728\. _𑀬𑁄 𑀘 𑀉𑀧𑁆𑀧𑀢𑀺𑀢𑀁 𑀅𑀢𑁆𑀣𑀁,_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀺𑀩𑁄𑀥𑀢𑀺;_  
_𑀫𑀼𑀘𑁆𑀘𑀢𑁂 𑀲𑀢𑁆𑀢𑀼𑀲𑀫𑁆𑀩𑀸𑀥𑀸,_  
_𑀢𑀤𑀸𑀳𑀁 𑀲𑀢𑁆𑀢𑀼𑀓𑀸 𑀬𑀣𑀸𑁇_  


(36.)

729\. _𑀢𑀤𑀸𑀳𑀁 𑀧𑀸𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀲𑀢𑁆𑀢𑀼𑀓𑀁;_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑀁 𑀲𑁂𑀢𑀯𑀢𑁆𑀣𑀸𑀦𑀁,_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀳𑀁𑁇_  


(37.)

730\. _𑀲𑀡𑁆𑀟𑀸𑀲𑁂𑀦 𑀘 𑀓𑁂𑀲𑁂 𑀫𑁂,_  
_𑀮𑀼𑀜𑁆𑀘𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀲𑁄 𑀢𑀤𑀸;_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀲𑀫𑀬𑀁,_  
_𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀁𑁇_  


(38.)

731\. _𑀢𑀢𑁄 𑀢𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀳𑀁,_  
_𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀏𑀓𑀺𑀓𑀸;_  
_𑀲𑀫𑀬𑀁 𑀢𑀁 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑁂𑀲𑀺𑀁,_  
_𑀲𑀼𑀯𑀸𑀦𑁄 𑀫𑀸𑀦𑀼𑀲𑀁 𑀓𑀭𑀁𑁇_  


(39.)

732\. _𑀙𑀺𑀦𑁆𑀦𑀁 𑀕𑀬𑁆𑀳 𑀲𑀫𑀻𑀧𑁂 𑀫𑁂,_  
_𑀧𑀸𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑀓𑁆𑀓𑀫𑀺;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀫𑀮𑀪𑀺𑀁,_  
_𑀳𑀢𑁆𑀣𑀁 𑀢𑀁 𑀧𑀼𑀍𑀅𑀯𑀸𑀓𑀼𑀮𑀁𑁇_  


(40.)

733\. _𑀢𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀸,_  
_𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑁂;_  
_𑀢𑁂 𑀅𑀯𑁄𑀘𑀼𑀁 ‘𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀅𑀢𑁆𑀣𑀁 𑀲𑀓𑁆𑀬𑀪𑀺𑀓𑁆𑀔𑀯𑁄’𑁇_  


(41.)

734\. _𑀲𑀸𑀳𑀁 𑀢𑀫𑀢𑁆𑀣𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀁,_  
_𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑁂;_  
_𑀢𑁂 𑀫𑀫𑀸𑀤𑀸𑀬 𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(42.)

735\. _𑀲𑁄 𑀫𑁂 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀔𑀦𑁆𑀥𑀸𑀬𑀢𑀦𑀥𑀸𑀢𑀼𑀬𑁄;_  
_𑀅𑀲𑀼𑀪𑀸𑀦𑀺𑀘𑁆𑀘𑀤𑀼𑀓𑁆𑀔𑀸𑀢𑀺,_  
_𑀅𑀦𑀢𑁆𑀢𑀸𑀢𑀺 𑀘 𑀦𑀸𑀬𑀓𑁄𑁇_  


(43.)

736\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸𑀳𑀁,_  
_𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀯𑀺𑀲𑁄𑀥𑀬𑀺𑀁;_  
_𑀢𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇_  


(44.)

737\. _𑀆𑀬𑀸𑀘𑀺𑀢𑁄 𑀢𑀤𑀸 𑀆𑀳,_  
_‘𑀏𑀳𑀺 𑀪𑀤𑁆𑀤𑁂’𑀢𑀺 𑀦𑀸𑀬𑀓𑁄;_  
_𑀢𑀤𑀸𑀳𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀢𑁄𑀬𑀫𑀤𑁆𑀤𑀲𑀁𑁇_  


(45.)

738\. _𑀧𑀸𑀤𑀧𑀓𑁆𑀔𑀸𑀮𑀦𑁂𑀦𑀸𑀳𑀁,_  
_𑀜𑀢𑁆𑀯𑀸 𑀲𑁅𑀤𑀬𑀩𑁆𑀩𑀬𑀁;_  
_𑀢𑀣𑀸 𑀲𑀩𑁆𑀩𑁂𑀧𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑁂,_  
_𑀈𑀤𑀺𑀲𑀁 𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀁 𑀢𑀤𑀸𑁇_  


(46.)

739\. _𑀢𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺 𑀫𑁂,_  
_𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀲𑀩𑁆𑀩𑀲𑁄;_  
_𑀔𑀺𑀧𑁆𑀧𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀫𑁂,_  
_𑀢𑀤𑀸 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀻 𑀚𑀺𑀦𑁄𑁇_  


(47.)

740\. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀓𑀸𑁇_  


(48.)

741\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀯𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀲𑀺𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀮𑀸𑁇_  


(49.)

742\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(50.)

743\. _𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄𑁇_  


(51.)

744\. _𑀅𑀢𑁆𑀣𑀥𑀫𑁆𑀫𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀻𑀲𑀼,_  
_𑀧𑀝𑀺𑀪𑀸𑀦𑁂 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀜𑀸𑀡𑀁 𑀫𑁂 𑀯𑀺𑀫𑀮𑀁 𑀲𑀼𑀤𑁆𑀥𑀁,_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


(52.)

745\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_𑀪𑀯𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀦𑀸𑀕𑀻𑀯 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(53.)

746\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_𑀫𑀫 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(54.)

747\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀧𑀺 𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑁂;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_
