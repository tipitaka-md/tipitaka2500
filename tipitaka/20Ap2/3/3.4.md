# 𑁩𑁇𑁪 𑀲𑀓𑀼𑀮𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

### 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀻𑀯𑀕𑁆𑀕

### 𑀲𑀓𑀼𑀮𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(131.)

831\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(132.)

832\. _𑀳𑀺𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀲𑀼𑀔𑀸𑀬 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄;_  
_𑀅𑀢𑁆𑀣𑀸𑀬 𑀧𑀼𑀭𑀺𑀲𑀸𑀚𑀜𑁆𑀜𑁄,_  
_𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀤𑁂𑀯𑀓𑁂𑁇_  


(133.)

833\. _𑀬𑀲𑀕𑁆𑀕𑀧𑀢𑁆𑀢𑁄 𑀲𑀺𑀭𑀺𑀫𑀸,_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀯𑀡𑁆𑀡𑀕𑀢𑁄 𑀚𑀺𑀦𑁄;_  
_𑀧𑀽𑀚𑀺𑀢𑁄 𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀤𑀺𑀲𑀸𑀲𑀩𑁆𑀩𑀸𑀲𑀼 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄𑁇_  


(134.)

834\. _𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀲𑁄,_  
_𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀓𑀣𑀁𑀓𑀣𑁄;_  
_𑀲𑀫𑁆𑀧𑀼𑀡𑁆𑀡𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀧𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(135.)

835\. _𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲,_  
_𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀸 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄;_  
_𑀅𑀦𑀓𑁆𑀔𑀸𑀢𑀜𑁆𑀘 𑀅𑀓𑁆𑀔𑀸𑀲𑀺,_  
_𑀅𑀲𑀜𑁆𑀚𑀸𑀢𑀜𑁆𑀘 𑀲𑀜𑁆𑀚𑀦𑀻𑁇_  


(136.)

836\. _𑀫𑀕𑁆𑀕𑀜𑁆𑀜𑀽 𑀘 𑀫𑀕𑁆𑀕𑀯𑀺𑀤𑀽,_  
_𑀫𑀕𑁆𑀕𑀓𑁆𑀔𑀸𑀬𑀻 𑀦𑀭𑀸𑀲𑀪𑁄;_  
_𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀲𑀸𑀭𑀣𑀻𑀦𑀁 𑀯𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(137.)

837\. _𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄 𑀲𑀢𑁆𑀣𑀸,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺 𑀦𑀸𑀬𑀓𑁄;_  
_𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑁂 𑀓𑀸𑀫𑀧𑀗𑁆𑀓𑀫𑁆𑀳𑀺,_  
_𑀲𑀫𑀼𑀤𑁆𑀥𑀭𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑁂𑁇_  


(138.)

838\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀚𑀸𑀢𑀸 𑀔𑀢𑁆𑀢𑀺𑀬𑀦𑀦𑁆𑀤𑀦𑀸;_  
_𑀲𑀼𑀭𑀽𑀧𑀸 𑀲𑀥𑀦𑀸 𑀘𑀸𑀧𑀺,_  
_𑀤𑀬𑀺𑀢𑀸 𑀘 𑀲𑀺𑀭𑀻𑀫𑀢𑀻𑁇_  


(139.)

839\. _𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀳𑀸𑀭𑀜𑁆𑀜𑁄,_  
_𑀥𑀻𑀢𑀸 𑀧𑀭𑀫𑀲𑁄𑀪𑀡𑀸;_  
_𑀯𑁂𑀫𑀸𑀢𑀸 𑀪𑀕𑀺𑀦𑀻 𑀘𑀸𑀧𑀺,_  
_𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀦𑀸𑀫𑀺𑀦𑁄𑁇_  


(140.)

840\. _𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀳𑀺 𑀲𑀳𑀺𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀭𑀡𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀉𑀧𑀸𑀕𑀫𑁆𑀫 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁𑁇_  


(141.)

841\. _𑀢𑀤𑀸 𑀳𑀺 𑀲𑁄 𑀮𑁄𑀓𑀕𑀭𑀼,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀓𑀁;_  
_𑀓𑀺𑀢𑁆𑀢𑀬𑀁 𑀧𑀭𑀺𑀲𑀸𑀫𑀚𑁆𑀛𑁂,_  
_𑀅𑀕𑁆𑀕𑀝𑁆𑀞𑀸𑀦𑁂 𑀞𑀧𑁂𑀲𑀺 𑀢𑀁𑁇_  


(142.)

842\. _𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀢𑀫𑀳𑀁 𑀳𑀝𑁆𑀞𑀸,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸𑀦 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀧𑀽𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀘 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀁 𑀅𑀧𑀢𑁆𑀣𑀬𑀺𑀁𑁇_  


(143.)

843\. _𑀢𑀢𑁄 𑀅𑀯𑁄𑀘 𑀫𑀁 𑀲𑀢𑁆𑀣𑀸,_  
_‘𑀦𑀦𑁆𑀤𑁂 𑀮𑀘𑁆𑀙𑀲𑀺 𑀧𑀢𑁆𑀣𑀺𑀢𑀁;_  
_𑀧𑀤𑀻𑀧𑀥𑀫𑁆𑀫𑀤𑀸𑀦𑀸𑀦𑀁,_  
_𑀨𑀮𑀫𑁂𑀢𑀁 𑀲𑀼𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀁𑁇_  


(144.)

844\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(145.)

845\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑀸,_  
_𑀑𑀭𑀲𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀲𑀓𑀼𑀮𑀸 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀺𑀓𑀸’𑁇_  


(146.)

846\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(147.)

847\. _𑀇𑀫𑀫𑁆𑀳𑀺 𑀪𑀤𑁆𑀤𑀓𑁂 𑀓𑀧𑁆𑀧𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀫𑀳𑀸𑀬𑀲𑁄;_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀯𑀤𑀢𑀁 𑀯𑀭𑁄𑁇_  


(148.)

848\. _𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀺𑀦𑀻 𑀆𑀲𑀺𑀁,_  
_𑀢𑀤𑀸𑀳𑀁 𑀏𑀓𑀘𑀸𑀭𑀺𑀦𑀻;_  
_𑀪𑀺𑀓𑁆𑀔𑀸𑀬 𑀯𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀮𑀪𑀺𑀁 𑀢𑁂𑀮𑀫𑀢𑁆𑀢𑀓𑀁𑁇_  


(149.)

849\. _𑀢𑁂𑀦 𑀤𑀻𑀧𑀁 𑀧𑀤𑀻𑀧𑁂𑀢𑁆𑀯𑀸,_  
_𑀉𑀧𑀝𑁆𑀞𑀺𑀁 𑀲𑀩𑁆𑀩𑀲𑀁𑀯𑀭𑀺𑀁;_  
_𑀘𑁂𑀢𑀺𑀬𑀁 𑀤𑁆𑀯𑀺𑀧𑀤𑀕𑁆𑀕𑀲𑁆𑀲,_  
_𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀘𑁂𑀢𑀲𑀸𑁇_  


(150.)

850\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(151.)

851\. _𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣𑀽𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸;_  
_𑀧𑀚𑁆𑀚𑀮𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀤𑀻𑀧𑀸,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀕𑀢𑀸𑀬 𑀫𑁂𑁇_  


(152.)

852\. _𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑀁 𑀢𑀺𑀭𑁄𑀲𑁂𑀮𑀁,_  
_𑀲𑀫𑀢𑀺𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀩𑁆𑀩𑀢𑀁;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀳𑀁 𑀬𑀤𑀺𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(153.)

853\. _𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀦𑀬𑀦𑀸 𑀳𑁄𑀫𑀺,_  
_𑀬𑀲𑀲𑀸 𑀘 𑀚𑀮𑀸𑀫𑀳𑀁;_  
_𑀲𑀤𑁆𑀥𑀸𑀧𑀜𑁆𑀜𑀸𑀯𑀢𑀻 𑀘𑁂𑀯,_  
_𑀤𑀻𑀧𑀤𑀸𑀦𑀲𑁆𑀲𑀺𑀤𑀁 𑀨𑀮𑀁𑁇_  


(154.)

854\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀘 𑀪𑀯𑁂 𑀤𑀸𑀦𑀺,_  
_𑀚𑀸𑀢𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀼𑀮𑁂 𑀅𑀳𑀁;_  
_𑀧𑀳𑀽𑀢𑀥𑀦𑀥𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀫𑀼𑀤𑀺𑀢𑁂 𑀭𑀸𑀚𑀧𑀽𑀚𑀺𑀢𑁂𑁇_  


(155.)

855\. _𑀅𑀳𑀁 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀭𑀡𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀧𑀼𑀭𑀧𑁆𑀧𑀯𑁂𑀲𑁂 𑀲𑀼𑀕𑀢𑀁,_  
_𑀯𑀸𑀢𑀧𑀸𑀦𑁂 𑀞𑀺𑀢𑀸 𑀅𑀳𑀁𑁇_  


(156.)

856\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀚𑀮𑀦𑁆𑀢𑀁 𑀬𑀲𑀲𑀸,_  
_𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀲𑀓𑁆𑀓𑀢𑀁;_  
_𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀮𑀓𑁆𑀔𑀡𑁂𑀳𑀺 𑀯𑀺𑀪𑀽𑀲𑀺𑀢𑀁𑁇_  


(157.)

857\. _𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀲𑀫𑀭𑁄𑀘𑀬𑀺𑀁;_  
_𑀦 𑀘𑀺𑀭𑁂𑀦𑁂𑀯 𑀓𑀸𑀮𑁂𑀦,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(158.)

858\. _𑀇𑀤𑁆𑀥𑀻𑀲𑀼 𑀘 𑀯𑀲𑀻 𑀳𑁄𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸;_  
_𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀓𑀸𑁇_  


(159.)

859\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀯𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀲𑀺𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀮𑀸𑁇_  


(160.)

860\. _𑀧𑀭𑀺𑀘𑀺𑀡𑁆𑀡𑁄 𑀫𑀬𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀑𑀳𑀺𑀢𑁄 𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀭𑁄,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(161.)

861\. _𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀲𑀁𑀬𑁄𑀚𑀦𑀓𑁆𑀔𑀬𑁄𑁇_  


(162.)

862\. _𑀢𑀢𑁄 𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑀺 𑀫𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼𑀓𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀸,_  
_𑀲𑀓𑀼𑀮𑀸𑀢𑀺 𑀦𑀭𑀼𑀢𑁆𑀢𑀫𑁄𑁇_  


(163.)

863\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(164.)

864\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(165.)

865\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


866\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀲𑀓𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---
