# 𑁩𑁇𑁫 𑀦𑀦𑁆𑀤𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

* 𑀓𑀼𑀡𑁆𑀟𑀮𑀓𑁂𑀲𑀻𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀸𑀣𑁂𑀭𑀻𑀅𑀧𑀤𑀸𑀦

(166.)

868\. _“𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄 𑀦𑀸𑀫 𑀚𑀺𑀦𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀇𑀢𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀓𑀧𑁆𑀧𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺 𑀦𑀸𑀬𑀓𑁄𑁇_  


(167.)

869\. _𑀑𑀯𑀸𑀤𑀓𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀓𑁄,_  
_𑀢𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀤𑁂𑀲𑀦𑀸𑀓𑀼𑀲𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(168.)

870\. _𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄 𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀳𑀺𑀢𑁂𑀲𑀻 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀜𑁆𑀘𑀲𑀻𑀮𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀧𑀺𑁇_  


(169.)

871\. _𑀏𑀯𑀁 𑀦𑀺𑀭𑀸𑀓𑀼𑀮𑀁 𑀆𑀲𑀺,_  
_𑀲𑀼𑀜𑁆𑀜𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀘;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑁂𑀳𑀺,_  
_𑀯𑀲𑀻𑀪𑀽𑀢𑁂𑀳𑀺 𑀢𑀸𑀤𑀺𑀪𑀺𑁇_  


(170.)

872\. _𑀭𑀢𑀦𑀸𑀦𑀝𑁆𑀞𑀧𑀜𑁆𑀜𑀸𑀲𑀁,_  
_𑀉𑀕𑁆𑀕𑀢𑁄𑀯 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀓𑀜𑁆𑀘𑀦𑀕𑁆𑀖𑀺𑀬𑀲𑀗𑁆𑀓𑀸𑀲𑁄,_  
_𑀩𑀸𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀭𑀮𑀓𑁆𑀔𑀡𑁄𑁇_  


(171.)

873\. _𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀆𑀬𑀼 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀢𑀸𑀯𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀢𑀸𑀭𑁂𑀲𑀺 𑀚𑀦𑀢𑀁 𑀩𑀳𑀼𑀁𑁇_  


(172.)

874\. _𑀢𑀤𑀸𑀳𑀁 𑀳𑀁𑀲𑀯𑀢𑀺𑀬𑀁,_  
_𑀚𑀸𑀢𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀓𑀼𑀮𑁂 𑀅𑀳𑀼𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀧𑀚𑁆𑀚𑁄𑀢𑁂,_  
_𑀫𑀳𑀸𑀲𑀼𑀔𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


(173.)

875\. _𑀉𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀫𑀳𑀸𑀯𑀻𑀭𑀁,_  
_𑀅𑀲𑁆𑀲𑁄𑀲𑀺𑀁 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀁;_  
_𑀅𑀫𑀢𑀁 𑀧𑀭𑀫𑀲𑁆𑀲𑀸𑀤𑀁,_  
_𑀧𑀭𑀫𑀢𑁆𑀣𑀦𑀺𑀯𑁂𑀤𑀓𑀁𑁇_  


(174.)

876\. _𑀢𑀤𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀲𑀁𑀖𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁;_  
_𑀤𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀫𑀳𑀸𑀤𑀸𑀦𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀲𑁂𑀳𑀺 𑀧𑀸𑀡𑀺𑀪𑀺𑁇_  


(175.)

877\. _𑀛𑀸𑀬𑀺𑀦𑀻𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁,_  
_𑀅𑀕𑁆𑀕𑀝𑁆𑀞𑀸𑀦𑀫𑀧𑀢𑁆𑀣𑀬𑀺𑀁;_  
_𑀦𑀺𑀧𑀘𑁆𑀘 𑀲𑀺𑀭𑀲𑀸 𑀥𑀻𑀭𑀁,_  
_𑀲𑀲𑀁𑀖𑀁 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑀁𑁇_  


(176.)

878\. _𑀢𑀤𑀸 𑀅𑀤𑀦𑁆𑀢𑀤𑀫𑀓𑁄,_  
_𑀢𑀺𑀮𑁄𑀓𑀲𑀭𑀡𑁄 𑀧𑀪𑀽;_  
_𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺 𑀦𑀭𑀲𑀸𑀭𑀣𑀺,_  
_‘𑀮𑀘𑁆𑀙𑀲𑁂 𑀢𑀁 𑀲𑀼𑀧𑀢𑁆𑀣𑀺𑀢𑀁𑁇_  


(177.)

879\. _𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀺𑀢𑁄 𑀓𑀧𑁆𑀧𑁂,_  
_𑀑𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀲𑀫𑁆𑀪𑀯𑁄;_  
_𑀕𑁄𑀢𑀫𑁄 𑀦𑀸𑀫 𑀕𑁄𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(178.)

880\. _𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀤𑀸𑀬𑀸𑀤𑀸,_  
_𑀑𑀭𑀲𑀸 𑀥𑀫𑁆𑀫𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀦𑀦𑁆𑀤𑀸𑀢𑀺 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀯𑀺𑀓𑀸’𑁇_  


(179.)

881\. _𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀫𑀼𑀤𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀢𑀤𑀸 𑀚𑀺𑀦𑀁;_  
_𑀫𑁂𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸 𑀧𑀭𑀺𑀘𑀭𑀺𑀁,_  
_𑀧𑀘𑁆𑀘𑀬𑁂𑀳𑀺 𑀯𑀺𑀦𑀸𑀬𑀓𑀁𑁇_  


(180.)

882\. _𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀓𑀢𑁂𑀦,_  
_𑀘𑁂𑀢𑀦𑀸𑀧𑀡𑀺𑀥𑀻𑀳𑀺 𑀘;_  
_𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀼𑀲𑀁 𑀤𑁂𑀳𑀁,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀫𑀕𑀘𑁆𑀙𑀳𑀁𑁇_  


(181.)

883\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀬𑀸𑀫𑀫𑀕𑀁,_  
_𑀢𑀢𑁄𑀳𑀁 𑀢𑀼𑀲𑀺𑀢𑀁 𑀕𑀢𑀸;_  
_𑀢𑀢𑁄 𑀘 𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀺𑀁,_  
_𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀧𑀼𑀭𑀁 𑀢𑀢𑁄𑁇_  


(182.)

884\. _𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣𑀽𑀧𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀸𑀳𑀲𑀸;_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣𑁂𑀯 𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(183.)

885\. _𑀢𑀢𑁄 𑀘𑀼𑀢𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑁂,_  
_𑀭𑀸𑀚𑀸𑀦𑀁 𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺𑀦𑀁;_  
_𑀫𑀡𑁆𑀟𑀮𑀻𑀦𑀜𑁆𑀘 𑀭𑀸𑀚𑀽𑀦𑀁,_  
_𑀫𑀳𑁂𑀲𑀺𑀢𑁆𑀢𑀫𑀓𑀸𑀭𑀬𑀺𑀁𑁇_  


(184.)

886\. _𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀼𑀪𑁄𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑁂𑀯𑁂𑀲𑀼 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀘;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀲𑀼𑀔𑀺𑀢𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀦𑁂𑀓𑀓𑀧𑁆𑀧𑁂𑀲𑀼 𑀲𑀁𑀲𑀭𑀺𑀁𑁇_  


(185.)

887\. _𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀪𑀯𑁂 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑁂,_  
_𑀲𑀼𑀭𑀫𑁆𑀫𑁂 𑀓𑀧𑀺𑀮𑀯𑁆𑀳𑀬𑁂;_  
_𑀭𑀜𑁆𑀜𑁄 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑀲𑁆𑀲𑀸𑀳𑀁,_  
_𑀥𑀻𑀢𑀸 𑀆𑀲𑀺𑀁 𑀅𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑀸𑁇_  


(186.)

888\. _𑀲𑀺𑀭𑀺𑀬𑀸 𑀭𑀽𑀧𑀺𑀦𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀦𑀦𑁆𑀤𑀺𑀢𑀁 𑀆𑀲𑀺 𑀢𑀁 𑀓𑀼𑀮𑀁;_  
_𑀢𑁂𑀦 𑀦𑀦𑁆𑀤𑀸𑀢𑀺 𑀫𑁂 𑀦𑀸𑀫𑀁,_  
_𑀲𑀼𑀦𑁆𑀤𑀭𑀁 𑀧𑀯𑀭𑀁 𑀅𑀳𑀼𑁇_  


(187.)

889\. _𑀬𑀼𑀯𑀢𑀻𑀦𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑀸𑀲𑀁,_  
_𑀓𑀮𑁆𑀬𑀸𑀡𑀻𑀢𑀺 𑀘 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀢𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀕𑀭𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀬𑀲𑁄𑀥𑀭𑀁𑁇_  


(188.)

890\. _𑀚𑁂𑀝𑁆𑀞𑁄 𑀪𑀸𑀢𑀸 𑀢𑀺𑀮𑁄𑀓𑀕𑁆𑀕𑁄,_  
_𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀅𑀭𑀳𑀸 𑀢𑀣𑀸;_  
_𑀏𑀓𑀸𑀓𑀺𑀦𑀻 𑀕𑀳𑀝𑁆𑀞𑀸𑀳𑀁,_  
_𑀫𑀸𑀢𑀭𑀸 𑀧𑀭𑀺𑀘𑁄𑀤𑀺𑀢𑀸𑁇_  


(189.)

891\. _‘𑀲𑀸𑀓𑀺𑀬𑀫𑁆𑀳𑀺 𑀓𑀼𑀮𑁂 𑀚𑀸𑀢𑀸,_  
_𑀧𑀼𑀢𑁆𑀢𑁂 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀚𑀸 𑀢𑀼𑀯𑀁;_  
_𑀦𑀦𑁆𑀤𑁂𑀦𑀧𑀺 𑀯𑀺𑀦𑀸 𑀪𑀽𑀢𑀸,_  
_𑀅𑀕𑀸𑀭𑁂 𑀓𑀺𑀁 𑀦𑀼 𑀅𑀘𑁆𑀙𑀲𑀺𑁇_  


(190.)

892\. _𑀚𑀭𑀸𑀯𑀲𑀸𑀦𑀁 𑀬𑁄𑀩𑁆𑀩𑀜𑁆𑀜𑀁,_  
_𑀭𑀽𑀧𑀁 𑀅𑀲𑀼𑀘𑀺𑀲𑀫𑁆𑀫𑀢𑀁;_  
_𑀭𑁄𑀕𑀦𑁆𑀢𑀫𑀧𑀺𑀘𑀸𑀭𑁄𑀕𑁆𑀬𑀁,_  
_𑀚𑀻𑀯𑀺𑀢𑀁 𑀫𑀭𑀡𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


(191.)

893\. _𑀇𑀤𑀫𑁆𑀧𑀺 𑀢𑁂 𑀲𑀼𑀪𑀁 𑀭𑀽𑀧𑀁,_  
_𑀲𑀲𑀻𑀓𑀦𑁆𑀢𑀁 𑀫𑀦𑁄𑀳𑀭𑀁;_  
_𑀪𑀽𑀲𑀦𑀸𑀦𑀁 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑀁,_  
_𑀲𑀺𑀭𑀺𑀲𑀗𑁆𑀖𑀸𑀝𑀲𑀦𑁆𑀦𑀺𑀪𑀁𑁇_  


(192.)

894\. _𑀧𑀼𑀜𑁆𑀚𑀺𑀢𑀁 𑀮𑁄𑀓𑀲𑀸𑀭𑀁𑀯,_  
_𑀦𑀬𑀦𑀸𑀦𑀁 𑀭𑀲𑀸𑀬𑀦𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀁 𑀓𑀺𑀢𑁆𑀢𑀺𑀚𑀦𑀦𑀁,_  
_𑀉𑀓𑁆𑀓𑀸𑀓𑀓𑀼𑀮𑀦𑀦𑁆𑀤𑀦𑀁𑁇_  


(193.)

895\. _𑀦 𑀘𑀺𑀭𑁂𑀦𑁂𑀯 𑀓𑀸𑀮𑁂𑀦,_  
_𑀚𑀭𑀸 𑀲𑀫𑀥𑀺𑀲𑁂𑀲𑁆𑀲𑀢𑀺;_  
_𑀯𑀺𑀳𑀸𑀬 𑀕𑁂𑀳𑀁 𑀓𑀸𑀭𑀼𑀜𑁆𑀜𑁂,_  
_𑀘𑀭 𑀥𑀫𑁆𑀫𑀫𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑁂’𑁇_  


(194.)

896\. _𑀲𑀼𑀢𑁆𑀯𑀸𑀳𑀁 𑀫𑀸𑀢𑀼 𑀯𑀘𑀦𑀁,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀤𑁂𑀳𑁂𑀦 𑀦 𑀢𑀼 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀭𑀽𑀧𑀬𑁄𑀩𑁆𑀩𑀦𑀮𑀸𑀍𑀇𑀢𑀸𑁇_  


(195.)

897\. _𑀫𑀳𑀢𑀸 𑀘 𑀧𑀬𑀢𑁆𑀢𑁂𑀦,_  
_𑀛𑀸𑀦𑀚𑁆𑀛𑁂𑀦 𑀧𑀭𑀁 𑀫𑀫;_  
_𑀓𑀸𑀢𑀼𑀜𑁆𑀘 𑀯𑀤𑀢𑁂 𑀫𑀸𑀢𑀸,_  
_𑀦 𑀘𑀸𑀳𑀁 𑀢𑀢𑁆𑀣 𑀉𑀲𑁆𑀲𑀼𑀓𑀸𑁇_  


(196.)

898\. _𑀢𑀢𑁄 𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀁 𑀓𑀸𑀫𑀮𑀸𑀮𑀲𑀁;_  
_𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀦𑀢𑁆𑀣𑀁 𑀭𑀽𑀧𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑀫 𑀘𑀓𑁆𑀔𑀼𑀧𑀣𑁂 𑀚𑀺𑀦𑁄𑁇_  


(197.)

899\. _𑀲𑀓𑁂𑀦 𑀆𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀇𑀢𑁆𑀣𑀺𑀁 𑀫𑀸𑀧𑁂𑀲𑀺 𑀲𑁄𑀪𑀺𑀦𑀺𑀁;_  
_𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀁 𑀲𑀼𑀭𑀼𑀘𑀺𑀭𑀁,_  
_𑀫𑀫𑀢𑁄𑀧𑀺 𑀲𑀼𑀭𑀽𑀧𑀺𑀦𑀺𑀁𑁇_  


(198.)

900\. _𑀢𑀫𑀳𑀁 𑀯𑀺𑀫𑁆𑀳𑀺𑀢𑀸 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀅𑀢𑀺𑀯𑀺𑀫𑁆𑀳𑀺𑀢𑀤𑁂𑀳𑀺𑀦𑀺𑀁;_  
_𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀁 ‘𑀲𑀨𑀮𑀁 𑀫𑁂𑀢𑀺,_  
_𑀦𑁂𑀢𑁆𑀢𑀮𑀸𑀪𑀜𑁆𑀘 𑀫𑀸𑀦𑀼𑀲𑀁𑁇_  


(199.)

901\. _𑀢𑀫𑀳𑀁 𑀏𑀳𑀺 𑀲𑀼𑀪𑀕𑁂,_  
_𑀬𑁂𑀦𑀢𑁆𑀣𑁄 𑀢𑀁 𑀯𑀤𑁂𑀳𑀺 𑀫𑁂;_  
_𑀓𑀼𑀮𑀁 𑀢𑁂 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀯𑀤 𑀫𑁂 𑀬𑀤𑀺 𑀢𑁂 𑀧𑀺𑀬𑀁𑁇_  


(200.)

902\. _𑀦 𑀯𑀜𑁆𑀘𑀓𑀸𑀮𑁄 𑀲𑀼𑀪𑀕𑁂,_  
_𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂 𑀫𑀁 𑀦𑀺𑀯𑀸𑀲𑀬;_  
_𑀲𑀻𑀤𑀦𑁆𑀢𑀻𑀯 𑀫𑀫𑀗𑁆𑀕𑀸𑀦𑀺,_  
_𑀧𑀲𑀼𑀧𑁆𑀧𑀬 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀓𑀁’𑁇_  


(201.)

903\. _𑀢𑀢𑁄 𑀲𑀻𑀲𑀁 𑀫𑀫𑀗𑁆𑀕𑁂 𑀲𑀸,_  
_𑀓𑀢𑁆𑀯𑀸 𑀲𑀬𑀺 𑀲𑀼𑀮𑁄𑀘𑀦𑀸;_  
_𑀢𑀲𑁆𑀲𑀸 𑀦𑀮𑀸𑀝𑁂 𑀧𑀢𑀺𑀢𑀸,_  
_𑀮𑀼𑀤𑁆𑀥𑀸 𑀧𑀭𑀫𑀤𑀸𑀭𑀼𑀡𑀸𑁇_  


(202.)

904\. _𑀲𑀳 𑀢𑀲𑁆𑀲𑀸 𑀦𑀺𑀧𑀸𑀢𑁂𑀦,_  
_𑀧𑀺𑀍𑀅𑀓𑀸 𑀉𑀧𑀧𑀚𑁆𑀚𑀣;_  
_𑀧𑀕𑁆𑀖𑀭𑀺𑀁𑀲𑀼 𑀧𑀪𑀺𑀦𑁆𑀦𑀸 𑀘,_  
_𑀓𑀼𑀡𑀧𑀸 𑀧𑀼𑀩𑁆𑀩𑀮𑁄𑀳𑀺𑀢𑀸𑁇_  


(203.)

905\. _𑀧𑀪𑀺𑀦𑁆𑀦𑀁 𑀯𑀤𑀦𑀜𑁆𑀘𑀸𑀧𑀺,_  
_𑀓𑀼𑀡𑀧𑀁 𑀧𑀽𑀢𑀺𑀕𑀦𑁆𑀥𑀦𑀁;_  
_𑀉𑀤𑁆𑀥𑀼𑀫𑀸𑀢𑀁 𑀯𑀺𑀦𑀺𑀮𑀜𑁆𑀘,_  
_𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀭𑀻𑀭𑀓𑀁𑁇_  


(204.)

906\. _𑀲𑀸 𑀧𑀯𑁂𑀤𑀺𑀢𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀻,_  
_𑀦𑀺𑀲𑁆𑀲𑀲𑀦𑁆𑀢𑀻 𑀫𑀼𑀳𑀼𑀁 𑀫𑀼𑀳𑀼𑀁;_  
_𑀯𑁂𑀤𑀬𑀦𑁆𑀢𑀻 𑀲𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁,_  
_𑀓𑀭𑀼𑀡𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑀬𑀺𑁇_  


(205.)

907\. _‘𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀫𑀺,_  
_𑀨𑀼𑀲𑀬𑀦𑁆𑀢𑀺 𑀘 𑀯𑁂𑀤𑀦𑀸;_  
_𑀫𑀳𑀸𑀤𑀼𑀓𑁆𑀔𑁂 𑀦𑀺𑀫𑀼𑀕𑁆𑀕𑀫𑁆𑀳𑀺,_  
_𑀲𑀭𑀡𑀁 𑀳𑁄𑀳𑀺 𑀫𑁂 𑀲𑀔𑀻𑁇_  


(206.)

908\. _𑀓𑀼𑀳𑀺𑀁 𑀯𑀤𑀦𑀲𑁄𑀪𑀁 𑀢𑁂,_  
_𑀓𑀼𑀳𑀺𑀁 𑀢𑁂 𑀢𑀼𑀗𑁆𑀕𑀦𑀸𑀲𑀺𑀓𑀸;_  
_𑀢𑀫𑁆𑀩𑀩𑀺𑀫𑁆𑀩𑀯𑀭𑁄𑀝𑁆𑀞𑀁 𑀢𑁂,_  
_𑀯𑀤𑀦𑀁 𑀢𑁂 𑀓𑀼𑀳𑀺𑀁 𑀕𑀢𑀁𑁇_  


(207.)

909\. _𑀓𑀼𑀳𑀺𑀁 𑀲𑀲𑀻𑀦𑀺𑀪𑀁 𑀯𑀡𑁆𑀡𑀁,_  
_𑀓𑀫𑁆𑀩𑀼𑀕𑀻𑀯𑀸 𑀓𑀼𑀳𑀺𑀁 𑀕𑀢𑀸;_  
_𑀤𑁄𑀍𑀆𑀮𑁄𑀮𑀸𑀯 𑀢𑁂 𑀓𑀡𑁆𑀡𑀸,_  
_𑀯𑁂𑀯𑀡𑁆𑀡𑀁 𑀲𑀫𑀼𑀧𑀸𑀕𑀢𑀸𑁇_  


(208.)

910\. _𑀫𑀓𑀼𑀍𑀅𑀔𑀸𑀭𑀓𑀸𑀓𑀸𑀭𑀸,_  
_𑀓𑀮𑀺𑀓𑀸𑀯 𑀧𑀬𑁄𑀥𑀭𑀸;_  
_𑀧𑀪𑀺𑀦𑁆𑀦𑀸 𑀧𑀽𑀢𑀺𑀓𑀼𑀡𑀧𑀸,_  
_𑀤𑀼𑀝𑁆𑀞𑀕𑀦𑁆𑀥𑀺𑀢𑁆𑀢𑀫𑀸𑀕𑀢𑀸𑁇_  


(209.)

911\. _𑀯𑁂𑀤𑀺𑀫𑀚𑁆𑀛𑀸𑀯 𑀲𑀼𑀲𑁆𑀲𑁄𑀡𑀻,_  
_𑀲𑀽𑀦𑀸𑀯 𑀦𑀻𑀢𑀓𑀺𑀩𑁆𑀩𑀺𑀲𑀸;_  
_𑀚𑀸𑀢𑀸 𑀅𑀫𑁂𑀚𑁆𑀛𑀪𑀭𑀺𑀢𑀸,_  
_𑀅𑀳𑁄 𑀭𑀽𑀧𑀫𑀲𑀲𑁆𑀲𑀢𑀁𑁇_  


(210.)

912\. _𑀲𑀩𑁆𑀩𑀁 𑀲𑀭𑀻𑀭𑀲𑀜𑁆𑀚𑀸𑀢𑀁,_  
_𑀧𑀽𑀢𑀺𑀕𑀦𑁆𑀥𑀁 𑀪𑀬𑀸𑀦𑀓𑀁;_  
_𑀲𑀼𑀲𑀸𑀦𑀫𑀺𑀯 𑀩𑀻𑀪𑀘𑁆𑀙𑀁,_  
_𑀭𑀫𑀦𑁆𑀢𑁂 𑀬𑀢𑁆𑀣 𑀩𑀸𑀮𑀺𑀲𑀸’𑁇_  


(211.)

913\. _𑀢𑀤𑀸 𑀫𑀳𑀸𑀓𑀸𑀭𑀼𑀡𑀺𑀓𑁄,_  
_𑀪𑀸𑀢𑀸 𑀫𑁂 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀁,_  
_𑀇𑀫𑀸 𑀕𑀸𑀣𑀸 𑀅𑀪𑀸𑀲𑀣𑁇_  


(212.)

914\. _‘𑀆𑀢𑀼𑀭𑀁 𑀓𑀼𑀡𑀧𑀁 𑀧𑀽𑀢𑀺𑀁,_  
_𑀧𑀲𑁆𑀲 𑀦𑀦𑁆𑀤𑁂 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀁;_  
_𑀅𑀲𑀼𑀪𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀳𑀺,_  
_𑀏𑀓𑀕𑁆𑀕𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀁𑁇_  


(213.)

915\. _𑀬𑀣𑀸 𑀇𑀤𑀁 𑀢𑀣𑀸 𑀏𑀢𑀁,_  
_𑀬𑀣𑀸 𑀏𑀢𑀁 𑀢𑀣𑀸 𑀇𑀤𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀧𑀽𑀢𑀺𑀓𑀁 𑀯𑀸𑀢𑀺,_  
_𑀩𑀸𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀁𑁇_  


(214.)

916\. _𑀏𑀯𑀫𑁂𑀢𑀁 𑀅𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀻,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑀸;_  
_𑀢𑀢𑁄 𑀲𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀺𑀚𑁆𑀛 𑀤𑀓𑁆𑀔𑀲𑀺’𑁇_  


(215.)

917\. _𑀢𑀢𑁄𑀳𑀁 𑀅𑀢𑀺𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑀸,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀕𑀸𑀣𑀸 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀸;_  
_𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑀺𑀢𑀸𑀯𑀳𑀁 𑀲𑀦𑁆𑀢𑀻,_  
_𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀧𑀸𑀧𑀼𑀡𑀺𑀁𑁇_  


(216.)

918\. _𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀳𑀁,_  
_𑀲𑀤𑀸 𑀛𑀸𑀦𑀧𑀭𑀸𑀬𑀡𑀸;_  
_𑀚𑀺𑀦𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀼𑀡𑁂 𑀢𑀼𑀝𑁆𑀞𑁄,_  
_𑀏𑀢𑀤𑀕𑁆𑀕𑁂 𑀞𑀧𑁂𑀲𑀺 𑀫𑀁𑁇_  


(217.)

919\. _𑀓𑀺𑀮𑁂𑀲𑀸 𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁,_  
_…𑀧𑁂…_  
_𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀦𑀸𑀲𑀯𑀸𑁇_  


(218.)

920\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀯𑀢 𑀫𑁂 𑀆𑀲𑀺,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(219.)

921\. _𑀧𑀝𑀺𑀲𑀫𑁆𑀪𑀺𑀤𑀸 𑀘𑀢𑀲𑁆𑀲𑁄,_  
_…𑀧𑁂…_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁”𑁇_  


922\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀦𑀧𑀤𑀓𑀮𑁆𑀬𑀸𑀡𑀻 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇

---

923\. 𑀦𑀦𑁆𑀤𑀸𑀣𑁂𑀭𑀺𑀬𑀸𑀧𑀤𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
