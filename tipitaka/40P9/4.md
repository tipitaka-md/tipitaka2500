# 𑁪 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀆𑀘𑀬𑀕𑀸𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀤𑀺

* 𑀳𑁂𑀢𑀼𑀤𑀼𑀓, 𑀆𑀘𑀬𑀕𑀸𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀤𑀺

(17.)

46\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

47\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

48\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀦𑁂𑀯𑀆𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀦𑀸𑀧𑀘𑀬𑀕𑀸𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

(18.)

49\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀲𑁂𑀓𑁆𑀔𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

50\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀲𑁂𑀓𑁆𑀔𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

51\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(19.)

52\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

53\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀫𑀳𑀕𑁆𑀕𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

54\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(20.)

55\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

56\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀫𑀳𑀕𑁆𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

57\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(21.)

58\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀳𑀻𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

59\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀫𑀚𑁆𑀛𑀺𑀫𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

60\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀧𑀡𑀻𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(22.)

61\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

62\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀦𑀺𑀬𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

63\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀦𑀺𑀬𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

(23.)

64\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

65\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

66\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(24.)

67\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

68\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑁅𑀧𑁆𑀧𑀸𑀤𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(25.)

69\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀢𑀻𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

70\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀦𑀸𑀕𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(26.)

71\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀢𑀻𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

72\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀦𑀸𑀕𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

73\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(27.)

74\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

75\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

(28.)

76\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)

77\. 𑀦𑀳𑁂𑀢𑀼𑀁 𑀦𑀩𑀳𑀺𑀤𑁆𑀥𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇)
