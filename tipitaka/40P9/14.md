# 𑁧𑁪 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀓𑀺𑀮𑁂𑀲𑀕𑁄𑀘𑁆𑀙𑀓, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

(68.)

169. 𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀦𑁄𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

170. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

(69.)

171. 𑀦𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(70.)

172. 𑀦𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

173. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀆𑀲𑁂𑀯𑀦𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇

(71.)

174. 𑀦𑀓𑀺𑀮𑁂𑀲𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

175. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇

(72.)

176. 𑀦𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

177. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

(73.)

178. 𑀦𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀓𑀺𑀮𑁂𑀲𑀜𑁆𑀘𑁂𑀯 𑀦𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀦𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀜𑁆𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂… 𑁇 (𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀦𑀯𑁇 𑀯𑀺𑀧𑀸𑀓𑀁 𑀦𑀢𑁆𑀣𑀺𑁇)

179. 𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀓𑀺𑀮𑁂𑀲𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀅𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀺𑀓𑀁 𑀦𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘… 𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀆𑀲𑁂𑀯𑀦𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇
