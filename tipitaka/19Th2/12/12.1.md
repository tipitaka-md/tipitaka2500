# 𑁧𑁨𑁇𑁧 𑀧𑀼𑀡𑁆𑀡𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀲𑁄𑀍𑀅𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀧𑀼𑀡𑁆𑀡𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(236.)

312\. _“𑀉𑀤𑀳𑀸𑀭𑀻 𑀅𑀳𑀁 𑀲𑀻𑀢𑁂,_  
_𑀲𑀤𑀸 𑀉𑀤𑀓𑀫𑁄𑀢𑀭𑀺𑀁;_  
_𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀤𑀡𑁆𑀟𑀪𑀬𑀪𑀻𑀢𑀸,_  
_𑀯𑀸𑀘𑀸𑀤𑁄𑀲𑀪𑀬𑀝𑁆𑀝𑀺𑀢𑀸𑁇_  


(237.)

313\. _𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀢𑁆𑀯𑀁 𑀪𑀻𑀢𑁄,_  
_𑀲𑀤𑀸 𑀉𑀤𑀓𑀫𑁄𑀢𑀭𑀺;_  
_𑀯𑁂𑀥𑀫𑀸𑀦𑁂𑀳𑀺 𑀕𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀲𑀻𑀢𑀁 𑀯𑁂𑀤𑀬𑀲𑁂 𑀪𑀼𑀲𑀁”𑁇_  


(238.)

314\. _“𑀚𑀸𑀦𑀦𑁆𑀢𑀻 𑀯𑀢 𑀫𑀁 𑀪𑁄𑀢𑀺,_  
_𑀧𑀼𑀡𑁆𑀡𑀺𑀓𑁂 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀲𑀺;_  
_𑀓𑀭𑁄𑀦𑁆𑀢𑀁 𑀓𑀼𑀲𑀮𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀭𑀼𑀦𑁆𑀥𑀦𑁆𑀢𑀁 𑀓𑀢𑀧𑀸𑀧𑀓𑀁𑁇_  


(239.)

315\. _𑀬𑁄 𑀘 𑀯𑀼𑀟𑁆𑀠𑁄 𑀤𑀳𑀭𑁄 𑀯𑀸,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀁 𑀧𑀓𑀼𑀩𑁆𑀩𑀢𑀺;_  
_𑀤𑀓𑀸𑀪𑀺𑀲𑁂𑀘𑀦𑀸 𑀲𑁄𑀧𑀺,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(240.)

316\. _“𑀓𑁄 𑀦𑀼 𑀢𑁂 𑀇𑀤𑀫𑀓𑁆𑀔𑀸𑀲𑀺,_  
_𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀚𑀸𑀦𑀓𑁄;_  
_𑀤𑀓𑀸𑀪𑀺𑀲𑁂𑀘𑀦𑀸 𑀦𑀸𑀫,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(241.)

317\. _𑀲𑀕𑁆𑀕𑀁 𑀦𑀽𑀦 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑀩𑁆𑀩𑁂 𑀫𑀡𑁆𑀟𑀽𑀓𑀓𑀘𑁆𑀙𑀧𑀸;_  
_𑀦𑀸𑀕𑀸 𑀘 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀬𑁂 𑀘𑀜𑁆𑀜𑁂 𑀉𑀤𑀓𑁂 𑀘𑀭𑀸𑁇_  


(242.)

318\. _𑀑𑀭𑀩𑁆𑀪𑀺𑀓𑀸 𑀲𑀽𑀓𑀭𑀺𑀓𑀸,_  
_𑀫𑀘𑁆𑀙𑀺𑀓𑀸 𑀫𑀺𑀕𑀩𑀦𑁆𑀥𑀓𑀸;_  
_𑀘𑁄𑀭𑀸 𑀘 𑀯𑀚𑁆𑀛𑀖𑀸𑀢𑀸 𑀘,_  
_𑀬𑁂 𑀘𑀜𑁆𑀜𑁂 𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀤𑀓𑀸𑀪𑀺𑀲𑁂𑀘𑀦𑀸 𑀢𑁂𑀧𑀺,_  
_𑀧𑀸𑀧𑀓𑀫𑁆𑀫𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀭𑁂𑁇_  


(243.)

319\. _𑀲𑀘𑁂 𑀇𑀫𑀸 𑀦𑀤𑀺𑀬𑁄 𑀢𑁂,_  
_𑀧𑀸𑀧𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀁 𑀯𑀳𑀼𑀁;_  
_𑀧𑀼𑀜𑁆𑀜𑀫𑁆𑀧𑀺𑀫𑀸 𑀯𑀳𑁂𑀬𑁆𑀬𑀼𑀁 𑀢𑁂,_  
_𑀢𑁂𑀦 𑀢𑁆𑀯𑀁 𑀧𑀭𑀺𑀩𑀸𑀳𑀺𑀭𑁄𑁇_  


(244.)

320\. _𑀬𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀢𑁆𑀯𑀁 𑀪𑀻𑀢𑁄,_  
_𑀲𑀤𑀸 𑀉𑀤𑀓𑀫𑁄𑀢𑀭𑀺;_  
_𑀢𑀫𑁂𑀯 𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀫𑀸 𑀓𑀸𑀲𑀺,_  
_𑀫𑀸 𑀢𑁂 𑀲𑀻𑀢𑀁 𑀙𑀯𑀺𑀁 𑀳𑀦𑁂”𑁇_  


(245.)

321\. _“𑀓𑀼𑀫𑁆𑀫𑀕𑁆𑀕𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀁 𑀫𑀁,_  
_𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀁 𑀲𑀫𑀸𑀦𑀬𑀺;_  
_𑀤𑀓𑀸𑀪𑀺𑀲𑁂𑀘𑀦𑀸 𑀪𑁄𑀢𑀺,_  
_𑀇𑀫𑀁 𑀲𑀸𑀝𑀁 𑀤𑀤𑀸𑀫𑀺 𑀢𑁂”𑁇_  


(246.)

322\. _“𑀢𑀼𑀬𑁆𑀳𑁂𑀯 𑀲𑀸𑀝𑀓𑁄 𑀳𑁄𑀢𑀼,_  
_𑀦𑀸𑀳𑀫𑀺𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀸𑀝𑀓𑀁;_  
_𑀲𑀘𑁂 𑀪𑀸𑀬𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀲𑀘𑁂 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀫𑀧𑁆𑀧𑀺𑀬𑀁𑁇_  


(247.)

323\. _𑀫𑀸𑀓𑀸𑀲𑀺 𑀧𑀸𑀧𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀆𑀯𑀺 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀳𑁄;_  
_𑀲𑀘𑁂 𑀘 𑀧𑀸𑀧𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺 𑀓𑀭𑁄𑀲𑀺 𑀯𑀸𑁇_  


(248.)

324\. _𑀦 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀢𑁆𑀬𑀢𑁆𑀣𑀺,_  
_𑀉𑀧𑁂𑀘𑁆𑀘𑀸𑀧𑀺 𑀧𑀮𑀸𑀬𑀢𑁄;_  
_𑀲𑀘𑁂 𑀪𑀸𑀬𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀲𑀘𑁂 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀫𑀧𑁆𑀧𑀺𑀬𑀁𑁇_  


(249.)

325\. _𑀉𑀧𑁂𑀳𑀺 𑀲𑀭𑀡𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀖𑀜𑁆𑀘 𑀢𑀸𑀤𑀺𑀦𑀁;_  
_𑀲𑀫𑀸𑀤𑀺𑀬𑀸𑀳𑀺 𑀲𑀻𑀮𑀸𑀦𑀺,_  
_𑀢𑀁 𑀢𑁂 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑁂𑀳𑀺𑀢𑀺”𑁇_  


(250.)

326\. _“𑀉𑀧𑁂𑀫𑀺 𑀲𑀭𑀡𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀖𑀜𑁆𑀘 𑀢𑀸𑀤𑀺𑀦𑀁;_  
_𑀲𑀫𑀸𑀤𑀺𑀬𑀸𑀫𑀺 𑀲𑀻𑀮𑀸𑀦𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑁂𑀳𑀺𑀢𑀺𑁇_  


(251.)

327\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀅𑀚𑁆𑀚𑀫𑁆𑀳𑀺 𑀲𑀘𑁆𑀘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑁄 𑀯𑁂𑀤𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑁄𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀫𑁆𑀳𑀺 𑀦𑁆𑀳𑀸𑀢𑀓𑁄”𑀢𑀺𑁇_  


---

328\. …  𑀧𑀼𑀡𑁆𑀡𑀸 𑀣𑁂𑀭𑀻… 𑁇
