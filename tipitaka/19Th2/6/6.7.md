# 𑁬𑁇𑁭 𑀕𑀼𑀢𑁆𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀕𑀼𑀢𑁆𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(163.)

224\. _“𑀕𑀼𑀢𑁆𑀢𑁂 𑀬𑀤𑀢𑁆𑀣𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀲𑀼𑀁 𑀧𑀺𑀬𑀁;_  
_𑀢𑀫𑁂𑀯 𑀅𑀦𑀼𑀩𑁆𑀭𑀽𑀳𑁂𑀳𑀺,_  
_𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀫𑀺𑁇_  


(164.)

225\. _𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀜𑁆𑀘𑀺𑀢𑀸 𑀲𑀢𑁆𑀢𑀸,_  
_𑀫𑀸𑀭𑀲𑁆𑀲 𑀯𑀺𑀲𑀬𑁂 𑀭𑀢𑀸;_  
_𑀅𑀦𑁂𑀓𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀲𑀦𑁆𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀅𑀯𑀺𑀤𑁆𑀤𑀲𑀽𑁇_  


(165.)

226\. _𑀓𑀸𑀫𑀘𑁆𑀙𑀦𑁆𑀤𑀜𑁆𑀘 𑀩𑁆𑀬𑀸𑀧𑀸𑀤𑀁,_  
_𑀲𑀓𑁆𑀓𑀸𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀫𑁂𑀯 𑀘;_  
_𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀧𑀭𑀸𑀫𑀸𑀲𑀁,_  
_𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀜𑁆𑀘 𑀧𑀜𑁆𑀘𑀫𑀁𑁇_  


(166.)

227\. _𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀏𑀢𑀸𑀦𑀺,_  
_𑀧𑀚𑀳𑀺𑀢𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻;_  
_𑀑𑀭𑀫𑁆𑀪𑀸𑀕𑀫𑀦𑀻𑀬𑀸𑀦𑀺,_  
_𑀦𑀬𑀺𑀤𑀁 𑀧𑀼𑀦𑀭𑁂𑀳𑀺𑀲𑀺𑁇_  


(167.)

228\. _𑀭𑀸𑀕𑀁 𑀫𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀜𑁆𑀘,_  
_𑀉𑀤𑁆𑀥𑀘𑁆𑀘𑀜𑁆𑀘 𑀯𑀺𑀯𑀚𑁆𑀚𑀺𑀬;_  
_𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺 𑀙𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(168.)

229\. _𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀬 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑀁;_  
_𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀦𑀺𑀘𑁆𑀙𑀸𑀢𑀸,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀸 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


---

230\. …  𑀕𑀼𑀢𑁆𑀢𑀸 𑀣𑁂𑀭𑀻… 𑁇
