# 𑁬𑁇𑁪 𑀲𑀼𑀚𑀸𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀼𑀚𑀸𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(145.)

203\. _“𑀅𑀮𑀗𑁆𑀓𑀢𑀸 𑀲𑀼𑀯𑀲𑀦𑀸,_  
_𑀫𑀸𑀮𑀺𑀦𑀻 𑀘𑀦𑁆𑀤𑀦𑁄𑀓𑁆𑀔𑀺𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀭𑀡𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀤𑀸𑀲𑀻𑀕𑀡𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀸𑁇_  


(146.)

204\. _𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀜𑁆𑀘 𑀆𑀤𑀸𑀬,_  
_𑀔𑀚𑁆𑀚𑀁 𑀪𑁄𑀚𑁆𑀚𑀁 𑀅𑀦𑀧𑁆𑀧𑀓𑀁;_  
_𑀕𑁂𑀳𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀬𑁆𑀬𑀸𑀦𑀫𑀪𑀺𑀳𑀸𑀭𑀬𑀺𑀁𑁇_  


(147.)

205\. _𑀢𑀢𑁆𑀣 𑀭𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀻𑀍𑀇𑀢𑁆𑀯𑀸,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀲𑀓𑀁 𑀖𑀭𑀁;_  
_𑀯𑀺𑀳𑀸𑀭𑀁 𑀤𑀝𑁆𑀞𑀼𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑀁,_  
_𑀲𑀸𑀓𑁂𑀢𑁂 𑀅𑀜𑁆𑀚𑀦𑀁 𑀯𑀦𑀁𑁇_  


(148.)

206\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀮𑁄𑀓𑀧𑀚𑁆𑀚𑁄𑀢𑀁,_  
_𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑀁;_  
_𑀲𑁄 𑀫𑁂 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀘𑀓𑁆𑀔𑀼𑀫𑀸𑁇_  


(149.)

207\. _𑀲𑀼𑀢𑁆𑀯𑀸 𑀘 𑀔𑁄 𑀫𑀳𑁂𑀲𑀺𑀲𑁆𑀲,_  
_𑀲𑀘𑁆𑀘𑀁 𑀲𑀫𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀳𑀁;_  
_𑀢𑀢𑁆𑀣𑁂𑀯 𑀯𑀺𑀭𑀚𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀨𑀼𑀲𑀬𑀺𑀁 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁𑁇_  


(150.)

208\. _𑀢𑀢𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀅𑀫𑁄𑀖𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

209\. …  𑀲𑀼𑀚𑀸𑀢𑀸 𑀣𑁂𑀭𑀻… 𑁇
