# 𑁬𑁇𑁩 𑀔𑁂𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀔𑁂𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(139.)

196\. _“𑀤𑀳𑀭𑀸 𑀢𑁆𑀯𑀁 𑀭𑀽𑀧𑀯𑀢𑀻,_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀤𑀳𑀭𑁄 𑀬𑀼𑀯𑀸;_  
_𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑁂𑀦 𑀢𑀼𑀭𑀺𑀬𑁂𑀦,_  
_𑀏𑀳𑀺 𑀔𑁂𑀫𑁂 𑀭𑀫𑀸𑀫𑀲𑁂”𑁇_  


(140.)

197\. _“𑀇𑀫𑀺𑀦𑀸 𑀧𑀽𑀢𑀺𑀓𑀸𑀬𑁂𑀦,_  
_𑀆𑀢𑀼𑀭𑁂𑀦 𑀧𑀪𑀗𑁆𑀕𑀼𑀦𑀸;_  
_𑀅𑀝𑁆𑀝𑀺𑀬𑀸𑀫𑀺 𑀳𑀭𑀸𑀬𑀸𑀫𑀺,_  
_𑀓𑀸𑀫𑀢𑀡𑁆𑀳𑀸 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(141.)

198\. _𑀲𑀢𑁆𑀢𑀺𑀲𑀽𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸,_  
_𑀔𑀦𑁆𑀥𑀸𑀲𑀁 𑀅𑀥𑀺𑀓𑀼𑀝𑁆𑀝𑀦𑀸;_  
_𑀬𑀁 ‘𑀢𑁆𑀯𑀁 𑀓𑀸𑀫𑀭𑀢𑀺𑀁’ 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_‘𑀅𑀭𑀢𑀻’ 𑀤𑀸𑀦𑀺 𑀲𑀸 𑀫𑀫𑁇_  


(142.)

199\. _𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀳𑀢𑀸 𑀦𑀦𑁆𑀤𑀻,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁄 𑀧𑀤𑀸𑀮𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦𑀺𑀳𑀢𑁄 𑀢𑁆𑀯𑀫𑀲𑀺 𑀅𑀦𑁆𑀢𑀓𑁇_  


(143.)

200\. _𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀸𑀦𑀺 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀸,_  
_𑀅𑀕𑁆𑀕𑀺𑀁 𑀧𑀭𑀺𑀘𑀭𑀁 𑀯𑀦𑁂;_  
_𑀬𑀣𑀸𑀪𑀼𑀘𑁆𑀘𑀫𑀚𑀸𑀦𑀦𑁆𑀢𑀸,_  
_𑀩𑀸𑀮𑀸 𑀲𑀼𑀤𑁆𑀥𑀺𑀫𑀫𑀜𑁆𑀜𑀣𑁇_  


(144.)

201\. _𑀅𑀳𑀜𑁆𑀘 𑀔𑁄 𑀦𑀫𑀲𑁆𑀲𑀦𑁆𑀢𑀻,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀼𑀭𑀺𑀲𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀧𑀫𑀼𑀢𑁆𑀢𑀸 𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺,_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀓𑀸𑀭𑀺𑀓𑀸”𑀢𑀺𑁇_  


---

202\. …  𑀔𑁂𑀫𑀸 𑀣𑁂𑀭𑀻… 𑁇
