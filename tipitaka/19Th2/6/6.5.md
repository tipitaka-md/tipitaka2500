# 𑁬𑁇𑁫 𑀅𑀦𑁄𑀧𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀙𑀓𑁆𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀅𑀦𑁄𑀧𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(151.)

210\. _“𑀉𑀘𑁆𑀘𑁂 𑀓𑀼𑀮𑁂 𑀅𑀳𑀁 𑀚𑀸𑀢𑀸,_  
_𑀩𑀳𑀼𑀯𑀺𑀢𑁆𑀢𑁂 𑀫𑀳𑀤𑁆𑀥𑀦𑁂;_  
_𑀯𑀡𑁆𑀡𑀭𑀽𑀧𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀥𑀻𑀢𑀸 𑀫𑀚𑁆𑀛𑀲𑁆𑀲 𑀅𑀢𑁆𑀭𑀚𑀸𑁇_  


(152.)

211\. _𑀧𑀢𑁆𑀣𑀺𑀢𑀸 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀕𑀺𑀚𑁆𑀛𑀺𑀢𑀸;_  
_𑀧𑀺𑀢𑀼 𑀫𑁂 𑀧𑁂𑀲𑀬𑀻 𑀤𑀽𑀢𑀁,_  
_𑀤𑁂𑀣 𑀫𑀬𑁆𑀳𑀁 𑀅𑀦𑁄𑀧𑀫𑀁𑁇_  


(153.)

212\. _𑀬𑀢𑁆𑀢𑀓𑀁 𑀢𑀼𑀮𑀺𑀢𑀸 𑀏𑀲𑀸,_  
_𑀢𑀼𑀬𑁆𑀳𑀁 𑀥𑀻𑀢𑀸 𑀅𑀦𑁄𑀧𑀫𑀸;_  
_𑀢𑀢𑁄 𑀅𑀝𑁆𑀞𑀕𑀼𑀡𑀁 𑀤𑀲𑁆𑀲𑀁,_  
_𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀭𑀢𑀦𑀸𑀦𑀺 𑀘𑁇_  


(154.)

213\. _𑀲𑀸𑀳𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀮𑁄𑀓𑀚𑁂𑀝𑁆𑀞𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀦𑀺 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑁇_  


(155.)

214\. _𑀲𑁄 𑀫𑁂 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀕𑁄𑀢𑀫𑁄;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀆𑀲𑀦𑁂 𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀨𑀼𑀲𑀬𑀺𑀁 𑀢𑀢𑀺𑀬𑀁 𑀨𑀮𑀁𑁇_  


(156.)

215\. _𑀢𑀢𑁄 𑀓𑁂𑀲𑀸𑀦𑀺 𑀙𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀅𑀚𑁆𑀚 𑀫𑁂 𑀲𑀢𑁆𑀢𑀫𑀻 𑀭𑀢𑁆𑀢𑀺,_  
_𑀬𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀯𑀺𑀲𑁄𑀲𑀺𑀢𑀸”𑀢𑀺𑁇_  


---

216\. …  𑀅𑀦𑁄𑀧𑀫𑀸 𑀣𑁂𑀭𑀻… 𑁇
