# 𑁨𑁇𑁩 𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑀫𑀸𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑀫𑀸𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(23.)

45\. _“𑀲𑀼𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀲𑀼𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀲𑀸𑀥𑀼𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑀸𑀫𑁆𑀳𑀺 𑀫𑀼𑀲𑀮𑀲𑁆𑀲;_  
_𑀅𑀳𑀺𑀭𑀺𑀓𑁄 𑀫𑁂 𑀙𑀢𑁆𑀢𑀓𑀁 𑀯𑀸𑀧𑀺,_  
_𑀉𑀓𑁆𑀔𑀮𑀺𑀓𑀸 𑀫𑁂 𑀤𑁂𑀟𑁆𑀟𑀼𑀪𑀁 𑀯𑀸𑀢𑀺𑁇_  


(24.)

46\. _𑀭𑀸𑀕𑀜𑁆𑀘 𑀅𑀳𑀁 𑀤𑁄𑀲𑀜𑁆𑀘,_  
_𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀺 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀻𑀢𑀺 𑀯𑀺𑀳𑀦𑀸𑀫𑀺;_  
_𑀲𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀫𑀼𑀧𑀕𑀫𑁆𑀫,_  
_𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆𑀢𑀺 𑀲𑀼𑀔𑀢𑁄 𑀛𑀸𑀬𑀸𑀫𑀻”𑀢𑀺𑁇_  


---

47\. …  𑀲𑀼𑀫𑀗𑁆𑀕𑀮𑀫𑀸𑀢𑀸 𑀣𑁂𑀭𑀻𑁇
