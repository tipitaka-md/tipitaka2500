# 𑁨𑁇𑁫 𑀘𑀺𑀢𑁆𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀘𑀺𑀢𑁆𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(27.)

51\. _“𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀔𑁄𑀫𑁆𑀳𑀺 𑀓𑀺𑀲𑀺𑀓𑀸,_  
_𑀕𑀺𑀮𑀸𑀦𑀸 𑀩𑀸𑀍𑀳𑀤𑀼𑀩𑁆𑀩𑀮𑀸;_  
_𑀤𑀡𑁆𑀟𑀫𑁄𑀮𑀼𑀩𑁆𑀪 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀧𑀩𑁆𑀩𑀢𑀁 𑀅𑀪𑀺𑀭𑀽𑀳𑀺𑀬𑁇_  


(28.)

52\. _𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀢𑁆𑀢𑀓𑀜𑁆𑀘 𑀦𑀺𑀓𑀼𑀚𑁆𑀚𑀺𑀬;_  
_𑀲𑁂𑀮𑁂 𑀔𑀫𑁆𑀪𑁂𑀲𑀺𑀫𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑀁 𑀧𑀤𑀸𑀮𑀺𑀬𑀸”𑀢𑀺𑁇_  


---

53\. …  𑀘𑀺𑀢𑁆𑀢𑀸 𑀣𑁂𑀭𑀻… 𑁇
