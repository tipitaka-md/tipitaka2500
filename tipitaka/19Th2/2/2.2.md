# 𑁨𑁇𑁨 𑀚𑁂𑀦𑁆𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀚𑁂𑀦𑁆𑀢𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(21.)

42\. _“𑀬𑁂 𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸,_  
_𑀫𑀕𑁆𑀕𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀪𑀸𑀯𑀺𑀢𑀸 𑀢𑁂 𑀫𑀬𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀬𑀣𑀸 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀸𑁇_  


(22.)

43\. _𑀤𑀺𑀝𑁆𑀞𑁄 𑀳𑀺 𑀫𑁂 𑀲𑁄 𑀪𑀕𑀯𑀸,_  
_𑀅𑀦𑁆𑀢𑀺𑀫𑁄𑀬𑀁 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑁄;_  
_𑀯𑀺𑀓𑁆𑀔𑀻𑀡𑁄 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇_  


44\. 𑀇𑀢𑁆𑀣𑀁 𑀲𑀼𑀤𑀁 𑀚𑁂𑀦𑁆𑀢𑀸 𑀣𑁂𑀭𑀻 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇
