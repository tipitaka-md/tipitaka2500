# 𑁨𑁇𑁧𑁦 𑀲𑀸𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀸𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(37.)

66\. _“𑀘𑀢𑀼𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀜𑁆𑀘𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁,_  
_𑀯𑀺𑀳𑀸𑀭𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀁;_  
_𑀅𑀮𑀤𑁆𑀥𑀸 𑀘𑁂𑀢𑀲𑁄 𑀲𑀦𑁆𑀢𑀺𑀁,_  
_𑀘𑀺𑀢𑁆𑀢𑁂 𑀅𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀦𑀻;_  
_𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀅𑀝𑁆𑀞𑀫𑀻 𑀭𑀢𑁆𑀢𑀺,_  
_𑀬𑀢𑁄 𑀢𑀡𑁆𑀳𑀸 𑀲𑀫𑀽𑀳𑀢𑀸𑁇_  


(38.)

67\. _𑀩𑀳𑀽𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀳𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀭𑀢𑀸𑀬 𑀫𑁂;_  
_𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

68\. …  𑀲𑀸𑀫𑀸 𑀣𑁂𑀭𑀻… 𑁇
