# 𑁨𑁇𑁯 𑀅𑀪𑀬𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀤𑀼𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀪𑀬𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(35.)

63\. _“𑀅𑀪𑀬𑁂 𑀪𑀺𑀤𑀼𑀭𑁄 𑀓𑀸𑀬𑁄,_  
_𑀬𑀢𑁆𑀣 𑀲𑀢𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸;_  
_𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀸 𑀲𑀢𑀻𑀫𑀢𑀻𑁇_  


(36.)

64\. _𑀩𑀳𑀽𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑀥𑀫𑁆𑀫𑁂𑀳𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑀭𑀢𑀸𑀬 𑀫𑁂;_  
_𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑁄 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_  


---

65\. …  𑀅𑀪𑀬𑀸 𑀣𑁂𑀭𑀻… 𑁇
