# 𑁮𑁇𑁧 𑀲𑀻𑀲𑀽𑀧𑀘𑀸𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀅𑀝𑁆𑀞𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀻𑀲𑀽𑀧𑀘𑀸𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(196.)

264\. _“𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁂𑀲𑀼 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑀸;_  
_𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑁂 𑀧𑀤𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀅𑀲𑁂𑀘𑀦𑀓𑀫𑁄𑀚𑀯𑀁”𑁇_  


(197.)

265\. _“𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀘 𑀬𑀸𑀫𑀸 𑀘,_  
_𑀢𑀼𑀲𑀺𑀢𑀸 𑀘𑀸𑀧𑀺 𑀤𑁂𑀯𑀢𑀸;_  
_𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀺𑀦𑁄 𑀤𑁂𑀯𑀸,_  
_𑀬𑁂 𑀤𑁂𑀯𑀸 𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀦𑁄;_  
_𑀢𑀢𑁆𑀣 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀡𑀻𑀥𑁂𑀳𑀺,_  
_𑀬𑀢𑁆𑀣 𑀢𑁂 𑀯𑀼𑀲𑀺𑀢𑀁 𑀧𑀼𑀭𑁂”𑁇_  


(198.)

266\. _“𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀘 𑀬𑀸𑀫𑀸 𑀘,_  
_𑀢𑀼𑀲𑀺𑀢𑀸 𑀘𑀸𑀧𑀺 𑀤𑁂𑀯𑀢𑀸;_  
_𑀦𑀺𑀫𑁆𑀫𑀸𑀦𑀭𑀢𑀺𑀦𑁄 𑀤𑁂𑀯𑀸,_  
_𑀬𑁂 𑀤𑁂𑀯𑀸 𑀯𑀲𑀯𑀢𑁆𑀢𑀺𑀦𑁄𑁇_  


(199.)

267\. _𑀓𑀸𑀮𑀁 𑀓𑀸𑀮𑀁 𑀪𑀯𑀸𑀪𑀯𑀁,_  
_𑀲𑀓𑁆𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀸;_  
_𑀅𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑀸 𑀲𑀓𑁆𑀓𑀸𑀬𑀁,_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀲𑀸𑀭𑀺𑀦𑁄𑁇_  


(200.)

268\. _𑀲𑀩𑁆𑀩𑁄 𑀆𑀤𑀻𑀧𑀺𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀲𑀩𑁆𑀩𑁄 𑀮𑁄𑀓𑁄 𑀧𑀤𑀻𑀧𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑁄 𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀲𑀩𑁆𑀩𑁄 𑀮𑁄𑀓𑁄 𑀧𑀓𑀫𑁆𑀧𑀺𑀢𑁄𑁇_  


(201.)

269\. _𑀅𑀓𑀫𑁆𑀧𑀺𑀬𑀁 𑀅𑀢𑀼𑀮𑀺𑀬𑀁,_  
_𑀅𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑁄 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀢𑀢𑁆𑀣 𑀫𑁂 𑀦𑀺𑀭𑀢𑁄 𑀫𑀦𑁄𑁇_  


(202.)

270\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀺𑀁 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑀸;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(203.)

271\. _𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀳𑀢𑀸 𑀦𑀦𑁆𑀤𑀻,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁄 𑀧𑀤𑀸𑀮𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦𑀺𑀳𑀢𑁄 𑀢𑁆𑀯𑀫𑀲𑀺 𑀅𑀦𑁆𑀢𑀓”𑁇_  


---

272\. …  𑀲𑀻𑀲𑀽𑀧𑀘𑀸𑀮𑀸 𑀣𑁂𑀭𑀻… 𑁇



---
