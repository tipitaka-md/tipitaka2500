# 𑁩𑁇𑁮 𑀲𑁄𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑁄𑀫𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(60.)

98\. _“𑀬𑀁 𑀢𑀁 𑀇𑀲𑀻𑀳𑀺 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀞𑀸𑀦𑀁 𑀤𑀼𑀭𑀪𑀺𑀲𑀫𑁆𑀪𑀯𑀁;_  
_𑀦 𑀢𑀁 𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀓𑁆𑀓𑀸 𑀧𑀧𑁆𑀧𑁄𑀢𑀼𑀫𑀺𑀢𑁆𑀣𑀺𑀬𑀸”𑁇_  


(61.)

99\. _“𑀇𑀢𑁆𑀣𑀺𑀪𑀸𑀯𑁄 𑀦𑁄 𑀓𑀺𑀁 𑀓𑀬𑀺𑀭𑀸,_  
_𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁂;_  
_𑀜𑀸𑀡𑀫𑁆𑀳𑀺 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀲𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀧𑀲𑁆𑀲𑀢𑁄𑁇_  


(62.)

100\. _𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀳𑀢𑀸 𑀦𑀦𑁆𑀤𑀻,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁄 𑀧𑀤𑀸𑀮𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦𑀺𑀳𑀢𑁄 𑀢𑁆𑀯𑀫𑀲𑀺 𑀅𑀦𑁆𑀢𑀓𑀸”𑀢𑀺𑁇_  


---

101\. …  𑀲𑁄𑀫𑀸 𑀣𑁂𑀭𑀻… 𑁇
