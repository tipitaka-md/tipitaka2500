# 𑁩𑁇𑁭 𑀲𑁂𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑁂𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(57.)

94\. _“𑀦𑀢𑁆𑀣𑀺 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀁 𑀮𑁄𑀓𑁂,_  
_𑀓𑀺𑀁 𑀯𑀺𑀯𑁂𑀓𑁂𑀦 𑀓𑀸𑀳𑀲𑀺;_  
_𑀪𑀼𑀜𑁆𑀚𑀸𑀳𑀺 𑀓𑀸𑀫𑀭𑀢𑀺𑀬𑁄,_  
_𑀫𑀸𑀳𑀼 𑀧𑀘𑁆𑀙𑀸𑀦𑀼𑀢𑀸𑀧𑀺𑀦𑀻”𑁇_  


(58.)

95\. _“𑀲𑀢𑁆𑀢𑀺𑀲𑀽𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸,_  
_𑀔𑀦𑁆𑀥𑀸𑀲𑀁 𑀅𑀥𑀺𑀓𑀼𑀝𑁆𑀝𑀦𑀸;_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 ‘𑀓𑀸𑀫𑀭𑀢𑀺𑀁’ 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_‘𑀅𑀭𑀢𑀻’ 𑀤𑀸𑀦𑀺 𑀲𑀸 𑀫𑀫𑁇_  


(59.)

96\. _𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀳𑀢𑀸 𑀦𑀦𑁆𑀤𑀻,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁄 𑀧𑀤𑀸𑀮𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦𑀺𑀳𑀢𑁄 𑀢𑁆𑀯𑀫𑀲𑀺 𑀅𑀦𑁆𑀢𑀓𑀸”𑀢𑀺𑁇_  


---

97\. …  𑀲𑁂𑀮𑀸 𑀣𑁂𑀭𑀻… 𑁇
