# 𑁩𑁇𑁬 𑀲𑀼𑀓𑁆𑀓𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀢𑀺𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀲𑀼𑀓𑁆𑀓𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(54.)

90\. _“𑀓𑀺𑀁𑀫𑁂 𑀓𑀢𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸,_  
_𑀫𑀥𑀼𑀁 𑀧𑀻𑀢𑀸𑀯 𑀅𑀘𑁆𑀙𑀭𑁂;_  
_𑀬𑁂 𑀲𑀼𑀓𑁆𑀓𑀁 𑀦 𑀉𑀧𑀸𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀲𑀦𑀁𑁇_  


(55.)

91\. _𑀢𑀜𑁆𑀘 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀸𑀦𑀻𑀬𑀁,_  
_𑀅𑀲𑁂𑀘𑀦𑀓𑀫𑁄𑀚𑀯𑀁;_  
_𑀧𑀺𑀯𑀦𑁆𑀢𑀺 𑀫𑀜𑁆𑀜𑁂 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀮𑀸𑀳𑀓𑀫𑀺𑀯𑀤𑁆𑀥𑀕𑀽𑁇_  


(56.)

92\. _𑀲𑀼𑀓𑁆𑀓𑀸 𑀲𑀼𑀓𑁆𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺,_  
_𑀯𑀻𑀢𑀭𑀸𑀕𑀸 𑀲𑀫𑀸𑀳𑀺𑀢𑀸;_  
_𑀥𑀸𑀭𑁂𑀢𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

93\. …  𑀲𑀼𑀓𑁆𑀓𑀸 𑀣𑁂𑀭𑀻… 𑁇
