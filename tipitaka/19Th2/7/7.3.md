# 𑁭𑁇𑁩 𑀉𑀧𑀘𑀸𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀉𑀧𑀘𑀸𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(189.)

255. _“𑀲𑀢𑀺𑀫𑀢𑀻 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑀻,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀸𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸;_  
_𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺 𑀧𑀤𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀅𑀓𑀸𑀧𑀼𑀭𑀺𑀲𑀲𑁂𑀯𑀺𑀢𑀁”𑁇_  


(190.)

256. _“𑀓𑀺𑀁 𑀦𑀼 𑀚𑀸𑀢𑀺𑀁 𑀦 𑀭𑁄𑀘𑁂𑀲𑀺,_  
_𑀚𑀸𑀢𑁄 𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀢𑀺;_  
_𑀪𑀼𑀜𑁆𑀚𑀸𑀳𑀺 𑀓𑀸𑀫𑀭𑀢𑀺𑀬𑁄,_  
_𑀫𑀸𑀳𑀼 𑀧𑀘𑁆𑀙𑀸𑀦𑀼𑀢𑀸𑀧𑀺𑀦𑀻”𑁇_  


(191.)

257. _“𑀚𑀸𑀢𑀲𑁆𑀲 𑀫𑀭𑀡𑀁 𑀳𑁄𑀢𑀺,_  
_𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀸𑀦 𑀙𑁂𑀤𑀦𑀁;_  
_𑀯𑀥𑀩𑀦𑁆𑀥𑀧𑀭𑀺𑀓𑁆𑀮𑁂𑀲𑀁,_  
_𑀚𑀸𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(192.)

258. _𑀅𑀢𑁆𑀣𑀺 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑁂 𑀚𑀸𑀢𑁄,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀅𑀧𑀭𑀸𑀚𑀺𑀢𑁄;_  
_𑀲𑁄 𑀫𑁂 𑀥𑀫𑁆𑀫𑀫𑀤𑁂𑀲𑁂𑀲𑀺,_  
_𑀚𑀸𑀢𑀺𑀬𑀸 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀁𑁇_  


(193.)

259. _𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀁;_  
_𑀅𑀭𑀺𑀬𑀁 𑀘𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀲𑀫𑀕𑀸𑀫𑀺𑀦𑀁𑁇_  


(194.)

260. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀯𑀺𑀳𑀭𑀺𑀁 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑀸;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(195.)

261. _𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀳𑀢𑀸 𑀦𑀦𑁆𑀤𑀻,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁄 𑀧𑀤𑀸𑀮𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦𑀺𑀳𑀢𑁄 𑀢𑁆𑀯𑀫𑀲𑀺 𑀅𑀦𑁆𑀢𑀓”𑁇_  


---

262. …  𑀉𑀧𑀘𑀸𑀮𑀸 𑀣𑁂𑀭𑀻… 𑁇



---

263. 𑀲𑀢𑁆𑀢𑀓𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
