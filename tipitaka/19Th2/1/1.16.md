# 𑁧𑁇𑁧𑁬 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑀼𑀫𑀦𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑀼𑀫𑀦𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(16.)

32\. _“𑀲𑀼𑀔𑀁 𑀢𑁆𑀯𑀁 𑀯𑀼𑀟𑁆𑀠𑀺𑀓𑁂 𑀲𑁂𑀳𑀺,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘𑁄𑀍𑀏𑀦 𑀧𑀸𑀭𑀼𑀢𑀸;_

_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀸𑀲𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸”𑀢𑀺𑁇_
