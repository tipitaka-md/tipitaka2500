# 𑁧𑁇𑁧𑁮 𑀲𑀁𑀖𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀁𑀖𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(18.)

36\. _“𑀳𑀺𑀢𑁆𑀯𑀸 𑀖𑀭𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀲𑀼𑀁 𑀧𑀺𑀬𑀁;_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀕𑀜𑁆𑀘 𑀤𑁄𑀲𑀜𑁆𑀘,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀜𑁆𑀘 𑀯𑀺𑀭𑀸𑀚𑀺𑀬;_  
_𑀲𑀫𑀽𑀮𑀁 𑀢𑀡𑁆𑀳𑀫𑀩𑁆𑀩𑀼𑀬𑁆𑀳,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀸𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸”𑀢𑀺𑁇_  


---

37\. …  𑀲𑀁𑀖𑀸 𑀣𑁂𑀭𑀻… 𑁇



---

38\. 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
