# 𑁧𑁇𑁭 𑀯𑀻𑀭𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀯𑀻𑀭𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(7.)

14. _“𑀯𑀻𑀭𑀸 𑀯𑀻𑀭𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀸𑀯𑀺𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸;_  
_𑀥𑀸𑀭𑁂𑀢𑀺 𑀅𑀦𑁆𑀢𑀺𑀫𑀁 𑀤𑁂𑀳𑀁,_  
_𑀚𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑀁 𑀲𑀯𑀸𑀳𑀺𑀦𑀺𑀦𑁆”𑀢𑀺𑁇_  


---

15. …  𑀯𑀻𑀭𑀸 𑀣𑁂𑀭𑀻… 𑁇
