# 𑁧𑁇𑁫 𑀅𑀜𑁆𑀜𑀢𑀭𑀸𑀢𑀺𑀲𑁆𑀲𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀅𑀜𑁆𑀜𑀢𑀭𑀸𑀢𑀺𑀲𑁆𑀲𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(5.)

10. _“𑀢𑀺𑀲𑁆𑀲𑁂 𑀬𑀼𑀜𑁆𑀚𑀲𑁆𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀳𑀺,_  
_𑀔𑀡𑁄 𑀢𑀁 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀔𑀡𑀸𑀢𑀻𑀢𑀸 𑀳𑀺 𑀲𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸”𑀢𑀺𑁇_  


---

11. …  𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀢𑀺𑀲𑁆𑀲𑀸 𑀣𑁂𑀭𑀻… 𑁇
