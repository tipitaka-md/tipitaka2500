# 𑁧𑁇𑁯 𑀪𑀤𑁆𑀭𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀏𑀓𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀪𑀤𑁆𑀭𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(9.)

18\. _“𑀲𑀤𑁆𑀥𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀪𑀤𑁆𑀭𑁂 𑀪𑀤𑁆𑀭𑀭𑀢𑀸 𑀪𑀯;_  
_𑀪𑀸𑀯𑁂𑀳𑀺 𑀓𑀼𑀲𑀮𑁂 𑀥𑀫𑁆𑀫𑁂,_  
_𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀫𑀦𑀼𑀢𑁆𑀢𑀭𑀦𑁆”𑀢𑀺𑁇_  


---

19\. …  𑀪𑀤𑁆𑀭𑀸 𑀣𑁂𑀭𑀻… 𑁇
