# 𑁧𑁧𑁇𑁧 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀤𑁆𑀯𑀸𑀤𑀲𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(224.)

298\. _“𑀉𑀪𑁄 𑀫𑀸𑀢𑀸 𑀘 𑀥𑀻𑀢𑀸 𑀘,_  
_𑀫𑀬𑀁 𑀆𑀲𑀼𑀁 𑀲𑀧𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀅𑀳𑀼 𑀲𑀁𑀯𑁂𑀕𑁄,_  
_𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑁄𑁇_  


(225.)

299\. _𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀓𑀸𑀫𑀸 𑀅𑀲𑀼𑀘𑀻,_  
_𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀸 𑀩𑀳𑀼𑀓𑀡𑁆𑀝𑀓𑀸;_  
_𑀬𑀢𑁆𑀣 𑀫𑀸𑀢𑀸 𑀘 𑀥𑀻𑀢𑀸 𑀘,_  
_𑀲𑀪𑀭𑀺𑀬𑀸 𑀫𑀬𑀁 𑀅𑀳𑀼𑀁𑁇_  


(226.)

300\. _𑀓𑀸𑀫𑁂𑀲𑁆𑀯𑀸𑀤𑀻𑀦𑀯𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀁 𑀤𑀝𑁆𑀞𑀼 𑀔𑁂𑀫𑀢𑁄;_  
_𑀲𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀺𑀁 𑀭𑀸𑀚𑀕𑀳𑁂,_  
_𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(227.)

301\. _𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁;_  
_𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀘𑁆𑀘𑀜𑀸𑀡𑀜𑁆𑀘,_  
_𑀲𑁄𑀢𑀥𑀸𑀢𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀸𑁇_  


(228.)

302\. _𑀇𑀤𑁆𑀥𑀻𑀧𑀺 𑀫𑁂 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀧𑀢𑁆𑀢𑁄 𑀫𑁂 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑁄;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(229.)

303\. _𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑁆𑀯𑀸,_  
_𑀘𑀢𑀼𑀭𑀲𑁆𑀲𑀁 𑀭𑀣𑀁 𑀅𑀳𑀁;_  
_𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀦𑁄”𑁇_  


(230.)

304\. _“𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀕𑁆𑀕𑀁 𑀉𑀧𑀕𑀫𑁆𑀫 𑀧𑀸𑀤𑀧𑀁,_  
_𑀏𑀓𑀸 𑀢𑀼𑀯𑀁 𑀢𑀺𑀝𑁆𑀞𑀲𑀺 𑀲𑀸𑀮𑀫𑀽𑀮𑁂;_  
_𑀦 𑀘𑀸𑀧𑀺 𑀢𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑁄𑀘𑀺,_  
_𑀩𑀸𑀮𑁂 𑀦 𑀢𑁆𑀯𑀁 𑀪𑀸𑀬𑀲𑀺 𑀥𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁”𑁇_  


(231.)

305\. _“𑀲𑀢𑀁 𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺𑀧𑀺 𑀥𑀼𑀢𑁆𑀢𑀓𑀸𑀦𑀁,_  
_𑀲𑀫𑀸𑀕𑀢𑀸 𑀏𑀤𑀺𑀲𑀓𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁;_  
_𑀮𑁄𑀫𑀁 𑀦 𑀇𑀜𑁆𑀚𑁂 𑀦𑀧𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑁂,_  
_𑀓𑀺𑀁 𑀫𑁂 𑀢𑀼𑀯𑀁 𑀫𑀸𑀭 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑁂𑀓𑁄𑁇_  


(232.)

306\. _𑀏𑀲𑀸 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀸𑀫𑀺,_  
_𑀓𑀼𑀘𑁆𑀙𑀺𑀁 𑀯𑀸 𑀧𑀯𑀺𑀲𑀸𑀫𑀺 𑀢𑁂;_  
_𑀪𑀫𑀼𑀓𑀦𑁆𑀢𑀭𑁂 𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺,_  
_𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑀁 𑀫𑀁 𑀦 𑀤𑀓𑁆𑀔𑀲𑀺𑁇_  


(233.)

307\. _𑀘𑀺𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀯𑀲𑀻𑀪𑀽𑀢𑀸𑀳𑀁,_  
_𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀸;_  
_𑀙𑀍𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


(234.)

308\. _𑀲𑀢𑁆𑀢𑀺𑀲𑀽𑀮𑀽𑀧𑀫𑀸 𑀓𑀸𑀫𑀸,_  
_𑀔𑀦𑁆𑀥𑀸𑀲𑀁 𑀅𑀥𑀺𑀓𑀼𑀝𑁆𑀝𑀦𑀸;_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 ‘𑀓𑀸𑀫𑀭𑀢𑀺𑀁’ 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_‘𑀆𑀭𑀢𑀻’ 𑀤𑀸𑀦𑀺 𑀲𑀸 𑀫𑀫𑁇_  


(235.)

309\. _𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀯𑀺𑀳𑀢𑀸 𑀦𑀦𑁆𑀤𑀻,_  
_𑀢𑀫𑁄𑀔𑀦𑁆𑀥𑁄 𑀧𑀤𑀸𑀮𑀺𑀢𑁄;_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀧𑀸𑀧𑀺𑀫,_  
_𑀦𑀺𑀳𑀢𑁄 𑀢𑁆𑀯𑀫𑀲𑀺 𑀅𑀦𑁆𑀢𑀓𑀸”𑀢𑀺𑁇_  


---

310\. …  𑀉𑀧𑁆𑀧𑀮𑀯𑀡𑁆𑀡𑀸 𑀣𑁂𑀭𑀻… 𑁇



---
