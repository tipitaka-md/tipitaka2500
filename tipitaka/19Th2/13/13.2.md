# 𑁧𑁩𑁇𑁨 𑀭𑁄𑀳𑀺𑀦𑀻𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

### 𑀭𑁄𑀳𑀺𑀦𑀻𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(271.)

350\. _“‘𑀲𑀫𑀡𑀸’𑀢𑀺 𑀪𑁄𑀢𑀺 𑀲𑀼𑀧𑀺,_  
_‘𑀲𑀫𑀡𑀸’𑀢𑀺 𑀧𑀩𑀼𑀚𑁆𑀛𑀲𑀺;_  
_𑀲𑀫𑀡𑀸𑀦𑁂𑀯 𑀓𑀺𑀢𑁆𑀢𑁂𑀲𑀺,_  
_𑀲𑀫𑀡𑀻 𑀦𑀽𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(272.)

351\. _𑀯𑀺𑀧𑀼𑀮𑀁 𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀸𑀦𑀜𑁆𑀘,_  
_𑀲𑀫𑀡𑀸𑀦𑀁 𑀧𑀯𑁂𑀘𑁆𑀘𑀲𑀺;_  
_𑀭𑁄𑀳𑀺𑀦𑀻 𑀤𑀸𑀦𑀺 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀓𑁂𑀦 𑀢𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(273.)

352\. _𑀅𑀓𑀫𑁆𑀫𑀓𑀸𑀫𑀸 𑀅𑀮𑀲𑀸,_  
_𑀧𑀭𑀤𑀢𑁆𑀢𑀽𑀧𑀚𑀻𑀯𑀺𑀦𑁄;_  
_𑀆𑀲𑀁𑀲𑀼𑀓𑀸 𑀲𑀸𑀤𑀼𑀓𑀸𑀫𑀸,_  
_𑀓𑁂𑀦 𑀢𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸”𑁇_  


(274.)

353\. _“𑀘𑀺𑀭𑀲𑁆𑀲𑀁 𑀯𑀢 𑀫𑀁 𑀢𑀸𑀢,_  
_𑀲𑀫𑀡𑀸𑀦𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀲𑀺;_  
_𑀢𑁂𑀲𑀁 𑀢𑁂 𑀓𑀺𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀧𑀜𑁆𑀜𑀸𑀲𑀻𑀮𑀧𑀭𑀓𑁆𑀓𑀫𑀁𑁇_  


(275.)

354\. _𑀓𑀫𑁆𑀫𑀓𑀸𑀫𑀸 𑀅𑀦𑀮𑀲𑀸,_  
_𑀓𑀫𑁆𑀫𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀸𑀭𑀓𑀸;_  
_𑀭𑀸𑀕𑀁 𑀤𑁄𑀲𑀁 𑀧𑀚𑀳𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(276.)

355\. _𑀢𑀻𑀡𑀺 𑀧𑀸𑀧𑀲𑁆𑀲 𑀫𑀽𑀮𑀸𑀦𑀺,_  
_𑀥𑀼𑀦𑀦𑁆𑀢𑀺 𑀲𑀼𑀘𑀺𑀓𑀸𑀭𑀺𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑀁 𑀧𑀸𑀧𑀁 𑀧𑀳𑀻𑀦𑁂𑀲𑀁,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(277.)

356\. _𑀓𑀸𑀬𑀓𑀫𑁆𑀫𑀁 𑀲𑀼𑀘𑀺 𑀦𑁂𑀲𑀁,_  
_𑀯𑀘𑀻𑀓𑀫𑁆𑀫𑀜𑁆𑀘 𑀢𑀸𑀤𑀺𑀲𑀁;_  
_𑀫𑀦𑁄𑀓𑀫𑁆𑀫𑀁 𑀲𑀼𑀘𑀺 𑀦𑁂𑀲𑀁,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(278.)

357\. _𑀯𑀺𑀫𑀮𑀸 𑀲𑀗𑁆𑀔𑀫𑀼𑀢𑁆𑀢𑀸𑀯,_  
_𑀲𑀼𑀤𑁆𑀥𑀸 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑀸;_  
_𑀧𑀼𑀡𑁆𑀡𑀸 𑀲𑀼𑀓𑁆𑀓𑀸𑀦 𑀥𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(279.)

358\. _𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸,_  
_𑀅𑀭𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀚𑀻𑀯𑀺𑀦𑁄;_  
_𑀅𑀢𑁆𑀣𑀁 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(280.)

359\. _𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸,_  
_𑀅𑀭𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀚𑀻𑀯𑀺𑀦𑁄;_  
_𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(281.)

360\. _𑀤𑀽𑀭𑀗𑁆𑀕𑀫𑀸 𑀲𑀢𑀺𑀫𑀦𑁆𑀢𑁄,_  
_𑀫𑀦𑁆𑀢𑀪𑀸𑀡𑀻 𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑀸;_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(282.)

361\. _𑀬𑀲𑁆𑀫𑀸 𑀕𑀸𑀫𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺,_  
_𑀦 𑀯𑀺𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀁;_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑀸𑀯 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(283.)

362\. _𑀦 𑀢𑁂 𑀲𑀁 𑀓𑁄𑀝𑁆𑀞𑁂 𑀑𑀧𑁂𑀦𑁆𑀢𑀺,_  
_𑀦 𑀓𑀼𑀫𑁆𑀪𑀺𑀁 𑀦 𑀔𑀍𑀑𑀧𑀺𑀬𑀁;_  
_𑀧𑀭𑀺𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀫𑁂𑀲𑀸𑀦𑀸,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(284.)

363\. _𑀦 𑀢𑁂 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺,_  
_𑀦 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀦 𑀭𑀽𑀧𑀺𑀬𑀁;_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀬𑀸𑀧𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸𑁇_  


(285.)

364\. _𑀦𑀸𑀦𑀸𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸,_  
_𑀦𑀸𑀦𑀸𑀚𑀦𑀧𑀤𑁂𑀳𑀺 𑀘;_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀺𑀬𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀫𑁂 𑀲𑀫𑀡𑀸 𑀧𑀺𑀬𑀸”𑁇_  


(286.)

365\. _“𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀢 𑀦𑁄 𑀪𑁄𑀢𑀺,_  
_𑀓𑀼𑀮𑁂 𑀚𑀸𑀢𑀸𑀲𑀺 𑀭𑁄𑀳𑀺𑀦𑀻;_  
_𑀲𑀤𑁆𑀥𑀸 𑀩𑀼𑀤𑁆𑀥𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘,_  
_𑀲𑀁𑀖𑁂 𑀘 𑀢𑀺𑀩𑁆𑀩𑀕𑀸𑀭𑀯𑀸𑁇_  


(287.)

366\. _𑀢𑀼𑀯𑀁 𑀳𑁂𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀅𑀫𑁆𑀳𑀫𑁆𑀧𑀺 𑀏𑀢𑁂 𑀲𑀫𑀡𑀸,_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁𑁇_  


(288.)

367\. _𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁂𑀢𑁆𑀣 𑀬𑀜𑁆𑀜𑁄,_  
_𑀯𑀺𑀧𑀼𑀮𑁄 𑀦𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺”;_  
_“𑀲𑀘𑁂 𑀪𑀸𑀬𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀲𑀘𑁂 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀫𑀧𑁆𑀧𑀺𑀬𑀁𑁇_  


(289.)

368\. _𑀉𑀧𑁂𑀳𑀺 𑀲𑀭𑀡𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀖𑀜𑁆𑀘 𑀢𑀸𑀤𑀺𑀦𑀁;_  
_𑀲𑀫𑀸𑀤𑀺𑀬𑀸𑀳𑀺 𑀲𑀻𑀮𑀸𑀦𑀺,_  
_𑀢𑀁 𑀢𑁂 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑁂𑀳𑀺𑀢𑀺”𑁇_  


(290.)

369\. _“𑀉𑀧𑁂𑀫𑀺 𑀲𑀭𑀡𑀁 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀁𑀖𑀜𑁆𑀘 𑀢𑀸𑀤𑀺𑀦𑀁;_  
_𑀲𑀫𑀸𑀤𑀺𑀬𑀸𑀫𑀺 𑀲𑀻𑀮𑀸𑀦𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑁂𑀳𑀺𑀢𑀺𑁇_  


(291.)

370\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀩𑀦𑁆𑀥𑀼 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀲𑁄 𑀇𑀤𑀸𑀦𑀺𑀫𑁆𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀢𑁂𑀯𑀺𑀚𑁆𑀚𑁄 𑀲𑁄𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀫𑁆𑀳𑀺,_  
_𑀯𑁂𑀤𑀕𑀽 𑀘𑀫𑁆𑀳𑀺 𑀦𑁆𑀳𑀸𑀢𑀓𑁄”𑁇_  


---

371\. …  𑀭𑁄𑀳𑀺𑀦𑀻 𑀣𑁂𑀭𑀻… 𑁇
