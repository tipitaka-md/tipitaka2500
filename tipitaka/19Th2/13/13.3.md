# 𑁧𑁩𑁇𑁩 𑀘𑀸𑀧𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀯𑀻𑀲𑀢𑀺𑀦𑀺𑀧𑀸𑀢

### 𑀘𑀸𑀧𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(292.)

372\. _“𑀮𑀝𑁆𑀞𑀺𑀳𑀢𑁆𑀣𑁄 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺,_  
_𑀲𑁄 𑀤𑀸𑀦𑀺 𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑀓𑁄;_  
_𑀆𑀲𑀸𑀬 𑀧𑀮𑀺𑀧𑀸 𑀖𑁄𑀭𑀸,_  
_𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀸𑀭𑀫𑁂𑀢𑀯𑁂𑁇_  


(293.)

373\. _𑀲𑀼𑀫𑀢𑁆𑀢𑀁 𑀫𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑀸,_  
_𑀘𑀸𑀧𑀸 𑀧𑀼𑀢𑁆𑀢𑀫𑀢𑁄𑀲𑀬𑀺;_  
_𑀘𑀸𑀧𑀸𑀬 𑀩𑀦𑁆𑀥𑀦𑀁 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀁 𑀧𑀼𑀦𑁄𑀧𑀳𑀁”𑁇_  


(294.)

374\. _“𑀫𑀸 𑀫𑁂 𑀓𑀼𑀚𑁆𑀛𑀺 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀫𑀸 𑀫𑁂 𑀓𑀼𑀚𑁆𑀛𑀺 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀦 𑀳𑀺 𑀓𑁄𑀥𑀧𑀭𑁂𑀢𑀲𑁆𑀲,_  
_𑀲𑀼𑀤𑁆𑀥𑀺 𑀅𑀢𑁆𑀣𑀺 𑀓𑀼𑀢𑁄 𑀢𑀧𑁄”𑁇_  


(295.)

375\. _“𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀜𑁆𑀘 𑀦𑀸𑀍𑀆𑀢𑁄,_  
_𑀓𑁄𑀥 𑀦𑀸𑀍𑀆𑀬 𑀯𑀘𑁆𑀙𑀢𑀺;_  
_𑀩𑀦𑁆𑀥𑀦𑁆𑀢𑀻 𑀇𑀢𑁆𑀣𑀺𑀭𑀽𑀧𑁂𑀦,_  
_𑀲𑀫𑀡𑁂 𑀥𑀫𑁆𑀫𑀚𑀻𑀯𑀺𑀦𑁄”𑁇_  


(296.)

376\. _“𑀏𑀳𑀺 𑀓𑀸𑀍𑀅 𑀦𑀺𑀯𑀢𑁆𑀢𑀲𑁆𑀲𑀼,_  
_𑀪𑀼𑀜𑁆𑀚 𑀓𑀸𑀫𑁂 𑀬𑀣𑀸 𑀧𑀼𑀭𑁂;_  
_𑀅𑀳𑀜𑁆𑀘 𑀢𑁂 𑀯𑀲𑀻𑀓𑀢𑀸,_  
_𑀬𑁂 𑀘 𑀫𑁂 𑀲𑀦𑁆𑀢𑀺 𑀜𑀸𑀢𑀓𑀸”𑁇_  


(297.)

377\. _“𑀏𑀢𑁆𑀢𑁄 𑀘𑀸𑀧𑁂 𑀘𑀢𑀼𑀩𑁆𑀪𑀸𑀕𑀁,_  
_𑀬𑀣𑀸 𑀪𑀸𑀲𑀲𑀺 𑀢𑁆𑀯𑀜𑁆𑀘 𑀫𑁂;_  
_𑀢𑀬𑀺 𑀭𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑁄𑀲𑀲𑁆𑀲,_  
_𑀉𑀍𑀆𑀭𑀁 𑀯𑀢 𑀢𑀁 𑀲𑀺𑀬𑀸”𑁇_  


(298.)

378\. _“𑀓𑀸𑀍𑀅𑀗𑁆𑀕𑀺𑀦𑀺𑀁𑀯 𑀢𑀓𑁆𑀓𑀸𑀭𑀺𑀁,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀕𑀺𑀭𑀺𑀫𑀼𑀤𑁆𑀥𑀦𑀺;_  
_𑀨𑀼𑀮𑁆𑀮𑀁 𑀤𑀸𑀮𑀺𑀫𑀮𑀝𑁆𑀞𑀺𑀁𑀯,_  
_𑀅𑀦𑁆𑀢𑁄𑀤𑀻𑀧𑁂𑀯 𑀧𑀸𑀝𑀮𑀺𑀁𑁇_  


(299.)

379\. _𑀳𑀭𑀺𑀘𑀦𑁆𑀤𑀦𑀮𑀺𑀢𑁆𑀢𑀗𑁆𑀕𑀺𑀁,_  
_𑀓𑀸𑀲𑀺𑀓𑀼𑀢𑁆𑀢𑀫𑀥𑀸𑀭𑀺𑀦𑀺𑀁;_  
_𑀢𑀁 𑀫𑀁 𑀭𑀽𑀧𑀯𑀢𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀁,_  
_𑀓𑀲𑁆𑀲 𑀑𑀳𑀸𑀬 𑀕𑀘𑁆𑀙𑀲𑀺”𑁇_  


(300.)

380\. _“𑀲𑀸𑀓𑀼𑀦𑁆𑀢𑀺𑀓𑁄𑀯 𑀲𑀓𑀼𑀡𑀺𑀁,_  
_𑀬𑀣𑀸 𑀩𑀦𑁆𑀥𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀢𑀺;_  
_𑀆𑀳𑀭𑀺𑀫𑁂𑀦 𑀭𑀽𑀧𑁂𑀦,_  
_𑀦 𑀫𑀁 𑀢𑁆𑀯𑀁 𑀩𑀸𑀥𑀬𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(301.)

381\. _“𑀇𑀫𑀜𑁆𑀘 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑀨𑀮𑀁,_  
_𑀓𑀸𑀍𑀅 𑀉𑀧𑁆𑀧𑀸𑀤𑀺𑀢𑀁 𑀢𑀬𑀸;_  
_𑀢𑀁 𑀫𑀁 𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀁,_  
_𑀓𑀲𑁆𑀲 𑀑𑀳𑀸𑀬 𑀕𑀘𑁆𑀙𑀲𑀺”𑁇_  


(302.)

382\. _“𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑁂 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑀸,_  
_𑀢𑀢𑁄 𑀜𑀸𑀢𑀻 𑀢𑀢𑁄 𑀥𑀦𑀁;_  
_𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑀸,_  
_𑀦𑀸𑀕𑁄 𑀙𑁂𑀢𑁆𑀯𑀸𑀯 𑀩𑀦𑁆𑀥𑀦𑀁”𑁇_  


(303.)

383\. _“𑀇𑀤𑀸𑀦𑀺 𑀢𑁂 𑀇𑀫𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀤𑀡𑁆𑀟𑁂𑀦 𑀙𑀼𑀭𑀺𑀓𑀸𑀬 𑀯𑀸;_  
_𑀪𑀽𑀫𑀺𑀬𑀁 𑀯𑀸 𑀦𑀺𑀲𑀼𑀫𑁆𑀪𑀺𑀲𑁆𑀲𑀁,_  
_𑀧𑀼𑀢𑁆𑀢𑀲𑁄𑀓𑀸 𑀦 𑀕𑀘𑁆𑀙𑀲𑀺”𑁇_  


(304.)

384\. _“𑀲𑀘𑁂 𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀸𑀦𑀁,_  
_𑀓𑀼𑀓𑁆𑀓𑀼𑀭𑀸𑀦𑀁 𑀧𑀤𑀸𑀳𑀺𑀲𑀺;_  
_𑀦 𑀫𑀁 𑀧𑀼𑀢𑁆𑀢𑀓𑀢𑁆𑀢𑁂 𑀚𑀫𑁆𑀫𑀺,_  
_𑀧𑀼𑀦𑀭𑀸𑀯𑀢𑁆𑀢𑀬𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(305.)

385\. _“𑀳𑀦𑁆𑀤 𑀔𑁄 𑀤𑀸𑀦𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀓𑀼𑀳𑀺𑀁 𑀓𑀸𑀍𑀅 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀺;_  
_𑀓𑀢𑀫𑀁 𑀕𑀸𑀫𑀦𑀺𑀕𑀫𑀁,_  
_𑀦𑀕𑀭𑀁 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑁄”𑁇_  


(306.)

386\. _“𑀅𑀳𑀼𑀫𑁆𑀳 𑀧𑀼𑀩𑁆𑀩𑁂 𑀕𑀡𑀺𑀦𑁄,_  
_𑀅𑀲𑁆𑀲𑀫𑀡𑀸 𑀲𑀫𑀡𑀫𑀸𑀦𑀺𑀦𑁄;_  
_𑀕𑀸𑀫𑁂𑀦 𑀕𑀸𑀫𑀁 𑀯𑀺𑀘𑀭𑀺𑀫𑁆𑀳,_  
_𑀦𑀕𑀭𑁂 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑁄𑁇_  


(307.)

387\. _𑀏𑀲𑁄 𑀳𑀺 𑀪𑀕𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀤𑀺𑀁 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀁 𑀧𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀳𑀸𑀦𑀸𑀬,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑀁 𑀕𑀘𑁆𑀙𑀁,_  
_𑀲𑁄 𑀫𑁂 𑀲𑀢𑁆𑀣𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(308.)

388\. _“𑀯𑀦𑁆𑀤𑀦𑀁 𑀤𑀸𑀦𑀺 𑀯𑀚𑁆𑀚𑀸𑀲𑀺,_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀜𑁆𑀘 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁”𑁇_  


(309.)

389\. _“𑀏𑀢𑀁 𑀔𑁄 𑀮𑀩𑁆𑀪𑀫𑀫𑁆𑀳𑁂𑀳𑀺,_  
_𑀬𑀣𑀸 𑀪𑀸𑀲𑀲𑀺 𑀢𑁆𑀯𑀜𑁆𑀘 𑀫𑁂;_  
_𑀯𑀦𑁆𑀤𑀦𑀁 𑀤𑀸𑀦𑀺 𑀢𑁂 𑀯𑀚𑁆𑀚𑀁,_  
_𑀮𑁄𑀓𑀦𑀸𑀣𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀜𑁆𑀘 𑀓𑀢𑁆𑀯𑀸𑀦,_  
_𑀆𑀤𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀓𑁆𑀔𑀺𑀡𑀁”𑁇_  


(310.)

390\. _𑀢𑀢𑁄 𑀘 𑀓𑀸𑀍𑀑 𑀧𑀓𑁆𑀓𑀸𑀫𑀺,_  
_𑀦𑀤𑀺𑀁 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀁 𑀧𑀢𑀺;_  
_𑀲𑁄 𑀅𑀤𑁆𑀤𑀲𑀸𑀲𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀁 𑀅𑀫𑀢𑀁 𑀧𑀤𑀁𑁇_  


(311.)

391\. _𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀘 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀁;_  
_𑀅𑀭𑀺𑀬𑀁 𑀘𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑀁 𑀫𑀕𑁆𑀕𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀲𑀫𑀕𑀸𑀫𑀺𑀦𑀁𑁇_  


(312.)

392\. _𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑀸𑀦𑀺 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀢𑁆𑀯𑀸𑀦 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀘𑀸𑀧𑀸𑀬 𑀆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁𑁇_  


---

393\. …  𑀘𑀸𑀧𑀸 𑀣𑁂𑀭𑀻… 𑁇
