# 𑁫𑁇𑁭 𑀲𑀓𑀼𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀓𑀼𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(97.)

145\. _“𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀲𑀦𑁆𑀢𑀻𑀳𑀁,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀅𑀤𑁆𑀤𑀲𑀁 𑀯𑀺𑀭𑀚𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀧𑀤𑀫𑀘𑁆𑀘𑀼𑀢𑀁𑁇_  


(98.)

146\. _𑀲𑀸𑀳𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀥𑀻𑀢𑀭𑀜𑁆𑀘,_  
_𑀥𑀦𑀥𑀜𑁆𑀜𑀜𑁆𑀘 𑀙𑀟𑁆𑀟𑀺𑀬;_  
_𑀓𑁂𑀲𑁂 𑀙𑁂𑀤𑀸𑀧𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁𑁇_  


(99.)

147\. _𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀅𑀳𑀁 𑀲𑀦𑁆𑀢𑀻,_  
_𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀻 𑀫𑀕𑁆𑀕𑀫𑀜𑁆𑀚𑀲𑀁;_  
_𑀧𑀳𑀸𑀲𑀺𑀁 𑀭𑀸𑀕𑀤𑁄𑀲𑀜𑁆𑀘,_  
_𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑁂 𑀘 𑀆𑀲𑀯𑁂𑁇_  


(100.)

148\. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚,_  
_𑀧𑀼𑀩𑁆𑀩𑀚𑀸𑀢𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀁;_  
_𑀤𑀺𑀩𑁆𑀩𑀘𑀓𑁆𑀔𑀼 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁,_  
_𑀯𑀺𑀫𑀮𑀁 𑀲𑀸𑀥𑀼𑀪𑀸𑀯𑀺𑀢𑀁𑁇_  


(101.)

149\. _𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀧𑀭𑀢𑁄 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀳𑁂𑀢𑀼𑀚𑀸𑀢𑁂 𑀧𑀮𑁄𑀓𑀺𑀢𑁂;_  
_𑀧𑀳𑀸𑀲𑀺𑀁 𑀆𑀲𑀯𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀸𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸”𑀢𑀺𑁇_  


---

150\. …  𑀲𑀓𑀼𑀮𑀸 𑀣𑁂𑀭𑀻… 𑁇
