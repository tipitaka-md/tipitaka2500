# 𑁫𑁇𑁪 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(82.)

127\. _“𑀆𑀢𑀼𑀭𑀁 𑀅𑀲𑀼𑀘𑀺𑀁 𑀧𑀽𑀢𑀺𑀁,_  
_𑀧𑀲𑁆𑀲 𑀦𑀦𑁆𑀤𑁂 𑀲𑀫𑀼𑀲𑁆𑀲𑀬𑀁;_  
_𑀅𑀲𑀼𑀪𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀳𑀺,_  
_𑀏𑀓𑀕𑁆𑀕𑀁 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑀁𑁇_  


(83.)

128\. _𑀬𑀣𑀸 𑀇𑀤𑀁 𑀢𑀣𑀸 𑀏𑀢𑀁,_  
_𑀬𑀣𑀸 𑀏𑀢𑀁 𑀢𑀣𑀸 𑀇𑀤𑀁;_  
_𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀧𑀽𑀢𑀺𑀓𑀁 𑀯𑀸𑀢𑀺,_  
_𑀩𑀸𑀮𑀸𑀦𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑀁𑁇_  


(84.)

129\. _𑀏𑀯𑀫𑁂𑀢𑀁 𑀅𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀻,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑀸;_  
_𑀢𑀢𑁄 𑀲𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬,_  
_𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀺𑀚𑁆𑀛 𑀤𑀓𑁆𑀔𑀺𑀲𑀁”𑁇_  


(85.)

130\. _“𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸𑀬,_  
_𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺𑀬𑀸 𑀬𑁄𑀦𑀺𑀲𑁄;_  
_𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀬𑀁 𑀓𑀸𑀬𑁄,_  
_𑀤𑀺𑀝𑁆𑀞𑁄 𑀲𑀦𑁆𑀢𑀭𑀩𑀸𑀳𑀺𑀭𑁄𑁇_  


(86.)

131\. _𑀅𑀣 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀳𑀁 𑀓𑀸𑀬𑁂,_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀜𑁆𑀘 𑀯𑀺𑀭𑀚𑁆𑀚𑀳𑀁;_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀸𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸”𑀢𑀺𑁇_  


---

132\. …  𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀣𑁂𑀭𑀻… 𑁇
