# 𑁫𑁇𑁫 𑀦𑀦𑁆𑀤𑀼𑀢𑁆𑀢𑀭𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

### 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

### 𑀦𑀦𑁆𑀤𑀼𑀢𑁆𑀢𑀭𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(87.)

133\. _“𑀅𑀕𑁆𑀕𑀺𑀁 𑀘𑀦𑁆𑀤𑀜𑁆𑀘 𑀲𑀽𑀭𑀺𑀬𑀜𑁆𑀘,_  
_𑀤𑁂𑀯𑀢𑀸 𑀘 𑀦𑀫𑀲𑁆𑀲𑀺𑀳𑀁;_  
_𑀦𑀤𑀻𑀢𑀺𑀢𑁆𑀣𑀸𑀦𑀺 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦,_  
_𑀉𑀤𑀓𑀁 𑀑𑀭𑀼𑀳𑀸𑀫𑀺𑀳𑀁𑁇_  


(88.)

134\. _𑀩𑀳𑀽𑀯𑀢𑀲𑀫𑀸𑀤𑀸𑀦𑀸,_  
_𑀅𑀟𑁆𑀠𑀁 𑀲𑀻𑀲𑀲𑁆𑀲 𑀑𑀮𑀺𑀔𑀺𑀁;_  
_𑀙𑀫𑀸𑀬 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀫𑀺,_  
_𑀭𑀢𑁆𑀢𑀺𑀁 𑀪𑀢𑁆𑀢𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑀳𑀁𑁇_  


(89.)

135\. _𑀯𑀺𑀪𑀽𑀲𑀸𑀫𑀡𑁆𑀟𑀦𑀭𑀢𑀸,_  
_𑀦𑁆𑀳𑀸𑀧𑀦𑀼𑀘𑁆𑀙𑀸𑀤𑀦𑁂𑀳𑀺 𑀘;_  
_𑀉𑀧𑀓𑀸𑀲𑀺𑀁 𑀇𑀫𑀁 𑀓𑀸𑀬𑀁,_  
_𑀓𑀸𑀫𑀭𑀸𑀕𑁂𑀦 𑀅𑀝𑁆𑀝𑀺𑀢𑀸𑁇_  


(90.)

136\. _𑀢𑀢𑁄 𑀲𑀤𑁆𑀥𑀁 𑀮𑀪𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀩𑁆𑀩𑀚𑀺𑀁 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀸𑀬𑀁 𑀬𑀣𑀸𑀪𑀽𑀢𑀁,_  
_𑀓𑀸𑀫𑀭𑀸𑀕𑁄 𑀲𑀫𑀽𑀳𑀢𑁄𑁇_  


(91.)

137\. _𑀲𑀩𑁆𑀩𑁂 𑀪𑀯𑀸 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸,_  
_𑀇𑀘𑁆𑀙𑀸 𑀘 𑀧𑀢𑁆𑀣𑀦𑀸𑀧𑀺 𑀘;_  
_𑀲𑀩𑁆𑀩𑀬𑁄𑀕𑀯𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸,_  
_𑀲𑀦𑁆𑀢𑀺𑀁 𑀧𑀸𑀧𑀼𑀡𑀺 𑀘𑁂𑀢𑀲𑁄”𑀢𑀺𑁇_  


---

138\. …  𑀦𑀦𑁆𑀤𑀼𑀢𑁆𑀢𑀭𑀸 𑀣𑁂𑀭𑀻… 𑁇
