# 𑁫𑁇𑁨 𑀯𑀺𑀫𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

* 𑀧𑀜𑁆𑀘𑀓𑀦𑀺𑀧𑀸𑀢

* 𑀯𑀺𑀫𑀮𑀸𑀣𑁂𑀭𑀻𑀕𑀸𑀣𑀸

(72.)

115\. _“𑀫𑀢𑁆𑀢𑀸 𑀯𑀡𑁆𑀡𑁂𑀦 𑀭𑀽𑀧𑁂𑀦,_  
_𑀲𑁄𑀪𑀕𑁆𑀕𑁂𑀦 𑀬𑀲𑁂𑀦 𑀘;_  
_𑀬𑁄𑀩𑁆𑀩𑀦𑁂𑀦 𑀘𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀸,_  
_𑀅𑀜𑁆𑀜𑀸𑀲𑀫𑀢𑀺𑀫𑀜𑁆𑀜𑀺𑀳𑀁𑁇_  


(73.)

116\. _𑀯𑀺𑀪𑀽𑀲𑁂𑀢𑁆𑀯𑀸 𑀇𑀫𑀁 𑀓𑀸𑀬𑀁,_  
_𑀲𑀼𑀘𑀺𑀢𑁆𑀢𑀁 𑀩𑀸𑀮𑀮𑀸𑀧𑀦𑀁;_  
_𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑀁 𑀯𑁂𑀲𑀺𑀤𑁆𑀯𑀸𑀭𑀫𑁆𑀳𑀺,_  
_𑀮𑀼𑀤𑁆𑀤𑁄 𑀧𑀸𑀲𑀫𑀺𑀯𑁄𑀟𑁆𑀟𑀺𑀬𑁇_  


(74.)

117\. _𑀧𑀺𑀍𑀅𑀦𑁆𑀥𑀦𑀁 𑀯𑀺𑀤𑀁𑀲𑁂𑀦𑁆𑀢𑀻,_  
_𑀕𑀼𑀬𑁆𑀳𑀁 𑀧𑀓𑀸𑀲𑀺𑀓𑀁 𑀩𑀳𑀼𑀁;_  
_𑀅𑀓𑀸𑀲𑀺𑀁 𑀯𑀺𑀯𑀺𑀥𑀁 𑀫𑀸𑀬𑀁,_  
_𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀦𑁆𑀢𑀻 𑀩𑀳𑀼𑀁 𑀚𑀦𑀁𑁇_  


(75.)

118\. _𑀲𑀸𑀚𑁆𑀚 𑀧𑀺𑀡𑁆𑀟𑀁 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀼𑀡𑁆𑀟𑀸 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀫𑁆𑀳𑀺,_  
_𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑁆𑀲 𑀮𑀸𑀪𑀺𑀦𑀻𑁇_  


(76.)

119\. _𑀲𑀩𑁆𑀩𑁂 𑀬𑁄𑀕𑀸 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸,_  
_𑀬𑁂 𑀤𑀺𑀩𑁆𑀩𑀸 𑀬𑁂 𑀘 𑀫𑀸𑀦𑀼𑀲𑀸;_  
_𑀔𑁂𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀯𑁂 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀻𑀢𑀺𑀪𑀽𑀢𑀸𑀫𑁆𑀳𑀺 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸”𑀢𑀺𑁇_  


---

120\. …  𑀯𑀺𑀫𑀮𑀸 𑀧𑀼𑀭𑀸𑀡𑀕𑀡𑀺𑀓𑀸 𑀣𑁂𑀭𑀻… 𑁇
