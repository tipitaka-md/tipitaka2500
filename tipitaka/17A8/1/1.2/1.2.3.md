# 𑁧𑁇𑁨𑁇𑁩 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(13.)

147\. “𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀪𑀤𑁆𑀤𑁄 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀭𑀸𑀚𑀸𑀭𑀳𑁄 𑀳𑁄𑀢𑀺 𑀭𑀸𑀚𑀪𑁄𑀕𑁆𑀕𑁄, 𑀭𑀜𑁆𑀜𑁄 𑀅𑀗𑁆𑀕𑀦𑁆𑀢𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀜𑁆𑀜𑁄 𑀪𑀤𑁆𑀤𑁄 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀉𑀪𑀢𑁄 𑀲𑀼𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀫𑀸𑀢𑀺𑀢𑁄 𑀘 𑀧𑀺𑀢𑀺𑀢𑁄 𑀘𑁇 𑀬𑀲𑁆𑀲𑀁 𑀤𑀺𑀲𑀸𑀬𑀁 𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀪𑀤𑁆𑀤𑀸 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑀸 𑀚𑀸𑀬𑀦𑁆𑀢𑀺, 𑀢𑀲𑁆𑀲𑀁 𑀤𑀺𑀲𑀸𑀬𑀁 𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀪𑁄𑀚𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑀺—  𑀅𑀮𑁆𑀮𑀁 𑀯𑀸 𑀲𑀼𑀓𑁆𑀔𑀁 𑀯𑀸—  𑀢𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀓𑀺𑀭𑀦𑁆𑀢𑁄𑁇 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀻 𑀳𑁄𑀢𑀺 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑀼𑀁 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁 𑀯𑀸𑁇 𑀲𑁄𑀭𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀔𑀲𑀁𑀯𑀸𑀲𑁄, 𑀦 𑀘 𑀅𑀜𑁆𑀜𑁂 𑀅𑀲𑁆𑀲𑁂 𑀉𑀩𑁆𑀩𑁂𑀚𑁂𑀢𑀸𑁇 𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀓𑀽𑀝𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀚𑀺𑀫𑁆𑀳𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀯𑀗𑁆𑀓𑁂𑀬𑁆𑀬𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀲𑀸𑀭𑀣𑀺𑀲𑁆𑀲 𑀆𑀯𑀺𑀓𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀲𑀫𑀲𑁆𑀲 𑀲𑀸𑀭𑀣𑀺 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀤𑀦𑀸𑀬 𑀯𑀸𑀬𑀫𑀢𑀺𑁇 𑀯𑀸𑀳𑀻 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺𑁇 ‘𑀓𑀸𑀫𑀜𑁆𑀜𑁂 𑀅𑀲𑁆𑀲𑀸 𑀯𑀳𑀦𑁆𑀢𑀼 𑀯𑀸 𑀫𑀸 𑀯𑀸, 𑀅𑀳𑀫𑁂𑀢𑁆𑀣 𑀯𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀉𑀚𑀼𑀫𑀕𑁆𑀕𑁂𑀦𑁂𑀯 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀣𑀸𑀫𑀯𑀸 𑀳𑁄𑀢𑀺 𑀬𑀸𑀯 𑀚𑀻𑀯𑀺𑀢𑀫𑀭𑀡𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀣𑀸𑀫𑀁 𑀉𑀧𑀤𑀁𑀲𑁂𑀢𑀸𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀪𑀤𑁆𑀤𑁄 𑀅𑀲𑁆𑀲𑀸𑀚𑀸𑀦𑀻𑀬𑁄 𑀭𑀸𑀚𑀸𑀭𑀳𑁄 𑀳𑁄𑀢𑀺 𑀭𑀸𑀚𑀪𑁄𑀕𑁆𑀕𑁄, 𑀭𑀜𑁆𑀜𑁄 𑀅𑀗𑁆𑀕𑀦𑁆𑀢𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

148\. 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻, 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀬𑀁 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀪𑁄𑀚𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑀺—  𑀮𑀽𑀔𑀁 𑀯𑀸 𑀧𑀡𑀻𑀢𑀁 𑀯𑀸—  𑀢𑀁 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁𑀬𑁂𑀯 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺 𑀅𑀯𑀺𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀻 𑀳𑁄𑀢𑀺 𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑁂𑀦; 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀸𑀦𑀁 𑀧𑀸𑀧𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀲𑁄𑀭𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀔𑀲𑀁𑀯𑀸𑀲𑁄, 𑀦 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀩𑁆𑀩𑁂𑀚𑁂𑀢𑀸𑁇 𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀸𑀞𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀓𑀽𑀝𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀚𑀺𑀫𑁆𑀳𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀯𑀗𑁆𑀓𑁂𑀬𑁆𑀬𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀆𑀯𑀺𑀓𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀢𑁆𑀣𑀭𑀺 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀽𑀲𑀼 𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀲𑀼𑁇 𑀢𑁂𑀲𑀫𑀲𑁆𑀲 𑀲𑀢𑁆𑀣𑀸 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀽 𑀯𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀤𑀦𑀸𑀬 𑀯𑀸𑀬𑀫𑀢𑀺𑁇 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺𑁇 ‘𑀓𑀸𑀫𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀼 𑀯𑀸 𑀫𑀸 𑀯𑀸, 𑀅𑀳𑀫𑁂𑀢𑁆𑀣 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀺𑁇 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀉𑀚𑀼𑀫𑀕𑁆𑀕𑁂𑀦𑁂𑀯 𑀕𑀘𑁆𑀙𑀢𑀺; 𑀢𑀢𑁆𑀭𑀸𑀬𑀁 𑀉𑀚𑀼𑀫𑀕𑁆𑀕𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑁇 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺—  ‘𑀓𑀸𑀫𑀁 𑀢𑀘𑁄 𑀘 𑀦𑁆𑀳𑀸𑀭𑀼 𑀘 𑀅𑀝𑁆𑀞𑀺 𑀘 𑀅𑀯𑀲𑀺𑀲𑁆𑀲𑀢𑀼, 𑀲𑀭𑀻𑀭𑁂 𑀉𑀧𑀲𑁆𑀲𑀼𑀲𑁆𑀲𑀢𑀼 𑀫𑀁𑀲𑀮𑁄𑀳𑀺𑀢𑀁; 𑀬𑀁 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀣𑀸𑀫𑁂𑀦 𑀧𑀼𑀭𑀺𑀲𑀯𑀻𑀭𑀺𑀬𑁂𑀦 𑀧𑀼𑀭𑀺𑀲𑀧𑀭𑀓𑁆𑀓𑀫𑁂𑀦 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀢𑀁 𑀅𑀧𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀡𑁆𑀞𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
