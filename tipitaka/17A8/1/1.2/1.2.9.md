# 𑁧𑁇𑁨𑁇𑁯 𑀧𑀳𑀸𑀭𑀸𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

* 𑀧𑀳𑀸𑀭𑀸𑀤𑀲𑀼𑀢𑁆𑀢

(19.)

180\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀭𑀜𑁆𑀚𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦𑀍𑀏𑀭𑀼𑀧𑀼𑀘𑀺𑀫𑀦𑁆𑀤𑀫𑀽𑀮𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀧𑀳𑀸𑀭𑀸𑀤𑁄 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀧𑀳𑀸𑀭𑀸𑀤𑀁 𑀅𑀲𑀼𑀭𑀺𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

181\. “𑀅𑀧𑀺 𑀧𑀦, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂”𑀢𑀺𑁇 “𑀓𑀢𑀺 𑀧𑀦, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀅𑀝𑁆𑀞, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀦𑁆𑀦𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑁄𑀡𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑀩𑁆𑀪𑀸𑀭𑁄, 𑀦 𑀆𑀬𑀢𑀓𑁂𑀦𑁂𑀯 𑀧𑀧𑀸𑀢𑁄𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀦𑁆𑀦𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑁄𑀡𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑀩𑁆𑀪𑀸𑀭𑁄, 𑀦 𑀆𑀬𑀢𑀓𑁂𑀦𑁂𑀯 𑀧𑀧𑀸𑀢𑁄; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀧𑀞𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁧)

182\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀞𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑁂𑀮𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀞𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑁂𑀮𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁨)

183\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀦 𑀫𑀢𑁂𑀦 𑀓𑀼𑀡𑀧𑁂𑀦 𑀲𑀁𑀯𑀲𑀢𑀺𑁇 𑀬𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀫𑀢𑀁 𑀓𑀼𑀡𑀧𑀁, 𑀢𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀢𑀻𑀭𑀁 𑀯𑀸𑀳𑁂𑀢𑀺, 𑀣𑀮𑀁 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀦 𑀫𑀢𑁂𑀦 𑀓𑀼𑀡𑀧𑁂𑀦 𑀲𑀁𑀯𑀲𑀢𑀺, 𑀬𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀫𑀢𑀁 𑀓𑀼𑀡𑀧𑀁, 𑀢𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀢𑀻𑀭𑀁 𑀯𑀸𑀳𑁂𑀢𑀺, 𑀣𑀮𑀁 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀢𑀢𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁩)

184\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀕𑀗𑁆𑀕𑀸 𑀬𑀫𑀼𑀦𑀸 𑀅𑀘𑀺𑀭𑀯𑀢𑀻 𑀲𑀭𑀪𑀽 𑀫𑀳𑀻, 𑀢𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀧𑀢𑁆𑀯𑀸 𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀦𑀺 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀸𑀦𑀺, ‘𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄’ 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀕𑀗𑁆𑀕𑀸 𑀬𑀫𑀼𑀦𑀸 𑀅𑀘𑀺𑀭𑀯𑀢𑀻 𑀲𑀭𑀪𑀽 𑀫𑀳𑀻, 𑀢𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀧𑀢𑁆𑀯𑀸 𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀦𑀺 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀸𑀦𑀺, ‘𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄’ 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁪)

185\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸 𑀘 𑀮𑁄𑀓𑁂 𑀲𑀯𑀦𑁆𑀢𑀺𑀬𑁄 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺 𑀬𑀸 𑀘 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀥𑀸𑀭𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁂𑀦 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀊𑀦𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀽𑀭𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸 𑀘 𑀮𑁄𑀓𑁂 𑀲𑀯𑀦𑁆𑀢𑀺𑀬𑁄 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀺 𑀬𑀸 𑀘 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀥𑀸𑀭𑀸 𑀧𑀧𑀢𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁂𑀦 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲 𑀊𑀦𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀽𑀭𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀧𑀜𑁆𑀘𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁫)

186\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀏𑀓𑀭𑀲𑁄 𑀮𑁄𑀡𑀭𑀲𑁄𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀏𑀓𑀭𑀲𑁄 𑀮𑁄𑀡𑀭𑀲𑁄; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀙𑀝𑁆𑀞𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁬)

187\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀩𑀳𑀼𑀭𑀢𑀦𑁄 𑀅𑀦𑁂𑀓𑀭𑀢𑀦𑁄𑁇 𑀢𑀢𑁆𑀭𑀺𑀫𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀡𑀺 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑁄 𑀲𑀗𑁆𑀔𑁄 𑀲𑀺𑀮𑀸 𑀧𑀯𑀸𑀍𑀅𑀁 𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑁄 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀁𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀩𑀳𑀼𑀭𑀢𑀦𑁄 𑀅𑀦𑁂𑀓𑀭𑀢𑀦𑁄; 𑀢𑀢𑁆𑀭𑀺𑀫𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀡𑀺 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑁄 𑀲𑀗𑁆𑀔𑁄 𑀲𑀺𑀮𑀸 𑀧𑀯𑀸𑀍𑀅𑀁 𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑁄 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀁; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀲𑀢𑁆𑀢𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁭)

188\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀫𑀳𑀢𑀁 𑀪𑀽𑀢𑀸𑀦𑀁 𑀆𑀯𑀸𑀲𑁄𑁇 𑀢𑀢𑁆𑀭𑀺𑀫𑁂 𑀪𑀽𑀢𑀸—  𑀢𑀺𑀫𑀺 𑀢𑀺𑀫𑀺𑀗𑁆𑀕𑀮𑁄 𑀢𑀺𑀫𑀺𑀭𑀧𑀺𑀗𑁆𑀕𑀮𑁄 𑀅𑀲𑀼𑀭𑀸 𑀦𑀸𑀕𑀸 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸𑁇 𑀲𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀤𑁆𑀯𑀺𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀢𑀺𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀘𑀢𑀼𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀫𑀳𑀢𑀁 𑀪𑀽𑀢𑀸𑀦𑀁 𑀆𑀯𑀸𑀲𑁄; 𑀢𑀢𑁆𑀭𑀺𑀫𑁂 𑀪𑀽𑀢𑀸—  𑀢𑀺𑀫𑀺 𑀢𑀺𑀫𑀺𑀗𑁆𑀕𑀮𑁄 𑀢𑀺𑀫𑀺𑀭𑀧𑀺𑀗𑁆𑀕𑀮𑁄 𑀅𑀲𑀼𑀭𑀸 𑀦𑀸𑀕𑀸 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸; 𑀲𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸…𑀧𑁂…  𑀢𑀺𑀬𑁄𑀚𑀦…  𑀘𑀢𑀼𑀬𑁄𑀚𑀦…  𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸; 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀝𑁆𑀞𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀝𑁆𑀞 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀼𑀭𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀻𑀢𑀺𑁇 (𑁮)

189\. 𑀅𑀧𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂”𑀢𑀺𑁇 “𑀓𑀢𑀺 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀅𑀝𑁆𑀞, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀦𑀺𑀦𑁆𑀦𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑁄𑀡𑁄 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑀩𑁆𑀪𑀸𑀭𑁄, 𑀦 𑀆𑀬𑀢𑀓𑁂𑀦𑁂𑀯 𑀧𑀧𑀸𑀢𑁄; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑀺𑀓𑁆𑀔𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑀝𑀺𑀧𑀤𑀸, 𑀦 𑀆𑀬𑀢𑀓𑁂𑀦𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀧𑀝𑀺𑀯𑁂𑀥𑁄𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑀺𑀓𑁆𑀔𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀧𑀝𑀺𑀧𑀤𑀸, 𑀦 𑀆𑀬𑀢𑀓𑁂𑀦𑁂𑀯 𑀅𑀜𑁆𑀜𑀸𑀧𑀝𑀺𑀯𑁂𑀥𑁄; 𑀅𑀬𑀁, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀞𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁧)

190\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀞𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑁂𑀮𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀬𑀁 𑀫𑀬𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀁 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸 𑀚𑀻𑀯𑀺𑀢𑀳𑁂𑀢𑀼𑀧𑀺 𑀦𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀫𑀬𑀸 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀁 𑀫𑀫 𑀲𑀸𑀯𑀓𑀸 𑀚𑀻𑀯𑀺𑀢𑀳𑁂𑀢𑀼𑀧𑀺 𑀦𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀅𑀬𑀁, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁨)

191\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀦 𑀫𑀢𑁂𑀦 𑀓𑀼𑀡𑀧𑁂𑀦 𑀲𑀁𑀯𑀲𑀢𑀺𑁇 𑀬𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀫𑀢𑀁 𑀓𑀼𑀡𑀧𑀁, 𑀢𑀁 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀢𑀻𑀭𑀁 𑀯𑀸𑀳𑁂𑀢𑀺 𑀣𑀮𑀁 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀼𑀘𑀺𑀲𑀗𑁆𑀓𑀲𑁆𑀲𑀭𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀲𑀫𑀡𑀧𑀝𑀺𑀜𑁆𑀜𑁄 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀧𑀝𑀺𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀢𑁄𑀧𑀽𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀓𑀲𑀫𑁆𑀩𑀼𑀚𑀸𑀢𑁄, 𑀦 𑀢𑁂𑀦 𑀲𑀁𑀖𑁄 𑀲𑀁𑀯𑀲𑀢𑀺; 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀢𑀺𑁇

192\. 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀫𑀚𑁆𑀛𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀆𑀭𑀓𑀸𑀯 𑀲𑀁𑀖𑀫𑁆𑀳𑀸 𑀲𑀁𑀖𑁄 𑀘 𑀢𑁂𑀦𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀬𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑁄 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀼𑀘𑀺𑀲𑀗𑁆𑀓𑀲𑁆𑀲𑀭𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀲𑀫𑀡𑀧𑀝𑀺𑀜𑁆𑀜𑁄 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀧𑀝𑀺𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀢𑁄𑀧𑀽𑀢𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑁄 𑀓𑀲𑀫𑁆𑀩𑀼𑀚𑀸𑀢𑁄, 𑀦 𑀢𑁂𑀦 𑀲𑀁𑀖𑁄 𑀲𑀁𑀯𑀲𑀢𑀺; 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀧𑀢𑀺; 𑀓𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑁄 𑀳𑁄𑀢𑀺 𑀫𑀚𑁆𑀛𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄, 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀆𑀭𑀓𑀸𑀯 𑀲𑀁𑀖𑀫𑁆𑀳𑀸 𑀲𑀁𑀖𑁄 𑀘 𑀢𑁂𑀦; 𑀅𑀬𑀁, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀢𑀢𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁩)

193\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀬𑀸 𑀓𑀸𑀘𑀺 𑀫𑀳𑀸𑀦𑀤𑀺𑀬𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀕𑀗𑁆𑀕𑀸 𑀬𑀫𑀼𑀦𑀸 𑀅𑀘𑀺𑀭𑀯𑀢𑀻 𑀲𑀭𑀪𑀽 𑀫𑀳𑀻, 𑀢𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑀁 𑀧𑀢𑁆𑀯𑀸 𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀦𑀺 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀸𑀦𑀺, ‘𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄’ 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂 𑀯𑀡𑁆𑀡𑀸—  𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸, 𑀯𑁂𑀲𑁆𑀲𑀸, 𑀲𑀼𑀤𑁆𑀤𑀸, 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀦𑀺 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀸𑀦𑀺, ‘𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸’ 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂 𑀯𑀡𑁆𑀡𑀸—  𑀔𑀢𑁆𑀢𑀺𑀬𑀸, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸, 𑀯𑁂𑀲𑁆𑀲𑀸, 𑀲𑀼𑀤𑁆𑀤𑀸, 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀸𑀦𑀺 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀸𑀦𑀺, ‘𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸’ 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀅𑀬𑀁, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁪)

196\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀩𑀳𑀼𑀭𑀢𑀦𑁄 𑀅𑀦𑁂𑀓𑀭𑀢𑀦𑁄; 𑀢𑀢𑁆𑀭𑀺𑀫𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀫𑀼𑀢𑁆𑀢𑀸 𑀫𑀡𑀺 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑁄 𑀲𑀗𑁆𑀔𑁄 𑀲𑀺𑀮𑀸 𑀧𑀯𑀸𑀍𑀅𑀁 𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑁄 𑀫𑀲𑀸𑀭𑀕𑀮𑁆𑀮𑀁; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄 𑀩𑀳𑀼𑀭𑀢𑀦𑁄 𑀅𑀦𑁂𑀓𑀭𑀢𑀦𑁄𑁇 𑀢𑀢𑁆𑀭𑀺𑀫𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄 𑀩𑀳𑀼𑀭𑀢𑀦𑁄 𑀅𑀦𑁂𑀓𑀭𑀢𑀦𑁄; 𑀢𑀢𑁆𑀭𑀺𑀫𑀸𑀦𑀺 𑀭𑀢𑀦𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀢𑀺𑀧𑀝𑁆𑀞𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀥𑀸𑀦𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸, 𑀧𑀜𑁆𑀘𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀜𑁆𑀘 𑀩𑀮𑀸𑀦𑀺, 𑀲𑀢𑁆𑀢 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀸, 𑀅𑀭𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀕𑁆𑀕𑁄; 𑀅𑀬𑀁, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀲𑀢𑁆𑀢𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁭)

197\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀫𑀳𑀢𑀁 𑀪𑀽𑀢𑀸𑀦𑀁 𑀆𑀯𑀸𑀲𑁄; 𑀢𑀢𑁆𑀭𑀺𑀫𑁂 𑀪𑀽𑀢𑀸—  𑀢𑀺𑀫𑀺 𑀢𑀺𑀫𑀺𑀗𑁆𑀕𑀮𑁄 𑀢𑀺𑀫𑀺𑀭𑀧𑀺𑀗𑁆𑀕𑀮𑁄 𑀅𑀲𑀼𑀭𑀸 𑀦𑀸𑀕𑀸 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀸; 𑀲𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀤𑁆𑀯𑀺𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀢𑀺𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀘𑀢𑀼𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸, 𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀲𑀢𑀺𑀓𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀪𑀸𑀯𑀸; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄 𑀫𑀳𑀢𑀁 𑀪𑀽𑀢𑀸𑀦𑀁 𑀆𑀯𑀸𑀲𑁄; 𑀢𑀢𑁆𑀭𑀺𑀫𑁂 𑀪𑀽𑀢𑀸—  𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀭𑀳𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄𑁇 𑀬𑀫𑁆𑀧𑀺, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄 𑀫𑀳𑀢𑀁 𑀪𑀽𑀢𑀸𑀦𑀁 𑀆𑀯𑀸𑀲𑁄; 𑀢𑀢𑁆𑀭𑀺𑀫𑁂 𑀪𑀽𑀢𑀸—  𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀭𑀳𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀅𑀬𑀁, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀝𑁆𑀞𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀬𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀺𑁇 (𑁮)

198\. 𑀇𑀫𑁂 𑀔𑁄, 𑀧𑀳𑀸𑀭𑀸𑀤, 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀝𑁆𑀞 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂 𑀤𑀺𑀲𑁆𑀯𑀸 𑀤𑀺𑀲𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀪𑀺𑀭𑀫𑀦𑁆𑀢𑀻”𑀢𑀺𑁇
