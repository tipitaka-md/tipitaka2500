# 𑁧𑁇𑁨𑁇𑁫 𑀫𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑀳𑀸𑀯𑀕𑁆𑀕

### 𑀫𑀮𑀲𑀼𑀢𑁆𑀢

(15.)

168. “𑀅𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀮𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞? 𑀅𑀲𑀚𑁆𑀛𑀸𑀬𑀫𑀮𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑁆𑀢𑀸; 𑀅𑀦𑀼𑀝𑁆𑀞𑀸𑀦𑀫𑀮𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀖𑀭𑀸; 𑀫𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀡𑁆𑀡𑀲𑁆𑀲 𑀓𑁄𑀲𑀚𑁆𑀚𑀁; 𑀧𑀫𑀸𑀤𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀓𑁆𑀔𑀢𑁄 𑀫𑀮𑀁; 𑀫𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁; 𑀫𑀘𑁆𑀙𑁂𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀤𑀢𑁄 𑀫𑀮𑀁; 𑀫𑀮𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀫𑁆𑀳𑀺 𑀘; 𑀢𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀮𑀸 𑀫𑀮𑀢𑀭𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀭𑀫𑀁 𑀫𑀮𑀁𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀫𑀮𑀸𑀦𑀻𑀢𑀺𑁇

169. _𑀅𑀲𑀚𑁆𑀛𑀸𑀬𑀫𑀮𑀸 𑀫𑀦𑁆𑀢𑀸,_  
_𑀅𑀦𑀼𑀝𑁆𑀞𑀸𑀦𑀫𑀮𑀸 𑀖𑀭𑀸;_  
_𑀫𑀮𑀁 𑀯𑀡𑁆𑀡𑀲𑁆𑀲 𑀓𑁄𑀲𑀚𑁆𑀚𑀁,_  
_𑀧𑀫𑀸𑀤𑁄 𑀭𑀓𑁆𑀔𑀢𑁄 𑀫𑀮𑀁𑁇_  


170. _𑀫𑀮𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁,_  
_𑀫𑀘𑁆𑀙𑁂𑀭𑀁 𑀤𑀤𑀢𑁄 𑀫𑀮𑀁;_  
_𑀫𑀮𑀸 𑀯𑁂 𑀧𑀸𑀧𑀓𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑁂 𑀧𑀭𑀫𑁆𑀳𑀺 𑀘;_  
_𑀢𑀢𑁄 𑀫𑀮𑀸 𑀫𑀮𑀢𑀭𑀁,_  
_𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀧𑀭𑀫𑀁 𑀫𑀮𑀦𑁆”𑀢𑀺𑁇_  


---

171. 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
