# 𑁧𑁇𑁫𑁇𑁮 𑀦𑀓𑀼𑀮𑀫𑀸𑀢𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀉𑀧𑁄𑀲𑀣𑀯𑀕𑁆𑀕

* 𑀦𑀓𑀼𑀮𑀫𑀸𑀢𑀸𑀲𑀼𑀢𑁆𑀢

(48.)

474\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀪𑀕𑁆𑀕𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀕𑀺𑀭𑁂 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀦𑀓𑀼𑀮𑀫𑀸𑀢𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀻 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸…𑀧𑁂… 𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀦𑀓𑀼𑀮𑀫𑀸𑀢𑀭𑀁 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

475\. “𑀅𑀝𑁆𑀞𑀳𑀺 𑀔𑁄, 𑀦𑀓𑀼𑀮𑀫𑀸𑀢𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀇𑀥, 𑀦𑀓𑀼𑀮𑀫𑀸𑀢𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀬𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀪𑀢𑁆𑀢𑀼𑀦𑁄 𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀢𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀼𑀝𑁆𑀞𑀸𑀬𑀺𑀦𑀻 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀧𑀸𑀢𑀺𑀦𑀻 𑀓𑀺𑀁𑀓𑀸𑀭𑀧𑀝𑀺𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑀻 𑀫𑀦𑀸𑀧𑀘𑀸𑀭𑀺𑀦𑀻 𑀧𑀺𑀬𑀯𑀸𑀤𑀺𑀦𑀻𑁇 (𑁧)

476\. 𑀬𑁂 𑀢𑁂 𑀪𑀢𑁆𑀢𑀼 𑀕𑀭𑀼𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀫𑀸𑀢𑀸𑀢𑀺 𑀯𑀸 𑀧𑀺𑀢𑀸𑀢𑀺 𑀯𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀢𑀺 𑀯𑀸—  𑀢𑁂 𑀲𑀓𑁆𑀓𑀭𑁄𑀢𑀺 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀢𑀺 𑀫𑀸𑀦𑁂𑀢𑀺 𑀧𑀽𑀚𑁂𑀢𑀺, 𑀅𑀩𑁆𑀪𑀸𑀕𑀢𑁂 𑀘 𑀆𑀲𑀦𑁄𑀤𑀓𑁂𑀦 𑀧𑀝𑀺𑀧𑀽𑀚𑁂𑀢𑀺𑁇 (𑁨)

477\. 𑀬𑁂 𑀢𑁂 𑀪𑀢𑁆𑀢𑀼 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸—  𑀉𑀡𑁆𑀡𑀸𑀢𑀺 𑀯𑀸 𑀓𑀧𑁆𑀧𑀸𑀲𑀸𑀢𑀺 𑀯𑀸—  𑀢𑀢𑁆𑀣 𑀤𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑀸 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 (𑁩)

478\. 𑀬𑁄 𑀲𑁄 𑀪𑀢𑁆𑀢𑀼 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑁄 𑀅𑀦𑁆𑀢𑁄𑀚𑀦𑁄—  𑀤𑀸𑀲𑀸𑀢𑀺 𑀯𑀸 𑀧𑁂𑀲𑁆𑀲𑀸𑀢𑀺 𑀯𑀸 𑀓𑀫𑁆𑀫𑀓𑀭𑀸𑀢𑀺 𑀯𑀸—  𑀢𑁂𑀲𑀁 𑀓𑀢𑀜𑁆𑀘 𑀓𑀢𑀢𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀅𑀓𑀢𑀜𑁆𑀘 𑀅𑀓𑀢𑀢𑁄 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀓𑀸𑀦𑀜𑁆𑀘 𑀩𑀮𑀸𑀩𑀮𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀁𑀲𑁂𑀦 𑀲𑀁𑀯𑀺𑀪𑀚𑀢𑀺𑁇 (𑁪)

479\. 𑀬𑀁 𑀪𑀢𑁆𑀢𑀸 𑀆𑀳𑀭𑀢𑀺 𑀥𑀦𑀁 𑀯𑀸 𑀥𑀜𑁆𑀜𑀁 𑀯𑀸 𑀭𑀚𑀢𑀁 𑀯𑀸 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀯𑀸 𑀢𑀁 𑀆𑀭𑀓𑁆𑀔𑁂𑀦 𑀕𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀥𑀼𑀢𑁆𑀢𑀻 𑀅𑀣𑁂𑀦𑀻 𑀅𑀲𑁄𑀡𑁆𑀟𑀻 𑀅𑀯𑀺𑀦𑀸𑀲𑀺𑀓𑀸𑁇 (𑁫)

480\. 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸𑁇 (𑁬)

481\. 𑀲𑀻𑀮𑀯𑀢𑀻 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺—  𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸…𑀧𑁂…  𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸…𑀧𑁂… 𑁇 (𑁭)

482\. 𑀘𑀸𑀕𑀯𑀢𑀻 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺 𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺 𑀫𑀼𑀢𑁆𑀢𑀘𑀸𑀕𑀸 𑀧𑀬𑀢𑀧𑀸𑀡𑀺𑀦𑀻 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀭𑀢𑀸 𑀬𑀸𑀘𑀬𑁄𑀕𑀸 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑀸𑁇 (𑁮)

483\. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀦𑀓𑀼𑀮𑀫𑀸𑀢𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺𑁇

484\. _𑀬𑁄 𑀦𑀁 𑀪𑀭𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀆𑀢𑀸𑀧𑀺 𑀉𑀲𑁆𑀲𑀼𑀓𑁄;_  
_𑀢𑀁 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀤𑀁 𑀧𑁄𑀲𑀁,_  
_𑀪𑀢𑁆𑀢𑀸𑀭𑀁 𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺𑁇_  


485\. _𑀦 𑀘𑀸𑀧𑀺 𑀲𑁄𑀢𑁆𑀣𑀺 𑀪𑀢𑁆𑀢𑀸𑀭𑀁,_  
_𑀇𑀲𑁆𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀭𑁄𑀲𑀬𑁂;_  
_𑀪𑀢𑁆𑀢𑀼 𑀘 𑀕𑀭𑀼𑀦𑁄 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀝𑀺𑀧𑀽𑀚𑁂𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


486\. _𑀉𑀝𑁆𑀞𑀸𑀳𑀺𑀓𑀸 𑀅𑀦𑀮𑀲𑀸,_  
_𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀭𑀺𑀚𑁆𑀚𑀦𑀸;_  
_𑀪𑀢𑁆𑀢𑀼 𑀫𑀦𑀸𑀧𑀁 𑀘𑀭𑀢𑀺,_  
_𑀲𑀫𑁆𑀪𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀢𑀺𑁇_  


487\. _𑀬𑀸 𑀏𑀯𑀁 𑀯𑀢𑁆𑀢𑀢𑀺 𑀦𑀸𑀭𑀻,_  
_𑀪𑀢𑁆𑀢𑀼 𑀙𑀦𑁆𑀤𑀯𑀲𑀸𑀦𑀼𑀕𑀸;_  
_𑀫𑀦𑀸𑀧𑀸 𑀦𑀸𑀫 𑀢𑁂 𑀤𑁂𑀯𑀸,_  
_𑀬𑀢𑁆𑀣 𑀲𑀸 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


---

488\. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
