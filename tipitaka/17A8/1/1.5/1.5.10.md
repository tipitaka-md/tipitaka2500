# 𑁧𑁇𑁫𑁇𑁧𑁦 𑀤𑀼𑀢𑀺𑀬𑁃𑀥𑀮𑁄𑀓𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑁄𑀲𑀣𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑁃𑀥𑀮𑁄𑀓𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(50.)

506. “𑀘𑀢𑀽𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀇𑀥𑀮𑁄𑀓𑀯𑀺𑀚𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀬𑀁𑀲 𑀮𑁄𑀓𑁄 𑀆𑀭𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀭𑀺𑀚𑀦𑁄, 𑀪𑀢𑁆𑀢𑀼 𑀫𑀦𑀸𑀧𑀁 𑀘𑀭𑀢𑀺, 𑀲𑀫𑁆𑀪𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀢𑀺𑁇

507. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀬𑁂 𑀢𑁂 𑀪𑀢𑁆𑀢𑀼 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

508. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀭𑀺𑀚𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀬𑁄 𑀲𑁄 𑀪𑀢𑁆𑀢𑀼 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑁄 𑀅𑀦𑁆𑀢𑁄𑀚𑀦𑁄…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀭𑀺𑀚𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

509. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀪𑀢𑁆𑀢𑀼 𑀫𑀦𑀸𑀧𑀁 𑀘𑀭𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀬𑀁 𑀪𑀢𑁆𑀢𑀼 𑀅𑀫𑀦𑀸𑀧𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀳𑁂𑀢𑀼𑀧𑀺 𑀦 𑀅𑀚𑁆𑀛𑀸𑀘𑀭𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀪𑀢𑁆𑀢𑀼 𑀫𑀦𑀸𑀧𑀁 𑀘𑀭𑀢𑀺𑁇

510. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀫𑁆𑀪𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀬𑀁 𑀪𑀢𑁆𑀢𑀸 𑀆𑀳𑀭𑀢𑀺…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀫𑁆𑀪𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀇𑀥𑀮𑁄𑀓𑀯𑀺𑀚𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀬𑀁𑀲 𑀮𑁄𑀓𑁄 𑀆𑀭𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇

511. 𑀘𑀢𑀽𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀭𑀮𑁄𑀓𑀯𑀺𑀚𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀮𑁄𑀓𑁄 𑀆𑀭𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

512. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

513. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

514. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

515. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀧𑀭𑀮𑁄𑀓𑀯𑀺𑀚𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀮𑁄𑀓𑁄 𑀆𑀭𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

516. _𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸,_  
_𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀭𑀺𑀚𑁆𑀚𑀦𑀸;_  
_𑀪𑀢𑁆𑀢𑀼 𑀫𑀦𑀸𑀧𑀁 𑀘𑀭𑀢𑀺,_  
_𑀲𑀫𑁆𑀪𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀢𑀺𑁇_  


517. _𑀲𑀤𑁆𑀥𑀸 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀯𑀤𑀜𑁆𑀜𑀽 𑀯𑀻𑀢𑀫𑀘𑁆𑀙𑀭𑀸;_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀫𑀕𑁆𑀕𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺,_  
_𑀲𑁄𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁𑁇_  


518. _𑀇𑀘𑁆𑀘𑁂𑀢𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀘,_  
_𑀬𑀲𑁆𑀲𑀸 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀦𑀸𑀭𑀺𑀬𑀸;_  
_𑀢𑀫𑁆𑀧𑀺 𑀲𑀻𑀮𑀯𑀢𑀺𑀁 𑀆𑀳𑀼,_  
_𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀁 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑀺𑀁𑁇_  


519. _𑀲𑁄𑀍𑀅𑀲𑀸𑀓𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀼𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀢𑀸𑀤𑀺𑀲𑀻 𑀲𑀻𑀮𑀯𑀢𑀻 𑀉𑀧𑀸𑀲𑀺𑀓𑀸,_  
_𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑁂𑀯𑀮𑁄𑀓𑀁 𑀫𑀦𑀸𑀧𑀦𑁆”𑀢𑀺𑁇_

_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀧𑀼𑀦 𑀯𑀺𑀲𑀸𑀔𑁂,_  
_𑀦𑀓𑀼𑀮𑀸 𑀇𑀥𑀮𑁄𑀓𑀺𑀓𑀸 𑀤𑁆𑀯𑁂𑀢𑀺𑁇_
