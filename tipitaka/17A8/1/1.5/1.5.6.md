# 𑁧𑁇𑁫𑁇𑁬 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑁄𑀲𑀣𑀯𑀕𑁆𑀕

### 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑀼𑀢𑁆𑀢

(46.)

443. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁂𑀯𑀢𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸 𑀦𑀸𑀫 𑀤𑁂𑀯𑀢𑀸 𑀢𑀻𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀓𑀸𑀭𑁂𑀫 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀫𑁇 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀬𑀸𑀤𑀺𑀲𑀓𑀁 𑀯𑀡𑁆𑀡𑀁 𑀆𑀓𑀗𑁆𑀔𑀸𑀫 𑀢𑀸𑀤𑀺𑀲𑀓𑀁 𑀯𑀡𑁆𑀡𑀁 𑀞𑀸𑀦𑀲𑁄 𑀧𑀝𑀺𑀮𑀪𑀸𑀫; 𑀬𑀸𑀤𑀺𑀲𑀓𑀁 𑀲𑀭𑀁 𑀆𑀓𑀗𑁆𑀔𑀸𑀫 𑀢𑀸𑀤𑀺𑀲𑀓𑀁 𑀲𑀭𑀁 𑀞𑀸𑀦𑀲𑁄 𑀧𑀝𑀺𑀮𑀪𑀸𑀫; 𑀬𑀸𑀤𑀺𑀲𑀓𑀁 𑀲𑀼𑀔𑀁 𑀆𑀓𑀗𑁆𑀔𑀸𑀫 𑀢𑀸𑀤𑀺𑀲𑀓𑀁 𑀲𑀼𑀔𑀁 𑀞𑀸𑀦𑀲𑁄 𑀧𑀝𑀺𑀮𑀪𑀸𑀫𑁇 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸 𑀦𑀸𑀫 𑀤𑁂𑀯𑀢𑀸 𑀇𑀫𑁂𑀲𑀼 𑀢𑀻𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀁 𑀓𑀸𑀭𑁂𑀫 𑀯𑀲𑀁 𑀯𑀢𑁆𑀢𑁂𑀫𑀸”𑀢𑀺𑁇

444. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑁄 𑀯𑀢𑀺𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀲𑀩𑁆𑀩𑀸𑀯 𑀦𑀻𑀮𑀸 𑀅𑀲𑁆𑀲𑀼 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸 𑀦𑀻𑀮𑀯𑀢𑁆𑀣𑀸 𑀦𑀻𑀮𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯 𑀦𑀻𑀮𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸 𑀦𑀻𑀮𑀯𑀢𑁆𑀣𑀸 𑀦𑀻𑀮𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑁇

445. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑁄 𑀯𑀢𑀺𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀲𑀩𑁆𑀩𑀸𑀯 𑀧𑀻𑀢𑀸 𑀅𑀲𑁆𑀲𑀼…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀸𑀯 𑀮𑁄𑀳𑀺𑀢𑀓𑀸 𑀅𑀲𑁆𑀲𑀼…  𑀲𑀩𑁆𑀩𑀸𑀯 𑀑𑀤𑀸𑀢𑀸 𑀅𑀲𑁆𑀲𑀼 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸 𑀑𑀤𑀸𑀢𑀯𑀢𑁆𑀣𑀸 𑀑𑀤𑀸𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀫𑀜𑁆𑀜𑀸𑀬 𑀲𑀩𑁆𑀩𑀸𑀯 𑀑𑀤𑀸𑀢𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸 𑀑𑀤𑀸𑀢𑀯𑀢𑁆𑀣𑀸 𑀑𑀤𑀸𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑁇

446. 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀓𑀸 𑀘 𑀕𑀸𑀬𑀺 𑀏𑀓𑀸 𑀘 𑀦𑀘𑁆𑀘𑀺 𑀏𑀓𑀸 𑀘 𑀅𑀘𑁆𑀙𑀭𑀁 𑀯𑀸𑀤𑁂𑀲𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀢𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑀸𑀍𑀇𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀲𑀼𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀓𑀫𑀦𑀻𑀬𑁄 𑀘 𑀧𑁂𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑀸𑀲𑀁 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀓𑀫𑀦𑀻𑀬𑁄 𑀘 𑀧𑁂𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀑𑀓𑁆𑀔𑀺𑀧𑀺𑁇

447. 𑀅𑀣 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 “𑀦 𑀔𑁆𑀯𑀬𑁆𑀬𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀤𑀺𑀬𑀢𑀻”𑀢𑀺 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

450. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀓𑀸 𑀘 𑀕𑀸𑀬𑀺 𑀏𑀓𑀸 𑀘 𑀦𑀘𑁆𑀘𑀺 𑀏𑀓𑀸 𑀘 𑀅𑀘𑁆𑀙𑀭𑀁 𑀯𑀸𑀤𑁂𑀲𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀜𑁆𑀘𑀗𑁆𑀕𑀺𑀓𑀲𑁆𑀲 𑀢𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀼𑀯𑀺𑀦𑀻𑀢𑀲𑁆𑀲 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑀸𑀍𑀇𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀲𑀼𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀢𑀲𑁆𑀲 𑀲𑀤𑁆𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀓𑀫𑀦𑀻𑀬𑁄 𑀘 𑀧𑁂𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑀸𑀲𑀁 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑀦𑀁 𑀲𑀤𑁆𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀕𑁆𑀕𑀼 𑀘 𑀭𑀚𑀦𑀻𑀬𑁄 𑀘 𑀓𑀫𑀦𑀻𑀬𑁄 𑀘 𑀧𑁂𑀫𑀦𑀻𑀬𑁄 𑀘 𑀫𑀤𑀦𑀻𑀬𑁄 𑀘𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺 𑀑𑀓𑁆𑀔𑀺𑀧𑀺𑁇

451. 𑀅𑀣 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 ‘𑀦 𑀔𑁆𑀯𑀬𑁆𑀬𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀲𑀸𑀤𑀺𑀬𑀢𑀻’𑀢𑀺 𑀢𑀢𑁆𑀣𑁂𑀯𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺?

452. “𑀅𑀝𑁆𑀞𑀳𑀺 𑀔𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀇𑀥, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀬𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀪𑀢𑁆𑀢𑀼𑀦𑁄 𑀤𑁂𑀦𑁆𑀢𑀺 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀢𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀼𑀝𑁆𑀞𑀸𑀬𑀺𑀦𑀻 𑀧𑀘𑁆𑀙𑀸𑀦𑀺𑀧𑀸𑀢𑀺𑀦𑀻 𑀓𑀺𑀁𑀓𑀸𑀭𑀧𑀝𑀺𑀲𑁆𑀲𑀸𑀯𑀺𑀦𑀻 𑀫𑀦𑀸𑀧𑀘𑀸𑀭𑀺𑀦𑀻 𑀧𑀺𑀬𑀯𑀸𑀤𑀺𑀦𑀻𑁇 (𑁧)

453. 𑀬𑁂 𑀢𑁂 𑀪𑀢𑁆𑀢𑀼 𑀕𑀭𑀼𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀫𑀸𑀢𑀸𑀢𑀺 𑀯𑀸 𑀧𑀺𑀢𑀸𑀢𑀺 𑀯𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀢𑀺 𑀯𑀸—  𑀢𑁂 𑀲𑀓𑁆𑀓𑀭𑁄𑀢𑀺, 𑀕𑀭𑀼𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀫𑀸𑀦𑁂𑀢𑀺, 𑀧𑀽𑀚𑁂𑀢𑀺, 𑀅𑀩𑁆𑀪𑀸𑀕𑀢𑁂 𑀘 𑀆𑀲𑀦𑁄𑀤𑀓𑁂𑀦 𑀧𑀝𑀺𑀧𑀽𑀚𑁂𑀢𑀺𑁇 (𑁨)

454. 𑀬𑁂 𑀢𑁂 𑀪𑀢𑁆𑀢𑀼 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸—  𑀉𑀡𑁆𑀡𑀸𑀢𑀺 𑀯𑀸 𑀓𑀧𑁆𑀧𑀸𑀲𑀸𑀢𑀺 𑀯𑀸—  𑀢𑀢𑁆𑀣 𑀤𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑀸 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 (𑁩)

455. 𑀬𑁄 𑀲𑁄 𑀪𑀢𑁆𑀢𑀼 𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑁄 𑀅𑀦𑁆𑀢𑁄𑀚𑀦𑁄—  𑀤𑀸𑀲𑀸𑀢𑀺 𑀯𑀸 𑀧𑁂𑀲𑁆𑀲𑀸𑀢𑀺 𑀯𑀸 𑀓𑀫𑁆𑀫𑀓𑀭𑀸𑀢𑀺 𑀯𑀸—  𑀢𑁂𑀲𑀁 𑀓𑀢𑀜𑁆𑀘 𑀓𑀢𑀢𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀅𑀓𑀢𑀜𑁆𑀘 𑀅𑀓𑀢𑀢𑁄 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀕𑀺𑀮𑀸𑀦𑀓𑀸𑀦𑀜𑁆𑀘 𑀩𑀮𑀸𑀩𑀮𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀜𑁆𑀘𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀁𑀲𑁂𑀦 𑀲𑀁𑀯𑀺𑀪𑀚𑀢𑀺𑁇 (𑁪)

456. 𑀬𑀁 𑀪𑀢𑁆𑀢𑀼 𑀆𑀳𑀭𑀢𑀺 𑀥𑀦𑀁 𑀯𑀸 𑀥𑀜𑁆𑀜𑀁 𑀯𑀸 𑀚𑀸𑀢𑀭𑀽𑀧𑀁 𑀯𑀸 𑀢𑀁 𑀆𑀭𑀓𑁆𑀔𑁂𑀦 𑀕𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀥𑀼𑀢𑁆𑀢𑀻 𑀅𑀣𑁂𑀦𑀻 𑀅𑀲𑁄𑀡𑁆𑀟𑀻 𑀅𑀯𑀺𑀦𑀸𑀲𑀺𑀓𑀸𑁇 (𑁫)

457. 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸𑁇 (𑁬)

458. 𑀲𑀻𑀮𑀯𑀢𑀻 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺—  𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀸𑁇 (𑁭)

459. 𑀘𑀸𑀕𑀯𑀢𑀻 𑀔𑁄 𑀧𑀦 𑀳𑁄𑀢𑀺𑁇 𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺 𑀫𑀼𑀢𑁆𑀢𑀘𑀸𑀕𑀸 𑀧𑀬𑀢𑀧𑀸𑀡𑀺𑀦𑀻 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀭𑀢𑀸 𑀬𑀸𑀘𑀬𑁄𑀕𑀸 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑀸𑁇 (𑁮)

460. 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀫𑀦𑀸𑀧𑀓𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀤𑁂𑀯𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺𑁇

461. _𑀬𑁄 𑀦𑀁 𑀪𑀭𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀸,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀆𑀢𑀸𑀧𑀺 𑀉𑀲𑁆𑀲𑀼𑀓𑁄;_  
_𑀢𑀁 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀤𑀁 𑀧𑁄𑀲𑀁,_  
_𑀪𑀢𑁆𑀢𑀸𑀭𑀁 𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺𑁇_  


462. _𑀦 𑀘𑀸𑀧𑀺 𑀲𑁄𑀢𑁆𑀣𑀺 𑀪𑀢𑁆𑀢𑀸𑀭𑀁,_  
_𑀇𑀲𑁆𑀲𑀸𑀯𑀸𑀤𑁂𑀦 𑀭𑁄𑀲𑀬𑁂;_  
_𑀪𑀢𑁆𑀢𑀼 𑀘 𑀕𑀭𑀼𑀦𑁄 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀝𑀺𑀧𑀽𑀚𑁂𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇_  


463. _𑀉𑀝𑁆𑀞𑀸𑀳𑀺𑀓𑀸 𑀅𑀦𑀮𑀲𑀸,_  
_𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀧𑀭𑀺𑀚𑁆𑀚𑀦𑀸;_  
_𑀪𑀢𑁆𑀢𑀼 𑀫𑀦𑀸𑀧𑀁 𑀘𑀭𑀢𑀺,_  
_𑀲𑀫𑁆𑀪𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀢𑀺𑁇_  


464. _𑀬𑀸 𑀏𑀯𑀁 𑀯𑀢𑁆𑀢𑀢𑀺 𑀦𑀸𑀭𑀻,_  
_𑀪𑀢𑁆𑀢𑀼 𑀙𑀦𑁆𑀤𑀯𑀲𑀸𑀦𑀼𑀕𑀸;_  
_𑀫𑀦𑀸𑀧𑀸 𑀦𑀸𑀫 𑀢𑁂 𑀤𑁂𑀯𑀸,_  
_𑀬𑀢𑁆𑀣 𑀲𑀸 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_  


---

465. 𑀙𑀝𑁆𑀞𑀁𑁇
