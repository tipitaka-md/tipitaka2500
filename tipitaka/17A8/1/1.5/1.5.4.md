# 𑁧𑁇𑁫𑁇𑁪 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀉𑀧𑁄𑀲𑀣𑀯𑀕𑁆𑀕

### 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀲𑀼𑀢𑁆𑀢

(44.)

426\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀉𑀧𑁄𑀲𑀣𑁄 𑀉𑀧𑀯𑀼𑀢𑁆𑀣𑁄 𑀫𑀳𑀧𑁆𑀨𑀮𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑀸 𑀲𑀕𑁆𑀕𑀫𑀼𑀧𑁂𑀦𑁆𑀢𑀺 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

427\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀺𑀬𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀧𑀺𑀬𑀸𑀦𑀫𑁆𑀧𑀺 𑀫𑁂 𑀅𑀲𑁆𑀲 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸…𑀧𑁂…  𑀯𑁂𑀲𑁆𑀲𑀸…  𑀲𑀼𑀤𑁆𑀤𑀸 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

428\. “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀏𑀯𑀫𑁂𑀢𑀁, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞𑁇 𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀔𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀲𑀩𑁆𑀩𑁂 𑀘𑁂𑀧𑀺, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸…𑀧𑁂…  𑀯𑁂𑀲𑁆𑀲𑀸…  𑀲𑀼𑀤𑁆𑀤𑀸 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀩𑁆𑀩𑁂𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀲𑀤𑁂𑀯𑀓𑁄 𑀘𑁂𑀧𑀺, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀮𑁄𑀓𑁄 𑀲𑀫𑀸𑀭𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁄 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀻 𑀧𑀚𑀸 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀲𑀤𑁂𑀯𑀓𑀲𑁆𑀲𑀧𑀺𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲 𑀲𑀫𑀸𑀭𑀓𑀲𑁆𑀲 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀲𑁆𑀲 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑁇 𑀇𑀫𑁂 𑀘𑁂𑀧𑀺, 𑀯𑀸𑀲𑁂𑀝𑁆𑀞, 𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀉𑀧𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀇𑀫𑁂𑀲𑀫𑁆𑀧𑀺𑀲𑁆𑀲 𑀫𑀳𑀸𑀲𑀸𑀮𑀸𑀦𑀁 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 ( )𑁇 𑀓𑁄 𑀧𑀦 𑀯𑀸𑀤𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
