# 𑁧𑁇𑁩𑁇𑁬 𑀚𑀻𑀯𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀚𑀻𑀯𑀓𑀲𑀼𑀢𑁆𑀢

(26.)

257. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑀻𑀯𑀓𑀫𑁆𑀩𑀯𑀦𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀬𑀢𑁄 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺; 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

258. “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀬𑀢𑁄 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺; 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

259. “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀢𑁆𑀢𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀧𑀭𑀳𑀺𑀢𑀸𑀬𑀸”𑀢𑀺? “𑀬𑀢𑁄 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀧𑀭𑀁 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀧𑀭𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀢𑁆𑀢𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀧𑀭𑀳𑀺𑀢𑀸𑀬𑀸”𑀢𑀺𑁇

260. “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀢𑁆𑀢𑀳𑀺𑀢𑀸𑀬 𑀘 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀳𑀺𑀢𑀸𑀬 𑀘𑀸”𑀢𑀺? “𑀬𑀢𑁄 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀲𑁄𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀼𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀥𑀸𑀭𑀡𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀲𑀼𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀬 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀘 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺𑁇 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀢𑁆𑀢𑀳𑀺𑀢𑀸𑀬 𑀘 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀳𑀺𑀢𑀸𑀬 𑀘𑀸”𑀢𑀺𑁇
