# 𑁧𑁇𑁩𑁇𑁯 𑀅𑀓𑁆𑀔𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀅𑀓𑁆𑀔𑀡𑀲𑀼𑀢𑁆𑀢

(29.)

272\. “‘𑀔𑀡𑀓𑀺𑀘𑁆𑀘𑁄 𑀮𑁄𑀓𑁄, 𑀔𑀡𑀓𑀺𑀘𑁆𑀘𑁄 𑀮𑁄𑀓𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀪𑀸𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀔𑁄 𑀲𑁄 𑀚𑀸𑀦𑀸𑀢𑀺 𑀔𑀡𑀁 𑀯𑀸 𑀅𑀓𑁆𑀔𑀡𑀁 𑀯𑀸𑁇 𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀓𑁆𑀔𑀡𑀸 𑀅𑀲𑀫𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸, 𑀥𑀫𑁆𑀫𑁄 𑀘 𑀤𑁂𑀲𑀺𑀬𑀢𑀺 𑀑𑀧𑀲𑀫𑀺𑀓𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀓𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀻 𑀲𑀼𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁄; 𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁧)

273\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸, 𑀥𑀫𑁆𑀫𑁄 𑀘 𑀤𑁂𑀲𑀺𑀬𑀢𑀺 𑀑𑀧𑀲𑀫𑀺𑀓𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀓𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀻 𑀲𑀼𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁄; 𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇 (𑁨)

274\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇 (𑁩)

275\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀁 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂… 𑁇 (𑁪)

276\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀭𑁂𑀲𑀼 𑀫𑀺𑀮𑀓𑁆𑀔𑁂𑀲𑀼, 𑀬𑀢𑁆𑀣 𑀦𑀢𑁆𑀣𑀺 𑀕𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀉𑀧𑀸𑀲𑀓𑀸𑀦𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁…𑀧𑁂… 𑁇 (𑁫)

277\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀯𑀺𑀧𑀭𑀻𑀢𑀤𑀲𑁆𑀲𑀦𑁄—  ‘𑀦𑀢𑁆𑀣𑀺 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀬𑀺𑀝𑁆𑀞𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀳𑀼𑀢𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀼𑀓𑀢𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀨𑀮𑀁 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀅𑀬𑀁 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀧𑀭𑁄 𑀮𑁄𑀓𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀫𑀸𑀢𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀧𑀺𑀢𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀲𑀢𑁆𑀢𑀸 𑀑𑀧𑀧𑀸𑀢𑀺𑀓𑀸, 𑀦𑀢𑁆𑀣𑀺 𑀮𑁄𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀫𑁆𑀫𑀕𑁆𑀕𑀢𑀸 𑀲𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀬𑁂 𑀇𑀫𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀧𑀭𑀜𑁆𑀘 𑀮𑁄𑀓𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀻’𑀢𑀺…𑀧𑁂… 𑁇 (𑁬)

278\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀚𑀍𑀑 𑀏𑀍𑀅𑀫𑀽𑀕𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀩𑀮𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀫𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁭)

279\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑁇 𑀥𑀫𑁆𑀫𑁄 𑀘 𑀦 𑀤𑁂𑀲𑀺𑀬𑀢𑀺 𑀑𑀧𑀲𑀫𑀺𑀓𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀓𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀻 𑀲𑀼𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁄𑁇 𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀜𑁆𑀜𑀯𑀸 𑀅𑀚𑀍𑀑 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑁄 𑀧𑀝𑀺𑀩𑀮𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀫𑁄 𑀅𑀓𑁆𑀔𑀡𑁄 𑀅𑀲𑀫𑀬𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 (𑁮)

280\. 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀅𑀓𑁆𑀔𑀡𑀸 𑀅𑀲𑀫𑀬𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇

281\. 𑀏𑀓𑁄𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀡𑁄 𑀘 𑀲𑀫𑀬𑁄 𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑁇 𑀓𑀢𑀫𑁄 𑀏𑀓𑁄? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀘 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑁇 𑀥𑀫𑁆𑀫𑁄 𑀘 𑀤𑁂𑀲𑀺𑀬𑀢𑀺 𑀑𑀧𑀲𑀫𑀺𑀓𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀺𑀓𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀕𑀸𑀫𑀻 𑀲𑀼𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁄𑁇 𑀅𑀬𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀲𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀧𑀜𑁆𑀜𑀯𑀸 𑀅𑀚𑀍𑀑 𑀅𑀦𑁂𑀍𑀅𑀫𑀽𑀕𑁄 𑀧𑀝𑀺𑀩𑀮𑁄 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁄𑀯 𑀔𑀡𑁄 𑀘 𑀲𑀫𑀬𑁄 𑀘 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸𑀲𑀸𑀬𑀸𑀢𑀺𑁇

282\. _𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑀸𑀪𑀁 𑀮𑀤𑁆𑀥𑀸𑀦,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂;_  
_𑀬𑁂 𑀔𑀡𑀁 𑀦𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀅𑀢𑀺𑀦𑀸𑀫𑁂𑀦𑁆𑀢𑀺 𑀢𑁂 𑀔𑀡𑀁𑁇_  


283\. _𑀩𑀳𑀽 𑀳𑀺 𑀅𑀓𑁆𑀔𑀡𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸;_  
_𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀮𑁄𑀓𑁂,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀸𑁇_  


284\. _𑀢𑀬𑀺𑀤𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀁,_  
_𑀬𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁 𑀲𑀼𑀤𑀼𑀮𑁆𑀮𑀪𑀁;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀧𑀝𑀺𑀮𑀸𑀪𑁄 𑀘,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀤𑁂𑀲𑀦𑀸;_  
_𑀅𑀮𑀁 𑀯𑀸𑀬𑀫𑀺𑀢𑀼𑀁 𑀢𑀢𑁆𑀣,_  
_𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑁂𑀦 𑀚𑀦𑁆𑀢𑀼𑀦𑀸𑁇_  


285\. _𑀓𑀣𑀁 𑀯𑀺𑀚𑀜𑁆𑀜𑀸 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀔𑀡𑁄 𑀯𑁂 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀔𑀡𑀸𑀢𑀻𑀢𑀸 𑀳𑀺 𑀲𑁄𑀘𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀸𑁇_  


286\. _𑀇𑀥 𑀘𑁂 𑀦𑀁 𑀯𑀺𑀭𑀸𑀥𑁂𑀢𑀺,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀬𑀸𑀫𑀢𑀁;_  
_𑀯𑀸𑀡𑀺𑀚𑁄𑀯 𑀅𑀢𑀻𑀢𑀢𑁆𑀣𑁄,_  
_𑀘𑀺𑀭𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀢𑀧𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


287\. _𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀺𑀯𑀼𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀅𑀧𑀭𑀸𑀥𑀺𑀓𑁄;_  
_𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀘𑀺𑀭𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀲𑁆𑀲𑀢𑀺𑁇_  


288\. _𑀬𑁂 𑀘 𑀮𑀤𑁆𑀥𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀢𑁆𑀢𑀁,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑁂;_  
_𑀅𑀓𑀁𑀲𑀼 𑀲𑀢𑁆𑀣𑀼 𑀯𑀘𑀦𑀁,_  
_𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀯𑀸𑁇_  


289\. _𑀔𑀡𑀁 𑀧𑀘𑁆𑀘𑀯𑀺𑀤𑀼𑀁 𑀮𑁄𑀓𑁂,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁;_  
_𑀬𑁂 𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼,_  
_𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑀁𑁇_  


290\. _𑀬𑁂 𑀲𑀁𑀯𑀭𑀸 𑀘𑀓𑁆𑀔𑀼𑀫𑀢𑀸,_  
_𑀤𑁂𑀲𑀺𑀢𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀢𑁂𑀲𑀼 𑀕𑀼𑀢𑁆𑀢𑁄 𑀲𑀤𑀸 𑀲𑀢𑁄,_  
_𑀯𑀺𑀳𑀭𑁂 𑀅𑀦𑀯𑀲𑁆𑀲𑀼𑀢𑁄𑁇_  


291\. _𑀲𑀩𑁆𑀩𑁂 𑀅𑀦𑀼𑀲𑀬𑁂 𑀙𑁂𑀢𑁆𑀯𑀸,_  
_𑀫𑀸𑀭𑀥𑁂𑀬𑁆𑀬𑀧𑀭𑀸𑀦𑀼𑀕𑁂;_  
_𑀢𑁂 𑀯𑁂 𑀧𑀸𑀭𑀗𑁆𑀕𑀢𑀸 𑀮𑁄𑀓𑁂,_  
_𑀬𑁂 𑀧𑀢𑁆𑀢𑀸 𑀆𑀲𑀯𑀓𑁆𑀔𑀬𑀦𑁆”𑀢𑀺𑁇_  


---

292\. 𑀦𑀯𑀫𑀁𑁇
