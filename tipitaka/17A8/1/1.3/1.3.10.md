# 𑁧𑁇𑁩𑁇𑁧𑁦 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀫𑀳𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀫𑀳𑀸𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀼𑀢𑁆𑀢

(30.)

293\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀪𑀕𑁆𑀕𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀼𑀁𑀲𑀼𑀫𑀸𑀭𑀕𑀺𑀭𑁂 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀘𑁂𑀢𑀻𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀺𑀘𑁆𑀙𑀲𑁆𑀲; 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲; 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀲𑁆𑀲; 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲; 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲; 𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲; 𑀧𑀜𑁆𑀜𑀯𑀢𑁄 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

294\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁—  𑀪𑀕𑁆𑀕𑁂𑀲𑀼 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀕𑀺𑀭𑁂 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀘𑁂𑀢𑀻𑀲𑀼 𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑀤𑀸𑀬𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

295\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀬𑀁 𑀢𑀁 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑀺—  ‘𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀺𑀘𑁆𑀙𑀲𑁆𑀲; 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲; 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀲𑁆𑀲; 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲; 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲; 𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲; 𑀧𑀜𑁆𑀜𑀯𑀢𑁄 𑀅𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀲𑁆𑀲𑀸’𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀇𑀫𑀫𑁆𑀧𑀺 𑀅𑀝𑁆𑀞𑀫𑀁 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺—  ‘𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲 𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄’𑀢𑀺𑁇

296\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑁆𑀲𑀲𑀺, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺, 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺 𑀯𑀺𑀯𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀯𑀺𑀯𑁂𑀓𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀧𑀞𑀫𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇

297\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑁆𑀲𑀲𑀺, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺, 𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀸𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀲𑀸𑀤𑀦𑀁 𑀘𑁂𑀢𑀲𑁄 𑀏𑀓𑁄𑀤𑀺𑀪𑀸𑀯𑀁 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀁 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀚𑀁 𑀧𑀻𑀢𑀺𑀲𑀼𑀔𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇

298\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑁆𑀲𑀲𑀺, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀬𑀸𑀯𑀤𑁂𑀯 𑀆𑀓𑀗𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺, 𑀧𑀻𑀢𑀺𑀬𑀸 𑀘 𑀯𑀺𑀭𑀸𑀕𑀸 𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀘 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀲𑀺 𑀲𑀢𑁄 𑀘 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀸𑀬𑁂𑀦 𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀲𑁆𑀲𑀲𑀺 𑀬𑀁 𑀢𑀁 𑀅𑀭𑀺𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀉𑀧𑁂𑀓𑁆𑀔𑀓𑁄 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀻’𑀢𑀺 𑀢𑀢𑀺𑀬𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇

302\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀇𑀫𑁂 𑀘 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑁆𑀲𑀲𑀺, 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻, 𑀢𑀢𑁄 𑀢𑀼𑀬𑁆𑀳𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀁 𑀉𑀮𑁆𑀮𑀺𑀢𑁆𑀢𑀸𑀯𑀮𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀯𑀸𑀢𑀁 𑀨𑀼𑀲𑀺𑀢𑀕𑁆𑀕𑀍𑀅𑀁 𑀧𑀺𑀳𑀺𑀢𑀯𑀸𑀢𑀧𑀸𑀦𑀁; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑁂 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀔𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀭𑀢𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀸𑀬 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀑𑀓𑁆𑀓𑀫𑀦𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲𑁇

303\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀇𑀫𑁂 𑀘 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑁆𑀲𑀲𑀺, 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻, 𑀢𑀢𑁄 𑀢𑀼𑀬𑁆𑀳𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁄 𑀕𑁄𑀦𑀓𑀢𑁆𑀣𑀢𑁄 𑀧𑀝𑀺𑀓𑀢𑁆𑀣𑀢𑁄 𑀧𑀝𑀮𑀺𑀓𑀢𑁆𑀣𑀢𑁄 𑀓𑀤𑀮𑀺𑀫𑀺𑀕𑀧𑀯𑀭𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀭𑀡𑁄 𑀲𑁅𑀢𑁆𑀢𑀭𑀘𑁆𑀙𑀤𑁄 𑀉𑀪𑀢𑁄𑀮𑁄𑀳𑀺𑀢𑀓𑀽𑀧𑀥𑀸𑀦𑁄; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑁂 𑀢𑀺𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑀓𑀲𑀬𑀦𑀸𑀲𑀦𑀁 𑀔𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀭𑀢𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀸𑀬 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀑𑀓𑁆𑀓𑀫𑀦𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲𑁇

304\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀇𑀫𑁂 𑀘 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀲𑁆𑀲𑀲𑀺, 𑀇𑀫𑁂𑀲𑀜𑁆𑀘 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻, 𑀢𑀢𑁄 𑀢𑀼𑀬𑁆𑀳𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀸 𑀦𑀸𑀦𑀸𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀧𑁆𑀧𑀺 𑀦𑀯𑀦𑀻𑀢𑀁 𑀢𑁂𑀮𑀁 𑀫𑀥𑀼 𑀨𑀸𑀡𑀺𑀢𑀁; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑁂 𑀧𑀽𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀔𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀭𑀢𑀺𑀬𑀸 𑀅𑀧𑀭𑀺𑀢𑀲𑁆𑀲𑀸𑀬 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀑𑀓𑁆𑀓𑀫𑀦𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑁆𑀲𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥, 𑀆𑀬𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀇𑀥𑁂𑀯 𑀘𑁂𑀢𑀻𑀲𑀼 𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑀤𑀸𑀬𑁂 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇

305\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀇𑀫𑀺𑀦𑀸 𑀑𑀯𑀸𑀤𑁂𑀦 𑀑𑀯𑀤𑀺𑀢𑁆𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁—  𑀘𑁂𑀢𑀻𑀲𑀼 𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑀤𑀸𑀬𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀪𑀕𑁆𑀕𑁂𑀲𑀼 𑀲𑀼𑀲𑀼𑀫𑀸𑀭𑀕𑀺𑀭𑁂 𑀪𑁂𑀲𑀓𑀍𑀆𑀯𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀻𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀝𑁆𑀞 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀁 𑀲𑀼𑀡𑀸𑀣…𑀧𑁂…  𑀓𑀢𑀫𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀸? 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀺𑀘𑁆𑀙𑀲𑁆𑀲; 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲; 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀲𑁆𑀲; 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲; 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲; 𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲; 𑀧𑀜𑁆𑀜𑀯𑀢𑁄 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀲𑁆𑀲; 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲 𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄𑁇

306\. ‘𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀸𑀦𑁄 ‘𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀧𑀜𑁆𑀜𑀯𑀸 𑀲𑀫𑀸𑀦𑁄 ‘𑀧𑀜𑁆𑀜𑀯𑀸𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑁄 𑀲𑀫𑀸𑀦𑁄 ‘𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑁄𑀢𑀺 𑀫𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺𑁇 ‘𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀺𑀘𑁆𑀙𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 (𑁧)

307\. ‘𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺 𑀇𑀢𑀭𑀻𑀢𑀭𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦𑁇 ‘𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 (𑁨)

308\. ‘𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀪𑀯𑀦𑁆𑀢𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀸𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄 𑀭𑀸𑀚𑀸𑀦𑁄 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑀸𑁇 𑀢𑀢𑁆𑀭 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑁂𑀓𑀦𑀺𑀦𑁆𑀦𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀯𑁂𑀓𑀧𑁄𑀡𑁂𑀦 𑀯𑀺𑀯𑁂𑀓𑀧𑀩𑁆𑀪𑀸𑀭𑁂𑀦 𑀯𑀺𑀯𑁂𑀓𑀝𑁆𑀞𑁂𑀦 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀸𑀪𑀺𑀭𑀢𑁂𑀦 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀼 𑀉𑀬𑁆𑀬𑁄𑀚𑀦𑀺𑀓𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀁𑀬𑁂𑀯 𑀓𑀣𑀁 𑀓𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 ‘𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 (𑁩)

309\. ‘𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁, 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 ‘𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀻𑀢𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 (𑁪)

310\. ‘𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑀺𑀫𑀸 𑀳𑁄𑀢𑀺 𑀧𑀭𑀫𑁂𑀦 𑀲𑀢𑀺𑀦𑁂𑀧𑀓𑁆𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀘𑀺𑀭𑀓𑀢𑀫𑁆𑀧𑀺 𑀘𑀺𑀭𑀪𑀸𑀲𑀺𑀢𑀫𑁆𑀧𑀺 𑀲𑀭𑀺𑀢𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀭𑀺𑀢𑀸𑁇 ‘𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄, 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 (𑁫)

311\. ‘𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀯𑀺𑀘𑁆𑀘𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺…𑀧𑁂…  𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀛𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 ‘𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 (𑁬)

312\. ‘𑀧𑀜𑁆𑀜𑀯𑀢𑁄 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺 𑀉𑀤𑀬𑀢𑁆𑀣𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸𑁇 ‘𑀧𑀜𑁆𑀜𑀯𑀢𑁄 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀲𑁆𑀲𑀸’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 (𑁭)

313\. ‘𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲 𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄’𑀢𑀺, 𑀇𑀢𑀺 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁𑁇 𑀓𑀺𑀜𑁆𑀘𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀁? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀧𑀜𑁆𑀘𑀦𑀺𑀭𑁄𑀥𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀢𑀺 𑀧𑀲𑀻𑀤𑀢𑀺 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇 ‘𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁄, 𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄, 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑁄 𑀧𑀧𑀜𑁆𑀘𑀸𑀭𑀸𑀫𑀲𑁆𑀲 𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑀺𑀦𑁄’𑀢𑀺, 𑀇𑀢𑀺 𑀬𑀁 𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀇𑀤𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 (𑁮)

314\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀢𑀢𑁆𑀣𑁂𑀯 𑀘𑁂𑀢𑀻𑀲𑀼 𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑀤𑀸𑀬𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

315\. _“𑀫𑀫 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀲𑀢𑁆𑀣𑀸 𑀮𑁄𑀓𑁂 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄;_  
_𑀫𑀦𑁄𑀫𑀬𑁂𑀦 𑀓𑀸𑀬𑁂𑀦,_  
_𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇_  


316\. _𑀬𑀣𑀸 𑀫𑁂 𑀅𑀳𑀼 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀢𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀺 𑀤𑁂𑀲𑀬𑀺;_  
_𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀭𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀦𑀺𑀧𑁆𑀧𑀧𑀜𑁆𑀘𑀁 𑀅𑀤𑁂𑀲𑀬𑀺𑁇_  


317\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀲𑀸𑀲𑀦𑁂 𑀭𑀢𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀦𑁆”𑀢𑀺𑁇_

321\. _𑀤𑁆𑀯𑁂 𑀉𑀕𑁆𑀕𑀸 𑀤𑁆𑀯𑁂 𑀘 𑀳𑀢𑁆𑀣𑀓𑀸,_

_𑀤𑁆𑀯𑁂 𑀩𑀮𑀸 𑀅𑀓𑁆𑀔𑀡𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁂𑀦 𑀢𑁂 𑀤𑀲𑀸𑀢𑀺𑁇_
