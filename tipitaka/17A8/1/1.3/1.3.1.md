# 𑁧𑁇𑁩𑁇𑁧 𑀧𑀞𑀫𑁅𑀕𑁆𑀕𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑁅𑀕𑁆𑀕𑀲𑀼𑀢𑁆𑀢

(21.)

213\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀕𑁆𑀕𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀁 𑀥𑀸𑀭𑁂𑀣𑀸”𑀢𑀺𑁇 𑀇𑀤𑀫𑀯𑁄𑀘 𑀪𑀕𑀯𑀸𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀲𑀼𑀕𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

214\. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀉𑀕𑁆𑀕𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀕𑁆𑀕𑁄 𑀕𑀳𑀧𑀢𑀺 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑁄 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀉𑀕𑁆𑀕𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀁 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀢𑀤𑀯𑁄𑀘—

215\. “𑀅𑀝𑁆𑀞𑀳𑀺 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀓𑀢𑀫𑁂 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀝𑁆𑀞 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸, 𑀬𑁂𑀳𑀺 𑀢𑁆𑀯𑀁 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄”𑀢𑀺? “𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄’𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂 𑀫𑁂 𑀅𑀝𑁆𑀞 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀳𑀺; 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀕𑀳𑀧𑀢𑀻”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀕𑁆𑀕𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀉𑀕𑁆𑀕𑁄 𑀕𑀳𑀧𑀢𑀺 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀤𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀞𑀫𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀅𑀤𑁆𑀤𑀲𑀁; 𑀲𑀳 𑀤𑀲𑁆𑀲𑀦𑁂𑀦𑁂𑀯 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀞𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧)

216\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁; 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑀁, 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄 𑀫𑁂 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁄 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑁄 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑁄 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀢𑀢𑁆𑀣𑁂𑀯 𑀩𑀼𑀤𑁆𑀥𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑀁𑀖𑀜𑁆𑀘 𑀲𑀭𑀡𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀜𑁆𑀘𑀫𑀸𑀦𑀺 𑀘 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨)

217\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀓𑁄𑀫𑀸𑀭𑀺𑀬𑁄 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀢𑀸 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀸 𑀧𑀚𑀸𑀧𑀢𑀺𑀬𑁄 𑀏𑀢𑀤𑀯𑀘𑀁—  ‘𑀫𑀬𑀸 𑀔𑁄, 𑀪𑀕𑀺𑀦𑀺𑀬𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀜𑁆𑀘𑀫𑀸𑀦𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺 𑀲𑀫𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑀦𑀺𑁇 𑀬𑀸 𑀇𑀘𑁆𑀙𑀢𑀺 𑀲𑀸 𑀇𑀥𑁂𑀯 𑀪𑁄𑀕𑁂 𑀘 𑀪𑀼𑀜𑁆𑀚𑀢𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀘 𑀓𑀭𑁄𑀢𑀼, 𑀲𑀓𑀸𑀦𑀺 𑀯𑀸 𑀜𑀸𑀢𑀺𑀓𑀼𑀮𑀸𑀦𑀺 𑀕𑀘𑁆𑀙𑀢𑀼𑁇 𑀳𑁄𑀢𑀺 𑀯𑀸 𑀧𑀦 𑀧𑀼𑀭𑀺𑀲𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄, 𑀓𑀲𑁆𑀲 𑀯𑁄 𑀤𑀫𑁆𑀫𑀻’𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑁂𑀝𑁆𑀞𑀸 𑀧𑀚𑀸𑀧𑀢𑀺 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀫𑀁, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀧𑀓𑁆𑀓𑁄𑀲𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸𑀫𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀚𑀸𑀧𑀢𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀪𑀺𑀗𑁆𑀕𑀸𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀑𑀡𑁄𑀚𑁂𑀲𑀺𑀁𑁇 𑀓𑁄𑀫𑀸𑀭𑀺𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀸𑀭𑀁 𑀧𑀭𑀺𑀘𑁆𑀘𑀚𑀦𑁆𑀢𑁄 𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀢𑀺𑀬𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁩)

218\. 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀔𑁄 𑀧𑀦 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀮𑁂 𑀪𑁄𑀕𑀸𑁇 𑀢𑁂 𑀘 𑀔𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀪𑀢𑁆𑀢𑀸 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑁂𑀳𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁪)

219\. 𑀬𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀸𑀫𑀺; 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁𑀬𑁂𑀯 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀸𑀫𑀺, 𑀦𑁄 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁫)

220\. 𑀲𑁄 𑀘𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺; 𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁𑀬𑁂𑀯 𑀲𑀼𑀡𑁄𑀫𑀺, 𑀦𑁄 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀫𑁂 𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀅𑀳𑀫𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀙𑀝𑁆𑀞𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁬)

221\. 𑀅𑀦𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀔𑁄 𑀧𑀦 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑁂𑀯𑀢𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺—  ‘𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀁 𑀯𑀤𑀸𑀫𑀺—  ‘𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀣 𑀯𑀸 𑀏𑀯𑀁 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂 𑀤𑁂𑀯𑀢𑀸 𑀦𑁄 𑀯𑀸 𑀯𑀤𑁂𑀬𑁆𑀬𑀸𑀣, 𑀅𑀣 𑀔𑁄 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑁄’𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀢𑀢𑁄𑀦𑀺𑀤𑀸𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀦𑁆𑀦𑀢𑀺𑀁—  ‘𑀫𑀁 𑀯𑀸 𑀤𑁂𑀯𑀢𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀅𑀳𑀁 𑀯𑀸 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀻’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀢𑁆𑀢𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁭)

222\. 𑀬𑀸𑀦𑀺𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀤𑁂𑀲𑀺𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑁄𑀭𑀫𑁆𑀪𑀸𑀕𑀺𑀬𑀸𑀦𑀺 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀺, 𑀦𑀸𑀳𑀁 𑀢𑁂𑀲𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀢𑁆𑀢𑀦𑀺 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀁 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀝𑁆𑀞𑀫𑁄 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁄 𑀅𑀩𑁆𑀪𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁮)

223\. 𑀇𑀫𑁂 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀝𑁆𑀞 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀦 𑀘 𑀔𑁄 𑀅𑀳𑀁 𑀚𑀸𑀦𑀸𑀫𑀺—  𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀸𑀳𑀁 𑀅𑀝𑁆𑀞𑀳𑀺 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄”𑀢𑀺𑁇

224\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀕𑁆𑀕𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀦𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀯𑀢𑀓𑁄 𑀅𑀳𑁄𑀲𑀺 𑀉𑀕𑁆𑀕𑁂𑀦 𑀕𑀳𑀧𑀢𑀺𑀦𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑁄, 𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀪𑀕𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇

225\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀬𑀣𑀸 𑀢𑀁 𑀉𑀕𑁆𑀕𑁄 𑀕𑀳𑀧𑀢𑀺 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑁄 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬, 𑀇𑀫𑁂𑀳𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀉𑀕𑁆𑀕𑁄 𑀕𑀳𑀧𑀢𑀺 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑁄 𑀫𑀬𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀇𑀫𑁂𑀳𑀺 𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑁂𑀳𑀺 𑀅𑀩𑁆𑀪𑀼𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀉𑀕𑁆𑀕𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀁 𑀥𑀸𑀭𑁂𑀳𑀻”𑀢𑀺𑁇
