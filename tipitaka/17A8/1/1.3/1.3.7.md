# 𑁧𑁇𑁩𑁇𑁭 𑀧𑀞𑀫𑀩𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀕𑀳𑀧𑀢𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀩𑀮𑀲𑀼𑀢𑁆𑀢

(27.)

262\. “𑀅𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀮𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞? 𑀭𑀼𑀡𑁆𑀡𑀩𑀮𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀭𑀓𑀸, 𑀓𑁄𑀥𑀩𑀮𑀸 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀸, 𑀆𑀯𑀼𑀥𑀩𑀮𑀸 𑀘𑁄𑀭𑀸, 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀩𑀮𑀸 𑀭𑀸𑀚𑀸𑀦𑁄, 𑀉𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀩𑀮𑀸 𑀩𑀸𑀮𑀸, 𑀦𑀺𑀚𑁆𑀛𑀢𑁆𑀢𑀺𑀩𑀮𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸, 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀦𑀩𑀮𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸, 𑀔𑀦𑁆𑀢𑀺𑀩𑀮𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀩𑀮𑀸𑀦𑀻”𑀢𑀺𑁇
