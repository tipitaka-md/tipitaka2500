# 𑁧𑁇𑁧𑁇𑁬 𑀤𑀼𑀢𑀺𑀬𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀲𑀼𑀢𑁆𑀢

(6.)

39. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀮𑁄𑀓𑁄 𑀘 𑀅𑀝𑁆𑀞 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀮𑀸𑀪𑁄 𑀘, 𑀅𑀮𑀸𑀪𑁄 𑀘, 𑀬𑀲𑁄 𑀘, 𑀅𑀬𑀲𑁄 𑀘, 𑀦𑀺𑀦𑁆𑀤𑀸 𑀘, 𑀧𑀲𑀁𑀲𑀸 𑀘, 𑀲𑀼𑀔𑀜𑁆𑀘, 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀸 𑀮𑁄𑀓𑀁 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺, 𑀮𑁄𑀓𑁄 𑀘 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑁂 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀢𑀺𑁇

40. 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑁄𑀧𑀺 𑀅𑀮𑀸𑀪𑁄𑀧𑀺 𑀬𑀲𑁄𑀧𑀺 𑀅𑀬𑀲𑁄𑀧𑀺 𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀺 𑀧𑀲𑀁𑀲𑀸𑀧𑀺 𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺𑁇 𑀲𑀼𑀢𑀯𑀢𑁄𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑁄𑀧𑀺 𑀅𑀮𑀸𑀪𑁄𑀧𑀺 𑀬𑀲𑁄𑀧𑀺 𑀅𑀬𑀲𑁄𑀧𑀺 𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀺 𑀧𑀲𑀁𑀲𑀸𑀧𑀺 𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑁄 𑀯𑀺𑀲𑁂𑀲𑁄 𑀓𑁄 𑀅𑀥𑀺𑀧𑁆𑀧𑀬𑀸𑀲𑁄 𑀓𑀺𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀦𑀸”𑀢𑀺? “𑀪𑀕𑀯𑀁𑀫𑀽𑀮𑀓𑀸 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀸 𑀪𑀕𑀯𑀁𑀦𑁂𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀕𑀯𑀁𑀧𑀝𑀺𑀲𑀭𑀡𑀸𑁇 𑀲𑀸𑀥𑀼 𑀯𑀢, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀏𑀢𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄𑁇 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

41. “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀡𑀸𑀣, 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀣, 𑀪𑀸𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑁄𑁇 𑀲𑁄 𑀦 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀮𑀸𑀪𑁄; 𑀲𑁄 𑀘 𑀔𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄 𑀤𑀼𑀓𑁆𑀔𑁄 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀮𑀸𑀪𑁄…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀬𑀲𑁄…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀬𑀲𑁄…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀺𑀦𑁆𑀤𑀸…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀲𑀁𑀲𑀸…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑁄 𑀦 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁; 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀦𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

42. 𑀢𑀲𑁆𑀲 𑀮𑀸𑀪𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀅𑀮𑀸𑀪𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀬𑀲𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀅𑀬𑀲𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀧𑀲𑀁𑀲𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇 𑀲𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀅𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀅𑀮𑀸𑀪𑁂 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀬𑀲𑀁 𑀅𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀅𑀬𑀲𑁂 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀲𑀁𑀲𑀁 𑀅𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑀺𑀦𑁆𑀤𑀸𑀬 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀅𑀦𑀼𑀭𑁄𑀥𑀯𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺𑁇 ‘𑀦 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇

43. 𑀲𑀼𑀢𑀯𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑁄𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀮𑀸𑀪𑁄; 𑀲𑁄 𑀘 𑀔𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑁄 𑀤𑀼𑀓𑁆𑀔𑁄 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁄’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀮𑀸𑀪𑁄…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀬𑀲𑁄…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀅𑀬𑀲𑁄…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀦𑀺𑀦𑁆𑀤𑀸…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀲𑀁𑀲𑀸…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀼𑀔𑀁…  𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀲𑁄 𑀇𑀢𑀺 𑀧𑀝𑀺𑀲𑀜𑁆𑀘𑀺𑀓𑁆𑀔𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀔𑁄 𑀫𑁂 𑀇𑀤𑀁 𑀤𑀼𑀓𑁆𑀔𑀁; 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀅𑀦𑀺𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀦𑁆’𑀢𑀺 𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺𑁇

44. 𑀢𑀲𑁆𑀲 𑀮𑀸𑀪𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀅𑀮𑀸𑀪𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀬𑀲𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀅𑀬𑀲𑁄𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀦𑀺𑀦𑁆𑀤𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀧𑀲𑀁𑀲𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀲𑀼𑀔𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀫𑁆𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇 𑀲𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀦𑀸𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀅𑀮𑀸𑀪𑁂 𑀦𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀬𑀲𑀁 𑀦𑀸𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀅𑀬𑀲𑁂 𑀦𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀲𑀁𑀲𑀁 𑀦𑀸𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦𑀺𑀦𑁆𑀤𑀸𑀬 𑀦𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀁 𑀦𑀸𑀦𑀼𑀭𑀼𑀚𑁆𑀛𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑁂 𑀦𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀅𑀦𑀼𑀭𑁄𑀥𑀯𑀺𑀭𑁄𑀥𑀯𑀺𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀚𑀭𑀸𑀬 𑀫𑀭𑀡𑁂𑀦 𑀲𑁄𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑁂𑀳𑀺 𑀉𑀧𑀸𑀬𑀸𑀲𑁂𑀳𑀺𑁇 ‘𑀧𑀭𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀸’𑀢𑀺 𑀯𑀤𑀸𑀫𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀲𑁂𑀲𑁄 𑀅𑀬𑀁 𑀅𑀥𑀺𑀧𑁆𑀧𑀬𑀸𑀲𑁄 𑀇𑀤𑀁 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀲𑀼𑀢𑀯𑀢𑁄 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀼𑀢𑀯𑀢𑀸 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁂𑀦𑀸𑀢𑀺𑁇

45. _𑀮𑀸𑀪𑁄 𑀅𑀮𑀸𑀪𑁄 𑀘 𑀬𑀲𑀸𑀬𑀲𑁄 𑀘,_  
_𑀦𑀺𑀦𑁆𑀤𑀸 𑀧𑀲𑀁𑀲𑀸 𑀘 𑀲𑀼𑀔𑀁 𑀤𑀼𑀔𑀜𑁆𑀘;_  
_𑀏𑀢𑁂 𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀫𑀦𑀼𑀚𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸,_  
_𑀅𑀲𑀲𑁆𑀲𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸𑁇_  


46. _𑀏𑀢𑁂 𑀘 𑀜𑀢𑁆𑀯𑀸 𑀲𑀢𑀺𑀫𑀸 𑀲𑀼𑀫𑁂𑀥𑁄,_  
_𑀅𑀯𑁂𑀓𑁆𑀔𑀢𑀺 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑁂;_  
_𑀇𑀝𑁆𑀞𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀫𑀣𑁂𑀦𑁆𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁,_  
_𑀅𑀦𑀺𑀝𑁆𑀞𑀢𑁄 𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑀫𑁂𑀢𑀺𑁇_  


47. _𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀭𑁄𑀥𑀸 𑀅𑀣 𑀯𑀸 𑀯𑀺𑀭𑁄𑀥𑀸,_  
_𑀯𑀺𑀥𑀽𑀧𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑀸 𑀦 𑀲𑀦𑁆𑀢𑀺;_  
_𑀧𑀤𑀜𑁆𑀘 𑀜𑀢𑁆𑀯𑀸 𑀯𑀺𑀭𑀚𑀁 𑀅𑀲𑁄𑀓𑀁,_  
_𑀲𑀫𑁆𑀫𑀧𑁆𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀪𑀯𑀲𑁆𑀲 𑀧𑀸𑀭𑀕𑀽”𑀢𑀺𑁇_  


---

48. 𑀙𑀝𑁆𑀞𑀁𑁇
