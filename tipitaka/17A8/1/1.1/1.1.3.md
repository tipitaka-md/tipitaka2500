# 𑁧𑁇𑁧𑁇𑁩 𑀧𑀞𑀫𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(3.)

28\. “𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀧𑁆𑀧𑀺𑀬𑀧𑀲𑀁𑀲𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀧𑀺𑀬𑀕𑀭𑀳𑀻 𑀘, 𑀮𑀸𑀪𑀓𑀸𑀫𑁄 𑀘, 𑀲𑀓𑁆𑀓𑀸𑀭𑀓𑀸𑀫𑁄 𑀘, 𑀅𑀳𑀺𑀭𑀺𑀓𑁄 𑀘, 𑀅𑀦𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀘, 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀘, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀅𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀅𑀕𑀭𑀼 𑀘 𑀅𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇

29\. 𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀅𑀧𑁆𑀧𑀺𑀬𑀧𑀲𑀁𑀲𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀦 𑀧𑀺𑀬𑀕𑀭𑀳𑀻 𑀘, 𑀦 𑀮𑀸𑀪𑀓𑀸𑀫𑁄 𑀘, 𑀦 𑀲𑀓𑁆𑀓𑀸𑀭𑀓𑀸𑀫𑁄 𑀘, 𑀳𑀺𑀭𑀻𑀫𑀸 𑀘 𑀳𑁄𑀢𑀺, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀘, 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀘, 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺 𑀘𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀧𑀺𑀬𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀫𑀦𑀸𑀧𑁄 𑀘 𑀕𑀭𑀼 𑀘 𑀪𑀸𑀯𑀦𑀻𑀬𑁄 𑀘𑀸”𑀢𑀺𑁇
