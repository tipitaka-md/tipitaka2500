# 𑁧𑁇𑁧𑁇𑁨 𑀧𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀜𑀸𑀲𑀼𑀢𑁆𑀢

(2.)

10\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀳𑁂𑀢𑀽 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀁 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑁂𑀫𑀜𑁆𑀘 𑀕𑀸𑀭𑀯𑁄 𑀘𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑁄 𑀳𑁂𑀢𑀼 𑀧𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

11\. 𑀲𑁄 𑀢𑀁 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀁 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑁂𑀫𑀜𑁆𑀘 𑀕𑀸𑀭𑀯𑁄 𑀘, 𑀢𑁂 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺 𑀧𑀭𑀺𑀧𑀜𑁆𑀳𑀢𑀺—  ‘𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀣𑀁; 𑀇𑀫𑀲𑁆𑀲 𑀓𑁄 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀯𑀺𑀯𑀝𑀜𑁆𑀘𑁂𑀯 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺, 𑀅𑀦𑀼𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀢𑀜𑁆𑀘 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀘 𑀓𑀗𑁆𑀔𑀸𑀞𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

12\. 𑀲𑁄 𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀬𑁂𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺—  𑀓𑀸𑀬𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀘𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼 𑀢𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

13\. 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻, 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀳𑁂𑀢𑀼 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

14\. 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀢𑀥𑀭𑁄 𑀲𑀼𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄𑁇 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀥𑀸𑀢𑀸 𑀯𑀘𑀲𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀳𑁂𑀢𑀼 𑀧𑀜𑁆𑀘𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

15\. 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀝𑁆𑀞𑁄 𑀳𑁂𑀢𑀼 𑀙𑀝𑁆𑀞𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

16\. 𑀲𑀁𑀖𑀕𑀢𑁄 𑀔𑁄 𑀧𑀦 𑀅𑀦𑀸𑀦𑀸𑀓𑀣𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀺𑀓𑁄𑁇 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺 𑀧𑀭𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀲𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑀁 𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀫𑁄 𑀳𑁂𑀢𑀼 𑀲𑀢𑁆𑀢𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

17\. 𑀧𑀜𑁆𑀘𑀲𑀼 𑀔𑁄 𑀧𑀦 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺—  ‘𑀇𑀢𑀺 𑀭𑀽𑀧𑀁, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸, 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸…𑀧𑁂…  𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄’𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀫𑁄 𑀳𑁂𑀢𑀼 𑀅𑀝𑁆𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

18\. 𑀢𑀫𑁂𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀏𑀯𑀁 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀁 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑁂𑀫𑀜𑁆𑀘 𑀕𑀸𑀭𑀯𑁄 𑀘𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁧)

19\. ‘𑀢𑀁 𑀔𑁄 𑀧𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀢𑁆𑀣𑀸𑀭𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀕𑀭𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀢𑀺𑀩𑁆𑀩𑀁 𑀳𑀺𑀭𑁄𑀢𑁆𑀢𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀧𑁂𑀫𑀜𑁆𑀘 𑀕𑀸𑀭𑀯𑁄 𑀘, 𑀢𑁂 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺 𑀧𑀭𑀺𑀧𑀜𑁆𑀳𑀢𑀺—  “𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀣𑀁; 𑀇𑀫𑀲𑁆𑀲 𑀓𑁄 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺? 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀯𑀺𑀯𑀝𑀜𑁆𑀘𑁂𑀯 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺, 𑀅𑀦𑀼𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀢𑀜𑁆𑀘 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑁂𑀲𑀼 𑀘 𑀓𑀗𑁆𑀔𑀸𑀞𑀸𑀦𑀺𑀬𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀗𑁆𑀔𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁨)

20\. ‘𑀢𑀁 𑀔𑁄 𑀧𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀬𑁂𑀦 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺—  𑀓𑀸𑀬𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀦 𑀘𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁩)

21\. ‘𑀲𑀻𑀮𑀯𑀸 𑀔𑁄 𑀧𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀁𑀯𑀭𑀲𑀁𑀯𑀼𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀡𑀼𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀯𑀚𑁆𑀚𑁂𑀲𑀼 𑀪𑀬𑀤𑀲𑁆𑀲𑀸𑀯𑀻, 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁪)

22\. ‘𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀼𑀢𑀥𑀭𑁄 𑀲𑀼𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑁄𑁇 𑀬𑁂 𑀢𑁂 𑀥𑀫𑁆𑀫𑀸 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀸 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀅𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀥𑀸𑀢𑀸 𑀯𑀘𑀲𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀫𑀦𑀲𑀸𑀦𑀼𑀧𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁫)

23\. ‘𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀳𑀸𑀦𑀸𑀬, 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀬, 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁬)

24\. ‘𑀲𑀁𑀖𑀕𑀢𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀸𑀦𑀸𑀓𑀣𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀺𑀓𑁄𑁇 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀢𑀺 𑀧𑀭𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀲𑀢𑀺 𑀅𑀭𑀺𑀬𑀁 𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑀁 𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀢𑀺𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁭)

25\. ‘𑀧𑀜𑁆𑀘𑀲𑀼 𑀔𑁄 𑀧𑀦𑀸𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀲𑀼 𑀉𑀤𑀬𑀩𑁆𑀩𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀢𑀺—  𑀇𑀢𑀺 𑀭𑀽𑀧𑀁, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀭𑀽𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄; 𑀇𑀢𑀺 𑀯𑁂𑀤𑀦𑀸…𑀧𑁂…  𑀇𑀢𑀺 𑀲𑀜𑁆𑀜𑀸…  𑀇𑀢𑀺 𑀲𑀗𑁆𑀔𑀸𑀭𑀸…  𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀇𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑁄𑀢𑀺𑁇 𑀅𑀤𑁆𑀥𑀸 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀚𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺 𑀧𑀲𑁆𑀲𑀁 𑀧𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀧𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀕𑀭𑀼𑀢𑁆𑀢𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀲𑀸𑀫𑀜𑁆𑀜𑀸𑀬 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 (𑁮)

26\. 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀺𑀓𑀸𑀬 𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀧𑁆𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬, 𑀧𑀝𑀺𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀸𑀬 𑀪𑀸𑀯𑀦𑀸𑀬 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀬𑀸 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

---

27\. 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
