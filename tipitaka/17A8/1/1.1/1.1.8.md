# 𑁧𑁇𑁧𑁇𑁮 𑀉𑀢𑁆𑀢𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀕𑁆𑀕

* 𑀉𑀢𑁆𑀢𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(8.)

56\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀫𑀳𑀺𑀲𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀝𑀚𑀸𑀮𑀺𑀓𑀸𑀬𑀁𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀭𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

57\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀤𑀺𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀫𑀳𑀺𑀲𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀝𑀚𑀸𑀮𑀺𑀓𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  “𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀭𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

58\. 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀫𑀳𑀺𑀲𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀝𑀚𑀸𑀮𑀺𑀓𑀸𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀲𑁆𑀲𑀯𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸 𑀬𑁂𑀦 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀕𑁆𑀖𑁂, 𑀫𑀸𑀭𑀺𑀲, 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀏𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀫𑀳𑀺𑀲𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀝𑀚𑀸𑀮𑀺𑀓𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺—  ‘𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁…  𑀧𑀭𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀻’”𑀢𑀺𑁇

59\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀳𑀺𑀲𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀲𑀗𑁆𑀔𑁂𑀬𑁆𑀬𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀝𑀚𑀸𑀮𑀺𑀓𑀸𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

60\. “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑀺—  ‘𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁…  𑀧𑀭𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀻’”𑀢𑀺? “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦𑀺𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀲𑀓𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀁, 𑀉𑀤𑀸𑀳𑀼 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀉𑀧𑀫𑀁 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀉𑀧𑀫𑀸𑀬 𑀫𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀼𑀭𑀺𑀲𑀸 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

61\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀕𑀸𑀫𑀲𑁆𑀲 𑀯𑀸 𑀦𑀺𑀕𑀫𑀲𑁆𑀲 𑀯𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀫𑀳𑀸𑀥𑀜𑁆𑀜𑀭𑀸𑀲𑀺𑁇 𑀢𑀢𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀥𑀜𑁆𑀜𑀁 𑀆𑀳𑀭𑁂𑀬𑁆𑀬—  𑀓𑀸𑀚𑁂𑀳𑀺𑀧𑀺 𑀧𑀺𑀝𑀓𑁂𑀳𑀺𑀧𑀺 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂𑀳𑀺𑀧𑀺 𑀅𑀜𑁆𑀚𑀮𑀻𑀳𑀺𑀧𑀺𑁇 𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀢𑀁 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬—  ‘𑀓𑀼𑀢𑁄 𑀇𑀫𑀁 𑀥𑀜𑁆𑀜𑀁 𑀆𑀳𑀭𑀣𑀸’𑀢𑀺, 𑀓𑀣𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄 𑀦𑀼 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “‘𑀅𑀫𑀼𑀫𑁆𑀳𑀸 𑀫𑀳𑀸𑀥𑀜𑁆𑀜𑀭𑀸𑀲𑀺𑀫𑁆𑀳𑀸 𑀆𑀳𑀭𑀸𑀫𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀲𑀫𑁆𑀫𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀫𑀸𑀦𑁄 𑀩𑁆𑀬𑀸𑀓𑀭𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤, 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀢𑀢𑁄 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀫𑀬𑀁 𑀘𑀜𑁆𑀜𑁂 𑀘 𑀪𑀡𑀸𑀫𑀸”𑀢𑀺𑁇

62\. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀪𑀦𑁆𑀢𑁂𑁇 𑀬𑀸𑀯 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀜𑁆𑀘𑀺𑀤𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀦—  ‘𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀢𑀢𑁄 𑀉𑀧𑀸𑀤𑀸𑀬𑀼𑀧𑀸𑀤𑀸𑀬 𑀫𑀬𑀜𑁆𑀘𑀜𑁆𑀜𑁂 𑀘 𑀪𑀡𑀸𑀫𑀸’𑀢𑀺𑁇 𑀏𑀓𑀫𑀺𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀉𑀢𑁆𑀢𑀭, 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

63\. ‘𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀧𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀢𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁…  𑀧𑀭𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀮𑀸𑀪𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄; 𑀅𑀮𑀸𑀪𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀬𑀲𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀬𑀲𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇

64\. 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀮𑀸𑀪𑀁…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇

65\. 𑀓𑀺𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀮𑀸𑀪𑀁…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬?

66\. 𑀬𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀅𑀦𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀏𑀯𑀁𑀲 𑀢𑁂 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑀁 𑀳𑀺𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀮𑀸𑀪𑀁…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀦𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀁 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀢𑁄 𑀏𑀯𑀁𑀲 𑀢𑁂 𑀆𑀲𑀯𑀸 𑀯𑀺𑀖𑀸𑀢𑀧𑀭𑀺𑀍𑀆𑀳𑀸 𑀦 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀇𑀤𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬; 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀮𑀸𑀪𑀁…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑁇

67\. 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  “𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀮𑀸𑀪𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀮𑀸𑀪𑀁…𑀧𑁂…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀬𑀲𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀁…  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇

68\. 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂 𑀉𑀢𑁆𑀢𑀭, 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀘𑀢𑀲𑁆𑀲𑁄 𑀧𑀭𑀺𑀲𑀸—  𑀪𑀺𑀓𑁆𑀔𑀽, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄, 𑀉𑀧𑀸𑀲𑀓𑀸, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄𑁇 𑀦𑀸𑀬𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀓𑀺𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑀺 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁄𑁇 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁𑁇 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀸𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁𑁇 𑀥𑀸𑀭𑁂𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀢𑁆𑀢𑀭𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁𑁇 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁄 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁄 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀓𑁄”𑀢𑀺𑁇

---

69\. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
