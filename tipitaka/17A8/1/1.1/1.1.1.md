# 𑁧𑁇𑁧𑁇𑁧 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀫𑁂𑀢𑁆𑀢𑀸𑀲𑀼𑀢𑁆𑀢

(1.)

2. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀪𑀤𑀦𑁆𑀢𑁂”𑀢𑀺 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

3. “𑀫𑁂𑀢𑁆𑀢𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀸𑀬 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀬 𑀬𑀸𑀦𑀻𑀓𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀬 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀘𑀺𑀢𑀸𑀬 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑀬 𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀲𑀼𑀔𑀁 𑀲𑀼𑀧𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦 𑀧𑀸𑀧𑀓𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀦𑀸𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀺 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀦𑁆𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀽𑀧𑀕𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑁂𑀢𑁆𑀢𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀸𑀬 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀬 𑀬𑀸𑀦𑀻𑀓𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀬 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀘𑀺𑀢𑀸𑀬 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑀬 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀢𑀺𑁇

4. _𑀬𑁄 𑀘 𑀫𑁂𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑀬𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀢𑁄;_  
_𑀢𑀦𑀽 𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺,_  
_𑀧𑀲𑁆𑀲𑀢𑁄 𑀉𑀧𑀥𑀺𑀓𑁆𑀔𑀬𑀁𑁇_  


5. _𑀏𑀓𑀫𑁆𑀧𑀺 𑀘𑁂 𑀧𑀸𑀡𑀫𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑀢𑀺 𑀓𑀼𑀲𑀮𑀻 𑀢𑁂𑀦 𑀳𑁄𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑁂 𑀘 𑀧𑀸𑀡𑁂 𑀫𑀦𑀲𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀻,_  
_𑀧𑀳𑀽𑀢𑀫𑀭𑀺𑀬𑁄 𑀧𑀓𑀭𑁄𑀢𑀺 𑀧𑀼𑀜𑁆𑀜𑀁𑁇_  


6. _𑀬𑁂 𑀲𑀢𑁆𑀢𑀲𑀡𑁆𑀟𑀁 𑀧𑀣𑀯𑀺𑀁 𑀯𑀺𑀚𑁂𑀢𑁆𑀯𑀸,_  
_𑀭𑀸𑀚𑀺𑀲𑀬𑁄 𑀬𑀚𑀫𑀸𑀦𑀸 𑀅𑀦𑀼𑀧𑀭𑀺𑀬𑀕𑀸;_  
_𑀅𑀲𑁆𑀲𑀫𑁂𑀥𑀁 𑀧𑀼𑀭𑀺𑀲𑀫𑁂𑀥𑀁,_  
_𑀲𑀫𑁆𑀫𑀸𑀧𑀸𑀲𑀁 𑀯𑀸𑀚𑀧𑁂𑀬𑁆𑀬𑀁 𑀦𑀺𑀭𑀕𑁆𑀕𑀍𑀅𑀁𑁇_  


7. _𑀫𑁂𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑀲𑁆𑀲,_  
_𑀓𑀮𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦𑀸𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁;_  
_𑀘𑀦𑁆𑀤𑀧𑁆𑀧𑀪𑀸 𑀢𑀸𑀭𑀕𑀡𑀸𑀯 𑀲𑀩𑁆𑀩𑁂,_  
_𑀬𑀣𑀸 𑀦 𑀅𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀓𑀮𑀫𑁆𑀧𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀁𑁇_  


8. _𑀬𑁄 𑀦 𑀳𑀦𑁆𑀢𑀺 𑀦 𑀖𑀸𑀢𑁂𑀢𑀺,_  
_𑀦 𑀚𑀺𑀦𑀸𑀢𑀺 𑀦 𑀚𑀸𑀧𑀬𑁂;_  
_𑀫𑁂𑀢𑁆𑀢𑀁𑀲𑁄 𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀁,_  
_𑀯𑁂𑀭𑀁 𑀢𑀲𑁆𑀲 𑀦 𑀓𑁂𑀦𑀘𑀻”𑀢𑀺𑁇_  


---

9. 𑀧𑀞𑀫𑀁𑁇
