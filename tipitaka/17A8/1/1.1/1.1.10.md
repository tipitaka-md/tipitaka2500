# 𑁧𑁇𑁧𑁇𑁧𑁦 𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀫𑁂𑀢𑁆𑀢𑀸𑀯𑀕𑁆𑀕

### 𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑀲𑀼𑀢𑁆𑀢

(10.)

77\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀕𑁆𑀕𑀭𑀸𑀬 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑀸 𑀢𑀻𑀭𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑀺𑀬𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑁂𑀦𑀸𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀢𑀺, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀓𑀣𑀁 𑀅𑀧𑀦𑀸𑀫𑁂𑀢𑀺, 𑀓𑁄𑀧𑀜𑁆𑀘 𑀤𑁄𑀲𑀜𑁆𑀘 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀜𑁆𑀘 𑀧𑀸𑀢𑀼𑀓𑀭𑁄𑀢𑀺𑁇

78\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦𑀺𑀤𑁆𑀥𑀫𑀣𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀁; 𑀦𑀺𑀤𑁆𑀥𑀫𑀣𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀁𑁇 𑀅𑀧𑀦𑁂𑀬𑁆𑀬𑁂𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇 𑀓𑀺𑀁 𑀯𑁄 𑀢𑁂𑀦 𑀧𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑁂𑀦𑁇 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀲𑀁𑀬𑁂𑀯 𑀳𑁄𑀢𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀆𑀮𑁄𑀓𑀺𑀢𑀁 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀤𑁆𑀤𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁—  𑀬𑀸𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀲𑀫𑀡𑀤𑀽𑀲𑀻𑀯𑀸𑀬𑀁 𑀲𑀫𑀡𑀧𑀮𑀸𑀧𑁄 𑀲𑀫𑀡𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑁄’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀦𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀫𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀤𑁆𑀤𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀲𑁂𑀲𑀻𑀢𑀺𑁇

79\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂 𑀬𑀯𑀓𑀭𑀡𑁂 𑀬𑀯𑀤𑀽𑀲𑀻 𑀚𑀸𑀬𑁂𑀣 𑀬𑀯𑀧𑀮𑀸𑀧𑁄 𑀬𑀯𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀲𑀁𑀬𑁂𑀯 𑀫𑀽𑀮𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀤𑁆𑀤𑀓𑀸𑀦𑀁 𑀬𑀯𑀸𑀦𑀁; 𑀢𑀸𑀤𑀺𑀲𑀁𑀬𑁂𑀯 𑀦𑀸𑀍𑀅𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀤𑁆𑀤𑀓𑀸𑀦𑀁 𑀬𑀯𑀸𑀦𑀁; 𑀢𑀸𑀤𑀺𑀲𑀁𑀬𑁂𑀯 𑀧𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀤𑁆𑀤𑀓𑀸𑀦𑀁 𑀬𑀯𑀸𑀦𑀁—  𑀬𑀸𑀯𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀦 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀲𑀻𑀲𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀬𑀯𑀤𑀽𑀲𑀻𑀯𑀸𑀬𑀁 𑀬𑀯𑀧𑀮𑀸𑀧𑁄 𑀬𑀯𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑁄’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀽𑀮𑀁 𑀉𑀧𑁆𑀧𑀸𑀝𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀬𑀯𑀓𑀭𑀡𑀲𑁆𑀲 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀫𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀤𑁆𑀤𑀓𑁂 𑀬𑀯𑁂 𑀤𑀽𑀲𑁂𑀲𑀻𑀢𑀺𑁇

80\. 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀲𑀁𑀬𑁂𑀯 𑀳𑁄𑀢𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀆𑀮𑁄𑀓𑀺𑀢𑀁 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀤𑁆𑀤𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁—  𑀬𑀸𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀲𑀫𑀡𑀤𑀽𑀲𑀻𑀯𑀸𑀬𑀁 𑀲𑀫𑀡𑀧𑀮𑀸𑀧𑁄 𑀲𑀫𑀡𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑁄’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀦𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀫𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀤𑁆𑀤𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀲𑁂𑀲𑀻𑀢𑀺𑁇

81\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀢𑁄 𑀥𑀜𑁆𑀜𑀭𑀸𑀲𑀺𑀲𑁆𑀲 𑀨𑀼𑀡𑀫𑀸𑀦𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀥𑀜𑁆𑀜𑀸𑀦𑀺 𑀤𑀍𑀳𑀸𑀦𑀺 𑀲𑀸𑀭𑀯𑀦𑁆𑀢𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀧𑀼𑀜𑁆𑀚𑀁 𑀳𑁄𑀢𑀺, 𑀬𑀸𑀦𑀺 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀥𑀜𑁆𑀜𑀸𑀦𑀺 𑀤𑀼𑀩𑁆𑀩𑀮𑀸𑀦𑀺 𑀧𑀮𑀸𑀧𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀯𑀸𑀢𑁄 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀯𑀳𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀲𑀸𑀫𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀺𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀧𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀫𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀤𑁆𑀤𑀓𑁂 𑀥𑀜𑁆𑀜𑁂 𑀤𑀽𑀲𑁂𑀲𑀻𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀲𑀁𑀬𑁂𑀯 𑀳𑁄𑀢𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀆𑀮𑁄𑀓𑀺𑀢𑀁 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀤𑁆𑀤𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁—  𑀬𑀸𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀲𑀫𑀡𑀤𑀽𑀲𑀻𑀯𑀸𑀬𑀁 𑀲𑀫𑀡𑀧𑀮𑀸𑀧𑁄 𑀲𑀫𑀡𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑁄’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀦𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀫𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀤𑁆𑀤𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀲𑁂𑀲𑀻𑀢𑀺𑁇

82\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑁄 𑀉𑀤𑀧𑀸𑀦𑀧𑀦𑀸𑀍𑀇𑀬𑀢𑁆𑀣𑀺𑀓𑁄 𑀢𑀺𑀡𑁆𑀳𑀁 𑀓𑀼𑀞𑀸𑀭𑀺𑀁 𑀆𑀤𑀸𑀬 𑀯𑀦𑀁 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑁄 𑀬𑀁 𑀬𑀤𑁂𑀯 𑀭𑀼𑀓𑁆𑀔𑀁 𑀓𑀼𑀞𑀸𑀭𑀺𑀧𑀸𑀲𑁂𑀦 𑀆𑀓𑁄𑀝𑁂𑀬𑁆𑀬 𑀢𑀢𑁆𑀣 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀭𑀼𑀓𑁆𑀔𑀸𑀦𑀺 𑀤𑀍𑀳𑀸𑀦𑀺 𑀲𑀸𑀭𑀯𑀦𑁆𑀢𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀓𑀼𑀞𑀸𑀭𑀺𑀧𑀸𑀲𑁂𑀦 𑀆𑀓𑁄𑀝𑀺𑀢𑀸𑀦𑀺 𑀓𑀓𑁆𑀔𑀍𑀅𑀁 𑀧𑀝𑀺𑀦𑀤𑀦𑁆𑀢𑀺; 𑀬𑀸𑀦𑀺 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀭𑀼𑀓𑁆𑀔𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑁄𑀧𑀽𑀢𑀻𑀦𑀺 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸𑀦𑀺 𑀓𑀲𑀫𑁆𑀩𑀼𑀚𑀸𑀢𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀓𑀼𑀞𑀸𑀭𑀺𑀧𑀸𑀲𑁂𑀦 𑀆𑀓𑁄𑀝𑀺𑀢𑀸𑀦𑀺 𑀤𑀤𑁆𑀤𑀭𑀁 𑀧𑀝𑀺𑀦𑀤𑀦𑁆𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀫𑀽𑀮𑁂 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀫𑀽𑀮𑁂 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑁂 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀅𑀕𑁆𑀕𑁂 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁄 𑀲𑀼𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀅𑀦𑁆𑀢𑁄 𑀲𑀼𑀯𑀺𑀲𑁄𑀥𑀺𑀢𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑁆𑀯𑀸 𑀉𑀤𑀧𑀸𑀦𑀧𑀦𑀸𑀍𑀇𑀁 𑀬𑁄𑀚𑁂𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀢𑀸𑀤𑀺𑀲𑀁𑀬𑁂𑀯 𑀳𑁄𑀢𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀆𑀮𑁄𑀓𑀺𑀢𑀁 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀁 𑀪𑀤𑁆𑀤𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁—  𑀬𑀸𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁆𑀯𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑀫𑁂𑀦𑀁 𑀏𑀯𑀁 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀲𑀫𑀡𑀤𑀽𑀲𑀻𑀯𑀸𑀬𑀁 𑀲𑀫𑀡𑀧𑀮𑀸𑀧𑁄 𑀲𑀫𑀡𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑁄’𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀦𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀫𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀤𑁆𑀤𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀽𑀲𑁂𑀲𑀻𑀢𑀺𑁇

83\. _𑀲𑀁𑀯𑀸𑀲𑀸𑀬𑀁 𑀯𑀺𑀚𑀸𑀦𑀸𑀣,_  
_𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀓𑁄𑀥𑀦𑁄 𑀇𑀢𑀺;_  
_𑀫𑀓𑁆𑀔𑀻 𑀣𑀫𑁆𑀪𑀻 𑀧𑀍𑀆𑀲𑀻 𑀘,_  
_𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀫𑀘𑁆𑀙𑀭𑀻 𑀲𑀞𑁄𑁇_  


84\. _𑀲𑀦𑁆𑀢𑀯𑀸𑀘𑁄 𑀚𑀦𑀯𑀢𑀺,_  
_𑀲𑀫𑀡𑁄 𑀯𑀺𑀬 𑀪𑀸𑀲𑀢𑀺;_  
_𑀭𑀳𑁄 𑀓𑀭𑁄𑀢𑀺 𑀓𑀭𑀡𑀁,_  
_𑀧𑀸𑀧𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀦𑀸𑀤𑀭𑁄𑁇_  


85\. _𑀲𑀁𑀲𑀧𑁆𑀧𑀻 𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻,_  
_𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀫𑀕𑁆𑀕𑀸 𑀳𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀚𑁆𑀚𑀬𑀸𑀣 𑀦𑀁𑁇_  


86\. _𑀓𑀸𑀭𑀡𑁆𑀟𑀯𑀁 𑀦𑀺𑀤𑁆𑀥𑀫𑀣,_  
_𑀓𑀲𑀫𑁆𑀩𑀼𑀁 𑀅𑀧𑀓𑀲𑁆𑀲𑀣;_  
_𑀢𑀢𑁄 𑀧𑀮𑀸𑀧𑁂 𑀯𑀸𑀳𑁂𑀣,_  
_𑀅𑀲𑁆𑀲𑀫𑀡𑁂 𑀲𑀫𑀡𑀫𑀸𑀦𑀺𑀦𑁂𑁇_  


87\. _𑀦𑀺𑀤𑁆𑀥𑀫𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁂,_  
_𑀧𑀸𑀧𑀆𑀘𑀸𑀭𑀕𑁄𑀘𑀭𑁂;_  
_𑀲𑀼𑀤𑁆𑀥𑀸𑀲𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀲𑀁𑀯𑀸𑀲𑀁,_  
_𑀓𑀧𑁆𑀧𑀬𑀯𑁆𑀳𑁄 𑀧𑀢𑀺𑀲𑁆𑀲𑀢𑀸;_  
_𑀢𑀢𑁄 𑀲𑀫𑀕𑁆𑀕𑀸 𑀦𑀺𑀧𑀓𑀸,_  
_𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀣𑀸”𑀢𑀺𑁇_
