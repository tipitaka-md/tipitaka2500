# 𑁧𑁇𑁪𑁇𑁧 𑀧𑀞𑀫𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀧𑀞𑀫𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(31.)

322\. “𑀅𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞? 𑀆𑀲𑀚𑁆𑀚 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀪𑀬𑀸 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, ‘𑀅𑀤𑀸𑀲𑀺 𑀫𑁂’𑀢𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, ‘𑀤𑀲𑁆𑀲𑀢𑀺 𑀫𑁂’𑀢𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, ‘𑀲𑀸𑀳𑀼 𑀤𑀸𑀦𑀦𑁆’𑀢𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, ‘𑀅𑀳𑀁 𑀧𑀘𑀸𑀫𑀺, 𑀇𑀫𑁂 𑀦 𑀧𑀘𑀦𑁆𑀢𑀺; 𑀦𑀸𑀭𑀳𑀸𑀫𑀺 𑀧𑀘𑀦𑁆𑀢𑁄 𑀅𑀧𑀘𑀦𑁆𑀢𑀸𑀦𑀁 𑀤𑀸𑀦𑀁 𑀅𑀤𑀸𑀢𑀼𑀦𑁆’𑀢𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, ‘𑀇𑀫𑀁 𑀫𑁂 𑀤𑀸𑀦𑀁 𑀤𑀤𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀻’𑀢𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀘𑀺𑀢𑁆𑀢𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀢𑁆𑀣𑀁 𑀤𑀸𑀦𑀁 𑀤𑁂𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀤𑀸𑀦𑀸𑀦𑀻”𑀢𑀺𑁇
