# 𑁧𑁇𑁪𑁇𑁮 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀲𑀼𑀢𑁆𑀢

(38.)

360\. “𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀫𑀸𑀦𑁄 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺—  𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀧𑁂𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀭𑀜𑁆𑀜𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺𑁇

361\. 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀲𑀩𑁆𑀩𑀲𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀓𑀼𑀮𑁂 𑀚𑀸𑀬𑀫𑀸𑀦𑁄 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺—  𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀧𑁂𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀭𑀜𑁆𑀜𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

362\. _𑀩𑀳𑀽𑀦𑀁 𑀯𑀢 𑀅𑀢𑁆𑀣𑀸𑀬,_  
_𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄 𑀖𑀭𑀫𑀸𑀯𑀲𑀁;_  
_𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀁 𑀧𑀼𑀩𑁆𑀩𑁂,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑁄𑁇_  


363\. _𑀧𑀽𑀚𑁂𑀢𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦,_  
_𑀧𑀼𑀩𑁆𑀩𑁂𑀓𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀁;_  
_𑀅𑀦𑀸𑀕𑀸𑀭𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂,_  
_𑀅𑀧𑀘𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀬𑁄𑁇_  


364\. _𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀲𑀤𑁆𑀥𑁄 𑀧𑀽𑀚𑁂𑀢𑀺,_  
_𑀜𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀧𑁂𑀲𑀮𑁄;_  
_𑀭𑀜𑁆𑀜𑁄 𑀳𑀺𑀢𑁄 𑀤𑁂𑀯𑀳𑀺𑀢𑁄,_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀲𑀔𑀺𑀦𑀁 𑀳𑀺𑀢𑁄𑁇_  


365\. _𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑁄 𑀳𑀺𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄;_  
_𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀁,_  
_𑀲 𑀮𑁄𑀓𑀁 𑀪𑀚𑀢𑁂 𑀲𑀺𑀯𑀦𑁆”𑀢𑀺𑁇_  


---

366\. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
