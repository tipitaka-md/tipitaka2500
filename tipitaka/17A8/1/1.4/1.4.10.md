# 𑁧𑁇𑁪𑁇𑁧𑁦 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀯𑀺𑀧𑀸𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀯𑀺𑀧𑀸𑀓𑀲𑀼𑀢𑁆𑀢

(40.)

374\. “𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑁂𑀯𑀺𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀅𑀧𑁆𑀧𑀸𑀬𑀼𑀓𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁧)

375\. 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑁂𑀯𑀺𑀢𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁨)

376\. 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑁂𑀯𑀺𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀲𑀧𑀢𑁆𑀢𑀯𑁂𑀭𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁩)

377\. 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑁂𑀯𑀺𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀅𑀪𑀽𑀢𑀩𑁆𑀪𑀓𑁆𑀔𑀸𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁪)

378\. 𑀧𑀺𑀲𑀼𑀡𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀸𑀘𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀧𑀺𑀲𑀼𑀡𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀪𑁂𑀤𑀦𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁫)

379\. 𑀨𑀭𑀼𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀸𑀘𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀨𑀭𑀼𑀲𑀸𑀬 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀅𑀫𑀦𑀸𑀧𑀲𑀤𑁆𑀤𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁬)

380\. 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑁂𑀯𑀺𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑁄 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀲𑀫𑁆𑀨𑀧𑁆𑀧𑀮𑀸𑀧𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀅𑀦𑀸𑀤𑁂𑀬𑁆𑀬𑀯𑀸𑀘𑀸𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 (𑁭)

381\. 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀧𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀲𑁂𑀯𑀺𑀢𑀁 𑀪𑀸𑀯𑀺𑀢𑀁 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀁 𑀦𑀺𑀭𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀧𑁂𑀢𑁆𑀢𑀺𑀯𑀺𑀲𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁𑁇 𑀬𑁄 𑀲𑀩𑁆𑀩𑀮𑀳𑀼𑀲𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀧𑀸𑀦𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑀲𑁆𑀲 𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇 (𑁮)

384\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



385\. _𑀤𑁆𑀯𑁂 𑀤𑀸𑀦𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘,_  
_𑀔𑁂𑀢𑁆𑀢𑀁 𑀤𑀸𑀦𑀽𑀧𑀧𑀢𑁆𑀢𑀺𑀬𑁄;_
