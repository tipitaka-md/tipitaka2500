# 𑁧𑁇𑁪𑁇𑁭 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(37.)

355\. “𑀅𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞? 𑀲𑀼𑀘𑀺𑀁 𑀤𑁂𑀢𑀺, 𑀧𑀡𑀻𑀢𑀁 𑀤𑁂𑀢𑀺, 𑀓𑀸𑀮𑁂𑀦 𑀤𑁂𑀢𑀺, 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀤𑁂𑀢𑀺, 𑀯𑀺𑀘𑁂𑀬𑁆𑀬 𑀤𑁂𑀢𑀺, 𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀤𑁂𑀢𑀺, 𑀤𑀤𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀸𑀤𑁂𑀢𑀺, 𑀤𑀢𑁆𑀯𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀤𑀸𑀦𑀸𑀦𑀻𑀢𑀺𑁇

356\. _𑀲𑀼𑀘𑀺𑀁 𑀧𑀡𑀻𑀢𑀁 𑀓𑀸𑀮𑁂𑀦,_  
_𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀧𑀸𑀦𑀪𑁄𑀚𑀦𑀁;_  
_𑀅𑀪𑀺𑀡𑁆𑀳𑀁 𑀤𑀤𑀸𑀢𑀺 𑀤𑀸𑀦𑀁,_  
_𑀲𑀼𑀔𑁂𑀢𑁆𑀢𑁂𑀲𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀲𑀼𑁇_  


357\. _𑀦𑁂𑀯 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀺𑀲𑁆𑀲,_  
_𑀘𑀚𑀺𑀢𑁆𑀯𑀸 𑀆𑀫𑀺𑀲𑀁 𑀩𑀳𑀼𑀁;_  
_𑀏𑀯𑀁 𑀤𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀤𑀸𑀦𑀸𑀦𑀺,_  
_𑀯𑀡𑁆𑀡𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀲𑁆𑀲𑀺𑀦𑁄𑁇_  


358\. _𑀏𑀯𑀁 𑀬𑀚𑀺𑀢𑁆𑀯𑀸 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀲𑀤𑁆𑀥𑁄 𑀫𑀼𑀢𑁆𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸;_  
_𑀅𑀩𑁆𑀬𑀸𑀩𑀚𑁆𑀛𑀁 𑀲𑀼𑀔𑀁 𑀮𑁄𑀓𑀁,_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇_
