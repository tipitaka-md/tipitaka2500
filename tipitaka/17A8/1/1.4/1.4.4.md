# 𑁧𑁇𑁪𑁇𑁪 𑀔𑁂𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀞𑀫𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀔𑁂𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(34.)

328. “𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑁂𑀢𑁆𑀢𑁂 𑀩𑀻𑀚𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀦 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀦 𑀫𑀳𑀲𑁆𑀲𑀸𑀤𑀁 𑀦 𑀨𑀸𑀢𑀺𑀲𑁂𑀬𑁆𑀬𑀁𑁇 𑀓𑀣𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑁂𑀢𑁆𑀢𑀁 𑀉𑀦𑁆𑀦𑀸𑀫𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑀺 𑀘 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀺𑀓𑀜𑁆𑀘 𑀳𑁄𑀢𑀺, 𑀊𑀲𑀭𑀜𑁆𑀘 𑀳𑁄𑀢𑀺, 𑀦 𑀘 𑀕𑀫𑁆𑀪𑀻𑀭𑀲𑀺𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀦 𑀆𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀦 𑀅𑀧𑀸𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀦 𑀫𑀸𑀢𑀺𑀓𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀦 𑀫𑀭𑀺𑀬𑀸𑀤𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑁂𑀢𑁆𑀢𑁂 𑀩𑀻𑀚𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀦 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀦 𑀫𑀳𑀲𑁆𑀲𑀸𑀤𑀁 𑀦 𑀨𑀸𑀢𑀺𑀲𑁂𑀬𑁆𑀬𑀁𑁇

329. 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀲𑀼 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀤𑀸𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀦 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀁 𑀦 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑀁 𑀦 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑀁𑁇 𑀓𑀣𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀲𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀘𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀆𑀚𑀻𑀯𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀯𑀸𑀬𑀸𑀫𑀸, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀢𑀺𑀦𑁄, 𑀫𑀺𑀘𑁆𑀙𑀸𑀲𑀫𑀸𑀥𑀺𑀦𑁄𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀤𑀸𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀁 𑀦 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀦 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀁 𑀦 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑀁 𑀦 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑀁𑁇

330. 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑁂𑀢𑁆𑀢𑁂 𑀩𑀻𑀚𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀲𑁆𑀲𑀸𑀤𑀁 𑀨𑀸𑀢𑀺𑀲𑁂𑀬𑁆𑀬𑀁𑁇 𑀓𑀣𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀦𑁆𑀦𑀸𑀫𑀸𑀦𑀺𑀦𑁆𑀦𑀸𑀫𑀺 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀸𑀲𑀸𑀡𑀲𑀓𑁆𑀔𑀭𑀺𑀓𑀜𑁆𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀦𑀽𑀲𑀭𑀜𑁆𑀘 𑀳𑁄𑀢𑀺, 𑀕𑀫𑁆𑀪𑀻𑀭𑀲𑀺𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀆𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀸𑀬𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀸𑀢𑀺𑀓𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀭𑀺𑀬𑀸𑀤𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑁂𑀢𑁆𑀢𑁂 𑀩𑀻𑀚𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀲𑁆𑀲𑀸𑀤𑀁 𑀨𑀸𑀢𑀺𑀲𑁂𑀬𑁆𑀬𑀁𑁇

331. 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀲𑀼 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀤𑀸𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀁 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑀁 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑀁𑁇 𑀓𑀣𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀲𑀼? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀲𑀫𑁆𑀫𑀸𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀸, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀘𑀸, 𑀲𑀫𑁆𑀫𑀸𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸, 𑀲𑀫𑁆𑀫𑀸𑀆𑀚𑀻𑀯𑀸, 𑀲𑀫𑁆𑀫𑀸𑀯𑀸𑀬𑀸𑀫𑀸, 𑀲𑀫𑁆𑀫𑀸𑀲𑀢𑀺𑀦𑁄, 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑀸𑀥𑀺𑀦𑁄𑁇 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀲𑀼, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀲𑀼 𑀤𑀸𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀁 𑀳𑁄𑀢𑀺 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀁 𑀫𑀳𑀸𑀚𑀼𑀢𑀺𑀓𑀁 𑀫𑀳𑀸𑀯𑀺𑀧𑁆𑀨𑀸𑀭𑀦𑁆𑀢𑀺𑁇

332. _𑀬𑀣𑀸𑀧𑀺 𑀔𑁂𑀢𑁆𑀢𑁂 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀧𑀯𑀼𑀢𑁆𑀢𑀸 𑀩𑀻𑀚𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀤𑁂𑀯𑁂 𑀲𑀫𑁆𑀧𑀸𑀤𑀬𑀦𑁆𑀢𑀫𑁆𑀳𑀺,_  
_𑀳𑁄𑀢𑀺 𑀥𑀜𑁆𑀜𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


333. _𑀅𑀦𑀻𑀢𑀺𑀲𑀫𑁆𑀧𑀤𑀸 𑀳𑁄𑀢𑀺,_  
_𑀯𑀺𑀭𑀽𑀍𑀳𑀻 𑀪𑀯𑀢𑀺 𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀯𑁂𑀧𑀼𑀮𑁆𑀮𑀲𑀫𑁆𑀧𑀤𑀸 𑀳𑁄𑀢𑀺,_  
_𑀨𑀮𑀁 𑀯𑁂 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


334. _𑀏𑀯𑀁 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀻𑀮𑁂𑀲𑀼,_  
_𑀤𑀺𑀦𑁆𑀦𑀸 𑀪𑁄𑀚𑀦𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀁 𑀉𑀧𑀦𑁂𑀢𑀺,_  
_𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑀺𑀲𑁆𑀲 𑀢𑀁 𑀓𑀢𑀁𑁇_  


335. _𑀢𑀲𑁆𑀫𑀸 𑀲𑀫𑁆𑀧𑀤𑀫𑀸𑀓𑀗𑁆𑀔𑀻,_  
_𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀢𑁆𑀣𑀽𑀥 𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑁂 𑀲𑁂𑀯𑁂𑀣,_  
_𑀏𑀯𑀁 𑀇𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


336. _𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀮𑀤𑁆𑀥𑀸 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀤𑀁;_  
_𑀓𑀭𑁄𑀢𑀺 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀤𑀁,_  
_𑀮𑀪𑀢𑀺 𑀘𑀢𑁆𑀣𑀲𑀫𑁆𑀧𑀤𑀁𑁇_  


337. _𑀮𑁄𑀓𑀁 𑀜𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀪𑀽𑀢𑀁,_  
_𑀧𑀧𑁆𑀧𑀼𑀬𑁆𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀤𑀁;_  
_𑀫𑀕𑁆𑀕𑀲𑀫𑁆𑀧𑀤𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀬𑀸𑀢𑀺 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀫𑀸𑀦𑀲𑁄𑁇_  


338. _𑀑𑀥𑀼𑀦𑀺𑀢𑁆𑀯𑀸 𑀫𑀮𑀁 𑀲𑀩𑁆𑀩𑀁,_  
_𑀧𑀢𑁆𑀯𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀲𑀫𑁆𑀧𑀤𑀁;_  
_𑀫𑀼𑀘𑁆𑀘𑀢𑀺 𑀲𑀩𑁆𑀩𑀤𑀼𑀓𑁆𑀔𑁂𑀳𑀺,_  
_𑀲𑀸 𑀳𑁄𑀢𑀺 𑀲𑀩𑁆𑀩𑀲𑀫𑁆𑀧𑀤𑀸”𑀢𑀺𑁇_  


---
