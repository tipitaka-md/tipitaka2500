# 𑁨𑁇𑁨𑁇𑁨 𑀅𑀮𑀁𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀅𑀮𑀁𑀲𑀼𑀢𑁆𑀢

(62.)

631\. “𑀙𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀙𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄, 𑀧𑁄𑀭𑀺𑀬𑀸 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀍𑀆𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 (𑁧)

632\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 (𑁨)

633\. 𑀘𑀢𑀽𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄, 𑀧𑁄𑀭𑀺𑀬𑀸 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀍𑀆𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀦𑁄 𑀘 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 (𑁩)

634\. 𑀘𑀢𑀽𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀥𑀸𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀦 𑀘 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 (𑁪)

635\. 𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄, 𑀧𑁄𑀭𑀺𑀬𑀸 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀍𑀆𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀦𑁄 𑀘 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 (𑁫)

636\. 𑀢𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀥𑀸𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 (𑁬)

637\. 𑀤𑁆𑀯𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀦𑁄 𑀘 𑀲𑀼𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀦𑁄 𑀘 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 (𑁭)

638\. 𑀤𑁆𑀯𑀻𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀦𑁄 𑀘 𑀲𑀼𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀥𑀸𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄, 𑀧𑁄𑀭𑀺𑀬𑀸 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀍𑀆𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄”𑀢𑀺𑁇 (𑁮)
