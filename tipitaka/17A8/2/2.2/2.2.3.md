# 𑁨𑁇𑁨𑁇𑁩 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(63.)

640\. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺…𑀧𑁂…  𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼, 𑀬𑀫𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀫𑁂𑀯𑀁 𑀧𑀦𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀫𑀫𑀜𑁆𑀜𑁂𑀯 𑀅𑀚𑁆𑀛𑁂𑀲𑀦𑁆𑀢𑀺𑁇 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀪𑀸𑀲𑀺𑀢𑁂 𑀫𑀫𑀜𑁆𑀜𑁂𑀯 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 “𑀤𑁂𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀁, 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀥𑀫𑁆𑀫𑀁𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀤𑀸𑀬𑀸𑀤𑁄 𑀅𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀫𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀼𑀲𑀡𑁆𑀞𑀺𑀢𑀁, 𑀦 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀞𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

641\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀼𑀲𑀡𑁆𑀞𑀺𑀢𑀁, 𑀦 𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀧𑀸𑀧𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀬 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀢𑀢𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀫𑁂𑀢𑁆𑀢𑀸 𑀫𑁂 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

642\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀫𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑁆𑀧𑀺 𑀲𑀯𑀺𑀘𑀸𑀭𑀁 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑁆𑀧𑀺 𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀁 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑁆𑀧𑀺 𑀅𑀯𑀺𑀘𑀸𑀭𑀁 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀦𑀺𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀸𑀢𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇

643\. 𑀬𑀢𑁄 𑀔𑁄, 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀓𑀭𑀼𑀡𑀸 𑀫𑁂 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺…  𑀫𑀼𑀤𑀺𑀢𑀸 𑀫𑁂 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺…  𑀉𑀧𑁂𑀓𑁆𑀔𑀸 𑀫𑁂 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺 𑀪𑀸𑀯𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

644\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀫𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀦𑀺𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀸𑀢𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇

645\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀓𑀸𑀬𑁂 𑀓𑀸𑀬𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

646\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀫𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀦𑀺𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀸𑀢𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇

647\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀯𑁂𑀤𑀦𑀸𑀲𑀼 …  𑀘𑀺𑀢𑁆𑀢𑁂…  𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀧𑀲𑁆𑀲𑀻 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀢𑀸𑀧𑀻 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀲𑀢𑀺𑀫𑀸, 𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀮𑁄𑀓𑁂 𑀅𑀪𑀺𑀚𑁆𑀛𑀸𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇

648\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀇𑀫𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀯𑀺𑀢𑀓𑁆𑀓𑀲𑀯𑀺𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀯𑀺𑀘𑀸𑀭𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀅𑀯𑀺𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀦𑀺𑀧𑁆𑀧𑀻𑀢𑀺𑀓𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀲𑀸𑀢𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀫𑁆𑀧𑀺 𑀪𑀸𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇

649\. 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀬𑀁 𑀲𑀫𑀸𑀥𑀺 𑀏𑀯𑀁 𑀪𑀸𑀯𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀪𑀸𑀯𑀺𑀢𑁄, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑁂𑀦 𑀬𑁂𑀦𑁂𑀯 𑀕𑀕𑁆𑀖𑀲𑀺 𑀨𑀸𑀲𑀼𑀁𑀬𑁂𑀯 𑀕𑀕𑁆𑀖𑀲𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀞𑀲𑁆𑀲𑀲𑀺 𑀨𑀸𑀲𑀼𑀁𑀬𑁂𑀯 𑀞𑀲𑁆𑀲𑀲𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀲𑀺 𑀨𑀸𑀲𑀼𑀁𑀬𑁂𑀯 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀲𑀺, 𑀬𑀢𑁆𑀣 𑀬𑀢𑁆𑀣 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀲𑀺 𑀨𑀸𑀲𑀼𑀁𑀬𑁂𑀯 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑁆𑀲𑀲𑀻”𑀢𑀺𑁇

650\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑀸 𑀇𑀫𑀺𑀦𑀸 𑀑𑀯𑀸𑀤𑁂𑀦 𑀑𑀯𑀤𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀘 𑀧𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀻𑀢𑀺𑁇

---

651\. 𑀢𑀢𑀺𑀬𑀁𑁇
