# 𑁨𑁇𑁨𑁇𑁪 𑀕𑀬𑀸𑀲𑀻𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀕𑀬𑀸𑀲𑀻𑀲𑀲𑀼𑀢𑁆𑀢

(64.)

652\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀕𑀬𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀬𑀸𑀲𑀻𑀲𑁂𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺…𑀧𑁂…  “𑀧𑀼𑀩𑁆𑀩𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑁄𑀥𑀸 𑀅𑀦𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄𑀯 𑀲𑀫𑀸𑀦𑁄 𑀑𑀪𑀸𑀲𑀜𑁆𑀜𑁂𑀯 𑀔𑁄 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑁄 𑀘 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀺𑁇 (𑁧)

653\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀘𑁂 𑀔𑁄 𑀅𑀳𑀁 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁; 𑀏𑀯𑀁 𑀫𑁂 𑀇𑀤𑀁 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑀭𑀁 𑀅𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

654\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺; 𑀦𑁄 𑀘 𑀔𑁄 𑀢𑀸𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀺𑁇 (𑁨)

655\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀘𑁂 𑀔𑁄 𑀅𑀳𑀁 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁, 𑀢𑀸𑀳𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀁 𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬𑀁 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀁; 𑀏𑀯𑀁 𑀫𑁂 𑀇𑀤𑀁 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑀭𑀁 𑀅𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

656\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀸𑀳𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀺; 𑀦𑁄 𑀘 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀸𑀢𑀺𑁇 (𑁩)

657\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀘𑁂 𑀔𑁄 𑀅𑀳𑀁 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁, 𑀢𑀸𑀳𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀁 𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬𑀁 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀁, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  “𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀸”𑀢𑀺; 𑀏𑀯𑀁 𑀫𑁂 𑀇𑀤𑀁 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑀭𑀁 𑀅𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

658\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀸𑀳𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀸’𑀢𑀺; 𑀦𑁄 𑀘 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀇𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀇𑀢𑁄 𑀘𑀼𑀢𑀸 𑀢𑀢𑁆𑀣 𑀉𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺…𑀧𑁂…  𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀇𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀇𑀢𑁄 𑀘𑀼𑀢𑀸 𑀢𑀢𑁆𑀣 𑀉𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺; 𑀦𑁄 𑀘 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀇𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑀺𑀬𑁄’𑀢𑀺…𑀧𑁂…  𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀇𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑀺𑀬𑁄’𑀢𑀺; 𑀦𑁄 𑀘 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀁𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀏𑀯𑀁𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀸’𑀢𑀺…𑀧𑁂…  𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀁𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀏𑀯𑀁𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀸’𑀢𑀺; 𑀦𑁄 𑀘 𑀔𑁄 𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀤𑀺 𑀯𑀸 𑀫𑁂 𑀇𑀫𑀸𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀼𑀢𑁆𑀣𑀧𑀼𑀩𑁆𑀩𑀁 𑀬𑀤𑀺 𑀯𑀸 𑀦 𑀲𑀦𑁆𑀦𑀺𑀯𑀼𑀢𑁆𑀣𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺𑁇 (𑁪--𑁭)

659\. 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀲𑀘𑁂 𑀔𑁄 𑀅𑀳𑀁 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀁, 𑀢𑀸𑀳𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀁 𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬𑀁 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀁, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  “𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀸”𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  “𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀇𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀇𑀢𑁄 𑀘𑀼𑀢𑀸 𑀢𑀢𑁆𑀣 𑀉𑀧𑀧𑀦𑁆𑀦𑀸”𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  “𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑀺𑀬𑁄”𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  “𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀁𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀏𑀯𑀁𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀸”𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁 𑀬𑀤𑀺 𑀯𑀸 𑀫𑁂 𑀇𑀫𑀸𑀳𑀺 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀼𑀢𑁆𑀣𑀧𑀼𑀩𑁆𑀩𑀁 𑀬𑀤𑀺 𑀯𑀸 𑀦 𑀲𑀦𑁆𑀦𑀺𑀯𑀼𑀢𑁆𑀣𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺; 𑀏𑀯𑀁 𑀫𑁂 𑀇𑀤𑀁 𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑀭𑀁 𑀅𑀲𑁆𑀲𑀸’𑀢𑀺𑁇

660\. 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀑𑀪𑀸𑀲𑀜𑁆𑀘𑁂𑀯 𑀲𑀜𑁆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀭𑀽𑀧𑀸𑀦𑀺 𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀸𑀳𑀺 𑀘 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀸𑀫𑀺 𑀲𑀮𑁆𑀮𑀧𑀸𑀫𑀺 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀸𑀫𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀅𑀫𑀼𑀓𑀫𑁆𑀳𑀸 𑀯𑀸 𑀤𑁂𑀯𑀦𑀺𑀓𑀸𑀬𑀸’𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀇𑀫𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀇𑀢𑁄 𑀘𑀼𑀢𑀸 𑀢𑀢𑁆𑀣 𑀉𑀧𑀧𑀦𑁆𑀦𑀸’𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀫𑀸𑀳𑀸𑀭𑀸 𑀏𑀯𑀁𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀺𑀦𑀺𑀬𑁄’𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺—  ‘𑀇𑀫𑀸 𑀤𑁂𑀯𑀢𑀸 𑀏𑀯𑀁𑀤𑀻𑀖𑀸𑀬𑀼𑀓𑀸 𑀏𑀯𑀁𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀸’𑀢𑀺, 𑀢𑀸 𑀘 𑀤𑁂𑀯𑀢𑀸 𑀚𑀸𑀦𑀸𑀫𑀺 𑀬𑀤𑀺 𑀯𑀸 𑀫𑁂 𑀤𑁂𑀯𑀢𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀼𑀢𑁆𑀣𑀧𑀼𑀩𑁆𑀩𑀁 𑀬𑀤𑀺 𑀯𑀸 𑀦 𑀲𑀦𑁆𑀦𑀺𑀯𑀼𑀢𑁆𑀣𑀧𑀼𑀩𑁆𑀩𑀦𑁆𑀢𑀺𑁇 (𑁮)

661\. 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀁 𑀅𑀥𑀺𑀤𑁂𑀯𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀦 𑀲𑀼𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀦𑁂𑀯 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄’𑀢𑀺 𑀧𑀘𑁆𑀘𑀜𑁆𑀜𑀸𑀲𑀺𑀁𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀁 𑀅𑀥𑀺𑀤𑁂𑀯𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀁 𑀲𑀼𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀅𑀣𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄’𑀢𑀺 𑀧𑀘𑁆𑀘𑀜𑁆𑀜𑀸𑀲𑀺𑀁; 𑀜𑀸𑀡𑀜𑁆𑀘 𑀧𑀦 𑀫𑁂 𑀤𑀲𑁆𑀲𑀦𑀁 𑀉𑀤𑀧𑀸𑀤𑀺; 𑀅𑀓𑀼𑀧𑁆𑀧𑀸 𑀫𑁂 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺; 𑀅𑀬𑀫𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀸𑀢𑀺 𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇
