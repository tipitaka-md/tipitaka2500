# 𑁨𑁇𑁨𑁇𑁧 𑀇𑀘𑁆𑀙𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀇𑀘𑁆𑀙𑀸𑀲𑀼𑀢𑁆𑀢

(61.)

621\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄 𑀖𑀝𑀢𑁄 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀲𑁄𑀘𑀢𑀺 𑀓𑀺𑀮𑀫𑀢𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀲𑁄𑀘𑀻 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁧)

622\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄 𑀖𑀝𑀢𑁄 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘 𑀫𑀤𑀻 𑀘 𑀧𑀫𑀸𑀤𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁨)

623\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀦 𑀖𑀝𑀢𑀺 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄 𑀅𑀖𑀝𑀢𑁄 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀲𑁄𑀘𑀢𑀺, 𑀓𑀺𑀮𑀫𑀢𑀺, 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀦 𑀖𑀝𑀢𑀺 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀲𑁄𑀘𑀻 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁩)

624\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄, 𑀅𑀖𑀝𑀢𑁄, 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀫𑀚𑁆𑀚𑀢𑀺, 𑀧𑀫𑀚𑁆𑀚𑀢𑀺, 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀦 𑀖𑀝𑀢𑀺 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘 𑀫𑀤𑀻 𑀘, 𑀧𑀫𑀸𑀤𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁪)

625\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄 𑀖𑀝𑀢𑁄 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀦 𑀲𑁄𑀘𑀢𑀺 𑀦 𑀓𑀺𑀮𑀫𑀢𑀺 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀦 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀦 𑀘 𑀲𑁄𑀘𑀻 𑀦 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁫)

626\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄 𑀖𑀝𑀢𑁄 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀧𑀫𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘, 𑀦 𑀘 𑀫𑀤𑀻 𑀦 𑀘 𑀧𑀫𑀸𑀤𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁬)

627\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄, 𑀅𑀖𑀝𑀢𑁄, 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀦 𑀲𑁄𑀘𑀢𑀺, 𑀦 𑀓𑀺𑀮𑀫𑀢𑀺, 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀦 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀦 𑀘 𑀲𑁄𑀘𑀻 𑀦 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁭)

628\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄 𑀅𑀖𑀝𑀢𑁄 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀧𑀫𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂—  ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘, 𑀦 𑀘 𑀫𑀤𑀻 𑀦 𑀘 𑀧𑀫𑀸𑀤𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁮)

629\. 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇

---
