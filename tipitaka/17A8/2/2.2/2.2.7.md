# 𑁨𑁇𑁨𑁇𑁭 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀲𑀼𑀢𑁆𑀢

(67.)

681\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀸𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀅𑀤𑀺𑀝𑁆𑀞𑁂 𑀤𑀺𑀝𑁆𑀞𑀯𑀸𑀤𑀺𑀢𑀸, 𑀅𑀲𑀼𑀢𑁂 𑀲𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀅𑀫𑀼𑀢𑁂 𑀫𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀤𑀺𑀝𑁆𑀞𑁂 𑀅𑀤𑀺𑀝𑁆𑀞𑀯𑀸𑀤𑀺𑀢𑀸, 𑀲𑀼𑀢𑁂 𑀅𑀲𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀫𑀼𑀢𑁂 𑀅𑀫𑀼𑀢𑀯𑀸𑀤𑀺𑀢𑀸, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀅𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀯𑀸𑀤𑀺𑀢𑀸𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀅𑀦𑀭𑀺𑀬𑀯𑁄𑀳𑀸𑀭𑀸”𑀢𑀺𑁇
