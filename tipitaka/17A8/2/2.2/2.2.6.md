# 𑁨𑁇𑁨𑁇𑁬 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀯𑀕𑁆𑀕

* 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀲𑀼𑀢𑁆𑀢

(66.)

672\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀭𑀽𑀧𑀻 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁧)

673\. 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻, 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁨)

674\. 𑀲𑀼𑀪𑀦𑁆𑀢𑁂𑀯 𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁩)

675\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑀸 𑀧𑀝𑀺𑀖𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀫𑀸 𑀦𑀸𑀦𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀸𑀦𑀁 𑀅𑀫𑀦𑀲𑀺𑀓𑀸𑀭𑀸 ‘𑀅𑀦𑀦𑁆𑀢𑁄 𑀆𑀓𑀸𑀲𑁄’𑀢𑀺 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁪)

676\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀸𑀲𑀸𑀦𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀅𑀦𑀦𑁆𑀢𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀦𑁆’𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀧𑀜𑁆𑀘𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁫)

677\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀜𑁆𑀘𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀻’𑀢𑀺 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀙𑀝𑁆𑀞𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁬)

678\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 (𑁭)

679\. 𑀲𑀩𑁆𑀩𑀲𑁄 𑀦𑁂𑀯𑀲𑀜𑁆𑀜𑀸𑀦𑀸𑀲𑀜𑁆𑀜𑀸𑀬𑀢𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁆𑀫 𑀲𑀜𑁆𑀜𑀸𑀯𑁂𑀤𑀬𑀺𑀢𑀦𑀺𑀭𑁄𑀥𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀬𑀁 𑀅𑀝𑁆𑀞𑀫𑁄 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸”𑀢𑀺𑁇 (𑁮)
