# 𑁨𑁇𑁨𑁇𑁧𑁦 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀯𑀕𑁆𑀕

### 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑀼𑀢𑁆𑀢

(70.)

688\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀲𑀻𑀤𑀦𑀁𑁇 𑀬𑁂𑀦 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀆𑀤𑀸𑀬 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑁇

689\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

690\. “𑀭𑀫𑀡𑀻𑀬𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀲𑀸𑀮𑀻, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀉𑀤𑁂𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀕𑁄𑀢𑀫𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁆𑀩𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀸𑀭𑀦𑁆𑀤𑀤𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑁇 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁; 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀬𑀸𑀘𑀺—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺, 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑁇

691\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀭𑀫𑀡𑀻𑀬𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀲𑀸𑀮𑀻, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀉𑀤𑁂𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀕𑁄𑀢𑀫𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁆𑀩𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀸𑀭𑀦𑁆𑀤𑀤𑀁 𑀘𑁂𑀢𑀺𑀬𑀁, 𑀭𑀫𑀡𑀻𑀬𑀁 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸…𑀧𑁂…  𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁; 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀬𑀸𑀘𑀺—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁, 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺, 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑁇

692\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

693\. “𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀸𑀯 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀲𑀸𑀯𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀢𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀸 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀢𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑁇

694\. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀸𑀯 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦 𑀲𑀸𑀯𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀫𑁂 𑀉𑀧𑀸𑀲𑀓𑀸 𑀦 𑀲𑀸𑀯𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀬𑀸𑀯 𑀫𑁂 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀦 𑀲𑀸𑀯𑀺𑀓𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀢𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀺𑀬𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀺𑀓𑀸 𑀯𑀺𑀬𑀢𑁆𑀢𑀸 𑀯𑀺𑀦𑀻𑀢𑀸 𑀯𑀺𑀲𑀸𑀭𑀤𑀸 𑀧𑀢𑁆𑀢𑀬𑁄𑀕𑀓𑁆𑀔𑁂𑀫𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀥𑀫𑁆𑀫𑀥𑀭𑀸 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀼𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀺𑀬𑁄, 𑀲𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀧𑁂𑀦𑁆𑀢𑀺 𑀧𑀝𑁆𑀞𑀧𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑀺 𑀯𑀺𑀪𑀚𑀦𑁆𑀢𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑀻𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀭𑀧𑁆𑀧𑀯𑀸𑀤𑀁 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀧𑁆𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺𑁇

695\. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀸𑀯 𑀫𑁂 𑀇𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀦 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁, 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁, 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀁𑁇

696\. 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄”𑀢𑀺𑁇 “𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑁆𑀯𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀳𑁄𑀳𑀺𑁇 𑀦𑀘𑀺𑀭𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

697\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀘𑀸𑀧𑀸𑀮𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀑𑀲𑁆𑀲𑀚𑀺𑁇 𑀑𑀲𑁆𑀲𑀝𑁆𑀞𑁂 𑀘 𑀪𑀕𑀯𑀢𑀸 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀫𑀳𑀸𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀅𑀳𑁄𑀲𑀺 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

698\. _“𑀢𑀼𑀮𑀫𑀢𑀼𑀮𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀁,_  
_𑀪𑀯𑀲𑀗𑁆𑀔𑀸𑀭𑀫𑀯𑀲𑁆𑀲𑀚𑀺 𑀫𑀼𑀦𑀺;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀭𑀢𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑁄,_  
_𑀅𑀪𑀺𑀦𑁆𑀤𑀺 𑀓𑀯𑀘𑀫𑀺𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇_  


699\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄; 𑀲𑀼𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑀸”𑀢𑀺?

700\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄; 𑀲𑀼𑀫𑀳𑀸 𑀯𑀢𑀸𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀲𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑀸”𑀢𑀺?

701\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀳𑁂𑀢𑀽, 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀳𑀸𑀧𑀣𑀯𑀻 𑀉𑀤𑀓𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸; 𑀉𑀤𑀓𑀁 𑀯𑀸𑀢𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁; 𑀯𑀸𑀢𑁄 𑀆𑀓𑀸𑀲𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀬𑁄 𑀬𑀁 𑀫𑀳𑀸𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀺; 𑀫𑀳𑀸𑀯𑀸𑀢𑀸 𑀯𑀸𑀬𑀦𑁆𑀢𑀸 𑀉𑀤𑀓𑀁 𑀓𑀫𑁆𑀧𑁂𑀦𑁆𑀢𑀺; 𑀉𑀤𑀓𑀁 𑀓𑀫𑁆𑀧𑀺𑀢𑀁 𑀧𑀣𑀯𑀺𑀁 𑀓𑀫𑁆𑀧𑁂𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀞𑀫𑁄 𑀳𑁂𑀢𑀼, 𑀧𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 (𑁧)

702\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀫𑀡𑁄 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀯𑀸 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀘𑁂𑀢𑁄𑀯𑀲𑀺𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀢𑀸 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀸𑁇 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀧𑀣𑀯𑀻𑀲𑀜𑁆𑀜𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸 𑀆𑀧𑁄𑀲𑀜𑁆𑀜𑀸𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀧𑀣𑀯𑀺𑀁 𑀓𑀫𑁆𑀧𑁂𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑁂𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑁂𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑁂𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 (𑁨)

703\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀢𑀼𑀲𑀺𑀢𑀸 𑀓𑀸𑀬𑀸 𑀘𑀯𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀢𑀺𑀬𑁄 𑀳𑁂𑀢𑀼, 𑀢𑀢𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 (𑁩)

704\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁄 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀫𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀳𑁂𑀢𑀼, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 (𑁪)

705\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀳𑁂𑀢𑀼, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 (𑁫)

706\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀓𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀙𑀝𑁆𑀞𑁄 𑀳𑁂𑀢𑀼, 𑀙𑀝𑁆𑀞𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 (𑁬)

707\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀑𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀲𑀢𑁆𑀢𑀫𑁄 𑀳𑁂𑀢𑀼, 𑀲𑀢𑁆𑀢𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 (𑁭)

708\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀤𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺, 𑀢𑀤𑀸𑀬𑀁 𑀧𑀣𑀯𑀻 𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀢𑀺 𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀝𑁆𑀞𑀫𑁄 𑀳𑁂𑀢𑀼, 𑀅𑀝𑁆𑀞𑀫𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀝𑁆𑀞 𑀳𑁂𑀢𑀽, 𑀅𑀝𑁆𑀞 𑀧𑀘𑁆𑀘𑀬𑀸 𑀫𑀳𑀢𑁄 𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑀲𑁆𑀲 𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑀸𑀬𑀸”𑀢𑀺𑁇 (𑁮)
