# 𑁨𑁇𑁧𑁇𑁮 𑀤𑀼𑀢𑀺𑀬𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀕𑁄𑀢𑀫𑀻𑀯𑀕𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀲𑀼𑀢𑁆𑀢

(58.)

608\. “𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀻𑀮𑀯𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼; 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀸; 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀣𑀸𑀫𑀯𑀸 𑀤𑀍𑀳𑀧𑀭𑀓𑁆𑀓𑀫𑁄 𑀅𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀥𑀼𑀭𑁄 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀳𑁄𑀢𑀺 𑀧𑀦𑁆𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑁄; 𑀅𑀭𑀢𑀺𑀭𑀢𑀺𑀲𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀭𑀢𑀺𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀪𑀬𑀪𑁂𑀭𑀯𑀲𑀳𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀪𑀬𑀪𑁂𑀭𑀯𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀯𑀺𑀳𑀭𑀢𑀺; 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀛𑀸𑀦𑀸𑀦𑀁 𑀆𑀪𑀺𑀘𑁂𑀢𑀲𑀺𑀓𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀸𑀦𑀁 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻; 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸…𑀧𑁂…  𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
