# 𑁨𑁇𑁧𑁇𑁪 𑀤𑀻𑀖𑀚𑀸𑀡𑀼𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀕𑁄𑀢𑀫𑀻𑀯𑀕𑁆𑀕

### 𑀤𑀻𑀖𑀚𑀸𑀡𑀼𑀲𑀼𑀢𑁆𑀢

(54.)

563. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀮𑀺𑀬𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀓𑁆𑀓𑀭𑀧𑀢𑁆𑀢𑀁 𑀦𑀸𑀫 𑀓𑁄𑀮𑀺𑀬𑀸𑀦𑀁 𑀦𑀺𑀕𑀫𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀤𑀻𑀖𑀚𑀸𑀡𑀼 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀤𑀻𑀖𑀚𑀸𑀡𑀼 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄 𑀧𑀼𑀢𑁆𑀢𑀲𑀫𑁆𑀩𑀸𑀥𑀲𑀬𑀦𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀸𑀫, 𑀓𑀸𑀲𑀺𑀓𑀘𑀦𑁆𑀤𑀦𑀁 𑀧𑀘𑁆𑀘𑀦𑀼𑀪𑁄𑀫, 𑀫𑀸𑀮𑀸𑀕𑀦𑁆𑀥𑀯𑀺𑀮𑁂𑀧𑀦𑀁 𑀥𑀸𑀭𑀬𑀸𑀫, 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀲𑀸𑀤𑀬𑀸𑀫𑁇 𑀢𑁂𑀲𑀁 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀢𑀣𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼 𑀬𑁂 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀲𑁆𑀲𑀼 𑀥𑀫𑁆𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀳𑀺𑀢𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀳𑀺𑀢𑀸𑀬 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

564. “𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀳𑀺𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀤𑀸, 𑀆𑀭𑀓𑁆𑀔𑀲𑀫𑁆𑀧𑀤𑀸, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸, 𑀲𑀫𑀚𑀻𑀯𑀺𑀢𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀤𑀸? 𑀇𑀥, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀓𑀫𑁆𑀫𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀚𑀻𑀯𑀺𑀓𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺—  𑀬𑀤𑀺 𑀓𑀲𑀺𑀬𑀸, 𑀬𑀤𑀺 𑀯𑀡𑀺𑀚𑁆𑀚𑀸𑀬, 𑀬𑀤𑀺 𑀕𑁄𑀭𑀓𑁆𑀔𑁂𑀦, 𑀬𑀤𑀺 𑀇𑀲𑁆𑀲𑀢𑁆𑀢𑁂𑀦, 𑀬𑀤𑀺 𑀭𑀸𑀚𑀧𑁄𑀭𑀺𑀲𑁂𑀦, 𑀬𑀤𑀺 𑀲𑀺𑀧𑁆𑀧𑀜𑁆𑀜𑀢𑀭𑁂𑀦—  𑀢𑀢𑁆𑀣 𑀤𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀮𑀲𑁄, 𑀢𑀢𑁆𑀭𑀼𑀧𑀸𑀬𑀸𑀬 𑀯𑀻𑀫𑀁𑀲𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀅𑀮𑀁 𑀓𑀸𑀢𑀼𑀁 𑀅𑀮𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀢𑀼𑀁𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀤𑀸𑁇 (𑁧:*𑁧*)

565. 𑀓𑀢𑀫𑀸 𑀘, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀆𑀭𑀓𑁆𑀔𑀲𑀫𑁆𑀧𑀤𑀸? 𑀇𑀥, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑁄𑀕𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀉𑀝𑁆𑀞𑀸𑀦𑀯𑀻𑀭𑀺𑀬𑀸𑀥𑀺𑀕𑀢𑀸 𑀩𑀸𑀳𑀸𑀩𑀮𑀧𑀭𑀺𑀘𑀺𑀢𑀸, 𑀲𑁂𑀤𑀸𑀯𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀸𑁇 𑀢𑁂 𑀆𑀭𑀓𑁆𑀔𑁂𑀦 𑀕𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀫𑁂 𑀇𑀫𑁂 𑀪𑁄𑀕𑁂 𑀦𑁂𑀯 𑀭𑀸𑀚𑀸𑀦𑁄 𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀘𑁄𑀭𑀸 𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀅𑀕𑁆𑀕𑀺 𑀟𑀳𑁂𑀬𑁆𑀬, 𑀦 𑀉𑀤𑀓𑀁 𑀯𑀳𑁂𑀬𑁆𑀬, 𑀦 𑀅𑀧𑁆𑀧𑀺𑀬𑀸 𑀤𑀸𑀬𑀸𑀤𑀸 𑀳𑀭𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀆𑀭𑀓𑁆𑀔𑀲𑀫𑁆𑀧𑀤𑀸𑁇 (𑁨:*𑁨*)

566. 𑀓𑀢𑀫𑀸 𑀘, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸? 𑀇𑀥, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑀲𑁆𑀫𑀺𑀁 𑀕𑀸𑀫𑁂 𑀯𑀸 𑀦𑀺𑀕𑀫𑁂 𑀯𑀸 𑀧𑀝𑀺𑀯𑀲𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀬𑁂 𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀕𑀳𑀧𑀢𑀻 𑀯𑀸 𑀕𑀳𑀧𑀢𑀺𑀧𑀼𑀢𑁆𑀢𑀸 𑀯𑀸 𑀤𑀳𑀭𑀸 𑀯𑀸 𑀯𑀼𑀤𑁆𑀥𑀲𑀻𑀮𑀺𑀦𑁄, 𑀯𑀼𑀤𑁆𑀥𑀸 𑀯𑀸 𑀯𑀼𑀤𑁆𑀥𑀲𑀻𑀮𑀺𑀦𑁄, 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸, 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸—  𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀢𑀺 𑀲𑀮𑁆𑀮𑀧𑀢𑀺 𑀲𑀸𑀓𑀘𑁆𑀙𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺; 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀢𑀺, 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀢𑀺, 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀢𑀺, 𑀬𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑀢𑀸𑁇 (𑁩:*𑁩*)

569. 𑀏𑀯𑀁 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀪𑁄𑀕𑀸𑀦𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀆𑀬𑀫𑀼𑀔𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀦 𑀇𑀢𑁆𑀣𑀺𑀥𑀼𑀢𑁆𑀢𑁄, 𑀦 𑀲𑀼𑀭𑀸𑀥𑀼𑀢𑁆𑀢𑁄, 𑀦 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀫𑀳𑀢𑁄 𑀢𑀍𑀆𑀓𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘𑁂𑀯 𑀆𑀬𑀫𑀼𑀔𑀸𑀦𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀅𑀧𑀸𑀬𑀫𑀼𑀔𑀸𑀦𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑁄 𑀬𑀸𑀦𑀺 𑀘𑁂𑀯 𑀆𑀬𑀫𑀼𑀔𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀬𑀸𑀦𑀺 𑀘 𑀅𑀧𑀸𑀬𑀫𑀼𑀔𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀧𑀺𑀤𑀳𑁂𑀬𑁆𑀬; 𑀤𑁂𑀯𑁄 𑀘 𑀲𑀫𑁆𑀫𑀸 𑀥𑀸𑀭𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀯𑁂𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑀲𑁆𑀲, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀫𑀳𑀢𑁄 𑀢𑀍𑀆𑀓𑀲𑁆𑀲 𑀯𑀼𑀤𑁆𑀥𑀺𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸, 𑀦𑁄 𑀧𑀭𑀺𑀳𑀸𑀦𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀏𑀯𑀁 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑁄𑀕𑀸𑀦𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀆𑀬𑀫𑀼𑀔𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀦 𑀇𑀢𑁆𑀣𑀺𑀥𑀼𑀢𑁆𑀢𑁄, 𑀦 𑀲𑀼𑀭𑀸𑀥𑀼𑀢𑁆𑀢𑁄, 𑀦 𑀅𑀓𑁆𑀔𑀥𑀼𑀢𑁆𑀢𑁄, 𑀓𑀮𑁆𑀬𑀸𑀡𑀫𑀺𑀢𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀳𑀸𑀬𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑁄𑁇 (𑁪:*𑁪*)

570. 𑀇𑀫𑁂 𑀔𑁄, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀳𑀺𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀸𑀬𑁇

571. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀫𑁂, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀳𑀺𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀲𑀼𑀔𑀸𑀬𑁇 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄? 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀤𑀸, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸, 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀤𑀸, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀤𑀸𑁇

572. 𑀓𑀢𑀫𑀸 𑀘, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀤𑀸? 𑀇𑀥, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀤𑁆𑀤𑀳𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀩𑁄𑀥𑀺𑀁—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀲𑀤𑁆𑀥𑀸𑀲𑀫𑁆𑀧𑀤𑀸𑁇 (𑁧:*𑁫*)

573. 𑀓𑀢𑀫𑀸 𑀘, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸? 𑀇𑀥, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸𑁇 (𑁨:*𑁬*)

574. 𑀓𑀢𑀫𑀸 𑀘, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀤𑀸? 𑀇𑀥, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀕𑀢𑀫𑀮𑀫𑀘𑁆𑀙𑁂𑀭𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺 𑀫𑀼𑀢𑁆𑀢𑀘𑀸𑀕𑁄 𑀧𑀬𑀢𑀧𑀸𑀡𑀺 𑀯𑁄𑀲𑁆𑀲𑀕𑁆𑀕𑀭𑀢𑁄 𑀬𑀸𑀘𑀬𑁄𑀕𑁄 𑀤𑀸𑀦𑀲𑀁𑀯𑀺𑀪𑀸𑀕𑀭𑀢𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀘𑀸𑀕𑀲𑀫𑁆𑀧𑀤𑀸𑁇 (𑁩:*𑁭*)

575. 𑀓𑀢𑀫𑀸 𑀘, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀤𑀸? 𑀇𑀥, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀉𑀤𑀬𑀢𑁆𑀣𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀭𑀺𑀬𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑁂𑀥𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑁆𑀧𑀤𑀸𑁇 (𑁪:*𑁮*)

576. 𑀇𑀫𑁂 𑀔𑁄, 𑀩𑁆𑀬𑀕𑁆𑀖𑀧𑀚𑁆𑀚, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀳𑀺𑀢𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀲𑀼𑀔𑀸𑀬𑀸𑀢𑀺𑁇

577. _𑀉𑀝𑁆𑀞𑀸𑀢𑀸 𑀓𑀫𑁆𑀫𑀥𑁂𑀬𑁆𑀬𑁂𑀲𑀼,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀥𑀸𑀦𑀯𑀸;_  
_𑀲𑀫𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺 𑀚𑀻𑀯𑀺𑀓𑀁,_  
_𑀲𑀫𑁆𑀪𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀢𑀺𑁇_  


578. _𑀲𑀤𑁆𑀥𑁄 𑀲𑀻𑀮𑁂𑀦 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀯𑀤𑀜𑁆𑀜𑀽 𑀯𑀻𑀢𑀫𑀘𑁆𑀙𑀭𑁄;_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀫𑀕𑁆𑀕𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺,_  
_𑀲𑁄𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀁𑁇_  


579. _𑀇𑀘𑁆𑀘𑁂𑀢𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀘,_  
_𑀲𑀤𑁆𑀥𑀲𑁆𑀲 𑀖𑀭𑀫𑁂𑀲𑀺𑀦𑁄;_  
_𑀅𑀓𑁆𑀔𑀸𑀢𑀸 𑀲𑀘𑁆𑀘𑀦𑀸𑀫𑁂𑀦,_  
_𑀉𑀪𑀬𑀢𑁆𑀣 𑀲𑀼𑀔𑀸𑀯𑀳𑀸𑁇_  


580. _𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀳𑀺𑀢𑀢𑁆𑀣𑀸𑀬,_  
_𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀲𑀼𑀔𑀸𑀬 𑀘;_  
_𑀏𑀯𑀫𑁂𑀢𑀁 𑀕𑀳𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀘𑀸𑀕𑁄 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀯𑀟𑁆𑀠𑀢𑀻”𑀢𑀺𑁇_  


---

581. 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
