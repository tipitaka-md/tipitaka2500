# 𑁨𑁇𑁧𑁇𑁯 𑀧𑀞𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀕𑁄𑀢𑀫𑀻𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀼𑀢𑁆𑀢

(59.)

610\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑀸 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲? 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄, 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀻, 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀦𑀸𑀕𑀸𑀫𑀻, 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀅𑀭𑀳𑀸, 𑀅𑀭𑀳𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺𑁇

611\. _𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀨𑀮𑁂 𑀞𑀺𑀢𑀸;_  
_𑀏𑀲 𑀲𑀁𑀖𑁄 𑀉𑀚𑀼𑀪𑀽𑀢𑁄,_  
_𑀧𑀜𑁆𑀜𑀸𑀲𑀻𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


612\. _𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀧𑁂𑀓𑁆𑀔𑀸𑀦 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀓𑀭𑁄𑀢𑀁 𑀑𑀧𑀥𑀺𑀓𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀲𑀁𑀖𑁂 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_
