# 𑁨𑁇𑁪𑁇𑁬 𑀬𑀲𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀬𑀲𑀲𑀼𑀢𑁆𑀢

(86.)

825\. 𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀁 𑀦𑀸𑀫 𑀓𑁄𑀲𑀮𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸—  “𑀲𑀫𑀡𑁄 𑀔𑀮𑀼 𑀪𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀯𑀦𑀲𑀡𑁆𑀟𑁂𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄…𑀧𑁂…  𑀲𑀸𑀥𑀼 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻’”𑀢𑀺𑁇

826\. 𑀅𑀣 𑀔𑁄 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀳𑀼𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀯𑀦𑀲𑀡𑁆𑀟𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀅𑀝𑁆𑀞𑀁𑀲𑀼 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀺𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀺𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑁂 𑀧𑀦 𑀢𑁂, 𑀦𑀸𑀕𑀺𑀢, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸 𑀓𑁂𑀯𑀝𑁆𑀝𑀸 𑀫𑀜𑁆𑀜𑁂 𑀫𑀘𑁆𑀙𑀯𑀺𑀮𑁄𑀧𑁂”𑀢𑀺? “𑀏𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀧𑀳𑀼𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀞𑀺𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 “𑀫𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀬𑀲𑁂𑀦 𑀲𑀫𑀸𑀕𑀫𑀁, 𑀫𑀸 𑀘 𑀫𑀬𑀸 𑀬𑀲𑁄𑁇 𑀬𑁄 𑀔𑁄, 𑀦𑀸𑀕𑀺𑀢, 𑀦𑀬𑀺𑀫𑀲𑁆𑀲 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻, 𑀲𑁄 𑀢𑀁 𑀫𑀻𑀍𑀳𑀲𑀼𑀔𑀁 𑀫𑀺𑀤𑁆𑀥𑀲𑀼𑀔𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑀼𑀔𑀁 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

827\. “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄𑁇 𑀬𑁂𑀦 𑀬𑁂𑀦𑁂𑀯 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀦𑁆𑀦𑀺𑀦𑁆𑀦𑀸𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀦𑁂𑀕𑀫𑀸 𑀘𑁂𑀯 𑀚𑀸𑀦𑀧𑀤𑀸 𑀘𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀣𑀼𑀮𑁆𑀮𑀨𑀼𑀲𑀺𑀢𑀓𑁂 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀬𑀣𑀸𑀦𑀺𑀦𑁆𑀦𑀁 𑀉𑀤𑀓𑀸𑀦𑀺 𑀧𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑁂𑀦 𑀬𑁂𑀦𑁂𑀯 𑀤𑀸𑀦𑀺 𑀪𑀕𑀯𑀸 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀦𑁆𑀦𑀺𑀦𑁆𑀦𑀸𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀦𑁂𑀕𑀫𑀸 𑀘𑁂𑀯 𑀚𑀸𑀦𑀧𑀤𑀸 𑀘𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀢𑀣𑀸 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀻𑀮𑀧𑀜𑁆𑀜𑀸𑀡𑀦𑁆”𑀢𑀺𑁇

828\. “𑀫𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀬𑀲𑁂𑀦 𑀲𑀫𑀸𑀕𑀫𑀁, 𑀫𑀸 𑀘 𑀫𑀬𑀸 𑀬𑀲𑁄𑁇 𑀬𑁄 𑀔𑁄, 𑀦𑀸𑀕𑀺𑀢, 𑀦𑀬𑀺𑀫𑀲𑁆𑀲 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀲𑁆𑀲 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻, 𑀲𑁄 𑀢𑀁 𑀫𑀻𑀍𑀳𑀲𑀼𑀔𑀁 𑀫𑀺𑀤𑁆𑀥𑀲𑀼𑀔𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀲𑀼𑀔𑀁 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑁇

829\. 𑀤𑁂𑀯𑀢𑀸𑀧𑀺 𑀔𑁄, 𑀦𑀸𑀕𑀺𑀢, 𑀏𑀓𑀘𑁆𑀘𑀸 𑀦𑀬𑀺𑀫𑀲𑁆𑀲 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀦𑀺𑀬𑁄 𑀅𑀲𑁆𑀲𑀼 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀺𑀦𑀺𑀬𑁄 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀺𑀦𑀺𑀬𑁄, 𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀔𑁄, 𑀦𑀸𑀕𑀺𑀢, 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀦 𑀳𑀺 𑀦𑀽𑀦𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀲 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀲𑁆𑀲𑀼 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀺𑀦𑁄𑁇 𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺’𑁇

830\. 𑀇𑀥𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀧𑀢𑁄𑀤𑀓𑁂𑀦 𑀲𑀜𑁆𑀚𑀕𑁆𑀖𑀦𑁆𑀢𑁂 𑀲𑀗𑁆𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑁂𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀦 𑀳𑀺 𑀦𑀽𑀦𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀲 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀲𑁆𑀲𑀼 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀺𑀦𑁄𑁇 𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀧𑀢𑁄𑀤𑀓𑁂𑀦 𑀲𑀜𑁆𑀚𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑀗𑁆𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺’𑁇 (𑁧)

831\. 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀉𑀤𑀭𑀸𑀯𑀤𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀲𑀼𑀔𑀁 𑀧𑀲𑁆𑀲𑀲𑀼𑀔𑀁 𑀫𑀺𑀤𑁆𑀥𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀦 𑀳𑀺 𑀦𑀽𑀦𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀫𑀲𑁆𑀲 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀲𑁆𑀲𑀼 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀺𑀦𑁄 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀺𑀦𑁄𑁇 𑀬𑀲𑁆𑀲𑀸𑀳𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀧𑀯𑀺𑀯𑁂𑀓𑀲𑀼𑀔𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑀲𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀩𑁄𑀥𑀲𑀼𑀔𑀲𑁆𑀲 𑀦𑀺𑀓𑀸𑀫𑀮𑀸𑀪𑀻 𑀅𑀓𑀺𑀘𑁆𑀙𑀮𑀸𑀪𑀻 𑀅𑀓𑀲𑀺𑀭𑀮𑀸𑀪𑀻𑁇 𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀉𑀤𑀭𑀸𑀯𑀤𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀬𑁆𑀬𑀲𑀼𑀔𑀁 𑀧𑀲𑁆𑀲𑀲𑀼𑀔𑀁 𑀫𑀺𑀤𑁆𑀥𑀲𑀼𑀔𑀁 𑀅𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺’𑁇 (𑁨)

832\. 𑀇𑀥𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀕𑀸𑀫𑀦𑁆𑀢𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀤𑀸𑀦𑀺 𑀇𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀡𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀯𑀸𑁇 𑀢𑀁 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀘𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀫𑀺 𑀕𑀸𑀫𑀦𑁆𑀢𑀯𑀺𑀳𑀸𑀭𑁂𑀦𑁇 (𑁩)

833\. 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀧𑀘𑀮𑀸𑀬𑀫𑀸𑀦𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀤𑀸𑀦𑀺 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀇𑀫𑀁 𑀦𑀺𑀤𑁆𑀤𑀸𑀓𑀺𑀮𑀫𑀣𑀁 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑀲𑀜𑁆𑀜𑀁𑀬𑁂𑀯 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀏𑀓𑀢𑁆𑀢𑀦𑁆’𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀫𑀺 𑀅𑀭𑀜𑁆𑀜𑀯𑀺𑀳𑀸𑀭𑁂𑀦𑁇 (𑁪)

834\. 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀤𑀸𑀦𑀺 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀳𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀫𑀺 𑀅𑀭𑀜𑁆𑀜𑀯𑀺𑀳𑀸𑀭𑁂𑀦𑁇 (𑁫)

835\. 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀲𑀫𑀸𑀳𑀺𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀤𑀸𑀦𑀺 𑀅𑀬𑀫𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀫𑀺 𑀅𑀭𑀜𑁆𑀜𑀯𑀺𑀳𑀸𑀭𑁂𑀦𑁇 (𑁬)

836\. 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀕𑀸𑀫𑀦𑁆𑀢𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀮𑀸𑀪𑀺𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀲𑁄 𑀢𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀦𑀺𑀓𑀸𑀫𑀬𑀫𑀸𑀦𑁄 𑀭𑀺𑀜𑁆𑀘𑀢𑀺 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀁, 𑀭𑀺𑀜𑁆𑀘𑀢𑀺 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺; 𑀕𑀸𑀫𑀦𑀺𑀕𑀫𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀁 𑀑𑀲𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸𑀲𑀁 𑀓𑀧𑁆𑀧𑁂𑀢𑀺𑁇 𑀢𑁂𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀫𑀺 𑀕𑀸𑀫𑀦𑁆𑀢𑀯𑀺𑀳𑀸𑀭𑁂𑀦𑁇 (𑁭)

837\. 𑀇𑀥 𑀧𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀁 𑀮𑀸𑀪𑀺𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀲𑁄 𑀢𑀁 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑀲𑀺𑀮𑁄𑀓𑀁 𑀧𑀝𑀺𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀦 𑀭𑀺𑀜𑁆𑀘𑀢𑀺 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀁, 𑀦 𑀭𑀺𑀜𑁆𑀘𑀢𑀺 𑀅𑀭𑀜𑁆𑀜𑀯𑀦𑀧𑀢𑁆𑀣𑀸𑀦𑀺 𑀧𑀦𑁆𑀢𑀸𑀦𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺𑁇 𑀢𑁂𑀦𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀳𑁄𑀫𑀺 𑀅𑀭𑀜𑁆𑀜𑀯𑀺𑀳𑀸𑀭𑁂𑀦𑁇 (𑁮)

838\. 𑀬𑀲𑁆𑀫𑀸𑀳𑀁, 𑀦𑀸𑀕𑀺𑀢, 𑀲𑀫𑀬𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀦 𑀓𑀜𑁆𑀘𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀧𑀼𑀭𑀢𑁄 𑀯𑀸 𑀧𑀘𑁆𑀙𑀢𑁄 𑀯𑀸, 𑀨𑀸𑀲𑀼 𑀫𑁂, 𑀦𑀸𑀕𑀺𑀢, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀧𑀲𑁆𑀲𑀸𑀯𑀓𑀫𑁆𑀫𑀸𑀬𑀸”𑀢𑀺𑁇
