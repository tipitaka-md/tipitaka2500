# 𑁨𑁇𑁪𑁇𑁮 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀧𑀯𑁂𑀤𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀧𑀯𑁂𑀤𑀦𑀻𑀬𑀲𑀼𑀢𑁆𑀢

(88.)

843. “𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀧𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀮𑀸𑀪𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀕𑀺𑀳𑀻 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀕𑀺𑀳𑀻 𑀕𑀺𑀳𑀻𑀳𑀺 𑀪𑁂𑀤𑁂𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀅𑀕𑁄𑀘𑀭𑁂 𑀘 𑀦𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀅𑀧𑁆𑀧𑀲𑀸𑀤𑀁 𑀧𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇

844. 𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀧𑀲𑀸𑀤𑀁 𑀧𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀦 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀮𑀸𑀪𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀦 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀦 𑀕𑀺𑀳𑀻 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀦 𑀕𑀺𑀳𑀻 𑀕𑀺𑀳𑀻𑀳𑀺 𑀪𑁂𑀤𑁂𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀕𑁄𑀘𑀭𑁂 𑀘 𑀦𑀁 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀧𑀲𑀸𑀤𑀁 𑀧𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇
