# 𑁨𑁇𑁪𑁇𑁭 𑀧𑀢𑁆𑀢𑀦𑀺𑀓𑀼𑀚𑁆𑀚𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀧𑀢𑁆𑀢𑀦𑀺𑀓𑀼𑀚𑁆𑀚𑀦𑀲𑀼𑀢𑁆𑀢

(87.)

840\. “𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀲𑀁𑀖𑁄 𑀧𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀮𑀸𑀪𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀯𑀸𑀲𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑁂𑀤𑁂𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀲𑀁𑀖𑁄 𑀧𑀢𑁆𑀢𑀁 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇

841\. 𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀲𑀁𑀖𑁄 𑀧𑀢𑁆𑀢𑀁 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀮𑀸𑀪𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀯𑀸𑀲𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑁂𑀤𑁂𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀲𑀁𑀖𑁄 𑀧𑀢𑁆𑀢𑀁 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

---
