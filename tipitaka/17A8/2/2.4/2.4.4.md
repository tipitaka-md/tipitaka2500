# 𑁨𑁇𑁪𑁇𑁪 𑀘𑁄𑀭𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀘𑁄𑀭𑀲𑀼𑀢𑁆𑀢

(84.)

817\. “𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀳𑀸𑀘𑁄𑀭𑁄 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀅𑀧𑁆𑀧𑀳𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀳𑀭𑀢𑀺, 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀇𑀢𑁆𑀣𑀺𑀁 𑀳𑀦𑀢𑀺, 𑀓𑀼𑀫𑀸𑀭𑀺𑀁 𑀤𑀽𑀲𑁂𑀢𑀺, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀁 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀢𑀺, 𑀭𑀸𑀚𑀓𑁄𑀲𑀁 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀢𑀺, 𑀅𑀘𑁆𑀘𑀸𑀲𑀦𑁆𑀦𑁂 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀦 𑀘 𑀦𑀺𑀥𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀳𑀸𑀘𑁄𑀭𑁄 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

818\. 𑀅𑀝𑁆𑀞𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀳𑀸𑀘𑁄𑀭𑁄 𑀦 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀳𑀺? 𑀦 𑀅𑀧𑁆𑀧𑀳𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀳𑀭𑀢𑀺, 𑀦 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀦 𑀇𑀢𑁆𑀣𑀺𑀁 𑀳𑀦𑀢𑀺, 𑀦 𑀓𑀼𑀫𑀸𑀭𑀺𑀁 𑀤𑀽𑀲𑁂𑀢𑀺, 𑀦 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀁 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀢𑀺, 𑀦 𑀭𑀸𑀚𑀓𑁄𑀲𑀁 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀢𑀺, 𑀦 𑀅𑀘𑁆𑀘𑀸𑀲𑀦𑁆𑀦𑁂 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀦𑀺𑀥𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀫𑀳𑀸𑀘𑁄𑀭𑁄 𑀦 𑀔𑀺𑀧𑁆𑀧𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
