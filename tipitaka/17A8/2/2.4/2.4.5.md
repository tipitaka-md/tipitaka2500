# 𑁨𑁇𑁪𑁇𑁫 𑀲𑀫𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑀺𑀯𑀕𑁆𑀕

### 𑀲𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(85.)

820\. “‘𑀲𑀫𑀡𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀯𑁂𑀤𑀕𑀽’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀪𑀺𑀲𑀓𑁆𑀓𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀦𑀺𑀫𑁆𑀫𑀮𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀯𑀺𑀫𑀮𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀜𑀸𑀡𑀻’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 ‘𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄’𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀅𑀥𑀺𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑀸𑀢𑀺𑁇

821\. _𑀬𑀁 𑀲𑀫𑀡𑁂𑀦 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂𑀦 𑀯𑀼𑀲𑀻𑀫𑀢𑀸;_  
_𑀬𑀁 𑀯𑁂𑀤𑀕𑀼𑀦𑀸 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀪𑀺𑀲𑀓𑁆𑀓𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


822\. _𑀬𑀁 𑀦𑀺𑀫𑁆𑀫𑀮𑁂𑀦 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀯𑀺𑀫𑀮𑁂𑀦 𑀲𑀼𑀘𑀻𑀫𑀢𑀸;_  
_𑀬𑀁 𑀜𑀸𑀡𑀺𑀦𑀸 𑀘 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁂𑀦 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁𑁇_  


823\. _𑀲𑁄𑀳𑀁 𑀯𑀺𑀚𑀺𑀢𑀲𑀗𑁆𑀕𑀸𑀫𑁄,_  
_𑀫𑀼𑀢𑁆𑀢𑁄 𑀫𑁄𑀘𑁂𑀫𑀺 𑀩𑀦𑁆𑀥𑀦𑀸;_  
_𑀦𑀸𑀕𑁄𑀫𑁆𑀳𑀺 𑀧𑀭𑀫𑀤𑀦𑁆𑀢𑁄,_  
_𑀅𑀲𑁂𑀔𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇_  


---

824\. 𑀧𑀜𑁆𑀘𑀫𑀁𑁇
