# 𑁨𑁇𑁩𑁇𑁧𑁦 𑀓𑀼𑀲𑀻𑀢𑀸𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀓𑀼𑀲𑀻𑀢𑀸𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑀲𑀼𑀢𑁆𑀢

(80.)

787\. “𑀅𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁧)

788\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑀁𑁇 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁨)

789\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀕𑁆𑀕𑁄 𑀫𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁩)

790\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀫𑀕𑁆𑀕𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁪)

791\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫)

792\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀕𑀭𑀼𑀓𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄 𑀫𑀸𑀲𑀸𑀘𑀺𑀢𑀁 𑀫𑀜𑁆𑀜𑁂𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀝𑁆𑀞𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬)

793\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄 𑀅𑀢𑁆𑀣𑀺 𑀓𑀧𑁆𑀧𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 (𑁭)

794\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀅𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀫𑀁 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀼𑁇 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀓𑀼𑀲𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 (𑁮)

795\. 𑀅𑀝𑁆𑀞𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞? 𑀇𑀥, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑁂 𑀓𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀫𑀬𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀦 𑀦 𑀲𑀼𑀓𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁧)

796\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺𑀁𑁇 𑀓𑀫𑁆𑀫𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀢𑀺𑀬𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁨)

797\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  𑀫𑀕𑁆𑀕𑁄 𑀔𑁄 𑀫𑁂 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦 𑀫𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑀦 𑀦 𑀲𑀼𑀓𑀭𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑀺𑀬𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁩)

798\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑀕𑁆𑀕𑁄 𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  𑀅𑀳𑀁 𑀔𑁄 𑀫𑀕𑁆𑀕𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀫𑀕𑁆𑀕𑀁 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁪)

799\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀦𑀸𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀮𑀳𑀼𑀓𑁄 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁫)

800\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  𑀅𑀳𑀁 𑀔𑁄 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀅𑀮𑀢𑁆𑀣𑀁 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀓𑀸𑀬𑁄 𑀩𑀮𑀯𑀸 𑀓𑀫𑁆𑀫𑀜𑁆𑀜𑁄𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀝𑁆𑀞𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁬)

801\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂 𑀅𑀬𑀁 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁄 𑀆𑀩𑀸𑀥𑁄𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀫𑁂 𑀆𑀩𑀸𑀥𑁄 𑀧𑀯𑀟𑁆𑀠𑁂𑀬𑁆𑀬𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺…𑀧𑁂…  𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁭)

802\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑀁 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀘𑀺𑀭𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀕𑁂𑀮𑀜𑁆𑀜𑀸𑁇 𑀞𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀬𑀁 𑀫𑁂 𑀆𑀩𑀸𑀥𑁄 𑀧𑀘𑁆𑀘𑀼𑀤𑀸𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑁇 𑀳𑀦𑁆𑀤𑀸𑀳𑀁 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀸𑀫𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀯𑀻𑀭𑀺𑀬𑀁 𑀆𑀭𑀪𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀥𑀺𑀕𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑀲𑁆𑀲 𑀲𑀘𑁆𑀙𑀺𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑁇 𑀇𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀫𑀁 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀼𑁇 (𑁮)

803\. 𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞 𑀆𑀭𑀫𑁆𑀪𑀯𑀢𑁆𑀣𑀽𑀦𑀻”𑀢𑀺𑁇
