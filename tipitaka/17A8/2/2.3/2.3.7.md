# 𑁨𑁇𑁩𑁇𑁭 𑀇𑀘𑁆𑀙𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

### 𑀇𑀘𑁆𑀙𑀸𑀲𑀼𑀢𑁆𑀢

(77.)

765\. 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀯𑀼𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺𑁇 “𑀆𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—

766\. “𑀅𑀝𑁆𑀞𑀺𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀖𑀝𑀢𑀺, 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄, 𑀖𑀝𑀢𑁄, 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀲𑁄𑀘𑀢𑀺 𑀓𑀺𑀮𑀫𑀢𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀖𑀝𑀢𑀺, 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀲𑁄𑀘𑀻 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁧)

767\. 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀖𑀝𑀢𑀺, 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄 𑀖𑀝𑀢𑁄 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘, 𑀫𑀤𑀻 𑀘 𑀧𑀫𑀸𑀤𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁨)

768\. 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄, 𑀅𑀖𑀝𑀢𑁄, 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀲𑁄𑀘𑀢𑀺 𑀓𑀺𑀮𑀫𑀢𑀺 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀲𑁄𑀘𑀻 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁩)

769\. 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄, 𑀅𑀖𑀝𑀢𑁄, 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀦 𑀖𑀝𑀢𑀺 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘, 𑀫𑀤𑀻 𑀘 𑀧𑀫𑀸𑀤𑀻 𑀘, 𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁪)

770\. 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀖𑀝𑀢𑀺, 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄, 𑀖𑀝𑀢𑁄, 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀦 𑀲𑁄𑀘𑀢𑀺 𑀦 𑀓𑀺𑀮𑀫𑀢𑀺 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀦 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺 𑀖𑀝𑀢𑀺 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀦 𑀘 𑀲𑁄𑀘𑀻 𑀦 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁫)

771\. 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀖𑀝𑀢𑀺, 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀉𑀝𑁆𑀞𑀳𑀢𑁄, 𑀖𑀝𑀢𑁄, 𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀖𑀝𑀢𑀺, 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘, 𑀦 𑀘 𑀫𑀤𑀻 𑀦 𑀘 𑀧𑀫𑀸𑀤𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁬)

772\. 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄, 𑀅𑀖𑀝𑀢𑁄, 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑁄 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀅𑀮𑀸𑀪𑁂𑀦 𑀦 𑀲𑁄𑀘𑀢𑀺 𑀦 𑀓𑀺𑀮𑀫𑀢𑀺 𑀦 𑀧𑀭𑀺𑀤𑁂𑀯𑀢𑀺, 𑀦 𑀉𑀭𑀢𑁆𑀢𑀸𑀍𑀇𑀁 𑀓𑀦𑁆𑀤𑀢𑀺, 𑀦 𑀲𑀫𑁆𑀫𑁄𑀳𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀘 𑀮𑀸𑀪𑀻, 𑀦 𑀘 𑀲𑁄𑀘𑀻 𑀦 𑀘 𑀧𑀭𑀺𑀤𑁂𑀯𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 (𑁭)

773\. 𑀇𑀥 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀦𑀺𑀭𑀸𑀬𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑀺𑀦𑁄 𑀇𑀘𑁆𑀙𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀲𑁄 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬𑁇 𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀝𑁆𑀞𑀳𑀢𑁄, 𑀅𑀖𑀝𑀢𑁄, 𑀅𑀯𑀸𑀬𑀫𑀢𑁄 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀮𑀸𑀪𑁂𑀦 𑀦 𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀚𑁆𑀚𑀢𑀺 𑀦 𑀧𑀫𑀸𑀤𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀸𑀯𑀼𑀲𑁄, ‘𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀘𑁆𑀙𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀦 𑀉𑀝𑁆𑀞𑀳𑀢𑀺, 𑀦 𑀖𑀝𑀢𑀺, 𑀦 𑀯𑀸𑀬𑀫𑀢𑀺 𑀮𑀸𑀪𑀸𑀬, 𑀮𑀸𑀪𑀻 𑀘, 𑀦 𑀘 𑀫𑀤𑀻 𑀦 𑀘 𑀧𑀫𑀸𑀤𑀻, 𑀅𑀘𑁆𑀘𑀼𑀢𑁄 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸’𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀝𑁆𑀞 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀦𑁆”𑀢𑀺𑁇 (𑁮)

---
