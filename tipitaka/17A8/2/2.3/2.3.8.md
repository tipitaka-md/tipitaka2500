# 𑁨𑁇𑁩𑁇𑁮 𑀅𑀮𑀁𑀲𑀼𑀢𑁆𑀢

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀬𑀫𑀓𑀯𑀕𑁆𑀕

* 𑀅𑀮𑀁𑀲𑀼𑀢𑁆𑀢

(78.)

775\. 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺…𑀧𑁂…  𑀙𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀙𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄, 𑀧𑁄𑀭𑀺𑀬𑀸 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀍𑀆𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀙𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇

776\. 𑀧𑀜𑁆𑀘𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇

777\. 𑀘𑀢𑀽𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀦𑁄 𑀘 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇

778\. 𑀘𑀢𑀽𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀥𑀸𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀢𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇

779\. 𑀢𑀻𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀦𑁄 𑀘 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇

780\. 𑀢𑀻𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀲𑀼𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀥𑀸𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸, 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇

781\. 𑀤𑁆𑀯𑀻𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀦𑁄 𑀘 𑀲𑀼𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀥𑀸𑀢𑀸𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀘 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀦𑁄 𑀘 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄, 𑀦𑀸𑀮𑀁 𑀧𑀭𑁂𑀲𑀁𑁇

782\. 𑀤𑁆𑀯𑀻𑀳𑀸𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑁆𑀯𑀻𑀳𑀺? 𑀇𑀥𑀸𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀳𑁂𑀯 𑀔𑁄 𑀔𑀺𑀧𑁆𑀧𑀦𑀺𑀲𑀦𑁆𑀢𑀺 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀼𑀲𑀮𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼; 𑀦𑁄 𑀘 𑀲𑀼𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀥𑀸𑀭𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀥𑀸𑀢𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀳𑁄𑀢𑀺; 𑀦𑁄 𑀘 𑀅𑀢𑁆𑀣𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀫𑀜𑁆𑀜𑀸𑀬 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀘𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀯𑀸𑀓𑁆𑀓𑀭𑀡𑁄, 𑀧𑁄𑀭𑀺𑀬𑀸 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀍𑀆𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸; 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑀓𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀓𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀓𑁄 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁆𑀯𑀻𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀮𑀁 𑀧𑀭𑁂𑀲𑀁, 𑀦𑀸𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄”𑀢𑀺𑁇

---
