# 𑁨𑁇𑁬𑁇𑁨

* 𑀭𑀸𑀕𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(118.)

858\. “𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀅𑀝𑁆𑀞? 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀸𑀦𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺𑁇 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀅𑀭𑀽𑀧𑀲𑀜𑁆𑀜𑀻 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀭𑀽𑀧𑀸𑀦𑀺 𑀧𑀲𑁆𑀲𑀢𑀺 𑀦𑀻𑀮𑀸𑀦𑀺 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀤𑀲𑁆𑀲𑀦𑀸𑀦𑀺 𑀦𑀻𑀮𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺 𑀧𑀻𑀢𑀸𑀦𑀺 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺…𑀧𑁂…  𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀡𑁆𑀡𑀸𑀦𑀺…𑀧𑁂…  𑀑𑀤𑀸𑀢𑀸𑀦𑀺 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸𑀦𑀺…𑀧𑁂…  𑀑𑀤𑀸𑀢𑀦𑀺𑀪𑀸𑀲𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀏𑀯𑀁𑀲𑀜𑁆𑀜𑀻 𑀳𑁄𑀢𑀺—  𑀭𑀸𑀕𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀇𑀫𑁂 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀸 𑀪𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀸”𑁇
