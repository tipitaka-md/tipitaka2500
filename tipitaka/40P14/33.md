# 𑁩𑁩 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀤𑀼𑀓

### 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀤𑀼𑀓

(48.)

153\. 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)

154\. 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀘 𑀦𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

155\. 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀜𑁆𑀘 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

156\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀧𑀜𑁆𑀘, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀧𑀜𑁆𑀘𑁇
