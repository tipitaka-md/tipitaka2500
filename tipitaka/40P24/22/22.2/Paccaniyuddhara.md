# 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀼𑀤𑁆𑀥𑀸𑀭

* 𑀲𑀭𑀡𑀤𑀼𑀓, 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀧𑀤

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀼𑀤𑁆𑀥𑀸𑀭

(92.)

195\. 𑀦𑀲𑀭𑀡𑁄 𑀦𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀭𑀡𑀲𑁆𑀲 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

196\. 𑀦𑀅𑀭𑀡𑁄 𑀦𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀭𑀡𑀲𑁆𑀲 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂, 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀦𑁅𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀏𑀓𑀁, 𑀦𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀦𑁄𑀅𑀯𑀺𑀕𑀢𑁂 𑀤𑁆𑀯𑁂𑁇

197\. 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

198\. 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀤𑁆𑀯𑁂, 𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

199\. (𑀬𑀣𑀸 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓𑁂 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑀁 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)

200\. 𑀅𑀦𑀼𑀮𑁄𑀫𑀤𑀼𑀓𑀢𑀺𑀓𑀧𑀝𑁆𑀞𑀸𑀦𑀢𑁄 𑀧𑀝𑁆𑀞𑀸𑀬 𑀬𑀸𑀯 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀸 𑀢𑀺𑀁𑀲𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀪𑀸𑀡𑀯𑀸𑀭𑁂𑀳𑀺 𑀧𑀝𑁆𑀞𑀸𑀦𑀁𑁇

---

201\. 𑀥𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑁂 𑀤𑀼𑀓𑀤𑀼𑀓𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

202\. 𑀧𑀘𑁆𑀘𑀦𑀻𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

203\. 𑀧𑀝𑁆𑀞𑀸𑀦𑀧𑀓𑀭𑀡𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
