# 𑁧𑁩 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓𑀸𑀤𑀺, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

### 𑀫𑀳𑀦𑁆𑀢𑀭𑀤𑀼𑀓𑀸𑀤𑀺, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

(59.)

133\. 𑀦𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀢𑀻𑀡𑀺, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀧𑀼𑀭𑁂𑀚𑀸𑀢𑁂 𑀆𑀲𑁂𑀯𑀦𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

134\. 𑀦𑀅𑀦𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀦𑀦𑀳𑁂𑀢𑀼𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀦𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇
