# 𑁮𑁫 𑀲𑀭𑀡𑀤𑀼𑀓, 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

### 𑀲𑀭𑀡𑀤𑀼𑀓, 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀢𑁆𑀢𑀺𑀓

(97.)

268. 𑀅𑀭𑀡𑀁 𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀲𑀭𑀡𑁄 𑀦𑀫𑀕𑁆𑀕𑀸𑀭𑀫𑁆𑀫𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

269. 𑀅𑀭𑀡𑀁 𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀲𑀭𑀡𑁄 𑀦𑀫𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

270. 𑀅𑀭𑀡𑀁 𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀲𑀭𑀡𑁄 𑀦𑀫𑀕𑁆𑀕𑀸𑀥𑀺𑀧𑀢𑀺 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇
