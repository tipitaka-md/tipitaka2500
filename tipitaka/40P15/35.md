# 𑁩𑁫 𑀆𑀲𑀯𑀲𑀸𑀲𑀯𑀤𑀼𑀓𑀸𑀤𑀺, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

* 𑀆𑀲𑀯𑀲𑀸𑀲𑀯𑀤𑀼𑀓𑀸𑀤𑀺, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

(44.)

128\. 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑀅𑀦𑀸𑀲𑀯𑁄 𑀘 𑀦𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇

129\. 𑀆𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀲𑀸𑀲𑀯𑀜𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁄𑀆𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑀅𑀦𑀸𑀲𑀯𑁄 𑀘 𑀦𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸… 𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

130\. 𑀲𑀸𑀲𑀯𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀘 𑀆𑀲𑀯𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀅𑀦𑀸𑀲𑀯𑁄 𑀘𑁂𑀯 𑀦𑀦𑁄 𑀘 𑀆𑀲𑀯𑁄 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸… 𑁇 𑀳𑁂𑀢𑀼𑀬𑀸 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑀻𑀡𑀺𑁇

131\. (𑀆𑀲𑀯𑀆𑀲𑀯𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓𑀁 𑀆𑀲𑀯𑀲𑀸𑀲𑀯𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀁𑁇)
