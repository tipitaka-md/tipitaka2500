# 𑁨𑁫 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓𑀸𑀤𑀺, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

### 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓𑀸𑀤𑀺, 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓

(32.)

89\. 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

90\. 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

91\. 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀦 𑀘 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘𑁂𑀯 𑀦 𑀘 𑀳𑁂𑀢𑀼 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀦 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀦𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁩)

92\. 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

(33.)

93\. 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀦𑀅𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀏𑀢𑁂𑀦 𑀉𑀧𑀸𑀬𑁂𑀦 𑀦𑀯 𑀧𑀜𑁆𑀳𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)

94\. 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀦𑀯, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀦𑀢𑁆𑀣𑀺𑀬𑀸 𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇

(34.)

95\. 𑀳𑁂𑀢𑀼 𑀘𑁂𑀯 𑀲𑀳𑁂𑀢𑀼𑀓𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑀅𑀳𑁂𑀢𑀼𑀓𑀲𑁆𑀲 𑀘 𑀦𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑀦𑀯 𑀧𑀜𑁆𑀳𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇)

96\. 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀳𑁂𑀢𑀼𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀤𑀼𑀓𑀁 𑀳𑁂𑀢𑀼𑀲𑀳𑁂𑀢𑀼𑀓𑀤𑀼𑀓𑀲𑀤𑀺𑀲𑀁𑁇 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇 𑀦𑀯 𑀧𑀜𑁆𑀳𑀸𑁇)
