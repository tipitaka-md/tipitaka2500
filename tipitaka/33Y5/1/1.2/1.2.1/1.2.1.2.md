# 𑁧𑁇𑁨𑁇𑁧𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(11.)

128\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

129\. 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁𑁇

130\. «  𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

131\. »  𑀦 𑀲𑀫𑀼𑀤𑀬𑁄 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

132\. «  𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦 𑀲𑀫𑀼𑀤𑀬𑁄𑀢𑀺?

133\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁, 𑀲𑀫𑀼𑀤𑀬𑁄𑁇 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀲𑀫𑀼𑀤𑀬𑁄 𑀦 𑀘 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁𑁇

134\. »  𑀦 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

135\. «  𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦 𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺?

136\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁, 𑀦𑀺𑀭𑁄𑀥𑁄𑁇 𑀦𑀺𑀭𑁄𑀥𑀜𑁆𑀘 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦 𑀘 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁𑁇

137\. »  𑀦 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

138\. «  𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦 𑀫𑀕𑁆𑀕𑁄𑀢𑀺?

139\. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁, 𑀫𑀕𑁆𑀕𑁄𑁇 𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦 𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀘 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁𑁇
