# 𑁧𑁇𑁨𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(10.)

116\. »  𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

117\. «  𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀤𑀼𑀓𑁆𑀔𑀦𑁆𑀢𑀺?

118\. 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁, 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁𑁇 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇

119\. »  𑀲𑀫𑀼𑀤𑀬𑁄 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

120\. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀘𑁂𑀯 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇

121\. «  𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀫𑀼𑀤𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

122\. »  𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

123\. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀘𑁂𑀯 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇

124\. «  𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑀺𑀭𑁄𑀥𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

125\. »  𑀫𑀕𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

126\. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀫𑀕𑁆𑀕𑁄 𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇

127\. «  𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀫𑀕𑁆𑀕𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
