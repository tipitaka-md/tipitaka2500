# 𑁧𑁇𑁨𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀲𑀘𑁆𑀘𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(23.)

201\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

202\. 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀘𑁆𑀘𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑁇

203\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

204\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

205\. 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀘𑁆𑀘𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑁇

206\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(24.)

207\. »  𑀦 𑀲𑀫𑀼𑀤𑀬𑁄 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

208\. 𑀲𑀫𑀼𑀤𑀬𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀲𑀘𑁆𑀘𑀸𑁇 𑀲𑀫𑀼𑀤𑀬𑀜𑁆𑀘 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀲𑀫𑀼𑀤𑀬𑁄 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑁇

209\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂… 𑁇

(25.)

210\. »  𑀦 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

211\. 𑀦𑀺𑀭𑁄𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂… 𑁇

(26.)

212\. »  𑀦 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

213\. 𑀫𑀕𑁆𑀕𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑁄, 𑀲𑀘𑁆𑀘𑀸𑁇 𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑁇

214\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

215\. »  𑀦 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

216\. 𑀫𑀕𑁆𑀕𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑁄, 𑀲𑀘𑁆𑀘𑀸𑁇 𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑁇

217\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 …𑀧𑁂… 𑁇

218\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

---
