# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀲𑀘𑁆𑀘𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(21.)

182\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

183\. 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀘𑁆𑀘𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑁇

184\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

185\. 𑀦 𑀲𑀫𑀼𑀤𑀬𑁄 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

186\. 𑀲𑀫𑀼𑀤𑀬𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀦𑀺𑀭𑁄𑀥𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸…𑀧𑁂… 𑁇

187\. »  𑀦 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

188\. 𑀫𑀕𑁆𑀕𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑁄, 𑀲𑀘𑁆𑀘𑀸𑁇 𑀫𑀕𑁆𑀕𑀜𑁆𑀘 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀘 𑀲𑀘𑁆𑀘𑀸𑁇

189\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
