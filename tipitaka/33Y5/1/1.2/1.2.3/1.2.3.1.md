# 𑁧𑁇𑁨𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀲𑀘𑁆𑀘𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫

(20.)

174\. »  𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

175\. «  𑀲𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

176\. 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁𑁇

177\. »  𑀲𑀫𑀼𑀤𑀬𑁄 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

178\. »  𑀦𑀺𑀭𑁄𑀥𑁄 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

179\. »  𑀫𑀕𑁆𑀕𑁄 𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

180\. «  𑀲𑀘𑁆𑀘𑀸 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

181\. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁𑁇
