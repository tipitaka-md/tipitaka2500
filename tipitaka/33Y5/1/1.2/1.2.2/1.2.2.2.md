# 𑁧𑁇𑁨𑁇𑁨𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(16.)

158\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

159\. 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁𑁇

160\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

161\. »  𑀦 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

162\. 𑀓𑀸𑀬𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀘𑁂𑀢𑀲𑀺𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁𑁇 𑀤𑀼𑀓𑁆𑀔𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀦 𑀘𑁂𑀯 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁𑁇

163\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(17.)

164\. »  𑀦 𑀲𑀫𑀼𑀤𑀬𑁄 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

165\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

166\. »  𑀦 𑀲𑀫𑀼𑀤𑀬𑁄 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

167\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(18.)

168\. »  𑀦 𑀦𑀺𑀭𑁄𑀥𑁄 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

169\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(19.)

170\. »  𑀦 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

171\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

172\. »  𑀦 𑀫𑀕𑁆𑀕𑁄 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

173\. «  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀦 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
