# 𑁧𑁇𑁨𑁇𑁨𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(12.)

140. »  𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

141. «  𑀲𑀘𑁆𑀘𑀸 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

142. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁂𑀯 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁𑁇

143. »  𑀤𑀼𑀓𑁆𑀔𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

144. «  𑀲𑀘𑁆𑀘𑀸 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

145. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁𑁇

(13.)

146. »  𑀲𑀫𑀼𑀤𑀬𑁄 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

147. 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀲𑀫𑀼𑀤𑀬𑁄, 𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁𑁇 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀲𑀫𑀼𑀤𑀬𑁄 𑀘𑁂𑀯 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇

148. «  𑀲𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

149. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁𑁇

(14.)

150. »  𑀦𑀺𑀭𑁄𑀥𑁄 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

151. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀦𑀺𑀭𑁄𑀥𑁄, 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁𑁇 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀦𑀺𑀭𑁄𑀥𑁄 𑀘𑁂𑀯 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇

152. «  𑀲𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

153. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁𑁇

(15.)

154. »  𑀫𑀕𑁆𑀕𑁄 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

155. 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑁄 𑀫𑀕𑁆𑀕𑁄, 𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁𑁇 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀫𑀕𑁆𑀕𑁄 𑀘𑁂𑀯 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇

156. «  𑀲𑀘𑁆𑀘𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺? …𑀧𑁂…  𑀲𑀘𑁆𑀘𑀸 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀦𑁆𑀢𑀺?

157. 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁 𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁂𑀯 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀜𑁆𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀲𑀘𑁆𑀘𑀸 𑀦 𑀦𑀺𑀭𑁄𑀥𑀲𑀘𑁆𑀘𑀁𑁇
