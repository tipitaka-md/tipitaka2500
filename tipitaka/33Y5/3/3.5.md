# 𑁩𑁇𑁫 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

(169.)

811\. »  𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

812\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

813\. »  𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

814\. 𑀬𑁂 𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀭𑀳𑀸 𑀬𑁂 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

815\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

816\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀭𑀳𑀸 𑀬𑁂 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇
