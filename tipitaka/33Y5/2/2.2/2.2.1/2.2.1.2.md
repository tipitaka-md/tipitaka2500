# 𑁨𑁇𑁨𑁇𑁧𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(90.)

490\. »  𑀬𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀻𑀢𑀺?

491\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀢𑀺…𑀧𑁂… 𑁇

492\. (𑀬𑀢𑁆𑀣𑀓𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀧𑀺 𑀦𑀺𑀭𑁄𑀥𑁂𑀧𑀺 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑁂𑀧𑀺 𑀲𑀤𑀺𑀲𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁𑁇)
