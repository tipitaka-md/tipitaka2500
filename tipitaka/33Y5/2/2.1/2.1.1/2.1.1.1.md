# 𑁨𑁇𑁧𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(27.)

220. »  𑀬𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

221. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀯𑀢𑁆𑀢𑁂 𑀢𑀡𑁆𑀳𑀸𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀡𑁆𑀳𑀸𑀬 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

222. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

223. »  𑀬𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

224. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀯𑀢𑁆𑀢𑁂 𑀫𑀕𑁆𑀕𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

225. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

226. 𑀅𑀭𑀽𑀧𑁂 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

(28.)

227. »  𑀬𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

228. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇
