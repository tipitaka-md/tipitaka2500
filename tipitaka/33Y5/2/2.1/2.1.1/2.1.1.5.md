# 𑁨𑁇𑁧𑁇𑁧𑁇𑁫 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

(34.)

251\. »  𑀬𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

252\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

253\. »  𑀬𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

254\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

(35.)

255\. »  𑀬𑀢𑁆𑀣 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

256\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀢𑀢𑁆𑀣 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀻𑀢𑀺?

257\. 𑀅𑀧𑀸𑀬𑁂 𑀢𑀢𑁆𑀣 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇
