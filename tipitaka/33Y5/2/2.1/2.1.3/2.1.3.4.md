# 𑁨𑁇𑁧𑁇𑁩𑁇𑁪 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮

(53.)

325\. »  𑀬𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

326\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

327\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

328\. »  𑀬𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

329\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

330\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑁂 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

(54.)

331\. »  𑀬𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

332\. 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇

333\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺?

334\. 𑀬𑁂 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸 𑀫𑀕𑁆𑀕𑀁 𑀦 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀁 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀫𑀕𑁆𑀕𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀫𑀼𑀤𑀬𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇
