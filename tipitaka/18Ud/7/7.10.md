# 𑁭𑁇𑁧𑁦 𑀉𑀢𑁂𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀉𑀢𑁂𑀦𑀲𑀼𑀢𑁆𑀢

(70.)

652\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀉𑀢𑁂𑀦𑀲𑁆𑀲 𑀉𑀬𑁆𑀬𑀸𑀦𑀕𑀢𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀤𑀟𑁆𑀠𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀘 𑀇𑀢𑁆𑀣𑀺𑀲𑀢𑀸𑀦𑀺 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀸𑀫𑀸𑀯𑀢𑀻𑀧𑀫𑀼𑀔𑀸𑀦𑀺𑁇

653\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀜𑁆𑀜𑁄 𑀉𑀢𑁂𑀦𑀲𑁆𑀲 𑀉𑀬𑁆𑀬𑀸𑀦𑀕𑀢𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀤𑀟𑁆𑀠𑀁, 𑀧𑀜𑁆𑀘 𑀘 𑀇𑀢𑁆𑀣𑀺𑀲𑀢𑀸𑀦𑀺 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸𑀦𑀺 𑀲𑀸𑀫𑀸𑀯𑀢𑀻𑀧𑀫𑀼𑀔𑀸𑀦𑀺𑁇 𑀢𑀸𑀲𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀓𑀸 𑀕𑀢𑀺 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄”𑀢𑀺?

654\. “𑀲𑀦𑁆𑀢𑁂𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑀸, 𑀲𑀦𑁆𑀢𑀺 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑁄, 𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑁄𑁇 𑀲𑀩𑁆𑀩𑀸 𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄 𑀅𑀦𑀺𑀧𑁆𑀨𑀮𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸”𑀢𑀺𑁇

655\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

656\. _“𑀫𑁄𑀳𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀦𑁄 𑀮𑁄𑀓𑁄,_  
_𑀪𑀩𑁆𑀩𑀭𑀽𑀧𑁄𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀉𑀧𑀥𑀺𑀩𑀦𑁆𑀥𑀦𑁄 𑀩𑀸𑀮𑁄,_  
_𑀢𑀫𑀲𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄;_  
_𑀲𑀲𑁆𑀲𑀢𑁄𑀭𑀺𑀯 𑀔𑀸𑀬𑀢𑀺,_  
_𑀧𑀲𑁆𑀲𑀢𑁄 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀦𑁆”𑀢𑀺𑁇 (𑁮𑁩)_

658\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



659\. _𑀤𑁆𑀯𑁂 𑀪𑀤𑁆𑀤𑀺𑀬𑀸 𑀤𑁆𑀯𑁂 𑀘 𑀲𑀢𑁆𑀢𑀸,_  
_𑀮𑀓𑀼𑀡𑁆𑀟𑀓𑁄 𑀢𑀡𑁆𑀳𑀸𑀓𑁆𑀔𑀬𑁄;_  
_𑀧𑀧𑀜𑁆𑀘𑀔𑀬𑁄 𑀘 𑀓𑀘𑁆𑀘𑀸𑀦𑁄,_  
_𑀉𑀤𑀧𑀸𑀦𑀜𑁆𑀘 𑀉𑀢𑁂𑀦𑁄𑀢𑀺𑁇_
