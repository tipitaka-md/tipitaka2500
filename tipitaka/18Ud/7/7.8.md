# 𑁭𑁇𑁮 𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

* 𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(68.)

639\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬𑁇

640\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬𑁇

641\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

642\. _“𑀬𑀲𑁆𑀲 𑀲𑀺𑀬𑀸 𑀲𑀩𑁆𑀩𑀤𑀸 𑀲𑀢𑀺,_  
_𑀲𑀢𑀢𑀁 𑀓𑀸𑀬𑀕𑀢𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀦𑁄 𑀘𑀲𑁆𑀲 𑀦𑁄 𑀘 𑀫𑁂 𑀲𑀺𑀬𑀸,_  
_𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀦 𑀘 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀯𑀺𑀳𑀸𑀭𑀺 𑀢𑀢𑁆𑀣 𑀲𑁄,_  
_𑀓𑀸𑀮𑁂𑀦𑁂𑀯 𑀢𑀭𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁮𑁧)_
