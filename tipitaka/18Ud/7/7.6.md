# 𑁭𑁇𑁬 𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

* 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

* 𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀲𑀼𑀢𑁆𑀢

(66.)

630\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑁄𑁇

631\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀜𑁆𑀜𑀸𑀲𑀺𑀓𑁄𑀡𑁆𑀟𑀜𑁆𑀜𑀁 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀢𑀡𑁆𑀳𑀸𑀲𑀗𑁆𑀔𑀬𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀁𑁇

632\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

633\. _“𑀬𑀲𑁆𑀲 𑀫𑀽𑀮𑀁 𑀙𑀫𑀸 𑀦𑀢𑁆𑀣𑀺,_  
_𑀧𑀡𑁆𑀡𑀸 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀢𑁄 𑀮𑀢𑀸;_  
_𑀢𑀁 𑀥𑀻𑀭𑀁 𑀩𑀦𑁆𑀥𑀦𑀸 𑀫𑀼𑀢𑁆𑀢𑀁,_  
_𑀓𑁄 𑀢𑀁 𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑀼𑀫𑀭𑀳𑀢𑀺;_  
_𑀤𑁂𑀯𑀸𑀧𑀺 𑀦𑀁 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀼𑀦𑀸𑀧𑀺 𑀧𑀲𑀁𑀲𑀺𑀢𑁄”𑀢𑀺𑁇 (𑁭𑁯)_
