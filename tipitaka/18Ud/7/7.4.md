# 𑁭𑁇𑁪 𑀤𑀼𑀢𑀺𑀬𑀲𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀲𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(64.)

618. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀓𑀸𑀫𑁂𑀲𑀼 𑀲𑀢𑁆𑀢𑀸 ( ) 𑀭𑀢𑁆𑀢𑀸 𑀕𑀺𑀤𑁆𑀥𑀸 𑀕𑀣𑀺𑀢𑀸 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑀸 𑀅𑀚𑁆𑀛𑁄𑀲𑀦𑁆𑀦𑀸 𑀅𑀦𑁆𑀥𑀻𑀓𑀢𑀸 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀚𑀸𑀢𑀸 𑀓𑀸𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇

619. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀓𑀸𑀫𑁂𑀲𑀼 𑀲𑀢𑁆𑀢𑁂 𑀭𑀢𑁆𑀢𑁂 𑀕𑀺𑀤𑁆𑀥𑁂 𑀕𑀣𑀺𑀢𑁂 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀅𑀚𑁆𑀛𑁄𑀧𑀦𑁆𑀦𑁂 𑀅𑀦𑁆𑀥𑀻𑀓𑀢𑁂 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀚𑀸𑀢𑁂 𑀓𑀸𑀫𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁂𑁇

620. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

621. _“𑀓𑀸𑀫𑀦𑁆𑀥𑀸 𑀚𑀸𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀢𑀡𑁆𑀳𑀸𑀙𑀤𑀦𑀙𑀸𑀤𑀺𑀢𑀸;_  
_𑀧𑀫𑀢𑁆𑀢𑀩𑀦𑁆𑀥𑀼𑀦𑀸 𑀩𑀤𑁆𑀥𑀸,_  
_𑀫𑀘𑁆𑀙𑀸𑀯 𑀓𑀼𑀫𑀺𑀦𑀸𑀫𑀼𑀔𑁂;_  
_𑀚𑀭𑀸𑀫𑀭𑀡𑀫𑀦𑁆𑀯𑁂𑀦𑁆𑀢𑀺,_  
_𑀯𑀘𑁆𑀙𑁄 𑀔𑀻𑀭𑀧𑀓𑁄𑀯 𑀫𑀸𑀢𑀭𑀦𑁆”𑀢𑀺𑁇 (𑁭𑁭)_
