# 𑁭𑁇𑁭 𑀧𑀧𑀜𑁆𑀘𑀔𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀧𑀧𑀜𑁆𑀘𑀔𑀬𑀲𑀼𑀢𑁆𑀢

(67.)

635\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸𑀧𑀳𑀸𑀦𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

636\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀧𑀜𑁆𑀘𑀲𑀜𑁆𑀜𑀸𑀲𑀗𑁆𑀔𑀸𑀧𑀳𑀸𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

637\. _“𑀬𑀲𑁆𑀲 𑀧𑀧𑀜𑁆𑀘𑀸 𑀞𑀺𑀢𑀺 𑀘 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀦𑁆𑀤𑀸𑀦𑀁 𑀧𑀮𑀺𑀖𑀜𑁆𑀘 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁄;_  
_𑀢𑀁 𑀦𑀺𑀢𑁆𑀢𑀡𑁆𑀳𑀁 𑀫𑀼𑀦𑀺𑀁 𑀘𑀭𑀦𑁆𑀢𑀁,_  
_𑀦𑀸𑀯𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑀤𑁂𑀯𑀓𑁄𑀧𑀺 𑀮𑁄𑀓𑁄”𑀢𑀺𑁇 (𑁮𑁦)_
