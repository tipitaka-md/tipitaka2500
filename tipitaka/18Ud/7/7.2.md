# 𑁭𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀮𑀓𑀼𑀡𑁆𑀟𑀓𑀪𑀤𑁆𑀤𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀘𑀽𑀍𑀅𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀮𑀓𑀼𑀡𑁆𑀟𑀓𑀪𑀤𑁆𑀤𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(62.)

608\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀮𑀓𑀼𑀡𑁆𑀟𑀓𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀲𑁂𑀔𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑀺𑁇

609\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀮𑀓𑀼𑀡𑁆𑀟𑀓𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀲𑁂𑀔𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀦𑁆𑀢𑀁 𑀲𑀫𑀸𑀤𑀧𑁂𑀦𑁆𑀢𑀁 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀦𑁆𑀢𑀁𑁇

610\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

611\. _“𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀺 𑀯𑀝𑁆𑀝𑀁 𑀩𑁆𑀬𑀕𑀸 𑀦𑀺𑀭𑀸𑀲𑀁,_  
_𑀯𑀺𑀲𑀼𑀓𑁆𑀔𑀸 𑀲𑀭𑀺𑀢𑀸 𑀦 𑀲𑀦𑁆𑀤𑀢𑀺;_  
_𑀙𑀺𑀦𑁆𑀦𑀁 𑀯𑀝𑁆𑀝𑀁 𑀦 𑀯𑀢𑁆𑀢𑀢𑀺,_  
_𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁭𑁫)_  


---
