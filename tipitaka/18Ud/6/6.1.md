# 𑁬𑁇𑁧 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀚𑀘𑁆𑀘𑀦𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑁄𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀲𑀼𑀢𑁆𑀢

(51.)

457\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀡𑁆𑀳𑀸𑀳𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀲𑀻𑀤𑀦𑀁𑁇 𑀬𑁂𑀦 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇

458\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀆𑀤𑀸𑀬 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

459\. “𑀭𑀫𑀡𑀻𑀬𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀲𑀸𑀮𑀻; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀉𑀤𑁂𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀕𑁄𑀢𑀫𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁆𑀩𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀸𑀭𑀦𑁆𑀤𑀤𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 ( ) 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑁇 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸”𑀢𑀺𑁇

460\. 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂, 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂, 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁; 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀬𑀸𑀘𑀺—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁; 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺, 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

461\. “𑀭𑀫𑀡𑀻𑀬𑀸, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑁂𑀲𑀸𑀮𑀻; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀉𑀤𑁂𑀦𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀕𑁄𑀢𑀫𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀢𑁆𑀢𑀫𑁆𑀩𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀩𑀳𑀼𑀧𑀼𑀢𑁆𑀢𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀸𑀭𑀦𑁆𑀤𑀤𑀁 𑀘𑁂𑀢𑀺𑀬𑀁; 𑀭𑀫𑀡𑀻𑀬𑀁 𑀘𑀸𑀧𑀸𑀮𑀁 𑀘𑁂𑀢𑀺𑀬𑀁𑁇 𑀬𑀲𑁆𑀲 𑀓𑀲𑁆𑀲𑀘𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸, 𑀲𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀤𑀸 𑀪𑀸𑀯𑀺𑀢𑀸 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸 𑀬𑀸𑀦𑀻𑀓𑀢𑀸 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀧𑀭𑀺𑀘𑀺𑀢𑀸 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑁇 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀓𑀧𑁆𑀧𑀸𑀯𑀲𑁂𑀲𑀁 𑀯𑀸”𑀢𑀺𑁇

462\. 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑀸 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂, 𑀑𑀍𑀆𑀭𑀺𑀓𑁂 𑀑𑀪𑀸𑀲𑁂 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂, 𑀦𑀸𑀲𑀓𑁆𑀔𑀺 𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀺𑀢𑀼𑀁; 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀬𑀸𑀘𑀺—  “𑀢𑀺𑀝𑁆𑀞𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀧𑁆𑀧𑀁; 𑀢𑀺𑀝𑁆𑀞𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀓𑀧𑁆𑀧𑀁 𑀩𑀳𑀼𑀚𑀦𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀲𑀼𑀔𑀸𑀬 𑀮𑁄𑀓𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀅𑀢𑁆𑀣𑀸𑀬 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀦𑁆”𑀢𑀺, 𑀬𑀣𑀸 𑀢𑀁 𑀫𑀸𑀭𑁂𑀦 𑀧𑀭𑀺𑀬𑀼𑀝𑁆𑀞𑀺𑀢𑀘𑀺𑀢𑁆𑀢𑁄𑁇

463\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

464\. 𑀅𑀣 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸, 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤𑁂, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀫𑀸𑀭𑁄 𑀧𑀸𑀧𑀺𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

469\. 𑀪𑀸𑀲𑀺𑀢𑀸 𑀔𑁄 𑀧𑀦𑁂𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀯𑀸𑀘𑀸—  ‘𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀸𑀯 𑀫𑁂 𑀇𑀤𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀦 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀻𑀢𑀜𑁆𑀘 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀓𑀁 𑀩𑀸𑀳𑀼𑀚𑀜𑁆𑀜𑀁 𑀧𑀼𑀣𑀼𑀪𑀽𑀢𑀁 𑀬𑀸𑀯 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀁𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸; 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀢𑀼 𑀲𑀼𑀕𑀢𑁄; 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄”𑀢𑀺𑁇

470\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀫𑀸𑀭𑀁 𑀧𑀸𑀧𑀺𑀫𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀢𑁆𑀯𑀁, 𑀧𑀸𑀧𑀺𑀫, 𑀳𑁄𑀳𑀺𑁇 𑀦 𑀘𑀺𑀭𑀁 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀢𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀫𑀸𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

471\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀘𑀸𑀧𑀸𑀮𑁂 𑀘𑁂𑀢𑀺𑀬𑁂 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀑𑀲𑁆𑀲𑀚𑀺𑁇 𑀑𑀲𑁆𑀲𑀝𑁆𑀞𑁂 𑀘 𑀪𑀕𑀯𑀢𑀸 𑀆𑀬𑀼𑀲𑀗𑁆𑀔𑀸𑀭𑁂 𑀫𑀳𑀸𑀪𑀽𑀫𑀺𑀘𑀸𑀮𑁄 𑀅𑀳𑁄𑀲𑀺 𑀪𑀺𑀁𑀲𑀦𑀓𑁄 𑀮𑁄𑀫𑀳𑀁𑀲𑁄, 𑀤𑁂𑀯𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀺𑀬𑁄 𑀘 𑀨𑀮𑀺𑀁𑀲𑀼𑁇

472\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

473\. _“𑀢𑀼𑀮𑀫𑀢𑀼𑀮𑀜𑁆𑀘 𑀲𑀫𑁆𑀪𑀯𑀁,_  
_𑀪𑀯𑀲𑀗𑁆𑀔𑀸𑀭𑀫𑀯𑀲𑁆𑀲𑀚𑀺 𑀫𑀼𑀦𑀺;_  
_𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀭𑀢𑁄 𑀲𑀫𑀸𑀳𑀺𑀢𑁄,_  
_𑀅𑀪𑀺𑀦𑁆𑀤𑀺 𑀓𑀯𑀘𑀫𑀺𑀯𑀢𑁆𑀢𑀲𑀫𑁆𑀪𑀯𑀦𑁆”𑀢𑀺𑁇 (𑁬𑁨)_
