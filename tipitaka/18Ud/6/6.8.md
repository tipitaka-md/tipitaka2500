# 𑁬𑁇𑁮 𑀕𑀡𑀺𑀓𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀚𑀘𑁆𑀘𑀦𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀕𑀡𑀺𑀓𑀸𑀲𑀼𑀢𑁆𑀢

(58.)

581. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀤𑁆𑀯𑁂 𑀧𑀽𑀕𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀕𑀡𑀺𑀓𑀸𑀬 𑀲𑀸𑀭𑀢𑁆𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸; 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀸𑀡𑀻𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀮𑁂𑀟𑁆𑀟𑀽𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀤𑀡𑁆𑀟𑁂𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀣𑁂𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀦𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀁𑁇

582. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—

583. “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀸𑀚𑀕𑀳𑁂 𑀤𑁆𑀯𑁂 𑀧𑀽𑀕𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀕𑀡𑀺𑀓𑀸𑀬 𑀲𑀸𑀭𑀢𑁆𑀢𑀸 𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑀸; 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀸𑀡𑀻𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀮𑁂𑀟𑁆𑀟𑀽𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀤𑀡𑁆𑀟𑁂𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀣𑁂𑀳𑀺𑀧𑀺 𑀉𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀢𑀢𑁆𑀣 𑀫𑀭𑀡𑀫𑁆𑀧𑀺 𑀦𑀺𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀭𑀡𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺𑁇

584. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

585. “𑀬𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀁 𑀬𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀉𑀪𑀬𑀫𑁂𑀢𑀁 𑀭𑀚𑀸𑀦𑀼𑀓𑀺𑀡𑁆𑀡𑀁, 𑀆𑀢𑀼𑀭𑀲𑁆𑀲𑀸𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀢𑁄𑁇 𑀬𑁂 𑀘 𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀸𑀭𑀸 𑀲𑀻𑀮𑀩𑁆𑀩𑀢𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀭𑀸, 𑀅𑀬𑀫𑁂𑀓𑁄 𑀅𑀦𑁆𑀢𑁄𑁇 𑀬𑁂 𑀘 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄—  ‘𑀦𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑁂𑀲𑀼 𑀤𑁄𑀲𑁄’𑀢𑀺, 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀦𑁆𑀢𑁄𑁇 𑀇𑀘𑁆𑀘𑁂𑀢𑁂 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑀸 𑀓𑀝𑀲𑀺𑀯𑀟𑁆𑀠𑀦𑀸, 𑀓𑀝𑀲𑀺𑀬𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀯𑀟𑁆𑀠𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀢𑁂𑀢𑁂 𑀉𑀪𑁄 𑀅𑀦𑁆𑀢𑁂 𑀅𑀦𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀑𑀮𑀻𑀬𑀦𑁆𑀢𑀺 𑀏𑀓𑁂, 𑀅𑀢𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀏𑀓𑁂𑁇 𑀬𑁂 𑀘 𑀔𑁄 𑀢𑁂 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀬 𑀢𑀢𑁆𑀭 𑀘 𑀦𑀸𑀳𑁂𑀲𑀼𑀁, 𑀢𑁂𑀦 𑀘 𑀦𑀸𑀫𑀜𑁆𑀜𑀺𑀁𑀲𑀼, 𑀯𑀝𑁆𑀝𑀁 𑀢𑁂𑀲𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀜𑀸𑀧𑀦𑀸𑀬𑀸”𑀢𑀺𑁇
