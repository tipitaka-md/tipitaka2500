# 𑁬𑁇𑁧𑁦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀚𑀘𑁆𑀘𑀦𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(60.)

592\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

593\. “𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀢𑀸𑀯 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀲𑀓𑁆𑀓𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑀸 𑀫𑀸𑀦𑀺𑀢𑀸 𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀧𑀘𑀺𑀢𑀸 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀣 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀅𑀲𑀓𑁆𑀓𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀕𑀭𑀼𑀓𑀢𑀸 𑀅𑀫𑀸𑀦𑀺𑀢𑀸 𑀅𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀦𑀧𑀘𑀺𑀢𑀸 𑀦 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀪𑀕𑀯𑀸𑀬𑁂𑀯 𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀘𑀸”𑀢𑀺𑁇

594\. “𑀏𑀯𑀫𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀸𑀯𑀓𑀻𑀯𑀜𑁆𑀘, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀢𑀸𑀯 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀲𑀓𑁆𑀓𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑀸 𑀫𑀸𑀦𑀺𑀢𑀸 𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀧𑀘𑀺𑀢𑀸 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀅𑀣 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀅𑀲𑀓𑁆𑀓𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀕𑀭𑀼𑀓𑀢𑀸 𑀅𑀫𑀸𑀦𑀺𑀢𑀸 𑀅𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀦𑀧𑀘𑀺𑀢𑀸 𑀦 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀢𑀣𑀸𑀕𑀢𑁄𑀯 𑀤𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀘𑀸”𑀢𑀺𑁇

595\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

596\. _“𑀑𑀪𑀸𑀲𑀢𑀺 𑀢𑀸𑀯 𑀲𑁄 𑀓𑀺𑀫𑀺,_  
_𑀬𑀸𑀯 𑀦 𑀉𑀦𑁆𑀦𑀫𑀢𑁂 𑀧𑀪𑀗𑁆𑀓𑀭𑁄;_  
_𑀲 𑀯𑁂𑀭𑁄𑀘𑀦𑀫𑁆𑀳𑀺 𑀉𑀕𑁆𑀕𑀢𑁂,_  
_𑀳𑀢𑀧𑁆𑀧𑀪𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀘𑀸𑀧𑀺 𑀪𑀸𑀲𑀢𑀺𑁇 (𑁭𑁨)_  


597\. _𑀏𑀯𑀁 𑀑𑀪𑀸𑀲𑀺𑀢𑀫𑁂𑀯 𑀢𑀓𑁆𑀓𑀺𑀓𑀸𑀦𑀁,_  
_𑀬𑀸𑀯 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀸 𑀮𑁄𑀓𑁂 𑀦𑀼𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺;_  
_𑀦 𑀢𑀓𑁆𑀓𑀺𑀓𑀸 𑀲𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀦 𑀘𑀸𑀧𑀺 𑀲𑀸𑀯𑀓𑀸,_  
_𑀤𑀼𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀻 𑀦 𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀭𑁂”𑀢𑀺𑁇 (𑁭𑁩)_

599\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



600\. _𑀆𑀬𑀼𑀚𑀝𑀺𑀮𑀯𑁂𑀓𑁆𑀔𑀡𑀸,_  
_𑀢𑀬𑁄 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀼𑀪𑀽𑀢𑀺;_  
_𑀕𑀡𑀺𑀓𑀸 𑀉𑀧𑀸𑀢𑀺 𑀦𑀯𑀫𑁄,_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀘 𑀢𑁂 𑀤𑀲𑀸𑀢𑀺𑁇_
