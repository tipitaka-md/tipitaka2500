# 𑁬𑁇𑁯 𑀉𑀧𑀸𑀢𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀚𑀘𑁆𑀘𑀦𑁆𑀥𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀸𑀢𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺𑀲𑀼𑀢𑁆𑀢

(59.)

587\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀢𑀺𑀫𑀺𑀲𑀸𑀬𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁂𑀲𑀼 𑀛𑀸𑀬𑀫𑀸𑀦𑁂𑀲𑀼𑁇

588\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀅𑀥𑀺𑀧𑀸𑀢𑀓𑀸 𑀢𑁂𑀲𑀼 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁂𑀲𑀼 𑀆𑀧𑀸𑀢𑀧𑀭𑀺𑀧𑀸𑀢𑀁 𑀅𑀦𑀬𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀅𑀥𑀺𑀧𑀸𑀢𑀓𑁂 𑀢𑁂𑀲𑀼 𑀢𑁂𑀮𑀧𑁆𑀧𑀤𑀻𑀧𑁂𑀲𑀼 𑀆𑀧𑀸𑀢𑀧𑀭𑀺𑀧𑀸𑀢𑀁 𑀅𑀦𑀬𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁂, 𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁂, 𑀅𑀦𑀬𑀩𑁆𑀬𑀲𑀦𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁂𑁇

589\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

590\. _“𑀉𑀧𑀸𑀢𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀺 𑀦 𑀲𑀸𑀭𑀫𑁂𑀦𑁆𑀢𑀺,_  
_𑀦𑀯𑀁 𑀦𑀯𑀁 𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑁆𑀭𑀽𑀳𑀬𑀦𑁆𑀢𑀺;_  
_𑀧𑀢𑀦𑁆𑀢𑀺 𑀧𑀚𑁆𑀚𑁄𑀢𑀫𑀺𑀯𑀸𑀥𑀺𑀧𑀸𑀢𑀓𑀸,_  
_𑀤𑀺𑀝𑁆𑀞𑁂 𑀲𑀼𑀢𑁂 𑀇𑀢𑀺𑀳𑁂𑀓𑁂 𑀦𑀺𑀯𑀺𑀝𑁆𑀞𑀸”𑀢𑀺𑁇 (𑁭𑁧)_
