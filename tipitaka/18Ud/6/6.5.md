# 𑁬𑁇𑁫 𑀤𑀼𑀢𑀺𑀬𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀚𑀘𑁆𑀘𑀦𑁆𑀥𑀯𑀕𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(55.)

524. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀺 𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀸𑀦𑀸𑀔𑀦𑁆𑀢𑀺𑀓𑀸 𑀦𑀸𑀦𑀸𑀭𑀼𑀘𑀺𑀓𑀸 𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀲𑁆𑀲𑀬𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸𑁇

525. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧)

526. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁨)

527. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀲𑁆𑀲𑀢𑁄 𑀘 𑀅𑀲𑀲𑁆𑀲𑀢𑁄 𑀘 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁩)

528. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀦𑁂𑀯 𑀲𑀲𑁆𑀲𑀢𑁄 𑀦𑀸𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁪)

529. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀬𑀁𑀓𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁫)

530. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀧𑀭𑀁𑀓𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁬)

531. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀬𑀁𑀓𑀢𑁄 𑀘 𑀧𑀭𑀁𑀓𑀢𑁄 𑀘 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁭)

532. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀅𑀲𑀬𑀗𑁆𑀓𑀸𑀭𑁄 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑁄 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁮)

533. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀲𑁆𑀲𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁯)

534. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀅𑀲𑀲𑁆𑀲𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁦)

535. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀲𑁆𑀲𑀢𑀜𑁆𑀘 𑀅𑀲𑀲𑁆𑀲𑀢𑀜𑁆𑀘 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁧)

536. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀦𑁂𑀯 𑀲𑀲𑁆𑀲𑀢𑀁 𑀦𑀸𑀲𑀲𑁆𑀲𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁨)

537. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀬𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁩)

538. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀧𑀭𑀁𑀓𑀢𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁪)

539. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀲𑀬𑀁𑀓𑀢𑀜𑁆𑀘 𑀧𑀭𑀁𑀓𑀢𑀜𑁆𑀘 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁫)

540. 𑀲𑀦𑁆𑀢𑀺 𑀧𑀦𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  “𑀅𑀲𑀬𑀁𑀓𑀸𑀭𑀁 𑀅𑀧𑀭𑀗𑁆𑀓𑀸𑀭𑀁 𑀅𑀥𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁬)

541. 𑀢𑁂 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  “𑀏𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀦𑁂𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄; 𑀦𑁂𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀏𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄”𑀢𑀺𑁇

542. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—

543. “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀦𑀸𑀦𑀸𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀺 𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀦𑀸𑀦𑀸𑀔𑀦𑁆𑀢𑀺𑀓𑀸 𑀦𑀸𑀦𑀸𑀭𑀼𑀘𑀺𑀓𑀸 𑀦𑀸𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑀺𑀲𑁆𑀲𑀬𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸𑁇

544. 𑀲𑀦𑁆𑀢𑁂𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀏𑀯𑀁𑀯𑀸𑀤𑀺𑀦𑁄 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄—  ‘𑀲𑀲𑁆𑀲𑀢𑁄 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀮𑁄𑀓𑁄 𑀘, 𑀇𑀤𑀫𑁂𑀯 𑀲𑀘𑁆𑀘𑀁 𑀫𑁄𑀖𑀫𑀜𑁆𑀜𑀦𑁆’𑀢𑀺…𑀧𑁂…  𑀢𑁂 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  ‘𑀏𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀦𑁂𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄; 𑀦𑁂𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀏𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄’”𑀢𑀺𑁇

545. “𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀅𑀦𑁆𑀥𑀸 𑀅𑀘𑀓𑁆𑀔𑀼𑀓𑀸; 𑀅𑀢𑁆𑀣𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀦𑀢𑁆𑀣𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀥𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀢𑁆𑀣𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀅𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸 𑀓𑀮𑀳𑀚𑀸𑀢𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀫𑀼𑀔𑀲𑀢𑁆𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀼𑀤𑀦𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  ‘𑀏𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀦𑁂𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄; 𑀦𑁂𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀏𑀤𑀺𑀲𑁄 𑀥𑀫𑁆𑀫𑁄’”𑀢𑀺𑁇

546. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

547. _“𑀇𑀫𑁂𑀲𑀼 𑀓𑀺𑀭 𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀏𑀓𑁂 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀅𑀦𑁆𑀢𑀭𑀸𑀯 𑀯𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺,_  
_𑀅𑀧𑁆𑀧𑀢𑁆𑀯𑀸𑀯 𑀢𑀫𑁄𑀕𑀥𑀦𑁆”𑀢𑀺𑁇 (𑁬𑁬)_
