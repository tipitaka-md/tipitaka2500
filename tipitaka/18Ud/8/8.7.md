# 𑁮𑁇𑁭 𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑀲𑀼𑀢𑁆𑀢

(77.)

732. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁂𑀦 𑀧𑀘𑁆𑀙𑀸𑀲𑀫𑀡𑁂𑀦𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀤𑁆𑀯𑀺𑀥𑀸𑀧𑀣𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀦𑁆𑀣𑁄; 𑀇𑀫𑀺𑀦𑀸 𑀕𑀘𑁆𑀙𑀸𑀫𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀦𑀸𑀕𑀲𑀫𑀸𑀮, 𑀧𑀦𑁆𑀣𑁄; 𑀇𑀫𑀺𑀦𑀸 𑀕𑀘𑁆𑀙𑀸𑀫𑀸”𑀢𑀺𑁇

733. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀦𑁆𑀣𑁄; 𑀇𑀫𑀺𑀦𑀸 𑀕𑀘𑁆𑀙𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀦𑀸𑀕𑀲𑀫𑀸𑀮, 𑀧𑀦𑁆𑀣𑁄; 𑀇𑀫𑀺𑀦𑀸 𑀕𑀘𑁆𑀙𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀢𑀢𑁆𑀣𑁂𑀯 𑀙𑀫𑀸𑀬𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺—  “𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺𑁇

734. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑀲𑁆𑀲 𑀢𑁂𑀦 𑀧𑀦𑁆𑀣𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀘𑁄𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀘 𑀧𑀸𑀤𑁂𑀳𑀺 𑀘 𑀆𑀓𑁄𑀝𑁂𑀲𑀼𑀁 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀺𑀦𑁆𑀤𑀺𑀁𑀲𑀼 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀜𑁆𑀘 𑀯𑀺𑀧𑁆𑀨𑀸𑀮𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀪𑀺𑀦𑁆𑀦𑁂𑀦 𑀧𑀢𑁆𑀢𑁂𑀦 𑀯𑀺𑀧𑁆𑀨𑀸𑀮𑀺𑀢𑀸𑀬 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀬𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀸𑀕𑀲𑀫𑀸𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑁂𑀦 𑀧𑀦𑁆𑀣𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀘𑁄𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀘 𑀧𑀸𑀤𑁂𑀳𑀺 𑀘 𑀆𑀓𑁄𑀝𑁂𑀲𑀼𑀁, 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀺𑀦𑁆𑀤𑀺𑀁𑀲𑀼, 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀜𑁆𑀘 𑀯𑀺𑀧𑁆𑀨𑀸𑀮𑁂𑀲𑀼𑀦𑁆”𑀢𑀺𑁇

735. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

736. _“𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀭𑀫𑁂𑀓𑀢𑁄 𑀯𑀲𑀁,_  
_𑀫𑀺𑀲𑁆𑀲𑁄 𑀅𑀜𑁆𑀜𑀚𑀦𑁂𑀦 𑀯𑁂𑀤𑀕𑀽;_  
_𑀯𑀺𑀤𑁆𑀯𑀸 𑀧𑀚𑀳𑀸𑀢𑀺 𑀧𑀸𑀧𑀓𑀁,_  
_𑀓𑁄𑀜𑁆𑀘𑁄 𑀔𑀻𑀭𑀧𑀓𑁄𑀯 𑀦𑀺𑀦𑁆𑀦𑀕𑀦𑁆”𑀢𑀺𑁇 (𑁯𑁦)_
