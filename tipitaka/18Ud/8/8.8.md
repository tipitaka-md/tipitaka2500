# 𑁮𑁇𑁮 𑀯𑀺𑀲𑀸𑀔𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀯𑀺𑀲𑀸𑀔𑀸𑀲𑀼𑀢𑁆𑀢

(78.)

738\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀸𑀭𑀸𑀫𑁂 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀧𑀸𑀲𑀸𑀤𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀦𑀢𑁆𑀢𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀺𑀬𑀸 𑀫𑀦𑀸𑀧𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀸 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀸 𑀤𑀺𑀯𑀸 𑀤𑀺𑀯𑀲𑁆𑀲 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

739\. “𑀳𑀦𑁆𑀤 𑀓𑀼𑀢𑁄 𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀆𑀕𑀘𑁆𑀙𑀲𑀺 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀸 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀸 𑀇𑀥𑀽𑀧𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀸 𑀤𑀺𑀯𑀸 𑀤𑀺𑀯𑀲𑁆𑀲𑀸”𑀢𑀺? “𑀦𑀢𑁆𑀢𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀺𑀬𑀸 𑀫𑀦𑀸𑀧𑀸 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸𑁇 𑀢𑁂𑀦𑀸𑀳𑀁 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀸 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀸 𑀇𑀥𑀽𑀧𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀸 𑀤𑀺𑀯𑀸 𑀤𑀺𑀯𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 “𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀸𑀯𑀢𑀺𑀓𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀘 𑀦𑀢𑁆𑀢𑀸𑀭𑁄 𑀘𑀸”𑀢𑀺? “𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀸𑀯𑀢𑀺𑀓𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀘 𑀦𑀢𑁆𑀢𑀸𑀭𑁄 𑀘𑀸”𑀢𑀺𑁇

740\. “𑀓𑀻𑀯𑀩𑀳𑀼𑀓𑀸 𑀧𑀦, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑁂𑀯𑀲𑀺𑀓𑀁 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀤𑀲𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑁂𑀯𑀲𑀺𑀓𑀁 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺; 𑀦𑀯𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂…  𑀅𑀝𑁆𑀞𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂…  𑀲𑀢𑁆𑀢𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂…  𑀙𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂…  𑀧𑀜𑁆𑀘𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂…  𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂…  𑀢𑀻𑀡𑀺𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂…  𑀤𑁆𑀯𑁂𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑁂𑀯𑀲𑀺𑀓𑀁 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑁄𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀤𑁂𑀯𑀲𑀺𑀓𑀁 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀅𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀳𑀻”𑀢𑀺𑁇

741\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀅𑀧𑀺 𑀦𑀼 𑀢𑁆𑀯𑀁 𑀓𑀤𑀸𑀘𑀺 𑀓𑀭𑀳𑀘𑀺 𑀅𑀦𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀸 𑀯𑀸 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀅𑀦𑀮𑁆𑀮𑀓𑁂𑀲𑀸 𑀯𑀸”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀅𑀮𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀸𑀯 𑀩𑀳𑀼𑀓𑁂𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀘 𑀦𑀢𑁆𑀢𑀸𑀭𑁂𑀳𑀺 𑀘𑀸”𑀢𑀺𑁇

742\. “𑀬𑁂𑀲𑀁 𑀔𑁄, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀲𑀢𑀁 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀲𑀢𑀁 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀦𑀯𑀼𑀢𑀺 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀦𑀯𑀼𑀢𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀅𑀲𑀻𑀢𑀺 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀅𑀲𑀻𑀢𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀢𑀺 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀲𑀢𑁆𑀢𑀢𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀲𑀝𑁆𑀞𑀺 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀲𑀝𑁆𑀞𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀁 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀧𑀜𑁆𑀜𑀸𑀲𑀁 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀁 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀁 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺, 𑀬𑁂𑀲𑀁 𑀢𑀺𑀁𑀲𑀁 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀢𑀺𑀁𑀲𑀁 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀯𑀻𑀲𑀢𑀺 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀯𑀻𑀲𑀢𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺, 𑀬𑁂𑀲𑀁 𑀤𑀲 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀤𑀲 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀦𑀯 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀦𑀯 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀅𑀝𑁆𑀞 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀅𑀝𑁆𑀞 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀲𑀢𑁆𑀢 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀲𑀢𑁆𑀢 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀙 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀙 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀧𑀜𑁆𑀘 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀧𑀜𑁆𑀘 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀢𑀻𑀡𑀺 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀢𑀻𑀡𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀤𑁆𑀯𑁂 𑀧𑀺𑀬𑀸𑀦𑀺, 𑀤𑁆𑀯𑁂 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀺; 𑀬𑁂𑀲𑀁 𑀏𑀓𑀁 𑀧𑀺𑀬𑀁, 𑀏𑀓𑀁 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁; 𑀬𑁂𑀲𑀁 𑀦𑀢𑁆𑀣𑀺 𑀧𑀺𑀬𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀢𑁂𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀁, 𑀅𑀲𑁄𑀓𑀸 𑀢𑁂 𑀯𑀺𑀭𑀚𑀸 𑀅𑀦𑀼𑀧𑀸𑀬𑀸𑀲𑀸𑀢𑀺 𑀯𑀤𑀸𑀫𑀻”𑀢𑀺𑁇

743\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

744\. _“𑀬𑁂 𑀓𑁂𑀘𑀺 𑀲𑁄𑀓𑀸 𑀧𑀭𑀺𑀤𑁂𑀯𑀺𑀢𑀸 𑀯𑀸,_  
_𑀤𑀼𑀓𑁆𑀔𑀸 𑀯 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀫𑀦𑁂𑀓𑀭𑀽𑀧𑀸;_  
_𑀧𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑀧𑁆𑀧𑀪𑀯𑀦𑁆𑀢𑀺 𑀏𑀢𑁂,_  
_𑀧𑀺𑀬𑁂 𑀅𑀲𑀦𑁆𑀢𑁂 𑀦 𑀪𑀯𑀦𑁆𑀢𑀺 𑀏𑀢𑁂𑁇 (𑁯𑁧)_  


745\. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀢𑁂 𑀲𑀼𑀔𑀺𑀦𑁄 𑀯𑀻𑀢𑀲𑁄𑀓𑀸,_  
_𑀬𑁂𑀲𑀁 𑀧𑀺𑀬𑀁 𑀦𑀢𑁆𑀣𑀺 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂;_  
_𑀢𑀲𑁆𑀫𑀸 𑀅𑀲𑁄𑀓𑀁 𑀯𑀺𑀭𑀚𑀁 𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄,_  
_𑀧𑀺𑀬𑀁 𑀦 𑀓𑀬𑀺𑀭𑀸𑀣 𑀓𑀼𑀳𑀺𑀜𑁆𑀘𑀺 𑀮𑁄𑀓𑁂”𑀢𑀺𑁇 (𑁯𑁨)_
