# 𑁮𑁇𑁫 𑀘𑀼𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

### 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀘𑀼𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(75.)

677\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀧𑀸𑀯𑀸 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀧𑀸𑀯𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀫𑁆𑀩𑀯𑀦𑁂𑁇

678\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀫𑀮𑁆𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀸𑀯𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀧𑀸𑀯𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀬𑁆𑀳𑀁 𑀅𑀫𑁆𑀩𑀯𑀦𑁂”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑀁 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

679\. 𑀅𑀣 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀓𑁂 𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀳𑀽𑀢𑀜𑁆𑀘 𑀲𑀽𑀓𑀭𑀫𑀤𑁆𑀤𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

680\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀬𑁂𑀦 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀘𑀼𑀦𑁆𑀤𑀁 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀁 𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀲𑀽𑀓𑀭𑀫𑀤𑁆𑀤𑀯𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀢𑁂𑀦 𑀫𑀁 𑀧𑀭𑀺𑀯𑀺𑀲, 𑀬𑀁 𑀧𑀦𑀜𑁆𑀜𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀭𑀺𑀯𑀺𑀲𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀲𑀽𑀓𑀭𑀫𑀤𑁆𑀤𑀯𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀭𑀺𑀯𑀺𑀲𑀺; 𑀬𑀁 𑀧𑀦𑀜𑁆𑀜𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀭𑀺𑀯𑀺𑀲𑀺𑁇

681\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀘𑀼𑀦𑁆𑀤𑀁 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀁 𑀢𑁂, 𑀘𑀼𑀦𑁆𑀤, 𑀲𑀽𑀓𑀭𑀫𑀤𑁆𑀤𑀯𑀁 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀁 𑀢𑀁 𑀲𑁄𑀩𑁆𑀪𑁂 𑀦𑀺𑀔𑀡𑀸𑀳𑀺𑁇 𑀦𑀸𑀳𑀁 𑀢𑀁, 𑀘𑀼𑀦𑁆𑀤, 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀬𑀲𑁆𑀲 𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑀁 𑀅𑀳𑁄𑀲𑀺 𑀲𑀽𑀓𑀭𑀫𑀤𑁆𑀤𑀯𑀁 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀁 𑀢𑀁 𑀲𑁄𑀩𑁆𑀪𑁂 𑀦𑀺𑀔𑀡𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀘𑀼𑀦𑁆𑀤𑀁 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

682\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀲𑁆𑀲 𑀔𑀭𑁄 𑀆𑀩𑀸𑀥𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑁇 𑀮𑁄𑀳𑀺𑀢𑀧𑀓𑁆𑀔𑀦𑁆𑀤𑀺𑀓𑀸 𑀧𑀩𑀸𑀍𑀳𑀸 𑀯𑁂𑀤𑀦𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀫𑀸𑀭𑀡𑀦𑁆𑀢𑀺𑀓𑀸𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀅𑀯𑀺𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄𑁇

683\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀬𑀸𑀫𑀸𑀦𑀦𑁆𑀤, 𑀬𑁂𑀦 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇

684\. _𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀫𑁆𑀫𑀸𑀭𑀲𑁆𑀲𑀸𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀆𑀩𑀸𑀥𑀁 𑀲𑀫𑁆𑀨𑀼𑀲𑀻 𑀥𑀻𑀭𑁄,_  
_𑀧𑀩𑀸𑀍𑀳𑀁 𑀫𑀸𑀭𑀡𑀦𑁆𑀢𑀺𑀓𑀁𑁇_  


685\. _𑀪𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀲𑀽𑀓𑀭𑀫𑀤𑁆𑀤𑀯𑁂𑀦,_  
_𑀩𑁆𑀬𑀸𑀥𑀺𑀧𑁆𑀧𑀩𑀸𑀍𑀳𑁄 𑀉𑀤𑀧𑀸𑀤𑀺 𑀲𑀢𑁆𑀣𑀼𑀦𑁄;_  
_𑀯𑀺𑀭𑀺𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘,_  
_“𑀕𑀘𑁆𑀙𑀸𑀫𑀳𑀁 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀁 𑀦𑀕𑀭𑀦𑁆”𑀢𑀺𑁇_  


686\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀺; 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀆𑀳𑀭; 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

687\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀓𑀝𑀲𑀢𑀸𑀦𑀺 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺𑁇 𑀢𑀁 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀉𑀤𑀓𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀮𑀼𑀍𑀇𑀢𑀁 𑀆𑀯𑀺𑀮𑀁 𑀲𑀦𑁆𑀤𑀢𑀺𑁇 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀓𑀼𑀥𑀸 𑀦𑀤𑀻 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑀸 𑀲𑀸𑀢𑁄𑀤𑀓𑀸 𑀲𑀻𑀢𑁄𑀤𑀓𑀸 𑀲𑁂𑀢𑁄𑀤𑀓𑀸 𑀲𑀼𑀧𑀢𑀺𑀢𑁆𑀣𑀸 𑀭𑀫𑀡𑀻𑀬𑀸𑁇 𑀏𑀢𑁆𑀣 𑀪𑀕𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀜𑁆𑀘 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀲𑀻𑀢𑀻𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

688\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀆𑀳𑀭; 𑀧𑀺𑀧𑀸𑀲𑀺𑀢𑁄𑀲𑁆𑀫𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸 𑀦𑀤𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀦𑀤𑀻 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀮𑀼𑀍𑀇𑀢𑀸 𑀆𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀅𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀅𑀦𑀸𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀢𑀺𑁇

689\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀸𑁇 𑀅𑀬𑀜𑁆𑀳𑀺 𑀲𑀸 𑀦𑀤𑀻 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀮𑀼𑀍𑀇𑀢𑀸 𑀆𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸 𑀫𑀬𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀅𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀅𑀦𑀸𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀢𑀻”𑀢𑀺𑁇 𑀧𑀢𑁆𑀢𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀸𑁇 𑀅𑀬𑀜𑁆𑀳𑀺 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀤𑀻 𑀘𑀓𑁆𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀧𑀭𑀺𑀢𑁆𑀢𑀸 𑀮𑀼𑀍𑀇𑀢𑀸 𑀆𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸 𑀫𑀬𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂 𑀅𑀘𑁆𑀙𑀸 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀅𑀦𑀸𑀯𑀺𑀮𑀸 𑀲𑀦𑁆𑀤𑀢𑀺𑁇 𑀧𑀺𑀯𑀢𑀼 𑀪𑀕𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀁, 𑀧𑀺𑀯𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀧𑀸𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇

690\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀁 𑀅𑀧𑀸𑀬𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦 𑀓𑀼𑀓𑀼𑀥𑀸 𑀦𑀤𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀓𑀼𑀥𑀁 𑀦𑀤𑀺𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑁆𑀳𑀢𑁆𑀯𑀸 𑀘 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘 𑀧𑀘𑁆𑀘𑀼𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀘𑀼𑀦𑁆𑀤𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀗𑁆𑀖 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀘𑀼𑀦𑁆𑀤𑀓, 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀺; 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑀲𑁆𑀫𑀺, 𑀘𑀼𑀦𑁆𑀤𑀓, 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

691\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘𑀼𑀦𑁆𑀤𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀻𑀳𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺 𑀧𑀸𑀤𑁂 𑀧𑀸𑀤𑀁 𑀅𑀘𑁆𑀘𑀸𑀥𑀸𑀬 𑀲𑀢𑁄 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑁄 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀜𑁆𑀜𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀦 𑀘𑀼𑀦𑁆𑀤𑀓𑁄 𑀢𑀢𑁆𑀣𑁂𑀯 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

692\. _𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑀤𑀺𑀓𑀁 𑀓𑀼𑀓𑀼𑀥𑀁,_  
_𑀅𑀘𑁆𑀙𑁄𑀤𑀓𑀁 𑀲𑀸𑀢𑀼𑀤𑀓𑀁 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀁;_  
_𑀑𑀕𑀸𑀳𑀺 𑀲𑀢𑁆𑀣𑀸 𑀲𑀼𑀓𑀺𑀮𑀦𑁆𑀢𑀭𑀽𑀧𑁄,_  
_𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀫𑁄𑀥 𑀮𑁄𑀓𑁂𑁇_  


693\. _𑀦𑁆𑀳𑀢𑁆𑀯𑀸 𑀘 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀘𑀼𑀤𑀢𑀸𑀭𑀺 𑀲𑀢𑁆𑀣𑀸,_  
_𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀕𑀡𑀲𑁆𑀲 𑀫𑀚𑁆𑀛𑁂;_  
_𑀲𑀢𑁆𑀣𑀸 𑀧𑀯𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀸 𑀇𑀥 𑀥𑀫𑁆𑀫𑁂,_  
_𑀉𑀧𑀸𑀕𑀫𑀺 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀫𑀳𑁂𑀲𑀺;_  
_𑀆𑀫𑀦𑁆𑀢𑀬𑀺 𑀘𑀼𑀦𑁆𑀤𑀓𑀁 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀁,_  
_“𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀦𑁆𑀣𑀭 𑀫𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀁”𑁇_  


694\. _𑀲𑁄 𑀘𑁄𑀤𑀺𑀢𑁄 𑀪𑀸𑀯𑀺𑀢𑀢𑁆𑀢𑁂𑀦 𑀘𑀼𑀦𑁆𑀤𑁄,_  
_𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀦𑁆𑀣𑀭𑀺 𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯;_  
_𑀦𑀺𑀧𑀚𑁆𑀚𑀺 𑀲𑀢𑁆𑀣𑀸 𑀲𑀼𑀓𑀺𑀮𑀦𑁆𑀢𑀭𑀽𑀧𑁄,_  
_𑀘𑀼𑀦𑁆𑀤𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀧𑀫𑀼𑀔𑁂 𑀦𑀺𑀲𑀻𑀤𑀻𑀢𑀺𑁇_  


695\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀺𑀬𑀸 𑀔𑁄 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑁄𑀘𑀺 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑀁 𑀉𑀧𑀤𑀳𑁂𑀬𑁆𑀬—  ‘𑀢𑀲𑁆𑀲 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀘𑀼𑀦𑁆𑀤, 𑀅𑀮𑀸𑀪𑀸, 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄’𑀢𑀺𑁇 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲𑀸𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀩𑁆𑀩𑁄—

696\. 𑀢𑀲𑁆𑀲 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀘𑀼𑀦𑁆𑀤, 𑀮𑀸𑀪𑀸, 𑀢𑀲𑁆𑀲 𑀢𑁂 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀬𑀲𑁆𑀲 𑀢𑁂 𑀢𑀣𑀸𑀕𑀢𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄𑁇 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀫𑁂𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀘𑀼𑀦𑁆𑀤, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀢𑀁, 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁—  𑀤𑁆𑀯𑁂𑀫𑁂 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀸 𑀲𑀫𑀲𑀫𑀨𑀮𑀸 𑀲𑀫𑀲𑀫𑀯𑀺𑀧𑀸𑀓𑀸 𑀅𑀢𑀺𑀯𑀺𑀬 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀳𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀢𑀭𑀸 𑀘 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀢𑀭𑀸 𑀘𑁇 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑁂? 𑀬𑀜𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀁 𑀅𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀬 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀥𑀸𑀢𑀼𑀬𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀢𑀺𑁇 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀸 𑀲𑀫𑀲𑀫𑀨𑀮𑀸 𑀲𑀫𑀲𑀫𑀯𑀺𑀧𑀸𑀓𑀸 𑀅𑀢𑀺𑀯𑀺𑀬 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀳𑀺 𑀫𑀳𑀧𑁆𑀨𑀮𑀢𑀭𑀸 𑀘 𑀫𑀳𑀸𑀦𑀺𑀲𑀁𑀲𑀢𑀭𑀸 𑀘𑁇

697\. ‘𑀆𑀬𑀼𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀯𑀡𑁆𑀡𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀲𑀼𑀔𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀲𑀕𑁆𑀕𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀬𑀲𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀁, 𑀆𑀥𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀘𑀼𑀦𑁆𑀤𑁂𑀦 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀢𑀦𑁆’𑀢𑀺𑁇 𑀘𑀼𑀦𑁆𑀤𑀲𑁆𑀲𑀸𑀦𑀦𑁆𑀤, 𑀓𑀫𑁆𑀫𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

698\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

699\. _“𑀤𑀤𑀢𑁄 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀯𑀟𑁆𑀠𑀢𑀺,_  
_𑀲𑀁𑀬𑀫𑀢𑁄 𑀯𑁂𑀭𑀁 𑀦 𑀘𑀻𑀬𑀢𑀺;_  
_𑀓𑀼𑀲𑀮𑁄 𑀘 𑀚𑀳𑀸𑀢𑀺 𑀧𑀸𑀧𑀓𑀁,_  
_𑀭𑀸𑀕𑀤𑁄𑀲𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑀸 𑀲𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄”𑀢𑀺𑁇 (𑁮𑁫)_
