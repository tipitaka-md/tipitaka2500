# 𑁮𑁇𑁯 𑀧𑀞𑀫𑀤𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

* 𑀧𑀸𑀝𑀮𑀺𑀕𑀸𑀫𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀤𑀩𑁆𑀩𑀲𑀼𑀢𑁆𑀢

(79.)

747\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑀩𑁆𑀩𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑀩𑁆𑀩𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀓𑀸𑀮𑁄 𑀫𑁂 𑀤𑀸𑀦𑀺, 𑀲𑀼𑀕𑀢𑀸”𑀢𑀺𑁇 “𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀢𑁆𑀯𑀁, 𑀤𑀩𑁆𑀩, 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇

748\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀤𑀩𑁆𑀩𑁄 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀸𑀬𑀺𑁇

749\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀩𑁆𑀩𑀲𑁆𑀲 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀲𑁆𑀲 𑀛𑀸𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑀲𑁆𑀲 𑀦𑁂𑀯 𑀙𑀸𑀭𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀢𑁆𑀣 𑀦 𑀫𑀲𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀧𑁆𑀧𑀺𑀲𑁆𑀲 𑀯𑀸 𑀢𑁂𑀮𑀲𑁆𑀲 𑀯𑀸 𑀛𑀸𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑀲𑁆𑀲 𑀦𑁂𑀯 𑀙𑀸𑀭𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺 𑀦 𑀫𑀲𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀤𑀩𑁆𑀩𑀲𑁆𑀲 𑀫𑀮𑁆𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀲𑁆𑀲 𑀛𑀸𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀟𑀬𑁆𑀳𑀫𑀸𑀦𑀲𑁆𑀲 𑀦𑁂𑀯 𑀙𑀸𑀭𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀢𑁆𑀣 𑀦 𑀫𑀲𑀻𑀢𑀺𑁇

750\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

751\. _“𑀅𑀪𑁂𑀤𑀺 𑀓𑀸𑀬𑁄 𑀦𑀺𑀭𑁄𑀥𑀺 𑀲𑀜𑁆𑀜𑀸,_  
_𑀯𑁂𑀤𑀦𑀸 𑀲𑀻𑀢𑀺𑀪𑀯𑀺𑀁𑀲𑀼 𑀲𑀩𑁆𑀩𑀸;_  
_𑀯𑀽𑀧𑀲𑀫𑀺𑀁𑀲𑀼 𑀲𑀗𑁆𑀔𑀸𑀭𑀸,_  
_𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀅𑀢𑁆𑀣𑀫𑀸𑀕𑀫𑀸”𑀢𑀺𑁇 (𑁯𑁩)_
