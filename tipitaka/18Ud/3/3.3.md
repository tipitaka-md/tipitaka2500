# 𑁩𑁇𑁩 𑀬𑀲𑁄𑀚𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀬𑀲𑁄𑀚𑀲𑀼𑀢𑁆𑀢

(23.)

177\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀬𑀲𑁄𑀚𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀢𑁂𑀥 𑀔𑁄 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑁂𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑀸𑀫𑀬𑀫𑀸𑀦𑀸 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸 𑀅𑀳𑁂𑀲𑀼𑀁𑁇

178\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑁂 𑀧𑀦𑁂𑀢𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸 𑀓𑁂𑀯𑀝𑁆𑀝𑀸 𑀫𑀜𑁆𑀜𑁂 𑀫𑀘𑁆𑀙𑀯𑀺𑀮𑁄𑀧𑁂”𑀢𑀺? “𑀏𑀢𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑁄𑀚𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀢𑁂𑀢𑁂 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑁂𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑀸𑀫𑀬𑀫𑀸𑀦𑀸 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦𑀳𑀸𑀦𑀦𑁆𑀤, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇

179\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

180\. “𑀓𑀺𑀁 𑀦𑀼 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸, 𑀓𑁂𑀯𑀝𑁆𑀝𑀸 𑀫𑀜𑁆𑀜𑁂 𑀫𑀘𑁆𑀙𑀯𑀺𑀮𑁄𑀧𑁂”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄𑀚𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀢𑁂𑀫𑁂 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑁂𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀧𑀜𑁆𑀜𑀸𑀧𑀬𑀫𑀸𑀦𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀧𑀝𑀺𑀲𑀸𑀫𑀬𑀫𑀸𑀦𑀸 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸”𑀢𑀺𑁇 “𑀕𑀘𑁆𑀙𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀸𑀫𑁂𑀫𑀺 𑀯𑁄; 𑀦 𑀯𑁄 𑀫𑀫 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

181\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀯𑀚𑁆𑀚𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀯𑀚𑁆𑀚𑀻𑀲𑀼 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑀸 𑀬𑁂𑀦 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸 𑀦𑀤𑀻 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀧𑀡𑁆𑀡𑀓𑀼𑀝𑀺𑀬𑁄 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇

182\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄𑀚𑁄 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀡𑀸𑀫𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑁂𑀦 𑀳𑀺𑀢𑁂𑀲𑀺𑀦𑀸, 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁂𑀦 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀣𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀧𑁆𑀧𑁂𑀫 𑀬𑀣𑀸 𑀦𑁄 𑀯𑀺𑀳𑀭𑀢𑀁 𑀪𑀕𑀯𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀬𑀲𑁄𑀚𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑀸 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀸 𑀢𑁂𑀦𑁂𑀯𑀦𑁆𑀢𑀭𑀯𑀲𑁆𑀲𑁂𑀦 𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑀺𑀲𑁆𑀲𑁄 𑀯𑀺𑀚𑁆𑀚𑀸 𑀲𑀘𑁆𑀙𑀸𑀓𑀁𑀲𑀼𑁇

183\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇

184\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀆𑀮𑁄𑀓𑀚𑀸𑀢𑀸 𑀯𑀺𑀬 𑀫𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀲𑀸 𑀤𑀺𑀲𑀸, 𑀑𑀪𑀸𑀲𑀚𑀸𑀢𑀸 𑀯𑀺𑀬 𑀫𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀲𑀸 𑀤𑀺𑀲𑀸; 𑀬𑀲𑁆𑀲𑀁 𑀤𑀺𑀲𑀸𑀬𑀁 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀕𑀦𑁆𑀢𑀼𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀽𑀮𑀸𑀲𑀺 𑀫𑁂 𑀫𑀦𑀲𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁—  ‘𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀺, 𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑁄’”𑀢𑀺𑁇

185\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁂𑀦 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀺, 𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑁄’”𑀢𑀺𑁇

186\. “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁—  𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀼𑀭𑀢𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀺, 𑀲𑀢𑁆𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑁄”𑀢𑀺𑁇

187\. “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁—  𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀬 𑀦𑀤𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑀸 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁂𑀲𑀼𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀆𑀦𑁂𑀜𑁆𑀚𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀢𑀫𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀏𑀢𑀭𑀳𑀺 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀆𑀦𑁂𑀜𑁆𑀚𑁂𑀦 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀏𑀢𑀭𑀳𑀺 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇 𑀲𑀩𑁆𑀩𑁂𑀯 𑀆𑀦𑁂𑀜𑁆𑀚𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇

188\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸, 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀧𑀞𑀫𑁂 𑀬𑀸𑀫𑁂, 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀢𑁆𑀢𑀺; 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀧𑀞𑀫𑁄 𑀬𑀸𑀫𑁄; 𑀘𑀺𑀭𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽; 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

189\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸, 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑁂 𑀬𑀸𑀫𑁂, 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀢𑁆𑀢𑀺; 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀫𑀚𑁆𑀛𑀺𑀫𑁄 𑀬𑀸𑀫𑁄; 𑀘𑀺𑀭𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽; 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

190\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸, 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂, 𑀉𑀤𑁆𑀥𑀲𑁆𑀢𑁂 𑀅𑀭𑀼𑀡𑁂, 𑀦𑀦𑁆𑀤𑀺𑀫𑀼𑀔𑀺𑀬𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀢𑁆𑀢𑀺; 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀬𑀸𑀫𑁄; 𑀉𑀤𑁆𑀥𑀲𑁆𑀢𑁄 𑀅𑀭𑀼𑀡𑁄; 𑀦𑀦𑁆𑀤𑀺𑀫𑀼𑀔𑀻 𑀭𑀢𑁆𑀢𑀺; 𑀘𑀺𑀭𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽; 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀻”𑀢𑀺𑁇

191\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀏𑀢𑁆𑀢𑀓𑀫𑁆𑀧𑀺 𑀢𑁂 𑀦𑀧𑁆𑀧𑀝𑀺𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀅𑀳𑀜𑁆𑀘, 𑀆𑀦𑀦𑁆𑀤, 𑀇𑀫𑀸𑀦𑀺 𑀘 𑀧𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑁂𑀯 𑀆𑀦𑁂𑀜𑁆𑀚𑀲𑀫𑀸𑀥𑀺𑀦𑀸 𑀦𑀺𑀲𑀻𑀤𑀺𑀫𑁆𑀳𑀸”𑀢𑀺𑁇

192\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

193\. _“𑀬𑀲𑁆𑀲 𑀚𑀺𑀢𑁄 𑀓𑀸𑀫𑀓𑀡𑁆𑀟𑀓𑁄,_  
_𑀅𑀓𑁆𑀓𑁄𑀲𑁄 𑀘 𑀯𑀥𑁄 𑀘 𑀩𑀦𑁆𑀥𑀦𑀜𑁆𑀘;_  
_𑀧𑀩𑁆𑀩𑀢𑁄𑀯 𑀲𑁄 𑀞𑀺𑀢𑁄 𑀅𑀦𑁂𑀚𑁄,_  
_𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑁂𑀲𑀼 𑀦 𑀯𑁂𑀥𑀢𑀻 𑀲 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺𑁇 (𑁨𑁭)_
