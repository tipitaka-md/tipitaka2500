# 𑁩𑁇𑁨 𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀦𑀦𑁆𑀤𑀲𑀼𑀢𑁆𑀢

(22.)

158\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀢𑀸 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀫𑀸𑀭𑁄𑀘𑁂𑀢𑀺—  “𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀺; 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀥𑀸𑀭𑁂𑀢𑀼𑀁, 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

159\. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀲𑁆𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀢𑀸 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀫𑀸𑀭𑁄𑀘𑁂𑀢𑀺—  ‘𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀺, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀥𑀸𑀭𑁂𑀢𑀼𑀁, 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇

160\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀦𑀦𑁆𑀤, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀦𑀦𑁆𑀤, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇

161\. “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

162\. “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀫𑀸𑀭𑁄𑀘𑁂𑀲𑀺—  ‘𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀺, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀥𑀸𑀭𑁂𑀢𑀼𑀁, 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇

163\. “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤, 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀲𑀺, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀲𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀥𑀸𑀭𑁂𑀢𑀼𑀁, 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? “𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀻 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀚𑀦𑀧𑀤𑀓𑀮𑁆𑀬𑀸𑀡𑀻 𑀖𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀟𑁆𑀠𑀼𑀮𑁆𑀮𑀺𑀔𑀺𑀢𑁂𑀳𑀺 𑀓𑁂𑀲𑁂𑀳𑀺 𑀅𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀢𑀼𑀯𑀝𑀁 𑀔𑁄, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀲𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀫𑀸𑀦𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀸𑀫𑀺, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀥𑀸𑀭𑁂𑀢𑀼𑀁, 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

164\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀩𑀸𑀳𑀸𑀬𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁—  𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇

165\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀺 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀆𑀕𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀧𑀲𑁆𑀲𑀲𑀺 𑀦𑁄 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤, 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀺 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀻”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇

166\. “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀦𑀦𑁆𑀤, 𑀓𑀢𑀫𑀸 𑀦𑀼 𑀔𑁄 𑀅𑀪𑀺𑀭𑀽𑀧𑀢𑀭𑀸 𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀢𑀭𑀸 𑀯𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀢𑀭𑀸 𑀯𑀸, 𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀻 𑀯𑀸 𑀚𑀦𑀧𑀤𑀓𑀮𑁆𑀬𑀸𑀡𑀻, 𑀇𑀫𑀸𑀦𑀺 𑀯𑀸 𑀧𑀜𑁆𑀘 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀺 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀻”𑀢𑀺? “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀮𑀼𑀝𑁆𑀞𑀫𑀓𑁆𑀓𑀝𑀻 𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀓𑀺𑀬𑀸𑀦𑀻 𑀚𑀦𑀧𑀤𑀓𑀮𑁆𑀬𑀸𑀡𑀻 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀁 𑀉𑀧𑀦𑀺𑀥𑀸𑀬 𑀲𑀗𑁆𑀔𑁆𑀬𑀫𑁆𑀧𑀺 𑀦𑁄𑀧𑁂𑀢𑀺 𑀓𑀮𑀪𑀸𑀕𑀫𑁆𑀧𑀺 𑀦𑁄𑀧𑁂𑀢𑀺 𑀉𑀧𑀦𑀺𑀥𑀺𑀫𑁆𑀧𑀺 𑀦𑁄𑀧𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀭𑀽𑀧𑀢𑀭𑀸𑀦𑀺 𑀘𑁂𑀯 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀢𑀭𑀸𑀦𑀺 𑀘 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀢𑀭𑀸𑀦𑀺 𑀘𑀸”𑀢𑀺𑁇

167\. “𑀅𑀪𑀺𑀭𑀫, 𑀦𑀦𑁆𑀤, 𑀅𑀪𑀺𑀭𑀫, 𑀦𑀦𑁆𑀤𑁇 𑀅𑀳𑀁 𑀢𑁂 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀘𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀁, 𑀅𑀪𑀺𑀭𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂”𑀢𑀺𑁇

168\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀩𑀸𑀳𑀸𑀬𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁—  𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀚𑁂𑀢𑀯𑀦𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇

169\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀺𑀭 𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀢𑀸 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀘𑁆𑀙𑀭𑀸𑀦𑀁 𑀳𑁂𑀢𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺; 𑀪𑀕𑀯𑀸 𑀓𑀺𑀭𑀲𑁆𑀲 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

170\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀪𑀢𑀓𑀯𑀸𑀤𑁂𑀦 𑀘 𑀉𑀧𑀓𑁆𑀓𑀺𑀢𑀓𑀯𑀸𑀤𑁂𑀦 𑀘 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀦𑁆𑀢𑀺—  “𑀪𑀢𑀓𑁄 𑀓𑀺𑀭𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀉𑀧𑀓𑁆𑀓𑀺𑀢𑀓𑁄 𑀓𑀺𑀭𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀅𑀘𑁆𑀙𑀭𑀸𑀦𑀁 𑀳𑁂𑀢𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺; 𑀪𑀕𑀯𑀸 𑀓𑀺𑀭𑀲𑁆𑀲 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀦𑁆”𑀢𑀺𑁇

171\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀲𑀳𑀸𑀬𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀢𑀓𑀯𑀸𑀤𑁂𑀦 𑀘 𑀉𑀧𑀓𑁆𑀓𑀺𑀢𑀓𑀯𑀸𑀤𑁂𑀦 𑀘 𑀅𑀝𑁆𑀝𑀻𑀬𑀫𑀸𑀦𑁄 𑀳𑀭𑀸𑀬𑀫𑀸𑀦𑁄 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄 𑀏𑀓𑁄 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀆𑀢𑀸𑀧𑀻 𑀧𑀳𑀺𑀢𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯—  𑀬𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀢𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀁—  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 “𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸”𑀢𑀺 𑀅𑀩𑁆𑀪𑀜𑁆𑀜𑀸𑀲𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀅𑀭𑀳𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇

172\. 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀲𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀬𑀲𑁆𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀢𑀸 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄𑀧𑀺 𑀔𑁄 𑀜𑀸𑀡𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀦𑀦𑁆𑀤𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻”𑀢𑀺𑁇

173\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀅𑀘𑁆𑀙𑀭𑀸𑀲𑀢𑀸𑀦𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬 𑀓𑀓𑀼𑀝𑀧𑀸𑀤𑀸𑀦𑀁, 𑀫𑀼𑀜𑁆𑀘𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀯𑀸”𑀢𑀺𑁇 “𑀫𑀬𑀸𑀧𑀺 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀦𑀦𑁆𑀤, 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀯𑀺𑀤𑀺𑀢𑁄—  ‘𑀦𑀦𑁆𑀤𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻’𑀢𑀺𑁇 𑀤𑁂𑀯𑀢𑀸𑀧𑀺 𑀫𑁂 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  ‘𑀆𑀬𑀲𑁆𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀢𑀸 𑀫𑀸𑀢𑀼𑀘𑁆𑀙𑀸𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀲𑀯𑀸𑀦𑀁 𑀔𑀬𑀸 𑀅𑀦𑀸𑀲𑀯𑀁 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀧𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀁 𑀤𑀺𑀝𑁆𑀞𑁂𑀯 𑀥𑀫𑁆𑀫𑁂 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚 𑀯𑀺𑀳𑀭𑀢𑀻’𑀢𑀺𑁇 𑀬𑀤𑁂𑀯 𑀔𑁄 𑀢𑁂, 𑀦𑀦𑁆𑀤, 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁, 𑀅𑀣𑀸𑀳𑀁 𑀫𑀼𑀢𑁆𑀢𑁄 𑀏𑀢𑀲𑁆𑀫𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀯𑀸”𑀢𑀺𑁇

174\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

175\. _“𑀬𑀲𑁆𑀲 𑀦𑀺𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑁄 𑀧𑀗𑁆𑀓𑁄,_  
_𑀫𑀤𑁆𑀤𑀺𑀢𑁄 𑀓𑀸𑀫𑀓𑀡𑁆𑀟𑀓𑁄;_  
_𑀫𑁄𑀳𑀓𑁆𑀔𑀬𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑁂𑀲𑀼 𑀦 𑀯𑁂𑀥𑀢𑀻 𑀲 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺𑁇 (𑁨𑁬)_
