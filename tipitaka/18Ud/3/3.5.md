# 𑁩𑁇𑁫 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(25.)

199\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀓𑀸𑀬𑀕𑀢𑀸𑀬 𑀲𑀢𑀺𑀬𑀸 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀸𑀬𑁇

200\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

201\. _“𑀲𑀢𑀺 𑀓𑀸𑀬𑀕𑀢𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸,_  
_𑀙𑀲𑀼 𑀨𑀲𑁆𑀲𑀸𑀬𑀢𑀦𑁂𑀲𑀼 𑀲𑀁𑀯𑀼𑀢𑁄;_  
_𑀲𑀢𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀫𑀸𑀳𑀺𑀢𑁄,_  
_𑀚𑀜𑁆𑀜𑀸 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀫𑀢𑁆𑀢𑀦𑁄”𑀢𑀺𑁇 (𑁨𑁯)_  


---
