# 𑁩𑁇𑁮 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀲𑀼𑀢𑁆𑀢

(28.)

217\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀭𑁂𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀏 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀅𑀬𑀫𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀉𑀤𑀧𑀸𑀤𑀺—

218\. “𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑁂 𑀲𑁄𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑁂 𑀖𑀸𑀬𑀺𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑁂 𑀲𑀸𑀬𑀺𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀨𑀼𑀲𑀺𑀢𑀼𑀁𑁇 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀓𑁆𑀓𑀢𑁄 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀸𑀯𑀼𑀲𑁄, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸 𑀳𑁄𑀫𑁇 𑀫𑀬𑀫𑁆𑀧𑀺 𑀮𑀘𑁆𑀙𑀸𑀫 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀮𑀘𑁆𑀙𑀸𑀫 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑁂 𑀲𑁄𑀢𑀼𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀮𑀘𑁆𑀙𑀸𑀫 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑁂 𑀖𑀸𑀬𑀺𑀢𑀼𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀮𑀘𑁆𑀙𑀸𑀫 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑁂 𑀲𑀸𑀬𑀺𑀢𑀼𑀁, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀮𑀘𑁆𑀙𑀸𑀫 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀨𑀼𑀲𑀺𑀢𑀼𑀁; 𑀫𑀬𑀫𑁆𑀧𑀺 𑀲𑀓𑁆𑀓𑀢𑀸 𑀕𑀭𑀼𑀓𑀢𑀸 𑀫𑀸𑀦𑀺𑀢𑀸 𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀧𑀘𑀺𑀢𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘𑀭𑀳𑀺 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸𑁇

219\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦 𑀓𑀭𑁂𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀑 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀸𑀬 𑀦𑀼𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸, 𑀓𑀸 𑀘 𑀧𑀦 𑀯𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸”𑀢𑀺?

220\. “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀓𑀭𑁂𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀏 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀅𑀬𑀫𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀉𑀤𑀧𑀸𑀤𑀺—

221\. ‘𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁄 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀲𑁄𑀢𑁂𑀦 𑀲𑀤𑁆𑀤𑁂 𑀲𑁄𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀖𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀥𑁂 𑀖𑀸𑀬𑀺𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀭𑀲𑁂 𑀲𑀸𑀬𑀺𑀢𑀼𑀁, 𑀮𑀪𑀢𑀺 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀨𑀼𑀲𑀺𑀢𑀼𑀁𑁇 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀓𑁆𑀓𑀢𑁄 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀸𑀯𑀼𑀲𑁄, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸 𑀳𑁄𑀫𑁇 𑀫𑀬𑀫𑁆𑀧𑀺 𑀮𑀘𑁆𑀙𑀸𑀫 𑀓𑀸𑀮𑁂𑀦 𑀓𑀸𑀮𑀁 𑀫𑀦𑀸𑀧𑀺𑀓𑁂 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀭𑀽𑀧𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁…𑀧𑁂…  𑀓𑀸𑀬𑁂𑀦 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂 𑀨𑀼𑀲𑀺𑀢𑀼𑀁𑁇 𑀫𑀬𑀫𑁆𑀧𑀺 𑀲𑀓𑁆𑀓𑀢𑀸 𑀕𑀭𑀼𑀓𑀢𑀸 𑀫𑀸𑀦𑀺𑀢𑀸 𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀧𑀘𑀺𑀢𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸, 𑀅𑀣 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇

222\. “𑀦 𑀔𑁆𑀯𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸𑀦𑀁 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀏𑀯𑀭𑀽𑀧𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀬𑀁 𑀓𑀭𑀡𑀻𑀬𑀁—  𑀥𑀫𑁆𑀫𑀻 𑀯𑀸 𑀓𑀣𑀸 𑀅𑀭𑀺𑀬𑁄 𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄”𑀢𑀺𑁇

223\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

224\. _“𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄,_  
_𑀅𑀢𑁆𑀢𑀪𑀭𑀲𑁆𑀲 𑀅𑀦𑀜𑁆𑀜𑀧𑁄𑀲𑀺𑀦𑁄;_  
_𑀤𑁂𑀯𑀸 𑀧𑀺𑀳𑀬𑀦𑁆𑀢𑀺 𑀢𑀸𑀤𑀺𑀦𑁄,_  
_𑀦𑁄 𑀘𑁂 𑀲𑀤𑁆𑀤𑀲𑀺𑀮𑁄𑀓𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄”𑀢𑀺𑁇 (𑁩𑁨)_
