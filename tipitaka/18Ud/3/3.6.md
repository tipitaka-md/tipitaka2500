# 𑁩𑁇𑁬 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑀼𑀢𑁆𑀢

(26.)

203\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑀮𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀆𑀬𑀲𑁆𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑀮𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀻”𑀢𑀺𑁇

204\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇

205\. “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀯𑀘𑁆𑀙, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑀮𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇

206\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂𑀦𑀺𑀯𑀸𑀲𑀁 𑀫𑀦𑀲𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀫𑀸 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀺𑀢𑁆𑀣𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀘𑁆𑀙𑁄 𑀤𑁄𑀲𑀦𑁆𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑀮𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀺𑁇 𑀯𑀘𑁆𑀙𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀜𑁆𑀘 𑀚𑀸𑀢𑀺𑀲𑀢𑀸𑀦𑀺 𑀅𑀩𑁆𑀩𑁄𑀓𑀺𑀡𑁆𑀡𑀸𑀦𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀼𑀮𑁂 𑀧𑀘𑁆𑀘𑀸𑀚𑀸𑀢𑀸𑀦𑀺𑁇 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀯𑀲𑀮𑀯𑀸𑀤𑁄 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀲𑀫𑀼𑀤𑀸𑀘𑀺𑀡𑁆𑀡𑁄𑁇 𑀢𑁂𑀦𑀸𑀬𑀁 𑀯𑀘𑁆𑀙𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑀮𑀯𑀸𑀤𑁂𑀦 𑀲𑀫𑀼𑀤𑀸𑀘𑀭𑀢𑀻”𑀢𑀺𑁇

207\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

208\. _“𑀬𑀫𑁆𑀳𑀺 𑀦 𑀫𑀸𑀬𑀸 𑀯𑀲𑀢𑀻 𑀦 𑀫𑀸𑀦𑁄,_  
_𑀬𑁄 𑀯𑀻𑀢𑀮𑁄𑀪𑁄 𑀅𑀫𑀫𑁄 𑀦𑀺𑀭𑀸𑀲𑁄;_  
_𑀧𑀡𑀼𑀦𑁆𑀦𑀓𑁄𑀥𑁄 𑀅𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀢𑁆𑀢𑁄,_  
_𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀲𑁄 𑀲𑀫𑀡𑁄 𑀲 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺𑁇 (𑁩𑁦)_
