# 𑁩𑁇𑁯 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀲𑀺𑀧𑁆𑀧𑀲𑀼𑀢𑁆𑀢

(29.)

226\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀏 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀅𑀬𑀫𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀺𑀧𑁆𑀧𑀁 𑀚𑀸𑀦𑀸𑀢𑀺? 𑀓𑁄 𑀓𑀺𑀁 𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀓𑁆𑀔𑀺? 𑀓𑀢𑀭𑀁 𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺?

227\. 𑀢𑀢𑁆𑀣𑁂𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀳𑀢𑁆𑀣𑀺𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀲𑁆𑀲𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀭𑀣𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀥𑀦𑀼𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀣𑀭𑀼𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀼𑀤𑁆𑀤𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀕𑀡𑀦𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀲𑀗𑁆𑀔𑀸𑀦𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀮𑁂𑀔𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀸𑀯𑁂𑀬𑁆𑀬𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀮𑁄𑀓𑀸𑀬𑀢𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀔𑀢𑁆𑀢𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘𑀭𑀳𑀺 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸𑁇

228\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦 𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀑 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀸𑀬 𑀦𑀼𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸, 𑀓𑀸 𑀘 𑀧𑀦 𑀯𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸”𑀢𑀺?

229\. “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀫𑀡𑁆𑀟𑀮𑀫𑀸𑀍𑀏 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀬𑀫𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀺𑀧𑁆𑀧𑀁 𑀚𑀸𑀦𑀸𑀢𑀺? 𑀓𑁄 𑀓𑀺𑀁 𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀓𑁆𑀔𑀺? 𑀓𑀢𑀭𑀁 𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺?

230\. 𑀢𑀢𑁆𑀣𑁂𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀳𑀢𑁆𑀣𑀺𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀲𑁆𑀲𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀭𑀣𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀥𑀦𑀼𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀣𑀭𑀼𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀫𑀼𑀤𑁆𑀤𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀕𑀡𑀦𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀗𑁆𑀔𑀸𑀦𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀮𑁂𑀔𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀓𑀸𑀯𑁂𑀬𑁆𑀬𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀮𑁄𑀓𑀸𑀬𑀢𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀔𑀢𑁆𑀢𑀯𑀺𑀚𑁆𑀚𑀸𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀧𑁆𑀧𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸, 𑀅𑀣 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇

231\. “𑀦 𑀔𑁆𑀯𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸𑀦𑀁 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀏𑀯𑀭𑀽𑀧𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀬𑀁 𑀓𑀭𑀡𑀻𑀬𑀁—  𑀥𑀫𑁆𑀫𑀻 𑀯𑀸 𑀓𑀣𑀸 𑀅𑀭𑀺𑀬𑁄 𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄”𑀢𑀺𑁇

232\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

233\. _“𑀅𑀲𑀺𑀧𑁆𑀧𑀚𑀻𑀯𑀻 𑀮𑀳𑀼 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑁄,_  
_𑀬𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄 𑀲𑀩𑁆𑀩𑀥𑀺 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄;_  
_𑀅𑀦𑁄𑀓𑀲𑀸𑀭𑀻 𑀅𑀫𑀫𑁄 𑀦𑀺𑀭𑀸𑀲𑁄,_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀦𑀁 𑀏𑀓𑀘𑀭𑁄 𑀲 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺𑁇 (𑁩𑁩)_
