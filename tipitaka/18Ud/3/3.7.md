# 𑁩𑁇𑁭 𑀲𑀓𑁆𑀓𑀼𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀦𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀲𑀓𑁆𑀓𑀼𑀤𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(27.)

210\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀧𑀺𑀧𑁆𑀧𑀮𑀺𑀕𑀼𑀳𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀏𑀓𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀫𑀸𑀥𑀺𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

211\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑀸𑀦𑀺 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀆𑀧𑀦𑁆𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑀸𑀦𑀺 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇

212\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑁂𑀲𑀓𑀸𑀭𑀯𑀡𑁆𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀦𑁆𑀢𑀁 𑀯𑀺𑀦𑀸𑀢𑀺𑁇 𑀲𑀼𑀚𑀸 𑀅𑀲𑀼𑀭𑀓𑀜𑁆𑀜𑀸 𑀢𑀲𑀭𑀁 𑀧𑀽𑀭𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑀧𑀤𑀸𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀖𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀳𑀢𑁆𑀣𑀢𑁄 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀖𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀖𑀝𑀺𑀬𑀸 𑀑𑀤𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀁 𑀧𑀽𑀭𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀅𑀤𑀸𑀲𑀺𑁇 𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁄 𑀅𑀦𑁂𑀓𑀲𑀽𑀧𑁄 𑀅𑀦𑁂𑀓𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁄 𑀅𑀦𑁂𑀓𑀭𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀁 𑀲𑀢𑁆𑀢𑁄 𑀬𑀲𑁆𑀲𑀸𑀬𑀁 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀓𑁆𑀓𑁄 𑀔𑁄 𑀅𑀬𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄”𑀢𑀺𑁇 𑀇𑀢𑀺 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀁 𑀔𑁄 𑀢𑁂 𑀇𑀤𑀁, 𑀓𑁄𑀲𑀺𑀬; 𑀫𑀸 𑀧𑀼𑀦𑀧𑀺 𑀏𑀯𑀭𑀽𑀧𑀫𑀓𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀅𑀢𑁆𑀣𑁄; 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀧𑀼𑀜𑁆𑀜𑁂𑀦 𑀓𑀭𑀡𑀻𑀬𑀦𑁆”𑀢𑀺𑁇

213\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  “𑀅𑀳𑁄 𑀤𑀸𑀦𑀁 𑀧𑀭𑀫𑀤𑀸𑀦𑀁 𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀅𑀳𑁄 𑀤𑀸𑀦𑀁 𑀧𑀭𑀫𑀤𑀸𑀦𑀁 𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀅𑀳𑁄 𑀤𑀸𑀦𑀁 𑀧𑀭𑀫𑀤𑀸𑀦𑀁 𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀺𑀩𑁆𑀩𑀸𑀬 𑀲𑁄𑀢𑀥𑀸𑀢𑀼𑀬𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀬 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀺𑀓𑀸𑀬 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀲𑁆𑀲 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  “𑀅𑀳𑁄 𑀤𑀸𑀦𑀁 𑀧𑀭𑀫𑀤𑀸𑀦𑀁 𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀅𑀳𑁄 𑀤𑀸𑀦𑀁 𑀧𑀭𑀫𑀤𑀸𑀦𑀁 𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀅𑀳𑁄 𑀤𑀸𑀦𑀁 𑀧𑀭𑀫𑀤𑀸𑀦𑀁 𑀓𑀲𑁆𑀲𑀧𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀦𑁆”𑀢𑀺𑁇

214\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

215\. _“𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄,_  
_𑀅𑀢𑁆𑀢𑀪𑀭𑀲𑁆𑀲 𑀅𑀦𑀜𑁆𑀜𑀧𑁄𑀲𑀺𑀦𑁄;_  
_𑀤𑁂𑀯𑀸 𑀧𑀺𑀳𑀬𑀦𑁆𑀢𑀺 𑀢𑀸𑀤𑀺𑀦𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑀸 𑀲𑀢𑀻𑀫𑀢𑁄”𑀢𑀺𑁇 (𑁩𑁧)_
