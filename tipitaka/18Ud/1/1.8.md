# 𑁧𑁇𑁮 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

### 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑀼𑀢𑁆𑀢

(8.)

37\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸—  “𑀅𑀬𑁆𑀬𑁄 𑀓𑀺𑀭 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀲𑀸 𑀤𑀸𑀭𑀓𑀁 𑀆𑀤𑀸𑀬 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇

38\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀔𑀼𑀤𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑀺, 𑀲𑀫𑀡, 𑀧𑁄𑀲 𑀫𑀦𑁆”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

39\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀔𑀼𑀤𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑀺, 𑀲𑀫𑀡, 𑀧𑁄𑀲 𑀫𑀦𑁆”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

40\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀔𑀼𑀤𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀜𑁆𑀳𑀺, 𑀲𑀫𑀡, 𑀧𑁄𑀲 𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀢𑀼𑀡𑁆𑀳𑀻 𑀅𑀳𑁄𑀲𑀺𑁇

41\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺—  “𑀏𑀲𑁄 𑀢𑁂, 𑀲𑀫𑀡, 𑀧𑀼𑀢𑁆𑀢𑁄; 𑀧𑁄𑀲 𑀦𑀦𑁆”𑀢𑀺𑁇

42\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀦𑁂𑀯 𑀑𑀮𑁄𑀓𑁂𑀲𑀺 𑀦𑀸𑀧𑀺 𑀆𑀮𑀧𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀧𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀻 𑀅𑀤𑁆𑀤𑀲 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀁 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀦𑁂𑀯 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑀁 𑀦𑀸𑀧𑀺 𑀆𑀮𑀧𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦 𑀘𑀸𑀬𑀁 𑀲𑀫𑀡𑁄 𑀧𑀼𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀅𑀢𑁆𑀣𑀺𑀓𑁄”𑀢𑀺𑁇 𑀢𑀢𑁄 𑀧𑀝𑀺𑀦𑀺𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀤𑀸𑀭𑀓𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀘𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀸𑀦𑀼𑀲𑀓𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀏𑀯𑀭𑀽𑀧𑀁 𑀯𑀺𑀧𑁆𑀧𑀓𑀸𑀭𑀁𑁇

43\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

44\. _“𑀆𑀬𑀦𑁆𑀢𑀺𑀁 𑀦𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀢𑀺,_  
_𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑀁 𑀦 𑀲𑁄𑀘𑀢𑀺;_  
_𑀲𑀗𑁆𑀕𑀸 𑀲𑀗𑁆𑀕𑀸𑀫𑀚𑀺𑀁 𑀫𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑁇 (𑁮)_
