# 𑁧𑁇𑁬 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑀼𑀢𑁆𑀢

(6.)

28\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀧𑀺𑀧𑁆𑀧𑀮𑀺𑀕𑀼𑀳𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀆𑀩𑀸𑀥𑀺𑀓𑁄 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀩𑀸𑀍𑀳𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀧𑀭𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀫𑁆𑀳𑀸 𑀆𑀩𑀸𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀢𑀫𑁆𑀳𑀸 𑀆𑀩𑀸𑀥𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

29\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑀸𑀦𑀺 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀆𑀧𑀦𑁆𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀮𑀸𑀪𑀸𑀬𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀤𑁂𑀯𑀢𑀸𑀲𑀢𑀸𑀦𑀺 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺—  𑀬𑁂𑀦 𑀤𑀮𑀺𑀤𑁆𑀤𑀯𑀺𑀲𑀺𑀔𑀸 𑀓𑀧𑀡𑀯𑀺𑀲𑀺𑀔𑀸 𑀧𑁂𑀲𑀓𑀸𑀭𑀯𑀺𑀲𑀺𑀔𑀸𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀁 𑀬𑁂𑀦 𑀤𑀮𑀺𑀤𑁆𑀤𑀯𑀺𑀲𑀺𑀔𑀸 𑀓𑀧𑀡𑀯𑀺𑀲𑀺𑀔𑀸 𑀧𑁂𑀲𑀓𑀸𑀭𑀯𑀺𑀲𑀺𑀔𑀸𑁇

30\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

31\. _“𑀅𑀦𑀜𑁆𑀜𑀧𑁄𑀲𑀺𑀫𑀜𑁆𑀜𑀸𑀢𑀁,_  
_𑀤𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀔𑀻𑀡𑀸𑀲𑀯𑀁 𑀯𑀦𑁆𑀢𑀤𑁄𑀲𑀁,_  
_𑀢𑀫𑀳𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀦𑁆”𑀢𑀺𑁇 (𑁬)_
