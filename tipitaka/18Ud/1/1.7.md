# 𑁧𑁇𑁭 𑀅𑀚𑀓𑀮𑀸𑀧𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

* 𑀅𑀚𑀓𑀮𑀸𑀧𑀓𑀲𑀼𑀢𑁆𑀢

(7.)

33\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀧𑀸𑀯𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀚𑀓𑀮𑀸𑀧𑀓𑁂 𑀘𑁂𑀢𑀺𑀬𑁂, 𑀅𑀚𑀓𑀮𑀸𑀧𑀓𑀲𑁆𑀲 𑀬𑀓𑁆𑀔𑀲𑁆𑀲 𑀪𑀯𑀦𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀢𑀺𑀫𑀺𑀲𑀸𑀬𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺; 𑀤𑁂𑀯𑁄 𑀘 𑀏𑀓𑀫𑁂𑀓𑀁 𑀨𑀼𑀲𑀸𑀬𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀚𑀓𑀮𑀸𑀧𑀓𑁄 𑀬𑀓𑁆𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀬𑀁 𑀙𑀫𑁆𑀪𑀺𑀢𑀢𑁆𑀢𑀁 𑀮𑁄𑀫𑀳𑀁𑀲𑀁 𑀉𑀧𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 “𑀅𑀓𑁆𑀓𑀼𑀮𑁄 𑀧𑀓𑁆𑀓𑀼𑀮𑁄”𑀢𑀺 𑀅𑀓𑁆𑀓𑀼𑀮𑀧𑀓𑁆𑀓𑀼𑀮𑀺𑀓𑀁 𑀅𑀓𑀸𑀲𑀺—  “𑀏𑀲𑁄 𑀢𑁂, 𑀲𑀫𑀡, 𑀧𑀺𑀲𑀸𑀘𑁄”𑀢𑀺𑁇

34\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

35\. _“𑀬𑀤𑀸 𑀲𑀓𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼,_  
_𑀧𑀸𑀭𑀕𑀽 𑀳𑁄𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀅𑀣 𑀏𑀢𑀁 𑀧𑀺𑀲𑀸𑀘𑀜𑁆𑀘,_  
_𑀧𑀓𑁆𑀓𑀼𑀮𑀜𑁆𑀘𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇 (𑁭)_
