# 𑁧𑁇𑁧 𑀧𑀞𑀫𑀩𑁄𑀥𑀺𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

* 𑀧𑀞𑀫𑀩𑁄𑀥𑀺𑀲𑀼𑀢𑁆𑀢

(1.)

2\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀦𑀚𑁆𑀚𑀸 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀸𑀬 𑀢𑀻𑀭𑁂 𑀩𑁄𑀥𑀺𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀏𑀓𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀼𑀔𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀞𑀫𑀁 𑀬𑀸𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘𑀲𑀫𑀼𑀧𑁆𑀧𑀸𑀤𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑀁 𑀲𑀸𑀥𑀼𑀓𑀁 𑀫𑀦𑀲𑀸𑀓𑀸𑀲𑀺—

3\. “𑀇𑀢𑀺 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑀢𑀺 𑀇𑀤𑀁 𑀳𑁄𑀢𑀺, 𑀇𑀫𑀲𑁆𑀲𑀼𑀧𑁆𑀧𑀸𑀤𑀸 𑀇𑀤𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀬𑀤𑀺𑀤𑀁—  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀗𑁆𑀔𑀸𑀭𑀸, 𑀲𑀗𑁆𑀔𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁, 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀸𑀫𑀭𑀽𑀧𑀁, 𑀦𑀸𑀫𑀭𑀽𑀧𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀍𑀆𑀬𑀢𑀦𑀁, 𑀲𑀍𑀆𑀬𑀢𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀨𑀲𑁆𑀲𑁄, 𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑁂𑀤𑀦𑀸, 𑀯𑁂𑀤𑀦𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀡𑁆𑀳𑀸, 𑀢𑀡𑁆𑀳𑀸𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑀸𑀤𑀸𑀦𑀁, 𑀉𑀧𑀸𑀤𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀪𑀯𑁄, 𑀪𑀯𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀸𑀢𑀺, 𑀚𑀸𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀚𑀭𑀸𑀫𑀭𑀡𑀁 𑀲𑁄𑀓𑀧𑀭𑀺𑀤𑁂𑀯𑀤𑀼𑀓𑁆𑀔𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀼𑀧𑀸𑀬𑀸𑀲𑀸 𑀲𑀫𑁆𑀪𑀯𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀓𑁂𑀯𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀦𑁆𑀥𑀲𑁆𑀲 𑀲𑀫𑀼𑀤𑀬𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

4\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

5\. _“𑀬𑀤𑀸 𑀳𑀯𑁂 𑀧𑀸𑀢𑀼𑀪𑀯𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀸,_  
_𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀛𑀸𑀬𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲;_  
_𑀅𑀣𑀲𑁆𑀲 𑀓𑀗𑁆𑀔𑀸 𑀯𑀧𑀬𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀸,_  
_𑀬𑀢𑁄 𑀧𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑀳𑁂𑀢𑀼𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 (𑁧)_
