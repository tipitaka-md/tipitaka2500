# 𑁧𑁇𑁧𑁦 𑀩𑀸𑀳𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

### 𑀩𑀸𑀳𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(10.)

51\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑁂 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀲𑀫𑀼𑀤𑁆𑀤𑀢𑀻𑀭𑁂 𑀲𑀓𑁆𑀓𑀢𑁄 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑁂 𑀔𑁄 𑀓𑁂𑀘𑀺 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸, 𑀅𑀳𑀁 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄”𑀢𑀺𑁇

52\. 𑀅𑀣 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀧𑀼𑀭𑀸𑀡𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀩𑀸𑀳𑀺𑀬𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀬𑁂𑀦 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀩𑀸𑀳𑀺𑀬𑀁 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑁂𑀯 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀩𑀸𑀳𑀺𑀬, 𑀅𑀭𑀳𑀸, 𑀦𑀸𑀧𑀺 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀲𑀸𑀧𑀺 𑀢𑁂 𑀧𑀝𑀺𑀧𑀤𑀸 𑀦𑀢𑁆𑀣𑀺 𑀬𑀸𑀬 𑀢𑁆𑀯𑀁 𑀅𑀭𑀳𑀸 𑀯𑀸 𑀅𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄”𑀢𑀺𑁇

53\. “𑀅𑀣 𑀓𑁂 𑀘𑀭𑀳𑀺 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀺, 𑀩𑀸𑀳𑀺𑀬, 𑀉𑀢𑁆𑀢𑀭𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀲𑀸𑀯𑀢𑁆𑀣𑀻 𑀦𑀸𑀫 𑀦𑀕𑀭𑀁𑁇 𑀢𑀢𑁆𑀣 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀭𑀳𑀺 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑁇 𑀲𑁄 𑀳𑀺, 𑀩𑀸𑀳𑀺𑀬, 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀸 𑀘𑁂𑀯 𑀅𑀭𑀳𑀢𑁆𑀢𑀸𑀬 𑀘 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀻”𑀢𑀺𑁇

54\. 𑀅𑀣 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀢𑀸𑀬 𑀤𑁂𑀯𑀢𑀸𑀬 𑀲𑀁𑀯𑁂𑀚𑀺𑀢𑁄 𑀢𑀸𑀯𑀤𑁂𑀯 𑀲𑀼𑀧𑁆𑀧𑀸𑀭𑀓𑀫𑁆𑀳𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀭𑀢𑁆𑀢𑀺𑀧𑀭𑀺𑀯𑀸𑀲𑁂𑀦 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀻 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀳𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄? 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑀫𑁆𑀳𑀸 𑀫𑀬𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀔𑁄, 𑀩𑀸𑀳𑀺𑀬, 𑀪𑀕𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬𑀸”𑀢𑀺𑁇

55\. 𑀅𑀣 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑁄 𑀚𑁂𑀢𑀯𑀦𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀤𑁆𑀤𑀲 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀁 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀧𑀲𑀸𑀤𑀦𑀻𑀬𑀁 𑀲𑀦𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀉𑀢𑁆𑀢𑀫𑀤𑀫𑀣𑀲𑀫𑀣𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁 𑀤𑀦𑁆𑀢𑀁 𑀕𑀼𑀢𑁆𑀢𑀁 𑀬𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀸𑀕𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑁂𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁; 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁, 𑀬𑀁 𑀫𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀩𑀸𑀳𑀺𑀬𑀁 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄 𑀢𑀸𑀯, 𑀩𑀸𑀳𑀺𑀬, 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬𑀸”𑀢𑀺𑁇

56\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀼𑀚𑁆𑀚𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀦𑀁, 𑀫𑀬𑁆𑀳𑀁 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀦𑀁𑁇 𑀤𑁂𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁; 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁, 𑀬𑀁 𑀫𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀸𑀳𑀺𑀬𑀁 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄 𑀢𑀸𑀯, 𑀩𑀸𑀳𑀺𑀬, 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬𑀸”𑀢𑀺𑁇

57\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀼𑀚𑁆𑀚𑀸𑀦𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀦𑀁, 𑀫𑀬𑁆𑀳𑀁 𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀦𑁆𑀢𑀭𑀸𑀬𑀸𑀦𑀁𑁇 𑀤𑁂𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁; 𑀤𑁂𑀲𑁂𑀢𑀼 𑀲𑀼𑀕𑀢𑁄 𑀥𑀫𑁆𑀫𑀁, 𑀬𑀁 𑀫𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

58\. “𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳 𑀢𑁂, 𑀩𑀸𑀳𑀺𑀬, 𑀏𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀤𑀺𑀝𑁆𑀞𑁂 𑀤𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀼𑀢𑁂 𑀲𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀼𑀢𑁂 𑀫𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀳𑀺 𑀢𑁂, 𑀩𑀸𑀳𑀺𑀬, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑀢𑁄 𑀔𑁄 𑀢𑁂, 𑀩𑀸𑀳𑀺𑀬, 𑀤𑀺𑀝𑁆𑀞𑁂 𑀤𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀼𑀢𑁂 𑀲𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀼𑀢𑁂 𑀫𑀼𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀫𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀩𑀸𑀳𑀺𑀬, 𑀦 𑀢𑁂𑀦; 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀩𑀸𑀳𑀺𑀬, 𑀦 𑀢𑁂𑀦 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀩𑀸𑀳𑀺𑀬, 𑀦 𑀢𑀢𑁆𑀣; 𑀬𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀩𑀸𑀳𑀺𑀬, 𑀦 𑀢𑀢𑁆𑀣, 𑀢𑀢𑁄 𑀢𑁆𑀯𑀁, 𑀩𑀸𑀳𑀺𑀬, 𑀦𑁂𑀯𑀺𑀥 𑀦 𑀳𑀼𑀭𑀁 𑀦 𑀉𑀪𑀬𑀫𑀦𑁆𑀢𑀭𑁂𑀦𑁇 𑀏𑀲𑁂𑀯𑀦𑁆𑀢𑁄 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

59\. 𑀅𑀣 𑀔𑁄 𑀩𑀸𑀳𑀺𑀬𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀇𑀫𑀸𑀬 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸𑀬 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸𑀬 𑀢𑀸𑀯𑀤𑁂𑀯 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑁇

60\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀸𑀳𑀺𑀬𑀁 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀁 𑀇𑀫𑀺𑀦𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀑𑀯𑀸𑀤𑁂𑀦 𑀑𑀯𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀩𑀸𑀳𑀺𑀬𑀁 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀁 𑀕𑀸𑀯𑀻 𑀢𑀭𑀼𑀡𑀯𑀘𑁆𑀙𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇

61\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀕𑀭𑀫𑁆𑀳𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀤𑁆𑀤𑀲 𑀩𑀸𑀳𑀺𑀬𑀁 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀁 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀁; 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀡𑁆𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀳𑀺𑀬𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀓𑀁; 𑀫𑀜𑁆𑀘𑀓𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀛𑀸𑀧𑁂𑀣; 𑀣𑀽𑀧𑀜𑁆𑀘𑀲𑁆𑀲 𑀓𑀭𑁄𑀣𑁇 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄”𑀢𑀺𑁇

62\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀩𑀸𑀳𑀺𑀬𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀓𑀁 𑀫𑀜𑁆𑀘𑀓𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀛𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀣𑀽𑀧𑀜𑁆𑀘𑀲𑁆𑀲 𑀓𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀤𑀟𑁆𑀠𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀸𑀳𑀺𑀬𑀲𑁆𑀲 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑀲𑁆𑀲 𑀲𑀭𑀻𑀭𑀁, 𑀣𑀽𑀧𑁄 𑀘𑀲𑁆𑀲 𑀓𑀢𑁄𑁇 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄”𑀢𑀺? “𑀧𑀡𑁆𑀟𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄 𑀧𑀘𑁆𑀘𑀧𑀸𑀤𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁; 𑀦 𑀘 𑀫𑀁 𑀥𑀫𑁆𑀫𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀲𑀺𑁇 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀳𑀺𑀬𑁄 𑀤𑀸𑀭𑀼𑀘𑀻𑀭𑀺𑀬𑁄”𑀢𑀺𑁇

63\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

64\. _“𑀬𑀢𑁆𑀣 𑀆𑀧𑁄 𑀘 𑀧𑀣𑀯𑀻,_  
_𑀢𑁂𑀚𑁄 𑀯𑀸𑀬𑁄 𑀦 𑀕𑀸𑀥𑀢𑀺;_  
_𑀦 𑀢𑀢𑁆𑀣 𑀲𑀼𑀓𑁆𑀓𑀸 𑀚𑁄𑀢𑀦𑁆𑀢𑀺,_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑁄 𑀦𑀧𑁆𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀦 𑀢𑀢𑁆𑀣 𑀘𑀦𑁆𑀤𑀺𑀫𑀸 𑀪𑀸𑀢𑀺,_  
_𑀢𑀫𑁄 𑀢𑀢𑁆𑀣 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧𑁦)_  


65\. _𑀬𑀤𑀸 𑀘 𑀅𑀢𑁆𑀢𑀦𑀸𑀯𑁂𑀤𑀺,_  
_𑀫𑀼𑀦𑀺 𑀫𑁄𑀦𑁂𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀅𑀣 𑀭𑀽𑀧𑀸 𑀅𑀭𑀽𑀧𑀸 𑀘,_  
_𑀲𑀼𑀔𑀤𑀼𑀓𑁆𑀔𑀸 𑀧𑀫𑀼𑀘𑁆𑀘𑀢𑀻”𑀢𑀺𑁇 (𑁧𑁧)_
