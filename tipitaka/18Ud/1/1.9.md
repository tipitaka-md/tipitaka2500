# 𑁧𑁇𑁯 𑀚𑀝𑀺𑀮𑀲𑀼𑀢𑁆𑀢

### 𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

### 𑀚𑀝𑀺𑀮𑀲𑀼𑀢𑁆𑀢

(9.)

46\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀕𑀬𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀬𑀸𑀲𑀻𑀲𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀚𑀝𑀺𑀮𑀸 𑀲𑀻𑀢𑀸𑀲𑀼 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀝𑁆𑀞𑀓𑁂 𑀳𑀺𑀫𑀧𑀸𑀢𑀲𑀫𑀬𑁂 𑀕𑀬𑀸𑀬𑀁 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀧𑀺 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀑𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺𑀧𑀺, 𑀅𑀕𑁆𑀕𑀺𑀫𑁆𑀧𑀺 𑀚𑀼𑀳𑀦𑁆𑀢𑀺—  “𑀇𑀫𑀺𑀦𑀸 𑀲𑀼𑀤𑁆𑀥𑀻”𑀢𑀺𑁇

47\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀚𑀝𑀺𑀮𑁂 𑀲𑀻𑀢𑀸𑀲𑀼 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀝𑁆𑀞𑀓𑁂 𑀳𑀺𑀫𑀧𑀸𑀢𑀲𑀫𑀬𑁂 𑀕𑀬𑀸𑀬𑀁 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑁂𑀧𑀺 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑁂𑀧𑀺 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑁂 𑀑𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑁂𑀧𑀺 𑀅𑀕𑁆𑀕𑀺𑀫𑁆𑀧𑀺 𑀚𑀼𑀳𑀦𑁆𑀢𑁂—  “𑀇𑀫𑀺𑀦𑀸 𑀲𑀼𑀤𑁆𑀥𑀻”𑀢𑀺𑁇

48\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

49\. _“𑀦 𑀉𑀤𑀓𑁂𑀦 𑀲𑀼𑀘𑀻 𑀳𑁄𑀢𑀺,_  
_𑀩𑀳𑁆𑀯𑁂𑀢𑁆𑀣 𑀦𑁆𑀳𑀸𑀬𑀢𑀻 𑀚𑀦𑁄;_  
_𑀬𑀫𑁆𑀳𑀺 𑀲𑀘𑁆𑀘𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁄 𑀘,_  
_𑀲𑁄 𑀲𑀼𑀘𑀻 𑀲𑁄 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄”𑀢𑀺𑁇 (𑁯)_
