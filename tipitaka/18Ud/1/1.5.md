# 𑁧𑁇𑁫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

* 𑀩𑁄𑀥𑀺𑀯𑀕𑁆𑀕

* 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑀼𑀢𑁆𑀢

(5.)

23\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀫𑀳𑀸𑀘𑀼𑀦𑁆𑀤𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀭𑁂𑀯𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇

24\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂; 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀚𑀸𑀢𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀢𑀫𑁂 𑀘 𑀧𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀓𑀭𑀡𑀸 𑀥𑀫𑁆𑀫𑀸”𑀢𑀺?

25\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

26\. _“𑀩𑀸𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀧𑀓𑁂 𑀥𑀫𑁆𑀫𑁂,_  
_𑀬𑁂 𑀘𑀭𑀦𑁆𑀢𑀺 𑀲𑀤𑀸 𑀲𑀢𑀸;_  
_𑀔𑀻𑀡𑀲𑀁𑀬𑁄𑀚𑀦𑀸 𑀩𑀼𑀤𑁆𑀥𑀸,_  
_𑀢𑁂 𑀯𑁂 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸”𑀢𑀺𑁇 (𑁫)_
