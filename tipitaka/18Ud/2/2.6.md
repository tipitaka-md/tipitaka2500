# 𑁨𑁇𑁬 𑀕𑀩𑁆𑀪𑀺𑀦𑀻𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀕𑀩𑁆𑀪𑀺𑀦𑀻𑀲𑀼𑀢𑁆𑀢

(16.)

100\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀤𑀳𑀭𑀫𑀸𑀡𑀯𑀺𑀓𑀸 𑀧𑀚𑀸𑀧𑀢𑀺 𑀳𑁄𑀢𑀺 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀉𑀧𑀯𑀺𑀚𑀜𑁆𑀜𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑁂𑀮𑀁 𑀆𑀳𑀭, 𑀬𑀁 𑀫𑁂 𑀯𑀺𑀚𑀸𑀢𑀸𑀬 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

101\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀼𑀢𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑁄𑀢𑀺, 𑀢𑁂𑀮𑀁 𑀆𑀳𑀭𑀸𑀫𑀻”𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑁂𑀮𑀁 𑀆𑀳𑀭, 𑀬𑀁 𑀫𑁂 𑀯𑀺𑀚𑀸𑀢𑀸𑀬 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀼𑀢𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑁄𑀢𑀺, 𑀢𑁂𑀮𑀁 𑀆𑀳𑀭𑀸𑀫𑀻”𑀢𑀺? 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀢𑁂𑀮𑀁 𑀆𑀳𑀭, 𑀬𑀁 𑀫𑁂 𑀯𑀺𑀚𑀸𑀢𑀸𑀬 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

102\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑁂 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀲𑀧𑁆𑀧𑀺𑀲𑁆𑀲 𑀯𑀸 𑀢𑁂𑀮𑀲𑁆𑀲 𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀢𑀼𑀁 𑀤𑀻𑀬𑀢𑀺, 𑀦𑁄 𑀦𑀻𑀳𑀭𑀺𑀢𑀼𑀁𑁇

103\. 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀭𑀜𑁆𑀜𑁄 𑀔𑁄 𑀧𑀦 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑁂 𑀲𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀸 𑀲𑀧𑁆𑀧𑀺𑀲𑁆𑀲 𑀯𑀸 𑀢𑁂𑀮𑀲𑁆𑀲 𑀯𑀸 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀢𑀼𑀁 𑀤𑀻𑀬𑀢𑀺, 𑀦𑁄 𑀦𑀻𑀳𑀭𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑁂𑀮𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀖𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀉𑀘𑁆𑀙𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀤𑀤𑁂𑀬𑁆𑀬𑀁, 𑀬𑀁 𑀇𑀫𑀺𑀲𑁆𑀲𑀸 𑀯𑀺𑀚𑀸𑀢𑀸𑀬 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

104\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑁂𑀮𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀺𑀯𑀺𑀢𑁆𑀯𑀸 𑀖𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀲𑀓𑁆𑀓𑁄𑀢𑀺 𑀉𑀤𑁆𑀥𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀦 𑀧𑀦 𑀅𑀥𑁄𑁇 𑀲𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀢𑀺𑀩𑁆𑀩𑀸𑀳𑀺 𑀔𑀭𑀸𑀳𑀺 𑀓𑀝𑀼𑀓𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑁄 𑀆𑀯𑀝𑁆𑀝𑀢𑀺 𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀢𑀺𑁇

105\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀢𑀺𑀩𑁆𑀩𑀸𑀳𑀺 𑀔𑀭𑀸𑀳𑀺 𑀓𑀝𑀼𑀓𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑀁 𑀆𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑀁 𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀫𑀸𑀦𑀁𑁇

106\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

107\. _“𑀲𑀼𑀔𑀺𑀦𑁄 𑀯𑀢 𑀬𑁂 𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑀸,_  
_𑀯𑁂𑀤𑀕𑀼𑀦𑁄 𑀳𑀺 𑀚𑀦𑀸 𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑀸;_  
_𑀲𑀓𑀺𑀜𑁆𑀘𑀦𑀁 𑀧𑀲𑁆𑀲 𑀯𑀺𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑀁,_  
_𑀚𑀦𑁄 𑀚𑀦𑀲𑁆𑀫𑀺𑀁 𑀧𑀝𑀺𑀩𑀦𑁆𑀥𑀘𑀺𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 (𑁧𑁯)_
