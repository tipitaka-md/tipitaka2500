# 𑁨𑁇𑁨 𑀭𑀸𑀚𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀭𑀸𑀚𑀲𑀼𑀢𑁆𑀢

(12.)

78\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀅𑀬𑀫𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀭𑀸𑀚𑀽𑀦𑀁 𑀫𑀳𑀤𑁆𑀥𑀦𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀪𑁄𑀕𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀓𑁄𑀲𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀯𑀺𑀚𑀺𑀢𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀯𑀸𑀳𑀦𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀭𑁄 𑀯𑀸 𑀭𑀸𑀚𑀸 𑀯𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄, 𑀭𑀸𑀚𑀸 𑀯𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄”𑀢𑀺? 𑀅𑀬𑀜𑁆𑀘𑀭𑀳𑀺 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸𑁇

79\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀸𑀬 𑀦𑀼𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸, 𑀓𑀸 𑀘 𑀧𑀦 𑀯𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸”𑀢𑀺?

80\. “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬𑀁 𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀅𑀬𑀫𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑁂𑀲𑀁 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀭𑀸𑀚𑀽𑀦𑀁 𑀫𑀳𑀤𑁆𑀥𑀦𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀪𑁄𑀕𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀓𑁄𑀲𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀯𑀺𑀚𑀺𑀢𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀯𑀸𑀳𑀦𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀩𑁆𑀩𑀮𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀭𑁄 𑀯𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀭𑁄 𑀯𑀸 𑀭𑀸𑀚𑀸 𑀯𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄, 𑀭𑀸𑀚𑀸 𑀯𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄’𑀢𑀺? 𑀅𑀬𑀁 𑀔𑁄 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁆𑀢𑀭𑀸𑀓𑀣𑀸 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀸, 𑀅𑀣 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇

81\. “𑀦 𑀔𑁆𑀯𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀤𑁆𑀥𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸𑀦𑀁 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀏𑀯𑀭𑀽𑀧𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀸𑀦𑀁 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀬𑀁 𑀓𑀭𑀡𑀻𑀬𑀁—  𑀥𑀫𑁆𑀫𑀻 𑀯𑀸 𑀓𑀣𑀸 𑀅𑀭𑀺𑀬𑁄 𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄”𑀢𑀺𑁇

82\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

83\. _“𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀲𑀼𑀔𑀁 𑀮𑁄𑀓𑁂,_  
_𑀬𑀜𑁆𑀘𑀺𑀤𑀁 𑀤𑀺𑀯𑀺𑀬𑀁 𑀲𑀼𑀔𑀁;_  
_𑀢𑀡𑁆𑀳𑀓𑁆𑀔𑀬𑀲𑀼𑀔𑀲𑁆𑀲𑁂𑀢𑁂,_  
_𑀓𑀮𑀁 𑀦𑀸𑀕𑁆𑀖𑀦𑁆𑀢𑀺 𑀲𑁄𑀍𑀅𑀲𑀺𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁪)_
