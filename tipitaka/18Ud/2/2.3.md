# 𑁨𑁇𑁩 𑀤𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀤𑀡𑁆𑀟𑀲𑀼𑀢𑁆𑀢

(13.)

85\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀸 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀳𑀺𑀁 𑀤𑀡𑁆𑀟𑁂𑀦 𑀳𑀦𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀓𑀼𑀫𑀸𑀭𑀓𑁂 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀳𑀺𑀁 𑀤𑀡𑁆𑀟𑁂𑀦 𑀳𑀦𑀦𑁆𑀢𑁂𑁇

86\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

87\. _“𑀲𑀼𑀔𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀬𑁄 𑀤𑀡𑁆𑀟𑁂𑀦 𑀯𑀺𑀳𑀺𑀁𑀲𑀢𑀺;_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀲𑀼𑀔𑀫𑁂𑀲𑀸𑀦𑁄,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑁄 𑀦 𑀮𑀪𑀢𑁂 𑀲𑀼𑀔𑀁𑁇 (𑁧𑁫)_  


88\. _𑀲𑀼𑀔𑀓𑀸𑀫𑀸𑀦𑀺 𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀬𑁄 𑀤𑀡𑁆𑀟𑁂𑀦 𑀦 𑀳𑀺𑀁𑀲𑀢𑀺;_  
_𑀅𑀢𑁆𑀢𑀦𑁄 𑀲𑀼𑀔𑀫𑁂𑀲𑀸𑀦𑁄,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑁄 𑀮𑀪𑀢𑁂 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁬)_  


---
