# 𑁨𑁇𑁫 𑀉𑀧𑀸𑀲𑀓𑀲𑀼𑀢𑁆𑀢

* 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀸𑀲𑀓𑀲𑀼𑀢𑁆𑀢

(15.)

95\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀇𑀘𑁆𑀙𑀸𑀦𑀗𑁆𑀕𑀮𑀓𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀻𑀭𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀘𑀺𑀭𑀲𑁆𑀲𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀸𑀲𑀓, 𑀇𑀫𑀁 𑀧𑀭𑀺𑀬𑀸𑀬𑀫𑀓𑀸𑀲𑀺 𑀬𑀤𑀺𑀤𑀁 𑀇𑀥𑀸𑀕𑀫𑀦𑀸𑀬𑀸”𑀢𑀺𑁇

96\. “𑀘𑀺𑀭𑀧𑀝𑀺𑀓𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀓𑀸𑀫𑁄, 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀓𑁂𑀳𑀺𑀘𑀺 𑀓𑁂𑀳𑀺𑀘𑀺 𑀓𑀺𑀘𑁆𑀘𑀓𑀭𑀡𑀻𑀬𑁂𑀳𑀺 𑀩𑁆𑀬𑀸𑀯𑀝𑁄𑁇 𑀏𑀯𑀸𑀳𑀁 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

97\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

98\. _“𑀲𑀼𑀔𑀁 𑀯𑀢 𑀢𑀲𑁆𑀲 𑀦 𑀳𑁄𑀢𑀺 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲;_  
_𑀲𑀓𑀺𑀜𑁆𑀘𑀦𑀁 𑀧𑀲𑁆𑀲 𑀯𑀺𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑀁,_  
_𑀚𑀦𑁄 𑀚𑀦𑀲𑁆𑀫𑀺𑀁 𑀧𑀝𑀺𑀩𑀦𑁆𑀥𑀭𑀽𑀧𑁄”𑀢𑀺𑁇 (𑁧𑁮)_
