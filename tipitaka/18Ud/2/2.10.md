# 𑁨𑁇𑁧𑁦 𑀪𑀤𑁆𑀤𑀺𑀬𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀪𑀤𑁆𑀤𑀺𑀬𑀲𑀼𑀢𑁆𑀢

(20.)

140\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀺𑀬𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀫𑁆𑀩𑀯𑀦𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  “𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇

141\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀪𑀤𑁆𑀤𑀺𑀬𑀲𑁆𑀲 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑀲𑁆𑀲𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑀲𑁆𑀲𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑀲𑁆𑀲𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  “𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀦𑁂𑀲𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺, 𑀬𑀁𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀕𑀸𑀭𑀺𑀬𑀪𑀽𑀢𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀲𑀼𑀔𑀁, 𑀲𑁄 𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀫𑀸𑀦𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  ‘𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆’”𑀢𑀺𑁇

142\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀆𑀬𑀲𑁆𑀫𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  ‘𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆’𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺𑁇 𑀬𑀁𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀕𑀸𑀭𑀺𑀬𑀪𑀽𑀢𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀲𑀼𑀔𑀁, 𑀲𑁄 𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀫𑀸𑀦𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  ‘𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆’”𑀢𑀺𑁇

143\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀳𑀺—  ‘𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀪𑀤𑁆𑀤𑀺𑀬, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻’”𑀢𑀺𑁇

144\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀪𑀤𑁆𑀤𑀺𑀬, 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁄 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—

145\. “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀪𑀤𑁆𑀤𑀺𑀬, 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  ‘𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇

146\. “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀪𑀤𑁆𑀤𑀺𑀬, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  ‘𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆’”𑀢𑀺𑁇 “𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀕𑀸𑀭𑀺𑀬𑀪𑀽𑀢𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄𑀧𑀺 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀭𑀓𑁆𑀔𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀩𑀳𑀺𑀧𑀺 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀭𑀓𑁆𑀔𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀅𑀦𑁆𑀢𑁄𑀧𑀺 𑀦𑀕𑀭𑁂 𑀭𑀓𑁆𑀔𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀩𑀳𑀺𑀧𑀺 𑀦𑀕𑀭𑁂 𑀭𑀓𑁆𑀔𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀅𑀦𑁆𑀢𑁄𑀧𑀺 𑀚𑀦𑀧𑀤𑁂 𑀭𑀓𑁆𑀔𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺, 𑀩𑀳𑀺𑀧𑀺 𑀚𑀦𑀧𑀤𑁂 𑀭𑀓𑁆𑀔𑀸 𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀁 𑀭𑀓𑁆𑀔𑀺𑀢𑁄 𑀕𑁄𑀧𑀺𑀢𑁄 𑀲𑀦𑁆𑀢𑁄 𑀪𑀻𑀢𑁄 𑀉𑀩𑁆𑀩𑀺𑀕𑁆𑀕𑁄 𑀉𑀲𑁆𑀲𑀗𑁆𑀓𑀻 𑀉𑀢𑁆𑀭𑀸𑀲𑀻 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀧𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀏𑀓𑁄 𑀅𑀪𑀻𑀢𑁄 𑀅𑀦𑀼𑀩𑁆𑀩𑀺𑀕𑁆𑀕𑁄 𑀅𑀦𑀼𑀲𑁆𑀲𑀗𑁆𑀓𑀻 𑀅𑀦𑀼𑀢𑁆𑀭𑀸𑀲𑀻 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀧𑀦𑁆𑀦𑀮𑁄𑀫𑁄 𑀧𑀭𑀤𑀢𑁆𑀢𑀯𑀼𑀢𑁆𑀢𑁄, 𑀫𑀺𑀕𑀪𑀽𑀢𑁂𑀦 𑀘𑁂𑀢𑀲𑀸 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁄 𑀅𑀭𑀜𑁆𑀜𑀕𑀢𑁄𑀧𑀺 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀕𑀢𑁄𑀧𑀺 𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀕𑀢𑁄𑀧𑀺 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺𑀁—  ‘𑀅𑀳𑁄 𑀲𑀼𑀔𑀁, 𑀅𑀳𑁄 𑀲𑀼𑀔𑀦𑁆’”𑀢𑀺𑁇

147\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

148\. _“𑀬𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀢𑁄 𑀦 𑀲𑀦𑁆𑀢𑀺 𑀓𑁄𑀧𑀸,_  
_𑀇𑀢𑀺𑀪𑀯𑀸𑀪𑀯𑀢𑀜𑁆𑀘 𑀯𑀻𑀢𑀺𑀯𑀢𑁆𑀢𑁄;_  
_𑀢𑀁 𑀯𑀺𑀕𑀢𑀪𑀬𑀁 𑀲𑀼𑀔𑀺𑀁 𑀅𑀲𑁄𑀓𑀁,_  
_𑀤𑁂𑀯𑀸 𑀦𑀸𑀦𑀼𑀪𑀯𑀦𑁆𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇_

151\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



152\. _𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑁄 𑀭𑀸𑀚𑀸 𑀤𑀡𑁆𑀟𑁂𑀦,_  
_𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀘;_  
_𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑁄 𑀘,_  
_𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀯𑀺𑀲𑀸𑀔𑀸 𑀘;_  
_𑀓𑀸𑀍𑀈𑀕𑁄𑀥𑀸𑀬 𑀪𑀤𑁆𑀤𑀺𑀬𑁄𑀢𑀺𑁇_
