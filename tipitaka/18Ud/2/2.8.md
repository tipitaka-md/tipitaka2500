# 𑁨𑁇𑁮 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀲𑀼𑀢𑁆𑀢

(18.)

116. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑀼𑀡𑁆𑀟𑀺𑀓𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀼𑀡𑁆𑀟𑀥𑀸𑀦𑀯𑀦𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀕𑀩𑁆𑀪𑀁 𑀥𑀸𑀭𑁂𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀽𑀍𑀳𑀕𑀩𑁆𑀪𑀸 𑀲𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀢𑀺𑀩𑁆𑀩𑀸𑀳𑀺 𑀔𑀭𑀸𑀳𑀺 𑀓𑀝𑀼𑀓𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑀸 𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀺—  “𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀢 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀬𑁄 𑀇𑀫𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺; 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀢 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀬𑁄 𑀇𑀫𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀬𑀢𑁆𑀣𑀺𑀤𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

117. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀸𑀫𑀺𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀸𑀳𑀺; 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀁 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀁 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀩𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀼𑀘𑁆𑀙—  ‘𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺; 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀁 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀁 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀩𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀳𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀕𑀩𑁆𑀪𑀁 𑀥𑀸𑀭𑁂𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀽𑀍𑀳𑀕𑀩𑁆𑀪𑀸 𑀲𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀢𑀺𑀩𑁆𑀩𑀸𑀳𑀺 𑀔𑀭𑀸𑀳𑀺 𑀓𑀝𑀼𑀓𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑀸 𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀺—  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀢 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀬𑁄 𑀇𑀫𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺; 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀢 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀬𑁄 𑀇𑀫𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀢𑀁 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀬𑀢𑁆𑀣𑀺𑀤𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀻’”𑀢𑀺𑁇

118. “𑀧𑀭𑀫𑀦𑁆”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺, 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀁 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀁 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀩𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺; 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀕𑀩𑁆𑀪𑀁 𑀥𑀸𑀭𑁂𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀽𑀍𑀳𑀕𑀩𑁆𑀪𑀸 𑀲𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀳𑀺 𑀢𑀺𑀩𑁆𑀩𑀸𑀳𑀺 𑀔𑀭𑀸𑀳𑀺 𑀓𑀝𑀼𑀓𑀸𑀳𑀺 𑀯𑁂𑀤𑀦𑀸𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑀸 𑀢𑀻𑀳𑀺 𑀯𑀺𑀢𑀓𑁆𑀓𑁂𑀳𑀺 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀺—  𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀢 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀬𑁄 𑀇𑀫𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺; 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀯𑀢 𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀲𑀸𑀯𑀓𑀲𑀁𑀖𑁄 𑀬𑁄 𑀇𑀫𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄; 𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀬𑀢𑁆𑀣𑀺𑀤𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀻’”𑀢𑀺𑁇

119. “𑀲𑀼𑀔𑀺𑀦𑀻 𑀳𑁄𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸; 𑀅𑀭𑁄𑀕𑀸 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀬𑀢𑀽”𑀢𑀺𑁇 𑀲𑀳 𑀯𑀘𑀦𑀸 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀼𑀔𑀺𑀦𑀻 𑀅𑀭𑁄𑀕𑀸 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀬𑀺𑁇

120. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀓𑀁 𑀖𑀭𑀁 𑀢𑁂𑀦 𑀧𑀘𑁆𑀘𑀸𑀬𑀸𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑁄 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀭𑀁 𑀲𑀼𑀔𑀺𑀦𑀺𑀁 𑀅𑀭𑁄𑀕𑀁 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀸, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫𑀸𑀬𑀁 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀳 𑀯𑀘𑀦𑀸 𑀘 𑀧𑀦 𑀪𑀕𑀯𑀢𑁄 𑀲𑀼𑀔𑀺𑀦𑀻 𑀅𑀭𑁄𑀕𑀸 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀧𑀫𑀼𑀤𑀺𑀢𑁄 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇

121. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀸𑀫𑀺𑀓𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀸𑀳𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀳𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀕𑀩𑁆𑀪𑀁 𑀥𑀸𑀭𑁂𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀽𑀍𑀳𑀕𑀩𑁆𑀪𑀸 𑀲𑀸 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀔𑀺𑀦𑀻 𑀅𑀭𑁄𑀕𑀸 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀢𑀸𑁇 𑀲𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸𑀬 𑀲𑀢𑁆𑀢 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸’”𑀢𑀺𑁇

122. “𑀧𑀭𑀫𑀦𑁆”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀓𑁄𑀮𑀺𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

123. “𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺; 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀢𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀕𑀩𑁆𑀪𑀁 𑀥𑀸𑀭𑁂𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀽𑀍𑀳𑀕𑀩𑁆𑀪𑀸 𑀲𑀸 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀔𑀺𑀦𑀻 𑀅𑀭𑁄𑀕𑀸 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀢𑀸𑁇 𑀲𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀓𑀺𑀭, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸𑀬 𑀲𑀢𑁆𑀢 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸’”𑀢𑀺𑁇

124. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘 𑀉𑀧𑀸𑀲𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦, 𑀬𑁂𑀦 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀺—  ‘𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀕𑀩𑁆𑀪𑀁 𑀥𑀸𑀭𑁂𑀲𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀽𑀍𑀳𑀕𑀩𑁆𑀪𑀸 𑀲𑀸 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀔𑀺𑀦𑀻 𑀅𑀭𑁄𑀕𑀸 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀢𑀸𑁇 𑀲𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺𑁇 𑀓𑀭𑁄𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀪𑀢𑁆𑀢𑀸𑀦𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀢𑁆𑀯𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇 𑀢𑀼𑀬𑁆𑀳𑁂𑀲𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄”𑀢𑀺𑁇

125. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀕𑀩𑁆𑀪𑀁 𑀥𑀸𑀭𑁂𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀫𑀽𑀍𑀳𑀕𑀩𑁆𑀪𑀸 𑀲𑀸 𑀏𑀢𑀭𑀳𑀺 𑀲𑀼𑀔𑀺𑀦𑀻 𑀅𑀭𑁄𑀕𑀸 𑀅𑀭𑁄𑀕𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀢𑀸𑁇 𑀲𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺𑁇 𑀓𑀭𑁄𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀪𑀢𑁆𑀢𑀸𑀦𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀢𑁆𑀯𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺𑁇

126. “𑀲𑀘𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄—  𑀪𑁄𑀕𑀸𑀦𑀜𑁆𑀘 𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲 𑀘 𑀲𑀤𑁆𑀥𑀸𑀬 𑀘, 𑀓𑀭𑁄𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀪𑀢𑁆𑀢𑀸𑀦𑀺, 𑀧𑀘𑁆𑀙𑀸𑀳𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀢𑁂 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄—  𑀪𑁄𑀕𑀸𑀦𑀜𑁆𑀘 𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲 𑀘𑁇 𑀲𑀤𑁆𑀥𑀸𑀬 𑀧𑀦 𑀢𑁆𑀯𑀁𑀬𑁂𑀯 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄”𑀢𑀺𑁇

127. “𑀲𑀘𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀸𑀝𑀺𑀪𑁄𑀕𑁄—  𑀪𑁄𑀕𑀸𑀦𑀜𑁆𑀘 𑀚𑀻𑀯𑀺𑀢𑀲𑁆𑀲 𑀘, 𑀓𑀭𑁄𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀪𑀢𑁆𑀢𑀸𑀦𑀺, 𑀧𑀘𑁆𑀙𑀸𑀳𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

128. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀜𑁆𑀜𑀢𑁆𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀫𑀬𑀸; 𑀓𑀭𑁄𑀢𑀼 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢 𑀪𑀢𑁆𑀢𑀸𑀦𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀲𑁄 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

129. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀲𑀺, 𑀢𑀜𑁆𑀘 𑀤𑀸𑀭𑀓𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀲𑀺 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁𑁇

130. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺 𑀢𑁂, 𑀤𑀸𑀭𑀓, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑀼𑀓𑁆𑀔𑀦𑁆”𑀢𑀺? “𑀓𑀼𑀢𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀼𑀢𑁄 𑀬𑀸𑀧𑀦𑀻𑀬𑀁𑁇 𑀲𑀢𑁆𑀢 𑀫𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀮𑁄𑀳𑀺𑀢𑀓𑀼𑀫𑁆𑀪𑀺𑀬𑀁 𑀯𑀼𑀢𑁆𑀣𑀸𑀦𑀻”𑀢𑀺𑁇

131. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀸 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀸—  “𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑁂 𑀥𑀫𑁆𑀫𑀲𑁂𑀦𑀸𑀧𑀢𑀺𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀫𑀦𑁆𑀢𑁂𑀢𑀻”𑀢𑀺 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀧𑀫𑀼𑀤𑀺𑀢𑀸 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀭𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑀁 𑀧𑀫𑀼𑀤𑀺𑀢𑀁 𑀧𑀻𑀢𑀺𑀲𑁄𑀫𑀦𑀲𑁆𑀲𑀚𑀸𑀢𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀓𑁄𑀮𑀺𑀬𑀥𑀻𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀲𑁂, 𑀅𑀜𑁆𑀜𑀫𑁆𑀧𑀺 𑀏𑀯𑀭𑀽𑀧𑀁 𑀧𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺? “𑀇𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀫𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀅𑀜𑁆𑀜𑀸𑀦𑀺𑀧𑀺 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀻”𑀢𑀺𑁇

132. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

133. _“𑀅𑀲𑀸𑀢𑀁 𑀲𑀸𑀢𑀭𑀽𑀧𑁂𑀦,_  
_𑀧𑀺𑀬𑀭𑀽𑀧𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀬𑀁;_  
_𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀼𑀔𑀲𑁆𑀲 𑀭𑀽𑀧𑁂𑀦,_  
_𑀧𑀫𑀢𑁆𑀢𑀫𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀻”𑀢𑀺𑁇 (𑁨𑁨)_  


---

134. 𑀅𑀝𑁆𑀞𑀫𑀁𑁇
