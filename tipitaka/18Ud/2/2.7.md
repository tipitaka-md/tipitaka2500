# 𑁨𑁇𑁭 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑀲𑀼𑀢𑁆𑀢

### 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀯𑀕𑁆𑀕

### 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑀲𑀼𑀢𑁆𑀢

(17.)

109\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

110\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀸 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀸 𑀤𑀺𑀯𑀸 𑀤𑀺𑀯𑀲𑁆𑀲 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀔𑁄 𑀢𑁂 𑀉𑀧𑀸𑀲𑀓𑁂 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀉𑀧𑀸𑀲𑀓𑀸, 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀸 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀸 𑀇𑀥𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀸 𑀤𑀺𑀯𑀸 𑀤𑀺𑀯𑀲𑁆𑀲𑀸”𑀢𑀺?

111\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀧𑀼𑀢𑁆𑀢𑀓𑁄 𑀧𑀺𑀬𑁄 𑀫𑀦𑀸𑀧𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀢𑁂𑀦 𑀫𑀬𑀁 𑀅𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀸 𑀅𑀮𑁆𑀮𑀓𑁂𑀲𑀸 𑀇𑀥𑀽𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀸 𑀤𑀺𑀯𑀸 𑀤𑀺𑀯𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

112\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

113\. _“𑀧𑀺𑀬𑀭𑀽𑀧𑀲𑁆𑀲𑀸𑀤𑀕𑀥𑀺𑀢𑀸𑀲𑁂,_  
_𑀤𑁂𑀯𑀓𑀸𑀬𑀸 𑀧𑀼𑀣𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘;_  
_𑀅𑀖𑀸𑀯𑀺𑀦𑁄 𑀧𑀭𑀺𑀚𑀼𑀦𑁆𑀦𑀸,_  
_𑀫𑀘𑁆𑀘𑀼𑀭𑀸𑀚𑀲𑁆𑀲 𑀯𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 (𑁨𑁦)_  


114\. _𑀬𑁂 𑀯𑁂 𑀤𑀺𑀯𑀸 𑀘 𑀭𑀢𑁆𑀢𑁄 𑀘,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀸 𑀚𑀳𑀦𑁆𑀢𑀺 𑀧𑀺𑀬𑀭𑀽𑀧𑀁;_  
_𑀢𑁂 𑀯𑁂 𑀔𑀡𑀦𑁆𑀢𑀺 𑀅𑀖𑀫𑀽𑀮𑀁,_  
_𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀆𑀫𑀺𑀲𑀁 𑀤𑀼𑀭𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 (𑁨𑁧)_  


---
