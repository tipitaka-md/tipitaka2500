# 𑁪𑁇𑁫 𑀦𑀸𑀕𑀲𑀼𑀢𑁆𑀢

* 𑀫𑁂𑀖𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀦𑀸𑀕𑀲𑀼𑀢𑁆𑀢

(35.)

299\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀖𑁄𑀲𑀺𑀢𑀸𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀳𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁂𑀳𑀺𑁇 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀏𑀢𑀭𑀳𑀺 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀳𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁂𑀳𑀺𑁇 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀏𑀓𑁄 𑀕𑀡𑀲𑁆𑀫𑀸 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

300\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑀸𑀫𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀅𑀦𑀸𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀁 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀏𑀓𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀬𑁂𑀦 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀁 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑀫𑀽𑀮𑁂𑁇

301\. 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑀧𑀺 𑀔𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀮𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀺𑁇 𑀙𑀺𑀦𑁆𑀦𑀕𑁆𑀕𑀸𑀦𑀺 𑀘𑁂𑀯 𑀢𑀺𑀡𑀸𑀦𑀺 𑀔𑀸𑀤𑀢𑀺, 𑀑𑀪𑀕𑁆𑀕𑁄𑀪𑀕𑁆𑀕𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀔𑀸𑀪𑀗𑁆𑀕𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀺, 𑀆𑀯𑀺𑀮𑀸𑀦𑀺 𑀘 𑀧𑀸𑀦𑀻𑀬𑀸𑀦𑀺 𑀧𑀺𑀯𑀢𑀺, 𑀑𑀕𑀸𑀳𑀸 𑀘𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀺𑀖𑀁𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀏𑀢𑀭𑀳𑀺 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀮𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀺, 𑀙𑀺𑀦𑁆𑀦𑀕𑁆𑀕𑀸𑀦𑀺 𑀘𑁂𑀯 𑀢𑀺𑀡𑀸𑀦𑀺 𑀔𑀸𑀤𑀸𑀫𑀺, 𑀑𑀪𑀕𑁆𑀕𑁄𑀪𑀕𑁆𑀕𑀜𑁆𑀘 𑀫𑁂 𑀲𑀸𑀔𑀸𑀪𑀗𑁆𑀕𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀺, 𑀆𑀯𑀺𑀮𑀸𑀦𑀺 𑀘 𑀧𑀸𑀦𑀻𑀬𑀸𑀦𑀺 𑀧𑀺𑀯𑀸𑀫𑀺, 𑀑𑀕𑀸𑀳𑀸 𑀘 𑀫𑁂 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀺𑀖𑀁𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀸𑀫𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀏𑀓𑁄 𑀕𑀡𑀲𑁆𑀫𑀸 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

302\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀬𑀽𑀣𑀸 𑀅𑀧𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀁 𑀭𑀓𑁆𑀔𑀺𑀢𑀯𑀦𑀲𑀡𑁆𑀟𑁄 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑀫𑀽𑀮𑀁 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀲𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑀁 𑀧𑀤𑁂𑀲𑀁 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀲𑁄𑀡𑁆𑀟𑀸𑀬 𑀘 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺𑁇

303\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀳𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁂𑀳𑀺, 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀲𑁄𑀫𑁆𑀳𑀺 𑀏𑀢𑀭𑀳𑀺 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀳𑀺 𑀭𑀸𑀚𑀽𑀳𑀺 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀳𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑁂𑀳𑀺, 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑁄 𑀲𑀼𑀔𑀁 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀸𑀫𑀻”𑀢𑀺𑁇

304\. 𑀢𑀲𑁆𑀲𑀧𑀺 𑀔𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀮𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀺, 𑀙𑀺𑀦𑁆𑀦𑀕𑁆𑀕𑀸𑀦𑀺 𑀘𑁂𑀯 𑀢𑀺𑀡𑀸𑀦𑀺 𑀔𑀸𑀤𑀺𑀁, 𑀑𑀪𑀕𑁆𑀕𑁄𑀪𑀕𑁆𑀕𑀜𑁆𑀘 𑀫𑁂 𑀲𑀸𑀔𑀸𑀪𑀗𑁆𑀕𑀁 𑀔𑀸𑀤𑀺𑀁𑀲𑀼, 𑀆𑀯𑀺𑀮𑀸𑀦𑀺 𑀘 𑀧𑀸𑀦𑀻𑀬𑀸𑀦𑀺 𑀅𑀧𑀸𑀬𑀺𑀁, 𑀑𑀕𑀸𑀳𑀸 𑀘 𑀫𑁂 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀺𑀖𑀁𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀅𑀕𑀫𑀁𑀲𑀼, 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀤𑀼𑀓𑁆𑀔𑀁 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀸𑀲𑀺𑀁𑁇 𑀲𑁄𑀫𑁆𑀳𑀺 𑀏𑀢𑀭𑀳𑀺 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀮𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀺, 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀕𑁆𑀕𑀸𑀦𑀺 𑀘𑁂𑀯 𑀢𑀺𑀡𑀸𑀦𑀺 𑀔𑀸𑀤𑀸𑀫𑀺, 𑀑𑀪𑀕𑁆𑀕𑁄𑀪𑀕𑁆𑀕𑀜𑁆𑀘 𑀫𑁂 𑀲𑀸𑀔𑀸𑀪𑀗𑁆𑀕𑀁 𑀦 𑀔𑀸𑀤𑀦𑁆𑀢𑀺, 𑀅𑀦𑀸𑀯𑀺𑀮𑀸𑀦𑀺 𑀘 𑀧𑀸𑀦𑀻𑀬𑀸𑀦𑀺 𑀧𑀺𑀯𑀸𑀫𑀺, 𑀑𑀕𑀸𑀳𑀸 𑀘 𑀫𑁂 𑀉𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀺𑀬𑁄 𑀦 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀺𑀖𑀁𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑀸𑀓𑀺𑀡𑁆𑀡𑁄 𑀲𑀼𑀔𑀁 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀸𑀫𑀻”𑀢𑀺𑁇

305\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀘 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

306\. _“𑀏𑀢𑀁 𑀦𑀸𑀕𑀲𑁆𑀲 𑀦𑀸𑀕𑁂𑀦,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑁄;_  
_𑀲𑀫𑁂𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀬𑀤𑁂𑀓𑁄 𑀭𑀫𑀢𑀻 𑀫𑀦𑁄”𑀢𑀺𑁇 (𑁪𑁪)_
