# 𑁪𑁇𑁯 𑀉𑀧𑀲𑁂𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀫𑁂𑀖𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀲𑁂𑀦𑀲𑀼𑀢𑁆𑀢

(39.)

338\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑁂𑀦𑀲𑁆𑀲 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀲𑀢𑁆𑀣𑀸 𑀘 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄; 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀘𑀫𑁆𑀳𑀺 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄; 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀘 𑀫𑁂 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸; 𑀲𑀻𑀮𑁂𑀲𑀼 𑀘𑀫𑁆𑀳𑀺 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻; 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁄 𑀘𑀫𑁆𑀳𑀺 𑀏𑀓𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄; 𑀅𑀭𑀳𑀸 𑀘𑀫𑁆𑀳𑀺 𑀔𑀻𑀡𑀸𑀲𑀯𑁄; 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀘𑀫𑁆𑀳𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄𑁇 𑀪𑀤𑁆𑀤𑀓𑀁 𑀫𑁂 𑀚𑀻𑀯𑀺𑀢𑀁, 𑀪𑀤𑁆𑀤𑀓𑀁 𑀫𑀭𑀡𑀦𑁆”𑀢𑀺𑁇

339\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑁂𑀦𑀲𑁆𑀲 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

340\. _“𑀬𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀦 𑀢𑀧𑀢𑀺,_  
_𑀫𑀭𑀡𑀦𑁆𑀢𑁂 𑀦 𑀲𑁄𑀘𑀢𑀺;_  
_𑀲 𑀯𑁂 𑀤𑀺𑀝𑁆𑀞𑀧𑀤𑁄 𑀥𑀻𑀭𑁄,_  
_𑀲𑁄𑀓𑀫𑀚𑁆𑀛𑁂 𑀦 𑀲𑁄𑀘𑀢𑀺𑁇 (𑁪𑁮)_  


341\. _𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀪𑀯𑀢𑀡𑁆𑀳𑀲𑁆𑀲,_  
_𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀯𑀺𑀓𑁆𑀔𑀻𑀡𑁄 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀢𑀲𑁆𑀲 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇 (𑁪𑁯)_
