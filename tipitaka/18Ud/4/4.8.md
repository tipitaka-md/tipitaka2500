# 𑁪𑁇𑁮 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀲𑀼𑀢𑁆𑀢

* 𑀫𑁂𑀖𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀲𑀼𑀢𑁆𑀢

(38.)

318\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄𑀧𑀺 𑀲𑀓𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺 𑀕𑀭𑀼𑀓𑀢𑁄 𑀫𑀸𑀦𑀺𑀢𑁄 𑀧𑀽𑀚𑀺𑀢𑁄 𑀅𑀧𑀘𑀺𑀢𑁄 𑀮𑀸𑀪𑀻 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀅𑀲𑀓𑁆𑀓𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀕𑀭𑀼𑀓𑀢𑀸 𑀅𑀫𑀸𑀦𑀺𑀢𑀸 𑀅𑀧𑀽𑀚𑀺𑀢𑀸 𑀅𑀦𑀧𑀘𑀺𑀢𑀸 𑀦 𑀮𑀸𑀪𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸𑀦𑀁𑁇

319\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀓𑁆𑀓𑀸𑀭𑀁 𑀅𑀲𑀳𑀫𑀸𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀘 𑀬𑁂𑀦 𑀲𑀼𑀦𑁆𑀤𑀭𑀻 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀦𑁆𑀤𑀭𑀺𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀉𑀲𑁆𑀲𑀳𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀕𑀺𑀦𑀺, 𑀜𑀸𑀢𑀻𑀦𑀁 𑀅𑀢𑁆𑀣𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀓𑁆𑀬𑀸𑀳𑀁, 𑀅𑀬𑁆𑀬𑀸, 𑀓𑀭𑁄𑀫𑀺? 𑀓𑀺𑀁 𑀫𑀬𑀸 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀓𑀸𑀢𑀼𑀁? 𑀚𑀻𑀯𑀺𑀢𑀫𑁆𑀧𑀺 𑀫𑁂 𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀁 𑀜𑀸𑀢𑀻𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬𑀸”𑀢𑀺𑁇

320\. “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀕𑀺𑀦𑀺, 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀕𑀘𑁆𑀙𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑀸”𑀢𑀺 𑀔𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀻 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸 𑀢𑁂𑀲𑀁 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸𑀦𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇

321\. 𑀬𑀤𑀸 𑀢𑁂 𑀅𑀜𑁆𑀜𑀺𑀁𑀲𑀼 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸—  “𑀯𑁄𑀤𑀺𑀝𑁆𑀞𑀸 𑀔𑁄 𑀲𑀼𑀦𑁆𑀤𑀭𑀻 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸 𑀩𑀳𑀼𑀚𑀦𑁂𑀦 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁 𑀚𑁂𑀢𑀯𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀦𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀚𑁂𑀢𑀯𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀔𑀸𑀓𑀽𑀧𑁂 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀧𑀲𑁂𑀦𑀤𑀺𑀁 𑀓𑁄𑀲𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀬𑀸 𑀲𑀸, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑀼𑀦𑁆𑀤𑀭𑀻 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸; 𑀲𑀸 𑀦𑁄 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀓𑀢𑁆𑀣 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀲𑀗𑁆𑀓𑀣𑀸”𑀢𑀺? “𑀚𑁂𑀢𑀯𑀦𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀚𑁂𑀢𑀯𑀦𑀁 𑀯𑀺𑀘𑀺𑀦𑀣𑀸”𑀢𑀺𑁇

322\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀚𑁂𑀢𑀯𑀦𑀁 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀔𑀸𑀓𑀽𑀧𑀸 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀜𑁆𑀘𑀓𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀯𑁂𑀲𑁂𑀢𑁆𑀯𑀸 𑀭𑀣𑀺𑀬𑀸𑀬 𑀭𑀣𑀺𑀬𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀲𑀼𑀁—

323\. “𑀧𑀲𑁆𑀲𑀣𑀸𑀬𑁆𑀬𑀸 𑀲𑀫𑀡𑀸𑀦𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀁𑁇 𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁𑁇 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁𑁇 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁? 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀲𑀸𑀫𑀜𑁆𑀜𑀸, 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀸𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀓𑀺𑀘𑁆𑀘𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀇𑀢𑁆𑀣𑀺𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

324\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑀦𑁆𑀢𑀺—

325\. “𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁𑁇 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁𑁇 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁? 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀲𑀸𑀫𑀜𑁆𑀜𑀸, 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀸𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀓𑀺𑀘𑁆𑀘𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀇𑀢𑁆𑀣𑀺𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

326\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—

327\. “𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑀦𑁆𑀢𑀺—  ‘𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀧𑀸𑀧𑀥𑀫𑁆𑀫𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀦𑀢𑁆𑀣𑀺 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁𑁇 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀦𑀝𑁆𑀞𑀁 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁𑁇 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀁, 𑀓𑀼𑀢𑁄 𑀇𑀫𑁂𑀲𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀁? 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀲𑀸𑀫𑀜𑁆𑀜𑀸, 𑀅𑀧𑀕𑀢𑀸 𑀇𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀜𑁆𑀜𑀸𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑀓𑀺𑀘𑁆𑀘𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀇𑀢𑁆𑀣𑀺𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑁆𑀲𑀢𑀻’”𑀢𑀺𑁇

328\. “𑀦𑁂𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀤𑁄 𑀘𑀺𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀳𑀫𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀇𑀫𑀸𑀬 𑀕𑀸𑀣𑀸𑀬 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀣—

329\. _‘𑀅𑀪𑀽𑀢𑀯𑀸𑀤𑀻 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑁂𑀢𑀺,_  
_𑀬𑁄 𑀯𑀸𑀧𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦 𑀓𑀭𑁄𑀫𑀺 𑀘𑀸𑀳;_  
_𑀉𑀪𑁄𑀧𑀺 𑀢𑁂 𑀧𑁂𑀘𑁆𑀘 𑀲𑀫𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀳𑀻𑀦𑀓𑀫𑁆𑀫𑀸 𑀫𑀦𑀼𑀚𑀸 𑀧𑀭𑀢𑁆𑀣𑀸’”𑀢𑀺𑁇_  


330\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀕𑀸𑀣𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀬𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀺𑀲𑁆𑀯𑀸 𑀅𑀲𑀩𑁆𑀪𑀸𑀳𑀺 𑀨𑀭𑀼𑀲𑀸𑀳𑀺 𑀯𑀸𑀘𑀸𑀳𑀺 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀺 𑀭𑁄𑀲𑀦𑁆𑀢𑀺 𑀯𑀺𑀳𑁂𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀇𑀫𑀸𑀬 𑀕𑀸𑀣𑀸𑀬 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺—

331\. _“𑀅𑀪𑀽𑀢𑀯𑀸𑀤𑀻 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑁂𑀢𑀺,_  
_𑀬𑁄 𑀯𑀸𑀧𑀺 𑀓𑀢𑁆𑀯𑀸 𑀦 𑀓𑀭𑁄𑀫𑀺 𑀘𑀸𑀳;_  
_𑀉𑀪𑁄𑀧𑀺 𑀢𑁂 𑀧𑁂𑀘𑁆𑀘 𑀲𑀫𑀸 𑀪𑀯𑀦𑁆𑀢𑀺,_  
_𑀦𑀺𑀳𑀻𑀦𑀓𑀫𑁆𑀫𑀸 𑀫𑀦𑀼𑀚𑀸 𑀧𑀭𑀢𑁆𑀣𑀸”𑀢𑀺𑁇_  


332\. 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀓𑀸𑀭𑀓𑀸 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸, 𑀦𑀬𑀺𑀫𑁂𑀳𑀺 𑀓𑀢𑀁, 𑀲𑀧𑀦𑁆𑀢𑀺𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺𑁇 𑀦𑁂𑀯 𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀘𑀺𑀭𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀫𑁂𑀯 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇

333\. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—

334\. “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀬𑀸𑀯 𑀲𑀼𑀪𑀸𑀲𑀺𑀢𑀜𑁆𑀘𑀺𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸—  ‘𑀦𑁂𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀤𑁄 𑀘𑀺𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀫𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀤𑁆𑀤𑁄”𑀢𑀺𑁇

335\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

336\. _“𑀢𑀼𑀤𑀦𑁆𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀚𑀦𑀸 𑀅𑀲𑀜𑁆𑀜𑀢𑀸,_  
_𑀲𑀭𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀸𑀫𑀕𑀢𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑀁;_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀯𑀸𑀓𑁆𑀬𑀁 𑀨𑀭𑀼𑀲𑀁 𑀉𑀤𑀻𑀭𑀺𑀢𑀁,_  
_𑀅𑀥𑀺𑀯𑀸𑀲𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀤𑀼𑀝𑁆𑀞𑀘𑀺𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 (𑁪𑁭)_
