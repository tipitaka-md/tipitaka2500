# 𑁪𑁇𑁭 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

### 𑀫𑁂𑀖𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑀼𑀢𑁆𑀢

(37.)

313\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑁄 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑁄 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑁄 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄𑁇

314\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀁 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀁 𑀧𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀁 𑀅𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑀁 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀁𑁇

315\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

316\. _“𑀅𑀥𑀺𑀘𑁂𑀢𑀲𑁄 𑀅𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑁄,_  
_𑀫𑀼𑀦𑀺𑀦𑁄 𑀫𑁄𑀦𑀧𑀣𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀢𑁄;_  
_𑀲𑁄𑀓𑀸 𑀦 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑀸𑀤𑀺𑀦𑁄,_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀤𑀸 𑀲𑀢𑀻𑀫𑀢𑁄”𑀢𑀺𑁇 (𑁪𑁬)_
