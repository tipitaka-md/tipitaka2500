# 𑁪𑁇𑁧𑁦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁅𑀧𑀲𑀫𑀲𑀼𑀢𑁆𑀢

### 𑀫𑁂𑀖𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁅𑀧𑀲𑀫𑀲𑀼𑀢𑁆𑀢

(40.)

343. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑁄 𑀉𑀧𑀲𑀫𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑁄𑁇

344. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑁄 𑀉𑀧𑀲𑀫𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀁𑁇

345. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

346. _“𑀉𑀧𑀲𑀦𑁆𑀢𑀲𑀦𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀦𑁂𑀢𑁆𑀢𑀺𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄;_  
_𑀯𑀺𑀓𑁆𑀔𑀻𑀡𑁄 𑀚𑀸𑀢𑀺𑀲𑀁𑀲𑀸𑀭𑁄,_  
_𑀫𑀼𑀢𑁆𑀢𑁄 𑀲𑁄 𑀫𑀸𑀭𑀩𑀦𑁆𑀥𑀦𑀸”𑀢𑀺𑁇 (𑁫𑁦)_

348. 𑀫𑁂𑀖𑀺𑀬𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇

349. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



350. _𑀫𑁂𑀖𑀺𑀬𑁄 𑀉𑀤𑁆𑀥𑀢𑀸 𑀕𑁄𑀧𑀸𑀮𑁄,_  
_𑀬𑀓𑁆𑀔𑁄 𑀦𑀸𑀕𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀁;_  
_𑀧𑀺𑀡𑁆𑀟𑁄𑀮𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀘,_  
_𑀲𑀼𑀦𑁆𑀤𑀭𑀻 𑀪𑀯𑀢𑀺 𑀅𑀝𑁆𑀞𑀫𑀁;_  
_𑀉𑀧𑀲𑁂𑀦𑁄 𑀯𑀗𑁆𑀕𑀦𑁆𑀢𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀢𑁂 𑀤𑀲𑀸𑀢𑀺𑁇_
