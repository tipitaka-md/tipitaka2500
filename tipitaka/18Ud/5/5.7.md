# 𑁫𑁇𑁭 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀡𑀯𑀕𑁆𑀕

* 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑀲𑀼𑀢𑁆𑀢

(47.)

433\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑁄 𑀓𑀗𑁆𑀔𑀸𑀯𑀺𑀢𑀭𑀡𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑁄𑁇

434\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑀁 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀅𑀢𑁆𑀢𑀦𑁄 𑀓𑀗𑁆𑀔𑀸𑀯𑀺𑀢𑀭𑀡𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀁𑁇

435\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

436\. _“𑀬𑀸 𑀓𑀸𑀘𑀺 𑀓𑀗𑁆𑀔𑀸 𑀇𑀥 𑀯𑀸 𑀳𑀼𑀭𑀁 𑀯𑀸,_  
_𑀲𑀓𑀯𑁂𑀤𑀺𑀬𑀸 𑀯𑀸 𑀧𑀭𑀯𑁂𑀤𑀺𑀬𑀸 𑀯𑀸;_  
_𑀬𑁂 𑀛𑀸𑀬𑀺𑀦𑁄 𑀢𑀸 𑀧𑀚𑀳𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀸,_  
_𑀆𑀢𑀸𑀧𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀸”𑀢𑀺𑁇 (𑁫𑁮)_
