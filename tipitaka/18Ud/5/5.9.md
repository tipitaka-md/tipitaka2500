# 𑁫𑁇𑁯 𑀲𑀥𑀸𑀬𑀫𑀸𑀦𑀲𑀼𑀢𑁆𑀢

* 𑀲𑁄𑀡𑀯𑀕𑁆𑀕

* 𑀲𑀥𑀸𑀬𑀫𑀸𑀦𑀲𑀼𑀢𑁆𑀢

(49.)

445\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀢𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀫𑀸𑀡𑀯𑀓𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀲𑀥𑀸𑀬𑀫𑀸𑀦𑀭𑀽𑀧𑀸 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀫𑀸𑀡𑀯𑀓𑁂 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀲𑀥𑀸𑀬𑀫𑀸𑀦𑀭𑀽𑀧𑁂 𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑁇

446\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

447\. _“𑀧𑀭𑀺𑀫𑀼𑀝𑁆𑀞𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑀪𑀸𑀲𑀸,_  
_𑀯𑀸𑀘𑀸𑀕𑁄𑀘𑀭𑀪𑀸𑀡𑀺𑀦𑁄;_  
_𑀬𑀸𑀯𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀼𑀔𑀸𑀬𑀸𑀫𑀁,_  
_𑀬𑁂𑀦 𑀦𑀻𑀢𑀸 𑀦 𑀢𑀁 𑀯𑀺𑀤𑀽”𑀢𑀺𑁇 (𑁬𑁦)_
