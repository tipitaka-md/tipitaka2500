# 𑁫𑁇𑁬 𑀲𑁄𑀡𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀡𑀯𑀕𑁆𑀕

### 𑀲𑁄𑀡𑀲𑀼𑀢𑁆𑀢

(46.)

412\. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀅𑀯𑀦𑁆𑀢𑀻𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀼𑀭𑀭𑀖𑀭𑁂 𑀧𑀯𑀢𑁆𑀢𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

413\. 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

414\. 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

415\. “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄”𑀢𑀺𑁇

416\. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑁄𑀡𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀔𑁄, 𑀲𑁄𑀡, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀏𑀓𑀪𑀢𑁆𑀢𑀁 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁𑁇 𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀆𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚 𑀓𑀸𑀮𑀬𑀼𑀢𑁆𑀢𑀁 𑀏𑀓𑀪𑀢𑁆𑀢𑀁 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀲𑁆𑀲 𑀬𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑁄 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇

417\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑁄𑀡𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀔𑁄, 𑀲𑁄𑀡, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀏𑀓𑀪𑀢𑁆𑀢𑀁 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁𑁇 𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀆𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚 𑀓𑀸𑀮𑀬𑀼𑀢𑁆𑀢𑀁 𑀏𑀓𑀪𑀢𑁆𑀢𑀁 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀲𑁆𑀲 𑀬𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑁄 𑀧𑀝𑀺𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇

418\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

419\. “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀬𑀣𑀸 𑀬𑀣𑀸 𑀔𑁄 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄”𑀢𑀺𑁇

420\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑁄𑀡𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁄 𑀅𑀧𑁆𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀓𑀲𑀺𑀭𑁂𑀦 𑀢𑀢𑁄 𑀢𑀢𑁄 𑀤𑀲𑀯𑀕𑁆𑀕𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁄𑀡𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑀺𑁇

421\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀦 𑀔𑁄 𑀫𑁂 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀅𑀧𑀺 𑀘 𑀲𑀼𑀢𑁄𑀬𑁂𑀯 𑀫𑁂—  ‘𑀲𑁄 𑀪𑀕𑀯𑀸 𑀈𑀤𑀺𑀲𑁄 𑀘 𑀈𑀤𑀺𑀲𑁄 𑀘𑀸’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀫𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀳𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀦𑁆”𑀢𑀺𑁇

422\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

423\. “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀦 𑀔𑁄 𑀫𑁂 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀅𑀧𑀺 𑀘 𑀲𑀼𑀢𑁄𑀬𑁂𑀯 𑀫𑁂—  𑀲𑁄 𑀪𑀕𑀯𑀸 𑀈𑀤𑀺𑀲𑁄 𑀘 𑀈𑀤𑀺𑀲𑁄 𑀘𑀸’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀫𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀳𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀦𑁆”𑀢𑀺𑁇 ( )

424\. “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑁄𑀡; 𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁𑁇 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀧𑀲𑀸𑀤𑀦𑀻𑀬𑀁 𑀲𑀦𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀉𑀢𑁆𑀢𑀫𑀤𑀫𑀣𑀲𑀫𑀣𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁 𑀤𑀦𑁆𑀢𑀁 𑀕𑀼𑀢𑁆𑀢𑀁 𑀬𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀸𑀕𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀸𑀳𑀺, 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀁 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀁 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀩𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀼𑀘𑁆𑀙—  ‘𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺, 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀁 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀁 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀩𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀻’”𑀢𑀺𑁇

425\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄, 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺, 𑀅𑀧𑁆𑀧𑀸𑀩𑀸𑀥𑀁 𑀅𑀧𑁆𑀧𑀸𑀢𑀗𑁆𑀓𑀁 𑀮𑀳𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀩𑀮𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇

426\. “𑀓𑀘𑁆𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺𑀲𑀺 𑀅𑀧𑁆𑀧𑀓𑀺𑀮𑀫𑀣𑁂𑀦 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀕𑀢𑁄, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑀲𑀻”𑀢𑀺? “𑀔𑀫𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸, 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸, 𑀅𑀧𑁆𑀧𑀓𑀺𑀮𑀫𑀣𑁂𑀦 𑀘𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀆𑀕𑀢𑁄, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀦𑁆𑀢𑁄𑀫𑁆𑀳𑀻”𑀢𑀺𑁇

427\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀫𑀲𑁆𑀲𑀸𑀦𑀦𑁆𑀤, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀲𑁆𑀲 𑀔𑁄 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀆𑀡𑀸𑀧𑁂𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲𑀸𑀦𑀦𑁆𑀤, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀻’𑀢𑀺, 𑀇𑀘𑁆𑀙𑀢𑀺 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀼𑀁, 𑀇𑀘𑁆𑀙𑀢𑀺 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑁄𑀡𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀼𑀦𑁆”𑀢𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺𑁇

428\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀳𑀼𑀤𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀩𑀳𑀼𑀤𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁄𑀡𑀁 𑀅𑀚𑁆𑀛𑁂𑀲𑀺—  “𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

429\. “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀲𑁄𑀍𑀅𑀲 𑀅𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕𑀺𑀓𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀦𑁂𑀯 𑀲𑀭𑁂𑀦 𑀅𑀪𑀡𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀲𑀭𑀪𑀜𑁆𑀜𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀅𑀩𑁆𑀪𑀦𑀼𑀫𑁄𑀤𑀺—  “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀲𑁄𑀍𑀅𑀲 𑀅𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕𑀺𑀓𑀸𑀦𑀺 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀸𑀦𑀺 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀸𑀦𑀺, 𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀬𑀸𑀲𑀺 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬 𑀅𑀦𑁂𑀮𑀕𑀍𑀆𑀬 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸𑁇 𑀓𑀢𑀺 𑀯𑀲𑁆𑀲𑁄𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀏𑀓𑀯𑀲𑁆𑀲𑁄 𑀅𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀯𑀁 𑀘𑀺𑀭𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺? “𑀘𑀺𑀭𑀁 𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑁄; 𑀅𑀧𑀺 𑀘 𑀲𑀫𑁆𑀩𑀸𑀥𑁄 𑀖𑀭𑀸𑀯𑀸𑀲𑁄 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄”𑀢𑀺𑁇

430\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

431\. _“𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀤𑀻𑀦𑀯𑀁 𑀮𑁄𑀓𑁂,_  
_𑀜𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁;_  
_𑀅𑀭𑀺𑀬𑁄 𑀦 𑀭𑀫𑀢𑀻 𑀧𑀸𑀧𑁂,_  
_𑀧𑀸𑀧𑁂 𑀦 𑀭𑀫𑀢𑀻 𑀲𑀼𑀘𑀻”𑀢𑀺𑁇 (𑁫𑁭)_
