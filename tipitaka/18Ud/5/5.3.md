# 𑁫𑁇𑁩 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀓𑀼𑀝𑁆𑀞𑀺𑀲𑀼𑀢𑁆𑀢

### 𑀲𑁄𑀡𑀯𑀕𑁆𑀕

### 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀓𑀼𑀝𑁆𑀞𑀺𑀲𑀼𑀢𑁆𑀢

(43.)

363. 𑀏𑀯𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁—  𑀏𑀓𑀁 𑀲𑀫𑀬𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑀸𑀫 𑀓𑀼𑀝𑁆𑀞𑀻 𑀅𑀳𑁄𑀲𑀺—  𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑀮𑀺𑀤𑁆𑀤𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀧𑀡𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀭𑀸𑀓𑁄𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀫𑀳𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇

364. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀢𑀁 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀏𑀢𑁆𑀣 𑀓𑀺𑀜𑁆𑀘𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑀸𑀚𑀻𑀬𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑁂𑀦 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀁𑁇 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫𑁂𑀢𑁆𑀣 𑀓𑀺𑀜𑁆𑀘𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀮𑀪𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

365. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀬𑁂𑀦 𑀲𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀫𑀳𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀭𑀺𑀯𑀼𑀢𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦 𑀔𑁄 𑀏𑀢𑁆𑀣 𑀓𑀺𑀜𑁆𑀘𑀺 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑀸𑀚𑀻𑀬𑀢𑀺𑁇 𑀲𑀫𑀡𑁄 𑀅𑀬𑀁 𑀕𑁄𑀢𑀫𑁄 𑀧𑀭𑀺𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀼𑀡𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑁂𑀯 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺—  “𑀅𑀳𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑁄𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇

366. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀩𑁆𑀩𑀸𑀯𑀦𑁆𑀢𑀁 𑀧𑀭𑀺𑀲𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄 𑀧𑀭𑀺𑀘𑁆𑀘 𑀫𑀦𑀲𑀸𑀓𑀸𑀲𑀺 “𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀇𑀥 𑀪𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀼𑀦𑁆”𑀢𑀺? 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀁 𑀓𑀼𑀝𑁆𑀞𑀺𑀁 𑀢𑀲𑁆𑀲𑀁 𑀧𑀭𑀺𑀲𑀸𑀬𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀬𑀁 𑀔𑁄 𑀇𑀥 𑀪𑀩𑁆𑀩𑁄 𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀁 𑀓𑀼𑀝𑁆𑀞𑀺𑀁 𑀆𑀭𑀩𑁆𑀪 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁; 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁; 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀁 𑀓𑀼𑀝𑁆𑀞𑀺𑀁 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑀁 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑀁, 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀓𑀼𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇

367. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁄 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑁄 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑁄 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀣𑀼 𑀲𑀸𑀲𑀦𑁂 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

368. “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

369. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑀁 𑀓𑀼𑀝𑁆𑀞𑀺𑀁 𑀕𑀸𑀯𑀻 𑀢𑀭𑀼𑀡𑀯𑀘𑁆𑀙𑀸 𑀅𑀥𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇

370. 𑀅𑀣 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀬𑁄 𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀦𑀸𑀫 𑀓𑀼𑀝𑁆𑀞𑀻 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄, 𑀲𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺, 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄”𑀢𑀺?

371. “𑀧𑀡𑁆𑀟𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻; 𑀧𑀘𑁆𑀘𑀧𑀸𑀤𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀸𑀦𑀼𑀥𑀫𑁆𑀫𑀁; 𑀦 𑀘 𑀫𑀁 𑀥𑀫𑁆𑀫𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀳𑁂𑀲𑁂𑀲𑀺𑁇 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀝𑁆𑀞𑀻 𑀢𑀺𑀡𑁆𑀡𑀁 𑀲𑀁𑀬𑁄𑀚𑀦𑀸𑀦𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀸 𑀲𑁄𑀢𑀸𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀥𑀫𑁆𑀫𑁄 𑀦𑀺𑀬𑀢𑁄 𑀲𑀫𑁆𑀩𑁄𑀥𑀺𑀧𑀭𑀸𑀬𑀡𑁄”𑀢𑀺𑁇

372. 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑁂𑀢𑀼, 𑀓𑁄 𑀧𑀘𑁆𑀘𑀬𑁄 𑀬𑁂𑀦 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀅𑀳𑁄𑀲𑀺—  𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑀮𑀺𑀤𑁆𑀤𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀧𑀡𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀭𑀸𑀓𑁄”𑀢𑀺?

373. “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀧𑁆𑀧𑀩𑀼𑀤𑁆𑀥𑁄 𑀓𑀼𑀝𑁆𑀞𑀻 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀲𑁄 𑀉𑀬𑁆𑀬𑀸𑀦𑀪𑀽𑀫𑀺𑀁 𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀢𑀕𑀭𑀲𑀺𑀔𑀺𑀁 𑀧𑀘𑁆𑀘𑁂𑀓𑀩𑀼𑀤𑁆𑀥𑀁 𑀦𑀕𑀭𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑁆𑀯𑀸𑀬𑀁 𑀓𑀼𑀝𑁆𑀞𑀻 𑀓𑀼𑀝𑁆𑀞𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀘𑀭𑀢𑀻’𑀢𑀺? 𑀦𑀺𑀝𑁆𑀞𑀼𑀪𑀺𑀢𑁆𑀯𑀸 𑀅𑀧𑀲𑀩𑁆𑀬𑀢𑁄 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀲𑁄 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑁂𑀦 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀭𑀬𑁂 𑀧𑀘𑁆𑀘𑀺𑀢𑁆𑀣𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀓𑀸𑀯𑀲𑁂𑀲𑁂𑀦 𑀇𑀫𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀓𑀼𑀝𑁆𑀞𑀻 𑀅𑀳𑁄𑀲𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀤𑀮𑀺𑀤𑁆𑀤𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀓𑀧𑀡𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀭𑀸𑀓𑁄𑁇 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑀤𑁆𑀥𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺 𑀲𑀼𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺 𑀘𑀸𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺 𑀧𑀜𑁆𑀜𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑁇 𑀲𑁄 𑀢𑀣𑀸𑀕𑀢𑀧𑁆𑀧𑀯𑁂𑀤𑀺𑀢𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑀁 𑀆𑀕𑀫𑁆𑀫 𑀲𑀤𑁆𑀥𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀻𑀮𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀢𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀦𑁆𑀦𑁄 𑀤𑁂𑀯𑀸𑀦𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸𑀦𑀁 𑀲𑀳𑀩𑁆𑀬𑀢𑀁𑁇 𑀲𑁄 𑀢𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑁂 𑀤𑁂𑀯𑁂 𑀅𑀢𑀺𑀭𑁄𑀘𑀢𑀺 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑁂𑀯 𑀬𑀲𑀲𑀸 𑀘𑀸”𑀢𑀺𑁇

374. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

375. _“𑀘𑀓𑁆𑀔𑀼𑀫𑀸 𑀯𑀺𑀲𑀫𑀸𑀦𑀻𑀯,_  
_𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑁂 𑀧𑀭𑀓𑁆𑀓𑀫𑁂;_  
_𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀚𑀻𑀯𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀸𑀧𑀸𑀦𑀺 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑀬𑁂”𑀢𑀺𑁇 (𑁫𑁩)_
