# 𑁫𑁇𑁧𑁪 𑀲𑀫𑀣𑀸𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭

### 𑀲𑀫𑀣𑀪𑁂𑀤

### 𑀲𑀫𑀣𑀸𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭

(309.)

1363\. 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺?

1364\. 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇

(310.)

1365\. 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁧)

1366\. 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀸𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁨)

1367\. 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁩)

1368\. 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁪)

1369\. 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁫)

1370\. 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 (𑁬)

(311.)

1371\. 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁧)

1372\. 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁨)

1373\. 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁩)

1374\. 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁪)

1375\. 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁫)

1376\. 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁬)

1377\. 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 (𑁭)

(312.)

1378\. 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁧)

1379\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁨)

1380\. 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁩)

1381\. 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 (𑁪)

(313.)

1382\. 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁧)

1383\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁨)

1384\. 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 (𑁩)

1385\. 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺; 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 (𑁪)

1386\. 𑀙𑀸𑀧𑀺 𑀲𑀫𑀣𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀧𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺; 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀦 𑀓𑁂𑀦𑀘𑀺 𑀲𑀫𑁆𑀫𑀢𑀺𑁇

---

1387\. 𑀲𑀫𑀣𑀸𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀏𑀓𑀽𑀦𑀯𑀻𑀲𑀢𑀺𑀫𑁄𑁇
