# 𑁫𑁇𑁧𑁩 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑀦𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀲𑀫𑀣𑀪𑁂𑀤

### 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑀦𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑀯𑀸𑀭

(308.)

1347\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺? 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺?

1348\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 𑀧𑀜𑁆𑀘𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁧)

1349\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁨)

1350\. 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 (𑁩)

1351\. 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀣𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦𑁇 𑀙𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁪)

1352\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀧𑀜𑁆𑀘𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁫)

1353\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 (𑁬)

1354\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 𑀧𑀜𑁆𑀘𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁭)

1355\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀙𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀣𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬𑁇 (𑁮)

1356\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁯)

1357\. 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 (𑁧𑁦)

1358\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀲𑀢𑁆𑀢𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁧𑁧)

1359\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀧𑀜𑁆𑀘𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁧𑁨)

1360\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀙𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀣𑁂𑀦 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬𑁇 (𑁧𑁩)

1361\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘 𑀲𑀢𑁆𑀢𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁧𑁪)
