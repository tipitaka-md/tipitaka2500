# 𑁫𑁇𑁪 𑀲𑀫𑀣𑀸𑀲𑀫𑀣𑀲𑁆𑀲𑀲𑀸𑀥𑀸𑀭𑀡𑀯𑀸𑀭

### 𑀲𑀫𑀣𑀪𑁂𑀤

### 𑀲𑀫𑀣𑀸𑀲𑀫𑀣𑀲𑁆𑀲𑀲𑀸𑀥𑀸𑀭𑀡𑀯𑀸𑀭

(299.)

1261\. 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇

1262\. 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀣𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 (𑁧)

1263\. 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑀲𑁆𑀲 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑁄𑁇 (𑁨)

1264\. 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑁄, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑀲𑁆𑀲 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑁄𑁇 (𑁩)

1265\. 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀁, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑀲𑁆𑀲 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁𑁇 (𑁪)

1266\. 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑀲𑁆𑀲 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇 (𑁫)

1267\. 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑁄, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑁄𑁇 (𑁬)

1268\. 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸; 𑀏𑀯𑀁 𑀲𑀺𑀬𑀸 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸𑁇

---

1269\. 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀦𑀯𑀫𑁄𑁇
