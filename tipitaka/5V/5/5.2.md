# 𑁫𑁇𑁨 𑀲𑀸𑀥𑀸𑀭𑀡𑀯𑀸𑀭

* 𑀲𑀫𑀣𑀪𑁂𑀤

* 𑀲𑀸𑀥𑀸𑀭𑀡𑀯𑀸𑀭

(297.)

1249\. 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸?

1250\. 𑀤𑁆𑀯𑁂 𑀲𑀫𑀣𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑁇 𑀧𑀜𑁆𑀘 𑀲𑀫𑀣𑀸 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸—  𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑁇 (𑁧)

1251\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑀣𑀸 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄; 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑁇 𑀢𑀬𑁄 𑀲𑀫𑀣𑀸 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑁇 (𑁨)

1252\. 𑀢𑀬𑁄 𑀲𑀫𑀣𑀸 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑀸—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑀣𑀸 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑁇 (𑁩)

1253\. 𑀏𑀓𑁄 𑀲𑀫𑀣𑁄 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀸𑀥𑀸𑀭𑀡𑁄—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑁇 𑀙 𑀲𑀫𑀣𑀸 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸—  𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑁇 (𑁪)

---

1254\. 𑀲𑀸𑀥𑀸𑀭𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇
