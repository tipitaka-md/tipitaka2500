# 𑁫𑁇𑁧𑁧 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭

### 𑀲𑀫𑀣𑀪𑁂𑀤

### 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭

(306.)

1328\. 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑁆𑀢𑀺 𑀯𑀸 𑀲𑀫𑀣𑀸𑀢𑀺 𑀯𑀸 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀉𑀤𑀸𑀳𑀼 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸? 𑀮𑀩𑁆𑀪𑀸 𑀘 𑀧𑀦𑀺𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆𑀢𑀺?

1329\. 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑁆𑀢𑀺 𑀯𑀸 𑀲𑀫𑀣𑀸𑀢𑀺 𑀯𑀸 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀦𑁄 𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀮𑀩𑁆𑀪𑀸 𑀘 𑀧𑀦𑀺𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆𑀢𑀺𑁇 𑀲𑁄—  “𑀫𑀸 𑀳𑁂𑀯𑀦𑁆”𑀢𑀺𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄𑁇 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑁆𑀢𑀺 𑀯𑀸 𑀲𑀫𑀣𑀸𑀢𑀺 𑀯𑀸 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸𑁇 𑀦𑁄 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲 𑀳𑁂𑀢𑀼? 𑀦𑀦𑀼 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  “𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀺, 𑀲𑀢𑁆𑀢 𑀲𑀫𑀣𑀸𑁇 𑀅𑀥𑀺𑀓𑀭𑀡𑀸 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀲𑀫𑀣𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁, 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸; 𑀦𑁄 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
