# 𑁫𑁇𑁭 𑀯𑀺𑀦𑀬𑀯𑀸𑀭

### 𑀲𑀫𑀣𑀪𑁂𑀤

### 𑀯𑀺𑀦𑀬𑀯𑀸𑀭

(302.)

1288\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄, 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀦𑀬𑁄? 𑀯𑀺𑀦𑀬𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑁄? 𑀯𑀺𑀦𑀬𑁄 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀦𑀬𑁄? 𑀯𑀺𑀦𑀬𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀦𑀬𑁄? 𑀯𑀺𑀦𑀬𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀯𑀺𑀦𑀬𑁄? 𑀯𑀺𑀦𑀬𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑁄? 𑀯𑀺𑀦𑀬𑁄 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀯𑀺𑀦𑀬𑁄?

1289\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀦 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑁇 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀦𑀬𑁄 𑀘𑁂𑀯 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀘𑁇 (𑁧)

1290\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑁇 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑁄 𑀘𑁂𑀯 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀘𑁇 (𑁨)

1291\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄𑁇 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀦𑀬𑁄 𑀘𑁂𑀯 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀘𑁇 (𑁩)

1292\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄𑁇 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀯𑀺𑀦𑀬𑁄 𑀘𑁂𑀯 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀘𑁇 (𑁪)

1293\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀲𑀺𑀬𑀸 𑀦 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁𑁇 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀯𑀺𑀦𑀬𑁄 𑀘𑁂𑀯 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀜𑁆𑀘𑁇 (𑁫)

1294\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀲𑀺𑀬𑀸 𑀦 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑁇 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑁄 𑀘𑁂𑀯 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀘𑁇 (𑁬)

1295\. 𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄, 𑀲𑀺𑀬𑀸 𑀦 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑁇 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀯𑀺𑀦𑀬𑁄 𑀘𑁂𑀯 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀘𑁇 (𑁭)

---

1296\. 𑀯𑀺𑀦𑀬𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑁄𑁇
