# 𑁫𑁇𑁮 𑀓𑀼𑀲𑀮𑀯𑀸𑀭

### 𑀲𑀫𑀣𑀪𑁂𑀤

### 𑀓𑀼𑀲𑀮𑀯𑀸𑀭

(303.)

1297\. 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀓𑀼𑀲𑀮𑁄 𑀅𑀓𑀼𑀲𑀮𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄? 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀓𑀼𑀲𑀮𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀓𑀼𑀲𑀮𑁄 𑀅𑀓𑀼𑀲𑀮𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄? 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀓𑀼𑀲𑀮𑁄 𑀅𑀓𑀼𑀲𑀮𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄? 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁? 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀓𑀼𑀲𑀮𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸? 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀓𑀼𑀲𑀮𑁄 𑀅𑀓𑀼𑀲𑀮𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄?

1298\. 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 𑀦𑀢𑁆𑀣𑀺 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀅𑀓𑀼𑀲𑀮𑁄𑁇 (𑁧)

1299\. 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 (𑁨)

1300\. 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 (𑁩)

1301\. 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 (𑁪)

1302\. 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 (𑁫)

1303\. 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀸, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑁇 (𑁬)

1304\. 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑁄, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄𑁇 (𑁭)

1305\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀼𑀲𑀮𑀁 𑀅𑀓𑀼𑀲𑀮𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇

1306\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 (𑁧)

1307\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 (𑁨)

1308\. 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 𑀦𑀢𑁆𑀣𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 (𑁩)

1309\. 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀺𑀬𑀸 𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀓𑀼𑀲𑀮𑀁, 𑀲𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁𑁇 (𑁪)

---

1310\. 𑀓𑀼𑀲𑀮𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀢𑁂𑀭𑀲𑀫𑁄𑁇
