# 𑁫𑁇𑁬 𑀲𑀫𑀣𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀯𑀸𑀭

* 𑀲𑀫𑀣𑀪𑁂𑀤

* 𑀲𑀫𑀣𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀯𑀸𑀭

(301.)

1279\. 𑀲𑀫𑀣𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄, 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑀣𑁄? 𑀲𑀫𑀣𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑀣𑁄? 𑀲𑀫𑀣𑁄 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑀣𑁄? 𑀲𑀫𑀣𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑀣𑁄? 𑀲𑀫𑀣𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀲𑀫𑀣𑁄? 𑀲𑀫𑀣𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑀣𑁄? 𑀲𑀫𑀣𑁄 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑀣𑁄?

1280\. 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄—  𑀇𑀫𑁂 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀸, 𑀦𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑁇 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑀣𑁄 𑀘𑁂𑀯 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀘𑁇 (𑁧)

1281\. 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄—  𑀇𑀫𑁂 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀸, 𑀦𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑁇 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑀣𑁄 𑀘𑁂𑀯 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀘𑁇 (𑁨)

1282\. 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸—  𑀇𑀫𑁂 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀸, 𑀦𑁄 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄𑁇 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑀣𑁄 𑀘𑁂𑀯 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀘𑁇 (𑁩)

1283\. 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄—  𑀇𑀫𑁂 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀸, 𑀦𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄𑁇 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀲𑀫𑀣𑁄 𑀘𑁂𑀯 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀘𑁇 (𑁪)

1284\. 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄—  𑀇𑀫𑁂 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀸, 𑀦𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁𑁇 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀲𑀫𑀣𑁄 𑀘𑁂𑀯 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀜𑁆𑀘𑁇 (𑁫)

1285\. 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁—  𑀇𑀫𑁂 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀸, 𑀦𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑁇 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀲𑀫𑀣𑁄 𑀘𑁂𑀯 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀘𑁇 (𑁬)

1286\. 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸—  𑀇𑀫𑁂 𑀲𑀫𑀣𑀸 𑀲𑀫𑀣𑀸, 𑀦𑁄 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑁇 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀲𑀫𑀣𑁄 𑀘𑁂𑀯 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀘𑁇 (𑁭)

---

1287\. 𑀲𑀫𑀣𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀏𑀓𑀸𑀤𑀲𑀫𑁄𑁇
