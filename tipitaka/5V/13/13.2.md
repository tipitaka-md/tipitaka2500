# 𑁧𑁩𑁇𑁨 𑀘𑁄𑀤𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺

* 𑀘𑁄𑀤𑀦𑀸𑀓𑀡𑁆𑀟

* 𑀘𑁄𑀤𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺

(363.)

1798\. 𑀘𑁄𑀤𑀓𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁? 𑀘𑀼𑀤𑀺𑀢𑀓𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁? 𑀲𑀁𑀖𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁? 𑀅𑀦𑀼𑀯𑀺𑀚𑁆𑀚𑀓𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁? 𑀘𑁄𑀤𑀓𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺? 𑀘𑁄𑀤𑀓𑁂𑀦 𑀧𑀜𑁆𑀘𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀬 𑀧𑀭𑁄 𑀘𑁄𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀸𑀮𑁂𑀦 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀦𑁄 𑀅𑀓𑀸𑀮𑁂𑀦, 𑀪𑀽𑀢𑁂𑀦 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀦𑁄 𑀅𑀪𑀽𑀢𑁂𑀦, 𑀲𑀡𑁆𑀳𑁂𑀦 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀦𑁄 𑀨𑀭𑀼𑀲𑁂𑀦, 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁂𑀦 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀦𑁄 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁂𑀦, 𑀫𑁂𑀢𑁆𑀢𑀸𑀘𑀺𑀢𑁆𑀢𑁄 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀦𑁄 𑀤𑁄𑀲𑀦𑁆𑀢𑀭𑁄𑀢𑀺—  𑀘𑁄𑀤𑀓𑁂𑀦 𑀏𑀯𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀘𑀼𑀤𑀺𑀢𑀓𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺? 𑀘𑀼𑀤𑀺𑀢𑀓𑁂𑀦 𑀤𑁆𑀯𑀻𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁆𑀘𑁂 𑀘 𑀅𑀓𑀼𑀧𑁆𑀧𑁂 𑀘—  𑀘𑀼𑀤𑀺𑀢𑀓𑁂𑀦 𑀏𑀯𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀁𑀖𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺? 𑀲𑀁𑀖𑁂𑀦 𑀑𑀢𑀺𑀡𑁆𑀡𑀸𑀦𑁄𑀢𑀺𑀡𑁆𑀡𑀁 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀁𑀖𑁂𑀦 𑀏𑀯𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀼𑀯𑀺𑀚𑁆𑀚𑀓𑁂𑀦 𑀓𑀣𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺? 𑀅𑀦𑀼𑀯𑀺𑀚𑁆𑀚𑀓𑁂𑀦 𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀬𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀬𑁂𑀦 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺 𑀢𑀣𑀸 𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀼𑀯𑀺𑀚𑁆𑀚𑀓𑁂𑀦 𑀏𑀯𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇

(364.)

1799\. _𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀺𑀫𑀢𑁆𑀣𑀸𑀬,_  
_𑀧𑀯𑀸𑀭𑀡𑀸 𑀓𑀺𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸;_  
_𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀓𑀺𑀫𑀢𑁆𑀣𑀸𑀬,_  
_𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸 𑀓𑀺𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸;_  
_𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀓𑀺𑀫𑀢𑁆𑀣𑀸𑀬,_  
_𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀓𑀺𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸𑁇_  


1800\. _𑀉𑀧𑁄𑀲𑀣𑁄 𑀲𑀸𑀫𑀕𑁆𑀕𑀢𑁆𑀣𑀸𑀬,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀸;_  
_𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀫𑀸𑀦𑀢𑁆𑀢𑀢𑁆𑀣𑀸𑀬,_  
_𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸 𑀦𑀺𑀕𑁆𑀕𑀳𑀢𑁆𑀣𑀸𑀬;_  
_𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀢𑁆𑀣𑀸𑀬,_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀢𑁆𑀣𑀸𑀬 𑀅𑀩𑁆𑀪𑀸𑀦𑀁𑁇_  


1801\. _𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀪𑀬𑀸 𑀫𑁄𑀳𑀸,_  
_𑀣𑁂𑀭𑁂 𑀘 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀔𑀢𑁄 𑀉𑀧𑀳𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀦𑀺𑀭𑀬𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑁄,_  
_𑀦 𑀘 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀕𑀸𑀭𑀯𑁄𑁇_  


1802\. _𑀦 𑀘 𑀆𑀫𑀺𑀲𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀦 𑀘 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀼𑀕𑁆𑀕𑀮𑀁;_  
_𑀉𑀪𑁄 𑀏𑀢𑁂 𑀯𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸,_  
_𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀢𑀣𑀸 𑀓𑀭𑁂𑁇_
