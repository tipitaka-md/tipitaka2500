# 𑁨𑁇𑁨𑁇𑁬 𑀅𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀕

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭

(254.)

1064\. 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂… 𑁇 (𑁧)

1065\. 𑀤𑀥𑀺𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀤𑀥𑀺𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁—  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 (𑁨)

---

1066\. 𑀅𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
