# 𑁨𑁇𑁨𑁇𑁧𑁇𑁧 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀕

### 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

(247.)

1011\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁? 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪? 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁…𑀧𑁂…  𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺?

1012\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀁 𑀓𑀢𑁆𑀣𑁄𑀕𑀥𑀁 𑀓𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺? 𑀦𑀺𑀤𑀸𑀦𑁄𑀕𑀥𑀁 𑀦𑀺𑀤𑀸𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁? 𑀓𑀢𑀫𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀫𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀫𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺? 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂…  𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺? 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀪𑀢𑀁—

1013\. _𑀉𑀧𑀸𑀮𑀺 𑀤𑀸𑀲𑀓𑁄 𑀘𑁂𑀯,_  
_𑀲𑁄𑀡𑀓𑁄 𑀲𑀺𑀕𑁆𑀕𑀯𑁄 𑀢𑀣𑀸;_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀸,_  
_𑀏𑀢𑁂 𑀚𑀫𑁆𑀩𑀼𑀲𑀺𑀭𑀺𑀯𑁆𑀳𑀬𑁂𑁇_  


---

1014\. …𑀧𑁂…



1015\. _𑀏𑀢𑁂 𑀦𑀸𑀕𑀸 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀺𑀦𑀬𑀜𑁆𑀜𑀽 𑀫𑀕𑁆𑀕𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀤𑀻𑀧𑁂 𑀧𑀓𑀸𑀲𑁂𑀲𑀼𑀁,_  
_𑀧𑀺𑀝𑀓𑀁 𑀢𑀫𑁆𑀩𑀧𑀡𑁆𑀡𑀺𑀬𑀸𑀢𑀺𑁇 (𑁧)_  


1016\. 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑀺 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁨)

1017\. 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁩)

1018\. 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁪)
