# 𑁨𑁇𑁧𑁇𑁨𑁇𑁧 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

(228.)

848\. 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑀁 𑀅𑀥𑁄𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁧)

849\. 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑀚𑁆𑀚𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀻 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑀚𑁆𑀚𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀻 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀯𑀚𑁆𑀚𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀯𑀚𑁆𑀚𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀻 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁨)

850\. 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀻 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀻 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸; 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲—  𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀻 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁩)

851\. 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀼𑀭𑀺𑀲𑁂𑀦—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀑𑀓𑀸𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀸”𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀫𑀢𑁆𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲—  𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 (𑁪)

---
