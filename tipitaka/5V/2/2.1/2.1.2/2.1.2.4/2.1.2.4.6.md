# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁬 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀆𑀭𑀸𑀫𑀯𑀕𑁆𑀕

(236.)

932\. 𑀚𑀸𑀦𑀁 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀁 𑀆𑀭𑀸𑀫𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀧𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

933\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀻 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀓𑁆𑀓𑁄𑀲𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

934\. 𑀘𑀡𑁆𑀟𑀻𑀓𑀢𑀸 𑀕𑀡𑀁 𑀧𑀭𑀺𑀪𑀸𑀲𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀭𑀺𑀪𑀸𑀲𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

935\. 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀯𑀸 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀯𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

936\. 𑀓𑀼𑀮𑀁 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀫𑀘𑁆𑀙𑀭𑀸𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀫𑀘𑁆𑀙𑀭𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

937\. 𑀅𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁂 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑀺 𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀲𑀳 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

938\. 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀪𑀢𑁄𑀲𑀁𑀖𑁂 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀦 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁭)

939\. 𑀑𑀯𑀸𑀤𑀸𑀬 𑀯𑀸 𑀲𑀁𑀯𑀸𑀲𑀸𑀬 𑀯𑀸 𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁮)

940\. 𑀉𑀧𑁄𑀲𑀣𑀫𑁆𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀑𑀯𑀸𑀤𑀫𑁆𑀧𑀺 𑀦 𑀬𑀸𑀘𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁯)

941\. 𑀧𑀲𑀸𑀔𑁂 𑀚𑀸𑀢𑀁 𑀕𑀡𑁆𑀟𑀁 𑀯𑀸 𑀭𑀼𑀥𑀺𑀢𑀁 𑀯𑀸 𑀅𑀦𑀧𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁 𑀯𑀸 𑀕𑀡𑀁 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀪𑁂𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑁂𑀤𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀪𑀺𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)
