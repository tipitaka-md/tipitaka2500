# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁯 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀯𑀕𑁆𑀕

(239.)

968\. 𑀙𑀢𑁆𑀢𑀼𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀥𑀸𑀭𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

969\. 𑀬𑀸𑀦𑁂𑀦 𑀬𑀸𑀬𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑀸𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀬𑀸𑀬𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

970\. 𑀲𑀗𑁆𑀖𑀸𑀡𑀺𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀥𑀸𑀭𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

971\. 𑀇𑀢𑁆𑀣𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀥𑀸𑀭𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

972\. 𑀕𑀦𑁆𑀥𑀯𑀡𑁆𑀡𑀓𑁂𑀦 𑀦𑀳𑀸𑀬𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀳𑀸𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀳𑀸𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

973\. 𑀯𑀸𑀲𑀺𑀢𑀓𑁂𑀦 𑀧𑀺𑀜𑁆𑀜𑀸𑀓𑁂𑀦 𑀦𑀳𑀸𑀬𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀳𑀸𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀳𑀸𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

974\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

975\. 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

976\. 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

977\. 𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀸 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀫𑁆𑀫𑀤𑁆𑀤𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)

978\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁧)

979\. 𑀅𑀦𑁄𑀓𑀸𑀲𑀓𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀜𑁆𑀳𑀁 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀼𑀘𑁆𑀙𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁨)

980\. 𑀅𑀲𑀗𑁆𑀓𑀘𑁆𑀘𑀺𑀓𑀸 𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀧𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁩)

---
