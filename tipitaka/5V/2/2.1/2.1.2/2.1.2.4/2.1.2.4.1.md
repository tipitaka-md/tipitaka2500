# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁧 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

(231.)

877\. 𑀮𑀲𑀼𑀡𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

878\. 𑀲𑀫𑁆𑀩𑀸𑀥𑁂 𑀮𑁄𑀫𑀁 𑀲𑀁𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀁𑀳𑀭𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀲𑀁𑀳𑀭𑀸𑀧𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

879\. 𑀢𑀮𑀖𑀸𑀢𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀭𑁄𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀓𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

880\. 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁 𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀆𑀤𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

881\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑀧𑀭𑀫𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀆𑀤𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

882\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁬)

883\. 𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

884\. 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀁 𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀁 𑀯𑀸 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀯𑀸 𑀢𑀺𑀭𑁄𑀧𑀸𑀓𑀸𑀭𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀙𑀟𑁆𑀟𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

885\. 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀁 𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀁 𑀯𑀸 𑀳𑀭𑀺𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀙𑀟𑁆𑀟𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

886\. 𑀦𑀘𑁆𑀘𑀁 𑀯𑀸 𑀕𑀻𑀢𑀁 𑀯𑀸 𑀯𑀸𑀤𑀺𑀢𑀁 𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀺 𑀯𑀸 𑀲𑀼𑀡𑀸𑀢𑀺 𑀯𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)
