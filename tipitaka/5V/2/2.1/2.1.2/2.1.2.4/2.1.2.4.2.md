# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁨 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕

(232.)

888\. 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀅𑀧𑁆𑀧𑀤𑀻𑀧𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁧)

889\. 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀑𑀓𑀸𑀲𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁨)

890\. 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁩)

891\. 𑀭𑀣𑀺𑀓𑀸𑀬 𑀯𑀸 𑀩𑁆𑀬𑀽𑀳𑁂 𑀯𑀸 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂 𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁂𑀦𑁂𑀓𑀸 𑀲𑀦𑁆𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀝𑁆𑀞𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁪)

892\. 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀓𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

893\. 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

894\. 𑀯𑀺𑀓𑀸𑀮𑁂 𑀓𑀼𑀮𑀸𑀦𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀸 𑀲𑀦𑁆𑀣𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀪𑀺𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

895\. 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑁂𑀦 𑀤𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑁂𑀦 𑀧𑀭𑀁 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀚𑁆𑀛𑀸𑀧𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

896\. 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀸 𑀧𑀭𑀁 𑀯𑀸 𑀦𑀺𑀭𑀬𑁂𑀦 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀲𑀧𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀪𑀺𑀲𑀧𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀪𑀺𑀲𑀧𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

897\. 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀯𑀥𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀥𑀢𑀺 𑀭𑁄𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀯𑀥𑀢𑀺 𑀦 𑀭𑁄𑀤𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁧𑁦)

---

898\. 𑀭𑀢𑁆𑀢𑀦𑁆𑀥𑀓𑀸𑀭𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
