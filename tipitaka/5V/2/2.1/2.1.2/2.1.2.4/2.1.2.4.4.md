# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁪 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕

(234.)

910\. 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀫𑀜𑁆𑀘𑁂 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀧𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

911\. 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀢𑁆𑀣𑀭𑀡𑀧𑀸𑀯𑀼𑀭𑀡𑀸 𑀢𑀼𑀯𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺𑀬𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀦𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀧𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

912\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀅𑀨𑀸𑀲𑀼𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀭𑁄𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀓𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

913\. 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀲𑀳𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀦𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑁂𑀦𑁆𑀢𑀻 𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁪)

914\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

915\. 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

916\. 𑀅𑀦𑁆𑀢𑁄𑀭𑀝𑁆𑀞𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

917\. 𑀢𑀺𑀭𑁄𑀭𑀝𑁆𑀞𑁂 𑀲𑀸𑀲𑀗𑁆𑀓𑀲𑀫𑁆𑀫𑀢𑁂 𑀲𑀧𑁆𑀧𑀝𑀺𑀪𑀬𑁂 𑀅𑀲𑀢𑁆𑀣𑀺𑀓𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

918\. 𑀅𑀦𑁆𑀢𑁄𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

919\. 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀘𑀸𑀭𑀺𑀓𑀁 𑀦 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁧𑁦)

---

920\. 𑀢𑀼𑀯𑀝𑁆𑀝𑀯𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
