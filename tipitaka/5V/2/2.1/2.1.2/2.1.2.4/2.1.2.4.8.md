# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁮 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

(238.)

954\. 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

955\. 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

956\. 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

957\. 𑀊𑀦𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

958\. 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

959\. “𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 “𑀲𑀸𑀥𑀽”𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀔𑀺𑀬𑁆𑀬𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

960\. 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁—  “𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀲𑀺, 𑀏𑀯𑀸𑀳𑀁 𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁭)

961\. 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁—  “𑀲𑀘𑁂 𑀫𑀁 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀲𑁆𑀲𑀲𑀺, 𑀏𑀯𑀸𑀳𑀁 𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁮)

962\. 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

963\. 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂𑀦 𑀯𑀸 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)

964\. 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀙𑀦𑁆𑀤𑀤𑀸𑀦𑁂𑀦 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁧)

965\. 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁨)

966\. 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁩)
