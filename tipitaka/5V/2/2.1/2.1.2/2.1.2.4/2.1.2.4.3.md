# 𑁨𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁩 𑀦𑀳𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀦𑀳𑀸𑀦𑀯𑀕𑁆𑀕

(233.)

899\. 𑀦𑀕𑁆𑀕𑀸 𑀦𑀳𑀸𑀬𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀳𑀸𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀳𑀸𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

900\. 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

901\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑁂𑀦𑁆𑀢𑀻 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁩)

902\. 𑀧𑀜𑁆𑀘𑀸𑀳𑀺𑀓𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀘𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁪)

903\. 𑀘𑀻𑀯𑀭𑀲𑀗𑁆𑀓𑀫𑀦𑀻𑀬𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀥𑀸𑀭𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀥𑀸𑀭𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

904\. 𑀕𑀡𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀮𑀸𑀪𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀭𑁄𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀓𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

905\. 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

906\. 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀬 𑀯𑀸 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀤𑀺𑀦𑁆𑀦𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

907\. 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀬 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

908\. 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀝𑀺𑀩𑀸𑀳𑀺𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)
