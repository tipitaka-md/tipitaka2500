# 𑁨𑁇𑁧𑁇𑁨𑁇𑁩 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

(230.)

864. 𑀧𑀢𑁆𑀢𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀻 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁧)

865. 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 “𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀸𑀚𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀸𑀚𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀪𑀸𑀚𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁨)

866. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁩)

867. 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁪)

868. 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁫)

869. 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀺𑀓𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦 𑀅𑀜𑁆𑀜𑀼𑀤𑁆𑀤𑀺𑀲𑀺𑀓𑁂𑀦 𑀲𑀁𑀖𑀺𑀓𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁬)

870. 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀺𑀓𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦 𑀅𑀜𑁆𑀜𑀼𑀤𑁆𑀤𑀺𑀲𑀺𑀓𑁂𑀦 𑀲𑀁𑀖𑀺𑀓𑁂𑀦 𑀲𑀁𑀬𑀸𑀘𑀺𑀓𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁭)

871. 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀺𑀓𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦 𑀅𑀜𑁆𑀜𑀼𑀤𑁆𑀤𑀺𑀲𑀺𑀓𑁂𑀦 𑀫𑀳𑀸𑀚𑀦𑀺𑀓𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁮)

872. 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀺𑀓𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦 𑀅𑀜𑁆𑀜𑀼𑀤𑁆𑀤𑀺𑀲𑀺𑀓𑁂𑀦 𑀫𑀳𑀸𑀚𑀦𑀺𑀓𑁂𑀦 𑀲𑀁𑀬𑀸𑀘𑀺𑀓𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁯)

873. 𑀅𑀜𑁆𑀜𑀤𑀢𑁆𑀣𑀺𑀓𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁂𑀦 𑀅𑀜𑁆𑀜𑀼𑀤𑁆𑀤𑀺𑀲𑀺𑀓𑁂𑀦 𑀧𑀼𑀕𑁆𑀕𑀮𑀺𑀓𑁂𑀦 𑀲𑀁𑀬𑀸𑀘𑀺𑀓𑁂𑀦 𑀅𑀜𑁆𑀜𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁧𑁦)

874. 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑀲𑀧𑀭𑀫𑀁 𑀕𑀭𑀼𑀧𑀸𑀯𑀼𑀭𑀡𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁧𑁧)

875. 𑀅𑀢𑀺𑀭𑁂𑀓𑀅𑀟𑁆𑀠𑀢𑁂𑀬𑁆𑀬𑀓𑀁𑀲𑀧𑀭𑀫𑀁 𑀮𑀳𑀼𑀧𑀸𑀯𑀼𑀭𑀡𑀁 𑀘𑁂𑀢𑀸𑀧𑁂𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀘𑁂𑀢𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀘𑁂𑀢𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁧𑁨)

---

876. 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
