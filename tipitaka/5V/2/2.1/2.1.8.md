# 𑁨𑁇𑁧𑁇𑁮 𑀲𑀫𑀼𑀘𑁆𑀘𑀬𑀯𑀸𑀭

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀲𑀫𑀼𑀘𑁆𑀘𑀬𑀯𑀸𑀭

(246.)

1006\. 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀥𑀓𑁆𑀔𑀓𑀁 𑀉𑀩𑁆𑀪𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑀁 𑀅𑀥𑁄𑀚𑀸𑀡𑀼𑀫𑀡𑁆𑀟𑀮𑀁 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑀁 𑀕𑀳𑀡𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀻 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1007\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘…𑀧𑁂… 𑁇 (𑁧)

1008\. 𑀤𑀥𑀺𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀤𑀥𑀺𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲—  𑀤𑀥𑀺𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻 𑀇𑀫𑀸 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1009\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀏𑀓𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀦 𑀯𑀸𑀘𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀦 𑀯𑀸𑀘𑀢𑁄, 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁—  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁨)
