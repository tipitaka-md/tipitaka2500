# 𑁨𑁇𑁧𑁇𑁧𑁇𑁪𑁇𑁩 𑀦𑀳𑀸𑀦𑀯𑀕𑁆𑀕

### 𑀦𑀳𑀸𑀦𑀯𑀕𑁆𑀕

(220.)

740\. 𑀦𑀕𑁆𑀕𑀸𑀬 𑀦𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦𑀕𑁆𑀕𑀸 𑀦𑀳𑀸𑀬𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇 (𑁧)

741\. 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁨)

742\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀲𑀺𑀩𑁆𑀩𑁂𑀲𑀺 𑀦 𑀲𑀺𑀩𑁆𑀩𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁩)

743\. 𑀧𑀜𑁆𑀘𑀸𑀳𑀺𑀓𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀘𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀲𑀦𑁆𑀢𑀭𑀼𑀢𑁆𑀢𑀭𑁂𑀦 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀓𑀣𑀺𑀦𑀓𑁂…𑀧𑁂… 𑁇 (𑁪)

744\. 𑀘𑀻𑀯𑀭𑀲𑀗𑁆𑀓𑀫𑀦𑀻𑀬𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀢𑀭𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀸𑀭𑀼𑀧𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀣𑀺𑀦𑀓𑁂…𑀧𑁂… 𑁇 (𑁫)

745\. 𑀕𑀡𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀮𑀸𑀪𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀡𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀮𑀸𑀪𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁬)

746\. 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁭)

747\. 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀬 𑀯𑀸 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀕𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀲𑀫𑀡𑀘𑀻𑀯𑀭𑀁 𑀅𑀤𑀸𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁮)

748\. 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀬 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀤𑀼𑀩𑁆𑀩𑀮𑀘𑀻𑀯𑀭𑀧𑀘𑁆𑀘𑀸𑀲𑀸𑀬 𑀘𑀻𑀯𑀭𑀓𑀸𑀮𑀲𑀫𑀬𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁯)

749\. 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁦)
