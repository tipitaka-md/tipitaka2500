# 𑁨𑁇𑁧𑁇𑁧𑁇𑁪𑁇𑁧 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀮𑀲𑀼𑀡𑀯𑀕𑁆𑀕

(218.)

718\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀮𑀲𑀼𑀡𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀮𑀲𑀼𑀡𑀁 𑀳𑀭𑀸𑀧𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇 (𑁧)

719\. 𑀲𑀫𑁆𑀩𑀸𑀥𑁂 𑀮𑁄𑀫𑀁 𑀲𑀁𑀳𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀲𑀫𑁆𑀩𑀸𑀥𑁂 𑀮𑁄𑀫𑀁 𑀲𑀁𑀳𑀭𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁨)

720\. 𑀢𑀮𑀖𑀸𑀢𑀓𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀢𑀮𑀖𑀸𑀢𑀓𑀁 𑀅𑀓𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂…𑀧𑁂… 𑁇 (𑁩)

721\. 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀢𑀼𑀫𑀝𑁆𑀞𑀓𑀁 𑀆𑀤𑀺𑀬𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂…𑀧𑁂… 𑁇 (𑁪)

722\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀧𑀩𑁆𑀩𑀧𑀭𑀫𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀢𑀺𑀕𑀫𑁆𑀪𑀻𑀭𑀁 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀁 𑀆𑀤𑀺𑀬𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂…𑀧𑁂… 𑁇 (𑁫)

723\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀯𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀘 𑀯𑀺𑀥𑀽𑀧𑀦𑁂𑀦 𑀘 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇 (𑁬)

724\. 𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀫𑀓𑀥𑀜𑁆𑀜𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁭)

725\. 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀁 𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀁 𑀯𑀸 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀢𑀺𑀭𑁄𑀓𑀼𑀝𑁆𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁮)

726\. 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀁 𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀁 𑀯𑀸 𑀳𑀭𑀺𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀲𑀗𑁆𑀓𑀸𑀭𑀫𑁆𑀧𑀺 𑀯𑀺𑀖𑀸𑀲𑀫𑁆𑀧𑀺 𑀳𑀭𑀺𑀢𑁂 𑀙𑀟𑁆𑀟𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁯)

727\. 𑀦𑀘𑁆𑀘𑀁 𑀯𑀸 𑀕𑀻𑀢𑀁 𑀯𑀸 𑀯𑀸𑀤𑀺𑀢𑀁 𑀯𑀸 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀦𑀘𑁆𑀘𑀫𑁆𑀧𑀺 𑀕𑀻𑀢𑀫𑁆𑀧𑀺 𑀯𑀸𑀤𑀺𑀢𑀫𑁆𑀧𑀺 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇 (𑁧𑁦)
