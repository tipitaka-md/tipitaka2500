# 𑁨𑁇𑁧𑁇𑁧𑁇𑁪𑁇𑁮 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀯𑀕𑁆𑀕

(225.)

795\. 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧)

796\. 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁨)

797\. 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀁 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀙𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁩)

798\. 𑀊𑀦𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀊𑀦𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁪)

799\. 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀸𑀬 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀤𑁆𑀯𑀸𑀤𑀲𑀯𑀲𑁆𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀅𑀲𑀫𑁆𑀫𑀢𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂…𑀧𑁂… 𑁇 (𑁫)

800\. “𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸𑀬 “𑀲𑀸𑀥𑀽”𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀇𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 “𑀅𑀮𑀁 𑀢𑀸𑀯 𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂, 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀺𑀢𑁂𑀦𑀸”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 “𑀲𑀸𑀥𑀽”𑀢𑀺 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁬)

801\. 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁—  “𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀲𑀺, 𑀏𑀯𑀸𑀳𑀁 𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁—  “𑀲𑀘𑁂 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑁆𑀲𑀲𑀺, 𑀏𑀯𑀸𑀳𑀁 𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁭)

802\. 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁—  “𑀲𑀘𑁂 𑀫𑀁 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀲𑁆𑀲𑀲𑀺, 𑀏𑀯𑀸𑀳𑀁 𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁—  “𑀲𑀘𑁂 𑀫𑀁 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀲𑁆𑀲𑀲𑀺, 𑀏𑀯𑀸𑀳𑀁 𑀢𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀦𑁂𑀯 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀸𑀬 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁮)

803\. 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀼𑀭𑀺𑀲𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀘𑀡𑁆𑀟𑀺𑀁 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁯)

804\. 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀯𑀸 𑀲𑀸𑀫𑀺𑀓𑁂𑀦 𑀯𑀸 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺𑀧𑀺 𑀲𑀸𑀫𑀺𑀓𑁂𑀦𑀸𑀧𑀺 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁦)

805\. 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀙𑀦𑁆𑀤𑀤𑀸𑀦𑁂𑀦 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀙𑀦𑁆𑀤𑀤𑀸𑀦𑁂𑀦 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁧)

806\. 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁨)

807\. 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀏𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀤𑁆𑀯𑁂 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁩)
