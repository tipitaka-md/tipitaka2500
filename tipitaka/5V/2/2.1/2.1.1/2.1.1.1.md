# 𑁨𑁇𑁧𑁇𑁧𑁇𑁧 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

662\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁? 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪? 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀁 𑀓𑀢𑁆𑀣𑁄𑀕𑀥𑀁 𑀓𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁? 𑀓𑀢𑀫𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀫𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀺? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀫𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺? 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀦𑀬𑁄? 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁄? 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁? 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀅𑀥𑀺𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁? 𑀓𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀺? 𑀓𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺? 𑀓𑀸 𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺? 𑀓𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁? 𑀓𑀸 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀓𑀸 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸? 𑀓𑀢𑁆𑀣 𑀞𑀺𑀢𑀁? 𑀓𑀸 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺? 𑀓𑀲𑁆𑀲 𑀯𑀘𑀦𑀁? 𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺?

663\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀼𑀦𑁆𑀤𑀭𑀻𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀯𑀲𑁆𑀲𑀼𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀁 𑀓𑀢𑁆𑀣𑁄𑀕𑀥𑀁 𑀓𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺? 𑀦𑀺𑀤𑀸𑀦𑁄𑀕𑀥𑀁 𑀦𑀺𑀤𑀸𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀫𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀫𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺? 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑁆𑀢𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀻𑀢𑀺? 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘𑁇 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀦𑀬𑁄, 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁄𑀢𑀺? 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀺𑀦𑀬𑁄, 𑀯𑀺𑀪𑀢𑁆𑀢𑀺 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁄𑁇 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁, 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀅𑀥𑀺𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁, 𑀯𑀺𑀪𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁𑁇 𑀓𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀅𑀲𑀁𑀯𑀭𑁄 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀁𑀯𑀭𑁄 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀸 𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? “𑀦 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀆𑀧𑀸𑀡𑀓𑁄𑀝𑀺𑀓𑀁 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀓𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀧𑀜𑁆𑀘𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬, 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑀢𑀸𑀬, 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬, 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀓𑀸 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻𑀢𑀺? 𑀲𑁂𑀓𑁆𑀔𑀸 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀓𑀸 𑀘 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀓𑀸 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀢𑀺? 𑀅𑀭𑀳𑀦𑁆𑀢𑀺𑀬𑁄 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑁇 𑀓𑀢𑁆𑀣 𑀞𑀺𑀢𑀦𑁆𑀢𑀺? 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑀸𑀲𑀼 𑀞𑀺𑀢𑀁𑁇 𑀓𑀸 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻𑀢𑀺? 𑀬𑀸𑀲𑀁 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑀸 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀲𑁆𑀲 𑀯𑀘𑀦𑀦𑁆𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺? 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀪𑀢𑀁—

664\. _𑀉𑀧𑀸𑀮𑀺 𑀤𑀸𑀲𑀓𑁄 𑀘𑁂𑀯,_  
_𑀲𑁄𑀡𑀓𑁄 𑀲𑀺𑀕𑁆𑀕𑀯𑁄 𑀢𑀣𑀸;_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀸,_  
_𑀏𑀢𑁂 𑀚𑀫𑁆𑀩𑀼𑀲𑀺𑀭𑀺𑀯𑁆𑀳𑀬𑁂𑁇_  


665\. _𑀢𑀢𑁄 𑀫𑀳𑀺𑀦𑁆𑀤𑁄 𑀇𑀝𑁆𑀝𑀺𑀬𑁄,_  
_𑀉𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀫𑁆𑀩𑀮𑁄 𑀢𑀣𑀸;_  
_𑀪𑀤𑁆𑀤𑀦𑀸𑀫𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


666\. _𑀏𑀢𑁂 𑀦𑀸𑀕𑀸 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸,_  
_𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑀸 𑀇𑀥𑀸𑀕𑀢𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀢𑁂 𑀯𑀸𑀘𑀬𑀺𑀁𑀲𑀼,_  
_𑀧𑀺𑀝𑀓𑀁 𑀢𑀫𑁆𑀩𑀧𑀡𑁆𑀡𑀺𑀬𑀸𑁇_  


667\. _𑀦𑀺𑀓𑀸𑀬𑁂 𑀧𑀜𑁆𑀘 𑀯𑀸𑀘𑁂𑀲𑀼𑀁,_  
_𑀲𑀢𑁆𑀢 𑀘𑁂𑀯 𑀧𑀓𑀭𑀡𑁂;_  
_𑀢𑀢𑁄 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀢𑀺𑀲𑁆𑀲𑀤𑀢𑁆𑀢𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


668\. _𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀓𑀸𑀍𑀅𑀲𑀼𑀫𑀦𑁄,_  
_𑀣𑁂𑀭𑁄 𑀘 𑀤𑀻𑀖𑀦𑀸𑀫𑀓𑁄;_  
_𑀤𑀻𑀖𑀲𑀼𑀫𑀦𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


669\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀓𑀸𑀍𑀅𑀲𑀼𑀫𑀦𑁄,_  
_𑀦𑀸𑀕𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀩𑀼𑀤𑁆𑀥𑀭𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀤𑁂𑀯𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


670\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀼𑀫𑀦𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀘𑀽𑀍𑀅𑀦𑀸𑀕𑁄,_  
_𑀕𑀚𑁄𑀯 𑀤𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀬𑁄𑁇_  


671\. _𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑀺𑀢𑀦𑀸𑀫𑁄 𑀘,_  
_𑀭𑁄𑀳𑀡𑁂 𑀲𑀸𑀥𑀼𑀧𑀽𑀚𑀺𑀢𑁄;_  
_𑀢𑀲𑁆𑀲 𑀲𑀺𑀲𑁆𑀲𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀔𑁂𑀫𑀦𑀸𑀫𑁄 𑀢𑀺𑀧𑁂𑀝𑀓𑁄𑁇_  


672\. _𑀤𑀻𑀧𑁂 𑀢𑀸𑀭𑀓𑀭𑀸𑀚𑀸𑀯,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀢𑀺𑀭𑁄𑀘𑀣;_  
_𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀨𑀼𑀲𑁆𑀲𑀤𑁂𑀯𑁄 𑀫𑀳𑀸𑀓𑀣𑀻𑁇_  


673\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀼𑀫𑀦𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀧𑀼𑀧𑁆𑀨𑀦𑀸𑀫𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀫𑀳𑀸𑀓𑀣𑀻 𑀫𑀳𑀸𑀲𑀺𑀯𑁄,_  
_𑀧𑀺𑀝𑀓𑁂 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


674\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀉𑀧𑀸𑀮𑀺 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀫𑀳𑀸𑀦𑀸𑀕𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀯𑀁𑀲𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


675\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀅𑀪𑀬𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀧𑀺𑀝𑀓𑁂 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄𑁇_  


676\. _𑀢𑀲𑁆𑀲 𑀲𑀺𑀲𑁆𑀲𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀧𑀼𑀧𑁆𑀨𑀦𑀸𑀫𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀲𑀸𑀲𑀦𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄,_  
_𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇_  


677\. _𑀘𑀽𑀍𑀆𑀪𑀬𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀯𑀁𑀲𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


678\. _𑀘𑀽𑀍𑀅𑀤𑁂𑀯𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀲𑀺𑀯𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


679\. _𑀏𑀢𑁂 𑀦𑀸𑀕𑀸 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀺𑀦𑀬𑀜𑁆𑀜𑀽 𑀫𑀕𑁆𑀕𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀤𑀻𑀧𑁂 𑀧𑀓𑀸𑀲𑁂𑀲𑀼𑀁,_  
_𑀧𑀺𑀝𑀓𑀁 𑀢𑀫𑁆𑀩𑀧𑀡𑁆𑀡𑀺𑀬𑀸𑀢𑀺𑁇 (𑁧)_  


(203.)

680\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀙𑀝𑁆𑀞𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀦𑁂𑀯𑀢𑁆𑀢𑀦𑀸 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀲𑀺 𑀦 𑀕𑀡𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁨)

(204.)

681\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀲𑀢𑁆𑀢𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀣𑀼𑀮𑁆𑀮𑀦𑀦𑁆𑀤𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀅𑀭𑀺𑀝𑁆𑀞𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀕𑀤𑁆𑀥𑀩𑀸𑀥𑀺𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁩)

(205.)

682\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀅𑀝𑁆𑀞𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀥𑀼𑀭𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑁂…𑀧𑁂… 𑁇 (𑁪)

---

683\. 𑀅𑀝𑁆𑀞 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇



---

684\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



685\. _𑀫𑁂𑀣𑀼𑀦𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀜𑁆𑀘,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀼𑀢𑁆𑀢𑀭𑀺;_  
_𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑀁 𑀙𑀸𑀤𑁂𑀢𑀺,_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀅𑀝𑁆𑀞 𑀯𑀢𑁆𑀣𑀼𑀓𑀸;_  
_𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑁄,_  
_𑀙𑁂𑀚𑁆𑀚𑀯𑀢𑁆𑀣𑀽 𑀅𑀲𑀁𑀲𑀬𑀸𑀢𑀺𑁇_
