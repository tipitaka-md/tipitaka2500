# 𑁧𑁫𑁇𑁩 𑀅𑀕𑀢𑀺𑀅𑀕𑀫𑀦

### 𑀫𑀳𑀸𑀲𑀗𑁆𑀕𑀸𑀫

### 𑀅𑀕𑀢𑀺𑀅𑀕𑀫𑀦

(383.)

1852. **𑀓𑀣𑀁 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺**? 𑀅𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀁 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

(384.)

1853. **𑀓𑀣𑀁 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺**? 𑀅𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀁 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

(385.)

1854. **𑀓𑀣𑀁 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺**? 𑀅𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀁 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

(386.)

1855. **𑀓𑀣𑀁 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺**? 𑀅𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑁄 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀏𑀯𑀁 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

1856. _𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀪𑀬𑀸 𑀫𑁄𑀳𑀸,_  
_𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺;_

_𑀲𑀼𑀓𑁆𑀓𑀧𑀓𑁆𑀔𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑀢𑀺𑁇_
