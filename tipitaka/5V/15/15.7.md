# 𑁧𑁫𑁇𑁭 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀺𑀪𑀸𑀕

### 𑀫𑀳𑀸𑀲𑀗𑁆𑀕𑀸𑀫

### 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀺𑀪𑀸𑀕

(398.)

1870\. 𑀓𑀺𑀁 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀓𑀢𑀫𑀸 𑀧𑀼𑀘𑁆𑀙𑀸? 𑀓𑀺𑀦𑁆𑀢𑀺 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀓𑀢𑀫𑀸 𑀧𑀼𑀘𑁆𑀙𑀸? 𑀓𑀤𑀸 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀓𑀢𑀫𑀸 𑀧𑀼𑀘𑁆𑀙𑀸? 𑀓𑀢𑁆𑀣 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀓𑀢𑀫𑀸 𑀧𑀼𑀘𑁆𑀙𑀸?

(399.)

1871\. **𑀓𑀺𑀁 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆**𑀢𑀺 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀅𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑀧𑀼𑀘𑁆𑀙𑀸𑁇 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺—  𑀅𑀝𑁆𑀞𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀢𑁂𑀯𑀻𑀲𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑁆𑀯𑁂𑀅𑀦𑀺𑀬𑀢𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀅𑀝𑁆𑀞𑀸𑀲𑀻𑀢𑀺𑀲𑀢𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑁆𑀯𑀸𑀤𑀲𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼𑀧𑀼𑀘𑁆𑀙𑀸𑁇 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺—  𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑁇 𑀆𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑁇 𑀅𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺—  𑀤𑁆𑀯𑀬𑀁𑀤𑁆𑀯𑀬𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑁇

(400.)

1872\. **𑀓𑀺𑀦𑁆𑀢𑀺 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆**𑀢𑀺 𑀮𑀺𑀗𑁆𑀕𑀧𑀼𑀘𑁆𑀙𑀸, 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀧𑀼𑀘𑁆𑀙𑀸, 𑀆𑀓𑀸𑀭𑀧𑀼𑀘𑁆𑀙𑀸, 𑀯𑀺𑀧𑁆𑀧𑀓𑀸𑀭𑀧𑀼𑀘𑁆𑀙𑀸𑁇 𑀮𑀺𑀗𑁆𑀕𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺—  𑀤𑀻𑀖𑀁 𑀯𑀸 𑀭𑀲𑁆𑀲𑀁 𑀯𑀸 𑀓𑀡𑁆𑀳𑀁 𑀯𑀸 𑀑𑀤𑀸𑀢𑀁 𑀯𑀸𑁇 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀯𑀸 𑀞𑀺𑀢𑀁 𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀯𑀸 𑀦𑀺𑀧𑀦𑁆𑀦𑀁 𑀯𑀸𑁇 𑀆𑀓𑀸𑀭𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀕𑀺𑀳𑀺𑀮𑀺𑀗𑁆𑀕𑁂 𑀯𑀸 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀮𑀺𑀗𑁆𑀕𑁂 𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀮𑀺𑀗𑁆𑀕𑁂 𑀯𑀸𑁇 𑀯𑀺𑀧𑁆𑀧𑀓𑀸𑀭𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁 𑀯𑀸 𑀞𑀺𑀢𑀁 𑀯𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀯𑀸 𑀦𑀺𑀧𑀦𑁆𑀦𑀁 𑀯𑀸𑁇

(401.)

1873\. **𑀓𑀤𑀸 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆**𑀢𑀺 𑀓𑀸𑀮𑀧𑀼𑀘𑁆𑀙𑀸, 𑀲𑀫𑀬𑀧𑀼𑀘𑁆𑀙𑀸, 𑀤𑀺𑀯𑀲𑀧𑀼𑀘𑁆𑀙𑀸, 𑀉𑀢𑀼𑀧𑀼𑀘𑁆𑀙𑀸𑁇 𑀓𑀸𑀮𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀓𑀸𑀮𑁂 𑀯𑀸 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀓𑀸𑀮𑁂 𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀓𑀸𑀮𑁂 𑀯𑀸𑁇 𑀲𑀫𑀬𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑁂 𑀯𑀸 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑀲𑀫𑀬𑁂 𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑁂 𑀯𑀸𑁇 𑀤𑀺𑀯𑀲𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀤𑀺𑀯𑀸 𑀯𑀸 𑀓𑀸𑀍𑀏 𑀯𑀸 𑀚𑀼𑀡𑁆𑀳𑁂 𑀯𑀸𑁇 𑀉𑀢𑀼𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀳𑁂𑀫𑀦𑁆𑀢𑁂 𑀯𑀸 𑀕𑀺𑀫𑁆𑀳𑁂 𑀯𑀸 𑀯𑀲𑁆𑀲𑁂 𑀯𑀸𑁇

(402.)

1874\. **𑀓𑀢𑁆𑀣 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀦𑁆**𑀢𑀺 𑀞𑀸𑀦𑀧𑀼𑀘𑁆𑀙𑀸, 𑀪𑀽𑀫𑀺𑀧𑀼𑀘𑁆𑀙𑀸, 𑀑𑀓𑀸𑀲𑀧𑀼𑀘𑁆𑀙𑀸, 𑀧𑀤𑁂𑀲𑀧𑀼𑀘𑁆𑀙𑀸𑁇 𑀞𑀸𑀦𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀪𑀽𑀫𑀺𑀬𑀸 𑀯𑀸 𑀧𑀣𑀯𑀺𑀬𑀸 𑀯𑀸 𑀥𑀭𑀡𑀺𑀬𑀸 𑀯𑀸 𑀚𑀕𑀢𑀺𑀬𑀸 𑀯𑀸𑁇 𑀪𑀽𑀫𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀪𑀽𑀫𑀺𑀬𑀸 𑀯𑀸 𑀧𑀣𑀯𑀺𑀬𑀸 𑀯𑀸 𑀧𑀩𑁆𑀩𑀢𑁂 𑀯𑀸 𑀧𑀸𑀲𑀸𑀡𑁂 𑀯𑀸 𑀧𑀸𑀲𑀸𑀤𑁂 𑀯𑀸𑁇 𑀑𑀓𑀸𑀲𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂 𑀯𑀸 𑀑𑀓𑀸𑀲𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀯𑀸 𑀑𑀓𑀸𑀲𑁂 𑀉𑀢𑁆𑀢𑀭𑁂 𑀯𑀸 𑀑𑀓𑀸𑀲𑁂 𑀤𑀓𑁆𑀔𑀺𑀡𑁂 𑀯𑀸 𑀑𑀓𑀸𑀲𑁂𑁇 𑀧𑀤𑁂𑀲𑀧𑀼𑀘𑁆𑀙𑀸𑀢𑀺 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑁂 𑀯𑀸 𑀧𑀤𑁂𑀲𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀯𑀸 𑀧𑀤𑁂𑀲𑁂 𑀉𑀢𑁆𑀢𑀭𑁂 𑀯𑀸 𑀧𑀤𑁂𑀲𑁂 𑀤𑀓𑁆𑀔𑀺𑀡𑁂 𑀯𑀸 𑀧𑀤𑁂𑀲𑁂𑀢𑀺𑁇

---

1875\. 𑀫𑀳𑀸𑀲𑀗𑁆𑀕𑀸𑀫𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

1876\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1877\. _𑀯𑀢𑁆𑀣𑀼 𑀦𑀺𑀤𑀸𑀦𑀁 𑀆𑀓𑀸𑀭𑁄,_  
_𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀁 𑀓𑀢𑀸𑀓𑀢𑀁;_  
_𑀓𑀫𑁆𑀫𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘𑁂𑀯,_  
_𑀲𑀫𑀣𑁄 𑀙𑀦𑁆𑀤𑀕𑀸𑀫𑀺 𑀘𑁇_  


1878\. _𑀤𑁄𑀲𑀸 𑀫𑁄𑀳𑀸 𑀪𑀬𑀸 𑀘𑁂𑀯,_  
_𑀲𑀜𑁆𑀜𑀸 𑀦𑀺𑀚𑁆𑀛𑀸𑀧𑀦𑁂𑀦 𑀘;_  
_𑀧𑁂𑀓𑁆𑀔𑀸 𑀧𑀲𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑁄𑀫𑁆𑀳𑀺,_  
_𑀲𑀼𑀢𑀣𑁂𑀭𑀢𑀭𑁂𑀦 𑀘𑁇_  


1879\. _𑀅𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀁,_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀘;_  
_𑀲𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑀦𑀸𑀧𑀺,_  
_𑀫𑀳𑀸𑀲𑀗𑁆𑀕𑀸𑀫𑀜𑀸𑀧𑀦𑀸𑀢𑀺𑁇_
