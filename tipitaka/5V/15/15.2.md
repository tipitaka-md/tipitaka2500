# 𑁧𑁫𑁇𑁨 𑀅𑀕𑀢𑀺𑀅𑀕𑀦𑁆𑀢𑀩𑁆𑀩

### 𑀫𑀳𑀸𑀲𑀗𑁆𑀕𑀸𑀫

### 𑀅𑀕𑀢𑀺𑀅𑀕𑀦𑁆𑀢𑀩𑁆𑀩

(379.)

1847\. **𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀸**𑀢𑀺 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀓𑀣𑀁 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄—  “𑀅𑀬𑀁 𑀫𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀯𑀸 𑀆𑀘𑀭𑀺𑀬𑁄 𑀯𑀸 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀯𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀯𑀸 𑀲𑀫𑀸𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀯𑀸 𑀲𑀫𑀸𑀦𑀸𑀘𑀭𑀺𑀬𑀓𑁄 𑀯𑀸 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑁄 𑀯𑀸 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑁄 𑀯𑀸 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑁄 𑀯𑀸”𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬 𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀭𑀓𑁆𑀔𑀸𑀬 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀸𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀸𑀲𑀼𑀔𑀸𑀬 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

(380.)

1848\. **𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀸**𑀢𑀺 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀓𑀣𑀁 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀅𑀘𑀭𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺, 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺, 𑀅𑀦𑀢𑁆𑀣𑀁 𑀫𑁂 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺, 𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…  𑀅𑀦𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺, 𑀅𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀫𑁂 𑀅𑀫𑀦𑀸𑀧𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀁 𑀅𑀘𑀭𑀺…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀢𑀺…  𑀅𑀢𑁆𑀣𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺 𑀆𑀖𑀸𑀢𑀁 𑀩𑀦𑁆𑀥𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀦𑀯𑀳𑀺 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀆𑀖𑀸𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀢𑁄 𑀓𑀼𑀤𑁆𑀥𑁄 𑀓𑁄𑀥𑀸𑀪𑀺𑀪𑀽𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀸𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀸𑀲𑀼𑀔𑀸𑀬 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

1849\. **𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀸**𑀢𑀺 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀓𑀣𑀁 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺? 𑀭𑀢𑁆𑀢𑁄 𑀭𑀸𑀕𑀯𑀲𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑁄 𑀤𑁄𑀲𑀯𑀲𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑀽𑀍𑀳𑁄 𑀫𑁄𑀳𑀯𑀲𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑁄 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀲𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑀽𑀍𑀳𑁄 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀫𑁄𑀳𑀸𑀪𑀺𑀪𑀽𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺…𑀧𑁂…  𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀸𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀸𑀲𑀼𑀔𑀸𑀬 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

1850\. **𑀦 𑀪𑀬𑀸𑀕𑀢𑀺 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀸**𑀢𑀺 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀓𑀣𑀁 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺? 𑀇𑀥𑁂𑀓𑀘𑁆𑀘𑁄—  “𑀅𑀬𑀁 𑀯𑀺𑀲𑀫𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀯𑀸 𑀕𑀳𑀦𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀯𑀸 𑀩𑀮𑀯𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁄 𑀯𑀸 𑀓𑀓𑁆𑀔𑀍𑀑 𑀨𑀭𑀼𑀲𑁄 𑀚𑀻𑀯𑀺𑀢𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑀁 𑀯𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀪𑀬𑀸 𑀪𑀻𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀸𑀳𑀺𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀸𑀲𑀼𑀔𑀸𑀬 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀇𑀫𑁂𑀳𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀳𑀺 𑀯𑀢𑁆𑀣𑀽𑀳𑀺 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀔𑀢𑀁 𑀉𑀧𑀳𑀢𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀢𑀺, 𑀲𑀸𑀯𑀚𑁆𑀚𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀲𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀘 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀏𑀯𑀁 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

1851\. _𑀙𑀦𑁆𑀤𑀸 𑀤𑁄𑀲𑀸 𑀪𑀬𑀸 𑀫𑁄𑀳𑀸,_  
_𑀬𑁄 𑀥𑀫𑁆𑀫𑀁 𑀅𑀢𑀺𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀦𑀺𑀳𑀻𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀬𑀲𑁄,_  
_𑀓𑀸𑀍𑀅𑀧𑀓𑁆𑀔𑁂𑀯 𑀘𑀦𑁆𑀤𑀺𑀫𑀸𑀢𑀺𑁇_
