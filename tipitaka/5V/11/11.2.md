# 𑁧𑁧𑁇𑁨 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑀺𑀤𑀸𑀦𑀸𑀤𑀺

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑀺𑀤𑀸𑀦𑀸𑀤𑀺

(342.)

1709\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁? 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁?

1710\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀺𑀯𑀸𑀤𑀦𑀺𑀤𑀸𑀦𑀁 𑀯𑀺𑀯𑀸𑀤𑀲𑀫𑀼𑀤𑀬𑀁 𑀯𑀺𑀯𑀸𑀤𑀚𑀸𑀢𑀺𑀓𑀁 𑀯𑀺𑀯𑀸𑀤𑀧𑀪𑀯𑀁 𑀯𑀺𑀯𑀸𑀤𑀲𑀫𑁆𑀪𑀸𑀭𑀁 𑀯𑀺𑀯𑀸𑀤𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀅𑀦𑀼𑀯𑀸𑀤𑀦𑀺𑀤𑀸𑀦𑀁 𑀅𑀦𑀼𑀯𑀸𑀤𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀦𑀼𑀯𑀸𑀤𑀚𑀸𑀢𑀺𑀓𑀁 𑀅𑀦𑀼𑀯𑀸𑀤𑀧𑀪𑀯𑀁 𑀅𑀦𑀼𑀯𑀸𑀤𑀲𑀫𑁆𑀪𑀸𑀭𑀁 𑀅𑀦𑀼𑀯𑀸𑀤𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀤𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀤𑀬𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀚𑀸𑀢𑀺𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀧𑀪𑀯𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀪𑀸𑀭𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀘𑁆𑀘𑀬𑀦𑀺𑀤𑀸𑀦𑀁 𑀓𑀺𑀘𑁆𑀘𑀬𑀲𑀫𑀼𑀤𑀬𑀁 𑀓𑀺𑀘𑁆𑀘𑀬𑀚𑀸𑀢𑀺𑀓𑀁 𑀓𑀺𑀘𑁆𑀘𑀬𑀧𑀪𑀯𑀁 𑀓𑀺𑀘𑁆𑀘𑀬𑀲𑀫𑁆𑀪𑀸𑀭𑀁 𑀓𑀺𑀘𑁆𑀘𑀬𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇

1711\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁?

1712\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀳𑁂𑀢𑀼𑀦𑀺𑀤𑀸𑀦𑀁, 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀤𑀬𑀁, 𑀳𑁂𑀢𑀼𑀚𑀸𑀢𑀺𑀓𑀁, 𑀳𑁂𑀢𑀼𑀧𑀪𑀯𑀁, 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀳𑁂𑀢𑀼𑀦𑀺𑀤𑀸𑀦𑀁, 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀤𑀬𑀁, 𑀳𑁂𑀢𑀼𑀚𑀸𑀢𑀺𑀓𑀁, 𑀳𑁂𑀢𑀼𑀧𑀪𑀯𑀁, 𑀳𑁂𑀢𑀼𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀳𑁂𑀢𑀼𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇

1713\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀓𑀺𑀁𑀦𑀺𑀤𑀸𑀦𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀤𑀬𑀁, 𑀓𑀺𑀁𑀚𑀸𑀢𑀺𑀓𑀁, 𑀓𑀺𑀁𑀧𑀪𑀯𑀁, 𑀓𑀺𑀁𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀓𑀺𑀁𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁?

1714\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀧𑀘𑁆𑀘𑀬𑀦𑀺𑀤𑀸𑀦𑀁, 𑀧𑀘𑁆𑀘𑀬𑀲𑀫𑀼𑀤𑀬𑀁, 𑀧𑀘𑁆𑀘𑀬𑀚𑀸𑀢𑀺𑀓𑀁, 𑀧𑀘𑁆𑀘𑀬𑀧𑀪𑀯𑀁, 𑀧𑀘𑁆𑀘𑀬𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀧𑀘𑁆𑀘𑀬𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂…  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀧𑀘𑁆𑀘𑀬𑀦𑀺𑀤𑀸𑀦𑀁, 𑀧𑀘𑁆𑀘𑀬𑀲𑀫𑀼𑀤𑀬𑀁, 𑀧𑀘𑁆𑀘𑀬𑀚𑀸𑀢𑀺𑀓𑀁, 𑀧𑀘𑁆𑀘𑀬𑀧𑀪𑀯𑀁, 𑀧𑀘𑁆𑀘𑀬𑀲𑀫𑁆𑀪𑀸𑀭𑀁, 𑀧𑀘𑁆𑀘𑀬𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁𑁇
