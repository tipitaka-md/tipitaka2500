# 𑁧𑁧𑁇𑁫 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬

(348.)

1730\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁; 𑀅𑀧𑀺 𑀘, 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀯𑀤𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀯𑀸 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀯𑀸 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺 𑀯𑀸𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀦𑀸𑀦𑀸𑀯𑀸𑀤𑁄 𑀅𑀜𑁆𑀜𑀣𑀸𑀯𑀸𑀤𑁄 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀸𑀬 𑀯𑁄𑀳𑀸𑀭𑁄 𑀫𑁂𑀥𑀓𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂 𑀲𑀁𑀖𑁄 𑀯𑀺𑀯𑀤𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀤𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀅𑀦𑀼𑀯𑀤𑀫𑀸𑀦𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀢𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀏𑀯𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 (𑁧)

1731\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁; 𑀅𑀧𑀺 𑀘 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀼𑀯𑀤𑀦𑁆𑀢𑀺 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸𑁇 𑀬𑁄 𑀢𑀢𑁆𑀣 𑀅𑀦𑀼𑀯𑀸𑀤𑁄 𑀅𑀦𑀼𑀯𑀤𑀦𑀸 𑀅𑀦𑀼𑀮𑁆𑀮𑀧𑀦𑀸 𑀅𑀦𑀼𑀪𑀡𑀦𑀸 𑀅𑀦𑀼𑀲𑀫𑁆𑀧𑀯𑀗𑁆𑀓𑀢𑀸 𑀅𑀩𑁆𑀪𑀼𑀲𑁆𑀲𑀳𑀦𑀢𑀸 𑀅𑀦𑀼𑀩𑀮𑀧𑁆𑀧𑀤𑀸𑀦𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑁂 𑀲𑀁𑀖𑁄 𑀯𑀺𑀯𑀤𑀢𑀺, 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀤𑀢𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀅𑀦𑀼𑀯𑀤𑀫𑀸𑀦𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀢𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀏𑀯𑀁 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 (𑁨)

1732\. 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁; 𑀅𑀧𑀺 𑀘 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬? 𑀧𑀜𑁆𑀘𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀲𑀢𑁆𑀢𑀧𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑁂 𑀲𑀁𑀖𑁄 𑀯𑀺𑀯𑀤𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀤𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀅𑀦𑀼𑀯𑀤𑀫𑀸𑀦𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀢𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀏𑀯𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 (𑁩)

1733\. 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀦 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁; 𑀅𑀧𑀺 𑀘 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀬𑀣𑀸 𑀓𑀣𑀁 𑀯𑀺𑀬? 𑀬𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀺𑀘𑁆𑀘𑀬𑀢𑀸 𑀓𑀭𑀡𑀻𑀬𑀢𑀸 𑀅𑀧𑀮𑁄𑀓𑀦𑀓𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑀓𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀫𑁆𑀫𑀁, 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂 𑀲𑀁𑀖𑁄 𑀯𑀺𑀯𑀤𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀯𑀤𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀅𑀦𑀼𑀯𑀤𑀫𑀸𑀦𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀢𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀏𑀯𑀁 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀳𑁄𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 (𑁪)
