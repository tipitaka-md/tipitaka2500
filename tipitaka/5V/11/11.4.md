# 𑁧𑁧𑁇𑁪 𑀅𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺

(344.)

1718\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀺𑀁 𑀧𑀦 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀢𑀺? 𑀆𑀫, 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀑𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀑𑀫𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀸 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1719\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺, 𑀓𑀢𑀺𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺?

1720\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀏𑀓𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦—  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦; 𑀢𑀻𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂, 𑀕𑀡𑀫𑀚𑁆𑀛𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

(345.)

1721\. 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀺𑀁 𑀧𑀦 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀢𑀺? 𑀆𑀫, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸, 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀅𑀫𑀽𑀮𑀺𑀓𑀸𑀬 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1722\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺, 𑀓𑀢𑀺𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺?

1723\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀕𑀭𑀼𑀓𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀏𑀓𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦—  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦; 𑀏𑀓𑀫𑁆𑀳𑀺 𑀞𑀸𑀦𑁂—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂; 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘𑁇 𑀬𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀮𑀳𑀼𑀓𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀏𑀓𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦—  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦; 𑀢𑀻𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀕𑀡𑀫𑀚𑁆𑀛𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

1724\. 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀺𑀁 𑀧𑀦 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀢𑀺? 𑀆𑀫, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1725\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺, 𑀓𑀢𑀺𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺?

1726\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁—  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀬𑀸 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀦 𑀓𑀢𑀫𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦, 𑀦 𑀓𑀢𑀫𑀫𑁆𑀳𑀺 𑀞𑀸𑀦𑁂, 𑀦 𑀓𑀢𑀫𑁂𑀦 𑀲𑀫𑀣𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀬𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀮𑀳𑀼𑀓𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀏𑀓𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦—  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦; 𑀢𑀻𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂, 𑀕𑀡𑀫𑀚𑁆𑀛𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

1727\. 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀺𑀁 𑀧𑀦 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀢𑀺? 𑀆𑀫, 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑁇 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀤𑁆𑀯𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸; 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀪𑁂𑀤𑀓𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲—  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀸 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1728\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀓𑀢𑀺𑀳𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀳𑀺, 𑀓𑀢𑀺𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼, 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺?

1729\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁—  𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀬𑀸 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀦 𑀓𑀢𑀫𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦, 𑀦 𑀓𑀢𑀫𑀫𑁆𑀳𑀺 𑀞𑀸𑀦𑁂, 𑀦 𑀓𑀢𑀫𑁂𑀦 𑀲𑀫𑀣𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀬𑀸 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀕𑀭𑀼𑀓𑀸 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀏𑀓𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦—  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦; 𑀏𑀓𑀫𑁆𑀳𑀺 𑀞𑀸𑀦𑁂—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂; 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘𑁇 𑀬𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀮𑀳𑀼𑀓𑀸 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀏𑀓𑁂𑀦 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦—  𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑁂𑀦; 𑀢𑀻𑀲𑀼 𑀞𑀸𑀦𑁂𑀲𑀼—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂, 𑀕𑀡𑀫𑀚𑁆𑀛𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂; 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇
