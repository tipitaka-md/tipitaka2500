# 𑁧𑁧𑁇𑁩 𑀅𑀥𑀺𑀓𑀭𑀡𑀫𑀽𑀮𑀸𑀤𑀺

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀫𑀽𑀮𑀸𑀤𑀺

(343.)

1715\. 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀺 𑀫𑀽𑀮𑀸𑀦𑀺, 𑀓𑀢𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀫𑀽𑀮𑀸𑀦𑀺, 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇

1716\. 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀸𑀦𑀺 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀫𑀽𑀮𑀸𑀦𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀤𑁆𑀯𑀸𑀤𑀲 𑀫𑀽𑀮𑀸𑀦𑀺, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀘𑀼𑀤𑁆𑀤𑀲 𑀫𑀽𑀮𑀸𑀦𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀙 𑀫𑀽𑀮𑀸𑀦𑀺, 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀏𑀓𑀁 𑀫𑀽𑀮𑀁, 𑀲𑀁𑀖𑁄—  𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀇𑀫𑀸𑀦𑀺 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀫𑀽𑀮𑀸𑀦𑀺𑁇

1717\. 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑁂 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀪𑁂𑀤𑀓𑀭𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸, 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸—  𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀇𑀫𑁂 𑀢𑁂𑀢𑁆𑀢𑀺𑀁𑀲 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇
