# 𑁧𑁧𑁇𑁬 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀸𑀭

* 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

* 𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀸𑀭

(349.)

1734\. 𑀬𑀢𑁆𑀣 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁? 𑀬𑀢𑁆𑀣 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸? 𑀬𑀢𑁆𑀣 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸? 𑀬𑀢𑁆𑀣 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄? 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄?
