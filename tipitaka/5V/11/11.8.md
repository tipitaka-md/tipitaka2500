# 𑁧𑁧𑁇𑁮 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

### 𑀲𑀁𑀲𑀝𑁆𑀞𑀯𑀸𑀭

(351.)

1736\. 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀉𑀤𑀸𑀳𑀼 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸? 𑀮𑀩𑁆𑀪𑀸 𑀘 𑀧𑀦𑀺𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁? 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀦𑁆𑀢𑀺 𑀯𑀸…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀢𑀺 𑀯𑀸…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀢𑀺 𑀯𑀸…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑀢𑀺 𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀉𑀤𑀸𑀳𑀼 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸? 𑀮𑀩𑁆𑀪𑀸 𑀘 𑀧𑀦𑀺𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁?

1737\. 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸; 𑀦 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀦𑁆𑀢𑀺 𑀯𑀸…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀢𑀺 𑀯𑀸…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀢𑀺 𑀯𑀸…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀯𑀸 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑀢𑀺 𑀯𑀸—  𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸 𑀲𑀁𑀲𑀝𑁆𑀞𑀸, 𑀦𑁄 𑀯𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑀸; 𑀦 𑀘 𑀮𑀩𑁆𑀪𑀸 𑀇𑀫𑁂𑀲𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀺𑀩𑁆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀁𑁇
