# 𑁧𑁧𑁇𑁭 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀸𑀯𑀸𑀭

* 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

* 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀸𑀯𑀸𑀭

(350.)

1735\. 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺—  𑀬𑀢𑁆𑀣 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄, 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀦 𑀢𑀢𑁆𑀣 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀦 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀦 𑀢𑀢𑁆𑀣 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀦 𑀢𑀢𑁆𑀣 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀦 𑀢𑀢𑁆𑀣 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑀬𑁂 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁂𑀦 𑀘…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸𑀬 𑀘…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀬 𑀘…𑀧𑁂…  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺—  𑀬𑀢𑁆𑀣 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄 𑀢𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄, 𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀢𑀢𑁆𑀣 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄, 𑀦 𑀢𑀢𑁆𑀣 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀦 𑀢𑀢𑁆𑀣 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀦 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀦 𑀢𑀢𑁆𑀣 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀦 𑀢𑀢𑁆𑀣 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑁇
