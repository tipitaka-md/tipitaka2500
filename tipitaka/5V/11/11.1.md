# 𑁧𑁧𑁇𑁧 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀪𑁂𑀤𑀸𑀤𑀺

* 𑀅𑀥𑀺𑀓𑀭𑀡𑀪𑁂𑀤

* 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀪𑁂𑀤𑀸𑀤𑀺

(340.)

1699\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀺𑁇

1700\. 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀸? 𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀤𑀲 𑀉𑀓𑁆𑀓𑁄𑀝𑀸𑁇 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑀸, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀉𑀓𑁆𑀓𑁄𑀝𑀸, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀢𑀬𑁄 𑀉𑀓𑁆𑀓𑁄𑀝𑀸, 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀏𑀓𑁄 𑀉𑀓𑁆𑀓𑁄𑀝𑁄—  𑀇𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀇𑀫𑁂 𑀤𑀲 𑀉𑀓𑁆𑀓𑁄𑀝𑀸𑁇

1701\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀓𑀢𑀺 𑀲𑀫𑀣𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺? 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀓𑀢𑀺 𑀲𑀫𑀣𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀓𑀢𑀺 𑀲𑀫𑀣𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺? 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀓𑀢𑀺 𑀲𑀫𑀣𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺?

1702\. 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀲𑀫𑀣𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺𑁇 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀫𑀣𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀢𑀬𑁄 𑀲𑀫𑀣𑁂 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺𑁇 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀲𑀫𑀣𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺𑁇

(341.)

1703\. 𑀓𑀢𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀸? 𑀓𑀢𑀺𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀁 𑀧𑀲𑀯𑀢𑀺? 𑀓𑀢𑀺𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺? 𑀓𑀢𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺?

1704\. 𑀤𑁆𑀯𑀸𑀤𑀲 𑀉𑀓𑁆𑀓𑁄𑀝𑀸𑁇 𑀤𑀲𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀁 𑀧𑀲𑀯𑀢𑀺𑁇 𑀘𑀢𑀽𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺?

1705\. 𑀓𑀢𑀫𑁂 𑀤𑁆𑀯𑀸𑀤𑀲 𑀉𑀓𑁆𑀓𑁄𑀝𑀸? 𑀅𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁, 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀫𑁆𑀫𑀁, 𑀧𑀼𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀓𑀫𑁆𑀫𑀁, 𑀅𑀦𑀺𑀳𑀢𑀁, 𑀤𑀼𑀦𑁆𑀦𑀺𑀳𑀢𑀁, 𑀧𑀼𑀦 𑀦𑀺𑀳𑀦𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀁, 𑀤𑀼𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀁, 𑀧𑀼𑀦 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁, 𑀤𑀼𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁, 𑀧𑀼𑀦 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺—  𑀇𑀫𑁂 𑀤𑁆𑀯𑀸𑀤𑀲 𑀉𑀓𑁆𑀓𑁄𑀝𑀸𑁇

1706\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀤𑀲𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀁 𑀧𑀲𑀯𑀢𑀺? 𑀢𑀢𑁆𑀣 𑀚𑀸𑀢𑀓𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀚𑀸𑀢𑀓𑀁 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀕𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀕𑀢𑀁 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀤𑀲𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀁 𑀧𑀲𑀯𑀢𑀺𑁇

1707\. 𑀓𑀢𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺? 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀘𑀢𑀽𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺𑁇

1708\. 𑀓𑀢𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺? 𑀢𑀤𑀳𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁄 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁, 𑀓𑀸𑀭𑀓𑁄 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁, 𑀙𑀦𑁆𑀤𑀤𑀸𑀬𑀓𑁄 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁—  𑀇𑀫𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇
