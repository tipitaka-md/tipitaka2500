# 𑁨𑁧𑁇𑁨 𑀅𑀢𑁆𑀣𑀯𑀲𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕

* 𑀅𑀢𑁆𑀣𑀯𑀲𑀯𑀕𑁆𑀕

(498.)

2410\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬, 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑀢𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2411\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬, 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2412\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2413\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀯𑁂𑀭𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀯𑁂𑀭𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2414\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀯𑀚𑁆𑀚𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀯𑀚𑁆𑀚𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2415\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀪𑀬𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀪𑀬𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2416\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2417\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀸𑀬, 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀸𑀦𑀁 𑀧𑀓𑁆𑀔𑀼𑀧𑀘𑁆𑀙𑁂𑀤𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2418\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

2419\. 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬—  𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇

---

2420\. 𑀅𑀢𑁆𑀣𑀯𑀲𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
