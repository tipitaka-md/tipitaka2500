# 𑁨𑁧𑁇𑁫 𑀦𑀯𑀲𑀗𑁆𑀕𑀳𑀯𑀕𑁆𑀕

* 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕

* 𑀦𑀯𑀲𑀗𑁆𑀕𑀳𑀯𑀕𑁆𑀕

(501.)

2434\. 𑀦𑀯𑀲𑀗𑁆𑀕𑀳𑀸—  𑀯𑀢𑁆𑀣𑀼𑀲𑀗𑁆𑀕𑀳𑁄, 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀲𑀗𑁆𑀕𑀳𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀗𑁆𑀕𑀳𑁄, 𑀦𑀺𑀤𑀸𑀦𑀲𑀗𑁆𑀕𑀳𑁄, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀗𑁆𑀕𑀳𑁄, 𑀔𑀦𑁆𑀥𑀲𑀗𑁆𑀕𑀳𑁄, 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀗𑁆𑀕𑀳𑁄, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀗𑁆𑀕𑀳𑁄, 𑀲𑀫𑀣𑀲𑀗𑁆𑀕𑀳𑁄𑀢𑀺𑁇

2435\. 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂 𑀲𑀘𑁂 𑀉𑀪𑁄 𑀅𑀢𑁆𑀣𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀣𑀼 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀯𑀢𑁆𑀣𑀼𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀧𑀝𑀺𑀜𑁆𑀜𑀸 𑀲𑁄𑀢𑀩𑁆𑀩𑀸𑁇 𑀉𑀪𑀺𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀧𑀝𑀺𑀜𑁆𑀜𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀪𑁄𑀧𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀯𑀽𑀧𑀲𑀫𑀺𑀢𑁂 𑀉𑀪𑁄𑀧𑀺 𑀢𑀼𑀝𑁆𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀣𑀸”𑀢𑀺𑁇 𑀲𑀘𑁂 𑀆𑀳𑀁𑀲𑀼—  “𑀉𑀪𑁄𑀧𑀺 𑀢𑀼𑀝𑁆𑀞𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺, 𑀲𑀁𑀖𑁂𑀦 𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀲𑀫𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀅𑀮𑀚𑁆𑀚𑀼𑀲𑁆𑀲𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀺𑀲𑀸 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀩𑀸𑀮𑀼𑀲𑁆𑀲𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀺𑀲𑀸 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄 𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀬𑁂𑀦 𑀯𑀺𑀦𑀬𑁂𑀦 𑀬𑁂𑀦 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂𑀦 𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇 𑀢𑀣𑀸 𑀢𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁂𑀢𑀩𑁆𑀩𑀁𑁇

2436\. 𑀯𑀢𑁆𑀣𑀼 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀁, 𑀕𑁄𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦𑀸𑀫𑀁 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀁, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇

2437\. 𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2438\. 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2439\. 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑁄𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2440\. 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2441\. 𑀲𑀼𑀓𑁆𑀓𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀻𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2442\. 𑀓𑀸𑀬𑀲𑀁𑀲𑀕𑁆𑀕𑁄𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2443\. 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀯𑀸𑀘𑀸𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2444\. 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2445\. 𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁆𑀢𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2446\. 𑀲𑀜𑁆𑀜𑀸𑀘𑀺𑀓𑀸𑀬 𑀓𑀼𑀝𑀺𑀁 𑀓𑀸𑀭𑀸𑀧𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2447\. 𑀫𑀳𑀮𑁆𑀮𑀓𑀁 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀸𑀭𑀸𑀧𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2448\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2449\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀓𑀺𑀜𑁆𑀘𑀺 𑀤𑁂𑀲𑀁 𑀮𑁂𑀲𑀫𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀤𑀸𑀬 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2450\. 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2451\. 𑀪𑁂𑀤𑀓𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2452\. 𑀤𑀼𑀩𑁆𑀩𑀘𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑁇

2453\. 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘…𑀧𑁂… 𑁇

2454\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀡𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀘𑁂𑀯 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘—  𑀤𑀼𑀓𑁆𑀓𑀝𑀦𑁆𑀢𑀺 𑀦𑀸𑀫𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑀸𑀢𑀺𑁇

---

2455\. 𑀦𑀯𑀲𑀗𑁆𑀕𑀳𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

2456\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2457\. _𑀅𑀧𑀮𑁄𑀓𑀦𑀁 𑀜𑀢𑁆𑀢𑀺 𑀘,_  
_𑀤𑀼𑀢𑀺𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀘;_  
_𑀯𑀢𑁆𑀣𑀼 𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀁,_  
_𑀲𑀻𑀫𑀸 𑀧𑀭𑀺𑀲𑀫𑁂𑀯 𑀘𑁇_  


2458\. _𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸 𑀘,_  
_𑀧𑀝𑀺𑀜𑁆𑀜𑀸 𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑁄;_  
_𑀯𑀢𑁆𑀣𑀼 𑀲𑀁𑀖𑀧𑀼𑀕𑁆𑀕𑀮𑀜𑁆𑀘,_  
_𑀜𑀢𑁆𑀢𑀺𑀁 𑀦 𑀧𑀘𑁆𑀙𑀸 𑀜𑀢𑁆𑀢𑀺 𑀘𑁇_  


2459\. _𑀯𑀢𑁆𑀣𑀼𑀁 𑀲𑀁𑀖𑀧𑀼𑀕𑁆𑀕𑀮𑀜𑁆𑀘,_  
_𑀲𑀸𑀯𑀦𑀁 𑀅𑀓𑀸𑀮𑁂𑀦 𑀘;_  
_𑀅𑀢𑀺𑀔𑀼𑀤𑁆𑀤𑀓𑀸 𑀫𑀳𑀦𑁆𑀢𑀸 𑀘,_  
_𑀔𑀡𑁆𑀟𑀘𑁆𑀙𑀸𑀬𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀓𑀸𑁇_  


2460\. _𑀩𑀳𑀺𑀦𑀤𑀻 𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀘,_  
_𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁂 𑀘 𑀪𑀺𑀦𑁆𑀤𑀢𑀺;_  
_𑀅𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑀢𑀺 𑀲𑀻𑀫𑀸𑀬,_  
_𑀘𑀢𑀼 𑀧𑀜𑁆𑀘 𑀘 𑀯𑀕𑁆𑀕𑀺𑀓𑀸𑁇_  


2461\. _𑀤𑀲 𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑀸 𑀘,_  
_𑀅𑀦𑀸𑀳𑀝𑀸 𑀘 𑀆𑀳𑀝𑀸;_  
_𑀓𑀫𑁆𑀫𑀧𑀢𑁆𑀢𑀸 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸,_  
_𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀸 𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑁇_  


2462\. _𑀅𑀧𑀮𑁄𑀓𑀦𑀁 𑀧𑀜𑁆𑀘𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀜𑀢𑁆𑀢𑀺 𑀘 𑀦𑀯𑀞𑀸𑀦𑀺𑀓𑀸;_  
_𑀜𑀢𑁆𑀢𑀺 𑀤𑀼𑀢𑀺𑀬𑀁 𑀲𑀢𑁆𑀢𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀘𑀢𑀼𑀢𑁆𑀣𑀸 𑀲𑀢𑁆𑀢𑀞𑀸𑀦𑀺𑀓𑀸𑁇_  


2463\. _𑀲𑀼𑀝𑁆𑀞𑀼 𑀨𑀸𑀲𑀼 𑀘 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀼,_  
_𑀧𑁂𑀲𑀮𑀸 𑀘𑀸𑀧𑀺 𑀆𑀲𑀯𑀸;_  
_𑀯𑁂𑀭𑀯𑀚𑁆𑀚𑀪𑀬𑀜𑁆𑀘𑁂𑀯,_  
_𑀅𑀓𑀼𑀲𑀮𑀁 𑀕𑀺𑀳𑀻𑀦𑀜𑁆𑀘𑁇_  


2464\. _𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁,_  
_𑀧𑀲𑀦𑁆𑀦𑀸 𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀧𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸 𑀘𑁂𑀯,_  
_𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀘𑁇_  


2465\. _𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀜𑁆𑀘 𑀞𑀧𑀦𑀸,_  
_𑀧𑀯𑀸𑀭𑀡𑀜𑁆𑀘 𑀞𑀧𑀦𑀁;_  
_𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀸 𑀦𑀺𑀬𑀲𑁆𑀲𑀜𑁆𑀘,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬 𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑁇_  


2466\. _𑀉𑀓𑁆𑀔𑁂𑀧𑀦 𑀧𑀭𑀺𑀯𑀸𑀲𑀁,_  
_𑀫𑀽𑀮𑀫𑀸𑀦𑀢𑁆𑀢𑀅𑀩𑁆𑀪𑀸𑀦𑀁;_  
_𑀑𑀲𑀸𑀭𑀡𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁,_  
_𑀢𑀣𑁂𑀯 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


2467\. _𑀅𑀧𑀮𑁄𑀓𑀦𑀜𑀢𑁆𑀢𑀺 𑀘,_  
_𑀤𑀼𑀢𑀺𑀬𑀜𑁆𑀘 𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀁;_  
_𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁,_  
_𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄 𑀲𑀢𑀺𑁇_  


2468\. _𑀅𑀫𑀽𑀍𑀳𑀧𑀝𑀺𑀬𑁂𑀪𑀼𑀬𑁆𑀬,_  
_𑀧𑀸𑀧𑀺𑀬 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑀁;_  
_𑀯𑀢𑁆𑀣𑀼 𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺,_  
_𑀦𑀺𑀤𑀸𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑀦 𑀘𑁇_  


2469\. _𑀔𑀦𑁆𑀥𑀸 𑀘𑁂𑀯 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸,_  
_𑀅𑀥𑀺𑀓𑀭𑀡𑀫𑁂𑀯 𑀘;_  
_𑀲𑀫𑀣𑀸 𑀲𑀗𑁆𑀕𑀳𑀸 𑀘𑁂𑀯,_  
_𑀦𑀸𑀫𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀸 𑀢𑀣𑀸𑀢𑀺𑁇_  


---

2470\. 𑀧𑀭𑀺𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

2471\. 𑀧𑀭𑀺𑀯𑀸𑀭𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇



---

2472\. 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀕𑀸𑀣𑀸



2473\. _𑀩𑀼𑀩𑁆𑀩𑀸𑀘𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀜𑁆𑀘,_  
_𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀘 𑀢𑀳𑀺𑀁 𑀢𑀳𑀺𑀁;_  
_𑀤𑀻𑀧𑀦𑀸𑀫𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀼𑀢𑀥𑀸𑀭𑀺𑀁 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄𑁇_  


2474\. _𑀇𑀫𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀲𑀗𑁆𑀔𑁂𑀧𑀁,_  
_𑀲𑀚𑁆𑀛𑀸𑀫𑀕𑁆𑀕𑁂𑀦 𑀫𑀚𑁆𑀛𑀺𑀫𑁂;_  
_𑀘𑀺𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸 𑀮𑀺𑀔𑀸𑀧𑁂𑀲𑀺,_  
_𑀲𑀺𑀲𑁆𑀲𑀓𑀸𑀦𑀁 𑀲𑀼𑀔𑀸𑀯𑀳𑀁𑁇_  


2475\. _𑀧𑀭𑀺𑀯𑀸𑀭𑀦𑁆𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀲𑀮𑀓𑁆𑀔𑀡𑀁;_  
_𑀅𑀢𑁆𑀣𑀁 𑀅𑀢𑁆𑀣𑁂𑀦 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂𑁇_  


2476\. _𑀲𑀸𑀲𑀦𑀁 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀲𑀺,_  
_𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑀁𑀯 𑀲𑀸𑀕𑀭𑁄;_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑀁 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑁄,_  
_𑀓𑀼𑀢𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀁𑁇_  


2477\. _𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀯𑀢𑁆𑀣𑀼 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺,_  
_𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀏𑀓𑀢𑁄 𑀉𑀪𑀢𑁄 𑀘𑁂𑀯,_  
_𑀮𑁄𑀓𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺 𑀯𑀚𑁆𑀚𑀢𑁄𑁇_  


2478\. _𑀬𑀲𑁆𑀲 𑀚𑀸𑀬𑀢𑀺 𑀯𑀺𑀫𑀢𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀘𑀓𑁆𑀓𑀯𑀢𑁆𑀢𑀺 𑀫𑀳𑀸𑀲𑁂𑀦𑁂,_  
_𑀫𑀺𑀕𑀫𑀚𑁆𑀛𑁂𑀯 𑀓𑁂𑀲𑀭𑀺𑁇_  


2479\. _𑀭𑀯𑀺 𑀭𑀁𑀲𑀺𑀧𑀭𑀺𑀓𑁆𑀓𑀺𑀡𑁆𑀡𑁄,_  
_𑀘𑀦𑁆𑀤𑁄 𑀢𑀸𑀭𑀕𑀡𑁂 𑀬𑀣𑀸;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀧𑀭𑀺𑀲𑀸𑀬,_  
_𑀕𑀡𑀫𑀚𑁆𑀛𑁂𑀯 𑀦𑀸𑀬𑀓𑁄;_  
_𑀏𑀯𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦 𑀲𑁄𑀪𑀢𑀻𑀢𑀺𑁇_
