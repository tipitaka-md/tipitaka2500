# 𑁧𑁦𑁇𑁨 𑀘𑀢𑀼𑀯𑀺𑀧𑀢𑁆𑀢𑀺

* 𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

* 𑀘𑀢𑀼𑀯𑀺𑀧𑀢𑁆𑀢𑀺

(336.)

1641\. _“𑀬𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳 𑀅𑀓𑀺𑀢𑁆𑀢𑀬𑀺 𑀦𑁄,_  
_𑀢𑀁 𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀅𑀦𑀜𑁆𑀜𑀣𑀸;_  
_𑀅𑀜𑁆𑀜𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀤𑀺𑀗𑁆𑀖 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀕𑀭𑀼𑀓 𑀮𑀳𑀼𑀓𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀸𑀯𑀲𑁂𑀲𑀁;_  
_𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀜𑁆𑀘 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁,_  
_𑀬𑁂 𑀘 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸𑁇_  


1642\. _𑀲𑀸𑀥𑀸𑀭𑀡𑀁 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀁,_  
_𑀯𑀺𑀪𑀢𑁆𑀢𑀺𑀬𑁄 𑀘 𑀬𑁂𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺;_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺𑀧𑁂𑀢𑀸𑀦𑀺 𑀯𑀺𑀬𑀸𑀓𑀭𑁄𑀳𑀺,_  
_𑀳𑀦𑁆𑀤 𑀯𑀸𑀓𑁆𑀬𑀁 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂”𑁇_  


1643\. _“𑀏𑀓𑀢𑀺𑀁𑀲𑀸 𑀬𑁂 𑀕𑀭𑀼𑀓𑀸,_  
_𑀅𑀝𑁆𑀞𑁂𑀢𑁆𑀣 𑀅𑀦𑀯𑀲𑁂𑀲𑀸;_  
_𑀬𑁂 𑀕𑀭𑀼𑀓𑀸 𑀢𑁂 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸,_  
_𑀬𑁂 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀲𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺;_  
_𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄,_  
_‘𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀻’𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1644\. _𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸,_  
_𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁;_  
_𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀬𑁄 𑀘𑀸𑀬𑀁,_  
_𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀳𑀲𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄;_  
_𑀅𑀬𑀁 𑀲𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀫𑀢𑀸𑁇_  


1645\. _𑀯𑀺𑀧𑀭𑀻𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺,_  
_𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑀸;_  
_𑀅𑀩𑁆𑀪𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀫𑁄𑀳𑀧𑀸𑀭𑀼𑀢𑀸;_  
_𑀅𑀬𑀁 𑀲𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀫𑀢𑀸𑁇_  


1646\. 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀅𑀲𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺, 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸—  ‘𑀬𑁄 𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀲𑀢𑀺, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀸’𑀢𑀺 𑀪𑀡𑀢𑀺, 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀲𑀽𑀧𑀁 𑀯𑀸 𑀑𑀤𑀦𑀁 𑀯𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇 𑀅𑀬𑀁 𑀲𑀸 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑁆𑀫𑀢𑀸𑁇

1647\. _𑀏𑀓𑀸𑀤𑀲 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸,_  
_𑀢𑁂 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀅𑀝𑁆𑀞 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸;_  
_𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀘𑀡𑁆𑀟𑀓𑀸𑀍𑀈 𑀘,_  
_𑀇𑀫𑁂 𑀢𑁂 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸”𑁇_
