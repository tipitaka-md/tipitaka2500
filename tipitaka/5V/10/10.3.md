# 𑁧𑁦𑁇𑁩 𑀙𑁂𑀤𑀦𑀓𑀸𑀤𑀺

### 𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

### 𑀙𑁂𑀤𑀦𑀓𑀸𑀤𑀺

(337.)

1648\. “𑀓𑀢𑀺 𑀙𑁂𑀤𑀦𑀓𑀸𑀦𑀺? 𑀓𑀢𑀺 𑀪𑁂𑀤𑀦𑀓𑀸𑀦𑀺? 𑀓𑀢𑀺 𑀉𑀤𑁆𑀤𑀸𑀮𑀦𑀓𑀸𑀦𑀺? 𑀓𑀢𑀺 𑀅𑀦𑀜𑁆𑀜𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺? 𑀓𑀢𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑁄? 𑀓𑀢𑀺 𑀲𑀸𑀫𑀻𑀘𑀺𑀬𑁄? 𑀓𑀢𑀺 𑀧𑀭𑀫𑀸𑀦𑀺?

1649\. _𑀓𑀢𑀺 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸”𑁇_  


1650\. “𑀙 𑀙𑁂𑀤𑀦𑀓𑀸𑀦𑀺𑁇 𑀏𑀓𑀁 𑀪𑁂𑀤𑀦𑀓𑀁𑁇 𑀏𑀓𑀁 𑀉𑀤𑁆𑀤𑀸𑀮𑀦𑀓𑀁𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀦𑀜𑁆𑀜𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺𑁇 𑀘𑀢𑀲𑁆𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑁄𑁇 𑀲𑀢𑁆𑀢 𑀲𑀸𑀫𑀻𑀘𑀺𑀬𑁄𑁇 𑀘𑀼𑀤𑁆𑀤𑀲 𑀧𑀭𑀫𑀸𑀦𑀺𑁇

1651\. _𑀲𑁄𑀤𑀲 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸”𑁇_
