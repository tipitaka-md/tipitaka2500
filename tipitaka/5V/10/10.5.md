# 𑁧𑁦𑁇𑁫 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀤𑀺𑀆𑀧𑀢𑁆𑀢𑀺

### 𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀤𑀺𑀆𑀧𑀢𑁆𑀢𑀺

(339.)

1677\. _“𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀘𑀼𑀢𑁄 𑀧𑀭𑀤𑁆𑀥𑁄 𑀪𑀝𑁆𑀞𑁄 𑀘,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸 𑀳𑀺 𑀦𑀺𑀭𑀗𑁆𑀓𑀢𑁄;_  
_𑀲𑀁𑀯𑀸𑀲𑁄𑀧𑀺 𑀢𑀳𑀺𑀁 𑀦𑀢𑁆𑀣𑀺,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1678\. _“𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁄”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀲𑀁𑀖𑁄𑀯 𑀤𑁂𑀢𑀺 𑀧𑀭𑀺𑀯𑀸𑀲𑀁,_  
_𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺 𑀅𑀩𑁆𑀪𑁂𑀢𑀺,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1679\. _“𑀅𑀦𑀺𑀬𑀢𑁄”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀅𑀦𑀺𑀬𑀢𑁄 𑀦 𑀦𑀺𑀬𑀢𑁄,_  
_𑀅𑀦𑁂𑀓𑀁𑀲𑀺𑀓𑀢𑀁 𑀧𑀤𑀁;_  
_𑀢𑀺𑀡𑁆𑀡𑀫𑀜𑁆𑀜𑀢𑀭𑀁 𑀞𑀸𑀦𑀁,_  
_“𑀅𑀦𑀺𑀬𑀢𑁄”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1680\. _“𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀏𑀓𑀲𑁆𑀲 𑀫𑀽𑀮𑁂 𑀬𑁄 𑀤𑁂𑀲𑁂𑀢𑀺,_  
_𑀬𑁄 𑀘 𑀢𑀁 𑀧𑀝𑀺𑀕𑀡𑁆𑀳𑀢𑀺;_  
_𑀅𑀘𑁆𑀘𑀬𑁄 𑀢𑁂𑀦 𑀲𑀫𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1681\. _“𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀕𑀡𑀫𑀚𑁆𑀛𑁂,_  
_𑀏𑀓𑀲𑁆𑀲𑁂𑀯 𑀘 𑀏𑀓𑀢𑁄;_  
_𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸𑀦 𑀤𑁂𑀲𑁂𑀢𑀺,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1682\. _“𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀧𑀸𑀢𑁂𑀢𑀺 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀁 𑀅𑀧𑀭𑀚𑁆𑀛𑀢𑀺;_  
_𑀘𑀺𑀢𑁆𑀢𑀲𑀫𑁆𑀫𑁄𑀳𑀦𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1683\. _“𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀓𑀺𑀘𑁆𑀙𑀸 𑀮𑀤𑁆𑀥𑀸𑀬 𑀪𑁄𑀚𑀦𑀁;_  
_𑀲𑀸𑀫𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬,_  
_“𑀕𑀸𑀭𑀬𑁆𑀳𑀦𑁆”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1684\. _𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀸𑀲𑀼 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸 𑀙𑀦𑁆𑀤𑀸𑀬,_  
_𑀯𑁄𑀲𑀸𑀲𑀢𑀺 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁;_  
_𑀅𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀢𑀳𑀺𑀁 𑀪𑀼𑀜𑁆𑀚𑁂,_  
_“𑀕𑀸𑀭𑀬𑁆𑀳𑀦𑁆”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1685\. _𑀲𑀤𑁆𑀥𑀸𑀘𑀺𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀅𑀧𑁆𑀧𑀪𑁄𑀕𑀁 𑀅𑀦𑀸𑀍𑀇𑀬𑀁;_  
_𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀢𑀳𑀺𑀁 𑀪𑀼𑀜𑁆𑀚𑁂,_  
_“𑀕𑀸𑀭𑀬𑁆𑀳𑀦𑁆”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1686\. _𑀬𑁄 𑀘𑁂 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑁄,_  
_𑀲𑀸𑀲𑀗𑁆𑀓𑁂 𑀲𑀪𑀬𑀸𑀦𑀓𑁂;_  
_𑀅𑀯𑀺𑀤𑀺𑀢𑀁 𑀢𑀳𑀺𑀁 𑀪𑀼𑀜𑁆𑀚𑁂,_  
_“𑀕𑀸𑀭𑀬𑁆𑀳𑀦𑁆”𑀢𑀺 𑀧𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1687\. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸 𑀲𑀦𑁆𑀢𑀸,_  
_𑀬𑀁 𑀧𑀭𑁂𑀲𑀁 𑀫𑀫𑀸𑀬𑀺𑀢𑀁;_  
_𑀲𑀧𑁆𑀧𑀺 𑀢𑁂𑀮𑀁 𑀫𑀥𑀼𑀁 𑀨𑀸𑀡𑀺𑀢𑀁,_  
_𑀫𑀘𑁆𑀙𑀫𑀁𑀲𑀁 𑀅𑀣𑁄 𑀔𑀻𑀭𑀁;_  
_𑀤𑀥𑀺𑀁 𑀲𑀬𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀬𑁆𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻,_  
_𑀕𑀸𑀭𑀬𑁆𑀳𑀧𑀢𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑁂𑁇_  


1688\. _“𑀤𑀼𑀓𑁆𑀓𑀝𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀅𑀧𑀭𑀤𑁆𑀥𑀁 𑀯𑀺𑀭𑀤𑁆𑀥𑀜𑁆𑀘,_  
_𑀔𑀮𑀺𑀢𑀁 𑀬𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇_  


1689\. _𑀬𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁄 𑀓𑀭𑁂 𑀧𑀸𑀧𑀁,_  
_𑀆𑀯𑀺 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀳𑁄;_  
_“𑀤𑀼𑀓𑁆𑀓𑀝𑀦𑁆”𑀢𑀺 𑀧𑀯𑁂𑀤𑁂𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1690\. _“𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀤𑀼𑀭𑀸𑀪𑀝𑁆𑀞𑀁,_  
_𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀜𑁆𑀘 𑀬𑀁 𑀧𑀤𑀁;_  
_𑀬𑀜𑁆𑀘 𑀯𑀺𑀜𑁆𑀜𑀽 𑀕𑀭𑀳𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1691\. _“𑀲𑁂𑀔𑀺𑀬𑀦𑁆”𑀢𑀺 𑀬𑀁 𑀯𑀼𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀳𑀺 𑀬𑀣𑀸𑀢𑀣𑀁;_  
_𑀲𑁂𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀉𑀚𑀼𑀫𑀕𑁆𑀕𑀸𑀦𑀼𑀲𑀸𑀭𑀺𑀦𑁄𑁇_  


1692\. _𑀆𑀤𑀺 𑀘𑁂𑀢𑀁 𑀘𑀭𑀡𑀜𑁆𑀘,_  
_𑀫𑀼𑀔𑀁 𑀲𑀜𑁆𑀜𑀫𑀲𑀁𑀯𑀭𑁄;_  
_𑀲𑀺𑀓𑁆𑀔𑀸 𑀏𑀢𑀸𑀤𑀺𑀲𑀻 𑀦𑀢𑁆𑀣𑀺,_  
_𑀢𑁂𑀦𑁂𑀢𑀁 𑀇𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


1693\. _𑀙𑀦𑁆𑀦𑀫𑀢𑀺𑀯𑀲𑁆𑀲𑀢𑀺,_  
_𑀯𑀺𑀯𑀝𑀁 𑀦𑀸𑀢𑀺𑀯𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀙𑀦𑁆𑀦𑀁 𑀯𑀺𑀯𑀭𑁂𑀣,_  
_𑀏𑀯𑀁 𑀢𑀁 𑀦𑀸𑀢𑀺𑀯𑀲𑁆𑀲𑀢𑀺𑁇_  


1694\. _𑀕𑀢𑀺 𑀫𑀺𑀕𑀸𑀦𑀁 𑀧𑀯𑀦𑀁,_  
_𑀆𑀓𑀸𑀲𑁄 𑀧𑀓𑁆𑀔𑀺𑀦𑀁 𑀕𑀢𑀺;_  
_𑀯𑀺𑀪𑀯𑁄 𑀕𑀢𑀺 𑀥𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀕𑀢𑀻𑀢𑀺𑁇_  


---

1695\. 𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

1696\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1697\. _𑀲𑀢𑁆𑀢𑀦𑀕𑀭𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸,_  
_𑀯𑀺𑀧𑀢𑁆𑀢𑀺 𑀘𑀢𑀼𑀭𑁄𑀧𑀺 𑀘;_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘,_  
_𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀸;_  
_𑀲𑀸𑀲𑀦𑀁 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬,_  
_𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀁 𑀇𑀤𑀦𑁆𑀢𑀺𑁇_  


---

1698\. 𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
