# 𑁧𑁇𑁨𑁇𑁭 𑀲𑀫𑀣𑀯𑀸𑀭

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀕

* 𑀲𑀫𑀣𑀯𑀸𑀭

(199.)

655\. 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘; 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘…𑀧𑁂… 𑁇 (𑁧)

656\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺? 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇 (𑁨)
