# 𑁧𑁇𑁨𑁇𑁧𑁇𑁧 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀕

* 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

(188.)

595\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁…𑀧𑁂…  𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺?

596\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀁 𑀓𑀢𑁆𑀣𑁄𑀕𑀥𑀁 𑀓𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺? 𑀦𑀺𑀤𑀸𑀦𑁄𑀕𑀥𑀁 𑀦𑀺𑀤𑀸𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀫𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀫𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺? 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂…  𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺? 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀪𑀢𑀁—

597\. _𑀉𑀧𑀸𑀮𑀺 𑀤𑀸𑀲𑀓𑁄 𑀘𑁂𑀯,_  
_𑀲𑁄𑀡𑀓𑁄 𑀲𑀺𑀕𑁆𑀕𑀯𑁄 𑀢𑀣𑀸;_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀸,_  
_𑀏𑀢𑁂 𑀚𑀫𑁆𑀩𑀼𑀲𑀺𑀭𑀺𑀯𑁆𑀳𑀬𑁂𑁇_  


---

598\. …𑀧𑁂…



599\. _𑀏𑀢𑁂 𑀦𑀸𑀕𑀸 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀺𑀦𑀬𑀜𑁆𑀜𑀽 𑀫𑀕𑁆𑀕𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀤𑀻𑀧𑁂 𑀧𑀓𑀸𑀲𑁂𑀲𑀼𑀁,_  
_𑀧𑀺𑀝𑀓𑀁 𑀢𑀫𑁆𑀩𑀧𑀡𑁆𑀡𑀺𑀬𑀸𑀢𑀺𑁇_  


(189.)

600\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀥𑀦𑀺𑀬𑀁 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀥𑀦𑀺𑀬𑁄 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀤𑀸𑀭𑀽𑀦𑀺 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(190.)

601\. 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(191.)

602\. 𑀅𑀲𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇
