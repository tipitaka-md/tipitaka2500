# 𑁧𑁇𑁨𑁇𑁪 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀯𑀸𑀭

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀕

* 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀯𑀸𑀭

(196.)

646\. 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦…𑀧𑁂… 𑁇 (𑁧)

647\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑀡𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 (𑁨)

---
