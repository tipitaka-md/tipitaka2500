# 𑁧𑁇𑁧𑁇𑁨𑁇𑁪𑁇𑁧 𑀓𑀣𑀺𑀦𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀓𑀣𑀺𑀦𑀯𑀕𑁆𑀕

(162.)

352\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀤𑀲𑀸𑀳𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁧)

353\. 𑀏𑀓𑀭𑀢𑁆𑀢𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁨)

354\. 𑀅𑀓𑀸𑀮𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀲𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁩)

355\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀧𑀼𑀭𑀸𑀡𑀘𑀻𑀯𑀭𑀁 𑀥𑁄𑀯𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀥𑁄𑀯𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀥𑁄𑀯𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁪)

356\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀳𑀢𑁆𑀣𑀢𑁄 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀕𑀳𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁫)

357\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀁 𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁬)

358\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀯𑀸 𑀕𑀳𑀧𑀢𑀸𑀦𑀺𑀁 𑀯𑀸 𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺 𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁭)

359\. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀓𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀦𑁆𑀦𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁮)

360\. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑁂 𑀕𑀳𑀧𑀢𑀺𑀓𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀆𑀧𑀦𑁆𑀦𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁯)

361\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀘𑁄𑀤𑀦𑀸𑀬 𑀅𑀢𑀺𑀭𑁂𑀓𑀙𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀞𑀸𑀦𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀪𑀺𑀦𑀺𑀧𑁆𑀨𑀸𑀤𑀺𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 (𑁧𑁦)

---

362\. 𑀓𑀣𑀺𑀦𑀯𑀕𑁆𑀕𑁄 𑀧𑀞𑀫𑁄𑁇
