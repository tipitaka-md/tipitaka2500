# 𑁧𑁇𑁧𑁇𑁨𑁇𑁭𑁇𑁬 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕

(180.)

546\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀓𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧)

547\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀢𑁆𑀣𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁨)

548\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀢𑁆𑀢𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁩)

549\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀑𑀝𑁆𑀞𑀦𑀺𑀮𑁆𑀮𑁂𑀳𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁪)

550\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀸𑀫𑀺𑀲𑁂𑀦 𑀳𑀢𑁆𑀣𑁂𑀦 𑀧𑀸𑀦𑀻𑀬𑀣𑀸𑀮𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁫)

551\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀲𑀺𑀢𑁆𑀣𑀓𑀁 𑀧𑀢𑁆𑀢𑀥𑁄𑀯𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀙𑀟𑁆𑀟𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁬)

552\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀙𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁭)

553\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁮)

554\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀢𑁆𑀣𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁯)

555\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧𑁦)

---

556\. 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼𑀯𑀕𑁆𑀕𑁄 𑀙𑀝𑁆𑀞𑁄𑁇
