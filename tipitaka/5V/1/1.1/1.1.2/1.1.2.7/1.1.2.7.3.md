# 𑁧𑁇𑁧𑁇𑁨𑁇𑁭𑁇𑁩 𑀔𑀫𑁆𑀪𑀓𑀢𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀔𑀫𑁆𑀪𑀓𑀢𑀯𑀕𑁆𑀕

(177.)

513\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀔𑀫𑁆𑀪𑀓𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧)

514\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀔𑀫𑁆𑀪𑀓𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁨)

515\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁩)

516\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁪)

517\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁫)

518\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁬)

519\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁭)

520\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁮)

521\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀽𑀧𑀜𑁆𑀜𑁂𑀯 𑀩𑀳𑀼𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁯)

522\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀽𑀧𑀻𑀓𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧𑁦)

---
