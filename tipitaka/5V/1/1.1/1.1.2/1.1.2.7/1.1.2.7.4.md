# 𑁧𑁇𑁧𑁇𑁨𑁇𑁭𑁇𑁪 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀯𑀕𑁆𑀕

(178.)

524\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧)

525\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁨)

526\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀑𑀫𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁩)

527\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀽𑀧𑀜𑁆𑀜𑁂𑀯 𑀩𑀳𑀼𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁪)

528\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀣𑀽𑀧𑀓𑀢𑁄 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁫)

529\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀽𑀧𑀁 𑀯𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀯𑀸 𑀑𑀤𑀦𑁂𑀦 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁬)

530\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀽𑀧𑀁 𑀯𑀸 𑀑𑀤𑀦𑀁 𑀯𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁭)

531\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀻 𑀧𑀭𑁂𑀲𑀁 𑀧𑀢𑁆𑀢𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁮)

532\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀫𑀳𑀦𑁆𑀢𑀁 𑀓𑀩𑀍𑀅𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁯)

533\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀻𑀖𑀁 𑀆𑀮𑁄𑀧𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧𑁦)

---
