# 𑁧𑁇𑁧𑁇𑁨𑁇𑁭𑁇𑁫 𑀓𑀩𑀍𑀅𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀓𑀩𑀍𑀅𑀯𑀕𑁆𑀕

(179.)

535\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀦𑀸𑀳𑀝𑁂 𑀓𑀩𑀍𑀏 𑀫𑀼𑀔𑀤𑁆𑀯𑀸𑀭𑀁 𑀯𑀺𑀯𑀭𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧)

536\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑁄 𑀲𑀩𑁆𑀩𑀁 𑀳𑀢𑁆𑀣𑀁 𑀫𑀼𑀔𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁨)

537\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀓𑀩𑀍𑀏𑀦 𑀫𑀼𑀔𑁂𑀦 𑀩𑁆𑀬𑀸𑀳𑀭𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁩)

538\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀼𑀓𑁆𑀔𑁂𑀧𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁪)

539\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀩𑀍𑀆𑀯𑀘𑁆𑀙𑁂𑀤𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁫)

540\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀯𑀕𑀡𑁆𑀟𑀓𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁬)

541\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀢𑁆𑀣𑀦𑀺𑀤𑁆𑀥𑀼𑀦𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁭)

542\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀺𑀢𑁆𑀣𑀸𑀯𑀓𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁮)

543\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀚𑀺𑀯𑁆𑀳𑀸𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁯)

544\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀘𑀧𑀼𑀘𑀧𑀼𑀓𑀸𑀭𑀓𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇 (𑁧𑁦)

---
