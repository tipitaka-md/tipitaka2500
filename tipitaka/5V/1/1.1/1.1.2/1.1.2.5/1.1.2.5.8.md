# 𑁧𑁇𑁧𑁇𑁨𑁇𑁫𑁇𑁮 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕

(172.)

462\. 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄—  “𑀦 𑀢𑀸𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀬𑀸𑀯 𑀦 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀩𑁆𑀬𑀢𑁆𑀢𑀁 𑀯𑀺𑀦𑀬𑀥𑀭𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀪𑀡𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀡𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀪𑀡𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

463\. 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀯𑀡𑁆𑀡𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀺𑀯𑀡𑁆𑀡𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

464\. 𑀫𑁄𑀳𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀦𑀸𑀭𑁄𑀧𑀺𑀢𑁂 𑀫𑁄𑀳𑁂 𑀫𑁄𑀳𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀆𑀭𑁄𑀧𑀺𑀢𑁂 𑀫𑁄𑀳𑁂 𑀫𑁄𑀳𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

465\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀧𑀳𑀸𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀳𑀭𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀳𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

466\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀢𑀮𑀲𑀢𑁆𑀢𑀺𑀓𑀁 𑀉𑀕𑁆𑀕𑀺𑀭𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀕𑁆𑀕𑀺𑀭𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀕𑁆𑀕𑀺𑀭𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

467\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

468\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀉𑀧𑀤𑀳𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀧𑀤𑀳𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀧𑀤𑀳𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

469\. 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀡𑁆𑀟𑀦𑀚𑀸𑀢𑀸𑀦𑀁 𑀓𑀮𑀳𑀚𑀸𑀢𑀸𑀦𑀁 𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺𑀁 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀲𑁄𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

470\. 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀔𑀺𑀬𑁆𑀬𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

471\. 𑀲𑀁𑀖𑁂 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀣𑀸𑀬 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑀸𑀬 𑀙𑀦𑁆𑀤𑀁 𑀅𑀤𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀭𑀺𑀲𑀸𑀬 𑀳𑀢𑁆𑀣𑀧𑀸𑀲𑀁 𑀯𑀺𑀚𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀯𑀺𑀚𑀳𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)

472\. 𑀲𑀫𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀔𑀻𑀬𑀦𑀥𑀫𑁆𑀫𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀔𑀺𑀬𑁆𑀬𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁧)

473\. 𑀚𑀸𑀦𑀁 𑀲𑀁𑀖𑀺𑀓𑀁 𑀮𑀸𑀪𑀁 𑀧𑀭𑀺𑀡𑀢𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀭𑀺𑀡𑀸𑀫𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁨)

---

474\. 𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀕𑁆𑀕𑁄 𑀅𑀝𑁆𑀞𑀫𑁄𑁇
