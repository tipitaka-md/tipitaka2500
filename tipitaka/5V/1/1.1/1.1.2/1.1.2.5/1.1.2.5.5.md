# 𑁧𑁇𑁧𑁇𑁨𑁇𑁫𑁇𑁫 𑀅𑀘𑁂𑀮𑀓𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀅𑀘𑁂𑀮𑀓𑀯𑀕𑁆𑀕

(169.)

429\. 𑀅𑀘𑁂𑀮𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀬 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀤𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀤𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

430\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁—  “𑀏𑀳𑀸𑀯𑀼𑀲𑁄, 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀬𑁆𑀬𑁄𑀚𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

431\. 𑀲𑀪𑁄𑀚𑀦𑁂 𑀓𑀼𑀮𑁂 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

432\. 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

433\. 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

434\. 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀲𑀪𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀉𑀫𑁆𑀫𑀸𑀭𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

435\. 𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

436\. 𑀉𑀬𑁆𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

437\. 𑀅𑀢𑀺𑀭𑁂𑀓𑀢𑀺𑀭𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀬 𑀯𑀲𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀯𑀲𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀯𑀲𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

438\. 𑀉𑀬𑁆𑀬𑁄𑀥𑀺𑀓𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀬𑀢𑁆𑀣 𑀞𑀺𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)

---
