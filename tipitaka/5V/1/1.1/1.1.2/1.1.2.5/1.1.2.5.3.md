# 𑁧𑁇𑁧𑁇𑁨𑁇𑁫𑁇𑁩 𑀑𑀯𑀸𑀤𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀑𑀯𑀸𑀤𑀯𑀕𑁆𑀕

(167.)

407\. 𑀅𑀲𑀫𑁆𑀫𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀑𑀯𑀤𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀑𑀯𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀑𑀯𑀤𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

408\. 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑁂 𑀲𑀽𑀭𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀑𑀯𑀤𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀑𑀯𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀑𑀯𑀤𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

409\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀼𑀧𑀲𑁆𑀲𑀬𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀑𑀯𑀤𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀑𑀯𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀑𑀯𑀤𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

410\. “𑀆𑀫𑀺𑀲𑀳𑁂𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀑𑀯𑀤𑀦𑁆𑀢𑀻”𑀢𑀺 𑀪𑀡𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀡𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀪𑀡𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

411\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀤𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

412\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀘𑀻𑀯𑀭𑀁 𑀲𑀺𑀩𑁆𑀩𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀺𑀩𑁆𑀩𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀆𑀭𑀸𑀧𑀣𑁂 𑀆𑀭𑀸𑀧𑀣𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

413\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

414\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀁𑀯𑀺𑀥𑀸𑀬 𑀏𑀓𑀁 𑀦𑀸𑀯𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀪𑀺𑀭𑀼𑀳𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀪𑀺𑀭𑀼𑀍𑀳𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

415\. 𑀚𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀧𑀭𑀺𑀧𑀸𑀘𑀺𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

416\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)

---

417\. 𑀑𑀯𑀸𑀤𑀯𑀕𑁆𑀕𑁄 𑀢𑀢𑀺𑀬𑁄𑁇
