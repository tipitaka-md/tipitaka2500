# 𑁧𑁇𑁧𑁇𑁨𑁇𑁫𑁇𑁪 𑀪𑁄𑀚𑀦𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀪𑁄𑀚𑀦𑀯𑀕𑁆𑀕

(168.)

418\. 𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺 𑀆𑀯𑀲𑀣𑀧𑀺𑀡𑁆𑀟𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

419\. 𑀕𑀡𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

420\. 𑀧𑀭𑀫𑁆𑀧𑀭𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

421\. 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀧𑀢𑁆𑀢𑀧𑀽𑀭𑁂 𑀧𑀽𑀯𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀕𑀳𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

422\. 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

423\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀧𑀯𑀸𑀭𑀺𑀢𑀁 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀯𑀸 𑀅𑀪𑀺𑀳𑀝𑁆𑀞𑀼𑀁 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀯𑀘𑀦𑁂𑀦 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀪𑁄𑀚𑀦𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

424\. 𑀯𑀺𑀓𑀸𑀮𑁂 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

425\. 𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀓𑀸𑀭𑀓𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

426\. 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

427\. 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀼𑀔𑀤𑁆𑀯𑀸𑀭𑀁 𑀆𑀳𑀸𑀭𑀁 𑀆𑀳𑀭𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 “𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀅𑀚𑁆𑀛𑁄𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)

---
