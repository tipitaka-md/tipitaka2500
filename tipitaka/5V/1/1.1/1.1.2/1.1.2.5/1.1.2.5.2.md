# 𑁧𑁇𑁧𑁇𑁨𑁇𑁫𑁇𑁨 𑀪𑀽𑀢𑀕𑀸𑀫𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀪𑀽𑀢𑀕𑀸𑀫𑀯𑀕𑁆𑀕

(166.)

396\. 𑀪𑀽𑀢𑀕𑀸𑀫𑀁 𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀧𑀳𑀸𑀭𑁂 𑀧𑀳𑀸𑀭𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧)

397\. 𑀅𑀜𑁆𑀜𑁂𑀦𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀦𑀸𑀭𑁄𑀧𑀺𑀢𑁂 𑀅𑀜𑁆𑀜𑀯𑀸𑀤𑀓𑁂 𑀅𑀜𑁆𑀜𑁂𑀦𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀆𑀭𑁄𑀧𑀺𑀢𑁂 𑀅𑀜𑁆𑀜𑀯𑀸𑀤𑀓𑁂 𑀅𑀜𑁆𑀜𑁂𑀦𑀜𑁆𑀜𑀁 𑀧𑀝𑀺𑀘𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁨)

398\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀉𑀚𑁆𑀛𑀸𑀧𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁩)

399\. 𑀲𑀁𑀖𑀺𑀓𑀁 𑀫𑀜𑁆𑀘𑀁 𑀯𑀸 𑀧𑀻𑀞𑀁 𑀯𑀸 𑀪𑀺𑀲𑀺𑀁 𑀯𑀸 𑀓𑁄𑀘𑁆𑀙𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀮𑁂𑀟𑁆𑀟𑀼𑀧𑀸𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁪)

400\. 𑀲𑀁𑀖𑀺𑀓𑁂 𑀯𑀺𑀳𑀸𑀭𑁂 𑀲𑁂𑀬𑁆𑀬𑀁 𑀲𑀦𑁆𑀣𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀤𑁆𑀥𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀞𑀫𑀁 𑀧𑀸𑀤𑀁 𑀧𑀭𑀺𑀓𑁆𑀔𑁂𑀧𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲; 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀤𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁫)

401\. 𑀲𑀁𑀖𑀺𑀓𑁂 𑀯𑀺𑀳𑀸𑀭𑁂 𑀚𑀸𑀦𑀁 𑀧𑀼𑀩𑁆𑀩𑀼𑀧𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀧𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁬)

402\. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀲𑀁𑀖𑀺𑀓𑀸 𑀯𑀺𑀳𑀸𑀭𑀸 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁭)

403\. 𑀲𑀁𑀖𑀺𑀓𑁂 𑀯𑀺𑀳𑀸𑀭𑁂 𑀉𑀧𑀭𑀺𑀯𑁂𑀳𑀸𑀲𑀓𑀼𑀝𑀺𑀬𑀸 𑀆𑀳𑀘𑁆𑀘𑀧𑀸𑀤𑀓𑀁 𑀫𑀜𑁆𑀘𑀁 𑀯𑀸 𑀧𑀻𑀞𑀁 𑀯𑀸 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀪𑀺𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀪𑀺𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁮)

404\. 𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀧𑀭𑀺𑀬𑀸𑀬𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀅𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁯)

405\. 𑀚𑀸𑀦𑀁 𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑀁 𑀉𑀤𑀓𑀁 𑀢𑀺𑀡𑀁 𑀯𑀸 𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀯𑀸 𑀲𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑀺𑀜𑁆𑀘𑀢𑀺, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 (𑁧𑁦)

---

406\. 𑀪𑀽𑀢𑀕𑀸𑀫𑀯𑀕𑁆𑀕𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇
