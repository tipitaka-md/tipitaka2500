# 𑁧𑁇𑁧𑁇𑁬 𑀅𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀅𑀥𑀺𑀓𑀭𑀡𑀯𑀸𑀭

(185.)

582\. 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁…𑀧𑁂… 𑁇 (𑁧)

583\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 (𑁨)

---
