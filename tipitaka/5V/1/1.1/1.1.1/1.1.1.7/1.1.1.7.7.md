# 𑁧𑁇𑁧𑁇𑁧𑁇𑁭𑁇𑁭 𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕

(156.)

305\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀸𑀤𑀼𑀓𑀸𑀭𑀼𑀍𑀳𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄…𑀧𑁂… 𑁇 (𑁧)

306\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑀸𑀳𑀦𑀸𑀭𑀼𑀍𑀳𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄…𑀧𑁂… 𑁇 (𑁨)

307\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀬𑀸𑀦𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄…𑀧𑁂… 𑁇 (𑁩)

308\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑀬𑀦𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄…𑀧𑁂… 𑁇 (𑁪)

309\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄…𑀧𑁂… 𑁇 (𑁫)

310\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀯𑁂𑀞𑀺𑀢𑀲𑀻𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄…𑀧𑁂… 𑁇 (𑁬)

311\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑀻𑀲𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄…𑀧𑁂… 𑁇 (𑁭)

312\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀙𑀫𑀸𑀬𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁮)

313\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑀻𑀘𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀘𑁆𑀘𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁯)

314\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀞𑀺𑀢𑁂𑀦 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁦)

315\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀘𑁆𑀙𑀢𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑀦 𑀧𑀼𑀭𑀢𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁧)

316\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀧𑁆𑀧𑀣𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑀦 𑀧𑀣𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑁧𑁨)

317\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀞𑀺𑀢𑁂𑀦 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁧𑁩)

318\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀭𑀺𑀢𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁧𑁪)

319\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀯𑀸 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀯𑀸 𑀔𑁂𑀍𑀅𑀁 𑀯𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑀓𑁂 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀔𑁂𑀍𑀅𑀫𑁆𑀧𑀺 𑀅𑀓𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁧𑁫)

---

320\. 𑀧𑀸𑀤𑀼𑀓𑀯𑀕𑁆𑀕𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇

  
𑀧𑀜𑁆𑀘𑀲𑀢𑁆𑀢𑀢𑀺 𑀲𑁂𑀔𑀺𑀬𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇



---

321\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



322\. _𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑁄𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀓𑁆𑀔𑀼;_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑀚𑁆𑀚𑀕𑁆𑀖𑀺𑀓𑀸 𑀲𑀤𑁆𑀤𑁄,_  
_𑀢𑀬𑁄 𑀘𑁂𑀯 𑀧𑀘𑀸𑀮𑀦𑀸𑁇_  


323\. _𑀔𑀫𑁆𑀪𑀁 𑀑𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑁄 𑀘𑁂𑀯𑀼𑀓𑁆-_  
_𑀓𑀼𑀝𑀺𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸𑀬 𑀘;_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀧𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀘,_  
_𑀲𑀫𑀲𑀽𑀧𑀁 𑀲𑀫𑀢𑀺𑀢𑁆𑀢𑀺𑀓𑀁𑁇_  


324\. _𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀧𑀢𑁆𑀢𑀲𑀜𑁆𑀜𑀻 𑀘,_  
_𑀲𑀧𑀤𑀸𑀦𑀁 𑀲𑀫𑀲𑀽𑀧𑀓𑀁;_  
_𑀣𑀽𑀧𑀓𑀢𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀼𑀚𑁆𑀛𑀸𑀦𑀲𑀜𑁆𑀜𑀺𑀦𑀸𑁇_  


325\. _𑀦 𑀫𑀳𑀦𑁆𑀢𑀁 𑀫𑀡𑁆𑀟𑀮𑀁 𑀤𑁆𑀯𑀸𑀭𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀳𑀢𑁆𑀣𑀁 𑀦 𑀩𑁆𑀬𑀸𑀳𑀭𑁂;_  
_𑀉𑀓𑁆𑀔𑁂𑀧𑁄 𑀙𑁂𑀤𑀦𑀸 𑀕𑀡𑁆𑀟𑁄,_  
_𑀥𑀼𑀦𑀁 𑀲𑀺𑀢𑁆𑀣𑀸𑀯𑀓𑀸𑀭𑀓𑀁𑁇_  


326\. _𑀚𑀺𑀯𑁆𑀳𑀸𑀦𑀺𑀘𑁆𑀙𑀸𑀭𑀓𑀜𑁆𑀘𑁂𑀯,_  
_𑀘𑀧𑀼𑀘𑀧𑀼 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼;_  
_𑀳𑀢𑁆𑀣𑁄 𑀧𑀢𑁆𑀢𑁄 𑀘 𑀑𑀝𑁆𑀞𑁄 𑀘,_  
_𑀲𑀸𑀫𑀺𑀲𑀁 𑀲𑀺𑀢𑁆𑀣𑀓𑁂𑀦 𑀘𑁇_  


327\. _𑀙𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀲𑁆𑀲 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀦 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑀸;_  
_𑀏𑀯𑀫𑁂𑀯 𑀤𑀡𑁆𑀟𑀧𑀸𑀡𑀺𑀲𑁆𑀲,_  
_𑀲𑀢𑁆𑀣𑀆𑀯𑀼𑀥𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


328\. _𑀧𑀸𑀤𑀼𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀘𑁂𑀯,_  
_𑀬𑀸𑀦𑀲𑁂𑀬𑁆𑀬𑀸𑀕𑀢𑀲𑁆𑀲 𑀘;_  
_𑀧𑀮𑁆𑀮𑀢𑁆𑀣𑀺𑀓𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀯𑁂𑀞𑀺𑀢𑁄𑀕𑀼𑀡𑁆𑀞𑀺𑀢𑀲𑁆𑀲 𑀘𑁇_  


329\. _𑀙𑀫𑀸 𑀦𑀻𑀘𑀸𑀲𑀦𑁂 𑀞𑀸𑀦𑁂,_  
_𑀧𑀘𑁆𑀙𑀢𑁄 𑀉𑀧𑁆𑀧𑀣𑁂𑀦 𑀘;_  
_𑀞𑀺𑀢𑀓𑁂𑀦 𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀳𑀭𑀺𑀢𑁂 𑀉𑀤𑀓𑀫𑁆𑀳𑀺 𑀘𑀸𑀢𑀺𑁇_  


---

330\. 𑀢𑁂𑀲𑀁 𑀯𑀕𑁆𑀕𑀸𑀦𑀫𑀼𑀤𑁆𑀤𑀸𑀦𑀁



331\. _𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑁅𑀚𑁆𑀚𑀕𑁆𑀖𑀺,_  
_𑀔𑀫𑁆𑀪𑀁 𑀧𑀺𑀡𑁆𑀟𑀁 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀓𑀩𑀍𑀆 𑀲𑀼𑀭𑀼𑀲𑀼𑀭𑀼 𑀘,_  
_𑀧𑀸𑀤𑀼𑀓𑁂𑀦 𑀘 𑀲𑀢𑁆𑀢𑀫𑀸𑀢𑀺𑁇_  


---

332\. 𑀫𑀳𑀸𑀯𑀺𑀪𑀗𑁆𑀕𑁂 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
