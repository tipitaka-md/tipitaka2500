# 𑁧𑁇𑁧𑁇𑁧𑁇𑁭𑁇𑁧 𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

* 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

* 𑀧𑀞𑀫𑀪𑀸𑀕

* 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀲𑁂𑀔𑀺𑀬𑀓𑀡𑁆𑀟

* 𑀧𑀭𑀺𑀫𑀡𑁆𑀟𑀮𑀯𑀕𑁆𑀕

(150.)

239\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀼𑀭𑀢𑁄 𑀯𑀸 𑀧𑀘𑁆𑀙𑀢𑁄 𑀯𑀸 𑀑𑀮𑀫𑁆𑀩𑁂𑀦𑁆𑀢𑁂𑀦 𑀦𑀺𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀢𑁄𑀧𑀺 𑀧𑀘𑁆𑀙𑀢𑁄𑀧𑀺 𑀑𑀮𑀫𑁆𑀩𑁂𑀦𑁆𑀢𑀸 𑀦𑀺𑀯𑀸𑀲𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁧)

240\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀧𑀼𑀭𑀢𑁄 𑀯𑀸 𑀧𑀘𑁆𑀙𑀢𑁄 𑀯𑀸 𑀑𑀮𑀫𑁆𑀩𑁂𑀦𑁆𑀢𑁂𑀦 𑀧𑀸𑀭𑀼𑀧𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀭𑀢𑁄𑀧𑀺 𑀧𑀘𑁆𑀙𑀢𑁄𑀧𑀺 𑀑𑀮𑀫𑁆𑀩𑁂𑀦𑁆𑀢𑀸 𑀧𑀸𑀭𑀼𑀧𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁨)

241\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀸𑀬𑀁 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁩)

242\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀸𑀬𑀁 𑀯𑀺𑀯𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁪)

243\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀢𑁆𑀣𑀁 𑀯𑀸 𑀧𑀸𑀤𑀁 𑀯𑀸 𑀓𑀻𑀍𑀆𑀧𑁂𑀦𑁆𑀢𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁫)

244\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀳𑀢𑁆𑀣𑀁 𑀯𑀸 𑀧𑀸𑀤𑀁 𑀯𑀸 𑀓𑀻𑀍𑀆𑀧𑁂𑀦𑁆𑀢𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁬)

245\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁭)

246\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀳𑀁 𑀢𑀳𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁮)

247\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁯)

248\. 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀓𑀝𑀁…𑀧𑁂…  𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇 (𑁧𑁦)
