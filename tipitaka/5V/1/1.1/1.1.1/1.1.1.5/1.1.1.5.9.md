# 𑁧𑁇𑁧𑁇𑁧𑁇𑁫𑁇𑁯 𑀭𑀸𑀚𑀯𑀕𑁆𑀕

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀭𑀸𑀚𑀯𑀕𑁆𑀕

(136.)

209. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀓𑀣𑀺𑀦𑀓𑁂…𑀧𑁂… 𑁇

(137.)

210. 𑀭𑀢𑀦𑀁 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀢𑀦𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(138.)

211. 𑀲𑀦𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀯𑀺𑀓𑀸𑀮𑁂 𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀓𑀸𑀮𑁂 𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀓𑀣𑀺𑀦𑀓𑁂…𑀧𑁂… 𑁇

(139.)

212. 𑀅𑀝𑁆𑀞𑀺𑀫𑀬𑀁 𑀯𑀸 𑀤𑀦𑁆𑀢𑀫𑀬𑀁 𑀯𑀸 𑀯𑀺𑀲𑀸𑀡𑀫𑀬𑀁 𑀯𑀸 𑀲𑀽𑀘𑀺𑀖𑀭𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀫𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀳𑀽 𑀲𑀽𑀘𑀺𑀖𑀭𑁂 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(140.)

213. 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀫𑀜𑁆𑀘𑀁 𑀯𑀸 𑀧𑀻𑀞𑀁 𑀯𑀸 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀉𑀘𑁆𑀘𑁂 𑀫𑀜𑁆𑀘𑁂 𑀲𑀬𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(141.)

214. 𑀫𑀜𑁆𑀘𑀁 𑀯𑀸 𑀧𑀻𑀞𑀁 𑀯𑀸 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀜𑁆𑀘𑀁 𑀯𑀸 𑀧𑀻𑀞𑀁 𑀯𑀸 𑀢𑀽𑀮𑁄𑀦𑀤𑁆𑀥𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(142.)

215. 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀦𑀺 𑀦𑀺𑀲𑀻𑀤𑀦𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(143.)

216. 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀓𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀓𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(144.)

217. 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀫𑀸𑀡𑀺𑀓𑀸𑀬𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(145.)

218. 𑀲𑀼𑀕𑀢𑀘𑀻𑀯𑀭𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀦𑀦𑁆𑀤𑁄 𑀲𑀼𑀕𑀢𑀘𑀻𑀯𑀭𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀘𑀻𑀯𑀭𑀁 𑀥𑀸𑀭𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

---

219. 𑀭𑀸𑀚𑀯𑀕𑁆𑀕𑁄 𑀦𑀯𑀫𑁄𑁇

  
𑀤𑁆𑀯𑁂𑀦𑀯𑀼𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇

  
𑀔𑀼𑀤𑁆𑀤𑀓𑀁 𑀲𑀫𑀢𑁆𑀢𑀁𑁇



---

220. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



221. _𑀫𑀼𑀲𑀸 𑀑𑀫𑀲𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑀁,_  
_𑀧𑀤𑀲𑁂𑀬𑁆𑀬𑀸 𑀘 𑀇𑀢𑁆𑀣𑀺𑀬𑀸;_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑀺𑀜𑁆𑀜𑀼𑀦𑀸 𑀪𑀽𑀢𑀸,_  
_𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀧𑀢𑁆𑀢𑀺 𑀔𑀡𑀦𑀸𑁇_  


222. _𑀪𑀽𑀢𑀁 𑀅𑀜𑁆𑀜𑀸𑀬 𑀉𑀚𑁆𑀛𑀸𑀬𑀺,_  
_𑀫𑀜𑁆𑀘𑁄 𑀲𑁂𑀬𑁆𑀬𑁄 𑀘 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀦𑀸𑀳𑀘𑁆𑀘,_  
_𑀤𑁆𑀯𑀸𑀭𑀁 𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑁂𑀦 𑀘𑁇_  


223. _𑀅𑀲𑀫𑁆𑀫𑀢𑀸 𑀅𑀢𑁆𑀣𑀗𑁆𑀕𑀢𑁂,_  
_𑀉𑀧𑀲𑁆𑀲𑀬𑀸𑀫𑀺𑀲𑁂𑀦 𑀘;_  
_𑀤𑀤𑁂 𑀲𑀺𑀩𑁆𑀩𑁂 𑀯𑀺𑀥𑀸𑀦𑁂𑀦,_  
_𑀦𑀸𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬 𑀏𑀓𑀢𑁄𑁇_  


224. _𑀧𑀺𑀡𑁆𑀟𑀁 𑀕𑀡𑀁 𑀧𑀭𑀁 𑀧𑀽𑀯𑀁,_  
_𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀧𑀯𑀸𑀭𑀺𑀢𑀁;_  
_𑀯𑀺𑀓𑀸𑀮𑀁 𑀲𑀦𑁆𑀦𑀺𑀥𑀺 𑀔𑀻𑀭𑀁,_  
_𑀤𑀦𑁆𑀢𑀧𑁄𑀦𑁂𑀦 𑀢𑁂 𑀤𑀲𑁇_  


225. _𑀅𑀘𑁂𑀮𑀓𑀁 𑀉𑀬𑁆𑀬𑁄𑀔𑀚𑁆𑀚,_  
_𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀭𑀳𑁂𑀦 𑀘;_  
_𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀧𑀘𑁆𑀘𑀬𑁂𑀳𑀺,_  
_𑀲𑁂𑀦𑀸𑀯𑀲𑀦𑀼𑀬𑁆𑀬𑁄𑀥𑀺𑀓𑀁𑁇_  


226. _𑀲𑀼𑀭𑀸 𑀅𑀗𑁆𑀕𑀼𑀮𑀺 𑀳𑀸𑀲𑁄 𑀘,_  
_𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀜𑁆𑀘 𑀪𑀺𑀁𑀲𑀦𑀁;_  
_𑀚𑁄𑀢𑀺 𑀦𑀳𑀸𑀦 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀁,_  
_𑀲𑀸𑀫𑀁 𑀅𑀧𑀦𑀺𑀥𑁂𑀦 𑀘𑁇_  


227. _𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘𑀼𑀤𑀓𑀓𑀫𑁆𑀫𑀸 𑀘,_  
_𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀊𑀦𑀯𑀻𑀲𑀢𑀺;_  
_𑀣𑁂𑀬𑁆𑀬𑁃𑀢𑁆𑀣𑀺𑀅𑀯𑀤𑁂𑀲𑀁,_  
_𑀲𑀁𑀯𑀸𑀲𑁂 𑀦𑀸𑀲𑀺𑀢𑁂𑀦 𑀘𑁇_  


228. _𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀺𑀮𑁂𑀔𑀸,_  
_𑀫𑁄𑀳𑁄 𑀧𑀳𑀸𑀭𑁂𑀦𑀼𑀕𑁆𑀕𑀺𑀭𑁂;_  
_𑀅𑀫𑀽𑀮𑀓𑀜𑁆𑀘 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘,_  
_𑀲𑁄𑀲𑁆𑀲𑀸𑀫𑀺 𑀔𑀺𑀬𑁆𑀬𑀧𑀓𑁆𑀓𑀫𑁂𑁇_  


229. _𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀬𑁆𑀬 𑀧𑀼𑀕𑁆𑀕𑀮𑁂;_  
_𑀭𑀜𑁆𑀜𑀜𑁆𑀘 𑀭𑀢𑀦𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀲𑀽𑀘𑀺 𑀫𑀜𑁆𑀘𑁄 𑀘 𑀢𑀽𑀮𑀺𑀓𑀸;_  
_𑀦𑀺𑀲𑀻𑀤𑀦𑀁 𑀓𑀡𑁆𑀟𑀼𑀘𑁆𑀙𑀸𑀤𑀺,_  
_𑀯𑀲𑁆𑀲𑀺𑀓𑀸 𑀲𑀼𑀕𑀢𑁂𑀦 𑀘𑀸𑀢𑀺𑁇_  


---

230. 𑀢𑁂𑀲𑀁 𑀯𑀕𑁆𑀕𑀸𑀦𑀁 𑀉𑀤𑁆𑀤𑀸𑀦𑀁



231. _𑀫𑀼𑀲𑀸 𑀪𑀽𑀢𑀸 𑀘 𑀑𑀯𑀸𑀤𑁄,_  
_𑀪𑁄𑀚𑀦𑀸𑀘𑁂𑀮𑀓𑁂𑀦 𑀘;_  
_𑀲𑀼𑀭𑀸 𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑀸 𑀥𑀫𑁆𑀫𑁄,_  
_𑀭𑀸𑀚𑀯𑀕𑁆𑀕𑁂𑀦 𑀢𑁂 𑀦𑀯𑀸𑀢𑀺𑁇_
