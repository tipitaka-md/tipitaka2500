# 𑁧𑁇𑁧𑁇𑁧𑁇𑁫𑁇𑁫 𑀅𑀘𑁂𑀮𑀓𑀯𑀕𑁆𑀕

### 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀅𑀘𑁂𑀮𑀓𑀯𑀕𑁆𑀕

(94.)

163. 𑀅𑀘𑁂𑀮𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀬 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀺𑀲𑁆𑀲𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀺𑀓𑀸𑀬 𑀏𑀓𑀁 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀤𑁆𑀯𑁂 𑀧𑀽𑀯𑁂 𑀅𑀤𑀸𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇

(95.)

164. 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 “𑀏𑀳𑀸𑀯𑀼𑀲𑁄, 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑀸 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 “𑀏𑀳𑀸𑀯𑀼𑀲𑁄, 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀅𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(96.)

165. 𑀲𑀪𑁄𑀚𑀦𑁂 𑀓𑀼𑀮𑁂 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀪𑁄𑀚𑀦𑁂 𑀓𑀼𑀮𑁂 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇

(97.)

166. 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀭𑀳𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇

(98.)

167. 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑁄 𑀏𑀓𑀸𑀬 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀁 𑀓𑀧𑁆𑀧𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇

(99.)

168. 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁂𑀦 𑀲𑀪𑀢𑁆𑀢𑁂𑀦 𑀲𑀦𑁆𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀲𑀪𑀢𑁆𑀢𑁄 𑀲𑀫𑀸𑀦𑁄 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀓𑀣𑀺𑀦𑀓𑁂…𑀧𑁂… 𑁇

(100.)

169. 𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀫𑀳𑀸𑀦𑀸𑀫𑁂𑀦 𑀲𑀓𑁆𑀓𑁂𑀦 “𑀅𑀚𑁆𑀚𑀡𑁆𑀳𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀕𑀫𑁂𑀣𑀸”𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 𑀦𑀸𑀕𑀫𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(101.)

170. 𑀉𑀬𑁆𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀬𑁆𑀬𑀼𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇

(102.)

171. 𑀅𑀢𑀺𑀭𑁂𑀓𑀢𑀺𑀭𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀬 𑀯𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀢𑀺𑀭𑁂𑀓𑀢𑀺𑀭𑀢𑁆𑀢𑀁 𑀲𑁂𑀦𑀸𑀬 𑀯𑀲𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇

(103.)

172. 𑀉𑀬𑁆𑀬𑁄𑀥𑀺𑀓𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀬𑁆𑀬𑁄𑀥𑀺𑀓𑀁 𑀅𑀕𑀫𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀓𑁂…𑀧𑁂… 𑁇
