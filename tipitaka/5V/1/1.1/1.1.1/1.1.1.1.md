# 𑁧𑁇𑁧𑁇𑁧𑁇𑁧 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀓𑀡𑁆𑀟

(1.)

2\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀞𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁, 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪, 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁? 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺? 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀁 𑀓𑀢𑁆𑀣𑁄𑀕𑀥𑀁 𑀓𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁? 𑀓𑀢𑀫𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀫𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀺? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀫𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺? 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀦𑀬𑁄, 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁄? 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁, 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀅𑀥𑀺𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁? 𑀓𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀺? 𑀓𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺? 𑀓𑀸 𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀺? 𑀓𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀧𑀞𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁? 𑀓𑁂 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀓𑁂 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸? 𑀓𑀢𑁆𑀣 𑀞𑀺𑀢𑀁? 𑀓𑁂 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺? 𑀓𑀲𑁆𑀲 𑀯𑀘𑀦𑀁? 𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺?

3\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀞𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑁄 𑀓𑀮𑀦𑁆𑀤𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀢𑁆𑀣𑀺𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀻𑀢𑀺? 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀁 𑀓𑀢𑁆𑀣𑁄𑀕𑀥𑀁 𑀓𑀢𑁆𑀣 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀦𑁆𑀢𑀺? 𑀦𑀺𑀤𑀸𑀦𑁄𑀕𑀥𑀁, 𑀦𑀺𑀤𑀸𑀦𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁𑁇 𑀓𑀢𑀫𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑁂𑀦 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀫𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀫𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑀢𑀺? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺? 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑁆𑀢𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀻𑀢𑀺? 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘𑁇 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀦𑀬𑁄, 𑀓𑁄 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁄𑀢𑀺? 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀯𑀺𑀦𑀬𑁄, 𑀯𑀺𑀪𑀢𑁆𑀢𑀺 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁄𑁇 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁, 𑀓𑀺𑀁 𑀢𑀢𑁆𑀣 𑀅𑀥𑀺𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀦𑁆𑀢𑀺? 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁, 𑀯𑀺𑀪𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁𑁇 𑀓𑀸 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀅𑀲𑀁𑀯𑀭𑁄 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀸 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀲𑀁𑀯𑀭𑁄 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑁇 𑀓𑀸 𑀧𑀝𑀺𑀧𑀢𑁆𑀢𑀻𑀢𑀺? “𑀦 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀆𑀧𑀸𑀡𑀓𑁄𑀝𑀺𑀓𑀁 𑀲𑀫𑀸𑀤𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑁂𑀲𑀼𑁇 𑀓𑀢𑀺 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀧𑀞𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀧𑀞𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬, 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑀢𑀸𑀬, 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬, 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀓𑁂 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀻𑀢𑀺? 𑀲𑁂𑀓𑁆𑀔𑀸 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀓𑀸 𑀘 𑀲𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑁇 𑀓𑁂 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑀢𑀺? 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀲𑀺𑀓𑁆𑀔𑀸𑁇 𑀓𑀢𑁆𑀣 𑀞𑀺𑀢𑀦𑁆𑀢𑀺? 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑁂𑀲𑀼 𑀞𑀺𑀢𑀁𑁇 𑀓𑁂 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻𑀢𑀺? 𑀬𑁂𑀲𑀁 𑀯𑀢𑁆𑀢𑀢𑀺 𑀢𑁂 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀲𑁆𑀲 𑀯𑀘𑀦𑀦𑁆𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀯𑀘𑀦𑀁 𑀅𑀭𑀳𑀢𑁄 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲𑁇 𑀓𑁂𑀦𑀸𑀪𑀢𑀦𑁆𑀢𑀺? 𑀧𑀭𑀫𑁆𑀧𑀭𑀸𑀪𑀢𑀁—

(3.)

4\. _𑀉𑀧𑀸𑀮𑀺 𑀤𑀸𑀲𑀓𑁄 𑀘𑁂𑀯,_  
_𑀲𑁄𑀡𑀓𑁄 𑀲𑀺𑀕𑁆𑀕𑀯𑁄 𑀢𑀣𑀸;_  
_𑀫𑁄𑀕𑁆𑀕𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘𑀫𑀸,_  
_𑀏𑀢𑁂 𑀚𑀫𑁆𑀩𑀼𑀲𑀺𑀭𑀺𑀯𑁆𑀳𑀬𑁂𑁇_  


5\. _𑀢𑀢𑁄 𑀫𑀳𑀺𑀦𑁆𑀤𑁄 𑀇𑀝𑁆𑀝𑀺𑀬𑁄,_  
_𑀉𑀢𑁆𑀢𑀺𑀬𑁄 𑀲𑀫𑁆𑀩𑀮𑁄 𑀢𑀣𑀸;_  
_𑀪𑀤𑁆𑀤𑀦𑀸𑀫𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


6\. _𑀏𑀢𑁂 𑀦𑀸𑀕𑀸 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸,_  
_𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑀸 𑀇𑀥𑀸𑀕𑀢𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀢𑁂 𑀯𑀸𑀘𑀬𑀺𑀁𑀲𑀼,_  
_𑀧𑀺𑀝𑀓𑀁 𑀢𑀫𑁆𑀩𑀧𑀡𑁆𑀡𑀺𑀬𑀸𑁇_  


7\. _𑀦𑀺𑀓𑀸𑀬𑁂 𑀧𑀜𑁆𑀘 𑀯𑀸𑀘𑁂𑀲𑀼𑀁,_  
_𑀲𑀢𑁆𑀢 𑀘𑁂𑀯 𑀧𑀓𑀭𑀡𑁂;_  
_𑀢𑀢𑁄 𑀅𑀭𑀺𑀝𑁆𑀞𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀢𑀺𑀲𑁆𑀲𑀤𑀢𑁆𑀢𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


8\. _𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀓𑀸𑀍𑀅𑀲𑀼𑀫𑀦𑁄,_  
_𑀣𑁂𑀭𑁄 𑀘 𑀤𑀻𑀖𑀦𑀸𑀫𑀓𑁄;_  
_𑀤𑀻𑀖𑀲𑀼𑀫𑀦𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


9\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀓𑀸𑀍𑀅𑀲𑀼𑀫𑀦𑁄,_  
_𑀦𑀸𑀕𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀩𑀼𑀤𑁆𑀥𑀭𑀓𑁆𑀔𑀺𑀢𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀤𑁂𑀯𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


10\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀼𑀫𑀦𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀘𑀽𑀍𑀅𑀦𑀸𑀕𑁄,_  
_𑀕𑀚𑁄𑀯 𑀤𑀼𑀧𑁆𑀧𑀥𑀁𑀲𑀺𑀬𑁄𑁇_  


11\. _𑀥𑀫𑁆𑀫𑀧𑀸𑀮𑀺𑀢𑀦𑀸𑀫𑁄 𑀘,_  
_𑀭𑁄𑀳𑀡𑁂 𑀲𑀸𑀥𑀼𑀧𑀽𑀚𑀺𑀢𑁄;_  
_𑀢𑀲𑁆𑀲 𑀲𑀺𑀲𑁆𑀲𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀔𑁂𑀫𑀦𑀸𑀫𑁄 𑀢𑀺𑀧𑁂𑀝𑀓𑁄𑁇_  


12\. _𑀤𑀻𑀧𑁂 𑀢𑀸𑀭𑀓𑀭𑀸𑀚𑀸𑀯,_  
_𑀧𑀜𑁆𑀜𑀸𑀬 𑀅𑀢𑀺𑀭𑁄𑀘𑀣;_  
_𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀨𑀼𑀲𑁆𑀲𑀤𑁂𑀯𑁄 𑀫𑀳𑀸𑀓𑀣𑀻𑁇_  


13\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀼𑀫𑀦𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀧𑀼𑀧𑁆𑀨𑀦𑀸𑀫𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀫𑀳𑀸𑀓𑀣𑀻 𑀫𑀳𑀸𑀲𑀺𑀯𑁄,_  
_𑀧𑀺𑀝𑀓𑁂 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


14\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀉𑀧𑀸𑀮𑀺 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀫𑀳𑀸𑀦𑀸𑀕𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀯𑀁𑀲𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


15\. _𑀧𑀼𑀦𑀤𑁂𑀯 𑀅𑀪𑀬𑁄 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀧𑀺𑀝𑀓𑁂 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄𑁇_  


16\. _𑀢𑀲𑁆𑀲 𑀲𑀺𑀲𑁆𑀲𑁄 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄,_  
_𑀧𑀼𑀧𑁆𑀨𑀦𑀸𑀫𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀲𑀸𑀲𑀦𑀁 𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑁄,_  
_𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁂 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇_  


17\. _𑀘𑀽𑀍𑀆𑀪𑀬𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀢𑀺𑀲𑁆𑀲𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀯𑀁𑀲𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


18\. _𑀘𑀽𑀍𑀅𑀤𑁂𑀯𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀘 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀲𑀺𑀯𑀢𑁆𑀣𑁂𑀭𑁄 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀯𑀺𑀦𑀬𑁂 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


19\. _𑀏𑀢𑁂 𑀦𑀸𑀕𑀸 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑀸,_  
_𑀯𑀺𑀦𑀬𑀜𑁆𑀜𑀽 𑀫𑀕𑁆𑀕𑀓𑁄𑀯𑀺𑀤𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀤𑀻𑀧𑁂 𑀧𑀓𑀸𑀲𑁂𑀲𑀼𑀁,_  
_𑀧𑀺𑀝𑀓𑀁 𑀢𑀫𑁆𑀩𑀧𑀡𑁆𑀡𑀺𑀬𑀸𑀢𑀺𑁇_  


(4.)

20\. 𑀬𑀁 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀚𑀸𑀦𑀢𑀸 𑀧𑀲𑁆𑀲𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀥𑀦𑀺𑀬𑀁 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀥𑀦𑀺𑀬𑁄 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀤𑀸𑀭𑀽𑀦𑀺 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺? 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(5.)

21\. 𑀢𑀢𑀺𑀬𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺? 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(6.)

22\. 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀯𑀕𑁆𑀕𑀼𑀫𑀼𑀤𑀸𑀢𑀻𑀭𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀺𑀳𑀻𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀻𑀢𑀺? 𑀢𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

24\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



25\. _𑀫𑁂𑀣𑀼𑀦𑀸𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀜𑁆𑀘,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀼𑀢𑁆𑀢𑀭𑀺;_  
_𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺,_  
_𑀙𑁂𑀚𑁆𑀚𑀯𑀢𑁆𑀣𑀽 𑀅𑀲𑀁𑀲𑀬𑀸𑀢𑀺𑁇_
