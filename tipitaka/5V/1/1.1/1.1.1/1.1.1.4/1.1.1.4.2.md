# 𑁧𑁇𑁧𑁇𑁧𑁇𑁪𑁇𑁨 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀪𑀺𑀓𑁆𑀔𑀼𑀯𑀺𑀪𑀗𑁆𑀕

### 𑀧𑀞𑀫𑀪𑀸𑀕

### 𑀓𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀓𑀡𑁆𑀟

### 𑀓𑁄𑀲𑀺𑀬𑀯𑀕𑁆𑀕

(34.)

93\. 𑀓𑁄𑀲𑀺𑀬𑀫𑀺𑀲𑁆𑀲𑀓𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀆𑀍𑀅𑀯𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀺𑀬𑀓𑀸𑀭𑀓𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼 “𑀩𑀳𑀽, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑁄𑀲𑀓𑀸𑀭𑀓𑁂 𑀧𑀘𑀣𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀤𑀲𑁆𑀲𑀣𑁇 𑀫𑀬𑀫𑁆𑀧𑀺 𑀇𑀘𑁆𑀙𑀸𑀫 𑀓𑁄𑀲𑀺𑀬𑀫𑀺𑀲𑁆𑀲𑀓𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(35.)

94\. 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(36.)

95\. 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀼𑀮𑀁 𑀑𑀤𑀸𑀢𑀸𑀦𑀁 𑀢𑀼𑀮𑀁 𑀕𑁄𑀘𑀭𑀺𑀬𑀸𑀦𑀁 𑀦𑀯𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁄𑀓𑀜𑁆𑀜𑁂𑀯 𑀑𑀤𑀸𑀢𑀁 𑀅𑀦𑁆𑀢𑁂 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀣𑁂𑀯 𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀍𑀅𑀓𑀸𑀦𑀁 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(37.)

96\. 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀁 𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀸 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(38.)

97\. 𑀅𑀦𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀸𑀡𑀲𑀦𑁆𑀣𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀁 𑀦𑀯𑀁 𑀦𑀺𑀲𑀻𑀤𑀦𑀲𑀦𑁆𑀣𑀢𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀣𑀢𑀸𑀦𑀺 𑀉𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀗𑁆𑀕𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀗𑁆𑀕𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀗𑁆𑀕𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(39.)

98\. 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀺𑀬𑁄𑀚𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀺𑀬𑁄𑀚𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄…𑀧𑁂… 𑁇

(40.)

99\. 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀏𑀍𑀅𑀓𑀮𑁄𑀫𑀸𑀦𑀺 𑀥𑁄𑀯𑀸𑀧𑁂𑀲𑀼𑀁, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(41.)

100\. 𑀭𑀽𑀧𑀺𑀬𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀽𑀧𑀺𑀬𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(42.)

101\. 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀭𑀽𑀧𑀺𑀬𑀲𑀁𑀯𑁄𑀳𑀸𑀭𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇

(43.)

102\. 𑀦𑀸𑀦𑀧𑁆𑀧𑀓𑀸𑀭𑀓𑀁 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀓𑀢𑁆𑀣 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀦𑁆𑀢𑀺? 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀓𑀁 𑀆𑀭𑀩𑁆𑀪𑀸𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀭𑀩𑁆𑀪𑁇 𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀦𑁆𑀢𑀺? 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀓𑀬𑀯𑀺𑀓𑁆𑀓𑀬𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺, 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁𑁇 𑀏𑀓𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺…𑀧𑁂… 𑁇
