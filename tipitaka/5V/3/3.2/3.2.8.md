# 𑁩𑁇𑁨𑁇𑁮 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

### 𑀢𑁂𑀭𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(265.)

1131\. _𑀅𑀤𑁆𑀥𑀸𑀦𑀦𑀸𑀯𑀁 𑀧𑀡𑀻𑀢𑀁,_  
_𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁂𑀦 𑀲𑀁𑀳𑀭𑁂;_  
_𑀥𑀜𑁆𑀜𑀁 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀘𑁂𑀯,_  
_𑀅𑀝𑁆𑀞 𑀘 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑁇_  


1132\. _𑀲𑀺𑀓𑁆𑀔𑀸 𑀧𑀦𑁆𑀦𑀭𑀲 𑀏𑀢𑁂,_  
_𑀓𑀸𑀬𑀸 𑀦 𑀯𑀸𑀘𑀸 𑀦 𑀫𑀦𑀸;_  
_𑀓𑀸𑀬𑀯𑀸𑀘𑀸𑀳𑀺 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀦 𑀢𑁂 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀚𑀸𑀬𑀭𑁂𑁇_  


1133\. _𑀓𑀸𑀬𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀦 𑀢𑁂 𑀚𑀸𑀬𑀦𑁆𑀢𑀺 𑀯𑀸𑀘𑀢𑁄;_  
_𑀓𑀸𑀬𑀯𑀸𑀘𑀸𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸 𑀘𑀢𑀼𑀩𑁆𑀩𑀺𑀥𑀸;_  
_𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀩𑀼𑀤𑁆𑀥𑀜𑀸𑀡𑁂𑀦,_  
_𑀅𑀤𑁆𑀥𑀸𑀦𑁂𑀦 𑀲𑀳𑀸 𑀲𑀫𑀸𑁇_  


---

1134\. 𑀅𑀤𑁆𑀥𑀸𑀦𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
