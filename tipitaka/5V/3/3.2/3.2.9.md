# 𑁩𑁇𑁨𑁇𑁯 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

* 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

* 𑀢𑁂𑀭𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

* 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(266.)

1135\. _𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀁 𑀉𑀧𑀲𑁆𑀲𑀼𑀢𑀺,_  
_𑀲𑀽𑀧𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑁂𑀦 𑀘;_  
_𑀭𑀢𑁆𑀢𑀺𑀙𑀦𑁆𑀦𑀜𑁆𑀘 𑀑𑀓𑀸𑀲𑀁,_  
_𑀏𑀢𑁂 𑀩𑁆𑀬𑀽𑀳𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑁇_  


1136\. _𑀓𑀸𑀬𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀦 𑀢𑁂 𑀚𑀸𑀬𑀦𑁆𑀢𑀺 𑀯𑀸𑀘𑀢𑁄;_  
_𑀢𑀻𑀳𑀺 𑀤𑁆𑀯𑀸𑀭𑁂𑀳𑀺 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀤𑁆𑀯𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑀸 𑀇𑀫𑁂;_  
_𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸,_  
_𑀤𑁂𑀲𑀺𑀢𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


---

1137\. 𑀣𑁂𑀬𑁆𑀬𑀲𑀢𑁆𑀣𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
