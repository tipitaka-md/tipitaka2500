# 𑁩𑁇𑁨𑁇𑁧 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

### 𑀢𑁂𑀭𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(258.)

1088\. _𑀫𑁂𑀣𑀼𑀦𑀁 𑀲𑀼𑀓𑁆𑀓𑀲𑀁𑀲𑀕𑁆𑀕𑁄,_  
_𑀅𑀦𑀺𑀬𑀢𑀸 𑀧𑀞𑀫𑀺𑀓𑀸;_  
_𑀧𑀼𑀩𑁆𑀩𑀽𑀧𑀧𑀭𑀺𑀧𑀸𑀘𑀺𑀢𑀸,_  
_𑀭𑀳𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀳𑁇_  


1089\. _𑀲𑀪𑁄𑀚𑀦𑁂 𑀭𑀳𑁄 𑀤𑁆𑀯𑁂 𑀘,_  
_𑀅𑀗𑁆𑀕𑀼𑀮𑀺 𑀉𑀤𑀓𑁂 𑀳𑀲𑀁;_  
_𑀧𑀳𑀸𑀭𑁂 𑀉𑀕𑁆𑀕𑀺𑀭𑁂 𑀘𑁂𑀯,_  
_𑀢𑁂𑀧𑀜𑁆𑀜𑀸𑀲𑀸 𑀘 𑀲𑁂𑀔𑀺𑀬𑀸𑁇_  


1090\. _𑀅𑀥𑀓𑁆𑀔𑀕𑀸𑀫𑀸𑀯𑀲𑁆𑀲𑀼𑀢𑀸,_  
_𑀢𑀮𑀫𑀝𑁆𑀞𑀜𑁆𑀘 𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀸;_  
_𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀘 𑀑𑀯𑀸𑀤𑀁,_  
_𑀦𑀸𑀦𑀼𑀩𑀦𑁆𑀥𑁂 𑀧𑀯𑀢𑁆𑀢𑀺𑀦𑀺𑀁𑁇_  


1091\. _𑀙𑀲𑀢𑁆𑀢𑀢𑀺 𑀇𑀫𑁂 𑀲𑀺𑀓𑁆𑀔𑀸,_  
_𑀓𑀸𑀬𑀫𑀸𑀦𑀲𑀺𑀓𑀸 𑀓𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀏𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸,_  
_𑀧𑀞𑀫𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀬𑀣𑀸𑁇_  


---

1092\. 𑀧𑀞𑀫𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
