# 𑁩𑁇𑁨𑁇𑁪 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

### 𑀢𑁂𑀭𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(261.)

1111\. _𑀪𑁂𑀤𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀩𑀘-_  
_𑀤𑀽𑀲𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀤𑀺𑀝𑁆𑀞𑀺 𑀘;_  
_𑀙𑀦𑁆𑀤𑀁 𑀉𑀚𑁆𑀚𑀕𑁆𑀖𑀺𑀓𑀸 𑀤𑁆𑀯𑁂 𑀘,_  
_𑀤𑁆𑀯𑁂 𑀘 𑀲𑀤𑁆𑀤𑀸 𑀦 𑀩𑁆𑀬𑀸𑀳𑀭𑁂𑁇_  


1112\. _𑀙𑀫𑀸 𑀦𑀻𑀘𑀸𑀲𑀦𑁂 𑀞𑀸𑀦𑀁,_  
_𑀧𑀘𑁆𑀙𑀢𑁄 𑀉𑀧𑁆𑀧𑀣𑁂𑀦 𑀘;_  
_𑀯𑀚𑁆𑀚𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀕𑀳𑀡𑀸,_  
_𑀑𑀲𑀸𑀭𑁂 𑀧𑀘𑁆𑀘𑀸𑀘𑀺𑀓𑁆𑀔𑀦𑀸𑁇_  


1113\. _𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀤𑁆𑀯𑁂 𑀯𑀥𑀺,_  
_𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀸𑀬 𑀘;_  
_𑀧𑀼𑀦 𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀦 𑀯𑀽𑀧𑀲𑀫𑁂,_  
_𑀆𑀭𑀸𑀫𑀜𑁆𑀘 𑀧𑀯𑀸𑀭𑀡𑀸𑁇_  


1114\. _𑀅𑀦𑁆𑀯𑀤𑁆𑀥𑀁 𑀲𑀳 𑀚𑀻𑀯𑀺𑀦𑀺𑀁 𑀤𑁆𑀯𑁂,_  
_𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀦𑀸;_  
_𑀲𑀢𑁆𑀢𑀢𑀺𑀁𑀲 𑀇𑀫𑁂 𑀥𑀫𑁆𑀫𑀸,_  
_𑀓𑀸𑀬𑀯𑀸𑀘𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑁂 𑀏𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸,_  
_𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸 𑀬𑀣𑀸𑁇_  


---

1115\. 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
