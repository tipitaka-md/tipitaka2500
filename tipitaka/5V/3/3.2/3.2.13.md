# 𑁩𑁇𑁨𑁇𑁧𑁩 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

* 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

* 𑀢𑁂𑀭𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

* 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(270.)

1147\. _𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀯𑀸𑀘𑀸𑀬,_  
_𑀦 𑀓𑀸𑀬𑀸 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄;_  
_𑀚𑀸𑀬𑀢𑀺 𑀓𑀸𑀬𑀯𑀸𑀘𑀸𑀬,_  
_𑀦 𑀢𑀁 𑀚𑀸𑀬𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀢𑁄𑁇_  


1148\. _𑀚𑀸𑀬𑀢𑀺 𑀯𑀸𑀘𑀸𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀦 𑀢𑀁 𑀚𑀸𑀬𑀢𑀺 𑀓𑀸𑀬𑀢𑁄;_  
_𑀚𑀸𑀬𑀢𑀺 𑀢𑀻𑀳𑀺 𑀤𑁆𑀯𑀸𑀭𑁂𑀳𑀺,_  
_𑀅𑀓𑀢𑀁 𑀘𑀢𑀼𑀞𑀸𑀦𑀺𑀓𑀁𑁇_  


---

1149\. 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



1150\. _𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀜𑁆𑀳𑀺 𑀲𑀗𑁆𑀔𑁂𑀧𑀁,_  
_𑀤𑀲 𑀢𑀻𑀡𑀺 𑀲𑀼𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀅𑀲𑀫𑁆𑀫𑁄𑀳𑀓𑀭𑀁 𑀞𑀸𑀦𑀁,_  
_𑀦𑁂𑀢𑁆𑀢𑀺𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁;_  
_𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑁄 𑀇𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀽,_  
_𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂 𑀦 𑀫𑀼𑀬𑁆𑀳𑀢𑀻𑀢𑀺𑁇_  


---

1151\. 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
