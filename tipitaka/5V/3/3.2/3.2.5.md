# 𑁩𑁇𑁨𑁇𑁫 𑀓𑀣𑀺𑀦𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

### 𑀢𑁂𑀭𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

### 𑀓𑀣𑀺𑀦𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(262.)

1116. _𑀉𑀩𑁆𑀪𑀢𑀁 𑀓𑀣𑀺𑀦𑀁 𑀢𑀻𑀡𑀺,_  
_𑀧𑀞𑀫𑀁 𑀧𑀢𑁆𑀢𑀪𑁂𑀲𑀚𑁆𑀚𑀁;_  
_𑀅𑀘𑁆𑀘𑁂𑀓𑀁 𑀘𑀸𑀧𑀺 𑀲𑀸𑀲𑀗𑁆𑀓𑀁,_  
_𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁂𑀦 𑀯𑀸 𑀤𑀼𑀯𑁂𑁇_  


1117. _𑀉𑀧𑀲𑁆𑀲𑀬𑀁 𑀧𑀭𑀫𑁆𑀧𑀭𑀸,_  
_𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀸;_  
_𑀯𑀺𑀓𑀧𑁆𑀧𑀁 𑀭𑀜𑁆𑀜𑁄 𑀯𑀺𑀓𑀸𑀮𑁂,_  
_𑀯𑁄𑀲𑀸𑀲𑀸𑀭𑀜𑁆𑀜𑀓𑁂𑀦 𑀘𑁇_  


1118. _𑀉𑀲𑁆𑀲𑀬𑀸𑀲𑀦𑁆𑀦𑀺𑀘𑀬𑀜𑁆𑀘,_  
_𑀧𑀼𑀭𑁂 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀓𑀸𑀮𑁂 𑀘;_  
_𑀧𑀜𑁆𑀘𑀸𑀳𑀺𑀓𑀸 𑀲𑀗𑁆𑀓𑀫𑀦𑀻,_  
_𑀤𑁆𑀯𑁂𑀧𑀺 𑀆𑀯𑀲𑀣𑁂𑀦 𑀘𑁇_  


1119. _𑀧𑀲𑀸𑀔𑁂 𑀆𑀲𑀦𑁂 𑀘𑁂𑀯,_  
_𑀢𑀺𑀁𑀲 𑀏𑀓𑀽𑀦𑀓𑀸 𑀇𑀫𑁂;_  
_𑀓𑀸𑀬𑀯𑀸𑀘𑀸 𑀦 𑀘 𑀘𑀺𑀢𑁆𑀢𑀸,_  
_𑀢𑀻𑀳𑀺 𑀤𑁆𑀯𑀸𑀭𑁂𑀳𑀺 𑀚𑀸𑀬𑀭𑁂;_  
_𑀤𑁆𑀯𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀓𑀣𑀺𑀦𑁂𑀦 𑀲𑀳𑀸 𑀲𑀫𑀸𑁇_  


---

1120. 𑀓𑀣𑀺𑀦𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
