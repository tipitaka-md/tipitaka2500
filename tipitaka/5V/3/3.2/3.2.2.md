# 𑁩𑁇𑁨𑁇𑁨 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

* 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

* 𑀢𑁂𑀭𑀲𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

* 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦

(259.)

1093\. _𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀯𑀺𑀕𑁆𑀕𑀳𑀼𑀢𑁆𑀢𑀭𑀺,_  
_𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀅𑀢𑁆𑀢𑀓𑀸𑀫𑀺𑀦𑀁;_  
_𑀅𑀫𑀽𑀮𑀸 𑀅𑀜𑁆𑀜𑀪𑀸𑀕𑀺𑀬𑀸,_  
_𑀅𑀦𑀺𑀬𑀢𑀸 𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑁇_  


1094\. _𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑁂 𑀧𑀭𑀺𑀡𑀸𑀫𑀦𑁂,_  
_𑀫𑀼𑀲𑀸 𑀑𑀫𑀲𑀧𑁂𑀲𑀼𑀡𑀸;_  
_𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀧𑀣𑀯𑀻𑀔𑀡𑁂,_  
_𑀪𑀽𑀢𑀁 𑀅𑀜𑁆𑀜𑀸𑀬 𑀉𑀚𑁆𑀛𑀸𑀧𑁂𑁇_  


1095\. _𑀦𑀺𑀓𑁆𑀓𑀟𑁆𑀠𑀦𑀁 𑀲𑀺𑀜𑁆𑀘𑀦𑀜𑁆𑀘,_  
_𑀆𑀫𑀺𑀲𑀳𑁂𑀢𑀼 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀻;_  
_𑀏𑀳𑀺 𑀅𑀦𑀸𑀤𑀭𑀺 𑀪𑀺𑀁𑀲𑀸,_  
_𑀅𑀧𑀦𑀺𑀥𑁂 𑀘 𑀚𑀻𑀯𑀺𑀢𑀁𑁇_  


1096\. _𑀚𑀸𑀦𑀁 𑀲𑀧𑁆𑀧𑀸𑀡𑀓𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀊𑀦𑀲𑀁𑀯𑀸𑀲𑀦𑀸𑀲𑀦𑀸;_  
_𑀲𑀳𑀥𑀫𑁆𑀫𑀺𑀓𑀯𑀺𑀮𑁂𑀔𑀸,_  
_𑀫𑁄𑀳𑁄 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀘𑁇_  


1097\. _𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀬𑁆𑀬 𑀧𑀼𑀕𑁆𑀕𑀮𑁂;_  
_𑀓𑀺𑀁 𑀢𑁂 𑀅𑀓𑀸𑀮𑀁 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑁂,_  
_𑀤𑀼𑀕𑁆𑀕𑀳𑀻 𑀦𑀺𑀭𑀬𑁂𑀦 𑀘𑁇_  


1098\. _𑀕𑀡𑀁 𑀯𑀺𑀪𑀗𑁆𑀕𑀁 𑀤𑀼𑀩𑁆𑀩𑀮𑀁,_  
_𑀓𑀣𑀺𑀦𑀸𑀨𑀸𑀲𑀼𑀧𑀲𑁆𑀲𑀬𑀁;_  
_𑀅𑀓𑁆𑀓𑁄𑀲𑀘𑀡𑁆𑀟𑀻 𑀫𑀘𑁆𑀙𑀭𑀻,_  
_𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀘 𑀧𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸𑁇_  


1099\. _𑀤𑁆𑀯𑁂𑀯𑀲𑁆𑀲𑀁 𑀲𑀺𑀓𑁆𑀔𑀸 𑀲𑀁𑀖𑁂𑀦,_  
_𑀢𑀬𑁄 𑀘𑁂𑀯 𑀕𑀺𑀳𑀻𑀕𑀢𑀸;_  
_𑀓𑀼𑀫𑀸𑀭𑀺𑀪𑀽𑀢𑀸 𑀢𑀺𑀲𑁆𑀲𑁄 𑀘,_  
_𑀊𑀦𑀤𑁆𑀯𑀸𑀤𑀲𑀲𑀫𑁆𑀫𑀢𑀸𑁇_  


1100\. _𑀅𑀮𑀁 𑀢𑀸𑀯 𑀲𑁄𑀓𑀸𑀯𑀸𑀲𑀁,_  
_𑀙𑀦𑁆𑀤𑀸 𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀸 𑀘 𑀤𑁆𑀯𑁂;_  
_𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸 𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑁂,_  
_𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸 𑀢𑀺𑀓𑀸 𑀓𑀢𑀸𑁇_  


1101\. _𑀓𑀸𑀬𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀦 𑀯𑀸𑀘𑀸,_  
_𑀯𑀸𑀘𑀸𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀓𑀸𑀬𑀺𑀓𑀁;_  
_𑀢𑀻𑀳𑀺 𑀤𑁆𑀯𑀸𑀭𑁂𑀳𑀺 𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁 𑀬𑀣𑀸𑁇_  


---

1102\. 𑀤𑀼𑀢𑀺𑀬𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
