# 𑁩𑁇𑁧 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦

### 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑀻𑀲𑀲𑀗𑁆𑀔𑁂𑀧

### 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦

(257.)

1076. _𑀅𑀦𑀺𑀘𑁆𑀘𑀸 𑀲𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀔𑀸𑀭𑀸,_  
_𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀢𑁆𑀢𑀸 𑀘 𑀲𑀗𑁆𑀔𑀢𑀸;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀜𑁆𑀘𑁂𑀯 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺,_  
_𑀅𑀦𑀢𑁆𑀢𑀸 𑀇𑀢𑀺 𑀦𑀺𑀘𑁆𑀙𑀬𑀸𑁇_  


1077. _𑀩𑀼𑀤𑁆𑀥𑀘𑀦𑁆𑀤𑁂 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀩𑀼𑀤𑁆𑀥𑀸𑀤𑀺𑀘𑁆𑀘𑁂 𑀅𑀦𑀼𑀕𑁆𑀕𑀢𑁂;_  
_𑀢𑁂𑀲𑀁 𑀲𑀪𑀸𑀕𑀥𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀦𑀸𑀫𑀫𑀢𑁆𑀢𑀁 𑀦 𑀦𑀸𑀬𑀢𑀺𑁇_  


1078. _𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀯𑀺𑀯𑀺𑀥𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀽𑀭𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀸𑀭𑀫𑀻;_  
_𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀫𑀳𑀸𑀯𑀻𑀭𑀸,_  
_𑀘𑀓𑁆𑀔𑀼𑀪𑀽𑀢𑀸 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂𑁇_  


1079. _𑀢𑁂 𑀤𑁂𑀲𑀬𑀦𑁆𑀢𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁,_  
_𑀤𑀼𑀓𑁆𑀔𑀳𑀸𑀦𑀺𑀁 𑀲𑀼𑀔𑀸𑀯𑀳𑀁;_  
_𑀅𑀗𑁆𑀕𑀻𑀭𑀲𑁄 𑀲𑀓𑁆𑀬𑀫𑀼𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄𑁇_  


1080. _𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀼𑀢𑁆𑀢𑀫𑁄 𑀲𑀻𑀳𑁄,_  
_𑀧𑀺𑀝𑀓𑁂 𑀢𑀻𑀡𑀺 𑀤𑁂𑀲𑀬𑀺;_  
_𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀫𑀪𑀺𑀥𑀫𑁆𑀫𑀜𑁆𑀘,_  
_𑀯𑀺𑀦𑀬𑀜𑁆𑀘 𑀫𑀳𑀸𑀕𑀼𑀡𑀁𑁇_  


1081. _𑀏𑀯𑀁 𑀦𑀻𑀬𑀢𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁄,_  
_𑀯𑀺𑀦𑀬𑁄 𑀬𑀤𑀺 𑀢𑀺𑀝𑁆𑀞𑀢𑀺;_  
_𑀉𑀪𑀢𑁄 𑀘 𑀯𑀺𑀪𑀗𑁆𑀕𑀸𑀦𑀺,_  
_𑀔𑀦𑁆𑀥𑀓𑀸 𑀬𑀸 𑀘 𑀫𑀸𑀢𑀺𑀓𑀸𑁇_  


1082. _𑀫𑀸𑀮𑀸 𑀲𑀼𑀢𑁆𑀢𑀕𑀼𑀡𑁂𑀦𑁂𑀯,_  
_𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦 𑀕𑀦𑁆𑀣𑀺𑀢𑀸;_  
_𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀭𑀺𑀯𑀸𑀭𑀲𑁆𑀲,_  
_𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦𑀺𑀬𑀢𑁄 𑀓𑀢𑀁𑁇_  


1083. _𑀲𑀫𑁆𑀪𑁂𑀤𑀁 𑀦𑀺𑀤𑀸𑀦𑀜𑁆𑀘𑀜𑁆𑀜𑀁,_  
_𑀲𑀼𑀢𑁆𑀢𑁂 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀧𑀭𑀺;_  
_𑀢𑀲𑁆𑀫𑀸 𑀲𑀺𑀓𑁆𑀔𑁂 𑀧𑀭𑀺𑀯𑀸𑀭𑀁,_  
_𑀥𑀫𑁆𑀫𑀓𑀸𑀫𑁄 𑀲𑀼𑀧𑁂𑀲𑀮𑁄𑀢𑀺𑁇_
