# 𑁭𑁇𑁬 𑀙𑀓𑁆𑀓𑀯𑀸𑀭

### 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

### 𑀙𑀓𑁆𑀓𑀯𑀸𑀭

(326.)

1520\. 𑀙 𑀅𑀕𑀸𑀭𑀯𑀸𑁇 𑀙 𑀕𑀸𑀭𑀯𑀸𑁇 𑀙 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀙 𑀲𑀸𑀫𑀻𑀘𑀺𑀬𑁄𑁇 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀙𑀘𑁆𑀙𑁂𑀤𑀦𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀙𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀙𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀯𑀺𑀦𑀬𑀥𑀭𑁂𑁇 𑀙 𑀧𑀭𑀫𑀸𑀦𑀺𑁇 𑀙𑀸𑀭𑀢𑁆𑀢𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀙 𑀘𑀻𑀯𑀭𑀸𑀦𑀺𑁇 𑀙 𑀭𑀚𑀦𑀸𑀦𑀺𑁇 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀦 𑀯𑀸𑀘𑀢𑁄𑁇 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀦 𑀓𑀸𑀬𑀢𑁄𑁇 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀙 𑀓𑀫𑁆𑀫𑀸𑀦𑀺𑁇 𑀙 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀙 𑀅𑀦𑀼𑀯𑀸𑀤𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀙 𑀲𑀸𑀭𑀡𑀻𑀬𑀸 𑀥𑀫𑁆𑀫𑀸 𑀤𑀻𑀖𑀲𑁄𑁇 𑀙 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄, 𑀲𑀼𑀕𑀢𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑀸, 𑀢𑀺𑀭𑀺𑀬𑀁 𑀙 𑀯𑀺𑀤𑀢𑁆𑀣𑀺𑀬𑁄𑁇 𑀙 𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑁄 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀸𑁇 𑀙 𑀦𑀳𑀸𑀦𑁂 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄—  𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀘𑀻𑀯𑀭𑀁 𑀲𑀫𑀸𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇

1521\. 𑀙𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀧𑀜𑁆𑀜𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇

1522\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀙𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀁 𑀅𑀲𑁂𑀓𑁆𑀔𑁂 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀁 𑀅𑀲𑁂𑀓𑁆𑀔𑁂 𑀲𑀫𑀸𑀥𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀧𑀜𑁆𑀜𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀁 𑀅𑀲𑁂𑀓𑁆𑀔𑁂 𑀧𑀜𑁆𑀜𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀁 𑀅𑀲𑁂𑀓𑁆𑀔𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸; 𑀅𑀢𑁆𑀢𑀦𑀸 𑀅𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀁 𑀅𑀲𑁂𑀓𑁆𑀔𑁂 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀜𑀸𑀡𑀤𑀲𑁆𑀲𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸; 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇

1523\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀙𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀲𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺, 𑀳𑀺𑀭𑀺𑀫𑀸 𑀳𑁄𑀢𑀺, 𑀑𑀢𑁆𑀢𑀧𑁆𑀧𑀻 𑀳𑁄𑀢𑀺, 𑀆𑀭𑀤𑁆𑀥𑀯𑀻𑀭𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺 𑀳𑁄𑀢𑀺, 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇

1524\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀙𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀦 𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀅𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑁂 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀅𑀢𑀺𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀜𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇

1525\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀙𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀧𑀝𑀺𑀩𑀮𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀁 𑀯𑀸 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀼𑀁 𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀼𑀁 𑀯𑀸, 𑀅𑀦𑀪𑀺𑀭𑀢𑀁 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀢𑀼𑀁 𑀯𑀸 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀢𑀼𑀁 𑀯𑀸, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀢𑁄 𑀯𑀺𑀦𑁄𑀤𑁂𑀢𑀼𑀁, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇

1526\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀙𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀧𑀝𑀺𑀩𑀮𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀁 𑀯𑀸 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀁 𑀯𑀸 𑀆𑀪𑀺𑀲𑀫𑀸𑀘𑀸𑀭𑀺𑀓𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑁂𑀢𑀼𑀁, 𑀆𑀤𑀺𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀬𑀓𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀯𑀺𑀦𑁂𑀢𑀼𑀁, 𑀅𑀪𑀺𑀥𑀫𑁆𑀫𑁂 𑀯𑀺𑀦𑁂𑀢𑀼𑀁, 𑀅𑀪𑀺𑀯𑀺𑀦𑀬𑁂 𑀯𑀺𑀦𑁂𑀢𑀼𑀁, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀥𑀫𑁆𑀫𑀢𑁄 𑀯𑀺𑀯𑁂𑀘𑁂𑀢𑀼𑀁, 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇

1527\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀙𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀪𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑁆𑀯𑀸𑀕𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀻𑀦𑀺 𑀲𑀼𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑀲𑁄 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑁄, 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇

1528\. 𑀙 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺, 𑀙 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀻𑀢𑀺𑁇

1529\. 𑀙𑀓𑁆𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

1530\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1531\. _𑀅𑀕𑀸𑀭𑀯𑀸 𑀕𑀸𑀭𑀯𑀸 𑀘,_  
_𑀯𑀺𑀦𑀻𑀢𑀸 𑀲𑀸𑀫𑀻𑀘𑀺𑀧𑀺 𑀘;_  
_𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸 𑀙𑁂𑀤𑀦𑀸 𑀘𑁂𑀯,_  
_𑀆𑀓𑀸𑀭𑀸𑀦𑀺𑀲𑀁𑀲𑁂𑀦 𑀘𑁇_  


1532\. _𑀧𑀭𑀫𑀸𑀦𑀺 𑀘 𑀙𑀸𑀭𑀢𑁆𑀢𑀁,_  
_𑀘𑀻𑀯𑀭𑀁 𑀭𑀚𑀦𑀸𑀦𑀺 𑀘;_  
_𑀓𑀸𑀬𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘𑀸𑀧𑀺,_  
_𑀯𑀸𑀘𑀢𑁄 𑀘𑀺𑀢𑁆𑀢𑀢𑁄𑀧𑀺 𑀘𑁇_  


1533\. _𑀓𑀸𑀬𑀯𑀸𑀘𑀸𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘,_  
_𑀓𑀫𑁆𑀫𑀯𑀺𑀯𑀸𑀤𑀫𑁂𑀯 𑀘;_  
_𑀅𑀦𑀼𑀯𑀸𑀤𑀸 𑀤𑀻𑀖𑀲𑁄 𑀘,_  
_𑀢𑀺𑀭𑀺𑀬𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁂𑀦 𑀘𑁇_  


1534\. _𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀆𑀤𑀸𑀬,_  
_𑀲𑀫𑀸𑀤𑀸𑀬 𑀢𑀣𑁂𑀯 𑀘;_  
_𑀅𑀲𑁂𑀓𑁆𑀔𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀸,_  
_𑀲𑀤𑁆𑀥𑁄 𑀅𑀥𑀺𑀲𑀻𑀮𑁂𑀦 𑀘;_  
_𑀕𑀺𑀮𑀸𑀦𑀸𑀪𑀺𑀲𑀫𑀸𑀘𑀸𑀭𑀻,_  
_𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀢𑀺𑁇_
