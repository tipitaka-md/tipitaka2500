# 𑁭𑁇𑁩 𑀢𑀺𑀓𑀯𑀸𑀭

* 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

* 𑀢𑀺𑀓𑀯𑀸𑀭

(323.)

1464\. 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂 𑀪𑀕𑀯𑀢𑀺 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁂; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁂𑀧𑀺 𑀪𑀕𑀯𑀢𑀺 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀭𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁂𑀧𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀓𑀸𑀮𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀯𑀺𑀓𑀸𑀮𑁂; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀺𑀓𑀸𑀮𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀓𑀸𑀮𑁂; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀓𑀸𑀮𑁂 𑀘𑁂𑀯 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀺𑀓𑀸𑀮𑁂 𑀘𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀭𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀤𑀺𑀯𑀸; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑀺𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀭𑀢𑁆𑀢𑀺𑀁; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀤𑀺𑀯𑀸 𑀘𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀊𑀦𑀤𑀲𑀯𑀲𑁆𑀲𑁄; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀊𑀦𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀤𑀲𑀯𑀲𑁆𑀲𑁄; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀘𑁂𑀯 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀊𑀦𑀤𑀲𑀯𑀲𑁆𑀲𑁄 𑀘𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀘𑁂𑀯 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀘𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀓𑀼𑀲𑀮𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀗𑁆𑀕𑀻 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀔𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀗𑁆𑀕𑀻 𑀆𑀧𑀚𑁆𑀚𑀢𑀺; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀯𑁂𑀤𑀦𑀸𑀲𑀫𑀗𑁆𑀕𑀻 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀻𑀡𑀺 𑀘𑁄𑀤𑀦𑀸𑀯𑀢𑁆𑀣𑀽𑀦𑀺—  𑀤𑀺𑀝𑁆𑀞𑁂𑀦, 𑀲𑀼𑀢𑁂𑀦, 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬𑁇 𑀢𑀬𑁄 𑀲𑀮𑀸𑀓𑀕𑁆𑀕𑀸𑀳𑀸—  𑀕𑀼𑀍𑀳𑀓𑁄, 𑀯𑀺𑀯𑀝𑀓𑁄, 𑀲𑀓𑀡𑁆𑀡𑀚𑀧𑁆𑀧𑀓𑁄𑁇 𑀢𑀬𑁄 𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀸—  𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸, 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸, 𑀅𑀲𑀮𑁆𑀮𑁂𑀔𑀢𑀸𑁇 𑀢𑀬𑁄 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸—  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸, 𑀲𑀮𑁆𑀮𑁂𑀔𑀢𑀸𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀢𑀬𑁄 𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀸—  𑀫𑀳𑀺𑀘𑁆𑀙𑀢𑀸, 𑀅𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸, 𑀅𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑁇 𑀢𑀬𑁄 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸—  𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸, 𑀫𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀢𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄—  𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄—  𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀬𑁄—  𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺, 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑁇

1466\. 𑀢𑀻𑀡𑀺 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺𑁇 𑀢𑀻𑀡𑀺 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺𑁇 𑀢𑀬𑁄 𑀧𑀭𑀺𑀯𑀸𑀲𑀸—  𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄, 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄, 𑀲𑀼𑀤𑁆𑀥𑀦𑁆𑀢𑀧𑀭𑀺𑀯𑀸𑀲𑁄𑁇 𑀢𑀬𑁄 𑀫𑀸𑀦𑀢𑁆𑀢𑀸—  𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀫𑀸𑀦𑀢𑁆𑀢𑀁, 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀫𑀸𑀦𑀢𑁆𑀢𑀁, 𑀧𑀓𑁆𑀔𑀫𑀸𑀦𑀢𑁆𑀢𑀁𑁇 𑀢𑀬𑁄 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀭𑀢𑁆𑀢𑀺𑀘𑁆𑀙𑁂𑀤𑀸—  𑀲𑀳𑀯𑀸𑀲𑁄, 𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁄, 𑀅𑀦𑀸𑀭𑁄𑀘𑀦𑀸𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀩𑀳𑀺; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑀳𑀺 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀅𑀦𑁆𑀢𑁄; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑁆𑀢𑁄 𑀘𑁂𑀯 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑀳𑀺 𑀘𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀸𑀬 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀩𑀳𑀺𑀲𑀻𑀫𑀸𑀬; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀩𑀳𑀺𑀲𑀻𑀫𑀸𑀬 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀸𑀬; 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀸𑀬 𑀘𑁂𑀯 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀩𑀳𑀺𑀲𑀻𑀫𑀸𑀬 𑀘𑁇 𑀢𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂, 𑀕𑀡𑀫𑀚𑁆𑀛𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀢𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺, 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂, 𑀕𑀡𑀫𑀚𑁆𑀛𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇 𑀢𑀻𑀡𑀺 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀤𑀸𑀦𑀸𑀦𑀺𑁇 𑀢𑀻𑀡𑀺 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀤𑀸𑀦𑀸𑀦𑀺𑁇

1468\. 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀲𑀁𑀖𑁄 𑀆𑀕𑀸𑀍𑀳𑀸𑀬 𑀘𑁂𑀢𑁂𑀬𑁆𑀬—  𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄, 𑀩𑀸𑀮𑁄 𑀳𑁄𑀢𑀺 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑁄 𑀅𑀦𑀧𑀤𑀸𑀦𑁄, 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀳𑀺 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑁂 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀢𑀺𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀤𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀤𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀤𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀅𑀦𑀸𑀘𑀸𑀭𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀅𑀦𑀸𑀘𑀸𑀭𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀅𑀦𑀸𑀘𑀸𑀭𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑀺𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑀺𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑀺𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀚𑀻𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀚𑀻𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀚𑀻𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀸𑀫𑀡𑁂𑀭𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀖𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀢𑀸𑀤𑀺𑀲𑀺𑀓𑀁, 𑀢𑀢𑁄 𑀯𑀸 𑀧𑀸𑀧𑀺𑀝𑁆𑀞𑀢𑀭𑀁𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁—  𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺𑁇

1469\. 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀧𑁄𑀲𑀣𑀁 𑀞𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  “𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁 𑀫𑀸 𑀓𑀮𑀳𑀁 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆”𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  “𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁 𑀫𑀸 𑀓𑀮𑀳𑀁 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆”𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦 𑀓𑀸𑀘𑀺 𑀲𑀁𑀖𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀤𑀸𑀢𑀩𑁆𑀩𑀸—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀓𑀺𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑀺 𑀧𑀘𑁆𑀘𑁂𑀓𑀝𑁆𑀞𑀸𑀦𑁂 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀑𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀦𑀸𑀮𑀁 𑀑𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀼𑀁—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀯𑀘𑀦𑀻𑀬𑀁 𑀦𑀸𑀤𑀸𑀢𑀩𑁆𑀩𑀁—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀦𑀬𑁄 𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀦𑀬𑁄 𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀦𑀬𑁄 𑀦 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀦𑀬𑁄 𑀦 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀯𑀺𑀦𑀬𑁄 𑀦 𑀲𑀸𑀓𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇 𑀢𑀻𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘𑁇

1470\. 𑀢𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀓𑁄, 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀓𑁄, 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀉𑀧𑁄𑀲𑀣𑁄𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀢𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑀸—  𑀲𑀁𑀖𑁂𑀉𑀧𑁄𑀲𑀣𑁄, 𑀕𑀡𑁂𑀉𑀧𑁄𑀲𑀣𑁄, 𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑀉𑀧𑁄𑀲𑀣𑁄𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀢𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑀸—  𑀲𑀼𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑁄𑀉𑀧𑁄𑀲𑀣𑁄, 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀉𑀧𑁄𑀲𑀣𑁄, 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀧𑁄𑀲𑀣𑁄𑁇

1471\. 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀯𑀸𑀭𑀡𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀓𑀸, 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀓𑀸, 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀯𑀸𑀭𑀡𑀸—  𑀲𑀁𑀖𑁂𑀧𑀯𑀸𑀭𑀡𑀸, 𑀕𑀡𑁂𑀧𑀯𑀸𑀭𑀡𑀸, 𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀅𑀧𑀭𑀸𑀧𑀺 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀯𑀸𑀭𑀡𑀸—  𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀧𑀯𑀸𑀭𑀡𑀸, 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀧𑀯𑀸𑀭𑀡𑀸, 𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀸𑀧𑀯𑀸𑀭𑀡𑀸𑁇

1472\. 𑀢𑀬𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸—  𑀇𑀤𑀫𑀧𑁆𑀧𑀳𑀸𑀬, 𑀬𑁄 𑀘 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀧𑀝𑀺𑀜𑁆𑀜𑁄, 𑀬𑁄 𑀘 𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀁 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑁂𑀢𑀺, 𑀬𑁄 𑀘𑀸𑀬𑀁 𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀏𑀯𑀁𑀤𑀺𑀝𑁆𑀞𑀺—  “𑀦𑀢𑁆𑀣𑀺 𑀓𑀸𑀫𑁂𑀲𑀼 𑀤𑁄𑀲𑁄”𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑁂𑀲𑀼 𑀧𑀸𑀢𑀩𑁆𑀬𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀻𑀡𑀺 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀮𑁄𑀪𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀤𑁄𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀫𑁄𑀳𑁄 𑀅𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁𑁇 𑀢𑀻𑀡𑀺 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀸𑀦𑀺—  𑀅𑀮𑁄𑀪𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀅𑀤𑁄𑀲𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁, 𑀅𑀫𑁄𑀳𑁄 𑀓𑀼𑀲𑀮𑀫𑀽𑀮𑀁𑁇 𑀢𑀻𑀡𑀺 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀸𑀦𑀺—  𑀓𑀸𑀬𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀯𑀘𑀻𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁, 𑀫𑀦𑁄𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁𑁇 𑀢𑀻𑀡𑀺 𑀲𑀼𑀘𑀭𑀺𑀢𑀸𑀦𑀺—  𑀓𑀸𑀬𑀲𑀼𑀘𑀭𑀺𑀢𑀁, 𑀯𑀘𑀻𑀲𑀼𑀘𑀭𑀺𑀢𑀁, 𑀫𑀦𑁄𑀲𑀼𑀘𑀭𑀺𑀢𑀁𑁇 𑀢𑀬𑁄 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀪𑀕𑀯𑀢𑀸 𑀓𑀼𑀮𑁂𑀲𑀼 𑀢𑀺𑀓𑀪𑁄𑀚𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬, 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬, “𑀫𑀸 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀸 𑀧𑀓𑁆𑀔𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀁𑀖𑀁 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺 𑀓𑀼𑀮𑀸𑀦𑀼𑀤𑁆𑀤𑀬𑀢𑀸𑀬 𑀘𑁇 𑀢𑀻𑀳𑀺 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑀢𑀸 𑀧𑀸𑀧𑀫𑀺𑀢𑁆𑀢𑀢𑀸 𑀑𑀭𑀫𑀢𑁆𑀢𑀓𑁂𑀦 𑀯𑀺𑀲𑁂𑀲𑀸𑀥𑀺𑀕𑀫𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀸 𑀯𑁄𑀲𑀸𑀦𑀁 𑀆𑀧𑀸𑀤𑀺𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑁄—  𑀤𑀡𑁆𑀟𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑀺𑀓𑁆𑀓𑀸𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀤𑀡𑁆𑀟𑀲𑀺𑀓𑁆𑀓𑀸𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀼𑀯𑀝𑁆𑀞𑀸𑀦𑀺𑀓𑀸 𑀅𑀲𑀗𑁆𑀓𑀫𑀦𑀻𑀬𑀸—  𑀯𑀘𑁆𑀘𑀧𑀸𑀤𑀼𑀓𑀸, 𑀧𑀲𑁆𑀲𑀸𑀯𑀧𑀸𑀤𑀼𑀓𑀸, 𑀆𑀘𑀫𑀦𑀧𑀸𑀤𑀼𑀓𑀸𑁇 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀸𑀤𑀖𑀁𑀲𑀦𑀺𑀬𑁄—  𑀲𑀓𑁆𑀔𑀭𑀁, 𑀓𑀣𑀮𑀸, 𑀲𑀫𑀼𑀤𑁆𑀤𑀨𑁂𑀡𑀓𑁄𑀢𑀺𑁇

1473\. 𑀢𑀺𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

1474\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1475\. _𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑁂 𑀓𑀸𑀮𑁂 𑀭𑀢𑁆𑀢𑀺𑀜𑁆𑀘,_  
_𑀤𑀲 𑀧𑀜𑁆𑀘 𑀓𑀼𑀲𑀮𑁂𑀦;_  
_𑀯𑁂𑀤𑀦𑀸 𑀘𑁄𑀤𑀦𑀸 𑀯𑀢𑁆𑀣𑀼,_  
_𑀲𑀮𑀸𑀓𑀸 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀸𑁇_  


1476\. _𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀧𑀭𑁂 𑀤𑁆𑀯𑁂 𑀘,_  
_𑀩𑀸𑀮𑁄 𑀓𑀸𑀍𑀏 𑀘 𑀓𑀧𑁆𑀧𑀢𑀺;_  
_𑀳𑁂𑀫𑀦𑁆𑀢𑁂 𑀲𑀁𑀖𑁄 𑀲𑀁𑀖𑀲𑁆𑀲,_  
_𑀙𑀸𑀤𑀦𑀸 𑀘 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑁇_  


1477\. _𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸 𑀯𑀺𑀯𑀝𑀸 𑀘,_  
_𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀬𑀦𑀸;_  
_𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀯𑀸𑀲𑀁,_  
_𑀫𑀸𑀦𑀢𑁆𑀢𑀸 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀸𑁇_  


1478\. _𑀅𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑁄 𑀘 𑀲𑀻𑀫𑀸𑀬,_  
_𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀧𑀼𑀦𑀸𑀧𑀭𑁂;_  
_𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀅𑀧𑀭𑁂 𑀘𑁂𑀯,_  
_𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸 𑀤𑀼𑀯𑁂𑁇_  


1479\. _𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀸 𑀦𑀺𑀬𑀲𑁆𑀲𑀸 𑀘,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻;_  
_𑀅𑀤𑀲𑁆𑀲𑀦𑀸 𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂,_  
_𑀅𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸𑁇_  


1480\. _𑀆𑀕𑀸𑀍𑀳𑀓𑀫𑁆𑀫𑀸𑀥𑀺𑀲𑀻𑀮𑁂,_  
_𑀤𑀯𑀸𑀦𑀸𑀘𑀸𑀭𑀖𑀸𑀢𑀺𑀓𑀸;_  
_𑀆𑀚𑀻𑀯𑀸𑀧𑀦𑁆𑀦𑀸 𑀢𑀸𑀤𑀺𑀲𑀺𑀓𑀸,_  
_𑀅𑀯𑀡𑁆𑀡𑀼𑀧𑁄𑀲𑀣𑁂𑀦 𑀘𑁇_  


1481\. _𑀧𑀯𑀸𑀭𑀡𑀸 𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀘,_  
_𑀯𑁄𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑁂𑀓𑁂𑀦 𑀘;_  
_𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀑𑀓𑀸𑀲𑀁 𑀦 𑀓𑀭𑁂 𑀢𑀣𑀸𑁇_  


1482\. _𑀦 𑀓𑀭𑁂 𑀲𑀯𑀘𑀦𑀻𑀬𑀁,_  
_𑀦 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀓𑀸 𑀤𑀼𑀯𑁂;_  
_𑀦 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂 𑀤𑀼𑀯𑁂 𑀘𑁂𑀯,_  
_𑀅𑀦𑀼𑀬𑁄𑀕𑀫𑁆𑀧𑀺 𑀦𑁄 𑀤𑀤𑁂𑁇_  


1483\. _𑀲𑀸𑀓𑀘𑁆𑀙𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸,_  
_𑀦𑀺𑀲𑁆𑀲𑀬𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀘;_  
_𑀉𑀧𑁄𑀲𑀣𑀢𑀺𑀓𑀸 𑀢𑀻𑀡𑀺,_  
_𑀧𑀯𑀸𑀭𑀡𑀢𑀺𑀓𑀸 𑀢𑀬𑁄𑁇_  


1484\. _𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀅𑀓𑀼𑀲𑀮𑀸,_  
_𑀓𑀼𑀲𑀮𑀸 𑀘𑀭𑀺𑀢𑀸 𑀤𑀼𑀯𑁂;_  
_𑀢𑀺𑀓𑀪𑁄𑀚𑀦𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂,_  
_𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀧𑀸𑀤𑀼𑀓𑁂𑀦 𑀘;_  
_𑀧𑀸𑀤𑀖𑀁𑀲𑀦𑀺𑀓𑀸 𑀘𑁂𑀯,_  
_𑀉𑀤𑁆𑀤𑀸𑀦𑀁 𑀢𑀺𑀓𑀓𑁂 𑀇𑀤𑀦𑁆𑀢𑀺𑁇_
