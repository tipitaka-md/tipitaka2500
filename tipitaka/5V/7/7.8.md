# 𑁭𑁇𑁮 𑀅𑀝𑁆𑀞𑀓𑀯𑀸𑀭

### 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

### 𑀅𑀝𑁆𑀞𑀓𑀯𑀸𑀭

(328.)

1552\. 𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁂𑀦 𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑁂𑀦 𑀧𑀭𑁂𑀲𑀫𑁆𑀧𑀺 𑀲𑀤𑁆𑀥𑀸𑀬 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀝𑁆𑀞 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸𑁇 𑀅𑀝𑁆𑀞𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀓𑀼𑀮𑀸𑀦𑀺 𑀤𑀽𑀲𑁂𑀢𑀺𑁇 𑀅𑀝𑁆𑀞 𑀫𑀸𑀢𑀺𑀓𑀸 𑀘𑀻𑀯𑀭𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀸𑀬𑁇 𑀅𑀝𑁆𑀞 𑀫𑀸𑀢𑀺𑀓𑀸 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀉𑀩𑁆𑀪𑀸𑀭𑀸𑀬𑁇 𑀅𑀝𑁆𑀞 𑀧𑀸𑀦𑀸𑀦𑀺𑁇 𑀅𑀝𑁆𑀞𑀳𑀺 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀪𑀺𑀪𑀽𑀢𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄 𑀤𑁂𑀯𑀤𑀢𑁆𑀢𑁄 𑀆𑀧𑀸𑀬𑀺𑀓𑁄 𑀦𑁂𑀭𑀬𑀺𑀓𑁄 𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑁄 𑀅𑀢𑁂𑀓𑀺𑀘𑁆𑀙𑁄𑁇 𑀅𑀝𑁆𑀞 𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀝𑁆𑀞 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀝𑁆𑀞 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑁇 𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄𑁇 𑀅𑀝𑁆𑀞 𑀉𑀧𑁄𑀲𑀣𑀗𑁆𑀕𑀸𑀦𑀺𑁇 𑀅𑀝𑁆𑀞 𑀤𑀽𑀢𑁂𑀬𑁆𑀬𑀗𑁆𑀕𑀸𑀦𑀺𑁇 𑀅𑀝𑁆𑀞 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀯𑀢𑁆𑀢𑀸𑀦𑀺𑁇 𑀅𑀝𑁆𑀞 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀥𑀫𑁆𑀫𑀸 𑀫𑀳𑀸𑀲𑀫𑀼𑀤𑁆𑀤𑁂𑁇 𑀅𑀝𑁆𑀞 𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀪𑀼𑀢𑀥𑀫𑁆𑀫𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂𑁇 𑀅𑀝𑁆𑀞 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀸𑁇 𑀅𑀝𑁆𑀞 𑀅𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀸𑁇 𑀅𑀝𑁆𑀞𑀫𑁂 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀝𑁆𑀞 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑁇 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀻 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑀺𑀬𑀸 𑀤𑁂𑀲𑀺𑀢𑀸𑀧𑀺 𑀅𑀤𑁂𑀲𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀝𑁆𑀞𑀯𑀸𑀘𑀺𑀓𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀆𑀲𑀦𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀅𑀝𑁆𑀞 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀢𑀺𑁇 𑀅𑀝𑁆𑀞𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀯𑀺𑀦𑀬𑀥𑀭𑁂𑁇 𑀅𑀝𑁆𑀞 𑀧𑀭𑀫𑀸𑀦𑀺𑁇 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀓𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀝𑁆𑀞𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀝𑁆𑀞 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺 𑀅𑀝𑁆𑀞 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀻𑀢𑀺𑁇

1553\. 𑀅𑀝𑁆𑀞𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

1554\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1555\. _𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑁂𑀲𑀫𑁆𑀧𑀺,_  
_𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀤𑀽𑀲𑀦𑀸;_  
_𑀫𑀸𑀢𑀺𑀓𑀸 𑀓𑀣𑀺𑀦𑀼𑀩𑁆𑀪𑀸𑀭𑀸,_  
_𑀧𑀸𑀦𑀸 𑀅𑀪𑀺𑀪𑀽𑀢𑁂𑀦 𑀘𑁇_  


1556\. _𑀮𑁄𑀓𑀥𑀫𑁆𑀫𑀸 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀸,_  
_𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸 𑀫𑀼𑀲𑀸;_  
_𑀉𑀧𑁄𑀲𑀣𑀸 𑀘 𑀤𑀽𑀢𑀗𑁆𑀕𑀸,_  
_𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁂𑀧𑀺 𑀘𑁇_  


1557\. _𑀅𑀩𑁆𑀪𑀼𑀢𑀸 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀁,_  
_𑀅𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁;_  
_𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀝𑁆𑀞𑀫𑀁 𑀯𑀢𑁆𑀣𑀼,_  
_𑀅𑀤𑁂𑀲𑀺𑀢𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑁇_  


1558\. _𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀲𑀦𑀜𑁆𑀘𑁂𑀯,_  
_𑀯𑀭𑀁 𑀑𑀯𑀸𑀤𑀓𑁂𑀦 𑀘;_  
_𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀭𑀫𑀸𑀦𑀺,_  
_𑀅𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀯𑀢𑁆𑀢𑀦𑀸;_  
_𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀘,_  
_𑀅𑀝𑁆𑀞𑀓𑀸 𑀲𑀼𑀧𑁆𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑀢𑀺𑁇_
