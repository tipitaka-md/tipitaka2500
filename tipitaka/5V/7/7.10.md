# 𑁭𑁇𑁧𑁦 𑀤𑀲𑀓𑀯𑀸𑀭

### 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

### 𑀤𑀲𑀓𑀯𑀸𑀭

(330.)

1565\. 𑀤𑀲 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀤𑀲 𑀆𑀖𑀸𑀢𑀧𑀝𑀺𑀯𑀺𑀦𑀬𑀸𑁇 𑀤𑀲 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀤𑀲𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀤𑀲𑀯𑀢𑁆𑀣𑀼𑀓𑀸 𑀲𑀫𑁆𑀫𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀤𑀲 𑀅𑀦𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀺𑀓𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑁇 𑀤𑀲 𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸𑁇 𑀤𑀲 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀸𑁇 𑀤𑀲 𑀅𑀓𑀼𑀲𑀮𑀓𑀫𑁆𑀫𑀧𑀣𑀸𑁇 𑀤𑀲 𑀓𑀼𑀲𑀮𑀓𑀫𑁆𑀫𑀧𑀣𑀸𑁇 𑀤𑀲 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀲𑀮𑀸𑀓𑀕𑁆𑀕𑀸𑀳𑀸𑁇 𑀤𑀲 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀲𑀮𑀸𑀓𑀕𑁆𑀕𑀸𑀳𑀸𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑀸𑀦𑀁 𑀤𑀲 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺𑁇 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1566\. 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀩𑀸𑀮𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀦 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀦 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀅𑀧𑁆𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀽𑀮𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀤𑀬𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1567\. 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀺, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀽𑀮𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀤𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1568\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀩𑀸𑀮𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀫𑀽𑀮𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀤𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀺𑀯𑀘𑀦𑀧𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1569\. 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀫𑀽𑀮𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀼𑀤𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀧𑀝𑀺𑀧𑀤𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀤𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀺𑀯𑀘𑀦𑀧𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1570\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀩𑀸𑀮𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀜𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀧𑀼𑀩𑁆𑀩𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀅𑀧𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀓𑀸𑀮𑀜𑁆𑀜𑀽 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀘𑀭𑀺𑀬𑀧𑀭𑀫𑁆𑀧𑀭𑀸 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀦 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀦 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀸 𑀦 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀸𑁇

1571\. 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀜𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀮𑀜𑁆𑀜𑀽 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀘𑀭𑀺𑀬𑀧𑀭𑀫𑁆𑀧𑀭𑀸 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀸 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀸𑁇

1572\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀩𑀸𑀮𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀪𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀦 𑀲𑁆𑀯𑀸𑀕𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀦 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀦 𑀲𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀻𑀦𑀺 𑀦 𑀲𑀼𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑀲𑁄 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑁄, 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀘 𑀦 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺𑁇

1573\. 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀪𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑁆𑀯𑀸𑀕𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀻𑀦𑀺 𑀲𑀼𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑀲𑁄 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑁄, 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀘 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺𑁇

1574\. 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁𑁇 𑀤𑀲 𑀆𑀤𑀻𑀦𑀯𑀸 𑀭𑀸𑀚𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀧𑁆𑀧𑀯𑁂𑀲𑀦𑁂𑁇 𑀤𑀲 𑀤𑀸𑀦𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀤𑀲 𑀭𑀢𑀦𑀸𑀦𑀺𑁇 𑀤𑀲𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄𑁇 𑀤𑀲𑀯𑀕𑁆𑀕𑁂𑀦 𑀕𑀡𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀤𑀲 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺𑁇 𑀤𑀲 𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑀸𑁇 𑀤𑀲𑀸𑀳𑀧𑀭𑀫𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀤𑀲 𑀲𑀼𑀓𑁆𑀓𑀸𑀦𑀺𑁇 𑀤𑀲 𑀇𑀢𑁆𑀣𑀺𑀬𑁄𑁇 𑀤𑀲 𑀪𑀭𑀺𑀬𑀸𑀬𑁄𑁇 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀸 𑀤𑀲 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀤𑀻𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀤𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀤𑀲 𑀅𑀓𑁆𑀓𑁄𑀲𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀤𑀲𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑀁 𑀉𑀧𑀲𑀁𑀳𑀭𑀢𑀺𑁇 𑀤𑀲 𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺𑁇 𑀤𑀲 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀺𑀁𑀲𑀼𑁇 𑀤𑀲 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺𑁇 𑀤𑀲 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺𑁇 𑀤𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀬𑀸𑀕𑀼𑀬𑀸𑁇 𑀤𑀲 𑀫𑀁𑀲𑀸 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑁇 𑀤𑀲 𑀧𑀭𑀫𑀸𑀦𑀺𑁇 𑀤𑀲𑀯𑀲𑁆𑀲𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀤𑀲𑀯𑀲𑁆𑀲𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀩𑀮𑀸𑀬 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑀲𑀯𑀲𑁆𑀲𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑀬 𑀧𑀝𑀺𑀩𑀮𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑀦𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀲𑀸𑀤𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑀲𑀯𑀲𑁆𑀲𑀸𑀬 𑀕𑀺𑀳𑀺𑀕𑀢𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸𑀢𑀺𑁇

1576\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1577\. _𑀆𑀖𑀸𑀢𑀁 𑀯𑀺𑀦𑀬𑀁 𑀯𑀢𑁆𑀣𑀼,_  
_𑀫𑀺𑀘𑁆𑀙𑀸 𑀲𑀫𑁆𑀫𑀸 𑀘 𑀅𑀦𑁆𑀢𑀕𑀸;_  
_𑀫𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀸 𑀘𑁂𑀯 𑀲𑀫𑁆𑀫𑀢𑁆𑀢𑀸,_  
_𑀅𑀓𑀼𑀲𑀮𑀸 𑀓𑀼𑀲𑀮𑀸𑀧𑀺 𑀘𑁇_  


1578\. _𑀲𑀮𑀸𑀓𑀸 𑀅𑀥𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀸,_  
_𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀘 𑀦𑀸𑀲𑀦𑀸;_  
_𑀪𑀸𑀲𑀸𑀥𑀺𑀓𑀭𑀡𑀜𑁆𑀘𑁂𑀯,_  
_𑀜𑀢𑁆𑀢𑀺𑀮𑀳𑀼𑀓𑀫𑁂𑀯 𑀘𑁇_  


1579\. _𑀮𑀳𑀼𑀓𑀸 𑀕𑀭𑀼𑀓𑀸 𑀏𑀢𑁂,_  
_𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀸 𑀯𑀺𑀚𑀸𑀦𑀣;_  
_𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸 𑀘 𑀲𑀺𑀓𑁆𑀔𑀸 𑀘,_  
_𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀘 𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇_  


1580\. _𑀭𑀢𑀦𑀁 𑀤𑀲𑀯𑀕𑁆𑀕𑁄 𑀘,_  
_𑀢𑀣𑁂𑀯 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀥𑀸𑀭𑀡𑀸 𑀘,_  
_𑀤𑀲𑀸𑀳𑀲𑀼𑀓𑁆𑀓𑁃𑀢𑁆𑀣𑀺𑀬𑁄𑁇_  


1581\. _𑀪𑀭𑀺𑀬𑀸 𑀤𑀲 𑀯𑀢𑁆𑀣𑀽𑀦𑀺,_  
_𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀓𑁆𑀓𑁄𑀲𑁂𑀦 𑀘;_  
_𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑀜𑁆𑀘𑁂𑀯 𑀲𑁂𑀦𑀸𑀦𑀺,_  
_𑀯𑀭𑀸𑀦𑀺 𑀘 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑁇_  


1582\. _𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀬𑀸𑀕𑀼𑀫𑀁𑀲𑀸 𑀘,_
