# 𑁭𑁇𑁫 𑀧𑀜𑁆𑀘𑀓𑀯𑀸𑀭

### 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

### 𑀧𑀜𑁆𑀘𑀓𑀯𑀸𑀭

(325.)

1502\. 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 𑀧𑀜𑁆𑀘 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀧𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀆𑀦𑀦𑁆𑀢𑀭𑀺𑀓𑀸𑀦𑀺𑁇 𑀧𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦𑀺𑀬𑀢𑀸𑁇 𑀧𑀜𑁆𑀘 𑀙𑁂𑀤𑀦𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑀁 𑀦 𑀉𑀧𑁂𑀢𑀺—  𑀲𑀬𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀦 𑀓𑀭𑁄𑀢𑀺, 𑀧𑀭𑀁 𑀯𑀸 𑀦 𑀅𑀚𑁆𑀛𑁂𑀲𑀢𑀺, 𑀙𑀦𑁆𑀤𑀁 𑀯𑀸 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀸 𑀦 𑀤𑁂𑀢𑀺, 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀓𑀫𑁆𑀫𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺, 𑀓𑀢𑁂 𑀯𑀸 𑀧𑀦 𑀓𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑁂𑀢𑀺—  𑀲𑀬𑀁 𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀧𑀭𑀁 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀲𑀢𑀺, 𑀙𑀦𑁆𑀤𑀁 𑀯𑀸 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀯𑀸 𑀤𑁂𑀢𑀺, 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑁂 𑀓𑀫𑁆𑀫𑁂 𑀦𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺, 𑀓𑀢𑁂 𑀯𑀸 𑀧𑀦 𑀓𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀤𑀺𑀝𑁆𑀞𑀺 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀧𑁆𑀧𑀦𑁆𑀢𑀺—  𑀅𑀦𑀸𑀫𑀦𑁆𑀢𑀘𑀸𑀭𑁄, 𑀕𑀡𑀪𑁄𑀚𑀦𑀁, 𑀧𑀭𑀫𑁆𑀧𑀭𑀪𑁄𑀚𑀦𑀁, 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀁, 𑀅𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀸𑁇 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀲𑁆𑀲𑀗𑁆𑀓𑀺𑀢𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀺𑀢𑁄 𑀳𑁄𑀢𑀺—  𑀧𑀸𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀺 𑀅𑀓𑀼𑀧𑁆𑀧𑀥𑀫𑁆𑀫𑁄𑀧𑀺 𑀯𑁂𑀲𑀺𑀬𑀸𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀥𑀯𑀸𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀣𑀼𑀮𑁆𑀮𑀓𑀼𑀫𑀸𑀭𑀺𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀧𑀡𑁆𑀟𑀓𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀕𑁄𑀘𑀭𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀢𑁂𑀮𑀸𑀦𑀺—  𑀢𑀺𑀮𑀢𑁂𑀮𑀁, 𑀲𑀸𑀲𑀧𑀢𑁂𑀮𑀁, 𑀫𑀥𑀼𑀓𑀢𑁂𑀮𑀁, 𑀏𑀭𑀡𑁆𑀟𑀓𑀢𑁂𑀮𑀁, 𑀯𑀲𑀸𑀢𑁂𑀮𑀁𑁇 𑀧𑀜𑁆𑀘 𑀯𑀲𑀸𑀦𑀺—  𑀅𑀘𑁆𑀙𑀯𑀲𑀁, 𑀫𑀘𑁆𑀙𑀯𑀲𑀁, 𑀲𑀼𑀲𑀼𑀓𑀸𑀯𑀲𑀁, 𑀲𑀽𑀓𑀭𑀯𑀲𑀁, 𑀕𑀤𑁆𑀭𑀪𑀯𑀲𑀁𑁇 𑀧𑀜𑁆𑀘 𑀩𑁆𑀬𑀲𑀦𑀸𑀦𑀺—  𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑀁, 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀁, 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀁, 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑀁, 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑀁𑁇 𑀧𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀤𑀸—  𑀜𑀸𑀢𑀺𑀲𑀫𑁆𑀧𑀤𑀸, 𑀪𑁄𑀕𑀲𑀫𑁆𑀧𑀤𑀸, 𑀅𑀭𑁄𑀕𑀲𑀫𑁆𑀧𑀤𑀸, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀤𑀸𑁇 𑀧𑀜𑁆𑀘 𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑁄 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀸—  𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑁄 𑀯𑀸, 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀯𑀸, 𑀧𑀓𑁆𑀔𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑁄 𑀯𑀸, 𑀆𑀡𑀢𑁆𑀢𑀺𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀫𑀻𑁇 𑀧𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸—  𑀅𑀤𑁆𑀥𑀸𑀦𑀳𑀻𑀦𑁄, 𑀅𑀗𑁆𑀕𑀳𑀻𑀦𑁄, 𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀧𑀦𑁆𑀦𑁄, 𑀓𑀭𑀡𑀤𑀼𑀓𑁆𑀓𑀝𑀓𑁄, 𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄𑁇 𑀧𑀜𑁆𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺—  𑀲𑁄𑀲𑀸𑀦𑀺𑀓𑀁, 𑀧𑀸𑀧𑀡𑀺𑀓𑀁, 𑀉𑀦𑁆𑀤𑀽𑀭𑀓𑁆𑀔𑀸𑀬𑀺𑀓𑀁, 𑀉𑀧𑀘𑀺𑀓𑀓𑁆𑀔𑀸𑀬𑀺𑀓𑀁, 𑀅𑀕𑁆𑀕𑀺𑀤𑀟𑁆𑀠𑀁𑁇 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀧𑀜𑁆𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺—  𑀕𑁄𑀔𑀸𑀬𑀺𑀓𑀁, 𑀅𑀚𑀓𑁆𑀔𑀸𑀬𑀺𑀓𑀁, 𑀣𑀽𑀧𑀘𑀻𑀯𑀭𑀁, 𑀆𑀪𑀺𑀲𑁂𑀓𑀺𑀓𑀁, 𑀕𑀢𑀧𑀝𑀺𑀬𑀸𑀕𑀢𑀁𑁇 𑀧𑀜𑁆𑀘 𑀅𑀯𑀳𑀸𑀭𑀸—  𑀣𑁂𑀬𑁆𑀬𑀸𑀯𑀳𑀸𑀭𑁄, 𑀧𑀲𑀬𑁆𑀳𑀸𑀯𑀳𑀸𑀭𑁄, 𑀧𑀭𑀺𑀓𑀧𑁆𑀧𑀸𑀯𑀳𑀸𑀭𑁄, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀸𑀯𑀳𑀸𑀭𑁄, 𑀓𑀼𑀲𑀸𑀯𑀳𑀸𑀭𑁄𑁇 𑀧𑀜𑁆𑀘 𑀫𑀳𑀸𑀘𑁄𑀭𑀸 𑀲𑀦𑁆𑀢𑁄 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀸 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁𑁇 𑀧𑀜𑁆𑀘 𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑀺𑀬𑀸𑀦𑀺𑁇 𑀧𑀜𑁆𑀘 𑀅𑀯𑁂𑀪𑀗𑁆𑀕𑀺𑀬𑀸𑀦𑀺𑁇 𑀧𑀜𑁆𑀘𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀓𑀸𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄𑁇 𑀧𑀜𑁆𑀘𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄𑁇 𑀧𑀜𑁆𑀘𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀤𑁂𑀲𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑁄𑁇 𑀧𑀜𑁆𑀘 𑀲𑀁𑀖𑀸𑁇 𑀧𑀜𑁆𑀘 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑁇 𑀲𑀩𑁆𑀩𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀯𑀺𑀦𑀬𑀥𑀭𑀧𑀜𑁆𑀘𑀫𑁂𑀦 𑀕𑀡𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀸𑀭𑁂𑁇 𑀧𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀸𑀦𑀺𑁇 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑁂 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲𑁇 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁—  𑀫𑀚𑁆𑀚𑀤𑀸𑀦𑀁, 𑀲𑀫𑀚𑁆𑀚𑀤𑀸𑀦𑀁, 𑀇𑀢𑁆𑀣𑀺𑀤𑀸𑀦𑀁, 𑀉𑀲𑀪𑀤𑀸𑀦𑀁, 𑀘𑀺𑀢𑁆𑀢𑀓𑀫𑁆𑀫𑀤𑀸𑀦𑀁𑁇 𑀧𑀜𑁆𑀘 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑀬𑀸—  𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀭𑀸𑀕𑁄 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑀬𑁄, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀤𑁄𑀲𑁄 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑀬𑁄, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀫𑁄𑀳𑁄 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑀬𑁄, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀝𑀺𑀪𑀸𑀦𑀁 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑀬𑀁, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀕𑀫𑀺𑀬𑀘𑀺𑀢𑁆𑀢𑀁 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀦𑁄𑀤𑀬𑀁𑁇 𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀺𑀬𑀸—  𑀲𑀓𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀢𑀺, 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀓𑀫𑁆𑀫𑀁 𑀉𑀧𑀘𑀺𑀦𑀢𑀺, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀺𑀬𑀸—  𑀲𑀓𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀢𑀺, 𑀧𑀭𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀲𑀻𑀤𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1503\. 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀩𑀸𑀮𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀦 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀦 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀦 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀦 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬𑁇 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀲𑁆𑀲 𑀪𑀸𑀲𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀓𑀸𑀭𑁂𑀢𑀺 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀩𑀸𑀮𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀫𑀽𑀮𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀤𑀬𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀁 𑀉𑀪𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦𑀲𑁆𑀲 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀲𑁆𑀯𑀸𑀕𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺 𑀲𑀼𑀯𑀺𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀼𑀧𑁆𑀧𑀯𑀢𑁆𑀢𑀻𑀦𑀺 𑀲𑀼𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀢𑀸𑀦𑀺 𑀲𑀼𑀢𑁆𑀢𑀲𑁄 𑀅𑀦𑀼𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀲𑁄𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀩𑀸𑀮𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀘 𑀦 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 “𑀧𑀡𑁆𑀟𑀺𑀢𑁄” 𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺—  𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑁂 𑀘 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺𑁇

1504\. 𑀧𑀜𑁆𑀘 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑀸—  𑀫𑀦𑁆𑀤𑀢𑁆𑀢𑀸 𑀫𑁄𑀫𑀽𑀳𑀢𑁆𑀢𑀸 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀫𑁆𑀫𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑁂𑀧𑀸 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, “𑀯𑀡𑁆𑀡𑀺𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑁂𑀳𑀻”𑀢𑀺 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀮𑁆𑀮𑁂𑀔𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀇𑀤𑀫𑀢𑁆𑀣𑀺𑀢𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀲𑁄𑀲𑀸𑀦𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀢𑁂𑀘𑀻𑀯𑀭𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀲𑀧𑀤𑀸𑀦𑀘𑀸𑀭𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀦𑁂𑀲𑀚𑁆𑀚𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀬𑀣𑀸𑀲𑀦𑁆𑀣𑀢𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀏𑀓𑀸𑀲𑀦𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀔𑀮𑀼𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀺𑀓𑀸…𑀧𑁂…  𑀧𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀸—  𑀫𑀦𑁆𑀤𑀢𑁆𑀢𑀸 𑀫𑁄𑀫𑀽𑀳𑀢𑁆𑀢𑀸 𑀧𑀢𑁆𑀢𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀧𑀢𑁆𑀢𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀉𑀫𑁆𑀫𑀸𑀤𑀸 𑀘𑀺𑀢𑁆𑀢𑀓𑁆𑀔𑁂𑀧𑀸 𑀧𑀢𑁆𑀢𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, “𑀯𑀡𑁆𑀡𑀺𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀲𑀸𑀯𑀓𑁂𑀳𑀻”𑀢𑀺 𑀧𑀢𑁆𑀢𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀺 𑀘 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀲𑀮𑁆𑀮𑁂𑀔𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀯𑀺𑀯𑁂𑀓𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀇𑀤𑀫𑀢𑁆𑀣𑀺𑀢𑀜𑁆𑀜𑁂𑀯 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀢𑁆𑀢𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

1505\. 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁—  𑀉𑀧𑁄𑀲𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁—  𑀉𑀧𑁄𑀲𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸𑁇 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁—  𑀉𑀧𑁄𑀲𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑀸 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁—  𑀉𑀧𑁄𑀲𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁—  𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑀸 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑀸 𑀯𑀸𑁇

1506\. 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂—  𑀅𑀢𑁆𑀢𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀯𑀤𑀢𑀺, 𑀅𑀦𑀼𑀯𑀺𑀘𑁆𑀘𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀽 𑀕𑀭𑀳𑀦𑁆𑀢𑀺, 𑀧𑀸𑀧𑀓𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀅𑀧𑀸𑀬𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁 𑀯𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀭𑀬𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂—  𑀅𑀢𑁆𑀢𑀸𑀧𑀺 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀉𑀧𑀯𑀤𑀢𑀺, 𑀅𑀦𑀼𑀯𑀺𑀘𑁆𑀘𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀽 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺, 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀧𑀭𑀁 𑀫𑀭𑀡𑀸 𑀲𑀼𑀕𑀢𑀺𑀁 𑀲𑀕𑁆𑀕𑀁 𑀮𑁄𑀓𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂—  𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀦 𑀧𑀲𑀻𑀤𑀦𑁆𑀢𑀺, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀏𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀜𑁆𑀜𑀣𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁 𑀅𑀓𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀦𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀦 𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂—  𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀧𑀲𑀻𑀤𑀦𑁆𑀢𑀺, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀳𑁄𑀢𑀺, 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀧𑀘𑁆𑀙𑀺𑀫𑀸 𑀚𑀦𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀕𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀘𑀺𑀢𑁆𑀢𑀫𑀲𑁆𑀲 𑀧𑀲𑀻𑀤𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀓𑀼𑀮𑀽𑀧𑀓𑁂—  𑀅𑀦𑀸𑀫𑀦𑁆𑀢𑀘𑀸𑀭𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀭𑀳𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀸𑀬 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑁂 𑀆𑀲𑀦𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀉𑀢𑁆𑀢𑀭𑀺𑀙𑀧𑁆𑀧𑀜𑁆𑀘𑀯𑀸𑀘𑀸𑀳𑀺 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀓𑀸𑀫𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀩𑀳𑀼𑀮𑁄 𑀘 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀓𑀼𑀮𑀽𑀧𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄—  𑀅𑀢𑀺𑀯𑁂𑀮𑀁 𑀓𑀼𑀮𑁂𑀲𑀼 𑀲𑀁𑀲𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑀺𑀳𑀭𑀢𑁄 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀅𑀪𑀺𑀡𑁆𑀳𑀤𑀲𑁆𑀲𑀦𑀁, 𑀤𑀲𑁆𑀲𑀦𑁂 𑀲𑀢𑀺 𑀲𑀁𑀲𑀕𑁆𑀕𑁄, 𑀲𑀁𑀲𑀕𑁆𑀕𑁂 𑀲𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑁄, 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑁂 𑀲𑀢𑀺 𑀑𑀢𑀸𑀭𑁄, 𑀑𑀢𑀺𑀡𑁆𑀡𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲𑁂𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀁 𑀅𑀦𑀪𑀺𑀭𑀢𑁄 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀯𑀸 𑀲𑀁𑀓𑀺𑀮𑀺𑀝𑁆𑀞𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀺𑀓𑁆𑀔𑀁 𑀯𑀸 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀢𑀺𑁇

1507\. 𑀧𑀜𑁆𑀘 𑀩𑀻𑀚𑀚𑀸𑀢𑀸𑀦𑀺—  𑀫𑀽𑀮𑀩𑀻𑀚𑀁, 𑀔𑀦𑁆𑀥𑀩𑀻𑀚𑀁, 𑀨𑀍𑀉𑀩𑀻𑀚𑀁, 𑀅𑀕𑁆𑀕𑀩𑀻𑀚𑀁, 𑀩𑀻𑀚𑀩𑀻𑀚𑀜𑁆𑀜𑁂𑀯 𑀧𑀜𑁆𑀘𑀫𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀺 𑀲𑀫𑀡𑀓𑀧𑁆𑀧𑁂𑀳𑀺 𑀨𑀮𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁—  𑀅𑀕𑁆𑀕𑀺𑀧𑀭𑀺𑀚𑀺𑀢𑀁, 𑀲𑀢𑁆𑀣𑀧𑀭𑀺𑀚𑀺𑀢𑀁, 𑀦𑀔𑀧𑀭𑀺𑀚𑀺𑀢𑀁, 𑀅𑀩𑀻𑀚𑀁, 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀩𑀻𑀚𑀜𑁆𑀜𑁂𑀯 𑀧𑀜𑁆𑀘𑀫𑀁𑁇 𑀧𑀜𑁆𑀘 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑁄—  𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀲𑀼𑀢𑁂𑀦 𑀲𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀬𑀁 𑀧𑀞𑀫𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺; 𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀲𑀼𑀢𑁂𑀦 𑀲𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀬𑀁 𑀤𑀼𑀢𑀺𑀬𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺; 𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀭𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀲𑀼𑀢𑁂𑀦 𑀲𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀬𑀁 𑀢𑀢𑀺𑀬𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺; 𑀦𑀺𑀤𑀸𑀦𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀭𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀅𑀦𑀺𑀬𑀢𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀁 𑀲𑀼𑀢𑁂𑀦 𑀲𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀬𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀸 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺; 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦𑁂𑀯 𑀧𑀜𑁆𑀘𑀫𑀻𑁇 𑀅𑀧𑀭𑀸𑀧𑀺 𑀧𑀜𑁆𑀘 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑁄—  𑀲𑀼𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑁄, 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀉𑀧𑁄𑀲𑀣𑁄, 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀧𑁄𑀲𑀣𑁄, 𑀧𑀯𑀸𑀭𑀡𑀸, 𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑀉𑀧𑁄𑀲𑀣𑁄𑀬𑁂𑀯 𑀧𑀜𑁆𑀘𑀫𑁄𑁇 𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀯𑀺𑀦𑀬𑀥𑀭𑁂—  𑀅𑀢𑁆𑀢𑀦𑁄 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀼𑀕𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀭𑀓𑁆𑀔𑀺𑀢𑁄, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀸𑀦𑀁 𑀧𑀝𑀺𑀲𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀯𑁄𑀳𑀭𑀢𑀺, 𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑁂 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀧𑀜𑁆𑀘 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺𑁇 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀻𑀢𑀺𑁇

_𑀯𑀺𑀦𑀻𑀢𑀸𑀦𑀦𑁆𑀢𑀭𑁂𑀦 𑀘;_  
_𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀙𑁂𑀤𑀦𑀸 𑀘𑁂𑀯,_  
_𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘 𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


1511\. _𑀦 𑀉𑀧𑁂𑀢𑀺 𑀉𑀧𑁂𑀢𑀺 𑀘,_  
_𑀓𑀧𑁆𑀧𑀦𑁆𑀢𑀼𑀲𑀗𑁆𑀓𑀺𑀢𑁂𑀮𑀜𑁆𑀘;_  
_𑀯𑀲𑀁 𑀩𑁆𑀬𑀲𑀦𑀸 𑀲𑀫𑁆𑀧𑀤𑀸,_  
_𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁂𑀦 𑀘𑁇_  


1512\. _𑀲𑁄𑀲𑀸𑀦𑀺𑀓𑀁 𑀔𑀸𑀬𑀺𑀢𑀜𑁆𑀘,_  
_𑀣𑁂𑀬𑁆𑀬𑀁 𑀘𑁄𑀭𑁄 𑀘 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺 𑀅𑀯𑁂𑀪𑀗𑁆𑀕𑀺,_  
_𑀓𑀸𑀬𑀢𑁄 𑀓𑀸𑀬𑀯𑀸𑀘𑀢𑁄𑁇_  


1513\. _𑀤𑁂𑀲𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑁂𑀲𑀁,_  
_𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀓𑀣𑀺𑀦𑁂𑀦 𑀘;_  
_𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁,_  
_𑀧𑀸𑀭𑀸𑀚𑀺𑀣𑀼𑀮𑁆𑀮𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇_  


1514\. _𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀓𑀧𑁆𑀧𑀺𑀬𑀜𑁆𑀘,_  
_𑀅𑀧𑀼𑀜𑁆𑀜𑀤𑀼𑀯𑀺𑀦𑁄𑀤𑀬𑀸;_  
_𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀦𑀻 𑀅𑀧𑀭𑁂 𑀘,_  
_𑀪𑀸𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀫𑁂𑀯 𑀘𑁇_  


1515\. _𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀢𑁆𑀣𑀼 𑀜𑀢𑁆𑀢𑀺,_  
_𑀆𑀧𑀢𑁆𑀢𑀸 𑀉𑀪𑀬𑀸𑀦𑀺 𑀘;_  
_𑀮𑀳𑀼𑀓𑀝𑁆𑀞𑀫𑀓𑀸 𑀏𑀢𑁂,_  
_𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀸 𑀯𑀺𑀚𑀸𑀦𑀣𑁇_  


1516\. _𑀅𑀭𑀜𑁆𑀜𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀧𑀁𑀲𑀼𑀭𑀼𑀓𑁆𑀔𑀲𑀼𑀲𑀸𑀦𑀺𑀓𑀸;_  
_𑀅𑀩𑁆𑀪𑁄𑀓𑀸𑀲𑁄 𑀘𑀻𑀯𑀭𑀜𑁆𑀘,_  
_𑀲𑀧𑀤𑀸𑀦𑁄 𑀦𑀺𑀲𑀚𑁆𑀚𑀺𑀓𑁄𑁇_  


1517\. _𑀲𑀦𑁆𑀣𑀢𑀺 𑀔𑀮𑀼 𑀧𑀘𑁆𑀙𑀸𑀧𑀺,_  
_𑀧𑀢𑁆𑀢𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀫𑁂𑀯 𑀘;_  
_𑀉𑀧𑁄𑀲𑀣𑀁 𑀧𑀯𑀸𑀭𑀡𑀁,_  
_𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀺 𑀘𑁇_  


1518\. _𑀓𑀡𑁆𑀳𑀲𑀼𑀓𑁆𑀓𑀧𑀤𑀸 𑀏𑀢𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀫𑁆𑀧𑀺 𑀢𑁂 𑀢𑀣𑀸;_  
_𑀅𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀧𑀸𑀲𑀸𑀤𑀺,_  
_𑀢𑀣𑁂𑀯 𑀅𑀧𑀭𑁂 𑀤𑀼𑀯𑁂𑁇_  


1519\. _𑀓𑀼𑀮𑀽𑀧𑀓𑁂 𑀅𑀢𑀺𑀯𑁂𑀮𑀁,_  
_𑀩𑀻𑀚𑀁 𑀲𑀫𑀡𑀓𑀧𑁆𑀧𑀺 𑀘;_  
_𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺 𑀅𑀧𑀭𑁂 𑀘𑁂𑀯,_  
_𑀯𑀺𑀦𑀬𑀸𑀥𑀫𑁆𑀫𑀺𑀓𑁂𑀦 𑀘;_  
_𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀘 𑀢𑀣𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀼𑀤𑁆𑀥𑀺𑀧𑀜𑁆𑀘𑀓𑀸𑀢𑀺𑁇_
