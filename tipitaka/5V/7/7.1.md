# 𑁭𑁇𑁧 𑀏𑀓𑀓𑀯𑀸𑀭

* 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

* 𑀏𑀓𑀓𑀯𑀸𑀭

(321.)

1440\. 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑀭𑀸 𑀥𑀫𑁆𑀫𑀸 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀭𑀸 𑀥𑀫𑁆𑀫𑀸 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑁂𑀲𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀤𑁂𑀲𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀻 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀸𑀯𑀚𑁆𑀚𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀯𑀚𑁆𑀚𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀺𑀭𑀺𑀬𑀸𑀓𑀺𑀭𑀺𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀧𑀭𑀸𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀧𑀭𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑁂𑀲𑀺𑀢𑀸 𑀕𑀡𑀦𑀽𑀧𑀕𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀤𑁂𑀲𑀺𑀢𑀸 𑀦 𑀕𑀡𑀦𑀽𑀧𑀕𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀧𑀤𑁂𑀲𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀏𑀓𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀉𑀪𑀢𑁄𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀣𑀼𑀮𑁆𑀮𑀯𑀚𑁆𑀚𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀣𑀼𑀮𑁆𑀮𑀯𑀚𑁆𑀚𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀕𑀺𑀳𑀺𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦 𑀕𑀺𑀳𑀺𑀧𑀝𑀺𑀲𑀁𑀬𑀼𑀢𑁆𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀬𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀺𑀬𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀆𑀤𑀺𑀓𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀦𑀸𑀤𑀺𑀓𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀥𑀺𑀘𑁆𑀘𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀪𑀺𑀡𑁆𑀳𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀘𑁄𑀤𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀘𑀼𑀤𑀺𑀢𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀥𑀫𑁆𑀫𑀘𑁄𑀤𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀥𑀫𑁆𑀫𑀘𑀼𑀤𑀺𑀢𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀥𑀫𑁆𑀫𑀘𑁄𑀤𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀥𑀫𑁆𑀫𑀘𑀼𑀤𑀺𑀢𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀦𑀺𑀬𑀢𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀦𑀺𑀬𑀢𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀪𑀩𑁆𑀩𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀪𑀩𑁆𑀩𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀦𑀸𑀲𑀺𑀢𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀦𑀸𑀲𑀺𑀢𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀅𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀞𑀧𑀦𑀁 𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺𑁇

1442\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1443\. _𑀓𑀭𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀮𑀳𑀼𑀓𑀸,_  
_𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀘 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸;_  
_𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀘,_  
_𑀅𑀦𑁆𑀢𑀭𑀸 𑀯𑀚𑁆𑀚𑀓𑀺𑀭𑀺𑀬𑀁𑁇_  


1444\. _𑀓𑀺𑀭𑀺𑀬𑀸𑀓𑀺𑀭𑀺𑀬𑀁 𑀧𑀼𑀩𑁆𑀩𑀸,_  
_𑀅𑀦𑁆𑀢𑀭𑀸 𑀕𑀡𑀦𑀽𑀧𑀕𑀸;_  
_𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺 𑀅𑀦𑀸𑀦𑀼𑀧𑁆𑀧𑀦𑁆𑀦,_  
_𑀲𑀩𑁆𑀩𑀲𑀸𑀥𑀸𑀭𑀡𑀸 𑀘 𑀏𑀓𑀢𑁄𑁇_  


1445\. _𑀣𑀼𑀮𑁆𑀮𑀕𑀺𑀳𑀺𑀦𑀺𑀬𑀢𑀸 𑀘,_  
_𑀆𑀤𑀺 𑀅𑀥𑀺𑀘𑁆𑀘𑀘𑁄𑀤𑀓𑁄;_  
_𑀅𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀦𑀺𑀬𑀢𑁄,_  
_𑀅𑀪𑀩𑁆𑀩𑁄𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀦𑀸𑀲𑀺𑀢𑀸;_  
_𑀲𑀫𑀸𑀦𑀁 𑀞𑀧𑀦𑀜𑁆𑀘𑁂𑀯,_  
_𑀉𑀤𑁆𑀤𑀸𑀦𑀁 𑀏𑀓𑀓𑁂 𑀇𑀤𑀦𑁆𑀢𑀺𑁇_
