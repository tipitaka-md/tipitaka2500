# 𑁭𑁇𑁨 𑀤𑀼𑀓𑀯𑀸𑀭

* 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

* 𑀤𑀼𑀓𑀯𑀸𑀭

1446\. 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀲𑀜𑁆𑀜𑀸 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦𑁄 𑀲𑀜𑁆𑀜𑀸𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀮𑀤𑁆𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀲𑁆𑀲, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀮𑀤𑁆𑀥𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀲𑁆𑀲𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀧𑀝𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀲𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀧𑀝𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀭𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀧𑀝𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀝𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑀸, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀭𑀧𑀼𑀕𑁆𑀕𑀮𑀧𑀝𑀺𑀲𑀜𑁆𑀜𑀼𑀢𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀦𑁆𑀢𑁄 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀫𑀼𑀲𑀸 𑀪𑀡𑀦𑁆𑀢𑁄 𑀮𑀳𑀼𑀓𑀁𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀼𑀲𑀸 𑀪𑀡𑀦𑁆𑀢𑁄 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀦𑁆𑀢𑁄 𑀮𑀳𑀼𑀓𑀁𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀪𑀽𑀫𑀺𑀕𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀯𑁂𑀳𑀸𑀲𑀕𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑁂𑀳𑀸𑀲𑀕𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀪𑀽𑀫𑀺𑀕𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁄𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀦𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀲𑀫𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀲𑀫𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀤𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀤𑁂𑀲𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀦 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀦 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀭𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀤𑀺𑀯𑀸𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑀺𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀭𑀢𑁆𑀢𑀺𑀁𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑁂 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀢𑁆𑀣𑀸𑀧𑀢𑁆𑀢𑀺 𑀦 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1447\. 𑀤𑁆𑀯𑁂 𑀉𑀧𑁄𑀲𑀣𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀓𑁄 𑀘 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀓𑁄 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀧𑀯𑀸𑀭𑀡𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀓𑀸 𑀘 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀓𑀸 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀸𑀦𑀺—  𑀅𑀧𑀮𑁄𑀓𑀦𑀓𑀫𑁆𑀫𑀁, 𑀜𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀫𑀁𑁇 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀸𑀦𑀺—  𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑀓𑀫𑁆𑀫𑀁, 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀫𑁆𑀫𑀁𑁇 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀯𑀢𑁆𑀣𑀽𑀦𑀺—  𑀅𑀧𑀮𑁄𑀓𑀦𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀜𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑁇 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀯𑀢𑁆𑀣𑀽𑀦𑀺—  𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼, 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑁇 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀸—  𑀅𑀧𑀮𑁄𑀓𑀦𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀤𑁄𑀲𑁄, 𑀜𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀤𑁄𑀲𑁄𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀸—  𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀤𑁄𑀲𑁄, 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀤𑁄𑀲𑁄𑁇 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄—  𑀅𑀧𑀮𑁄𑀓𑀦𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑁇 𑀅𑀧𑀭𑀸𑀧𑀺 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄—  𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀪𑀽𑀫𑀺𑀬𑁄—  𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀯𑀸 𑀦𑀁 𑀲𑀁𑀖𑁄 𑀉𑀓𑁆𑀔𑀺𑀧𑀢𑀺 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀯𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀯𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀯𑀸𑁇 𑀤𑁆𑀯𑁂 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀪𑀽𑀫𑀺𑀬𑁄—  𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀯𑀸 𑀦𑀁 𑀲𑀁𑀖𑁄 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀑𑀲𑀸𑀭𑁂𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑁂 𑀯𑀸 𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀯𑀸 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀯𑀸𑁇 𑀤𑁆𑀯𑁂 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸—  𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘𑁇 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸, 𑀤𑁆𑀯𑁂 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸, 𑀤𑁆𑀯𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸, 𑀤𑁆𑀯𑁂 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸, 𑀤𑁆𑀯𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀸, 𑀤𑁆𑀯𑁂 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸—  𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘𑁇 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄, 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 𑀤𑁆𑀯𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀲𑀁𑀖𑁄 𑀪𑀺𑀚𑁆𑀚𑀢𑀺—  𑀓𑀫𑁆𑀫𑁂𑀦 𑀯𑀸 𑀲𑀮𑀸𑀓𑀕𑁆𑀕𑀸𑀳𑁂𑀦 𑀯𑀸𑁇

1448\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸—  𑀅𑀤𑁆𑀥𑀸𑀦𑀳𑀻𑀦𑁄, 𑀅𑀗𑁆𑀕𑀳𑀻𑀦𑁄𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸—  𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀧𑀦𑁆𑀦𑁄, 𑀓𑀭𑀡𑀤𑀼𑀓𑁆𑀓𑀝𑀓𑁄𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸—  𑀅𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀧𑀭𑀺𑀧𑀽𑀭𑁄 𑀦𑁄 𑀘 𑀬𑀸𑀘𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁—  𑀅𑀮𑀚𑁆𑀚𑀺𑀲𑁆𑀲 𑀘 𑀩𑀸𑀮𑀲𑁆𑀲 𑀘𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—  𑀅𑀮𑀚𑁆𑀚𑀺𑀲𑁆𑀲 𑀘 𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀘 𑀦 𑀬𑀸𑀘𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄—  𑀩𑀸𑀮𑀲𑁆𑀲 𑀘 𑀮𑀚𑁆𑀚𑀺𑀲𑁆𑀲 𑀘 𑀬𑀸𑀘𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀪𑀩𑁆𑀩𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁—  𑀩𑀼𑀤𑁆𑀥𑀸 𑀘 𑀧𑀘𑁆𑀘𑁂𑀓𑀩𑀼𑀤𑁆𑀥𑀸 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀪𑀩𑁆𑀩𑀸, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁—  𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀪𑀩𑁆𑀩𑀸 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁—  𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘 𑀅𑀭𑀺𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀪𑀩𑁆𑀩𑀸 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀁—  𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀪𑀩𑁆𑀩𑀸 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀘𑀭𑀺𑀢𑀼𑀁—  𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘 𑀅𑀭𑀺𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀪𑀩𑁆𑀩𑀸 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀘𑀭𑀺𑀢𑀼𑀁—  𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀘 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀸𑁇

1449\. 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀸—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀸—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺 𑀏𑀓𑀘𑁆𑀘𑁄 𑀲𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀤𑀼𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀤𑁆𑀯𑁂 𑀑𑀲𑀸𑀭𑀡𑀸—  𑀅𑀢𑁆𑀣𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺 𑀏𑀓𑀘𑁆𑀘𑁄 𑀲𑁄𑀲𑀸𑀭𑀺𑀢𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀜𑁆𑀜𑀸—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀸—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀓𑀸𑀬𑀧𑀝𑀺𑀩𑀤𑁆𑀥𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀸—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀉𑀧𑀖𑀸𑀢𑀺𑀓𑀸—  𑀲𑀺𑀓𑁆𑀔𑀽𑀧𑀖𑀸𑀢𑀺𑀓𑀸 𑀘 𑀪𑁄𑀕𑀽𑀧𑀖𑀸𑀢𑀺𑀓𑀸 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀘𑁄𑀤𑀦𑀸—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀘𑁄𑀤𑁂𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀘𑁄𑀤𑁂𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀧𑀮𑀺𑀩𑁄𑀥𑀸—  𑀆𑀯𑀸𑀲𑀧𑀮𑀺𑀩𑁄𑀥𑁄 𑀘 𑀘𑀻𑀯𑀭𑀧𑀮𑀺𑀩𑁄𑀥𑁄 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀅𑀧𑀮𑀺𑀩𑁄𑀥𑀸—  𑀆𑀯𑀸𑀲𑀅𑀧𑀮𑀺𑀩𑁄𑀥𑁄 𑀘 𑀘𑀻𑀯𑀭𑀅𑀧𑀮𑀺𑀩𑁄𑀥𑁄 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀘𑀻𑀯𑀭𑀸𑀦𑀺—  𑀕𑀳𑀧𑀢𑀺𑀓𑀜𑁆𑀘 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀜𑁆𑀘𑁇 𑀤𑁆𑀯𑁂 𑀧𑀢𑁆𑀢𑀸—  𑀅𑀬𑁄𑀧𑀢𑁆𑀢𑁄 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑁄𑁇 𑀤𑁆𑀯𑁂 𑀫𑀡𑁆𑀟𑀮𑀸𑀦𑀺—  𑀢𑀺𑀧𑀼𑀫𑀬𑀁, 𑀲𑀻𑀲𑀫𑀬𑀁𑁇 𑀤𑁆𑀯𑁂 𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀸—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀘𑀻𑀯𑀭𑀲𑁆𑀲 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀸—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀸—  𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀸 𑀘 𑀧𑀭𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀸 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀯𑀺𑀦𑀬𑀸—  𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘𑁇 𑀤𑁆𑀯𑁂 𑀯𑁂𑀦𑀬𑀺𑀓𑀸—  𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑀦𑀼𑀮𑁄𑀫𑀜𑁆𑀘𑁇 𑀤𑁆𑀯𑁂 𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀲𑀮𑁆𑀮𑁂𑀔𑀸—  𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄, 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀫𑀢𑁆𑀢𑀓𑀸𑀭𑀺𑀢𑀸𑁇 𑀤𑁆𑀯𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀤𑁆𑀯𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑁂𑀦 𑀯𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀯𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀭𑀺𑀯𑀸𑀲𑀸—  𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄, 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀧𑀭𑀺𑀯𑀸𑀲𑀸—  𑀲𑀼𑀤𑁆𑀥𑀦𑁆𑀢𑀧𑀭𑀺𑀯𑀸𑀲𑁄 𑀲𑀫𑁄𑀥𑀸𑀦𑀧𑀭𑀺𑀯𑀸𑀲𑁄𑁇 𑀤𑁆𑀯𑁂 𑀫𑀸𑀦𑀢𑁆𑀢𑀸—  𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀫𑀸𑀦𑀢𑁆𑀢𑀁, 𑀅𑀧𑁆𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀫𑀸𑀦𑀢𑁆𑀢𑀁𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀫𑀸𑀦𑀢𑁆𑀢𑀸—  𑀧𑀓𑁆𑀔𑀫𑀸𑀦𑀢𑁆𑀢𑀁, 𑀲𑀫𑁄𑀥𑀸𑀦𑀫𑀸𑀦𑀢𑁆𑀢𑀁𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀭𑀢𑁆𑀢𑀺𑀘𑁆𑀙𑁂𑀤𑁄—  𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀲𑁆𑀲 𑀘 𑀫𑀸𑀦𑀢𑁆𑀢𑀘𑀸𑀭𑀺𑀓𑀲𑁆𑀲 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑀸𑀦𑀺—  𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀸𑀤𑀭𑀺𑀬𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀸𑀤𑀭𑀺𑀬𑀜𑁆𑀘𑁇 𑀤𑁆𑀯𑁂 𑀮𑁄𑀡𑀸𑀦𑀺—  𑀚𑀸𑀢𑀺𑀫𑀜𑁆𑀘 𑀓𑀸𑀭𑀺𑀫𑀜𑁆𑀘𑁇 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀤𑁆𑀯𑁂 𑀮𑁄𑀡𑀸𑀦𑀺—  𑀲𑀸𑀫𑀼𑀤𑁆𑀤𑀁 𑀓𑀸𑀍𑀅𑀮𑁄𑀡𑀁𑁇 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀤𑁆𑀯𑁂 𑀮𑁄𑀡𑀸𑀦𑀺—  𑀲𑀺𑀦𑁆𑀥𑀯𑀁, 𑀉𑀩𑁆𑀪𑀺𑀤𑀁𑁇 𑀅𑀧𑀭𑀸𑀦𑀺𑀧𑀺 𑀤𑁆𑀯𑁂 𑀮𑁄𑀡𑀸𑀦𑀺—  𑀭𑁄𑀫𑀓𑀁, 𑀧𑀓𑁆𑀓𑀸𑀮𑀓𑀁𑁇 𑀤𑁆𑀯𑁂 𑀧𑀭𑀺𑀪𑁄𑀕𑀸—  𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀧𑀭𑀺𑀪𑁄𑀕𑁄 𑀘 𑀩𑀸𑀳𑀺𑀭𑀧𑀭𑀺𑀪𑁄𑀕𑁄 𑀘𑁇 𑀤𑁆𑀯𑁂 𑀅𑀓𑁆𑀓𑁄𑀲𑀸—  𑀳𑀻𑀦𑁄 𑀘 𑀅𑀓𑁆𑀓𑁄𑀲𑁄 𑀉𑀓𑁆𑀓𑀝𑁆𑀞𑁄 𑀘 𑀅𑀓𑁆𑀓𑁄𑀲𑁄𑁇 𑀤𑁆𑀯𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑀁 𑀳𑁄𑀢𑀺—  𑀧𑀺𑀬𑀓𑀫𑁆𑀬𑀲𑁆𑀲 𑀯𑀸 𑀪𑁂𑀤𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀲𑁆𑀲 𑀯𑀸𑁇 𑀤𑁆𑀯𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀕𑀡𑀪𑁄𑀚𑀦𑀁 𑀧𑀲𑀯𑀢𑀺—  𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀢𑁄 𑀯𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀢𑁄 𑀯𑀸𑁇 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸—  𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸, 𑀧𑀘𑁆𑀙𑀺𑀫𑀺𑀓𑀸𑁇 𑀤𑁆𑀯𑁂 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺𑁇 𑀤𑁆𑀯𑁂 𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀝𑁆𑀞𑀧𑀦𑀸𑀦𑀺𑁇

1450\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀩𑀸𑀮𑀸—  𑀬𑁄 𑀘 𑀅𑀦𑀸𑀕𑀢𑀁 𑀪𑀸𑀭𑀁 𑀯𑀳𑀢𑀺, 𑀬𑁄 𑀘 𑀆𑀕𑀢𑀁 𑀪𑀸𑀭𑀁 𑀦 𑀯𑀳𑀢𑀺𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸—  𑀬𑁄 𑀘 𑀅𑀦𑀸𑀕𑀢𑀁 𑀪𑀸𑀭𑀁 𑀦 𑀯𑀳𑀢𑀺, 𑀬𑁄 𑀘 𑀆𑀕𑀢𑀁 𑀪𑀸𑀭𑀁 𑀯𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀩𑀸𑀮𑀸—  𑀬𑁄 𑀘 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸—  𑀬𑁄 𑀘 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀩𑀸𑀮𑀸—  𑀬𑁄 𑀘 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸—  𑀬𑁄 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀩𑀸𑀮𑀸—  𑀬𑁄 𑀘 𑀅𑀥𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸—  𑀬𑁄 𑀘 𑀅𑀥𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻𑁇 𑀅𑀧𑀭𑁂𑀧𑀺 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀩𑀸𑀮𑀸—  𑀬𑁄 𑀘 𑀅𑀯𑀺𑀦𑀬𑁂 𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀅𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸—  𑀬𑁄 𑀘 𑀅𑀯𑀺𑀦𑀬𑁂 𑀅𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻𑁇

1451\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀦 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀢𑀺, 𑀬𑁄 𑀘 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁 𑀦 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀦 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁 𑀦 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀢𑀺, 𑀬𑁄 𑀘 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀢𑀺𑁇 𑀅𑀧𑀭𑁂𑀲𑀫𑁆𑀧𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻𑁇 𑀅𑀧𑀭𑁂𑀲𑀫𑁆𑀧𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀜𑁆𑀜𑀻𑁇 𑀅𑀧𑀭𑁂𑀲𑀫𑁆𑀧𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀥𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀥𑀫𑁆𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀲𑀜𑁆𑀜𑀻𑁇 𑀅𑀧𑀭𑁂𑀲𑀫𑁆𑀧𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀯𑀺𑀦𑀬𑁂 𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀅𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀆𑀲𑀯𑀸 𑀦 𑀯𑀟𑁆𑀠𑀦𑁆𑀢𑀺—  𑀬𑁄 𑀘 𑀅𑀯𑀺𑀦𑀬𑁂 𑀅𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻, 𑀬𑁄 𑀘 𑀯𑀺𑀦𑀬𑁂 𑀯𑀺𑀦𑀬𑀲𑀜𑁆𑀜𑀻𑁇

1453\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1454\. _𑀲𑀜𑁆𑀜𑀸 𑀮𑀤𑁆𑀥𑀸 𑀘 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀸,_  
_𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀸 𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀸;_  
_𑀲𑀘𑁆𑀘𑀁 𑀪𑀽𑀫𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁄,_  
_𑀆𑀤𑀺𑀬𑀦𑁆𑀢𑁄 𑀲𑀫𑀸𑀤𑀺𑀬𑀁𑁇_  


1455\. _𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀤𑁂𑀦𑁆𑀢𑁄 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑁄,_  
_𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀘;_  
_𑀅𑀭𑀼𑀡𑀸𑀙𑀺𑀦𑁆𑀤𑀁 𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑁄,_  
_𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀘 𑀉𑀧𑁄𑀲𑀣𑀸𑁇_  


1456\. _𑀧𑀯𑀸𑀭𑀡𑀸 𑀓𑀫𑁆𑀫𑀸𑀧𑀭𑀸,_  
_𑀯𑀢𑁆𑀣𑀼 𑀅𑀧𑀭𑀸 𑀤𑁄𑀲𑀸 𑀘;_  
_𑀅𑀧𑀭𑀸 𑀤𑁆𑀯𑁂 𑀘 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺,_  
_𑀦𑀸𑀦𑀸 𑀲𑀫𑀸𑀦𑀫𑁂𑀯 𑀘𑁇_  


1457\. _𑀧𑀸𑀭𑀸𑀚𑀺𑀲𑀁𑀖𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬,_  
_𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀸;_  
_𑀤𑀼𑀓𑁆𑀓𑀝𑀸 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸 𑀘𑁂𑀯,_  
_𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸 𑀘𑁇_  


1458\. _𑀪𑀺𑀚𑁆𑀚𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸,_  
_𑀢𑀣𑁂𑀯 𑀅𑀧𑀭𑁂 𑀤𑀼𑀯𑁂;_  
_𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀅𑀪𑀩𑁆𑀩𑀸𑀪𑀩𑁆𑀩𑀫𑁂𑀯 𑀘𑁇_  


1459\. _𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑀸 𑀘,_  
_𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀸 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀸;_  
_𑀑𑀲𑀸𑀭𑀡𑀸 𑀧𑀝𑀺𑀜𑁆𑀜𑀸 𑀘,_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀸𑁇_  


1460\. _𑀉𑀧𑀖𑀸𑀢𑀺 𑀘𑁄𑀤𑀦𑀸 𑀘,_  
_𑀓𑀣𑀺𑀦𑀸 𑀘 𑀤𑀼𑀯𑁂 𑀢𑀣𑀸;_  
_𑀘𑀻𑀯𑀭𑀸 𑀧𑀢𑁆𑀢𑀫𑀡𑁆𑀟𑀮𑀸,_  
_𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀸 𑀢𑀣𑁂𑀯 𑀤𑁆𑀯𑁂𑁇_  


1461\. _𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀸 𑀘 𑀯𑀺𑀦𑀬𑀸,_  
_𑀯𑁂𑀦𑀬𑀺𑀓𑀸 𑀘 𑀲𑀮𑁆𑀮𑁂𑀔𑀸;_  
_𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀘 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺,_  
_𑀧𑀭𑀺𑀯𑀸𑀲𑀸𑀧𑀭𑁂 𑀤𑀼𑀯𑁂𑁇_  


1462\. _𑀤𑁆𑀯𑁂 𑀫𑀸𑀦𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑁂 𑀘,_  
_𑀭𑀢𑁆𑀢𑀺𑀘𑁆𑀙𑁂𑀤𑁄 𑀅𑀦𑀸𑀤𑀭𑀺;_  
_𑀤𑁆𑀯𑁂 𑀮𑁄𑀡𑀸 𑀢𑀬𑁄 𑀅𑀧𑀭𑁂,_  
_𑀧𑀭𑀺𑀪𑁄𑀕𑀓𑁆𑀓𑁄𑀲𑁂𑀦 𑀘𑁇_  


1463\. _𑀧𑁂𑀲𑀼𑀜𑁆𑀜𑁄 𑀘 𑀕𑀡𑀸𑀯𑀲𑁆𑀲,_  
_𑀞𑀧𑀦𑀸 𑀪𑀸𑀭𑀓𑀧𑁆𑀧𑀺𑀬𑀁;_  
_𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀅𑀥𑀫𑁆𑀫𑀥𑀫𑁆𑀫𑀸,_  
_𑀯𑀺𑀦𑀬𑁂 𑀆𑀲𑀯𑁂 𑀢𑀣𑀸𑀢𑀺𑁇_
