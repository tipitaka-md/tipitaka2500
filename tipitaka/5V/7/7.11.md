# 𑁭𑁇𑁧𑁧 𑀏𑀓𑀸𑀤𑀲𑀓𑀯𑀸𑀭

### 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀦𑀬

### 𑀏𑀓𑀸𑀤𑀲𑀓𑀯𑀸𑀭

(331.)

1583. 𑀏𑀓𑀸𑀤𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸, 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀧𑀸𑀤𑀼𑀓𑀸 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀧𑀢𑁆𑀢𑀸 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑀦𑀺𑁇 𑀏𑀓𑀸𑀤𑀲 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀁 𑀏𑀓𑀸𑀤𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀏𑀓𑀸𑀤𑀲 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦 𑀯𑀺𑀓𑀧𑁆𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀏𑀓𑀸𑀤𑀲𑁂 𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑀫𑀦𑁂 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀳𑁄𑀢𑀺𑁇 𑀏𑀓𑀸𑀤𑀲 𑀕𑀡𑁆𑀞𑀺𑀓𑀸 𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀯𑀺𑀥𑀸 𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀧𑀣𑀯𑀻 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀧𑀣𑀯𑀻 𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀦𑀺𑀲𑁆𑀲𑀬𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑁄𑁇 𑀏𑀓𑀸𑀤𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀏𑀓𑀸𑀤𑀲 𑀧𑀭𑀫𑀸𑀦𑀺𑁇 𑀏𑀓𑀸𑀤𑀲 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀸𑀤𑀲 𑀲𑀻𑀫𑀸𑀤𑁄𑀲𑀸𑁇 𑀅𑀓𑁆𑀓𑁄𑀲𑀓𑀧𑀭𑀺𑀪𑀸𑀲𑀓𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁂 𑀏𑀓𑀸𑀤𑀲𑀸𑀤𑀻𑀦𑀯𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀸𑀬 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀬 𑀬𑀸𑀦𑀻𑀓𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀬 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀘𑀺𑀢𑀸𑀬 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑀬 𑀏𑀓𑀸𑀤𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑁇 𑀲𑀼𑀔𑀁 𑀲𑀼𑀧𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦 𑀧𑀸𑀧𑀓𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀺𑀬𑁄 𑀳𑁄𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀦𑀸𑀲𑁆𑀲 𑀅𑀕𑁆𑀕𑀺 𑀯𑀸 𑀯𑀺𑀲𑀁 𑀯𑀸 𑀲𑀢𑁆𑀣𑀁 𑀯𑀸 𑀓𑀫𑀢𑀺, 𑀢𑀼𑀯𑀝𑁆𑀝𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀢𑀺, 𑀫𑀼𑀔𑀯𑀡𑁆𑀡𑁄 𑀯𑀺𑀧𑁆𑀧𑀲𑀻𑀤𑀢𑀺, 𑀅𑀲𑀫𑁆𑀫𑀽𑀍𑀳𑁄 𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀉𑀢𑁆𑀢𑀭𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀚𑁆𑀛𑀦𑁆𑀢𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀮𑁄𑀓𑀽𑀧𑀕𑁄 𑀳𑁄𑀢𑀺—  𑀫𑁂𑀢𑁆𑀢𑀸𑀬 𑀘𑁂𑀢𑁄𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀆𑀲𑁂𑀯𑀺𑀢𑀸𑀬 𑀪𑀸𑀯𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀮𑀻𑀓𑀢𑀸𑀬 𑀬𑀸𑀦𑀻𑀓𑀢𑀸𑀬 𑀯𑀢𑁆𑀣𑀼𑀓𑀢𑀸𑀬 𑀅𑀦𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀘𑀺𑀢𑀸𑀬 𑀲𑀼𑀲𑀫𑀸𑀭𑀤𑁆𑀥𑀸𑀬 𑀇𑀫𑁂 𑀏𑀓𑀸𑀤𑀲𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀧𑀸𑀝𑀺𑀓𑀗𑁆𑀔𑀸𑀢𑀺𑁇

---

1584. 𑀏𑀓𑀸𑀤𑀲𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

1585. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1586. _𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀘,_  
_𑀧𑀢𑁆𑀢𑀸 𑀘 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀘;_  
_𑀢𑀢𑀺𑀬𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸 𑀘,_  
_𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀺𑀓𑀧𑁆𑀧𑀦𑀸𑁇_  


1587. _𑀅𑀭𑀼𑀡𑀸 𑀕𑀡𑁆𑀞𑀺𑀓𑀸 𑀯𑀺𑀥𑀸,_  
_𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀸 𑀘 𑀓𑀧𑁆𑀧𑀺𑀬𑀁;_  
_𑀦𑀺𑀲𑁆𑀲𑀬𑀸𑀯𑀦𑁆𑀤𑀺𑀬𑀸 𑀘𑁂𑀯,_  
_𑀧𑀭𑀫𑀸𑀦𑀺 𑀯𑀭𑀸𑀦𑀺 𑀘;_  
_𑀲𑀻𑀫𑀸𑀤𑁄𑀲𑀸 𑀘 𑀅𑀓𑁆𑀓𑁄𑀲𑀸,_  
_𑀫𑁂𑀢𑁆𑀢𑀸𑀬𑁂𑀓𑀸𑀤𑀲𑀸 𑀓𑀢𑀸𑀢𑀺𑁇_  


---

1588. 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀁𑁇



---

1589. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1590. _𑀏𑀓𑀓𑀸 𑀘 𑀤𑀼𑀓𑀸 𑀘𑁂𑀯,_  
_𑀢𑀺𑀓𑀸 𑀘 𑀘𑀢𑀼𑀧𑀜𑁆𑀘𑀓𑀸;_  
_𑀙𑀲𑀢𑁆𑀢𑀝𑁆𑀞𑀦𑀯𑀓𑀸 𑀘,_  
_𑀤𑀲 𑀏𑀓𑀸𑀤𑀲𑀸𑀦𑀺 𑀘𑁇_  


1591. _𑀳𑀺𑀢𑀸𑀬 𑀲𑀩𑁆𑀩𑀲𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀜𑀸𑀢𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀸;_  
_𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀸 𑀯𑀺𑀫𑀮𑀸,_  
_𑀫𑀳𑀸𑀯𑀻𑀭𑁂𑀦 𑀤𑁂𑀲𑀺𑀢𑀸𑀢𑀺𑁇_  


---

1592. 𑀏𑀓𑀼𑀢𑁆𑀢𑀭𑀺𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
