# 𑁨𑁦𑁇𑁧 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀧𑀜𑁆𑀳𑀸

### 𑀲𑁂𑀤𑀫𑁄𑀘𑀦𑀕𑀸𑀣𑀸

### 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀧𑀜𑁆𑀳𑀸

(479.)

2342\. _𑀅𑀲𑀁𑀯𑀸𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀳𑀺 𑀘,_  
_𑀲𑀫𑁆𑀪𑁄𑀕𑁄 𑀏𑀓𑀘𑁆𑀘𑁄 𑀢𑀳𑀺𑀁 𑀦 𑀮𑀩𑁆𑀪𑀢𑀺;_  
_𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁂𑀦 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2343\. _𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀬𑀁 𑀅𑀯𑁂𑀪𑀗𑁆𑀕𑀺𑀬𑀁,_  
_𑀧𑀜𑁆𑀘 𑀯𑀼𑀢𑁆𑀢𑀸 𑀫𑀳𑁂𑀲𑀺𑀦𑀸;_  
_𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2344\. _𑀤𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑁂 𑀦 𑀯𑀤𑀸𑀫𑀺,_  
_𑀏𑀓𑀸𑀤𑀲 𑀯𑀺𑀯𑀚𑁆𑀚𑀺𑀬;_  
_𑀯𑀼𑀟𑁆𑀠𑀁 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2345\. _𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀦 𑀘 𑀧𑀦 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑁄,_  
_𑀦 𑀲𑀁𑀖𑀪𑀺𑀦𑁆𑀦𑁄 𑀦 𑀘 𑀧𑀦 𑀧𑀓𑁆𑀔𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑁄;_  
_𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀪𑀽𑀫𑀺𑀬𑀸 𑀞𑀺𑀢𑁄,_  
_𑀓𑀣𑀁 𑀦𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀅𑀲𑀸𑀥𑀸𑀭𑀡𑁄 𑀲𑀺𑀬𑀸;_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2346\. _𑀉𑀧𑁂𑀢𑀺 𑀥𑀫𑁆𑀫𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀫𑀸𑀦𑁄,_  
_𑀓𑀼𑀲𑀮𑀁 𑀅𑀢𑁆𑀣𑀽𑀧𑀲𑀜𑁆𑀳𑀺𑀢𑀁;_  
_𑀦 𑀚𑀻𑀯𑀢𑀺 𑀦 𑀫𑀢𑁄 𑀦 𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑁄,_  
_𑀢𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀓𑀢𑀫𑀁 𑀯𑀤𑀦𑁆𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀸;_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2347\. _𑀉𑀩𑁆𑀪𑀓𑁆𑀔𑀓𑁂 𑀦 𑀯𑀤𑀸𑀫𑀺,_  
_𑀅𑀥𑁄 𑀦𑀸𑀪𑀺𑀁 𑀯𑀺𑀯𑀚𑁆𑀚𑀺𑀬;_  
_𑀫𑁂𑀣𑀼𑀦𑀥𑀫𑁆𑀫𑀧𑀘𑁆𑀘𑀬𑀸,_  
_𑀓𑀣𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄 𑀲𑀺𑀬𑀸;_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2348\. _𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀜𑀸𑀘𑀺𑀓𑀸𑀬 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀢𑀺,_  
_𑀅𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁;_  
_𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀅𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2349\. _𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀜𑀸𑀘𑀺𑀓𑀸𑀬 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀢𑀺,_  
_𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀺𑀓𑀁;_  
_𑀅𑀦𑀸𑀭𑀫𑁆𑀪𑀁 𑀲𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2350\. _𑀦 𑀓𑀸𑀬𑀺𑀓𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀧𑀬𑁄𑀕𑀫𑀸𑀘𑀭𑁂,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀯𑀸𑀘𑀸𑀬 𑀧𑀭𑁂 𑀪𑀡𑁂𑀬𑁆𑀬;_  
_𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀕𑀭𑀼𑀓𑀁 𑀙𑁂𑀚𑁆𑀚𑀯𑀢𑁆𑀣𑀼𑀁,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2351\. _𑀦 𑀓𑀸𑀬𑀺𑀓𑀁 𑀯𑀸𑀘𑀲𑀺𑀓𑀜𑁆𑀘 𑀓𑀺𑀜𑁆𑀘𑀺,_  
_𑀫𑀦𑀲𑀸𑀧𑀺 𑀲𑀦𑁆𑀢𑁄 𑀦 𑀓𑀭𑁂𑀬𑁆𑀬 𑀧𑀸𑀧𑀁;_  
_𑀲𑁄 𑀦𑀸𑀲𑀺𑀢𑁄 𑀓𑀺𑀦𑁆𑀢𑀺 𑀲𑀼𑀦𑀸𑀲𑀺𑀢𑁄 𑀪𑀯𑁂,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2352\. _𑀅𑀦𑀸𑀮𑀧𑀦𑁆𑀢𑁄 𑀫𑀦𑀼𑀚𑁂𑀦 𑀓𑁂𑀦𑀘𑀺,_  
_𑀯𑀸𑀘𑀸𑀕𑀺𑀭𑀁 𑀦𑁄 𑀘 𑀧𑀭𑁂 𑀪𑀡𑁂𑀬𑁆𑀬;_  
_𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀯𑀸𑀘𑀲𑀺𑀓𑀁 𑀦 𑀓𑀸𑀬𑀺𑀓𑀁,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2353\. _𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸 𑀩𑀼𑀤𑁆𑀥𑀯𑀭𑁂𑀦 𑀯𑀡𑁆𑀡𑀺𑀢𑀸,_  
_𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀘𑀢𑀼𑀭𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀼𑀁;_  
_𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀏𑀓𑀧𑀬𑁄𑀕𑁂𑀦 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2354\. _𑀉𑀪𑁄 𑀏𑀓𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀳𑀢𑁆𑀣𑀢𑁄 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬;_  
_𑀲𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀦𑀸𑀦𑀸,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_  


2355\. _𑀘𑀢𑀼𑀭𑁄 𑀚𑀦𑀸 𑀲𑀁𑀯𑀺𑀥𑀸𑀬,_  
_𑀕𑀭𑀼𑀪𑀡𑁆𑀟𑀁 𑀅𑀯𑀸𑀳𑀭𑀼𑀁;_  
_𑀢𑀬𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀏𑀓𑁄 𑀦 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁄,_  
_𑀧𑀜𑁆𑀳𑀸 𑀫𑁂𑀲𑀸 𑀓𑀼𑀲𑀮𑁂𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀸𑁇_
