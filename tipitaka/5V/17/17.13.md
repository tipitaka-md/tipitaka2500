# 𑁧𑁭𑁇𑁧𑁩 𑀆𑀯𑀸𑀲𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀘𑀓

### 𑀆𑀯𑀸𑀲𑀺𑀓𑀯𑀕𑁆𑀕

(461.)

2164\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂”𑀢𑀺?

2165\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀁𑀖𑀺𑀓𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀺𑀓𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇

2166\. 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀁𑀖𑀺𑀓𑀁 𑀦 𑀧𑀼𑀕𑁆𑀕𑀮𑀺𑀓𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇

(462.)

2167\. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸”𑀢𑀺?

2168\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀇𑀥𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑁇 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀇𑀥𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 𑀅𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀬𑁄𑀢𑀺 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀢𑀺, 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀦 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀦 𑀲𑀫𑀼𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀯𑀺𑀦𑀬𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸”𑀢𑀺𑁇

(463.)

2169\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂”𑀢𑀺?

2170\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇

2171\. 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀢𑁆𑀢𑀼𑀤𑁆𑀤𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇

(464.)

2172\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑁄…𑀧𑁂…  𑀪𑀡𑁆𑀟𑀸𑀕𑀸𑀭𑀺𑀓𑁄…𑀧𑁂…  𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑁄…𑀧𑁂…  𑀘𑀻𑀯𑀭𑀪𑀸𑀚𑀓𑁄…𑀧𑁂…  𑀬𑀸𑀕𑀼𑀪𑀸𑀚𑀓𑁄…𑀧𑁂…  𑀨𑀮𑀪𑀸𑀚𑀓𑁄…𑀧𑁂…  𑀔𑀚𑁆𑀚𑀪𑀸𑀚𑀓𑁄…𑀧𑁂…  𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀓𑁄…𑀧𑁂…  𑀲𑀸𑀝𑀺𑀬𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄…𑀧𑁂…  𑀧𑀢𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀸𑀧𑀓𑁄…𑀧𑁂…  𑀆𑀭𑀸𑀫𑀺𑀓𑀧𑁂𑀲𑀓𑁄…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂”𑀢𑀺?

2173\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀧𑁂𑀲𑀺𑀢𑀸𑀧𑁂𑀲𑀺𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀦𑀺𑀭𑀬𑁂𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀧𑁂𑀲𑀺𑀢𑀸𑀧𑁂𑀲𑀺𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑁂𑀲𑀓𑁄 𑀬𑀣𑀸𑀪𑀢𑀁 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀯𑀁 𑀲𑀕𑁆𑀕𑁂”𑀢𑀺𑁇
