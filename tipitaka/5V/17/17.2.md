# 𑁧𑁭𑁇𑁨 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀘𑀓

### 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀦𑀯𑀕𑁆𑀕

(420.)

1960\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺?

1961\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀸𑀫𑀡𑁂𑀭𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀲𑀫𑁆𑀫𑀢𑁄𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀑𑀯𑀤𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀁𑁇 (𑁧)

1962\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀖𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀢𑀸𑀤𑀺𑀲𑀺𑀓𑀁, 𑀢𑀢𑁄 𑀯𑀸 𑀧𑀸𑀧𑀺𑀝𑁆𑀞𑀢𑀭𑀁, 𑀓𑀫𑁆𑀫𑀁 𑀕𑀭𑀳𑀢𑀺, 𑀓𑀫𑁆𑀫𑀺𑀓𑁂 𑀕𑀭𑀳𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀁𑁇 (𑁨)

1963\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀁𑁇 (𑁩)

1964\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘, 𑀑𑀫𑀤𑁆𑀤𑀓𑀸𑀭𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀯𑀢𑁆𑀢𑁂𑀲𑀼 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀘 𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 (𑁪)

(421.)

1965\. “𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂𑀦 𑀓𑀢𑀺 𑀥𑀫𑁆𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺?

1966\. “𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑁂𑀦, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑁂𑀦, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂𑀦 𑀦𑀻𑀘𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀲𑀁𑀖𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄, 𑀭𑀚𑁄𑀳𑀭𑀡𑀲𑀫𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦, 𑀆𑀲𑀦𑀓𑀼𑀲𑀮𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀓𑀼𑀲𑀮𑁂𑀦, 𑀣𑁂𑀭𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀧𑀔𑀚𑁆𑀚𑀦𑁆𑀢𑁂𑀦, 𑀦𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀲𑀦𑁂𑀦 𑀅𑀧𑁆𑀧𑀝𑀺𑀩𑀸𑀳𑀦𑁆𑀢𑁂𑀦 𑀬𑀣𑀸𑀧𑀢𑀺𑀭𑀽𑀧𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑀩𑁆𑀩𑀁, 𑀅𑀦𑀸𑀦𑀸𑀓𑀣𑀺𑀓𑁂𑀦 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀺𑀓𑁂𑀦, 𑀲𑀸𑀫𑀁 𑀯𑀸 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄, 𑀧𑀭𑁄 𑀯𑀸 𑀅𑀚𑁆𑀛𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀭𑀺𑀬𑁄 𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁄 𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀺𑀢𑀩𑁆𑀩𑁄, 𑀲𑀘𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑁄 𑀲𑀫𑀕𑁆𑀕𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀢𑀺; 𑀢𑀢𑁆𑀭 𑀘𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀅𑀧𑀺 𑀤𑀺𑀝𑁆𑀞𑀸𑀯𑀺𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸 𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑁇 𑀢𑀁 𑀓𑀺𑀲𑁆𑀲𑀳𑁂𑀢𑀼? 𑀫𑀸𑀳𑀁 𑀲𑀁𑀖𑁂𑀦 𑀦𑀸𑀦𑀢𑁆𑀢𑁄 𑀅𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑀗𑁆𑀕𑀸𑀫𑀸𑀯𑀘𑀭𑁂𑀦𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑁂𑀦 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

(422.)

1967\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀅𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀅𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑀸”𑀢𑀺?

1969\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀅𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀅𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀉𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀸 𑀘 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀲𑀸𑀤𑁂𑀢𑀸 𑀘, 𑀅𑀥𑀫𑁆𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀲𑀫𑁆𑀨𑀜𑁆𑀘 𑀩𑀳𑀼𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀅𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀅𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀉𑀲𑁆𑀲𑀸𑀤𑁂𑀢𑀸 𑀘 𑀳𑁄𑀢𑀺, 𑀦 𑀅𑀧𑀲𑀸𑀤𑁂𑀢𑀸 𑀘, 𑀥𑀫𑁆𑀫𑀁 𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀲𑀫𑁆𑀨𑀜𑁆𑀘 𑀦 𑀩𑀳𑀼𑀁 𑀪𑀸𑀲𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑁇 (𑁨)

1970\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀅𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀅𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀲𑀬𑁆𑀳𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀅𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀅𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀅𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀧𑀲𑀬𑁆𑀳𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀑𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀦𑁆𑀢𑁄 𑀩𑀳𑀼𑀚𑀦𑀓𑀦𑁆𑀢𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀩𑀳𑀼𑀚𑀦𑀫𑀦𑀸𑀧𑁄 𑀘 𑀩𑀳𑀼𑀚𑀦𑀭𑀼𑀘𑀺𑀢𑁄 𑀘𑀸”𑀢𑀺𑁇 (𑁩)

1971\. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀺𑀬𑀸”𑀢𑀺?

1972\. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀢𑁆𑀢𑀦𑁄 𑀲𑀻𑀮𑀓𑁆𑀔𑀦𑁆𑀥𑁄 𑀲𑀼𑀕𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀼𑀭𑀓𑁆𑀔𑀺𑀢𑁄, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀸𑀦𑀁 𑀧𑀝𑀺𑀲𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀯𑁄𑀳𑀭𑀢𑀺, 𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑁂 𑀲𑀳𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀸𑀦𑀺𑀲𑀁𑀲𑀸 𑀯𑀺𑀦𑀬𑀧𑀭𑀺𑀬𑀢𑁆𑀢𑀺𑀬𑀸”𑁇
