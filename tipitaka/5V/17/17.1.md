# 𑁧𑁭𑁇𑁧 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀘𑀓

* 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀯𑀕𑁆𑀕

(417.)

1939\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆”𑀢𑀺?

1940\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀉𑀧𑁄𑀲𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀉𑀧𑁄𑀲𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇

1941\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀯𑀸𑀭𑀡𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀯𑀸𑀭𑀡𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇

1942\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀊𑀦𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀦𑀸𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸 𑀳𑁄𑀢𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑁄 𑀯𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁”𑁇

1943\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺?

1947\. 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁧)

1948\. “𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀲𑀻𑀮𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀚𑁆𑀛𑀸𑀘𑀸𑀭𑁂 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀢𑀺𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁨)

1949\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀤𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀤𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀤𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁩)

1950\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀅𑀦𑀸𑀘𑀸𑀭𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀅𑀦𑀸𑀘𑀸𑀭𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀅𑀦𑀸𑀘𑀸𑀭𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁪)

1951\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑀺𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑀺𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀉𑀧𑀖𑀸𑀢𑀺𑀓𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁫)

1952\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀸𑀬𑀺𑀓𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀚𑀻𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀚𑀻𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑀸𑀬𑀺𑀓𑀯𑀸𑀘𑀲𑀺𑀓𑁂𑀦 𑀫𑀺𑀘𑁆𑀙𑀸𑀚𑀻𑀯𑁂𑀦 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁬)

1953\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀢𑀺, 𑀲𑀸𑀫𑀡𑁂𑀭𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀺, 𑀲𑀫𑁆𑀫𑀢𑁄𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀑𑀯𑀤𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁭)

1954\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀬𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀁𑀖𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑀁 𑀳𑁄𑀢𑀺 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀢𑀸𑀤𑀺𑀲𑀺𑀓𑀁, 𑀢𑀢𑁄 𑀯𑀸 𑀧𑀸𑀧𑀺𑀝𑁆𑀞𑀢𑀭𑀁, 𑀓𑀫𑁆𑀫𑀁 𑀕𑀭𑀳𑀢𑀺, 𑀓𑀫𑁆𑀫𑀺𑀓𑁂 𑀕𑀭𑀳𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 (𑁮)

1955\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀘 𑀳𑁄𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀦𑁆𑀦𑁄 𑀘—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 (𑁯)

---

1956\. 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀞𑀫𑁄𑁇



---

1957\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1958\. _𑀉𑀧𑁄𑀲𑀣𑀁 𑀧𑀯𑀸𑀭𑀡𑀁,_  
_𑀆𑀧𑀢𑁆𑀢𑀺 𑀘 𑀕𑀺𑀮𑀸𑀦𑀓𑀁;_  
_𑀅𑀪𑀺𑀲𑀫𑀸𑀘𑀸𑀭𑀮𑀚𑁆𑀚𑀻 𑀘,_  
_𑀅𑀥𑀺𑀲𑀻𑀮𑁂 𑀤𑀯𑁂𑀦 𑀘𑁇_  


1959\. _𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀉𑀧𑀖𑀸𑀢𑀺,_  
_𑀫𑀺𑀘𑁆𑀙𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀫𑁂𑀯 𑀘;_  
_𑀬𑀸𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲,_  
_𑀧𑀞𑀫𑁄 𑀯𑀕𑁆𑀕𑀲𑀗𑁆𑀕𑀳𑁄𑀢𑀺𑁇_
