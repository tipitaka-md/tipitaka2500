# 𑁧𑁭𑁇𑁧𑁪 𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀸𑀭𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀘𑀓

### 𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀸𑀭𑀯𑀕𑁆𑀕

(465.)

2178. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀸𑀭𑁂”𑀢𑀺?

2179. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀸𑀭𑁂𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀦𑀸𑀫𑀦𑁆𑀢𑀘𑀸𑀭𑁄, 𑀅𑀲𑀫𑀸𑀤𑀸𑀦𑀘𑀸𑀭𑁄, 𑀕𑀡𑀪𑁄𑀚𑀦𑀁, 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀘𑀻𑀯𑀭𑀁, 𑀬𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀘𑀻𑀯𑀭𑀼𑀧𑁆𑀧𑀸𑀤𑁄 𑀲𑁄 𑀦𑁂𑀲𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀓𑀣𑀺𑀦𑀢𑁆𑀣𑀸𑀭𑁂”𑀢𑀺𑁇

(466.)

2180. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑁄”𑀢𑀺?

2181. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀆𑀤𑀻𑀦𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑁄𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀼𑀧𑀢𑀺, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀧𑀸𑀧𑀓𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸 𑀦 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀅𑀲𑀼𑀘𑀺 𑀫𑀼𑀘𑁆𑀘𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀆𑀤𑀻𑀦𑀯𑀸 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑁄𑁇

2182. 𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑁄𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀲𑀼𑀔𑀁 𑀲𑀼𑀧𑀢𑀺, 𑀲𑀼𑀔𑀁 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀢𑀺, 𑀦 𑀧𑀸𑀧𑀓𑀁 𑀲𑀼𑀧𑀺𑀦𑀁 𑀧𑀲𑁆𑀲𑀢𑀺, 𑀤𑁂𑀯𑀢𑀸 𑀭𑀓𑁆𑀔𑀦𑁆𑀢𑀺, 𑀅𑀲𑀼𑀘𑀺 𑀦 𑀫𑀼𑀘𑁆𑀘𑀢𑀺—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀆𑀦𑀺𑀲𑀁𑀲𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀦𑀺𑀤𑁆𑀤𑀁 𑀑𑀓𑁆𑀓𑀫𑀢𑁄”𑀢𑀺𑁇

(467.)

2183. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸”𑀢𑀺?

2184. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀭𑀘𑁆𑀙𑀕𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀑𑀢𑀫𑀲𑀺𑀓𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀅𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀦𑁆𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀲𑀼𑀢𑁆𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇

2185. 𑀅𑀧𑀭𑁂𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀬𑀸𑀕𑀼𑀧𑀸𑀦𑁂 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀏𑀓𑀸𑀯𑀢𑁆𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀅𑀜𑁆𑀜𑀯𑀺𑀳𑀺𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀦𑀕𑁆𑀕𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇

2186. 𑀅𑀧𑀭𑁂𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀔𑀸𑀤𑀦𑁆𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀉𑀘𑁆𑀘𑀸𑀭𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀧𑀲𑁆𑀲𑀸𑀯𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇

2187. 𑀅𑀧𑀭𑁂𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀧𑀼𑀭𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀧𑀘𑁆𑀙𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑁄 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀧𑀡𑁆𑀟𑀓𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇

2188. 𑀅𑀧𑀭𑁂𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀫𑀸𑀦𑀢𑁆𑀢𑀘𑀸𑀭𑀺𑀓𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑁄 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀅𑀯𑀦𑁆𑀤𑀺𑀬𑀸”𑀢𑀺𑁇

(468.)

2189. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀦𑁆𑀤𑀺𑀬𑀸”𑀢𑀺?

2190. “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀯𑀦𑁆𑀤𑀺𑀬𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀧𑀘𑁆𑀙𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀧𑀼𑀭𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑁄 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑁄 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀆𑀘𑀭𑀺𑀬𑁄 𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀯𑀦𑁆𑀤𑀺𑀬𑁄, 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀢𑀣𑀸𑀕𑀢𑁄 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀦𑁆𑀤𑀺𑀬𑁄—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀯𑀦𑁆𑀤𑀺𑀬𑀸”𑀢𑀺𑁇

(469.)

2191. “𑀦𑀯𑀓𑀢𑀭𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑁂𑀦 𑀓𑀢𑀺 𑀥𑀫𑁆𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑀸 𑀯𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀸”𑀢𑀺?

2192. “𑀦𑀯𑀓𑀢𑀭𑁂𑀦𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑁂𑀦 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑀸 𑀯𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀦𑀯𑀓𑀢𑀭𑁂𑀦𑀼𑀧𑀸𑀮𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑁂𑀦 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸, 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸, 𑀉𑀪𑁄𑀳𑀺 𑀧𑀸𑀡𑀺𑀢𑀮𑁂𑀳𑀺 𑀧𑀸𑀤𑀸𑀦𑀺 𑀧𑀭𑀺𑀲𑀫𑁆𑀩𑀸𑀳𑀦𑁆𑀢𑁂𑀦, 𑀧𑁂𑀫𑀜𑁆𑀘 𑀕𑀸𑀭𑀯𑀜𑁆𑀘 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑀸 𑀯𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀸—  𑀦𑀯𑀓𑀢𑀭𑁂𑀦𑀼𑀧𑀸𑀮𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀦𑁆𑀢𑁂𑀦 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀸𑀤𑀸 𑀯𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇
