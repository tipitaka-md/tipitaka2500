# 𑁧𑁭𑁇𑁯 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀯𑀕𑁆𑀕

### 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀘𑀓

### 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀯𑀕𑁆𑀕

(455.)

2107. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺?

2108. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀅𑀢𑁆𑀣𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀥𑀫𑁆𑀫𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀢𑁆𑀣𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇

2109. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑁄𑀥𑀦𑁄 𑀳𑁄𑀢𑀺 𑀓𑁄𑀥𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀫𑀓𑁆𑀔𑀻 𑀳𑁄𑀢𑀺 𑀫𑀓𑁆𑀔𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀧𑀍𑀆𑀲𑀻 𑀳𑁄𑀢𑀺 𑀧𑀍𑀆𑀲𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀳𑁄𑀢𑀺 𑀇𑀲𑁆𑀲𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀸𑀫𑀸𑀲𑀻 𑀳𑁄𑀢𑀺 𑀆𑀥𑀸𑀦𑀕𑁆𑀕𑀸𑀳𑀻 𑀤𑀼𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀻—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀓𑁄𑀥𑀦𑁄 𑀳𑁄𑀢𑀺 𑀦 𑀓𑁄𑀥𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀦 𑀫𑀓𑁆𑀔𑀻 𑀳𑁄𑀢𑀺 𑀦 𑀫𑀓𑁆𑀔𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀦 𑀧𑀍𑀆𑀲𑀻 𑀳𑁄𑀢𑀺 𑀦 𑀧𑀍𑀆𑀲𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀦 𑀇𑀲𑁆𑀲𑀼𑀓𑀻 𑀳𑁄𑀢𑀺 𑀦 𑀇𑀲𑁆𑀲𑀸𑀪𑀺𑀪𑀽𑀢𑁄, 𑀅𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀧𑀭𑀸𑀫𑀸𑀲𑀻 𑀳𑁄𑀢𑀺 𑀅𑀦𑀸𑀥𑀸𑀦𑀕𑁆𑀕𑀸𑀳𑀻 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀻—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇

2110. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀼𑀧𑁆𑀧𑀢𑀺, 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀬𑀢𑀺, 𑀓𑁄𑀧𑀁 𑀚𑀦𑁂𑀢𑀺, 𑀅𑀓𑁆𑀔𑀫𑁄 𑀳𑁄𑀢𑀺 𑀅𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀓𑀼𑀧𑁆𑀧𑀢𑀺, 𑀦 𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀦 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀬𑀢𑀺, 𑀦 𑀓𑁄𑀧𑀁 𑀚𑀦𑁂𑀢𑀺, 𑀔𑀫𑁄 𑀳𑁄𑀢𑀺 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀕𑁆𑀕𑀸𑀳𑀻 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑀺𑀁—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀩𑁆𑀩𑀸𑀳𑀺𑀓𑀸𑀬 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇

2117. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀤𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀤𑀧𑀘𑁆𑀘𑀸𑀪𑀝𑁆𑀞𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀺𑀯𑀘𑀦𑀧𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀯𑀢𑁆𑀣𑀼𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀤𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀤𑀧𑀘𑁆𑀘𑀸𑀪𑀝𑁆𑀞𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀺𑀯𑀘𑀦𑀧𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

2118. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀬𑁄𑀕𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀽𑀧𑀲𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀬𑁄𑀕𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀽𑀧𑀲𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇

2119. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀧𑀬𑁄𑀕𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀦 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀧𑀬𑁄𑀕𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇

_𑀧𑀲𑀸𑀭𑁂𑀢𑀸 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁,_  
_𑀅𑀓𑀼𑀲𑀮𑁄 𑀢𑀣𑁂𑀯 𑀘𑁇_
