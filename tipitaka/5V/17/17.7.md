# 𑁧𑁭𑁇𑁭 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀘𑀓

* 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀯𑀕𑁆𑀕

(444.)

2069\. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸”𑀢𑀺? “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀅𑀢𑁆𑀣𑀺 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀕𑀸𑀫𑀻, 𑀅𑀢𑁆𑀣𑀺 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀤𑀼𑀓𑁆𑀓𑀝𑀕𑀸𑀫𑀻—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸”𑀢𑀺𑁇

(445.)

2070\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀁 𑀯𑀸 𑀞𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  ‘𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁, 𑀫𑀸 𑀓𑀮𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆’𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀸 𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸”𑀢𑀺?

2071\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀁 𑀯𑀸 𑀞𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  ‘𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁, 𑀫𑀸 𑀓𑀮𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆’𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀸 𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀘, 𑀅𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀘, 𑀘𑀸𑀯𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄 𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦𑁄 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑁄—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀁 𑀯𑀸 𑀞𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  ‘𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁, 𑀫𑀸 𑀓𑀮𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆’𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀸 𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇

2072\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀁 𑀯𑀸 𑀞𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  ‘𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁, 𑀫𑀸 𑀓𑀮𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆’𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀸 𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀯𑀘𑀻𑀲𑀫𑀸𑀘𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀩𑀸𑀮𑁄 𑀳𑁄𑀢𑀺 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄, 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀁 𑀯𑀸 𑀞𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲—  ‘𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁, 𑀫𑀸 𑀓𑀮𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆’𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀯𑀸 𑀧𑀯𑀸𑀭𑀡𑀸 𑀯𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇

(446.)

2073\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺?

2074\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇

2075\. 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑀼𑀬𑁄𑀕𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

(447.)

2076\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀮𑀚𑁆𑀚𑀺𑀢𑀸, 𑀅𑀜𑁆𑀜𑀸𑀡𑀢𑀸, 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀢𑀸, 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀺𑀢𑀸, 𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀺𑀢𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

2077\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀦, 𑀅𑀲𑁆𑀲𑀯𑀦𑁂𑀦, 𑀧𑀲𑀼𑀢𑁆𑀢𑀓𑀢𑀸, 𑀢𑀣𑀸𑀲𑀜𑁆𑀜𑀻, 𑀲𑀢𑀺𑀲𑀫𑁆𑀫𑁄𑀲𑀸—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺𑁇

(448.)

2078\. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀭𑀸”𑀢𑀺? “𑀧𑀜𑁆𑀘𑀺𑀫𑁂, 𑀉𑀧𑀸𑀮𑀺, 𑀯𑁂𑀭𑀸𑁇 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀁, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀁—  𑀇𑀫𑁂 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀯𑁂𑀭𑀸”𑀢𑀺𑁇

2079\. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑁄”𑀢𑀺? “𑀧𑀜𑁆𑀘𑀺𑀫𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑁄𑁇 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀘? 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀻, 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑁆𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀻—  𑀇𑀫𑀸 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀯𑁂𑀭𑀫𑀡𑀺𑀬𑁄”𑀢𑀺𑁇

(449.)

2080\. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀬𑀲𑀦𑀸𑀦𑀻”𑀢𑀺? “𑀧𑀜𑁆𑀘𑀺𑀫𑀸𑀦𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀩𑁆𑀬𑀲𑀦𑀸𑀦𑀺𑁇 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘? 𑀜𑀸𑀢𑀺𑀩𑁆𑀬𑀲𑀦𑀁, 𑀪𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀁, 𑀭𑁄𑀕𑀩𑁆𑀬𑀲𑀦𑀁, 𑀲𑀻𑀮𑀩𑁆𑀬𑀲𑀦𑀁, 𑀤𑀺𑀝𑁆𑀞𑀺𑀩𑁆𑀬𑀲𑀦𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀩𑁆𑀬𑀲𑀦𑀸𑀦𑀻”𑀢𑀺𑁇

2081\. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑁆𑀧𑀤𑀸”𑀢𑀺? “𑀧𑀜𑁆𑀘𑀺𑀫𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀫𑁆𑀧𑀤𑀸𑁇 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀘? 𑀜𑀸𑀢𑀺𑀲𑀫𑁆𑀧𑀤𑀸, 𑀪𑁄𑀕𑀲𑀫𑁆𑀧𑀤𑀸, 𑀆𑀭𑁄𑀕𑁆𑀬𑀲𑀫𑁆𑀧𑀤𑀸, 𑀲𑀻𑀮𑀲𑀫𑁆𑀧𑀤𑀸, 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀫𑁆𑀧𑀤𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘 𑀲𑀫𑁆𑀧𑀤𑀸”𑀢𑀺𑁇

---

2082\. 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀲𑀢𑁆𑀢𑀫𑁄𑁇

2083\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2084\. _𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀘 𑀑𑀫𑀤𑁆𑀤𑀺,_  
_𑀅𑀧𑀭𑁂𑀳𑀺 𑀅𑀦𑀼𑀬𑁄𑀕𑁄;_  
_𑀆𑀧𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀅𑀧𑀭𑁂𑀳𑀺,_  
_𑀯𑁂𑀭𑀸 𑀯𑁂𑀭𑀫𑀡𑀻𑀧𑀺 𑀘;_  
_𑀩𑁆𑀬𑀲𑀦𑀁 𑀲𑀫𑁆𑀧𑀤𑀸 𑀘𑁂𑀯,_  
_𑀲𑀢𑁆𑀢𑀫𑁄 𑀯𑀕𑁆𑀕𑀲𑀗𑁆𑀕𑀳𑁄𑀢𑀺𑁇_
