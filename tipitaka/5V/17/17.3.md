# 𑁧𑁭𑁇𑁩 𑀯𑁄𑀳𑀸𑀭𑀯𑀕𑁆𑀕

* 𑀉𑀧𑀸𑀮𑀺𑀧𑀜𑁆𑀘𑀓

* 𑀯𑁄𑀳𑀸𑀭𑀯𑀕𑁆𑀕

(424.)

1977\. “𑀓𑀢𑀺𑀳𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺

1978\. “𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀬𑁄𑀕𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀽𑀧𑀲𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀬𑁄𑀕𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀽𑀧𑀲𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁧)

1979\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀧𑀬𑁄𑀕𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀦 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀧𑀬𑁄𑀕𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀥𑀺𑀓𑀭𑀡𑀲𑁆𑀲 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁨)

1980\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀧𑀲𑀬𑁆𑀳𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀅𑀦𑁄𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀦 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀧𑀲𑀬𑁆𑀳𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀑𑀓𑀸𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀯𑀢𑁆𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁂 𑀬𑀣𑀸𑀯𑀺𑀦𑀬𑁂 𑀬𑀣𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑁂𑀢𑀸 𑀳𑁄𑀢𑀺, 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁩)

1981\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀆𑀧𑀢𑁆𑀢𑀸𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀮𑀳𑀼𑀓𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀸𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁪)

1982\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀓𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀭𑀡𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁫)

1983\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀯𑀢𑁆𑀣𑀼𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀤𑀸𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀤𑀧𑀘𑁆𑀘𑀸𑀪𑀝𑁆𑀞𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀺𑀯𑀘𑀦𑀧𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀯𑀢𑁆𑀣𑀼𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀺𑀤𑀸𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀤𑀧𑀘𑁆𑀘𑀸𑀪𑀝𑁆𑀞𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀼𑀲𑀦𑁆𑀥𑀺𑀯𑀘𑀦𑀧𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁬)

1984\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀮𑀚𑁆𑀚𑀻 𑀘 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁭)

1985\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀅𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀯𑀺𑀦𑀬𑁂—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺, 𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺 𑀯𑀺𑀦𑀬𑁂—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁮)

1986\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀜𑀢𑁆𑀢𑀺𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀣𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀽𑀧𑀲𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀜𑀢𑁆𑀢𑀺𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀫𑀣𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀽𑀧𑀲𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁯)

1987\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀲𑀼𑀢𑁆𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀘 𑀞𑀸𑀦𑀸𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀲𑀼𑀢𑁆𑀢𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀼𑀢𑁆𑀢𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀞𑀸𑀦𑀸𑀞𑀸𑀦𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 (𑁧𑁦)

1988\. 𑀅𑀧𑀭𑁂𑀳𑀺𑀧𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦 𑀘 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀓𑀼𑀲𑀮𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀦 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀜𑁆𑀘𑀳𑀼𑀧𑀸𑀮𑀺, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀢𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀺? 𑀥𑀫𑁆𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀥𑀫𑁆𑀫𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀮𑁄𑀫𑀁 𑀚𑀸𑀦𑀸𑀢𑀺, 𑀧𑀼𑀩𑁆𑀩𑀸𑀧𑀭𑀓𑀼𑀲𑀮𑁄 𑀘 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑁂 𑀯𑁄𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 (𑁧𑁧)

---

1989\. 𑀯𑁄𑀳𑀸𑀭𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀢𑀢𑀺𑀬𑁄𑁇



---

1990\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1991\. _𑀆𑀧𑀢𑁆𑀢𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀁,_  
_𑀧𑀲𑀬𑁆𑀳𑀸𑀧𑀢𑁆𑀢𑀺 𑀚𑀸𑀦𑀦𑀸;_  
_𑀓𑀫𑁆𑀫𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀮𑀚𑁆𑀚𑀻 𑀘,_  
_𑀅𑀓𑀼𑀲𑀮𑁄 𑀘 𑀜𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀼𑀢𑁆𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀥𑀫𑁆𑀫𑀁,_  
_𑀢𑀢𑀺𑀬𑁄 𑀯𑀕𑁆𑀕𑀲𑀗𑁆𑀕𑀳𑁄𑀢𑀺𑁇_
