# 𑁧𑁯𑁇𑁪 𑀅𑀯𑀦𑁆𑀤𑀦𑀻𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀤𑀺

### 𑀤𑀼𑀢𑀺𑀬𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

### 𑀅𑀯𑀦𑁆𑀤𑀦𑀻𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀤𑀺

(477.)

2294. _𑀓𑀢𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦𑀸𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀲𑀸𑀫𑀺𑀘𑁂𑀦 𑀘;_  
_𑀓𑀢𑀺𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀳𑁄𑀢𑀺,_  
_𑀓𑀢𑀺 𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑀸𑁇_  


2295. _𑀤𑀲 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦𑀸𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀲𑀸𑀫𑀺𑀘𑁂𑀦 𑀘;_  
_𑀤𑀲𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀳𑁄𑀢𑀺,_  
_𑀤𑀲 𑀘𑀻𑀯𑀭𑀥𑀸𑀭𑀡𑀸𑁇_  


2296. _𑀓𑀢𑀺𑀦𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀤𑀸𑀢𑀩𑁆𑀩𑀁 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁;_  
_𑀓𑀢𑀺𑀦𑀁 𑀲𑀦𑁆𑀢𑁂 𑀤𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀓𑀢𑀺𑀦𑀜𑁆𑀘𑁂𑀯 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑀁𑁇_  


2297. _𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀤𑀸𑀢𑀩𑁆𑀩𑀁 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁;_  
_𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀦𑁆𑀢𑁂 𑀤𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀲𑁄𑀍𑀅𑀲𑀦𑁆𑀦𑀁 𑀦 𑀤𑀸𑀢𑀩𑁆𑀩𑀁𑁇_  


2298. _𑀓𑀢𑀺𑀲𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀲𑀢𑀁,_  
_𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀙𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸𑀦;_  
_𑀓𑀢𑀺 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀲𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑁄𑁇_  


2299. _𑀤𑀲𑀲𑀢𑀁 𑀭𑀢𑁆𑀢𑀺𑀲𑀢𑀁,_  
_𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀙𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸𑀦;_  
_𑀤𑀲 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀲𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀼𑀘𑁆𑀘𑁂𑀬𑁆𑀬 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑁄𑁇_  


2300. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀺𑁇_  


2301. _𑀤𑁆𑀯𑀸𑀤𑀲 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀸𑁇_  


2302. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀺𑁇_  


2303. _𑀘𑀢𑀲𑁆𑀲𑁄 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀩𑁆𑀩𑁂𑀯 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀸𑁇_  


2304. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀺𑁇_  


2305. _𑀙 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀏𑀓𑁂𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀸;_  
_𑀧𑀜𑁆𑀘 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2306. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀺𑁇_  


2307. _𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀯𑀺𑀦𑀬𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁,_  
_𑀏𑀓𑁂𑀢𑁆𑀣 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀓𑀢𑀸;_  
_𑀢𑀬𑁄 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2308. _𑀬𑀁 𑀤𑁂𑀲𑀺𑀢𑀁𑀦𑀦𑁆𑀢𑀚𑀺𑀦𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀸,_  
_𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀺 𑀯𑀺𑀯𑁂𑀓𑀤𑀲𑁆𑀲𑀺𑀦𑀸;_  
_𑀓𑀢𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀯𑀺𑀦𑀸 𑀲𑀫𑀣𑁂𑀳𑀺,_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀩𑁆𑀭𑀽𑀳𑀺 𑀯𑀺𑀪𑀗𑁆𑀕𑀓𑁄𑀯𑀺𑀤𑁇_  


2309. _𑀬𑀁 𑀤𑁂𑀲𑀺𑀢𑀁𑀦𑀦𑁆𑀢𑀚𑀺𑀦𑁂𑀦 𑀢𑀸𑀤𑀺𑀦𑀸,_  
_𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀺 𑀯𑀺𑀯𑁂𑀓𑀤𑀲𑁆𑀲𑀺𑀦𑀸;_  
_𑀏𑀓𑁂𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀢𑀺 𑀯𑀺𑀦𑀸 𑀲𑀫𑀣𑁂𑀳𑀺,_  
_𑀏𑀢𑀁 𑀢𑁂 𑀅𑀓𑁆𑀔𑀸𑀫𑀺 𑀯𑀺𑀪𑀗𑁆𑀕𑀓𑁄𑀯𑀺𑀤𑁇_  


2310. _𑀓𑀢𑀺 𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2311. _𑀙𑀊𑀦𑀤𑀺𑀬𑀟𑁆𑀠𑀲𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀆𑀧𑀸𑀬𑀺𑀓𑀸 𑀦𑁂𑀭𑀬𑀺𑀓𑀸,_  
_𑀓𑀧𑁆𑀧𑀝𑁆𑀞𑀸 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2312. _𑀓𑀢𑀺 𑀦𑀸𑀧𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2313. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀦𑀸𑀧𑀸𑀬𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2314. _𑀓𑀢𑀺 𑀅𑀝𑁆𑀞𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2315. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀅𑀝𑁆𑀞𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_
