# 𑁧𑁯𑁇𑁩 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬

* 𑀤𑀼𑀢𑀺𑀬𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

* 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬

(476.)

2262\. _𑀓𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁,_  
_𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀏𑀓𑀢𑁄𑁇_  


2263\. _𑀧𑀜𑁆𑀘 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁,_  
_𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀏𑀓𑀢𑁄𑁇_  


2264\. _𑀓𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁,_  
_𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀏𑀓𑀢𑁄𑁇_  


2265\. _𑀦𑀯 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀅𑀧𑀼𑀩𑁆𑀩𑀁 𑀅𑀘𑀭𑀺𑀫𑀁,_  
_𑀆𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬 𑀏𑀓𑀢𑁄𑁇_  


2266\. _𑀓𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀓𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2267\. _𑀧𑀜𑁆𑀘 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀏𑀓𑀯𑀸𑀘𑀸𑀬 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2268\. _𑀓𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀓𑀢𑀺 𑀯𑀸𑀘𑀸𑀬 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2269\. _𑀦𑀯 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀏𑀓𑀯𑀸𑀘𑀸𑀬 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2270\. _𑀓𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀓𑀺𑀜𑁆𑀘 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2271\. _𑀧𑀜𑁆𑀘 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀯𑀢𑁆𑀣𑀼𑀁 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2272\. _𑀓𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀓𑀺𑀜𑁆𑀘 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2273\. _𑀦𑀯 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀦𑀺;_  
_𑀯𑀢𑁆𑀣𑀼𑀁 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀤𑁂𑀲𑁂𑀬𑁆𑀬,_  
_𑀯𑀼𑀢𑁆𑀢𑀸 𑀆𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2274\. _𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑁂 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀯𑁄𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸;_  
_𑀔𑀸𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀪𑁄𑀚𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2275\. _𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑁂 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀙 𑀯𑁄𑀳𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸;_  
_𑀔𑀸𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀧𑀜𑁆𑀘 𑀪𑁄𑀚𑀦𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2276\. _𑀲𑀩𑁆𑀩𑀸 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸,_  
_𑀓𑀢𑀺 𑀞𑀸𑀦𑀸𑀦𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺;_  
_𑀓𑀢𑀺𑀦𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺,_  
_𑀓𑀢𑀺𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀘𑁇_  


2277\. _𑀲𑀩𑁆𑀩𑀸 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸,_  
_𑀧𑀜𑁆𑀘 𑀞𑀸𑀦𑀸𑀦𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺;_  
_𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀜𑁆𑀘𑁂𑀯 𑀆𑀧𑀢𑁆𑀢𑀺,_  
_𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑁂𑀦 𑀘𑁇_  


2278\. _𑀓𑀢𑀺𑀦𑀁 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑁄 𑀳𑁄𑀢𑀺,_  
_𑀓𑀢𑀺𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁂𑀦 𑀘;_  
_𑀓𑀢𑀺𑀦𑀜𑁆𑀘𑁂𑀯 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺,_  
_𑀓𑀢𑀺𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑁄𑀪𑀢𑀺𑁇_  


2279\. _𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑁄 𑀳𑁄𑀢𑀺,_  
_𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀯𑀽𑀧𑀲𑀫𑁂𑀦 𑀘;_  
_𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀜𑁆𑀘𑁂𑀯 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺,_  
_𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑁄𑀪𑀢𑀺𑁇_  


2280\. _𑀓𑀢𑀺 𑀓𑀸𑀬𑀺𑀓𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀓𑀢𑀺 𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑀺𑀯𑀸;_  
_𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺,_  
_𑀓𑀢𑀺 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2281\. _𑀤𑁆𑀯𑁂 𑀓𑀸𑀬𑀺𑀓𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀤𑁆𑀯𑁂 𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑀺𑀯𑀸;_  
_𑀦𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀏𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀺,_  
_𑀏𑀓𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2282\. _𑀓𑀢𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁,_  
_𑀧𑀭𑁂𑀲𑀁 𑀲𑀤𑁆𑀥𑀸𑀬 𑀤𑁂𑀲𑀬𑁂;_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀸 𑀓𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀺 𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑀸𑁇_  


2283\. _𑀅𑀝𑁆𑀞𑀸𑀦𑀺𑀲𑀁𑀲𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁,_  
_𑀧𑀭𑁂𑀲𑀁 𑀲𑀤𑁆𑀥𑀸𑀬 𑀤𑁂𑀲𑀬𑁂;_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀸 𑀢𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀢𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲 𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑀸𑁇_  


2284\. _𑀓𑀢𑀺 𑀞𑀸𑀦𑁂 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄,_  
_𑀓𑀢𑀺 𑀧𑀭𑀫𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀓𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸,_  
_𑀓𑀢𑀺𑀦𑀁 𑀤𑁂𑀲𑀦𑀸𑀬 𑀘𑁇_  


2285\. _𑀧𑀜𑁆𑀘 𑀞𑀸𑀦𑁂 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄,_  
_𑀘𑀼𑀤𑁆𑀤𑀲 𑀧𑀭𑀫𑀦𑁆𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀤𑁆𑀯𑀸𑀤𑀲 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸,_  
_𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀤𑁂𑀲𑀦𑀸𑀬 𑀘𑁇_  


2286\. _𑀓𑀢𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄,_  
_𑀓𑀢𑀺 𑀉𑀧𑁄𑀲𑀣𑀗𑁆𑀕𑀸𑀦𑀺;_  
_𑀓𑀢𑀺 𑀤𑀽𑀢𑁂𑀬𑁆𑀬𑀗𑁆𑀕𑀸𑀦𑀺,_  
_𑀓𑀢𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀯𑀢𑁆𑀢𑀦𑀸𑁇_  


2287\. _𑀅𑀝𑁆𑀞𑀗𑁆𑀕𑀺𑀓𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄,_  
_𑀅𑀝𑁆𑀞 𑀉𑀧𑁄𑀲𑀣𑀗𑁆𑀕𑀸𑀦𑀺;_  
_𑀅𑀝𑁆𑀞 𑀤𑀽𑀢𑁂𑀬𑁆𑀬𑀗𑁆𑀕𑀸𑀦𑀺,_  
_𑀅𑀝𑁆𑀞 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀯𑀢𑁆𑀢𑀦𑀸𑁇_  


2288\. _𑀓𑀢𑀺𑀯𑀸𑀘𑀺𑀓𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸,_  
_𑀓𑀢𑀺𑀦𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑀁;_  
_𑀓𑀢𑀺𑀦𑀁 𑀆𑀲𑀦𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑁄 𑀓𑀢𑀺𑀳𑀺𑁇_  


2289\. _𑀅𑀝𑁆𑀞𑀯𑀸𑀘𑀺𑀓𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸,_  
_𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑀁;_  
_𑀅𑀝𑁆𑀞𑀦𑁆𑀦𑀁 𑀆𑀲𑀦𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀁,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑁄 𑀅𑀝𑁆𑀞𑀳𑀺𑁇_  


2290\. _𑀓𑀢𑀺𑀦𑀁 𑀙𑁂𑀚𑁆𑀚𑀁 𑀳𑁄𑀢𑀺,_  
_𑀓𑀢𑀺𑀦𑀁 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁;_  
_𑀓𑀢𑀺𑀦𑀜𑁆𑀘𑁂𑀯 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀏𑀓𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑁇_  


2291\. _𑀏𑀓𑀲𑁆𑀲 𑀙𑁂𑀚𑁆𑀚𑀁 𑀳𑁄𑀢𑀺,_  
_𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁;_  
_𑀘𑀢𑀼𑀦𑁆𑀦𑀜𑁆𑀘𑁂𑀯 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀏𑀓𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑁇_  


2292\. _𑀓𑀢𑀺 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺,_  
_𑀓𑀢𑀺𑀳𑀺 𑀲𑀁𑀖𑁄 𑀪𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀓𑀢𑁂𑀢𑁆𑀣 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀸 𑀓𑀢𑀺𑁇_  


2293\. _𑀦𑀯 𑀆𑀖𑀸𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺,_  
_𑀦𑀯𑀳𑀺 𑀲𑀁𑀖𑁄 𑀪𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀦𑀯𑁂𑀢𑁆𑀣 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀸 𑀦𑀯𑁇_
