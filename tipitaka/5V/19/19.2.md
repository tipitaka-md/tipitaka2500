# 𑁧𑁯𑁇𑁨 𑀤𑁂𑀲𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸𑀤𑀺𑀆𑀧𑀢𑁆𑀢𑀺

### 𑀤𑀼𑀢𑀺𑀬𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

### 𑀤𑁂𑀲𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸𑀤𑀺𑀆𑀧𑀢𑁆𑀢𑀺

(475.)

2242. _𑀓𑀢𑀺 𑀤𑁂𑀲𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀲𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸 𑀓𑀢𑀸;_  
_𑀓𑀢𑁂𑀢𑁆𑀣 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2243. _𑀧𑀜𑁆𑀘 𑀤𑁂𑀲𑀦𑀸𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑁄,_  
_𑀙 𑀲𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸 𑀓𑀢𑀸;_  
_𑀏𑀓𑁂𑀢𑁆𑀣 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸𑁇_  


2244. _𑀯𑀺𑀦𑀬𑀕𑀭𑀼𑀓𑀸 𑀓𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀓𑀸𑀬𑀯𑀸𑀘𑀲𑀺𑀓𑀸𑀦𑀺 𑀘;_  
_𑀓𑀢𑀺 𑀯𑀺𑀓𑀸𑀮𑁂 𑀥𑀜𑁆𑀜𑀭𑀲𑁄,_  
_𑀓𑀢𑀺 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑁇_  


2245. _𑀯𑀺𑀦𑀬𑀕𑀭𑀼𑀓𑀸 𑀤𑁆𑀯𑁂 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀓𑀸𑀬𑀯𑀸𑀘𑀲𑀺𑀓𑀸𑀦𑀺 𑀘;_  
_𑀏𑀓𑁄 𑀯𑀺𑀓𑀸𑀮𑁂 𑀥𑀜𑁆𑀜𑀭𑀲𑁄,_  
_𑀏𑀓𑀸 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑁇_  


2246. _𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀓𑀸𑀬𑀺𑀓𑀸 𑀓𑀢𑀺,_  
_𑀓𑀢𑀺 𑀲𑀁𑀯𑀸𑀲𑀓𑀪𑀽𑀫𑀺𑀬𑁄;_  
_𑀓𑀢𑀺𑀦𑀁 𑀭𑀢𑁆𑀢𑀺𑀘𑁆𑀙𑁂𑀤𑁄,_  
_𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀸 𑀓𑀢𑀺𑁇_  


2247. _𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀓𑀸𑀬𑀺𑀓𑀸 𑀤𑁆𑀯𑁂,_  
_𑀤𑁆𑀯𑁂 𑀲𑀁𑀯𑀸𑀲𑀓𑀪𑀽𑀫𑀺𑀬𑁄;_  
_𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀭𑀢𑁆𑀢𑀺𑀘𑁆𑀙𑁂𑀤𑁄,_  
_𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸 𑀤𑁆𑀯𑀗𑁆𑀕𑀼𑀮𑀸 𑀤𑀼𑀯𑁂𑁇_  


2248. _𑀓𑀢𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑀺𑀳𑀺 𑀲𑀁𑀖𑁄 𑀪𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀓𑀢𑁂𑀢𑁆𑀣 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀸 𑀓𑀢𑀺𑁇_  


2249. _𑀤𑁆𑀯𑁂 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀯𑀥𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀁𑀖𑁄 𑀪𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀤𑁆𑀯𑁂𑀢𑁆𑀣 𑀧𑀞𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸,_  
_𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀭𑀡𑀸 𑀤𑀼𑀯𑁂𑁇_  


2250. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁂 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀯𑀸𑀘𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀓𑀢𑀺;_  
_𑀑𑀪𑀸𑀲𑀦𑀸 𑀓𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀓𑀢𑀺𑁇_  


2251. _𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁂 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀯𑀸𑀘𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀢𑀬𑁄;_  
_𑀑𑀪𑀸𑀲𑀦𑀸 𑀢𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀢𑀬𑁄𑁇_  


2252. _𑀓𑀢𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸,_  
_𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀗𑁆𑀕𑀳𑀸;_  
_𑀦𑀸𑀲𑀺𑀢𑀓𑀸 𑀓𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀺𑀦𑀁 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀸𑁇_  


2253. _𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑀸,_  
_𑀢𑀬𑁄 𑀓𑀫𑁆𑀫𑀸𑀦𑀁 𑀲𑀗𑁆𑀕𑀳𑀸;_  
_𑀦𑀸𑀲𑀺𑀢𑀓𑀸 𑀢𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀢𑀺𑀡𑁆𑀡𑀦𑁆𑀦𑀁 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀸𑁇_  


2254. _𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑁂 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀫𑁂𑀣𑀼𑀦𑀧𑀘𑁆𑀘𑀬𑀸;_  
_𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀙𑀟𑁆𑀟𑀺𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2255. _𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀦𑁂 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀘𑀢𑀲𑁆𑀲𑁄 𑀫𑁂𑀣𑀼𑀦𑀧𑀘𑁆𑀘𑀬𑀸;_  
_𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀧𑀜𑁆𑀘 𑀙𑀟𑁆𑀟𑀺𑀢𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2256. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑀯𑀕𑁆𑀕𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀝𑀸;_  
_𑀓𑀢𑁂𑀢𑁆𑀣 𑀦𑀯𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀓𑀢𑀺𑀦𑀁 𑀘𑀻𑀯𑀭𑁂𑀦 𑀘𑁇_  


2257. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑀯𑀸𑀤𑀓𑀯𑀕𑁆𑀕𑀲𑁆𑀫𑀺𑀁,_  
_𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀝𑀸 𑀓𑀢𑀸;_  
_𑀘𑀢𑀼𑀭𑁂𑀢𑁆𑀣 𑀦𑀯𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀘𑀻𑀯𑀭𑁂𑀦 𑀘𑁇_  


2258. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘 𑀅𑀓𑁆𑀔𑀸𑀢𑀸,_  
_𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀺𑀬𑀸 𑀓𑀢𑀺;_  
_𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸𑀫𑀓𑀥𑀜𑁆𑀜𑁂𑀦,_  
_𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀝𑀸 𑀓𑀢𑀺𑁇_  


2259. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘 𑀅𑀓𑁆𑀔𑀸𑀢𑀸,_  
_𑀅𑀝𑁆𑀞 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸 𑀓𑀢𑀸;_  
_𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸𑀫𑀓𑀥𑀜𑁆𑀜𑁂𑀦,_  
_𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦 𑀤𑀼𑀓𑁆𑀓𑀝𑀸 𑀓𑀢𑀸𑁇_  


2260. _𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀞𑀺𑀢𑀲𑁆𑀲 𑀘𑀸𑀧𑀺 𑀓𑀺𑀢𑁆𑀢𑀓𑀸;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀦𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀓𑀺𑀢𑁆𑀢𑀓𑀸𑁇_  


2261. _𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀲𑁆𑀲 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀞𑀺𑀢𑀲𑁆𑀲 𑀘𑀸𑀧𑀺 𑀢𑀢𑁆𑀢𑀓𑀸;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀦𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲𑀸𑀧𑀺 𑀢𑀢𑁆𑀢𑀓𑀸𑁇_
