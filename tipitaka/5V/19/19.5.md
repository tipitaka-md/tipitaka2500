# 𑁧𑁯𑁇𑁫 𑀲𑁄𑀍𑀅𑀲𑀓𑀫𑁆𑀫𑀸𑀤𑀺

* 𑀤𑀼𑀢𑀺𑀬𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

* 𑀲𑁄𑀍𑀅𑀲𑀓𑀫𑁆𑀫𑀸𑀤𑀺

(478.)

2316\. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2317\. _𑀲𑁄𑀍𑀅𑀲 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2318\. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2319\. _𑀤𑁆𑀯𑀸𑀤𑀲 𑀓𑀫𑁆𑀫𑀤𑁄𑀲𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2320\. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2321\. _𑀘𑀢𑀲𑁆𑀲𑁄 𑀓𑀫𑁆𑀫𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2322\. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2323\. _𑀙 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2324\. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2325\. _𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2326\. _𑀓𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_

_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2327\. _𑀅𑀝𑁆𑀞 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2328\. _𑀓𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2329\. _𑀢𑁂𑀯𑀻𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2330\. _𑀓𑀢𑀺 𑀅𑀦𑀺𑀬𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_

2331\. _𑀤𑁆𑀯𑁂 𑀅𑀦𑀺𑀬𑀢𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2332\. _𑀓𑀢𑀺 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2333\. _𑀤𑁆𑀯𑁂𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2334\. _𑀓𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_  
_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2335\. _𑀅𑀝𑁆𑀞𑀸𑀲𑀻𑀢𑀺𑀲𑀢𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_

_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂𑁇_  


2336\. _𑀓𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_

_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2337\. _𑀤𑁆𑀯𑀸𑀤𑀲 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_

2338\. _𑀓𑀢𑀺 𑀲𑁂𑀔𑀺𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_

_𑀯𑀺𑀦𑀬𑀸𑀦𑀺 𑀲𑀼𑀡𑁄𑀫 𑀢𑁂𑁇_  


2339\. _𑀧𑀜𑁆𑀘𑀲𑀢𑁆𑀢𑀢𑀺 𑀲𑁂𑀔𑀺𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀩𑀼𑀤𑁆𑀥𑁂𑀦𑀸𑀤𑀺𑀘𑁆𑀘𑀩𑀦𑁆𑀥𑀼𑀦𑀸;_  
_𑀯𑀺𑀦𑀬𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲,_

2340\. _𑀬𑀸𑀯 𑀲𑀼𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀁 𑀢𑀬𑀸,_
