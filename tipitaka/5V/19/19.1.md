# 𑁧𑁯𑁇𑁧 𑀓𑀸𑀬𑀺𑀓𑀸𑀤𑀺𑀆𑀧𑀢𑁆𑀢𑀺

### 𑀤𑀼𑀢𑀺𑀬𑀕𑀸𑀣𑀸𑀲𑀗𑁆𑀕𑀡𑀺𑀓

### 𑀓𑀸𑀬𑀺𑀓𑀸𑀤𑀺𑀆𑀧𑀢𑁆𑀢𑀺

(474.)

2232. _𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀓𑀸𑀬𑀺𑀓𑀸,_  
_𑀓𑀢𑀺 𑀯𑀸𑀘𑀲𑀺𑀓𑀸 𑀓𑀢𑀸;_  
_𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀲𑀁𑀲𑀕𑁆𑀕𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2233. _𑀙𑀸𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀓𑀸𑀬𑀺𑀓𑀸,_  
_𑀙 𑀯𑀸𑀘𑀲𑀺𑀓𑀸 𑀓𑀢𑀸;_  
_𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀧𑀜𑁆𑀘 𑀲𑀁𑀲𑀕𑁆𑀕𑀧𑀘𑁆𑀘𑀬𑀸𑁇_  


2234. _𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑁂 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸;_  
_𑀓𑀢𑁂𑀢𑁆𑀣 𑀅𑀝𑁆𑀞 𑀯𑀢𑁆𑀣𑀼𑀓𑀸,_  
_𑀓𑀢𑀺𑀳𑀺 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀕𑀳𑁄𑁇_  


2235. _𑀅𑀭𑀼𑀡𑀼𑀕𑁆𑀕𑁂 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀤𑁆𑀯𑁂 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀓𑀸;_  
_𑀏𑀓𑁂𑀢𑁆𑀣 𑀅𑀝𑁆𑀞 𑀯𑀢𑁆𑀣𑀼𑀓𑀸,_  
_𑀏𑀓𑁂𑀦 𑀲𑀩𑁆𑀩𑀲𑀗𑁆𑀕𑀳𑁄𑁇_  


2236. _𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀓𑀢𑀺 𑀫𑀽𑀮𑀸𑀦𑀺,_  
_𑀬𑀸𑀦𑀺 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸;_  
_𑀯𑀺𑀦𑀬𑀕𑀭𑀼𑀓𑀸 𑀓𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀘𑁆𑀙𑀸𑀤𑀦𑀸 𑀓𑀢𑀺𑁇_  


2237. _𑀯𑀺𑀦𑀬𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀫𑀽𑀮𑀸𑀦𑀺,_  
_𑀬𑀸𑀦𑀺 𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸;_  
_𑀯𑀺𑀦𑀬𑀕𑀭𑀼𑀓𑀸 𑀤𑁆𑀯𑁂 𑀯𑀼𑀢𑁆𑀢𑀸,_  
_𑀤𑁆𑀯𑁂 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀘𑁆𑀙𑀸𑀤𑀦𑀸𑁇_  


2238. _𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀓𑀢𑀺 𑀦𑀤𑀺𑀧𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸;_  
_𑀓𑀢𑀺𑀫𑀁𑀲𑁂𑀲𑀼 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁,_  
_𑀓𑀢𑀺𑀫𑀁𑀲𑁂𑀲𑀼 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇_  


2239. _𑀕𑀸𑀫𑀦𑁆𑀢𑀭𑁂 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀘𑀢𑀲𑁆𑀲𑁄 𑀦𑀤𑀺𑀧𑀸𑀭𑀧𑀘𑁆𑀘𑀬𑀸;_  
_𑀏𑀓𑀫𑀁𑀲𑁂 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁,_  
_𑀦𑀯𑀫𑀁𑀲𑁂𑀲𑀼 𑀤𑀼𑀓𑁆𑀓𑀝𑀁𑁇_  


2240. _𑀓𑀢𑀺 𑀯𑀸𑀘𑀲𑀺𑀓𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀓𑀢𑀺 𑀯𑀸𑀘𑀲𑀺𑀓𑀸 𑀤𑀺𑀯𑀸;_  
_𑀤𑀤𑀫𑀸𑀦𑀲𑁆𑀲 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀺𑀢𑁆𑀢𑀓𑀸𑁇_  


2241. _𑀤𑁆𑀯𑁂 𑀯𑀸𑀘𑀲𑀺𑀓𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀤𑁆𑀯𑁂 𑀯𑀸𑀘𑀲𑀺𑀓𑀸 𑀤𑀺𑀯𑀸;_  
_𑀤𑀤𑀫𑀸𑀦𑀲𑁆𑀲 𑀢𑀺𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑁇_
