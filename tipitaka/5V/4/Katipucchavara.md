# 𑀓𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀸𑀭

### 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮

### 𑀓𑀢𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀸𑀭

(271.)

1152. 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄? 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸? 𑀓𑀢𑀺 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀓𑀢𑀺 𑀅𑀕𑀸𑀭𑀯𑀸? 𑀓𑀢𑀺 𑀕𑀸𑀭𑀯𑀸? 𑀓𑀢𑀺 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄? 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸? 𑀓𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀸𑀦𑀺? 𑀓𑀢𑀺 𑀅𑀦𑀼𑀯𑀸𑀤𑀫𑀽𑀮𑀸𑀦𑀺? 𑀓𑀢𑀺 𑀲𑀸𑀭𑀡𑀻𑀬𑀸 𑀥𑀫𑁆𑀫𑀸? 𑀓𑀢𑀺 𑀪𑁂𑀤𑀓𑀭𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀓𑀢𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀺? 𑀓𑀢𑀺 𑀲𑀫𑀣𑀸?

1153. 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 𑀧𑀜𑁆𑀘 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 𑀲𑀢𑁆𑀢 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀙 𑀅𑀕𑀸𑀭𑀯𑀸𑁇 𑀙 𑀕𑀸𑀭𑀯𑀸𑁇 𑀙 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑁇 𑀙 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀙 𑀅𑀦𑀼𑀯𑀸𑀤𑀫𑀽𑀮𑀸𑀦𑀺𑁇 𑀙 𑀲𑀸𑀭𑀡𑀻𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀪𑁂𑀤𑀓𑀭𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀺𑁇 𑀲𑀢𑁆𑀢 𑀲𑀫𑀣𑀸𑁇

1154. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺, 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺, 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺—  𑀇𑀫𑀸 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 (𑁧)

1155. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 (𑁨)

1156. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀆𑀭𑀢𑀺 𑀯𑀺𑀭𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀺 𑀯𑁂𑀭𑀫𑀡𑀻 𑀅𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀓𑀭𑀡𑀁 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑁂𑀮𑀸𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄—  𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 (𑁩)

1157. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺, 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺, 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺, 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺, 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸𑀧𑀢𑁆𑀢𑀺—  𑀇𑀫𑀸 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 (𑁪)

1158. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸? 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄—  𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇 (𑁫)

1159. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀲𑀢𑁆𑀢𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀆𑀭𑀢𑀺 𑀯𑀺𑀭𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀺 𑀯𑁂𑀭𑀫𑀡𑀻 𑀅𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀓𑀭𑀡𑀁 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑁂𑀮𑀸𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄—  𑀇𑀫𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 (𑁬)

1160. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀙 𑀅𑀕𑀸𑀭𑀯𑀸? 𑀩𑀼𑀤𑁆𑀥𑁂 𑀅𑀕𑀸𑀭𑀯𑁄, 𑀥𑀫𑁆𑀫𑁂 𑀅𑀕𑀸𑀭𑀯𑁄, 𑀲𑀁𑀖𑁂 𑀅𑀕𑀸𑀭𑀯𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀅𑀕𑀸𑀭𑀯𑁄, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂 𑀅𑀕𑀸𑀭𑀯𑁄, 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀸𑀭𑁂 𑀅𑀕𑀸𑀭𑀯𑁄—  𑀇𑀫𑁂 𑀙 𑀅𑀕𑀸𑀭𑀯𑀸𑁇 (𑁭)

1161. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀙 𑀕𑀸𑀭𑀯𑀸? 𑀩𑀼𑀤𑁆𑀥𑁂 𑀕𑀸𑀭𑀯𑁄, 𑀥𑀫𑁆𑀫𑁂 𑀕𑀸𑀭𑀯𑁄, 𑀲𑀁𑀖𑁂 𑀕𑀸𑀭𑀯𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀕𑀸𑀭𑀯𑁄, 𑀅𑀧𑁆𑀧𑀫𑀸𑀤𑁂 𑀕𑀸𑀭𑀯𑁄, 𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀸𑀭𑁂 𑀕𑀸𑀭𑀯𑁄—  𑀇𑀫𑁂 𑀙 𑀕𑀸𑀭𑀯𑀸𑁇 (𑁮)

1162. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀙 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀙𑀳𑀺 𑀅𑀕𑀸𑀭𑀯𑁂𑀳𑀺 𑀆𑀭𑀢𑀺 𑀯𑀺𑀭𑀢𑀺 𑀧𑀝𑀺𑀯𑀺𑀭𑀢𑀺 𑀯𑁂𑀭𑀫𑀡𑀻 𑀅𑀓𑀺𑀭𑀺𑀬𑀸 𑀅𑀓𑀭𑀡𑀁 𑀅𑀦𑀚𑁆𑀛𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑁂𑀮𑀸𑀅𑀦𑀢𑀺𑀓𑁆𑀓𑀫𑁄 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄—  𑀇𑀫𑀸𑀦𑀺 𑀙 𑀯𑀺𑀦𑀻𑀢𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 (𑁯)

1163. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄? 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺, 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺, 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀢𑁆𑀢𑀺—  𑀇𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄𑁇 (𑁧𑁦)

1165. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀙 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀸𑀦𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀥𑀦𑁄 𑀳𑁄𑀢𑀺 𑀉𑀧𑀦𑀸𑀳𑀻𑁇 𑀬𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀥𑀦𑁄 𑀳𑁄𑀢𑀺 𑀉𑀧𑀦𑀸𑀳𑀻 𑀲𑁄 𑀲𑀢𑁆𑀣𑀭𑀺𑀧𑀺 𑀅𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂𑀧𑀺 𑀅𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀲𑀁𑀖𑁂𑀧𑀺 𑀅𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀲𑀺𑀓𑁆𑀔𑀸𑀬𑀧𑀺 𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻 𑀳𑁄𑀢𑀺𑁇 𑀬𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀣𑀭𑀺 𑀅𑀕𑀸𑀭𑀯𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄, 𑀥𑀫𑁆𑀫𑁂…𑀧𑁂…  𑀲𑀁𑀖𑁂…𑀧𑁂…  𑀲𑀺𑀓𑁆𑀔𑀸𑀬 𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑀓𑀸𑀭𑀻, 𑀲𑁄 𑀲𑀁𑀖𑁂 𑀯𑀺𑀯𑀸𑀤𑀁 𑀚𑀦𑁂𑀢𑀺𑁇 𑀬𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀯𑀸𑀤𑁄 𑀩𑀳𑀼𑀚𑀦𑀸𑀳𑀺𑀢𑀸𑀬 𑀩𑀳𑀼𑀚𑀦𑀸𑀲𑀼𑀔𑀸𑀬 𑀩𑀳𑀼𑀦𑁄 𑀚𑀦𑀲𑁆𑀲 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀅𑀳𑀺𑀢𑀸𑀬 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁𑁇 𑀏𑀯𑀭𑀽𑀧𑀜𑁆𑀘𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀣 𑀢𑀢𑁆𑀭 𑀢𑀼𑀫𑁆𑀳𑁂 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀸𑀧𑀓𑀲𑁆𑀲 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀸𑀬 𑀯𑀸𑀬𑀫𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀏𑀯𑀭𑀽𑀧𑀜𑁆𑀘𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀁 𑀅𑀚𑁆𑀛𑀢𑁆𑀢𑀁 𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀥𑀸 𑀯𑀸 𑀦 𑀲𑀫𑀦𑀼𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀣 𑀢𑀢𑁆𑀭 𑀢𑀼𑀫𑁆𑀳𑁂 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀸𑀧𑀓𑀲𑁆𑀲 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀯𑀸𑀬 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑀲𑁆𑀲 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀲𑁆𑀲 𑀧𑀳𑀸𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀏𑀯𑀫𑁂𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑀲𑁆𑀲 𑀯𑀺𑀯𑀸𑀤𑀫𑀽𑀮𑀲𑁆𑀲 𑀆𑀬𑀢𑀺𑀁 𑀅𑀦𑀯𑀲𑁆𑀲𑀯𑁄 𑀳𑁄𑀢𑀺𑁇

1172. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂 𑀢𑁂 𑀮𑀸𑀪𑀸 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀸 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀧𑀢𑁆𑀢𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀳𑀺 𑀮𑀸𑀪𑁂𑀳𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑀺𑀪𑀢𑁆𑀢𑀪𑁄𑀕𑀻 𑀳𑁄𑀢𑀺 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀲𑀸𑀥𑀸𑀭𑀡𑀪𑁄𑀕𑀻𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

1173. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀲𑀻𑀮𑀸𑀦𑀺 𑀅𑀔𑀡𑁆𑀟𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀺𑀤𑁆𑀤𑀸𑀦𑀺 𑀅𑀲𑀩𑀮𑀸𑀦𑀺 𑀅𑀓𑀫𑁆𑀫𑀸𑀲𑀸𑀦𑀺 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑀸𑀦𑀺 𑀯𑀺𑀜𑁆𑀜𑀼𑀧𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀧𑀭𑀸𑀫𑀝𑁆𑀞𑀸𑀦𑀺 𑀲𑀫𑀸𑀥𑀺𑀲𑀁𑀯𑀢𑁆𑀢𑀦𑀺𑀓𑀸𑀦𑀺, 𑀢𑀣𑀸𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀻𑀮𑀲𑀸𑀫𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇

1174. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀺 𑀅𑀭𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀦𑀺𑀓𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀢𑀺 𑀢𑀓𑁆𑀓𑀭𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀓𑁆𑀔𑀬𑀸𑀬 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀲𑀸𑀫𑀜𑁆𑀜𑀕𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀳𑀺 𑀆𑀯𑀺 𑀘𑁂𑀯 𑀭𑀳𑁄 𑀘𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑁄 𑀲𑀸𑀭𑀡𑀻𑀬𑁄 𑀧𑀺𑀬𑀓𑀭𑀡𑁄 𑀕𑀭𑀼𑀓𑀭𑀡𑁄 𑀲𑀗𑁆𑀕𑀳𑀸𑀬 𑀅𑀯𑀺𑀯𑀸𑀤𑀸𑀬 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀏𑀓𑀻𑀪𑀸𑀯𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀇𑀫𑁂 𑀙 𑀲𑀸𑀭𑀡𑀻𑀬𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 (𑁧𑁪)

(275.)

1175. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀪𑁂𑀤𑀓𑀭𑀯𑀢𑁆𑀣𑀽𑀦𑀺? 𑀇𑀥 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀥𑀫𑁆𑀫𑀁 “𑀥𑀫𑁆𑀫𑁄”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀁 “𑀅𑀥𑀫𑁆𑀫𑁄”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀯𑀺𑀦𑀬𑀁 “𑀯𑀺𑀦𑀬𑁄”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀯𑀺𑀦𑀬𑀁 “𑀅𑀯𑀺𑀦𑀬𑁄”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 “𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀪𑀸𑀲𑀺𑀢𑀁 𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 “𑀅𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀮𑀧𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 “𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 “𑀅𑀦𑀸𑀘𑀺𑀡𑁆𑀡𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 “𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 “𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦𑀸”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 “𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀁 “𑀆𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀮𑀳𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 “𑀕𑀭𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀕𑀭𑀼𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 “𑀮𑀳𑀼𑀓𑀸 𑀆𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀲𑀸𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 “𑀅𑀦𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀦𑀯𑀲𑁂𑀲𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 “𑀲𑀸𑀯𑀲𑁂𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 “𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺, 𑀅𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 “𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀸 𑀆𑀧𑀢𑁆𑀢𑀻”𑀢𑀺 𑀤𑀻𑀧𑁂𑀢𑀺𑁇 𑀇𑀫𑀸𑀦𑀺 𑀅𑀝𑁆𑀞𑀸𑀭𑀲 𑀪𑁂𑀤𑀓𑀭𑀯𑀢𑁆𑀣𑀽𑀦𑀺𑁇 (𑁧𑁫)

1176. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀺? 𑀯𑀺𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀅𑀦𑀼𑀯𑀸𑀤𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁, 𑀓𑀺𑀘𑁆𑀘𑀸𑀥𑀺𑀓𑀭𑀡𑀁—  𑀇𑀫𑀸𑀦𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀺𑁇 (𑁧𑁬)

1177. 𑀢𑀢𑁆𑀣 𑀓𑀢𑀫𑁂 𑀲𑀢𑁆𑀢 𑀲𑀫𑀣𑀸? 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁄, 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑁄, 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑁄, 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑀁, 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑀲𑀺𑀓𑀸, 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸, 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁄—  𑀇𑀫𑁂 𑀲𑀢𑁆𑀢 𑀲𑀫𑀣𑀸𑁇 (𑁧𑁭)
