# 𑁪𑁇𑁪 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭

### 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮

### 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀯𑀸𑀭

(284.)

1215. 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀚𑀸𑀦𑀁 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀅𑀢𑁆𑀢𑀦𑁄 𑀤𑀼𑀝𑁆𑀞𑀼𑀮𑁆𑀮𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1216. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁧)

(285.)

1217. 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀁 𑀏𑀓𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1218. 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺? 𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀏𑀓𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀪𑀚𑀢𑀺—  𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁨)

1219. 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀲𑀫𑀦𑀼𑀪𑀸𑀲𑀦𑀸𑀬 𑀦 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀢𑀺, 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲—  𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀸 𑀤𑁆𑀯𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1220. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀏𑀓𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀤𑁆𑀯𑀻𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁩)

1221. 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀅𑀲𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀲𑀜𑁆𑀘𑀭𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 “𑀬𑁄 𑀢𑁂 𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀲𑀢𑀺, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀳𑀸”𑀢𑀺 𑀪𑀡𑀢𑀺, 𑀧𑀝𑀺𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲; 𑀆𑀚𑀻𑀯𑀳𑁂𑀢𑀼 𑀆𑀚𑀻𑀯𑀓𑀸𑀭𑀡𑀸 𑀲𑀽𑀧𑀁 𑀯𑀸 𑀑𑀤𑀦𑀁 𑀯𑀸 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲—  𑀆𑀚𑀻𑀯𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑀸 𑀇𑀫𑀸 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1222. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀙𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀙𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄; 𑀲𑀺𑀬𑀸 𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇 (𑁪)
