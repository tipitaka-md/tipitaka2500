# 𑁪𑁇𑁨 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀅𑀦𑁆𑀢𑀭𑀧𑁂𑀬𑁆𑀬𑀸𑀮

### 𑀓𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀸𑀭

(277.)

1189\. 𑀧𑀞𑀫𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀧𑀞𑀫𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀲𑀜𑁆𑀜𑀸𑀘𑀺𑀓𑀸𑀬 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀢𑀺 𑀅𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀅𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀏𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀁 𑀅𑀦𑀸𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑁂 𑀆𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲; 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀓𑀸𑀮𑁂 𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲; 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀸𑀬 𑀳𑀢𑁆𑀣𑀢𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲—  𑀧𑀞𑀫𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀇𑀫𑀸 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1190\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

(278.)

1191\. 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀤𑀺𑀲𑀢𑀺—  “𑀓𑀼𑀝𑀺𑀁 𑀫𑁂 𑀓𑀭𑁄𑀣𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀅𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀅𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁𑁇 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀏𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀁 𑀅𑀦𑀸𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑁂 𑀆𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀤𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀯𑀸𑀘𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲—  𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀇𑀫𑀸 𑀘𑀢𑀲𑁆𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1192\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀘𑀢𑀽𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

(279.)

1193\. 𑀢𑀢𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀢𑀢𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀢𑀺 𑀅𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀅𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁𑁇 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀏𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀁 𑀅𑀦𑀸𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑁂 𑀆𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲—  𑀢𑀢𑀺𑀬𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀇𑀫𑀸 𑀧𑀜𑁆𑀘 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1194\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀺𑀢𑁆𑀢𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

(280.)

1195\. 𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀲𑀜𑁆𑀜𑀸𑀘𑀺𑀓𑀸𑀬 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀢𑀺 𑀅𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀅𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀏𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀁 𑀅𑀦𑀸𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑁂 𑀆𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀯𑀺𑀓𑀸𑀮𑁂 𑀪𑁄𑀚𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀅𑀜𑁆𑀜𑀸𑀢𑀺𑀓𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀸𑀬 𑀳𑀢𑁆𑀣𑀢𑁄 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀯𑀸 𑀲𑀳𑀢𑁆𑀣𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀇𑀫𑀸 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1196\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀙𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

1197\. 𑀧𑀜𑁆𑀘𑀫𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀧𑀜𑁆𑀘𑀫𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀅𑀲𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀲𑀫𑀸𑀤𑀺𑀲𑀢𑀺—  “𑀓𑀼𑀝𑀺𑀁 𑀫𑁂 𑀓𑀭𑁄𑀣𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀅𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀅𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁𑁇 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀏𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀁 𑀅𑀦𑀸𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑁂 𑀆𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀧𑀤𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀯𑀸𑀘𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀦 𑀔𑀼𑀁𑀲𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀦 𑀯𑀫𑁆𑀪𑁂𑀢𑀼𑀓𑀸𑀫𑁄 𑀦 𑀫𑀗𑁆𑀓𑀼𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄 𑀤𑀯𑀓𑀫𑁆𑀬𑀢𑀸 𑀳𑀻𑀦𑁂𑀦 𑀳𑀻𑀦𑀁 𑀯𑀤𑁂𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲—  𑀧𑀜𑁆𑀘𑀫𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀇𑀫𑀸 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1198\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀙𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀦 𑀓𑀸𑀬𑀢𑁄𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑁇

(282.)

1199\. 𑀙𑀝𑁆𑀞𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀓𑀢𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺? 𑀙𑀝𑁆𑀞𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺—  𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀡𑁆𑀟𑀁 𑀅𑀯𑀳𑀭𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀲𑁆𑀲; 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀝𑀺𑀁 𑀓𑀭𑁄𑀢𑀺 𑀅𑀤𑁂𑀲𑀺𑀢𑀯𑀢𑁆𑀣𑀼𑀓𑀁 𑀧𑀫𑀸𑀡𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀫𑁆𑀪𑀁 𑀅𑀧𑀭𑀺𑀓𑁆𑀓𑀫𑀦𑀁, 𑀧𑀬𑁄𑀕𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑀁; 𑀏𑀓𑀁 𑀧𑀺𑀡𑁆𑀟𑀁 𑀅𑀦𑀸𑀕𑀢𑁂 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲; 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑁂 𑀆𑀕𑀢𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀧𑀡𑀻𑀢𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀲𑁆𑀲𑁇 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀲𑀜𑁆𑀜𑀻 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀯𑁄𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀦 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀲𑁆𑀲—  𑀙𑀝𑁆𑀞𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀇𑀫𑀸 𑀙 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇

1200\. 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀓𑀢𑀺 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸? 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀳𑀺 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺? 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁 𑀓𑀢𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁? 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀓𑀢𑀺𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺? 𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀯𑀺𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀤𑁆𑀯𑁂 𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑁄 𑀪𑀚𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀲𑀺𑀬𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑀦𑀁 𑀙𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀳𑀺 𑀲𑀗𑁆𑀕𑀳𑀺𑀢𑀸—  𑀲𑀺𑀬𑀸 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦, 𑀲𑀺𑀬𑀸 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦𑁇 𑀙𑀦𑁆𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑀸𑀦𑀁 𑀏𑀓𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺—  𑀓𑀸𑀬𑀢𑁄 𑀘 𑀯𑀸𑀘𑀢𑁄 𑀘 𑀘𑀺𑀢𑁆𑀢𑀢𑁄 𑀘 𑀲𑀫𑀼𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀸𑀦𑀁, 𑀆𑀧𑀢𑁆𑀢𑀸𑀥𑀺𑀓𑀭𑀡𑀁𑁇 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀫𑀣𑀸𑀦𑀁 𑀢𑀻𑀳𑀺 𑀲𑀫𑀣𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺—  𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀢𑀓𑀭𑀡𑁂𑀦 𑀘, 𑀲𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀯𑀺𑀦𑀬𑁂𑀦 𑀘 𑀢𑀺𑀡𑀯𑀢𑁆𑀣𑀸𑀭𑀓𑁂𑀦 𑀘𑀸𑀢𑀺𑁇
