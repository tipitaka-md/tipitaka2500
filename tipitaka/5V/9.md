# 𑁯 𑀅𑀢𑁆𑀣𑀯𑀲𑀧𑀓𑀭𑀡

* 𑀅𑀢𑁆𑀣𑀯𑀲𑀧𑀓𑀭𑀡

(334.)

1604\. 𑀤𑀲 𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀝𑀺𑀘𑁆𑀘 𑀢𑀣𑀸𑀕𑀢𑁂𑀦 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑀢𑀸𑀬, 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑀢𑀸𑀬, 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬, 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬, 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬, 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬, 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇

1605\. 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀬𑀁 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀬𑀁 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬𑁇 𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬 𑀢𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬𑁇 𑀬𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬 𑀢𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬𑁇 𑀬𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬 𑀢𑀁 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬𑁇 𑀬𑀁 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬 𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸𑁇 𑀬𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸 𑀢𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇

1606\. 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸𑁇 𑀬𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼 𑀢𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇

1607\. 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀬𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼 𑀢𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑁇

1608\. 𑀬𑀁 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬…𑀧𑁂…  𑀬𑀁 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬…  𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬…  𑀬𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬…  𑀬𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬…  𑀬𑀁 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬…  𑀬𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸…  𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀲𑀁𑀖𑀲𑀼𑀝𑁆𑀞𑀼𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀲𑀁𑀖𑀨𑀸𑀲𑀼𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀤𑀼𑀫𑁆𑀫𑀗𑁆𑀓𑀽𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑀸𑀬𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀧𑁂𑀲𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸𑀬𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀸𑀬𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑀸𑀦𑀁 𑀆𑀲𑀯𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀢𑀸𑀬𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀧𑀲𑀸𑀤𑀸𑀬𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬𑁇 𑀬𑀁 𑀯𑀺𑀦𑀬𑀸𑀦𑀼𑀕𑁆𑀕𑀳𑀸𑀬 𑀢𑀁 𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀝𑁆𑀞𑀺𑀢𑀺𑀬𑀸𑀢𑀺𑁇

1609\. _𑀅𑀢𑁆𑀣𑀲𑀢𑀁 𑀥𑀫𑁆𑀫𑀲𑀢𑀁,_  
_𑀤𑁆𑀯𑁂 𑀘 𑀦𑀺𑀭𑀼𑀢𑁆𑀢𑀺𑀲𑀢𑀸𑀦𑀺;_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀜𑀸𑀡𑀲𑀢𑀸𑀦𑀺,_  
_𑀅𑀢𑁆𑀣𑀯𑀲𑁂 𑀧𑀓𑀭𑀡𑁂𑀢𑀺𑁇_  


---

1610\. 𑀅𑀢𑁆𑀣𑀯𑀲𑀧𑀓𑀭𑀡𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

  
𑀫𑀳𑀸𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

1611\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1612\. _𑀧𑀞𑀫𑀁 𑀅𑀝𑁆𑀞𑀧𑀼𑀘𑁆𑀙𑀸𑀬𑀁,_  
_𑀧𑀘𑁆𑀘𑀬𑁂𑀲𑀼 𑀧𑀼𑀦𑀝𑁆𑀞 𑀘;_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑁄𑀍𑀅𑀲 𑀏𑀢𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀦𑀜𑁆𑀘 𑀲𑁄𑀍𑀅𑀲𑁇_
