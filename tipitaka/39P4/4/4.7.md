# 𑁪𑁇𑁭 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀤𑀼𑀓𑀸𑀤𑀺

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀆𑀲𑀯𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

### 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀤𑀼𑀓𑀸𑀤𑀺

(34.)

1329\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀲𑀜𑁆𑀜𑁄𑀚𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘…𑀧𑁂…  𑀅𑀓𑀼𑀲𑀮𑀁 𑀕𑀦𑁆𑀣𑀁…  𑀑𑀖𑀁…  𑀬𑁄𑀕𑀁…  𑀦𑀻𑀯𑀭𑀡𑀁… 𑁇

1330\. 𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀦𑁄𑀧𑀭𑀸𑀫𑀸𑀲𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀆𑀲𑀯𑀕𑁄𑀘𑁆𑀙𑀓𑀲𑀤𑀺𑀲𑀁𑁇) 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀺 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇
