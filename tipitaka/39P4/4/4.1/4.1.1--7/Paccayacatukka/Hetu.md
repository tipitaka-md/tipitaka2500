# 𑀳𑁂𑀢𑀼

* 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀆𑀲𑀯𑀧𑀭𑀸𑀫𑀸𑀲𑀕𑁄𑀘𑁆𑀙𑀓

* 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀆𑀲𑀯𑀤𑀼𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

* 𑀳𑁂𑀢𑀼

(1.)

1224\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1225\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1226\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁂𑀧𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭𑁂𑀧𑀺 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

(2.)

1227\. 𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

1228\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀓𑀼𑀲𑀮𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

1229\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀦𑁄𑀆𑀲𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)

1230\. 𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀆𑀲𑀯𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀦𑁄𑀆𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)

1231\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀦𑁄𑀆𑀲𑀯𑀜𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀦𑁄𑀆𑀲𑀯𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀦𑁄𑀆𑀲𑀯𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)
