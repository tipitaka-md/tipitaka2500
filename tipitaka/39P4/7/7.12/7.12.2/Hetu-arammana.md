# 𑀳𑁂𑀢𑀼-𑀆𑀭𑀫𑁆𑀫𑀡

* 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀧𑀺𑀝𑁆𑀞𑀺𑀤𑀼𑀓

* 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓, 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀤𑀼𑀓

* 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀧𑀤

* 𑀳𑁂𑀢𑀼-𑀆𑀭𑀫𑁆𑀫𑀡

(45.)

1650\. 𑀓𑀼𑀲𑀮𑀁 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀓𑀼𑀲𑀮𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)

1651\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1652\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀤𑁆𑀯𑁂𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇) (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀁𑁇)

(46.)

1653\. 𑀓𑀼𑀲𑀮𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

1654\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧)

1655\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀲𑁆𑀲 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑁄 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀓𑀼𑀲𑀮𑀲𑁆𑀲 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁨)

(47.)

1656\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑁂, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑁆𑀯𑁂, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀢𑀻𑀡𑀺, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀤𑁆𑀯𑁂…𑀧𑁂…  𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀤𑁆𑀯𑁂𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1657\. (𑀬𑀣𑀸 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓𑁂 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑀁, 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
