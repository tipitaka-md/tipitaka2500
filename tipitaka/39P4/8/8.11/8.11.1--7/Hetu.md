# 𑀳𑁂𑀢𑀼

* 𑀢𑀺𑀓, 𑀲𑀭𑀡𑀤𑀼𑀓

* 𑀲𑁂𑀓𑁆𑀔𑀢𑁆𑀢𑀺𑀓, 𑀲𑀭𑀡𑀤𑀼𑀓

* 𑀧𑀝𑀺𑀘𑁆𑀘𑀸𑀤𑀺𑀯𑀸𑀭

* 𑀳𑁂𑀢𑀼

(39.)

1815\. 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀁 𑀲𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄 𑀲𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1816\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀏𑀓𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀏𑀓𑀁𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1817\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑁂𑀧𑀺…𑀧𑁂…  𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑁂𑀧𑀺 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀏𑀓𑀁𑁇)

(40.)

1818\. 𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀘 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁩)

1819\. 𑀅𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸…  𑀢𑀻𑀡𑀺𑁇

1820\. 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧)

1821\. 𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀜𑁆𑀘 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁…𑀧𑁂…  𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇

1822\. 𑀅𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀜𑁆𑀘 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀁 𑀅𑀭𑀡𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀘𑁆𑀘 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸𑁇 (𑁧) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(41.)

1823\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀦𑀯, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀢𑀻𑀡𑀺…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀦𑀯𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1824\. (𑀲𑀳𑀚𑀸𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀲𑀫𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀯𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀝𑀺𑀘𑁆𑀘𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀁𑁇)

(42.)

1825\. 𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀲𑁂𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

1826\. 𑀅𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀅𑀲𑁂𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

1827\. 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑁄 𑀅𑀭𑀡𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦𑁂𑀯𑀲𑁂𑀓𑁆𑀔𑀦𑀸𑀲𑁂𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀭𑀡𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 (𑁧) (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

(43.)

1828\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀲𑀢𑁆𑀢, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀧𑀜𑁆𑀘, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀦𑀯, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀅𑀝𑁆𑀞…𑀧𑁂…  𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀅𑀝𑁆𑀞…𑀧𑁂…  𑀆𑀲𑁂𑀯𑀦𑁂 𑀤𑁆𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀅𑀝𑁆𑀞, 𑀯𑀺𑀧𑀸𑀓𑁂 𑀲𑀢𑁆𑀢…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀢𑁂𑀭𑀲𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

1829\. (𑀬𑀣𑀸 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓𑁂 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑀁, 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
