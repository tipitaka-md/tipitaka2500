# 𑀳𑁂𑀢𑀼-𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀤𑀺

### 𑀢𑀺𑀓, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

### 𑀧𑀻𑀢𑀺𑀢𑁆𑀢𑀺𑀓, 𑀳𑁂𑀢𑀼𑀤𑀼𑀓

### 𑀳𑁂𑀢𑀼𑀧𑀤

### 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀬𑀘𑀢𑀼𑀓𑁆𑀓

### 𑀳𑁂𑀢𑀼-𑀆𑀭𑀫𑁆𑀫𑀡𑀸𑀤𑀺

(104.)

351\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

352\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

353\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀏𑀓𑀁𑁇

354\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀢𑀻𑀡𑀺𑁇

(105.)

355\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑁇

356\. 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑁇

357\. 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇 𑀉𑀧𑁂𑀓𑁆𑀔𑀸𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑁇

358\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀘 𑀲𑀼𑀔𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀆𑀭𑀫𑁆𑀫𑀡𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄…  𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑁇

359\. 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑁄 𑀳𑁂𑀢𑀼 𑀥𑀫𑁆𑀫𑁄 𑀧𑀻𑀢𑀺𑀲𑀳𑀕𑀢𑀲𑁆𑀲 𑀳𑁂𑀢𑀼𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀥𑀺𑀧𑀢𑀺𑀧𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀘𑁆𑀘𑀬𑁄𑁇

(106.)

360\. 𑀳𑁂𑀢𑀼𑀬𑀸 𑀤𑀲, 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑁄𑀍𑀅𑀲, 𑀅𑀥𑀺𑀧𑀢𑀺𑀬𑀸 𑀲𑁄𑀍𑀅𑀲, 𑀅𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑁄𑀍𑀅𑀲, 𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑁂 𑀲𑁄𑀍𑀅𑀲, 𑀲𑀳𑀚𑀸𑀢𑁂 𑀤𑀲, 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑁂 𑀤𑀲, 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀤𑀲, 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀲𑁄𑀍𑀅𑀲…𑀧𑁂…  𑀯𑀺𑀧𑀸𑀓𑁂 𑀤𑀲…𑀧𑁂…  𑀅𑀯𑀺𑀕𑀢𑁂 𑀤𑀲𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

361\. 𑀦𑀳𑁂𑀢𑀼𑀬𑀸 𑀲𑁄𑀍𑀅𑀲, 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑁄𑀍𑀅𑀲𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

362\. 𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀦𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀤𑀲𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

363\. 𑀦𑀳𑁂𑀢𑀼𑀧𑀘𑁆𑀘𑀬𑀸 𑀆𑀭𑀫𑁆𑀫𑀡𑁂 𑀲𑁄𑀍𑀅𑀲𑁇 (𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀁𑁇)

364\. (𑀬𑀣𑀸 𑀓𑀼𑀲𑀮𑀢𑁆𑀢𑀺𑀓𑁂 𑀧𑀜𑁆𑀳𑀸𑀯𑀸𑀭𑀁, 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
