# 𑁧 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀸𑀭

(2.)

3\. 𑀓𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺? 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

4\. 𑀓𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺? 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

5\. 𑀓𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺? 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

6\. 𑀓𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺? 𑀲𑀩𑁆𑀩𑀲𑀓𑁆𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀏𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

7\. 𑀓𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺? 𑀲𑀩𑁆𑀩𑀲𑀓𑁆𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

8\. 𑀓𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺? 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

9\. 𑀓𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺? 𑀲𑀩𑁆𑀩𑀲𑀓𑁆𑀓𑀸𑀬𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂𑀲𑀼 𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

---

10\. 𑀉𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑀯𑀸𑀭𑁄𑁇
