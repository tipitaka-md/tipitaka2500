# 𑁨𑁇𑁨𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀲𑀸𑀦𑀼𑀲𑀬𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(77.)

525\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

526\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

527\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

528\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

529\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

530\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

531\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

532\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

533\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

534\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

535\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

536\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

537\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

(78.)

538\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

539\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

540\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

541\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

542\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

543\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

544\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

545\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

546\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

547\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

(79.)

548\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

549\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

550\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

551\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

552\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

553\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

554\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

555\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

556\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

(80.)

557\. »  𑀬𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

558\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(81.)

559\. »  𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

560\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

561\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

562\. »  𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

563\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

564\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

565\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

566\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

567\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇
