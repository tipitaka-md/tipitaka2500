# 𑁨𑁇𑁨𑁇𑁫 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀲𑀸𑀦𑀼𑀲𑀬𑀯𑀸𑀭

* 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(110.)

735\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

736\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

737\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

738\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

739\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

740\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

741\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

742\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

743\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

744\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

745\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

746\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

747\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

748\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

749\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

750\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

751\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(111.)

752\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

753\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

754\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

755\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

756\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

757\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

758\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

759\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

760\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

761\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

762\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

763\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

764\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

765\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(112.)

766\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

767\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

768\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

769\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

770\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

771\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

772\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

773\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

774\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

776\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

777\. »  𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

779\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

780\. »  𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

781\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

782\. »  𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

783\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

784\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

785\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

786\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

787\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

788\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

789\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

790\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

791\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

792\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

793\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

794\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

795\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

796\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

797\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

798\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀤𑀼𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(117.)

799\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

800\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

801\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

802\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

808\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

809\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

811\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
