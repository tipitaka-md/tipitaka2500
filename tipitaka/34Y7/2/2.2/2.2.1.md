# 𑁨𑁇𑁨𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀲𑀸𑀦𑀼𑀲𑀬𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(66.)

470\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

471\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

472\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

473\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

474\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

475\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

476\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

477\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

478\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

479\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

480\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

(67.)

481\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

482\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

483\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

484\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

485\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

486\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

487\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

488\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

489\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

(68.)

490\. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

491\. 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

492\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

493\. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

494\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(69.)

495\. »  𑀬𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

496\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(70.)

497\. »  𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

498\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

(71.)

500\. »  𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

501\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

502\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

503\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

504\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

505\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

506\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

507\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

508\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

509\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

510\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 (𑀤𑀼𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

511\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

512\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

513\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

514\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

515\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

516\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 (𑀢𑀺𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(74.)

517\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

518\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇 (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(75.)

519\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

520\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

521\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(76.)

522\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

523\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

524\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
