# 𑁨𑁇𑁨𑁇𑁪 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀲𑀸𑀦𑀼𑀲𑀬𑀯𑀸𑀭

### 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(99.)

682. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

683. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

684. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

685. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀭𑀳𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

686. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

687. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

688. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

689. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸𑁇

690. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

691. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀭𑀳𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

692. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(100.)

693. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

694. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀭𑀳𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

695. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

696. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

697. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

698. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸𑁇 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸𑁇

699. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

700. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀭𑀳𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

701. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(101.)

702. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

703. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

704. 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸𑁇 𑀅𑀭𑀳𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇

705. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

706. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

728. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

729. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

730. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑀸𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑁇 𑀅𑀭𑀳𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄…𑀧𑁂… 𑁇 (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(108.)

731. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

732. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(109.)

733. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

734. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀦𑀺𑀭𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
