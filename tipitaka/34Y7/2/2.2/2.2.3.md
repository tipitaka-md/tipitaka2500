# 𑁨𑁇𑁨𑁇𑁩 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀲𑀸𑀦𑀼𑀲𑀬𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

(88.)

602\. »  𑀬𑁄 𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

603\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

604\. »  𑀬𑁄 𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

605\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

606\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀢𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

607\. »  𑀬𑁄 𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

608\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

609\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

610\. 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀲𑁆𑀯𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

611\. »  𑀬𑁄 𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

612\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

613\. »  𑀬𑁄 𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

614\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

615\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀲𑁄 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀢𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

(89.)

616\. »  𑀬𑁄 𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

617\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

618\. »  𑀬𑁄 𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

619\. 𑀤𑁆𑀯𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

620\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

621\. 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀲𑁆𑀯𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

622\. »  𑀬𑁄 𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

623\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀦𑁄𑁇

624\. »  𑀬𑁄 𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

625\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

626\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀻 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀲𑁄 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀢𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

(90.)

627\. »  𑀬𑁄 𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁂𑀦…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

628\. 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

629\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

630\. 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄, 𑀦𑁄 𑀘 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇 𑀲𑁆𑀯𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

631\. »  𑀬𑁄 𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

632\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀢𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

633\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

634\. »  𑀬𑁄 𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

635\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

636\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀢𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇

639\. »  𑀬𑁄 𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

641\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

642\. 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

643\. »  𑀬𑁄 𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

644\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?

645\. 𑀢𑀬𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸, 𑀦𑁄 𑀘 𑀢𑁂 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑀸𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀘 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑁇

646\. »  𑀬𑁄 𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

647\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄 𑀲𑁄 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁂𑀦 𑀲𑀸𑀦𑀼𑀲𑀬𑁄𑀢𑀺?
