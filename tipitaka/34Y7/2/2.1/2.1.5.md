# 𑁨𑁇𑁧𑁇𑁫 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀲𑀬𑀯𑀸𑀭

### 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(46.)

301. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

302. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

303. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

304. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

305. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

306. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

307. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

308. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

309. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

310. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

311. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

312. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

313. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

314. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

315. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

316. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

317. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(47.)

318. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

319. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺𑁇

320. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

375. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺?

376. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀏𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇

377. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

378. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦…𑀧𑁂… ? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

379. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦𑀸𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀦𑀸𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
