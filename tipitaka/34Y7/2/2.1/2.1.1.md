# 𑁨𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀲𑀬𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(3.)

11\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

12\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

13\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

14\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

15\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

16\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

17\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

18\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

19\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

20\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

21\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

22\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

23\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

24\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

25\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

26\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

27\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

(4.)

28\. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

29\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

30\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

31\. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

32\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

33\. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

34\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

35\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

(5.)

36\. »  𑀬𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

37\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

38\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

39\. »  𑀬𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

40\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(6.)

41\. »  𑀬𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

42\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

43\. »  𑀬𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

44\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

45\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

(7.)

46\. »  𑀬𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

47\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

48\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

(8.)

49\. »  𑀬𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

50\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(9.)

51\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

52\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺?

53\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇

54\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺?

55\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

56\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

57\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

58\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺?

61\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺𑁇

62\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

63\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

64\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺?

65\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 (𑀢𑀺𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(11.)

66\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

67\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

68\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

69\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺?

(12.)

71\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

72\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺?

73\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(13.)

74\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

75\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀻𑀢𑀺?

76\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀦𑀼𑀲𑁂𑀦𑁆𑀢𑀺𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
