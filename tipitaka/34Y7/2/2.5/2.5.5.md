# 𑁨𑁇𑁫𑁇𑁫 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀳𑀻𑀦𑀯𑀸𑀭

* 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(308.)

2093\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2094\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2095\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2096\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2097\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2098\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2099\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2100\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2101\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2102\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2103\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2104\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2105\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2106\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2107\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2108\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2109\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

(309.)

2110\. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2111\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2112\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2113\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2114\. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2115\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2116\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2117\. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2118\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2119\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2120\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2121\. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2122\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2123\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

(310.)

2124\. »  𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2125\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2126\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2127\. »  𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2128\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2129\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2130\. »  𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2131\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2132\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

(311.)

2133\. »  𑀬𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2134\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(312.)

2135\. »  𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2136\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2137\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2138\. »  𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2139\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2140\. »  𑀬𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2141\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2167\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀢𑀺?

2168\. 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇

2169\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

2170\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀢𑀺?

2171\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

2172\. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

2173\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀢𑀺?
