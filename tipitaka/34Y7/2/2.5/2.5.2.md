# 𑁨𑁇𑁫𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀳𑀻𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(275.)

1864. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1865. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1866. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1867. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1868. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1869. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1870. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1871. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1872. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1873. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1874. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1875. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1876. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1877. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1878. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1879. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1880. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

(276.)

1881. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1882. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1883. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1884. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1885. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1886. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1887. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1888. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1889. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1890. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1891. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

1892. »  𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1943. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(285.)

1944. »  𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1945. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸𑀢𑀺?

1946. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀏𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
