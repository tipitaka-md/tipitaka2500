# 𑁨𑁇𑁫𑁇𑁪 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀳𑀻𑀦𑀯𑀸𑀭

* 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(297.)

2038\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2039\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2040\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2041\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2042\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2043\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2044\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2045\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2046\. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2047\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2048\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

(298.)

2049\. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2050\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2051\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2052\. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2053\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2054\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2055\. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2056\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2057\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

(299.)

2058\. »  𑀬𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2059\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇
