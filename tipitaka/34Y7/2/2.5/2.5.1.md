# 𑁨𑁇𑁫𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀳𑀻𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(264.)

1811. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1812. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1813. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1814. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀅𑀭𑀳𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1815. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1816. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1817. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1818. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1819. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1820. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀅𑀭𑀳𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1821. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(265.)

1822. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1823. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀅𑀭𑀳𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1824. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1825. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1826. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1827. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1828. »  𑀬𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1829. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀅𑀭𑀳𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1830. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(266.)

1831. »  𑀬𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1832. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1833. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀅𑀭𑀳𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄𑁇

1834. »  𑀬𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1835. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(267.)

1836. »  𑀬𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1837. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(268.)

1838. »  𑀬𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

1857. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸 𑀢𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1858. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸𑀢𑀺?

1859. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑁇 𑀅𑀭𑀳𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸…𑀧𑁂… 𑁇 (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(273.)

1860. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸 𑀢𑀲𑁆𑀲 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1861. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(274.)

1862. »  𑀬𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸 𑀢𑀲𑁆𑀲 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1863. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀳𑀻𑀦𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
