# 𑁨𑁇𑁫𑁇𑁬 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀳𑀻𑀦𑀯𑀸𑀭

* 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

(319.)

2175\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2176\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2177\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2178\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2179\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2180\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2181\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑁂𑀲𑀜𑁆𑀜𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2182\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2183\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2184\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2185\. 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2186\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2187\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2188\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2189\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2190\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2191\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2193\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2194\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2195\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2196\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2197\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2198\. 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2199\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2200\. 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2201\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2202\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2203\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2204\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇

2205\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2208\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2209\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2210\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2211\. 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2212\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2213\. 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑁂𑀲𑀜𑁆𑀜𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2214\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2215\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2216\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2217\. 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑁂𑀲𑀜𑁆𑀜𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2219\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

2220\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2221\. 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2222\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2223\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

2224\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2225\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2226\. 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇

(324.)

2227\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

2228\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺?

2229\. 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑁂𑀲𑀜𑁆𑀜𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

2230\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

2231\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸𑀢𑀺?

2232\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀲𑁆𑀲 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 𑀢𑀲𑁆𑀲𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑁆𑀲 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄, 𑀦𑁄 𑀘 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄; 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸 “𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸”𑀢𑀺 𑀯𑀸𑁇 𑀢𑁂𑀲𑀜𑁆𑀜𑁂𑀯 𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑁄 𑀘 𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑀸; 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑁄 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄 “𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸 “𑀅𑀧𑁆𑀧𑀳𑀻𑀦𑁄”𑀢𑀺 𑀯𑀸𑁇
