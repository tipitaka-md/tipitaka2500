# 𑁨𑁇𑁪𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(198.)

1359\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1360\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1361\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1362\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1363\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1364\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1365\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1366\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1367\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1368\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1369\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

(199.)

1370\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1371\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1372\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1373\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1374\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1375\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1376\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1377\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1378\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

(200.)

1379\. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1380\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1381\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1382\. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1383\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(201.)

1384\. »  𑀬𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1385\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(202.)

1386\. »  𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1387\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1388\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

(203.)

1389\. »  𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1390\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(204.)

1391\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1392\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1393\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1394\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

1395\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1396\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1397\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1398\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1399\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇 (𑀤𑀼𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(205.)

1400\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1401\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1402\. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇

1403\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1404\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1405\. 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑀢𑀺𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(206.)

1406\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1407\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1408\. 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺…𑀧𑁂… 𑁇 (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(207.)

1409\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1410\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1411\. 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(208.)

1412\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

1413\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1414\. 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
