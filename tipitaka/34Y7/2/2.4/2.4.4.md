# 𑁨𑁇𑁪𑁇𑁪 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀭𑀺𑀜𑁆𑀜𑀸𑀯𑀸𑀭

* 𑀧𑀝𑀺𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(231.)

1562\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1563\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1564\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1565\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1566\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1567\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1568\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1569\. 𑀅𑀝𑁆𑀞𑀫𑀓𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1570\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1571\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1572\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1573\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1574\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1575\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

(232.)

1576\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1577\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1578\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1579\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1580\. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1598\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1599\. 𑀅𑀝𑁆𑀞𑀫𑀓𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

(236.)

1600\. »  𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1601\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(237.)

1602\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1603\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1604\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1605\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1606\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1607\. 𑀅𑀝𑁆𑀞𑀫𑀓𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1608\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1609\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1610\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1611\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇

1612\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1621\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1622\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1623\. 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 …𑀧𑁂… 𑁇 (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(240.)

1624\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1625\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1626\. 𑀅𑀝𑁆𑀞𑀫𑀓𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(241.)

1627\. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1628\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀻𑀢𑀺?

1629\. 𑀅𑀝𑁆𑀞𑀫𑀓𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑁄 𑀘 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀫𑀕𑁆𑀕𑀲𑀫𑀗𑁆𑀕𑀻𑀦𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀫𑀓𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦 𑀧𑀭𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
