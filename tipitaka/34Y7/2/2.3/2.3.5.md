# 𑁨𑁇𑁩𑁇𑁫 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

### 𑀫𑀳𑀸𑀯𑀸𑀭

### 𑀧𑀚𑀳𑀦𑀯𑀸𑀭

### 𑀧𑀝𑀺𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(176.)

1178\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1179\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1180\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1181\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1182\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1183\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1184\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1185\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1186\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1187\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1188\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1189\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1190\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1191\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1192\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1193\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1194\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(177.)

1195\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1196\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1197\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1198\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1199\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1200\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1201\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1202\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1203\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1204\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1205\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1206\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1207\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1208\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(178.)

1209\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1210\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1211\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1212\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1213\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1214\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1215\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1216\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1217\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(179.)

1218\. »  𑀬𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1219\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1220\. »  𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1221\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1222\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺; 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1223\. »  𑀬𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1224\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1225\. »  𑀬𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1226\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1227\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

1228\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1229\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1230\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1231\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1232\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1233\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1234\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1235\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1236\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1237\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1238\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1239\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1240\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1241\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀤𑀼𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(183.)

1242\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1243\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1244\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1245\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1246\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1247\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1248\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀢𑀺𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(184.)

1249\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1250\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇 (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(185.)

1251\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1252\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1253\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑁂 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺𑁇

1254\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1255\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(186.)

1256\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

1257\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
