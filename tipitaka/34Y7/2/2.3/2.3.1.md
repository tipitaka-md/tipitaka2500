(132.)

906. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

911. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

912. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

(133.)

917. »  𑀬𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

921. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

924. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀚𑀳𑀢𑀺𑁇

925. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

(134.)

926. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

927. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

928. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀚𑀳𑀢𑀺𑁇

929. »  𑀬𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

930. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(135.)

931. »  𑀬𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

932. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇

(136.)

933. »  𑀬𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

(137.)

936. »  𑀬𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

937. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 (𑀏𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

942. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

(139.)

947. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

948. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

949. 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀚𑀳𑀢𑀺𑁇

950. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

951. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

952. 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 (𑀢𑀺𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(140.)

953. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

954. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

955. 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀢𑀤𑁂𑀓𑀝𑁆𑀞𑀁 𑀧𑀚𑀳𑀢𑀺…𑀧𑁂… 𑁇 (𑀘𑀢𑀼𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(141.)

956. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

957. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

958. 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 (𑀧𑀜𑁆𑀘𑀓𑀫𑀽𑀮𑀓𑀁𑁇)

(142.)

959. »  𑀬𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

960. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

961. 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺𑁇 (𑀙𑀓𑁆𑀓𑀫𑀽𑀮𑀓𑀁𑁇)
