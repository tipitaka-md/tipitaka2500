# 𑁨𑁇𑁩𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀫𑀳𑀸𑀯𑀸𑀭

* 𑀧𑀚𑀳𑀦𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(143.)

962\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

963\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

964\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

965\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

966\. 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺𑁇

967\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

968\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

969\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺𑁇

970\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

971\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

972\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

973\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

974\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺𑁇

(144.)

975\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

976\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

977\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

978\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

979\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺𑁇

980\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

981\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑁄𑁇

982\. »  𑀬𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

983\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

984\. 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺𑁇

(145.)

985\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

986\. «  𑀬𑀢𑁄 𑀯𑀸 𑀧𑀦 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

987\. 𑀤𑀼𑀓𑁆𑀔𑀸𑀬 𑀯𑁂𑀤𑀦𑀸𑀬 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺𑁇 𑀓𑀸𑀫𑀥𑀸𑀢𑀼𑀬𑀸 𑀤𑁆𑀯𑀻𑀲𑀼 𑀯𑁂𑀤𑀦𑀸𑀲𑀼 𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀅𑀭𑀽𑀧𑀥𑀸𑀢𑀼𑀬𑀸 𑀢𑀢𑁄 𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺𑁇

988\. »  𑀬𑀢𑁄 𑀫𑀸𑀦𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀪𑀯𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺?

1007\. »  𑀬𑀢𑁄 𑀓𑀸𑀫𑀭𑀸𑀕𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀖𑀸𑀦𑀼𑀲𑀬𑀜𑁆𑀘 𑀧𑀚𑀳𑀢𑀺 𑀢𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑀬𑀁…𑀧𑁂…  𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀦𑀼𑀲𑀬𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇
