# 𑁧𑁇𑁧 𑀯𑀢𑁆𑀣𑀼𑀕𑀸𑀣𑀸

### 𑀧𑀸𑀭𑀸𑀬𑀦𑀯𑀕𑁆𑀕𑀕𑀸𑀣𑀸

### 𑀯𑀢𑁆𑀣𑀼𑀕𑀸𑀣𑀸

(1.)

2. _𑀓𑁄𑀲𑀮𑀸𑀦𑀁 𑀧𑀼𑀭𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_𑀅𑀕𑀫𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑀁;_  
_𑀆𑀓𑀺𑀜𑁆𑀘𑀜𑁆𑀜𑀁 𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀫𑀦𑁆𑀢𑀧𑀸𑀭𑀕𑀽𑁇_  


(2.)

3. _𑀲𑁄 𑀅𑀲𑁆𑀲𑀓𑀲𑁆𑀲 𑀯𑀺𑀲𑀬𑁂,_  
_𑀫𑀍𑀅𑀓𑀲𑁆𑀲 𑀲𑀫𑀸𑀲𑀦𑁂;_  
_𑀯𑀲𑀺 𑀕𑁄𑀥𑀸𑀯𑀭𑀻𑀓𑀽𑀮𑁂,_  
_𑀉𑀜𑁆𑀙𑁂𑀦 𑀘 𑀨𑀮𑁂𑀦 𑀘𑁇_  


(3.)

4. _𑀢𑀲𑁆𑀲𑁂𑀯 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬,_  
_𑀕𑀸𑀫𑁄 𑀘 𑀯𑀺𑀧𑀼𑀮𑁄 𑀅𑀳𑀼;_  
_𑀢𑀢𑁄 𑀚𑀸𑀢𑁂𑀦 𑀆𑀬𑁂𑀦,_  
_𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀫𑀓𑀧𑁆𑀧𑀬𑀺𑁇_  


(4.)

5. _𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑀁 𑀬𑀚𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀧𑀼𑀦 𑀧𑀸𑀯𑀺𑀲𑀺 𑀅𑀲𑁆𑀲𑀫𑀁;_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀝𑀺𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀺,_  
_𑀅𑀜𑁆𑀜𑁄 𑀆𑀕𑀜𑁆𑀙𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


(5.)

6. _𑀉𑀕𑁆𑀖𑀝𑁆𑀝𑀧𑀸𑀤𑁄 𑀢𑀲𑀺𑀢𑁄,_  
_𑀧𑀗𑁆𑀓𑀤𑀦𑁆𑀢𑁄 𑀭𑀚𑀲𑁆𑀲𑀺𑀭𑁄;_  
_𑀲𑁄 𑀘 𑀦𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀲𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀬𑀸𑀘𑀢𑀺𑁇_  


(6.)

7. _𑀢𑀫𑁂𑀦𑀁 𑀩𑀸𑀯𑀭𑀻 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀆𑀲𑀦𑁂𑀦 𑀦𑀺𑀫𑀦𑁆𑀢𑀬𑀺;_  
_𑀲𑀼𑀔𑀜𑁆𑀘 𑀓𑀼𑀲𑀮𑀁 𑀧𑀼𑀘𑁆𑀙𑀺,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑁇_  


(7.)

8. _“𑀬𑀁 𑀔𑁄 𑀫𑀫 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑀁,_  
_𑀲𑀩𑁆𑀩𑀁 𑀯𑀺𑀲𑀚𑁆𑀚𑀺𑀢𑀁 𑀫𑀬𑀸;_  
_𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀳𑀺 𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀦𑀢𑁆𑀣𑀺 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺 𑀫𑁂”𑁇_  


(8.)

9. _“𑀲𑀘𑁂 𑀫𑁂 𑀬𑀸𑀘𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀪𑀯𑀁 𑀦𑀸𑀦𑀼𑀧𑀤𑀲𑁆𑀲𑀢𑀺;_  
_𑀲𑀢𑁆𑀢𑀫𑁂 𑀤𑀺𑀯𑀲𑁂 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀫𑀼𑀤𑁆𑀥𑀸 𑀨𑀮𑀢𑀼 𑀲𑀢𑁆𑀢𑀥𑀸”𑁇_  


(9.)

10. _𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀼𑀳𑀓𑁄,_  
_𑀪𑁂𑀭𑀯𑀁 𑀲𑁄 𑀅𑀓𑀺𑀢𑁆𑀢𑀬𑀺;_  
_𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀩𑀸𑀯𑀭𑀻 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑁄 𑀅𑀳𑀼𑁇_  


(10.)

11. _𑀉𑀲𑁆𑀲𑀼𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑀸𑀳𑀸𑀭𑁄,_  
_𑀲𑁄𑀓𑀲𑀮𑁆𑀮𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑁄;_  
_𑀅𑀣𑁄𑀧𑀺 𑀏𑀯𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀛𑀸𑀦𑁂 𑀦 𑀭𑀫𑀢𑀻 𑀫𑀦𑁄𑁇_  


(11.)

12. _𑀉𑀢𑁆𑀭𑀲𑁆𑀢𑀁 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀤𑁂𑀯𑀢𑀸 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀺𑀦𑀻;_  
_𑀩𑀸𑀯𑀭𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑁆𑀫,_  
_𑀇𑀤𑀁 𑀯𑀘𑀦𑀫𑀩𑁆𑀭𑀯𑀺𑁇_  


(12.)

13. _“𑀦 𑀲𑁄 𑀫𑀼𑀤𑁆𑀥𑀁 𑀧𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀓𑀼𑀳𑀓𑁄 𑀲𑁄 𑀥𑀦𑀢𑁆𑀣𑀺𑀓𑁄;_  
_𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀫𑀼𑀤𑁆𑀥𑀧𑀸𑀢𑁂 𑀯𑀸,_  
_𑀜𑀸𑀡𑀁 𑀢𑀲𑁆𑀲 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(13.)

14. _“𑀪𑁄𑀢𑀻 𑀘𑀭𑀳𑀺 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀸;_  
_𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀢𑀁 𑀲𑀼𑀡𑁄𑀫 𑀯𑀘𑁄 𑀢𑀯”𑁇_  


(14.)

15. _“𑀅𑀳𑀫𑁆𑀧𑁂𑀢𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀜𑀸𑀡𑀁 𑀫𑁂𑀢𑁆𑀣 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑁂 𑀘,_  
_𑀚𑀺𑀦𑀸𑀦𑀁 𑀳𑁂𑀢𑁆𑀣 𑀤𑀲𑁆𑀲𑀦𑀁”𑁇_  


(15.)

16. _“𑀅𑀣 𑀓𑁄 𑀘𑀭𑀳𑀺 𑀚𑀸𑀦𑀸𑀢𑀺,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀧𑀣𑀯𑀺𑀫𑀡𑁆𑀟𑀮𑁂;_  
_𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀤𑁂𑀯𑀢𑁂”𑁇_  


(16.)

17. _“𑀧𑀼𑀭𑀸 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀫𑁆𑀳𑀸,_  
_𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀮𑁄𑀓𑀦𑀸𑀬𑀓𑁄;_  
_𑀅𑀧𑀘𑁆𑀘𑁄 𑀑𑀓𑁆𑀓𑀸𑀓𑀭𑀸𑀚𑀲𑁆𑀲,_  
_𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀪𑀗𑁆𑀓𑀭𑁄𑁇_  


(17.)

18. _𑀲𑁄 𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦 𑀧𑀸𑀭𑀕𑀽;_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀺𑀜𑁆𑀜𑀸𑀩𑀮𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑁂𑀲𑀼 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀓𑁆𑀔𑀬𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀉𑀧𑀥𑀺𑀓𑁆𑀔𑀬𑁂𑁇_  


(18.)

19. _𑀩𑀼𑀤𑁆𑀥𑁄 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀮𑁄𑀓𑁂,_  
_𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀫𑀸;_  
_𑀢𑀁 𑀢𑁆𑀯𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀧𑀼𑀘𑁆𑀙𑀲𑁆𑀲𑀼,_  
_𑀲𑁄 𑀢𑁂 𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(19.)

20. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀯𑀘𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀉𑀤𑀕𑁆𑀕𑁄 𑀩𑀸𑀯𑀭𑀻 𑀅𑀳𑀼;_  
_𑀲𑁄𑀓𑀲𑁆𑀲 𑀢𑀦𑀼𑀓𑁄 𑀆𑀲𑀺,_  
_𑀧𑀻𑀢𑀺𑀜𑁆𑀘 𑀯𑀺𑀧𑀼𑀮𑀁 𑀮𑀪𑀺𑁇_  


(20.)

21. _𑀲𑁄 𑀩𑀸𑀯𑀭𑀻 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀉𑀤𑀕𑁆𑀕𑁄,_  
_𑀢𑀁 𑀤𑁂𑀯𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺 𑀯𑁂𑀤𑀚𑀸𑀢𑁄;_  
_“𑀓𑀢𑀫𑀫𑁆𑀳𑀺 𑀕𑀸𑀫𑁂 𑀦𑀺𑀕𑀫𑀫𑁆𑀳𑀺 𑀯𑀸 𑀧𑀦,_  
_𑀓𑀢𑀫𑀫𑁆𑀳𑀺 𑀯𑀸 𑀚𑀦𑀧𑀤𑁂 𑀮𑁄𑀓𑀦𑀸𑀣𑁄;_  
_𑀬𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀧𑀲𑁆𑀲𑁂𑀫𑀼,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑀁”𑁇_  


(21.)

22. _“𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀓𑁄𑀲𑀮𑀫𑀦𑁆𑀤𑀺𑀭𑁂 𑀚𑀺𑀦𑁄,_  
_𑀧𑀳𑀽𑀢𑀧𑀜𑁆𑀜𑁄 𑀯𑀭𑀪𑀽𑀭𑀺𑀫𑁂𑀥𑀲𑁄;_  
_𑀲𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀥𑀼𑀭𑁄 𑀅𑀦𑀸𑀲𑀯𑁄,_  
_𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀲𑁆𑀲 𑀯𑀺𑀤𑀽 𑀦𑀭𑀸𑀲𑀪𑁄”𑁇_  


(22.)

23. _𑀢𑀢𑁄 𑀆𑀫𑀦𑁆𑀢𑀬𑀻 𑀲𑀺𑀲𑁆𑀲𑁂,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀫𑀦𑁆𑀢𑀧𑀸𑀭𑀕𑀽;_  
_“𑀏𑀣 𑀫𑀸𑀡𑀯𑀸 𑀅𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀼𑀡𑀸𑀣 𑀯𑀘𑀦𑀁 𑀫𑀫𑁇_  


(23.)

24. _𑀬𑀲𑁆𑀲𑁂𑀲𑁄 𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀮𑁄𑀓𑁂,_  
_𑀧𑀸𑀢𑀼𑀪𑀸𑀯𑁄 𑀅𑀪𑀺𑀡𑁆𑀳𑀲𑁄;_  
_𑀲𑁆𑀯𑀸𑀚𑁆𑀚 𑀮𑁄𑀓𑀫𑁆𑀳𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀇𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁,_  
_𑀧𑀲𑁆𑀲𑀯𑁆𑀳𑁄 𑀤𑁆𑀯𑀺𑀧𑀤𑀼𑀢𑁆𑀢𑀫𑀁”𑁇_  


(24.)

25. _“𑀓𑀣𑀁 𑀘𑀭𑀳𑀺 𑀚𑀸𑀦𑁂𑀫𑀼,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑁄𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀅𑀚𑀸𑀦𑀢𑀁 𑀦𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀬𑀣𑀸 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(25.)

26. _“𑀆𑀕𑀢𑀸𑀦𑀺 𑀳𑀺 𑀫𑀦𑁆𑀢𑁂𑀲𑀼,_  
_𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀸;_  
_𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀦𑀺 𑀘 𑀩𑁆𑀬𑀸𑀓𑁆𑀔𑀸𑀢𑀸,_  
_𑀲𑀫𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀲𑁄𑁇_  


(26.)

27. _𑀬𑀲𑁆𑀲𑁂𑀢𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀕𑀢𑁆𑀢𑁂𑀲𑀼,_  
_𑀫𑀳𑀸𑀧𑀼𑀭𑀺𑀲𑀮𑀓𑁆𑀔𑀡𑀸;_  
_𑀤𑁆𑀯𑁂𑀬𑁂𑀯 𑀢𑀲𑁆𑀲 𑀕𑀢𑀺𑀬𑁄,_  
_𑀢𑀢𑀺𑀬𑀸 𑀳𑀺 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(27.)

28. _𑀲𑀘𑁂 𑀅𑀕𑀸𑀭𑀁 𑀆𑀯𑀲𑀢𑀺,_  
_𑀯𑀺𑀚𑁂𑀬𑁆𑀬 𑀧𑀣𑀯𑀺𑀁 𑀇𑀫𑀁;_  
_𑀅𑀤𑀡𑁆𑀟𑁂𑀦 𑀅𑀲𑀢𑁆𑀣𑁂𑀦,_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺𑁇_  


(28.)

29. _𑀲𑀘𑁂 𑀘 𑀲𑁄 𑀧𑀩𑁆𑀩𑀚𑀢𑀺,_  
_𑀅𑀕𑀸𑀭𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁;_  
_𑀯𑀺𑀯𑀝𑁆𑀝𑀘𑁆𑀙𑀤𑁄 𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄,_  
_𑀅𑀭𑀳𑀸 𑀪𑀯𑀢𑀺 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄𑁇_  


(29.)

30. _𑀚𑀸𑀢𑀺𑀁 𑀕𑁄𑀢𑁆𑀢𑀜𑁆𑀘 𑀮𑀓𑁆𑀔𑀡𑀁,_  
_𑀫𑀦𑁆𑀢𑁂 𑀲𑀺𑀲𑁆𑀲𑁂 𑀧𑀼𑀦𑀸𑀧𑀭𑁂;_  
_𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀫𑀦𑀲𑀸𑀬𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀣𑁇_  


(30.)

31. _𑀅𑀦𑀸𑀯𑀭𑀡𑀤𑀲𑁆𑀲𑀸𑀯𑀻,_  
_𑀬𑀤𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑀦𑀲𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀧𑀜𑁆𑀳𑁂,_  
_𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀲𑀚𑁆𑀚𑁂𑀲𑁆𑀲𑀢𑀺”𑁇_  


(31.)

32. _𑀩𑀸𑀯𑀭𑀺𑀲𑁆𑀲 𑀯𑀘𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀲𑁆𑀲𑀸 𑀲𑁄𑀍𑀅𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀅𑀚𑀺𑀢𑁄 𑀢𑀺𑀲𑁆𑀲𑀫𑁂𑀢𑁆𑀢𑁂𑀬𑁆𑀬𑁄,_  
_𑀧𑀼𑀡𑁆𑀡𑀓𑁄 𑀅𑀣 𑀫𑁂𑀢𑁆𑀢𑀕𑀽𑁇_  


(32.)

33. _𑀥𑁄𑀢𑀓𑁄 𑀉𑀧𑀲𑀻𑀯𑁄 𑀘,_  
_𑀦𑀦𑁆𑀤𑁄 𑀘 𑀅𑀣 𑀳𑁂𑀫𑀓𑁄;_  
_𑀢𑁄𑀤𑁂𑀬𑁆𑀬𑀓𑀧𑁆𑀧𑀸 𑀤𑀼𑀪𑀬𑁄,_  
_𑀚𑀢𑀼𑀓𑀡𑁆𑀡𑀻 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


(33.)

34. _𑀪𑀤𑁆𑀭𑀸𑀯𑀼𑀥𑁄 𑀉𑀤𑀬𑁄 𑀘,_  
_𑀧𑁄𑀲𑀸𑀮𑁄 𑀘𑀸𑀧𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀫𑁄𑀖𑀭𑀸𑀚𑀸 𑀘 𑀫𑁂𑀥𑀸𑀯𑀻,_  
_𑀧𑀺𑀗𑁆𑀕𑀺𑀬𑁄 𑀘 𑀫𑀳𑀸𑀇𑀲𑀺𑁇_  


(34.)

35. _𑀧𑀘𑁆𑀘𑁂𑀓𑀕𑀡𑀺𑀦𑁄 𑀲𑀩𑁆𑀩𑁂,_  
_𑀲𑀩𑁆𑀩𑀮𑁄𑀓𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀼𑀢𑀸;_  
_𑀛𑀸𑀬𑀻 𑀛𑀸𑀦𑀭𑀢𑀸 𑀥𑀻𑀭𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑀯𑀸𑀲𑀦𑀯𑀸𑀲𑀺𑀢𑀸𑁇_  


(35.)

36. _𑀩𑀸𑀯𑀭𑀺𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀚𑀝𑀸𑀚𑀺𑀦𑀥𑀭𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀼𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀫𑀼𑀔𑀸𑁇_  


(36.)

37. _𑀫𑀍𑀅𑀓𑀲𑁆𑀲 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀦𑀁,_  
_𑀧𑀼𑀭𑀫𑀸𑀳𑀺𑀲𑁆𑀲𑀢𑀺𑀁 𑀢𑀤𑀸;_  
_𑀉𑀚𑁆𑀚𑁂𑀦𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀕𑁄𑀦𑀤𑁆𑀥𑀁,_  
_𑀯𑁂𑀤𑀺𑀲𑀁 𑀯𑀦𑀲𑀯𑁆𑀳𑀬𑀁𑁇_  


(37.)

38. _𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀸𑀓𑁂𑀢𑀁,_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀜𑁆𑀘 𑀧𑀼𑀭𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀲𑁂𑀢𑀩𑁆𑀬𑀁 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀁,_  
_𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀜𑁆𑀘 𑀫𑀦𑁆𑀤𑀺𑀭𑀁𑁇_  


(38.)

39. _𑀧𑀸𑀯𑀜𑁆𑀘 𑀪𑁄𑀕𑀦𑀕𑀭𑀁,_  
_𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀫𑀸𑀕𑀥𑀁 𑀧𑀼𑀭𑀁;_  
_𑀧𑀸𑀲𑀸𑀡𑀓𑀁 𑀘𑁂𑀢𑀺𑀬𑀜𑁆𑀘,_  
_𑀭𑀫𑀡𑀻𑀬𑀁 𑀫𑀦𑁄𑀭𑀫𑀁𑁇_  


(39.)

40. _𑀢𑀲𑀺𑀢𑁄𑀯𑀼𑀤𑀓𑀁 𑀲𑀻𑀢𑀁,_  
_𑀫𑀳𑀸𑀮𑀸𑀪𑀁𑀯 𑀯𑀸𑀡𑀺𑀚𑁄;_  
_𑀙𑀸𑀬𑀁 𑀖𑀫𑁆𑀫𑀸𑀪𑀺𑀢𑀢𑁆𑀢𑁄𑀯,_  
_𑀢𑀼𑀭𑀺𑀢𑀸 𑀧𑀩𑁆𑀩𑀢𑀫𑀸𑀭𑀼𑀳𑀼𑀁𑁇_  


(40.)

41. _𑀪𑀕𑀯𑀸 𑀢𑀫𑁆𑀳𑀺 𑀲𑀫𑀬𑁂,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺,_  
_𑀲𑀻𑀳𑁄𑀯 𑀦𑀤𑀢𑀻 𑀯𑀦𑁂𑁇_  


(41.)

42. _𑀅𑀚𑀺𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀩𑀼𑀤𑁆𑀥𑀁,_  
_𑀧𑀻𑀢𑀭𑀁𑀲𑀺𑀁𑀯 𑀪𑀸𑀡𑀼𑀫𑀁;_  
_𑀘𑀦𑁆𑀤𑀁 𑀬𑀣𑀸 𑀧𑀦𑁆𑀦𑀭𑀲𑁂,_  
_𑀧𑀭𑀺𑀧𑀽𑀭𑀁 𑀉𑀧𑀸𑀕𑀢𑀁𑁇_  


(42.)

43. _𑀅𑀣𑀲𑁆𑀲 𑀕𑀢𑁆𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀧𑀭𑀺𑀧𑀽𑀭𑀜𑁆𑀘 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁;_  
_𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀳𑀝𑁆𑀞𑁄,_  
_𑀫𑀦𑁄𑀧𑀜𑁆𑀳𑁂 𑀅𑀧𑀼𑀘𑁆𑀙𑀣𑁇_  


(43.)

44. _“𑀆𑀤𑀺𑀲𑁆𑀲 𑀚𑀫𑁆𑀫𑀦𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀕𑁄𑀢𑁆𑀢𑀁 𑀩𑁆𑀭𑀽𑀳𑀺 𑀲𑀮𑀓𑁆𑀔𑀡𑀁;_  
_𑀫𑀦𑁆𑀢𑁂𑀲𑀼 𑀧𑀸𑀭𑀫𑀺𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀢𑀺 𑀯𑀸𑀘𑁂𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄”𑁇_  


(44.)

45. _“𑀯𑀻𑀲𑀁 𑀯𑀲𑁆𑀲𑀲𑀢𑀁 𑀆𑀬𑀼,_  
_𑀲𑁄 𑀘 𑀕𑁄𑀢𑁆𑀢𑁂𑀦 𑀩𑀸𑀯𑀭𑀻;_  
_𑀢𑀻𑀡𑀺𑀲𑁆𑀲 𑀮𑀓𑁆𑀔𑀡𑀸 𑀕𑀢𑁆𑀢𑁂,_  
_𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑁂𑀤𑀸𑀦 𑀧𑀸𑀭𑀕𑀽𑁇_  


(45.)

46. _𑀮𑀓𑁆𑀔𑀡𑁂 𑀇𑀢𑀺𑀳𑀸𑀲𑁂 𑀘,_  
_𑀲𑀦𑀺𑀖𑀡𑁆𑀟𑀼𑀲𑀓𑁂𑀝𑀼𑀪𑁂;_  
_𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺 𑀯𑀸𑀘𑁂𑀢𑀺,_  
_𑀲𑀥𑀫𑁆𑀫𑁂 𑀧𑀸𑀭𑀫𑀺𑀁 𑀕𑀢𑁄”𑁇_  


(46.)

47. _“𑀮𑀓𑁆𑀔𑀡𑀸𑀦𑀁 𑀧𑀯𑀺𑀘𑀬𑀁,_  
_𑀩𑀸𑀯𑀭𑀺𑀲𑁆𑀲 𑀦𑀭𑀼𑀢𑁆𑀢𑀫;_  
_𑀢𑀡𑁆𑀳𑀘𑁆𑀙𑀺𑀤 𑀧𑀓𑀸𑀲𑁂𑀳𑀺,_  
_𑀫𑀸 𑀦𑁄 𑀓𑀗𑁆𑀔𑀸𑀬𑀺𑀢𑀁 𑀅𑀳𑀼”𑁇_  


(47.)

48. _“𑀫𑀼𑀔𑀁 𑀚𑀺𑀯𑁆𑀳𑀸𑀬 𑀙𑀸𑀤𑁂𑀢𑀺,_  
_𑀉𑀡𑁆𑀡𑀲𑁆𑀲 𑀪𑀫𑀼𑀓𑀦𑁆𑀢𑀭𑁂;_  
_𑀓𑁄𑀲𑁄𑀳𑀺𑀢𑀁 𑀯𑀢𑁆𑀣𑀕𑀼𑀬𑁆𑀳𑀁,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀫𑀸𑀡𑀯”𑁇_  


(48.)

49. _𑀧𑀼𑀘𑁆𑀙𑀜𑁆𑀳𑀺 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀲𑀼𑀡𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀳𑁂 𑀯𑀺𑀬𑀸𑀓𑀢𑁂;_  
_𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑁂𑀢𑀺 𑀚𑀦𑁄 𑀲𑀩𑁆𑀩𑁄,_  
_𑀯𑁂𑀤𑀚𑀸𑀢𑁄 𑀓𑀢𑀜𑁆𑀚𑀮𑀻𑁇_  


(49.)

50. _“𑀓𑁄 𑀦𑀼 𑀤𑁂𑀯𑁄 𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀯𑀸,_  
_𑀇𑀦𑁆𑀤𑁄 𑀯𑀸𑀧𑀺 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀺;_  
_𑀫𑀦𑀲𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁂 𑀧𑀜𑁆𑀳𑁂,_  
_𑀓𑀫𑁂𑀢𑀁 𑀧𑀝𑀺𑀪𑀸𑀲𑀢𑀺”𑁇_  


(50.)

51. _“𑀫𑀼𑀤𑁆𑀥𑀁 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀜𑁆𑀘,_  
_𑀩𑀸𑀯𑀭𑀻 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀢𑀺;_  
_𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀳𑀺 𑀪𑀕𑀯𑀸,_  
_𑀓𑀗𑁆𑀔𑀁 𑀯𑀺𑀦𑀬 𑀦𑁄 𑀇𑀲𑁂”𑁇_  


(51.)

52. _“𑀅𑀯𑀺𑀚𑁆𑀚𑀸 𑀫𑀼𑀤𑁆𑀥𑀸𑀢𑀺 𑀚𑀸𑀦𑀸𑀳𑀺,_  
_𑀯𑀺𑀚𑁆𑀚𑀸 𑀫𑀼𑀤𑁆𑀥𑀸𑀥𑀺𑀧𑀸𑀢𑀺𑀦𑀻;_  
_𑀲𑀤𑁆𑀥𑀸𑀲𑀢𑀺𑀲𑀫𑀸𑀥𑀻𑀳𑀺,_  
_𑀙𑀦𑁆𑀤𑀯𑀻𑀭𑀺𑀬𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑀸”𑁇_  


(52.)

53. _𑀢𑀢𑁄 𑀯𑁂𑀤𑁂𑀦 𑀫𑀳𑀢𑀸,_  
_𑀲𑀦𑁆𑀣𑀫𑁆𑀪𑁂𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀡𑀯𑁄;_  
_𑀏𑀓𑀁𑀲𑀁 𑀅𑀚𑀺𑀦𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀧𑀢𑀺𑁇_  


(53.)

54. _“𑀩𑀸𑀯𑀭𑀻 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑁄𑀢𑁄,_  
_𑀲𑀳 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀫𑀸𑀭𑀺𑀲;_  
_𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁄 𑀲𑀼𑀫𑀦𑁄,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀢𑀺 𑀘𑀓𑁆𑀔𑀼𑀫”𑁇_  


(54.)

55. _“𑀲𑀼𑀔𑀺𑀢𑁄 𑀩𑀸𑀯𑀭𑀻 𑀳𑁄𑀢𑀼,_  
_𑀲𑀳 𑀲𑀺𑀲𑁆𑀲𑁂𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀢𑁆𑀯𑀜𑁆𑀘𑀸𑀧𑀺 𑀲𑀼𑀔𑀺𑀢𑁄 𑀳𑁄𑀳𑀺,_  
_𑀘𑀺𑀭𑀁 𑀚𑀻𑀯𑀸𑀳𑀺 𑀫𑀸𑀡𑀯𑁇_  


(55.)

56. _𑀩𑀸𑀯𑀭𑀺𑀲𑁆𑀲 𑀘 𑀢𑀼𑀬𑁆𑀳𑀁 𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀩𑁆𑀩𑀲𑀁𑀲𑀬𑀁;_  
_𑀓𑀢𑀸𑀯𑀓𑀸𑀲𑀸 𑀧𑀼𑀘𑁆𑀙𑀯𑁆𑀳𑁄,_  
_𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀫𑀦𑀲𑀺𑀘𑁆𑀙𑀣”𑁇_  


(56.)

57. _𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀓𑀢𑁄𑀓𑀸𑀲𑁄,_  
_𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀜𑁆𑀚𑀮𑀻;_  
_𑀅𑀚𑀺𑀢𑁄 𑀧𑀞𑀫𑀁 𑀧𑀜𑁆𑀳𑀁,_  
_𑀢𑀢𑁆𑀣 𑀧𑀼𑀘𑁆𑀙𑀺 𑀢𑀣𑀸𑀕𑀢𑀁𑁇_  


---

58. 𑀯𑀢𑁆𑀣𑀼𑀕𑀸𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
