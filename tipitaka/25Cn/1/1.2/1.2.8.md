# 𑁧𑁇𑁨𑁇𑁮 𑀳𑁂𑀫𑀓𑀫𑀸𑀡𑀯𑀧𑀼𑀘𑁆𑀙𑀸

* 𑀧𑀸𑀭𑀸𑀬𑀦𑀯𑀕𑁆𑀕𑀕𑀸𑀣𑀸

* 𑀧𑀼𑀘𑁆𑀙𑀸

* 𑀳𑁂𑀫𑀓𑀫𑀸𑀡𑀯𑀧𑀼𑀘𑁆𑀙𑀸

(109.)

118\. _“𑀬𑁂 𑀫𑁂 𑀧𑀼𑀩𑁆𑀩𑁂 𑀯𑀺𑀬𑀸𑀓𑀁𑀲𑀼, __(𑀇𑀘𑁆𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀳𑁂𑀫𑀓𑁄)___  
_𑀳𑀼𑀭𑀁 𑀕𑁄𑀢𑀫𑀲𑀸𑀲𑀦𑀸;_  
_𑀇𑀘𑁆𑀘𑀸𑀲𑀺 𑀇𑀢𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺,_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀇𑀢𑀺𑀳𑀻𑀢𑀺𑀳𑀁;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀢𑀓𑁆𑀓𑀯𑀟𑁆𑀠𑀦𑀁,_  
_𑀦𑀸𑀳𑀁 𑀢𑀢𑁆𑀣 𑀅𑀪𑀺𑀭𑀫𑀺𑀁𑁇 (𑁧)_  


(110.)

119\. _𑀢𑁆𑀯𑀜𑁆𑀘 𑀫𑁂 𑀥𑀫𑁆𑀫𑀫𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀢𑀡𑁆𑀳𑀸𑀦𑀺𑀕𑁆𑀖𑀸𑀢𑀦𑀁 𑀫𑀼𑀦𑀺;_  
_𑀬𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁄 𑀘𑀭𑀁,_  
_𑀢𑀭𑁂 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀁”𑁇 (𑁨)_  


(111.)

120\. _“𑀇𑀥 𑀤𑀺𑀝𑁆𑀞𑀲𑀼𑀢𑀫𑀼𑀢𑀯𑀺𑀜𑁆𑀜𑀸𑀢𑁂𑀲𑀼,_  
_𑀧𑀺𑀬𑀭𑀽𑀧𑁂𑀲𑀼 𑀳𑁂𑀫𑀓;_  
_𑀙𑀦𑁆𑀤𑀭𑀸𑀕𑀯𑀺𑀦𑁄𑀤𑀦𑀁,_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀧𑀤𑀫𑀘𑁆𑀘𑀼𑀢𑀁𑁇 (𑁩)_  


(112.)

121\. _𑀏𑀢𑀤𑀜𑁆𑀜𑀸𑀬 𑀬𑁂 𑀲𑀢𑀸,_  
_𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀦𑀺𑀩𑁆𑀩𑀼𑀢𑀸;_  
_𑀉𑀧𑀲𑀦𑁆𑀢𑀸 𑀘 𑀢𑁂 𑀲𑀤𑀸,_  
_𑀢𑀺𑀡𑁆𑀡𑀸 𑀮𑁄𑀓𑁂 𑀯𑀺𑀲𑀢𑁆𑀢𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 (𑁪)_  


---
