# 𑁧𑁇𑁨𑁇𑁧 𑀅𑀚𑀺𑀢𑀫𑀸𑀡𑀯𑀧𑀼𑀘𑁆𑀙𑀸

### 𑀧𑀸𑀭𑀸𑀬𑀦𑀯𑀕𑁆𑀕𑀕𑀸𑀣𑀸

### 𑀧𑀼𑀘𑁆𑀙𑀸

### 𑀅𑀚𑀺𑀢𑀫𑀸𑀡𑀯𑀧𑀼𑀘𑁆𑀙𑀸

(57.)

59\. _“𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀺𑀯𑀼𑀢𑁄 𑀮𑁄𑀓𑁄, __(𑀇𑀘𑁆𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑀺𑀢𑁄)___  
_𑀓𑁂𑀦𑀲𑁆𑀲𑀼 𑀦𑀧𑁆𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀓𑀺𑀲𑁆𑀲𑀸𑀪𑀺𑀮𑁂𑀧𑀦𑀁 𑀩𑁆𑀭𑀽𑀲𑀺,_  
_𑀓𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀁”𑁇 (𑁧)_  


(58.)

60\. _“𑀅𑀯𑀺𑀚𑁆𑀚𑀸𑀬 𑀦𑀺𑀯𑀼𑀢𑁄 𑀮𑁄𑀓𑁄, __(𑀅𑀚𑀺𑀢𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀯𑁂𑀯𑀺𑀘𑁆𑀙𑀸 𑀧𑀫𑀸𑀤𑀸 𑀦𑀧𑁆𑀧𑀓𑀸𑀲𑀢𑀺;_  
_𑀚𑀧𑁆𑀧𑀸𑀪𑀺𑀮𑁂𑀧𑀦𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑁆𑀲 𑀫𑀳𑀩𑁆𑀪𑀬𑀁”𑁇 (𑁨)_  


(59.)

61\. _“𑀲𑀯𑀦𑁆𑀢𑀺 𑀲𑀩𑁆𑀩𑀥𑀺 𑀲𑁄𑀢𑀸, __(𑀇𑀘𑁆𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑀺𑀢𑁄)___  
_𑀲𑁄𑀢𑀸𑀦𑀁 𑀓𑀺𑀁 𑀦𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀲𑁄𑀢𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀁 𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑁂𑀦 𑀲𑁄𑀢𑀸 𑀧𑀺𑀥𑀺𑀬𑁆𑀬𑀭𑁂”𑁇 (𑁩)_  


(60.)

62\. _“𑀬𑀸𑀦𑀺 𑀲𑁄𑀢𑀸𑀦𑀺 𑀮𑁄𑀓𑀲𑁆𑀫𑀺𑀁, __(𑀅𑀚𑀺𑀢𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀲𑀢𑀺 𑀢𑁂𑀲𑀁 𑀦𑀺𑀯𑀸𑀭𑀡𑀁;_  
_𑀲𑁄𑀢𑀸𑀦𑀁 𑀲𑀁𑀯𑀭𑀁 𑀩𑁆𑀭𑀽𑀫𑀺,_  
_𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀢𑁂 𑀧𑀺𑀥𑀺𑀬𑁆𑀬𑀭𑁂”𑁇 (𑁪)_  


(61.)

63\. _“𑀧𑀜𑁆𑀜𑀸 𑀘𑁂𑀯 𑀲𑀢𑀺 𑀘𑀸𑀧𑀺, __(𑀇𑀘𑁆𑀘𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀚𑀺𑀢𑁄)___  
_𑀦𑀸𑀫𑀭𑀽𑀧𑀜𑁆𑀘 𑀫𑀸𑀭𑀺𑀲;_  
_𑀏𑀢𑀁 𑀫𑁂 𑀧𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺,_  
_𑀓𑀢𑁆𑀣𑁂𑀢𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺”𑁇 (𑁫)_  


(62.)

64\. _“𑀬𑀫𑁂𑀢𑀁 𑀧𑀜𑁆𑀳𑀁 𑀅𑀧𑀼𑀘𑁆𑀙𑀺,_  
_𑀅𑀚𑀺𑀢 𑀢𑀁 𑀯𑀤𑀸𑀫𑀺 𑀢𑁂;_  
_𑀬𑀢𑁆𑀣 𑀦𑀸𑀫𑀜𑁆𑀘 𑀭𑀽𑀧𑀜𑁆𑀘,_  
_𑀅𑀲𑁂𑀲𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺;_  
_𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀦𑀺𑀭𑁄𑀥𑁂𑀦,_  
_𑀏𑀢𑁆𑀣𑁂𑀢𑀁 𑀉𑀧𑀭𑀼𑀚𑁆𑀛𑀢𑀺”𑁇 (𑁬)_  


(63.)

65\. _“𑀬𑁂 𑀘 𑀲𑀗𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀸𑀲𑁂,_  
_𑀬𑁂 𑀘 𑀲𑁂𑀔𑀸 𑀧𑀼𑀣𑀽 𑀇𑀥;_  
_𑀢𑁂𑀲𑀁 𑀫𑁂 𑀦𑀺𑀧𑀓𑁄 𑀇𑀭𑀺𑀬𑀁,_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑀸𑀭𑀺𑀲”𑁇 (𑁭)_  


(64.)

66\. _“𑀓𑀸𑀫𑁂𑀲𑀼 𑀦𑀸𑀪𑀺𑀕𑀺𑀚𑁆𑀛𑁂𑀬𑁆𑀬,_  
_𑀫𑀦𑀲𑀸𑀦𑀸𑀯𑀺𑀮𑁄 𑀲𑀺𑀬𑀸;_  
_𑀓𑀼𑀲𑀮𑁄 𑀲𑀩𑁆𑀩𑀥𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀲𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁂”𑀢𑀺𑁇 (𑁮)_  


---
