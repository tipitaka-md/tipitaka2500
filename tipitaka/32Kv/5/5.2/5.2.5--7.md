# 𑁫𑁇𑁨𑁇𑁫--𑁭 𑀢𑀺𑀲𑁆𑀲𑁄𑀧𑀺𑀓𑀣𑀸

### 𑀔𑀼𑀤𑁆𑀤𑀓𑀅𑀟𑁆𑀠𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀩𑀸𑀯𑀻𑀲𑀢𑀺𑀫𑀯𑀕𑁆𑀕

### 𑀢𑀺𑀲𑁆𑀲𑁄𑀧𑀺𑀓𑀣𑀸

(898.)

2753\. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀕𑀩𑁆𑀪𑀲𑁂𑀬𑁆𑀬𑀸𑀬 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀲𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

(899.)

2754\. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀧𑀺𑀦𑀕𑀢𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀲𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀥𑀫𑁆𑀫𑀸𑀪𑀺𑀲𑀫𑀬𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

(900.)

2755\. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀲𑀼𑀧𑀺𑀦𑀕𑀢𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀲𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀼𑀝𑁆𑀞𑀲𑁆𑀲𑀢𑀺𑀲𑁆𑀲 𑀅𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀲𑁆𑀲 𑀅𑀭𑀳𑀢𑁆𑀢𑀧𑁆𑀧𑀢𑁆𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

---

2756\. 𑀢𑀺𑀲𑁆𑀲𑁄𑀧𑀺𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
