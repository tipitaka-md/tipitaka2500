# 𑁪𑁇𑁫𑁇𑁪 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀸

* 𑀦𑀺𑀕𑁆𑀕𑀳𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀯𑀻𑀲𑀢𑀺𑀫𑀯𑀕𑁆𑀕

* 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀸

(869.)

2676\. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁂𑀲𑀼 𑀤𑁂𑀯𑀸𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀤𑁂𑀯𑀮𑁄𑀓𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀬𑁄𑀦𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂…  𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀓𑀻𑀝𑀸 𑀧𑀝𑀗𑁆𑀕𑀸 𑀫𑀓𑀲𑀸 𑀫𑀓𑁆𑀔𑀺𑀓𑀸 𑀅𑀳𑀻 𑀯𑀺𑀘𑁆𑀙𑀺𑀓𑀸 𑀲𑀢𑀧𑀤𑀻 𑀕𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀸𑀤𑀸𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

(870.)

2677\. 𑁆 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀏𑀭𑀸𑀯𑀡𑁄 𑀦𑀸𑀫 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀤𑀺𑀩𑁆𑀩𑀁 𑀬𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀏𑀭𑀸𑀯𑀡𑁄 𑀦𑀸𑀫 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀲𑀳𑀲𑁆𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀤𑀺𑀩𑁆𑀩𑀁 𑀬𑀸𑀦𑀁, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸”𑀢𑀺𑁇

(871.)

2678\. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀳𑀢𑁆𑀣𑀺𑀩𑀦𑁆𑀥𑀸 𑀅𑀲𑁆𑀲𑀩𑀦𑁆𑀥𑀸 𑀬𑀸𑀯𑀲𑀺𑀓𑀸 𑀓𑀸𑀭𑀡𑀺𑀓𑀸 𑀪𑀢𑁆𑀢𑀓𑀸𑀭𑀓𑀸𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀢𑀺𑁇

---

2679\. 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
