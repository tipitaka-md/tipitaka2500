# 𑁪𑁇𑁨𑁇𑁭 𑀦𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁𑀲𑀁𑀖𑁄𑀤𑀓𑁆𑀔𑀺𑀡𑀁𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻𑀢𑀺𑀓𑀣𑀸

### 𑀦𑀺𑀕𑁆𑀕𑀳𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀲𑀢𑁆𑀢𑀭𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀦𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁𑀲𑀁𑀖𑁄𑀤𑀓𑁆𑀔𑀺𑀡𑀁𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻𑀢𑀺𑀓𑀣𑀸

(793.)

2473\. ๐ 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

2474\. ๐ 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

2475\. ๐ 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀘𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀆𑀭𑀸𑀥𑁂𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀘𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀆𑀭𑀸𑀥𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

2476\. ๐ 𑀦𑀦𑀼 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀘𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀤𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀤𑀢𑁆𑀯𑀸…  𑀕𑀺𑀮𑀸𑀦𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀤𑀢𑁆𑀯𑀸…  𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀤𑀢𑁆𑀯𑀸…  𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀤𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀧𑀸𑀦𑀻𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀆𑀭𑀸𑀥𑁂𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀅𑀢𑁆𑀣𑀺 𑀓𑁂𑀘𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀸𑀦𑀻𑀬𑀁 𑀤𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀆𑀭𑀸𑀥𑁂𑀦𑁆𑀢𑀺, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻”𑀢𑀺𑁇

(794.)

2477\. 𑁆 𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀫𑀕𑁆𑀕𑁄 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀺, 𑀨𑀮𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

---

2478\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀺𑀲𑁄𑀥𑁂𑀢𑀻𑀢𑀺𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
