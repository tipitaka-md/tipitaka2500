# 𑁪𑁇𑁨𑁇𑁯 𑀦𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁𑀲𑀁𑀖𑀲𑁆𑀲𑀤𑀺𑀦𑁆𑀦𑀁𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆𑀢𑀺𑀓𑀣𑀸

* 𑀦𑀺𑀕𑁆𑀕𑀳𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀲𑀢𑁆𑀢𑀭𑀲𑀫𑀯𑀕𑁆𑀕

* 𑀦𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁𑀲𑀁𑀖𑀲𑁆𑀲𑀤𑀺𑀦𑁆𑀦𑀁𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆𑀢𑀺𑀓𑀣𑀸

(797.)

2484\. ๐ 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀧𑀸𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑁄 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀲𑀁𑀖𑁄 𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑁄…𑀧𑁂…  𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀁 𑀧𑀼𑀜𑁆𑀜𑀓𑁆𑀔𑁂𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺𑁇

2485\. ๐ 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀕𑀸 𑀅𑀝𑁆𑀞 𑀧𑀼𑀭𑀺𑀲𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀬𑁆𑀬𑀸 𑀯𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺𑁇

(798.)

2486\. ๐ 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸—  “𑀲𑀁𑀖𑁂, 𑀕𑁄𑀢𑀫𑀺, 𑀤𑁂𑀳𑀺, 𑀲𑀁𑀖𑁂 𑀢𑁂 𑀤𑀺𑀦𑁆𑀦𑁂 𑀅𑀳𑀜𑁆𑀘𑁂𑀯 𑀧𑀽𑀚𑀺𑀢𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀁𑀖𑁄 𑀘𑀸”𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑁂𑀯 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆𑀢𑀺𑁇

2487\. ๐ 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—

2488\. _“𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀧𑁂𑀓𑁆𑀔𑀸𑀦 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀓𑀭𑁄𑀢𑀁 𑀑𑀧𑀥𑀺𑀓𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀓𑀢𑁆𑀣 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆”𑀢𑀺𑁇_  


2489\. _“𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸,_  
_𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀘 𑀨𑀮𑁂 𑀞𑀺𑀢𑀸;_  
_𑀏𑀲 𑀲𑀁𑀖𑁄 𑀉𑀚𑀼𑀪𑀽𑀢𑁄,_  
_𑀧𑀜𑁆𑀜𑀸𑀲𑀻𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑁄𑁇_  


2490\. _𑀬𑀚𑀫𑀸𑀦𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀼𑀜𑁆𑀜𑀧𑁂𑀓𑁆𑀔𑀸𑀦 𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀓𑀭𑁄𑀢𑀁 𑀑𑀧𑀥𑀺𑀓𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀲𑀁𑀖𑁂 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆𑀢𑀺𑁇_  


2491\. _𑀏𑀲𑁄 𑀳𑀺 𑀲𑀁𑀖𑁄 𑀯𑀺𑀧𑀼𑀮𑁄 𑀫𑀳𑀕𑁆𑀕𑀢𑁄,_  
_𑀏𑀲𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁄 𑀉𑀤𑀥𑀻𑀯 𑀲𑀸𑀕𑀭𑁄;_  
_𑀏𑀢𑁂 𑀳𑀺 𑀲𑁂𑀝𑁆𑀞𑀸 𑀦𑀭𑀯𑀻𑀭𑀲𑀸𑀯𑀓𑀸,_  
_𑀧𑀪𑀗𑁆𑀓𑀭𑀸 𑀥𑀫𑁆𑀫𑀫𑀼𑀤𑀻𑀭𑀬𑀦𑁆𑀢𑀺𑁇_  


2492\. _𑀢𑁂𑀲𑀁 𑀲𑀼𑀤𑀺𑀦𑁆𑀦𑀁 𑀲𑀼𑀳𑀼𑀢𑀁 𑀲𑀼𑀬𑀺𑀝𑁆𑀞𑀁,_  
_𑀬𑁂 𑀲𑀁𑀖𑀫𑀼𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀤𑀤𑀦𑁆𑀢𑀺 𑀤𑀸𑀦𑀁;_  
_𑀲𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀲𑀁𑀖𑀕𑀢𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸,_  
_𑀫𑀳𑀧𑁆𑀨𑀮𑀸 𑀮𑁄𑀓𑀯𑀺𑀤𑀽𑀦 𑀯𑀡𑁆𑀡𑀺𑀢𑀸𑁇_  


2493\. _𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀬𑀜𑁆𑀜𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀸,_  
_𑀬𑁂 𑀯𑁂𑀤𑀚𑀸𑀢𑀸 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀺 𑀮𑁄𑀓𑁂;_  
_𑀯𑀺𑀦𑁂𑀬𑁆𑀬 𑀫𑀘𑁆𑀙𑁂𑀭𑀫𑀮𑀁 𑀲𑀫𑀽𑀮𑀁,_  
_𑀅𑀦𑀺𑀦𑁆𑀤𑀺𑀢𑀸 𑀲𑀕𑁆𑀕𑀫𑀼𑀧𑁂𑀦𑁆𑀢𑀺 𑀞𑀸𑀦𑀦𑁆”𑀢𑀺𑁇_  


2494\. 𑀅𑀢𑁆𑀣𑁂𑀯 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆𑀢𑀺𑁇

---

2495\. 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀺𑀦𑁆𑀦𑀁 𑀫𑀳𑀧𑁆𑀨𑀮𑀦𑁆𑀢𑀺𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
