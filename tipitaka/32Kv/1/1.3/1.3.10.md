# 𑁧𑁇𑁩𑁇𑁧𑁦 𑀲𑀁𑀯𑀭𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

### 𑀲𑀁𑀯𑀭𑀓𑀣𑀸

(379.)

1236. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1237. ๐ 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1238. ๐ 𑀦𑀦𑀼 𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀭𑁄, 𑀬𑀫𑁆𑀳𑀸 𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1239. 𑀆𑀚𑀸𑀦𑀸𑀳𑀺 𑀦𑀺𑀕𑁆𑀕𑀳𑀁𑁇 𑀳𑀜𑁆𑀘𑀺 𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀭𑁄, 𑀬𑀫𑁆𑀳𑀸 𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀦𑁆”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀭𑁄, 𑀬𑀫𑁆𑀳𑀸 𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀦𑁆’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

1240. 𑀦𑁄 𑀘𑁂 𑀧𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀭𑁄, 𑀬𑀫𑁆𑀳𑀸 𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀦𑁆”𑀢𑀺, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄 ‘𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀁, 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀅𑀲𑀁𑀯𑀭𑁄, 𑀬𑀫𑁆𑀳𑀸 𑀅𑀲𑀁𑀯𑀭𑀸 𑀲𑀁𑀯𑀭𑁄 𑀲𑀻𑀮𑀦𑁆’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

1241. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀅𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀅𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1242. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀅𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀅𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

(380.)

1243. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1244. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1245. ๐ 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1246. ๐ 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1247. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1248. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀅𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1249. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1250. ๐ 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀢𑁆𑀣𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀲𑀼 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀸 𑀯𑁂𑀭𑀫𑀡𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀢𑀢𑁆𑀣 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1251. 𑁆 𑀦𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀲𑀩𑁆𑀩𑁂 𑀤𑁂𑀯𑀸 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀺𑀦𑁄 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀺𑀦𑁄 𑀓𑀸𑀫𑁂𑀲𑀼𑀫𑀺𑀘𑁆𑀙𑀸𑀘𑀸𑀭𑀺𑀦𑁄 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀼𑀭𑀸𑀫𑁂𑀭𑀬𑀫𑀚𑁆𑀚𑀧𑀫𑀸𑀤𑀝𑁆𑀞𑀸𑀬𑀺𑀦𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂…  𑀢𑁂𑀦 𑀳𑀺 𑀅𑀢𑁆𑀣𑀺 𑀤𑁂𑀯𑁂𑀲𑀼 𑀲𑀁𑀯𑀭𑁄𑀢𑀺𑁇
