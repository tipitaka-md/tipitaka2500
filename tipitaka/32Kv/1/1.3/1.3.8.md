# 𑁧𑁇𑁩𑁇𑁮 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀓𑀣𑀸

* 𑀫𑀳𑀸𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀢𑀢𑀺𑀬𑀯𑀕𑁆𑀕

* 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀓𑀣𑀸

(375.)

1213\. ๐ 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁, 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀫𑀁𑀲𑀲𑁄𑀢𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1214\. ๐ 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀬𑀸𑀤𑀺𑀲𑀁 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁, 𑀬𑀸𑀤𑀺𑀲𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀢𑀸𑀤𑀺𑀲𑀁 𑀫𑀁𑀲𑀲𑁄𑀢𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1215\. ๐ 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀢𑀜𑁆𑀜𑁂𑀯 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀢𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁, 𑀢𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀢𑀁 𑀫𑀁𑀲𑀲𑁄𑀢𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1216\. ๐ 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀬𑀸𑀤𑀺𑀲𑁄 𑀫𑀁𑀲𑀲𑁄𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑀬𑁄 𑀆𑀦𑀼𑀪𑀸𑀯𑁄 𑀕𑁄𑀘𑀭𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀤𑀺𑀩𑁆𑀩𑀲𑁆𑀲 𑀲𑁄𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑀬𑁄 𑀆𑀦𑀼𑀪𑀸𑀯𑁄 𑀕𑁄𑀘𑀭𑁄𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1217\. ๐ 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1218\. ๐ 𑀉𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀤𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1219\. ๐ 𑀓𑀸𑀫𑀸𑀯𑀘𑀭𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1220\. ๐ 𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀅𑀭𑀽𑀧𑀸𑀯𑀘𑀭𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀳𑀼𑀢𑁆𑀯𑀸 𑀅𑀧𑀭𑀺𑀬𑀸𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

(376.)

1221\. ๐ 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1222\. ๐ 𑀫𑀁𑀲𑀲𑁄𑀢𑀁 𑀥𑀫𑁆𑀫𑀼𑀧𑀢𑁆𑀣𑀤𑁆𑀥𑀁 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁 𑀳𑁄𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀏𑀓𑀁𑀬𑁂𑀯 𑀲𑁄𑀢𑀦𑁆𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂…𑀧𑁂… 𑁇

1223\. ๐ 𑀏𑀓𑀁𑀬𑁂𑀯 𑀲𑁄𑀢𑀦𑁆𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀤𑁆𑀯𑁂 𑀲𑁄𑀢𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑀸—  “𑀫𑀁𑀲𑀲𑁄𑀢𑀁, 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀦𑁆”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀤𑁆𑀯𑁂 𑀲𑁄𑀢𑀸𑀦𑀺 𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀢𑀸—  𑀫𑀁𑀲𑀲𑁄𑀢𑀁, 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀁, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀏𑀓𑀜𑁆𑀜𑁂𑀯 𑀲𑁄𑀢𑀦𑁆”𑀢𑀺𑁇

---

1224\. 𑀤𑀺𑀩𑁆𑀩𑀲𑁄𑀢𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
