# 𑁧𑁇𑁧𑁇𑁧𑁇𑁭𑁇𑁧 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀸𑀦𑀼𑀮𑁄𑀫

* 𑀫𑀳𑀸𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀕𑁆𑀕𑀮𑀓𑀣𑀸

* 𑀅𑀯𑀬𑀯𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀸𑀦𑀼𑀮𑁄𑀫

(16.)

60\. 𑁆 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂𑁇

61\. 𑀆𑀚𑀸𑀦𑀸𑀳𑀺 𑀦𑀺𑀕𑁆𑀕𑀳𑀁𑁇 𑀳𑀜𑁆𑀘𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

62\. 𑀦𑁄 𑀘𑁂 𑀧𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀲𑀩𑁆𑀩𑁂𑀲𑀼 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸…𑀧𑁂… 𑁇
