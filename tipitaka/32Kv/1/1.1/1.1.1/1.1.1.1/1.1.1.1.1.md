# 𑁧𑁇𑁧𑁇𑁧𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀫𑀳𑀸𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀧𑀞𑀫𑀯𑀕𑁆𑀕

* 𑀧𑀼𑀕𑁆𑀕𑀮𑀓𑀣𑀸

* 𑀲𑀼𑀤𑁆𑀥𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(1.)

2\. ๐ 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂𑁇

3\. 𑀆𑀚𑀸𑀦𑀸𑀳𑀺 𑀦𑀺𑀕𑁆𑀕𑀳𑀁𑁇 𑀳𑀜𑁆𑀘𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

4\. 𑀦𑁄 𑀘𑁂 𑀧𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

---

5\. 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀜𑁆𑀘𑀓𑀁𑁇



(2.)

6\. 𑁆 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸𑀢𑀺? 𑀦 𑀳𑁂𑀯𑀁 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂𑁇

7\. 𑀆𑀚𑀸𑀦𑀸𑀳𑀺 𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀁𑁇 𑀳𑀜𑁆𑀘𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

8\. 𑀦𑁄 𑀘𑁂 𑀧𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

---

9\. 𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑁇



(3.)

10\. 𑁆 𑀢𑁆𑀯𑀜𑁆𑀘𑁂 𑀧𑀦 𑀫𑀜𑁆𑀜𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺, 𑀢𑁂𑀦 𑀢𑀯 𑀢𑀢𑁆𑀣 𑀳𑁂𑀢𑀸𑀬 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀳𑁂𑀯𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀁 𑀳𑁂𑀯𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁂𑁇 𑀅𑀣 𑀢𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫𑁇 𑀲𑀼𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑁄 𑀘 𑀳𑁄𑀲𑀺𑁇

11\. 𑀳𑀜𑁆𑀘𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

12\. 𑀦𑁄 𑀘𑁂 𑀧𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀦𑀼𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀇𑀤𑀁 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

---

13\. 𑀦𑀺𑀕𑁆𑀕𑀳𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑁇



(4.)

14\. 𑁆 𑀏𑀲𑁂 𑀘𑁂 𑀤𑀼𑀦𑁆𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑁂 𑀳𑁂𑀯𑀫𑁂𑀯𑀁 𑀢𑀢𑁆𑀣 𑀤𑀓𑁆𑀔𑁇 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  “𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦”, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀦𑁄 𑀘 𑀫𑀬𑀁 𑀢𑀬𑀸 𑀢𑀢𑁆𑀣 𑀳𑁂𑀢𑀸𑀬 𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀳𑁂𑀯𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀳𑁂𑀯𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀣 𑀫𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀲𑀺𑁇 𑀤𑀼𑀦𑁆𑀦𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀘 𑀳𑁄𑀫𑁇

15\. 𑀳𑀜𑁆𑀘𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

16\. 𑀦𑁄 𑀘𑁂 𑀧𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀇𑀤𑀁 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

---

17\. 𑀉𑀧𑀦𑀬𑀦𑀘𑀢𑀼𑀓𑁆𑀓𑀁𑁇



(5.)

18\. 𑁆 𑀦 𑀳𑁂𑀯𑀁 𑀦𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁂𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀬𑀁 𑀦𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀲𑀺—  “𑀳𑀜𑁆𑀘𑀺 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  ‘𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀫𑀺𑀘𑁆𑀙𑀸𑁇

19\. 𑀦𑁄 𑀘𑁂 𑀧𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺, 𑀦𑁄 𑀘 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸”𑀢𑀺𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀯𑀤𑁂𑀲𑀺—  “𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂 𑀔𑁄—  ‘𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦’, 𑀦𑁄 𑀘 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  ‘𑀬𑁄 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑁄 𑀧𑀭𑀫𑀢𑁆𑀣𑁄, 𑀢𑀢𑁄 𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀮𑀩𑁆𑀪𑀢𑀺 𑀲𑀘𑁆𑀘𑀺𑀓𑀝𑁆𑀞𑀧𑀭𑀫𑀢𑁆𑀣𑁂𑀦𑀸’”𑀢𑀺 𑀇𑀤𑀁 𑀢𑁂 𑀫𑀺𑀘𑁆𑀙𑀸𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀬𑁂 𑀓𑀢𑁂 𑀦𑀺𑀕𑁆𑀕𑀳𑁂 𑀲𑁂 𑀦𑀺𑀕𑁆𑀕𑀳𑁂 𑀤𑀼𑀓𑁆𑀓𑀝𑁂𑁇 𑀲𑀼𑀓𑀢𑁂 𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂𑁇 𑀲𑀼𑀓𑀢𑀸 𑀧𑀝𑀺𑀧𑀸𑀤𑀦𑀸𑀢𑀺𑁇
