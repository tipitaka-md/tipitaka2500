# 𑁩𑁇𑁨𑁇𑁭 𑀏𑀓𑀩𑀻𑀚𑀻𑀓𑀣𑀸

* 𑀅𑀦𑀼𑀲𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

* 𑀏𑀓𑀩𑀻𑀚𑀻𑀓𑀣𑀸

(647.)

2121\. 𑁆 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀏𑀓𑀩𑀻𑀚𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀏𑀓𑀩𑀻𑀚𑀺𑀢𑀸𑀦𑀺𑀬𑀢𑁄”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀲𑁄 𑀏𑀓𑀩𑀻𑀚𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀲𑁄 𑀏𑀓𑀩𑀻𑀚𑀻, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀏𑀓𑀩𑀻𑀚𑀻 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀏𑀓𑀩𑀻𑀚𑀺𑀢𑀸𑀦𑀺𑀬𑀢𑁄”𑀢𑀺𑁇
