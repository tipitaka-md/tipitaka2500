# 𑁩𑁇𑁨𑁇𑁬 𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑀓𑀣𑀸

### 𑀅𑀦𑀼𑀲𑀬𑀧𑀡𑁆𑀡𑀸𑀲𑀓

### 𑀤𑁆𑀯𑀸𑀤𑀲𑀫𑀯𑀕𑁆𑀕

### 𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑀓𑀣𑀸

(646.)

2119\. 𑁆 𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁—  “𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑀢𑀸𑀦𑀺𑀬𑀢𑁄”𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀦𑀦𑀼 𑀲𑁄 𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑁄𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇 𑀳𑀜𑁆𑀘𑀺 𑀲𑁄 𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑁄, 𑀢𑁂𑀦 𑀯𑀢 𑀭𑁂 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁂—  “𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀓𑁄𑀮𑀁𑀓𑁄𑀮𑀢𑀸𑀦𑀺𑀬𑀢𑁄”𑀢𑀺𑁇
