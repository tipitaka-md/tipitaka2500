# 𑁩 𑀪𑀸𑀯𑀦𑀸𑀯𑀸𑀭

### 𑀪𑀸𑀯𑀦𑀸𑀯𑀸𑀭

(229.)

954\. »  𑀬𑁄 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

955\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀯𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

956\. »  𑀬𑁄 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀺 𑀲𑁄 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

957\. «  𑀬𑁄 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀦𑀧𑁆𑀧𑀚𑀳𑀢𑀺 𑀲𑁄 𑀓𑀼𑀲𑀮𑀁 𑀥𑀫𑁆𑀫𑀁 𑀦 𑀪𑀸𑀯𑁂𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸…𑀧𑁂… 𑁇
