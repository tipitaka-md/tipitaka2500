# 𑁧𑁇𑁨𑁇𑁧𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀧𑀤𑀲𑁄𑀥𑀦𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫

(17.)

75\. »  𑀓𑀼𑀲𑀮𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

76\. «  𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

77\. »  𑀅𑀓𑀼𑀲𑀮𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

78\. «  𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

79\. »  𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

80\. «  𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
