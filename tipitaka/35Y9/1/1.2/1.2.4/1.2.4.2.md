# 𑁧𑁇𑁨𑁇𑁪𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑀫𑀽𑀮𑀘𑀓𑁆𑀓𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(30.)

153\. »  𑀦 𑀓𑀼𑀲𑀮𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

154\. 𑀓𑀼𑀲𑀮𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

155\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

156\. »  𑀦 𑀓𑀼𑀲𑀮𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

157\. 𑀓𑀼𑀲𑀮𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

158\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(31.)

159\. »  𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

160\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

161\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

162\. »  𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

163\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

164\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(32.)

165\. »  𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

166\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

167\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

168\. »  𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

169\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

170\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

---
