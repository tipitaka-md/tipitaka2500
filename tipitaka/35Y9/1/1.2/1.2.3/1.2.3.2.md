# 𑁧𑁇𑁨𑁇𑁩𑁇𑁨 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

### 𑀲𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓

(26.)

126\. »  𑀦 𑀓𑀼𑀲𑀮𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

127\. 𑀓𑀼𑀲𑀮𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀼𑀲𑀮𑀸, 𑀥𑀫𑁆𑀫𑀸𑁇 𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

128\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

129\. »  𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

130\. 𑀅𑀓𑀼𑀲𑀮𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀓𑀼𑀲𑀮𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

131\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

132\. »  𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀦 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

133\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑁇 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀯𑀲𑁂𑀲𑀸 𑀦 𑀘𑁂𑀯 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀦 𑀘 𑀥𑀫𑁆𑀫𑀸𑁇

134\. «  𑀦 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
