# 𑁧𑁇𑁨𑁇𑁩𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀧𑀡𑁆𑀡𑀢𑁆𑀢𑀺𑀯𑀸𑀭𑀦𑀺𑀤𑁆𑀤𑁂𑀲

* 𑀲𑀼𑀤𑁆𑀥𑀥𑀫𑁆𑀫𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫

(25.)

117\. »  𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

118\. «  𑀥𑀫𑁆𑀫𑀸 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

119\. 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇

120\. »  𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

121\. «  𑀥𑀫𑁆𑀫𑀸 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

122\. 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸𑁇

123\. »  𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

124\. «  𑀥𑀫𑁆𑀫𑀸 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑀢𑀺?

125\. 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑁂𑀯 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀘𑁇 𑀅𑀯𑀲𑁂𑀲𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸𑁇
