# 𑁨𑁇𑁨𑁇𑁧𑁇𑁫 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

(107.)

479\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

480\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

481\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺? 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

482\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

(108.)

483\. »  𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺? 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

484\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇
