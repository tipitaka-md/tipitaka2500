# 𑁨𑁇𑁨𑁇𑁫𑁇𑁧 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀅𑀦𑀼𑀮𑁄𑀫𑀧𑀼𑀕𑁆𑀕𑀮

(143.)

614\. »  𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺?

615\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀢𑀭𑁂𑀲𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

616\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺?

617\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

618\. »  𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

619\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺?

620\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇

(144.)

621\. »  𑀬𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

622\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀻𑀢𑀺?

623\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀦𑀺𑀭𑁄𑀥𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇 𑀅𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺𑁇
