497\. »  𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

498\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

499\. »  𑀬𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

500\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

501\. »  𑀬𑀲𑁆𑀲 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
