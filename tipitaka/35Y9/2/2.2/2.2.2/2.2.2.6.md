# 𑁨𑁇𑁨𑁇𑁨𑁇𑁬 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

* 𑀅𑀢𑀻𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

(121.)

532\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

533\. 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑁂 𑀅𑀓𑀼𑀲𑀮𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇

534\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

535\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

536\. 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇

537\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(122.)

538\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

539\. 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀸𑀦𑀁 𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇 𑀲𑀼𑀤𑁆𑀥𑀸𑀯𑀸𑀲𑀁 𑀉𑀧𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇

540\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
