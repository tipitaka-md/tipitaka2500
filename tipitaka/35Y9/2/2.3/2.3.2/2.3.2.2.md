# 𑁨𑁇𑁩𑁇𑁨𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

* 𑀅𑀢𑀻𑀢𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(177.)

750\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

751\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

752\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

753\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

754\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇 𑀘𑀢𑀼𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇

(178.)

755\. »  𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

756\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺?

757\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇 𑀘𑀢𑀼𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇
