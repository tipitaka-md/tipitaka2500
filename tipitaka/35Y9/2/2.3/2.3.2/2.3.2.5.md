# 𑁨𑁇𑁩𑁇𑁨𑁇𑁫 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

* 𑀅𑀢𑀻𑀢𑀯𑀸𑀭

* 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀑𑀓𑀸𑀲

(183.)

773\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

774\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

775\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇

776\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇

(184.)

777\. »  𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣𑁇

778\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺? 𑀦𑀢𑁆𑀣𑀺𑁇
