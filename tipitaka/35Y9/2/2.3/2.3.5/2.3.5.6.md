# 𑁨𑁇𑁩𑁇𑁫𑁇𑁬 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

### 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

### 𑀉𑀧𑁆𑀧𑀸𑀤𑀦𑀺𑀭𑁄𑀥𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀕𑀢𑀯𑀸𑀭

### 𑀧𑀘𑁆𑀘𑀦𑀻𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑀓𑀸𑀲

(217.)

905\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺?

906\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

907\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻𑀢𑀺?

908\. 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑀸𑀦𑀁 𑀬𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀕𑁆𑀕𑀫𑀕𑁆𑀕𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇

909\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺?

910\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

911\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

(218.)

912\. »  𑀬𑀲𑁆𑀲 𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺?

913\. 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀅𑀓𑀼𑀲𑀮𑀯𑀺𑀧𑁆𑀧𑀬𑀼𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀸𑀤𑀓𑁆𑀔𑀡𑁂 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑀸𑀦𑀁 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀧𑀘𑁆𑀙𑀺𑀫𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀗𑁆𑀕𑀓𑁆𑀔𑀡𑁂 𑀢𑁂𑀲𑀁 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀺 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

914\. «  𑀬𑀲𑁆𑀲 𑀯𑀸 𑀧𑀦 𑀬𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀦𑀺𑀭𑀼𑀚𑁆𑀛𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀲𑁆𑀲 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀦 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇
