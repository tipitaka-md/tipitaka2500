# 𑁨𑁇𑁧𑁇𑁩𑁇𑁨 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

* 𑀧𑀯𑀢𑁆𑀢𑀺𑀯𑀸𑀭

* 𑀉𑀧𑁆𑀧𑀸𑀤𑀯𑀸𑀭

* 𑀅𑀦𑀸𑀕𑀢𑀯𑀸𑀭

* 𑀅𑀦𑀼𑀮𑁄𑀫𑀑𑀓𑀸𑀲

(59.)

283\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

284\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

285\. »  𑀬𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

286\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺?

287\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇

(60.)

288\. »  𑀬𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺? 𑀆𑀫𑀦𑁆𑀢𑀸𑁇

289\. «  𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺?

290\. 𑀅𑀲𑀜𑁆𑀜𑀲𑀢𑁆𑀢𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀘 𑀢𑀢𑁆𑀣 𑀅𑀓𑀼𑀲𑀮𑀸 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀘𑀢𑀼𑀯𑁄𑀓𑀸𑀭𑁂 𑀧𑀜𑁆𑀘𑀯𑁄𑀓𑀸𑀭𑁂 𑀢𑀢𑁆𑀣 𑀅𑀩𑁆𑀬𑀸𑀓𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀓𑀼𑀲𑀮𑀸 𑀘 𑀥𑀫𑁆𑀫𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇
