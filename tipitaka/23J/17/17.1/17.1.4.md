# 𑁧𑁭𑁇𑁧𑁇𑁪 𑀲𑀗𑁆𑀔𑀧𑀸𑀮𑀚𑀸𑀢𑀓

* 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀦𑀺𑀧𑀸𑀢

* 𑀢𑁂𑀲𑀓𑀼𑀡𑀯𑀕𑁆𑀕

* 𑀲𑀗𑁆𑀔𑀧𑀸𑀮𑀚𑀸𑀢𑀓

(143.)

147\. _“𑀅𑀭𑀺𑀬𑀸𑀯𑀓𑀸𑀲𑁄𑀲𑀺 𑀧𑀲𑀦𑁆𑀦𑀦𑁂𑀢𑁆𑀢𑁄,_  
_𑀫𑀜𑁆𑀜𑁂 𑀪𑀯𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀓𑀼𑀮𑀫𑁆𑀳𑀸;_  
_𑀓𑀣𑀁 𑀦𑀼 𑀯𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀳𑀸𑀬 𑀪𑁄𑀕𑁂,_  
_𑀧𑀩𑁆𑀩𑀚𑀺 𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫 𑀖𑀭𑀸 𑀲𑀧𑀜𑁆𑀜”𑁇_  


(144.)

148\. _“𑀲𑀬𑀁 𑀯𑀺𑀫𑀸𑀦𑀁 𑀦𑀭𑀤𑁂𑀯 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀲𑁆𑀲 𑀫𑀳𑁄𑀭𑀕𑀲𑁆𑀲;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀧𑀼𑀜𑁆𑀜𑀸𑀦 𑀫𑀳𑀸𑀯𑀺𑀧𑀸𑀓𑀁,_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀭𑀸𑀚”𑁇_  


(145.)

149\. _“𑀦 𑀓𑀸𑀫𑀓𑀸𑀫𑀸 𑀦 𑀪𑀬𑀸 𑀦 𑀤𑁄𑀲𑀸,_  
_𑀯𑀸𑀘𑀁 𑀫𑀼𑀲𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀪𑀡𑀦𑁆𑀢𑀺;_  
_𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀫𑁂 𑀚𑀸𑀬𑀺𑀳𑀺𑀢𑀺𑀧𑁆𑀧𑀲𑀸𑀤𑁄”𑁇_  


(146.)

150\. _“𑀯𑀸𑀡𑀺𑀚𑁆𑀚 𑀭𑀝𑁆𑀞𑀸𑀥𑀺𑀧 𑀕𑀘𑁆𑀙𑀫𑀸𑀦𑁄,_  
_𑀧𑀣𑁂 𑀅𑀤𑁆𑀤𑀲𑀸𑀲𑀺𑀫𑁆𑀳𑀺 𑀪𑁄𑀚𑀧𑀼𑀢𑁆𑀢𑁂;_  
_𑀧𑀯𑀤𑁆𑀥𑀓𑀸𑀬𑀁 𑀉𑀭𑀕𑀁 𑀫𑀳𑀦𑁆𑀢𑀁,_  
_𑀆𑀤𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀧𑀫𑁄𑀤𑀫𑀸𑀦𑁂𑁇_  


(147.)

151\. _𑀲𑁄𑀳𑀁 𑀲𑀫𑀸𑀕𑀫𑁆𑀫 𑀚𑀦𑀺𑀦𑁆𑀤 𑀢𑁂𑀳𑀺,_  
_𑀧𑀳𑀝𑁆𑀞𑀮𑁄𑀫𑁄 𑀅𑀯𑀘𑀫𑁆𑀳𑀺 𑀪𑀻𑀢𑁄;_  
_‘𑀓𑀼𑀳𑀺𑀁 𑀅𑀬𑀁 𑀦𑀻𑀬𑀢𑀺 𑀪𑀻𑀫𑀓𑀸𑀬𑁄,_  
_𑀦𑀸𑀕𑁂𑀦 𑀓𑀺𑀁 𑀓𑀸𑀳𑀣 𑀪𑁄𑀚𑀧𑀼𑀢𑁆𑀢𑀸’𑁇_  


(148.)

152\. _‘𑀦𑀸𑀕𑁄 𑀅𑀬𑀁 𑀦𑀻𑀬𑀢𑀺 𑀪𑁄𑀚𑀦𑀢𑁆𑀣𑀸,_  
_𑀧𑀯𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀉𑀭𑀕𑁄 𑀫𑀳𑀦𑁆𑀢𑁄;_  
_𑀲𑀸𑀤𑀼𑀜𑁆𑀘 𑀣𑀽𑀮𑀜𑁆𑀘 𑀫𑀼𑀤𑀼𑀜𑁆𑀘 𑀫𑀁𑀲𑀁,_  
_𑀦 𑀢𑁆𑀯𑀁 𑀭𑀲𑀜𑁆𑀜𑀸𑀲𑀺 𑀯𑀺𑀤𑁂𑀳𑀧𑀼𑀢𑁆𑀢𑁇_  


(149.)

153\. _𑀇𑀢𑁄 𑀫𑀬𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀓𑀁 𑀦𑀺𑀓𑁂𑀢𑀁,_  
_𑀆𑀤𑀸𑀬 𑀲𑀢𑁆𑀣𑀸𑀦𑀺 𑀯𑀺𑀓𑁄𑀧𑀬𑀺𑀢𑁆𑀯𑀸;_  
_𑀫𑀁𑀲𑀸𑀦𑀺 𑀪𑁄𑀓𑁆𑀔𑀸𑀫 𑀧𑀫𑁄𑀤𑀫𑀸𑀦𑀸,_  
_𑀫𑀬𑀜𑁆𑀳𑀺 𑀯𑁂 𑀲𑀢𑁆𑀢𑀯𑁄 𑀧𑀦𑁆𑀦𑀕𑀸𑀦𑀁’𑁇_  


(150.)

154\. _‘𑀲𑀘𑁂 𑀅𑀬𑀁 𑀦𑀻𑀬𑀢𑀺 𑀪𑁄𑀚𑀦𑀢𑁆𑀣𑀸,_  
_𑀧𑀯𑀤𑁆𑀥𑀓𑀸𑀬𑁄 𑀉𑀭𑀕𑁄 𑀫𑀳𑀦𑁆𑀢𑁄;_  
_𑀤𑀤𑀸𑀫𑀺 𑀯𑁄 𑀩𑀮𑀺𑀩𑀤𑁆𑀤𑀸𑀦𑀺 𑀲𑁄𑀍𑀅𑀲,_  
_𑀦𑀸𑀕𑀁 𑀇𑀫𑀁 𑀫𑀼𑀜𑁆𑀘𑀣 𑀩𑀦𑁆𑀥𑀦𑀲𑁆𑀫𑀸’𑁇_  


(151.)

155\. _‘𑀅𑀤𑁆𑀥𑀸 𑀳𑀺 𑀦𑁄 𑀪𑀓𑁆𑀔𑁄 𑀅𑀬𑀁 𑀫𑀦𑀸𑀧𑁄,_  
_𑀩𑀳𑀽 𑀘 𑀦𑁄 𑀉𑀭𑀕𑀸 𑀪𑀼𑀢𑁆𑀢𑀧𑀼𑀩𑁆𑀩𑀸;_  
_𑀓𑀭𑁄𑀫 𑀢𑁂 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀅𑀍𑀆𑀭,_  
_𑀫𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀦𑁄 𑀳𑁄𑀳𑀺 𑀯𑀺𑀤𑁂𑀳𑀧𑀼𑀢𑁆𑀢’𑁇_  


(152.)

156\. _𑀢𑀤𑀸𑀲𑁆𑀲𑀼 𑀢𑁂 𑀩𑀦𑁆𑀥𑀦𑀸 𑀫𑁄𑀘𑀬𑀺𑀁𑀲𑀼,_  
_𑀬𑀁 𑀦𑀢𑁆𑀣𑀼𑀢𑁄 𑀧𑀝𑀺𑀫𑁄𑀓𑁆𑀓𑀲𑁆𑀲 𑀧𑀸𑀲𑁂;_  
_𑀫𑀼𑀢𑁆𑀢𑁄 𑀘 𑀲𑁄 𑀩𑀦𑁆𑀥𑀦𑀸 𑀦𑀸𑀕𑀭𑀸𑀚𑀸,_  
_𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀧𑀸𑀘𑀻𑀦𑀫𑀼𑀔𑁄 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁𑁇_  


(153.)

157\. _𑀕𑀦𑁆𑀢𑁆𑀯𑀸𑀦 𑀧𑀸𑀘𑀻𑀦𑀫𑀼𑀔𑁄 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁,_  
_𑀧𑀼𑀡𑁆𑀡𑁂𑀳𑀺 𑀦𑁂𑀢𑁆𑀢𑁂𑀳𑀺 𑀧𑀮𑁄𑀓𑀬𑀻 𑀫𑀁;_  
_𑀢𑀤𑀸𑀲𑁆𑀲𑀳𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑁆𑀯𑀕𑀘𑁆𑀙𑀺𑀁,_  
_𑀤𑀲𑀗𑁆𑀕𑀼𑀮𑀺𑀁 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑁇_  


(154.)

158\. _‘𑀕𑀘𑁆𑀙𑁂𑀯 𑀔𑁄 𑀢𑁆𑀯𑀁 𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑁄,_  
_𑀫𑀸 𑀢𑀁 𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀧𑀼𑀦𑀭𑀕𑁆𑀕𑀳𑁂𑀲𑀼𑀁;_  
_𑀤𑀼𑀓𑁆𑀔𑁄 𑀳𑀺 𑀮𑀼𑀤𑁆𑀤𑁂𑀳𑀺 𑀧𑀼𑀦𑀸 𑀲𑀫𑀸𑀕𑀫𑁄,_  
_𑀅𑀤𑀲𑁆𑀲𑀦𑀁 𑀪𑁄𑀚𑀧𑀼𑀢𑁆𑀢𑀸𑀦 𑀕𑀘𑁆𑀙’𑁇_  


(155.)

159\. _𑀅𑀕𑀫𑀸𑀲𑀺 𑀲𑁄 𑀭𑀳𑀤𑀁 𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀁,_  
_𑀦𑀻𑀮𑁄𑀪𑀸𑀲𑀁 𑀭𑀫𑀡𑀻𑀬𑀁 𑀲𑀼𑀢𑀺𑀢𑁆𑀣𑀁;_  
_𑀲𑀫𑁄𑀢𑀢𑀁 𑀚𑀫𑁆𑀩𑀼𑀳𑀺 𑀯𑁂𑀢𑀲𑀸𑀳𑀺,_  
_𑀧𑀸𑀯𑁂𑀓𑁆𑀔𑀺 𑀦𑀺𑀢𑁆𑀢𑀺𑀡𑁆𑀡𑀪𑀬𑁄 𑀧𑀢𑀻𑀢𑁄𑁇_  


(156.)

160\. _𑀲𑁄 𑀢𑀁 𑀧𑀯𑀺𑀲𑁆𑀲 𑀦 𑀘𑀺𑀭𑀲𑁆𑀲 𑀦𑀸𑀕𑁄,_  
_𑀤𑀺𑀩𑁆𑀩𑁂𑀦 𑀫𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑀼𑀁 𑀚𑀦𑀺𑀦𑁆𑀤;_  
_𑀉𑀧𑀝𑁆𑀞𑀳𑀻 𑀫𑀁 𑀧𑀺𑀢𑀭𑀁𑀯 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀳𑀤𑀬𑀗𑁆𑀕𑀫𑀁 𑀓𑀡𑁆𑀡𑀲𑀼𑀔𑀁 𑀪𑀡𑀦𑁆𑀢𑁄𑁇_  


(157.)

161\. _‘𑀢𑁆𑀯𑀁 𑀫𑁂𑀲𑀺 𑀫𑀸𑀢𑀸 𑀘 𑀧𑀺𑀢𑀸 𑀅𑀍𑀆𑀭,_  
_𑀅𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑁄 𑀧𑀸𑀡𑀤𑀤𑁄 𑀲𑀳𑀸𑀬𑁄;_  
_𑀲𑀓𑀜𑁆𑀘 𑀇𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀮𑀸𑀪𑀓𑁄𑀲𑁆𑀫𑀺,_  
_𑀅𑀍𑀆𑀭 𑀧𑀲𑁆𑀲 𑀫𑁂 𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺;_  
_𑀧𑀳𑀽𑀢𑀪𑀓𑁆𑀔𑀁 𑀩𑀳𑀼𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀁,_  
_𑀫𑀲𑀓𑁆𑀓𑀲𑀸𑀭𑀁 𑀯𑀺𑀬 𑀯𑀸𑀲𑀯𑀲𑁆𑀲𑁇_  


(158.)

162\. _𑀢𑀁 𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑁂𑀳𑀺 𑀉𑀧𑁂𑀢𑀭𑀽𑀧𑀁,_  
_𑀅𑀲𑀓𑁆𑀔𑀭𑀸 𑀘𑁂𑀯 𑀫𑀼𑀤𑀽 𑀲𑀼𑀪𑀸 𑀘;_  
_𑀦𑀻𑀘𑀢𑁆𑀢𑀺𑀡𑀸 𑀅𑀧𑁆𑀧𑀭𑀚𑀸 𑀘 𑀪𑀽𑀫𑀺,_  
_𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸 𑀬𑀢𑁆𑀣 𑀚𑀳𑀦𑁆𑀢𑀺 𑀲𑁄𑀓𑀁𑁇_  


(159.)

163\. _𑀅𑀦𑀸𑀯𑀓𑀼𑀮𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀽𑀧𑀦𑀻𑀮𑀸,_  
_𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀁 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀲𑀼𑀭𑀫𑁆𑀫𑀁;_  
_𑀧𑀓𑁆𑀓𑀸 𑀘 𑀧𑁂𑀲𑀻 𑀘 𑀨𑀮𑀸 𑀲𑀼𑀨𑀼𑀮𑁆𑀮𑀸,_  
_𑀦𑀺𑀘𑁆𑀘𑁄𑀢𑀼𑀓𑀸 𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑀻 𑀨𑀮𑀸𑀦𑀺’𑁇_  


(160.)

164\. _𑀢𑁂𑀲𑀁 𑀯𑀦𑀸𑀦𑀁 𑀦𑀭𑀤𑁂𑀯 𑀫𑀚𑁆𑀛𑁂,_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀪𑀲𑁆𑀲𑀭𑀲𑀦𑁆𑀦𑀺𑀓𑀸𑀲𑀁;_  
_𑀭𑀚𑀢𑀕𑁆𑀕𑀍𑀅𑀁 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀁 𑀉𑀍𑀆𑀭𑀁,_  
_𑀑𑀪𑀸𑀲𑀢𑀻 𑀯𑀺𑀚𑁆𑀚𑀼𑀭𑀺𑀯𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂𑁇_  


(161.)

165\. _𑀫𑀡𑀻𑀫𑀬𑀸 𑀲𑁄𑀡𑁆𑀡𑀫𑀬𑀸 𑀉𑀍𑀆𑀭𑀸,_  
_𑀅𑀦𑁂𑀓𑀘𑀺𑀢𑁆𑀢𑀸 𑀲𑀢𑀢𑀁 𑀲𑀼𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀧𑀭𑀺𑀧𑀽𑀭𑀸 𑀓𑀜𑁆𑀜𑀸𑀳𑀺 𑀅𑀮𑀗𑁆𑀓𑀢𑀸𑀪𑀺,_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀸𑀬𑀽𑀭𑀥𑀭𑀸𑀳𑀺 𑀭𑀸𑀚𑁇_  


(162.)

166\. _𑀲𑁄 𑀲𑀗𑁆𑀔𑀧𑀸𑀮𑁄 𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑁄,_  
_𑀧𑀸𑀲𑀸𑀤𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀅𑀦𑁄𑀫𑀯𑀡𑁆𑀡𑁄;_  
_𑀲𑀳𑀲𑁆𑀲𑀣𑀫𑁆𑀪𑀁 𑀅𑀢𑀼𑀮𑀸𑀦𑀼𑀪𑀸𑀯𑀁,_  
_𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸 𑀫𑀳𑁂𑀲𑀻 𑀅𑀳𑁄𑀲𑀺𑁇_  


(163.)

167\. _𑀏𑀓𑀸 𑀘 𑀦𑀸𑀭𑀻 𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑀸,_  
_𑀆𑀤𑀸𑀬 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀁 𑀫𑀳𑀕𑁆𑀖𑀁;_  
_𑀲𑀼𑀪𑀁 𑀫𑀡𑀺𑀁 𑀚𑀸𑀢𑀺𑀫𑀦𑁆𑀢𑀽𑀧𑀧𑀦𑁆𑀦𑀁,_  
_𑀅𑀘𑁄𑀤𑀺𑀢𑀸 𑀆𑀲𑀦𑀫𑀩𑁆𑀪𑀺𑀳𑀸𑀲𑀺𑁇_  


(164.)

168\. _𑀢𑀢𑁄 𑀫𑀁 𑀉𑀭𑀕𑁄 𑀳𑀢𑁆𑀣𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸,_  
_𑀦𑀺𑀲𑀻𑀤𑀬𑀻 𑀧𑀸𑀫𑀼𑀔𑀆𑀲𑀦𑀲𑁆𑀫𑀺𑀁;_  
_‘𑀇𑀤𑀫𑀸𑀲𑀦𑀁 𑀅𑀢𑁆𑀭 𑀪𑀯𑀁 𑀦𑀺𑀲𑀻𑀤𑀢𑀼,_  
_𑀪𑀯𑀜𑁆𑀳𑀺 𑀫𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀕𑀭𑀽𑀦𑀁’𑁇_  


(165.)

169\. _𑀅𑀜𑁆𑀜𑀸 𑀘 𑀦𑀸𑀭𑀻 𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑀸,_  
_𑀆𑀤𑀸𑀬 𑀯𑀸𑀭𑀺𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸;_  
_𑀧𑀸𑀤𑀸𑀦𑀺 𑀧𑀓𑁆𑀔𑀸𑀮𑀬𑀻 𑀫𑁂 𑀚𑀦𑀺𑀦𑁆𑀤,_  
_𑀪𑀭𑀺𑀬𑀸𑀯 𑀪𑀢𑁆𑀢𑀽 𑀧𑀢𑀺𑀦𑁄 𑀧𑀺𑀬𑀲𑁆𑀲𑁇_  


(166.)

170\. _𑀅𑀧𑀭𑀸 𑀘 𑀦𑀸𑀭𑀻 𑀢𑀭𑀫𑀸𑀦𑀭𑀽𑀧𑀸,_  
_𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸𑀬 𑀧𑀸𑀢𑀺𑀬𑀸;_  
_𑀅𑀦𑁂𑀓𑀲𑀽𑀧𑀁 𑀯𑀺𑀯𑀺𑀥𑀁 𑀯𑀺𑀬𑀜𑁆𑀚𑀦𑀁,_  
_𑀉𑀧𑀦𑀸𑀫𑀬𑀻 𑀪𑀢𑁆𑀢 𑀫𑀦𑀼𑀜𑁆𑀜𑀭𑀽𑀧𑀁𑁇_  


(167.)

171\. _𑀢𑀼𑀭𑀺𑀬𑁂𑀳𑀺 𑀫𑀁 𑀪𑀸𑀭𑀢 𑀪𑀼𑀢𑁆𑀢𑀯𑀦𑁆𑀢𑀁,_  
_𑀉𑀧𑀝𑁆𑀞𑀳𑀼𑀁 𑀪𑀢𑁆𑀢𑀼 𑀫𑀦𑁄 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸;_  
_𑀢𑀢𑀼𑀢𑁆𑀢𑀭𑀺𑀁 𑀫𑀁 𑀦𑀺𑀧𑀢𑀻 𑀫𑀳𑀦𑁆𑀢𑀁,_  
_𑀤𑀺𑀩𑁆𑀩𑁂𑀳𑀺 𑀓𑀸𑀫𑁂𑀳𑀺 𑀅𑀦𑀧𑁆𑀧𑀓𑁂𑀳𑀺𑁇_  


(168.)

172\. _‘𑀪𑀭𑀺𑀬𑀸 𑀫𑀫𑁂𑀢𑀸 𑀢𑀺𑀲𑀢𑀸 𑀅𑀍𑀆𑀭,_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀢𑀫𑀚𑁆𑀛𑀸 𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑀸𑀪𑀸;_  
_𑀅𑀍𑀆𑀭 𑀏𑀢𑀸𑀲𑁆𑀲𑀼 𑀢𑁂 𑀓𑀸𑀫𑀓𑀸𑀭𑀸,_  
_𑀤𑀤𑀸𑀫𑀺 𑀢𑁂 𑀢𑀸 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀲𑁆𑀲𑀼’𑁇_  


(169.)

173\. _𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑀁 𑀤𑀺𑀩𑁆𑀩𑀭𑀲𑀸𑀦𑀼𑀪𑀼𑀢𑁆𑀯𑀸,_  
_𑀢𑀤𑀸𑀲𑁆𑀲𑀼𑀳𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀚𑁆𑀛𑀪𑀸𑀲𑀺𑀁;_  
_‘𑀦𑀸𑀕𑀲𑁆𑀲𑀺𑀤𑀁 𑀓𑀺𑀦𑁆𑀢𑀺 𑀓𑀣𑀜𑁆𑀘 𑀮𑀤𑁆𑀥𑀁,_  
_𑀓𑀣𑀚𑁆𑀛𑀕𑀫𑀸𑀲𑀺 𑀯𑀺𑀫𑀸𑀦𑀲𑁂𑀝𑁆𑀞𑀁𑁇_  


(170.)

174\. _𑀅𑀥𑀺𑀘𑁆𑀘𑀮𑀤𑁆𑀥𑀁 𑀧𑀭𑀺𑀡𑀸𑀫𑀚𑀁 𑀢𑁂,_  
_𑀲𑀬𑀁𑀓𑀢𑀁 𑀉𑀤𑀸𑀳𑀼 𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀺𑀦𑁆𑀦𑀁;_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀦𑀸𑀕𑀭𑀸𑀚𑁂𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀓𑀣𑀚𑁆𑀛𑀕𑀫𑀸𑀲𑀺 𑀯𑀺𑀫𑀸𑀦𑀲𑁂𑀝𑁆𑀞𑀁’𑁇_  


(171.)

175\. _‘𑀦𑀸𑀥𑀺𑀘𑁆𑀘𑀮𑀤𑁆𑀥𑀁 𑀦 𑀧𑀭𑀺𑀡𑀸𑀫𑀚𑀁 𑀫𑁂,_  
_𑀦 𑀲𑀬𑀁𑀓𑀢𑀁 𑀦𑀸𑀧𑀺 𑀤𑁂𑀯𑁂𑀳𑀺 𑀤𑀺𑀦𑁆𑀦𑀁;_  
_𑀲𑀓𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀳𑀺 𑀅𑀧𑀸𑀧𑀓𑁂𑀳𑀺,_  
_𑀧𑀼𑀜𑁆𑀜𑁂𑀳𑀺 𑀫𑁂 𑀮𑀤𑁆𑀥𑀫𑀺𑀤𑀁 𑀯𑀺𑀫𑀸𑀦𑀁’𑁇_  


(172.)

176\. _‘𑀓𑀺𑀁 𑀢𑁂 𑀯𑀢𑀁 𑀓𑀺𑀁 𑀧𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀓𑀺𑀲𑁆𑀲 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄;_  
_𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀫𑁂 𑀦𑀸𑀕𑀭𑀸𑀚𑁂𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀓𑀣𑀁 𑀦𑀼 𑀢𑁂 𑀮𑀤𑁆𑀥𑀫𑀺𑀤𑀁 𑀯𑀺𑀫𑀸𑀦𑀁’𑁇_  


(173.)

177\. _‘𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺𑀁 𑀫𑀕𑀥𑀸𑀦𑀫𑀺𑀲𑁆𑀲𑀭𑁄,_  
_𑀤𑀼𑀬𑁆𑀬𑁄𑀥𑀦𑁄 𑀦𑀸𑀫 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄;_  
_𑀲𑁄 𑀇𑀢𑁆𑀢𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑀁 𑀲𑀁𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀲𑀲𑁆𑀲𑀢𑀁 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀁𑁇_  


(174.)

178\. _𑀅𑀦𑁆𑀦𑀜𑁆𑀘 𑀧𑀸𑀦𑀜𑁆𑀘 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁄,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀤𑀸𑀦𑀁 𑀯𑀺𑀧𑀼𑀮𑀁 𑀅𑀤𑀸𑀲𑀺𑀁;_  
_𑀑𑀧𑀸𑀦𑀪𑀽𑀢𑀁 𑀫𑁂 𑀖𑀭𑀁 𑀢𑀤𑀸𑀲𑀺,_  
_𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀸 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀘𑁇_  


(175.)

179\. _𑀫𑀸𑀮𑀜𑁆𑀘 𑀕𑀦𑁆𑀥𑀜𑁆𑀘 𑀯𑀺𑀮𑁂𑀧𑀦𑀜𑁆𑀘,_  
_𑀧𑀤𑀻𑀧𑀺𑀬𑀁 𑀬𑀸𑀦𑀫𑀼𑀧𑀲𑁆𑀲𑀬𑀜𑁆𑀘;_  
_𑀅𑀘𑁆𑀙𑀸𑀤𑀦𑀁 𑀲𑁂𑀬𑁆𑀬𑀫𑀣𑀦𑁆𑀦𑀧𑀸𑀦𑀁,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘 𑀤𑀸𑀦𑀸𑀦𑀺 𑀅𑀤𑀫𑁆𑀳 𑀢𑀢𑁆𑀣𑁇_  


(176.)

180\. _𑀢𑀁 𑀫𑁂 𑀯𑀢𑀁 𑀢𑀁 𑀧𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀢𑀲𑁆𑀲 𑀲𑀼𑀘𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀅𑀬𑀁 𑀯𑀺𑀧𑀸𑀓𑁄;_  
_𑀢𑁂𑀦𑁂𑀯 𑀫𑁂 𑀮𑀤𑁆𑀥𑀫𑀺𑀤𑀁 𑀯𑀺𑀫𑀸𑀦𑀁,_  
_𑀧𑀳𑀽𑀢𑀪𑀓𑁆𑀔𑀁 𑀩𑀳𑀼𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀁;_  
_𑀦𑀘𑁆𑀘𑁂𑀳𑀺 𑀕𑀻𑀢𑁂𑀳𑀺 𑀘𑀼𑀧𑁂𑀢𑀭𑀽𑀧𑀁,_  
_𑀘𑀺𑀭𑀝𑁆𑀞𑀺𑀢𑀺𑀓𑀁 𑀦 𑀘 𑀲𑀲𑁆𑀲𑀢𑀸𑀬𑀁’𑁇_  


(177.)

181\. _‘𑀅𑀧𑁆𑀧𑀸𑀦𑀼𑀪𑀸𑀯𑀸 𑀢𑀁 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀁,_  
_𑀢𑁂𑀚𑀲𑁆𑀲𑀺𑀦𑀁 𑀳𑀦𑁆𑀢𑀺 𑀅𑀢𑁂𑀚𑀯𑀦𑁆𑀢𑁄;_  
_𑀓𑀺𑀫𑁂𑀯 𑀤𑀸𑀞𑀸𑀯𑀼𑀥 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘,_  
_𑀳𑀢𑁆𑀣𑀢𑁆𑀢𑀫𑀸𑀕𑀘𑁆𑀙𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀸𑀦𑀁𑁇_  


(178.)

182\. _𑀪𑀬𑀁 𑀦𑀼 𑀢𑁂 𑀅𑀦𑁆𑀯𑀕𑀢𑀁 𑀫𑀳𑀦𑁆𑀢𑀁,_  
_𑀢𑁂𑀚𑁄 𑀦𑀼 𑀢𑁂 𑀦𑀸𑀦𑁆𑀯𑀕𑀁 𑀤𑀦𑁆𑀢𑀫𑀽𑀮𑀁;_  
_𑀓𑀺𑀫𑁂𑀯 𑀤𑀸𑀞𑀸𑀯𑀼𑀥 𑀓𑀺𑀁 𑀧𑀝𑀺𑀘𑁆𑀘,_  
_𑀓𑀺𑀮𑁂𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀸𑀦𑀁’𑁇_  


(179.)

183\. _‘𑀦 𑀫𑁂 𑀪𑀬𑀁 𑀅𑀦𑁆𑀯𑀕𑀢𑀁 𑀫𑀳𑀦𑁆𑀢𑀁,_  
_𑀢𑁂𑀚𑁄 𑀦 𑀲𑀓𑁆𑀓𑀸 𑀫𑀫 𑀢𑁂𑀳𑀺 𑀳𑀦𑁆𑀢𑀼𑀁;_  
_𑀲𑀢𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀸𑀦𑀺 𑀲𑀼𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑀦𑀺,_  
_𑀲𑀫𑀼𑀤𑁆𑀤𑀯𑁂𑀮𑀸𑀯 𑀤𑀼𑀭𑀘𑁆𑀘𑀬𑀸𑀦𑀺𑁇_  


(180.)

184\. _𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀁 𑀧𑀜𑁆𑀘𑀤𑀲𑀺𑀁 𑀅𑀍𑀆𑀭,_  
_𑀉𑀧𑁄𑀲𑀣𑀁 𑀦𑀺𑀘𑁆𑀘𑀫𑀼𑀧𑀸𑀯𑀲𑀸𑀫𑀺;_  
_𑀅𑀣𑀸𑀕𑀫𑀼𑀁 𑀲𑁄𑀍𑀅𑀲 𑀪𑁄𑀚𑀧𑀼𑀢𑁆𑀢𑀸,_  
_𑀭𑀚𑁆𑀚𑀼𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦 𑀤𑀍𑀳𑀜𑁆𑀘 𑀧𑀸𑀲𑀁𑁇_  


(181.)

185\. _𑀪𑁂𑀢𑁆𑀯𑀸𑀦 𑀦𑀸𑀲𑀁 𑀅𑀢𑀺𑀓𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀼𑀁,_  
_𑀦𑀬𑀺𑀁𑀲𑀼 𑀫𑀁 𑀲𑀫𑁆𑀧𑀭𑀺𑀕𑀬𑁆𑀳 𑀮𑀼𑀤𑁆𑀤𑀸;_  
_𑀏𑀢𑀸𑀤𑀺𑀲𑀁 𑀤𑀼𑀓𑁆𑀔𑀫𑀳𑀁 𑀢𑀺𑀢𑀺𑀓𑁆𑀔𑀁,_  
_𑀉𑀧𑁄𑀲𑀣𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑁄𑀧𑀬𑀦𑁆𑀢𑁄’𑁇_  


(182.)

186\. _‘𑀏𑀓𑀸𑀬𑀦𑁂 𑀢𑀁 𑀧𑀣𑁂 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼,_  
_𑀩𑀮𑁂𑀦 𑀯𑀡𑁆𑀡𑁂𑀦 𑀘𑀼𑀧𑁂𑀢𑀭𑀽𑀧𑀁;_  
_𑀲𑀺𑀭𑀺𑀬𑀸 𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀪𑀸𑀯𑀺𑀢𑁄𑀲𑀺,_  
_𑀓𑀺𑀁 𑀧𑀢𑁆𑀣𑀬𑀁 𑀦𑀸𑀕 𑀢𑀧𑁄 𑀓𑀭𑁄𑀲𑀺’𑁇_  


(183.)

187\. _‘𑀦 𑀧𑀼𑀢𑁆𑀢𑀳𑁂𑀢𑀽 𑀦 𑀥𑀦𑀲𑁆𑀲 𑀳𑁂𑀢𑀼,_  
_𑀦 𑀆𑀬𑀼𑀦𑁄 𑀘𑀸𑀧𑀺 𑀅𑀍𑀆𑀭 𑀳𑁂𑀢𑀼;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀬𑁄𑀦𑀺𑀁 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑀸𑀦𑁄,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀭𑀓𑁆𑀓𑀫𑁆𑀫 𑀢𑀧𑁄 𑀓𑀭𑁄𑀫𑀺’𑁇_  


(184.)

188\. _‘𑀢𑁆𑀯𑀁 𑀮𑁄𑀳𑀺𑀢𑀓𑁆𑀔𑁄 𑀯𑀺𑀳𑀢𑀦𑁆𑀢𑀭𑀁𑀲𑁄,_  
_𑀅𑀮𑀗𑁆𑀓𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀢𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼;_  
_𑀲𑀼𑀭𑁄𑀲𑀺𑀢𑁄 𑀮𑁄𑀳𑀺𑀢𑀘𑀦𑁆𑀤𑀦𑁂𑀦,_  
_𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑀭𑀸𑀚𑀸𑀯 𑀤𑀺𑀲𑀸 𑀧𑀪𑀸𑀲𑀲𑀺𑁇_  


(185.)

189\. _𑀤𑁂𑀯𑀺𑀤𑁆𑀥𑀺𑀧𑀢𑁆𑀢𑁄𑀲𑀺 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄,_  
_𑀲𑀩𑁆𑀩𑁂𑀳𑀺 𑀓𑀸𑀫𑁂𑀳𑀺 𑀲𑀫𑀗𑁆𑀕𑀺𑀪𑀽𑀢𑁄;_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀦𑀸𑀕𑀭𑀸𑀚𑁂𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀲𑁂𑀬𑁆𑀬𑁄 𑀇𑀢𑁄 𑀓𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁄’𑁇_  


(186.)

190\. _‘𑀅𑀍𑀆𑀭 𑀦𑀸𑀜𑁆𑀜𑀢𑁆𑀭 𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑀸,_  
_𑀲𑀼𑀤𑁆𑀥𑀻 𑀯 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀲𑀁𑀬𑀫𑁄 𑀯𑀸;_  
_𑀅𑀳𑀜𑁆𑀘 𑀮𑀤𑁆𑀥𑀸𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀬𑁄𑀦𑀺𑀁,_  
_𑀓𑀸𑀳𑀸𑀫𑀺 𑀚𑀸𑀢𑀺𑀫𑀭𑀡𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀁’𑁇_  


(187.)

191\. _‘𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑁄 𑀫𑁂 𑀯𑀲𑀢𑁄 𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑁂,_  
_𑀅𑀦𑁆𑀦𑁂𑀦 𑀧𑀸𑀦𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁄𑀲𑁆𑀫𑀺;_  
_𑀆𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀮𑁂𑀫𑀺 𑀦𑀸𑀕,_  
_𑀘𑀺𑀭𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁄𑀲𑁆𑀫𑀺 𑀅𑀳𑀁 𑀚𑀦𑀺𑀦𑁆𑀤’𑁇_  


(188.)

192\. _‘𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀤𑀸𑀭𑀸 𑀅𑀦𑀼𑀚𑀻𑀯𑀺𑀦𑁄 𑀘,_  
_𑀦𑀺𑀘𑁆𑀘𑀸𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀸 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀣𑁂𑀢𑀁;_  
_𑀓𑀘𑁆𑀘𑀺𑀦𑁆𑀦𑀼 𑀢𑀁 𑀦𑀸𑀪𑀺𑀲𑀧𑀺𑀢𑁆𑀣 𑀓𑁄𑀘𑀺,_  
_𑀧𑀺𑀬𑀜𑁆𑀳𑀺 𑀫𑁂 𑀤𑀲𑁆𑀲𑀦𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀅𑀍𑀆𑀭’𑁇_  


(189.)

193\. _‘𑀬𑀣𑀸𑀧𑀺 𑀫𑀸𑀢𑀽 𑀘 𑀧𑀺𑀢𑀽 𑀅𑀕𑀸𑀭𑁂,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀺𑀬𑁄 𑀧𑀝𑀺𑀯𑀺𑀳𑀺𑀢𑁄 𑀯𑀲𑁂𑀬𑁆𑀬;_  
_𑀢𑀢𑁄𑀧𑀺 𑀫𑀬𑁆𑀳𑀁 𑀇𑀥𑀫𑁂𑀯 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀘𑀺𑀢𑁆𑀢𑀜𑁆𑀳𑀺 𑀢𑁂 𑀦𑀸𑀕 𑀫𑀬𑀻 𑀧𑀲𑀦𑁆𑀦𑀁’𑁇_  


(190.)

194\. _‘𑀫𑀡𑀻 𑀫𑀫𑀁 𑀯𑀺𑀚𑁆𑀚𑀢𑀺 𑀮𑁄𑀳𑀺𑀢𑀗𑁆𑀓𑁄,_  
_𑀥𑀦𑀸𑀳𑀭𑁄 𑀫𑀡𑀺𑀭𑀢𑀦𑀁 𑀉𑀍𑀆𑀭𑀁;_  
_𑀆𑀤𑀸𑀬 𑀢𑁆𑀯𑀁 𑀕𑀘𑁆𑀙 𑀲𑀓𑀁 𑀦𑀺𑀓𑁂𑀢𑀁,_  
_𑀮𑀤𑁆𑀥𑀸 𑀥𑀦𑀁 𑀢𑀁 𑀫𑀡𑀺𑀫𑁄𑀲𑁆𑀲𑀚𑀲𑁆𑀲𑀼’𑁇_  


(191.)

195\. _𑀤𑀺𑀝𑁆𑀞𑀸 𑀫𑀬𑀸 𑀫𑀸𑀦𑀼𑀲𑀓𑀸𑀧𑀺 𑀓𑀸𑀫𑀸,_  
_𑀅𑀲𑀲𑁆𑀲𑀢𑀸 𑀯𑀺𑀧𑀭𑀺𑀡𑀸𑀫𑀥𑀫𑁆𑀫𑀸;_  
_𑀆𑀤𑀻𑀦𑀯𑀁 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀲𑀼 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀳𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀭𑀸𑀚𑁇_  


(192.)

196\. _𑀤𑀼𑀫𑀧𑁆𑀨𑀮𑀸𑀦𑀻𑀯 𑀧𑀢𑀦𑁆𑀢𑀺 𑀫𑀸𑀡𑀯𑀸,_  
_𑀤𑀳𑀭𑀸 𑀘 𑀯𑀼𑀤𑁆𑀥𑀸 𑀘 𑀲𑀭𑀻𑀭𑀪𑁂𑀤𑀸;_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀺 𑀭𑀸𑀚,_  
_𑀅𑀧𑀡𑁆𑀡𑀓𑀁 𑀲𑀸𑀫𑀜𑁆𑀜𑀫𑁂𑀯 𑀲𑁂𑀬𑁆𑀬𑁄”𑁇_  


(193.)

197\. _“𑀅𑀤𑁆𑀥𑀸 𑀳𑀯𑁂 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀲𑀧𑀜𑁆𑀜𑀸,_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀬𑁂 𑀩𑀳𑀼𑀞𑀸𑀦𑀘𑀺𑀦𑁆𑀢𑀺𑀦𑁄;_  
_𑀦𑀸𑀕𑀜𑁆𑀘 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀢𑀯𑀜𑁆𑀘𑀍𑀆𑀭,_  
_𑀓𑀸𑀳𑀸𑀫𑀺 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀅𑀦𑀧𑁆𑀧𑀓𑀸𑀦𑀺”𑁇_  


(194.)

198\. _“𑀅𑀤𑁆𑀥𑀸 𑀳𑀯𑁂 𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑀸 𑀲𑀧𑀜𑁆𑀜𑀸,_  
_𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸 𑀬𑁂 𑀩𑀳𑀼𑀞𑀸𑀦𑀘𑀺𑀦𑁆𑀢𑀺𑀦𑁄;_  
_𑀦𑀸𑀕𑀜𑁆𑀘 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀫𑀫𑀜𑁆𑀘 𑀭𑀸𑀚,_  
_𑀓𑀭𑁄𑀳𑀺 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀅𑀦𑀧𑁆𑀧𑀓𑀸𑀦𑀻”𑀢𑀺𑁇_  


---

199\. 𑀲𑀗𑁆𑀔𑀧𑀸𑀮𑀚𑀸𑀢𑀓𑀁 𑀘𑀢𑀼𑀢𑁆𑀣𑀁𑁇
