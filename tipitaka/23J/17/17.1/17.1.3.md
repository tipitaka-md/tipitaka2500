# 𑁧𑁭𑁇𑁧𑁇𑁩 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀸𑀚𑀸𑀢𑀓

### 𑀘𑀢𑁆𑀢𑀸𑀮𑀻𑀲𑀦𑀺𑀧𑀸𑀢

### 𑀢𑁂𑀲𑀓𑀼𑀡𑀯𑀕𑁆𑀕

### 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀸𑀚𑀸𑀢𑀓

(95.)

98\. _𑀅𑀣 𑀩𑁆𑀭𑀯𑀺 𑀩𑁆𑀭𑀳𑀸 𑀇𑀦𑁆𑀤𑁄,_  
_𑀯𑀢𑁆𑀭𑀪𑀽 𑀚𑀬𑀢𑀁 𑀧𑀺𑀢𑀸;_  
_𑀤𑁂𑀯𑀓𑀜𑁆𑀜𑀁 𑀧𑀭𑀸𑀪𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀥𑀫𑁆𑀫𑀸𑀬𑀁 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀁𑁇_  


(96.)

99\. _“𑀫𑀺𑀲𑁆𑀲𑁂 𑀤𑁂𑀯𑀸 𑀢𑀁 𑀬𑀸𑀘𑀦𑁆𑀢𑀺,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀲𑁃𑀦𑁆𑀤𑀓𑀸;_  
_𑀇𑀲𑀺𑀧𑁆𑀧𑀮𑁄𑀪𑀦𑁂 𑀕𑀘𑁆𑀙,_  
_𑀇𑀲𑀺𑀲𑀺𑀗𑁆𑀕𑀁 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑁂𑁇_  


(97.)

100\. _𑀧𑀼𑀭𑀸𑀬𑀁 𑀅𑀫𑁆𑀳𑁂 𑀅𑀘𑁆𑀘𑁂𑀢𑀺,_  
_𑀯𑀢𑁆𑀢𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀯𑀸;_  
_𑀦𑀺𑀩𑁆𑀩𑀸𑀦𑀸𑀪𑀺𑀭𑀢𑁄 𑀯𑀼𑀤𑁆𑀥𑁄,_  
_𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀸𑀦𑀺 𑀆𑀯𑀭”𑁇_  


(98.)

101\. _“𑀤𑁂𑀯𑀭𑀸𑀚 𑀓𑀺𑀫𑁂𑀯 𑀢𑁆𑀯𑀁,_  
_𑀫𑀫𑁂𑀯 𑀢𑀼𑀯𑀁 𑀲𑀺𑀓𑁆𑀔𑀲𑀺;_  
_𑀇𑀲𑀺𑀧𑁆𑀧𑀮𑁄𑀪𑀦𑁂 𑀕𑀘𑁆𑀙,_  
_𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀅𑀘𑁆𑀙𑀭𑀸𑁇_  


(99.)

102\. _𑀫𑀸𑀤𑀺𑀲𑀺𑀬𑁄 𑀧𑀯𑀭𑀸 𑀘𑁂𑀯,_  
_𑀅𑀲𑁄𑀓𑁂 𑀦𑀦𑁆𑀤𑀦𑁂 𑀯𑀦𑁂;_  
_𑀢𑀸𑀲𑀫𑁆𑀧𑀺 𑀳𑁄𑀢𑀼 𑀧𑀭𑀺𑀬𑀸𑀬𑁄,_  
_𑀢𑀸𑀧𑀺 𑀬𑀦𑁆𑀢𑀼 𑀧𑀮𑁄𑀪𑀦𑀸”𑁇_  


(100.)

103\. _“𑀅𑀤𑁆𑀥𑀸 𑀳𑀺 𑀲𑀘𑁆𑀘𑀁 𑀪𑀡𑀲𑀺,_  
_𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀸𑀧𑀺 𑀅𑀘𑁆𑀙𑀭𑀸;_  
_𑀢𑀸𑀤𑀺𑀲𑀺𑀬𑁄 𑀧𑀯𑀭𑀸 𑀘𑁂𑀯,_  
_𑀅𑀲𑁄𑀓𑁂 𑀦𑀦𑁆𑀤𑀦𑁂 𑀯𑀦𑁂𑁇_  


(101.)

104\. _𑀦 𑀢𑀸 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀁 𑀧𑀼𑀫𑀁 𑀕𑀢𑀸;_  
_𑀬𑀸𑀤𑀺𑀲𑀁 𑀢𑁆𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀦𑀸𑀭𑀺 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑁂𑁇_  


(102.)

105\. _𑀢𑁆𑀯𑀫𑁂𑀯 𑀕𑀘𑁆𑀙 𑀓𑀮𑁆𑀬𑀸𑀡𑀺,_  
_𑀇𑀢𑁆𑀣𑀻𑀦𑀁 𑀧𑀯𑀭𑀸 𑀘𑀲𑀺;_  
_𑀢𑀯𑁂𑀯 𑀯𑀡𑁆𑀡𑀭𑀽𑀧𑁂𑀦,_  
_𑀲𑀯𑀲𑀫𑀸𑀦𑀬𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(103.)

106\. _“𑀦 𑀯𑀸𑀳𑀁 𑀦 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑁂𑀦 𑀧𑁂𑀲𑀺𑀢𑀸;_  
_𑀯𑀺𑀪𑁂𑀫𑀺 𑀘𑁂𑀢𑀁 𑀆𑀲𑀸𑀤𑀼𑀁,_  
_𑀉𑀕𑁆𑀕𑀢𑁂𑀚𑁄 𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄𑁇_  


(104.)

107\. _𑀅𑀦𑁂𑀓𑁂 𑀦𑀺𑀭𑀬𑀁 𑀧𑀢𑁆𑀢𑀸,_  
_𑀇𑀲𑀺𑀫𑀸𑀲𑀸𑀤𑀺𑀬𑀸 𑀚𑀦𑀸;_  
_𑀆𑀧𑀦𑁆𑀦𑀸 𑀫𑁄𑀳𑀲𑀁𑀲𑀸𑀭𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀮𑁄𑀫𑀸𑀦𑀺 𑀳𑀁𑀲𑀬𑁂”𑁇_  


(105.)

108\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺,_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀺𑀦𑀻;_  
_𑀫𑀺𑀲𑁆𑀲𑀸 𑀫𑀺𑀲𑁆𑀲𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀻,_  
_𑀇𑀲𑀺𑀲𑀺𑀗𑁆𑀕𑀁 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀸𑁇_  


(106.)

109\. _𑀲𑀸 𑀘 𑀢𑀁 𑀯𑀦𑀫𑁄𑀕𑀬𑁆𑀳,_  
_𑀇𑀲𑀺𑀲𑀺𑀗𑁆𑀕𑁂𑀦 𑀭𑀓𑁆𑀔𑀺𑀢𑀁;_  
_𑀩𑀺𑀫𑁆𑀩𑀚𑀸𑀮𑀓𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀅𑀤𑁆𑀥𑀬𑁄𑀚𑀦𑀁𑁇_  


(107.)

110\. _𑀧𑀸𑀢𑁄𑀯 𑀧𑀸𑀢𑀭𑀸𑀲𑀫𑁆𑀳𑀺,_  
_𑀉𑀤𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀢𑀺;_  
_𑀅𑀕𑁆𑀕𑀺𑀝𑁆𑀞𑀁 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀦𑁆𑀢𑀁,_  
_𑀇𑀲𑀺𑀲𑀺𑀗𑁆𑀕𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(108.)

111\. _“𑀓𑀸 𑀦𑀼 𑀯𑀺𑀚𑁆𑀚𑀼𑀭𑀺𑀯𑀸𑀪𑀸𑀲𑀺,_  
_𑀑𑀲𑀥𑀻 𑀯𑀺𑀬 𑀢𑀸𑀭𑀓𑀸;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀸𑀪𑀭𑀡𑀸,_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑀸𑁇_  


(109.)

112\. _𑀆𑀤𑀺𑀘𑁆𑀘𑀯𑀡𑁆𑀡𑀲𑀗𑁆𑀓𑀸𑀲𑀸,_  
_𑀳𑁂𑀫𑀘𑀦𑁆𑀤𑀦𑀕𑀦𑁆𑀥𑀺𑀦𑀻;_  
_𑀲𑀜𑁆𑀜𑀢𑀽𑀭𑀽 𑀫𑀳𑀸𑀫𑀸𑀬𑀸,_  
_𑀓𑀼𑀫𑀸𑀭𑀻 𑀘𑀸𑀭𑀼𑀤𑀲𑁆𑀲𑀦𑀸𑁇_  


(110.)

113\. _𑀯𑀺𑀮𑀕𑁆𑀕𑀸 𑀫𑀼𑀤𑀼𑀓𑀸 𑀲𑀼𑀤𑁆𑀥𑀸,_  
_𑀧𑀸𑀤𑀸 𑀢𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀕𑀫𑀦𑀸 𑀓𑀸𑀫𑀦𑀻𑀬𑀸 𑀢𑁂,_  
_𑀳𑀭𑀦𑁆𑀢𑀺𑀬𑁂𑀯 𑀫𑁂 𑀫𑀦𑁄𑁇_  


(111.)

114\. _𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀸𑀯 𑀢𑁂 𑀊𑀭𑀽,_  
_𑀦𑀸𑀕𑀦𑀸𑀲𑀲𑀫𑀽𑀧𑀫𑀸;_  
_𑀯𑀺𑀫𑀝𑁆𑀞𑀸 𑀢𑀼𑀬𑁆𑀳𑀁 𑀲𑀼𑀲𑁆𑀲𑁄𑀡𑀻,_  
_𑀅𑀓𑁆𑀔𑀲𑁆𑀲 𑀨𑀮𑀓𑀁 𑀬𑀣𑀸𑁇_  


(112.)

115\. _𑀉𑀧𑁆𑀧𑀮𑀲𑁆𑀲𑁂𑀯 𑀓𑀺𑀜𑁆𑀚𑀓𑁆𑀔𑀸,_  
_𑀦𑀸𑀪𑀺 𑀢𑁂 𑀲𑀸𑀥𑀼 𑀲𑀡𑁆𑀞𑀺𑀢𑀸;_  
_𑀧𑀽𑀭𑀸 𑀓𑀡𑁆𑀳𑀜𑁆𑀚𑀦𑀲𑁆𑀲𑁂𑀯,_  
_𑀤𑀽𑀭𑀢𑁄 𑀧𑀝𑀺𑀤𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(113.)

116\. _𑀤𑀼𑀯𑀺𑀥𑀸 𑀚𑀸𑀢𑀸 𑀉𑀭𑀚𑀸,_  
_𑀅𑀯𑀡𑁆𑀝𑀸 𑀲𑀸𑀥𑀼𑀧𑀘𑁆𑀘𑀼𑀤𑀸;_  
_𑀧𑀬𑁄𑀥𑀭𑀸 𑀅𑀧𑀢𑀺𑀢𑀸,_  
_𑀅𑀟𑁆𑀠𑀮𑀸𑀩𑀼𑀲𑀫𑀸 𑀣𑀦𑀸𑁇_  


(114.)

117\. _𑀤𑀻𑀖𑀸 𑀓𑀫𑁆𑀩𑀼𑀢𑀮𑀸𑀪𑀸𑀲𑀸,_  
_𑀕𑀻𑀯𑀸 𑀏𑀡𑁂𑀬𑁆𑀬𑀓𑀸 𑀬𑀣𑀸;_  
_𑀧𑀡𑁆𑀟𑀭𑀸𑀯𑀭𑀡𑀸 𑀯𑀕𑁆𑀕𑀼,_  
_𑀘𑀢𑀼𑀢𑁆𑀣𑀫𑀦𑀲𑀦𑁆𑀦𑀺𑀪𑀸𑁇_  


(115.)

118\. _𑀉𑀤𑁆𑀥𑀕𑁆𑀕𑀸 𑀘 𑀅𑀥𑀕𑁆𑀕𑀸 𑀘,_  
_𑀤𑀼𑀫𑀕𑁆𑀕𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀺𑀢𑀸;_  
_𑀤𑀼𑀯𑀺𑀚𑀸 𑀦𑁂𑀮𑀲𑀫𑁆𑀪𑀽𑀢𑀸,_  
_𑀤𑀦𑁆𑀢𑀸 𑀢𑀯 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀸𑁇_  


(116.)

119\. _𑀅𑀧𑀡𑁆𑀟𑀭𑀸 𑀮𑁄𑀳𑀺𑀢𑀦𑁆𑀢𑀸,_  
_𑀚𑀺𑀜𑁆𑀚𑀽𑀓𑀨𑀮𑀲𑀦𑁆𑀦𑀺𑀪𑀸;_  
_𑀆𑀬𑀢𑀸 𑀘 𑀯𑀺𑀲𑀸𑀮𑀸 𑀘,_  
_𑀦𑁂𑀢𑁆𑀢𑀸 𑀢𑀯 𑀲𑀼𑀤𑀲𑁆𑀲𑀦𑀸𑁇_  


(117.)

120\. _𑀦𑀸𑀢𑀺𑀤𑀻𑀖𑀸 𑀲𑀼𑀲𑀫𑁆𑀫𑀝𑁆𑀞𑀸,_  
_𑀓𑀦𑀓𑀩𑁆𑀬𑀸 𑀲𑀫𑁄𑀘𑀺𑀢𑀸;_  
_𑀉𑀢𑁆𑀢𑀫𑀗𑁆𑀕𑀭𑀼𑀳𑀸 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀓𑁂𑀲𑀸 𑀘𑀦𑁆𑀤𑀦𑀕𑀦𑁆𑀥𑀺𑀓𑀸𑁇_  


(118.)

121\. _𑀬𑀸𑀯𑀢𑀸 𑀓𑀲𑀺𑀕𑁄𑀭𑀓𑁆𑀔𑀸,_  
_𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀘 𑀬𑀸 𑀕𑀢𑀺;_  
_𑀇𑀲𑀻𑀦𑀜𑁆𑀘 𑀧𑀭𑀓𑁆𑀓𑀦𑁆𑀢𑀁,_  
_𑀲𑀜𑁆𑀜𑀢𑀸𑀦𑀁 𑀢𑀧𑀲𑁆𑀲𑀺𑀦𑀁𑁇_  


(119.)

122\. _𑀦 𑀢𑁂 𑀲𑀫𑀲𑀫𑀁 𑀧𑀲𑁆𑀲𑁂,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀧𑀣𑀯𑀺𑀫𑀡𑁆𑀟𑀮𑁂;_  
_𑀓𑁄 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(120.)

123\. _“𑀦 𑀧𑀜𑁆𑀳𑀓𑀸𑀮𑁄 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀓𑀲𑁆𑀲𑀧𑁂𑀯𑀁 𑀕𑀢𑁂 𑀲𑀢𑀺;_  
_𑀏𑀳𑀺 𑀲𑀫𑁆𑀫 𑀭𑀫𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀉𑀪𑁄 𑀅𑀲𑁆𑀫𑀸𑀓𑀫𑀲𑁆𑀲𑀫𑁂;_  
_𑀏𑀳𑀺 𑀢𑀁 𑀉𑀧𑀕𑀽𑀳𑀺𑀲𑁆𑀲𑀁,_  
_𑀭𑀢𑀻𑀦𑀁 𑀓𑀼𑀲𑀮𑁄 𑀪𑀯”𑁇_  


(121.)

124\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺,_  
_𑀅𑀘𑁆𑀙𑀭𑀸 𑀓𑀸𑀫𑀯𑀡𑁆𑀡𑀺𑀦𑀻;_  
_𑀫𑀺𑀲𑁆𑀲𑀸 𑀫𑀺𑀲𑁆𑀲𑀺𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀻,_  
_𑀇𑀲𑀺𑀲𑀺𑀗𑁆𑀕𑀁 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀸𑁇_  


(122.)

125\. _𑀲𑁄 𑀘 𑀯𑁂𑀕𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀫𑁆𑀫,_  
_𑀙𑁂𑀢𑁆𑀯𑀸 𑀤𑀦𑁆𑀥𑀧𑀭𑀓𑁆𑀓𑀫𑀁;_  
_𑀢𑀫𑀼𑀢𑁆𑀢𑀫𑀸𑀲𑀼 𑀯𑁂𑀡𑀻𑀲𑀼,_  
_𑀅𑀚𑁆𑀛𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀧𑀭𑀸𑀫𑀲𑀺𑁇_  


(123.)

126\. _𑀢𑀫𑀼𑀤𑀸𑀯𑀢𑁆𑀢 𑀓𑀮𑁆𑀬𑀸𑀡𑀻,_  
_𑀧𑀮𑀺𑀲𑁆𑀲𑀚𑀺 𑀲𑀼𑀲𑁄𑀪𑀦𑀸;_  
_𑀘𑀯𑀺𑀢𑀫𑁆𑀳𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀸,_  
_𑀬𑀣𑀸 𑀢𑀁 𑀅𑀣 𑀢𑁄𑀲𑀺𑀢𑀸𑁇_  


(124.)

127\. _𑀫𑀦𑀲𑀸 𑀅𑀕𑀫𑀸 𑀇𑀦𑁆𑀤𑀁,_  
_𑀯𑀲𑀦𑁆𑀢𑀁 𑀦𑀦𑁆𑀤𑀦𑁂 𑀯𑀦𑁂;_  
_𑀢𑀲𑁆𑀲𑀸 𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑀫𑀜𑁆𑀜𑀸𑀬,_  
_𑀫𑀖𑀯𑀸 𑀤𑁂𑀯𑀓𑀼𑀜𑁆𑀚𑀭𑁄𑁇_  


(125.)

128\. _𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀧𑀳𑀺𑀡𑀻 𑀔𑀺𑀧𑁆𑀧𑀁,_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑀁 𑀲𑁄𑀧𑀯𑀸𑀳𑀦𑀁;_  
_𑀲𑁅𑀢𑁆𑀢𑀭𑀘𑁆𑀙𑀤𑀧𑀜𑁆𑀜𑀸𑀲𑀁,_  
_𑀲𑀳𑀲𑁆𑀲𑀧𑀝𑀺𑀬𑀢𑁆𑀣𑀢𑀁𑁇_  


(126.)

129\. _𑀢𑀫𑁂𑀦𑀁 𑀢𑀢𑁆𑀣 𑀥𑀸𑀭𑁂𑀲𑀺,_  
_𑀉𑀭𑁂 𑀓𑀢𑁆𑀯𑀸𑀦 𑀲𑁄𑀪𑀦𑀸;_  
_𑀬𑀣𑀸 𑀏𑀓𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁𑀯,_  
_𑀢𑀻𑀡𑀺 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀥𑀸𑀭𑀬𑀺𑁇_  


(127.)

130\. _𑀯𑀺𑀫𑀤𑁄 𑀢𑀻𑀳𑀺 𑀯𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀧𑀩𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸𑀦 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄;_  
_𑀅𑀤𑁆𑀤𑀲𑀸𑀲𑀺 𑀳𑀭𑀺𑀢𑀭𑀼𑀓𑁆𑀔𑁂,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀅𑀕𑁆𑀕𑀺𑀬𑀸𑀬𑀦𑀁𑁇_  


(128.)

131\. _𑀦𑀯𑀧𑀢𑁆𑀢𑀯𑀦𑀁 𑀨𑀼𑀮𑁆𑀮𑀁,_  
_𑀓𑁄𑀓𑀺𑀮𑀕𑁆𑀕𑀡𑀖𑁄𑀲𑀺𑀢𑀁;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀯𑀺𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸,_  
_𑀭𑀼𑀤𑀁 𑀅𑀲𑁆𑀲𑀽𑀦𑀺 𑀯𑀢𑁆𑀢𑀬𑀺𑁇_  


(129.)

132\. _“𑀦 𑀚𑀼𑀳𑁂 𑀦 𑀚𑀧𑁂 𑀫𑀦𑁆𑀢𑁂,_  
_𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑁆𑀢𑀁 𑀧𑀳𑀸𑀧𑀺𑀢𑀁;_  
_𑀓𑁄 𑀦𑀼 𑀫𑁂 𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀮𑁄𑀪𑀬𑀺𑁇_  


(130.)

133\. _𑀅𑀭𑀜𑁆𑀜𑁂 𑀫𑁂 𑀯𑀺𑀳𑀭𑀢𑁄,_  
_𑀬𑁄 𑀫𑁂 𑀢𑁂𑀚𑀸 𑀳 𑀲𑀫𑁆𑀪𑀼𑀢𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑁆𑀦𑀧𑀭𑀺𑀧𑀽𑀭𑀁,_  
_𑀦𑀸𑀯𑀁𑀯 𑀕𑀡𑁆𑀳𑀺 𑀅𑀡𑁆𑀡𑀯𑁂”𑁇_  


(131.)

134\. _“𑀅𑀳𑀁 𑀢𑁂 𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑁂𑀦 𑀧𑁂𑀲𑀺𑀢𑀸;_  
_𑀅𑀯𑀥𑀺𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀧𑀫𑀸𑀤𑁄 𑀢𑁆𑀯𑀁 𑀦 𑀩𑀼𑀚𑁆𑀛𑀲𑀺”𑁇_  


(132.)

135\. _“𑀇𑀫𑀸𑀦𑀺 𑀓𑀺𑀭 𑀫𑀁 𑀢𑀸𑀢𑁄,_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀺;_  
_𑀓𑀫𑀮𑀸𑀲𑀤𑀺𑀲𑀺𑀢𑁆𑀣𑀺𑀬𑁄,_  
_𑀢𑀸𑀬𑁄 𑀩𑀼𑀚𑁆𑀛𑁂𑀲𑀺 𑀫𑀸𑀡𑀯𑁇_  


(133.)

136\. _𑀉𑀭𑁂𑀕𑀡𑁆𑀟𑀸𑀬𑁄 𑀩𑀼𑀚𑁆𑀛𑁂𑀲𑀺,_  
_𑀢𑀸𑀬𑁄 𑀩𑀼𑀚𑁆𑀛𑁂𑀲𑀺 𑀫𑀸𑀡𑀯;_  
_𑀇𑀘𑁆𑀘𑀸𑀦𑀼𑀲𑀸𑀲𑀺 𑀫𑀁 𑀢𑀸𑀢𑁄,_  
_𑀬𑀣𑀸 𑀫𑀁 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑁄𑁇_  


(134.)

137\. _𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀯𑀘𑀦𑀁 𑀦𑀸𑀓𑀁,_  
_𑀧𑀺𑀢𑀼 𑀯𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀲𑀸𑀲𑀦𑀁;_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀦𑀺𑀫𑁆𑀫𑀦𑀼𑀲𑁆𑀲𑀫𑁆𑀳𑀺,_  
_𑀲𑁆𑀯𑀚𑁆𑀚 𑀛𑀸𑀬𑀸𑀫𑀺 𑀏𑀓𑀓𑁄𑁇_  


(135.)

138\. _𑀲𑁄𑀳𑀁 𑀢𑀣𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀥𑀺𑀭𑀢𑁆𑀣𑀼 𑀚𑀻𑀯𑀺𑀢𑁂𑀦 𑀫𑁂;_  
_𑀧𑀼𑀦 𑀯𑀸 𑀢𑀸𑀤𑀺𑀲𑁄 𑀳𑁂𑀲𑁆𑀲𑀁,_  
_𑀫𑀭𑀡𑀁 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(136.)

139\. _𑀢𑀲𑁆𑀲 𑀢𑁂𑀚𑀁 𑀯𑀻𑀭𑀺𑀬𑀜𑁆𑀘,_  
_𑀥𑀺𑀢𑀺𑀁 𑀜𑀢𑁆𑀯𑀸 𑀅𑀯𑀝𑁆𑀞𑀺𑀢𑀁;_  
_𑀲𑀺𑀭𑀲𑀸 𑀅𑀕𑁆𑀕𑀳𑀻 𑀧𑀸𑀤𑁂,_  
_𑀇𑀲𑀺𑀲𑀺𑀗𑁆𑀕𑀁 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀸𑁇_  


(137.)

140\. _“𑀫𑀸 𑀫𑁂 𑀓𑀼𑀚𑁆𑀛 𑀫𑀳𑀸𑀯𑀻𑀭,_  
_𑀫𑀸 𑀫𑁂 𑀓𑀼𑀚𑁆𑀛 𑀫𑀳𑀸𑀇𑀲𑁂;_  
_𑀫𑀳𑀸 𑀅𑀢𑁆𑀣𑁄 𑀫𑀬𑀸 𑀘𑀺𑀡𑁆𑀡𑁄,_  
_𑀢𑀺𑀤𑀲𑀸𑀦𑀁 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀁;_  
_𑀢𑀬𑀸 𑀲𑀗𑁆𑀓𑀫𑁆𑀧𑀺𑀢𑀁 𑀆𑀲𑀺,_  
_𑀲𑀩𑁆𑀩𑀁 𑀤𑁂𑀯𑀧𑀼𑀭𑀁 𑀢𑀤𑀸”𑁇_  


(138.)

141\. _“𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀘 𑀬𑁂 𑀤𑁂𑀯𑀸,_  
_𑀢𑀺𑀤𑀲𑀸𑀦𑀜𑁆𑀘 𑀯𑀸𑀲𑀯𑁄;_  
_𑀢𑁆𑀯𑀜𑁆𑀘 𑀪𑀤𑁆𑀤𑁂 𑀲𑀼𑀔𑀻 𑀳𑁄𑀳𑀺,_  
_𑀕𑀘𑁆𑀙 𑀓𑀜𑁆𑀜𑁂 𑀬𑀣𑀸𑀲𑀼𑀔𑀁”𑁇_  


(139.)

142\. _𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀢𑀫𑁆𑀳𑀸 𑀞𑀸𑀦𑀸 𑀅𑀧𑀓𑁆𑀓𑀫𑀺𑁇_  


(140.)

143\. _𑀬𑁄 𑀘 𑀢𑀲𑁆𑀲𑀸𑀲𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁄,_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑁄 𑀲𑁄𑀧𑀯𑀸𑀳𑀦𑁄;_  
_𑀲𑁅𑀢𑁆𑀢𑀭𑀘𑁆𑀙𑀤𑀧𑀜𑁆𑀜𑀸𑀲𑁄,_  
_𑀲𑀳𑀲𑁆𑀲𑀧𑀝𑀺𑀬𑀢𑁆𑀣𑀢𑁄;_  
_𑀢𑀫𑁂𑀯 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀅𑀕𑀸 𑀤𑁂𑀯𑀸𑀦 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(141.)

144\. _𑀢𑀫𑁄𑀓𑁆𑀓𑀫𑀺𑀯 𑀆𑀬𑀦𑁆𑀢𑀺𑀁,_  
_𑀚𑀮𑀦𑁆𑀢𑀺𑀁 𑀯𑀺𑀚𑁆𑀚𑀼𑀢𑀁 𑀬𑀣𑀸;_  
_𑀧𑀢𑀻𑀢𑁄 𑀲𑀼𑀫𑀦𑁄 𑀯𑀺𑀢𑁆𑀢𑁄,_  
_𑀤𑁂𑀯𑀺𑀦𑁆𑀤𑁄 𑀅𑀤𑀤𑀸 𑀯𑀭𑀁𑁇_  


(142.)

145\. _“𑀯𑀭𑀜𑁆𑀘𑁂 𑀫𑁂 𑀅𑀤𑁄 𑀲𑀓𑁆𑀓,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀫𑀺𑀲𑁆𑀲𑀭;_  
_𑀦𑀺𑀲𑀺𑀧𑁆𑀧𑀮𑁄𑀪𑀺𑀓𑀸 𑀕𑀘𑁆𑀙𑁂,_  
_𑀏𑀢𑀁 𑀲𑀓𑁆𑀓 𑀯𑀭𑀁 𑀯𑀭𑁂”𑀢𑀺𑁇_  


---

146\. 𑀅𑀮𑀫𑁆𑀩𑀼𑀲𑀸𑀚𑀸𑀢𑀓𑀁 𑀢𑀢𑀺𑀬𑀁𑁇
