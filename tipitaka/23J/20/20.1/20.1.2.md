# 𑁨𑁦𑁇𑁧𑁇𑁨 𑀲𑁄𑀡𑀦𑀦𑁆𑀤𑀚𑀸𑀢𑀓

* 𑀲𑀢𑁆𑀢𑀢𑀺𑀦𑀺𑀧𑀸𑀢

* 𑀓𑀼𑀲𑀯𑀕𑁆𑀕

* 𑀲𑁄𑀡𑀦𑀦𑁆𑀤𑀚𑀸𑀢𑀓

(92.)

675\. _“𑀤𑁂𑀯𑀢𑀸 𑀦𑀼𑀲𑀺 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀅𑀤𑀼 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀫𑀸,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(93.)

676\. _“𑀦𑀸𑀧𑀺 𑀤𑁂𑀯𑁄 𑀦 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀦𑀸𑀧𑀺 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀪𑀽𑀢𑁄 𑀇𑀤𑁆𑀥𑀺𑀫𑀸,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀪𑀸𑀭𑀥”𑁇_  


(94.)

677\. _“𑀓𑀢𑀭𑀽𑀧𑀫𑀺𑀤𑀁 𑀪𑁄𑀢𑁄,_  
_𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀅𑀦𑀧𑁆𑀧𑀓𑀁;_  
_𑀤𑁂𑀯𑀫𑁆𑀳𑀺 𑀯𑀲𑁆𑀲𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀁 𑀪𑀯𑀁 𑀅𑀓𑀸𑁇_  


(95.)

678\. _𑀢𑀢𑁄 𑀯𑀸𑀢𑀸𑀢𑀧𑁂 𑀖𑁄𑀭𑁂,_  
_𑀲𑀻𑀢𑀘𑁆𑀙𑀸𑀬𑀁 𑀪𑀯𑀁 𑀅𑀓𑀸;_  
_𑀢𑀢𑁄 𑀅𑀫𑀺𑀢𑁆𑀢𑀫𑀚𑁆𑀛𑁂𑀲𑀼,_  
_𑀲𑀭𑀢𑀸𑀡𑀁 𑀪𑀯𑀁 𑀅𑀓𑀸𑁇_  


(96.)

679\. _𑀢𑀢𑁄 𑀨𑀻𑀢𑀸𑀦𑀺 𑀭𑀝𑁆𑀞𑀸𑀦𑀺,_  
_𑀯𑀲𑀺𑀦𑁄 𑀢𑁂 𑀪𑀯𑀁 𑀅𑀓𑀸;_  
_𑀢𑀢𑁄 𑀏𑀓𑀲𑀢𑀁 𑀔𑀢𑁆𑀬𑁂,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑁂 𑀪𑀯𑀁 𑀅𑀓𑀸𑁇_  


(97.)

680\. _𑀧𑀢𑀻𑀢𑀸𑀲𑁆𑀲𑀼 𑀫𑀬𑀁 𑀪𑁄𑀢𑁄,_  
_𑀯𑀤 𑀢𑀁 𑀪𑀜𑁆𑀚𑀫𑀺𑀘𑁆𑀙𑀲𑀺;_  
_𑀳𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀭𑀣𑀁,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀘 𑀅𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀭𑀫𑁆𑀫𑀸𑀦𑀺,_  
_𑀫𑀬𑀁 𑀪𑁄𑀢𑁄 𑀤𑀤𑀸𑀫𑀲𑁂𑁇_  


(98.)

681\. _𑀅𑀣 𑀯𑀗𑁆𑀕𑁂 𑀯𑀸 𑀫𑀕𑀥𑁂,_  
_𑀫𑀬𑀁 𑀪𑁄𑀢𑁄 𑀤𑀤𑀸𑀫𑀲𑁂;_  
_𑀅𑀣 𑀯𑀸 𑀅𑀲𑁆𑀲𑀓𑀸𑀯𑀦𑁆𑀢𑀻,_  
_𑀲𑀼𑀫𑀦𑀸 𑀤𑀫𑁆𑀫 𑀢𑁂 𑀫𑀬𑀁𑁇_  


(99.)

682\. _𑀉𑀧𑀟𑁆𑀠𑀁 𑀯𑀸𑀧𑀺 𑀭𑀚𑁆𑀚𑀲𑁆𑀲,_  
_𑀫𑀬𑀁 𑀪𑁄𑀢𑁄 𑀤𑀤𑀸𑀫𑀲𑁂;_  
_𑀲𑀘𑁂 𑀢𑁂 𑀅𑀢𑁆𑀣𑁄 𑀭𑀚𑁆𑀚𑁂𑀦,_  
_𑀅𑀦𑀼𑀲𑀸𑀲 𑀬𑀤𑀺𑀘𑁆𑀙𑀲𑀺”𑁇_  


(100.)

683\. _“𑀦 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄𑀧𑀺 𑀭𑀚𑁆𑀚𑁂𑀦,_  
_𑀦𑀕𑀭𑁂𑀦 𑀥𑀦𑁂𑀦 𑀯𑀸;_  
_𑀅𑀣𑁄𑀧𑀺 𑀚𑀦𑀧𑀤𑁂𑀦,_  
_𑀅𑀢𑁆𑀣𑁄 𑀫𑀬𑁆𑀳𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(101.)

684\. _𑀪𑁄𑀢𑁄𑀯 𑀭𑀝𑁆𑀞𑁂 𑀯𑀺𑀚𑀺𑀢𑁂,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀅𑀢𑁆𑀣𑀺 𑀅𑀲𑁆𑀲𑀫𑁄;_  
_𑀧𑀺𑀢𑀸 𑀫𑀬𑁆𑀳𑀁 𑀚𑀦𑁂𑀢𑁆𑀢𑀻 𑀘,_  
_𑀉𑀪𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀅𑀲𑁆𑀲𑀫𑁂𑁇_  


(102.)

685\. _𑀢𑁂𑀲𑀸𑀳𑀁 𑀧𑀼𑀩𑁆𑀩𑀸𑀘𑀭𑀺𑀬𑁂𑀲𑀼,_  
_𑀧𑀼𑀜𑁆𑀜𑀁 𑀦 𑀮𑀪𑀸𑀫𑀺 𑀓𑀸𑀢𑀯𑁂;_  
_𑀪𑀯𑀦𑁆𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀲𑁄𑀡𑀁 𑀬𑀸𑀘𑁂𑀫𑀼 𑀲𑀁𑀯𑀭𑀁”𑁇_  


(103.)

686\. _“𑀓𑀭𑁄𑀫𑀺 𑀢𑁂 𑀢𑀁 𑀯𑀘𑀦𑀁,_  
_𑀬𑀁 𑀫𑀁 𑀪𑀡𑀲𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀏𑀢𑀜𑁆𑀘 𑀔𑁄 𑀦𑁄 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀓𑀻𑀯𑀦𑁆𑀢𑁄 𑀳𑁄𑀦𑁆𑀢𑀼 𑀬𑀸𑀘𑀓𑀸”𑁇_  


(104.)

687\. _“𑀧𑀭𑁄𑀲𑀢𑀁 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀇𑀫𑁂 𑀘 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀪𑀯𑀜𑁆𑀘 𑀭𑀸𑀚𑀸 𑀫𑀦𑁄𑀚𑁄,_  
_𑀅𑀮𑀁 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀬𑀸𑀘𑀓𑀸”𑁇_  


(105.)

688\. _“𑀳𑀢𑁆𑀣𑀻 𑀅𑀲𑁆𑀲𑁂 𑀘 𑀬𑁄𑀚𑁂𑀦𑁆𑀢𑀼,_  
_𑀭𑀣𑀁 𑀲𑀦𑁆𑀦𑀬𑁆𑀳 𑀲𑀸𑀭𑀣𑀺;_  
_𑀆𑀩𑀦𑁆𑀥𑀦𑀸𑀦𑀺 𑀕𑀡𑁆𑀳𑀸𑀣,_  
_𑀧𑀸𑀤𑀸𑀲𑀼𑀲𑁆𑀲𑀸𑀭𑀬𑀤𑁆𑀥𑀚𑁂;_  
_𑀅𑀲𑁆𑀲𑀫𑀁 𑀢𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀢𑀺 𑀓𑁄𑀲𑀺𑀬𑁄”𑁇_  


(106.)

689\. _𑀢𑀢𑁄 𑀘 𑀭𑀸𑀚𑀸 𑀧𑀸𑀬𑀸𑀲𑀺,_  
_𑀲𑁂𑀦𑀸𑀬 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀻;_  
_𑀅𑀕𑀫𑀸 𑀅𑀲𑁆𑀲𑀫𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀢𑀺 𑀓𑁄𑀲𑀺𑀬𑁄𑁇_  


(107.)

690\. _“𑀓𑀲𑁆𑀲 𑀓𑀸𑀤𑀫𑁆𑀩𑀬𑁄 𑀓𑀸𑀚𑁄,_  
_𑀯𑁂𑀳𑀸𑀲𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀼𑀮𑀁;_  
_𑀅𑀁𑀲𑀁 𑀅𑀲𑀫𑁆𑀨𑀼𑀲𑀁 𑀏𑀢𑀺,_  
_𑀉𑀤𑀳𑀸𑀭𑀸𑀬 𑀕𑀘𑁆𑀙𑀢𑁄”𑁇_  


(108.)

691\. _“𑀅𑀳𑀁 𑀲𑁄𑀡𑁄 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀢𑀸𑀧𑀲𑁄 𑀲𑀳𑀺𑀢𑀩𑁆𑀩𑀢𑁄;_  
_𑀪𑀭𑀸𑀫𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀫𑀢𑀦𑁆𑀤𑀺𑀢𑁄𑁇_  


(109.)

692\. _𑀯𑀦𑁂 𑀨𑀮𑀜𑁆𑀘 𑀫𑀽𑀮𑀜𑁆𑀘,_  
_𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀧𑁄𑀲𑁂𑀫𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀓𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀁”𑁇_  


(110.)

693\. _“𑀇𑀘𑁆𑀙𑀸𑀫 𑀅𑀲𑁆𑀲𑀫𑀁 𑀕𑀦𑁆𑀢𑀼𑀁,_  
_𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀢𑀺 𑀓𑁄𑀲𑀺𑀬𑁄;_  
_𑀫𑀕𑁆𑀕𑀁 𑀦𑁄 𑀲𑁄𑀡 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑁂𑀫𑀼 𑀅𑀲𑁆𑀲𑀫𑀁”𑁇_  


(111.)

694\. _“𑀅𑀬𑀁 𑀏𑀓𑀧𑀤𑀻 𑀭𑀸𑀚,_  
_𑀬𑁂𑀦𑁂𑀢𑀁 𑀫𑁂𑀖𑀲𑀦𑁆𑀦𑀺𑀪𑀁;_  
_𑀓𑁄𑀯𑀺𑀍𑀆𑀭𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀏𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀢𑀺 𑀓𑁄𑀲𑀺𑀬𑁄”𑁇_  


(112.)

695\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺,_  
_𑀢𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀸𑀇𑀲𑀺;_  
_𑀯𑁂𑀳𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸𑀦 𑀔𑀢𑁆𑀢𑀺𑀬𑁂𑁇_  


(113.)

696\. _𑀅𑀲𑁆𑀲𑀫𑀁 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑀦 𑀆𑀲𑀦𑀁;_  
_𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸,_  
_𑀧𑀺𑀢𑀭𑀁 𑀧𑀝𑀺𑀩𑁄𑀥𑀬𑀺𑁇_  


(114.)

697\. _“𑀇𑀫𑁂 𑀆𑀬𑀦𑁆𑀢𑀺 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀅𑀲𑁆𑀲𑀫𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀲𑀻𑀤 𑀢𑁆𑀯𑀁 𑀫𑀳𑀸𑀇𑀲𑁂”𑁇_  


(115.)

698\. _𑀢𑀲𑁆𑀲 𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀢𑀭𑀫𑀸𑀦𑁄 𑀫𑀳𑀸𑀇𑀲𑀺;_  
_𑀅𑀲𑁆𑀲𑀫𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀲𑀤𑁆𑀯𑀸𑀭𑀫𑁆𑀳𑀺 𑀉𑀧𑀸𑀯𑀺𑀲𑀺𑁇_  


(116.)

699\. _𑀢𑀜𑁆𑀘 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀦𑁆𑀢𑀁,_  
_𑀚𑀮𑀦𑁆𑀢𑀁𑀭𑀺𑀯 𑀢𑁂𑀚𑀲𑀸;_  
_𑀔𑀢𑁆𑀬𑀲𑀗𑁆𑀖𑀧𑀭𑀺𑀩𑁆𑀬𑀽𑀍𑀳𑀁,_  
_𑀓𑁄𑀲𑀺𑀬𑁄 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀺𑁇_  


(117.)

700\. _𑀓𑀲𑁆𑀲 𑀪𑁂𑀭𑀻 𑀫𑀼𑀤𑀺𑀗𑁆𑀕𑀸 𑀘,_  
_𑀲𑀗𑁆𑀔𑀸 𑀧𑀡𑀯𑀤𑀺𑀦𑁆𑀤𑀺𑀫𑀸;_  
_𑀧𑀼𑀭𑀢𑁄 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸𑀦𑀺,_  
_𑀳𑀸𑀲𑀬𑀦𑁆𑀢𑀸 𑀭𑀣𑁂𑀲𑀪𑀁𑁇_  


(118.)

701\. _𑀓𑀲𑁆𑀲 𑀓𑀜𑁆𑀘𑀦𑀧𑀝𑁆𑀝𑁂𑀦,_  
_𑀧𑀼𑀣𑀼𑀦𑀸 𑀯𑀺𑀚𑁆𑀚𑀼𑀯𑀡𑁆𑀡𑀺𑀦𑀸;_  
_𑀬𑀼𑀯𑀸 𑀓𑀮𑀸𑀧𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑁄,_  
_𑀓𑁄 𑀏𑀢𑀺 𑀲𑀺𑀭𑀺𑀬𑀸 𑀚𑀮𑀁𑁇_  


(119.)

702\. _𑀉𑀓𑁆𑀓𑀸𑀫𑀼𑀔𑀧𑀳𑀝𑁆𑀞𑀁𑀯,_  
_𑀔𑀤𑀺𑀭𑀗𑁆𑀕𑀸𑀭𑀲𑀦𑁆𑀦𑀺𑀪𑀁;_  
_𑀫𑀼𑀔𑀜𑁆𑀘 𑀭𑀼𑀘𑀺𑀭𑀸 𑀪𑀸𑀢𑀺,_  
_𑀓𑁄 𑀏𑀢𑀺 𑀲𑀺𑀭𑀺𑀬𑀸 𑀚𑀮𑀁𑁇_  


(120.)

703\. _𑀓𑀲𑁆𑀲 𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀙𑀢𑁆𑀢𑀁,_  
_𑀲𑀲𑀮𑀸𑀓𑀁 𑀫𑀦𑁄𑀭𑀫𑀁;_  
_𑀆𑀤𑀺𑀘𑁆𑀘𑀭𑀁𑀲𑀸𑀯𑀭𑀡𑀁,_  
_𑀓𑁄 𑀏𑀢𑀺 𑀲𑀺𑀭𑀺𑀬𑀸 𑀚𑀮𑀁𑁇_  


(121.)

704\. _𑀓𑀲𑁆𑀲 𑀅𑀗𑁆𑀕𑀁 𑀧𑀭𑀺𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀯𑀸𑀍𑀅𑀩𑀻𑀚𑀦𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀁;_  
_𑀘𑀭𑀦𑁆𑀢𑀺 𑀯𑀭𑀧𑀼𑀜𑁆𑀜𑀲𑁆𑀲,_  
_𑀳𑀢𑁆𑀣𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁂𑀦 𑀆𑀬𑀢𑁄𑁇_  


(122.)

705\. _𑀓𑀲𑁆𑀲 𑀲𑁂𑀢𑀸𑀦𑀺 𑀙𑀢𑁆𑀢𑀸𑀦𑀺,_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑀸 𑀘 𑀯𑀫𑁆𑀫𑀺𑀢𑀸;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀻𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀓𑁄 𑀏𑀢𑀺 𑀲𑀺𑀭𑀺𑀬𑀸 𑀚𑀮𑀁𑁇_  


(123.)

706\. _𑀓𑀲𑁆𑀲 𑀏𑀓𑀲𑀢𑀁 𑀔𑀢𑁆𑀬𑀸,_  
_𑀅𑀦𑀼𑀬𑀦𑁆𑀢𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀲𑀫𑀦𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺,_  
_𑀓𑁄 𑀏𑀢𑀺 𑀲𑀺𑀭𑀺𑀬𑀸 𑀚𑀮𑀁𑁇_  


(124.)

707\. _𑀳𑀢𑁆𑀣𑀺 𑀅𑀲𑁆𑀲𑀭𑀣 𑀧𑀢𑁆𑀢𑀺,_  
_𑀲𑁂𑀦𑀸 𑀘 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀻;_  
_𑀲𑀫𑀦𑁆𑀢𑀸𑀦𑀼𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀺,_  
_𑀓𑁄 𑀏𑀢𑀺 𑀲𑀺𑀭𑀺𑀬𑀸 𑀚𑀮𑀁𑁇_  


(125.)

708\. _𑀓𑀲𑁆𑀲𑁂𑀲𑀸 𑀫𑀳𑀢𑀻 𑀲𑁂𑀦𑀸,_  
_𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀅𑀓𑁆𑀔𑁄𑀪𑀡𑀻 𑀅𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀸,_  
_𑀲𑀸𑀕𑀭𑀲𑁆𑀲𑁂𑀯 𑀊𑀫𑀺𑀬𑁄𑁇_  


(126.)

709\. _𑀭𑀸𑀚𑀸𑀪𑀺𑀭𑀸𑀚𑀸 𑀫𑀦𑁄𑀚𑁄,_  
_𑀇𑀦𑁆𑀤𑁄𑀯 𑀚𑀬𑀢𑀁 𑀧𑀢𑀺;_  
_𑀦𑀦𑁆𑀤𑀲𑁆𑀲𑀚𑁆𑀛𑀸𑀯𑀭𑀁 𑀏𑀢𑀺,_  
_𑀅𑀲𑁆𑀲𑀫𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀁𑁇_  


(127.)

710\. _𑀢𑀲𑁆𑀲𑁂𑀲𑀸 𑀫𑀳𑀢𑀻 𑀲𑁂𑀦𑀸,_  
_𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀅𑀓𑁆𑀔𑁄𑀪𑀡𑀻 𑀅𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀸,_  
_𑀲𑀸𑀕𑀭𑀲𑁆𑀲𑁂𑀯 𑀊𑀫𑀺𑀬𑁄𑁇_  


(128.)

711\. _𑀅𑀦𑀼𑀮𑀺𑀢𑁆𑀢𑀸 𑀘𑀦𑁆𑀤𑀦𑁂𑀦,_  
_𑀓𑀸𑀲𑀺𑀓𑀼𑀢𑁆𑀢𑀫𑀥𑀸𑀭𑀺𑀦𑁄;_  
_𑀲𑀩𑁆𑀩𑁂 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀇𑀲𑀻𑀦𑀁 𑀅𑀚𑁆𑀛𑀼𑀧𑀸𑀕𑀫𑀼𑀁𑁇_  


(129.)

712\. _“𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀪𑁄𑀢𑁄 𑀓𑀼𑀲𑀮𑀁,_  
_𑀓𑀘𑁆𑀘𑀺 𑀪𑁄𑀢𑁄 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀓𑀘𑁆𑀘𑀺 𑀉𑀜𑁆𑀙𑁂𑀦 𑀬𑀸𑀧𑁂𑀣,_  
_𑀓𑀘𑁆𑀘𑀺 𑀫𑀽𑀮𑀨𑀮𑀸 𑀩𑀳𑀽𑁇_  


(130.)

713\. _𑀓𑀘𑁆𑀘𑀺 𑀟𑀁𑀲𑀸 𑀫𑀓𑀲𑀸 𑀘,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀭𑀻𑀲𑀧𑀸;_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀓𑀘𑁆𑀘𑀺 𑀳𑀺𑀁𑀲𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(131.)

714\. _“𑀓𑀼𑀲𑀮𑀜𑁆𑀘𑁂𑀯 𑀦𑁄 𑀭𑀸𑀚,_  
_𑀅𑀣𑁄 𑀭𑀸𑀚 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀅𑀣𑁄 𑀉𑀜𑁆𑀙𑁂𑀦 𑀬𑀸𑀧𑁂𑀫,_  
_𑀅𑀣𑁄 𑀫𑀽𑀮𑀨𑀮𑀸 𑀩𑀳𑀽𑁇_  


(132.)

715\. _𑀅𑀣𑁄 𑀟𑀁𑀲𑀸 𑀫𑀓𑀲𑀸 𑀘,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀭𑀻𑀲𑀧𑀸;_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀳𑀺𑀁𑀲𑀸 𑀫𑀬𑁆𑀳𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(133.)

716\. _𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀧𑀽𑀕𑀸𑀦𑀺,_  
_𑀅𑀲𑁆𑀲𑀫𑁂 𑀲𑀫𑁆𑀫𑀢𑀁 𑀇𑀥;_  
_𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀆𑀩𑀸𑀥𑀁 𑀅𑀫𑀦𑁄𑀭𑀫𑀁𑁇_  


(134.)

717\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀅𑀣𑁄 𑀢𑁂 𑀅𑀤𑀼𑀭𑀸𑀕𑀢𑀁;_  
_𑀇𑀲𑁆𑀲𑀭𑁄𑀲𑀺 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀬𑀁 𑀇𑀥𑀢𑁆𑀣𑀺 𑀧𑀯𑁂𑀤𑀬𑁇_  


(135.)

718\. _𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀸𑀦𑀺 𑀧𑀺𑀬𑀸𑀮𑀸𑀦𑀺,_  
_𑀫𑀥𑀼𑀓𑁂 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀬𑁄;_  
_𑀨𑀮𑀸𑀦𑀺 𑀔𑀼𑀤𑁆𑀤𑀓𑀧𑁆𑀧𑀸𑀦𑀺,_  
_𑀪𑀼𑀜𑁆𑀚 𑀭𑀸𑀚 𑀯𑀭𑀁 𑀯𑀭𑀁𑁇_  


(136.)

719\. _𑀇𑀤𑀫𑁆𑀧𑀺 𑀧𑀸𑀦𑀻𑀬𑀁 𑀲𑀻𑀢𑀁,_  
_𑀆𑀪𑀢𑀁 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑀸;_  
_𑀢𑀢𑁄 𑀧𑀺𑀯 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀲𑀘𑁂 𑀢𑁆𑀯𑀁 𑀅𑀪𑀺𑀓𑀗𑁆𑀔𑀲𑀺”𑁇_  


(137.)

720\. _“𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀬𑀁 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀅𑀕𑁆𑀖𑀺𑀬𑀁 𑀓𑀢𑀁;_  
_𑀦𑀦𑁆𑀤𑀲𑁆𑀲𑀸𑀧𑀺 𑀦𑀺𑀲𑀸𑀫𑁂𑀣,_  
_𑀯𑀘𑀦𑀁 𑀲𑁄 𑀧𑀯𑀓𑁆𑀔𑀢𑀺𑁇_  


(138.)

721\. _𑀅𑀚𑁆𑀛𑀸𑀯𑀭𑀫𑁆𑀳𑀸 𑀦𑀦𑁆𑀤𑀲𑁆𑀲,_  
_𑀪𑁄𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑀫𑀸𑀕𑀢𑀸;_  
_𑀲𑀼𑀡𑀸𑀢𑀼 𑀪𑀯𑀁 𑀯𑀘𑀦𑀁,_  
_𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀧𑀭𑀺𑀲𑀸𑀬 𑀘”𑁇_  


(139.)

722\. _“𑀧𑀭𑁄𑀲𑀢𑀁 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀫𑀳𑀸𑀲𑀸𑀮𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀇𑀫𑁂 𑀘 𑀔𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀩𑁆𑀩𑁂,_  
_𑀅𑀪𑀺𑀚𑀸𑀢𑀸 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑁄;_  
_𑀪𑀯𑀜𑁆𑀘 𑀭𑀸𑀚𑀸 𑀫𑀦𑁄𑀚𑁄,_  
_𑀅𑀦𑀼𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀼 𑀫𑁂 𑀯𑀘𑁄𑁇_  


(140.)

723\. _𑀬𑁂 𑀘 𑀲𑀦𑁆𑀢𑀺 𑀲𑀫𑀻𑀢𑀸𑀭𑁄,_  
_𑀬𑀓𑁆𑀔𑀸𑀦𑀺 𑀇𑀥 𑀫𑀲𑁆𑀲𑀫𑁂;_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀪𑀽𑀢𑀪𑀩𑁆𑀬𑀸𑀦𑀺,_  
_𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀯𑀘𑀦𑀁 𑀫𑀫𑁇_  


(141.)

724\. _𑀦𑀫𑁄 𑀓𑀢𑁆𑀯𑀸𑀦 𑀪𑀽𑀢𑀸𑀦𑀁,_  
_𑀇𑀲𑀺𑀁 𑀯𑀓𑁆𑀔𑀸𑀫𑀺 𑀲𑀼𑀩𑁆𑀩𑀢𑀁;_  
_𑀲𑁄 𑀢𑁆𑀬𑀸𑀳𑀁 𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀩𑀸𑀳𑀼,_  
_𑀢𑀯 𑀓𑁄𑀲𑀺𑀬 𑀲𑀫𑁆𑀫𑀢𑁄𑁇_  


(142.)

725\. _𑀧𑀺𑀢𑀭𑀁 𑀫𑁂 𑀚𑀦𑁂𑀢𑁆𑀢𑀺𑀜𑁆𑀘,_  
_𑀪𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑀲𑁆𑀲 𑀫𑁂 𑀲𑀢𑁄;_  
_𑀯𑀻𑀭 𑀧𑀼𑀜𑁆𑀜𑀫𑀺𑀤𑀁 𑀞𑀸𑀦𑀁,_  
_𑀫𑀸 𑀫𑀁 𑀓𑁄𑀲𑀺𑀬 𑀯𑀸𑀭𑀬𑁇_  


(143.)

726\. _𑀲𑀩𑁆𑀪𑀺 𑀳𑁂𑀢𑀁 𑀉𑀧𑀜𑁆𑀜𑀸𑀢𑀁,_  
_𑀫𑀫𑁂𑀢𑀁 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀚;_  
_𑀉𑀝𑁆𑀞𑀸𑀦𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀢𑀬𑀸 𑀓𑀢𑀁;_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺,_  
_𑀫𑀫 𑀮𑁄𑀓𑀤𑀤𑁄 𑀪𑀯𑁇_  


(144.)

727\. _𑀢𑀣𑁂𑀯 𑀲𑀦𑁆𑀢𑀺 𑀫𑀦𑀼𑀚𑀸,_  
_𑀥𑀫𑁆𑀫𑁂 𑀥𑀫𑁆𑀫𑀧𑀤𑀁 𑀯𑀺𑀤𑀽;_  
_𑀫𑀕𑁆𑀕𑁄 𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀬𑀣𑀸 𑀚𑀸𑀦𑀸𑀲𑀺 𑀢𑁆𑀯𑀁 𑀇𑀲𑁂𑁇_  


(145.)

728\. _𑀉𑀝𑁆𑀞𑀸𑀦𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀼𑀲𑀼𑀔𑀸𑀯𑀳𑀁;_  
_𑀢𑀁 𑀫𑀁 𑀧𑀼𑀜𑁆𑀜𑀸 𑀦𑀺𑀯𑀸𑀭𑁂𑀢𑀺,_  
_𑀅𑀭𑀺𑀬𑀫𑀕𑁆𑀕𑀸𑀯𑀭𑁄 𑀦𑀭𑁄”𑁇_  


(146.)

729\. _“𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀪𑁄𑀦𑁆𑀢𑁄 𑀯𑀘𑀦𑀁,_  
_𑀪𑀸𑀢𑀼𑀭𑀚𑁆𑀛𑀸𑀯𑀭𑀸 𑀫𑀫;_  
_𑀓𑀼𑀮𑀯𑀁𑀲𑀁 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀧𑁄𑀭𑀸𑀡𑀁 𑀧𑀭𑀺𑀳𑀸𑀧𑀬𑀁;_  
_𑀅𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀻 𑀚𑁂𑀝𑁆𑀞𑁂𑀲𑀼,_  
_𑀦𑀺𑀭𑀬𑀁 𑀲𑁄𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇_  


(147.)

730\. _𑀬𑁂 𑀘 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀓𑀼𑀲𑀮𑀸,_  
_𑀧𑁄𑀭𑀸𑀡𑀲𑁆𑀲 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀘𑀸𑀭𑀺𑀢𑁆𑀢𑁂𑀦 𑀘 𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀦 𑀢𑁂 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁𑁇_  


(148.)

731\. _𑀫𑀸𑀢𑀸 𑀧𑀺𑀢𑀸 𑀘 𑀪𑀸𑀢𑀸 𑀘,_  
_𑀪𑀕𑀺𑀦𑀻 𑀜𑀸𑀢𑀺𑀩𑀦𑁆𑀥𑀯𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀚𑁂𑀝𑁆𑀞𑀲𑁆𑀲 𑀢𑁂 𑀪𑀸𑀭𑀸,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀪𑀸𑀭𑀥𑁇_  


(149.)

732\. _𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀕𑀭𑀼𑀁 𑀪𑀸𑀭𑀁,_  
_𑀦𑀸𑀯𑀺𑀓𑁄 𑀯𑀺𑀬 𑀉𑀲𑁆𑀲𑀳𑁂;_  
_𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀚𑁂𑀝𑁆𑀞𑁄 𑀘𑀲𑁆𑀫𑀺 𑀭𑀣𑁂𑀲𑀪”𑁇_  


(150.)

733\. _“𑀅𑀥𑀺𑀕𑀫𑀸 𑀢𑀫𑁂 𑀜𑀸𑀡𑀁,_  
_𑀚𑀸𑀮𑀁𑀯 𑀚𑀸𑀢𑀯𑁂𑀤𑀢𑁄;_  
_𑀏𑀯𑀫𑁂𑀯 𑀦𑁄 𑀪𑀯𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀓𑁄𑀲𑀺𑀬𑁄 𑀧𑀯𑀺𑀤𑀁𑀲𑀬𑀺𑁇_  


(151.)

734\. _𑀬𑀣𑀸 𑀉𑀤𑀬𑀫𑀸𑀤𑀺𑀘𑁆𑀘𑁄,_  
_𑀯𑀸𑀲𑀼𑀤𑁂𑀯𑁄 𑀧𑀪𑀗𑁆𑀓𑀭𑁄;_  
_𑀧𑀸𑀡𑀻𑀦𑀁 𑀧𑀯𑀺𑀤𑀁𑀲𑁂𑀢𑀺,_  
_𑀭𑀽𑀧𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑀧𑀸𑀧𑀓𑀁;_  
_𑀏𑀯𑀫𑁂𑀯 𑀦𑁄 𑀪𑀯𑀁 𑀥𑀫𑁆𑀫𑀁,_  
_𑀓𑁄𑀲𑀺𑀬𑁄 𑀧𑀯𑀺𑀤𑀁𑀲𑀬𑀺”𑁇_  


(152.)

735\. _“𑀏𑀯𑀁 𑀫𑁂 𑀬𑀸𑀘𑀫𑀸𑀦𑀲𑁆𑀲,_  
_𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀦𑀸𑀯𑀩𑀼𑀚𑁆𑀛𑀣;_  
_𑀢𑀯 𑀧𑀤𑁆𑀥𑀘𑀭𑁄 𑀳𑁂𑀲𑁆𑀲𑀁,_  
_𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁄”𑁇_  


(153.)

736\. _“𑀅𑀤𑁆𑀥𑀸 𑀦𑀦𑁆𑀤 𑀯𑀺𑀚𑀸𑀦𑀸𑀲𑀺,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀪𑀺 𑀤𑁂𑀲𑀺𑀢𑀁;_  
_𑀅𑀭𑀺𑀬𑁄 𑀅𑀭𑀺𑀬𑀲𑀫𑀸𑀘𑀸𑀭𑁄,_  
_𑀩𑀸𑀍𑀳𑀁 𑀢𑁆𑀯𑀁 𑀫𑀫 𑀭𑀼𑀘𑁆𑀘𑀲𑀺𑁇_  


(154.)

737\. _𑀪𑀯𑀦𑁆𑀢𑀁 𑀯𑀤𑀸𑀫𑀺 𑀪𑁄𑀢𑀺𑀜𑁆𑀘,_  
_𑀲𑀼𑀡𑀸𑀣 𑀯𑀘𑀦𑀁 𑀫𑀫;_  
_𑀦𑀸𑀬𑀁 𑀪𑀸𑀭𑁄 𑀪𑀸𑀭𑀫𑀢𑁄,_  
_𑀅𑀳𑀼 𑀫𑀬𑁆𑀳𑀁 𑀓𑀼𑀤𑀸𑀘𑀦𑀁𑁇_  


(155.)

738\. _𑀢𑀁 𑀫𑀁 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀁 𑀲𑀦𑁆𑀢𑀁,_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀼𑀲𑀼𑀔𑀸𑀯𑀳𑀁;_  
_𑀦𑀦𑁆𑀤𑁄 𑀅𑀚𑁆𑀛𑀸𑀯𑀭𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀸𑀬 𑀬𑀸𑀘𑀢𑀺𑁇_  


(156.)

739\. _𑀬𑁄 𑀯𑁂 𑀇𑀘𑁆𑀙𑀢𑀺 𑀓𑀸𑀫𑁂𑀦,_  
_𑀲𑀦𑁆𑀢𑀸𑀦𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀁;_  
_𑀦𑀦𑁆𑀤𑀁 𑀯𑁄 𑀯𑀭𑀣 𑀏𑀓𑁄,_  
_𑀓𑀁 𑀦𑀦𑁆𑀤𑁄 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀼”𑁇_  


(157.)

740\. _“𑀢𑀬𑀸 𑀢𑀸𑀢 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸,_  
_𑀲𑁄𑀡 𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑀸 𑀫𑀬𑀁;_  
_𑀉𑀧𑀖𑀸𑀢𑀼𑀁 𑀮𑀪𑁂 𑀦𑀦𑁆𑀤𑀁,_  
_𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑀁𑁇_  


(158.)

741\. _𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀲𑁆𑀲𑁂𑀯 𑀢𑀭𑀼𑀡𑀁,_  
_𑀧𑀯𑀸𑀍𑀅𑀁 𑀫𑀸𑀮𑀼𑀢𑁂𑀭𑀺𑀢𑀁;_  
_𑀘𑀺𑀭𑀲𑁆𑀲𑀁 𑀦𑀦𑁆𑀤𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦,_  
_𑀳𑀤𑀬𑀁 𑀫𑁂 𑀧𑀯𑁂𑀥𑀢𑀺𑁇_  


(159.)

742\. _𑀬𑀤𑀸 𑀲𑀼𑀢𑁆𑀢𑀸𑀧𑀺 𑀲𑀼𑀧𑀺𑀦𑁂,_  
_𑀦𑀦𑁆𑀤𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀕𑀢𑀁;_  
_𑀉𑀤𑀕𑁆𑀕𑀸 𑀲𑀼𑀫𑀦𑀸 𑀳𑁄𑀫𑀺,_  
_𑀦𑀦𑁆𑀤𑁄 𑀦𑁄 𑀆𑀕𑀢𑁄 𑀅𑀬𑀁𑁇_  


(160.)

743\. _𑀬𑀤𑀸 𑀘 𑀧𑀝𑀺𑀩𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸,_  
_𑀦𑀦𑁆𑀤𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀦𑀸𑀕𑀢𑀁;_  
_𑀪𑀺𑀬𑁆𑀬𑁄 𑀆𑀯𑀺𑀲𑀢𑀻 𑀲𑁄𑀓𑁄,_  
_𑀤𑁄𑀫𑀦𑀲𑁆𑀲𑀜𑁆𑀘𑀦𑀧𑁆𑀧𑀓𑀁𑁇_  


(161.)

744\. _𑀲𑀸𑀳𑀁 𑀅𑀚𑁆𑀚 𑀘𑀺𑀭𑀲𑁆𑀲𑀫𑁆𑀧𑀺,_  
_𑀦𑀦𑁆𑀤𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀆𑀕𑀢𑀁;_  
_𑀪𑀢𑁆𑀢𑀼𑀘𑁆𑀘 𑀫𑀬𑁆𑀳𑀜𑁆𑀘 𑀧𑀺𑀬𑁄,_  
_𑀦𑀦𑁆𑀤𑁄 𑀦𑁄 𑀧𑀸𑀯𑀺𑀲𑀻 𑀖𑀭𑀁𑁇_  


(162.)

745\. _𑀧𑀺𑀢𑀼𑀧𑀺 𑀦𑀦𑁆𑀤𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄,_  
_𑀬𑀁 𑀦𑀦𑁆𑀤𑁄 𑀦𑀧𑁆𑀧𑀯𑀲𑁂 𑀖𑀭𑀸;_  
_𑀮𑀪𑀢𑀽 𑀢𑀸𑀢 𑀦𑀦𑁆𑀤𑁄 𑀢𑀁,_  
_𑀫𑀁 𑀦𑀦𑁆𑀤𑁄 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀢𑀼”𑁇_  


(163.)

746\. _“𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀺𑀓𑀸 𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀘,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀭𑀲𑀤𑀤𑀻 𑀘 𑀦𑁄;_  
_𑀫𑀕𑁆𑀕𑁄 𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀫𑀸𑀢𑀸 𑀢𑀁 𑀯𑀭𑀢𑁂 𑀇𑀲𑁂𑁇_  


(164.)

747\. _𑀧𑀼𑀩𑁆𑀩𑁂 𑀭𑀲𑀤𑀤𑀻 𑀕𑁄𑀢𑁆𑀢𑀻,_  
_𑀫𑀸𑀢𑀸 𑀧𑀼𑀜𑁆𑀜𑀽𑀧𑀲𑀁𑀳𑀺𑀢𑀸;_  
_𑀫𑀕𑁆𑀕𑁄 𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀫𑀸𑀢𑀸 𑀢𑀁 𑀯𑀭𑀢𑁂 𑀇𑀲𑁂”𑁇_  


(165.)

748\. _“𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑀸 𑀧𑀼𑀢𑁆𑀢𑀨𑀮𑀁,_  
_𑀤𑁂𑀯𑀢𑀸𑀬 𑀦𑀫𑀲𑁆𑀲𑀢𑀺;_  
_𑀦𑀓𑁆𑀔𑀢𑁆𑀢𑀸𑀦𑀺 𑀘 𑀧𑀼𑀘𑁆𑀙𑀢𑀺,_  
_𑀉𑀢𑀼𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑀸𑀦𑀺 𑀘𑁇_  


(166.)

749\. _𑀢𑀲𑁆𑀲𑀸 𑀉𑀢𑀼𑀫𑁆𑀳𑀺 𑀦𑁆𑀳𑀸𑀢𑀸𑀬,_  
_𑀳𑁄𑀢𑀺 𑀕𑀩𑁆𑀪𑀲𑁆𑀲 𑀯𑁄𑀓𑁆𑀓𑀫𑁄;_  
_𑀢𑁂𑀦 𑀤𑁄𑀳𑀍𑀇𑀦𑀻 𑀳𑁄𑀢𑀺,_  
_𑀲𑀼𑀳𑀤𑀸 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(167.)

750\. _𑀲𑀁𑀯𑀘𑁆𑀙𑀭𑀁 𑀯𑀸 𑀊𑀦𑀁 𑀯𑀸,_  
_𑀧𑀭𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀚𑀸𑀬𑀢𑀺;_  
_𑀢𑁂𑀦 𑀲𑀸 𑀚𑀦𑀬𑀦𑁆𑀢𑀻𑀢𑀺,_  
_𑀚𑀦𑁂𑀢𑁆𑀢𑀺 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(168.)

751\. _𑀣𑀦𑀔𑀻𑀭𑁂𑀦 𑀕𑀻𑀢𑁂𑀦,_  
_𑀅𑀗𑁆𑀕𑀧𑀸𑀯𑀼𑀭𑀡𑁂𑀦 𑀘;_  
_𑀭𑁄𑀤𑀦𑁆𑀢𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀢𑁄𑀲𑁂𑀢𑀺,_  
_𑀢𑁄𑀲𑁂𑀦𑁆𑀢𑀻 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(169.)

752\. _𑀢𑀢𑁄 𑀯𑀸𑀢𑀸𑀢𑀧𑁂 𑀖𑁄𑀭𑁂,_  
_𑀫𑀫𑀁 𑀓𑀢𑁆𑀯𑀸 𑀉𑀤𑀺𑀓𑁆𑀔𑀢𑀺;_  
_𑀤𑀸𑀭𑀓𑀁 𑀅𑀧𑁆𑀧𑀚𑀸𑀦𑀦𑁆𑀢𑀁,_  
_𑀧𑁄𑀲𑁂𑀦𑁆𑀢𑀻 𑀢𑁂𑀦 𑀯𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(170.)

753\. _𑀬𑀜𑁆𑀘 𑀫𑀸𑀢𑀼𑀥𑀦𑀁 𑀳𑁄𑀢𑀺,_  
_𑀬𑀜𑁆𑀘 𑀳𑁄𑀢𑀺 𑀧𑀺𑀢𑀼𑀤𑁆𑀥𑀦𑀁;_  
_𑀉𑀪𑀬𑀫𑁆𑀧𑁂𑀢𑀲𑁆𑀲 𑀕𑁄𑀧𑁂𑀢𑀺,_  
_𑀅𑀧𑀺 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀦𑁄 𑀲𑀺𑀬𑀸𑁇_  


(171.)

754\. _𑀏𑀯𑀁 𑀧𑀼𑀢𑁆𑀢 𑀅𑀤𑀼𑀁 𑀧𑀼𑀢𑁆𑀢,_  
_𑀇𑀢𑀺 𑀫𑀸𑀢𑀸 𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀺;_  
_𑀧𑀫𑀢𑁆𑀢𑀁 𑀧𑀭𑀤𑀸𑀭𑁂𑀲𑀼,_  
_𑀦𑀺𑀲𑀻𑀣𑁂 𑀧𑀢𑁆𑀢𑀬𑁄𑀩𑁆𑀩𑀦𑁂;_  
_𑀲𑀸𑀬𑀁 𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀦𑀸𑀬𑀦𑁆𑀢𑀁,_  
_𑀇𑀢𑀺 𑀫𑀸𑀢𑀸 𑀯𑀺𑀳𑀜𑁆𑀜𑀢𑀺𑁇_  


(172.)

755\. _𑀏𑀯𑀁 𑀓𑀺𑀘𑁆𑀙𑀸 𑀪𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀫𑀸𑀢𑀼 𑀅𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁄;_  
_𑀫𑀸𑀢𑀭𑀺 𑀫𑀺𑀘𑁆𑀙𑀸 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀭𑀬𑀁 𑀲𑁄𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇_  


(173.)

756\. _𑀏𑀯𑀁 𑀓𑀺𑀘𑁆𑀙𑀸 𑀪𑀢𑁄 𑀧𑁄𑀲𑁄,_  
_𑀧𑀺𑀢𑀼 𑀅𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁄;_  
_𑀧𑀺𑀢𑀭𑀺 𑀫𑀺𑀘𑁆𑀙𑀸 𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀭𑀬𑀁 𑀲𑁄𑀧𑀧𑀚𑁆𑀚𑀢𑀺𑁇_  


(174.)

757\. _𑀥𑀦𑀸𑀧𑀺 𑀥𑀦𑀓𑀸𑀫𑀸𑀦𑀁,_  
_𑀦𑀲𑁆𑀲𑀢𑀺 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀫𑀸𑀢𑀭𑀁 𑀅𑀧𑀭𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀺𑀘𑁆𑀙𑀁 𑀯𑀸 𑀲𑁄 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(175.)

758\. _𑀥𑀦𑀸𑀧𑀺 𑀥𑀦𑀓𑀸𑀫𑀸𑀦𑀁,_  
_𑀦𑀲𑁆𑀲𑀢𑀺 𑀇𑀢𑀺 𑀫𑁂 𑀲𑀼𑀢𑀁;_  
_𑀧𑀺𑀢𑀭𑀁 𑀅𑀧𑀭𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀺𑀘𑁆𑀙𑀁 𑀯𑀸 𑀲𑁄 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(176.)

759\. _𑀆𑀦𑀦𑁆𑀤𑁄 𑀘 𑀧𑀫𑁄𑀤𑁄 𑀘,_  
_𑀲𑀤𑀸 𑀳𑀲𑀺𑀢𑀓𑀻𑀍𑀇𑀢𑀁;_  
_𑀫𑀸𑀢𑀭𑀁 𑀧𑀭𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀮𑀩𑁆𑀪𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀢𑁄𑁇_  


(177.)

760\. _𑀆𑀦𑀦𑁆𑀤𑁄 𑀘 𑀧𑀫𑁄𑀤𑁄 𑀘,_  
_𑀲𑀤𑀸 𑀳𑀲𑀺𑀢𑀓𑀻𑀍𑀇𑀢𑀁;_  
_𑀧𑀺𑀢𑀭𑀁 𑀧𑀭𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀮𑀩𑁆𑀪𑀫𑁂𑀢𑀁 𑀯𑀺𑀚𑀸𑀦𑀢𑁄𑁇_  


(178.)

761\. _𑀤𑀸𑀦𑀜𑁆𑀘 𑀧𑀺𑀬𑀯𑀸𑀘𑀸 𑀘,_  
_𑀅𑀢𑁆𑀣𑀘𑀭𑀺𑀬𑀸 𑀘 𑀬𑀸 𑀇𑀥;_  
_𑀲𑀫𑀸𑀦𑀢𑁆𑀢𑀢𑀸 𑀘 𑀥𑀫𑁆𑀫𑁂𑀲𑀼,_  
_𑀢𑀢𑁆𑀣 𑀢𑀢𑁆𑀣 𑀬𑀣𑀸𑀭𑀳𑀁;_  
_𑀏𑀢𑁂 𑀔𑁄 𑀲𑀗𑁆𑀕𑀳𑀸 𑀮𑁄𑀓𑁂,_  
_𑀭𑀣𑀲𑁆𑀲𑀸𑀡𑀻𑀯 𑀬𑀸𑀬𑀢𑁄𑁇_  


(179.)

762\. _𑀏𑀢𑁂 𑀘 𑀲𑀗𑁆𑀕𑀳𑀸 𑀦𑀸𑀲𑁆𑀲𑀼,_  
_𑀦 𑀫𑀸𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀓𑀸𑀭𑀡𑀸;_  
_𑀮𑀪𑁂𑀣 𑀫𑀸𑀦𑀁 𑀧𑀽𑀚𑀁 𑀯𑀸,_  
_𑀧𑀺𑀢𑀸 𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑀓𑀸𑀭𑀡𑀸𑁇_  


(180.)

763\. _𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀗𑁆𑀕𑀳𑀸 𑀏𑀢𑁂,_  
_𑀲𑀫𑁆𑀫𑀧𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀺 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀢𑀲𑁆𑀫𑀸 𑀫𑀳𑀢𑁆𑀢𑀁 𑀧𑀧𑁆𑀧𑁄𑀦𑁆𑀢𑀺,_  
_𑀧𑀸𑀲𑀁𑀲𑀸 𑀘 𑀪𑀯𑀦𑁆𑀢𑀺 𑀢𑁂𑁇_  


(181.)

764\. _𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀢𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄,_  
_𑀧𑀼𑀩𑁆𑀩𑀸𑀘𑀭𑀺𑀬𑀸𑀢𑀺 𑀯𑀼𑀘𑁆𑀘𑀭𑁂;_  
_𑀆𑀳𑀼𑀦𑁂𑀬𑁆𑀬𑀸 𑀘 𑀧𑀼𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀧𑀚𑀸𑀬 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀓𑀸𑁇_  


(182.)

765\. _𑀢𑀲𑁆𑀫𑀸 𑀳𑀺 𑀦𑁂 𑀦𑀫𑀲𑁆𑀲𑁂𑀬𑁆𑀬,_  
_𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬 𑀘 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀅𑀦𑁆𑀦𑁂𑀦 𑀅𑀣𑁄 𑀧𑀸𑀦𑁂𑀦,_  
_𑀯𑀢𑁆𑀣𑁂𑀦 𑀲𑀬𑀦𑁂𑀦 𑀘;_  
_𑀉𑀘𑁆𑀙𑀸𑀤𑀦𑁂𑀦 𑀦𑁆𑀳𑀸𑀧𑀦𑁂𑀦,_  
_𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑁄𑀯𑀦𑁂𑀦 𑀘𑁇_  


(183.)

766\. _𑀢𑀸𑀬 𑀦𑀁 𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀲𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑀸;_  
_𑀇𑀥𑁂𑀯 𑀦𑀁 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺,_  
_𑀧𑁂𑀘𑁆𑀘 𑀲𑀕𑁆𑀕𑁂 𑀧𑀫𑁄𑀤𑀢𑀻”𑀢𑀺𑁇 (𑁩𑁭𑁧𑁮)_  


---

767\. 𑀲𑁄𑀡𑀦𑀦𑁆𑀤𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇



---

768\. 𑀲𑀢𑁆𑀢𑀢𑀺𑀦𑀺𑀧𑀸𑀢𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇



---

769\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



770\. _𑀅𑀣 𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑀫𑁆𑀳𑀺 𑀦𑀺𑀧𑀸𑀢𑀯𑀭𑁂,_  
_𑀲𑀪𑀸𑀯𑀦𑁆𑀢𑀼 𑀓𑀼𑀲𑀸𑀯𑀢𑀺𑀭𑀸𑀚𑀯𑀭𑁄;_  
_𑀅𑀣 𑀲𑁄𑀡𑀲𑀼𑀦𑀦𑁆𑀤𑀯𑀭𑁄 𑀘 𑀧𑀼𑀦,_  
_𑀅𑀪𑀺𑀯𑀸𑀲𑀺𑀢𑀲𑀢𑁆𑀢𑀢𑀺𑀫𑀫𑁆𑀳𑀺 𑀲𑀼𑀢𑁂𑀢𑀺𑁇_
