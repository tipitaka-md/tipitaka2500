# 𑁨𑁨𑁇𑁧𑁇𑁨 𑀫𑀳𑀸𑀚𑀦𑀓𑀚𑀸𑀢𑀓

### 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀫𑀳𑀸𑀚𑀦𑀓𑀚𑀸𑀢𑀓

(123.)

1438\. _“𑀓𑁄𑀬𑀁 𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀧𑀲𑁆𑀲𑀁 𑀢𑀻𑀭𑀫𑀸𑀬𑀼𑀳𑁂;_  
_𑀓𑀁 𑀢𑁆𑀯𑀁 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀜𑀢𑁆𑀯𑀸,_  
_𑀏𑀯𑀁 𑀯𑀸𑀬𑀫𑀲𑁂 𑀪𑀼𑀲𑀁”𑁇_  


(124.)

1439\. _“𑀦𑀺𑀲𑀫𑁆𑀫 𑀯𑀢𑁆𑀢𑀁 𑀮𑁄𑀓𑀲𑁆𑀲,_  
_𑀯𑀸𑀬𑀸𑀫𑀲𑁆𑀲 𑀘 𑀤𑁂𑀯𑀢𑁂;_  
_𑀢𑀲𑁆𑀫𑀸 𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀧𑀲𑁆𑀲𑀁 𑀢𑀻𑀭𑀫𑀸𑀬𑀼𑀳𑁂”𑁇_  


(125.)

1440\. _“𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑀲𑁆𑀫𑀺𑀁,_  
_𑀢𑀻𑀭𑀁 𑀬𑀲𑁆𑀲 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑁄𑀖𑁄 𑀢𑁂 𑀧𑀼𑀭𑀺𑀲𑀯𑀸𑀬𑀸𑀫𑁄,_  
_𑀅𑀧𑁆𑀧𑀢𑁆𑀯𑀸𑀯 𑀫𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(126.)

1441\. _“𑀅𑀦𑀡𑁄 𑀜𑀸𑀢𑀺𑀦𑀁 𑀳𑁄𑀢𑀺,_  
_𑀤𑁂𑀯𑀸𑀦𑀁 𑀧𑀺𑀢𑀼𑀦𑀜𑁆𑀘 𑀲𑁄;_  
_𑀓𑀭𑀁 𑀧𑀼𑀭𑀺𑀲𑀓𑀺𑀘𑁆𑀘𑀸𑀦𑀺,_  
_𑀦 𑀘 𑀧𑀘𑁆𑀙𑀸𑀦𑀼𑀢𑀧𑁆𑀧𑀢𑀺”𑁇_  


(127.)

1442\. _“𑀅𑀧𑀸𑀭𑀦𑁂𑀬𑁆𑀬𑀁 𑀬𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀅𑀨𑀮𑀁 𑀓𑀺𑀮𑀫𑀣𑀼𑀤𑁆𑀤𑀬𑀁;_  
_𑀢𑀢𑁆𑀣 𑀓𑁄 𑀯𑀸𑀬𑀫𑁂𑀦𑀢𑁆𑀣𑁄,_  
_𑀫𑀘𑁆𑀘𑀼 𑀬𑀲𑁆𑀲𑀸𑀪𑀺𑀦𑀺𑀧𑁆𑀧𑀢𑀁”𑁇_  


(128.)

1443\. _“𑀅𑀧𑀸𑀭𑀦𑁂𑀬𑁆𑀬𑀫𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀬𑁄 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸𑀦 𑀤𑁂𑀯𑀢𑁂;_  
_𑀦 𑀭𑀓𑁆𑀔𑁂 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑀸𑀡𑀁,_  
_𑀚𑀜𑁆𑀜𑀸 𑀲𑁄 𑀬𑀤𑀺 𑀳𑀸𑀧𑀬𑁂𑁇_  


(129.)

1444\. _𑀅𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀨𑀮𑀁 𑀏𑀓𑁂,_  
_𑀅𑀲𑁆𑀫𑀺𑀁 𑀮𑁄𑀓𑀲𑁆𑀫𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀧𑀬𑁄𑀚𑀬𑀦𑁆𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺,_  
_𑀢𑀸𑀦𑀺 𑀇𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀯𑀸 𑀦 𑀯𑀸𑁇_  


(130.)

1445\. _𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑀁 𑀓𑀫𑁆𑀫𑀨𑀮𑀁,_  
_𑀦𑀦𑀼 𑀧𑀲𑁆𑀲𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀲𑀦𑁆𑀦𑀸 𑀅𑀜𑁆𑀜𑁂 𑀢𑀭𑀸𑀫𑀳𑀁,_  
_𑀢𑀜𑁆𑀘 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


(131.)

1446\. _𑀲𑁄 𑀅𑀳𑀁 𑀯𑀸𑀬𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀣𑀸𑀲𑀢𑁆𑀢𑀺 𑀬𑀣𑀸𑀩𑀮𑀁;_  
_𑀕𑀘𑁆𑀙𑀁 𑀧𑀸𑀭𑀁 𑀲𑀫𑀼𑀤𑁆𑀤𑀲𑁆𑀲,_  
_𑀓𑀲𑁆𑀲𑀁 𑀧𑀼𑀭𑀺𑀲𑀓𑀸𑀭𑀺𑀬𑀁”𑁇_  


(132.)

1447\. _“𑀬𑁄 𑀢𑁆𑀯𑀁 𑀏𑀯𑀁 𑀕𑀢𑁂 𑀑𑀖𑁂,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀬𑁆𑀬𑁂 𑀫𑀳𑀡𑁆𑀡𑀯𑁂;_  
_𑀥𑀫𑁆𑀫𑀯𑀸𑀬𑀸𑀫𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀓𑀫𑁆𑀫𑀼𑀦𑀸 𑀦𑀸𑀯𑀲𑀻𑀤𑀲𑀺;_  
_𑀲𑁄 𑀢𑁆𑀯𑀁 𑀢𑀢𑁆𑀣𑁂𑀯 𑀕𑀘𑁆𑀙𑀸𑀳𑀺,_  
_𑀬𑀢𑁆𑀣 𑀢𑁂 𑀦𑀺𑀭𑀢𑁄 𑀫𑀦𑁄𑁇_  


1448\. _𑀲𑀼𑀭𑀺𑀬𑀼𑀕𑁆𑀕𑀫𑀦𑁂 𑀦𑀺𑀥𑀺,_  
_𑀅𑀣𑁄 𑀑𑀕𑁆𑀕𑀫𑀦𑁂 𑀦𑀺𑀥𑀺;_  
_𑀅𑀦𑁆𑀢𑁄 𑀦𑀺𑀥𑀺 𑀩𑀳𑀺 𑀦𑀺𑀥𑀺,_  
_𑀦 𑀅𑀦𑁆𑀢𑁄 𑀦 𑀩𑀳𑀺 𑀦𑀺𑀥𑀺𑁇_  


1449\. _𑀆𑀭𑁄𑀳𑀦𑁂 𑀫𑀳𑀸𑀦𑀺𑀥𑀺,_  
_𑀅𑀣𑁄 𑀑𑀭𑁄𑀳𑀦𑁂 𑀦𑀺𑀥𑀺;_  
_𑀘𑀢𑀽𑀲𑀼 𑀫𑀳𑀸𑀲𑀸𑀮𑁂𑀲𑀼,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑁂 𑀦𑀺𑀥𑀺𑁇_  


1450\. _𑀤𑀦𑁆𑀢𑀕𑁆𑀕𑁂𑀲𑀼 𑀫𑀳𑀸𑀦𑀺𑀥𑀺,_  
_𑀩𑀸𑀮𑀕𑁆𑀕𑁂𑀲𑀼 𑀘 𑀓𑁂𑀧𑀼𑀓𑁂;_  
_𑀭𑀼𑀓𑁆𑀔𑀕𑁆𑀕𑁂𑀲𑀼 𑀫𑀳𑀸𑀦𑀺𑀥𑀺,_  
_𑀲𑁄𑀍𑀅𑀲𑁂𑀢𑁂 𑀫𑀳𑀸𑀦𑀺𑀥𑀻;_  
_𑀲𑀳𑀲𑁆𑀲𑀣𑀸𑀫𑁄 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁄,_  
_𑀲𑀻𑀯𑀮𑀸𑀭𑀸𑀥𑀦𑁂𑀦 𑀘𑁇_  


(133.)

1451\. _𑀆𑀲𑀻𑀲𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀬𑀣𑀸 𑀇𑀘𑁆𑀙𑀺𑀁 𑀢𑀣𑀸 𑀅𑀳𑀼𑁇_  


(134.)

1452\. _𑀆𑀲𑀻𑀲𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀉𑀤𑀓𑀸 𑀣𑀮𑀫𑀼𑀩𑁆𑀪𑀢𑀁𑁇_  


(135.)

1453\. _𑀯𑀸𑀬𑀫𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀬𑀣𑀸 𑀇𑀘𑁆𑀙𑀺𑀁 𑀢𑀣𑀸 𑀅𑀳𑀼𑁇_  


(136.)

1454\. _𑀯𑀸𑀬𑀫𑁂𑀣𑁂𑀯 𑀧𑀼𑀭𑀺𑀲𑁄,_  
_𑀦 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁,_  
_𑀉𑀤𑀓𑀸 𑀣𑀮𑀫𑀼𑀩𑁆𑀪𑀢𑀁𑁇_  


(137.)

1455\. _𑀤𑀼𑀓𑁆𑀔𑀽𑀧𑀦𑀻𑀢𑁄𑀧𑀺 𑀦𑀭𑁄 𑀲𑀧𑀜𑁆𑀜𑁄,_  
_𑀆𑀲𑀁 𑀦 𑀙𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬 𑀲𑀼𑀔𑀸𑀕𑀫𑀸𑀬;_  
_𑀩𑀳𑀽 𑀳𑀺 𑀨𑀲𑁆𑀲𑀸 𑀅𑀳𑀺𑀢𑀸 𑀳𑀺𑀢𑀸 𑀘,_  
_𑀅𑀯𑀺𑀢𑀓𑁆𑀓𑀺𑀢𑀸 𑀫𑀘𑁆𑀘𑀼𑀫𑀼𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺𑁇_  


(138.)

1456\. _𑀅𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀫𑁆𑀧𑀺 𑀪𑀯𑀢𑀺,_  
_𑀘𑀺𑀦𑁆𑀢𑀺𑀢𑀫𑁆𑀧𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀺;_  
_𑀦 𑀳𑀺 𑀘𑀺𑀦𑁆𑀢𑀸𑀫𑀬𑀸 𑀪𑁄𑀕𑀸,_  
_𑀇𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀯𑀸”𑁇_  


(139.)

1457\. _“𑀅𑀧𑁄𑀭𑀸𑀡𑀁 𑀯𑀢 𑀪𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀩𑁆𑀩𑀪𑀼𑀫𑁆𑀫𑁄 𑀤𑀺𑀲𑀫𑁆𑀧𑀢𑀺;_  
_𑀦𑀸𑀚𑁆𑀚 𑀦𑀘𑁆𑀘𑁂 𑀦𑀺𑀲𑀸𑀫𑁂𑀢𑀺,_  
_𑀦 𑀕𑀻𑀢𑁂 𑀓𑀼𑀭𑀼𑀢𑁂 𑀫𑀦𑁄𑁇_  


(140.)

1458\. _𑀦 𑀫𑀺𑀕𑁂 𑀦𑀧𑀺 𑀉𑀬𑁆𑀬𑀸𑀦𑁂,_  
_𑀦𑀧𑀺 𑀳𑀁𑀲𑁂 𑀉𑀤𑀺𑀓𑁆𑀔𑀢𑀺;_  
_𑀫𑀽𑀕𑁄𑀯 𑀢𑀼𑀡𑁆𑀳𑀺𑀫𑀸𑀲𑀻𑀦𑁄,_  
_𑀦 𑀅𑀢𑁆𑀣𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀺”𑁇_  


(141.)

1459\. _“𑀲𑀼𑀔𑀓𑀸𑀫𑀸 𑀭𑀳𑁄𑀲𑀻𑀮𑀸,_  
_𑀯𑀥𑀩𑀦𑁆𑀥𑀸 𑀉𑀧𑀸𑀭𑀢𑀸;_  
_𑀓𑀲𑁆𑀲 𑀦𑀼 𑀅𑀚𑁆𑀚 𑀆𑀭𑀸𑀫𑁂,_  
_𑀤𑀳𑀭𑀸 𑀯𑀼𑀤𑁆𑀥𑀸 𑀘 𑀅𑀘𑁆𑀙𑀭𑁂𑁇_  


(142.)

1460\. _𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀦𑀣𑀸 𑀥𑀻𑀭𑀸,_  
_𑀦𑀫𑁄 𑀢𑁂𑀲𑀁 𑀫𑀳𑁂𑀲𑀺𑀦𑀁;_  
_𑀬𑁂 𑀉𑀲𑁆𑀲𑀼𑀓𑀫𑁆𑀳𑀺 𑀮𑁄𑀓𑀫𑁆𑀳𑀺,_  
_𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀼𑀓𑀸𑁇_  


(143.)

1461\. _𑀢𑁂 𑀙𑁂𑀢𑁆𑀯𑀸 𑀫𑀘𑁆𑀘𑀼𑀦𑁄 𑀚𑀸𑀮𑀁,_  
_𑀢𑀢𑀁 𑀫𑀸𑀬𑀸𑀯𑀺𑀦𑁄 𑀤𑀍𑀅𑀁;_  
_𑀙𑀺𑀦𑁆𑀦𑀸𑀮𑀬𑀢𑁆𑀢𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀓𑁄 𑀢𑁂𑀲𑀁 𑀕𑀢𑀺𑀫𑀸𑀧𑀬𑁂𑁇_  


(144.)

1462\. _𑀓𑀤𑀸𑀳𑀁 𑀫𑀺𑀣𑀺𑀮𑀁 𑀨𑀻𑀢𑀁,_  
_𑀯𑀺𑀪𑀢𑁆𑀢𑀁 𑀪𑀸𑀕𑀲𑁄 𑀫𑀺𑀢𑀁;_  
_𑀧𑀳𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(145.)

_𑀧𑀳𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_

1481\. _𑀓𑀤𑀸𑀳𑀁 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑁂 𑀘,_

1482\. _𑀓𑀤𑀸𑀳𑀁 𑀲𑁄𑀡𑁆𑀡𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂,_  
_𑀕𑁄𑀦𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑀦𑁆𑀣𑀢𑁂;_

(167.)

1485\. _𑀓𑀤𑀸𑀳𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀭𑀫𑁆𑀫𑀸,_

_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀓𑁂𑀳𑀺 𑀘;_  
_𑀧𑀳𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_

_𑀧𑀳𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_

(170.)

1488\. _𑀓𑀤𑀸𑀳𑀁 𑀅𑀲𑁆𑀲𑀕𑀼𑀫𑁆𑀩𑁂 𑀘,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑁂;_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑁂𑀯 𑀚𑀸𑀢𑀺𑀬𑀸,_  
_𑀲𑀺𑀦𑁆𑀥𑀯𑁂 𑀲𑀻𑀖𑀯𑀸𑀳𑀦𑁂𑁇_

_𑀘𑀸𑀧𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀯𑀫𑁆𑀫𑀺𑀪𑀺;_

_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(221.)

1539\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀫𑀺𑀕𑀭𑀣𑀸,_  
_𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀸 𑀉𑀲𑁆𑀲𑀺𑀢𑀤𑁆𑀥𑀚𑀸;_  
_𑀤𑀻𑀧𑀸 𑀅𑀣𑁄𑀧𑀺 𑀯𑁂𑀬𑁆𑀬𑀕𑁆𑀖𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸𑁇_  


(222.)

1540\. _𑀆𑀭𑀽𑀍𑀳𑀸 𑀕𑀸𑀫𑀡𑀻𑀬𑁂𑀳𑀺,_  
_𑀘𑀸𑀧𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀯𑀫𑁆𑀫𑀺𑀪𑀺;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(223.)

1541\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀦𑀻𑀮𑀯𑀫𑁆𑀫𑀥𑀭𑀸 𑀲𑀽𑀭𑀸,_  
_𑀢𑁄𑀫𑀭𑀗𑁆𑀓𑀼𑀲𑀧𑀸𑀡𑀺𑀦𑁄;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(224.)

1542\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀅𑀲𑁆𑀲𑀸𑀭𑁄𑀳𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀦𑀻𑀮𑀯𑀫𑁆𑀫𑀥𑀭𑀸 𑀲𑀽𑀭𑀸,_  
_𑀇𑀮𑁆𑀮𑀺𑀬𑀸𑀘𑀸𑀧𑀥𑀸𑀭𑀺𑀦𑁄;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(225.)

1543\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀭𑀣𑀸𑀭𑁄𑀳𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀦𑀻𑀮𑀯𑀫𑁆𑀫𑀥𑀭𑀸 𑀲𑀽𑀭𑀸,_  
_𑀘𑀸𑀧𑀳𑀢𑁆𑀣𑀸 𑀓𑀮𑀸𑀧𑀺𑀦𑁄;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(226.)

1544\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀥𑀦𑀼𑀕𑁆𑀕𑀳𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀦𑀻𑀮𑀯𑀫𑁆𑀫𑀥𑀭𑀸 𑀲𑀽𑀭𑀸,_  
_𑀘𑀸𑀧𑀳𑀢𑁆𑀣𑀸 𑀓𑀮𑀸𑀧𑀺𑀦𑁄;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(227.)

1545\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀘𑀺𑀢𑁆𑀭𑀯𑀫𑁆𑀫𑀥𑀭𑀸 𑀲𑀽𑀭𑀸,_  
_𑀓𑀜𑁆𑀘𑀦𑀸𑀯𑁂𑀍𑀅𑀥𑀸𑀭𑀺𑀦𑁄;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(228.)

1546\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀅𑀭𑀺𑀬𑀕𑀡𑀸,_  
_𑀯𑀢𑀯𑀦𑁆𑀢𑀸 𑀅𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀳𑀭𑀺𑀘𑀦𑁆𑀤𑀦𑀮𑀺𑀢𑁆𑀢𑀗𑁆𑀕𑀸,_  
_𑀓𑀸𑀲𑀺𑀓𑀼𑀢𑁆𑀢𑀫𑀥𑀸𑀭𑀺𑀦𑁄;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(229.)

1547\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀅𑀫𑀘𑁆𑀘𑀕𑀡𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀧𑀻𑀢𑀯𑀫𑁆𑀫𑀥𑀭𑀸 𑀲𑀽𑀭𑀸,_  
_𑀧𑀼𑀭𑀢𑁄 𑀕𑀘𑁆𑀙𑀫𑀸𑀮𑀺𑀦𑁄;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(230.)

1548\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(231.)

1549\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀸 𑀢𑀦𑀼𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(232.)

1550\. _𑀓𑀤𑀸𑀲𑁆𑀲𑀼 𑀫𑀁 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀅𑀲𑁆𑀲𑀯𑀸 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑀻;_  
_𑀬𑀦𑁆𑀢𑀁 𑀫𑀁 𑀦𑀸𑀦𑀼𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(233.)

1551\. _𑀓𑀤𑀸𑀳𑀁 𑀧𑀢𑁆𑀢𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀼𑀡𑁆𑀟𑁄 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀧𑀸𑀭𑀼𑀢𑁄;_  
_𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(234.)

1552\. _𑀓𑀤𑀸𑀳𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀁,_  
_𑀉𑀚𑁆𑀛𑀺𑀢𑀸𑀦𑀁 𑀫𑀳𑀸𑀧𑀣𑁂;_  
_𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀥𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(235.)

1553\. _𑀓𑀤𑀸𑀳𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑀫𑁆𑀫𑁂𑀖𑁂,_  
_𑀑𑀯𑀝𑁆𑀞𑁄 𑀅𑀮𑁆𑀮𑀘𑀻𑀯𑀭𑁄;_  
_𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(236.)

1554\. _𑀓𑀤𑀸𑀳𑀁 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸,_  
_𑀭𑀼𑀓𑁆𑀔𑀸 𑀭𑀼𑀓𑁆𑀔𑀁 𑀯𑀦𑀸 𑀯𑀦𑀁;_  
_𑀅𑀦𑀧𑁂𑀓𑁆𑀔𑁄 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(237.)

1555\. _𑀓𑀤𑀸𑀳𑀁 𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑁂𑀲𑀼,_  
_𑀧𑀳𑀻𑀦𑀪𑀬𑀪𑁂𑀭𑀯𑁄;_  
_𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(238.)

1556\. _𑀓𑀤𑀸𑀳𑀁 𑀯𑀻𑀡𑀁𑀯 𑀭𑀼𑀚𑁆𑀚𑀓𑁄,_  
_𑀲𑀢𑁆𑀢𑀢𑀦𑁆𑀢𑀺𑀁 𑀫𑀦𑁄𑀭𑀫𑀁;_  
_𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀚𑀼𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀁 𑀓𑀼𑀤𑀸𑀲𑁆𑀲𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(239.)

1557\. _𑀓𑀤𑀸𑀳𑀁 𑀭𑀣𑀓𑀸𑀭𑁄𑀯,_  
_𑀧𑀭𑀺𑀓𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀳𑀦𑀁;_  
_𑀓𑀸𑀫𑀲𑀁𑀬𑁄𑀚𑀦𑁂 𑀙𑁂𑀘𑁆𑀙𑀁,_  
_𑀬𑁂 𑀤𑀺𑀩𑁆𑀩𑁂 𑀬𑁂 𑀘 𑀫𑀸𑀦𑀼𑀲𑁂”𑁇_  


(240.)

1558\. _𑀢𑀸 𑀘 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_“𑀓𑀲𑁆𑀫𑀸 𑀦𑁄 𑀯𑀺𑀚𑀳𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(241.)

1559\. _𑀢𑀸 𑀘 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀸 𑀢𑀦𑀼𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_“𑀓𑀲𑁆𑀫𑀸 𑀦𑁄 𑀯𑀺𑀚𑀳𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(242.)

1560\. _𑀢𑀸 𑀘 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀅𑀲𑁆𑀲𑀯𑀸 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑀻;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_“𑀓𑀲𑁆𑀫𑀸 𑀦𑁄 𑀯𑀺𑀚𑀳𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(243.)

1561\. _𑀢𑀸 𑀘 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀤𑁆𑀤𑀯𑀻 𑀭𑀸𑀚𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(244.)

1562\. _𑀢𑀸 𑀘 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀸 𑀢𑀦𑀼𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀤𑁆𑀤𑀯𑀻 𑀭𑀸𑀚𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(245.)

1563\. _𑀢𑀸 𑀘 𑀲𑀢𑁆𑀢𑀲𑀢𑀸 𑀪𑀭𑀺𑀬𑀸,_  
_𑀅𑀲𑁆𑀲𑀯𑀸 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑀻;_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀤𑁆𑀤𑀯𑀻 𑀭𑀸𑀚𑀸,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬 𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄𑁇_  


(246.)

1564\. _𑀳𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑀧𑀮𑀁 𑀓𑀁𑀲𑀁,_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑀁 𑀲𑀢𑀭𑀸𑀚𑀺𑀓𑀁;_  
_𑀅𑀕𑁆𑀕𑀳𑀻 𑀫𑀢𑁆𑀢𑀺𑀓𑀁 𑀧𑀢𑁆𑀢𑀁,_  
_𑀢𑀁 𑀤𑀼𑀢𑀺𑀬𑀸𑀪𑀺𑀲𑁂𑀘𑀦𑀁𑁇_  


(247.)

1565\. _“𑀪𑁂𑀲𑁆𑀫𑀸 𑀅𑀕𑁆𑀕𑀺𑀲𑀫𑀸 𑀚𑀸𑀮𑀸,_  
_𑀓𑁄𑀲𑀸 𑀟𑀬𑁆𑀳𑀦𑁆𑀢𑀺 𑀪𑀸𑀕𑀲𑁄;_  
_𑀭𑀚𑀢𑀁 𑀚𑀸𑀢𑀭𑀽𑀧𑀜𑁆𑀘,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀩𑀳𑀽𑁇_  


(248.)

1566\. _𑀫𑀡𑀬𑁄 𑀲𑀗𑁆𑀔𑀫𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀯𑀢𑁆𑀣𑀺𑀓𑀁 𑀳𑀭𑀺𑀘𑀦𑁆𑀤𑀦𑀁;_  
_𑀅𑀚𑀺𑀦𑀁 𑀤𑀦𑁆𑀢𑀪𑀡𑁆𑀟𑀜𑁆𑀘,_  
_𑀮𑁄𑀳𑀁 𑀓𑀸𑀍𑀆𑀬𑀲𑀁 𑀩𑀳𑀽;_  
_𑀏𑀳𑀺 𑀭𑀸𑀚 𑀦𑀺𑀯𑀢𑁆𑀢𑀲𑁆𑀲𑀼,_  
_𑀫𑀸 𑀢𑁂𑀢𑀁 𑀯𑀺𑀦𑀲𑀸 𑀥𑀦𑀁”𑁇_  


(249.)

1567\. _“𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀚𑀻𑀯𑀸𑀫,_  
_𑀬𑁂𑀲𑀁 𑀦𑁄 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀁;_  
_𑀫𑀺𑀣𑀺𑀮𑀸 𑀤𑀬𑁆𑀳𑀫𑀸𑀦𑀸𑀬,_  
_𑀦 𑀫𑁂 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀤𑀬𑁆𑀳𑀣”𑁇_  


(250.)

1568\. _“𑀅𑀝𑀯𑀺𑀬𑁄 𑀲𑀫𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀭𑀝𑁆𑀞𑀁 𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑀬𑀦𑁆𑀢𑀺 𑀢𑀁;_  
_𑀏𑀳𑀺 𑀭𑀸𑀚 𑀦𑀺𑀯𑀢𑁆𑀢𑀲𑁆𑀲𑀼,_  
_𑀫𑀸 𑀭𑀝𑁆𑀞𑀁 𑀯𑀺𑀦𑀲𑀸 𑀇𑀤𑀁”𑁇_  


(251.)

1569\. _“𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀚𑀻𑀯𑀸𑀫,_  
_𑀬𑁂𑀲𑀁 𑀦𑁄 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀁;_  
_𑀭𑀝𑁆𑀞𑁂 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀫𑀸𑀦𑀫𑁆𑀳𑀺,_  
_𑀦 𑀫𑁂 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀳𑀻𑀭𑀣𑁇_  


(252.)

1570\. _𑀲𑀼𑀲𑀼𑀔𑀁 𑀯𑀢 𑀚𑀻𑀯𑀸𑀫,_  
_𑀬𑁂𑀲𑀁 𑀦𑁄 𑀦𑀢𑁆𑀣𑀺 𑀓𑀺𑀜𑁆𑀘𑀦𑀁;_  
_𑀧𑀻𑀢𑀺𑀪𑀓𑁆𑀔𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀤𑁂𑀯𑀸 𑀆𑀪𑀲𑁆𑀲𑀭𑀸 𑀬𑀣𑀸”𑁇_  


(253.)

1571\. _“𑀓𑀺𑀫𑁆𑀳𑁂𑀲𑁄 𑀫𑀳𑀢𑁄 𑀖𑁄𑀲𑁄,_  
_𑀓𑀸 𑀦𑀼 𑀕𑀸𑀫𑁂𑀯 𑀓𑀻𑀍𑀇𑀬𑀸;_  
_𑀲𑀫𑀡 𑀢𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀸𑀫,_  
_𑀓𑀢𑁆𑀣𑁂𑀲𑁄 𑀅𑀪𑀺𑀲𑀝𑁄 𑀚𑀦𑁄”𑁇_  


(254.)

1572\. _“𑀫𑀫𑀁 𑀑𑀳𑀸𑀬 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁,_  
_𑀏𑀢𑁆𑀣𑁂𑀲𑁄 𑀅𑀪𑀺𑀲𑀝𑁄 𑀚𑀦𑁄;_  
_𑀲𑀻𑀫𑀸𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑀁 𑀬𑀦𑁆𑀢𑀁,_  
_𑀫𑀼𑀦𑀺𑀫𑁄𑀦𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀫𑀺𑀲𑁆𑀲𑀁 𑀦𑀦𑁆𑀤𑀻𑀳𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁,_  
_𑀓𑀺𑀁 𑀚𑀸𑀦𑀫𑀦𑀼𑀧𑀼𑀘𑁆𑀙𑀲𑀺”𑁇_  


(255.)

1573\. _“𑀫𑀸𑀲𑁆𑀲𑀼 𑀢𑀺𑀡𑁆𑀡𑁄 𑀅𑀫𑀜𑁆𑀜𑀺𑀢𑁆𑀣,_  
_𑀲𑀭𑀻𑀭𑀁 𑀥𑀸𑀭𑀬𑀁 𑀇𑀫𑀁;_  
_𑀅𑀢𑀻𑀭𑀡𑁂𑀬𑁆𑀬 𑀬𑀫𑀺𑀤𑀁,_  
_𑀩𑀳𑀽 𑀳𑀺 𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑀬𑁄”𑁇_  


(256.)

1574\. _“𑀓𑁄 𑀦𑀼 𑀫𑁂 𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑀲𑁆𑀲,_  
_𑀫𑀫𑀁 𑀏𑀯𑀁𑀯𑀺𑀳𑀸𑀭𑀺𑀦𑁄;_  
_𑀬𑁄 𑀦𑁂𑀯 𑀤𑀺𑀝𑁆𑀞𑁂 𑀦𑀸𑀤𑀺𑀝𑁆𑀞𑁂,_  
_𑀓𑀸𑀫𑀸𑀦𑀫𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑁂”𑁇_  


(257.)

1575\. _“𑀦𑀺𑀤𑁆𑀤𑀸 𑀢𑀦𑁆𑀤𑀻 𑀯𑀺𑀚𑀫𑁆𑀪𑀺𑀢𑀸,_  
_𑀅𑀭𑀢𑀻 𑀪𑀢𑁆𑀢𑀲𑀫𑁆𑀫𑀤𑁄;_  
_𑀆𑀯𑀲𑀦𑁆𑀢𑀺 𑀲𑀭𑀻𑀭𑀝𑁆𑀞𑀸,_  
_𑀩𑀳𑀽 𑀳𑀺 𑀧𑀭𑀺𑀧𑀦𑁆𑀣𑀬𑁄”𑁇_  


(258.)

1576\. _“𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀯𑀢 𑀫𑀁 𑀪𑀯𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀺;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡 𑀢𑁂𑀯 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀓𑁄 𑀦𑀼 𑀢𑁆𑀯𑀫𑀲𑀺 𑀫𑀸𑀭𑀺𑀲”𑁇_  


(259.)

1577\. _“𑀦𑀸𑀭𑀤𑁄 𑀇𑀢𑀺 𑀫𑁂 𑀦𑀸𑀫𑀁,_  
_𑀓𑀲𑁆𑀲𑀧𑁄 𑀇𑀢𑀺 𑀫𑀁 𑀯𑀺𑀤𑀽;_  
_𑀪𑁄𑀢𑁄 𑀲𑀓𑀸𑀲𑀫𑀸𑀕𑀘𑁆𑀙𑀺𑀁,_  
_𑀲𑀸𑀥𑀼 𑀲𑀩𑁆𑀪𑀺 𑀲𑀫𑀸𑀕𑀫𑁄𑁇_  


(260.)

1578\. _𑀢𑀲𑁆𑀲 𑀢𑁂 𑀲𑀩𑁆𑀩𑁄 𑀆𑀦𑀦𑁆𑀤𑁄,_  
_𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑀯𑀢𑁆𑀢𑀢𑀼;_  
_𑀬𑀁 𑀊𑀦𑀁 𑀢𑀁 𑀧𑀭𑀺𑀧𑀽𑀭𑁂𑀳𑀺,_  
_𑀔𑀦𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑀲𑀫𑁂𑀦 𑀘𑁇_  


(261.)

1579\. _𑀧𑀲𑀸𑀭𑀬 𑀲𑀦𑁆𑀦𑀢𑀜𑁆𑀘,_  
_𑀉𑀦𑁆𑀦𑀢𑀜𑁆𑀘 𑀧𑀲𑀸𑀭𑀬;_  
_𑀓𑀫𑁆𑀫𑀁 𑀯𑀺𑀚𑁆𑀚𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀜𑁆𑀘,_  
_𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀚𑁇_  


(262.)

1580\. _𑀩𑀳𑀽 𑀳𑀢𑁆𑀣𑀻 𑀘 𑀅𑀲𑁆𑀲𑁂 𑀘,_  
_𑀦𑀕𑀭𑁂 𑀚𑀦𑀧𑀤𑀸𑀦𑀺 𑀘;_  
_𑀳𑀺𑀢𑁆𑀯𑀸 𑀚𑀦𑀓 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀓𑀧𑀸𑀮𑁂 𑀭𑀢𑀺𑀫𑀚𑁆𑀛𑀕𑀸𑁇_  


(263.)

1581\. _𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀢𑁂 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑀸 𑀘 𑀜𑀸𑀢𑀓𑀸;_  
_𑀤𑀼𑀩𑁆𑀪𑀺𑀫𑀓𑀁𑀲𑀼 𑀚𑀦𑀓,_  
_𑀓𑀲𑁆𑀫𑀸 𑀢𑁂𑀢𑀁 𑀅𑀭𑀼𑀘𑁆𑀘𑀣”𑁇_  


(264.)

1582\. _“𑀦 𑀫𑀺𑀕𑀸𑀚𑀺𑀦 𑀚𑀸𑀢𑀼𑀘𑁆𑀙𑁂,_  
_𑀅𑀳𑀁 𑀓𑀜𑁆𑀘𑀺 𑀓𑀼𑀤𑀸𑀘𑀦𑀁;_  
_𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀚𑀺𑀦𑁂 𑀜𑀸𑀢𑀺𑀁,_  
_𑀦 𑀘𑀸𑀧𑀺 𑀜𑀸𑀢𑀬𑁄 𑀫𑀫𑀁𑁇_  


(265.)

1583\. _𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀮𑁄𑀓𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀔𑀚𑁆𑀚𑀦𑁆𑀢𑀁 𑀓𑀤𑁆𑀤𑀫𑀻𑀓𑀢𑀁;_  
_𑀳𑀜𑁆𑀜𑀭𑁂 𑀩𑀚𑁆𑀛𑀭𑁂 𑀘𑁂𑀢𑁆𑀣,_  
_𑀬𑀢𑁆𑀣 𑀲𑀦𑁆𑀦𑁄 𑀧𑀼𑀣𑀼𑀚𑁆𑀚𑀦𑁄;_  
_𑀏𑀢𑀸𑀳𑀁 𑀉𑀧𑀫𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀪𑀺𑀓𑁆𑀔𑀓𑁄𑀲𑁆𑀫𑀺 𑀫𑀺𑀕𑀸𑀚𑀺𑀦”𑁇_  


(266.)

1584\. _“𑀓𑁄 𑀦𑀼 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸,_  
_𑀓𑀲𑁆𑀲𑁂𑀢𑀁 𑀯𑀘𑀦𑀁 𑀲𑀼𑀘𑀺;_  
_𑀦 𑀳𑀺 𑀓𑀧𑁆𑀧𑀁 𑀯𑀸 𑀯𑀺𑀚𑁆𑀚𑀁 𑀯𑀸,_  
_𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀬 𑀭𑀣𑁂𑀲𑀪;_  
_𑀲𑀫𑀡𑀁 𑀆𑀳𑀼 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀬𑀣𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲𑀢𑀺𑀓𑁆𑀓𑀫𑁄𑁇_  


(267.)

1585\. _𑀦 𑀫𑀺𑀕𑀸𑀚𑀺𑀦 𑀚𑀸𑀢𑀼𑀘𑁆𑀙𑁂,_  
_𑀅𑀳𑀁 𑀓𑀜𑁆𑀘𑀺 𑀓𑀼𑀤𑀸𑀘𑀦𑀁;_  
_𑀲𑀫𑀡𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀯𑀸𑀧𑀺,_  
_𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀧𑀸𑀯𑀺𑀲𑀺𑀁”𑁇_  


(268.)

1586\. _“𑀫𑀳𑀢𑀸 𑀘𑀸𑀦𑀼𑀪𑀸𑀯𑁂𑀦,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀲𑀺𑀭𑀺𑀬𑀸 𑀚𑀮𑀁;_  
_𑀕𑀻𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀕𑀻𑀢𑁂𑀲𑀼,_  
_𑀯𑀚𑁆𑀚𑀫𑀸𑀦𑁂𑀲𑀼 𑀯𑀕𑁆𑀕𑀼𑀲𑀼𑁇_  


(269.)

1587\. _𑀢𑀽𑀭𑀺𑀬𑀢𑀸𑀍𑀅𑀲𑀗𑁆𑀖𑀼𑀝𑁆𑀞𑁂,_  
_𑀲𑀫𑁆𑀫𑀢𑀸𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑁂;_  
_𑀲 𑀫𑀺𑀕𑀸𑀚𑀺𑀦 𑀫𑀤𑁆𑀤𑀓𑁆𑀔𑀺𑀁,_  
_𑀨𑀮𑀺𑀁 𑀅𑀫𑁆𑀩𑀁 𑀢𑀺𑀭𑁄𑀘𑁆𑀙𑀤𑀁;_  
_𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺,_  
_𑀨𑀮𑀓𑀸𑀫𑁂𑀳𑀺 𑀚𑀦𑁆𑀢𑀼𑀪𑀺𑁇_  


(270.)

1588\. _𑀲𑁄 𑀔𑁄𑀳𑀁 𑀢𑀁 𑀲𑀺𑀭𑀺𑀁 𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀺𑀕𑀸𑀚𑀺𑀦;_  
_𑀫𑀽𑀮𑀁 𑀅𑀫𑁆𑀩𑀲𑁆𑀲𑀼𑀧𑀸𑀕𑀘𑁆𑀙𑀺𑀁,_  
_𑀨𑀮𑀺𑀦𑁄 𑀦𑀺𑀧𑁆𑀨𑀮𑀲𑁆𑀲 𑀘𑁇_  


(271.)

1589\. _𑀨𑀮𑀺𑀁 𑀅𑀫𑁆𑀩𑀁 𑀳𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀯𑀺𑀤𑁆𑀥𑀁𑀲𑁆𑀢𑀁 𑀯𑀺𑀦𑀍𑀈𑀓𑀢𑀁;_  
_𑀅𑀣𑁂𑀓𑀁 𑀇𑀢𑀭𑀁 𑀅𑀫𑁆𑀩𑀁,_  
_𑀦𑀻𑀮𑁄𑀪𑀸𑀲𑀁 𑀫𑀦𑁄𑀭𑀫𑀁𑁇_  


(272.)

1590\. _𑀏𑀯𑀫𑁂𑀯 𑀦𑀽𑀦𑀫𑁆𑀳𑁂𑀧𑀺,_  
_𑀇𑀲𑁆𑀲𑀭𑁂 𑀩𑀳𑀼𑀓𑀡𑁆𑀝𑀓𑁂;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀦𑁄 𑀯𑀥𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀬𑀣𑀸 𑀅𑀫𑁆𑀩𑁄 𑀨𑀮𑀻 𑀳𑀢𑁄𑁇_  


(273.)

1591\. _𑀅𑀚𑀺𑀦𑀫𑁆𑀳𑀺 𑀳𑀜𑁆𑀜𑀢𑁂 𑀤𑀻𑀧𑀺,_  
_𑀦𑀸𑀕𑁄 𑀤𑀦𑁆𑀢𑁂𑀳𑀺 𑀳𑀜𑁆𑀜𑀢𑁂;_  
_𑀥𑀦𑀫𑁆𑀳𑀺 𑀥𑀦𑀺𑀦𑁄 𑀳𑀦𑁆𑀢𑀺,_  
_𑀅𑀦𑀺𑀓𑁂𑀢𑀫𑀲𑀦𑁆𑀣𑀯𑀁;_  
_𑀨𑀮𑀻 𑀅𑀫𑁆𑀩𑁄 𑀅𑀨𑀮𑁄 𑀘,_  
_𑀢𑁂 𑀲𑀢𑁆𑀣𑀸𑀭𑁄 𑀉𑀪𑁄 𑀫𑀫”𑁇_  


(274.)

1592\. _“𑀲𑀩𑁆𑀩𑁄 𑀚𑀦𑁄 𑀧𑀩𑁆𑀬𑀣𑀺𑀢𑁄,_  
_𑀭𑀸𑀚𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀇𑀢𑀺;_  
_𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸𑁇_  


(275.)

1593\. _𑀅𑀲𑁆𑀲𑀸𑀲𑀬𑀺𑀢𑁆𑀯𑀸 𑀚𑀦𑀢𑀁,_  
_𑀞𑀧𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀘𑁆𑀙𑀤𑀁;_  
_𑀧𑀼𑀢𑁆𑀢𑀁 𑀭𑀚𑁆𑀚𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀣 𑀧𑀘𑁆𑀙𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(276.)

1594\. _“𑀘𑀢𑁆𑀢𑀸 𑀫𑀬𑀸 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀫𑀺𑀢𑁆𑀢𑀸𑀫𑀘𑁆𑀘𑀸 𑀘 𑀜𑀸𑀢𑀓𑀸;_  
_𑀲𑀦𑁆𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀤𑁂𑀳𑀸𑀦𑀁,_  
_𑀤𑀻𑀖𑀸𑀯𑀼 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁄;_  
_𑀢𑁂 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀫𑀺𑀣𑀺𑀮𑀸𑀬𑀁 𑀧𑀚𑀸𑀧𑀢𑀺𑁇_  


(277.)

1595\. _𑀏𑀳𑀺 𑀢𑀁 𑀅𑀦𑀼𑀲𑀺𑀓𑁆𑀔𑀸𑀫𑀺,_  
_𑀬𑀁 𑀯𑀸𑀓𑁆𑀬𑀁 𑀫𑀫 𑀭𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀭𑀚𑁆𑀚𑀁 𑀢𑀼𑀯𑀁 𑀓𑀸𑀭𑀬𑀲𑀺,_  
_𑀧𑀸𑀧𑀁 𑀤𑀼𑀘𑁆𑀘𑀭𑀺𑀢𑀁 𑀩𑀳𑀼𑀁;_  
_𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸 𑀫𑀦𑀲𑀸,_  
_𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀲𑀺 𑀤𑀼𑀕𑁆𑀕𑀢𑀺𑀁𑁇_  


(278.)

1596\. _𑀧𑀭𑀤𑀺𑀦𑁆𑀦𑀓𑁂𑀦 𑀧𑀭𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁂𑀦,_  
_𑀧𑀺𑀡𑁆𑀟𑁂𑀦 𑀬𑀸𑀧𑁂𑀳𑀺 𑀲 𑀥𑀻𑀭𑀥𑀫𑁆𑀫𑁄”𑁇_  


(279.)

1597\. _“𑀬𑁄𑀧𑀺 𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀪𑀢𑁆𑀢𑀓𑀸𑀮𑁂 𑀦 𑀪𑀼𑀜𑁆𑀚𑁂,_  
_𑀅𑀚𑀼𑀝𑁆𑀞𑀫𑀸𑀭𑀻𑀯 𑀔𑀼𑀤𑀸𑀬 𑀫𑀺𑀬𑁆𑀬𑁂;_  
_𑀦 𑀢𑁆𑀯𑁂𑀯 𑀧𑀺𑀡𑁆𑀟𑀁 𑀮𑀼𑀍𑀇𑀢𑀁 𑀅𑀦𑀭𑀺𑀬𑀁,_  
_𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀭𑀽𑀧𑁄 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁄 𑀦 𑀲𑁂𑀯𑁂;_  
_𑀢𑀬𑀺𑀤𑀁 𑀦 𑀲𑀸𑀥𑀼 𑀢𑀬𑀺𑀤𑀁 𑀦 𑀲𑀼𑀝𑁆𑀞𑀼,_  
_𑀲𑀼𑀦𑀔𑀼𑀘𑁆𑀙𑀺𑀝𑁆𑀞𑀓𑀁 𑀚𑀦𑀓 𑀪𑀼𑀜𑁆𑀚𑀲𑁂 𑀢𑀼𑀯𑀁”𑁇_  


(280.)

1598\. _“𑀦 𑀘𑀸𑀧𑀺 𑀫𑁂 𑀲𑀻𑀯𑀮𑀺 𑀲𑁄 𑀅𑀪𑀓𑁆𑀔𑁄,_  
_𑀬𑀁 𑀳𑁄𑀢𑀺 𑀘𑀢𑁆𑀢𑀁 𑀕𑀺𑀳𑀺𑀦𑁄 𑀲𑀼𑀦𑀲𑁆𑀲 𑀯𑀸;_  
_𑀬𑁂 𑀓𑁂𑀘𑀺 𑀪𑁄𑀕𑀸 𑀇𑀥 𑀥𑀫𑁆𑀫𑀮𑀤𑁆𑀥𑀸,_  
_𑀲𑀩𑁆𑀩𑁄 𑀲𑁄 𑀪𑀓𑁆𑀔𑁄 𑀅𑀦𑀯𑀬𑁄𑀢𑀺 𑀯𑀼𑀢𑁆𑀢𑁄”𑁇_  


(281.)

1599\. _“𑀓𑀼𑀫𑀸𑀭𑀺𑀓𑁂 𑀉𑀧𑀲𑁂𑀦𑀺𑀬𑁂,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀦𑀺𑀕𑁆𑀕𑀍𑀅𑀫𑀡𑁆𑀟𑀺𑀢𑁂;_  
_𑀓𑀲𑁆𑀫𑀸 𑀢𑁂 𑀏𑀓𑁄 𑀪𑀼𑀚𑁄 𑀚𑀦𑀢𑀺,_  
_𑀏𑀓𑁄 𑀢𑁂 𑀦 𑀚𑀦𑀢𑀻 𑀪𑀼𑀚𑁄”𑁇_  


(282.)

1600\. _“𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀫𑁂 𑀲𑀫𑀡 𑀳𑀢𑁆𑀣𑁂,_  
_𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑀸 𑀤𑀼𑀦𑀻𑀯𑀭𑀸;_  
_𑀲𑀗𑁆𑀖𑀸𑀢𑀸 𑀚𑀸𑀬𑀢𑁂 𑀲𑀤𑁆𑀤𑁄,_  
_𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲𑁂𑀯 𑀲𑀸 𑀕𑀢𑀺𑁇_  


(283.)

1601\. _𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀫𑁂 𑀲𑀫𑀡 𑀳𑀢𑁆𑀣𑁂,_  
_𑀧𑀝𑀺𑀫𑀼𑀓𑁆𑀓𑁄 𑀏𑀓𑀦𑀻𑀯𑀭𑁄;_  
_𑀲𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀦 𑀚𑀦𑀢𑀺,_  
_𑀫𑀼𑀦𑀺𑀪𑀽𑀢𑁄𑀯 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


(284.)

1602\. _𑀯𑀺𑀯𑀸𑀤𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀤𑀼𑀢𑀺𑀬𑁄,_  
_𑀓𑁂𑀦𑁂𑀓𑁄 𑀯𑀺𑀯𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀲𑁆𑀲 𑀢𑁂 𑀲𑀕𑁆𑀕𑀓𑀸𑀫𑀲𑁆𑀲,_  
_𑀏𑀓𑀢𑁆𑀢𑀫𑀼𑀧𑀭𑁄𑀘𑀢𑀁”𑁇_  


(285.)

1603\. _“𑀲𑀼𑀡𑀸𑀲𑀺 𑀲𑀻𑀯𑀮𑀺 𑀓𑀣𑀸,_  
_𑀓𑀼𑀫𑀸𑀭𑀺𑀬𑀸 𑀧𑀯𑁂𑀤𑀺𑀢𑀸;_  
_𑀧𑁂𑀲𑀺𑀬𑀸 𑀫𑀁 𑀕𑀭𑀳𑀺𑀢𑁆𑀣𑁄,_  
_𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲𑁂𑀯 𑀲𑀸 𑀕𑀢𑀺𑁇_  


(286.)

1604\. _𑀅𑀬𑀁 𑀤𑁆𑀯𑁂𑀥𑀸𑀧𑀣𑁄 𑀪𑀤𑁆𑀤𑁂,_  
_𑀅𑀦𑀼𑀘𑀺𑀡𑁆𑀡𑁄 𑀧𑀣𑀸𑀯𑀺𑀳𑀺;_  
_𑀢𑁂𑀲𑀁 𑀢𑁆𑀯𑀁 𑀏𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺,_  
_𑀅𑀳𑀫𑁂𑀓𑀁 𑀧𑀼𑀦𑀸𑀧𑀭𑀁𑁇_  


(287.)

1605\. _𑀫𑀸𑀯𑀘 𑀫𑀁 𑀢𑁆𑀯𑀁 𑀧𑀢𑀺 𑀫𑁂𑀢𑀺,_  
_𑀦𑀸𑀳𑀁 𑀪𑀭𑀺𑀬𑀸𑀢𑀺 𑀯𑀸 𑀧𑀼𑀦”;_  
_𑀇𑀫𑀫𑁂𑀯 𑀓𑀣𑀬𑀦𑁆𑀢𑀸,_  
_𑀣𑀽𑀡𑀁 𑀦𑀕𑀭𑀼𑀧𑀸𑀕𑀫𑀼𑀁𑁇_  


(288.)

1606\. _𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀉𑀲𑀼𑀓𑀸𑀭𑀲𑁆𑀲,_  
_𑀪𑀢𑁆𑀢𑀓𑀸𑀮𑁂 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑁂;_  
_𑀢𑀢𑁆𑀭𑀸 𑀘 𑀲𑁄 𑀉𑀲𑀼𑀓𑀸𑀭𑁄,_  
_𑀏𑀓𑀁 𑀤𑀡𑁆𑀟𑀁 𑀉𑀚𑀼𑀁 𑀓𑀢𑀁;_  
_𑀏𑀓𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀼𑀁 𑀦𑀺𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀚𑀺𑀫𑁆𑀳𑀫𑁂𑀓𑁂𑀦 𑀧𑁂𑀓𑁆𑀔𑀢𑀺𑁇_  


(289.)

1607\. _“𑀏𑀯𑀁 𑀦𑁄 𑀲𑀸𑀥𑀼 𑀧𑀲𑁆𑀲𑀲𑀺,_  
_𑀉𑀲𑀼𑀓𑀸𑀭 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂;_  
_𑀬𑀤𑁂𑀓𑀁 𑀘𑀓𑁆𑀔𑀼𑀁 𑀦𑀺𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀚𑀺𑀫𑁆𑀳𑀫𑁂𑀓𑁂𑀦 𑀧𑁂𑀓𑁆𑀔𑀲𑀺”𑁇_  


(290.)

1608\. _“𑀤𑁆𑀯𑀻𑀳𑀺 𑀲𑀫𑀡 𑀘𑀓𑁆𑀔𑀽𑀳𑀺,_  
_𑀯𑀺𑀲𑀸𑀮𑀁 𑀯𑀺𑀬 𑀔𑀸𑀬𑀢𑀺;_  
_𑀅𑀲𑀫𑁆𑀧𑀢𑁆𑀯𑀸 𑀧𑀭𑀫𑀁 𑀮𑀺𑀗𑁆𑀕𑀁,_  
_𑀦𑀼𑀚𑀼𑀪𑀸𑀯𑀸𑀬 𑀓𑀧𑁆𑀧𑀢𑀺𑁇_  


(291.)

1609\. _𑀏𑀓𑀜𑁆𑀘 𑀘𑀓𑁆𑀔𑀼𑀁 𑀦𑀺𑀕𑁆𑀕𑀬𑁆𑀳,_  
_𑀚𑀺𑀫𑁆𑀳𑀫𑁂𑀓𑁂𑀦 𑀧𑁂𑀓𑁆𑀔𑀢𑁄;_  
_𑀲𑀫𑁆𑀧𑀢𑁆𑀯𑀸 𑀧𑀭𑀫𑀁 𑀮𑀺𑀗𑁆𑀕𑀁,_  
_𑀉𑀚𑀼𑀪𑀸𑀯𑀸𑀬 𑀓𑀧𑁆𑀧𑀢𑀺𑁇_  


(292.)

1610\. _𑀯𑀺𑀯𑀸𑀤𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀤𑀼𑀢𑀺𑀬𑁄,_  
_𑀓𑁂𑀦𑁂𑀓𑁄 𑀯𑀺𑀯𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀢𑀲𑁆𑀲 𑀢𑁂 𑀲𑀕𑁆𑀕𑀓𑀸𑀫𑀲𑁆𑀲,_  
_𑀏𑀓𑀢𑁆𑀢𑀫𑀼𑀧𑀭𑁄𑀘𑀢𑀁”𑁇_  


(293.)

1611\. _“𑀲𑀼𑀡𑀸𑀲𑀺 𑀲𑀻𑀯𑀮𑀺 𑀓𑀣𑀸,_  
_𑀉𑀲𑀼𑀓𑀸𑀭𑁂𑀦 𑀯𑁂𑀤𑀺𑀢𑀸;_  
_𑀧𑁂𑀲𑀺𑀬𑀸 𑀫𑀁 𑀕𑀭𑀳𑀺𑀢𑁆𑀣𑁄,_  
_𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲𑁂𑀯 𑀲𑀸 𑀕𑀢𑀺𑁇_  


(294.)

1612\. _𑀅𑀬𑀁 𑀤𑁆𑀯𑁂𑀥𑀸𑀧𑀣𑁄 𑀪𑀤𑁆𑀤𑁂,_  
_𑀅𑀦𑀼𑀘𑀺𑀡𑁆𑀡𑁄 𑀧𑀣𑀸𑀯𑀺𑀳𑀺;_  
_𑀢𑁂𑀲𑀁 𑀢𑁆𑀯𑀁 𑀏𑀓𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺,_  
_𑀅𑀳𑀫𑁂𑀓𑀁 𑀧𑀼𑀦𑀸𑀧𑀭𑀁𑁇_  


(295.)

1613\. _𑀫𑀸𑀯𑀘 𑀫𑀁 𑀢𑁆𑀯𑀁 𑀧𑀢𑀺 𑀫𑁂𑀢𑀺,_  
_𑀦𑀸𑀳𑀁 𑀪𑀭𑀺𑀬𑀸𑀢𑀺 𑀯𑀸 𑀧𑀼𑀦;_  
_𑀫𑀼𑀜𑁆𑀚𑀸𑀯𑁂𑀲𑀺𑀓𑀸 𑀧𑀯𑀸𑀍𑀳𑀸,_  
_𑀏𑀓𑀸 𑀯𑀺𑀳𑀭 𑀲𑀻𑀯𑀮𑀻”𑀢𑀺𑁇_  


---

1614\. 𑀫𑀳𑀸𑀚𑀦𑀓𑀚𑀸𑀢𑀓𑀁 𑀤𑀼𑀢𑀺𑀬𑀁𑁇
