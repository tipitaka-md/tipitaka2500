# 𑁨𑁨𑁇𑁧𑁇𑁯𑁇𑁬 𑀭𑀸𑀚𑀯𑀲𑀢𑀺

### 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀯𑀺𑀥𑀼𑀭𑀚𑀸𑀢𑀓

### 𑀭𑀸𑀚𑀯𑀲𑀢𑀺

(1471.)

2810\. _𑀲𑁄 𑀘 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀫𑀘𑁆𑀘𑁂 𑀘,_  
_𑀜𑀸𑀢𑀬𑁄 𑀲𑀼𑀳𑀤𑀚𑁆𑀚𑀦𑁂;_  
_𑀅𑀮𑀻𑀦𑀫𑀦𑀲𑀗𑁆𑀓𑀧𑁆𑀧𑁄,_  
_𑀯𑀺𑀥𑀼𑀭𑁄 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀺𑁇_  


(1472.)

2811\. _“𑀏𑀣𑀬𑁆𑀬𑁄 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁,_  
_𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀡𑀸𑀣 𑀫𑁂;_  
_𑀬𑀣𑀸 𑀭𑀸𑀚𑀓𑀼𑀮𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀬𑀲𑀁 𑀧𑁄𑀲𑁄 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(1473.)

2812\. _𑀦 𑀳𑀺 𑀭𑀸𑀚𑀓𑀼𑀮𑀁 𑀧𑀢𑁆𑀢𑁄,_  
_𑀅𑀜𑁆𑀜𑀸𑀢𑁄 𑀮𑀪𑀢𑁂 𑀬𑀲𑀁;_  
_𑀦𑀸𑀲𑀽𑀭𑁄 𑀦𑀸𑀧𑀺 𑀤𑀼𑀫𑁆𑀫𑁂𑀥𑁄,_  
_𑀦𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀓𑀼𑀤𑀸𑀘𑀦𑀁𑁇_  


(1474.)

2813\. _𑀬𑀤𑀸𑀲𑁆𑀲 𑀲𑀻𑀮𑀁 𑀧𑀜𑁆𑀜𑀜𑁆𑀘,_  
_𑀲𑁄𑀘𑁂𑀬𑁆𑀬𑀁 𑀘𑀸𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀅𑀣 𑀯𑀺𑀲𑁆𑀲𑀲𑀢𑁂 𑀢𑁆𑀬𑀫𑁆𑀳𑀺,_  
_𑀕𑀼𑀬𑁆𑀳𑀜𑁆𑀘𑀲𑁆𑀲 𑀦 𑀭𑀓𑁆𑀔𑀢𑀺𑁇_  


(1475.)

2814\. _𑀢𑀼𑀮𑀸 𑀬𑀣𑀸 𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀸,_  
_𑀲𑀫𑀤𑀡𑁆𑀟𑀸 𑀲𑀼𑀥𑀸𑀭𑀺𑀢𑀸;_  
_𑀅𑀚𑁆𑀛𑀺𑀝𑁆𑀞𑁄 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1476.)

2815\. _𑀢𑀼𑀮𑀸 𑀬𑀣𑀸 𑀧𑀕𑁆𑀕𑀳𑀺𑀢𑀸,_  
_𑀲𑀫𑀤𑀡𑁆𑀟𑀸 𑀲𑀼𑀥𑀸𑀭𑀺𑀢𑀸;_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀪𑁄𑀦𑁆𑀢𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1477.)

2816\. _𑀤𑀺𑀯𑀸 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀭𑀸𑀚𑀓𑀺𑀘𑁆𑀘𑁂𑀲𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀅𑀚𑁆𑀛𑀺𑀝𑁆𑀞𑁄 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1478.)

2817\. _𑀤𑀺𑀯𑀸 𑀯𑀸 𑀬𑀤𑀺 𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀭𑀸𑀚𑀓𑀺𑀘𑁆𑀘𑁂𑀲𑀼 𑀧𑀡𑁆𑀟𑀺𑀢𑁄;_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀪𑁄𑀦𑁆𑀢𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1479.)

2818\. _𑀬𑁄 𑀘𑀲𑁆𑀲 𑀲𑀼𑀓𑀢𑁄 𑀫𑀕𑁆𑀕𑁄,_  
_𑀭𑀜𑁆𑀜𑁄 𑀲𑀼𑀧𑁆𑀧𑀝𑀺𑀬𑀸𑀤𑀺𑀢𑁄;_  
_𑀦 𑀢𑁂𑀦 𑀯𑀼𑀢𑁆𑀢𑁄 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1480.)

2819\. _𑀦 𑀭𑀜𑁆𑀜𑁄 𑀲𑀤𑀺𑀲𑀁 𑀪𑀼𑀜𑁆𑀚𑁂,_  
_𑀓𑀸𑀫𑀪𑁄𑀕𑁂 𑀓𑀼𑀤𑀸𑀘𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀘𑁆𑀙𑀢𑁄 𑀕𑀘𑁆𑀙𑁂,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1481.)

2820\. _𑀦 𑀭𑀜𑁆𑀜𑁄 𑀲𑀤𑀺𑀲𑀁 𑀯𑀢𑁆𑀣𑀁,_  
_𑀦 𑀫𑀸𑀮𑀁 𑀦 𑀯𑀺𑀮𑁂𑀧𑀦𑀁;_  
_𑀆𑀓𑀧𑁆𑀧𑀁 𑀲𑀭𑀓𑀼𑀢𑁆𑀢𑀺𑀁 𑀯𑀸,_  
_𑀦 𑀭𑀜𑁆𑀜𑁄 𑀲𑀤𑀺𑀲𑀫𑀸𑀘𑀭𑁂;_  
_𑀅𑀜𑁆𑀜𑀁 𑀓𑀭𑁂𑀬𑁆𑀬 𑀆𑀓𑀧𑁆𑀧𑀁,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1482.)

2821\. _𑀓𑀻𑀍𑀏 𑀭𑀸𑀚𑀸 𑀅𑀫𑀘𑁆𑀘𑁂𑀳𑀺,_  
_𑀪𑀭𑀺𑀬𑀸𑀳𑀺 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄;_  
_𑀦𑀸𑀫𑀘𑁆𑀘𑁄 𑀭𑀸𑀚𑀪𑀭𑀺𑀬𑀸𑀲𑀼,_  
_𑀪𑀸𑀯𑀁 𑀓𑀼𑀩𑁆𑀩𑁂𑀣 𑀧𑀡𑁆𑀟𑀺𑀢𑁄𑁇_  


(1483.)

2822\. _𑀅𑀦𑀼𑀤𑁆𑀥𑀢𑁄 𑀅𑀘𑀧𑀮𑁄,_  
_𑀦𑀺𑀧𑀓𑁄 𑀲𑀁𑀯𑀼𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀫𑀦𑁄𑀧𑀡𑀺𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1484.)

2823\. _𑀦𑀸𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸𑀳𑀺 𑀓𑀻𑀍𑀏𑀬𑁆𑀬,_  
_𑀦 𑀫𑀦𑁆𑀢𑁂𑀬𑁆𑀬 𑀭𑀳𑁄𑀕𑀢𑁄;_  
_𑀦𑀸𑀲𑁆𑀲 𑀓𑁄𑀲𑀸 𑀥𑀦𑀁 𑀕𑀡𑁆𑀳𑁂,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1485.)

2824\. _𑀦 𑀦𑀺𑀤𑁆𑀤𑀁 𑀩𑀳𑀼 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬,_  
_𑀦 𑀫𑀤𑀸𑀬 𑀲𑀼𑀭𑀁 𑀧𑀺𑀯𑁂;_  
_𑀦𑀸𑀲𑁆𑀲 𑀤𑀸𑀬𑁂 𑀫𑀺𑀕𑁂 𑀳𑀜𑁆𑀜𑁂,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1486.)

2825\. _𑀦𑀸𑀲𑁆𑀲 𑀧𑀻𑀞𑀁 𑀦 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁,_  
_𑀦 𑀓𑁄𑀘𑁆𑀙𑀁 𑀦 𑀦𑀸𑀯𑀁 𑀭𑀣𑀁;_  
_𑀲𑀫𑁆𑀫𑀢𑁄𑀫𑁆𑀳𑀻𑀢𑀺 𑀆𑀭𑀽𑀳𑁂,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1487.)

2826\. _𑀦𑀸𑀢𑀺𑀤𑀽𑀭𑁂 𑀪𑀚𑁂 𑀭𑀜𑁆𑀜𑁄,_  
_𑀦𑀸𑀘𑁆𑀘𑀸𑀲𑀦𑁆𑀦𑁂 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄;_  
_𑀲𑀫𑁆𑀫𑀼𑀔𑀜𑁆𑀘𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬,_  
_𑀲𑀦𑁆𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀲𑀪𑀢𑁆𑀢𑀼𑀦𑁄𑁇_  


(1488.)

2827\. _𑀦 𑀯𑁂 𑀭𑀸𑀚𑀸 𑀲𑀔𑀸 𑀳𑁄𑀢𑀺,_  
_𑀦 𑀭𑀸𑀚𑀸 𑀳𑁄𑀢𑀺 𑀫𑁂𑀣𑀼𑀦𑁄;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀓𑀼𑀚𑁆𑀛𑀦𑁆𑀢𑀺 𑀭𑀸𑀚𑀸𑀦𑁄,_  
_𑀲𑀽𑀓𑁂𑀦𑀓𑁆𑀔𑀻𑀯 𑀖𑀝𑁆𑀝𑀺𑀢𑀁𑁇_  


(1489.)

2828\. _𑀦 𑀧𑀽𑀚𑀺𑀢𑁄 𑀫𑀜𑁆𑀜𑀫𑀸𑀦𑁄,_  
_𑀫𑁂𑀥𑀸𑀯𑀻 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀦𑀭𑁄;_  
_𑀨𑀭𑀼𑀲𑀁 𑀧𑀢𑀺𑀫𑀦𑁆𑀢𑁂𑀬𑁆𑀬,_  
_𑀭𑀸𑀚𑀸𑀦𑀁 𑀧𑀭𑀺𑀲𑀗𑁆𑀕𑀢𑀁𑁇_  


(1490.)

2829\. _𑀮𑀤𑁆𑀥𑀤𑁆𑀯𑀸𑀭𑁄 𑀮𑀪𑁂 𑀤𑁆𑀯𑀸𑀭𑀁,_  
_𑀦𑁂𑀯 𑀭𑀸𑀚𑀽𑀲𑀼 𑀯𑀺𑀲𑁆𑀲𑀲𑁂;_  
_𑀅𑀕𑁆𑀕𑀻𑀯 𑀲𑀁𑀬𑀢𑁄 𑀢𑀺𑀝𑁆𑀞𑁂,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1491.)

2830\. _𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀪𑀸𑀢𑀭𑀁 𑀯𑀸 𑀲𑀁,_  
_𑀲𑀫𑁆𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀕𑀸𑀫𑁂𑀳𑀺 𑀦𑀺𑀕𑀫𑁂𑀳𑀺 𑀯𑀸,_  
_𑀭𑀝𑁆𑀞𑁂𑀳𑀺 𑀚𑀦𑀧𑀤𑁂𑀳𑀺 𑀯𑀸;_  
_𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑁄 𑀉𑀧𑁂𑀓𑁆𑀔𑁂𑀬𑁆𑀬,_  
_𑀦 𑀪𑀡𑁂 𑀙𑁂𑀓𑀧𑀸𑀧𑀓𑀁𑁇_  


(1492.)

2831\. _𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑁂 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑁂,_  
_𑀭𑀣𑀺𑀓𑁂 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑁂;_  
_𑀢𑁂𑀲𑀁 𑀓𑀫𑁆𑀫𑀸𑀯𑀤𑀸𑀦𑁂𑀦,_  
_𑀭𑀸𑀚𑀸 𑀯𑀟𑁆𑀠𑁂𑀢𑀺 𑀯𑁂𑀢𑀦𑀁;_  
_𑀦 𑀢𑁂𑀲𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀕𑀘𑁆𑀙𑁂,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1493.)

2832\. _𑀘𑀸𑀧𑁄𑀯𑀽𑀦𑀼𑀤𑀭𑁄 𑀥𑀻𑀭𑁄,_  
_𑀯𑀁𑀲𑁄𑀯𑀸𑀧𑀺 𑀧𑀓𑀫𑁆𑀧𑀬𑁂;_  
_𑀧𑀝𑀺𑀮𑁄𑀫𑀁 𑀦 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1494.)

2833\. _𑀘𑀸𑀧𑁄𑀯𑀽𑀦𑀼𑀤𑀭𑁄 𑀅𑀲𑁆𑀲,_  
_𑀫𑀘𑁆𑀙𑁄𑀯𑀲𑁆𑀲 𑀅𑀚𑀺𑀯𑁆𑀳𑀯𑀸;_  
_𑀅𑀧𑁆𑀧𑀸𑀲𑀻 𑀦𑀺𑀧𑀓𑁄 𑀲𑀽𑀭𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1495.)

2834\. _𑀦 𑀩𑀸𑀍𑀳𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬,_  
_𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀁 𑀢𑁂𑀚𑀲𑀗𑁆𑀔𑀬𑀁;_  
_𑀓𑀸𑀲𑀁 𑀲𑀸𑀲𑀁 𑀤𑀭𑀁 𑀩𑀮𑁆𑀬𑀁,_  
_𑀔𑀻𑀡𑀫𑁂𑀥𑁄 𑀦𑀺𑀕𑀘𑁆𑀙𑀢𑀺𑁇_  


(1496.)

2835\. _𑀦𑀸𑀢𑀺𑀯𑁂𑀮𑀁 𑀧𑀪𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀦 𑀢𑀼𑀡𑁆𑀳𑀻 𑀲𑀩𑁆𑀩𑀤𑀸 𑀲𑀺𑀬𑀸;_  
_𑀅𑀯𑀺𑀓𑀺𑀡𑁆𑀡𑀁 𑀫𑀺𑀢𑀁 𑀯𑀸𑀘𑀁,_  
_𑀧𑀢𑁆𑀢𑁂 𑀓𑀸𑀮𑁂 𑀉𑀤𑀻𑀭𑀬𑁂𑁇_  


(1497.)

2836\. _𑀅𑀓𑁆𑀓𑁄𑀥𑀦𑁄 𑀅𑀲𑀗𑁆𑀖𑀝𑁆𑀝𑁄,_  
_𑀲𑀘𑁆𑀘𑁄 𑀲𑀡𑁆𑀳𑁄 𑀅𑀧𑁂𑀲𑀼𑀡𑁄;_  
_𑀲𑀫𑁆𑀨𑀁 𑀕𑀺𑀭𑀁 𑀦 𑀪𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1498.)

2837\. _𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑁄 𑀅𑀲𑁆𑀲,_  
_𑀓𑀼𑀮𑁂 𑀚𑁂𑀝𑁆𑀞𑀸𑀧𑀘𑀸𑀬𑀺𑀓𑁄;_  
_𑀲𑀡𑁆𑀳𑁄 𑀲𑀔𑀺𑀮𑀲𑀫𑁆𑀪𑀸𑀲𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1499.)

2838\. _𑀯𑀺𑀦𑀻𑀢𑁄 𑀲𑀺𑀧𑁆𑀧𑀯𑀸 𑀤𑀦𑁆𑀢𑁄,_  
_𑀓𑀢𑀢𑁆𑀢𑁄 𑀦𑀺𑀬𑀢𑁄 𑀫𑀼𑀤𑀼;_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀲𑀼𑀘𑀺 𑀤𑀓𑁆𑀔𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1500.)

2839\. _𑀦𑀺𑀯𑀸𑀢𑀯𑀼𑀢𑁆𑀢𑀺 𑀯𑀼𑀤𑁆𑀥𑁂𑀲𑀼,_  
_𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀲𑀕𑀸𑀭𑀯𑁄;_  
_𑀲𑀼𑀭𑀢𑁄 𑀲𑀼𑀔𑀲𑀁𑀯𑀸𑀲𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1501.)

2840\. _𑀆𑀭𑀓𑀸 𑀧𑀭𑀺𑀯𑀚𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀲𑀳𑀺𑀢𑀼𑀁 𑀧𑀳𑀺𑀢𑀁 𑀚𑀦𑀁;_  
_𑀪𑀢𑁆𑀢𑀸𑀭𑀜𑁆𑀜𑁂𑀯𑀼𑀤𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬,_  
_𑀦 𑀘 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀭𑀸𑀚𑀺𑀦𑁄𑁇_  


(1502.)

2841\. _𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀘𑀸𑀧𑀺,_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂;_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1503.)

2842\. _𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀘𑀸𑀧𑀺,_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂;_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀯𑀸𑀲𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1504.)

2843\. _𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀘𑀸𑀧𑀺,_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂;_  
_𑀢𑀧𑁆𑀧𑁂𑀬𑁆𑀬 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1505.)

2844\. _𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂 𑀘𑀸𑀧𑀺,_  
_𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁂;_  
_𑀆𑀲𑀚𑁆𑀚 𑀧𑀜𑁆𑀜𑁂 𑀲𑁂𑀯𑁂𑀣,_  
_𑀆𑀓𑀗𑁆𑀔𑀁 𑀯𑀼𑀤𑁆𑀥𑀺𑀫𑀢𑁆𑀢𑀦𑁄𑁇_  


(1506.)

2845\. _𑀤𑀺𑀦𑁆𑀦𑀧𑀼𑀩𑁆𑀩𑀁 𑀦 𑀳𑀸𑀧𑁂𑀬𑁆𑀬,_  
_𑀤𑀸𑀦𑀁 𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁂;_  
_𑀦 𑀘 𑀓𑀺𑀜𑁆𑀘𑀺 𑀦𑀺𑀯𑀸𑀭𑁂𑀬𑁆𑀬,_  
_𑀤𑀸𑀦𑀓𑀸𑀮𑁂 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁂𑁇_  


(1507.)

2846\. _𑀧𑀜𑁆𑀜𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀯𑀺𑀥𑀸𑀦𑀯𑀺𑀥𑀺𑀓𑁄𑀯𑀺𑀤𑁄;_  
_𑀓𑀸𑀮𑀜𑁆𑀜𑀽 𑀲𑀫𑀬𑀜𑁆𑀜𑀽 𑀘,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1508.)

2847\. _𑀉𑀝𑁆𑀞𑀸𑀢𑀸 𑀓𑀫𑁆𑀫𑀥𑁂𑀬𑁆𑀬𑁂𑀲𑀼,_  
_𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑁄 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄;_  
_𑀲𑀼𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑁄,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1509.)

2848\. _𑀔𑀮𑀁 𑀲𑀸𑀮𑀁 𑀧𑀲𑀼𑀁 𑀔𑁂𑀢𑁆𑀢𑀁,_  
_𑀕𑀦𑁆𑀢𑀸 𑀘𑀲𑁆𑀲 𑀅𑀪𑀺𑀓𑁆𑀔𑀡𑀁;_  
_𑀫𑀺𑀢𑀁 𑀥𑀜𑁆𑀜𑀁 𑀦𑀺𑀥𑀸𑀧𑁂𑀬𑁆𑀬,_  
_𑀫𑀺𑀢𑀁𑀯 𑀧𑀸𑀘𑀬𑁂 𑀖𑀭𑁂𑁇_  


(1510.)

2849\. _𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀸 𑀪𑀸𑀢𑀭𑀁 𑀯𑀸 𑀲𑀁,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀅𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀅𑀦𑀗𑁆𑀕𑀯𑀸 𑀳𑀺 𑀢𑁂 𑀩𑀸𑀮𑀸,_  
_𑀬𑀣𑀸 𑀧𑁂𑀢𑀸 𑀢𑀣𑁂𑀯 𑀢𑁂;_  
_𑀘𑁄𑀍𑀅𑀜𑁆𑀘 𑀦𑁂𑀲𑀁 𑀧𑀺𑀡𑁆𑀟𑀜𑁆𑀘,_  
_𑀆𑀲𑀻𑀦𑀸𑀦𑀁 𑀧𑀤𑀸𑀧𑀬𑁂𑁇_  


(1511.)

2850\. _𑀤𑀸𑀲𑁂 𑀓𑀫𑁆𑀫𑀓𑀭𑁂 𑀧𑁂𑀲𑁆𑀲𑁂,_  
_𑀲𑀻𑀮𑁂𑀲𑀼 𑀲𑀼𑀲𑀫𑀸𑀳𑀺𑀢𑁂;_  
_𑀤𑀓𑁆𑀔𑁂 𑀉𑀝𑁆𑀞𑀸𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂,_  
_𑀆𑀥𑀺𑀧𑀘𑁆𑀘𑀫𑁆𑀳𑀺 𑀞𑀸𑀧𑀬𑁂𑁇_  


(1512.)

2851\. _𑀲𑀻𑀮𑀯𑀸 𑀘 𑀅𑀮𑁄𑀮𑁄 𑀘,_  
_𑀅𑀦𑀼𑀭𑀓𑁆𑀔𑁄 𑀘 𑀭𑀸𑀚𑀺𑀦𑁄;_  
_𑀆𑀯𑀻 𑀭𑀳𑁄 𑀳𑀺𑀢𑁄 𑀢𑀲𑁆𑀲,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1513.)

2852\. _𑀙𑀦𑁆𑀤𑀜𑁆𑀜𑀽 𑀭𑀸𑀚𑀺𑀦𑁄 𑀘𑀲𑁆𑀲,_  
_𑀘𑀺𑀢𑁆𑀢𑀝𑁆𑀞𑁄 𑀅𑀲𑁆𑀲 𑀭𑀸𑀚𑀺𑀦𑁄;_  
_𑀅𑀲𑀗𑁆𑀓𑀼𑀲𑀓𑀯𑀼𑀢𑁆𑀢𑀺’𑀲𑁆𑀲,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1514.)

2853\. _𑀉𑀘𑁆𑀙𑀸𑀤𑀬𑁂 𑀘 𑀦𑁆𑀳𑀸𑀧𑀬𑁂,_  
_𑀥𑁄𑀯𑁂 𑀧𑀸𑀤𑁂 𑀅𑀥𑁄𑀲𑀺𑀭𑀁;_  
_𑀆𑀳𑀢𑁄𑀧𑀺 𑀦 𑀓𑀼𑀧𑁆𑀧𑁂𑀬𑁆𑀬,_  
_𑀲 𑀭𑀸𑀚𑀯𑀲𑀢𑀺𑀁 𑀯𑀲𑁂𑁇_  


(1515.)

2854\. _𑀓𑀼𑀫𑁆𑀪𑀫𑁆𑀧𑀜𑁆𑀚𑀮𑀺𑀁 𑀓𑀭𑀺𑀬𑀸,_  
_𑀘𑀸𑀝𑀜𑁆𑀘𑀸𑀧𑀺 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀓𑀺𑀫𑁂𑀯 𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀸𑀦𑀁,_  
_𑀤𑀸𑀢𑀸𑀭𑀁 𑀥𑀻𑀭𑀫𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(1516.)

2855\. _𑀬𑁄 𑀤𑁂𑀢𑀺 𑀲𑀬𑀦𑀁 𑀯𑀢𑁆𑀣𑀁,_  
_𑀬𑀸𑀦𑀁 𑀆𑀯𑀲𑀣𑀁 𑀖𑀭𑀁;_  
_𑀧𑀚𑁆𑀚𑀼𑀦𑁆𑀦𑁄𑀭𑀺𑀯 𑀪𑀽𑀢𑀸𑀦𑀺,_  
_𑀪𑁄𑀕𑁂𑀳𑀺 𑀅𑀪𑀺𑀯𑀲𑁆𑀲𑀢𑀺𑁇_  


(1517.)

2856\. _𑀏𑀲𑀬𑁆𑀬𑁄 𑀭𑀸𑀚𑀯𑀲𑀢𑀺,_  
_𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁄 𑀬𑀣𑀸 𑀦𑀭𑁄;_  
_𑀆𑀭𑀸𑀥𑀬𑀢𑀺 𑀭𑀸𑀚𑀸𑀦𑀁,_  
_𑀧𑀽𑀚𑀁 𑀮𑀪𑀢𑀺 𑀪𑀢𑁆𑀢𑀼𑀲𑀼”𑁇_  


---

2857\. 𑀭𑀸𑀚𑀯𑀲𑀢𑀺 𑀦𑀸𑀫𑁇
