# 𑁨𑁨𑁇𑁧𑁇𑁧𑁦𑁇𑁩 𑀤𑀸𑀦𑀓𑀡𑁆𑀟

### 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀚𑀸𑀢𑀓

### 𑀤𑀸𑀦𑀓𑀡𑁆𑀟

(1744.)

3092\. _𑀢𑁂𑀲𑀁 𑀮𑀸𑀮𑀧𑁆𑀧𑀺𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀼𑀡𑀺𑀲𑀸𑀬 𑀘;_  
_𑀓𑀮𑀼𑀦𑀁 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀲𑀺,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀻𑁇_  


(1745.)

3093\. _“𑀲𑁂𑀬𑁆𑀬𑁄 𑀯𑀺𑀲𑀁 𑀫𑁂 𑀔𑀸𑀬𑀺𑀢𑀁,_  
_𑀧𑀧𑀸𑀢𑀸 𑀧𑀧𑀢𑁂𑀬𑁆𑀬𑀳𑀁;_  
_𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀩𑀚𑁆𑀛 𑀫𑀺𑀬𑁆𑀬𑀸𑀳𑀁,_  
_𑀓𑀲𑁆𑀫𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀧𑀼𑀢𑁆𑀢𑀁;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(1746.)

3094\. _𑀅𑀚𑁆𑀛𑀸𑀬𑀓𑀁 𑀤𑀸𑀦𑀧𑀢𑀺𑀁,_  
_𑀬𑀸𑀘𑀬𑁄𑀕𑀁 𑀅𑀫𑀘𑁆𑀙𑀭𑀺𑀁;_  
_𑀧𑀽𑀚𑀺𑀢𑀁 𑀧𑀝𑀺𑀭𑀸𑀚𑀽𑀳𑀺,_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀫𑀦𑁆𑀢𑀁 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀁;_  
_𑀓𑀲𑁆𑀫𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(1747.)

3095\. _𑀫𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀢𑀺𑀪𑀭𑀁 𑀚𑀦𑁆𑀢𑀼𑀁,_  
_𑀓𑀼𑀮𑁂 𑀚𑁂𑀝𑁆𑀞𑀸𑀧𑀘𑀸𑀬𑀺𑀓𑀁;_  
_𑀓𑀲𑁆𑀫𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(1748.)

3096\. _𑀭𑀜𑁆𑀜𑁄 𑀳𑀺𑀢𑀁 𑀤𑁂𑀯𑀺𑀳𑀺𑀢𑀁,_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀲𑀔𑀺𑀦𑀁 𑀳𑀺𑀢𑀁;_  
_𑀳𑀺𑀢𑀁 𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀭𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀓𑀲𑁆𑀫𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀧𑀼𑀢𑁆𑀢𑀁;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(1749.)

3097\. _𑀫𑀥𑀽𑀦𑀺𑀯 𑀧𑀮𑀸𑀢𑀸𑀦𑀺,_  
_𑀅𑀫𑁆𑀩𑀸𑀯 𑀧𑀢𑀺𑀢𑀸 𑀙𑀫𑀸;_  
_𑀏𑀯𑀁 𑀳𑁂𑀲𑁆𑀲𑀢𑀺 𑀢𑁂 𑀭𑀝𑁆𑀞𑀁,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(1750.)

3098\. _𑀳𑀁𑀲𑁄 𑀦𑀺𑀔𑀻𑀡𑀧𑀢𑁆𑀢𑁄𑀯,_  
_𑀧𑀮𑁆𑀮𑀮𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀽𑀤𑀓𑁂;_  
_𑀅𑀧𑀯𑀺𑀤𑁆𑀥𑁄 𑀅𑀫𑀘𑁆𑀘𑁂𑀳𑀺,_  
_𑀏𑀓𑁄 𑀭𑀸𑀚𑀸 𑀯𑀺𑀳𑀺𑀬𑁆𑀬𑀲𑀺𑁇_  


(1751.)

3099\. _𑀢𑀁 𑀢𑀁 𑀩𑁆𑀭𑀽𑀫𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀅𑀢𑁆𑀣𑁄 𑀢𑁂 𑀫𑀸 𑀉𑀧𑀘𑁆𑀘𑀕𑀸;_  
_𑀫𑀸 𑀦𑀁 𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀸,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀺 𑀅𑀤𑀽𑀲𑀓𑀁”𑁇_  


(1752.)

3100\. _“𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀸𑀧𑀘𑀺𑀢𑀺𑀁 𑀓𑀼𑀫𑁆𑀫𑀺,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀺𑀦𑀬𑀁 𑀥𑀚𑀁;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀫𑀺 𑀲𑀓𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀧𑀸𑀡𑀸 𑀧𑀺𑀬𑀢𑀭𑁄 𑀳𑀺 𑀫𑁂”𑁇_  


(1753.)

3101\. _“𑀬𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀥𑀚𑀕𑁆𑀕𑀸𑀦𑀺,_  
_𑀓𑀡𑀺𑀓𑀸𑀭𑀸𑀯 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀬𑀸𑀬𑀦𑁆𑀢𑀫𑀦𑀼𑀬𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑁆𑀯𑀚𑁆𑀚𑁂𑀓𑁄𑀯 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(1754.)

3102\. _𑀬𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀥𑀚𑀕𑁆𑀕𑀸𑀦𑀺,_  
_𑀓𑀡𑀺𑀓𑀸𑀭𑀯𑀦𑀸𑀦𑀺𑀯;_  
_𑀬𑀸𑀬𑀦𑁆𑀢𑀫𑀦𑀼𑀬𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑁆𑀯𑀚𑁆𑀚𑁂𑀓𑁄𑀯 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(1755.)

3103\. _𑀬𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀻𑀓𑀸𑀦𑀺,_  
_𑀓𑀡𑀺𑀓𑀸𑀭𑀸𑀯 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀬𑀸𑀬𑀦𑁆𑀢𑀫𑀦𑀼𑀬𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑁆𑀯𑀚𑁆𑀚𑁂𑀓𑁄𑀯 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(1756.)

3104\. _𑀬𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀦𑀻𑀓𑀸𑀦𑀺,_  
_𑀓𑀡𑀺𑀓𑀸𑀭𑀯𑀦𑀸𑀦𑀺𑀯;_  
_𑀬𑀸𑀬𑀦𑁆𑀢𑀫𑀦𑀼𑀬𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑁆𑀯𑀚𑁆𑀚𑁂𑀓𑁄𑀯 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(1757.)

3105\. _𑀇𑀦𑁆𑀤𑀕𑁄𑀧𑀓𑀯𑀡𑁆𑀡𑀸𑀪𑀸,_  
_𑀕𑀦𑁆𑀥𑀸𑀭𑀸 𑀧𑀡𑁆𑀟𑀼𑀓𑀫𑁆𑀩𑀮𑀸;_  
_𑀬𑀸𑀬𑀦𑁆𑀢𑀫𑀦𑀼𑀬𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀲𑁆𑀯𑀚𑁆𑀚𑁂𑀓𑁄𑀯 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(1758.)

3106\. _𑀬𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀳𑀢𑁆𑀣𑀺𑀦𑀸 𑀬𑀸𑀢𑀺,_  
_𑀲𑀺𑀯𑀺𑀓𑀸𑀬 𑀭𑀣𑁂𑀦 𑀘;_  
_𑀲𑁆𑀯𑀚𑁆𑀚 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀓𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀢𑁆𑀢𑀺𑀓𑁄𑁇_  


(1759.)

3107\. _𑀓𑀣𑀁 𑀘𑀦𑁆𑀤𑀦𑀮𑀺𑀢𑁆𑀢𑀗𑁆𑀕𑁄,_  
_𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑀧𑁆𑀧𑀩𑁄𑀥𑀦𑁄;_  
_𑀔𑀼𑀭𑀸𑀚𑀺𑀦𑀁 𑀨𑀭𑀲𑀼𑀜𑁆𑀘,_  
_𑀔𑀸𑀭𑀺𑀓𑀸𑀚𑀜𑁆𑀘 𑀳𑀸𑀳𑀺𑀢𑀺𑁇_  


(1760.)

3108\. _𑀓𑀲𑁆𑀫𑀸 𑀦𑀸𑀪𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀓𑀸𑀲𑀸𑀯𑀸 𑀅𑀚𑀺𑀦𑀸𑀦𑀺 𑀘;_  
_𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀁 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑀁,_  
_𑀓𑀲𑁆𑀫𑀸 𑀘𑀻𑀭𑀁 𑀦 𑀩𑀚𑁆𑀛𑀭𑁂𑁇_  


(1761.)

3109\. _𑀓𑀣𑀁 𑀦𑀼 𑀘𑀻𑀭𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺,_  
_𑀭𑀸𑀚𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀚𑀦𑀸;_  
_𑀓𑀣𑀁 𑀓𑀼𑀲𑀫𑀬𑀁 𑀘𑀻𑀭𑀁,_  
_𑀫𑀤𑁆𑀤𑀻 𑀧𑀭𑀺𑀤𑀳𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(1762.)

3110\. _𑀓𑀸𑀲𑀺𑀬𑀸𑀦𑀺 𑀘 𑀥𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀔𑁄𑀫𑀓𑁄𑀝𑀼𑀫𑁆𑀩𑀭𑀸𑀦𑀺 𑀘;_  
_𑀓𑀼𑀲𑀘𑀻𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀻,_  
_𑀓𑀣𑀁 𑀫𑀤𑁆𑀤𑀻 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇_  


(1763.)

3111\. _𑀯𑀬𑁆𑀳𑀸𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀯𑀺𑀓𑀸𑀬 𑀭𑀣𑁂𑀦 𑀘;_  
_𑀲𑀸 𑀓𑀣𑀚𑁆𑀚 𑀅𑀦𑀼𑀚𑁆𑀛𑀗𑁆𑀕𑀻,_  
_𑀧𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑁇_  


(1764.)

3112\. _𑀬𑀲𑁆𑀲𑀸 𑀫𑀼𑀤𑀼𑀢𑀮𑀸 𑀳𑀢𑁆𑀣𑀸,_  
_𑀘𑀭𑀡𑀸 𑀘 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸;_  
_𑀲𑀸 𑀓𑀣𑀚𑁆𑀚 𑀅𑀦𑀼𑀚𑁆𑀛𑀗𑁆𑀕𑀻,_  
_𑀧𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑁇_  


(1765.)

3113\. _𑀬𑀲𑁆𑀲𑀸 𑀫𑀼𑀤𑀼𑀢𑀮𑀸 𑀧𑀸𑀤𑀸,_  
_𑀘𑀭𑀡𑀸 𑀘 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸;_  
_𑀧𑀸𑀤𑀼𑀓𑀸𑀳𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀸𑀳𑀺,_  
_𑀧𑀻𑀍𑀅𑀫𑀸𑀦𑀸𑀯 𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀲𑀸 𑀓𑀣𑀚𑁆𑀚 𑀅𑀦𑀼𑀚𑁆𑀛𑀗𑁆𑀕𑀻,_  
_𑀧𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑁇_  


(1766.)

3114\. _𑀬𑀸𑀲𑁆𑀲𑀼 𑀇𑀢𑁆𑀣𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀧𑀼𑀭𑀢𑁄 𑀕𑀘𑁆𑀙𑀢𑀺 𑀫𑀸𑀮𑀺𑀦𑀻;_  
_𑀲𑀸 𑀓𑀣𑀚𑁆𑀚 𑀅𑀦𑀼𑀚𑁆𑀛𑀗𑁆𑀕𑀻,_  
_𑀯𑀦𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀏𑀓𑀺𑀓𑀸𑁇_  


(1767.)

3115\. _𑀬𑀸𑀲𑁆𑀲𑀼 𑀲𑀺𑀯𑀸𑀬 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀼𑀳𑀼𑀁 𑀉𑀢𑁆𑀢𑀲𑀢𑁂 𑀧𑀼𑀭𑁂;_  
_𑀲𑀸 𑀓𑀣𑀚𑁆𑀚 𑀅𑀦𑀼𑀚𑁆𑀛𑀗𑁆𑀕𑀻,_  
_𑀯𑀦𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀪𑀻𑀭𑀼𑀓𑀸𑁇_  


(1768.)

3116\. _𑀬𑀸𑀲𑁆𑀲𑀼 𑀇𑀦𑁆𑀤𑀲𑀕𑁄𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀉𑀮𑀽𑀓𑀲𑁆𑀲 𑀧𑀯𑀲𑁆𑀲𑀢𑁄;_  
_𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀦𑀤𑀢𑁄 𑀪𑀻𑀢𑀸,_  
_𑀯𑀸𑀭𑀼𑀡𑀻𑀯 𑀧𑀯𑁂𑀥𑀢𑀺;_  
_𑀲𑀸 𑀓𑀣𑀚𑁆𑀚 𑀅𑀦𑀼𑀚𑁆𑀛𑀗𑁆𑀕𑀻,_  
_𑀯𑀦𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀪𑀻𑀭𑀼𑀓𑀸𑁇_  


(1769.)

3117\. _𑀲𑀓𑀼𑀡𑀻 𑀳𑀢𑀧𑀼𑀢𑁆𑀢𑀸𑀯,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀘𑀺𑀭𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀛𑀸𑀬𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀆𑀕𑀫𑁆𑀫𑀺𑀫𑀁 𑀧𑀼𑀭𑀁𑁇_  


(1770.)

3118\. _𑀲𑀓𑀼𑀡𑀻 𑀳𑀢𑀧𑀼𑀢𑁆𑀢𑀸𑀯,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀓𑀺𑀲𑀸 𑀧𑀡𑁆𑀟𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀧𑀺𑀬𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑀲𑁆𑀲𑀢𑀻𑁇_  


(1771.)

3119\. _𑀲𑀓𑀼𑀡𑀻 𑀳𑀢𑀧𑀼𑀢𑁆𑀢𑀸𑀯,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀢𑁂𑀦 𑀢𑁂𑀦 𑀧𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀁,_  
_𑀧𑀺𑀬𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑀲𑁆𑀲𑀢𑀻𑁇_  


(1772.)

3120\. _𑀓𑀼𑀭𑀭𑀻 𑀳𑀢𑀙𑀸𑀧𑀸𑀯,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀘𑀺𑀭𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀛𑀸𑀬𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀆𑀕𑀫𑁆𑀫𑀺𑀫𑀁 𑀧𑀼𑀭𑀁𑁇_  


(1773.)

3121\. _𑀓𑀼𑀭𑀭𑀻 𑀳𑀢𑀙𑀸𑀧𑀸𑀯,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀓𑀺𑀲𑀸 𑀧𑀡𑁆𑀟𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀧𑀺𑀬𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑀲𑁆𑀲𑀢𑀻𑁇_  


(1774.)

3122\. _𑀓𑀼𑀭𑀭𑀻 𑀳𑀢𑀙𑀸𑀧𑀸𑀯,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀓𑀼𑀮𑀸𑀯𑀓𑀁;_  
_𑀢𑁂𑀦 𑀢𑁂𑀦 𑀧𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀁,_  
_𑀧𑀺𑀬𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑀲𑁆𑀲𑀢𑀻𑁇_  


(1775.)

3123\. _𑀲𑀸 𑀦𑀽𑀦 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀻𑀯,_  
_𑀧𑀮𑁆𑀮𑀮𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀽𑀤𑀓𑁂;_  
_𑀘𑀺𑀭𑀁 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀛𑀸𑀬𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀼𑀜𑁆𑀜𑀁 𑀆𑀕𑀫𑁆𑀫𑀺𑀫𑀁 𑀧𑀼𑀭𑀁𑁇_  


(1776.)

3124\. _𑀲𑀸 𑀦𑀽𑀦 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀻𑀯,_  
_𑀧𑀮𑁆𑀮𑀮𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀽𑀤𑀓𑁂;_  
_𑀓𑀺𑀲𑀸 𑀧𑀡𑁆𑀟𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀧𑀺𑀬𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑀲𑁆𑀲𑀢𑀻𑁇_  


(1777.)

3125\. _𑀲𑀸 𑀦𑀽𑀦 𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀻𑀯,_  
_𑀧𑀮𑁆𑀮𑀮𑀲𑁆𑀫𑀺𑀁 𑀅𑀦𑀽𑀤𑀓𑁂;_  
_𑀢𑁂𑀦 𑀢𑁂𑀦 𑀧𑀥𑀸𑀯𑀺𑀲𑁆𑀲𑀁,_  
_𑀧𑀺𑀬𑁂 𑀧𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑀲𑁆𑀲𑀢𑀻𑁇_  


(1778.)

3126\. _𑀏𑀯𑀁 𑀫𑁂 𑀯𑀺𑀮𑀧𑀦𑁆𑀢𑀺𑀬𑀸,_  
_𑀭𑀸𑀚𑀸 𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀤𑀽𑀲𑀓𑀁;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀺 𑀯𑀦𑀁 𑀭𑀝𑁆𑀞𑀸,_  
_𑀫𑀜𑁆𑀜𑁂 𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀚𑀻𑀯𑀺𑀢𑀁”𑁇_  


(1779.)

3127\. _𑀢𑀲𑁆𑀲𑀸 𑀮𑀸𑀮𑀧𑁆𑀧𑀺𑀢𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑀸 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀩𑀳𑀽;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_𑀲𑀺𑀯𑀺𑀓𑀜𑁆𑀜𑀸 𑀲𑀫𑀸𑀕𑀢𑀸𑁇_  


(1780.)

3128\. _𑀲𑀸𑀮𑀸𑀯 𑀲𑀫𑁆𑀧𑀫𑀣𑀺𑀢𑀸,_  
_𑀫𑀸𑀮𑀼𑀢𑁂𑀦 𑀧𑀫𑀤𑁆𑀤𑀺𑀢𑀸;_  
_𑀲𑁂𑀦𑁆𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑀸 𑀘 𑀤𑀸𑀭𑀸 𑀘,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀦𑀺𑀯𑁂𑀲𑀦𑁂𑁇_  


(1781.)

3129\. _𑀑𑀭𑁄𑀥𑀸 𑀘 𑀓𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀦𑀺𑀯𑁂𑀲𑀦𑁂𑁇_  


(1782.)

3130\. _𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀦𑀺𑀯𑁂𑀲𑀦𑁂𑁇_  


(1783.)

3131\. _𑀢𑀢𑁄 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀧𑀢𑀺;_  
_𑀅𑀣 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀤𑀸𑀦𑀁 𑀤𑀸𑀢𑀼𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(1784.)

3132\. _“𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀓𑀸𑀫𑀸𑀦𑀁,_  
_𑀲𑁄𑀡𑁆𑀟𑀸𑀦𑀁 𑀤𑁂𑀣 𑀯𑀸𑀭𑀼𑀡𑀺𑀁;_  
_𑀪𑁄𑀚𑀦𑀁 𑀪𑁄𑀚𑀦𑀢𑁆𑀣𑀻𑀦𑀁,_  
_𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀧𑀯𑁂𑀘𑁆𑀙𑀣𑁇_  


(1785.)

3133\. _𑀫𑀸 𑀘 𑀓𑀜𑁆𑀘𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑁂,_  
_𑀳𑁂𑀞𑀬𑀺𑀢𑁆𑀣 𑀇𑀥𑀸𑀕𑀢𑁂;_  
_𑀢𑀧𑁆𑀧𑁂𑀣 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀦,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀧𑀝𑀺𑀧𑀽𑀚𑀺𑀢𑀸”𑁇_  


(1786.)

3134\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀯𑀢𑁆𑀢𑀢𑀻 𑀲𑀤𑁆𑀤𑁄,_  
_𑀢𑀼𑀫𑀼𑀮𑁄 𑀪𑁂𑀭𑀯𑁄 𑀫𑀳𑀸;_  
_“𑀤𑀸𑀦𑁂𑀦 𑀢𑀁 𑀦𑀻𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀦 𑀤𑀸𑀦𑀁 𑀅𑀤𑀸 𑀢𑀼𑀯𑀁”𑁇_  


(1787.)

3135\. _𑀢𑁂 𑀲𑀼 𑀫𑀢𑁆𑀢𑀸 𑀓𑀺𑀮𑀦𑁆𑀢𑀸𑀯,_  
_𑀲𑀫𑁆𑀧𑀢𑀦𑁆𑀢𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀸;_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚𑁂,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(1788.)

3136\. _“𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀼𑀁 𑀯𑀢 𑀪𑁄 𑀭𑀼𑀓𑁆𑀔𑀁,_  
_𑀦𑀸𑀦𑀸𑀨𑀮𑀥𑀭𑀁 𑀤𑀼𑀫𑀁;_  
_𑀬𑀣𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀭𑀝𑁆𑀞𑀸,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(1789.)

3137\. _𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀼𑀁 𑀯𑀢 𑀪𑁄 𑀭𑀼𑀓𑁆𑀔𑀁,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀤𑀤𑀁 𑀤𑀼𑀫𑀁;_  
_𑀬𑀣𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀭𑀝𑁆𑀞𑀸,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁𑁇_  


(1790.)

3138\. _𑀅𑀘𑁆𑀙𑁂𑀘𑁆𑀙𑀼𑀁 𑀯𑀢 𑀪𑁄 𑀭𑀼𑀓𑁆𑀔𑀁,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀭𑀲𑀸𑀳𑀭𑀁;_  
_𑀬𑀣𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀭𑀝𑁆𑀞𑀸,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀺 𑀅𑀤𑀽𑀲𑀓𑀁”𑁇_  


(1791.)

3139\. _𑀬𑁂 𑀯𑀼𑀟𑁆𑀠𑀸 𑀬𑁂 𑀘 𑀤𑀳𑀭𑀸,_  
_𑀬𑁂 𑀘 𑀫𑀚𑁆𑀛𑀺𑀫𑀧𑁄𑀭𑀺𑀲𑀸;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚𑁂;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(1792.)

3140\. _𑀅𑀢𑀺𑀬𑀓𑁆𑀔𑀸 𑀯𑀲𑁆𑀲𑀯𑀭𑀸,_  
_𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀸 𑀘 𑀭𑀸𑀚𑀺𑀦𑁄;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚𑁂;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(1793.)

3141\. _𑀣𑀺𑀬𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_𑀬𑀸 𑀢𑀫𑁆𑀳𑀺 𑀦𑀕𑀭𑁂 𑀅𑀳𑀼;_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚𑁂,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(1794.)

3142\. _𑀬𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀬𑁂 𑀘 𑀲𑀫𑀡𑀸,_  
_𑀅𑀜𑁆𑀜𑁂 𑀯𑀸𑀧𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀸;_  
_𑀩𑀸𑀳𑀸 𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀧𑀓𑁆𑀓𑀦𑁆𑀤𑀼𑀁,_  
_“𑀅𑀥𑀫𑁆𑀫𑁄 𑀓𑀺𑀭 𑀪𑁄 𑀇𑀢𑀺𑁇_  


(1795.)

3143\. _𑀬𑀣𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀬𑀚𑀫𑀸𑀦𑁄 𑀲𑀓𑁂 𑀧𑀼𑀭𑁂;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀢𑁆𑀣𑁂𑀦,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺”𑁇_  


(1796.)

3144\. _𑀲𑀢𑁆𑀢 𑀳𑀢𑁆𑀣𑀺𑀲𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑁂;_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑀓𑀘𑁆𑀙𑁂 𑀫𑀸𑀢𑀗𑁆𑀕𑁂,_  
_𑀳𑁂𑀫𑀓𑀧𑁆𑀧𑀦𑀯𑀸𑀲𑀲𑁂𑁇_  


(1797.)

3145\. _𑀆𑀭𑀽𑀍𑀳𑁂 𑀕𑀸𑀫𑀡𑀻𑀬𑁂𑀳𑀺,_  
_𑀢𑁄𑀫𑀭𑀗𑁆𑀓𑀼𑀲𑀧𑀸𑀡𑀺𑀪𑀺;_  
_𑀏𑀲 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1798.)

3146\. _𑀲𑀢𑁆𑀢 𑀅𑀲𑁆𑀲𑀲𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑁂;_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑁂𑀯 𑀚𑀸𑀢𑀺𑀬𑀸,_  
_𑀲𑀺𑀦𑁆𑀥𑀯𑁂 𑀲𑀻𑀖𑀯𑀸𑀳𑀦𑁂𑁇_  


(1799.)

3147\. _𑀆𑀭𑀽𑀍𑀳𑁂 𑀕𑀸𑀫𑀡𑀻𑀬𑁂𑀳𑀺,_  
_𑀇𑀮𑁆𑀮𑀺𑀬𑀸𑀘𑀸𑀧𑀥𑀸𑀭𑀺𑀪𑀺;_  
_𑀏𑀲 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1800.)

3148\. _𑀲𑀢𑁆𑀢 𑀭𑀣𑀲𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑁂 𑀉𑀲𑁆𑀲𑀺𑀢𑀤𑁆𑀥𑀚𑁂;_  
_𑀤𑀻𑀧𑁂 𑀅𑀣𑁄𑀧𑀺 𑀯𑁂𑀬𑀕𑁆𑀖𑁂,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑁂𑁇_  


(1801.)

3149\. _𑀆𑀭𑀽𑀍𑀳𑁂 𑀕𑀸𑀫𑀡𑀻𑀬𑁂𑀳𑀺,_  
_𑀘𑀸𑀧𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀯𑀫𑁆𑀫𑀺𑀪𑀺;_  
_𑀏𑀲 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1802.)

3150\. _𑀲𑀢𑁆𑀢 𑀇𑀢𑁆𑀣𑀺𑀲𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀏𑀓𑀫𑁂𑀓𑀸 𑀭𑀣𑁂 𑀞𑀺𑀢𑀸;_  
_𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀸 𑀦𑀺𑀓𑁆𑀔𑀭𑀚𑁆𑀚𑀽𑀳𑀺,_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑁂𑀳𑀺 𑀅𑀮𑀗𑁆𑀓𑀢𑀸𑁇_  


(1803.)

3151\. _𑀧𑀻𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸 𑀧𑀻𑀢𑀯𑀲𑀦𑀸,_  
_𑀧𑀻𑀢𑀸𑀪𑀭𑀡𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀅𑀍𑀆𑀭𑀧𑀫𑁆𑀳𑀸 𑀳𑀲𑀼𑀮𑀸,_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀸 𑀢𑀦𑀼𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀏𑀲 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1804.)

3152\. _𑀲𑀢𑁆𑀢 𑀥𑁂𑀦𑀼𑀲𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀩𑁆𑀩𑀸 𑀓𑀁𑀲𑀼𑀧𑀥𑀸𑀭𑀡𑀸;_  
_𑀏𑀲 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1805.)

3153\. _𑀲𑀢𑁆𑀢 𑀤𑀸𑀲𑀺𑀲𑀢𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀲𑀢𑁆𑀢 𑀤𑀸𑀲𑀲𑀢𑀸𑀦𑀺 𑀘;_  
_𑀏𑀲 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1806.)

3154\. _𑀳𑀢𑁆𑀣𑀻 𑀅𑀲𑁆𑀲𑀭𑀣𑁂 𑀤𑀢𑁆𑀯𑀸,_  
_𑀦𑀸𑀭𑀺𑀬𑁄 𑀘 𑀅𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀏𑀲 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1807.)

3155\. _𑀢𑀤𑀸𑀲𑀺 𑀬𑀁 𑀪𑀺𑀁𑀲𑀦𑀓𑀁,_  
_𑀢𑀤𑀸𑀲𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀫𑀳𑀸𑀤𑀸𑀦𑁂 𑀧𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀳𑀺,_  
_𑀫𑁂𑀤𑀦𑀻 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀣𑁇_  


(1808.)

3156\. _𑀢𑀤𑀸𑀲𑀺 𑀬𑀁 𑀪𑀺𑀁𑀲𑀦𑀓𑀁,_  
_𑀢𑀤𑀸𑀲𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀬𑀁 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀢𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀢𑀺𑁇_  


(1809.)

3157\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀯𑀢𑁆𑀢𑀢𑀻 𑀲𑀤𑁆𑀤𑁄,_  
_𑀢𑀼𑀫𑀼𑀮𑁄 𑀪𑁂𑀭𑀯𑁄 𑀫𑀳𑀸;_  
_𑀤𑀸𑀦𑁂𑀦 𑀢𑀁 𑀦𑀻𑀳𑀭𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀦 𑀤𑀸𑀦𑀁 𑀅𑀤𑀸 𑀢𑀼𑀯𑀁𑁇_  


(1810.)

3158\. _𑀢𑁂 𑀲𑀼 𑀫𑀢𑁆𑀢𑀸 𑀓𑀺𑀮𑀦𑁆𑀢𑀸𑀯,_  
_𑀲𑀫𑁆𑀧𑀢𑀦𑁆𑀢𑀺 𑀯𑀦𑀺𑀩𑁆𑀩𑀓𑀸;_  
_𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚𑁂,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(1811.)

3159\. _𑀆𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀣 𑀭𑀸𑀚𑀸𑀦𑀁,_  
_𑀲𑀜𑁆𑀚𑀬𑀁 𑀥𑀫𑁆𑀫𑀺𑀦𑀁 𑀯𑀭𑀁;_  
_“𑀅𑀯𑀭𑀼𑀤𑁆𑀥𑀲𑀺 𑀫𑀁 𑀤𑁂𑀯,_  
_𑀯𑀗𑁆𑀓𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺 𑀧𑀩𑁆𑀩𑀢𑀁𑁇_  


(1812.)

3160\. _𑀬𑁂 𑀳𑀺 𑀓𑁂𑀘𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀪𑀽𑀢𑀸 𑀬𑁂 𑀘 𑀪𑀯𑀺𑀲𑁆𑀲𑀭𑁂;_  
_𑀅𑀢𑀺𑀢𑁆𑀢𑀸𑀬𑁂𑀯 𑀓𑀸𑀫𑁂𑀳𑀺,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀬𑀫𑀲𑀸𑀥𑀦𑀁𑁇_  


(1813.)

3161\. _𑀲𑁆𑀯𑀸𑀳𑀁 𑀲𑀓𑁂 𑀅𑀪𑀺𑀲𑁆𑀲𑀲𑀺𑀁,_  
_𑀬𑀚𑀫𑀸𑀦𑁄 𑀲𑀓𑁂 𑀧𑀼𑀭𑁂;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀢𑁆𑀣𑁂𑀦,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀳𑀁𑁇_  


(1814.)

3162\. _𑀅𑀖𑀁 𑀢𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀲𑁆𑀲𑀁,_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂;_  
_𑀔𑀕𑁆𑀕𑀤𑀻𑀧𑀺𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑁂,_  
_𑀅𑀳𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀫𑀺;_  
_𑀢𑀼𑀫𑁆𑀳𑁂 𑀧𑀗𑁆𑀓𑀫𑁆𑀳𑀺 𑀲𑀻𑀤𑀣𑁇_  


(1815.)

3163\. _𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀳𑀺 𑀫𑀁 𑀅𑀫𑁆𑀫,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀫𑀫 𑀭𑀼𑀘𑁆𑀘𑀢𑀺;_  
_𑀲𑁆𑀯𑀸𑀳𑀁 𑀲𑀓𑁂 𑀅𑀪𑀺𑀲𑁆𑀲𑀲𑀺𑀁,_  
_𑀬𑀚𑀫𑀸𑀦𑁄 𑀲𑀓𑁂 𑀧𑀼𑀭𑁂;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀢𑁆𑀣𑁂𑀦,_  
_𑀲𑀫𑁆𑀳𑀸 𑀭𑀝𑁆𑀞𑀸 𑀦𑀺𑀭𑀚𑁆𑀚𑀳𑀁𑁇_  


(1816.)

3164\. _𑀅𑀖𑀁 𑀢𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀲𑁆𑀲𑀁,_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂;_  
_𑀔𑀕𑁆𑀕𑀤𑀻𑀧𑀺𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑁂,_  
_𑀅𑀳𑀁 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑁄𑀫𑀺;_  
_𑀢𑀼𑀫𑁆𑀳𑁂 𑀧𑀗𑁆𑀓𑀫𑁆𑀳𑀺 𑀲𑀻𑀤𑀣”𑁇_  


(1817.)

3165\. _“𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺 𑀢𑀁 𑀧𑀼𑀢𑁆𑀢,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀢𑁂 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀼;_  
_𑀅𑀬𑀜𑁆𑀘 𑀫𑀤𑁆𑀤𑀻 𑀓𑀮𑁆𑀬𑀸𑀡𑀻,_  
_𑀲𑀼𑀲𑀜𑁆𑀜𑀸 𑀢𑀦𑀼𑀫𑀚𑁆𑀛𑀺𑀫𑀸;_  
_𑀅𑀘𑁆𑀙𑀢𑀁 𑀲𑀳 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀓𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(1818.)

3166\. _“𑀦𑀸𑀳𑀁 𑀅𑀓𑀸𑀫𑀸 𑀤𑀸𑀲𑀺𑀫𑁆𑀧𑀺,_  
_𑀅𑀭𑀜𑁆𑀜𑀁 𑀦𑁂𑀢𑀼𑀫𑀼𑀲𑁆𑀲𑀳𑁂;_  
_𑀲𑀘𑁂 𑀇𑀘𑁆𑀙𑀢𑀺 𑀅𑀦𑁆𑀯𑁂𑀢𑀼,_  
_𑀲𑀘𑁂 𑀦𑀺𑀘𑁆𑀙𑀢𑀺 𑀅𑀘𑁆𑀙𑀢𑀼”𑁇_  


(1819.)

3167\. _𑀢𑀢𑁄 𑀲𑀼𑀡𑁆𑀳𑀁 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀬𑀸𑀘𑀺𑀢𑀼𑀁 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀣;_  
_“𑀫𑀸 𑀘𑀦𑁆𑀤𑀦𑀲𑀫𑀸𑀘𑀸𑀭𑁂,_  
_𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀅𑀥𑀸𑀭𑀬𑀺𑁇_  


(1820.)

3168\. _𑀫𑀸 𑀓𑀸𑀲𑀺𑀬𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀓𑀼𑀲𑀘𑀻𑀭𑀁 𑀅𑀥𑀸𑀭𑀬𑀺;_  
_𑀤𑀼𑀓𑁆𑀔𑁄 𑀯𑀸𑀲𑁄 𑀅𑀭𑀜𑁆𑀜𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑀸 𑀳𑀺 𑀢𑁆𑀯𑀁 𑀮𑀓𑁆𑀔𑀡𑁂 𑀕𑀫𑀺”𑁇_  


(1821.)

3169\. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸;_  
_“𑀦𑀸𑀳𑀁 𑀢𑀁 𑀲𑀼𑀔𑀫𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁,_  
_𑀬𑀁 𑀫𑁂 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀯𑀺𑀦𑀸”𑁇_  


(1822.)

3170\. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁄;_  
_“𑀇𑀗𑁆𑀖 𑀫𑀤𑁆𑀤𑀺 𑀦𑀺𑀲𑀸𑀫𑁂𑀳𑀺,_  
_𑀯𑀦𑁂 𑀬𑁂 𑀳𑁄𑀦𑁆𑀢𑀺 𑀤𑀼𑀲𑁆𑀲𑀳𑀸𑁇_  


(1823.)

3171\. _𑀩𑀳𑀽 𑀓𑀻𑀝𑀸 𑀧𑀝𑀗𑁆𑀕𑀸 𑀘,_  
_𑀫𑀓𑀲𑀸 𑀫𑀥𑀼𑀫𑀓𑁆𑀔𑀺𑀓𑀸;_  
_𑀢𑁂𑀧𑀺 𑀢𑀁 𑀢𑀢𑁆𑀣 𑀳𑀺𑀁𑀲𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀢𑀁 𑀢𑁂 𑀤𑀼𑀓𑁆𑀔𑀢𑀭𑀁 𑀲𑀺𑀬𑀸𑁇_  


(1824.)

3172\. _𑀅𑀧𑀭𑁂 𑀧𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀸𑀧𑁂,_  
_𑀦𑀤𑀻𑀦𑀼𑀧𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑁂;_  
_𑀲𑀧𑁆𑀧𑀸 𑀅𑀚𑀕𑀭𑀸 𑀦𑀸𑀫,_  
_𑀅𑀯𑀺𑀲𑀸 𑀢𑁂 𑀫𑀳𑀩𑁆𑀩𑀮𑀸𑁇_  


(1825.)

3173\. _𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀫𑀺𑀕𑀁 𑀯𑀸𑀧𑀺,_  
_𑀅𑀧𑀺 𑀫𑀸𑀲𑀦𑁆𑀦𑀫𑀸𑀕𑀢𑀁;_  
_𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀪𑁄𑀕𑁂𑀳𑀺,_  
_𑀯𑀲𑀫𑀸𑀦𑁂𑀦𑁆𑀢𑀺 𑀅𑀢𑁆𑀢𑀦𑁄𑁇_  


(1826.)

3174\. _𑀅𑀜𑁆𑀜𑁂𑀧𑀺 𑀓𑀡𑁆𑀳𑀚𑀝𑀺𑀦𑁄,_  
_𑀅𑀘𑁆𑀙𑀸 𑀦𑀸𑀫 𑀅𑀖𑀫𑁆𑀫𑀺𑀕𑀸;_  
_𑀦 𑀢𑁂𑀳𑀺 𑀧𑀼𑀭𑀺𑀲𑁄 𑀤𑀺𑀝𑁆𑀞𑁄,_  
_𑀭𑀼𑀓𑁆𑀔𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀫𑀼𑀘𑁆𑀘𑀢𑀺𑁇_  


(1827.)

3175\. _𑀲𑀗𑁆𑀖𑀝𑁆𑀝𑀬𑀦𑁆𑀢𑀸 𑀲𑀺𑀗𑁆𑀕𑀸𑀦𑀺,_  
_𑀢𑀺𑀓𑁆𑀔𑀕𑁆𑀕𑀸𑀢𑀺𑀧𑁆𑀧𑀳𑀸𑀭𑀺𑀦𑁄;_  
_𑀫𑀳𑀺𑀁𑀲𑀸 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑁂𑀢𑁆𑀣,_  
_𑀦𑀤𑀺𑀁 𑀲𑁄𑀢𑀼𑀫𑁆𑀩𑀭𑀁 𑀧𑀢𑀺𑁇_  


(1828.)

3176\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀫𑀺𑀕𑀸𑀦𑀁 𑀬𑀽𑀣𑀸𑀦𑀁,_  
_𑀕𑀯𑀁 𑀲𑀜𑁆𑀘𑀭𑀢𑀁 𑀯𑀦𑁂;_  
_𑀥𑁂𑀦𑀼𑀯 𑀯𑀘𑁆𑀙𑀕𑀺𑀤𑁆𑀥𑀸𑀯,_  
_𑀓𑀣𑀁 𑀫𑀤𑁆𑀤𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1829.)

3177\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀢𑀺𑀢𑁂 𑀖𑁄𑀭𑁂,_  
_𑀤𑀼𑀫𑀕𑁆𑀕𑁂𑀲𑀼 𑀧𑁆𑀮𑀯𑀗𑁆𑀕𑀫𑁂;_  
_𑀅𑀔𑁂𑀢𑁆𑀢𑀜𑁆𑀜𑀸𑀬 𑀢𑁂 𑀫𑀤𑁆𑀤𑀺,_  
_𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑁂 𑀫𑀳𑀩𑁆𑀪𑀬𑀁𑁇_  


(1830.)

3178\. _𑀬𑀸 𑀢𑁆𑀯𑀁 𑀲𑀺𑀯𑀸𑀬 𑀲𑀼𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀼𑀳𑀼𑀁 𑀉𑀢𑁆𑀢𑀲𑀬𑀻 𑀧𑀼𑀭𑁂;_  
_𑀲𑀸 𑀢𑁆𑀯𑀁 𑀯𑀗𑁆𑀓𑀫𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀸,_  
_𑀓𑀣𑀁 𑀫𑀤𑁆𑀤𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1831.)

3179\. _𑀞𑀺𑀢𑁂 𑀫𑀚𑁆𑀛𑀦𑁆𑀳𑀺𑀓𑁂 𑀓𑀸𑀮𑁂,_  
_𑀲𑀦𑁆𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀧𑀓𑁆𑀔𑀺𑀲𑀼;_  
_𑀲𑀡𑀢𑁂𑀯 𑀩𑁆𑀭𑀳𑀸𑀭𑀜𑁆𑀜𑀁,_  
_𑀢𑀢𑁆𑀣 𑀓𑀺𑀁 𑀕𑀦𑁆𑀢𑀼𑀫𑀺𑀘𑁆𑀙𑀲𑀺”𑁇_  


(1832.)

3180\. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸;_  
_“𑀬𑀸𑀦𑀺 𑀏𑀢𑀸𑀦𑀺 𑀅𑀓𑁆𑀔𑀸𑀲𑀺,_  
_𑀯𑀦𑁂 𑀧𑀝𑀺𑀪𑀬𑀸𑀦𑀺 𑀫𑁂;_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀅𑀪𑀺𑀲𑀫𑁆𑀪𑁄𑀲𑁆𑀲𑀁,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1833.)

3181\. _𑀓𑀸𑀲𑀁 𑀓𑀼𑀲𑀁 𑀧𑁄𑀝𑀓𑀺𑀮𑀁,_  
_𑀉𑀲𑀺𑀭𑀁 𑀫𑀼𑀜𑁆𑀚𑀧𑀩𑁆𑀩𑀚𑀁;_  
_𑀉𑀭𑀲𑀸 𑀧𑀦𑀼𑀤𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀦𑀲𑁆𑀲 𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀼𑀦𑁆𑀦𑀬𑀸𑁇_  


(1834.)

3182\. _𑀩𑀳𑀽𑀳𑀺 𑀯𑀢 𑀘𑀭𑀺𑀬𑀸𑀳𑀺,_  
_𑀓𑀼𑀫𑀸𑀭𑀻 𑀯𑀺𑀦𑁆𑀤𑀢𑁂 𑀧𑀢𑀺𑀁;_  
_𑀉𑀤𑀭𑀲𑁆𑀲𑀼𑀧𑀭𑁄𑀥𑁂𑀦,_  
_𑀕𑁄𑀳𑀦𑀼𑀯𑁂𑀞𑀦𑁂𑀦 𑀘𑁇_  


(1835.)

3183\. _𑀅𑀕𑁆𑀕𑀺𑀲𑁆𑀲 𑀧𑀸𑀭𑀺𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀉𑀤𑀓𑀼𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑁂𑀦 𑀘;_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀓𑁂,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1836.)

3184\. _𑀅𑀧𑀺𑀲𑁆𑀲𑀸 𑀳𑁄𑀢𑀺 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄,_  
_𑀉𑀘𑁆𑀙𑀺𑀝𑁆𑀞𑀫𑀧𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁;_  
_𑀬𑁄 𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀓𑀸𑀫𑀁 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀢𑀺;_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀓𑁂,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1837.)

3185\. _𑀓𑁂𑀲𑀕𑁆𑀕𑀳𑀡𑀫𑀼𑀓𑁆𑀔𑁂𑀧𑀸,_  
_𑀪𑀽𑀫𑁆𑀬𑀸 𑀘 𑀧𑀭𑀺𑀲𑀼𑀫𑁆𑀪𑀦𑀸;_  
_𑀤𑀢𑁆𑀯𑀸 𑀘 𑀦𑁄𑀧𑀓𑁆𑀓𑀫𑀢𑀺,_  
_𑀩𑀳𑀼𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀦𑀧𑁆𑀧𑀓𑀁;_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀓𑁂,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1838.)

3186\. _𑀲𑀼𑀓𑁆𑀓𑀘𑁆𑀙𑀯𑀻 𑀯𑁂𑀥𑀯𑁂𑀭𑀸,_  
_𑀤𑀢𑁆𑀯𑀸 𑀲𑀼𑀪𑀕𑀫𑀸𑀦𑀺𑀦𑁄;_  
_𑀅𑀓𑀸𑀫𑀁 𑀧𑀭𑀺𑀓𑀟𑁆𑀠𑀦𑁆𑀢𑀺,_  
_𑀉𑀮𑀽𑀓𑀜𑁆𑀜𑁂𑀯 𑀯𑀸𑀬𑀲𑀸;_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀓𑁂,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1839.)

3187\. _𑀅𑀧𑀺 𑀜𑀸𑀢𑀺𑀓𑀼𑀮𑁂 𑀨𑀻𑀢𑁂,_  
_𑀓𑀁𑀲𑀧𑀚𑁆𑀚𑁄𑀢𑀦𑁂 𑀯𑀲𑀁;_  
_𑀦𑁂𑀯𑀸𑀪𑀺𑀯𑀸𑀓𑁆𑀬𑀁 𑀦 𑀮𑀪𑁂,_  
_𑀪𑀸𑀢𑀽𑀳𑀺 𑀲𑀔𑀺𑀦𑀻𑀳𑀺𑀧𑀺;_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀓𑁂,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1840.)

3188\. _𑀦𑀕𑁆𑀕𑀸 𑀦𑀤𑀻 𑀅𑀦𑀼𑀤𑀓𑀸,_  
_𑀦𑀕𑁆𑀕𑀁 𑀭𑀝𑁆𑀞𑀁 𑀅𑀭𑀸𑀚𑀓𑀁;_  
_𑀇𑀢𑁆𑀣𑀻𑀧𑀺 𑀯𑀺𑀥𑀯𑀸 𑀦𑀕𑁆𑀕𑀸,_  
_𑀬𑀲𑁆𑀲𑀸𑀧𑀺 𑀤𑀲 𑀪𑀸𑀢𑀭𑁄;_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀓𑁂,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1841.)

3189\. _𑀥𑀚𑁄 𑀭𑀣𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀥𑀽𑀫𑁄 𑀧𑀜𑁆𑀜𑀸𑀡𑀫𑀕𑁆𑀕𑀺𑀦𑁄;_  
_𑀭𑀸𑀚𑀸 𑀭𑀣𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀡𑀁,_  
_𑀪𑀢𑁆𑀢𑀸 𑀧𑀜𑁆𑀜𑀸𑀡𑀫𑀺𑀢𑁆𑀣𑀺𑀬𑀸;_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀁 𑀮𑁄𑀓𑁂,_  
_𑀕𑀘𑁆𑀙𑀜𑁆𑀜𑁂𑀯 𑀭𑀣𑁂𑀲𑀪𑁇_  


(1842.)

3190\. _𑀬𑀸 𑀤𑀮𑀺𑀤𑁆𑀤𑀻 𑀤𑀮𑀺𑀤𑁆𑀤𑀲𑁆𑀲,_  
_𑀅𑀟𑁆𑀠𑀸 𑀅𑀟𑁆𑀠𑀲𑁆𑀲 𑀓𑀺𑀢𑁆𑀢𑀺𑀫𑀸;_  
_𑀢𑀁 𑀯𑁂 𑀤𑁂𑀯𑀸 𑀧𑀲𑀁𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑀼𑀓𑁆𑀓𑀭𑀜𑁆𑀳𑀺 𑀓𑀭𑁄𑀢𑀺 𑀲𑀸𑁇_  


(1843.)

3191\. _𑀲𑀸𑀫𑀺𑀓𑀁 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀤𑀸 𑀓𑀸𑀲𑀸𑀬𑀯𑀸𑀲𑀺𑀦𑀻;_  
_𑀧𑀣𑀩𑁆𑀬𑀸𑀧𑀺 𑀅𑀪𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑁆𑀬𑀸,_  
_𑀯𑁂𑀥𑀩𑁆𑀬𑀁 𑀓𑀝𑀼𑀓𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑁇_  


(1844.)

3192\. _𑀅𑀧𑀺 𑀲𑀸𑀕𑀭𑀧𑀭𑀺𑀬𑀦𑁆𑀢𑀁,_  
_𑀩𑀳𑀼𑀯𑀺𑀢𑁆𑀢𑀥𑀭𑀁 𑀫𑀳𑀺𑀁;_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑀦𑀧𑀭𑀺𑀧𑀽𑀭𑀁,_  
_𑀦𑀺𑀘𑁆𑀙𑁂 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀯𑀺𑀦𑀸𑁇_  


(1845.)

3193\. _𑀓𑀣𑀁 𑀦𑀼 𑀢𑀸𑀲𑀁 𑀳𑀤𑀬𑀁,_  
_𑀲𑀼𑀔𑀭𑀸 𑀯𑀢 𑀇𑀢𑁆𑀣𑀺𑀬𑁄;_  
_𑀬𑀸 𑀲𑀸𑀫𑀺𑀓𑁂 𑀤𑀼𑀓𑁆𑀔𑀺𑀢𑀫𑁆𑀳𑀺,_  
_𑀲𑀼𑀔𑀫𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀅𑀢𑁆𑀢𑀦𑁄𑁇_  


(1846.)

3194\. _𑀦𑀺𑀓𑁆𑀔𑀫𑀦𑁆𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚𑁂,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂;_  
_𑀢𑀫𑀳𑀁 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑀲𑁆𑀲𑀁,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀤𑀤𑁄 𑀳𑀺 𑀫𑁂”𑁇_  


(1847.)

3195\. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀫𑀤𑁆𑀤𑀺𑀁 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀁;_  
_“𑀇𑀫𑁂 𑀢𑁂 𑀤𑀳𑀭𑀸 𑀧𑀼𑀢𑁆𑀢𑀸,_  
_𑀚𑀸𑀮𑀻 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀸 𑀘𑀼𑀪𑁄;_  
_𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁆𑀧 𑀮𑀓𑁆𑀔𑀡𑁂 𑀕𑀘𑁆𑀙,_  
_𑀫𑀬𑀁 𑀢𑁂 𑀧𑁄𑀲𑀬𑀸𑀫𑀲𑁂”𑁇_  


(1848.)

3196\. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸;_  
_“𑀧𑀺𑀬𑀸 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑀓𑀸 𑀤𑁂𑀯,_  
_𑀚𑀸𑀮𑀻 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀸 𑀘𑀼𑀪𑁄;_  
_𑀢𑁆𑀬𑀫𑁆𑀳𑀁 𑀢𑀢𑁆𑀣 𑀭𑀫𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀚𑀻𑀯𑀲𑁄𑀓𑀺𑀦𑀁”𑁇_  


(1849.)

3197\. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁄;_  
_“𑀲𑀸𑀮𑀻𑀦𑀁 𑀑𑀤𑀦𑀁 𑀪𑀼𑀢𑁆𑀯𑀸,_  
_𑀲𑀼𑀘𑀺𑀁 𑀫𑀁𑀲𑀽𑀧𑀲𑁂𑀘𑀦𑀁;_  
_𑀭𑀼𑀓𑁆𑀔𑀨𑀮𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(1850.)

3198\. _𑀪𑀼𑀢𑁆𑀯𑀸 𑀲𑀢𑀧𑀮𑁂 𑀓𑀁𑀲𑁂,_  
_𑀲𑁄𑀯𑀡𑁆𑀡𑁂 𑀲𑀢𑀭𑀸𑀚𑀺𑀓𑁂;_  
_𑀭𑀼𑀓𑁆𑀔𑀧𑀢𑁆𑀢𑁂𑀲𑀼 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀸,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(1851.)

3199\. _𑀓𑀸𑀲𑀺𑀬𑀸𑀦𑀺 𑀘 𑀥𑀸𑀭𑁂𑀢𑁆𑀯𑀸,_  
_𑀔𑁄𑀫𑀓𑁄𑀝𑀼𑀫𑁆𑀩𑀭𑀸𑀦𑀺 𑀘;_  
_𑀓𑀼𑀲𑀘𑀻𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀸,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(1852.)

3200\. _𑀯𑀬𑁆𑀳𑀸𑀳𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑀺𑀢𑁆𑀯𑀸,_  
_𑀲𑀺𑀯𑀺𑀓𑀸𑀬 𑀭𑀣𑁂𑀦 𑀘;_  
_𑀧𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀭𑀺𑀥𑀸𑀯𑀦𑁆𑀢𑀸,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(1853.)

3201\. _𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑁂 𑀲𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀦𑀺𑀯𑀸𑀢𑁂 𑀨𑀼𑀲𑀺𑀢𑀕𑁆𑀕𑀍𑀏;_  
_𑀲𑀬𑀦𑁆𑀢𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀲𑁆𑀫𑀺𑀁,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(1854.)

3202\. _𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀲𑀼 𑀲𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀕𑁄𑀦𑀓𑁂 𑀘𑀺𑀢𑁆𑀢𑀲𑀦𑁆𑀣𑀢𑁂;_  
_𑀲𑀬𑀦𑁆𑀢𑀸 𑀢𑀺𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑁂,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(1855.)

3203\. _𑀕𑀦𑁆𑀥𑀓𑁂𑀦 𑀯𑀺𑀮𑀺𑀫𑁆𑀧𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀕𑀭𑀼𑀘𑀦𑁆𑀤𑀦𑁂𑀦 𑀘;_  
_𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀸,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸𑁇_  


(1856.)

3204\. _𑀘𑀸𑀫𑀭𑀫𑁄𑀭𑀳𑀢𑁆𑀣𑁂𑀳𑀺,_  
_𑀩𑀻𑀚𑀺𑀢𑀗𑁆𑀕𑀸 𑀲𑀼𑀔𑁂𑀥𑀺𑀢𑀸;_  
_𑀨𑀼𑀝𑁆𑀞𑀸 𑀟𑀁𑀲𑁂𑀳𑀺 𑀫𑀓𑀲𑁂𑀳𑀺,_  
_𑀓𑀣𑀁 𑀓𑀸𑀳𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸”𑁇_  


(1857.)

3205\. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸;_  
_“𑀫𑀸 𑀤𑁂𑀯 𑀧𑀭𑀺𑀤𑁂𑀯𑁂𑀲𑀺,_  
_𑀫𑀸 𑀘 𑀢𑁆𑀯𑀁 𑀯𑀺𑀫𑀦𑁄 𑀅𑀳𑀼;_  
_𑀬𑀣𑀸 𑀫𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫,_  
_𑀢𑀣𑀸 𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀤𑀸𑀭𑀓𑀸”𑁇_  


(1858.)

3206\. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸;_  
_𑀲𑀺𑀯𑀺𑀫𑀕𑁆𑀕𑁂𑀦 𑀅𑀦𑁆𑀯𑁂𑀲𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑁂 𑀆𑀤𑀸𑀬 𑀮𑀓𑁆𑀔𑀡𑀸𑁇_  


(1859.)

3207\. _𑀢𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸𑀦 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀧𑀺𑀢𑀼 𑀫𑀸𑀢𑀼 𑀘 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁𑁇_  


(1860.)

3208\. _𑀘𑀢𑀼𑀯𑀸𑀳𑀺𑀁 𑀭𑀣𑀁 𑀬𑀼𑀢𑁆𑀢𑀁,_  
_𑀲𑀻𑀖𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀲𑀦𑁆𑀤𑀦𑀁;_  
_𑀆𑀤𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑀤𑀸𑀭𑀜𑁆𑀘,_  
_𑀯𑀗𑁆𑀓𑀁 𑀧𑀸𑀬𑀸𑀲𑀺 𑀧𑀩𑁆𑀩𑀢𑀁𑁇_  


(1861.)

3209\. _𑀢𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀬𑁂𑀦𑀸𑀲𑀺 𑀩𑀳𑀼𑀓𑁄 𑀚𑀦𑁄;_  
_“𑀆𑀫𑀦𑁆𑀢 𑀔𑁄 𑀢𑀁 𑀕𑀘𑁆𑀙𑀸𑀫,_  
_𑀅𑀭𑁄𑀕𑀸 𑀳𑁄𑀦𑁆𑀢𑀼 𑀜𑀸𑀢𑀬𑁄𑁇_  


(1862.)

3210\. _𑀇𑀗𑁆𑀖 𑀫𑀤𑁆𑀤𑀺 𑀦𑀺𑀲𑀸𑀫𑁂𑀳𑀺,_  
_𑀭𑀫𑁆𑀫𑀭𑀽𑀧𑀁𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀆𑀯𑀸𑀲𑀁 𑀲𑀺𑀯𑀺𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀁 𑀪𑀯𑀦𑀁 𑀫𑀫”𑁇_  


(1863.)

3211\. _𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀅𑀦𑁆𑀯𑀕𑀫𑀼𑀁,_  
_𑀢𑁂 𑀦𑀁 𑀅𑀲𑁆𑀲𑁂 𑀅𑀬𑀸𑀘𑀺𑀲𑀼𑀁;_  
_𑀬𑀸𑀘𑀺𑀢𑁄 𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀺,_  
_𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀘𑀢𑀼𑀭𑁄 𑀳𑀬𑁂𑁇_  


(1864.)

3212\. _“𑀇𑀗𑁆𑀖 𑀫𑀤𑁆𑀤𑀺 𑀦𑀺𑀲𑀸𑀫𑁂𑀳𑀺,_  
_𑀘𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑀁𑀯 𑀤𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀫𑀺𑀕𑀭𑁄𑀳𑀺𑀘𑁆𑀘𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀤𑀓𑁆𑀔𑀺𑀡𑀲𑁆𑀲𑀸 𑀯𑀳𑀦𑁆𑀢𑀺 𑀫𑀁”𑁇_  


(1865.)

3213\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀧𑀜𑁆𑀘𑀫𑁄 𑀆𑀕𑀸,_  
_𑀲𑁄 𑀢𑀁 𑀭𑀣𑀫𑀬𑀸𑀘𑀣;_  
_𑀢𑀲𑁆𑀲 𑀢𑀁 𑀬𑀸𑀘𑀺𑀢𑁄𑀤𑀸𑀲𑀺,_  
_𑀦 𑀘𑀲𑁆𑀲𑀼𑀧𑀳𑀢𑁄 𑀫𑀦𑁄𑁇_  


(1866.)

3214\. _𑀢𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀑𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑀁 𑀚𑀦𑀁;_  
_𑀅𑀲𑁆𑀲𑀸𑀲𑀬𑀺 𑀅𑀲𑁆𑀲𑀭𑀣𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀦𑁂𑀲𑀺𑀦𑁄𑁇_  


(1867.)

3215\. _“𑀢𑁆𑀯𑀁 𑀫𑀤𑁆𑀤𑀺 𑀓𑀡𑁆𑀳𑀁 𑀕𑀡𑁆𑀳𑀸𑀳𑀺,_  
_𑀮𑀳𑀼 𑀏𑀲𑀸 𑀓𑀦𑀺𑀝𑁆𑀞𑀺𑀓𑀸;_  
_𑀅𑀳𑀁 𑀚𑀸𑀮𑀺𑀁 𑀕𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀕𑀭𑀼𑀓𑁄 𑀪𑀸𑀢𑀺𑀓𑁄 𑀳𑀺 𑀲𑁄”𑁇_  


(1868.)

3216\. _𑀭𑀸𑀚𑀸 𑀓𑀼𑀫𑀸𑀭𑀫𑀸𑀤𑀸𑀬,_
