# 𑁨𑁨𑁇𑁧𑁇𑁧𑁦𑁇𑁧𑁨 𑀙𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀫𑁆𑀫

### 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀚𑀸𑀢𑀓

### 𑀙𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀫𑁆𑀫

(2368.)

3727\. _𑀢𑁂𑀲𑀁 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁,_  
_𑀪𑀻𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀅𑀳𑀼;_  
_𑀧𑀩𑁆𑀩𑀢𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀪𑀻𑀢𑁄 𑀲𑁂𑀦𑀁 𑀉𑀤𑀺𑀓𑁆𑀔𑀢𑀺𑁇_  


(2369.)

3728\. _“𑀇𑀗𑁆𑀖 𑀫𑀤𑁆𑀤𑀺 𑀦𑀺𑀲𑀸𑀫𑁂𑀳𑀺,_  
_𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑁄 𑀬𑀸𑀤𑀺𑀲𑁄 𑀯𑀦𑁂;_  
_𑀆𑀚𑀸𑀦𑀻𑀬𑀸 𑀳𑀲𑀺𑀬𑀦𑁆𑀢𑀺,_  
_𑀥𑀚𑀕𑁆𑀕𑀸𑀦𑀺 𑀘 𑀤𑀺𑀲𑁆𑀲𑀭𑁂𑁇_  


(2370.)

3729\. _𑀇𑀫𑁂 𑀦𑀽𑀦 𑀅𑀭𑀜𑁆𑀜𑀲𑁆𑀫𑀺𑀁,_  
_𑀫𑀺𑀕𑀲𑀗𑁆𑀖𑀸𑀦𑀺 𑀮𑀼𑀤𑁆𑀤𑀓𑀸;_  
_𑀯𑀸𑀕𑀼𑀭𑀸𑀳𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀧𑁆𑀧,_  
_𑀲𑁄𑀩𑁆𑀪𑀁 𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀯𑀤𑁂;_  
_𑀯𑀺𑀓𑁆𑀓𑁄𑀲𑀫𑀸𑀦𑀸 𑀢𑀺𑀩𑁆𑀩𑀸𑀳𑀺,_  
_𑀳𑀦𑁆𑀢𑀺 𑀦𑁂𑀲𑀁 𑀯𑀭𑀁 𑀯𑀭𑀁𑁇_  


(2371.)

3730\. _𑀬𑀣𑀸 𑀫𑀬𑀁 𑀅𑀤𑀽𑀲𑀓𑀸,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀅𑀯𑀭𑀼𑀤𑁆𑀥𑀓𑀸;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀢𑁆𑀢𑀁 𑀕𑀢𑀸,_  
_𑀧𑀲𑁆𑀲 𑀤𑀼𑀩𑁆𑀩𑀮𑀖𑀸𑀢𑀓𑀁”𑁇_  


(2372.)

3731\. _“𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀦𑀧𑁆𑀧𑀲𑀸𑀳𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀅𑀕𑁆𑀕𑀻𑀯 𑀉𑀤𑀓𑀡𑁆𑀡𑀯𑁂;_  
_𑀢𑀤𑁂𑀯 𑀢𑁆𑀯𑀁 𑀯𑀺𑀘𑀺𑀦𑁆𑀢𑁂𑀳𑀺,_  
_𑀅𑀧𑀺 𑀲𑁄𑀢𑁆𑀣𑀺 𑀇𑀢𑁄 𑀲𑀺𑀬𑀸”𑁇_  


(2373.)

3732\. _𑀢𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸𑀦 𑀧𑀩𑁆𑀩𑀢𑀸;_  
_𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀸𑀬𑀁,_  
_𑀤𑀍𑀳𑀁 𑀓𑀢𑁆𑀯𑀸𑀦 𑀫𑀸𑀦𑀲𑀁𑁇_  


(2374.)

3733\. _𑀦𑀺𑀯𑀢𑁆𑀢𑀬𑀺𑀢𑁆𑀯𑀸𑀦 𑀭𑀣𑀁,_  
_𑀯𑀼𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦 𑀲𑁂𑀦𑀺𑀬𑁄;_  
_𑀏𑀓𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀁,_  
_𑀧𑀺𑀢𑀸 𑀧𑀼𑀢𑁆𑀢𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(2375.)

3734\. _𑀳𑀢𑁆𑀣𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀢𑁄 𑀑𑀭𑀼𑀬𑁆𑀳,_  
_𑀏𑀓𑀁𑀲𑁄 𑀧𑀜𑁆𑀚𑀮𑀻𑀓𑀢𑁄;_  
_𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑁄 𑀅𑀫𑀘𑁆𑀘𑁂𑀳𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑀁 𑀲𑀺𑀜𑁆𑀘𑀺𑀢𑀼𑀫𑀸𑀕𑀫𑀺𑁇_  


(2376.)

3735\. _𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲 𑀓𑀼𑀫𑀸𑀭𑀁 𑀲𑁄,_  
_𑀭𑀫𑁆𑀫𑀭𑀽𑀧𑀁 𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀧𑀡𑁆𑀡𑀲𑀸𑀮𑀸𑀬𑀁,_  
_𑀛𑀸𑀬𑀦𑁆𑀢𑀁 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑀁𑁇_  


(2377.)

3736\. _𑀢𑀜𑁆𑀘 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀦𑁆𑀢𑀁,_  
_𑀧𑀺𑀢𑀭𑀁 𑀧𑀼𑀢𑁆𑀢𑀕𑀺𑀤𑁆𑀥𑀺𑀦𑀁;_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀘 𑀫𑀤𑁆𑀤𑀻 𑀘,_  
_𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀯𑀦𑁆𑀤𑀺𑀲𑀼𑀁𑁇_  


(2378.)

3737\. _𑀫𑀤𑁆𑀤𑀻 𑀘 𑀲𑀺𑀭𑀲𑀸 𑀧𑀸𑀤𑁂,_  
_𑀲𑀲𑀼𑀭𑀲𑁆𑀲𑀸𑀪𑀺𑀯𑀸𑀤𑀬𑀺;_  
_“𑀫𑀤𑁆𑀤𑀻 𑀅𑀳𑀜𑁆𑀳𑀺 𑀢𑁂 𑀤𑁂𑀯,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀫𑀺 𑀢𑁂 𑀲𑀼𑀡𑁆𑀳𑀸”;_  
_𑀢𑁂 𑀲𑀼 𑀢𑀢𑁆𑀣 𑀧𑀮𑀺𑀲𑀚𑁆𑀚,_  
_𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑀭𑀺𑀫𑀚𑁆𑀚𑀣𑁇_  


(2379.)

3738\. _“𑀓𑀘𑁆𑀘𑀺 𑀯𑁄 𑀓𑀼𑀲𑀮𑀁 𑀧𑀼𑀢𑁆𑀢,_  
_𑀓𑀘𑁆𑀘𑀺 𑀧𑀼𑀢𑁆𑀢 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀓𑀘𑁆𑀘𑀺 𑀉𑀜𑁆𑀙𑁂𑀦 𑀬𑀸𑀧𑁂𑀣,_  
_𑀓𑀘𑁆𑀘𑀺 𑀫𑀽𑀮𑀨𑀮𑀸 𑀩𑀳𑀽𑁇_  


(2380.)

3739\. _𑀓𑀘𑁆𑀘𑀺 𑀟𑀁𑀲𑀸 𑀫𑀓𑀲𑀸 𑀘,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀭𑀻𑀲𑀧𑀸;_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀓𑀘𑁆𑀘𑀺 𑀳𑀺𑀁𑀲𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(2381.)

3740\. _“𑀅𑀢𑁆𑀣𑀺 𑀦𑁄 𑀚𑀻𑀯𑀺𑀓𑀸 𑀤𑁂𑀯,_  
_𑀲𑀸 𑀘 𑀬𑀸𑀤𑀺𑀲𑀓𑀻𑀤𑀺𑀲𑀸;_  
_𑀓𑀲𑀺𑀭𑀸 𑀚𑀻𑀯𑀺𑀓𑀸 𑀳𑁄𑀫,_  
_𑀉𑀜𑁆𑀙𑀸𑀘𑀭𑀺𑀬𑀸𑀬 𑀚𑀻𑀯𑀺𑀢𑀁𑁇_  


(2382.)

3741\. _𑀅𑀦𑀺𑀤𑁆𑀥𑀺𑀦𑀁 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀤𑀫𑁂𑀢𑀲𑁆𑀲𑀁𑀯 𑀲𑀸𑀭𑀣𑀺;_  
_𑀢𑁆𑀬𑀫𑁆𑀳𑀸 𑀅𑀦𑀺𑀤𑁆𑀥𑀺𑀓𑀸 𑀤𑀦𑁆𑀢𑀸,_  
_𑀅𑀲𑀫𑀺𑀤𑁆𑀥𑀺 𑀤𑀫𑁂𑀢𑀺 𑀦𑁄𑁇_  


(2383.)

3742\. _𑀅𑀧𑀺 𑀦𑁄 𑀓𑀺𑀲𑀸𑀦𑀺 𑀫𑀁𑀲𑀸𑀦𑀺,_  
_𑀧𑀺𑀢𑀼 𑀫𑀸𑀢𑀼 𑀅𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀅𑀯𑀭𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀅𑀭𑀜𑁆𑀜𑁂 𑀚𑀻𑀯𑀲𑁄𑀓𑀺𑀦𑀁𑁇_  


(2384.)

3743\. _𑀬𑁂𑀧𑀺 𑀢𑁂 𑀲𑀺𑀯𑀺𑀲𑁂𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀤𑀸𑀬𑀸𑀤𑀸𑀧𑀢𑁆𑀢𑀫𑀸𑀦𑀲𑀸;_  
_𑀚𑀸𑀮𑀻 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀸 𑀘𑀼𑀪𑁄,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀯𑀲𑀸𑀦𑀼𑀕𑀸;_  
_𑀅𑀘𑁆𑀘𑀸𑀬𑀺𑀓𑀲𑁆𑀲 𑀮𑀼𑀤𑁆𑀤𑀲𑁆𑀲,_  
_𑀬𑁄 𑀦𑁂 𑀕𑀸𑀯𑁄𑀯 𑀲𑀼𑀫𑁆𑀪𑀢𑀺𑁇_  


(2385.)

3744\. _𑀢𑁂 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀼𑀢𑁆𑀢𑁂,_  
_𑀬𑀤𑀺 𑀚𑀸𑀦𑀸𑀣 𑀲𑀁𑀲𑀣;_  
_𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀸𑀣 𑀦𑁄 𑀔𑀺𑀧𑁆𑀧𑀁,_  
_𑀲𑀧𑁆𑀧𑀤𑀝𑁆𑀞𑀁𑀯 𑀫𑀸𑀡𑀯𑀁”𑁇_  


(2386.)

3745\. _“𑀉𑀪𑁄 𑀓𑀼𑀫𑀸𑀭𑀸 𑀦𑀺𑀓𑁆𑀓𑀻𑀢𑀸,_  
_𑀚𑀸𑀮𑀻 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀸 𑀘𑀼𑀪𑁄;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀥𑀦𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀧𑀼𑀢𑁆𑀢 𑀫𑀸 𑀪𑀸𑀬𑀺 𑀅𑀲𑁆𑀲𑀲”𑁇_  


(2387.)

3746\. _“𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀢𑀸𑀢 𑀓𑀼𑀲𑀮𑀁,_  
_𑀓𑀘𑁆𑀘𑀺 𑀢𑀸𑀢 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀢𑀸𑀢 𑀫𑁂 𑀫𑀸𑀢𑀼,_  
_𑀘𑀓𑁆𑀔𑀼 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺”𑁇_  


(2388.)

3747\. _“𑀓𑀼𑀲𑀮𑀜𑁆𑀘𑁂𑀯 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢,_  
_𑀅𑀣𑁄 𑀧𑀼𑀢𑁆𑀢 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀅𑀣𑁄 𑀘 𑀧𑀼𑀢𑁆𑀢 𑀢𑁂 𑀫𑀸𑀢𑀼,_  
_𑀘𑀓𑁆𑀔𑀼 𑀦 𑀧𑀭𑀺𑀳𑀸𑀬𑀢𑀺”𑁇_  


(2389.)

3748\. _“𑀓𑀘𑁆𑀘𑀺 𑀅𑀭𑁄𑀕𑀁 𑀬𑁄𑀕𑁆𑀕𑀁 𑀢𑁂,_  
_𑀓𑀘𑁆𑀘𑀺 𑀯𑀳𑀢𑀺 𑀯𑀸𑀳𑀦𑀁;_  
_𑀓𑀘𑁆𑀘𑀺 𑀨𑀻𑀢𑁄 𑀚𑀦𑀧𑀤𑁄,_  
_𑀓𑀘𑁆𑀘𑀺 𑀯𑀼𑀝𑁆𑀞𑀺 𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(2390.)

3749\. _“𑀅𑀣𑁄 𑀅𑀭𑁄𑀕𑀁 𑀬𑁄𑀕𑁆𑀕𑀁 𑀫𑁂,_  
_𑀅𑀣𑁄 𑀯𑀳𑀢𑀺 𑀯𑀸𑀳𑀦𑀁;_  
_𑀅𑀣𑁄 𑀨𑀻𑀢𑁄 𑀚𑀦𑀧𑀤𑁄,_  
_𑀅𑀣𑁄 𑀯𑀼𑀝𑁆𑀞𑀺 𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(2391.)

3750\. _𑀇𑀘𑁆𑀘𑁂𑀯𑀁 𑀫𑀦𑁆𑀢𑀬𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀫𑀸𑀢𑀸 𑀦𑁂𑀲𑀁 𑀅𑀤𑀺𑀲𑁆𑀲𑀣;_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻 𑀕𑀺𑀭𑀺𑀤𑁆𑀯𑀸𑀭𑁂,_  
_𑀧𑀢𑁆𑀢𑀺𑀓𑀸 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑀸𑁇_  


(2392.)

3751\. _𑀢𑀜𑁆𑀘 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀦𑁆𑀢𑀺𑀁,_  
_𑀫𑀸𑀢𑀭𑀁 𑀧𑀼𑀢𑁆𑀢𑀕𑀺𑀤𑁆𑀥𑀺𑀦𑀺𑀁;_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀘 𑀫𑀤𑁆𑀤𑀻 𑀘,_  
_𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀯𑀦𑁆𑀤𑀺𑀲𑀼𑀁𑁇_  


(2393.)

3752\. _𑀫𑀤𑁆𑀤𑀻 𑀘 𑀲𑀺𑀭𑀲𑀸 𑀧𑀸𑀤𑁂,_  
_𑀲𑀲𑁆𑀲𑀼𑀬𑀸 𑀅𑀪𑀺𑀯𑀸𑀤𑀬𑀺;_  
_“𑀫𑀤𑁆𑀤𑀻 𑀅𑀳𑀜𑁆𑀳𑀺 𑀢𑁂 𑀅𑀬𑁆𑀬𑁂,_  
_𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀫𑀺 𑀢𑁂 𑀲𑀼𑀡𑁆𑀳𑀸”𑁇_  


(2394.)

3753\. _𑀫𑀤𑁆𑀤𑀺𑀜𑁆𑀘 𑀧𑀼𑀢𑁆𑀢𑀓𑀸 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀤𑀽𑀭𑀢𑁄 𑀲𑁄𑀢𑁆𑀣𑀺𑀫𑀸𑀕𑀢𑀸;_  
_𑀓𑀦𑁆𑀤𑀦𑁆𑀢𑀸 𑀫𑀪𑀺𑀥𑀸𑀯𑀺𑀁𑀲𑀼,_  
_𑀯𑀘𑁆𑀙𑀩𑀸𑀮𑀸𑀯 𑀫𑀸𑀢𑀭𑀁𑁇_  


(2395.)

3754\. _𑀫𑀤𑁆𑀤𑀻 𑀘 𑀧𑀼𑀢𑁆𑀢𑀓𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀤𑀽𑀭𑀢𑁄 𑀲𑁄𑀢𑁆𑀣𑀺𑀫𑀸𑀕𑀢𑁂;_  
_𑀯𑀸𑀭𑀼𑀡𑀻𑀯 𑀧𑀯𑁂𑀥𑁂𑀦𑁆𑀢𑀻,_  
_𑀣𑀦𑀥𑀸𑀭𑀸𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑀣𑁇_  


(2396.)

3755\. _𑀲𑀫𑀸𑀕𑀢𑀸𑀦𑀁 𑀜𑀸𑀢𑀻𑀦𑀁,_  
_𑀫𑀳𑀸𑀖𑁄𑀲𑁄 𑀅𑀚𑀸𑀬𑀣;_  
_𑀧𑀩𑁆𑀩𑀢𑀸 𑀲𑀫𑀦𑀸𑀤𑀺𑀁𑀲𑀼,_  
_𑀫𑀳𑀻 𑀧𑀓𑀫𑁆𑀧𑀺𑀢𑀸 𑀅𑀳𑀼𑁇_  


(2397.)

3756\. _𑀯𑀼𑀝𑁆𑀞𑀺𑀥𑀸𑀭𑀁 𑀧𑀯𑀢𑁆𑀢𑁂𑀦𑁆𑀢𑁄,_  
_𑀤𑁂𑀯𑁄 𑀧𑀸𑀯𑀲𑁆𑀲𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀅𑀣 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀜𑀸𑀢𑀻𑀳𑀺 𑀲𑀫𑀕𑀘𑁆𑀙𑀣𑁇_  


(2398.)

3757\. _𑀦𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑀼𑀡𑀺𑀲𑀸 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀭𑀸𑀚𑀸 𑀤𑁂𑀯𑀻 𑀘 𑀏𑀓𑀢𑁄;_  
_𑀬𑀤𑀸 𑀲𑀫𑀸𑀕𑀢𑀸 𑀆𑀲𑀼𑀁,_  
_𑀢𑀤𑀸𑀲𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁𑁇_  


(2399.)

3758\. _𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑀸 𑀢𑀲𑁆𑀲 𑀬𑀸𑀘𑀦𑁆𑀢𑀺,_  
_𑀭𑁄𑀤𑀦𑁆𑀢𑀸 𑀪𑁂𑀭𑀯𑁂 𑀯𑀦𑁂;_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀜𑁆𑀘 𑀫𑀤𑁆𑀤𑀺𑀜𑁆𑀘,_  
_𑀲𑀩𑁆𑀩𑁂 𑀭𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀢𑁆𑀯𑀁 𑀦𑁄𑀲𑀺 𑀇𑀲𑁆𑀲𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀣 𑀦𑁄 𑀉𑀪𑁄𑁇_  


---

3759\. 𑀙𑀔𑀢𑁆𑀢𑀺𑀬𑀓𑀫𑁆𑀫𑀁 𑀦𑀸𑀫𑁇



(2400.)

3760\. _“𑀥𑀫𑁆𑀫𑁂𑀦 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀦𑁆𑀢𑀁,_  
_𑀭𑀝𑁆𑀞𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀺𑀢𑁆𑀣 𑀫𑀁;_  
_𑀢𑁆𑀯𑀜𑁆𑀘 𑀚𑀸𑀦𑀧𑀤𑀸 𑀘𑁂𑀯,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸”𑁇_  


(2401.)

3761\. _“𑀤𑀼𑀓𑁆𑀓𑀝𑀜𑁆𑀘 𑀳𑀺 𑀦𑁄 𑀧𑀼𑀢𑁆𑀢,_  
_𑀪𑀽𑀦𑀳𑀘𑁆𑀘𑀁 𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀬𑁄𑀳𑀁 𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀸,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀺𑀫𑀤𑀽𑀲𑀓𑀁𑁇_  


(2402.)

3762\. _𑀬𑁂𑀦 𑀓𑁂𑀦𑀘𑀺 𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀧𑀺𑀢𑀼 𑀤𑀼𑀓𑁆𑀔𑀁 𑀉𑀤𑀩𑁆𑀩𑀳𑁂;_  
_𑀫𑀸𑀢𑀼 𑀪𑀕𑀺𑀦𑀺𑀬𑀸 𑀘𑀸𑀧𑀺,_  
_𑀅𑀧𑀺 𑀧𑀸𑀡𑁂𑀳𑀺 𑀅𑀢𑁆𑀢𑀦𑁄”𑁇_  


(2403.)

3763\. _𑀢𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀧𑀯𑀸𑀳𑀬𑀺;_  
_𑀭𑀚𑁄𑀚𑀮𑁆𑀮𑀁 𑀧𑀯𑀸𑀳𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀗𑁆𑀔𑀯𑀡𑁆𑀡𑀁 𑀅𑀥𑀸𑀭𑀬𑀺𑁇_  


(2404.)

3764\. _𑀲𑀻𑀲𑀁 𑀦𑁆𑀳𑀸𑀢𑁄 𑀲𑀼𑀘𑀺𑀯𑀢𑁆𑀣𑁄,_  
_𑀲𑀩𑁆𑀩𑀸𑀪𑀭𑀡𑀪𑀽𑀲𑀺𑀢𑁄;_  
_𑀧𑀘𑁆𑀘𑀬𑀁 𑀦𑀸𑀕𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀔𑀕𑁆𑀕𑀁 𑀩𑀦𑁆𑀥𑀺 𑀧𑀭𑀦𑁆𑀢𑀧𑀁𑁇_  


(2405.)

3765\. _𑀢𑀢𑁄 𑀲𑀝𑁆𑀞𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀬𑁄𑀥𑀺𑀦𑁄 𑀘𑀸𑀭𑀼𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀲𑀳𑀚𑀸𑀢𑀸 𑀧𑀓𑀺𑀭𑀺𑀁𑀲𑀼,_  
_𑀦𑀦𑁆𑀤𑀬𑀦𑁆𑀢𑀸 𑀭𑀣𑁂𑀲𑀪𑀁𑁇_  


(2406.)

3766\. _𑀢𑀢𑁄 𑀫𑀤𑁆𑀤𑀺𑀫𑁆𑀧𑀺 𑀦𑁆𑀳𑀸𑀧𑁂𑀲𑀼𑀁,_  
_𑀲𑀺𑀯𑀺𑀓𑀜𑁆𑀜𑀸 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_“𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀢𑀁 𑀧𑀸𑀮𑁂𑀢𑀼,_  
_𑀚𑀸𑀮𑀻 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀸 𑀘𑀼𑀪𑁄;_  
_𑀅𑀣𑁄𑀧𑀺 𑀢𑀁 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀲𑀜𑁆𑀚𑀬𑁄 𑀅𑀪𑀺𑀭𑀓𑁆𑀔𑀢𑀼”𑁇_  


(2407.)

3767\. _𑀇𑀤𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀁 𑀮𑀤𑁆𑀥𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀓𑁆𑀮𑁂𑀲𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀆𑀦𑀦𑁆𑀤𑀺𑀬𑀁 𑀆𑀘𑀭𑀺𑀁𑀲𑀼,_  
_𑀭𑀫𑀡𑀻𑀬𑁂 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑁂𑁇_  


(2408.)

3768\. _𑀇𑀤𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀁 𑀮𑀤𑁆𑀥𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀓𑁆𑀮𑁂𑀲𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀆𑀦𑀦𑁆𑀤𑀺 𑀯𑀺𑀢𑁆𑀢𑀸 𑀲𑀼𑀫𑀦𑀸,_  
_𑀧𑀼𑀢𑁆𑀢𑁂 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀮𑀓𑁆𑀔𑀡𑀸𑁇_  


(2409.)

3769\. _𑀇𑀤𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀬𑀁 𑀮𑀤𑁆𑀥𑀸,_  
_𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀗𑁆𑀓𑁆𑀮𑁂𑀲𑀫𑀢𑁆𑀢𑀦𑁄;_  
_𑀆𑀦𑀦𑁆𑀤𑀺 𑀯𑀺𑀢𑁆𑀢𑀸 𑀧𑀢𑀻𑀢𑀸,_  
_𑀲𑀳 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀮𑀓𑁆𑀔𑀡𑀸𑁇_  


(2410.)

3770\. _“𑀏𑀓𑀪𑀢𑁆𑀢𑀸 𑀧𑀼𑀭𑁂 𑀆𑀲𑀺𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀣𑀡𑁆𑀟𑀺𑀮𑀲𑀸𑀬𑀺𑀦𑀻;_  
_𑀇𑀢𑀺 𑀫𑁂𑀢𑀁 𑀯𑀢𑀁 𑀆𑀲𑀺,_  
_𑀢𑀼𑀫𑁆𑀳𑀁 𑀓𑀸𑀫𑀸 𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑀓𑀸𑁇_  


(2411.)

3771\. _𑀢𑀁 𑀫𑁂 𑀯𑀢𑀁 𑀲𑀫𑀺𑀤𑁆𑀥𑀚𑁆𑀚,_  
_𑀢𑀼𑀫𑁆𑀳𑁂 𑀲𑀗𑁆𑀕𑀫𑁆𑀫 𑀧𑀼𑀢𑁆𑀢𑀓𑀸;_  
_𑀫𑀸𑀢𑀼𑀚𑀫𑁆𑀧𑀺 𑀢𑀁 𑀧𑀸𑀮𑁂𑀢𑀼,_  
_𑀧𑀺𑀢𑀼𑀚𑀫𑁆𑀧𑀺 𑀘 𑀧𑀼𑀢𑁆𑀢𑀓;_  
_𑀅𑀣𑁄𑀧𑀺 𑀢𑀁 𑀫𑀳𑀸𑀭𑀸𑀚𑀸,_  
_𑀲𑀜𑁆𑀚𑀬𑁄 𑀅𑀪𑀺𑀭𑀓𑁆𑀔𑀢𑀼𑁇_  


(2412.)

3772\. _𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺𑀢𑁆𑀣𑀺 𑀓𑀢𑀁 𑀧𑀼𑀜𑁆𑀜𑀁,_  
_𑀫𑀬𑁆𑀳𑀜𑁆𑀘𑁂𑀯 𑀧𑀺𑀢𑀼𑀘𑁆𑀘 𑀢𑁂;_  
_𑀲𑀩𑁆𑀩𑁂𑀦 𑀢𑁂𑀦 𑀓𑀼𑀲𑀮𑁂𑀦,_  
_𑀅𑀚𑀭𑁄 𑀅𑀫𑀭𑁄 𑀪𑀯”𑁇_  


(2413.)

3773\. _𑀓𑀧𑁆𑀧𑀸𑀲𑀺𑀓𑀜𑁆𑀘 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀁,_  
_𑀔𑁄𑀫𑀓𑁄𑀝𑀼𑀫𑁆𑀩𑀭𑀸𑀦𑀺 𑀘;_  
_𑀲𑀲𑁆𑀲𑀼 𑀲𑀼𑀡𑁆𑀳𑀸𑀬 𑀧𑀸𑀳𑁂𑀲𑀺,_  
_𑀬𑁂𑀳𑀺 𑀫𑀤𑁆𑀤𑀻 𑀅𑀲𑁄𑀪𑀣𑁇_  


(2414.)

3774\. _𑀢𑀢𑁄 𑀳𑁂𑀫𑀜𑁆𑀘 𑀓𑀸𑀬𑀽𑀭𑀁,_  
_𑀕𑀻𑀯𑁂𑀬𑁆𑀬𑀁 𑀭𑀢𑀦𑀸𑀫𑀬𑀁;_  
_𑀲𑀲𑁆𑀲𑀼 𑀲𑀼𑀡𑁆𑀳𑀸𑀬 𑀧𑀸𑀳𑁂𑀲𑀺,_  
_𑀬𑁂𑀳𑀺 𑀫𑀤𑁆𑀤𑀻 𑀅𑀲𑁄𑀪𑀣𑁇_  


(2415.)

3775\. _𑀢𑀢𑁄 𑀳𑁂𑀫𑀜𑁆𑀘 𑀓𑀸𑀬𑀽𑀭𑀁,_  
_𑀅𑀗𑁆𑀕𑀤𑀁 𑀫𑀡𑀺𑀫𑁂𑀔𑀮𑀁;_  
_𑀲𑀲𑁆𑀲𑀼 𑀲𑀼𑀡𑁆𑀳𑀸𑀬 𑀧𑀸𑀳𑁂𑀲𑀺,_  
_𑀬𑁂𑀳𑀺 𑀫𑀤𑁆𑀤𑀻 𑀅𑀲𑁄𑀪𑀣𑁇_  


(2416.)

3776\. _𑀉𑀡𑁆𑀡𑀢𑀁 𑀫𑀼𑀔𑀨𑀼𑀮𑁆𑀮𑀜𑁆𑀘,_  
_𑀦𑀸𑀦𑀸𑀭𑀢𑁆𑀢𑁂 𑀘 𑀫𑀸𑀡𑀺𑀓𑁂;_  
_𑀲𑀲𑁆𑀲𑀼 𑀲𑀼𑀡𑁆𑀳𑀸𑀬 𑀧𑀸𑀳𑁂𑀲𑀺,_  
_𑀬𑁂𑀳𑀺 𑀫𑀤𑁆𑀤𑀻 𑀅𑀲𑁄𑀪𑀣𑁇_  


(2417.)

3777\. _𑀉𑀕𑁆𑀕𑀢𑁆𑀣𑀦𑀁 𑀕𑀺𑀗𑁆𑀕𑀫𑀓𑀁,_  
_𑀫𑁂𑀔𑀮𑀁 𑀧𑀸𑀝𑀺𑀧𑀸𑀤𑀓𑀁;_  
_𑀲𑀲𑁆𑀲𑀼 𑀲𑀼𑀡𑁆𑀳𑀸𑀬 𑀧𑀸𑀳𑁂𑀲𑀺,_  
_𑀬𑁂𑀳𑀺 𑀫𑀤𑁆𑀤𑀻 𑀅𑀲𑁄𑀪𑀣𑁇_  


(2418.)

3778\. _𑀲𑀼𑀢𑁆𑀢𑀜𑁆𑀘 𑀲𑀼𑀢𑁆𑀢𑀯𑀚𑁆𑀚𑀜𑁆𑀘,_  
_𑀉𑀧𑀦𑀺𑀚𑁆𑀛𑀸𑀬 𑀲𑁂𑀬𑁆𑀬𑀲𑀺;_  
_𑀅𑀲𑁄𑀪𑀣 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀤𑁂𑀯𑀓𑀜𑁆𑀜𑀸𑀯 𑀦𑀦𑁆𑀤𑀦𑁂𑁇_  


(2419.)

3779\. _𑀲𑀻𑀲𑀁 𑀦𑁆𑀳𑀸𑀢𑀸 𑀲𑀼𑀘𑀺𑀯𑀢𑁆𑀣𑀸,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀸;_  
_𑀅𑀲𑁄𑀪𑀣 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑁂𑀯 𑀅𑀘𑁆𑀙𑀭𑀸𑁇_  


(2420.)

3780\. _𑀓𑀤𑀮𑀻𑀯 𑀯𑀸𑀢𑀘𑁆𑀙𑀼𑀧𑀺𑀢𑀸,_  
_𑀚𑀸𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀮𑀢𑀸𑀯𑀦𑁂;_  
_𑀤𑀦𑁆𑀢𑀸𑀯𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻 𑀅𑀲𑁄𑀪𑀣𑁇_  


(2421.)

3781\. _𑀲𑀓𑀼𑀡𑀻 𑀫𑀸𑀦𑀼𑀲𑀺𑀦𑀻𑀯,_  
_𑀚𑀸𑀢𑀸 𑀘𑀺𑀢𑁆𑀢𑀧𑀢𑁆𑀢𑀸 𑀧𑀢𑀻;_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀧𑀓𑁆𑀓𑀩𑀺𑀫𑁆𑀩𑁄𑀝𑁆𑀞𑀻,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻 𑀅𑀲𑁄𑀪𑀣𑁇_  


(2422.)

3782\. _𑀢𑀲𑁆𑀲𑀸 𑀘 𑀦𑀸𑀕𑀫𑀸𑀦𑁂𑀲𑀼𑀁,_  
_𑀦𑀸𑀢𑀺𑀩𑀤𑁆𑀥𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑀁;_  
_𑀲𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀫𑀁 𑀲𑀭𑀓𑁆𑀔𑀫𑀁,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑀁 𑀉𑀭𑀽𑀍𑀳𑀯𑀁𑁇_  


(2423.)

3783\. _𑀲𑀸 𑀫𑀤𑁆𑀤𑀻 𑀦𑀸𑀕𑀫𑀸𑀭𑀼𑀳𑀺,_  
_𑀦𑀸𑀢𑀺𑀯𑀤𑁆𑀥𑀁𑀯 𑀓𑀼𑀜𑁆𑀚𑀭𑀁;_  
_𑀲𑀢𑁆𑀢𑀺𑀓𑁆𑀔𑀫𑀁 𑀲𑀭𑀓𑁆𑀔𑀫𑀁,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑀁 𑀉𑀭𑀽𑀍𑀳𑀯𑀁𑁇_  


(2424.)

3784\. _𑀲𑀩𑁆𑀩𑀫𑁆𑀳𑀺 𑀢𑀁𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀫𑀺𑀕𑀸 𑀅𑀳𑀼𑀁;_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀲𑁆𑀲 𑀢𑁂𑀚𑁂𑀦,_  
_𑀦𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀯𑀺𑀳𑁂𑀞𑀬𑀼𑀁𑁇_  


(2425.)

3785\. _𑀲𑀩𑁆𑀩𑀫𑁆𑀳𑀺 𑀢𑀁𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀤𑀺𑀚𑀸 𑀅𑀳𑀼𑀁;_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀲𑁆𑀲 𑀢𑁂𑀚𑁂𑀦,_  
_𑀦𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀯𑀺𑀳𑁂𑀞𑀬𑀼𑀁𑁇_  


(2426.)

3786\. _𑀲𑀩𑁆𑀩𑀫𑁆𑀳𑀺 𑀢𑀁𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀫𑀺𑀕𑀸 𑀅𑀳𑀼𑀁;_  
_𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2427.)

3787\. _𑀲𑀩𑁆𑀩𑀫𑁆𑀳𑀺 𑀢𑀁𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀤𑀺𑀚𑀸 𑀅𑀳𑀼𑀁;_  
_𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2428.)

3788\. _𑀲𑀩𑁆𑀩𑀫𑁆𑀳𑀺 𑀢𑀁𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀫𑀺𑀕𑀸 𑀅𑀳𑀼𑀁;_  
_𑀦𑀸𑀲𑁆𑀲𑀼 𑀫𑀜𑁆𑀚𑀽 𑀦𑀺𑀓𑀽𑀚𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2429.)

3789\. _𑀲𑀩𑁆𑀩𑀫𑁆𑀳𑀺 𑀢𑀁𑀅𑀭𑀜𑁆𑀜𑀫𑁆𑀳𑀺,_  
_𑀬𑀸𑀯𑀦𑁆𑀢𑁂𑀢𑁆𑀣 𑀤𑀺𑀚𑀸 𑀅𑀳𑀼𑀁;_  
_𑀦𑀸𑀲𑁆𑀲𑀼 𑀫𑀜𑁆𑀚𑀽 𑀦𑀺𑀓𑀽𑀚𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2430.)

3790\. _𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑁄 𑀭𑀸𑀚𑀫𑀕𑁆𑀕𑁄,_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀼𑀧𑁆𑀨𑀲𑀦𑁆𑀣𑀢𑁄;_  
_𑀯𑀲𑀺 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀬𑀢𑁆𑀣,_  
_𑀬𑀸𑀯𑀢𑀸𑀯 𑀚𑁂𑀢𑀼𑀢𑁆𑀢𑀭𑀸𑁇_  


(2431.)

3791\. _𑀢𑀢𑁄 𑀲𑀝𑁆𑀞𑀺𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀬𑁄𑀥𑀺𑀦𑁄 𑀘𑀸𑀭𑀼𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀺𑀭𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2432.)

3792\. _𑀑𑀭𑁄𑀥𑀸 𑀘 𑀓𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀺𑀭𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2433.)

3793\. _𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀺𑀭𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2434.)

3794\. _𑀲𑀫𑀸𑀕𑀢𑀸 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀧𑀭𑀺𑀓𑀺𑀭𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2435.)

3795\. _𑀓𑀭𑁄𑀝𑀺𑀬𑀸 𑀘𑀫𑁆𑀫𑀥𑀭𑀸,_  
_𑀇𑀮𑁆𑀮𑀻𑀳𑀢𑁆𑀣𑀸 𑀲𑀼𑀯𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀧𑀼𑀭𑀢𑁄 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀬𑀸𑀢𑀫𑁆𑀳𑀺;_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2436.)

3796\. _𑀢𑁂 𑀧𑀸𑀯𑀺𑀲𑀼𑀁 𑀧𑀼𑀭𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀫𑀳𑀸𑀧𑀸𑀓𑀸𑀭𑀢𑁄𑀭𑀡𑀁;_  
_𑀉𑀧𑁂𑀢𑀁 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑁂𑀳𑀺,_  
_𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑁂𑀳𑀺 𑀘𑀽𑀪𑀬𑀁𑁇_  


(2437.)

3797\. _𑀯𑀺𑀢𑁆𑀢𑀸 𑀚𑀸𑀦𑀧𑀤𑀸 𑀆𑀲𑀼𑀁,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀓𑀼𑀫𑀸𑀭𑀫𑁆𑀳𑀺,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2438.)

3798\. _𑀘𑁂𑀮𑀼𑀓𑁆𑀔𑁂𑀧𑁄 𑀅𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀣,_  
_𑀆𑀕𑀢𑁂 𑀥𑀦𑀤𑀸𑀬𑀓𑁂;_  
_𑀦𑀦𑁆𑀤𑀺𑀁 𑀧𑀯𑁂𑀲𑀺 𑀦𑀕𑀭𑁂,_  
_𑀩𑀦𑁆𑀥𑀦𑀸 𑀫𑁄𑀓𑁆𑀔𑁄 𑀅𑀖𑁄𑀲𑀣𑁇_  


(2439.)

3799\. _𑀚𑀸𑀢𑀭𑀽𑀧𑀫𑀬𑀁 𑀯𑀲𑁆𑀲𑀁,_  
_𑀤𑁂𑀯𑁄 𑀧𑀸𑀯𑀲𑁆𑀲𑀺 𑀢𑀸𑀯𑀤𑁂;_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁂 𑀧𑀯𑀺𑀝𑁆𑀞𑀫𑁆𑀳𑀺,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(2440.)

3800\. _𑀢𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀤𑀸𑀦𑀁 𑀤𑀢𑁆𑀯𑀸𑀦 𑀔𑀢𑁆𑀢𑀺𑀬𑁄;_  
_𑀓𑀸𑀬𑀲𑁆𑀲 𑀪𑁂𑀤𑀸 𑀲𑀧𑁆𑀧𑀜𑁆𑀜𑁄,_  
_𑀲𑀕𑁆𑀕𑀁 𑀲𑁄 𑀉𑀧𑀧𑀚𑁆𑀚𑀣𑀸𑀢𑀺𑁇 (𑁬𑁬𑁫𑁨)_  


---

3801\. 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀚𑀸𑀢𑀓𑀁 𑀤𑀲𑀫𑀁𑁇
