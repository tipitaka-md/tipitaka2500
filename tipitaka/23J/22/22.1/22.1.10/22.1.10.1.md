# 𑁨𑁨𑁇𑁧𑁇𑁧𑁦𑁇𑁧 𑀤𑀲𑀯𑀭𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

### 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀯𑀕𑁆𑀕

### 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀚𑀸𑀢𑀓

### 𑀤𑀲𑀯𑀭𑀓𑀣𑀸

(1655.)

3001. _“𑀨𑀼𑀲𑁆𑀲𑀢𑀻 𑀯𑀭𑀯𑀡𑁆𑀡𑀸𑀪𑁂,_  
_𑀯𑀭𑀲𑁆𑀲𑀼 𑀤𑀲𑀥𑀸 𑀯𑀭𑁂;_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀘𑀸𑀭𑀼𑀧𑀼𑀩𑁆𑀩𑀗𑁆𑀕𑀺,_  
_𑀬𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀫𑀦𑀲𑁄 𑀧𑀺𑀬𑀁”𑁇_  


(1656.)

3002. _“𑀤𑁂𑀯𑀭𑀸𑀚 𑀦𑀫𑁄 𑀢𑁆𑀬𑀢𑁆𑀣𑀼,_  
_𑀓𑀺𑀁 𑀧𑀸𑀧𑀁 𑀧𑀓𑀢𑀁 𑀫𑀬𑀸;_  
_𑀭𑀫𑁆𑀫𑀸 𑀘𑀸𑀯𑁂𑀲𑀺 𑀫𑀁 𑀞𑀸𑀦𑀸,_  
_𑀯𑀸𑀢𑁄𑀯 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑀁”𑁇_  


(1657.)

3003. _“𑀦 𑀘𑁂𑀯 𑀢𑁂 𑀓𑀢𑀁 𑀧𑀸𑀧𑀁,_  
_𑀦 𑀘 𑀫𑁂 𑀢𑁆𑀯𑀫𑀲𑀺 𑀅𑀧𑁆𑀧𑀺𑀬𑀸;_  
_𑀧𑀼𑀜𑁆𑀜𑀜𑁆𑀘 𑀢𑁂 𑀧𑀭𑀺𑀓𑁆𑀔𑀻𑀡𑀁,_  
_𑀬𑁂𑀦 𑀢𑁂𑀯𑀁 𑀯𑀤𑀸𑀫𑀳𑀁𑁇_  


(1658.)

3004. _𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀫𑀭𑀡𑀁 𑀢𑀼𑀬𑁆𑀳𑀁,_  
_𑀯𑀺𑀦𑀸𑀪𑀸𑀯𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀧𑀝𑀺𑀕𑀡𑁆𑀳𑀸𑀳𑀺 𑀫𑁂 𑀏𑀢𑁂,_  
_𑀯𑀭𑁂 𑀤𑀲 𑀧𑀯𑁂𑀘𑁆𑀙𑀢𑁄”𑁇_  


(1659.)

3005. _“𑀯𑀭𑀜𑁆𑀘𑁂 𑀫𑁂 𑀅𑀤𑁄 𑀲𑀓𑁆𑀓,_  
_𑀲𑀩𑁆𑀩𑀪𑀽𑀢𑀸𑀦𑀫𑀺𑀲𑁆𑀲𑀭;_  
_𑀲𑀺𑀯𑀺𑀭𑀸𑀚𑀲𑁆𑀲 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀢𑀢𑁆𑀣 𑀅𑀲𑁆𑀲𑀁 𑀦𑀺𑀯𑁂𑀲𑀦𑁂𑁇_  


(1660.)

3006. _𑀦𑀻𑀮𑀦𑁂𑀢𑁆𑀢𑀸 𑀦𑀻𑀮𑀪𑀫𑀼,_  
_𑀦𑀺𑀮𑀓𑁆𑀔𑀻 𑀘 𑀬𑀣𑀸 𑀫𑀺𑀕𑀻;_  
_𑀨𑀼𑀲𑁆𑀲𑀢𑀻 𑀦𑀸𑀫 𑀦𑀸𑀫𑁂𑀦,_  
_𑀢𑀢𑁆𑀣𑀧𑀲𑁆𑀲𑀁 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁇_  


(1661.)

3007. _𑀧𑀼𑀢𑁆𑀢𑀁 𑀮𑀪𑁂𑀣 𑀯𑀭𑀤𑀁,_  
_𑀬𑀸𑀘𑀬𑁄𑀕𑀁 𑀅𑀫𑀘𑁆𑀙𑀭𑀺𑀁;_  
_𑀧𑀽𑀚𑀺𑀢𑀁 𑀧𑀝𑀺𑀭𑀸𑀚𑀽𑀳𑀺,_  
_𑀓𑀺𑀢𑁆𑀢𑀺𑀫𑀦𑁆𑀢𑀁 𑀬𑀲𑀲𑁆𑀲𑀺𑀦𑀁𑁇_  


(1662.)

3008. _𑀕𑀩𑁆𑀪𑀁 𑀫𑁂 𑀥𑀸𑀭𑀬𑀦𑁆𑀢𑀺𑀬𑀸,_  
_𑀫𑀚𑁆𑀛𑀺𑀫𑀗𑁆𑀕𑀁 𑀅𑀦𑀼𑀦𑁆𑀦𑀢𑀁;_  
_𑀓𑀼𑀘𑁆𑀙𑀺 𑀅𑀦𑀼𑀦𑁆𑀦𑀢𑁄 𑀅𑀲𑁆𑀲,_  
_𑀘𑀸𑀧𑀁𑀯 𑀮𑀺𑀔𑀺𑀢𑀁 𑀲𑀫𑀁𑁇_  


(1663.)

3009. _𑀣𑀦𑀸 𑀫𑁂 𑀦𑀧𑁆𑀧𑀧𑀢𑁂𑀬𑁆𑀬𑀼𑀁,_  
_𑀧𑀮𑀺𑀢𑀸 𑀦 𑀲𑀦𑁆𑀢𑀼 𑀯𑀸𑀲𑀯;_  
_𑀓𑀸𑀬𑁂 𑀭𑀚𑁄 𑀦 𑀮𑀺𑀫𑁆𑀧𑁂𑀣,_  
_𑀯𑀚𑁆𑀛𑀜𑁆𑀘𑀸𑀧𑀺 𑀧𑀫𑁄𑀘𑀬𑁂𑁇_  


(1664.)

3010. _𑀫𑀬𑀽𑀭𑀓𑁄𑀜𑁆𑀘𑀸𑀪𑀺𑀭𑀼𑀤𑁂,_  
_𑀦𑀸𑀭𑀺𑀯𑀭𑀕𑀡𑀸𑀬𑀼𑀢𑁂;_  
_𑀔𑀼𑀚𑁆𑀚𑀘𑁂𑀮𑀸𑀧𑀓𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀲𑀽𑀢𑀫𑀸𑀕𑀥𑀯𑀡𑁆𑀡𑀺𑀢𑁂𑁇_  


(1665.)

3011. _𑀘𑀺𑀢𑁆𑀭𑀕𑁆𑀕𑀍𑀏𑀭𑀼𑀖𑀼𑀲𑀺𑀢𑁂,_  
_𑀲𑀼𑀭𑀸𑀫𑀁𑀲𑀧𑀩𑁄𑀥𑀦𑁂;_  
_𑀲𑀺𑀯𑀺𑀭𑀸𑀚𑀲𑁆𑀲 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀢𑀢𑁆𑀣𑀲𑁆𑀲𑀁 𑀫𑀳𑁂𑀲𑀻 𑀧𑀺𑀬𑀸”𑁇_  


(1666.)

3012. _“𑀬𑁂 𑀢𑁂 𑀤𑀲 𑀯𑀭𑀸 𑀤𑀺𑀦𑁆𑀦𑀸,_  
_𑀫𑀬𑀸 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑁂;_  
_𑀲𑀺𑀯𑀺𑀭𑀸𑀚𑀲𑁆𑀲 𑀯𑀺𑀚𑀺𑀢𑁂,_  
_𑀲𑀩𑁆𑀩𑁂 𑀢𑁂 𑀮𑀘𑁆𑀙𑀲𑀻 𑀯𑀭𑁂”𑁇_  


(1667.)

3013. _𑀇𑀤𑀁 𑀯𑀢𑁆𑀯𑀸𑀦 𑀫𑀖𑀯𑀸,_  
_𑀤𑁂𑀯𑀭𑀸𑀚𑀸 𑀲𑀼𑀚𑀫𑁆𑀧𑀢𑀺;_  
_𑀨𑀼𑀲𑁆𑀲𑀢𑀺𑀬𑀸 𑀯𑀭𑀁 𑀤𑀢𑁆𑀯𑀸,_  
_𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀣 𑀯𑀸𑀲𑀯𑁄𑁇_  


---

3014. 𑀤𑀲𑀯𑀭𑀓𑀣𑀸 𑀦𑀸𑀫𑁇



(1668.)

3015. _“𑀧𑀭𑀽𑀍𑀳𑀓𑀘𑁆𑀙𑀦𑀔𑀮𑁄𑀫𑀸,_  
_𑀧𑀗𑁆𑀓𑀤𑀦𑁆𑀢𑀸 𑀭𑀚𑀲𑁆𑀲𑀺𑀭𑀸;_  
_𑀧𑀕𑁆𑀕𑀬𑁆𑀳 𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀩𑀸𑀳𑀼𑀁,_  
_𑀓𑀺𑀁 𑀫𑀁 𑀬𑀸𑀘𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸”𑁇_  


(1669.)

3016. _“𑀭𑀢𑀦𑀁 𑀤𑁂𑀯 𑀬𑀸𑀘𑀸𑀫,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑀁;_  
_𑀤𑀤𑀸𑀳𑀺 𑀧𑀯𑀭𑀁 𑀦𑀸𑀕𑀁,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑀁 𑀉𑀭𑀽𑀍𑀳𑀯𑀁”𑁇_  


(1670.)

3017. _“𑀤𑀤𑀸𑀫𑀺 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑀸𑀫𑀺,_  
_𑀬𑀁 𑀫𑀁 𑀬𑀸𑀘𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀧𑀪𑀺𑀦𑁆𑀦𑀁 𑀓𑀼𑀜𑁆𑀚𑀭𑀁 𑀤𑀦𑁆𑀢𑀺𑀁,_  
_𑀑𑀧𑀯𑀬𑁆𑀳𑀁 𑀕𑀚𑀼𑀢𑁆𑀢𑀫𑀁”𑁇_  


(1671.)

3018. _𑀳𑀢𑁆𑀣𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀢𑁄 𑀑𑀭𑀼𑀬𑁆𑀳,_  
_𑀭𑀸𑀚𑀸 𑀘𑀸𑀕𑀸𑀥𑀺𑀫𑀸𑀦𑀲𑁄;_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀅𑀤𑀸 𑀤𑀸𑀦𑀁,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁄𑁇_  


(1672.)

3019. _𑀢𑀤𑀸𑀲𑀺 𑀬𑀁 𑀪𑀺𑀁𑀲𑀦𑀓𑀁,_  
_𑀢𑀤𑀸𑀲𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁂 𑀧𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀳𑀺,_  
_𑀫𑁂𑀤𑀦𑀻 𑀲𑀫𑁆𑀧𑀓𑀫𑁆𑀧𑀣𑁇_  


(1673.)

3020. _𑀢𑀤𑀸𑀲𑀺 𑀬𑀁 𑀪𑀺𑀁𑀲𑀦𑀓𑀁,_  
_𑀢𑀤𑀸𑀲𑀺 𑀮𑁄𑀫𑀳𑀁𑀲𑀦𑀁;_  
_𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁂 𑀧𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀳𑀺,_  
_𑀔𑀼𑀪𑀺𑀢𑁆𑀣 𑀦𑀕𑀭𑀁 𑀢𑀤𑀸𑁇_  


(1674.)

3021. _𑀲𑀫𑀸𑀓𑀼𑀮𑀁 𑀧𑀼𑀭𑀁 𑀆𑀲𑀺,_  
_𑀖𑁄𑀲𑁄 𑀘 𑀯𑀺𑀧𑀼𑀮𑁄 𑀫𑀳𑀸;_  
_𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁂 𑀧𑀤𑀺𑀦𑁆𑀦𑀫𑁆𑀳𑀺,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁂𑁇_  


(1675.)

3022. _𑀉𑀕𑁆𑀕𑀸 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸𑁇_  


(1676.)

3023. _𑀓𑁂𑀯𑀮𑁄 𑀘𑀸𑀧𑀺 𑀦𑀺𑀕𑀫𑁄,_  
_𑀲𑀺𑀯𑀬𑁄 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀦𑀸𑀕𑀁 𑀦𑀻𑀬𑀫𑀸𑀦𑀁,_  
_𑀢𑁂 𑀭𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀯𑁂𑀤𑀬𑀼𑀁𑁇_  


(1677.)

3024. _“𑀯𑀺𑀥𑀫𑀁 𑀤𑁂𑀯 𑀢𑁂 𑀭𑀝𑁆𑀞𑀁,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀢𑀯;_  
_𑀓𑀣𑀁 𑀦𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀁 𑀤𑀚𑁆𑀚𑀸,_  
_𑀦𑀸𑀕𑀁 𑀭𑀝𑁆𑀞𑀲𑁆𑀲 𑀧𑀽𑀚𑀺𑀢𑀁𑁇_  


(1678.)

3025. _𑀓𑀣𑀁 𑀦𑁄 𑀓𑀼𑀜𑁆𑀚𑀭𑀁 𑀤𑀚𑁆𑀚𑀸,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑀁 𑀉𑀭𑀽𑀍𑀳𑀯𑀁;_  
_𑀔𑁂𑀢𑁆𑀢𑀜𑁆𑀜𑀼𑀁 𑀲𑀩𑁆𑀩𑀬𑀼𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀲𑀩𑁆𑀩𑀲𑁂𑀢𑀁 𑀕𑀚𑀼𑀢𑁆𑀢𑀫𑀁𑁇_  


(1679.)

3026. _𑀧𑀡𑁆𑀟𑀼𑀓𑀫𑁆𑀩𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀧𑀪𑀺𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀼𑀫𑀤𑁆𑀤𑀦𑀁;_  
_𑀤𑀦𑁆𑀢𑀺𑀁 𑀲𑀯𑀸𑀍𑀅𑀩𑀻𑀚𑀦𑀺𑀁,_  
_𑀲𑁂𑀢𑀁 𑀓𑁂𑀮𑀸𑀲𑀲𑀸𑀤𑀺𑀲𑀁𑁇_  


(1680.)

3027. _𑀲𑀲𑁂𑀢𑀘𑁆𑀙𑀢𑁆𑀢𑀁 𑀲𑁅𑀧𑀸𑀥𑁂𑀬𑁆𑀬𑀁,_  
_𑀲𑀸𑀣𑀩𑁆𑀩𑀦𑀁 𑀲𑀳𑀢𑁆𑀣𑀺𑀧𑀁;_  
_𑀅𑀕𑁆𑀕𑀬𑀸𑀦𑀁 𑀭𑀸𑀚𑀯𑀸𑀳𑀺𑀁,_  
_𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀦𑀁 𑀅𑀤𑀸 𑀕𑀚𑀁𑁇_  


(1681.)

3028. _𑀅𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀜𑁆𑀘 𑀬𑁄 𑀤𑀚𑁆𑀚𑀸,_  
_𑀯𑀢𑁆𑀣𑀲𑁂𑀦𑀸𑀲𑀦𑀸𑀦𑀺 𑀘;_  
_𑀏𑀢𑀁 𑀔𑁄 𑀤𑀸𑀦𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁,_  
_𑀏𑀢𑀁 𑀔𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸𑀭𑀳𑀁𑁇_  


(1682.)

3029. _𑀅𑀬𑀁 𑀢𑁂 𑀯𑀁𑀲𑀭𑀸𑀚𑀸 𑀦𑁄,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀭𑀝𑁆𑀞𑀯𑀟𑁆𑀠𑀦𑁄;_  
_𑀓𑀣𑀁 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀕𑀚𑀁 𑀪𑀸𑀚𑁂𑀢𑀺 𑀲𑀜𑁆𑀚𑀬𑁇_  


(1683.)

3030. _𑀲𑀘𑁂 𑀢𑁆𑀯𑀁 𑀦 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺,_  
_𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀁 𑀇𑀤𑀁;_  
_𑀫𑀜𑁆𑀜𑁂 𑀢𑀁 𑀲𑀳 𑀧𑀼𑀢𑁆𑀢𑁂𑀦,_  
_𑀲𑀺𑀯𑀻 𑀳𑀢𑁆𑀣𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀭𑁂”𑁇_  


(1684.)

3031. _“𑀓𑀸𑀫𑀁 𑀚𑀦𑀧𑀤𑁄 𑀫𑀸𑀲𑀺,_  
_𑀭𑀝𑁆𑀞𑀜𑁆𑀘𑀸𑀧𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀼;_  
_𑀦𑀸𑀳𑀁 𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀸,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀤𑀽𑀲𑀓𑀁;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬𑀁 𑀲𑀓𑀸 𑀭𑀝𑁆𑀞𑀸,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀳𑀺 𑀫𑀫 𑀑𑀭𑀲𑁄𑁇_  


(1685.)

3032. _𑀓𑀸𑀫𑀁 𑀚𑀦𑀧𑀤𑁄 𑀫𑀸𑀲𑀺,_  
_𑀭𑀝𑁆𑀞𑀜𑁆𑀘𑀸𑀧𑀺 𑀯𑀺𑀦𑀲𑁆𑀲𑀢𑀼;_  
_𑀦𑀸𑀳𑀁 𑀲𑀺𑀯𑀻𑀦𑀁 𑀯𑀘𑀦𑀸,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀤𑀽𑀲𑀓𑀁;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬𑀁 𑀲𑀓𑀸 𑀭𑀝𑁆𑀞𑀸,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀳𑀺 𑀫𑀫 𑀅𑀢𑁆𑀭𑀚𑁄𑁇_  


(1686.)

3033. _𑀦 𑀘𑀸𑀳𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀤𑀼𑀩𑁆𑀪𑁂𑀬𑁆𑀬𑀁,_  
_𑀅𑀭𑀺𑀬𑀲𑀻𑀮𑀯𑀢𑁄 𑀳𑀺 𑀲𑁄;_  
_𑀅𑀲𑀺𑀮𑁄𑀓𑁄𑀧𑀺 𑀫𑁂 𑀅𑀲𑁆𑀲,_  
_𑀧𑀸𑀧𑀜𑁆𑀘 𑀧𑀲𑀯𑁂 𑀩𑀳𑀼𑀁;_  
_𑀓𑀣𑀁 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀲𑀢𑁆𑀣𑁂𑀦 𑀖𑀸𑀢𑀬𑀸𑀫𑀲𑁂”𑁇_  


(1687.)

3034. _“𑀫𑀸 𑀦𑀁 𑀤𑀡𑁆𑀟𑁂𑀦 𑀲𑀢𑁆𑀣𑁂𑀦,_  
_𑀦 𑀳𑀺 𑀲𑁄 𑀩𑀦𑁆𑀥𑀦𑀸𑀭𑀳𑁄;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀳𑀺 𑀘 𑀦𑀁 𑀭𑀝𑁆𑀞𑀸,_  
_𑀯𑀗𑁆𑀓𑁂 𑀯𑀲𑀢𑀼 𑀧𑀩𑁆𑀩𑀢𑁂”𑁇_  


(1688.)

3035. _“𑀏𑀲𑁄 𑀘𑁂 𑀲𑀺𑀯𑀻𑀦𑀁 𑀙𑀦𑁆𑀤𑁄,_  
_𑀙𑀦𑁆𑀤𑀁 𑀦 𑀧𑀦𑀼𑀤𑀸𑀫𑀲𑁂;_  
_𑀇𑀫𑀁 𑀲𑁄 𑀯𑀲𑀢𑀼 𑀭𑀢𑁆𑀢𑀺𑀁,_  
_𑀓𑀸𑀫𑁂 𑀘 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀼𑁇_  


(1689.)

3036. _𑀢𑀢𑁄 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀧𑀢𑀺;_  
_𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀺𑀯𑀬𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀭𑀝𑁆𑀞𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀦𑁆𑀢𑀼 𑀦𑀁𑁇_  


(1690.)

3037. _𑀉𑀝𑁆𑀞𑁂𑀳𑀺 𑀓𑀢𑁆𑀢𑁂 𑀢𑀭𑀫𑀸𑀦𑁄,_  
_𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁 𑀯𑀤;_  
_‘𑀲𑀺𑀯𑀬𑁄 𑀤𑁂𑀯 𑀢𑁂 𑀓𑀼𑀤𑁆𑀥𑀸,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸𑁇_  


(1691.)

3038. _𑀉𑀕𑁆𑀕𑀸 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀓𑁂𑀯𑀮𑁄 𑀘𑀸𑀧𑀺 𑀦𑀺𑀕𑀫𑁄,_  
_𑀲𑀺𑀯𑀬𑁄 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸𑁇_  


(1692.)

3039. _𑀅𑀲𑁆𑀫𑀸 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀧𑀢𑀺;_  
_𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀺𑀯𑀬𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀭𑀝𑁆𑀞𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀦𑁆𑀢𑀺 𑀢𑀁’”𑁇_  


(1693.)

3040. _𑀲 𑀓𑀢𑁆𑀢𑀸 𑀢𑀭𑀫𑀸𑀦𑁄𑀯,_  
_𑀲𑀺𑀯𑀺𑀭𑀸𑀚𑁂𑀦 𑀧𑁂𑀲𑀺𑀢𑁄;_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀸𑀪𑀭𑀡𑁄,_  
_𑀲𑀼𑀯𑀢𑁆𑀣𑁄 𑀘𑀦𑁆𑀤𑀦𑀪𑀽𑀲𑀺𑀢𑁄𑁇_  


(1694.)

3041. _𑀲𑀻𑀲𑀁 𑀦𑁆𑀳𑀸𑀢𑁄 𑀉𑀤𑀓𑁂 𑀲𑁄,_  
_𑀆𑀫𑀼𑀢𑁆𑀢𑀫𑀡𑀺𑀓𑀼𑀡𑁆𑀟𑀮𑁄;_  
_𑀉𑀧𑀸𑀕𑀫𑀺 𑀧𑀼𑀭𑀁 𑀭𑀫𑁆𑀫𑀁,_  
_𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀦𑀺𑀯𑁂𑀲𑀦𑀁𑁇_  


(1695.)

3042. _𑀢𑀢𑁆𑀣𑀤𑁆𑀤𑀲 𑀓𑀼𑀫𑀸𑀭𑀁 𑀲𑁄,_  
_𑀭𑀫𑀫𑀸𑀦𑀁 𑀲𑀓𑁂 𑀧𑀼𑀭𑁂;_  
_𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀁 𑀅𑀫𑀘𑁆𑀘𑁂𑀳𑀺,_  
_𑀢𑀺𑀤𑀲𑀸𑀦𑀁𑀯 𑀯𑀸𑀲𑀯𑀁𑁇_  


(1696.)

3043. _𑀲𑁄 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀭𑀫𑀸𑀦𑁄,_  
_𑀓𑀢𑁆𑀢𑀸 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀁𑀩𑁆𑀭𑀯𑀺;_  
_“𑀤𑀼𑀓𑁆𑀔𑀁 𑀢𑁂 𑀯𑁂𑀤𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀫𑀸 𑀫𑁂 𑀓𑀼𑀚𑁆𑀛𑀺 𑀭𑀣𑁂𑀲𑀪”𑁇_  


(1697.)

3044. _𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀫𑀸𑀦𑁄 𑀲𑁄,_  
_𑀓𑀢𑁆𑀢𑀸 𑀭𑀸𑀚𑀸𑀦𑀫𑀩𑁆𑀭𑀯𑀺;_  
_“𑀪𑀢𑁆𑀢𑀸 𑀫𑁂𑀲𑀺 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀲𑀩𑁆𑀩𑀓𑀸𑀫𑀭𑀲𑀸𑀳𑀭𑁄𑁇_  


(1698.)

3045. _𑀤𑀼𑀓𑁆𑀔𑀁 𑀢𑁂 𑀯𑁂𑀤𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀢𑀢𑁆𑀣 𑀅𑀲𑁆𑀲𑀸𑀲𑀬𑀦𑁆𑀢𑀼 𑀫𑀁;_  
_𑀲𑀺𑀯𑀬𑁄 𑀤𑁂𑀯 𑀢𑁂 𑀓𑀼𑀤𑁆𑀥𑀸,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸𑁇_  


(1699.)

3046. _𑀉𑀕𑁆𑀕𑀸 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀓𑁂𑀯𑀮𑁄 𑀘𑀸𑀧𑀺 𑀦𑀺𑀕𑀫𑁄,_  
_𑀲𑀺𑀯𑀬𑁄 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸𑁇_  


(1700.)

3047. _𑀅𑀲𑁆𑀫𑀸 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀧𑀢𑀺;_  
_𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀺𑀯𑀬𑁄 𑀳𑀼𑀢𑁆𑀯𑀸,_  
_𑀭𑀝𑁆𑀞𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀦𑁆𑀢𑀺 𑀢𑀁”𑁇_  


(1701.)

3048. _“𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀫𑁂 𑀲𑀺𑀯𑀬𑁄 𑀓𑀼𑀤𑁆𑀥𑀸,_  
_𑀦𑀸𑀳𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀁;_  
_𑀢𑀁 𑀫𑁂 𑀓𑀢𑁆𑀢𑁂 𑀯𑀺𑀬𑀸𑀘𑀺𑀓𑁆𑀔,_  
_𑀓𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀦𑁆𑀢𑀺 𑀫𑀁”𑁇_  


(1702.)

3049. _“𑀉𑀕𑁆𑀕𑀸 𑀘 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀦𑀸𑀕𑀤𑀸𑀦𑁂𑀦 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺,_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑀬𑀦𑁆𑀢𑀺 𑀢𑀁”𑁇_  


(1703.)

3050. _“𑀳𑀤𑀬𑀁 𑀘𑀓𑁆𑀔𑀼𑀫𑁆𑀧𑀳𑀁 𑀤𑀚𑁆𑀚𑀁,_  
_𑀓𑀺𑀁 𑀫𑁂 𑀩𑀸𑀳𑀺𑀭𑀓𑀁 𑀥𑀦𑀁;_  
_𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀯𑀸 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀯𑀸,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀫𑀡𑀺𑁇_  


(1704.)

3051. _𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀯𑀸𑀧𑀳𑀁 𑀩𑀸𑀳𑀼𑀁,_  
_𑀤𑀺𑀲𑁆𑀯𑀸 𑀬𑀸𑀘𑀓𑀫𑀸𑀕𑀢𑁂;_  
_𑀤𑀤𑁂𑀬𑁆𑀬𑀁 𑀦 𑀯𑀺𑀓𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀁,_  
_𑀤𑀸𑀦𑁂 𑀫𑁂 𑀭𑀫𑀢𑁂 𑀫𑀦𑁄𑁇_  


(1705.)

3052. _𑀓𑀸𑀫𑀁 𑀫𑀁 𑀲𑀺𑀯𑀬𑁄 𑀲𑀩𑁆𑀩𑁂,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀦𑁆𑀢𑀼 𑀳𑀦𑀦𑁆𑀢𑀼 𑀯𑀸;_  
_𑀦𑁂𑀯 𑀤𑀸𑀦𑀸 𑀯𑀺𑀭𑀫𑀺𑀲𑁆𑀲𑀁,_  
_𑀓𑀸𑀫𑀁 𑀙𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀼 𑀲𑀢𑁆𑀢𑀥𑀸”𑁇_  


(1706.)

3053. _𑀏𑀯𑀁 𑀢𑀁 𑀲𑀺𑀯𑀬𑁄 𑀆𑀳𑀼,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_“𑀓𑁄𑀦𑁆𑀢𑀺𑀫𑀸𑀭𑀸𑀬 𑀢𑀻𑀭𑁂𑀦,_  
_𑀕𑀺𑀭𑀺𑀫𑀸𑀭𑀜𑁆𑀚𑀭𑀁 𑀧𑀢𑀺;_  
_𑀬𑁂𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑀺𑀢𑀸 𑀬𑀦𑁆𑀢𑀺,_  
_𑀢𑁂𑀦 𑀕𑀘𑁆𑀙𑀢𑀼 𑀲𑀼𑀩𑁆𑀩𑀢𑁄”𑁇_  


(1707.)

3054. _“𑀲𑁄𑀳𑀁 𑀢𑁂𑀦 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀤𑀽𑀲𑀓𑀸;_  
_𑀭𑀢𑁆𑀢𑀺𑀦𑁆𑀤𑀺𑀯𑀁 𑀫𑁂 𑀔𑀫𑀣,_  
_𑀬𑀸𑀯 𑀤𑀸𑀦𑀁 𑀤𑀤𑀸𑀫𑀳𑀁”𑁇_  


(1708.)

3055. _𑀆𑀫𑀦𑁆𑀢𑀬𑀺𑀢𑁆𑀣 𑀭𑀸𑀚𑀸 𑀦𑀁,_  
_𑀫𑀤𑁆𑀤𑀺𑀁 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀁;_  
_“𑀬𑀁 𑀢𑁂 𑀓𑀺𑀜𑁆𑀘𑀺 𑀫𑀬𑀸 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀥𑀦𑀁 𑀥𑀜𑁆𑀜𑀜𑁆𑀘 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(1709.)

3056. _𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀯𑀸 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀯𑀸,_  
_𑀫𑀼𑀢𑁆𑀢𑀸 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸 𑀩𑀳𑀽;_  
_𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀤𑀳𑁂𑀬𑁆𑀬𑀸𑀲𑀺,_  
_𑀬𑀜𑁆𑀘 𑀢𑁂 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀁 𑀥𑀦𑀁”𑁇_  


(1710.)

3057. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸;_  
_“𑀓𑀼𑀳𑀺𑀁 𑀤𑁂𑀯 𑀦𑀺𑀤𑀳𑀸𑀫𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄”𑁇_  


(1711.)

3058. _“𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁂𑀲𑀼 𑀤𑀚𑁆𑀚𑀸𑀲𑀺,_  
_𑀤𑀸𑀦𑀁 𑀫𑀤𑁆𑀤𑀺 𑀬𑀣𑀸𑀭𑀳𑀁;_  
_𑀦 𑀳𑀺 𑀤𑀸𑀦𑀸 𑀧𑀭𑀁 𑀅𑀢𑁆𑀣𑀺,_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀸 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁𑁇_  


(1712.)

3059. _𑀧𑀼𑀢𑁆𑀢𑁂𑀲𑀼 𑀫𑀤𑁆𑀤𑀺 𑀤𑀬𑁂𑀲𑀺,_  
_𑀲𑀲𑁆𑀲𑀼𑀬𑀸 𑀲𑀲𑀼𑀭𑀫𑁆𑀳𑀺 𑀘;_  
_𑀬𑁄 𑀘 𑀢𑀁 𑀪𑀢𑁆𑀢𑀸 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬,_  
_𑀲𑀓𑁆𑀓𑀘𑁆𑀘𑀁 𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑁂𑁇_  


(1713.)

3060. _𑀦𑁄 𑀘𑁂 𑀢𑀁 𑀪𑀢𑁆𑀢𑀸 𑀫𑀜𑁆𑀜𑁂𑀬𑁆𑀬,_  
_𑀫𑀬𑀸 𑀯𑀺𑀧𑁆𑀧𑀯𑀲𑁂𑀦 𑀢𑁂;_  
_𑀅𑀜𑁆𑀜𑀁 𑀪𑀢𑁆𑀢𑀸𑀭𑀁 𑀧𑀭𑀺𑀬𑁂𑀲,_  
_𑀫𑀸 𑀓𑀺𑀲𑀺𑀢𑁆𑀣𑁄 𑀫𑀬𑀸 𑀯𑀺𑀦𑀸𑁇_  


(1714.)

3061. _𑀅𑀳𑀜𑁆𑀳𑀺 𑀯𑀦𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀖𑁄𑀭𑀁 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀬𑀼𑀢𑀁;_  
_𑀲𑀁𑀲𑀬𑁄 𑀚𑀻𑀯𑀺𑀢𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀏𑀓𑀓𑀲𑁆𑀲 𑀩𑁆𑀭𑀳𑀸𑀯𑀦𑁂”𑁇_  


(1715.)

3062. _𑀢𑀫𑀩𑁆𑀭𑀯𑀺 𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀻,_  
_𑀫𑀤𑁆𑀤𑀻 𑀲𑀩𑁆𑀩𑀗𑁆𑀕𑀲𑁄𑀪𑀦𑀸;_  
_“𑀅𑀪𑀼𑀫𑁆𑀫𑁂 𑀓𑀣𑀁 𑀦𑀼 𑀪𑀡𑀲𑀺,_  
_𑀧𑀸𑀧𑀓𑀁 𑀯𑀢 𑀪𑀸𑀲𑀲𑀺𑁇_  


(1716.)

3063. _𑀦𑁂𑀲 𑀥𑀫𑁆𑀫𑁄 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀏𑀓𑀓𑁄;_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀢𑁂𑀦 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀬𑁂𑀦 𑀕𑀘𑁆𑀙𑀲𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁇_  


(1717.)

3064. _𑀫𑀭𑀡𑀁 𑀯𑀸 𑀢𑀬𑀸 𑀲𑀤𑁆𑀥𑀺𑀁,_  
_𑀚𑀻𑀯𑀺𑀢𑀁 𑀯𑀸 𑀢𑀬𑀸 𑀯𑀺𑀦𑀸;_  
_𑀢𑀤𑁂𑀯 𑀫𑀭𑀡𑀁 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑀜𑁆𑀘𑁂 𑀚𑀻𑀯𑁂 𑀢𑀬𑀸 𑀯𑀺𑀦𑀸𑁇_  


(1718.)

3065. _𑀅𑀕𑁆𑀕𑀺𑀁 𑀉𑀚𑁆𑀚𑀸𑀮𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀏𑀓𑀚𑀸𑀮𑀲𑀫𑀸𑀳𑀺𑀢𑀁;_  
_𑀢𑀢𑁆𑀣 𑀫𑁂 𑀫𑀭𑀡𑀁 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀬𑀜𑁆𑀘𑁂 𑀚𑀻𑀯𑁂 𑀢𑀬𑀸 𑀯𑀺𑀦𑀸𑁇_  


(1719.)

3066. _𑀬𑀣𑀸 𑀆𑀭𑀜𑁆𑀜𑀓𑀁 𑀦𑀸𑀕𑀁,_  
_𑀤𑀦𑁆𑀢𑀺𑀁 𑀅𑀦𑁆𑀯𑁂𑀢𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻;_  
_𑀚𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀁 𑀕𑀺𑀭𑀺𑀤𑀼𑀕𑁆𑀕𑁂𑀲𑀼,_  
_𑀲𑀫𑁂𑀲𑀼 𑀯𑀺𑀲𑀫𑁂𑀲𑀼 𑀘𑁇_  


(1720.)

3067. _𑀏𑀯𑀁 𑀢𑀁 𑀅𑀦𑀼𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀧𑀼𑀢𑁆𑀢𑁂 𑀆𑀤𑀸𑀬 𑀧𑀘𑁆𑀙𑀢𑁄;_  
_𑀲𑀼𑀪𑀭𑀸 𑀢𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀦 𑀢𑁂 𑀳𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀤𑀼𑀩𑁆𑀪𑀭𑀸𑁇_  


(1721.)

3068. _𑀇𑀫𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑀜𑁆𑀚𑀼𑀓𑁂 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑁂;_  
_𑀆𑀲𑀻𑀦𑁂 𑀯𑀦𑀕𑀼𑀫𑁆𑀩𑀲𑁆𑀫𑀺𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1722.)

3069. _𑀇𑀫𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑀜𑁆𑀚𑀼𑀓𑁂 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑁂;_  
_𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑁂 𑀯𑀦𑀕𑀼𑀫𑁆𑀩𑀲𑁆𑀫𑀺𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1723.)

3070. _𑀇𑀫𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑀜𑁆𑀚𑀼𑀓𑁂 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑁂;_  
_𑀅𑀲𑁆𑀲𑀫𑁂 𑀭𑀫𑀡𑀻𑀬𑀫𑁆𑀳𑀺,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1724.)

3071. _𑀇𑀫𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑀜𑁆𑀚𑀼𑀓𑁂 𑀧𑀺𑀬𑀪𑀸𑀡𑀺𑀦𑁂;_  
_𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑁂 𑀅𑀲𑁆𑀲𑀫𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1725.)

3072. _𑀇𑀫𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑀸𑀮𑀥𑀸𑀭𑀻 𑀅𑀮𑀗𑁆𑀓𑀢𑁂;_  
_𑀅𑀲𑁆𑀲𑀫𑁂 𑀭𑀫𑀡𑀻𑀬𑀫𑁆𑀳𑀺,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1726.)

3073. _𑀇𑀫𑁂 𑀓𑀼𑀫𑀸𑀭𑁂 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄,_  
_𑀫𑀸𑀮𑀥𑀸𑀭𑀻 𑀅𑀮𑀗𑁆𑀓𑀢𑁂;_  
_𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑁂 𑀅𑀲𑁆𑀲𑀫𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1727.)

3074. _𑀬𑀤𑀸 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑁂,_  
_𑀓𑀼𑀫𑀸𑀭𑁂 𑀫𑀸𑀮𑀥𑀸𑀭𑀺𑀦𑁂;_  
_𑀅𑀲𑁆𑀲𑀫𑁂 𑀭𑀫𑀡𑀻𑀬𑀫𑁆𑀳𑀺,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1728.)

3075. _𑀬𑀤𑀸 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑁂,_  
_𑀓𑀼𑀫𑀸𑀭𑁂 𑀫𑀸𑀮𑀥𑀸𑀭𑀺𑀦𑁂;_  
_𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑁂 𑀅𑀲𑁆𑀲𑀫𑁂 𑀭𑀫𑁆𑀫𑁂,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1729.)

3076. _𑀬𑀤𑀸 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀫𑀸𑀢𑀗𑁆𑀕𑀁,_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀁 𑀲𑀝𑁆𑀞𑀺𑀳𑀸𑀬𑀦𑀁;_  
_𑀏𑀓𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀘𑀭𑀦𑁆𑀢𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1730.)

3077. _𑀬𑀤𑀸 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀫𑀸𑀢𑀗𑁆𑀕𑀁,_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑀁 𑀲𑀝𑁆𑀞𑀺𑀳𑀸𑀬𑀦𑀁;_  
_𑀲𑀸𑀬𑀁 𑀧𑀸𑀢𑁄 𑀯𑀺𑀘𑀭𑀦𑁆𑀢𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1731.)

3078. _𑀬𑀤𑀸 𑀓𑀭𑁂𑀡𑀼𑀲𑀗𑁆𑀖𑀲𑁆𑀲,_  
_𑀬𑀽𑀣𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀯𑀚𑀁;_  
_𑀓𑁄𑀜𑁆𑀘𑀁 𑀓𑀸𑀳𑀢𑀺 𑀫𑀸𑀢𑀗𑁆𑀕𑁄,_  
_𑀓𑀼𑀜𑁆𑀚𑀭𑁄 𑀲𑀝𑁆𑀞𑀺𑀳𑀸𑀬𑀦𑁄;_  
_𑀢𑀲𑁆𑀲 𑀢𑀁 𑀦𑀤𑀢𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1732.)

3079. _𑀤𑀼𑀪𑀢𑁄 𑀯𑀦𑀯𑀺𑀓𑀸𑀲𑁂,_  
_𑀬𑀤𑀸 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀓𑀸𑀫𑀤𑁄;_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1733.)

3080. _𑀫𑀺𑀕𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀸𑀬𑀦𑁆𑀳𑀁,_  
_𑀧𑀜𑁆𑀘𑀫𑀸𑀮𑀺𑀦𑀫𑀸𑀕𑀢𑀁;_  
_𑀓𑀺𑀫𑁆𑀧𑀼𑀭𑀺𑀲𑁂 𑀘 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑁂,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1734.)

3081. _𑀬𑀤𑀸 𑀲𑁄𑀲𑁆𑀲𑀲𑀺 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁,_  
_𑀲𑀦𑁆𑀤𑀫𑀸𑀦𑀸𑀬 𑀲𑀺𑀦𑁆𑀥𑀼𑀬𑀸;_  
_𑀕𑀻𑀢𑀁 𑀓𑀺𑀫𑁆𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀜𑁆𑀘,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1735.)

3082. _𑀬𑀤𑀸 𑀲𑁄𑀲𑁆𑀲𑀲𑀺 𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁,_  
_𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑀘𑀸𑀭𑀺𑀦𑁄;_  
_𑀯𑀲𑁆𑀲𑀫𑀸𑀦𑀲𑁆𑀲𑀼𑀮𑀽𑀓𑀲𑁆𑀲,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1736.)

3083. _𑀬𑀤𑀸 𑀲𑀻𑀳𑀲𑁆𑀲 𑀩𑁆𑀬𑀕𑁆𑀖𑀲𑁆𑀲,_  
_𑀔𑀕𑁆𑀕𑀲𑁆𑀲 𑀕𑀯𑀬𑀲𑁆𑀲 𑀘;_  
_𑀯𑀦𑁂 𑀲𑁄𑀲𑁆𑀲𑀲𑀺 𑀯𑀸𑀍𑀆𑀦𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1737.)

3084. _𑀬𑀤𑀸 𑀫𑁄𑀭𑀻𑀳𑀺 𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀩𑀭𑀺𑀳𑀻𑀦𑀁 𑀫𑀢𑁆𑀣𑀓𑀸𑀲𑀺𑀦𑀁;_  
_𑀫𑁄𑀭𑀁 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1738.)

3085. _𑀬𑀤𑀸 𑀫𑁄𑀭𑀻𑀳𑀺 𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀅𑀡𑁆𑀟𑀚𑀁 𑀘𑀺𑀢𑁆𑀭𑀧𑀓𑁆𑀔𑀺𑀦𑀁;_  
_𑀫𑁄𑀭𑀁 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1739.)

3086. _𑀬𑀤𑀸 𑀫𑁄𑀭𑀻𑀳𑀺 𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀦𑀻𑀮𑀕𑀻𑀯𑀁 𑀲𑀺𑀔𑀡𑁆𑀟𑀺𑀦𑀁;_  
_𑀫𑁄𑀭𑀁 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀦𑀘𑁆𑀘𑀦𑁆𑀢𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1740.)

3087. _𑀬𑀤𑀸 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀳𑁂𑀫𑀦𑁆𑀢𑁂,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁂 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑁂;_  
_𑀲𑀼𑀭𑀪𑀺𑀁 𑀲𑀫𑁆𑀧𑀯𑀸𑀬𑀦𑁆𑀢𑁂,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1741.)

3088. _𑀬𑀤𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑁂 𑀫𑀸𑀲𑁂,_  
_𑀳𑀭𑀺𑀢𑀁 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀫𑁂𑀤𑀦𑀺𑀁;_  
_𑀇𑀦𑁆𑀤𑀕𑁄𑀧𑀓𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1742.)

3089. _𑀬𑀤𑀸 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀳𑁂𑀫𑀦𑁆𑀢𑁂,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑁂 𑀥𑀭𑀡𑀻𑀭𑀼𑀳𑁂;_  
_𑀓𑀼𑀝𑀚𑀁 𑀩𑀺𑀫𑁆𑀩𑀚𑀸𑀮𑀜𑁆𑀘,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁 𑀮𑁄𑀤𑁆𑀤𑀧𑀤𑁆𑀫𑀓𑀁;_  
_𑀲𑀼𑀭𑀪𑀺𑀁 𑀲𑀫𑁆𑀧𑀯𑀸𑀬𑀦𑁆𑀢𑁂,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(1743.)

3090. _𑀬𑀤𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑁂 𑀫𑀸𑀲𑁂,_  
_𑀯𑀦𑀁 𑀤𑀓𑁆𑀔𑀺𑀲𑀺 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀁;_  
_𑀑𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺 𑀘 𑀧𑀤𑁆𑀫𑀸𑀦𑀺,_  
_𑀦 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


---

3091. 𑀳𑁂𑀫𑀯𑀦𑁆𑀢𑀁 𑀦𑀸𑀫𑁇
