# 𑁨𑁨𑁇𑁧𑁇𑁧𑁦𑁇𑁭 𑀫𑀳𑀸𑀯𑀦𑀯𑀡𑁆𑀡𑀦𑀸

* 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

* 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀯𑀕𑁆𑀕

* 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀚𑀸𑀢𑀓

* 𑀫𑀳𑀸𑀯𑀦𑀯𑀡𑁆𑀡𑀦𑀸

(2007.)

3359\. _𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀲𑁄 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄,_  
_𑀅𑀤𑁆𑀤𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀼𑀢𑀁 𑀇𑀲𑀺𑀁;_  
_𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀢𑀁 𑀪𑀸𑀭𑀤𑁆𑀯𑀸𑀚𑁄,_  
_𑀲𑀫𑁆𑀫𑁄𑀤𑀺 𑀇𑀲𑀺𑀦𑀸 𑀲𑀳𑁇_  


(2008.)

3360\. _“𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀪𑁄𑀢𑁄 𑀓𑀼𑀲𑀮𑀁,_  
_𑀓𑀘𑁆𑀘𑀺 𑀪𑁄𑀢𑁄 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀓𑀘𑁆𑀘𑀺 𑀉𑀜𑁆𑀙𑁂𑀦 𑀬𑀸𑀧𑁂𑀲𑀺,_  
_𑀓𑀘𑁆𑀘𑀺 𑀫𑀽𑀮𑀨𑀮𑀸 𑀩𑀳𑀽𑁇_  


(2009.)

3361\. _𑀓𑀘𑁆𑀘𑀺 𑀟𑀁𑀲𑀸 𑀫𑀓𑀲𑀸 𑀘,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀭𑀻𑀲𑀧𑀸;_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀓𑀘𑁆𑀘𑀺 𑀳𑀺𑀁𑀲𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺”𑁇_  


(2010.)

3362\. _“𑀓𑀼𑀲𑀮𑀜𑁆𑀘𑁂𑀯 𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀅𑀣𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀅𑀣𑁄 𑀉𑀜𑁆𑀙𑁂𑀦 𑀬𑀸𑀧𑁂𑀫𑀺,_  
_𑀅𑀣𑁄 𑀫𑀽𑀮𑀨𑀮𑀸 𑀩𑀳𑀽𑁇_  


(2011.)

3363\. _𑀅𑀣𑁄 𑀟𑀁𑀲𑀸 𑀫𑀓𑀲𑀸 𑀘,_  
_𑀅𑀧𑁆𑀧𑀫𑁂𑀯 𑀲𑀭𑀻𑀲𑀧𑀸;_  
_𑀯𑀦𑁂 𑀯𑀸𑀍𑀅𑀫𑀺𑀕𑀸𑀓𑀺𑀡𑁆𑀡𑁂,_  
_𑀳𑀺𑀁𑀲𑀸 𑀫𑀬𑁆𑀳𑀁 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(2012.)

3364\. _𑀩𑀳𑀽𑀦𑀺 𑀯𑀲𑁆𑀲𑀧𑀽𑀕𑀸𑀦𑀺,_  
_𑀅𑀲𑁆𑀲𑀫𑁂 𑀯𑀲𑀢𑁄 𑀫𑀫;_  
_𑀦𑀸𑀪𑀺𑀚𑀸𑀦𑀸𑀫𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁,_  
_𑀆𑀩𑀸𑀥𑀁 𑀅𑀫𑀦𑁄𑀭𑀫𑀁𑁇_  


(2013.)

3365\. _𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀢𑁂 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀅𑀣𑁄 𑀢𑁂 𑀅𑀤𑀼𑀭𑀸𑀕𑀢𑀁;_  
_𑀅𑀦𑁆𑀢𑁄 𑀧𑀯𑀺𑀲 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑀬𑀲𑁆𑀲𑀼 𑀢𑁂𑁇_  


(2014.)

3366\. _𑀢𑀺𑀦𑁆𑀤𑀼𑀓𑀸𑀦𑀺 𑀧𑀺𑀬𑀸𑀮𑀸𑀦𑀺,_  
_𑀫𑀥𑀼𑀓𑁂 𑀓𑀸𑀲𑀼𑀫𑀸𑀭𑀺𑀬𑁄;_  
_𑀨𑀮𑀸𑀦𑀺 𑀔𑀼𑀤𑁆𑀤𑀓𑀧𑁆𑀧𑀸𑀦𑀺,_  
_𑀪𑀼𑀜𑁆𑀚 𑀩𑁆𑀭𑀳𑁆𑀫𑁂 𑀯𑀭𑀁 𑀯𑀭𑀁𑁇_  


(2015.)

3367\. _𑀇𑀤𑀫𑁆𑀧𑀺 𑀧𑀸𑀦𑀻𑀬𑀁 𑀲𑀻𑀢𑀁,_  
_𑀆𑀪𑀢𑀁 𑀕𑀺𑀭𑀺𑀕𑀩𑁆𑀪𑀭𑀸;_  
_𑀢𑀢𑁄 𑀧𑀺𑀯 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀲𑀘𑁂 𑀢𑁆𑀯𑀁 𑀅𑀪𑀺𑀓𑀗𑁆𑀔𑀲𑀺”𑁇_  


(2016.)

3368\. _“𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀬𑀁 𑀤𑀺𑀦𑁆𑀦𑀁,_  
_𑀲𑀩𑁆𑀩𑀲𑁆𑀲 𑀅𑀕𑁆𑀖𑀺𑀬𑀁 𑀓𑀢𑀁;_  
_𑀲𑀜𑁆𑀚𑀬𑀲𑁆𑀲 𑀲𑀓𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀲𑀺𑀯𑀻𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀺𑀢𑀁;_  
_𑀢𑀫𑀳𑀁 𑀤𑀲𑁆𑀲𑀦𑀫𑀸𑀕𑀢𑁄,_  
_𑀬𑀤𑀺 𑀚𑀸𑀦𑀸𑀲𑀺 𑀲𑀁𑀲 𑀫𑁂”𑁇_  


(2017.)

3369\. _“𑀦 𑀪𑀯𑀁 𑀏𑀢𑀺 𑀧𑀼𑀜𑁆𑀜𑀢𑁆𑀣𑀁,_  
_𑀲𑀺𑀯𑀺𑀭𑀸𑀚𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀫𑀜𑁆𑀜𑁂 𑀪𑀯𑀁 𑀧𑀢𑁆𑀣𑀬𑀢𑀺,_  
_𑀭𑀜𑁆𑀜𑁄 𑀪𑀭𑀺𑀬𑀁 𑀧𑀢𑀺𑀩𑁆𑀩𑀢𑀁;_  
_𑀫𑀜𑁆𑀜𑁂 𑀓𑀡𑁆𑀳𑀸𑀚𑀺𑀦𑀁 𑀤𑀸𑀲𑀺𑀁,_  
_𑀚𑀸𑀮𑀺𑀁 𑀤𑀸𑀲𑀜𑁆𑀘 𑀇𑀘𑁆𑀙𑀲𑀺𑁇_  


(2018.)

3370\. _𑀅𑀣 𑀯𑀸 𑀢𑀬𑁄 𑀫𑀸𑀢𑀸𑀧𑀼𑀢𑁆𑀢𑁂,_  
_𑀅𑀭𑀜𑁆𑀜𑀸 𑀦𑁂𑀢𑀼𑀫𑀸𑀕𑀢𑁄;_  
_𑀦 𑀢𑀲𑁆𑀲 𑀪𑁄𑀕𑀸 𑀯𑀺𑀚𑁆𑀚𑀦𑁆𑀢𑀺,_  
_𑀥𑀦𑀁 𑀥𑀜𑁆𑀜𑀜𑁆𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡”𑁇_  


(2019.)

3371\. _“𑀅𑀓𑀼𑀤𑁆𑀥𑀭𑀽𑀧𑁄𑀳𑀁 𑀪𑁄𑀢𑁄,_  
_𑀦𑀸𑀳𑀁 𑀬𑀸𑀘𑀺𑀢𑀼𑀫𑀸𑀕𑀢𑁄;_  
_𑀲𑀸𑀥𑀼 𑀤𑀲𑁆𑀲𑀦𑀫𑀭𑀺𑀬𑀸𑀦𑀁,_  
_𑀲𑀦𑁆𑀦𑀺𑀯𑀸𑀲𑁄 𑀲𑀤𑀸 𑀲𑀼𑀔𑁄𑁇_  


(2020.)

3372\. _𑀅𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑁄 𑀲𑀺𑀯𑀺𑀭𑀸𑀚𑀸,_  
_𑀲𑀺𑀯𑀻𑀳𑀺 𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀺𑀢𑁄;_  
_𑀢𑀫𑀳𑀁 𑀤𑀲𑁆𑀲𑀦𑀫𑀸𑀕𑀢𑁄,_  
_𑀬𑀤𑀺 𑀚𑀸𑀦𑀸𑀲𑀺 𑀲𑀁𑀲 𑀫𑁂”𑁇_  


(2021.)

3373\. _“𑀏𑀲 𑀲𑁂𑀮𑁄 𑀫𑀳𑀸𑀩𑁆𑀭𑀳𑁆𑀫𑁂,_  
_𑀧𑀩𑁆𑀩𑀢𑁄 𑀕𑀦𑁆𑀥𑀫𑀸𑀤𑀦𑁄;_  
_𑀬𑀢𑁆𑀣 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀳 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


(2022.)

3374\. _𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀡𑁆𑀡𑀁,_  
_𑀆𑀲𑀤𑀜𑁆𑀘 𑀫𑀲𑀁 𑀚𑀝𑀁;_  
_𑀘𑀫𑁆𑀫𑀯𑀸𑀲𑀻 𑀙𑀫𑀸 𑀲𑁂𑀢𑀺,_  
_𑀚𑀸𑀢𑀯𑁂𑀤𑀁 𑀦𑀫𑀲𑁆𑀲𑀢𑀺𑁇_  


(2023.)

3375\. _𑀏𑀢𑁂 𑀦𑀻𑀮𑀸 𑀧𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀦𑀸𑀨𑀮𑀥𑀭𑀸 𑀤𑀼𑀫𑀸;_  
_𑀉𑀕𑁆𑀕𑀢𑀸 𑀅𑀩𑁆𑀪𑀓𑀽𑀝𑀸𑀯,_  
_𑀦𑀻𑀮𑀸 𑀅𑀜𑁆𑀚𑀦𑀧𑀩𑁆𑀩𑀢𑀸𑁇_  


(2024.)

3376\. _𑀥𑀯𑀲𑁆𑀲𑀓𑀡𑁆𑀡𑀸 𑀔𑀤𑀺𑀭𑀸,_  
_𑀲𑀸𑀮𑀸 𑀨𑀦𑁆𑀤𑀦𑀫𑀸𑀮𑀼𑀯𑀸;_  
_𑀲𑀫𑁆𑀧𑀯𑁂𑀥𑀦𑁆𑀢𑀺 𑀯𑀸𑀢𑁂𑀦,_  
_𑀲𑀓𑀺𑀁 𑀧𑀻𑀢𑀸𑀯 𑀫𑀸𑀡𑀯𑀸𑁇_  


(2025.)

3377\. _𑀉𑀧𑀭𑀺 𑀤𑀼𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀲𑀼,_  
_𑀲𑀗𑁆𑀕𑀻𑀢𑀺𑀬𑁄𑀯 𑀲𑀼𑀬𑁆𑀬𑀭𑁂;_  
_𑀦𑀚𑁆𑀚𑀼𑀳𑀸 𑀓𑁄𑀓𑀺𑀮𑀲𑀗𑁆𑀖𑀸,_  
_𑀲𑀫𑁆𑀧𑀢𑀦𑁆𑀢𑀺 𑀤𑀼𑀫𑀸 𑀤𑀼𑀫𑀁𑁇_  


(2026.)

3378\. _𑀅𑀯𑁆𑀳𑀬𑀦𑁆𑀢𑁂𑀯 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁,_  
_𑀲𑀸𑀔𑀸𑀧𑀢𑁆𑀢𑀲𑀫𑀻𑀭𑀺𑀢𑀸;_  
_𑀭𑀫𑀬𑀦𑁆𑀢𑁂𑀯 𑀆𑀕𑀦𑁆𑀢𑀁,_  
_𑀫𑁄𑀤𑀬𑀦𑁆𑀢𑀺 𑀦𑀺𑀯𑀸𑀲𑀺𑀦𑀁;_  
_𑀬𑀢𑁆𑀣 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀳 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


(2027.)

3379\. _𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀡𑁆𑀡𑀁,_  
_𑀆𑀲𑀤𑀜𑁆𑀘 𑀫𑀲𑀁 𑀚𑀝𑀁;_  
_𑀘𑀫𑁆𑀫𑀯𑀸𑀲𑀻 𑀙𑀫𑀸 𑀲𑁂𑀢𑀺,_  
_𑀚𑀸𑀢𑀯𑁂𑀤𑀁 𑀦𑀫𑀲𑁆𑀲𑀢𑀺𑁇_  


(2028.)

3380\. _𑀓𑀭𑁂𑀭𑀺𑀫𑀸𑀮𑀸 𑀯𑀺𑀢𑀢𑀸,_  
_𑀪𑀽𑀫𑀺𑀪𑀸𑀕𑁂 𑀫𑀦𑁄𑀭𑀫𑁂;_  
_𑀲𑀤𑁆𑀤𑀮𑀸𑀳𑀭𑀺𑀢𑀸 𑀪𑀽𑀫𑀺,_  
_𑀦 𑀢𑀢𑁆𑀣𑀼𑀤𑁆𑀥𑀁𑀲𑀢𑁂 𑀭𑀚𑁄𑁇_  


(2029.)

3381\. _𑀫𑀬𑀽𑀭𑀕𑀻𑀯𑀲𑀗𑁆𑀓𑀸𑀲𑀸,_  
_𑀢𑀽𑀮𑀨𑀲𑁆𑀲𑀲𑀫𑀽𑀧𑀫𑀸;_  
_𑀢𑀺𑀡𑀸𑀦𑀺 𑀦𑀸𑀢𑀺𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀼𑀮𑀸𑁇_  


(2030.)

3382\. _𑀅𑀫𑁆𑀩𑀸 𑀚𑀫𑁆𑀩𑀽 𑀓𑀧𑀺𑀢𑁆𑀣𑀸 𑀘,_  
_𑀦𑀻𑀘𑁂 𑀧𑀓𑁆𑀓𑀸 𑀘𑀼𑀤𑀼𑀫𑁆𑀩𑀭𑀸;_  
_𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀳𑀺 𑀭𑀼𑀓𑁆𑀔𑁂𑀳𑀺,_  
_𑀯𑀦𑀁 𑀢𑀁 𑀭𑀢𑀺𑀯𑀟𑁆𑀠𑀦𑀁𑁇_  


(2031.)

3383\. _𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀯𑀡𑁆𑀡𑀲𑀦𑁆𑀦𑀺𑀪𑀁,_  
_𑀫𑀘𑁆𑀙𑀕𑀼𑀫𑁆𑀩𑀦𑀺𑀲𑁂𑀯𑀺𑀢𑀁;_  
_𑀲𑀼𑀘𑀺𑀁 𑀲𑀼𑀕𑀦𑁆𑀥𑀁 𑀲𑀮𑀺𑀮𑀁,_  
_𑀆𑀧𑁄 𑀢𑀢𑁆𑀣𑀧𑀺 𑀲𑀦𑁆𑀤𑀢𑀺𑁇_  


(2032.)

3384\. _𑀢𑀲𑁆𑀲𑀸𑀯𑀺𑀤𑀽𑀭𑁂 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻,_

_𑀧𑀤𑀼𑀫𑀼𑀧𑁆𑀧𑀮𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_

(2033.)

3385\. _𑀢𑀻𑀡𑀺 𑀉𑀧𑁆𑀧𑀮𑀚𑀸𑀢𑀸𑀦𑀺,_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀭𑀲𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀦𑀻𑀮𑀸𑀦𑁂𑀓𑀸𑀦𑀺,_  
_𑀲𑁂𑀢𑀸 𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀘𑁇_  


(2034.)

_𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑁄 𑀦𑀸𑀫 𑀲𑁄 𑀲𑀭𑁄𑁇_  


(2035.)

3387\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀧𑀤𑀼𑀫𑀸 𑀨𑀼𑀮𑁆𑀮𑀸,_

_𑀕𑀺𑀫𑁆𑀳𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸 𑀨𑀼𑀮𑁆𑀮𑀸,_  
_𑀚𑀡𑁆𑀡𑀼𑀢𑀕𑁆𑀖𑀸 𑀉𑀧𑀢𑁆𑀣𑀭𑀸𑁇_  


(2036.)

3388\. _𑀲𑀼𑀭𑀪𑀻 𑀲𑀫𑁆𑀧𑀯𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀯𑀺𑀘𑀺𑀢𑁆𑀢𑀧𑀼𑀧𑁆𑀨𑀲𑀦𑁆𑀣𑀢𑀸;_  
_𑀪𑀫𑀭𑀸 𑀧𑀼𑀧𑁆𑀨𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀫𑀪𑀺𑀦𑀸𑀤𑀺𑀢𑀸𑁇_  


(2037.)

_𑀓𑀤𑀫𑁆𑀩𑀸 𑀧𑀸𑀝𑀮𑀻 𑀨𑀼𑀮𑁆𑀮𑀸,_

(2038.)

_𑀧𑀸𑀭𑀺𑀚𑀜𑁆𑀜𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀯𑀸𑀭𑀡𑀸 𑀯𑀬𑀦𑀸 𑀭𑀼𑀓𑁆𑀔𑀸,_  
_𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀫𑀼𑀪𑀢𑁄 𑀲𑀭𑀁𑁇_  


(2039.)

3391\. _𑀲𑀺𑀭𑀻𑀲𑀸 𑀲𑁂𑀢𑀧𑀸𑀭𑀺𑀲𑀸,_  
_𑀲𑀸𑀥𑀼 𑀯𑀸𑀬𑀦𑁆𑀢𑀺 𑀧𑀤𑁆𑀫𑀓𑀸;_  
_𑀦𑀺𑀕𑁆𑀕𑀼𑀡𑁆𑀟𑀻 𑀲𑀺𑀭𑀻𑀦𑀺𑀕𑁆𑀕𑀼𑀡𑁆𑀟𑀻,_  
_𑀅𑀲𑀦𑀸 𑀘𑁂𑀢𑁆𑀣 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸𑁇_  


(2040.)

3392\. _𑀧𑀗𑁆𑀕𑀼𑀭𑀸 𑀩𑀳𑀼𑀮𑀸 𑀲𑁂𑀮𑀸,_  
_𑀲𑁄𑀪𑀜𑁆𑀚𑀦𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀓𑁂𑀢𑀓𑀸 𑀓𑀡𑀺𑀓𑀸𑀭𑀸 𑀘,_  
_𑀓𑀦𑀯𑁂𑀭𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸𑁇_  


(2041.)

3393\. _𑀅𑀚𑁆𑀚𑀼𑀦𑀸 𑀅𑀚𑁆𑀚𑀼𑀓𑀡𑁆𑀡𑀸 𑀘,_  
_𑀫𑀳𑀸𑀦𑀸𑀫𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀲𑀼𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀕𑁆𑀕𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀧𑀚𑁆𑀚𑀮𑀦𑁆𑀢𑁂𑀯 𑀓𑀺𑀁𑀲𑀼𑀓𑀸𑁇_  


(2042.)

3394\. _𑀲𑁂𑀢𑀧𑀡𑁆𑀡𑀻 𑀲𑀢𑁆𑀢𑀧𑀡𑁆𑀡𑀸,_  
_𑀓𑀤𑀮𑀺𑀬𑁄 𑀓𑀼𑀲𑀼𑀫𑁆𑀪𑀭𑀸;_  
_𑀥𑀦𑀼𑀢𑀓𑁆𑀓𑀸𑀭𑀻 𑀧𑀼𑀧𑁆𑀨𑁂𑀳𑀺,_

_𑀲𑀮𑁆𑀮𑀓𑀺𑀬𑁄 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀲𑁂𑀢𑀕𑁂𑀭𑀼 𑀘 𑀢𑀕𑀭𑀸,_  
_𑀫𑀁𑀲𑀺𑀓𑀼𑀝𑁆𑀞𑀸 𑀓𑀼𑀮𑀸𑀯𑀭𑀸𑁇_  


(2044.)

3396\. _𑀤𑀳𑀭𑀸 𑀭𑀼𑀓𑁆𑀔𑀸 𑀘 𑀯𑀼𑀤𑁆𑀥𑀸 𑀘,_  
_𑀅𑀓𑀼𑀝𑀺𑀮𑀸 𑀘𑁂𑀢𑁆𑀣 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀅𑀲𑁆𑀲𑀫𑀁 𑀉𑀪𑀢𑁄 𑀞𑀦𑁆𑀢𑀺,_  
_𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑀦𑁆𑀢𑀢𑁄𑁇_  


(2045.)

3397\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀉𑀤𑀓𑀦𑁆𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀩𑀳𑀼𑀚𑀸𑀢𑁄 𑀨𑀡𑀺𑀚𑁆𑀚𑀓𑁄;_

(2046.)

3398\. _𑀉𑀤𑁆𑀤𑀸𑀧𑀯𑀢𑁆𑀢𑀁 𑀉𑀮𑁆𑀮𑀼𑀍𑀇𑀢𑀁,_  
_𑀫𑀓𑁆𑀔𑀺𑀓𑀸 𑀳𑀺𑀗𑁆𑀕𑀼𑀚𑀸𑀮𑀺𑀓𑀸;_  
_𑀤𑀸𑀲𑀺𑀫𑀓𑀜𑁆𑀚𑀓𑁄 𑀘𑁂𑀢𑁆𑀣,_  
_𑀩𑀳𑀽 𑀦𑀻𑀘𑁂𑀓𑀍𑀅𑀫𑁆𑀩𑀓𑀸𑁇_  


(2047.)

3399\. _𑀏𑀮𑀫𑁆𑀨𑀼𑀭𑀓𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀸,_  
_𑀭𑀼𑀓𑁆𑀔𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀥𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀕𑀦𑁆𑀥𑁄 𑀢𑁂𑀲𑀁 𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


(2048.)

3400\. _𑀉𑀪𑀢𑁄 𑀲𑀭𑀁 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀁,_

_𑀇𑀦𑁆𑀤𑀻𑀯𑀭𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀯𑀦𑀁 𑀢𑀁 𑀉𑀧𑀲𑁄𑀪𑀢𑀺𑁇_  


(2049.)

3401\. _𑀅𑀟𑁆𑀠𑀫𑀸𑀲𑀁 𑀥𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀕𑀦𑁆𑀥𑁄 𑀢𑁂𑀲𑀁 𑀦 𑀙𑀺𑀚𑁆𑀚𑀢𑀺;_  
_𑀦𑀻𑀮𑀧𑀼𑀧𑁆𑀨𑀻 𑀲𑁂𑀢𑀯𑀸𑀭𑀻,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀕𑀺𑀭𑀺𑀓𑀡𑁆𑀡𑀺𑀓𑀸;_  
_𑀓𑀮𑁂𑀭𑀼𑀓𑁆𑀔𑁂𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀯𑀦𑀁 𑀢𑀁 𑀢𑀼𑀮𑀲𑀻𑀳𑀺 𑀘𑁇_  


(2050.)

3402\. _𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀢𑁂𑀯 𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀧𑀼𑀧𑁆𑀨𑀲𑀸𑀔𑀸𑀳𑀺 𑀢𑀁 𑀯𑀦𑀁;_  
_𑀪𑀫𑀭𑀸 𑀧𑀼𑀧𑁆𑀨𑀕𑀦𑁆𑀥𑁂𑀦,_  
_𑀲𑀫𑀦𑁆𑀢𑀸 𑀫𑀪𑀺𑀦𑀸𑀤𑀺𑀢𑀸𑁇_  


(2051.)

3403\. _𑀢𑀻𑀡𑀺 𑀓𑀓𑁆𑀓𑀸𑀭𑀼𑀚𑀸𑀢𑀸𑀦𑀺,_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀭𑀲𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡;_  
_𑀓𑀼𑀫𑁆𑀪𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀘𑁂𑀓𑀸𑀦𑀺,_  
_𑀫𑀼𑀭𑀚𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀢𑀸 𑀉𑀪𑁄𑁇_  


(2052.)

3404\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀸𑀲𑀧𑁄 𑀩𑀳𑀼𑀓𑁄,_  
_𑀦𑀸𑀤𑀺𑀬𑁄 𑀳𑀭𑀺𑀢𑀸𑀬𑀼𑀢𑁄;_  
_𑀅𑀲𑀻 𑀢𑀸𑀮𑀸𑀯 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀙𑁂𑀚𑁆𑀚𑀸 𑀇𑀦𑁆𑀤𑀻𑀯𑀭𑀸 𑀩𑀳𑀽𑁇_  


(2053.)

3405\. _𑀅𑀧𑁆𑀨𑁄𑀝𑀸 𑀲𑀼𑀭𑀺𑀬𑀯𑀮𑁆𑀮𑀻 𑀘,_  
_𑀓𑀸𑀍𑀈𑀬𑀸 𑀫𑀥𑀼𑀕𑀦𑁆𑀥𑀺𑀬𑀸;_  
_𑀅𑀲𑁄𑀓𑀸 𑀫𑀼𑀤𑀬𑀦𑁆𑀢𑀻 𑀘,_  
_𑀯𑀮𑁆𑀮𑀺𑀪𑁄 𑀔𑀼𑀤𑁆𑀤𑀧𑀼𑀧𑁆𑀨𑀺𑀬𑁄𑁇_  


(2054.)

3406\. _𑀓𑁄𑀭𑀡𑁆𑀟𑀓𑀸 𑀅𑀦𑁄𑀚𑀸 𑀘,_  
_𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸 𑀦𑀸𑀕𑀫𑀮𑁆𑀮𑀺𑀓𑀸;_  
_𑀭𑀼𑀓𑁆𑀔𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺,_  
_𑀨𑀼𑀮𑁆𑀮𑀸 𑀓𑀺𑀁𑀲𑀼𑀓𑀯𑀮𑁆𑀮𑀺𑀬𑁄𑁇_  


(2055.)

3407\. _𑀓𑀝𑁂𑀭𑀼𑀳𑀸 𑀘 𑀯𑀸𑀲𑀦𑁆𑀢𑀻,_  
_𑀬𑀽𑀣𑀺𑀓𑀸 𑀫𑀥𑀼𑀕𑀦𑁆𑀥𑀺𑀬𑀸;_  
_𑀦𑀺𑀮𑀺𑀬𑀸 𑀲𑀼𑀫𑀦𑀸 𑀪𑀡𑁆𑀟𑀻,_  
_𑀲𑁄𑀪𑀢𑀺 𑀧𑀤𑀼𑀫𑀼𑀢𑁆𑀢𑀭𑁄𑁇_  


(2056.)

3408\. _𑀧𑀸𑀝𑀮𑀻 𑀲𑀫𑀼𑀤𑁆𑀤𑀓𑀧𑁆𑀧𑀸𑀲𑀻,_  
_𑀓𑀡𑀺𑀓𑀸𑀭𑀸 𑀘 𑀧𑀼𑀧𑁆𑀨𑀺𑀢𑀸;_  
_𑀳𑁂𑀫𑀚𑀸𑀮𑀸𑀯 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀭𑀼𑀘𑀺𑀭𑀕𑁆𑀕𑀺 𑀲𑀺𑀔𑀽𑀧𑀫𑀸𑁇_  


(2057.)

3409\. _𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀘 𑀧𑀼𑀧𑁆𑀨𑀸𑀦𑀺,_  
_𑀣𑀮𑀚𑀸𑀦𑀼𑀤𑀓𑀸𑀦𑀺 𑀘;_  
_𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀢𑀢𑁆𑀣 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀏𑀯𑀁 𑀭𑀫𑁆𑀫𑁄 𑀫𑀳𑁄𑀤𑀥𑀺𑁇_  


(2058.)

3410\. _𑀅𑀣𑀲𑁆𑀲𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑀸,_  
_𑀩𑀳𑀼𑀓𑀸 𑀯𑀸𑀭𑀺𑀕𑁄𑀘𑀭𑀸;_  
_𑀭𑁄𑀳𑀺𑀢𑀸 𑀦𑀍𑀅𑀧𑀻 𑀲𑀺𑀗𑁆𑀕𑀽,_  
_𑀓𑀼𑀫𑁆𑀪𑀺𑀮𑀸 𑀫𑀓𑀭𑀸 𑀲𑀼𑀲𑀽𑁇_  


(2059.)

3411\. _𑀫𑀥𑀼 𑀘 𑀫𑀥𑀼𑀮𑀝𑁆𑀞𑀺 𑀘,_  
_𑀢𑀸𑀮𑀺𑀲𑀸 𑀘 𑀧𑀺𑀬𑀗𑁆𑀕𑀼𑀓𑀸;_  
_𑀓𑀼𑀝𑀦𑁆𑀤𑀚𑀸 𑀪𑀤𑁆𑀤𑀫𑀼𑀢𑁆𑀢𑀸,_  
_𑀲𑁂𑀢𑀧𑀼𑀧𑁆𑀨𑀸 𑀘 𑀮𑁄𑀮𑀼𑀧𑀸𑁇_  


(2060.)

3412\. _𑀲𑀼𑀭𑀪𑀻 𑀘 𑀭𑀼𑀓𑁆𑀔𑀸 𑀢𑀕𑀭𑀸,_  
_𑀩𑀳𑀼𑀓𑀸 𑀢𑀼𑀗𑁆𑀕𑀯𑀡𑁆𑀝𑀓𑀸;_  
_𑀧𑀤𑁆𑀫𑀓𑀸 𑀦𑀭𑀤𑀸 𑀓𑀼𑀝𑁆𑀞𑀸,_  
_𑀛𑀸𑀫𑀓𑀸 𑀘 𑀳𑀭𑁂𑀡𑀼𑀓𑀸𑁇_  


(2061.)

3413\. _𑀳𑀮𑀺𑀤𑁆𑀤𑀓𑀸 𑀕𑀦𑁆𑀥𑀲𑀺𑀮𑀸,_  
_𑀳𑀺𑀭𑀺𑀯𑁂𑀭𑀸 𑀘 𑀕𑀼𑀕𑁆𑀕𑀼𑀮𑀸;_  
_𑀯𑀺𑀪𑁂𑀤𑀺𑀓𑀸 𑀘𑁄𑀭𑀓𑀸 𑀓𑀼𑀝𑁆𑀞𑀸,_  
_𑀓𑀧𑁆𑀧𑀽𑀭𑀸 𑀘 𑀓𑀮𑀺𑀗𑁆𑀕𑀼𑀓𑀸𑁇_  


(2062.)

3414\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀻𑀳𑀩𑁆𑀬𑀕𑁆𑀖𑀸 𑀘,_  
_𑀧𑀼𑀭𑀺𑀲𑀸𑀮𑀽 𑀘 𑀳𑀢𑁆𑀣𑀺𑀬𑁄;_  
_𑀏𑀡𑁂𑀬𑁆𑀬𑀸 𑀧𑀲𑀤𑀸 𑀘𑁂𑀯,_  
_𑀭𑁄𑀳𑀺𑀘𑁆𑀘𑀸 𑀲𑀭𑀪𑀸 𑀫𑀺𑀕𑀸𑁇_  


(2063.)

3415\. _𑀓𑁄𑀝𑁆𑀞𑀲𑀼𑀡𑀸 𑀲𑀼𑀡𑁄𑀧𑀺 𑀘,_  
_𑀢𑀼𑀮𑀺𑀬𑀸 𑀦𑀍𑀅𑀲𑀦𑁆𑀦𑀺𑀪𑀸;_  
_𑀘𑀸𑀫𑀭𑀻 𑀘𑀮𑀦𑀻 𑀮𑀗𑁆𑀖𑀻,_  
_𑀛𑀸𑀧𑀺𑀢𑀸 𑀫𑀓𑁆𑀓𑀝𑀸 𑀧𑀺𑀘𑀼𑁇_  


(2064.)

3416\. _𑀓𑀓𑁆𑀓𑀝𑀸 𑀓𑀝𑀫𑀸𑀬𑀸 𑀘,_  
_𑀇𑀓𑁆𑀓𑀸 𑀕𑁄𑀡𑀲𑀺𑀭𑀸 𑀩𑀳𑀽;_  
_𑀔𑀕𑁆𑀕𑀸 𑀯𑀭𑀸𑀳𑀸 𑀦𑀓𑀼𑀮𑀸,_  
_𑀓𑀸𑀍𑀅𑀓𑁂𑀢𑁆𑀣 𑀩𑀳𑀽𑀢𑀲𑁄𑁇_  


(2065.)

3417\. _𑀫𑀳𑀺𑀁𑀲𑀸 𑀲𑁄𑀡𑀲𑀺𑀗𑁆𑀕𑀸𑀮𑀸,_  
_𑀧𑀫𑁆𑀧𑀓𑀸 𑀘 𑀲𑀫𑀦𑁆𑀢𑀢𑁄;_  
_𑀆𑀓𑀼𑀘𑁆𑀙𑀸 𑀧𑀘𑀮𑀸𑀓𑀸 𑀘,_  
_𑀘𑀺𑀢𑁆𑀭𑀓𑀸 𑀘𑀸𑀧𑀺 𑀤𑀻𑀧𑀺𑀬𑁄𑁇_  


(2066.)

3418\. _𑀧𑁂𑀮𑀓𑀸 𑀘 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸,_  
_𑀲𑀻𑀳𑀸 𑀕𑁄𑀕𑀡𑀺𑀲𑀸𑀤𑀓𑀸;_  
_𑀅𑀝𑁆𑀞𑀧𑀸𑀤𑀸 𑀘 𑀫𑁄𑀭𑀸 𑀘,_  
_𑀪𑀲𑁆𑀲𑀭𑀸 𑀘 𑀓𑀼𑀓𑀼𑀢𑁆𑀣𑀓𑀸𑁇_  


(2067.)

3419\. _𑀘𑀗𑁆𑀓𑁄𑀭𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀸 𑀦𑀸𑀕𑀸,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀓𑀽𑀚𑀺𑀦𑁄;_  
_𑀩𑀓𑀸 𑀩𑀮𑀸𑀓𑀸 𑀦𑀚𑁆𑀚𑀼𑀳𑀸,_  
_𑀤𑀺𑀦𑁆𑀤𑀺𑀪𑀸 𑀓𑀼𑀜𑁆𑀚𑀯𑀸𑀚𑀺𑀢𑀸𑁇_  


(2068.)

3420\. _𑀩𑁆𑀬𑀕𑁆𑀖𑀺𑀦𑀲𑀸 𑀮𑁄𑀳𑀧𑀺𑀝𑁆𑀞𑀸,_  
_𑀧𑀫𑁆𑀫𑀓𑀸 𑀚𑀻𑀯𑀚𑀻𑀯𑀓𑀸;_  
_𑀓𑀧𑀺𑀜𑁆𑀚𑀭𑀸 𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀸𑀬𑁄,_  
_𑀓𑀼𑀮𑀸 𑀘 𑀧𑀝𑀺𑀓𑀼𑀢𑁆𑀣𑀓𑀸𑁇_  


(2069.)

3421\. _𑀫𑀦𑁆𑀤𑀸𑀮𑀓𑀸 𑀘𑁂𑀮𑀓𑁂𑀝𑀼,_  
_𑀪𑀡𑁆𑀟𑀼𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀦𑀸𑀫𑀓𑀸;_  
_𑀘𑁂𑀮𑀸𑀯𑀓𑀸 𑀧𑀺𑀗𑁆𑀕𑀮𑀸𑀬𑁄,_  
_𑀕𑁄𑀝𑀓𑀸 𑀅𑀗𑁆𑀕𑀳𑁂𑀢𑀼𑀓𑀸𑁇_  


(2070.)

3422\. _𑀓𑀭𑀯𑀺𑀬𑀸 𑀘 𑀲𑀕𑁆𑀕𑀸 𑀘,_  
_𑀉𑀳𑀼𑀗𑁆𑀓𑀸𑀭𑀸 𑀘 𑀓𑀼𑀓𑁆𑀓𑀼𑀳𑀸;_  
_𑀦𑀸𑀦𑀸𑀤𑀺𑀚𑀕𑀡𑀸𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀦𑀸𑀦𑀸𑀲𑀭𑀦𑀺𑀓𑀽𑀚𑀺𑀢𑀁𑁇_  


(2071.)

3423\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀓𑀼𑀡𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀻𑀮𑀓𑀸 𑀫𑀜𑁆𑀚𑀼𑀪𑀸𑀡𑀓𑀸;_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀲𑀳 𑀪𑀭𑀺𑀬𑀸𑀳𑀺,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀓𑀽𑀚𑀺𑀦𑁄𑁇_  


(2072.)

3424\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀓𑀼𑀡𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑀺𑀚𑀸 𑀫𑀜𑁆𑀚𑀼𑀲𑁆𑀲𑀭𑀸 𑀲𑀺𑀢𑀸;_  
_𑀲𑁂𑀢𑀘𑁆𑀙𑀺𑀓𑀽𑀝𑀸 𑀪𑀤𑁆𑀭𑀓𑁆𑀔𑀸,_  
_𑀅𑀡𑁆𑀟𑀚𑀸 𑀘𑀺𑀢𑁆𑀭𑀧𑁂𑀔𑀼𑀡𑀸𑁇_  


(2073.)

3425\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀓𑀼𑀡𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀤𑀺𑀚𑀸 𑀫𑀜𑁆𑀚𑀼𑀲𑁆𑀲𑀭𑀸 𑀲𑀺𑀢𑀸;_  
_𑀲𑀺𑀔𑀡𑁆𑀟𑀻 𑀦𑀻𑀮𑀕𑀻𑀯𑀸𑀳𑀺,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀓𑀽𑀚𑀺𑀦𑁄𑁇_  


(2074.)

3426\. _𑀓𑀼𑀓𑀼𑀢𑁆𑀣𑀓𑀸 𑀓𑀼𑀍𑀈𑀭𑀓𑀸,_  
_𑀓𑁄𑀝𑁆𑀞𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀲𑀸𑀢𑀓𑀸;_  
_𑀓𑀸𑀮𑀸𑀫𑁂𑀬𑁆𑀬𑀸 𑀩𑀮𑀺𑀬𑀓𑁆𑀔𑀸,_  
_𑀓𑀤𑀫𑁆𑀩𑀸 𑀲𑀼𑀯𑀲𑀸𑀍𑀇𑀓𑀸𑁇_  


(2075.)

3427\. _𑀳𑀮𑀺𑀤𑁆𑀤𑀸 𑀮𑁄𑀳𑀺𑀢𑀸 𑀲𑁂𑀢𑀸,_  
_𑀅𑀣𑁂𑀢𑁆𑀣 𑀦𑀮𑀓𑀸 𑀩𑀳𑀽;_  
_𑀯𑀸𑀭𑀡𑀸 𑀪𑀺𑀗𑁆𑀕𑀭𑀸𑀚𑀸 𑀘,_  
_𑀓𑀤𑀫𑁆𑀩𑀸 𑀲𑀼𑀯𑀓𑁄𑀓𑀺𑀮𑀸𑁇_  


(2076.)

3428\. _𑀉𑀓𑁆𑀓𑀼𑀲𑀸 𑀓𑀼𑀭𑀭𑀸 𑀳𑀁𑀲𑀸,_  
_𑀆𑀝𑀸 𑀧𑀭𑀺𑀯𑀤𑁂𑀦𑁆𑀢𑀺𑀓𑀸;_  
_𑀧𑀸𑀓𑀳𑀁𑀲𑀸 𑀅𑀢𑀺𑀩𑀮𑀸,_  
_𑀦𑀚𑁆𑀚𑀼𑀳𑀸 𑀚𑀻𑀯𑀚𑀻𑀯𑀓𑀸𑁇_  


(2077.)

3429\. _𑀧𑀸𑀭𑁂𑀯𑀢𑀸 𑀭𑀯𑀺𑀳𑀁𑀲𑀸,_  
_𑀘𑀓𑁆𑀓𑀯𑀸𑀓𑀸 𑀦𑀤𑀻𑀘𑀭𑀸;_  
_𑀯𑀸𑀭𑀡𑀸𑀪𑀺𑀭𑀼𑀤𑀸 𑀭𑀫𑁆𑀫𑀸,_  
_𑀉𑀪𑁄 𑀓𑀸𑀮𑀽𑀧𑀓𑀽𑀚𑀺𑀦𑁄𑁇_  


(2078.)

3430\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀓𑀼𑀡𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀦𑀸𑀯𑀡𑁆𑀡𑀸 𑀩𑀳𑀽 𑀤𑀺𑀚𑀸;_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀲𑀳 𑀪𑀭𑀺𑀬𑀸𑀳𑀺,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀓𑀽𑀚𑀺𑀦𑁄𑁇_  


(2079.)

3431\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀓𑀼𑀡𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀦𑀸𑀯𑀡𑁆𑀡𑀸 𑀩𑀳𑀽 𑀤𑀺𑀚𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀫𑀜𑁆𑀚𑀽 𑀦𑀺𑀓𑀽𑀚𑀦𑁆𑀢𑀺,_  
_𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀫𑀼𑀪𑀢𑁄 𑀲𑀭𑀁𑁇_  


(2080.)

3432\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀓𑀼𑀡𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀓𑀭𑀯𑀺𑀬𑀸 𑀦𑀸𑀫 𑀢𑁂 𑀤𑀺𑀚𑀸;_  
_𑀫𑁄𑀤𑀦𑁆𑀢𑀺 𑀲𑀳 𑀪𑀭𑀺𑀬𑀸𑀳𑀺,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀓𑀽𑀚𑀺𑀦𑁄𑁇_  


(2081.)

3433\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀓𑀼𑀡𑀸 𑀲𑀦𑁆𑀢𑀺,_  
_𑀓𑀭𑀯𑀺𑀬𑀸 𑀦𑀸𑀫 𑀢𑁂 𑀤𑀺𑀚𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀫𑀜𑁆𑀚𑀽 𑀦𑀺𑀓𑀽𑀚𑀦𑁆𑀢𑀺,_  
_𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀫𑀼𑀪𑀢𑁄 𑀲𑀭𑀁𑁇_  


(2082.)

3434\. _𑀏𑀡𑁂𑀬𑁆𑀬𑀧𑀲𑀤𑀸𑀓𑀺𑀡𑁆𑀡𑀁,_  
_𑀦𑀸𑀕𑀲𑀁𑀲𑁂𑀯𑀺𑀢𑀁 𑀯𑀦𑀁;_  
_𑀦𑀸𑀦𑀸𑀮𑀢𑀸𑀳𑀺 𑀲𑀜𑁆𑀙𑀦𑁆𑀦𑀁,_  
_𑀓𑀤𑀮𑀻𑀫𑀺𑀕𑀲𑁂𑀯𑀺𑀢𑀁𑁇_  


(2083.)

3435\. _𑀅𑀣𑁂𑀢𑁆𑀣 𑀲𑀸𑀲𑀧𑁄 𑀩𑀳𑀼𑀓𑁄,_  
_𑀦𑀻𑀯𑀸𑀭𑁄 𑀯𑀭𑀓𑁄 𑀩𑀳𑀼;_  
_𑀲𑀸𑀮𑀺 𑀅𑀓𑀝𑁆𑀞𑀧𑀸𑀓𑁄 𑀘,_  
_𑀉𑀘𑁆𑀙𑀼 𑀢𑀢𑁆𑀣 𑀅𑀦𑀧𑁆𑀧𑀓𑁄𑁇_  


(2084.)

3436\. _𑀅𑀬𑀁 𑀏𑀓𑀧𑀤𑀻 𑀏𑀢𑀺,_  
_𑀉𑀚𑀼𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀅𑀲𑁆𑀲𑀫𑀁;_  
_𑀔𑀼𑀤𑀁 𑀧𑀺𑀧𑀸𑀲𑀁 𑀅𑀭𑀢𑀺𑀁,_  
_𑀢𑀢𑁆𑀣 𑀧𑀢𑁆𑀢𑁄 𑀦 𑀯𑀺𑀦𑁆𑀤𑀢𑀺;_  
_𑀬𑀢𑁆𑀣 𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑁄 𑀭𑀸𑀚𑀸,_  
_𑀲𑀳 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


(2085.)

3437\. _𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀯𑀡𑁆𑀡𑀁,_

_𑀚𑀸𑀢𑀯𑁂𑀤𑀁 𑀦𑀫𑀲𑁆𑀲𑀢𑀺”𑁇_
