# 𑁨𑁨𑁇𑁧𑁇𑁧 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀚𑀸𑀢𑀓

* 𑀫𑀳𑀸𑀦𑀺𑀧𑀸𑀢

* 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀯𑀕𑁆𑀕

* 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀚𑀸𑀢𑀓

(1.)

1315\. _“𑀫𑀸 𑀧𑀡𑁆𑀟𑀺𑀘𑁆𑀘𑀬𑀁 𑀯𑀺𑀪𑀸𑀯𑀬,_  
_𑀩𑀸𑀮𑀫𑀢𑁄 𑀪𑀯 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑀁;_  
_𑀲𑀩𑁆𑀩𑁄 𑀢𑀁 𑀚𑀦𑁄 𑀑𑀘𑀺𑀦𑀸𑀬𑀢𑀼,_  
_𑀏𑀯𑀁 𑀢𑀯 𑀅𑀢𑁆𑀣𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺”𑁇_  


(2.)

1316\. _“𑀓𑀭𑁄𑀫𑀺 𑀢𑁂 𑀢𑀁 𑀯𑀘𑀦𑀁,_  
_𑀬𑀁 𑀫𑀁 𑀪𑀡𑀲𑀺 𑀤𑁂𑀯𑀢𑁂;_  
_𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸𑀲𑀺 𑀫𑁂 𑀅𑀫𑁆𑀫,_  
_𑀳𑀺𑀢𑀓𑀸𑀫𑀸𑀲𑀺 𑀤𑁂𑀯𑀢𑁂”𑁇_  


(3.)

1317\. _“𑀓𑀺𑀁 𑀦𑀼 𑀲𑀦𑁆𑀢𑀭𑀫𑀸𑀦𑁄𑀯,_  
_𑀓𑀸𑀲𑀼𑀁 𑀔𑀡𑀲𑀺 𑀲𑀸𑀭𑀣𑀺;_  
_𑀧𑀼𑀝𑁆𑀞𑁄 𑀫𑁂 𑀲𑀫𑁆𑀫 𑀅𑀓𑁆𑀔𑀸𑀳𑀺,_  
_𑀓𑀺𑀁 𑀓𑀸𑀲𑀼𑀬𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(4.)

1318\. _“𑀭𑀜𑁆𑀜𑁄 𑀫𑀽𑀕𑁄 𑀘 𑀧𑀓𑁆𑀔𑁄 𑀘,_  
_𑀧𑀼𑀢𑁆𑀢𑁄 𑀚𑀸𑀢𑁄 𑀅𑀘𑁂𑀢𑀲𑁄;_  
_𑀲𑁄𑀫𑁆𑀳𑀺 𑀭𑀜𑁆𑀜𑀸 𑀲𑀫𑀚𑁆𑀛𑀺𑀝𑁆𑀞𑁄,_  
_𑀧𑀼𑀢𑁆𑀢𑀁 𑀫𑁂 𑀦𑀺𑀔𑀡𑀁 𑀯𑀦𑁂”𑁇_  


(5.)

1319\. _“𑀦 𑀩𑀥𑀺𑀭𑁄 𑀦 𑀫𑀽𑀕𑁄𑀲𑁆𑀫𑀺,_  
_𑀦 𑀧𑀓𑁆𑀔𑁄 𑀦 𑀘 𑀯𑀻𑀓𑀮𑁄;_  
_𑀅𑀥𑀫𑁆𑀫𑀁 𑀲𑀸𑀭𑀣𑀺 𑀓𑀬𑀺𑀭𑀸,_  
_𑀫𑀜𑁆𑀘𑁂 𑀢𑁆𑀯𑀁 𑀦𑀺𑀔𑀡𑀁 𑀯𑀦𑁂𑁇_  


(6.)

1320\. _𑀊𑀭𑀽 𑀩𑀸𑀳𑀼𑀜𑁆𑀘 𑀫𑁂 𑀧𑀲𑁆𑀲,_  
_𑀪𑀸𑀲𑀺𑀢𑀜𑁆𑀘 𑀲𑀼𑀡𑁄𑀳𑀺 𑀫𑁂;_  
_𑀅𑀥𑀫𑁆𑀫𑀁 𑀲𑀸𑀭𑀣𑀺 𑀓𑀬𑀺𑀭𑀸,_  
_𑀫𑀜𑁆𑀘𑁂 𑀢𑁆𑀯𑀁 𑀦𑀺𑀔𑀡𑀁 𑀯𑀦𑁂”𑁇_  


(7.)

1321\. _“𑀤𑁂𑀯𑀢𑀸 𑀦𑀼𑀲𑀺 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀅𑀤𑀼 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀓𑁄 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀓𑀣𑀁 𑀚𑀸𑀦𑁂𑀫𑀼 𑀢𑀁 𑀫𑀬𑀁”𑁇_  


(8.)

1322\. _“𑀦𑀫𑁆𑀳𑀺 𑀤𑁂𑀯𑁄 𑀦 𑀕𑀦𑁆𑀥𑀩𑁆𑀩𑁄,_  
_𑀦𑀸𑀧𑀺 𑀲𑀓𑁆𑀓𑁄 𑀧𑀼𑀭𑀺𑀦𑁆𑀤𑀤𑁄;_  
_𑀓𑀸𑀲𑀺𑀭𑀜𑁆𑀜𑁄 𑀅𑀳𑀁 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀬𑀁 𑀓𑀸𑀲𑀼𑀬𑀸 𑀦𑀺𑀔𑀜𑁆𑀜𑀲𑀺𑁇_  


(9.)

1323\. _𑀢𑀲𑁆𑀲 𑀭𑀜𑁆𑀜𑁄 𑀅𑀳𑀁 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀬𑀁 𑀢𑁆𑀯𑀁 𑀲𑀫𑁆𑀫𑀽𑀧𑀚𑀻𑀯𑀲𑀺;_  
_𑀅𑀥𑀫𑁆𑀫𑀁 𑀲𑀸𑀭𑀣𑀺 𑀓𑀬𑀺𑀭𑀸,_  
_𑀫𑀜𑁆𑀘𑁂 𑀢𑁆𑀯𑀁 𑀦𑀺𑀔𑀡𑀁 𑀯𑀦𑁂𑁇_  


(10.)

1324\. _𑀬𑀲𑁆𑀲 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀙𑀸𑀬𑀸𑀬,_  
_𑀦𑀺𑀲𑀻𑀤𑁂𑀬𑁆𑀬 𑀲𑀬𑁂𑀬𑁆𑀬 𑀯𑀸;_  
_𑀦 𑀢𑀲𑁆𑀲 𑀲𑀸𑀔𑀁 𑀪𑀜𑁆𑀚𑁂𑀬𑁆𑀬,_  
_𑀫𑀺𑀢𑁆𑀢𑀤𑀼𑀩𑁆𑀪𑁄 𑀳𑀺 𑀧𑀸𑀧𑀓𑁄𑁇_  


(11.)

1325\. _𑀬𑀣𑀸 𑀭𑀼𑀓𑁆𑀔𑁄 𑀢𑀣𑀸 𑀭𑀸𑀚𑀸,_  
_𑀬𑀣𑀸 𑀲𑀸𑀔𑀸 𑀢𑀣𑀸 𑀅𑀳𑀁;_  
_𑀬𑀣𑀸 𑀙𑀸𑀬𑀽𑀧𑀕𑁄 𑀧𑁄𑀲𑁄,_  
_𑀏𑀯𑀁 𑀢𑁆𑀯𑀫𑀲𑀺 𑀲𑀸𑀭𑀣𑀺;_  
_𑀅𑀥𑀫𑁆𑀫𑀁 𑀲𑀸𑀭𑀣𑀺 𑀓𑀬𑀺𑀭𑀸,_  
_𑀫𑀜𑁆𑀘𑁂 𑀢𑁆𑀯𑀁 𑀦𑀺𑀔𑀡𑀁 𑀯𑀦𑁂𑁇_  


(12.)

1326\. _𑀧𑀳𑀽𑀢𑀪𑀓𑁆𑀔𑁄 𑀪𑀯𑀢𑀺,_  
_𑀯𑀺𑀧𑁆𑀧𑀯𑀼𑀢𑁆𑀣𑁄 𑀲𑀓𑀁 𑀖𑀭𑀸;_  
_𑀩𑀳𑀽 𑀦𑀁 𑀉𑀧𑀚𑀻𑀯𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(13.)

1327\. _𑀬𑀁 𑀬𑀁 𑀚𑀦𑀧𑀤𑀁 𑀬𑀸𑀢𑀺,_  
_𑀦𑀺𑀕𑀫𑁂 𑀭𑀸𑀚𑀥𑀸𑀦𑀺𑀬𑁄;_  
_𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀧𑀽𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(14.)

1328\. _𑀦𑀸𑀲𑁆𑀲 𑀘𑁄𑀭𑀸 𑀧𑀲𑀸𑀳𑀦𑁆𑀢𑀺,_  
_𑀦𑀸𑀢𑀺𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑀸;_  
_𑀲𑀩𑁆𑀩𑁂 𑀅𑀫𑀺𑀢𑁆𑀢𑁂 𑀢𑀭𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(15.)

1329\. _𑀅𑀓𑁆𑀓𑀼𑀤𑁆𑀥𑁄 𑀲𑀖𑀭𑀁 𑀏𑀢𑀺,_  
_𑀲𑀪𑀸𑀬𑀁 𑀧𑀝𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁄;_  
_𑀜𑀸𑀢𑀻𑀦𑀁 𑀉𑀢𑁆𑀢𑀫𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(16.)

1330\. _𑀲𑀓𑁆𑀓𑀢𑁆𑀯𑀸 𑀲𑀓𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀕𑀭𑀼 𑀳𑁄𑀢𑀺 𑀲𑀕𑀸𑀭𑀯𑁄;_  
_𑀯𑀡𑁆𑀡𑀓𑀺𑀢𑁆𑀢𑀺𑀪𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(17.)

1331\. _𑀧𑀽𑀚𑀓𑁄 𑀮𑀪𑀢𑁂 𑀧𑀽𑀚𑀁,_  
_𑀯𑀦𑁆𑀤𑀓𑁄 𑀧𑀝𑀺𑀯𑀦𑁆𑀤𑀦𑀁;_  
_𑀬𑀲𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀜𑁆𑀘 𑀧𑀧𑁆𑀧𑁄𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(18.)

1332\. _𑀅𑀕𑁆𑀕𑀺 𑀬𑀣𑀸 𑀧𑀚𑁆𑀚𑀮𑀢𑀺,_  
_𑀤𑁂𑀯𑀢𑀸𑀯 𑀯𑀺𑀭𑁄𑀘𑀢𑀺;_  
_𑀲𑀺𑀭𑀺𑀬𑀸 𑀅𑀚𑀳𑀺𑀢𑁄 𑀳𑁄𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(19.)

1333\. _𑀕𑀸𑀯𑁄 𑀢𑀲𑁆𑀲 𑀧𑀚𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀔𑁂𑀢𑁆𑀢𑁂 𑀯𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀭𑀽𑀳𑀢𑀺;_  
_𑀯𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀨𑀮𑀫𑀲𑁆𑀦𑀸𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(20.)

1334\. _𑀤𑀭𑀺𑀢𑁄 𑀧𑀩𑁆𑀩𑀢𑀸𑀢𑁄 𑀯𑀸,_  
_𑀭𑀼𑀓𑁆𑀔𑀢𑁄 𑀧𑀢𑀺𑀢𑁄 𑀦𑀭𑁄;_  
_𑀘𑀼𑀢𑁄 𑀧𑀢𑀺𑀝𑁆𑀞𑀁 𑀮𑀪𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺𑁇_  


(21.)

1335\. _𑀯𑀺𑀭𑀽𑀍𑀳𑀫𑀽𑀮𑀲𑀦𑁆𑀢𑀸𑀦𑀁,_  
_𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀫𑀺𑀯 𑀫𑀸𑀮𑀼𑀢𑁄;_  
_𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀦𑀧𑁆𑀧𑀲𑀸𑀳𑀦𑁆𑀢𑀺,_  
_𑀬𑁄 𑀫𑀺𑀢𑁆𑀢𑀸𑀦𑀁 𑀦 𑀤𑀼𑀩𑁆𑀪𑀢𑀺”𑁇_  


(22.)

1336\. _“𑀏𑀳𑀺 𑀢𑀁 𑀧𑀝𑀺𑀦𑁂𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀲𑀓𑀁 𑀖𑀭𑀁;_  
_𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀓𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(23.)

1337\. _“𑀅𑀮𑀁 𑀫𑁂 𑀢𑁂𑀦 𑀭𑀚𑁆𑀚𑁂𑀦,_  
_𑀜𑀸𑀢𑀓𑁂𑀳𑀺 𑀥𑀦𑁂𑀦 𑀯𑀸;_  
_𑀬𑀁 𑀫𑁂 𑀅𑀥𑀫𑁆𑀫𑀘𑀭𑀺𑀬𑀸𑀬,_  
_𑀭𑀚𑁆𑀚𑀁 𑀮𑀩𑁆𑀪𑁂𑀣 𑀲𑀸𑀭𑀣𑀺”𑁇_  


(24.)

1338\. _“𑀧𑀼𑀡𑁆𑀡𑀧𑀢𑁆𑀢𑀁 𑀫𑀁 𑀮𑀸𑀪𑁂𑀳𑀺,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀇𑀢𑁄 𑀕𑀢𑁄;_  
_𑀧𑀺𑀢𑀸 𑀫𑀸𑀢𑀸 𑀘 𑀫𑁂 𑀤𑀚𑁆𑀚𑀼𑀁,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀢𑀬𑀻 𑀕𑀢𑁂𑁇_  


(25.)

1339\. _𑀑𑀭𑁄𑀥𑀸 𑀘 𑀓𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀢𑁂𑀧𑀺 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀤𑀚𑁆𑀚𑀼𑀁,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀢𑀬𑀻 𑀕𑀢𑁂𑁇_  


(26.)

1340\. _𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀢𑁂𑀧𑀺 𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀤𑀚𑁆𑀚𑀼𑀁,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀢𑀬𑀻 𑀕𑀢𑁂𑁇_  


(27.)

1341\. _𑀩𑀳𑀼𑀥𑀜𑁆𑀜𑀸 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀉𑀧𑀸𑀬𑀦𑀸𑀦𑀺 𑀫𑁂 𑀤𑀚𑁆𑀚𑀼𑀁,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀢𑀬𑀻 𑀕𑀢𑁂”𑁇_  


(28.)

1342\. _“𑀧𑀺𑀢𑀼 𑀫𑀸𑀢𑀼 𑀘𑀳𑀁 𑀘𑀢𑁆𑀢𑁄,_  
_𑀭𑀝𑁆𑀞𑀲𑁆𑀲 𑀦𑀺𑀕𑀫𑀲𑁆𑀲 𑀘;_  
_𑀅𑀣𑁄 𑀲𑀩𑁆𑀩𑀓𑀼𑀫𑀸𑀭𑀸𑀦𑀁,_  
_𑀦𑀢𑁆𑀣𑀺 𑀫𑀬𑁆𑀳𑀁 𑀲𑀓𑀁 𑀖𑀭𑀁𑁇_  


(29.)

1343\. _𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄 𑀅𑀳𑀁 𑀫𑀢𑁆𑀬𑀸,_  
_𑀲𑀜𑁆𑀘𑀢𑁆𑀢𑁄 𑀧𑀺𑀢𑀭𑀸 𑀫𑀳𑀁;_  
_𑀏𑀓𑁄𑀭𑀜𑁆𑀜𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄,_  
_𑀦 𑀓𑀸𑀫𑁂 𑀅𑀪𑀺𑀧𑀢𑁆𑀣𑀬𑁂𑁇_  


(30.)

1344\. _𑀅𑀧𑀺 𑀅𑀢𑀭𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀨𑀮𑀸𑀲𑀸𑀯 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀯𑀺𑀧𑀓𑁆𑀓𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁄𑀲𑁆𑀫𑀺,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀲𑀸𑀭𑀣𑀺𑁇_  


(31.)

1345\. _𑀅𑀧𑀺 𑀅𑀢𑀭𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀲𑀫𑁆𑀫𑀤𑀢𑁆𑀣𑁄 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀺;_  
_𑀯𑀺𑀧𑀓𑁆𑀓𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁄𑀲𑁆𑀫𑀺,_  
_𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑁄”𑁇_  


(32.)

1346\. _“𑀏𑀯𑀁 𑀯𑀕𑁆𑀕𑀼𑀓𑀣𑁄 𑀲𑀦𑁆𑀢𑁄,_  
_𑀯𑀺𑀲𑀝𑁆𑀞𑀯𑀘𑀦𑁄 𑀘𑀲𑀺;_  
_𑀓𑀲𑁆𑀫𑀸 𑀧𑀺𑀢𑀼 𑀘 𑀫𑀸𑀢𑀼𑀘𑁆𑀘,_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦 𑀪𑀡𑀻 𑀢𑀤𑀸”𑁇_  


(33.)

1347\. _“𑀦𑀸𑀳𑀁 𑀅𑀲𑀦𑁆𑀥𑀺𑀢𑀸 𑀧𑀓𑁆𑀔𑁄,_  
_𑀦 𑀩𑀥𑀺𑀭𑁄 𑀅𑀲𑁄𑀢𑀢𑀸;_  
_𑀦𑀸𑀳𑀁 𑀅𑀚𑀺𑀯𑁆𑀳𑀢𑀸 𑀫𑀽𑀕𑁄,_  
_𑀫𑀸 𑀫𑀁 𑀫𑀽𑀕𑀫𑀥𑀸𑀭𑀬𑀺𑁇_  


(34.)

1348\. _𑀧𑀼𑀭𑀺𑀫𑀁 𑀲𑀭𑀸𑀫𑀳𑀁 𑀚𑀸𑀢𑀺𑀁,_  
_𑀬𑀢𑁆𑀣 𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀓𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀳𑀺𑀁 𑀭𑀚𑁆𑀚𑀁,_  
_𑀧𑀸𑀧𑀢𑁆𑀣𑀁 𑀦𑀺𑀭𑀬𑀁 𑀪𑀼𑀲𑀁𑁇_  


(35.)

1349\. _𑀯𑀻𑀲𑀢𑀺𑀜𑁆𑀘𑁂𑀯 𑀯𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀳𑀺𑀁 𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺𑀁;_  
_𑀅𑀲𑀻𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀅𑀧𑀘𑁆𑀘𑀺𑀲𑀁𑁇_  


(36.)

1350\. _𑀢𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀲𑁆𑀲𑀳𑀁 𑀪𑀻𑀢𑁄,_  
_𑀫𑀸 𑀫𑀁 𑀭𑀚𑁆𑀚𑀸𑀪𑀺𑀲𑁂𑀘𑀬𑀼𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀺𑀢𑀼 𑀘 𑀫𑀸𑀢𑀼𑀘𑁆𑀘,_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦 𑀪𑀡𑀺𑀁 𑀢𑀤𑀸𑁇_  


(37.)

1351\. _𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂 𑀫𑀁 𑀦𑀺𑀲𑀸𑀤𑁂𑀢𑁆𑀯𑀸,_  
_𑀧𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀸𑀦𑀼𑀲𑀸𑀲𑀢𑀺;_  
_𑀏𑀓𑀁 𑀳𑀦𑀣 𑀩𑀦𑁆𑀥𑀣,_  
_𑀏𑀓𑀁 𑀔𑀸𑀭𑀸𑀧𑀢𑀘𑁆𑀙𑀺𑀓𑀁;_  
_𑀏𑀓𑀁 𑀲𑀽𑀮𑀲𑁆𑀫𑀺𑀁 𑀉𑀧𑁆𑀧𑁂𑀣,_  
_𑀇𑀘𑁆𑀘𑀲𑁆𑀲 𑀫𑀦𑀼𑀲𑀸𑀲𑀢𑀺𑁇_  


(38.)

1352\. _𑀢𑀸𑀬𑀸𑀳𑀁 𑀨𑀭𑀼𑀲𑀁 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀯𑀸𑀘𑀸𑀬𑁄 𑀲𑀫𑀼𑀤𑀻𑀭𑀺𑀢𑀸;_  
_𑀅𑀫𑀽𑀕𑁄 𑀫𑀽𑀕𑀯𑀡𑁆𑀡𑁂𑀦,_  
_𑀅𑀧𑀓𑁆𑀔𑁄 𑀧𑀓𑁆𑀔𑀲𑀫𑁆𑀫𑀢𑁄;_  
_𑀲𑀓𑁂 𑀫𑀼𑀢𑁆𑀢𑀓𑀭𑀻𑀲𑀲𑁆𑀫𑀺𑀁,_  
_𑀅𑀘𑁆𑀙𑀸𑀳𑀁 𑀲𑀫𑁆𑀧𑀭𑀺𑀧𑁆𑀮𑀼𑀢𑁄𑁇_  


(39.)

1353\. _𑀓𑀲𑀺𑀭𑀜𑁆𑀘 𑀧𑀭𑀺𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀢𑀜𑁆𑀘 𑀤𑀼𑀓𑁆𑀔𑁂𑀦 𑀲𑀁𑀬𑀼𑀢𑀁;_  
_𑀓𑁄𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀯𑁂𑀭𑀁 𑀓𑀬𑀺𑀭𑀸𑀣 𑀓𑁂𑀦𑀘𑀺𑁇_  


(40.)

1354\. _𑀧𑀜𑁆𑀜𑀸𑀬 𑀘 𑀅𑀮𑀸𑀪𑁂𑀦,_  
_𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀘 𑀅𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀓𑁄𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀫𑀸𑀕𑀫𑁆𑀫,_  
_𑀯𑁂𑀭𑀁 𑀓𑀬𑀺𑀭𑀸𑀣 𑀓𑁂𑀦𑀘𑀺𑁇_  


(41.)

1355\. _𑀅𑀧𑀺 𑀅𑀢𑀭𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀨𑀮𑀸𑀲𑀸𑀯 𑀲𑀫𑀺𑀚𑁆𑀛𑀢𑀺;_  
_𑀯𑀺𑀧𑀓𑁆𑀓𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁄𑀲𑁆𑀫𑀺,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀲𑀸𑀭𑀣𑀺𑁇_  


(42.)

1356\. _𑀅𑀧𑀺 𑀅𑀢𑀭𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀲𑀫𑁆𑀫𑀤𑀢𑁆𑀣𑁄 𑀯𑀺𑀧𑀘𑁆𑀘𑀢𑀺;_  
_𑀯𑀺𑀧𑀓𑁆𑀓𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑁄𑀲𑁆𑀫𑀺,_  
_𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁄 𑀅𑀓𑀼𑀢𑁄𑀪𑀬𑁄”𑁇_  


(43.)

1357\. _“𑀅𑀳𑀫𑁆𑀧𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢 𑀢𑀯𑀦𑁆𑀢𑀺𑀓𑁂;_  
_𑀅𑀯𑁆𑀳𑀸𑀬𑀲𑁆𑀲𑀼 𑀫𑀁 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸 𑀫𑀫 𑀭𑀼𑀘𑁆𑀘𑀢𑀺”𑁇_  


(44.)

1358\. _“𑀭𑀣𑀁 𑀦𑀺𑀬𑁆𑀬𑀸𑀤𑀬𑀺𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀦𑀡𑁄 𑀏𑀳𑀺 𑀲𑀸𑀭𑀣𑀺;_  
_𑀅𑀦𑀡𑀲𑁆𑀲 𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸,_  
_𑀏𑀢𑀁 𑀇𑀲𑀻𑀳𑀺 𑀯𑀡𑁆𑀡𑀺𑀢𑀁”𑁇_  


(45.)

1359\. _“𑀬𑀤𑁂𑀯 𑀢𑁆𑀬𑀸𑀳𑀁 𑀯𑀘𑀦𑀁,_  
_𑀅𑀓𑀭𑀁 𑀪𑀤𑁆𑀤𑀫𑀢𑁆𑀣𑀼 𑀢𑁂;_  
_𑀢𑀤𑁂𑀯 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀯𑀘𑀦𑀁,_  
_𑀬𑀸𑀘𑀺𑀢𑁄 𑀓𑀢𑁆𑀢𑀼𑀫𑀭𑀳𑀲𑀺𑁇_  


(46.)

1360\. _𑀇𑀥𑁂𑀯 𑀢𑀸𑀯 𑀅𑀘𑁆𑀙𑀲𑁆𑀲𑀼,_  
_𑀬𑀸𑀯 𑀭𑀸𑀚𑀸𑀦𑀫𑀸𑀦𑀬𑁂;_  
_𑀅𑀧𑁆𑀧𑁂𑀯 𑀢𑁂 𑀧𑀺𑀢𑀸 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀧𑀢𑀻𑀢𑁄 𑀲𑀼𑀫𑀦𑁄 𑀲𑀺𑀬𑀸”𑁇_  


(47.)

1361\. _“𑀓𑀭𑁄𑀫𑀺 𑀢𑁂𑀢𑀁 𑀯𑀘𑀦𑀁,_  
_𑀬𑀁 𑀫𑀁 𑀪𑀡𑀲𑀺 𑀲𑀸𑀭𑀣𑀺;_  
_𑀅𑀳𑀫𑁆𑀧𑀺 𑀤𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑁄𑀲𑁆𑀫𑀺,_  
_𑀧𑀺𑀢𑀭𑀁 𑀫𑁂 𑀇𑀥𑀸𑀕𑀢𑀁𑁇_  


(48.)

1362\. _𑀏𑀳𑀺 𑀲𑀫𑁆𑀫 𑀦𑀺𑀯𑀢𑁆𑀢𑀲𑁆𑀲𑀼,_  
_𑀓𑀼𑀲𑀮𑀁 𑀯𑀚𑁆𑀚𑀸𑀲𑀺 𑀜𑀸𑀢𑀺𑀦𑀁;_  
_𑀫𑀸𑀢𑀭𑀁 𑀧𑀺𑀢𑀭𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀯𑀼𑀢𑁆𑀢𑁄 𑀯𑀚𑁆𑀚𑀸𑀲𑀺 𑀯𑀦𑁆𑀤𑀦𑀁”𑁇_  


(49.)

1363\. _𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑁂 𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀢𑁆𑀯𑀸 𑀘 𑀦𑀁 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁;_  
_𑀲𑀸𑀭𑀣𑀺 𑀭𑀣𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀭𑀸𑀚𑀤𑁆𑀯𑀸𑀭𑀁 𑀉𑀧𑀸𑀕𑀫𑀺𑁇_  


(50.)

1364\. _𑀲𑀼𑀜𑁆𑀜𑀁 𑀫𑀸𑀢𑀸 𑀭𑀣𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀏𑀓𑀁 𑀲𑀸𑀭𑀣𑀺𑀫𑀸𑀕𑀢𑀁;_  
_𑀅𑀲𑁆𑀲𑀼𑀧𑀼𑀡𑁆𑀡𑁂𑀳𑀺 𑀦𑁂𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀭𑁄𑀤𑀦𑁆𑀢𑀻 𑀦𑀁 𑀉𑀤𑀺𑀓𑁆𑀔𑀢𑀺𑁇_  


(51.)

1365\. _“𑀅𑀬𑀁 𑀲𑁄 𑀲𑀸𑀭𑀣𑀺 𑀏𑀢𑀺,_  
_𑀦𑀺𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀫 𑀅𑀢𑁆𑀭𑀚𑀁;_  
_𑀦𑀺𑀳𑀢𑁄 𑀦𑀽𑀦 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑁄,_  
_𑀧𑀣𑀩𑁆𑀬𑀸 𑀪𑀽𑀫𑀺𑀯𑀟𑁆𑀠𑀦𑁄𑁇_  


(52.)

1366\. _𑀅𑀫𑀺𑀢𑁆𑀢𑀸 𑀦𑀽𑀦 𑀦𑀦𑁆𑀤𑀦𑁆𑀢𑀺,_  
_𑀧𑀢𑀻𑀢𑀸 𑀦𑀽𑀦 𑀯𑁂𑀭𑀺𑀦𑁄;_  
_𑀆𑀕𑀢𑀁 𑀲𑀸𑀭𑀣𑀺𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀦𑀺𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀫𑀫 𑀅𑀢𑁆𑀭𑀚𑀁”𑁇_  


(53.)

1367\. _𑀲𑀼𑀜𑁆𑀜𑀁 𑀫𑀸𑀢𑀸 𑀭𑀣𑀁 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀏𑀓𑀁 𑀲𑀸𑀭𑀣𑀺𑀫𑀸𑀕𑀢𑀁;_  
_𑀅𑀲𑁆𑀲𑀼𑀧𑀼𑀡𑁆𑀡𑁂𑀳𑀺 𑀦𑁂𑀢𑁆𑀢𑁂𑀳𑀺,_  
_𑀭𑁄𑀤𑀦𑁆𑀢𑀻 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀺 𑀦𑀁𑁇_  


(54.)

1368\. _“𑀓𑀺𑀁 𑀦𑀼 𑀫𑀽𑀕𑁄 𑀓𑀺𑀁 𑀦𑀼 𑀧𑀓𑁆𑀔𑁄,_  
_𑀓𑀺𑀁 𑀦𑀼 𑀲𑁄 𑀯𑀺𑀮𑀧𑀻 𑀢𑀤𑀸;_  
_𑀦𑀺𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀪𑀽𑀫𑀺𑀬𑀸,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀲𑀸𑀭𑀣𑀺𑁇_  


(55.)

1369\. _𑀓𑀣𑀁 𑀳𑀢𑁆𑀣𑁂𑀳𑀺 𑀧𑀸𑀤𑁂𑀳𑀺,_  
_𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑁄 𑀯𑀺𑀯𑀚𑁆𑀚𑀬𑀺;_  
_𑀦𑀺𑀳𑀜𑁆𑀜𑀫𑀸𑀦𑁄 𑀪𑀽𑀫𑀺𑀬𑀸,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄”𑁇_  


(56.)

1370\. _“𑀅𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀁 𑀢𑁂 𑀅𑀳𑀁 𑀅𑀬𑁆𑀬𑁂,_  
_𑀤𑀚𑁆𑀚𑀸𑀲𑀺 𑀅𑀪𑀬𑀁 𑀫𑀫;_  
_𑀬𑀁 𑀫𑁂 𑀲𑀼𑀢𑀁 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑁇_  


(57.)

1371\. _“𑀅𑀪𑀬𑀁 𑀲𑀫𑁆𑀫 𑀢𑁂 𑀤𑀫𑁆𑀫𑀺,_  
_𑀅𑀪𑀻𑀢𑁄 𑀪𑀡 𑀲𑀸𑀭𑀣𑀺;_  
_𑀬𑀁 𑀢𑁂 𑀲𑀼𑀢𑀁 𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀁 𑀯𑀸,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂”𑁇_  


(58.)

1372\. _“𑀦 𑀲𑁄 𑀫𑀽𑀕𑁄 𑀦 𑀲𑁄 𑀧𑀓𑁆𑀔𑁄,_  
_𑀯𑀺𑀲𑀝𑁆𑀞𑀯𑀘𑀦𑁄 𑀘 𑀲𑁄;_  
_𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀓𑀺𑀭 𑀲𑁄 𑀪𑀻𑀢𑁄,_  
_𑀅𑀓𑀭𑀸 𑀆𑀮𑀬𑁂 𑀩𑀳𑀽𑁇_  


(59.)

1373\. _𑀧𑀼𑀭𑀺𑀫𑀁 𑀲𑀭𑀢𑀺 𑀲𑁄 𑀚𑀸𑀢𑀺𑀁,_  
_𑀬𑀢𑁆𑀣 𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺;_  
_𑀓𑀸𑀭𑀬𑀺𑀢𑁆𑀯𑀸 𑀢𑀳𑀺𑀁 𑀭𑀚𑁆𑀚𑀁,_  
_𑀧𑀸𑀧𑀢𑁆𑀣 𑀦𑀺𑀭𑀬𑀁 𑀪𑀼𑀲𑀁𑁇_  


(60.)

1374\. _𑀯𑀻𑀲𑀢𑀺𑀜𑁆𑀘𑁂𑀯 𑀯𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀢𑀳𑀺𑀁 𑀭𑀚𑁆𑀚𑀫𑀓𑀸𑀭𑀬𑀺;_  
_𑀅𑀲𑀻𑀢𑀺𑀯𑀲𑁆𑀲𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺,_  
_𑀦𑀺𑀭𑀬𑀫𑁆𑀳𑀺 𑀅𑀧𑀘𑁆𑀘𑀺 𑀲𑁄𑁇_  


(61.)

1375\. _𑀢𑀲𑁆𑀲 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀲𑁄 𑀪𑀻𑀢𑁄,_  
_𑀫𑀸 𑀫𑀁 𑀭𑀚𑁆𑀚𑀸𑀪𑀺𑀲𑁂𑀘𑀬𑀼𑀁;_  
_𑀢𑀲𑁆𑀫𑀸 𑀧𑀺𑀢𑀼 𑀘 𑀫𑀸𑀢𑀼𑀘𑁆𑀘,_  
_𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦 𑀪𑀡𑀻 𑀢𑀤𑀸𑁇_  


(62.)

1376\. _𑀅𑀗𑁆𑀕𑀧𑀘𑁆𑀘𑀗𑁆𑀕𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄,_  
_𑀆𑀭𑁄𑀳𑀧𑀭𑀺𑀡𑀸𑀳𑀯𑀸;_  
_𑀯𑀺𑀲𑀝𑁆𑀞𑀯𑀘𑀦𑁄 𑀧𑀜𑁆𑀜𑁄,_  
_𑀫𑀕𑁆𑀕𑁂 𑀲𑀕𑁆𑀕𑀲𑁆𑀲 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇_  


(63.)

1377\. _𑀲𑀘𑁂 𑀢𑁆𑀯𑀁 𑀤𑀝𑁆𑀞𑀼𑀓𑀸𑀫𑀸𑀲𑀺,_  
_𑀭𑀸𑀚𑀧𑀼𑀢𑁆𑀢𑀁 𑀢𑀯𑀢𑁆𑀭𑀚𑀁;_  
_𑀏𑀳𑀺 𑀢𑀁 𑀧𑀸𑀧𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀢𑀺 𑀢𑁂𑀫𑀺𑀬𑁄”𑁇_  


(64.)

1378\. _“𑀬𑁄𑀚𑀬𑀦𑁆𑀢𑀼 𑀭𑀣𑁂 𑀅𑀲𑁆𑀲𑁂,_  
_𑀓𑀘𑁆𑀙𑀁 𑀦𑀸𑀕𑀸𑀦 𑀩𑀦𑁆𑀥𑀣;_  
_𑀉𑀤𑀻𑀭𑀬𑀦𑁆𑀢𑀼 𑀲𑀗𑁆𑀔𑀧𑀡𑀯𑀸,_  
_𑀯𑀸𑀤𑀦𑁆𑀢𑀼 𑀏𑀓𑀧𑁄𑀓𑁆𑀔𑀭𑀸𑁇_  


(65.)

1379\. _𑀯𑀸𑀤𑀦𑁆𑀢𑀼 𑀪𑁂𑀭𑀻 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀸,_  
_𑀯𑀕𑁆𑀕𑀽 𑀯𑀸𑀤𑀦𑁆𑀢𑀼 𑀤𑀼𑀦𑁆𑀤𑀼𑀪𑀻;_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀫𑀁 𑀅𑀦𑁆𑀯𑁂𑀦𑁆𑀢𑀼,_  
_𑀕𑀘𑁆𑀙𑀁 𑀧𑀼𑀢𑁆𑀢𑀦𑀺𑀯𑁂𑀤𑀓𑁄𑁇_  


(66.)

1380\. _𑀑𑀭𑁄𑀥𑀸 𑀘 𑀓𑀼𑀫𑀸𑀭𑀸 𑀘,_  
_𑀯𑁂𑀲𑀺𑀬𑀸𑀦𑀸 𑀘 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑁂𑀦𑁆𑀢𑀼,_  
_𑀕𑀘𑁆𑀙𑀁 𑀧𑀼𑀢𑁆𑀢𑀦𑀺𑀯𑁂𑀤𑀓𑁄𑁇_  


(67.)

1381\. _𑀳𑀢𑁆𑀣𑀸𑀭𑁄𑀳𑀸 𑀅𑀦𑀻𑀓𑀝𑁆𑀞𑀸,_  
_𑀭𑀣𑀺𑀓𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀭𑀓𑀸;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑁂𑀦𑁆𑀢𑀼,_  
_𑀕𑀘𑁆𑀙𑀁 𑀧𑀼𑀢𑁆𑀢𑀦𑀺𑀯𑁂𑀤𑀓𑁄𑁇_  


(68.)

1382\. _𑀲𑀫𑀸𑀕𑀢𑀸 𑀚𑀸𑀦𑀧𑀤𑀸,_  
_𑀦𑁂𑀕𑀫𑀸 𑀘 𑀲𑀫𑀸𑀕𑀢𑀸;_  
_𑀔𑀺𑀧𑁆𑀧𑀁 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑁂𑀦𑁆𑀢𑀼,_  
_𑀕𑀘𑁆𑀙𑀁 𑀧𑀼𑀢𑁆𑀢𑀦𑀺𑀯𑁂𑀤𑀓𑁄”𑁇_  


(69.)

1383\. _𑀅𑀲𑁆𑀲𑁂 𑀘 𑀲𑀸𑀭𑀣𑀻 𑀬𑀼𑀢𑁆𑀢𑁂,_  
_𑀲𑀺𑀦𑁆𑀥𑀯𑁂 𑀲𑀻𑀖𑀯𑀸𑀳𑀦𑁂;_  
_𑀭𑀸𑀚𑀤𑁆𑀯𑀸𑀭𑀁 𑀉𑀧𑀸𑀕𑀘𑁆𑀙𑀼𑀁,_  
_“𑀬𑀼𑀢𑁆𑀢𑀸 𑀤𑁂𑀯 𑀇𑀫𑁂 𑀳𑀬𑀸”𑁇_  


(70.)

1384\. _“𑀣𑀽𑀮𑀸 𑀚𑀯𑁂𑀦 𑀳𑀸𑀬𑀦𑁆𑀢𑀺,_  
_𑀓𑀺𑀲𑀸 𑀳𑀸𑀬𑀦𑁆𑀢𑀺 𑀣𑀸𑀫𑀼𑀦𑀸;_  
_𑀓𑀺𑀲𑁂 𑀣𑀽𑀮𑁂 𑀯𑀺𑀯𑀚𑁆𑀚𑁂𑀢𑁆𑀯𑀸,_  
_𑀲𑀁𑀲𑀝𑁆𑀞𑀸 𑀬𑁄𑀚𑀺𑀢𑀸 𑀳𑀬𑀸”𑁇_  


(71.)

1385\. _𑀢𑀢𑁄 𑀭𑀸𑀚𑀸 𑀢𑀭𑀫𑀸𑀦𑁄,_  
_𑀬𑀼𑀢𑁆𑀢𑀫𑀸𑀭𑀼𑀬𑁆𑀳 𑀲𑀦𑁆𑀤𑀦𑀁;_  
_𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺,_  
_“𑀲𑀩𑁆𑀩𑀸𑀯 𑀅𑀦𑀼𑀬𑀸𑀣 𑀫𑀁𑁇_  


(72.)

1386\. _𑀯𑀸𑀮𑀩𑀻𑀚𑀦𑀺𑀫𑀼𑀡𑁆𑀳𑀻𑀲𑀁,_  
_𑀔𑀕𑁆𑀕𑀁 𑀙𑀢𑁆𑀢𑀜𑁆𑀘 𑀧𑀡𑁆𑀟𑀭𑀁;_  
_𑀉𑀧𑀸𑀥𑀻 𑀭𑀣𑀫𑀸𑀭𑀼𑀬𑁆𑀳,_  
_𑀲𑀼𑀯𑀡𑁆𑀡𑁂𑀳𑀺 𑀅𑀮𑀗𑁆𑀓𑀢𑀸”𑁇_  


(73.)

1387\. _𑀢𑀢𑁄 𑀲 𑀭𑀸𑀚𑀸 𑀧𑀸𑀬𑀸𑀲𑀺,_  
_𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁆𑀯𑀸𑀦 𑀲𑀸𑀭𑀣𑀺𑀁;_  
_𑀔𑀺𑀧𑁆𑀧𑀫𑁂𑀯 𑀉𑀧𑀸𑀕𑀘𑁆𑀙𑀺,_  
_𑀬𑀢𑁆𑀣 𑀲𑀫𑁆𑀫𑀢𑀺 𑀢𑁂𑀫𑀺𑀬𑁄𑁇_  


(74.)

1388\. _𑀢𑀜𑁆𑀘 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀦𑁆𑀢𑀁,_  
_𑀚𑀮𑀦𑁆𑀢𑀫𑀺𑀯 𑀢𑁂𑀚𑀲𑀸;_  
_𑀔𑀢𑁆𑀢𑀲𑀗𑁆𑀖𑀧𑀭𑀺𑀩𑁆𑀬𑀽𑀍𑀳𑀁,_  
_𑀢𑁂𑀫𑀺𑀬𑁄 𑀏𑀢𑀤𑀩𑁆𑀭𑀯𑀺𑁇_  


(75.)

1389\. _“𑀓𑀘𑁆𑀘𑀺 𑀦𑀼 𑀢𑀸𑀢 𑀓𑀼𑀲𑀮𑀁,_  
_𑀓𑀘𑁆𑀘𑀺 𑀢𑀸𑀢 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀲𑀩𑁆𑀩𑀸 𑀘 𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀬𑁄,_  
_𑀅𑀭𑁄𑀕𑀸 𑀫𑀬𑁆𑀳 𑀫𑀸𑀢𑀭𑁄”𑁇_  


(76.)

1390\. _“𑀓𑀼𑀲𑀮𑀜𑁆𑀘𑁂𑀯 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢,_  
_𑀅𑀣𑁄 𑀧𑀼𑀢𑁆𑀢 𑀅𑀦𑀸𑀫𑀬𑀁;_  
_𑀲𑀩𑁆𑀩𑀸 𑀘 𑀭𑀸𑀚𑀓𑀜𑁆𑀜𑀸𑀬𑁄,_  
_𑀅𑀭𑁄𑀕𑀸 𑀢𑀼𑀬𑁆𑀳 𑀫𑀸𑀢𑀭𑁄”𑁇_  


(77.)

1391\. _“𑀓𑀘𑁆𑀘𑀺 𑀅𑀫𑀚𑁆𑀚𑀧𑁄 𑀢𑀸𑀢,_  
_𑀓𑀘𑁆𑀘𑀺 𑀢𑁂 𑀲𑀼𑀭𑀫𑀧𑁆𑀧𑀺𑀬𑀁;_  
_𑀓𑀘𑁆𑀘𑀺 𑀲𑀘𑁆𑀘𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘,_  
_𑀤𑀸𑀦𑁂 𑀢𑁂 𑀭𑀫𑀢𑁂 𑀫𑀦𑁄”𑁇_  


(78.)

1392\. _“𑀅𑀫𑀚𑁆𑀚𑀧𑁄 𑀅𑀳𑀁 𑀧𑀼𑀢𑁆𑀢,_  
_𑀅𑀣𑁄 𑀫𑁂 𑀲𑀼𑀭𑀫𑀧𑁆𑀧𑀺𑀬𑀁;_  
_𑀅𑀣𑁄 𑀲𑀘𑁆𑀘𑁂 𑀘 𑀥𑀫𑁆𑀫𑁂 𑀘,_  
_𑀤𑀸𑀦𑁂 𑀫𑁂 𑀭𑀫𑀢𑁂 𑀫𑀦𑁄”𑁇_  


(79.)

1393\. _“𑀓𑀘𑁆𑀘𑀺 𑀅𑀭𑁄𑀕𑀁 𑀬𑁄𑀕𑁆𑀕𑀁 𑀢𑁂,_  
_𑀓𑀘𑁆𑀘𑀺 𑀯𑀳𑀢𑀺 𑀯𑀸𑀳𑀦𑀁;_  
_𑀓𑀘𑁆𑀘𑀺 𑀢𑁂 𑀩𑁆𑀬𑀸𑀥𑀬𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀭𑀻𑀭𑀲𑁆𑀲𑀼𑀧𑀢𑀸𑀧𑀦𑀸”𑁇_  


(80.)

1394\. _“𑀅𑀣𑁄 𑀅𑀭𑁄𑀕𑀁 𑀬𑁄𑀕𑁆𑀕𑀁 𑀫𑁂,_  
_𑀅𑀣𑁄 𑀯𑀳𑀢𑀺 𑀯𑀸𑀳𑀦𑀁;_  
_𑀅𑀣𑁄 𑀫𑁂 𑀩𑁆𑀬𑀸𑀥𑀬𑁄 𑀦𑀢𑁆𑀣𑀺,_  
_𑀲𑀭𑀻𑀭𑀲𑁆𑀲𑀼𑀧𑀢𑀸𑀧𑀦𑀸”𑁇_  


(81.)

1395\. _“𑀓𑀘𑁆𑀘𑀺 𑀅𑀦𑁆𑀢𑀸 𑀘 𑀢𑁂 𑀨𑀻𑀢𑀸,_  
_𑀫𑀚𑁆𑀛𑁂 𑀘 𑀩𑀳𑀮𑀸 𑀢𑀯;_  
_𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀓𑁄𑀲𑀜𑁆𑀘,_  
_𑀓𑀘𑁆𑀘𑀺 𑀢𑁂 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀢𑀁”𑁇_  


(82.)

1396\. _“𑀅𑀣𑁄 𑀅𑀦𑁆𑀢𑀸 𑀘 𑀫𑁂 𑀨𑀻𑀢𑀸,_  
_𑀫𑀚𑁆𑀛𑁂 𑀘 𑀩𑀳𑀮𑀸 𑀫𑀫;_  
_𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀓𑁄𑀲𑀜𑁆𑀘,_  
_𑀲𑀩𑁆𑀩𑀁 𑀫𑁂 𑀧𑀝𑀺𑀲𑀦𑁆𑀣𑀢𑀁”𑁇_  


(83.)

1397\. _“𑀲𑁆𑀯𑀸𑀕𑀢𑀁 𑀢𑁂 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀅𑀣𑁄 𑀢𑁂 𑀅𑀤𑀼𑀭𑀸𑀕𑀢𑀁;_  
_𑀧𑀢𑀺𑀝𑁆𑀞𑀧𑁂𑀦𑁆𑀢𑀼 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁,_  
_𑀬𑀢𑁆𑀣 𑀭𑀸𑀚𑀸 𑀦𑀺𑀲𑀓𑁆𑀓𑀢𑀺𑁇_  


(84.)

1398\. _𑀇𑀥𑁂𑀯 𑀢𑁂 𑀦𑀺𑀲𑀻𑀤𑀲𑁆𑀲𑀼,_  
_𑀦𑀺𑀬𑀢𑁂 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂;_  
_𑀏𑀢𑁆𑀢𑁄 𑀉𑀤𑀓𑀫𑀸𑀤𑀸𑀬,_  
_𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑀬𑀲𑁆𑀲𑀼 𑀢𑁂𑁇_  


(85.)

1399\. _𑀇𑀤𑀫𑁆𑀧𑀺 𑀧𑀡𑁆𑀡𑀓𑀁 𑀫𑀬𑁆𑀳𑀁,_  
_𑀭𑀦𑁆𑀥𑀁 𑀭𑀸𑀚 𑀅𑀮𑁄𑀡𑀓𑀁;_  
_𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚 𑀫𑀳𑀸𑀭𑀸𑀚,_  
_𑀧𑀸𑀳𑀼𑀦𑁄 𑀫𑁂𑀲𑀺𑀥𑀸𑀕𑀢𑁄”𑁇_  


(86.)

1400\. _“𑀦 𑀘𑀸𑀳𑀁 𑀧𑀡𑁆𑀡𑀁 𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀺,_  
_𑀦 𑀳𑁂𑀢𑀁 𑀫𑀬𑁆𑀳 𑀪𑁄𑀚𑀦𑀁;_  
_𑀲𑀸𑀮𑀻𑀦𑀁 𑀑𑀤𑀦𑀁 𑀪𑀼𑀜𑁆𑀚𑁂,_  
_𑀲𑀼𑀘𑀺𑀁 𑀫𑀁𑀲𑀽𑀧𑀲𑁂𑀘𑀦𑀁”𑁇_  


(87.)

1401\. _“𑀅𑀘𑁆𑀙𑁂𑀭𑀓𑀁 𑀫𑀁 𑀧𑀝𑀺𑀪𑀸𑀢𑀺,_  
_𑀏𑀓𑀓𑀫𑁆𑀧𑀺 𑀭𑀳𑁄𑀕𑀢𑀁;_  
_𑀏𑀤𑀺𑀲𑀁 𑀪𑀼𑀜𑁆𑀚𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀓𑁂𑀦 𑀯𑀡𑁆𑀡𑁄 𑀧𑀲𑀻𑀤𑀢𑀺”𑁇_  


(88.)

1402\. _“𑀏𑀓𑁄 𑀭𑀸𑀚 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀺,_  
_𑀦𑀺𑀬𑀢𑁂 𑀧𑀡𑁆𑀡𑀲𑀦𑁆𑀣𑀭𑁂;_  
_𑀢𑀸𑀬 𑀫𑁂 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀸𑀬,_  
_𑀭𑀸𑀚 𑀯𑀡𑁆𑀡𑁄 𑀧𑀲𑀻𑀤𑀢𑀺𑁇_  


(89.)

1403\. _𑀦 𑀘 𑀦𑁂𑀢𑁆𑀢𑀺𑀁𑀲𑀩𑀦𑁆𑀥𑀸 𑀫𑁂,_  
_𑀭𑀸𑀚𑀭𑀓𑁆𑀔𑀸 𑀉𑀧𑀝𑁆𑀞𑀺𑀢𑀸;_  
_𑀢𑀸𑀬 𑀫𑁂 𑀲𑀼𑀔𑀲𑁂𑀬𑁆𑀬𑀸𑀬,_  
_𑀭𑀸𑀚 𑀯𑀡𑁆𑀡𑁄 𑀧𑀲𑀻𑀤𑀢𑀺𑁇_  


(90.)

1404\. _𑀅𑀢𑀻𑀢𑀁 𑀦𑀸𑀦𑀼𑀲𑁄𑀘𑀸𑀫𑀺,_  
_𑀦𑀧𑁆𑀧𑀚𑀧𑁆𑀧𑀸𑀫𑀺𑀦𑀸𑀕𑀢𑀁;_  
_𑀧𑀘𑁆𑀘𑀼𑀧𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀬𑀸𑀧𑁂𑀫𑀺,_  
_𑀢𑁂𑀦 𑀯𑀡𑁆𑀡𑁄 𑀧𑀲𑀻𑀤𑀢𑀺𑁇_  


(91.)

1405\. _𑀅𑀦𑀸𑀕𑀢𑀧𑁆𑀧𑀚𑀧𑁆𑀧𑀸𑀬,_  
_𑀅𑀢𑀻𑀢𑀲𑁆𑀲𑀸𑀦𑀼𑀲𑁄𑀘𑀦𑀸;_  
_𑀏𑀢𑁂𑀦 𑀩𑀸𑀮𑀸 𑀲𑀼𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀦𑀍𑀑𑀯 𑀳𑀭𑀺𑀢𑁄 𑀮𑀼𑀢𑁄”𑁇_  


(92.)

1406\. _“𑀳𑀢𑁆𑀣𑀸𑀦𑀻𑀓𑀁 𑀭𑀣𑀸𑀦𑀻𑀓𑀁,_  
_𑀅𑀲𑁆𑀲𑁂 𑀧𑀢𑁆𑀢𑀻 𑀘 𑀯𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀭𑀫𑁆𑀫𑀸𑀦𑀺,_  
_𑀅𑀳𑀁 𑀧𑀼𑀢𑁆𑀢 𑀤𑀤𑀸𑀫𑀺 𑀢𑁂𑁇_  


(93.)

1407\. _𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀫𑁆𑀧𑀺 𑀢𑁂 𑀤𑀫𑁆𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀁;_  
_𑀢𑀸 𑀧𑀼𑀢𑁆𑀢 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀲𑁆𑀲𑀼,_  
_𑀢𑁆𑀯𑀁 𑀦𑁄 𑀭𑀸𑀚𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(94.)

1408\. _𑀓𑀼𑀲𑀮𑀸 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑀲𑁆𑀲,_  
_𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀘𑀸𑀢𑀼𑀭𑀺𑀢𑁆𑀣𑀺𑀬𑁄;_  
_𑀓𑀸𑀫𑁂 𑀢𑀁 𑀭𑀫𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀓𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(95.)

1409\. _𑀧𑀝𑀺𑀭𑀸𑀚𑀽𑀳𑀺 𑀢𑁂 𑀓𑀜𑁆𑀜𑀸,_  
_𑀆𑀦𑀬𑀺𑀲𑁆𑀲𑀁 𑀅𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀢𑀸𑀲𑀼 𑀧𑀼𑀢𑁆𑀢𑁂 𑀚𑀦𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀣 𑀧𑀘𑁆𑀙𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(96.)

1410\. _𑀬𑀼𑀯𑀸 𑀘 𑀤𑀳𑀭𑁄 𑀘𑀸𑀲𑀺,_  
_𑀧𑀞𑀫𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀲𑀼𑀲𑀼;_  
_𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀓𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(97.)

1411\. _“𑀬𑀼𑀯𑀸 𑀘𑀭𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀬𑀼𑀯𑀸 𑀲𑀺𑀬𑀸;_  
_𑀤𑀳𑀭𑀲𑁆𑀲 𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸,_  
_𑀏𑀢𑀁 𑀇𑀲𑀻𑀳𑀺 𑀯𑀡𑁆𑀡𑀺𑀢𑀁𑁇_  


(98.)

1412\. _𑀬𑀼𑀯𑀸 𑀘𑀭𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁,_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀬𑀼𑀯𑀸 𑀲𑀺𑀬𑀸;_  
_𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺,_  
_𑀦𑀸𑀳𑀁 𑀭𑀚𑁆𑀚𑁂𑀦 𑀫𑀢𑁆𑀣𑀺𑀓𑁄𑁇_  


(99.)

1413\. _𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀤𑀳𑀭𑀁,_  
_𑀅𑀫𑁆𑀫 𑀢𑀸𑀢 𑀯𑀤𑀦𑁆𑀢𑀭𑀁;_  
_𑀓𑀺𑀘𑁆𑀙𑀸𑀮𑀤𑁆𑀥𑀁 𑀧𑀺𑀬𑀁 𑀧𑀼𑀢𑁆𑀢𑀁,_  
_𑀅𑀧𑁆𑀧𑀢𑁆𑀯𑀸𑀯 𑀚𑀭𑀁 𑀫𑀢𑀁𑁇_  


(100.)

1414\. _𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑁄𑀳𑀁 𑀤𑀳𑀭𑀺𑀁,_  
_𑀓𑀼𑀫𑀸𑀭𑀺𑀁 𑀘𑀸𑀭𑀼𑀤𑀲𑁆𑀲𑀦𑀺𑀁;_  
_𑀦𑀯𑀯𑀁𑀲𑀓𑀍𑀈𑀭𑀁𑀯,_  
_𑀧𑀮𑀼𑀕𑁆𑀕𑀁 𑀚𑀻𑀯𑀺𑀢𑀓𑁆𑀔𑀬𑀁𑁇_  


(101.)

1415\. _𑀤𑀳𑀭𑀸𑀧𑀺 𑀳𑀺 𑀫𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺,_  
_𑀦𑀭𑀸 𑀘 𑀅𑀣 𑀦𑀸𑀭𑀺𑀬𑁄;_  
_𑀢𑀢𑁆𑀣 𑀓𑁄 𑀯𑀺𑀲𑁆𑀲𑀲𑁂 𑀧𑁄𑀲𑁄,_  
_𑀤𑀳𑀭𑁄𑀫𑁆𑀳𑀻𑀢𑀺 𑀚𑀻𑀯𑀺𑀢𑁂𑁇_  


(102.)

1416\. _𑀬𑀲𑁆𑀲 𑀭𑀢𑁆𑀬𑀸 𑀯𑀺𑀯𑀲𑀸𑀦𑁂,_  
_𑀆𑀬𑀼 𑀅𑀧𑁆𑀧𑀢𑀭𑀁 𑀲𑀺𑀬𑀸;_  
_𑀅𑀧𑁆𑀧𑁄𑀤𑀓𑁂𑀯 𑀫𑀘𑁆𑀙𑀸𑀦𑀁,_  
_𑀓𑀺𑀁 𑀦𑀼 𑀓𑁄𑀫𑀸𑀭𑀓𑀁 𑀢𑀳𑀺𑀁𑁇_  


(103.)

1417\. _𑀦𑀺𑀘𑁆𑀘𑀫𑀩𑁆𑀪𑀸𑀳𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀦𑀺𑀘𑁆𑀘𑀜𑁆𑀘 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄;_  
_𑀅𑀫𑁄𑀖𑀸𑀲𑀼 𑀯𑀚𑀦𑁆𑀢𑀻𑀲𑀼,_  
_𑀓𑀺𑀁 𑀫𑀁 𑀭𑀚𑁆𑀚𑁂𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑀲𑀺”𑁇_  


(104.)

1418\. _“𑀓𑁂𑀦 𑀫𑀩𑁆𑀪𑀸𑀳𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀓𑁂𑀦 𑀘 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄;_  
_𑀓𑀸𑀬𑁄 𑀅𑀫𑁄𑀖𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀢𑀁 𑀫𑁂 𑀅𑀓𑁆𑀔𑀸𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄”𑁇_  


(105.)

1419\. _“𑀫𑀘𑁆𑀘𑀼𑀦𑀸𑀩𑁆𑀪𑀸𑀳𑀢𑁄 𑀮𑁄𑀓𑁄,_  
_𑀚𑀭𑀸𑀬 𑀧𑀭𑀺𑀯𑀸𑀭𑀺𑀢𑁄;_  
_𑀭𑀢𑁆𑀬𑁄 𑀅𑀫𑁄𑀖𑀸 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺,_  
_𑀏𑀯𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀔𑀢𑁆𑀢𑀺𑀬𑁇_  


(106.)

1420\. _𑀬𑀣𑀸𑀧𑀺 𑀢𑀦𑁆𑀢𑁂 𑀯𑀺𑀢𑀢𑁂,_  
_𑀬𑀁 𑀬𑀤𑁂𑀯𑀽𑀧𑀯𑀺𑀬𑁆𑀬𑀢𑀺;_  
_𑀅𑀧𑁆𑀧𑀓𑀁 𑀳𑁄𑀢𑀺 𑀯𑁂𑀢𑀩𑁆𑀩𑀁,_  
_𑀏𑀯𑀁 𑀫𑀘𑁆𑀘𑀸𑀦 𑀚𑀻𑀯𑀺𑀢𑀁𑁇_  


(107.)

1421\. _𑀬𑀣𑀸 𑀯𑀸𑀭𑀺𑀯𑀳𑁄 𑀧𑀽𑀭𑁄,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀦𑀼𑀧𑀦𑀺𑀯𑀢𑁆𑀢𑀢𑀺;_  
_𑀏𑀯𑀫𑀸𑀬𑀼 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁,_  
_𑀕𑀘𑁆𑀙𑀦𑁆𑀦𑀼𑀧𑀦𑀺𑀯𑀢𑁆𑀢𑀢𑀺𑁇_  


(108.)

1422\. _𑀬𑀣𑀸 𑀯𑀸𑀭𑀺𑀯𑀳𑁄 𑀧𑀽𑀭𑁄,_  
_𑀯𑀳𑁂 𑀭𑀼𑀓𑁆𑀔𑁂𑀧𑀓𑀽𑀮𑀚𑁂;_  
_𑀏𑀯𑀁 𑀚𑀭𑀸𑀫𑀭𑀡𑁂𑀦,_  
_𑀯𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑁂 𑀲𑀩𑁆𑀩𑀧𑀸𑀡𑀺𑀦𑁄”𑁇_  


(109.)

1423\. _“𑀳𑀢𑁆𑀣𑀸𑀦𑀻𑀓𑀁 𑀭𑀣𑀸𑀦𑀻𑀓𑀁,_  
_𑀅𑀲𑁆𑀲𑁂 𑀧𑀢𑁆𑀢𑀻 𑀘 𑀯𑀫𑁆𑀫𑀺𑀦𑁄;_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀭𑀫𑁆𑀫𑀸𑀦𑀺,_  
_𑀅𑀳𑀁 𑀧𑀼𑀢𑁆𑀢 𑀤𑀤𑀸𑀫𑀺 𑀢𑁂𑁇_  


(110.)

1424\. _𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀫𑁆𑀧𑀺 𑀢𑁂 𑀤𑀫𑁆𑀫𑀺,_  
_𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀪𑀽𑀲𑀺𑀢𑀁;_  
_𑀢𑀸 𑀧𑀼𑀢𑁆𑀢 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀲𑁆𑀲𑀼,_  
_𑀢𑁆𑀯𑀁 𑀦𑁄 𑀭𑀸𑀚𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(111.)

1425\. _𑀓𑀼𑀲𑀮𑀸 𑀦𑀘𑁆𑀘𑀕𑀻𑀢𑀲𑁆𑀲,_  
_𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀸 𑀘𑀸𑀢𑀼𑀭𑀺𑀢𑁆𑀣𑀺𑀬𑁄;_  
_𑀓𑀸𑀫𑁂 𑀢𑀁 𑀭𑀫𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀓𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(112.)

1426\. _𑀧𑀝𑀺𑀭𑀸𑀚𑀽𑀳𑀺 𑀢𑁂 𑀓𑀜𑁆𑀜𑀸,_  
_𑀆𑀦𑀬𑀺𑀲𑁆𑀲𑀁 𑀅𑀮𑀗𑁆𑀓𑀢𑀸;_  
_𑀢𑀸𑀲𑀼 𑀧𑀼𑀢𑁆𑀢𑁂 𑀚𑀦𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀅𑀣 𑀧𑀘𑁆𑀙𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(113.)

1427\. _𑀬𑀼𑀯𑀸 𑀘 𑀤𑀳𑀭𑁄 𑀘𑀸𑀲𑀺,_  
_𑀧𑀞𑀫𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀲𑀼𑀲𑀼;_  
_𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀓𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺𑁇_  


(114.)

1428\. _𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀓𑁄𑀲𑀜𑁆𑀘,_  
_𑀯𑀸𑀳𑀦𑀸𑀦𑀺 𑀩𑀮𑀸𑀦𑀺 𑀘;_  
_𑀦𑀺𑀯𑁂𑀲𑀦𑀸𑀦𑀺 𑀭𑀫𑁆𑀫𑀸𑀦𑀺,_  
_𑀅𑀳𑀁 𑀧𑀼𑀢𑁆𑀢 𑀤𑀤𑀸𑀫𑀺 𑀢𑁂𑁇_  


(115.)

1429\. _𑀕𑁄𑀫𑀡𑁆𑀟𑀮𑀧𑀭𑀺𑀩𑁆𑀬𑀽𑀍𑀳𑁄,_  
_𑀤𑀸𑀲𑀺𑀲𑀗𑁆𑀖𑀧𑀼𑀭𑀓𑁆𑀔𑀢𑁄;_  
_𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑀦𑁆𑀢𑁂,_  
_𑀓𑀺𑀁 𑀅𑀭𑀜𑁆𑀜𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀲𑀺”𑁇_  


(116.)

1430\. _“𑀓𑀺𑀁 𑀥𑀦𑁂𑀦 𑀬𑀁 𑀔𑀻𑀬𑁂𑀣,_  
_𑀓𑀺𑀁 𑀪𑀭𑀺𑀬𑀸𑀬 𑀫𑀭𑀺𑀲𑁆𑀲𑀢𑀺;_  
_𑀓𑀺𑀁 𑀬𑁄𑀩𑁆𑀩𑀦𑁂𑀦 𑀚𑀺𑀡𑁆𑀡𑁂𑀦,_  
_𑀬𑀁 𑀚𑀭𑀸𑀬𑀸𑀪𑀺𑀪𑀼𑀬𑁆𑀬𑀢𑀺𑁇_  


(117.)

1431\. _𑀢𑀢𑁆𑀣 𑀓𑀸 𑀦𑀦𑁆𑀤𑀺 𑀓𑀸 𑀔𑀺𑀟𑁆𑀟𑀸,_  
_𑀓𑀸 𑀭𑀢𑀺 𑀓𑀸 𑀥𑀦𑁂𑀲𑀦𑀸;_  
_𑀓𑀺𑀁 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀤𑀸𑀭𑁂𑀳𑀺,_  
_𑀭𑀸𑀚 𑀫𑀼𑀢𑁆𑀢𑁄𑀲𑁆𑀫𑀺 𑀩𑀦𑁆𑀥𑀦𑀸𑁇_  


(118.)

1432\. _𑀬𑁄𑀳𑀁 𑀏𑀯𑀁 𑀧𑀚𑀸𑀦𑀸𑀫𑀺,_  
_𑀫𑀘𑁆𑀘𑀼 𑀫𑁂 𑀦𑀧𑁆𑀧𑀫𑀚𑁆𑀚𑀢𑀺;_  
_𑀅𑀦𑁆𑀢𑀓𑁂𑀦𑀸𑀥𑀺𑀧𑀦𑁆𑀦𑀲𑁆𑀲,_  
_𑀓𑀸 𑀭𑀢𑀻 𑀓𑀸 𑀥𑀦𑁂𑀲𑀦𑀸𑁇_  


(119.)

1433\. _𑀨𑀮𑀸𑀦𑀫𑀺𑀯 𑀧𑀓𑁆𑀓𑀸𑀦𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀧𑀢𑀦𑀢𑁄 𑀪𑀬𑀁;_  
_𑀏𑀯𑀁 𑀚𑀸𑀢𑀸𑀦 𑀫𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀦𑀺𑀘𑁆𑀘𑀁 𑀫𑀭𑀡𑀢𑁄 𑀪𑀬𑀁𑁇_  


(120.)

1434\. _𑀲𑀸𑀬𑀫𑁂𑀓𑁂 𑀦 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀧𑀸𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑀸 𑀩𑀳𑀽 𑀚𑀦𑀸;_  
_𑀧𑀸𑀢𑁄 𑀏𑀓𑁂 𑀦 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺,_  
_𑀲𑀸𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀸 𑀩𑀳𑀽 𑀚𑀦𑀸𑁇_  


(121.)

1435\. _𑀅𑀚𑁆𑀚𑁂𑀯 𑀓𑀺𑀘𑁆𑀘𑀁 𑀆𑀢𑀧𑀁,_  
_𑀓𑁄 𑀚𑀜𑁆𑀜𑀸 𑀫𑀭𑀡𑀁 𑀲𑀼𑀯𑁂;_  
_𑀦 𑀳𑀺 𑀦𑁄 𑀲𑀗𑁆𑀕𑀭𑀁 𑀢𑁂𑀦,_  
_𑀫𑀳𑀸𑀲𑁂𑀦𑁂𑀦 𑀫𑀘𑁆𑀘𑀼𑀦𑀸𑁇_  


(122.)

1436\. _𑀘𑁄𑀭𑀸 𑀥𑀦𑀲𑁆𑀲 𑀧𑀢𑁆𑀣𑁂𑀦𑁆𑀢𑀺,_  
_𑀭𑀸𑀚 𑀫𑀼𑀢𑁆𑀢𑁄𑀲𑁆𑀫𑀺 𑀩𑀦𑁆𑀥𑀦𑀸;_  
_𑀏𑀳𑀺 𑀭𑀸𑀚 𑀦𑀺𑀯𑀢𑁆𑀢𑀲𑁆𑀲𑀼,_  
_𑀦𑀸𑀳𑀁 𑀭𑀚𑁆𑀚𑁂𑀦 𑀫𑀢𑁆𑀣𑀺𑀓𑁄”𑀢𑀺𑁇_  


---

1437\. 𑀫𑀽𑀕𑀧𑀓𑁆𑀔𑀚𑀸𑀢𑀓𑀁 𑀧𑀞𑀫𑀁𑁇
