# 𑁯𑁇𑁬 𑀤𑁆𑀯𑁂𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀸𑀤𑀺𑀓𑀣𑀸

* 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀤𑁆𑀯𑁂𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀸𑀤𑀺𑀓𑀣𑀸

(395.)

1809\. 𑀤𑁆𑀯𑁂𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀸𑁇 𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁𑁇 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀲𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀤𑀼𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑀼𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄? 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄𑁇 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑀼𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑀼𑀦𑁆𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄𑁇

1810\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺—  𑀲𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄? 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄 𑀳𑁄𑀢𑀺 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑁄 𑀅𑀦𑀧𑀤𑀸𑀦𑁄, 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀳𑀺 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀳𑀺, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺—  𑀲𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑀺—  𑀲𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄𑁇

(396.)

1811\. 𑀤𑁆𑀯𑁂𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀲𑀸𑀭𑀡𑀸𑁇 𑀅𑀢𑁆𑀣𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀲𑁄𑀲𑀸𑀭𑀺𑀢𑁄, 𑀏𑀓𑀘𑁆𑀘𑁄 𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑁄? 𑀧𑀡𑁆𑀟𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄, 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑁂 𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀤𑁄𑀲𑀸𑀭𑀺𑀢𑀸𑁇

1812\. 𑀓𑀢𑀫𑁄 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀲𑁄𑀲𑀸𑀭𑀺𑀢𑁄? 𑀳𑀢𑁆𑀣𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺, 𑀲𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀧𑀸𑀤𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀓𑀡𑁆𑀡𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀦𑀸𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀍𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀓𑀡𑁆𑀟𑀭𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀨𑀡𑀳𑀢𑁆𑀣𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀔𑀼𑀚𑁆𑀚𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀯𑀸𑀫𑀦𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀕𑀮𑀕𑀡𑁆𑀟𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀮𑀓𑁆𑀔𑀡𑀸𑀳𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀓𑀲𑀸𑀳𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀮𑀺𑀔𑀺𑀢𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀻𑀧𑀤𑀺𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀸𑀧𑀭𑁄𑀕𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀭𑀺𑀲𑀤𑀽𑀲𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀓𑀸𑀡𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀓𑀼𑀡𑀻, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀔𑀜𑁆𑀚𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀓𑁆𑀔𑀳𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀙𑀺𑀦𑁆𑀦𑀺𑀭𑀺𑀬𑀸𑀧𑀣𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀚𑀭𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀦𑁆𑀥𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀫𑀽𑀕𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀩𑀥𑀺𑀭𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀦𑁆𑀥𑀫𑀽𑀕𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀦𑁆𑀥𑀩𑀥𑀺𑀭𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀫𑀽𑀕𑀩𑀥𑀺𑀭𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀦𑁆𑀥𑀫𑀽𑀕𑀩𑀥𑀺𑀭𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀲𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀲𑁄𑀲𑀸𑀭𑀺𑀢𑁄𑁇 𑀇𑀫𑁂 𑀯𑀼𑀘𑁆𑀘𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀑𑀲𑀸𑀭𑀡𑀁, 𑀢𑁂 𑀘𑁂 𑀲𑀁𑀖𑁄 𑀑𑀲𑀸𑀭𑁂𑀢𑀺—  𑀲𑁄𑀲𑀸𑀭𑀺𑀢𑀸𑁇
