# 𑁯𑁇𑁧𑁬 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀓𑀣𑀸

* 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀓𑀣𑀸

(418.)

1858\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇

(419.)

1859\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇

1860\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇

1861\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇

1862\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀇𑀫𑀲𑁆𑀲 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇
