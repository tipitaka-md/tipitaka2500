# 𑁯𑁇𑁯 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀣𑀸

* 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀣𑀸

(407.)

1841\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇

(408.)

1842\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇

(409.)

1843\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇

1844\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇

1845\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀳𑁄𑀢𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁄 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁄 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁄𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸𑀧𑀺 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇
