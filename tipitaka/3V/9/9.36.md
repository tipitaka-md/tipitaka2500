# 𑁯𑁇𑁩𑁬 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀓𑀣𑀸

(450.)

1896\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…𑀧𑁂…  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑁄 𑀲𑀁𑀖𑁄 𑀯𑀺𑀯𑀤𑀢𑀺—  ‘𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀅𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀼𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺, 𑀬𑁂 𑀘 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀼𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺, 𑀇𑀫𑁂 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀦𑁄𑁇 𑀇𑀫𑁂𑀧𑀺 𑀧𑀜𑁆𑀘 𑀯𑀸𑀭𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸”𑁇

---

1897\. 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀦𑀯𑀫𑁄𑁇



---

1898\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1899\. _𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀪𑀕𑀯𑀸 𑀆𑀲𑀺,_  
_𑀯𑀢𑁆𑀣𑀼 𑀯𑀸𑀲𑀪𑀕𑀸𑀫𑀓𑁂;_  
_𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀫𑀼𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁,_  
_𑀅𑀓𑀸𑀲𑀺 𑀇𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀓𑁂𑁇_  


1900\. _𑀧𑀓𑀢𑀜𑁆𑀜𑀼𑀦𑁄𑀢𑀺 𑀜𑀢𑁆𑀯𑀸,_  
_𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀦 𑀓𑀭𑀻 𑀢𑀤𑀸;_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀦 𑀓𑀭𑁄𑀢𑀻𑀢𑀺,_  
_𑀲𑀸𑀕𑀫𑀸 𑀚𑀺𑀦𑀲𑀦𑁆𑀢𑀺𑀓𑁂𑁇_  


1901\. _𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀫𑀕𑁆𑀕𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀘;_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀜𑁆𑀘,_  
_𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀺𑀓𑀁𑁇_  


1902\. _𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀁,_  
_𑀏𑀓𑁄 𑀉𑀓𑁆𑀔𑀺𑀧𑀢𑁂𑀓𑀓𑀁;_  
_𑀏𑀓𑁄 𑀘 𑀤𑁆𑀯𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂,_  
_𑀲𑀁𑀖𑀁 𑀉𑀓𑁆𑀔𑀺𑀧𑀢𑁂𑀓𑀓𑁄𑁇_  


1903\. _𑀤𑀼𑀯𑁂𑀧𑀺 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀧𑀺,_  
_𑀲𑀁𑀖𑁄 𑀲𑀁𑀖𑀜𑁆𑀘 𑀉𑀓𑁆𑀔𑀺𑀧𑀺;_  
_𑀲𑀩𑁆𑀩𑀜𑁆𑀜𑀼𑀧𑀯𑀭𑁄 𑀲𑀼𑀢𑁆𑀯𑀸,_  
_𑀅𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑁇_  


1904\. _𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀁 𑀬𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀲𑀸𑀯𑀦𑀁;_  
_𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀯𑀺𑀧𑀦𑁆𑀦𑀁,_  
_𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀜𑀢𑁆𑀢𑀺𑀬𑀸 𑀘 𑀬𑀁𑁇_  


1905\. _𑀉𑀪𑀬𑁂𑀦 𑀯𑀺𑀧𑀦𑁆𑀦𑀜𑁆𑀘,_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀥𑀫𑁆𑀫𑀫𑁂𑀯 𑀘;_  
_𑀯𑀺𑀦𑀬𑀸 𑀲𑀢𑁆𑀣𑀼 𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑀁,_  
_𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀸𑀭𑀳𑀺𑀓𑀁𑁇_  


1906\. _𑀅𑀥𑀫𑁆𑀫𑀯𑀕𑁆𑀕𑀁 𑀲𑀫𑀕𑁆𑀕𑀁,_  
_𑀥𑀫𑁆𑀫 𑀧𑀢𑀺𑀭𑀽𑀧𑀸𑀦𑀺 𑀬𑁂 𑀤𑀼𑀯𑁂;_  
_𑀥𑀫𑁆𑀫𑁂𑀦𑁂𑀯 𑀘 𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁,_  
_𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀲𑀺 𑀢𑀣𑀸𑀕𑀢𑁄𑁇_  


1907\. _𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑁄 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑁄,_  
_𑀤𑀲𑀯𑀕𑁆𑀕𑁄 𑀘 𑀯𑀻𑀲𑀢𑀺;_  
_𑀧𑀭𑁄𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑁄 𑀘,_  
_𑀲𑀁𑀖𑁄 𑀧𑀜𑁆𑀘𑀯𑀺𑀥𑁄 𑀢𑀣𑀸𑁇_  


1908\. _𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁,_  
_𑀬𑀜𑁆𑀘 𑀓𑀫𑁆𑀫𑀁 𑀧𑀯𑀸𑀭𑀡𑀁;_  
_𑀅𑀩𑁆𑀪𑀸𑀦𑀓𑀫𑁆𑀫𑁂𑀦 𑀲𑀳,_  
_𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑁂𑀳𑀺 𑀓𑀫𑁆𑀫𑀺𑀓𑁄𑁇_  


1909\. _𑀤𑀼𑀯𑁂 𑀓𑀫𑁆𑀫𑁂 𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀫𑀚𑁆𑀛𑀤𑁂𑀲𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀁;_  
_𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀺𑀓𑁄,_  
_𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀺𑀓𑁄𑁇_  


1910\. _𑀅𑀩𑁆𑀪𑀸𑀦𑁂𑀓𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀲𑀯𑀕𑁆𑀕𑀺𑀓𑀸;_  
_𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑀓𑀭𑁄 𑀲𑀁𑀖𑁄,_  
_𑀯𑀻𑀲𑁄 𑀲𑀩𑁆𑀩𑀢𑁆𑀣 𑀓𑀫𑁆𑀫𑀺𑀓𑁄𑁇_  


1911\. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀘,_  
_𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀻;_  
_𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀽,_  
_𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀤𑀲𑁆𑀲𑀦𑁂𑁇_  


1912\. _𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸,_  
_𑀧𑀡𑁆𑀟𑀓𑁄 𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀁;_  
_𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀁,_  
_𑀫𑀸𑀢𑀼 𑀧𑀺𑀢𑀼 𑀘 𑀖𑀸𑀢𑀓𑀁𑁇_  


1913\. _𑀅𑀭𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻𑀤𑀽𑀲𑀺,_  
_𑀪𑁂𑀤𑀓𑀁 𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀁;_  
_𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀁,_  
_𑀦𑀸𑀦𑀸𑀲𑀻𑀫𑀸𑀬 𑀇𑀤𑁆𑀥𑀺𑀬𑀸𑁇_  


1914\. _𑀬𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀓𑀭𑁂 𑀓𑀫𑁆𑀫𑀁,_  
_𑀳𑁄𑀦𑁆𑀢𑁂𑀢𑁂 𑀘𑀢𑀼𑀯𑀻𑀲𑀢𑀺;_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸,_  
_𑀦 𑀳𑁂𑀢𑁂 𑀕𑀡𑀧𑀽𑀭𑀓𑀸𑁇_  


1915\. _𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄,_  
_𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀤𑁂𑀬𑁆𑀬 𑀯𑀸;_  
_𑀫𑀽𑀮𑀸 𑀫𑀸𑀦𑀢𑁆𑀢𑀫𑀩𑁆𑀪𑁂𑀬𑁆𑀬,_  
_𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀁𑁇_  


1916\. _𑀫𑀽𑀮𑀸 𑀅𑀭𑀳𑀫𑀸𑀦𑀢𑁆𑀢𑀸,_  
_𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑀫𑁂𑀯 𑀘;_  
_𑀦 𑀓𑀫𑁆𑀫𑀓𑀸𑀭𑀓𑀸 𑀧𑀜𑁆𑀘,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀓𑀸𑀲𑀺𑀢𑀸𑁇_  


1917\. _𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀘,_  
_𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀓𑀸;_  
_𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀫𑁅𑀫𑁆𑀫𑀢𑁆𑀢𑀸,_  
_𑀔𑀺𑀢𑁆𑀢𑀸𑀯𑁂𑀤𑀦𑀤𑀲𑁆𑀲𑀦𑁂𑁇_  


1918\. _𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸,_  
_𑀧𑀡𑁆𑀟𑀓𑀸𑀧𑀺 𑀘 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀸;_  
_𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀸 𑀲𑀻𑀫𑀸,_  
_𑀯𑁂𑀳𑀸𑀲𑀁 𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫 𑀘𑁇_  


1919\. _𑀅𑀝𑁆𑀞𑀸𑀭𑀲𑀦𑁆𑀦𑀫𑁂𑀢𑁂𑀲𑀁,_  
_𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀁 𑀦 𑀭𑀼𑀳𑀢𑀺;_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀓𑀢𑀢𑁆𑀢𑀲𑁆𑀲,_  
_𑀭𑀼𑀳𑀢𑀺 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸𑁇_  


1920\. _𑀲𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀤𑀼𑀦𑁆𑀦𑀺𑀲𑀸𑀭𑀺𑀢𑁄,_  
_𑀩𑀸𑀮𑁄 𑀳𑀺 𑀲𑀼𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄;_  
_𑀧𑀡𑁆𑀟𑀓𑁄 𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑁄,_  
_𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄𑁇_  


1921\. _𑀫𑀸𑀢𑀼 𑀧𑀺𑀢𑀼 𑀅𑀭𑀳𑀦𑁆𑀢,_  
_𑀤𑀽𑀲𑀓𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄;_  
_𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑁄 𑀘𑁂𑀯,_  
_𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑁄 𑀘 𑀬𑁄𑁇_  


1922\. _𑀏𑀓𑀸𑀤𑀲𑀦𑁆𑀦𑀁 𑀏𑀢𑁂𑀲𑀁,_  
_𑀑𑀲𑀸𑀭𑀡𑀁 𑀦 𑀬𑀼𑀚𑁆𑀚𑀢𑀺;_  
_𑀳𑀢𑁆𑀣𑀧𑀸𑀤𑀁 𑀢𑀤𑀼𑀪𑀬𑀁,_  
_𑀓𑀡𑁆𑀡𑀦𑀸𑀲𑀁 𑀢𑀤𑀽𑀪𑀬𑀁𑁇_  


1923\. _𑀅𑀗𑁆𑀕𑀼𑀮𑀺 𑀅𑀍𑀅𑀓𑀡𑁆𑀟𑀭𑀁,_  
_𑀨𑀡𑀁 𑀔𑀼𑀚𑁆𑀚𑁄 𑀘 𑀯𑀸𑀫𑀦𑁄;_  
_𑀕𑀡𑁆𑀟𑀻 𑀮𑀓𑁆𑀔𑀡𑀓𑀲𑀸 𑀘,_  
_𑀮𑀺𑀔𑀺𑀢𑀓𑁄 𑀘 𑀲𑀻𑀧𑀤𑀻𑁇_  


1924\. _𑀧𑀸𑀧𑀸 𑀧𑀭𑀺𑀲𑀓𑀸𑀡𑁄 𑀘,_  
_𑀓𑀼𑀡𑀻 𑀔𑀜𑁆𑀚𑁄 𑀳𑀢𑁄𑀧𑀺 𑀘;_  
_𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀤𑀼𑀩𑁆𑀩𑀮𑁄,_  
_𑀅𑀦𑁆𑀥𑁄 𑀫𑀽𑀕𑁄 𑀘 𑀩𑀥𑀺𑀭𑁄𑁇_  


1925\. _𑀅𑀦𑁆𑀥𑀫𑀽𑀕𑀦𑁆𑀥𑀩𑀥𑀺𑀭𑁄,_  
_𑀫𑀽𑀕𑀩𑀥𑀺𑀭𑀫𑁂𑀯 𑀘;_  
_𑀅𑀦𑁆𑀥𑀫𑀽𑀕𑀩𑀥𑀺𑀭𑁄 𑀘,_  
_𑀤𑁆𑀯𑀢𑁆𑀢𑀺𑀁𑀲𑁂𑀢𑁂 𑀅𑀦𑀽𑀦𑀓𑀸𑁇_  


1926\. _𑀢𑁂𑀲𑀁 𑀑𑀲𑀸𑀭𑀡𑀁 𑀳𑁄𑀢𑀺,_  
_𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦 𑀧𑀓𑀸𑀲𑀺𑀢𑀁;_  
_𑀤𑀝𑁆𑀞𑀩𑁆𑀩𑀸 𑀧𑀝𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸,_  
_𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀸 𑀦 𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇_  


1927\. _𑀢𑀲𑁆𑀲 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀸 𑀓𑀫𑁆𑀫𑀸,_  
_𑀲𑀢𑁆𑀢 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸;_  
_𑀆𑀧𑀦𑁆𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀲𑀢𑁆𑀢 𑀢𑁂𑀧𑀺 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑁇_  


1928\. _𑀆𑀧𑀦𑁆𑀦𑀁 𑀦𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁,_  
_𑀲𑀢𑁆𑀢 𑀓𑀫𑁆𑀫𑀸 𑀲𑀼𑀥𑀫𑁆𑀫𑀺𑀓𑀸;_  
_𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸 𑀘,_  
_𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀘 𑀓𑀸𑀭𑀡𑀸𑁇_  


1929\. _𑀲𑀢𑀺 𑀅𑀫𑀽𑀍𑀳𑀧𑀸𑀧𑀺𑀓𑀸,_  
_𑀢𑀚𑁆𑀚𑀦𑀻𑀦𑀺𑀬𑀲𑁆𑀲𑁂𑀦 𑀘;_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬 𑀧𑀝𑀺𑀲𑀸𑀭𑁄,_  
_𑀉𑀓𑁆𑀔𑁂𑀧𑀧𑀭𑀺𑀯𑀸𑀲 𑀘𑁇_  


1930\. _𑀫𑀽𑀮𑀸 𑀫𑀸𑀦𑀢𑁆𑀢𑀅𑀩𑁆𑀪𑀸𑀦𑀸,_  
_𑀢𑀣𑁂𑀯 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸;_  
_𑀅𑀜𑁆𑀜𑀁 𑀓𑀭𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀲𑁆𑀲,_  
_𑀲𑁄𑀍𑀅𑀲𑁂𑀢𑁂 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑁇_  


1931\. _𑀢𑀁 𑀢𑀁 𑀓𑀭𑁂𑀬𑁆𑀬 𑀢𑀁 𑀢𑀲𑁆𑀲,_  
_𑀲𑁄𑀍𑀅𑀲𑁂𑀢𑁂 𑀲𑀼𑀥𑀫𑁆𑀫𑀺𑀓𑀸;_  
_𑀧𑀘𑁆𑀘𑀸𑀭𑁄𑀧𑁂𑀬𑁆𑀬 𑀅𑀜𑁆𑀜𑀜𑁆𑀜𑀁,_  
_𑀲𑁄𑀍𑀅𑀲𑁂𑀢𑁂 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸𑁇_  


1932\. _𑀤𑁆𑀯𑁂 𑀤𑁆𑀯𑁂 𑀢𑀫𑁆𑀫𑀽𑀮𑀓𑀁 𑀢𑀲𑁆𑀲,_  
_𑀢𑁂𑀧𑀺 𑀲𑁄𑀍𑀅𑀲 𑀥𑀫𑁆𑀫𑀺𑀓𑀸;_  
_𑀏𑀓𑁂𑀓𑀫𑀽𑀮𑀓𑀁 𑀘𑀓𑁆𑀓𑀁,_  
_“𑀅𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺 𑀚𑀺𑀦𑁄𑀩𑁆𑀭𑀯𑀺𑁇_  


1933\. _𑀅𑀓𑀸𑀲𑀺 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀁𑀖𑁄 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁄;_  
_𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁,_  
_𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀺 𑀲𑁄𑁇_  


1934\. _𑀢𑀢𑁆𑀣𑀸𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸,_  
_𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀁 𑀓𑀭𑀼𑀁;_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀣 𑀯𑀕𑁆𑀕𑀸𑀥𑀫𑁆𑀫𑁂𑀦,_  
_𑀢𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀁 𑀓𑀭𑀼𑀁𑁇_  


1935\. _𑀧𑀢𑀺𑀭𑀽𑀧𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑀧𑀺,_  
_𑀲𑀫𑀕𑁆𑀕𑀸𑀧𑀺 𑀢𑀣𑀸 𑀓𑀭𑀼𑀁;_  
_𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸 𑀘,_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀫𑁂𑀯 𑀘𑁇_  


1936\. _𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸 𑀘,_  
_𑀲𑀫𑀕𑁆𑀕𑀸 𑀘 𑀇𑀫𑁂 𑀧𑀤𑀸;_  
_𑀏𑀓𑁂𑀓𑀫𑀽𑀮𑀓𑀁 𑀓𑀢𑁆𑀯𑀸,_  
_𑀘𑀓𑁆𑀓𑀁 𑀩𑀦𑁆𑀥𑁂 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄𑁇_  


1937\. _𑀩𑀸𑀮𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀁,_  
_𑀧𑀩𑁆𑀩𑀸𑀚𑁂 𑀓𑀼𑀮𑀤𑀽𑀲𑀓𑀁;_  
_𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁,_  
_𑀓𑀭𑁂 𑀅𑀓𑁆𑀓𑁄𑀲𑀓𑀲𑁆𑀲 𑀘𑁇_  


1938\. _𑀅𑀤𑀲𑁆𑀲𑀦𑀸𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂,_  
_𑀬𑁄 𑀘 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀦 𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂;_  
_𑀢𑁂𑀲𑀁 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁,_  
_𑀲𑀢𑁆𑀣𑀯𑀸𑀳𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀁𑁇_  


1939\. _𑀉𑀧𑀭𑀺 𑀦𑀬𑀓𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀧𑀜𑁆𑀜𑁄 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀁 𑀦𑀬𑁂;_  
_𑀢𑁂𑀲𑀁𑀬𑁂𑀯 𑀅𑀦𑀼𑀮𑁄𑀫𑀁,_  
_𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺 𑀬𑀸𑀘𑀺𑀢𑁂𑁇_  


1940\. _𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺 𑀢𑁂𑀲𑀁 𑀓𑀫𑁆𑀫𑀸𑀦𑀁,_  
_𑀳𑁂𑀝𑁆𑀞𑀸 𑀓𑀫𑁆𑀫𑀦𑀬𑁂𑀦 𑀘;_  
_𑀢𑀲𑁆𑀫𑀺𑀁 𑀢𑀲𑁆𑀫𑀺𑀁 𑀢𑀼 𑀓𑀫𑁆𑀫𑁂𑀲𑀼,_  
_𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑁄 𑀘 𑀯𑀺𑀯𑀤𑀢𑀺𑁇_  


1941\. _𑀅𑀓𑀢𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀜𑁆𑀘𑁂𑀯,_  
_𑀧𑀼𑀦𑀓𑀸𑀢𑀩𑁆𑀩𑀓𑀦𑁆𑀢𑀺 𑀘;_  
_𑀓𑀫𑁆𑀫𑁂 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀬𑀸 𑀘𑀸𑀧𑀺,_  
_𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀦𑁄𑁇_  


1942\. _𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀩𑁆𑀬𑀸𑀥𑀺𑀢𑁂 𑀤𑀺𑀲𑁆𑀯𑀸,_  
_𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀫𑀳𑀸𑀫𑀼𑀦𑀺;_  
_𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀫𑀓𑁆𑀔𑀸𑀲𑀺,_  
_𑀲𑀮𑁆𑀮𑀓𑀢𑁆𑀢𑁄𑀯 𑀑𑀲𑀥𑀦𑁆𑀢𑀺𑁇_  


---

1943\. 𑀇𑀫𑀫𑁆𑀳𑀺 𑀔𑀦𑁆𑀥𑀓𑁂 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀙𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀢𑀺𑁇



---

1944\. 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
