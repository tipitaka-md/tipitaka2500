# 𑁯𑁇𑁮 𑀉𑀧𑀸𑀮𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀉𑀧𑀸𑀮𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀓𑀣𑀸

(400.)

1829. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺? “𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀢𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 “𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀓𑀭𑁄𑀢𑀺…  𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀁 𑀤𑁂𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀺…  𑀧𑀭𑀺𑀯𑀸𑀲𑀸𑀭𑀳𑀁 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺…  𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑀲𑁆𑀲 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺…  𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀺…  𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺? “𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀢𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀁𑁇

1830. 𑀬𑁄 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀲𑀫𑁆𑀫𑀼𑀔𑀸𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀓𑀭𑁄𑀢𑀺, 𑀏𑀯𑀁 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀁, 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀲𑀁𑀖𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑁄 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀸 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬𑀓𑀭𑀡𑀻𑀬𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀧𑁆𑀧𑀝𑀺𑀜𑁆𑀜𑀸𑀬 𑀓𑀭𑁄𑀢𑀺…  𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀁 𑀤𑁂𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀺…  𑀧𑀭𑀺𑀯𑀸𑀲𑀸𑀭𑀳𑀁 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺…  𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑀲𑁆𑀲 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺…  𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀺…  𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺, 𑀏𑀯𑀁 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀲𑀁𑀖𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1838. 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀲𑀁𑀖𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀁 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺…  𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀸𑀭𑀳𑀲𑁆𑀲 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀁 𑀤𑁂𑀢𑀺, 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀲𑀁𑀖𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇

(406.)

1839. 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…𑀧𑁂…  𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲…  𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲…  𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲…  𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲…  𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀸𑀭𑀳𑀲𑁆𑀲…  𑀧𑀭𑀺𑀯𑀸𑀲𑀸𑀭𑀳𑀁…  𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑀲𑁆𑀲…  𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑀁…  𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑀁…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀲𑀢𑀺𑀯𑀺𑀦𑀬𑀁 𑀤𑁂𑀢𑀺, 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀲𑀁𑀖𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀬𑁄 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀁𑀖𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀅𑀫𑀽𑀍𑀳𑀯𑀺𑀦𑀬𑀁 𑀤𑁂𑀢𑀺…𑀧𑁂…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀲𑁆𑀲𑀧𑀸𑀧𑀺𑀬𑀲𑀺𑀓𑀸𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑁂𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀁 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀲𑁆𑀲 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺…  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀭𑀳𑀁 𑀅𑀩𑁆𑀪𑁂𑀢𑀺, 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀦𑀬𑀓𑀫𑁆𑀫𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦 𑀲𑀁𑀖𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

---
