# 𑁯𑁇𑁩𑁪 𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀤𑀲𑁆𑀲𑀦𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀓𑀣𑀸

(448.)

1894\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀢𑀺, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑀢𑀺, 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀸𑀘𑀢𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…𑀧𑁂…  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑁄 𑀲𑀁𑀖𑁄 𑀯𑀺𑀯𑀤𑀢𑀺—  ‘𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀅𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀼𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺, 𑀬𑁂 𑀘 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀼𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺, 𑀇𑀫𑁂 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀦𑁄𑁇 𑀇𑀫𑁂𑀧𑀺 𑀧𑀜𑁆𑀘 𑀯𑀸𑀭𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸𑁇
