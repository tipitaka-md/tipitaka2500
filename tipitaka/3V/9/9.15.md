# 𑁯𑁇𑁧𑁫 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀣𑀸

(417.)

1856\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺 𑀧𑀸𑀧𑀺𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀇𑀘𑁆𑀙𑀢𑀺 𑀧𑀸𑀧𑀺𑀓𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸…𑀧𑁂… 𑁇
