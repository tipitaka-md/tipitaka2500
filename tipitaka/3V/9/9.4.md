# 𑁯𑁇𑁪 𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀸𑀤𑀺𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀸𑀤𑀺𑀓𑀣𑀸

(388.)

1797\. 𑀧𑀜𑁆𑀘 𑀲𑀁𑀖𑀸—  𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀤𑀲𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀅𑀢𑀺𑀭𑁂𑀓𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀻𑀡𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺—  𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀧𑀯𑀸𑀭𑀡𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀓𑀫𑁆𑀫𑀸𑀦𑀺—  𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀩𑁆𑀪𑀸𑀦𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀤𑀲𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀁 𑀓𑀫𑁆𑀫𑀁—  𑀅𑀩𑁆𑀪𑀸𑀦𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁆𑀯𑀸𑀬𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑁄 𑀲𑀩𑁆𑀩𑀓𑀫𑁆𑀫𑁂𑀲𑀼 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑁄𑁇

(389.)

1798\. 𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁…𑀧𑁂… 𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀅𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀧𑀡𑁆𑀟𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀦𑀸𑀦𑀸𑀲𑀻𑀫𑀸𑀬 𑀞𑀺𑀢𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀯𑁂𑀳𑀸𑀲𑁂 𑀞𑀺𑀢𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀬𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀢𑀁𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇

1800\. 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀧𑀜𑁆𑀘𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…𑀧𑁂… 𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀧𑀡𑁆𑀟𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀦𑀸𑀦𑀸𑀲𑀻𑀫𑀸𑀬 𑀞𑀺𑀢𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀯𑁂𑀳𑀸𑀲𑁂 𑀞𑀺𑀢𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀬𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀢𑀁𑀧𑀜𑁆𑀘𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇

1802\. 𑀤𑀲𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀲𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀤𑀲𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀤𑀲𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁…𑀧𑁂… 𑁇 𑀤𑀲𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀬𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀢𑀁𑀤𑀲𑀫𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇

1804\. 𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀯𑀻𑀲𑀢𑀺𑀯𑀕𑁆𑀕𑀓𑀭𑀡𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀧𑀡𑁆𑀟𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀦𑀸𑀦𑀸𑀲𑀻𑀫𑀸𑀬 𑀞𑀺𑀢𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀯𑁂𑀳𑀸𑀲𑁂 𑀞𑀺𑀢𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬…  𑀬𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀢𑀁𑀯𑀻𑀲𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇
