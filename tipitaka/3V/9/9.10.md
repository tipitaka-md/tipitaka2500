# 𑁯𑁇𑁧𑁦 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀓𑀣𑀸

(412.)

1846\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄 𑀳𑁄𑀢𑀺 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑁄 𑀅𑀦𑀧𑀤𑀸𑀦𑁄, 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀳𑀺 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑁄 𑀅𑀦𑀧𑀤𑀸𑀦𑁄, 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀳𑀺 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸𑁇 𑀲𑁄 𑀢𑀫𑁆𑀳𑀸 𑀆𑀯𑀸𑀲𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑁂𑀦 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀓𑀢𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸…𑀧𑁂… 𑁇

---

1847\. 𑀬𑀣𑀸 𑀳𑁂𑀝𑁆𑀞𑀸, 𑀢𑀣𑀸 𑀘𑀓𑁆𑀓𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇
