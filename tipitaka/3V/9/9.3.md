# 𑁯𑁇𑁩 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀓𑀫𑁆𑀫𑀸𑀤𑀺𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀓𑀫𑁆𑀫𑀸𑀤𑀺𑀓𑀣𑀸

(385.)

1786\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺; 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺; 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀜𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁, 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺; 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀯𑀺𑀦𑀬𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺; 𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑀓𑀢𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀸𑀭𑀳𑀁𑁇

1787\. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀜𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁, 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀯𑀺𑀧𑀦𑁆𑀦𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀯𑀺𑀦𑀬𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑀓𑀢𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀸𑀭𑀳𑀦𑁆”𑀢𑀺𑁇

1788\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑀓𑀢𑀫𑁆𑀧𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀸𑀭𑀳𑀦𑁆”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

(386.)

1789\. “𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀯𑀺𑀧𑀦𑁆𑀦𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀜𑀢𑁆𑀢𑀺𑀯𑀺𑀧𑀦𑁆𑀦𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀯𑀺𑀧𑀦𑁆𑀦𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀥𑀫𑁆𑀫𑀸 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀯𑀺𑀦𑀬𑀸 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑀸𑀧𑀺 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑀸 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁; 𑀧𑀝𑀺𑀓𑀼𑀝𑁆𑀞𑀓𑀢𑀜𑁆𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀸𑀭𑀳𑀁 𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇

(387.)

1790\. 𑀙𑀬𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑀸𑀦𑀺—  𑀅𑀥𑀫𑁆𑀫𑀓𑀫𑁆𑀫𑀁, 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇

1792\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁? 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀅𑀦𑀸𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀅𑀦𑀸𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀅𑀦𑀸𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀅𑀦𑀸𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀅𑀦𑀸𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀅𑀦𑀸𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 (𑁨)

1793\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁? 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 (𑁩)

1795\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁? 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀧𑀞𑀫𑀁 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀜𑀢𑁆𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀧𑀞𑀫𑀁 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀜𑀢𑁆𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 (𑁫)

1796\. 𑀓𑀢𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁? 𑀜𑀢𑁆𑀢𑀺𑀤𑀼𑀢𑀺𑀬𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀧𑀞𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀏𑀓𑀸𑀬 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀬 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺—  𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀫𑁂 𑀧𑀞𑀫𑀁 𑀜𑀢𑁆𑀢𑀺𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀢𑀻𑀳𑀺 𑀓𑀫𑁆𑀫𑀯𑀸𑀘𑀸𑀳𑀺 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀫𑁆𑀫𑀧𑁆𑀧𑀢𑁆𑀢𑀸, 𑀢𑁂 𑀆𑀕𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀙𑀦𑁆𑀤𑀸𑀭𑀳𑀸𑀦𑀁 𑀙𑀦𑁆𑀤𑁄 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑀸 𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑁆𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 𑀇𑀤𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁𑁇 (𑁬)
