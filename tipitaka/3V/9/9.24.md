# 𑁯𑁇𑁨𑁪 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀯𑀺𑀯𑀸𑀤𑀓𑀣𑀸

* 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀯𑀺𑀯𑀸𑀤𑀓𑀣𑀸

(434.)

1880\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄 𑀳𑁄𑀢𑀺 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑁄 𑀅𑀦𑀧𑀤𑀸𑀦𑁄, 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀳𑀺 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀸𑀮𑁄 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑁄 𑀅𑀦𑀧𑀤𑀸𑀦𑁄, 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑁂𑀳𑀺 𑀕𑀺𑀳𑀺𑀲𑀁𑀲𑀕𑁆𑀕𑁂𑀳𑀺𑁇 𑀳𑀦𑁆𑀤𑀲𑁆𑀲 𑀫𑀬𑀁 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀢𑀲𑁆𑀲 𑀦𑀺𑀬𑀲𑁆𑀲𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…𑀧𑁂…  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀸…  𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸𑁇 𑀢𑀢𑁆𑀭𑀝𑁆𑀞𑁄 𑀲𑀁𑀖𑁄 𑀯𑀺𑀯𑀤𑀢𑀺—  ‘𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀯𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀁, 𑀅𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀼𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀥𑀫𑁆𑀫𑀧𑀢𑀺𑀭𑀽𑀧𑀓𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺, 𑀬𑁂 𑀘 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀅𑀓𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀼𑀦 𑀓𑀸𑀢𑀩𑁆𑀩𑀁 𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺, 𑀇𑀫𑁂 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀺𑀦𑁄𑁇 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀯𑀸𑀭𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑀸𑁇
