# 𑁯𑁇𑁫 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀸𑀤𑀺𑀓𑀣𑀸

### 𑀘𑀫𑁆𑀧𑁂𑀬𑁆𑀬𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀸𑀤𑀺𑀓𑀣𑀸

(393.)

1806\. 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀢𑀁𑀯𑀻𑀲𑁄 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀢𑀁𑀯𑀻𑀲𑁄 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀢𑀁𑀯𑀻𑀲𑁄 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀫𑀸𑀦𑀢𑁆𑀢𑀘𑀸𑀭𑀺𑀓𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀢𑀁𑀯𑀻𑀲𑁄 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀯𑀸𑀲𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑁂𑀬𑁆𑀬, 𑀫𑀸𑀦𑀢𑁆𑀢𑀁 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀢𑀁𑀯𑀻𑀲𑁄 𑀅𑀩𑁆𑀪𑁂𑀬𑁆𑀬—  𑀅𑀓𑀫𑁆𑀫𑀁 𑀦 𑀘 𑀓𑀭𑀡𑀻𑀬𑀁𑁇

(394.)

1807\. 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀭𑀼𑀳𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑀲𑁆𑀲 𑀦 𑀭𑀼𑀳𑀢𑀺𑁇 𑀓𑀲𑁆𑀲 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀦 𑀭𑀼𑀳𑀢𑀺? 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀦 𑀭𑀼𑀳𑀢𑀺𑁇 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀦𑀸𑀦𑀸𑀲𑀻𑀫𑀸𑀬 𑀞𑀺𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀇𑀤𑁆𑀥𑀺𑀬𑀸 𑀯𑁂𑀳𑀸𑀲𑁂 𑀞𑀺𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀦 𑀭𑀼𑀳𑀢𑀺𑁇 𑀇𑀫𑁂𑀲𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀦 𑀭𑀼𑀳𑀢𑀺𑁇

1808\. 𑀓𑀲𑁆𑀲 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀭𑀼𑀳𑀢𑀺? 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀓𑀢𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀲𑁆𑀲 𑀲𑀫𑀸𑀦𑀲𑀻𑀫𑀸𑀬 𑀞𑀺𑀢𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀆𑀦𑀦𑁆𑀢𑀭𑀺𑀓𑀲𑁆𑀲𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀭𑀼𑀳𑀢𑀺𑁇 𑀇𑀫𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸 𑀭𑀼𑀳𑀢𑀺𑁇
