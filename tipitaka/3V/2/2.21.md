# 𑁨𑁇𑁨𑁧 𑀙𑀦𑁆𑀤𑀤𑀸𑀦𑀓𑀣𑀸

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀙𑀦𑁆𑀤𑀤𑀸𑀦𑀓𑀣𑀸

(165.)

685. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀢𑁆𑀣𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀲𑁄 𑀅𑀦𑀸𑀕𑀢𑁄”𑀢𑀺𑁇

686. “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀙𑀦𑁆𑀤𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀙𑀦𑁆𑀤𑀁 𑀤𑀫𑁆𑀫𑀺, 𑀙𑀦𑁆𑀤𑀁 𑀫𑁂 𑀳𑀭, 𑀙𑀦𑁆𑀤𑀁 𑀫𑁂 𑀆𑀭𑁄𑀘𑁂𑀳𑀻’𑀢𑀺𑁇 𑀓𑀸𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀤𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇 𑀦 𑀓𑀸𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀤𑀺𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀜𑁆𑀘𑁂𑀦 𑀯𑀸 𑀧𑀻𑀞𑁂𑀦 𑀯𑀸 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀆𑀦𑁂𑀢𑁆𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀲𑀘𑁂 𑀔𑁄 𑀫𑀬𑀁 𑀕𑀺𑀮𑀸𑀦𑀁 𑀞𑀸𑀦𑀸 𑀘𑀸𑀯𑁂𑀲𑁆𑀲𑀸𑀫, 𑀆𑀩𑀸𑀥𑁄 𑀯𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺 𑀓𑀸𑀮𑀁𑀓𑀺𑀭𑀺𑀬𑀸 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀞𑀸𑀦𑀸 𑀘𑀸𑀯𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀁𑀖𑁂𑀦 𑀢𑀢𑁆𑀣 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀯𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀓𑀭𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

687. 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀢𑀢𑁆𑀣𑁂𑀯 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀙𑀦𑁆𑀤𑁄𑁇 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀢𑀢𑁆𑀣𑁂𑀯 𑀯𑀺𑀩𑁆𑀪𑀫𑀢𑀺…𑀧𑁂…  𑀓𑀸𑀮𑀁 𑀓𑀭𑁄𑀢𑀺…  𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀅𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀉𑀫𑁆𑀫𑀢𑁆𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀔𑀺𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀯𑁂𑀤𑀦𑀸𑀝𑁆𑀝𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀧𑀡𑁆𑀟𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀙𑀦𑁆𑀤𑁄𑁇

688. 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀅𑀦𑀸𑀳𑀝𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀯𑀺𑀩𑁆𑀪𑀫𑀢𑀺…𑀧𑁂…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀅𑀦𑀸𑀳𑀝𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇

689. 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀲𑀁𑀖𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺, 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀲𑀁𑀖𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀯𑀺𑀩𑁆𑀪𑀫𑀢𑀺…𑀧𑁂…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑁄 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇

690. 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀲𑀁𑀖𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀼𑀢𑁆𑀢𑁄 𑀦 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀧𑀫𑀢𑁆𑀢𑁄 𑀦 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄 𑀦 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑀲𑁆𑀲 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

691. 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀦𑁆𑀦𑁂 𑀙𑀦𑁆𑀤𑁂 𑀲𑀁𑀖𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀦 𑀆𑀭𑁄𑀘𑁂𑀢𑀺, 𑀆𑀳𑀝𑁄 𑀳𑁄𑀢𑀺 𑀙𑀦𑁆𑀤𑁄𑁇 𑀙𑀦𑁆𑀤𑀳𑀸𑀭𑀓𑀲𑁆𑀲 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀤𑁂𑀦𑁆𑀢𑁂𑀦 𑀙𑀦𑁆𑀤𑀫𑁆𑀧𑀺 𑀤𑀸𑀢𑀼𑀁, 𑀲𑀦𑁆𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀭𑀡𑀻𑀬𑀦𑁆”𑀢𑀺𑁇
