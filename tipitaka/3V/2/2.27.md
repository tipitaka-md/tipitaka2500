# 𑁨𑁇𑁨𑁭 𑀲𑀪𑀸𑀕𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀯𑀺𑀥𑀺

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀪𑀸𑀕𑀸𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀯𑀺𑀥𑀺

(171.)

717\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀲𑀪𑀸𑀕𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑀪𑀸𑀕𑀸 𑀆𑀧𑀢𑁆𑀢𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸’𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

718\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀚𑁆𑀚𑀼𑀓𑀁 𑀧𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄—  𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙; 𑀫𑀬𑀁 𑀢𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

719\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀬𑀤𑀸 𑀅𑀜𑁆𑀜𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀲𑀼𑀤𑁆𑀥𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀤𑀸 𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁄𑀲𑀣𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

720\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

721\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑁄𑁇 𑀬𑀤𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀫𑀢𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀤𑀸 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁; 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁄𑀲𑀣𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

722\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀚𑁆𑀚𑀼𑀓𑀁 𑀧𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄—  𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙; 𑀫𑀬𑀁 𑀢𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀧𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄—  𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙; 𑀫𑀬𑀁 𑀢𑁂 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

723\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁𑁇 𑀢𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀆𑀕𑀢𑀸𑀕𑀫𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 𑀫𑀸𑀢𑀺𑀓𑀸𑀥𑀭𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀮𑀚𑁆𑀚𑀻 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀓𑁄 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀜𑁆𑀘𑁂𑀯𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀺𑀁 𑀦𑀸𑀫 𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀜𑁆𑀘𑁂𑀯𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺, 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄; 𑀧𑀝𑀺𑀓𑀭𑁄𑀳𑀺 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀓𑁄𑀯 𑀇𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄; 𑀅𑀬𑀁 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀇𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀓𑀺𑀁 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑁄 𑀆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀦𑀸𑀧𑀦𑁆𑀦𑁄 𑀯𑀸𑁇 𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀓𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀘𑀦𑁂𑀦 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑁄 𑀓𑀺𑀭, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀜𑁆𑀘𑁂𑀯𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺, 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑀸; 𑀧𑀝𑀺𑀓𑀭𑁄𑀣 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀇𑀘𑁆𑀙𑀺𑀁𑀲𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀘𑀦𑁂𑀦 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀸𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

724\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀲𑀪𑀸𑀕𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀦 𑀚𑀸𑀦𑀸𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁𑁇 𑀢𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀆𑀕𑀢𑀸𑀕𑀫𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 𑀫𑀸𑀢𑀺𑀓𑀸𑀥𑀭𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀮𑀚𑁆𑀚𑀻 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀓𑁄 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀬𑁄 𑀦𑀼 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀜𑁆𑀘𑁂𑀯𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺, 𑀓𑀺𑀁 𑀦𑀸𑀫 𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀬𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀜𑁆𑀘𑁂𑀯𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺, 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄; 𑀧𑀝𑀺𑀓𑀭𑁄𑀳𑀺 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀓𑁄𑀯 𑀇𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀬𑀁 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀇𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀓𑀺𑀁 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀭𑁄 𑀆𑀧𑀦𑁆𑀦𑁄 𑀯𑀸 𑀅𑀦𑀸𑀧𑀦𑁆𑀦𑁄 𑀯𑀸𑁇 𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀓𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀘𑀦𑁂𑀦 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  ‘𑀬𑁄 𑀓𑀺𑀭, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀜𑁆𑀘𑁂𑀯𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀚𑁆𑀚𑀢𑀺, 𑀇𑀫𑀁 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀯𑀼𑀲𑁄 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑀸, 𑀧𑀝𑀺𑀓𑀭𑁄𑀣 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀦𑁆’𑀢𑀺𑁇 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀯𑀘𑀦𑁂𑀦 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀧𑀝𑀺𑀓𑀭𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀓𑀸𑀫𑀸 𑀯𑀘𑀦𑀻𑀬𑀸”𑀢𑀺𑁇
