# 𑁨𑁇𑁭 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀻𑀫𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀻𑀫𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(143.)

607. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀅𑀦𑁆𑀥𑀓𑀯𑀺𑀦𑁆𑀤𑀸 𑀭𑀸𑀚𑀕𑀳𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀦𑀤𑀺𑀁 𑀢𑀭𑀦𑁆𑀢𑁄 𑀫𑀦𑀁 𑀯𑀽𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀘𑀻𑀯𑀭𑀸𑀦𑀺𑀲𑁆𑀲 𑀅𑀮𑁆𑀮𑀸𑀦𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀮𑁆𑀮𑀸𑀦𑀻”𑀢𑀺? “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀦𑁆𑀥𑀓𑀯𑀺𑀦𑁆𑀤𑀸 𑀭𑀸𑀚𑀕𑀳𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀦𑀤𑀺𑀁 𑀢𑀭𑀦𑁆𑀢𑁄 𑀫𑀦𑀫𑁆𑀳𑀺 𑀯𑀽𑀍𑀳𑁄𑁇 𑀢𑁂𑀦 𑀫𑁂 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀮𑁆𑀮𑀸𑀦𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀬𑀸 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

608. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸 𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸, 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

609. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸 𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀲𑀻𑀫𑀸𑀬 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀸 𑀲𑀻𑀫𑀸 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

610. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑀸 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀟𑀬𑁆𑀳𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀦𑁆𑀤𑀽𑀭𑁂𑀳𑀺𑀧𑀺 𑀔𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀘𑁆𑀘𑁄𑀍𑀆 𑀳𑁄𑀦𑁆𑀢𑀺 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑀼𑀘𑁆𑀘𑁄𑀍𑀆 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸”𑀢𑀺? “𑀇𑀥 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑁂 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀫𑁆𑀳𑀸𑁇 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀦𑀝𑁆𑀞𑀸𑀦𑀺𑀧𑀺 𑀤𑀟𑁆𑀠𑀸𑀦𑀺𑀧𑀺, 𑀉𑀦𑁆𑀤𑀽𑀭𑁂𑀳𑀺𑀧𑀺 𑀔𑀸𑀬𑀺𑀢𑀸𑀦𑀺, 𑀢𑁂𑀦 𑀫𑀬𑀁 𑀤𑀼𑀘𑁆𑀘𑁄𑀍𑀆 𑀮𑀽𑀔𑀘𑀻𑀯𑀭𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀬𑀸 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀼, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀜𑁆𑀘 𑀕𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑀜𑁆𑀘𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(144.)

611. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸 𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀜𑁆𑀘 𑀕𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑀜𑁆𑀘𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

612. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸 𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀜𑁆𑀘 𑀕𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑀜𑁆𑀘𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀲𑀻𑀫𑀸𑀬 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀸𑀬 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀜𑁆𑀘 𑀕𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑀜𑁆𑀘, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀸 𑀲𑀻𑀫𑀸 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀸, 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀕𑀸𑀫𑀜𑁆𑀘 𑀕𑀸𑀫𑀽𑀧𑀘𑀸𑀭𑀜𑁆𑀘𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇
