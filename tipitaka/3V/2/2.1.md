# 𑁨𑁇𑁧 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(132.)

580\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑁇 𑀢𑁂 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀲𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀲𑀼 𑀧𑁂𑀫𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀧𑀲𑀸𑀤𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀧𑀓𑁆𑀔𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑁇 𑀢𑁂 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀲𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀲𑀼 𑀧𑁂𑀫𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀧𑀲𑀸𑀤𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀧𑀓𑁆𑀔𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀅𑀬𑁆𑀬𑀸𑀧𑀺 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇

581\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺 ‘𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑁇 𑀢𑁂 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀲𑀼 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂𑀲𑀼 𑀧𑁂𑀫𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀧𑀲𑀸𑀤𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀧𑀓𑁆𑀔𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀅𑀬𑁆𑀬𑀸𑀧𑀺 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸𑀧𑀺 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

582\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀼𑀦𑁆”𑀢𑀺—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀦𑀺𑀲𑀻𑀤𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑁇 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀽𑀕𑀲𑀽𑀓𑀭𑀸𑁇 𑀦𑀦𑀼 𑀦𑀸𑀫 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀪𑀸𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
