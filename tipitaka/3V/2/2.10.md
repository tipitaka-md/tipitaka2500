# 𑁨𑁇𑁧𑁦 𑀉𑀧𑁄𑀲𑀣𑀪𑁂𑀤𑀸𑀤𑀺

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀉𑀧𑁄𑀲𑀣𑀪𑁂𑀤𑀸𑀤𑀺

(149.)

622\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀉𑀧𑁄𑀲𑀣𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀤𑁆𑀯𑁂𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁄𑀲𑀣𑀸—  𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀓𑁄 𑀘 𑀧𑀦𑁆𑀦𑀭𑀲𑀺𑀓𑁄 𑀘𑁇 𑀇𑀫𑁂 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀉𑀧𑁄𑀲𑀣𑀸”𑀢𑀺𑁇

623\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀸𑀦𑀻”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀫𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀸𑀦𑀺—  𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁, 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁, 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀤𑀺𑀤𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁, 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀦 𑀘 𑀫𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀤𑀺𑀤𑀁 𑀅𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁, 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀦 𑀘 𑀫𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀤𑀺𑀤𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀯𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁, 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀦 𑀘 𑀫𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀤𑀺𑀤𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁, 𑀏𑀯𑀭𑀽𑀧𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁, 𑀏𑀯𑀭𑀽𑀧𑀜𑁆𑀘 𑀫𑀬𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁𑁇 𑀢𑀲𑁆𑀫𑀸𑀢𑀺𑀳, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫 𑀬𑀤𑀺𑀤𑀁 𑀥𑀫𑁆𑀫𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀦𑁆𑀢𑀺—  𑀏𑀯𑀜𑁆𑀳𑀺 𑀯𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
