# 𑁨𑁇𑁫 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀓𑀣𑀸

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀓𑀣𑀸

(141.)

596\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑁂𑀡𑀺𑀬𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺 𑀅𑀲𑀗𑁆𑀓𑁂𑀢𑁂𑀦𑁇 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  “𑀓𑀢𑁆𑀣 𑀯𑀸 𑀅𑀚𑁆𑀚𑀼𑀧𑁄𑀲𑀣𑁄 𑀓𑀭𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑁂𑀡𑀺𑀬𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁 𑀅𑀲𑀗𑁆𑀓𑁂𑀢𑁂𑀦𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀬𑀁 𑀲𑀁𑀖𑁄 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀸 𑀅𑀟𑁆𑀠𑀬𑁄𑀕𑀁 𑀯𑀸 𑀧𑀸𑀲𑀸𑀤𑀁 𑀯𑀸 𑀳𑀫𑁆𑀫𑀺𑀬𑀁 𑀯𑀸 𑀕𑀼𑀳𑀁 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

597\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀯𑀺𑀳𑀸𑀭𑀁 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

598\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀯𑀺𑀳𑀸𑀭𑀁 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑁆𑀫𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

599\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀤𑁆𑀯𑁂 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀪𑀬𑀢𑁆𑀣 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺—  “𑀇𑀥 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀭𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀇𑀥 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀭𑀻𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀤𑁆𑀯𑁂 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀬𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀁 𑀲𑀫𑀽𑀳𑀦𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁆𑀣 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀽𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

600\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑀽𑀳𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

601\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑀽𑀳𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀲𑁆𑀲 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑀽𑀳𑀢𑀁 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇
