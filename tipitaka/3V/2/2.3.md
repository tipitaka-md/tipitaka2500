# 𑁨𑁇𑁩 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀯𑀢𑁆𑀣𑀼

(137.)

590\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀤𑁆𑀤𑀓𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀺 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀯𑀸𑀳𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀦 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀯𑀸𑀳𑀁 𑀲𑀁𑀖𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀧𑀭𑀫𑀸𑀬 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀤𑁆𑀤𑀓𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀺 𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀦𑀼 𑀢𑁂, 𑀓𑀧𑁆𑀧𑀺𑀦, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀯𑀸𑀳𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀦 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁 𑀯𑀸𑀳𑀁 𑀲𑀁𑀖𑀓𑀫𑁆𑀫𑀁 𑀦 𑀯𑀸 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁, 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀧𑀭𑀫𑀸𑀬 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀢𑀼𑀫𑁆𑀳𑁂 𑀘𑁂 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀦 𑀲𑀓𑁆𑀓𑀭𑀺𑀲𑁆𑀲𑀣 𑀦 𑀕𑀭𑀼𑀓𑀭𑀺𑀲𑁆𑀲𑀣 𑀦 𑀫𑀸𑀦𑁂𑀲𑁆𑀲𑀣 𑀦 𑀧𑀽𑀚𑁂𑀲𑁆𑀲𑀣, 𑀅𑀣 𑀓𑁄 𑀘𑀭𑀳𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀲𑀓𑁆𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀕𑀭𑀼𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀫𑀸𑀦𑁂𑀲𑁆𑀲𑀢𑀺 𑀧𑀽𑀚𑁂𑀲𑁆𑀲𑀢𑀺? 𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡, 𑀉𑀧𑁄𑀲𑀣𑀁, 𑀫𑀸 𑀦𑁄 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁 𑀲𑀁𑀖𑀓𑀫𑁆𑀫𑀁, 𑀫𑀸 𑀦𑁄 𑀅𑀕𑀫𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀫𑀤𑁆𑀤𑀓𑀼𑀘𑁆𑀙𑀺𑀫𑁆𑀳𑀺 𑀫𑀺𑀕𑀤𑀸𑀬𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀧𑁆𑀧𑀺𑀦𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀔𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇
