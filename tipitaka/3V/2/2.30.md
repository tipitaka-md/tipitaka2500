# 𑁨𑁇𑁩𑁦 𑀯𑁂𑀫𑀢𑀺𑀓𑀧𑀦𑁆𑀦𑀭𑀲𑀓

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀯𑁂𑀫𑀢𑀺𑀓𑀧𑀦𑁆𑀦𑀭𑀲𑀓

(174.)

748\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀼𑀁 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂, 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀢𑀭𑀸𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀼𑀦 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀉𑀤𑁆𑀤𑁂𑀲𑀓𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁧)

749\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀼𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂, 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀫𑀲𑀫𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁, 𑀅𑀯𑀲𑁂𑀲𑀁 𑀲𑁄𑀢𑀩𑁆𑀩𑀁𑁇 𑀉𑀤𑁆𑀤𑁂𑀲𑀓𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁨)

750\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀼𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀣𑁄𑀓𑀢𑀭𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁, 𑀅𑀯𑀲𑁂𑀲𑀁 𑀲𑁄𑀢𑀩𑁆𑀩𑀁𑁇 𑀉𑀤𑁆𑀤𑁂𑀲𑀓𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁩)

751\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀼𑀁 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑁂 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂…𑀧𑁂…  𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀏𑀓𑀘𑁆𑀘𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬, 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀢𑀭𑀸…𑀧𑁂…  𑀲𑀫𑀲𑀫𑀸…𑀧𑁂…  𑀣𑁄𑀓𑀢𑀭𑀸𑁇 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁 𑀲𑀼𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀁, 𑀢𑁂𑀲𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺 𑀆𑀭𑁄𑀘𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀉𑀤𑁆𑀤𑁂𑀲𑀓𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁪--𑁧𑁫)
