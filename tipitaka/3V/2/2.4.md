# 𑁨𑁇𑁪 𑀲𑀻𑀫𑀸𑀦𑀼𑀚𑀸𑀦𑀦

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀻𑀫𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(138.)

591\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀬𑀸𑀯𑀢𑀸 𑀏𑀓𑀸𑀯𑀸𑀲𑁄’𑀢𑀺, 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄 𑀏𑀓𑀸𑀯𑀸𑀲𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸—  𑀧𑀞𑀫𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑀩𑁆𑀩𑀸—  𑀧𑀩𑁆𑀩𑀢𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀧𑀸𑀲𑀸𑀡𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀯𑀦𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀭𑀼𑀓𑁆𑀔𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀫𑀕𑁆𑀕𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀯𑀫𑁆𑀫𑀺𑀓𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀦𑀤𑀻𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀉𑀤𑀓𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁𑁇 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(139.)

592\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸𑀯𑀢𑀸 𑀲𑀫𑀦𑁆𑀢𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀁 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀁𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

593\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸𑀯𑀢𑀸 𑀲𑀫𑀦𑁆𑀢𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇 𑀲𑀁𑀖𑁄 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀁 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀁𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀻𑀫𑀸𑀬 𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸𑀬 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸𑀬, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀻𑀫𑀸 𑀲𑀁𑀖𑁂𑀦 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(140.)

594\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀻𑀫𑀸𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸”𑀢𑀺—  𑀅𑀢𑀺𑀫𑀳𑀢𑀺𑀬𑁄 𑀲𑀻𑀫𑀸𑀬𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀦𑁆𑀢𑀺, 𑀘𑀢𑀼𑀬𑁄𑀚𑀦𑀺𑀓𑀸𑀧𑀺 𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀺𑀓𑀸𑀧𑀺 𑀙𑀬𑁄𑀚𑀦𑀺𑀓𑀸𑀧𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑁄𑀲𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀸 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂𑀧𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑁂𑀧𑀺 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀦𑁆𑀢𑀭𑀸𑀧𑀺 𑀧𑀭𑀺𑀯𑀲𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀫𑀳𑀢𑀻 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸, 𑀘𑀢𑀼𑀬𑁄𑀚𑀦𑀺𑀓𑀸 𑀯𑀸 𑀧𑀜𑁆𑀘𑀬𑁄𑀚𑀦𑀺𑀓𑀸 𑀯𑀸 𑀙𑀬𑁄𑀚𑀦𑀺𑀓𑀸 𑀯𑀸𑁇 𑀬𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀬𑁄𑀚𑀦𑀧𑀭𑀫𑀁 𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

595\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀤𑀻𑀧𑀸𑀭𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑁄𑀲𑀣𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀯𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑀺, 𑀧𑀢𑁆𑀢𑀸𑀧𑀺 𑀯𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑀺, 𑀘𑀻𑀯𑀭𑀸𑀦𑀺𑀧𑀺 𑀯𑀼𑀬𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀤𑀻𑀧𑀸𑀭𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲 𑀥𑀼𑀯𑀦𑀸𑀯𑀸 𑀯𑀸 𑀥𑀼𑀯𑀲𑁂𑀢𑀼 𑀯𑀸, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀦𑀤𑀻𑀧𑀸𑀭𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
