# 𑁨𑁇𑁧𑁭 𑀧𑀓𑁆𑀔𑀕𑀡𑀦𑀸𑀤𑀺𑀉𑀕𑁆𑀕𑀳𑀡𑀸𑀦𑀼𑀚𑀸𑀦𑀦

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀓𑁆𑀔𑀕𑀡𑀦𑀸𑀤𑀺𑀉𑀕𑁆𑀕𑀳𑀡𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(156.)

653\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀘𑁄𑀤𑀦𑀸𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀜𑁆𑀙𑀺𑁇

654\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀓𑀢𑀺𑀫𑀻, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀓𑁆𑀔𑀲𑁆𑀲𑀸”𑀢𑀺? 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀸𑀦𑀸𑀫𑀸”𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀧𑀓𑁆𑀔𑀕𑀡𑀦𑀫𑀢𑁆𑀢𑀫𑁆𑀧𑀺𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀓𑀺𑀁 𑀧𑀦𑀺𑀫𑁂 𑀅𑀜𑁆𑀜𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀓𑁆𑀔𑀕𑀡𑀦𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀧𑀓𑁆𑀔𑀕𑀡𑀦𑀸 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀧𑀓𑁆𑀔𑀕𑀡𑀦𑀁 𑀉𑀕𑁆𑀕𑀳𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(157.)

655\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀓𑀻𑀯𑀢𑀺𑀓𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀸𑀦𑀸𑀫𑀸”𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀫𑁆𑀧𑀺𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺, 𑀓𑀺𑀁 𑀧𑀦𑀺𑀫𑁂 𑀅𑀜𑁆𑀜𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀡𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

656\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀤𑀸 𑀦𑀼 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀡𑁂𑀢𑀩𑁆𑀩𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀦𑀸𑀫𑀕𑁆𑀕𑁂𑀦 𑀯𑀸 𑀕𑀡𑁂𑀢𑀼𑀁, 𑀲𑀮𑀸𑀓𑀁 𑀯𑀸 𑀕𑀸𑀳𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(158.)

657\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀚𑀸𑀦𑀦𑁆𑀢𑀸 𑀅𑀚𑁆𑀚𑀼𑀧𑁄𑀲𑀣𑁄𑀢𑀺 𑀤𑀽𑀭𑀁 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂𑀧𑀺 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀉𑀤𑁆𑀤𑀺𑀝𑁆𑀞𑀫𑀢𑁆𑀢𑁂𑀧𑀺 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑁄𑀘𑁂𑀢𑀼𑀁 ‘𑀅𑀚𑁆𑀚𑀼𑀧𑁄𑀲𑀣𑁄’”𑀢𑀺𑁇

658\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀆𑀭𑁄𑀘𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑁂𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀓𑀸𑀮𑀯𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

659\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀣𑁂𑀭𑁄 𑀓𑀸𑀮𑀯𑀢𑁄 𑀦𑀲𑁆𑀲𑀭𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀢𑁆𑀢𑀓𑀸𑀮𑁂𑀧𑀺 𑀆𑀭𑁄𑀘𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

660\. 𑀪𑀢𑁆𑀢𑀓𑀸𑀮𑁂𑀧𑀺 𑀦𑀲𑁆𑀲𑀭𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀁 𑀓𑀸𑀮𑀁 𑀲𑀭𑀢𑀺, 𑀢𑀁 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
