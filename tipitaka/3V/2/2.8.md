# 𑁨𑁇𑁮 𑀲𑀻𑀫𑀸𑀲𑀫𑀽𑀳𑀦𑀦

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀻𑀫𑀸𑀲𑀫𑀽𑀳𑀦𑀦

(145.)

613\. “𑀲𑀻𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀦𑁆𑀢𑁂𑀦 𑀧𑀞𑀫𑀁 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸, 𑀧𑀘𑁆𑀙𑀸 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀻𑀫𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀽𑀳𑀦𑀦𑁆𑀢𑁂𑀦 𑀧𑀞𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁄 𑀲𑀫𑀽𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑁄, 𑀧𑀘𑁆𑀙𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀲𑀻𑀫𑀸 𑀲𑀫𑀽𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁄 𑀲𑀫𑀽𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

614\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑁄 𑀲𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁄 𑀲𑀫𑁆𑀫𑀢𑁄, 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑀽𑀳𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

615\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑁄 𑀲𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁄 𑀲𑀫𑁆𑀫𑀢𑁄, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀁 𑀲𑀫𑀽𑀳𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀏𑀢𑀲𑁆𑀲 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀲𑁆𑀲 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑀽𑀳𑀢𑁄 𑀲𑁄 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑁄𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇

616\. 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸 𑀲𑀫𑀽𑀳𑀦𑁆𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(146.)

617\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸 𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸, 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀲𑀫𑀽𑀳𑀦𑁂𑀬𑁆𑀬 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀁 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀁𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

618\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸 𑀲𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀲𑀻𑀫𑀁 𑀲𑀫𑀽𑀳𑀦𑀢𑀺 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀁 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀁𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀏𑀢𑀺𑀲𑁆𑀲𑀸 𑀲𑀻𑀫𑀸𑀬 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸𑀬 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸𑀬 𑀲𑀫𑀼𑀕𑁆𑀖𑀸𑀢𑁄, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑀽𑀳𑀢𑀸 𑀲𑀸 𑀲𑀻𑀫𑀸 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇
