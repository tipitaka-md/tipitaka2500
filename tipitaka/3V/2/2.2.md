# 𑁨𑁇𑁨 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(133.)

583\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑀸𑀦𑀺 𑀫𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀦𑁂𑀲𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁𑁇 𑀲𑁄 𑀦𑁂𑀲𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺 ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀬𑀸𑀦𑀺 𑀫𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀸𑀦𑀺, 𑀢𑀸𑀦𑀺 𑀦𑁂𑀲𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁𑁇 𑀲𑁄 𑀦𑁂𑀲𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(134.)

584\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑁇 𑀓𑀺𑀁 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑀓𑀺𑀘𑁆𑀘𑀁? 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀆𑀭𑁄𑀘𑁂𑀣𑁇 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀁 𑀲𑀩𑁆𑀩𑁂𑀯 𑀲𑀦𑁆𑀢𑀸 𑀲𑀸𑀥𑀼𑀓𑀁 𑀲𑀼𑀡𑁄𑀫 𑀫𑀦𑀲𑀺 𑀓𑀭𑁄𑀫𑁇 𑀬𑀲𑁆𑀲 𑀲𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺, 𑀲𑁄 𑀆𑀯𑀺𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀅𑀲𑀦𑁆𑀢𑀺𑀬𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀼𑀡𑁆𑀳𑀻 𑀪𑀯𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀢𑀺 𑀯𑁂𑀤𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀬𑀣𑀸 𑀔𑁄 𑀧𑀦 𑀧𑀘𑁆𑀘𑁂𑀓𑀧𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀁 𑀳𑁄𑀢𑀺, 𑀏𑀯𑀫𑁂𑀯𑀁 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀺𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀬𑁄 𑀧𑀦 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀯𑀢𑀢𑀺𑀬𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀺𑀬𑀫𑀸𑀦𑁂 𑀲𑀭𑀫𑀸𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀦𑀸𑀯𑀺𑀓𑀭𑁂𑀬𑁆𑀬, 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀲𑁆𑀲 𑀳𑁄𑀢𑀺𑁇 𑀲𑀫𑁆𑀧𑀚𑀸𑀦𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀔𑁄 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸𑁇 𑀢𑀲𑁆𑀫𑀸, 𑀲𑀭𑀫𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀆𑀧𑀦𑁆𑀦𑁂𑀦 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀧𑁂𑀓𑁆𑀔𑁂𑀦 𑀲𑀦𑁆𑀢𑀻 𑀆𑀧𑀢𑁆𑀢𑀺 𑀆𑀯𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸; 𑀆𑀯𑀺𑀓𑀢𑀸 𑀳𑀺𑀲𑁆𑀲 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀻’”𑀢𑀺𑁇

𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀳𑁄𑀢𑀺𑁇 **𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄 𑀥𑀫𑁆𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸**𑀢𑀺𑁇 𑀓𑀺𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄? 𑀧𑀞𑀫𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄, 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄, 𑀢𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄, 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄, 𑀛𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑀫𑀸𑀥𑀻𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀸𑀦𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁄𑁇 **𑀢𑀲𑁆𑀫𑀸**𑀢𑀺 𑀢𑀁𑀓𑀸𑀭𑀡𑀸𑁇 **𑀲𑀭𑀫𑀸𑀦𑁂𑀦𑀸**𑀢𑀺 𑀚𑀸𑀦𑀫𑀸𑀦𑁂𑀦 𑀲𑀜𑁆𑀚𑀸𑀦𑀫𑀸𑀦𑁂𑀦𑁇 **𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀧𑁂𑀓𑁆𑀔𑁂𑀦𑀸**𑀢𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀼𑀓𑀸𑀫𑁂𑀦 𑀯𑀺𑀲𑀼𑀚𑁆𑀛𑀺𑀢𑀼𑀓𑀸𑀫𑁂𑀦𑁇 **𑀲𑀦𑁆𑀢𑀻** 𑀦𑀸𑀫 **𑀆𑀧𑀢𑁆𑀢𑀺** 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀸 𑀯𑀸 𑀳𑁄𑀢𑀺, 𑀆𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀸 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑁇 **𑀆𑀯𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸**𑀢𑀺 𑀆𑀯𑀺𑀓𑀸𑀢𑀩𑁆𑀩𑀸 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀯𑀸 𑀕𑀡𑀫𑀚𑁆𑀛𑁂 𑀯𑀸 𑀏𑀓𑀧𑀼𑀕𑁆𑀕𑀮𑁂 𑀯𑀸𑁇 **𑀆𑀯𑀺𑀓𑀢𑀸 𑀳𑀺𑀲𑁆𑀲 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀻**𑀢𑀺 𑀓𑀺𑀲𑁆𑀲 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺? 𑀧𑀞𑀫𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺, 𑀤𑀼𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺, 𑀢𑀢𑀺𑀬𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺, 𑀘𑀢𑀼𑀢𑁆𑀣𑀲𑁆𑀲 𑀛𑀸𑀦𑀲𑁆𑀲 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺, 𑀛𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑀫𑀸𑀥𑀻𑀦𑀁 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀻𑀦𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑀸𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀭𑀡𑀸𑀦𑀁 𑀧𑀯𑀺𑀯𑁂𑀓𑀸𑀦𑀁 𑀓𑀼𑀲𑀮𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀸𑀦𑀁 𑀅𑀥𑀺𑀕𑀫𑀸𑀬 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀻𑀢𑀺𑁇

(136.)

586\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄”𑀢𑀺—  𑀤𑁂𑀯𑀲𑀺𑀓𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁂𑀯𑀲𑀺𑀓𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑁄𑀲𑀣𑁂 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

587\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀉𑀧𑁄𑀲𑀣𑁂 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑁄 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄”𑀢𑀺—  𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺, 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀅𑀝𑁆𑀞𑀫𑀺𑀬𑀸 𑀘 𑀧𑀓𑁆𑀔𑀲𑁆𑀲𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀢𑀺𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀓𑀺𑀁 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀯𑀸 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

588\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑀣𑀸𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀦𑁆𑀢𑀺 𑀲𑀓𑀸𑀬 𑀲𑀓𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀣𑀸𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁 𑀲𑀓𑀸𑀬 𑀲𑀓𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇

589\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀦𑁆’𑀢𑀺𑁇 𑀓𑀺𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀦𑀼 𑀔𑁄 𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀳𑁄𑀢𑀺, 𑀬𑀸𑀯𑀢𑀸 𑀏𑀓𑀸𑀯𑀸𑀲𑁄, 𑀉𑀤𑀸𑀳𑀼 𑀲𑀩𑁆𑀩𑀸 𑀧𑀣𑀯𑀻”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑁆𑀢𑀸𑀯𑀢𑀸 𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀬𑀸𑀯𑀢𑀸 𑀏𑀓𑀸𑀯𑀸𑀲𑁄”𑀢𑀺𑁇
