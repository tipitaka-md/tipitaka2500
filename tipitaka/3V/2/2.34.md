# 𑁨𑁇𑁩𑁪 𑀮𑀺𑀗𑁆𑀕𑀸𑀤𑀺𑀤𑀲𑁆𑀲𑀦

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀮𑀺𑀗𑁆𑀕𑀸𑀤𑀺𑀤𑀲𑁆𑀲𑀦

(179.)

780\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀓𑀸𑀭𑀁, 𑀆𑀯𑀸𑀲𑀺𑀓𑀮𑀺𑀗𑁆𑀕𑀁, 𑀆𑀯𑀸𑀲𑀺𑀓𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀆𑀯𑀸𑀲𑀺𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑀁, 𑀲𑀼𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀫𑀜𑁆𑀘𑀧𑀻𑀞𑀁, 𑀪𑀺𑀲𑀺𑀩𑀺𑀩𑁆𑀩𑁄𑀳𑀦𑀁, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀲𑀽𑀧𑀝𑁆𑀞𑀺𑀢𑀁, 𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀲𑀼𑀲𑀫𑁆𑀫𑀝𑁆𑀞𑀁; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄’𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀦 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀅𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  ‘𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀓𑁄 𑀢𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺—  𑀪𑁂𑀤𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

781\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀓𑀸𑀭𑀁, 𑀆𑀯𑀸𑀲𑀺𑀓𑀮𑀺𑀗𑁆𑀕𑀁, 𑀆𑀯𑀸𑀲𑀺𑀓𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀆𑀯𑀸𑀲𑀺𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑀁, 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀤𑀲𑀤𑁆𑀤𑀁, 𑀲𑀚𑁆𑀛𑀸𑀬𑀲𑀤𑁆𑀤𑀁, 𑀉𑀓𑁆𑀓𑀸𑀲𑀺𑀢𑀲𑀤𑁆𑀤𑀁, 𑀔𑀺𑀧𑀺𑀢𑀲𑀤𑁆𑀤𑀁; 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄’𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀦 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀅𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  ‘𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀓𑁄 𑀢𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺—  𑀪𑁂𑀤𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

782\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀓𑀸𑀭𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀮𑀺𑀗𑁆𑀕𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑀁, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀧𑀢𑁆𑀢𑀁, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀘𑀻𑀯𑀭𑀁, 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀦𑀁, 𑀧𑀸𑀤𑀸𑀦𑀁 𑀥𑁄𑀢𑀁, 𑀉𑀤𑀓𑀦𑀺𑀲𑁆𑀲𑁂𑀓𑀁; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄’𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀦 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀅𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  ‘𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀓𑁄 𑀢𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺—  𑀪𑁂𑀤𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇

783\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀓𑀸𑀭𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀮𑀺𑀗𑁆𑀕𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀼𑀤𑁆𑀤𑁂𑀲𑀁, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀤𑀲𑀤𑁆𑀤𑀁, 𑀉𑀧𑀸𑀳𑀦𑀧𑀧𑁆𑀨𑁄𑀝𑀦𑀲𑀤𑁆𑀤𑀁, 𑀉𑀓𑁆𑀓𑀸𑀲𑀺𑀢𑀲𑀤𑁆𑀤𑀁, 𑀔𑀺𑀧𑀺𑀢𑀲𑀤𑁆𑀤𑀁; 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀢𑁆𑀣𑀺 𑀦𑀼 𑀔𑁄’𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀦 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀅𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀅𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀯𑁂𑀫𑀢𑀺𑀓𑀸 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀺; 𑀯𑀺𑀘𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  ‘𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀯𑀺𑀦𑀲𑁆𑀲𑀦𑁆𑀢𑁂𑀢𑁂, 𑀓𑁄 𑀢𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄’𑀢𑀺—  𑀪𑁂𑀤𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀸 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇
