# 𑁨𑁇𑁯 𑀕𑀸𑀫𑀲𑀻𑀫𑀸𑀤𑀺

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀕𑀸𑀫𑀲𑀻𑀫𑀸𑀤𑀺

(147.)

619. “𑀅𑀲𑀫𑁆𑀫𑀢𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸𑀬 𑀅𑀝𑁆𑀞𑀧𑀺𑀢𑀸𑀬, 𑀬𑀁 𑀕𑀸𑀫𑀁 𑀯𑀸 𑀦𑀺𑀕𑀫𑀁 𑀯𑀸 𑀉𑀧𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀬𑀸 𑀢𑀲𑁆𑀲 𑀯𑀸 𑀕𑀸𑀫𑀲𑁆𑀲 𑀕𑀸𑀫𑀲𑀻𑀫𑀸, 𑀦𑀺𑀕𑀫𑀲𑁆𑀲 𑀯𑀸 𑀦𑀺𑀕𑀫𑀲𑀻𑀫𑀸, 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸𑁇 𑀅𑀕𑀸𑀫𑀓𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀜𑁆𑀜𑁂 𑀲𑀫𑀦𑁆𑀢𑀸 𑀲𑀢𑁆𑀢𑀩𑁆𑀪𑀦𑁆𑀢𑀭𑀸, 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸𑁇 𑀲𑀩𑁆𑀩𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀤𑀻 𑀅𑀲𑀻𑀫𑀸; 𑀲𑀩𑁆𑀩𑁄 𑀲𑀫𑀼𑀤𑁆𑀤𑁄 𑀅𑀲𑀻𑀫𑁄; 𑀲𑀩𑁆𑀩𑁄 𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁄 𑀅𑀲𑀻𑀫𑁄𑁇 𑀦𑀤𑀺𑀬𑀸 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀯𑀸 𑀚𑀸𑀢𑀲𑁆𑀲𑀭𑁂 𑀯𑀸 𑀬𑀁 𑀫𑀚𑁆𑀛𑀺𑀫𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀲𑀫𑀦𑁆𑀢𑀸 𑀉𑀤𑀓𑀼𑀓𑁆𑀔𑁂𑀧𑀸, 𑀅𑀬𑀁 𑀢𑀢𑁆𑀣 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀸 𑀏𑀓𑀼𑀧𑁄𑀲𑀣𑀸”𑀢𑀺𑁇

(148.)

620. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀫𑀸𑀬 𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀬𑁂𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸 𑀧𑀞𑀫𑀁 𑀲𑀫𑁆𑀫𑀢𑀸 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀅𑀓𑀼𑀧𑁆𑀧𑀁 𑀞𑀸𑀦𑀸𑀭𑀳𑀁𑁇 𑀬𑁂𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸 𑀧𑀘𑁆𑀙𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀸𑀭𑀳𑀁𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸𑀬 𑀲𑀻𑀫𑀸 𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

621. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀫𑀸𑀬 𑀲𑀻𑀫𑀁 𑀅𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀬𑁂𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸 𑀧𑀞𑀫𑀁 𑀲𑀫𑁆𑀫𑀢𑀸 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀅𑀓𑀼𑀧𑁆𑀧𑀁 𑀞𑀸𑀦𑀸𑀭𑀳𑀁𑁇 𑀬𑁂𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸 𑀧𑀘𑁆𑀙𑀸 𑀲𑀫𑁆𑀫𑀢𑀸 𑀢𑁂𑀲𑀁 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀁 𑀓𑀼𑀧𑁆𑀧𑀁 𑀅𑀝𑁆𑀞𑀸𑀦𑀸𑀭𑀳𑀁𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀸𑀬 𑀲𑀻𑀫𑀸 𑀅𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀅𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀦𑁆𑀢𑁂𑀦 𑀲𑀻𑀫𑀦𑁆𑀢𑀭𑀺𑀓𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀻𑀫𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
