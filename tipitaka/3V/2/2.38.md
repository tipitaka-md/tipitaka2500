# 𑁨𑁇𑁩𑁮 𑀯𑀚𑁆𑀚𑀦𑀻𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀦

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀯𑀚𑁆𑀚𑀦𑀻𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀦

(183.)

797\. 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬…𑀧𑁂…  𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 …𑀧𑁂…  𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸…𑀧𑁂…  𑀦 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀅𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

798\. 𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

799\. 𑀦 𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

800\. 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀤𑀸𑀦𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑁄𑀲𑀣𑁂 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸”𑀢𑀺𑁇

---

801\. 𑀢𑀢𑀺𑀬𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇

  
𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀤𑀼𑀢𑀺𑀬𑁄𑁇



---

802\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



803\. _𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀘,_  
_𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀼𑀁 𑀢𑀼𑀡𑁆𑀳𑀺𑀓𑀸;_  
_𑀥𑀫𑁆𑀫𑀁 𑀭𑀳𑁄 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁,_  
_𑀤𑁂𑀯𑀲𑀺𑀓𑀁 𑀢𑀤𑀸 𑀲𑀓𑀺𑀁𑁇_  


804\. _𑀬𑀣𑀸𑀧𑀭𑀺𑀲𑀸 𑀲𑀫𑀕𑁆𑀕𑀁,_  
_𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀫𑀤𑁆𑀤𑀓𑀼𑀘𑁆𑀙𑀺 𑀘;_  
_𑀲𑀻𑀫𑀸 𑀫𑀳𑀢𑀻 𑀦𑀤𑀺𑀬𑀸,_  
_𑀅𑀦𑀼 𑀤𑁆𑀯𑁂 𑀔𑀼𑀤𑁆𑀤𑀓𑀸𑀦𑀺 𑀘𑁇_  


805\. _𑀦𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀘𑁂𑀯,_  
_𑀲𑀻𑀫𑀸 𑀅𑀯𑀺𑀧𑁆𑀧𑀯𑀸𑀲𑀦𑀸;_  
_𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂 𑀧𑀞𑀫𑀁 𑀲𑀻𑀫𑀁,_  
_𑀧𑀘𑁆𑀙𑀸 𑀲𑀻𑀫𑀁 𑀲𑀫𑀽𑀳𑀦𑁂𑁇_  


806\. _𑀅𑀲𑀫𑁆𑀫𑀢𑀸 𑀕𑀸𑀫𑀲𑀻𑀫𑀸,_  
_𑀦𑀤𑀺𑀬𑀸 𑀲𑀫𑀼𑀤𑁆𑀤𑁂 𑀲𑀭𑁂;_  
_𑀉𑀤𑀓𑀼𑀓𑁆𑀔𑁂𑀧𑁄 𑀪𑀺𑀦𑁆𑀤𑀦𑁆𑀢𑀺,_  
_𑀢𑀣𑁂𑀯𑀚𑁆𑀛𑁄𑀢𑁆𑀣𑀭𑀦𑁆𑀢𑀺 𑀘𑁇_  


807\. _𑀓𑀢𑀺 𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑁄,_  
_𑀲𑀯𑀭𑀸 𑀅𑀲𑀢𑀻𑀧𑀺 𑀘;_  
_𑀥𑀫𑁆𑀫𑀁 𑀯𑀺𑀦𑀬𑀁 𑀢𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑀺,_  
_𑀧𑀼𑀦 𑀯𑀺𑀦𑀬𑀢𑀚𑁆𑀚𑀦𑀸𑁇_  


808\. _𑀘𑁄𑀤𑀦𑀸 𑀓𑀢𑁂 𑀑𑀓𑀸𑀲𑁂,_  
_𑀅𑀥𑀫𑁆𑀫𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀓𑁄𑀲𑀦𑀸;_  
_𑀘𑀢𑀼𑀧𑀜𑁆𑀘𑀧𑀭𑀸 𑀆𑀯𑀺,_  
_𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀘𑁂𑀧𑀺 𑀯𑀸𑀬𑀫𑁂𑁇_  


809\. _𑀲𑀕𑀳𑀝𑁆𑀞𑀸 𑀅𑀦𑀚𑁆𑀛𑀺𑀝𑁆𑀞𑀸,_  
_𑀘𑁄𑀤𑀦𑀫𑁆𑀳𑀺 𑀦 𑀚𑀸𑀦𑀢𑀺;_  
_𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀲𑀚𑁆𑀚𑀼𑀓𑀁 𑀦 𑀘 𑀕𑀘𑁆𑀙𑀭𑁂𑁇_  


810\. _𑀓𑀢𑀺𑀫𑀻 𑀓𑀻𑀯𑀢𑀺𑀓𑀸 𑀤𑀽𑀭𑁂,_  
_𑀆𑀭𑁄𑀘𑁂𑀢𑀼𑀜𑁆𑀘 𑀦𑀲𑁆𑀲𑀭𑀺;_  
_𑀉𑀓𑁆𑀮𑀸𑀧𑀁 𑀆𑀲𑀦𑀁 𑀤𑀻𑀧𑁄,_  
_𑀤𑀺𑀲𑀸 𑀅𑀜𑁆𑀜𑁄 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄𑁇_  


811\. _𑀲𑀚𑁆𑀚𑀼𑀓𑀁 𑀯𑀲𑁆𑀲𑀼𑀧𑁄𑀲𑀣𑁄,_  
_𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘 𑀜𑀸𑀢𑀓𑀸;_  
_𑀕𑀕𑁆𑀕𑁄 𑀘𑀢𑀼𑀢𑀬𑁄 𑀤𑁆𑀯𑁂𑀓𑁄,_  
_𑀆𑀧𑀢𑁆𑀢𑀺𑀲𑀪𑀸𑀕𑀸 𑀲𑀭𑀺𑁇_  


812\. _𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀯𑁂𑀫𑀢𑀺𑀓𑁄,_  
_𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄;_  
_𑀩𑀳𑀽 𑀲𑀫𑀲𑀫𑀸 𑀣𑁄𑀓𑀸,_  
_𑀧𑀭𑀺𑀲𑀸 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀘𑁇_  


813\. _𑀏𑀓𑀘𑁆𑀘𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸 𑀲𑀩𑁆𑀩𑀸,_  
_𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀘 𑀯𑁂𑀫𑀢𑀺𑀓𑀸;_  
_𑀓𑀧𑁆𑀧𑀢𑁂𑀯𑀸𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸,_  
_𑀚𑀸𑀦𑀁 𑀧𑀲𑁆𑀲𑀁 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 𑀘𑁇_  


814\. _𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀦 𑀆𑀕𑀦𑁆𑀢𑀼,_  
_𑀘𑀸𑀢𑀼𑀧𑀦𑁆𑀦𑀭𑀲𑁄 𑀧𑀼𑀦;_  
_𑀧𑀸𑀝𑀺𑀧𑀤𑁄 𑀧𑀦𑁆𑀦𑀭𑀲𑁄,_  
_𑀮𑀺𑀗𑁆𑀕𑀲𑀁𑀯𑀸𑀲𑀓𑀸 𑀉𑀪𑁄𑁇_  


815\. _𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀸𑀦𑀼𑀧𑁄𑀲𑀣𑁄,_  
_𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸;_  
_𑀏𑀢𑁂 𑀯𑀺𑀪𑀢𑁆𑀢𑀸 𑀉𑀤𑁆𑀤𑀸𑀦𑀸,_  
_𑀯𑀢𑁆𑀣𑀼𑀯𑀺𑀪𑀽𑀢𑀓𑀸𑀭𑀡𑀸𑀢𑀺𑁇_  


---
