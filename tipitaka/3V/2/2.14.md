# 𑁨𑁇𑁧𑁪 𑀘𑁄𑀤𑀦𑀸𑀓𑀣𑀸

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀘𑁄𑀤𑀦𑀸𑀓𑀣𑀸

(153.)

639. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁄𑀓𑀸𑀲𑀓𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁄𑀓𑀸𑀲𑀓𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀘𑁄𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀼𑀁—  𑀓𑀭𑁄𑀢𑀼 𑀆𑀬𑀲𑁆𑀫𑀸 𑀑𑀓𑀸𑀲𑀁, 𑀅𑀳𑀁 𑀢𑀁 𑀯𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄”𑀢𑀺𑁇

640. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑁂𑀲𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀮𑀪𑀦𑁆𑀢𑀺 𑀆𑀖𑀸𑀢𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁, 𑀯𑀥𑁂𑀦 𑀢𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑁂𑀧𑀺 𑀑𑀓𑀸𑀲𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀢𑀼𑀮𑀬𑀺𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

641. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀧𑀼𑀭𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑁂𑀲𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀁 𑀅𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀭𑀡𑁂 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀁 𑀅𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀭𑀡𑁂 𑀑𑀓𑀸𑀲𑁄 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀓𑀸𑀭𑀸𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀢𑀼𑀮𑀬𑀺𑀢𑁆𑀯𑀸 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
