# 𑁨𑁇𑁨𑁬 𑀆𑀧𑀢𑁆𑀢𑀺𑀆𑀯𑀺𑀓𑀭𑀡𑀯𑀺𑀥𑀺

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀆𑀧𑀢𑁆𑀢𑀺𑀆𑀯𑀺𑀓𑀭𑀡𑀯𑀺𑀥𑀺

(170.)

714\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

715\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀲𑀭𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀸𑀫𑀦𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀇𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀲𑁄𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁄𑀲𑀣𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

716\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲𑀫𑀸𑀦𑁂 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀸𑀫𑀦𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑁄𑁇 𑀬𑀤𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀫𑀢𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀤𑀸 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀲𑁄𑀢𑀩𑁆𑀩𑀁; 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁄𑀲𑀣𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
