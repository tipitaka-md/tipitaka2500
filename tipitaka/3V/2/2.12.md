# 𑁨𑁇𑁧𑁨 𑀯𑀺𑀦𑀬𑀧𑀼𑀘𑁆𑀙𑀦𑀓𑀣𑀸

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀯𑀺𑀦𑀬𑀧𑀼𑀘𑁆𑀙𑀦𑀓𑀣𑀸

(151.)

629\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀅𑀲𑀫𑁆𑀫𑀢𑀸 𑀯𑀺𑀦𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀅𑀲𑀫𑁆𑀫𑀢𑁂𑀦 𑀯𑀺𑀦𑀬𑁄 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑁆𑀫𑀢𑁂𑀦 𑀯𑀺𑀦𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄—  𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁, 𑀧𑀭𑁂𑀦 𑀯𑀸 𑀧𑀭𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀣𑀜𑁆𑀘 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁? 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

630\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀅𑀳𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀯𑀺𑀦𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇

631\. 𑀓𑀣𑀜𑁆𑀘 𑀧𑀭𑁂𑀦 𑀧𑀭𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄? 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

632\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀯𑀺𑀦𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀭𑁂𑀦 𑀧𑀭𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

633\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀧𑁂𑀲𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑁆𑀫𑀢𑀸 𑀯𑀺𑀦𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀮𑀪𑀦𑁆𑀢𑀺 𑀆𑀖𑀸𑀢𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀁, 𑀯𑀥𑁂𑀦 𑀢𑀚𑁆𑀚𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀲𑀫𑁆𑀫𑀢𑁂𑀦𑀧𑀺 𑀧𑀭𑀺𑀲𑀁 𑀑𑀮𑁄𑀓𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀢𑀼𑀮𑀬𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀦𑀬𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
