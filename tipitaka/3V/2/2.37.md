# 𑁨𑁇𑁩𑁭 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀯𑀸𑀭

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀯𑀸𑀭

(182.)

794\. 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀆𑀯𑀸𑀲𑁄, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀸, 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸—  ‘𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀅𑀚𑁆𑀚𑁂𑀯 𑀕𑀦𑁆𑀢𑀼𑀦𑁆’𑀢𑀺𑁇 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀅𑀦𑀸𑀯𑀸𑀲𑁄…𑀧𑁂…  𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀆𑀯𑀸𑀲𑁄 𑀯𑀸 𑀅𑀦𑀸𑀯𑀸𑀲𑁄 𑀯𑀸, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀸, 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸—  ‘𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀅𑀚𑁆𑀚𑁂𑀯 𑀕𑀦𑁆𑀢𑀼𑀦𑁆’𑀢𑀺𑁇

795\. 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀸 𑀅𑀦𑀸𑀯𑀸𑀲𑀸 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀆𑀯𑀸𑀲𑁄…𑀧𑁂…  𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀅𑀦𑀸𑀯𑀸𑀲𑁄…𑀧𑁂…  𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀆𑀯𑀸𑀲𑁄 𑀯𑀸 𑀅𑀦𑀸𑀯𑀸𑀲𑁄 𑀯𑀸, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀸, 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸—  ‘𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀅𑀚𑁆𑀚𑁂𑀯 𑀕𑀦𑁆𑀢𑀼𑀦𑁆’𑀢𑀺𑁇

796\. 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑀸 𑀆𑀯𑀸𑀲𑀸 𑀯𑀸 𑀅𑀦𑀸𑀯𑀸𑀲𑀸 𑀯𑀸 𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀆𑀯𑀸𑀲𑁄…𑀧𑁂…  𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀅𑀦𑀸𑀯𑀸𑀲𑁄…𑀧𑁂…  𑀲𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀆𑀯𑀸𑀲𑁄 𑀯𑀸 𑀅𑀦𑀸𑀯𑀸𑀲𑁄 𑀯𑀸, 𑀬𑀢𑁆𑀣𑀲𑁆𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀸, 𑀬𑀁 𑀚𑀜𑁆𑀜𑀸—  ‘𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀅𑀚𑁆𑀚𑁂𑀯 𑀕𑀦𑁆𑀢𑀼𑀦𑁆’𑀢𑀺𑁇
