# 𑁨𑁇𑁬 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(142.)

602. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀅𑀢𑀺𑀔𑀼𑀤𑁆𑀤𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀫𑀳𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀲𑀫𑁆𑀫𑀢𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀉𑀧𑁄𑀲𑀣𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄’𑀢𑀺, 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳𑀸 𑀅𑀲𑀫𑁆𑀫𑀢𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀅𑀲𑁆𑀲𑀼𑀫𑁆𑀳𑀸, 𑀓𑀢𑁄 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀧𑁄𑀲𑀣𑁄, 𑀅𑀓𑀢𑁄 𑀦𑀼 𑀔𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀲𑀫𑁆𑀫𑀢𑀸𑀬 𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀽𑀫𑀺𑀬𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀅𑀲𑀫𑁆𑀫𑀢𑀸𑀬 𑀯𑀸 𑀬𑀢𑁄 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀲𑀼𑀡𑀸𑀢𑀺, 𑀓𑀢𑁄𑀯𑀲𑁆𑀲 𑀉𑀧𑁄𑀲𑀣𑁄𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀬𑀸𑀯 𑀫𑀳𑀦𑁆𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀆𑀓𑀗𑁆𑀔𑀢𑀺, 𑀢𑀸𑀯 𑀫𑀳𑀦𑁆𑀢𑀁 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀞𑀫𑀁 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂 𑀓𑀺𑀢𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

603. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸𑀯𑀢𑀸 𑀲𑀫𑀦𑁆𑀢𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

604. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀸𑀯𑀢𑀸 𑀲𑀫𑀦𑁆𑀢𑀸 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑀸 𑀓𑀺𑀢𑁆𑀢𑀺𑀢𑀸𑁇 𑀲𑀁𑀖𑁄 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇 𑀲𑀫𑁆𑀫𑀢𑀁 𑀲𑀁𑀖𑁂𑀦 𑀏𑀢𑁂𑀳𑀺 𑀦𑀺𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀉𑀧𑁄𑀲𑀣𑀧𑁆𑀧𑀫𑀼𑀔𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

605. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀦𑀯𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀞𑀫𑀢𑀭𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸—  “𑀦 𑀢𑀸𑀯 𑀣𑁂𑀭𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀉𑀧𑁄𑀲𑀣𑁄 𑀯𑀺𑀓𑀸𑀮𑁂 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀣𑁂𑀭𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀞𑀫𑀢𑀭𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

606. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀫𑀸𑀦𑀲𑀻𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀯𑀸𑀲𑁂 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀭𑀻𑀬𑀢𑀼, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀯𑀸𑀲𑁂 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀭𑀻𑀬𑀢𑀽”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀫𑀸𑀦𑀲𑀻𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀯𑀤𑀦𑁆𑀢𑀺—  ‘𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀯𑀸𑀲𑁂 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀭𑀻𑀬𑀢𑀼, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀯𑀸𑀲𑁂 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀭𑀻𑀬𑀢𑀽’𑀢𑀺𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀯𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
