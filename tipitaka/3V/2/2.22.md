# 𑁨𑁇𑁨𑁨 𑀜𑀸𑀢𑀓𑀸𑀤𑀺𑀕𑁆𑀕𑀳𑀡𑀓𑀣𑀸

* 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀜𑀸𑀢𑀓𑀸𑀤𑀺𑀕𑁆𑀕𑀳𑀡𑀓𑀣𑀸

(166.)

692\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀜𑀸𑀢𑀓𑀸 𑀕𑀡𑁆𑀳𑀺𑀁𑀲𑀼𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

693\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀜𑀸𑀢𑀓𑀸 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀜𑀸𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀇𑀗𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀫𑀼𑀜𑁆𑀘𑀣, 𑀬𑀸𑀯𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀢𑁂 𑀜𑀸𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀇𑀗𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀳𑁄𑀣, 𑀬𑀸𑀯𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀤𑁂𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀢𑁂 𑀜𑀸𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀇𑀗𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀦𑁂𑀣, 𑀬𑀸𑀯 𑀲𑀁𑀖𑁄 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀯𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀭𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

694\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀭𑀸𑀚𑀸𑀦𑁄 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀘𑁄𑀭𑀸 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺…  𑀥𑀼𑀢𑁆𑀢𑀸 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺…  𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀇𑀗𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀫𑀼𑀜𑁆𑀘𑀣, 𑀬𑀸𑀯𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀇𑀗𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀳𑁄𑀣, 𑀬𑀸𑀯𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀁 𑀤𑁂𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀇𑀗𑁆𑀖 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀦𑁂𑀣, 𑀬𑀸𑀯 𑀲𑀁𑀖𑁄 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁄𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀯𑀕𑁆𑀕𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀓𑀭𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
