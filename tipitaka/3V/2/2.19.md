# 𑁨𑁇𑁧𑁯 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑀸𑀤𑀺𑀯𑀢𑁆𑀣𑀼

### 𑀉𑀧𑁄𑀲𑀣𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑀸𑀤𑀺𑀯𑀢𑁆𑀣𑀼

(163.)

673. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑀸 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

674. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑀸 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂 𑀆𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀳𑀺 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸—  ‘𑀓𑀳𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀣, 𑀓𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀣𑀸’𑀢𑀺? 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑁂 𑀩𑀸𑀮𑁂 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁂 𑀅𑀧𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀳𑀺 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀳𑀺 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

675. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑁇 𑀢𑁂 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀯𑀸, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑁄 𑀆𑀕𑀢𑀸𑀕𑀫𑁄 𑀥𑀫𑁆𑀫𑀥𑀭𑁄 𑀯𑀺𑀦𑀬𑀥𑀭𑁄 𑀫𑀸𑀢𑀺𑀓𑀸𑀥𑀭𑁄 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀩𑁆𑀬𑀢𑁆𑀢𑁄 𑀫𑁂𑀥𑀸𑀯𑀻 𑀮𑀚𑁆𑀚𑀻 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀓𑁄 𑀲𑀺𑀓𑁆𑀔𑀸𑀓𑀸𑀫𑁄𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀗𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄 𑀅𑀦𑀼𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄 𑀉𑀧𑀮𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄 𑀘𑀼𑀡𑁆𑀡𑁂𑀦 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑁂𑀦 𑀫𑀼𑀔𑁄𑀤𑀓𑁂𑀦𑁇 𑀦𑁄 𑀘𑁂 𑀲𑀗𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀼𑀁 𑀅𑀦𑀼𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀼𑀁 𑀉𑀧𑀮𑀸𑀧𑁂𑀬𑁆𑀬𑀼𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬𑀼𑀁 𑀘𑀼𑀡𑁆𑀡𑁂𑀦 𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑁂𑀦 𑀫𑀼𑀔𑁄𑀤𑀓𑁂𑀦, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

676. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼𑀧𑁄𑀲𑀣𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑁇 𑀢𑁂 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀯𑀸, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀚𑁆𑀚𑀼𑀓𑀁 𑀧𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀬𑀢𑁆𑀣 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀯𑀸, 𑀲𑁄 𑀆𑀯𑀸𑀲𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑁄𑁇 𑀦𑁄 𑀘𑁂 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀁, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

677. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀺 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑁇 𑀢𑁂 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀯𑀸 𑀉𑀧𑁄𑀲𑀣𑀓𑀫𑁆𑀫𑀁 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀼𑀤𑁆𑀤𑁂𑀲𑀁 𑀯𑀸𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀆𑀯𑀸𑀲𑀸 𑀲𑀚𑁆𑀚𑀼𑀓𑀁 𑀧𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁 𑀧𑀸𑀳𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀯𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀯𑀸 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀸’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
