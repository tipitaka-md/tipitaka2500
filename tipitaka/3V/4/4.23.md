# 𑁪𑁇𑁨𑁩 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀤𑀺

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀯𑀢𑁆𑀣𑀼𑀓𑀸𑀤𑀺

(238.)

1098\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄, 𑀢𑁂𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄, 𑀲𑀸𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀢𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

1099\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺…  𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀓𑁆𑀓𑀝𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺…  𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄, 𑀢𑁂𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄, 𑀲𑀸𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀢𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

1100\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…  𑀤𑀼𑀓𑁆𑀓𑀝𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄, 𑀢𑁂𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄, 𑀲𑀸𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀢𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

1101\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺…  𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺…  𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀓𑁆𑀓𑀝𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀤𑀺𑀝𑁆𑀞𑀺𑀦𑁄, 𑀢𑁂𑀳𑀺 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀧𑀦𑁂𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀲𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄, 𑀲𑀸𑀲𑁆𑀲 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀓𑀢𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇
