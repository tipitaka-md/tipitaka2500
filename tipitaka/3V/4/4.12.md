# 𑁪𑁇𑁧𑁨 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀧𑀦𑁆𑀦𑀭𑀲𑀓

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀧𑀦𑁆𑀦𑀭𑀲𑀓

(225.)

1026. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑁂𑀯 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀦𑀸𑀫𑁆𑀳𑀸𑀓𑀁 𑀦 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀸 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀢𑀭𑀸𑁇 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀼𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁧)

1027. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑁂𑀯 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀦𑀸𑀫𑁆𑀳𑀸𑀓𑀁 𑀦 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀸 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀲𑀫𑀲𑀫𑀸𑁇 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑀸, 𑀅𑀯𑀲𑁂𑀲𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁨)

1028. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑁂𑀯 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀦𑀸𑀫𑁆𑀳𑀸𑀓𑀁 𑀦 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀸 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀣𑁄𑀓𑀢𑀭𑀸𑁇 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑀸, 𑀅𑀯𑀲𑁂𑀲𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁩)

1029. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀢𑁆𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀸𑀕𑀢𑀸’𑀢𑀺𑁇 𑀢𑁂 ‘𑀓𑀧𑁆𑀧𑀢𑁂𑀯 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀦𑀸𑀫𑁆𑀳𑀸𑀓𑀁 𑀦 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀸 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀢𑀫𑀢𑁆𑀢𑁂…𑀧𑁂…  𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀏𑀓𑀘𑁆𑀘𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀸𑀬 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬, 𑀅𑀣𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀩𑀳𑀼𑀢𑀭𑀸…𑀧𑁂…  𑀲𑀫𑀲𑀫𑀸…𑀧𑁂…  𑀣𑁄𑀓𑀢𑀭𑀸𑁇 𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀲𑀼𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑀸, 𑀢𑁂𑀲𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀁 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 (𑁪--𑁧𑁫)

---

1030. 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀧𑀓𑀢𑀧𑀦𑁆𑀦𑀭𑀲𑀓𑀁 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
