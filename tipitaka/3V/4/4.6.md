# 𑁪𑁇𑁬 𑀆𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀯𑀺𑀥𑀺

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀧𑀢𑁆𑀢𑀺𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑀯𑀺𑀥𑀺

(219.)

987\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

988\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀆𑀧𑀦𑁆𑀦𑁄, 𑀢𑀁 𑀧𑀝𑀺𑀤𑁂𑀲𑁂𑀫𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—  ‘𑀧𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇 ‘𑀆𑀫 𑀧𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 ‘𑀆𑀬𑀢𑀺𑀁 𑀲𑀁𑀯𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀻’𑀢𑀺𑁇

989\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀸𑀬 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀫𑀢𑀺𑀓𑁄; 𑀬𑀤𑀸 𑀦𑀺𑀩𑁆𑀩𑁂𑀫𑀢𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀢𑀤𑀸 𑀢𑀁 𑀆𑀧𑀢𑁆𑀢𑀺𑀁 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀯𑀢𑁆𑀯𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁; 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
