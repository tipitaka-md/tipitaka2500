# 𑁪𑁇𑁨𑁨 𑀧𑀯𑀸𑀭𑀡𑀸𑀞𑀧𑀦

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀞𑀧𑀦

(235.)

1088. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁄 𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄 𑀧𑀯𑀸𑀭𑁂𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀘𑁄𑀤𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1089. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀬𑀫𑀸𑀦𑀸 𑀦 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀑𑀓𑀸𑀲𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀓𑀸𑀲𑀁 𑀅𑀓𑀭𑁄𑀦𑁆𑀢𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁂 𑀯𑀸 𑀧𑀦𑁆𑀦𑀭𑀲𑁂 𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁂 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑁂 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀤𑀸𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀲𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑁄𑁇 𑀢𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀫𑀺𑁇 𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸”𑀢𑀺𑁇

1090. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀧𑀼𑀭𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑁂𑀲𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀁 𑀅𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀭𑀡𑁂 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀦𑁆𑀢𑀺, 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀦𑀁 𑀅𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀓𑀸𑀭𑀡𑁂 𑀧𑀯𑀸𑀭𑀡𑀸 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀞𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀺𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀧𑀯𑀸𑀭𑀡𑀸 𑀞𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀞𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

(236.)

1091. 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸, 𑀏𑀯𑀁 𑀅𑀝𑁆𑀞𑀧𑀺𑀢𑀸𑁇 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸? 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀸𑀬 𑀮𑀧𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀅𑀝𑁆𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀸𑀬 𑀮𑀧𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀅𑀝𑁆𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀝𑁆𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑁇

1092. 𑀓𑀣𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸? 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀸𑀬 𑀮𑀧𑀺𑀢𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…  𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀸𑀬 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀪𑀸𑀲𑀺𑀢𑀸𑀬 𑀮𑀧𑀺𑀢𑀸𑀬 𑀅𑀧𑀭𑀺𑀬𑁄𑀲𑀺𑀢𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀞𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑁇

1093. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺𑁇 𑀢𑀜𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀓𑀸𑀬𑀲𑀫𑀸𑀘𑀸𑀭𑁄, 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀯𑀘𑀻𑀲𑀫𑀸𑀘𑀸𑀭𑁄, 𑀅𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀚𑀻𑀯𑁄, 𑀩𑀸𑀮𑁄, 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁄, 𑀦 𑀧𑀝𑀺𑀩𑀮𑁄 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀻𑀬𑀫𑀸𑀦𑁄 𑀅𑀦𑀼𑀬𑁄𑀕𑀁 𑀤𑀸𑀢𑀼𑀦𑁆’𑀢𑀺, ‘𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁, 𑀫𑀸 𑀓𑀮𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁, 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆’𑀢𑀺 𑀑𑀫𑀤𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇

𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀞𑀧𑁂𑀫𑀺, 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀞𑀧𑁂𑀫𑀺, 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑀸 𑀞𑀧𑁂𑀫𑀻’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀚𑀸𑀦𑀸𑀲𑀺 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀚𑀸𑀦𑀸𑀲𑀺 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀚𑀸𑀦𑀸𑀲𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀦𑁆’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀚𑀸𑀦𑀸𑀫𑀺 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀚𑀸𑀦𑀸𑀫𑀺 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀁, 𑀚𑀸𑀦𑀸𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀺𑀦𑁆’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀓𑀢𑀫𑀸 𑀧𑀦𑀸𑀯𑀼𑀲𑁄, 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺, 𑀓𑀢𑀫𑀸 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺, 𑀓𑀢𑀫𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀸𑀦𑀺, 𑀢𑁂𑀭𑀲 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀸, 𑀅𑀬𑀁 𑀲𑀻𑀮𑀯𑀺𑀧𑀢𑁆𑀢𑀺; 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁, 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁, 𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁, 𑀤𑀼𑀓𑁆𑀓𑀝𑀁, 𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁, 𑀅𑀬𑀁 𑀆𑀘𑀸𑀭𑀯𑀺𑀧𑀢𑁆𑀢𑀺; 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀦𑁆𑀢𑀕𑁆𑀕𑀸𑀳𑀺𑀓𑀸𑀤𑀺𑀝𑁆𑀞𑀺, 𑀅𑀬𑀁 𑀤𑀺𑀝𑁆𑀞𑀺𑀯𑀺𑀧𑀢𑁆𑀢𑀻’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀲𑀺, 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀞𑀧𑁂𑀲𑀺, 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀞𑀧𑁂𑀲𑀺, 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸 𑀞𑀧𑁂𑀲𑀻’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀞𑀧𑁂𑀫𑀺, 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀞𑀧𑁂𑀫𑀺, 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸 𑀞𑀧𑁂𑀫𑀻’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀲𑀺, 𑀓𑀺𑀁 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀓𑀤𑀸 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀓𑀢𑁆𑀣 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀁, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁…  𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁…  𑀤𑀼𑀓𑁆𑀓𑀝𑀁…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑁄 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀓𑀢𑁆𑀣 𑀘 𑀢𑁆𑀯𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀓𑀢𑁆𑀣 𑀘𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀳𑁄𑀲𑀺, 𑀓𑀺𑀜𑁆𑀘 𑀢𑁆𑀯𑀁 𑀓𑀭𑁄𑀲𑀺, 𑀓𑀺𑀜𑁆𑀘𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀭𑁄𑀢𑀻’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀫𑀺, 𑀅𑀧𑀺 𑀘 𑀲𑀼𑀢𑁂𑀦 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀫𑀻’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀼𑀢𑁂𑀦 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀲𑀺, 𑀓𑀺𑀁 𑀢𑁂 𑀲𑀼𑀢𑀁, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀢𑁂 𑀲𑀼𑀢𑀁, 𑀓𑀤𑀸 𑀢𑁂 𑀲𑀼𑀢𑀁, 𑀓𑀢𑁆𑀣 𑀢𑁂 𑀲𑀼𑀢𑀁, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀲𑀼𑀢𑀁, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀲𑀼𑀢𑀁, 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁…  𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁…  𑀤𑀼𑀓𑁆𑀓𑀝𑀁…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀲𑀼𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀼𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀼𑀢𑀁, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀲𑀼𑀢𑀁, 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀲𑀼𑀢𑀁, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀲𑀼𑀢𑀁, 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀲𑀼𑀢𑀁, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬 𑀲𑀼𑀢𑀁, 𑀭𑀸𑀚𑀽𑀦𑀁 𑀲𑀼𑀢𑀁, 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀢𑀁, 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀲𑀼𑀢𑀁, 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀼𑀢𑀦𑁆’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀼𑀢𑁂𑀦 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀫𑀺, 𑀅𑀧𑀺 𑀘 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀫𑀻’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀬𑀁 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀲𑀺, 𑀓𑀺𑀁 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀓𑀺𑀦𑁆𑀢𑀺 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀓𑀤𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀓𑀢𑁆𑀣 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁…  𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁…  𑀤𑀼𑀓𑁆𑀓𑀝𑀁…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀭𑀸𑀚𑀽𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀭𑀸𑀚𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀺, 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀲𑀻’𑀢𑀺? 𑀲𑁄 𑀘𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀦 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀫𑀺, 𑀅𑀧𑀺 𑀘 𑀅𑀳𑀫𑁆𑀧𑀺 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺 𑀓𑁂𑀦 𑀧𑀦𑀸𑀳𑀁 𑀇𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁄𑀤𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀬𑁄𑀕𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀦 𑀆𑀭𑀸𑀥𑁂𑀢𑀺, 𑀅𑀦𑀦𑀼𑀯𑀸𑀤𑁄 𑀘𑀼𑀤𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁄𑀤𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀬𑁄𑀕𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀽𑀦𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻𑀦𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀆𑀭𑀸𑀥𑁂𑀢𑀺, 𑀲𑀸𑀦𑀼𑀯𑀸𑀤𑁄 𑀘𑀼𑀤𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀢𑀺 𑀅𑀮𑀁𑀯𑀘𑀦𑀸𑀬𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁄𑀤𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑀺𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁄𑀤𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑀺𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑁄𑀤𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀫𑀽𑀮𑀓𑁂𑀦 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑁂𑀦…  𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑁂𑀦…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑁂𑀦…  𑀤𑀼𑀓𑁆𑀓𑀝𑁂𑀦…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑁂𑀦 𑀅𑀦𑀼𑀤𑁆𑀥𑀁𑀲𑀺𑀢𑀁 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀼𑀤𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀭𑀸𑀚𑀺𑀓𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀦𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀼𑀤𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀲𑀁𑀖𑀸𑀤𑀺𑀲𑁂𑀲𑀁 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀼𑀤𑀺𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁…  𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁…  𑀧𑀸𑀝𑀺𑀤𑁂𑀲𑀦𑀻𑀬𑀁…  𑀤𑀼𑀓𑁆𑀓𑀝𑀁…  𑀤𑀼𑀩𑁆𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀸𑀢𑀺, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇
