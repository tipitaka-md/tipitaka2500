# 𑁪𑁇𑁫 𑀲𑀁𑀖𑀧𑀯𑀸𑀭𑀡𑀸𑀤𑀺𑀧𑁆𑀧𑀪𑁂𑀤

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀁𑀖𑀧𑀯𑀸𑀭𑀡𑀸𑀤𑀺𑀧𑁆𑀧𑀪𑁂𑀤

(215.)

973\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳𑀸 𑀧𑀜𑁆𑀘 𑀚𑀦𑀸𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀁𑀖𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(216.)

974\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀁𑀖𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀦𑁆𑀢𑀺𑁇 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀚𑀦𑀸𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸—

975\. ‘𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄𑁇 𑀅𑀚𑁆𑀚 𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀬𑀤𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇

976\. 𑀣𑁂𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇

977\. 𑀦𑀯𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇

978\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀢𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀁𑀖𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳𑀸 𑀢𑀬𑁄 𑀚𑀦𑀸𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸—

979\. ‘𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑁇 𑀅𑀚𑁆𑀚 𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀬𑀤𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇

982\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀁𑀖𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳𑀸 𑀤𑁆𑀯𑁂 𑀚𑀦𑀸𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀣𑁂𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀯𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑁇

983\. 𑀦𑀯𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀣𑁂𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇

(218.)

984\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀁𑀖𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁, 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀏𑀓𑀓𑁄𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

985\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀬𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺 𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸𑀬 𑀯𑀸 𑀫𑀡𑁆𑀟𑀧𑁂 𑀯𑀸 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀯𑀸, 𑀲𑁄 𑀤𑁂𑀲𑁄 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀻𑀧𑀁 𑀓𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁; 𑀦𑁄 𑀘𑁂 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, ‘𑀅𑀚𑁆𑀚 𑀫𑁂 𑀧𑀯𑀸𑀭𑀡𑀸’𑀢𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

986\. 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣 𑀧𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀏𑀓𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀽𑀳𑀺 𑀲𑀁𑀖𑁂 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀏𑀓𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀢𑀻𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣 𑀢𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀏𑀓𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑀻𑀳𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀼𑀜𑁆𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑀢𑁆𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀏𑀓𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑁂𑀦 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀥𑀺𑀝𑁆𑀞𑁂𑀬𑁆𑀬 𑀘𑁂, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
