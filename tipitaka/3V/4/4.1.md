# 𑁪𑁇𑁧 𑀅𑀨𑀸𑀲𑀼𑀓𑀯𑀺𑀳𑀸𑀭

* 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀅𑀨𑀸𑀲𑀼𑀓𑀯𑀺𑀳𑀸𑀭

(209.)

953\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀉𑀧𑀸𑀬𑁂𑀦 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀫𑀬𑀁 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑁂𑀯 𑀆𑀮𑀧𑁂𑀬𑁆𑀬𑀸𑀫 𑀦 𑀲𑀮𑁆𑀮𑀧𑁂𑀬𑁆𑀬𑀸𑀫—  𑀬𑁄 𑀧𑀞𑀫𑀁 𑀕𑀸𑀫𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬 𑀲𑁄 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀣𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬, 𑀅𑀯𑀓𑁆𑀓𑀸𑀭𑀧𑀸𑀢𑀺𑀁 𑀥𑁄𑀯𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬; 𑀬𑁄 𑀧𑀘𑁆𑀙𑀸 𑀕𑀸𑀫𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀲𑀘𑀲𑁆𑀲 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀲𑁂𑀲𑁄, 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀦𑁄 𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑁂𑀬𑁆𑀬 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀬𑁆𑀬, 𑀅𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀑𑀧𑀺𑀮𑀸𑀧𑁂𑀬𑁆𑀬; 𑀲𑁄 𑀆𑀲𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀣𑀮𑀺𑀓𑀁 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀬𑁆𑀬, 𑀅𑀯𑀓𑁆𑀓𑀸𑀭𑀧𑀸𑀢𑀺𑀁 𑀥𑁄𑀯𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀬𑁆𑀬, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀬𑁆𑀬, 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑁂𑀬𑁆𑀬; 𑀬𑁄 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬 𑀧𑀸𑀦𑀻𑀬𑀖𑀝𑀁 𑀯𑀸 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀖𑀝𑀁 𑀯𑀸 𑀯𑀘𑁆𑀘𑀖𑀝𑀁 𑀯𑀸 𑀭𑀺𑀢𑁆𑀢𑀁 𑀢𑀼𑀘𑁆𑀙𑀁 𑀲𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬; 𑀲𑀘𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀲𑀬𑁆𑀳𑀁, 𑀳𑀢𑁆𑀣𑀯𑀺𑀓𑀸𑀭𑁂𑀦 𑀤𑀼𑀢𑀺𑀬𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀳𑀢𑁆𑀣𑀯𑀺𑀮𑀗𑁆𑀖𑀓𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬; 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀸𑀘𑀁 𑀪𑀺𑀦𑁆𑀤𑁂𑀬𑁆𑀬—  𑀏𑀯𑀁 𑀔𑁄 𑀫𑀬𑀁 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇

954\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀦𑁂𑀯 𑀆𑀮𑀧𑀺𑀁𑀲𑀼, 𑀦 𑀲𑀮𑁆𑀮𑀧𑀺𑀁𑀲𑀼𑁇 𑀬𑁄 𑀧𑀞𑀫𑀁 𑀕𑀸𑀫𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑁄 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺, 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀣𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀢𑀺, 𑀅𑀯𑀓𑁆𑀓𑀸𑀭𑀧𑀸𑀢𑀺𑀁 𑀥𑁄𑀯𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺𑁇 𑀬𑁄 𑀧𑀘𑁆𑀙𑀸 𑀕𑀸𑀫𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀲𑀘𑁂 𑀳𑁄𑀢𑀺 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀲𑁂𑀲𑁄, 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀪𑀼𑀜𑁆𑀚𑀢𑀺, 𑀦𑁄 𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀢𑀺, 𑀅𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀑𑀧𑀺𑀮𑀸𑀧𑁂𑀢𑀺; 𑀲𑁄 𑀆𑀲𑀦𑀁 𑀉𑀤𑁆𑀥𑀭𑀢𑀺, 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀣𑀮𑀺𑀓𑀁 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀢𑀺, 𑀅𑀯𑀓𑁆𑀓𑀸𑀭𑀧𑀸𑀢𑀺𑀁 𑀥𑁄𑀯𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀢𑀺, 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀢𑀺, 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀢𑀺𑁇 𑀬𑁄 𑀧𑀲𑁆𑀲𑀢𑀺 𑀧𑀸𑀦𑀻𑀬𑀖𑀝𑀁 𑀯𑀸 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀖𑀝𑀁 𑀯𑀸 𑀯𑀘𑁆𑀘𑀖𑀝𑀁 𑀯𑀸 𑀭𑀺𑀢𑁆𑀢𑀁 𑀢𑀼𑀘𑁆𑀙𑀁 𑀲𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺𑁇 𑀲𑀘𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀅𑀯𑀺𑀲𑀬𑁆𑀳𑀁, 𑀳𑀢𑁆𑀣𑀯𑀺𑀓𑀸𑀭𑁂𑀦 𑀤𑀼𑀢𑀺𑀬𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀢𑁆𑀯𑀸 𑀳𑀢𑁆𑀣𑀯𑀺𑀮𑀗𑁆𑀖𑀓𑁂𑀦 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀢𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀯𑀸𑀘𑀁 𑀪𑀺𑀦𑁆𑀤𑀢𑀺𑁇

955\. 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀢𑁂𑀫𑀸𑀲𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀆𑀘𑀺𑀡𑁆𑀡𑀁 𑀔𑁄 𑀧𑀦𑁂𑀢𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀸𑀦𑀁 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀫𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀢𑁆𑀣, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? “𑀔𑀫𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸, 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸𑁇 𑀲𑀫𑀕𑁆𑀕𑀸 𑀘 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀫𑁆𑀳𑀸, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑀺𑀫𑁆𑀳𑀸”𑀢𑀺𑁇 𑀚𑀸𑀦𑀦𑁆𑀢𑀸𑀧𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀚𑀸𑀦𑀦𑁆𑀢𑀸𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀓𑀸𑀮𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀓𑀸𑀮𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁𑁇 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁂 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀸𑀓𑀸𑀭𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀥𑀫𑁆𑀫𑀁 𑀯𑀸 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫, 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀣𑀸𑀓𑀣𑀁 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀕𑁆𑀕𑀸 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀓𑀁 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑀢𑁆𑀣, 𑀦 𑀘 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺𑁇

957\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀨𑀸𑀲𑀼𑀜𑁆𑀜𑁂𑀯 𑀓𑀺𑀭𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀯𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀧𑀲𑀼𑀲𑀁𑀯𑀸𑀲𑀜𑁆𑀜𑁂𑀯 𑀓𑀺𑀭𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀯𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀏𑀍𑀅𑀓𑀲𑀁𑀯𑀸𑀲𑀜𑁆𑀜𑁂𑀯 𑀓𑀺𑀭𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀯𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀲𑀧𑀢𑁆𑀢𑀲𑀁𑀯𑀸𑀲𑀜𑁆𑀜𑁂𑀯 𑀓𑀺𑀭𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀯𑀼𑀝𑁆𑀞𑀸 𑀲𑀫𑀸𑀦𑀸 𑀨𑀸𑀲𑀼𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺 𑀧𑀝𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫𑀺𑀫𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀫𑀽𑀕𑀩𑁆𑀩𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀫𑀸𑀤𑀸𑀦𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀽𑀕𑀩𑁆𑀩𑀢𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀲𑀫𑀸𑀤𑀸𑀦𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀲𑀫𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀢𑀻𑀳𑀺 𑀞𑀸𑀦𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁—  𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀲𑀸 𑀯𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀸𑀦𑀼𑀮𑁄𑀫𑀢𑀸 𑀆𑀧𑀢𑁆𑀢𑀺𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀢𑀸 𑀯𑀺𑀦𑀬𑀧𑀼𑀭𑁂𑀓𑁆𑀔𑀸𑀭𑀢𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

958\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀚𑁆𑀚 𑀧𑀯𑀸𑀭𑀡𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

959\. 𑀣𑁂𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀲𑀁𑀖𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀁 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀁 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇

960\. 𑀦𑀯𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀲𑀁𑀖𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀧𑀯𑀸𑀭𑁂𑀫𑀺 𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀯𑀸 𑀲𑀼𑀢𑁂𑀦 𑀯𑀸 𑀧𑀭𑀺𑀲𑀗𑁆𑀓𑀸𑀬 𑀯𑀸𑁇 𑀯𑀤𑀦𑁆𑀢𑀼 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’”𑀢𑀺𑁇

(211.)

961\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀧𑀯𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀅𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀧𑀯𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀅𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀧𑀯𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀅𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀣𑁂𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀧𑀯𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀅𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑁂𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀲𑀼 𑀧𑀯𑀸𑀭𑀬𑀫𑀸𑀦𑁂𑀲𑀼 𑀆𑀲𑀦𑁂𑀲𑀼 𑀅𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀅𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂𑀳𑀺 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

962\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀣𑁂𑀭𑁄 𑀚𑀭𑀸𑀤𑀼𑀩𑁆𑀩𑀮𑁄 𑀬𑀸𑀯 𑀲𑀩𑁆𑀩𑁂 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀻𑀢𑀺 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀆𑀕𑀫𑀬𑀫𑀸𑀦𑁄 𑀫𑀼𑀘𑁆𑀙𑀺𑀢𑁄 𑀧𑀧𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀫𑀦𑁆𑀢𑀭𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑀼𑀁 𑀬𑀸𑀯 𑀧𑀯𑀸𑀭𑁂𑀢𑀺, 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
