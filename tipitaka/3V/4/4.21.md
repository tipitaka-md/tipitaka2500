# 𑁪𑁇𑁨𑁧 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀤𑀺𑀧𑀯𑀸𑀭𑀡𑀸

* 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑀤𑀺𑀧𑀯𑀸𑀭𑀡𑀸

(234.)

1075\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀯𑀭𑀪𑀬𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1076\. 𑀩𑀸𑀍𑀳𑀢𑀭𑀁 𑀲𑀯𑀭𑀪𑀬𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1077\. 𑀩𑀸𑀍𑀳𑀢𑀭𑀁 𑀲𑀯𑀭𑀪𑀬𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1078\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀭𑀢𑁆𑀢𑀺 𑀯𑀺𑀪𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

1079\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀭𑀢𑁆𑀢𑀺 𑀯𑀺𑀪𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1080\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀤𑀸𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀭𑀢𑁆𑀢𑀺 𑀯𑀺𑀪𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀁, 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀁, 𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

1081\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀥𑀫𑁆𑀫𑀁 𑀪𑀡𑀦𑁆𑀢𑁂𑀳𑀺…𑀧𑁂…  𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑀓𑁂𑀳𑀺 𑀲𑀼𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀲𑀗𑁆𑀕𑀸𑀬𑀦𑁆𑀢𑁂𑀳𑀺…  𑀯𑀺𑀦𑀬𑀥𑀭𑁂𑀳𑀺 𑀯𑀺𑀦𑀬𑀁 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀦𑀦𑁆𑀢𑁂𑀳𑀺…  𑀥𑀫𑁆𑀫𑀓𑀣𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀸𑀓𑀘𑁆𑀙𑀦𑁆𑀢𑁂𑀳𑀺…  𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀓𑀮𑀳𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀓𑀮𑀳𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀭𑀢𑁆𑀢𑀺 𑀯𑀺𑀪𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1082\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀓𑀮𑀳𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁂𑀳𑀺 𑀬𑁂𑀪𑀼𑀬𑁆𑀬𑁂𑀦 𑀭𑀢𑁆𑀢𑀺 𑀔𑁂𑀧𑀺𑀢𑀸𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀭𑀢𑁆𑀢𑀺 𑀯𑀺𑀪𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀁, 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀁, 𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’”𑀢𑀺𑁇

1083\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀫𑀳𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀳𑀸 𑀘 𑀫𑁂𑀖𑁄 𑀉𑀕𑁆𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀬𑀁 𑀔𑁄 𑀫𑀳𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄, 𑀧𑀭𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀺𑀓𑀁, 𑀫𑀳𑀸 𑀘 𑀫𑁂𑀖𑁄 𑀉𑀕𑁆𑀕𑀢𑁄𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀫𑁂𑀖𑁄 𑀧𑀯𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

1084\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀫𑀳𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀭𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀳𑁄𑀢𑀺, 𑀫𑀳𑀸 𑀘 𑀫𑁂𑀖𑁄 𑀉𑀕𑁆𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀫𑀳𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄, 𑀧𑀭𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀺𑀓𑀁, 𑀫𑀳𑀸 𑀘 𑀫𑁂𑀖𑁄 𑀉𑀕𑁆𑀕𑀢𑁄𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀫𑁂𑀖𑁄 𑀧𑀯𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1085\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀫𑀳𑀸𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁄, 𑀧𑀭𑀺𑀢𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑁄𑀯𑀲𑁆𑀲𑀺𑀓𑀁, 𑀫𑀳𑀸 𑀘 𑀫𑁂𑀖𑁄 𑀉𑀕𑁆𑀕𑀢𑁄𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀫𑁂𑀖𑁄 𑀧𑀯𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀁, 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀁, 𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇

1086\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀭𑀸𑀚𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀘𑁄𑀭𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀅𑀕𑁆𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀉𑀤𑀓𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀫𑀦𑀼𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀯𑀸𑀍𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀲𑀭𑀻𑀲𑀧𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀚𑀻𑀯𑀺𑀢𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺…  𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀬𑀁 𑀔𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1087\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑁇 𑀲𑀘𑁂 𑀲𑀁𑀖𑁄 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀺𑀢𑁄𑀯 𑀲𑀁𑀖𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀣𑀸𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆𑀢𑀭𑀸𑀬𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀁, 𑀏𑀓𑀯𑀸𑀘𑀺𑀓𑀁, 𑀲𑀫𑀸𑀦𑀯𑀲𑁆𑀲𑀺𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’”𑀢𑀺𑁇
