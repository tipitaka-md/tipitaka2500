# 𑁪𑁇𑁧𑁭 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀸𑀤𑀻𑀳𑀺𑀧𑀯𑀸𑀭𑀡𑀸

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀸𑀤𑀻𑀳𑀺𑀧𑀯𑀸𑀭𑀡𑀸

(230.)

1057\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑁂𑁇 𑀢𑁂 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀦𑁆𑀢𑀺, 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀦𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀦𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

1058\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑁂𑁇 𑀢𑁂 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀦𑁆𑀢𑀺, 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

1059\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑁂𑁇 𑀢𑁂 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀦𑁆𑀢𑀺, 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀦𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀦𑀸𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀦𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇

1060\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀫𑀸𑀦𑀲𑀁𑀯𑀸𑀲𑀓𑁂𑁇 𑀢𑁂 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀦𑁆𑀢𑀺, 𑀦𑀸𑀦𑀸𑀲𑀁𑀯𑀸𑀲𑀓𑀤𑀺𑀝𑁆𑀞𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀸𑀝𑁂𑀓𑁆𑀓𑀁 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀦𑁆𑀢𑀺, 𑀅𑀪𑀺𑀯𑀺𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇
