# 𑁪𑁇𑁨𑁦 𑀯𑀚𑁆𑀚𑀦𑀻𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀦

* 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀯𑀚𑁆𑀚𑀦𑀻𑀬𑀧𑀼𑀕𑁆𑀕𑀮𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀦

(233.)

1070\. 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬…𑀧𑁂…  𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸…𑀧𑁂…  𑀦 𑀲𑀺𑀓𑁆𑀔𑀁 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀅𑀦𑁆𑀢𑀺𑀫𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1071\. 𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀤𑀲𑁆𑀲𑀦𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀦 𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑀫𑁆𑀫𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀧𑀸𑀧𑀺𑀓𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄𑁇

1072\. 𑀦 𑀧𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀲𑁆𑀲…𑀧𑁂…  𑀦 𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀲𑁆𑀲 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀧𑀭𑀺𑀲𑀸𑀬 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

1073\. 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀭𑀺𑀯𑀸𑀲𑀺𑀓𑀧𑀯𑀸𑀭𑀡𑀸𑀤𑀸𑀦𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀅𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸”𑀢𑀺𑁇

---

1074\. 𑀤𑀼𑀢𑀺𑀬𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
