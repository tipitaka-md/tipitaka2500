# 𑁪𑁇𑁨𑁫 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀯𑀢𑁆𑀣𑀼

* 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀯𑀢𑁆𑀣𑀼

(240.)

1106\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼—  “𑀫𑀬𑀁 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀓𑀺𑀭 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀢𑀸—  ‘𑀫𑀬𑀁 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

1107\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀲𑀸𑀫𑀦𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  ‘𑀫𑀬𑀁 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀤𑁆𑀯𑁂 𑀢𑀬𑁄 𑀉𑀧𑁄𑀲𑀣𑁂 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑀺𑀓𑁂 𑀓𑀸𑀢𑀼𑀁—  ‘𑀓𑀣𑀁 𑀫𑀬𑀁 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀞𑀫𑀢𑀭𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀮𑀳𑀼𑀁 𑀮𑀳𑀼𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁, 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  ‘𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄; 𑀬𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀽’𑀢𑀺𑁇 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀅𑀲𑀁𑀯𑀺𑀳𑀺𑀢𑀸 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀩𑁆𑀩𑀁, 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀣𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑀩𑁆𑀩𑀁, 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀧𑀸𑀦𑀻𑀬𑁂𑀦 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑀸; 𑀢𑁂𑀲𑀁 𑀯𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁, 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀸—  ‘𑀧𑀯𑀸𑀭𑀺𑀢𑀸 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄; 𑀬𑀣𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀽’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂𑀢𑀁 𑀮𑀪𑁂𑀣, 𑀇𑀘𑁆𑀘𑁂𑀢𑀁 𑀓𑀼𑀲𑀮𑀁𑁇 𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸—

1108\. ‘𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑁂, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑁇 𑀬𑀤𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀇𑀤𑀸𑀦𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀫, 𑀆𑀕𑀫𑁂 𑀓𑀸𑀍𑀏 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀇𑀤𑀸𑀦𑁂𑀯 𑀦𑁄 𑀧𑀯𑀸𑀭𑁂𑀣𑀸’𑀢𑀺, 𑀢𑁂 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑀸 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀯𑀸𑀭𑀡𑀸𑀬; 𑀦 𑀢𑀸𑀯 𑀫𑀬𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀢𑀁 𑀓𑀸𑀍𑀅𑀁 𑀅𑀦𑀼𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸—

1109\. ‘𑀲𑀼𑀡𑀦𑁆𑀢𑀼 𑀫𑁂, 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑁇 𑀬𑀤𑀸𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀇𑀤𑀸𑀦𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬𑀸𑀫, 𑀆𑀕𑀫𑁂 𑀚𑀼𑀡𑁆𑀳𑁂 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬𑀼𑀁—  ‘𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀇𑀤𑀸𑀦𑁂𑀯 𑀦𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑀸’𑀢𑀺, 𑀢𑁂 𑀏𑀯𑀫𑀲𑁆𑀲𑀼 𑀯𑀘𑀦𑀻𑀬𑀸—  ‘𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑀸 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀧𑀯𑀸𑀭𑀡𑀸𑀬, 𑀦 𑀢𑀸𑀯 𑀫𑀬𑀁 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑀸 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑀸 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑀸 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑀸 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑀸 𑀢𑀫𑁆𑀧𑀺 𑀚𑀼𑀡𑁆𑀳𑀁 𑀅𑀦𑀼𑀯𑀲𑁂𑀬𑁆𑀬𑀼𑀁, 𑀢𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀆𑀕𑀫𑁂 𑀚𑀼𑀡𑁆𑀳𑁂 𑀓𑁄𑀫𑀼𑀤𑀺𑀬𑀸 𑀘𑀸𑀢𑀼𑀫𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀅𑀓𑀸𑀫𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇

1110\. 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀕𑀺𑀮𑀸𑀦𑁄 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀆𑀬𑀲𑁆𑀫𑀸 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀕𑀺𑀮𑀸𑀦𑁄 𑀘 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑀓𑁆𑀔𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸𑁇 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀯 𑀅𑀭𑁄𑀕𑁄 𑀳𑁄𑀲𑀺𑁇 𑀅𑀭𑁄𑀕𑁄 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑁂𑀢𑀺, 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀬𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄𑁇 𑀕𑀺𑀮𑀸𑀦𑁄 𑀘 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑀓𑁆𑀔𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸𑁇 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀯𑀸𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑁄𑀕𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀭𑁄𑀕𑀁 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑁂𑀢𑀺, 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀕𑀺𑀮𑀸𑀦𑁄 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀸𑁇 𑀕𑀺𑀮𑀸𑀦𑁄 𑀘 𑀅𑀦𑀦𑀼𑀬𑁄𑀕𑀓𑁆𑀔𑀫𑁄 𑀯𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑀸𑁇 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀯 𑀅𑀭𑁄𑀕𑀸 𑀳𑁄𑀣𑁇 𑀅𑀭𑁄𑀕𑁄 𑀅𑀭𑁄𑀕𑀁 𑀆𑀓𑀗𑁆𑀔𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑁂𑀲𑁆𑀲𑀲𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘𑁂 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑁄 𑀘𑁄𑀤𑁂𑀢𑀺, 𑀅𑀦𑀸𑀤𑀭𑀺𑀬𑁂 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀁𑁇 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀅𑀕𑀺𑀮𑀸𑀦𑁄 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀉𑀪𑁄 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
