# 𑁪𑁇𑁧𑁪 𑀲𑀻𑀫𑁄𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑀓𑀧𑁂𑀬𑁆𑀬𑀸𑀮

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀻𑀫𑁄𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑀓𑀧𑁂𑀬𑁆𑀬𑀸𑀮

(227.)

1047\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀦𑁆𑀢𑀺, 𑀧𑀜𑁆𑀘 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀸 𑀯𑀸𑁇 𑀢𑁂 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀻’𑀢𑀺…𑀧𑁂…  𑀢𑁂 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺—  ‘𑀅𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀸’𑀢𑀺…𑀧𑁂…  𑀢𑁂 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀅𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀢𑁂 𑀦 𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀅𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀢𑁂 𑀦 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 ‘𑀅𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀁 𑀑𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀻’𑀢𑀺…𑀧𑁂…  𑀢𑁂 𑀦 𑀲𑀼𑀡𑀦𑁆𑀢𑀺 ‘𑀅𑀜𑁆𑀜𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀸’𑀢𑀺…𑀧𑁂… 𑁇

1048\. 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀦 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀏𑀓𑀲𑀢𑀧𑀜𑁆𑀘𑀲𑀢𑁆𑀢𑀢𑀺 𑀢𑀺𑀓𑀦𑀬𑀢𑁄, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀦 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀆𑀯𑀸𑀲𑀺𑀓𑀸, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀦 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸, 𑀧𑁂𑀬𑁆𑀬𑀸𑀮𑀫𑀼𑀔𑁂𑀦 𑀲𑀢𑁆𑀢 𑀢𑀺𑀓𑀲𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇
