# 𑁪𑁇𑁧𑁫 𑀤𑀺𑀯𑀲𑀦𑀸𑀦𑀢𑁆𑀢

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀤𑀺𑀯𑀲𑀦𑀸𑀦𑀢𑁆𑀢

(228.)

1049\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀧𑀦𑁆𑀦𑀭𑀲𑁄𑁇 𑀲𑀘𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀲𑀫𑀲𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1050\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀦𑁆𑀦𑀭𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲𑁄𑁇 𑀲𑀘𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀲𑀫𑀲𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇

1051\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀝𑀺𑀧𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀧𑀦𑁆𑀦𑀭𑀲𑁄𑁇 𑀲𑀘𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀦𑀸𑀓𑀸𑀫𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸 𑀲𑀸𑀫𑀕𑁆𑀕𑀻; 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀲𑀫𑀲𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀦𑀸𑀓𑀸𑀫𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸 𑀲𑀸𑀫𑀕𑁆𑀕𑀻; 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀯𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸, 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀯𑀸 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇

1052\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀦𑁆𑀦𑀭𑀲𑁄 𑀳𑁄𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀦𑀁 𑀧𑀸𑀝𑀺𑀧𑀤𑁄𑁇 𑀲𑀘𑁂 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀯𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸, 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀯𑀸 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀲𑀫𑀲𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀯𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸, 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀯𑀸 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀘𑁂 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁂𑀳𑀺 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀦𑀸𑀓𑀸𑀫𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑀸 𑀲𑀸𑀫𑀕𑁆𑀕𑀻; 𑀆𑀯𑀸𑀲𑀺𑀓𑁂𑀳𑀺 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇
