# 𑁪𑁇𑁨𑁪 𑀯𑀢𑁆𑀣𑀼𑀞𑀧𑀦𑀸𑀤𑀺

### 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀯𑀢𑁆𑀣𑀼𑀞𑀧𑀦𑀸𑀤𑀺

(239.)

1102\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀤𑀸𑀳𑀭𑁂𑀬𑁆𑀬—  ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀣𑀼 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦 𑀧𑀼𑀕𑁆𑀕𑀮𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀪𑀕𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀧𑀯𑀸𑀭𑀡𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑁇 𑀲𑀘𑁂 𑀯𑀢𑁆𑀣𑀼 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦 𑀧𑀼𑀕𑁆𑀕𑀮𑁄, 𑀇𑀤𑀸𑀦𑁂𑀯 𑀦𑀁 𑀯𑀤𑁂𑀳𑀻’𑀢𑀺𑁇

1103\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀤𑀸𑀳𑀭𑁂𑀬𑁆𑀬—  ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦 𑀯𑀢𑁆𑀣𑀼𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀪𑀕𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀧𑀯𑀸𑀭𑀡𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑁇 𑀲𑀘𑁂 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀦 𑀯𑀢𑁆𑀣𑀼, 𑀇𑀤𑀸𑀦𑁂𑀯 𑀦𑀁 𑀯𑀤𑁂𑀳𑀻’𑀢𑀺𑁇

1104\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀤𑀳𑀼 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀉𑀤𑀸𑀳𑀭𑁂𑀬𑁆𑀬—  ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀤𑀁 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀘 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀯𑀢𑁆𑀣𑀼𑀜𑁆𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑀜𑁆𑀘 𑀞𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀪𑀕𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀸𑀦𑀜𑁆𑀘 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀜𑁆𑀘 𑀧𑀯𑀸𑀭𑀡𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀸𑁇 𑀲𑀘𑁂 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀘 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀇𑀤𑀸𑀦𑁂𑀯 𑀦𑀁 𑀯𑀤𑁂𑀳𑀻’𑀢𑀺𑁇

1105\. 𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀯𑀢𑁆𑀣𑀼 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀧𑀼𑀕𑁆𑀕𑀮𑁄, 𑀓𑀮𑁆𑀮𑀁 𑀯𑀘𑀦𑀸𑀬𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀧𑀘𑁆𑀙𑀸 𑀯𑀢𑁆𑀣𑀼, 𑀓𑀮𑁆𑀮𑀁 𑀯𑀘𑀦𑀸𑀬𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀯𑀢𑁆𑀣𑀼 𑀘 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀘 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀢𑀜𑁆𑀘𑁂 𑀓𑀢𑀸𑀬 𑀧𑀯𑀸𑀭𑀡𑀸𑀬 𑀉𑀓𑁆𑀓𑁄𑀝𑁂𑀢𑀺, 𑀉𑀓𑁆𑀓𑁄𑀝𑀦𑀓𑀁 𑀧𑀸𑀘𑀺𑀢𑁆𑀢𑀺𑀬𑀦𑁆”𑀢𑀺𑁇
