# 𑁪𑁇𑁨𑁬 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳

* 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳

(241.)

1111\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀔𑁄 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄𑁇 𑀲𑀘𑁂 𑀫𑀬𑀁 𑀇𑀤𑀸𑀦𑀺 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫, 𑀲𑀺𑀬𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀇𑀫𑀫𑁆𑀳𑀸 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸 𑀧𑀭𑀺𑀩𑀸𑀳𑀺𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

1112\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀦𑁆𑀤𑀺𑀝𑁆𑀞𑀸 𑀲𑀫𑁆𑀪𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀔𑁄 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄𑁇 𑀲𑀘𑁂 𑀫𑀬𑀁 𑀇𑀤𑀸𑀦𑀺 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫, 𑀲𑀺𑀬𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀇𑀫𑀫𑁆𑀳𑀸 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸 𑀧𑀭𑀺𑀩𑀸𑀳𑀺𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀩𑁆𑀩𑀁—  𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1113\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄𑁇 𑀲𑀘𑁂 𑀫𑀬𑀁 𑀇𑀤𑀸𑀦𑀺 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫, 𑀲𑀺𑀬𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀇𑀫𑀫𑁆𑀳𑀸 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸 𑀧𑀭𑀺𑀩𑀸𑀳𑀺𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀇𑀤𑀸𑀦𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁂𑀬𑁆𑀬, 𑀆𑀕𑀫𑁂 𑀚𑀼𑀡𑁆𑀳𑁂 𑀓𑁄𑀫𑀼𑀤𑀺𑀬𑀸 𑀘𑀸𑀢𑀼𑀫𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀧𑀯𑀸𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1114\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀫𑀕𑁆𑀕𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀯𑀺𑀯𑀤𑀫𑀸𑀦𑀸𑀦𑀁 𑀯𑀺𑀳𑀭𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑁄 𑀅𑀥𑀺𑀕𑀢𑁄𑁇 𑀲𑀘𑁂 𑀫𑀬𑀁 𑀇𑀤𑀸𑀦𑀺 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫, 𑀲𑀺𑀬𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀇𑀫𑀫𑁆𑀳𑀸 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀸 𑀧𑀭𑀺𑀩𑀸𑀳𑀺𑀭𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀇𑀤𑀸𑀦𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀆𑀕𑀫𑁂 𑀚𑀼𑀡𑁆𑀳𑁂 𑀓𑁄𑀫𑀼𑀤𑀺𑀬𑀸 𑀘𑀸𑀢𑀼𑀫𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳𑀲𑁆𑀲 𑀓𑀭𑀡𑀁, 𑀇𑀤𑀸𑀦𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀆𑀕𑀫𑁂 𑀚𑀼𑀡𑁆𑀳𑁂 𑀓𑁄𑀫𑀼𑀤𑀺𑀬𑀸 𑀘𑀸𑀢𑀼𑀫𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1115\. 𑀓𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳𑁄, 𑀇𑀤𑀸𑀦𑀺 𑀉𑀧𑁄𑀲𑀣𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀲𑁆𑀲𑀢𑀺, 𑀆𑀕𑀫𑁂 𑀚𑀼𑀡𑁆𑀳𑁂 𑀓𑁄𑀫𑀼𑀤𑀺𑀬𑀸 𑀘𑀸𑀢𑀼𑀫𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇

1116\. 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀓𑀢𑁂 𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀯𑀤𑁂𑀬𑁆𑀬—  ‘𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀚𑀦𑀧𑀤𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁; 𑀅𑀢𑁆𑀣𑀺 𑀫𑁂 𑀚𑀦𑀧𑀤𑁂 𑀓𑀭𑀡𑀻𑀬𑀦𑁆’𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀲𑀸𑀥𑀸𑀯𑀼𑀲𑁄, 𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀕𑀘𑁆𑀙𑀸𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀯𑀸𑀭𑀬𑀫𑀸𑀦𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑁄 𑀔𑁄 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬, 𑀦 𑀢𑀸𑀯𑀸𑀳𑀁 𑀧𑀯𑀸𑀭𑁂𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀸𑀭𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀉𑀪𑁄 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀚𑀦𑀧𑀤𑁂 𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀻𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀅𑀦𑁆𑀢𑁄 𑀓𑁄𑀫𑀼𑀤𑀺𑀬𑀸 𑀘𑀸𑀢𑀼𑀫𑀸𑀲𑀺𑀦𑀺𑀬𑀸 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀲𑁄 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑁄 𑀔𑁄 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀯𑀸𑀭𑀡𑀸𑀬; 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀅𑀳𑀦𑁆’𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀧𑀯𑀸𑀭𑀺𑀬𑀫𑀸𑀦𑁂 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀯𑀸𑀭𑀡𑀁 𑀞𑀧𑁂𑀢𑀺, 𑀉𑀪𑁄 𑀲𑀁𑀖𑁂𑀦 𑀲𑀫𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑀫𑀦𑀼𑀕𑀸𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁂𑀦 𑀧𑀯𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇

---

1117\. 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇



---

1118\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1119\. _𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀓𑁄𑀲𑀮𑁂𑀲𑀼,_  
_𑀅𑀕𑀫𑀼𑀁 𑀲𑀢𑁆𑀣𑀼 𑀤𑀲𑁆𑀲𑀦𑀁;_  
_𑀅𑀨𑀸𑀲𑀼𑀁 𑀧𑀲𑀼𑀲𑀁𑀯𑀸𑀲𑀁,_  
_𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀸𑀦𑀼𑀮𑁄𑀫𑀢𑀸𑁇_  


1120\. _𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀸 𑀧𑀡𑀸𑀫𑀜𑁆𑀘,_  
_𑀓𑀫𑁆𑀫𑀁 𑀕𑀺𑀮𑀸𑀦𑀜𑀸𑀢𑀓𑀸;_  
_𑀭𑀸𑀚𑀸 𑀘𑁄𑀭𑀸 𑀘 𑀥𑀼𑀢𑁆𑀢𑀸 𑀘,_  
_𑀪𑀺𑀓𑁆𑀔𑀼𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀢𑀣𑀸𑁇_  


1121\. _𑀧𑀜𑁆𑀘 𑀘𑀢𑀼𑀢𑀬𑁄 𑀤𑁆𑀯𑁂𑀓𑁄,_  
_𑀆𑀧𑀦𑁆𑀦𑁄 𑀯𑁂𑀫𑀢𑀻 𑀲𑀭𑀺;_  
_𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑁄 𑀯𑁂𑀫𑀢𑀺𑀓𑁄,_  
_𑀩𑀳𑀽 𑀲𑀫𑀸 𑀘 𑀣𑁄𑀓𑀺𑀓𑀸𑁇_  


1122\. _𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀲,_  
_𑀮𑀺𑀗𑁆𑀕𑀲𑀁𑀯𑀸𑀲𑀓𑀸 𑀉𑀪𑁄;_  
_𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁 𑀦 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸𑀬,_  
_𑀙𑀦𑁆𑀤𑀤𑀸𑀦𑁂 𑀧𑀯𑀸𑀭𑀡𑀸𑁇_  


1123\. _𑀲𑀯𑀭𑁂𑀳𑀺 𑀔𑁂𑀧𑀺𑀢𑀸 𑀫𑁂𑀖𑁄,_  
_𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀧𑀯𑀸𑀭𑀡𑀸;_  
_𑀦 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀧𑀼𑀭𑀫𑁆𑀳𑀸𑀓𑀁,_  
_𑀅𑀝𑁆𑀞𑀧𑀺𑀢𑀸 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄𑁇_  


1124\. _𑀓𑀺𑀫𑁆𑀳𑀺 𑀯𑀸𑀢𑀺 𑀓𑀢𑀫𑀜𑁆𑀘,_  
_𑀤𑀺𑀝𑁆𑀞𑁂𑀦 𑀲𑀼𑀢𑀲𑀗𑁆𑀓𑀸𑀬;_  
_𑀘𑁄𑀤𑀓𑁄 𑀘𑀼𑀤𑀺𑀢𑀓𑁄 𑀘,_  
_𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀁 𑀯𑀢𑁆𑀣𑀼 𑀪𑀡𑁆𑀟𑀦𑀁;_  
_𑀧𑀯𑀸𑀭𑀡𑀸𑀲𑀗𑁆𑀕𑀳𑁄 𑀘,_  
_𑀅𑀦𑀺𑀲𑁆𑀲𑀭𑁄 𑀧𑀯𑀸𑀭𑀬𑁂𑀢𑀺𑁇_  


---

1125\. 𑀇𑀫𑀫𑁆𑀳𑀺 𑀔𑀦𑁆𑀥𑀓𑁂 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀙𑀘𑀢𑁆𑀢𑀸𑀭𑀻𑀲𑀸𑀢𑀺𑁇



---

1126\. 𑀧𑀯𑀸𑀭𑀡𑀸𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
