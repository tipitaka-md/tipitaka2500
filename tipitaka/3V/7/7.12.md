# 𑁭𑁇𑁧𑁨 𑀅𑀧𑀯𑀺𑀮𑀸𑀬𑀦𑀦𑀯𑀓

### 𑀓𑀣𑀺𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀧𑀯𑀺𑀮𑀸𑀬𑀦𑀦𑀯𑀓

(321.)

1508. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑀁 𑀅𑀧𑀯𑀺𑀮𑀸𑀬𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀺𑀲𑀁𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀓𑀳𑀁 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄, 𑀓𑀢𑁆𑀣 𑀘 𑀢𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄𑀫𑁆𑀳𑀺𑁇 𑀢𑀢𑁆𑀣 𑀘 𑀫𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀳𑀭, 𑀫𑀬𑀁 𑀢𑁂 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀢𑀺—  “𑀓𑀳𑀁 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀬𑀁 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄; 𑀓𑀳𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀅𑀫𑀼𑀓𑀁 𑀦𑀸𑀫 𑀆𑀯𑀸𑀲𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁆𑀣 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀫𑀬𑀁 𑀢𑁂 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁧)

1509. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺…𑀧𑁂…  “𑀦𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁨)

1510. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺…𑀧𑁂…  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑀁 𑀦𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀸𑀲𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁩)

1511. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑀁 𑀅𑀧𑀯𑀺𑀮𑀸𑀬𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀺𑀲𑀁𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀓𑀳𑀁 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄, 𑀓𑀢𑁆𑀣 𑀘 𑀢𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄𑀫𑁆𑀳𑀺, 𑀢𑀢𑁆𑀣 𑀘 𑀫𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀳𑀭, 𑀫𑀬𑀁 𑀢𑁂 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀢𑀺—  “𑀓𑀳𑀁 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀬𑀁 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀆𑀯𑀼𑀲𑁄, 𑀓𑀳𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀅𑀫𑀼𑀓𑀁 𑀦𑀸𑀫 𑀆𑀯𑀸𑀲𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁆𑀣 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸 𑀅𑀕𑀫𑀸𑀲𑀺, 𑀫𑀬𑀁 𑀢𑁂 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁪)

1512. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑀁 𑀅𑀧𑀯𑀺𑀮𑀸𑀬𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀺𑀲𑀁𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀓𑀳𑀁 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄, 𑀓𑀢𑁆𑀣 𑀘 𑀢𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄𑀫𑁆𑀳𑀺, 𑀢𑀢𑁆𑀣 𑀘 𑀫𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀳𑀭, 𑀫𑀬𑀁 𑀢𑁂 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀢𑀺—  “𑀓𑀳𑀁 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀬𑀁 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀫𑁂𑀦𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀆𑀯𑀼𑀲𑁄, 𑀓𑀳𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀲𑀻”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀅𑀫𑀼𑀓𑀁 𑀦𑀸𑀫 𑀆𑀯𑀸𑀲𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀢𑁆𑀣 𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸 𑀅𑀕𑀫𑀸𑀲𑀺, 𑀫𑀬𑀁 𑀢𑁂 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀦𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁫)

1513. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺…𑀧𑁂…  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑀁 𑀦𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀸𑀲𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁬)

1514. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑀁 𑀅𑀧𑀯𑀺𑀮𑀸𑀬𑀫𑀸𑀦𑁄𑁇 𑀢𑀫𑁂𑀦𑀁 𑀤𑀺𑀲𑀁𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀓𑀳𑀁 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄, 𑀓𑀢𑁆𑀣 𑀘 𑀢𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀲𑁄 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄𑀫𑁆𑀳𑀺, 𑀢𑀢𑁆𑀣 𑀘 𑀫𑁂 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀕𑀘𑁆𑀙𑀸𑀯𑀼𑀲𑁄, 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀳𑀭, 𑀫𑀬𑀁 𑀢𑁂 𑀇𑀥 𑀘𑀻𑀯𑀭𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀢𑀺—  “𑀓𑀳𑀁 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺? 𑀢𑁂 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀬𑀁 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁭)

1515. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺…𑀧𑁂…  “𑀦𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁮)

1516. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀤𑀺𑀲𑀗𑁆𑀕𑀫𑀺𑀓𑁄 𑀧𑀓𑁆𑀓𑀫𑀢𑀺…𑀧𑁂…  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑀁 𑀦𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀸𑀲𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁯)
