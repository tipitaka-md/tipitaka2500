# 𑁭𑁇𑁧𑁧 𑀓𑀭𑀡𑀻𑀬𑀤𑁄𑀍𑀅𑀲𑀓

### 𑀓𑀣𑀺𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀓𑀭𑀡𑀻𑀬𑀤𑁄𑀍𑀅𑀲𑀓

(320.)

1495. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁧)

1496. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀦𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁨)

1497. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑀁 𑀦𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀸𑀲𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁩)

1498. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀸 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀲𑀸𑀯𑀘𑁆𑀙𑁂𑀤𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁪)

1499. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁫)

1500. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀦𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁬)

1501. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑀁 𑀦𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀸𑀲𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁭)

1502. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀸 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀲𑀸𑀯𑀘𑁆𑀙𑁂𑀤𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁮)

1503. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁂𑀦; 𑀦𑁂𑀯𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺, 𑀦 𑀧𑀦𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁯)

1504. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁂𑀦; 𑀦𑁂𑀯𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺, 𑀦 𑀧𑀦𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀦𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀦𑁆𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁧𑁦)

1505. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁂𑀦; 𑀦𑁂𑀯𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺, 𑀦 𑀧𑀦𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀅𑀦𑀸𑀲𑀸𑀬 𑀮𑀪𑀢𑀺, 𑀆𑀲𑀸𑀬 𑀦 𑀮𑀪𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀬𑀺𑀭𑀫𑀸𑀦𑀁 𑀦𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀸𑀲𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁧𑁧)

1506. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀓𑁆𑀓𑀫𑀢𑀺 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁂𑀦; 𑀦𑁂𑀯𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺, 𑀦 𑀧𑀦𑀲𑁆𑀲 𑀳𑁄𑀢𑀺—  “𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀸𑀲𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑀸 𑀘𑀻𑀯𑀭𑀸𑀲𑀸 𑀉𑀧𑀘𑁆𑀙𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀲𑀸𑀯𑀘𑁆𑀙𑁂𑀤𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄𑁇 (𑁧𑁨)
