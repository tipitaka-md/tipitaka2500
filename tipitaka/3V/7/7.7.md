# 𑁭𑁇𑁭 𑀲𑀫𑀸𑀤𑀸𑀬𑀧𑀦𑁆𑀦𑀭𑀲𑀓𑀸𑀤𑀺

### 𑀓𑀣𑀺𑀦𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀫𑀸𑀤𑀸𑀬𑀧𑀦𑁆𑀦𑀭𑀲𑀓𑀸𑀤𑀺

(316.)

1446\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀘𑀻𑀯𑀭𑀁 𑀲𑀫𑀸𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀢𑀺…𑀧𑁂… 𑁇

---

1447\. (𑀆𑀤𑀸𑀬𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀁 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)



1448\. 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀣𑀢𑀓𑀣𑀺𑀦𑁄 𑀯𑀺𑀧𑁆𑀧𑀓𑀢𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀩𑀳𑀺𑀲𑀻𑀫𑀕𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  “𑀇𑀥𑁂𑀯𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀲𑁆𑀲𑀁, 𑀦 𑀧𑀘𑁆𑀘𑁂𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀦𑀺𑀝𑁆𑀞𑀸𑀦𑀦𑁆𑀢𑀺𑀓𑁄 𑀓𑀣𑀺𑀦𑀼𑀤𑁆𑀥𑀸𑀭𑁄…𑀧𑁂… 𑁇

---

1449\. (𑀲𑀫𑀸𑀤𑀸𑀬𑀯𑀸𑀭𑀲𑀤𑀺𑀲𑀁 𑀏𑀯𑀁 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇)
