# 𑁮𑁇𑁧 𑀚𑀻𑀯𑀓𑀯𑀢𑁆𑀣𑀼

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀚𑀻𑀯𑀓𑀯𑀢𑁆𑀣𑀼

(326.)

1558\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀳𑁄𑀢𑀺 𑀨𑀺𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘; 𑀲𑀢𑁆𑀢 𑀘 𑀧𑀸𑀲𑀸𑀤𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑀸𑀲𑀸𑀤𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑀸𑀲𑀸𑀤𑀸; 𑀲𑀢𑁆𑀢 𑀘 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸𑀦𑀺; 𑀲𑀢𑁆𑀢 𑀘 𑀆𑀭𑀸𑀫𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀆𑀭𑀸𑀫𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀆𑀭𑀸𑀫𑀸; 𑀲𑀢𑁆𑀢 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑁄; 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀻 𑀘 𑀕𑀡𑀺𑀓𑀸 𑀅𑀪𑀺𑀭𑀽𑀧𑀸 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸 𑀧𑀭𑀫𑀸𑀬 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸, 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀦𑀘𑁆𑀘𑁂 𑀘 𑀕𑀻𑀢𑁂 𑀘 𑀯𑀸𑀤𑀺𑀢𑁂 𑀘, 𑀅𑀪𑀺𑀲𑀝𑀸 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀸𑀬 𑀘 𑀭𑀢𑁆𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺; 𑀢𑀸𑀬 𑀘 𑀯𑁂𑀲𑀸𑀮𑀻 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀉𑀧𑀲𑁄𑀪𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑀓𑁄 𑀦𑁂𑀕𑀫𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀅𑀕𑀫𑀸𑀲𑀺 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑀓𑁄 𑀦𑁂𑀕𑀫𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀇𑀤𑁆𑀥𑀜𑁆𑀘𑁂𑀯 𑀨𑀺𑀢𑀜𑁆𑀘 𑀩𑀳𑀼𑀚𑀦𑀜𑁆𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀜𑁆𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀜𑁆𑀘; 𑀲𑀢𑁆𑀢 𑀘 𑀧𑀸𑀲𑀸𑀤𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑀸𑀲𑀸𑀤𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑀸𑀲𑀸𑀤𑁂; 𑀲𑀢𑁆𑀢 𑀘 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀸𑀦𑀺; 𑀲𑀢𑁆𑀢 𑀘 𑀆𑀭𑀸𑀫𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀆𑀭𑀸𑀫𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀆𑀭𑀸𑀫𑁂; 𑀲𑀢𑁆𑀢 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑀲𑀢𑀸𑀦𑀺 𑀲𑀢𑁆𑀢 𑀘 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑁄; 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀜𑁆𑀘 𑀕𑀡𑀺𑀓𑀁 𑀅𑀪𑀺𑀭𑀽𑀧𑀁 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀁 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀧𑀭𑀫𑀸𑀬 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁, 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀦𑀘𑁆𑀘𑁂 𑀘 𑀕𑀻𑀢𑁂 𑀘 𑀯𑀸𑀤𑀺𑀢𑁂 𑀘, 𑀅𑀪𑀺𑀲𑀝𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀲𑀸𑀬 𑀘 𑀭𑀢𑁆𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑀁, 𑀢𑀸𑀬 𑀘 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀉𑀧𑀲𑁄𑀪𑀦𑁆𑀢𑀺𑀁𑁇

(327.)

1559\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑀓𑁄 𑀦𑁂𑀕𑀫𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁 𑀢𑀻𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀜𑁆𑀙𑀺𑁇 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑁂𑀲𑀸𑀮𑀻, 𑀤𑁂𑀯, 𑀇𑀤𑁆𑀥𑀸 𑀘𑁂𑀯 𑀨𑀺𑀢𑀸 𑀘 𑀩𑀳𑀼𑀚𑀦𑀸 𑀘 𑀆𑀓𑀺𑀡𑁆𑀡𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀘; 𑀲𑀢𑁆𑀢 𑀘 𑀧𑀸𑀲𑀸𑀤𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺…𑀧𑁂…  𑀢𑀸𑀬 𑀘 𑀯𑁂𑀲𑀸𑀮𑀻 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀉𑀧𑀲𑁄𑀪𑀢𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀤𑁂𑀯, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀕𑀡𑀺𑀓𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀢𑀸𑀤𑀺𑀲𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀁 𑀚𑀸𑀦𑀸𑀣 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀕𑀡𑀺𑀓𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺𑁇

1560\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑀸𑀮𑀯𑀢𑀻 𑀦𑀸𑀫 𑀓𑀼𑀫𑀸𑀭𑀻 𑀅𑀪𑀺𑀭𑀽𑀧𑀸 𑀳𑁄𑀢𑀺 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸 𑀧𑀭𑀫𑀸𑀬 𑀯𑀡𑁆𑀡𑀧𑁄𑀓𑁆𑀔𑀭𑀢𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑀓𑁄 𑀦𑁂𑀕𑀫𑁄 𑀲𑀸𑀮𑀯𑀢𑀺𑀁 𑀓𑀼𑀫𑀸𑀭𑀺𑀁 𑀕𑀡𑀺𑀓𑀁 𑀯𑀼𑀝𑁆𑀞𑀸𑀧𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀮𑀯𑀢𑀻 𑀕𑀡𑀺𑀓𑀸 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀸 𑀅𑀳𑁄𑀲𑀺 𑀦𑀘𑁆𑀘𑁂 𑀘 𑀕𑀻𑀢𑁂 𑀘 𑀯𑀸𑀤𑀺𑀢𑁂 𑀘, 𑀅𑀪𑀺𑀲𑀝𑀸 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀝𑀺𑀲𑀢𑁂𑀦 𑀘 𑀭𑀢𑁆𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀮𑀯𑀢𑀻 𑀕𑀡𑀺𑀓𑀸 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀮𑀯𑀢𑀺𑀬𑀸 𑀕𑀡𑀺𑀓𑀸𑀬 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀢𑁆𑀣𑀻 𑀔𑁄 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀧𑀼𑀭𑀺𑀲𑀸𑀦𑀁 𑀅𑀫𑀦𑀸𑀧𑀸𑁇 𑀲𑀘𑁂 𑀫𑀁 𑀓𑁄𑀘𑀺 𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺—  𑀲𑀸𑀮𑀯𑀢𑀻 𑀕𑀡𑀺𑀓𑀸 𑀕𑀩𑁆𑀪𑀺𑀦𑀻𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀫𑁂 𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀪𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀕𑀺𑀮𑀸𑀦𑀁 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀮𑀯𑀢𑀻 𑀕𑀡𑀺𑀓𑀸 𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀫𑀸, 𑀪𑀡𑁂 𑀤𑁄𑀯𑀸𑀭𑀺𑀓, 𑀓𑁄𑀘𑀺 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀬𑁄 𑀘 𑀫𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺, ‘𑀕𑀺𑀮𑀸𑀦𑀸’𑀢𑀺 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀤𑁄𑀯𑀸𑀭𑀺𑀓𑁄 𑀲𑀸𑀮𑀯𑀢𑀺𑀬𑀸 𑀕𑀡𑀺𑀓𑀸𑀬 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀮𑀯𑀢𑀻 𑀕𑀡𑀺𑀓𑀸 𑀢𑀲𑁆𑀲 𑀕𑀩𑁆𑀪𑀲𑁆𑀲 𑀧𑀭𑀺𑀧𑀸𑀓𑀫𑀦𑁆𑀯𑀸𑀬 𑀧𑀼𑀢𑁆𑀢𑀁 𑀯𑀺𑀚𑀸𑀬𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀮𑀯𑀢𑀻 𑀕𑀡𑀺𑀓𑀸 𑀤𑀸𑀲𑀺𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀳𑀦𑁆𑀤, 𑀚𑁂, 𑀇𑀫𑀁 𑀤𑀸𑀭𑀓𑀁 𑀓𑀢𑁆𑀢𑀭𑀲𑀼𑀧𑁆𑀧𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀓𑀽𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀲𑀸 𑀤𑀸𑀲𑀻 𑀲𑀸𑀮𑀯𑀢𑀺𑀬𑀸 𑀕𑀡𑀺𑀓𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀓𑀢𑁆𑀢𑀭𑀲𑀼𑀧𑁆𑀧𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀲𑀗𑁆𑀓𑀸𑀭𑀓𑀽𑀝𑁂 𑀙𑀟𑁆𑀟𑁂𑀲𑀺𑁇

1561\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀪𑀬𑁄 𑀦𑀸𑀫 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑁄 𑀓𑀸𑀮𑀲𑁆𑀲𑁂𑀯 𑀭𑀸𑀚𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀓𑀸𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀓𑀺𑀁 𑀏𑀢𑀁, 𑀪𑀡𑁂, 𑀓𑀸𑀓𑁂𑀳𑀺 𑀲𑀫𑁆𑀧𑀭𑀺𑀓𑀺𑀡𑁆𑀡𑀦𑁆”𑀢𑀺? “𑀤𑀸𑀭𑀓𑁄, 𑀤𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀚𑀻𑀯𑀢𑀺, 𑀪𑀡𑁂”𑀢𑀺? “𑀚𑀻𑀯𑀢𑀺, 𑀤𑁂𑀯𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀦𑁂𑀢𑁆𑀯𑀸 𑀥𑀸𑀢𑀻𑀦𑀁 𑀤𑁂𑀣 𑀧𑁄𑀲𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀅𑀪𑀬𑀲𑁆𑀲 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀢𑀁 𑀤𑀸𑀭𑀓𑀁 𑀅𑀪𑀬𑀲𑁆𑀲 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑀁 𑀦𑁂𑀢𑁆𑀯𑀸 𑀥𑀸𑀢𑀻𑀦𑀁 𑀅𑀤𑀁𑀲𑀼—  “𑀧𑁄𑀲𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀚𑀻𑀯𑀢𑀻𑀢𑀺 “𑀚𑀻𑀯𑀓𑁄”𑀢𑀺 𑀦𑀸𑀫𑀁 𑀅𑀓𑀁𑀲𑀼𑁇 𑀓𑀼𑀫𑀸𑀭𑁂𑀦 𑀧𑁄𑀲𑀸𑀧𑀺𑀢𑁄𑀢𑀺 “𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄”𑀢𑀺 𑀦𑀸𑀫𑀁 𑀅𑀓𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀯𑀺𑀜𑁆𑀜𑀼𑀢𑀁 𑀧𑀸𑀧𑀼𑀡𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀬𑁂𑀦 𑀅𑀪𑀬𑁄 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀪𑀬𑀁 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀸 𑀫𑁂, 𑀤𑁂𑀯, 𑀫𑀸𑀢𑀸, 𑀓𑁄 𑀧𑀺𑀢𑀸”𑀢𑀺? “𑀅𑀳𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑁂, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓, 𑀫𑀸𑀢𑀭𑀁 𑀦 𑀚𑀸𑀦𑀸𑀫𑀺; 𑀅𑀧𑀺 𑀘𑀸𑀳𑀁 𑀢𑁂 𑀧𑀺𑀢𑀸; 𑀫𑀬𑀸𑀲𑀺 𑀧𑁄𑀲𑀸𑀧𑀺𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀔𑁄 𑀭𑀸𑀚𑀓𑀼𑀮𑀸𑀦𑀺 𑀦 𑀲𑀼𑀓𑀭𑀸𑀦𑀺 𑀅𑀲𑀺𑀧𑁆𑀧𑁂𑀦 𑀉𑀧𑀚𑀻𑀯𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

1562\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀓𑁆𑀓𑀲𑀺𑀮𑀸𑀬𑀁 𑀤𑀺𑀲𑀸𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑁄 𑀯𑁂𑀚𑁆𑀚𑁄 𑀧𑀝𑀺𑀯𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀅𑀪𑀬𑀁 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀬𑁂𑀦 𑀢𑀓𑁆𑀓𑀲𑀺𑀮𑀸 𑀢𑁂𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀬𑁂𑀦 𑀢𑀓𑁆𑀓𑀲𑀺𑀮𑀸, 𑀬𑁂𑀦 𑀯𑁂𑀚𑁆𑀚𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀯𑁂𑀚𑁆𑀚𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀲𑀺𑀧𑁆𑀧𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓, 𑀲𑀺𑀓𑁆𑀔𑀲𑁆𑀲𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀩𑀳𑀼𑀜𑁆𑀘 𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀮𑀳𑀼𑀜𑁆𑀘 𑀕𑀡𑁆𑀳𑀸𑀢𑀺 𑀲𑀼𑀝𑁆𑀞𑀼 𑀘 𑀉𑀧𑀥𑀸𑀭𑁂𑀢𑀺, 𑀕𑀳𑀺𑀢𑀜𑁆𑀘𑀲𑁆𑀲 𑀦 𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀩𑀳𑀼𑀜𑁆𑀘 𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀮𑀳𑀼𑀜𑁆𑀘 𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀲𑀼𑀝𑁆𑀞𑀼 𑀘 𑀉𑀧𑀥𑀸𑀭𑁂𑀫𑀺, 𑀕𑀳𑀺𑀢𑀜𑁆𑀘 𑀫𑁂 𑀦 𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀘 𑀫𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀥𑀻𑀬𑀦𑁆𑀢𑀲𑁆𑀲, 𑀦𑀬𑀺𑀫𑀲𑁆𑀲 𑀲𑀺𑀧𑁆𑀧𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀓𑀤𑀸 𑀇𑀫𑀲𑁆𑀲 𑀲𑀺𑀧𑁆𑀧𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1563\. 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀬𑁂𑀦 𑀲𑁄 𑀯𑁂𑀚𑁆𑀚𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀯𑁂𑀚𑁆𑀚𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀘𑀭𑀺𑀬, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀮𑀳𑀼𑀜𑁆𑀘 𑀕𑀡𑁆𑀳𑀸𑀫𑀺 𑀲𑀼𑀝𑁆𑀞𑀼 𑀘 𑀉𑀧𑀥𑀸𑀭𑁂𑀫𑀺, 𑀕𑀳𑀺𑀢𑀜𑁆𑀘 𑀫𑁂 𑀦 𑀲𑀫𑁆𑀫𑀼𑀲𑁆𑀲𑀢𑀺, 𑀲𑀢𑁆𑀢 𑀘 𑀫𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀅𑀥𑀻𑀬𑀦𑁆𑀢𑀲𑁆𑀲, 𑀦𑀬𑀺𑀫𑀲𑁆𑀲 𑀲𑀺𑀧𑁆𑀧𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀓𑀤𑀸 𑀇𑀫𑀲𑁆𑀲 𑀲𑀺𑀧𑁆𑀧𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓, 𑀔𑀡𑀺𑀢𑁆𑀢𑀺𑀁 𑀆𑀤𑀸𑀬 𑀢𑀓𑁆𑀓𑀲𑀺𑀮𑀸𑀬 𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀺𑀢𑁆𑀯𑀸 𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀺 𑀢𑀁 𑀆𑀳𑀭𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀆𑀘𑀭𑀺𑀬𑀸”𑀢𑀺 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀲𑁆𑀲 𑀯𑁂𑀚𑁆𑀚𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀔𑀡𑀺𑀢𑁆𑀢𑀺𑀁 𑀆𑀤𑀸𑀬 𑀢𑀓𑁆𑀓𑀲𑀺𑀮𑀸𑀬 𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀅𑀤𑁆𑀤𑀲𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀬𑁂𑀦 𑀲𑁄 𑀯𑁂𑀚𑁆𑀚𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀯𑁂𑀚𑁆𑀚𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄𑀫𑁆𑀳𑀺, 𑀆𑀘𑀭𑀺𑀬, 𑀢𑀓𑁆𑀓𑀲𑀺𑀮𑀸𑀬 𑀲𑀫𑀦𑁆𑀢𑀸 𑀬𑁄𑀚𑀦𑀁, 𑀦 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀅𑀤𑁆𑀤𑀲𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀼𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄𑀲𑀺, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓𑁇 𑀅𑀮𑀁 𑀢𑁂 𑀏𑀢𑁆𑀢𑀓𑀁 𑀚𑀻𑀯𑀺𑀓𑀸𑀬𑀸”𑀢𑀺 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀁 𑀧𑀸𑀤𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑁂𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀢𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀲𑀸𑀓𑁂𑀢𑁂 𑀧𑀭𑀺𑀓𑁆𑀔𑀬𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑁂 𑀔𑁄 𑀫𑀕𑁆𑀕𑀸 𑀓𑀦𑁆𑀢𑀸𑀭𑀸 𑀅𑀧𑁆𑀧𑁄𑀤𑀓𑀸 𑀅𑀧𑁆𑀧𑀪𑀓𑁆𑀔𑀸, 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀅𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑁂𑀦 𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇
