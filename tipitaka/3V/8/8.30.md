# 𑁮𑁇𑁩𑁦 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀤𑀺𑀓𑀣𑀸

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀤𑀺𑀓𑀣𑀸

(377.)

1735\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀭𑁂𑀯𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀳𑀢𑁆𑀣𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀣𑁂𑀭𑀲𑁆𑀲 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀭𑁂𑀯𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀭𑁂𑀯𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀲𑀫𑀸𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀣𑁂𑀭𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑀁𑁇 𑀲𑀫𑁆𑀧𑀢𑁆𑀢𑀁 𑀢𑀁 𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺? “𑀦𑀸𑀳𑀁 𑀢𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀘𑀻𑀯𑀭𑀁 𑀧𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀭𑁂𑀯𑀢𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀳𑀢𑁆𑀣𑁂 𑀣𑁂𑀭𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀳𑁂𑀲𑀺𑀁𑁇 𑀓𑀳𑀁 𑀢𑀁 𑀘𑀻𑀯𑀭𑀦𑁆”𑀢𑀺? “𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑀦𑁆”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

(378.)

1736\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1737\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1738\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀲𑀼𑀡𑀸𑀢𑀺—  ‘𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀲𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀲𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1739\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀲𑀼𑀡𑀸𑀢𑀺—  ‘𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀲𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1740\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀳𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀲𑀼𑀡𑀸𑀢𑀺—  ‘𑀉𑀪𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸’𑀢𑀺𑁇 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀲𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇

1741\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑀫𑁆𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1742\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑀫𑁆𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1743\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑀫𑁆𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀲𑀼𑀡𑀸𑀢𑀺—  ‘𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀲𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1744\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑀫𑁆𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀲𑀼𑀡𑀸𑀢𑀺—  ‘𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀲𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀲𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀺𑀲𑁆𑀲𑀸𑀲𑀸 𑀕𑀡𑁆𑀳𑀸𑀢𑀺𑁇 𑀤𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀁𑁇

1745\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑁂 𑀘𑀻𑀯𑀭𑀁 𑀧𑀳𑀺𑀡𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑀫𑁆𑀫𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀲𑀼𑀡𑀸𑀢𑀺 ‘𑀉𑀪𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀸’𑀢𑀺𑁇 𑀬𑁄 𑀧𑀳𑀺𑀡𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀤𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇 𑀬𑀲𑁆𑀲 𑀧𑀳𑀺𑀬𑁆𑀬𑀢𑀺 𑀢𑀲𑁆𑀲 𑀫𑀢𑀓𑀘𑀻𑀯𑀭𑀁 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀲𑁆𑀯𑀸𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
