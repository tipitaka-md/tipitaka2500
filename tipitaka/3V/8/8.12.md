# 𑁮𑁇𑁧𑁨 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀓𑀣𑀸

(342.)

1602\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀘𑀻𑀯𑀭𑀁 𑀆𑀤𑀸𑀬 𑀆𑀭𑀸𑀫𑀁 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑀸 𑀧𑀝𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀼𑀁—  𑀬𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀕𑀳𑀺𑀢𑀸𑀕𑀳𑀺𑀢𑀜𑁆𑀘 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀧𑀞𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄; 𑀬𑀸𑀘𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1603\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1604\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1605\. 𑀲𑀫𑁆𑀫𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑁄𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

1606\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀕𑁆𑀕𑀸𑀳𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀉𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀘𑀻𑀯𑀭𑀁 𑀦𑀲𑁆𑀲𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑀻𑀯𑀭𑀦𑀺𑀤𑀳𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀼𑀁—  𑀬𑁄 𑀦 𑀙𑀦𑁆𑀤𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀤𑁄𑀲𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀫𑁄𑀳𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦 𑀪𑀬𑀸𑀕𑀢𑀺𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀦𑀺𑀳𑀺𑀢𑀸𑀦𑀺𑀳𑀺𑀢𑀜𑁆𑀘 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀧𑀞𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄; 𑀬𑀸𑀘𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1607\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑀻𑀯𑀭𑀦𑀺𑀤𑀳𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1608\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀘𑀻𑀯𑀭𑀦𑀺𑀤𑀳𑀓𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀘𑀻𑀯𑀭𑀦𑀺𑀤𑀳𑀓𑀲𑁆𑀲 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1609\. 𑀲𑀫𑁆𑀫𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘𑀻𑀯𑀭𑀦𑀺𑀤𑀳𑀓𑁄𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇
