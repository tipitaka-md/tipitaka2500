# 𑁮𑁇𑁭 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀓𑀣𑀸

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀓𑀣𑀸

(335.)

1583\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀧𑀚𑁆𑀚𑁄𑀢𑀲𑁆𑀲 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺—  𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀢𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀜𑁆𑀘 𑀲𑁂𑀝𑁆𑀞𑀜𑁆𑀘 𑀫𑁄𑀓𑁆𑀔𑀜𑁆𑀘 𑀉𑀢𑁆𑀢𑀫𑀜𑁆𑀘 𑀧𑀯𑀭𑀜𑁆𑀘𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀧𑀚𑁆𑀚𑁄𑀢𑁄 𑀢𑀁 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀧𑀸𑀳𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀤𑀁 𑀔𑁄 𑀫𑁂 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁 𑀭𑀜𑁆𑀜𑀸 𑀧𑀚𑁆𑀚𑁄𑀢𑁂𑀦 𑀧𑀳𑀺𑀢𑀁—  𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀢𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀜𑁆𑀘 𑀲𑁂𑀝𑁆𑀞𑀜𑁆𑀘 𑀫𑁄𑀓𑁆𑀔𑀜𑁆𑀘 𑀉𑀢𑁆𑀢𑀫𑀜𑁆𑀘 𑀧𑀯𑀭𑀜𑁆𑀘𑁇 𑀦𑀬𑀺𑀤𑀁 𑀅𑀜𑁆𑀜𑁄 𑀓𑁄𑀘𑀺 𑀧𑀘𑁆𑀘𑀸𑀭𑀳𑀢𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀅𑀭𑀳𑀢𑀸 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁂𑀦, 𑀭𑀜𑁆𑀜𑀸 𑀯𑀸 𑀫𑀸𑀕𑀥𑁂𑀦 𑀲𑁂𑀦𑀺𑀬𑁂𑀦 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁂𑀦𑀸”𑀢𑀺𑁇
