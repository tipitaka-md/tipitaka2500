# 𑁮𑁇𑁨𑁭 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

(372.)

1720\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀧𑀻𑀢𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀓𑀡𑁆𑀳𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀭𑀗𑁆𑀕𑀭𑀢𑁆𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀦𑀸𑀫𑀭𑀢𑁆𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀤𑀻𑀖𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀼𑀧𑁆𑀨𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀨𑀡𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀓𑀜𑁆𑀘𑀼𑀓𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀢𑀺𑀭𑀻𑀝𑀓𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀯𑁂𑀞𑀦𑀁 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀞𑀦𑀁 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀧𑀻𑀢𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀮𑁄𑀳𑀺𑀢𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀓𑀡𑁆𑀳𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀭𑀗𑁆𑀕𑀭𑀢𑁆𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀦𑀸𑀫𑀭𑀢𑁆𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀤𑀻𑀖𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀧𑀼𑀧𑁆𑀨𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀨𑀡𑀤𑀲𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀦 𑀓𑀜𑁆𑀘𑀼𑀓𑀁 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀢𑀺𑀭𑀻𑀝𑀓𑀁 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀯𑁂𑀞𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
