# 𑁮𑁇𑁯 𑀯𑀭𑀬𑀸𑀘𑀦𑀸𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀯𑀭𑀬𑀸𑀘𑀦𑀸𑀓𑀣𑀸

(337.)

1587\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑁄 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀁 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀓𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀭𑀁 𑀬𑀸𑀘𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀭𑀸 𑀔𑁄, 𑀚𑀻𑀯𑀓, 𑀢𑀣𑀸𑀕𑀢𑀸”𑀢𑀺𑁇 “𑀬𑀜𑁆𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀧𑁆𑀧𑀢𑀺 𑀬𑀜𑁆𑀘 𑀅𑀦𑀯𑀚𑁆𑀚𑀦𑁆”𑀢𑀺𑁇 “𑀯𑀤𑁂𑀳𑀺, 𑀚𑀻𑀯𑀓𑀸”𑀢𑀺𑁇 “𑀪𑀕𑀯𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀘𑁇 𑀇𑀤𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁 𑀭𑀜𑁆𑀜𑀸 𑀧𑀚𑁆𑀚𑁄𑀢𑁂𑀦 𑀧𑀳𑀺𑀢𑀁—  𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀢𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀳𑀽𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀕𑁆𑀕𑀜𑁆𑀘 𑀲𑁂𑀝𑁆𑀞𑀜𑁆𑀘 𑀫𑁄𑀓𑁆𑀔𑀜𑁆𑀘 𑀉𑀢𑁆𑀢𑀫𑀜𑁆𑀘 𑀧𑀯𑀭𑀜𑁆𑀘𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁; 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀘 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀲𑀺𑀯𑁂𑀬𑁆𑀬𑀓𑀁 𑀤𑀼𑀲𑁆𑀲𑀬𑀼𑀕𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁𑁇 𑀬𑁄 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀺𑀓𑁄 𑀳𑁄𑀢𑀼𑁇 𑀬𑁄 𑀇𑀘𑁆𑀙𑀢𑀺, 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀲𑀸𑀤𑀺𑀬𑀢𑀼𑁇 𑀇𑀢𑀭𑀻𑀢𑀭𑁂𑀦𑀧𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀁 𑀯𑀡𑁆𑀡𑁂𑀫𑀻”𑀢𑀺𑁇

1588\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  “𑀪𑀕𑀯𑀢𑀸 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀳𑀝𑁆𑀞𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀉𑀤𑀕𑁆𑀕𑀸—  “𑀇𑀤𑀸𑀦𑀺 𑀔𑁄 𑀫𑀬𑀁 𑀤𑀸𑀦𑀸𑀦𑀺 𑀤𑀲𑁆𑀲𑀸𑀫 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫, 𑀬𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀸𑀳𑁂𑀦𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀩𑀳𑀽𑀦𑀺 𑀘𑀻𑀯𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼𑁇

1589\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀚𑀸𑀦𑀧𑀤𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  “𑀪𑀕𑀯𑀢𑀸 𑀓𑀺𑀭 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀳𑀝𑁆𑀞𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀉𑀤𑀕𑁆𑀕𑀸—  “𑀇𑀤𑀸𑀦𑀺 𑀔𑁄 𑀫𑀬𑀁 𑀤𑀸𑀦𑀸𑀦𑀺 𑀤𑀲𑁆𑀲𑀸𑀫 𑀧𑀼𑀜𑁆𑀜𑀸𑀦𑀺 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫, 𑀬𑀢𑁄 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺𑁇 𑀚𑀦𑀧𑀤𑁂𑀧𑀺 𑀏𑀓𑀸𑀳𑁂𑀦𑁂𑀯 𑀩𑀳𑀽𑀦𑀺 𑀘𑀻𑀯𑀭𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀁𑀲𑀼𑁇

1590\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀸𑀯𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀯𑀸𑀭𑀦𑁆”𑀢𑀺𑁇

1591\. 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀧𑀸𑀯𑀸𑀭𑁄 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀧𑀸𑀯𑀸𑀭𑀦𑁆”𑀢𑀺𑁇

1592\. 𑀓𑁄𑀚𑀯𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑁄𑀚𑀯𑀦𑁆”𑀢𑀺𑁇

---

1593\. 𑀧𑀞𑀫𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
