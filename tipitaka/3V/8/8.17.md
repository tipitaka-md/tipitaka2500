# 𑁮𑁇𑁧𑁭 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀓𑀣𑀸

(347.)

1646\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺—  𑀅𑀜𑁆𑀜𑁂𑀦𑁂𑀯 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀕𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀆𑀭𑀸𑀫𑁂 𑀅𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀅𑀜𑁆𑀜𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀦𑀳𑀸𑀦𑀁 𑀑𑀢𑀭𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

1647\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀓𑁂𑀢𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀇𑀤𑀜𑁆𑀘 𑀫𑁂 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀅𑀳𑀜𑁆𑀘𑀺𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀓𑁂𑀢𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀓𑀻𑀯𑀘𑀺𑀭𑀁 𑀧𑀦𑀸𑀦𑀦𑁆𑀤, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀆𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀦𑀯𑀫𑀁 𑀯𑀸, 𑀪𑀕𑀯𑀸, 𑀤𑀺𑀯𑀲𑀁, 𑀤𑀲𑀫𑀁 𑀯𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀲𑀸𑀳𑀧𑀭𑀫𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀥𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1648\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑁂 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀢𑀺𑀭𑁂𑀓𑀘𑀻𑀯𑀭𑀁 𑀯𑀺𑀓𑀧𑁆𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(348.)

1649\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁄 𑀙𑀺𑀤𑁆𑀤𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁—  𑀤𑀺𑀕𑀼𑀡𑀸 𑀲𑀗𑁆𑀖𑀸𑀝𑀺, 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑁄 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑁄, 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑁄 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁄𑁇 𑀅𑀬𑀜𑁆𑀘 𑀫𑁂 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁄 𑀙𑀺𑀤𑁆𑀤𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀅𑀘𑁆𑀙𑀼𑀧𑁂𑀬𑁆𑀬𑀁, 𑀲𑀫𑀦𑁆𑀢𑀢𑁄 𑀤𑀼𑀧𑀝𑁆𑀝𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀚𑁆𑀛𑁂 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀅𑀘𑁆𑀙𑀼𑀧𑁂𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀅𑀘𑁆𑀙𑀼𑀧𑁂𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀓𑀭𑁄𑀲𑀻”𑀢𑀺? “𑀅𑀕𑁆𑀕𑀍𑀅𑀁, 𑀪𑀕𑀯𑀸, 𑀅𑀘𑁆𑀙𑀼𑀧𑁂𑀫𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀼; 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀅𑀘𑁆𑀙𑀼𑀧𑁂𑀲𑀻”𑀢𑀺𑁇

1650\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀳𑀢𑀸𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀳𑀢𑀓𑀧𑁆𑀧𑀸𑀦𑀁 𑀤𑀺𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁, 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁, 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁; 𑀉𑀢𑀼𑀤𑁆𑀥𑀝𑀸𑀦𑀁 𑀤𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀘𑀢𑀼𑀕𑁆𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁, 𑀤𑀺𑀕𑀼𑀡𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁, 𑀤𑀺𑀕𑀼𑀡𑀁 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀁; 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑁂 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁; 𑀧𑀸𑀧𑀡𑀺𑀓𑁂 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀢𑀼𑀦𑁆𑀦𑀁 𑀑𑀯𑀝𑁆𑀝𑀺𑀓𑀁 𑀓𑀡𑁆𑀟𑀼𑀲𑀓𑀁 𑀤𑀍𑀳𑀻𑀓𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇
