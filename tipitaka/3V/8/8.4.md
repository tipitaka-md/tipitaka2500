# 𑁮𑁇𑁪 𑀭𑀸𑀚𑀕𑀳𑀲𑁂𑀝𑁆𑀞𑀺𑀯𑀢𑁆𑀣𑀼

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀭𑀸𑀚𑀕𑀳𑀲𑁂𑀝𑁆𑀞𑀺𑀯𑀢𑁆𑀣𑀼

(332.)

1570\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀓𑀲𑁆𑀲 𑀲𑁂𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀲𑀢𑁆𑀢𑀯𑀲𑁆𑀲𑀺𑀓𑁄 𑀲𑀻𑀲𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀩𑀳𑀽 𑀫𑀳𑀦𑁆𑀢𑀸 𑀫𑀳𑀦𑁆𑀢𑀸 𑀤𑀺𑀲𑀸𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑀸 𑀯𑁂𑀚𑁆𑀚𑀸 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀦𑀸𑀲𑀓𑁆𑀔𑀺𑀁𑀲𑀼 𑀅𑀭𑁄𑀕𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀩𑀳𑀼𑀁 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀆𑀤𑀸𑀬 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀅𑀧𑀺 𑀘 𑀯𑁂𑀚𑁆𑀚𑁂𑀳𑀺 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑁂𑀚𑁆𑀚𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑁂𑀚𑁆𑀚𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀲𑀢𑁆𑀢𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑀓𑀲𑁆𑀲 𑀦𑁂𑀕𑀫𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀬𑀁 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀭𑀜𑁆𑀜𑁄 𑀘𑁂𑀯 𑀦𑁂𑀕𑀫𑀲𑁆𑀲 𑀘𑁇 𑀅𑀧𑀺 𑀘 𑀯𑁂𑀚𑁆𑀚𑁂𑀳𑀺 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑁂𑀚𑁆𑀚𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑁂𑀚𑁆𑀚𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀢𑁆𑀢𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘 𑀭𑀜𑁆𑀜𑁄 𑀚𑀻𑀯𑀓𑁄 𑀯𑁂𑀚𑁆𑀚𑁄 𑀢𑀭𑀼𑀡𑁄 𑀪𑀤𑁆𑀭𑀓𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀭𑀸𑀚𑀸𑀦𑀁 𑀚𑀻𑀯𑀓𑀁 𑀯𑁂𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑁂𑀬𑁆𑀬𑀸𑀫 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1571\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀕𑀳𑀓𑁄 𑀦𑁂𑀕𑀫𑁄 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀤𑁂𑀯, 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑁄 𑀤𑁂𑀯𑀲𑁆𑀲 𑀘𑁂𑀯 𑀦𑁂𑀕𑀫𑀲𑁆𑀲 𑀘; 𑀅𑀧𑀺 𑀘 𑀯𑁂𑀚𑁆𑀚𑁂𑀳𑀺 𑀧𑀘𑁆𑀘𑀓𑁆𑀔𑀸𑀢𑁄𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑁂𑀚𑁆𑀚𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑁂𑀚𑁆𑀚𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀢𑁆𑀢𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀤𑁂𑀯𑁄 𑀚𑀻𑀯𑀓𑀁 𑀯𑁂𑀚𑁆𑀚𑀁 𑀆𑀡𑀸𑀧𑁂𑀢𑀼 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1572\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓, 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀯𑀺𑀓𑀸𑀭𑀁 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑁂𑀢𑁆𑀯𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀭𑁄𑀕𑁄 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀓𑀺𑀁 𑀫𑁂 𑀅𑀲𑁆𑀲 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄”𑀢𑀺? “𑀲𑀩𑁆𑀩𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀢𑁂, 𑀆𑀘𑀭𑀺𑀬, 𑀳𑁄𑀢𑀼, 𑀅𑀳𑀜𑁆𑀘 𑀢𑁂 𑀤𑀸𑀲𑁄”𑀢𑀺𑁇 “𑀲𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀏𑀓𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀓𑁆𑀓𑁄𑀫𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀏𑀓𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀓𑁆𑀓𑁄𑀫𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀓𑁆𑀓𑁄𑀫𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1573\. 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀫𑀜𑁆𑀘𑀓𑁂 𑀦𑀺𑀧𑀸𑀢𑁂𑀢𑁆𑀯𑀸 𑀫𑀜𑁆𑀘𑀓𑁂 𑀲𑀫𑁆𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀲𑀻𑀲𑀘𑁆𑀙𑀯𑀺𑀁 𑀉𑀧𑁆𑀧𑀸𑀝𑁂𑀢𑁆𑀯𑀸 𑀲𑀺𑀩𑁆𑀩𑀺𑀦𑀺𑀁 𑀯𑀺𑀦𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀤𑁆𑀯𑁂 𑀧𑀸𑀡𑀓𑁂 𑀦𑀻𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀳𑀸𑀚𑀦𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑁂𑀲𑀺—  “𑀧𑀲𑁆𑀲𑀣𑀬𑁆𑀬𑁂, 𑀇𑀫𑁂 𑀤𑁆𑀯𑁂 𑀧𑀸𑀡𑀓𑁂, 𑀏𑀓𑀁 𑀔𑀼𑀤𑁆𑀤𑀓𑀁 𑀏𑀓𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀆𑀘𑀭𑀺𑀬𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺—  𑀢𑁂𑀳𑀸𑀬𑀁 𑀫𑀳𑀮𑁆𑀮𑀓𑁄 𑀧𑀸𑀡𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑁄𑁇 𑀧𑀜𑁆𑀘𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀫𑀢𑁆𑀣𑀮𑀼𑀗𑁆𑀕𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀫𑀢𑁆𑀣𑀮𑀼𑀗𑁆𑀕𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄 𑀢𑁂𑀳𑀺 𑀆𑀘𑀭𑀺𑀬𑁂𑀳𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀆𑀘𑀭𑀺𑀬𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀲𑀢𑁆𑀢𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺—  𑀢𑁂𑀳𑀸𑀬𑀁 𑀔𑀼𑀤𑁆𑀤𑀓𑁄 𑀧𑀸𑀡𑀓𑁄 𑀤𑀺𑀝𑁆𑀞𑁄𑁇 𑀲𑀢𑁆𑀢𑀫𑀁 𑀤𑀺𑀯𑀲𑀁 𑀲𑁂𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀫𑀢𑁆𑀣𑀮𑀼𑀗𑁆𑀕𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀫𑀢𑁆𑀣𑀮𑀼𑀗𑁆𑀕𑀲𑁆𑀲 𑀧𑀭𑀺𑀬𑀸𑀤𑀸𑀦𑀸 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀓𑀸𑀮𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀼𑀤𑀺𑀝𑁆𑀞𑁄 𑀢𑁂𑀳𑀺 𑀆𑀘𑀭𑀺𑀬𑁂𑀳𑀻”𑀢𑀺𑁇 𑀲𑀺𑀩𑁆𑀩𑀺𑀦𑀺𑀁 𑀲𑀫𑁆𑀧𑀝𑀺𑀧𑀸𑀝𑁂𑀢𑁆𑀯𑀸 𑀲𑀻𑀲𑀘𑁆𑀙𑀯𑀺𑀁 𑀲𑀺𑀩𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀆𑀮𑁂𑀧𑀁 𑀅𑀤𑀸𑀲𑀺𑁇

1574\. 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀸𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀏𑀓𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺—  𑀲𑀓𑁆𑀓𑁄𑀫𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀏𑀓𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀸𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀁, 𑀅𑀧𑀸𑀳𑀁 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀏𑀓𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀳𑀻”𑀢𑀺𑁇

1575\. 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀸𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺—  𑀲𑀓𑁆𑀓𑁄𑀫𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀸𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀁, 𑀅𑀧𑀸𑀳𑀁 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀸𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀤𑀼𑀢𑀺𑀬𑁂𑀦 𑀧𑀲𑁆𑀲𑁂𑀦 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀸𑀳𑀻”𑀢𑀺𑁇

1576\. 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀸𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀫𑁂 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺—  𑀲𑀓𑁆𑀓𑁄𑀫𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀸𑀳𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀁, 𑀅𑀧𑀸𑀳𑀁 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀉𑀢𑁆𑀢𑀸𑀦𑁄 𑀲𑀢𑁆𑀢𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀳𑀜𑁆𑀘𑁂 𑀢𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀦 𑀯𑀤𑁂𑀬𑁆𑀬𑀁, 𑀏𑀢𑁆𑀢𑀓𑀫𑁆𑀧𑀺 𑀢𑁆𑀯𑀁 𑀦 𑀦𑀺𑀧𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀲𑀺, 𑀅𑀧𑀺 𑀘 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀫𑀬𑀸 𑀜𑀸𑀢𑁄—  𑀢𑀻𑀳𑀺 𑀲𑀢𑁆𑀢𑀸𑀳𑁂𑀳𑀺 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺𑁇 𑀉𑀝𑁆𑀞𑁂𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀅𑀭𑁄𑀕𑁄𑀲𑀺𑁇 𑀚𑀸𑀦𑀸𑀲𑀺 𑀓𑀺𑀁 𑀫𑁂 𑀤𑁂𑀬𑁆𑀬𑀥𑀫𑁆𑀫𑁄”𑀢𑀺? “𑀲𑀩𑁆𑀩𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀢𑁂, 𑀆𑀘𑀭𑀺𑀬, 𑀳𑁄𑀢𑀼, 𑀅𑀳𑀜𑁆𑀘 𑀢𑁂 𑀤𑀸𑀲𑁄”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀫𑀸 𑀫𑁂 𑀢𑁆𑀯𑀁 𑀲𑀩𑁆𑀩𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀁 𑀅𑀤𑀸𑀲𑀺, 𑀫𑀸 𑀘 𑀫𑁂 𑀤𑀸𑀲𑁄𑁇 𑀭𑀜𑁆𑀜𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀤𑁂𑀳𑀺, 𑀫𑀬𑁆𑀳𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀅𑀭𑁄𑀕𑁄 𑀲𑀫𑀸𑀦𑁄 𑀭𑀜𑁆𑀜𑁄 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁 𑀅𑀤𑀸𑀲𑀺, 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑀁𑁇
