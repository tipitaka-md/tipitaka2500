# 𑁮𑁇𑁧𑁪 𑀘𑀻𑀯𑀭𑀭𑀚𑀦𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀘𑀻𑀯𑀭𑀭𑀚𑀦𑀓𑀣𑀸

(344.)

1629\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀓𑀡𑁂𑀦𑀧𑀺 𑀧𑀡𑁆𑀟𑀼𑀫𑀢𑁆𑀢𑀺𑀓𑀸𑀬𑀧𑀺 𑀘𑀻𑀯𑀭𑀁 𑀭𑀚𑀦𑁆𑀢𑀺𑁇 𑀘𑀻𑀯𑀭𑀁 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀭𑀚𑀦𑀸𑀦𑀺—  𑀫𑀽𑀮𑀭𑀚𑀦𑀁, 𑀔𑀦𑁆𑀥𑀭𑀚𑀦𑀁, 𑀢𑀘𑀭𑀚𑀦𑀁, 𑀧𑀢𑁆𑀢𑀭𑀚𑀦𑀁, 𑀧𑀼𑀧𑁆𑀨𑀭𑀚𑀦𑀁, 𑀨𑀮𑀭𑀚𑀦𑀦𑁆”𑀢𑀺𑁇

1630\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀢𑀼𑀤𑀓𑀸𑀬 𑀘𑀻𑀯𑀭𑀁 𑀭𑀚𑀦𑁆𑀢𑀺𑁇 𑀘𑀻𑀯𑀭𑀁 𑀤𑀼𑀕𑁆𑀕𑀦𑁆𑀥𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀚𑀦𑀁 𑀧𑀘𑀺𑀢𑀼𑀁 𑀘𑀼𑀮𑁆𑀮𑀁 𑀭𑀚𑀦𑀓𑀼𑀫𑁆𑀪𑀺𑀦𑁆”𑀢𑀺𑁇 𑀭𑀚𑀦𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀬𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀢𑁆𑀢𑀭𑀸𑀍𑀉𑀫𑁆𑀧𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1631\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀚𑀸𑀦𑀦𑁆𑀢𑀺 𑀭𑀚𑀦𑀁 𑀧𑀓𑁆𑀓𑀁 𑀯𑀸 𑀅𑀧𑀓𑁆𑀓𑀁 𑀯𑀸𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀓𑁂 𑀯𑀸 𑀦𑀔𑀧𑀺𑀝𑁆𑀞𑀺𑀓𑀸𑀬 𑀯𑀸 𑀣𑁂𑀯𑀓𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1632\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀚𑀦𑀁 𑀑𑀭𑁄𑀧𑁂𑀦𑁆𑀢𑀸 𑀓𑀼𑀫𑁆𑀪𑀺𑀁 𑀆𑀯𑀺𑀜𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀓𑀼𑀫𑁆𑀪𑀻 𑀪𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀚𑀦𑀼𑀍𑀉𑀗𑁆𑀓𑀁 𑀤𑀡𑁆𑀟𑀓𑀣𑀸𑀮𑀓𑀦𑁆”𑀢𑀺𑁇

1633\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀭𑀚𑀦𑀪𑀸𑀚𑀦𑀁 𑀦 𑀲𑀁𑀯𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀚𑀦𑀓𑁄𑀮𑀫𑁆𑀩𑀁 𑀭𑀚𑀦𑀖𑀝𑀦𑁆”𑀢𑀺𑁇

1634\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀢𑀺𑀬𑀸𑀧𑀺 𑀧𑀢𑁆𑀢𑁂𑀧𑀺 𑀘𑀻𑀯𑀭𑀁 𑀑𑀫𑀤𑁆𑀤𑀦𑁆𑀢𑀺𑁇 𑀘𑀻𑀯𑀭𑀁 𑀧𑀭𑀺𑀪𑀺𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀚𑀦𑀤𑁄𑀡𑀺𑀓𑀦𑁆”𑀢𑀺𑁇

1635\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀫𑀸𑀬 𑀘𑀻𑀯𑀭𑀁 𑀧𑀢𑁆𑀣𑀭𑀦𑁆𑀢𑀺𑁇 𑀘𑀻𑀯𑀭𑀁 𑀧𑀁𑀲𑀼𑀓𑀺𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑀓𑀦𑁆”𑀢𑀺𑁇

1636\. 𑀢𑀺𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑀓𑁄 𑀉𑀧𑀘𑀺𑀓𑀸𑀳𑀺 𑀔𑀚𑁆𑀚𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀻𑀯𑀭𑀯𑀁𑀲𑀁 𑀘𑀻𑀯𑀭𑀭𑀚𑁆𑀚𑀼𑀦𑁆”𑀢𑀺𑁇

1637\. 𑀫𑀚𑁆𑀛𑁂𑀦 𑀮𑀕𑁆𑀕𑁂𑀦𑁆𑀢𑀺𑁇 𑀭𑀚𑀦𑀁 𑀉𑀪𑀢𑁄 𑀕𑀮𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀡𑁆𑀡𑁂 𑀩𑀦𑁆𑀥𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1638\. 𑀓𑀡𑁆𑀡𑁄 𑀚𑀻𑀭𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀡𑁆𑀡𑀲𑀼𑀢𑁆𑀢𑀓𑀦𑁆”𑀢𑀺𑁇

1639\. 𑀭𑀚𑀦𑀁 𑀏𑀓𑀢𑁄 𑀕𑀮𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁 𑀲𑀫𑁆𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀓𑀁 𑀭𑀚𑁂𑀢𑀼𑀁, 𑀦 𑀘 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁂 𑀣𑁂𑀯𑁂 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1640\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀧𑀢𑁆𑀣𑀺𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀤𑀓𑁂 𑀑𑀲𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1641\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀻𑀯𑀭𑀁 𑀨𑀭𑀼𑀲𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀡𑀺𑀦𑀸 𑀆𑀓𑁄𑀝𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1642\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺 𑀤𑀦𑁆𑀢𑀓𑀸𑀲𑀸𑀯𑀸𑀦𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
