# 𑁮𑁇𑁧𑁮 𑀯𑀺𑀲𑀸𑀔𑀸𑀯𑀢𑁆𑀣𑀼

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀯𑀺𑀲𑀸𑀔𑀸𑀯𑀢𑁆𑀣𑀼

(349.)

1651\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1652\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀻𑀧𑀺𑀓𑁄 𑀫𑀳𑀸𑀫𑁂𑀖𑁄 𑀧𑀸𑀯𑀲𑁆𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀣𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑁂𑀢𑀯𑀦𑁂 𑀯𑀲𑁆𑀲𑀢𑀺 𑀏𑀯𑀁 𑀘𑀢𑀽𑀲𑀼 𑀤𑀻𑀧𑁂𑀲𑀼 𑀯𑀲𑁆𑀲𑀢𑀺𑁇 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀬𑀁𑁇 𑀅𑀬𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀓𑁄 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀻𑀧𑀺𑀓𑁄 𑀫𑀳𑀸𑀫𑁂𑀖𑁄”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇

1653\. 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀺𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙, 𑀚𑁂𑁇 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀺—  𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀲𑀸 𑀤𑀸𑀲𑀻 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀤𑁆𑀤𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑁂 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦—  “𑀦𑀢𑁆𑀣𑀺 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀆𑀚𑀻𑀯𑀓𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀬𑁂𑀦 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀢𑁆𑀣𑀬𑁆𑀬𑁂, 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀆𑀚𑀻𑀯𑀓𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑀬 𑀯𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑀺𑀬𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀅𑀬𑁆𑀬𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀬𑀁 𑀩𑀸𑀮𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑁆𑀣—  𑀦𑀢𑁆𑀣𑀺 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀆𑀚𑀻𑀯𑀓𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀧𑀼𑀦 𑀤𑀸𑀲𑀺𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙, 𑀚𑁂𑁇 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀺—  𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

1654\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀻𑀢𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀮𑁆𑀮𑀓𑀸𑀬𑀸 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀤𑀸𑀲𑀻 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻—  “𑀦𑀢𑁆𑀣𑀺 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀲𑀼𑀜𑁆𑀜𑁄 𑀆𑀭𑀸𑀫𑁄”𑀢𑀺—  𑀬𑁂𑀦 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀢𑁆𑀣𑀬𑁆𑀬𑁂, 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀲𑀼𑀜𑁆𑀜𑁄 𑀆𑀭𑀸𑀫𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑀬 𑀯𑀺𑀬𑀢𑁆𑀢𑀸𑀬 𑀫𑁂𑀥𑀸𑀯𑀺𑀦𑀺𑀬𑀸 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀅𑀬𑁆𑀬𑀸 𑀕𑀢𑁆𑀢𑀸𑀦𑀺 𑀲𑀻𑀢𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀮𑁆𑀮𑀓𑀸𑀬𑀸 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑀣𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀸𑁇 𑀲𑀸𑀬𑀁 𑀩𑀸𑀮𑀸 𑀫𑀜𑁆𑀜𑀺𑀢𑁆𑀣—  𑀦𑀢𑁆𑀣𑀺 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀲𑀼𑀜𑁆𑀜𑁄 𑀆𑀭𑀸𑀫𑁄”𑀢𑀺, 𑀧𑀼𑀦 𑀤𑀸𑀲𑀺𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙, 𑀚𑁂𑁇 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀺—  𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

1655\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀦𑁆𑀤𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁; 𑀓𑀸𑀮𑁄 𑀪𑀢𑁆𑀢𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀸, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀚𑀡𑁆𑀡𑀼𑀓𑀫𑀢𑁆𑀢𑁂𑀲𑀼𑀧𑀺 𑀑𑀖𑁂𑀲𑀼 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂𑀲𑀼, 𑀓𑀝𑀺𑀫𑀢𑁆𑀢𑁂𑀲𑀼𑀧𑀺 𑀑𑀖𑁂𑀲𑀼 𑀧𑀯𑀢𑁆𑀢𑀫𑀸𑀦𑁂𑀲𑀼, 𑀦 𑀳𑀺 𑀦𑀸𑀫 𑀏𑀓𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲𑀧𑀺 𑀧𑀸𑀤𑀸 𑀯𑀸 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀯𑀸 𑀅𑀮𑁆𑀮𑀸𑀦𑀺 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀳𑀝𑁆𑀞𑀸 𑀉𑀤𑀕𑁆𑀕𑀸 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀝𑁆𑀞𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀭𑀸 𑀔𑁄, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀢𑀣𑀸𑀕𑀢𑀸”𑀢𑀺𑁇 “𑀬𑀸𑀦𑀺 𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀧𑁆𑀧𑀺𑀬𑀸𑀦𑀺 𑀬𑀸𑀦𑀺 𑀘 𑀅𑀦𑀯𑀚𑁆𑀚𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀯𑀤𑁂𑀳𑀺, 𑀯𑀺𑀲𑀸𑀔𑁂”𑀢𑀺𑁇 “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀕𑀫𑀺𑀓𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀥𑀼𑀯𑀬𑀸𑀕𑀼𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀝𑁆𑀞 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀲𑀻”𑀢𑀺?

1656\. “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀸𑀲𑀺𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺𑀁—  ‘𑀕𑀘𑁆𑀙, 𑀚𑁂𑁇 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀺—  𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀸𑀲𑀻 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀤𑁆𑀤𑀲 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑁂 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦—  ‘𑀦𑀢𑁆𑀣𑀺 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀆𑀚𑀻𑀯𑀓𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀻’𑀢𑀺—  𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀦𑀢𑁆𑀣𑀬𑁆𑀬𑁂, 𑀆𑀭𑀸𑀫𑁂 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀆𑀚𑀻𑀯𑀓𑀸 𑀓𑀸𑀬𑀁 𑀑𑀯𑀲𑁆𑀲𑀸𑀧𑁂𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀅𑀲𑀼𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀕𑁆𑀕𑀺𑀬𑀁 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀁 𑀧𑀝𑀺𑀓𑀽𑀮𑀁𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀤𑀸𑀢𑀼𑀁𑁇

1657\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀦 𑀯𑀻𑀣𑀺𑀓𑀼𑀲𑀮𑁄 𑀦 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺𑁇 𑀲𑁄 𑀫𑁂 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀻𑀣𑀺𑀓𑀼𑀲𑀮𑁄 𑀕𑁄𑀘𑀭𑀓𑀼𑀲𑀮𑁄 𑀅𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁𑁇

1658\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀫𑀺𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀲𑀢𑁆𑀣𑀸 𑀯𑀸 𑀯𑀺𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀯𑀸𑀲𑀁 𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀢𑁆𑀣 𑀯𑀺𑀓𑀸𑀮𑁂 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁄 𑀫𑁂 𑀕𑀫𑀺𑀓𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀣𑀸 𑀦 𑀯𑀺𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀯𑀸𑀲𑀁 𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑀢𑁆𑀣 𑀓𑀸𑀮𑁂 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀓𑀺𑀮𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀁 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀕𑀫𑀺𑀓𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁𑁇

1659\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀺 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀅𑀮𑀪𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀩𑀸𑀥𑁄 𑀯𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀸𑀮𑀁𑀓𑀺𑀭𑀺𑀬𑀸 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀩𑀸𑀥𑁄 𑀦 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀸𑀮𑀁𑀓𑀺𑀭𑀺𑀬𑀸 𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁𑁇

1660\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀢𑁆𑀢𑀦𑁄 𑀪𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀫𑀸𑀦𑁄 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀉𑀲𑁆𑀲𑀽𑀭𑁂 𑀪𑀢𑁆𑀢𑀁 𑀦𑀻𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀢𑁆𑀢𑀘𑁆𑀙𑁂𑀤𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁄 𑀫𑁂 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀓𑀸𑀮𑁂𑀦 𑀪𑀢𑁆𑀢𑀁 𑀦𑀻𑀳𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀪𑀢𑁆𑀢𑀘𑁆𑀙𑁂𑀤𑀁 𑀦 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀤𑀸𑀢𑀼𑀁𑁇

1661\. 𑀧𑀼𑀦 𑀘𑀧𑀭𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀅𑀮𑀪𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀩𑀸𑀥𑁄 𑀯𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀸𑀮𑀁𑀓𑀺𑀭𑀺𑀬𑀸 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀩𑀸𑀥𑁄 𑀦 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀸𑀮𑀁𑀓𑀺𑀭𑀺𑀬𑀸 𑀦 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀤𑀸𑀢𑀼𑀁𑁇

1663\. 𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀅𑀘𑀺𑀭𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀯𑁂𑀲𑀺𑀬𑀸𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀦𑀕𑁆𑀕𑀸 𑀏𑀓𑀢𑀺𑀢𑁆𑀣𑁂 𑀦𑀳𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀲𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑁂𑀲𑀼𑀁—  ‘𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀦𑀸𑀫 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁, 𑀅𑀬𑁆𑀬𑁂, 𑀤𑀳𑀭𑀸𑀦𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀺𑀡𑁆𑀡𑁂𑀦, 𑀦𑀦𑀼 𑀦𑀸𑀫 𑀓𑀸𑀫𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸; 𑀬𑀤𑀸 𑀚𑀺𑀡𑁆𑀡𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀣 𑀢𑀤𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀣𑁇 𑀏𑀯𑀁 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀉𑀪𑁄 𑀅𑀢𑁆𑀣𑀸 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻’𑀢𑀺𑁇 𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀯𑁂𑀲𑀺𑀬𑀸𑀳𑀺 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀺𑀬𑀫𑀸𑀦𑀸 𑀫𑀗𑁆𑀓𑀽 𑀅𑀳𑁂𑀲𑀼𑀁𑁇 𑀅𑀲𑀼𑀘𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀸𑀢𑀼𑀕𑀸𑀫𑀲𑁆𑀲 𑀦𑀕𑁆𑀕𑀺𑀬𑀁 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑀁 𑀧𑀝𑀺𑀓𑀽𑀮𑀁𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀢𑁆𑀣𑀯𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(351.)

1664\. “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀝𑁆𑀞 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀲𑀻”𑀢𑀺? “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑀸𑀲𑀼 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀆𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀢𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀢𑀲𑁆𑀲 𑀓𑀸 𑀕𑀢𑀺 𑀓𑁄 𑀅𑀪𑀺𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑁄’𑀢𑀺? 𑀢𑀁 𑀪𑀕𑀯𑀸 𑀩𑁆𑀬𑀸𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑁄𑀢𑀸𑀧𑀢𑁆𑀢𑀺𑀨𑀮𑁂 𑀯𑀸 𑀲𑀓𑀤𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑁂 𑀯𑀸 𑀅𑀦𑀸𑀕𑀸𑀫𑀺𑀨𑀮𑁂 𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑁂 𑀯𑀸𑁇 𑀢𑁆𑀬𑀸𑀳𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀆𑀕𑀢𑀧𑀼𑀩𑁆𑀩𑀸 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑁂𑀦 𑀅𑀬𑁆𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀻’𑀢𑀺? 𑀲𑀘𑁂 𑀫𑁂 𑀯𑀓𑁆𑀔𑀦𑁆𑀢𑀺—  ‘𑀆𑀕𑀢𑀧𑀼𑀩𑁆𑀩𑀸 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀻’𑀢𑀺 𑀦𑀺𑀝𑁆𑀞𑀫𑁂𑀢𑁆𑀣 𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺—  ‘𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀁 𑀢𑁂𑀦 𑀅𑀬𑁆𑀬𑁂𑀦 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀸 𑀯𑀸 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀕𑀫𑀺𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀥𑀼𑀯𑀬𑀸𑀕𑀼 𑀯𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀢𑀤𑀦𑀼𑀲𑁆𑀲𑀭𑀦𑁆𑀢𑀺𑀬𑀸 𑀧𑀸𑀫𑀼𑀚𑁆𑀚𑀁 𑀚𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀫𑀼𑀤𑀺𑀢𑀸𑀬 𑀧𑀻𑀢𑀺 𑀚𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀻𑀢𑀺𑀫𑀦𑀸𑀬 𑀓𑀸𑀬𑁄 𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀲𑁆𑀲𑀤𑁆𑀥𑀓𑀸𑀬𑀸 𑀲𑀼𑀔𑀁 𑀯𑁂𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀲𑀼𑀔𑀺𑀦𑀺𑀬𑀸 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀫𑀸𑀥𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀸 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀪𑀸𑀯𑀦𑀸 𑀩𑀮𑀪𑀸𑀯𑀦𑀸 𑀩𑁄𑀚𑁆𑀛𑀗𑁆𑀕𑀪𑀸𑀯𑀦𑀸𑁇 𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀝𑁆𑀞 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀯𑀺𑀲𑀸𑀔𑁂; 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀇𑀫𑀁 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀫𑀸𑀦𑀸 𑀢𑀣𑀸𑀕𑀢𑀁 𑀅𑀝𑁆𑀞 𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀘𑀲𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺 𑀢𑁂, 𑀯𑀺𑀲𑀸𑀔𑁂, 𑀅𑀝𑁆𑀞 𑀯𑀭𑀸𑀦𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺—

1665\. _“𑀬𑀸 𑀅𑀦𑁆𑀦𑀧𑀸𑀦𑀁 𑀤𑀤𑀢𑀺𑀧𑁆𑀧𑀫𑁄𑀤𑀺𑀢𑀸,_  
_𑀲𑀻𑀮𑀽𑀧𑀧𑀦𑁆𑀦𑀸 𑀲𑀼𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀯𑀺𑀓𑀸;_  
_𑀤𑀤𑀸𑀢𑀺 𑀤𑀸𑀦𑀁 𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀫𑀘𑁆𑀙𑀭𑀁,_  
_𑀲𑁄𑀯𑀕𑁆𑀕𑀺𑀓𑀁 𑀲𑁄𑀓𑀦𑀼𑀤𑀁 𑀲𑀼𑀔𑀸𑀯𑀳𑀁𑁇_  


1666\. _𑀤𑀺𑀩𑁆𑀩𑀁 𑀲𑀸 𑀮𑀪𑀢𑁂 𑀆𑀬𑀼𑀁,_  
_𑀆𑀕𑀫𑁆𑀫 𑀫𑀕𑁆𑀕𑀁 𑀯𑀺𑀭𑀚𑀁 𑀅𑀦𑀗𑁆𑀕𑀡𑀁;_  
_𑀲𑀸 𑀧𑀼𑀜𑁆𑀜𑀓𑀸𑀫𑀸 𑀲𑀼𑀔𑀺𑀦𑀻 𑀅𑀦𑀸𑀫𑀬𑀸,_  
_𑀲𑀕𑁆𑀕𑀫𑁆𑀳𑀺 𑀓𑀸𑀬𑀫𑁆𑀳𑀺 𑀘𑀺𑀭𑀁 𑀧𑀫𑁄𑀤𑀢𑀻”𑀢𑀺𑁇_  


(352.)

1667\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀲𑀸𑀔𑀁 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀭𑀁 𑀇𑀫𑀸𑀳𑀺 𑀕𑀸𑀣𑀸𑀳𑀺 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀺𑀓𑀲𑀸𑀝𑀺𑀓𑀁, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀪𑀢𑁆𑀢𑀁, 𑀕𑀫𑀺𑀓𑀪𑀢𑁆𑀢𑀁, 𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁, 𑀥𑀼𑀯𑀬𑀸𑀕𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀤𑀓𑀲𑀸𑀝𑀺𑀓𑀦𑁆”𑀢𑀺𑁇
