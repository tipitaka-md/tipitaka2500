# 𑁮𑁇𑁨𑁪 𑀫𑀢𑀲𑀦𑁆𑀢𑀓𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀫𑀢𑀲𑀦𑁆𑀢𑀓𑀓𑀣𑀸

(367.)

1707. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀆𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀢𑀢𑁆𑀣 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀦𑀁 𑀯𑀡𑁆𑀡𑀺𑀢𑀁𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑁂𑀫𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀢𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑀲𑀼𑁇 𑀲𑁄 𑀢𑁂𑀳𑀺 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀲𑀁𑀖𑁄 𑀲𑀸𑀫𑀻 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑁂, 𑀅𑀧𑀺 𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀸𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘𑀸’𑀢𑀺𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1708. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1709. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘𑁇 𑀲𑀁𑀖𑁄 𑀇𑀫𑀁 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀫𑀲𑁆𑀲 𑀢𑀺𑀘𑀻𑀯𑀭𑀲𑁆𑀲 𑀘 𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1710. 𑀤𑀺𑀦𑁆𑀦𑀁 𑀇𑀤𑀁 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(368.)

1711. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀲𑀁𑀖𑁄 𑀲𑀸𑀫𑀻 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑁂, 𑀅𑀧𑀺 𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀸𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀢𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄, 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘𑀸’𑀢𑀺𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1712. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1713. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄𑁇 𑀇𑀤𑀁 𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘𑁇 𑀲𑀁𑀖𑁄 𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑁂𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀫𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀲𑁆𑀲 𑀘 𑀧𑀢𑁆𑀢𑀲𑁆𑀲 𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑀸𑀦𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1714. 𑀤𑀺𑀦𑁆𑀦𑀁 𑀇𑀤𑀁 𑀲𑀁𑀖𑁂𑀦 𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑁄 𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(369.)

1715. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀘 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀘 𑀕𑀺𑀮𑀸𑀦𑀁 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑀲𑀼𑁇 𑀲𑁄 𑀢𑁂𑀳𑀺 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀬𑀫𑀸𑀦𑁄 𑀓𑀸𑀮𑀫𑀓𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀲𑁆𑀲 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀲𑁆𑀲 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀲𑀫𑀓𑀁 𑀧𑀝𑀺𑀯𑀻𑀲𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1716. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀩𑀳𑀼𑀪𑀡𑁆𑀟𑁄 𑀩𑀳𑀼𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑁂 𑀲𑀁𑀖𑁄 𑀲𑀸𑀫𑀻 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑁂, 𑀅𑀧𑀺 𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸 𑀩𑀳𑀽𑀧𑀓𑀸𑀭𑀸𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁂𑀦 𑀢𑀺𑀘𑀻𑀯𑀭𑀜𑁆𑀘 𑀧𑀢𑁆𑀢𑀜𑁆𑀘 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑀦𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀮𑀳𑀼𑀪𑀡𑁆𑀟𑀁 𑀮𑀳𑀼𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀢𑀁 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀪𑀸𑀚𑁂𑀢𑀼𑀁𑁇 𑀬𑀁 𑀢𑀢𑁆𑀣 𑀕𑀭𑀼𑀪𑀡𑁆𑀟𑀁 𑀕𑀭𑀼𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁 𑀢𑀁 𑀆𑀕𑀢𑀸𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀘𑀸𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀲𑁆𑀲 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑀺𑀓𑀁 𑀅𑀯𑁂𑀪𑀗𑁆𑀕𑀺𑀓𑀦𑁆”𑀢𑀺𑁇
