# 𑁮𑁇𑁨𑁨 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(364.)

1696. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀸𑀫𑀓𑀸𑀯𑀸𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀢𑀢𑁆𑀣 𑀘 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀪𑀸𑀚𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀪𑀸𑀕𑀦𑁆”𑀢𑀺? “𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀢𑀢𑁄 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀧𑀺 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀪𑀸𑀚𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀪𑀸𑀕𑀦𑁆”𑀢𑀺? “𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀢𑀢𑁄𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀧𑀺 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀪𑀸𑀚𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀪𑀸𑀕𑀦𑁆”𑀢𑀺? “𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀢𑀢𑁄𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀫𑀳𑀦𑁆𑀢𑀁 𑀘𑀻𑀯𑀭𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀆𑀤𑀸𑀬 𑀧𑀼𑀦𑀤𑁂𑀯 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀜𑁆𑀙𑀺𑁇

1697. 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀳𑀸𑀧𑀼𑀜𑁆𑀜𑁄𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀁 𑀢𑁂 𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀓𑀼𑀢𑁄 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀼𑀜𑁆𑀜𑀁? 𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀸𑀫𑀓𑀸𑀯𑀸𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀫𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀪𑀸𑀚𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀪𑀸𑀕𑀦𑁆’𑀢𑀺? ‘𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀢𑁄 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀧𑀺 𑀫𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀪𑀸𑀚𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀪𑀸𑀕𑀦𑁆’𑀢𑀺? ‘𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀢𑁄𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀁 𑀆𑀯𑀸𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑀁𑁇 𑀢𑀢𑁆𑀣𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀧𑀺 𑀫𑀁 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀪𑀸𑀚𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀲𑀺 𑀪𑀸𑀕𑀦𑁆’𑀢𑀺? ‘𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀢𑀢𑁄𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑀁𑁇 𑀏𑀯𑀁 𑀫𑁂 𑀩𑀳𑀼𑀁 𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀻”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇

1698. 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁄 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑀁 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀘𑀻𑀯𑀭𑀪𑀸𑀕𑁄 𑀲𑀸𑀤𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀲𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1699. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀓𑁄 𑀤𑁆𑀯𑀻𑀲𑀼 𑀆𑀯𑀸𑀲𑁂𑀲𑀼 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺—  “𑀏𑀯𑀁 𑀫𑁂 𑀩𑀳𑀼𑀁 𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀤𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀏𑀓𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀁𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑁄 𑀤𑁆𑀯𑀻𑀲𑀼 𑀆𑀯𑀸𑀲𑁂𑀲𑀼 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀢𑀺—  ‘𑀏𑀯𑀁 𑀫𑁂 𑀩𑀳𑀼𑀁 𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀧𑀟𑁆𑀠𑀁 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀧𑀟𑁆𑀠𑀁 𑀯𑀲𑀢𑀺, 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀧𑀟𑁆𑀠𑁄 𑀅𑀫𑀼𑀢𑁆𑀭 𑀉𑀧𑀟𑁆𑀠𑁄 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀦 𑀩𑀳𑀼𑀢𑀭𑀁 𑀯𑀲𑀢𑀺, 𑀢𑀢𑁄 𑀘𑀻𑀯𑀭𑀧𑀝𑀺𑀯𑀻𑀲𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
