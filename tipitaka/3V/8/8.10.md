# 𑁮𑁇𑁧𑁦 𑀓𑀫𑁆𑀩𑀮𑀸𑀦𑀼𑀚𑀸𑀦𑀦𑀸𑀤𑀺𑀓𑀣𑀸

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀓𑀫𑁆𑀩𑀮𑀸𑀦𑀼𑀚𑀸𑀦𑀦𑀸𑀤𑀺𑀓𑀣𑀸

(338.)

1594\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀓𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀧𑀸𑀳𑁂𑀲𑀺 𑀉𑀧𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀦𑀁 𑀔𑀫𑀫𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀁 𑀅𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀓𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀓𑁄 𑀓𑀫𑁆𑀩𑀮𑁄 𑀓𑀸𑀲𑀺𑀭𑀜𑁆𑀜𑀸 𑀧𑀳𑀺𑀢𑁄 𑀉𑀧𑀟𑁆𑀠𑀓𑀸𑀲𑀺𑀦𑀁 𑀔𑀫𑀫𑀸𑀦𑁄𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀫𑁆𑀩𑀮𑀁, 𑀬𑀁 𑀫𑀫𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀓𑀫𑁆𑀩𑀮𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺…𑀧𑁂…  𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀫𑁆𑀩𑀮𑀦𑁆”𑀢𑀺𑁇

(339.)

1595\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀖𑀲𑁆𑀲 𑀉𑀘𑁆𑀘𑀸𑀯𑀘𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀢𑀸 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁, 𑀓𑀺𑀁 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙 𑀘𑀻𑀯𑀭𑀸𑀦𑀺—  𑀔𑁄𑀫𑀁 𑀓𑀧𑁆𑀧𑀸𑀲𑀺𑀓𑀁 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀁 𑀓𑀫𑁆𑀩𑀮𑀁 𑀲𑀸𑀡𑀁 𑀪𑀗𑁆𑀕𑀦𑁆”𑀢𑀺𑁇

(340.)

1596\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺 𑀢𑁂 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀦 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺—  “𑀏𑀓𑀁𑀬𑁂𑀯 𑀪𑀕𑀯𑀢𑀸 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁, 𑀦 𑀤𑁆𑀯𑁂”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀳𑀧𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑁂𑀦 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀫𑁆𑀧𑀺 𑀲𑀸𑀤𑀺𑀬𑀺𑀢𑀼𑀁; 𑀢𑀤𑀼𑀪𑀬𑁂𑀦𑀧𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀁 𑀯𑀡𑁆𑀡𑁂𑀫𑀻”𑀢𑀺𑁇
