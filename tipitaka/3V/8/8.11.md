# 𑁮𑁇𑁧𑁧 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀓𑀣𑀸

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀧𑀭𑀺𑀬𑁂𑀲𑀦𑀓𑀣𑀸

(341.)

1597\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀕𑀫𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬 𑀢𑁂 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀮𑀪𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀕𑀫𑁂𑀲𑀼𑀁 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀸𑀕𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀸𑀕𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑁇 𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦𑀸𑀕𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀸𑀕𑀫𑁂𑀦𑁆𑀢𑀸𑀦𑀁 𑀦𑀸𑀓𑀸𑀫𑀸 𑀪𑀸𑀕𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1598\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀫𑁂𑀲𑀼𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬 𑀢𑁂 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀮𑀪𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀕𑀫𑁂𑀲𑀼𑀁 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀸𑀕𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀸𑀕𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑁇 𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀕𑀫𑁂𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀓𑀸𑀫𑀸 𑀪𑀸𑀕𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1599\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀞𑀫𑀁 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀘𑁆𑀙𑀸 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀞𑀫𑀁 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬 𑀢𑁂 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀮𑀪𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀘𑁆𑀙𑀸 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀢𑁂 𑀦 𑀮𑀪𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀸𑀕𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀸𑀕𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑁇 𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀧𑀘𑁆𑀙𑀸 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀘𑁆𑀙𑀸 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀦𑀸𑀓𑀸𑀫𑀸 𑀪𑀸𑀕𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1600\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀲𑀤𑀺𑀲𑀸 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀮𑀪𑀺𑀁𑀲𑀼, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀮𑀪𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀮𑀪𑀺𑀁𑀲𑀼, 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀸𑀕𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀸𑀕𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑁇 𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦 𑀮𑀪𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑀺𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀓𑀸𑀫𑀸 𑀪𑀸𑀕𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1601\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀓𑀢𑀺𑀓𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑀼𑀲𑀸𑀦𑀁 𑀑𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀬𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀸𑀦𑀺 𑀮𑀪𑀺𑀁𑀲𑀼, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀮𑀪𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀮𑀪𑀺𑀁𑀲𑀼 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀸𑀕𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀦 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀪𑀸𑀕𑀁 𑀤𑀲𑁆𑀲𑀸𑀫𑁇 𑀓𑀺𑀲𑁆𑀲 𑀢𑀼𑀫𑁆𑀳𑁂 𑀦 𑀮𑀪𑀺𑀢𑁆𑀣𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀢𑀺𑀓𑀁 𑀓𑀢𑁆𑀯𑀸 𑀑𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀓𑀸𑀫𑀸 𑀪𑀸𑀕𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
