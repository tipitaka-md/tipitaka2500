# 𑁮𑁇𑁮 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀺𑀭𑁂𑀘𑀦𑀓𑀣𑀸

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀯𑀺𑀭𑁂𑀘𑀦𑀓𑀣𑀸

(336.)

1584\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑁄 𑀤𑁄𑀲𑀸𑀪𑀺𑀲𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀤𑁄𑀲𑀸𑀪𑀺𑀲𑀦𑁆𑀦𑁄 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄𑁇 𑀇𑀘𑁆𑀙𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀧𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑁄𑀲𑀸𑀪𑀺𑀲𑀦𑁆𑀦𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀚𑀻𑀯𑀓, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄𑁇 𑀇𑀘𑁆𑀙𑀢𑀺 𑀢𑀣𑀸𑀕𑀢𑁄 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀧𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑀁 𑀓𑀢𑀺𑀧𑀸𑀳𑀁 𑀲𑀺𑀦𑁂𑀳𑁂𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑀁 𑀓𑀢𑀺𑀧𑀸𑀳𑀁 𑀲𑀺𑀦𑁂𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀺𑀦𑀺𑀤𑁆𑀥𑁄 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀚𑀻𑀯𑀓, 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄𑁇 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦 𑀔𑁄 𑀫𑁂𑀢𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀬𑁄𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀑𑀍𑀆𑀭𑀺𑀓𑀁 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀤𑀤𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀢𑀻𑀡𑀺 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀸𑀦𑀺 𑀦𑀸𑀦𑀸𑀪𑁂𑀲𑀚𑁆𑀚𑁂𑀳𑀺 𑀧𑀭𑀺𑀪𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀇𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀞𑀫𑀁 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀁 𑀉𑀧𑀲𑀺𑀗𑁆𑀖𑀢𑀼𑁇 𑀇𑀤𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀇𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀁 𑀉𑀧𑀲𑀺𑀗𑁆𑀖𑀢𑀼𑁇 𑀇𑀤𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀇𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀢𑀢𑀺𑀬𑀁 𑀉𑀧𑁆𑀧𑀮𑀳𑀢𑁆𑀣𑀁 𑀉𑀧𑀲𑀺𑀗𑁆𑀖𑀢𑀼𑁇 𑀇𑀤𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀤𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀩𑀳𑀺 𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑀸 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀬𑀸 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑁄𑀲𑀸𑀪𑀺𑀲𑀦𑁆𑀦𑁄 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄𑁇 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺, 𑀏𑀓𑀽𑀦𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀪𑀕𑀯𑀸 𑀯𑀺𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑀳𑀸𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀓𑀺𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1585\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥𑀸𑀦𑀦𑁆𑀤, 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀩𑀳𑀺 𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑀸 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀫𑀬𑀸 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑁄𑀲𑀸𑀪𑀺𑀲𑀦𑁆𑀦𑁄 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄𑁇 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀫𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺, 𑀏𑀓𑀽𑀦𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀪𑀕𑀯𑀸 𑀯𑀺𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑀳𑀸𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀓𑀺𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀸𑀦𑀦𑁆𑀤, 𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀲𑀺𑁇

1586\. 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀺𑀭𑀺𑀢𑁆𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸”𑀢𑀺? “𑀯𑀺𑀭𑀺𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺, 𑀚𑀻𑀯𑀓𑀸”𑀢𑀺𑁇 “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀺 𑀤𑁆𑀯𑀸𑀭𑀓𑁄𑀝𑁆𑀞𑀓𑀸 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀫𑀬𑀸 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑁄𑀲𑀸𑀪𑀺𑀲𑀦𑁆𑀦𑁄 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑁄𑁇 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺, 𑀏𑀓𑀽𑀦𑀢𑁆𑀢𑀺𑀁𑀲𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀧𑀺 𑀘 𑀪𑀕𑀯𑀸 𑀯𑀺𑀭𑀺𑀢𑁆𑀢𑁄 𑀦𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑀳𑀸𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀓𑀺𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀦𑀳𑀸𑀬𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀦𑀳𑀸𑀬𑀢𑀼 𑀲𑀼𑀕𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀡𑁆𑀳𑁄𑀤𑀓𑀁 𑀦𑀳𑀸𑀬𑀺𑁇 𑀦𑀳𑀸𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀓𑀺𑀁 𑀯𑀺𑀭𑁂𑀘𑁂𑀲𑀺𑁇 𑀏𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀲𑀫𑀢𑁆𑀢𑀺𑀁𑀲𑀸𑀬 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀸𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑁄 𑀧𑀓𑀢𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀮𑀁 𑀬𑀽𑀲𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑁂𑀦𑀸”𑀢𑀺𑁇
