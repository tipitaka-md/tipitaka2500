# 𑁮𑁇𑁨𑁧 𑀲𑀁𑀖𑀺𑀓𑀘𑀻𑀯𑀭𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀁𑀖𑀺𑀓𑀘𑀻𑀯𑀭𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀣𑀸

(363.)

1692. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑁄 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑁇 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀤𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀲𑀁𑀖𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀏𑀓𑀓𑁄𑁇 𑀇𑀫𑁂 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀤𑀁𑀲𑀼𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀦𑀺 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀆𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀢𑀼𑀬𑁆𑀳𑁂𑀯, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀯 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀉𑀩𑁆𑀪𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀓𑁄 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  ‘𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸’𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲𑁂𑀯 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀯 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀉𑀩𑁆𑀪𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇

1693. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀢𑀼𑀓𑀸𑀮𑀁 𑀏𑀓𑁄 𑀯𑀲𑀺𑁇 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀤𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀲𑀁𑀖𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀏𑀓𑀓𑁄𑁇 𑀇𑀫𑁂 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  ‘𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸’𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀤𑀁𑀲𑀼𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀸𑀦𑀺 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀆𑀤𑀸𑀬 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀼𑀔𑀻𑀪𑀽𑀢𑁂𑀦 𑀲𑀁𑀖𑁂𑀦 𑀪𑀸𑀚𑁂𑀢𑀼𑀁𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀢𑀼𑀓𑀸𑀮𑀁 𑀏𑀓𑁄 𑀯𑀲𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  ‘𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸’𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀢𑀼𑀁—  ‘𑀫𑀬𑁆𑀳𑀺𑀫𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀻’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀥𑀺𑀝𑁆𑀞𑀺𑀢𑁂 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀫𑀓𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀪𑀸𑀕𑁄𑁇 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑀺𑀬𑀫𑀸𑀦𑁂, 𑀅𑀧𑀸𑀢𑀺𑀢𑁂 𑀓𑀼𑀲𑁂, 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑀫𑀓𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀪𑀸𑀕𑁄𑁇 𑀢𑁂𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀢𑀁 𑀘𑀻𑀯𑀭𑀁 𑀪𑀸𑀚𑀺𑀬𑀫𑀸𑀦𑁂, 𑀧𑀸𑀢𑀺𑀢𑁂 𑀓𑀼𑀲𑁂, 𑀅𑀜𑁆𑀜𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀦𑀸𑀓𑀸𑀫𑀸 𑀤𑀸𑀢𑀩𑁆𑀩𑁄 𑀪𑀸𑀕𑁄”𑀢𑀺𑁇

1694. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀸𑀢𑀺𑀓𑀸 𑀣𑁂𑀭𑀸, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀇𑀲𑀺𑀤𑀸𑀲𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀇𑀲𑀺𑀪𑀝𑁄, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀕𑀸𑀫𑀓𑀸𑀯𑀸𑀲𑀁 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  “𑀘𑀺𑀭𑀲𑁆𑀲𑀸𑀧𑀺 𑀣𑁂𑀭𑀸 𑀆𑀕𑀢𑀸”𑀢𑀺—  𑀲𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀅𑀤𑀁𑀲𑀼𑁇 𑀆𑀯𑀸𑀲𑀺𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼—  “𑀇𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀺𑀓𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀣𑁂𑀭𑁂 𑀆𑀕𑀫𑁆𑀫 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀺, 𑀲𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀣𑁂𑀭𑀸 𑀪𑀸𑀕𑀦𑁆”𑀢𑀺𑁇 𑀣𑁂𑀭𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀬𑀣𑀸 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁𑀬𑁂𑀯 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀯 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀉𑀩𑁆𑀪𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇

1695. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀘𑀢𑀼𑀯𑀕𑁆𑀕𑁄 𑀧𑀘𑁆𑀙𑀺𑀫𑁄 𑀲𑀁𑀖𑁄’𑀢𑀺𑁇 𑀫𑀬𑀜𑁆𑀘𑀫𑁆𑀳𑀸 𑀢𑀬𑁄 𑀚𑀦𑀸𑁇 𑀇𑀫𑁂 𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑁂𑀫𑀸𑀢𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀣𑁂𑀭𑀸, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀦𑀺𑀮𑀯𑀸𑀲𑀻 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀲𑀸𑀡𑀯𑀸𑀲𑀻 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀕𑁄𑀢𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀪𑀕𑀼 𑀆𑀬𑀲𑁆𑀫𑀸 𑀘 𑀨𑀍𑀇𑀓𑀲𑀦𑁆𑀢𑀸𑀦𑁄, 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀝𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀸𑀝𑀮𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀣𑁂𑀭𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀣𑁂𑀭𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀬𑀣𑀸 𑀔𑁄 𑀫𑀬𑀁 𑀆𑀯𑀼𑀲𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫, 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁𑀬𑁂𑀯 𑀢𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀬𑀸𑀯 𑀓𑀣𑀺𑀦𑀲𑁆𑀲 𑀉𑀩𑁆𑀪𑀸𑀭𑀸𑀬𑀸”𑀢𑀺𑁇
