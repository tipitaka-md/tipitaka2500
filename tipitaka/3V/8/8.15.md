# 𑁮𑁇𑁧𑁫 𑀙𑀺𑀦𑁆𑀦𑀓𑀘𑀻𑀯𑀭𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀙𑀺𑀦𑁆𑀦𑀓𑀘𑀻𑀯𑀭𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(345.)

1643\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀕𑀺𑀭𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀕𑀥𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀘𑁆𑀙𑀺𑀩𑀤𑁆𑀥𑀁 𑀧𑀸𑀍𑀇𑀩𑀤𑁆𑀥𑀁 𑀫𑀭𑀺𑀬𑀸𑀤𑀩𑀤𑁆𑀥𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀩𑀤𑁆𑀥𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀧𑀲𑁆𑀲𑀲𑀺 𑀦𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀕𑀥𑀔𑁂𑀢𑁆𑀢𑀁 𑀅𑀘𑁆𑀙𑀺𑀩𑀤𑁆𑀥𑀁 𑀧𑀸𑀍𑀇𑀩𑀤𑁆𑀥𑀁 𑀫𑀭𑀺𑀬𑀸𑀤𑀩𑀤𑁆𑀥𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀩𑀤𑁆𑀥𑀦𑁆”𑀢𑀺? “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺𑁇 “𑀉𑀲𑁆𑀲𑀳𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀺 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀉𑀲𑁆𑀲𑀳𑀸𑀫𑀺, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀕𑀺𑀭𑀺𑀲𑁆𑀫𑀺𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀦𑀤𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀜𑁆𑀙𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀲𑁆𑀲𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀘𑀻𑀯𑀭𑀸𑀦𑀺 𑀲𑀁𑀯𑀺𑀤𑀳𑀺𑀢𑀸𑀦𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀧𑀡𑁆𑀟𑀺𑀢𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀦𑁆𑀤𑁄; 𑀫𑀳𑀸𑀧𑀜𑁆𑀜𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀦𑀦𑁆𑀤𑁄; 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀫𑀬𑀸 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀼𑀲𑀺𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀟𑁆𑀠𑀓𑀼𑀲𑀺𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀡𑁆𑀟𑀮𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀟𑁆𑀠𑀫𑀡𑁆𑀟𑀮𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀯𑀺𑀯𑀝𑁆𑀝𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀦𑀼𑀯𑀺𑀯𑀝𑁆𑀝𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀕𑀻𑀯𑁂𑀬𑁆𑀬𑀓𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀚𑀗𑁆𑀖𑁂𑀬𑁆𑀬𑀓𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀩𑀸𑀳𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀦𑀸𑀫 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀙𑀺𑀦𑁆𑀦𑀓𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀢𑁆𑀣𑀮𑀽𑀔𑀁 𑀲𑀫𑀡𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀁 𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀜𑁆𑀘 𑀅𑀦𑀪𑀺𑀘𑁆𑀙𑀺𑀢𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀺𑀦𑁆𑀦𑀓𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁 𑀙𑀺𑀦𑁆𑀦𑀓𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀙𑀺𑀦𑁆𑀦𑀓𑀁 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀦𑁆”𑀢𑀺𑁇
