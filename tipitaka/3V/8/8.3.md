# 𑁮𑁇𑁩 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀭𑀸𑀚𑀯𑀢𑁆𑀣𑀼

* 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀭𑀸𑀚𑀯𑀢𑁆𑀣𑀼

(331.)

1568\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀪𑀕𑀦𑁆𑀤𑀮𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸𑀝𑀓𑀸 𑀮𑁄𑀳𑀺𑀢𑁂𑀦 𑀫𑀓𑁆𑀔𑀺𑀬𑀦𑁆𑀢𑀺𑁇 𑀤𑁂𑀯𑀺𑀬𑁄 𑀤𑀺𑀲𑁆𑀯𑀸 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑁂𑀦𑁆𑀢𑀺—  “𑀉𑀢𑀼𑀦𑀻 𑀤𑀸𑀦𑀺 𑀤𑁂𑀯𑁄, 𑀧𑀼𑀧𑁆𑀨𑀁 𑀤𑁂𑀯𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀦 𑀘𑀺𑀭𑀁 𑀤𑁂𑀯𑁄 𑀯𑀺𑀚𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀗𑁆𑀓𑀼 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀅𑀪𑀬𑀁 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄, 𑀪𑀡𑁂 𑀅𑀪𑀬, 𑀢𑀸𑀤𑀺𑀲𑁄 𑀆𑀩𑀸𑀥𑁄, 𑀲𑀸𑀝𑀓𑀸 𑀮𑁄𑀳𑀺𑀢𑁂𑀦 𑀫𑀓𑁆𑀔𑀺𑀬𑀦𑁆𑀢𑀺, 𑀤𑁂𑀯𑀺𑀬𑁄 𑀫𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑁂𑀦𑁆𑀢𑀺—  ‘𑀉𑀢𑀼𑀦𑀻 𑀤𑀸𑀦𑀺 𑀤𑁂𑀯𑁄, 𑀧𑀼𑀧𑁆𑀨𑀁 𑀤𑁂𑀯𑀲𑁆𑀲 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀦 𑀘𑀺𑀭𑀁 𑀤𑁂𑀯𑁄 𑀯𑀺𑀚𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀇𑀗𑁆𑀖, 𑀪𑀡𑁂 𑀅𑀪𑀬, 𑀢𑀸𑀤𑀺𑀲𑀁 𑀯𑁂𑀚𑁆𑀚𑀁 𑀚𑀸𑀦𑀸𑀳𑀺 𑀬𑁄 𑀫𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀅𑀬𑀁, 𑀤𑁂𑀯, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀚𑀻𑀯𑀓𑁄 𑀯𑁂𑀚𑁆𑀚𑁄 𑀢𑀭𑀼𑀡𑁄 𑀪𑀤𑁆𑀭𑀓𑁄𑁇 𑀲𑁄 𑀤𑁂𑀯𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂 𑀅𑀪𑀬, 𑀚𑀻𑀯𑀓𑀁 𑀯𑁂𑀚𑁆𑀚𑀁 𑀆𑀡𑀸𑀧𑁂𑀳𑀺; 𑀲𑁄 𑀫𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1569\. 𑀅𑀣 𑀔𑁄 𑀅𑀪𑀬𑁄 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑁄 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓, 𑀭𑀸𑀚𑀸𑀦𑀁 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀅𑀪𑀬𑀲𑁆𑀲 𑀭𑀸𑀚𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀦𑀔𑁂𑀦 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀩𑀸𑀥𑀁 𑀢𑁂, 𑀤𑁂𑀯, 𑀧𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀪𑀕𑀦𑁆𑀤𑀮𑀸𑀩𑀸𑀥𑀁 𑀏𑀓𑁂𑀦𑁂𑀯 𑀆𑀮𑁂𑀧𑁂𑀦 𑀅𑀧𑀓𑀟𑁆𑀠𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀅𑀭𑁄𑀕𑁄 𑀲𑀫𑀸𑀦𑁄 𑀧𑀜𑁆𑀘 𑀇𑀢𑁆𑀣𑀺𑀲𑀢𑀸𑀦𑀺 𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀁 𑀪𑀽𑀲𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀑𑀫𑀼𑀜𑁆𑀘𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀼𑀜𑁆𑀚𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀢𑀁, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀇𑀢𑁆𑀣𑀺𑀲𑀢𑀸𑀦𑀁 𑀲𑀩𑁆𑀩𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀁 𑀢𑀼𑀬𑁆𑀳𑀁 𑀳𑁄𑀢𑀽”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀤𑁂𑀯, 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀫𑁂 𑀤𑁂𑀯𑁄 𑀲𑀭𑀢𑀽”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂 𑀚𑀻𑀯𑀓, 𑀫𑀁 𑀉𑀧𑀝𑁆𑀞𑀳, 𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀜𑁆𑀘, 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇
