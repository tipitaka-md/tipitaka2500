# 𑁮𑁇𑁨𑁬 𑀓𑀼𑀲𑀘𑀻𑀭𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀓𑀼𑀲𑀘𑀻𑀭𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

(371.)

1718. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀼𑀲𑀘𑀻𑀭𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀯𑀸𑀓𑀘𑀻𑀭𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀨𑀮𑀓𑀘𑀻𑀭𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀓𑁂𑀲𑀓𑀫𑁆𑀩𑀮𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀯𑀸𑀍𑀅𑀓𑀫𑁆𑀩𑀮𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀉𑀮𑀽𑀓𑀧𑀓𑁆𑀔𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀅𑀚𑀺𑀦𑀓𑁆𑀔𑀺𑀧𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀕𑀯𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑁆𑀲 𑀥𑀼𑀢𑀲𑁆𑀲 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑁆𑀲 𑀅𑀧𑀘𑀬𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻𑁇 𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀚𑀺𑀦𑀓𑁆𑀔𑀺𑀧𑀁 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀮𑁆𑀮𑁂𑀔𑀸𑀬 𑀥𑀼𑀢𑀢𑀸𑀬 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀢𑀸𑀬 𑀅𑀧𑀘𑀬𑀸𑀬 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀚𑀺𑀦𑀓𑁆𑀔𑀺𑀧𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀚𑀺𑀦𑀓𑁆𑀔𑀺𑀧𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀥𑀚𑀁 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑀺𑀦𑀓𑁆𑀔𑀺𑀧𑀁 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀥𑀚𑀁 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1719. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀓𑁆𑀓𑀦𑀸𑀍𑀅𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀧𑁄𑀢𑁆𑀣𑀓𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀕𑀯𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀲𑀮𑁆𑀮𑁂𑀔𑀲𑁆𑀲 𑀥𑀼𑀢𑀲𑁆𑀲 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀲𑁆𑀲 𑀅𑀧𑀘𑀬𑀲𑁆𑀲 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀲𑁆𑀲, 𑀯𑀡𑁆𑀡𑀯𑀸𑀤𑀻𑁇 𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑁄𑀢𑁆𑀣𑀓𑁄 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀢𑀸𑀬 𑀲𑀦𑁆𑀢𑀼𑀝𑁆𑀞𑀺𑀢𑀸𑀬 𑀲𑀮𑁆𑀮𑁂𑀔𑀸𑀬 𑀥𑀼𑀢𑀢𑀸𑀬 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀢𑀸𑀬 𑀅𑀧𑀘𑀬𑀸𑀬 𑀯𑀻𑀭𑀺𑀬𑀸𑀭𑀫𑁆𑀪𑀸𑀬 𑀲𑀁𑀯𑀢𑁆𑀢𑀢𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑁄𑀢𑁆𑀣𑀓𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀧𑁄𑀢𑁆𑀣𑀓𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑁄𑀢𑁆𑀣𑀓𑁄 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀦𑀺𑀯𑀸𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
