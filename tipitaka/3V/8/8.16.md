# 𑁮𑁇𑁧𑁬 𑀢𑀺𑀘𑀻𑀯𑀭𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀘𑀻𑀯𑀭𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀢𑀺𑀘𑀻𑀯𑀭𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(346.)

1644\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲 𑀪𑀕𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀭𑀸𑀚𑀕𑀳𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑁂𑀳𑀺 𑀉𑀩𑁆𑀪𑀡𑁆𑀟𑀺𑀢𑁂 𑀲𑀻𑀲𑁂𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀺𑀲𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀦𑁆𑀥𑁂𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀺𑀲𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀝𑀺𑀬𑀸𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀺𑀲𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀢𑀺𑀮𑀳𑀼𑀁 𑀔𑁄 𑀇𑀫𑁂 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀘𑀻𑀯𑀭𑁂 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀆𑀯𑀢𑁆𑀢𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑁂 𑀲𑀻𑀫𑀁 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑀁, 𑀫𑀭𑀺𑀬𑀸𑀤𑀁 𑀞𑀧𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑁄𑀢𑀫𑀓𑁂 𑀘𑁂𑀢𑀺𑀬𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀲𑀻𑀢𑀸𑀲𑀼 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀝𑁆𑀞𑀓𑀸𑀲𑀼 𑀳𑀺𑀫𑀧𑀸𑀢𑀲𑀫𑀬𑁂 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀏𑀓𑀘𑀻𑀯𑀭𑁄 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀧𑀞𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀻𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀁 𑀪𑀕𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑁇 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀻𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀢𑀺𑀬𑀁 𑀪𑀕𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑁇 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀉𑀤𑁆𑀥𑀲𑁆𑀢𑁂 𑀅𑀭𑀼𑀡𑁂 𑀦𑀦𑁆𑀤𑀺𑀫𑀼𑀔𑀺𑀬𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀻𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑀁 𑀪𑀕𑀯𑀸 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑁇 𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑁂𑀧𑀺 𑀔𑁄 𑀢𑁂 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀲𑀻𑀢𑀸𑀮𑀼𑀓𑀸 𑀲𑀻𑀢𑀪𑀻𑀭𑀼𑀓𑀸 𑀢𑁂𑀧𑀺 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑁂 𑀲𑀻𑀫𑀁 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑀁, 𑀫𑀭𑀺𑀬𑀸𑀤𑀁 𑀞𑀧𑁂𑀬𑁆𑀬𑀁, 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇

1645\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀭𑀸𑀚𑀕𑀳𑀁 𑀅𑀦𑁆𑀢𑀭𑀸 𑀘 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀅𑀤𑁆𑀤𑀲𑀁 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀻𑀯𑀭𑁂𑀳𑀺 𑀉𑀩𑁆𑀪𑀡𑁆𑀟𑀺𑀢𑁂 𑀲𑀻𑀲𑁂𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀺𑀲𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀦𑁆𑀥𑁂𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀺𑀲𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀓𑀝𑀺𑀬𑀸𑀧𑀺 𑀘𑀻𑀯𑀭𑀪𑀺𑀲𑀺𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀢𑀺𑀮𑀳𑀼𑀁 𑀔𑁄 𑀇𑀫𑁂 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀘𑀻𑀯𑀭𑁂 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀆𑀯𑀢𑁆𑀢𑀸𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑁂 𑀲𑀻𑀫𑀁 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑀁, 𑀫𑀭𑀺𑀬𑀸𑀤𑀁 𑀞𑀧𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀇𑀥𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀢𑀸𑀲𑀼 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀝𑁆𑀞𑀓𑀸𑀲𑀼 𑀳𑀺𑀫𑀧𑀸𑀢𑀲𑀫𑀬𑁂 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀏𑀓𑀘𑀻𑀯𑀭𑁄 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑁇 𑀦 𑀫𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀧𑀞𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀻𑀢𑀁 𑀫𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀸𑀳𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀁𑁇 𑀦 𑀫𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀫𑀚𑁆𑀛𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀲𑀻𑀢𑀁 𑀫𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀢𑀺𑀬𑀸𑀳𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀁𑁇 𑀦 𑀫𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑁂 𑀧𑀘𑁆𑀙𑀺𑀫𑁂 𑀬𑀸𑀫𑁂 𑀉𑀤𑁆𑀥𑀲𑁆𑀢𑁂 𑀅𑀭𑀼𑀡𑁂 𑀦𑀦𑁆𑀤𑀺𑀫𑀼𑀔𑀺𑀬𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀻𑀢𑀁 𑀫𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀘𑀢𑀼𑀢𑁆𑀣𑀸𑀳𑀁 𑀘𑀻𑀯𑀭𑀁 𑀧𑀸𑀭𑀼𑀧𑀺𑀁𑁇 𑀦 𑀫𑀁 𑀲𑀻𑀢𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂𑀧𑀺 𑀔𑁄 𑀢𑁂 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀲𑀻𑀢𑀸𑀮𑀼𑀓𑀸 𑀲𑀻𑀢𑀪𑀻𑀭𑀼𑀓𑀸 𑀢𑁂𑀧𑀺 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀢𑀺𑀘𑀻𑀯𑀭𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑁂 𑀲𑀻𑀫𑀁 𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑀁, 𑀫𑀭𑀺𑀬𑀸𑀤𑀁 𑀞𑀧𑁂𑀬𑁆𑀬𑀁, 𑀢𑀺𑀘𑀻𑀯𑀭𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀘𑀻𑀯𑀭𑀁—  𑀤𑀺𑀕𑀼𑀡𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺𑀁, 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁, 𑀏𑀓𑀘𑁆𑀘𑀺𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑀦𑁆”𑀢𑀺𑁇
