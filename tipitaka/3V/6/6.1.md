# 𑁬𑁇𑁧 𑀧𑀜𑁆𑀘𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑀣𑀸

* 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀜𑁆𑀘𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑀣𑀸

(260.)

1198\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀸𑀭𑀤𑀺𑀓𑁂𑀦 𑀆𑀩𑀸𑀥𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀬𑀸𑀕𑀼𑀧𑀺 𑀧𑀻𑀢𑀸 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀪𑀼𑀢𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂 𑀢𑁂𑀦 𑀓𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀺𑀲𑁂 𑀮𑀽𑀔𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁂 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀺𑀲𑀸, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸”𑀢𑀺? “𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀸𑀭𑀤𑀺𑀓𑁂𑀦 𑀆𑀩𑀸𑀥𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀬𑀸𑀕𑀼𑀧𑀺 𑀧𑀻𑀢𑀸 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀪𑀼𑀢𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂 𑀢𑁂𑀦 𑀓𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸”𑀢𑀺𑁇

1199\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀸𑀭𑀤𑀺𑀓𑁂𑀦 𑀆𑀩𑀸𑀥𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀬𑀸𑀕𑀼𑀧𑀺 𑀧𑀻𑀢𑀸 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀪𑀼𑀢𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂 𑀢𑁂𑀦 𑀓𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀬𑀁 𑀪𑁂𑀲𑀚𑁆𑀚𑀜𑁆𑀘𑁂𑀯 𑀅𑀲𑁆𑀲 𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑀫𑁆𑀫𑀢𑀜𑁆𑀘 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀆𑀳𑀸𑀭𑀢𑁆𑀣𑀜𑁆𑀘 𑀨𑀭𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀑𑀍𑀆𑀭𑀺𑀓𑁄 𑀆𑀳𑀸𑀭𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑀸𑀦𑀺 𑀔𑁄 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀧𑁆𑀧𑀺, 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀢𑁂𑀮𑀁, 𑀫𑀥𑀼, 𑀨𑀸𑀡𑀺𑀢𑀁; 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀘𑁂𑀯 𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀘 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀆𑀳𑀸𑀭𑀢𑁆𑀣𑀜𑁆𑀘 𑀨𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀑𑀍𑀆𑀭𑀺𑀓𑁄 𑀆𑀳𑀸𑀭𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀓𑀸𑀮𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1200\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀸𑀭𑀤𑀺𑀓𑁂𑀦 𑀆𑀩𑀸𑀥𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀸𑀦𑀁 𑀬𑀸𑀕𑀼𑀧𑀺 𑀧𑀻𑀢𑀸 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺, 𑀪𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀪𑀼𑀢𑁆𑀢𑀁 𑀉𑀕𑁆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂 𑀢𑁂𑀦 𑀓𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀬𑀁 𑀪𑁂𑀲𑀚𑁆𑀚𑀜𑁆𑀘𑁂𑀯 𑀅𑀲𑁆𑀲 𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑀫𑁆𑀫𑀢𑀜𑁆𑀘 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀆𑀳𑀸𑀭𑀢𑁆𑀣𑀜𑁆𑀘 𑀨𑀭𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀑𑀍𑀆𑀭𑀺𑀓𑁄 𑀆𑀳𑀸𑀭𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀇𑀫𑀸𑀦𑀺 𑀔𑁄 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀧𑁆𑀧𑀺, 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀢𑁂𑀮𑀁, 𑀫𑀥𑀼, 𑀨𑀸𑀡𑀺𑀢𑀁; 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀘𑁂𑀯 𑀪𑁂𑀲𑀚𑁆𑀚𑀲𑀫𑁆𑀫𑀢𑀸𑀦𑀺 𑀘 𑀮𑁄𑀓𑀲𑁆𑀲, 𑀆𑀳𑀸𑀭𑀢𑁆𑀣𑀜𑁆𑀘 𑀨𑀭𑀦𑁆𑀢𑀺, 𑀦 𑀘 𑀑𑀍𑀆𑀭𑀺𑀓𑁄 𑀆𑀳𑀸𑀭𑁄 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀇𑀫𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁, 𑀓𑀸𑀮𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆’𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀓𑀸𑀮𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(261.)

1201\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀓𑀸𑀮𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀬𑀸𑀦𑀺𑀧𑀺 𑀢𑀸𑀦𑀺 𑀧𑀸𑀓𑀢𑀺𑀓𑀸𑀦𑀺 𑀮𑀽𑀔𑀸𑀦𑀺 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀢𑀸𑀦𑀺𑀧𑀺 𑀦𑀘𑁆𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀧𑀕𑁂𑀯 𑀲𑁂𑀦𑁂𑀲𑀺𑀢𑀸𑀦𑀺𑁇 𑀢𑁂 𑀢𑁂𑀦 𑀘𑁂𑀯 𑀲𑀸𑀭𑀤𑀺𑀓𑁂𑀦 𑀆𑀩𑀸𑀥𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀪𑀢𑁆𑀢𑀸𑀘𑁆𑀙𑀸𑀤𑀓𑁂𑀦, 𑀢𑀤𑀼𑀪𑀬𑁂𑀦 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀓𑀺𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀓𑀺𑀲𑁂 𑀮𑀽𑀔𑁂 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁂 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁂 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀢𑀭𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀓𑀺𑀲𑀸, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸”𑀢𑀺? “𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀸𑀦𑀺 𑀘 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀓𑀸𑀮𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀲𑀁 𑀬𑀸𑀦𑀺𑀧𑀺 𑀢𑀸𑀦𑀺 𑀧𑀸𑀓𑀢𑀺𑀓𑀸𑀦𑀺 𑀮𑀽𑀔𑀸𑀦𑀺 𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀢𑀸𑀦𑀺𑀧𑀺 𑀦𑀘𑁆𑀙𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀧𑀕𑁂𑀯 𑀲𑁂𑀦𑁂𑀲𑀺𑀢𑀸𑀦𑀺𑁇 𑀢𑁂 𑀢𑁂𑀦 𑀘𑁂𑀯 𑀲𑀸𑀭𑀤𑀺𑀓𑁂𑀦 𑀆𑀩𑀸𑀥𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑀸, 𑀇𑀫𑀺𑀦𑀸 𑀘 𑀪𑀢𑁆𑀢𑀸𑀘𑁆𑀙𑀸𑀤𑀓𑁂𑀦, 𑀢𑀤𑀼𑀪𑀬𑁂𑀦 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀓𑀺𑀲𑀸, 𑀮𑀽𑀔𑀸, 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀸, 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀸, 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀦𑀺 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀮𑁂𑀧𑀺 𑀯𑀺𑀓𑀸𑀮𑁂𑀧𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1202\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀕𑀺𑀮𑀸𑀦𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀲𑁂𑀳𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑁂𑀳𑀺 𑀅𑀢𑁆𑀣𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑀸𑀦𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺—  𑀅𑀘𑁆𑀙𑀯𑀲𑀁, 𑀫𑀘𑁆𑀙𑀯𑀲𑀁, 𑀲𑀼𑀲𑀼𑀓𑀸𑀯𑀲𑀁, 𑀲𑀽𑀓𑀭𑀯𑀲𑀁, 𑀕𑀤𑁆𑀭𑀪𑀯𑀲𑀁—  𑀓𑀸𑀮𑁂 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀓𑀸𑀮𑁂 𑀦𑀺𑀧𑁆𑀧𑀓𑁆𑀓𑀁 𑀓𑀸𑀮𑁂 𑀲𑀁𑀲𑀝𑁆𑀞𑀁 𑀢𑁂𑀮𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀯𑀺𑀓𑀸𑀮𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀯𑀺𑀓𑀸𑀮𑁂 𑀦𑀺𑀧𑁆𑀧𑀓𑁆𑀓𑀁 𑀯𑀺𑀓𑀸𑀮𑁂 𑀲𑀁𑀲𑀝𑁆𑀞𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀢𑀺𑀡𑁆𑀡𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇 𑀓𑀸𑀮𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀯𑀺𑀓𑀸𑀮𑁂 𑀦𑀺𑀧𑁆𑀧𑀓𑁆𑀓𑀁 𑀯𑀺𑀓𑀸𑀮𑁂 𑀲𑀁𑀲𑀝𑁆𑀞𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇 𑀓𑀸𑀮𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀓𑀸𑀮𑁂 𑀦𑀺𑀧𑁆𑀧𑀓𑁆𑀓𑀁 𑀯𑀺𑀓𑀸𑀮𑁂 𑀲𑀁𑀲𑀝𑁆𑀞𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀓𑀸𑀮𑁂 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀓𑀸𑀮𑁂 𑀦𑀺𑀧𑁆𑀧𑀓𑁆𑀓𑀁 𑀓𑀸𑀮𑁂 𑀲𑀁𑀲𑀝𑁆𑀞𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻”𑀢𑀺𑁇
