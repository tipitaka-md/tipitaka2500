# 𑁬𑁇𑁧𑁩 𑀩𑁂𑀮𑀝𑁆𑀞𑀓𑀘𑁆𑀘𑀸𑀦𑀯𑀢𑁆𑀣𑀼

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀩𑁂𑀮𑀝𑁆𑀞𑀓𑀘𑁆𑀘𑀸𑀦𑀯𑀢𑁆𑀣𑀼

(284.)

1288\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑁆𑀥𑀓𑀯𑀺𑀦𑁆𑀤𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺, 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁, 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀭𑀸𑀚𑀕𑀳𑀸 𑀅𑀦𑁆𑀥𑀓𑀯𑀺𑀦𑁆𑀤𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀓𑀝𑀲𑀢𑁂𑀳𑀺, 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀕𑀼𑀍𑀅𑀓𑀼𑀫𑁆𑀪𑀧𑀽𑀭𑁂𑀳𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑁂𑀮𑀝𑁆𑀞𑀁 𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

1289\. 𑀅𑀣 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀫𑁂𑀓𑀁 𑀕𑀼𑀍𑀅𑀓𑀼𑀫𑁆𑀪𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀏𑀓𑀁𑀬𑁂𑀯 𑀕𑀼𑀍𑀅𑀓𑀼𑀫𑁆𑀪𑀁 𑀆𑀳𑀭𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀬𑁂𑀯 𑀕𑀼𑀍𑀅𑀓𑀼𑀫𑁆𑀪𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀪𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀼𑀍𑀅𑀓𑀼𑀫𑁆𑀪𑁄; 𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀤𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀺𑀦𑁆𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀼𑀍𑀑, 𑀩𑀳𑀼 𑀘𑀸𑀬𑀁 𑀕𑀼𑀍𑀑 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑁄𑁇 𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀤𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀺𑀦𑁆𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀼𑀍𑀑 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄, 𑀩𑀳𑀼 𑀘𑀸𑀬𑀁 𑀕𑀼𑀍𑀑 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑁄𑁇 𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀼𑀍𑀏𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀼𑀍𑀏𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀢𑁆𑀢𑁂𑀧𑀺 𑀧𑀽𑀭𑁂𑀲𑀼𑀁 𑀧𑀭𑀺𑀲𑁆𑀲𑀸𑀯𑀦𑀸𑀦𑀺𑀧𑀺 𑀣𑀯𑀺𑀓𑀸𑀬𑁄𑀧𑀺 𑀧𑀽𑀭𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀼𑀍𑀏𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀼𑀍𑀏𑀳𑀺, 𑀩𑀳𑀼 𑀘𑀸𑀬𑀁 𑀕𑀼𑀍𑀑 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑁄𑁇 𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀤𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀺𑀦𑁆𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸𑀦𑀁 𑀕𑀼𑀍𑀑, 𑀩𑀳𑀼 𑀘𑀸𑀬𑀁 𑀕𑀼𑀍𑀑 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑁄𑁇 𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸𑀦𑀁 𑀕𑀼𑀍𑀅𑀁 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀤𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀤𑀺𑀦𑁆𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸𑀦𑀁 𑀕𑀼𑀍𑀑 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑁄, 𑀩𑀳𑀼 𑀘𑀸𑀬𑀁 𑀕𑀼𑀍𑀑 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑁄𑁇 𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑁂 𑀕𑀼𑀍𑀏𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑁂 𑀕𑀼𑀍𑀏𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸 𑀓𑁄𑀮𑀫𑁆𑀩𑁂𑀧𑀺 𑀖𑀝𑁂𑀧𑀺 𑀧𑀽𑀭𑁂𑀲𑀼𑀁, 𑀧𑀺𑀝𑀓𑀸𑀦𑀺𑀧𑀺 𑀉𑀘𑁆𑀙𑀗𑁆𑀕𑁂𑀧𑀺 𑀧𑀽𑀭𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑁂 𑀕𑀼𑀍𑀏𑀳𑀺 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑀺𑀢𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀺𑀖𑀸𑀲𑀸𑀤𑀸 𑀕𑀼𑀍𑀏𑀳𑀺, 𑀩𑀳𑀼 𑀘𑀸𑀬𑀁 𑀕𑀼𑀍𑀑 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑁄𑁇 𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀸𑀫𑀻”𑀢𑀺? “𑀦𑀸𑀳𑀁 𑀢𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀧𑀲𑁆𑀲𑀸𑀫𑀺 𑀲𑀤𑁂𑀯𑀓𑁂 𑀮𑁄𑀓𑁂 𑀲𑀫𑀸𑀭𑀓𑁂 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑁂 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀬𑀸 𑀧𑀚𑀸𑀬 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀬 𑀬𑀲𑁆𑀲 𑀲𑁄 𑀕𑀼𑀍𑀑 𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑁄 𑀲𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑀁 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀯𑀸 𑀢𑀣𑀸𑀕𑀢𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀯𑀸𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀓𑀘𑁆𑀘𑀸𑀦, 𑀢𑀁 𑀕𑀼𑀍𑀅𑀁 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑁂 𑀯𑀸 𑀙𑀟𑁆𑀟𑁂𑀳𑀺, 𑀅𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀯𑀸 𑀉𑀤𑀓𑁂 𑀑𑀧𑀺𑀮𑀸𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀕𑀼𑀍𑀅𑀁 𑀅𑀧𑁆𑀧𑀸𑀡𑀓𑁂 𑀉𑀤𑀓𑁂 𑀑𑀧𑀺𑀮𑀸𑀧𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀕𑀼𑀍𑀑 𑀉𑀤𑀓𑁂 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀧𑀥𑀽𑀧𑀸𑀬𑀢𑀺 𑀲𑀫𑁆𑀧𑀥𑀽𑀧𑀸𑀬𑀢𑀺𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀨𑀸𑀮𑁄 𑀤𑀺𑀯𑀲𑀁𑀲𑀦𑁆𑀢𑀢𑁆𑀢𑁄 𑀉𑀤𑀓𑁂 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀧𑀥𑀽𑀧𑀸𑀬𑀢𑀺 𑀲𑀫𑁆𑀧𑀥𑀽𑀧𑀸𑀬𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀲𑁄 𑀕𑀼𑀍𑀑 𑀉𑀤𑀓𑁂 𑀧𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘𑀺𑀘𑁆𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀘𑀺𑀝𑀺𑀘𑀺𑀝𑀸𑀬𑀢𑀺 𑀧𑀥𑀽𑀧𑀸𑀬𑀢𑀺 𑀲𑀫𑁆𑀧𑀥𑀽𑀧𑀸𑀬𑀢𑀺𑁇

1290\. 𑀅𑀣 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑀁𑀯𑀺𑀕𑁆𑀕𑁄 𑀮𑁄𑀫𑀳𑀝𑁆𑀞𑀚𑀸𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁, 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀩𑁂𑀮𑀝𑁆𑀞𑀁 𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑀁, 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑀁, 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑀁, 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀁, 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀯 𑀩𑁂𑀮𑀝𑁆𑀞𑀲𑁆𑀲 𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁, 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀩𑁂𑀮𑀝𑁆𑀞𑁄 𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁄 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑁄 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑁄 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬…𑀧𑁂…  𑀏𑀯𑀫𑁂𑀯𑀁 𑀔𑁄 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀕𑀢𑀦𑁆”𑀢𑀺𑁇

1291\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀕𑀼𑀍𑀑 𑀉𑀲𑁆𑀲𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲𑁂𑀯 𑀪𑀕𑀯𑀢𑀸 𑀕𑀼𑀍𑀑 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄, 𑀦𑁄 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲𑀸”𑀢𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀕𑀼𑀍𑀅𑀁 𑀦 𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀕𑀼𑀍𑀅𑀁, 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀕𑀼𑀍𑀑𑀤𑀓𑀦𑁆”𑀢𑀺𑁇
