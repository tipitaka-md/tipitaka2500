# 𑁬𑁇𑁨𑁫 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀯𑀢𑁆𑀣𑀼

* 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀯𑀢𑁆𑀣𑀼

(303.)

1363\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀆𑀢𑀼𑀫𑀸 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀆𑀢𑀼𑀫𑀸𑀬𑀁 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀦𑀳𑀸𑀧𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑁄𑁇 𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀤𑀸𑀭𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀫𑀜𑁆𑀚𑀼𑀓𑀸 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀬𑁆𑀬𑀓𑀸, 𑀤𑀓𑁆𑀔𑀸 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑀲𑀺𑀧𑁆𑀧𑀸 𑀲𑀓𑁂 𑀆𑀘𑀭𑀺𑀬𑀓𑁂 𑀦𑀳𑀸𑀧𑀺𑀢𑀓𑀫𑁆𑀫𑁂𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀲𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀆𑀢𑀼𑀫𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀢𑁂 𑀤𑀸𑀭𑀓𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭, 𑀢𑀸𑀢𑀸, 𑀆𑀢𑀼𑀫𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀕𑀘𑁆𑀙𑀣 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀢𑀸𑀢𑀸, 𑀔𑀼𑀭𑀪𑀡𑁆𑀟𑀁 𑀆𑀤𑀸𑀬 𑀦𑀸𑀍𑀇𑀬𑀸𑀯𑀸𑀧𑀓𑁂𑀦 𑀅𑀦𑀼𑀖𑀭𑀓𑀁 𑀅𑀦𑀼𑀖𑀭𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀣, 𑀮𑁄𑀡𑀫𑁆𑀧𑀺, 𑀢𑁂𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀳𑀭𑀣, 𑀪𑀕𑀯𑀢𑁄 𑀆𑀕𑀢𑀲𑁆𑀲 𑀬𑀸𑀕𑀼𑀧𑀸𑀦𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀢𑀸𑀢𑀸”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀤𑀸𑀭𑀓𑀸 𑀢𑀲𑁆𑀲 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀪𑀡𑁆𑀟𑀁 𑀆𑀤𑀸𑀬 𑀦𑀸𑀍𑀇𑀬𑀸𑀯𑀸𑀧𑀓𑁂𑀦 𑀅𑀦𑀼𑀖𑀭𑀓𑀁 𑀅𑀦𑀼𑀖𑀭𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀺, 𑀮𑁄𑀡𑀫𑁆𑀧𑀺, 𑀢𑁂𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀳𑀭𑀦𑁆𑀢𑀸𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑁂 𑀤𑀸𑀭𑀓𑁂 𑀫𑀜𑁆𑀚𑀼𑀓𑁂 𑀧𑀝𑀺𑀪𑀸𑀦𑁂𑀬𑁆𑀬𑀓𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀧𑀺 𑀦 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀼𑀓𑀸𑀫𑀸 𑀢𑁂𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺, 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸𑀧𑀺 𑀩𑀳𑀼𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀤𑀸𑀭𑀓𑀸 𑀩𑀳𑀼𑀁 𑀮𑁄𑀡𑀫𑁆𑀧𑀺, 𑀢𑁂𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀳𑀭𑀺𑀁𑀲𑀼𑁇

1364\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀆𑀢𑀼𑀫𑀸 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀆𑀢𑀼𑀫𑀸𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀪𑀼𑀲𑀸𑀕𑀸𑀭𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀳𑀽𑀢𑀁 𑀬𑀸𑀕𑀼𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀬𑀸𑀕𑀼𑀦𑁆”𑀢𑀺𑁇 𑀚𑀸𑀦𑀦𑁆𑀢𑀸𑀧𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀁 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀼𑀢𑀸𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀬𑀸𑀕𑀽”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀯𑀼𑀟𑁆𑀠𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂𑀦 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁂 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑀩𑁆𑀩𑀁, 𑀬𑁄 𑀲𑀫𑀸𑀤𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀳𑀸𑀧𑀺𑀢𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀔𑀼𑀭𑀪𑀡𑁆𑀟𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀳𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1365\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀢𑀼𑀫𑀸𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀩𑀳𑀼𑀁 𑀨𑀮𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀢𑀸 𑀨𑀮𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁, 𑀓𑀺𑀁 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀁 𑀨𑀮𑀔𑀸𑀤𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇
