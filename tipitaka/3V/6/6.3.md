# 𑁬𑁇𑁩 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀢𑁆𑀣𑀼

* 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀯𑀢𑁆𑀣𑀼

(270.)

1240\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀩𑁆𑀪𑀸𑀭𑀁 𑀲𑁄𑀥𑀸𑀧𑁂𑀢𑀺 𑀮𑁂𑀡𑀁 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀣𑁂𑀭𑁄 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑀻”𑀢𑀺? “𑀧𑀩𑁆𑀪𑀸𑀭𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀲𑁄𑀥𑀸𑀧𑁂𑀫𑀺, 𑀮𑁂𑀡𑀁 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄”𑀢𑀺𑁇 “𑀅𑀢𑁆𑀣𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀺𑀓𑁂𑀦𑀸”𑀢𑀺? “𑀦 𑀔𑁄, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀪𑀕𑀯𑀢𑀸 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁆𑀯𑀸 𑀫𑀫 𑀆𑀭𑁄𑀘𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺, 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺, 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺, 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁂𑀦 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1241\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀭𑀸𑀚𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀆𑀭𑀸𑀫𑀺𑀓𑀁 𑀤𑀸𑀢𑀼𑀓𑀸𑀫𑁄𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑀸𑀫𑀺𑀓𑀦𑁆”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑀸 𑀆𑀭𑀸𑀫𑀺𑀓𑁄”𑀢𑀺? “𑀏𑀯𑀁, 𑀫𑀳𑀸𑀭𑀸𑀚𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀺𑀓𑀁 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺𑁇

1242\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀺𑀓𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸, 𑀯𑀺𑀲𑁆𑀲𑀭𑀺𑀢𑁆𑀯𑀸, 𑀘𑀺𑀭𑁂𑀦 𑀲𑀢𑀺𑀁 𑀧𑀝𑀺𑀮𑀪𑀺𑀢𑁆𑀯𑀸, 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀓𑀁 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑁄 𑀫𑀬𑀸, 𑀪𑀡𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁄, 𑀤𑀺𑀦𑁆𑀦𑁄 𑀲𑁄 𑀆𑀭𑀸𑀫𑀺𑀓𑁄”𑀢𑀺? “𑀦 𑀔𑁄, 𑀤𑁂𑀯, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀺𑀓𑁄 𑀤𑀺𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 “𑀓𑀻𑀯 𑀘𑀺𑀭𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀡𑁂, 𑀇𑀢𑁄 𑀳𑀺 𑀢𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀕𑀡𑁂𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀜𑁆𑀘, 𑀤𑁂𑀯, 𑀭𑀢𑁆𑀢𑀺𑀲𑀢𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀆𑀭𑀸𑀫𑀺𑀓𑀲𑀢𑀸𑀦𑀺 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀆𑀭𑀸𑀫𑀺𑀓𑀲𑀢𑀸𑀦𑀺 𑀧𑀸𑀤𑀸𑀲𑀺, 𑀧𑀸𑀝𑀺𑀬𑁂𑀓𑁆𑀓𑁄 𑀕𑀸𑀫𑁄 𑀦𑀺𑀯𑀺𑀲𑀺𑁇 “𑀆𑀭𑀸𑀫𑀺𑀓𑀕𑀸𑀫𑀓𑁄”𑀢𑀺𑀧𑀺 𑀦𑀁 𑀆𑀳𑀁𑀲𑀼, “𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀕𑀸𑀫𑀓𑁄”𑀢𑀺𑀧𑀺 𑀦𑀁 𑀆𑀳𑀁𑀲𑀼𑁇

1243\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀸𑀫𑀓𑁂 𑀓𑀼𑀮𑀽𑀧𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀲𑁆𑀫𑀺𑀁 𑀕𑀸𑀫𑀓𑁂 𑀉𑀲𑁆𑀲𑀯𑁄 𑀳𑁄𑀢𑀺𑁇 𑀤𑀸𑀭𑀓𑀸 𑀅𑀮𑀗𑁆𑀓𑀢𑀸 𑀫𑀸𑀮𑀸𑀓𑀺𑀢𑀸 𑀓𑀻𑀍𑀅𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀕𑀸𑀫𑀓𑁂 𑀲𑀧𑀤𑀸𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀲𑁆𑀲𑀸 𑀆𑀭𑀸𑀫𑀺𑀓𑀺𑀦𑀺𑀬𑀸 𑀥𑀻𑀢𑀸 𑀅𑀜𑁆𑀜𑁂 𑀤𑀸𑀭𑀓𑁂 𑀅𑀮𑀗𑁆𑀓𑀢𑁂 𑀫𑀸𑀮𑀸𑀓𑀺𑀢𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀢𑀺—  “𑀫𑀸𑀮𑀁 𑀫𑁂 𑀤𑁂𑀣, 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑀁 𑀫𑁂 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑀺𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲𑀸𑀬𑀁 𑀤𑀸𑀭𑀺𑀓𑀸 𑀭𑁄𑀤𑀢𑀻”𑀢𑀺? “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀸𑀭𑀺𑀓𑀸 𑀅𑀜𑁆𑀜𑁂 𑀤𑀸𑀭𑀓𑁂 𑀅𑀮𑀗𑁆𑀓𑀢𑁂 𑀫𑀸𑀮𑀸𑀓𑀺𑀢𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀭𑁄𑀤𑀢𑀺—  ‘𑀫𑀸𑀮𑀁 𑀫𑁂 𑀤𑁂𑀣, 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑀁 𑀫𑁂 𑀤𑁂𑀣𑀸’𑀢𑀺𑁇 𑀓𑀼𑀢𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀸𑀦𑀁 𑀫𑀸𑀮𑀸, 𑀓𑀼𑀢𑁄 𑀅𑀮𑀗𑁆𑀓𑀸𑀭𑁄”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀢𑀺𑀡𑀡𑁆𑀟𑀼𑀧𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑀺𑀦𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀳𑀦𑁆𑀤𑀺𑀫𑀁 𑀢𑀺𑀡𑀡𑁆𑀟𑀼𑀧𑀓𑀁 𑀢𑀲𑁆𑀲𑀸 𑀤𑀸𑀭𑀺𑀓𑀸𑀬 𑀲𑀻𑀲𑁂 𑀧𑀝𑀺𑀫𑀼𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀆𑀭𑀸𑀫𑀺𑀓𑀺𑀦𑀻 𑀢𑀁 𑀢𑀺𑀡𑀡𑁆𑀟𑀼𑀧𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀤𑀸𑀭𑀺𑀓𑀸𑀬 𑀲𑀻𑀲𑁂 𑀧𑀝𑀺𑀫𑀼𑀜𑁆𑀘𑀺𑁇 𑀲𑀸 𑀅𑀳𑁄𑀲𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀸𑀮𑀸 𑀅𑀪𑀺𑀭𑀽𑀧𑀸, 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸, 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸; 𑀦𑀢𑁆𑀣𑀺 𑀢𑀸𑀤𑀺𑀲𑀸 𑀭𑀜𑁆𑀜𑁄𑀧𑀺 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀸𑀮𑀸𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁—  “𑀅𑀫𑀼𑀓𑀲𑁆𑀲, 𑀤𑁂𑀯, 𑀆𑀭𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀖𑀭𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀸𑀮𑀸 𑀅𑀪𑀺𑀭𑀽𑀧𑀸, 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸, 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸; 𑀦𑀢𑁆𑀣𑀺 𑀢𑀸𑀤𑀺𑀲𑀸 𑀤𑁂𑀯𑀲𑁆𑀲𑀧𑀺 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀸𑀮𑀸; 𑀓𑀼𑀢𑁄 𑀢𑀲𑁆𑀲 𑀤𑀼𑀕𑁆𑀕𑀢𑀲𑁆𑀲? 𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀘𑁄𑀭𑀺𑀓𑀸𑀬 𑀆𑀪𑀢𑀸”𑀢𑀺𑁇

1244\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑀓𑀼𑀮𑀁 𑀩𑀦𑁆𑀥𑀸𑀧𑁂𑀲𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀕𑀸𑀫𑀓𑁂 𑀲𑀧𑀤𑀸𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀢𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀯𑀺𑀲𑁆𑀲𑀓𑁂 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀓𑀳𑀁 𑀇𑀫𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑀓𑀼𑀮𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺? “𑀏𑀢𑀺𑀲𑁆𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀸𑀮𑀸𑀬 𑀓𑀸𑀭𑀡𑀸 𑀭𑀜𑁆𑀜𑀸 𑀩𑀦𑁆𑀥𑀸𑀧𑀺𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀆𑀭𑀸𑀫𑀺𑀓𑀓𑀼𑀮𑀁 𑀩𑀦𑁆𑀥𑀸𑀧𑀺𑀢𑀦𑁆”𑀢𑀺? “𑀢𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀭𑀸𑀫𑀺𑀓𑀲𑁆𑀲 𑀖𑀭𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀸𑀮𑀸 𑀅𑀪𑀺𑀭𑀽𑀧𑀸, 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑀸, 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀸; 𑀦𑀢𑁆𑀣𑀺 𑀢𑀸𑀤𑀺𑀲𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀫𑁆𑀧𑀺 𑀅𑀦𑁆𑀢𑁂𑀧𑀼𑀭𑁂 𑀲𑀼𑀯𑀡𑁆𑀡𑀫𑀸𑀮𑀸; 𑀓𑀼𑀢𑁄 𑀢𑀲𑁆𑀲 𑀤𑀼𑀕𑁆𑀕𑀢𑀲𑁆𑀲? 𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀘𑁄𑀭𑀺𑀓𑀸𑀬 𑀆𑀪𑀢𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀸𑀲𑀸𑀤𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑀦𑁆𑀢𑀺 𑀅𑀥𑀺𑀫𑀼𑀘𑁆𑀘𑀺; 𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑀩𑁆𑀩𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑁄𑁇 “𑀇𑀤𑀁 𑀧𑀦 𑀢𑁂, 𑀫𑀳𑀸𑀭𑀸𑀚, 𑀢𑀸𑀯 𑀩𑀳𑀼𑀁 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀓𑀼𑀢𑁄”𑀢𑀺? “𑀅𑀜𑁆𑀜𑀸𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲𑁂𑀯𑁂𑀲𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄”𑀢𑀺 𑀢𑀁 𑀆𑀭𑀸𑀫𑀺𑀓𑀓𑀼𑀮𑀁 𑀫𑀼𑀜𑁆𑀘𑀸𑀧𑁂𑀲𑀺𑁇

1245\. 𑀫𑀦𑀼𑀲𑁆𑀲𑀸—  “𑀅𑀬𑁆𑀬𑁂𑀦 𑀓𑀺𑀭 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁂𑀦 𑀲𑀭𑀸𑀚𑀺𑀓𑀸𑀬 𑀧𑀭𑀺𑀲𑀸𑀬 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀤𑀲𑁆𑀲𑀺𑀢𑀦𑁆”𑀢𑀺—  𑀅𑀢𑁆𑀢𑀫𑀦𑀸 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑀲𑁆𑀲 𑀧𑀜𑁆𑀘 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀅𑀪𑀺𑀳𑀭𑀺𑀁𑀲𑀼, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀧𑁆𑀧𑀺𑀁, 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀢𑁂𑀮𑀁, 𑀫𑀥𑀼𑀁, 𑀨𑀸𑀡𑀺𑀢𑀁𑁇 𑀧𑀓𑀢𑀺𑀬𑀸𑀧𑀺 𑀘 𑀆𑀬𑀲𑁆𑀫𑀸 𑀧𑀺𑀮𑀺𑀦𑁆𑀤𑀯𑀘𑁆𑀙𑁄 𑀮𑀸𑀪𑀻 𑀳𑁄𑀢𑀺 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀁; 𑀮𑀤𑁆𑀥𑀁 𑀮𑀤𑁆𑀥𑀁 𑀧𑀭𑀺𑀲𑀸𑀬 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀺𑁇 𑀧𑀭𑀺𑀲𑀸 𑀘𑀲𑁆𑀲 𑀳𑁄𑀢𑀺 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀺𑀓𑀸; 𑀮𑀤𑁆𑀥𑀁 𑀮𑀤𑁆𑀥𑀁 𑀓𑁄𑀮𑀫𑁆𑀩𑁂𑀧𑀺, 𑀖𑀝𑁂𑀧𑀺, 𑀧𑀽𑀭𑁂𑀢𑁆𑀯𑀸 𑀧𑀝𑀺𑀲𑀸𑀫𑁂𑀢𑀺; 𑀧𑀭𑀺𑀲𑁆𑀲𑀸𑀯𑀦𑀸𑀦𑀺𑀧𑀺, 𑀣𑀯𑀺𑀓𑀸𑀬𑁄𑀧𑀺, 𑀧𑀽𑀭𑁂𑀢𑁆𑀯𑀸 𑀯𑀸𑀢𑀧𑀸𑀦𑁂𑀲𑀼 𑀮𑀕𑁆𑀕𑁂𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀑𑀮𑀻𑀦𑀯𑀺𑀮𑀻𑀦𑀸𑀦𑀺 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺𑁇 𑀉𑀦𑁆𑀤𑀽𑀭𑁂𑀳𑀺𑀧𑀺 𑀯𑀺𑀳𑀸𑀭𑀸 𑀑𑀓𑀺𑀡𑁆𑀡𑀯𑀺𑀓𑀺𑀡𑁆𑀡𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀸𑀭𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀦𑁆𑀢𑁄𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀺𑀓𑀸 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄”𑀢𑀺𑁇

1246\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸, 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀘𑁂𑀢𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀘𑁂𑀢𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀸𑀦𑀺 𑀔𑁄 𑀧𑀦 𑀢𑀸𑀦𑀺 𑀕𑀺𑀮𑀸𑀦𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀝𑀺𑀲𑀸𑀬𑀦𑀻𑀬𑀸𑀦𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀲𑀧𑁆𑀧𑀺, 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀢𑁂𑀮𑀁, 𑀫𑀥𑀼, 𑀨𑀸𑀡𑀺𑀢𑀁, 𑀢𑀸𑀦𑀺 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀧𑀭𑀫𑀁 𑀲𑀦𑁆𑀦𑀺𑀥𑀺𑀓𑀸𑀭𑀓𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸𑀦𑀺𑁇 𑀢𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀸𑀫𑀬𑀢𑁄 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
