# 𑁬𑁇𑁨𑁦 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(295.)

1334\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀯𑁂𑀲𑀸𑀮𑀻 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀸 𑀳𑁄𑀢𑀺 𑀲𑀼𑀲𑀲𑁆𑀲𑀸 𑀲𑀼𑀮𑀪𑀧𑀺𑀡𑁆𑀟𑀸, 𑀲𑀼𑀓𑀭𑀸 𑀉𑀜𑁆𑀙𑁂𑀦 𑀧𑀕𑁆𑀕𑀳𑁂𑀦 𑀬𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺 𑀫𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸𑀦𑀺 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀤𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑁂 𑀤𑀼𑀮𑁆𑀮𑀪𑀧𑀺𑀡𑁆𑀟𑁂 𑀅𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀁 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑀺𑀢𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀓𑀁 𑀢𑀢𑁄 𑀦𑀻𑀳𑀝𑀁 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀯𑀦𑀝𑁆𑀞𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀝𑁆𑀞𑀁, 𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀫𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸𑀦𑀺 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀤𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑁂 𑀤𑀼𑀮𑁆𑀮𑀪𑀧𑀺𑀡𑁆𑀟𑁂 𑀅𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀁 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑀺𑀢𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀓𑀁 𑀢𑀢𑁄 𑀦𑀻𑀳𑀝𑀁 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀯𑀦𑀝𑁆𑀞𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀝𑁆𑀞𑀁, 𑀅𑀚𑁆𑀚𑀸𑀧𑀺 𑀦𑀼 𑀔𑁄 𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇

1335\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀸𑀦𑀺 𑀢𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸𑀦𑀺 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀤𑀼𑀲𑁆𑀲𑀲𑁆𑀲𑁂 𑀤𑀼𑀮𑁆𑀮𑀪𑀧𑀺𑀡𑁆𑀟𑁂 𑀅𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀁 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑀺𑀢𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀓𑀁 𑀢𑀢𑁄 𑀦𑀻𑀳𑀝𑀁 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀯𑀦𑀝𑁆𑀞𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀝𑁆𑀞𑀁, 𑀢𑀸𑀦𑀸𑀳𑀁 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀸𑀫𑀺𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀁 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁 𑀉𑀕𑁆𑀕𑀳𑀺𑀢𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀓𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀢𑁄 𑀦𑀻𑀳𑀝𑀁 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀯𑀦𑀝𑁆𑀞𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀝𑁆𑀞𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀦𑀸 𑀧𑀯𑀸𑀭𑀺𑀢𑁂𑀦 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

1336\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀚𑀸𑀦𑀧𑀤𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀩𑀳𑀼𑀁 𑀮𑁄𑀡𑀫𑁆𑀧𑀺, 𑀢𑁂𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀲𑀓𑀝𑁂𑀲𑀼 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀸𑀭𑀸𑀫𑀓𑁄𑀝𑁆𑀞𑀓𑁂 𑀲𑀓𑀝𑀧𑀭𑀺𑀯𑀝𑁆𑀝𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀬𑀤𑀸 𑀧𑀝𑀺𑀧𑀸𑀝𑀺𑀁 𑀮𑀪𑀺𑀲𑁆𑀲𑀸𑀫, 𑀢𑀤𑀸 𑀪𑀢𑁆𑀢𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀫𑀳𑀸 𑀘 𑀫𑁂𑀖𑁄 𑀉𑀕𑁆𑀕𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀁 𑀮𑁄𑀡𑀫𑁆𑀧𑀺, 𑀢𑁂𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀲𑀓𑀝𑁂𑀲𑀼 𑀆𑀭𑁄𑀧𑀺𑀢𑀸 𑀢𑀺𑀝𑁆𑀞𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸 𑀘 𑀫𑁂𑀖𑁄 𑀉𑀕𑁆𑀕𑀢𑁄; 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀸𑀦𑀦𑁆𑀤, 𑀲𑀁𑀖𑁄 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀁 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣 𑀯𑀸𑀲𑁂𑀢𑀼, 𑀬𑀁 𑀲𑀁𑀖𑁄 𑀆𑀓𑀗𑁆𑀔𑀢𑀺 𑀯𑀺𑀳𑀸𑀭𑀁 𑀯𑀸 𑀅𑀟𑁆𑀠𑀬𑁄𑀕𑀁 𑀯𑀸 𑀧𑀸𑀲𑀸𑀤𑀁 𑀯𑀸 𑀳𑀫𑁆𑀫𑀺𑀬𑀁 𑀯𑀸 𑀕𑀼𑀳𑀁 𑀯𑀸𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

1337\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑁂𑀬𑁆𑀬, 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

1338\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀯𑀺𑀳𑀸𑀭𑀁 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑀲𑁆𑀲 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀢𑀺, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

1339\. 𑀲𑀫𑁆𑀫𑀢𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

1340\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀬𑀸 𑀬𑀸𑀕𑀼𑀬𑁄 𑀧𑀘𑀦𑁆𑀢𑀺, 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀘𑀦𑁆𑀢𑀺, 𑀲𑀽𑀧𑀸𑀦𑀺 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀫𑀁𑀲𑀸𑀦𑀺 𑀓𑁄𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀨𑀸𑀮𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀁 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀁 𑀓𑀸𑀓𑁄𑀭𑀯𑀲𑀤𑁆𑀤𑀁, 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑁄 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑁄 𑀓𑀸𑀓𑁄𑀭𑀯𑀲𑀤𑁆𑀤𑁄”𑀢𑀺? “𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀬𑀸 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀬𑀸 𑀬𑀸𑀕𑀼𑀬𑁄 𑀧𑀘𑀦𑁆𑀢𑀺, 𑀪𑀢𑁆𑀢𑀸𑀦𑀺 𑀧𑀘𑀦𑁆𑀢𑀺, 𑀲𑀽𑀧𑀸𑀦𑀺 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺, 𑀫𑀁𑀲𑀸𑀦𑀺 𑀓𑁄𑀝𑁆𑀝𑁂𑀦𑁆𑀢𑀺, 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀨𑀸𑀮𑁂𑀦𑁆𑀢𑀺𑁇 𑀲𑁄 𑀏𑀲𑁄, 𑀪𑀕𑀯𑀸, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑁄 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑁄 𑀓𑀸𑀓𑁄𑀭𑀯𑀲𑀤𑁆𑀤𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀼𑀢𑀺 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀬𑁄—  𑀉𑀲𑁆𑀲𑀸𑀯𑀦𑀦𑁆𑀢𑀺𑀓𑀁 𑀕𑁄𑀦𑀺𑀲𑀸𑀤𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀦𑁆”𑀢𑀺𑁇

1341\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄𑀚𑁄 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺 𑀆𑀳𑀭𑀺𑀬𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀺 𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀉𑀓𑁆𑀓𑀧𑀺𑀡𑁆𑀟𑀓𑀸𑀧𑀺 𑀔𑀸𑀤𑀦𑁆𑀢𑀺, 𑀘𑁄𑀭𑀸𑀧𑀺 𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀁 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀲𑁆𑀲𑁄 𑀓𑀧𑁆𑀧𑀺𑀬𑀪𑀽𑀫𑀺𑀬𑁄—  𑀉𑀲𑁆𑀲𑀸𑀯𑀦𑀦𑁆𑀢𑀺𑀓𑀁 𑀕𑁄𑀦𑀺𑀲𑀸𑀤𑀺𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀲𑀫𑁆𑀫𑀼𑀢𑀺𑀦𑁆”𑀢𑀺𑁇

1342\. 𑀲𑀻𑀳𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀘𑀢𑀼𑀢𑁆𑀣𑁄𑁇
