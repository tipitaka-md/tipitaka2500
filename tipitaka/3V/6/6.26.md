# 𑁬𑁇𑁨𑁬 𑀘𑀢𑀼𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀓𑀣𑀸

* 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀘𑀢𑀼𑀫𑀳𑀸𑀧𑀤𑁂𑀲𑀓𑀣𑀸

(305.)

1367\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀓𑀺𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑀺 𑀓𑀺𑀲𑁆𑀫𑀺𑀜𑁆𑀘𑀺 𑀞𑀸𑀦𑁂 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁, 𑀓𑀺𑀁 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 ‘𑀇𑀤𑀁 𑀦 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀢𑀜𑁆𑀘𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑀺, 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀢𑀁 𑀯𑁄 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 ‘𑀇𑀤𑀁 𑀦 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺 𑀅𑀧𑁆𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀢𑀜𑁆𑀘𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑀺, 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀢𑀁 𑀯𑁄 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 ‘𑀇𑀤𑀁 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀢𑀜𑁆𑀘𑁂 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑀺, 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀢𑀁 𑀯𑁄 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 ‘𑀇𑀤𑀁 𑀓𑀧𑁆𑀧𑀢𑀻’𑀢𑀺 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀅𑀦𑀼𑀮𑁄𑀫𑁂𑀢𑀺, 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀧𑀝𑀺𑀩𑀸𑀳𑀢𑀺, 𑀢𑀁 𑀯𑁄 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇

1368\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀬𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁂𑀦 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀺? 𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀬𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀺? 𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀬𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁂𑀦 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀺? 𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑁂𑀦 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀺? 𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑁂𑀦 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀺? 𑀓𑀧𑁆𑀧𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑁂𑀦 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁, 𑀦 𑀦𑀼 𑀔𑁄 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

1369\. “𑀬𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑀁, 𑀢𑀤𑀳𑀼 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀓𑀸𑀮𑁂 𑀓𑀧𑁆𑀧𑀢𑀺, 𑀯𑀺𑀓𑀸𑀮𑁂 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀬𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁, 𑀢𑀤𑀳𑀼 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀓𑀸𑀮𑁂 𑀓𑀧𑁆𑀧𑀢𑀺, 𑀯𑀺𑀓𑀸𑀮𑁂 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀬𑀸𑀯𑀓𑀸𑀮𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁, 𑀢𑀤𑀳𑀼 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀓𑀸𑀮𑁂 𑀓𑀧𑁆𑀧𑀢𑀺, 𑀯𑀺𑀓𑀸𑀮𑁂 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑀁, 𑀢𑀤𑀳𑀼 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀬𑀸𑀫𑁂 𑀓𑀧𑁆𑀧𑀢𑀺, 𑀬𑀸𑀫𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀬𑀸𑀫𑀓𑀸𑀮𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁, 𑀢𑀤𑀳𑀼 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀬𑀸𑀫𑁂 𑀓𑀧𑁆𑀧𑀢𑀺, 𑀬𑀸𑀫𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀸𑀮𑀺𑀓𑁂𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀯𑀚𑀻𑀯𑀺𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁, 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀓𑀧𑁆𑀧𑀢𑀺, 𑀲𑀢𑁆𑀢𑀸𑀳𑀸𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀦 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇

1371\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1372\. _𑀲𑀸𑀭𑀤𑀺𑀓𑁂 𑀯𑀺𑀓𑀸𑀮𑁂𑀧𑀺,_  
_𑀯𑀲𑀁 𑀫𑀽𑀮𑁂 𑀧𑀺𑀝𑁆𑀞𑁂𑀳𑀺 𑀘;_  
_𑀓𑀲𑀸𑀯𑁂𑀳𑀺 𑀧𑀡𑁆𑀡𑀁 𑀨𑀮𑀁,_  
_𑀚𑀢𑀼 𑀮𑁄𑀡𑀁 𑀙𑀓𑀡𑀜𑁆𑀘𑁇_  


1373\. _𑀘𑀼𑀡𑁆𑀡𑀁 𑀘𑀸𑀮𑀺𑀦𑀺 𑀫𑀁𑀲𑀜𑁆𑀘,_  
_𑀅𑀜𑁆𑀚𑀦𑀁 𑀉𑀧𑀧𑀺𑀲𑀦𑀻;_  
_𑀅𑀜𑁆𑀚𑀦𑀻 𑀉𑀘𑁆𑀘𑀸𑀧𑀸𑀭𑀼𑀢𑀸,_  
_𑀲𑀮𑀸𑀓𑀸 𑀲𑀮𑀸𑀓𑀞𑀸𑀦𑀺𑀁𑁇_  


1374\. _𑀣𑀯𑀺𑀓𑀁𑀲𑀩𑀤𑁆𑀥𑀓𑀁 𑀲𑀼𑀢𑁆𑀢𑀁,_  
_𑀫𑀼𑀤𑁆𑀥𑀦𑀺𑀢𑁂𑀮𑀦𑀢𑁆𑀣𑀼 𑀘;_  
_𑀦𑀢𑁆𑀣𑀼𑀓𑀭𑀡𑀻 𑀥𑀽𑀫𑀜𑁆𑀘,_  
_𑀦𑁂𑀢𑁆𑀢𑀜𑁆𑀘𑀸𑀧𑀺𑀥𑀦𑀢𑁆𑀣𑀯𑀺𑁇_  


1375\. _𑀢𑁂𑀮𑀧𑀸𑀓𑁂𑀲𑀼 𑀫𑀚𑁆𑀚𑀜𑁆𑀘,_  
_𑀅𑀢𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀅𑀩𑁆𑀪𑀜𑁆𑀚𑀦𑀁;_  
_𑀢𑀼𑀫𑁆𑀩𑀁 𑀲𑁂𑀤𑀁 𑀲𑀫𑁆𑀪𑀸𑀭𑀜𑁆𑀘,_  
_𑀫𑀳𑀸 𑀪𑀗𑁆𑀕𑁄𑀤𑀓𑀁 𑀢𑀣𑀸𑁇_  


1376\. _𑀤𑀓𑀓𑁄𑀝𑁆𑀞𑀁 𑀮𑁄𑀳𑀺𑀢𑀜𑁆𑀘,_  
_𑀯𑀺𑀲𑀸𑀡𑀁 𑀧𑀸𑀤𑀩𑁆𑀪𑀜𑁆𑀚𑀦𑀁;_  
_𑀧𑀚𑁆𑀚𑀁 𑀲𑀢𑁆𑀣𑀁 𑀓𑀲𑀸𑀯𑀜𑁆𑀘,_  
_𑀢𑀺𑀮𑀓𑀓𑁆𑀓𑀁 𑀓𑀩𑀍𑀇𑀓𑀁𑁇_  


1377\. _𑀘𑁄𑀍𑀅𑀁 𑀲𑀸𑀲𑀧𑀓𑀼𑀝𑁆𑀝𑀜𑁆𑀘,_  
_𑀥𑀽𑀫 𑀲𑀓𑁆𑀔𑀭𑀺𑀓𑀸𑀬 𑀘;_  
_𑀯𑀡𑀢𑁂𑀮𑀁 𑀯𑀺𑀓𑀸𑀲𑀺𑀓𑀁,_  
_𑀯𑀺𑀓𑀝𑀜𑁆𑀘 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀁𑁇_  


1378\. _𑀕𑀽𑀣𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑁄 𑀮𑁄𑀍𑀇𑀜𑁆𑀘,_  
_𑀔𑀸𑀭𑀁 𑀫𑀼𑀢𑁆𑀢𑀳𑀭𑀻𑀢𑀓𑀁;_  
_𑀕𑀦𑁆𑀥𑀸 𑀯𑀺𑀭𑁂𑀘𑀦𑀜𑁆𑀘𑁂𑀯,_  
_𑀅𑀘𑁆𑀙𑀸𑀓𑀝𑀁 𑀓𑀝𑀸𑀓𑀝𑀁𑁇_  


1379\. _𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀺 𑀧𑀩𑁆𑀪𑀸𑀭𑀸,_  
_𑀆𑀭𑀸𑀫 𑀲𑀢𑁆𑀢𑀸𑀳𑁂𑀦 𑀘;_  
_𑀕𑀼𑀍𑀅𑀁 𑀫𑀼𑀕𑁆𑀕𑀁 𑀲𑁄𑀯𑀻𑀭𑀜𑁆𑀘,_  
_𑀲𑀸𑀫𑀁𑀧𑀸𑀓𑀸 𑀧𑀼𑀦𑀸𑀧𑀘𑁂𑁇_  


1380\. _𑀧𑀼𑀦𑀸𑀦𑀼𑀜𑁆𑀜𑀸𑀲𑀺 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂,_  
_𑀨𑀮𑀜𑁆𑀘 𑀢𑀺𑀮𑀔𑀸𑀤𑀦𑀻;_  
_𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀓𑀸𑀬𑀟𑀸𑀳𑁄,_  
_𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀜𑁆𑀘 𑀪𑀕𑀦𑁆𑀤𑀮𑀁𑁇_  


1381\. _𑀯𑀢𑁆𑀣𑀺𑀓𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑀼𑀧𑁆𑀧𑀺𑀜𑁆𑀘,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀫𑀁𑀲𑀫𑁂𑀯 𑀘;_  
_𑀳𑀢𑁆𑀣𑀺𑀅𑀲𑁆𑀲𑀸 𑀲𑀼𑀦𑀔𑁄 𑀘,_  
_𑀅𑀳𑀺 𑀲𑀻𑀳𑀜𑁆𑀘 𑀤𑀻𑀧𑀺𑀓𑀁𑁇_  


1382\. _𑀅𑀘𑁆𑀙𑀢𑀭𑀘𑁆𑀙𑀫𑀁𑀲𑀜𑁆𑀘,_  
_𑀧𑀝𑀺𑀧𑀸𑀝𑀺 𑀘 𑀬𑀸𑀕𑀼 𑀘;_  
_𑀢𑀭𑀼𑀡𑀁 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀕𑀼𑀍𑀅𑀁,_  
_𑀲𑀼𑀦𑀺𑀥𑀸𑀯𑀲𑀣𑀸𑀕𑀸𑀭𑀁𑁇_  


1383\. _𑀕𑀗𑁆𑀕𑀸 𑀓𑁄𑀝𑀺𑀲𑀘𑁆𑀘𑀓𑀣𑀸,_  
_𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀻 𑀘 𑀮𑀺𑀘𑁆𑀙𑀯𑀻;_  
_𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀓𑀢𑀁 𑀲𑀼𑀪𑀺𑀓𑁆𑀔𑀁,_  
_𑀧𑀼𑀦𑀤𑁂𑀯 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑁇_  


1384\. _𑀫𑁂𑀖𑁄 𑀬𑀲𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀘,_  
_𑀕𑁄𑀭𑀲𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀓𑁂𑀦 𑀘;_  
_𑀓𑁂𑀡𑀺 𑀅𑀫𑁆𑀩𑁄 𑀚𑀫𑁆𑀩𑀼 𑀘𑁄𑀘,_  
_𑀫𑁄𑀘𑀫𑀥𑀼𑀫𑀼𑀤𑁆𑀤𑀺𑀓𑀲𑀸𑀮𑀼𑀓𑀁𑁇_  


1385\. _𑀨𑀸𑀭𑀼𑀲𑀓𑀸 𑀟𑀸𑀓𑀧𑀺𑀝𑁆𑀞𑀁,_  
_𑀆𑀢𑀼𑀫𑀸𑀬𑀁 𑀦𑀳𑀸𑀧𑀺𑀢𑁄;_  
_𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀨𑀮𑀁 𑀩𑀻𑀚𑀁,_  
_𑀓𑀺𑀲𑁆𑀫𑀺𑀁 𑀞𑀸𑀦𑁂 𑀘 𑀓𑀸𑀮𑀺𑀓𑁂𑀢𑀺𑁇_  


---

1386\. 𑀇𑀫𑀫𑁆𑀳𑀺 𑀔𑀦𑁆𑀥𑀓𑁂 𑀯𑀢𑁆𑀣𑀽 𑀏𑀓𑀲𑀢𑀁 𑀙𑀯𑀢𑁆𑀣𑀼𑁇



---

1387\. 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
