# 𑁬𑁇𑁫 𑀅𑀦𑁆𑀢𑁄𑀯𑀼𑀝𑁆𑀞𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀦𑁆𑀢𑁄𑀯𑀼𑀝𑁆𑀞𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

(274.)

1251. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀉𑀤𑀭𑀯𑀸𑀢𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄—  “𑀧𑀼𑀩𑁆𑀩𑁂𑀧𑀺 𑀪𑀕𑀯𑀢𑁄 𑀉𑀤𑀭𑀯𑀸𑀢𑀸𑀩𑀸𑀥𑁄 𑀢𑁂𑀓𑀝𑀼𑀮𑀬𑀸𑀕𑀼𑀬𑀸 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀻”𑀢𑀺—  𑀲𑀸𑀫𑀁 𑀢𑀺𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀫𑀼𑀕𑁆𑀕𑀫𑁆𑀧𑀺 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑁆𑀯𑀸, 𑀅𑀦𑁆𑀢𑁄 𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸, 𑀅𑀦𑁆𑀢𑁄 𑀲𑀸𑀫𑀁 𑀧𑀘𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀧𑀺𑀯𑀢𑀼 𑀪𑀕𑀯𑀸 𑀢𑁂𑀓𑀝𑀼𑀮𑀬𑀸𑀕𑀼𑀦𑁆”𑀢𑀺𑁇 𑀚𑀸𑀦𑀦𑁆𑀢𑀸𑀧𑀺 𑀢𑀣𑀸𑀕𑀢𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀚𑀸𑀦𑀦𑁆𑀢𑀸𑀧𑀺 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀓𑀸𑀮𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀓𑀸𑀮𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀦 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺; 𑀅𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁 𑀢𑀣𑀸𑀕𑀢𑀸 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑀁𑁇 𑀅𑀦𑀢𑁆𑀣𑀲𑀁𑀳𑀺𑀢𑁂 𑀲𑁂𑀢𑀼𑀖𑀸𑀢𑁄 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀁𑁇 𑀤𑁆𑀯𑀻𑀳𑀺 𑀆𑀓𑀸𑀭𑁂𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺—  “𑀥𑀫𑁆𑀫𑀁 𑀯𑀸 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫, 𑀲𑀸𑀯𑀓𑀸𑀦𑀁 𑀯𑀸 𑀲𑀺𑀓𑁆𑀔𑀸𑀧𑀤𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

1252. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀼𑀢𑀸𑀬𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀬𑀸𑀕𑀽”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸—  “𑀅𑀦𑀦𑀼𑀘𑁆𑀙𑀯𑀺𑀓𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑀦𑀼𑀮𑁄𑀫𑀺𑀓𑀁, 𑀅𑀧𑁆𑀧𑀢𑀺𑀭𑀽𑀧𑀁, 𑀅𑀲𑁆𑀲𑀸𑀫𑀡𑀓𑀁, 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁, 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀏𑀯𑀭𑀽𑀧𑀸𑀬 𑀩𑀸𑀳𑀼𑀮𑁆𑀮𑀸𑀬 𑀘𑁂𑀢𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀬𑀤𑀧𑀺, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀁 𑀢𑀤𑀧𑀺 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁; 𑀬𑀤𑀧𑀺 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀤𑀧𑀺 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁; 𑀬𑀤𑀧𑀺 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀤𑀧𑀺 𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀁𑁇 𑀦𑁂𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁, 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑁆𑀢𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁, 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀢𑀺𑀡𑁆𑀡𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇 𑀅𑀦𑁆𑀢𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁, 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇 𑀅𑀦𑁆𑀢𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀩𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇 𑀩𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁, 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀦𑀁𑁇 𑀅𑀦𑁆𑀢𑁄 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀩𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀩𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁, 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀩𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀩𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀩𑀳𑀺 𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀩𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀅𑀜𑁆𑀜𑁂𑀳𑀺 𑀧𑀓𑁆𑀓𑀁, 𑀢𑀜𑁆𑀘𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀻”𑀢𑀺𑁇

1253. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀲𑀸𑀫𑀁𑀧𑀸𑀓𑁄 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄”𑀢𑀺—  𑀧𑀼𑀦 𑀧𑀸𑀓𑁂 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀦 𑀧𑀸𑀓𑀁 𑀧𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1254. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑀁 𑀳𑁄𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀮𑁄𑀡𑀫𑁆𑀧𑀺, 𑀢𑁂𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀆𑀭𑀸𑀫𑀁 𑀆𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀺 𑀯𑀸𑀲𑁂𑀦𑁆𑀢𑀺; 𑀉𑀓𑁆𑀓𑀧𑀺𑀡𑁆𑀟𑀓𑀸𑀧𑀺 𑀔𑀸𑀤𑀦𑁆𑀢𑀺, 𑀘𑁄𑀭𑀸𑀧𑀺 𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑁄 𑀯𑀸𑀲𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑁄 𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺 𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺𑁇 𑀤𑀫𑀓𑀸 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑁄 𑀧𑀘𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀓𑀸𑀭𑀓𑀸 𑀩𑀳𑀼𑀢𑀭𑀁 𑀳𑀭𑀦𑁆𑀢𑀺, 𑀅𑀧𑁆𑀧𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀁 𑀧𑀘𑀺𑀢𑀼𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑁄 𑀯𑀼𑀝𑁆𑀞𑀁, 𑀅𑀦𑁆𑀢𑁄 𑀧𑀓𑁆𑀓𑀁, 𑀲𑀸𑀫𑀁 𑀧𑀓𑁆𑀓𑀦𑁆”𑀢𑀺𑁇
