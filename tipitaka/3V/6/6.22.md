# 𑁬𑁇𑁨𑁨 𑀧𑀜𑁆𑀘𑀕𑁄𑀭𑀲𑀸𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀜𑁆𑀘𑀕𑁄𑀭𑀲𑀸𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(299.)

1350. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑁂𑀡𑁆𑀟𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀬𑁂𑀦 𑀅𑀗𑁆𑀕𑀼𑀢𑁆𑀢𑀭𑀸𑀧𑁄 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀬𑁂𑀦 𑀅𑀗𑁆𑀕𑀼𑀢𑁆𑀢𑀭𑀸𑀧𑁄 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀤𑀸𑀲𑁂 𑀘 𑀓𑀫𑁆𑀫𑀓𑀭𑁂 𑀘 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀩𑀳𑀼𑀁 𑀮𑁄𑀡𑀫𑁆𑀧𑀺, 𑀢𑁂𑀮𑀫𑁆𑀧𑀺, 𑀢𑀡𑁆𑀟𑀼𑀮𑀫𑁆𑀧𑀺, 𑀔𑀸𑀤𑀦𑀻𑀬𑀫𑁆𑀧𑀺 𑀲𑀓𑀝𑁂𑀲𑀼 𑀆𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀣, 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑀸𑀦𑀺 𑀘 𑀕𑁄𑀧𑀸𑀮𑀓𑀲𑀢𑀸𑀦𑀺 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑀸𑀦𑀺 𑀘 𑀥𑁂𑀦𑀼𑀲𑀢𑀸𑀦𑀺 𑀆𑀤𑀸𑀬 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼, 𑀬𑀢𑁆𑀣 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫 𑀢𑀢𑁆𑀣 𑀢𑀭𑀼𑀡𑁂𑀦 𑀔𑀻𑀭𑁂𑀦 𑀪𑁄𑀚𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀓𑀦𑁆𑀢𑀸𑀭𑁂 𑀲𑀫𑁆𑀪𑀸𑀯𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1351. 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀧𑀭𑀺𑀯𑁂𑀲𑀦𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑀸𑀦𑀺 𑀕𑁄𑀧𑀸𑀮𑀓𑀲𑀢𑀸𑀦𑀺 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀏𑀓𑀫𑁂𑀓𑀁 𑀥𑁂𑀦𑀼𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀧𑀢𑀺𑀝𑁆𑀞𑀣 𑀢𑀭𑀼𑀡𑁂𑀦 𑀔𑀻𑀭𑁂𑀦 𑀪𑁄𑀚𑁂𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀲𑀺, 𑀢𑀭𑀼𑀡𑁂𑀦 𑀘 𑀔𑀻𑀭𑁂𑀦𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀔𑀻𑀭𑀁 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀢𑀭𑀼𑀡𑁂𑀦 𑀘 𑀔𑀻𑀭𑁂𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀕𑁆𑀕𑀸 𑀓𑀦𑁆𑀢𑀸𑀭𑀸, 𑀅𑀧𑁆𑀧𑁄𑀤𑀓𑀸 𑀅𑀧𑁆𑀧𑀪𑀓𑁆𑀔𑀸, 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀅𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑁂𑀦 𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑁂𑀡𑁆𑀟𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1352. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘 𑀕𑁄𑀭𑀲𑁂—  𑀔𑀻𑀭𑀁, 𑀤𑀥𑀺𑀁, 𑀢𑀓𑁆𑀓𑀁, 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀲𑀧𑁆𑀧𑀺𑀁𑁇 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀕𑁆𑀕𑀸 𑀓𑀦𑁆𑀢𑀸𑀭𑀸 𑀅𑀧𑁆𑀧𑁄𑀤𑀓𑀸 𑀅𑀧𑁆𑀧𑀪𑀓𑁆𑀔𑀸, 𑀦 𑀲𑀼𑀓𑀭𑀸 𑀅𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑁂𑀦 𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀸𑀣𑁂𑀬𑁆𑀬𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀼𑀁 𑀢𑀡𑁆𑀟𑀼𑀮𑁄 𑀢𑀡𑁆𑀟𑀼𑀮𑀢𑁆𑀣𑀺𑀓𑁂𑀦, 𑀫𑀼𑀕𑁆𑀕𑁄 𑀫𑀼𑀕𑁆𑀕𑀢𑁆𑀣𑀺𑀓𑁂𑀦, 𑀫𑀸𑀲𑁄 𑀫𑀸𑀲𑀢𑁆𑀣𑀺𑀓𑁂𑀦, 𑀮𑁄𑀡𑀁 𑀮𑁄𑀡𑀢𑁆𑀣𑀺𑀓𑁂𑀦, 𑀕𑀼𑀍𑀑 𑀕𑀼𑀍𑀅𑀢𑁆𑀣𑀺𑀓𑁂𑀦, 𑀢𑁂𑀮𑀁 𑀢𑁂𑀮𑀢𑁆𑀣𑀺𑀓𑁂𑀦, 𑀲𑀧𑁆𑀧𑀺 𑀲𑀧𑁆𑀧𑀺𑀢𑁆𑀣𑀺𑀓𑁂𑀦𑁇 𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸, 𑀲𑀤𑁆𑀥𑀸 𑀧𑀲𑀦𑁆𑀦𑀸, 𑀢𑁂 𑀓𑀧𑁆𑀧𑀺𑀬𑀓𑀸𑀭𑀓𑀸𑀦𑀁 𑀳𑀢𑁆𑀣𑁂 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀺—  ‘𑀇𑀫𑀺𑀦𑀸 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀬𑀁 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀢𑀁 𑀤𑁂𑀣𑀸’𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀁 𑀢𑀢𑁄 𑀓𑀧𑁆𑀧𑀺𑀬𑀁 𑀢𑀁 𑀲𑀸𑀤𑀺𑀢𑀼𑀁; 𑀦 𑀢𑁆𑀯𑁂𑀯𑀸𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑁂𑀦𑀘𑀺 𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀚𑀸𑀢𑀭𑀽𑀧𑀭𑀚𑀢𑀁 𑀲𑀸𑀤𑀺𑀢𑀩𑁆𑀩𑀁 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀢𑀩𑁆𑀩𑀦𑁆𑀢𑀺 𑀯𑀤𑀸𑀫𑀻”𑀢𑀺𑁇
