# 𑁬𑁇𑁧𑁦 𑀳𑀢𑁆𑀣𑀺𑀫𑀁𑀲𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀳𑀢𑁆𑀣𑀺𑀫𑀁𑀲𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

(281.)

1271\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀳𑀢𑁆𑀣𑀻 𑀫𑀭𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀳𑀢𑁆𑀣𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀳𑀢𑁆𑀣𑀺𑀫𑀁𑀲𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑀢𑁆𑀣𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑀢𑁆𑀣𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀭𑀸𑀚𑀗𑁆𑀕𑀁 𑀳𑀢𑁆𑀣𑀻, 𑀲𑀘𑁂 𑀭𑀸𑀚𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀦 𑀦𑁂𑀲𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀳𑀢𑁆𑀣𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1272\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀅𑀲𑁆𑀲𑀸 𑀫𑀭𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀅𑀲𑁆𑀲𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀫𑀁𑀲𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀲𑁆𑀲𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀲𑁆𑀲𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀭𑀸𑀚𑀗𑁆𑀕𑀁 𑀅𑀲𑁆𑀲𑀸, 𑀲𑀘𑁂 𑀭𑀸𑀚𑀸 𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀦 𑀦𑁂𑀲𑀁 𑀅𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁆𑀲𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1273\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀲𑀼𑀦𑀔𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀼𑀦𑀔𑀫𑀁𑀲𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀼𑀦𑀔𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀦𑀔𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑁄 𑀲𑀼𑀦𑀔𑁄 𑀧𑀝𑀺𑀓𑀽𑀮𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀦𑀔𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1274\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀅𑀳𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀳𑀺𑀫𑀁𑀲𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀳𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀳𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀚𑁂𑀕𑀼𑀘𑁆𑀙𑁄 𑀅𑀳𑀺 𑀧𑀝𑀺𑀓𑀽𑀮𑁄”𑀢𑀺𑁇 𑀲𑀼𑀧𑀲𑁆𑀲𑁄𑀧𑀺 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀲𑀼𑀧𑀲𑁆𑀲𑁄 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀸𑀕𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑁇 𑀢𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀳𑀺𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑁂𑀞𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀅𑀳𑀺𑀫𑀁𑀲𑀁 𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀧𑀲𑁆𑀲𑀁 𑀦𑀸𑀕𑀭𑀸𑀚𑀸𑀦𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺…𑀧𑁂…  𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀳𑀺𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1275\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀮𑀼𑀤𑁆𑀤𑀓𑀸 𑀲𑀻𑀳𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀻𑀳𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀲𑀻𑀳𑀫𑀁𑀲𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀳𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀲𑀻𑀳𑀸 𑀲𑀻𑀳𑀫𑀁𑀲𑀕𑀦𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀭𑀺𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀳𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1276\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀮𑀼𑀤𑁆𑀤𑀓𑀸 𑀩𑁆𑀬𑀕𑁆𑀖𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀤𑀻𑀧𑀺𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀅𑀘𑁆𑀙𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀢𑀭𑀘𑁆𑀙𑀁 𑀳𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀭𑀘𑁆𑀙𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀸𑀦𑀁 𑀢𑀭𑀘𑁆𑀙𑀫𑀁𑀲𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀭𑀘𑁆𑀙𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑀭𑀘𑁆𑀙𑀸 𑀢𑀭𑀘𑁆𑀙𑀫𑀁𑀲𑀕𑀦𑁆𑀥𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀭𑀺𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀭𑀘𑁆𑀙𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
