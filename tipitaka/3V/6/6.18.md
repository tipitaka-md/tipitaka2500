# 𑁬𑁇𑁧𑁮 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀯𑀢𑁆𑀣𑀼

* 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀯𑀢𑁆𑀣𑀼

(289.)

1313\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀻—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀯𑁂𑀲𑀸𑀮𑀺𑀓𑀸 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀪𑀤𑁆𑀭𑀸𑀦𑀺 𑀪𑀤𑁆𑀭𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀭𑀁 𑀪𑀤𑁆𑀭𑀁 𑀬𑀸𑀦𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀭𑁂𑀳𑀺 𑀪𑀤𑁆𑀭𑁂𑀳𑀺 𑀬𑀸𑀦𑁂𑀳𑀺 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀼𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀦𑀻𑀮𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀦𑀻𑀮𑀯𑀡𑁆𑀡𑀸 𑀦𑀻𑀮𑀯𑀢𑁆𑀣𑀸 𑀦𑀻𑀮𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀧𑀻𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀻𑀢𑀯𑀡𑁆𑀡𑀸 𑀧𑀻𑀢𑀯𑀢𑁆𑀣𑀸 𑀧𑀻𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀮𑁄𑀳𑀺𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀮𑁄𑀳𑀺𑀢𑀯𑀡𑁆𑀡𑀸 𑀮𑁄𑀳𑀺𑀢𑀯𑀢𑁆𑀣𑀸 𑀮𑁄𑀳𑀺𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀑𑀤𑀸𑀢𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀑𑀤𑀸𑀢𑀯𑀡𑁆𑀡𑀸 𑀑𑀤𑀸𑀢𑀯𑀢𑁆𑀣𑀸 𑀑𑀤𑀸𑀢𑀸𑀮𑀗𑁆𑀓𑀸𑀭𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀻 𑀕𑀡𑀺𑀓𑀸 𑀤𑀳𑀭𑀸𑀦𑀁 𑀤𑀳𑀭𑀸𑀦𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀦𑀁 𑀈𑀲𑀸𑀬 𑀈𑀲𑀁 𑀬𑀼𑀕𑁂𑀦 𑀬𑀼𑀕𑀁 𑀘𑀓𑁆𑀓𑁂𑀦 𑀘𑀓𑁆𑀓𑀁 𑀅𑀓𑁆𑀔𑁂𑀦 𑀅𑀓𑁆𑀔𑀁 𑀧𑀝𑀺𑀯𑀝𑁆𑀝𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀁 𑀕𑀡𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀺𑀲𑁆𑀲, 𑀚𑁂 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺, 𑀤𑀳𑀭𑀸𑀦𑀁 𑀤𑀳𑀭𑀸𑀦𑀁 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀦𑀁 𑀈𑀲𑀸𑀬 𑀈𑀲𑀁 𑀬𑀼𑀕𑁂𑀦 𑀬𑀼𑀕𑀁 𑀘𑀓𑁆𑀓𑁂𑀦 𑀘𑀓𑁆𑀓𑀁 𑀅𑀓𑁆𑀔𑁂𑀦 𑀅𑀓𑁆𑀔𑀁 𑀧𑀝𑀺𑀯𑀝𑁆𑀝𑁂𑀲𑀻”𑀢𑀺? “𑀢𑀣𑀸 𑀳𑀺 𑀧𑀦 𑀫𑀬𑀸, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀸, 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄”𑀢𑀺𑁇 “𑀤𑁂𑀳𑀺, 𑀚𑁂 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀏𑀢𑀁 𑀪𑀢𑁆𑀢𑀁 𑀲𑀢𑀲𑀳𑀲𑁆𑀲𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀲𑀘𑁂𑀧𑀺 𑀫𑁂, 𑀅𑀬𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀸, 𑀯𑁂𑀲𑀸𑀮𑀺𑀁 𑀲𑀸𑀳𑀸𑀭𑀁 𑀤𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀣, 𑀦𑁂𑀯 𑀤𑀚𑁆𑀚𑀸𑀳𑀁 𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀁 𑀨𑁄𑀝𑁂𑀲𑀼𑀁—  “𑀚𑀺𑀢𑀫𑁆𑀳 𑀯𑀢 𑀪𑁄 𑀅𑀫𑁆𑀩𑀓𑀸𑀬, 𑀧𑀭𑀸𑀚𑀺𑀢𑀫𑁆𑀳 𑀯𑀢 𑀪𑁄 𑀅𑀫𑁆𑀩𑀓𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑁂𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀤𑁂𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀸 𑀅𑀤𑀺𑀝𑁆𑀞𑀧𑀼𑀩𑁆𑀩𑀸, 𑀑𑀮𑁄𑀓𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀧𑀭𑀺𑀲𑀁; 𑀅𑀧𑀮𑁄𑀓𑁂𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀧𑀭𑀺𑀲𑀁; 𑀉𑀧𑀲𑀁𑀳𑀭𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀺𑀘𑁆𑀙𑀯𑀻𑀧𑀭𑀺𑀲𑀁 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀧𑀭𑀺𑀲𑀦𑁆”𑀢𑀺𑁇

1314\. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀬𑀸𑀦𑀲𑁆𑀲 𑀪𑀽𑀫𑀺, 𑀬𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑀸𑀦𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑀸𑀯 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁂 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻, 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑀸 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀅𑀥𑀺𑀯𑀼𑀝𑁆𑀞𑁄𑀫𑁆𑀳𑀺, 𑀮𑀺𑀘𑁆𑀙𑀯𑀻, 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀬𑀸 𑀕𑀡𑀺𑀓𑀸𑀬 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀁 𑀨𑁄𑀝𑁂𑀲𑀼𑀁—  “𑀚𑀺𑀢𑀫𑁆𑀳 𑀯𑀢 𑀪𑁄 𑀅𑀫𑁆𑀩𑀓𑀸𑀬, 𑀧𑀭𑀸𑀚𑀺𑀢𑀫𑁆𑀳 𑀯𑀢 𑀪𑁄 𑀅𑀫𑁆𑀩𑀓𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀮𑀺𑀘𑁆𑀙𑀯𑀻 𑀪𑀕𑀯𑀢𑁄 𑀪𑀸𑀲𑀺𑀢𑀁 𑀅𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑀼𑀫𑁄𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇

1315\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀦𑀸𑀢𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀦𑀸𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀜𑁆𑀚𑀓𑀸𑀯𑀲𑀣𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀻 𑀕𑀡𑀺𑀓𑀸 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀓𑁂 𑀆𑀭𑀸𑀫𑁂 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀬𑀸 𑀕𑀡𑀺𑀓𑀸𑀬 𑀧𑀭𑀺𑀯𑁂𑀲𑀦𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀻 𑀕𑀡𑀺𑀓𑀸 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀻 𑀕𑀡𑀺𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀩𑀯𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀆𑀭𑀸𑀫𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀫𑁆𑀩𑀧𑀸𑀮𑀺𑀁 𑀕𑀡𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀫𑀳𑀸𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀫𑀳𑀸𑀯𑀦𑁂 𑀓𑀽𑀝𑀸𑀕𑀸𑀭𑀲𑀸𑀮𑀸𑀬𑀁𑁇
