# 𑁬𑁇𑁧𑁬 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁂𑀲𑀘𑁆𑀘𑀓𑀣𑀸

* 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁂𑀲𑀘𑁆𑀘𑀓𑀣𑀸

(287.)

1309\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀓𑁄𑀝𑀺𑀕𑀸𑀫𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀘𑀢𑀼𑀦𑁆𑀦𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀓𑀢𑀫𑁂𑀲𑀁 𑀘𑀢𑀼𑀦𑁆𑀦𑀁? 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀲𑁆𑀲 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲…𑀧𑁂…  𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀺𑀬𑀸 𑀧𑀝𑀺𑀧𑀤𑀸𑀬 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀲𑁆𑀲 𑀅𑀦𑀦𑀼𑀩𑁄𑀥𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀥𑀸 𑀏𑀯𑀫𑀺𑀤𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀦𑁆𑀥𑀸𑀯𑀺𑀢𑀁 𑀲𑀁𑀲𑀭𑀺𑀢𑀁 𑀫𑀫𑀜𑁆𑀘𑁂𑀯 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀘𑁇 𑀢𑀬𑀺𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀤𑀼𑀓𑁆𑀔𑀲𑀫𑀼𑀤𑀬𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀤𑀼𑀓𑁆𑀔𑀦𑀺𑀭𑁄𑀥𑀕𑀸𑀫𑀺𑀦𑀻 𑀧𑀝𑀺𑀧𑀤𑀸 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀁 𑀅𑀦𑀼𑀩𑀼𑀤𑁆𑀥𑀁 𑀧𑀝𑀺𑀯𑀺𑀤𑁆𑀥𑀁, 𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀪𑀯𑀢𑀡𑁆𑀳𑀸, 𑀔𑀻𑀡𑀸 𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺, 𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄𑀢𑀺𑁇

1310\. _𑀘𑀢𑀼𑀦𑁆𑀦𑀁 𑀅𑀭𑀺𑀬𑀲𑀘𑁆𑀘𑀸𑀦𑀁,_  
_𑀬𑀣𑀸𑀪𑀽𑀢𑀁 𑀅𑀤𑀲𑁆𑀲𑀦𑀸;_  
_𑀲𑀁𑀲𑀺𑀢𑀁 𑀤𑀻𑀖𑀫𑀤𑁆𑀥𑀸𑀦𑀁,_  
_𑀢𑀸𑀲𑀼 𑀢𑀸𑀲𑁆𑀯𑁂𑀯 𑀚𑀸𑀢𑀺𑀲𑀼𑁇_  


1311\. _𑀢𑀸𑀦𑀺 𑀏𑀢𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀸𑀦𑀺,_  
_𑀪𑀯𑀦𑁂𑀢𑁆𑀢𑀺 𑀲𑀫𑀽𑀳𑀢𑀸;_  
_𑀉𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀁 𑀫𑀽𑀮𑀁 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲,_  
_𑀦𑀢𑁆𑀣𑀺 𑀤𑀸𑀦𑀺 𑀧𑀼𑀦𑀩𑁆𑀪𑀯𑁄”𑀢𑀺𑁇_
