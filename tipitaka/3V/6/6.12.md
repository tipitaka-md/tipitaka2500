# 𑁬𑁇𑁧𑁨 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

1286\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀕𑀯𑀢𑀸 𑀓𑀺𑀭 𑀬𑀸𑀕𑀼 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸 𑀫𑀥𑀼𑀕𑁄𑀍𑀅𑀓𑀜𑁆𑀘𑀸𑀢𑀺𑁇 𑀢𑁂 𑀓𑀸𑀮𑀲𑁆𑀲𑁂𑀯, 𑀪𑁄𑀚𑁆𑀚𑀬𑀸𑀕𑀼𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀫𑀥𑀼𑀕𑁄𑀍𑀅𑀓𑀜𑁆𑀘𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀸𑀮𑀲𑁆𑀲𑁂𑀯 𑀪𑁄𑀚𑁆𑀚𑀬𑀸𑀕𑀼𑀬𑀸 𑀥𑀸𑀢𑀸 𑀫𑀥𑀼𑀕𑁄𑀍𑀅𑀓𑁂𑀦 𑀘 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀦 𑀘𑀺𑀢𑁆𑀢𑀭𑀽𑀧𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁂𑀦 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂𑀦 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑀦𑁆𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑀸𑀦𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑀸𑀦𑀺 𑀫𑀁𑀲𑀧𑀸𑀢𑀺𑀲𑀢𑀸𑀦𑀺 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀬𑁆𑀬𑀁, 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀫𑁂𑀓𑀁 𑀫𑀁𑀲𑀧𑀸𑀢𑀺𑀁 𑀉𑀧𑀦𑀸𑀫𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑀸𑀦𑀺 𑀘 𑀫𑀁𑀲𑀧𑀸𑀢𑀺𑀲𑀢𑀸𑀦𑀺, 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀢𑀲𑁆𑀲 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺, 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀭𑀺𑀯𑀺𑀲𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀣𑁄𑀓𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁂𑀳𑀺; 𑀣𑁄𑀓𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀫𑀸 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀪𑀦𑁆𑀢𑁂—  ‘𑀅𑀬𑀁 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄’𑀢𑀺—  𑀣𑁄𑀓𑀁 𑀣𑁄𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀣𑁇 𑀩𑀳𑀼𑀁 𑀫𑁂 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀢𑁆𑀢𑀁, 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑀸𑀦𑀺 𑀘 𑀫𑀁𑀲𑀧𑀸𑀢𑀺𑀲𑀢𑀸𑀦𑀺𑁇 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀓𑀫𑁂𑀓𑀁 𑀫𑀁𑀲𑀧𑀸𑀢𑀺𑀁 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀦𑁆”𑀢𑀺𑁇 “𑀦 𑀔𑁄 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀁𑀓𑀸𑀭𑀡𑀸 𑀣𑁄𑀓𑀁 𑀣𑁄𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫, 𑀅𑀧𑀺 𑀘 𑀫𑀬𑀁 𑀓𑀸𑀮𑀲𑁆𑀲𑁂𑀯 𑀪𑁄𑀚𑁆𑀚𑀬𑀸𑀕𑀼𑀬𑀸 𑀥𑀸𑀢𑀸 𑀫𑀥𑀼𑀕𑁄𑀍𑀅𑀓𑁂𑀦 𑀘𑁇 𑀢𑁂𑀦 𑀫𑀬𑀁 𑀣𑁄𑀓𑀁 𑀣𑁄𑀓𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀤𑀦𑁆𑀢𑀸 𑀫𑀬𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑁄𑀚𑁆𑀚𑀬𑀸𑀕𑀼𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀘𑀸𑀳𑀁 𑀧𑀝𑀺𑀩𑀮𑁄 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀲𑀸𑀤𑀦𑀸𑀧𑁂𑀓𑁆𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀢𑁆𑀢𑁂 𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑁄 𑀅𑀕𑀫𑀸𑀲𑀺—  “𑀪𑀼𑀜𑁆𑀚𑀣 𑀯𑀸 𑀳𑀭𑀣 𑀯𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀢𑀁 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑀁 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1287\. 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑀲𑁆𑀲 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀳𑀼𑀤𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁, 𑀅𑀳𑀼 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄—  “𑀅𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀮𑀸𑀪𑀸; 𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀲𑀼𑀮𑀤𑁆𑀥𑀁; 𑀬𑁄𑀳𑀁 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀲𑀸𑀤𑀦𑀸𑀧𑁂𑀓𑁆𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀢𑁆𑀢𑁂 𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑁄 𑀅𑀕𑀫𑀸𑀲𑀺𑀁—  ‘𑀪𑀼𑀜𑁆𑀚𑀣 𑀯𑀸 𑀳𑀭𑀣 𑀯𑀸’𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀩𑀳𑀼𑀁 𑀧𑀲𑀼𑀢𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑀸 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀢𑁄 𑀅𑀳𑀼𑀤𑁂𑀯 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁, 𑀅𑀳𑀼 𑀯𑀺𑀧𑁆𑀧𑀝𑀺𑀲𑀸𑀭𑁄 ‘𑀅𑀮𑀸𑀪𑀸 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀮𑀸𑀪𑀸; 𑀤𑀼𑀮𑁆𑀮𑀤𑁆𑀥𑀁 𑀯𑀢 𑀫𑁂, 𑀦 𑀯𑀢 𑀫𑁂 𑀲𑀼𑀮𑀤𑁆𑀥𑀁; 𑀬𑁄𑀳𑀁 𑀓𑀼𑀧𑀺𑀢𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀆𑀲𑀸𑀤𑀦𑀸𑀧𑁂𑀓𑁆𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀢𑁆𑀢𑁂 𑀧𑀽𑀭𑁂𑀦𑁆𑀢𑁄 𑀅𑀕𑀫𑀸𑀲𑀺𑀁—  𑀪𑀼𑀜𑁆𑀚𑀣 𑀯𑀸 𑀳𑀭𑀣 𑀯𑀸𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀩𑀳𑀼𑀁 𑀧𑀲𑀼𑀢𑀁, 𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑀸 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑀸’𑀢𑀺𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑀳𑀼𑀁 𑀧𑀲𑀼𑀢𑀁, 𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑀸 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀯𑀸”𑀢𑀺? “𑀬𑀤𑀕𑁆𑀕𑁂𑀦 𑀢𑀬𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁄 𑀢𑀤𑀕𑁆𑀕𑁂𑀦 𑀢𑁂 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀼𑀢𑀁𑁇 𑀬𑀤𑀕𑁆𑀕𑁂𑀦 𑀢𑁂 𑀏𑀓𑀫𑁂𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀏𑀓𑀫𑁂𑀓𑀁 𑀲𑀺𑀢𑁆𑀣𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀢𑀤𑀕𑁆𑀕𑁂𑀦 𑀢𑁂 𑀩𑀳𑀼𑀁 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀼𑀢𑀁, 𑀲𑀕𑁆𑀕𑀸 𑀢𑁂 𑀆𑀭𑀤𑁆𑀥𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀢𑀭𑀼𑀡𑀧𑀲𑀦𑁆𑀦𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄—  “𑀮𑀸𑀪𑀸 𑀓𑀺𑀭 𑀫𑁂, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁 𑀓𑀺𑀭 𑀫𑁂, 𑀩𑀳𑀼𑀁 𑀓𑀺𑀭 𑀫𑀬𑀸 𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀼𑀢𑀁, 𑀲𑀕𑁆𑀕𑀸 𑀓𑀺𑀭 𑀫𑁂 𑀆𑀭𑀤𑁆𑀥𑀸”𑀢𑀺—  𑀳𑀝𑁆𑀞𑁄 𑀉𑀤𑀕𑁆𑀕𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑁄𑀚𑁆𑀚𑀬𑀸𑀕𑀼𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑀸 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑁄𑀚𑁆𑀚𑀬𑀸𑀕𑀼𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀢𑁆𑀭 𑀦𑀺𑀫𑀦𑁆𑀢𑀺𑀢𑁂𑀦 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀪𑁄𑀚𑁆𑀚𑀬𑀸𑀕𑀼 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
