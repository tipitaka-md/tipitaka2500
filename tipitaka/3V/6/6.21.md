# 𑁬𑁇𑁨𑁧 𑀫𑁂𑀡𑁆𑀟𑀓𑀕𑀳𑀧𑀢𑀺𑀯𑀢𑁆𑀣𑀼

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀫𑁂𑀡𑁆𑀟𑀓𑀕𑀳𑀧𑀢𑀺𑀯𑀢𑁆𑀣𑀼

(296.)

1343\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀺𑀬𑀦𑀕𑀭𑁂 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀧𑀝𑀺𑀯𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺—  𑀲𑀻𑀲𑀁 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀥𑀜𑁆𑀜𑀲𑁆𑀲 𑀥𑀸𑀭𑀸 𑀑𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀽𑀭𑁂𑀢𑀺𑁇 𑀪𑀭𑀺𑀬𑀸𑀬 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺—  𑀏𑀓𑀁𑀬𑁂𑀯 𑀆𑀍𑀳𑀓𑀣𑀸𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀜𑁆𑀘 𑀲𑀽𑀧𑀪𑀺𑀜𑁆𑀚𑀦𑀓𑀁 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀢𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀬𑀸𑀯 𑀲𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺—  𑀏𑀓𑀁𑀬𑁂𑀯 𑀲𑀳𑀲𑁆𑀲𑀣𑀯𑀺𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀲𑁆𑀲 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀯𑁂𑀢𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀬𑀸𑀯𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑀢𑀸𑁇 𑀲𑀼𑀡𑀺𑀲𑀸𑀬 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺—  𑀏𑀓𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀤𑁄𑀡𑀺𑀓𑀁 𑀧𑀺𑀝𑀓𑀁 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀲𑁆𑀲 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀬𑀸𑀯 𑀲𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀤𑀸𑀲𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀳𑁄𑀢𑀺—  𑀏𑀓𑁂𑀦 𑀦𑀗𑁆𑀕𑀮𑁂𑀦 𑀓𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀲𑀻𑀢𑀸𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇

1344\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀓𑀺𑀭 𑀯𑀺𑀚𑀺𑀢𑁂 𑀪𑀤𑁆𑀤𑀺𑀬𑀦𑀕𑀭𑁂 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀧𑀝𑀺𑀯𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄—  𑀲𑀻𑀲𑀁 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀥𑀜𑁆𑀜𑀲𑁆𑀲 𑀥𑀸𑀭𑀸 𑀑𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀽𑀭𑁂𑀢𑀺𑁇 𑀪𑀭𑀺𑀬𑀸𑀬 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄—  𑀏𑀓𑀁𑀬𑁂𑀯 𑀆𑀍𑀳𑀓𑀣𑀸𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀜𑁆𑀘 𑀲𑀽𑀧𑀪𑀺𑀜𑁆𑀚𑀦𑀓𑀁 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀢𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀬𑀸𑀯 𑀲𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄—  𑀏𑀓𑀁𑀬𑁂𑀯 𑀲𑀳𑀲𑁆𑀲𑀣𑀯𑀺𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀲𑁆𑀲 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀯𑁂𑀢𑀦𑀁 𑀤𑁂𑀢𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀬𑀸𑀯𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑀢𑀸𑁇 𑀲𑀼𑀡𑀺𑀲𑀸𑀬 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄—  𑀏𑀓𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀤𑁄𑀡𑀺𑀓𑀁 𑀧𑀺𑀝𑀓𑀁 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀓𑀫𑁆𑀫𑀓𑀭𑀧𑁄𑀭𑀺𑀲𑀲𑁆𑀲 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀢𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀬𑀸𑀯 𑀲𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 𑀤𑀸𑀲𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄—  𑀏𑀓𑁂𑀦 𑀦𑀗𑁆𑀕𑀮𑁂𑀦 𑀓𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀲𑀻𑀢𑀸𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀲𑀩𑁆𑀩𑀢𑁆𑀣𑀓𑀁 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀓𑀺𑀭, 𑀪𑀡𑁂, 𑀯𑀺𑀚𑀺𑀢𑁂 𑀪𑀤𑁆𑀤𑀺𑀬𑀦𑀕𑀭𑁂 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀧𑀝𑀺𑀯𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄—  𑀲𑀻𑀲𑀁 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀢𑀺, 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀥𑀜𑁆𑀜𑀲𑁆𑀲 𑀥𑀸𑀭𑀸 𑀑𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀽𑀭𑁂𑀢𑀺𑁇 𑀪𑀭𑀺𑀬𑀸𑀬…𑀧𑁂…  𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲…  𑀲𑀼𑀡𑀺𑀲𑀸𑀬…  𑀤𑀸𑀲𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀏𑀓𑁂𑀦 𑀦𑀗𑁆𑀕𑀮𑁂𑀦 𑀓𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀲𑀻𑀢𑀸𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻𑀢𑀺𑁇 𑀕𑀘𑁆𑀙, 𑀪𑀡𑁂, 𑀚𑀸𑀦𑀸𑀳𑀺𑁇 𑀬𑀣𑀸 𑀫𑀬𑀸 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀏𑀯𑀁 𑀢𑀯 𑀤𑀺𑀝𑁆𑀞𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

1345\. “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬 𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀢𑁂𑀦 𑀧𑀸𑀬𑀸𑀲𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀬𑁂𑀦 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑁂𑀡𑁆𑀟𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀜𑁆𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀭𑀜𑁆𑀜𑀸 𑀆𑀡𑀢𑁆𑀢𑁄—  ‘𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀓𑀺𑀭, 𑀪𑀡𑁂, 𑀯𑀺𑀚𑀺𑀢𑁂 𑀪𑀤𑁆𑀤𑀺𑀬𑀦𑀕𑀭𑁂 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀧𑀝𑀺𑀯𑀲𑀢𑀺, 𑀢𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀲𑀻𑀲𑀁 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀪𑀭𑀺𑀬𑀸𑀬…  𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲…  𑀲𑀼𑀡𑀺𑀲𑀸𑀬…  𑀤𑀸𑀲𑀲𑁆𑀲 𑀏𑀯𑀭𑀽𑀧𑁄 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀏𑀓𑁂𑀦 𑀦𑀗𑁆𑀕𑀮𑁂𑀦 𑀓𑀲𑀦𑁆𑀢𑀲𑁆𑀲 𑀲𑀢𑁆𑀢 𑀲𑀻𑀢𑀸𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀻𑀢𑀺, 𑀕𑀘𑁆𑀙, 𑀪𑀡𑁂, 𑀚𑀸𑀦𑀸𑀳𑀺𑁇 𑀬𑀣𑀸 𑀫𑀬𑀸 𑀲𑀸𑀫𑀁 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀏𑀯𑀁 𑀢𑀯 𑀤𑀺𑀝𑁆𑀞𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀧𑀲𑁆𑀲𑀸𑀫 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀲𑀻𑀲𑀁 𑀦𑀳𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀲𑀫𑁆𑀫𑀚𑁆𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀩𑀳𑀺𑀤𑁆𑀯𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀺, 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑀸 𑀥𑀜𑁆𑀜𑀲𑁆𑀲 𑀥𑀸𑀭𑀸 𑀑𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀥𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑀁 𑀧𑀽𑀭𑁂𑀲𑀺𑁇 “𑀤𑀺𑀝𑁆𑀞𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄𑁇 𑀪𑀭𑀺𑀬𑀸𑀬 𑀢𑁂 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

1346\. 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀭𑀺𑀬𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀁 𑀲𑁂𑀦𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸 𑀏𑀓𑀁𑀬𑁂𑀯 𑀆𑀍𑀳𑀓𑀣𑀸𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀜𑁆𑀘 𑀲𑀽𑀧𑀪𑀺𑀜𑁆𑀚𑀦𑀓𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀁 𑀲𑁂𑀦𑀁 𑀪𑀢𑁆𑀢𑁂𑀦 𑀧𑀭𑀺𑀯𑀺𑀲𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀬𑀸𑀯 𑀲𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 “𑀤𑀺𑀝𑁆𑀞𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀪𑀭𑀺𑀬𑀸𑀬𑀧𑀺 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄𑁇 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑁂 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀯𑁂𑀢𑀦𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀧𑀼𑀢𑁆𑀢𑁄 𑀏𑀓𑀁𑀬𑁂𑀯 𑀲𑀳𑀲𑁆𑀲𑀣𑀯𑀺𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀯𑁂𑀢𑀦𑀁 𑀅𑀤𑀸𑀲𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺, 𑀬𑀸𑀯𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀕𑀢𑀸𑁇 “𑀤𑀺𑀝𑁆𑀞𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲𑀧𑀺 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄𑁇 𑀲𑀼𑀡𑀺𑀲𑀸𑀬 𑀢𑁂 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀲𑀼𑀡𑀺𑀲𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀲𑀼𑀡𑀺𑀲𑀸 𑀏𑀓𑀁𑀬𑁂𑀯 𑀘𑀢𑀼𑀤𑁄𑀡𑀺𑀓𑀁 𑀧𑀺𑀝𑀓𑀁 𑀉𑀧𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬 𑀙𑀫𑀸𑀲𑀺𑀓𑀁 𑀪𑀢𑁆𑀢𑀁 𑀅𑀤𑀸𑀲𑀺, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀬𑀸𑀯 𑀲𑀸 𑀦 𑀯𑀼𑀝𑁆𑀞𑀸𑀢𑀺𑁇 “𑀤𑀺𑀝𑁆𑀞𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀲𑀼𑀡𑀺𑀲𑀸𑀬𑀧𑀺 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄𑁇 𑀤𑀸𑀲𑀲𑁆𑀲 𑀢𑁂 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀫𑀬𑁆𑀳𑀁 𑀔𑁄, 𑀲𑀸𑀫𑀺, 𑀤𑀸𑀲𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄 𑀔𑁂𑀢𑁆𑀢𑁂 𑀧𑀲𑁆𑀲𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀤𑀺𑀝𑁆𑀞𑁄 𑀢𑁂 𑀤𑀸𑀲𑀲𑁆𑀲𑀧𑀺 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁄 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀬𑀸 𑀲𑁂𑀦𑀸𑀬 𑀧𑀼𑀦𑀤𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀘𑁆𑀘𑀸𑀕𑀜𑁆𑀙𑀺𑁇 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇

1347\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑁂𑀲𑀸𑀮𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀦𑁂𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺—  “𑀲𑀫𑀡𑁄 𑀔𑀮𑀼 𑀪𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀪𑀤𑁆𑀤𑀺𑀬𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀪𑀤𑁆𑀤𑀺𑀬𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑀸𑀢𑀺𑀬𑀸 𑀯𑀦𑁂𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  ‘𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸’𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀲𑀤𑁂𑀯𑀓𑀁 𑀲𑀫𑀸𑀭𑀓𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀁 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀧𑀚𑀁 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

1348\. 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀤𑁆𑀭𑀸𑀦𑀺 𑀪𑀤𑁆𑀭𑀸𑀦𑀺 𑀬𑀸𑀦𑀸𑀦𑀺 𑀬𑁄𑀚𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀭𑀁 𑀪𑀤𑁆𑀭𑀁 𑀬𑀸𑀦𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀤𑁆𑀭𑁂𑀳𑀺 𑀪𑀤𑁆𑀭𑁂𑀳𑀺 𑀬𑀸𑀦𑁂𑀳𑀺 𑀪𑀤𑁆𑀤𑀺𑀬𑀸 𑀦𑀺𑀬𑁆𑀬𑀸𑀲𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑁇 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀔𑁄 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀫𑁂𑀡𑁆𑀟𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑁂𑀡𑁆𑀟𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀳𑀁 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀕𑀘𑁆𑀙𑀲𑀻”𑀢𑀺? “𑀕𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀕𑀳𑀧𑀢𑀺, 𑀓𑀺𑀭𑀺𑀬𑀯𑀸𑀤𑁄 𑀲𑀫𑀸𑀦𑁄 𑀅𑀓𑀺𑀭𑀺𑀬𑀯𑀸𑀤𑀁 𑀲𑀫𑀡𑀁 𑀕𑁄𑀢𑀫𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀲𑀺? 𑀲𑀫𑀡𑁄 𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀕𑁄𑀢𑀫𑁄 𑀅𑀓𑀺𑀭𑀺𑀬𑀯𑀸𑀤𑁄 𑀅𑀓𑀺𑀭𑀺𑀬𑀸𑀬 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺, 𑀢𑁂𑀦 𑀘 𑀲𑀸𑀯𑀓𑁂 𑀯𑀺𑀦𑁂𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀬𑀣𑀬𑀺𑀫𑁂 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀉𑀲𑀽𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀬𑀸𑀯𑀢𑀺𑀓𑀸 𑀬𑀸𑀦𑀲𑁆𑀲 𑀪𑀽𑀫𑀺, 𑀬𑀸𑀦𑁂𑀦 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀬𑀸𑀦𑀸 𑀧𑀘𑁆𑀘𑁄𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀺𑀓𑁄𑀯 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁…𑀧𑁂…  𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀁𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1349\. 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀪𑀭𑀺𑀬𑀸 𑀘 𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀲𑀼𑀡𑀺𑀲𑀸 𑀘 𑀤𑀸𑀲𑁄 𑀘 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁…𑀧𑁂…  𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀏𑀢𑁂 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑁂 𑀲𑀭𑀡𑀁 𑀕𑀢𑁂”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀫𑁂𑀡𑁆𑀟𑀓𑁄 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀸𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑀸𑀯 𑀅𑀳𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀥𑀼𑀯𑀪𑀢𑁆𑀢𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑁂𑀡𑁆𑀟𑀓𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇
