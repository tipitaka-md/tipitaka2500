# 𑁬𑁇𑁯 𑀫𑀦𑀼𑀲𑁆𑀲𑀫𑀁𑀲𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

* 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀫𑀦𑀼𑀲𑁆𑀲𑀫𑀁𑀲𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀓𑀣𑀸

(280.)

1266\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀘 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀉𑀪𑀢𑁄𑀧𑀲𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀤𑀸𑀬𑀓𑀸, 𑀓𑀸𑀭𑀓𑀸, 𑀲𑀁𑀖𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀭𑀺𑀯𑁂𑀡𑁂𑀦 𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀢𑀺—  “𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀓𑀲𑁆𑀲 𑀓𑀺𑀁 𑀆𑀳𑀭𑀺𑀬𑀢𑀽”𑀢𑀺? 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀧𑀻𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑀸 𑀔𑁄, 𑀪𑀕𑀺𑀦𑀺, 𑀯𑀺𑀭𑁂𑀘𑀦𑀁 𑀧𑀻𑀢𑀁𑁇 𑀅𑀢𑁆𑀣𑁄 𑀫𑁂 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀻𑀬𑁂𑀦𑀸”𑀢𑀺𑁇 “𑀲𑀼𑀝𑁆𑀞𑀼, 𑀅𑀬𑁆𑀬, 𑀆𑀳𑀭𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺 𑀖𑀭𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙, 𑀪𑀡𑁂, 𑀧𑀯𑀢𑁆𑀢𑀫𑀁𑀲𑀁 𑀚𑀸𑀦𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀅𑀬𑁆𑀬𑁂”𑀢𑀺 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸𑀬 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄 𑀦 𑀅𑀤𑁆𑀤𑀲 𑀧𑀯𑀢𑁆𑀢𑀫𑀁𑀲𑀁𑁇

1267\. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀬𑁂𑀦 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀢𑁆𑀣𑀬𑁆𑀬𑁂 𑀧𑀯𑀢𑁆𑀢𑀫𑀁𑀲𑀁𑁇 𑀫𑀸𑀖𑀸𑀢𑁄 𑀅𑀚𑁆𑀚𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸𑀬 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀢𑀲𑁆𑀲 𑀔𑁄 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀝𑀺𑀘𑁆𑀙𑀸𑀤𑀦𑀻𑀬𑀁 𑀅𑀮𑀪𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀩𑀸𑀥𑁄 𑀯𑀸 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺, 𑀓𑀸𑀮𑀁𑀓𑀺𑀭𑀺𑀬𑀸 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀫𑁂𑀢𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀬𑀸𑀳𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀦 𑀳𑀭𑀸𑀧𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀧𑁄𑀢𑁆𑀣𑀦𑀺𑀓𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀊𑀭𑀼𑀫𑀁𑀲𑀁 𑀉𑀓𑁆𑀓𑀦𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀺𑀬𑀸 𑀅𑀤𑀸𑀲𑀺—  “𑀳𑀦𑁆𑀤, 𑀚𑁂, 𑀇𑀫𑀁 𑀫𑀁𑀲𑀁 𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀫𑀼𑀓𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀢𑀲𑁆𑀲 𑀤𑀚𑁆𑀚𑀸𑀳𑀺𑁇 𑀬𑁄 𑀘 𑀫𑀁 𑀧𑀼𑀘𑁆𑀙𑀢𑀺, ‘𑀕𑀺𑀮𑀸𑀦𑀸’𑀢𑀺 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀳𑀻”𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑁂𑀦 𑀊𑀭𑀼𑀁 𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀑𑀯𑀭𑀓𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀫𑀜𑁆𑀘𑀓𑁂 𑀦𑀺𑀧𑀚𑁆𑀚𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀖𑀭𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀤𑀸𑀲𑀺𑀁 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀓𑀳𑀁 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸”𑀢𑀺? “𑀏𑀲𑀸𑀬𑁆𑀬, 𑀑𑀯𑀭𑀓𑁂 𑀦𑀺𑀧𑀦𑁆𑀦𑀸”𑀢𑀺𑁇

1268\. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑁂𑀦 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲 𑀦𑀺𑀧𑀦𑁆𑀦𑀸𑀲𑀻”𑀢𑀺? “𑀕𑀺𑀮𑀸𑀦𑀸𑀫𑁆𑀳𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀢𑁂 𑀆𑀩𑀸𑀥𑁄”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀬𑀸𑀯 𑀲𑀤𑁆𑀥𑀸𑀬𑀁 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀧𑀲𑀦𑁆𑀦𑀸, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀅𑀢𑁆𑀢𑀦𑁄𑀧𑀺 𑀫𑀁𑀲𑀸𑀦𑀺 𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀸𑀦𑀺𑁇 𑀓𑀺𑀫𑁆𑀧𑀺𑀫𑀸𑀬 𑀅𑀜𑁆𑀜𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀅𑀤𑁂𑀬𑁆𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺—  𑀳𑀝𑁆𑀞𑁄 𑀉𑀤𑀕𑁆𑀕𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁, 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

1269\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺, 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀳𑀁, 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸”𑀢𑀺? “𑀕𑀺𑀮𑀸𑀦𑀸, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀆𑀕𑀘𑁆𑀙𑀢𑀽”𑀢𑀺𑁇 “𑀦, 𑀪𑀕𑀯𑀸, 𑀉𑀲𑁆𑀲𑀳𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸𑀧𑀺 𑀆𑀦𑁂𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀁 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀆𑀦𑁂𑀲𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀲𑀳 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀢𑀸𑀯 𑀫𑀳𑀸𑀯𑀡𑁄 𑀭𑀼𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀺 𑀲𑀼𑀘𑁆𑀙𑀯𑀺𑀮𑁄𑀫𑀚𑀸𑀢𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑁄 𑀘 𑀉𑀧𑀸𑀲𑀓𑁄 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀸 𑀘 𑀉𑀧𑀸𑀲𑀺𑀓𑀸—  “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀢𑀣𑀸𑀕𑀢𑀲𑁆𑀲 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑀢𑀸 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑀢𑀸, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀳 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀢𑀸𑀯 𑀫𑀳𑀸𑀯𑀡𑁄 𑀭𑀼𑀍𑀳𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀲𑀼𑀘𑁆𑀙𑀯𑀺𑀮𑁄𑀫𑀚𑀸𑀢𑁄”𑀢𑀺—  𑀳𑀝𑁆𑀞𑀸 𑀉𑀤𑀕𑁆𑀕𑀸 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀓𑀁 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀜𑁆𑀘 𑀉𑀧𑀸𑀲𑀺𑀓𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1270\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀁 𑀫𑀁𑀲𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀻”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀼𑀧𑁆𑀧𑀺𑀬𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀁 𑀫𑀁𑀲𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺𑀦𑁆”𑀢𑀺𑁇 “𑀆𑀳𑀭𑀺𑀬𑀺𑀢𑁆𑀣, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀆𑀳𑀭𑀺𑀬𑀺𑀢𑁆𑀣, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀸𑀫𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀧𑀝𑀺𑀯𑁂𑀓𑁆𑀔𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀦𑀸𑀳𑀁, 𑀪𑀕𑀯𑀸, 𑀧𑀝𑀺𑀯𑁂𑀓𑁆𑀔𑀺𑀦𑁆”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀫𑀁𑀲𑀁 𑀔𑁄 𑀢𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀧𑀭𑀺𑀪𑀼𑀢𑁆𑀢𑀁𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀤𑁆𑀥𑀸 𑀧𑀲𑀦𑁆𑀦𑀸, 𑀢𑁂𑀳𑀺 𑀅𑀢𑁆𑀢𑀦𑁄𑀧𑀺 𑀫𑀁𑀲𑀸𑀦𑀺 𑀧𑀭𑀺𑀘𑁆𑀘𑀢𑁆𑀢𑀸𑀦𑀺𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀝𑀺𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
