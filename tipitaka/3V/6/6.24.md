# 𑁬𑁇𑁨𑁪 𑀭𑁄𑀚𑀫𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀼

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀭𑁄𑀚𑀫𑀮𑁆𑀮𑀯𑀢𑁆𑀣𑀼

1360\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀧𑀡𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸—  “𑀪𑀕𑀯𑀸 𑀓𑀺𑀭 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀁 𑀆𑀕𑀘𑁆𑀙𑀢𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀮𑀲𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀢𑁂𑀳𑀻”𑀢𑀺𑁇 𑀢𑁂 𑀲𑀗𑁆𑀕𑀭𑀁 𑀅𑀓𑀁𑀲𑀼—  “𑀬𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀦 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺𑀲𑁆𑀲 𑀤𑀡𑁆𑀟𑁄”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀳𑀸𑀬𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀺𑀦𑀸𑀭𑀓𑀸 𑀫𑀮𑁆𑀮𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀅𑀓𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀁 𑀔𑁄 𑀭𑁄𑀚𑀁 𑀫𑀮𑁆𑀮𑀁 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀉𑀍𑀆𑀭𑀁 𑀔𑁄 𑀢𑁂 𑀇𑀤𑀁, 𑀆𑀯𑀼𑀲𑁄 𑀭𑁄𑀚, 𑀬𑀁 𑀢𑁆𑀯𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀩𑀳𑀼𑀓𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑁂 𑀯𑀸 𑀥𑀫𑁆𑀫𑁂 𑀯𑀸 𑀲𑀁𑀖𑁂 𑀯𑀸; 𑀅𑀧𑀺 𑀘 𑀜𑀸𑀢𑀻𑀳𑀺 𑀲𑀗𑁆𑀕𑀭𑁄 𑀓𑀢𑁄—  ‘𑀬𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀦 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀜𑁆𑀘𑀲𑀢𑀸𑀦𑀺𑀲𑁆𑀲 𑀤𑀡𑁆𑀟𑁄’𑀢𑀺; 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀜𑀸𑀢𑀻𑀦𑀁 𑀤𑀡𑁆𑀟𑀪𑀬𑀸 𑀏𑀯𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀫𑀦𑀁 𑀅𑀓𑀸𑀲𑀺𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀅𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀅𑀳𑁄𑀲𑀺 “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀏𑀯𑀁 𑀯𑀓𑁆𑀔𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑁄 𑀜𑀸𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑁇 𑀫𑀳𑀢𑁆𑀣𑀺𑀓𑁄 𑀔𑁄 𑀧𑀦 𑀏𑀯𑀭𑀽𑀧𑀸𑀦𑀁 𑀜𑀸𑀢𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀲𑀸𑀤𑁄𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀢𑀣𑀸 𑀓𑀭𑁄𑀢𑀼, 𑀬𑀣𑀸 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀲𑀻𑀤𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 “𑀦 𑀔𑁄 𑀢𑀁, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀢𑀣𑀸𑀕𑀢𑁂𑀦, 𑀬𑀣𑀸 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀲𑀻𑀤𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

1361\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑁄𑀚𑀁 𑀫𑀮𑁆𑀮𑀁 𑀫𑁂𑀢𑁆𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀨𑀭𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀪𑀕𑀯𑀢𑁄 𑀫𑁂𑀢𑁆𑀢𑁂𑀦 𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀨𑀼𑀝𑁆𑀞𑁄, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀕𑀸𑀯𑀺𑀁 𑀢𑀭𑀼𑀡𑀯𑀘𑁆𑀙𑁄; 𑀏𑀯𑀫𑁂𑀯 𑀯𑀺𑀳𑀸𑀭𑁂𑀦 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀭𑀺𑀯𑁂𑀡𑁂𑀦 𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀢𑀺—  “𑀓𑀳𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀢𑀭𑀳𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄, 𑀤𑀲𑁆𑀲𑀦𑀓𑀸𑀫𑀸 𑀳𑀺 𑀫𑀬𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀲𑀸𑀯𑀼𑀲𑁄 𑀭𑁄𑀚, 𑀯𑀺𑀳𑀸𑀭𑁄 𑀲𑀁𑀯𑀼𑀢𑀤𑁆𑀯𑀸𑀭𑁄, 𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀭𑀫𑀸𑀦𑁄 𑀆𑀍𑀇𑀦𑁆𑀤𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀆𑀓𑁄𑀝𑁂𑀳𑀺, 𑀯𑀺𑀯𑀭𑀺𑀲𑁆𑀲𑀢𑀺 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀬𑁂𑀦 𑀲𑁄 𑀯𑀺𑀳𑀸𑀭𑁄 𑀲𑀁𑀯𑀼𑀢𑀤𑁆𑀯𑀸𑀭𑁄, 𑀢𑁂𑀦 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀢𑀭𑀫𑀸𑀦𑁄 𑀆𑀍𑀇𑀦𑁆𑀤𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀕𑀍𑀅𑀁 𑀆𑀓𑁄𑀝𑁂𑀲𑀺𑁇 𑀯𑀺𑀯𑀭𑀺 𑀪𑀕𑀯𑀸 𑀤𑁆𑀯𑀸𑀭𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀔𑁄 𑀭𑁄𑀚𑀲𑁆𑀲 𑀫𑀮𑁆𑀮𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁…𑀧𑁂…  𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀫𑀫𑀜𑁆𑀜𑁂𑀯 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀼𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁, 𑀦𑁄 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆”𑀢𑀺𑁇 “𑀬𑁂𑀲𑀁 𑀔𑁄, 𑀭𑁄𑀚, 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑀬𑀸, 𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀅𑀳𑁄 𑀦𑀽𑀦 𑀅𑀬𑁆𑀬𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀜𑁆𑀜𑁂𑀯 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀼𑀁 𑀘𑀻𑀯𑀭𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀲𑁂𑀦𑀸𑀲𑀦𑀕𑀺𑀮𑀸𑀦𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀪𑁂𑀲𑀚𑁆𑀚𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑀁, 𑀦𑁄 𑀅𑀜𑁆𑀜𑁂𑀲𑀦𑁆’𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀭𑁄𑀚, 𑀢𑀯 𑀘𑁂𑀯 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀅𑀜𑁆𑀜𑁂𑀲𑀜𑁆𑀘𑀸”𑀢𑀺𑁇

1362\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑀼𑀲𑀺𑀦𑀸𑀭𑀸𑀬𑀁 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀪𑀢𑁆𑀢𑀸𑀦𑀁 𑀪𑀢𑁆𑀢𑀧𑀝𑀺𑀧𑀸𑀝𑀺 𑀅𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑀲𑁆𑀲 𑀫𑀮𑁆𑀮𑀲𑁆𑀲 𑀧𑀝𑀺𑀧𑀸𑀝𑀺𑀁 𑀅𑀮𑀪𑀦𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑀁 𑀑𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀁, 𑀬𑀁 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀦𑀸𑀲𑁆𑀲, 𑀢𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀦𑀸𑀤𑁆𑀤𑀲—  𑀟𑀸𑀓𑀜𑁆𑀘 𑀧𑀺𑀝𑁆𑀞𑀔𑀸𑀤𑀦𑀻𑀬𑀜𑁆𑀘𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀝𑀺𑀧𑀸𑀝𑀺𑀁 𑀅𑀮𑀪𑀦𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑀁 𑀑𑀮𑁄𑀓𑁂𑀬𑁆𑀬𑀁, 𑀬𑀁 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑁂 𑀦𑀸𑀲𑁆𑀲, 𑀢𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀪𑀢𑁆𑀢𑀕𑁆𑀕𑀁 𑀑𑀮𑁄𑀓𑁂𑀦𑁆𑀢𑁄 𑀤𑁆𑀯𑁂 𑀦𑀸𑀤𑁆𑀤𑀲𑀁—  𑀟𑀸𑀓𑀜𑁆𑀘 𑀧𑀺𑀝𑁆𑀞𑀔𑀸𑀤𑀦𑀻𑀬𑀜𑁆𑀘𑁇 𑀲𑀘𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀬𑁆𑀬𑀁 𑀟𑀸𑀓𑀜𑁆𑀘 𑀧𑀺𑀝𑁆𑀞𑀔𑀸𑀤𑀦𑀻𑀬𑀜𑁆𑀘, 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬 𑀫𑁂 𑀪𑀕𑀯𑀸”𑀢𑀺? “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑁄𑀚, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀸𑀦𑀦𑁆𑀤, 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀢𑀽”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑁄𑀚, 𑀧𑀝𑀺𑀬𑀸𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀳𑀽𑀢𑀁 𑀟𑀸𑀓𑀜𑁆𑀘 𑀧𑀺𑀝𑁆𑀞𑀔𑀸𑀤𑀦𑀻𑀬𑀜𑁆𑀘 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀟𑀸𑀓𑀜𑁆𑀘 𑀧𑀺𑀝𑁆𑀞𑀔𑀸𑀤𑀦𑀻𑀬𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀭𑁄𑀚, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀤𑁂𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑁄𑀚𑁄 𑀫𑀮𑁆𑀮𑁄 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀳𑀽𑀢𑁂𑀳𑀺 𑀟𑀸𑀓𑁂𑀳𑀺 𑀘 𑀧𑀺𑀝𑁆𑀞𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀳𑀺 𑀘 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀥𑁄𑀢𑀳𑀢𑁆𑀣𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀭𑁄𑀚𑀁 𑀫𑀮𑁆𑀮𑀁 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀟𑀸𑀓𑀁 𑀲𑀩𑁆𑀩𑀜𑁆𑀘 𑀧𑀺𑀝𑁆𑀞𑀔𑀸𑀤𑀦𑀻𑀬𑀦𑁆”𑀢𑀺𑁇
