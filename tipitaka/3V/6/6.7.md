# 𑁬𑁇𑁭 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(277.)

1260. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀲𑀁𑀖𑀲𑁆𑀲𑀢𑁆𑀣𑀸𑀬 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼—  “𑀓𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑁄”𑀢𑀺? “𑀏𑀲𑀸𑀯𑀼𑀲𑁄, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀕𑀸𑀫𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀯𑀺𑀝𑁆𑀞𑁄”𑀢𑀺𑁇 “𑀇𑀤𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀣 𑀬𑀸𑀯 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀆𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀓𑀼𑀮𑀸𑀦𑀺 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀤𑀺𑀯𑀸 𑀆𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀧𑀺 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀧𑀺 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀘 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀣𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑁂𑀪𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀺𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀦𑀸 𑀧𑀯𑀸𑀭𑀺𑀢𑁂𑀦 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(278.)

1261. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑀟𑀸𑀳𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀼𑀩𑁆𑀩𑁂 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀸𑀬𑀟𑀸𑀳𑀸𑀩𑀸𑀥𑁄 𑀓𑁂𑀦 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀻”𑀢𑀺? “𑀪𑀺𑀲𑁂𑀳𑀺 𑀘 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀸𑀳𑀺 𑀘𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀦𑁆𑀤𑀸𑀓𑀺𑀦𑀺𑀬𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀦𑀸𑀕𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀢𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄𑁇 𑀲𑁆𑀯𑀸𑀕𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲𑁇 𑀓𑁂𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑁄; 𑀓𑀺𑀁 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺? “𑀪𑀺𑀲𑁂𑀳𑀺 𑀘 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀢𑁆𑀣𑁄, 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀸𑀳𑀺 𑀘𑀸”𑀢𑀺𑁇

1262. 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀦𑀸𑀕𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀦𑀸𑀕𑀁 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀅𑀬𑁆𑀬𑀲𑁆𑀲 𑀪𑀺𑀲𑁂 𑀘 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀸𑀬𑁄 𑀘 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀦𑀸𑀕𑁄 𑀫𑀦𑁆𑀤𑀸𑀓𑀺𑀦𑀺𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸, 𑀲𑁄𑀡𑁆𑀟𑀸𑀬 𑀪𑀺𑀲𑀜𑁆𑀘 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀜𑁆𑀘 𑀅𑀩𑁆𑀩𑀸𑀳𑀺𑀢𑁆𑀯𑀸, 𑀲𑀼𑀯𑀺𑀓𑁆𑀔𑀸𑀮𑀺𑀢𑀁 𑀯𑀺𑀓𑁆𑀔𑀸𑀮𑁂𑀢𑁆𑀯𑀸, 𑀪𑀡𑁆𑀟𑀺𑀓𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄—  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀩𑀮𑀯𑀸 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬, 𑀧𑀲𑀸𑀭𑀺𑀢𑀁 𑀯𑀸 𑀩𑀸𑀳𑀁 𑀲𑀫𑀺𑀜𑁆𑀚𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯—  𑀫𑀦𑁆𑀤𑀸𑀓𑀺𑀦𑀺𑀬𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀚𑁂𑀢𑀯𑀦𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀲𑁄𑀧𑀺 𑀔𑁄 𑀦𑀸𑀕𑁄 𑀫𑀦𑁆𑀤𑀸𑀓𑀺𑀦𑀺𑀬𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀚𑁂𑀢𑀯𑀦𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀦𑀸𑀕𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀲𑁂 𑀘 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀸𑀬𑁄 𑀘 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑁆𑀢𑀭𑀳𑀺𑀢𑁄 𑀫𑀦𑁆𑀤𑀸𑀓𑀺𑀦𑀺𑀬𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀬𑀸 𑀢𑀻𑀭𑁂 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀲𑁂 𑀘 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀸𑀬𑁄 𑀘 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀺𑀲𑁂 𑀘 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀸𑀬𑁄 𑀘 𑀪𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀬𑀟𑀸𑀳𑀸𑀩𑀸𑀥𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇 𑀩𑀳𑀽 𑀪𑀺𑀲𑀸 𑀘 𑀫𑀼𑀍𑀆𑀮𑀺𑀓𑀸𑀬𑁄 𑀘 𑀅𑀯𑀲𑀺𑀝𑁆𑀞𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀼𑀩𑁆𑀪𑀺𑀓𑁆𑀔𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀧𑀺 𑀧𑀯𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀧𑀝𑀺𑀲𑀗𑁆𑀔𑀸𑀧𑀺 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁄 𑀘 𑀲𑀁𑀖𑁄 𑀧𑀯𑀸𑀭𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀣, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀣𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀦𑀝𑁆𑀞𑀁 𑀧𑁄𑀓𑁆𑀔𑀭𑀝𑁆𑀞𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀦𑀸 𑀧𑀯𑀸𑀭𑀺𑀢𑁂𑀦 𑀅𑀦𑀢𑀺𑀭𑀺𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1263. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀩𑀳𑀼𑀁 𑀨𑀮𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺, 𑀓𑀧𑁆𑀧𑀺𑀬𑀓𑀸𑀭𑀓𑁄 𑀘 𑀦 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀨𑀮𑀁 𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀩𑀻𑀚𑀁 𑀦𑀺𑀩𑁆𑀩𑀢𑁆𑀢𑀩𑀻𑀚𑀁 𑀅𑀓𑀢𑀓𑀧𑁆𑀧𑀁 𑀨𑀮𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
