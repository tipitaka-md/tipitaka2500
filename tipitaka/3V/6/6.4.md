# 𑁬𑁇𑁪 𑀕𑀼𑀍𑀆𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀪𑁂𑀲𑀚𑁆𑀚𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀕𑀼𑀍𑀆𑀤𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(272.)

1248\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀕𑀼𑀍𑀅𑀓𑀭𑀡𑀁, 𑀑𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀕𑀼𑀍𑀏 𑀧𑀺𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀙𑀸𑀭𑀺𑀓𑀫𑁆𑀧𑀺 𑀧𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦—  “𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑁄 𑀕𑀼𑀍𑀑 𑀲𑀸𑀫𑀺𑀲𑁄, 𑀦 𑀓𑀧𑁆𑀧𑀢𑀺 𑀕𑀼𑀍𑀑 𑀯𑀺𑀓𑀸𑀮𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑁄 𑀲𑀧𑀭𑀺𑀲𑁄 𑀕𑀼𑀍𑀅𑀁 𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇 𑀬𑁂𑀧𑀺𑀲𑁆𑀲 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀢𑁂𑀧𑀺 𑀕𑀼𑀍𑀅𑀁 𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀓𑀺𑀫𑀢𑁆𑀣𑀸𑀬, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀼𑀍𑀏 𑀧𑀺𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀙𑀸𑀭𑀺𑀓𑀫𑁆𑀧𑀺 𑀧𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀣𑀤𑁆𑀥𑀢𑁆𑀣𑀸𑀬, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑀤𑁆𑀥𑀢𑁆𑀣𑀸𑀬 𑀕𑀼𑀍𑀏 𑀧𑀺𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀙𑀸𑀭𑀺𑀓𑀫𑁆𑀧𑀺 𑀧𑀓𑁆𑀔𑀺𑀧𑀦𑁆𑀢𑀺, 𑀲𑁄 𑀘 𑀕𑀼𑀍𑀑𑀢𑁆𑀯𑁂𑀯 𑀲𑀗𑁆𑀔𑀁 𑀕𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀣𑀸𑀲𑀼𑀔𑀁 𑀕𑀼𑀍𑀅𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1249\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀗𑁆𑀔𑀸𑀭𑁂𑀯𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀯𑀘𑁆𑀘𑁂 𑀫𑀼𑀕𑁆𑀕𑀁 𑀚𑀸𑀢𑀁, 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸—  “𑀅𑀓𑀧𑁆𑀧𑀺𑀬𑀸 𑀫𑀼𑀕𑁆𑀕𑀸; 𑀧𑀓𑁆𑀓𑀸𑀧𑀺 𑀫𑀼𑀕𑁆𑀕𑀸 𑀚𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺—  𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑁄 𑀲𑀧𑀭𑀺𑀲𑁄 𑀫𑀼𑀕𑁆𑀕𑀁 𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀢𑀺𑁇 𑀬𑁂𑀧𑀺𑀲𑁆𑀲 𑀲𑁄𑀢𑀩𑁆𑀩𑀁 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺, 𑀢𑁂𑀧𑀺 𑀫𑀼𑀕𑁆𑀕𑀁 𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀲𑀘𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀓𑁆𑀓𑀸𑀧𑀺 𑀫𑀼𑀕𑁆𑀕𑀸 𑀚𑀸𑀬𑀦𑁆𑀢𑀺, 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀣𑀸𑀲𑀼𑀔𑀁 𑀫𑀼𑀕𑁆𑀕𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

(273.)

1250\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀉𑀤𑀭𑀯𑀸𑀢𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀮𑁄𑀡𑀲𑁄𑀯𑀻𑀭𑀓𑀁 𑀅𑀧𑀸𑀬𑀺𑁇 𑀢𑀲𑁆𑀲 𑀲𑁄 𑀉𑀤𑀭𑀯𑀸𑀢𑀸𑀩𑀸𑀥𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀮𑁄𑀡𑀲𑁄𑀯𑀻𑀭𑀓𑀁; 𑀅𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀉𑀤𑀓𑀲𑀫𑁆𑀪𑀺𑀦𑁆𑀦𑀁 𑀧𑀸𑀦𑀧𑀭𑀺𑀪𑁄𑀕𑁂𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
