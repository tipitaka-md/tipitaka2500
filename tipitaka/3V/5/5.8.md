# 𑁫𑁇𑁮 𑀉𑀘𑁆𑀘𑀸𑀲𑀬𑀦𑀫𑀳𑀸𑀲𑀬𑀦𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀉𑀘𑁆𑀘𑀸𑀲𑀬𑀦𑀫𑀳𑀸𑀲𑀬𑀦𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

(254.)

1168. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀘𑁆𑀘𑀸𑀲𑀬𑀦𑀫𑀳𑀸𑀲𑀬𑀦𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀆𑀲𑀦𑁆𑀤𑀺𑀁, 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁, 𑀕𑁄𑀦𑀓𑀁, 𑀘𑀺𑀢𑁆𑀢𑀓𑀁, 𑀧𑀝𑀺𑀓𑀁, 𑀧𑀝𑀮𑀺𑀓𑀁, 𑀢𑀽𑀮𑀺𑀓𑀁, 𑀯𑀺𑀓𑀢𑀺𑀓𑀁, 𑀉𑀤𑁆𑀥𑀮𑁄𑀫𑀺𑀁, 𑀏𑀓𑀦𑁆𑀢𑀮𑁄𑀫𑀺𑀁, 𑀓𑀝𑁆𑀝𑀺𑀲𑁆𑀲𑀁, 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀁, 𑀓𑀼𑀢𑁆𑀢𑀓𑀁, 𑀳𑀢𑁆𑀣𑀢𑁆𑀣𑀭𑀁, 𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀭𑀁, 𑀭𑀣𑀢𑁆𑀣𑀭𑀁, 𑀅𑀚𑀺𑀦𑀧𑀯𑁂𑀡𑀺𑀁, 𑀓𑀤𑀮𑀺𑀫𑀺𑀕𑀧𑀯𑀭𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀭𑀡𑀁, 𑀲𑁅𑀢𑁆𑀢𑀭𑀘𑁆𑀙𑀤𑀁, 𑀉𑀪𑀢𑁄𑀮𑁄𑀳𑀺𑀢𑀓𑀽𑀧𑀥𑀸𑀦𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀯𑀺𑀳𑀸𑀭𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀘𑁆𑀘𑀸𑀲𑀬𑀦𑀫𑀳𑀸𑀲𑀬𑀦𑀸𑀦𑀺 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑀦𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀆𑀲𑀦𑁆𑀤𑀺, 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁄, 𑀕𑁄𑀦𑀓𑁄, 𑀘𑀺𑀢𑁆𑀢𑀓𑁄, 𑀧𑀝𑀺𑀓𑀸, 𑀧𑀝𑀮𑀺𑀓𑀸, 𑀢𑀽𑀮𑀺𑀓𑀸, 𑀯𑀺𑀓𑀢𑀺𑀓𑀸, 𑀉𑀤𑁆𑀥𑀮𑁄𑀫𑀺, 𑀏𑀓𑀦𑁆𑀢𑀮𑁄𑀫𑀺, 𑀓𑀝𑁆𑀝𑀺𑀲𑁆𑀲𑀁, 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀁, 𑀓𑀼𑀢𑁆𑀢𑀓𑀁, 𑀳𑀢𑁆𑀣𑀢𑁆𑀣𑀭𑀁, 𑀅𑀲𑁆𑀲𑀢𑁆𑀣𑀭𑀁, 𑀭𑀣𑀢𑁆𑀣𑀭𑀁, 𑀅𑀚𑀺𑀦𑀧𑀯𑁂𑀡𑀺, 𑀓𑀤𑀮𑀺𑀫𑀺𑀕𑀧𑀯𑀭𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀭𑀡𑀁, 𑀲𑁅𑀢𑁆𑀢𑀭𑀘𑁆𑀙𑀤𑀁, 𑀉𑀪𑀢𑁄𑀮𑁄𑀳𑀺𑀢𑀓𑀽𑀧𑀥𑀸𑀦𑀁𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
