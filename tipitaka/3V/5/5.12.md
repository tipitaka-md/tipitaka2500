# 𑁫𑁇𑁧𑁨 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲𑀧𑀜𑁆𑀘𑀯𑀭𑀧𑀭𑀺𑀤𑀲𑁆𑀲𑀦

* 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲𑀧𑀜𑁆𑀘𑀯𑀭𑀧𑀭𑀺𑀤𑀲𑁆𑀲𑀦

(258.)

1179\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀲𑁆𑀲 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀲𑀼𑀢𑁄𑀬𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀏𑀤𑀺𑀲𑁄 𑀘 𑀏𑀤𑀺𑀲𑁄 𑀘𑀸𑀢𑀺, 𑀦 𑀘 𑀫𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀳𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁, 𑀲𑀘𑁂 𑀫𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  ‘𑀲𑀼𑀢𑁄𑀬𑁂𑀯 𑀔𑁄 𑀫𑁂 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀏𑀤𑀺𑀲𑁄 𑀘 𑀏𑀤𑀺𑀲𑁄 𑀘𑀸𑀢𑀺, 𑀦 𑀘 𑀫𑀬𑀸 𑀲𑀫𑁆𑀫𑀼𑀔𑀸 𑀤𑀺𑀝𑁆𑀞𑁄, 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀳𑀁 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁, 𑀲𑀘𑁂 𑀫𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸’𑀢𑀺; 𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁, 𑀲𑀘𑁂 𑀫𑀁 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑁄𑀡𑁇 𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀭𑀳𑀦𑁆𑀢𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑀁𑁇 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑀁 𑀧𑀲𑀸𑀤𑀦𑀻𑀬𑀁 𑀲𑀦𑁆𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀦𑁆𑀢𑀫𑀸𑀦𑀲𑀁 𑀉𑀢𑁆𑀢𑀫𑀤𑀫𑀣𑀲𑀫𑀣𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑀁 𑀤𑀦𑁆𑀢𑀁 𑀕𑀼𑀢𑁆𑀢𑀁 𑀬𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀦𑀸𑀕𑀁𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀫𑀫 𑀯𑀘𑀦𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤—  ‘𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀻’𑀢𑀺𑁇 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀳𑀺—  ‘𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄, 𑀢𑀺𑀡𑁆𑀡𑀁 𑀫𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀓𑀲𑀺𑀭𑁂𑀦 𑀢𑀢𑁄 𑀢𑀢𑁄 𑀤𑀲𑀯𑀕𑁆𑀕𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀮𑀢𑁆𑀣𑀁; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀅𑀧𑁆𑀧𑀢𑀭𑁂𑀦 𑀕𑀡𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀡𑁆𑀳𑀼𑀢𑁆𑀢𑀭𑀸 𑀪𑀽𑀫𑀺 𑀔𑀭𑀸 𑀕𑁄𑀓𑀡𑁆𑀝𑀓𑀳𑀢𑀸; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀳𑀸𑀦𑀕𑀭𑀼𑀓𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀸; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀥𑀼𑀯𑀦𑀳𑀸𑀦𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀏𑀭𑀕𑀽 𑀫𑁄𑀭𑀕𑀽 𑀫𑀚𑁆𑀚𑀸𑀭𑀽 𑀚𑀦𑁆𑀢𑀽; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑀺—  “𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀫𑀸”𑀢𑀺𑁇 𑀢𑁂 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀳𑀺 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀘𑀻𑀯𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑀦𑁆”𑀢𑀺 𑀢𑁂 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺—  “𑀫𑀸 𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀚𑁂𑀢𑀯𑀦𑀁 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

1180\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀫𑀲𑁆𑀲, 𑀆𑀦𑀦𑁆𑀤, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄—  “𑀬𑀲𑁆𑀲 𑀔𑁄 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀆𑀡𑀸𑀧𑁂𑀢𑀺, ‘𑀇𑀫𑀲𑁆𑀲, 𑀆𑀦𑀦𑁆𑀤, 𑀆𑀕𑀦𑁆𑀢𑀼𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀳𑀻’𑀢𑀺, 𑀇𑀘𑁆𑀙𑀢𑀺 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀼𑀁, 𑀇𑀘𑁆𑀙𑀢𑀺 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀲𑁄𑀡𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀏𑀓𑀯𑀺𑀳𑀸𑀭𑁂 𑀯𑀢𑁆𑀣𑀼𑀦𑁆”𑀢𑀺—  𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀳𑀸𑀭𑁂 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀸𑀧𑁂𑀲𑀺𑁇

1181\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀳𑀼𑀤𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀩𑀳𑀼𑀤𑁂𑀯 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀯𑀻𑀢𑀺𑀦𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁄𑀡𑀁 𑀅𑀚𑁆𑀛𑁂𑀲𑀺—  “𑀧𑀝𑀺𑀪𑀸𑀢𑀼 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀥𑀫𑁆𑀫𑁄 𑀪𑀸𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀲𑀩𑁆𑀩𑀸𑀦𑁂𑀯 𑀅𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕𑀺𑀓𑀸𑀦𑀺 𑀲𑀭𑁂𑀦 𑀅𑀪𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀲𑀭𑀪𑀜𑁆𑀜𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑁂 𑀅𑀩𑁆𑀪𑀸𑀦𑀼𑀫𑁄𑀤𑀺—  “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀪𑀺𑀓𑁆𑀔𑀼𑁇 𑀲𑀼𑀕𑁆𑀕𑀳𑀺𑀢𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀅𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕𑀺𑀓𑀸𑀦𑀺, 𑀲𑀼𑀫𑀦𑀲𑀺𑀓𑀢𑀸𑀦𑀺 𑀲𑀽𑀧𑀥𑀸𑀭𑀺𑀢𑀸𑀦𑀺𑁇 𑀓𑀮𑁆𑀬𑀸𑀡𑀺𑀬𑀸𑀧𑀺 𑀯𑀸𑀘𑀸𑀬 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄, 𑀯𑀺𑀲𑁆𑀲𑀝𑁆𑀞𑀸𑀬, 𑀅𑀦𑁂𑀮𑀕𑀮𑀸𑀬, 𑀅𑀢𑁆𑀣𑀲𑁆𑀲 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑀦𑀺𑀬𑀸𑁇 𑀓𑀢𑀺𑀯𑀲𑁆𑀲𑁄𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺? “𑀏𑀓𑀯𑀲𑁆𑀲𑁄𑀳𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀏𑀯𑀁 𑀘𑀺𑀭𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺? “𑀘𑀺𑀭𑀁 𑀤𑀺𑀝𑁆𑀞𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀸𑀫𑁂𑀲𑀼 𑀆𑀤𑀻𑀦𑀯𑁄, 𑀅𑀧𑀺 𑀘 𑀲𑀫𑁆𑀩𑀸𑀥𑀸 𑀖𑀭𑀸𑀯𑀸𑀲𑀸 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑀸 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

1182\. _“𑀤𑀺𑀲𑁆𑀯𑀸 𑀆𑀤𑀻𑀦𑀯𑀁 𑀮𑁄𑀓𑁂,_  
_𑀜𑀢𑁆𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀦𑀺𑀭𑀽𑀧𑀥𑀺𑀁;_  
_𑀅𑀭𑀺𑀬𑁄 𑀦 𑀭𑀫𑀢𑀻 𑀧𑀸𑀧𑁂,_  
_𑀧𑀸𑀧𑁂 𑀦 𑀭𑀫𑀢𑀻 𑀲𑀼𑀘𑀻”𑀢𑀺𑁇_  


1183\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑁄𑀡𑁄—  “𑀧𑀝𑀺𑀲𑀫𑁆𑀫𑁄𑀤𑀢𑀺 𑀔𑁄 𑀫𑀁 𑀪𑀕𑀯𑀸, 𑀅𑀬𑀁 𑀔𑁆𑀯𑀲𑁆𑀲 𑀓𑀸𑀮𑁄 𑀬𑀁 𑀫𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀧𑀭𑀺𑀤𑀲𑁆𑀲𑀻”𑀢𑀺—  𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂 𑀲𑀺𑀭𑀲𑀸 𑀯𑀦𑁆𑀤𑀢𑀺, 𑀏𑀯𑀜𑁆𑀘 𑀯𑀤𑁂𑀢𑀺 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀧𑁆𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄𑁇 𑀢𑀺𑀡𑁆𑀡𑀁 𑀫𑁂 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀓𑀲𑀺𑀭𑁂𑀦 𑀢𑀢𑁄 𑀢𑀢𑁄 𑀤𑀲𑀯𑀕𑁆𑀕𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀮𑀢𑁆𑀣𑀁, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀅𑀧𑁆𑀧𑀢𑀭𑁂𑀦 𑀕𑀡𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀡𑁆𑀳𑀼𑀢𑁆𑀢𑀭𑀸 𑀪𑀽𑀫𑀺 𑀔𑀭𑀸 𑀕𑁄𑀓𑀡𑁆𑀝𑀓𑀳𑀢𑀸; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀳𑀸𑀦𑀕𑀭𑀼𑀓𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀸, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀥𑀼𑀯𑀦𑀳𑀸𑀦𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀏𑀭𑀕𑀽 𑀫𑁄𑀭𑀕𑀽 𑀫𑀚𑁆𑀚𑀸𑀭𑀽 𑀚𑀦𑁆𑀢𑀽; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁𑁇 𑀏𑀢𑀭𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑁂 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀭𑁄𑀘𑁂𑀦𑁆𑀢𑀺—  ‘𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀳𑀺 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀳𑀺 𑀘𑀻𑀯𑀭𑀁 𑀤𑀺𑀦𑁆𑀦𑀦𑁆’𑀢𑀺𑁇 𑀢𑁂 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑀸 𑀦 𑀲𑀸𑀤𑀺𑀬𑀦𑁆𑀢𑀺—  ‘𑀫𑀸 𑀦𑁄 𑀦𑀺𑀲𑁆𑀲𑀕𑁆𑀕𑀺𑀬𑀁 𑀅𑀳𑁄𑀲𑀻’𑀢𑀺; 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀪𑀕𑀯𑀸 𑀘𑀻𑀯𑀭𑁂 𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

(259.)

1184\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀯𑀺𑀦𑀬𑀥𑀭𑀧𑀜𑁆𑀘𑀫𑁂𑀦 𑀕𑀡𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀢𑀢𑁆𑀭𑀺𑀫𑁂 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀦𑀧𑀤𑀸—  𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀕𑀚𑀗𑁆𑀕𑀮𑀁 𑀦𑀸𑀫 𑀦𑀺𑀕𑀫𑁄, 𑀢𑀲𑁆𑀲 𑀧𑀭𑁂𑀦 𑀫𑀳𑀸𑀲𑀸𑀮𑀸, 𑀢𑀢𑁄 𑀧𑀭𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀦𑀧𑀤𑀸, 𑀑𑀭𑀢𑁄 𑀫𑀚𑁆𑀛𑁂; 𑀧𑀼𑀭𑀢𑁆𑀣𑀺𑀫𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀲𑀮𑁆𑀮𑀯𑀢𑀻 𑀦𑀸𑀫 𑀦𑀤𑀻, 𑀢𑀢𑁄 𑀧𑀭𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀦𑀧𑀤𑀸, 𑀑𑀭𑀢𑁄 𑀫𑀚𑁆𑀛𑁂; 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀲𑁂𑀢𑀓𑀡𑁆𑀡𑀺𑀓𑀁 𑀦𑀸𑀫 𑀦𑀺𑀕𑀫𑁄, 𑀢𑀢𑁄 𑀧𑀭𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀦𑀧𑀤𑀸, 𑀑𑀭𑀢𑁄 𑀫𑀚𑁆𑀛𑁂; 𑀧𑀘𑁆𑀙𑀺𑀫𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀣𑀽𑀡𑀁 𑀦𑀸𑀫 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀸𑀫𑁄, 𑀢𑀢𑁄 𑀧𑀭𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀦𑀧𑀤𑀸, 𑀑𑀭𑀢𑁄 𑀫𑀚𑁆𑀛𑁂; 𑀉𑀢𑁆𑀢𑀭𑀸𑀬 𑀤𑀺𑀲𑀸𑀬 𑀉𑀲𑀻𑀭𑀤𑁆𑀥𑀚𑁄 𑀦𑀸𑀫 𑀧𑀩𑁆𑀩𑀢𑁄, 𑀢𑀢𑁄 𑀧𑀭𑀸 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑀸 𑀚𑀦𑀧𑀤𑀸, 𑀑𑀭𑀢𑁄 𑀫𑀚𑁆𑀛𑁂𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀭𑀽𑀧𑁂𑀲𑀼 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀯𑀺𑀦𑀬𑀥𑀭𑀧𑀜𑁆𑀘𑀫𑁂𑀦 𑀕𑀡𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀡𑁆𑀳𑀼𑀢𑁆𑀢𑀭𑀸 𑀪𑀽𑀫𑀺 𑀔𑀭𑀸 𑀕𑁄𑀓𑀡𑁆𑀝𑀓𑀳𑀢𑀸𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀁𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀳𑀸𑀦𑀕𑀭𑀼𑀓𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀤𑀓𑀲𑀼𑀤𑁆𑀥𑀺𑀓𑀸𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀥𑀼𑀯𑀦𑀳𑀸𑀦𑀁𑁇 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀏𑀭𑀕𑀽 𑀫𑁄𑀭𑀕𑀽 𑀫𑀚𑁆𑀚𑀸𑀭𑀽 𑀚𑀦𑁆𑀢𑀽; 𑀏𑀯𑀫𑁂𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁂 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀘𑀫𑁆𑀫𑀸𑀦𑀺 𑀅𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀺, 𑀏𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀁 𑀅𑀚𑀘𑀫𑁆𑀫𑀁 𑀫𑀺𑀕𑀘𑀫𑁆𑀫𑀁𑁇 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀦𑀺𑀲𑁆𑀲𑀻𑀫𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀘𑀻𑀯𑀭𑀁 𑀤𑁂𑀦𑁆𑀢𑀺—  ‘𑀇𑀫𑀁 𑀘𑀻𑀯𑀭𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀤𑁂𑀫𑀸’𑀢𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀤𑀺𑀢𑀼𑀁, 𑀦 𑀢𑀸𑀯 𑀢𑀁 𑀕𑀡𑀦𑀽𑀧𑀕𑀁 𑀬𑀸𑀯 𑀦 𑀳𑀢𑁆𑀣𑀁 𑀕𑀘𑁆𑀙𑀢𑀻”𑀢𑀺𑁇

---

1185\. 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀧𑀜𑁆𑀘𑀫𑁄𑁇



---

1186\. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



1187\. _𑀭𑀸𑀚𑀸 𑀘 𑀫𑀸𑀕𑀥𑁄 𑀲𑁄𑀡𑁄,_  
_𑀅𑀲𑀻𑀢𑀺𑀲𑀳𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀭𑁄;_  
_𑀲𑀸𑀕𑀢𑁄 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑀲𑁆𑀫𑀺𑀁,_  
_𑀩𑀳𑀼𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑀺 𑀉𑀢𑁆𑀢𑀭𑀺𑀁𑁇_  


1188\. _𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀭𑀤𑁆𑀥𑀪𑀺𑀚𑁆𑀚𑀺𑀁𑀲𑀼,_  
_𑀯𑀻𑀡𑀁 𑀏𑀓𑀧𑀮𑀸𑀲𑀺𑀓𑀁;_  
_𑀦𑀻𑀮𑀸 𑀧𑀻𑀢𑀸 𑀮𑁄𑀳𑀺𑀢𑀺𑀓𑀸,_  
_𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀸 𑀓𑀡𑁆𑀳𑀫𑁂𑀯 𑀘𑁇_  


1189\. _𑀫𑀳𑀸𑀭𑀗𑁆𑀕𑀫𑀳𑀸𑀦𑀸𑀫𑀸,_  
_𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀘 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀺;_  
_𑀔𑀮𑁆𑀮𑀓𑀸 𑀧𑀼𑀝𑀧𑀸𑀮𑀺 𑀘,_  
_𑀢𑀽𑀮𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀫𑁂𑀡𑁆𑀟𑀚𑀸𑁇_  


1190\. _𑀯𑀺𑀘𑁆𑀙𑀺𑀓𑀸 𑀫𑁄𑀭𑀘𑀺𑀢𑁆𑀭𑀸 𑀘,_  
_𑀲𑀻𑀳𑀩𑁆𑀬𑀕𑁆𑀖𑀸 𑀘 𑀤𑀻𑀧𑀺𑀓𑀸;_  
_𑀅𑀚𑀺𑀦𑀼𑀤𑁆𑀤𑀸 𑀫𑀚𑁆𑀚𑀸𑀭𑀻 𑀘,_  
_𑀓𑀸𑀍𑀅𑀮𑀼𑀯𑀓𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸𑁇_  


1191\. _𑀨𑀮𑀺𑀢𑀼𑀧𑀸𑀳𑀦𑀸 𑀔𑀺𑀮𑀸,_  
_𑀥𑁄𑀢𑀔𑀸𑀡𑀼𑀔𑀝𑀔𑀝𑀸;_  
_𑀢𑀸𑀮𑀯𑁂𑀍𑀉𑀢𑀺𑀡𑀁 𑀘𑁂𑀯,_  
_𑀫𑀼𑀜𑁆𑀚𑀧𑀩𑁆𑀩𑀚𑀳𑀺𑀦𑁆𑀢𑀸𑀮𑀸𑁇_  


1192\. _𑀓𑀫𑀮𑀓𑀫𑁆𑀩𑀮𑀲𑁄𑀯𑀡𑁆𑀡𑀸,_  
_𑀭𑀽𑀧𑀺𑀓𑀸 𑀫𑀡𑀺𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀸;_  
_𑀨𑀮𑀺𑀓𑀸 𑀓𑀁𑀲𑀓𑀸𑀘𑀸 𑀘,_  
_𑀢𑀺𑀧𑀼𑀲𑀻𑀲𑀜𑁆𑀘 𑀢𑀫𑁆𑀩𑀓𑀸𑁇_  


1193\. _𑀕𑀸𑀯𑀻 𑀬𑀸𑀦𑀁 𑀕𑀺𑀮𑀸𑀦𑁄 𑀘,_  
_𑀧𑀼𑀭𑀺𑀲𑀸𑀬𑀼𑀢𑁆𑀢𑀲𑀺𑀯𑀺𑀓𑀸;_  
_𑀲𑀬𑀦𑀸𑀦𑀺 𑀫𑀳𑀸𑀘𑀫𑁆𑀫𑀸,_  
_𑀕𑁄𑀘𑀫𑁆𑀫𑁂𑀳𑀺 𑀘 𑀧𑀸𑀧𑀓𑁄𑁇_  


1194\. _𑀕𑀺𑀳𑀻𑀦𑀁 𑀘𑀫𑁆𑀫𑀯𑀤𑁆𑀥𑁂𑀳𑀺,_  
_𑀧𑀯𑀺𑀲𑀦𑁆𑀢𑀺 𑀕𑀺𑀮𑀸𑀬𑀦𑁄;_  
_𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀬𑀦𑁄 𑀲𑁄𑀡𑁄,_  
_𑀲𑀭𑁂𑀦 𑀅𑀝𑁆𑀞𑀓𑀯𑀕𑁆𑀕𑀺𑀓𑀁𑁇_  


1195\. _𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀧𑀜𑁆𑀘𑀳𑀺,_  
_𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀸 𑀥𑀼𑀯𑀲𑀺𑀦𑀸;_  
_𑀘𑀫𑁆𑀫𑀢𑁆𑀣𑀭𑀡𑀸𑀦𑀼𑀜𑁆𑀜𑀸𑀲𑀺,_  
_𑀦 𑀢𑀸𑀯 𑀕𑀡𑀦𑀽𑀧𑀕𑀁;_  
_𑀅𑀤𑀸𑀲𑀺 𑀫𑁂 𑀯𑀭𑁂 𑀧𑀜𑁆𑀘,_  
_𑀲𑁄𑀡𑀢𑁆𑀣𑁂𑀭𑀲𑁆𑀲 𑀦𑀸𑀬𑀓𑁄𑀢𑀺𑁇_  


---

1196\. 𑀇𑀫𑀫𑁆𑀳𑀺 𑀔𑀦𑁆𑀥𑀓𑁂 𑀯𑀢𑁆𑀣𑀽𑀦𑀺 𑀢𑁂𑀲𑀝𑁆𑀞𑀺𑁇



---
