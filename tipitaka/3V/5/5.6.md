# 𑁫𑁇𑁬 𑀓𑀝𑁆𑀞𑀧𑀸𑀤𑀼𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀓𑀝𑁆𑀞𑀧𑀸𑀤𑀼𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

(250.)

1157. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀓𑀝𑁆𑀞𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸 𑀔𑀝𑀔𑀝𑀲𑀤𑁆𑀤𑀸, 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀸𑀚𑀓𑀣𑀁, 𑀘𑁄𑀭𑀓𑀣𑀁, 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀓𑀣𑀁, 𑀲𑁂𑀦𑀸𑀓𑀣𑀁, 𑀪𑀬𑀓𑀣𑀁, 𑀬𑀼𑀤𑁆𑀥𑀓𑀣𑀁, 𑀅𑀦𑁆𑀦𑀓𑀣𑀁, 𑀧𑀸𑀦𑀓𑀣𑀁, 𑀯𑀢𑁆𑀣𑀓𑀣𑀁, 𑀲𑀬𑀦𑀓𑀣𑀁, 𑀫𑀸𑀮𑀸𑀓𑀣𑀁, 𑀕𑀦𑁆𑀥𑀓𑀣𑀁, 𑀜𑀸𑀢𑀺𑀓𑀣𑀁, 𑀬𑀸𑀦𑀓𑀣𑀁, 𑀕𑀸𑀫𑀓𑀣𑀁, 𑀦𑀺𑀕𑀫𑀓𑀣𑀁, 𑀦𑀕𑀭𑀓𑀣𑀁, 𑀚𑀦𑀧𑀤𑀓𑀣𑀁, 𑀇𑀢𑁆𑀣𑀺𑀓𑀣𑀁, 𑀲𑀽𑀭𑀓𑀣𑀁, 𑀯𑀺𑀲𑀺𑀔𑀸𑀓𑀣𑀁, 𑀓𑀼𑀫𑁆𑀪𑀝𑁆𑀞𑀸𑀦𑀓𑀣𑀁, 𑀧𑀼𑀩𑁆𑀩𑀧𑁂𑀢𑀓𑀣𑀁, 𑀦𑀸𑀦𑀢𑁆𑀢𑀓𑀣𑀁, 𑀮𑁄𑀓𑀓𑁆𑀔𑀸𑀬𑀺𑀓𑀁, 𑀲𑀫𑀼𑀤𑁆𑀤𑀓𑁆𑀔𑀸𑀬𑀺𑀓𑀁, 𑀇𑀢𑀺𑀪𑀯𑀸𑀪𑀯𑀓𑀣𑀁 𑀇𑀢𑀺 𑀯𑀸; 𑀓𑀻𑀝𑀓𑀫𑁆𑀧𑀺 𑀅𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀘𑀸𑀯𑁂𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀓𑀝𑁆𑀞𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸 𑀔𑀝𑀔𑀝𑀲𑀤𑁆𑀤𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀸𑀚𑀓𑀣𑀁, 𑀘𑁄𑀭𑀓𑀣𑀁…𑀧𑁂…  𑀇𑀢𑀺𑀪𑀯𑀸𑀪𑀯𑀓𑀣𑀁 𑀇𑀢𑀺 𑀯𑀸, 𑀓𑀻𑀝𑀓𑀫𑁆𑀧𑀺 𑀅𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀘𑀸𑀯𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀓𑀝𑁆𑀞𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺, 𑀉𑀘𑁆𑀘𑀸𑀲𑀤𑁆𑀤𑀸 𑀫𑀳𑀸𑀲𑀤𑁆𑀤𑀸 𑀔𑀝𑀔𑀝𑀲𑀤𑁆𑀤𑀸, 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀓𑀣𑀁 𑀓𑀣𑁂𑀦𑁆𑀢𑀸, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀭𑀸𑀚𑀓𑀣𑀁, 𑀘𑁄𑀭𑀓𑀣𑀁…𑀧𑁂…  𑀇𑀢𑀺𑀪𑀯𑀸𑀪𑀯𑀓𑀣𑀁 𑀇𑀢𑀺 𑀯𑀸, 𑀓𑀻𑀝𑀓𑀫𑁆𑀧𑀺 𑀅𑀓𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀧𑀺 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀘𑀸𑀯𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀝𑁆𑀞𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1158. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀇𑀲𑀺𑀧𑀢𑀦𑁂 𑀫𑀺𑀕𑀤𑀸𑀬𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀓𑀝𑁆𑀞𑀧𑀸𑀤𑀼𑀓𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸”𑀢𑀺—  𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑁂 𑀙𑁂𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺; 𑀢𑀸𑀦𑀺 𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑀸𑀦𑀺 𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀫𑀺𑀮𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑁂 𑀙𑁂𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀢𑀸𑀦𑀺 𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑀸𑀦𑀺 𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀫𑀺𑀮𑀸𑀬𑀦𑁆𑀢𑀺; 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

1159. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑁂 𑀙𑁂𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺; 𑀢𑀸𑀦𑀺 𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑀸𑀦𑀺 𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀫𑀺𑀮𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑁂 𑀙𑁂𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺; 𑀢𑀸𑀦𑀺 𑀢𑀸𑀮𑀢𑀭𑀼𑀡𑀸𑀦𑀺 𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀫𑀺𑀮𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀚𑀻𑀯𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀁𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1160. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽—  “𑀪𑀕𑀯𑀢𑀸 𑀢𑀸𑀮𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸”𑀢𑀺—  𑀯𑁂𑀍𑀉𑀢𑀭𑀼𑀡𑁂 𑀙𑁂𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀍𑀉𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀯𑁂𑀍𑀉𑀢𑀭𑀼𑀡𑀸𑀦𑀺 𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀫𑀺𑀮𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀯𑁂𑀍𑀉𑀢𑀭𑀼𑀡𑁂 𑀙𑁂𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀯𑁂𑀍𑀉𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑀸𑀦𑀺 𑀯𑁂𑀍𑀉𑀢𑀭𑀼𑀡𑀸𑀦𑀺 𑀙𑀺𑀦𑁆𑀦𑀸𑀦𑀺 𑀫𑀺𑀮𑀸𑀬𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀚𑀻𑀯𑀲𑀜𑁆𑀜𑀺𑀦𑁄 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀭𑀼𑀓𑁆𑀔𑀲𑁆𑀫𑀺𑀁…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑁂𑀍𑀉𑀧𑀢𑁆𑀢𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1162. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀤𑁆𑀤𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀸𑀤𑀼𑀓𑀫𑀡𑁆𑀟𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑀺𑀡𑀧𑀸𑀤𑀼𑀓𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑀧𑀺…𑀧𑁂…  𑀭𑀺𑀜𑁆𑀘𑀦𑁆𑀢𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀁 𑀅𑀥𑀺𑀲𑀻𑀮𑀁 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀦𑁆”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀧𑀸𑀤𑀼𑀓𑀫𑀡𑁆𑀟𑀦𑀸𑀦𑀼𑀬𑁄𑀕𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑀺𑀡𑀧𑀸𑀤𑀼𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀓𑀸𑀭𑀸𑀧𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑀧𑀺…𑀧𑁂…  𑀭𑀺𑀜𑁆𑀘𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀉𑀤𑁆𑀤𑁂𑀲𑀁 𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀁 𑀅𑀥𑀺𑀲𑀻𑀮𑀁 𑀅𑀥𑀺𑀘𑀺𑀢𑁆𑀢𑀁 𑀅𑀥𑀺𑀧𑀜𑁆𑀜𑀁𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀡𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑀼𑀜𑁆𑀚𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀧𑀩𑁆𑀩𑀚𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀳𑀺𑀦𑁆𑀢𑀸𑀮𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀓𑀫𑀮𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀓𑀫𑁆𑀩𑀮𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀭𑀽𑀧𑀺𑀬𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑀡𑀺𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀯𑁂𑀍𑀉𑀭𑀺𑀬𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀨𑀮𑀺𑀓𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀓𑀁𑀲𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀓𑀸𑀘𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀢𑀺𑀧𑀼𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑀻𑀲𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀢𑀫𑁆𑀩𑀮𑁄𑀳𑀫𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀘𑀺 𑀲𑀗𑁆𑀓𑀫𑀦𑀺𑀬𑀸 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀲𑁆𑀲𑁄 𑀧𑀸𑀤𑀼𑀓𑀸 𑀥𑀼𑀯𑀝𑁆𑀞𑀸𑀦𑀺𑀬𑀸 𑀅𑀲𑀗𑁆𑀓𑀫𑀦𑀺𑀬𑀸𑀬𑁄—  𑀯𑀘𑁆𑀘𑀧𑀸𑀤𑀼𑀓𑀁, 𑀧𑀲𑁆𑀲𑀸𑀯𑀧𑀸𑀤𑀼𑀓𑀁, 𑀆𑀘𑀫𑀦𑀧𑀸𑀤𑀼𑀓𑀦𑁆”𑀢𑀺𑁇

1163. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀤𑁆𑀤𑀺𑀬𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀘𑀺𑀭𑀯𑀢𑀺𑀬𑀸 𑀦𑀤𑀺𑀬𑀸 𑀕𑀸𑀯𑀻𑀦𑀁 𑀢𑀭𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀺𑀲𑀸𑀡𑁂𑀲𑀼𑀧𑀺 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀓𑀡𑁆𑀡𑁂𑀲𑀼𑀧𑀺 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀕𑀻𑀯𑀸𑀬𑀧𑀺 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀙𑁂𑀧𑁆𑀧𑀸𑀧𑀺 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺, 𑀧𑀺𑀝𑁆𑀞𑀺𑀫𑁆𑀧𑀺 𑀅𑀪𑀺𑀭𑀼𑀳𑀦𑁆𑀢𑀺, 𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑀸𑀧𑀺 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀙𑀼𑀧𑀦𑁆𑀢𑀺, 𑀯𑀘𑁆𑀙𑀢𑀭𑀺𑀫𑁆𑀧𑀺 𑀑𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀫𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀕𑀸𑀯𑀻𑀦𑀁 𑀢𑀭𑀦𑁆𑀢𑀻𑀦𑀁 𑀯𑀺𑀲𑀸𑀡𑁂𑀲𑀼𑀧𑀺 𑀕𑀳𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇

1164. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀸𑀯𑀻𑀦𑀁 𑀯𑀺𑀲𑀸𑀡𑁂𑀲𑀼 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀓𑀡𑁆𑀡𑁂𑀲𑀼 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀕𑀻𑀯𑀸𑀬 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀙𑁂𑀧𑁆𑀧𑀸𑀬 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑀁, 𑀦 𑀧𑀺𑀝𑁆𑀞𑀺 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀅𑀪𑀺𑀭𑀼𑀳𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀢𑁆𑀢𑀘𑀺𑀢𑁆𑀢𑁂𑀦 𑀅𑀗𑁆𑀕𑀚𑀸𑀢𑀁 𑀙𑀼𑀧𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀙𑀼𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀣𑀼𑀮𑁆𑀮𑀘𑁆𑀘𑀬𑀲𑁆𑀲𑁇 𑀦 𑀯𑀘𑁆𑀙𑀢𑀭𑀻 𑀫𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀫𑀸𑀭𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
