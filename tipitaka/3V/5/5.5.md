# 𑁫𑁇𑁫 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂𑀉𑀧𑀸𑀳𑀦𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂𑀉𑀧𑀸𑀳𑀦𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

(248.)

1153\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁄 𑀘𑀗𑁆𑀓𑀫𑀢𑀺𑁇 “𑀲𑀢𑁆𑀣𑀸 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁄 𑀘𑀗𑁆𑀓𑀫𑀢𑀻”𑀢𑀺—  𑀣𑁂𑀭𑀸𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑀸 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀲𑀢𑁆𑀣𑀭𑀺 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂, 𑀣𑁂𑀭𑁂𑀲𑀼𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂𑀲𑀼 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂𑀲𑀼, 𑀲𑁅𑀧𑀸𑀳𑀦𑀸 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀲𑀢𑁆𑀣𑀭𑀺 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂, 𑀣𑁂𑀭𑁂𑀲𑀼𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂𑀲𑀼 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂𑀲𑀼, 𑀲𑁅𑀧𑀸𑀳𑀦𑀸 𑀘𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀲𑀢𑁆𑀣𑀭𑀺 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂, 𑀣𑁂𑀭𑁂𑀲𑀼𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂𑀲𑀼 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂𑀲𑀼, 𑀲𑁅𑀧𑀸𑀳𑀦𑀸 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀲𑀢𑁆𑀣𑀭𑀺 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂, 𑀣𑁂𑀭𑁂𑀲𑀼𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂𑀲𑀼 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂𑀲𑀼, 𑀲𑁅𑀧𑀸𑀳𑀦𑀸 𑀘𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀳𑀻 𑀑𑀤𑀸𑀢𑀯𑀢𑁆𑀣𑀯𑀲𑀦𑀓𑀸 𑀅𑀪𑀺𑀚𑀻𑀯𑀦𑀺𑀓𑀲𑁆𑀲 𑀲𑀺𑀧𑁆𑀧𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸 𑀆𑀘𑀭𑀺𑀬𑁂𑀲𑀼 𑀲𑀕𑀸𑀭𑀯𑀸 𑀲𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀸 𑀲𑀪𑀸𑀕𑀯𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀥 𑀔𑁄 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀪𑁂𑀣, 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀆𑀘𑀭𑀺𑀬𑁂𑀲𑀼 𑀆𑀘𑀭𑀺𑀬𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀲𑀼 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀅𑀕𑀸𑀭𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀲𑀪𑀸𑀕𑀯𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀸𑀣𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀘𑀭𑀺𑀬𑁂𑀲𑀼 𑀆𑀘𑀭𑀺𑀬𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀲𑀼 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀅𑀦𑀼𑀧𑀸𑀳𑀦𑁂𑀲𑀼 𑀘𑀗𑁆𑀓𑀫𑀫𑀸𑀦𑁂𑀲𑀼 𑀲𑁅𑀧𑀸𑀳𑀦𑁂𑀦 𑀘𑀗𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀘𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

(249.)

1154\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀸𑀤𑀔𑀺𑀮𑀸𑀩𑀸𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑁂𑀦𑀸𑀲𑀦𑀘𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀁 𑀇𑀫𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀆𑀩𑀸𑀥𑁄”𑀢𑀺? “𑀇𑀫𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀧𑀸𑀤𑀔𑀺𑀮𑀸𑀩𑀸𑀥𑁄; 𑀇𑀫𑀁 𑀫𑀬𑀁 𑀧𑀭𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀉𑀘𑁆𑀘𑀸𑀭𑀫𑁆𑀧𑀺 𑀧𑀲𑁆𑀲𑀸𑀯𑀫𑁆𑀧𑀺 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀲𑁆𑀲 𑀧𑀸𑀤𑀸 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀸, 𑀧𑀸𑀤𑀸 𑀯𑀸 𑀨𑀮𑀺𑀢𑀸, 𑀧𑀸𑀤𑀔𑀺𑀮𑁄 𑀯𑀸 𑀆𑀩𑀸𑀥𑁄 𑀉𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1155\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀥𑁄𑀢𑁂𑀳𑀺 𑀧𑀸𑀤𑁂𑀳𑀺 𑀫𑀜𑁆𑀘𑀫𑁆𑀧𑀺 𑀧𑀻𑀞𑀫𑁆𑀧𑀺 𑀅𑀪𑀺𑀭𑀼𑀳𑀦𑁆𑀢𑀺; 𑀘𑀻𑀯𑀭𑀫𑁆𑀧𑀺 𑀲𑁂𑀦𑀸𑀲𑀦𑀫𑁆𑀧𑀺 𑀤𑀼𑀲𑁆𑀲𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, ‘𑀇𑀤𑀸𑀦𑀺 𑀫𑀜𑁆𑀘𑀁 𑀯𑀸 𑀧𑀻𑀞𑀁 𑀯𑀸 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺 𑀉𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

1156\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀉𑀧𑁄𑀲𑀣𑀕𑁆𑀕𑀫𑁆𑀧𑀺 𑀲𑀦𑁆𑀦𑀺𑀲𑀚𑁆𑀚𑀫𑁆𑀧𑀺 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀔𑀸𑀡𑀼𑀫𑁆𑀧𑀺 𑀓𑀡𑁆𑀝𑀓𑀫𑁆𑀧𑀺 𑀅𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺; 𑀧𑀸𑀤𑀸 𑀤𑀼𑀓𑁆𑀔𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑀸𑀭𑀸𑀫𑁂 𑀉𑀧𑀸𑀳𑀦𑀁 𑀥𑀸𑀭𑁂𑀢𑀼𑀁, 𑀉𑀓𑁆𑀓𑀁, 𑀧𑀤𑀻𑀧𑀁, 𑀓𑀢𑁆𑀢𑀭𑀤𑀡𑁆𑀟𑀦𑁆”𑀢𑀺𑁇
