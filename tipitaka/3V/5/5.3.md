# 𑁫𑁇𑁩 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀺𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀺𑀓𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

(246.)

1148. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀩𑁆𑀩𑀧𑀻𑀢𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…  𑀲𑀩𑁆𑀩𑀮𑁄𑀳𑀺𑀢𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…  𑀲𑀩𑁆𑀩𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…  𑀲𑀩𑁆𑀩𑀓𑀡𑁆𑀳𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…  𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀭𑀗𑁆𑀕𑀭𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…  𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀦𑀸𑀫𑀭𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑀦𑀻𑀮𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸…𑀧𑁂…  𑀦 𑀲𑀩𑁆𑀩𑀧𑀻𑀢𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑀩𑁆𑀩𑀮𑁄𑀳𑀺𑀢𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑀩𑁆𑀩𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑀩𑁆𑀩𑀓𑀡𑁆𑀳𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀭𑀗𑁆𑀕𑀭𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀲𑀩𑁆𑀩𑀫𑀳𑀸𑀦𑀸𑀫𑀭𑀢𑁆𑀢𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1149. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀻𑀮𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀻𑀢𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀺𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀓𑀡𑁆𑀳𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸𑀭𑀗𑁆𑀕𑀭𑀢𑁆𑀢𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀫𑀳𑀸𑀦𑀸𑀫𑀭𑀢𑁆𑀢𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀻𑀮𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸…𑀧𑁂…  𑀦 𑀧𑀻𑀢𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀮𑁄𑀳𑀺𑀢𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀺𑀓𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀓𑀡𑁆𑀳𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑀳𑀸𑀭𑀗𑁆𑀕𑀭𑀢𑁆𑀢𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑀳𑀸𑀦𑀸𑀫𑀭𑀢𑁆𑀢𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1150. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀔𑀮𑁆𑀮𑀓𑀩𑀤𑁆𑀥𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀼𑀝𑀩𑀤𑁆𑀥𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀧𑀸𑀮𑀺𑀕𑀼𑀡𑁆𑀞𑀺𑀫𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀢𑀽𑀮𑀧𑀼𑀡𑁆𑀡𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀧𑀢𑁆𑀢𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀫𑁂𑀡𑁆𑀟𑀯𑀺𑀲𑀸𑀡𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀅𑀚𑀯𑀺𑀲𑀸𑀡𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀯𑀺𑀘𑁆𑀙𑀺𑀓𑀸𑀍𑀇𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀫𑁄𑀭𑀧𑀺𑀜𑁆𑀙𑀧𑀭𑀺𑀲𑀺𑀩𑁆𑀩𑀺𑀢𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀘𑀺𑀢𑁆𑀭𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀮𑁆𑀮𑀓𑀩𑀤𑁆𑀥𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸…𑀧𑁂…  𑀦 𑀧𑀼𑀝𑀩𑀤𑁆𑀥𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀧𑀸𑀮𑀺𑀕𑀼𑀡𑁆𑀞𑀺𑀫𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀢𑀽𑀮𑀧𑀼𑀡𑁆𑀡𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀢𑀺𑀢𑁆𑀢𑀺𑀭𑀧𑀢𑁆𑀢𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑁂𑀡𑁆𑀟𑀯𑀺𑀲𑀸𑀡𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀅𑀚𑀯𑀺𑀲𑀸𑀡𑀯𑀤𑁆𑀥𑀺𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀯𑀺𑀘𑁆𑀙𑀺𑀓𑀸𑀍𑀇𑀓𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑁄𑀭𑀧𑀺𑀜𑁆𑀙𑀧𑀭𑀺𑀲𑀺𑀩𑁆𑀩𑀺𑀢𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀘𑀺𑀢𑁆𑀭𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1151. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀩𑁆𑀬𑀕𑁆𑀖𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀤𑀻𑀧𑀺𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀅𑀚𑀺𑀦𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀉𑀤𑁆𑀤𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀫𑀚𑁆𑀚𑀸𑀭𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀓𑀸𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺, 𑀮𑀼𑀯𑀓𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄 𑀥𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀺𑀳𑀻 𑀓𑀸𑀫𑀪𑁄𑀕𑀺𑀦𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀻𑀳𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸…𑀧𑁂…  𑀦 𑀩𑁆𑀬𑀕𑁆𑀖𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀤𑀻𑀧𑀺𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀅𑀚𑀺𑀦𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀉𑀤𑁆𑀤𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀫𑀚𑁆𑀚𑀸𑀭𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀓𑀸𑀍𑀅𑀓𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸, 𑀦 𑀮𑀼𑀯𑀓𑀘𑀫𑁆𑀫𑀧𑀭𑀺𑀓𑁆𑀔𑀝𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
