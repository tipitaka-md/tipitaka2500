# 𑁫𑁇𑁪 𑀑𑀫𑀼𑀓𑁆𑀓𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀸𑀦𑀼𑀚𑀸𑀦𑀦

* 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀑𑀫𑀼𑀓𑁆𑀓𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(247.)

1152\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺, 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀘𑁆𑀙𑀸𑀲𑀫𑀡𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀔𑀜𑁆𑀚𑀫𑀸𑀦𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀸 𑀆𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁; 𑀤𑀺𑀲𑁆𑀯𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀆𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺; 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀺𑀲𑁆𑀲, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀔𑀜𑁆𑀚𑀢𑀻”𑀢𑀺? “𑀧𑀸𑀤𑀸 𑀫𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀨𑀮𑀺𑀢𑀸”𑀢𑀺𑁇 “𑀳𑀦𑁆𑀤, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀸 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀸”𑀢𑀺𑁇 “𑀕𑀡𑁆𑀳𑀸𑀳𑁂𑀢𑀸, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀉𑀧𑀸𑀳𑀦𑀸𑀬𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀑𑀫𑀼𑀓𑁆𑀓𑀁 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀁𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀦𑀯𑀸 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
