# 𑁫𑁇𑁨 𑀤𑀺𑀕𑀼𑀡𑀸𑀤𑀺𑀉𑀧𑀸𑀳𑀦𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

### 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀤𑀺𑀕𑀼𑀡𑀸𑀤𑀺𑀉𑀧𑀸𑀳𑀦𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

(245.)

1147\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀯𑀁 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀅𑀢𑁆𑀣𑁄 𑀘 𑀯𑀼𑀢𑁆𑀢𑁄, 𑀅𑀢𑁆𑀢𑀸 𑀘 𑀅𑀦𑀼𑀧𑀦𑀻𑀢𑁄𑁇 𑀅𑀣 𑀘 𑀧𑀦𑀺𑀥𑁂𑀓𑀘𑁆𑀘𑁂 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀳𑀲𑀫𑀸𑀦𑀓𑀁 𑀫𑀜𑁆𑀜𑁂 𑀅𑀜𑁆𑀜𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑁄𑀦𑁆𑀢𑀺, 𑀢𑁂 𑀧𑀘𑁆𑀙𑀸 𑀯𑀺𑀖𑀸𑀢𑀁 𑀆𑀧𑀚𑁆𑀚𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁄𑀡𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀲𑁄𑀡, 𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑁄𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺 𑀢𑁂, 𑀲𑁄𑀡, 𑀏𑀓𑀧𑀮𑀸𑀲𑀺𑀓𑀁 𑀉𑀧𑀸𑀳𑀦𑀦𑁆”𑀢𑀺𑁇 “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀲𑀻𑀢𑀺𑀲𑀓𑀝𑀯𑀸𑀳𑁂 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀑𑀳𑀸𑀬 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀲𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀺𑀓𑀜𑁆𑀘 𑀅𑀦𑀻𑀓𑀁𑁇 𑀅𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀓𑀧𑀮𑀸𑀲𑀺𑀓𑀜𑁆𑀘𑁂 𑀉𑀧𑀸𑀳𑀦𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀢𑀲𑁆𑀲 𑀫𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀢𑁆𑀢𑀸𑀭𑁄 ‘𑀲𑁄𑀡𑁄 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑁄 𑀅𑀲𑀻𑀢𑀺𑀲𑀓𑀝𑀯𑀸𑀳𑁂 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀑𑀳𑀸𑀬 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀲𑀢𑁆𑀢𑀳𑀢𑁆𑀣𑀺𑀓𑀜𑁆𑀘 𑀅𑀦𑀻𑀓𑀁𑁇 𑀲𑁄 𑀤𑀸𑀦𑀸𑀬𑀁 𑀏𑀓𑀧𑀮𑀸𑀲𑀺𑀓𑀸𑀲𑀼 𑀉𑀧𑀸𑀳𑀦𑀸𑀲𑀼 𑀲𑀢𑁆𑀢𑁄’𑀢𑀺𑁇 𑀲𑀘𑁂 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀳𑀫𑁆𑀧𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺; 𑀦𑁄 𑀘𑁂 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀳𑀫𑁆𑀧𑀺 𑀦 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀧𑀮𑀸𑀲𑀺𑀓𑀁 𑀉𑀧𑀸𑀳𑀦𑀁𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀺𑀕𑀼𑀡𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀦 𑀢𑀺𑀕𑀼𑀡𑀸 𑀉𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀦 𑀕𑀼𑀡𑀗𑁆𑀕𑀼𑀡𑀽𑀧𑀸𑀳𑀦𑀸 𑀥𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀥𑀸𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
