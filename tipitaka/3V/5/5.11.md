# 𑁫𑁇𑁧𑁧 𑀲𑁄𑀡𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀯𑀢𑁆𑀣𑀼

* 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀲𑁄𑀡𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀯𑀢𑁆𑀣𑀼

(257.)

1176\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀅𑀯𑀦𑁆𑀢𑀻𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀼𑀭𑀭𑀖𑀭𑁂 𑀧𑀧𑀢𑀓𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀳𑁄𑀢𑀺𑁇

1177\. 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀣𑀸 𑀬𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂𑀦 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄”𑀢𑀺𑁇 ( ) “𑀤𑀼𑀓𑁆𑀓𑀭𑀁 𑀔𑁄, 𑀲𑁄𑀡, 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀁 𑀏𑀓𑀪𑀢𑁆𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀲𑁄𑀡, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀅𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑁄 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀲𑀦𑀁 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚, 𑀓𑀸𑀮𑀬𑀼𑀢𑁆𑀢𑀁 𑀏𑀓𑀲𑁂𑀬𑁆𑀬𑀁 𑀏𑀓𑀪𑀢𑁆𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀲𑁆𑀲 𑀬𑁄 𑀅𑀳𑁄𑀲𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑁄 𑀲𑁄 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑀺𑁇

1178\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀣𑀸 𑀬𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁂𑀦 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑀺𑀢𑀁 𑀆𑀚𑀸𑀦𑀸𑀫𑀺, 𑀦𑀬𑀺𑀤𑀁 𑀲𑀼𑀓𑀭𑀁 𑀅𑀕𑀸𑀭𑀁 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀸 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀏𑀓𑀦𑁆𑀢𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀲𑀗𑁆𑀔𑀮𑀺𑀔𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁𑁇 𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀼𑀁𑁇 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀲𑁄𑀡𑀁 𑀉𑀧𑀸𑀲𑀓𑀁 𑀓𑀼𑀝𑀺𑀓𑀡𑁆𑀡𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀯𑀦𑁆𑀢𑀺𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀧𑀣𑁄 𑀅𑀧𑁆𑀧𑀪𑀺𑀓𑁆𑀔𑀼𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀘𑁆𑀘𑀸𑀦𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀯𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀓𑀺𑀘𑁆𑀙𑁂𑀦 𑀓𑀲𑀺𑀭𑁂𑀦 𑀢𑀢𑁄 𑀢𑀢𑁄 𑀤𑀲𑀯𑀕𑁆𑀕𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑁄𑀡𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑀺𑁇
