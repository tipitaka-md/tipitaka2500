# 𑁫𑁇𑁧 𑀲𑁄𑀡𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀯𑀢𑁆𑀣𑀼

### 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑁄𑀡𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀯𑀢𑁆𑀣𑀼

(242.)

1127\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁂 𑀧𑀩𑁆𑀩𑀢𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀕𑀸𑀫𑀲𑀳𑀲𑁆𑀲𑁂𑀲𑀼 𑀇𑀲𑁆𑀲𑀭𑀺𑀬𑀸𑀥𑀺𑀧𑀘𑁆𑀘𑀁 𑀭𑀚𑁆𑀚𑀁 𑀓𑀸𑀭𑁂𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀘𑀫𑁆𑀧𑀸𑀬𑀁 𑀲𑁄𑀡𑁄 𑀦𑀸𑀫 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀸𑀤𑀢𑀮𑁂𑀲𑀼 𑀮𑁄𑀫𑀸𑀦𑀺 𑀚𑀸𑀢𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑀸𑀦𑀺 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑁂𑀦𑀘𑀺𑀤𑁂𑀯 𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀲𑁄𑀡𑀲𑁆𑀲 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀲𑁄𑀡𑁄, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑁄𑀡𑀲𑁆𑀲 𑀆𑀕𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀲𑁄𑀡𑀁 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀭𑀸𑀚𑀸 𑀢𑁂, 𑀢𑀸𑀢 𑀲𑁄𑀡, 𑀧𑀸𑀤𑁂 𑀤𑀓𑁆𑀔𑀺𑀢𑀼𑀓𑀸𑀫𑁄𑁇 𑀫𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀲𑁄𑀡, 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀢𑁂𑀦 𑀧𑀸𑀤𑁂 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀸𑀭𑁂𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀭𑀜𑁆𑀜𑁄 𑀧𑀼𑀭𑀢𑁄 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑁇 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀢𑁂 𑀭𑀸𑀚𑀸 𑀧𑀸𑀤𑁂 𑀤𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑀁 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀁 𑀲𑀺𑀯𑀺𑀓𑀸𑀬 𑀆𑀦𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄𑀡𑁄 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑁄 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀭𑀜𑁆𑀜𑁄 𑀧𑀼𑀭𑀢𑁄 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀲𑁄𑀡𑀲𑁆𑀲 𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀲𑁆𑀲 𑀧𑀸𑀤𑀢𑀮𑁂𑀲𑀼 𑀮𑁄𑀫𑀸𑀦𑀺 𑀚𑀸𑀢𑀸𑀦𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑀸𑀦𑀺 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀅𑀢𑁆𑀣𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀲𑀺—  “𑀢𑀼𑀫𑁆𑀳𑁂 𑀔𑁆𑀯𑀢𑁆𑀣, 𑀪𑀡𑁂, 𑀫𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀺𑀓𑁂 𑀅𑀢𑁆𑀣𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀸; 𑀕𑀘𑁆𑀙𑀣, 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑀣; 𑀲𑁄 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑁆𑀧𑀭𑀸𑀬𑀺𑀓𑁂 𑀅𑀢𑁆𑀣𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

### 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑁄𑀡𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀯𑀢𑁆𑀣𑀼

1128\. 𑀅𑀣 𑀔𑁄 𑀢𑀸𑀦𑀺 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀬𑁂𑀦 𑀕𑀺𑀚𑁆𑀛𑀓𑀽𑀝𑁄 𑀧𑀩𑁆𑀩𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀕𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸𑀦𑀺 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀕𑀢𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀕𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀇𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀥𑀽𑀧𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬; 𑀲𑀸𑀥𑀼 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑀼𑀫𑁆𑀳𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀫𑀼𑀳𑀼𑀢𑁆𑀢𑀁 𑀇𑀥𑁂𑀯 𑀢𑀸𑀯 𑀳𑁄𑀣, 𑀬𑀸𑀯𑀸𑀳𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀕𑀢𑁄 𑀢𑁂𑀲𑀁 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀼𑀭𑀢𑁄 𑀧𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀸𑀬 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀇𑀥𑀽𑀧𑀲𑀗𑁆𑀓𑀦𑁆𑀢𑀸𑀦𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬; 𑀬𑀲𑁆𑀲𑀤𑀸𑀦𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀓𑀸𑀮𑀁 𑀫𑀜𑁆𑀜𑀢𑀻”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀕𑀢, 𑀯𑀺𑀳𑀸𑀭𑀧𑀘𑁆𑀙𑀸𑀬𑀸𑀬𑀁 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀕𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀧𑀻𑀞𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑁂𑀲𑀁 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀧𑀼𑀭𑀢𑁄 𑀧𑁂𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀦𑀁 𑀧𑀸𑀝𑀺𑀓𑀸𑀬 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀧𑀘𑁆𑀙𑀸𑀬𑀸𑀬𑀁 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀳𑀸𑀭𑀧𑀘𑁆𑀙𑀸𑀬𑀸𑀬𑀁 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸𑀦𑀺 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸𑀦𑀺 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀲𑀸𑀕𑀢𑀁 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀢𑀣𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀲𑀁 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀕𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀕𑀢, 𑀪𑀺𑀬𑁆𑀬𑁄𑀲𑁄 𑀫𑀢𑁆𑀢𑀸𑀬 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀤𑀲𑁆𑀲𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀕𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺𑀧𑀺, 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑀧𑀺, 𑀦𑀺𑀲𑀻𑀤𑀢𑀺𑀧𑀺, 𑀲𑁂𑀬𑁆𑀬𑀫𑁆𑀧𑀺 𑀓𑀧𑁆𑀧𑁂𑀢𑀺, 𑀥𑀽𑀫𑀸𑀬𑀢𑀺𑀧𑀺 𑀧𑀚𑁆𑀚𑀮𑀢𑀺𑀧𑀺, 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀢𑀺𑀧𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀕𑀢𑁄 𑀆𑀓𑀸𑀲𑁂 𑀅𑀦𑁆𑀢𑀮𑀺𑀓𑁆𑀔𑁂 𑀅𑀦𑁂𑀓𑀯𑀺𑀳𑀺𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺 𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸; 𑀲𑀸𑀯𑀓𑁄𑀳𑀫𑀲𑁆𑀫𑀺𑁇 𑀲𑀢𑁆𑀣𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸; 𑀲𑀸𑀯𑀓𑁄𑀳𑀫𑀲𑁆𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀸𑀦𑀺 𑀅𑀲𑀻𑀢𑀺 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 “𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀲𑀸𑀯𑀓𑁄𑀧𑀺 𑀦𑀸𑀫 𑀏𑀯𑀁 𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀏𑀯𑀁 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀅𑀳𑁄 𑀦𑀽𑀦 𑀲𑀢𑁆𑀣𑀸”𑀢𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁𑀬𑁂𑀯 𑀲𑀫𑀦𑁆𑀦𑀸𑀳𑀭𑀦𑁆𑀢𑀺, 𑀦𑁄 𑀢𑀣𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀕𑀢𑀁𑁇

### 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑁄𑀡𑀓𑁄𑀍𑀇𑀯𑀺𑀲𑀯𑀢𑁆𑀣𑀼

1129\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀲𑀁 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁, 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑁂, 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑁂, 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑁂, 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁂, 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁂, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑀁, 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀢𑁂𑀲𑀁 𑀅𑀲𑀻𑀢𑀺𑀬𑀸 𑀕𑀸𑀫𑀺𑀓𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀁 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑀸 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑀸 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀢𑁂 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑁇 𑀥𑀫𑁆𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑁂 𑀲𑀭𑀡𑀁 𑀕𑀢𑁂”𑀢𑀺𑁇
