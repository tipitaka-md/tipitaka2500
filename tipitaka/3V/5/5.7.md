# 𑁫𑁇𑁭 𑀬𑀸𑀦𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

* 𑀘𑀫𑁆𑀫𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀬𑀸𑀦𑀸𑀤𑀺𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧

(253.)

1165\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑀸𑀦𑁂𑀦 𑀬𑀸𑀬𑀦𑁆𑀢𑀺, 𑀇𑀢𑁆𑀣𑀺𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀧𑀼𑀭𑀺𑀲𑀦𑁆𑀢𑀭𑁂𑀦, 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀢𑀭𑁂𑀦𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑀗𑁆𑀕𑀸𑀫𑀳𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀦𑁂𑀦 𑀬𑀸𑀬𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀬𑀸𑀬𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

1166\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀓𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑁄 𑀕𑀫𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺? “𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀦𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀓𑁆𑀓𑁄𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑁄𑀫𑁆𑀳𑀻”𑀢𑀺𑁇 “𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀦𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸 𑀬𑀸𑀦𑀦𑁆”𑀢𑀺 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀸𑀬𑀦𑁆𑀢𑁄 𑀬𑀸𑀦𑀁 𑀦𑀸𑀪𑀺𑀭𑀼𑀳𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀬𑀸𑀦𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀢𑁆𑀣𑀺𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀼 𑀔𑁄 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀦𑀼 𑀔𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀭𑀺𑀲𑀬𑀼𑀢𑁆𑀢𑀁 𑀳𑀢𑁆𑀣𑀯𑀝𑁆𑀝𑀓𑀦𑁆”𑀢𑀺𑁇

1167\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀬𑀸𑀦𑀼𑀕𑁆𑀖𑀸𑀢𑁂𑀦 𑀩𑀸𑀍𑀳𑀢𑀭𑀁 𑀅𑀨𑀸𑀲𑀼 𑀅𑀳𑁄𑀲𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀯𑀺𑀓𑀁 𑀧𑀸𑀝𑀗𑁆𑀓𑀺𑀦𑁆”𑀢𑀺𑁇
