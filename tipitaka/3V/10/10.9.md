# 𑁧𑁦𑁇𑁯 𑀉𑀧𑀸𑀮𑀺𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑀧𑀼𑀘𑁆𑀙𑀸

### 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀉𑀧𑀸𑀮𑀺𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑀧𑀼𑀘𑁆𑀙𑀸

(476.)

2021. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀬𑀲𑁆𑀫𑀺𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑀽𑀮𑀸 𑀫𑀽𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀦𑀼 𑀔𑁄 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻”𑀢𑀺? “𑀬𑀲𑁆𑀫𑀺𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑀽𑀮𑀸 𑀫𑀽𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀲𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻”𑀢𑀺𑁇

2022. “𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀦, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀫𑀽𑀮𑀸 𑀫𑀽𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀦𑀼 𑀔𑁄 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻”𑀢𑀺? “𑀬𑀲𑁆𑀫𑀺𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀫𑀽𑀮𑀸 𑀫𑀽𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀥𑀫𑁆𑀫𑀺𑀓𑀸 𑀲𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻”𑀢𑀺𑁇

2023. “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑁄”𑀢𑀺? “𑀤𑁆𑀯𑁂𑀫𑀸, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑁄—  𑀅𑀢𑁆𑀣𑀼𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀅𑀢𑁆𑀣𑀸𑀧𑁂𑀢𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀼𑀧𑁂𑀢𑀸; 𑀅𑀢𑁆𑀣𑀼𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀅𑀢𑁆𑀣𑀼𑀧𑁂𑀢𑀸 𑀘 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀼𑀧𑁂𑀢𑀸 𑀘𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀅𑀢𑁆𑀣𑀸𑀧𑁂𑀢𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀼𑀧𑁂𑀢𑀸? 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀅𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀅𑀫𑀽𑀮𑀸 𑀫𑀽𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀅𑀢𑁆𑀣𑀸𑀧𑁂𑀢𑀸 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀼𑀧𑁂𑀢𑀸𑁇 𑀓𑀢𑀫𑀸 𑀘, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀅𑀢𑁆𑀣𑀼𑀧𑁂𑀢𑀸 𑀘 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀼𑀧𑁂𑀢𑀸 𑀘? 𑀬𑀲𑁆𑀫𑀺𑀁, 𑀉𑀧𑀸𑀮𑀺, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀫𑀽𑀮𑀸 𑀫𑀽𑀮𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀺, 𑀅𑀬𑀁 𑀯𑀼𑀘𑁆𑀘𑀢𑀺, 𑀉𑀧𑀸𑀮𑀺, 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀅𑀢𑁆𑀣𑀼𑀧𑁂𑀢𑀸 𑀘 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀼𑀧𑁂𑀢𑀸 𑀘𑁇 𑀇𑀫𑀸 𑀔𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀤𑁆𑀯𑁂 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑁄”𑀢𑀺𑁇

(477.)

2024. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

2025. _“𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀺𑀘𑁆𑀘𑁂𑀲𑀼 𑀘 𑀫𑀦𑁆𑀢𑀦𑀸𑀲𑀼 𑀘,_  
_𑀅𑀢𑁆𑀣𑁂𑀲𑀼 𑀚𑀸𑀢𑁂𑀲𑀼 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑁂𑀲𑀼 𑀘;_  
_𑀓𑀣𑀁𑀧𑀓𑀸𑀭𑁄𑀥 𑀦𑀭𑁄 𑀫𑀳𑀢𑁆𑀣𑀺𑀓𑁄,_  
_𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑀣𑀁 𑀳𑁄𑀢𑀺𑀥 𑀧𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑁄”𑀢𑀺𑁇_  


2026. _“𑀅𑀦𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀧𑀞𑀫𑁂𑀦 𑀲𑀻𑀮𑀢𑁄,_  
_𑀅𑀯𑁂𑀓𑁆𑀔𑀺𑀢𑀸𑀘𑀸𑀭𑁄 𑀲𑀼𑀲𑀁𑀯𑀼𑀢𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑁄;_  
_𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀦𑀽𑀧𑀯𑀤𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀢𑁄,_  
_𑀦 𑀳𑀺𑀲𑁆𑀲 𑀢𑀁 𑀳𑁄𑀢𑀺 𑀯𑀤𑁂𑀬𑁆𑀬𑀼 𑀬𑁂𑀦 𑀦𑀁𑁇_  


2027. _𑀲𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀲𑀻𑀮𑀯𑀺𑀲𑀼𑀤𑁆𑀥𑀺𑀬𑀸 𑀞𑀺𑀢𑁄,_  
_𑀯𑀺𑀲𑀸𑀭𑀤𑁄 𑀳𑁄𑀢𑀺 𑀯𑀺𑀲𑀬𑁆𑀳 𑀪𑀸𑀲𑀢𑀺;_  
_𑀦𑀘𑁆𑀙𑀫𑁆𑀪𑀢𑀺 𑀧𑀭𑀺𑀲𑀕𑀢𑁄 𑀦 𑀯𑁂𑀥𑀢𑀺,_  
_𑀅𑀢𑁆𑀣𑀁 𑀦 𑀳𑀸𑀧𑁂𑀢𑀺 𑀅𑀦𑀼𑀬𑁆𑀬𑀼𑀢𑀁 𑀪𑀡𑀁𑁇_  


2028. _𑀢𑀣𑁂𑀯 𑀧𑀜𑁆𑀳𑀁 𑀧𑀭𑀺𑀲𑀸𑀲𑀼 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄,_  
_𑀦 𑀘𑁂𑀯 𑀧𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀦 𑀫𑀗𑁆𑀓𑀼 𑀳𑁄𑀢𑀺;_  
_𑀲𑁄 𑀓𑀸𑀮𑀸𑀕𑀢𑀁 𑀩𑁆𑀬𑀸𑀓𑀭𑀡𑀸𑀭𑀳𑀁 𑀯𑀘𑁄,_  
_𑀭𑀜𑁆𑀚𑁂𑀢𑀺 𑀯𑀺𑀜𑁆𑀜𑀽𑀧𑀭𑀺𑀲𑀁 𑀯𑀺𑀘𑀓𑁆𑀔𑀡𑁄𑁇_  


2029. _𑀲𑀕𑀸𑀭𑀯𑁄 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼,_  
_𑀆𑀘𑁂𑀭𑀓𑀫𑁆𑀳𑀺 𑀘 𑀲𑀓𑁂 𑀯𑀺𑀲𑀸𑀭𑀤𑁄;_  
_𑀅𑀮𑀁 𑀧𑀫𑁂𑀢𑀼𑀁 𑀧𑀕𑀼𑀡𑁄 𑀓𑀣𑁂𑀢𑀯𑁂,_  
_𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀜𑁆𑀘 𑀯𑀺𑀭𑀤𑁆𑀥𑀺𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


2030. _𑀧𑀘𑁆𑀘𑀢𑁆𑀣𑀺𑀓𑀸 𑀬𑁂𑀦 𑀯𑀚𑀦𑁆𑀢𑀺 𑀦𑀺𑀕𑁆𑀕𑀳𑀁,_  
_𑀫𑀳𑀸𑀚𑀦𑁄 𑀲𑀜𑁆𑀜𑀧𑀦𑀜𑁆𑀘 𑀕𑀘𑁆𑀙𑀢𑀺;_  
_𑀲𑀓𑀜𑁆𑀘 𑀆𑀤𑀸𑀬𑀫𑀬𑀁 𑀦 𑀭𑀺𑀜𑁆𑀘𑀢𑀺,_  
_𑀯𑀺𑀬𑀸𑀓𑀭𑀁 𑀧𑀜𑁆𑀳𑀫𑀦𑀽𑀧𑀖𑀸𑀢𑀺𑀓𑀁𑁇_  


2031. _𑀤𑀽𑀢𑁂𑀬𑁆𑀬𑀓𑀫𑁆𑀫𑁂𑀲𑀼 𑀅𑀮𑀁 𑀲𑀫𑀼𑀕𑁆𑀕𑀳𑁄,_  
_𑀲𑀁𑀖𑀲𑁆𑀲 𑀓𑀺𑀘𑁆𑀘𑁂𑀲𑀼 𑀘 𑀆𑀳𑀼 𑀦𑀁 𑀬𑀣𑀸;_  
_𑀓𑀭𑀁 𑀯𑀘𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀕𑀡𑁂𑀦 𑀧𑁂𑀲𑀺𑀢𑁄,_  
_𑀅𑀳𑀁 𑀓𑀭𑁄𑀫𑀻𑀢𑀺 𑀦 𑀢𑁂𑀦 𑀫𑀜𑁆𑀜𑀢𑀺𑁇_  


2032. _𑀆𑀧𑀚𑁆𑀚𑀢𑀺 𑀬𑀸𑀯𑀢𑀓𑁂𑀲𑀼 𑀯𑀢𑁆𑀣𑀼𑀲𑀼,_  
_𑀆𑀧𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸 𑀘 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑀺;_  
_𑀏𑀢𑁂 𑀯𑀺𑀪𑀗𑁆𑀕𑀸 𑀉𑀪𑀬𑀲𑁆𑀲 𑀲𑁆𑀯𑀸𑀕𑀢𑀸,_  
_𑀆𑀧𑀢𑁆𑀢𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀦𑀧𑀤𑀲𑁆𑀲 𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


2033. _𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀢𑀺 𑀬𑀸𑀦𑀺 𑀘𑀸𑀘𑀭𑀁,_  
_𑀦𑀺𑀲𑁆𑀲𑀸𑀭𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀬𑀣𑀸 𑀘 𑀯𑀢𑁆𑀢𑀦𑀸;_  
_𑀑𑀲𑀸𑀭𑀡𑀁 𑀢𑀁𑀯𑀼𑀲𑀺𑀢𑀲𑁆𑀲 𑀚𑀦𑁆𑀢𑀼𑀦𑁄,_  
_𑀏𑀢𑀫𑁆𑀧𑀺 𑀚𑀸𑀦𑀸𑀢𑀺 𑀯𑀺𑀪𑀗𑁆𑀕𑀓𑁄𑀯𑀺𑀤𑁄𑁇_  


2034. _𑀲𑀕𑀸𑀭𑀯𑁄 𑀯𑀼𑀟𑁆𑀠𑀢𑀭𑁂𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀼,_  
_𑀦𑀯𑁂𑀲𑀼 𑀣𑁂𑀭𑁂𑀲𑀼 𑀘 𑀫𑀚𑁆𑀛𑀺𑀫𑁂𑀲𑀼 𑀘;_  
_𑀫𑀳𑀸𑀚𑀦𑀲𑁆𑀲𑀢𑁆𑀣𑀘𑀭𑁄𑀥 𑀧𑀡𑁆𑀟𑀺𑀢𑁄,_  
_𑀲𑁄 𑀢𑀸𑀤𑀺𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀥 𑀧𑀕𑁆𑀕𑀳𑀸𑀭𑀳𑁄”𑀢𑀺𑁇_  


---

2035. 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀤𑀲𑀫𑁄𑁇



---

2036. 𑀢𑀲𑁆𑀲𑀼𑀤𑁆𑀤𑀸𑀦𑀁



2037. _𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀚𑀺𑀦𑀯𑀭𑁄,_  
_𑀯𑀺𑀯𑀸𑀤𑀸𑀧𑀢𑁆𑀢𑀺𑀤𑀲𑁆𑀲𑀦𑁂;_  
_𑀦𑀼𑀓𑁆𑀔𑀺𑀧𑁂𑀬𑁆𑀬 𑀬𑀲𑁆𑀫𑀺𑀁 𑀢𑀲𑁆𑀫𑀺𑀁,_  
_𑀲𑀤𑁆𑀥𑀸𑀬𑀸𑀧𑀢𑁆𑀢𑀺 𑀤𑁂𑀲𑀬𑁂𑁇_  


2038. _𑀅𑀦𑁆𑀢𑁄𑀲𑀻𑀫𑀸𑀬𑀁 𑀢𑀢𑁆𑀣𑁂𑀯,_  
_𑀩𑀸𑀮𑀓𑀜𑁆𑀘𑁂𑀯 𑀯𑀁𑀲𑀤𑀸;_  
_𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀸 𑀘 𑀲𑀸𑀯𑀢𑁆𑀣𑀺,_  
_𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀘 𑀓𑁄𑀮𑀺𑀢𑁄𑁇_  


2039. _𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀓𑀘𑁆𑀘𑀸𑀦𑀸,_  
_𑀓𑁄𑀝𑁆𑀞𑀺𑀓𑁄 𑀓𑀧𑁆𑀧𑀺𑀦𑁂𑀦 𑀘;_  
_𑀫𑀳𑀸𑀘𑀼𑀦𑁆𑀤𑁄 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑁄,_  
_𑀭𑁂𑀯𑀢𑁄 𑀉𑀧𑀸𑀮𑀺 𑀘𑀼𑀪𑁄𑁇_  


2040. _𑀆𑀦𑀦𑁆𑀤𑁄 𑀭𑀸𑀳𑀼𑀮𑁄 𑀘𑁂𑀯,_  
_𑀕𑁄𑀢𑀫𑀻𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑁄;_  
_𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀯𑀺𑀯𑀺𑀢𑁆𑀢𑀜𑁆𑀘,_  
_𑀆𑀫𑀺𑀲𑀁 𑀲𑀫𑀓𑀫𑁆𑀧𑀺 𑀘𑁇_  


2041. _𑀦 𑀓𑁂𑀳𑀺 𑀙𑀦𑁆𑀤𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄,_  
_𑀉𑀧𑀸𑀮𑀺𑀧𑀭𑀺𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑁄;_  
_𑀅𑀦𑀸𑀦𑀼𑀯𑀚𑁆𑀚𑁄 𑀲𑀻𑀮𑁂𑀦,_  
_𑀲𑀸𑀫𑀕𑁆𑀕𑀻 𑀚𑀺𑀦𑀲𑀸𑀲𑀦𑁂𑀢𑀺𑁇_  


---

2042. 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

2043. 𑀫𑀳𑀸𑀯𑀕𑁆𑀕𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇



---

2044. 𑀫𑀳𑀸𑀯𑀕𑁆𑀕𑀧𑀸𑀍𑀇 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
