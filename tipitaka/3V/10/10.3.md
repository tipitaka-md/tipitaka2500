# 𑁧𑁦𑁇𑁩 𑀩𑀸𑀮𑀓𑀮𑁄𑀡𑀓𑀕𑀫𑀦𑀓𑀣𑀸

### 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀩𑀸𑀮𑀓𑀮𑁄𑀡𑀓𑀕𑀫𑀦𑀓𑀣𑀸

(465.)

1991\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀞𑀺𑀢𑀓𑁄𑀯 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀪𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀩𑀸𑀮𑀓𑀮𑁄𑀡𑀓𑀕𑀸𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀕𑀼 𑀩𑀸𑀮𑀓𑀮𑁄𑀡𑀓𑀕𑀸𑀫𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀪𑀕𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀆𑀲𑀦𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀲𑀺, 𑀧𑀸𑀤𑁄𑀤𑀓𑀁 𑀧𑀸𑀤𑀧𑀻𑀞𑀁 𑀧𑀸𑀤𑀓𑀣𑀮𑀺𑀓𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺, 𑀧𑀘𑁆𑀘𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺𑁇 𑀦𑀺𑀲𑀻𑀤𑀺 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂, 𑀦𑀺𑀲𑀚𑁆𑀚 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀸𑀤𑁂 𑀧𑀓𑁆𑀔𑀸𑀮𑁂𑀲𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸𑀧𑀺 𑀔𑁄 𑀪𑀕𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀪𑀕𑀼𑀁 𑀪𑀕𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀘𑁆𑀘𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀔𑀫𑀦𑀻𑀬𑀁; 𑀓𑀘𑁆𑀘𑀺 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀓𑀘𑁆𑀘𑀺 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀦 𑀓𑀺𑀮𑀫𑀲𑀻”𑀢𑀺? “𑀔𑀫𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸, 𑀬𑀸𑀧𑀦𑀻𑀬𑀁, 𑀪𑀕𑀯𑀸; 𑀦 𑀘𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀪𑀕𑀼𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀧𑀸𑀘𑀻𑀦𑀯𑀁𑀲𑀤𑀸𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇
