# 𑁧𑁦𑁇𑁫 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀕𑀫𑀦𑀓𑀣𑀸

### 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀕𑀫𑀦𑀓𑀣𑀸

(467.)

1997\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀅𑀦𑀼𑀭𑀼𑀤𑁆𑀥𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀦𑀦𑁆𑀤𑀺𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀓𑀺𑀫𑀺𑀮𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀬𑁂𑀦 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀁 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀭𑀓𑁆𑀔𑀺𑀢𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑀫𑀽𑀮𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀢𑁂𑀳𑀺 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁂𑀳𑀺𑁇 𑀲𑁄𑀫𑁆𑀳𑀺 𑀏𑀢𑀭𑀳𑀺 𑀏𑀓𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀼𑀔𑀁 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯 𑀢𑁂𑀳𑀺 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑁂𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑀡𑁆𑀟𑀦𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀓𑀮𑀳𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀯𑀺𑀯𑀸𑀤𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀪𑀲𑁆𑀲𑀓𑀸𑀭𑀓𑁂𑀳𑀺 𑀲𑀁𑀖𑁂 𑀅𑀥𑀺𑀓𑀭𑀡𑀓𑀸𑀭𑀓𑁂𑀳𑀻”𑀢𑀺𑁇

1998\. 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑀧𑀺 𑀔𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀢𑀺 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀍𑀅𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀺, 𑀙𑀺𑀦𑁆𑀦𑀕𑁆𑀕𑀸𑀦𑀺 𑀘𑁂𑀯 𑀢𑀺𑀡𑀸𑀦𑀺 𑀔𑀸𑀤𑀢𑀺, 𑀑𑀪𑀕𑁆𑀕𑁄𑀪𑀕𑁆𑀕𑀜𑁆𑀘𑀲𑁆𑀲 𑀲𑀸𑀔𑀸𑀪𑀗𑁆𑀕𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀺, 𑀆𑀯𑀺𑀮𑀸𑀦𑀺 𑀘 𑀧𑀸𑀦𑀻𑀬𑀸𑀦𑀺 𑀧𑀺𑀯𑀢𑀺, 𑀑𑀕𑀸𑀳𑀸 𑀘𑀲𑁆𑀲 𑀑𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀺𑀖𑀁𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀍𑀅𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀺, 𑀙𑀺𑀦𑁆𑀦𑀕𑁆𑀕𑀸𑀦𑀺 𑀘𑁂𑀯 𑀢𑀺𑀡𑀸𑀦𑀺 𑀔𑀸𑀤𑀸𑀫𑀺, 𑀑𑀪𑀕𑁆𑀕𑁄𑀪𑀕𑁆𑀕𑀜𑁆𑀘 𑀫𑁂 𑀲𑀸𑀔𑀸𑀪𑀗𑁆𑀕𑀁 𑀔𑀸𑀤𑀦𑁆𑀢𑀺, 𑀆𑀯𑀺𑀮𑀸𑀦𑀺 𑀘 𑀧𑀸𑀦𑀻𑀬𑀸𑀦𑀺 𑀧𑀺𑀯𑀸𑀫𑀺, 𑀑𑀕𑀸𑀳𑀸 𑀘 𑀫𑁂 𑀑𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀺𑀖𑀁𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀏𑀓𑁄𑀯 𑀕𑀡𑀲𑁆𑀫𑀸 𑀯𑀽𑀧𑀓𑀝𑁆𑀞𑁄 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑁄 𑀬𑀽𑀣𑀸 𑀅𑀧𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑀁 𑀭𑀓𑁆𑀔𑀺𑀢𑀯𑀦𑀲𑀡𑁆𑀟𑁄 𑀪𑀤𑁆𑀤𑀲𑀸𑀮𑀫𑀽𑀮𑀁 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑁄𑀡𑁆𑀟𑀸𑀬 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀦𑀻𑀬𑀁 𑀧𑀭𑀺𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀺, 𑀅𑀧𑁆𑀧𑀳𑀭𑀺𑀢𑀜𑁆𑀘 𑀓𑀭𑁄𑀢𑀺𑁇

1999\. 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀳𑀁 𑀔𑁄 𑀧𑀼𑀩𑁆𑀩𑁂 𑀆𑀓𑀺𑀡𑁆𑀡𑁄 𑀦 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀸𑀲𑀺𑀁 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀍𑀅𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀺, 𑀙𑀺𑀦𑁆𑀦𑀕𑁆𑀕𑀸𑀦𑀺 𑀘𑁂𑀯 𑀢𑀺𑀡𑀸𑀦𑀺 𑀔𑀸𑀤𑀺𑀁, 𑀑𑀪𑀕𑁆𑀕𑁄𑀪𑀕𑁆𑀕𑀜𑁆𑀘 𑀫𑁂 𑀲𑀸𑀔𑀸𑀪𑀗𑁆𑀕𑀁 𑀔𑀸𑀤𑀺𑀁𑀲𑀼, 𑀆𑀯𑀺𑀮𑀸𑀦𑀺 𑀘 𑀧𑀸𑀦𑀻𑀬𑀸𑀦𑀺 𑀅𑀧𑀸𑀬𑀺𑀁 𑀑𑀕𑀸𑀳𑀸 𑀘 𑀫𑁂 𑀑𑀢𑀺𑀡𑁆𑀡𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑀺𑀬𑁄 𑀓𑀸𑀬𑀁 𑀉𑀧𑀦𑀺𑀖𑀁𑀲𑀦𑁆𑀢𑀺𑀬𑁄 𑀅𑀕𑀫𑀁𑀲𑀼𑁇 𑀲𑁄𑀫𑁆𑀳𑀺 𑀏𑀢𑀭𑀳𑀺 𑀏𑀓𑁄 𑀅𑀤𑀼𑀢𑀺𑀬𑁄 𑀲𑀼𑀔𑀁 𑀨𑀸𑀲𑀼 𑀯𑀺𑀳𑀭𑀸𑀫𑀺 𑀅𑀜𑁆𑀜𑀢𑁆𑀭𑁂𑀯 𑀳𑀢𑁆𑀣𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀦𑀻𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀓𑀍𑀅𑀪𑁂𑀳𑀺 𑀳𑀢𑁆𑀣𑀺𑀘𑁆𑀙𑀸𑀧𑁂𑀳𑀻”𑀢𑀺𑁇

2000\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀢𑁆𑀢𑀦𑁄 𑀘 𑀧𑀯𑀺𑀯𑁂𑀓𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀲 𑀘 𑀳𑀢𑁆𑀣𑀺𑀦𑀸𑀕𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

2001\. _“𑀏𑀢𑀁 𑀦𑀸𑀕𑀲𑁆𑀲 𑀦𑀸𑀕𑁂𑀦,_  
_𑀈𑀲𑀸𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀳𑀢𑁆𑀣𑀺𑀦𑁄;_  
_𑀲𑀫𑁂𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀘𑀺𑀢𑁆𑀢𑁂𑀦,_  
_𑀬𑀤𑁂𑀓𑁄 𑀭𑀫𑀢𑀻 𑀯𑀦𑁂”𑀢𑀺𑁇_  


2002\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀸𑀮𑀺𑀮𑁂𑀬𑁆𑀬𑀓𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸—  “𑀇𑀫𑁂 𑀔𑁄 𑀅𑀬𑁆𑀬𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀼𑀦𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀦𑀢𑁆𑀣𑀲𑁆𑀲 𑀓𑀸𑀭𑀓𑀸𑁇 𑀇𑀫𑁂𑀳𑀺 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑁄 𑀪𑀕𑀯𑀸 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁 𑀅𑀬𑁆𑀬𑁂 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑁂𑀯 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀫𑁆𑀫𑀁 𑀲𑀸𑀫𑀻𑀘𑀺𑀓𑀫𑁆𑀫𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀲𑀓𑁆𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀕𑀭𑀼𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀫𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀪𑀚𑁂𑀬𑁆𑀬𑀸𑀫, 𑀦 𑀧𑀽𑀚𑁂𑀬𑁆𑀬𑀸𑀫, 𑀉𑀧𑀕𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀧𑀺𑀡𑁆𑀟𑀓𑀁 𑀦 𑀤𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀸𑀫—  𑀏𑀯𑀁 𑀇𑀫𑁂 𑀅𑀫𑁆𑀳𑁂𑀳𑀺 𑀅𑀲𑀓𑁆𑀓𑀭𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀕𑀭𑀼𑀓𑀭𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀫𑀸𑀦𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀪𑀚𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀧𑀽𑀚𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑀧𑀓𑀢𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀸 𑀉𑀧𑀸𑀲𑀓𑀸 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑁂𑀯 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀲𑀼𑀁, 𑀦 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑁂𑀲𑀼𑀁, 𑀦 𑀅𑀜𑁆𑀚𑀮𑀺𑀓𑀫𑁆𑀫𑀁 𑀲𑀸𑀫𑀻𑀘𑀺𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀁𑀲𑀼, 𑀦 𑀲𑀓𑁆𑀓𑀭𑀺𑀁𑀲𑀼, 𑀦 𑀕𑀭𑀼𑀁 𑀓𑀭𑀺𑀁𑀲𑀼, 𑀦 𑀫𑀸𑀦𑁂𑀲𑀼𑀁, 𑀦 𑀪𑀚𑁂𑀲𑀼𑀁, 𑀦 𑀧𑀽𑀚𑁂𑀲𑀼𑀁, 𑀉𑀧𑀕𑀢𑀸𑀦𑀫𑁆𑀧𑀺 𑀧𑀺𑀡𑁆𑀟𑀓𑀁 𑀦 𑀅𑀤𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑁂𑀳𑀺 𑀉𑀧𑀸𑀲𑀓𑁂𑀳𑀺 𑀅𑀲𑀓𑁆𑀓𑀭𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀕𑀭𑀼𑀓𑀭𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀫𑀸𑀦𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀪𑀚𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀧𑀽𑀚𑀺𑀬𑀫𑀸𑀦𑀸 𑀅𑀲𑀓𑁆𑀓𑀸𑀭𑀧𑀓𑀢𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀇𑀫𑀁 𑀅𑀥𑀺𑀓𑀭𑀡𑀁 𑀯𑀽𑀧𑀲𑀫𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇
