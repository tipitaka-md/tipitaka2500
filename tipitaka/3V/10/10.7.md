# 𑁧𑁦𑁇𑁭 𑀑𑀲𑀸𑀭𑀡𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀑𑀲𑀸𑀭𑀡𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(474.)

2014\. 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀯𑀺𑀦𑀬𑀜𑁆𑀘 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀆𑀧𑀢𑁆𑀢𑀺 𑀏𑀲𑀸, 𑀦𑁂𑀲𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀆𑀧𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀺, 𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺, 𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂𑀦𑀫𑁆𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀬𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀧𑀢𑁆𑀢𑀺 𑀏𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄; 𑀦𑁂𑀲𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀆𑀧𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀺, 𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺, 𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂𑀦𑀫𑁆𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦𑁇 𑀏𑀣 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑁂𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀁 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀫𑀸𑀳—  ‘𑀆𑀧𑀢𑁆𑀢𑀺 𑀏𑀲𑀸, 𑀆𑀯𑀼𑀲𑁄; 𑀦𑁂𑀲𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀆𑀧𑀦𑁆𑀦𑁄𑀫𑁆𑀳𑀺, 𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑀫𑁆𑀳𑀺, 𑀦𑀫𑁆𑀳𑀺 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂𑀦𑀫𑁆𑀳𑀺 𑀓𑀫𑁆𑀫𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦𑁇 𑀏𑀣 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀑𑀲𑀸𑀭𑁂𑀣𑀸’𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀆𑀧𑀢𑁆𑀢𑀺 𑀏𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂; 𑀦𑁂𑀲𑀸 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑁇 𑀆𑀧𑀦𑁆𑀦𑁄 𑀏𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀏𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀦𑁂𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀦𑀼𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄𑁇 𑀥𑀫𑁆𑀫𑀺𑀓𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀅𑀓𑀼𑀧𑁆𑀧𑁂𑀦 𑀞𑀸𑀦𑀸𑀭𑀳𑁂𑀦𑁇 𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘 𑀧𑀲𑁆𑀲𑀢𑀺 𑀘, 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀑𑀲𑀸𑀭𑁂𑀣𑀸”𑀢𑀺𑁇
