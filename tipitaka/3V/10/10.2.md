# 𑁧𑁦𑁇𑁨 𑀤𑀻𑀖𑀸𑀯𑀼𑀯𑀢𑁆𑀣𑀼

* 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀤𑀻𑀖𑀸𑀯𑀼𑀯𑀢𑁆𑀣𑀼

(458.)

1960\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀪𑀽𑀢𑀧𑀼𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀦𑀸𑀫 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺 𑀅𑀟𑁆𑀠𑁄 𑀫𑀳𑀤𑁆𑀥𑀦𑁄 𑀫𑀳𑀸𑀪𑁄𑀕𑁄 𑀫𑀳𑀩𑁆𑀩𑀮𑁄 𑀫𑀳𑀸𑀯𑀸𑀳𑀦𑁄 𑀫𑀳𑀸𑀯𑀺𑀚𑀺𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀓𑁄𑀲𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑁄𑁇 𑀤𑀻𑀖𑀻𑀢𑀺 𑀦𑀸𑀫 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸 𑀅𑀳𑁄𑀲𑀺 𑀤𑀮𑀺𑀤𑁆𑀤𑁄 𑀅𑀧𑁆𑀧𑀥𑀦𑁄 𑀅𑀧𑁆𑀧𑀪𑁄𑀕𑁄 𑀅𑀧𑁆𑀧𑀩𑀮𑁄 𑀅𑀧𑁆𑀧𑀯𑀸𑀳𑀦𑁄 𑀅𑀧𑁆𑀧𑀯𑀺𑀚𑀺𑀢𑁄 𑀅𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀓𑁄𑀲𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑁄𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀁 𑀲𑁂𑀦𑀁 𑀲𑀦𑁆𑀦𑀬𑁆𑀳𑀺𑀢𑁆𑀯𑀸 𑀤𑀻𑀖𑀻𑀢𑀺𑀁 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸𑀦𑀁 𑀅𑀩𑁆𑀪𑀼𑀬𑁆𑀬𑀸𑀲𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸—  ‘𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀺𑀭 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀁 𑀲𑁂𑀦𑀁 𑀲𑀦𑁆𑀦𑀬𑁆𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀫𑀁 𑀅𑀩𑁆𑀪𑀼𑀬𑁆𑀬𑀸𑀢𑁄’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀭𑀜𑁆𑀜𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀔𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀅𑀟𑁆𑀠𑁄 𑀫𑀳𑀤𑁆𑀥𑀦𑁄 𑀫𑀳𑀸𑀪𑁄𑀕𑁄 𑀫𑀳𑀩𑁆𑀩𑀮𑁄 𑀫𑀳𑀸𑀯𑀸𑀳𑀦𑁄 𑀫𑀳𑀸𑀯𑀺𑀚𑀺𑀢𑁄 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀓𑁄𑀲𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑁄, 𑀅𑀳𑀁 𑀧𑀦𑀫𑁆𑀳𑀺 𑀤𑀮𑀺𑀤𑁆𑀤𑁄 𑀅𑀧𑁆𑀧𑀥𑀦𑁄 𑀅𑀧𑁆𑀧𑀪𑁄𑀕𑁄 𑀅𑀧𑁆𑀧𑀩𑀮𑁄 𑀅𑀧𑁆𑀧𑀯𑀸𑀳𑀦𑁄 𑀅𑀧𑁆𑀧𑀯𑀺𑀚𑀺𑀢𑁄 𑀅𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀓𑁄𑀲𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑁄, 𑀦𑀸𑀳𑀁 𑀧𑀝𑀺𑀩𑀮𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁂𑀦 𑀓𑀸𑀲𑀺𑀭𑀜𑁆𑀜𑀸 𑀏𑀓𑀲𑀗𑁆𑀖𑀸𑀢𑀫𑁆𑀧𑀺 𑀲𑀳𑀺𑀢𑀼𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀦𑀕𑀭𑀫𑁆𑀳𑀸 𑀦𑀺𑀧𑁆𑀧𑀢𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇

1961\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸 𑀫𑀳𑁂𑀲𑀺𑀁 𑀆𑀤𑀸𑀬 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀦𑀕𑀭𑀫𑁆𑀳𑀸 𑀦𑀺𑀧𑁆𑀧𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀤𑀻𑀖𑀻𑀢𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀭𑀜𑁆𑀜𑁄 𑀩𑀮𑀜𑁆𑀘 𑀯𑀸𑀳𑀦𑀜𑁆𑀘 𑀚𑀦𑀧𑀤𑀜𑁆𑀘 𑀓𑁄𑀲𑀜𑁆𑀘 𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀅𑀪𑀺𑀯𑀺𑀚𑀺𑀬 𑀅𑀚𑁆𑀛𑀸𑀯𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑁄 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑁂𑀦 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀻 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂 𑀑𑀓𑀸𑀲𑁂 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀯𑁂𑀲𑁂𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀘𑁆𑀙𑀦𑁆𑀦𑁂𑀦 𑀧𑀝𑀺𑀯𑀲𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑁂𑀲𑀻 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀕𑀩𑁆𑀪𑀺𑀦𑀻 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀲𑀸 𑀏𑀯𑀭𑀽𑀧𑁄 𑀤𑁄𑀳𑀍𑀑 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺—  𑀇𑀘𑁆𑀙𑀢𑀺 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀉𑀕𑁆𑀕𑀫𑀦𑀓𑀸𑀮𑁂 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀁 𑀲𑁂𑀦𑀁 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀁 𑀯𑀫𑁆𑀫𑀺𑀓𑀁 𑀲𑀼𑀪𑀽𑀫𑁂 𑀞𑀺𑀢𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁, 𑀔𑀕𑁆𑀕𑀸𑀦𑀜𑁆𑀘 𑀥𑁄𑀯𑀦𑀁 𑀧𑀸𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀭𑀜𑁆𑀜𑁄 𑀫𑀳𑁂𑀲𑀻 𑀤𑀻𑀖𑀻𑀢𑀺𑀁 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀕𑀩𑁆𑀪𑀺𑀦𑀻𑀫𑁆𑀳𑀺, 𑀤𑁂𑀯𑁇 𑀢𑀲𑁆𑀲𑀸 𑀫𑁂 𑀏𑀯𑀭𑀽𑀧𑁄 𑀤𑁄𑀳𑀍𑀑 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑁄—  𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀲𑀽𑀭𑀺𑀬𑀲𑁆𑀲 𑀉𑀕𑁆𑀕𑀫𑀦𑀓𑀸𑀮𑁂 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀺𑀁 𑀲𑁂𑀦𑀁 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀁 𑀯𑀫𑁆𑀫𑀺𑀓𑀁 𑀲𑀼𑀪𑀽𑀫𑁂 𑀞𑀺𑀢𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀁, 𑀔𑀕𑁆𑀕𑀸𑀦𑀜𑁆𑀘 𑀥𑁄𑀯𑀦𑀁 𑀧𑀸𑀢𑀼𑀦𑁆’𑀢𑀺𑁇 ‘𑀓𑀼𑀢𑁄, 𑀤𑁂𑀯𑀺, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀤𑀼𑀕𑁆𑀕𑀢𑀸𑀦𑀁 𑀘𑀢𑀼𑀭𑀗𑁆𑀕𑀺𑀦𑀻 𑀲𑁂𑀦𑀸 𑀲𑀦𑁆𑀦𑀤𑁆𑀥𑀸 𑀯𑀫𑁆𑀫𑀺𑀓𑀸 𑀲𑀼𑀪𑀽𑀫𑁂 𑀞𑀺𑀢𑀸, 𑀔𑀕𑁆𑀕𑀸𑀦𑀜𑁆𑀘 𑀥𑁄𑀯𑀦𑀁 𑀧𑀸𑀢𑀼𑀦𑁆’𑀢𑀺 ‘𑀲𑀘𑀸𑀳𑀁, 𑀤𑁂𑀯, 𑀦 𑀮𑀪𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀫𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇

1964\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑀻𑀖𑀻𑀢𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀭𑀜𑁆𑀜𑁄 𑀓𑀧𑁆𑀧𑀓𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁂 𑀓𑀸𑀲𑀺𑀭𑀜𑁆𑀜𑁂 𑀧𑀝𑀺𑀯𑀲𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀭𑀜𑁆𑀜𑁄 𑀓𑀧𑁆𑀧𑀓𑁄 𑀤𑀻𑀖𑀻𑀢𑀺𑀁 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸𑀦𑀁 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀁 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂 𑀑𑀓𑀸𑀲𑁂 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀯𑁂𑀲𑁂𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀘𑁆𑀙𑀦𑁆𑀦𑁂𑀦 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀁 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸𑀦𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀤𑀻𑀖𑀻𑀢𑀺, 𑀤𑁂𑀯, 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑁄 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀺𑀫𑁂 𑀑𑀓𑀸𑀲𑁂 𑀓𑀼𑀫𑁆𑀪𑀓𑀸𑀭𑀦𑀺𑀯𑁂𑀲𑀦𑁂 𑀅𑀜𑁆𑀜𑀸𑀢𑀓𑀯𑁂𑀲𑁂𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀘𑁆𑀙𑀦𑁆𑀦𑁂𑀦 𑀧𑀝𑀺𑀯𑀲𑀢𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺𑀁 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸𑀦𑀁 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀁 𑀆𑀦𑁂𑀣𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀲𑀺𑀭𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀤𑀻𑀖𑀻𑀢𑀺𑀁 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸𑀦𑀁 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀁 𑀆𑀦𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  ‘𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀡𑁂, 𑀤𑀻𑀖𑀻𑀢𑀺𑀁 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸𑀦𑀁 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀁 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀓𑀸𑀬 𑀭𑀣𑀺𑀓𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑁂𑀦 𑀤𑁆𑀯𑀸𑀭𑁂𑀦 𑀦𑀺𑀓𑁆𑀔𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀢𑁄 𑀦𑀕𑀭𑀲𑁆𑀲 𑀘𑀢𑀼𑀥𑀸 𑀙𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀩𑀺𑀮𑀸𑀦𑀺 𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀣𑀸’𑀢𑀺𑁇 ‘𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸’𑀢𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀸𑀲𑀺𑀭𑀜𑁆𑀜𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁆𑀯𑀸 𑀤𑀻𑀖𑀻𑀢𑀺𑀁 𑀓𑁄𑀲𑀮𑀭𑀸𑀚𑀸𑀦𑀁 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀁 𑀤𑀍𑀳𑀸𑀬 𑀭𑀚𑁆𑀚𑀼𑀬𑀸 𑀧𑀘𑁆𑀙𑀸𑀩𑀸𑀳𑀁 𑀕𑀸𑀍𑀳𑀩𑀦𑁆𑀥𑀦𑀁 𑀩𑀦𑁆𑀥𑀺𑀢𑁆𑀯𑀸 𑀔𑀼𑀭𑀫𑀼𑀡𑁆𑀟𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀔𑀭𑀲𑁆𑀲𑀭𑁂𑀦 𑀧𑀡𑀯𑁂𑀦 𑀭𑀣𑀺𑀓𑀸𑀬 𑀭𑀣𑀺𑀓𑀁 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑁂𑀦 𑀲𑀺𑀗𑁆𑀖𑀸𑀝𑀓𑀁 𑀧𑀭𑀺𑀦𑁂𑀦𑁆𑀢𑀺𑁇

1971\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀻𑀖𑀸𑀯𑀼𑀲𑁆𑀲 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀧𑀺𑀢𑀸 𑀔𑁄 𑀫𑀁 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘 “𑀫𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀤𑀻𑀖𑀁 𑀧𑀲𑁆𑀲, 𑀫𑀸 𑀭𑀲𑁆𑀲𑀁𑁇 𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑁂𑀦 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺; 𑀅𑀯𑁂𑀭𑁂𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀦 𑀔𑁄 𑀫𑁂𑀢𑀁 𑀧𑀢𑀺𑀭𑀽𑀧𑀁, 𑀬𑁆𑀯𑀸𑀳𑀁 𑀧𑀺𑀢𑀼𑀯𑀘𑀦𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺 𑀓𑁄𑀲𑀺𑀬𑀸 𑀔𑀕𑁆𑀕𑀁 𑀧𑀯𑁂𑀲𑁂𑀲𑀺𑁇

1976\. 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸 𑀤𑀻𑀖𑀸𑀯𑀼𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀬𑀁 𑀔𑁄 𑀢𑁂, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀧𑀺𑀢𑀸 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘 “𑀫𑀸 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀤𑀻𑀖𑀁 𑀧𑀲𑁆𑀲, 𑀫𑀸 𑀭𑀲𑁆𑀲𑀁𑁇 𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑁂𑀦 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺; 𑀅𑀯𑁂𑀭𑁂𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀻”𑀢𑀺, 𑀓𑀺𑀁 𑀢𑁂 𑀧𑀺𑀢𑀸 𑀲𑀦𑁆𑀥𑀸𑀬 𑀅𑀯𑀘𑀸’𑀢𑀺? ‘𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀤𑁂𑀯, 𑀧𑀺𑀢𑀸 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘 “𑀫𑀸 𑀤𑀻𑀖𑀦𑁆”𑀢𑀺 𑀫𑀸 𑀘𑀺𑀭𑀁 𑀯𑁂𑀭𑀁 𑀅𑀓𑀸𑀲𑀻𑀢𑀺𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂, 𑀤𑁂𑀯, 𑀧𑀺𑀢𑀸 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘 “𑀫𑀸 𑀤𑀻𑀖𑀦𑁆”𑀢𑀺𑁇 𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀤𑁂𑀯, 𑀧𑀺𑀢𑀸 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘 “𑀫𑀸 𑀭𑀲𑁆𑀲𑀦𑁆”𑀢𑀺 𑀫𑀸 𑀔𑀺𑀧𑁆𑀧𑀁 𑀫𑀺𑀢𑁆𑀢𑁂𑀳𑀺 𑀪𑀺𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑀸𑀢𑀺𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂, 𑀤𑁂𑀯, 𑀧𑀺𑀢𑀸 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘 “𑀫𑀸 𑀭𑀲𑁆𑀲𑀦𑁆”𑀢𑀺𑁇 𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀤𑁂𑀯, 𑀧𑀺𑀢𑀸 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘 “𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑁂𑀦 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺, 𑀅𑀯𑁂𑀭𑁂𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀻”𑀢𑀺 𑀤𑁂𑀯𑁂𑀦 𑀫𑁂 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀳𑀢𑀸𑀢𑀺𑁇 𑀲𑀘𑀸𑀳𑀁 𑀤𑁂𑀯𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑀁, 𑀬𑁂 𑀤𑁂𑀯𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀢𑁂 𑀫𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑀼𑀁, 𑀬𑁂 𑀫𑁂 𑀅𑀢𑁆𑀣𑀓𑀸𑀫𑀸 𑀢𑁂 𑀢𑁂 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀬𑁆𑀬𑀼𑀁—  𑀏𑀯𑀁 𑀢𑀁 𑀯𑁂𑀭𑀁 𑀯𑁂𑀭𑁂𑀦 𑀦 𑀯𑀽𑀧𑀲𑀫𑁂𑀬𑁆𑀬𑁇 𑀇𑀤𑀸𑀦𑀺 𑀘 𑀧𑀦 𑀫𑁂 𑀤𑁂𑀯𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀁 𑀤𑀺𑀦𑁆𑀦𑀁, 𑀫𑀬𑀸 𑀘 𑀤𑁂𑀯𑀲𑁆𑀲 𑀚𑀻𑀯𑀺𑀢𑀁 𑀤𑀺𑀦𑁆𑀦𑀁𑁇 𑀏𑀯𑀁 𑀢𑀁 𑀯𑁂𑀭𑀁 𑀅𑀯𑁂𑀭𑁂𑀦 𑀯𑀽𑀧𑀲𑀦𑁆𑀢𑀁𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀫𑁂, 𑀤𑁂𑀯, 𑀧𑀺𑀢𑀸 𑀫𑀭𑀡𑀓𑀸𑀮𑁂 𑀅𑀯𑀘—  “𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑁂𑀦 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺; 𑀅𑀯𑁂𑀭𑁂𑀦 𑀳𑀺, 𑀢𑀸𑀢 𑀤𑀻𑀖𑀸𑀯𑀼, 𑀯𑁂𑀭𑀸 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀻”’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀳𑁆𑀫𑀤𑀢𑁆𑀢𑁄 𑀓𑀸𑀲𑀺𑀭𑀸𑀚𑀸—  ‘𑀅𑀘𑁆𑀙𑀭𑀺𑀬𑀁 𑀯𑀢 𑀪𑁄, 𑀅𑀩𑁆𑀪𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄𑁇 𑀬𑀸𑀯 𑀧𑀡𑁆𑀟𑀺𑀢𑁄 𑀅𑀬𑀁 𑀤𑀻𑀖𑀸𑀯𑀼 𑀓𑀼𑀫𑀸𑀭𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀧𑀺𑀢𑀼𑀦𑁄 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀪𑀸𑀲𑀺𑀢𑀲𑁆𑀲 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀆𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺 𑀧𑁂𑀢𑁆𑀢𑀺𑀓𑀁 𑀩𑀮𑀜𑁆𑀘 𑀯𑀸𑀳𑀦𑀜𑁆𑀘 𑀚𑀦𑀧𑀤𑀜𑁆𑀘 𑀓𑁄𑀲𑀜𑁆𑀘 𑀓𑁄𑀝𑁆𑀞𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀧𑀝𑀺𑀧𑀸𑀤𑁂𑀲𑀺, 𑀥𑀻𑀢𑀭𑀜𑁆𑀘 𑀅𑀤𑀸𑀲𑀺𑁇 𑀢𑁂𑀲𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀚𑀽𑀦𑀁 𑀆𑀤𑀺𑀦𑁆𑀦𑀤𑀡𑁆𑀟𑀸𑀦𑀁 𑀆𑀤𑀺𑀦𑁆𑀦𑀲𑀢𑁆𑀣𑀸𑀦𑀁 𑀏𑀯𑀭𑀽𑀧𑀁 𑀔𑀦𑁆𑀢𑀺𑀲𑁄𑀭𑀘𑁆𑀘𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀇𑀥 𑀔𑁄 𑀧𑀦 𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄𑀪𑁂𑀣 𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀲𑀫𑀸𑀦𑀸 𑀔𑀫𑀸 𑀘 𑀪𑀯𑁂𑀬𑁆𑀬𑀸𑀣 𑀲𑁄𑀭𑀢𑀸 𑀘𑀸”𑀢𑀺?

1977\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀮𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸 𑀪𑀡𑁆𑀟𑀦𑀁 𑀫𑀸 𑀓𑀮𑀳𑀁 𑀫𑀸 𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀫𑀸 𑀯𑀺𑀯𑀸𑀤𑀦𑁆”𑀢𑀺𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑁄 𑀅𑀥𑀫𑁆𑀫𑀯𑀸𑀤𑀻 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀆𑀕𑀫𑁂𑀢𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀸𑀫𑀻; 𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀲𑀼𑀔𑀯𑀺𑀳𑀸𑀭𑀫𑀦𑀼𑀬𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀳𑀭𑀢𑀼𑁇 𑀫𑀬𑀫𑁂𑀢𑁂𑀦 𑀪𑀡𑁆𑀟𑀦𑁂𑀦 𑀓𑀮𑀳𑁂𑀦 𑀯𑀺𑀕𑁆𑀕𑀳𑁂𑀦 𑀯𑀺𑀯𑀸𑀤𑁂𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸—  “𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑀭𑀽𑀧𑀸 𑀔𑁄 𑀇𑀫𑁂 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸, 𑀦𑀬𑀺𑀫𑁂 𑀲𑀼𑀓𑀭𑀸 𑀲𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺—  𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

1979\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺𑁇 𑀓𑁄𑀲𑀫𑁆𑀩𑀺𑀬𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸𑀪𑀢𑁆𑀢𑀁 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁄 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀲𑀁𑀲𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀞𑀺𑀢𑀓𑁄𑀯 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀅𑀪𑀸𑀲𑀺—

1980\. _“𑀧𑀼𑀣𑀼𑀲𑀤𑁆𑀤𑁄 𑀲𑀫𑀚𑀦𑁄,_  
_𑀦 𑀩𑀸𑀮𑁄 𑀓𑁄𑀘𑀺 𑀫𑀜𑁆𑀜𑀣;_  
_𑀲𑀁𑀖𑀲𑁆𑀫𑀺𑀁 𑀪𑀺𑀚𑁆𑀚𑀫𑀸𑀦𑀲𑁆𑀫𑀺𑀁,_  
_𑀦𑀸𑀜𑁆𑀜𑀁 𑀪𑀺𑀬𑁆𑀬𑁄 𑀅𑀫𑀜𑁆𑀜𑀭𑀼𑀁𑁇_  


1981\. _𑀧𑀭𑀺𑀫𑀼𑀝𑁆𑀞𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑀪𑀸𑀲𑀸,_  
_𑀯𑀸𑀘𑀸𑀕𑁄𑀘𑀭𑀪𑀸𑀡𑀺𑀦𑁄;_  
_𑀬𑀸𑀯𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀫𑀼𑀔𑀸𑀬𑀸𑀫𑀁,_  
_𑀬𑁂𑀦 𑀦𑀻𑀢𑀸 𑀦 𑀢𑀁 𑀯𑀺𑀤𑀽𑁇_  


1982\. _𑀅𑀓𑁆𑀓𑁄𑀘𑁆𑀙𑀺 𑀫𑀁 𑀅𑀯𑀥𑀺 𑀫𑀁,_  
_𑀅𑀚𑀺𑀦𑀺 𑀫𑀁 𑀅𑀳𑀸𑀲𑀺 𑀫𑁂;_  
_𑀬𑁂 𑀘 𑀢𑀁 𑀉𑀧𑀦𑀬𑁆𑀳𑀦𑁆𑀢𑀺,_  
_𑀯𑁂𑀭𑀁 𑀢𑁂𑀲𑀁 𑀦 𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


1983\. _𑀅𑀓𑁆𑀓𑁄𑀘𑁆𑀙𑀺 𑀫𑀁 𑀅𑀯𑀥𑀺 𑀫𑀁,_  
_𑀅𑀚𑀺𑀦𑀺 𑀫𑀁 𑀅𑀳𑀸𑀲𑀺 𑀫𑁂;_  
_𑀬𑁂 𑀘 𑀢𑀁 𑀦𑀼𑀧𑀦𑀬𑁆𑀳𑀦𑁆𑀢𑀺,_  
_𑀯𑁂𑀭𑀁 𑀢𑁂𑀲𑀽𑀧𑀲𑀫𑁆𑀫𑀢𑀺𑁇_  


1984\. _𑀦 𑀳𑀺 𑀯𑁂𑀭𑁂𑀦 𑀯𑁂𑀭𑀸𑀦𑀺,_  
_𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀻𑀥 𑀓𑀼𑀤𑀸𑀘𑀦𑀁;_  
_𑀅𑀯𑁂𑀭𑁂𑀦 𑀘 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺,_  
_𑀏𑀲𑀥𑀫𑁆𑀫𑁄 𑀲𑀦𑀦𑁆𑀢𑀦𑁄𑁇_  


1985\. _𑀧𑀭𑁂 𑀘 𑀦 𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀫𑀬𑀫𑁂𑀢𑁆𑀣 𑀬𑀫𑀸𑀫𑀲𑁂;_  
_𑀬𑁂 𑀘 𑀢𑀢𑁆𑀣 𑀯𑀺𑀚𑀸𑀦𑀦𑁆𑀢𑀺,_  
_𑀢𑀢𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀢𑀺 𑀫𑁂𑀥𑀕𑀸𑁇_  


1986\. _𑀅𑀝𑁆𑀞𑀺𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑀸 𑀧𑀸𑀡𑀳𑀭𑀸,_  
_𑀕𑀯𑀸𑀲𑁆𑀲𑀥𑀦𑀳𑀸𑀭𑀺𑀦𑁄;_  
_𑀭𑀝𑁆𑀞𑀁 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀢𑁂𑀲𑀫𑁆𑀧𑀺 𑀳𑁄𑀢𑀺 𑀲𑀗𑁆𑀕𑀢𑀺𑁇_  


---

1987\. 𑀓𑀲𑁆𑀫𑀸 𑀢𑀼𑀫𑁆𑀳𑀸𑀓 𑀦𑁄 𑀲𑀺𑀬𑀸𑁇



1988\. _𑀲𑀘𑁂 𑀮𑀪𑁂𑀣 𑀦𑀺𑀧𑀓𑀁 𑀲𑀳𑀸𑀬𑀁,_  
_𑀲𑀤𑁆𑀥𑀺𑀜𑁆𑀘𑀭𑀁 𑀲𑀸𑀥𑀼𑀯𑀺𑀳𑀸𑀭𑀺 𑀥𑀻𑀭𑀁;_  
_𑀅𑀪𑀺𑀪𑀼𑀬𑁆𑀬 𑀲𑀩𑁆𑀩𑀸𑀦𑀺 𑀧𑀭𑀺𑀲𑁆𑀲𑀬𑀸𑀦𑀺,_  
_𑀘𑀭𑁂𑀬𑁆𑀬 𑀢𑁂𑀦𑀢𑁆𑀢𑀫𑀦𑁄 𑀲𑀢𑀻𑀫𑀸𑁇_  


1989\. _𑀦𑁄 𑀘𑁂 𑀮𑀪𑁂𑀣 𑀦𑀺𑀧𑀓𑀁 𑀲𑀳𑀸𑀬𑀁,_  
_𑀲𑀤𑁆𑀥𑀺𑀜𑁆𑀘𑀭𑀁 𑀲𑀸𑀥𑀼𑀯𑀺𑀳𑀸𑀭𑀺 𑀥𑀻𑀭𑀁;_  
_𑀭𑀸𑀚𑀸𑀯 𑀭𑀝𑁆𑀞𑀁 𑀯𑀺𑀚𑀺𑀢𑀁 𑀧𑀳𑀸𑀬,_  
_𑀏𑀓𑁄 𑀘𑀭𑁂 𑀫𑀸𑀢𑀗𑁆𑀕𑀭𑀜𑁆𑀜𑁂𑀯 𑀦𑀸𑀕𑁄𑁇_  


1990\. _𑀏𑀓𑀲𑁆𑀲 𑀘𑀭𑀺𑀢𑀁 𑀲𑁂𑀬𑁆𑀬𑁄,_  
_𑀦𑀢𑁆𑀣𑀺 𑀩𑀸𑀮𑁂 𑀲𑀳𑀸𑀬𑀢𑀸;_  
_𑀏𑀓𑁄 𑀘𑀭𑁂 𑀦 𑀘 𑀧𑀸𑀧𑀸𑀦𑀺 𑀓𑀬𑀺𑀭𑀸,_  
_𑀅𑀧𑁆𑀧𑁄𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑁄 𑀫𑀸𑀢𑀗𑁆𑀕𑀭𑀜𑁆𑀜𑁂𑀯 𑀦𑀸𑀕𑁄”𑀢𑀺𑁇_
