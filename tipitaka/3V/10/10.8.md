# 𑁧𑁦𑁇𑁮 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑀓𑀣𑀸

### 𑀓𑁄𑀲𑀫𑁆𑀩𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑀓𑀣𑀸

(475.)

2015. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀁 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀑𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀉𑀓𑁆𑀔𑁂𑀧𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀔𑁂𑀧𑀓𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑁄 𑀏𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘 𑀧𑀲𑁆𑀲𑀺 𑀘 𑀑𑀲𑀸𑀭𑀺𑀢𑁄 𑀘𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀫𑀸”𑀢𑀺𑁇

2016. 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀉𑀓𑁆𑀔𑁂𑀧𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀸 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀢𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑀦𑀼𑀯𑀢𑁆𑀢𑀓𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  ‘𑀬𑀲𑁆𑀫𑀺𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑁄 𑀏𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘 𑀧𑀲𑁆𑀲𑀺 𑀘 𑀑𑀲𑀸𑀭𑀺𑀢𑁄 𑀘𑁇 𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀫𑀸’𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? “𑀬𑀢𑁄 𑀘 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘 𑀧𑀲𑁆𑀲𑀺 𑀘 𑀑𑀲𑀸𑀭𑀺𑀢𑁄 𑀘, 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁄 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀼𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀢𑀩𑁆𑀩𑀸𑁇 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀏𑀓𑀚𑁆𑀛𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑀩𑁆𑀩𑀁 𑀕𑀺𑀮𑀸𑀦𑁂𑀳𑀺 𑀘 𑀅𑀕𑀺𑀮𑀸𑀦𑁂𑀳𑀺 𑀘𑁇 𑀦 𑀓𑁂𑀳𑀺𑀘𑀺 𑀙𑀦𑁆𑀤𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀲𑀦𑁆𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

2017. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑁄 𑀏𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘 𑀧𑀲𑁆𑀲𑀺 𑀘 𑀑𑀲𑀸𑀭𑀺𑀢𑁄 𑀘𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

2018. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀅𑀳𑁄𑀲𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀪𑀡𑁆𑀟𑀦𑀁 𑀓𑀮𑀳𑁄 𑀯𑀺𑀕𑁆𑀕𑀳𑁄 𑀯𑀺𑀯𑀸𑀤𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄 𑀲𑀁𑀖𑀭𑀸𑀚𑀺 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁, 𑀲𑁄 𑀏𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀆𑀧𑀦𑁆𑀦𑁄 𑀘 𑀉𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑁄 𑀘 𑀧𑀲𑁆𑀲𑀺 𑀘 𑀑𑀲𑀸𑀭𑀺𑀢𑁄 𑀘𑁇 𑀲𑀁𑀖𑁄 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀁 𑀓𑀭𑁄𑀢𑀺𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀺𑀬𑀸 𑀓𑀭𑀡𑀁, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲, 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

2019. 𑀓𑀢𑀸 𑀲𑀁𑀖𑁂𑀦 𑀢𑀲𑁆𑀲 𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀲 𑀯𑀽𑀧𑀲𑀫𑀸𑀬 𑀲𑀁𑀖𑀲𑀸𑀫𑀕𑁆𑀕𑀻𑁇 𑀦𑀺𑀳𑀢𑁄 𑀲𑀁𑀖𑀪𑁂𑀤𑁄, 𑀦𑀺𑀳𑀢𑀸 𑀲𑀁𑀖𑀭𑀸𑀚𑀺, 𑀦𑀺𑀳𑀢𑀁 𑀲𑀁𑀖𑀯𑀯𑀢𑁆𑀣𑀸𑀦𑀁, 𑀦𑀺𑀳𑀢𑀁 𑀲𑀁𑀖𑀦𑀸𑀦𑀸𑀓𑀭𑀡𑀁𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’𑀢𑀺𑁇

2020. 𑀢𑀸𑀯𑀤𑁂𑀯 𑀉𑀧𑁄𑀲𑀣𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄, 𑀧𑀸𑀢𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇
