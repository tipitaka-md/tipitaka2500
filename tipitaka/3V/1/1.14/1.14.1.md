# 𑁧𑁇𑁧𑁪𑁇𑁧 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀦𑀁𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

* 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀦𑀁𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸

(63.)

239\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀫𑀸𑀕𑀥𑀺𑀓𑀸 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀪𑀕𑀯𑀢𑀺 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀧𑀼𑀢𑁆𑀢𑀓𑀢𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄, 𑀯𑁂𑀥𑀩𑁆𑀬𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄, 𑀓𑀼𑀮𑀼𑀧𑀘𑁆𑀙𑁂𑀤𑀸𑀬 𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀲𑀫𑀡𑁄 𑀕𑁄𑀢𑀫𑁄, 𑀇𑀤𑀸𑀦𑀺 𑀅𑀦𑁂𑀦 𑀚𑀝𑀺𑀮𑀲𑀳𑀲𑁆𑀲𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑀺𑀢𑀁, 𑀇𑀫𑀸𑀦𑀺 𑀘 𑀅𑀟𑁆𑀠𑀢𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀢𑀸𑀦𑀺 𑀲𑀜𑁆𑀘𑀬𑀸𑀦𑀺 𑀧𑀩𑁆𑀩𑀸𑀚𑀺𑀢𑀸𑀦𑀺𑁇 𑀇𑀫𑁂 𑀘 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸 𑀫𑀸𑀕𑀥𑀺𑀓𑀸 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀸 𑀲𑀫𑀡𑁂 𑀕𑁄𑀢𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀧𑀺𑀲𑁆𑀲𑀼 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀺𑀲𑁆𑀯𑀸 𑀇𑀫𑀸𑀬 𑀕𑀸𑀣𑀸𑀬 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺—

240\. _“𑀆𑀕𑀢𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄,_  
_𑀫𑀸𑀕𑀥𑀸𑀦𑀁 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀜𑁆𑀘𑀬𑁂 𑀦𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀁𑀲𑀼 𑀤𑀸𑀦𑀺 𑀦𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


241\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀘𑀺𑀭𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀢𑁆𑀢𑀸𑀳𑀫𑁂𑀯 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑁂𑀦 𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀇𑀫𑀸𑀬 𑀕𑀸𑀣𑀸𑀬 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺—

242\. _‘𑀆𑀕𑀢𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄,_  
_𑀫𑀸𑀕𑀥𑀸𑀦𑀁 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀜𑁆𑀘𑀬𑁂 𑀦𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀁𑀲𑀼 𑀤𑀸𑀦𑀺 𑀦𑀬𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇_  


243\. 𑀢𑁂 𑀢𑀼𑀫𑁆𑀳𑁂 𑀇𑀫𑀸𑀬 𑀕𑀸𑀣𑀸𑀬 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀣—

244\. _‘𑀦𑀬𑀦𑁆𑀢𑀺 𑀯𑁂 𑀫𑀳𑀸𑀯𑀻𑀭𑀸,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀸;_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀦𑀬𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀓𑀸 𑀉𑀲𑀽𑀬𑀸 𑀯𑀺𑀚𑀸𑀦𑀢𑀦𑁆’”𑀢𑀺𑁇_  


245\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀤𑀺𑀲𑁆𑀯𑀸 𑀇𑀫𑀸𑀬 𑀕𑀸𑀣𑀸𑀬 𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺—

246\. _“𑀆𑀕𑀢𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄,_  
_𑀫𑀸𑀕𑀥𑀸𑀦𑀁 𑀕𑀺𑀭𑀺𑀩𑁆𑀩𑀚𑀁;_  
_𑀲𑀩𑁆𑀩𑁂 𑀲𑀜𑁆𑀘𑀬𑁂 𑀦𑁂𑀢𑁆𑀯𑀸𑀦,_  
_𑀓𑀁𑀲𑀼 𑀤𑀸𑀦𑀺 𑀦𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇_  


247\. 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀇𑀫𑀸𑀬 𑀕𑀸𑀣𑀸𑀬 𑀧𑀝𑀺𑀘𑁄𑀤𑁂𑀦𑁆𑀢𑀺—

248\. _“𑀦𑀬𑀦𑁆𑀢𑀺 𑀯𑁂 𑀫𑀳𑀸𑀯𑀻𑀭𑀸,_  
_𑀲𑀤𑁆𑀥𑀫𑁆𑀫𑁂𑀦 𑀢𑀣𑀸𑀕𑀢𑀸;_  
_𑀥𑀫𑁆𑀫𑁂𑀦 𑀦𑀬𑀫𑀸𑀦𑀸𑀦𑀁,_  
_𑀓𑀸 𑀉𑀲𑀽𑀬𑀸 𑀯𑀺𑀚𑀸𑀦𑀢𑀦𑁆”𑀢𑀺𑁇_  


249\. 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀥𑀫𑁆𑀫𑁂𑀦 𑀓𑀺𑀭 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀦𑁂𑀦𑁆𑀢𑀺 𑀦𑁄 𑀅𑀥𑀫𑁆𑀫𑁂𑀦𑀸𑀢𑀺 𑀲𑀢𑁆𑀢𑀸𑀳𑀫𑁂𑀯 𑀲𑁄 𑀲𑀤𑁆𑀤𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑀭𑀥𑀸𑀬𑀺𑁇

---

250\. 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇

  
𑀘𑀢𑀼𑀢𑁆𑀣𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
