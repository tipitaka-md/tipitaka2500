# 𑁧𑁇𑁩𑁯 𑀆𑀳𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀆𑀳𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀯𑀢𑁆𑀣𑀼

(102.)

443. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺, 𑀢𑀢𑁆𑀣 𑀳𑁂𑀫𑀦𑁆𑀢𑀁, 𑀢𑀢𑁆𑀣 𑀕𑀺𑀫𑁆𑀳𑀁𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀆𑀳𑀼𑀦𑁆𑀤𑀭𑀺𑀓𑀸 𑀲𑀫𑀡𑀸𑀦𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑀦𑀁 𑀤𑀺𑀲𑀸 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑀸, 𑀦 𑀇𑀫𑁂𑀲𑀁 𑀤𑀺𑀲𑀸 𑀧𑀓𑁆𑀔𑀸𑀬𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙𑀸𑀦𑀦𑁆𑀤, 𑀅𑀯𑀸𑀧𑀼𑀭𑀡𑀁 𑀆𑀤𑀸𑀬 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑁂𑀡𑀺𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀳𑀺—  ‘𑀇𑀘𑁆𑀙𑀢𑀸𑀯𑀼𑀲𑁄 𑀪𑀕𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀕𑀺𑀭𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀢𑁆𑀣𑁄, 𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑀢𑀽’”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀅𑀯𑀸𑀧𑀼𑀭𑀡𑀁 𑀆𑀤𑀸𑀬 𑀅𑀦𑀼𑀧𑀭𑀺𑀯𑁂𑀡𑀺𑀬𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀇𑀘𑁆𑀙𑀢𑀸𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀸 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀕𑀺𑀭𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀁𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀅𑀢𑁆𑀣𑁄, 𑀲𑁄 𑀆𑀕𑀘𑁆𑀙𑀢𑀽”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀪𑀕𑀯𑀢𑀸, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁 𑀤𑀲𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀢𑁆𑀣𑀼𑀁, 𑀤𑀲𑀯𑀲𑁆𑀲𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑀸𑀢𑀼𑀁𑁇 𑀢𑀢𑁆𑀣 𑀘 𑀦𑁄 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀘 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀇𑀢𑁆𑀢𑀭𑁄 𑀘 𑀯𑀸𑀲𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀼𑀦 𑀘 𑀧𑀘𑁆𑀘𑀸𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺, 𑀧𑀼𑀦 𑀘 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀕𑀳𑁂𑀢𑀩𑁆𑀩𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀘𑁂 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀸 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫; 𑀦𑁄 𑀘𑁂 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀆𑀘𑀭𑀺𑀬𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀸 𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀦 𑀕𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑁇 𑀮𑀳𑀼𑀘𑀺𑀢𑁆𑀢𑀓𑀢𑀸 𑀦𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀆𑀦𑀦𑁆𑀤, 𑀧𑀜𑁆𑀜𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀑𑀕𑀡𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀤𑀓𑁆𑀔𑀺𑀡𑀸𑀕𑀺𑀭𑀺𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇
