# 𑁧𑁇𑁩𑁫 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼𑀯𑀢𑁆𑀣𑀼

(98.)

436\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀡𑁆𑀟𑀺𑀢𑁆𑀯𑀸 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀢𑀁 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼𑀁 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀸 𑀆𑀭𑀸𑀫𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼—  “𑀅𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀤𑀸𑀭𑀓𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺? 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀚𑀸𑀦𑀁𑀬𑁂𑀯 𑀆𑀳𑀁𑀲𑀼—  “𑀦 𑀚𑀸𑀦𑀸𑀫𑀸”𑀢𑀺, 𑀅𑀧𑀲𑁆𑀲𑀁𑀬𑁂𑀯 𑀆𑀳𑀁𑀲𑀼—  “𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀢𑀁 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼𑀁 𑀯𑀺𑀘𑀺𑀦𑀦𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀁 𑀤𑀺𑀲𑁆𑀯𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸, 𑀤𑀼𑀲𑁆𑀲𑀻𑀮𑀸 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀺𑀦𑁄𑁇 𑀚𑀸𑀦𑀁𑀬𑁂𑀯 𑀆𑀳𑀁𑀲𑀼—  ‘𑀦 𑀚𑀸𑀦𑀸𑀫𑀸’𑀢𑀺, 𑀧𑀲𑁆𑀲𑀁𑀬𑁂𑀯 𑀆𑀳𑀁𑀲𑀼—  ‘𑀦 𑀧𑀲𑁆𑀲𑀸𑀫𑀸’𑀢𑀺𑁇 𑀅𑀬𑀁 𑀤𑀸𑀭𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀸𑀭𑀪𑀡𑁆𑀟𑀼𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀁 𑀅𑀧𑀮𑁄𑀓𑁂𑀢𑀼𑀁 𑀪𑀡𑁆𑀟𑀼𑀓𑀫𑁆𑀫𑀸𑀬𑀸”𑀢𑀺𑁇
