# 𑁧𑁇𑁫𑁮 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀢𑁆𑀣𑀼

(120.)

504\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

505\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀲𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀧𑀺 𑀦𑀘𑀺𑀭𑀲𑁆𑀲𑁂𑀯 𑀅𑀮𑀚𑁆𑀚𑀺𑀦𑁄 𑀳𑁄𑀦𑁆𑀢𑀺 𑀧𑀸𑀧𑀓𑀸𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀯𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

506\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄, 𑀦 𑀅𑀮𑀚𑁆𑀚𑀻𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀸𑀫 𑀮𑀚𑁆𑀚𑀺𑀁 𑀯𑀸 𑀅𑀮𑀚𑁆𑀚𑀺𑀁 𑀯𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀢𑀽𑀳𑀧𑀜𑁆𑀘𑀸𑀳𑀁 𑀆𑀕𑀫𑁂𑀢𑀼𑀁 𑀬𑀸𑀯 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀪𑀸𑀕𑀢𑀁 𑀚𑀸𑀦𑀸𑀫𑀻”𑀢𑀺𑁇
