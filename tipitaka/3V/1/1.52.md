# 𑁧𑁇𑁫𑁨 𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀯𑀢𑁆𑀣𑀼

(114.)

488\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀓𑁂𑀢𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀘𑁄𑀭𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀘𑁆𑀙𑀺𑀦𑁆𑀤𑀺𑀁𑀲𑀼, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑀦𑀺𑀁𑀲𑀼𑁇 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀭𑀸𑀚𑀪𑀝𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘𑁄𑀭𑁂 𑀅𑀕𑁆𑀕𑀳𑁂𑀲𑀼𑀁, 𑀏𑀓𑀘𑁆𑀘𑁂 𑀘𑁄𑀭𑀸 𑀧𑀮𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀬𑁂 𑀢𑁂 𑀧𑀮𑀸𑀬𑀺𑀁𑀲𑀼 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀁𑀲𑀼, 𑀬𑁂 𑀢𑁂 𑀕𑀳𑀺𑀢𑀸 𑀢𑁂 𑀯𑀥𑀸𑀬 𑀑𑀦𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀔𑁄 𑀢𑁂 𑀧𑀮𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀢𑁂 𑀘𑁄𑀭𑁂 𑀯𑀥𑀸𑀬 𑀑𑀦𑀺𑀬𑁆𑀬𑀫𑀸𑀦𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀲𑀸𑀥𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀧𑀮𑀸𑀬𑀺𑀫𑁆𑀳𑀸, 𑀲𑀘𑀸 𑀘 𑀫𑀬𑀁 𑀕𑀬𑁆𑀳𑁂𑀬𑁆𑀬𑀸𑀫, 𑀫𑀬𑀫𑁆𑀧𑀺 𑀏𑀯𑀫𑁂𑀯 𑀳𑀜𑁆𑀜𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀁 𑀧𑀦 𑀢𑀼𑀫𑁆𑀳𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀓𑀢𑁆𑀣𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽𑁇 𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄, 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 (𑁭)
