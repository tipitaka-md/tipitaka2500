# 𑁧𑁇𑁬𑁨 𑀕𑀩𑁆𑀪𑀯𑀻𑀲𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀕𑀩𑁆𑀪𑀯𑀻𑀲𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(124.)

514\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀸 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑁄 𑀕𑀩𑁆𑀪𑀯𑀻𑀲𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀓𑀼𑀫𑀸𑀭𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀕𑀩𑁆𑀪𑀯𑀻𑀲𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀦𑀼 𑀔𑁄𑀫𑁆𑀳𑀺, 𑀦𑀦𑀼 𑀔𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀬𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀓𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀫𑀺𑀁 𑀧𑀞𑀫𑀁 𑀘𑀺𑀢𑁆𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀧𑀞𑀫𑀁 𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑀁 𑀧𑀸𑀢𑀼𑀪𑀽𑀢𑀁, 𑀢𑀤𑀼𑀧𑀸𑀤𑀸𑀬 𑀲𑀸𑀯𑀲𑁆𑀲 𑀚𑀸𑀢𑀺𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀩𑁆𑀪𑀯𑀻𑀲𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
