# 𑁧𑁇𑁧𑁪 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

(60.)

220\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀜𑁆𑀘𑀬𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀝𑀺𑀯𑀲𑀢𑀺 𑀫𑀳𑀢𑀺𑀬𑀸 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀧𑀭𑀺𑀲𑀸𑀬 𑀲𑀤𑁆𑀥𑀺𑀁 𑀅𑀟𑁆𑀠𑀢𑁂𑀬𑁆𑀬𑁂𑀳𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀢𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸 𑀲𑀜𑁆𑀘𑀬𑁂 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀢𑁂𑀳𑀺 𑀓𑀢𑀺𑀓𑀸 𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  “𑀬𑁄 𑀧𑀞𑀫𑀁 𑀅𑀫𑀢𑀁 𑀅𑀥𑀺𑀕𑀘𑁆𑀙𑀢𑀺, 𑀲𑁄 𑀇𑀢𑀭𑀲𑁆𑀲 𑀆𑀭𑁄𑀘𑁂𑀢𑀽”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

221\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀚𑀺 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀧𑀸𑀯𑀺𑀲𑀺 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂𑀦 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀆𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂𑀦 𑀧𑀲𑀸𑀭𑀺𑀢𑁂𑀦, 𑀑𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀓𑁆𑀔𑀼 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀁 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂𑀦 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀆𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂𑀦 𑀧𑀲𑀸𑀭𑀺𑀢𑁂𑀦 𑀑𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀓𑁆𑀔𑀼𑀁 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑁂 𑀯𑀢 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸, 𑀅𑀬𑀁 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁—  ‘𑀓𑀁𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀓𑁄 𑀯𑀸 𑀢𑁂 𑀲𑀢𑁆𑀣𑀸, 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀲𑀻’”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀅𑀓𑀸𑀮𑁄 𑀔𑁄 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁, 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑀁, 𑀅𑀢𑁆𑀣𑀺𑀓𑁂𑀳𑀺 𑀉𑀧𑀜𑁆𑀜𑀸𑀢𑀁 𑀫𑀕𑁆𑀕𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀚𑀺 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀆𑀤𑀸𑀬 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄𑀧𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀚𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀅𑀲𑁆𑀲𑀚𑀺𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀓𑀁𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀓𑁄 𑀯𑀸 𑀢𑁂 𑀲𑀢𑁆𑀣𑀸, 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀲𑀻”𑀢𑀺? “𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀢𑀸𑀳𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀲𑁄 𑀘 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸, 𑀢𑀲𑁆𑀲 𑀘𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀫𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁𑀯𑀸𑀤𑀻 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀢𑁆𑀣𑀸, 𑀓𑀺𑀫𑀓𑁆𑀔𑀸𑀬𑀻”𑀢𑀺? “𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀯𑁄 𑀅𑀘𑀺𑀭𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀅𑀥𑀼𑀦𑀸𑀕𑀢𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑀁, 𑀦 𑀢𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼𑀁, 𑀅𑀧𑀺 𑀘 𑀢𑁂 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀓𑁆𑀔𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀳𑁄𑀢𑀼, 𑀆𑀯𑀼𑀲𑁄—

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

222\. _𑀅𑀧𑁆𑀧𑀁 𑀯𑀸 𑀩𑀳𑀼𑀁 𑀯𑀸 𑀪𑀸𑀲𑀲𑁆𑀲𑀼,_  
_𑀅𑀢𑁆𑀣𑀁𑀬𑁂𑀯 𑀫𑁂 𑀩𑁆𑀭𑀽𑀳𑀺;_  
_𑀅𑀢𑁆𑀣𑁂𑀦𑁂𑀯 𑀫𑁂 𑀅𑀢𑁆𑀣𑁄,_  
_𑀓𑀺𑀁 𑀓𑀸𑀳𑀲𑀺 𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀩𑀳𑀼𑀦𑁆”𑀢𑀺𑁇_  


* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

223\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀅𑀲𑁆𑀲𑀚𑀺 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀅𑀪𑀸𑀲𑀺—

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

224\. _“𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀼𑀧𑁆𑀧𑀪𑀯𑀸,_

227\. 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀬𑁂𑀦 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀓𑀘𑁆𑀘𑀺 𑀦𑁄 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀫𑀢𑀁 𑀅𑀥𑀺𑀕𑀢𑁄”𑀢𑀺? “𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀅𑀫𑀢𑀁 𑀅𑀥𑀺𑀕𑀢𑁄”𑀢𑀺𑁇 “𑀬𑀣𑀸𑀓𑀣𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀫𑀢𑀁 𑀅𑀥𑀺𑀕𑀢𑁄”𑀢𑀺? “𑀇𑀥𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀤𑁆𑀤𑀲𑀁 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀁 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁂𑀦 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀧𑀝𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑁂𑀦 𑀆𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀯𑀺𑀮𑁄𑀓𑀺𑀢𑁂𑀦 𑀲𑀫𑀺𑀜𑁆𑀚𑀺𑀢𑁂𑀦 𑀧𑀲𑀸𑀭𑀺𑀢𑁂𑀦 𑀑𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀘𑀓𑁆𑀔𑀼𑀁 𑀇𑀭𑀺𑀬𑀸𑀧𑀣𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀫𑁂 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀬𑁂 𑀯𑀢 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀯𑀸 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸, 𑀅𑀬𑀁 𑀢𑁂𑀲𑀁 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀁—  𑀓𑀁𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀓𑁄 𑀯𑀸 𑀢𑁂 𑀲𑀢𑁆𑀣𑀸, 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀲𑀻’𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀓𑀸𑀮𑁄 𑀔𑁄 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁 𑀅𑀦𑁆𑀢𑀭𑀖𑀭𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀢𑀺, 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑁂𑀬𑁆𑀬𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑁂𑀳𑀺 𑀉𑀧𑀜𑁆𑀜𑀸𑀢𑀁 𑀫𑀕𑁆𑀕𑀦𑁆’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀲𑁆𑀲𑀚𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀆𑀤𑀸𑀬 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑁂𑀦 𑀅𑀲𑁆𑀲𑀚𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀚𑀺𑀦𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺𑀁𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀯𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀺 𑀔𑁄 𑀢𑁂, 𑀆𑀯𑀼𑀲𑁄, 𑀇𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀸𑀦𑀺, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀙𑀯𑀺𑀯𑀡𑁆𑀡𑁄 𑀧𑀭𑀺𑀬𑁄𑀤𑀸𑀢𑁄𑁇 𑀓𑀁𑀲𑀺 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀓𑁄 𑀯𑀸 𑀢𑁂 𑀲𑀢𑁆𑀣𑀸, 𑀓𑀲𑁆𑀲 𑀯𑀸 𑀢𑁆𑀯𑀁 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀲𑀻’𑀢𑀺? ‘𑀅𑀢𑁆𑀣𑀸𑀯𑀼𑀲𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀢𑀸𑀳𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄, 𑀲𑁄 𑀘 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸, 𑀢𑀲𑁆𑀲 𑀘𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀭𑁄𑀘𑁂𑀫𑀻’𑀢𑀺𑁇 ‘𑀓𑀺𑀁𑀯𑀸𑀤𑀻 𑀧𑀦𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀢𑁆𑀣𑀸 𑀓𑀺𑀫𑀓𑁆𑀔𑀸𑀬𑀻’𑀢𑀺𑁇 ‘𑀅𑀳𑀁 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀦𑀯𑁄 𑀅𑀘𑀺𑀭𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀥𑀼𑀦𑀸𑀕𑀢𑁄 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑀁, 𑀦 𑀢𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑁂𑀦 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀼𑀁, 𑀅𑀧𑀺 𑀘 𑀢𑁂 𑀲𑀁𑀔𑀺𑀢𑁆𑀢𑁂𑀦 𑀅𑀢𑁆𑀣𑀁 𑀯𑀓𑁆𑀔𑀸𑀫𑀻’𑀢𑀺𑁇 𑀅𑀣 𑀔𑁆𑀯𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀲𑁆𑀲𑀚𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀁—  ‘𑀳𑁄𑀢𑀼, 𑀆𑀯𑀼𑀲𑁄,

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

229\. 𑀅𑀣 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀲𑁆𑀲𑀚𑀺 𑀪𑀺𑀓𑁆𑀔𑀼 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀅𑀪𑀸𑀲𑀺—

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

230\. _‘𑀬𑁂 𑀥𑀫𑁆𑀫𑀸 𑀳𑁂𑀢𑀼𑀧𑁆𑀧𑀪𑀯𑀸,_  
_𑀢𑁂𑀲𑀁 𑀳𑁂𑀢𑀼𑀁 𑀢𑀣𑀸𑀕𑀢𑁄 𑀆𑀳;_  
_𑀢𑁂𑀲𑀜𑁆𑀘 𑀬𑁄 𑀦𑀺𑀭𑁄𑀥𑁄,_  
_𑀏𑀯𑀁𑀯𑀸𑀤𑀻 𑀫𑀳𑀸𑀲𑀫𑀡𑁄’”𑀢𑀺𑁇_  


* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

231\. 𑀅𑀣 𑀔𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀲𑁆𑀲 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀇𑀫𑀁 𑀥𑀫𑁆𑀫𑀧𑀭𑀺𑀬𑀸𑀬𑀁 𑀲𑀼𑀢𑁆𑀯𑀸 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

232\. _𑀏𑀲𑁂𑀯 𑀥𑀫𑁆𑀫𑁄 𑀬𑀤𑀺 𑀢𑀸𑀯𑀤𑁂𑀯,_  
_𑀧𑀘𑁆𑀘𑀩𑁆𑀬𑀢𑁆𑀣 𑀧𑀤𑀫𑀲𑁄𑀓𑀁;_  
_𑀅𑀤𑀺𑀝𑁆𑀞𑀁 𑀅𑀩𑁆𑀪𑀢𑀻𑀢𑀁,_  
_𑀩𑀳𑀼𑀓𑁂𑀳𑀺 𑀓𑀧𑁆𑀧𑀦𑀳𑀼𑀢𑁂𑀳𑀻”𑀢𑀺𑁇_  


* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

(62.)

233\. 𑀅𑀣 𑀔𑁄 𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀕𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂, 𑀲𑁄 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 “𑀇𑀫𑀸𑀦𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀅𑀟𑁆𑀠𑀢𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀢𑀸𑀦𑀺 𑀅𑀫𑁆𑀳𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀅𑀫𑁆𑀳𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀇𑀥 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺, 𑀢𑁂𑀧𑀺 𑀢𑀸𑀯 𑀅𑀧𑀮𑁄𑀓𑁂𑀫𑁇 𑀬𑀣𑀸 𑀢𑁂 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀢𑀣𑀸 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸 𑀬𑁂𑀦 𑀢𑁂 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁂 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀕𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂, 𑀲𑁄 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 “𑀫𑀬𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀲𑀫𑁆𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀸 𑀇𑀥 𑀯𑀺𑀳𑀭𑀸𑀫, 𑀲𑀘𑁂 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀫𑀬𑀁 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

234\. 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸 𑀬𑁂𑀦 𑀲𑀜𑁆𑀘𑀬𑁄 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑀜𑁆𑀘𑀬𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀕𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂, 𑀲𑁄 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸 𑀅𑀕𑀫𑀺𑀢𑁆𑀣, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑀬𑁄 𑀇𑀫𑀁 𑀕𑀡𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸 𑀲𑀜𑁆𑀘𑀬𑀁 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀕𑀘𑁆𑀙𑀸𑀫 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂, 𑀲𑁄 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀫𑀸 𑀅𑀕𑀫𑀺𑀢𑁆𑀣, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀢𑀬𑁄 𑀇𑀫𑀁 𑀕𑀡𑀁 𑀧𑀭𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸 𑀢𑀸𑀦𑀺 𑀅𑀟𑁆𑀠𑀢𑁂𑀬𑁆𑀬𑀸𑀦𑀺 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑀢𑀸𑀦𑀺 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀲𑀜𑁆𑀘𑀬𑀲𑁆𑀲 𑀧𑀦 𑀧𑀭𑀺𑀩𑁆𑀩𑀸𑀚𑀓𑀲𑁆𑀲 𑀢𑀢𑁆𑀣𑁂𑀯 𑀉𑀡𑁆𑀳𑀁 𑀮𑁄𑀳𑀺𑀢𑀁 𑀫𑀼𑀔𑀢𑁄 𑀉𑀕𑁆𑀕𑀜𑁆𑀙𑀺𑁇

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

235\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑁂 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀏𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀲𑀳𑀸𑀬𑀓𑀸 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺, 𑀓𑁄𑀮𑀺𑀢𑁄 𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀘𑁇 𑀏𑀢𑀁 𑀫𑁂 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀅𑀕𑁆𑀕𑀁 𑀪𑀤𑁆𑀤𑀬𑀼𑀕𑀦𑁆”𑀢𑀺𑁇

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

236\. _𑀕𑀫𑁆𑀪𑀻𑀭𑁂 𑀜𑀸𑀡𑀯𑀺𑀲𑀬𑁂,_  
_𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁂 𑀉𑀧𑀥𑀺𑀲𑀗𑁆𑀔𑀬𑁂;_  
_𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑁂 𑀅𑀧𑁆𑀧𑀢𑁆𑀢𑁂 𑀯𑁂𑀍𑀉𑀯𑀦𑀁,_  
_𑀅𑀣 𑀦𑁂 𑀲𑀢𑁆𑀣𑀸 𑀩𑁆𑀬𑀸𑀓𑀸𑀲𑀺𑁇_  


* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

237\. _“𑀏𑀢𑁂 𑀤𑁆𑀯𑁂 𑀲𑀳𑀸𑀬𑀓𑀸,_  
_𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀓𑁄𑀮𑀺𑀢𑁄 𑀉𑀧𑀢𑀺𑀲𑁆𑀲𑁄 𑀘;_  
_𑀏𑀢𑀁 𑀫𑁂 𑀲𑀸𑀯𑀓𑀬𑀼𑀕𑀁,_

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

238\. 𑀅𑀣 𑀔𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀫𑁄𑀕𑁆𑀕𑀮𑁆𑀮𑀸𑀦𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇
