# 𑁧𑁇𑁫𑁯 𑀕𑀫𑀺𑀓𑀸𑀤𑀺𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀕𑀫𑀺𑀓𑀸𑀤𑀺𑀦𑀺𑀲𑁆𑀲𑀬𑀯𑀢𑁆𑀣𑀼

(121.)

507. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀦𑀺𑀲𑁆𑀲𑀬𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁄, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁂𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑁆”𑀢𑀺𑁇

508. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀤𑁆𑀯𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀆𑀯𑀸𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀺𑀁𑀲𑀼𑁇 𑀢𑀢𑁆𑀣 𑀏𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀕𑀺𑀮𑀸𑀦𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀦𑀺𑀲𑁆𑀲𑀬𑀓𑀭𑀡𑀻𑀬𑁄 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁂𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑁆”𑀢𑀺𑁇

509. 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀦𑀺𑀲𑁆𑀲𑀬𑀓𑀭𑀡𑀻𑀬𑁄, 𑀅𑀬𑀜𑁆𑀘 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁂𑀦 𑀬𑀸𑀘𑀺𑀬𑀫𑀸𑀦𑁂𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀦𑁆”𑀢𑀺𑁇

510. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀘 𑀢𑀲𑁆𑀫𑀺𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑁂 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀘𑀫𑁆𑀳𑀺 𑀦𑀺𑀲𑁆𑀲𑀬𑀓𑀭𑀡𑀻𑀬𑁄 𑀅𑀭𑀜𑁆𑀜𑁂 𑀯𑀺𑀳𑀭𑀸𑀫𑀺, 𑀫𑀬𑁆𑀳𑀜𑁆𑀘 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑁂 𑀨𑀸𑀲𑀼 𑀳𑁄𑀢𑀺, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑀜𑁆𑀜𑀺𑀓𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀨𑀸𑀲𑀼𑀯𑀺𑀳𑀸𑀭𑀁 𑀲𑀮𑁆𑀮𑀓𑁆𑀔𑁂𑀦𑁆𑀢𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁂𑀦 𑀅𑀦𑀺𑀲𑁆𑀲𑀺𑀢𑁂𑀦 𑀯𑀢𑁆𑀣𑀼𑀁—  𑀬𑀤𑀸 𑀧𑀢𑀺𑀭𑀽𑀧𑁄 𑀦𑀺𑀲𑁆𑀲𑀬𑀤𑀸𑀬𑀓𑁄 𑀆𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺, 𑀢𑀤𑀸 𑀢𑀲𑁆𑀲 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀯𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇
