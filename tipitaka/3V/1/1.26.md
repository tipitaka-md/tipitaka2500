# 𑁧𑁇𑁨𑁬 𑀧𑀜𑁆𑀘𑀸𑀩𑀸𑀥𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀜𑁆𑀘𑀸𑀩𑀸𑀥𑀯𑀢𑁆𑀣𑀼

(88.)

422. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀕𑀥𑁂𑀲𑀼 𑀧𑀜𑁆𑀘 𑀆𑀩𑀸𑀥𑀸 𑀉𑀲𑁆𑀲𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀓𑀼𑀝𑁆𑀞𑀁, 𑀕𑀡𑁆𑀟𑁄, 𑀓𑀺𑀮𑀸𑀲𑁄, 𑀲𑁄𑀲𑁄, 𑀅𑀧𑀫𑀸𑀭𑁄𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀩𑀸𑀥𑁂𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑀸 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀲𑀸𑀥𑀼 𑀦𑁄, 𑀆𑀘𑀭𑀺𑀬, 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀳𑀁 𑀔𑁆𑀯𑀬𑁆𑀬𑁄, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄; 𑀭𑀸𑀚𑀸 𑀘 𑀫𑁂 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄 𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄; 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀩𑁆𑀩𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀢𑁂, 𑀆𑀘𑀭𑀺𑀬, 𑀳𑁄𑀢𑀼; 𑀫𑀬𑀜𑁆𑀘 𑀢𑁂 𑀤𑀸𑀲𑀸; 𑀲𑀸𑀥𑀼, 𑀦𑁄, 𑀆𑀘𑀭𑀺𑀬, 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀳𑀁 𑀔𑁆𑀯𑀬𑁆𑀬𑁄, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄 𑀭𑀸𑀚𑀸 𑀘 𑀫𑁂 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄 𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄; 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑁂 𑀔𑁄 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀻𑀮𑀸 𑀲𑀼𑀔𑀲𑀫𑀸𑀘𑀸𑀭𑀸, 𑀲𑀼𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀢𑁂𑀲𑀼 𑀲𑀬𑀦𑁂𑀲𑀼 𑀲𑀬𑀦𑁆𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦 𑀫𑀬𑀁 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀸𑀫𑁇 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀚𑀻𑀯𑀓𑁄 𑀘 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀼𑀁, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑀲𑀼 𑀚𑀻𑀯𑀓𑁄 𑀘 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀩𑀳𑀽 𑀕𑀺𑀮𑀸𑀦𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀝𑁆𑀞𑀳𑀦𑁆𑀢𑀸 𑀬𑀸𑀘𑀦𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀜𑁆𑀜𑀢𑁆𑀢𑀺𑀩𑀳𑀼𑀮𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺—  “𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀣, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀣, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀚𑀻𑀯𑀓𑁄𑀧𑀺 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀩𑀳𑀽 𑀕𑀺𑀮𑀸𑀦𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀭𑀸𑀚𑀓𑀺𑀘𑁆𑀘𑀁 𑀧𑀭𑀺𑀳𑀸𑀧𑁂𑀲𑀺𑁇

(89.)

423. 𑀅𑀜𑁆𑀜𑀢𑀭𑁄𑀧𑀺 𑀧𑀼𑀭𑀺𑀲𑁄 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀩𑀸𑀥𑁂𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑁄 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼 𑀫𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀳𑀁 𑀔𑁆𑀯𑀬𑁆𑀬𑁄, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄, 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄, 𑀭𑀸𑀚𑀸 𑀘 𑀫𑁂 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄 𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄; 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 “𑀲𑀩𑁆𑀩𑀁 𑀲𑀸𑀧𑀢𑁂𑀬𑁆𑀬𑀜𑁆𑀘 𑀢𑁂, 𑀆𑀘𑀭𑀺𑀬, 𑀳𑁄𑀢𑀼, 𑀅𑀳𑀜𑁆𑀘 𑀢𑁂 𑀤𑀸𑀲𑁄; 𑀲𑀸𑀥𑀼 𑀫𑀁, 𑀆𑀘𑀭𑀺𑀬, 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀸𑀳𑀻”𑀢𑀺𑁇 “𑀅𑀳𑀁 𑀔𑁆𑀯𑀬𑁆𑀬𑁄, 𑀩𑀳𑀼𑀓𑀺𑀘𑁆𑀘𑁄 𑀩𑀳𑀼𑀓𑀭𑀡𑀻𑀬𑁄, 𑀭𑀸𑀚𑀸 𑀘 𑀫𑁂 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄 𑀇𑀢𑁆𑀣𑀸𑀕𑀸𑀭𑀜𑁆𑀘 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑁄 𑀘 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁄, 𑀦𑀸𑀳𑀁 𑀲𑀓𑁆𑀓𑁄𑀫𑀺 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

424. 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑁂 𑀔𑁄 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀻𑀮𑀸 𑀲𑀼𑀔𑀲𑀫𑀸𑀘𑀸𑀭𑀸, 𑀲𑀼𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀢𑁂𑀲𑀼 𑀲𑀬𑀦𑁂𑀲𑀼 𑀲𑀬𑀦𑁆𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀁𑁇 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀚𑀻𑀯𑀓𑁄 𑀘 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑁄𑀫𑁆𑀳𑀺 𑀅𑀭𑁄𑀕𑁄 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀼𑀁, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁂𑀯 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀁𑀲𑀼, 𑀚𑀻𑀯𑀓𑁄 𑀘 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀺𑀓𑀺𑀘𑁆𑀙𑀺𑁇 𑀲𑁄 𑀅𑀭𑁄𑀕𑁄 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑁇

425. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀯𑀺𑀩𑁆𑀪𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀢𑀁 𑀧𑀼𑀭𑀺𑀲𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁄, 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺? “𑀏𑀯𑀁, 𑀆𑀘𑀭𑀺𑀬𑀸”𑀢𑀺𑁇 “𑀓𑀺𑀲𑁆𑀲 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬𑁄, 𑀏𑀯𑀭𑀽𑀧𑀫𑀓𑀸𑀲𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀚𑀻𑀯𑀓𑀲𑁆𑀲 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀲𑁆𑀲 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀤𑀦𑁆𑀢𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀩𑀸𑀥𑁂𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀩𑀸𑀥𑁂𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑀁 𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀚𑀻𑀯𑀓𑀁 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀚𑀻𑀯𑀓𑁄 𑀓𑁄𑀫𑀸𑀭𑀪𑀘𑁆𑀘𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀺 𑀆𑀩𑀸𑀥𑁂𑀳𑀺 𑀨𑀼𑀝𑁆𑀞𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
