# 𑁧𑁇𑁪𑁩 𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑀯𑀢𑁆𑀣𑀼

(107.)

474\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀕𑀸𑀭𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀲𑀪𑀸𑀕𑀯𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀅𑀕𑀸𑀭𑀯𑀸 𑀅𑀧𑁆𑀧𑀢𑀺𑀲𑁆𑀲𑀸 𑀅𑀲𑀪𑀸𑀕𑀯𑀼𑀢𑁆𑀢𑀺𑀓𑀸 𑀯𑀺𑀳𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀮𑀸𑀪𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀢𑁆𑀣𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀯𑀸𑀲𑀸𑀬 𑀧𑀭𑀺𑀲𑀓𑁆𑀓𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀓𑁆𑀓𑁄𑀲𑀢𑀺 𑀧𑀭𑀺𑀪𑀸𑀲𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀪𑁂𑀤𑁂𑀢𑀺—  𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀳𑀺 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀲𑁆𑀲 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

475\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀯𑀭𑀡𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

476\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀫𑀡𑁂𑀭𑀸𑀦𑀁 𑀲𑀩𑁆𑀩𑀁 𑀲𑀁𑀖𑀸𑀭𑀸𑀫𑀁 𑀆𑀯𑀭𑀡𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀆𑀭𑀸𑀫𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑀼𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑀧𑀺, 𑀯𑀺𑀩𑁆𑀪𑀫𑀦𑁆𑀢𑀺𑀧𑀺, 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀲𑀼𑀧𑀺 𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀩𑁆𑀩𑁄 𑀲𑀁𑀖𑀸𑀭𑀸𑀫𑁄 𑀆𑀯𑀭𑀡𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀯𑀲𑀢𑀺, 𑀬𑀢𑁆𑀣 𑀯𑀸 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀆𑀯𑀭𑀡𑀁 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

477\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀸𑀫𑀡𑁂𑀭𑀸𑀦𑀁 𑀫𑀼𑀔𑀤𑁆𑀯𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀸𑀭𑀁 𑀆𑀯𑀭𑀡𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀬𑀸𑀕𑀼𑀧𑀸𑀦𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀸 𑀲𑀸𑀫𑀡𑁂𑀭𑁂 𑀏𑀯𑀁 𑀯𑀤𑁂𑀦𑁆𑀢𑀺—  “𑀏𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀸𑀕𑀼𑀁 𑀧𑀺𑀯𑀣; 𑀏𑀣, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀣𑀸”𑀢𑀺𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀦𑁆𑀢𑀺—  “𑀦𑀸𑀯𑀼𑀲𑁄, 𑀮𑀩𑁆𑀪𑀸𑁇 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀯𑀭𑀡𑀁 𑀓𑀢𑀦𑁆”𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀤𑀦𑁆𑀢𑀸 𑀲𑀸𑀫𑀡𑁂𑀭𑀸𑀦𑀁 𑀫𑀼𑀔𑀤𑁆𑀯𑀸𑀭𑀺𑀓𑀁 𑀆𑀳𑀸𑀭𑀁 𑀆𑀯𑀭𑀡𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀼𑀔𑀤𑁆𑀯𑀸𑀭𑀺𑀓𑁄 𑀆𑀳𑀸𑀭𑁄 𑀆𑀯𑀭𑀡𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
