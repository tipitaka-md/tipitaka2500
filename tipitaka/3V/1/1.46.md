# 𑁧𑁇𑁪𑁬 𑀓𑀡𑁆𑀝𑀓𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀓𑀡𑁆𑀝𑀓𑀲𑀸𑀫𑀡𑁂𑀭𑀯𑀢𑁆𑀣𑀼

481. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑀡𑁆𑀝𑀓𑁄 𑀦𑀸𑀫 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀓𑀡𑁆𑀝𑀓𑀺𑀁 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀤𑀽𑀲𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀁 𑀦𑀸𑀲𑁂𑀢𑀼𑀁𑁇 𑀧𑀸𑀡𑀸𑀢𑀺𑀧𑀸𑀢𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀤𑀺𑀦𑁆𑀦𑀸𑀤𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀻 𑀳𑁄𑀢𑀺, 𑀫𑀚𑁆𑀚𑀧𑀸𑀬𑀻 𑀳𑁄𑀢𑀺, 𑀩𑀼𑀤𑁆𑀥𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀲𑀁𑀖𑀲𑁆𑀲 𑀅𑀯𑀡𑁆𑀡𑀁 𑀪𑀸𑀲𑀢𑀺, 𑀫𑀺𑀘𑁆𑀙𑀸𑀤𑀺𑀝𑁆𑀞𑀺𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑁄 𑀳𑁄𑀢𑀺—  𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀳𑀺 𑀤𑀲𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀁 𑀦𑀸𑀲𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
