# 𑁧𑁇𑁬𑁩 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀯𑀺𑀥𑀺

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀯𑀺𑀥𑀺

(125.)

515\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀓𑀼𑀝𑁆𑀞𑀺𑀓𑀸𑀧𑀺 𑀕𑀡𑁆𑀟𑀺𑀓𑀸𑀧𑀺 𑀓𑀺𑀮𑀸𑀲𑀺𑀓𑀸𑀧𑀺 𑀲𑁄𑀲𑀺𑀓𑀸𑀧𑀺 𑀅𑀧𑀫𑀸𑀭𑀺𑀓𑀸𑀧𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑁂𑀦 𑀢𑁂𑀭𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀩𑁆𑀩𑁄—  ‘𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀸 𑀆𑀩𑀸𑀥𑀸—  𑀓𑀼𑀝𑁆𑀞𑀁, 𑀕𑀡𑁆𑀟𑁄, 𑀓𑀺𑀮𑀸𑀲𑁄, 𑀲𑁄𑀲𑁄, 𑀅𑀧𑀫𑀸𑀭𑁄? 𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑀲𑀺? 𑀧𑀼𑀭𑀺𑀲𑁄𑀲𑀺? 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑁄𑀲𑀺? 𑀅𑀡𑀡𑁄𑀲𑀺? 𑀦𑀲𑀺 𑀭𑀸𑀚𑀪𑀝𑁄? 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄𑀲𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺? 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄𑀲𑀺? 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁? 𑀓𑀺𑀁𑀦𑀸𑀫𑁄𑀲𑀺? 𑀓𑁄𑀦𑀸𑀫𑁄 𑀢𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄’”𑀢𑀺?

516\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀬𑀦𑁆𑀢𑀺, 𑀫𑀗𑁆𑀓𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀞𑀫𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀧𑀘𑁆𑀙𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

517\. 𑀢𑀢𑁆𑀣𑁂𑀯 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀢𑀣𑁂𑀯 𑀯𑀺𑀢𑁆𑀣𑀸𑀬𑀦𑁆𑀢𑀺, 𑀫𑀗𑁆𑀓𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄—

(126.)

518\. 𑀧𑀞𑀫𑀁 𑀉𑀧𑀚𑁆𑀛𑀁 𑀕𑀸𑀳𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀉𑀧𑀚𑁆𑀛𑀁 𑀕𑀸𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁—  ‘𑀅𑀬𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑁄, 𑀅𑀬𑀁 𑀲𑀗𑁆𑀖𑀸𑀝𑀺, 𑀅𑀬𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑁄, 𑀅𑀬𑀁 𑀅𑀦𑁆𑀢𑀭𑀯𑀸𑀲𑀓𑁄𑁇 𑀕𑀘𑁆𑀙, 𑀅𑀫𑀼𑀫𑁆𑀳𑀺 𑀑𑀓𑀸𑀲𑁂 𑀢𑀺𑀝𑁆𑀞𑀸𑀳𑀻’”𑀢𑀺𑁇

519\. 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀤𑀼𑀭𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀯𑀺𑀢𑁆𑀣𑀸𑀬𑀦𑁆𑀢𑀺, 𑀫𑀗𑁆𑀓𑀽 𑀳𑁄𑀦𑁆𑀢𑀺, 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀮𑁂𑀦 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

520\. 𑀅𑀲𑀫𑁆𑀫𑀢𑀸 𑀅𑀦𑀼𑀲𑀸𑀲𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀫𑁆𑀫𑀢𑁂𑀦 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀦𑀼𑀲𑀸𑀲𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀢𑁂𑀦 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄—  𑀅𑀢𑁆𑀢𑀦𑀸 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁, 𑀧𑀭𑁂𑀦 𑀯𑀸 𑀧𑀭𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇

521\. 𑀓𑀣𑀜𑁆𑀘 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁? 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀅𑀳𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀅𑀢𑁆𑀢𑀦𑀸𑀯 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑀁𑁇

522\. 𑀓𑀣𑀜𑁆𑀘 𑀧𑀦 𑀧𑀭𑁂𑀦 𑀧𑀭𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄? 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀏𑀯𑀁 𑀧𑀭𑁂𑀦 𑀧𑀭𑁄 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀺𑀢𑀩𑁆𑀩𑁄𑁇

523\. 𑀢𑁂𑀦 𑀲𑀫𑁆𑀫𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀲𑀼𑀡𑀲𑀺, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫, 𑀅𑀬𑀁 𑀢𑁂 𑀲𑀘𑁆𑀘𑀓𑀸𑀮𑁄 𑀪𑀽𑀢𑀓𑀸𑀮𑁄𑁇 𑀬𑀁 𑀚𑀸𑀢𑀁 𑀢𑀁 𑀲𑀁𑀖𑀫𑀚𑁆𑀛𑁂 𑀧𑀼𑀘𑁆𑀙𑀦𑁆𑀢𑁂 𑀲𑀦𑁆𑀢𑀁 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀦𑀢𑁆𑀣𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀫𑀸 𑀔𑁄 𑀯𑀺𑀢𑁆𑀣𑀸𑀬𑀺, 𑀫𑀸 𑀔𑁄 𑀫𑀗𑁆𑀓𑀼 𑀅𑀳𑁄𑀲𑀺𑁇 𑀏𑀯𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀸 𑀆𑀩𑀸𑀥𑀸—  𑀓𑀼𑀝𑁆𑀞𑀁, 𑀕𑀡𑁆𑀟𑁄, 𑀓𑀺𑀮𑀸𑀲𑁄, 𑀲𑁄𑀲𑁄, 𑀅𑀧𑀫𑀸𑀭𑁄? 𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑀲𑀺? 𑀧𑀼𑀭𑀺𑀲𑁄𑀲𑀺? 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑁄𑀲𑀺? 𑀅𑀡𑀡𑁄𑀲𑀺? 𑀦𑀲𑀺 𑀭𑀸𑀚𑀪𑀝𑁄? 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄𑀲𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺? 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄𑀲𑀺? 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁? 𑀓𑀺𑀁𑀦𑀸𑀫𑁄𑀲𑀺? 𑀓𑁄𑀦𑀸𑀫𑁄 𑀢𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄’”𑀢𑀺?

524\. 𑀏𑀓𑀢𑁄 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑀢𑁄 𑀆𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀼𑀲𑀸𑀲𑀓𑁂𑀦 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀅𑀦𑀼𑀲𑀺𑀝𑁆𑀞𑁄 𑀲𑁄 𑀫𑀬𑀸𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 ‘𑀆𑀕𑀘𑁆𑀙𑀸𑀳𑀻’𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄𑁇

525\. 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀲𑀁𑀖𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀸𑀫𑀺𑁇 𑀉𑀮𑁆𑀮𑀼𑀫𑁆𑀧𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀸𑀫𑀺𑁇 𑀉𑀮𑁆𑀮𑀼𑀫𑁆𑀧𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀸𑀫𑀺𑁇 𑀉𑀮𑁆𑀮𑀼𑀫𑁆𑀧𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

526\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀅𑀳𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂 𑀥𑀫𑁆𑀫𑁂 𑀧𑀼𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀦𑁆𑀢𑀺? 𑀲𑀼𑀡𑀲𑀺, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫, 𑀅𑀬𑀁 𑀢𑁂 𑀲𑀘𑁆𑀘𑀓𑀸𑀮𑁄 𑀪𑀽𑀢𑀓𑀸𑀮𑁄𑁇 𑀬𑀁 𑀚𑀸𑀢𑀁 𑀢𑀁 𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺𑁇 𑀲𑀦𑁆𑀢𑀁 𑀅𑀢𑁆𑀣𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁, 𑀅𑀲𑀦𑁆𑀢𑀁 𑀦𑀢𑁆𑀣𑀻𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀲𑀦𑁆𑀢𑀺 𑀢𑁂 𑀏𑀯𑀭𑀽𑀧𑀸 𑀆𑀩𑀸𑀥𑀸—  𑀓𑀼𑀝𑁆𑀞𑀁 𑀕𑀡𑁆𑀟𑁄 𑀓𑀺𑀮𑀸𑀲𑁄 𑀲𑁄𑀲𑁄 𑀅𑀧𑀫𑀸𑀭𑁄? 𑀫𑀦𑀼𑀲𑁆𑀲𑁄𑀲𑀺? 𑀧𑀼𑀭𑀺𑀲𑁄𑀲𑀺? 𑀪𑀼𑀚𑀺𑀲𑁆𑀲𑁄𑀲𑀺? 𑀅𑀡𑀡𑁄𑀲𑀺? 𑀦𑀲𑀺 𑀭𑀸𑀚𑀪𑀝𑁄? 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄𑀲𑀺 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺? 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄𑀲𑀺? 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁? 𑀓𑀺𑀁𑀦𑀸𑀫𑁄𑀲𑀺? 𑀓𑁄𑀦𑀸𑀫𑁄 𑀢𑁂 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄’𑀢𑀺? 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(127.)

527\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺, 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

528\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺, 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

529\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺, 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

530\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄, 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀺𑀓𑁂𑀳𑀺 𑀥𑀫𑁆𑀫𑁂𑀳𑀺, 𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀁𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

531\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇
