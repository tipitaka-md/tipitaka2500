# 𑁧𑁇𑁪 𑀭𑀸𑀚𑀸𑀬𑀢𑀦𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀭𑀸𑀚𑀸𑀬𑀢𑀦𑀓𑀣𑀸

(6.)

21. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀫𑀽𑀮𑀸 𑀬𑁂𑀦 𑀭𑀸𑀚𑀸𑀬𑀢𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀭𑀸𑀚𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑁂 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀏𑀓𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀼𑀔𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀧𑀼𑀲𑁆𑀲 𑀪𑀮𑁆𑀮𑀺𑀓𑀸 𑀯𑀸𑀡𑀺𑀚𑀸 𑀉𑀓𑁆𑀓𑀮𑀸 𑀢𑀁 𑀤𑁂𑀲𑀁 𑀅𑀤𑁆𑀥𑀸𑀦𑀫𑀕𑁆𑀕𑀧𑁆𑀧𑀝𑀺𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀧𑀼𑀲𑁆𑀲𑀪𑀮𑁆𑀮𑀺𑀓𑀸𑀦𑀁 𑀯𑀸𑀡𑀺𑀚𑀸𑀦𑀁 𑀜𑀸𑀢𑀺𑀲𑀸𑀮𑁄𑀳𑀺𑀢𑀸 𑀤𑁂𑀯𑀢𑀸 𑀢𑀧𑀼𑀲𑁆𑀲𑀪𑀮𑁆𑀮𑀺𑀓𑁂 𑀯𑀸𑀡𑀺𑀚𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀬𑀁, 𑀫𑀸𑀭𑀺𑀲𑀸, 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀸𑀬𑀢𑀦𑀫𑀽𑀮𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀧𑀞𑀫𑀸𑀪𑀺𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄; 𑀕𑀘𑁆𑀙𑀣 𑀢𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀫𑀦𑁆𑀣𑁂𑀦 𑀘 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀸𑀬 𑀘 𑀧𑀢𑀺𑀫𑀸𑀦𑁂𑀣; 𑀢𑀁 𑀯𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇

22. 𑀅𑀣 𑀔𑁄 𑀢𑀧𑀼𑀲𑁆𑀲𑀪𑀮𑁆𑀮𑀺𑀓𑀸 𑀯𑀸𑀡𑀺𑀚𑀸 𑀫𑀦𑁆𑀣𑀜𑁆𑀘 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀜𑁆𑀘 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀞𑀺𑀢𑀸 𑀔𑁄 𑀢𑀧𑀼𑀲𑁆𑀲𑀪𑀮𑁆𑀮𑀺𑀓𑀸 𑀯𑀸𑀡𑀺𑀚𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀦𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀫𑀦𑁆𑀣𑀜𑁆𑀘 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀜𑁆𑀘, 𑀬𑀁 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀲𑁆𑀲 𑀤𑀻𑀖𑀭𑀢𑁆𑀢𑀁 𑀳𑀺𑀢𑀸𑀬 𑀲𑀼𑀔𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦 𑀔𑁄 𑀢𑀣𑀸𑀕𑀢𑀸 𑀳𑀢𑁆𑀣𑁂𑀲𑀼 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑁇 𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬𑀁 𑀫𑀦𑁆𑀣𑀜𑁆𑀘 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀜𑁆𑀘𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀪𑀕𑀯𑀢𑁄 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀲𑁂𑀮𑀫𑀬𑁂 𑀧𑀢𑁆𑀢𑁂 𑀪𑀕𑀯𑀢𑁄 𑀉𑀧𑀦𑀸𑀫𑁂𑀲𑀼𑀁—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀢𑀼 𑀫𑀦𑁆𑀣𑀜𑁆𑀘 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀜𑁆𑀘𑀸”𑀢𑀺𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀧𑀘𑁆𑀘𑀕𑁆𑀖𑁂 𑀲𑁂𑀮𑀫𑀬𑁂 𑀧𑀢𑁆𑀢𑁂 𑀫𑀦𑁆𑀣𑀜𑁆𑀘 𑀫𑀥𑀼𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀜𑁆𑀘, 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑁇

23. 𑀅𑀣 𑀔𑁄 𑀢𑀧𑀼𑀲𑁆𑀲𑀪𑀮𑁆𑀮𑀺𑀓𑀸 𑀯𑀸𑀡𑀺𑀚𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀏𑀢𑁂 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫 𑀥𑀫𑁆𑀫𑀜𑁆𑀘, 𑀉𑀧𑀸𑀲𑀓𑁂 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑁂 𑀲𑀭𑀡𑀁 𑀕𑀢𑁂”𑀢𑀺𑁇 𑀢𑁂 𑀘 𑀮𑁄𑀓𑁂 𑀧𑀞𑀫𑀁 𑀉𑀧𑀸𑀲𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀤𑁆𑀯𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑁇

---

24. 𑀭𑀸𑀚𑀸𑀬𑀢𑀦𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
