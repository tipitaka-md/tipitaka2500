# 𑁧𑁇𑁪𑁧 𑀭𑀸𑀳𑀼𑀮𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀭𑀸𑀳𑀼𑀮𑀯𑀢𑁆𑀣𑀼

(105.)

465\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀓𑁆𑀓𑁂𑀲𑀼 𑀯𑀺𑀳𑀭𑀢𑀺 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀸𑀭𑀸𑀫𑁂𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀬𑁂𑀦 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑀲𑁆𑀲 𑀲𑀓𑁆𑀓𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀳𑀼𑀮𑀫𑀸𑀢𑀸 𑀤𑁂𑀯𑀻 𑀭𑀸𑀳𑀼𑀮𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀲𑁄 𑀢𑁂, 𑀭𑀸𑀳𑀼𑀮, 𑀧𑀺𑀢𑀸𑁇 𑀕𑀘𑁆𑀙𑀲𑁆𑀲𑀼, 𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀸𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀳𑀼𑀮𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄, 𑀅𑀝𑁆𑀞𑀸𑀲𑀺—  “𑀲𑀼𑀔𑀸 𑀢𑁂, 𑀲𑀫𑀡, 𑀙𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀳𑀼𑀮𑁄 𑀓𑀼𑀫𑀸𑀭𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀧𑀺𑀝𑁆𑀞𑀺𑀢𑁄 𑀅𑀦𑀼𑀩𑀦𑁆𑀥𑀺—  “𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀫𑁂, 𑀲𑀫𑀡, 𑀤𑁂𑀳𑀺; 𑀤𑀸𑀬𑀚𑁆𑀚𑀁 𑀫𑁂, 𑀲𑀫𑀡, 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀭𑀸𑀳𑀼𑀮𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀸𑀳𑀼𑀮𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀫𑀻”𑀢𑀺?

466\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀻𑀳𑀺 𑀲𑀭𑀡𑀕𑀫𑀦𑁂𑀳𑀺 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄—  𑀧𑀞𑀫𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 ‘𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀻’𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—

467\. _‘𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇_  


468\. _𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇_  


469\. _𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀻’𑀢𑀺𑁇_  


470\. 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀲𑀭𑀡𑀕𑀫𑀦𑁂𑀳𑀺 𑀲𑀸𑀫𑀡𑁂𑀭𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀸𑀳𑀼𑀮𑀁 𑀓𑀼𑀫𑀸𑀭𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀺𑁇

471\. 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀲𑀓𑁆𑀓𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀓𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀯𑀭𑀁 𑀬𑀸𑀘𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀅𑀢𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀭𑀸 𑀔𑁄, 𑀕𑁄𑀢𑀫, 𑀢𑀣𑀸𑀕𑀢𑀸”𑀢𑀺𑁇 “𑀬𑀜𑁆𑀘, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀧𑁆𑀧𑀢𑀺, 𑀬𑀜𑁆𑀘 𑀅𑀦𑀯𑀚𑁆𑀚𑀦𑁆”𑀢𑀺𑁇 “𑀯𑀤𑁂𑀳𑀺, 𑀕𑁄𑀢𑀫𑀸”𑀢𑀺𑁇 “𑀪𑀕𑀯𑀢𑀺 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀅𑀦𑀧𑁆𑀧𑀓𑀁 𑀤𑀼𑀓𑁆𑀔𑀁 𑀅𑀳𑁄𑀲𑀺, 𑀢𑀣𑀸 𑀦𑀦𑁆𑀤𑁂, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀭𑀸𑀳𑀼𑀮𑁂𑁇 𑀧𑀼𑀢𑁆𑀢𑀧𑁂𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀙𑀯𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀙𑀯𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀘𑀫𑁆𑀫𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀘𑀫𑁆𑀫𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀫𑀁𑀲𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀫𑀁𑀲𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑁆𑀳𑀸𑀭𑀼𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑀺𑀦𑁆𑀤𑀢𑀺, 𑀅𑀝𑁆𑀞𑀺𑀁 𑀙𑁂𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀁 𑀆𑀳𑀘𑁆𑀘 𑀢𑀺𑀝𑁆𑀞𑀢𑀺𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑀁 𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑀁 𑀲𑀓𑁆𑀓𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀼𑀤𑁆𑀥𑁄𑀤𑀦𑁄 𑀲𑀓𑁆𑀓𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑁄 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀳𑀺 𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

472\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀓𑀧𑀺𑀮𑀯𑀢𑁆𑀣𑀼𑀲𑁆𑀫𑀺𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀸𑀭𑀓𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀇𑀫𑀁 𑀤𑀸𑀭𑀓𑀁 𑀣𑁂𑀭𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸’𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘 𑀫𑁂 𑀭𑀸𑀳𑀼𑀮𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁄𑁇 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀲𑀸𑀫𑀡𑁂𑀭𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀼𑀁, 𑀬𑀸𑀯𑀢𑀓𑁂 𑀯𑀸 𑀧𑀦 𑀉𑀲𑁆𑀲𑀳𑀢𑀺 𑀑𑀯𑀤𑀺𑀢𑀼𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀢𑀼𑀁 𑀢𑀸𑀯𑀢𑀓𑁂 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
