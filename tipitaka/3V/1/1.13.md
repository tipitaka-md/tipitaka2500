# 𑁧𑁇𑁧𑁩 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑀫𑀸𑀕𑀫𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑀫𑀸𑀕𑀫𑀓𑀣𑀸

(55.)

202\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀕𑀬𑀸𑀲𑀻𑀲𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺, 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀧𑀼𑀭𑀸𑀡𑀚𑀝𑀺𑀮𑁂𑀳𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑀁 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀝𑁆𑀞𑀺𑀯𑀦𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀢𑀺𑀬𑁂𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄—  “𑀲𑀫𑀡𑁄 𑀔𑀮𑀼 𑀪𑁄 𑀕𑁄𑀢𑀫𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀓𑁆𑀬𑀓𑀼𑀮𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑀁 𑀅𑀦𑀼𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀮𑀝𑁆𑀞𑀺𑀯𑀦𑁂 𑀲𑀼𑀧𑁆𑀧𑀢𑀺𑀝𑁆𑀞𑁂 𑀘𑁂𑀢𑀺𑀬𑁂𑁇 𑀢𑀁 𑀔𑁄 𑀧𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀕𑁄𑀢𑀫𑀁 𑀏𑀯𑀁 𑀓𑀮𑁆𑀬𑀸𑀡𑁄 𑀓𑀺𑀢𑁆𑀢𑀺𑀲𑀤𑁆𑀤𑁄 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀢𑁄—  𑀇𑀢𑀺𑀧𑀺 𑀲𑁄 𑀪𑀕𑀯𑀸 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀚𑁆𑀚𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀽 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑀤𑀫𑁆𑀫𑀲𑀸𑀭𑀣𑀺 𑀲𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑁇 𑀲𑁄 𑀇𑀫𑀁 𑀮𑁄𑀓𑀁 𑀲𑀤𑁂𑀯𑀓𑀁 𑀲𑀫𑀸𑀭𑀓𑀁 𑀲𑀩𑁆𑀭𑀳𑁆𑀫𑀓𑀁 𑀲𑀲𑁆𑀲𑀫𑀡𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀺𑀁 𑀧𑀚𑀁 𑀲𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀁 𑀲𑀬𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸 𑀲𑀘𑁆𑀙𑀺𑀓𑀢𑁆𑀯𑀸 𑀧𑀯𑁂𑀤𑁂𑀢𑀺𑁇 𑀲𑁄 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀢𑀺 𑀆𑀤𑀺𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀫𑀚𑁆𑀛𑁂𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀧𑀭𑀺𑀬𑁄𑀲𑀸𑀦𑀓𑀮𑁆𑀬𑀸𑀡𑀁 𑀲𑀸𑀢𑁆𑀣𑀁 𑀲𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀁 𑀓𑁂𑀯𑀮𑀧𑀭𑀺𑀧𑀼𑀡𑁆𑀡𑀁 𑀧𑀭𑀺𑀲𑀼𑀤𑁆𑀥𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀧𑀓𑀸𑀲𑁂𑀢𑀺𑁇 𑀲𑀸𑀥𑀼 𑀔𑁄 𑀧𑀦 𑀢𑀣𑀸𑀭𑀽𑀧𑀸𑀦𑀁 𑀅𑀭𑀳𑀢𑀁 𑀤𑀲𑁆𑀲𑀦𑀁 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

203\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑀳𑀼𑀢𑁂𑀳𑀺 𑀫𑀸𑀕𑀥𑀺𑀓𑁂𑀳𑀺 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑁂𑀳𑀺 𑀧𑀭𑀺𑀯𑀼𑀢𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀢𑁂𑀧𑀺 𑀔𑁄 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑀳𑀼𑀢𑀸 𑀫𑀸𑀕𑀥𑀺𑀓𑀸 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀢𑀸 𑀲𑀤𑁆𑀥𑀺𑀁 𑀲𑀫𑁆𑀫𑁄𑀤𑀺𑀁𑀲𑀼, 𑀲𑀫𑁆𑀫𑁄𑀤𑀦𑀻𑀬𑀁 𑀓𑀣𑀁 𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀻𑀢𑀺𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀦𑀸𑀫𑀕𑁄𑀢𑁆𑀢𑀁 𑀲𑀸𑀯𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼, 𑀅𑀧𑁆𑀧𑁂𑀓𑀘𑁆𑀘𑁂 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀽𑀢𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑀳𑀼𑀢𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀺𑀓𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀺, 𑀉𑀤𑀸𑀳𑀼 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀲𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑀳𑀼𑀢𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀺𑀓𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

204\. _“𑀓𑀺𑀫𑁂𑀯 𑀤𑀺𑀲𑁆𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀯𑀸𑀲𑀺,_  
_𑀧𑀳𑀸𑀲𑀺 𑀅𑀕𑁆𑀕𑀺𑀁 𑀓𑀺𑀲𑀓𑁄𑀯𑀤𑀸𑀦𑁄;_  
_𑀧𑀼𑀘𑁆𑀙𑀸𑀫𑀺 𑀢𑀁 𑀓𑀲𑁆𑀲𑀧 𑀏𑀢𑀫𑀢𑁆𑀣𑀁,_  
_𑀓𑀣𑀁 𑀧𑀳𑀻𑀦𑀁 𑀢𑀯 𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇_  


205\. _“𑀭𑀽𑀧𑁂 𑀘 𑀲𑀤𑁆𑀤𑁂 𑀘 𑀅𑀣𑁄 𑀭𑀲𑁂 𑀘,_  
_𑀓𑀸𑀫𑀺𑀢𑁆𑀣𑀺𑀬𑁄 𑀘𑀸𑀪𑀺𑀯𑀤𑀦𑁆𑀢𑀺 𑀬𑀜𑁆𑀜𑀸;_  
_𑀏𑀢𑀁 𑀫𑀮𑀦𑁆𑀢𑀺 𑀉𑀧𑀥𑀻𑀲𑀼 𑀜𑀢𑁆𑀯𑀸,_  
_𑀢𑀲𑁆𑀫𑀸 𑀦 𑀬𑀺𑀝𑁆𑀞𑁂 𑀦 𑀳𑀼𑀢𑁂 𑀅𑀭𑀜𑁆𑀚𑀺𑀦𑁆”𑀢𑀺𑁇_  


206\. _“𑀏𑀢𑁆𑀣𑁂𑀯 𑀢𑁂 𑀫𑀦𑁄 𑀦 𑀭𑀫𑀺𑀢𑁆𑀣, __(𑀓𑀲𑁆𑀲𑀧𑀸𑀢𑀺 𑀪𑀕𑀯𑀸)___  
_𑀭𑀽𑀧𑁂𑀲𑀼 𑀲𑀤𑁆𑀤𑁂𑀲𑀼 𑀅𑀣𑁄 𑀭𑀲𑁂𑀲𑀼;_  
_𑀅𑀣 𑀓𑁄 𑀘𑀭𑀳𑀺 𑀤𑁂𑀯𑀫𑀦𑀼𑀲𑁆𑀲𑀮𑁄𑀓𑁂,_  
_𑀭𑀢𑁄 𑀫𑀦𑁄 𑀓𑀲𑁆𑀲𑀧 𑀩𑁆𑀭𑀽𑀳𑀺 𑀫𑁂𑀢𑀦𑁆”𑀢𑀺𑁇_  


207\. _“𑀤𑀺𑀲𑁆𑀯𑀸 𑀧𑀤𑀁 𑀲𑀦𑁆𑀢𑀫𑀦𑀽𑀧𑀥𑀻𑀓𑀁,_  
_𑀅𑀓𑀺𑀜𑁆𑀘𑀦𑀁 𑀓𑀸𑀫𑀪𑀯𑁂 𑀅𑀲𑀢𑁆𑀢𑀁;_  
_𑀅𑀦𑀜𑁆𑀜𑀣𑀸𑀪𑀸𑀯𑀺𑀫𑀦𑀜𑁆𑀜𑀦𑁂𑀬𑁆𑀬𑀁,_  
_𑀢𑀲𑁆𑀫𑀸 𑀦 𑀬𑀺𑀝𑁆𑀞𑁂 𑀦 𑀳𑀼𑀢𑁂 𑀅𑀭𑀜𑁆𑀚𑀺𑀦𑁆”𑀢𑀺𑁇_  


(56.)

208\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀢𑁆𑀣𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀯𑀓𑁄𑀳𑀫𑀲𑁆𑀫𑀺; 𑀲𑀢𑁆𑀣𑀸 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀲𑀸𑀯𑀓𑁄𑀳𑀫𑀲𑁆𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑀳𑀼𑀢𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀺𑀓𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂𑀲𑀁 𑀤𑁆𑀯𑀸𑀤𑀲𑀦𑀳𑀼𑀢𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀺𑀓𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑁂 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑁂 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁂, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑀁, 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀏𑀓𑀸𑀤𑀲𑀦𑀳𑀼𑀢𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀺𑀓𑀸𑀦𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀕𑀳𑀧𑀢𑀺𑀓𑀸𑀦𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀧𑁆𑀧𑀫𑀼𑀔𑀸𑀦𑀁 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀏𑀓𑀦𑀳𑀼𑀢𑀁 𑀉𑀧𑀸𑀲𑀓𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀯𑁂𑀤𑁂𑀲𑀺𑁇

209\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁄 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑁄 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑁄 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀲𑀢𑁄 𑀧𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀲𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀢𑁂 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑀸𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀲𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀅𑀳𑁄 𑀯𑀢 𑀫𑀁 𑀭𑀚𑁆𑀚𑁂 𑀅𑀪𑀺𑀲𑀺𑀜𑁆𑀘𑁂𑀬𑁆𑀬𑀼𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀞𑀫𑁄 𑀅𑀲𑁆𑀲𑀸𑀲𑀓𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑁄𑁇 ‘𑀢𑀲𑁆𑀲 𑀘 𑀫𑁂 𑀯𑀺𑀚𑀺𑀢𑀁 𑀅𑀭𑀳𑀁 𑀲𑀫𑁆𑀫𑀸𑀲𑀫𑁆𑀩𑀼𑀤𑁆𑀥𑁄 𑀑𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀸’𑀢𑀺, 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀤𑀼𑀢𑀺𑀬𑁄 𑀅𑀲𑁆𑀲𑀸𑀲𑀓𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑁄𑁇 ‘𑀢𑀜𑁆𑀘𑀸𑀳𑀁 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀬𑀺𑀭𑀼𑀧𑀸𑀲𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀢𑀺𑀬𑁄 𑀅𑀲𑁆𑀲𑀸𑀲𑀓𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑁄𑁇 ‘𑀲𑁄 𑀘 𑀫𑁂 𑀪𑀕𑀯𑀸 𑀥𑀫𑁆𑀫𑀁 𑀤𑁂𑀲𑁂𑀬𑁆𑀬𑀸’𑀢𑀺, 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀢𑀼𑀢𑁆𑀣𑁄 𑀅𑀲𑁆𑀲𑀸𑀲𑀓𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑁄𑁇 ‘𑀢𑀲𑁆𑀲 𑀘𑀸𑀳𑀁 𑀪𑀕𑀯𑀢𑁄 𑀥𑀫𑁆𑀫𑀁 𑀆𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺, 𑀅𑀬𑀁 𑀔𑁄 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀘𑀫𑁄 𑀅𑀲𑁆𑀲𑀸𑀲𑀓𑁄 𑀅𑀳𑁄𑀲𑀺, 𑀲𑁄 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀓𑀼𑀫𑀸𑀭𑀲𑁆𑀲 𑀲𑀢𑁄 𑀇𑀫𑁂 𑀧𑀜𑁆𑀘 𑀅𑀲𑁆𑀲𑀸𑀲𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀁, 𑀢𑁂 𑀫𑁂 𑀏𑀢𑀭𑀳𑀺 𑀲𑀫𑀺𑀤𑁆𑀥𑀸𑁇 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁, 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀁, 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀘 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸, 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀧𑀡𑀻𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀧𑀝𑀺𑀬𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑀸𑀧𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇

210\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀸𑀯𑀺𑀲𑀺 𑀫𑀳𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑁂𑀦 𑀲𑀩𑁆𑀩𑁂𑀳𑁂𑀯 𑀧𑀼𑀭𑀸𑀡𑀚𑀝𑀺𑀮𑁂𑀳𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀸𑀡𑀯𑀓𑀯𑀡𑁆𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀼𑀭𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀕𑀘𑁆𑀙𑀢𑀺 𑀇𑀫𑀸 𑀕𑀸𑀣𑀸𑀬𑁄 𑀕𑀸𑀬𑀫𑀸𑀦𑁄—

211\. _“𑀤𑀦𑁆𑀢𑁄 𑀤𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀳 𑀧𑀼𑀭𑀸𑀡𑀚𑀝𑀺𑀮𑁂𑀳𑀺,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁂𑀳𑀺;_  
_𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑀯𑀡𑁆𑀡𑁄,_  
_𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀸𑀯𑀺𑀲𑀺 𑀪𑀕𑀯𑀸𑁇_  


212\. _𑀫𑀼𑀢𑁆𑀢𑁄 𑀫𑀼𑀢𑁆𑀢𑁂𑀳𑀺 𑀲𑀳 𑀧𑀼𑀭𑀸𑀡𑀚𑀝𑀺𑀮𑁂𑀳𑀺,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁂𑀳𑀺;_  
_𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑀯𑀡𑁆𑀡𑁄,_  
_𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀸𑀯𑀺𑀲𑀺 𑀪𑀕𑀯𑀸𑁇_  


213\. _𑀢𑀺𑀡𑁆𑀡𑁄 𑀢𑀺𑀡𑁆𑀡𑁂𑀳𑀺 𑀲𑀳 𑀧𑀼𑀭𑀸𑀡𑀚𑀝𑀺𑀮𑁂𑀳𑀺,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁂𑀳𑀺;_  
_𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑀯𑀡𑁆𑀡𑁄,_  
_𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀸𑀯𑀺𑀲𑀺 𑀪𑀕𑀯𑀸𑁇_  


214\. _𑀲𑀦𑁆𑀢𑁄 𑀲𑀦𑁆𑀢𑁂𑀳𑀺 𑀲𑀳 𑀧𑀼𑀭𑀸𑀡𑀚𑀝𑀺𑀮𑁂𑀳𑀺,_  
_𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁄 𑀯𑀺𑀧𑁆𑀧𑀫𑀼𑀢𑁆𑀢𑁂𑀳𑀺;_  
_𑀲𑀺𑀗𑁆𑀕𑀻𑀦𑀺𑀓𑁆𑀔𑀲𑀯𑀡𑁆𑀡𑁄,_  
_𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀸𑀯𑀺𑀲𑀺 𑀪𑀕𑀯𑀸𑁇_  


215\. _𑀤𑀲𑀯𑀸𑀲𑁄 𑀤𑀲𑀩𑀮𑁄,_  
_𑀤𑀲𑀥𑀫𑁆𑀫𑀯𑀺𑀤𑀽 𑀤𑀲𑀪𑀺 𑀘𑀼𑀧𑁂𑀢𑁄;_  
_𑀲𑁄 𑀤𑀲𑀲𑀢𑀧𑀭𑀺𑀯𑀸𑀭𑁄,_  
_𑀭𑀸𑀚𑀕𑀳𑀁 𑀧𑀸𑀯𑀺𑀲𑀺 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇_  


216\. 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀲𑀓𑁆𑀓𑀁 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑀁 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀯𑀢𑀸𑀬𑀁 𑀫𑀸𑀡𑀯𑀓𑁄, 𑀤𑀲𑁆𑀲𑀦𑀻𑀬𑁄 𑀯𑀢𑀸𑀬𑀁 𑀫𑀸𑀡𑀯𑀓𑁄, 𑀧𑀸𑀲𑀸𑀤𑀺𑀓𑁄 𑀯𑀢𑀸𑀬𑀁 𑀫𑀸𑀡𑀯𑀓𑁄𑁇 𑀓𑀲𑁆𑀲 𑀦𑀼 𑀔𑁄 𑀅𑀬𑀁 𑀫𑀸𑀡𑀯𑀓𑁄”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀢𑁂 𑀫𑀦𑀼𑀲𑁆𑀲𑁂 𑀕𑀸𑀣𑀸𑀬 𑀅𑀚𑁆𑀛𑀪𑀸𑀲𑀺—

217\. _“𑀬𑁄 𑀥𑀻𑀭𑁄 𑀲𑀩𑁆𑀩𑀥𑀺 𑀤𑀦𑁆𑀢𑁄,_  
_𑀲𑀼𑀤𑁆𑀥𑁄 𑀅𑀧𑁆𑀧𑀝𑀺𑀧𑀼𑀕𑁆𑀕𑀮𑁄;_  
_𑀅𑀭𑀳𑀁 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑁂,_  
_𑀢𑀲𑁆𑀲𑀸𑀳𑀁 𑀧𑀭𑀺𑀘𑀸𑀭𑀓𑁄”𑀢𑀺𑁇_  


(59.)

218\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑁂𑀦𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀔𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀢𑁆𑀣 𑀦𑀼 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑁂𑀬𑁆𑀬? 𑀬𑀁 𑀅𑀲𑁆𑀲 𑀕𑀸𑀫𑀢𑁄 𑀦𑁂𑀯 𑀅𑀢𑀺𑀤𑀽𑀭𑁂 𑀦 𑀅𑀘𑁆𑀘𑀸𑀲𑀦𑁆𑀦𑁂, 𑀕𑀫𑀦𑀸𑀕𑀫𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁, 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑀁, 𑀤𑀺𑀯𑀸 𑀅𑀧𑁆𑀧𑀸𑀓𑀺𑀡𑁆𑀡𑀁, 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁 𑀯𑀺𑀚𑀦𑀯𑀸𑀢𑀁, 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀳𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀓𑀁, 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀤𑀁 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀕𑀸𑀫𑀢𑁄 𑀦𑁂𑀯 𑀅𑀢𑀺𑀤𑀽𑀭𑁂 𑀦 𑀅𑀘𑁆𑀘𑀸𑀲𑀦𑁆𑀦𑁂 𑀕𑀫𑀦𑀸𑀕𑀫𑀦𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀺𑀓𑀸𑀦𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀦𑀻𑀬𑀁 𑀤𑀺𑀯𑀸 𑀅𑀧𑁆𑀧𑀸𑀓𑀺𑀡𑁆𑀡𑀁 𑀭𑀢𑁆𑀢𑀺𑀁 𑀅𑀧𑁆𑀧𑀲𑀤𑁆𑀤𑀁 𑀅𑀧𑁆𑀧𑀦𑀺𑀕𑁆𑀖𑁄𑀲𑀁 𑀯𑀺𑀚𑀦𑀯𑀸𑀢𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀭𑀸𑀳𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀓𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀲𑀸𑀭𑀼𑀧𑁆𑀧𑀁𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀤𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀲𑁄𑀯𑀡𑁆𑀡𑀫𑀬𑀁 𑀪𑀺𑀗𑁆𑀓𑀸𑀭𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀑𑀡𑁄𑀚𑁂𑀲𑀺—  “𑀏𑀢𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀍𑀉𑀯𑀦𑀁 𑀉𑀬𑁆𑀬𑀸𑀦𑀁 𑀩𑀼𑀤𑁆𑀥𑀧𑁆𑀧𑀫𑀼𑀔𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀲𑁆𑀲 𑀤𑀫𑁆𑀫𑀻”𑀢𑀺𑁇 𑀧𑀝𑀺𑀕𑁆𑀕𑀳𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀆𑀭𑀸𑀫𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀆𑀭𑀸𑀫𑀦𑁆”𑀢𑀺𑁇
