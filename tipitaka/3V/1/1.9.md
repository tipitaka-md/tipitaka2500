# 𑁧𑁇𑁯 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀽𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀸

(34.)

139\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀸𑀤𑀺𑀲𑀸 𑀦𑀸𑀦𑀸𑀚𑀦𑀧𑀤𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀧𑁂𑀓𑁆𑀔𑁂 𑀘 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁂 𑀘 𑀆𑀦𑁂𑀦𑁆𑀢𑀺—  “𑀪𑀕𑀯𑀸 𑀦𑁂 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁂𑀯 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀘 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀘𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀸𑀤𑀺𑀲𑀸 𑀦𑀸𑀦𑀸𑀚𑀦𑀧𑀤𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀧𑁂𑀓𑁆𑀔𑁂 𑀘 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁂 𑀘 𑀆𑀦𑁂𑀦𑁆𑀢𑀺—  ‘𑀪𑀕𑀯𑀸 𑀦𑁂 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁂𑀯 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀘 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀘𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁—  𑀢𑀼𑀫𑁆𑀳𑁂𑀯 𑀤𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀲𑀼 𑀢𑀸𑀲𑀼 𑀤𑀺𑀲𑀸𑀲𑀼 𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀣𑀸”𑀢𑀺𑁇

140\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀸𑀬𑀦𑁆𑀳𑀲𑀫𑀬𑀁 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀸𑀦𑀸 𑀯𑀼𑀝𑁆𑀞𑀺𑀢𑁄 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀇𑀥 𑀫𑀬𑁆𑀳𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀳𑁄𑀕𑀢𑀲𑁆𑀲 𑀧𑀝𑀺𑀲𑀮𑁆𑀮𑀻𑀦𑀲𑁆𑀲 𑀏𑀯𑀁 𑀘𑁂𑀢𑀲𑁄 𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑁄 𑀉𑀤𑀧𑀸𑀤𑀺—  𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦𑀸𑀦𑀸𑀤𑀺𑀲𑀸 𑀦𑀸𑀦𑀸𑀚𑀦𑀧𑀤𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀧𑁂𑀓𑁆𑀔𑁂 𑀘 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁂 𑀘 𑀆𑀦𑁂𑀦𑁆𑀢𑀺—  ‘𑀪𑀕𑀯𑀸 𑀦𑁂 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀢𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑁆𑀲𑀢𑀻’𑀢𑀺, 𑀢𑀢𑁆𑀣 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁂𑀯 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀘 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀘𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀅𑀦𑀼𑀚𑀸𑀦𑁂𑀬𑁆𑀬𑀁 𑀢𑀼𑀫𑁆𑀳𑁂𑀯 𑀤𑀸𑀦𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀸𑀲𑀼 𑀢𑀸𑀲𑀼 𑀤𑀺𑀲𑀸𑀲𑀼 𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀣𑀸𑀢𑀺, 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀼𑀫𑁆𑀳𑁂𑀯 𑀤𑀸𑀦𑀺 𑀢𑀸𑀲𑀼 𑀢𑀸𑀲𑀼 𑀤𑀺𑀲𑀸𑀲𑀼 𑀢𑁂𑀲𑀼 𑀢𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀣 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀣𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄—

141\. 𑀧𑀞𑀫𑀁 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸, 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸, 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀸𑀭𑀸𑀧𑁂𑀢𑁆𑀯𑀸, 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸, 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀸𑀧𑁂𑀢𑁆𑀯𑀸, 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀡𑁆𑀳𑀸𑀧𑁂𑀢𑁆𑀯𑀸 ‘𑀏𑀯𑀁 𑀯𑀤𑁂𑀳𑀻’𑀢𑀺 𑀯𑀢𑁆𑀢𑀩𑁆𑀩𑁄—

142\. _‘𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇_  


143\. _𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺𑁇_  


144\. _𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀥𑀫𑁆𑀫𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺,_  
_𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀲𑀁𑀖𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀻’𑀢𑀺𑁇_  


145\. 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀳𑀺 𑀢𑀻𑀳𑀺 𑀲𑀭𑀡𑀕𑀫𑀦𑁂𑀳𑀺 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇

---

146\. 𑀢𑀻𑀳𑀺 𑀲𑀭𑀡𑀕𑀫𑀦𑁂𑀳𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
