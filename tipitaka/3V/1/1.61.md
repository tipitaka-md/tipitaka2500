# 𑁧𑁇𑁬𑁧 𑀤𑁆𑀯𑁂𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸𑀤𑀺𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀤𑁆𑀯𑁂𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸𑀤𑀺𑀯𑀢𑁆𑀣𑀼

(123.)

512\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑀺𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀏𑀓𑀸𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑁂 𑀓𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

513\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸𑀦𑀁 𑀣𑁂𑀭𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀯𑀺𑀯𑀤𑀦𑁆𑀢𑀺—  “𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀅𑀳𑀁 𑀧𑀞𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀣𑁂𑀭𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀳𑀦𑁆𑀤 𑀫𑀬𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀏𑀓𑀸𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑁂 𑀓𑀭𑁄𑀫𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀢𑀬𑁄 𑀏𑀓𑀸𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑁂 𑀓𑀸𑀢𑀼𑀁, 𑀢𑀜𑁆𑀘 𑀔𑁄 𑀏𑀓𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀦𑀸𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑀸”𑀢𑀺𑁇
