# 𑁧𑁇𑁩𑁬 𑀉𑀧𑀸𑀮𑀺𑀤𑀸𑀭𑀓𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀉𑀧𑀸𑀮𑀺𑀤𑀸𑀭𑀓𑀯𑀢𑁆𑀣𑀼

(99.)

437\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀲𑀢𑁆𑀢𑀭𑀲𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀤𑀸𑀭𑀓𑀸 𑀲𑀳𑀸𑀬𑀓𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑀸𑀮𑀺𑀤𑀸𑀭𑀓𑁄 𑀢𑁂𑀲𑀁 𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀉𑀧𑀸𑀬𑁂𑀦 𑀉𑀧𑀸𑀮𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀼𑀔𑀜𑁆𑀘 𑀚𑀻𑀯𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀮𑁂𑀔𑀁 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀏𑀯𑀁 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀼𑀔𑀜𑁆𑀘 𑀚𑀻𑀯𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀮𑁂𑀔𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀗𑁆𑀕𑀼𑀮𑀺𑀬𑁄 𑀤𑀼𑀓𑁆𑀔𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀕𑀡𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀏𑀯𑀁 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀼𑀔𑀜𑁆𑀘 𑀚𑀻𑀯𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀕𑀡𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀺, 𑀉𑀭𑀲𑁆𑀲 𑀤𑀼𑀓𑁆𑀔𑁄 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀭𑀽𑀧𑀁 𑀲𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀏𑀯𑀁 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀼𑀔𑀜𑁆𑀘 𑀚𑀻𑀯𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀲𑀘𑁂 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀭𑀽𑀧𑀁 𑀲𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀓𑁆𑀔𑀻𑀦𑀺 𑀤𑀼𑀓𑁆𑀔𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀔𑁄 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀻𑀮𑀸 𑀲𑀼𑀔𑀲𑀫𑀸𑀘𑀸𑀭𑀸, 𑀲𑀼𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀢𑁂𑀲𑀼 𑀲𑀬𑀦𑁂𑀲𑀼 𑀲𑀬𑀦𑁆𑀢𑀺𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬, 𑀏𑀯𑀁 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀲𑀼𑀔𑀜𑁆𑀘 𑀚𑀻𑀯𑁂𑀬𑁆𑀬, 𑀦 𑀘 𑀓𑀺𑀮𑀫𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇

438\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺𑀤𑀸𑀭𑀓𑁄 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀽𑀦𑀁 𑀇𑀫𑀁 𑀓𑀣𑀸𑀲𑀮𑁆𑀮𑀸𑀧𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀧𑀸𑀮𑀺𑀤𑀸𑀭𑀓𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀤𑀸𑀭𑀓𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀤𑀸𑀭𑀓𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀏𑀣 𑀫𑀬𑀁, 𑀅𑀬𑁆𑀬𑀸, 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀲𑀘𑁂 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀅𑀬𑁆𑀬, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺, 𑀏𑀯𑀁 𑀫𑀬𑀫𑁆𑀧𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀤𑀸𑀭𑀓𑀸 𑀏𑀓𑀫𑁂𑀓𑀲𑁆𑀲 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀣 𑀫𑀁 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀸𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀤𑀸𑀭𑀓𑀸𑀦𑀁 𑀫𑀸𑀢𑀸𑀧𑀺𑀢𑀭𑁄—  “𑀲𑀩𑁆𑀩𑁂𑀧𑀺𑀫𑁂 𑀤𑀸𑀭𑀓𑀸 𑀲𑀫𑀸𑀦𑀘𑁆𑀙𑀦𑁆𑀤𑀸 𑀓𑀮𑁆𑀬𑀸𑀡𑀸𑀥𑀺𑀧𑁆𑀧𑀸𑀬𑀸”𑀢𑀺—  𑀅𑀦𑀼𑀚𑀸𑀦𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀼𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀢𑁂 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀭𑁄𑀤𑀦𑁆𑀢𑀺—  “𑀬𑀸𑀕𑀼𑀁 𑀤𑁂𑀣, 𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀣, 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀆𑀕𑀫𑁂𑀣, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀯 𑀭𑀢𑁆𑀢𑀺 𑀯𑀺𑀪𑀸𑀬𑀢𑀺𑁇 𑀲𑀘𑁂 𑀬𑀸𑀕𑀼 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀧𑀺𑀯𑀺𑀲𑁆𑀲𑀣, 𑀲𑀘𑁂 𑀪𑀢𑁆𑀢𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀣, 𑀲𑀘𑁂 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀔𑀸𑀤𑀺𑀲𑁆𑀲𑀣; 𑀦𑁄 𑀘𑁂 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺 𑀬𑀸𑀕𑀼 𑀯𑀸 𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀯𑀸, 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀣𑀸”𑀢𑀺𑁇 𑀏𑀯𑀫𑁆𑀧𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀯𑀼𑀘𑁆𑀘𑀫𑀸𑀦𑀸 𑀭𑁄𑀤𑀦𑁆𑀢𑀺𑀬𑁂𑀯 “𑀬𑀸𑀕𑀼𑀁 𑀤𑁂𑀣, 𑀪𑀢𑁆𑀢𑀁 𑀤𑁂𑀣, 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀤𑁂𑀣𑀸”𑀢𑀺; 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀳𑀤𑀦𑁆𑀢𑀺𑀧𑀺 𑀉𑀫𑁆𑀫𑀺𑀳𑀦𑁆𑀢𑀺𑀧𑀺𑁇

439\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀤𑀸𑀭𑀓𑀲𑀤𑁆𑀤𑀁𑁇 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀆𑀦𑀦𑁆𑀤𑀁 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑀸𑀭𑀓𑀲𑀤𑁆𑀤𑁄”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀚𑀸𑀦𑀁 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁂, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀸 𑀚𑀸𑀦𑀁 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑀁 𑀧𑀼𑀕𑁆𑀕𑀮𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀅𑀓𑁆𑀔𑀫𑁄 𑀳𑁄𑀢𑀺 𑀲𑀻𑀢𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑀲𑁆𑀲 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀬 𑀧𑀺𑀧𑀸𑀲𑀸𑀬 𑀟𑀁𑀲𑀫𑀓𑀲𑀯𑀸𑀢𑀸𑀢𑀧𑀲𑀭𑀻𑀲𑀧𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀦𑀁 𑀤𑀼𑀭𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀼𑀭𑀸𑀕𑀢𑀸𑀦𑀁 𑀯𑀘𑀦𑀧𑀣𑀸𑀦𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀁 𑀢𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀔𑀭𑀸𑀦𑀁 𑀓𑀝𑀼𑀓𑀸𑀦𑀁 𑀅𑀲𑀸𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀸𑀧𑀸𑀦𑀁 𑀧𑀸𑀡𑀳𑀭𑀸𑀦𑀁 𑀅𑀦𑀥𑀺𑀯𑀸𑀲𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄𑀯 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀔𑀫𑁄 𑀳𑁄𑀢𑀺 𑀲𑀻𑀢𑀲𑁆𑀲 𑀉𑀡𑁆𑀳𑀲𑁆𑀲 𑀚𑀺𑀖𑀘𑁆𑀙𑀸𑀬 𑀧𑀺𑀧𑀸𑀲𑀸𑀬 𑀟𑀁𑀲𑀫𑀓𑀲𑀯𑀸𑀢𑀸𑀢𑀧𑀲𑀭𑀻𑀲𑀧𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀸𑀦𑀁 𑀤𑀼𑀭𑀼𑀢𑁆𑀢𑀸𑀦𑀁 𑀤𑀼𑀭𑀸𑀕𑀢𑀸𑀦𑀁 𑀯𑀘𑀦𑀧𑀣𑀸𑀦𑀁, 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀁 𑀲𑀸𑀭𑀻𑀭𑀺𑀓𑀸𑀦𑀁 𑀯𑁂𑀤𑀦𑀸𑀦𑀁 𑀤𑀼𑀓𑁆𑀔𑀸𑀦𑀁 𑀢𑀺𑀩𑁆𑀩𑀸𑀦𑀁 𑀔𑀭𑀸𑀦𑀁 𑀓𑀝𑀼𑀓𑀸𑀦𑀁 𑀅𑀲𑀸𑀢𑀸𑀦𑀁 𑀅𑀫𑀦𑀸𑀧𑀸𑀦𑀁 𑀧𑀸𑀡𑀳𑀭𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀸𑀲𑀓𑀚𑀸𑀢𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬, 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀚𑀸𑀦𑀁 𑀊𑀦𑀯𑀻𑀲𑀢𑀺𑀯𑀲𑁆𑀲𑁄 𑀧𑀼𑀕𑁆𑀕𑀮𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬, 𑀬𑀣𑀸𑀥𑀫𑁆𑀫𑁄 𑀓𑀸𑀭𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
