# 𑁧𑁇𑁨𑁦 𑀧𑀡𑀸𑀫𑀦𑀸𑀔𑀫𑀸𑀧𑀦𑀸

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀡𑀸𑀫𑀦𑀸𑀔𑀫𑀸𑀧𑀦𑀸

(80.)

352\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀆𑀘𑀭𑀺𑀬𑁂𑀲𑀼 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁂𑀦 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀦𑁂𑀯 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄—  𑀧𑀡𑀸𑀫𑁂𑀫𑀺 𑀢𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀫𑀸𑀬𑀺𑀥 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀻𑀢𑀺 𑀯𑀸, 𑀦𑀻𑀳𑀭 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀦𑁆𑀢𑀺 𑀯𑀸, 𑀦𑀸𑀳𑀁 𑀢𑀬𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄𑀢𑀺 𑀯𑀸𑁇 𑀓𑀸𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀡𑀸𑀫𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄; 𑀦 𑀓𑀸𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀧𑀡𑀸𑀫𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄”𑀢𑀺𑁇

353\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀧𑀡𑀸𑀫𑀺𑀢𑀸 𑀦 𑀔𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀫𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀦𑁂𑀯 𑀔𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀸𑀫𑀺𑀢𑁂𑀦 𑀦 𑀔𑀫𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀦 𑀔𑀫𑀸𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

354\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀘𑀭𑀺𑀬𑀸 𑀔𑀫𑀸𑀧𑀺𑀬𑀫𑀸𑀦𑀸 𑀦 𑀔𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀫𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀦𑁂𑀯 𑀔𑀫𑀦𑁆𑀢𑀺𑁇 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀩𑁆𑀪𑀫𑀦𑁆𑀢𑀺𑀧𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀲𑀼𑀧𑀺 𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀫𑀸𑀧𑀺𑀬𑀫𑀸𑀦𑁂𑀦 𑀦 𑀔𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀦 𑀔𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

355\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀘𑀭𑀺𑀬𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑀺, 𑀅𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀦 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑁄 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀡𑀸𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸𑀢𑀺𑁇

(81.)

356\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀺 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇

357\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀺 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇

358\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀅𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀺 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀅𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇

359\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀦𑀸𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀺 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑁄 𑀦𑀸𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇

360\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀺 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇

361\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀁 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀆𑀘𑀭𑀺𑀬𑀫𑁆𑀳𑀺 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀁 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀆𑀘𑀭𑀺𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇
