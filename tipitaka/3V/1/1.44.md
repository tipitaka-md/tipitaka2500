# 𑁧𑁇𑁪𑁪 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀭𑀡𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸𑀯𑀭𑀡𑀯𑀢𑁆𑀣𑀼

(108.)

479. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀲𑀸𑀫𑀡𑁂𑀭𑀸𑀦𑀁 𑀆𑀯𑀭𑀡𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀦 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑁂𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀆𑀯𑀭𑀡𑀁 𑀓𑀢𑀦𑁆”𑀢𑀺𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀫𑁆𑀳𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀸𑀦𑀁 𑀆𑀯𑀭𑀡𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂 𑀅𑀦𑀸𑀧𑀼𑀘𑁆𑀙𑀸 𑀆𑀯𑀭𑀡𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
