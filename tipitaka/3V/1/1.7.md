# 𑁧𑁇𑁭 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀓𑀣𑀸

(25.)

107\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀬𑀲𑁄 𑀦𑀸𑀫 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀔𑀼𑀫𑀸𑀮𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀢𑀬𑁄 𑀧𑀸𑀲𑀸𑀤𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀏𑀓𑁄 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑁄, 𑀏𑀓𑁄 𑀕𑀺𑀫𑁆𑀳𑀺𑀓𑁄, 𑀏𑀓𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁄𑁇 𑀲𑁄 𑀯𑀲𑁆𑀲𑀺𑀓𑁂 𑀧𑀸𑀲𑀸𑀤𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀸𑀲𑁂 𑀦𑀺𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑁂𑀳𑀺 𑀢𑀽𑀭𑀺𑀬𑁂𑀳𑀺 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀫𑀸𑀦𑁄 𑀦 𑀳𑁂𑀝𑁆𑀞𑀸𑀧𑀸𑀲𑀸𑀤𑀁 𑀑𑀭𑁄𑀳𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀜𑁆𑀘𑀳𑀺 𑀓𑀸𑀫𑀕𑀼𑀡𑁂𑀳𑀺 𑀲𑀫𑀧𑁆𑀧𑀺𑀢𑀲𑁆𑀲 𑀲𑀫𑀗𑁆𑀕𑀻𑀪𑀽𑀢𑀲𑁆𑀲 𑀧𑀭𑀺𑀘𑀸𑀭𑀬𑀫𑀸𑀦𑀲𑁆𑀲 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀦𑀺𑀤𑁆𑀤𑀸 𑀑𑀓𑁆𑀓𑀫𑀺, 𑀧𑀭𑀺𑀚𑀦𑀲𑁆𑀲𑀧𑀺 𑀦𑀺𑀤𑁆𑀤𑀸 𑀑𑀓𑁆𑀓𑀫𑀺, 𑀲𑀩𑁆𑀩𑀭𑀢𑁆𑀢𑀺𑀬𑁄 𑀘 𑀢𑁂𑀮𑀧𑀤𑀻𑀧𑁄 𑀛𑀸𑀬𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀧𑀝𑀺𑀓𑀘𑁆𑀘𑁂𑀯 𑀧𑀩𑀼𑀚𑁆𑀛𑀺𑀢𑁆𑀯𑀸 𑀅𑀤𑁆𑀤𑀲 𑀲𑀓𑀁 𑀧𑀭𑀺𑀚𑀦𑀁 𑀲𑀼𑀧𑀦𑁆𑀢𑀁—  𑀅𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀓𑀘𑁆𑀙𑁂 𑀯𑀻𑀡𑀁, 𑀅𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀓𑀡𑁆𑀞𑁂 𑀫𑀼𑀤𑀺𑀗𑁆𑀕𑀁, 𑀅𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀸 𑀓𑀘𑁆𑀙𑁂 𑀆𑀍𑀅𑀫𑁆𑀩𑀭𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀓𑁂𑀲𑀺𑀓𑀁, 𑀅𑀜𑁆𑀜𑀁 𑀯𑀺𑀓𑁆𑀔𑁂𑀍𑀇𑀓𑀁, 𑀅𑀜𑁆𑀜𑀸 𑀯𑀺𑀧𑁆𑀧𑀮𑀧𑀦𑁆𑀢𑀺𑀬𑁄, 𑀳𑀢𑁆𑀣𑀧𑁆𑀧𑀢𑁆𑀢𑀁 𑀲𑀼𑀲𑀸𑀦𑀁 𑀫𑀜𑁆𑀜𑁂𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀆𑀤𑀻𑀦𑀯𑁄 𑀧𑀸𑀢𑀼𑀭𑀳𑁄𑀲𑀺, 𑀦𑀺𑀩𑁆𑀩𑀺𑀤𑀸𑀬 𑀘𑀺𑀢𑁆𑀢𑀁 𑀲𑀡𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  “𑀉𑀧𑀤𑁆𑀤𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀉𑀧𑀲𑁆𑀲𑀝𑁆𑀞𑀁 𑀯𑀢 𑀪𑁄”𑀢𑀺𑁇

108\. 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀲𑀼𑀯𑀡𑁆𑀡𑀧𑀸𑀤𑀼𑀓𑀸𑀬𑁄 𑀆𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀦𑀺𑀯𑁂𑀲𑀦𑀤𑁆𑀯𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑁆𑀯𑀸𑀭𑀁 𑀯𑀺𑀯𑀭𑀺𑀁𑀲𑀼—  “𑀫𑀸 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑁄𑀘𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀫𑀓𑀸𑀲𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀦𑀕𑀭𑀤𑁆𑀯𑀸𑀭𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀤𑁆𑀯𑀸𑀭𑀁 𑀯𑀺𑀯𑀭𑀺𑀁𑀲𑀼—  “𑀫𑀸 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀓𑁄𑀘𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑀫𑀓𑀸𑀲𑀺 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀬𑁂𑀦 𑀇𑀲𑀺𑀧𑀢𑀦𑀁 𑀫𑀺𑀕𑀤𑀸𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇

(26.)

109\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀧𑀘𑁆𑀘𑀽𑀲𑀲𑀫𑀬𑀁 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀬 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀘𑀗𑁆𑀓𑀫𑀢𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀘𑀗𑁆𑀓𑀫𑀸 𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—  “𑀉𑀧𑀤𑁆𑀤𑀼𑀢𑀁 𑀯𑀢 𑀪𑁄, 𑀉𑀧𑀲𑁆𑀲𑀝𑁆𑀞𑀁 𑀯𑀢 𑀪𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀔𑁄, 𑀬𑀲, 𑀅𑀦𑀼𑀧𑀤𑁆𑀤𑀼𑀢𑀁, 𑀇𑀤𑀁 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀝𑁆𑀞𑀁𑁇 𑀏𑀳𑀺, 𑀬𑀲, 𑀦𑀺𑀲𑀻𑀤, 𑀥𑀫𑁆𑀫𑀁 𑀢𑁂 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄—  “𑀇𑀤𑀁 𑀓𑀺𑀭 𑀅𑀦𑀼𑀧𑀤𑁆𑀤𑀼𑀢𑀁, 𑀇𑀤𑀁 𑀅𑀦𑀼𑀧𑀲𑁆𑀲𑀝𑁆𑀞𑀦𑁆”𑀢𑀺 𑀳𑀝𑁆𑀞𑁄 𑀉𑀤𑀕𑁆𑀕𑁄 𑀲𑀼𑀯𑀡𑁆𑀡𑀧𑀸𑀤𑀼𑀓𑀸𑀳𑀺 𑀑𑀭𑁄𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

110\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀔𑁄 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁, 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑀁, 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑀁, 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑀁, 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀁, 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑀁, 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇

(27.)

111\. 𑀅𑀣 𑀔𑁄 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀧𑀸𑀲𑀸𑀤𑀁 𑀅𑀪𑀺𑀭𑀼𑀳𑀺𑀢𑁆𑀯𑀸 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀅𑀧𑀲𑁆𑀲𑀦𑁆𑀢𑀻 𑀬𑁂𑀦 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀧𑀼𑀢𑁆𑀢𑁄 𑀢𑁂, 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀲𑁄 𑀦 𑀤𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀅𑀲𑁆𑀲𑀤𑀽𑀢𑁂 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀸𑀫𑀁𑀬𑁂𑀯 𑀬𑁂𑀦 𑀇𑀲𑀺𑀧𑀢𑀦𑀁 𑀫𑀺𑀕𑀤𑀸𑀬𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀲𑀼𑀯𑀡𑁆𑀡𑀧𑀸𑀤𑀼𑀓𑀸𑀦𑀁 𑀦𑀺𑀓𑁆𑀔𑁂𑀧𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀢𑀁𑀬𑁂𑀯 𑀅𑀦𑀼𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀤𑀽𑀭𑀢𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀇𑀤𑁆𑀥𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁂𑀬𑁆𑀬𑀁 𑀬𑀣𑀸 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀇𑀥 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀇𑀥 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀦 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀇𑀤𑁆𑀥𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑁂𑀲𑀺𑁇

112\. 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸”𑀢𑀺?

113\. “𑀢𑁂𑀦 𑀳𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀦𑀺𑀲𑀻𑀤, 𑀅𑀧𑁆𑀧𑁂𑀯 𑀦𑀸𑀫 𑀇𑀥 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀇𑀥 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸𑀲𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺—  “𑀇𑀥𑁂𑀯 𑀓𑀺𑀭𑀸𑀳𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀇𑀥 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀲𑁆𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺 𑀳𑀝𑁆𑀞𑁄 𑀉𑀤𑀕𑁆𑀕𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇

114\. 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀲𑁆𑀲 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁, 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑀁, 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑀁, 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑀁, 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀁, 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀁, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑀁, 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀲𑁂𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇

115\. 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑁄 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑁄 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑁄 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑁄 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑁄 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑁄 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑁄 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑁄 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀦𑀺𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑀺𑀢𑀁 𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀚𑁆𑀚𑁂𑀬𑁆𑀬, 𑀧𑀝𑀺𑀘𑁆𑀙𑀦𑁆𑀦𑀁 𑀯𑀸 𑀯𑀺𑀯𑀭𑁂𑀬𑁆𑀬, 𑀫𑀽𑀍𑀳𑀲𑁆𑀲 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑁂𑀬𑁆𑀬, 𑀅𑀦𑁆𑀥𑀓𑀸𑀭𑁂 𑀯𑀸 𑀢𑁂𑀮𑀧𑀚𑁆𑀚𑁄𑀢𑀁 𑀥𑀸𑀭𑁂𑀬𑁆𑀬—  ‘𑀘𑀓𑁆𑀔𑀼𑀫𑀦𑁆𑀢𑁄 𑀭𑀽𑀧𑀸𑀦𑀺 𑀤𑀓𑁆𑀔𑀦𑁆𑀢𑀻’𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯𑀁 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀧𑀓𑀸𑀲𑀺𑀢𑁄𑁇 𑀏𑀲𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫𑀺, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀓𑀁 𑀫𑀁 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀢𑀦𑁆”𑀢𑀺𑁇 𑀲𑁄𑀯 𑀮𑁄𑀓𑁂 𑀧𑀞𑀫𑀁 𑀉𑀧𑀸𑀲𑀓𑁄 𑀅𑀳𑁄𑀲𑀺 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑁄𑁇

116\. 𑀅𑀣 𑀔𑁄 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀺𑀢𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑁂 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀁 𑀬𑀣𑀸𑀯𑀺𑀤𑀺𑀢𑀁 𑀪𑀽𑀫𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀲𑀲𑁆𑀲 𑀔𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀧𑀺𑀢𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑁂 𑀤𑁂𑀲𑀺𑀬𑀫𑀸𑀦𑁂 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀁 𑀬𑀣𑀸𑀯𑀺𑀤𑀺𑀢𑀁 𑀪𑀽𑀫𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁𑁇 𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁, 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑁄; 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀢𑀁 𑀇𑀤𑁆𑀥𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀁 𑀇𑀤𑁆𑀥𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀲𑀺𑁇

117\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁 𑀤𑀺𑀲𑁆𑀯𑀸𑀦, 𑀬𑀲𑀁 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀸𑀢𑀸 𑀢𑁂, 𑀢𑀸𑀢 𑀬𑀲, 𑀧𑀭𑀺𑀤𑁂𑀯𑀲𑁄𑀓𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑀸, 𑀤𑁂𑀳𑀺 𑀫𑀸𑀢𑀼𑀬𑀸 𑀚𑀻𑀯𑀺𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀮𑁆𑀮𑁄𑀓𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑁂𑀝𑁆𑀞𑀺𑀁 𑀕𑀳𑀧𑀢𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀲𑀺, 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀲𑁆𑀲 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀯𑀺𑀤𑀺𑀢𑁄 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑀬𑀸, 𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀁 𑀬𑀣𑀸𑀯𑀺𑀤𑀺𑀢𑀁 𑀪𑀽𑀫𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁𑁇 𑀪𑀩𑁆𑀩𑁄 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑁄”𑀢𑀺? “𑀦𑁄 𑀳𑁂𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂”𑁇 “𑀬𑀲𑀲𑁆𑀲 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀜𑀸𑀡𑁂𑀦 𑀲𑁂𑀓𑁆𑀔𑁂𑀦 𑀤𑀲𑁆𑀲𑀦𑁂𑀦 𑀥𑀫𑁆𑀫𑁄 𑀤𑀺𑀝𑁆𑀞𑁄 𑀯𑀺𑀤𑀺𑀢𑁄 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑀬𑀸, 𑀢𑀲𑁆𑀲 𑀬𑀣𑀸𑀤𑀺𑀝𑁆𑀞𑀁 𑀬𑀣𑀸𑀯𑀺𑀤𑀺𑀢𑀁 𑀪𑀽𑀫𑀺𑀁 𑀧𑀘𑁆𑀘𑀯𑁂𑀓𑁆𑀔𑀦𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁𑁇 𑀅𑀪𑀩𑁆𑀩𑁄 𑀔𑁄, 𑀕𑀳𑀧𑀢𑀺, 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀳𑀻𑀦𑀸𑀬𑀸𑀯𑀢𑁆𑀢𑀺𑀢𑁆𑀯𑀸 𑀓𑀸𑀫𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀼𑀩𑁆𑀩𑁂 𑀅𑀕𑀸𑀭𑀺𑀓𑀪𑀽𑀢𑁄”𑀢𑀺𑁇 “𑀮𑀸𑀪𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲, 𑀲𑀼𑀮𑀤𑁆𑀥𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲, 𑀬𑀣𑀸 𑀬𑀲𑀲𑁆𑀲 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀁𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀅𑀚𑁆𑀚𑀢𑀦𑀸𑀬 𑀪𑀢𑁆𑀢𑀁 𑀬𑀲𑁂𑀦 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀧𑀘𑁆𑀙𑀸𑀲𑀫𑀡𑁂𑀦𑀸”𑀢𑀺𑁇 𑀅𑀥𑀺𑀯𑀸𑀲𑁂𑀲𑀺 𑀪𑀕𑀯𑀸 𑀢𑀼𑀡𑁆𑀳𑀻𑀪𑀸𑀯𑁂𑀦𑁇

118\. 𑀅𑀣 𑀔𑁄 𑀲𑁂𑀝𑁆𑀞𑀺 𑀕𑀳𑀧𑀢𑀺 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀯𑀸𑀲𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀘𑀺𑀭𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁂 𑀲𑁂𑀝𑁆𑀞𑀺𑀫𑁆𑀳𑀺 𑀕𑀳𑀧𑀢𑀺𑀫𑁆𑀳𑀺 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀳𑀺 𑀪𑀺𑀓𑁆𑀔𑀽”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀢𑁆𑀢 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

120\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀼𑀩𑁆𑀩𑀡𑁆𑀳𑀲𑀫𑀬𑀁 𑀦𑀺𑀯𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀫𑀸𑀤𑀸𑀬 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀬𑀲𑁂𑀦 𑀧𑀘𑁆𑀙𑀸𑀲𑀫𑀡𑁂𑀦 𑀬𑁂𑀦 𑀲𑁂𑀝𑁆𑀞𑀺𑀲𑁆𑀲 𑀕𑀳𑀧𑀢𑀺𑀲𑁆𑀲 𑀦𑀺𑀯𑁂𑀲𑀦𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁂 𑀆𑀲𑀦𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀬𑀲𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀘 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀘 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇

121\. 𑀢𑀸𑀲𑀁 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁, 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁, 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀢𑀸 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑀸, 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑀸, 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑀸, 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑀸, 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑀸, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺—  𑀤𑀼𑀓𑁆𑀔𑀁, 𑀲𑀫𑀼𑀤𑀬𑀁, 𑀦𑀺𑀭𑁄𑀥𑀁, 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀢𑀸𑀲𑀁 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑀸 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑀸 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑀸 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀁, 𑀪𑀦𑁆𑀢𑁂…𑀧𑁂…  𑀏𑀢𑀸 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀭𑀡𑀁 𑀕𑀘𑁆𑀙𑀸𑀫, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀜𑁆𑀘𑁇 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬𑁄 𑀦𑁄 𑀪𑀕𑀯𑀸 𑀥𑀸𑀭𑁂𑀢𑀼 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀸𑀡𑀼𑀧𑁂𑀢𑀸 𑀲𑀭𑀡𑀁 𑀕𑀢𑀸”𑀢𑀺𑁇 𑀢𑀸 𑀘 𑀮𑁄𑀓𑁂 𑀧𑀞𑀫𑀁 𑀉𑀧𑀸𑀲𑀺𑀓𑀸 𑀅𑀳𑁂𑀲𑀼𑀁 𑀢𑁂𑀯𑀸𑀘𑀺𑀓𑀸𑁇

122\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀬𑀲𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀘 𑀧𑀺𑀢𑀸 𑀘 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸 𑀘 𑀪𑀕𑀯𑀦𑁆𑀢𑀜𑁆𑀘 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀜𑁆𑀘 𑀬𑀲𑀁 𑀧𑀡𑀻𑀢𑁂𑀦 𑀔𑀸𑀤𑀦𑀻𑀬𑁂𑀦 𑀪𑁄𑀚𑀦𑀻𑀬𑁂𑀦 𑀲𑀳𑀢𑁆𑀣𑀸 𑀲𑀦𑁆𑀢𑀧𑁆𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸, 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀪𑀼𑀢𑁆𑀢𑀸𑀯𑀺𑀁 𑀑𑀦𑀻𑀢𑀧𑀢𑁆𑀢𑀧𑀸𑀡𑀺𑀁, 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀬𑀲𑀲𑁆𑀲 𑀫𑀸𑀢𑀭𑀜𑁆𑀘 𑀧𑀺𑀢𑀭𑀜𑁆𑀘 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀜𑁆𑀘 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀸𑀤𑀧𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀢𑁆𑀯𑀸 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀢𑁆𑀯𑀸 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇

123\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀬𑀲𑀲𑁆𑀲 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀕𑀺𑀳𑀺𑀲𑀳𑀸𑀬𑀓𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀲𑁂𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑁂𑀝𑁆𑀞𑀻𑀦𑀁 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀼𑀢𑁆𑀢𑀸—  𑀯𑀺𑀫𑀮𑁄, 𑀲𑀼𑀩𑀸𑀳𑀼, 𑀧𑀼𑀡𑁆𑀡𑀚𑀺, 𑀕𑀯𑀫𑁆𑀧𑀢𑀺—  𑀬𑀲𑁄 𑀓𑀺𑀭 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀢𑀺𑁇 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀦𑁂𑀲𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦 𑀳𑀺 𑀦𑀽𑀦 𑀲𑁄 𑀑𑀭𑀓𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄, 𑀦 𑀲𑀸 𑀑𑀭𑀓𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀬𑀢𑁆𑀣 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀬𑀲𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇

124\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄 𑀢𑁂 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀕𑀺𑀳𑀺𑀲𑀳𑀸𑀬𑀓𑁂 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀕𑀺𑀳𑀺𑀲𑀳𑀸𑀬𑀓𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀲𑁂𑀝𑁆𑀞𑀸𑀦𑀼𑀲𑁂𑀝𑁆𑀞𑀻𑀦𑀁 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀼𑀢𑁆𑀢𑀸—  𑀯𑀺𑀫𑀮𑁄, 𑀲𑀼𑀩𑀸𑀳𑀼, 𑀧𑀼𑀡𑁆𑀡𑀚𑀺, 𑀕𑀯𑀫𑁆𑀧𑀢𑀺𑁇 𑀇𑀫𑁂 𑀪𑀕𑀯𑀸 𑀑𑀯𑀤𑀢𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀽”𑀢𑀺𑁇

125\. 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑁂 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑁂 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁂, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀺𑀭𑁄𑀥𑀁 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀢𑁂𑀲𑀁 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑀸 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑀸 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀑𑀯𑀤𑀺 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑁇 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀑𑀯𑀤𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀏𑀓𑀸𑀤𑀲 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

127\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀬𑀲𑀲𑁆𑀲 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀢𑁆𑀢𑀸 𑀕𑀺𑀳𑀺𑀲𑀳𑀸𑀬𑀓𑀸 𑀚𑀸𑀦𑀧𑀤𑀸 𑀧𑀼𑀩𑁆𑀩𑀸𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀼𑀢𑁆𑀢𑀸—  𑀬𑀲𑁄 𑀓𑀺𑀭 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀢𑀺𑁇 𑀲𑀼𑀢𑁆𑀯𑀸𑀦 𑀦𑁂𑀲𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦 𑀳𑀺 𑀦𑀽𑀦 𑀲𑁄 𑀑𑀭𑀓𑁄 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁄, 𑀦 𑀲𑀸 𑀑𑀭𑀓𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀬𑀢𑁆𑀣 𑀬𑀲𑁄 𑀓𑀼𑀮𑀧𑀼𑀢𑁆𑀢𑁄 𑀓𑁂𑀲𑀫𑀲𑁆𑀲𑀼𑀁 𑀑𑀳𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀓𑀸𑀲𑀸𑀬𑀸𑀦𑀺 𑀯𑀢𑁆𑀣𑀸𑀦𑀺 𑀅𑀘𑁆𑀙𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀅𑀕𑀸𑀭𑀲𑁆𑀫𑀸 𑀅𑀦𑀕𑀸𑀭𑀺𑀬𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺𑁇 𑀢𑁂 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀬𑀲𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀁𑀲𑀼𑁇

128\. 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄 𑀢𑁂 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀢𑁆𑀢𑁂 𑀕𑀺𑀳𑀺𑀲𑀳𑀸𑀬𑀓𑁂 𑀆𑀤𑀸𑀬 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀬𑀲𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀫𑁂 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀜𑁆𑀜𑀸𑀲𑀫𑀢𑁆𑀢𑀸 𑀕𑀺𑀳𑀺𑀲𑀳𑀸𑀬𑀓𑀸 𑀚𑀸𑀦𑀧𑀤𑀸 𑀧𑀼𑀩𑁆𑀩𑀸𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀓𑀸𑀦𑀁 𑀓𑀼𑀮𑀸𑀦𑀁 𑀧𑀼𑀢𑁆𑀢𑀸𑁇 𑀇𑀫𑁂 𑀪𑀕𑀯𑀸 𑀑𑀯𑀤𑀢𑀼 𑀅𑀦𑀼𑀲𑀸𑀲𑀢𑀽”𑀢𑀺𑁇

129\. 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑁂 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑁂 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁂, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀺𑀭𑁄𑀥𑀁 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀢𑁂𑀲𑀁 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑀸 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑀸 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀑𑀯𑀤𑀺 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑁇 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀑𑀯𑀤𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀦𑀼𑀲𑀸𑀲𑀺𑀬𑀫𑀸𑀦𑀸𑀦𑀁 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀏𑀓𑀲𑀝𑁆𑀞𑀺 𑀮𑁄𑀓𑁂 𑀅𑀭𑀳𑀦𑁆𑀢𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇
