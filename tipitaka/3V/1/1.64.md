# 𑁧𑁇𑁬𑁪 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀦𑀺𑀲𑁆𑀲𑀬

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀦𑀺𑀲𑁆𑀲𑀬

(128.)

533\. “𑀢𑀸𑀯𑀤𑁂𑀯 𑀙𑀸𑀬𑀸 𑀫𑁂𑀢𑀩𑁆𑀩𑀸, 𑀉𑀢𑀼𑀧𑁆𑀧𑀫𑀸𑀡𑀁 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀁, 𑀤𑀺𑀯𑀲𑀪𑀸𑀕𑁄 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑁄, 𑀲𑀗𑁆𑀕𑀻𑀢𑀺 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀸, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀦𑀺𑀲𑁆𑀲𑀬𑀸 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀩𑁆𑀩𑀸—

534\. 𑀧𑀺𑀡𑁆𑀟𑀺𑀬𑀸𑀮𑁄𑀧𑀪𑁄𑀚𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑁇 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑀁, 𑀉𑀤𑁆𑀤𑁂𑀲𑀪𑀢𑁆𑀢𑀁, 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀁, 𑀲𑀮𑀸𑀓𑀪𑀢𑁆𑀢𑀁, 𑀧𑀓𑁆𑀔𑀺𑀓𑀁, 𑀉𑀧𑁄𑀲𑀣𑀺𑀓𑀁, 𑀧𑀸𑀝𑀺𑀧𑀤𑀺𑀓𑀁𑁇 (𑁧)

535\. 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑁇 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀔𑁄𑀫𑀁, 𑀓𑀧𑁆𑀧𑀸𑀲𑀺𑀓𑀁, 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀁, 𑀓𑀫𑁆𑀩𑀮𑀁, 𑀲𑀸𑀡𑀁, 𑀪𑀗𑁆𑀕𑀁𑁇 (𑁨)

536\. 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑁇 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀯𑀺𑀳𑀸𑀭𑁄, 𑀅𑀟𑁆𑀠𑀬𑁄𑀕𑁄, 𑀧𑀸𑀲𑀸𑀤𑁄, 𑀳𑀫𑁆𑀫𑀺𑀬𑀁, 𑀕𑀼𑀳𑀸𑁇 (𑁩)

537\. 𑀧𑀽𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸𑁇 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄𑁇 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀲𑀧𑁆𑀧𑀺, 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀢𑁂𑀮𑀁, 𑀫𑀥𑀼, 𑀨𑀸𑀡𑀺𑀢𑀦𑁆”𑀢𑀺𑁇 (𑁪)

---

538\. 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀦𑀺𑀲𑁆𑀲𑀬𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
