# 𑁧𑁇𑁩𑁮 𑀓𑀡𑁆𑀝𑀓𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀓𑀡𑁆𑀝𑀓𑀯𑀢𑁆𑀣𑀼

(101.)

442\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀤𑁆𑀯𑁂 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀳𑁄𑀦𑁆𑀢𑀺—  𑀓𑀡𑁆𑀝𑀓𑁄 𑀘 𑀫𑀳𑀓𑁄 𑀘𑁇 𑀢𑁂 𑀅𑀜𑁆𑀜𑀫𑀜𑁆𑀜𑀁 𑀤𑀽𑀲𑁂𑀲𑀼𑀁𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀓𑁂𑀦 𑀤𑁆𑀯𑁂 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀉𑀧𑀝𑁆𑀞𑀸𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
