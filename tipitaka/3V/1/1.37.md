# 𑁧𑁇𑁩𑁭 𑀅𑀳𑀺𑀯𑀸𑀢𑀓𑀭𑁄𑀕𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀳𑀺𑀯𑀸𑀢𑀓𑀭𑁄𑀕𑀯𑀢𑁆𑀣𑀼

(100.)

440\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀓𑀼𑀮𑀁 𑀅𑀳𑀺𑀯𑀸𑀢𑀓𑀭𑁄𑀕𑁂𑀦 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀢𑀲𑁆𑀲 𑀧𑀺𑀢𑀸𑀧𑀼𑀢𑁆𑀢𑀓𑀸 𑀲𑁂𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀢𑁄𑀯 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀤𑀸𑀭𑀓𑁄 𑀧𑀺𑀢𑀼𑀦𑁄 𑀪𑀺𑀓𑁆𑀔𑀸𑀬 𑀤𑀺𑀦𑁆𑀦𑀸𑀬 𑀉𑀧𑀥𑀸𑀯𑀺𑀢𑁆𑀯𑀸 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺, 𑀢𑀸𑀢, 𑀤𑁂𑀳𑀺; 𑀫𑀬𑁆𑀳𑀫𑁆𑀧𑀺, 𑀢𑀸𑀢, 𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀅𑀬𑀫𑁆𑀧𑀺 𑀤𑀸𑀭𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀚𑀸𑀢𑁄”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀊𑀦𑀧𑀦𑁆𑀦𑀭𑀲𑀯𑀲𑁆𑀲𑁄 𑀤𑀸𑀭𑀓𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

441\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀓𑀼𑀮𑀁 𑀲𑀤𑁆𑀥𑀁 𑀧𑀲𑀦𑁆𑀦𑀁 𑀅𑀳𑀺𑀯𑀸𑀢𑀓𑀭𑁄𑀕𑁂𑀦 𑀓𑀸𑀮𑀗𑁆𑀓𑀢𑀁 𑀳𑁄𑀢𑀺, 𑀤𑁆𑀯𑁂 𑀘 𑀤𑀸𑀭𑀓𑀸 𑀲𑁂𑀲𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀧𑁄𑀭𑀸𑀡𑀓𑁂𑀦 𑀆𑀘𑀺𑀡𑁆𑀡𑀓𑀧𑁆𑀧𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀥𑀸𑀯𑀦𑁆𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀅𑀧𑀲𑀸𑀤𑀺𑀬𑀫𑀸𑀦𑀸 𑀭𑁄𑀤𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀊𑀦𑀧𑀦𑁆𑀦𑀭𑀲𑀯𑀲𑁆𑀲𑁄 𑀤𑀸𑀭𑀓𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄’𑀢𑀺𑁇 𑀇𑀫𑁂 𑀘 𑀤𑀸𑀭𑀓𑀸 𑀊𑀦𑀧𑀦𑁆𑀦𑀭𑀲𑀯𑀲𑁆𑀲𑀸𑁇 𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀉𑀧𑀸𑀬𑁂𑀦 𑀇𑀫𑁂 𑀤𑀸𑀭𑀓𑀸 𑀦 𑀯𑀺𑀦𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 “𑀉𑀲𑁆𑀲𑀳𑀦𑁆𑀢𑀺 𑀧𑀦 𑀢𑁂, 𑀆𑀦𑀦𑁆𑀤, 𑀤𑀸𑀭𑀓𑀸 𑀓𑀸𑀓𑁂 𑀉𑀟𑁆𑀟𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺? “𑀉𑀲𑁆𑀲𑀳𑀦𑁆𑀢𑀺, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀊𑀦𑀧𑀦𑁆𑀦𑀭𑀲𑀯𑀲𑁆𑀲𑀁 𑀤𑀸𑀭𑀓𑀁 𑀓𑀸𑀓𑀼𑀟𑁆𑀟𑁂𑀧𑀓𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
