# 𑁧𑁇𑁧𑁭 𑀧𑀡𑀸𑀫𑀺𑀢𑀓𑀣𑀸

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀡𑀸𑀫𑀺𑀢𑀓𑀣𑀸

(68.)

286\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀸 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀲𑀼 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀸 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀲𑀼 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀸 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀲𑀼 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀸 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀲𑀼 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻𑀢𑀺…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀺 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀦 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 𑀦𑁂𑀯 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀺𑁇

287\. 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄—  ‘𑀧𑀡𑀸𑀫𑁂𑀫𑀺 𑀢𑀦𑁆’𑀢𑀺 𑀯𑀸, ‘𑀫𑀸𑀬𑀺𑀥 𑀧𑀝𑀺𑀓𑁆𑀓𑀫𑀻’𑀢𑀺 𑀯𑀸, ‘𑀦𑀻𑀳𑀭 𑀢𑁂 𑀧𑀢𑁆𑀢𑀘𑀻𑀯𑀭𑀦𑁆’𑀢𑀺 𑀯𑀸, ‘𑀦𑀸𑀳𑀁 𑀢𑀬𑀸 𑀉𑀧𑀝𑁆𑀞𑀸𑀢𑀩𑁆𑀩𑁄’𑀢𑀺 𑀯𑀸, 𑀓𑀸𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀧𑀡𑀸𑀫𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄; 𑀦 𑀓𑀸𑀬𑁂𑀦 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀓𑀸𑀬𑁂𑀦 𑀯𑀸𑀘𑀸𑀬 𑀯𑀺𑀜𑁆𑀜𑀸𑀧𑁂𑀢𑀺, 𑀦 𑀧𑀡𑀸𑀫𑀺𑀢𑁄 𑀳𑁄𑀢𑀺 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄”𑀢𑀺𑁇

288\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀸 𑀧𑀡𑀸𑀫𑀺𑀢𑀸 𑀦 𑀔𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀫𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀦𑁂𑀯 𑀔𑀫𑀸𑀧𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑀸𑀫𑀺𑀢𑁂𑀦 𑀦 𑀔𑀫𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀦 𑀔𑀫𑀸𑀧𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

289\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸 𑀔𑀫𑀸𑀧𑀺𑀬𑀫𑀸𑀦𑀸 𑀦 𑀔𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀫𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀦𑁂𑀯 𑀔𑀫𑀦𑁆𑀢𑀺𑁇 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀸 𑀧𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑀧𑀺 𑀯𑀺𑀩𑁆𑀪𑀫𑀦𑁆𑀢𑀺𑀧𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑁂𑀲𑀼𑀧𑀺 𑀲𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀔𑀫𑀸𑀧𑀺𑀬𑀫𑀸𑀦𑁂𑀦 𑀦 𑀔𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀦 𑀔𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

290\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀸 𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑀺, 𑀅𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑀁 𑀦 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑁄 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀡𑀸𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀦 𑀘, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑀫𑁆𑀫𑀸𑀯𑀢𑁆𑀢𑀦𑁆𑀢𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸𑀢𑀺𑁇

291\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀺 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇

292\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀺 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀦 𑀧𑀡𑀸𑀫𑁂𑀢𑀩𑁆𑀩𑁄𑁇

293\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀅𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀺 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀅𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇

294\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀦𑀸𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀺 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑁄 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑁄 𑀦𑀸𑀮𑀁 𑀧𑀡𑀸𑀫𑁂𑀢𑀼𑀁𑁇

295\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀺 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀦𑀸𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀁 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇

296\. 𑀧𑀜𑁆𑀘𑀳𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀁 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺𑁇 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀫𑁆𑀳𑀺 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀁 𑀧𑁂𑀫𑀁 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀧𑀲𑀸𑀤𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀳𑀺𑀭𑀻 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑁄 𑀕𑀸𑀭𑀯𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀥𑀺𑀫𑀢𑁆𑀢𑀸 𑀪𑀸𑀯𑀦𑀸 𑀳𑁄𑀢𑀺—  𑀇𑀫𑁂𑀳𑀺 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀳𑀗𑁆𑀕𑁂𑀳𑀺 𑀲𑀫𑀦𑁆𑀦𑀸𑀕𑀢𑀁 𑀲𑀤𑁆𑀥𑀺𑀯𑀺𑀳𑀸𑀭𑀺𑀓𑀁 𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁄 𑀲𑀸𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀺, 𑀅𑀧𑁆𑀧𑀡𑀸𑀫𑁂𑀦𑁆𑀢𑁄 𑀅𑀦𑀢𑀺𑀲𑀸𑀭𑁄 𑀳𑁄𑀢𑀻”𑀢𑀺𑁇

(69.)

297\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀇𑀘𑁆𑀙𑀺𑀁𑀲𑀼 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼𑀁𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀓𑀺𑀲𑁄 𑀅𑀳𑁄𑀲𑀺 𑀮𑀽𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁄 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁄𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀓𑀺𑀲𑀁 𑀮𑀽𑀔𑀁 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑀁 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑀁 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀓𑀺𑀲𑁄 𑀮𑀽𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁄 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁄”𑀢𑀺? “𑀏𑀲𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀇𑀘𑁆𑀙𑀺𑀁𑀲𑀼 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀼𑀁𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀅𑀮𑀪𑀫𑀸𑀦𑁄 𑀓𑀺𑀲𑁄 𑀮𑀽𑀔𑁄 𑀤𑀼𑀩𑁆𑀩𑀡𑁆𑀡𑁄 𑀉𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀧𑁆𑀧𑀡𑁆𑀟𑀼𑀓𑀚𑀸𑀢𑁄 𑀥𑀫𑀦𑀺𑀲𑀦𑁆𑀣𑀢𑀕𑀢𑁆𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀓𑁄 𑀦𑀼 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀲𑀭𑀲𑀻”𑀢𑀺? 𑀏𑀯𑀁 𑀯𑀼𑀢𑁆𑀢𑁂, 𑀆𑀬𑀲𑁆𑀫𑀸 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀅𑀳𑀁 𑀔𑁄, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀲𑀭𑀲𑀻”𑀢𑀺? “𑀇𑀥 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀦𑁆𑀢𑀲𑁆𑀲 𑀓𑀝𑀘𑁆𑀙𑀼𑀪𑀺𑀓𑁆𑀔𑀁 𑀤𑀸𑀧𑁂𑀲𑀺𑁇 𑀇𑀫𑀁 𑀔𑁄 𑀅𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀅𑀥𑀺𑀓𑀸𑀭𑀁 𑀲𑀭𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀲𑀸𑀥𑀼 𑀲𑀸𑀥𑀼, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀓𑀢𑀜𑁆𑀜𑀼𑀦𑁄 𑀳𑀺, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀲𑀧𑁆𑀧𑀼𑀭𑀺𑀲𑀸 𑀓𑀢𑀯𑁂𑀤𑀺𑀦𑁄𑁇 𑀢𑁂𑀦 𑀳𑀺 𑀢𑁆𑀯𑀁, 𑀲𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀢, 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀳𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀳𑀻”𑀢𑀺𑁇 “𑀓𑀣𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀫𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀫𑀻”𑀢𑀺?

298\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀬𑀸 𑀲𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀬𑀸 𑀢𑀻𑀳𑀺 𑀲𑀭𑀡𑀕𑀫𑀦𑁂𑀳𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀸, 𑀢𑀁 𑀅𑀚𑁆𑀚𑀢𑀕𑁆𑀕𑁂 𑀧𑀝𑀺𑀓𑁆𑀔𑀺𑀧𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀜𑀢𑁆𑀢𑀺𑀘𑀢𑀼𑀢𑁆𑀣𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(70.)

299\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

300\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

301\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

302\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺—  𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

303\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(71.)

304\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀸 𑀅𑀦𑀸𑀘𑀸𑀭𑀁 𑀆𑀘𑀭𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀸𑀯𑀼𑀲𑁄, 𑀏𑀯𑀭𑀽𑀧𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀦𑁂𑀢𑀁 𑀓𑀧𑁆𑀧𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀦𑁂𑀯𑀸𑀳𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁂 𑀬𑀸𑀘𑀺𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀣 𑀫𑀦𑁆𑀢𑀺𑁇 𑀓𑀺𑀲𑁆𑀲 𑀫𑀁 𑀢𑀼𑀫𑁆𑀳𑁂 𑀅𑀬𑀸𑀘𑀺𑀢𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑀺𑀢𑁆𑀣𑀸”𑀢𑀺?

305\. 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀬𑀸𑀘𑀺𑀢𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀘𑀺𑀢𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀼𑀁𑁇 𑀏𑀯𑀜𑁆𑀘 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁂𑀦 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀓𑀁𑀲𑀁 𑀉𑀢𑁆𑀢𑀭𑀸𑀲𑀗𑁆𑀕𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀧𑀸𑀤𑁂 𑀯𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀉𑀓𑁆𑀓𑀼𑀝𑀺𑀓𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀚𑀮𑀺𑀁 𑀧𑀕𑁆𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀲𑁆𑀲 𑀯𑀘𑀦𑀻𑀬𑁄—  ‘𑀲𑀁𑀖𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀸𑀫𑀺, 𑀉𑀮𑁆𑀮𑀼𑀫𑁆𑀧𑀢𑀼 𑀫𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄 𑀅𑀦𑀼𑀓𑀫𑁆𑀧𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑀸’𑀢𑀺𑁇 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀬𑀸𑀘𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀲𑀁𑀖𑁄 𑀜𑀸𑀧𑁂𑀢𑀩𑁆𑀩𑁄—

(72.)

306\. ‘𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀤𑀺 𑀲𑀁𑀖𑀲𑁆𑀲 𑀧𑀢𑁆𑀢𑀓𑀮𑁆𑀮𑀁, 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀏𑀲𑀸 𑀜𑀢𑁆𑀢𑀺𑁇

307\. 𑀲𑀼𑀡𑀸𑀢𑀼 𑀫𑁂, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀁𑀖𑁄𑁇 𑀅𑀬𑀁 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄𑁇 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀲𑀁𑀖𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀬𑀸𑀘𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀲𑀁𑀖𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀬𑀲𑁆𑀲𑀸𑀬𑀲𑁆𑀫𑀢𑁄 𑀔𑀫𑀢𑀺 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦, 𑀲𑁄 𑀢𑀼𑀡𑁆𑀳𑀲𑁆𑀲; 𑀬𑀲𑁆𑀲 𑀦𑀓𑁆𑀔𑀫𑀢𑀺, 𑀲𑁄 𑀪𑀸𑀲𑁂𑀬𑁆𑀬𑁇

308\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂…  𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀤𑀸𑀫𑀺…𑀧𑁂… 𑁇

309\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀲𑀁𑀖𑁂𑀦 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦𑁇 𑀔𑀫𑀢𑀺 𑀲𑀁𑀖𑀲𑁆𑀲, 𑀢𑀲𑁆𑀫𑀸 𑀢𑀼𑀡𑁆𑀳𑀻, 𑀏𑀯𑀫𑁂𑀢𑀁 𑀥𑀸𑀭𑀬𑀸𑀫𑀻’”𑀢𑀺𑁇

(73.)

310\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀕𑀳𑁂 𑀧𑀡𑀻𑀢𑀸𑀦𑀁 𑀪𑀢𑁆𑀢𑀸𑀦𑀁 𑀪𑀢𑁆𑀢𑀧𑀝𑀺𑀧𑀸𑀝𑀺 𑀅𑀝𑁆𑀞𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑁂 𑀔𑁄 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀲𑀼𑀔𑀲𑀻𑀮𑀸 𑀲𑀼𑀔𑀲𑀫𑀸𑀘𑀸𑀭𑀸, 𑀲𑀼𑀪𑁄𑀚𑀦𑀸𑀦𑀺 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀦𑀺𑀯𑀸𑀢𑁂𑀲𑀼 𑀲𑀬𑀦𑁂𑀲𑀼 𑀲𑀬𑀦𑁆𑀢𑀺𑁇 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑁇 𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀼𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁂 𑀪𑀢𑁆𑀢𑀧𑀝𑀺𑀧𑀸𑀝𑀺 𑀔𑀻𑀬𑀺𑀢𑁆𑀣𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀏𑀳𑀺 𑀤𑀸𑀦𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀦𑀸𑀳𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀢𑀁𑀓𑀸𑀭𑀡𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀸𑀬 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻𑀢𑀺𑁇 𑀲𑀘𑁂 𑀫𑁂 𑀤𑀲𑁆𑀲𑀣 𑀪𑀼𑀜𑁆𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀦𑁄 𑀘𑁂 𑀫𑁂 𑀤𑀲𑁆𑀲𑀣 𑀯𑀺𑀩𑁆𑀪𑀫𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀓𑀺𑀁 𑀧𑀦 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀉𑀤𑀭𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺? “𑀏𑀯𑀫𑀸𑀯𑀼𑀲𑁄”𑀢𑀺𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀼 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀉𑀤𑀭𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇

311\. 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀪𑀺𑀓𑁆𑀔𑀼, 𑀉𑀤𑀭𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀏𑀯𑀁 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁂 𑀥𑀫𑁆𑀫𑀯𑀺𑀦𑀬𑁂 𑀉𑀤𑀭𑀲𑁆𑀲 𑀓𑀸𑀭𑀡𑀸 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬 𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀪𑀺𑀬𑁆𑀬𑁄𑀪𑀸𑀯𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑁂𑀦 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀦𑀺𑀲𑁆𑀲𑀬𑁂 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀼𑀁—  𑀧𑀺𑀡𑁆𑀟𑀺𑀬𑀸𑀮𑁄𑀧𑀪𑁄𑀚𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄; 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀲𑀁𑀖𑀪𑀢𑁆𑀢𑀁, 𑀉𑀤𑁆𑀤𑁂𑀲𑀪𑀢𑁆𑀢𑀁, 𑀦𑀺𑀫𑀦𑁆𑀢𑀦𑀁, 𑀲𑀮𑀸𑀓𑀪𑀢𑁆𑀢𑀁, 𑀧𑀓𑁆𑀔𑀺𑀓𑀁, 𑀉𑀧𑁄𑀲𑀣𑀺𑀓𑀁, 𑀧𑀸𑀝𑀺𑀧𑀤𑀺𑀓𑀁𑁇 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀘𑀻𑀯𑀭𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄; 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀔𑁄𑀫𑀁, 𑀓𑀧𑁆𑀧𑀸𑀲𑀺𑀓𑀁, 𑀓𑁄𑀲𑁂𑀬𑁆𑀬𑀁, 𑀓𑀫𑁆𑀩𑀮𑀁, 𑀲𑀸𑀡𑀁, 𑀪𑀗𑁆𑀕𑀁𑁇 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄; 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀯𑀺𑀳𑀸𑀭𑁄, 𑀅𑀟𑁆𑀠𑀬𑁄𑀕𑁄, 𑀧𑀸𑀲𑀸𑀤𑁄, 𑀳𑀫𑁆𑀫𑀺𑀬𑀁, 𑀕𑀼𑀳𑀸𑁇 𑀧𑀽𑀢𑀺𑀫𑀼𑀢𑁆𑀢𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀦𑀺𑀲𑁆𑀲𑀸𑀬 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀸, 𑀢𑀢𑁆𑀣 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀉𑀲𑁆𑀲𑀸𑀳𑁄 𑀓𑀭𑀡𑀻𑀬𑁄; 𑀅𑀢𑀺𑀭𑁂𑀓𑀮𑀸𑀪𑁄—  𑀲𑀧𑁆𑀧𑀺, 𑀦𑀯𑀦𑀻𑀢𑀁, 𑀢𑁂𑀮𑀁, 𑀫𑀥𑀼, 𑀨𑀸𑀡𑀺𑀢𑀦𑁆”𑀢𑀺𑁇
