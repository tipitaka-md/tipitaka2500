# 𑁧𑁇𑁧𑁨 𑀉𑀭𑀼𑀯𑁂𑀮𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀉𑀭𑀼𑀯𑁂𑀮𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀓𑀣𑀸

(37.)

155\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀉𑀭𑀼𑀯𑁂𑀮𑀸 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀭𑀼𑀯𑁂𑀮𑀸𑀬𑀁 𑀢𑀬𑁄 𑀚𑀝𑀺𑀮𑀸 𑀧𑀝𑀺𑀯𑀲𑀦𑁆𑀢𑀺—  𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄, 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑁄, 𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑁄𑀢𑀺𑁇 𑀢𑁂𑀲𑀼 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀚𑀝𑀺𑀮𑀲𑀢𑀸𑀦𑀁 𑀦𑀸𑀬𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀦𑀸𑀬𑀓𑁄 𑀅𑀕𑁆𑀕𑁄 𑀧𑀫𑀼𑀔𑁄 𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀺𑀡𑁆𑀡𑀁 𑀚𑀝𑀺𑀮𑀲𑀢𑀸𑀦𑀁 𑀦𑀸𑀬𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀦𑀸𑀬𑀓𑁄 𑀅𑀕𑁆𑀕𑁄 𑀧𑀫𑀼𑀔𑁄 𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀤𑁆𑀯𑀺𑀦𑁆𑀦𑀁 𑀚𑀝𑀺𑀮𑀲𑀢𑀸𑀦𑀁 𑀦𑀸𑀬𑀓𑁄 𑀳𑁄𑀢𑀺, 𑀯𑀺𑀦𑀸𑀬𑀓𑁄 𑀅𑀕𑁆𑀕𑁄 𑀧𑀫𑀼𑀔𑁄 𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑁄𑁇

156\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀬𑁂𑀦 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀕𑀭𑀼, 𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫 𑀏𑀓𑀭𑀢𑁆𑀢𑀁 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂”𑀢𑀺? “𑀦 𑀔𑁄 𑀫𑁂, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀕𑀭𑀼, 𑀘𑀡𑁆𑀟𑁂𑀢𑁆𑀣 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀆𑀲𑀺𑀯𑀺𑀲𑁄 𑀖𑁄𑀭𑀯𑀺𑀲𑁄, 𑀲𑁄 𑀢𑀁 𑀫𑀸 𑀯𑀺𑀳𑁂𑀞𑁂𑀲𑀻”𑀢𑀺𑁇

157\. 𑀤𑀼𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀕𑀭𑀼, 𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫 𑀏𑀓𑀭𑀢𑁆𑀢𑀁 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂”𑀢𑀺? “𑀦 𑀔𑁄 𑀫𑁂, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀕𑀭𑀼, 𑀘𑀡𑁆𑀟𑁂𑀢𑁆𑀣 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀆𑀲𑀺𑀯𑀺𑀲𑁄 𑀖𑁄𑀭𑀯𑀺𑀲𑁄, 𑀲𑁄 𑀢𑀁 𑀫𑀸 𑀯𑀺𑀳𑁂𑀞𑁂𑀲𑀻”𑀢𑀺𑁇

158\. 𑀢𑀢𑀺𑀬𑀫𑁆𑀧𑀺 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀘𑁂 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀕𑀭𑀼, 𑀯𑀲𑁂𑀬𑁆𑀬𑀸𑀫 𑀏𑀓𑀭𑀢𑁆𑀢𑀁 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂”𑀢𑀺? “𑀦 𑀔𑁄 𑀫𑁂, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀕𑀭𑀼, 𑀘𑀡𑁆𑀟𑁂𑀢𑁆𑀣 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀆𑀲𑀺𑀯𑀺𑀲𑁄 𑀖𑁄𑀭𑀯𑀺𑀲𑁄, 𑀲𑁄 𑀢𑀁 𑀫𑀸 𑀯𑀺𑀳𑁂𑀞𑁂𑀲𑀻”𑀢𑀺𑁇 “𑀅𑀧𑁆𑀧𑁂𑀯 𑀫𑀁 𑀦 𑀯𑀺𑀳𑁂𑀞𑁂𑀬𑁆𑀬, 𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀳𑀺 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀦𑁆”𑀢𑀺𑁇 “𑀯𑀺𑀳𑀭, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀬𑀣𑀸𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀁 𑀧𑀯𑀺𑀲𑀺𑀢𑁆𑀯𑀸 𑀢𑀺𑀡𑀲𑀦𑁆𑀣𑀸𑀭𑀓𑀁 𑀧𑀜𑁆𑀜𑀧𑁂𑀢𑁆𑀯𑀸 𑀦𑀺𑀲𑀻𑀤𑀺 𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑀁 𑀆𑀪𑀼𑀚𑀺𑀢𑁆𑀯𑀸 𑀉𑀚𑀼𑀁 𑀓𑀸𑀬𑀁 𑀧𑀡𑀺𑀥𑀸𑀬 𑀧𑀭𑀺𑀫𑀼𑀔𑀁 𑀲𑀢𑀺𑀁 𑀉𑀧𑀝𑁆𑀞𑀧𑁂𑀢𑁆𑀯𑀸𑁇

(38.)

159\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀲𑁄 𑀦𑀸𑀕𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀧𑀥𑀽𑀧𑀸𑀬𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀲𑁆𑀲 𑀦𑀸𑀕𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀳𑀘𑁆𑀘 𑀙𑀯𑀺𑀜𑁆𑀘 𑀘𑀫𑁆𑀫𑀜𑁆𑀘 𑀫𑀁𑀲𑀜𑁆𑀘 𑀦𑁆𑀳𑀸𑀭𑀼𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀜𑁆𑀘 𑀢𑁂𑀚𑀲𑀸 𑀢𑁂𑀚𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀣𑀸𑀭𑀽𑀧𑀁 𑀇𑀤𑁆𑀥𑀸𑀪𑀺𑀲𑀗𑁆𑀔𑀸𑀭𑀁 𑀅𑀪𑀺𑀲𑀗𑁆𑀔𑀭𑀺𑀢𑁆𑀯𑀸 𑀧𑀥𑀽𑀧𑀸𑀬𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀦𑀸𑀕𑁄 𑀫𑀓𑁆𑀔𑀁 𑀅𑀲𑀳𑀫𑀸𑀦𑁄 𑀧𑀚𑁆𑀚𑀮𑀺𑁇 𑀪𑀕𑀯𑀸𑀧𑀺 𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀁 𑀲𑀫𑀸𑀧𑀚𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀧𑀚𑁆𑀚𑀮𑀺𑁇 𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀸𑀦𑀁 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀁 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀯𑀺𑀬 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀁 𑀧𑀭𑀺𑀯𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀯𑀢 𑀪𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀦𑀸𑀕𑁂𑀦 𑀯𑀺𑀳𑁂𑀞𑀺𑀬𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀲𑁆𑀲 𑀦𑀸𑀕𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀳𑀘𑁆𑀘 𑀙𑀯𑀺𑀜𑁆𑀘 𑀘𑀫𑁆𑀫𑀜𑁆𑀘 𑀫𑀁𑀲𑀜𑁆𑀘 𑀦𑁆𑀳𑀸𑀭𑀼𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀺𑀜𑁆𑀘 𑀅𑀝𑁆𑀞𑀺𑀫𑀺𑀜𑁆𑀚𑀜𑁆𑀘 𑀢𑁂𑀚𑀲𑀸 𑀢𑁂𑀚𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀧𑀢𑁆𑀢𑁂 𑀧𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑁂𑀲𑀺—  “𑀅𑀬𑀁 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀦𑀸𑀕𑁄 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑁄 𑀅𑀲𑁆𑀲 𑀢𑁂𑀚𑀲𑀸 𑀢𑁂𑀚𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀘𑀡𑁆𑀟𑀲𑁆𑀲 𑀦𑀸𑀕𑀭𑀸𑀚𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀺𑀫𑀢𑁄 𑀆𑀲𑀺𑀯𑀺𑀲𑀲𑁆𑀲 𑀖𑁄𑀭𑀯𑀺𑀲𑀲𑁆𑀲 𑀢𑁂𑀚𑀲𑀸 𑀢𑁂𑀚𑀁 𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇

(39.)

160\. _𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀸𑀬𑀁 𑀪𑀕𑀯𑀸,_  
_𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀅𑀯𑁄𑀘;_  
_“𑀲𑀘𑁂 𑀢𑁂 𑀓𑀲𑁆𑀲𑀧 𑀅𑀕𑀭𑀼,_  
_𑀯𑀺𑀳𑀭𑁂𑀫𑀼 𑀅𑀚𑁆𑀚𑀡𑁆𑀳𑁄 𑀅𑀕𑁆𑀕𑀺𑀲𑀸𑀮𑀫𑁆𑀳𑀻”𑀢𑀺𑁇_  


161\. _“𑀦 𑀔𑁄 𑀫𑁂 𑀫𑀳𑀸𑀲𑀫𑀡 𑀕𑀭𑀼,_  
_𑀨𑀸𑀲𑀼𑀓𑀸𑀫𑁄𑀯 𑀢𑀁 𑀦𑀺𑀯𑀸𑀭𑁂𑀫𑀺;_  
_𑀘𑀡𑁆𑀟𑁂𑀢𑁆𑀣 𑀦𑀸𑀕𑀭𑀸𑀚𑀸,_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀸 𑀆𑀲𑀺𑀯𑀺𑀲𑁄 𑀖𑁄𑀭𑀯𑀺𑀲𑁄;_  
_𑀲𑁄 𑀢𑀁 𑀫𑀸 𑀯𑀺𑀳𑁂𑀞𑁂𑀲𑀻”𑀢𑀺𑁇_  


162\. _“𑀅𑀧𑁆𑀧𑁂𑀯 𑀫𑀁 𑀦 𑀯𑀺𑀳𑁂𑀞𑁂𑀬𑁆𑀬,_  
_𑀇𑀗𑁆𑀖 𑀢𑁆𑀯𑀁 𑀓𑀲𑁆𑀲𑀧 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀳𑀺 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀦𑁆”𑀢𑀺;_  
_𑀤𑀺𑀦𑁆𑀦𑀦𑁆𑀢𑀺 𑀦𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀪𑀻𑀢𑁄 𑀧𑀸𑀯𑀺𑀲𑀺 𑀪𑀬𑀫𑀢𑀻𑀢𑁄𑁇_  


163\. _𑀤𑀺𑀲𑁆𑀯𑀸 𑀇𑀲𑀺𑀁 𑀧𑀯𑀺𑀝𑁆𑀞𑀁,_  
_𑀅𑀳𑀺𑀦𑀸𑀕𑁄 𑀤𑀼𑀫𑁆𑀫𑀦𑁄 𑀧𑀥𑀽𑀧𑀸𑀬𑀺;_  
_𑀲𑀼𑀫𑀦𑀫𑀦𑀲𑁄 𑀅𑀥𑀺𑀫𑀦𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀦𑀸𑀕𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀧𑀥𑀽𑀧𑀸𑀬𑀺𑁇_  


164\. _𑀫𑀓𑁆𑀔𑀜𑁆𑀘 𑀅𑀲𑀳𑀫𑀸𑀦𑁄,_  
_𑀅𑀳𑀺𑀦𑀸𑀕𑁄 𑀧𑀸𑀯𑀓𑁄𑀯 𑀧𑀚𑁆𑀚𑀮𑀺;_  
_𑀢𑁂𑀚𑁄𑀥𑀸𑀢𑀼𑀲𑀼𑀓𑀼𑀲𑀮𑁄,_  
_𑀫𑀦𑀼𑀲𑁆𑀲𑀦𑀸𑀕𑁄𑀧𑀺 𑀢𑀢𑁆𑀣 𑀧𑀚𑁆𑀚𑀮𑀺𑁇_  


165\. _𑀉𑀪𑀺𑀦𑁆𑀦𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀸𑀦𑀁,_  
_𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑀁 𑀆𑀤𑀺𑀢𑁆𑀢𑀁 𑀳𑁄𑀢𑀺 𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀮𑀺𑀢𑀁 𑀲𑀚𑁄𑀢𑀺𑀪𑀽𑀢𑀁;_  
_𑀉𑀤𑀺𑀘𑁆𑀙𑀭𑁂 𑀚𑀝𑀺𑀮𑀸,_  
_“𑀅𑀪𑀺𑀭𑀽𑀧𑁄 𑀯𑀢 𑀪𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄;_  
_𑀦𑀸𑀕𑁂𑀦 𑀯𑀺𑀳𑁂𑀞𑀺𑀬𑀢𑀻”𑀢𑀺 𑀪𑀡𑀦𑁆𑀢𑀺𑁇_  


166\. _𑀅𑀣 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦,_  
_𑀳𑀢𑀸 𑀦𑀸𑀕𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀺𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺;_  
_𑀇𑀤𑁆𑀥𑀺𑀫𑀢𑁄 𑀧𑀦 𑀞𑀺𑀢𑀸,_  
_𑀅𑀦𑁂𑀓𑀯𑀡𑁆𑀡𑀸 𑀅𑀘𑁆𑀘𑀺𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇_  


167\. _𑀦𑀻𑀮𑀸 𑀅𑀣 𑀮𑁄𑀳𑀺𑀢𑀺𑀓𑀸,_  
_𑀫𑀜𑁆𑀚𑀺𑀝𑁆𑀞𑀸 𑀧𑀻𑀢𑀓𑀸 𑀨𑀮𑀺𑀓𑀯𑀡𑁆𑀡𑀸𑀬𑁄;_  
_𑀅𑀗𑁆𑀕𑀻𑀭𑀲𑀲𑁆𑀲 𑀓𑀸𑀬𑁂,_  
_𑀅𑀦𑁂𑀓𑀯𑀡𑁆𑀡𑀸 𑀅𑀘𑁆𑀘𑀺𑀬𑁄 𑀳𑁄𑀦𑁆𑀢𑀺𑁇_  


168\. _𑀧𑀢𑁆𑀢𑀫𑁆𑀳𑀺 𑀑𑀤𑀳𑀺𑀢𑁆𑀯𑀸,_  
_𑀅𑀳𑀺𑀦𑀸𑀕𑀁 𑀩𑁆𑀭𑀸𑀳𑁆𑀫𑀡𑀲𑁆𑀲 𑀤𑀲𑁆𑀲𑁂𑀲𑀺;_  
_“𑀅𑀬𑀁 𑀢𑁂 𑀓𑀲𑁆𑀲𑀧 𑀦𑀸𑀕𑁄,_  
_𑀧𑀭𑀺𑀬𑀸𑀤𑀺𑀦𑁆𑀦𑁄 𑀅𑀲𑁆𑀲 𑀢𑁂𑀚𑀲𑀸 𑀢𑁂𑀚𑁄”𑀢𑀺𑁇_  


169\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀪𑀕𑀯𑀢𑁄 𑀇𑀫𑀺𑀦𑀸 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑁂𑀦 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥𑁂𑀯, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀯𑀺𑀳𑀭, 𑀅𑀳𑀁 𑀢𑁂 𑀥𑀼𑀯𑀪𑀢𑁆𑀢𑁂𑀦𑀸”𑀢𑀺𑁇

---

170\. 𑀧𑀞𑀫𑀁 𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁𑁇



(40.)

171\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀅𑀲𑁆𑀲𑀫𑀲𑁆𑀲 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀅𑀝𑁆𑀞𑀁𑀲𑀼 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀦𑁆𑀢𑀸 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸𑁇

172\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀸𑀮𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀁𑁇 𑀓𑁂 𑀦𑀼 𑀔𑁄 𑀢𑁂, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑀸 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁆𑀯𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀘𑀢𑀼𑀤𑁆𑀤𑀺𑀲𑀸 𑀅𑀝𑁆𑀞𑀁𑀲𑀼 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀦𑁆𑀢𑀸 𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑀸”𑀢𑀺𑁇 “𑀏𑀢𑁂 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀘𑀢𑁆𑀢𑀸𑀭𑁄 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀘𑀢𑁆𑀢𑀸𑀭𑁄𑀧𑀺 𑀫𑀳𑀸𑀭𑀸𑀚𑀸𑀦𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇

---

173\. 𑀤𑀼𑀢𑀺𑀬𑀁 𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁𑁇

174\. 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑁄 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀼𑀭𑀺𑀫𑀸𑀳𑀺 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸𑀳𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

175\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀸𑀮𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀁𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑁄 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁆𑀯𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀼𑀭𑀺𑀫𑀸𑀳𑀺 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸𑀳𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺? “𑀏𑀲𑁄 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀓𑁆𑀓𑁄𑀧𑀺 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇

177\. 𑀅𑀣 𑀔𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑁄 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀼𑀭𑀺𑀫𑀸𑀳𑀺 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸𑀳𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑁇

178\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀸𑀮𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀁𑁇 𑀓𑁄 𑀦𑀼 𑀔𑁄 𑀲𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀸𑀬 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀯𑀡𑁆𑀡𑁄 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀑𑀪𑀸𑀲𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀢𑁆𑀯𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀳𑀸𑀅𑀕𑁆𑀕𑀺𑀓𑁆𑀔𑀦𑁆𑀥𑁄, 𑀧𑀼𑀭𑀺𑀫𑀸𑀳𑀺 𑀯𑀡𑁆𑀡𑀦𑀺𑀪𑀸𑀳𑀺 𑀅𑀪𑀺𑀓𑁆𑀓𑀦𑁆𑀢𑀢𑀭𑁄 𑀘 𑀧𑀡𑀻𑀢𑀢𑀭𑁄 𑀘𑀸”𑀢𑀺? “𑀏𑀲𑁄 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀩𑁆𑀭𑀳𑁆𑀫𑀸 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀬𑁂𑀦𑀸𑀳𑀁 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀩𑁆𑀭𑀳𑁆𑀫𑀸𑀧𑀺 𑀲𑀳𑀫𑁆𑀧𑀢𑀺 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀢𑀺 𑀥𑀫𑁆𑀫𑀲𑁆𑀲𑀯𑀦𑀸𑀬, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇

180\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄 𑀳𑁄𑀢𑀺, 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀸 𑀘 𑀅𑀗𑁆𑀕𑀫𑀕𑀥𑀸 𑀧𑀳𑀽𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀫𑁂 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄, 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀸 𑀘 𑀅𑀗𑁆𑀕𑀫𑀕𑀥𑀸 𑀧𑀳𑀽𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑀘𑁂 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁂 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀳𑀸𑀲𑀫𑀡𑀲𑁆𑀲 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀫 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀧𑀭𑀺𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺𑁇 𑀅𑀳𑁄 𑀦𑀽𑀦 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀦𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀉𑀢𑁆𑀢𑀭𑀓𑀼𑀭𑀼𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁄𑀢𑀢𑁆𑀢𑀤𑀳𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀓𑀸𑀲𑀺𑁇

181\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀸𑀮𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀁𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀳𑀺𑀬𑁆𑀬𑁄 𑀦𑀸𑀕𑀫𑀸𑀲𑀺? 𑀅𑀧𑀺 𑀘 𑀫𑀬𑀁 𑀢𑀁 𑀲𑀭𑀸𑀫—  𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀦𑀸𑀕𑀘𑁆𑀙𑀢𑀻𑀢𑀺? 𑀔𑀸𑀤𑀦𑀻𑀬𑀲𑁆𑀲 𑀘 𑀪𑁄𑀚𑀦𑀻𑀬𑀲𑁆𑀲 𑀘 𑀢𑁂 𑀧𑀝𑀺𑀯𑀻𑀲𑁄 𑀞𑀧𑀺𑀢𑁄”𑀢𑀺𑁇 “𑀦𑀦𑀼 𑀢𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀏𑀢𑀭𑀳𑀺 𑀔𑁄 𑀫𑁂 𑀫𑀳𑀸𑀬𑀜𑁆𑀜𑁄 𑀧𑀘𑁆𑀘𑀼𑀧𑀝𑁆𑀞𑀺𑀢𑁄, 𑀓𑁂𑀯𑀮𑀓𑀧𑁆𑀧𑀸 𑀘 𑀅𑀗𑁆𑀕𑀫𑀕𑀥𑀸 𑀧𑀳𑀽𑀢𑀁 𑀔𑀸𑀤𑀦𑀻𑀬𑀁 𑀪𑁄𑀚𑀦𑀻𑀬𑀁 𑀆𑀤𑀸𑀬 𑀅𑀪𑀺𑀓𑁆𑀓𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀲𑀘𑁂 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀚𑀦𑀓𑀸𑀬𑁂 𑀇𑀤𑁆𑀥𑀺𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀳𑀸𑀲𑀫𑀡𑀲𑁆𑀲 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀅𑀪𑀺𑀯𑀟𑁆𑀠𑀺𑀲𑁆𑀲𑀢𑀺, 𑀫𑀫 𑀮𑀸𑀪𑀲𑀓𑁆𑀓𑀸𑀭𑁄 𑀧𑀭𑀺𑀳𑀸𑀬𑀺𑀲𑁆𑀲𑀢𑀺, 𑀅𑀳𑁄 𑀦𑀽𑀦 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀲𑁆𑀯𑀸𑀢𑀦𑀸𑀬 𑀦𑀸𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑁄 𑀔𑁄 𑀅𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀢𑀯 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀉𑀢𑁆𑀢𑀭𑀓𑀼𑀭𑀼𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀢𑀢𑁄 𑀧𑀺𑀡𑁆𑀟𑀧𑀸𑀢𑀁 𑀆𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀦𑁄𑀢𑀢𑁆𑀢𑀤𑀳𑁂 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀢𑁆𑀣𑁂𑀯 𑀤𑀺𑀯𑀸𑀯𑀺𑀳𑀸𑀭𑀁 𑀅𑀓𑀸𑀲𑀺𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀘𑁂𑀢𑀲𑀸𑀧𑀺 𑀘𑀺𑀢𑁆𑀢𑀁 𑀧𑀚𑀸𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇

182\. 𑀧𑀜𑁆𑀘𑀫𑀁 𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀁𑁇

183\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑁄 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀢𑁆𑀣 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀥𑁄𑀯𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁 𑀔𑀡𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀥𑁄𑀯𑀢𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀫𑀳𑀢𑀺𑀁 𑀲𑀺𑀮𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀢𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀓𑀓𑀼𑀥𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀪𑀕𑀯𑀢𑁄 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀲𑀸𑀔𑀁 𑀑𑀦𑀸𑀫𑁂𑀲𑀺—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀢𑀽”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀪𑀕𑀯𑀢𑁄 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀫𑀳𑀢𑀺𑀁 𑀲𑀺𑀮𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺—  “𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀽”𑀢𑀺𑁇

184\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀸𑀮𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀁𑁇 𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀸𑀬𑀁 𑀧𑀼𑀩𑁆𑀩𑁂 𑀇𑀥 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻, 𑀲𑀸𑀬𑀁 𑀇𑀥 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑁇 𑀦𑀬𑀺𑀫𑀸 𑀲𑀺𑀮𑀸 𑀧𑀼𑀩𑁆𑀩𑁂 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸𑁇 𑀓𑁂𑀦𑀺𑀫𑀸 𑀲𑀺𑀮𑀸 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸? 𑀦𑀬𑀺𑀫𑀲𑁆𑀲 𑀓𑀓𑀼𑀥𑀲𑁆𑀲 𑀧𑀼𑀩𑁆𑀩𑁂 𑀲𑀸𑀔𑀸 𑀑𑀦𑀢𑀸, 𑀲𑀸𑀬𑀁 𑀲𑀸𑀔𑀸 𑀑𑀦𑀢𑀸”𑀢𑀺𑁇 “𑀇𑀥 𑀫𑁂, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀅𑀳𑁄𑀲𑀺𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀢𑁆𑀣 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀥𑁄𑀯𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺? 𑀅𑀣 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀫 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀧𑀸𑀡𑀺𑀦𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀺𑀁 𑀔𑀡𑀺𑀢𑁆𑀯𑀸 𑀫𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  ‘𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀥𑁄𑀯𑀢𑀽’𑀢𑀺𑁇 𑀲𑀸𑀬𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀦 𑀧𑀸𑀡𑀺𑀦𑀸 𑀔𑀡𑀺𑀢𑀸 𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺? 𑀅𑀣 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀫 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀫𑀳𑀢𑀺𑀁 𑀲𑀺𑀮𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺—  ‘𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀧𑀭𑀺𑀫𑀤𑁆𑀤𑀢𑀽’𑀢𑀺𑁇 𑀲𑀸𑀬𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀦 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀲𑀺𑀮𑀸𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺? 𑀅𑀣 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀓𑀓𑀼𑀥𑁂 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀸 𑀤𑁂𑀯𑀢𑀸 𑀫𑀫 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀲𑀸𑀔𑀁 𑀑𑀦𑀸𑀫𑁂𑀲𑀺—  ‘𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀆𑀮𑀫𑁆𑀩𑀺𑀢𑁆𑀯𑀸 𑀉𑀢𑁆𑀢𑀭𑀢𑀽’𑀢𑀺𑁇 𑀲𑁆𑀯𑀸𑀬𑀁 𑀆𑀳𑀭𑀳𑀢𑁆𑀣𑁄 𑀓𑀓𑀼𑀥𑁄𑁇 𑀢𑀲𑁆𑀲 𑀫𑀬𑁆𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  ‘𑀓𑀺𑀫𑁆𑀳𑀺 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀬𑁆𑀬𑀦𑁆’𑀢𑀺? 𑀅𑀣 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑀓𑁆𑀓𑁄 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀫𑀫 𑀘𑁂𑀢𑀲𑀸 𑀘𑁂𑀢𑁄𑀧𑀭𑀺𑀯𑀺𑀢𑀓𑁆𑀓𑀫𑀜𑁆𑀜𑀸𑀬 𑀫𑀳𑀢𑀺𑀁 𑀲𑀺𑀮𑀁 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀧𑀺—  ‘𑀇𑀥, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀧𑀁𑀲𑀼𑀓𑀽𑀮𑀁 𑀯𑀺𑀲𑁆𑀲𑀚𑁆𑀚𑁂𑀢𑀽’𑀢𑀺𑁇 𑀲𑀸𑀬𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀫𑀦𑀼𑀲𑁆𑀲𑁂𑀦 𑀉𑀧𑀦𑀺𑀓𑁆𑀔𑀺𑀢𑁆𑀢𑀸 𑀲𑀺𑀮𑀸” 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀓𑁆𑀓𑁄𑀧𑀺 𑀤𑁂𑀯𑀸𑀦𑀫𑀺𑀦𑁆𑀤𑁄 𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇

185\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀆𑀬𑀸𑀫𑀳𑀦𑁆”𑀢𑀺 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀬 𑀚𑀫𑁆𑀩𑀼𑀬𑀸 “𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁄” 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀢𑀢𑁄 𑀨𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀫𑁂𑀦 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀫𑀕𑁆𑀕𑁂𑀦 𑀆𑀕𑀢𑁄? 𑀅𑀳𑀁 𑀢𑀬𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄, 𑀲𑁄 𑀢𑁆𑀯𑀁 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 “𑀇𑀥𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀢𑀁 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀬 𑀚𑀫𑁆𑀩𑀼𑀬𑀸 ‘𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁄’ 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀢𑀢𑁄 𑀨𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑁇 𑀇𑀤𑀁 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀚𑀫𑁆𑀩𑀼𑀨𑀮𑀁 𑀯𑀡𑁆𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀕𑀦𑁆𑀥𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀭𑀲𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀲𑀺 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀢𑁆𑀯𑀁𑀬𑁂𑀯 𑀢𑀁 𑀅𑀭𑀳𑀲𑀺, 𑀢𑁆𑀯𑀁𑀬𑁂𑀯 𑀢𑀁 𑀧𑀭𑀺𑀪𑀼𑀜𑁆𑀚𑀸𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀫𑀁 𑀧𑀞𑀫𑀢𑀭𑀁 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀬 𑀚𑀫𑁆𑀩𑀼𑀬𑀸 ‘𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁄’ 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀢𑀢𑁄 𑀨𑀮𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀪𑀢𑁆𑀢𑀁 𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑁆𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀯𑀺𑀳𑀸𑀲𑀺𑁇

(45.)

186\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀢𑀲𑁆𑀲𑀸 𑀭𑀢𑁆𑀢𑀺𑀬𑀸 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀮𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺—  “𑀓𑀸𑀮𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁 𑀪𑀢𑁆𑀢𑀦𑁆”𑀢𑀺𑁇 “𑀕𑀘𑁆𑀙 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀆𑀬𑀸𑀫𑀳𑀦𑁆”𑀢𑀺 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀬𑀸𑀬 𑀚𑀫𑁆𑀩𑀼𑀬𑀸 “𑀚𑀫𑁆𑀩𑀼𑀤𑀻𑀧𑁄” 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀢𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀅𑀫𑁆𑀩𑁄…𑀧𑁂…  𑀢𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀆𑀫𑀮𑀓𑀻…𑀧𑁂…  𑀢𑀲𑁆𑀲𑀸 𑀅𑀯𑀺𑀤𑀽𑀭𑁂 𑀳𑀭𑀻𑀢𑀓𑀻…𑀧𑁂…  𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀧𑀼𑀧𑁆𑀨𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀓𑀢𑀫𑁂𑀦 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀫𑀕𑁆𑀕𑁂𑀦 𑀆𑀕𑀢𑁄? 𑀅𑀳𑀁 𑀢𑀬𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑁄, 𑀲𑁄 𑀢𑁆𑀯𑀁 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 “𑀇𑀥𑀸𑀳𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀢𑀁 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀧𑀼𑀧𑁆𑀨𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄𑁇 𑀇𑀤𑀁 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀧𑀼𑀧𑁆𑀨𑀁 𑀯𑀡𑁆𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁 𑀕𑀦𑁆𑀥𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀁𑁇 𑀲𑀘𑁂 𑀆𑀓𑀗𑁆𑀔𑀲𑀺 𑀕𑀡𑁆𑀳𑀸”𑀢𑀺𑁇 “𑀅𑀮𑀁, 𑀫𑀳𑀸𑀲𑀫𑀡, 𑀢𑁆𑀯𑀁𑀬𑁂𑀯 𑀢𑀁 𑀅𑀭𑀳𑀲𑀺, 𑀢𑁆𑀯𑀁𑀬𑁂𑀯 𑀢𑀁 𑀕𑀡𑁆𑀳𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀫𑀁 𑀧𑀞𑀫𑀢𑀭𑀁 𑀉𑀬𑁆𑀬𑁄𑀚𑁂𑀢𑁆𑀯𑀸 𑀢𑀸𑀯𑀢𑀺𑀁𑀲𑀁 𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀸𑀭𑀺𑀘𑁆𑀙𑀢𑁆𑀢𑀓𑀧𑀼𑀧𑁆𑀨𑀁 𑀕𑀳𑁂𑀢𑁆𑀯𑀸 𑀧𑀞𑀫𑀢𑀭𑀁 𑀆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀕𑁆𑀬𑀸𑀕𑀸𑀭𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇

(46.)

187\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀅𑀕𑁆𑀕𑀺𑀁 𑀧𑀭𑀺𑀘𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀨𑀸𑀮𑁂𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀚𑀝𑀺𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀣𑀸 𑀫𑀬𑀁 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫 𑀓𑀝𑁆𑀞𑀸𑀦𑀺 𑀨𑀸𑀮𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀨𑀸𑀮𑀺𑀬𑀦𑁆𑀢𑀼, 𑀓𑀲𑁆𑀲𑀧, 𑀓𑀝𑁆𑀞𑀸𑀦𑀻”𑀢𑀺𑁇 “𑀨𑀸𑀮𑀺𑀬𑀦𑁆𑀢𑀼, 𑀫𑀳𑀸𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 𑀲𑀓𑀺𑀤𑁂𑀯 𑀧𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀲𑀢𑀸𑀦𑀺 𑀨𑀸𑀮𑀺𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀓𑀝𑁆𑀞𑀸𑀦𑀺𑀧𑀺 𑀨𑀸𑀮𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇

(47.)

188\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀅𑀕𑁆𑀕𑀺𑀁 𑀧𑀭𑀺𑀘𑀭𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀺𑀁 𑀉𑀚𑁆𑀚𑀮𑁂𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀚𑀝𑀺𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀣𑀸 𑀫𑀬𑀁 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫 𑀅𑀕𑁆𑀕𑀺𑀁 𑀉𑀚𑁆𑀚𑀮𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀉𑀚𑁆𑀚𑀮𑀺𑀬𑀦𑁆𑀢𑀼, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀕𑁆𑀕𑀻”𑀢𑀺𑁇 “𑀉𑀚𑁆𑀚𑀮𑀺𑀬𑀦𑁆𑀢𑀼, 𑀫𑀳𑀸𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 𑀲𑀓𑀺𑀤𑁂𑀯 𑀧𑀜𑁆𑀘 𑀅𑀕𑁆𑀕𑀺𑀲𑀢𑀸𑀦𑀺 𑀉𑀚𑁆𑀚𑀮𑀺𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀅𑀕𑁆𑀕𑀻𑀧𑀺 𑀉𑀚𑁆𑀚𑀮𑀺𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇

(48.)

189\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀅𑀕𑁆𑀕𑀺𑀁 𑀧𑀭𑀺𑀘𑀭𑀺𑀢𑁆𑀯𑀸 𑀦 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺 𑀅𑀕𑁆𑀕𑀺𑀁 𑀯𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀚𑀝𑀺𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀣𑀸 𑀫𑀬𑀁 𑀦 𑀲𑀓𑁆𑀓𑁄𑀫 𑀅𑀕𑁆𑀕𑀺𑀁 𑀯𑀺𑀚𑁆𑀛𑀸𑀧𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀯𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀼, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀕𑁆𑀕𑀻”𑀢𑀺𑁇 “𑀯𑀺𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀼, 𑀫𑀳𑀸𑀲𑀫𑀡𑀸”𑀢𑀺𑁇 𑀲𑀓𑀺𑀤𑁂𑀯 𑀧𑀜𑁆𑀘 𑀅𑀕𑁆𑀕𑀺𑀲𑀢𑀸𑀦𑀺 𑀯𑀺𑀚𑁆𑀛𑀸𑀬𑀺𑀁𑀲𑀼𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀅𑀕𑁆𑀕𑀻𑀧𑀺 𑀯𑀺𑀚𑁆𑀛𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇

(49.)

190\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀲𑀻𑀢𑀸𑀲𑀼 𑀳𑁂𑀫𑀦𑁆𑀢𑀺𑀓𑀸𑀲𑀼 𑀭𑀢𑁆𑀢𑀻𑀲𑀼 𑀅𑀦𑁆𑀢𑀭𑀝𑁆𑀞𑀓𑀸𑀲𑀼 𑀳𑀺𑀫𑀧𑀸𑀢𑀲𑀫𑀬𑁂 𑀦𑀚𑁆𑀚𑀸 𑀦𑁂𑀭𑀜𑁆𑀚𑀭𑀸𑀬 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀧𑀺, 𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀦𑁆𑀢𑀺𑀧𑀺, 𑀉𑀫𑁆𑀫𑀼𑀚𑁆𑀚𑀦𑀦𑀺𑀫𑀼𑀚𑁆𑀚𑀦𑀫𑁆𑀧𑀺 𑀓𑀭𑁄𑀦𑁆𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀧𑀜𑁆𑀘𑀫𑀢𑁆𑀢𑀸𑀦𑀺 𑀫𑀦𑁆𑀤𑀸𑀫𑀼𑀔𑀺𑀲𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺, 𑀬𑀢𑁆𑀣 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀉𑀢𑁆𑀢𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀺𑀩𑁆𑀩𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀚𑀝𑀺𑀮𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀦𑀺𑀲𑁆𑀲𑀁𑀲𑀬𑀁 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑀲𑁆𑀲 𑀇𑀤𑁆𑀥𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀣𑀬𑀺𑀫𑀸 𑀫𑀦𑁆𑀤𑀸𑀫𑀼𑀔𑀺𑀬𑁄 𑀦𑀺𑀫𑁆𑀫𑀺𑀢𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀢𑀸𑀯 𑀩𑀳𑀽 𑀫𑀦𑁆𑀤𑀸𑀫𑀼𑀔𑀺𑀬𑁄𑀧𑀺 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇

(50.)

191\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀳𑀸 𑀅𑀓𑀸𑀮𑀫𑁂𑀖𑁄 𑀧𑀸𑀯𑀲𑁆𑀲𑀺, 𑀫𑀳𑀸 𑀉𑀤𑀓𑀯𑀸𑀳𑀓𑁄 𑀲𑀜𑁆𑀚𑀸𑀬𑀺𑁇 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺, 𑀲𑁄 𑀧𑀤𑁂𑀲𑁄 𑀉𑀤𑀓𑁂𑀦 𑀦 𑀑𑀢𑁆𑀣𑀝𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀲𑀫𑀦𑁆𑀢𑀸 𑀉𑀤𑀓𑀁 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀭𑁂𑀡𑀼𑀳𑀢𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀘𑀗𑁆𑀓𑀫𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀫𑀦𑁆𑀢𑀸 𑀉𑀤𑀓𑀁 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀭𑁂𑀡𑀼𑀳𑀢𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀘𑀗𑁆𑀓𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄—  “𑀫𑀸 𑀳𑁂𑀯 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀉𑀤𑀓𑁂𑀦 𑀯𑀽𑀍𑀳𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺 𑀦𑀸𑀯𑀸𑀬 𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂𑀳𑀺 𑀚𑀝𑀺𑀮𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑀲𑁆𑀫𑀺𑀁 𑀧𑀤𑁂𑀲𑁂 𑀪𑀕𑀯𑀸 𑀯𑀺𑀳𑀭𑀢𑀺 𑀢𑀁 𑀧𑀤𑁂𑀲𑀁 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀫𑀦𑁆𑀢𑀸 𑀉𑀤𑀓𑀁 𑀉𑀲𑁆𑀲𑀸𑀭𑁂𑀢𑁆𑀯𑀸 𑀫𑀚𑁆𑀛𑁂 𑀭𑁂𑀡𑀼𑀳𑀢𑀸𑀬 𑀪𑀽𑀫𑀺𑀬𑀸 𑀘𑀗𑁆𑀓𑀫𑀦𑁆𑀢𑀁, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀦𑀼 𑀢𑁆𑀯𑀁, 𑀫𑀳𑀸𑀲𑀫𑀡𑀸”𑀢𑀺? “𑀅𑀬𑀫𑀳𑀫𑀲𑁆𑀫𑀺, 𑀓𑀲𑁆𑀲𑀧𑀸”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀯𑁂𑀳𑀸𑀲𑀁 𑀅𑀩𑁆𑀪𑀼𑀕𑁆𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀦𑀸𑀯𑀸𑀬 𑀧𑀘𑁆𑀘𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀚𑀝𑀺𑀮𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀬𑀢𑁆𑀭 𑀳𑀺 𑀦𑀸𑀫 𑀉𑀤𑀓𑀫𑁆𑀧𑀺 𑀦 𑀧𑀯𑀸𑀳𑀺𑀲𑁆𑀲𑀢𑀺, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆”𑀢𑀺𑁇

(51.)

192\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀘𑀺𑀭𑀫𑁆𑀧𑀺 𑀔𑁄 𑀇𑀫𑀲𑁆𑀲 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲𑀲𑁆𑀲 𑀏𑀯𑀁 𑀪𑀯𑀺𑀲𑁆𑀲𑀢𑀺—  ‘𑀫𑀳𑀺𑀤𑁆𑀥𑀺𑀓𑁄 𑀔𑁄 𑀫𑀳𑀸𑀲𑀫𑀡𑁄 𑀫𑀳𑀸𑀦𑀼𑀪𑀸𑀯𑁄, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀘 𑀔𑁄 𑀅𑀭𑀳𑀸 𑀬𑀣𑀸 𑀅𑀳𑀦𑁆’𑀢𑀺; 𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑀁 𑀚𑀝𑀺𑀮𑀁 𑀲𑀁𑀯𑁂𑀚𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀚𑀝𑀺𑀮𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀦𑁂𑀯 𑀘 𑀔𑁄 𑀢𑁆𑀯𑀁, 𑀓𑀲𑁆𑀲𑀧, 𑀅𑀭𑀳𑀸, 𑀦𑀸𑀧𑀺 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄𑁇 𑀲𑀸𑀧𑀺 𑀢𑁂 𑀧𑀝𑀺𑀧𑀤𑀸 𑀦𑀢𑁆𑀣𑀺, 𑀬𑀸𑀬 𑀢𑁆𑀯𑀁 𑀅𑀭𑀳𑀸 𑀯𑀸 𑀅𑀲𑁆𑀲𑀲𑀺, 𑀅𑀭𑀳𑀢𑁆𑀢𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀦𑁆𑀦𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀢𑁆𑀯𑀁 𑀔𑁄𑀲𑀺, 𑀓𑀲𑁆𑀲𑀧, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀚𑀝𑀺𑀮𑀲𑀢𑀸𑀦𑀁 𑀦𑀸𑀬𑀓𑁄 𑀯𑀺𑀦𑀸𑀬𑀓𑁄 𑀅𑀕𑁆𑀕𑁄 𑀧𑀫𑀼𑀔𑁄 𑀧𑀸𑀫𑁄𑀓𑁆𑀔𑁄𑁇 𑀢𑁂𑀧𑀺 𑀢𑀸𑀯 𑀅𑀧𑀮𑁄𑀓𑁂𑀳𑀺, 𑀬𑀣𑀸 𑀢𑁂 𑀫𑀜𑁆𑀜𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀢𑁂 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇

193\. 𑀅𑀣 𑀔𑁄 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀬𑁂𑀦 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑁂 𑀚𑀝𑀺𑀮𑁂 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀘𑁆𑀙𑀸𑀫𑀳𑀁, 𑀪𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀢𑀼𑀁, 𑀬𑀣𑀸 𑀪𑀯𑀦𑁆𑀢𑁄 𑀫𑀜𑁆𑀜𑀦𑁆𑀢𑀺 𑀢𑀣𑀸 𑀓𑀭𑁄𑀦𑁆𑀢𑀽”𑀢𑀺𑁇 “𑀘𑀺𑀭𑀧𑀝𑀺𑀓𑀸 𑀫𑀬𑀁, 𑀪𑁄, 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀅𑀪𑀺𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸, 𑀲𑀘𑁂 𑀪𑀯𑀁, 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀢𑀺, 𑀲𑀩𑁆𑀩𑁂𑀯 𑀫𑀬𑀁 𑀫𑀳𑀸𑀲𑀫𑀡𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀓𑁂𑀲𑀫𑀺𑀲𑁆𑀲𑀁 𑀚𑀝𑀸𑀫𑀺𑀲𑁆𑀲𑀁 𑀔𑀸𑀭𑀺𑀓𑀸𑀚𑀫𑀺𑀲𑁆𑀲𑀁 𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑀫𑀺𑀲𑁆𑀲𑀁 𑀉𑀤𑀓𑁂 𑀧𑀯𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇

(52.)

194\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀦𑀤𑀻𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀓𑁂𑀲𑀫𑀺𑀲𑁆𑀲𑀁 𑀚𑀝𑀸𑀫𑀺𑀲𑁆𑀲𑀁 𑀔𑀸𑀭𑀺𑀓𑀸𑀚𑀫𑀺𑀲𑁆𑀲𑀁 𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑀫𑀺𑀲𑁆𑀲𑀁 𑀉𑀤𑀓𑁂 𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀸 𑀳𑁂𑀯 𑀫𑁂 𑀪𑀸𑀢𑀼𑀦𑁄 𑀉𑀧𑀲𑀕𑁆𑀕𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺𑁇 𑀚𑀝𑀺𑀮𑁂 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙𑀣 𑀫𑁂 𑀪𑀸𑀢𑀭𑀁 𑀚𑀸𑀦𑀸𑀣𑀸”𑀢𑀺𑁇 𑀲𑀸𑀫𑀜𑁆𑀘 𑀢𑀻𑀳𑀺 𑀚𑀝𑀺𑀮𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀦𑀼 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑁂𑀬𑁆𑀬𑁄”𑀢𑀺? “𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀇𑀤𑀁 𑀲𑁂𑀬𑁆𑀬𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀓𑁂𑀲𑀫𑀺𑀲𑁆𑀲𑀁 𑀚𑀝𑀸𑀫𑀺𑀲𑁆𑀲𑀁 𑀔𑀸𑀭𑀺𑀓𑀸𑀚𑀫𑀺𑀲𑁆𑀲𑀁 𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑀫𑀺𑀲𑁆𑀲𑀁 𑀉𑀤𑀓𑁂 𑀧𑀯𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇

(53.)

195\. 𑀅𑀤𑁆𑀤𑀲𑀸 𑀔𑁄 𑀕𑀬𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀚𑀝𑀺𑀮𑁄 𑀓𑁂𑀲𑀫𑀺𑀲𑁆𑀲𑀁 𑀚𑀝𑀸𑀫𑀺𑀲𑁆𑀲𑀁 𑀔𑀸𑀭𑀺𑀓𑀸𑀚𑀫𑀺𑀲𑁆𑀲𑀁 𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑀫𑀺𑀲𑁆𑀲𑀁 𑀉𑀤𑀓𑁂 𑀯𑀼𑀬𑁆𑀳𑀫𑀸𑀦𑁂, 𑀤𑀺𑀲𑁆𑀯𑀸𑀦𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀸 𑀳𑁂𑀯 𑀫𑁂 𑀪𑀸𑀢𑀽𑀦𑀁 𑀉𑀧𑀲𑀕𑁆𑀕𑁄 𑀅𑀳𑁄𑀲𑀻”𑀢𑀺𑁇 𑀚𑀝𑀺𑀮𑁂 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙𑀣 𑀫𑁂 𑀪𑀸𑀢𑀭𑁄 𑀚𑀸𑀦𑀸𑀣𑀸”𑀢𑀺𑁇 𑀲𑀸𑀫𑀜𑁆𑀘 𑀤𑁆𑀯𑀻𑀳𑀺 𑀚𑀝𑀺𑀮𑀲𑀢𑁂𑀳𑀺 𑀲𑀤𑁆𑀥𑀺𑀁 𑀬𑁂𑀦𑀸𑀬𑀲𑁆𑀫𑀸 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀭𑀼𑀯𑁂𑀮𑀓𑀲𑁆𑀲𑀧𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀇𑀤𑀁 𑀦𑀼 𑀔𑁄, 𑀓𑀲𑁆𑀲𑀧, 𑀲𑁂𑀬𑁆𑀬𑁄”𑀢𑀺? “𑀆𑀫𑀸𑀯𑀼𑀲𑁄, 𑀇𑀤𑀁 𑀲𑁂𑀬𑁆𑀬𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀚𑀝𑀺𑀮𑀸 𑀓𑁂𑀲𑀫𑀺𑀲𑁆𑀲𑀁 𑀚𑀝𑀸𑀫𑀺𑀲𑁆𑀲𑀁 𑀔𑀸𑀭𑀺𑀓𑀸𑀚𑀫𑀺𑀲𑁆𑀲𑀁 𑀅𑀕𑁆𑀕𑀺𑀳𑀼𑀢𑀫𑀺𑀲𑁆𑀲𑀁 𑀉𑀤𑀓𑁂 𑀧𑀯𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀸𑀤𑁂𑀲𑀼 𑀲𑀺𑀭𑀲𑀸 𑀦𑀺𑀧𑀢𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇

196\. 𑀪𑀕𑀯𑀢𑁄 𑀅𑀥𑀺𑀝𑁆𑀞𑀸𑀦𑁂𑀦 𑀧𑀜𑁆𑀘 𑀓𑀝𑁆𑀞𑀲𑀢𑀸𑀦𑀺 𑀦 𑀨𑀸𑀮𑀺𑀬𑀺𑀁𑀲𑀼, 𑀨𑀸𑀮𑀺𑀬𑀺𑀁𑀲𑀼; 𑀅𑀕𑁆𑀕𑀻 𑀦 𑀉𑀚𑁆𑀚𑀮𑀺𑀬𑀺𑀁𑀲𑀼, 𑀉𑀚𑁆𑀚𑀮𑀺𑀬𑀺𑀁𑀲𑀼; 𑀦 𑀯𑀺𑀚𑁆𑀛𑀸𑀬𑀺𑀁𑀲𑀼, 𑀯𑀺𑀚𑁆𑀛𑀸𑀬𑀺𑀁𑀲𑀼; 𑀧𑀜𑁆𑀘𑀫𑀦𑁆𑀤𑀸𑀫𑀼𑀔𑀺𑀲𑀢𑀸𑀦𑀺 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑁇 𑀏𑀢𑁂𑀦 𑀦𑀬𑁂𑀦 𑀅𑀟𑁆𑀠𑀼𑀟𑁆𑀠𑀧𑀸𑀝𑀺𑀳𑀸𑀭𑀺𑀬𑀲𑀳𑀲𑁆𑀲𑀸𑀦𑀺 𑀳𑁄𑀦𑁆𑀢𑀺𑁇

199\. 𑀏𑀯𑀁 𑀧𑀲𑁆𑀲𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀼𑀢𑀯𑀸 𑀅𑀭𑀺𑀬𑀲𑀸𑀯𑀓𑁄 𑀘𑀓𑁆𑀔𑀼𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀭𑀽𑀧𑁂𑀲𑀼𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀘𑀓𑁆𑀔𑀼𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀬𑀫𑀺𑀤𑀁 𑀘𑀓𑁆𑀔𑀼𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸, 𑀢𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺𑁇 𑀲𑁄𑀢𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀲𑀤𑁆𑀤𑁂𑀲𑀼𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀖𑀸𑀦𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀕𑀦𑁆𑀥𑁂𑀲𑀼𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀚𑀺𑀯𑁆𑀳𑀸𑀬𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀭𑀲𑁂𑀲𑀼𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀓𑀸𑀬𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀨𑁄𑀝𑁆𑀞𑀩𑁆𑀩𑁂𑀲𑀼𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺…𑀧𑁂…  𑀫𑀦𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀥𑀫𑁆𑀫𑁂𑀲𑀼𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀫𑀦𑁄𑀯𑀺𑀜𑁆𑀜𑀸𑀡𑁂𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁂𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀬𑀫𑀺𑀤𑀁 𑀫𑀦𑁄𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑀧𑀘𑁆𑀘𑀬𑀸 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀢𑀺 𑀯𑁂𑀤𑀬𑀺𑀢𑀁 𑀲𑀼𑀔𑀁 𑀯𑀸 𑀤𑀼𑀓𑁆𑀔𑀁 𑀯𑀸 𑀅𑀤𑀼𑀓𑁆𑀔𑀫𑀲𑀼𑀔𑀁 𑀯𑀸 𑀢𑀲𑁆𑀫𑀺𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀢𑀺, 𑀦𑀺𑀩𑁆𑀩𑀺𑀦𑁆𑀤𑀁 𑀯𑀺𑀭𑀚𑁆𑀚𑀢𑀺, 𑀯𑀺𑀭𑀸𑀕𑀸 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀢𑀺, 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀫𑀺𑀢𑀺 𑀜𑀸𑀡𑀁 𑀳𑁄𑀢𑀺𑁇 𑀔𑀻𑀡𑀸 𑀚𑀸𑀢𑀺, 𑀯𑀼𑀲𑀺𑀢𑀁 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁, 𑀓𑀢𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀦𑀸𑀧𑀭𑀁 𑀇𑀢𑁆𑀣𑀢𑁆𑀢𑀸𑀬𑀸𑀢𑀺 𑀧𑀚𑀸𑀦𑀸𑀢𑀻”𑀢𑀺𑁇

200\. 𑀇𑀫𑀲𑁆𑀫𑀺𑀜𑁆𑀘 𑀧𑀦 𑀯𑁂𑀬𑁆𑀬𑀸𑀓𑀭𑀡𑀲𑁆𑀫𑀺𑀁 𑀪𑀜𑁆𑀜𑀫𑀸𑀦𑁂 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀳𑀲𑁆𑀲𑀲𑁆𑀲 𑀅𑀦𑀼𑀧𑀸𑀤𑀸𑀬 𑀆𑀲𑀯𑁂𑀳𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀦𑀺 𑀯𑀺𑀫𑀼𑀘𑁆𑀘𑀺𑀁𑀲𑀼𑁇
