# 𑁧𑁇𑁨𑁭 𑀭𑀸𑀚𑀪𑀝𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀭𑀸𑀚𑀪𑀝𑀯𑀢𑁆𑀣𑀼

(90.)

426\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑁄 𑀓𑀼𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀲𑁂𑀦𑀸𑀦𑀸𑀬𑀓𑁂 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂 𑀆𑀡𑀸𑀧𑁂𑀲𑀺—  “𑀕𑀘𑁆𑀙𑀣, 𑀪𑀡𑁂, 𑀧𑀘𑁆𑀘𑀦𑁆𑀢𑀁 𑀉𑀘𑁆𑀘𑀺𑀦𑀣𑀸”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀤𑁂𑀯𑀸”𑀢𑀺 𑀔𑁄 𑀲𑁂𑀦𑀸𑀦𑀸𑀬𑀓𑀸 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀧𑀘𑁆𑀘𑀲𑁆𑀲𑁄𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀦𑀁 𑀅𑀪𑀺𑀜𑁆𑀜𑀸𑀢𑀸𑀦𑀁 𑀬𑁄𑀥𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀫𑀬𑀁 𑀔𑁄 𑀬𑀼𑀤𑁆𑀥𑀸𑀪𑀺𑀦𑀦𑁆𑀤𑀺𑀦𑁄 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀸 𑀧𑀸𑀧𑀜𑁆𑀘 𑀓𑀭𑁄𑀫, 𑀩𑀳𑀼𑀜𑁆𑀘 𑀅𑀧𑀼𑀜𑁆𑀜𑀁 𑀧𑀲𑀯𑀸𑀫𑁇 𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀁 𑀉𑀧𑀸𑀬𑁂𑀦 𑀧𑀸𑀧𑀸 𑀘 𑀯𑀺𑀭𑀫𑁂𑀬𑁆𑀬𑀸𑀫 𑀓𑀮𑁆𑀬𑀸𑀡𑀜𑁆𑀘 𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺?

427\. 𑀅𑀣 𑀔𑁄 𑀢𑁂𑀲𑀁 𑀬𑁄𑀥𑀸𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑁂 𑀔𑁄 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸𑁇 𑀲𑀘𑁂 𑀔𑁄 𑀫𑀬𑀁 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀸𑀫, 𑀏𑀯𑀁 𑀫𑀬𑀁 𑀧𑀸𑀧𑀸 𑀘 𑀯𑀺𑀭𑀫𑁂𑀬𑁆𑀬𑀸𑀫 𑀓𑀮𑁆𑀬𑀸𑀡𑀜𑁆𑀘 𑀓𑀭𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀬𑁄𑀥𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑀁𑀲𑀼𑁇 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑀼𑀁, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀲𑀼𑀁𑁇 𑀲𑁂𑀦𑀸𑀦𑀸𑀬𑀓𑀸 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀭𑀸𑀚𑀪𑀝𑁂 𑀧𑀼𑀘𑁆𑀙𑀺𑀁𑀲𑀼—  “𑀓𑀺𑀁 𑀦𑀼 𑀔𑁄, 𑀪𑀡𑁂, 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀬𑁄𑀥𑀸 𑀦 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺? “𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘 𑀇𑀢𑁆𑀣𑀦𑁆𑀦𑀸𑀫𑁄 𑀘, 𑀲𑀸𑀫𑀺, 𑀬𑁄𑀥𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸”𑀢𑀺𑁇 𑀲𑁂𑀦𑀸𑀦𑀸𑀬𑀓𑀸 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀭𑀸𑀚𑀪𑀝𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀲𑁂𑀦𑀸𑀦𑀸𑀬𑀓𑀸 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑀸 𑀭𑀜𑁆𑀜𑁄 𑀫𑀸𑀕𑀥𑀲𑁆𑀲 𑀲𑁂𑀦𑀺𑀬𑀲𑁆𑀲 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀲𑁆𑀲 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

428\. 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀯𑁄𑀳𑀸𑀭𑀺𑀓𑁂 𑀫𑀳𑀸𑀫𑀢𑁆𑀢𑁂 𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀬𑁄, 𑀪𑀡𑁂, 𑀭𑀸𑀚𑀪𑀝𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀺, 𑀓𑀺𑀁 𑀲𑁄 𑀧𑀲𑀯𑀢𑀻”𑀢𑀺? “𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑀲𑁆𑀲, 𑀤𑁂𑀯, 𑀲𑀻𑀲𑀁 𑀙𑁂𑀢𑀩𑁆𑀩𑀁, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀓𑀲𑁆𑀲 𑀚𑀺𑀯𑁆𑀳𑀸 𑀉𑀤𑁆𑀥𑀭𑀺𑀢𑀩𑁆𑀩𑀸, 𑀕𑀡𑀲𑁆𑀲 𑀉𑀧𑀟𑁆𑀠𑀨𑀸𑀲𑀼𑀓𑀸 𑀪𑀜𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑁄 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘—  “𑀲𑀦𑁆𑀢𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀭𑀸𑀚𑀸𑀦𑁄 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑁇 𑀢𑁂 𑀅𑀧𑁆𑀧𑀫𑀢𑁆𑀢𑀓𑁂𑀦𑀧𑀺 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀳𑁂𑀞𑁂𑀬𑁆𑀬𑀼𑀁𑁇 𑀲𑀸𑀥𑀼, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀬𑁆𑀬𑀸 𑀭𑀸𑀚𑀪𑀝𑀁 𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀸𑀦𑀁 𑀫𑀸𑀕𑀥𑀁 𑀲𑁂𑀦𑀺𑀬𑀁 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑀁 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑁂𑀲𑀺 𑀲𑀫𑀸𑀤𑀧𑁂𑀲𑀺 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑁂𑀲𑀺 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑁂𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀪𑀕𑀯𑀢𑀸 𑀥𑀫𑁆𑀫𑀺𑀬𑀸 𑀓𑀣𑀸𑀬 𑀲𑀦𑁆𑀤𑀲𑁆𑀲𑀺𑀢𑁄 𑀲𑀫𑀸𑀤𑀧𑀺𑀢𑁄 𑀲𑀫𑀼𑀢𑁆𑀢𑁂𑀚𑀺𑀢𑁄 𑀲𑀫𑁆𑀧𑀳𑀁𑀲𑀺𑀢𑁄 𑀉𑀝𑁆𑀞𑀸𑀬𑀸𑀲𑀦𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀧𑀤𑀓𑁆𑀔𑀺𑀡𑀁 𑀓𑀢𑁆𑀯𑀸 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀚𑀪𑀝𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
