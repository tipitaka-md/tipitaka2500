# 𑁧𑁇𑁫𑁧 𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀯𑀢𑁆𑀣𑀼

(113.)

487. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀧𑀺𑀢𑀭𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀲𑀺𑁇 𑀲𑁄 𑀢𑁂𑀦 𑀧𑀸𑀧𑀓𑁂𑀦 𑀓𑀫𑁆𑀫𑁂𑀦 𑀅𑀝𑁆𑀝𑀻𑀬𑀢𑀺 𑀳𑀭𑀸𑀬𑀢𑀺 𑀚𑀺𑀕𑀼𑀘𑁆𑀙𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀡𑀯𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑁂𑀦 𑀦𑀼 𑀔𑁄 𑀅𑀳𑀁 𑀉𑀧𑀸𑀬𑁂𑀦 𑀇𑀫𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀫𑀸𑀡𑀯𑀓𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀇𑀫𑁂 𑀔𑁄 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀥𑀫𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄 𑀲𑀘𑁆𑀘𑀯𑀸𑀤𑀺𑀦𑁄 𑀲𑀻𑀮𑀯𑀦𑁆𑀢𑁄 𑀓𑀮𑁆𑀬𑀸𑀡𑀥𑀫𑁆𑀫𑀸, 𑀲𑀘𑁂 𑀔𑁄 𑀅𑀳𑀁 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑁂𑀬𑁆𑀬𑀁, 𑀏𑀯𑀸𑀳𑀁 𑀇𑀫𑀲𑁆𑀲 𑀧𑀸𑀧𑀓𑀲𑁆𑀲 𑀓𑀫𑁆𑀫𑀲𑁆𑀲 𑀦𑀺𑀓𑁆𑀔𑀦𑁆𑀢𑀺𑀁 𑀓𑀭𑁂𑀬𑁆𑀬𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀸𑀮𑀺𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀧𑀼𑀩𑁆𑀩𑁂𑀧𑀺 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄 𑀉𑀧𑀸𑀮𑀺, 𑀦𑀸𑀕𑁄 𑀫𑀸𑀡𑀯𑀓𑀯𑀡𑁆𑀡𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑁇 𑀇𑀗𑁆𑀖𑀸𑀯𑀼𑀲𑁄, 𑀉𑀧𑀸𑀮𑀺, 𑀇𑀫𑀁 𑀫𑀸𑀡𑀯𑀓𑀁 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀸𑀳𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀫𑀸𑀡𑀯𑀓𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀧𑀸𑀮𑀺𑀦𑀸 𑀅𑀦𑀼𑀬𑀼𑀜𑁆𑀚𑀻𑀬𑀫𑀸𑀦𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀸𑀮𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄, 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 (𑁬)
