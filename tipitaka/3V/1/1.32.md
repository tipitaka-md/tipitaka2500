# 𑁧𑁇𑁩𑁨 𑀮𑀓𑁆𑀔𑀡𑀸𑀳𑀢𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀮𑀓𑁆𑀔𑀡𑀸𑀳𑀢𑀯𑀢𑁆𑀣𑀼

(95.)

433\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀮𑀓𑁆𑀔𑀡𑀸𑀳𑀢𑁄 𑀓𑀢𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀮𑀓𑁆𑀔𑀡𑀸𑀳𑀢𑀁 𑀓𑀢𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑀓𑁆𑀔𑀡𑀸𑀳𑀢𑁄 𑀓𑀢𑀤𑀡𑁆𑀟𑀓𑀫𑁆𑀫𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
