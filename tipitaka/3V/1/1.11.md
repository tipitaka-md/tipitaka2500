# 𑁧𑁇𑁧𑁧 𑀪𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑀺𑀬𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀪𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑀺𑀬𑀯𑀢𑁆𑀣𑀼

(36.)

152\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀩𑀸𑀭𑀸𑀡𑀲𑀺𑀬𑀁 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀉𑀭𑀼𑀯𑁂𑀮𑀸 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀫𑀕𑁆𑀕𑀸 𑀑𑀓𑁆𑀓𑀫𑁆𑀫 𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀯𑀦𑀲𑀡𑁆𑀟𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀢𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀅𑀚𑁆𑀛𑁄𑀕𑀸𑀳𑁂𑀢𑁆𑀯𑀸 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀻𑀤𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀢𑀺𑀁𑀲𑀫𑀢𑁆𑀢𑀸 𑀪𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀲𑀳𑀸𑀬𑀓𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀸 𑀢𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀓𑀲𑁆𑀲 𑀧𑀚𑀸𑀧𑀢𑀺 𑀦𑀸𑀳𑁄𑀲𑀺; 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑁂𑀲𑀻 𑀆𑀦𑀻𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸 𑀯𑁂𑀲𑀻 𑀢𑁂𑀲𑀼 𑀧𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁂𑀲𑀼 𑀪𑀡𑁆𑀟𑀁 𑀆𑀤𑀸𑀬 𑀧𑀮𑀸𑀬𑀺𑀢𑁆𑀣𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀲𑀳𑀸𑀬𑀓𑀸 𑀲𑀳𑀸𑀬𑀓𑀲𑁆𑀲 𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸, 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑀸, 𑀢𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀦𑁆𑀢𑀸 𑀅𑀤𑁆𑀤𑀲𑀁𑀲𑀼 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑁂 𑀦𑀺𑀲𑀺𑀦𑁆𑀦𑀁𑁇 𑀤𑀺𑀲𑁆𑀯𑀸𑀦 𑀬𑁂𑀦 𑀪𑀕𑀯𑀸 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀁𑀲𑀼, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀅𑀧𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀸 𑀏𑀓𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀧𑀲𑁆𑀲𑁂𑀬𑁆𑀬𑀸”𑀢𑀺? “𑀓𑀺𑀁 𑀧𑀦 𑀯𑁄, 𑀓𑀼𑀫𑀸𑀭𑀸, 𑀇𑀢𑁆𑀣𑀺𑀬𑀸”𑀢𑀺? “𑀇𑀥 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀢𑀺𑀁𑀲𑀫𑀢𑁆𑀢𑀸 𑀪𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀲𑀳𑀸𑀬𑀓𑀸 𑀲𑀧𑀚𑀸𑀧𑀢𑀺𑀓𑀸 𑀇𑀫𑀲𑁆𑀫𑀺𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑁂 𑀧𑀭𑀺𑀘𑀸𑀭𑀺𑀫𑁆𑀳𑀸𑁇 𑀏𑀓𑀲𑁆𑀲 𑀧𑀚𑀸𑀧𑀢𑀺 𑀦𑀸𑀳𑁄𑀲𑀺; 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑁂𑀲𑀻 𑀆𑀦𑀻𑀢𑀸 𑀅𑀳𑁄𑀲𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀲𑀸, 𑀪𑀦𑁆𑀢𑁂, 𑀯𑁂𑀲𑀻 𑀅𑀫𑁆𑀳𑁂𑀲𑀼 𑀧𑀫𑀢𑁆𑀢𑁂𑀲𑀼 𑀧𑀭𑀺𑀘𑀸𑀭𑁂𑀦𑁆𑀢𑁂𑀲𑀼 𑀪𑀡𑁆𑀟𑀁 𑀆𑀤𑀸𑀬 𑀧𑀮𑀸𑀬𑀺𑀢𑁆𑀣𑁇 𑀢𑁂 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀲𑀳𑀸𑀬𑀓𑀸 𑀲𑀳𑀸𑀬𑀓𑀲𑁆𑀲 𑀯𑁂𑀬𑁆𑀬𑀸𑀯𑀘𑁆𑀘𑀁 𑀓𑀭𑁄𑀦𑁆𑀢𑀸, 𑀢𑀁 𑀇𑀢𑁆𑀣𑀺𑀁 𑀕𑀯𑁂𑀲𑀦𑁆𑀢𑀸, 𑀇𑀫𑀁 𑀯𑀦𑀲𑀡𑁆𑀟𑀁 𑀆𑀳𑀺𑀡𑁆𑀟𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀢𑀁 𑀓𑀺𑀁 𑀫𑀜𑁆𑀜𑀣 𑀯𑁄, 𑀓𑀼𑀫𑀸𑀭𑀸, 𑀓𑀢𑀫𑀁 𑀦𑀼 𑀔𑁄 𑀢𑀼𑀫𑁆𑀳𑀸𑀓𑀁 𑀯𑀭𑀁—  𑀬𑀁 𑀯𑀸 𑀢𑀼𑀫𑁆𑀳𑁂 𑀇𑀢𑁆𑀣𑀺𑀁 𑀕𑀯𑁂𑀲𑁂𑀬𑁆𑀬𑀸𑀣, 𑀬𑀁 𑀯𑀸 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀕𑀯𑁂𑀲𑁂𑀬𑁆𑀬𑀸𑀣𑀸”𑀢𑀺? “𑀏𑀢𑀤𑁂𑀯, 𑀪𑀦𑁆𑀢𑁂, 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀯𑀭𑀁 𑀬𑀁 𑀫𑀬𑀁 𑀅𑀢𑁆𑀢𑀸𑀦𑀁 𑀕𑀯𑁂𑀲𑁂𑀬𑁆𑀬𑀸𑀫𑀸”𑀢𑀺𑁇 “𑀢𑁂𑀦 𑀳𑀺 𑀯𑁄, 𑀓𑀼𑀫𑀸𑀭𑀸, 𑀦𑀺𑀲𑀻𑀤𑀣, 𑀥𑀫𑁆𑀫𑀁 𑀯𑁄 𑀤𑁂𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 “𑀏𑀯𑀁, 𑀪𑀦𑁆𑀢𑁂”𑀢𑀺 𑀔𑁄 𑀢𑁂 𑀪𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑀺𑀬𑀸 𑀲𑀳𑀸𑀬𑀓𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀅𑀪𑀺𑀯𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀫𑀦𑁆𑀢𑀁 𑀦𑀺𑀲𑀻𑀤𑀺𑀁𑀲𑀼𑁇

153\. 𑀢𑁂𑀲𑀁 𑀪𑀕𑀯𑀸 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑀺𑀁 𑀓𑀣𑀁 𑀓𑀣𑁂𑀲𑀺, 𑀲𑁂𑀬𑁆𑀬𑀣𑀺𑀤𑀁—  𑀤𑀸𑀦𑀓𑀣𑀁 𑀲𑀻𑀮𑀓𑀣𑀁 𑀲𑀕𑁆𑀕𑀓𑀣𑀁 𑀓𑀸𑀫𑀸𑀦𑀁 𑀆𑀤𑀻𑀦𑀯𑀁 𑀑𑀓𑀸𑀭𑀁 𑀲𑀁𑀓𑀺𑀮𑁂𑀲𑀁 𑀦𑁂𑀓𑁆𑀔𑀫𑁆𑀫𑁂 𑀆𑀦𑀺𑀲𑀁𑀲𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺𑁇 𑀬𑀤𑀸 𑀢𑁂 𑀪𑀕𑀯𑀸 𑀅𑀜𑁆𑀜𑀸𑀲𑀺 𑀓𑀮𑁆𑀮𑀘𑀺𑀢𑁆𑀢𑁂 𑀫𑀼𑀤𑀼𑀘𑀺𑀢𑁆𑀢𑁂 𑀯𑀺𑀦𑀻𑀯𑀭𑀡𑀘𑀺𑀢𑁆𑀢𑁂 𑀉𑀤𑀕𑁆𑀕𑀘𑀺𑀢𑁆𑀢𑁂 𑀧𑀲𑀦𑁆𑀦𑀘𑀺𑀢𑁆𑀢𑁂, 𑀅𑀣 𑀬𑀸 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀁 𑀲𑀸𑀫𑀼𑀓𑁆𑀓𑀁𑀲𑀺𑀓𑀸 𑀥𑀫𑁆𑀫𑀤𑁂𑀲𑀦𑀸, 𑀢𑀁 𑀧𑀓𑀸𑀲𑁂𑀲𑀺 𑀤𑀼𑀓𑁆𑀔𑀁 𑀲𑀫𑀼𑀤𑀬𑀁 𑀦𑀺𑀭𑁄𑀥𑀁 𑀫𑀕𑁆𑀕𑀁𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀲𑀼𑀤𑁆𑀥𑀁 𑀯𑀢𑁆𑀣𑀁 𑀅𑀧𑀕𑀢𑀓𑀸𑀍𑀅𑀓𑀁 𑀲𑀫𑁆𑀫𑀤𑁂𑀯 𑀭𑀚𑀦𑀁 𑀧𑀝𑀺𑀕𑁆𑀕𑀡𑁆𑀳𑁂𑀬𑁆𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀢𑁂𑀲𑀁 𑀢𑀲𑁆𑀫𑀺𑀁𑀬𑁂𑀯 𑀆𑀲𑀦𑁂 𑀯𑀺𑀭𑀚𑀁 𑀯𑀻𑀢𑀫𑀮𑀁 𑀥𑀫𑁆𑀫𑀘𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑀧𑀸𑀤𑀺—  “𑀬𑀁 𑀓𑀺𑀜𑁆𑀘𑀺 𑀲𑀫𑀼𑀤𑀬𑀥𑀫𑁆𑀫𑀁 𑀲𑀩𑁆𑀩𑀁 𑀢𑀁 𑀦𑀺𑀭𑁄𑀥𑀥𑀫𑁆𑀫𑀦𑁆”𑀢𑀺𑁇 𑀢𑁂 𑀤𑀺𑀝𑁆𑀞𑀥𑀫𑁆𑀫𑀸 𑀧𑀢𑁆𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀺𑀤𑀺𑀢𑀥𑀫𑁆𑀫𑀸 𑀧𑀭𑀺𑀬𑁄𑀕𑀸𑀍𑀳𑀥𑀫𑁆𑀫𑀸 𑀢𑀺𑀡𑁆𑀡𑀯𑀺𑀘𑀺𑀓𑀺𑀘𑁆𑀙𑀸 𑀯𑀺𑀕𑀢𑀓𑀣𑀁𑀓𑀣𑀸 𑀯𑁂𑀲𑀸𑀭𑀚𑁆𑀚𑀧𑁆𑀧𑀢𑁆𑀢𑀸 𑀅𑀧𑀭𑀧𑁆𑀧𑀘𑁆𑀘𑀬𑀸 𑀲𑀢𑁆𑀣𑀼𑀲𑀸𑀲𑀦𑁂 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀫𑀬𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀪𑀕𑀯𑀢𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁, 𑀮𑀪𑁂𑀬𑁆𑀬𑀸𑀫 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀦𑁆”𑀢𑀺𑁇 “𑀏𑀣 𑀪𑀺𑀓𑁆𑀔𑀯𑁄”𑀢𑀺 𑀪𑀕𑀯𑀸 𑀅𑀯𑁄𑀘—  “𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀣 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀲𑀸𑀯 𑀢𑁂𑀲𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸 𑀅𑀳𑁄𑀲𑀺𑁇

---

154\. 𑀪𑀤𑁆𑀤𑀯𑀕𑁆𑀕𑀺𑀬𑀲𑀳𑀸𑀬𑀓𑀸𑀦𑀁 𑀯𑀢𑁆𑀣𑀼 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀁𑁇
