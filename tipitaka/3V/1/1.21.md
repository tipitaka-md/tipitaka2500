# 𑁧𑁇𑁨𑁧 𑀩𑀸𑀮𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀩𑀸𑀮𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀯𑀢𑁆𑀣𑀼

(82.)

363. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀤𑀲𑀯𑀲𑁆𑀲𑀫𑁆𑀳𑀸 𑀤𑀲𑀯𑀲𑁆𑀲𑀫𑁆𑀳𑀸𑀢𑀺, 𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀩𑀸𑀮𑀸, 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸𑁇 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸, 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑀸𑁇 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀸, 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸𑁇 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸, 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁄𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀺𑀓𑁆𑀔𑀽—  𑀤𑀲𑀯𑀲𑁆𑀲𑀫𑁆𑀳𑀸 𑀤𑀲𑀯𑀲𑁆𑀲𑀫𑁆𑀳𑀸𑀢𑀺—  𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀩𑀸𑀮𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀧𑀡𑁆𑀟𑀺𑀢𑀸, 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑁆𑀬𑀢𑁆𑀢𑀸, 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀅𑀧𑁆𑀧𑀲𑁆𑀲𑀼𑀢𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀩𑀳𑀼𑀲𑁆𑀲𑀼𑀢𑀸, 𑀤𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀆𑀘𑀭𑀺𑀬𑀸 𑀤𑀼𑀧𑁆𑀧𑀜𑁆𑀜𑀸 𑀅𑀦𑁆𑀢𑁂𑀯𑀸𑀲𑀺𑀓𑀸 𑀧𑀜𑁆𑀜𑀯𑀦𑁆𑀢𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀽—  𑀤𑀲𑀯𑀲𑁆𑀲𑀫𑁆𑀳𑀸 𑀤𑀲𑀯𑀲𑁆𑀲𑀫𑁆𑀳𑀸𑀢𑀺—  𑀩𑀸𑀮𑀸 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑀸𑀮𑁂𑀦 𑀅𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀦𑀺𑀲𑁆𑀲𑀬𑁄 𑀤𑀸𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀤𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇 𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀩𑁆𑀬𑀢𑁆𑀢𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀧𑀝𑀺𑀩𑀮𑁂𑀦 𑀤𑀲𑀯𑀲𑁆𑀲𑁂𑀦 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀤𑀲𑀯𑀲𑁆𑀲𑁂𑀦 𑀯𑀸 𑀦𑀺𑀲𑁆𑀲𑀬𑀁 𑀤𑀸𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
