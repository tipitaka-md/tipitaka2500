# 𑁧𑁇𑁪𑁭 𑀧𑀡𑁆𑀟𑀓𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀡𑁆𑀟𑀓𑀯𑀢𑁆𑀣𑀼

(109.)

482. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀡𑁆𑀟𑀓𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀤𑀳𑀭𑁂 𑀤𑀳𑀭𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀏𑀣, 𑀫𑀁 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑁄 𑀤𑀽𑀲𑁂𑀣𑀸”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺—  “𑀦𑀲𑁆𑀲, 𑀧𑀡𑁆𑀟𑀓, 𑀯𑀺𑀦𑀲𑁆𑀲, 𑀧𑀡𑁆𑀟𑀓, 𑀓𑁄 𑀢𑀬𑀸 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀳𑀺 𑀅𑀧𑀲𑀸𑀤𑀺𑀢𑁄 𑀫𑀳𑀦𑁆𑀢𑁂 𑀫𑀳𑀦𑁆𑀢𑁂 𑀫𑁄𑀍𑀇𑀕𑀮𑁆𑀮𑁂 𑀲𑀸𑀫𑀡𑁂𑀭𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀏𑀣, 𑀫𑀁 𑀆𑀯𑀼𑀲𑁄 𑀤𑀽𑀲𑁂𑀣𑀸”𑀢𑀺𑁇 𑀲𑀸𑀫𑀡𑁂𑀭𑀸 𑀅𑀧𑀲𑀸𑀤𑁂𑀦𑁆𑀢𑀺—  “𑀦𑀲𑁆𑀲, 𑀧𑀡𑁆𑀟𑀓, 𑀯𑀺𑀦𑀲𑁆𑀲, 𑀧𑀡𑁆𑀟𑀓, 𑀓𑁄 𑀢𑀬𑀸 𑀅𑀢𑁆𑀣𑁄”𑀢𑀺𑁇 𑀲𑁄 𑀲𑀸𑀫𑀡𑁂𑀭𑁂𑀳𑀺 𑀅𑀧𑀲𑀸𑀤𑀺𑀢𑁄 𑀳𑀢𑁆𑀣𑀺𑀪𑀡𑁆𑀟𑁂 𑀅𑀲𑁆𑀲𑀪𑀡𑁆𑀟𑁂 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀁 𑀯𑀤𑁂𑀢𑀺—  “𑀏𑀣, 𑀫𑀁 𑀆𑀯𑀼𑀲𑁄 𑀤𑀽𑀲𑁂𑀣𑀸”𑀢𑀺𑁇 𑀳𑀢𑁆𑀣𑀺𑀪𑀡𑁆𑀟𑀸 𑀅𑀲𑁆𑀲𑀪𑀡𑁆𑀟𑀸 𑀤𑀽𑀲𑁂𑀲𑀼𑀁𑁇 𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀧𑀡𑁆𑀟𑀓𑀸 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸𑁇 𑀬𑁂𑀧𑀺 𑀇𑀫𑁂𑀲𑀁 𑀦 𑀧𑀡𑁆𑀟𑀓𑀸, 𑀢𑁂𑀧𑀺 𑀇𑀫𑁂 𑀧𑀡𑁆𑀟𑀓𑁂 𑀤𑀽𑀲𑁂𑀦𑁆𑀢𑀺𑁇 𑀏𑀯𑀁 𑀇𑀫𑁂 𑀲𑀩𑁆𑀩𑁂𑀯 𑀅𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀸𑀭𑀺𑀦𑁄”𑀢𑀺𑁇 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀳𑀢𑁆𑀣𑀺𑀪𑀡𑁆𑀟𑀸𑀦𑀁 𑀅𑀲𑁆𑀲𑀪𑀡𑁆𑀟𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀧𑀡𑁆𑀟𑀓𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄, 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀦𑀸𑀲𑁂𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇 (𑁧)
