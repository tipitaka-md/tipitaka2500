# 𑁧𑁇𑁫𑁫 𑀅𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑀸𑀤𑀺𑀯𑀢𑁆𑀣𑀼

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑀸𑀤𑀺𑀯𑀢𑁆𑀣𑀼

(117.)

491. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀦𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑀓𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁨)

492. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁩)

493. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀕𑀡𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑀡𑁂𑀦 𑀉𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁪)

494. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀡𑁆𑀟𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺…𑀧𑁂…  𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀺 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀡𑁆𑀟𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀣𑁂𑀬𑁆𑀬𑀲𑀁𑀯𑀸𑀲𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀧𑀓𑁆𑀓𑀦𑁆𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀫𑀸𑀢𑀼𑀖𑀸𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀺𑀢𑀼𑀖𑀸𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀭𑀳𑀦𑁆𑀢𑀖𑀸𑀢𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀤𑀽𑀲𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀪𑁂𑀤𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀮𑁄𑀳𑀺𑀢𑀼𑀧𑁆𑀧𑀸𑀤𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄…𑀧𑁂…  𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀪𑀢𑁄𑀩𑁆𑀬𑀜𑁆𑀚𑀦𑀓𑀼𑀧𑀚𑁆𑀛𑀸𑀬𑁂𑀦 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇 (𑁧𑁫)
