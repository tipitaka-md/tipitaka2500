# 𑁧𑁇𑁬𑁦 𑀕𑁄𑀢𑁆𑀢𑁂𑀦𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀸𑀦𑀼𑀚𑀸𑀦𑀦

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀕𑁄𑀢𑁆𑀢𑁂𑀦𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑀦𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(122.)

511\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑀲𑁆𑀲 𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀸𑀧𑁂𑀓𑁆𑀔𑁄 𑀳𑁄𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀆𑀬𑀲𑁆𑀫𑀸 𑀫𑀳𑀸𑀓𑀲𑁆𑀲𑀧𑁄 𑀆𑀬𑀲𑁆𑀫𑀢𑁄 𑀆𑀦𑀦𑁆𑀤𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀆𑀦𑀦𑁆𑀤𑁄 𑀇𑀫𑀁 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑁆𑀲𑀢𑀽”𑀢𑀺𑁇 𑀆𑀬𑀲𑁆𑀫𑀸 𑀆𑀦𑀦𑁆𑀤𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀦𑀸𑀳𑀁 𑀉𑀲𑁆𑀲𑀳𑀸𑀫𑀺 𑀣𑁂𑀭𑀲𑁆𑀲 𑀦𑀸𑀫𑀁 𑀕𑀳𑁂𑀢𑀼𑀁, 𑀕𑀭𑀼 𑀫𑁂 𑀣𑁂𑀭𑁄”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀕𑁄𑀢𑁆𑀢𑁂𑀦𑀧𑀺 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
