# 𑁧𑁇𑁬𑁫 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀅𑀓𑀭𑀡𑀻𑀬

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀘𑀢𑁆𑀢𑀸𑀭𑀺𑀅𑀓𑀭𑀡𑀻𑀬

(129.)

539\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀜𑁆𑀜𑀢𑀭𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀏𑀓𑀓𑀁 𑀑𑀳𑀸𑀬 𑀧𑀓𑁆𑀓𑀫𑀺𑀁𑀲𑀼𑁇 𑀲𑁄 𑀧𑀘𑁆𑀙𑀸 𑀏𑀓𑀓𑁄𑀯 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀧𑀼𑀭𑀸𑀡𑀤𑀼𑀢𑀺𑀬𑀺𑀓𑀸𑀬 𑀲𑀫𑀸𑀕𑀜𑁆𑀙𑀺𑁇 𑀲𑀸 𑀏𑀯𑀫𑀸𑀳—  “𑀓𑀺𑀁 𑀤𑀸𑀦𑀺 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀲𑀻”𑀢𑀺? “𑀆𑀫 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄𑀫𑁆𑀳𑀻”𑀢𑀺𑁇 “𑀤𑀼𑀮𑁆𑀮𑀪𑁄 𑀔𑁄 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑀸𑀦𑀁 𑀫𑁂𑀣𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑁄; 𑀏𑀳𑀺, 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀸”𑀢𑀺𑁇 𑀲𑁄 𑀢𑀲𑁆𑀲𑀸 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀘𑀺𑀭𑁂𑀦 𑀅𑀕𑀫𑀸𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀓𑀺𑀲𑁆𑀲 𑀢𑁆𑀯𑀁, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀁 𑀘𑀺𑀭𑀁 𑀅𑀓𑀸𑀲𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀲𑀫𑁆𑀧𑀸𑀤𑁂𑀢𑁆𑀯𑀸 𑀤𑀼𑀢𑀺𑀬𑀁 𑀤𑀸𑀢𑀼𑀁, 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀘 𑀅𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀢𑀼𑀁—

540\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀫𑁂𑀣𑀼𑀦𑁄 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀺𑀭𑀘𑁆𑀙𑀸𑀦𑀕𑀢𑀸𑀬𑀧𑀺𑁇 𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀢𑀺, 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀼𑀭𑀺𑀲𑁄 𑀲𑀻𑀲𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀢𑁂𑀦 𑀲𑀭𑀻𑀭𑀩𑀦𑁆𑀥𑀦𑁂𑀦 𑀚𑀻𑀯𑀺𑀢𑀼𑀁; 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑁂𑀣𑀼𑀦𑀁 𑀥𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀲𑁂𑀯𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀢𑀁 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 (𑁧)

541\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀦 𑀆𑀤𑀸𑀢𑀩𑁆𑀩𑀁, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀢𑀺𑀡𑀲𑀮𑀸𑀓𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀤𑀁 𑀯𑀸 𑀧𑀸𑀤𑀸𑀭𑀳𑀁 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀸𑀤𑀁 𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑀢𑀺, 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀡𑁆𑀟𑀼𑀧𑀮𑀸𑀲𑁄 𑀩𑀦𑁆𑀥𑀦𑀸 𑀧𑀯𑀼𑀢𑁆𑀢𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀳𑀭𑀺𑀢𑀢𑁆𑀣𑀸𑀬; 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀤𑀁 𑀯𑀸 𑀧𑀸𑀤𑀸𑀭𑀳𑀁 𑀯𑀸 𑀅𑀢𑀺𑀭𑁂𑀓𑀧𑀸𑀤𑀁 𑀯𑀸 𑀅𑀤𑀺𑀦𑁆𑀦𑀁 𑀣𑁂𑀬𑁆𑀬𑀲𑀗𑁆𑀔𑀸𑀢𑀁 𑀆𑀤𑀺𑀬𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀢𑀁 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 (𑁨)

542\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀧𑀸𑀡𑁄 𑀚𑀻𑀯𑀺𑀢𑀸 𑀦 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀩𑁆𑀩𑁄, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀓𑀼𑀦𑁆𑀣𑀓𑀺𑀧𑀺𑀮𑁆𑀮𑀺𑀓𑀁 𑀉𑀧𑀸𑀤𑀸𑀬𑁇 𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑀺, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 𑀕𑀩𑁆𑀪𑀧𑀸𑀢𑀦𑀁 𑀉𑀧𑀸𑀤𑀸𑀬, 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀧𑀼𑀣𑀼𑀲𑀺𑀮𑀸 𑀤𑁆𑀯𑁂𑀥𑀸 𑀪𑀺𑀦𑁆𑀦𑀸 𑀅𑀧𑁆𑀧𑀝𑀺𑀲𑀦𑁆𑀥𑀺𑀓𑀸 𑀳𑁄𑀢𑀺; 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀫𑀦𑀼𑀲𑁆𑀲𑀯𑀺𑀕𑁆𑀕𑀳𑀁 𑀚𑀻𑀯𑀺𑀢𑀸 𑀯𑁄𑀭𑁄𑀧𑁂𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀢𑀁 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀁𑁇 (𑁩)

543\. 𑀉𑀧𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑁄 𑀦 𑀉𑀮𑁆𑀮𑀧𑀺𑀢𑀩𑁆𑀩𑁄, 𑀅𑀦𑁆𑀢𑀫𑀲𑁄 ‘𑀲𑀼𑀜𑁆𑀜𑀸𑀕𑀸𑀭𑁂 𑀅𑀪𑀺𑀭𑀫𑀸𑀫𑀻’𑀢𑀺𑁇 𑀬𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀅𑀲𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀢𑀺 𑀛𑀸𑀦𑀁 𑀯𑀸 𑀯𑀺𑀫𑁄𑀓𑁆𑀔𑀁 𑀯𑀸 𑀲𑀫𑀸𑀥𑀺𑀁 𑀯𑀸 𑀲𑀫𑀸𑀧𑀢𑁆𑀢𑀺𑀁 𑀯𑀸 𑀫𑀕𑁆𑀕𑀁 𑀯𑀸 𑀨𑀮𑀁 𑀯𑀸, 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀦𑀸𑀫 𑀢𑀸𑀮𑁄 𑀫𑀢𑁆𑀣𑀓𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀅𑀪𑀩𑁆𑀩𑁄 𑀧𑀼𑀦 𑀯𑀺𑀭𑀼𑀍𑀳𑀺𑀬𑀸; 𑀏𑀯𑀫𑁂𑀯 𑀪𑀺𑀓𑁆𑀔𑀼 𑀧𑀸𑀧𑀺𑀘𑁆𑀙𑁄 𑀇𑀘𑁆𑀙𑀸𑀧𑀓𑀢𑁄 𑀅𑀲𑀦𑁆𑀢𑀁 𑀅𑀪𑀽𑀢𑀁 𑀉𑀢𑁆𑀢𑀭𑀺𑀫𑀦𑀼𑀲𑁆𑀲𑀥𑀫𑁆𑀫𑀁 𑀉𑀮𑁆𑀮𑀧𑀺𑀢𑁆𑀯𑀸 𑀅𑀲𑁆𑀲𑀫𑀡𑁄 𑀳𑁄𑀢𑀺 𑀅𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁄𑁇 𑀢𑀁 𑀢𑁂 𑀬𑀸𑀯𑀚𑀻𑀯𑀁 𑀅𑀓𑀭𑀡𑀻𑀬𑀦𑁆”𑀢𑀺𑁇 (𑁪)

---

544\. 𑀘𑀢𑁆𑀢𑀸𑀭𑀺 𑀅𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀺 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑀦𑀺𑁇
