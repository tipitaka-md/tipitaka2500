# 𑁧𑁇𑁨𑁯 𑀓𑀸𑀭𑀪𑁂𑀤𑀓𑀘𑁄𑀭𑀯𑀢𑁆𑀣𑀼

* 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

* 𑀓𑀸𑀭𑀪𑁂𑀤𑀓𑀘𑁄𑀭𑀯𑀢𑁆𑀣𑀼

(92.)

430\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀜𑁆𑀜𑀸 𑀫𑀸𑀕𑀥𑁂𑀦 𑀲𑁂𑀦𑀺𑀬𑁂𑀦 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁂𑀦 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀳𑁄𑀢𑀺—  “𑀬𑁂 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁂 𑀮𑀩𑁆𑀪𑀸 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀸𑀢𑀼𑀁; 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀦𑁆𑀢𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸”𑀢𑀺𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀧𑀼𑀭𑀺𑀲𑁄 𑀘𑁄𑀭𑀺𑀓𑀁 𑀓𑀢𑁆𑀯𑀸 𑀓𑀸𑀭𑀸𑀬 𑀩𑀤𑁆𑀥𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀓𑀸𑀭𑀁 𑀪𑀺𑀦𑁆𑀤𑀺𑀢𑁆𑀯𑀸 𑀧𑀮𑀸𑀬𑀺𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀧𑀲𑁆𑀲𑀺𑀢𑁆𑀯𑀸 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀅𑀬𑀁 𑀲𑁄 𑀓𑀸𑀭𑀪𑁂𑀤𑀓𑁄 𑀘𑁄𑀭𑁄𑁇 𑀳𑀦𑁆𑀤 𑀦𑀁 𑀦𑁂𑀫𑀸”𑀢𑀺𑁇 𑀏𑀓𑀘𑁆𑀘𑁂 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀫𑀸𑀬𑁆𑀬𑁄, 𑀏𑀯𑀁 𑀅𑀯𑀘𑀼𑀢𑁆𑀣𑁇 𑀅𑀦𑀼𑀜𑁆𑀜𑀸𑀢𑀁 𑀭𑀜𑁆𑀜𑀸 𑀫𑀸𑀕𑀥𑁂𑀦 𑀲𑁂𑀦𑀺𑀬𑁂𑀦 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁂𑀦—  ‘𑀬𑁂 𑀲𑀫𑀡𑁂𑀲𑀼 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑁂𑀲𑀼 𑀧𑀩𑁆𑀩𑀚𑀦𑁆𑀢𑀺, 𑀦 𑀢𑁂 𑀮𑀩𑁆𑀪𑀸 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀸𑀢𑀼𑀁; 𑀲𑁆𑀯𑀸𑀓𑁆𑀔𑀸𑀢𑁄 𑀥𑀫𑁆𑀫𑁄, 𑀘𑀭𑀦𑁆𑀢𑀼 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀁 𑀲𑀫𑁆𑀫𑀸 𑀤𑀼𑀓𑁆𑀔𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀓𑀺𑀭𑀺𑀬𑀸𑀬𑀸’”𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀅𑀪𑀬𑀽𑀯𑀭𑀸 𑀇𑀫𑁂 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸, 𑀦𑀬𑀺𑀫𑁂 𑀮𑀩𑁆𑀪𑀸 𑀓𑀺𑀜𑁆𑀘𑀺 𑀓𑀸𑀢𑀼𑀁𑁇 𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀓𑀸𑀭𑀪𑁂𑀤𑀓𑀁 𑀘𑁄𑀭𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀓𑀸𑀭𑀪𑁂𑀤𑀓𑁄 𑀘𑁄𑀭𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
