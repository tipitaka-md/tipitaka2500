# 𑁧𑁇𑁪𑁫 𑀅𑀧𑀮𑀸𑀍𑀅𑀦𑀯𑀢𑁆𑀣𑀼

* 𑀅𑀧𑀮𑀸𑀍𑀅𑀦𑀯𑀢𑁆𑀣𑀼

480\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀣𑁂𑀭𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑁂 𑀅𑀧𑀮𑀸𑀍𑀏𑀦𑁆𑀢𑀺𑁇 𑀣𑁂𑀭𑀸 𑀲𑀸𑀫𑀁 𑀤𑀦𑁆𑀢𑀓𑀝𑁆𑀞𑀫𑁆𑀧𑀺 𑀫𑀼𑀔𑁄𑀤𑀓𑀫𑁆𑀧𑀺 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀸 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀜𑁆𑀜𑀲𑁆𑀲 𑀧𑀭𑀺𑀲𑀸 𑀅𑀧𑀮𑀸𑀍𑀏𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀅𑀧𑀮𑀸𑀍𑀏𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
