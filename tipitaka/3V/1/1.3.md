# 𑁧𑁇𑁩 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀓𑀣𑀸

### 𑀫𑀳𑀸𑀔𑀦𑁆𑀥𑀓

### 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀓𑀣𑀸

(5.)

17. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀢𑀫𑁆𑀳𑀸 𑀲𑀫𑀸𑀥𑀺𑀫𑁆𑀳𑀸 𑀯𑀼𑀝𑁆𑀞𑀳𑀺𑀢𑁆𑀯𑀸 𑀅𑀚𑀧𑀸𑀮𑀦𑀺𑀕𑁆𑀭𑁄𑀥𑀫𑀽𑀮𑀸 𑀬𑁂𑀦 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑁄 𑀢𑁂𑀦𑀼𑀧𑀲𑀗𑁆𑀓𑀫𑀺, 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀫𑀽𑀮𑁂 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀏𑀓𑀧𑀮𑁆𑀮𑀗𑁆𑀓𑁂𑀦 𑀦𑀺𑀲𑀻𑀤𑀺 𑀯𑀺𑀫𑀼𑀢𑁆𑀢𑀺𑀲𑀼𑀔𑀧𑀝𑀺𑀲𑀁𑀯𑁂𑀤𑀻𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀫𑀳𑀸 𑀅𑀓𑀸𑀮𑀫𑁂𑀖𑁄 𑀉𑀤𑀧𑀸𑀤𑀺, 𑀲𑀢𑁆𑀢𑀸𑀳𑀯𑀤𑁆𑀤𑀮𑀺𑀓𑀸 𑀲𑀻𑀢𑀯𑀸𑀢𑀤𑀼𑀤𑁆𑀤𑀺𑀦𑀻𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑁄 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀲𑀓𑀪𑀯𑀦𑀸 𑀦𑀺𑀓𑁆𑀔𑀫𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑀁 𑀲𑀢𑁆𑀢𑀓𑁆𑀔𑀢𑁆𑀢𑀼𑀁 𑀪𑁄𑀕𑁂𑀳𑀺 𑀧𑀭𑀺𑀓𑁆𑀔𑀺𑀧𑀺𑀢𑁆𑀯𑀸 𑀉𑀧𑀭𑀺𑀫𑀼𑀤𑁆𑀥𑀦𑀺 𑀫𑀳𑀦𑁆𑀢𑀁 𑀨𑀡𑀁 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀅𑀝𑁆𑀞𑀸𑀲𑀺—  “𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀲𑀻𑀢𑀁, 𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀉𑀡𑁆𑀳𑀁, 𑀫𑀸 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀟𑀁𑀲𑀫𑀓𑀲𑀯𑀸𑀢𑀸𑀢𑀧𑀲𑀭𑀻𑀲𑀧𑀲𑀫𑁆𑀨𑀲𑁆𑀲𑁄”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑁄 𑀦𑀸𑀕𑀭𑀸𑀚𑀸 𑀲𑀢𑁆𑀢𑀸𑀳𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀬𑁂𑀦 𑀯𑀺𑀤𑁆𑀥𑀁 𑀯𑀺𑀕𑀢𑀯𑀮𑀸𑀳𑀓𑀁 𑀤𑁂𑀯𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀓𑀸𑀬𑀸 𑀪𑁄𑀕𑁂 𑀯𑀺𑀦𑀺𑀯𑁂𑀞𑁂𑀢𑁆𑀯𑀸 𑀲𑀓𑀯𑀡𑁆𑀡𑀁 𑀧𑀝𑀺𑀲𑀁𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀫𑀸𑀡𑀯𑀓𑀯𑀡𑁆𑀡𑀁 𑀅𑀪𑀺𑀦𑀺𑀫𑁆𑀫𑀺𑀦𑀺𑀢𑁆𑀯𑀸 𑀪𑀕𑀯𑀢𑁄 𑀧𑀼𑀭𑀢𑁄 𑀅𑀝𑁆𑀞𑀸𑀲𑀺 𑀧𑀜𑁆𑀚𑀮𑀺𑀓𑁄 𑀪𑀕𑀯𑀦𑁆𑀢𑀁 𑀦𑀫𑀲𑁆𑀲𑀫𑀸𑀦𑁄𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀯𑀺𑀤𑀺𑀢𑁆𑀯𑀸 𑀢𑀸𑀬𑀁 𑀯𑁂𑀮𑀸𑀬𑀁 𑀇𑀫𑀁 𑀉𑀤𑀸𑀦𑀁 𑀉𑀤𑀸𑀦𑁂𑀲𑀺—

18. _“𑀲𑀼𑀔𑁄 𑀯𑀺𑀯𑁂𑀓𑁄 𑀢𑀼𑀝𑁆𑀞𑀲𑁆𑀲,_  
_𑀲𑀼𑀢𑀥𑀫𑁆𑀫𑀲𑁆𑀲 𑀧𑀲𑁆𑀲𑀢𑁄;_  
_𑀅𑀩𑁆𑀬𑀸𑀧𑀚𑁆𑀚𑀁 𑀲𑀼𑀔𑀁 𑀮𑁄𑀓𑁂,_  
_𑀧𑀸𑀡𑀪𑀽𑀢𑁂𑀲𑀼 𑀲𑀁𑀬𑀫𑁄𑁇_  


19. _𑀲𑀼𑀔𑀸 𑀯𑀺𑀭𑀸𑀕𑀢𑀸 𑀮𑁄𑀓𑁂,_  
_𑀓𑀸𑀫𑀸𑀦𑀁 𑀲𑀫𑀢𑀺𑀓𑁆𑀓𑀫𑁄;_  
_𑀅𑀲𑁆𑀫𑀺𑀫𑀸𑀦𑀲𑁆𑀲 𑀬𑁄 𑀯𑀺𑀦𑀬𑁄,_  
_𑀏𑀢𑀁 𑀯𑁂 𑀧𑀭𑀫𑀁 𑀲𑀼𑀔𑀦𑁆”𑀢𑀺𑁇_  


---

20. 𑀫𑀼𑀘𑀮𑀺𑀦𑁆𑀤𑀓𑀣𑀸 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
