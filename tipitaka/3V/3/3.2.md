# 𑁩𑁇𑁨 𑀯𑀲𑁆𑀲𑀸𑀦𑁂𑀘𑀸𑀭𑀺𑀓𑀸𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀸𑀤𑀺

### 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀯𑀲𑁆𑀲𑀸𑀦𑁂𑀘𑀸𑀭𑀺𑀓𑀸𑀧𑀝𑀺𑀓𑁆𑀔𑁂𑀧𑀸𑀤𑀺

(185.)

821. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀢𑀣𑁂𑀯 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀕𑀺𑀫𑁆𑀳𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀳𑀭𑀺𑀢𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀦𑁆𑀢𑀸, 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀸, 𑀩𑀳𑀽 𑀔𑀼𑀤𑁆𑀤𑀓𑁂 𑀧𑀸𑀡𑁂 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀸𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀤𑀼𑀭𑀓𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀅𑀮𑁆𑀮𑀻𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀗𑁆𑀓𑀲𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀓𑀼𑀦𑁆𑀢𑀓𑀸 𑀭𑀼𑀓𑁆𑀔𑀕𑁆𑀕𑁂𑀲𑀼 𑀓𑀼𑀮𑀸𑀯𑀓𑀸𑀦𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀅𑀮𑁆𑀮𑀻𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀗𑁆𑀓𑀲𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀧𑀦 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀕𑀺𑀫𑁆𑀳𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀺, 𑀳𑀭𑀺𑀢𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀦𑁆𑀢𑀸, 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀸, 𑀩𑀳𑀽 𑀔𑀼𑀤𑁆𑀤𑀓𑁂 𑀧𑀸𑀡𑁂 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀸”𑀢𑀺𑁇

822. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀅𑀦𑁆𑀢𑀭𑀸𑀯𑀲𑁆𑀲𑀁 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺? 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀅𑀯𑀲𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸𑁇 𑀬𑁄 𑀧𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

(186.)

823. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀇𑀘𑁆𑀙𑀦𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀁 𑀦 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀦 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

824. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀙𑀩𑁆𑀩𑀕𑁆𑀕𑀺𑀬𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑀤𑀳𑀼 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸𑀬 𑀯𑀲𑁆𑀲𑀁 𑀅𑀦𑀼𑀧𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑀸 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀆𑀯𑀸𑀲𑀁 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀢𑀤𑀳𑀼 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸𑀬 𑀯𑀲𑁆𑀲𑀁 𑀅𑀦𑀼𑀧𑀕𑀦𑁆𑀢𑀼𑀓𑀸𑀫𑁂𑀦 𑀲𑀜𑁆𑀘𑀺𑀘𑁆𑀘 𑀆𑀯𑀸𑀲𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑁄𑁇 𑀬𑁄 𑀅𑀢𑀺𑀓𑁆𑀓𑀫𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

825. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀭𑀸𑀚𑀸 𑀫𑀸𑀕𑀥𑁄 𑀲𑁂𑀦𑀺𑀬𑁄 𑀩𑀺𑀫𑁆𑀩𑀺𑀲𑀸𑀭𑁄 𑀯𑀲𑁆𑀲𑀁 𑀉𑀓𑁆𑀓𑀟𑁆𑀠𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀬𑀤𑀺 𑀧𑀦𑀸𑀬𑁆𑀬𑀸 𑀆𑀕𑀫𑁂 𑀚𑀼𑀡𑁆𑀳𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬𑀼𑀦𑁆”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀸𑀚𑀽𑀦𑀁 𑀅𑀦𑀼𑀯𑀢𑁆𑀢𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇
