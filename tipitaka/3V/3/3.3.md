# 𑁩𑁇𑁩 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(187.)

826\. 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀬𑀣𑀸𑀪𑀺𑀭𑀦𑁆𑀢𑀁 𑀯𑀺𑀳𑀭𑀺𑀢𑁆𑀯𑀸 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀧𑀓𑁆𑀓𑀸𑀫𑀺𑁇 𑀅𑀦𑀼𑀧𑀼𑀩𑁆𑀩𑁂𑀦 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀫𑀸𑀦𑁄 𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺 𑀢𑀤𑀯𑀲𑀭𑀺𑁇 𑀢𑀢𑁆𑀭 𑀲𑀼𑀤𑀁 𑀪𑀕𑀯𑀸 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀁 𑀯𑀺𑀳𑀭𑀢𑀺 𑀚𑁂𑀢𑀯𑀦𑁂 𑀅𑀦𑀸𑀣𑀧𑀺𑀡𑁆𑀟𑀺𑀓𑀲𑁆𑀲 𑀆𑀭𑀸𑀫𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀉𑀤𑁂𑀦𑁂𑀦 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀤𑀦𑁆𑀢𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀤𑀸𑀦𑀜𑁆𑀘 𑀤𑀸𑀢𑀼𑀁, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑁄𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀪𑀕𑀯𑀢𑀸, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀦 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑁆𑀯𑀸 𑀧𑀼𑀭𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀧𑀘𑁆𑀙𑀺𑀫𑀁 𑀯𑀸 𑀢𑁂𑀫𑀸𑀲𑀁 𑀅𑀯𑀲𑀺𑀢𑁆𑀯𑀸 𑀘𑀸𑀭𑀺𑀓𑀸 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀸’𑀢𑀺𑁇 𑀆𑀕𑀫𑁂𑀢𑀼 𑀉𑀤𑁂𑀦𑁄 𑀉𑀧𑀸𑀲𑀓𑁄, 𑀬𑀸𑀯 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀺𑁇 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀆𑀕𑀫𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀲𑀘𑁂 𑀧𑀦𑀲𑁆𑀲 𑀅𑀘𑁆𑀘𑀸𑀬𑀺𑀓𑀁 𑀓𑀭𑀡𑀻𑀬𑀁, 𑀢𑀢𑁆𑀣𑁂𑀯 𑀆𑀯𑀸𑀲𑀺𑀓𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀯𑀺𑀳𑀸𑀭𑀁 𑀧𑀢𑀺𑀝𑁆𑀞𑀸𑀧𑁂𑀢𑀽”𑀢𑀺𑁇 𑀉𑀤𑁂𑀦𑁄 𑀉𑀧𑀸𑀲𑀓𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀪𑀤𑀦𑁆𑀢𑀸 𑀫𑀬𑀸 𑀧𑀳𑀺𑀢𑁂 𑀦 𑀆𑀕𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀅𑀳𑀜𑁆𑀳𑀺 𑀤𑀸𑀬𑀓𑁄 𑀓𑀸𑀭𑀓𑁄 𑀲𑀁𑀖𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑁄”𑀢𑀺𑁇

827\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑁂𑀦𑀲𑁆𑀲 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀳𑀺𑀢𑁂 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬, 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸, 𑀉𑀧𑀸𑀲𑀓𑀲𑁆𑀲, 𑀉𑀧𑀸𑀲𑀺𑀓𑀸𑀬—  𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀳𑀺𑀢𑁂 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

(188.)

828\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀤𑀦𑁆𑀢𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀤𑀸𑀦𑀜𑁆𑀘 𑀤𑀸𑀢𑀼𑀁, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑁄𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀧𑀳𑀺𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

829\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲 𑀅𑀟𑁆𑀠𑀬𑁄𑀕𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀧𑀸𑀲𑀸𑀤𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀳𑀫𑁆𑀫𑀺𑀬𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺…  𑀕𑀼𑀳𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺…  𑀓𑁄𑀝𑁆𑀞𑀓𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀅𑀕𑁆𑀕𑀺𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀓𑀧𑁆𑀧𑀺𑀬𑀓𑀼𑀝𑀺 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀯𑀘𑁆𑀘𑀓𑀼𑀝𑀺 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀘𑀗𑁆𑀓𑀫𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀘𑀗𑁆𑀓𑀫𑀦𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀉𑀤𑀧𑀸𑀦𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀉𑀤𑀧𑀸𑀦𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺…  𑀚𑀦𑁆𑀢𑀸𑀖𑀭𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀫𑀡𑁆𑀟𑀧𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀆𑀭𑀸𑀫𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀆𑀭𑀸𑀫𑀯𑀢𑁆𑀣𑀼 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀤𑀦𑁆𑀢𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀤𑀸𑀦𑀜𑁆𑀘 𑀤𑀸𑀢𑀼𑀁, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑁄𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀧𑀳𑀺𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

837\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…𑀧𑁂…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀸𑀫𑀡𑁂𑀭𑁂𑀦 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀲𑀁𑀖𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀏𑀓𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀏𑀓𑀁 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑁂 𑀲𑀸𑀫𑀡𑁂𑀭𑁂 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀏𑀓𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀲𑀫𑁆𑀩𑀳𑀼𑀮𑀸 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑁄 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀏𑀓𑀁 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀁 𑀉𑀤𑁆𑀤𑀺𑀲𑁆𑀲…  𑀅𑀢𑁆𑀢𑀦𑁄 𑀅𑀢𑁆𑀣𑀸𑀬 𑀯𑀺𑀳𑀸𑀭𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀅𑀟𑁆𑀠𑀬𑁄𑀕𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀧𑀸𑀲𑀸𑀤𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀳𑀫𑁆𑀫𑀺𑀬𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺…  𑀕𑀼𑀳𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀧𑀭𑀺𑀯𑁂𑀡𑀁 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺…  𑀓𑁄𑀝𑁆𑀞𑀓𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀉𑀧𑀝𑁆𑀞𑀸𑀦𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀅𑀕𑁆𑀕𑀺𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀓𑀧𑁆𑀧𑀺𑀬𑀓𑀼𑀝𑀺 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀘𑀗𑁆𑀓𑀫𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀘𑀗𑁆𑀓𑀫𑀦𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀉𑀤𑀧𑀸𑀦𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀉𑀤𑀧𑀸𑀦𑀲𑀸𑀮𑀸 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀧𑁄𑀓𑁆𑀔𑀭𑀡𑀻 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺…  𑀫𑀡𑁆𑀟𑀧𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀆𑀭𑀸𑀫𑁄 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑁄 𑀳𑁄𑀢𑀺…  𑀆𑀭𑀸𑀫𑀯𑀢𑁆𑀣𑀼 𑀓𑀸𑀭𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬…  ‘𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀤𑀸𑀦𑀜𑁆𑀘 𑀤𑀸𑀢𑀼𑀁, 𑀥𑀫𑁆𑀫𑀜𑁆𑀘 𑀲𑁄𑀢𑀼𑀁, 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘 𑀧𑀲𑁆𑀲𑀺𑀢𑀼𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀧𑀳𑀺𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
