# 𑁩𑁇𑁧𑁧 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀓𑀢𑀺𑀓𑀸

### 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀅𑀥𑀫𑁆𑀫𑀺𑀓𑀓𑀢𑀺𑀓𑀸

(205.)

916. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀸𑀯𑀢𑁆𑀣𑀺𑀬𑀸 𑀲𑀁𑀖𑁂𑀦 𑀏𑀯𑀭𑀽𑀧𑀸 𑀓𑀢𑀺𑀓𑀸 𑀓𑀢𑀸 𑀳𑁄𑀢𑀺—  “𑀅𑀦𑁆𑀢𑀭𑀸𑀯𑀲𑁆𑀲𑀁 𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺𑁇 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀦𑀢𑁆𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀉𑀧𑀲𑀗𑁆𑀓𑀫𑀺𑀢𑁆𑀯𑀸 𑀧𑀩𑁆𑀩𑀚𑁆𑀚𑀁 𑀬𑀸𑀘𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀏𑀯𑀫𑀸𑀳𑀁𑀲𑀼—  “𑀲𑀁𑀖𑁂𑀦 𑀔𑁄, 𑀆𑀯𑀼𑀲𑁄, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀓𑀢𑀺𑀓𑀸 𑀓𑀢𑀸 ‘𑀅𑀦𑁆𑀢𑀭𑀸𑀯𑀲𑁆𑀲𑀁 𑀦 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀆𑀕𑀫𑁂𑀳𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀬𑀸𑀯 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀦𑁆𑀢𑀺𑁇 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀲𑁆𑀲𑀦𑁆𑀢𑀻”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀲𑁆𑀲𑀁𑀯𑀼𑀝𑁆𑀞𑀸 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀦𑀢𑁆𑀢𑀸𑀭𑀁 𑀏𑀢𑀤𑀯𑁄𑀘𑀼𑀁—  “𑀏𑀳𑀺 𑀤𑀸𑀦𑀺, 𑀆𑀯𑀼𑀲𑁄, 𑀧𑀩𑁆𑀩𑀚𑀸𑀳𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀏𑀯𑀫𑀸𑀳—  “𑀲𑀘𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀩𑁆𑀩𑀚𑀺𑀢𑁄 𑀅𑀲𑁆𑀲𑀁, 𑀅𑀪𑀺𑀭𑀫𑁂𑀬𑁆𑀬𑀸𑀫𑀳𑀁𑁇 𑀦 𑀤𑀸𑀦𑀸𑀳𑀁, 𑀪𑀦𑁆𑀢𑁂, 𑀧𑀩𑁆𑀩𑀚𑀺𑀲𑁆𑀲𑀸𑀫𑀻”𑀢𑀺𑁇 𑀯𑀺𑀲𑀸𑀔𑀸 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑀸 𑀏𑀯𑀭𑀽𑀧𑀁 𑀓𑀢𑀺𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺—  ‘𑀦 𑀅𑀦𑁆𑀢𑀭𑀸𑀯𑀲𑁆𑀲𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀓𑀁 𑀓𑀸𑀮𑀁 𑀥𑀫𑁆𑀫𑁄 𑀦 𑀘𑀭𑀺𑀢𑀩𑁆𑀩𑁄”𑀢𑀺?

917. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀺𑀲𑀸𑀔𑀸𑀬 𑀫𑀺𑀕𑀸𑀭𑀫𑀸𑀢𑀼𑀬𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺𑀬𑀸 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺𑀬𑀸𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀏𑀯𑀭𑀽𑀧𑀸 𑀓𑀢𑀺𑀓𑀸 𑀓𑀸𑀢𑀩𑁆𑀩𑀸—  ‘𑀦 𑀅𑀦𑁆𑀢𑀭𑀸𑀯𑀲𑁆𑀲𑀁 𑀧𑀩𑁆𑀩𑀸𑀚𑁂𑀢𑀩𑁆𑀩𑀦𑁆’𑀢𑀺𑁇 𑀬𑁄 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
