# 𑁩𑁇𑁫 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(198.)

867\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀫𑀸𑀢𑀸 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑀸, 𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑁄, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀕𑀢𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑀲𑁆𑀲 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀪𑀕𑀯𑀢𑀸 𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑀁—  ‘𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀧𑀳𑀺𑀢𑁂 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂; 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂’𑀢𑀺𑁇 𑀅𑀬𑀜𑁆𑀘 𑀫𑁂 𑀫𑀸𑀢𑀸 𑀕𑀺𑀮𑀸𑀦𑀸, 𑀲𑀸 𑀘 𑀅𑀦𑀼𑀧𑀸𑀲𑀺𑀓𑀸, 𑀓𑀣𑀁 𑀦𑀼 𑀔𑁄 𑀫𑀬𑀸 𑀧𑀝𑀺𑀧𑀚𑁆𑀚𑀺𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬, 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸, 𑀫𑀸𑀢𑀼𑀬𑀸 𑀘 𑀧𑀺𑀢𑀼𑀲𑁆𑀲 𑀘—  𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀲𑀢𑁆𑀢𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

868\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀫𑀸𑀢𑀸 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑀸, 𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑁄, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

869\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀧𑀺𑀢𑀸 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀫𑁂 𑀧𑀼𑀢𑁆𑀢𑁄, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀧𑀼𑀢𑁆𑀢𑀲𑁆𑀲 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇
