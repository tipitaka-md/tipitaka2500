# 𑁩𑁇𑁧𑁨 𑀧𑀝𑀺𑀲𑁆𑀲𑀯𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺

### 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀝𑀺𑀲𑁆𑀲𑀯𑀤𑀼𑀓𑁆𑀓𑀝𑀸𑀧𑀢𑁆𑀢𑀺

(206.)

918. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀆𑀬𑀲𑁆𑀫𑀢𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁂𑀦 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁂𑀦 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸𑀬𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀅𑀤𑁆𑀤𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀤𑁆𑀯𑁂 𑀆𑀯𑀸𑀲𑁂 𑀩𑀳𑀼𑀘𑀻𑀯𑀭𑀓𑁂𑁇 𑀢𑀲𑁆𑀲 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼 𑀆𑀯𑀸𑀲𑁂𑀲𑀼 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑁂𑀬𑁆𑀬𑀁𑁇 𑀏𑀯𑀁 𑀫𑁂 𑀩𑀳𑀼𑀁 𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻”𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼 𑀆𑀯𑀸𑀲𑁂𑀲𑀼 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀺𑁇 𑀭𑀸𑀚𑀸 𑀧𑀲𑁂𑀦𑀤𑀺 𑀓𑁄𑀲𑀮𑁄 𑀉𑀚𑁆𑀛𑀸𑀬𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀅𑀬𑁆𑀬𑁄 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀅𑀫𑁆𑀳𑀸𑀓𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀁𑀯𑀸𑀤𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑀦𑀼 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀕𑀭𑀳𑀺𑀢𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀧𑀲𑀢𑁆𑀣𑀸”𑀢𑀺𑁇

919. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀲𑁆𑀲 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀲𑁆𑀲𑁇 𑀬𑁂 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀧𑁆𑀧𑀺𑀘𑁆𑀙𑀸…𑀧𑁂…  𑀢𑁂 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀆𑀬𑀲𑁆𑀫𑀸 𑀉𑀧𑀦𑀦𑁆𑀤𑁄 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑁄 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀁𑀯𑀸𑀤𑁂𑀲𑁆𑀲𑀢𑀺𑁇 𑀦𑀦𑀼 𑀪𑀕𑀯𑀢𑀸 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀕𑀭𑀳𑀺𑀢𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀧𑀲𑀢𑁆𑀣𑀸”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁…𑀧𑁂…  𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑀁𑀖𑀁 𑀲𑀦𑁆𑀦𑀺𑀧𑀸𑀢𑀸𑀧𑁂𑀢𑁆𑀯𑀸 𑀆𑀬𑀲𑁆𑀫𑀦𑁆𑀢𑀁 𑀉𑀧𑀦𑀦𑁆𑀤𑀁 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀁 𑀧𑀝𑀺𑀧𑀼𑀘𑁆𑀙𑀺—  “𑀲𑀘𑁆𑀘𑀁 𑀓𑀺𑀭 𑀢𑁆𑀯𑀁, 𑀉𑀧𑀦𑀦𑁆𑀤, 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀁𑀯𑀸𑀤𑁂𑀲𑀻”𑀢𑀺? “𑀲𑀘𑁆𑀘𑀁, 𑀪𑀕𑀯𑀸”𑀢𑀺𑁇 𑀯𑀺𑀕𑀭𑀳𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸…𑀧𑁂…  𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀢𑁆𑀯𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀭𑀜𑁆𑀜𑁄 𑀧𑀲𑁂𑀦𑀤𑀺𑀲𑁆𑀲 𑀓𑁄𑀲𑀮𑀲𑁆𑀲 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀡𑀺𑀢𑁆𑀯𑀸 𑀯𑀺𑀲𑀁𑀯𑀸𑀤𑁂𑀲𑁆𑀲𑀲𑀺𑁇 𑀦𑀦𑀼 𑀫𑀬𑀸, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀦𑁂𑀓𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑁄 𑀕𑀭𑀳𑀺𑀢𑁄, 𑀫𑀼𑀲𑀸𑀯𑀸𑀤𑀸 𑀯𑁂𑀭𑀫𑀡𑀻 𑀧𑀲𑀢𑁆𑀣𑀸𑁇 𑀦𑁂𑀢𑀁, 𑀫𑁄𑀖𑀧𑀼𑀭𑀺𑀲, 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑀦𑀁 𑀯𑀸 𑀧𑀲𑀸𑀤𑀸𑀬…𑀧𑁂…  𑀯𑀺𑀕𑀭𑀳𑀺𑀢𑁆𑀯𑀸…𑀧𑁂…  𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—

(207.)

920. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀸 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑁄 𑀧𑀝𑀺𑀲𑁆𑀲𑀼𑀢𑁄 𑀳𑁄𑀢𑀺 𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸𑀬𑁇 𑀲𑁄 𑀢𑀁 𑀆𑀯𑀸𑀲𑀁 𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑁄 𑀧𑀲𑁆𑀲𑀢𑀺 𑀅𑀦𑁆𑀢𑀭𑀸𑀫𑀕𑁆𑀕𑁂 𑀤𑁆𑀯𑁂 𑀆𑀯𑀸𑀲𑁂 𑀩𑀳𑀼𑀘𑀻𑀯𑀭𑀓𑁂𑁇 𑀢𑀲𑁆𑀲 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀬𑀦𑁆𑀦𑀽𑀦𑀸𑀳𑀁 𑀇𑀫𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼 𑀆𑀯𑀸𑀲𑁂𑀲𑀼 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑁂𑀬𑁆𑀬𑀁𑁇 𑀏𑀯𑀁 𑀫𑁂 𑀩𑀳𑀼𑀁 𑀘𑀻𑀯𑀭𑀁 𑀉𑀧𑁆𑀧𑀚𑁆𑀚𑀺𑀲𑁆𑀲𑀢𑀻’𑀢𑀺𑁇 𑀲𑁄 𑀢𑁂𑀲𑀼 𑀤𑁆𑀯𑀻𑀲𑀼 𑀆𑀯𑀸𑀲𑁂𑀲𑀼 𑀯𑀲𑁆𑀲𑀁 𑀯𑀲𑀢𑀺𑁇 𑀢𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸 𑀘 𑀦 𑀧𑀜𑁆𑀜𑀸𑀬𑀢𑀺, 𑀧𑀝𑀺𑀲𑁆𑀲𑀯𑁂 𑀘 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑁇

_𑀆𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬 𑀦 𑀏𑀬𑁆𑀬 𑀯𑀸;_

_𑀢𑀦𑁆𑀢𑀺𑀫𑀕𑁆𑀕𑀁 𑀦𑀺𑀲𑀸𑀫𑀬𑁂𑀢𑀺𑁇_
