# 𑁩𑁇𑁭 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀯𑀸𑀭

* 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁂𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀯𑀸𑀭

(200.)

876\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀸𑀍𑀏𑀳𑀺 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀕𑀡𑁆𑀳𑀺𑀁𑀲𑀼𑀧𑀺 𑀧𑀭𑀺𑀧𑀸𑀢𑀺𑀁𑀲𑀼𑀧𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

877\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀯𑀸𑀍𑀏𑀳𑀺 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀕𑀡𑁆𑀳𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

878\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑀭𑀻𑀲𑀧𑁂𑀳𑀺 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀟𑀁𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀧𑀭𑀺𑀧𑀸𑀢𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

879\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑁄𑀭𑁂𑀳𑀺 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀯𑀺𑀮𑀼𑀫𑁆𑀧𑀦𑁆𑀢𑀺𑀧𑀺 𑀆𑀓𑁄𑀝𑁂𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

880\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀺𑀲𑀸𑀘𑁂𑀳𑀺 𑀉𑀩𑁆𑀩𑀸𑀍𑀳𑀸 𑀳𑁄𑀦𑁆𑀢𑀺𑁇 𑀆𑀯𑀺𑀲𑀦𑁆𑀢𑀺𑀧𑀺 𑀳𑀦𑀦𑁆𑀢𑀺𑀧𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

881\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀸𑀫𑁄 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀤𑀟𑁆𑀠𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

882\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀅𑀕𑁆𑀕𑀺𑀦𑀸 𑀤𑀟𑁆𑀠𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀦 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

883\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀸𑀫𑁄 𑀉𑀤𑀓𑁂𑀦 𑀯𑀽𑀍𑀳𑁄 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀧𑀺𑀡𑁆𑀟𑀓𑁂𑀦 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

884\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑁂𑀦𑀸𑀲𑀦𑀁 𑀉𑀤𑀓𑁂𑀦 𑀯𑀽𑀍𑀳𑀁 𑀳𑁄𑀢𑀺𑁇 𑀪𑀺𑀓𑁆𑀔𑀽 𑀲𑁂𑀦𑀸𑀲𑀦𑁂𑀦 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

(201.)

885\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸𑀦𑀁 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀕𑀸𑀫𑁄 𑀘𑁄𑀭𑁂𑀳𑀺 𑀯𑀼𑀝𑁆𑀞𑀸𑀲𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀕𑀸𑀫𑁄 𑀢𑁂𑀦 𑀕𑀦𑁆𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

886\. 𑀕𑀸𑀫𑁄 𑀤𑁆𑀯𑁂𑀥𑀸 𑀪𑀺𑀚𑁆𑀚𑀺𑀢𑁆𑀣𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀩𑀳𑀼𑀢𑀭𑀸 𑀢𑁂𑀦 𑀕𑀦𑁆𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

887\. 𑀩𑀳𑀼𑀢𑀭𑀸 𑀅𑀲𑁆𑀲𑀤𑁆𑀥𑀸 𑀳𑁄𑀦𑁆𑀢𑀺 𑀅𑀧𑁆𑀧𑀲𑀦𑁆𑀦𑀸𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀬𑁂𑀦 𑀲𑀤𑁆𑀥𑀸 𑀧𑀲𑀦𑁆𑀦𑀸 𑀢𑁂𑀦 𑀕𑀦𑁆𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

888\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀓𑁄𑀲𑀮𑁂𑀲𑀼 𑀚𑀦𑀧𑀤𑁂 𑀅𑀜𑁆𑀜𑀢𑀭𑀲𑁆𑀫𑀺𑀁 𑀆𑀯𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀮𑀪𑀺𑀁𑀲𑀼 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇

889\. “𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

890\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀮𑀪𑀦𑁆𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺 𑀲𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀺 𑀪𑁄𑀚𑀦𑀸𑀦𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

891\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀮𑀪𑀦𑁆𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀲𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀺 𑀪𑁄𑀚𑀦𑀸𑀦𑀺, 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺 𑀲𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

892\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀮𑀪𑀦𑁆𑀢𑀺 𑀮𑀽𑀔𑀲𑁆𑀲 𑀯𑀸 𑀧𑀡𑀻𑀢𑀲𑁆𑀲 𑀯𑀸 𑀪𑁄𑀚𑀦𑀲𑁆𑀲 𑀬𑀸𑀯𑀤𑀢𑁆𑀣𑀁 𑀧𑀸𑀭𑀺𑀧𑀽𑀭𑀺𑀁, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀲𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀺 𑀪𑁄𑀚𑀦𑀸𑀦𑀺, 𑀮𑀪𑀦𑁆𑀢𑀺 𑀲𑀧𑁆𑀧𑀸𑀬𑀸𑀦𑀺 𑀪𑁂𑀲𑀚𑁆𑀚𑀸𑀦𑀺, 𑀦 𑀮𑀪𑀦𑁆𑀢𑀺 𑀧𑀢𑀺𑀭𑀽𑀧𑀁 𑀉𑀧𑀝𑁆𑀞𑀸𑀓𑀁𑁇 𑀏𑀲𑁂𑀯 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄𑀢𑀺 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

893\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀇𑀢𑁆𑀣𑀻 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺—  ‘𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀔𑁂𑀢𑁆𑀢𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀕𑀸𑀯𑀼𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀕𑀸𑀯𑀺𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀤𑀸𑀲𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀤𑀸𑀲𑀺𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺, 𑀥𑀻𑀢𑀭𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫𑀺 𑀪𑀭𑀺𑀬𑀢𑁆𑀣𑀸𑀬, 𑀅𑀳𑀁 𑀯𑀸 𑀢𑁂 𑀪𑀭𑀺𑀬𑀸 𑀳𑁄𑀫𑀺, 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀢𑁂 𑀪𑀭𑀺𑀬𑀁 𑀆𑀦𑁂𑀫𑀻’𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀮𑀳𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀁 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸, 𑀲𑀺𑀬𑀸𑀧𑀺 𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄’𑀢𑀺, 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

894\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑀁 𑀪𑀺𑀓𑁆𑀔𑀼𑀁 𑀯𑁂𑀲𑀻 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺…𑀧𑁂…  𑀣𑀼𑀮𑁆𑀮𑀓𑀼𑀫𑀸𑀭𑀻 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺…  𑀧𑀡𑁆𑀟𑀓𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀢𑀺…  𑀜𑀸𑀢𑀓𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺…  𑀭𑀸𑀚𑀸𑀦𑁄 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺…  𑀘𑁄𑀭𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺…  𑀥𑀼𑀢𑁆𑀢𑀸 𑀦𑀺𑀫𑀦𑁆𑀢𑁂𑀦𑁆𑀢𑀺—  ‘𑀏𑀳𑀺, 𑀪𑀦𑁆𑀢𑁂, 𑀳𑀺𑀭𑀜𑁆𑀜𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀲𑀼𑀯𑀡𑁆𑀡𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀔𑁂𑀢𑁆𑀢𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀯𑀢𑁆𑀣𑀼𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀕𑀸𑀯𑀼𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀕𑀸𑀯𑀺𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀤𑀸𑀲𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀤𑀸𑀲𑀺𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫, 𑀥𑀻𑀢𑀭𑀁 𑀯𑀸 𑀢𑁂 𑀤𑁂𑀫 𑀪𑀭𑀺𑀬𑀢𑁆𑀣𑀸𑀬, 𑀅𑀜𑁆𑀜𑀁 𑀯𑀸 𑀢𑁂 𑀪𑀭𑀺𑀬𑀁 𑀆𑀦𑁂𑀫𑀸’𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀮𑀳𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀁 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸, 𑀲𑀺𑀬𑀸𑀧𑀺 𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄’𑀢𑀺, 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇

895\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀕𑀢𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀲𑁆𑀲𑀸𑀫𑀺𑀓𑀁 𑀦𑀺𑀥𑀺𑀁 𑀧𑀲𑁆𑀲𑀢𑀺𑁇 𑀢𑀢𑁆𑀭 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑁄 𑀏𑀯𑀁 𑀳𑁄𑀢𑀺—  ‘𑀮𑀳𑀼𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀁 𑀔𑁄 𑀘𑀺𑀢𑁆𑀢𑀁 𑀯𑀼𑀢𑁆𑀢𑀁 𑀪𑀕𑀯𑀢𑀸, 𑀲𑀺𑀬𑀸𑀧𑀺 𑀫𑁂 𑀩𑁆𑀭𑀳𑁆𑀫𑀘𑀭𑀺𑀬𑀲𑁆𑀲 𑀅𑀦𑁆𑀢𑀭𑀸𑀬𑁄’𑀢𑀺, 𑀧𑀓𑁆𑀓𑀫𑀺𑀢𑀩𑁆𑀩𑀁𑁇 𑀅𑀦𑀸𑀧𑀢𑁆𑀢𑀺 𑀯𑀲𑁆𑀲𑀘𑁆𑀙𑁂𑀤𑀲𑁆𑀲𑁇
