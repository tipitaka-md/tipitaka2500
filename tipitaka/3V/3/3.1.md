# 𑁩𑁇𑁧 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸𑀦𑀼𑀚𑀸𑀦𑀦

* 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸𑀦𑀼𑀚𑀸𑀦𑀦

(184.)

818\. 𑀢𑁂𑀦 𑀲𑀫𑀬𑁂𑀦 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸 𑀭𑀸𑀚𑀕𑀳𑁂 𑀯𑀺𑀳𑀭𑀢𑀺 𑀯𑁂𑀍𑀉𑀯𑀦𑁂 𑀓𑀮𑀦𑁆𑀤𑀓𑀦𑀺𑀯𑀸𑀧𑁂𑁇 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀕𑀯𑀢𑀸 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑁄 𑀅𑀧𑀜𑁆𑀜𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇 𑀢𑁂𑀥 𑀪𑀺𑀓𑁆𑀔𑀽 𑀳𑁂𑀫𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀕𑀺𑀫𑁆𑀳𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀓𑀣𑀜𑁆𑀳𑀺 𑀦𑀸𑀫 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀕𑀺𑀫𑁆𑀳𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺, 𑀳𑀭𑀺𑀢𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀦𑁆𑀢𑀸, 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀸, 𑀩𑀳𑀽 𑀔𑀼𑀤𑁆𑀤𑀓𑁂 𑀧𑀸𑀡𑁂 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀸𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀅𑀜𑁆𑀜𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸 𑀤𑀼𑀭𑀓𑁆𑀔𑀸𑀢𑀥𑀫𑁆𑀫𑀸 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀅𑀮𑁆𑀮𑀻𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀗𑁆𑀓𑀲𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀳𑀺 𑀦𑀸𑀫 𑀲𑀓𑀼𑀦𑁆𑀢𑀓𑀸 𑀭𑀼𑀓𑁆𑀔𑀕𑁆𑀕𑁂𑀲𑀼 𑀓𑀼𑀮𑀸𑀯𑀓𑀸𑀦𑀺 𑀓𑀭𑀺𑀢𑁆𑀯𑀸 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀁 𑀅𑀮𑁆𑀮𑀻𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺 𑀲𑀗𑁆𑀓𑀲𑀸𑀬𑀺𑀲𑁆𑀲𑀦𑁆𑀢𑀺𑁇 𑀇𑀫𑁂 𑀧𑀦 𑀲𑀫𑀡𑀸 𑀲𑀓𑁆𑀬𑀧𑀼𑀢𑁆𑀢𑀺𑀬𑀸 𑀳𑁂𑀫𑀦𑁆𑀢𑀫𑁆𑀧𑀺 𑀕𑀺𑀫𑁆𑀳𑀫𑁆𑀧𑀺 𑀯𑀲𑁆𑀲𑀫𑁆𑀧𑀺 𑀘𑀸𑀭𑀺𑀓𑀁 𑀘𑀭𑀦𑁆𑀢𑀺, 𑀳𑀭𑀺𑀢𑀸𑀦𑀺 𑀢𑀺𑀡𑀸𑀦𑀺 𑀲𑀫𑁆𑀫𑀤𑁆𑀤𑀦𑁆𑀢𑀸, 𑀏𑀓𑀺𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀁 𑀚𑀻𑀯𑀁 𑀯𑀺𑀳𑁂𑀞𑁂𑀦𑁆𑀢𑀸, 𑀩𑀳𑀽 𑀔𑀼𑀤𑁆𑀤𑀓𑁂 𑀧𑀸𑀡𑁂 𑀲𑀗𑁆𑀖𑀸𑀢𑀁 𑀆𑀧𑀸𑀤𑁂𑀦𑁆𑀢𑀸”𑀢𑀺𑁇

819\. 𑀅𑀲𑁆𑀲𑁄𑀲𑀼𑀁 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽 𑀢𑁂𑀲𑀁 𑀫𑀦𑀼𑀲𑁆𑀲𑀸𑀦𑀁 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀸𑀦𑀁 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀸𑀦𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀢𑁂 𑀪𑀺𑀓𑁆𑀔𑀽 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀕𑀯𑀸 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀦𑀺𑀤𑀸𑀦𑁂 𑀏𑀢𑀲𑁆𑀫𑀺𑀁 𑀧𑀓𑀭𑀡𑁂 𑀥𑀫𑁆𑀫𑀺𑀁 𑀓𑀣𑀁 𑀓𑀢𑁆𑀯𑀸 𑀪𑀺𑀓𑁆𑀔𑀽 𑀆𑀫𑀦𑁆𑀢𑁂𑀲𑀺—  “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀦𑁆”𑀢𑀺𑁇 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀤𑀸 𑀦𑀼 𑀔𑁄 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀦𑁆”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀸𑀦𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀼𑀦𑁆”𑀢𑀺𑁇

820\. 𑀅𑀣 𑀔𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀏𑀢𑀤𑀳𑁄𑀲𑀺—  “𑀓𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸”𑀢𑀺? 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀤𑁆𑀯𑁂𑀫𑀸, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸—  𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸, 𑀧𑀘𑁆𑀙𑀺𑀫𑀺𑀓𑀸𑁇 𑀅𑀧𑀭𑀚𑁆𑀚𑀼𑀕𑀢𑀸𑀬 𑀆𑀲𑀸𑀍𑀳𑀺𑀬𑀸 𑀧𑀼𑀭𑀺𑀫𑀺𑀓𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀸, 𑀫𑀸𑀲𑀕𑀢𑀸𑀬 𑀆𑀲𑀸𑀍𑀳𑀺𑀬𑀸 𑀧𑀘𑁆𑀙𑀺𑀫𑀺𑀓𑀸 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀸—  𑀇𑀫𑀸 𑀔𑁄, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀤𑁆𑀯𑁂 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀸”𑀢𑀺𑁇
