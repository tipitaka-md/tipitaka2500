# 𑁩𑁇𑁧𑁦 𑀯𑀲𑁆𑀲𑀁𑀅𑀦𑀼𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦

### 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀯𑀲𑁆𑀲𑀁𑀅𑀦𑀼𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀝𑁆𑀞𑀸𑀦

(204.)

909\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀼𑀓𑁆𑀔𑀲𑀼𑀲𑀺𑀭𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀧𑀺𑀲𑀸𑀘𑀺𑀮𑁆𑀮𑀺𑀓𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀼𑀓𑁆𑀔𑀲𑀼𑀲𑀺𑀭𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

910\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀭𑀼𑀓𑁆𑀔𑀯𑀺𑀝𑀪𑀺𑀬𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀫𑀺𑀕𑀮𑀼𑀤𑁆𑀤𑀓𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀭𑀼𑀓𑁆𑀔𑀯𑀺𑀝𑀪𑀺𑀬𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

911\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀤𑁂𑀯𑁂 𑀯𑀲𑁆𑀲𑀦𑁆𑀢𑁂 𑀭𑀼𑀓𑁆𑀔𑀫𑀽𑀮𑀫𑁆𑀧𑀺 𑀦𑀺𑀩𑁆𑀩𑀓𑁄𑀲𑀫𑁆𑀧𑀺 𑀉𑀧𑀥𑀸𑀯𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀚𑁆𑀛𑁄𑀓𑀸𑀲𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

912\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀅𑀲𑁂𑀦𑀸𑀲𑀦𑀺𑀓𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀲𑀻𑀢𑁂𑀦𑀧𑀺 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺, 𑀉𑀡𑁆𑀳𑁂𑀦𑀧𑀺 𑀓𑀺𑀮𑀫𑀦𑁆𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀅𑀲𑁂𑀦𑀸𑀲𑀦𑀺𑀓𑁂𑀦 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

913\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀯𑀓𑀼𑀝𑀺𑀓𑀸𑀬 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀙𑀯𑀟𑀸𑀳𑀓𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀯𑀓𑀼𑀝𑀺𑀓𑀸𑀬 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

914\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀙𑀢𑁆𑀢𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀕𑁄𑀧𑀸𑀮𑀓𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀙𑀢𑁆𑀢𑁂 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇

915\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀪𑀺𑀓𑁆𑀔𑀽 𑀘𑀸𑀝𑀺𑀬𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀺𑁇 𑀫𑀦𑀼𑀲𑁆𑀲𑀸 𑀉𑀚𑁆𑀛𑀸𑀬𑀦𑁆𑀢𑀺 𑀔𑀺𑀬𑁆𑀬𑀦𑁆𑀢𑀺 𑀯𑀺𑀧𑀸𑀘𑁂𑀦𑁆𑀢𑀺—  “𑀲𑁂𑀬𑁆𑀬𑀣𑀸𑀧𑀺 𑀢𑀺𑀢𑁆𑀣𑀺𑀬𑀸”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀘𑀸𑀝𑀺𑀬𑀸 𑀯𑀲𑁆𑀲𑀁 𑀉𑀧𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁𑁇 𑀬𑁄 𑀉𑀧𑀕𑀘𑁆𑀙𑁂𑀬𑁆𑀬, 𑀆𑀧𑀢𑁆𑀢𑀺 𑀤𑀼𑀓𑁆𑀓𑀝𑀲𑁆𑀲𑀸”𑀢𑀺𑁇
