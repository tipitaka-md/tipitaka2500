# 𑁩𑁇𑁪 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

* 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

* 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(193.)

838\. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀅𑀜𑁆𑀜𑀢𑀭𑁄 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂𑁇 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬, 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸—  𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀇𑀫𑁂𑀲𑀁 𑀧𑀜𑁆𑀘𑀦𑁆𑀦𑀁 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺 𑀕𑀦𑁆𑀢𑀼𑀁, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

839\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

840\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀫𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀅𑀦𑀪𑀺𑀭𑀢𑀁 𑀯𑀽𑀧𑀓𑀸𑀲𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀯𑀽𑀧𑀓𑀸𑀲𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀯𑀸𑀲𑁆𑀲 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

841\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀫𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀯𑀺𑀦𑁄𑀤𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀯𑀺𑀦𑁄𑀤𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀯𑀸𑀲𑁆𑀲 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

842\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀫𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀯𑀺𑀯𑁂𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀯𑀺𑀯𑁂𑀘𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀯𑀸𑀲𑁆𑀲 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

843\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀳𑁄𑀢𑀺 𑀧𑀭𑀺𑀯𑀸𑀲𑀸𑀭𑀳𑁄𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑁄 𑀧𑀭𑀺𑀯𑀸𑀲𑀸𑀭𑀳𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀧𑀭𑀺𑀯𑀸𑀲𑀤𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀕𑀡𑀧𑀽𑀭𑀓𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

844\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀫𑀽𑀮𑀸𑀬𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀕𑀡𑀧𑀽𑀭𑀓𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

845\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀫𑀸𑀦𑀢𑁆𑀢𑀤𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀕𑀡𑀧𑀽𑀭𑀓𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

846\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼 𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀕𑀡𑀧𑀽𑀭𑀓𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

849\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

852\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀫𑁂 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀯𑀺𑀯𑁂𑀘𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀯𑀺𑀯𑁂𑀘𑀸𑀧𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀥𑀫𑁆𑀫𑀓𑀣𑀁 𑀯𑀸𑀲𑁆𑀲𑀸 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

853\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺 𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑀸𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀭𑀼𑀥𑀫𑁆𑀫𑀁 𑀅𑀚𑁆𑀛𑀸𑀧𑀦𑁆𑀦𑀸 𑀫𑀸𑀦𑀢𑁆𑀢𑀸𑀭𑀳𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀫𑀸𑀦𑀢𑁆𑀢𑀤𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

854\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀸𑀭𑀳𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀫𑀽𑀮𑀸𑀬 𑀧𑀝𑀺𑀓𑀲𑁆𑀲𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

855\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀻 𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀅𑀩𑁆𑀪𑀸𑀦𑀸𑀭𑀳𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀅𑀩𑁆𑀪𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

856\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀦𑀺𑀬𑀸 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺—  𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀁 𑀯𑀸, 𑀦𑀺𑀬𑀲𑁆𑀲𑀁 𑀯𑀸, 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀁 𑀯𑀸, 𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀸, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀁 𑀯𑀸𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀲𑀁𑀖𑁄 𑀫𑁂 𑀓𑀫𑁆𑀫𑀁 𑀓𑀢𑁆𑀢𑀼𑀓𑀸𑀫𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀲𑀁𑀖𑁄 𑀓𑀫𑁆𑀫𑀁 𑀦 𑀓𑀭𑁂𑀬𑁆𑀬, 𑀮𑀳𑀼𑀓𑀸𑀬 𑀯𑀸 𑀧𑀭𑀺𑀡𑀸𑀫𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

857\. 𑀓𑀢𑀁 𑀯𑀸 𑀧𑀦𑀲𑁆𑀲𑀸 𑀳𑁄𑀢𑀺 𑀲𑀁𑀖𑁂𑀦 𑀓𑀫𑁆𑀫𑀁—  𑀢𑀚𑁆𑀚𑀦𑀻𑀬𑀁 𑀯𑀸, 𑀦𑀺𑀬𑀲𑁆𑀲𑀁 𑀯𑀸, 𑀧𑀩𑁆𑀩𑀸𑀚𑀦𑀻𑀬𑀁 𑀯𑀸, 𑀧𑀝𑀺𑀲𑀸𑀭𑀡𑀻𑀬𑀁 𑀯𑀸, 𑀉𑀓𑁆𑀔𑁂𑀧𑀦𑀻𑀬𑀁 𑀯𑀸𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀲𑀁𑀖𑁄 𑀫𑁂 𑀓𑀫𑁆𑀫𑀁 𑀅𑀓𑀸𑀲𑀺, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀓𑀺𑀦𑁆𑀢𑀺 𑀦𑀼 𑀔𑁄 𑀲𑀫𑁆𑀫𑀸 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀮𑁄𑀫𑀁 𑀧𑀸𑀢𑁂𑀬𑁆𑀬, 𑀦𑁂𑀢𑁆𑀣𑀸𑀭𑀁 𑀯𑀢𑁆𑀢𑁂𑀬𑁆𑀬, 𑀲𑀁𑀖𑁄 𑀢𑀁 𑀓𑀫𑁆𑀫𑀁 𑀧𑀝𑀺𑀧𑁆𑀧𑀲𑁆𑀲𑀫𑁆𑀪𑁂𑀬𑁆𑀬𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

(195.)

858\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

859\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸𑀬 𑀲𑀺𑀓𑁆𑀔𑀸 𑀓𑀼𑀧𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀲𑀺𑀓𑁆𑀔𑀸 𑀫𑁂 𑀓𑀼𑀧𑀺𑀢𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀫𑀸𑀤𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

860\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀺𑀓𑁆𑀔𑀫𑀸𑀦𑀸 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀓𑀸𑀫𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀕𑀡𑀧𑀽𑀭𑀓𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

861\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

862\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀲𑀸𑀫𑀡𑁂𑀭𑀲𑁆𑀲 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀯𑀲𑁆𑀲𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀯𑀲𑁆𑀲𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀓𑀸𑀫𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

863\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑁄 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀓𑀸𑀫𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀉𑀧𑀲𑀫𑁆𑀧𑀚𑁆𑀚𑀺𑀢𑀼𑀓𑀸𑀫𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀺𑀓𑁆𑀔𑀽, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀉𑀧𑀲𑀫𑁆𑀧𑀤𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀅𑀦𑀼𑀲𑁆𑀲𑀸𑀯𑁂𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀕𑀡𑀧𑀽𑀭𑀓𑁄 𑀯𑀸 𑀪𑀯𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

(197.)

864\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀕𑀺𑀮𑀸𑀦𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀼𑀧𑀝𑁆𑀞𑀸𑀓𑀪𑀢𑁆𑀢𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀕𑀺𑀮𑀸𑀦𑀪𑁂𑀲𑀚𑁆𑀚𑀁 𑀯𑀸 𑀧𑀭𑀺𑀬𑁂𑀲𑀺𑀲𑁆𑀲𑀸𑀫𑀺, 𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀉𑀧𑀝𑁆𑀞𑀳𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

865\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀅𑀦𑀪𑀺𑀭𑀢𑀺 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀸 𑀳𑁄𑀢𑀺…𑀧𑁂…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀓𑀼𑀓𑁆𑀓𑀼𑀘𑁆𑀘𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀲𑀸𑀫𑀡𑁂𑀭𑀺𑀬𑀸 𑀤𑀺𑀝𑁆𑀞𑀺𑀕𑀢𑀁 𑀉𑀧𑁆𑀧𑀦𑁆𑀦𑀁 𑀳𑁄𑀢𑀺…  𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀯𑀲𑁆𑀲𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀯𑀲𑁆𑀲𑀁 𑀧𑀼𑀘𑁆𑀙𑀺𑀢𑀼𑀓𑀸𑀫𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀧𑀼𑀘𑁆𑀙𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸, 𑀆𑀘𑀺𑀓𑁆𑀔𑀺𑀲𑁆𑀲𑀸𑀫𑀺 𑀯𑀸’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

866\. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀸𑀫𑀡𑁂𑀭𑀻 𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑀼𑀓𑀸𑀫𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀲𑀺𑀓𑁆𑀔𑀁 𑀲𑀫𑀸𑀤𑀺𑀬𑀺𑀢𑀼𑀓𑀸𑀫𑀸, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀅𑀬𑁆𑀬𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀅𑀬𑁆𑀬𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑀧𑀺, 𑀧𑀕𑁂𑀯 𑀧𑀳𑀺𑀢𑁂—  ‘𑀲𑀺𑀓𑁆𑀔𑀸𑀲𑀫𑀸𑀤𑀸𑀦𑀁 𑀉𑀲𑁆𑀲𑀼𑀓𑁆𑀓𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻’𑀢𑀺𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇
