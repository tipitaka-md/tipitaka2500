# 𑁩𑁇𑁬 𑀧𑀳𑀺𑀢𑁂𑀬𑁂𑀯𑀅𑀦𑀼𑀚𑀸𑀦𑀦

### 𑀯𑀲𑁆𑀲𑀽𑀧𑀦𑀸𑀬𑀺𑀓𑀓𑁆𑀔𑀦𑁆𑀥𑀓

### 𑀧𑀳𑀺𑀢𑁂𑀬𑁂𑀯𑀅𑀦𑀼𑀚𑀸𑀦𑀦

(199.)

870. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀪𑀸𑀢𑀸 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀸𑀢𑀼𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀫𑁂 𑀪𑀸𑀢𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀸𑀢𑀼𑀦𑁄 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀧𑀳𑀺𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

871. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀪𑀕𑀺𑀦𑀻 𑀕𑀺𑀮𑀸𑀦𑀸 𑀳𑁄𑀢𑀺𑁇 𑀲𑀸 𑀘𑁂 𑀪𑀸𑀢𑀼𑀦𑁄 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑀸, 𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀫𑁂 𑀪𑀸𑀢𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀸𑀢𑀼𑀦𑁄 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀧𑀳𑀺𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

872. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀜𑀸𑀢𑀓𑁄 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀼𑀲𑁆𑀲 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀆𑀕𑀘𑁆𑀙𑀢𑀼 𑀪𑀤𑀦𑁆𑀢𑁄, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀤𑀦𑁆𑀢𑀲𑁆𑀲 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀧𑀳𑀺𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄𑁇

873. 𑀇𑀥 𑀧𑀦, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀪𑀺𑀓𑁆𑀔𑀼𑀕𑀢𑀺𑀓𑁄 𑀕𑀺𑀮𑀸𑀦𑁄 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀘𑁂 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀳𑀺𑀡𑁂𑀬𑁆𑀬—  ‘𑀅𑀳𑀜𑁆𑀳𑀺 𑀕𑀺𑀮𑀸𑀦𑁄, 𑀆𑀕𑀘𑁆𑀙𑀦𑁆𑀢𑀼 𑀪𑀤𑀦𑁆𑀢𑀸, 𑀇𑀘𑁆𑀙𑀸𑀫𑀺 𑀪𑀤𑀦𑁆𑀢𑀸𑀦𑀁 𑀆𑀕𑀢𑀦𑁆’𑀢𑀺, 𑀕𑀦𑁆𑀢𑀩𑁆𑀩𑀁, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀢𑁆𑀢𑀸𑀳𑀓𑀭𑀡𑀻𑀬𑁂𑀦, 𑀧𑀳𑀺𑀢𑁂, 𑀦 𑀢𑁆𑀯𑁂𑀯 𑀅𑀧𑁆𑀧𑀳𑀺𑀢𑁂𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

874. 𑀢𑁂𑀦 𑀔𑁄 𑀧𑀦 𑀲𑀫𑀬𑁂𑀦 𑀲𑀁𑀖𑀲𑁆𑀲 𑀯𑀺𑀳𑀸𑀭𑁄 𑀉𑀦𑁆𑀤𑁆𑀭𑀺𑀬𑀢𑀺𑁇 𑀅𑀜𑁆𑀜𑀢𑀭𑁂𑀦 𑀉𑀧𑀸𑀲𑀓𑁂𑀦 𑀅𑀭𑀜𑁆𑀜𑁂 𑀪𑀡𑁆𑀟𑀁 𑀙𑁂𑀤𑀸𑀧𑀺𑀢𑀁 𑀳𑁄𑀢𑀺𑁇 𑀲𑁄 𑀪𑀺𑀓𑁆𑀔𑀽𑀦𑀁 𑀲𑀦𑁆𑀢𑀺𑀓𑁂 𑀤𑀽𑀢𑀁 𑀧𑀸𑀳𑁂𑀲𑀺—  “𑀲𑀘𑁂 𑀪𑀤𑀦𑁆𑀢𑀸 𑀢𑀁 𑀪𑀡𑁆𑀟𑀁 𑀆𑀯𑀳𑀸𑀧𑁂𑀬𑁆𑀬𑀼𑀁, 𑀤𑀚𑁆𑀚𑀸𑀳𑀁 𑀢𑀁 𑀪𑀡𑁆𑀟𑀦𑁆”𑀢𑀺𑁇 𑀪𑀕𑀯𑀢𑁄 𑀏𑀢𑀫𑀢𑁆𑀣𑀁 𑀆𑀭𑁄𑀘𑁂𑀲𑀼𑀁𑁇 “𑀅𑀦𑀼𑀚𑀸𑀦𑀸𑀫𑀺, 𑀪𑀺𑀓𑁆𑀔𑀯𑁂, 𑀲𑀁𑀖𑀓𑀭𑀡𑀻𑀬𑁂𑀦 𑀕𑀦𑁆𑀢𑀼𑀁𑁇 𑀲𑀢𑁆𑀢𑀸𑀳𑀁 𑀲𑀦𑁆𑀦𑀺𑀯𑀢𑁆𑀢𑁄 𑀓𑀸𑀢𑀩𑁆𑀩𑁄”𑀢𑀺𑁇

---

875. 𑀯𑀲𑁆𑀲𑀸𑀯𑀸𑀲𑀪𑀸𑀡𑀯𑀸𑀭𑁄 𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑁄𑁇
